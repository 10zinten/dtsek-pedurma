དེ་དའི་དོན་བསྟན་འགྱུར། །རྒྱུད། %གུp:::«པེ་»«སྣར་»ཡི་དམ། @«པེ་»«སྣར་པ་སྟེ། «པེ་»«སྣར་»+ལས་སྒྲུབ་པའི། ྋག་«པེ་»«སྣར་»གྱིས། །Q«པེ་《སྣར་གཟུང་། «པེ་《སྣར་»པ། །«པེ་《སྣར་། །ཞུ་«སྣར་»འདི། «པེ་»«སྣར་»ཀྱི། སྐུ་དྲིན། ་བ་ལས་སྐད་«པེ་»«སྣར་»དུ་ཉིད། «པེ་»སྐལ་བ། 《སྣར་སྐབས། :«པེ་»«སྣར་»བོང་། ®«པེ་»«སྣར་པའི་སེམས་ཅན། (མ«པེ་»«སྣར་དཀར། «པེ་»«སྣར་—པ། @«པེ་»《སྣར་»སྟོན། །«པེ་»«སྣར་+བ། པར། ཚུ་པ་ནས། འོ་པོ་ལ་ཞི་བའི་ཚེ་སྟེ་ཕྱིན། (པེ་»«སྣར་འགྱུར་བ། «པེ་»«སྣར་རྟོག«པེ་《སྣར་»+ཡང་། 《ཅོ་པས། «པེ་»《སྣར་རྟོག་«པེ་»འགྲོལ། «པེ་»«སྣར་གྱིས་བརླབས། ཚེ«པེ་»《སྣར་++རབ་བསྔགས་་ཆོས་ཐམས་ཅད་རང་ངོས་རང་བཞིན་གྱིས་རྣམ་པར་དག་པའི་རྡོ་རྗེ། «པེ་»《སྣར་»ཨོཾ་ཨ་ཨོཾ་ཨ་པྲ་ཀྲི། «པེ་»སྣར་+hཆོས་ཐམས་ཅད་རང་བཞིན་གྱིས་ཡོངས་སུ་དག་པ། :ཀྱི(«པེ་»«སྣར་»ཏུ་ཏ་འདི་ལྟ་སྟེ། ཀྱེ«པེ་《སྣར་ཐ་ག་ཏ། །དེ། «པེ་»«སྣར་»ཡེ་ཏི། ཨ། +རྒདེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཡེ་ཤེས་ཀྱི་སྐུས་འཇམ་དཔལ་ཡོངས་སུ་དག་པ་ཉིད་དུ་བཟུང་བར་བྱའོ་ཞེས་སོ། །»«པེ་»«སྣར་»ཏ་ཧྲི། «པེ་»«སྣར་»+ང་དེ་བཞིན་་་་་་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྙིང་པོ། «པེ་»«སྣར་མཱནྟི། ཚེ«པེ་»«སྣར་»+དཀྱེ་བཅོམ་ལྡན་་་་་་འདས་ཡེ་ཤེས་སྐུ་ཅན་གསུང་གི་དབང་ཕྱུག«པེ་»«སྣར་+དསྨྲ་བ་ཆེན་པོ། «པེ་»《སྣར་»དྷརྨཱ། །+རྒཆོས་ཐམས་ཅད་ནམ་མཁའ་ལྟར་དྲི་མ་མེད་ཅིང་ཤིན་ཏུ་ཡོངས་སུ་དག་པའི་ཆོས་དབྱིངས་ཡེ་ཤེས་ཀྱི་སྙིང་པོ། («པེ་»«སྣར་»དྷརྨཱ། «པེ་»«སྣར་མཚན། །འགྱེ«པེ་»«སྣར་ཨོ། («པེ་»«སྣར་»ཀྱིས་བྱིན་་་དུ་གྱིས། (མེ«པེ་»«སྣར་»—པས། ་«པེ་—པས། «སྣར་ནས། དང་ས་ཊ«པེ་»«སྣར་»ཀྱི། «པེ་»《སྣར་བསྒྱུར། »«པེ་»《སྣར་བསྒྱུར། «པེ་»«སྣར་»ཀྱིས། །ལ་ན་བླ་ན་བ་དང་། ར་་ག་ཟླ་་འས་པ་ཚེ་།
