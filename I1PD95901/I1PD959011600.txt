ནི་སྟོང་པ་ཉིད་མ་ཡིན། །ལྟ་བ་རྣམས་ལས་ལོག་རྣམས་དང༌། །བདག་ཏུ་ལྟ་བ་ཚོལ་རྣམས་ཀྱི། །བདག་ཏུ་འཛིན་པ་བཟློག་པའི་ཕྱིར། །རྒྱལ་བ་རྣམས་ཀྱིས་སྟོང་པ་བསྟན། །སྲིད་ཞི་མཉམ་ཉིད་དུས་གསུམ་རྒྱལ་བའི་ཡུམ། །འགྲོ་བའི་་་བདེ་སྒྲུབ་ཐབས་མཁས་སྟོན་པར་བྱེད། །གཞུང་གི་བདག་ཉིད་རིན་ཆེན་ལྟ་བུ་ལ། །དད་དང་གུས་པས་ཡང་ཡང་ཕྱག་འཚལ་ལོ། །ཕན་སེམས་སྤྲོ་དགས་བྲིས་པའི་་་འགྲེལ་པ་འདི། །གང་ཟག་དམ་པ་རྣམས་ཀྱི་ཕྱག་ཏུ་བརྒྱུད་པ་དང་། །བསྟན་པའི་བྱ་བ་མཛད་པ་བདག་གི་སྨོན་ལམ་ཡིན། དང་། ཆོས་ཉིད་བདེན་པས་དེ་ལྟར་འགྲུབ་གྱུར་ཅིག། རྒྱ་གར་གྱི་མཁན་པོ་དེ་ཉིད་དང༌། སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་་་་དགེ་སློང་སེང་གེ་རྒྱལ་མཚན་གྱིས་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །་། །སདྲ་པg《་ཀྱི་《་། །།ཀ་རཡི་བྱཚོ:ཀ་བ་ཏུ་བྱེད་པའི་ཁ་ཁྱི་ཆཟླ་ཚོ། བྱིས་ཏེ་འོས་དང་བ་ས་ཀྱིoa། བསྡུར་མཆན། ཀྱི། ཡོ་བ། ནགས་ཀྱི་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་༄༅། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྙིང་པོའི་འགྲེལ་པ་དོན་གྱི་སྒྲོན་མ་བཞུགས་སོ། །«པེ་»«སྣར་»ཡཱརྦ། ཉི«པེ་»«སྣར་པ། ༄༅«སྣར་རྣམས། མ«པེ་»«སྣར་མཱ། «སྣར་»+ལ། «པེ་»«སྣར་+གནས། @«སྣར་»གནོད། །ཚེ«པེ་»«སྣར་བློ་བུར། «པེ་»«སྣར་མོའི། @«པེ་»«སྣར་»སོ། «སྣར་+ཐ། «སྣར་ལས། «པེ་»«སྣར་»ནི། ཉི«པེ་»«སྣར་»+དེ། ཉི«པེ་»«སྣར་»དངང་། «པེ་»«སྣར་གྱིས། «པེ་»«སྣར་ནས། ་«པེ་»«སྣར་པར། །«པེ་»«སྣར་ན། །«སྣར་པ། «སྣར་»དོ། །«སྣར་འགག«པེ་»ནོ། །«པེ་»«སྣར་»དོ། །«པེ་»སྣར་»«ཅོ་+ནི། «པེ་»«སྣར་+ནི། «པེ་»«སྣར་»—མི། ཉི«སྣར་»སྦྱངས། «པེ་»«སྣར་»—ཐོབ་བྱ། —།
