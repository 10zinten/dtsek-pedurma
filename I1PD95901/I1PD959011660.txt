སྨན་ལྟ་བུའི། །ཡོངས་ཤེས་སྨན་པ་ལྟ་བུ་འདིས། །ཉོན་མོངས་འཇིག་རྟེན་ནད་པ་ཡི། །སྡུག་བསྔལ་ནད་རྣམས་སེལ་བར་ཤོག། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་་པའི་སྙིང་པོ་ཡོངས་སུ་ཤེས་པ་པཎྜི་ཏ་ཤྲཱི་མ་ཧཱ་ཛ་ནས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་དེ་ཉིད་དང་། སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་སེང་གེ་རྒྱལ་མཚན་གྱིས་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།།བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་ཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་༄༅། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་གྲེལ་པ་ཤྲཱི་མ་ཧཱ་ཛ་ནས་མཛད་པའོ། །ཞེས་འཁོད། ︽«པེ་»«སྣར་»ཏ། «པེ་»«སྣར་»ཡ་པ་རི་ཙེ་ཡ། ༄༅«པེ་»«སྣར་»འབྲེང་། «པེ་»«སྣར་བ་ར་ཎཱ་སེར། །«པེ་»«སྣར་»ཀྱིས། ཉི«སྣར་»ཉེར།  ི«སྣར་བསྡུད། «སྣར་བརྩོམ། «པེ་»«སྣར་»རིང་། ཚེ«སྣར་རིམས་ཀྱིས། «པེ་»«སྣར་»གྱི། «སྣར་གཟུང་། «པེ་»«སྣར་ལྟར། «པེ་«སྣར་བསླང་། «པེ་《སྣར་+བདག«པེ་»«སྣར་བསྡུད། །«པེ་»«སྣར་རྟོགས། ཉི«པེ་»«སྣར་བསླང་། ཉི«པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་སྐྱབས། གྱི«སྣར་+བ། «པེ་»«སྣར་པའི། །«སྣར་»—འཛིན། «པེ་»«སྣར་—བཅོམ། «པེ་»«སྣར་»—ཡེ་ཤེས་ཀྱི། «པེ་»«སྣར་—ནི། ཉི«སྣར་»ཀྱི། «པེ་»«སྣར་པ། ཚུ«པེ་»«སྣར་པོ་ནི། «པེ་»«སྣར་པོའི། །«པེ་»«སྣར་དེ། @«པེ་»«སྣར་»«ཅོ་»པའི། ་འ«པེ་»«སྣར་བསླུ། «སྣར་»པ་སྟེ། @«པེ་»«སྣར་»བ། «པེ་»«སྣར་སྐྱབས། «པེ་»«སྣར་—པ། «པེ་»«སྣར་དེ། ཉི«པེ་»«སྣར་»+ལ། «པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་གཞན། ཉི«སྣར་»—རྒྱས། «པེ་»«སྣར་»+པ། «པེ་»«སྣར་»བརྗོད། །«པེ་»«སྣར་པ། «པེ་»«སྣར་»གོ། །«པེ་»«སྣར་»གྱིས་བརླབས། ༄༅«པེ་»«སྣར་དེས་ན། —་ན་་་། ་་་་ན་་་་་་།
