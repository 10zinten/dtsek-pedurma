རྟོགས་པར་གྱུར་ནས། ཐེག་པ་ཆེན་པོ་ལ་སྒྲིབ་པ་མེད་པའི་ཕྱིར་དགྱེས་ཤིང་དེ་་་སྐད་ཅེས་བཀའ་སྩལ་པའོ། །ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དང་། བྱང་ཆུབ་་་སེམས་དཔའ་སྤྱན་རས་གཟིགས་ཀྱི་དབང་པོ་དང་། ལྷ་དང་། །མི་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟར་བཅས་པའི་འཁོར་ཐམས་ཅད་བཅོམ་ལྡན་འདས་ཀྱིས་་་་གསུངས་པ་ལ་མངོན་པར་དགའོ། །བཅོམ་ལྡན་འདས་མ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྙིང་པོའི་འགྲེལ་པ་སློབ་དཔོན་ཛྙཱ་ན་མི་ཏྲས་མཛད་པ་རྫོགས་སོ། །།།བསྡུར་མཆན། །::དེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་༄༅། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྙིང་པོའི་རྣམ་པར་བཤད་པ་ཛྙཱ་ན་མི་ཏྲས་མཛད་པའོ། །ཞེས་འཁོད། «པེ་»«སྣར་»ཧྲི། ཚེ«པེ་»«སྣར་བྱ་ཁྱ། «པེ་»《སྣར་པ། «པེ་»《སྣར་+ལྡན། «པེ་»«སྣར་བཞི། །«པེ་»«སྣར་བརླབས། ཉི«པེ་»«སྣར་པའོ། «པེ་»«སྣར་»གྱིས། «སྣར་»པར། །ཚེ«སྣར་བསལ། «པེ་»«སྣར་»དེ། «པེ་»«སྣར་»—ཀྱི། ︽«སྣར་ན། «པེ་»«སྣར་བྱའི། «པེ་»«སྣར་གཅིག«པེ་»«སྣར་རིའོ། །«པེ་»«སྣར་»གཅིག་«པེ་»«སྣར་+དག«པེ་»«སྣར་»ལྟའོ། «པེ་»«སྣར་»—འཕགས་པ། «པེ་»«སྣར་»ལྟའོ། ཚུ«པེ་»«སྣར་»+ལྔ་པོ། «པེ་»«སྣར་+གྱི་སྟོང་པ། ཚེ«པེ་»«སྣར་»+རང་གི། «པེ་»གྱིས། དུས་ཚོད་«པེ་»«སྣར་པའོ། «པེ་»བརྟགས། ཚེ«པེ་»ལྔ་པོ། «པེ་»«སྣར་དེ། །ཚེ«པེ་»«སྣར་»+པ། «སྣར་»བུ། «པེ་»«སྣར་སོ། «པེ་»གྱིས། «སྣར་»བརྟགས། «པེ་»«སྣར་»—ཉིད། «པེ་»«སྣར་བྱ་བ། «པེ་»«སྣར་བཙལ། །«པེ་»«སྣར་»—རྣམས། «པེ་《སྣར་བདགམི«སྣར་»ནི། སྐྱེ་བ། :ཚེ«པེ་»«སྣར་+མ་ཡིན་པ། ཚེ«པེ་»ཤིའི། ཚེ«པེ་+ནི། །།
