སློབ་དཔོན་གང་བ་སྤེལ་བ་ཞེས་བྱ་བ་ཆོས་མངོན་པའི་གཞུང་ལུགས་རྒྱ་མཚོའི་ཕ་་་རོལ་དུ་སོན་པས་མཛད་པ་ཆོས་མངོན་པའི་འགྲེལ་བཤད་མཚན་ཉིད་ཀྱི་རྗེས་སུ་་་འབྲང་བ་ཞེས་བྱ་བ་རྫོགས་སོ། །།།འདི་གངས་ཆེན་པོ་ཏི་སེ་དང་ཡིད་བཞིན་གྱི་མཚོའི་ལྷོ་ངོས་རི་བོ་ཆེན་པོ་སྤོས་ཀྱི་ངད་ལྡང་བའི་འདབ། ཡུལ་པུ་རངས་སུ་དགེ་འདུན་གཙོ་བོར་གྱུར་པ་བརྒྱ་ཕྲག་གཅིག་དང་། དགེ་འདུན་སྤྱིར་གྱིས་གཡུ་:ཕག་སྒུར་ལ་སོགས་པ་ཕུལ་ཏེ་བཀས་གཉེར་ནས་རྒྱ་གར་གྱི་མཁན་པོ་ཀ་ན་ཀ་བརྨ་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་པ་ཚབ་ཉི་མ་གྲགས་ཀྱིས་བསྒྱུར་བ་ཞིག་གོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ཆོས་མངོན་པ་མཛོད་ཀྱི་འགྲེལ་བཤད་མཚན་ཉིད་ཀྱི་རྗེས་སུ་འབྲང་བ་གང་བ་སྤེལ། ཞེས་འཁོད། «པེ་»ཙྩཱ་ནུ། 《སྣར་ནྟཱ་ནུ། ཚེ«སྣར་»ཐོས། ︽«སྣར་»རྗེ། (ི《སྣར་»+མ། «པེ་ཐོན། «པེ་»«སྣར་པས། «པེ་»«སྣར་ལས་བརྒལ། །«པེ་»བ་གཞན། «སྣར་གྱི། ༄༅«པེ་»«སྣར་»རང༌། «པེ་»«སྣར་བཞག(ཆོས་ཚན་འདིའི་ནང་་དུ་གཞག་ཅེས་པ་རྣམས་«པེ་»«སྣར་ལ་བཞག་ཅེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་»ཕྱིར། «སྣར་པ། །ི«སྣར་»རྟེན། «པེ་»«སྣར་པས། «པེ་»«སྣར་+ཏེ། ®«པེ་»«ཅོ་»འཇུག། «པེ་»«སྣར་»+ཅན། །«པེ་»«སྣར་»—རྣམས། Q«པེ་—པས། «པེ་»«སྣར་བྱའོ། «པེ་»གོམ། («པེ་»«སྣར་»པ་དེ་གཏན༑«པེ་»«སྣར་གཅོད། «སྣར་སྨྲོས། «པེ་»«སྣར་—ལ། «པེ་»«སྣར་ལ། «པེ་»«སྣར་ལ། «པེ་»«སྣར་པས། «པེ་《སྣར་»—གིས། «སྣར་»རིག«པེ་»ཕྱི་«པེ་»《སྣར་»སྟན། ་། །ཚེ«པེ་《སྣར་ཀྱི། ཉི«སྣར་བརྟེན། ཚེ«པེ་ཡིན། ཆོ«པེ་»«སྣར་»དེ་ཉིད། «པེ་»།
