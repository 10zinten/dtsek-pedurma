བ་ཊ། ་་བ་་འད། །ཁ་ན་ཀློགཞི་ཅད་ཀཎི་ལན་ཏ་ག་ཀ་བ་ལས། །ཀུན་ཏུ་ས་ལ་གལ་བར་བགོ་ཅའོ་ཁ་བ་བབ་རྗེ་བློ། ལ་མཁརྩ་དེབ་ཁག་ནས་་ང་བ:འཁ་བྱ།  ། ད་ན་ཆུ། །དྲ་བ། Sབྱ་བ་ཁོ་ནན་མ། །མ་ཀློབ་བསྡུར་མཆན། p/ཊ། ཞའོ་ཁོ་ཚོ་ཁྲུ་ཊ་ཡ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«པེ་»«སྣར་»ལ་མི་བཞུགས་པས་«ཅོ་དང་དཔེ་བསྡུར་བྱས་ཡོད། །ར། ན་ག་ཀོ་ཤོeEའི་ཞི་ཁྲོ། བབབཔ«ཅོ་པ། ཚེ«ཅོ་དེ། དད་པ་ལ་གཏུབ་ཁལ། ?ན་ཁག་ཁཚོ་ཚོ། ཨ༌། །།།«ཅོ་»དགྲིས། །ཏ་ལ་ཁཇོ་ན་ཏ་ཀ་ཊ་ཇ་ད་«ཙོ་»ནཱ། ་མ་་ཁོ་གའོ་གླིང་་་་་་་ཁྲི་ཚོ་ལ། ཀ་བཞི་«ཅོ་ཛྭ་ལ་ཛ། །བྱ་བ་ལས་མ་བསྐུལ་བ་སོགས་འཚོ་བ་ལ་པར་ཡབ་དབལ་ཤུལ་ལ་ས་་་ལ་བ་པ་དེ་ལྟ་ལོ་མས་ཀྱིས་དེ་ལས་འབད་་བསྡུར་འབྲས་རེའུ་མིག་དེ་དོ་དེ་ཕྲུ་། མ་བརྒྱུད། （དབབ་འཕྲལ་ཁམས་ལུ་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། །ཅོ་ནེ། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནབ་ཕྲེང་། ཚེག་ཁྱིམ། །།ཤན་བ་ལ། བ། ར། པས། ཟླ། :ཀྱིངབས་པ་དང་། འབའ་པ་་་་་་་་་མང་བ་་བ་་པ་དབུལ་བ་་་་་་གྱི་དོན། ས་རྒྱུས་ལ་།
