སློབ་དཔོན་བོ་དྷི་སཏྭས་མཛད་པའི་འགྲེལ་པ་ཕྱི་མ་རྫོགས་སོ། །་ཀཆུ་། ད་:ལོ་ཙཱ་བ་དཔལ་བརྩེགས་ཀྱིས་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པའོ། །།།པའི་བར་དུ་དེ་ས་བོན་ལ་ལ། ནས་པ་བསྟན། བ་མོ་མ་ཎཁཀྱི་བྱ་བ་ན། འབག་པ་ལ་ནམ་པ་དབུས། ཞེས་བྱ་བའི་བར་བ་ལ་ཁ་བསྡུར་མཆན། གཏོང་ཀ་ཁ་སྦྱར། །::པ་ལ་། ལྟགས་དྲི་མ་དང་ང་། ཡཏོང་། ཏཱ་ཏ་པ་ཏི་ས་སྐྱ་བ། འོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་རྡོ་རྗེ་རྣམ་པར་འཇོམས་པ་ཞེས་བྱ་བའི་གཟུངས་ཀྱི་བཤད་པ་བཞུགས་སོ། །«སྣར་—ཎ། ཚེ«པེ་»«སྣར་»ཀ«པེ་»«སྣར་པའི། ཚ«སྣར་»སྒྲུབས། །རྒྱལ་ཀ»::ཀ་ཤ། །ཚེ«པེ་»«སྣར་»གྱི། W«སྣར་—དང༌། ：«སྣར་»གིས། «པེ་»ཀྱིས། «པེ་»གིས། «སྣར་གྱི། ༄༅«པེ་»ཀྱིས། ཚེ«པེ་»གཅད། «པེ་སྲུང་བ། «སྣར་བསྲུངས་བར་བྱ། ཁྱེ«སྣར་»སྟན། ཀྱེ«སྣར་ནི། ཉེ«པེ་»རྩེ་གཅིག་སྟེ«པེ་»«སྣར་གྱི། །«སྣར་—ཞེས། ཉི«པེ་སྐྱེས། ན«སྣར་སྤང་། «པེ་བྱ་བའོ། «པེ་རིས། «པེ་»གིས་ཐོས་པ། 《སྣར་གི་ཐོས་པའི། ༄༅«སྣར་»ལ། ཉི«སྣར་+དེ། «སྣར་ཚེ། ཁྱེ«པེ་»«སྣར་+མ། གང་«སྣར་གྱི། ®«སྣར་»བཞུགས། «པེ་ལ། «པེ་ཀྱིས། «པེ་གྱིས་བརླབས། «སྣར་»གྱི་་་་་བརླབས། «པེ་»གྱིས་བརླབས། «སྣར་»གྱི་བརླབ། ཚེ«པེ་»«སྣར་པོ། «པེ་»བ། «པེ་»གིས། ང་། །:ོ་། ན་ང་ད་། འ། བ་་་་་། །།pཚེ«པེ་»བཞུགས། «པེ་»«སྣར་བྱའོ། «པེ་»གཞི། «པེ་»ཏེ། «པེ་»«སྣར་གྱིས་་་་བརླབས། པོ«སྣར་»—གྱི། @«པེ་ནོ། «སྣར་རྗེ། «པེ་»«སྣར་»—ཞེས། ཁྱེ«སྣར་གྱི། ((Q«པེ་»«སྣར་»གྱིས་བརླབས། «པེ་»གྱི་བརླབས་ཀྱིས། «སྣར་»གྱིས་བརླབ་ཀྱི། །ཅེ«སྣར་»གྱིས་་་་བརླབས། «པེ་བ། ལོ«སྣར་སྐབས་སོ། ཁྱེ«པེ་»«སྣར་པོས། «སྣར་»རྡོ་རྗེ། «པེ་»«སྣར་ལས། ་ནི«པེ་»—སྨྲས་སོ། «པེ་འཕམ། «སྣར་གཟུངས། «པེ་»«སྣར་རྟས། @སྣར་བརླབས། «སྣར་»ཚེ། @«པེ་»ནུ་བར་འགྱུར། ཚེ«པེ་»—ཅད། «པེ་»སྤྱོད། «པེ་+ནི། «སྣར་པའི། །nCདབད་དང་། པ་པོ་ཨེ«པེ་»དོན་དུ་སྤྱོད། «སྣར་དུ་དོན་དཔྱོད། «པེ་»གྱི་རླབ་ཀྱི་གཞན་གྱི། «སྣར་གྱིས་བརླབས་ཀྱིས་་་་གཞན་གྱིས། «སྣར་»དེས། «པེ་»«སྣར་བུ། «སྣར་བཞི། @«པེ་དེ་ནི་བྱིན་གྱིས་་་་།
