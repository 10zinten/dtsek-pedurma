ཏཝོག་ལ་ཆ་བློ་ལྟོཤོག། རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་གཟུངས་ཀྱི་རྣམ་པར་བཤད་པ་རྡོ་རྗེ་སྒྲོན་མ་ཞེས་བྱ་བ། གསང་སྔགས་མཐའ་དག་ལ་མངའ་བརྙེས་པའི་སློབ་དཔོན་ཆེན་པོ་་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པའོ། །།།སྐྱེ་ཁམིང་གའོ་ཏེ་ག་ཏོ་ཁ་མི་ཚང་ཁ་པ་ਸਲਮਨਵੀਸਸੁਸਲੋਂਨਵੇਂਸੌਸੇਵਾਵਣਹੀਰੋਸਸਲਾਨਿਸਬਨਚ'ਲੈਸਬਸਭਾਨੂੰਸਾਬਕਾਲਾਸਗੋਂਉਸདེ་བདེ་བ་ལ་གནས་པ། ཁྲོ་ལ་ཁོ་བསྡུར་མཆན། ལོ་རྒྱུས་ལོན་བྱེད་ཁ་ཀ་ཁ་ཁ། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»རྡོ་རྗེ་རྣམ་པར་འཇོམས་་པ་ཞེས་བྱ་བའི་གཟུངས་ཀྱི་རྣམ་པར་བཤད་པ་རྡོ་རྗེ་སྒྲོན་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །《སྣར་རྡོ་རྗེ་རྣམ་འཇོམས་ཀྱི་གཟུངས་ཀྱི་རྣམ་བཤད་སློབ་དཔོན་ཆེན་པོ་པདྨས་མཛད་པ་བཞུགས་སོ། །@«སྣར་»འཇོམ། «སྣར་»གསུངས། «པེ་སྤྲུལ། ཚེ«པེ་པ། «སྣར་»། །ཚེ«པེ་»—འདི་དག་གིས་བརྗོད་པར་བྱ་བ་ལ་སོགས་པ། ཉི«སྣར་»ཞེས། «པེ་»འདུལ། །།ད། «སྣར་བརླབ། Q«པེ་»འཐུན། «པེ་བྱ་བའི་བར་གྱིས་ནི་བྱིན་གྱིས་བརླབས། «སྣར་»པའི་བར་གྱི་་ནི་བྱིན་གྱིས་བརླབས། «པེ་»«སྣར་»དུ། མ«པེ་»+བྱ། «སྣར་བཅད། ཚེ«པེ་»《སྣར་»བསྲུང་། ཁྱི«པེ་བྱའི། «པེ་»གྱི། «སྣར་»། ༎གྱི«སྣར་རམཱ། ཉེ«པེ་»«སྣར་»ཏ། ཉེ«སྣར་གྱི། «པེ་བརྩལ། །དུག་ལྔ་དང་། ཡོད་འོང་།  ། བྱ་ཁn:ཏི་ཏུ་«སྣར་»གྱི། «པེ་ཀྱིས། «པེ་»«སྣར་ལུ། &«སྣར་»གྱི། «པེ་»«སྣར་»ཎཱ། «པེ་»ཀི་ལི་ཀི་ལ། «སྣར་ཀཱི་ལ་ཀཱི་ལི། ༄༅«སྣར་»གྱི། ཉི«པེ་བསྐུལ་བ། «པེ་རཱ། ཚེ«པེ་»«སྣར་ནམས། ཀྱེ«པེ་»ནཾ། («པེ་བཛཱ་ཀུ་ཤ་ཛཱ་ལཱ་ཡ། «སྣར་བཛྲ་ཨརྐུ་ཤ་ཛྭ་ལ། བུ«པེ་བིནྡ་བིནྡ། ཁྱི«པེ་མ་ཙནྟ། །«སྣར་མཙནྡ། ཨེ«སྣར་»གྱི། ཚེ«པེ་»སྡུས། «སྣར་། །པེ་ཁེ«སྣར་»གྱི། ད་དུང་ཀྱང་། གལ་«སྣར་གྱི། «པེ་»དགཉི«སྣར་གྱི། «པེ་»«སྣར་»«ཅོ་»དུར། «སྣར་ཀྱི། «སྣར་»བཀང་། @«སྣར་»གྱི། «པེ་»ཀླགཁྱེ«སྣར་བྱས། ཀྱེ«སྣར་»གི། ལུ«སྣར་»། ༎«པེ་»«སྣར་»བརྗོད། «པེ་»«སྣར་འདི། ཨེ«པེ་»དགོངས། ཨེ«སྣར་། །།«པེ་»—གྱུར། །«པེ་»པའི་གཞི། «པེ་»གིས། ན«པེ་»ནི། «པེ་»«སྣར་ཞེས། ཚེ«སྣར་ཀྱི། «པེ་ལས། «པེ་»གང་གིས། «སྣར་བདག་གི། ཉི«སྣར་»ཀྱི། «པེ་»«སྣར་ལ། གྱི«སྣར་»ཀྱི། འགྱོ«པེ་བྱའོ། །«པེ་»གི། སྟེ«པེ་པ། «པེ་»—ཚོགས། ཁྱེ«པེ་»«སྣར་»འགྱུར། ཚེ«པེ་—ལྡན་པ་ནི། ཁྲི«པེ་»«སྣར་»ཏེ། ཉི«པེ་བརྗོད་ཅེས། «སྣར་རྗེ། «སྣར་པོ། ཚུ«པེ་»གཞུགས། «སྣར་»། ཚེ«སྣར་»གྱི། O«པེ་ལྟར། ཚེ«སྣར་རིགཚེ«སྣར་»གི། ཀྱེ«སྣར་བཞི། །
