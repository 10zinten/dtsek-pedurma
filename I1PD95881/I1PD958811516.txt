སྐད་ཅིག་མ་སུམ་བརྒྱ་སུམ་བརྒྱ་ཡོད་དོ། །ལར་ཏུག། དེ་བཞིན་དུ་ལོ་དང་ཟླ་བ་དང༌། དུས་ཚོད་ཀུན་ལ་ཤེས་པར་བྱའོ། ་་། རྒྱལ་མཚན་ནི་འཇིག་རྟེན་གྱི་ཁམས་སུ་ནི་གྲགས་པའོ། བ། སེང་གེ་ནི་གཅན་གཟན་ཐམས་ཅད་ལ་མི་འཇིགས་པའོ། །ཁྱི་ནི་སྙིང་ཉེ་བའོ། ལ་། གླང་ནི་བླུན་པའོ། །།རུས་སྦལ་ནི་ཐེག་པ་ཆེ་བའོ། །བོང་བུ་ནི་འཇིག་རྟེན་གྱི་ཁམས་སུ་ངན་པར་གྲགས་པའོ། །།ཆུ་སྲིན་ནི་ཉེ་བར་གྲུབ་པའོ། །་་། དུང་ནི་བརྟན་པ་སྟེ་མི་འགྱུར་བའོ། །་ལ་། རྔ་མོ་ནི་ལས་ལ་ངན་པའོ༑་ལ། གཞུ་ནི་འཁྲུལ་ལ་བཟང་བའོ། །མདའ་ནི་འབིགས་པའོ། །གླང་པོ་ཆེ་ནི་འདོད་ཆགས་ལ་དགའ་བའོ། །དེ་བཞིན་དུ་ས་རི་བཅུ་གཉིས་བརྩི་བར་བྱའོ། །།ཅ་ཌ་་། །ཞp། །ཉང་རོ་ལ་ཁ་བའི། །བྱ་བ་::ཀུཀུན་དགའ་བོ་ཞེས་བྱ་བ་ནི། ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་གསུངས་པའོ། །།ཁྱོད་ཀྱི་གཟའ་གང་དག་རྒྱུ་སྐར་རྣམས་ཀྱི་ཁྲོད་ནས་རྒྱུ་ཞིང་། ཞེས་བྱ་བ་ནི། བྱང་ཤར་མཚམས་ན་བྲ་ཉེ་དང་སྨིན་དྲུག་གནས་ཏེ། །ཤར་དྲང་པོ་ནི་སྣར་མ་དང་མགོ་དང་ལག་:དང་ནབ་སོ་དང་རྒྱལ་གནས་སོ། །ཤར་ལྷོ་་་མཚམས་ན་སྐག་དང་མཆུ་གནས་སོ། །།ལྷོ་ཕྱོགས་ན་བྲེ་དང་མེ་བཞི་དང་དབོ་དང་ནག་པ་དང་ས་རི་གནས་སོ། །ཨུ། ལྷོ་ནུབ་མཚམས་ན་ས་ག་དང་ལྷ་མཚམས་་་་། གནས་སོ། །།ནུབ་ཕྱོགས་ན་སྨོན་དང་སྣུབས་དང་ཆུ་སྟོད་དང་ྋཆུ་སྨད་དང་བྱི་བཞིན་གནས་སོ༄༅། །ར། །།ནུབ་བྱང་མཚམས་ན་གྲོ་བཞིན་དང་མོན་གྲུ་གནས་སོ། །བྱང་ཕྱོགས་ན་མོན་གྲེ་དང་ཁྲུམས་སྟོད་དང་ཁྲུམས་སྨད་དང་ནམ་གྲུ་དང་ཐ་་་སྐར་གནས་སོ། །ཞུ། དེ་བཞིན་དུ་གཟའ་བདུན་པོ་ཡང་གནས་སོ། ::། དཔྱིད་ཟླ་ར་བའི་ཚེས་གཅིག་ལ་ཉི་མའོ། །།ཚེས་གཉིས་ལ་ཟླ་བ། ཚེས་གསུམ་ལ་མིག་དམར། ཚེས་བཞི་ལ་པ་བ་སངས། ཚེས་ལྔ་ལ་ཕུར་བུ། ཚེས་དྲུག་ལ་སྤེན་པ། :ཚེས་་—།
