ཟླ་བློ་ཁོཚེ«སྣར་»—པས། ཉི«སྣར་གྱིས། «སྣར་»—པ། ཆེ«སྣར་»—ཀྱི། འགེ«སྣར་»ཀྱིས། །«སྣར་སྩལ། «སྣར་ཕྲོ། «སྣར་བརླགས། «སྣར་»སྐྱེད་པ། «སྣར་»གནས། «སྣར་འགུག་༄༅«སྣར་ཀྱི། «སྣར་བརྟན། «སྣར་»ནི། ཆེ«སྣར་བ་སྒྱུར། ཚེ«སྣར་»ཀྱི། ཉེ«སྣར་»བསམ། «སྣར་»ནི། ལེ«སྣར་སྐུ། ཚེ《སྣར་བསམ། ཚེ«སྣར་ལ། «སྣར་—བ། ཉེ«སྣར་ཉམས། ཚེ«སྣར་བར། །«སྣར་_ལོངས་སྤྱོད་པས་ན། ®«སྣར་»—ནི། མི«སྣར་རིགས། «སྣར་ལས། མུ«སྣར་»མར། :@«སྣར་»མའི། «སྣར་»ཞིག@«སྣར་གྱིས་བརླབས། ཀྱི་»ལོ་རྒྱུ«སྣར་»སྐྱེད། «སྣར་»གི། «སྣར་»ལས། «སྣར་གསུང་སྟེ། བྱེད་པ་དང་རྒྱལ་བས་བཤད་པ་བརྒྱབ་ཨེ«སྣར་གདེང༌། «སྣར་བསྒྲུབ། «སྣར་ལས། «སྣར་འཚལ། «སྣར་པས། ༄༅«སྣར་»ཁམ། ༄༅«སྣར་རེ་རེ་ཞིང་ཚན། «སྣར་ན་བྱའོ། «སྣར་བསལ། ཚེ«སྣར་»སྤྲོ་བས། གྱི«སྣར་»མ། «སྣར་—བུ། «སྣར་བསྒྲུབ། @«སྣར་»—ནི། «སྣར་—པའི། ::«སྣར་བའི། «སྣར་+པོ། @«སྣར་བུར། .«སྣར་ཞེས། «སྣར་ཡིས། «སྣར་པོར། ༄༅«སྣར་—པ། O«སྣར་—དེ། «སྣར་མཉེ། ནOE:ལ་ནས་ཀྱི་ཁའི་ང་«སྣར་»+ནི། «སྣར་ལྷག《སྣར་»—རིག︽«སྣར་»+ཞི་བས། མ«སྣར་»ལས། @«སྣར་གསུངས། ༄༅«སྣར་»བའི། «སྣར་»སྨིཾ། ཨ«སྣར་»སྒྱུར། ཀ་བསྡུར་འབྲས་རེའུ་མིགདག་རྒྱུད། （ཀྱིལ། ལ་བའི། banཔེ་ཅིན། །དཔེ་གཞི། །སྡེ་དགེ། །སྡེ་ཚན། ཁ་རྒྱུད། པོད་རྟགས། ཐུཚན་གྲངས། །ལྡེབ། ན་བ་ཕྲེང་། །ཚེག་ཁྱིམ། །་སྣར་ཐང་། འབབ་རྒྱུད༌《རྒྱུད། (ལས་ཀ་བྱས། ར/ཆུ་པ། དང་། བ་_བ་ཅོ་ནེ། jazz་རྒྱུད། །@ཐུg། ་ནབ་(P། །
