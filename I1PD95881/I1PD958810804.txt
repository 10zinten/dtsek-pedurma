ཌཚོ། ཎཎ་ཙཱ་ཎ་ཊ་ཀ་ཋ་ཌ་ཌཚོ། །་ཁ་ཌ། ཙོ་ཌ་ཚོ་ཏ་ཀ༌་་ཀwཊ་ཋ་ཡཊོ། ཅ་ཅ་ཅཎྜ་ལ་ཁ་ཎ་ཁ། pཚོཌ་ཚོ་ཎ་འཛིཀ་ར་ཌཅཁཚོ། ༄༅། །ཀུན་ནས་སྒོར་འཇུག་པའི་འོད་ཟེར་གཙུག་::ཀཎྜི་ཁཛ་ཁ་ཏོར་དྲི་མ་མེད་པར་སྣང་བ་དེ་བཞིན་བྷོ་ཌི་ཏོཎ་ཁ་ཁ་འཉྩ་RIད་གཤིགས་པ་ཐམས་ཅད་ཀྱི་སྙིང་པོ་རཀྱང་རལ་། །/jzལ་ཌ་ཌོ་ཎnཆུ་ཚོ་ཎཌ། g/*ཡནྟོ་ཥཀ་ར་ཎཏྲཀ་བ་ཀ་ཁ། །ཌཱ་ཥ་ཁྲོ་ཌོ། །ཀཎི་འཇུཊ་དང་དམ་ཚིག་ལ་རྣམ་པར་བལྟ་ཁྲ་རྫོགས་་་་ག་ཀླུ་འམ། ཁ་ཎཁན། །བ་ཞེས་བྱ་བའི་གཟུངས་ཀྱི་ཁྱི་ལྔ་:སྐྱོ་ཌ་ཁཛོརཥ་ཌ་ཎཁ་ཊཁ་»ཁཎྜ། རྣམ་པར་བཤད་པ། སྐྱའོ། །་ཁཀྱི་ཌ་ཎོ་:ཌོནབོ་བི་ཟཡིག་ཀ་ཅཎ་ཌ་ཎ་ཌཌ། ་་། ཀཎོཌ་ཚོ་ཀཎྜོ་པ་ཏཁ་། །ཎ་ཀོཌ་ཀྱི་ཆ་ཆ། ༄༅། །རྒྱ་གར་སྐད་དུ། །ས་མནྟ་མུ་ཁ་པྲ་བེ་ཤ་རསྨི་བི་མ་ལོཙྪིཥ་ཕྲ་བྷཱ་སབྦ་ཏ་ཐཱ་ག་ཏཱ་ཧྲ་སཱི་ད་ཡ་ས་མ་ཡ་བི་ལོ་ཀི་ཏེ་ནཱ་མ་དྷཱ་ར་ཎི་སཾ་པྲྀཏྟི། ཚེ་ཁོ། བོད་་་སྐད་དུ། ། ། ཀུན་ནས་སྒོར་འཇུག་པའི་འོད་ཟེར་གཙུག་ཏོར་དྲི་མ་མེད་པར་སྣང་བ་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྙིང་པོ་དང་དམ་ཚིག་ལ་རྣམ་པར་བལྟ་ཞིབ་་་ཞེས་བྱ་བའི་གཟུངས་ཀྱི་རྣམ་པར་བ་ཤད་པ། ་ཌ་ཌ་ཎ། བྱང་ཆུབ་སེམས་དཔའ་སེམས་་དཔའ་ཆེན་པོ་ཐུགས་རྗེ་ཆེན་པོ་དང་ལྡན་པ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། ་་་་། དབུའི་གཙུག་ཏོར་འོད་ཟེར་གྱིས། །རྨོངས་པའི་དྲི་མ་སེལ་མཛད་རྒྱལཌ་ཀགཟུངས་ཀྱི་རྒྱན་དང་ལྡན་པའི་སྐུ། ཎཌོ། ཐུགས་རྗེ་མངའ་ལ་་་ཕྱག་འཚལ་ནས། །ལུང་དང་བླ་མའི་མན་ངག་ལས། ་་གཟུངས་ལ་རྟོགས་པོ་སྐྱེས་པས་འདིར། །ཆ་ཁ། ཤེས་རབ་གཟུངས་དང་བྲལ་རྣམས་ལ། ཎཎཌོ། བརྩེ་བས་གཟུངས་ཀྱི་རྣམ་འགྲེལ་བཤད། །ལོག་རྟོག་ཅན་རྣམས་ཐ་དད་དུ། །རྩོད་།
