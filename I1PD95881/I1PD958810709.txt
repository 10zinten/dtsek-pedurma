་ལོ་ཏོལས་ཀྱི་ཞབས་བསྲེག་པ་དང་མནན་པ་གང་འདོད་པ་བྱའོ། །འཁྲུལ་འཁོར་་འདི་ནི་ཕལ་ཆེར་བགེགས་ལ་:བྱེད་པར་བཤད་ཀྱི་འོན་ཀྱང་རྣལ་འབྱོར་ལས་སུ་རུང་བའི་སྔགས་པས་འཁྲུལ་འཁོར་གཅིག་གིས་ཀྱང་ལས་ཐམས་ཅད་ལ་སྦྱར་བར་་ནུས་སོ། །།འདིས་གདོན་རླག་པར་བྱེད་པ་དང༌། རྨུགས་པར་བྱེད་པའི་ལས་་་གཉིས་འགྲུབ་བོ། །དེ་ལྟར་ཕྲིན་ལས་བཅུ་བདུནི། གཞན་གྱི་ལས་འཇིག་པ་་ནི་སྡིག་པ་སྟེ་དེ་དག་པར་བྱེད་པ་ནི་བུམ་པ་ལ་སོགས་པའི་ཁྲུས་དང་འབྲེལ་པའི་:：ཕྱིར་འདིར་མ་སྨོས་སོ། །ཐམས་ཅད་ལ་:ཡང་སྤེལ་ཚིག་ནི་སོ་སོའི་སྐབས་དང་སྦྱར་རོ། །རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་མན་ངག་པཎྜི་ཏ་སྨྲི་ཏི་ཛྙཱ་ན་ཀཱིརྟིས་མཛད་པ་རྫོགས་སོ། །།།་་་་་་་་་་་་ལ་ལ་བཏོན་པའི་ས་བབ་ཁོབ་དང་ག་པཎྜི་ཏ་ཆེན་པོ་དེ་ཉིད་ཀྱིས་བསྒྱུར་བའོ། །ཨ། ་ས་མ་ཕྱི་ཁ་ལས་མ་དག་བ་ལ། ད་དུ། གླས་། ་ཆོ་གྲོས་ཀྱི་ཁྲོ་བུའི་ཤོ་ལག་ཆ་མི་སྤྲོ་འང་། །པནངཔ་བ། ནང་མདུད་པ་རེ་རེའི་ནང་དུ་ཕ་རོལ་གྱིས་་བ་པོ། བདག་དང་བྱ་བ་དང་གོ་བསྡུར་མཆན། ་་་བྱ་ཚུལ་བཞཚོ་ཁྲི་ཁ་བྱུང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་རྡོ་རྗེ་རྣམ་པར་འཇོམས་་པ་ཞེས་བྱ་བའི་གཟུངས་བཞུགས་སོ། །ཚུ«པེ་ཎཱ་ན་མདྷཱ། «སྣར་ནཱ་མ་དྷ། «པེ་»མའི། «སྣར་དགོས། «པེ་»ནི། @«སྣར་»གིས། @«པེ་བསྡུ། མི«པེ་གདགས། «པེ་»དང་། «པེ་»«སྣར་+པ། («པེ་དང༌། གེ«སྣར་»ན། ཉེ«པེ་»འོག་ཏུ། །«པེ་»«སྣར་པོ། Q«སྣར་»ཡིན། «པེ་ལ། «པེ་པས། ཚུ«པེ་པར་བྱའོ། «སྣར་པས་བྲིའོ། @«པེ་»ད་ན། ༄༅«སྣར་ན་ཡ། ཉི«སྣར་ན་པཱ། «པེ་»ཏྲ་ཊ་ཊ་ཏྲོ་ཊ་ཡ། «པེ་+ལ། ་(«པེ་སྟེ། «པེ་རྡོ་རྗེ། «པེ་»ཡང་། «སྣར་»—ལ། «པེ་»«སྣར་»བཏབ། ཚེ«པེ་»འདི། ཁ«པེ་+ཡང་། «པེ་»བརློབ་པར། འཚེ«སྣར་»ཀྱིས། «པེ་»—པོ། «པེ་»ལ། «པེ་»གཞིལ། «པེ་»«སྣར་བསྲུང་། ཁྱི«པེ་»དབྱེ་བ། ཚེ«སྣར་»+༄༅། །རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་མན་ངག་བཞུགས་སོ། །༄༅«པེ་འདི། «པེ་བཞི་སེར་པོ་ལ་རིན་པོ་ཆེ། «སྣར་»གཞི་སེར་པོ་ལ་རིན་པོ་ཆེའི། «པེ་གདབ་པ་ལ་སོགས། «པེ་»འདི། ཀྱེ«པེ་»གདམས། དགེ«པེ་བས། ཚེ«པེ་»+འཁྲུལ། «པེ་»«སྣར་—ལ། «སྣར་»ལྟ། ཚེ«པེ་»འབྱུང་། །
