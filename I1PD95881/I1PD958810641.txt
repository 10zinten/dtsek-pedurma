རྫོགས་ད་ཁ་བའི་གནའ་བ་ངན་པ་ང་བ་ཁགེ་བགེ་འི་རྟེཊི་ཏི་ཊི་ཁཀཱ་ཊི་ཝི་ཁའོ། RP'ਰੂਸRਕੋਕਕਓਸਸਲਆਕਰਕੇਰਣਬਇA་ཊི་བལ་:ཀ་ཁ་ཁར་བསྡུར་མཆན། །ཀྱང་ཀ་ཀཚེ་ཎཌོ་ན་པོ་དང་། །སྤETག་ཎཱ། aཏན་བྱ་ཨ། ངག་ཀུན་ཏུ་དུ་བ་ཁ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»རྡོ་རྗེ་རྣམ་པར་འཇོམས་་པའི་གཟུངས་ཀྱི་འགྲེལ་པ་བཞུགས། «སྣར་རྡོ་རྗེ་རྣམ་འཇོམས་ཀྱི་གཟུངས་ཀྱི་འགྲེལ་པ་བཞུགས་སོ། །«པེ་»ཁ། དྷཱ་ར་ཎཱ་མ་དྷཱ་ར་ཎི་བྷི་ཏི། སྣར་»དཱ་ར་ནཱ་མ་བྷ་ར་ཎི་པྲྀཏྟི། ནི«པེ་»གདམ། «པེ་བསྟེན། «སྣར་འདོན། «པེ་»ལྷའི། ༄༅«པེ་»«སྣར་ཀྱི། །དུ་བརཔནཔ༈D་། ག«པེ་»«སྣར་གྱིས་བརླབས། ཉི«པེ་ཀྱི་བྱིན་གྱིས་བརླབས། «སྣར་ཀྱིས་བྱིན་གྱིས་བརླབས། མི«པེ་»ལ་སྟོབས། །«སྣར་པ། ག«པེ་ཞེས། «པེ་»《སྣར་»ཤིགས། @«སྣར་བསླུ། ཉི«སྣར་»ཀྱི། ན@«སྣར་གྱི། ::ཀྱི«སྣར་པ་སའི། 《ཀྱེ«པེ་»«ཅོ་»གནོད། ，ཨེ«པེ་གྷུརྞ་གྷུརྞ་ཧཱུྃ། ཀཱ། «སྣར་གླུཎྜ་་་གླུཎ་ཧཱུྃ་ཀཙེ«པེ་པས། སྐྱེཔེ་»གྷཎྞཱ་པ་ཡ་གྷཱཥྛ། «སྣར་གྱུརྞ་པ་ཡ་གྷཱཎ། ཚེ«པེ་»བྷོ་ཊ་ཡ་བྷོ་ཊ། ་ཉི«པེ་»བྷོ་ཊ་ཡ་མམ་བྷོ་ཊཱིཉི«པེ་ལ། «པེ་བླ་ཧྥ་བྲ་ཧྥ། ཚེ«པེ་བླ་ནརབན་ཧྥ་ཡ་སམ་བྲ་ཧྥ། །དུ«པེ་པས། «སྣར་བསྲུངས། ་Q«པེ་»«སྣར་»ཀུ་ཊ་ཀུ་ཊ། «པེ་«སྣར་»ཀུ་ཊ། གྱོ«པེ་ནན། «ཅོ་དྲི་ནཾ། ཨེ«པེ་»«སྣར་གནོན། «པེ་»ག་ཊ་ག«པེ་»«སྣར་བསྲུང་། མེ«སྣར་»རྒྱུ། ཚེ«པེ་»གྱུར་པས། «སྣར་»འགྱུར་བའི། །ད་བ་ས་«པེ་རརྣ་ཏྲ་ཡཱ་ཡ། ན་མཙནྡ་བཛྲ་པཱ་ན། «སྣར་རཏྣ་ཏྲ་ཡཱ་ཡ། ན་མ་རྩ་ཙཎྜ་བཛྲ་པཱ་ཎི། «པེ་ན། «པེ་»«སྣར་»ཅེས། «པེ་»ཏད་ཡ། «པེ་དང་སརྦ་བིརྱ། «སྣར་ལ་སརྦ་བིདྱ། @«པེ་«སྣར་ཅད། ༄༅«པེ་བརླབ། ཉི«པེ་གྷ་ཛ། «པེ་»སྐྱེད། «པེ་བིརྱ་ནས་སུ་བཛྲ་ཡན་ཅད་་བཛྲ་ཨགྷ་ཛ་ལཱ་ཡ་ཀི་ལི་ཀི་ལི། «སྣར་བིདྡྷཱ་ནས། བཛྲ་ཡན་ཆད་བཛྲ་ཨཾ་ཀུ་ཤ་ཛཱ་ལ་ཡ་ཀཱི་ལི་ཀཱི་ལཱ། ཀྱེ«པེ་»ཀི་ལི་ཀི་ལཱ་ཡཱ། «སྣར་»ལཱི་ཀཱི་ལི་ཀཱི་ལཱ་ཡ། («པེ་»ཏྲ་ས་ཏྭ་ས་ཡ་བཛྲ་ཀི་ལི་ཀི་ལཱ། «སྣར་ཏཱ་ས་ཡ་ཏཱ་ཡ་ས་བཛྲ་་ཀཱི་ལི་ཀཱི་ལ། ཉེ«པེ་+དོ། «པེ་དྷཱ་ར། «སྣར་»དྷ་རཱ། (ི«སྣར་—པྲ། ཚེ«སྣར་བྷཉྩ། «ཅོ་པྲ་བྷཉྩ། ཁྱེ«པེ་»ཚིནྡྷ་ཚིནྡྷ། བྷིནྡྷ་བྷིནྡྷ། ཏེཚེ«པེ་»ཀི་ལི་ཀི། དགྱེ«པེ་»ཙནྡྲ་ཀི་ལི་ཀི། «སྣར་»ཙཎྜ་ལི་ཀཱི་ལི་ཀཱི། འགྱེ«པེ་ཏྲ་ས་ཡ་ཏྲཱ་ས་ཡ་བཛྲ་ཀི་ལི་ཀི། འཉེ«པེ་»ཤི་གྲོ་བཛྲ། «སྣར་»ཤཱི་གྷཾ་བཛྲ། འཉེ«སྣར་»«ཅོ་»ཀྱི། །མེ«པེ་»ནཱ་མམནྟ་བཛྲ་ནན། «པེ་གྷ་ཛཱ། །«སྣར་»ཡང་། «སྣར་»—པས། «པེ་»གདམ། &«པེ་ལུ་ནི་རུ་ཕུ་ལུ་གླི་ན་ཀུ་ལུ། མི་ལི་ཙུ་ལུ། ཀུ་རུ་་ཀུ་ལུ་བཛྲ་བི་ཛ་ཡཱ། «སྣར་ལུ་ནི་ཕུ་ལུ་རུ་གྲི་སྒ་ཀུ་ལུ་མི་ལུ་ཙུ་ལུ་ཀུ་རུ་ཀུ་རུ་བཛྲ་བི་ཛ་ཡ། «པེ་ཉུངས་་་་་དཀར། «པེ་ད་ར་ད་ར། དི་རི་དི་རི། དུ་རུ་དུ་རུ། སརྦ་ཀ་ལམ་ཨེ་བར་ཏ། «སྣར་»དྷ་ར་དྷ་ར་དྷི་རི་དྷི་རི་དྷུ་རུ་དྷུ་རུ་སརྦ་བཛྲ་ཀུ་ལཾ་ཨཱ་བར་ཏ། @«པེ་བོས། ཉི«པེ་»ཤ་མུ་གེ་བློ། སརྦ་ཏད་ཁཾ་ཨུད་ཁ། «སྣར་»ཤཱ་ལྟ་།
