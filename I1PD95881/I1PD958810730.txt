དབང་འབྱོར་ཟད་མི་ཤེས་པའི་གཏེར། །དངོས་གྲུབ་འབྲས་བུ་བདག་གིས་ཐོབ། །དགེ་བ་དྲི་མེད་ཡོད་པ་དེས། །བླ་མ་རྣམས་དང་འགྲོགས་པར་ཤོག། རྡོ་རྗེ་་་རྣམ་པར་འཇོམས་པའི་:གཟུངས་ཀྱི་འགྲེལ་པ་རིན་ཆེན་ཕྲེང་བ་ཞེས་བྱ་བ་སློབ་་་དཔོན་ཆེན་པོ་ས་ཝ་རི་པས་མཛད་པ་རྫོགས་སོ། །།།པཎྜི་ཏ་དེ་བ་པུརྞ་མ་ཏི་དང་། ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་ཀྱི་བཟང་པོས་་་་སཨིན། ག་ད་བ་མ་འ་ལ་བློ་འགྱུར་ཞpད་ཏེབསྒྱུར་བའོ། །I"་ཆེ་བདེ་བ་ཅན་དུ་། །བདག། ཤར་ཕྱོགས་པ་ན་བུ། ::ཀྱིབད་བ་ས་་ཉིན་ཏེ། ཀ་བ་གང་འཚུ། »དོ་བདག་ནས་དགེ། ལྔལ་དང་ང་ག་ལ་ཆེ་ཁག་བ་ཏུ་དབྱེ་བ་བྱས་པ་ནི་གལ་གྱི་བུ་རེ་མད། །ལེ་ལོ་མེད་པས་བདག་དབང་། དེ་དེ་ན་ང་ཚོ་ལ་བསྡུར་མཆན། ག་ད་བ་ས་ཀྱི་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»རྡོ་རྗེ་རྣམ་པར་འཇོམས་་པ་ཞེས་བྱ་བའི་གཟུངས་ཀྱི་འགྲེལ་པ་རིན་ཆེན་ཕྲེང་བ་བཞུགས་སོ། །:«སྣར་»རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་་་་་གཟུངས་ཀྱི་འགྲེལ་པ་བཞུགས་སོ། །«པེ་»ཎཱ། Q«པེ་རན། «པེ་»བྷི་ཏི། «པེ་»«སྣར་»རྗེས། @«སྣར་འཇོམས། 《ཚེ«སྣར་བཞིན། ཉི«སྣར་པའི། «པེ་»གྱི། འི«སྣར་»ཀྱི། +«པེ་»གྱིས། ་«པེ་»གྱི་བརླབས། 《སྣར་»གྱིས་བརླབས། །«པེ་»«སྣར་—དེ། འ«སྣར་ལ། «པེ་»ནི། ཚེ«པེ་»«སྣར་པ། O«སྣར་»བརླག«སྣར་»འགུག«པེ་བསྲུང༌། །(ི«པེ་»+ནི། @«པེ་»«སྣར་བསྲུང༌། ན«པེ་»«སྣར་རྨུགས། །«པེ་»བསྲུང་། «པེ་»《སྣར་»བྱེད་པའི། «པེ་»«སྣར་»འོ་ན། །ཚེ«པེ་»འོ་ན། མི«པེ་»«སྣར་»མ་ཧཱ་ཡཀྵ་སེ་ན་་::པ་ཏ་ཡེ་ཧཱུཾ་ཧཱུཾ། «སྣར་བསམ་སྟེ། གྱི«པེ་《སྣར་ཧཱུཾ། «པེ་»«སྣར་»རྡོ་རྗེ། ཉེ«སྣར་ལས། ཀྱིལ་《«སྣར་»རྗེ། རྩེ«པེ་»དཀྱིལ་མོ་དཀྲུང་། བོད་དང་། བྱའི་་་་་་་བར་དང་། «པེ་»«སྣར་—དེའི། @«པེ་»«སྣར་—ཧཱུྃ། «སྣར་བསྒོམས། །ཚེ«པེ་»གནག་མུ«པེ་»བ《སྣར་ན། «པེ་»《སྣར་»—གི། །ཚེ«སྣར་»ནི། «པེ་ཁར་ཧཱུྃ་བསམ་སྟེ། «སྣར་ཀྱང་ཧཱུྃ་་་བསམས་ཏེ། «པེ་»གྷཎྞ་གྷཱཥྛ། ཌཱ་པ་ཡ་གྷཱཥྛ། «སྣར་ཧཱུྃ། །དགྱེ«པེ་»«སྣར་»རྡོ་རྗེ། སྟེ«པེ་»བྷོ་ཊ་ཡ་བྷོ་ཊ་ཡ་ཞེས་པས་སོ། །དང་པ་ཊ་དོ། །དེའི། ན་མ་ར་ཚེ༑་«སྣར་»སཾ། @«པེ་»གྱི། «པེ་»«སྣར་»ཞི་བར་གྱུར་ཅིགཚེ«པེ་པས། Q@«སྣར་»སཾ། «སྣར་ཧཱུྃ། (@སྣར་»ཞིག། «པེ་བསྟན་ནས། ད་སྒོ་བ་སྟོན། «སྣར་བལྟམས། ད་སྒོ་་་་་་བསྟན། ཞི«སྣར་ཡོན་ཏན། «སྣར་ན། །ཉེ«པེ་ན། སྐྱེ«སྣར་»—མདོགཨུ«པེ་»།
