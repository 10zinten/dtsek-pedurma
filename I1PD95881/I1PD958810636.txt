ཏྲ་ཡཱ་ཡ། ན་མཙཎྜ་བཛྲ་པཱ་ཎ་ཡེ། མ་ཧཱ་ཡཀྵ་སེ་ནཱ་ཅིཔ་ཏ་ཡེ། ཞེས་པས་སྙིང་པོ་གསང་སྔགས་དང༌། ཏདྱ་ཐཱ། ཨོཾ་ཏྲ་ཊ་ཏྲ་ཊ་ཞེས་པས་ཁྲོ་བོ་རྣམས་:དང་། ་་་སརྦ་བིདྱ་བཛྲ་བཛྲ་སུ་བཛྲ་ཡན་ཆད་ཡང་སྙིང་དང་བཅས་པས་ལུས་བྱིན་གྱིས་བརླབས་པའོ། ་་། སྙིང་པོའི་གསང་སྔགས་དང་། །ཨོཾ་ཧ་ར་ཧ་ར་བཛྲ་ལ་སོགས་པ་ཕྱོགས་སྐྱོང་བཅུས་མཚམས་བཅད་དོ། །བཛྲ་ཨཾ་:ཀུ་ཤ་ཛཱ་ལཱ་ཡ་སྭཱ་ཧཱ་ཞེས་གཞལ་ཡས་:ཁང་བསྐྱེད་དོ། །སརྦ་:བིཏྭཱ་ནས། ་་་་སུ་བཛྲ་ཡན་་ཆད་བཛྲ་ཨོཾ་གུ་ཤ་ཛཱལཱ་ཡ་ཀཱི་ལི་ཀཱི་ལཱ་ཎིཡ་སྭཱ་ཧཱ། ཞེས་པས་གཙོ་བོ་བསྐྱེད། ་ཙུ་རུ་ཙུ་རུ་ཙཎྜ་:ཀཱི་ལི་ཀཱི་ལཱ་ཡ་སཱ་ཧཱ། ཞེས་པས་རྡོ་རྗེ་གཏུམ་པོ་བསྐྱེད། །:ཏཱ་ས་ཡ་ཏྲཱ་ས་ཡ་བཛྲ་ཀཱི་ལི་ཀཱི་ལཱ་ཝིཡ་སྭཱ་ཧཱ་ཞེས་པས་རྡོ་རྗེ་ཕུར་བུ་བསྐྱེད། ་ཧ་ར་ཧ་་ར་བཛྲ་:དྷཱ་རཱ་ཡ་སྭཱ་ཧཱ་ཞེས་པས་རྡོ་རྗེ་བེ་ཅོན་བསྐྱེད་དོ། ལ་། པྲ་ཧ་ར་པྲ་ཧ་ར་བཛྲ་:པྲ་བྷཉྫ་ཝེནཱ་ཡ་སྭཱ་ཧཱ། ཀ་ཞེས་པས་རྡོ་རྗེ་ཐོ་བ་བསྐྱེད་དོ། ་། གསང་སྔགས་འདི་ཐམས་ཅད་བཅོམ་ལྡན་འདས་རྡོ་རྗེ་རྣམ་པར་འཇོམས་པ་ཡིན་པས་ཁྱད་པར་མེད་ཀྱང་། །བླ་མ་དག་གི་ཞལ་ནས་ཕྱོགས་སྐྱོང་བ་ལ་སོགས་པའི་ལྷ་དགུག་པ་ཤིས་པའི་བསྙེན་པའི་རྩ་སྔགས་ཡིན་གསུངས་སོ། །དེ་ནས་རྡུལ་ཚོན་བཞེངས་སོ། །གཙོ་བོའི་སྔགས་གོང་དང་མཐུན། :：ཚིནྡ་ཚིནྡ། བྷིནྡ་བྷིནྡ། མ་ཧཱ་ཀཱི་ལི་ཀཱི་ལཱ་ཡ་སྭཱ་ཧཱ། བནྡྷ་བནྡྷ། ཀྲོ་དྷ་ཀྲོ་དྷ་:ཀཱི་ལི་ཀཱིྋལཱ་ཡ་སྭཱ་ཧཱ། ཙུ་རུ་ཙུ་རུ་:ཙཎྜ་ཀཱི་ལི་ཀཱི་ཨེལཱ་ཡ་སྭཱ་ཧཱ། :·ཏཱ་ས་ཡ་ཏྲཱ་ས་ཡ་བཛྲ་ཀཱི་ལི་ཀཱི་ཨེལཱ་ཡ་སྭཱ་ཧཱ་ཞེས་རྡོ་རྗེ་གཏུམ་པོ་དང༌། རྡོ་རྗེ་ཕུར་བུ་དང་རྡོ་རྗེ་བེ་ཅོན་དང་། རྡོ་རྗེ་ཐོ་བ་རྣམས་བསྐྱེད་དོ། །མ་ཏིསྦི་ར་བཛྲ། ་་:ཤྲཱི་གྷཾ་བཛཱ་ཡ་ྋསྭཱ་ཧཱ་ཞེས་བརྒྱད་ཀྱིས་ཕེབརྟན་པར་བཞུགས་སུ་གསོལ་ལོ། །:ན་མཿས་མནྟ་བཛཱ་ཎཾ་ཞེས་པ་དང་། བཛྲ་ཨོཾ་:ཀུ་ཤ་ཛ་ཨེལཱ་ཡ་སྭཱ་ཧཱ་ཞེས་དབང་བསྐུར་རོ། །གཞན་ཡང་འདི་ནི་ལས་ཐམས་—།
