པའི་དལ་བློའི་ཀ་ཏོ་དང་ཁ་བྱ་མི་ལ་མོའི་འདི་ལ་བསྐྱར་དུ་ཟླ་《@«པེ་»ཏ་ནིར་མ་ཎ་ཀ«སྣར་»ཏཱ་ནི་རྨ་ཎ་ཀཱ། ཚེ«པེ་»«སྣར་དེའི། །སྣར་བསོབ། «པེ་»། །ཁ་ཨཱ་ཡུ་ཤེ། 《སྣར་ཨཱ་ཡུཥ། «པེ་མརྟ། 《སྣར་མརྡ། ཨེ«པེ་»གཞགཨེ«པེ་»«སྣར་»ཀཚེ«སྣར་ཊ་ཁརྟ། །ཁྱེ«པེ་»—ཞེས་པས། ཏེ«པེ་»«སྣར་དུ་བོ། ཚེ«པེ་»+གྱི། །གེ«པེ་»ན། ཚེ«པེ་»—ཞེས། དང་འགེ«པེ་»«སྣར་»ཏྭཾ། འདེ«པེ་»ད། ཚེ«པེ་g། སོ་མྱཀ་སཾ། «སྣར་བྷེ། སམྱག་སཾ། གདུ«པེ་ཕྲེ། ཉེ«པེ་»དྷརྨ། འགུ«པེ་»«སྣར་»ཡེ། འཉེ«པེ་»དྷ་དུ། འཚེ«པེ་»དྷིཥྛ་ཀྵ། འགུ«པེ་བརྟན། «སྣར་བསྟན། ཚེ«པེ་»མཚོ་བདབ། །ང་དང་ང་བཏོན་ཏེ། །འཚེ་ཤིཏུ། །འོག་ན། བོད་མདོ་«པེ་»དུ། «སྣར་»སྐོར། «སྣར་པ། «པེ་»ཚ་ཚའི། «སྣར་»ཚ་ཚ། «པེ་»+ཡང་། བར་མ། @«པེ་མཆུ་འཆང་། «སྣར་ཆུ་འཕྱང་«པེ་»གིས། ཉི«པེ་»དགོད། «པེ་བའི། ཁྱེ«པེ་»་མ་དང་། །སིང་གྲི་བྱཱ་གི་རི་ཀར་ནི་ས། 《སྣར་»སི་གྲི་བྱ་གྲི་གར་ཀར་ནི་པ། (འགྱེ«སྣར་བྲི། །ཡེ«པེ་སྒྲུང་། (ཉེ«པེ་»ཏ། «པེ་སྨྲས། »@«པེ་《སྣར་»«ཅོ་»འིགྲྭ། O«པེ་»དུ་རུ་ཀ《སྣར་»ཏུ་རུ་ཀར། ༄༅«པེ་ལྷོ་རུའོ། ་«པེ་ལ་རེ་ཤིག་ཚ་ཚ། «སྣར་རེ་་་ཞིག་ཚ་ཚ། «པེ་ནས། ཁྱོ«པེ་»ཏེ། འགེ«སྣར་»+མཆོད། (《སྣར་»—དང་། ཙེ«སྣར་རྗེ། :ཀྱིཚེཚེ«པེ་པའི། །ཕྱི་མའི་:::བཞི་ཁྱོད་ལ་དགའ་བ་དང་། «པེ་»པོའི་འདིའི། «སྣར་པོ་འདི། «པེ་བསྒྲུབ། Q«པེ་»སྤྱན་གྱིས། «པེ་»ཁ། ཚེ«པེ་»བྱ་ཀྱི་བལྟས་དམན་ནི་རག《སྣར་»ལྟས་དམན་ན་རིག@«པེ་མཆེད་པ་དང་གཏོར་མས་ཚིག«སྣར་གཏོར་མ་ཀོང་དང་མཆོད་པ་དབུལ་ཞིང་ཆ། @«པེ་»«སྣར་»ཚ་ཚ། «པེ་རེའུ། @«པེ་»འཁོར། །ཁྱོ«པེ་པ། :《ཀྱེ«པེ་»ཚ་ཚ་རེ་རེ་བརྒྱན། «སྣར་»ཚ་ཚ་རེ་རེ་རྒྱན། འགུ«པེ་ལ་ལེར། །(ཉེ«པེ་»+གི། «སྣར་རྣམས་ཀྱི། །«པེ་»ཆུང་ངུའི། 《སྣར་ཆུང་ངུན་གྱི། ཚེ«པེ་»པའི། «སྣར་པ། །སྐྱེ«པེ་སྣོད་དང་རྒྱ་སྤོས། ལེ«པེ་»—འམ། ་འཁོ«པེ་ལ། དགྱེ«པེ་»བྲི་སྟེ། དབྱེ«པེ་»«སྣར་»ཚ་ཚ། (བྱེ་བ། །«པེ་»«སྣར་+དང་། «པེ་»«སྣར་»«ཅོ་»བཅད། «པེ་»པ། «སྣར་»གི། «པེ་»གསལ། ཡ་ཏིs/ ོ«པེ་»—འམ། «པེ་ཡང་། «པེ་»བྱ་དང་སྒྲུབ། «པེ་ལྟས། ཁྱེ«སྣར་»ཀྱི། ཀྱེ«པེ་»«སྣར་»ཊ་ཊི+གྱི། (ཉེ«པེ་བསྒྱུར། །ཆཀ་ཀྱེ«པེ་»སྣར་»ཚ་ཚ། ཨུ«པེ་རྣམ་པར་སྣང་མཛད་སངས་རྒྱས་ཀྱི་སྐུ་ཁོ་ནར་བསམས་ཏེ། སཱཙྪ་རེ་རེ་ཡང་། «སྣར་རྣམ་པར་སྣང་མཛད་སངས་རྒྱས་ཀྱི་སྐུ་ཁོ་ནར་བསམས་ཏེ། ཚ་ཚ་རེ་རེ་ལ་ཡང་། ཙེ《སྣར་རེ། ༄༅«པེ་»བ་སྤུའི་གསེབས།
