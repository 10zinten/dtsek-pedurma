ལ་ན་ཡང་། རྒྱུན་སྐད་དོན། དེ«པེ་»གི་རྒྱུ་ནི་བསྔགས། 《སྣར་»གིས་རྒྱུ་ནི་སྔགས། (《སྣར་»སྨྲས། །ཞི་ལེགས་པར་ཚོད་ཏེ་ལྟཞན་ན་(«པེ་»ཀྱིས་ཐོབ་པར་གྱུར། «སྣར་»ཀྱི་ཐོབ་པར་འགྱུར། «པེ་»གིས། 《སྣར་གི། །《སྣར་»མི་ཟད་བ་ཀྱིས། «པེ་སྒོམ། ས《སྣར་གྲུབ། 《སྣར་རྣམས། «པེ་+རོ། ཝོ《«པེ་རྟགས། ་«པེ་»དུ་བརྟག::ེ«པེ་»མཁའི། །(«པེ་པའི། པི«པེ་»ཅིག་གིས། 《སྣར་གཅིག་གིས། 《པེ་»«སྣར་+ན། /ཊ། ཁ་རལེཏྟཾ་ག་སྐྱེ་བ། སྐྱེ་དང་། ལོ་ལྟར་《སྣར་ཀྱི། «པེ་སྒོམ། 《སྣར་བ། «པེ་རླུང་། (《པེ་»«སྣར་»སྐྱེད། «པེ་»བས་སྐྱེད། །《སྣར་རྒྱུན། «པེ་བྱ་བའི། 《སྣར་བྱའི། །རྒྱ་མི་ཁ། ཁན་«པེ་»ཤེང༌། «སྣར་»ཤེས། །འ«པེ་བལྟ། ་འ«པེ་»རྟགས། ༄༅«པེ་ཞེས་བྱ། ས«པེ་འབྱུང་། ་《སྣར་པར། @《སྣར་»མེ་དང་ཆུ་དང༌། 《སྣར་»+ཀྱང་། །«པེ་རྟགས། ་ཁོ«པེ་»བྱིན། «པེ་གཤིག«པེ་《སྣར་»འང་། །(《སྣར་ནས། དང་། 《གངས་ཏེ། དང་། 《དཔེ་ཀར་《སྣར་རྒྱུ། །«པེ་གྱི། །འ«པེ་»«སྣར་»ལ་ལྟ། ་«པེ་»«སྣར་བས། (ས«པེ་»མ་ལ། །《སྣར་»ཆ། པོ«པེ་»བཟློས། །«པེ་གྲུབ། 《སྣར་བབས། ་《སྣར་རང་། :«པེ་»འཁྲིས། «སྣར་བབས་པའི། སྣར་བསལ། «པེ་སྐུར། །།།་སྒྲུབ་«པེ་བར། «པེ་ལས། «པེ་»གཉེས། «པེ་»«སྣར་»པའི། ཚེ་(«པེ་»བ་ལ་སོགས་པ། །«སྣར་»བའི་ལ་སོགས་པས། པ«སྣར་»གྱིས། «པེ་»སྒྲུབས། («པེ་»བོད། «སྣར་གསུམ། :《སྣར་+གྲུབ། །«པེ་»《སྣར་གནོད། (《སྣར་»ཡིན། ::«པེ་+རྣམ། «སྣར་+རྣམས། པ་«པེ་ལྷུང༌། «པེ་དེ་ཉིད། 《སྣར་མིང་། །+«པེ་དེ། ས«པེ་»འབྲང་མི། 《སྣར་»འབྲངས་མི། «པེ་»གིས་སྒྲུབ་པ། 《སྣར་གྲུབ་པ། །(«པེ་»འགྲོལ། །e(«པེ་»«སྣར་»བ། །:«པེ་»ཐམས་ཅད་དུ། 《སྣར་»དེ་ཐམས་ཅད་དུ། («པེ་བསོགས། ::«པེ་《སྣར་ན་སྒྲུབ། «པེ་»ཏེ། 《པེ་»«སྣར་»ཀྱིས། «པེ་ཟློག(«པེ་»པར། ་པ«པེ་»འདིར། «པེ་+ཡེ༑གོས་ན། 《སྣར་གྱི། «པེ་སྙེན། 《སྣར་ཆོ་ག«པེ་»བསྒྲུབ། ས«པེ་རྗོད། ་«པེ་པལ། །«པེ་»སྟར། ་《སྣར་བྱུགས། /«པེ་»«སྣར་སྒྲུབ། །(«པེ་»«སྣར་»+ཞེས། ཀ་:(«པེ་རྡ། །（《པེ་»ཞིམ། ལ། །ནས། སྤྱི་ལ་ས་བ་ནད། ད་སྣར་—སྣ། «པེ་»སྒྲུབ། «པེ་»《སྣར་བྱ། །«པེ་སྐུར། («པེ་»«སྣར་—དང་། «པེ་ལས། «པེ་+ཐམ་པར། ལ། འ«སྣར་»གྱིས་བརླབས། «པེ་»ད་རྭ། ་«སྣར་དར། །ཆོ་ག(«པེ་»«སྣར་བསྟན། «པེ་ལྟ། 《པེ་»སྐྲད། སྟེ«པེ་»«སྣར་»ཁོར། 《པེ་»ཅད། །《སྣར་བཅད། ལ«པེ་»རྗེ། ལ«པེ་བ། གྱི«པེ་»«སྣར་རྣམས། «པེ་»རྗོད། «པེ་《སྣར་»ལས། «པེ་རྗོད། 《སྣར་»གདུགས་འདེབས་པ། ད་དེ་ར་རྒྱུ་བ་་«པེ་བྱ། «པེ་»«སྣར་»+ནི། ::«པེ་»«སྣར་»གིས། «པེ་»ཟློག(«པེ་»«སྣར་།
