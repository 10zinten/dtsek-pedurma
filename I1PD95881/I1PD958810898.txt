འགྱུར་བ་ཡིན། །དེ་ནས་གཉིས་སུ་བགོས་ནས་ནི། །ཁོ་། ཆ་ཆེན་ཉི་ཤུ་བཞི་རུ་་་་འགྱུར། །ཁྱོ། དེ་ནས་ཆ་ཕྲན་བགོས་ནས་?ནི། ཎ་། ཚངས་པའི་:ཐིག་གི་ངོས་་་གཉིས་ཀྱི། །ཆ་ཆེན་དང་:པོའི་ཕྱེད་དུ་བགོ༄༅། །ཕྱི་ཡི་ཆ་ཕྱེད་དེ་བཞིན་བགོས། །ཆ་ཆེན་གཉིས་པའང་ཕྱེད་དུ་བགོ། །ནང་གི་ཆ་ཕྱེད་དེ་བཞིན་བགོ༄༅། །ཆ་ཆེན་གསུམ་པའི་མཐའ་ལ་གཅིག་། ཆ་ཆེན་བཞི་པའི་ཕྱི་ནང་གི། །ཆ་རེ་་་བཏབ་ཅིང་ལྔ་པ་ཕྱེད། །ཆ་ཆེན་དྲུག་པ་ལྷ་སྣམ་འགྱུར། །།ཆ་ཆེན་བདུན་པ་ཕྱི་་་ནང་གི། །ཆ་རེ་གདབ་ཅིང་དབུས་གཉིས་བཞག། ཆ་ཆེན་བརྒྱད་:པ་ཕྱེད་དུ་བགོ༑ཨོ་་། ཆ་ཆེན་དགུ་པ་བཞི་རུ་བགོ། །ཆ་ཆེན་བཅུ་པ་ཕྱི་ནང་གི། །།ཆ་རེ་་་གདབ་ཅིང་དབུས་གཉིས་གཞག། ཆ་ཆེན་བཅུ་གཅིག་ཕྱེད་དུ་བགོ། །ཆ་ཆེན་བཅུ་གཉིས་ཁོར་ཡུག་འགྱུར། །།ཕྱོགས་གསུམ་དུའང་དེ་བཞིན་ནོ། ཡང་། །སྒོ་གཅིག་བྱེད་ན་ཚངས་ཐིག་ཆེདང༌། །་གི་། ཕྱོགས་གཅིག་ཟུར་ཐིག་གདབ་བྱས་ནས། །།ཚངས་པའི་ཐིག་གི་ངོས་གཉིས་སུ། །ཆ་ཆེན་ལྔ་ལྔར་བགོ་བྱས་ནས། །།ཆ་ཆེན་བཅུ་རུ་འགྱུར་བ་ཡིན། །དེ་ནས་གཉིས་སུ་བགོ་བྱས་ནས། །ཆ་ཆེན་ཉི་ཤུར་་་་འགྱུར་བ་ཡིན། །དེ་ནས་ཆ་ཕྲན་བགོ་བར་བྱ། །དེ་། ཚངས་པའི་ཐིག་གི་ངོས་་་གཉིས་ཀྱི། །ཆ་ཆེན་དང་པོ་ཕྱེད་དུ་བགོ། །ཕྱི་ཡི་ཆ་ཕྱེད་དེ་བཞིན་བགོ། །ཆ་ཆེན་གཉིས་ནི་ཕྱེད་དུ་བགོ། །ནང་གི་ཆ་ཕྱེད་དེ་བཞིན་བགོ། །ཆ་ཆེན་གསུམ་པའི་མཐར་ཆ་གཅིག། ཆ་ཆེན་བཞི་པ་ལྷ་སྣམ་འགྱུར། །ཆ་ཆེན་ལྔ་པ་ཕྱི་ནང་གི། ྋ་། །ཆ་རེ་གདབ་ཅིང་དབུས་གཉིས་གཞག། ཆ་ཆེན་དྲུག་པ་ཕྱེད་དུ་བགོ། །ཆ་ཆེན་བདུན་པ་བཞི་རུ་བགོ། །ཆ་ཆེན་བརྒྱད་པ་ཕྱི་ནང་གི། །ཆ་རེ་གདབ་་་ཅིང་དབུས་གཉིས་གཞག། ཆ་ཆེན་དགུ་པ་ཕྱེད་དུ་བགོ། །།དེ་ལ་བསྐོར་བ་རིམ་གཉིས་བྱ། །གསུམ་པ་མཐའ་ཡི་:ཆ་ཕྲན་སྐོར། ་་། ཆ་ཆེན་བཅུ་པ་ཁོར་ྋཡུག་་་་།
