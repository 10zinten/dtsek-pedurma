་བསྟན་འགྱུར། རྒྱུད། ཏིཁུ་བས་བསྟན་ཏོ། །གཟུངས་སྔགས་ཀྱི་སྦྱོར་:བའི་ནུས་པ་ནི་གོ་སླ་བས་མ་བཤད་དོ། །གོས་སྔོན་ཕྱག་ན་རྡོ་རྗེ་ཡི། །གཟུངས་ཀྱི་འགྲེལ་པ་འདི་བརྩམས་པས། །དགེ་་བ་ཅུང་ཟད་གང་ཐོབ་པ། །དེ་ཡིས་འགྲོ་ཀུན་ནད་མེད་ཤོག། །པ་བྱ་བ་ལ། དཔལ་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་གཟུངས་ཀྱི་འགྲེལ་པ། །རིག་པའི་གནས་ཐམས་ཅད་ལ་མཁས་པའི་པཎྜི་ཏ་ཆེན་པོ་ཀརྨ་བཛྲས་མཛད་པ་རྫོགས་སོ༑། ་། རྟ་ར་ཊའོ། ཞེས་གཏོང་ཚོའི་རབ་ཁའི་ལ་བ་ཏུ་པ་ལ་ལ་ཁརཚཚ། ཁ་ལུ་རྒྱུད་དེ་ང་ད་ན། ས་ལ་ལོ་ཏཚོ་དུས་ལུ་བྱ་ཆུའི་ཁོང་ཁང་ན་ཏིག་བྱེད། །ལ་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་འཕགས་པ་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་གཟུངས་ཀྱི་འགྲེལ་པ་བཞུགས། «སྣར་»འཕགས་པ་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་ཅན་གྱི་གཟུངས་་ཀྱི་འགྲེལ་པ་པཎྜི་ཏ་ཆེན་པོ་ཀརྨ་བཛྲ་གྱིས་མཛད་པ་བཞུགས་སོ། །®«པེ་»དྷ་ར་ཎི་ནི་བེ་བྲི་ཏི། «སྣར་»དྷཱ་ར་ཎི་ནི་བེ་བྲིཏྟི། «པེ་»པོ་ཅན་གྱིས། «སྣར་པོ་ཅན་གྱི། «པེ་»+བྱེད་པར། «པེ་»«སྣར་»མདོ་ལས། @«པེ་བརྗོད། ༄༅«སྣར་»སྟོབ། «པེ་ཀྱིས། «སྣར་བྱའོ། ་་་་«སྣར་བརྗོད། ་«པེ་གྲོ། «པེ་གྲོ། «སྣར་འགྲོ་མ། «སྣར་»ཀྱི། «སྣར་»རྗེད། «སྣར་»དགོད། ཚེ«པེ་རིག«པེ་»བ། «པེ་»«སྣར་»ཆོས། ཁྱེ«པེ་»མཛེ། ཀྱེ«པེ་གྱིས། «པེ་»—དང༌། ཉེ«པེ་»—པས། ཚེ«པེ་»ནོ། ཁྱེ«སྣར་ཀྱི། ཁྱེ«པེ་ཆེན་པོ། ཁྱེ«པེ་»—དཔོན། ལེ«པེ་»ཀྱིས། ཨེ«སྣར་ཟ། འ«པེ་+རྗེས་སུ། ཉི«པེ་»ཚིགས། ་«པེ་»བ། «པེ་»ཀྱིས། @«སྣར་»བསྲེགས། «པེ་»གི་དབུགས་ཀྱིས། «སྣར་»གིས་དབུགས་ཀྱི། «སྣར་»ཀྱི། @«སྣར་བསྐྱེད། །«པེ་»ཀྱིས། འགེ«སྣར་བརྒྱན། ཀུན་ཀྱང་དམ་པའི་«པེ་»ཉེར།  ེ«པེ་+པས། «པེ་»པོའི། «པེ་»དུག«པེ་»གྱིས། ཚེ«པེ་»—པ། «པེ་གྲོ། «པེ་»སྡིགས། «པེ་»«སྣར་ཐོག་མ། «སྣར་»འགྱུར་རོ། «པེ་»«སྣར་ལས། «སྣར་སྒུགས། «པེ་»ཐོགས། «སྣར་»འཇིགཚེ«སྣར་»རྗེའི། ཚེ«པེ་»མདོག«པེ་»རྡབས། དགྱེདདགྱེ«སྣར་»བཟང་། རྒྱལ་པོ་ལ་སོགས་ཏེ་—།
