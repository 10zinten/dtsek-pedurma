ཁཝ་བློ་རྫོགས་པ་་བ་ཚོ་ཁག་ལ་ད་མ། ་ཏུ་ང་ཀོ། ཀྱི་ནང་ཚོ་ཁ་ཆེ་བ་ན་བ་ལཊ་ཋ་ཌ་ཀ་ཏ་ས་ཁན་ལམས་ཐག་དུམ་གྱི་བདེན་པ་བུ་ག་བམའོ་བྱ་བ་ལ་བཀའ་མ་པ་ལ་གོ་བ་གཏིང་:ལྷ་ཁང་ར་ཞབས་ཀྱི་ཆུས་ཀྱི་བསྡུར་མཆན། :ཆར་བ་ཀྱི་ནང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་རྡོ་རྗེ་རྣམ་པར་འཇོམ་་་་པའི་གཟུངས་ཞེས་བྱ་བའི་རྣམ་པར་བཤད་པ་བཞུགས་སོ། །《སྣར་རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་དོན་རྣམ་པར་་བཤད་པ་བཞུགས་སོ། །«སྣར་ནཱ་མ་དྷ། «པེ་»+བ། «སྣར་»གི། «སྣར་+དང་། «པེ་དེས་ན༑«པེ་»«སྣར་མདོ་ཡིས། «སྣར་དགོངས། «པེ་»—ཏོ། «སྣར་ནོ། «སྣར་པར། «སྣར་»གི། «པེ་རྒྱ་ལམ་གསུངས་པས། «སྣར་»བརྒྱ་ལམ་དུ་གསུངས་པར། །«སྣར་»ཞུགས། «པེ་བྱ་བའོ། །@«པེ་»ཀྱིས། 《«སྣར་»«ཅོ་»གྱིས་བརླབས། །*བླ་ཙཚ༽། ནས་པ་p?_«སྣར་»བ། Q«སྣར་གྱི། @«སྣར་—ཀ«པེ་སྟོན་ཏེ། ཨུ@«སྣར་གི། ༄༅«པེ་པས། ཉི«སྣར་ཀྱི། །»པོ། །ཚེ*ཀློ་ཀྱ་པ༽*། ་་འགྱུར་ལ་་D«པེ་»+ཞེས་པ། «སྣར་»གྱི། «པེ་»«སྣར་»ཏོ། «པེ་»+ལས། ག«པེ་»འཇིགས། ：«སྣར་»གི། ཚེ«སྣར་ཀྱི། ཉི«པེ་བརླག་«སྣར་»«ཅོ་»ཏོ། «པེ་»«སྣར་ནས། ཉི«པེ་»ཅིང་། &«སྣར་»—པའི། ཚུ«པེ་བསྐོང་། ཕྱི་ཚེ«པེ་བསྲུང་། @«པེ་»འདོགས། «པེ་བསྲུང་། @«པེ་ཚིནྡ། ཚེ«པེ་»ཚིནྡ་ཚིནྡ། ཀྱེ«པེ་རྣམས། གེ«སྣར་»+མཛད།  ེ«པེ་»«སྣར་ཅན། @«སྣར་»སྟེ། ན«པེ་པ་ནི། «པེ་»«སྣར་»ཀྱི། «སྣར་ལྔ། @«སྣར་གྱི། :པེ་»«སྣར་»ན། «པེ་»སྟོན། «པེ་»པ་དེ། ཀྱེ«པེ་»པ་ཞེས། ཀྱེ«སྣར་»+གྱི། 《སྣར་བསྲེགས་པ་བསྲེགས། དེ，P། །}ས་སྐྱ་ཚུན་བ་«སྣར་བརྡུངས་ཤིག་བརྡུངས། @«པེ་»ཎྜ་གླུ་ཌཱ། Q«པེ་»ཀྱུས་ཟུང་ལ་སྐྱོད་ཅིག་སྐྱོད། «སྣར་ཀྱི་་་་ཟུང་ལ་སྐྱོབ་ཅིག་སྐྱོབ། «སྣར་བསྐྱོད། @«པེ་»«སྣར་ཁོང་དུ། «སྣར་—མི། «པེ་»ཅིག«པེ་»བླུད། «སྣར་གླུ། ཁོ«པེ་གཏུགས། ཀྱེ«སྣར་བརྡུངས། ཉེ«པེ་《སྣར་»ཏུཾ། སྐྱེ«སྣར་»དགྲ། «སྣར་»བསྲེགས། ཁྱེ«སྣར་—པའི། །«སྣར་»རྗེ་བདག་གིས། «པེ་ཊ་ཏ་ཧ་ས་བཛྲ་བཛྲ། ནཱི་ལ། མི«སྣར་གྲོལ། «སྣར་ཀྱིས། ཚེ«པེ་»«སྣར་ཕུ་ལུ་ཎི་རུ་ཕུ་ལུ། པོ«པེ་»གླི་ན་ཀུ་ལུ། «སྣར་བྷིཉྩ་ཀུ་ལྷུ། ༄༅«སྣར་»གསུམ། ཉི«པེ་»ཆོམ། «པེ་»ཙུ་ལུ། ཀྱེ«སྣར་»ཀླུ། འགྱེ«པེ་»«སྣར་སྦྱོངས། ཉེ«པེ་ཀི་ལི་ཀི། ཉེ«པེ་»བོའི། སྐྱེ«པེ་སྐོངས། ཨེ«པེ་ན་པྲ་མོ་ཊ་ནཱ། དེ་—།
