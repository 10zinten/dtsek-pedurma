བསྟན་འགྱུར། རྒྱུད། ཚོ་དུའི་pgnཔོདུག་གོ། །མཐོང་བ་དང་གཉིས་སོ། །:ཀཱ་ར་ཊ་ནི་ཀླུའི་དུག་རེག་པའི་དུག་གོ༑། འཛིན་པའི་དུག་གོ། །།ཀེ་ཡུ་རཱ་ནི་བུམ་པའི་དུག་ཡིན་ཏེ། དེ་ནི་རེག་པའི་དུག་གོ། ཁ་། :བྷཱུ་ཏངྒ་མཱ་ནི་འབྱུང་པོ་རྣམས་ཀྱི་མཐོང་བའི་དུག་ཡིན་ཏེ། དེ་ཡང་དུག་བཞི་གནས་པའོ། །བྷཱུ་ཏ་པ་ཏི་ནི་མི་མ་ཡིན་གྱི་དུག་ཡིན་ཏེ། དེ་ཡང་མཐོང་བ་དང་། །རེག་པ་དང༌། བསམ་པ་ལས་བྱུང་བའི་དུག་:ཡིན་ཏེ། བིནྡྲ་པ་ཏི་ནི་ཐིགས་པའི་དུག་ཡིན་ཏེ། དེ་ཡང་བསམ་པ་དང་མཐོང་བ་ལས་བྱུང་བ། ཀླུའི་དུག་ཕོལ་མིག་གོ། །སི་རི་པ་ཏི་ནི་དེ་ཡང་ཀླུའི་དུག་ཡིན་ཏེ། དམུ་ཆུའི་ནད་དོ། ་། ཏེ་ཛོ་ཨེཔ་ཏི་ནི་ཀླུའི་དུག་མེ་དབལ་གྱི་ནད་ཡིན་ནོ། །ཡ་ཤོ་བ་ཏི་ནི་དེ་ཡང་ཀླུའི་ནད་ཡིན་ཏེ། མཛེའི་བྱེ་བྲག་གོ། ༽། ཡ་ཤོ་གྲ་པ་ཏི''ནི་ས་བདག་གི་དུག་ཡིན་ཏེ། །།འབྲས་ཀྱི་ནད་དོ། །:ཨ་ར་ཌཱ་ནི་ས་བདག་གི་་རིགས་གཡའ་བའོ། །ཏ་ར་ཌཱ་ནི་ས་བདག་གི་རིགས་ཤ་བཀྲའི་ནད་དོ། །ཏ་ར་ད་ནི་འབྲུམ་བུའོ། །ཏུ་ཏུ་རཱ་ནི་ལྷོག་པའི་ནད་དོ། །:དནྟ་ཧ་རཱ་ནི་མཆེ་་བ་ལ་སོགས་པ་ལས་བྱུང་བ་སེམས་ཅན་དུག་གི་སོའི་དུག་གོ། འགའ་། ཧ་ལ་ཧ་ལ་ནི་་འཇིག་རྟེན་དྲག་པོའི་དུག་ཡིན་ཏེ། །སེམས་ཅན་ནང་གི་དུག་གོ། །མེ་:ལཱ་ཧ་ལ་གྲ་ནི་སྦྱོར་བའི་དུག་གོ། །པཱ་ལ་ནི་འབྲས་བུ་ལས་བྱུང་བའོ། །e/། གུ་ཧ་ནི་་་་གསང་བའི་དུག་གོ། །གསང་ཡིབའི་དུག་ནི་དམ་ཚིག་ལས་བྱུང་བའོ། །ཙ་རི་ནི་བརྟན་པ་སྟེ་མི་འགྱུར་བའོ། །:དནྡཱ་རཱ་ནི་བཙན་དུག་གོ། །ཨི་རི་:ཀི་ཙི་ཀཱ་ཏི་ནི་རྩ་བའི་དུག་གོ། །ཀམྦྷ་ནི་བལྟ་བའི་དུག་གོ། །ཤ་:ཏ་དརྫུ་རཱ་ནི་ལོག་པའི་དུག་གོ། །བི་པུ་ལེ་ནི་སྤྲིན་གྱི་དུག་གོ། །ན་ཀུ་ལེ་ནི་སྲེ་མོང་ལས་བྱུང་བའོ། །ཀི་རཱི་པི་ནི་ཕྱི་བའི་དུག་གོ། །།ཏ་:རངྒཱ་རིཊྚ་ནི་སྲིན་བུ་ཀི་ཏིའི་དུག་གོ། །ཨ་་མྲ་མ་ཏི་ནི་སྦྲུལ་ནག་པོའི་དུག་གོ། །།ཛམྦུ་ཙེམ་ཏི་ནི་ཤ་དུག་གོ། ལ་། མ་དྷུ་མ་ཏི་ནི་།
