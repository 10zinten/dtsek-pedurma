བསྟན་འགྱུར། རྒྱུད། ཆུ་སྐྱེས་བསྣམས་པ། སྐུ་མདོག་དཀར་མོའོ། །ལས་ཀྱི་རྗེས་སུ་འབྲང་བའི་གཟུགས་ཁ་དོག་དཀར་པོ་ལ་སོགས་པ་དང་ལྡན་པར་བསྒོམས་ནས། རང་གི་སྙིང་གའི་པཾ་ཡིག་གི་ཡེ་ཤེས་ཀྱི་འཁོར་ལོ་སྤྱན་དྲངས་ཤིང་མདུན་དུ་ཡང་དག་པར་བཞུགས་ནས། མཆོད་པ་ལ་སོགས་པ་བྱས་ཏེ་རབ་ཏུ་བཅུག་ནས། དེ་ནས་སྔགས་བཟླས་པར་བྱ་སྟེ། ཨོཾ་བཛྲ་ཙརྩི་ཀེ་ཧཱུྃ་སྭཱ་ཧཱ་ཞེས་སོ། ་། རྡོ་རྗེ་:ཙརྩི་ཀའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།'ཧྥ་ཟོ། །བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཉི«སྣར་ཙར་ཙི་ཀེ། «སྣར་»ཙར་ཙི་ཀེའི། «སྣར་ཙར་ཙི་ཀེ། ཚེ«སྣར་»གས། «སྣར་རྗེ། «པེ་»«སྣར་»ཙ་ར་ཙི་ཀེ། ད་ནརར། «པེ་»«སྣར་»བརྒྱན། །དབུ། །ནང་ན། །ལན། «པེ་»«སྣར་»དམར། «སྣར་»ཙར་ཙི་ཀེའི། །མཚོ་བྱ་ནི་ཅིཊོ། ༼ལཥ་ཁཎྜ་ང་ཚོའི་ཟླ་ཁོ་དང་། ལས་ཀ་རྒྱུད་ཊིཀ་ཏོག་ལ་ལ་༼ཀ་ཀ་ཁྲོ་ཀའི་ང་ཁ་ཚོ་ངོས་ཏན་། དེ་ན་དེ་བློ་དེ་ལ་ལྟ་བསྡུར་འབྲས་རེའུ་མིག་སྤྱད་དབང་ངམ་མག་ཀ། ཁ་ཏོ་ལ་འཝི་དི་སུ༌ཞུ་ཁརྒྱུད། （ཁབྱེད་ཚུལ་བློ་། pད་ལ་་ས་ལ་ལཱ་ཎ་ཏepཀ་བ༌ཀ་ཀ་ཎཁ་ཁཀྱི་:་བདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། .། ，སྣར་ཐང་། ཅོ་ནེ། །ཚུལ་སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། བོད་རྟགས། ཚན་གྲངས། །ཤ་%ར་སྐདེལྡེབ། བན། བཙན་བ་ན། བ་ན་ང་ད་ཕྲེང་། །ཚེག་ཁྱིམ། །་། ལ་ལ་ཅི་། །བྱིས་པ་ལ།
