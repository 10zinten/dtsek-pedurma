རྡོ་རྗེ་དབྱངས་ཅན་མའི་སྒྲུབ་ཐབས། མཐའ་ཨི་ཡི་དབྱངས་ཀྱིས་མཛེས། །ཟླ་ཕྱེད་ཐིག་ལེས་མནན་གྱུར་པ། །བླ་མ་ལས་བལྟ་བྲི་ཞིང་བཟླས། །རྡོ་རྗེ་དབྱངས་ཅན་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།།དུན་བརྗོད་པ། འབ་བྱའོ། །་་་་་ས་བ་ཚུ་ལུ་ལྟ་བ་འདྲ་བ། །དེས་ཚོད་ཐམས་རུ་སྟོང་ཡོད་དུ་བསྒྲོན་པ་ལ་བྱའོ། །དེ་ནས་མདུན་དུ་མ་བོ་ཀླུ་བདུན་གྱི་ཚུལ་བསྡུར་མཆན། བའོ། །ཞེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»སྟི། ཚེ«པེ་»«སྣར་»+པ། ༄༅«པེ་»«སྣར་ཧྲི། ཚེ«པེ་»«སྣར་དག«པེ་»«སྣར་པོ་ཏི། འོ་༼བལ། @«པེ་»«སྣར་ཨཱི། །ང་བྱང་བུའི་འོད་ཟེར་དང་འགྲོ་བ་དུ་དོ་དམ་དང་དཔོག་པར་བསྡུས་ནས། །ས་བུ་ལྟར་བག་མེད་པ་ཡང་དག་པར་བ། ་པའི་བྱང་ཆུབ་ལ་བསྡུར་འབྲས་རེའུ་མིག་བར་ནང་དུ་། ་་ན་ཁོ་ན་ཊི་ཊ་ཀ་རྒྱུད། （ནགས་ཞུ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། ་རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུ་བརྒྱུད། །པོད་རྟགས། མུ་ཡ། དུ་ཚན་གྲངས། །།།ལ། ་ལྡེབ། /བཙ—ནན་བ?ན་_བ་བན་རྒྱུད། །A/འཚེག་ཁྱིམ། ཚན་:དང་པ། ས་གུདག། བོ། ཝ་ཁན་གྱི་་་་་་་་་་་་་་་་་་་་་འོ་ཏུམཁརྒྱ་མཎན་ཏག་པ་བ་ཕྲན་བུ་འཚོ་ཌ་བོདྲིའ་འཁ་བའིཏོ། །ཏིང་ངེ་། ཟླཊ་ཕྱི་འཁ་ཎ་་་་་་ནང་བཞུགས་པའི་ཁང་པ་ལ་ཊ་ལག་ཁབ་ཁག་ནང་། ཟ་ཁང་ཀ་ཁ་ག་ང་། ཊ་ཁཎྜིཏྟཾ།
