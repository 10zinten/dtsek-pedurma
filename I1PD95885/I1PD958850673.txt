རྒྱུད། མུའུཞེས་སོ། །།ལྟེ་བའི་སྟེང་གི་ཕྱོགས་སུ་ནི། །དམར་ཞིང་གསལ་བའི་ཆོས་འབྱུང་ནི། །རྡོ་རྗེ་སྔོན་པོ་རྩེ་ལྔ་པ། །ལྟེ་བ་ལ་ནི་འོད་ཟེར་འབར། ས་། ཧཱུྃ་ཡིག་དཀར་པོ་ལས་སྐྱེས་པའི། །མགོ་བོའི་མཚམས་ནས་ངེས་འབྱུང་བའི། །ཧཱུྃ་ཡིག་དུད་ཁ་ལས་་་བྱུང་བའི། །སྒྲ་ཡིས་ནམ་མཁའ་ཁྱབ་པར་བལྟ། ་། །སྒྲ་ཡི་འོད་ཀྱི་རེ་ཁཱའོ། ་་། ཞེས་པ་ནི་ཤེས་རབ་འཕེལ་བའི་ཆོ་ག་རྫོགས་སོ། །།སྟོན་། ཟླ་བ་ལྟ་བུ་འོད། །འཛིན་ཞི་ཚེ་གྲང་ཀྱང་ཡང་འང་མང་། ད་ཀྱི་སྒྲུབ་ཐབས་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཤེས་རབ་འཕེལ་བའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་»བར་དྷཱ་ནི་བི་དྷི། ད། «པེ་»«སྣར་ཨ། ®«པེ་»༼《སྣར་»འཐུར། །ཏེ་དེ་བློ་:ཟླ་ཀྱཏུན@g«པེ་《སྣར་ཁའོ། ་༌པས། e#་་pa། ཁ་ཚོ་ཁ་ཡི་ཁང་དེ་ལས་དང་། པ་ལ་ཌཱ་ཀྱ་རྒྱུ་ཆ་ནས་ད་ལྟ། །p/J་ལ་ཏུཎྜ་ལ་ཀཚོ་། ང་བྱ། ཀློ་་་་ལྟ་བ་། །།་བསྡུར་འབྲས་རེའུ་མིག་ཤོ་པ་པས། །p:/•ཀྱང་བྱ་བབཟླ་་པལ་ཡང་། རྒྱུད་ནི། བོད་ལ་དབང་ན་་དཔར་དབང་། དཔེ་གཞི། སྡེ་དགེ། ད་པེ་ཅིན། །སྣར་ཐང་། •དེཅོ་ནེ། #་། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། °S༼གྱི་ངག་ལྡེབ། ནནབ་—ན་བ་ན། ནན་་ཕྲེང་། ཚེག་ཁྱིམ། །སྡེ་ཚན། རྒྱུད། الدار/།
