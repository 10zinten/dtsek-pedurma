བསྟན་འགྱུར། །རྒྱུད། མུ་ཚེ། །བློ་རིཊ་ཀྱི་དོ། ཝང་ད་ཀ་ཀོ་དར་གྱི་ཚོང་ར་བདེ་བ་དང་། བཏབྱཁ་ཏོནོསྤྲོད། sgཔའོ་། སྤྱི་ཏེང་། །ཏཆ་ཚ་ཁཚོན་པ། པ། །ན་ཏུལ་ཚུ་བསྡུར་མཆན། ཉ་ཙ་ཚུལ་ཁྲི་བར་ཀྱི་བྱ་པ་པ་སྤྱི་ལ་ཕྱོགས་བྱས་ཀྱི་ནགrapཁོ་ཚོ་ལ་སྐུ་ཚེ་བྱ་ནག་གི་ཆ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཨ་ར་པ་ཙ་ནའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་ལོ། འ«པེ་»ཡིན། «སྣར་»ཡིས། «པེ་མཚམས། ་《སྣར་ཚད། རི་བོ་བྱ་བས་གདོལ་བླ་ན་ཏེ། རྒྱུ་གཏོང«པེ་»«སྣར་གིས། «པེ་»«སྣར་དེ། «པེ་»«སྣར་པར། «པེ་《སྣར་»། «པེ་»«སྣར་པ། པ«སྣར་ཞུགས། ་«པེ་»«སྣར་བཞུགས། «པེ་»«སྣར་»—ཟུར་གྱིས། «པེ་《སྣར་སྒྲ། ༼པ་པའི༌། །སྤྲང་། སྤྱི་ལ(«པེ་»+སུ། འ«པེ་»«སྣར་»འགྲོ། («པེ་»«སྣར་རྣམས། «པེ་»«སྣར་བསྒོམས། །«པེ་»«སྣར་ནས། «པེ་»«སྣར་ཕན་ཕྱིར། «པེ་»«སྣར་གང་རྣམས། «པེ་»«སྣར་»ནས། དེབ་ཀspཀgཏོང། ང་ཚོ། ་དེབ་ག་ཚོལྷོ་བའི་དཎཾབ་བྱ་བ་ན་སྐྱོ་ཁར་ངེས་ཡིན། གང་ཡིན་ཀྱང་། །ད་ཚེ། །ཁྲཚོ། ཆ་ཁ་ཚོན་ཁྲོ། ༽པ་ལ་གནས་པ། །ཏུ་བློ་ཟསར་འཐས་རེ་མིག་ཚུམ་ཞེནགྱི་ལོ་ཁ་བ་དུ་བག་ཆགས་དག་རབ་རྒྱུད། (འབཟཏ་ཀའི་དང་ག་ང་པེ་ཅིན། །ས་་དཔེ་གཞི། །སྡེ་དགེ། །།སྣར་ཐང་། །ཅོ་ནེ། །སྡེ་ཚན། །་རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། །ར། བོད་རྒྱུད། དུཔོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། ན_ན། ན་_ན། ན་_བ། ནན་ཕྲེང་། །་，། །ཚིག་རྐང་། །།།ཚེག་ཁྱིམ། །།་ད་ལ་འད། །ཡི་དམ་ལྷ་སྐུ་མི་འགྱུ་བད་ཉིད་བལྟས་ནས། །།
