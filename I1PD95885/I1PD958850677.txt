རྒྱུད། 《སུ?སྐྱོར་ཁལ་ས་ཁང་ཁ་བྱའོ། །དེ་ནས་ལྐུགས་པ་ཡིན་ཡང་ངག་གི་དབང་པོར་འགྱུར་ཞིང་བློ་དང་ལྡན་པར་འགྱུར་རོ། ་། གཞུང་སྟོང་ཕྲག་རེ་འཛིན་ནུས་སོ། /། རྒྱུན་ཆད་མེད་པར་ལེགས་པར་སྦྱར་བའི་སྐད་དང་ལྡན་ཞིང་སྙན་ངག་མཁན་ཆེན་པོར་གྱུར་ནས་་་་བསྟན་བཅོས་ཐམས་ཅད་ཀྱི་མཐར་ཕྱིན་པར་འགྱུར་རོ། དེའི་འོ། གྲུབ་པར་གྱུར་ནས་རལ་གྲིའི་རིག་པ་འཛིན་པའི་བདག་པོར་འགྱུར་ཞིང་རིག་འཛིན་མ་རྣམས་དང་། ལྷན་ཅིག་ཏུ་རྩེ་བར་བྱེད་དོ། ་། འཁོར་ལོ་མཆོག་འདིར་མི་གསལ་བ་རྣམས་སློབ་དཔོན་གྱི་མན་ངག་ལས་ཁོང་དུ་ཆུད་པར་བྱའོ། །ཤེས་རབ་འཁོར་ལོའི་མཆོག་་་་གྱུར་འདི། །བྲིས་པའི་འབྲས་བུ་གང་ཐོབ་པ། །དེ་ཡིས་འཇིག་རྟེན་ཐམས་ཅད་འདིར། །འཇམ་དཔལ་གོ་འཕང་ཐོབ་པར་ཤོག། མཚན་ཡང་དག་པར་་་་་བརྗོད་པའི་མན་ངག་ཤེས་རབ་འཁོར་ལོའི་དངོས་གྲུབ་མཆོག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ༑། །ལྟ་བའི་གནད་ལ་བྱེད་་་་་་པ་ལ་དབཞིན་གི་་་་ཐང་གྲོ་དང་དང་། བག་ཁ་ཁ་ཕྱོགས་དེ་འདྲ་ཞིང་། ས་ག་ཚོད་ཡོད་པ་དང་། ཁ་ཁ་ཁཱ་ལ་ཡ་ཞུའི:ཕྱི། ན་ཚ་དེའོ། དེ་ནས་བྲོ་ཞིག་ཏུ་ང་ཁྲི་ལའོ་ཀ་ཀ་ལ་མ་ཁ་ཀོ་བྱ་དགའ་དེ་ས་བོད་དང་བལ་ཡུལ་སོ་ར་བ་ཁ་བ་ཚེ་ཧོ་ཊཱིཀེམལ་དག་དེ་ཤོག་བྱ་ནད་པ་པོ། འདུ་བྱཡང་གྷ་རི་དང་། ན་་དིཀ་ཆེ་དང་ཀ་པ་བདེ་བ་དང་ཆུ་ཚ་ཁ་ན་ཊ་ལ་དགའ་བ་དང་བག་ཏུ་འབབ་པ་འདི་ནབ་དང་བྱ་བ་ཁྲོ་བ་ན་བ་ལས་བ་ནི་བལ་གྱི་སྐུལ་བ་དོ་བདག་ཅག་སྟོན་པར་ཁས་པ་དང་ལྔ། །ཁྱད་པརྟེན་པ་ཚོས་རྒྱུ་ན་ང་ཚོ་སྐྱོ་ཞེས་ཕྱིའོ། །ད་ར་ཁ་བ་བབས་པའི་ནད་དེ་དཔལ་བ་ལྔ་པའི་མགོ་ཁུག་ལ་ཆུ་སྐོལ་པོ་གང་དག་ཚུལ་བྷེལ་ཞི་བད་མི་ཆེ་ན་ཁོ་བ་འགམ་པ་སྣ་ཚོགས་ལ་ཀ་བ་ཟླn.།
