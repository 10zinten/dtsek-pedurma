ཨ་ཀཱ་ལ་མྲ་སྐྱུ་:བརྟ་ཡ་བརྟ་ཡ་སྭཱ་ཧཱ། འཕགས་པ་ཁྲོ་བོ་སྟོབས་པོ་ཆེའི་སྒྲུབ་་་ཐབས། དགེ་སློང་ཤཱཀྱ་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །།།ཨཱ་ཙརྱ་ནནྡས་རང་འགྱུར་དུ་མཛད་པའོ། །སྒྲ་བག་ཙམ་བཅོས་སོ། །།།#'བསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་+བཛྲ། «སྣར་»+པ། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»—གྱིས། «པེ་»«སྣར་དྷ་ན་ས། «པེ་»«སྣར་+ལ། ་ ོ«པེ་»«སྣར་བསྲུང་། «པེ་»«སྣར་ཀར། ཚེ«པེ་»«སྣར་»—། ཀྱེ«པེ་»«སྣར་»ཤུདྡྷེ་ཤུདྡྷེ། ལེ«པེ་»«སྣར་ཟློག«པེ་»«སྣར་»ཀ«པེ་»«སྣར་བ་བརྟ་ཡརྟ། «པེ་»«སྣར་ནནྟས། ਮਰਸ਼ੋਰਸਥਾਨਨੂੰਸਰਬਸਿੰਘལ་ཁྱད་ཀྱང་་ལང་ཁRབསྡུར་འབྲས་རེའུ་མིག༌ཁང་ནང་ཀ་བཏདང་། །ཀ་བ་བདུ། ཆུ་རྒྱུད། （བློ་ས་བ་བརྒྱུད་དཔེ་གཞི། །་སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། བརྒྱུད། །རྒྱུད། །རབ་པོད་རྟགས། མུ་ཚན་གྲངས། ལྡེབ། བ_ན། ན?ན། བན་བན་ཕྲེང་། །།ོ་ལ། ལོ་ལ་ཚེག་ཁྱིམ། །ten:ཆུ་ਸਿੰpeosਣਝੋਨਾਲਿ॥ਹਾਰਾਲਉਸਮਾਨਸ਼ਨམི་བན། །།
