རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས། བཅིང་བར་བྱ་བ་ནི་རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་ཕྱག་རྒྱའོ། །ཨོཾ་བཛྲ་:ཤྲཱི་ཁ་ལེ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ་ཞེས་པ་ནི་བཟླས་པའོ། །རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ། །་དེ་རྗེ། །པ། ས་སྒྲོག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས་རིམས་གྲངས་བཞུགས་སོ། །ཚུ«པེ་»ཛཾ་ཁ་ལ། «སྣར་»འི་ཁ་ལ། ཚེ«པེ་»«སྣར་»ཤྲཱི་ཁཱ། བ་ལྟ། དེཕྱིར་བྱོན། བསབསྡུར་འབྲས་རེའུ་མིགརྒྱུད། （ལ་བཏང་། པ་དང་། ལ་འོ། ཎཾ་ཏུ་། ་ཅོ་ནེ། /ཀྱིས་དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང༌། །སྡེ་ཚན། #*རྒྱུད། རྒྱུད་འགྲེལ། ཀརྒྱུད། ཀརྒྱུད། ཤ་བོད་རྟགས། །ཚན་གྲངས། ་་་ལྡེབ། ན་བ་ན?ན་ན་ན་ན་བ་ཕྲེང་། :*ཚེག་ཁྱིམ། ཨཱ་༼་n::ནཔ:པpཀའི་ཁ་%། ཆཏྟ་བུ་ཁྱི་S%zཁ་མཁ་བ་ལྔ་ལ་ཁྲོ་ཀngਸਰਣਸਮਾਨਬਸਉਨਾਲੋਂਸਿੰਗਲཁཊོ་ཥnཀ་ཏིག་ཏི། ༼ཀ་ཁprཀ་ལ་ཀའོ། །ངོ་ཏཎཎྜ་ཙཞུ་རྒྱུ་ཎཱ་ཌ། ཀ་ཅ་ཏུལ་ཀ་ཌོ་ཎཎོ། །npཏུ་?pཔོn།
