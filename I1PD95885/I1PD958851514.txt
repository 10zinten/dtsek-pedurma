གྱིས་མི་དམིགས་པར་བྱས་ལ་ཡོད་པ་དང་མེད་པའི་རྟོག་པ་སྤངས་ལ། སྟོང་པ་་་ཉིད་ཀྱི་བདེ་བ་ལ་རྒྱས་གདབ་པོ། །རྟོག་པ་ལ་ཤེས་པ་འཕྲོ་ན་སྔ་མ་ལྟར་བདེ་བ་བསྒོམ། ཡང་། བདེ་བ་ཤས་ཆེ་ན་ཕྱི་མ་ལྟར་སྟོང་པ་བསྒོམ། རྟོག་པ་གཞན་འབྲེལ་པ་:སྤང་སྟེ་ཟླ་བ་དྲུག་ཏུ་བསྒོམས་ན་མཆོག་ཐོབ་པར་འགྱུར་རོ། །སྒྲོལ་མ་མཆོག་སྒྲུབ་པ་ཞེས་བྱ་བ་དངོས་གྲུབ་བརྙེས་པའི་སློབ་དཔོན་བསོད་སྙོམས་པས་མཛད་པ་རྫོགས་སོ། །་བྱེ་། བལ། ལྟ་ས་དྲ་དེ་དོན་ལ་ཞེ་བདེ་ཕྲའོ། ས་ཀྱིས་བཅོམ་ལྡན་འདས་སྤྱན་དྲངས་མདུན་ན་གནས་པ་ལ་ཁོ་ན་ར་མཐོང་ན་མེད། ད་ལྟ་བ་ལ་བསྡུར་མཆན། བཱནུབཏེ་ཏཱ་རེ་ས་མེད་དེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཉི«པེ་»«སྣར་ར། «པེ་»«སྣར་ན། «པེ་»«སྣར་»ནི། «པེ་»«སྣར་»+པ། «པེ་སྲས། «སྣར་སྲང་། @«སྣར་»སྦྱར། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»སྤྲོས། ཁྱེ«པེ་»«སྣར་ལ། །ད་«པེ་»«སྣར་»སྤངས་ཏེ། མ་བསད་པ་དང་ས་གནས་དཔའ་མོ་::ཀྱཎཁ་ཤསྤྱན་ཏུ་ཁོ་བོ་ལ། །་pཏིང་། རྗེ་འཁ་ཎ་ཊབསྡུར་འབྲས་རེའུ་མིག་ཏུ་བཀལ་བ། འཊ་ཀ་ཊ། ཚོང་ཀ་ཁ་སཏོང༌་་་་རྒྱུད། （ཆ༽་ན་e/pན་ཅོ་ནེ། པེ་ཅིན། རྒྱུད་འགྲེལ། སྣར་ཐང་། རྒྱུད། ས་རྒྱུད། རྒྱུད་དུང་དང་། དཔེ་གཞི། སྡེ་དགེ། །སྡེ་ཚན། _རྒྱུད། །་་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནན་ཕྲེང་། །ཚེག་ཁྱིམ། ན་—རབསནབ། །ན?ན། །
