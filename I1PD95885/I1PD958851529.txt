བསྟན་འགྱུར། རྒྱུད། »ཨུ༽ཆུ་དེ་མར་བླུགས་པས་སྒྲིབ་པ་བྱང་། ཡར་འཁྱིལ་བས་སྤྱི་བོར་དོན་ཡོད་གྲུབ་པ་་བཞུགས་པར་བསམ། དེ་ནས་མི་རྟོག་པ་ཅུང་ཟད་བསྒོམ། ་ཤེས་པ་འཕྲོ་་་་བའམ་བྱིང་བ་ན་ལྷའི་:སྐུ་རིམ་གྱིས་བསམ། དེ་ནས་སྙིང་པདྨ་འདབ་མ་བརྒྱད་ལ། དབུས་ཀྱི་ཟླ་བ་ལ་ཨོ། ཤར་གྱི་འདབ་མ་ལ་ཏཱ། ་ལྷོའི་འདབ་མ་ལ་རེ། ནུབ་ཀྱི་འདབ་མ་ལ་ཏུཏ། བྱང་གི་འདབ་མ་ལ་ཏཱ། བྱང་ཤར་དུ་རེ། ཌ་ཤར་་ལྷོར་ཏུ། །ལྷོ་ནུབ་ཏུ་རེ། ནུབ་བྱང་དུ་སཱ་ཧཱ་བཀོད་ལ་རིམ་པས་བཟླས་སོ། །ལག་པ་ཨུཏྤ་ལའི་ཕྲེང་བ་བགྲང་ངོ༌། །བརྒྱའམ་སྟོང་བཟླས་ལ་འཕྲོ་གཅོད་ཙམ་ན། ཡེ་ཤེས་པ་ནམ་མཁའ་ལ་བཏོན་ལ། །བདག་ཐལ་མོ་སྦྱར་ལ། བདག་གི་དགེ་བའི་རྩ་བ་འདིས་བདག་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་ནས། སེམས་ཅན་་་ཐམས་ཅད་ཀྱི་དོན་བྱེད་པར་གྱུར་ཅིག་གོ། བདག་སངས་རྒྱས་མ་ཐོབ་བར་ལ་་་ཐེག་པ་ཆེན་པོའི་ཆོས་སྤྱོད་ཅིང་། །སེམས་ཅན་གྱི་དོན་བྱེད་པར་གྱུར་ཅིག་ཅེས་བརྗོད་ལ། གཤེགས་སུ་གསོལ་ལོ། །སྒྲོལ་མའི་སྒྲུབ་ཐབས་དྷ་བྱང་ཆུབ་སེམས་དཔའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།་་་་་་་་ཁྲ་གླིང་དུ་:པཎྜི་ཏ་ཏ་ཐཱ་ག་ཏ་རཀྵི་ཏ་དང་། བོད་ཀྱི་:ལོ་ཙཱ་བ་རིན་ཆེན་གྲགས་པས་བསྒྱུར་བའོ། །།།བྱོན་པ་ལ། ལོ་ལ་ལོ་བའི་ལཙོ་ཀ་ཁག་ཏུཔ། ་ངལྔ་ནི། །རྒྱ་ནག་པ་:་་་་་་་་དང་་། ཊ་ཁྲིད་གལ་ཏོའདི། དེ། ཁང་པ་དེ་ལ་ཉོན་ཡ*Sཚོ་ལ་ཁ་ཚོ་ཚདེད་ཚོད་ཀྱིས་/། །་ཁཊའི་བདུན་གྱི་བཀའ་སྟེ། ཚོགས་ཀྱི་ཡི་ཏ་ཀ་བ་ཀ་དག་ཆ་ལ། །ཡ/zན་པ་པོཙཁབུལ། ཎ་་་་་་་་་་་་་་་་་་་་་ཁེལ་བ་ལ་དཔྱད་ཚོགས་པའི་—།
