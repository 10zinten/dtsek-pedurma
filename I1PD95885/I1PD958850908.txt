རྟོག་པ་ལས་གསུངས་པའི་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་ཐབས། ཡ་སཏྭ་ཨཱའདིས་དུས་ཐམས་ཅད་དུ་གཤེགས་སུ་གསོལ་ཞིང་དགེ་བ་ཡོངས་སུ་བསྔོ་ཞིང་མངོན་པར་འདོད་པའི་དངོས་གྲུབ་ཞུས་ཏེ་རང་གི་ཕྱག་རྒྱ་སྤྱི་བོར་བཀྲོལ་ནས་ཨོཾ་མཱ་རཱི་ཙྪི་མུཞེས་སོ། །རང་གི་ལྷག་པའི་ལྷའི་རྣལ་འབྱོར་གྱིས་ཟས་དང་ཁྲུས་དང་། ཉལ་བ་ལ་སོགས་པ་ཐམས་ཅད་དུ་རབ་ཏུ་བརྟག་པར་བྱའོ། །དེའི་ཕྱིར་རིམ་པ་འདིས་ལུས་ངག་ཡིད་གསུམ་བསྲུང་བར་བྱས་ནས་ཅི་བདེར་་གནས་པར་བྱའོ། །འཕགས་མའི་སྒྲུབ་ཐབས་བྱས་པ་ཡིས། །དགེ་བ་ཉེར་བསགས་གང་དེ་ཡིས། །ཡེ་ཤེས་དྲི་མེད་འོད་ཟེར་ཅན། །སྐྱེ་བོས་གོ་འཕང་ཐོབ་པར་ཤོག། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་གྲིཉྩ་གྲིཉྩ་ཏུ་སརྦ་:བྷཱུ་ཏ་ཀཱི་བ་ལི་མ་མ་ཤཱནྟི་ཀུ་རུ་བནྡྷ་སྭཱ་ཧཱ། ཞེས་པའི་སྔགས་ཀྱིས་གཏོར་མ་འབུལ་ལོ། །རྟོག་པ་ལས་གསུངས་པའི་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།པཎྜི་ཏ་སློབ་དཔོན་སྙིང་པོའི་ཞབས་ཀྱིས་མཛད་པའོ། །ད་ན། འོད་ཤེད་ཅན་གྱི་སྒྲུབ་ཐབས། Rཁོང་ཚRཙ་བ་ན་ལ་ཕྱག་འཚལ་ལོ། །དང་པོ་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་《+Pཆུ་བནདུམ་པའི་ས་བོན་ཕྱེད་དང་བདག་པ། ཕྱི་ལོ་ཡབསྡུར་མཆན། བཀྱིལ། བཀག་པ་ཁོག་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་འདུས། ཚེ«པེ་»«སྣར་པའི། «པེ་»«སྣར་པ། ༼Pཏིཥ། «པེ་སརྦ་བིགྷ། 《སྣར་སརྦ་བིགྷ། ཉི«སྣར་»འཚལ། ཉེ«པེ་»«སྣར་»ཏིཙྪ། «པེ་»«སྣར་དྷ། ཨེ«པེ་《སྣར་»ཏིཙྪ། @«པེ་»སྣར་བཟུང་། «པེ་»«སྣར་»སྒྲུབ། «སྣར་»བསྐྱོད། @«སྣར་»ཤུ། ཁྱེ«པེ་»«སྣར་»ཞུ། ཀྱེ«པེ་»«སྣར་»+སྨོན་ལམ་སྔོན་དུ་འགྲོ་བས་བདག་་མེད་པ། ་ཉེ«པེ་»«སྣར་+ཀྱང་། །«པེ་»«སྣར་»—ཉ། «པེ་»«སྣར་—པ། @«པེ་»«སྣར་དང་། ︽«པེ་»«སྣར་»ལཾ། །གྱི«པེ་»«སྣར་»—པ། །«སྣར་བུར། ཚེ«པེ་»«སྣར་ཐུང་། «སྣར་བརྒྱན། «པེ་»«སྣར་གའི། ཚེ«པེ་»«སྣར་བན་དྷུ། །འགྱེ«པེ་《སྣར་+དབུ། ད་པའི་བདེ་—།
