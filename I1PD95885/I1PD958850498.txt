ཧྥཏ་བྱང་ཆུབ་ལྟ་བུ། ཀཋ་མ་ཁོ་ཚོས་ང་ལ་ཁོག་ཏོ་་སྐupཝན་བློ། ཀབ་ཕྲེང་བཞི་པ་ཡོ་བྱད་ཙམ་བའི་སེ་ར་བ་ཚོགpད་འཁོབྱ། །དེ་ཏཎསཱ་ཏ་ཆེ་ཞུ་དང་དེ་རྗེས་ཚོ་ཁམི་བཛྲ་ད°ཞེ་ལོ་རྗེ་ཆོས་ཀྱི་རྒྱུའི་ཚོད་དཔggjད། ང་ཚོ། ་སོ། །ཀླུ་དགས་པ་ཆོཀ་ཏའི་ཆུ་ལོག་ཎཋ། །p་sོད་ཁའི་ཏའོ་ཆི་དང་ལ་ཏ། །ཊ་ཚོ་ལྟ་ཁབློ་སྤྲང་། ཏུ་ཏན་། ༄༅། །རྡོ་རྗེ་ཆོས་ཀྱི་སྒྲུབ་ཐབས། ཀ་བ་ཀ་ཀ་ཁfਲੂਨੂੰਛੁਟਵਸੰਮਲਿਲਫ਼ਨਤਮਸਮgਸਮਾਨਸਨzན་nsཁྱུ་ཀུན་༼ང་ལ་བློན་པBའི་ངག་ནས། །ཁྱའི་ཁྲི་ཀའི་ཁྱུ་»་ཟླ་ཚོ། །བྱ་ཀ་ང་ཀ་}ཆཥ་z་:ག་ལུང་འ་ར་ལ་འདིའི་འོད་ལ་ད་ཀ༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་དྷརྨ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་ཆོས་ཀྱི་སྒྲུབ་ཐབས། རྡོ་རྗེ་ཆོས་ལ་ཕྱག་འཚལ་ལོ། །བྷིཡིག་ཡེ་ཤེས་རྫོགས་པ་ལས། m། ཧཱ་ལ་ཧཱ་ལའི་སྐུ་ཆེན་པོ། །ཞལ་གསུམ་སྤྱན་ཡང་གསུམ་པ་སྟེ། །རལ་པའི་ཅོད་པན་གྱིས་བརྒྱན་ཅིང་། །ཟླ་བ་ཕྱེད་པ་འཛིན་པའི་ལྷ། །དབུ་ལ་ཐོད་པའི་ཕྲེང་བས་བརྒྱན། །རལ་པའི་ནང་ནས་རྒྱལ་བའི་སྐུ། །རྒྱན་རྣམས་ཀུན་གྱིས་བརྒྱན་གྱུར་པ། n། ཆུ་སྐྱེས་དཀར་པོ་དང་མཚུངས་ཤིང༌། །།སྒེག་པ་་ལ་སོགས་ཉམས་དང་ལྡན། །ཕྱག་དྲུག་ཞལ་ནི་འཛུམ་པ་དང༌། ཀ། སྟག་གི་་པགས་པའི་ན་བཟར་བཅས། །ཕྱག་གཡས་ན་ནི་མཆོག་སྦྱིན་དང་། ་་། གཉིས་པ་ན་ནི་བགྲང་ཕྲེང་དང༌། །གསུམ་པ་མདའ་དང་གར་མཛད་དོ། །གཡོན་པ་ན་ནི་པདྨས་བརྒྱན། »། གཉིས་པ་དེ་བཞིན་ནུ་མ་འཛིན། །གསུམ་པ་ལ་ནི་དེ་བཞིན་གཞུ། ་ཁ། སྦྲུལ་གྱིས་ཡོངས་སུ་དཀྲིས་པ་ཡི༄༅། །རྩེ་གསུམ་་་།
