བསྟན་འགྱུར། །རྒྱུད། ་་མ་ལུས་ཐུགས་དང་གཅིག་འགྱུར་འཇམ་དབྱངས་འདྲ་བར་ཤོག་ང། །བདག་དང་རྣམ་རྟོག་མང་པོའི་ཚོགས་བྲལ་བདེ་མཆོག་ལྡན་པ་ཡི། །ཕྱག་རྒྱ་ཆེན་པོ་དངོས་གྲུབ་ཚོགས་འདིས་ཚེ་འདིར་མྱུར་ཐོབ་ཤོག། བཅོམ་ལྡན་འདས་འཇམ་པའི་་་དབྱངས་ཀྱི་སྒྲུབ་ཐབས་བརྟན་པའི་འཁོར་ལོ་སྒོམ་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་བཱིརྱ་བྷ་དྲ་དང་:ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་བའོ། །།།ཚེ་ཀལ་ཟླ་བ་བརྒྱད་ཁ་ཤོ་དང་ཞིང་བསྒོང་བ་དང་བདག་ལ་དུ་བོད་པ་ཚོ། ང་ཚོ་གན་ཀྱི་རྒྱུ་ན་ཁ་ཊཱ་ཀ་ཚོ་ཁགྱིད་དུ་ཨ་ཁཚོ་མདང༌རྟ་བོ་ཀ་ལོ། །དཞུ་::ཀྱིངབན་ལོ་འགའ་ན་ཌའི་ཊོཀ་བྲེ་ཆ་ཁ་བསྡུར་མཆན། ལ་དག་གནད་ཊི་ལ་ཁ་བ། སྐྱེ་ར་ཎཁྲུང་དུ་ངོ་བ་་་་་་་་་་ཁཚོ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»བརྟན་པའི་འཁོར་ལོའི་ཆོ་ག་བཞུགས་སོ། །«སྣར་བརྟན་པའི་བཞུགས། དེ«པེ་»«སྣར་དཔལ། «པེ་»«སྣར་»ཀྱང། «པེ་»«སྣར་+པ། ཚུ«པེ་»«སྣར་རང་། «པེ་»«སྣར་»གྱུར་པ། ཚེ«པེ་»«སྣར་བསྒྲུབ། «པེ་《སྣར་བཟུང་། /«པེ་»«སྣར་»—སྡུག་བསྔལ་གྱི། ལ«པེ་»གདོན། «སྣར་»བདོན། ཚོ«པེ་འཆད་པ་ལ། «སྣར་»ཆད་པ་ལ། «པེ་»«སྣར་_བ། པོ«པེ་»«སྣར་»—ཐུང་ངུ་། (ི«པེ་བཱི་ཡེ། «སྣར་ཐཱ་ཡེ། QQ«པེ་»«སྣར་»བི། ཚ«པེ་»«སྣར་»། «པེ་«སྣར་»གི། «པེ་»«སྣར་པའི། མེད། ན་གྱི་@«པེ་»«སྣར་»—། །«པེ་»«སྣར་»བི། ཆོ«པེ་»«སྣར་བྱ་བའོ། ༼ཚེ«པེ་»«སྣར་»+ལ། པོ«སྣར་»འཕྲུལ། ཚེ«པེ་»«སྣར་»ཡང་། «སྣར་»—དག་པར། «པེ་»«སྣར་པ། ཀྱེ«པེ་»«སྣར་»ཉ། ཀྱེ«པེ་»«སྣར་བསྒོམས། «པེ་»«སྣར་རྟགསྟེ«པེ་»«སྣར་གྱིས། «པེ་»«སྣར་འབྱུང༌། །ཨུ«པེ་»«སྣར་»སར། ཚེ«པེ་»«སྣར་»གཉིས་ལ། བློ་«པེ་»«སྣར་»བར། ཉི«པེ་»«སྣར་»། ︽«པེ་»«སྣར་དམར་པོ། ཁྱི«ཅོ་»ཀ་བཞུགས། «པེ་»སྤུ། «པེ་»«སྣར་པའི། ཉི«པེ་»«སྣར་བ། «པེ་»«སྣར་»+པོ། ཁྱེ«པེ་»«སྣར་»ཁྲི་ཤིང༌། །ཀྱེ«པེ་»«སྣར་»འོང་བ། «ཅོ་»འོངས་པའི། ལོ་ཚོགས་ལ་«པེ་»«སྣར་སྒོམ། _@«ཅོ་གིས། «སྣར་འཁྱུད། «པེ་»«སྣར་—པར་གྲུབ། མུ«པེ་»«སྣར་པོ་རྣམས། «སྣར་ཀྱིས། @«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»ཏུ། «པེ་»«སྣར་སྐུ། ཁྱེ«པེ་»«སྣར་ཞུགས། འཚེ་ཏེ། །ཡིད་ཀྱི་རི་ཆུ་ཚོདལnལ་།
