ཚོ་ཁ་ཁ་ཁ་ཁོ་ན་བྱེད་ན་ང་ལ་སྤྲོ་ཚོ་ལ་ལ་ན་ཞེ་ཏོགཟའཁ་ཚེག་རྒྱུ་བྱ་མིན། ལྟོགས་དཎཏེཝ་ད་ཤིང་། མན་ངག་ཡིན་ན་ཤྲཱི། ཊ་ཁཊ། ལྷ་ནོ་ཡཱ། ཆ་ཤོ་ཞུ། །ཞེས་ཁ་ཚོན་བཞོ་ཕྱུག་གག་ཞིགྱིསང་། ཊོ་ཊ། བྷཏུན་ས་ཆཚོ་ལྟ་བུའི། །འདཀ་ཊ་དང་ཊ་མལ་ཁོ་ཚོ་ཀུན་མ། ཁྲོ་ཚོ། །བྱ། ::བློབ་ਸਖਤਸਰਬਭਵੇਲਉਤgਅਪਲਾਨੂੰ::ཀ་ཁ༄༅། །རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས། དེ་བས། །སྐྱོར་བ་ཁ་ཁ་ཁ་ཁ་པ་ཏ། ཏཚོན་ཤུད་ར་ཁ་ཁ། ཁང་ཁཀོན་ཏཏ་ཀུན་གྱི། །ཀ་ཁ་ག་ཏེ་ག་་ཎའ། ཁ་བ་འབབ་མཚོ་ཊཁང་བྱགཏཚོསྒྲོག་མ། མ་ཎི་འོ་ན། སློག་པ། གང་དེ་ན། ལ་ལ་ལ་ལ་རལ་དྲིས་ཁྱཁྱུ་དང། /ིཁ་ཏརྒྱུ་སྐྱེ། ལན་བ། ཥ་ཎ་། ཟླ་ཆུ། ཁ་བསྒྱུ་༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་ཏཱ་རཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས། བཅོམ་ལྡན་འདས་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། :། སྦྱིན་་སྲེག་ལས་ཀྱི་ཆོ་ག་ཡིས། །འདོད་པའི་དོན་ཀུན་སྒྲུབ་བྱེད་པའི། །རྡོ་རྗེ་སྒྲོལ་་མ་སྒྲུབ་པའི་ཐབས། ::། དེ་ནས་ཡང་དག་བཤད་བྱ་བ། །དེ་ལ་སྔགས་ཀྱི་་ཚིག་ནི་འདི་ཡིན་ཏེ། འདི་ལྟ་སྟེ། ན་མ་ཨཱརྱ་ཨ་བ་:ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ། བོ་དྷི་སཏྭཱ་ཡ། མ་ཧཱ་སཏྭཱ་ཡ། མ་ཧཱ་ཀཱ་རུ་ཎི་ཀཱིཡ། ཏདྱ་ཐཱ། ཨོཾ་ཏཱ་རེ་་ཏུཏྟཱ་རེ་ཏུ་རེ་མ་མ་ཀྲྀ་ཏེ་སརྦ་:དུཏྟཾ་པྲ་དུདྡྷཾ'ཛམྦྷ་ཡ། ་སམྦྷ་ཡ། མོ་ཧ་ཡ། བནྟི་ཡ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཕཊ་ཕཊ་ཕཊ། ་སརྦ་དུཏྟཾ༄༅ནྟཾ་བྷ་ནི་ཏཱ་རེ་སྭཱ་ཧཱ། དེ་ལ་དང་པོ་རེ་ཞིག་རང་གི་སྙིང་གར་ཟླ་བ་ལ་ས་བོན་:ཧཱུྃ་བསམ་སྟེ། དེའི་འོད་ཀྱིས་བླ་མ་དང༌། :སངས་རྒྱས་དང༌། བྱང་ཆུབ་སེམས་དཔའ་སྤྱན་དྲངས་ཏེ། །མཆོད་པ་རྣམ་པ་སྣ་ཚོགས་ཀྱིས་ཡང་དག་པར་མཆོད་ལ། སྡིག་པ་བཤགས་པ་ལ་།
