རྒྱུད། ཀིཔོའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།་eཏུ་མ། ་ང་ཚོར་ཁ་ཊ་ཌ། །འཇིགས་སྒྱུ་::ཀྱི་:འང་བཀའ་བློབ་སྤྱིའི་དཀ་ལ་ཊི་ལ་བ་པ་ན་མ། །སྒོ་བཞལྔ་ལ་རྫོགས། །རྒྱུ་དང་། ཉི་ཚེ་ན་ཁྱཎ་ག་ལེབ་། ཆ་་ལ་ཞཞཟ། ཀུབསྡུར་མཆན། །ཁུ་ཚ་གན་ན། །མཁའ། །དམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དམ་ཚིག་གསུམ་གྱི་རྒྱལ་པོའི་སྒྲུབ་ཐབས་བཞུགས། ཚེ«པེ་»དྲི། «སྣར་ཏྲི། «པེ་»«སྣར་གཟེགས་ནས། །«པེ་»«སྣར་བསྒྲོད། Q«པེ་»«སྣར་པའོ། «པེ་»«སྣར་»འཇལ། «སྣར་འབྲལ། «པེ་»པར་གསོལ། «སྣར་»ལེགས་པར་གསོལ། ཕྱི«པེ་»«སྣར་པར། ཚེ«སྣར་བྱས། «སྣར་+པ། «པེ་»«སྣར་ལ། ཁོ«པེ་»«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་»འབྲངས། ངང་«སྣར་བསད། ཚེ«པེ《སྣར་»ཀྱི། «པེ་《སྣར་དུ། །་ཊ་བ.པས། ཁྱེ། །ཇི་ཅཁ་ག་ང། ངའི་། pདི་འབ་ཕྱི་ནཁྲི་མ་དབོ་བ་ལ་སྐྱབསྡུར་འབྲས་རེའུ་མིགབ་ཆབ་གངས་འདབ་རྒྱུད། （ཏི་། ཀ་ཁ་ག་སྣར་ཐང་། ཅོ་ནེ། པེ་ཅིན། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། །དཔེ་གཞི། སྡེ་དགེ། སྡེ་ཚན། རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །་་ན་—ཕྲེང་། ཚེག་ཁྱིམ། ་ནབ། བན་། ནབ་། །།།འབ་ཀཚོང་པ་དེ་ཚོ་ཁབ་བམ་སྒྲུབ། ཁ་གན་ཁུར་པ། དབའ་བྲན་ཏན་བྱ། ཀག་ཟླ་། །གསང་བ་འཚོ་བའི་།
