་་་ཚོ་་། ང་། ཁpད། །ppད་རྒྱལ་ལ་ཕྱོགས་པ་གད་པ་དང་། བྱ་བར། །༄༅། །འཕགས་མ་ལྷ་མོ་སྒྲོལ་མ་ལ་བསྟོད་པ། ཀ་ནིའe/gཀྱི་རིགས་ལ་༼དབང་སྒྱུར་བ་ན་ཀོ་ཀ་བ་ཀ་རྒྱུན་ལ་ལྟ་ཀའོ་འཆོལ་བའི་ཚ་མོ་ཡིན་པ་བདུན་པཆs༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏཱ་:རཱ་དེ་བཱི་སྟོ་ཏྲ། བོད་སྐད་དུ། །འཕགས་མ་ལྷ་མོ་སྒྲོལ་མ་ལ་བསྟོད་པ། རྗེ་བཙུན་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། ་། ལྷ་མོ་སྒྲོལ་མ་ལྷ་ཡི་ལྷ། །འཇིག་རྟེན་གསུམ་ན་མཆོག་གྱུར་པ། ལ། །ཐུགས་རྗེ་ཆེན་པོའི་དབང་གྱུར་པས། །ཁམས་གསུམ་འགྲོ་བ་སྒྲོལ་ལ་མཁས། །ཚོར་བ་བཏང་སྙོམས་ངོ་བོ་ཉིད། །མི་མཐུན་ཚོར་བ་བདེ་་སྡུག་ལྡན། །།ཕྲག་དོག་འཇོམས་པའི་ཕྲིན་ལས་ཅན། །མཁའ་ལྟར་:དངས་པ་ཅིབླ་ན་མེད། བྱ་། གཟུང་བ་དང་ནི་འཛིན་པ་ཡི། །རྣམ་རྟོག་སྤངས་པའི་བདེ་བ་ཅན༑། པདྨ་ཆུས་མི་ཆགས་པ་བཞིན༄༅། །འཁོར་བའི་སྐྱོན་གྱིས་མ་གོས་པ། །རང་བཞིན་གྱིས་དག་བླ་ན་མེད། །རྒོལ་བ་དག་ནི་ཚར་གཅོད་པ། །འགྲོ་བ་་་་ཡིད་འོང་ཕུན་ཚོགས་པ། ལ་། སྐུ་མདོག་མཚན་དང་དཔེ་བྱད་ལྡན། །སྒྲོལ་མ་སྣ་་ཚོགས་མདོག་ཅན་ནི། །ས། འབྱུང་པོ་ལས་བྱུང་གདོན་ཆེན་དང༌། ་། འདུ་བ་ལས་།
