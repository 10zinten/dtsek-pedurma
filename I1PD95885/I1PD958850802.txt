ཀྱི་ཀྲང་། ཁུལ་ཁོ་ཀ་བ་ཀ་ར་ཀ་ཎ་། བྱཁང་ལ་ཊི་'ਠੋਰਸਸਟਲੋਂਨੌਇਮਾਰਤਮਲਤ། །།ལོ་ཀ་ཀོ་ཊ་ཀ་བ་ཀ་རྒྱལ་ཏོ་ཆེ་འ་། ཀ་ཁ་ག་ང་། ༄༅། །འོད་དཔག་མེད་ཀྱི་སྙིང་པོའི་རྒྱུད་ལས་བཅོམ་། ལྡན་འདས་མ་འཕགས་མ་སྒྲོལ་མའི་རྟོག་པ་གསུངས་པ། དོའིབྱསབ:༄༅། །ན་མོ་རཏྣ་ཏྲ་ཡཱ་ཡ། ན་མ་་ཨཱཙཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ། བོ་དྷི་སཏྭཱ་ཡ། །མ་ཧཱ་སཏྭཱ་ཡ། མ་ཧཱ་ཀཱ་རུ་ཎི་ཀཱ་ཡ། ན་མཏཱ་ར་ཡི། ཏདྱ་ཐཱ། ཨོཾ་ཏཱ་རེ་ཏཱ་རཱ་ཡ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ས་མ་ཡ་སྦི་ཏེ་བྷ་ར་བྷ་ར། སརྦ་བྷ་ར་ཎ་བྷུ་ཏི་ཏེ་པདྨེ་པདྨེ་པདྨ་བྷུ་ཛེ་པ་ད་མཱ་ས་ན་སྦེ་ཧ་ས་ཧ་ས། ཏྲི་ལོཀྱ་བ་རི་དེ་སརྦ་དེ་བ་དཱ་ན་བ་སཱུ་ཛི་ཏེ་སྣ་རཱ་ཧི་བྷ་ག་བ་ཏ་སུ་ར་ཏས་མ་ཡདྷ་ན་དྷ་ན་མ་ཧཱ་སཏྭཱ་བ་ལོ་ཀི་ཏེ་མ་ཎི་ཀ་ན་ཀ་བི་ཙི་ཏཱ་བྷ་ར་ཎེ། ཨོ་བི་ལོ་ཀི་ཡ་བྷ་ག་བ་ཏི་ཏཱ་རེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། རྩ་བའི་སྔགས་སོ། །།སྐུ་གདུང་དང་བཅས་པའི་མཆོད་རྟེན་གྱི་དྲུང་དུ་བརྒྱད་ལྷག་པའི་སྟོང་བཟླས་པར་བྱའོ། །དེ་ནས་རྫོགས་པར་གྱུར་ནས་བཅོམ་ལྡན་འདས་འཕགས་མ་སྒྲོལ་མ་ལ་བལྟས་པས་གང་འདོད་པ་དེའི་མཆོག་ཐོབ་པར་འགྱུར་རོ། ་པོ། ཨོཾ་མ་ཎི་ཏཱ་རེ་ཧཱུྃ་འབུམ་བཟླས་ནས་འཕགས་མའི་དྲུང་དུ་ཉེ་བར་་་་གནས་ན་གང་འདོད་པ་དེ་ཐམས་ཅད་སྟེར་བར་བྱེད་དོ། །དཀྱིལ་འཁོར་མེད་ཀྱང་—།
