དྲས་མཛད་པ་རྗེ་བཙུན་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།བས། །རྒྱ་གར་ཤར་ཕྱོགས་ཛ་ག་ཏ་ལའི་པཎྜི་ཏ་ཆེན་པོ་བི་བྷུ་ཏི་ཙནྡྲས་རང་་འགྱུར་དུ་མཛད་པའོ། །།བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་མ་སྒྲོལ་མ་སྒྲུབ་པའི་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་ར། «པེ་»«སྣར་པོར་རེ་ཞིགས། ༄༅«སྣར་པར། «པེ་»«སྣར་»ཉ། ཚེ«པེ་»«སྣར་+དུ། «པེ་《སྣར་+གདན། ༄༅«པེ་»«སྣར་གྱིས། «པེ་»«སྣར་ནས། ཁོ«པེ་»«སྣར་»—སྒྲོལ་མ། ཚེ«པེ་»«སྣར་བསྒོམ། «པེ་»«སྣར་».བསྲེག«པེ་»«སྣར་བཟད། ཁྱེ«པེ་»«སྣར་+སྐུ། ཀྱེ«པེ་»«སྣར་»—ལོ། །མེ«པེ་»«སྣར་+ལོ། །?«པེ་«སྣར་»ཛ། ཉི«པེ་«སྣར་། 《། ང་འ་ལྟ་བར་ཏུ། །དོ་ཀླད་པ་དུ་ག་ཎ་བད་བ་་་ཁོ་ན། ཁ་བྱང་བ་ལ་ཏུ་ཁ་། བོ་སོགས་གོད་ཁ་ཎ་ཏ༌ར། ཐུན་ཏུ་རང་ལ་ཁ་n:ཚོང་བའི་རང་པ་འདྲ་བ་ས་ཁཟླབ། བསྡུར་འབྲས་རེའུ་མིགPཀz་ཁོ་ཏི། mབས་མས་ཀྱིཟླ་་་་་་་ང་ཡ་ཀྱnལ་ཟླ་ཆུང་ཙལ་རྒྱུ་བ་རྒྱུ་དང་། ད་མ་རའགལap?དཔེ་གཞི། སྡེ་དགེ། \/པེ་ཅིན། སྣར་ཐང༌། ་བྱ། ཅོ་ནེ། ་ཡ། སསྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། ན:ན། ནན་བས：བན་ན། ༽ཕྲེང་། ཚེག་ཁྱིམ། །རྒྱུད་འགྲེལ། །རྒྱུད། གྲོང་པ། ཡི། —།
