སྐུ་བ་ར་རྐོ་ཎeཆུ་བོ་རྒྱུ་ཆའི་བཞི་ཁྲི་ལ་འགྱིག་་་་ཁའི་། ::ཁོང་ཁགིས་སྤྱང་ཀ་ཀ་ཛཱ་ཊཏན་ཏན་ཏིག་ཀོའོག་རྒྱུ་ཆ་གེ་ལྔ་པ་་་་༌་ཀོ་བ་ལ་ཚད་དྲཎཁ་ཚོ་བཅུ་དེ་རི་ཏ་ཀུན་དང་། མདོ་བ་བཀའ་འོད་ཁ་ཟླ་བ་་པོarnaཕྱི་ལོ་པ་ཁ་མི་རིགས་nཏུ་རིལ་པ་་་་་་བདག་འདྲ་ལོ་ཁཡུག་གོ། ཁེ་ཀ༄༅། །།དཔལ་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྒྲུབ་འགྱུར་ཁ་བལ་ཝལ། ཐབས་མདོར་བསྡུས་པ། ལ་ཕྱིར་ཏུ་ཟ་ཁའི་དོན་ཚན་ཚང་འོ། ཟ་ཁ་དར་བས་ལྟ་ན་གཙོ། ད་པ་བྱས་པ་ང་ཚོ་ཁ་ཁྲོམ་ཐོག་མ་ཀུན་:ཏུ་འཁྲུ་བ་དའི་ཁྱོ་ག་ག་ཕྱབ། །ག། ཚེ་ཆ་ལང་བ་ཁོག་པ་ཐ། ཏ་ཏིམལ་ན་སྔགས་ཉq*jpin་ལོན་པ་དང་པོ། དེ་ཁོ་༄༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་:བྷུ་ཏ་ཌ་ོམ་ར་སཾ་ཀིསྟཱ་སཱ་དྷ་ནཾ། བོད་་་སྐད་དུ། དཔལ་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྒྲུབ་ཐབས་མདོར་བསྡུས་པ། དཔལ་འབྱུང་པོ་འདུལ་བྱེད་ལ་ཕྱག་འཚལ་ལོ། །ཚངས་པ་བརྒྱ་བྱིན་ལུས་ངན་སོགས། །མྱོས་རྣམས་འདོམས་པར་མཁས་པ་ཡི། །དཔལ་ལྡན་འབྱུང་འདུལ་ཕྱག་འཚལ་་་ནས། །དེ་ཡི་སྒྲུབ་ཐབས་བྲི་བར་བྱ། །མདོར་བསྡུས་མན་ངག་རྣམ་དག་པས། །མདོར་བསྡུས་འདོད་ཕྱིར་སྦྱར་བར་བྱ། །གནོད་སྦྱིན་གནོད་སྦྱིན་མོ་སོགས་འགྲུབ། །བདག་སྙིང་ཡ་མཚན་སྐྱེས་ནས་བྲིས། །དང་པོར་རེ་ཞིག་སྔགས་པས་ཐུགས་ཀར་ཟླ་བ་ལ་ཁམས་གསུམ་རྣམ་པར་གསལ་བའི་འོད་ཟེར་ཡང་དག་པར་འཁྲིགས་པ། །ལུས་ལ་རཾ་ཡིག་དམར་པོའི་ས་བོན་རྣམ་པར་བསམ་པར་བྱའོ། །དེ་ནས་དག་པའི་ལྕགས་ཀྱུའི་རྣམ་པས་མདུན་དུ་འབྱུང་པོ་འདུལ་བྱེད་གདན་དྲངས་ནས་བླ་ན་མེད་པའི་མཆོད་པ་རྣམ་པ་བདུན་གྱིས་མངོན་པར་མཆོད་ནས་སྟོང་པ་ཉིད་བསྒོམ་པར་།
