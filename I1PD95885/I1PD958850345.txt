དོན་ཡོད་གྲུབ་པས་དབུ་རྒྱན་བྱས་པ། ་ད་སྐད་ཅིག་གིས་་་་་་བདག་ཉིད་ཡོངས་སུ་རྫོགས་པར་བྱས་ལ། དེའི་སྤྱི་བོ་དང༌། མགྲིན་པ་དང་། སྙིང་ག་དང་། ལྟེ་བ་དང་། ཟླ་བ་ལ་གནས་པ་ཇི་ལྟ་བའི་གོ་རིམས་བཞིན་དུ་ཨོ་ཨཱམཱིཧཱུྃ་ཞེས་བྱ་བའི་ཡི་གེ་བཞི་བསྒོམས་ལ་སྤྲོ་བར་བྱའོ། །《། དེ་ནས་སྐྱོ་ན་་་་་སྔགས་བཟླས་པར་བྱའོ། །འདི་ལྟ་སྟེ་ཨོཾ་མ་ཧཱ་མཱ་ཡུ་རཱི་:བིཏྟཱ་རཱ་ཛཱི་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཅེས་བྱའོ། །འཕགས་མ་རྨ་བྱ་ཆེན་མོའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་་་་་བསྒྱུར་བའོ། །།།ཏུབསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་《སྣར་»གྱོང་། @«པེ་»«སྣར་+མོ། །«སྣར་»ལྟར། ཚེ«པེ་»«སྣར་»མཾ། «པེ་»བིདྱ་རི་ཙཱ། སྣར་»བིདྱ་རི་ཛྷ། «པེ་»«སྣར་»ལོཙྪ། དྲི་བཀ་ཊི་་རྒྱུ། གནས་ཆཚོཀ་ཊ་ཀོད་པ་ཁྱི་བྱཁྲོ། །དང་། ལྟ་བསྡུར་འབྲས་རེའུ་མིག་ཁ་ཆ་་༌་་ཏ་་སྐnཚོ་འོ་ཏོད། བྱ་རྒྱུད། （aF་ཏ་ཀ་ཁའི་ང་ཚོ་ཚང་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད། པར། རྒྱུད་འགྲེལ། །རྒྱུད། ་s:ཁ་༼རྒྱུད། དེས/པོད་རྟགས། །ཚན་གྲངས། ལྡེབ། བ_ན་། ན། _བ*བ་—ན?/བ་_ན་ངོ་། །ཕྲེང་། སྣ་ཚེག་ཁྱིམ། །།།།།།
