འདིས་སྙིང་ག་དང༌། བྱང་། དཔྲལ་བ་དང༌། ལྐོག་མ་དང་སྤྱི་བོར་གཏུགས་པར་བྱའོ༑འ། ཨོཾ་ཝཱ་གཱི་ཤྭ་རི་ཧཱུྃ། ཞེ་མགོན་པོ་འཇམ་པའི་དབྱངས་ཀྱི་རྒྱལ་པོ་་་་ཆེན་པོ་སྒེག་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།།%ཥ་ཊྚཎཏp། །བསྡུར་མཆན། ::མཚན་བྱང་འདི་དག་ཀྱང་དུས་ལྟར་བཀོད། དེར་ཞིབ་ལས་དྲུག། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྒྱལ་པོ་་ཆེན་པོ་འཇམ་པའི་དབྱངས་དང་འཕགས་པ་གཤིན་རྗེ་གཤེད་ཀྱི་སྒྲུབ་ཐབས་བཞུགས་སོ། །@«སྣར་ཁ། Q«པེ་»རཱ་ཛ་ལི། «སྣར་ར་ཛ་ཛ་ལཱི། «པེ་རྒྱལ་པོ་ཆེན་པོ། «སྣར་»རྒྱལ་པོ་ཆེན་པོ་རོལ་པའི། «སྣར་+རྒྱལ་པོ་ཆེན་པོ་སྒེག་པའི། པོ«སྣར་བྱས་ཏེ། ཚེསྣར་—པ་རོལ། ཉི«སྣར་»བསམ། «སྣར་»—ལ། དགྱེ«སྣར་»ཀྱིས། འགེ«པེ་»«སྣར་»ཀྱིས། 《«པེ་»«སྣར་གཏུག་%«སྣར་»ཀྱིས། འོ«སྣར་+པའི། །@gཌི་ཎཁ་ཁོ་ཀོ་ཌ་ཁཀྱང་ཁཊ་་ཏ་་ཀྱི་འཁ་བློ་ཁཁ་ཀ་ཁ་}*དཀོ་་དྲིལ་ཀླུ་pཚོགས་ཀྱི་མ་ཎ:ཀ་། དྲོ་་་nལ་དབང་འཁ་དྲོ་ཁ་ཁ་བ་བསྡུར་འབྲས་རེའུ་མིག་ཁ་གསབ་མ་དང་ཁོ་ཁྱོད་ཆགཞན་ཌ་ཎའི་ཟླ་ཁརྒྱུད། ལྔ་ལ་སྐུ་ལྔ་དང་། སྤྲོ་གླུ་ཁང་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། ་་། ཅོ་ནེ། སྡེ་ཚན། _རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། པོད་རྟགས། །ཚན་གྲངས། %ryལྔ་ལྡེབ། /བབ་བ་—ན། །ནབ་བབ་ཕྲེང་། །ཚེག་ཁྱིམ། །བྱུང་། ་།
