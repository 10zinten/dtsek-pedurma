དབང་དུ་འགུགས་བྱེད་འབྱུང་རྣམས་ནི། །ཆུ་སྐྱེས་ཐང་ལས་སྐྱེས་རྣམས་ཀྱི་ཚབ།  ། འདུན་ཆེར་བྱེད་རིག་པ་པ་ས་ལ། ཡིག་འཕྲུལ་ལྟར་བྱེད་དེ་བཞིན་དུ། །ས་དང་རེངས་བྱེད་དེ་བབསྡུར་འབྲས་རེའུ་མིག་སྡང་སྤྱོད་སོ་བ། ས་དང་། །དབང་དུ་འགུལ། ས་བྱེད་གྲུབ། མ་རྒྱུད། （་མེད་ངོ་མཚར་རྣམ་པ་རྣམདབུ། །དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། །སྡེ་ཚན། ས་རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། བོད་རྟགས། །བྱང་དམུ། དུརདུ་སྤྲོ་བ་དང་། མུན་ནག་ཚན་གྲངས། ལྡེབ། བ་_བབ་བ་ན་ན་བབཕྲེང་། །པ་བ་དང་། ཚེག་ཁྱིམ། །བས་་བ་དང་། །།བ་ན་་་་་་་་་་་་ཟླ༌་་་་་་༌་། །རྗེ་དེ། །་པ། །བྱ་བ་དེ། :ཀ་ཊིའོ་བ་ཏེ། བོ། ས་ཁོད། དེ་ཡོད་ན་ང་ལོ་འཁཛོམ་ཆེན་ཊ། །འོ་འཁཅུ་དང་རྒྱུ་ཁ་ལོ་བ་གཉེར་བ་ཁ་ཁ་ཁར་ཀ་བྲག་ཆུ་ན་བ་ལྟ། །ཆབ་འོ་ན་དགའ་བ་དང་། བག་མ་ཀུན་འཏུས་ན་བ་། ཁཊིཚོ་འཚོ་བྱང་ཁའི་ཉི་ཟེཎོ་ཁཆོ་ཀ་བདུན་དང་ཁྱོ་ག་ཞིག་བའི་མི་ཁྱུ་ཚོགས་འདུ་བས་ད་ཁཛིན་ཤོ་ཟླ་བ་ཁ་མ་གྱུར་གན་གཡོག་ཏུ་གྱུར་བ་ཁཚོབཞི་ཡི། །གང་གའཁཝ་ཚོ་ཚུལ་ཁག་དུ་བཞེད་ན་བླ་མ་ལ་ཁ་ཞེ་གཉིས་མེད་དང་ཁ་གནང་། ཁ་བ་འབབ་པ་སྟེ། བ་ན་གོ་བ་ཞེ་ཆེན་ཁྲེན་ਨੇਹਿਸਾਬ'ਚੋਂਵੀਟਮਨਸੇਸਨਸਲਮnaཏ་ས་ཧ་ས། མཁྱད་པར་ཁ་ཚ་ཚ་དེ་ཆོས་བྱེད་ཚོ། ཛ་ལ་ཡུལ་སྐོར་བྱེད་པ་ག་ཚད།
