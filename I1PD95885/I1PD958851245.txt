བསྟན་འགྱུར། རྒྱུད། ཁ་མགུལ་རྒྱན་དང་། རྣ་རྒྱན་དང་། ཅོད་པན་དུ་བརྒྱན་པ། སྒྱུ་མ་ཆེན་པོའི་འཁོར་ལོར་བྱས་པ་བཞི་ལ་ཆུང་མ་དང་བཅས་པའི་ཁྱབ་འཇུག་གཡས་བརྐྱང་བའི་ཞབས་ཀྱིས་མནན་པ་དབང་དུ་:བྱེད་པ་བསྒོམ་པར་བྱའོ། །ཟེའུ་། དེ་ནས་ཕྱག་རྒྱ་་་བཅིང་བར་བྱ་སྟེ། །ལག་མཐིལ་རྡོ་རྗེ་བཅིངས་པའི་སོར་མོ་འབར་བའི་སྙིང་པོའི་དབུས་སུ་མཐེ་བོང་གཉིས་བླང་བར་བྱའོ། །་ད་། དེ་ནས་སྔགས་བཟླས་པར་བྱ་སྟེ། ཨོཾ་བཛྲ་ཛྭ་ལཱ་ན་ལཱརྐ་ེཧཱུྃ་ཞེས་སོ། །རྡོ་རྗེ་མེ་དང་ཉི་མ་འབར་བའི་སྒྲུབ་ཐབས་་་་རྫོགས་སོ། །།།བསྡུར་མཆན། ན་མ་མ་དངམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་མེ་དང་ཉི་མ་འབར་བའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །༽༼«པེ་»«སྣར་ལཱ་ན་ལསྐ། «སྣར་»—པ། «པེ་»«སྣར་བརྒྱན། «པེ་»«སྣར་མོའི། «པེ་»«སྣར་»བྱེད་པ། «པེ་»«སྣར་»ལརྐ། ਸਮੇਂਨੂੰਉਸਨੂੰਉਸਤੋਂਨਾਲਸੰਨਾਟਨਇੰਫੋਲARRepਉਸਤੌਲਸpਲལ་ཁབ་ཀྱི་་ཁ་རྒྱུད། (ལ་ཏཁང་ཀ་བའི་གླ་ར་ཁpབསྟར་འབས་རེ་མིག་ཀྱང་འ་འH（ཨཡ། དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། །རྒྱུད། །པོད་རྟགས། །མུ/Rཚན་གྲངས། །གང་ལ་་་་། ་པ། ལྡེབ། ན?ནན་བ?ན་_བ་ནནཀྱི་བལ་ཚེག་ཁྱིམ། ་༌༌་ས༌ཀ་?བློ་།
