བསྟན་འགྱུར། རྒྱུད། མུਹੋਕਾਖ਼ਲਸਾਵਲੋਂਮਰਨਾਅਸਲ་་་:བོད་ཀྱི་འཚོ་བ་ཀུས་པ། བོད་ཀཔ་ལ་རབ་ཏུ་བསྡུར་མཆན། གང་གི་ཁང་པ་ཁགམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་ཧི། «སྣར་»ཧྲི། «སྣར་པ། «པེ་ལི་དང་། ཀ་ལིའི། «སྣར་»ལའི། @«སྣར་ཟད། ཁོ«སྣར་གྱུར་པའི། །ག་ང་། @«པེ་»«སྣར་ཉ། «སྣར་»ཀྱིས། @«སྣར་གནས། «པེ་»«སྣར་»ཡཾ་རཾ་བཾ་ལཾ། ཚེ«པེ་»«སྣར་»ཅོག@«པེ་»«སྣར་+ལཾ། ༄༅«སྣར་བཛྫི། «སྣར་བནྡ། @«སྣར་»ཏྭཾ། ཁྱེ«པེ་»«སྣར་ལརྐ། ཀྱེ«པེ་»《སྣར་བསྒྲུབ། «པེ་»«སྣར་ཟེ། «པེ་»«སྣར་»ལཱི། ཙཱ«སྣར་བསྐོར། ཉི«པེ་»«སྣར་»མ། «པེ་»«སྣར་པོ། «པེ་པོ། «པེ་»«སྣར་»ཀྵ། «པེ་»«སྣར་པ། ༄༅«པེ་»«སྣར་»གདན། «པེ་»«སྣར་»+ལ། །«པེ་»«སྣར་གྷཾ། «པེ་»«སྣར་»+། བློབ་ཀྱི་་«པེ་»«སྣར་ཨོཾ་ཧཱུྃ། «པེ་»«སྣར་ལའོ། «པེ་»«སྣར་ནི། «པེ་ལའོ། །ཧཱུཾ་ཧཱུཾ། «སྣར་»ལའོ། །ཧི་ཧི། «པེ་»«སྣར་རྣམས། ་«པེ་»«སྣར་ནི། «པེ་»«སྣར་»གོ་་རིམས་ཇི་ལྟ་བ་བཞིན་དུ། ཉི«པེ་»«སྣར་»ཏི། །བློ་འབར་བའོས་ལ་འབད་པ་དག། མ་པ། དབུ་སྐྲ་གྲོལ་བ། པཉེན་དུ་བ་བསྡུར་འབྲས་རེའུ་མིགལྔས་རྣམ་པར་བརྒྱན་པ་་་་་་་་རྒྱུད། (རྣམས་ལ་བྱང་ཆུབ་ཀྱི་སེམས་ལ་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། བརྒྱུད། །བོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། །བ་བས། བན། བབ་བརབ་ཕྲེང་། །།བ་ཚེག་ཁྱིམ། །།།།་།
