لولهགིས་བཙོ། །རྒྱུ་སྐར་རྒྱལ་ལ་དྲག་པོའང་དབང་དུ་བྱེད། །དབང་དུ་:བྱ་བའི་རྒྱུད་དོ། །།ཨོ་ཀུ་རུ་ཀུལླེ་སརྦ་དུཥྛན་ནཱ་ཤ་:ཡ་ནཱ་ཤ་ཅིཡ་ཀཱི་ལ་ཡ་ཀཱི'ིལ་ཡ། :བྷཉྫ་ཡ་བྷཉྫ་ཡ། མར་ད་ཡ་མར་ད་ཡ། དྷྭཾ་ས་ཡ་དྷྭཾ་ས་ཡ། ཨ་པ་སཱ་ར་ཡ། ཨ་པ་སཱ་ར་ཡ། ཨ་བི་མི་ཏཾ་མེ་ཀུ་རུ་ཤཱནྟི་མེ་ཀུ་རུ་པུཎྚི་མེ་ཀུ་རུ་སརྦ་ས་པ་ཧུཾ་མེ་སྭཱ་ཧཱ། ཨོཾ་:ཧཱུྃ་ཤྲཱི་ཉྙཱཧཱུྃ་ཧཾ་ཧོཧསྔགས་འདིས་ཡང་ཉམེའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་:མགོ་བོ་ྋགསུམ་ལ་གནས་པའི་པདྨའི་སྣོད་དུ་ཟན་ལ་སོགས་པ་རྣམས་བདུད་རྩིའི་རང་བཞིན་དངུལ་ཆུ་བཞིན་དུ་བལྟས་ལ་ཡི་གེ་གསུམ་གྱི་སྔགས་ཀྱིས་བྱིན་གྱིས་བརླབས་ལ་བཅོམ་ལྡན་འདས་མ་དང་འབྱུང་པོ་ཐམས་ཅད་ལ་འབུལ་ཏེ་བར་བྱའོ། །ཨོཾ་མུཞེས་པ་འདིས་མཆོད་པ་ལ་སོགས་པ་ཐམས་ཅད་གཤེགས་པར་བྱའོ༑ཡོ། འདིས་ནི་བགེགས་ཆེན་པོ་རྣམས་ཞི་བར་འགྱུར་རོ། །གཏོར་མའི་ཆོ་གའོ༑། ཞི་ལ་ྋཞི་བའི་སེམས་:ཀྱིས་ཏེ༄༅། །རྒྱས་ལ་རྒྱས་པའི་སེམས་ཀྱིས་སོ། །ཤིན་ཏུ་ཆགས་པས་དབང་དུ་དགུག། ཁྲོ་བའི་སེམས་ཀྱིས་གསད་པའོ། །ཞི་་ལ་མཎྜལ་རྣམ་པ་སྟེ། ྋ། རྒྱས་ལ་རྫིང་གི་རྣམ་པའོ། །དབང་ལ་ཟླ་བ་ཕྱེད་པ་སྟེ། །ནམ་མཁའི་ཁམས་འདྲ་གསད་པ་ལ། ཌ་། ཞི་བ་ཁྲུ་གང་ཚད་ཀྱིས་ཏེ། །རྒྱས་་་པ་:ལ་ནི་ཁྲུ་དོའོ། །རྒྱས་པ་ཇི་ལྟ་དེ་བཞིན་དབང་། །གསད་@པ་སོར་མོ་ཉི་ཤུའོ༑། ཞི་ལ་ཁྲུ་ཕྱེད་གཏིང་དུའོ། །ཁྲུ་ཡི་ཚད་ཀྱིས་རྒྱས་པ་ལ། །མཐེ་བོ་་་ལྔ་:ཡིས་གསད་@པ་ལ། ང་། །རྒྱས་པ་ཇི་ལྟ་སྐྱེ་དེ་བཞིན་དབང༌། །ཞི་བའི་སྦྱིན་སྲེག་ཚེས་གཅིག་ལ། །རྒྱས་པ་ཟླ་བ་གང་བ་ལ། །བཅུ་བཞི་ལ་ནི་མངོན་སྤྱོད་དོ། །བརྒྱད་ལ་དབང་གི་ཁེལས་ཡིན་ནོ། །ཞེས་པ་ནི་གཞན་ནོ། །དེ་ནས་ཡང་་་་སྔགས་ཀྱིས་བྲིསོ། །སྦྱིན་:སྲེག་ཐབ་ཁུང་ཚད་ལྡན་པ། །ཞི་བ་ལ་ནི་ཟླུམ་པོར་བྱ། །ཁྲུ་གང་གི་ནི་ཚད་ཀྱི་ཐིག་། ཁྲུ་ཕྱེད་གཏིང་དུ་ས་:བརྐོས་ནསོ། །ཚོན་རྩི་་་།
