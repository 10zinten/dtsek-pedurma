གྲི་གུག་གཡས་ཀྱི་ཕྱག་ན་བསྣམས་པ། སྐུ་ན་ཨུཏྤལ་དང་ཐོད་པ་གཡོན་གྱི་ཕྱག་ན་་་བསྣམས་པ། མི་བསྐྱོད་པས་དབུ་བརྒྱན་ཅིང་རལ་པ་ཁམ་པའི་ཐོར་ཚུགས་ཅན། མི་བཟད་པའི་གད་རྒྱངས་ཅན་ཤིན་ཏུ་འཇིགས་པའི་སྐུར་བསྐྱེད་ནས། ཨོ་ཧཱིཧཱུྃ་ཕཊ་ཅེས་པའི་སྔགས་ཤེས་རབ་ལྷག་པར་གསལ་བར་བྱེད་པས་བཟླས་པར་བྱའོ། །མཁས་པས་རལ་གྲིའི་གནས་སུ་ནི། །མི་ཡི་རུས་པའི་ཕྲེང་བ་བསམ། །སྤྲོ་དང་བསྡུ་བའི་སྦྱོར་བ་ཡིས། །སྔགས་མཆོག་ཡང་དག་བཟླས་པར་བྱ། །ཞེས་སོ། །རྒྱ་ནག་རིམ་པའི་འཕགས་མ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།།བསྡུར་མཆན། ཀྱི་ང་ད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ་«པེ་»«སྣར་ཙི། 《སྣར་དུ། «པེ་»སྣར་པདྨ་ཞེས། སཔེ་མ། («པེ་»«སྣར་ཟད། «པེ་»«སྣར་»སྩོལ། །«པེ་»དབསྡུར་འབྲས་རེའུ་མིག། དྲ་བ། ལྔ་པ་ན། སྒྲོགརྒྱུད། (ཁོ་བོའི་་ཚེ་༽,་་། །།དཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། སྣར་ཐང་། །སྡེ་ཚན། རྒྱུད། རྒྱུད་འགྲེལ། པོད་རྟེགས། །མུ་༼དུ་བ་ཚན་གྲངས། ，། །ལྡེབ། ན_བ། བ—ན། བ—ན་ནན་ཕྲེང་། རྒྱུད། བར། །
