བསྟན་འགྱུར། རྒྱུད། ༼ཆུའིizjpདོན་རྟོགས་ཏེ། །བདེ་བ་ཅན་གྱི་ཞིང་དུ་མྱུར་བགྲོད་ཤོག་ཀཚོ། རྗེ་བཙུན་མ་་་འཕགས་མ་སྒྲོལ་མའི་དཀྱིལ་འཁོར་དུ་ཚེའཇུག་པའི་ཆོ་ག་ཞེས་བྱ་བ། དངོས་གྲུབ་བརྙེས་པའི་སློབ་དཔོན་ཆེན་པོ་བྷ་བ་བྷ་ཊས་མཛད་པ་རྫོགས་སོ། །།།/ཀའམ་ཧཱ་པཎྜི་ཏ་ཤྲཱི་བི་མ་ལ་ཀཱིརྟིའི་ཞལ་སྔ་ནས་དང་། :ལོ་ཙཱ་ཅིབ་ཤཱཀྱའི་དགེ་སློང་རྡོ་རྗེ་འཛིན་པ་ཞང་གྲུབ་པ་དཔལ་བཟང་པོས་བལ་པོའི་གྲོང་ཁྱེར་ཡམ་བུའི་་་ནུབ་ཕྱོགས་ཉི་མའི་གཙུག་ལག་ཁང་དུ་བསྒྱུར་ཅིང་ཞུས་ནས་གཏན་ལ་ཕབ་པའོ། །།།བ། ང་དང་ན་ལ་ཁོ་ཚོངེས་ཏེ། །ལ་ལ་ནི༌ཆེ། sཏེ་ཁགབ། བྱ་ཁྱུ། པའི་འཚོ་བ་ལ་། ་གྱི་ལོ་བའི་རྒྱུཀྱིས་ཆ་དེ་བབསྡུར་མཆན། ས་ཁོའི་དེ་རིང་ཁ་ཆེ་nམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །འཕགས་མ་སྒྲོལ་མའི་དཀྱིལ་འཁོར་དུ་འཇུག་པའི་བྱ་བ་ལ་སོགས་པ་བཞུགས། s::Q«པེ་»«སྣར་ར་མཎྜ་ལཱ། «པེ་»«སྣར་ཀྲྀ། «པེ་»《སྣར་»—། «པེ་»«སྣར་»«ཅོ་མ། @«པེ་»«སྣར་པའི་བྱ་བ། ཚེ«པེ་»«སྣར་—དུ། «པེ་»«སྣར་+ལ། «སྣར་ནུ། ཨེ«པེ་»«སྣར་»ལ། ཀྱེ«པེ་»«སྣར་+ལ། «སྣར་གཙོང་། «པེ་»«སྣར་+པ། «པེ་《སྣར་བསམ། «སྣར་»གྱིས། སྤྱི་«སྣར་བརྫུན། «པེ་»«སྣར་བརྟེན་པས། «སྣར་བརྒྱན། ཅེ«པེ་»«སྣར་དཀྲུག«པེ་»«སྣར་བསྙུང་། @«པེ་»«སྣར་»ལ། «སྣར་»ཀྱིས། ཉི«པེ་»«སྣར་པས། «པེ་»«སྣར་»ཀྱིས། «པེ་»སྣར་བརླབ། «པེ་»«སྣར་»+ཀྱང་། «པེ་»«སྣར་ཕྲ། «པེ་»«སྣར་བདུན། «པེ་»སྣར་ཅོང་། ཚེ«པེ་»«སྣར་དུམ་བ། @«པེ་»«སྣར་ས། འ«པེ་»«སྣར་གཡུར་ཟ། «པེ་《སྣར་»ཀྲྀ། ཁོ«པེ་»«སྣར་+བ། འགྱེ«པེ་»«སྣར་བརྩེ། «པེ་»«སྣར་»ཏ་དང་པུན་ད་རི། «པེ་»«སྣར་»ཀཚེ«པེ་»«སྣར་པ། ན«པེ་»«སྣར་རྒྱུ། «པེ་»«སྣར་»སྐྱེས། «པེ་»«སྣར་»ཅིང་། ཡོཉི«པེ་»«སྣར་»གྱིས་བརླབས། «པེ་»«སྣར་»ཏ། དགྱེ«པེ་»«སྣར་ཡེ། འགྱེ«པེ་»«སྣར་ནནྡྷེ། །(«པེ་»«སྣར་»ཏིཊྛི། །་་་་་ཛིལ་་་་་་«སྣར་རྣམ། ཉི«པེ་»«སྣར་པ། «སྣར་བཞག«སྣར་བཞིའི། «པེ་»«སྣར་»བཞི་ཡིས། @«པེ་»«སྣར་ལས། ཚི«པེ་»«སྣར་ཀྱི། ། ེ«པེ་»«སྣར་—པའི། །«སྣར་གྱིས། «སྣར་ལ། »«པེ་»«སྣར་—ལྔ། མེ«སྣར་མཚམས། @«སྣར་»ཀྱི། @«པེ་»«སྣར་པ་མིའི། ཉོ«པེ་»«སྣར་»ག། «པེ་ཀྱིས། 《སྣར་ཀྱི། ཉེ«པེ་»«སྣར་»སྤྱངས། «པེ་»«སྣར་གྱི། «པེ་»«སྣར་»མ། །
