།།བོང་ཁ་ལ་ཏཊིཀུཎྜ་ལ་བློ་ཁོག་དང་ཁ་ཏ་་་་་་ཆཀཉིདན༌ཁང་། ས་ཏཚེ། །བུg《་ཀྱི། »། ཁpཀ་ལ་ཁལ་ཆlཀྱི་། །ཀ། 《་《་༄༅། །རིག་པ་འཛིན་པའི་སྡེ་སྣོད་དུ་གསུངས་པའི་མདོར་བསྡུས་པའི་འཇམ་དཔལ་གྱི་སྒྲུབ་པའི་ཐབསག°བར་བྱུང་། སམལ༄༅། །རྒྱ་གར་སྐད་དུ། བིཅུ་དྷ་ར་པི་ཊ་ཀ་སཾ་:ཀིསྟ་མཉྩ་ཤྲཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རིག་པ་འཛིན་པའི་སྡེ་སྣོད་དུ་གསུངས་པའི་མདོར་བསྡུས་པའི་་་་འཇམ་དཔལ་གྱི་སྒྲུབ་པའི་ཐབས། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་བསྟན་པའི་ཆོ་གས་སྟོང་པ་ཉིད་བསྒོམས་པའི་རྗེས་སུ་པདྨ་དང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཐིག་ལེ་དཀར་པོ་རྣམ་པར་བསྒོམས་ནས་:སྤྲོ་བ་ལ་སོགས་པ་སྔོན་དུ་འགྲོ་བས་དེ་ཡོངས་སུ་གྱུར་པ་ལས་ཡི་གེ་:དྷཱི@དཀར་པོ་དེ་ཡོངས་སུ་་་གྱུར་པ་ལས་འཇམ་དཔལ་ཨ་ར་པ་ཙ་ན་གཅིག་པུ་འབའ་ཞིག་ཉིད་དོ། །སྙིང་གར་ས་བོན་:རྣམ་པར་བསམས་ནས་ཡང་དང་ཡང་དུ་སྤྲོ་བ་ལ་སོགས་པའི་རྣམ་་་པར་བྱའོ། །ཨོ་དྷིསྭཱ་ཧཱ་ཞེས་བྱ་བའི་སྔགས་ཀྱིས་བཟླས་པ་བྱའོ། །།རིག་་པ་འཛིན་པའི་སྡེ་སྣོད་དུ་གསུངས་པའི་མདོར་བསྡུས་པའི་འཇམ་དཔལ་གྱི་སྒྲུབ་ཐབས་རྫོགས་སོ། །ཀ་བ། །འདི་ན་ཡོངས་སུ་གྱུར་པ་།
