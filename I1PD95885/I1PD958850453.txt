རྒྱུད། ༽འགྲུས་ལྡན་པས་མངོན་སུམ་བྱེད། ཀེ། བསྒོམས་པས་ངལ་ན་སེམས་རྩེ་གཅིག་་་པས་སྔགས་བཟླས་པར་བྱའོ། འོ། །ལྟ་། དེ་ལ་སྔགས་ནི་འདིའོ། །ལ་ལ་། ཨོ་བྷིསྭཱ་ཧཱ། །འདིའི་ཁ་དོག་སྟོན་ཀའི་སྐར་མའི་ཕྲེང་བ་དྲི་མེད་ལྟར། །རབ་ཏུ་འོད་གསལ་་་་དཔག་ཏུ་མེད་པའི་འོད་ཟེར་འཕྲོ་བ་ཅན། །az། བཟླས་པས་སྐྱོ་བར་གྱུར་ན་བསོད་་་ནམས་ཡོངས་སུ་བསྔོས་ནས་ལྡང་། །ལྷ་ཡི་ང་རྒྱལ་དང་བཅས་ཅི་བདེ་བར་ནི་་་་གནས་པར་བྱ། །དཔལ་ལྡན་འཇིག་རྟེན་དབང་ཕྱུག་གིསོ། །།སྒྲུབ་ཐབས་ཆོ་་་ག་:འདི་བྱས་པའི། །བདག་གིས་ཅིབསོད་ནམས་གང་བསགས་དེས། །འགྲོ་་་་་རྣམས་དེ་ཡི་གནས་ཐོབ་ཤོག། པོ་ཏ་ལའི་:ཁརྨ་པཱེཎིའི་སྒྲུབ་ཐབས། སློབ་་་་དཔོན་པདྨ་འབྱུང་གནས་བློ་གྲོས་ཞབས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།ཁྱོནགཏོ་:མཛོད་དང་པོ་དང་བ་ཡིད་ལ་གྱི། ཁོ་ཏན་འཛོལ་བྲེལ་ཚཚོ་ཁ་བའི་ཀ་རྫོང་། མཚོ་མ་པང་དུ་Nyeཏཔཆུ་ཆེན་པོ་ལང་ཀ་བའི་ཆུ་ལ་ཆོ་ག་ན་ག་ན་ས་མཁོཀྱི་བལ་བ་ཚོ། ཀ་ཁའི་ཁ་བོ་གནམ་བསྡུར་མཆན། གཏུག་ཚོ་མི་ཁ་་གནས་དང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་ཁ་སམ། །ཚེ«པེ་《སྣར་»གྱིས། ༽«པེ་»«སྣར་དངོས། ༄༅«པེ་»«སྣར་»ཁ་སཟ། །ཁྱི་ཞས། ཞེལ། Y«པེ་»«སྣར་བགྱིད། «པེ་»«སྣར་»—དང་། ཉི«པེ་»«སྣར་—ཉ། «པེ་»«སྣར་»—དུ། «པེ་བརླབས། «སྣར་བརླབ། «པེ་»«སྣར་»—། ༄༅«པེ་»«སྣར་»—ཉ། «པེ་《སྣར་ཨོཾ། «པེ་»«སྣར་ཟེ། ཉི«སྣར་»«ཅོ་བྱེད་པའི། ཚོན«པེ་»«སྣར་»བྲེ་ཏ། «པེ་»«སྣར་»རིད་པ། ཉི«ཅོ་»པོ། «སྣར་ན། ལ་ན་གཉིས་བར་དུ་བདག་«པེ་»«སྣར་»དབྱུག ི«པེ་»«སྣར་»«ཅོ་»པ། «པེ་»«སྣར་ལྤགས། «པེ་»«སྣར་»ལྟ། ཚུ«པེ་»«སྣར་པས། «པེ་»«སྣར་ནས། /@«པེ་»«སྣར་པས། «པེ་»«སྣར་ན། བ་བ་བ་«པེ་»«སྣར་»—བ། ་ཁྱེ«པེ་»«སྣར་»ཆད། «པེ་»«སྣར་བསྐལ། «པེ་»«སྣར་བསྒོམ། །«པེ་»«སྣར་»བསྒོམས། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་བྱས་པ་ཡི། «པེ་»«སྣར་»གི། ལོ@«པེ་»«སྣར་»ཁ་སཟ། ཡིད་ཀྱིས་ཀ་སྤུ་ཁོ། ངེད་ཁྲོམ་ར་ད་ཀ་ལོ་ཀ་བ་པ་སྦྱོར་བ། ཞི་མའོ། རའཁ། ཁ་ཚོ་ཚང་མ་གང་ལ་ལ་།
