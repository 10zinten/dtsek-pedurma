པ་ཁྲོ་དང་ཆ་བློན་་་བ་དག། པའང་་བ་དེ་དུས་གཤེར་སྒོའི་དེའི་རྒྱབ་ཏུའོ། །ལ་དོན་དེ་དེའི་བྱ་བ་ཏུའོ། །དེ་བཞིན་དུ་ལྷའི་གནས་དེ་ཁབ་ལྟར་རང་རང་གི་ཕྲགས་ཀྱིས་བདུག་gਲਐਲਤੋਂ।ਟਨੇਤੋਂਨੂੰਰੋਕਣਨਰੇਸ਼ਨਮੰਨਦੇਹਨਜੋਸਦਕਰਲਏ:::ཚེ་མི་༄༅། །སྒྲོལ་མ་མཆོག་ཐོབ་པ་ཞེས་བྱ་བ། །པ་འདྲ་བ་ན། །ཆོས་བརྒྱུད་པའི་འགྲོ་འོ། །བདུད་པ། །༄༅། །རྒྱ་གར་སྐད་དུ། ཏཱ་རཱ་ཨ་གྲྭ་པྲཔཱི་ནཱ་མ། ་བོད་སྐད་དུ༑སྒྲོལ་མ་མཆོག་ཐོབ་པ་ཞེས་བྱ་བ། རྗེ་བཙུན་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་གསུངས་པའི་རིམ་པས་སྟོང་པ་ཉིད་ཀྱི་བར་བསྒོམས་ཏེ། སྐད་ཅིག་ལ་་་རང་སྒྲོལ་མར་བསྒོམས་པའི་ཐུགས་ཀར་ཧཱུྃ་ཨིནྡྲ་ནི་ལའི་མདོག་ལྟ་བུ་བལྟ་བར་བྱའོ། །ལུས་སྐྱིལ་ཀྲུང་ལ་ཚིགས་དྲང་པོར་བསྲང་སྟེ་མཉམ་གཞག་གི་ཕྱག་རྒྱ་ལ་སོགས་་ཏེ་ལུས་གནད་ལྔ་དང་ལྡན་པས་རླུང་རྒྱུ་བ་སྤངས་ཏེ་ལྟེ་བ་ལས་མེ་རབ་ཏུ་སྦྱར་ྋནས། སྤྱི་གཙུག་ནས་ཐིག་ལེ་ཕབ་སྟེ་ཏཱཾ་ལ་ཕོག། ཏཱཾ་ལས་བྱང་ཆུབ་ཀྱི་སེམས་ཞུན་ཚུལ་་དུ་ལུས་ཐམས་ཅད་ལ་ཁྱབ་པས་བདེ་བ་ཆེན་པོའི་ཏིང་ངེ་འཛིན་སྐྱེས་པར་བསམ་ཞིང་སྲོག་རྩོལ་ལ་འབད་ན་ཞག་བདུན་གྱིས་དངོས་སུ་སྐྱེ་བར་འགྱུར་ཏེ། བྱང་སེམས་འཛག་ན་གདམས་ངག་དྲན་པར་བྱ། དེ་ནས་ཧཱུྃ་ལས་འོད་འཕྲོས་ྋཏེ་་སྣོད་བཅུད་ཐམས་ཅད་རང་ལ་བསྡུས་རང་ཡང་ཏྭཾ་ལས་ཀྱི་བསྡུས་ཏེ་ཏཱ་ཡང་རིམ་་་—།
