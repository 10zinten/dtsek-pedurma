ག་ག་ཎཅ་ཀཀྱe། །ཆ། z་ཁ་ཎ་ཌཁ་ཧྥ་ཎ་ཀྱང་། །བློ་༼ས་ਛgਗਵਰੁਣਾਨੋਂਉਸੰਨਨਵདང་། འཚེ་ཎ་ཅུ་ཊ་རྒྱལ། །ང་ཁ་པའི་ཡཁོ་པོ་། །ཞུ་ਮBgਉਠਨ।ਵਰੁਨਾਨੂੰਸਰਮਲਰੋਗਨਰੂਸ༄༅། །འཕགས་མ་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་ཌ་ཀ་པའི་ཐབས་ཞེས་བྱ་བ། །ཆesཏུ་ཀར»@ངཅ། ཁྲགཏོད། ་་་་་་་་་་ཀོ་བeyཆ་འི་ཆ་nཚོ་ཝིདཀྲུ་ཁཆཔ་། ༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་མཱ་རཱི་ཙྪི་སཱ་དྷ་ནཾ་ནཱ་མ། བོད་སྐད་དུ། འཕགས་མ་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་པའི་ཐབས་ཞེས་བྱ་བ། འཕགས་མ་འོད་ཟེར་ཅན་ཉིད་ལ་ཕྱག་འཚལ་ལོ། ་། བློ་ལྡན་སེམས་དཔའི་སྐྱིལ་ཀྲུང་གིས། །སྙིང་རྗེ་་་ཆེན་པོ་རབ་དམིགས་ལ། རྡོ་། སྔ་བར་སྙིང་གར་ཟླ་བ་ལ། །ཨོ་ཡིག་དཀར་པོ་་་་བསྒོམ་པར་བྱ། །རྣམ་སྣང་ཞེས་པའི་བདག་ཉིད་ཅན། །ཏི་ཡི་འོད་ཟེར་ཀུན་་དུ་དཀར། །ར་བ་ཟླུམ་པོ་མཆོད་རྟེན་ཏེ། །འབར་བའི་འོད་ཀྱིས་ཀུན་དུ་བརྒྱན། །བསྲུང་བའི་དོན་དུ་མདུན་དང་རྒྱབ། །གཡས་དང་གཡོན་དང་གཞན་ཡང་སྟེང་། །རིམ་བཞིན་མ་སོགས་ཀྱི་བྲན་ལ་སོགས། །བུ་མོ་ལ་སོགས་བུམ་སོགས་དང༌། །།ལུགས་མ་ཐང་སྐུ་བཤེས་གཉེན་དང་། །པོ་ཏི་ལ་སོགས་ཇི་སྙེད་པའོ། །ཏི་ནས་སྙིང་གའི་ཟླ་བར་མཾ། ་། མྱ་ངན་མེད་ཀྱི་ཡལ་གར་སྐྱེས། །འཕགས་མ་འོད་ཟེར་ཅན་མ་བསམ། །སྐུད་པ་ཁབ་རྣམས་བསྣམས་པ་མོ། །ལྡང་བའི་ཚུལ་གྱིས་།
