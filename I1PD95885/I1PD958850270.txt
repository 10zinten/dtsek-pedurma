ར་ཚོགས་ཀྱིས་རྣམ་པར་སྤྲུལ་པ། བས་བག་པ་ག་ལ་ཡ། ག་པ། བཎ་པ་ཡས་པ་ཅུང་ཞིག་བཏེ་པད་པ། ཉི་མའི་འོད་ཀྱི་ཕྲེང་བ་ད་ལུ་བདག་ཉིད་དམ་མ་བསམ་པར་བྱའི་ཟས་སྨན་མ་ཡོད་པ་དེར་སེམས་ནད་ཀུ་ཕྱུ་པ་སྔོན་དུ་འགྲོ་བ་ཡིག་་་་་་བུ་མ་གྱུར་བ་ལ་ལོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །མི་གཡོ་བའི་་་་སྒྲུབ་ཐབས་བཞུགས་སོ། །@«པེ་»«སྣར་»དྷཱ། ་བྱང་«པེ་»«སྣར་བསྟན། «པེ་»«སྣར་རང་། ཚ«པེ་»«སྣར་གཉིས། «པེ་»«སྣར་»བསྐོང་། གྱི«པེ་དམིག《སྣར་»དམིགས། «སྣར་གཟིགས། :པེ་»«སྣར་»འཐོབ། «པེ་»«སྣར་»བསྲུང་། «པེ་»«སྣར་གྱིས། «པེ་»ཀྲུང་། «སྣར་»བསྲུང་། ཁྱི«སྣར་གྱི། «པེ་»«སྣར་»སྒྲུབ། «པེ་»«སྣར་བཙུག«པེ་»«སྣར་བུའམ་ལྕགས་ཀྱི་ཕུར་བུས། «པེ་»«སྣར་+བམ། ༄༅«པེ་»«སྣར་+རོ། །«པེ་«སྣར་བཏང་ཏེ། «པེ་»«སྣར་»—སྔགས་པ་....འགྱུར་རོ། །(ཚེ«པེ་»«སྣར་ཐོབ། འགུ«པེ་»«སྣར་དེས། ཚེ«པེ་»«སྣར་+གཏུམ་པོ་ཆེན་པོ། «པེ་《སྣར་»ཀ་ར་ཀིརྟིས། ཀྱེ«པེ་»«སྣར་»ཙ། ་ཚེ«པེ་»«སྣར་»སོ། །འབྱོར་ལ། སྤྱི་ਤੋਂਨਹੀਂਸਨਲਾਲੋਂਸਰਬਸੰਸਨੇਨਸ਼ਾਬਹਾਲཌི་ཡ་?ཆ། ལ་ལཊི་བསྡུར་འབྲས་རེའུ་མིགཁ་ར་ཏ། །།བོདོ་ཆོག་ཁ། ཆཚོn་ཀ་བ་ཁ་ན་ཏ༌རྒྱུད། (,་ཆ་ཊི་ཎ་ཀ་ཀོ་ཀ་བ་ལ་ཀླུ་དཔེ་གཞི། །སྡེ་དགེ། །བེ་ཅིན། སྣར་ཐང་། གཅོ་ནེ། །ཎཏོསྡེ་ཚན། }རྒྱུད། ་རྒྱུད་འགྲེལ། ་༼་རྒྱུད། aaEརྒྱུད། *བོད་རྟགས། མུ་ཚན་གྲངས། །ལྡེབ། བནརབ། བབས། ཤབཤབ། བ། _བཔ}ཕྲེང་། :ཀ་ཁImpeཟླཚེག་ཁྱིམ། དང་ལ་:@raངག་པ། །ཚ་ཤོ་ཊའི་ཁྱབ་ཏཁལ་ཁ་ས་སྐཁ་ཁ་བའི་ན་ག་ས་ག་ལ་བཏོད་པ་ཚོ། བྱ་བ་བསླང་།
