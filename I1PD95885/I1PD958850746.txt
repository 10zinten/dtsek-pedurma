བསྐྱོད་པས་དབུ་བརྒྱན་པའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་མའི་སྤྲུལ་པའི་སྐུས་་་་ཕྱོགས་ཐམས་ཅད་དུ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་མཛད་པར་བལྟ་ཞིང་སྔགས་་བཟླས་པར་བྱའོ། །ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། སྔགས་མཆོག་འདི་ནི་རབ་་་མཆོག་སྟེ། །སངས་རྒྱས་ཀུན་གྱིས་ཕྱག་བྱས་པ། །བཀླགས་པ་ཡིས་ནི་དངོས་གྲུབ་བྱེད། ་། གསལ་པོར་རྡོ་རྗེ་གུར་ལས་གསུངས། །དེ་ནས་གཟུངས་སྔགས་ཅིགཞན་ཡང་། ན་མོ་ཨཱརྱ་བ་ལོ་ཀི་ཏེ་ཤྭ་ར་ཡ་བོ་དྷི་སཏྭཱ་ཡ་མ་ཧཱ་སཏྭཱ་ཡ། །མ་ཧཱ་ཀཱ་རུ་ཎི་ཀཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སརྦ་:དུཏྟཱན་པྲ་དུཏྟཱན་མ་མ་ཀྲྀ་ཏེ་ཛཾ་བྷ་ཡ་སྟོ་བྷ་ཡ་མོ་ཧ་ཡ་བནྡྷ་ཡ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ་སརྦ་དུཥྛན་་སྟེ་བྷ་ནི་ཏཱ་རེ་སྭཱ་ཧཱ། རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།པཎྜི་ཏ་ཆེན་པོ་ཆོས་འབྱུང་:བློ་གྲོས་ཞབས་ཀྱིས་མཛད་པའོ། །།།་དང་་འབར་དང་པོ་བ་དག་རྒྱང་། :་པ་ག་ངབབ། །བསྡུར་མཆན། །།མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ་འ«པེ་《སྣར་»ར། ན«པེ་»«སྣར་པོར། «སྣར་རྒྱན། །«པེ་»«སྣར་གི། །«པེ་»«སྣར་»—རྣམས། «པེ་»«སྣར་—ཉ། «པེ་»«སྣར་ནས། ®«པེ་»«སྣར་»+ཡི་གེ། ས«པེ་»«སྣར་»བཅིངས། ་@«པེ་»«སྣར་ཟེ། <@«པེ་»«སྣར་»ཁྲིས། «པེ་»《སྣར་—རྣམ་པར། ་་་། ཀྱི«པེ་»«སྣར་»དང༌། «པེ་»«སྣར་ཟེ། «པེ་»«སྣར་»དུ། «པེ་»«སྣར་»ནི། །«པེ་»«སྣར་»ནོ། ་པོ«པེ་»«སྣར་ནས། ་«པེ་»«སྣར་»—ཅན། །«པེ་»«སྣར་»འཛུབ། ་«པེ་»«སྣར་གྱི། «པེ་»«སྣར་»པའི། «པེ་»«སྣར་ཉེ་བའི། །«པེ་»«སྣར་ན། «པེ་»«སྣར་མ། ་«པེ་»«སྣར་ཕྱོགས། «པེ་»བ་ལ། «སྣར་»བ། «པེ་»«སྣར་གཅིག—།
