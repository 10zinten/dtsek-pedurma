བསྟན་འགྱུར། རྒྱུད། ཆུ༽དང་མཚུངས་པའི་སྐུར་བསམ་པར་བྱའོ། །དེ་བཞིན་དུ་བསྒོམས་ནས་སྔགས་་་་་བཟླས་པར་བྱའོ། །ཨོཾ་བ་སུ་:དྷཱ་རི་ཎཱི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་བ་སུ་སྭཱ་ཧཱ། :ཨོཾ་བ་སུ་ཤྲཱི་ཡེ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཅིབ་སུ་མུ་ཁི་སཱ་ཧཱ། ཨོཾ་བ་སུ་བྷི་ཡེ་སྭཱ་ཧཱ། སོ་སོའི་་་་བདག་ཉིད་ལ་བའི་ལྕི་བའི་མཎྜལ་གྲུ་བཞི་པ་ཁྲུ་གཉིས་པའི་ཚད་དུ་བྱས་ནས་ཐུན་མཚམས་གསུམ་དུ་མེ་ཏོག་དྲི་བཟང་པོས་སྟོང་ཕྲག་བཞིར་ྋམཆོད་པར་བྱའོ། །ཟླ་བ་དྲུག་ཏུ་བཟླས་པས་ཡིད་ཀྱི་ཤིང་རྟ་ཡོངས་སུ་གང་བར་འགྱུར་རོ། །ཇི་ལྟར་རྙེད་པའི་མེ་ཏོག་:གི་སྦྱིན་སྲེག་འབུམ་ཕྲག་བཞི་བྱས་པས་དཔལ་ཆེན་མོར་འགྱུར་བའོ༑། ནོར་རྒྱུན་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།།འབྲས། དེར་ཡིག་་་ཟླ་ར་ཊ་བསྡུར་མཆན། །་་་་ང་ཁ་ཆུ། །eeཀྱང་དར་ས་་༌་་ཀཱ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོད་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ལྷ་མོ་ནོར་་་་རྒྱུན་མའི་སྒྲུབ་ཐབས་རིམ་པ་བཞུགས་སོ། །«པེ་»«སྣར་ཝི། @«སྣར་ར། &«སྣར་»—རྣམ་པར། «སྣར་»དཀྱིལ་འཁོར་གྱི། པོ《སྣར་བར། །«སྣར་»མི་བསྐྱོད་པས་དབུ་བརྒྱན་པའོ། །«པེ་སྟེར་རྟེན། «སྣར་»སྟེན། «པེ་ནོར་བུའི། «སྣར་»ནི་ནོར་བུའི། «པེ་»«སྣར་རྟེན། ཀྱེ«སྣར་»གྱིས། གེ«སྣར་བཙོ། །«པེ་»«སྣར་»དྷ་རི་ཎི། «སྣར་»ཤི། «སྣར་ཨོཾ་བ་སུ་ཤྲཱི་ཡེ་སྭཱ་ཧཱ། «སྣར་»བྲི། («སྣར་ཁྲེ་@«སྣར་»ཤི། @«པེ་»དུ། «སྣར་»ཏུ། «སྣར་»སྦྱིན་བསྲེག«སྣར་གསུམ། ཁྱེ«པེ་རོ། «སྣར་»རོ། །ཞེས་སོ། །བ་། དོན་ཅཏུའི་སའ་་ལ་འཚོ་ར་ཚེ། །།ཁ་ཀ་འཊྚ་ཚོਰgਸਥਾਨਨਾਲੋਂਸਪਾੜਨੂੰਉਮਰਸੰਸཁོ་ཚོའི་ང་:ཚོ་ཁ་ཕྱོགས་ཏོན། །ཁ་F་གཙོ་་n/ཀའ་ས་ཏཥཕྱོགས་ਨੌਲਤੋਂਇਨਸਾਨਨੂੰਹਿੰRਰਨਸਲਫਰਇੰਨਾਲਾਲཡིཀྱག་།
