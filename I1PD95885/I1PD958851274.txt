རསྟོབས་པོ་ཆེའི་སྒྲུབ་ཐབས། ཕུང་པོ་ཆེའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །་་་། །དེ་དག་་་བ། དེ་ལ་ལམ་དུ་སྔོན་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སྟོབས་པོ་ཆེའི་སྒྲུབ་ཐབས་གཉིས་བཞུགས་སོ། །«པེ་»«སྣར་རྒྱན། «པེ་»«སྣར་»ཨོཾ་མ་ཧཱ་བ་ལ་སྭཱ་ཧཱ། ཅེ«པེ་»«སྣར་པ་ནི། །:བད་ཀྱང་འདི་ཚོས་ད་ལྟ་བ། དབང་དུ་བས་པ་ལ་དག་པ་སྟེ། དེ་དེ་བཞིན་ཉིད། ::དེ་བ་དེ་བྱབསྡར་འབྲས་རེའུ་མིགབ། བྱེད་རྒྱུད། （དཔེ་གཞི། །ཁོསྡེ་དགེ། ༼པེ་ཅིན། སྣར་ཐང་། །ཅོ་ནེ། འོ་སྡེ་ཚན། *Sརྒྱུད། རྒྱུད་འགྲེལ། །པོད་རྟགས། །བ་མུརརཔ་བར་དུ་ནང་པ་ཚན་གྲངས། །།།ལྡེབ། བ་བབ་ན?་བ་བབ་—བ*ཕྲེང༌། ་། །ཚེག་ཁྱིམ། །དཔལ་ਇਟੀਲਿਓediਉਸਆਲਮgਈਲਸ਼ਰਬਰਫਟੋਨਉਨਉਲੋਂਸਪਾਸལོ་ར་ཌ་ཁོ་ཏི་ཟླ་་་་་ལ་ལ་ཎ་ཊ་བ་ལ་བླ་ཆ་ཟླ་བ་ਨਾਨੂੰਇਸਨੂੰਇੰਤਰਫ਼ਰਿਸਤੋਂਚའཥ་ཎི་ཀཟློས་ཁ་ཚོད། ཀ་ཁ་ཁྲོ་ལ་ན་མ་ར་ཎ་ཀ་ལ་པ༌མས་ཀྱི་ཚེ།
