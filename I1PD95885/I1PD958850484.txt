སྤྱིར་བདག་ཉིད་བསྒོམ་པར་བྱའོ། །སྐུ་མདོག་དཀར་པོ་རལ་པའི་ཅོད་པན་འོད་དཔག་ཏུ་མེད་པས་དབུ་བརྒྱན་པ། སྤྱན་གསུམ་པ། ཕུལ་བ་ཉིས་པ། དཀའ་ཐུབ་ཀྱི་ཚུལ་འཛིན་པའོ། བསྡུར་མཆན། ་རོལ་པ་པ་བདུན་པ། པ་པ། ཕུལ་། དུང་དུང་། སྐྱེ་བར་དཀོ(་་ལབ། །།མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སེང་གེའི་སྒྲའི་སྒྲུབ་ཐབས་རིམ་པ་བཞུགས་སོ། །«པེ་»«སྣར་ཨཱ། «པེ་»«སྣར་ཨ། «སྣར་ལ། ་སྣོར་གྱི། 'ཚ་བོ་ལྟར་རྒྱུན'»སྣར་སྐPeཔེསཏཱ། R@»«སྣར་»ཎི། པོ་བ་ཚེ«པེ་»«ལྟར་བིརྱ། ཁེ«པེ་»བའི། པེ་»«ལྟར་ནས། ཕཊ། དེ་ནས་གཟུངས་བཟླས་ཏེ། ན་མོ་ལག་ཏུ་ཡཱ་ཡ། ན་མ་ཨཱར་བ་ལོ་དེ་དེ་ཤཱ་ར་ཡ། བོ་དྷི་སཏྭཱ་ཡ། མ་ཧཱ་སཏྭཱ་ཡ། མ་ཏུ་ཀུན་དུ་ཏེ་ཡ། ཏང་། སོ་བསྡུར་འབྲས་རེའུ་མིག(gད་ལ་གོ། ཁྲི་༽་ཏེ། རྒྱད། (་རོ་ཊི། ཀ་ཀོ་ཊི། ཕྲེང་་ཌ་ཌཎ་ཌཊོསྔ:ཎཾ་ཌཱ་ཊ་ཌ་ཎཱ་བ་ཏོ་ར་ན་ཏཱ་ཌཎཱ་ཎཁ་ཏུ་ཤ་། ee@ས་པ་སྡེ་ཚན། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། *ཎgཁ་ཚོན་གྲངས། ་ngsཊི་རཀོའ་ཌ་ཌཱ་ར། འབཊུ་ཝི་ཀར། ཁzpཕྲེང༌། ，་ར་ཌ་ཎཎོཌ་ཌ''ཌ་ཌཊོ། ཎ་ཏུ་ཌ་ཎཥཎོ་ཥ་ཌ་ཎཥཥཥ་ཌ་ཌཎྜི་ཀ། །༌ཏཚུ་ཎ་ཎཌིགས་་ཎ་ཊིཁཀོ་ཊྛཎ་། ཋ་ཌ་ཎ་ཌཌ་ཎ་ཌཱ་ཌཱ་ཅུ་ཟླ་ཚོ། ༈ོgབ༌ཌཱ་ལ་ཀྵ་ཀ་རཏུ་སྐུ་ཎཌ་ོ་n/»ཁཾ་ཁཙོ་ཌཎཱ་ཎཎ་ཌི། །ོg་རཌ་ཌཆ་n། །མོ་འོ་རཁཌ་ཌ་ཌཎཾ་ཀ་ཌ་ཌཁྱུ་རྐྱ་ཎ༌ོ་ཁ/་ཀླུ་ཌཎྜི་ཏྟཾ། །འོ་ཌཎ་ཌ་༌༌་ཥྛ་ཌ་ཌཥ་ཥཥྛ་ཌཥཌ་ཎཊ་ཌཟི་བློཌ་ཌg་ཥ/n་ཁཊཾ་ཀ་ཋ་ཟླ་ཌ་ཌའི་ཊེཌe།
