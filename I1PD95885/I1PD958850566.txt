#་མ་ཁྲི་བ་ཁོ་ན་ལ་ཐུནངོ་་་དུ་བཙན། ཚེ་དང་ཁ་བ་བབས། ད་ལྟ་དབྱར་ཕྱི་::། ཚོ། །རྗེ་དེའི་ས་ན་བདུད་བློན་དུ་ཚོགས་ལ་ལོ་དཔེ་ཚོགས་་་་་་་་་་་། ད་ལོ་ཡངས་དང་ལོ་རྒྱལ་ཐོག་ལྔ་བ་ལོ་ཡོདཔ་ད་རང་ལུ་པ༄༅། །དཔལ་ལྡན་ངག་གི་རྒྱལ་པོའི་སྒྲུབ་ཐབས། ཏུ་བྱི་བ་ཁྲི་བ་ཚོ་བོ་རྗེ་བསྐོ་བྱ་བ་དང་། pཁ་:དཔང་ལོ་ཚུ་འཚོལ་ཡོད་མི་འདོད་ཀྱི་དྲན་པ་རང་གྲོས་བོད་དེ༄༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་:མན་ཝཱ་གཱི་རཱ་ཛ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། དཔལ་ལྡན་ངག་གི་རྒྱལ་པོའི་སྒྲུབ་ཐབས། འཕགས་པ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །འགྲོ་བ་མ་ལུས་པ་ལ་ཡང་དག་བདེ་བའི་མཆོག་་་་འདོད་རང་གི་བདེ་བ་ནི། །ཆུང་ངུ་ཡང་ནི་འདོད་མིན་རང་བཞིན་ཆེན་པོ་ཡང་། དག་བླ་མའི་བཀའ་ཡིས་ནི། །བློ་གྲོས་དམན་ཤིང་ཆུང་ལ་ཤེས་རབ་ཆུ་སྐྱེས་རྣམ་་པར་རྒྱས་བྱེད་མཁས་པ་ཡིས། །བཅོམ་ལྡན་འཇམ་པའི་དཔལ་གྱི་སྒྲུབ་ཐབས་་བདག་གིས་ཡང་དག་རབ་ཏུ་བྲི་བར་བྱ། །དད་པས་དགེ་བའི་མ་མ་ཡང་དག་་ཉེར་སྟོན་ཅིང༌། །གང་གིས་མི་དགེའི་བྱ་བ་ཡང་ནི་ཡོངས་སྤངས་ཤིང༌། །དེ་་་བཞིན་ཡིད་ནི་དག་པས་བླ་མ་ལ་ཐོས་ནས། །བཅོམ་ལྡན་རྟག་ཏུ་བསྒོམ་པ་ལ་སོགས་རྣམ་པར་བྱ། །དང་པོ་ཡིད་ལ་མངོན་འདོད་དྲི་མེད་དབེན་པའི་ཕྱོགས། །དྲི་བཟངས་མེ་ཏོག་ལ་སོགས་རྣམ་པ་མང་པོ་དང་། ལy། ཆོམ་པོ་དགྲ་དང་འཚེ་བ་།
