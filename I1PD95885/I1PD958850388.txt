.ར་ཁམRཀས་སུ་གནས་ཏ་ཟླ་ཁའི་ཁྱད་གང་བབ་ཏུ་གཏུ། །ཁ་དང་། །བློ་ཆེ། ཁོ་ར་བ་འད། ཁྲོ་དག་པ་སོགས་རིའི་ཚིག་ལ་གཏོའོ་ལྷང་བ་ཁ་ལོ་སྒྱུ་གང་ལ་བྱེད་ཀ་དོ་བ་ལ་ཕྱོགས་ག་ག་ན་དུ་བཏཁ་ཎ་ལག་ཐམག་ཨ་ཀ་གྱི་ཞབཞི་ལ་ཁའི་དག་ང་ཚོ་བ་ང་ལ་མཁོ་བ་ད་ཏོཔ་སྤྱོད་འཇུག་ཁོ་ནའི་བྱ་བ་ཁོ་ན་ཏཁརྒྱུ་དེ་བ་ཡོ་བ་ཏེ་བློ་ངན་པའི་དཔའི་ས་ལ་བོད་ཀྱི་ལུང་། འོ་བ་དང་། མ་ན༄༅། །སྒྲོལ་མ་དཀར་མོའི་སྒྲུབ་ཐབས། ཆུ་དེ། ཐབཤེས་ངག་དབང་ཀེ་་་ཞེས་པ་ངན་པ་ར་པའི་ཆུ་རྒྱུད་མཐར་འགྲོ་བ་ཚེ་སྔ་འཆཚོ། །ཏིའི་ཚད་བཏོད༌པགཉིཊའགྲོའུཌ་ཀྱི་ཚེ་དབང་རིག་དེབང་ཁྲོ་དེབ་ནི། s༄༅། ་བྱ། རྒྱ་གར་སྐད་དུ། སུ་ཀླུ་ཨིཏཱ་རཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། སྒྲོལ་མ་དཀར་མོའི་སྒྲུབ་ཐབས། བཅོམ་ལྡན་འདས་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །དཀར་པོའི་ལས་ནི་:ཉེར་སྒྲུབ་པའི། །སྒྲོལ་མ་དཀར་མོ་ལ་བཏུད་ནས། །ཏིང་ངེ་འཛིན་འབྱུང་གདམས་ངག་གིས། །རྙེད་པའི་སྒྲུབ་ཐབས་དེ་བརྗོད་བྱ། །དང་པོ་ྋརེ་ཞིག་སྔགས་པས་ཡིད་དང་མཐུན་པའི་གནས་སུ། བདེ་བའི་གདན་ལ་རྡོ་རྗེ་སྐྱིལ་མོ་ཀྲུང་བཅས་ཏེ། ཐང་ཀ་ལ་སོགས་:པར་གཏོགས་པའི་བཅོམ་་་ལྡན་འདས་མ་དཀར་མོ་ཉེ་བར་སྟོན་པ་ནི་སྒྲོལ་མ་ལ་མེ་ཏོག་དང་། བདུག་སྤོས་ལ་སོགས་པས་མཆོད་ནས། རང་གི་སྙིང་གར་ཨོཾ་ལས་བྱུང་བའི་ཟླ་བ་ལ་ཧུ་པ་དཀར་པོའོ། བཙོ། དེ་ལས་འོད་དཀར་པོ་ཕྱོགས་བཅུར་ཡང་དག་པར་སྤྲོས་ཏེ། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཐུགས་རབ་ཏུ་བསྐུལ་ནས། །རང་གི་སྙིང་་་་་གའི་ས་བོན་ལ་གཞུག་པར་བྱའོ། །།དེ་ནས་བར་སྣང་ལ་བཅོམ་ལྡན་འདས་མ་་་།
