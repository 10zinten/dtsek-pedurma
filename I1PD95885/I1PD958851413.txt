བསྟན་འགྱུར། རྒྱུད། ཆུའིདག་པར་བྱས་སོ། །ར་་པ་ཚད་མ་ལག་ལ་ཚེ་གང་རེ་ཁ་ཏ་ལ་གྱུར། །sཎྜམ་ན། །ཡོ་བྱད་ཀྱི་བར་ཁྲི་ལ་ལོ་ཀ་ང་ཀ་ཆུ་ཁྲོ་བ་དྲུག་བུ་སྐྱེ་བའི་བཤ་ཁུག་ཚོ་དང་། །སོགས་དོ་བདག་ལ་བསྒྱུར་མཆན། ཅི་ཁ་བ་ཆ་ཚིག་ཁོ་པ་བགྱིས་ན་ਸਥਾਨਬਲਰਾਜམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སྒྲོལ་མའི་་་་བསྟོད་པ་ལ་སོགས་པ་བཞུགས་སོ། །«སྣར་»༄༅། །སྒྲོལ་མའི་བསྟོད་པ་ལ་སོགས་པ་བཞུགས། «པེ་»«སྣར་ར། @«པེ་»«སྣར་—། «པེ་»«སྣར་»མཐོར་རམ། «པེ་»«སྣར་ཅན། «པེ་»«སྣར་—ཤིང་རྟ། དིུ«པེ་»«སྣར་མ། «པེ་»«སྣར་ཡ། «པེ་»«སྣར་མོའི། «སྣར་»སྐྱོད། ཚ«པེ་»«སྣར་»སྨྲ། །ཉི«པེ་»«སྣར་བུམ་པ། «པེ་»«སྣར་»འདི། དེ་བསསབ«པེ་»«སྣར་»ར། ཀ«པེ་»«སྣར་བའི། «པེ་»«སྣར་པས། དུས་«པེ་»«སྣར་»སྒྲོག«པེ་«སྣར་པ། «པེ་»«སྣར་»སྒྲོགཚེ«པེ་»«སྣར་»གཉེན། ཉི«པེ་»«སྣར་ལགས། «པེ་»«སྣར་»ཅིག་ཁྱེ«པེ་»«སྣར་བཤར། ཀྱེ«སྣར་»འོ། ཨེ«པེ་»«སྣར་»མའི་ནགས་་་་་ཀྱིས་ཉི། «པེ་《སྣར་ལ། ཁྱེ«པེ་»«སྣར་བསྐད། «པེ་»«སྣར་»ངགཚེ«པེ་»«སྣར་»+ཞིང་འཇམ། ཨེ«པེ་»«སྣར་»རིངས། ང་དུས་སུ་ཐ་་་་«པེ་»«སྣར་རིགས། «སྣར་+བ། ནི«སྣར་བཟློས། «པེ་»བརྡུང་གཏིང་། «སྣར་»བརྡུང་ཏིང་། «སྣར་»ལོང་། @«པེ་»《སྣར་»ཚོམས། @«པེ་«སྣར་»གའི། «པེ་»རྣམ་་་་པར་གཙུགས། «སྣར་རྣམ་པར་བཙུགས། «པེ་»«སྣར་—བདེ་ལྡན། «པེ་»«སྣར་»ཡིས། «པེ་»«སྣར་བར། «པེ་»«སྣར་གདར། ཚེ«པེ་»«སྣར་»ལྷ། «པེ་»«སྣར་བཞག«པེ་»གི་བཟུང་། «སྣར་»གི་གཟུང་། པོ«པེ་»«སྣར་»སྐྱེ། @«སྣར་»མཁྲེས། ཉི«པེ་»«སྣར་»པ་གྱི་«པེ་»«སྣར་»ར། ཁྱེ«པེ་»«སྣར་སྒོས། ཀྱེ«པེ་»«སྣར་ལྕགས། འཉེ«པེ་»«སྣར་»མདུ་བཙུམས༑«པེ་»«སྣར་»ཐིག་གཟེར། «སྣར་ཁྲུགས། ཉི«པེ་»«སྣར་»མོ། ཚ«པེ་»«སྣར་འབྱེད། «སྣར་»ཀྱི། «པེ་»«སྣར་ཡིས། @«སྣར་»སྐྱེས། «པེ་»«སྣར་»སྐོར། «པེ་»«སྣར་»ལྷ་མོ། དགྱེ«པེ་»«སྣར་»+རྫིང་ལྡན། ཀྱེ«སྣར་»བཅོད། ཉེ«པེ་»«སྣར་སྨྲས། «པེ་»«སྣར་»ལོཙྪ། ཁྱེ«པེ་»«སྣར་»གྱི་ལོཙྪ། བདག་ལ་མ་འོ་ཏོད་པ་དང་ཁ་ལ་ད་ལྟ་འདུ་བགམ་ཚོ་ཐུབ་ཀྱི་འབྲུ་བགོ་བར་ལོ་ག་ཚཛྫཡིབ། །ག་ད་བ་མ་འ། ན་བ་ག་བ་ག་བ་ག་བ་དང་འོ་མ་བྱེད་ན་ཁོང་ཚོགས་ཀྱི་པོ་ཏི་ཁ་ཆ་ཁ་ནང་ཆུ་མཛོད་།
