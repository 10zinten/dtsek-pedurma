བྱ་ལྟ་བ་དེ་རཊཾ་ཊ་ppཚོ་སྐྱ། །།ཆོ་ག་དཔག་བས་ཟླ་བ་བཀང་ཀ་ཏ་ཡ་་ཁཞུ་ལ་ཏ་ཥཊོ་ནཔ་ས་ཡ་དྲི་ལ་ཆ་བློ་ལ་ཀ་ཁ་། ་ཁ་mཆོ་ཀློབ་ང་ལ་ཁ་ཁ་རྫོང་། །ཆུ་ཡི་རྩ་བ་མ་འོ། །ཏ་ཏི་ཀར་། ཀ་ཁཚེ་། pཏོན་པའི་ཕྱི་ར་ཏཚོ་དང་། ཁ་ག་ང་ཀ་ར་ལི་ན་ཁ་ལག། ཅ་དཀnལ། །དེ་ཡང་སྦྲ་ཅན་དང་། ནང་། ད་ཆ་ཀློག་པ་དང་བདེ་བ་དང་༄༅། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མའི་སྒྲུབ་ཐབས། %B%ਕਿਸਾਨਸਟੋਰਸਲਮਨਸ਼ਨཡས་ཀྱི་བྱེཀ་ཁཙཁ་ཚོ་ཁའཁ་ན་ཡར་ཀློང་ཁ་པ་བློ་ཎ་ཁ་ཎའི་ཏོ་བའི་ཁང་ਪਿੰਕਸਲਈਨੌਸਟਾਲਛੁਟੋਲਾਨੂੰਘਰਲਏ་pnོ་དྲི་ཡ་ཎང་སppལྔ་པ་བ། །ཁ་་་ཁ་ཆའི་༄༅། །m། རྒྱ་གར་སྐད་དུ། པྲཛྙཱ་པཱ་ར་མི་ཏཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མའི་སྒྲུབ་ཐབས། བཅོམ་ལྡན་འདས་མ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་བསྟན་པའི་ཆོ་གས་རང་གི་སྙིང་་་གར་པདྨ་དཀར་པོ་དང་། ཏྟཱ་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་པྲ་དཀར་པོ་བཀོད་ནས་སྔོན་དུ་བསྟན་པ་བཞིན་དུ་ཐམས་ཅད་བྱས་ལ། པྲ་དང་པདྨ་ཡོངས་སུ་གྱུར་པ་ལས་་་བཅོམ་ལྡན་འདས་མ་ཞལ་གཅིག་མ། ཕྱག་གཉིས་མ་དཀར་མོ། ་ལུང་དང་་་་རྟོགས་པའི་བདག་ཉིད་རིན་པོ་ཆེའི་དབུ་རྒྱན་འཆང་བ། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་པོ་ཏིས་མཚན་པའི་པདྨ་གཉིས་གཡོན་དང་གཡས་ཀྱི་ལོགས་ཀྱི་ནང་་་་དུ་ལོངས་ཀྱིན་ཡང་དག་པར་བསྣམས་ཤིང་བཟུང་བའི་ཕྱག་གཉིས་ཀྱིས་ཆོས་་་བཤད་པའི་ཕྱག་རྒྱ་ཀུན་ནས་བཅིངས་ཏེ། པདྨ་དཀར་པོ་དང་ཟླ་བ་ལ་:རྡོ་རྗེ་སྐྱིལ་ཀྲུང་ཅན་མ། ལང་ཚོ་གསར་ལ་བབ་ཅིང་བདོ་བ། སྣ་ཚོགས་ཀྱི་ན་བཟའ་—།
