རྒྱུད། ཆུ༽འཕྱར་བ་ནི་གཡོན་གྱི་ཕྱག་བཞི་ནའོ། །མི་བསྐྱོད་པས་མནན་པའི་དབུ་རྒྱན་ཅན་བསམ་པར་བྱའོ་ཞེས་བྱ་བ་ནི་རྟ་མཆོག་ཡེ་ཤེས་རྡོ་རྗེ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ཏོ། །སྔགས་ཀྱི་བཟླས་པ་ནི། ཨོཾ་ཧ་ཡ་གི་བ་སྭཱ་ཧཱ། རྟ་མགྲིན་གྱི་སྒྲུབ་ཐབས་རྫོགས་སོ༑། །ལ་ད་ཤ་མ་ཏཱ་ཤ་ཡ། ཧ་ན་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་་་བསྒྱུར་བའོ། །།།བྱ་འགས་ཚོ། །:བ་པ་ལོ་བ་དགེ་སློངང་བ་རིས་བསྒྱུར་བསྡུར་མཆན། ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་ལ། ཉི«པེ་»«སྣར་—གྱི། «པེ་»«སྣར་འགྲིལ། ཚེ«པེ་»«སྣར་ནི། «པེ་»«སྣར་ལོཙྪ། ནད་པའི་བདག་དང་ཕྱི་ལྟ་བ་། །བeਨੂੰਸਮਾਨਨਾਲਸ਼ਰਤਲਾਇਲ་་:སྐྱ་བ་དང༌། བསྡུར་འབྲས་རེའུ་མིགarrང་ཟླ་་ཚོ་བྱིས་པར་ན། ་་་་་ན་་དང་། དང་ppལཀ་བློན་»nཁྲོ་སྐུ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། ན་མལྡེབ། ནབ། བ་ན། །བ་ན། ནབ། །aཕྲེང་། །།ཚེག་ཁྱིམ། །།།སྡེ་ཚན། རྒྱུད། རྒྱུད། བརྒྱུད། དམ་པ་གང་མང་//མུ།
