་འབད་ཧ། །ཁ་ཎ་ཁཚུ་ཁ། anཎའི་སྒྲུབ་པ་གནང་བ་neཟླ་བ་ཀུན་ཏཚོཀྱི་བཞི་སྒྲོག་ལསོ། །འོ་བ་ཆེ་བ་ཁ་སྐོe། ཀའ་བ། ཁ་མ་གྱུབ་ས་ཐོག་ཁཟླ་དང་བཛྲ་ཟེར་ལ་nཚུ་པཎཱ་ཎོབཟླ། བད་ཁྲི་ཡོན་ཚོ་ཀུན་ལ་བགླིང་ཚུན་ཏུ་ཏུ་ཁལ་ཁ་ཏུ་དཔའི་སྔ་ལྟས་བདག་གི་ཁ་ལ་ཀ་ཎ། ཏདེ་ནས་རྒྱུན་མ་བཞི་ལ་མཁོ་བ་གང་དེ་ཕྲལ་གསལ་ཏེ། དg༄༅། །།ཨུ་ལྕན་གྱི་རིམ་པའི་འོད་ཟེར་ཅན་གཡོ་གྱི་སྒྲུབ་པའི་ཐབས། །བ་ཏི་ཏི་ར་ཞེས་པ་ཕཊ། བློ་ལ་བློ་ཁལ་ཙཱཚོ་ལ་ཡང་འཐེ་ལ་བྱམ། །་ཁ་ཚུ་ཞི་ཚལ་ལ་སོར༌དབུ་འབྱེད། དཔྱད་རིངས་པར་ཚོན་འe/Srna::དཀྱིgམ། ཀ་ཤིང་། ཁ་བ་འབྱེད་པ། ཏི་ཊིཀ་ད་ན་ཁྱrnཏོ༄༅། །རྒྱ་གར་སྐད་དུ། ཨོ་ཌི་ཡཱ་ན་་ཀྲ་མཱ་རཱི་ཙེ་སཱ་དྷ་ནཾ། བོད་་་སྐད་དུ། ཨུ་༼ན་གྱི་རིམ་པའི་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་པའི་ཐབས། འཕགས་མ་འོད་ཟེར་ཅན་ལ་ཕྱག་འཚལ་ལོ། །དང་པོར་སྙིང་གར་ཉི་མའི་གདན་ལ་ཧཱུྃ་ངོ༌། །དེའི་འོད་ཟེར་གྱིས་བཅོམ་ལྡན་འདས་མ་བཀུག་སྟེ། སྣ་ཚོགས་པའི་མཆོད་པས་་ཡང་དག་པར་མཆོད་ལ། སྡིག་པ་བཤགས་པ་ལ་སོགས་པ་བྱས་ནས། སྟོང་པ་ཉིད་བསྒོམས་པའི་རྗེས་སུ་སྣ་ཚོགས་རྡོ་རྗེའི་རྣམ་པ་སེམས་ཀྱིས་བསམས་ལ། དེ་ཡོངས་སུ་གྱུར་པ་ལས་རྡོ་རྗེའི་ས་གཞི་དང་། རྡོ་རྗེའི་ར་བ་ལ་སོགས་པ་བསམས་ལ༑དེའི་དབུས་སུ་དཔལ་ཨུ་རྒྱུན་གྱི་གནས་གྲུ་གསུམ་དཀར་ཞིང་དམར་བའོ། །དེའི་དབུས་སུ་ཉི་མ་ལ་གནས་པའི་རྡོ་རྗེ་རྩེ་ལྔ་པ་ཧཱུཾ་གིས་དེའི་ལྟེ་བ་མཚན་པའོ། །དེ་ཡོངས་སུ་གྱུར་པ་ལས་བྱང་ཆུབ་སེམས་ཀྱི་ངོ་བོ་ཉིད་འཇིག་པའི་དུས་ཆེན་པོའི་མེ་ལྟ་བུ་གྱེན་དུ་རྣམ་པར་འཕྲོ་བ། ་ཟིལ་གྱིས་མི་ནོན་པའོ། །ཧ་ཧ་ཞེས་དགོད་།
