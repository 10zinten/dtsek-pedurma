བསོད་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་་་་་་བསྒྱུར་ཅིང་ཞུས་པའོ། །སུ་འགྲོ། །ལས་པ་བས་ཀྱང་ངེས་པར་བསྟེན་པ། འོངས་པ་ཡི་ནི་གནས་བཞི་པོ་དེ་ལ་བདག་བསྟོད་དུ་བར་བྱ། །དེ་ཚོ་དང་བྲལ་བར་པ་བསུར་མཆན། ལོ་ཡི་མེད་པ་ག་ཀྱ་རྣམ་གྲོལ་བའི། །་འོ་བ། སྐྱོ་མ། །ཚོད། :རྒྱ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་པའི། ཚེ«པེ་»«སྣར་ཤེས་རབ་...འཚལ་ལོ། །«པེ་»«སྣར་»ལས། «པེ་»«སྣར་ཅན། «པེ་»«སྣར་»ཛཱ། དཔ«པེ་«སྣར་»པའི། པས་བོད་ལ་དར་ཨེ«པེ་»«སྣར་»ཙ། དགཔཆོས་ཀུན། །ལྔ་བའི་མེ་ཏོགད་ལྟ་བུའི་པ། །བུདས་རྒྱལ་ཁྲི་འབསྡུར་འབྲས་རེའུ་མིག::ཀྱིངོ་མ་རྒྱུ་ཕྱོགས་པར། །༽ཚུན་རྒྱུད། （ནི། །དཔལ་ལྡན་ཁྲོ་བོ་བ། མ་། སྣར་ཐང་། །དཔེ་གཞི། །སྡེ་ཚན། སྡེ་དགེ། རྒྱུད། །།པེ་ཅིན། རྒྱུད་འགྲེལ། །:རྒྱུད། རྒྱུད། པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། །ནབ་། ་་::ན་ཕབ། །ཤན་བ། ནབ་ལན་། ་//ཚེག་ཁྱིམ། །ངས་བྲང་རྒྱུ་ཁོ་ཏོ་གེ་ཕྱི་ཤིང་འབབ་པ། ་་་་་རྒྱབ་ན་ཀ་ཁ་འབའ་བ་ཁ། །ཁྱི་ཁ་ཚོདྨ་ཀ་རྫོང་ཏོ་ཆེ། བྲག་པས། །ལས་ཀྱི་མ་འདྲ་བ་ཟླ། པས་ཀ་ཁ་ག་ང་ལ་ཁྲིབ་ཁཚོ་ཁང་བཞེངས། ཁྲའི་ཆུ་ལ་རབ་ཏྟ། གཏོ་སྐཥ་ཚོར་ཞིང་ཁ་བཀབ་ཁང་ལ་ཁ། གྱིས། །ཀཾ་ཀ་ཞིག་ལ་གྱི་གཞི་རྟེན། །འ&ཁསྒོ་ཞི་ཝཎ་བཀློག་འབྱེཋཀ་མ་ཏུ་ཀ་ན་ཏུད་ཏག་།
