བསྟན་འགྱུར། རྒྱུད། མུསྨོན་ལམ་དང་བསྟོད་པ་རྣམས་རྫོགས་སོ། །འཚེ་ཥ། །།ཆpཁ། །ཁར་ལ་ཎ། །ཚུལ་འདི་ནི་དཔང་:ལོ་ཙཱ'ིབ་དཔལ་ལྡན་བློ་གྲོས་བརྟན་པས་ཞོལ་ཕུག་ཏུ། རྒྱ་དཔེ་ལས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའི་ཡི་གེ་པ་དང་བསྐུལ་བ་པོ་ནི་དགེ་བའི་བཤེས་གཉེན་རིན་ཆེན་མཐའ་ཅན་ཡིན་ནོ། །།ཎ་ཆེ། ཁའི་ཁ་ཏོདུ་དུང་། །ཡབ་ཀ་ར་ཀ་ཏཁ»m་ཚོ་། ཁ་རལ་ཏོ། ཁཀཱ་ན་དབསྡུར་མཆན། pཚོ་མཚན་བྱང་འདི་སྡེ་»«ཅོ་ལྟར་བཀོད། «སྣར་ལས། «སྣར་མནན། ༄༅«སྣར་དངས། «པེ་»«སྣར་»—མ། ཚེ«པེ་»«སྣར་+པ། @«སྣར་བདག་ཀྱང་། «སྣར་»སྟེན། «པེ་»སྤྲོ་དང་། 《སྣར་པ་དང༌། །སྤྲོས་ལྡན། དགྱེ«སྣར་+ཐམས་ཅད། ཀྱེ«པེ་»«སྣར་»ཐམས་ཅད། 《སྣར་བཙ། ནས་བར་པ་དང་། ད་ང་བར་ན་་ས་ག་པརབབབབནབབངལྟ་ཀྱ:(«པེ་»«སྣར་སྟེ། འ«སྣར་»འཕངས། «སྣར་»དུ། «སྣར་དག@«སྣར་»—ལ། «སྣར་མ། བདག་@«སྣར་པོ་ཆེ་གསེར་གྱི། /«སྣར་གསུམ། 《སྣར་»+དོལ་ཤལ་ཅོད་པན་ལྡན་ཞིང་ནོར་དང་འབྲུ་རྣམས་ཕུན་སུམ་ཚོགས་པའི་རྒྱུ། །ཀྱེ«པེ་རྣམས་ཀྱི། སྣར་རྣམ་ཀྱིས། 《སྣར་བསྐལ། (ཉེ«སྣར་གི། །ཉེ«པེ་པ་བདེ་འགྲོའི། «སྣར་པའི་བདེ་འགྲོའི། སྐྱེ«སྣར་+སྤྱོད་པ། །ལེ«སྣར་ལ། ཚེ«པེ་ལ་ནོ། «སྣར་ལ་ནི། «སྣར་»སྒྲུབས། དེ་དག་གབ་པ་ཞེས་ཀྱི་«སྣར་སྟོ། «པེ་»«སྣར་»ལོཙྪ། «སྣར་»ཞིང་། «པེ་»སྣར་»+སེམས་ཅན་ཐམས་ཅད་བདེ་སྐྱིད་ལྡན་པར་གྱུར་ཅིག། བཀྱིག། S**འི་ཡོལབ། ཤེས་ཀྱང་སྦྱང་བྱ་བསྡུར་འབྲས་རེའུ་མིག་རྒྱུད། (སྡེ་དགེ། པེ་ཅིན། རྒྱུད་འགྲེལ། སྣར་ཐང་། རྒྱུད། །།རྒྱུད། རྒྱུད་རྒྱུད། དཔེ་གཞི། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང་། ཚེག་ཁྱིམ། ན་བན༽ནན?བ*ན་—བ་། །ལ་༼m/ོ་།
