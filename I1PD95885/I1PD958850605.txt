བསྟན་འགྱུར། རྒྱུད། མུདང་བསྡུ་བ་ལ་སོགས་པ་བྱས་ནས། །དམ་ཚིག་སེམས་དཔའི་ང་རྒྱལ་གྱིས་ཅི་་་་བདེར་གནས་པར་བྱའོ། །ཏིང་ངེ་འཛིན་དང་ལྡན་པའི་རྣལ་འབྱོར་པས་ཡང་་་་དག་པའི་ཡེ་ཤེས་ཀྱི་རྒྱ་མཚོའི་ཕ་རོལ་དུ་ཕྱིན་པར་གྱུར་ནས་དུས་རིང་པོར་མི་ས་ཐོགས་པར་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཐོབ་པར་འགྱུར་རོ་ཞེས་སོ། །ཚིག་ལྷུག་་་་གསལ་པོ་འདི་ཡིས་ནི། ཡེ། །དཀར་པོའི་སྒྲུབ་ཐབས་ཡང་དག་བྲིས། ་། པདྨའི་་་འབྱུང་གནས་བསོད་ནམས་གང་། །དེ་ཡིས་སྲིད་ལས་གྲོལ་བར་ཤོག་ཚོ། ཁྱི། ཚིག་ལྷུག་པའི་ཨ་ར་པ་ཙ་ནའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །་དང་དེ་། ང་ཚོ་དང་། དེ་ཙམ་ས་ད་དཔོན་འབར་བ་ལྔའི་སྟེང་གནས་དང་དེ་དག་ད་དུང་དེང་ང་ཚོ་ནང་འཚོལ་། ཞུན་ནི། དེ་ཚོ་ཁ་བདུན་པ་ཏེ་ཐེག་པ་དང་ལྗོ:ཀྱིང་། བྱཚོད་བསྡུར་མཆན། ཕྱི་ཁྲི་རིམ། །ཁྲིབ་པ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»༄༅། །ཚིག་ལྷུག་པའི་ཨ་ར་པ་ཙ་ནའི་སྒྲུབ་ཐབས་རིམ་པ་བཞུགས་སོ། །«པེ་»«སྣར་མུཀྟ། ན«པེ་»སྣར་»པོར། «པེ་»«སྣར་འཐུན། ༽«པེ་»«སྣར་ཉེ་བར། «པེ་»«སྣར་»པ། «པེ་《སྣར་»—རྣམས། «པེ་»«སྣར་»སྐྱེ། །(«པེ་»«སྣར་»ོQ«པེ་»«སྣར་ཉ། ༼《པེ་»«སྣར་པའི། «པེ་»«སྣར་»པ། །«པེ་»«སྣར་»ཅན། «པེ་»«སྣར་ན། བློ(«པེ་»«སྣར་»—ལ། «པེ་»«སྣར་པའི། «པེ་»«སྣར་»བུའི། «པེ་»《སྣར་+རང་གི་སྙིང་གི་ཟླ་བ་ལ། «པེ་»«སྣར་»དཀར་མོ། ན«པེ་《སྣར་སྒོམ། པ་ད་ལོ་མ་«པེ་»«སྣར་»ལ། ་་(འ«པེ་»《སྣར་པ། པོ«པེ་《སྣར་»—རྣམ་པར། །ཡ་༽@«པེ་»«སྣར་དྲི། །«པེ་»«སྣར་སིདྡྷི་མེ། «པེ་《སྣར་»ཀརྨ། ཚེ«པེ་»«སྣར་»ཤཱི། «པེ་»«སྣར་+། ཉེ«པེ་»«སྣར་བཛི། «པེ་《སྣར་»ཏུ། པེ་《སྣར་པོ། :ཉེ་ར། ང་ཚོའི་དབང་ཚད་ང་། པའི་ར་སྤྲོད་དང་དེ་ནས་བྱ་བ་རྟ་ལྟ་བུའོ་ཊ་བ་སྔ་ཤོག་། རྒྱུ་བ་པ་དེ་་་::་ན་མ་ཡིན་ནམ། ་ཆ་པ་ཚ་བ་འབྲིང་། །
