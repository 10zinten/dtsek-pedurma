《སྣར་བསྐུངས། ཀྱེ«སྣར་འབྱུང༌། «སྣར་བཀུས། 《སྣར་»—མ། «སྣར་»ཨུཏྤལ༑། 《སྣར་རྣམས་རྫས་ལ། ལེ«པེ་»«སྣར་»བཞེས། 《སྣར་+ཡང་། /«སྣར་བསྒོམས། ན་ ོ《སྣར་+ཏུ་རེ། འགེ«པེ་ཀཱི་མཱ་མ་ཨི་པ་ཙི། «སྣར་»གི་མ་མ་ཨཱི་བ་མི། འི«པེ་»«སྣར་ཐ། 《སྣར་»སྒྲུབ། འདེ«པེ་»«སྣར་བསམ། «སྣར་སྤྱོད་པར། དཔེ《སྣར་+དུ། 《སྣར་སྔགས། ེ«པེ་»ནས༑«སྣར་ནས་མི། འ«སྣར་+མི། འེ《སྣར་བས། འགེ«པེ་»«སྣར་རོ། ཀྱེ《སྣར་ན། ?«པེ་»«སྣར་ཨ་ཀའི། «པེ་»«སྣར་»རྗེའི། «པེ་»«སྣར་»གི། «པེ་»«སྣར་—ཡང་། །«པེ་»«སྣར་»རོ། །«སྣར་བྷིང་ག་ར། «པེ་»«སྣར་»འགི། «པེ་»«སྣར་རྗེའི། «པེ་»《སྣར་»—མི། གྱོ«པེ་»«སྣར་»ས། ཀྱེ«སྣར་»གཟིན། «པེ་»«སྣར་སྤང་། «པེ་»«སྣར་བརྒྱ་རྩ། འ«པེ་»《སྣར་—དེ། ཆེ«པེ་»«སྣར་»ནས་་བྱུགས། «པེ་»«སྣར་ལས་གཞན་ལ། «པེ་»«སྣར་མུཀྟི། («པེ་»«སྣར་»+མུ། ལ་འོ། པ་ར་སྤྲོ་བ་དག་མི་ལ་གྲུ་བར་དོབ་གཞན་ལ་ཡང་རབ་ཏུ་སྦྱོར་བ་ལ་བྱ་བ་་བའི་ཁྲི་ལུ་པ་སྒྲོལ་བ་ལ་ལན་བདུན་མངོན་པར་བསྔལོག་བདེ་བ་ན་་་་་་་བསྡུར་འབྲས་རེའུ་མིག་བརྗེ་བ་བྱས་ཀྱི་ས་ྋརྒྱུད། (ཚོ་དཀྱི»ཐའརཚུལ་ད་ལུ་ཡི་འགྲོ་བ་བ་བ་ན་དཔེ་གཞི། སྡེ་དགེ། མང་། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། །གང་ལ་སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། :་བརྒྱུད། འོeཔོད་རྟགས། ཚན་གྲངས། ་ལོཔ་ནང་ལྡེབ། བ་བ?ན་བ། བ*ན་བ་བར་ཕྲེང་། །ཉིན། །ཚེག་ཁྱིམ། །།
