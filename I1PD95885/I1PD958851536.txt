བ། སུ་ཏོདྡྷོ་མེ་བྷ་བ། སུ་པོ་མེ་བྷ་བ། ཨ་ནུ་རཀྟ་མེ་བྷ་བ། :སརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪ། སརྦ་ཀརྨ་སུ་ཙ་མེ། ཙིཏྟཾ་:ཤེ་ཡིཀུ་རུ་ཧཱུཾ། ཧ་ཧ་ཧ་ཧ་ཧོ། བྷ་ག་ཝཱན་:སརྦ་ཏ་ཐཱ་ག་ཏ། བཛྲ་:མཱ་མེ་མུཉྩ། བཛཱི་བྷ་བ་མ་ཧཱ་ས་མ་ཡ་སཏྭ་་ཨཱ༑ཞེས་བྱ་བ་འདི་ཡིན་ནོ། །བསོན་ནམས་འདི་ཡིས་འགྲོ་བ་ཀུན། །རྗེ་བཙུན་སྒྲོལ་མ་འདྲ་བར་ཤོག། བདག་ཀྱང་ཏིང་འཛིན་འདི་བསྒྲུབས་ནས། །འགྲོ་བའི་དོན་དུ་སངས་རྒྱས་ཤོག། རྗེ་བཙུན་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སློབ་དཔོན་་ཆེན་པོ་དཔལ་མར་མེ་མཛད་ཡེ་ཤེས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་དཱི་པཾ་ཀ་ར་ཤྲཱི་ཛྙཱ་ན་དང་། བོད་ཀྱི་:ལོ་ཙཱ་བ་དགེ་་་སློང་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།བསྡུར་མཆན། ཀུན་ལ་ལོང་མེད་ཀྱང་གྲང་དུ་གདོང་གི་པའི་བྱ་བ་བྱུང་། །མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད། «པེ་»«སྣར་»མ།  ི«པེ་མ། «སྣར་བ། «པེ་»«སྣར་བིགྷཱན། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་གོང་། །«པེ་»«སྣར་»སྒྲོལ། «པེ་བདུན། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་དྷ། ཨུ«པེ་»«སྣར་སྤྲོ་བའི་བདག་པོའི། @«པེ་»«སྣར་ལ། ཚེ«པེ་《སྣར་»—དྲིའི། «པེ་»«སྣར་+སུ། «པེ་»«སྣར་སརྦ། ཚེ«པེ་ར་མ། «སྣར་མུ། «སྣར་»གཾ་ཏྲེ། («པེ་»«སྣར་»པར། «པེ་»«སྣར་»སརྦ། ཉི«པེ་»«སྣར་»བྷ། «པེ་»«སྣར་»ཉ། ༄༅«པེ་»«སྣར་»ལས། «སྣར་པས། «སྣར་»སྟིམ། ཚེ«པེ་»«སྣར་པོ། O«པེ་»«སྣར་པ། «པེ་»«སྣར་»སྐུ། (གྱེ«པེ་»«སྣར་ཡིས། «པེ་»«སྣར་བརྟན། ཉི«པེ་»«སྣར་གཞག«པེ་»«སྣར་ཧཱ། «པེ་»«སྣར་བསྐུལ། «པེ་»«སྣར་ལ། ་པོ«པེ་《སྣར་»ཉ། ༄༅«པེ་»《སྣར་»བ། ་ཉི«པེ་»«སྣར་གསོལ། །«པེ་»«སྣར་སྲུང་། ཁྱོ«པེ་»«སྣར་»ཙ། «པེ་»གང༌། ཨེ«པེ་»«སྣར་»—ཏུ། «སྣར་བརླབས། «པེ་»«སྣར་བསྟིམ་སྟེ། ཆེ«པེ་»«སྣར་»+དེ། —།
