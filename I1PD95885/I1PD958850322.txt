བར་བྱེད་དོ། །ཨོཾ་ཛཱི་བློ་སྭཱ་ཧཱ། སཞེས་བྱ་བ་བརྗོད་ཅིང་ཁྱོར་བས་སྦྱིན་་པར་བྱའོ། །མ་གྲུབ་པར་གྱུར་ན་དཔལ་རྡོ་རྗེ་ཧཱུྃ་མཛད་དུ་བདག་ཉིད་བསྒོམས་ལ་གནོད་གནས་ཀྱི་སྐུ་གཟུགས་ལ་དུག་དང་སྐེ་ཚེ་དང༌། ནིམ་པའི་ལོ་མ་དང་། དྷ་དུ་ར་དང༌། དུར་ཁྲོད་ཀྱི་སོལ་བས་རབ་ཏུ་བྱུགས་ལ། བསྲོ་ཤིང་་་སྐེམ་པར་བྱའོ། ཀ། ཕྱིས་ནི་དུར་ཁྲོད་ཀྱི་མེ་ལ་ཁྲོས་པས་གདུང་བར་བྱས་ན་་་་ངེས་པར་:འགྲུབ་པར་འགྱུར་རོ། ༽། དཔལ་གནོད་གནས་ཀྱི་སྒྲུབ་ཐབས་་་་་རྫོགས་སོ། །།།པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་ལྟེ་བའོ༑། །།།(༽དོ༽རོ། ལན། བསྡུར་མཆན། སམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །གནོད་གནས་་ཀྱི་སྒྲུབ་ཐབས་བཞུགས་སོ། །ི«སྣར་»ཛཾ། «སྣར་»སྭ། «སྣར་»ཀྱིས། ཚེ«པེ་»«སྣར་»—ཕྱི་རོལ་གྱི། «སྣར་གིས། ཚེ«སྣར་»+གཉིད། «སྣར་»ཊི། «པེ་»ཁཾ་གིས། 《སྣར་སུ་དབང་གི། ཁྱེ«སྣར་»བཟང་མོ་དང༌། གྱེ«སྣར་»སྦ། ཉེ«སྣར་»ཀྱི། བདག་ཆོ་གང་། «པེ་གྱི། «སྣར་གི། «པེ་»ཏོག«སྣར་»རྟོག«པེ་»«སྣར་»མགོ་བོའི། ༄༅«སྣར་—པོ། «པེ་»གྱི་ཕྱོགས་སུ། «སྣར་»གྱིས་ཕྱོགས་གཅིག་ཏུ། །«སྣར་བསམས། @«སྣར་»ཨུད་པ་ལ། «སྣར་»+དང་། «སྣར་»མཁྱུད། ཚེ«སྣར་འབབ། འགྱེ《སྣར་»པོའི། «པེ་»པོའི་པི་ཙི། །«སྣར་»བའི་ཙི་པི། འ«སྣར་»མ། སྐྱེ«པེ་»ཙེ་རེན་དྲི་ང༌། «སྣར་»ཙ་རེན་དྲམ་པོ། «པེ་»«སྣར་ལེན་དའོ། ཚེ«སྣར་»སྣོད། ཁྱེ«སྣར་གྱིས། ཨེ«སྣར་»པོ། འེ«སྣར་»ཆེར། འགྱེ«སྣར་»འིཀི་ལི་དང༌། ནགེ«སྣར་»ཀྱིས། །«སྣར་»ག་@«པེ་ཡ་ན་བརླ། «སྣར་»ཡི་ནི་བརླབ་པ། ི«པེ་»ལེན་དྲ། 《སྣར་ལེན་ད་བ། «སྣར་ནོར་སྦྱིན། «པེ་»«སྣར་མཱི་ལ་ནི། «པེ་པི་ཙི་ཀུཎྜ་ལི། «སྣར་ཙི་པི་ཀུཎ་་ཊ་ཡི། @«པེ་»ཛ་ལན། «སྣར་»ཙིལེན་དྲ། «པེ་»མགོའི། «སྣར་»གོའི། «སྣར་པར། བ།
