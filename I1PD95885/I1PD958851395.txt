བསྟན་འགྱུར། །རྒྱུད། ཆུ། པདྨ་དང་ཟླ་བའི་གདན་ལ་སཱ་ལས་བྱུང་བའི་ལྷ་མོ་མདོག་ལྗང་གུ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་གཡས་ན་རིན་པོ་ཆེ་བརྩིགས་པ་བསྣམས་པ། ཨུཁ་གཡོན་ཨུཏྤ་ལ་མཆོག་སྦྱིན་དང་བཅས་པའོ། །རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པ། ཞབས་རོལ་པས་་བཞུགས་པའོ། །གསོ་བྱེད་མ་ཞེས་བྱ་བའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སོ། །PP། །པདྨ་དང་ཟླ་བའི་གདན་ལ་ཙུ་ལས་བྱུང་བའི་ལྷ་མོ་མདོག་སེར་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། གཡས་རིན་པོ་ཆེ་དང་བཅས་པའི་མཆོག་སྦྱིན། །གཡོན་རིན་་པོ་ཆེས་གང་བའི་ནོར་བུའི་རྐྱལ་པ་བསྣམས་པ། ཞབས་རོལ་པས་བཞུགས་པའོ། །ནོར་སྟེར་མ་ཞེས་བྱ་བའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སོ། །ཁ་ཁ། ་རོལ་བོ་::ཀན་པའི་དེ་འདྲ་བ་ཞེས་བྱ་བའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སོ། །།།གཏོ་པདྨ་དང་ཟླ་བའི་གདན་ལ་ཧཱུྃ་ལས་བྱུང་བའི་ལྷ་མོ་མདོག་སྔོན་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པ། གཡས་དང་པོ་སྐྱབས་སྦྱིན་པ། །གཡོན་ཆོས་འཆད་པའི་ཕྱག་རྒྱ་ཅན་ཞབས་རོལ་པས་བཞུགས་པའོ། །མི་འཇིགས་པ་ཞེས་བྱ་བའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སོ། །།།ལ། ཀུན་ཁཞུ་བ་ཕྱི་ལ་ཞཎཥ། བྱ་བའི་ལ་ལུའི་ནང་དུ་བྲབས་དོ་ཏོ་ཥ་་་་་སྤྲོ་བ་ལུ་པདྨ་དང་ཟླ་བའི་གདན་ལ་ཁཾ་ལས་བྱུང་བའི་ལྷ་མོ་མདོག་སྔོན་མོ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། །གཡས་ན་ཨུཏྤ་ལ་སྔོན་པོ། གཡོན་ཤེལ་གྱི་ཕྲེང་བ་བསྣམས་པ། རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པ། ཞབས་རོལ་པས་བཞུགས་པའོ། ་ཀུ་། བསོད་་་ནམས་ཅན་ཞེས་བྱ་བའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སོ། །ཁམཚོ། ཀོ་ཎ། །ཀུ། ཚོ་་འི་སྒྲོལ་བ་ལ་ཐུབ་ཚོ། །།།ཎོ་nraང་ལ། རྒྱུ་S/s?ཊོ་ངrn་་ཎཔདྨ་དང་ཟླ་བའི་གདན་ལ་ཏྲ་ལས་བྱུང་བའི་ལྷ་མོ་མདོག་སེར་མོ་ཞལ་གཅིག་_།
