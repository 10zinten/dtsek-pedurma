བསྟན་འགྱུར། རྒྱུད། ཁ༽རྫོགས་སོ། །།་་་་་་་་་་་ང་ཚོ་ཁ་ཁཌ་ལ། །p/ཀྱི་དང་ལ་ལ་གང་ཞིག་འདི་འདོན་པ་ལ་ཤེས་རབ་སྩོལ་བར་ཁས་བླངས་སོ། །རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་ཛྷཱ་ན་ཤི་ལ་དང་། ཁ་:ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་འབར་གྱིས་་་བསྒྱུར་བའོ། །།།དེ་ལ་གང་ལ་:ཁ་ཚིག་ཁས་བ་ལ་སྤྲོ། །ན་ང་དཀའ་བས་ན་ཁ་དབང༌། :ཚོ་ལ་ངང་ཡཾ་ཎ་ཀཞུང་དེཀ་ཎg་་་བསྡུར་མཆན། །།་བ་ག་བ་ག་བ་ཏི་ཏ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»དབྱངས་ཅན་མའི་བསྟོད་པ་པཎྜི་ཏ་ནག་མོའི་ཁོལ་གྱི་མཛད་པ་པཎྜི་ཏ་ཛྙཱ་ན་ཤཱི་ལ་དང༌། དགེ་སློང་ཆོས་འབར་གྱི་འགྱུར་་་་་བཞུགས་སོ། །«པེ་»ས་ར་སསྟོཏྟཱ་ན། «སྣར་སྭ་ར་སྭ་སྟོཏྟཱ་ན། «པེ་»«སྣར་»—ལ། «པེ་»《སྣར་ལོ། །མི«པེ་»«སྣར་»—པ་དང་། «སྣར་འབྱེད། ཚེ«པེ་»སྣར་»ལྡན་པའི། ཚེ«པེ་»«སྣར་དང་། «པེ་»«སྣར་»གཏེར། ཉི«པེ་»«སྣར་+ནས། (ི«པེ་»«སྣར་»རབ་མཛེས། «པེ་»«སྣར་རྣམས། །«པེ་»«སྣར་»ཅིའི། «པེ་»«སྣར་བསྐྱོད། @«པེ་»«སྣར་མུད། «པེ་»«སྣར་པས། ་«པེ་»«སྣར་»སེལ། །༼ཡི༼གང་ང་:ཡ*S/«པེ་»«སྣར་སི། ཚེ«པེ་»«སྣར་»ལོཙྪ། བློ་གྲོས། ཀྱི། སྤྱིབསྡུར་འབྲས་རེའུ་མིགཉིན་ལན་ཁང་བཞི་བབབ། བརྒྱད། (ང་ལ་གཏེ། ཀཅི་ཡདཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། །སྡེ་ཚན། སྔགས། །སྔགས། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བ་ན་། བ་བ་བ?ནབ་ན་ཕྲེང་། །ཀྱི་ཚིག་རྐང་། ཚེག་ཁྱིམ། ་།
