རྒྱུད། ༼མུ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་ཨ་མུ་ཀཱ་སུ་བནྡྷ་:ན་མུཀྟི་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་པའི་སྔགས་་་འདིས་ཨརྐའི་ལོ་མ་ལ་བསྒྲུབ་བྱའི་མིང་བར་དུ་བཅུག་སྟེ་བྲིས་ནས་སྡིག་པའི་ཚང་དུ་སྦས་ན་དེ་བཅིངས་པ་ལས་གྲོལ་བར་འགྱུར་རོ། །རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་པའི་་ཐབས། ཁ་ཁ། ཡང་དག་བྲིས་ལས་ཉེར་བསགས་པའི། །ཀ། བསོད་ནམས་གང་་་་དེས་འཇིག་རྟེན་འདི། །བདེ་གཤེགས་གནས་སུ་འཇུག་པར་ཤོག། རྡོ་རྗེ་་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །ན་ཏེ། །ཀུན་ཏུ། །ཚུལ་ཁ་ཎ་ཁོ་ན་ད་ཁྲུཏུཥའབཔཎྜི་ཏ་ཆེན་པོ་རིན་ཆེན་འབྱུང་གནས་ཞི་བའི་ཞབས་ཀྱིས་མཛད་པའོ། །ལྷ་མཆོད་ཀ་ར་བཱི་རའི་མེ་ཏོག་ལ། །ཐུན་མཚམས་གསུམ་དུ་བརྒྱ་རྩ་བརྒྱད་་་བཟླས་ནས། །འདོད་པའི་མཆོག་ཉིད་བླང་བྱ་ཟླ་གཅིག་གིས། །འཕགས་མ་སྒྲོལ་མས་ངེས་པར་སྩོལ་བར་མཛད། །ཨོ་ཛཾ་བྷེ་མོ་ཧེ་སྭཱ་ཧཱ། འཁ། །ཁག་བྱེ་ཁམ་བའི་ས། བ་རུ་བརྒྱུད་རྒྱ་གར་བ་འབབ་ཀློབ་ཁ་བ་དཀབ་ལས་སླ་བ་འདུག་་་་ང་ཁགབ་འདེབསྡུར་མཆན། བའི་ངཏའོ་ཁ་དང་དེ་རྗེལ་པཚོ། བ་འདུལ་བ་ལ་ར་བ་བློ་ག་ལ་བརྒྱུཎྜི་ཏག་བློ་བྷའི་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། («པེ་»སྣར་ར། «པེ་»《སྣར་པ། «པེ་»《སྣར་—མ། («པེ་»«སྣར་—འཕགས་པ། «པེ་»«སྣར་»གི། པ། «པེ་»«སྣར་»སྦྱོང་བར། །《སྣར་»པོ། («པེ་»«སྣར་»ཀྱི། བདག་(«པེ་»«སྣར་»—ཀྱི། «པེ་»སྣར་»ཏུ་བཟུང༌། «པེ་»«སྣར་པ། «པེ་»«སྣར་»ཀྱི། ས«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་ཡས། ་«པེ་»«སྣར་»བཞིན། «པེ་»«སྣར་པ། «པེ་《སྣར་རྟོགས། བྱེ«པེ་»«སྣར་»ཉ། ་་《པེ་»«སྣར་མཆོག«པེ་《སྣར་»འཛག། «པེ་»«སྣར་མ་བསྲུང་ཞིང༌། «པེ་»«སྣར་མདུན། ,«པེ་རྣམས་བསྒོམས། 《སྣར་རྣམས་བསྒོམ། «པེ་»«སྣར་»—ཀྱིས། ོ«པེ་»«སྣར་འགྱུར་བས། «པེ་»《སྣར་»ལྡན། «པེ་《སྣར་+ཕྱོགས། ཁཉེ་བ་ཟབ་མོ་བགྱི་བ་སོགས་།
