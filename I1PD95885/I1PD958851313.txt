བདུད་མ་ལངས་པ་ན་པ་བ་དུ་ལ་ཏེ་་་་་་་་་ལོངས་པ། རིག་ས་པ་དང་བུད་མེད་ཐམས་ཅད་དར་བར་འགྱུར་བ་མ་འོ། །ཁན་ཊི་ན་རྒྱ་ཁ་འགའ་ཞིག་བཅུ་དམཚོ་འཁ་ཎཱཎའི་《དཆུ། ཀན་ཏ་ཀ་། ཆ་ཁ་ཚོ་ཉིད། ར་ཎ་ཊ་། རཀྵ་ཁ་དེfthཀླུ་ཀ་པར་༄༅། །ཡིག་བརྒྱའི་ཆོ་ག་བཞུགས། ལྷག་མ་/ཡིན་ལ། རདར《*[pཀྱང་། *། ་ཚུ་། ཁ་མ་མང་དུ་དུ་བ་གླུ་མ་བསྟན་པ་ད་་་། བ་་་་/དབྱང་སུང་དང་《(། z་ལྔ་ན་རྒྱ(། །ཨོཾ་བཛྲ་སཏྭ་ས་མ་ཡ། མ་ནུ་པཱ་ལ་ཡ་བཛྲ་སཏྭ་ཏྭེ་ནོ་པ་ཏིཥྛ། དྲི་དྷོ་མེ་བྷ་བ། སུ་ཏོདྡྷོ་མེ་བྷ་བ། ཨ་ནུ་རཀྟ་མེ་བྷ་བ། སུ་པོདྡྷོ་མེ་བྷ་བ། སརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪ། སརྦ་ཀརྨ་སུ་ཙ་མེ། ཙིཏྟཾ་ཤྲཱི་ཡཀུ་རུ་ཧཱུཾ། ཧ་ཧ་ཧ་ཧ་ཧོབྷ་ག་བཱན་སརྦ་ཏ་ཐཱ་ག་ཏ། བཛྲ་མཱ་མེ་མུཉྩ། བཛཱི་ཝབྷ་བ། མ་ཧཱ་ས་མ་་ཡ་སཏྭ་ཨཱིསཱ་ཧཱ། ཨོཾ་ཧ་ཕཊ། སྟོང་དང་ལྷ་བསྐྱེད་སྤྱན་དྲང་དང་། །གདན་བཞིར་བྱིན་བརླབས་དབང་བསྐུར་བའོ། །ཞེས་གཞན་ནས་བཤད་དོ། །གསང་སྔགས་དང་རིག་སྔགས་ཐམས་ཅད་ཀྱི་བདག་པོ་ཕྱག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ༑། རིག་སྔགས་བརྒྱ་པའི་ཆོ་ག་མདོར་བསྡུས་པ་བཤད་པར་བྱའོ། །ལྟ་བའོ། གང་ཞིག་སེམས་ཅན་དོན་མཛད་དང་། །སངས་རྒྱས་སྲས་རྣམས་ཐོབ་བྱའི་ཕྱིར། །བྱང་ཆུབ་ཏུ་ནི་སེམས་བསྐྱེད་ནས། །།བདག་གིས་རིག་སྔགས་བསྒྲུབ་པར་བྱའོ། །སྡོམ་པ་གསུམ་ལ་གནས་པ་དང་། །ཤེས་བཞིན་དྲན་པར་ལྡན་པ་དང༌། །།
