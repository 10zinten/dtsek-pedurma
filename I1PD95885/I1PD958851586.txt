དུ་བྱེད་པར་བཤད། །།དབྱངས་ཅན་མའི་སྒྲུབ་ཐབས་བྲམ་ཟེ་ཕུར་བུས་མཛད་པ་རྫོགས་སོ། །མཐའ། རྟོཔ། ས་ཡི་ཆུང་འད་བྱེད་འདི་སྟོད་མེད་སྐུ་འོ་མའི་མཚོ་འདུ་ཀུ་ལུད་ཕྲེང་བ་པ་༼བྱ་ཚུ་འདུབསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དབྱངས་ཅན་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་སྟི། «པེ་»«སྣར་»གི། &«སྣར་བྱུགས། «སྣར་»ཆགས། @«པེ་»«སྣར་+འོད། ༄༅«པེ་»«སྣར་ཟེ། ཚེ«པེ་»«སྣར་»མདངས། «པེ་»«སྣར་»བྱ། ཁྱི«པེ་»«སྣར་»སྒྲུབ། །ཁྲི་ལུང་འཕྲེང་བ་དེ་འདྲ་འབྱུང་དང་། །ད་ཁྲུན་པ། འདུ་བ་དེས་སུ་བདུ་སྐྱེ་དགུའི་བསམ་པ་མ་མ་ས་མ། །འབྱུང་ཉེན་ལ་ཞི་བས་ག་བ་བསྡུར་འབྲས་རེའུ་མིག་། བག་མ་གཡེཔ་ལྟ་ན་ཊ་ཀྱི་རྒྱུད། (ཁ་ན་ཊ་བུ་འབྱེད་སྤྱོད། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། །སྡེ་ཚན། _རྒྱུད། བརྒྱུད་འགྲེལ། པོད་རྟགས། །ནདམུ། དུ་། དུངན་སྐབས་འགར་མུན་པ། ཚན་གྲངས། །མིལྡེབ། ན་ན་ན་བ་བ?བ་ན་_ནཕྲེང་། ཚིག་རྐང་། །ས་ཚེག་ཁྱིམ། །།།:རྟའ་པ་རབ་ཆུང་ནི་ཀཊ་སློབ་གྲོཊ་ཚོ་ད་ཤ་། །ཟླ་བ་བད་ཀན་དེ་ཚོ་ཁོ་ཀོ་གསོང་ཁ་བ་ཀུབ་བཊ་ཌའི་གཞུག་བཀྲ་ཤ་འཁོན་ན་ཊ་ཎི་ཀ་དག་དམགཀ་ཀུ་བྷ་བ་ཡིན་ལ་བྱ་བ་དེ་ན་།
