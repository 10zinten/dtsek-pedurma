བསྟན་འགྱུར། ་རྒྱུད། ཆུ་༽བསྔགས་ནས་གང་གི་ལག་ཏུ་བྱིན་ན་བླུན་པོ་ཡང་སྙན་ངག་མཁན་དུ་འགྱུར་རོ། །དེའི་དྲི་ཙམ་གྱིས་ཀྱང་སྦྲུལ་ལ་སོགས་པ་རྣམས་ཉེ་ནུས་པ་མ་ཡིན་ནོ། འའི་། འདི་་་རྣམས་ཀྱི་བར་དུ་གཅོད་པས་བར་ཆད་བྱེད་པ་རྣམས་:དང་གཞན་རྣམས་ཀྱི་ཡང་ཐམས་ཅད་ལ་ཕན་པར་བྱེད་ནུས་པར་འགྱུར་རོ། །བདག་གིས་མང་པོ་སྨོས་པས་ཅི་ཞིག་བྱ་སྟེ། །བེམས་པོའི་བློ་ཡང་བསྟན་བཅོས་ལ་ཉོན་མོངས་པར་བྱ་མི་་་་དགོས་ཏེ། མི་གཡོ་བའི་སེམས་ཀྱིས་སོ། །ཁ་ཁ། སྔགས་འདི་འབུམ་ཕྲག་གཉིས་་བཟླས་ན་འདིས་ཤེས་རབ་ཀྱི་སྣང་བ་སྒྲུབ་པར་བྱེད་དོ། །འདིར་ཡང་གང་:ཞིག་རྣམས་ཀྱིས་སེམས་བརྟན་པར་གྱུར་པས་ཡོན་ཏན་མཐའ་ཡས་པ་གསུངས་པ་་་་དང་ལྡན་པར་འགྱུར་རོ། །དེ་རྣམས་ཀྱིས་ཤིན་ཏུ་མཆོག་གི་དངོས་གྲུབ་ཏུ་་་འགྱུར་རོ་ཞེས་སོ། །ཤེས་རབ་ཀྱི་སྣང་བ་སྒྲུབ་པའི་ཆོ་ག་རྫོགས་སོ། །།ད་ཆ་དེ་དག་ལ་རྟོག་འབད་ན་མི་དེ་ཚོའི་ནང་དུ་བྱ་བ་ལ་ཁག་ཚོ་ཏི་ཎ་ཁ་ལ་ཡིག་ཤེས་བྱས་ན་་་་་་་་་་་་་དེ་བས་ཀྱང་གལ་ཏེ་ང་ལ་བྱ་ཚོ་ལདགའ་བ་ལ་དགག་པ་བྱེ་བསྡུར་མཆན། འཁོང་ཁཁ་བཁ་འབབ་ཆ་ཁ་གཏོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཤེས་རབ་ཀྱི་་་་སྣང་བ་སྒྲུབ་པའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་»—། «པེ་»«སྣར་»མཁན་སྲེག«པེ་»«སྣར་»ཀཾ་ཀ་«པེ་»«སྣར་»ལ། «པེ་»«སྣར་»+དེ། ཚེ«པེ་»«སྣར་»ནས། ཉི«པེ་»བ་ཀ་《སྣར་»གྱིས། ང༽་ཀུན་དང་ཁྱོད་ལ་དགའ་བར་བྱ་བ་་་་། །།:ཀ་«པེ་»«སྣར་པ། O«པེ་»«སྣར་སྒྲིབ། «པེ་»«སྣར་»—དང་། «པེ་»«སྣར་སྲུངས། «པེ་»«སྣར་»སྣ། «པེ་»«སྣར་མོ། །«པེ་《སྣར་བསྟེན། ཚེ«པེ་»སྣར་སྐུལ། «པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་སྐུལ། །་༌ཚུ«སྣར་བྷ། @«པེ་»«སྣར་»དྷརྨ། ༄༅«སྣར་བྷ། «པེ་»«སྣར་»ཞུ། ཁྱེ«པེ་»«སྣར་»བརླབ། ཀྱེ«པེ་»«སྣར་»བ། །«སྣར་གྱིས། «པེ་»«སྣར་བརླབ། །
