རྒྱུད། ཆ༽བཟླས་པར་བྱ་སྟེ། ཨོ་བཛྲ་བཱ་:རཱ་ཧཱི། ཨཱ་བེ་ཤ་ཡ། ཨཱ་བེ་ཤ་ཡ། སརྦ་:ཛ་རཱ་ཨིན་དྷིསྭཱ་ཧཱ། ཨོཾ་:དྷཱིཧཱུྃ་ཧ@སྭཱ་ཧཱ། རྟག་ཏུ་བཟླས་པར་བྱའོ། །མདོར་བསྡུས་པའི་ཕག་མོའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་ཅིང་ཞུས་སོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་ཕག་མོའི་་་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་+པ། «སྣར་»སྟེངས་སུ། «སྣར་»ལྤགས། ཨུ«སྣར་»པ། «སྣར་བསྒོམས། ཚེ«སྣར་པ། «པེ་»«སྣར་»ཨོ། «པེ་རཱ་ཧི་ཨཱ་བེ་ཤ། ཨཱ། «སྣར་ར་ཧི་ཨ་བེ་ཤཨ། ཚ«སྣར་ཨ། «པེ་»ཛཱ་རཱ། «སྣར་»ཛཱ་ར། «པེ་»དྷཱིཧཱུྃ་ཧ། «སྣར་ཧྲིཧཱུཾ་ད། @@«པེ་»ད་«སྣར་ལོཙྪ། །།རྡོ་གན་འཁྲུ་དང་། ས་རྡོ་ཎོ། ཏཥ་ཚོ་ལོ་ལ་བ། ་་་་དཀྱིལ་ལ། །མཁོzབ་དང་ysp། ཁ་ན་དེ་ལ་ཀ་གྱི་ཁྱི་བསྡུར་འབྲས་རེའུ་མིགརྗེ་ཡབ་དྲ་བ་ལ། །དོནརྒྱུད། （ད་པལ་པ་ད་དབང་ཞུ་ན། ལ། བློ་ལ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། འདུས་པ། སྣར་ཐང་། ཅོ་ནེ། །སླaཀྱི་བྱ་སྡེ་ཚན། རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། ར་ལ་༼པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང་། །ཚེག་ཁྱིམ། །ན་བ་ན་བ་། ནབན་བ་དེ་ལ་གང་། འོ། ངརརཔ་། །
