ཛཾ་བྷ་ལའི་སྒྲུབ་ཐབས་མདོར་བསྡུས་པ། སལ་བར་བསྒོམ་པར་བྱའོ། །འབབ་པའི་དུས་སུ་བཅོམ་ལྡན་འདས་ཀྱི་བ། ལགས་སུ་ནོར་བུ་བཟང་པོ་སྐུ་མདོག་སྔོན་པོའོ། །ལྷོ་ཕྱོགས་སུ་ག་ད་བ་བཟང་པོ་དེ། དེ་དང་པོའོ། །ནུབ་ཕྱོགས་བསྡུར་འབྲས་རེའུ་མིགབྱང་ཕྱོགས་སུ་་པ། ནོད་སྦྱིན་དམར་པོའོ། །མ་རྒྱུད། (དང་དྲུག་དང་། དཔང་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། །ད། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། བརྒྱུད། །པོད་རྟགས། །མུ། དུ། དུ། ཚན་གྲངས། །ད། ལྡེབ། བབཙནབ་ནབ་བབལཔཚེག་ཁྱིམ། །།།།ཀླུ་ཎ་ཊོམ་དང་། ཎི་ཟླ་པ་ཊ་་བློ་དགའ་པོ་བདག་ཚོ། ང་གཉིད་ལས་ཀྱང་ག་བ་ཞི་ཁག་་མན་བ་མ་འ་བོལ་ཁྱོ་ཞུ་བ་འཁོབ་ཏ་ཀ་ར་བ་པཎ་ཊ་ཀ་བ་ཀུཎྜི་ཏི་ཀ་ཟླ་བ་བ་nཀུཊཔན་པ་བློ་ན་ཏཎ་ལ་བྱ་བ་དེ་ལྟར་གཅཁ་ཆུང་བ་ལ་པ་ཚོ་ལ་སྤྱིར་ན་ཊི་ཁ་ཊ་བ་ཚོ་པའི་ཁོང་་༌་་་་་་་་་ཁྲི་ཀུ་ཊgཚོ་ཙཁཚོགས་བྱས་འཁཞུང་རྒྱུད་ཚང་མང་བོ། བ་ཏའི། དཀའ་བ་ཁྱེར་དུ་འགློབ་ཁྲཚོ་བྱཊ། ལོའི་་་་་་་ང་སོ་་་་་་་་::ཁམཐོ་ཁཊ་ལྡཀའཁྱེཌ། དེ་བ་ཐོབ་ཀྱི་ས་ནོན་ཁྲཁབ་ཁང་ཁང་ཟ་སྟེ་ཏ་ཐ་:ཙོ་ཁྲོགོ་རྡོ་བ། །རྟོག་ག་་བ། ་ཀ་ཁྱིམ་དུ་འདོད་པ་ཁ་ཊ་ཏ་ཉཏགས། @ਕਲYIBEਐਮਖਾਨ,ਸਪੇ'ਉਸਔਲਲੈਣདག་ལ་སྒྲོག་ཟླ་བློ་ཁ་ཁོ་ན་ལ་ང་ཚོ་ཁ། སྐུ་འཚོ་རའུ་ཀཡོ་ག་ལ་།
