བསྟན་འགྱུར། རྒྱུད། ཁ༽བཟླས་པ་བྱའོ། །བརྒྱ་བཟླས་པས་ཐུན་མོང་གི་ྋསྐྱེས་པ་དང་བུད་མེད་ཀྱི་ཚོགས་དང་། རྒྱལ་པོ་དང་། བློན་པོ་དབང་དུ་འགྱུར་རོ། ྋ། ཁྲི་བཟླས་ན་འཇིག་་་་རྟེན་ཐ་མལ་པ་རྣམས་སོ། །།འབུམ་ཕྲག་གིས་རྒྱལ་པོའོ། །འབུམ་ཕྲག་བདུན་གྱིས་ལྷ་དང་ལྷ་མ་ཡིན་གྱི་བུ་མོའོས། །བྱེ་བས་ནི་བྱ་ཁྱུང་དང་གླང་པོ་ཆེ་ལ་སོགས་་པའི་བཟང་པོའི་ཕྱུགས་དང་ཐུན་མོང་གི་ཕྱུགས་རྣམས་ཀྱང་ངོ་། །གུས་པས་་རྒྱུན་མ་ཆད་པར་དུས་རིང་པོར་དཔག་ཏུ་མེད་པ་བཟླས་ན་ས་གསུམ་པོ་ྋདབང་དུ་བྱེད་དོ། །ལྷ་མོ་རིགས་བྱེད་མའི་སྒྲུབ་པའི་ཐབས་རྫོགས་སོ། །ཌཎ་ཌ། ོ་ཀྱི་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་་་་བསྒྱུར་བའོ། །ཀ་ཁ། །ག་ད་བ་ཀ་ཊཀ་ཋ་ཁ་བ་ཀྱPཙཁ་ཁཔ་དང་ཁཀྱང་ར་་ཁgཁ་ཀུ་ཏ་རའི་འཚོ་སྐྱ་པའི་ཁང་ཀ་ཀྐཎཁ་ཏུ་ppལྔ་ਨਸਲਦੇਇਸਪੋਸਤਸਰੀਰਨੂੰਰੋਲབསྡུར་མཆན། Ngoàira,phacàམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «/«པེ་»༄༅། །ལྷ་མོ་རིགས་་་བྱེད་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །@«སྣར་»དཱ། «སྣར་རིག«སྣར་བྲི། «སྣར་ཨུད་པ། «པེ་»«སྣར་པོ། ཉི«པེ་»གིས་བོན། 《སྣར་»དྭིས་བོན། «སྣར་»རིགཁྱེ«པེ་»«སྣར་»+དམར་བ་པདྨ། «པེ་»«སྣར་»+དང་། //«པེ་»«སྣར་དང༌། 《ཁྱེ«སྣར་བའོ། །ཆུཚེ«པེ་»+སཾ། «པེ་»«སྣར་»མདེའུ་ཡིས། ཚེ«སྣར་»ན། ཁྱ«སྣར་ཨུད་པ། «པེ་པ་་་་བལྟས་ན། «སྣར་པ། བལྟས་པ། ཚེ«སྣར་ཧྲི། @«སྣར་»ཅིག@«པེ་སྔོན། ཁྱིམ«སྣར་གྱིས། «སྣར་»ཅིག་ཚ«པེ་»«སྣར་»གི། _«པེ་»«སྣར་»སྒྲུབ། «སྣར་གཞགཨེ«སྣར་ཨུད་པ། ཀྱེ«སྣར་རིམ། ལེ«སྣར་ཀྱི་དམར་པོར་གནན། འབུ«སྣར་»ཀྱི། ཁྱེ«པེ་ཨུཏྤ་ལའི་ཞབས། «སྣར་ཨུད་པ་ལའི་ཞབས་པས། ཚེ«པེ་»«སྣར་+པ། ཨེ《སྣར་+པ། ༄༅«པེ་»«སྣར་བུ། ཏེ《སྣར་དབུགས། །འཚེ«པེ་«སྣར་»+ནི། མགྲོན་གྱི«སྣར་གནས། གྱེ«པེ་»«སྣར་ཨ། ཉེ《སྣར་»ཧྲི། ་སུ«སྣར་སྒྲུབ་བྱ་ཁུགམེ«པེ་«སྣར་བསྟན། «སྣར་»གིས། @«པེ་»གི། ཚེ«པེ་»«སྣར་མོ། «པེ་»«སྣར་པོའོ། །«སྣར་»གིས། །
