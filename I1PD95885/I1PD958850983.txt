རྒྱུད། ཆུ༽པ་ལ་རྡོ་རྗེ་དབྱངས་ཅན་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།ས། །མ་ཎཊ། བ་དང་། དེ་དག་ཀྱང་གང་ན་ཡི་བ་ཡང་། བྱ་བ་བྱེད་ཁ་ལོ་བ་ཞུ་དུས། འོ་ན་བ་དང་ར་ལུཏཚ། ་་་་་་་། འབའ་དོན་གབསྡུར་མཆན། ཁ་ན་ཏའི་རྒྱུད་པས་དེ་ཡང་གལ། ཚེ་དབང་། །དུ་བས་/T། ཁོ། བཀ::ཀྱིས་ཁ་ཚོན་ངོ་ལ་བབ་ཁྱད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་»ཏ་ཡི། ︽«པེ་»«སྣར་བརྟན། ཚེ«པེ་»«སྣར་རྩེ། ཚེ«པེ་»«སྣར་»སྙིང་། /ེ«སྣར་»ཆུས། ®«སྣར་»སྟིང་། ཚེ«པེ་»«སྣར་ཡིས། པའིས«པེ་»《སྣར་གོམ། «པེ་»«སྣར་»སྟེགས། «པེ་»«སྣར་འཕྲེང་། «པེ་»«སྣར་རྣམས། །«པེ་»«སྣར་པ། @«པེ་»«སྣར་བཀླག«པེ་»«སྣར་»བརྩོམ། «པེ་»«སྣར་བསྟན། «པེ་»«སྣར་བསྲུང་། «པེ་»«སྣར་»སྨིན། ཁྱེ«པེ་»«སྣར་»གཏམ། /«པེ་»ཡིས། @«པེ་»«སྣར་རྒྱལ་བར། །ནོ«པེ་»«སྣར་བར། ཚེ«སྣར་»ལྟར། ཚེ«པེ་»«སྣར་ལས། @«པེ་»«སྣར་»སྔགས། @«པེ་《སྣར་»བཅས། _«པེ་»«སྣར་»མ། «པེ་»«སྣར་ནི། །ཚེ«པེ་»«སྣར་»ལ། །འདུས། ཀྱེ«སྣར་ཀྱིས། ཉེ«པེ་བདྷ། «སྣར་»བརྡ། ཁ་ལོ«པེ་»«སྣར་བུདྷི་བརྙ། «པེ་«སྣར་ཀྱི། །ཆཚན་ཉིད་དང་། མཁས་པ་ਨਾਓਹਲਾਫ਼ਨਸਲਇਸਸਮੇਂਤਕਕੋਮਲཔ་དང་ཀུན་བསྡུར་འབྲས་རེའུ་མིག་དང་། ཆེ་འདི་ཊ་འདུགད་པ་ན་དང་ན་བ་མ་འ་རྒྱུད། （རྒྱལ་བ་ལ་ཚ་བ་དང་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། །ཙ་ན། བ་བ་ན་བ་དང་སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། ཕ་རྒྱུད། a/ཀྱི་རྒྱུད། གདུགས་བོད་རྟགས། །མུ་། ་ད་དུང་པ་དུ། ཀུད་པ་དམུ་ར:༼འོ། །ཚན་གྲངས། ་ལྡེབ། བ*ན་བ་བ?ནབབཙབ་ཕྲེང་། །།།L། ཚེག་ཁྱིམ། །།
