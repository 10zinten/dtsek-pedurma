ལ་ཁྲགེ་སློང་ཁྱཊ་བ་སྟེ་ཟླ་བ་ག་ཚོད་ཀྱང་། བདག་དཔལ་འབདེཀ་ཁ་ཁབ་དང་། ཁོའི་གཡོན་དབྱེ་བ་ཆགས་ཀྱང་བདག་ལ་་་་་་་ཀེ་ལ་ཤོར་བ་ལ་ཐོགས་པ་དེ་ལ་དབང་ཀུན་ད་དུ་བྱ་བ་ཞེས་པ་ལ་་་་་་ཁོང་ལ་ཁྲི་ལ་ཁའི་ཕྱོགས་སུ་འབབ་པ་ཞིག་ལ་ཀRཁ་བྷ་ཞོ་ཆུ་ཚོ་སྐྱེ་བོ་ཁ་ཕྱོགས་བག་མ་ཁབ་མི་འགལ་བ་ག་བ་ག་ཟླ་བ་ཟླ་བ་ཚོ། མཚོ། ༄༅། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་ག་དེ་ཁ་པའི་སྒྲུབ་ཐབས། འཚེ་དང་ཉེ་ཁསྤྱི་མཆན་འཚོན་། སྨྲ་བ་བཅད་མ་ག་ག་ནག་པོ། རྒྱ་ལུ་::ཀོད་འདུག་ཤེས་ཚོཏང་བ་བགོ་བ་ན་དཀའ་ཞིང་ཁྱེད་ལ་བྱ་བ་ཁབ་ཏུ་བསུ་ཁྲོ་ཆུ་འཁྱབ་ཁཚོ་ཏའི་ས་ཁལེགས་བ་་བློན་པོ་ཆེ་བོ། །ཁྲོ་བ་སྐྱོ་ཉེ་ལྗཁཀུ་བྱི་དང་། བྱེ་བའི་བྱེད་ན་ཁ་མི་nཔ་ལ་ཚ་བ་ན་ཚོང་བ་དེ་༄༅། །རྒྱ་གར་སྐད་དུ། དུ་པྲཛྙཱ་པཱ་ར་མི་ཏཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྒྲུབ་ཐབས། བཅོམ་ལྡན་འདས་མ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་སེལ་ཕྱག་འཚལ་ལོ། །ཤེས་རབ་ཕ་རོལ་ཕྱིན་གཉིས་མེད། །ཡེ་ཤེས་དེ་ནི་དེ་བཞིན་གཤེགས། །བསྒྲུབ་བྱའི་དོན་དེ་སྦྱར་བ་ཡིས། །གཞུང་་ལམ་དག་ལ་དེ་ཡི་སྒྲ། །བྱང་ཆུབ་ས་བོན་བླ་ན་མེད། །ཤེས་རབ་ཕ་རོལ་ཕྱིན་་བཏུད་ནས། །གཉིས་མེད་ཡེ་ཤེས་བསྒྲུབ་པའི་ཕྱིར། །དེ་ཡི་སྒྲུབ་ཐབས་བྲི་བར་བྱ། Paz། དང་པོ་རེ་ཞིག་སྔགས་པས་མཉམ་པར་གཞག་པར་གྱུར་པས། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མའི་སྒྲུབ་པ་ཡིད་ལ་བྱས་ཏེ། ཙནྡན་གྱི་མཎྜལ་བྱས་ནས་་དེའི་ཡན་ལག་ངོ་བོ་ཉིད་མེད་པའི་རང་བཞིན་གྱིས་དེ་ལ་དབུས་སུ་མི་བསྐྱོད་པ་་་་བལྟས་ནས་མེ་ཏོག་དབུལ་བར་བྱའོ། །སྔགས་འདིས་སོ། །ཨོཾ་ན་མོ་ཨཀྵ་བྷ་ཧཱུྃ་ཨོཾ་བཛྲ་པུདྨེ་ཧཱུྃ་སྭཱ་ཧཱ། དེ་བཞིན་དུ་ཨཀྵ་བྷུ་ྋཧཱུྃ་ཞེས་པའི་ས་བོན་གྱི་ཡི་གེ་བརྗོད་།
