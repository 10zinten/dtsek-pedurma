ཀ་ཀ་ནི་ཀ་ཀོ་ཊ་ལ་སྐད་ཀྱི་ཁ་ཁོརྩེ་འཛིན་ཡར་ཆེད་དུ་བྱེད་ན། །རྒྱུ་ਸਮਾਧਦੀਸਿਆਸੀਪਾਰਟੀਕੋਸਾਇਟਲੀਲਉਦੇਓཁོ་གི་ནང་ཁ་ནRམི་ཁས་བ་བོད་ཀྱི་དང་བ་དབང་པོ་དཔལ། ངར༄༅། །རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས། གང་། །ཁ་འཚེ་བུ་ཚེ་ཚད་ཀའི་བྲས་ཐོབ་ཀྱང་། །དུམ་ཏེ་ཁྱེད་ཆ་བོས་བ་ན་ལག་པ་ཁྱིལ་ਸਨੂੰਰਸੋਈਅਬਲਾਈਨਲਬੈਠੇਸੀ::ਇ༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་ཏཱ་རཱ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས། འཕགས་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །སྐུ་མདོག་པ་དཀར་མོ་ཡིད་འོང་མ། །རྡོ་རྗེ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ། །སེམས་ཅན་ཀུན་གྱི་་་༌་དོན་རྒྱུའི་ཕྱིར། །དེ་ཡི་སྒྲུབ་ཐབས་བཤད་པར་བྱ། །འདིར་ལྷག་པར་བསྒོམ་པའི་རྣལ་འབྱོར་པས། །རང་གི་སྙིང་གར་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་ཏཱཾ་ཡིག་་་་བསམས་ལ་ཞལ་བསང་བ་ལ་སོགས་པ་བྱའོ། །ོ། ཤིན་ཏུ་བདེ་བའི་སྟན་ལ་ཉེ་བར་འདུག་སྟེ། རང་གི་སྙིང་གའི་ས་བོན་ལས་འོད་ཟེར་འཕྲོས་ཏེ། འོད་ཟེར་དེས་རྗེ་བཙུན་སྒྲོལ་མ་གདན་དྲངས་ཏེ། །མདུན་གྱི་ནམ་མཁའ་ལ་བཞུགས་པའི་བླ་མ་དང༌། ལུ་སངས་རྒྱས་དང༌། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་ཡང་མདུན་དུ་་་སྡིག་པ་བཤགས་པ་ལ་སོགས་པ་བྱས་ཏེ། སྡིག་པ་བཤགས་ཤིང་མི་བྱེད་པར་སྡོམ་པ་དང་། རྗེས་སུ་ཡི་རང་བ་དང་། ཡོངས་སུ་བསྔོ་བ་དང་། གསུམ་ལ་་་་།
