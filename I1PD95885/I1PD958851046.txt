བབློམ་པར་བྱའོ། །ཤར་གྱི་ལ་དབ་མ་རབ་ཏུ་དང་བའི་སྒྲོལ་མའོ། དེ་འདབ་མར་རྫོགས་པའི་སྒྲོལ་མའོ། ནུབ་ཀྱི་འདབ་མས་སྤྱལ་བའི་ཕྱོགས་ལ་མའོ། །བྱང་གི་འདབ་མར་ཞུ་བ་སྒྲོལ་བསྡུར་མཆན། ་ལྡན་འདབ་མ་ལྟ་འོ། །ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཉི«སྣར་བསམས་ཤིང་། «སྣར་»རིག་«སྣར་བཀྲུང༌། བསམ་:«སྣར་»གི། «པེ་»«སྣར་པ། «པེ་»«སྣར་ཟེ། ལུ་«སྣར་»ལ། @«པེ་ཕུགས་ཏེ། །«པེ་»«སྣར་མྱོས། ༄༅«སྣར་གྱུར། ཉི«པེ་རིགལ་བས་་་་་་གདན་ས་མེད་པ། ་བ། མ་ན་སྤྱི་བ་རྒྱུ་དང་། མེད་སྙམ་གྱི་ཚུ་རལ་ངས་བའོ། །ལུག་པ་ལོ་བྱེད་པ་ད་པ། ་པ་དེས། ལ་དང་། བགཞུ་བསྣམས་པའོ། བ་བློ་བརྗེ་དང་མ་འབྲེག་པ་དང་བསྡུར་འབྲས་རེའུ་མིགདེ་ལ་ལྟོ་བ་འཕྱིང་བ་སྐུ་མདོབ། །ཀྱི་རྒྱུད། (པ་དབུ་བགྱི། །མ་རྒྱུད། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། པོད་རྟགས། མུ། ཚན་གྲངས། །བར་ལe། །ལྡེབ། བ་ན། ན་བ༴། ནབ?། བ་ནཚེག་ཁྱིམ། བ། ་ཀྱིས་། བྱ་:ཀ་ཀ་ཀུན་:ཇི་ད་པ་དང་། བརྡ་ཁ་ཟས་སྐོཁ་ཞུ་རྒྱན་ཆཟླ་ཎཐུ་བ་དང་། ཁང་པའི་ཀ་ལ་ཎཊཚོད་བཞི་ཆ་ངར་རིང་དུ་ནད་དུ། །མཎཎཁ་གཏདང་། རང་ཀའ:ཁྱབཝང་ཁཡུལ་དུ་ལྷ་ཚོགས་རན་ལ་འགྲོ་ལྟ་རྒྱུ་བ་བ་འཚོ། །སྐུ་བཞུགས་ཀྱི་སྐད་བྱེ་བ་ག་ཎ་ག་ད་། ཅཎྜག་པ། མགར་གྱི་ཕྱག་སྟེ་མ་ཞི་།
