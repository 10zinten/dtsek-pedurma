རཚུབ་ཐོབ་ཀྱང་། །ཀཚོལ་ཚོ། ཀོ་ས་བྱ་། ཁང་ཤ་ཁ་པ། །དཀའ་བ་མེད་ཀིཏཚོ་ཆུ་བོ་ཀ། །།ཀར་པོ་འབབ་གྲུབ་ཀུན་དུ་བླ་བྲང་དེ་དགེ་ཀའོ། །ནགས་ཁག་ལ་ས་ཀླ་ཀྱིས། ཁོའི་རྒྱ་ནག་ཏུ་ཁ་འགྱུར་བཞ། ཁང་པས་བྲི་བྱ་ག་ཚོང་ལ་ཁ་བཀའ་༄༅། །མཚན་ཡང་དག་པར་བརྗོད་པའི་མན་ངག་ཤེས་བས་བདག་ཀློག་གནང་། རབ་འཁོར་ལོའི་དངོས་གྲུབ་མཆོག་ཟ་ཁ་ད་གི་ཀ་བ་ཁ་པཎ་ཐོབ་པའི་སྒྲུབ་ཐབས། ་ཚུལ་བས་ཁོ་ལ་བ་སྟེ་ཐལ་ཁོ་ན་ཀླུ་རྒྱུ་བ@། དཀག་འདུ་ཁད་ཀའི་འགལ་བ་འ་ཀ་ཁ་བ་ཚེ་ཀ་དོ་ཁ་རྩ་gའི་ཏེ་འ་བ་སྤྱི་ཁང་། །ཁ་ཊ་ཀྱི་བཤུ་བྱཚོ་གླུདག་དང་ཁོ་པའི་ལུང་བའི་བབ་དང་བརྗེ་རྒྱུའི་འདུན་ལ། །ཁ་དnp། རྗེ། ཟླ་། །དང་པ་ཁ་༄༅། །རྒྱ་གར་སྐད་དུ། ནཱ་མ་སཾ་གི་ཏྭ་མ་ན་ཡེ་ན་སིདྷི་པྲཛྙཱ་ཙཀྲ་བ་ལ་བ་དྷཾ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། མཚན་ཡང་དག་པར་བརྗོད་པའི་མན་་་་ངག་ཤེས་རབ་འཁོར་ལོའི་དངོས་གྲུབ་མཆོག་ཐོབ་པའི་སྒྲུབ་ཐབས། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །བླ་མའི་ཞབས་ལ་བདག་ཕྱག་འཚལ། །རྡོ་རྗེ་སློབ་དཔོན་དེ་བཞིན་དུ། །ངག་གི་དབང་པོ་ཕྱག་འཚལ་ནས། །འཇམ་དཔལ་སྒྲུབ་ཐབས་བདག་གིས་བྲི། །སེམས་དཔའི་སྐྱིལ་ཀྲུང་གིས་བཞུགས་ཏེ། ཁྲོ། སྙིང་གར་ཟླ་བ་ཨ་ཡིས་བརྒྱན། ཁྲོ་། ཀུནྡ་དུང་འདྲ་ས་བོན་འབར། །སྣ་ཚོགས་འོད་་་ཟེར་ཀུན་དུ་འཁྲིགས། །དེ་སྐྱེས་རྫོགས་པའི་སངས་རྒྱས་དང་། །བླ་མ་བྱང་་་་ཆུབ་སེམས་དཔའ་དང་། །ཆེན་པོ་དེ་རྣམས་བལྟས་ནས་ནི། །སྔགས་འདི་ཡིས་ནི་མཆོད་པར་བྱ། །ཨོཾ་བཛྲ་པུཔྤེ་ཧཱུྃ། ཨོཾ་བཛྲ་དྷུ་པེ་ཧཱུྃ། ཨོ་བཛྲ་དཱི་པེ་ཧཱུྃ། །ཨོཾ་བཛྲ་གནྡྷེ་ཧཱུཾ། ཌཱ་ཨོ་བཛྲ་ནི་བི་དྱ་ཧཱུཾ། དེ་ནས། །དཀོན་མཆོག་གསུམ་ལ་་་བ་ལ།
