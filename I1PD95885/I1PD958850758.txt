ཋولمیམོ་རཏྣ་ཏྲ་ཡཱ་ཡ། ན་མ་:ཨཱརྱ་ཝ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ། བོ་དྷི་སཏྭཱ་ཡ། མ་ཧཱ་སཏྭཱ་ཡ། མ་ཧཱ་ཀཱ་རུ་ཎི་ཀཱ་ཡ་ཏདྱ་ཐཱ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སརྦ་དུཥྛན་་མ་མ་ཀྲྀ་ཏེ་ཛཾ་བྷ་ཡ། སྟམྦྷ་ཡ། མོ་ཧ་ཡ། བནྡྷ་ཡ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ་སརྦ་་དུཥྛན་སྟམྦྷ་ནི་ཏཱ་རེ་སྭཱ་ཧཱ། །ཡི་གེ་བཅུའི་ཆོ་གའི་སྔགས་འདི་གོས་ཀྱི་སྣ་ལ་ལན་་་་བདུན་དུ་མངོན་པར་བསྔགས་ནས་མདུད་པ་བྱས་ན། འཇིགས་པ་དང་བཅས་པའི་ས་གང་དུ་འགྲོ་ཡང་འཇིགས་པར་མི་འགྱུར་རོ། །སྟག་དང༌། ཆོམ་པོ་དང་། ཆུ་སྲིན་དང༌། །སེང་གེ་དང་། *སྦྲུལ་དང༌། མ་ཧེ་དང་། དོམ་དང་། དྲེད་ལ་སོགས་པ་མིང་དྲན་པ་ཙམ་གྱིས་ཞི་ཞིང་འཇིགས་པར་མི་འགྱུར་རོ། །ཀ། སྔགས་འདིས་ཨུཏྤ་ལའི་སྦྱིན་སྲེག་བརྒྱ་རྩ་བརྒྱད་བྱས་ཏེ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་་ཨ་མུ་ཀཾ་མེ་བ་ས་:མ་ནཱ'ཡ་སྭཱ་ཧཱ། ཌ་ཡང་སྔགས་དེས་དེ་བཞིན་དུ་བྱ་རོག་གི་སྒོ་ལ་ལན་སུམ་ཅུ་རྩ་གཉིས་ཡོངས་སུ་བཟླས་ཏེ་དགྲའི་ཁྱིམ་དུ་སྦས་ན་ཞག་བདུན་ན་སྐྱོད་པར་འགྱུར་རོ། །འོ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་ཙ་ལ་:པྲ་ཙ་ལ་ཤཱི་གྷཾ་གཱ་མི་ནི་དེ་བ་:དཏྟ་ཨུཙྪཱ་ཊ་ཡ་ཧཱུྃ་ཕཊ། དེ་ནས་གཞན་ལ་ཡང་:རབ་ཏུ་སྦྱོར་བར་འགྱུར་ཏེ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་ཨ་མུ་ཀཱ་བྷི་དྷཱ་ནཾ་ཀུ་མཱ་རཱི་མ་:ཧཱ་མུད་བཱ་ཧེ་ན་ཏ་མཿཔི་ཏཱ་པྲ་་ཡཙྪནྟུ་སྭཱ་ཧཱ། ཌམྱོས་བྱེད་གཏུམ་པོའི་ས་བོན་དང་། ་ཏེ། མྱོས་བྱེད་འབྲས་བུ་དེ་བཞིན་དུ། །མྱ་ངན་མེད་ཤིང་མེ་ཏོག་བཅས། །སྦྱིན་སྲེག་སྟོང་ཕྲག་ལྔ་བྱས་ཏེ། །མར་སྦྲང་བུ་རམ་སྦྱིན་སྲེག་ནི། །བུ་མོ་སྒྲུབ་ལ་མཆོག་ཏུ་བསྔགས། །དེ་ནས་ཞག་བདུན་སྦྱོར་བ་ཡིས། །རང་འདོད་བུ་མོ་ཐོབ་པར་འགྱུར། །དེ་ནས་གཞན་ལ་ཡང་རབ་ཏུ་སྦྱོར་བར་འགྱུར་ཏེ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་ཨ་མུ་ཀཱ་:བྷི་དྷ་ནཾ་སུམྦྷ་ན་ཏ་ཨཱ་ཀརྟ་ཡ་མ་མཱནྟི་ཀེ་ཛསྔགས་འདིས་རག་ཆུང་གི་ཚེར་མ་རང་བྱུང་གི་་་མེ་ཏོག་ལ་བསྒུགས་ཏེ་སྦྱིན་སྲེག་སྟོང་ཕྲག་ལྔ་བྱས་ན་འདིས་རྒྱལ་པོའི་བུ་མོ་ཡང་་་་I།
