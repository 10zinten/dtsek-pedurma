སུ་པོ༌ནས་རབ་ས་། །བྱ་བ་ས་བོན་ལས་བྱུང་བའི་ཨུཏྤ་ལ་དམར་་དང་། ་བཞིན་པ་ག་ceཆུང་། སྤྱི་ལ་ཐུག་'«p:ਮਰੋਂਦੇਲਲਾਨਰੰਗਸਵਾਨਰੇਸ਼ਉੱਨཁོདང་། ་ཡ་eig་བ་༄༅། །རིགས་བྱེད་མ་དཀར་མོའི་སྒྲུབ་ཐབས། ཀེ་པਲਈབློ་་ཏe*ང་། p/ང་། ཆོས་ལ་དགའ་དང་ད་དུང་ང་བའི་ཟླ་ར་བའི་ནང་དག་པ་བ་ཞིག་ལ་། །ཧྥ་ཊི་ལ་ཁ་ཅ་Fijཐ་ལའི་ཁཚེ། །ཏོ་ཏgཚོཔའི་དབྱེ་བ་ཕྱེས་ས་བླངས་རུང་ཁྱབ་ཅིང་འཛབ། ་པས། ན་༄༅། །རྒྱ་གར་སྐད་དུ། ལཤུཀླུ་ཀུ་རུ་ཀུལླེ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རིགས་བྱེད་མ་དཀར་མོའི་སྒྲུབ་ཐབས། རིགས་བྱེད་མ་ལ་ཕྱག་འཚལ་ནས། །བྱིས་པ་རྣམས་ཀྱི་བློ་འཕེལ་ཕྱིར། །བླ་མའི་ཞལ་གྱི་སྐུ་དྲིན་གྱིས། །བསྒོམ་པའི་རིམ་པ་བརྗོད་པར་བྱ། །དང་པོར་སྔགས་པས་བདག་དང་སེམས་ཅན་ཐམས་་ཅད་སྡུག་བསྔལ་བ་རྣམས་རྣམ་པར་གྲོལ་བར་འདོད་པས་སྔ་དྲོའི་ཐུན་ལ་ལངས་་ནས་གདོང་བཀྲུ་བ་ལ་སོགས་པ་བྱས་ནས། ་བཡིད་དང་མཐུན་པའི་གནས་སུ་་སྟན་བདེ་བ་ལ་འདུག་སྟེ། བསམ་པ་ཐམས་ཅད་སྔ་བར་བྱས་ནས་བདག་ཉིད་རྣལ་འབྱོར་ལ་གནས་ནས་ཞལ་བསིལ་གྱིས་དེ་ཉིད་གསུམ་གྱི་སྔགས་ཀྱིས་སྲུང་་བ་་བྱས་ནས། རང་གི་སྙིང་གར་རི་བོང་འཛིན་པའི་དཀྱིལ་འཁོར་དཀར་པོ་བསྒོམ་་པར་བྱའོ། །དེའི་སྟེང་དུ་རང་གི་ས་བོན་ལས་འོད་ཟེར་འཕྲོས་ཏེ། འོད་ཟེར་་་དེ་རྣམས་:ཀྱིས་ཆོས་ཀྱི་དབྱིངས་ནས་བཅོམ་ལྡན་འདས་མ་སྤྱན་དྲངས་ཏེ། །
