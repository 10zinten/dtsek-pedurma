།བསྟན་འགྱུར། རྒྱུད། མུ::ཀྱི། ད་དུང་་བ་་་་་ཁ་བའི་བཏོན་གནང་། བལ་པོ་ཀ་ཁ་ག་་ཆ་་ཁེ་ལོ་འབུ་རྒྱ་མཚོབ་དང་ན་བ་མ་འ་ར་ལ་ས། །བྱེར་ཁ་པ། ཁ་ཏ་ཀ་སུ་མཚོ་བ་འགན་འོབསྡུར་མཆན། ཀླུ་བས་ཁོ་ཊོན་ལག་ཁང་། ད་དུང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་མ་་་་གཙུག་ཏོར་རྣམ་པར་རྒྱལ་མའི་སྒྲུབ་ཐབས་རིམ་པ་བཞུགས་སོ། །«པེ་»«སྣར་དྡྷཾ། འ«པེ་»«སྣར་»«པེ་»«སྣར་»གྷཾ། «པེ་»«སྣར་ནང་ན། «པེ་»«སྣར་»+གསུམ། ︽«པེ་»«སྣར་»—པ། «སྣར་དབུས་རྒྱན། _@«པེ་»«སྣར་»—སྐུ་མདོག་དཀར་པོ། ཨ«པེ་»«སྣར་བས། །་(QQ«པེ་»«སྣར་ནི། «པེ་»«སྣར་པ། «པེ་»སྣར་»དྡྷཾ། «སྣར་_སོ། @«པེ་«སྣར་མྲྀ་ཏ་ཨ། «པེ་»«སྣར་ཡཾ་ཏཱ་ནཱཾ་བྷ་བ་ཡ་ཤྲཱི། «པེ་»«སྣར་»ཀཱནྟ་ཡ་མྲ། ཚེ«པེ་«སྣར་ཡ་ཀྲྀ་ཏ་ག་མི་རྒྱ། ཀྱེ«པེ་»«སྣར་ཡ་ཀཱ་རཾ་ཡེ། ་་ར་ཏཁྲུགས་པ་ཀུན་བཏགས་དུན་པ་བསྡུར་འབྲས་རེའུ་མིག་ཁྲི་སུ་འོས་གར་འཁཔ་བརྒྱུད། （ལག་ཁྲ་ཞིག་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ས་ཅོ་ནེ། །།སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། །བ?ན*བརབན་ནབ?ན་ཕྲེང་། ,[ཚེག་ཁྱིམ། zའིལ་པ་ལ་ཡན་པ་དུ་མ་པ། ཞབས་གཡོན་བརྐྱང་བའི་དམག་གི་པགས་པའི་ན་བ་དང་། དབུ་སྐྲ་གྱེན་དུ་མ་ལས་པ། ཀླུ་བརྒྱད་ཀྱིས་བརྒྱན་པ། བདག་།
