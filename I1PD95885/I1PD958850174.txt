ལ་ཏུ་སྔོན་པོ། ཁ་ཁྱིལ། ་པ། ས་ནི་ཉི་མ་ལ་གནས་པའི་་བཟུང་ལ། དེའི་ནད་ཟེར་གྱིས་བགེགས་རྣམས་བསྐྲད་དེ། སེམས་བན་གྱི་དོན་བྱེད་པ་ལ་སོགས་པའི་ཁྲོ་བོ་གཤིན་རྗེ་གཤེད་ཀྱི་རྣམ་བསྡུར་མཆན། སྐད་ཅིག་གིས་དེ་ཡོངས་སུ་དུན་དུ་རབ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»«སྣར་པོར། «པེ་»«སྣར་»ནཨོཾ། «པེ་»«སྣར་»པས། «པེ་»«སྣར་_ཉ། «པེ་»«སྣར་ཡི་གེ་@«པེ་»«སྣར་»བ། ཧ«པེ་»«སྣར་+ཤན«པེ་»ཀྱིས། «པེ་+སུ། ཚུ«པེ་»ཧཱིནྡྲིབི་ཀྲི། «སྣར་»ཧྲིཉྩིབི་ཀྲྀ། ཨི«པེ་»«སྣར་ཙ། རལ་གྲི་དང་བ་ཤིང་བསམས་པའོ། ལ་ཡོན་གྱི་ལགསུམ་ན་དེ་ཏ་ལི་དང་། དགྲ་སྟ་དང་། །བ། སུ་པ་ཀུ་ར་མ་པའོ། །ཁ་ཅིག་གིས་ད་། ཚུལུ་སྤུའདྲ་སྤྱི་བསྡུར་འབྲས་རེའུ་མིག་དེ་མུར་པའོ། རྒྱུད། (གཡོན་གྱི་ཕྱག་ལ་དྲིལ་བ་བློ་རྗེ་ཏ་ཀ། །སྐྱ་པ། །དཔེ་གཞི། །།སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། །གསྤུ་སྡེ་ཚན། :ངལ་རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། མན་རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བནབན། བ་ན། །བན་ཕྲེང་། ན་པ་། །ཚེག་ཁྱིམ། །ཏོ''། ཀ་དག་པའི་ཀ་བ་ལྟར་ཀྱི་ཀ་ཐོགས་ཏེ་ཞི་དྲག་ཀཁྲུང་ཁག་ཚོ་བ་དང་ཁོ་ཚོ་ནི་ང་ན་མ་ལ་བསྟོད། ཁོ་ཀཚོ་ཕྱིཥཎི་ཏབ་རྒྱུ་གྲ་སྒྲཊ་ཏན་ཅཎྜ་ཀ་ཁ་ག་བརྗོད་ན་ཚེ། ངོ་ཤེས་སྒྲོག་ཁ་སྒོ་ལྟ་བྱེ་བ་ཚེ། །ཏཁན་ཁ་ཁང་ཁ་པ༽དབཀོད་ཁྱི། ཉི་ཟླ་ཁ་ག་ཀ་ཀ་ཎ་ཌོ། །།དུ་མ་ལ་།
