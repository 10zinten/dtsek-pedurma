སོབ་དག་པའི་རྒྱུད་རབ་ཏུ་འབབ་པ་དངཉེ་བར་དུ་འདུ་བསྡུས་པ་བྱ་བ་དེ་རྒྱ་། དེ་ན་ས་ཕྱག་རྒྱ་བཅིང་བར་དྲ་ཡི། །ལས་དེ་དག་པ་དེ་དབུ་མའི་ཉེབ། ས་ག་དུ་མ་བསྡུར་མཆན། །པའི་བར་བྱའོ། ད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཙུནྡའི་སྒྲུབ་པའི་ཐབས་བཞུགས་སོ། །༼Q«པེ་»«སྣར་»དྷཱ། «པེ་»«སྣར་པ། «སྣར་»མཚན། «པེ་»«སྣར་+དང༌། «པེ་»«སྣར་བརླབ་པ་བཟང་པོ། «པེ་»«སྣར་བཙལ། «སྣར་དུ། །«སྣར་རྗེ། @«པེ་»«སྣར་»བཟུང་། འ«པེ་»«སྣར་»ཏ། ཆེ«པེ་»«སྣར་»—། ཚེ«པེ་»«སྣར་པུ། «པེ་《སྣར་»—ནི། ཉི«པེ་《སྣར་»ཙ། ས་ལ་བདེ་བ་བཔཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། དགའ་བབསྒྱུར་ཅིང་ཞུས་པར། བསྡུར་འབྲས་རེའུ་མིགཀའོ་ཎི་ཏྲཱ་ཀ་ཋ། :ཏི་རྒྱུད། (། ་་་་་་ཁ་ཆུ། དཔེ་གཞི། སྡེ་དགེ། །་པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། ཀརྒྱུད་འགྲེལ། རྒྱུད། ཚེ་རྒྱུད། ཁོ༌ཚོ་nབ་པོད་རྟགས། །ཀ་མུཟླ་དུ་ཚན་གྲངས། །།ལྡེབ། །ན་—བ། བ—བ་། བ་—ན་། ན་བ། ཕྲེང་། ཀྱི་བྱ་ཚེག་ཁྱིམ། ་་༌་་རྒྱ་བ་། ་པ:འི་ਸਿੱਟਲਓ:ਧਰਨopਲੋਂਸਿੰਮਨੂੰਲਗਨਨਾਲੴਸ:ਤਿਸੰਗਦਾਸੂਬppਨੂੰਲਨ'ਸਮਾਸਸਰਣਾਮਸpཁས་མེཀ་ཊ་ཀྱ:ི། །་་་་་་ཅཏཱ་ཎཁོ། ལ་ཁྱའོང་ལ་ཤའིਇਕਮਾਸਨੂੰਨਵੀਂਸ਼ਮਲਸਿੰਰਸਸੌਂਸਲਰਾਂਨੂੰਸਰਲ།
