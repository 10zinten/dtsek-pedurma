ཧ་ཧོབྷ་ག་བཱན་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་མཱ་མེ་མུཉྩ་:བཛཱི་བྷ་བ་མ་ཧཱ་ས་མ་ཡ་སཏྭ་ཨཞེས་བྱ་བའི་ཡི་གེ་བརྒྱ་པ་བརྗོད་པའི་རྗེས་ལ་བསྡུ་སྤྲོ་ལ་སོགས་པ་རྫོགས་པར་བྱས་ཏེ། དམ་ཚིག་སེམས་དཔའི་གཟུགས་ཀྱིས་སྤྱོད་ལམ་ཐམས་ཅད་དུ་་་རྣལ་འབྱོར་པས་མཉམ་པར་མ་བཞག་པའི་སྐབས་སུ་གནས་པར་བྱས་ནས། ཡང་དག་པའི་ཡེ་ཤེས་རྒྱ་མཚོའི་མཐར་ཕྱིན་པར་གྱུར་ནས་དུས་རིང་དུ་མི་ཐོགས་པར་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཐོབ་པར་འགྱུར་རོ། །ཚིག་ལྷུག་པ་དག་གསལ་པོར་ནི། །དཀར་པོ་སྒྲུབ་ཐབས་ཡང་དག་བྲིས། ༽། པདྨའི་འབྱུང་གནས་བསོད་ནམས་གང་། །དེ་ཡིས་སྲིད་ལས་གྲོལ་བར་ཤོག། ཚིག་ལྷུག་པའི་ཨ་ར་པ་ཙ་ནའི་སྒྲུབ་ཐབས་་སློབ་དཔོན་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།།པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། ཁམས་པ་ལོ་ཙཱ་བ་བ་རིས་བསྒྱུར་བའོ། །།།པ་དང་བར་དུ་བས་(ཎ་བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ཚིག་ལྷུག་པའི་ཨ་ར་པ་ཙ་་་ནའི་སྒྲུབ་ཐབས་ཀྱི་རིམ་པ་བཞུགས་སོ། །༄༅«པེ་»«སྣར་»ནཱ། @«པེ་»ཅིག«པེ་»«སྣར་པ། «པེ་»«སྣར་བྱུང་། །«པེ་»«སྣར་གྱི། །ཚུ«པེ་»«སྣར་»བསྡུ་བར། ་འ«པེ་»«སྣར་»+ཏུ། «པེ་»«སྣར་»དཀྱིལ་་་་་འཁོར། «པེ་»«སྣར་»དང༌། «པེ་»སྣར་—ཀྱིས། ཚེ«པེ་»«སྣར་»«ཅོ་»ནི། «པེ་»«སྣར་»+རྣམས། «པེ་»«སྣར་»གི། ཚེ«སྣར་བསྟེན། «སྣར་བསྟན། «པེ་»«སྣར་»ཤུདྡྷ། ཚུ«པེ་»«སྣར་»ཀུ་བ། (འདིའི་འོག་གི་ཝ་མང་ཆེ་བ་«པེ་»«སྣར་»ལ་བ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་དུས། ཚེ«པེ་»ཀྱིས། །ཨེ«པེ་»«སྣར་ཡེ། «པེ་»«སྣར་པར། ་«པེ་»«སྣར་པ། ༄༅«པེ་»«སྣར་+མོ། །
