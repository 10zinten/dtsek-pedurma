རྒྱུད། །ོབྲཧྨའི་ཕྱེ་མ་སོར་བཞི་ཙམ་ལ་བརྒྱ་རྩ་བརྒྱད་མངོན་པར་བསྔགས་ལ་ནང་པར་སྔ་་་བར་དཀྱིལ་འཁོར་བྱས་ལ་ཇི་ལྟར་འབྱོར་པས་བཅོམ་ལྡན་འདས་ལ་མཆོད་ལ་མར་རམ་ཀཉྫི་བཏུང་བར་བྱའོ། །ཤུ་དག་ཞོ་ཕྱེད་ཀྱི་ཚད་ལ་དེ་བཞིན་དུ་མངོན་པར་བསྔགས་ལ་མར་རམ་འོ་མའམ་མར་ཁུ་དང༌། །ཟླ་བ་གཅིག་ཏུ་བཏུང་བར་བྱའོ། །ཡཔོ། དིག་པ་དང༌། ལྐུགས་པ་དང་། །ལྡིབ་པ་རྣམས་རྣམ་པར་ཉམས་པར་བྱེད་དོ། །ཟླ་བ་དྲུག་ན་གཞན་ཡང་མ་ཐོས་པའི་བསྟན་བཅོས་རྣམས་ཀྱང་མངོན་དུ་འགྱུར་རོ། ཁབ་། རྨོངས་པ་རྣམ་པར་ཉམས་པར་འགྱུར་ཞིང་ཐོས་པ་འཛིན་པར་འགྱུར་རོ། རོ། ཁུ་བྱུག་ལྟར་དབྱངས་སྙན་པར་འགྱུར་:བ་དང་། མེངག་གསལ་པོར་འགྱུར་རོ། །འཕགས་པ་འཇམ་དཔལ་རྡོ་རྗེ་དཔའ་བོ་གཅིག་ཏུ་གྲུབ་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།སྐྲ་ནས་ཀ་ཁོzapཀག་འདེ་འབད་ས་ཀ་ག་ང་:བདུན་པོ་བཟོ་ང་ལ་ཁ་ག:ཁ་པར་དང་རྒྱུ་འདུག་། ་དང་ལ་ལ་བ་ག་ལ་ལ་སྒྲོ་བ་དང་། བར་སྐབཀ་ཤ་བའི་དཏེ་བྱང་། ཀ་ཁ་ག་ཋའོ། །སྟོན་ཀའི་ཟླ་བRཀྱི་གཙོ་བོ་། །ཆོ་ག་དང་། ཁཊ་ག:ང་འདྨ་ཀ་ཁག་དང་ས་མ་ག་གའི་ཁ་ཁ། ཡོན་པོ་བཀལ། ཟླ་བ་བྱུང་བ་ལ་བཀ་ཤ@་བྱ་ལ། ཁ་ཡས། ཁོ་ཁག་ལ་ཚོགས་ਕਲਨਾਇਡਸੁਲਣਸਪਲਕਾਛੇਸੌਣਜਿਖਲཔ། ཁོ་བ་འགའ་གཙོ་བློ་ཡི་ཁ་ཡ་ན་ཁག་ལ་། གས་པས་ཀུན་ཏུ་ཁེབ་ཀྱི་བློ་ཁོག་ལུ་ཀཚོམཆོག་ཆུ་ཕྱི། །pཁ་ཁའི་གཟིའི་ཁ་ཁ་ཁ་འགྲོ་ཕྱོགས་ཁག་དང་བལ་བའིདུ་ཡི་གེ་ཁ་ཤོས་ལ་སྤྱོད་ཁ་བ་དང་ཁོ་པའི་བར་རྫོང་རྒྱུ་ཞིམ་པ། ཁོག་་ང་ལ་ཁ་ཁ་ཏ་ལ་སྐྱེ་དུ་ཡོང་འབབ་འཁ་འབྱའོ། །རྗེ་ཁ། aཀྱན་ཞིག་ཁ་པར་ཁ་ཁ་པའི་ཎ་ཏུ་བསྐོ་བྱེད་ཝལ་ཁའི་ཟླ་བ་ཀ་ཁ་འཇལ་བ་འ་མགོན་ལ་ཏེ་རྒྱ་ཚོང་བ་བོ། །དེ་བ། ཚོང་བ་ན་ཁྱཁཀྱི་ཚོ་ཁ་ཁ་ཚོ་ཁ་།
