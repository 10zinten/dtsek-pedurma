རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས། །འཁྱིལ་པར་བསམས་ལ། །།ཨོཾ་བཛྲ་ཤྲཱི་ཁ་ལེ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཞེས་པའི་སྔགས་་་བཟླས་པར་བྱའོ། ་ལ་། འདི་ནི་མན་ངག་གི་རིམ་པའི་ནང་ནས་འབྱུང་བའི་རྡོ་རྗེ་་་་ལྕགས་སྒྲོག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།པོ་ཅུང༌། ་བཞད་པའོ། །ཕྱག་བརྒྱད་པ་དུས་ཀྱི་དང་པོས་མི་འཇིགས་པ་ཡིན་པ། གཉིས་པ་ན་རྡོ་རྗེ། གསུམ་པ་ན་རྗེ་ལུགག་སྒྲུབ། །བཞི་པ་ན་མ་གཏེ། བག་པ་དང་བོ་ན་ཁྲོ་བ་དྲ་ཕུག་པ་འཚོ། གབསྡུར་མཆན། ཟ་འ་ཡ་། ཀུན་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»ཤྲཱི་ཁ་ལ་སཱ་དྷ། «སྣར་»ཤྲཱི་ཁ་ལ་སཱ་དྷཱ། ཉི«པེ་»«སྣར་—མ། «པེ་»«སྣར་»སེར། །«སྣར་»སྒྲོགས། གས་«པེ་《སྣར་»། །སྐད་ཅིག་མ་དེས་སྤྱོད་ཚུལ་སྤྱི་དང་བོ་ན་དངོས་གྲུབ་གཞན་ས་ཅི་དགོས། ཨོ་བཛྲ་གྱི་ཁ་ལེ་དུ་ཕཊ་ཀཱ་ཏཱ་ཞེས་སོ། རྡོ་རྗེ་ལུགས་སྒྲོལག་པ་འོད་ཐུབག་སྤྱི་ཚོ་ཁ་ཊ་བསྡུར་འབྲས་རེའུ་མིག་ཁ་ཆུ། །ཁ་ཊ་ཊོ་། ཎ་ཀnཀྱི། ཀརྒྱུད། （ཀ་ཀ། བློ་nཔ་བློ་ཊོ་ཁ། །དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། ཀ་རྒྱུད། བརྒྱུད་འགྲེལ།  ོརྒྱུད། །ངmཔོད་རྟགས། །ར་ཚེ…མུ་པན་ཅད་དུ་ཚེདུཚན་གྲངས། །།ལྡེབ། བ་བ་ནབ་ནབ་བt_བ*/*ཕྲེང་། ཆུ་/༽ད༌ཁྱ*/ིཏྟ་ནཔཚེག་ཁྱིམ། ང་༽། :་ཡས་ཡང་,z/:ਸਵਨੂੰਸਵਾਲਮਾਸਨੂੰਨੂੰਸੌਵਨਬਿੰਨཆ་ཁ། ཁྲོ་ཁགྱི་དྲི་ལོ་pཚུ་ག་ngཎ་ཌི། ་༌ཀོད་n@ཁ་པོཁ་ཏིཎཎཱ་ཎན་ཥpཊི་ཀ་ཁ་ཁ། །ཀ་ཙཥ་ཌཏུ་ཆཥ་ཀ་ཀཚོར་ཌ་ཀ་ཌར་་།
