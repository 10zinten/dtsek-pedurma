།དོན་ཁྱུ་བས་ཚོ་བ་ལ་རྒྱུ་དཀའ་ནམ་ཁར་སྐྱེ་ན་ཚོ། ::ལྔ་ཚོ་ལཌ་ཎཊི་ལྟར་གཏགས་པ་དང་ཁཁཁྲོལ་nཁང་nep། ་མ་ཆག་༄༅། །འཕགས་མ་སྒྲོལ་མ་འཇིགས་པ་བརྒྱད་འོ་ན་མ་གཅིག་པ་ལས་སྐྱོབ་པ་ཞེས་བྱ་བའི་་ལ། །གང་ད་ཀྱི་ཁ་ཡི་ཁྲི་པ། །ཎསྒྲུབ་པའི་ཐབས། །བག་ཚོ་ལ་ཁ་ཚོའིਬਾਨੋਓਟਲਏਮਸਐਕਸਨਾਇਸ:། ཆུ་ཏོ་ཆཔ་ར་ཁཏྟ་ཀie/ཏ་ཏ་ན་ག་ཊི་ལ་ན་རོའིབ་བཏཏེ་ཏི་ང་། །ཚུད། ཁ་ཅིགསྤྲོ་ན་nཥཎ་ཊི་ཁྲོད་དེས་ཁོ་ཁp༄༅། །རྒྱ་གར་སྐད་དུ། པཨཱརྱ་ཏཱ་རཱཥྛ་བྷ་ཡ་ཏྲ་ཏ་ནཱ་མ་སཱ་དྷ་ནཾ། །བོད་སྐད་དུ། :འཕགས་མ་སྒྲོལ་མ་འཇིགས་པ་བརྒྱད་ལས་སྐྱོབ་པ་ཞེས་བྱ་བའི་སྒྲུབ་པའི་ཐབས། རྗེ་བཙུན་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །ཀུན་མཁྱེན་རྒྱལ་བའི་བདག་པོ་ལྷན་ཅིག་སྐྱེས་པའི་ཆོས་སྐུ་དེ་བཞིན་དུ། །ལོངས་སྤྱོད་རྫོགས་དང་སྤྲུལ་པའི་སྐུ་སྟེ་པདྨའི་འདབ་མ་བཞིན་དུ་ཡངས་པའི་སྤྱན། །རྣལ་འབྱོར་དག་་་པར་རྣམ་པར་ཐར་གྱུར་སྲིད་དང་སྲིད་ལས་བྲལ་བའི་སྐུ་གསུང་ཐུགས་ཆགས་པ། །ཐབས་དང་ཤེས་རབ་གཉིས་མེད་ལྷ་དང་ལྷ་མིན་མིས་མཆོད་དེ་ལ་ཕྱག་འཚལ་ཏེ། །སྐྱེ་འཇིག་རྣམ་པར་སྤངས་ཕྱིར་ཆོས་ཉིད་མི་འགྱུར་ཡང་དག་ཉིད་དུ་དབྱེར་མེད་་་་ཀྱང་། :། འཁྲུལ་བཅས་འགྲོ་བའི་དོན་ཕྱིར་སྐུ་གསུང་ཐུགས་ཀྱི་ཕྲིན་ལས་ཁྱད་པར་གསང་སྔགས་ལམ། །སྒྲིབ་པ་མྱུར་འཇོམས་སྣ་ཚོགས་ཕྱག་རྒྱའི་གཟུགས་འཕྲོས་་་སངས་རྒྱས་ཐམས་ཅད་སྐྱེད་ཅབྱེད་ཡུམ། །འཇིགས་པ་བརྒྱད་ལས་སྒྲོལ་བ་འདི་་།
