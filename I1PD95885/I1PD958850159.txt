རྒྱུད། ཨུའི::ཀ། བསྔགས་ལ། ་ནང་པར་སྔ་བར་དཀྱིལ་འཁོར་བྱས་ལ་ཇི་ལྟར་འབྱོར་པས་བཅོམ་་་ལྡན་འདས་མཆོད་ལ། མར་རམ་ཀན་ཛི་བཏུང་བར་བྱའོ། འ། ཤུ་དག་ཞོ་་་་ཕྱེད་ཚད་དེ་བཞིན་དུ་མངོན་པར་བསྔགས་ལ། །ཌ་མར་རམ་ཅི། འོ་མའམ་ཚིམར་ནག་དང་ཟླ་བ་གཅིག་ཏུ་བཏུང་བར་བྱའོ། །རིག་པ་དང་། །ལྐུགས་པ་རྣམ་་་པར་ཉམས་པར་བྱེད་དོ། །ཟླ་བ་དྲུག་ན་ཐོས་པ་དང༌། ཀ་གཞན་ཡང་བསྟན་་་་་བཅོས་མངོན་དུ་འགྱུར་རོ། །མི་ཤེས་པ་རྣམ་པར་ཉམས་པར་འགྱུར་རོ། །ཐོས་པ་འཛིན་པ་དང༌། །ཁུ་བྱུག་ལྟར་འཇམ་པའི་དབྱངས་གསལ་བའི་ངག་ཏུ་འགྱུར་རོ༑། འཕགས་པ་འཇམ་:དཔལ་རྡོ་རྗེ་དཔའ་བོ་གཅིག་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ༑། འ། །།།:::ད་སྐབས་ལ་ང་ལ་དང་གཉན། །ཤེལ་ཁྱོམ་ཁམས་པ་:ལོ་ཙཱ་བ་བ་རི་དགེ་སློང་ཆོས་ཀྱི་གྲགས་པས་བསྒྱུར་བའོ། །།།པ་དེ་དག་ལ་་། ཉེ་དུའི་ལན་ལ་སྔ་ཀྱང་། ཁའ་དང་ཁོང་ལ་ག་ད་བ་ཛི་ལ་ਮੌਸਮਝਸਵਾਸਛਲੇਸਾਲ,ਕਿਉਕਿཔ་ནས་དར་བའི་དུས་སྐལ་དང་བསྡུར་མཆན། ཁྲོ་དང་ཀློག་བརྒྱ་འཚོཔ་རཏུ་བབ་ཁོ་ན་དང་བ་འདྲ་་་་་་་་་་་་་་་་་་་་་་་་་་་་་ཐུག་པ་ལོ་དང་་་རཀྵ་ཀུ་དང་ལ་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་པའི། «སྣར་དཔའི། །@«པེ་»«སྣར་བསྒྲུབ། ༄༅«པེ་»«སྣར་བའི། «པེ་»«སྣར་+ཏེ། «པེ་»«སྣར་བྱ། །@«པེ་»«སྣར་ཨ། «པེ་»«སྣར་བཀོད། «པེ་»«སྣར་»གྱིས། གྱི«པེ་»«སྣར་ཀྱི། །«པེ་《སྣར་+ནི། གྱི«པེ་«སྣར་སྤྱད། «པེ་»«སྣར་»། ;&«པེ་སྣར་»ན། «པེ་»བདག་ཉིད་བཅུག་སྟེ། «པེ་»«སྣར་ནི། ༄༅«པེ་»«སྣར་པ། O«པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»ཁོངས། ཁྱེ«པེ་»«སྣར་སྙེད། གྱེ«པེ་»«སྣར་བསྒོམ། «པེ་»«སྣར་བཟླས་ཏེ། «པེ་«སྣར་»+། 《ཚེ«པེ་»«སྣར་གནས། །ཚོ་གཏིབ། །ལར་སྐྱེ་ཤིའི་བར་ཡི་«པེ་»«སྣར་གྲུབ། «པེ་»«སྣར་»—གྱི། @«པེ་སྣར་+པ། «པེ་»«སྣར་མ། གྱི«སྣར་རིགས། «པེ་»«སྣར་བྱ་བའི། @«པེ་»«སྣར་»སྒྲུབ་པ། ཉི«པེ་»«སྣར་»པར། «པེ་»«སྣར་»བ། ཁྱེ«པེ་»«སྣར་»ཀ་རའི། «པེ་»«སྣར་»གཞག་གེ«པེ་»«སྣར་བརྒྱ། །
