བསྟན་འགྱུར། རྒྱུད། མུའིབྱུང་ནད་རྣམས་དང་། །བཀྲ་མི་ཤིས་ནི་སྣ་ཚོགས་དང་། །།མཚོན་དང་དུག་དང་རིག་སྔགས་དང་། །ཕྲ་མེན་ལས་བྱུང་སྦྱོར་བ་ཡིས། །ཤིན་ཏུ་མི་བཟད་་ཐབས་ལྡན་པའི། །སྟོབས་ནི་སྣ་ཚོགས་འཇོམས་པ་དང་། མང་། སེང་གེ་གླང་པོ་སྟག་དང་སྦྲུལ། །ཆོམ་རྐུན་མེ་ཆུ་ཤ་ཟ་ཡི། །འཇིགས་པ་ཆེན་པོ་དེ་རྣམས་ལས། །སྐྱོབ་མཛད་དེ་ལ་བསྟོད་ཕྱག་འཚལ། །།འཕགས་མ་ལྷ་མོ་སྒྲོལ་མ་ལ་བསྟོད་པ་རྫོགས་སོ། །།སྐྲག་དང་དུད་འདུམ་བྱེད་ཡུམ། བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་ར་དེ་བི། @«སྣར་དངོས་པོ། «ཅོ་»མདངས་པ། ཆོ«པེ་»«སྣར་ལྟར། དེ་«སྣར་»བ། «པེ་སྐྱབ། 《སྣར་»སྐྱབས། །ད། ལ། ::གཚུ་ར་་གནད་ཀྱི་ཚོར་། རྒྱུ་། །བ་ཆུ། ད་་་་་་་དུངོ་། །ནོར༽︽g'?p། །ར་བསྡུར་འབྲས་རེའུ་མིག་ར་། །ཟླ་/། ན*#རྒྱུད། （ཙ་ནི། རྒྱུད། དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །ཏེ་༼་ཚོ་ལྡེབ། ན༽ན་བ་ན་ནབ་ན་ན་། པན་ཕྲེང་། །ཚིག་རྐང༌། །།ཚེག་ཁྱིམ། །།་་། ལM/ཚེ་ལོཚོ—།
