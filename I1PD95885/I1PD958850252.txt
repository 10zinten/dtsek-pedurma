ལག་གཉིས་སོ་སོ་དག་ཏུ་བྱ། ས། །ཁུ་ཚུར་བཅིངས་ཏེ་:སྡིགས་མཛུབ་གཉིས། །མཐེ་བོའི་དབུས་སུ་གཞག་བྱ་སྟེ། །མེ་ཏོག་སྣ་ཚོགས་མཆོད་པར་བྱ། །ཕྱག་རྒྱའོས། །ཨོ་མོ་མི་ཏྲི་ཡ་སྭཱ་ཧཱ། བྱམས་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །ཁྱད་པ། རྒྱུ་ནི་བཔཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། །ཁམས་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་་་བསྒྱུར་བའོ༄༅། །ག་ནི་། ཡིན་ཏེ། ::ཀ་ཁརབ་་༌༌། །བག་ང་ལ་སྔགས་འདིས་མ་བསླངས་བསྡུར་མཆན། ལག་བདུན་དུ་བྲལ། པད་ཡེ་ས་པ་དག་དག་་ག་བ་:འཛད་པར་འབག་འབགས་པའི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བྱམས་པ་དང་་་་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཕྱིན་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«སྣར་»+པའི། «སྣར་»བྱམས་ལ། «པེ་»མོ་་་གྱིས། «སྣར་»མེ་གི། ཚུ«སྣར་ཕྱགཚེ«སྣར་»རྣ་བར། ཚེ«སྣར་བཅིངས། /«སྣར་»བསྡིགས་འཛུབ། ཉི«པེ་»བཅའོ། «སྣར་བཅའ་བ་བྱའོ། «པེ་»ལོ་ཙ། «སྣར་»ལོཙྪ། །ཙཡསྣར་ཙོང་ཞུས་སོ། ཀར་ཏོ། nཥ་་པ་ལ་ཁ་ཎ་ཊ་ཁོ་ལ་ཀྱི་ཀར་ཏོ་བྱའི་རོ་ད་ཀང་ཀཚོ། །ཚེ་ཏུཏུ་བྱ་ཀ་ར་། །།དགbཔ་ལ་ཀཀ་n/ཚེ་ཊཾ། དང་བསྡུར་འབྲས་རེའུ་མིགབྱང་ད། །པའི་nསསོགས་འབབ་ན་ཚེ་np། རྒྱད། ་་་་་་་༌་་ཀR.nདpདཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། ཡར་སྡེ་ཚན། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། །z་༼ལྡེབ། ན་ན་ན་+བབན་ན་ན་སུརྒྱུབཕྲེང་། །དང་དང་གཙས་གནས་ཚེག་ཁྱིམ། །རྒྱུད། ་། །
