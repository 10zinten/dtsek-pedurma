སྒྱུཔར་བྱ། །བ། སངས་རྒྱས་སྤྲུལ་ཚོགས་འཕྲོ་བ་ཡིས། །འགྲོ་བ་གསུམ་པོ་རབ་ཏུ་་བསྒྲུབ། །མཾ་ཡིག་ལས་བྱུང་རྟག་ལས་ནི། །འཇམ་པའི་རྡོ་རྗེ་རང་བྱོན་ཉིད། །སྐུ་མདོག་དམར་པོ་གུར་གུམ་འདྲ། ཊ། སྔོ་དང་དཀར་པོ་ཞལ་གསུམ་སྟེ། །ཕྱག་་གཉིས་རང་འོད་རིག་མ་ལ། །མཉམ་འཁྱུད་མཆུ་༔ནི་ཞལ་གྱིས་འཛིན། ཞོ་། ཕྱག་ན་རལ་གྲི་མདའ་དང་ནི། :། གཞུ་དང་ཨུཏྤལ་སྔོན་པོ་བསྣམས། །སྣ་ཚོགས་ཆུ་་སྐྱེས་ཟླ་བའི་གདན། ང། རྡོ་རྗེ་སྐྱིལ་ཀྲུང་ཟླ་འོད་བཞིན། །སྲིད་པར་ཆགས་ཤིང་འཁོར་བ་ལས། །ཡང་དག་ཆགས་ཤིང་ཆགས་བྲལ་མཛད། །དཔལ་འཇིམ་རྡོ་རྗེར་ཀུན་བདག་ཉིད། པ། ཐམས་ཅད་སྒྱུ་མ་ལྟ་བུར་བསམ། །ཨོཾ་དྷརྨ་དྷཱ་ཏུ་སྭ་། བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། མིག་སོགས་རྣམས་ལ་འོད་ཟེར་གྱིས། །དེ་ནས་ས་བོན་ལ་གཞུག་བྱ། །:：ཊཱི་ཛི་ཁཾ་གཾ་སྐོ་སཾ་ཞེས། །ས་བོན་རྣམས་ཀྱིས་ཉེར་བསྒྲུབ་པོ། །ས་དབང་རྡོ་རྗེ་ནམ་མཁའ་དང༌། །འཇིགས་དབང་སྒྲིབ་སེལ་ཀུན་དུ་བཟང་། །དེ་བཞིན་རྡོ་རྗེ་སྐྱེ་མཆེད་རྣམས། །ཉི་མའི་དཀྱིལ་འཁོར་ལ་ཞུགས་སོ། །རང་གི་མགོ་བོ་མགྲིན་སྙིང་ཟླ། ཀྵ་། ཨོྃ་ཨཱ༔ཧཱུྃ་གི་སྔགས་འོད་ཀྱིས། །སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་ཡི། །སྔགས་ཀྱི་བདག་པོ་བསྒོམ་པར་བྱ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཀཱ་ཡ་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། ཏུཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཝཱཀ་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཙིཏྟ་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། །རྡོ་རྗེ་གསུམ་གྱི་མཚན་ཉིད་བལྟས༑་། སྙིང་གར་ཡེ་ཤེས་ས་བོན་ལས། །ཕྱོགས་རྣམས་ཀུན་བྱོན་རྫོགས་་སངས་རྒྱས། ཚོ། རིག་མ་མཆོད་ཅིང་རིག་དབང་འཛིན། །སྤྱན་དང་སྐུ་སོགས་་་བྱིན་གྱིས་བརླབ། ཁ། །རང་སྔགས་ལས་བྱུང་འོད་གསལ་མ། །མགོ་བོ་སྙིང་ག་ལྟེ་གསང་ཞབས། །སྐུ་སོགས་རྡོ་རྗེ་རྣམ་བལྟ་ཞིང་། །ཨོཾ་ཧཱུཾ་སྭཱ་ཨཱཧཱ་ཞེས་པའི། །ཡན་ལག་ལྔ་འོད་རབ་ཏུ་གསལ། །སྐུ་དབང་མི་གཡོ་རིན་ཆེན་དབང་། །འོད་།
