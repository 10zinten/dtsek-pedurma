ལྷའི་གོས་དང་། །རྒྱན་དང་འོད་དཔག་ཏུ་མེད་པ་དང་། ཚོརལ་པའི་ཅོད་པན་གྱིས་བརྒྱན་པ། སྙིང་གའི་ས་བོན་གྱི་འོད་ཟེར་གྱིས་སེམས་ཅན་རྣམས་བཀུག་ནས་རང་དབང་:མེད་པའི་གནས་ལ་བཀོད་དེ་སྔགས་བཟླས་པར་བྱའོ། །ༀ་མ་ཧཱ་རཱ་ག་བཛྲ་རཱ་ག་ཡ་སརྦ་སཏྭཱ་་ན་ཧོ། ཞེས་སོ། །ཆོས་ཀྱི་དབྱིངས་ངག་གི་དབང་ཕྱུག་གི་སྒྲུབ་ཐབས་ཀྱིས་དབང་དུ་བྱ་བའི་ཆོ་ག་་་རྫོགས་སོ། །སོམ་པ། ན་བ། ས་བ་པ་ལ་བར་ད་བར་མི་བ། ་པདྲ་བ། ག་ཡ་སརྦ་སཏྭཱ་ན་ཉིད་ནས་སེམས་ཅན་མས་དང་། བར་དབང་དུ་འགྱུར་བ་འགྲོ་ཞེས་པའི་ངེས་པ་ཤིན་ཏུ་བསྒོམས་པ་ལས་མང་བྱེ་བའི་གཏུབ། ས་བྱེད་སའི་མཚོབ་བསྡུར་མཆན། ནང་པའི་རྡོ་རྗེའོ། །ཞེས་དང་། དགའ་བ་བྱེད་པའི་ཏིང་ངེ་འཛིན་(.མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་《སྣར་»བ། «པེ་»«སྣར་»—། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»། «པེ་»«སྣར་»ལ། /«པེ་»«སྣར་བའི། «པེ་»«སྣར་»སཏཱ། «པེ་»«སྣར་»+། «པེ་»«སྣར་»ཀྱི། །།ཏུཏྟ། ཎཛོད་ཡ་ཀཱཊ་ཀ་ལ་པ་ཀ་ལ་པ་ཀ་མ་ཅ'ཁྱར་ཆུ། །ཟླ་ཁ་ཚོས་ཏ་ཞོ་ཀ་ཅོཀ་ཊི་ས་རྒྱུ། །ད་ཀྱི་གཞས་མ་ཀོ་ཎཁོ་ཆ་བཀལ་སབ། mཚ། ད་ཁ་ཚུ༌བསྡུར་འབྲས་རེའུ་མིག་ལ་་་་་་ང་ལ་nསྐུ་བར་ན་ངོས། ་་་༌་ངལ༽རྒྱུད། (་། ཆོལ་ངང་ལ་ཏཁང། ཁ་ངོ་ཡོ་ཆཚོཁདཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ::ཅོ་ནེ། སྡེ་ཚན། ནང་ང་རྒྱུད། ང་རྒྱུད་འགྲེལ། །རྒྱུད། །དུང་རྒྱུད། །་ང་སྐཔོད་རྟགས། ཚན་གྲངས། །།ལྡེབ། ན/°བ་། ན་བ། ན་_བསན་བ་ན་རྩ་བ་བྱཕྲེང་། /་། sཀ་ཀོབཡི་པ་ཚེག་ཁྱིམ། ཀྱིརབ་བྱ་་_།
