བལྟས་པ་རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པའི་གནོད་གནས་སུ་བདག་ཉིད་བསྒོམ་པར་བྱའོ༑། དེ་ནས་ཕྱག་རྒྱ་ནི་ལག་པ་ཐལ་མོ་ཁོང་སྟོང་དུ་བྱས་ལ། གུང་མོ་་་གཉིས་ཀྱི་ཚིགས་གསུམ་པ་བཅག་པའི་ཕྱག་རྒྱས་སྙིང་ག་དང་། དཔྲལ་བ་དང་། ལྐོག་མ་དང་། སྤྱི་བོར་གཏུགས་ཏེ་བཟླས་བརྗོད་བྱའོ། །ཨོཾ་ཛཾ་བྷ་ལ་ཛ་:ལེནྡྲ་ཡ་སཱ་ཧཱ༑འདི་ཉིད་ཀྱི་ཆུ་སྦྱིན་བླུགས་བརྒྱ་རྩ་བརྒྱད་དབུལ་ལོ། །མཆོད་པ་ཡང་བྱས་ལ་བཟླས་པ་བརྩམ་པར་བྱའོ། །དེ་ནས་ནོར་གྱི་བདག་པོ་ཆེན་པོར་:འགྱུར་བར་ཐེ་ཚོམ་མེད་དོ། །མགོན་པོ་གནོད་གནས་ཀྱི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །གནོད་གནས་དང་ཁྲོ་གཉེར་ཅན་གྱི་སྒྲུབ་པའི་ཐབས་བཞུགས་སོ། །«སྣར་ཀྱིས། ་ ི«པེ་»བི་ཛ་པུ། «སྣར་བི་ཛ་སྤུ། ་ཀྱི«སྣར་»རྒྱ། ལོ་《སྣར་»སྟོང༌། 《སྣར་»ཀྱིས། །«སྣར་ཁའི། །«པེ་ལེནྡྲ་ཡ། །ptསྣར་ལེན་ཏྲ་ཡེ། ཚེ«སྣར་བརྩམས། «སྣར་»གྱུར་པར། ཉི«པེ་»«སྣར་»+པའི། ༽ས/}}ཆུང་ང༼སཱ་བྱང་ཏཱ་ལ་་་་་་འོསར་། :ཀཔལ་པོསཔས་བསྡུར་འབྲས་རེའུ་མིགAnPARGARITA"ỤCཁ་ཁོ་ཚོ་ནི་དེ་མ་་གལ་རྒྱུད། （རྒྱུད། དཔེ་གཞི། ་སྡེ་དགེ། །།པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། །སྡེ་ཚན། _རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། །ཀྱི་པོད་རྟགས། །་མུང་བདུ་་༼དུ་°། ཏི་མུའོ། །ཚན་གྲངས། དང་་། ག་པ། ལྡེབ། བབས། ན་བ་བ?བ་བབ་{ཚེག་ཁྱིམ། །།་།
