བསྟན་འགྱུར། ་རྒྱུད། ཨུའིབརྩེ་བའི་གནས། །རང་ཉིད་སྒྲོལ་མ་ལས་སྐྱེས་པའི། །ཀུ་རུ་ཀུལླེར་བསྒོམ་པར་བྱ། །བསྒྲུབ་བྱ་དེ་ཡི་ཁ་ནང་སོན། །འདོད་པས་མྱོས་པར་བསམ་པར་བྱ། །ཡེ་ཤེས་སེམས་དཔའི་ཆུ་སྐྱེས་ལས། །སྤྱི་བོར་བདུད་རྩིའི་རྒྱུན་འབབ་བསམ། ་། བརྟུལ་ཞུགས་ཅན་གྱིས་ཐུན་གསུམ་བསྒོམས། །ཉི་མ་བདུན་གྱིས་ལྟུང་བར་འགྱུར། །རྒྱུ། རིམ་པ་གོམས་པས་དངོས་གྲུབ་ཕྲན། །མཁས་པ་ཡིས་ནི་བསྒྲུབ་པར་བྱ། རྟ་། འདིར་ནི་དཔལ་དང་བཅས་གྱུར་པའི། །དངོས་གྲུབ་ཆེན་པོ་ཉིད་ཀྱང་འགྲུབ། །རྟོག་:པ་ལས་གསུངས་པའི་ཀུ་རུ་ཀུལླེའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །བྱ་ན། །འ་ཎ། ཁོ་ན་དང་པ་རང་ལོ་བསག་པ་དང་། །བྷཏྟན་པོ་དང་བྱ་བ་གང་བའི་འབོབསྣམ་པ་དང་། རྒྱུ་བ་ང་ཚོབ་འདུ་བ་བྱེད་དུ་བག་གོས་པ། །ཀཎ་དང་། །བད་པ་ལྟ་ལོག་ཞིང་ཁམི་ཁང། ་ནི། །འདབ་བྱ་བའི་བྱ་བ་ལས་ཏེ། །བསྡུར་མཆན། །ད་ཁྲོ་ལ་ཁྲོ་ཅན་དུ་བ་ག་ནས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྟོག་པ་ལས་་་་གསུངས་པའི་ཀུ་རུ་ཀུལླེའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«སྣར་»སྤྱིའི། gཀྲས་ལ«པེ་»«སྣར་»འགྲོ། ::སྣར་»གྱིས། ་་༌@«པེ་བཟུང་། «སྣར་གཟུངས། ་་་«པེ་»«སྣར་པ། «སྣར་བཅིངས། ་ ོ«པེ་»«སྣར་བཞག«སྣར་བྱས་ཏེ། «པེ་ཨཱཏྨ་ཀོ་ཧཾ། «སྣར་ཨད་མ་ཀོ་ཧཾ། ཚེ«སྣར་»ཀྱིས། ཚེ«སྣར་སོ། ཀྱེ«སྣར་གྱིས། འུ«སྣར་འདོན། ༼ལོ《སྣར་རྗེ། «པེ་»«སྣར་མེད། 《སྣར་བྱེད། «སྣར་བརླབས། ་/«སྣར་ཨུད་དཔལ། @«སྣར་ཨཱ། Q«སྣར་ལ་སླར། «སྣར་ཡིས་རྣམ་པར། ཁྱེ«སྣར་པ། «སྣར་སྦྱང་བར་རྟོག«སྣར་བྱེད་པ། མི«སྣར་»དེའི། «པེ་»«སྣར་བྱ་སྟེ། གྱེ«པེ་»«སྣར་མོ། ཀྱེ«སྣར་བརྒྱན། «སྣར་ཨ། ཙེ《སྣར་ཡིས། /ཚེ«སྣར་རྗེ་དཀྱིལ་དཀྲུང་། ལ«སྣར་པའོ། ཁྱེ«སྣར་དྲངས། །ཁྱེ«སྣར་»+། ལེ«སྣར་བཅིངས། བ་ཚེ«སྣར་»ཚིགས། ཚེ«སྣར་»གྱིས། གྱི《སྣར་»སྦྱོར་སྟེ། དེའབྱེ«སྣར་»མོ། ནཙེ«སྣར་བཅིངས། «པེ་»«སྣར་»ལོར། གསུ«སྣར་»དེ་ནས་དབང་བསྐུར་ཞུ་བྱ་སྟེ། །ད་ཡ#RNཚོ«སྣར་རྗེ། «སྣར་»+སོ། «པེ་»སྣར་—ཀུན། «སྣར་»གྱིས། ཚེསྣར་བའི། «སྣར་བྱ། «པེ་»«སྣར་ཡིས། «སྣར་ཡིས། «སྣར་འཚམས། «སྣར་གྱིས། ཀྱེ«པེ་ནི། «སྣར་»ཧྲི། ཝེ«སྣར་གསུམ། ཉེ«པེ་»«སྣར་བརྩམས། རྩེ«པེ་གྱི་ནི། «སྣར་++།
