པ《་ཁ། 《་@། ་ངསི། བསྒྱུ། ཌའང་《་། 《@། peng་ཁ»། །ཀ་ཋsཀཌ་སུ་བས《ཁ། ངོ་ཁསྒྲ[《་ག། 《ཁ?བསྡུར་མཆན། 《་ང་《ར་《ཏིཔེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཀུ་རུ་ཀུལླེའི་་སྒྲུབ་ཐབས་གཉིས་བཞུགས་སོ། ཉི«སྣར་པ། O«པེ་»«སྣར་གསང་། «སྣར་+ལོ། «པེ་»ཞིག་བསྒོམ། «སྣར་»ཤིག་བསྒོམ། ༄༅«སྣར་རི། @«སྣར་ལ། ཉི«པེ་»«སྣར་»+ལས། «སྣར་མིག་«སྣར་མིག་མི«སྣར་བསྒོམས། /Q«སྣར་»ཕྱིར། ︽«སྣར་—དང་། ཚེ«སྣར་»—དོག@«སྣར་»གདུབ། ཨེ«པེ་»མནབས་པ། «སྣར་མནབས་པས། «སྣར་སྐོར། «པེ་»«སྣར་»ཟེ། ཚེ«སྣར་བསྒོམས། «སྣར་»—པ། བརྒྱད། «སྣར་»རྣམ་པ། «སྣར་བྱས་པ། «སྣར་གྱིས། ཚེ«སྣར་པས། ཚེ«སྣར་»+ཕྱག་ཨེ«སྣར་ཡིད། «སྣར་»གི། «སྣར་»ཅི། ︽«སྣར་བགྱིད། མོ«པེ་»«སྣར་»—སྤྱན་སྔར། @«སྣར་པས། ༄༅«སྣར་»གདོམ། «སྣར་པར། «སྣར་»ཀྱིས་དོན་སྔོན་དུ་བྱས། ཁྱེ«སྣར་»གྱིས། ཀྱེ«སྣར་བརླབས། འགུ«པེ་བྷ་བ་ཨཱཏྨ་ཀོ་ཧཾ། «སྣར་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། ཙེ«སྣར་»གྱིས། ཅེ«སྣར་ལྟ། ཅེ«སྣར་»སྐྱེས། ཚེ«སྣར་རིམས། ཁ«སྣར་བསྐྱེད། ཁྱེ«སྣར་»གཞིའི། འཚེ«པེ་»མེད། «པེ་»ཟེ། «སྣར་ཟེར། ཉི«སྣར་»དུ། Q«སྣར་པོའི། «པེ་»གིས་དབུས། «སྣར་»གི་་དབུས། «སྣར་»དང་ཨཱ་ལི། «སྣར་པའོ། @«སྣར་»བར། ཚེ«སྣར་»ལས། «སྣར་»གྱིས། ཀྱི«པེ་»«སྣར་»+དེ། ཁྱེ«པེ་»ཥིཉྩནྟུ་མོ། ཨ་མི་ཏཱ་བྷཱ་ནཱ། «སྣར་»ཥིཉྩནྟུ་ཨན་མ་མ་ཏཱི་བྷཱ་ནཱ། ཨུ«པེ་འཕྲོ་བ། «སྣར་འཕོས་པ། «པེ་»«སྣར་»+དེ། «པེ་»«སྣར་»བསྒོམས་ཤིང་། «སྣར་»ཀྱིས། «སྣར་»གཏོར། @«སྣར་»བའི། «སྣར་»གསུམ་པ། «སྣར་»+ལ། @«སྣར་མེས། ༄༅«སྣར་»པ་ལ། «པེ་»«སྣར་ཐུགས། «སྣར་»ལས། ཝ«སྣར་»ཐུགས། _ཀྱི«པེ་»«སྣར་པཾ། ༼«སྣར་མགོན་པོ། ཁྱེ«སྣར་པོའི། «སྣར་»ལ། ཚེ«སྣར་ལ། ཚེ«སྣར་ལས། ཁྱེ«སྣར་པ། ་བ་«སྣར་»དུ། ®«སྣར་བསྲེགམི«སྣར་»པ་བཅིངས་པ། «སྣར་+དང་། @«སྣར་འགྲང་ཕྲེང་གི། «སྣར་རྒྱུ། «སྣར་བསྐྱེད། «པེ་འཇུག་པ་ན། «སྣར་འཇུག་པ་ནི། «སྣར་»པོའི། ཚེ«པེ་»ཛ་ཊ། «སྣར་ཛི་ཌི། ཀྱེ《སྣར་»རན། ཉེ«སྣར་»ཙཀྵ། /«སྣར་»མན་ག་ཏཱ། འཁྱེ«པེ་»«སྣར་བྷ། ཚེ«སྣར་»ཏྲ་སེ་ཏྲ་ས་ནི་ཏྲ། ཁྱེ«སྣར་»ཏྲ། ཁྱེ«པེ་བྷུ་ཏན་བྷ་ཊི་བེ་ར། «སྣར་བྷུ་ཏན་བྷི་ཊི་ཏྲ་བི་ཥ། ལེ«སྣར་»ཛ་ཌི། ཚེ«སྣར་—བདུད་རྩི་ལྔའི་བཟའ་བ་སྔགས་འདིས་སྦྱང་་་་བར་བྱའོ། །འཁྱེ«སྣར་ཧཱུྃ། «སྣར་»—གསུམ་གྱི་དེ་ཉིད། ཚེ«སྣར་གྲུབ། ན«སྣར་པུ། །
