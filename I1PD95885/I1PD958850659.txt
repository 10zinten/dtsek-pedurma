ཚོ་ལྟ་ཁོ་ཁRwཁྲོ། ཆ་ཀ་ཎཞུ་། xaeཝཁབ་gu་n:;ཎ་ཚོnཀའི་འཌ་། ཀ་ཎRཆ་ཀ་ཁ་ཁོ། ཥཎ་་ལ་ཁ་ཎ་ཁཚོ་ལཎ་ཌཎཎྜལ་ཁgལ། །ཁོང་དུ་ཁ་ཁ་ཁ་ཏི་ཀ་ཏ་ཙིལླ་ཚོ་སྐྱ་ཎ་ཁ་ད་ཁ་ཟླ་བ། ལ་ཟ་ཁ་ཎ་ཁོ་ལ་ཚོགpའི་ཁ་ངོ་་nylsཎྜི་་སྐུ་ཁཀཀ་ཁ་ན་ཌ་ཀnfoཀྱི་ཡཌ་Tਨਡਾਲਰਤੋਂਵਟਾਇਗਰਾਨੇਝਟਸੌਵਾਨਸngਰਨདེ་བ་༄༅། །བུང་བ་མྱོས་པའི་སྒྲུབ་ཐབས། རྗེ། །།།ཏོ་ཥ་བློ་། ཁོ་ལག་༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨ་ལི་མན་མ་ཐ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། བུང་བ་མྱོས་པའི་སྒྲུབ་ཐབས། །བུང་བ་མྱོས་པ་ལ་ཕྱག་འཚལ་ལོ། །ཤིན་ཏུ་ཡིད་འོང་བདེ་མི་འགྱུར། ཕྱག། བསྟན་བཅོས་ལོངས་སྤྱོད་སྐུར་གྱུར་ལ། །།ཕྱག་་་་འཚལ་བུང་བ་ལུས་མེད་ཀྱི། །སྒྲུབ་ཐབས་བསྡུས་པ་བདག་གིས་བཤད། །དང་པོར་རེ་ཞིག་ཆུ་སྐྱེས་དམར་པོའི་ཟེའུ་འབྲུའི་གཟུགས་བརྙན་ལ་དཔལ་ཧེ་རུ་ཀའི་་སྐུར་བདག་ཉིད་བསྐྱེད་ནས་རྣམ་པར་རྟོག་པ་ཙམ་གྱིས་ཁམས་གསུམ་པོ་ཀུན་ནས་བརྟག་པར་བྱའོ། །རང་གི་སྙིང་གར་ཤིན་ཏུ་དམར་བའི་ཨུཏྤ་ལ་ཆེན་པོ་འདབ་མ་བརྒྱད་པའི་ཟེའུ་འབྲུ་ལ་ཨཱ་ལི་དང་ཀཱ་ལི་བསྡུས་པའི་གཟུགས་ལ་སངས་རྒྱས་་་ཐམས་ཅད་ཀྱི་ཡོན་ཏན་རྗེས་སུ་ཆགས་པའི་ཨཱཡིག་དང་ཧཱུྃ་ཡིག་ཡིད་ཀྱིས་མངོན་་པར་བྲི་བར་བྱའོ། །དེ་ནས་ཨཱཡིག་དང་ཧཱུྃ་ཡིག་དེ་དག་ལས་ཤིན་ཏུ་དམར་་་་ཞིང་མཛེས་པའི་འོད་ཟེར་གྱི་ཚོགས་འཕྲོས་ཏེ་དེ་བཞིན་དུ་ཡང་གུར་གུམ་རློན་པ་།
