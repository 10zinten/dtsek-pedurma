སྡུད་པར་བསམ་པར་བྱའོ། །བསམ་གཏན་གྱིས་སྐྱོ་བར་གྱུར་ན་སྔགས་བཟླས་པ་བྱ་སྟེ། དེ་ལ་སྔགས་ཀྱི་རྒྱལ་པོ་འདི་ལྟ་སྟེ། ཨ་ར་པ་ཙ་ན་དྷཱིཞེས་བྱའོ། །འཕྲལ་དུ་ཉམས་མྱོང་འཇམ་དཔལ་གྱི། །སྒྲུབ་ཐབས་བརྗོད་པའི་བསོད་ནམས་་་གང་། །བསགས་འགྱུར་དེ་ཡིས་འཇིག་རྟེན་ཀུན། །རྟག་ཏུ་འཇམ་དབྱངས་་་་མཚུངས་པར་ཤོག། འཕྲལ་དུ་ཉམས་སུ་མྱོང་བའི་ཨ་ར་པ་ཙ་ནའི་སྒྲུབ་ཐབས་་་་རྫོགས་སོ། །།།།ཁམས་པ་བ་རི་ལོ་ཙཱ་བ་དགེ་སློང་དྷརྨ་ཀིརྟིས་བསྒྱུར་བའོ། །དེaསྐྱེཅཎ—ཎེ། །།ཨུ། བསྡུར་མཆན། བྱ་ན་་་་་་་་་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «སྣར་»འདོད། «པེ་»«སྣར་»པོར། «པེ་»«སྣར་»གཅིག«པེ་»«སྣར་པའི། «པེ་»«སྣར་»ལས། @«པེ་»«སྣར་»ལ། «པེ་»«སྣར་»སེམས་དཔའི། །«པེ་»«སྣར་»གཅིག«པེ་《སྣར་རྭ། «པེ་»«སྣར་གྱི། «པེ་«སྣར་»རའི་རྣམ་པར། «པེ་»«སྣར་»ལཱི། ༄༅«པེ་»«སྣར་»ལྟ། ཉི«པེ་»«སྣར་ཀྱེ། «པེ་»«སྣར་ཁུར། «པེ་»«སྣར་ཀྱེ་ཤེ། བཀྱེ«པེ་»«སྣར་»—ཀྱི། གེ«པེ་»«སྣར་»རང་། ཉེ«པེ་»«སྣར་འབྱུང་། ཁྱི«པེ་»«སྣར་—མོ། ་རང་«པེ་»«སྣར་»—ལ། «པེ་»«སྣར་+ནི། «པེ་»«སྣར་»ཙ། «པེ་»«སྣར་»དྷརྨ། རོལ་གྱི་དང་ཁྲོ་བ་དང་། རྡོ་རྗེའོ། །དེ་ལ་དེ་ལ་བུ་དང་། དེ་ཞིབགས་པ་དང་། གཏུན་ཤིང་བས་ས་པ། དབུ་ན་མི་བུ་སྤྱོད་པའི་འདུ་ཁང་བསྒོམ་པར་བྱའོ། །དེ་ནས་དམ་ཚིག་མི་བྱ་བ་རྒྱ་བཅིང་བར་བྱའོ། བལ་མོ་ལ་ག་བ་བཅིངས་ཏེ། མཐེབ་ཆེ་བ་ཉིས་ཁ་ཕྱེ་སྟེ་རིགས་མཛུབ་གཉིས་ཀྱང་ཁ་ཕྱེ་ཞིང་གྲུང་མོ་བ་ཉིས་ཚིགས་གསུམ་པ་བརུང། པར་བྱའོ། དེའི་ཀྲི་ཏཱ་ན་།
