རྫོགས་སོ། །་ཨུཏྤ་། །སྐྱེ་མ་བསྡུ་བ་བདག་པ་དག་བྱའོ། དེའི་པ། ::ས་སྤྱོད་པ། ལོགས་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང༌། སེར་པ་:ལོ་ཙཱ་བ་དགེ་སློང་བ་རིས་བསྒྱུར་བའོ༑། ༽་པོ་ཕྱ། །།ཉིས་པ། ན་པ་ག་ད་མེད་དམར་པོའི་:ལ་ག་ད་ན་བསྣམས་པ། གཡས་པ་ན་བ་དང་དཀར་པོ་དང་ལྡན་པ། ཤེར་བ། དམན་ལ་པོས་བརྒྱན་པ་བསམ་པར་བྱའོ། བསམལོ་ལ། ས་དུ་མ་བྱ་བ་ནི་འདབ་ལྗང་:བར་ཕྱིང་། །དཀྱིལ་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»སྒྲོལ་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»ཏ་ར། ཚེ«པེ་»«སྣར་»ཏཾ། «པེ་»«སྣར་»ཏཾ་གྱིས་མཚན་པ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»པ། ཚེ«པེ་»«སྣར་རྒྱན་པ། ཉི«པེ་»«སྣར་»ན། «པེ་»«སྣར་»ལ་ན་ལ་ཡིས། ཆེ«པེ་»«སྣར་སྐེ་བར། གྱེ«པེ་《སྣར་»+འོད་ཀྱི། ཞེས་བྱ་བ། ་བདུད་«པེ་»«སྣར་ཁམས་པ་ལོཙྪ། ནས་ད་ཆའི་ཁ་བྲབ། སློབ་དཔོན་ཡིད་བཞིན་ནོར་བུ་རྒྱལ་པོ། མ་ཟན་ཟླ་ཚོགས་ཞི་ཁྲོ་རབ་བ་ལྔ་ལ་ཁ་ཊ་ར་ཁ་ཆུ། ཁཀླུ་༽ཁྱོ་ག་བ་ག་ཚོད་བསྡུར་འབྲས་རེའུ་མིག་བདུན་དུ་བག་ལ་བླ། །ཁྲོ་ད་བ་ཅིང་། ནང་དུ་ཎསཱ་རྒྱད། གནས་ཁྱད་དུ་ཚོ། །དོ་ལ། དཔེ་གཞི། ཛྫ་སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ས་སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། །པའི་་་༼ལྡེབ། ནབ་ན་བ། ན་ན་ན་བ། །ཚེག་ཁྱིམ། །།།།།ਸਲਾਸੁਣਸਬAAPgਲਉਹཀ་ཁའི་ང་ལ་nཤོ་ཀྱ་ཏ་་ཨ་pཏཁ་ཆཎཱ་ཎའི་ཆཏི་ག་ཎའི་ཀཚོ་ཆུ་ཚོགས་།
