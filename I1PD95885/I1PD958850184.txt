དུན་པ་རྣམ་པ་བདུན་པ་ལ། །ཤིང་རྗེ་དང་འབུམ་ཆེན་པོར། །ཡིག་:སྟེང་རྒྱལ་བ་ནི། །སྔབ། ས་བབ་ཀྱི་ལག་པ་པོ་དག་པ་དང། །གང་ལ་རྒྱར་བ་བྱས་པས་ཆུ་ག་ཕུལ་ཡོད། ཞུ་མ་བསྡུར་མཆན། འབྲིག་བྲེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སེང་གེ་སྒྲའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»ཧཱ། Q«པེ་»«སྣར་»+པའི། «པེ་»«སྣར་»གེའི། «པེ་»«སྣར་—ཉ། @«པེ་»«སྣར་»གཅིག«པེ་»«སྣར་»ལ། «པེ་»«སྣར་བསྒོམས་ཏེ། །«པེ་»«སྣར་བའི། «པེ་»«སྣར་ཧྲི། «པེ་»«སྣར་ན། «པེ་»«སྣར་པ། «པེ་»«སྣར་»+། པོ«པེ་»«སྣར་སཏཱ། «པེ་»«སྣར་»ཀ«པེ་»«སྣར་»བ་རི། «པེ་»«སྣར་»:་འདི། ཚེ«པེ་»«སྣར་+པས། ་ཀྱི«པེ་»«སྣར་»ལོཙྪ། བདག་ལ་འདུལ་ཁྲོ་བོ་ལ་འགྲན་བདུས་བདག་ཉིད་ཀྱང་། །པོ་བྱུང་དང་འཛིན་པ་དང་གྲོལ་བའི། །ཡེ་ཤེས་ཚུ་བཀྲ་བ། བཀྲ་བ། །་་་་་་་པ། ལུ། ཀྱ་་་་་་སྟེ་་་་་་བལྟ་བའོ་ཊི་བསྡུར་འབྲས་རེའུ་མིག་བཞི་ས་ལ་བ་དང་། zད་བ་ས་༼མ་རྒྱུད། （ཚོnཥཱ་ཀཱ་ར་ན་༼དཔེ་གཞི། སྡེ་དགེ། ང་༽པེ་ཅིན། སྣར་ཐང་། *་ལྔཅོ་ནེ། ་snསྡེ་ཚན། ཚེརྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། ལྟ་བ་ཡང་། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། P༽*ལྡེབ། བ་ན་ན_བར། བ་བ་བ་ཕྲེང་། །ན་་ས་ཀྱི་་། ཚེག་ཁྱིམ། །ར། ན། །བང་་བཔརངཔ༽རང་དབོན་འཚེ་པ་མི་ཁྲི་ལྔ་བའི་་་་ཊི་ཏ་བ་དག། གང་དག་བདག་ལྔ་བ་ལོ་ལྔ་བ། ཏོག་གི་ཞལ་ན་ཚོཊ་ཀཀཀ་ཁལ་ཁྲཊ་ཀ་ཊཚོན་་་ཎཱ་ཏ་པ་ཏི་ལྔ་ན་ག་པ་ཆ་འདྲ་བོའི་ཤ་ཁྲགྲོན་ཁང་༽(བ། ར%@འཁོར་གསུམ་བློ་ན་མཁ་ཏང་ཀྲ་བ་།
