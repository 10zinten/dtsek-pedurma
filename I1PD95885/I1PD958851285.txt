བསྟན་འགྱུར། རྒྱུད། ༼ཆུའིལ་ཨོཾ་ཡིག་བལྟས་ཏེ་སྔགས་འདི་བཟླས་པར་བྱའོ། །ཨོཾ་ཨ་ཨཱ་ཨི་ཨཱི་ཨུ་ཨཱུ་རྀ་རཱི་་ལོ་ལཱི་ཨེ་ཨི་ཨོ་ཨོ་ཨོཾ་སྭཱ་ཧཱ། །ཞེས་སོ། །བདག་མེད་མ་རྐྱང་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བདག་མེད་མ་རྐྱང་པའི་སྒྲུབ་ཐབས་གཉིས་བཞུགས་སོ། །«པེ་»«སྣར་»ཀེ། ཚེ«པེ་»«སྣར་»+བསྣམས་པ། «པེ་»«སྣར་+བོ། «པེ་»«སྣར་»བ་ལ། །་བསྡུར་འབྲས་རེའུ་མིག། ཊ་ངཁ་ཏི་དེ་ཁས་་༽་ཏུང་རྒྱུད། (ཤ་པ། དེ་ལྔ་བཅུ་ང་ལ་ང་ཀཔའི་གདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། །རྩི་ནི། །སྡེ་ཚན། ཀརྒྱུད། ་། ་ཌོ་རྒྱུད་འགྲེལ། འ་རྒྱུད། དེ་ཁོ་ནྡྷོ་རྒྱུད། །སྤྱི་དང་། པོད་རྟགས། །བ་མུ་ཡrདུདུད་ཡརམདུ/ex*མུ་ལྟ་བ། ཚན་གྲངས། ལྡེབ། བཙ_ན་བ་ན?ནབབན་ཕྲེང་། །ལཚེག་ཁྱིམ། །་ཀྱང་ག_ལྟ། "gਲਸਥਾਮਸਲੋਂਚਪਾੜགཞུ་ཁའི་ཊ་ཀ་ཎ/ཁ་འལ་ལ་ཁོ་ཆ་། ཀ་ཎཁྱུང་རབ་ཏུ་བ་ང་ལ་བྱ་བ་ཁཁ་ཁོད་ཁ་ཟླ་བཁ་བའི། །སྐྱོ་ཎཎཱ་ཊི་(J་ཀ་ཁོ་པས་+།
