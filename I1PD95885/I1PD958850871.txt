རྒྱུད། ༼ཁུའིཔ་ཅན་བརླག་སྟེ། །སེམས་ཅན་དཔག་ཏུ་མེད་པའི་དོན་སྣ་ཚོགས་བྱེད་པའོ། །ཨོཾ་མཱ་རཱི་ཙྪ་ཧཱུཾ་སྭཱ་ཧཱ། ཞེས་པ་ནི་བཟླས་པའི་སྔགས་སོ། །མྱ་ངན་མེད་པའི་ཤིང་གི་འོད་ཟེར་ཅན་གྱི་སྒྲུབ་ཐབས་རྫོགས་སོ། །ལ་ན་། ་བརྒྱ་པའི་སྔགས་བརྗོད་ལ། པར་བྱའོ། ཏོག་ལ་ལ་ལ་སུངས་པའི་འོད་ཟེར་ཅན་དཀར་མོ་བབབཔདེཡ། རྐལ་ལྟ། །པ། །དམ༌བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»ཀཱནྟཱ་མ་རི། «སྣར་»ཀཱནྟཱ་མཱ་རི༑«པེ་»«སྣར་»—ལས། «པེ་»སྣར་བརྒྱན། ཚེ«པེ་»«སྣར་པས། «པེ་»«སྣར་»མགོན། དེ་ན་བོད་སྐད་དུ་གད་པ་དང་། 《པེ་། འོད་ཟེར་པད་མ་དངོ། །མེ་《སྦྱོར་བྱེད་བའོ། ཆུ་ལ་བསྡུར་འབྲས་རེའུ་མིག་ཁ་དན་དེ། ད་པ། ་་་བརྒྱད། །ཀྱང་། རྒྱུད། ་(ཀ་བ་ཁོ་ན་ཌཎི་ཏེ་ལ་ཏུ། དེ་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། རྩོནེ། །ཡོད་སྡེ་ཚན། ་རྒྱུད། རྒྱུད་འགྲེལ། བརྒྱུད། ་zz། །་རྒྱུད། m/brང་བོད་རྟགས། །:དུང་མུ་་དུཚན་གྲངས། །ལྡེབ། བ་—ན། བབན་_བ་། རབ་—ན། ཕྲེང་། ལ་E། ་ན་ར་་་་པཎ་::ཚེག་ཁྱིམ། .་ལ་་ཁསྤྱིpn། ཀ་ཀུ་ར་བ་སྔོ་་ཚ་ཏཚོ་ཚ། ཀ་ལppro། །ཀ་p:/་ཀའི་ཚོ་ད་ཆ་རང་བ་ཀ་ཐོ་འང་ཁོ་ཚོ་ལ་ཁ་ཚོའི་བ་པ་ཡས་པ་བོད་པཎྜི་ཏ་བློ་དེ་ཎཁོ་ན།
