ཟཔ《་ཀྱི། prང་ཀྱི། ཁའི་ཁ་དོག་ལg༽ཀྱི་སྐྱེ་ཚེ། ་ཊབསྡུར་མཆན། :ན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»༄༅། །རིགས་བྱེད་མ་དཀར་མོའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»ཤུཀླ། «སྣར་རིག«པེ་»«སྣར་»+འཕགས་མ་རིག་བྱེད་མ་ལ་ཕྱག་འཚལ་ལོ། །།«པེ་ནི། @«སྣར་པོ། @«པེ་»«སྣར་+ལས། ཉི«སྣར་»འཐུན། «པེ་»«སྣར་»སྤྲོ། ཚེ«པེ་»«སྣར་བསྲུང་། གྱི«པེ་ཀྱིས་ཆོས་ཀྱིས། «སྣར་ཀྱི་ཆོས་ཀྱིས། །«སྣར་»རྣམ་པ། ®«སྣར་»+མཆོད་པ་སྣ་ཚོགས་པའི། «སྣར་»ཏེ། «པེ་»«སྣར་»ཉ། གྱི«སྣར་»ཀྱིས། «པེ་»གྱི་བརླབ་པོ། «སྣར་»གྱི་བརླབ་པའོ། «སྣར་»གྱིས། «སྣར་»སའི། «སྣར་»+གྱིས། ཁྱེསྣར་ཀྱི། ཀྱེ«སྣར་+བ། «སྣར་»སྣམས། «སྣར་མ། «པེ་»«སྣར་གི། ཚེ«སྣར་རྒྱན། ཚེ«སྣར་ཀྱི། ཁྱེ«སྣར་»དཀྱིལ་དཀྲུངས། ལེ«སྣར་པོས། ཚེ«སྣར་»—ཁ་་དོག་ལྟ། དགྱེ«པེ་»«སྣར་བརྒྱན། ཚེ«སྣར་»ཀྱིས། །ཨེ«སྣར་བུ། QQ«སྣར་»གྱིས། «སྣར་»འཕྲོ་བར། «སྣར་»—། °«པེ་»—པའི། «སྣར་པ། «སྣར་—ག་བུར་དང་། «སྣར་བཅས། ཁྱེ«པེ་»«སྣར་རྗེ། དགྱེ«སྣར་རྩོལ། (གེ«སྣར་རྣམས། «སྣར་འབྱུང་། @«སྣར་ལོ། «སྣར་»ཏ། ︽པེ་»«སྣར་»བི། «སྣར་»སྐུར། «སྣར་»—ཆུ། ཨ«སྣར་»ཟླ་བའི། ཉི«པེ་»«སྣར་པའི། «སྣར་»+འབར་ཞིང་། 《སྣར་»གྱིས། ཀྱེ«སྣར་»བརྒྱས་པ་སྟེ། མེ«སྣར་པོ་དེ་རྩེའོ། ལེ«པེ་»«སྣར་+མམ། ཨེ«སྣར་བྱས་ཏེ། ཉེ«སྣར་གྱུར། «སྣར་»+དེ་ལ། *®«སྣར་མངགས། ན«སྣར་»གིས། «པེ་»«སྣར་»པ། ཚུ«སྣར་གྱི། @«སྣར་+པ། ཚེ«སྣར་»གིས། སྣར་ཧྲི། «སྣར་»ཧྲིཁ་དཱི། ཚེ«སྣར་བསྲེག་བསྲུངས། འཀྱེ«སྣར་བརྒྱ་བཅུས་སྦྱིན་བསྲེག《སྣར་»བླངས། «སྣར་+བ། ཨེ«སྣར་»ཀྱིས། ཨེ«སྣར་»བཞིན་དུ། ཨེ«པེ་»«སྣར་ཐགས། «པེ་»མདུན་དུ་བྱས། 《སྣར་»མདུན་དུ་བྱས་ཏེ། «པེ་»«སྣར་ནས། ཚེ«སྣར་རླགགྱི《སྣར་+པ། ཀྱེ«སྣར་»+ལྔ། འབུ《སྣར་»གསུངས། «སྣར་གཟུང་། ཚེ«སྣར་སྤང་བར། «སྣར་ཐམས་ཅད། «སྣར་ལགས། «སྣར་གས། «སྣར་»གྱིས། ཚེ«སྣར་བྱ་སྟེ། «པེ་»《སྣར་»སྔགས། «སྣར་ནས། ཁྱེ«སྣར་ཐུན། ཀྱེ«སྣར་བརྒྱད། «སྣར་+བ། ཉེ«པེ་»«སྣར་»དོ། །«སྣར་»ཀྱི། «པེ་»«སྣར་»+པ། །
