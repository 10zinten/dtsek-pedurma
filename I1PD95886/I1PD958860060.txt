རྒྱུད་དང་རྟོག་པ་རྣམས་ལས་བྱུང༌། །དཀྱིལ་འཁོར་དབང་དང་དམ་ཚིག་གསུམ། །རྒྱུད་རྣམས་ཀུན་གྱི་སྲོག་ཤིང་པས། །རྣལ་འབྱོར་གསང་སྒོར་ཞུགས་རྣམས་ཀྱིས། །:གཅེས་སྤྲས་ཡིད་ཀྱི་གཞུང་འདི་སྤྱད། །།དཀྱིལ་འཁོར་དོན་བསྡུས་རྒྱུད་ཀྱི་མན་ངག་འདི། །ནོངས་སུ་ངེས་ཀྱིས་ལྷ་ལ་བཟོད་པར་གསོལ། །འདི་བརྩམས་དགེ་བས་སེམས་ཅན་མ་ལུས་ཀུན། །དཀྱིལ་འཁོར་གསུམ་གྱི་ཀློང་དུ་གནས་གྱུར་ཅིག། སློབ་དཔོན་སངས་རྒྱས་གསང་བས་མཛད་པ། དཀྱིལ་འཁོར་གྱི་་ཆོས་མདོར་་་བསྡུས་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་སངས་རྒྱས་གསང་བས། དབས་མཉྫ་ཤྲཱི་དང༌། བྲན་ཀ་་མུ་ཏི་ཏ་གཉིས་ལ་བརྫངས་པ། བོད་ཡུལ་དུ་ལོ་ཙཱ་་བ་ཀ་བ་དཔལ་བརྩེགས་ལ་་སོགས་པས་བསྒྱུར་བའོ། །།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»རྟོག། «པེ་»«སྣར་»སྒོམ། «པེ་ལྟེར། །«པེ་»《སྣར་»ལ། «པེ་»«སྣར་»ཀྱིས། «སྣར་བླང་། «པེ་»«སྣར་ལས། «པེ་《སྣར་བརྐྱང་བ། ་«པེ་»«སྣར་»སྤགས། «པེ་»«སྣར་ཡི། །«པེ་»བཏབ། «པེ་《སྣར་»གཞུ། «པེ་»«སྣར་»ཀྱིས། «པེ་»གསུང་གདུལ། «སྣར་»གསུང་་་འདུལ༑«པེ་»«སྣར་ཐབས། «པེ་»«སྣར་བརྩི། «པེ་»«སྣར་ཐབས། «པེ་»«སྣར་སྟངས། ་«པེ་»དགུར། «པེ་»གཟུགས«པེ་»«སྣར་»རྣམ། «པེ་»«སྣར་སྤྱོང་བཅུད་སྦྱིན། བརྒྱུད་དེ་ད་«པེ་»«སྣར་»པས། འ«པེ་»བཙུན་མོའི། «སྣར་»བརྩུན་མོའི། ་«པེ་»«སྣར་»ལས་བཞི། །«པེ་»«སྣར་»ཞི། «པེ་»གཞི། «པེ་»«སྣར་སྲུངས། 《ཅོ་རྗེ། སོ«པེ་»རྟོགས། །«པེ་»«སྣར་བརྟག«པེ་»«སྣར་»རྐྱང་པ་རང་བྱུང་ཡེ་ཤེས་ཏེ། །།
