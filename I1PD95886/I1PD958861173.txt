བསྟན་འགྱུར། རྒྱུད། ༼ཚུ་རགྷཱཉྩ་མེ་ཀུ་རུ། ཨོ་ཨཱཧཱིཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། །ཁ་སམྤ་ཎ་བ་ལི། དང་རྒྱས་པ་དང་བསྲུང་བ། རབ་གནས་མཁའ་ལ་གཤེགས་པའི་གཏོར་མའོ། །།་ལོ། །།ཀུན་ཨོཾ་ན་མོ་བུདྡྷཱ་ཡ་མ་ཧཱ་ཀཱ་རུ་ཎཱ་བྷཱ་ར་བྷ་རི་ཏཱ་ཧྲི་ད་ཡཱ་ཡ། པ་རཱརྦ་ས་མ་ཏཱ་ག་སངས་རྒྱས་ཐུགས་རྗེ་ཆེན་པོའི་ཁུར་གྱིས་གང་བའི་ཐུགས་ཅན་གཞན་གྱི་དོན་མཉམ་པ་ཉིད་རྟོགས་ཏ་ཙིཏྟཱ་ཡ། ཏྲི་དྷཱ་ཏུ་ཀི་ཀ་མཱནྟ་ཡེ་སྭཱརྦ་མ་ཧཱརྦ་ཀ་ར་ཀཱ་རི་ཎེ་ཨི་དཾ་བ་ལི་དི་བྱ་པའི་ཐུགས་ཅན་ཁམས་གསུམ་པའི་སྐུ་གཅིག་པུ་རང་གི་དོན་དོན་ཆེན་པོའི་མཛད་པ་མཛད་པ། ལྷའི་གནྣཱ་བཎ་ར་སོ་པེ་ཏཾ་བུ་བྷུ་ཛུ་་སརྔ་སཀཾ༔་ནུཏྟ་རཱ་ཡཾ་དྲི་དང་མདོག་དང་རོ་དང་ལྡན་པའི་གཏོར་མ་འདི་བཞེས་ཏེ། བདག་དང་སེམས་ཅན་ཐམས་ཅད་བླ་ན་མེད་སམྱཀཾ་བོ་དྷོ་པྲ་ཏིཥྛ་པ་ཡ། ཨོཾ་ཨཱཿསུ་ག་ཏ་བཛྲ་ཏུ་ཉྩ་ཧོཧཱུྃ་སྭཱ་ཧཱ། ཤྲཱི་བུདྡྷ་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལ་རབ་ཏུ་ཁོད་ཅིག་བདེ་གཤེགས་རྡོ་རྗེ་དགྱེས་པ། དཔལ་སངས་རྒྱས་བྷཏྟཱ་ར་ཀ་བ་ལི། །།།པ་འོ། དེཀ་ཀ་ལཱཀྱ་ཚོ་ང་ཚོ་རྗེ་བཙན་གྱི་གཏོར་མའོ། །ལ་ལ་་་ཉིད་ཁྲོད་ཀུན་ཏེང་་ཨོཾ་མྲྀ་ཏྱཱ་པཉྩ་ནི་ཏཱ་རེ་ཀཱ་ལཱ་ཀཱ་ལ་མྲྀ་ཏུ་ཧ་རེ། དཱ་རི་དྲུ་དུ༔ཁ་འཆི་བ་བསླུ་བའི་སྒྲོལ་མ་དུས་དང་དུས་མ་ཡིན་པའི་འཆི་བ་འཕྲོག་མ་དབུལ་བ་དང་སྡུག་བསྔལ་གྱི་བྷ་ཡ་བྷ་ཡཾ་ཀ་རེ་ཨཀྵ་མ་ཧཱ་བྷ་ཡ་ཧ་རེ་ཁཟླཉྫ་ན་པཱ་དུ་ཀ་འཇིགས་པ་འཇིགས་པར་བྱེད་མ། འཇིགས་པ་ཆེན་པོ་བརྒྱད་འཕྲོག་མ། རལ་གྲི་དང་མིག་སྨན་དང་ལྷམ་དང་བྷ་དྲ་གྷ་ཊཱ་དི་སིདྷི་དེ་བྷ་ག་ཝ་ཏི་ཏཱ་རེ་ཨི་དཾ་བ་ལིཾ་པཉྩ་མྲྀ་ཏཾ་ཀྲྀཥྞ་གྲིཉྩ་མ་མ་བུམ་པ་བཟང་པོ་ལ་སོགས་པའི་དངོས་གྲུབ་སྩོལ་མ་བཅོམ་ལྡན་འདས་མ་སྒྲོལ་མ་། ཆུ་དང་ཡུ་ཀུན་མྲྀ་ཏྱཾནཱ་ཤ་ཡ་ནཱ་ཤ་ཡ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། ཊ་མྲྀ་ཏུ་བཉྫ་ནཱི་ཏཱ་རཱ་ནས་བདག་བ་དན་སྡུག་བསྔལ་སྒྲོལ་མ་མྱུར་མ། ན་བ་དང་འཆི་བ་བསླུ་བའི་སྒྲོལ་མའི་་།
