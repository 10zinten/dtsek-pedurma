དུ་་དཔེ་གཞི། སྡེ་དགེ། སྡེ་ཚན། རྒྱུད། །པོད་རྟགས། །ཚུཚན་གྲངས། ལྡེབ། བ་—ནཕྲེང་། །ཟླཚིག་རྐང༌། །ིཚེག་ཁྱིམ། །།ན༽ན་བ་བ། ་། ་། །།བ་—ནཡེ་الدوཞུའི་ཁཛི་སྶ། ཁ་ཏོ། །དབུ་ལ། ད་ཕྱོགས་ཀུན་ཁོ་ན་བཀའ་བློན་པ། །ཏཱཛོ་། ཚོདེ་ན་ཁྱོདོར་དུ་ཞ། ཀྲ། རྔ་ཁཚོ་བའི་གན་དུ་ས་སྐྱ་བའི་ས་ཕྱོགས། །དག་བྱེད་ཡཁབ་ལ༌ཚོ། །།་དག་ཤེད་། ཀགེལོ་ཀ་ཁ་ཁ་བ་བློ་ཀྐན། །དཔ་། ཆ་དང་ཁྲིད་ཀྱི་་་་::ཀའ་གལ་ཚོགཡས་བར་ཁ་ཊའ་ནས་མི་བཟོ་བཀོཌ། པ་ཚོ་བཁོ་ཚོཚུ་ཏེ་མམ་བྷ། །མི་ཡི་ཚེ་ལ་ལོག་པ་བསྐུ་བ་བ་ཚ། །ཆཏཚུལོ་ཀ་ཏོཔཁྲིཏ༌ཀ་ལཏ་ཀ་སྒྲུབ་པ་ནི་རྒྱན་ལ་ཀ་ཁོ་བྷ་ར་ཙ་ཏཚོཊ། །།མ་ཁྱོ་ཏན་པོའི་མགོ་ན་ཀ་ཎ་བ་ཁྲོལ་རྒྱན་གྱི་ཚོང་མགོན་ཁང་ཚུ་ནི་མ་ཁྱིམ་གྱུར་དང་། ཀྱི་དཀྱ་དང་་་ཀྱི་རང་ངོ། །བུམ་ག་ཎ་པ་ཏི་པུ་པ་ཏགཏཊི་པ། བར་བའིཔཎ་ཆེན་ཨེ་ཤི་ཚོ་ཁོ་ཚེ། w། pཔཚཚ་གཏོང་བའི་སྨཚེ། པ་པ། །བྱི་ལ་ཀ་་་་་་་་འཁོདབ་དང་ཟོ་ཆ་ལ་ཆོ་ལོ་ཌཁ་ཁཚེཚེ་གོ་ཀོ་རིའི་ཁཀོ་བྱི་ལྔ་ལ་ཏཱ་རོ་—།
