རྒྱུད། །༼ཞི། དྷཱ་ནྱ་ཀ་རི་སྭཱ་ཧཱ། ཨཱ་གཙྪ་ཨཱ་གཙྪ་ཧཱུཾ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཨུཀེ་ཏུཀྵ་བྷཀྵེ་བྷུཀེ་བནྟ་ཎི་པྲ་བནྟ་ཎི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་བ་སུ་དྷེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུ་དྷཱ་དཱིཔཱ་ཡེ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ས་མ་ཡེ་སོ་མྱེ་ས་མ་ཡངྐ་རི་མ་ཧཱ་ས་མ་ཡེ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་བ་སུ་དྷཱ་ར་ཨེ་ཧེ་ཧི་བྷ་ག་བ་ཏི་ས་མ་ཡེ་ས་མ་ཡ་མ་ནུ་སྨ་ར་སིདྡྷིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ལོ་ཀ་པཱ་ལཱ་ཡ་ན་་མསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་མཱ་ཏ་རཱི་བློ་ན་མསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་བྷཾ་གཱ་རཱི་སྦྱོ་ན་མསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་བྷཱུ་ཏི་ནི་སྦྱོ་ན་མསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ཡཀྵི་ཎཱི་བློ་ན་མསྭཱ་ཧཱ། ༀ་ཤྲཱི་གུ་རྒྱ་ས་ཀ་ཊི་ཀེ་སརྦ་དུཥྛ་ནཱ་ཀརྟ་ཡ་ཨཱ་ཀརྟ་ཡ་ཧཱུྃ་སྭཱ་ཧཱ། ལ་འཨོ་བྷིཧཱུྃ་ཙི་སསཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཛམྦྷ་ལ་ཛ་ལེནྡྲ་ཡ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཨཱནཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ་སྭཱ་ཧཱ། ༈ཨོཾ་ཤྲཱི་མ་ཎི་པདྨེ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བཛྲ་དྷ་རཱ་ཡ་ཧཱུྃ་སྭཱ་ཧཱ། །་ཨོཾ་ཤྲཱི་མ་ཎི་བྷ་དྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་བི་པུརྞ་བྷ་དཱ་ཡ་སྭཱ་ཧཱ། ་ཨོ་བི་དྷ་ན་དཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བི་ཤ་བ་ཎཱ་ཡ་སྭཱ་ཧཱ། ཨོ་བི་ཙི་པི་ཀུཎྜ་ལི་ནེ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཀེ་ལི་མཱ་ལི་ནེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མུ་ཁེནྡྲ་ཡ་སྭཱ་ཧཱ༑ཨོཾ་ཤྲཱི་ཙ་རེནྡྲ་ཡ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཕུཤཾ་ཁ་པཱ་ལ་ནཱ་ག་རཱ་ཛཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཨི་ལཱ་དེ་བྱི་སྭཱ་ཧཱ། ཨོཾ་མ་ཎི་པདྨེ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷ་ར་སཱ་ག་ར་ནིགྷོ་ཥཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷརྨེ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོ་ཏ་བྷ་དེ་བྱི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་གུཔཱ་དེ་བྱི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ས་ར་སཱ་ཏཱི་དེ་བྱི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཙནྡྲ་ཀཱནྟཱ་དེ་བྱི་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཤཾ་ཁ་ནི་དྷཱ་ནཱ་ཡ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་པདྨ་ནེ་དྷཱ་ནཱ་ཡ་སྭཱ་ཧཱ། ༀ་ཤྲཱི་ཨཊྚཾ་མ་ཧཱ་ཡཀྵི་ཎཱི་བྷོ་ན་མསྭཱ་ཧཱ། །་ཐམས་་ཅད་མཆོད་པའི་དོན་དུ་ཧཱུྃ་སྭཱ་ཧཱའོ། །ཨོཾ་བཅོམ་ལྡན་འདས་མ་དཔལ་ནོར་རྒྱུན་མ་ཡེ་ཤེས་ཀྱི་སྐུ་མངའ་བ་ལ་ཕྱག་འཚལ་ལོ། །ནམ་མཁར་འགྲོ་མ་ཐད་ཀར་མཛེས་་ཤིང་ཡིད་དུ་འོང་བ་སྐྱེ་གནས་མཉམ་པ་མེད་པའི་ཡོན་ཏན་གྱི་གཏེར་གསེར་དང་་་རིན་པོ་ཆེ་འཁྱོག་པ། སྐྱེ་བོ་མང་པོའི་ནོར་སྤེལ་བའི་རྒྱ་མཚོའི་རིན་པོ་ཆེའི་ཆར་་ལྟ་བུ། མཆོག་ཏུ་དྲི་མ་མེད་པའི་གཟི་བརྗིད་མཐའ་ཡས་མཛད་མ། ནི་རྒྱལ་བ་DIAlor།
