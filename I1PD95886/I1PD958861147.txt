བསོགས་བྱ་བ་་་་་་ནས་ལས་་་་་་། དེ་དང་། ཁང་ཚོ་ལ་ཁ་བ་བཀང་ཁཀར་ཀར་བ་དང་ག་འགག་པ། །གྲུབ་པ་བཀའ་གནང་བ་མ་ཟུབ་པ་ཚོ་ཆ་བྱ་ལ་བསྡུར་མཆན། ་ཁྱད་པར་དོགས་པ་ན། །བྱཁཱ་། གྲོལ་ཀྱི་ཞིང་་ཚ། །།།ད། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་ཆོས་སྐྱོང་ཐམས་ཅད་པའི་གཏོར་མའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་»། «པེ་»«སྣར་བརྩམ། «པེ་»«སྣར་»བརྩོམ། ས«པེ་»«སྣར་བསང༌། @«པེ་»«སྣར་འཁུར། ཕྱི་ལོ། །«པེ་»«སྣར་སྲུངས། འ«པེ་»ནམ། «པེ་»«སྣར་གཞག«པེ་»«སྣར་བསྟབ། «པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་དྷརྨཱ། «པེ་»«སྣར་དྷརྨ། «པེ་»ནུཏ་བནྣ་ཏྭཱ་ཏ་ཏཱ། 《སྣར་ནུད་་་པནྣ་ཏཱིཏ་ཏཱ། «པེ་»«སྣར་བྷ། ཙེ«པེ་»«སྣར་»། «པེ་»«སྣར་»དུཋཎ། 《སྣར་བསྐུར། ཁོ:/«པེ་《སྣར་»གསོལ། །«པེ་»《སྣར་བསྒྲུབས། «པེ་»《སྣར་»ཚད་གཅད། །«པེ་»《སྣར་»ཙ། །ཀན་ལྔ་ལ་ཟླ་བ་ཚོན་ཏེ། ཡུ་བ་ར་པ་ཀ་ཁང་ཏུ་སྦྲང་འཁོན་རྒྱུ་དང་ལས་ལོ་ལ་་་། །།ཚེ། ་་ཏུ་འཚོ་་་་་་ཁ་ཀཚོ། རྒྱུད། (བྱ་ངང་པས་། །འདཁའ་དཔེ་གཞི། སྡེ་དགེ། །།པེ་ཅིན། སྣར་ཐང༌། བཅོ་ནེ། ད་བར་འབད་སྡེ་ཚན། རྒྱུད་འགྲེལ། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བ—ན། ན་ན་། ན་ན་ན?བ་། ཨ། །།ན། འོན་ཀྱང་། གང་ལ་བསྡུར་འབྲས་རེའུ་མིག་རྒྱུད། ཕྲེང་། ཚེག་ཁྱིམ། །།
