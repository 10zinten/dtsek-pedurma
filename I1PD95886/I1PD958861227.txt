བསྟན་འགྱུར། རྒྱུད། ཚུ། ནོར་སྐྱོང་གདོན་གྱི་བདག་པོ་སྟེ། །བྱང་ཕྱོགས་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་ཅིང་བསྟོད་དེ་ཕྱག་ཀྱང་འཚལ། །དབང་དང་ལྡན་པ་འདྲེན་པའི་ལྷ། །འབྱུང་པོའི་རྒྱལ་པོ་གདོན་གྱི་བདག། དབང་ལྡན་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་ཅིང་བསྟོད་དེ་ཕྱག་ཀྱང་འཚལ། །སྟེང་གི་ཚངས་པ་ཆེན་པོ་དང༌། །ཉི་མ་ཟླ་བ་གཟའ་ཆེན་པོ། :། གཟའ་དང་རྒྱུ་སྐར་བཅས་པ་ལ། :། མཆོད་ཅིང་་་བསྟོད་དེ་ཕྱག་ཀྱང་འཚལ། །འོག་གི་ཀླུ་སྟེ་ས་ཡི་ལྷ། །ས་འཛིན་ལག་པ་ཆེན་པོ་སྟེ༑༽། དམ་ཚིག་ཤེས་པའི་ཚོགས་བཅས་ལ། །མཆོད་ཅིང་བསྟོད་དེ་ཕྱག་ཀྱང་འཚལ། །ཞེས་བསྟོད་དེ་གཏོར་མ་སྦྱིན་ནོ། །ཨོ་པ་ཤུ་པ་ཏི་ནཱི་ལ་ཀཎྜ་ཨུ་མ་པྲ་ཡཱ་ཡ་ན་མསརྦ་ཏ་ཐཱ་ག་ཏེ་བྷོ་བི་ཤྭ་མུ་ཁེ་བྷྱསརྦ་ཐ་ཁཾ་ཨུངྒ་ཏེ་སྤ་ར་ཎ་ཨི་'མཾ་ག་ག་ན་ཁཾ་གྲིདྡྷེ་དཾ་བ་ལིཾ་ཏ་སྭཱ་ཧཱ། ཨོཾ་:ས་ཧ་ཤ་ཨཀྵ་ཡེ་ཨོ་ཨཧྥེ་མ་ཧཱ་ཏེ་ཛ་ས་མ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཀརྨ་ནུ་ཤ་མ་:དྷརྨ་རཱ་ཛཱ་ཡ། ཨོ་རཱཀྵ་ས་ཨ་དྷི་པ་ཏ་ཡེ་རཀྟ་མཱ་ལ་རཀྟ་པྲ་ཡཱ་ཡ། ༀ་མ་ཧཱ་ནཱ་གཱ་སྲི་ཡ་པདྨ་དྲི་ཡ། །ཞི་:པདྨ་དྷ་རཱ་ཡ། ཨོ་ག་ག་ན་བི་ས་རི་སརྦ་སཏྭ་ཨནྟ་ཀཱ་ཡ། ཨོཾ་དུརྒ་ཧ་པྲི་ཡ་སརྦ་དེ་བ་ཏ་ལཱ་ཙེ་ཡ། ཨོ་ཨུ་དྷཾ་བྲ་ཧྥ་ཎཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་སཱུརྱ་གྲ་ཧ་ཨ་དྷི་པ་ཏ་ཡེ་སྭཱ་ཧཱ། བོ་ཨོ་ཙནྡྲ་ནཀྵ་ཏྲ་ཨ་དྷི་པ་ཏ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཨ་:དྷཱ་པྲ་བྷི་བྷུཉྩ་ཨ་སུ་རེ་:བྷྱནཱ་གེ་སྭཱ་ཧཱ་ཞེས་བྱ་བའི་སྔགས་ཀྱིས་གཏོར་མ་སྦྱིན་པར་བྱའོ། །ཤར་ཕྱོགས་རྡོ་རྗེ་ཅན་གྱི་གྱི་ལྷ་ཚོགས་འཁོར་དང་བཅས། །གཏོར་མ་ཕུལ་བ་འདི་དག་བཞེས་སུ་གསོལ། །མེ་དང་གཤིན་རྗེ་བདེན་བྲལ་སྲིན་པོའི་བདག་་། ཆུ་བདག་རླུང་བདག་ནོར་གྱི་བདག་པོ་དང་། །དབང་ལྡན་ལྷ་སྟེ་འབྱུང་པོའི་བདག་པོ་དང་། །།སྟེང་གི་ཉི་ཟླ་ཚངས་པ་ཆེན་པོ་དང༌། །ལྷ་རྣམས་ཀུན་དང་ས་ཡི་ཀླུ་རྣམས་དང་། །རི་ཡི་ལྷ་དང་གཙང་མའི་ཚོགས་བཅས་ལ། །དེ་དག་སོ་སོར་གཏོར་མ་ཕུལ་བ་འདི། །རབ་+།
