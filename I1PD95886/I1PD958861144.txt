བག་ལ་དོས་དང་གཞི་ལྔ། །ཞི་བྱ། །ཆང་ཚོད་བཀོག་ལ་བག་ལ་བྱ་ནན་། ཁཚུལ་་་་་་་་་་་་་་རྒྱུ། །སྤུ་བཏང་བ་ལ་སྐྱོ་བ་དང་ཁ་ལ་པ་བློ་གྲོཥཎཱ་ཐམ་ནའི་ལོ་རྒྱུས་དང་་་་ལ་་ཡོད་ན་མགཨབ་ཆ་བ་བཏ། ཁོཔའིལོ་སྐད། རྒྱལ་བྱེད། (ཞཁྱོ་པདཞུ། །ལག་ན་། །ལྟ་བ་ལོ་ན་ཚེ་ད་པར། ༄༅། །ཆོས་སྐྱོང་ཐམས་ཅད་པའི་ཡོན་ཏན་ལ་ཀ། ཥ་ཏཆགཏོར་མའི་ཆོ་ག་དང་བྱ་བའི་པ་ཆ། །།ཅཊ་འདུ་ག་ཚཏྟ། དེམས་་༌་ཊཀ་མ་ཏོ་ལོ། །སྐྱོན་དང་ལ་ཡི་གནང། ས་སོང་། གཏཡན་ཁརྒྱུ་དང་། གཙང་:ཁའ་སྤྲོས་གཙོ་ག་ལ་ལ་ཅུང་ཁྲོག་བློས་ཐན་ཚོ་བ་སྐྱོབ་པ་དང་དེབསམ་པ་དེ་འདྲའི་གྱི་ཚེ། །ཀ་དག་ཀྱང་་ཁྲོན་པོ་སྲུང་དུ་གཙམི་ཀ་ཚོདེ༄༅། །རྒྱ་གར་སྐད་དུ། སརྦ་དྷརྨ་པཱ་ལ་བ་ལིནྟ་བི་དྷི། བོད་སྐད་དུ། ཆོས་སྐྱོང་ཐམས་ཅད་པའི་གཏོར་མའི་ཆོ་ག་ལ༌། དཔལ་ཕྱག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། ཏད། བསོད་ནམས་ཚོགས་སོགས་བར་ཆད་ཞི། །ཕྲིན་ལས་་་་རྣམ་བཞི་འགྲུབ་པ་དང༌། པ་ལ། ཆོས་སྐྱོང་ཐམས་ཅད་མཉེས་པར་འགྱུར། །དེ་ཕྱིར་གཏོར་མའི་ཆོ་ག་རྩོམ་། །དད་དང་ལྡན་པའི་སྒྲུབ་པ་པོས། །སྙིང་རྗེ་ཆེ་ཞིང་གཞན་ཕན་དགའ། །སེར་སྣ་མེད་ཅིང་གཏོར་བརྩམས་པས། །བག་། གསང་བའི་བཏུང་བ་ལྕགས་ཀྱུ་རྣམས། །ཉེ་དང་ཁུར་བ་འབྲས་ཆན་རྣམས། །ལ་ཕུག་ལ་སོགས་བཟའ་བཏུང་རྣམས། །།བརྩོན་པས་ཉེ་བར་ཚོགས་པ་ན། །འདོད་ལྷའི་རྣལ་འབྱོར་དང་ལྡན་པས། །དབེན་ཞིང་གཙང་བའི་ས་ཕྱོགས་སུ། །དབང་ཆེན་དཀྱིལ་འཁོར་རྣམ་ལྔས་བྱུག་ག། དེ་སྟེང་པད་སྣོད་མཛེས་པའི་ནང་། །སྔར་བསགས་རྫས་ཀྱིས་ཡོངས་སུ་དགང་། །བྱང་ཆུབ་མཆོག་ཏུ་སེམས་བསྐྱེད་ལ། །\།
