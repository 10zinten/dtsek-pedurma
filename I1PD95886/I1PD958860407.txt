རྒྱུད། ༽༽བསྐུལ་བ། སངས་རྒྱས་རྣམས་དང་བྱང་ཆུབ་སེམས། །དེ་དག་ཚོགས་ལ་བདག་བསྐུལ་ན། །སེམས་ཅན་རང་དགར་མི་གཏོང་བར། །ཆོས་ཀྱིས་རྒྱུད་འབྱོངས་འཁོར་ལོ་བསྐོར། །འོ་ཅཊི་ཡོད། །ཀ་ར་ཎ། མཆོད་པ་དང་། ཞང་གསོལ་བ་གདབ་པ། ཕྱོགས་བཅུའི་སངས་རྒྱས་ཐམས་ཅད་དང༌། །བྱང་ཆུབ་སེམས་དཔའ་འདོན་མཛད་ལ། །།སེམས་ཅན་བདེ་ལ་མ་བཀོད་པར། །མྱ་ངན་མི་འདའ་བཞུགས་པར་གསོལ། །དང་ཀྱི་འཁ་བརྒྱ་བཅུ་པ་མཉེན། དང་དེ་རྣམས་ཀྱི་དགེ་བ་བསྔོ་བ། བདག་གིས་དེ་ལྟར་དགེ་སྤྱད་པས། །ནམ་མཁའ་ཁྱབ་པའི་སེམས་ཅན་རྣམས། །དགེ་བ་འདི་ཡི་བྱིན་རླབས་་ཀྱིས། །ཕ་རོལ་ཕྱིན་དྲུག་མཐར་ཕྱིན་ནས། །ཐམས་ཅད་བྱང་ཆུབ་ཆེན་པོར་ཤོག། གལ་ཏེ་བྱང་ཆུབ་མ་ཐོབ་པར། །ཐེག་ཆེན་ཆོས་ལ་སྤྱོད་པ་དང༌། །འགྲོ་དོན་མི་འདོར་འཁོར་བ་ལས། །།འགྲོ་རྣམས་མྱུར་དུ་སྒྲོལ་བར་ཤོག། མདོ་གཞན་ལྟར་ཡང་གདབ་བོ། །འོ་ཎའི། །ཆེ་བ་ལྔར་ཅིཊ་ཡ། ཏཋདདའི་འཚོལ་བ། བདག་་་ར་འཚོ་ཁའོ། །ཏིལ་ལ་ཏནི་ཝེད་ཟཚེདངོས་ལ་གཉིས་ཏེ། སྨོན་པ་དང་འཇུག་པའོ། །།སྨོན་པ་ནི་སློབ་དཔོན་དགོངས་སུ་གསོལ། བདག་མིང་འདི་ཞེས་བགྱི་བ་དུས་འདི་ནས་བཟུང་ནས་སྙིང་་པོ་བྱང་ཆུབ་ལ་མཆིས་ཀྱི་བར་དུ་སྔོན་གྱི་སངས་རྒྱས་བཅོམ་ལྡན་འདས་རྣམས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་བསླབ་པ་ལ་རྗེས་སུ་སློབ་བོ། །རྗེས་སུ་སྒྲུབ་བོ། །སྔོན་གྱི་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་འགྲོ་བ་རྣམས་ལ་ཇི་ལྟར་ཕན་པ་དེ་ལྟར་བདག་ཀྱང་ཕན་པར་བགྱིད་དོ་ཞེས་ལན་གསུམ་བརྗོད་དོ། །།ང་ཏི་ਨੂੰਸਮੈਸਟਰਸਿੰਬਈ'ਚਈਏਸਣདངོས་འཇུག་པ་ནི། སློབ་དཔོན་དགོངས་སུ་གསོལ། །དེ་བཞིན་་་—།
