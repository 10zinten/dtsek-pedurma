བསྟན་འགྱུར། །རྒྱུད། ཚུ། དག་གིས་དམ་ཚིག་དང་སྡོམ་པ་མཐའ་དག་ཚུལ་བཞིན་བསྲུངས་ལ། །རྒྱུད་དང་། སྒྲུབ་ཐབས་དང་མཐུན་པར་དགོངས་པའི་གནས་སུ་ནམ་ལྷུན་གྱིས་གྲུབ་པའི་བར་དུ། རྡོ་ཤིང་ངམ་རི་དྭགས་ཤི་བའི་རོ་བཞིན་དུ་གནས་ཏེ་བསྒྲུབ་པར་བྱའོ། །བདག་་་ཉིད་དྲན་གསོ་རྣལ་འབྱོར་པ་ལ་ཕན་གདགས་ཕྱིར། །རྒྱུད་དང་སྒྲུབ་ཐབས་དག་ལས་དམ་ཚིག་མདོར་བསྡུས་པས། །འཕགས་དང་མཁའ་འགྲོའི་ཚོགས་ལ་གནོང་སྟེ་བཟོད་པར་གསོལ། །དེས་ནི་འཇིག་རྟེན་གསུམ་བཅས་རྡོ་རྗེ་འཛིན་ཐོབ་ཤོག། དབང་པོས་མཚན་:པའི་ཡུལ་གྱི་རྩ་ན་གནས་པའི་སློབ་དཔོན་ཛྙཱ་ན་བོ་དྷིས་མཛད་པ༑་ལས་དང་པོ་པའི་དམ་ཚིག་མདོར་བསྡུས་པ་རྫོགས་སོ། །ཁ་ཀོ༎་བདེ་རྒྱ་གར་གྱི་མཁན་པོ་ཀ་མ་ལ་གུཔྟ་དང་། ལོ་ཙཱ་བ་བསོད་ནམས་རྒྱལ་་་་བས་བསྒྱུར་བའོ། །་་ག་ང། ་ཀྱི་ཁཀོལ་ཀ་ཊོ་ཁོ་པ་ཚོའི་ཁ་ལ། ན་ལོ་ཞཀྱང་ཡོན་ཏུ་ཁང་ཀ་ཀ་ཀོ། ཎ་ལན་གདབ་པ་ལ། ནང་ཆེན་ཏེ༌ਮਿਲਪਾਇਲਲਈਸਲਮਾਨਤੇਸਰਕਾਰਸਮਝਾਇਲਕਰਨਵਿਚਸਮਰབསྡུར་མཆན། ཚེ་སྐཁ་འཁོ། ཀའདུ་གལ་འགའ་ལ་དྲ་བ་ལས་ཁོ་ན་གཏེ་པོ་ཀ་ཁ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»«སྣར་»༄༅། །ལས་ས་དང་པོ་པའི་དམ་ཚིག་མདོར་བསྡུས་པ་ཡེ་ཤེས་བྱང་ཆུབ་ཀྱིས་མཛད་པ་བཞུགས་སོ། །ད«པེ་《སྣར་—པ། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་པའི། «པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་རྫུན། །་«པེ་»«སྣར་»ས་ཏཱས། 《སྣར་»པའི། «པེ་»«སྣར་»སོ། ས«སྣར་»བཟུང་། «པེ་»«སྣར་»ཡིས། «སྣར་བཟུང༌། །ཀུན་ལ་དགེ་སར་ཀླད་ppཀྱི«པེ་»«སྣར་»བསྒྲལ། འི«པེ་»«སྣར་ནི། །«པེ་»«སྣར་»གི། «པེ་»«སྣར་»གཞོམ། ས«པེ་»«སྣར་གཟུང༌། པོ«པེ་»«སྣར་»+དུ། «པེ་»འཁྲུ། («པེ་»《སྣར་»གྱིས། «པེ་»«སྣར་ན། (ན«པེ་»—པའི། ཚ«པེ་《སྣར་»བོ་དྷི། ས«པེ་«སྣར་»བདེ། «པེ་»«སྣར་ཀྱི། @«པེ་བསྲུངས། «པེ་»«སྣར་+ནི། «པེ་»འཚལ། ཚེ《སྣར་བརྟེན། «པེ་»གསུངས། ་.«པེ་»《སྣར་དེ། «པེ་《སྣར་»བདེར། །«པེ་»《སྣར་»བསྔགས། ་་་།
