རྒྱུད། ་ཚུཏི་'གྱི་ལུས་དཔལ་ལྡན་རྒྱལ་བའི་དབང་པོ་མྱུར་འགྲུབ་ཤོག་པ་། ཚུལ་ཁྲིམས་དང་སྡོམ་པ་དང་དམ་ཚིག་མི་འགལ་བ་ཞེས་བྱ་བ། སློབ་དཔོན་མཁས་པ་ཆེན་པོ་ཡེ་ཤེས་་དཔལ་བཟང་པོའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།་ཁཀྱིས་ཕྱོཊ་::ཁ་ཆེའི་མཁན་པོ་ཛྙཱ་ན་ཤྲཱི་བྷ་དྲ་དེ་ཉིད་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་་སློང་བློ་གྲོས་སྙིང་པོ་དང་། གྲུབ་འི་དགེ་སློང་རབ་ཞི་བཤེས་གཉེན་གྱིས་བསྒྱུར་ཅིང་་་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །དང་པ་ལ། ཚ་ཁོ་བ་སེ་ཁོང་ཚོའི་མི་ཁ་ས་ར་མ་ཆེ་ལ་གནང་བ་ང་ལ་བྱ་བ་གསུམ། །ཕྱི་ལོ་/རྒྱུ་ལ། གྲོང་པའི་ཆུ། ?། སྐུ་n:epའོ་ཆུsཨ་པའི་ཚps:། དmp། ་་ན་ཏན་བ་ན་དུ་བ་མང་། །།ཚབསྡུར་མཆན། ་ཟེར་s/ི་ཁྱུ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»《སྣར་ཚུལ་ཁྲིམས་་་དང་སྡོམ་པ་དང་དམ་ཚིག་མི་འགལ་བ་ཡེ་ཤེས་དཔལ་གྱིས་མཛད་པ་བཞུགས་སོ། །«པེ་《སྣར་བསྲུངས། །བ་«པེ་»«སྣར་བརླབས། །རཔ:་་་་་་་་་བུ། ན་གཞི་བྱས་པ། །ལྷ་«པེ་»«སྣར་གནས། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»ཟློག«པེ་»རིགས། །«པེ་»《སྣར་»སྒོམ། «པེ་»«སྣར་»བཏང་། ་་་པ་ཚབ་ལ་(«པེ་»«སྣར་ཡི། ་«པེ་»བརྟེན། «པེ་»སློང༌། ང་། །ཚེ་«པེ་»«སྣར་པར། «པེ་»«སྣར་»སྒྲུབ། །«པེ་»སྟེ། །དེས་བསྒྲུབ་འདི་འགྲུབ། «སྣར་»ཏེ། །དེས་བསྒྲུབ་འདི་བསྒྲུབ། (.«པེ་»«སྣར་ཡི། བ་ལུ་«པེ་»སྣབས། 《སྣར་མནབས། («པེ་»«སྣར་»སྤྱད་པའི། །སྲིད་ན་གད་ལྟའི། བོད་«པེ་»«སྣར་»ཀྱི། .«པེ་»«སྣར་བསྲུང་། «པེ་»«སྣར་»དུ། ཚེ«པེ་ཡི། །«པེ་»«སྣར་»གིས། །«པེ་»གི། «སྣར་»གི་ནི། «པེ་»«སྣར་—དོན། «པེ་»«སྣར་»གི། ༄༅«པེ་《སྣར་བརྟན། ས«པེ་»«སྣར་བརྟེན། པོ«པེ་»«སྣར་»བཏང་། /«པེ་»«སྣར་ལ། «པེ་»«སྣར་བཏང་། «པེ་»«སྣར་»རྣམས། «པེ་»«སྣར་རིག་པརབསམ་«པེ་»«སྣར་གྲུབ། ་«པེ་»«སྣར་»ཙ། :«པེ་《སྣར་པ། ་བའམའ་/J་ལ་རྗེ་དང་བཀའ་ཁྲིད་ལྟ་ཁྱjཁཞེ་ཟླ་། །སྐུ་ལས་ར་བ་དང་ཚོ་ཟླ་ནག་པ་བ་དང་། —།
