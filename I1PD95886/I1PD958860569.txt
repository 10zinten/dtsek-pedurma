རྒྱུད། རྩེགཅན་ལ་ཡང་གྲུབ་ལ། །དེ་རྟོགས་པའི་ཐབས་ལ་ཡང་གྲུབ་པ་དེ་སྟོན་པར་བྱེད་་་་པའི་ཚིག་ལ་གྲུབ་སྟེ། གཅིག་གི་མིང་གཅིག་ལ་བཏགས་པའོ། །ཀུན་རྫོབ་ཀྱི་སྔགས་ནི་དེ་ཐོབ་པའི་ཐབས་ཏེ། །དེ་ཡང་སྔགས་ཀྱི་གཟུངས་དང་། སྔགས་ཀྱི་མིང་དང་། ་བརྡ་དང་། ཚིག་ལ་སོགས་པ་ཐམས་ཅད་དོ། ཞ། རྒྱུད་ཅེས་བྱ་བ་ནི་འབྲེལ་པའི་ཕྱིར་རྒྱུད་དེ། །དེ་ཡང་གཞི་དང༌། ཐབས་དང་། ལའབྲས་བུ་རྣམ་པ་གསུམ་འབྲེལ་པའི་ཕྱིར་རོ། །དབྱེ་བ་རྣམ་པ་གཉིས་ཏེ། ཚིག་གི་རྒྱུད་་། དང༌། །དོན་གྱི་རྒྱུད་དོ། །ཚིག་ནི་སྒྲའོ། ཁཟླ་། དོན་ལ་རྣམ་པ་གསུམ་སྟེ། གཞིའི་རྒྱུད་དང་། ཐབས་ཀྱི་རྒྱུད་དང་། འབྲས་བུའི་རྒྱུད་དོ། །དེ་ལ་:གཞིའི་རྒྱུད་ནི་བདེན་པ་གཉིས་ཀྱི་རང་བཞིན་དུ་གནས་པའོ། །ཐབས་ནི་སྒྲུབ་པའི་རིམ་པ་གཉིས་སོ། །འབྲས་བུ་ནི་སྐུ་གཉིས་ཏེ། ཌ་ཆོས་ཀྱི་སྐུ་དང་། གཟུགས་ཀྱི་སྐུའོ། །དེ་བས་ན་གཞི་ལ་བརྟེན་ནས་ཐབས་བསྒོམས། འབྲས་བུ་ཐོབ་པས་ན་འབྲེལ་པའི་ཕྱིར་རྒྱུད་ཅེས་དེ་ལ་བྱའོ། །བསྐྱེད་པ་ནི་ཟླ་བ་ལ་སོགས་པ་བསྐྱེད་པས་ན་བསྐྱེད་པའོ། །།གཞན་ནི་རིམ་པའོ། །རྫོགས་པ་ནི་སྟོང་པ་ཆེན་པོའི་ཡེ་ཤེས་ཏེ། དེ་ནི་ཐབས་ལ་རྫོགས་པར་ཡོད་པའམ། ཡང་ན་དེ་རྫོགས་པས་ཡོན་ཏན་རྫོགས་པའི་ཕྱིར་རྫོགས་པའོ། །གཞན་ནི་རིམ་པའོ། །མ་ཆགས་པར་གཏོང་བས་ནི་གཏོར་མའོ། །རྟོག་པ་ལ་སོགས་པ་སྲེག་་ཅིང་གང་་། འདོད་པའི་དོན་སྦྱིན་པས་ན་སྦྱིན་སྲེག་གོ། ལོ་། མཆོད་པའི་ཡུལ་དུ་གྱུར་པས་ན་སཱཙྪའོ། །ཁྱད་པར་དུ་གྱུར་པས་ན་རབ་ཏུ་གནས་པའོ། །ཡོན་ཏན་ཐམས་་ཅད་འདུས་པས་ན་དགོས་པ་བྱེད་པའི་ཕྱིར་བུམ་པའོ། །དགའ་བར་བྱེད་པས་ན་མཆོད་པའོ། །ཡོན་ཏན་བརྗོད་པས་ན་ེབསྟོད་པའོ། །གཏོར་མ་ནི་གཉིས་ཏེ༑འབྱུང་པོ་སྤྱིའི་གཏོར་མ་དང་། སོ་སོའི་གཏོར་མའོ། །སྦྱིན་སྲེག་ལ་་།
