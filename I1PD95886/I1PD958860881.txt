རྒྱུད། ཚུ་དེ་ནི་མུ་ཁཾ་?བྷ་མ་རི་སྭཱ་ཧཱ་ཞེས་བྱ་བས་སོར་མོ་ལྔ་ལ་ལན་རེ་རེ་དང༌། ཁ་དང་་་་ལག་པ་བྱིན་གྱིས་བརླབ་པ་ལ་གཉིས་ཏེ་བདུན་གྱིས་བྱིན་གྱིས་བརླབ་བོ། །ལ། དེའི་འོག་ཏུ་ནག་པོ་ཆེན་པོ་མཛེས་པའི་སྐུ་ཡོངས་སུ་རྫོགས་པ་ཞིག་མདུན་གྱི་:པདྨའི་་་་རིན་པོ་ཆེའི་སྣོད་དུ་ཞུགས་ཏེ། ལོ་ལྔ་རང་གི་ཐུགས་ཀ་ནས་ཅི་:འདྲ་བའི་སྐུ་གཟུགས་ཕྱུང་སྟེ་བསྟིམས་ལ། ཀརིན་པོ་ཆེའི་བུམ་པ་རྒྱན་ཐམས་ཅད་ཀྱིས་སྤྲས་པ་ཡོངས་སུ་གང་བས་ྋརྒྱུན་མི་འཆད་པས་ཆུ་སྦྱིན་པར་བྱའོ། ་ནས། འདི་ལྟར་བྱ་སྟེ། ཨོ་ཛྷེ་སྭཱ་ཧཱ་ཞེས་བྱ་བས་་རིན་པོ་ཆེས་བརྒྱན་པའི་ལག་པ་ནས་ཞལ་དུའོ། །ཞལ་ནས་རྡོ་རྗེ་ནོར་བུའི་རྩེ་མོ་རུའོ། འོ་། དེ་ནས་སླར་ལག་མཐིལ་དུ་འབབ་པའི་རིམ་པས་མཐར་ཆུའི་བདག་པོའི་ཞལ་དུ་འཇུག་པར་བསམས་ཏེ། །ཆོ་ག་དེ་ཉིད་ཀྱིས་ཁྱོར་བ་བརྒྱ་རྩ་བརྒྱད་འཁྲུལ་པ་མེད་པར་སྦྱིན་པར་བྱའོ། །དེའི་ཚེ་ཕྱག་གི་ལྟམཐིལ་གྱི་བུམ་པ་དང་། །ང་ནེའུ་ལེ་དང་ལྟོས་འགྲོ་རྣམས་ཀྱི་ཁ་དང་སྣའི་སྒོ་ལ་་་སོགས་པ་བུ་གའི་རྣམ་པ་ཐམས་ཅད་ལས་རིན་པོ་ཆེའི་རིགས་མི་འདྲ་བ་སྡུག་ཅིང་་ཡིད་དུ་འོང་བ་རྣམ་པ་དུ་མའི་ཆར་ཆེན་པོ་ཕབ་པས་གནས་ཁང་ལ་སོགས་པ་་་མཐའ་དག་:ཡོག་ཅིང་ནོར་གྱིས་གང་བར་གྱུར་པར་བསམ་མོ། །དེའི་རྗེས་ལ་ཡི་གེ་བཅུ་པ་ལ་མཆོད་པ་:རྣམ་པ་ལྔའི་སྔགས་སྦྱར་བས་མཆོད་ཅིང༌། ་ཚེ་ཐུན་མོང་དང་ཁྱད་པར་གྱི་སྒོ་ནས་བསྟོད་དོ། །སྔགས་དེ་དག་ཉིད་ཀྱིས་གཏོར་མ་བྱིན་གྱིས་བརླབས་ཏེ་དབུལ་བར་བྱའོ། །།ན་མོ་རཏྣ་ཏྲ་ཡཱ་ཡ། ་ཨོ་མ་ཎི་བྷ་:དཱ་ཡ། མ་ཧཱ་ཡཀྵ་སེ་ནཱ་པ་ཏ་ཡེའོ། །ཨོཾ་ཛམྦྷ་ལ་ཛ་ལེནྡྲ་ཡེ་སྭཱ་ཧཱ། །ཞེས་ལན་བདུན་ནམ་བརྒྱ་རྩ་བརྒྱད་ཀྱི་བར་དུ་བཟླས་ཏེ་ཇི་ལྟར་འདོད་པའི་དོན་ལ་གསོལ་བ་གདབ་པོ། །དེ་ནས་ཡང་མཆོད་པས་གཏང་རག་ཕུལ་ཏེ་མེ་ཏོག་དབུ་ལ་བཞག་ནས་ཡི་གེ་བརྒྱ་པས་ལྷག་ཆད་ཁ་བསྐང་ལ་རྟེན་སྤྱི་བོ་དང་ཆུའི་ཕུད་སྲིན་ལག་གིས་རེག་སྟེ་ལྕེ་ལ་།
