ཚོད་བསྟན་འགྱུར། ་རྒྱུད། ༼ཅུ། ཌཱ་ཎཁྱོབ་ཀ། གཙང་མའི་གནས་ནི་རྣམ་སྤངས་ནས། །འོག་མིན་གནས་ནི་ཉམས་དགའ་བར། །ཡང་དག་སངས་རྒྱས་དེར་སངས་རྒྱས། །སྒྲུབ་པ་པོ་དག་འདིར་སངས་རྒྱས། །ཞེས་གསུངས་པ་ཡིན་ནོ། །སྤྲུལ་པའི་སྐུས་ཀྱང་འཁོར་བ་ཇི་སྲིད་མ་སྟོངས་ཀྱི་བར་དུ་འཇིག་རྟེན་གྱི་ཁམས་མཐའ་ཡས་པར་གདུལ་བྱ་རྣམས་ཀྱི་མོས་པ་དང་། སྐལ་བ་ཇི་ལྟ་བ་བཞིན་དུ་སྤྲུལ་སྐུ་འདྲ་བ་དང་། ཌ་མི་འདྲ་བ་དང་། །ངེས་པ་དང་། མ་ངེས་པ་དང་། རྟག་པ་དང་། ཚམི་རྟག་པས་དོན་མཛད་ཅིང་བཞུགས་པ་ཡིན་ནོ། །དེ་ཡང་ཇི་སྐད་དུ་རྡོ་རྗེ་དྲ་བ་ལས། ལ་གང་དང་གང་གི་གཟུགས་ཀྱིས་ནི༑། མི་རྣམས་ཡིད་ཀྱིས་འཇུག་འགྱུར་བ། །དེ་དང་དེ་ཡི་གཟུགས་ཀྱིས་ནི། །སྣ་ཚོགས་ནོར་བུ་ཇི་བཞིན་ནོ། གློ་། ཞེས་གསུངས་པ་ཡིན་ནོ། ཁྱོ། དེ་ལྟར་ན་སྐུ་་་གསུམ་གྱིས་ལམ་ཐམས་ཅད་ཀྱི་འབྲས་བུར་གྱུར་པ་བསྡུས་པ་ཡིན་ཏེ། ང་ཚོ་བདག་དང་གཞན་གྱི་དོན་ཕུན་སུམ་ཚོགས་པའི་རབ་ཀྱི་མཐར་ཐུག་པའི་ཕྱིར་རོ། །དེ་ཡང་ཇི་སྐད་དུ། །ལུས་དང་ངག་སེམས་རྣམ་དག་པས། །སྐུ་གསུམ་དུ་ནི་བསྟན་པ་ཡིན། །ཞེས་གསུངས་པ་ཡིན་ནོ། །ཁོ་ཚེ་འོ། །དུས་ཀྱི་ཤེལ་སྒྲོལ་དུ་ཀ་ཀཚོ། ན་མ་ན་མ་ར་ཁང་དང་གསང་བ་བྱ་བ་བྱུང་བདུན་པ་དང་:། ཊདེ་ལྟར་བདེན་པ་གཉིས་ཀྱི་མཚན་ཉིད་ཤེས་ཤིང་འབྲས་བུ་སྐུ་གསུམ་གྱི་སྤྱིའི་རང་བཞིན་རྟོགས་ན་དེ་ལ་མངོན་པར་འདུན་ཅིང་དེའི་ཐོབ་པའི་ཐབས་ལ་འཇུག་པར་འགྱུར་རོ། །དེ་བས་ན་དེའི་རྒྱུར་གྱུར་པ་ནི་སྔགས་ཀྱི་ལམ་རྣམ་པ་གཉིས་སུ་འདོད་དེ། །སྨིན་པར་བྱེད་པའི་ལམ་དང༌། གྲོལ་བར་བྱེད་པའི་ལམ་མོ། །སྨིན་པར་བྱེད་པའི་ལམ་ལ་ཡང་བཞི་སྟེ། ཡབོད། བུམ་པའི་དབང་བསྐུར་བ་དང་། །གསང་བའི་དབང་བསྐུར་བ་དང་། །ཤེས་རབ་ཡེ་ཤེས་ཀྱི་དབང་བསྐུར་བ་དང་། བཞི་པའི་དབང་བསྐུར་བའོ། །དེ་ཡང་དཔལ་གསང་བ་འདུས་པའི་རྒྱུད་ཕྱི་མ་།
