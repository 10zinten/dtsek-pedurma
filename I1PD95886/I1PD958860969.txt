ད་ཕོ་བྲང་སྐྱིད་ཀྱི་བྱང་མཁར་དུ་ལྷ་བླ་མ་ཞི་བའི་བློ་གྲོས་ཀྱི་ལུང་གིས། རྒྱ་་་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཨ་བྷི་ཡུཀྟ་དེ་བ་ཤྲཱི་དང་། ལ་བོད་ཀྱི་ལོ་:ཙཱ་བ་བནྡེ་དྷརྨ་ཡོན་ཏན་གྱིས་ཇི་ལྟར་རྙེད་པར་བསྒྱུར་བའོ། །ཉཁཚོས་དེ་ལྡན་ཏན་་་:འདི་ནི་གཞུ་མིན་རི་དྭགས་མ་ཡིན་ཕག་གི་ཕྲུག་གུ་མིན། །ཡོངས་སུ་་་རྫོགས་པའི་ཟླ་བའི་གདོང་ཅན་མཛེས་མ་མ་ཡིན་ཏེ། །སྤྲུལ་པ་འདི་དག་ཐམས་ཅད་དོན་གཉེར་སྐྱེ་བོའི་འཇིག་རྟེན་དུ། །རི་ཁྲོད་ལྷག་པའི་རྒྱལ་པོ་རི་བོ་དཔེ་ཡི་མཆོག་ལ་ཡང་དག་བཞུགས། །།།ག་ཏེ་ག་ཎཏཏེ་ད་ལྟའི་བླ་ਰਾਸਟਸਟੋਨਸਿੰਘਨੂੰਸ਼ਹਿਰੋਂਲਾਲੇਬਖ਼ਸ਼། །ཏེ་དེས་རབ་དང་ལྔ་དང་པ་ལ་ལ་འདང་བ་ཡིག་བཅཡིན་པའི་དྲི་བ། འཛམ་བསྡུར་མཆན། ད་ཚུན་གྱི་བ་དང་། ནམ་ནམ་ཚ། ཚེ་ཆོ་གའི་་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་»པོ་ཏ་ལར་འགྲོ་བའི་ལམ་ཡིག་«པེ་»«སྣར་»པོའི། «པེ་»«སྣར་»། «པེ་»«སྣར་བ། །«པེ་»«སྣར་ཨ། «པེ་»«སྣར་ཨཱ། «པེ་»«སྣར་ཝ་ཏེ། «པེ་»«སྣར་»བི། «སྣར་»པུརྦ། («པེ་»«སྣར་»པིཉྩ། «པེ་»《སྣར་ཨཱ། །«པེ་»«སྣར་»དྷི་ཤ«པེ་+ཤ«པེ་»«སྣར་སཱ་དྷ་ནི་པཱ་དཾ་བནྡྷ། ཨེའི་«པེ་»«སྣར་»སིདྷ། «པེ་»«སྣར་»དཾ། («པེ་»«སྣར་»པྲ། «པེ་»«སྣར་པྲིཀྵ་དཾ་བན་མུ་ལ། སུ«པེ་»«སྣར་དཾ། པོ«པེ་»«སྣར་»བི། «པེ་»«སྣར་+རྫོགས། ཉེ«པེ་»«སྣར་—གྱི། དགྱེ«པེ་»«སྣར་བཟླས་པར། «པེ་»«སྣར་ནི། ེ«པེ་»«སྣར་»ཏ། «པེ་»«སྣར་པ་དེ་རྣམས། «པེ་»《སྣར་»བཏང་། «པེ་»«སྣར་»ཏ། བྱ་ཛི་ན་བ་གནས་པ་:«པེ་»བགྲོ་བ། «སྣར་འགྲོ་བ། «པེ་»«སྣར་དུམ་བ། «པེ་»«སྣར་བཟུང་། «པེ་»《སྣར་ཡང་། «པེ་»«སྣར་»རྩེགས། «པེ་»«སྣར་བཟླས་པར། ྋ«པེ་»«སྣར་»+བ། «པེ་»ག《སྣར་»རྒལ། «པེ་»«སྣར་རྟ། ལེ«པེ་»《སྣར་»ས། ཀྱེ«པེ་»«སྣར་སྤྱི། འབུ«པེ་»«སྣར་»ཤ་ཀོས་ཊ། ཉེ«པེ་»«སྣར་»+ཤིང་། «པེ་»«སྣར་བཟུང༌། ཨེ«པེ་»«སྣར་བྱ་བའི། །ལེ«པེ་»«སྣར་»ཏེ། འགྲོ་(«པེ་»«སྣར་»བསྒྲུབ། Q«པེ་»«སྣར་»འདི་ཡིས་སྤྱི་བོ། «པེ་»«སྣར་བཟླས་པར། «པེ་»《སྣར་»རྒོད། «པེ་»«སྣར་»སྐྱུ་རུ་རའི་ནགས་མཐོང་བར་འགྱུར་ཏེ། «པེ་»«སྣར་བཟུང་། །
