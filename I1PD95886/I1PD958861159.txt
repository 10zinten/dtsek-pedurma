བསྟན་འགྱུར། རྒྱུད།  ོ༼ཚུ་ན་ར་ཀཱ་ཀ་ར་ག་ཆུཊ་་་་་་་། མི་ཚེ་ལ་ས་ཆུ་དང་ནགས་ཀླུ་དབང་བྱེ་བྲག་ས་ཡ། ཨོ་བཛཱ་རལླི་ཧོཛཧཱུྃ་བཾ་ཧོབཛྲ་ཌཱ་ཀི་ནྱས་མ་ཡ་ཧཱུཾ་དྲྀ་ཤྱ་ཧོ། བཛཱཉྫ་ལྟ་རྡོ་རྗེ་རྩིབས་འཕུར་རྡོ་རྗེ་མཁའ་འགྲོ་མ་རྣམས་དམ་ཚིག་ཉིད་ལ་ལྟོས། རྡོ་རྗེ་ཐལ་མོ་ཁ་ཨུཛྷ་བི་ཀ་ཙ་ཡཱ་བ་ལཾ་ད་ཏུཏྲ་ནི་ཤཱདྷ་ཀེ། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་སརྦ་ཡཀྵ་རཱཀྵ་གྱེན་དུ་ཕྱེ་བས་མཚན་ཕྱེད་ན་གཏོར་མ་སྦྱིན་པར་བྱའོ། །གནོད་སྦྱིན་དང་སྲིན་པོ་དང་ས་བྷཱུ་ཏ་པྲ་ཏ་པི་ཤཱ་ཙོནཱ་དཱ་པསྨཱ་ར་ཌཱ་ཀ་ཌཱ་ཀི་ནཱ་ད་ཡ་ཨི་དཾ་བ་ལིཾ་ཤྲཱིཉྩནྟུ་འབྱུང་པོ་དང་ཡི་དྭགས་དང་ཤ་ཟ་དང་སྨྱོ་བྱེད་དང་བརྗེད་བྱེད་དང་མཁའ་འགྲོ་དང་མཁའ་འགྲོ་མ་ལ་སོགས་བ་དང་། ས་མ་ཡ་རཀྵནྟུ་མ་མ་སརྦ་སིདྡྷཾ་པྲ་ཡཙྪནྟུ། ཡ་ཐི་ཐཾ། ཡ་ཐེབྦཾ། པ་ཐམས་ཅད། དམ་ཚིག་སྲུངས་ཤིག། བདག་ལ་དངོས་གྲུབ་ཐམས་ཅད་སྩོལ་ཅིག། །བྷུཉྫ་ཐ་ཁཱ་ད་ཐ་པ་བ་ཐ་ཛི་སྒྲ་ཐ། མཱ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ་ཀཱ་ར་ཏ་ཡ་སཏུ་བདག་ལ་རྣམ་པ་ཐམས་ཅད་དུ། ལཁ་བི་བྲྀདྡྷ་ཡེ་ས་ཧཱ་ཡ་ཀཱ་བྷ་ཝནྟུ་ཧུཾ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ་ཙཀྲ་སཾ་བ་ར་དམ་པའི་བདེ་བ་རྣམ་པར་འཕེལ་བའི་ཆེད་དུ་སྟོང་གྲོགས་མཛོད་ཅིག། འཁོར་ལོ་སྡོམ་པའི་བ་ལི། །_། ་ཀ་ལྕི་ཏཱ་པ་ཏ་ལ་ཡི་ན་ཚ། གཏོར་མའོ། ལམ་། བླ་མ་ང་ཉ་ཏའ་ལྔའི་ཀཋ་ཌཏུ་ཁོཊ། དེན་བྲལ་ལ་འདུད། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཀྱི་རཀྟ་ནེ་ཏཱ་ཡི་མ་ཧཱ་ཙཎྜ་བཛྲ་པཱ་ཎ་ཡེ་མ་མ་སརྦ་ཤ་ཏུ་ཎཱ་་བཅོམ་ལྡན་འདས་མ་སྤྱན་དམར་མ་ལ་ཕྱག་འཚལ་ལོ། །ཕྱག་ན་རྡོ་རྗེ་གཏུམ་པོ་ལ་བདག་གི་དགྲ་སརྦ་ག་ཏཱ་ཎི། སཾ་ཀོ་ཙ་ཡ་སཾ་ཀོ་ཙ་ཡ། བྷཉྫ་ཡ་བྷཉྫ་ཡ། མོ་ཊ་ཡ་མོ་ཊ་ཡ། ཐམས་ཅད་ཀྱི་ལུས་རྣམས་ཐམས་ཅད་ཡང་དག་པར་སྐུམས་སྐུམས། །ཀྱི་ཀོ་ཊ་བྷ་ག་ཝ་ཏི་བཛྲ་ཤྲཱི་ཁ་ལེ་བཛྲ་པཱ་ཎི་ནཱ་ཀ་ཙུརྨ། ཨཱ་ཀརྟ་ཡ་ཨཱ་ཀརྟ་ཡ་ཙཀཱརྦནྡྷ་བཅོམ་ལྡན་འདས་མ་རྡོ་རྗེ་ལྕགས་སྒྲོག་མ། ཕྱག་ན་རྡོ་རྗེ། ལྷ་ཡུལ་པ་ཕྱེ་མར་གྱིས་ལ་།
