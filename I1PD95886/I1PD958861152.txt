བིགྷཱནྟ་ཀ་། ཨ་ཙ་ལ། ཊཀྐི་རཱ་ཛ། །ནཱི་ལ་དཎྜ། }ཆམ་ཧཱ་བ་ལེ་སྦྱ། བགེགས་མཐར་བྱེད། མི་གཡོ་བ། འདོད་པའི་རྒྱལ་པོ། དབྱུག་སྔོན། སྟོབས་པོ་ཆེ་རྣམས། ས་པ་རི་བཱ་རེ་བྷུ། ཨི་དཾ་བ་ལིཾ། གནྡྷཾ། པུཤྲཱི། དྷཱུ་པཾ། དཱི་པཉྩཀྵ་ཏཾ་ད་དཱ་མ་ཧེ། འཁོར་དང་བཅས་པ་རྣམས་འདི་གཏོར་མ། དྲི་མེ་ཏོག་བདུག་པ། མར་མེ་མ་ཉམས་པ་སྦྱིན་གྱིས་དེ་རྣམས་ཏེ་ཙཱ་ག་ཏྱ་ས་པ་རི་ཝཱ་རཱ་ཤཱི་གྲྭ་མི་དཾ་བ་ལིཾ་གྲིཉྩནྟ་ཁཱ་དནྟུ། ཛཧཱུྃ་བཾ་ཧོ༔ཀྱང་བྱོན་ནས་འཁོར་བཅས་མྱུར་བར་གཏོར་མ་འདི་གཟུང་དུ་གསོལ་བཞེས་ཤིག་འཐུང་ཤིག། ཡང་དག་པར་སནྣཾཔ་མ་མ་སརྦ་སཏྭཱ་ནཱཉྩ་ཤཱནྟི་པུཎྜི་རཀྵ་བ་ར་ཎ་གུཏྟིཾ་ཀུརྔནྟུ་ཧུཾ་ཧཱུྃ་ཕཊ་ཕཊ། ཚིམ་སྟེ། བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཞི་བ་དང་རྒྱས་པ་དང་བསྲུང་བ་དང་བསྐྱབ་པ་དང་སྦ་བ་མཛོད། བཛྲ་དྷ་ར་ཨཱཛྫཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། ད་ཤ་ཀྲོ་དྷ་བ་ལི་། ང་ལ་ཨ་ན་འངརྡོ་རྗེ་འཆང་གི་བཀའ་བཞིན་དུ། །རྒྱུ་། ཁྲོ་བོ་བཅུའི་གཏོར་མའོ། ་ཀ་ཁོ་ན་ལ་ག་། ་དང་། བས་བ་། །།ད་ལོ་ངོ་མང་ཨོཾ་ཨཱཿཧཱུྃ་ཧོཨི་ཧཱ་བ་སྦི་ཏེ་ཀྵེ་ཏྲ་པཱ་ལེ་བྷོཉརྱ་བྷུཉྩ་ཨི་དཾ་གནྡྷཾ། པུཤྲཱི། འདིར་གནས་པའི་ཞིང་སྐྱོང་རྣམས་དང་གཞན་རྣམས་ལ་ཡང་འདི་དྲི་དང་མེ་ཏོག་དྷཱ་པཾ། དཱི་པཾ། མ་ཀླུ་ཏཾ་ད་དཱ་མ་ཧེ། ཏེ་ཙཱ་ག་ཏྱ་ས་པ་རི་ཝཱ་རཱཤྲཱི་གྲྭ་མི་དཾ་བ་ལི་བདུག་པ་མར་མེ་དང་མ་ཉམས་པ་སྦྱིན་གྱི་དེ་རྣམས་ཀྱང་བྱོན་ནས་འཁོར་དང་བཅས་པ་རྣམས་མྱུར་བར་གྲིཙྪནྟུ་ཁཱ་དནྟུ་པི་བནྟུ་ཛཧཱུྃ་བཾ་ཧོ། སམྦྷོཔཱ་མ་མ་སརྦ་སཏྭཱ་ནཱཉྩ་ཤཱནྟི་པུཊྛི་རགྷཱ་བ་གཏོར་མ་འདི་ཟུང་གསོལ་འཐུང་ཀུན་དུ་ཚིམ་སྟེ་བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཞི་བ་དང་རྒྱས་པ་དང་ར་ཎ་གུཏྟིཾ་ཀུརྦནྡྷ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། བཛྲ་དྷ་ར་ཨཱ་ཙཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། ཀྵེ་ཏྲ་པཱ་ལ་བསྲུང་བ་དང་བསྐྱབ་པ་དང་སྦ་བ་མཛོད། ཟླ་། རྡོ་རྗེ་འཆང་གི་བཀའ་བཞིན་དུ། ཀ། ཞིང་སྐྱོང་གི་བ་ལི། །།.་ཀྱི་མི་དབང་། བདག་ཟ་ཀ་ལ་དེ་ཀ་ཏ་ན་ཀ་མ་ལ་དད་པ་གཏོར་མའོ། །།།sg། ཆུ་དགདྲིའི་ཚིག་ཉིད་ཀྱི་ཕྱི་མ། སྐྱིད་ཤ་རུ་ཎི་ཊི་དེ་ཙའ་ལ་བྱ་ཆའི་བཀའ་དབང་གནང༌། ལྷ་མཚོའི་མཚོ་ཉིའི་ཁམ་བུ། དབྱེ་ན་བཞི། ཚེ་/།
