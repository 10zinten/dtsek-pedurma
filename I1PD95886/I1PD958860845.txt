དང། །རྣམ་པ་ད་དང་བསྟན། ་དང་གཡོག་དང་པོ་བཏབ། །དབ་བྱེ་བ་ལ། །དེ་ལྟར་མགོན་པོ་ཞེས་འ༄༅། །བགེགས་ཀྱི་རྒྱལ་པོའི་སྒྲུབ་ཐབས་བཞུགས། །བ་ཀྱི་བར་ལ་ད་པ་ཚོགས་པ། །བ་:བ། བདུད་པ་ཤེད། འདས་པ་། དུ་བསཔཕྱིང་༄༅། །དཔལ་རྡོ་རྗེ་འཛིན་པ་་ལ་ཕྱག་འཚལ་ལོ། །བགེགས་ཀྱི་རྒྱལ་་་པོའི་སྒྲུབ་ཐབས་ནི། །ལྷ་མོ་:སྐལ་བ་་བཟང་མོ་ཉོན། །མ་མོའི་ཁྱིམ་དང་ཉམས་དགའ་བར། །བསྒོམ་པ་བཟང་པོ་བྱ་བ་ཉིད། །འབར་བའི་བྲིས་སྐུར་་དམིགས་ནས་ནི། །སྔར་མཆོད་བློ་གྲོས་ལྡན་པ་ཡིས། །གར་སྟབས་འགྱིང་་་་བས་སྟན་གནས་པ། །སྐུ་མདོག་དམར་པོ་འོད་ཟེར་དམར། །ཞལ་གཅིག་པ་ལ་སྤྱན་གསུམ་པ། །རལ་པའི་ཅོད་པན་དག་གིས་བརྒྱན། །དམར་པོ་འབར་འཁྱིལ་འཇིགས་པ་སྟེ། །མཆེ་བ་གཙིགས་པ་ཡང་དག་བྲི། །དགྲ་སྟ་་་གཏུན་ཤིང་ཉིད་དང་ནི། །དེ་བཞིན་དུ་ཡང་མདའ་གཞུ་ཉིད། །ཁ་ཊཾ་ག་དང་ལྕགས་ཀྱུ་དང་། །ཁྲག་གིས་གང་བའི་ཐོད་པ་དང༌། །རྡོ་རྗེ་རལ་གྲི་དེ་་བཞིན་གཞན། །ཤ་ཆེན་ཐོད་པ་འཛིན་པ་དང་། །ཕུབ་ཆུང་བ་དན་རྩེ་་་གསུམ་དང་། །འཕང་མདུང་དང་ནི་བཅུ་གཉིས་སོ། །ཨོ་གགསིདྡྷི་སིདྷི། །
