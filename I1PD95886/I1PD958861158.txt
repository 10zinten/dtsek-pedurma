པ་ཨོ་ཏཱ་རི་ཎི་སརྦ་དུཁ་བྷ་ཡ་ཧཱ་རི་ཎི། ཙ་ཏུརྨ་ར་ནི་བཱ་རི་ཎི་སརྦ་དེ་བཱ་སྒྲོལ་མ་སྡུག་བསྔལ་དང་འཇིགས་པ་ཐམས་ཅད་འཕྲོག་མ། བདུད་བཞི་བཟློག་པར་མཛད་མ། ལྷ་སུ་ར་ན་ར་གནྡྷཆ་ཀི་ར་མ་ཧོ་ར་གཱརླུ་པ་པྲ་བ་པྲ་མརྡ་ནི་དང་ལྷ་མ་ཡིན་དང་མི་དང་དྲི་ཟ་དང་མིའམ་ཅི་དང་ལྟོ་འཕྱེ་ཆེན་པོ་ལ་སོགས་པའི་ཉེར་འཚེ་རབ་ཏུ་འཇོམས་མ། བྷཱུ་ཏི། པྲ་ཏ། ཡཀྵ། རཱཀྵ་ས། ཌཱ་ཀ་ཌཱ་ཀི་རྩྭ་འབྱུང་པོ་དང་ཡི་དྭགས་དང་གནོད་སྦྱིན་དང་སྲིན་པོ་དང་མཁའ་འགྲོ་དང་མཁའ་འགྲོ་མ་ལ་སོགས་པའི་དི་བྷ་ཡ་བི་དྷནྡ་ནི་པ་ར་ཀྲྀ་ཏ་ཡ་མན་ཏན་པྲ་ཡོ་ག་འཇིགས་པ་རྣམས་འཇོམས་མ། གཞན་གྱིས་བྱས་པའི་འཁྲུལ་འཁོར་དང་སྔགས་དང་རྒྱུད་ཀྱི་སྦྱོར་བ་བི་ནཱ་ཤི་ནི་བྷ་ག་ཝ་ཏི་དུརྒོཏྟཱ་རི་ཎི་ཨཱ་གཙྪ་ཨཱ་གཙྪ་ཨི་དཾ་བ་ལི་རྣམ་པར་འཇིག་མ། བཅོམ་ལྡན་འདས་མ་ངན་འགྲོ་ལས་སྒྲོལ་མ། ཚུར་བྱོན་ཚུར་བྱོན། གཏོར་མ་འདི་གྲིཉྩ་གྲིཉྩ་མ་མ་སརྦ་ཤ་ཧྲུན། ཧ་ན་ཧ་ན། ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་བནྡྷ་ན་བྱཱ་དྷི་་་ཟུང་ཟུང་བདག་གི་དགྲ་རྣམས། པ་འཚུབ་ཏེ་འཆིང་བ་དང་ནད་དང་གྲ་ཧཱ་ན། ད་ཧ་ད་ཧ། ཙྪེ་ད་ཡ་ཙྪེ་ད་ཡ། བྷེ་ད་ཡ་བྷེ་ད་ཡ། མོ་ཊ་ཡ་མོ་གདོན་ཐམས་ཅད་ཆོམས་ཆོམས་ཁོས་ཁོས། ཥཁོ་ཌཊཾ་ཀྱི་ཌཆི་ཀྱི་ཁ་ཅོཎ་ཥ་ཌཁ་ཀ་ཁཊ་ཡ། སྤོ་ཊ་ཡ་སྤོ་ཊ་ཡ། བསརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪ་ཤཱནྟི་་པུཎྜི་ཕྱག་འཚལ་ལོ། །བཅོདངོས་གྲུབ་ཐམས་ཅད་བདག་ལ་རབ་ཏུ་སྩོལ། ཞི་བ་དང་རྒྱས་པ་དང་རཀྵ་མེ་ཀུ་རུ། བློ་སརྦ་ས་ཏཱན། ་བ་ཧུཾ་མེ་ཀུ་རུ། ངསརྦ་ཀ་ལི་སྲུང་བ་བདག་མཛོད། སེམས་ཅན་ཐམས་ཅད། བདག་གི་དབང་དུ་མཛོད། འཐབ་པ་དང་འཐབ་ཀ་ལ་ཧ་བི་གྲ་ཧ་བི་བཱ་ད་བིགྷཱཏྟཱ་ད་བྱཀྟི་ཀ་རཱ་ཤི་ཙྪིནྡ་ཙྪིནྡ། བྷིནྡ་བྷིནྡ། བྷཉྫ་ཡ་མོ་དང་འགྱེད་པ་དང་རྩོད་པ་དང༌། བགེགས་སྐྱེད་ཅིང་གསལ་བར་བྱེད་པ་རྣམས་ནི་བླ་ཀ་nབྷཉྫ་ཡ། གློ་བྷ་ཡ་ཀྵ་བྷ་ཡ། སྟམྦྷ་ཡ་སྟམྦྷ་ཡ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཨཱཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། དུརྒོཏྟཱ་ར་བ་ལི། །།་ཊི་ཀ་ཁོ་ན་ཊི་ལ་ཁ་%@་་་ག་ངངན་འགྲོ་ལས་སྒྲོལ་མའི་གཏོར་མའོ། །།ཚེ་། ་རྡོ་༼ཆ་༽་ཎ་། ཅཙེདུནི+།
