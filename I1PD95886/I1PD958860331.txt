བློ་བསྟན་འགྱུར། ་རྒྱུད། ༼ཙིཏིཟླ་ཎཁྱོབཀཋਰਸਸਰਕਬਜਾਲੈਕੇ,ਲੋਂ'ਸਬਸੋདང་དེ་ལ་སྤྱོད་པའི་རྒྱུད་ལ་ནི་མངོན་པར་བྱང་ཆུབ་པ་ལ་སོགས་པ་སྟེ། མདོར་བསྡུ་ན་སྟོང་ཕྲག་བརྒྱད་ཡིན་ལ། ཚེརྒྱས་པར་ནི་དབྱེ་བ་བསམ་གྱིས་མི་ཁྱབ་པ་ཡིན་ནོ༑ཊ། གང་དག་ལས་ཀྱི་བྱ་བ་སྣ་ཚོགས་རྒྱ་ཆེན་པོ་ཕྱག་རྒྱ་དམིགས་པའི་མཚན་ཉིད་ལ་སོགས་པ་དང༌། །རྣལ་འབྱོར་གཉིས་ཀས་དངོས་གྲུབ་ཀྱི་དོན་ལ་སྒྲུབ་པར་བྱེད་པའི་དོན་དུ་གསུངས་པ་ཡིན་ནོ། །་་་་་་ག་པ་ཤ་བ་ཁྲི་ཀ་ལ་། ཁོ་བའི་zainཟླ་། བོད་ཡིག་གཙོ་ཆེའི་སྐྱེ་ཁན་ཁ་ང། (། ཚོ་བར་དགའ་ལ་བྱ་ལདེ་ལ་རྣལ་འབྱོར་གྱི་རྒྱུད་ནི་དེ་ཁོ་ན་ཉིད་བསྡུས་པ་ལ་སོགས་གསུངས་ཏེ། མདོར་བསྡུ་ན་སྟོང་ཕྲག་དྲུག་པ་ཡིན་ལ། རྒྱས་པར་ནི་དབྱེ་བ་བསམ་གྱིས་མི་ཁྱབ་པ་ཡིན་ནོ། །གང་དག་ལྷའི་དཀྱིལ་འཁོར་གྱི་འཁོར་ལོ་དགོད་པ་དང་། སྨྱུ་དེ་ལས་བྱུང་བའི་ལས་སྒྲུབ་པ་དེ་ཁོ་ན་ཉིད་རྣམ་པ་བཅུ་རབ་ཏུ་དཔྱད་པས་དེ་རྣལ་འབྱོར་དང་ཕྱིའི་བྱ་བས་ཀྱང་ཏིང་ངེ་འཛིན་གྱི་ཡན་ལག་ཏུ་བསྒྱུར་བས་དངོས་གྲུབ་ཐམས་ཅད་ཐོབ་པར་འདོད་པའི་དོན་དུ་གསུངས་པ་ཡིན་ནོ། །།།དེ་ལ་ཞའོ་ལས་ཀྱི་ར་བའི་ལོ་པ། ན་པ་ལགས་དང་ཁ་ལ་བཀའ་བློན་འདུགཁྲོ་སྒྱུ་གང་དེའང་རྣལ་འབྱོར་ཆེན་པོའི་རྒྱུད་ནི་གདུལ་བྱ་དབང་པོ་མཆོག་དང༌། རྣལ་་འབྱོར་གཙོ་བོར་བྱེད་པའི་དོན་གསུངས་པ་ཡིན་ཏེ། མདོར་བསྡུ་ན་སྟོང་ཕྲག་བཅུ་གཉིས་ཡིན་ལ། ྋརྒྱས་པར་ནི་དབྱེ་བ་བསམ་གྱིས་མི་ཁྱབ་པ་ཡིན་ནོ། །དེ་ལ་རྣལ་འབྱོར་ཆེན་པོའི་རྒྱུད་ནི་རྣམ་པ་གཉིས་ཏེ། རང་བཞིན་གྱིས་རྒྱུད་དང་བརྟག་པའི་རྒྱུད་དོ། །དེ་ལ་རང་བཞིན་གྱི་རྒྱུད་ནི་མདོར་བསྡུ་ན་རྣམ་པ་གསུམ་སྟེ། རྒྱུའི་རྒྱུད་དང་། འབྲས་བུའི་རྒྱུད་དང་། ཐབས་ཀྱི་རྒྱུད་དོ། །དེ་ཡང་གསང་བ་འདུས་པའི་རྒྱུད་ཕྱི་མ་ལས། རྒྱུད་ནི་རྒྱུན་ཅེས་བྱ་བར་གྲགས། །།
