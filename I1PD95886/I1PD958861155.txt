བསྟན་འགྱུར། རྒྱུད། ཚུ་། ཀུ་རུ་སརྦ་ཛ་ནོ་པ་ཤཱནྟི་ཀུ་རུ་ཕཊ་སྭཱ་ཧཱ། ཀཱ་མ་རཱ་ཛ་བ་ལཾ། །ཏ་གྱིས། སྐྱེ་བོ་ཐམས་ཅད་ལ་ཉེ་བར་ཞི་བ་མཛོད་ཅིག། འདོད་པའི་རྒྱལ་པོའི་གཏོར་མའོ། །།ཁ་བློན་Aཨོ་ཧཱིཨེ་ཧེ་ཧི་ཨེ་ཀ་ཛ་ཊཱི། མ་མོ་པ་ཅུ་ཏཱན་ཤཱནྟི་ཀུ་རུ། པུཎྚི་ཀུ་རུ། ཚུར་བྱོན་རལ་གཅིག་མ། བདག་འཁོར་དང་བཅས་པ་ལ་ཞི་བར་མཛོད། རྒྱས་པར་མཛོད། མ་ཧཱ་ཡཀྵ་དྷི་པ་ཏ་ཡེ་ཨི་དཾ་བ་ལིཾ་གྲིཉྩ་གྲིཥྛ། དྲྀཏྟཱ་པ་ཡ་གྲོཏྟཱ་པ་ཡ། པ་ར་བི་དྷཾ་ཨཱ་གནོད་སྦྱིན་གྱི་བདག་པོ་ཆེན་པོ། འདི་གཏོར་མ་ཟུང་ཟུང་། འཛིན་དུ་ཆུག་འཛིན་དུ་ཆུག། གཞན་གྱི་ཀནྟ་ཡ་ཨཱ་ཀརྟ་ཡ། ཏུ་ཊ་ཏུ་ཊ། མུ་ཊ་སྤུ་ཊ། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨུ་གྲ་ཏཱ་རཱ་རིག་པ་ཁུག་ཁུག། ཆོད་ཆོད། །ཆོམས་ཆོམས། ་བླ་སྒྲོལ་མ་དྲག་མོའི་་བ་ལི། །།,དབའདེའི་ལས་མ་བྱུ་རྒྱུ། ཏེ་ཁ་ཁྱི་ཁ་ར་ཊ་ཁོ་ཊ་གཏར་མའ། །་རྒྱ་གར་དང་། གཡས་པལ་བློཏོ། །jnppypཚོ་༼papnཀ་ལ་། ་(Jའི་ཨོ་ཨཱཧཱུཾ་ཧཱུཾ་ཀུ་རུ་ཀུལླེ་སརྦ་སཏྭཾ་བ་ཤཾ་ཀ་རི། སརྦ་བྷ་ཡ་བྷ་ཡཾ་ཀ་རི། རིགས་སུ་བྱེད་མ་སེམས་ཅན་ཐམས་ཅད་དབང་དུ་བྱེད་མ། འཇིགས་པ་ཐམས་ཅད་འཇིགས་པར་ཨི་དཾ་བ་ལིཾ་གནྡྷཾ་པུཏྠཾ་དུ་པཾ་དཱི་པཉྩ་ཀླུ་ཏཾ་ད་དཱ་མ་ཧེ་སརྦ། དེ་བ་ནཱ་ག་ཡཀྵ་བྱེད་མ། གཏོར་མ་དྲི་དང་མེ་ཏོག་བདུག་པ་མར་མེ་དང་མ་ཉམས་པ་འདི་འབུལ་གྱི། ་ལྷ་ཀླུ་གནོད་སྦྱིན་གནྡྷཧཱ་སུ་ར་ག་རུ་ཌ་ཀིནྣ་ར་མ་ཧོ་ར་གཱ་ད་ཡོ། འ་ཡ་ཐིཏྟཾ་བྷུཉྫ་ཐ། ་་ཁཱ་ད་དྲི་ཟ་ལྷ་མིན་མཁའ་ལྡིང་མིའམ་ཅི་དང་ལྟོ་འཕྱེ་ཆེན་པོ་ལ་སོགས་ཐམས་ཅད། ཅི་འདོད་ཟོ་ཤིག། འཆོ་ཐ་པི་བ་ཐ་ཛི་གྲྭ་ཐ་མ་མ་མ་ནསིདྡྷཾ་པྲ་ཡཙྪ་ཐ་ཨོ་ཨཱཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཀུ་རུ་ཤིག། འཐུངས་དྲི་སྣོམ། བདག་གི་ཡིད་ལ་དངོས་གྲུབ་སྩོལ་ཅིག་བ་། རིགས་སུ་བྱེད་ཀུལླ་བ་ལི། །།།་པ་རེན་བློ་བོ་མེད་པ་དང་མཛོད། སངས་རྒྱས། པའི་གཏོར་མའོ། །མཁྱེན་ཏེ། །དཎམ་བྷཚོང་བ་ཚོར་བ་སྐྱུར་འཛིན་བ::ཀྱི་བྱ་བུཥན་ཏུ་ཆེ་བ་ལ། འདི་དག་ལ་ཁ་པ་བཞི་neཎ་མཆཡོepབ་པ་བཀག་པས། ཁ་ན་—།
