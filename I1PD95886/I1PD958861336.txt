པར། དང་ནི་ས་ཡི་སྟེང་གནས་ཀོ་ཀ་མེད་དང་བྱང་ཆུབ་ཤིང་། །དཔལ་གྱི་བེའུ་དང་ཨ་ཤད་འདབ་མ་ཞོལ་འབྲས་མེ་ཏོག་དཔལ་འབྲས་དུང་དང་འཁོར་ལོ་ཅན། །དགྱེས་པ་རྣམས་ཀྱིས་དྲི་མེད་ཡིད་བདེ་དོན་ཀུན་གྲུབ་པའི་བཀྲ་ཤིས་ཁྱོད་ལ་སྩལ་དུ་གསོལ། །གདུགས་དང་དྲུར་བ་པདྨ་དང་ནི་རྒྱལ་མཚན་དག་དང་ཉ་དང་པི་པ་ཝང་སིལ་སྙན་འཛིན། །ཕེག་པ་དག་དང་རབ་གང་བུམ་པ་ཐུབ་པ་གསུང་་་་མཆོག་རྡོ་རྗེ་དྲིལ་བུའི་སྒྲ་དང་ནི། །སངས་རྒྱས་ཆོ་འཕྲུལ་འཇོ་སྒེག་བཞད་འགྲོས་རྣམ་པར་རོལ་པ་ལྷ་དང་མི་ཡིས་ཕྱག་བྱས་པ། །དགྱེས་པ་རྣམས་ཀྱིས་དྲི་མེད་ཡིད་བདེ་དོན་ཀུན་གྲུབ་པའི་བཀྲ་ཤིས་ཁྱོད་ལ་སྩལ་དུ་གསོལ། །རྗེ་བཙུན་འཕགས་མ་སྒྲོལ་མས་གསུངས་པའི་བཀྲ་ཤིས་བརྒྱད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་བི་ན་ཡ་ཤྲཱི་མི་ཏྲ་དང་། ལོ་:ཙཱ་བ་ལྡན་མ་ཚུལ་་ཁྲིམས་སེང་གེས་བསྒྱུར་བའོ། །།།ལ་ག་བྲབོད་སྐད་དུ། བཤེས་བ། ས་སུ་བཅད་པ། དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོདམ་པ་ཏོག་ལ་དགའ་ལྡན་ཕོ་བྲང་དང་། བྱིང་བ་ལ། །གས་ཡག་དེ་བས་མགོན་ལ་སྤྱི་ལ་ཚབ་དམ་པས་ས་དབང་བཟུང་ཞིང་། །འདས་དང་ལ་་ལ་དང་པ་ལག་པ་གང་། །::ཀ་བ་གང་དང་། །དག་ལ། །།་བ་ལ་བ། ་་་པད་མ་འབྱུང་བ་དང་། བག་མ་བབ་པ་མ་བྱ། དག་ལ་ག་ཡི། བ་དང་ལོ་དག་ཡིགས་པོ། །།
