རཀྵ་རཀྵ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། སྲུང་སྲུང་། ཏིལ་ལ་ཏ་ཀམ་ཧཱ་སཱ་ཧཱ་ས་པྲ་མརྡ་ནི་བ་ལི། །་། །།སྟོང་ཆེན་རབ་ཏུ་འཇོམས་མའི་གཏོར་མའོ། །།།ཨོཾ་མ་ཧཱ་མཱ་ཡུ་རི་བི་རཱུ་རཱ་ཛྷི་སརྦ་བི་ཥ་པྲ་ཤ་མ་ནི་མ་མ་ཤཱནྟིཾ་ཀུ་རུ་པུཊྚི་ཀུ་རུ་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་རྨ་བྱ་ཆེན་མོ་་་་་་་་༌་་་་་་བ་ཧུཾ་ཀུ་རུ་སཆ་དེ་བ་ནཱ་ག་ཡཀྵ་གནྣཱཁྭ་སུ་ར་དུག་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བྱེད་མ་ལྷ་དང་ཀླུ་དང་གནོད་སྦྱིན་དང་དྲི་ཟ་དང་ལྷ་མ་ཡིན་དང་ག་རུ་ཌ་ཀིནྡ་ར་མ་ཧོ་ར་གཱརླུ་བ་དྲ་བཱན་ཧ་ན་ཧ་ན་ད་ཧ་ད་ཧ་མཱ་ར་ཡ་མཱ་ར་ཡ་ནམ་མཁའ་ལྡིང་དང་མིའམ་ཅི་དང་ལྟོ་འཕྱེ་ཆེན་པོ་ལ་སོགས་པའི་ཉེ་བར་འཚེ་བ་རྣམས་གརྫ་ཡ་གརྫ་ཡ་ཏརྫ་ཡ་ཏརྫ་ཡ་ཨི་དཾ་བ་ལིཾ་གྲིཉྩ་ཤྲཱིཉྩ་མཱི་སརྦ་སཏྭཾཉྩ་རཀྵ་རཀྵ་ཧཱུཾ་ཧཱུཾ་ཏང་ཁའི་ལོ་བདག་དང་སེམས་ཅན་ཐམས་ཅད་སྲུང་སྲུང་། ཕཊ་སྭཱ་ཧཱ། མ་ཧཱ་མཱ་ཡུ་རཱི་བ་ལི། །།་་་་་ར་ཎ་ཁ་ཀྱིའིརྨ་བྱ་ཆེན་མོའི་གཏོར་མའོ། །ར་འཁ་ཅང་ཏེ། ལོཀྱི་འཁོ། /དེ་ཁ་ནི་ང་ལ་ཏ་ཟླར་གྱུར་ཏོ། ཊ་པ་ཏ་ལ་ར་་ང་བ་ཨོ་མ་ཧཱ་ཤཱི་ཏ་བ་ནི་བི་，་རཱ་ཛཱི་སརྦ་དུཋ་པྲ་མརྡ་ནི་མ་མ་རིག་སྔགས་ཀྱི་རྒྱལ་མོ་བསིལ་བའི་ཚལ་ཆེན་མོ་གདུག་པ་ཐམས་ཅད་རབ་ཏུ་འཇོམས་མ་བདག་དང་སརྦ་སཏྭཱ་ནཱཉྩ་ཤཱནྟི་ཀུ་རུ་པུཊྚི་ཀུ་རུ་པ་ཤཾ་ཀུ་རུ་སརྦ་དུཥཱ་པྲ་དུཏྟཱ་མཱ་ར་ཡ་མཱ་ར་ཡ། སེམས་ཅན་ཐམས་ཅད་ལ་ཞི་བར་མཛོད། རྒྱས་པར་མཛོད་དབང་དུ་མཛོད། ཟླ་བ་བར་གྱི་དབང་ཀཱ་ར་ཡ་ཀཱ་ར་ཡ། ཏརྫ་ཡ་ཏརྫ་ཡ། ཨི་དཾ་བ་ལིཾ་ཀྲྀཥྞ་གྲིནྟ་རཀྵ་རཀྵ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྣ་ཧཱ། མ་ཧཱ་ཤྲཱི་ཏ་བ་ནི་བ་ལི། །།།བསིལ་བའི་ཚལ་ཆེན་མོའི་གཏོར་མའོ། །།།།m།
