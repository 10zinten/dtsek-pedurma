བསྟན་འགྱུར། རྒྱུད། ་ཚུ་ཟླ་་སྐུ་nཕན་ཕྱིར་བློ་ཡི་སྟོབས་བསྐྱེད་དེ། །ཅི་ནུས་མཐུ་ཡིས་རིམ་པར་གསལ་བཤད་པའི༑། །བསོད་ནམས་ཚོགས་གྱུར་འགྲོ་བས་རབ་ཐོབ་ནས། །སྒྲིབ་པ་གཉིས་སྤངས་ཚོགས་ཆེན་ཡོངས་རྫོགས་ཏེ། ཁྱི། །བླ་མེད་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བར་་་ཤོག་་། ཆུ་གཏོར་གྱི་་ལས་བྱང་རིམ་པར་ཕྱེ་བ་གསལ་བར་བཤད་པ་སློབ་དཔོན་ཤཱཀྱ་མི་ཏྲས་མཛད་པ་རྫོགས་སོ། །།།Fཡུ''e::ཀྱི་པཥབྱཁོ་nཚོ་ལྷོའི་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་པའོ། །ཡ་། ཁོའི་རྒྱུ་ད་པ་ན་ཤཚ་པོ། བཀའ་བསྡུ་ན། །གས་པོ་འཚོ་བའོ་ཏིལ་ལ་ལྔ་པོའི་་་་་་་་པའིལ་བྱུང་པ་སྟེ། ཁོ་ད་ལྟ་མ་ཟ་སྡེ་པ། །རྒྱུ་ཅི::ཀྱིད་འོ་ན། བློ་བཀའ་བསྡུར་མཆན། གཏོརེདུ་འོ་བྱེད་ས་། པ་གཞི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཕྱག་རྒྱ་ཆེན་་་པོའི་རྣལ་འབྱོར་ལ་འཇུག་པའི་མན་ངག་བཞུགས། «པེ་»ག་བ་ཏ་ར་པིན་དརླ། «སྣར་»གཱ་པ་ཏ་ར་པིན་དནྡྲ། «པེ་»ཞུགས། «སྣར་»རབ་ཞུགས། «པེ་»ཉིད། «པེ་»«སྣར་+གང་། པ«པེ་《སྣར་འཚལ། །ཟླ་བམརརབནབ་བློ་་་་་་་་མཚོ་«པེ་《སྣར་པ་བསྡུ་བ། «པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་—དུ། «པེ་»སྣར་—པར། ན། ས«པེ་»«སྣར་གྱིས། ་པ«པེ་»«སྣར་»པ། «པེ་»«སྣར་»དུམ་བ། «པེ་གྲུབ་བོ། །འདིས། རྗེད། «སྣར་འགྲུབ། །བོ་འདིས། ཀཔདང་གནོདང་ནཔ:«པེ་»«སྣར་»གྱིས། «པེ་»རྟོག«སྣར་རྟོགས། «པེ་»«སྣར་བཞུགས། «པེ་ལེགས། «སྣར་བདེ་ལེགས། ས«པེ་»«སྣར་ཀྱིས། པ«པེ་»«སྣར་»རྟོགས། «པེ་»«སྣར་པ། «པེ་»«སྣར་པར། «པེ་»«སྣར་»འགྱུར་བའི། །«པེ་»«སྣར་པའི། «པེ་»«སྣར་དེ། འགྲོ་བའི«པེ་»«སྣར་ན། «སྣར་བཟུང་། «པེ་+པ། «པེ་»«སྣར་»གྱིས། ས«པེ་»«སྣར་»བྱའི༑«པེ་»«སྣར་»རྟོགས། «པེ་»«སྣར་གྲུབ། «པེ་»གིས། «པེ་»《སྣར་གྱུར། ལ«པེ་«སྣར་»གིས་རྟོགས། ::«པེ་»«སྣར་པ། /«པེ་»«སྣར་བསླང་བའི། ཡི་(ལོའི་«པེ་»«སྣར་པས། «པེ་»«སྣར་»སྤངས་པར། བདུན«པེ་»«སྣར་པོ། «པེ་»«སྣར་»རྣམས༑ས«པེ་»«སྣར་»ཀྱིས། པོ«པེ་»«སྣར་དང༌། «པེ་»«སྣར་པར་རོ། 《སྣར་»ལས། «པེ་»«སྣར་པའི། «པེ་»«སྣར་»པ། «པེ་བརྩོན། «པེ་»«སྣར་བྱ་བའོ། །
