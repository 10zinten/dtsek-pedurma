རྒྱུད། བཅུ?ཉིད་དང་། དྲིལ་བུའི་དེ་ཉིད་དང་། ་ཌ་ཕྱག་རྒྱའི་དེ་ཉིད་བསྐུར་བ་ལྟར་མན་་་ངག་གིས་བྱིན་གྱིས་བརླབས་ལ་དབང་བླང་ངོ༌། །གང་ཀཚིག་བྱེད་ཀྱི་ལོ་ਉਤਨਸਿੰਘਭੂਸਤਨਸੰਸਥੋਂਉਸਨਾਲགསང་དབང་མན་ངག་བཞིན་བྱིན་གྱིས་བརླབ། ་གཏང་། ཝཱ་བླང་ངོ་། །ཤེས་རབ་ཡེ་ཤེས་ཀྱང་དེ་བཞིན་ནོ། །།བཞི་པ་ཡང་དེ་བཞིན་ནོ། །་འཚོ། ་། སུ་ཕྱོགས། ལ་ལ་དག་འདུལ་བ་ཀུན་གྱི་གང་། །བཀའ་སློབ་མ་ལ་དབང་བསྐུར་བ་ཡང་དེའི་རིམ་པས་ཏེ། དེ་ཡང་དཀྱིལ་འཁོར་དུ་གཞུག་པ་དང་དབང་བསྐུར་བའོ། །།།གཞུག་པ་ལ་ནང་དུ་གཞུག་པ་དང་། གསང་བ་བསྒོ་བ་དང༌། །དཀྱིལ་འཁོར་དུ་གཞུག་པའོ། །།ནང་དུ་གཞུག་པ་ནི་་་ལས་བུམ་:ཆུས་བསངས་པའི་མིག་རས་ཨཱ་:ཁཾ་གིས་བྱིན་གྱིས་བརླབས་ལ་བཅིང་བ་དང་། ཨཱ་ཁཾ་ཟློས་སུ་གཞུག་པ་དང༌། དགའ་ཆེན་ཞེས་གསོལ་བ་གདབ་པ་དང་། ལས་ཀྱི་སློབ་དཔོན་གྱིས་རྡོ་རྗེ་དང་མེ་ཏོག་སུ་ར་ཏ་ཧཱུཾ། ཞེས་སྙིང་གར་སྦྱིན༑ཨཱ་ཁཾ་གིས་ནང་དུ་གཞུག་གང་ཡིན་ཞེས་དྲི། ནང་དུ་གཞུག་པའོ། །གསང་བ་བསྒོ་བ་ལ་གསུམ་སྟེ། །རྒྱུད་སྦྱང་བ། བསྒོ་བ་དངོས་དང་ཡེ་ཤེས་་་དབབ་པའོ། །རྒྱུད་སྦྱང་བ་ལ་ལྔ་སྟེ། ཐོག་མ་ཞེས་པ་་སྡིག་པ་སྦྱང་བ་དང་། ཨུཏྟཱ་ད་ཡས་སེམས་བསྐྱེད་པ་དང་། སྡོམ་པ་འཛིན་དུ་གཞུག་པ་དང༌། མཁའ་གསང་ནས་དབྱུང་བ་དང་། བགེགས་བསལ་བའོ། །ཨ།  ོ་ཀོ་ལོ་ཏཎ་ཌི་ཏི། ་་་་་་་་་་རྒྱུ་མ་ལ་ལ་་་་་ལ། །ངོལྟ་ཀཀྐོར་ལ་ཎnn/p། ཅཎཱ་ཥ་nཏརབསྒོ་བ་དངོས་ལ་བདུན་ཏེ། འདོད་ཆགས་ཆེན་པོ་ལས་བྱུང་བའི། །ཁྲུས། འཁོར་ལོའི་དེ་ཉིད་གཟུགས་བརྙན་བཞིན། །།ཞེས་བྱིན་གྱིས་བརླབ་ཅིང་གནང་བ་ནོད་པ་དང༌། སངས་རྒྱས་ཐམས་ཅད་:ཀྱི་བྱིན་གྱི་རླབས་ཀྱིས་བརླབས་པས་།
