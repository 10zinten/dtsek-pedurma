རྒྱུད། །ཡི?ཁ་ན་ཁ་ཀོགས་བཞིན་ལ་མི་དམིགས། །།འོད་གསལ་སེམས་ནི་དག་པ་ལ། །བ། གཅིག་དང་དུ་མ་བྲལ་བའི་ཕྱིར། །དོན་གཞན་འགྱུར་བ་ཡང་མི་དམིགས། ཏུ། སྒྱུ་མ་ཙམ་གྱི་སེམས་དེ་ནི། །དག་དང་མ་དག་གཉིས་མི་རུང་། །དེ་བས་རང་བཞིན་རྣམ་དག་པའི། །བྱང་ཆུབ་སེམས་སུ་ངེས་པའོ། །།དེ་ཡི་དེ་བཞིན་ཉིད་ངོ་བོ། །གང་དང་འབྲེལ་པ་མ་ཡིན་ནོ། །མ་འབྲེལ་པ་ཡང་མ་ཡིན་ནོ། །ཞེ་། ཀུན་རྟོག་ཡང་ནི་སྙམ་མི་སེམས། །ནམ་མཁའ་རྟག་མེད་བལྟར་མི་སྣང་། ད་། ཆོས་་་་རྣམས་ཀུན་ཀྱང་དེ་བཞིན་ནོ། །ཆོས་རྣམས་སྐུ་གསུམ་མཉམ་པ་སྟེ། །།།གང་ཡིན་སེམས་ཀྱི་ངོ་བོའོ། ་་། སེམས་དེ་དེ་བཞིན་ཉིད་གནས་ལ། །་། དེ་ནི་ཁམས་གསུམ་གནས་མ་ཡིན། །མི་གནས་དེ་ནི་ཀུན་བརྟགས་ཀྱི། །སྐྱེ་མེད་ངོ་བོ་་མེད་པའོ། །ཡིན་ཡང་བདེ་གཤེགས་ཐམས་ཅད་ཀྱི། །ཡེ་ཤེས་འབྱུང་བའི་རྒྱུ་ཉིད་དེ། །རྒྱུ་གསུམ་མི་གནས་ཆོས་དེ་ནི། འཁྲུ། སྐྱེ་མེད་ཡེ་ཤེས་འབྱུང་བའི་གནས། །སད་དང་རྨི་ལམ་དབྱེར་མེད་པས། །འདས་པ་རྨི་ལམ་ལྟ་བུ་སྟེ། །ད་ལྟར་བྱུང་བ་རྨི་ལམ་འདྲ། །མ་འོངས་རྨི་ལམ་ལས་བྱུང་བའོ། །བདེ་་་གཤེགས་བྱང་ཆུབ་སེམས་དཔའ་དང་། །སེམས་ཅན་རྣམས་ནི་ཇི་སྙེད་པ། །རྨི་ལམ་ལྟ་བུར་བདག་མེད་དེ། །སྒྱུ་མ་ལྟ་བུར་དེ་བཞིན་ནོ། །འོ་ཎ། དཔེར་ན་་་་ཡིད་བཞིན་རིན་པོ་ཆེ། །རྟོག་པ་ཐམས་ཅད་རབ་བྲལ་ཡང་། ་ཁ། སྐལ་བ་་་་ལྡན་པའི་སེམས་དབང་གིས། །འདོད་པ་ཐམས་ཅད་འབྱུང་བར་འགྱུར། ་་ཁྱུ། །དེ་བཞིན་བྱང་ཆུབ་སེམས་དེ་ལས། །།ཆོས་རྣམས་ཐམས་ཅད་འབྱུང་བས་ན། །ནམ་མཁའ་རྗེས་འབྲང་ཆོས་རྣམས་ལ། །སངས་རྒྱས་ཆོས་རྣམས་རྗེས་སུ་འབྲང་། །དེ་བས་དངོས་གྲུབ་ཐམས་ཅད་དང༌། །སངས་རྒྱས་ཆོས་རྣམས་ཐམས་ཅད་ཀྱིས། །ལུས་དང་ངག་དང་སེམས་ལ་གནས། །དེ་ནི་ཁམས་གསུམ་མི་གནས་སོ། །ནམ་།
