བསྟན་འགྱུར། ་རྒྱུད། ་༼པི་རོ་མཚུངས་རིག་མས་ཉེར་འཁྱུད་པའི། །དཔེ་མེད་བདེ་ཆེན་གང་བའི་དབང་ཕྱུག་་དཔལ་གྱུར་ཤོག། དེ་ཁོ་ན་ཉིད་རིན་པོ་ཆེ་སྣང་བའི་:རྣམ་པར་་བཤད་པ། སློབ་དཔོན་ངག་གི་དབང་ཕྱུག་གྲགས་པས་མཛད་པ་རྫོགས་སོ། །་་་། ཁ་པ། །ལོ་:ཙཱ་བ་འགོས་་ལྷས་བཙས་ཀྱིས་རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་དཔེ་ལས་ཞུ་:གཏུགས་གཡར་ཁྲལ་འཚལ་བ་ལགས་སོ། །ལ་བ། ཀོ་དང་ཁྲོ་བའི་རྒྱུ་བ་པོ་ਉਸਨੂੰਨਿਰਨਾਇਗੀਲਨੇਊਸਿੰਘਨੇਸੋਨੇਦੀਸਿੱਧམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»དེ་ཁོ་ན་ཉིད་རིན་པོ་ཆེ་སྣང་བའི་རྣམ་པར་བཤད་པ་བཞུགས་སོ། །«པེ་»《སྣར་»ཏཏྟཱ་རནྟཱ། «པེ་»«སྣར་གྱི། «པེ་»རད་་:«སྣར་»ཀྱི། ས«པེ་»«སྣར་»གི། ཀྱི་པ་ནས་དེ་ན་ཁྱེད་བློ་ལ་བར་བྱེད་པ་«པེ་»«སྣར་པའི། «པེ་»སྣར་»ཀྱི། (《པེ་»«སྣར་ལོ། «པེ་»སྣར་པ། ས《པེ་»ས་སྐྱིད་ལགས་ཀྱ་ཀྱིཀྱི་ཡང་ས་ལ་། བ///༽«པེ་»«སྣར་ན། «པེ་»«སྣར་»—པར། «པེ་—པར་བྱེད། །«པེ་»«སྣར་—བའི། 《པེ་《སྣར་གསལ་བར། །་«པེ་《སྣར་»བྱའི། «པེ་»«སྣར་རྟོག«པེ་»«སྣར་བསམ། «པེ་»«སྣར་»འདིའི། །/དངདེ«པེ་《སྣར་»ཚིགས། 《སྣར་»ཟློག་«པེ་»«སྣར་»ལས། དོན་ལ་ལོ་་«པེ་»《སྣར་»—ལ། འ«པེ་»«སྣར་»ཏེ། («སྣར་»འགག་པར་འགྱུར་བ། «པེ་»«སྣར་—དུ། ས«པེ་《སྣར་སྐ། །«སྣར་»སྒྲུབ། «པེ་»«སྣར་པའི། «པེ་»«སྣར་»+ཅེས་བྱ་བ་ནི་བརང་རིག་ལ་་:འ་ལ་དབློར་། ཀརྒྱུད། ༽བག་་«པེ་»«སྣར་+བ། «པེ་»«སྣར་པ་ནི། «པེ་《སྣར་»དགོས། »«པེ་»«སྣར་»དང་། ས«པེ་»«སྣར་སྐུ། «པེ་»«སྣར་སྤྱད། «པེ་《སྣར་མ། «པེ་»«སྣར་ཅེས། ་མ་«པེ་སྤྱོད། «པེ་«སྣར་+པ། པ«པེ་《སྣར་བུ་ནི། «པེ་»«སྣར་པ་མཐའི། /ནན་་«པེ་《སྣར་ཀྱི། འ«པེ་»«སྣར་»—ལྷའི། «པེ་»«སྣར་བལྟོས། 《པེ་»«སྣར་»བསྒོམ། ས«པེ་《སྣར་པ། «པེ་《སྣར་+བ། 《སྣར་»བསྒྲུབ། བཀཁཊ།
