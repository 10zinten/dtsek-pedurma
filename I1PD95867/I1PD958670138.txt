ཞལ་སྔ་ནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།པཎྜི་ཏ་ཆེན་པོ་རྒྱ་ཡ་ན་ཤྲཱི་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ཆེན་པོ་ཤཱཀྱའི་དགེ་་་སློང་བློ་ལྡན་ཤེས་རབ་ཀྱིས་བལ་ཡུལ་མཐིལ་དུ་བསྒྱུར་བའོ། །།།:《ཏུ་བཀྲ། །ཀ་དག་འོ་བ་ཀྱིས། པའི་དོ་ར། བབས་སpནཡེ་བབས་དང་། ད་དུང་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཡན་ལག་བཞི་པའི་་སྒྲུབ་ཐབས་ཀྱི་འགྲེལ་པ་སྙིང་པོ་སྙེ་མ་ཞེས་བྱ་བ་སློབ་དཔོན་ཆེན་པོ་དཔལ་ཀུན་ཏུ་བཟང་པོའི་ཞལ་སྔ་ནས་མཛད་པ་་་བཞུགས་སོ། །«པེ་»«སྣར་»རཱ་ག་སཱ་དྷ་ནཾ་ཊཱི་ཀཱ་ས། །«པེ་»«སྣར་»འདི། «པེ་»«སྣར་»དོ། །«པེ་»«སྣར་ན། «པེ་»«སྣར་»དང༌། །O«པེ་《སྣར་ན།  ེ«པེ་»«སྣར་པ། ཉི«པེ་»«སྣར་ཚོགས་པའི་རྒྱུ་བརྗོད་དེ། ...འདི་ལྟར་འགྲོ་་་བའི་དོན་ཕུན་སུམ་ཚོགས་པ་ནི་འདིར་རབ་ཏུ་བྱེད་པའི་བརྗོད་པ། (ཞེས་པའི་གཡུང་དྲུང་རྟགས་བར་གྱི་ཡི་གེ་རྣམས་དེབ་འདིའི་ལྡེབ་པའི་བསྡུར་མཆན་པར་དཔེ་བསྡུར་བྱས་ཡོད། «པེ་»《སྣར་ཡིས། ། ི«པེ་《སྣར་གྲག«པེ་»«སྣར་རིང་། ཚེ«པེ་»«སྣར་ལུགས། «པེ་»དུམྨ། «སྣར་དུམ་མ། །«པེ་《སྣར་འགྱུར། ༄༅«པེ་»«སྣར་»—དེ་བཞིན་དུ། བབ་བསྟེགས་བུའི་«པེ་»«སྣར་བྱ་བའི། «པེ་»«སྣར་+ཚོགས་པའི་རྒྱུ་བརྗོད་དོ། །འདི་ལྟར་འགྲོ་བའི་དོན་ཕུན་་་་སུམ་ཚོགས་པ་ནི་འདིར་རབ་ཏུ་བྱེད་པའི་བརྗོད་པར། (ཞེས་པའི་ཡི་གེ་འདི་རྣམས་དེབ་འདིའི་ལྡེབ་ང—ངབར་དཔེ་བསྡུར་བྱས་ཡོད། «པེ་»«སྣར་»དག«པེ་»«སྣར་»—པས། «པེ་»《་《སྣར་རྗེ་ཡི། །@«པེ་»«སྣར་—ཉིད། །སྐྱེ་བའི་ཡན་ལག་«པེ་»«སྣར་བསྐྱེད། Q«པེ་»«སྣར་»དང་ལྔ།  ི«པེ་»«སྣར་»+པ། «པེ་»«སྣར་»—པ། ®«པེ་»«སྣར་»སྒྲུབ། འོ«པེ་»«སྣར་བསྒོམས། ་«པེ་»«སྣར་»བརྟན། གུ«པེ་»«སྣར་། །བྱེ་བ་དུ་མར་དྲན་པ་དང་། ཀྱི«པེ་»«སྣར་དོར། «པེ་»«སྣར་»པར་རོ། «པེ་»«སྣར་དགྲ། «པེ་»«སྣར་འཇོམས་་་པའི་ཕྱིར་ན། གུ«པེ་»«སྣར་རོ། @«པེ་»«སྣར་»བ། ༄༅«པེ་»«སྣར་»—གི། «པེ་»«སྣར་»ད། །་ཚེ«པེ་»«སྣར་པ། «སྣར་བ། ད་་«པེ་»《སྣར་»གི། «པེ་»«སྣར་»པ། ་་་་.།
