བསྐུར་བྱས་ནས་བསྒྲུབ་བྱ་དེ། །རྡོ་རྗེ་སེམས་དཔའི་སྤྲུལ་པ་ཡང་། །སྲུང་ན་གཞན་སྦྱར་སྨོས་ཅི་དགོས། །གལ་ཏེ་:ཁྱིས་གཙེས་ལ་སོགས་ལ། །མཚོན་ཆ་་་ནོར་ནི་གཞན་ལྟ་བསྒྲུབ། །དེ་ཚེ་ས་གསུམ་སྒྲས་འགེངས་ཚེ། །རྣལ་འབྱོར་་་འདི་ཡིས་བསྲུང་བ་བྱ། །རང་གི་ལྷ་ཡི་སྦྱོར་བ་ཡི། །སྤྱི་བོར་ཧཱུྃ་ལས་རྡོ་རྗེ་བསྐྱེད། །དེ་ཡི་ལྟེ་བར་ཟླ་བ་ལ། ཌ། མཱ་མ་ཀཱི་ནི་སྔོན་བཞིན་བསམ། །དེ་ཡི་ཡེ་ཤེས་ལྷ་མོ་ནི། །ཐུགས་ཀར་མོ་ལ་བཟླས་པའི་སྔགས། །ཡང་འགྲེང་རིག་བཅས་བསྐོར་བར་བསམ། །དེ་ཉིད་བཟླས་པས་འོད་ཟེར་འཕྲོས། །དེ་ཡིས་སྤྱི་བོ་ནས་བརྩམས་ཏེ། །རྐང་མཐིལ་བར་གྱི་རྡོ་རྗེ་ཡི། །གོ་ཆས་བར་མཚམས་མེད་གཏམས་ཏེ། ཚོ། འབྱི་ལོ་ཀ་ནི་ཆུར་བླུགས་བཞིན། ཀོ། སྦྱོར་བ་ཡང་དང་ཡང་བསམས་ཏེ། ་། བདག་དང་གཞན་ནི་སྲུང་བར་བྱེད། །ོ། འཇིགས་པ་སྟོང་གསུམ་གང་བ་ཡང་། །སྲུང་བྱེད་སྟོབས་སུ་འགྱུར་བར་བཤད། །མངོན་རྟོགས་ཉེ་བའི་རྣལ་འབྱོར་པས། །ལས་ནི་བཞི་སྟེ་གཅིག་ལ་གསུམ། ཁ། བསྒྲུབ་ཕྱིར་ཉེ་དང་མི་ཉེ་བས། །བསམ་གཏན་རི་མོ་བསྲུང་བའི་ཕྱིར། །།རང་བཞིན་དག་པའི་ཆོས་རྣམས་ཀྱི། །དོན་དང་་་འབྲས་བུར་བཅས་པ་རྣམས། །དག་པའི་ཁྲོ་བོ་རྣམས་ཀྱི་ནི། །བསྒོམ་པའི་ཆོ་་ག་བཤད་པར་བྱ། ཌ། སྔོན་བཞིན་རྟེན་གྱི་དཀྱིལ་འཁོར་དང་། །།དང་པོ་མགོན་་པོ་བསྐྱེད་བྱས་ནས། །སེམས་ཅན་སྐྱེ་རིམ་བཞི་སྤངས་ནས། །།བར་མའི་སེམས་ཀྱིས་དང་པོར་འགྱུར། །།།བསྐུལ་བས་ཧཱུྃ་ལྔ་ཧཱུཾ་ེདུ་བྱུང་། ་་། །ཧཱུྃ་ལྔ་ལྕགས་ཀྱུ་དབྱུག་ཏོ་དང༌། །རྩེ་གསུམ་རལ་གྲི་རྡོ་རྗེར་འགྱུར། །དྡྷཾ་ནི་འཁོར་ལོ་རྩིབས་བརྒྱད་འགྱུར། །དེ་ཡང་དེ་དག་རྣམས་ཀྱིས་མཚན། འོ། འོད་ཀྱི་ཁམས་བཞུགས་སངས་རྒྱས་རྣམས། །རང་དང་འདྲ་:བས་བསྐུལ་ཅིབྱས་ཏེ། ་། བཅུག་སྟེ་ས་བོན་ནང་དུ་ཐིམ། །ཊཱཀྐི་རྒྱལ་པོ་ཁྲོ་བོ་ཆེ། །།སྐུ་དང་དབུ་ཞལ་མཐིང་ག་ལ། །།
