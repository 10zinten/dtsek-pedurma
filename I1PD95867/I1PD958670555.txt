བྱ་སྟེ་སྔགས་འདི་བརྗོད་དོ། །ཨོཾ་པྲ་ཏི་:སྲྀནྣ་སཱ་མཾ་མ་ཧཱ་བ་ལ་ཞེས་བརྗོད་པར་བྱའོ༑། ཕྱག་མཚན་ཉེའམ་ཞེས་པ་ནི་རེའུ་ཆའི་གནས་ལས་ནི་ཤེས་པར་བྱ་སྟེ་སྐུ་དང་དབུ་ལ་བབ་ོན་མཆོག་གོ། །ཞལ་ལ་བབ་ན་སྔགས་ཀྱི་དངོས་གྲུབ་པོ། །སྐུའི་དབུས་སུ་བབ་ན་ཐ་མལ་པ་བརྒྱད་དོ། ་། སྨད་དུ་བབ་ན་ལས་རྣམས་འགྲུབ་པོ༑། ཉེ་ན་དངོས་གྲུབ་ཀྱང་ཉེའོ། །རིང་ན་དངོས་གྲུབ་ཀྱང་རིང་ངོ་། རྗེ། བར་དུ་བབ་ན་ཉེ་བའི་རིགས་སོ། །དཀྱིལ་འཁོར་ལས་ཐལ་ན་དད་པ་ཆུང་ངུ་་སྟེ་དམན་པའི་དངོས་གྲུབ་པོ། །སྙིང་པོ་ལ་བབ་ན་གང་ཡང་འགྲུབ་པོ། །རེ་ཁཱ་ལ་ཕོག་ན་ཡང་དོར་རོ། །མཉམ་པའམ་བདག་གི་མདུན་དུ་ལྷུང་ན་ལན་གསུམ་་་དོར་རོ། །དེར་མ་ཆུད་ན་སླར་གཏང་ངོ་ཞེས་སོ། །སྤྱན་ལ་ཕྱག་འཚལ་འོད་་འབར་ཅན། །ཞེས་པ་ནི་མིག་གཉིས་ལ་འོད་ཟེར་འབར་བའི་ཨོ་རེ་རེ་བཀོད་དོ། །ཨོཾ་བཛྲ་སཏྭ་ས་མནྟེ་:དེ་ཏྱ་ཙཥྛདྷ་ཊ་ན་ཏ་པ་ར་པ་:ཨུཧྥུ་ཊ་ཡ་ཏི་སརྦ་ཨཀྵ་བཛྲ་་ཙཀླུ་ཨ་ནུཏྟ་ྋརཾ་ཞེས་བརྗོད་ནས་གདོང་གཡོགས་བཀྲོལ་ཏེ་ཁྱོད་ཀྱི་མིག་གིས་ཇི་་་ལྟར་མཐོང་ཞེས་དྲིས་ཏེ། དཀར་པོ་མཐོང་ན་མཆོག་ཐོབ་པའོ། །ཚོ། སེར་པོ་་་་མཐོང་ན་རྒྱས་པའོ། །དམར་པོ་མཐོང་ན་:དབང་ངོ་ལ། །ནག་པོ་ནི་མངོན་་་་་སྤྱོད་དོ། །།སྣ་ཚོགས་པ་ནི་གང་ཡང་རུང་བའོ། །ཤེས་བྱའི་རྣམ་པ་ཐམས་ཅད་ལ་མིག་ཕྱེ་བར་བསམས་ནས་དཀྱིལ་འཁོར་བསྟན་པར་བྱའོ། །ཇི་ལྟར་ཞེ་ན། །འཁོར་ལོའི་བདག་པོ་ནས་བརྩམས་ནས་བདུད་རྩི་འཁྱིལ་པ་ཡན་ཆད་དོ་ཞེས་པ་།
