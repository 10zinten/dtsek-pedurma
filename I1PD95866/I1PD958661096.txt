སེམས་ཉིད་ཀྱིས་བརྟགས། །རིམ་པས་བརྒྱུད་ཅིང་ཡང་དག་རྫོགས། །འཁོར་ལོ་བྲིས་པའི་དགེ་བ་གང་། །འདིས་འགྲོ་འཇམ་དཔལ་རྡོ་རྗེར་ཤོག། :བདག་ཀྱང་རྒྱལ་བ་ཉིད་གྱུར་ཅིག༄༅། སྡེབ་སྦྱོར་བྱས་པའི་ཚིགས་བཅད་ནི། །བཞི་བརྒྱ་ལྔ་བཅུ་པ་ཉིད་དུ། །རབ་བྱེད་འཇིག་རྟེན་ལེགས་དྲན་ཕྱིར། །དཀྱིལ་འཁོར་ཆོ་ག་འདི་བྱས་སོ། །དཔལ་གསང་བ་འདུས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག། སློབ་དཔོན་མར་མེ་མཛད་བཟང་པོའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་པདྨ་ཀ་ར་ཝརྨ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་་་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།བ::ཀྱིངབ::ཀྱིངབད་པརྒྱུབསྐྱེ་བ་དང་ནོར་འཚོ་བ། ::ཀ་ར་བའི་མཚན་པ། བསྡུར་མཆན། 《མེ་འོང་། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»དཔལ་་་་་གསང་འདུས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་སློབ་དཔོན་མར་མེ་མཛད་བཟང་པོའི་ཞལ་སྔ་ནས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་མ། «པེ་»«སྣར་རིག«སྣར་རྣམས། །«པེ་»«སྣར་པ། «པེ་»«སྣར་»ཟློས། ཚེ«སྣར་»སྙེན། «པེ་»«སྣར་»བཅལ། གྱི«སྣར་»ལོ་ཏོག«པེ་»《སྣར་»ཅིག«སྣར་»གསལ། «པེ་»«སྣར་—ཤ། «པེ་»«སྣར་»གཞོལ། «པེ་»«སྣར་ཧཱུཾ། «པེ་»«སྣར་»བ། «པེ་རྣམ་བསྒོམས་དེར། «སྣར་རྣམས་བསྒོམས་དེར། «ཅོ་རྣམ་བསྒོམས་དང་། («སྣར་»ཤེས། «སྣར་»བསྔོ། ཚེ«སྣར་»བརྒྱན། «སྣར་»ཀྱིས། «སྣར་གཞུགས། @«པེ་»«སྣར་རིག@«པེ་»«སྣར་ཕྱུང་། «པེ་»«སྣར་»—ཐོབ། «པེ་»«སྣར་ཕྱིན། ཀྱེ«པེ་»ཕྲོ། «སྣར་»ཛི། ཀྱེ«པེ་»«སྣར་»རཾ་གྱི། སྟེ«སྣར་རིགས། «པེ་»«སྣར་བྲོ་ཨོ་ཛ།  ི«སྣར་ཨ། «པེ་»«སྣར་—། «པེ་»«སྣར་»+ཁྱོད་ནི། པ་དང་། «པེ་»«སྣར་»—ཁྱོད་ནི། «བ་@«པེ་»«སྣར་»བས། «པེ་»«སྣར་»སྐྱབ། ཉི@«པེ་»«སྣར་རྗེ། །«སྣར་»གིས། ཀྱེ«པེ་»«སྣར་»རྗེ། །སྐྱེ་བ་།
