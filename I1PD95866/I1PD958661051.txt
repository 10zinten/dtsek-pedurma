་བསྟན་འགྱུར། ་རྒྱུད། ༼དིਇਸਸੁਆਲੇਨੂੰਲੈਬਧਾਥਾਉ॥ਬਿਲਕਈਲਛੀਨਾਮལ་། །རལ་་མ་ཏི་་ན་་ས་བབ་གg:བཞི་ཁོ་ར་ཆབསྡུར་མཆན། གཏེ། ཐུབ་ཚེ་ལོ་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་»སྟང་སྟབས་ཀྱི་བཀོད་པ་སློབ་དཔོན་ཆེན་པོ་་་་་སངས་རྒྱས་ཡེ་ཤེས་ཀྱི་ཞལ་སྔ་ནས་མཛད་པ་བཞུགས་སོ། །《སྣར་»སྟང་སྟབས་ཀྱིས་བཀོད་པ་བཞུགས་སོ། །«སྣར་མདུན། @«པེ་»«སྣར་»བར། ®«སྣར་མཐེབས། ། ེ«པེ་»《སྣར་བརླབ། «པེ་»«སྣར་»གདུང་། །«སྣར་»སྟང་། «པེ་»«སྣར་+སོགས་པ་ལ། «པེ་»«སྣར་»ལྟ། «པེ་»«སྣར་བ། 《སྣར་མཛུག—པོ«པེ་»«སྣར་»དགུག་པར་བྱེད། «པེ་»«སྣར་»གདམས། «པེ་»«སྣར་»མཐེབ། «པེ་གྱི། ཀྱེ«པེ་»«སྣར་»མཐེ། «པེ་»«སྣར་སྐོར། «པེ་»«སྣར་»ལ། «སྣར་»བར། «པེ་»«སྣར་བཟུང་ན། «པེ་»«སྣར་»ལྟ། «པེ་»«སྣར་བརྡུང༌། «སྣར་»གིས། ༄༅«པེ་»«སྣར་»ལག་པ། ཉི«པེ་《སྣར་»+པ་བྱེད། །«པེ་»«སྣར་»+དུ། དགྱེ«པེ་»«སྣར་དབྱུག་དང་། གདན་དང་པ་，དང་། །ས་«པེ་»འདེགས་སོ། «སྣར་འདེབས་སོ། «ཅོ་འདེབས་པའོ། «པེ་»«སྣར་རྟིང་པར། «པེ་»«སྣར་»—ཧ་ཧ། «པེ་»«སྣར་+པ། ་ག«པེ་»«སྣར་»འདུད། @«པེ་»«སྣར་»མཐེ་བོ། །«པེ་»«སྣར་»—རྡོ་རྗེ། «པེ་»«སྣར་ལ། «སྣར་ཐེབ། ཚེ«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་»པ། «པེ་»«སྣར་མཐེབ། །«པེ་»«སྣར་»—པ། «སྣར་»འདེབས། འ«པེ་»«སྣར་རིམ། ཚེ«པེ་»དཔའ་བའི། «སྣར་»དཔའ་བོའི། «པེ་»«སྣར་»དབང་པོའི། «སྣར་»དང་། །ཀུན་(«པེ་»«སྣར་པར་རྐྱང་། ཁྱི«པེ་»«སྣར་རྗེ། «པེ་»«སྣར་+གནས་ནས་ནི། །«པེ་»«སྣར་»པ་རྣམས་ཀྱི། «སྣར་»བོ། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་སྐ། །«པེ་»«སྣར་»—ཞབས། །་ཀྱི་འབ་་་་་དཔག་པ་ནི་དཔའ་བོའི་འདུལ་སྟངས་ཡིན་ལ་གསདེ་དྲག་པོ་ཉིད་དུ་་་་དུ་ད་པ་པོ་ལ་བབ་དུ་འཇོག་པ་དེ། བཟང་པོའི་འདུ་བ་་་་་་་། །ལུ་བ་བྱེད་ཀ་མཐོ་བ་དག་ལ་ག་ཡི། །།
