སྐྱེ་ལ། ར་མི་ག་ཅ་ཌ་ཀ་མེད་པདང་། དེ་བས། ད་བར་གནས་ཀྱི་བཟླས་ནས། དེ་དག་ལ་འབད་འབད་དུ་::ཀྱིང་ལོ་ཁྱད་པས་ཤིན་ཏུ་མི་ཀ་ཁཚོ་བ། འ་པ་་་་པའི་Tਲਸb'ਤੇਕਿਵੇਂਸਿਲੇਨੂੰད་པ་དཀygའདོ། ཚོ་ལ། གཞན་ਲੱਗੇਲਟਲਈད་པ་དང་། ༄༅། །གྲོལ་བའི་ཐིག་ལེ་ཞེས་བྱ་བ། །གན་བ་ཏུ་། ཀོ་ཀཌ་ཀྱི་ཟླཚོང། ་རབ། Rཚེལ་བ་ལས། ཆེ་བ་སའི་བས། ངབཔ། །ད་བ་། བྱབྷ་ཁཆུ་ཁྱོད་ཁ་དང་བྱས་ནས་ས་དྲ་བ་དང་། ལེགས་དེ་བ་ཆེ་བ་གང་དུ་གདབ་པ་དང་བྱི་ལ་ག་མ་ཚང་ཞིང་། དབྱེ་བ་དང་ཀོ་ནོཔ་ཀྱིpsའི་ལོ། ཆོ་པར་དབང་བདག་ལ་་་་་་་་་༄༅། །རྒྱ་གར་སྐད་དུ། མུཀྟི་ཏི་ལ་ཀ་ནཱ་མ། བོད་སྐད་དུ། གྲོལ་བའི་ཐིག་ལེ་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་འཁྲུལ་པ་བསལ་བས་དྲི་མ་མེད། །འོད་ཟེར་མཐའ་ཡས་རིགས་ཀྱི་་་སྤྲུལ་འཕྲོ་བ། །ཤེས་རབ་ཡན་ལག་གིས་འཁྱུད་འབར་ཞིང་མཛེས། །འཇམ་་་དཔལ་དེ་ལ་དངོས་ཀུན་རབ་ཏུ་འདུད། །དེ་ཡི་སྙིང་དབུས་གཞོམ་དུ་མེད་པ་ཡི། །དངོས་ཀུན་རོ་གཅིག་མི་ཤིགས་ཐིག་ལེའི་གཟུགས། །མི་གསལ་་གསལ་བའི་བདག་ཉིད་དུ་གནས་པ། །བརྒྱུད་རིམ་ལས་བྱུང་དེ་ཉིད་ལ་འདུད་བྱ། །དེ་་་འདྲ་སྟོན་བྱེད་:བླ་མ་གསུང་རབ་མཆོག། འཇིག་རྟེན་གསུམ་ཁོང་ནང་གནས་དྲི་མ་སེལ། །དེ་ལ་མེ་ཏོག་སྙིམ་བཀང་མགྲིན་དམའ་བས། །མཐའ་མེད་་་་་འཁོར་བ་ཡན་ཆད་བར་དུ་འདུད། །དཔལ་ལྡན་གསང་བ་འདུས་པའི་དམ་པའི་གཞུང༌། །བླ་མས་གསོལ་བཏབ་ཡིད་ཆེས་བྲོད་ལྡན་པས། །དངོས་པོ་ཀུན་གྲོལ་།
