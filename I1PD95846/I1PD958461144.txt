བསྟན་འགྱུར། རྒྱུད། ཀ«པེ་»«སྣར་»ཙ། @«པེ་»«སྣར་»+རྒྱ་གར་ཡུལ་དུ་དཀའ་ལས་ཆེར་མཛད། དགེ་ཤེས་ལོ་ཙཱ་བ་ལ། །ཡུལ་དབུས་སུ་བྱོན་ནས། བདག་གིས་གསོལ་བ་བཏབ་ནས། དཀའ་ལས་བགྱིས་ཏེ་བྱང་ཆུབ་སེམས་དཔའི་འབྲེལ་པ་བསྒྱུར། །ཞལ་ངོ་ཆེ་ནས་ཉིད་ལ་བརྫངས་པ་ལགས་ཏེ་དགྱེས་པར་དགོངས། །བྱེད་ཀུན། །ཤེས་བག་མ་གསར་དུ་སྐྱེ་བ་དགག་པའི་རྒྱ་མ་་་་་་་་རི་དང་། དེའི་ནང་ལ་བསྡུར་འབྲས་རེའུ་མིགཚོ་བ་དང་འབའ་བ་ལ། རྒྱུད། （གབ། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། ཚན་གྲངས། པོད་རྟགས། ལེའུ། ཕྲེང༌། །ལྡེབ། ནན་ན་བ་ནན?། བ་བ?། ཚེག་ཁྱིམ། །པར་བྱའི་་པ:་་་་་::པོ་མེནི། །་བ་《ཀྱིཡདེ་ལ་ལྷག་གྱི་དྲ་བ་དང་། །།ན་ལ་ཀ་རྒྱུད་པའི༌སཀྱིས་ཁོང་ལ་༽༼ལངོཀྱིལས་དོན། བ་དོ། །་ཊི་དང་དེ་བཞིན་བ::་། སྤྱི་་་་།
