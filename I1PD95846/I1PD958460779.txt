་ལོ་pའི་ཡཁ། །ཁ༽༄༅། །བླ་མ་དམ་པ་ཆོས་ཀྱི་རྒྱལ་པོ་ལ་བསྟོད་པ། པ༌་་ཚཎཎཱ་ཎའ་ཀྱི་ལ་ཁ་ཎ་ལ་བློ་ཁ་ལ་དsཆུ་ཚེ་ཁྱཁྱུ་ཚེ་ལ་ལ་ལ་ཁའི་ཚོགས་འའབུ་སའི་པཎ་། ཁང། ༽༼པོ་། ངཥྛ་ཏི་ཎཀ་ལོ། །དེའཌ་་༌ང་མ་ཙཎྜི་ཀ་::ཀoཀླ་ཚོ་དཀླུ་དeཌ་ཁ་། །ཡང་། སྐྱེས་པ་གང་ག་བྱ་ཁྱ/z་ཟོལ་༄༅། །རྒྱ་གར་སྐད་དུ། ཨ་:པ་ར་མ་ན་གུ་རུ་དྷརྨཱ་རཱ་ཛ་སྟོ་ཏྲ་ནཱ་མ། བོད་སྐད་དུ། བླ་མ་དམ་པ་ཆོས་ཀྱི་རྒྱལ་པོ་ལ་བསྟོད་པ་ཞེས་བྱ་བ། དཀོན་་་་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །རབ་རྒྱས་རྟ་ཡི་དཔུང་གི་དབའ་རླབས་ཕྲེང་བ་མཐོ་ཞིང་ཞགས་པ་རལ་གྲི་མདུང་རྣོན་གྱིས། །ཆུ་ལ་རྒྱུ་བ་རབ་མང་འཇིགས་་་རུང་རྔ་ཆེན་སྒྲ་ལྡན་ཧོར་ཞེས་བྱ་བ་དུས་མཐའ་ཡི། །ཆུ་གཏེར་འདི་ནི་མཐའ་དག་མཁས་པའི་བླ་མ་གང་གི་ཟུར་མིག་རོལ་པས་འཇིག་རྟེན་རྣམས། །:ཞི་སླད་རབ་་་མཆོག་རྒྱལ་བའི་ཚུལ་ལ་ཞི་མཛད་ཆོས་ཀྱི་རྒྱལ་པོ་དེ་ནི་རྒྱལ་གྱུར་ཅིག། ནུབ་་་་ཕྱོགས་བོད་ཀྱི་རི་ལ་ཤིན་ཏུ་དྲི་མེད་འཆར་བ་ཐོབ་ཅིང་ཤར་ཕྱོགས་དག་གི་ཡུལ། །རིམ་པས་མནན་པ་གང་གི་རྨོངས་པ་ལས་སྐྱེས་ཤིན་ཏུ་རབ་རྒྱས་མུན་པ་བཅོམ་བྱས་ནས། །སེམས་ཅན་བྱང་ཆུབ་གྲོང་ཁྱེར་ཕྱིན་སླད་ཆོས་གསུང་:འོད་ཟེར་དྲ་བས་་་རབ་མཚུངས་སངས་རྒྱས་ལ། །རབ་ཏུ་གསལ་མཛད་ཆོས་ཀྱི་རྒྱལ་པོ་མཆོག་་་་།
