པ་འདྲ་བ་ཁྲི་ཚའི་བདག་། ་བ་་་་་་ལ་ཆ་ཁོ་ལ་བྱིས་བ་གྲུ་ཀmnཞིག་བསྡུ་ཁོ་ཚོ་བ། བསམ་མཁོ། ལོ་བ། །ཚོཀྱི་རྒྱ་གར་ནི་ཤི། །།།།མཚན་བྱང་འདི་«པེ་《སྣར་སྩལྟར་བཀོད། @«པེ་»«སྣར་བ། «པེ་བགྱིད། ཚོན་D«པེ་»«སྣར་»ཀྱི། @«པེ་»«སྣར་བསྟེན། «པེ་»«སྣར་»བརྟེན། ངས་སྐབས་ཀྱི་ཕྱི་«སྣར་»ལྟར། ®«པེ་»«སྣར་»གི། ལ་ཚSའིལ:::ཀྱིངབ::ཀྱིངབ《གམ་པ་དང་པ་ཐོན་པ་ན་རྒྱུད་བསྡུར་འབྲས་རེའུ་མིག་དག་བློར་བསྟོད་ཚོགས། (བ་མ་ལུང་ཏེ། དེ་ལོ་ཚོ་དཔེ་གཞི། །སྡེ་དགེ། སྡེ་ཚན། བསྟོད་ཚོགས། །བསྟོད་ཚོགས། །བསྟོད་ཚོགས། ནང་དུ་འབག་ར་མག་སྣར་ཐང་། ཚན་གྲངས། པོད་རྟགས། ན་_བ། ནན་ཕྲེང་། ཚིག་རྐང་། ཚེག་ཁྱིམ། དག་པ་དང། །ད་ལ་བྱས་པ་དང་། དུ་བྱུང་བ། དེ་དང:ང་། དེ་་ཡང་། ད་པ་པ། མ་ཞུ་བ་དང་་་་བ་་ག་པ་བའི་ལམ་བར་ན་༽ཞེས་ཀྱིན་འདུགཞེ་མ་།
