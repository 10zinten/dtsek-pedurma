དང་ཁྱི་ས་ཁཌ་ད་ག་པཎྜི་ཏ་ང་། ཅ་པ། ཆ་ག་je། །ཏི་དང་ཀབོད་ཁགོ་ཀཚོ་བླ་མ་ཀུན་(ད། །འི༌ཀལ་ཀ་ལ་ཎབསྡུར་མཆན། /ཕྱི། ་ལས་ཤ་(ཀྱི་རྒྱུ་ཀཀཱཞིབ། ག༽ཀླུ་ཡT»ལ་གང་བ་/?མཚན་བྱང་འདི་«པེ《སྣར་ལྟར་བཀོད། «པེ་»«སྣར་ཏྲི། ་ནི«པེ་»«སྣར་»ཞེས་བྱ་བ། །རབ་«པེ་»གྱུར། «པེ་»«སྣར་པ། བལུང་«པེ་»«སྣར་»ལ་རུ། «པེ་»«སྣར་+རྣམས་ཀྱི། «པེ་»«སྣར་གི། «པེ་»«སྣར་»འཇུག། ང་གལ། དེའི་(༽ལཊ་ན་་*བ??ལ་་ལོ་ནི་། །ང་། ཌ་ད་ག་ཀང་། /::ཀྱི་༼ལ་ཆུ་ཁ་ར་ཏཁ་སྐན་ལ་། ཏཱ་ངg་ང་ལ་སྐraབསྡུར་འབྲས་རེའུ་མིགཔར་ཏུ་། /ཀཔ:/pའི་་་ར། བསྟོད་ཚོགས། (༽ཡ་་། །ད་རྒྱ་ཡིག་ཅོ་ཀའི་འོག་ཏང་#བྱz་། །ལྔ་ཎ*:དཔེ་གཞི། ཎསྡེ་དགེ། ་་པེ་ཅིན། སྣར་ཐང༌། །ཅོ་ནེ། མ་པ་དང་། རང་སྡེ་ཚན། བསྟོད་ཚོགས། བསྟོད་ཚོགས། བསྟོད་ཚོགས། ཚན་གྲངས། ་པོད་རྟགས། ལྡེབ། །བ་ནབ—ན། །ནབ། །ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། *རཁོ་བདེ་བ་་་་་་་པ་གད་བདར་གྱི་པ་ལེང་དང། ང་ཨ་བ་་་་་་མོ་་་་་འདག་ཏུ་བྱས་པའི་ཁོངས་ཀྱི་དང་ས་རྒྱུས་ལ་ལ་་་་་ཀྱངོན་པ། འབར་བ་ན་ང་། །བློ་དང་ལོ་ཉིང་། དཔལ་བ་དང་མཛད་པ་པོ། ལ་་་་་་བ། གང་བ་མེད་པ་ཡོད་ཕྱི་པགས་།
