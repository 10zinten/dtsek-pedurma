བརྒྱ་ལྔ་བཅུ་པ་ཞེས་བྱ་བའི་བསྟོད་པའི་འགྲེལ་པ། ལྔ་བཅུ་པའི་བསྟོད་པའི་འགྲེལ་པའོ། །བདེ་གཤེགས་ལ་བསྟོད་བཤད་པ་ཡིས། །བདག་གིས་དགེ་བ་གང་ཐོབ་པ། །འགྲོ་བའི་སྡུག་བསྔལ་ཞི་བའི་ཕྱིར། །དེ་ལས་བདེ་གཤེགས་འགྱུར་བར་ཤོག། བརྒྱ་ལྔ་བཅུ་པ་ཞེས་བྱ་བའི་བསྟོད་པའི་འགྲེལ་པ་དགེ་སློང་དགའ་བྱེད་སྙན་པས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་ཤདྡྷཱ་ཀ་ར་ཝརྨ་དང་། ལོ་ཙཱ་བ་ཤཱཀྱ་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།པ་འི་ས་ཏུ་ཀྱང་དམ་པ་དང་། །བདེ་བའི་འོ་ན་ལམ་བསྡུར་མཆན། aཚོ་བབནི:ཀཚ་སgaཔ་དང་ལུག་ཡིན། ད་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»བརྒྱ་ལྔ་བཅུ་པ་ཞེས་བྱ་བའི་བསྟོད་པའི་འགྲེལ་པ་བཞུགས་སོ། །《སྣར་»བརྒྱ་ལྔ་བཅུ་པ་ཞེས་བྱ་བའི་བསྟོད་པའི་འགྲེལ་པ་བཞུགས་པ་ལགས། ®«པེ་»«སྣར་»ཏ། འ«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་»བསླང་། །།།།''། «པེ་»«སྣར་»ཚོགས། @«པེ་»«སྣར་བྱ་བའོ། །།།«པེ་»«སྣར་བཅད། ་་་་་«པེ་»«སྣར་»གསལ། ཉི«སྣར་»འདིར། «པེ་»«སྣར་»སྤངས་པར། «པེ་»«སྣར་»བ་ལ་པསོགས་པའི། མི«པེ་»«སྣར་»པ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»འདས། བདག་«པེ་»«སྣར་»རིམ་གྱིས། «པེ་»«སྣར་རིམ་དུ། «པེ་»«སྣར་»ཡང་། «པེ་»«སྣར་»ཐམས་ཅད། «པེ་སུ། ས་«པེ་»«སྣར་པས། :«པེ་»«སྣར་+ནོ། «པེ་»«སྣར་»ཞེ། «པེ་»«སྣར་»བ། @«པེ་»«སྣར་»སྐྱབས་ཀྱི་དོན། །«པེ་»«སྣར་»བཟུང་། ཉི«པེ་»«སྣར་»ལྟ། ཞེས་དང་། «པེ་»«སྣར་»དེ། ཚུ«པེ་»«སྣར་»ལ། ཉི«པེ་»ཅིག་པུ། «སྣར་»ཅིག་བུ། «པེ་»«སྣར་»+དང་། དེ་དག་«པེ་»«སྣར་པར། ་ཉི«པེ་»«སྣར་»མོངྒལ། «པེ་»བརྡེག་ཅིང་། «སྣར་»རྡེག་ཅིང་། «པེ་»«སྣར་བསོས། མི«པེ་»«སྣར་»བརྡེག@«པེ་»«སྣར་»འདོད། «པེ་»«སྣར་»མོངྒལ། །།
