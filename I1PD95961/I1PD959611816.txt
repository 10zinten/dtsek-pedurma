:བསྟན་འགྱུར། སྣ་ཚོགས། ༼ཅུའི་དཀར་འོ་མའི་རྒྱུད་ཀདེ་ནི༑། བདུད་རྩི་འདྲ་བའི་སྨན་གྱི་རྣམས། །སྟོན་པའི་ཕྱག་ན་བསྣམས་གྱུར་པ། །བདེན་པའི་ཚིག་ནི་དེ་དག་གིས། །སྨན་རྣམས་བདུད་རྩིར་:འགྱུར་བར་ཤོག། དེ་བཞིན་ལྷ་མོ་འཕྲོག་མ་ཡིས། །ལྷ་རྫས་དགེ་བ་ཨ་རུ་ར། །།བླངས་ནས་སྟོན་ལ་རབ་ཕུལ་བ། །བདུད་རྩི་འདྲ་བའི་སྨན་གྱི་རྣམས། །བདེན་པའི་ཚིག་ནི་དེ་དག་གིས། །ནད་པ་རྣམས་ཀྱིས་ནད་སེལ་ཅིང་། །ཉེས་པ་ཐམས་ཅད་འཕྲོག་བྱེད་པ། །སྨན་རྣམས་བདུད་རྩིར་:འགྱུར་བར་ཤོག། སངས་རྒྱས་རྣམ་གཟིགས་གཟི་བརྗིད་དང་། །གཙུག་ཕུད་ཅན་གྱི་སྟོབས་རྣམས་དང༌། །།ཐམས་ཅད་སྐྱོབ་ཀྱི་བདེན་ཚིག་དང་། །ལོག་དད་སེལ་གྱི་ཏིང་འཛིན་དང༌། །གསེར་ཞེས་བྱ་བའི་ཡེ་ཤེས་དང་། །འོད་སྲུངས་ཀྱི་ནི་རྫུ་འཕྲུལ་དང་། །ཤཱཀྱ་སེང་གེའི་མཐུ་རྣམས་ཀྱིས། །སྨན་རྣམས་བདུད་རྩིར་:འགྱུར་བར་ཤོག། །།།:བདེ་བ་དང་། །ཀ་བ་ཏེ། ਸਨਪਸੰਪਰਮਲਸਾਇੰਸਪਸਟਹੈਕਰབསྡུར་མཆན། ཁ་བདེ་བྱེད་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད། «པེ་»«སྣར་»+དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དེའི་ཚེ་ཚིགས་སུ་བཅད་པ་འདི་དག་གསུངས་སོ། །«པེ་»«སྣར་ན། «པེ་»«སྣར་ཡིས། །«པེ་»«སྣར་»འདི༑《སྣར་»དུ། «པེ་»«སྣར་ཡི། ཉི«པེ་»«སྣར་དེ། «སྣར་»དུ། «པེ་»«སྣར་+ཟད་པ་ཆགས་བྲལ་བདུད་རྩི་འདུས་མ་བྱས། །།བར་དུ་@«པེ་»«སྣར་བཞིན། «པེ་»སྣར་»པ། «པེ་ཆེ། «སྣར་»ཆེན། «པེ་»«སྣར་»ཤིན། «སྣར་ན། «སྣར་གསུང་། «སྣར་རྣམས། «པེ་»«སྣར་གསུང་། @«པེ་»སྣར་རབ། ཁྱེ«པེ་»«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་ཡིས། འབུ«པེ་»«སྣར་གསུང༌། «པེ་»«སྣར་ཡི། སྐྱེ«པེ་སྒོམ། «སྣར་»སྐོམ། བཀྲག་ལ་འགན་འཁྲི། །:(ི«པེ་»སྣར་མི་ལྡན། «པེ་»«སྣར་གསུང་། «སྣར་»དུ། «པེ་»«སྣར་»དེ་ཀུན། །«པེ་»«སྣར་བཏང་། «པེ་»«སྣར་»གསུང་། «སྣར་བདེ། ་ཀྱི«པེ་»«སྣར་དེང་རིང༌། པ་།
