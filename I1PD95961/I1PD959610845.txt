བློ་ཀ་དགུ་བརྒྱ་སྟེ་བམ་པོ་གསུམ། དབུ་མ་དེ་ཁོ་ན་ཉིད་སྣང་བ། །བློ་ཀ་དྲུག་་་བརྒྱ་སྟེ་བམ་པོ་གཉིས། དབུ་མ་སྣང་བ། བློ་ཀ་ཉིས་སྟོང་དགུ་བརྒྱ་སྟེ་བམ་པོ་དགུ། དེ་ཁོ་ན་ཉིད་ལ་འཇུག་པའི་རབ་ཏུ་བྱེད་པ། བློ་ཀ་སུམ་ཅུ། དེ་ཁོ་ན་ལ་འཇུག་པའི་འགྲེལ་པ། བློ་ཀ་ཉིས་བརྒྱ། དེ་ཁོ་ན་ཉིད་ལ་རྩོད་པ་བཟློག་པ༑་བློ་ཀ་སུམ་བརྒྱ་སྟེ་བམ་པོ་གཅིག། རྩོད་པ་བཟློག་པའི་ཚིག་ལེའུར་བྱས་པ། བློ་ཀ་བདུན་ཅུ། རྩོད་པ་བཟློག་པའི་འགྲེལ་པ། བློ་ཀ་བཞི་བརྒྱ་སྟེ་བམ་པོ་གཅིག་དང་། བློ་ཀ་བརྒྱ། རིགས་པ་དྲུག་ཅུ་པའི་ཚིག་ལེའུར་བྱས་པ། བློ་ཀ་དྲུག་ཅུ་རྩ་གཉིས། རིགས་པ་དྲུག་ཅུ་པའི་འགྲེལ་པ། བློ་ཀ་བདུན་བརྒྱ་དྲུག་ཅུ་ྋསྟེ་བམ་པོ་གཉིས་དང༌། བློ་ཀ་བརྒྱ་དྲུག་ཅུ། སྟོང་པ་ཉིད་བདུན་ཅུ་པའི་ཚིག་ལེའུར་བྱས་པ། བློ་ཀ་བདུན་ཅུ་རྩ་བཞི། སྟོང་པ་ཉིད་བདུན་ཅུ་པའི་འགྲེལ་པ། ྋབློ་ཀ་ཉིས་བརྒྱ་བརྒྱད་ཅུ། སྟོང་པ་ཉིད་ཀྱི་སྒོ་བཅུ་གཉིས་པ་རྩ་བ་དང་འགྲེལ་པར་བཅས་པ། བློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་གཉིས། །རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བའི་སྙིང་པོའི་ཚིག་ལེའུར་བྱས་པ་དང༌། དེའི་རྣམ་པར་་་བཤད་པ། བློ་ཀ་ལྔ་བཅུ་རྩ་བཞི། ལག་པའི་ཚད་ཀྱི་ཚིག་ལེའུར་བྱས་པ། ཐོ་ཀ་ལྔ། ལག་པའི་ཚད་ཀྱི་འགྲེལ་པ་བློ་ཀ་ལྔ། ལག་པའི་ཚད་ཀྱི་འགྲེལ་པ། བློ་ཀ་ཉི་ཤུ། ཡང་ལག་པའི་:ཆད་ཀྱི་ཁྱེའགྲེལ་པ། ཐོ་ཀ་བདུན་ཅུ། བྱང་ཆུབ་སེམས་ཀྱི་རྣམ་པར་བཤད་པ། བློ་ཀ་ཉིས་བརྒྱ། །དབུ་མའི་སྙིང་པོ་མདོར་བསྡུས་པའི་རབ་ཏུ་བྱེད་པ། བློ་ཀ་བརྒྱད་ཅུ། ཆོས་ཐམས་ཅད་ངོ་བོ་ཉིད་མེད་པར་གྲུབ་པ། སློབ་དཔོན་ཀ་མ་ལ་ཤཱི་ལས་མཛད་པ། བློ་ཀ་ལྔ་བརྒྱ་སྟེ་བམ་པོ་གཅིག་དང་། བློ་ཀ་ཉིས་བརྒྱ། སྐུ་གསུམ་ལ་འཇུག་པའི་ཚིག་ལེའུར་བྱས་པ། བློ་ཀ་བརྒྱ་ལྔ་བཅུ་རྩ་དྲུག། སྐུ་གསུམ་ལ་འཇུག་པའི་འགྲེལ་པ། བློ་ཀ་བདུན་་།
