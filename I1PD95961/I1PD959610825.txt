ཆུབ་སེམས་དཔའི་སྤྱོད་པ་བསྟན་པ། བློ་ཀ་ཉིས་བརྒྱ་ཉི་ཤུ། འཕགས་པ་མར་མེ་མཛད་ཀྱིས་ལུང་བསྟན་པ། བློ་ཀ་ཉིས་བརྒྱ་བཅུ། འཕགས་པ་རིན་པོ་ཆེའི་མཐའ། བློ་ཀ་ཉིས་བརྒྱ་བཅུ། འཕགས་པ་ཁྱེའུ་རིན་ཆེན་ཟླ་བས་ཞུས་པ། བློ་ཀ་ཉིས་བརྒྱ་རྩ་དྲུག:། འཕགས་པ་སངས་རྒྱས་ཀྱི་ས། བློ་ཀ་ཉིས་བརྒྱ། འཕགས་པ་བློ་གྲོས་མི་ཟད་པས་ཞུས་པ། ་བློ་ཀ་ཉིས་བརྒྱ། འཕགས་པ་ཀླུའི་་་རྒྱལ་པོ་རྒྱ་མཚོས་ཞུས་པ་ཆུང་ངུ་། བློ་ཀ་ཉིས་བརྒྱ། འཕགས་པ་སངས་རྒྱས་་་ཀྱི་དབུ་རྒྱན། བློ་ཀ་ཉིས་བརྒྱ། འཕགས་པ་ཚངས་པས་ཞུས་པ། བློ་ཀ་ཉིས་བརྒྱ། འཕགས་པ་ག་ཡ་མགོའི་རི། བློ་ཀ་བརྒྱ་བརྒྱད་ཅུ་རྩ་བརྒྱད། འཕགས་པ་རྒྱལ་པོ་ལ་གདམས་པ། བློ་ཀ་བརྒྱ་དྲུག་ཅུ། འཕགས་པ་དེ་བཞིན་གཤེགས་པའི་བསྔོ་བ་ལ་མཁས་པ། བློ་ཀ་བརྒྱ་དྲུག་ཅུ། འཕགས་པ་ཕྱོགས་་་བཅུའི་མུན་པ་རྣམ་པར་སེལ་བ། བློ་ཀ་བརྒྱ་ལྔ་བཅུ། འཕགས་པ་ཆོས་ཀྱི་་་ཕུང་པོ། བློ་ཀ་བརྒྱ་ལྔ་བཅུ༄༅། འཕགས་པ་འཇམ་དཔལ་གནས་པ། བློ་ཀ་་བརྒྱ་བཞི་བཅུ་། འཕགས་པ་བདེ་བ་ཅན་གྱི་བཀོད་པ། བློ་ཀ་བརྒྱ་སུམ་ཅུ། འཕགས་པ་རྣམ་པར་མི་རྟོག་པར་འཇུག་པའི་གཟུངས་ཤློ་ཀ་བརྒྱ་ཉི་ཤུ་རྩ་གཅིག། འཕགས་པ་སངས་རྒྱས་མི་:སྤངས་པི། བློ་ཀ་བརྒྱ་ཉི་ཤུ། འཕགས་པ་དཔལ་དབྱིག་གིས་@ཞུས་པ། །བློ་ཀ་བརྒྱ་རྩ་བཅོ་ལྔ། འཕགས་པ་བགྲེས་མོས་ཞུས་པ་བློ་ཀ་བརྒྱ་རྩ་བཅོ་ལྔ། འཕགས་པ་འཇིག་རྟེན་གྱི་རྗེས་སུ་མཐུན་པར་འཇུག་པ་བློ་ཀ་བརྒྱ་རྩ་བཅུ་གཅིག། འཕགས་པ་སངས་རྒྱས་བདུན་པ་བློ་ཀ་བརྒྱ་རྩ་ལྔ། བློ་ཀ་བརྒྱ་མན་ཆད་ལ། །འཕགས་པ་ཡི་གེ་མེད་པའི་ཟ་མ་ཏོག་རྣམ་པར་སྣང་མཛད་ཀྱི་སྙིང་པོ། བློ་ཀ་བརྒྱ། །འཕགས་པ་གྲོང་ཁྱེར་གྱི་འཚོ་བའི་མདོ། བློ་ཀ་བརྒྱ། །འཕགས་པ་ལྷ་མོ་ཆེན་མོ་དཔལ་ལུང་བསྟན་པ། བློ་ཀ་བརྒྱ། །
