མདོ་ཀུན་ལས་བཏུས་པ། བློ་ཀ་སྟོང་ལྔ་བརྒྱ་སྟེ་བམ་པོ་ལྔ། བྱང་ཆུབ་སེམས་་དཔའི་སྤྱོད་པ་ལ་འཇུག་པ། བློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་གཉིས། རིན་པོ་ཆེའི་ཕྲེང་བ། བློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་གཉིས། དེའི་འགྲེལ་པ་བློ་ཀ་སྟོང་ཉིས་་་་བརྒྱ་སྟེ་བམ་པོ་བཞི། ཕ་རོལ་ཏུ་ཕྱིན་པ་བཏུས་པ། བློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་་་གཉིས། ལོ་ཚིགས་སུ་བཅད་པའི་དོན་བཏུས་པ། བློ་ཀ་དགུ་བརྒྱ་སྟེ་བམ་པོ་་་་གསུམ། ་སྦྱངས་པའི་ཡོན་ཏན་གྱི་ཕན་ཡོན་སློབ་དཔོན་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ། བློ་ཀ་བརྒྱ། ཤེས་རབ་བརྒྱ་པའི་རབ་ཏུ་བྱེད་པ། བློ་ཀ་བརྒྱ། བྱང་ཆུབ་ཏུ་་སེམས་བསྐྱེད་པ། བློ་ཀ་བཞི་བཅུ་རྩ་བདུན། དེ་བཤད་པ། བློ་ཀ་ཉིས་བརྒྱ། གྲོགས་པོའི་སྤྲིང་ཡིག། བློ་ཀ་བརྒྱ། དེའི་རྒྱ་ཆེར་བཤད་པ་སློབ་དཔོན་བློ་གྲོས་ཆེན་པོས་མཛད་པ། ་བློ་ཀ་སྟོང་ཉིས་བརྒྱ་ལྔ་བཅུ་སྟེ་བམ་པོ་བཞི་དང༌། བློ་ཀ་ལྔ་བཅུ། སློབ་མ་ལ་སྤྲིང་བའི་སྤྲིང་ཡིག་སློབ་དཔོན་ཙནྡྲ་གོ་མིས་མཛད་པ། བློ་ཀ་ཉིས་བརྒྱ། དེའི་དཀའ་འགྲེལ་སློབ་དཔོན་བཻ་རོ་ཙ་ན་རཀྵི་ཏས་མཛད་པ། བློ་ཀ་བརྒྱ། །རྒྱལ་པོ་ཀ་ནི་ཀ་ལ་:སྤྲིང་བའི་སྤྲིང་ཡིག་སློབ་དཔོན་:མ་ཏི་ཙི་ཏྲས་མཛད་པ། བློ་ཀ་བརྒྱ། འཕགས་པ་སྤྱན་རས་གཟིགས་ཀྱིས་དགེ་སློང་རབ་གསལ་གཞོན་ནུ་ལ་སྤྲིང་བའི་སྤྲིང་ཡིག། བློ་ཀ་བཅུ་གཉིས། གཅེས་པ་བརྒྱད་ཅུ་པ། བློ་ཀ་བརྒྱད་ཅུ། སྡོམ་པ་ཉི་ཤུ་པ། བློ་ཀ་ཉི་ཤུ། དེའི་འགྲེལ་པ་་་་སློབ་དཔོན་ཤཱནྟ་རཀྵི་ཏས་མཛད་པ། ་བམ་པོ་གཉིས། ཚིགས་སུ་བཅད་པ་བརྒྱ་པ་སློབ་དཔོན་འཕགས་པ་ལྷས་མཛད་པ། བློ་ཀ་བརྒྱ། །ཚིགས་སུ་བཅད་པའི་མཛོད། སློབ་དཔོན་ཉི་མ་སྦས་པས་མཛད་པ། བློ་ཀ་བརྒྱ། ་ལྟ་བའི་བྱེ་བྲག་སློབ་དཔོན་ཉི་མའི་འོད་ཀྱིས་མཛད་པ། ་བློ་ཀ་བརྒྱ་ལྔ་བཅུ། འཇིག་རྟེན་ཕ་རོལ་བསྒྲུབ་པ། མཁན་པོ་ཡེ་ཤེས་སྡེས་མཛད་པ། བློ་ཀ་སུམ་ཅུ། །
