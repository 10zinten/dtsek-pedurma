.བསྟན་འགྱུར། རྒྱུད། །༼ཕིའི་ཙོ་ཌ་ཏུ་བརྗོད་པར་བྱ་བའི་རྒྱུད་འདི་ཡི། །དགོངས་པ་ཉེ་བར་ལེན་པའི་འགྲེལ། །རྡོ་རྗེ་བཞད་པའི་སློབ་དཔོན་གྱིས། །སེམས་ཅན་ཡེ་ཤེས་བསྐྱེད་ཕྱིར་ཏེ། །འདི་བརྩམས་པ་ལས་བྱུང་བ་ཡིས། །དགེ་བ་དྲི་མེད་གང་ཡིན་པ། །ཡང་དག་སློབ་མ་དྲི་མེད་དང་། །སེམས་ཅན་ཀུན་གྱིས་ཐོབ་པར་ཤོག་ལོ། །རྒྱུད་ཀྱི་རྒྱལ་པོ་་་་དཔལ་གསང་བ་འདུས་པའི་དགོངས་པ་ཐམས་ཅད་ཉེ་བར་ལེན་པའི་འགྲེལ་པ་་་་ਵੱਧ.ਸਗੋਂਮਾਲਸ਼ਕਮਰਥਨਉਪਲ། རྒྱ་གར་གྱི་མཁན་པོ་ཛྙཱ་ན་གབྷ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་བཻ་རོ་ཙ་ནས་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པའི། །ང་། ལ་་་་་ལ་་ཎ་ཊ་ལྔ་ཕྲག་ཁ་ਲਝਵਸਨੂੰਸਰਵਸੋਨੇਸਵਕਮਾਈਸਖ਼ਤਵgਲਲਲਲਲਲਲਲਤੋਂਬਈARਕ/Qམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །རྒྱུད་་ཀྱི་རྒྱལ་པོ་དཔལ་གསང་བ་འདུས་པའི་རྣམ་པར་བཤད་པའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་མ་ཛ་ཊཱི་ཀ«པེ་»«སྣར་»པའི། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་+བདག«པེ་»«སྣར་བལྟམ། «པེ་»«སྣར་ཆོ། «པེ་»«སྣར་»འདུལ། S«པེ་»«སྣར་»སྤེལ་སོགས་_པ། །«པེ་»«སྣར་»—གཞན་ནི། «པེ་»«སྣར་»ཅིག་་། ཀ་པ་་མཀམ། ནཔཨ་བ་བ«པེ་»«སྣར་པོར། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ད། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་ཆེན་པོ་སྟེ། «པེ་»«སྣར་»+སེམས་ཅན་གྱི་དོན་ལ་གནས་སོ་བའི་ཐ་ཚིག་གོ། །@«པེ་»«སྣར་»། སྐྱབས་གཅིག་ཏུ་བདག་གིས། ཡ། འབར་«པེ་»«སྣར་—དག་པ། «པེ་»«སྣར་བསྐོད།  ི«པེ་»སྣར་»བསོད། །ཨེ«པེ་སྟོན། «སྣར་»སྟན། ་ ེ«སྣར་རྗེས། ན«པེ་»«སྣར་»—དུ། «པེ་»«སྣར་»སྒྲུབ། ་་། «པེ་»«སྣར་བསྒོམ། ་་་་་་། «པེ་»«སྣར་བརྟེན། «པེ་»«སྣར་»མཁའ། །«པེ་»«སྣར་»དངོས། «པེ་»«སྣར་»གཅིག«པེ་»«སྣར་»བོ། Q«སྣར་»ལས། ན«པེ་»«སྣར་ནི། «པེ་»«སྣར་»གསུང་བ། «པེ་»«སྣར་»ཀྱི། སྣར་»སྐྱེད། ན་། དེ་དང་། མང་—།
