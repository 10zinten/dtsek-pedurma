།གསུང་དང་། ཐུགས་མཉམ་པ་ཉིད་ཀྱི་སྦྱོར་བས་ཡང་དག་པར་བསམ་པར་བྱའོ། །ཀུན་:དགའ་སྙིང་པོས་རྡོ་རྗེ་མཆོག། གུས་པས་དཀའ་འགྲེལ་འདི་བྱས་པའི། །བདག་གིས་དགེ་བ་འདི་བསགས་དེས། །འཇིག་རྟེན་རྡོ་རྗེ་འཛིན་པར་་་་་ཤོག། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྐུ་དང༌། གསུང་དང་། ཐུགས་ཀྱི་གསང་བ་འདུས་པ་ཞེས་བྱ་བ་རྟོག་པའི་རྒྱལ་པོ་ཆེན་པོའི་དཀའ་འགྲེལ་་་སློབ་དཔོན་ཀུན་དགའ་སྙིང་པོས་མཛད་པ་རྫོགས་སོ། །།།::རྒྱ་གར་གྱི་མཁན་པོ་བི་ཛ་ཡ་ཤྲཱི་:བྷ་དྲ་་དང་། ཞུ་ཆེན་:ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཏེ། སླད་ནས་:ཨཱ་ཙཱརྱ་ཤདྡྷ་ཀ་ར་ཝར་མས་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །།།གང་བ་མ་འ། དྲ་ལ་རྒྱུ་བྱེད། བསྡུར་མཆན། ཎཱ་ལེ་སྤྱི་ཚོདུ། །མ་སྒྲའི་ཁང་བ་ང་དང་ང་་་་རྒྱས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ་«པེ་»«སྣར་མ། །《སྣར་»འདུག(«པེ་»«སྣར་རྫུན། «པེ་»«སྣར་»ཀྱི་གསུང་། «པེ་»«སྣར་»ལེའུའི། ན་«པེ་པོ། དམག་གཏོས། ཀ་:འུ་ཏི་་་་་་་་་་་་(«པེ་བོ། «པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་ནས། «པེ་《སྣར་—རྣམ། ན་ནི་«པེ་»《སྣར་—བསྐྱེད་པར། Q«པེ་»«སྣར་སྒྲུབ། «པེ་»«སྣར་གསལ། «པེ་»«སྣར་»། མཐོབ། »«པེ་»«སྣར་རྡོ་རྗེ། སྐད་དང་ཡོད། དེ་རིང་ན་་«པེ་»«སྣར་ཀྱི། («སྣར་བརྟོད། («པེ་»«སྣར་»བ། ་ག་བར་ན་«པེ་»«སྣར་»ལ། །»«པེ་»«སྣར་»ཏུ། «པེ་»«སྣར་»དེ། «པེ་»《སྣར་པོ། ས«པེ་»«སྣར་»པ། «པེ་»«སྣར་»ཏོ། «པེ་»«སྣར་»གི། དམ་པ་དང་། བཙུན་པ། «པེ་»«སྣར་འདོམས། ་ཀ་ནི། །ཁམས་པའི་བར། བྱ་(«པེ་»«སྣར་»དང་གཉིས་ཀའི། །(«པེ་»«སྣར་པའི། 《འ«པེ་《སྣར་བྱེད་པས། «པེ་»《སྣར་»རྒྱ། །«པེ་»《སྣར་བསྟན། «སྣར་+ན། @«པེ་»«སྣར་»—སྐུ་རྡོ་རྗེ་...ལ་འཇོག་པའོ། །(«པེ་《སྣར་བལྟོས། །(«པེ་»«སྣར་ཀཱི། «པེ་»«སྣར་རེག། »«པེ་»«སྣར་ཅིང་། །།
