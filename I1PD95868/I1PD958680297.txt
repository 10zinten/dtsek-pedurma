རྒྱུད།  ོ༼ཝིཏིདེ་བཞིན་སྙིང་ག་ནས་བྱུང་ནས། ་། འཇིག་རྟེན་ཁམས་ནི་རྣམ་པར་གཟིགས། ་ཀ་། ང་རྒྱལ་གནས་ལ་གནས་པ་ཡི། །སེམས་ཅན་ཐམས་ཅད་སྦྱོང་བར་མཛད། ཁེ། རིན་ཆེན་ཆེན་པོ་སླར་གཤེགས་ནས། །ཐམས་ཅད་དབང་བསྐུར་ཕུན་སུམ་ཚོགས། །རྡོ་རྗེ་སེམས་དཔའི་གཡས་ཕྱོགས་ཀྱི། །རིན་ཆེན་དབུས་སུ་རྣམ་པར་བཞུགས། །ཨ་རོ་ལིཀ་ཏེ། ས་བོན་དག་ལས་གྲུབ་པ་ཡིས། །རྡོ་རྗེ་སྣང་བ་མཐའ་ཡིས་་བསྐྱེད། །པདྨ་རཱ་གའི་མདོག་འདྲ་བའི། །མི་འཁྲུགས་པས་ནི་ཅོད་པན་བརྒྱན། །ཞི་བ་རལ་པ་ཅོད་པན་འཆང༌། །དམར་དང་གནག་དང་དཀར་བའི་ཞལ། །ཐམས་ཅད་རྒྱན་གྱིས་རྣམ་བརྒྱན་པའི། །ཕྱག་དྲུག་པ་ནི་བསྒོམ་པར་བྱ། ཀ༌། གཡོན་པ་འགྱིང་བས་པདྨ་འཛིན། །དྲིལ་བུ་དཀུར་བརྟེན་ཕན་ཕྱིར་ཏེ། ད། གཡས་པར་པདྨ་སྐྱོད་མཛད་ཅིང་། །ཐུགས་ཀར་པདྨ་ཁ་ཡང་འབྱེད། ་་། ཕྱག་་་གཉིས་རྡོ་རྗེ་འཁོར་ལོ་དང༌། །གཡོན་གཉིས་རིན་ཆེན་རལ་གྲི་བསྣམས། །སྙིང་ག་ནས་ནི་བྱུང་གྱུར་ནས། །ཐམས་ཅད་ཕྱོགས་སུ་རྣམ་པར་གཟིགས། །སེམས་ཅན་འདོད་ཆགས་ལྡན་སྦྱོང་སྟེ། །འདམ་ཅན་ཆུ་ཡི་པདྨ་བཞིན། པ། །སླར་་་་་གཤེགས་པདྨའི་དབུས་སུ་ནི། །རྒྱབ་ཀྱི་ཕྱོགས་སུ་མངོན་པར་བཞུགས། །པྲཛྙཱ་ཛཱིཀ། ཁ། ས་བོན་ལས་ནི་གྲུབ་པ་ཡི། ཞ། རྡོ་རྗེ་གདོན་མི་ཟ་བ་བསྐྱེད། །ལྗང་་གུ་བཻ་ཌཱུརྱ་འདྲ་བ། །།རལ་པ་ཅོད་པན་རྡོ་རྗེ་གནས། །ཐམས་ཅད་རྒྱན་གྱིས་་་་རྣམ་པར་བརྒྱན། ཚེ། །ལྗང་གུ་གནག་དང་དཀར་བའི་ཞལ། །།འོད་ཟེར་རྣམ་ལྔ་འཁྲིགས་པར་འབར། །ཕྱག་དྲུག་རྣམ་པར་བསམ་པར་བྱ། །།རལ་གྲི་རྡོ་རྗེ་རྒྱ་གྲམ་དང་། །།འཁོར་ལོ་གཡས་པའི་ཕྱག་ཏུ་བསྒོམ། །དྲིལ་བུ་པདྨའི་སྡོང་བཅས་དང༌། །ནོར་བུ་གཡོན་པའི་ཕྱག་ཏུ་བསམ། །དེ་བཞིན་སྙིང་ག་ནས་བྱུང་ནས། །ཕྱོགས་བཅུ་ཀུན་དུ་གཟིགས་མཛད་དེ། །བརྫུན་ཚིག་སྨྲ་བའི་ལམ་གནས་པའི། །—།
