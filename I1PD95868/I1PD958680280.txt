དཔལ་གསང་བ་འདུས་པའི་རྒྱུད་ཀྱི་གླེང་གཞི་བླ་མའི་མན་ངག་གིས་བཤད་པ། བཞིན་གྱིས་ཀུན་འདི་ལྟར་ལྷུན་གྱིས་གྲུབ་པའི་སྒྱུ་མ་སྟོང་པའི་སྦྱོར་གཞི་མགོན་པོ་ཐམས་ཅད་མཁྱེན་པའི་བདག་གོ་ཞེས་འདི་ལྟར་ཤེས་པར་བྱས་ན། རབ་ཏུ་་་་་འགྲུབ་པར་འགྱུར་རོ། །དཀྱིལ་འཁོར་གྱི་སློབ་དཔོན་སྒེག་པའི་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ། །།།།paR,མི་ཁ་་་་་ངོ་བོད་ཀྱི་་ཁབ་ཀྱིསྐྱེ་ཁམཤེས་བར་བ་བ་ད°བསྡུར་མཆན། ::ཀྱིང་ར་ཀླུ་ཤིང་དེའི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»དེ་ནས་གསང་བ་འདུས་པའི་རྒྱུད་འགྲེལ་བཞུགས་སོ། །«པེ་»《སྣར་»༄༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་གུ་ཧྱ་ས་མཱ་་ཛ་ཏན་ནི་དཱ་ན་གུ་རཱུ་པ་དེ་ཤ་ནཱ་རྱ་ཏ། བོད་སྐད་དུ། དཔལ་གསང་བ་འདུས་པའི་རྒྱུད་ཀྱི་གླེང་གཞི་བླ་མའི་མན་ངག་གིས་བཤད་པ། འ«པེ་»«སྣར་»བ། །།།(«པེ་»«སྣར་»+སུམ། /«པེ་»«སྣར་+ལས། །«པེ་»«སྣར་པ། «པེ་»«སྣར་ནས། ན«པེ་»«སྣར་»ལ་གནས། «པེ་»«སྣར་»བསྐྱེད། «པེ་»«སྣར་བསྐྱེད། །«པེ་»«སྣར་རྗེའི། །«པེ་»«སྣར་»། །ན@«པེ་»«སྣར་»། གྱིས་@«པེ་»«སྣར་»+ནི། ན«པེ་»«སྣར་ཡ། «པེ་»«སྣར་»ཤུ། །དམ་«པེ་»«སྣར་»+པ། «པེ་»«སྣར་མ་རྣམས། «པེ་»«སྣར་དེ། «པེ་»«སྣར་དང༌། །«པེ་»«སྣར་ཞུགས། «པེ་»«སྣར་ཤུ། «པེ་»«སྣར་ཐཾ། «པེ་»«སྣར་»དོ། །ན་«སྣར་»གསུམ། «པེ་»«སྣར་»གི། འ«པེ་»«སྣར་གསལ། «པེ་»«སྣར་»སོ། «པེ་»«སྣར་པ། པ«པེ་»«སྣར་»གི། (@«པེ་»«སྣར་པ། «པེ་»«སྣར་»གི། «པེ་»«སྣར་»—རྒྱུར་བྱེད་པའི། །«པེ་»«སྣར་»+ཞེས། «པེ་»«སྣར་»སྐྱེད། «པེ་»«སྣར་»+ཞེས། ཆ«པེ་»《སྣར་»བྱ་བའི། (ི«པེ་»«སྣར་མྱོང་། «པེ་»«སྣར་»པར། །«པེ་»«སྣར་»དེ་ནི། «པེ་»«སྣར་»དགདེ་ཕྱིར་བདག་ལ་དལ་(«པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་»ཀྱི། ་འ«སྣར་བསྐྱེད། །ཚེ«པེ་»«སྣར་»སྒྲུབ། སུ«པེ་»«སྣར་»བརླབ། ཀྱི«པེ་»«སྣར་པོ། «པེ་»«སྣར་»ནི། «པེ་»«སྣར་»—པ། །
