་་༄༅། །དབུ་མའི་མན་ངག་ཅེས་བྱ་བའི་འགྲེལ་པ། ་མང་བ་་བྱེད་༄༅། །རྒྱ་གར་སྐད་དུ། མ་དྷ་མ་ཨུ་པ་དེ་ཤ་བྲྀཏྟི། བོད་སྐད་དུ། དབུ་མའི་མན་ངག་ཅེས་བྱ་བའི་འགྲེལ་པ། འཕགས་པ་འཇིག་རྟེན་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །འགྲོ་བའི་སྡུག་བསྔལ་སེལ་མཛད་ཅིང༌། །དཀར་པོ་རྣམ་་པར་འཕེལ་བའི་རྒྱུ། །བྱང་ཆུབ་སེམས་ལ་ཕྱག་འཚལ་ནས། །དབུ་མའི་མན་་་་ངག་གསལ་བར་བྱ། །ཕྱིན་ཅི་ལོག་གི་རྣམ་རྟོག་གིས། །འཁོར་བའི་འདམ་དུ་བྱིང་བ་རྣམས། །མན་ངག་ལམ་ལ་རབ་བརྟེན་ནས། །ཡང་དག་བྱང་ཆུབ་བསྒྲུབ་པའི་ཕྱིར། །མན་ངག་དོན་ཙམ་བཤད་པར་བྱ། །།དེ་ལ། གང་གི་གསུང་གི་འོད་ཟེར་གྱིས། །ཞེས་བྱ་བ་ལ་སོགས་པས་ནི་ཡུལ་ཡོན་ཏན་དང་ལྡན་པ་ལ་ཕྱག་འཚལ་བ་རྩོམ་པར་བྱེད་དོ། །དེ་སློབ་་་་དཔོན་:དེ་དག་ཉིད་ཡ་རབས་དམ་པ་ཉིད་དུ་ཤེས་པར་བྱ་བའི་ཕྱིར་རམ། བར་ཆད་ཀྱི་བགེགས་ཞི་བར་བྱ་བའི་ཕྱིར་རམ། བཤད་པར་དམ་བཅའ་བའི་ཕྱིར་།
