སོགས་པའི་སྤྱོད་པ་སྤྱད་ཅིང་། ་འདི་ལྟར་སངས་རྒྱས་བཅོམ་ལྡན་འདས་རྣལ་་་་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་སྐྱིལ་མོ་ཀྲུང་བཅས་ཏེ། བཞུགས་པ་བསམ་གཏན་་་གྱི་སྙོམས་པར་ཞུགས་ཤིང་འགྲོ་བ་ལ་གཟིགས་པ་མདུན་དུ་བསམ་པར་བྱའོ། ་་། རྣལ་འབྱོར་བསྒོམ་པའི་ལམ། ་འཕགས་པ་འཇམ་དཔལ་གྱི་ཞབས་ཀྱི་པདྨའི་་་ཟེའུ་འབྲུ་སྤྱི་བོས་ལེན་པ། སྡེ་སྣོད་གསུམ་ལ་མཁས་པ་སློབ་དཔོན་ཡེ་ཤེས་་་་སྙིང་པོས་མཛད་པ་རྫོགས་སོ། །།།གང་བ་བརྒྱད་ལ་ཡི་དམ་ལ་སྐྱེབར་་བ་འག་པ། ང་ཡི་དམ་དུ་བྱ་བར་གཙོ་མཔ། ཁ་ད་ར་ད་ར་ཁ་བ་དན་གནམ་ལ་ཁྱཚེ་དབང་བདག་དེས་ན་མ་བསྡུར་མཆན། བཀ་ར་བྱེད་ཀ་དེ་ལ་ཕྱིན་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། །རྣལ་འབྱོར་སྒོམ་པའི་ལམ་ཡེ་ཤེས་སྙིང་པོས་མཛད་པའོ། །ཞེས་འཁོད། «པེ་»«སྣར་»སྒོམ། ®«པེ་»«སྣར་»ལ། &«པེ་»«སྣར་»ལྗོངས། ་སུ«ཅོ་»འགོད། «སྣར་»ཀྱིས་ཚན། །ཤི། :/«སྣར་—པ། «པེ་«སྣར་»བཏགས། «པེ་»«སྣར་རྟེན། «པེ་»«སྣར་»«ཅོ་»དགག«པེ»དུ་བཏགས། 《སྣར་»དུ་བརྟགས། «པེ་《སྣར་བཞག་པས་ཆད་པ། «པེ་»«སྣར་བསླངས་པའི། །«པེ་+ཉིད་ཀྱི། «པེ་»«སྣར་ཡང་དག་པར། «པེ་»«སྣར་»—ཞི། «པེ་»«སྣར་»ཤེས། «པེ་»སྣར་བཞག（པོད་འདིའི་ནང་དུ་གཞག་ཅེས་པ་རྣམས་«པེ་»«སྣར་»ལ་་་བཞག་ཅེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་པའི་རྒོད་པར་འགྱུར། ་ག«པེ་»《སྣར་པར། «པེ་»«སྣར་»བ། «པེ་»«སྣར་བླང་། «པེ་»«སྣར་གིས། «པེ་»«སྣར་»དག་ནི། «པེ་»«སྣར་ཤེས་རབ། O«པེ་»«སྣར་»ན། «པེ་»«སྣར་»གྱིས་དམིགས། «པེ་»«སྣར་བཏང་། »«པེ་»«སྣར་»«ཅོ་པས། པོ«པེ་»«སྣར་»པར། །O«པེ་»«སྣར་བསྐྱེད། ཉི«པེ་»«སྣར་ཀྱི། «པེ་བསྡུ་བར་བྱའོ། །ཞེས་སེམས་ཅན། «པེ་»«སྣར་བསྐྱེད། ག«པེ་»«སྣར་བསམས་པའི་དོན་སྒྲུབ། @«པེ་»«སྣར་»—འདི། ཚེ«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་རྡུབ། «ཅོ་བརྔུབ། །།
