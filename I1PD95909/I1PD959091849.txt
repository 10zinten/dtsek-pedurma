བསྟན་འགྱུར། དབུ་མ། ཁིབྱེད་ཅིང་། །དེ་ཡིས་ཉེས་པ་གྲངས་མེད་སྐྱེད། ངོ་། །འཁོར་བའི་སྐྱོན་ནི་མཐའ་ཡས་ཀྱིས། །སྔོན་མཐའ་ད་ལྟར་ཕྱི་མཐའ་རུ། །ོ། གདུང་བྱས་གདུང་བྱེད་གདུང་བྱེད་འགྱུར། །དེ་དོན་རྟག་ཏུ་བསྒོམ་བྱ་ཞིང་། །བཙོན་ར་ལྟ་བུའི་འཁོར་བ་ལ། །བཙོན་ལྟར་སྐྱོ་བ་གོམས་པར་བྱ། །ལ་། ཇི་ལྟར་བཙོན་ནི་བཙོན་ར་ཡི། །སྡུག་་བསྔལ་མང་མྱོང་དེ་དྲན་པས། །སྐྱོ་སེམས་རྟག་ཏུ་ངང་གིས་འབྱུང་། །ཇི་ལྟར་བཙོན་ནི་བཙོན་ར་ལས། ཀོ་། འབྲོས་པའི་གླགས་ནི་བྱུང་བའི་ཚེ། །ཆེད་ཀ་གཞན་དང་མི་སྙོམས་པར། །དེ་ཡི་གནས་ནས་འབྲོས་པ་ལྟར། །འཁོར་བའི་རྒྱ་མཚོ་་ཆེ་འདི་ལས། །རྒལ་ཅིབའི་གླགས་དག་བྱུང་གྱུར་ན། །།ཆེད་དཀའ་གཞན་དང་མི་སྙོམས་པར། །སྲིད་པའི་ཁྱིམ་ནས་འབྱུང་བར་བྱ། །བདག་གཞན་འཁོར་བའི་སྡུག་བསྔལ་དག། མཐར་དབྱུང་ཕྱིར་ཡང་བློ་ལྡན་དག། འདོད་པ་མེ་བཞིན་མྱུར་སྤངས་ཏེ། ཌ། ཆོས་སྒྲུབ་གཙོར་བྱ་མི་རིགས་སམ། ་་། སྒྲུབ་པར་འདོད་པའི་གང་ཟག་གིས། །ས་ལྟར་ཆོས་ཉིད་གཞིར་གྱུར་པས། །དང་པོ་སྐྱབས་སུ་་ལེགས་འགྲོ་ཞིང་། །དེ་ཡི་བསླབ་པའི་ཆོས་ལ་བསླབ། །རྗེས་སུ་དྲན་དྲུག་རྟག་ཏུ་བསྒོམ། །སངས་རྒྱས་ཡོན་ཏན་མཐའ་ཡས་པ། །སྒྲུབ་པ་པོས་ནི་དྲན་བྱས་ལ། །དེ་ལ་དད་པ་བརྟན་པར་བྱ། །ཆོས་ཀུན་རྩ་བ་དད་པ་ཡིན། །རྩྭ་རྩེའི་ཟིལ་ཐིགས་ལྟ་བུར་ནི། །རིང་དུ་མི་གནས་ཚེ་འདིའི་ཕྱིར། །མཁས་པས་སྡིག་པ་ཡོང་མི་སྤྱོད། །དེ་འབྲས་ངན་འགྲོའི་སྡུག་བསྔལ་འབྱུང༌། །དགེ་བཅུ་ཚུལ་ཁྲིམས་དག་བྱ་ཞིང་། །སོ་སོར་ཐར་པའི་སྡོམ་དག་ཀྱང་། ད། །ཅི་ནུས་རིམ་གྱིས་དག་པར་བསྲུང་། །དེ་ནི་འཕགས་ནོར་ཀུན་གྱི་རྒྱུ། ཎ། མང་ཐོས་ནོར་ཡིན་དེ་ལྡན་བྱ། །ཚུལ་ཁྲིམས་ནོར་ཉམས་གྱུར་པ་བཞིན། །མང་ཐོས་ནོར་ལས་ཉམས་པའི་མི། །།ཕྱི་ཕྱིར་འགྱོད་པར་གདོན་མི་ཟ། །རང་ལ་ང་རྒྱལ་གཞོམ་པ་དང་། །_།
