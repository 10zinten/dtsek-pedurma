འ་ཎ་བསྟན་འགྱུར། དབུ་མ། Pཌ་ལ་་་་་་་་་་་་་་་་་་་་་་་་་་་ཀློག་དང་བར་ལ་གྲང་ཞོ་། ਸੰਸਥਾਨੂੰਸਬਕਸpਲਓਧਕਵੀਰਲੋਂਬਲਨਾབ་ལ་བསྡུར་མཆན། བློན་ངན་ཁྲི་ལ་བྱ་ཁགཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་བྱང་ཆུབ་་་སེམས་དཔའ་ལས་དང་པོའི་ལམ་ལ་འཇུག་པ་བསྟན་པ་བཞུགས། ཞེས་འཁོད། «པེ་»«སྣར་པོའི། ནམ་ག་། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་གཤེས། :«པེ་»པུ་ཐེག་པ། «སྣར་པུ་ཐེག«པེ་«སྣར་གི་དམ་པ་གཅིག་«པེ་»«སྣར་»+པ། «པེ་ཀྱིས། ན«པེ་»«སྣར་»བསྡམས། ༄༅«པེ་»«སྣར་འཇིག་གྱི«པེ་《སྣར་»གི། @«སྣར་—དང༌། @«པེ་»«སྣར་»—མེད་ཅིང་། རྒྱུ་«པེ་»«སྣར་»སྐོར། «པེ་»«སྣར་མཆིའོ། །བདག་གི། «པེ་»«སྣར་»+དེ། །«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་མ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་རྟོགས། །«པེ་»ཀྱིས། «པེ་»«སྣར་བསམ། ཚེ«པེ་»«སྣར་བའི། «པེ་»«སྣར་—རླུང་ལ་སོགས་པའི་་དང་། ནད་གནས་སུ་ནི། །«པེ་»སྣར་བསྒོ། (ཚེ«པེ་»«སྣར་»སྤྱོད། (ཀྱེ«པེ་»«སྣར་ཆ། བུ«པེ་»«སྣར་རོགས། (ཉེ«པེ་»«སྣར་བཞན། «པེ་»«སྣར་ལྟ་བུའི། «པེ་»«སྣར་སྐྱེད། ཨི«པེ་»«སྣར་+དེ། «པེ་»«སྣར་ཀྱི། །ཕྱི་ཕྱོགས་བལ་བུ་ཁ་ནང་ལ་། E་ལ་ངོ་ས་བ་ཡིད་ਛਲਤੋਂਸਾਲਾਂਲਸਣਨੂੰਇਸཔར། ལ་། ངད། ༼ཀ་་བསྡུར་འབྲས་རེའུ་མིག་དེ། བ་རུ་རྒྱུ་བ་དབུ་མ། （ཡ་ཟླ་sདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཏ་ཅོ་ནེ། །སྡེ་ཚན། །དབུ་མ། །།བསླབ་པ་ཀུན་བཏུས། །མདོ། ,དབུ་མ། འབྱུང་བོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན?བབ?བ་བན། བབ་ཕྲེང་། །ཚེག་ཁྱིམ། །།།།།_།
