སྤྱོད་ལམ་གྱི་གནས་དང་། སྤྱོད་ལམ་རྣམ་པར་མི་འཁྲུགས་པ་དང་། ལུས་ཀྱི་ལས་དང་། རིགས་ངག་གི་ལས་དང་། ཡིད་ཀྱི་ལས་ལེགས་པར་སྤྱད་པ་དང་། །དབང་པོ་དྲུག་སྡོམ་པ་དང་། རང་གི་ལུས་གཡོགས་པ་དང༌། མཉེ་བ་དང་། བཀྲུ་བའི་ལས་དང་། ཟོས་པ་དང༌། འཆོས་པ་དང༌། མྱངས་པ་དང༌། བསྐུམ་པ་དང་། བརྐྱང་བ་དང༌། ལྟ་བ་དང་། རྣམ་པར་ལྟ་བ་དང༌། །གཉིད་ཀྱིས་ལོག་པ་དང་། །གཉིད་ཀྱིས་ཅིམ་ལོག་པ་དང༌། སྐུ་རང་གི་ལུས་སུ་གཏོགས་པའི་བསྙེན་བཀུར་:ཅི་གང་ཡང་རུང་སྟེ། དེ་ཐམས་ཅད་ཐམས་ཅད་མཁྱེན་པ་ཉིད་དུ་ཡོངས་སུ་མ་བསྔོས་པ་ཅུང་ཟད་ཀྱང་མེད་དོ་ཞེས་བྱ་བ་ནས། རང་གི་བ་རྣམས་དང་དེ་ལས་གཞན་དུད་འགྲོའི་སྐྱེ་གནས་སུ་སོང་བ་རྣམས་ལ། འབྲས་:ཆན་པོ་ཚུམ་གང་ངམ་ཟན་:ཆང་ཅིག་ཨིཙམ་ཡོངས་སུ་གཏོང་བ་ཡང་སྐྱེ་བ་དེ་དང་དེ་དག་ཏུ་དེ་དག་ཉིད་ལ་ཕན་པ་དང་དེ་དག་ཉིད་ཡོངས་སུ་དགྲོལ་བའི་ཕྱིར་དེ་ཐམས་ཅད་ཡོངས་སུ་བསྔོའོ། །སེམས་ཅན་དེ་དག་དུད་འགྲོའི་སྐྱེ་གནས་དེ་དང་སྡུག་བསྔལ་གྱི་རྒྱ་མཚོ་དང་སྡུག་བསྔལ་ཉེ་བར་ལེན་པ་དེ་དང་སྡུག་བསྔལ་གྱི་ཕུང་པོ་དེ་དང་སྡུག་བསྔལ་གྱི་ཚོར་བ་དེ་དང་། སྡུག་བསྔལ་ཉེ་བར་བསགས་པ་དེ་དང་། སྡུག་བསྔལ་མངོན་པར་འདུ་བྱེད་པ་དང་། །སྡུག་བསྔལ་གྱི་གཞི་ཉེདང་། །སྡུག་བསྔལ་གྱི་རྩ་བ་དེ་དང༌། སྡུག་བསྔལ་གྱི་གནས་དེ་ལས་རྣམ་པར་བཟློག་པའི་ཕྱིར་ཡོངས་སུ་བསྔོའོ། །དེ་ལ་དམིགས་པས་སེམས་ཅན་ཐམས་ཅད་་ལ་དམིགས་པར་བྱེད་དོ། །སེམས་ཅན་རྣམས་ལ་མངོན་དུ་བྱེད་དོ། །ཡིད་ལ་བྱེད་དོ། །འདི་ལྟ་སྟེ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་དུ་ཡོངས་སུ་སྔོ་བར་བྱེད་པས་་་དགེ་བའི་རྩ་བ་དེ་ལ་མངོན་དུ་འགྲོ་བར་བྱེད་དོ། འོ་། བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་་པས་རབ་ཏུ་འཛིན་ཏོ། །དགེ་བའི་རྩ་བ་དེ་ལ་སྦྱོར་རོ། །།འཁོར་བའི་དགོན་པ་།
