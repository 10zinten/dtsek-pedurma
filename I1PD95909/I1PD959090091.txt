བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམ་པ་རྫོགས་སོ། །།།ཞ་ལུ་ལོ་ཙཱ་བ་དགེ་སློང་དྷརྨ་པཱ་ལ་བྷ་དྲ་ཞེས་བགྱི་བས་གྲྭ་ཐང་དུ་བསྒྱུར་བའོ། །།།པ་དང་བར་་་་་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«པེ་»«སྣར་»ལ་མི་བཞུགས་པས་«ཅོ་དང་དཔེ་བསྡུར་བྱས་ཡོད། «ཅོ་མ་གྱུར། །དེ་དེ་བས་ནད་པའི་«ཅོ་»གྱི། བསྡུར་འབྲས་རེའུ་མིགདབུ་མ། (དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། དབུ་མ། དབུ་མ། སྡེ་ཚན། །པོད་རྟགས། །ཚན་གྲངས། ལྡེབ། ཕྲེང་། ཚེག་ཁྱིམ། །བ་—ན་་། །བ། ན་ལོ་། །འི་དེ་བས་མང་བ་དང་ལྔ་།
