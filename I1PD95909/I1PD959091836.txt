་་་་་་་་་་་་་་་་་་:,་་་་་་་། :བྱེ་ན། །དད་པ་ལ། ནང་བ་དང་། ཚེ་བློ་དང་མ་ནང་། ::ཀྱི་མ་འོངས་ལ། ད་བར་༄༅། །བྱང་ཆུབ་སེམས་དཔའ་ལས་དང་པོ་པའི་་ལམ་ལ་འཇུག་པ་བསྟན་པ། ?ཁ་བཏུས། ན་བ་བན་པ་འགལ་བ་ਸੈਕਸਲੈਸਗਲੇਨਾਲਨਾਲਉਸ༄༅། །རྒྱ་གར་སྐད་དུ། བོ་དྷི་སཏྭ་ཀརྨ་དི་མརྒ་པ་ཏཱ་ར་དེ་ཤ་ན། བོད་སྐད་དུ། བྱང་ཆུབ་སེམས་དཔའ་ལས་དང་:པོ་པའི་ལམ་ལ་འཇུག་པ་བསྟན་པ། བླ་མ་དམ་པ་ལ་ཕྱག་འཚལ་ལོ། །ངན་འགྲོ་འཁོར་བ་འཇོམས་པ་དང་། །མངོན་མཐོ་ངེས་འབྱུང་རྩོལ་མཛད་པ། །ཕྱོགས་བཅུའི་དཀོན་མཆོག་ཐམས་ཅད་དང༌། །བླ་མའི་ཞབས་ལ་ཕྱག་འཚལ་ལོ། །འདི་ན་གང་དག་རང་བཞིན་གྱིས་སྙིང་རྗེ་དང་ཤེས་རབ་ཆེ་བ་ཐེག་པ་ཆེན་པོའི་རིགས་ཅན་དང་། ཡང་གང་དག་སྐྱེ་བ་སྔ་མ་དག་ཏུ་ཐེག་པ་ཆེན་པོ་ལ་གོམས་པར་བྱས་པའི་སྐྱེས་བུ། འཁོར་བ་ལ་ཡིད་ལེགས་པར་བྱུང་བ་འཆི་བ་རྗེས་སུ་དྲན་པ། ཕྱི་ནང་གི་དངོས་པོ་ཐམས་ཅད་ལ་ཞེན་པ་ཆུང་བ་དང་། དགེ་བའི་བཤེས་ཅིགཉེན་དམ་པ་ཅིག་་:：ལ་ཐེག་པ་ཆེན་པོའི་ལམ་ཕྱིན་ཅི་མ་ལོག་པ་ཅིག་ཐོས་པར་གྱུར་ན་ཅི་མ་རུང༌། བདག་:གིས་དམ་པ་ཅིག་བཙལ་བར་བྱའོ་སྙམ་དུ་སེམས་པ་དེས། དགེ་བའི་—།
