བསྟན་འགྱུར། དབུ་མ། །ཁིཉེས་པ་མེད། ་། གནས་བརྟན་མར་མེ་མཛད་དཔལ་གྱིས། །མདོ་སོགས་ཆོས་་ལས་བཤད་མཐོང་བ། །བྱང་ཆུབ་འོད་ཀྱིས་གསོལ་བཏབ་ནས། ཁོ་། བྱང་ཆུབ་ལམ་བཤད་མདོར་བསྡུས་བྱས། །བྱང་ཆུབ་ལམ་གྱི་སྒྲོན་མ་སློབ་དཔོན་ཆེན་པོ་དཔལ་མར་མེ་མཛད་ཡེ་ཤེས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།་་་་་རྒྱ་གར་གྱི་མཁན་པོ་ཆེན་པོ་དེ་ཉིད་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཞིང་གཏན་ལ་ཕབ་པ། ཆོས་འདི་ནི་ཞང་ཞུང་གི་ཐོ་ལིང་། གཙུག་ལག་ཁང་དུ་མཛད་པའོ། །ལ་། འལ་ལན་ਨਵਸਸਿਰਤੇਕਨਿਰਮਲਸਿਉਸਤਾਰਡਕਸਨਸਸਸ་་ང་ནིར་»ཀྱིས། ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་ལ་བྱང་ཆུབ་ལམ་སྒྲོན་ཇོ་བོ་རྗེས་མཛད་པ་བཞུགས་སོ། །ཞེས་དང་། 《སྣར་ལ་བྱང་ཆུབ་ལམ་སྒྲོན་ཇོ་བོ་རྗེས་མཛད་པ། ཞེས་འཁོད། «པེ་»«སྣར་པ་ཐཱ། ་་། @«པེ་»«སྣར་བཟངས། ®«པེ་»«སྣར་མོའི། @«སྣར་རྒྱུན་ཆིན། «པེ་ལྟོག«པེ་»《སྣར་བྱའོ། «པེ་»«སྣར་»ཀྱིས། གྱེ«པེ་»«སྣར་ནས། «པེ་གང་གའི་བྱེ་མ། «སྣར་གང་གཱའི་བྱེ་མ། ཉི«པེ་»«སྣར་»གི། «པེ་»«སྣར་»རིགས། «པེ་»«སྣར་ལེའུར། «པེ་»«སྣར་མ་པ་ར། གསར་པ་དང་། ད་ལ་་ས་གནང་བ་དང་ ེ«པེ་»«སྣར་བྱས། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་བསྒོམ། ཚེ«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»ཅིག། «སྣར་»ཀྱིས། ་«པེ་»«སྣར་བསྒོམ། ཚེ«པེ་»«སྣར་བསྒོམ། ་་མ་བ་ཞི་ཁྲོ་ཚེ«པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་»རྟོགས། «པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་»+ངེ་། «པེ་»«སྣར་—པ། །«པེ་»«སྣར་བསམ། «པེ་»«སྣར་བཀུར། «པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་»བླངས་། «པེ་»«སྣར་བསྲེག། ཚེ«པེ་»«སྣར་»གྱི། «སྣར་+ཤུ་བྷཾ། །ਮਸ਼ੀਸ਼ਰੀਜੇਸਨਕਰਮਕਾਹੂਨਮੁਸ਼ਕਲਨਾਲੋਂ།
