པས། །ཁམས་གསུམ་སེམས་ཅན་མཐའ་ཡས་ཇི་སྙེད་ཀུན། །ཤེས་རབ་མིག་་ཐོབ་དབུ་མའི་དོན་རྟོགས་ནས། །ཐམས་ཅད་མཁྱེན་པ་མྱུར་དུ་ཐོབ་པར་ཤོག། ཅིག་ཅར་འཇུག་པའི་རྣམ་པར་མི་རྟོག་པའི་བསྒོམ་དོན་སློབ་དཔོན་ཆེན་པོ་བི་མ་ལ་མི་ཏྲས་མཛད་པ་རྫོགས་སོ། །།།འད། ཅི་ལ་ཡོང་བ་དང་། དེ་སྔ་བོད་:རྒྱ་གར་གྱི་མཁན་པོ་དྷརྨ་ཏཱ་ཤཱི་ལ་དང་། ཞུ་ཆེན་གྱི་:ལོ་ཙཱ་བ་བནྡེ་ཡེ་་་ཤེས་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཚ་། །བ་ཀཞན་ཏཚོ་ཁོbཏོ། ཅཁ་བ་ཚེ་ཚེ་བདུན་རྒྱ་་། ཙོ། །ག་ད་བ་ལ་བ་ད་ན་བ་ཚོ་ཏུ་སློབ་བ་ཚབ། དེ་དང་བསྡུར་མཆན། ་བ་དང་། །ལྟ་བ་ཕྱོགས་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། །མི་རྟོག་པ་བསྒོམ་དོན་བི་མ་ལ་མི་་་ཏྲས་མཛད་པའོ། །ཞེས་འཁོད། «པེ་»«སྣར་»བསྒྲུབ། «སྣར་འབྱེད། །།«པེ་»«སྣར་མཐར། «པེ་»«སྣར་»དེ་བས། མི«པེ་»«སྣར་»—པ། ཕྱིའི་«པེ་»«སྣར་ཆགས་པར། «པེ་»«སྣར་དང༌། «པེ་»«སྣར་པོ། «པེ་»«སྣར་བསྒོམ་པ། Q«པེ་»«སྣར་»—ལ། «པེ་»«སྣར་»ཡང་། «པེ་»«སྣར་བསྟད། ཨ«པེ་»«སྣར་»བསྲངས་ཏེ། ཉི«པེ་»«སྣར་བསྔམ། «པེ་»བཏང༌། «པེ་བུ། ནོ«པེ་»《སྣར་»གྱིས། @«ཅོ་»ཀྱི། @«པེ་»«སྣར་བཞིན། «པེ་»«སྣར་བསྟད། «སྣར་བྱའི། །དགུ«པེ་»«སྣར་བསྟད་ལ། བསྟད། བཀྱེ«པེ་»«སྣར་»—གྱི། «པེ་»«སྣར་དེ། ®«སྣར་»—ཞེས་བྱ། །«པེ་»«སྣར་—དེ། «པེ་»«སྣར་མཁར། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»བཏགས། «པེ་»«སྣར་»ཉིད། ་་་་་་«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་བསམ། «པེ་»«སྣར་»—ཉིད། «པེ་»«སྣར་»གློད། གྱི«པེ་—ཁོ་ནར་དེ། @«པེ་»«སྣར་གཏོགཚེ«པེ་»«སྣར་»—ལ། །ཀྱི«པེ་»«སྣར་ཀྱིས། ®«པེ་»«སྣར་»དང༌། «པེ་»«སྣར་གཏང༌། «ཅོ་བཏང་། «པེ་»/དང་། བ་《སྣར་བརྟག«པེ་»«སྣར་»པའི། པོ«པེ་»«སྣར་+མ། @«པེ་»«སྣར་མནོ་བ། མན་བག་«པེ་»«སྣར་»རྣམས་བསལ། «པེ་»«སྣར་རྣམས། ན«པེ་»«སྣར་»ཡི་ནི། «པེ་»«སྣར་»«ཅོ་»ལྟ། ཉེ«པེ་»«སྣར་ཉིད། ་«པེ་»«སྣར་»+པ། ཚེ«པེ་»«སྣར་—འཇུག་པར། །།
