རང་གི་དངོས་པོའམ། གཞན་གྱི་དངོས་པོ་ཡང་མེད་ན། ངོ་བོ་ཉིད་སྟོང་པ་་་ཉིད་ཅེས་བྱ་བའི་ཆོས་ཉིད་གཅིག་ཡོད་པར་འགྱུར་རེ་སྐན་ནོ། །ཆོས་ཉིད་མི་་་འགྱུར་བ་བསྟན་པའི་ལེའུ་སྟེ་བདུན་ཅུ་དོན་གཉིས་པའོ། །།།།སྟོང་ཕྲག་བརྒྱ་པའི་རྣམ་པར་བཤད་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །འདི་ནི་ཁ་ཆེའི་པཎྜི་ཏ་དྷརྨ་ཤྲཱིས་མཛད་དོ་ཞེས་གྲག་གོ། པི་ཝེ་ཏིཔ་ལྟ། །ཌ་ཎ་ལ་། ད་དང་ས་སྐྱེ་བྱེད་ཀྱི། ལུ་ཁ། འི་ཉིན་། ད་ལ། བསར་མཆན། བའི་ཚུ་པ། :ཏུ་»མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་བརྒྱད་སྟོང་པའི་རྣམ་བཤད་མངོན་རྟོག་རྒྱན་དང་མཐུན་པར་བཤད་པའི་བཤད་འབུམ་ཁ་ཆེ་ཆོས་གྲགས་ཀྱིས་མཛད་པ་བཞུགས། ཉི«པེ་»«སྣར་གྲུབ། །<ན«པེ་»«སྣར་»གསུང་བ། «པེ་བསྲུང་། 《སྣར་བསྲངས། (། ཨ་ཏ་ལ«@། པ་པ་»}«པེ་»ནི། ®«པེ་»«སྣར་»དོགས། «པེ་»«སྣར་»བལྟ། «པེ་»«སྣར་གྱི། @«པེ་»ཅིང་། །«པེ་»«སྣར་»འགྲེའོ། Q«པེ་སྐྱབསྟན། །::ཀྱི(《་«པེ་»«སྣར་»སློང་། །@«པེ་»«སྣར་»+པ། «པེ་»«སྣར་»བཞག@«སྣར་ཤ། །«པེ་»«སྣར་»འདྲེས། ་ཀཚེ། ཁཎ་ཉིད་ཀྱི་ཀྱི་«པེ་»«སྣར་»བུ། ®«པེ་»གི། ཉི«པེ་»«སྣར་»—པར་རྟོགས། «པེ་»དང་། ་ན«པེ་»སྣར་»སྒོམ། ཉི«པེ་བསྟན། «པེ་»«སྣར་»ཡིས། །ཚེ«པེ་»«སྣར་»གྱིས། པར། ངའི་ད/pཊཀ་བ་དདེ་མ«པེ་»«སྣར་»+དེ་ནི། «པེ་»མེད། «སྣར་མེར། «པེ་»«སྣར་བུའི། «པེ་»«སྣར་བཞག«པེ་»«སྣར་»འགོད། ®«པེ་»«སྣར་»དགོད། «པེ་»«སྣར་»དང་། <<འི༽སྐྱོར་«པེ་«སྣར་བསྐྱེད། འཕྲོད་མ་ཞུ་བ་དང་། མི་ལ་ལས་རང་ལོ་ན་ཚེ«པེ་»«སྣར་»གི། ཉི«པེ་»«སྣར་»བང༌། «པེ་»སྣར་པའི། «པེ་»དང་། «པེ་»«སྣར་»གི། མམ། དེ་དེ་བདམས་«པེ་»«སྣར་+ཐོབ་པར། «པེ་»འགོད། །བརྒྱ་བ། སྐྱེ་བ་《@«པེ་»མོས། @«པེ་»«སྣར་»ནས། Q«པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་ཀྱིས། «པེ་»པར། །།
