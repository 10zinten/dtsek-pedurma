འཕགས་པ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བརྒྱད་སྟོང་པའི་དཀའ་འགྲེལ་་་་སྙིང་པོ་མཆོག་ཅེས་བྱ་བ་རྒྱལ་པོའི་སློབ་དཔོན་མཁས་པ་ཆེན་པོ་རིན་ཆེན་འབྱུང་་་་གནས་ཞི་བའི་ཞལ་སྔ་ནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།པཎྜི་ཏ་:སུ་བྷཱུ་ཏི་ཤཱན་ཏི་དང་། ལོ་ཙཱ་བ་ཤཱཀྱ་བློ་གྲོས་ཀྱིས་བསྒྱུར་་་་བའོ༑། །།།།།།།་ཁ་གསུ། གང་། དྲ་བས༌བཅར་ཏི་་ན་་ས་མུ་ཁ་ས་མ་བྱུང་། །ར་་བཞཏཎ་;ཚེ་ཁོའི་ཁ་ཏིག། ཚོན་བྱེད་དགོས་དང་། བོད་ཟླ་བསྡུར་མཆན། :། དེ་བས། དྲ་ཁ་ག་ལ་ལུ་པཁོདངོའིདང༌། །ར《དཀྱི། །p་་་་་་པ་མཁན《་དངཎ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ཏ། «པེ་»::ང་《སྣར་རོད་ཏ། :ཁ་ཁ། :། ་། མཁས་དབང་དྲུག་དུ་འབྱུང་སྨྲ་བ་པོ་ཚོགས་པ། 《«པེ་+ཏུ། «པེ་+པ། «པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་ཉིད། «པེ་»«སྣར་བ། ཀ《@«པེ་»«སྣར་»སྒྱུར་བ། ཚེ«པེ་»དར། «པེ་»སྣར་རངས། @«པེ་《སྣར་རྟོགས། །དེ་«སྣར་+བ། «པེ་»«སྣར་སྟོན། «པེ་»«སྣར་»ཀྱི། @«པེ་»«སྣར་རྣམ། ད་«པེ་»«སྣར་རིམ་སྟེ། «པེ་»«སྣར་+ན། །@«པེ་»«སྣར་ཡི་གོ་རིམ། ༄༅«པེ་»«སྣར་རིམ། །ཚུ«པེ་»«སྣར་»+ཀྱི། @«པེ་»《སྣར་»འདི། @«སྣར་»འཕྲིན། «སྣར་དེ་བརྒྱའོ། «པེ་»ད་ཁང་། 《སྣར་»—ལ། ཁྱེ«པེ་»«སྣར་»ཅིགཀྱེ«པེ་»«སྣར་སེའུ། །«པེ་»«སྣར་+གིས། འུ«པེ་»:ནས་འབྲེལ། «སྣར་ན་འབྲེལ། །།ཟླ་་པ:ཁ་:ལྔ་བརྒྱུད་《«པེ་»«སྣར་»དེ། /«པེ་»བདག། @«པེ་»«སྣར་+པ། «པེ་»«སྣར་+མི། །@«པེ་»«སྣར་ནི། «པེ་»«སྣར་»ཏེ། །«པེ་»«སྣར་»ཏོ། །«པེ་»«སྣར་»ཅེས། «པེ་《སྣར་གཙོ་བོ། ་ཆེ«པེ་»«སྣར་»དོ། །ལ@«པེ་»«སྣར་གཞུག@«པེ་»«སྣར་ནོ། །«པེ་»སྣར་»འདི། ཚེ«པེ་»«སྣར་»དེས་ན། <@«པེ་»«སྣར་གྲུབ། ཚེ«པེ་»«སྣར་གྲུབ་ཀྱང་། །ཕྱོགས་ཀུན་བྱ་ཀཚུ་ད་«པེ་»སྣར་»གི། །«པེ་»«སྣར་བརྒྱུད་པའི། །«པེ་»«སྣར་པ། «པེ་»«སྣར་པ། »«པེ་»«སྣར་བྱ་བའི། ཚེ«པེ་»«སྣར་»ཏོ༎@«པེ་»«སྣར་པ། །ལ་ལ་བྱ་བྱེད་པའི་རྒྱཚུ་། «པེ་»«སྣར་བ། »«པེ་»«སྣར་ཀྱི། ཀྱི@«པེ་»«སྣར་བརྟེན། འི། «པེ་»«སྣར་—དུ། «པེ་«སྣར་+དུ་རྣམ་པར། «པེ་»«སྣར་ལས། «པེ་»«སྣར་བརྟེན། «པེ་»«སྣར་ནི། «པེ་»«སྣར་ཀྱི། ཁྱེ«སྣར་ཆད། ཀྱེ«པེ་»སྣར་གཞུགས། ::ཀྱང་བྱར་མིང་ཀ་མེད་།
