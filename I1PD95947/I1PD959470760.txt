ས་ཆ་ངོ་ལས་འདི་རྟོགས་པ་དེ་བརྗོད་པའོ། །ཡང་དག་མིན་པ་ནི་ལོག་པ་སྟེ་དེའི་ཡུལ་་་ཅན་ཡིན་པས་ན་ཤེས་པ་ཡང་ཡང་དག་པ་མིན་ཞེས་བྱ་བ་སྟེ། འཁྲུལ་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ཉན་པ་པོའི་རྒྱུད་ལས་འཕགས་པའི་བདེན་པ་བཞི་ལ་དམིགས་པ་ཡང་དག་པ་ཉིད་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་ཏེ་བསྟན་ཏོ། །བྱ། །FRਈਨੇਉਸਨੂੰਵਾਰdਰੀਕਦੋਂਝauxਮਨਨੂੰਮਨཔ་བསྡུར་མཆན། :ཁ་བའི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་༄༅། །ཚད་མ་རྣམ་་་འགྲེལ་རྒྱན་གྱི་འགྲེལ་བཤད་ཤིན་ཏུ་ཡོངས་དག་ཛ་མཱ་རས་མཛད་སྟོད་དུམ། ཞེས་དང་། «སྣར་»ལ༄༅། །ཚད་་མ་རྣམ་འགྲེལ་...མཛད་པའི་སྟོད་དུམ། ཞེས་འཁོད། «པེ་»«སྣར་»བརྟི་ཀཱ་ལངྐཱ། ཨི«པེ་»«སྣར་»ཀ«པེ་»«སྣར་»འགའ། «སྣར་བཙམས། «པེ་»«སྣར་»—ནུས་པ། «པེ་»«སྣར་»བཀུར། «སྣར་ཐར། །«པེ་»«སྣར་བལྟོས། མི«པེ་»«སྣར་»ཉིད་ལ་མཁས་པ། «པེ་»«སྣར་གཟུང་། @«སྣར་»ཐུག་@«པེ་»«སྣར་རིགས། ༄༅«སྣར་»དེའི། «སྣར་»གཞེད། «པེ་»«སྣར་བ། ཚེ«པེ་»«སྣར་»—ཀྱང་། གྱེ«པེ་»«སྣར་»སྟེ། ཉེ«པེ་»«སྣར་བརྗོད། «པེ་»«སྣར་»རྟོག་«པེ་»«སྣར་རྩོལ་བ། @«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»བདག«པེ་»«སྣར་»གི། ཚུ«པེ་»སྣར་རྗོད། པོ«སྣར་»ཀྱིས་ཆེའི། @«པེ་»«སྣར་»+དེ་མི་ཤེས་པ། «སྣར་»པའི་ཕེན། «སྣར་མང་། «པེ་»«སྣར་»རོ། །ཀྱེ«སྣར་»པོ། ཉེ«པེ་»«སྣར་»ཛི། ཨེ«པེ་»«སྣར་»དགོས། «པེ་»«སྣར་»གི། ་«པེ་»«སྣར་རྟོག«པེ་»«སྣར་»ཉིད། «པེ་»«སྣར་པ། «པེ་»«སྣར་»ཕྱིར། @«སྣར་»ཀྱི། ༄༅«པེ་»«སྣར་»འདིའི། ཉི«པེ་»ཀྱི། «པེ་»«སྣར་»བསྒྲུབ། །«སྣར་བྱ་བ། «པེ་»འགྲུབ། ཚེ«སྣར་པ་གྲུབ། ༄༅«པེ་»ཡིས། ཚུ«པེ་»ཀྱང་བརྗོད། «སྣར་བརྗོད། @«སྣར་»དགོངས། ༄༅«སྣར་»+བ། «པེ་»《སྣར་»ཕྱིར། «པེ་»«སྣར་»+ན། «སྣར་བརྟགས། ཚེ«པེ་»«སྣར་»དགོས། «པེ་»«སྣར་བརྗོད། «སྣར་ནས། @«པེ་»«སྣར་»རྟོགས་པའམ། «པེ་»«སྣར་»—དེའི། ༄༅«པེ་»«སྣར་»མོད། «པེ་»པོའི། «པེ་»«སྣར་»ངེས། «པེ་»«སྣར་པོར། ཀྱེ«པེ་»བརྗོད། ཨེ«པེ་»པར་ཅན། «སྣར་པ་ཅན། W«པེ་»«སྣར་»+འདི། «པེ་»«སྣར་»ལ། «པེ་»།
