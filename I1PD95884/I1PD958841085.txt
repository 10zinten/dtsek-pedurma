བསྟན་འགྱུར། །རྒྱུད། ༼བུའིལ་སོགས་པའི་གླུ་དབྱངས་ཇི་ལྟ་བ་བཞིན་དུ་ཤེས་:པར་བྱའོ། ་། ཛཧཱུྃ་བཾ་ཧོརུ་ཊ་མ་:ཊ། །མ་ར་རི་ཌ་བཞིའོ། །ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་བཞིའོ། །།ཨ་ལ་ལ་ལ་ལ་ཧོ@དྲུག་གོ། །སིདྨ་སིདྡྷ་:བུདྡྷ་བུདྡྷ་དན་ཡ་སརྦ་གཉིས་སོ། །ཀ་ཎི་ཀ་ཎི་དྷུ་ནི་དྷུ་ནི་ཧཱུྃ་གཅིག་གོ། །རོལ་པའོ། །ཀ་ཊི་ཧཱུྃ་ཛབཞིའོ། །བྷེ་གྷ་ན་དོ་མ་གཉིས་སོ། །ཊཀྐི་ཧཱུྃ་ཧཱུྃ་བཞིའོ། །ཊཀྐི་ཧཱུཾ་ཛ་ཊིབཞིའོ། །ཧུ་ལུ་ཧུ་ལུ་་་་བཞིའོ། །ཨ་ལ་ལ་ལ་ལ་ཧོདྲུག་གོ། །རུ་ཊ་མ་ཊ་ཉིབཞིའོ། །བརྟན་པར་ཏིང་ཏིང་ཤགས་ཀྱི་རོལ་མོའོ། ་། ཊ་གྷི་གྷི་གཉིས་སོ། ·། ཊ་གྷི་གྷི་གྷི་གྷི་བཞིའོ། །ཊ་གྷི་གྷི་གཉིས་སོ། །ཧཱུྃ་གཅིག་གོ། །ཊ་གྷོ་གྷོིགཉིས་སོ། གྷི་ཊ་གྷོ་གཉིས་སོ། །གཡོ་བའི་རོལ་མོའོ། །::ཨོཾ་ཨོཾ་ཨོཾ་གསུམ་མོ། ་། ཋཐཾ་ཐཾ་ཁཾ་གསུམ་མོ། །དྷཾ་དྷཾ་དྷཾ། གསུམ་མོ། །དམ་ཚིག་གསུམ་གྱི་རོལ་མོའོ། །བི་ཀཾ་ཀཾ་ཀཾ་ཀཾ་ཀཾ་ཀཾ་ལྕེ་དྲུག་གོ། །།བི་:བ་ཛ་བི་བ་ཛ་དྲུག་གོ། །བི་བ་ཛ་བ་ཛ་ལྔའོ། །བི་ཀོ་ཀོ་ཀོ་ཀོ་ཀོ་དྲུག་གོ། །བི་ཧཱུྃ་བྷོ་གསུམ་མོ། །ཧཱུྃ་བྷྱོ་ཧཱུྃ་བྷོ་བཞིའོ། །བི་རུ་ལུ་རུ་ལུ་ཧཱུྃ་བྷོ་དྲུག་གོ། །བི་ཀོ་ཀོ་ཀོ་ཀོ་ཀོ་དྲུག་གོ། །དཔལ་ཧེ་རུ་ཀ་རྡོ་རྗེའི་རོལ་མོའོ། །པདྨ་པདྨ་ཧཱུྃ་ལྔའོ། །པདྨ་:སྟི་པདྨ་སྟི་ཨེལྔའོ། །པདྨ་:པདྨ་དྷཾ་ཨེལྔའོ། །པདྨ་:པདྨེ་ཧྲི་ལྔའོ། །པདྨ་པདྨ་ཤྲཱི་སེལྔའོ། །དཔལ་ལྡན་པདྨའི་ཏིང་ཏིང་ཤགས་ཀྱི་རོལ་མོའོ། །།:'ཏཱཾ་ཏཱཾ་ཏཱཾ། ཏཱཾ་བཞིའོ། །།བབབབབཞིའོ། ་ཊཊཊཊབཞིའོ། །ཏྲི་ཏྲ་ཀྲོ་ཏྲ་ཊ་ལྔའོ༑། རྒྱལ་པོའི་རོལ་མོའོ། །ཊཀྐི་ཧཱུཾ་ཛའགྱིང་བས་:འགྲོ་བའོ༄༅། །ཏྲཿཏྲཿཏྲཿཏྲབཞིའོ༑། མགྱོགས་པས་རབ་ཏུ་མགྱོགས་པའོ། །ཏ་:ཌ་བ་ཏཌ་བ་སྨིཾ་ཏ་བདུན་ནོ། །ཏ་:ཌ་བ་ཏ་ཌ་བ་ཏི་ཌ་བ་ཏ་ཌ་ེབ་བཅུ་གཉིས་སོ། །ང་། ཤིན་ཏུ་མགྱོགས་པའི་རོལ་མོའི་ཕྱག་རྒྱའོ་ཞེས་པ་ལ་སོགས་པའི་རོལ་མོ་རྣམས་དང་པོར་ཏེ། :མུ་ར་ཛ་ལ་སོགས་པ་ཚད་ཇི་ལྟ་བ་བཞིན་དུ་དཀྲོལ་བར་བྱའོ། །དེའི་རྗེས་ལ་།
