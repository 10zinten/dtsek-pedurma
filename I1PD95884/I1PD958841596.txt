《སྣར་ཡི་རང་། «པེ་»ཟླུམས། 《སྣར་ཟླུམ། །།ཏ་《འི་མ། བ(«སྣར་»གཟུངས་། ི«པེ་»«སྣར་པོས། «སྣར་»—སྐུ། «པེ་»«སྣར་»ལས། «པེ་»ཏི་འི་ཝ་སྐྱབས་་་་་་བཏབ་པ། མ་བུ་དག། ཀུན་མའི་འགྲོས་དང་(«པེ་»«སྣར་གྱི། («པེ་པ། «པེ་《སྣར་+ཏུ། ®«སྣར་ནང་། «པེ་»«སྣར་»སློབ། :བ། །་ང་《སྣར་—ཆང༌། ་་་་་་ཡེ་ཤེས་རྒྱན་ཀྱི«སྣར་»བ། /Q《སྣར་+གནས་པའི། 《སྣར་དག«པེ་»«སྣར་+སྤངས་ཏེ། མ་བྱིན་པར་ལེན་པ་དང་། འདོད་པས་ལོག་པར་གཡེམ་པ་དང༌། བརྫུན་དུ་སྨྲ་བ་དང་། འབྲུའི་ཆང་དང་། :་»བཅོས་པའི་ཆང་མྱོས་པར་འགྱུར་བ་བག་མེད་པའི་གནས། ཀྱི། «པེ་བསླབ། (པོ«པེ་བདག«པེ་»བརླབ། «པེ་»«སྣར་བྱིན། སྣར་ལྟུང༌། «པེ་《སྣར་ཕྲེང་། 《སྣར་དང༌། «སྣར་»གི། །་་«པེ་གི། /Q«སྣར་»བླང་། སྤྱི་ཚ«པེ་+ཅན། «པེ་»«སྣར་»ཀྱི། །::ཀྱི་འ«སྣར་—མ་ལུས། @«པེ་གསུང་། །《སྣར་མིའི་འོ། @《སྣར་+བ། (《སྣར་བསླབས། :ང་། 《སྣར་»+སྡོམ་བའི། (ཉེ«པེ་»«སྣར་པ། (ཉེ«པེ་»གནས་པ། 《སྣར་»བཅས་པར། ད་དྲག་«སྣར་+མ། ་«པེ་»གཞུགས། «པེ་+པ། །®«སྣར་དགའི། །«པེ་»+ཞེས་པ། པོ«པེ་»«སྣར་སྤངས་པའི། །ཁ«པེ་《སྣར་»གྱི། ་«སྣར་བསོད། །«པེ་»«སྣར་»ཀྱི། ཀ་ར། འཁོ«པེ་»«སྣར་»ཞིང༌། ་འཁྱེ«སྣར་པར། འབུ«པེ་»«སྣར་»+པ། (སྟེ«པེ་+དང་། 《སྣར་+བ་་ཀྱི་བྱ་བ་དང་། །(«པེ་»«སྣར་»གི། །༼(《སྣར་»ཀྱིས། ཨེ«པེ་»འདུལ་བར་བསྟན་པར། 《སྣར་འདུལ་བ་་::ཀྱིལྟར་བསྟན་པར། །བསྐྱེ«པེ་»«སྣར་»བོ། ་། ཁ་བསུར་དུ་འི། འི་(《སྣར་»+པ། «སྣར་བསལ་བར་བསྒོམས། «པེ་བསྟན། ། ི་ཚ«པེ་»མཆོག《སྣར་གྱིས། ལ་ཆཟླ«པེ་སྒྲགས། ་ཀྱང་། །《སྣར་»ཀྱི། «པེ་བསམ། «སྣར་ཡིས། ®«པེ་»«སྣར་»སྒྲོལ། («པེ་»«སྣར་བརྟན། རྒྱུ་ར། «པེ་ཞུགས་སུ། @«སྣར་»གྱིས། ཉི«སྣར་བྱུང་། ::@«པེ་»ཡི། །ས(«པེ་+དེ་ནས། :ལ་དུ《(«པེ་»«སྣར་མཆི། (《སྣར་+བ། །ཀའི། ༼ཁྲི''ལོ། །(«པེ་»«སྣར་པར། ཉེ«པེ་»བྱ། ལ«པེ་»གཟུང༌། པེ་ཚ«པེ་»གི། 《ས«སྣར་བསྡུད། །ཁོར་་«པེ་»བཞིན། །དདཔལབ་་་་་་པ(«པེ་—མཆོད་པའི་ལས་ཀྱང་ཅི་ནུས་བགྱི། །«པེ་»གྱིས། «པེ་»«སྣར་»བསྒྲལ། ཚེ«པེ་»བ་འགོད། <«པེ་»«སྣར་རྣམས། «པེ་»«སྣར་»ཤཱཏྠ་ལ་སོགས་པའི། །དུས་ཀྱི་ཀྱི༌(ི«པེ་»«སྣར་ཕྱི། «པེ་»«སྣར་»དྲིའི། །@«པེ་»ཤུདྷ་སརྦ་དྷརྨསརྦ་པཱ་པཾ་ནི་ཙཱ། སྣར་»ཤུདྡྷ་་ཚེསརྦ་དྷརྨསབྦ་པཱ་པཾ་ནི་ཙཱ། ?«པེ་»སརྦ་བི་ཀལྤན་ཨ་པ་ན། «སྣར་»སབྦ་བི་ཀཝཱན་ཨ་པ་ནཾ། («པེ་»《སྣར་»ཏཱིཀྵ། @«པེ་—བརྗོད། ཚེ《སྣར་བསྐལ། «པེ་»གསུམ་ལྟ། 《སྣར་»སུམ་བལྟ། «པེ་དཔུངས། འཚེ«པེ་»བུདྷ། ::ཀྱི«སྣར་སརྦ། །(། «པེ་བཅིང་། སྟེ《སྣར་»བྱས། ོ«སྣར་»།
