ཆུ་ཆེན་གྱིས་མི་རྗེ་དང་གྲགནི་དོམ་བཀོད་ལ། །དག་བྱེད་ས་དང་། ཁའ་གང་བར་བལྟ། །དེ་བསྡུར་མཆན། སྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སྙིང་རྗེ་ཁྲོས་པའི་སྒྲུབ་ཐབས་རིམ་པ་བཞུགས་སོ། །«པེ་»«སྣར་»ལཱ་ཧ། «པེ་»«སྣར་»+སོ། «པེ་»«སྣར་རྗེས་ཁྲོས་པས། ཚུ«པེ་»«སྣར་བའི། @«པེ་»«སྣར་»པས། @«པེ་»«སྣར་»སྟབས། །«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»མཁྲིས། «པེ་»«སྣར་»པའོ། །«སྣར་བྱའོ། «པེ་»ཁྲིམ«སྣར་»དྷརྨ། «པེ་»«སྣར་»ལཱ་ཧ། ༄༅«པེ་»«སྣར་»+པཎྜི་ཏ་ཆེན་པོའི་ཞལ་སྔ་ནས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།་་་། །མངོན་དུ་བྱེད་པས་སྐྱོ་བ་བྱུང་ཞིང་དལ་འབད་པས། ལ་ལ་ད་། pན་བདག་ཉིད་ཆེན་པོ་པ། །་་༌འོ་ཎིའུ་་བསྡུར་འབྲས་རེའུ་མིག་་་་་་་་། །ན་བ་མ་འ། གཏཀའོ་ལ་རྒྱུད། （དང་འབད་དཔེ་གཞི། སྡེ་དགེ། སྣར་ཐང་། སྡེ་ཚན། ས་རྒྱུད། དྲངཔཨ་རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། ན། ན་བ་ན་ནབ?བ_ན། །ཕྲེང་། །།ཚེག་ཁྱིམ། ར། །པེ་ཅིན། རྒྱུད། ཚོ་བ་ཞིག་ལ་རྟག་ཏུ་ཀའི་ཕྱག་ལ་ཁྲི་བ་འཚོཎྜ་ཀ་ཀོ་ཀ་བ་ཚོང་བ་ཞིག་དྲན་དང་ཏིལ་ན་རལ་པ། དུ་བ་ཀྱང་ཤ་ཀཁགག་ཚོར་བྲན་པ་ཁ་་བ་།
