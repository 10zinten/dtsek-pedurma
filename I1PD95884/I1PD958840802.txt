ཀབར་ལྟེ་བ་ནས་ཁར་ཁ་ནས་ཕྱིར་ཡང་ལྟེ་བར་ཞེས་བྱ་བ་ལ་སོགས་པས་འཁོར་ལོ་་་འཁོར་བའི་རིམ་པས་སྔགས་བགྲང་བར་བྱ་བའོ། །དེ་ལ་སྔགས་ནི་འདི་ཡིན་ཏེ། ཨོ་མ་རཱི་ཙི་ཚྭ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཟས་ལ་སོགས་པ་རྣམ་པ་ཐམས་ཅད་ལ་་་ཡང་རང་གི་ས་བོན་གྱིས་:བྱིན་གྱིས་བརླབས་པ་རྗེས་སུ་བསྒྲུབ་པར་བྱ་བའོ། །སྐྱོ་ན་ཡི་གེ་བརྒྱ་པའི་སྔགས་ཉེ་བར་བརྗོད་ལ་ཇི་ལྟར་བདེ་བར་རྒྱུ་བར་བྱ་བ་ཡིན་ནོ། །རྟོག་པ་ལས་གསུངས་པའི་འོད་ཟེར་ཅན་དཀར་མོའི་སྒྲུབ་ཐབས་སོ། །་མ། བྱང་ཆུབ་སེམས་དཔའ་དཱ་ན་ཤྲཱིའི་ཞལ་སྔ་ནས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འོད་ཟེར་་་ཅན་གྱི་སྒྲུབ་ཐབས་བསྐོར་བཞུགས་སོ། །«སྣར་ནཾ། «སྣར་པཾ། ®《སྣར་»མཾ། (འདིའི་འོག་གི་པ་མཾ་ཞེས་པ་རྣམས་«པེ་»«སྣར་ལ་ཟླ་ཚེས་ཡོད་མེད་ཅི་རིགས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། །«པེ་»«སྣར་གཙོ་བོའི། པོ«པེ་»«སྣར་—ཞལ། ་«པེ་»«སྣར་པ་དང་། །«པེ་»«སྣར་»མ། «པེ་»«སྣར་»+ཀ«པེ་»«སྣར་»ཤིང་། «པེ་»«སྣར་»ན། ་བ«པེ་»«སྣར་»དམར་མོ། ་«པེ་»«སྣར་བསྡིགས། («པེ་»«སྣར་བརྒྱན། «པེ་»«སྣར་»པའི༑@«པེ་»«སྣར་+དང༌། ཚེ«པེ་»སྣར་»གི། ཀྱེ«པེ་»གནོད། «པེ་»«སྣར་»+དུ། «ཅོ་»མཾ། «པེ་»བརླབ། «སྣར་བྱིན་གྱིས་བརླབ། «པེ་»«སྣར་»+ཡང་། འབད། ཀ་པས་བཞུགས་པ། ཤིང་མྱ་ངན་མེད་སྔོན་དེ་བ། ཕྱུག་བདག་བསྡུས་པ། བྱ་བ་བ་འདུས་པ་ལ་ལ་བ་པད་མ་་པ་:་་་་་བརྒྱ་ཕྱིང་། །ཤིང་ལས་བོད་ལ་བབ་པ་ཡིད་དུ་འོང་བའི་བདག་ཉིད་བསྒོམས་ལ། ག་གའི་ས་བོན་གྱི་འོད་ཟེར་།
