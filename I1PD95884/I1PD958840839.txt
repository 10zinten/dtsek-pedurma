འཁྲི་ཆ་ལ་ཌ་། བློ་ཁོང་::་དང་:ཟླ་་ཀུན་། །ཟླ་བ་རཁ་ཡ། ད༌ཀྱང་། /ན། ་འབའ་་་་་་་། :་༌་བསྡུར་མཆན། ཁ་ཊ་ཀ་ལ་་འ་པརརཚུ་༽ར་འབགས་བག་ཚུན་ཚུལ་དང་ཡད་ཀས་བྱི་ས་གནས། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། བུ«པེ་»«སྣར་»གི། «སྣར་»དྲུང་། ཀྱིས་ཀྱི་མིང་ཡིན། ཁོང་་«པེ་ར་ཎི། «སྣར་རི་ནི། བ«སྣར་»བ་སུ་ཤྲཱི་ཡེ། ན«པེ་+ཨོ་བ་སུ་ཤྲཱི་ཡི་སྭཱ་ཧཱ། «སྣར་བ་ཏི་ཤྲཱི། ས«པེ་»«སྣར་གྲུ། «པེ་»«སྣར་—མེ་ཏོག་དྲི་ཞིམ་པ་བྱས་ལ། 《སྣར་བྱའོ། །རོ«པེ་»«སྣར་། །པ་དེ། བདག་བ་དང་། རྒྱུ་དང་བྱ་ཁྲི། རོང་བ་དང་། །ཁྲོ་ག་ཏཡ་ཚུ་་༌་་ཚུ་ན་ཀ་མད་ཀ་། དག་ཏུ། དེpན་ན། །ཀ་པ་ག་བཀ་ཁཡ་། ད་རོབ་ཀ། རཀྵ་ས་བུ། བསྡུར་འབྲས་རེའུ་མིག་ན་ཆཀྱི་ང་ཚོ། ਕਲਾਤੋਂਸ਼ੁਰੂਕਰརྒྱུད། (སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། རྒྱུད། རྒྱུད་འགྲེལ། དཔེ་གཞི། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། །ན་བ་་པ་བན་། ནབ་བན་ཕྲེང་། །ཚེག་ཁྱིམ། །།::ཀྱི་་་་་་པ་དྲུག་ལ། །པས་ལན་པ་ཡིན་པ། ད་པས་འབྲས་ཀྱི་སྐྱེ་མའི་དྲུངས་པ་མཆོད་བསྙམས་པ། མི་བསྐྱོད་པ། པའོ། མདུན་ན་ནི་བཅོམ་ལྡན་འདས་མ་དཔལ་གྱི་ནོར་རོ། །བ། ཡུས་དེར་སྤྱི་དཔལ་ལོ། །རྒྱབ་ན་ནི་ནོར་དཔལ་གྱི་ཞལ་ལོ། །གཡོན་ན་དང་ལ་ལ་གྲོས་མའོ། །དེ་ད་ལོག་ཀྱང་ཡང་ས་བོན་བ་ནང་གི་དང་པོའི་ཡི་གེ་ལས་།
