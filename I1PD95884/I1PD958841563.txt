བསྟན་འགྱུར། །རྒྱུད། འབུད་::ང་འོ་བྱེད། ::《སྣར་»པ་རྣམས་ཀྱི་ཕྱག་རྒྱའི་སྔགས། ་ཎ་《སྣར་བྱིས། །ཏཁན་བཞི་«པེ་བཅའ། འི«པེ་»+དང༌། ་ནོ«པེ་»ན། «པེ་བཟུང་། (འདིའི་འོག་གི་གཟུང་ཞེས་པ་་ཊ་ཏན་རྣམས་«པེ་»ལ་བཟུང་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་དུ། ་«པེ་པོ་ཆེའོ། བློ་ཁ::«པེ་»རྗེའི། །ར《སྣར་པའི། 《པེ་»+ཀྱི། །་《པེ་བཟུང་བ་ནི་བཟང་བསྐྱོད། 《སྣར་གཟུང་::ཀ་མ་བ་ནི་བཟང་སྐྱོང་། «པེ་རླབས་ལྟར། །«པེ་»མཆོད། ཁ་《སྣར་དང༌། །བཏ«པེ་པ། པp::«པེ་ཀྱིའོ། ::《པེ་པད་མ། «པེ་»འཛུབ་སྡིག་པ་རབ་ཏུ་བསྐྱེད་པའི་ཚུལ་བྱ་བ་ནི་ངན་སོང་མེད་་པའིའོ༑། ལག་པ་གཉིས་ཡང་དག་པར་སྦྱར་བ་ནས་རབ་ཏུ་བསྣམ་པའི་ཚུལ་བྱ་བ་ནི་འབར་བའི་འོད་ཀྱིའོ། །།ཉི་ཟླ་ད་ལག་པ་གཡས་པ་མཆོད། «སྣར་བུམ་པའི་ཚུལ་བྱས་ཏེ་གཡོན་པའི་རྡོ་རྗེ་ཁུ་ཚུར་བྱ་བ་ནི་དྲ་བ་ཅན་གྱི་འོད་ཀྱིའོ། །ཀཚོལག་པ་གཡས་མཆོག་«པེ་»རི་བོང་གིས་མཚན་པའི་ཚུལ། «པེ་+བའི། ོ《པེ་ཁེངས་པའི་་་་ཟླ་ཊ་སྦྱོར་བས་དཀུར་གཞག《སྣར་བསྙེམས་པའི་ཚུལ་གྱིས་བཀུར་བཞག ི་《པེ་+ལ། །པཎ་ཊིས་«པེ་ལ། 《སྣར་གྱིས། «པེ་ཨཱ་མིༀ་མུༀ་གན་དྷ་ཧ་སྟི་ནེ་ཧཱུྃཨོཾ་ཛྙཱ་ན་ཀེ་ཏུ་ཛྙཱ་ན་པ་ཏི་པ་ཏ་ཧཱུྃ་ཨོཾ་བྷ་དྲ་པ་ཏི་པྲ་དྲ་པཱ་ལེ་སྭཱ་ཧཱ། ཨོཾ་སཱ་ག་རེ་མ་ཧཱ་སཱ་ག་རེ་ཧཱུྃ། ཨོཾ་ཨཀྵ་ཡེ་ཧཱུཾ་ཨཀཱ་ཡཀྵ་ར་མ་ཨ་བ་ར་་། ་ཎ་པ་རི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཨོཾ་པྲ་ཏི་བྷ་ནེ་པྲ་ཏི་བྷཱ་ན་ཀྲུ་ཌེ་ཧཱུྃ། ཨོཾ་སྦྲ་མེ་མ་ཧཱ་སྦྲ་མ་ར་ཏི་སཱ་ནི། ཨོཾ་སརྦ་ཨ་བ་་་བྱ་བའི་ཡང་ཛ་ཧི་སརྦ་ཨ་བི་ཤོ་དྷ་ནི་ཧཱུྃ། ཨོཾ་སརྦ་ཤོ་ཀད་མོ་ནིརགྷ་ཏ་མ་ཏི་ཧཱུྃཨོཾ་ཛ་ལི་ནེ་མ་ཧཱ་ཛཱ་ལེ་ནི་ཧཱུཾༀ་ཙན་ཏྲེ་ཙན་དྲ་བྱ་བ་ལ་གི་ནེ་ཏེ་སྭཱ་ཧཱ། ཨོཾ་ཨ་མི་ཏ་བྲ་བྷེ་ཨ་མི་ཏ་མ་ཧི་ཧཱུྃཨོཾ་ག་ག་ནི་ག་ག་ནི་པ་ར་ལོ་་ཙ་ནི་ཧཱུཾ། ཨོཾ་སརྦ་ནི་བ་ར་ཎ་བི་སྐམ་བྷི་ནེ་ཧཱུྃ། 《སྣར་ཨོ་མིༀ་མུༀ་གནྡྷ་ཧ་སྟི་ནེ་ཧཱུྃ། ཨོཾ་ཛྙཱ་ན་ཀེ་ཏུ་ཛྙཱ་ན་བ་ཏི་ཧཱུྃ། ཨོ་བྷ་དྲ་བ་ཏི་བྷ་དྲ་པཱ་ལེ་ཧཱུྃ། ཨོཾ་སཱ་ག་རེ་མ་ཧཱ་སཱ་ག་རེ་ཧཱུྃ། ཨོ་ཀླུ་ཡེ་ཧཱུཾ་ཨཀྵ་ཡ་ཀརྨ་ཨཱ་བ་ཎ་པ་རི་ཤོ་དྷ་ནེ་སྭཱ་ཧཱ། ཨོཾ་པྲ་ཏི་བྷཱ་ནེ་པྲ་ཏི་བྷ་ན་ཀུ་ཊེ་ཧཱུྃ། ཨོ་སྦུ་མེ་མ་ཧཱ་སྦྲ་མ་ར་ཏི་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཨ་པཱ་ཡ་། ཀ་ང་ཛ་ཧེ་སརྦ་ཨ་པཱི་ཡ་བི་ཤོ་དྷ་ནེ་ཧཱུྃ། ཨོཾ་སརྦ་ཤོ་ཀ་ཏ་མོ་ནི་རྒ་ན་མ་ཏི་ཧཱུྃ། :ༀ་ཛཱ་ལི་ནི་མ་ཧཱ་ཛཱ་ལ་ནེ་ཧཱུྃ། ཨོ་ལྕེ་ཙནྡྲ་ཙནྡྲ་བྱ་བ་ལོ་ཀི་ནི་ཏེ་སྭཱ་ཧཱ། ཨོཾ་ཨ་མི་ཏ་པྲ་བྷེ་ཨ་མི་ཏ་མ་ཏི་ཧཱུྃ། ཨོཾ་ག་ག་ནེ་ག་ག་ན་བ་ར་ལོ་ཙ་ནེ་ཧཱུྃ། །ཨོཾ་སརྦ་ནི་བ་ར་ཎ་བི་སྐམ་བྷི་ནི་ཧཱུཾ། «པེ་བཅིངས་པར། «པེ་»ཀྱི། «པེ་»མོ། «པེ་»བསྡམས༑«པེ་ཨོཾ་བཛྲ་ཨང་གུ་ཤ་ཧཱུཾ་ཛཱ། ཨོཾ་བཛྲ་བ་ཤ་ཧཱུྃ་ཧཱུྃ༔ཨོཾ་བཛྲ་སྭོ་ཊ་ཧཱུྃ་བཾཨོཾ་བཛྲ་ཨ་བེ་::ཀྱིངོཤ་ཧཱུཾ་ཧོ། 《སྣར་ཨོཾ་བཛྲ་ཨངྐུ་ཤ་ཛཨོ་.....ཨོཾ་བཛྲ་ཨ། 《པེ་»+རྣམས་ཀྱི། «པེ་མོ། །བ། «པེ་དབུས་མར་འཁྱིལ་པའི། «པེ་»«སྣར་བསྐུལ། «པེ་པ། །:«པེ་པས། :ཀ«པེ་ཚོརྫོ@འོད་ཀྱིའོ། ་«པེ་《སྣར་བ། ་་་་་བ་འདྲེན་པ་དང་འཁོད་ཀྱང་། «པེ་»རང་། ::«པེ་《སྣར་སོར། །«པེ་སྒྲོག་མའོ། 《སྣར་»སྒོགས་མའི་འོ། འཊ«པེ་»སཱ། a/?(འདིའི་འོག་གི་སཱ་ཞེས་པ་རྣམས་《པེ་ལ་སཱ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»རད་་ལ་པའི་ན་ཡུལ་ཀེ་རཏ་ན་ཛཱ། ::«པེ་བྷི་ཀུ་ཊི་བྷི། གང«པེ་»ཤིང་ཁ། «སྣར་ཤིཌ། །།ོར་«པེ་»སྭཱ་ཧཱ། ཀྱི་«པེ་+ལ། «པེ་»«སྣར་ལ། 《པེ་ཤུགས་ལ། :«པེ་»བའི་འདྲེན་པའི་ཚུལ་བྱས་ཏེ། རལ་ཚR། གྲི་འདྲེན་པའི་ཚུལ་དུ་བཟུང་བ་ནི། ཏེ། འདྲེན་པ་བྱེད་པ་མོ། 《སྣར་པས་རལ་གྲི་....གཟུང་བ་ནི་མི། །
