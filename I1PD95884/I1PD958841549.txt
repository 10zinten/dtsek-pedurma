བསྟན་འགྱུར། ་རྒྱུད། བུAਨਾਪਾਸਕੋਲਰਸਚਸਲਵਰਲਬਨੂੰਨਸ਼ਨ། བདག་དང་ཁ་ཞེ་ག་ང་ལ་བྱ་ནག་འུ། །་་། ཀ་བ་ཚོ་ཕྱོགས་བྱང་ཕྱོགས་པའི་ནང་། །འབག་ཏ་བསྡུར་མཆན། ང་པར་བྱའོ་པ་ཁ་ཏུ་ཀ་ཡ་མང་པལོ་STན་་ག་བ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སློབ་དཔོན་་གྱི་བྱ་བ་ཀུན་ལས་བཏུས་པ་ཨ་བ་དྷུ་ཏི་དཔལ་ལྡན་འགྲོ་བའི་མེ་ལོང་གིས་མཛད་པ་བཞུགས་སོ། །«སྣར་»རྡོ་རྗེ་སློབ་དཔོན་གྱི་བྱ་བ་ཀུན་ལས་བཏུས་པ་བཞུགས་སོ། །།།«པེ་ཨ་ཙརྒྱ་ཀྲི། «སྣར་ཙཱརྱ་ཀྲི། «པེ་+ས་མུ་ཙྪ་ཡ། «པེ་»«སྣར་+དཔལ། .«པེ་»ཁ་ཡི། པོ«པེ་+རྣམ་པར། «པེ་»ནི་དེ་ནས་གྲངས་མེད་པའི་གནས་སུ་ཞབས་ནི་འཇོག་པ། «པེ་བྲིས། <«པེ་»གཡོ་བའི་དཔུང་བའི་ཚོགས་ལྡན། «སྣར་དཔུང་བའི་བཀོད་པ་གཡོ་བའི། ལ«པེ་»གཡོ། «པེ་»བཅོམ། «པེ་ལྟར། «སྣར་»ལྟ། ནསཉེ«པེ་ནི། པད་མ་བདུན། བལ་དུ་གནས་པ་དང་པ་བ། །T:པ་དེ་དང་འདྲ་ན། བས་བར་། འཡརབདདྷདགཝདགཔལ་(«པེ་+ཀུན་གྱིས། «པེ་སྒྲུབ། «པེ་»རྡོ་རྗེ། «པེ་»རྗེའི། །«པེ་»ཕྱེ་པ་ཉིད། «པེ་+པ། «པེ་»«སྣར་མྱོང་། «པེ་བསྟན། «པེ་»སློབ་དཔོན་རྣམ་གསུམ། ::(«པེ་བདེ་མཆོག་རྒྱ་་་མཚོར་ཇི་བཞིན་གསུངས། །་(ཀྱེ«པེ་»སློང་། (《སྣར་པ། ་ཡས་ཀྱ་ང་ཚོ། ་.ལ་(«པེ་པ། 《«པེ་»ལ། «པེ་ལས་མཆོད་པར། ®«པེ་»+ནས། སུ«སྣར་»—ལྡན། «པེ་སྦྱིན་ཞིང་བདེ་བ་འོང་བ། ༄༅«པེ་»«སྣར་སྨྲ་བ་པོ། «པེ་»གྲུབ། །:«པེ་»བར། «པེ་པ། «པེ་སོ། །«པེ་པར། @«པེ་《སྣར་ན། @«པེ་»«སྣར་བ། ཉི«པེ་»གྱིས། «པེ་པ། ་འཁྱེ«པེ་+ལ། («པེ་»གི་ཕྱིར། «སྣར་»གི། «པེ་དང༌། «པེ་»«སྣར་»བཟུང་། «པེ་»གིས། ལོ«པེ་»བཅངས། (ི«པེ་དང་། འབྱེ་«པེ་»བསྒྲུབ། («པེ་པར། །(«པེ་»+ལས། «སྣར་»འཚོལ། «པེ་»བ་ཉིད་ཀྱིས། «སྣར་»དང་། «པེ་»ཚེ། «སྣར་»—རྣལ་འབྱོར་ཆེན་པོའི། @«པེ་»«སྣར་ཟད། «སྣར་བསྐམས་འགྱུར། «པེ་»གཡེངས་པའི༑དགྱེ«པེ་»གཡེངས། ཀྱེ«པེ་»+དམ། སྟེ«པེ་»—པ། 《སྣར་»བ། ་«པེ་»འདི། །«པེ་བསྒྱུར། །«པེ་ལ། འི«པེ་+ལ། «སྣར་ལྟར། །པ་®«པེ་»རིགས། །«སྣར་»—སྤྱོད་པ་དང་། «པེ་»པར་བྱས་པ། «སྣར་པ་བྱས། «པེ་+མ། «པེ་»ཡིན། «པེ་»གིས། (བྱེ«པེ་པ། («པེ་»སྐེམ། «པེ་»ལོ་བསྒྱུར། «སྣར་ལོ་སྒྱུར། «པེ་»ཀྱིས་ནི། སྐྱེ«པེ་»ཀྱིས། «པེ་»ཀྱིས་གསུངས་སྙན་ལས། «པེ་དང་། ཤེ«སྣར་»+གི། «སྣར་»གྱིས། «པེ་པ། «པེ་»བརྒྱན། «པེ་པ། ་འ«པེ་»ཀུ་ལེའི། «པེ་»རྒྱ། །ན«སྣར་»ཀྱི། ་«པེ་»«སྣར་»།
