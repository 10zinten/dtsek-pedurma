འདི་ཡིས་ལེགས་པར་བསྐུལ་བྱས་ལ། །ཨོ་ཨཱ་ཡིག་སྒོར་ནི་ཆོས་རྣམས་ཀུན། །གདོད་ནས་:མ་སྐྱེས་མེད་ཕྱིར་རོ། །མིུཞེས་བྱ་བའོ། །བསམ་གཏན་ལེགས་བླངས་རྣལ་འབྱོར་པས། །རང་གི་ནུས་པའི་སྦྱིན་པ་གཏང་ཚེ། །གཟུགས་སོགས་འདོད་པ་སྤྱོད་པ་ན། །མ་ལུས་རྟོག་པ་གཏན་ནས་སྤངས། །བླ་མེད་བྱང་ཆུབ་རིང་མིན་པར། །སྒྲུབ་པ་པོ་ནི་དམ་པས་ཐོབ། །སྒྲུབ་ཐབས་བྲིས་ལས་བདག་གིས་ནི། །དགེ་བ་བླ་མེད་ཐོབ་པ་གང་། །དེས་ནི་འཇིག་རྟེན་མ་ལུས་པ། །མྱུར་བར་སྙིང་རྗེའི་ཁྲོ་ཐོབ་ཤོག། ཧ་ལཱ་ཧ་ལའི་སྒྲུབ་ཐབས་དགེ་་་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།།།རྒྱ་བོད་བསྡུར་མཆན། པ་འདྲ་བ་ཀུན་གྱི་གདམསཀྱཀུན་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ལཱ་ཧ། «པེ་»མེ«སྣར་»རྗེ། «པེ་»སྣར་པ། གྱི«པེ་»«སྣར་»། @«པེ་»«སྣར་»གིས། «པེ་»«སྣར་»སྙིང་། «པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»—། «པེ་»«སྣར་»ལྟས། «པེ་»«སྣར་བའི། «པེ་»«སྣར་»བཞིར། «པེ་»«སྣར་ཚེ། «པེ་»བསྣབས། «སྣར་བསྣམས། «པེ་»«སྣར་པས། འགྱེ«པེ་»«སྣར་»དང་། («པེ་»«སྣར་»གི། @«པེ་《སྣར་རྟོགས། «པེ་»«སྣར་རྣམས། །«པེ་»«སྣར་»འགྲོ། ་«པེ་»«སྣར་»མགྲིན་་་སྙིང་། མུ«པེ་»«སྣར་»བ། @«པེ་»«སྣར་བཞག@«པེ་»«སྣར་ཧཱ། «པེ་»«སྣར་ནང་། «པེ་»«སྣར་བརྐྱང༌། (གེ«པེ་»«སྣར་པར། ་འཁྱེ«པེ་བྲལ་བར་མཛོད། «སྣར་བྲལ་བར་མཛད། ང(«པེ་《སྣར་བཤགས། །ལོག་«པེ་»«སྣར་སྐུལ། ཉེ«པེ་»«སྣར་»སྐྱེ་བ། «པེ་»«སྣར་»—། «པེ་»«སྣར་»བཏང༌། «སྣར་ལ། ་པོ«པེ་»«སྣར་རྗེས། རྣམས་ཟུམས་ཅད་ཀྱི་སྐུ་དང་གསུམ་དང་བྱ་བ་བྱེད་དང་བཞེད་པད་པ། ཉེད་ནི་ང་ཡིན་ནོ་ཞེས་སྔབ། ད་བཟོད་པར་བྱ་བའོ། །དེའི་དུ་བ་ལ་—།
