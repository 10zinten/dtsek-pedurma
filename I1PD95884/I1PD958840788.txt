སྨན་རྣམས་སྲང་རེ་ཙམ་དུ་བྱས་པས་ནི། །དལ་གྱིས་འཇམ་པོར་མིང་ནི་བློ་གྲོས་བཟང་། །ཟླ་བ་ཙམ་དུ་རབ་ཏུ་སྦྱར་བྱས་པས། །ངག་གི་དབང་ནི་བླ་ན་མེད་པ་འཐོབ། །ཟླ་བ་དྲུག་ཏུ་ཉེ་བར་སྦྱར་བྱས་པས། །དངོས་སུ་ངག་གི་དབང་་་་་ཕྱུག་ཉིད་དུ་འགྱུར། །མཉམ་མེད་ཁུ་བྱུག་སྐད་ནི་ངེས་སྒྲོགས་པའི། །སྙན་པའི་དབྱངས་ནི་སྐྱེ་བར་འགྱུར་བ་ཡིན། །དངོས་རྣམས་སྣ་ཚོགས་པ་ཡི་ནུས་ལྡན་་་་ཕྱིར༑། འདི་ལ་ཐེ་ཚོམ་ཟ་བར་བྱ་བ་མིན། །ཞེས་བྱ་བ་ནི་དབྱངས་ཅན་མ་་་་ཆེན་མོའི་སྒྲུབ་ཐབས་སོ། །མཁས་པ་ཨ་བྷ་ཡ་ལས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།ལུག་བསྡུར་མཆན། ་་་་་་:མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དབྱངས་ཅན་་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»ཧྲི། «པེ་»«སྣར་བྱ་བའོ། ་འོན་«སྣར་»མའི། @«པེ་»«སྣར་སྤྲོས་པར། «པེ་»«སྣར་+མ། «པེ་»«སྣར་ལྟ། །«པེ་»«སྣར་བས། «པེ་»«སྣར་»—། ༄༅«པེ་»«སྣར་ཏུ་བཟླས། «པེ་»«སྣར་པཱ། «པེ་»«སྣར་ཐགདགྱེ«པེ་»«སྣར་སྔགས། །«པེ་»«སྣར་མི། 《སྣར་མདའ། «པེ་»«སྣར་ཡིས། གཡུ་བུ་ག་ང་བ་མ་འ་ར་ལ་འདུ་བ་བར་བ་བྱ་བ་བསྐུམས་པའི་བོད་པ་དག་ན་བདག་བདེ་ལ་དེ་ད་བར་གས་པའོ། ད་ལན་བལ་འབད་འབག་འབག་དང་ལྡན་པ་བྱ་བ་དང་ན་་དང་དྲག་དང་::ནས་སྤྱིན་གྱིས་གངས་ལ། ད་ལན་དམ་པ་་་་་་།
