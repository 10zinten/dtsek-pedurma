སརྦ་བྷི་གྷ་ནན་ཧཱུཾ་ཕཊ། ཨོཾ་པད་མ་ཨཱ་ས་ན་ཨཱ་ས་ན་བྷེ་ན་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་དྷྭཾ་ས་ཡ་སརྦ་ཤ་དྲུན་ཧཱུྃ་ཕཊ། ཨོཾ་སརྭ་བ་ར་ཡ་ཀ་ཨ་བ་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏར་ཛ་ཡ་དར་ཛ་སརྦ་མ་ཧོ་ར་གན་ཧཱུྃ་ཕཊ། ༀ་བཱི་ར་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་དྲ་ས་ཡ་དྲ་ས་ཡ་སརྦ་དུཥྚ་ཧཱུྃ་ཕཊ། ༀ་བྷ་དྲ་ཨཱ་ས་ན་ཨ་བྷི་ན་ཡ་བྷི་ན་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་སུ་ག་ཏ་སུ་ག་ཏ་སརྦ་སཏྭ་ན་ཧཱུྃ་ཕཊ། ༀ་ཨུད་ཀུ་ཅུ་ཀ་ཨ་པ་ནི་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨུད་ཀུ་ཊུ་ཡ་ཨུད་ཀུ་ཊུ་ཡ་སརྦ་མ་རཱན་ཧཱུཾ་ཕཊ༑ཨོཾ་ས་ལི་ལ་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨ་བེ་ཤ་ཡ་ཨ་བེ་ཤ་ཡ་སརྦ་མཱ་རཱན་ཧཱུྃ་ཕཊ། ཨོཾ་སྭཱ་སྟི་ཀ་ཨ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་སྭ་སྟི་སྭ་སྟི་སརྦ་སརྦ་སཏྭ་ནཱཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨ་བྷི་མུ་ཁ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨ་བེ་ཤ་ཡ་ཨ་བེ་ཤ་ཡ་སརྦ་དུཊྚ་ས་དྲན་ཧཱུཾ་ཕཊ། ཨོཾ་པ་ར་མུ་ཁ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་མུ་ཁ་ཡ་པ་རཱ་ད་མུ་ཁ་ཡ་སརྦ་བྷི་གྷ་ནན་ཧཱུྃ་ཕཊ། «སྣར་»:ཨོ་བཛྲ་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་པྲ་མརྡ་ཡ་པྲ་མརྡ་སརྦ་བིགྷཱན་ཧཱུྃ་ཕཊ། ཨོཾ་དྷུ་ན། ཨཱ་ས་ན་བཛྲ་མཧཱ་ཀྲོ་དྷ་ལྕི་ར་ཡ་སརྦ་ཙེ་ཏཱ་ནི་ས་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པརྨ་ཀ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ནཱ་ཤ་ཡ་ནཱ་ཤ་ཡ་སརྦ་བིགྷཱན་ཧཱུྃ་ཕཊ། ཨོཾ་པདྨ་ཨཱ་བྷི་ན་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བྷི་དྷ་ན་ས་ཡ་སརྦ་ཤ་ཏུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་སཏྭ་པདྨ་ཀ་ཨཱ་ས་ནི་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏ་རྫ་ཡ་་་ཎ་སརྦ་མ་ཧོ་རཱ་གཱན་ཧཱུྃ་ཕཊ། ཨོ་བཱ་རི་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏཱ་ས་ཡ་ཏྲཱ་ས་ཡ་སརྦ་དུཥྛ་ན་ཧཱུྃ་ཕཊ། ཨོ་བྷ་དྲ་ཨཱ་ས་ན་ཨ་བྷི་ན་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་སུ་ག་ཏ་སརྦ་སཏྭཱ་ན་ཧཱུཾ་ཕཊ། །ཨོ་ཨུངྒུ་ཏུ་ཀ་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨུཏྲུ་ཊ་ཡ་ན་སུ་ཏ:ཨུརླུ་ཊ་ཡ་སརྦ་མཱ་རཱན་ཧཱུྃ་ཕཊ། ཨོཾ་ས་ལཱི་ལ་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་བེ་ཤ་ཡ་ཨཱ་བེ་ཤ་ཡ་སརྦ་མཱ་རཱན་ཧཱུྃ་ཕཊ། ་་ཨོཾ་སྭ་སྟི་ཀ་ཨཱ་ས་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་སྭསྟི་སྭསྟི་སརྦ་སཏྭཱ་ནཾ་ཧཱུྃ་ཕཊ།
