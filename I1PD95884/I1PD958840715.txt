རྒྱུད། ༼ཕུ་ཡིན་ཏེ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། སྒྲོལ་མའི་སྙིང་གར་འཁོར་ལོ་སྒོམ་པ་པོ༑། འཆི་བདག་སྐྱོན་གྱིས་གོས་པར་འགྱུར་མ་ཡིན། །ཟླ་བ་དྲུག་ཏུ་སྒྲུབ་ལ་་བརྩོན་པ་ལ། །དངོས་གྲུབ་ཀུན་ཡང་འབྱུང་བར་འགྱུར་བ་ཡིན། །འཆི་བ་སླུ་བར་བྱེད་པའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།ཨ་:བྷ་ཡ་ལས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།༈ས། །ཁྲོ་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། Q«པེ་《སྣར་བསྟན། «སྣར་»ངས། «པེ་»«སྣར་»འདིས། །«པེ་»«སྣར་ཏ། «པེ《སྣར་ལས། ཛཱ་ན་་་་་་་་་་། ་་་སྤྲོ་བ་ཚཁོ་ར་ནང་། ཁོ་རང་ལ་ཆ་། ཉྙོ། ་༌ཥཱ་ཏྟཱ་ར་གས་པ་མས་ལ་་་་་་། །ཚོད་མ་་་་་ནཔ:དོ་བསྡུར་འབྲས་རེའུ་མིག། ནང་། ད་ལས་རྒྱུད། (དགཔ་དུ་འོ། ༈། །དུ་ཕྱིར་ལ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། ཀ་:སྣར་ཐང་། །བ་ཅོ་ནེ། སྡེ་ཚན། རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། པ་ལྡེབ། བ་ན། །ནན། བརབ། ན་བར། ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། །།ང་ཉིདརྒྱུད། ཏུབུ་།
