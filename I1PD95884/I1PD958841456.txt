ནབཟུང་སྟེ་འཆད་པར་འགྱུར་བའི་རང་རང་གི་སྔགས་ཀྱིས་ཡམ་ཤིང་དང་དྲུར་ཅིབ་དང་ཀུ་ཤ་རྣམས་མིང་མེད་དང་མཐེ་བོང་གིས་བླངས་ནས་སོ། །:། རྩ་བ་དང་ནི་རྩེ་མོ་ཡིས། །།ཞི་གྱུར་རྒྱས་ལ་དེ་བཞིན་ནོ། །རྩེ་མོས་དབང་སྟེ་དེ་བཞིན་དགུག། རེངས་དང་རྨོངས་པ་དག་ལའང་དེ། །རྩ་བའི་ཆ་ཡིས་འཆི་བར་འགྱུར། །དབུས་ཀྱིས་བསྐྲད་པར་འདོད་པ་ཡིན། །སྙིང་རྗེ་ལྡན་པས་དེ་ཤེས་ནས། །སྦྱིན་སྲེག་ཆོ་ག་ཇི་བཞིན་བྱ། །ཨོཾ་བཛྲ་བྲཧྨ་ནེ་སྭཱ་ཧཱ། །པ་ལཱ་ཤའི་འོ། །ཨོ་བཛྲ་ཡཙྪཱ་ཡ་སྭཱ་ཧཱ། ་ཨུ་:དུམྨཱ་རའི་འོ། །ཨོཾ་ས་ཧ་ཀ་རཱ་ཡ་སྭཱ་ཧཱ། ཨ་མྲའི་འོ། །ཨོ་ཀ་དམཱ་ཡ་སྭཱ་ཧཱ། ཀ་དམྤའི་འོ། །ཨོཾ་སརྦ་ཏོ་བྷ་དྲཱ་ཡ་སྭཱ་ཧཱ། གམྦྷིརིའི་འོ༑། །ཨོཾ་བོ་དྷི་བྲྀཀྵ་ཡ་སྭཱ་ཧཱ། :ཨ་ཤཱདྨའི་ཨེའོ། །ཨོཾ་བཛྲ་ལཱ་ཏཱ་ཡ་སྭཱ་ཧཱ། པྤཀའི་འོ། །ཨོཾ་བཛྲ་ཀཱི་བཱ་སྭཱ་ཧཱ། ས་མཱིའི་འོ། །ཨོ་བཛྲ་ཀུ་བེ་རཱ་ཡ་སྭཱ་ཧཱ། ནྱ་གྲོ་དྷའི་འོ། །ཨོཾ་ག་ཏཱ་ཡ་སྭཱ་ཧཱ། སེང་ལྡེང་གིའོ། །་། ཨོཾ་བཛྲ་ཨ་ཤོ་ཀཱ་ཡ་སྭཱ་ཧཱ། མྱ་ངན་མེད་པའི་འོ། །ཨོཾ་བཛྲ་ལ་ཀུ་ཙཱ་ཡ་སྭཱ་ཧཱ། །ལ་ཀུ་ཙའིའོ། །ཨོཾ་བཛྲ་ཨརྐ་ཙེཡ་སྭཱ་ཧཱ། ཨརྐའི་འོ། །ཨོཾ་བཛྲ་སུ་བཱ་ཡ་སྭཱ་ཧཱ། :·བེ་ཀངྐའི་འོ། །ཨོཾ་བཛྲ་བི་ལོ་ཌྷ་ཀཱ་ཡ་སྭཱ་ཧཱ། བི་ལོ་ཌྷ་ཀའི་འོ། །།ཨོཾ་བཛྲ་མ་ད་ནཱ་ཡ་སྭཱ་ཧཱ། མྱོས་བྱེད་ཚེར་མ་ཅན་གྱིའོ། །ཨོཾ་བཛྲ་ཤི་ཁ་རཱ་ཡ་སྭཱ་ཧཱ། ཨ་པ་མརྒའིའོ། །ༀ་བཛྲ་ཨངྒཱ་ནྡྷེཤཱ་ཡ་སྭཱ་ཧཱ། པ་འདི་ནི་ཚེར་མ་ཅན་གྱི་ཤིང་ཐམས་ཅད་ཀྱིའོ། ལ། ཨོ་ཨག་ཡེ་སྭཱ་ཧཱ། མར་གྱིའོ། །།ཨོཾ་སརྦ་པཱ་པཾ་ད་ཧ་ན་བཛྲ་ཡ་སྭཱ་ཧཱ། ཏིལ་གྱིའོ། །ཨོཾ་བཛྲ་ཨཱ་ཡུ་ཥེ་སྭཱ་ཧཱ། དྲུར་བ་ཀུན་དའི་འོ། །ར། ཨོཾ་བཛྲ་པུཥཱ་ཡེ་སྭཱ་ཧཱ། འབྲས་མ་ཆག་པ་རྣམས་ཀྱིའོ། །ཨོཾ་སརྦ་སཾ་པ་དེ་སྭཱ་ཧཱ། །།ཞོ་ཟན་དང་བུར་ཟན་དང་ཟས་མཆོག་གིའོ། །ཨོཾ་ཨ་པྲ་ཏི་ཧ་ཏ་:བཛཱ་ཡ་སྭཱ་ཧཱ། ཏིལ་མར་གྱིས་བྱུགས་པའི་ཀུ་ཤ་རྣམས་ཀྱིའོ། །ཨོཾ་བཛྲ་བི་ཛ་ཡ་སྭཱ་ཧཱ། སོ་བའི་འོ། །ཨོ་།
