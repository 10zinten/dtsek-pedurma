ཡ༌ཨ་བྷ་ཡ་ལས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །དུས་སུ་ནི་བཅོམ་ལྡན་འདས་ཀྱི་ཤར་ཕྱོགས་སུ་ནི་ནོར་བུ་བཟང་པོ་ཁ་དོག་མཐིལ་དག་གོ། །ལྷོ་ཕྱོགས་སུ་ནི་ག་ང་བ་བཟང་པོ་ཁ་དོག་སེར་པོའོ། །ནུབ་ཕྱོགས་སུ་རྣམ་ཐོས་དམར་པོའོ། །བསྡུར་མཆན། རོལ་དུ་དམག་དོའོ། །མི་འགལ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད།  ེ«སྣར་»ལྷའི། ་«སྣར་ལྷ། «སྣར་སྲས། །«སྣར་»ལེགས་པར་སྦྱིན་མ་དང་། «པེ་+ལྷ་མོ་དང་། «པེ་+དག®«པེ་»ཀྱིས། །«པེ་»ནི་བྷ་དྲ། «སྣར་»ཎི་བྷ་དྲ། @«པེ་»ལེནྡ། @«པེ་པཱུ་ཎ་བྷ་དྲ། «སྣར་པུརྞ་བྷ་དཱ། «སྣར་»བེ། «པེ་»ཎ། །(ཁྱོ«པེ་»«སྣར་མ། ཀྱེ«པེ་པི་ཙི་ཏྲ་ཀུཎྜ་ལ་ནི། «སྣར་བྷི་ཙི་ཏྲ་ཀུན་ཌ་ལི་ནེ། ཉེ«པེ་»ཁེནྡྲ། (ཉེ«པེ་»བ་རེནྡྲ། «སྣར་བྷ་རེན་དཱ། «པེ་»བ་སུ་དྷ་ནི། «སྣར་བྷ་སུ་དྷ་ར། «པེ་»ཙནྟཱ་ཀཱ་ཏེ། 《སྣར་ཝཱ་ཀཱན་ཏེ། ་ཚེ«པེ་«སྣར་»ད་ཏྲ། བྱེ«སྣར་»ཏཏྟ། «པེ་»—ཨོ། ས(ཚེ«སྣར་»ད། ལོ་《སྣར་གུཔ་ཏཱ། འགྱེ«པེ་»«སྣར་»དྷེ་བེ། །ནགེ«པེ་སཱ་ཀྱི་ཡི། «སྣར་»སཱ་ཏེ་ཡི། འབྱེ«པེ་«སྣར་ཚོད་གསུམ་དུ་བསྒོམས་ཤིང་། 《《སྣར་པར། །ག《སྣར་»སྣང་། ན«པེ་»སྣར་གྲགས། འགྱི«སྣར་»ན། འ«སྣར་ནི། གྱི《སྣར་། །ཀོང་ཡོད་ས་དྲ་མ། །འདུ་བྱེད་ཀྱང། དར་ལུང་། /བྱ་ནི༽ཌོ་ལསྐ/#*། ཀཱ་ཡ། ཡུལ་གལནད་ཀྱི་མན་ཙེབད་པ། །དང་༼བར་་། པ་*་་*[་ལྟ། རངས་བསྡུར་འབྲས་རེའུ་མིག་བློ་བདེ་བ་ར་ན་་བརྒྱུད། _(་ང་གsgn*/a་། ཆུ་ལ་ང་veHeདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། ://རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །བྱ་བ་རྒྱུད། ང་༽པོད་རྟགས། སུ་ལ་བུཚན་གྲངས། ལྡེབ། བ་བ་ན་བནབ་ན། ན་བ། ཕྲེང༌། བྱ། །ལ། ་་་sཚེག་ཁྱིམ། ཛིཟླ་བརར:ཝཱའོ་།
