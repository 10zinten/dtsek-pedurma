ཚུ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་《སྣར་»༄༅། །པ་ཚབ་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའི་སྒྲུབ་ཐབས་བརྒྱ་རྩ་བཞུགས་སོ། །ཉི«པེ་»«སྣར་—ཨ། ན«པེ་»དེར«སྣར་—དགྲ་བཅོམ་པ། །དམ་པsཡ/s:ར་ཏེ་བད་ཀན་«པེ་»«སྣར་རྫི། «པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་»ལྟ། །།ཀ་ཁ་ག་ཎི་ཀལ་ཀ་ཆུང་e། །ཁ་nཊོ་དང་། ཀ་ཁ་ཁ་ཁང་་ཀ༌་་གོ་ཌར་ཀྱང་ཀ་ཤ་ཀྱི་ཊཁཙོ་ལ། །སྤ་/sz་། ཁྱཟླ་b། pབསྡུར་འབྲས་རེའུ་མིག། ཀ་ཀོ་རྒྱ་ལྔ། །pཁོ་ན་ཡe/ཚོ་ལས་། pབླ་། ་༌༌་རྒྱུད། (པེ་ཅིན། སྣར་ཐང་། །ཅོ་ནེ། རྒྱུད། རྒྱུད། །ཨཐུ། དཔེ་གཞི། །།སྡེ་དགེ། སྡེ་ཚན། རྒྱུད་འགྲེལ། པོད་རྟགས། །ཕུ་ཚན་གྲངས། །ལྡེབ། བ—བ་ནབ་ནབབ?ན?ཕྲེང༌། །ཚེག་ཁྱིམ། །།།་པ་དང་མ་འདས་དེ་བཞིན་གཤེགས་པ་དེ་ནི་བསོད་ནང་ད་བོད་པ་དེ་རྒྱུ་མཐུན་པ། དགེ་བའི་བྱ་བ་རྣམས་ཆུད་མི་ཟ་བ། བཟོད་པས་ཕྱིན། བསོད་ནམས་ཀྱི་གཏེར་གནས་ཀྱི་གཞི། དཔེ་བྱད་བཟང་པོ་དམ་པ་ཀྱིས་པ་པ་མ་མེ་ཏོག་རྒྱས་པ་སྤྱོད་ཡུལ་རན་པར་མཐུན་པ། མཐོང་ལ་པ་མེད་པ། དད་པས་མོས་པ་རྣམས་ལ་མངོན་པར་དགའ་བ། །
