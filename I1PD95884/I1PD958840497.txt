རྒྱུད། །ཚཡུ༽ཀྱི་པཎྜི་ཏ་ཆེན་པོ་རིག་པའི་གནས་རྣམས་ཐུགས་སུ་ཆུད་པའི་མཁས་པ་དག་དང༌། གྲུབ་པ་བརྙེས་པའི་སྐྱེས་བུ་ཆེན་པོ་རྣམས་ཀྱི་གསུང་གི་བདུད་རྩིའི་རྒྱུན་གྱིས་ཐུགས་གཏམས་ཤིང་དེ་ཁོ་ན་ཉིད་ལ་གཞོལ་བའི་སློབ་དཔོན་ཉི་མའི་དབང་པོའི་འོད་ཟེར་དང་། ཆག་ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་རྗེ་དཔལ་གྱིས་ལེགས་པར་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པ། འ་དུས་ཕྱིས་ཤོང་ལོ་ཙཱ་ེབ་དགེ་སློང་བློ་བརྟན་གྱིས་ལེགས་པར་བཅོས་ཤིང་དག་པར་བྱས་སོ། །ཚིགས་སུ་བཅད་པ་ཉིས་བརྒྱ་བདུན་ཅུ་རྩ་་་ལྔ་པའོ། །།།ཅེ་ན་ཁྱེ། ལ་ཁའི། །འཁ་ལ་མྱ་ངའི་ཞི་འགྲོ་བ། དེ་ལྟ་བ་མ་བྱེ་་བར་བྱེད་ཚོ་ཕྲན་ཐོས་གྲོལ་ཤོལ་ཀ་མ་གྲོལ་ཏུ་ཟ་བའི་ནས་ས་གནས་ཁ་ཀྱི་ཞིབ་བསྡུར་མཆན། ཚོ། ོགས་ཀྱི་ཐད་ལཎཎས་བབ་་་་་་་་་་་་་་་་། དེ་དག་ལ་གྲོ་གནང་བ་བ་ཀཀལྔ་ཁྲི་ལྔ། །ཀྱང་། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །སྦྱིན་་སྲེག་གི་ཆོ་ག་འོད་ཀྱི་སྙེ་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་གྱི། «པེ་»«སྣར་ནི། «པེ་»«སྣར་»འདི། ལ«པེ་»«སྣར་ཐབས། ད་«པེ་»«སྣར་བཟུང་། འ«པེ་《སྣར་པ་ཡིས། །«པེ་»«སྣར་»འཆན། «པེ་»«སྣར་»ཀོ། «པེ་»«སྣར་»—དུ། «པེ་»«སྣར་»ཀྱི། ༄༅«པེ་»«སྣར་པོ། «པེ་»«སྣར་ལས་ཕྱིར། «པེ་»«སྣར་»ཚེ། །«པེ་»«སྣར་གང་། /Q«པེ་»«སྣར་»གི། །«པེ་པ། «སྣར་ག་«པེ་»«སྣར་»དུ། «སྣར་»མེའི། །«པེ་»«སྣར་»མཆུར། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»དང་། «པེ་»ཕྱིར་ཕྱིའི་(«པེ་»སྣར་»འབྱེད། «པེ་»«སྣར་དུམ་བ། «པེ་»«སྣར་»ཤཱད་ཐ། ༄༅«པེ་»«སྣར་»ཀ། །«པེ་»«སྣར་པ། @«པེ་»«སྣར་བཅས། «པེ་»«སྣར་གཟུང་། °«པེ་»«སྣར་»ཤ«པེ་»«སྣར་དུར་བ། ཁྱེ«པེ་»«སྣར་»ཀྱི། ཀྱེ«པེ་»«སྣར་བྲན་ཆགས། «པེ་»«སྣར་»རྙིངས་པ་ཡིས། །ལེ«པེ་»«སྣར་བསྐལ་པ། ཨེ«པེ་»«སྣར་པ། ་«པེ་»«སྣར་»ས། ཨེ«པེ་»«སྣར་»དུར་བ་ཅན་དང་། སྐྱེ«པེ་»«སྣར་བསྲེག་ ི«པེ་»«སྣར་ལྟ་པུཏྤ་པ། «པེ་»«སྣར་ས། «པེ་»«སྣར་ན། «པེ་»«སྣར་ན། མི«པེ་»།
