ལོངས་སྤྱོད་རྫོགས་པ་དང་། པསྤྲུལ་པ་དང་། བདེ་བ་ཆེན་པོ་ཞེས་:བྱ་བ་འདི་དག་གོ། ་། ༀ་སརྦ་བུདྡྷ་:ཌཱ་ཀི་ནཱི་ཡེ། ཨོཾ་བཛྲ་ཝརྞ་ནཱི་ཡེ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་འདི་ནི་བཟླས་པའི་སྔགས་སོ། །རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །པ་ཏེ། ཀུན་ནས་བསལ་བྱ་བད་བ་::ས་དཱ་དང་ཨ་བྷ་ཡའི་ཞལ་སྔ་ནས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་པ་ཚོ་དཔོན་ད་དུང་ང་ལ་་བའི་ད་ཀྱིལ་འཁོར་གྱི་བསྡུར་མཆན། བའོ་པོ་ལ་ད་བའོ༑། །ར་བ། །གཡུང་བ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྒྲུབ་ཐབས་བཞུགས། ཚེ«པེ་»«སྣར་པོ། «པེ་»སྒོམ། «པེ་»«སྣར་ཧྲི། །«པེ་»《སྣར་»+མ། «པེ་»«སྣར་»ཧྲི། «པེ་»«སྣར་ན། «པེ་»«སྣར་པས་ནི་གྲི་གུགབཔ་«པེ་»«སྣར་»འབབ། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་+བར། «པེ་»«སྣར་གཞུགས། E:«པེ་»ཌ་ཀི་ནི། «སྣར་»ཌཱ་ཀི་ནི། @«པེ་བུདྡྷ་ཌཱ་ཀི་ནི་ཡེ་ཧཱུྃ། «སྣར་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ་ཧཱུྃ། «པེ་»:《སྣར་»+ཧཱུྃ། «པེ་»«སྣར་»ནཱ། @«པེ་»«སྣར་ཤི། ད་དུ། གང་། ང་། །O«པེ་»«སྣར་པ། ཉི«པེ་»ཌ་ཀི་ནི་ཡེ། «སྣར་ཊཱ་ཀི་ནི་ཡེ། འགན་མ་ཁ་ལ་ཀ་ཀོ་ལ། བ་བླགས་ཞུ་མ་ཚེ། གབསྡུར་འབྲས་རེའུ་མིགག་ལ་ཁ། དང་། རྒྱུ་བའི་ཚེ་ས་རྒྱུད། （རྒྱ་མཀ་བ་ཀ་བ་འདྲ་བ་དང་། པེ་ཅིན། སྣར་ཐང་། ཨེཅོ་ནེ། ལུགསྡེ་ཚན། :ཀུན་རྒྱུད་འགྲེལ། རྒྱུད། ་བ་བརྒྱུད། ་་་་་་་་པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། ནབ་བ་_བ་ན་_ན?ན་བ། ཕྲེང༌། །།།།པའི་ཚེག་ཁྱིམ། དཔེ་གཞི། ་་དབག་རྒྱུད། ཐུ།
