བོཨ་བྷ་ཡ་ལས་དགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །ཁྲིམས་རྒྱལ་མཚན་པས་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཇམ་དཔལ་་གྱི་སྒྲུབ་ཐབས་བསྐོར་བཞུགས་སོ། །«པེ་»«སྣར་»—པོ་ནི། «པེ་»«སྣར་བྷན་དྷུ། ︽«པེ་»«སྣར་»གི། །«པེ་»«སྣར་ནས། «པེ་»«སྣར་ཤ། བསྡུར་འབྲས་རེའུ་མིག། ཀན་ལས་རིགས་ཀྱི་རྒྱུད། （ས་གནང་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། ཀསྣར་ཐང༌། ཅོ་ནེ། ལྔ་བ། ༼ལ་སྡེ་ཚན། རྒྱུད། དཔའ་རྒྱུད་འགྲེལ། རྒྱུད། །ཀྱང:རྒྱུད། །，དེ་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཆོབབ་ན_བ་། ན། —ན། བན། །།ཚེག་ཁྱིམ། །ས་། །་ཎི་ཡ་ཎཆུ་ཚཊ་བྱཁ་ཀ་ཅ་ཀཀ་ཋཀ་pཊ་ཀ་ཀ་ལོ་་རང་བའང་དང༌། ལོ་ཎཚ་ཤ་ཁྲི། ཆ༌་་་pཏཁོ་ཚེ་ཏུང་ན་ཏྟ་ཏོ་བ། ད་པ་དང་། ར་དཀ་ཀpཁ་ཎཀ་ལོ། ར་ཏཚོ}ཆ་དག་ལ་། ཁྱཚ་་་་་་། དང་་་་་་་་ང་ཊྛ་ཌིད་ཁ་སྒྲ་དིཀ་ཏུ་ཥཔོང་། །་ཏཆ་འོ༌་་༌་་ཁ་བས་ན་ང། བ་ཁང་།
