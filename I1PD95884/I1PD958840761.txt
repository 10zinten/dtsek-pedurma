ཆཚོལ་བ་ལ་བསྡུར་མཆན། བློ་འབྱེད་ཀོལ་ཏུn:ཁ་ལ་ད་ཡང་ཡ་བ་རྒྱུ་ལ་ཊ་ཋpཚོ་Eཁ་དཀར་ང་ཀོང་དུ་བན་དnz་བལ་པ། ༼ལ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའ་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ཀུལླེ། «སྣར་»ནས། «སྣར་ལའོ། «པེ་ཨཱ། ཨཱི། ཨུ། «སྣར་ཨཱ། ཨཱི། ཨུ། @«སྣར་ཨེ་ཨོ་ཨ། «སྣར་»པ། ཡིག༼ི《སྣར་»རིམས་ཀྱིས། «པེ་»«སྣར་ན། །ངན་པས་ཀྱང་ན་ལ་«སྣར་—པ། @«སྣར་»ཡི། «པེ་《སྣར་ན། «སྣར་འབྲུའི། 《སྣར་པར། «སྣར་བྱའོ། «སྣར་»ཧྲི། ཉི«སྣར་སྒྲུབས། ARABAਮਾਸਟgਸਸਸਸਲਾਇਲbyR'ਸੌਸਟਮਵਿੱਚਰੋਣਕਲਾਈਨਮਾਨਵੀལར་ཡོ་ག་ལ་ཡ་། ་ཁོ་ཀྱི་་བསྡུར་འབྲས་རེའ་མིག་་རཀ་Eར་རྩའ། ་་ཁ་རྒྱུད། (ཅོ་ནེ། སྡེ་ཚན། རྒྱུད། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། རྒྱུད། །རྒྱུད་འགྲེལ། །::རྒྱུད། རྒྱུད། །པོད་རྟགས། ཡིག་ཚན་གྲངས། ནབནc_བསན_བ། ན_བཕྲེང་། །བར་ལ་དེ་། ཚེག་ཁྱིམ། །།འདངདང་། ལ་བ་བལྡེབ། །བ་ལ་ས་པ། འདས་པ། །ལ། དེའི་འདབ་མ་བརྒྱད་ལ་ནི་སྐུ་ཡི་ས་བ་ལ་ཡི་ས་བྱ་བ་དབྱངས་ལ་འདི་བརྒྱད་ཡོངས་འདུ་བྱ་ར་པ་ལས་བུང་བ་བརྒྱད་དེ། དགའ་ཞིང་རྒྱས་ལ་ཁྲོ་བར་གྱུར་པ་ཁ་དོག་དམར་དེ་རིམ་གྱིས་འགན་ས་པ་རྣམས་པ་དང་པོ་སྟེ་བདག་གི་རླུང་གིས་བསྐུལ་ནས་རང་གི་སྣ་བུག་གཡོན་པ་ནས་བྱུང་སྟེ། །
