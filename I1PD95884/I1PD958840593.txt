བདེ་ཆེབསྟན་འགྱུར། རྒྱུད། ༼ཕུ་ཎ་ཁཙོ་ཥཁོ། ལ་བའི་གནས་ག་ལ་བ་དེ་ཚོ་ਇਸਸਈਅਰਸਮਾਰਮਲਸਿੰਘਬਖਸਖਸਮਾਰན་བསྡུར་མཆན། རུ་ཀཁགདང་ལ་ཊ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་ཀུམ། 《སྣར་ཀུན། །«པེ་»«སྣར་བཞུགས་པར་བྱ་བའོ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་+རྣམ་པར། །«པེ་»«སྣར་བསྐྱེད། ་«པེ་»«སྣར་གྲགས། ༄༅«པེ་»«སྣར་མོ། ：/«པེ་»«སྣར་»སྐྱོང་། །གྱི«པེ་《སྣར་གྲོནྣ་གྲོསྣ། པོ«པེ་»«སྣར་ཨོ་ཊ་ཊི། «པེ་»«སྣར་+འཇིག་རྟེན་དབང་ཕྱུག«པེ་»«སྣར་ཨ་ཀ་ར་ལས། BEਜ਼ਰੂygpਕਮਲਇਵਅਖਾ'ਕੁਲਸཀྱི་ཤེས་བྱ་ཁྱབ་ཀྱི་ཁ་བསྡུར་འབྲས་རེའུ་མིག། རྒྱུད། (Spਰਪਨདཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། རྒྱུད། སྒྱུ་པོད་རྟགས། ཚན་གྲངས། །ལ་བར། །ལྡེབ། ནནབན། ན་ན་བབ་ཕྲེང་། །།།ཚེག་ཁྱིམ། རཔ་པ་ཡིན་ཀྱང་། །བཟོ་ཁོདབྱངས་ཡོད་བ་་་འཁང་། ཉིད་ལ། ཞེས་ས་བབས་བ་དང་་་་བདག་ཉེ་ལམ་གསུམ་པ་འགྲོ་བྱ་སྟེ། །
