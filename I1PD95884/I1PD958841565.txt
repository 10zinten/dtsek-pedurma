བསྟན་འགྱུར། རྒྱུད། ༼བུ༽«སྣར་གྱིས། «པེ་པར། «པེ་+ཡང་། དབྱ«པེ་+པ། 《སྣར་བས་བྱའོ། །།།བ་པེ་»ས། «པེ་སྐ། །:«པེ་»སྔར་བ་ལང་ས་བ་འགྲོ་བ་རྡོ་རྗེའི་ཆང་། «པེ་»«སྣར་»བསམས་ཏེ། 《སྣར་»གི། «པེ་»«སྣར་རྣམས། ་«པེ་»«སྣར་བཅིངས་པ། «པེ་»དེབསྒོམས། །«པེ་རྣམས། དམ་པ་་་་ཏ་ཀྱི་མ་་་°། ་(«པེ་»+ཀྱིས། ་(«པེ་»+མོ། ་«པེ་+སྦོམ་སུ། ས«པེ་»དུ། «སྣར་»སྦོམས་སོ། ?«པེ་»ཕྱེད་དང་གསུམ། སེང་ལྡེང་ལས་བྱས་པའམ། རླུས་པ་ལས་སྤྲོས་པའི་སྲོ་པོ་བཅུ་ལ་གདང་བྱས། «པེ་»པ། «པེ་»དམར་པོས། 《སྣར་དམར་པོ། ་ ི«པེ་སྐུར་ལྔས།  ོ་《སྣར་བསྔགས། «པེ་»ད་པ། །}:«པེ་»དུ། «པེ་པར། 《སྣར་པ་རྣོ་བ། །«པེ་»ཀྱིས་གཟུགས་ཅན་ཁྲོ་བོ། ལ«པེ་»ཀྱིབར། 《པེ་»ནས། ་«པེ་རྗེ། ་ཏུ«པེ་བ་བཅིངས། །་ེ«པེ་ཅན་གྱིས། ་འགྱེ«པེ་»—ལ། («པེ་ཨོ་ཨཧཱུཾ་ཡ་མ་ཨ་ན་ཏ་ཀྲིཏ་སརྦ་དུཥྛ་ཨི་ན་ཨུ་པ་ཨི་ན་དཱ་ན་ས་པ་རི་བ་རན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨ། །ཧཱུྃ་པྲ་ཛྷ་ཨན་ཏ་ཀྲི་ཏ་སརྦ་དུཥྚ་ཡ་མན་ས་པ་རི་བ་རན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཧཱུཾ་པད་མ་ཨན་དྲ་ཀྲི་ཏ་སརྦ་་་དུཥྤ་ནཱ་ག་ན་ས་པ་རི་བ་ར་ན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོ་ཨཧཱུྃ་བགྷིན་ཨན་དཱ་ཀྲི་ཏ་སརྦ་དུཥྚ་ཀུ་བེ་ར་ན་ས་པ་རི་བ་་རན་གི་ལ་ཡ་ཧཱུྃ་ཕཊ། །ཨོ་ཨཧཱུྃཨ་ཙ་ལ་སརྦ་དུཥྚ་ཨི་ཥ་ནན་ས་པ་རི་བ་རན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཧཱུཾ་ཊ་ཀ་ཀི་སརྦ་དུཥྚ་ཨ་ག་ནཱི་ན་སརྦ་ས་པ་རི་བ་རན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཧཱུྃནི་ལ་ཏ་ཎ་ད་སརྦ་དུཥྚ་ནི་རི་དཱི་ན་ས་པ་རི་བ་རན་ཀི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཱཿཧཱུྃ་མ་ཧཱ་བ་ལ་སརྦ་དུཥྚ་བ་ཡུ་ན་ས་པ་རི་བ་ར་ནི་ཀ་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་་ཨཱཧཱུྃ་ཨིཎྜ་ཥ་ཙ་ཀྲ་པ་ར་ཏི་ན་སརྦ་དུཥྚ་ཨར་ཀ་ཙན་དྲི་བི་དཱ་ན་ས་པ་རི་བ་རན་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཧཱུཾ་སུམ་བྷ་རཱ་་ཨཛཱ་སརྦ་དུཥྚ་ཧྥེ་ཙི་དྲི་བྲི་བཱི་ཐི་བཱི་དེ་བ་ད། ས་པ་རི་བ་རན་ཧཱུཾ་ཕཊ། «སྣར་»ཨོཾ་ཨཱཧཱུཾ་ཡ་མཱནྟ་ཀྲྀཊ་སརྦ་དུཥྚ་་་ཨིནྡྲ་ཨུ་སེནྡྲ་ནཱ། ་ས་པ་...པ་རི་ཝཱ་རཱི་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། ...ཨཱི་ཤ་ན་ས་པ་རི་...དུཋ་ནི་རིནྡྷ་ན་ས་པ་རི་ཝཱ་ར་ན་»ཏོག་ཀཱིཙནྡྲ་པཱི་ཏ་མ་ཧཱ..ཝཱ་རཱན་ཀཱི་ལ་ཡ་ཅེས། (《པེ་བཅུ་བྱས་པས་བར། «པེ་»གྱི། «པེ་»ད་ལ་བརྡུངས་པར། ་ས«པེ་གྱིས། །«པེ་ལ། «པེ་»པོའི་གཟུགས་ཅན། («པེ་པ་རྣམས་ཀྱིས། །ཁ་«པེ་པའི། «པེ་»བྱའོ། །«པེ་གེགས་ལ་ཕུར་པས་གདབ་པར་ཆོ་གའོ། །་(«པེ་»—དཀྱིལ་ལ་འཁོར་ལ། «པེ་བ་ཡི་དཀྱིལ་འཁོར་ལ་ཐིག་གདབ་པའི་ཚོགས་ཀྱིས། 《སྣར་»བའི་ཆོ་གས་ཐིག་གདབ་་བ་ལ་ཁ་པའི་ཆོ་གས། «པེ་ལ་ཇི་ལྟ་བ་རྗེས་སུ་འབྲང་། །«པེ་»མོ། «པེ་དཔའི། །(«པེ་བ། ་འ«པེ་»པོ་བསྣམས་པ་རྒྱན། 《སྣར་པོ་བསྣམས་པ་བརྒྱན། །«པེ་»+མ། །«པེ་»ཀྱི། བ་ཀུ་ལ། (《པེ་»+འཛིན། ་པོ«པེ་»«སྣར་»པའི། «པེ་»པོ། ན«པེ་»སྣར་ལས། ་ཚེ«པེ་ཧཱུཾ། «པེ་ཧི། (《སྣར་»། །«པེ་སྐ། །«པེ་»སྔོན་པོའི་འོ། །དེ་ནས་སྔ་བར་ལངས་ལ་་་་་་་་རྡོ་རྗེ་འཆང་གི་གོ་འཕང་ལ་དགོད་པའི་གནས་པའི་ཆོ་གའོ། །ཌ།
