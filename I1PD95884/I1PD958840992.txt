ལ་སྔགས་འདི་དག་བཟླས་པར་བྱ་སྟེ། སྤྱི་ཨོ་ཨ་ཨཱ་ཨི་ཨཱི་ཨུ་ཨཱུ་:རྀ་རཱི་ལ་ལཱི་ཨེ་ཨི་ཨོ་ཨོ་ཨོཾ་སྭཱ་ཧཱ། བདག་མེད་མ་འབའ་ཞིག་གི་སྒྲུབ་ཐབས་རྫོགས་སོ། །དེ་ཡི། པ་བིདགེ་སློང་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །།།ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། 《་《་ང་པ་བ། ནི་༄༅། །'g'བསྡུར་མཆན། ཁཡ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བདག་མེད་་མའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་»རཱདྨ། «པེ་»«སྣར་»མའི། &«པེ་བརྒྱན། «པེ་»《སྣར་»སྣོམས། པོ«པེ་»ཊཾ་ག་བསྣོམས། 《སྣར་»ཊཾ་ག་བསྣོམས། «པེ་»—པ། «པེ་»《སྣར་»ཟུངས། («པེ་རི་རཱི། «པེ་»«སྣར་+དཱ་དང་ཨ་བྷ་ཡ་ལས། །ལྔ་ལ་་་ཊ་ཋ་ཌ་ཆུང་ཚེ། སྐྱེ་བ་ལྔ་པོ་ཁཆ་ཚez་བློ་ཥ་ཥJeཥ་ཅཆུ་ལོ། ན་ས་ན་ཚ་ལང་། རང་ཀ་་་་་་༌ཎ་ཌལ་བསྡུར་འབྲས་རེའུ་མིག་ང་། །ཁཎྜ་ཚོབ་་ཁད་ཀོཁ་རྒྱུད། （ང་དང་ཁྱེར་དུ་མ་བཏོ་། །དཔེ་གཞི། ༽སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། ཏ་སྡེ་ཚན། རྒྱུད། སོགས་ཀྱི་རྒྱུད་འགྲེལ། རྒྱུད། ་་པོད་རྟགས། །བུ::(ཚན་གྲངས། །ག་ད་བ་མ་ལ་་། ལྡེབ། ནབ་བརནབ_ན?། ན་བར་ཕྲེང་། ཆུa*ཟླ་དང་ཚེག་ཁྱིམ། །,་་་ཀླ་ཀུན་ཟླ་ཁ་སྤུ་ཎི་ཀ་བ་ཀ་ཟླ་ཀ་ལ་མི་གཏོ་་ཏཎྜ་ཁོ་/འི་ལ་ཀྱི་ཚེ་རྒྱུད། ཐུ།
