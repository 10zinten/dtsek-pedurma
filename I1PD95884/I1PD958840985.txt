་། ཁ། ཁྲོ་ལ་ཁ་ཎ་ཙི་ཎཊི་ཏུ་#*། །བློ་ཙ::ཀཌ་། ་ཌ་ཚོ་ལ་ਲੌen?''ਨੂੰਯg'ਸੋਸ਼ਲਸੜੋ'ਸਮਟਰਮਬਣ། །།་ཏཏ་ལྷོ་། །ས་སྐྱ་མཁ་བ་ལོ་ཆ་ཁཁ་ཁཟླ་། ས་ཏཁ་ཁ་ཁ་ར་ཊ། །།ཀ་ཁ་ལ་ཀ་ཎe། ཆ་ཏུ་ys''pཆུ་ཊོ་ཏིས་་་༄༅། །དགྱེས་པའི་རྡོ་རྗེའི་སྒྲུབ་ཐབས། ༼ཚB/ཡང་ཁཊོ་ཁུ་ལ་</aཀྱི། ཁ་རོ་ཁ་བ་*བློ་ཌ་ཌ། རྟ་ལ་ཀོ་ཎeཆ་ཡིན་ཏཆི་ཚོ་༼སྐད་ཡཏུ་ཁཆ/??༽། ཁརྒྱུ་དང་། ཚེpའིལ་《་ཁཏ་ཁ་《་བnཟློསདྲ་ཁ་ཀཾ། [ཞུང་ལ་《་༼། 《་དཔ《་དོའི»༽༄༅། །རྒྱ་གར་སྐད་དུ། ཧེ་བཛྲ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། དགྱེས་པའི་རྡོ་རྗེའི་སྒྲུབ་ཐབས། ཞི་འཇིག་རྟེན་གསུམ་སྐྱོང་ཧེ་རུ་ཀདོ། བླ་མ་་་མགོན་ལ་ཕྱག་འཚལ་ནས། །ཤཱནྟིའི་ཞལ་སྔ་ནས་གསུངས་དེའི། །སྒྲུབ་ཐབས་ཆོ་:གར་རྣམ་པར་དབྱེ། །དཔའ་བོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། །མགུལ་པར་བདག་མེད་མ་ཡིས་འཁྱུད། །འཁོར་ལོ་ལ་སོགས་རྒྱན་དང་ལྡན། །སྤྱན་གསུམ་རྐང་གཉིས་རྒྱལ་པོའོ། །རྒྱུད་འདིར་བསྒོམ་པའི་དབང་དུ་བྱས་པའི་རྣལ་འབྱོར་པས༑ནམ་མཁའ་བྱུང་བ་དང་སྟན་ལས་ལངས་ལ། རང་གི་སྙིང་གར་ཉི་མ་ལ་གནས་པའི་ཡི་གེ་ཧཱུྃ་བལྟས་ཏེ། དེའི་འོད་ཟེར་གྱིས་དཔལ་ཧེ་རུ་ཀ་ལེགས་་་པར་གསལ་བར་བྱས་ཤིང་སྤྱན་དྲངས་ལ། མེ་ཏོག་ལ་སོགས་པའི་ཉེ་བར་མཁོ་བའི་ཡོ་བྱད་ལྔས་ཡང་དག་པར་མཆོད་དེ། དེའི་སྤྱན་སྔར་སྡིག་པ་བཤགས་པ་་་དང་། བསོད་ནམས་ལ་རྗེས་སུ་ཡི་རང་བ་དང་། །གསུམ་ལ་སྐྱབས་སུ་འགྲོ་བ་ས།
