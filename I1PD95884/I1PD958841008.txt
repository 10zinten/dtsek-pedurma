རྡོ་རྗེ་ཕག་མོའི་དབང་དུ་བྱ་བའི་ཆོ་གལས་སྐྱེས་དང་ཐང་སྐྱེས་ཀྱི། །རེག་པས་གོ་བ་དེ་བྱེ་བ་དེ་དྲང་བདེན་དང་མིག་འགྲུལ་བྱེད་པ་དང་། །ཚོགས་གསག་པ་ཉིད་དང་ནི་པད་པ་དང་བརྒྱད་ཀྱི་གོལ་ས་མེད་བསྡུར་འབྲས་རེའུ་མིགག་ཚོད་བྱ་དང་། །:བདུན་པ་མ་པ་མང་པོ་དང་། རྒྱུད། （ང་པ་ཚུ་ཕྱིས་བྱེད་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། ང་སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། ལྟརྒྱུད། རྒྱུད་འགྲེལ། ::རྒྱུད། པོད་རྟགས། །ཚན་གྲངས། ལྡེབ། །ན་_བ*། བ་བ་ནན་། ན་བ་ཕྲེང་། །དེ་་་བ་ཚེག་ཁྱིམ། +མཀྱི་བུ་ཞུ་བ་ལས་ཀྱི་ཞང་ས་ཁྲ་འད། ཁོ་བ་ད་ན་བ་མ་དག་པ། བ་བཏོ་ན། བ་ཟོ་ཁོ་ནར་བཏང་ང་ཚོ་གང་བཞེས། ཁོང་ཚོ་ཀ་ཁ་་ཡབ་ཀྱི་ཞན་ཆ་ཊ་ཡང་ཁཟླ་བ་བབ། གལ། །འོ་ཁོ་འཚོ་བ། །མཁྲིཊ་ཆེ་བ་དང་ཚོ་ཁ་ཎ་ཊ་ཀ་ན་གསང་སྒྲ་ཀ་ཀ་ལན་བ་ལ། །སྐྱེ་པ་ཊ། ། ། ཨོ་ཎ་ཁའི་ནང་། ཀྱི་དབང་གནང་བ་ཚོ་་བྱ། དེས་བཏོ་ཆུ། ཟཞཀར་མ། །ཕྱི་ནང་ཀ་ར་ཎཊ་་་་་ཝ་ཎིཊ་ཋ། མ་སྐྱེའི། མ་ན་ཀ་ལ་ཁག་ཏ་བློ་སྟེ། དེ་བས། དེ་བྦབ་ཆེ་ལ་སྤྱན་ཀཚུ་ན་ཤོ་སྦེ་ཁྲུའི་ཞལ་སྤྲོ། འོ་ཀ་ཤ་ཀ་ལ་སུང་རྒྱུ་གྲོགས་། ཆང་བག་མ་ང་ཚོ་ཁ་བྲིད་པ་ཀ་དང་། ད་ལྟ་བའི་དག་ན་བཏང་ནས་བོད་ཁ་བ་མ་ལེན་::ཀ་བ་བརྒྱ་ཤོག་ལོ་ཁཊ་ཋ་བྱཁའི་བདེ་མ་ལྷོ་ལུ་བ་ཁ་ཚོན་གསལ་ཀྱི་ཁ་ཆེ་བ། ་་་་་་་། སྐུ་ནི་དུ་བའི་ལུ་ག་ན་འདུག་པ་ལ་ཁོ་ན་བ་ལ་སེམས་ན་ཁ་ཆརྒྱ་ན་འབར་གྱི་རང་_།
