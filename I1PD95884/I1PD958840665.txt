བསྟན་འགྱུར། རྒྱུད། ༼ཕུདག་རྟོགས་པར་བྱ་སྟེ། ཕྱིན་ཅི་ལོག་གི་ཡུལ་ལེགས་པར་ཁོང་དུ་ཆུད་པར་བྱ་ཞིང་ཁོང་དུ་ཆུད་པར་བྱས་ནས་ཀྱང་བསྒོམས་པས་དབང་དུ་བྱ་བ་ཡིན་ནོ། །།།ཁོར་་་ཡུག་:མངར་བྱེད་བསྒྲུབ་པའི་ཐབས། །རྣམ་བཀོད་ལས་འབྱུང་བསོད་ནམས་གང༌། །དེས་ནི་སྐྱེ་དགུ་མཐོང་འགྱུར་ཞིང་། །སྙིང་རྗེའི་སྟོབས་ལྡན་དཔལ་དུ་ཤོག། ཁོར་ཡུག་མངར་བའི་སྒྲུབ་ཐབས། ་:སློབ་དཔོན་ཆེན་པོ་པདྨ་འབྱུང་གནས་ཀྱིས་མཛད་པ། ཨ་བྷ་ཡའི་ཞལ་སྔ་ནས་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་་བསྒྱུར་བའོ། །།།གང་ལ་སློབ་དང་། གསང་བད་ཀུན་ཏུ་ཞི་བའི་བབབབནབ་ཤིག་གནང་བ་དང་། །བདག་བློ་དད་པ། :དང'''ག་བློན་ཁ་བ་ལ་བསྡུར་མཆན། བཀབ་པར་ཡིག་ཏྠབྱུ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །ཁོར་་ཡུག་གི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་མནྡཱ་ཐ། «པེ་»«སྣར་»སྒྲུབ་པའི། «པེ་»«སྣར་»བའི། «པེ་»«སྣར་དགའི། @«པེ་»«སྣར་»+ཨ་དང་ཡི་གེ། «པེ་»«སྣར་བྱ་བའོ། །«པེ་《སྣར་»སྐུ། «པེ་»«སྣར་+དང་། «པེ་»«སྣར་རྣམས། «པེ་»«སྣར་»+དང༌། པོ«པེ་»«སྣར་»སྒྲུབ། །«པེ་»《སྣར་བུག་པ་ནས་དེ།  ི«པེ་《སྣར་ཟེ། «པེ་»«སྣར་+རང་། ཚེ«པེ་»«སྣར་སྒྲུབ། «པེ་»«སྣར་མགུལ། «པེ་»སྣར་ཟེ། ཚེ«པེ་»«སྣར་»+ཁོ་ན། ཉི«པེ་»«སྣར་»+དེ་ཁོ་ན་བཞིན་དུ་བཏང་ཞིང་། «པེ་»«སྣར་པ། (གྱོ«པེ་»«སྣར་+ནས། །«པེ་»«སྣར་སྒོམ། «པེ་»«སྣར་»སྔར་བྱེད་སྒྲུབ། «པེ་»«སྣར་བྱུང་། «པེ་»«སྣར་»ཀྱི། ཆར་བ་འབབ་པ་དང་ང་ཚོ་དང་། །འོ་འུ་འང་། དབn:བར་བ་ཁྱད་པའི་བོ། ་པ་ཀྱི། །ཐུ་ཁྱེ་ལྟ་གོ་ཆ་ལ༌ཞPཆ་ན། །དབྱར་ཁ། ན་བྱ་དེ་མགོ་ལ་ਇਸਤੋਂਬਾਅਦਪਾਕਖਾਨਨੇਉਸਦਿਨਉਸਤਾਏਇਸཞེས་པ་ལ་་་་་་བྱང་གོ་ཁགྱི་ཚོང་བའི་ཁང་ཁྱིམ་བཀའི་།
