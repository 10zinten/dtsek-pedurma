ཀྲྀ་ད་ཡཱཧཱུཾ། ཧྲིདྡྷཾ། ཊཱིཀྲྀ་ད་ཡི། ཧྲིདྡྷི། ཏྲཾ་ད་ཡཱ་ཡ། ཧྲ་དེ། ཀྲྀ་ད་ཡཱ། ཧཱུཾ། ཧྲིདྡྷཾ། ཧྲི་ད་ཡེ་ཤྩ། །ཧྲིདྡྷ། ཧྲི་ད་ཡཱཏ། ཀྲྀ་ད། ཀྲྀ་ད་ཡཱཧཱུཾ། ཧྲིདྡྷཾ། ཧྲི་ད་ཡེ་། ཧྲིདྡྷ། །ཏྲཾ་ད་ཡ་སྱ། ཀྲྀ་ད། ཀྲྀ་ད་ཡ་ཡོ། ཀྲྀ་དོ། ཧྲི་ད་ཡཱ་ནཾ། ཧུཾ་དཾ། ཧྲི་ད་ཡེ། ཧྲ་དི། ཧྲི་ད་ཡ་ཡོ། ཧྲི་དོ། ཀྲྀ་ད་ཡེ་། བྲྀཏུ། ཧེ་ཧྲི་ད་ཡ། ཧེ་ཧྲི་ད་ཡོ། ཧེ་ཧྲི་ད་ཡ། :ས་དྲི་ཤ་ཞེས་པ་འདིའི། ཎསུའི་འོ། །སྲྀ་ཛ་ཚྭ་ཛ་ཡ་ཛ་རཱ་ཛ། བྷ་ཛ་ཤ་རྣམས་ཀྱི་ཥའོ་་ཞེས་པ་ཥ་ཉིད་དོ། །ཛྷལའི་ཛ་ཤའོ་ཞེས་པ་ཥ་ཡིག་གི་ཌ་ཡིག་གོ། །ཀྭནའི་འོ་་ཞེས་པ་ཀུ་ཉིད་ཀྱི་ག་ཡིག་གོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པ་ཀ་ཡིག་གོ། །ཁར་ལ་ཛྷལའི་ཙརའོ། །ཀུ་ལས་ཀྱང་ཞེས་པ་ཥ་ཉིད་དོ། །ས་དྲིག། ས་དྲི་ཤོ། །ས་དྲི་ཤ། ཁོས་དྲི་ཤོ། ་ས་དྲི་ཤོ། ས་དྲི་ཤ། ས་དྲྀ་ཤཱ། :：ས་དྲིགཧཱུཾ། ས་དྲེགྲ་བྷི༄༅། ས་དྲི་ཤེ། །:：ས་དྲིག། ས་ཕྲོག་བྷིུ། །ས་དྲི་ཤ། ་་:ས་་ཕྲོག་ཧཱུཾ། ཌས་ག་°། ཟོ་ས་དྲི་ཤ། ས་དྲི་ཤོ། ས་དྲི་ཧཱུཾ། །ས་དྲི་ཤི། ས་དྲི་ཤོ། ས་དྲེགླུ། ཧེ་ས་དྲྀཀྲ། ཧེ་ས་དྲི་ཤོ། ཧེ་ས་པྲ་ཤ། དིཝ་ཞེས་པ་འདིའི། དིཝའི་ཨོཏའོ་ཞེས་པ་ཝ་ཡིག་གི་ཨོ་ཉིད་དོ། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ། །དིཝའི་མཐའ་ལའང་ཨུཏའོ་ཞེས་པ་ཝ་ཡིག་གི་ཨུ་ཉིད་དོ། །ཡཎ་རུ་་་བསྒྱུར་བའོ། །ཀུ་ལས་ཀྱང་ཞེས་པ་ཥ་ཉིད་དོ། །རྣམ་པར་གཅད་པའོ། །བྱོ༑བོདི་ལོ། ༼དི་ཝ། །དི་ཝཾ། དི་ལོ། དི་ཝ། དི་ཝཱ། བྱུ་ཧཱུཾ། དྷུ་བྷི། དི་ཝེ། ཨོ། དྷུ་བྷཾ། །བྱུ་། དི་ཝ། །དྷུདྡྷཾ། བྷུདྡྷ། །དི་ཝ། དི་ཝོ?། དི་ཝཾ། "དི་ཝི། དི་ཝོ། དྷུ་ཥཱ། ཧེ་དྲོ། ཧེ་དི་ལོ། ཧེ་དི་ཝ༑ཏེ་སྤྲོད་ཅེས་པ་འདིའི། སུའི་འོ། །རྫོགས་པ་ལ་ཡང་ནའོ། །ཁར་ལ་ཛྷལའི་ཙརའོ། །རྣམ་པར་གཅད་པའོ། །སྲིད། སྲྀཏ། ཀྲྀ་དོ། ཀྲྀ་ད། །
