བསྟན་འགྱུར། སྒྲ་མདོ། རེཔའི་མདོའི་ཥ་ཉིད་དོ། །ས་ལ་ཥཌྷ་དག་གི་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཀུ་ལས་ཀྱང་ཞེས་པས་ཥ་ཉིད་དོ། །ཨཀྵེ། མིག་མཾ་ཀ་ཎི་དག་ལས་ཨཱི་ཙིའོ། །མྲང་སྲོག་གཏོང་བ་ལའོ། ནི། ང་ཡིག་རིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། ལོ་ཀ་ཎ་རྒྱུན་ཆགས་པའི་བྱིངས་སོ། །འདི་དག་ལས་ཨཱི་ཙི་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨི་ཀའི་་་་ཨཏ་ཨེངའོ་ཞེས་པས་ཨ་ཉིད། རཾཀའི་ཨཎ་ལ་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད༑མ་རི་ཙིཏྟཾ། སྨིག་རྒྱུའམ་འོད་ཟེར། ཀཎྜི་༄༅། འཁྲི་ཤིང་། རཱ་ཤ་དི་དག་ལས་ཀྲྀ་སའོ། །རཱ་ལཱ་ལེན་པ་ལའོ། །།ཤ་དྲི་རྣམ་པར་རྒྱུ་བ་ལའོ། །ལོ་ྋཡིག་ནི་ལཾ་དིཏ་རླུཏ་སོགས། པུ་ཥྛ་ཏི་སོགས་རྣམས་ལས་ཨ་ཏང་ལའོ་ཞེས་པའི་རྟགས་དོན་ཏོ༑། འདི་དག་ལས་དྲྀས་ཕ་རོལ་ཏུ་འགྱུར་ཏེ། ས་ཡིག་ནི་དབྱངས་ཀྱི་དོན། རྭ་ཏྲི༑ཡམཚན་མོ། ཁར་ལ་ཛྷ་ལའི་ཙརའོ་ཞེས་པས་ད་ཡིག་གི་ཏའོ། །ཤཎྜི། གླང་པོ་ཆེ། །བྷུ་སྲུང་ཨ་དི་རྣམས་ལས་ཀྲྀནའོ། །བྷུ་ཡོད་པ་ལའོ། །སྐྱུད་རབ་ཏུ་སྐྱེ་བ་ལའོ༑། ང་ཡིག་སྦེ་ྋསྲུང་ཨུ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། ཊྛི་བུ་ཥཀྐ་མ་ཡིན་པའི་དང་པོའི་ཥའི་སའོ་ཞེས་པས་ཥའི་ས་ཨ་ད་ས་ཟ་བ་ལའོ། །འདི་རྣམས་ལས་ཀྲིན་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་ན་ཡིག་དབྱངས་དོན། ་ཀ་ཡིག་ཨཏ་ཨེང་དགག་པའི་དོན། བྷཱུ་རི། ཡམང་པོ། སཱུ་རི། ཡཉི་མ། ཨ་དྲི། རི་བོ། ཤ་ཀ། བྷཱུ་དག་ལས་ཨུནྟི། ་ཨནྟི་ཙ་དག་གོ། །ཤཀྑཾ་ནུས་པ་ལའོ། །བྷུ་ཡོད་པ་ལའོ། །འདི་དག་ལས་གྲངས་བཞིན་དུ་ཨུནྟི། ་ཨནྟི་ཙ་དག་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཙ་་དབྱངས་དོན། ཤ་ཀུནྟི། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨོ་ཉིད། »ཨེ་ཙའི་ཨ་་ཡ་ཨ་བ་ཨཱ་ཡ་ཨཱ་བ། རྣམས་སོ་ཞེས་པས་ཨབ་རུ་བསྒྱུར་རོ། །བྷ་བནྟི། དུས། ཨརྟི་ལས་ཨནི་ཙའོ། །རྀ་འགྲོ་བ་ལའོ། །།འདི་ལས་ཨནི་ཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཙ་ཡིག་དབྱངས་ཀྱི་དོན། ཨི་ཀའི་ཨཏ་ཨེངའོ་ཞེས་པས་ཨ་ཉིད་དང་། ར་ཕ་—།
