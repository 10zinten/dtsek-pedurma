བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིཔོ་ཉིད་དོ། པུཎྞ། ...གང་བ། །ཀྲྀ་ཏི་ལས་སུཀ་ཡང་ངོ༌། །ཀྲྀ་ཏི་གཅོད་པ་ལའོ། །འདི་ནས་ནཀ་ཕ་རོལ་དང་སུཀའང་ངོ༌། །ཀ་ཡིག་ཀིཏ་བྱ་བའི་དོན་ཏོ། །དེས་ན་ཀིཏ་ལའོ་ཞེས་པས་ཨཏ་ཨེང་བཀག་གོ། །ཀྲྀནྣཾ། ཟད་པར་རམ་མཐའ་དག་གམ་མ་ལུས་པ། ཟླ་ཥི་ལས་ཨིཏའི་ཨཏ་ཀྱང་ངོ༌། །ཀྲྀ་ཥ་འཁྱུད་པ་ལའོ། །འདི་ལས་ནཀ་ཕ་རོལ་ཏུ་འགྱུར་ཏེ། །སུཀའང་ངོ་། །ཨིཏ་ཀྱི་ཨཏ་ཅེས་པས་ཨི་ཡིག་གི་ཨ་ཡིག་གོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །།ཀུ་ལས་ཀྱང་ཞེས་པས་ཥ་ཉིད་དོ། །ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །ཤཱཀྵ། འཇམ་པོ། ཌ་ཏི་ཛིའི་ཨཱིཏ་ཀྱང་ངོ༌། །ཏི་ཛ་རྣོ་བ་ལའོ། །།འདི་ལས་ནག་ཕ་རོལ་དང་སུཀའང་ངོ་། །ཨི་ཡིག་གི་ཨཱི་་་་ཡིག་གོ། །།སྔ་མ་བཞིན་ཥ་ཉིད་དོ། །ཀུ་ཉིད་དང་ཎ་ཉིད་ཀྱང་ངོ༌། །ཏིལླཱ། ཡརྣོན་པོ། རསྣ་ལ་སོགས་པ་རྣམས་སོ། །རཱ་ལཱ་ལེན་པ་ལའོ། །།ཥ་མཐའ་ལའོ། །ཊྛི་བུ་ཥཀྵ་མ་ཡིན་པའི་དང་པོའི་ཥའི་སའོ། །།རྒྱུ་མཚན་མེད་ན་སོགས་ཀྱིས་ཋའི་་ཐའོ༑། ཨི་ཎ། བཱི་བཱ་འགྲོ་བ་ལའོ། །ཀྲྀ་ཥ་སྐོམ་པ་ལའོ། །རྀ་འགྲོ་བ་ལའོ། །ཛྲ་རྒ་བ་ལའོ། །འདི་རྣམས་ལས་ནག་ཕ་རོལ་དང་སུགའང་ངོ༌། ་། ཌཱཔའོ། །རསྣ༑ཡསྨན། སཱ་སྣ། ལྐོག་ཤལ། ཏིཥྛ་ཏིའི་ས་ཡིགའོ། །ཥ་དབྱི་བ་དང་ཎ་་ཉིད་ཀྱང་ཚིག་ཕྲད་དོ། །སྦྲ་ནཱ། ཀ་བ། བཱི་ནཱ། གི་ཝཾ། ཀྲྀ་ཥའི་ཥ་དབྱི་བ་་་་ཚིག་ཕྲད་དོ། །ཏཱཾ་ཎཾ། ཙུ། ཨཏིའི་ཨཏ་ཨེང་དང་ར་ཕ་རོལ་ཉིད་དང་ཎ་ཉིད་ཀྱང་ངོ༌། །།ཨཎ། བིཊ་པ། ཛཱིཙྪ། རྒས་པ། དེ་བཞིན་དུ་གཞན་ཡང་་་བལྟ་བར་བྱའོ། །ཀྲྀ། ཤྲཱི། ཙཾ། ཡ་མི། དཱ་རི། ཨརྫི་རྣམས་ལས་ཨུ་་་ནནའོ༑། །རྔུ་ཀྲྀཉ་བྱེད་པ་ལའོ། །བྲོཉ་སྒྲིབ་པ་ལའོ། །ཏཱཾ་རྐྱལ་པ་ལའོ། །ཡ་མུ་ཉེ་བར་འགོག་པ་ལའོ། །ཏཱཾ་རྣམ་པར་འཇོམས་པ་ལའོ། །།སྦྱོར་བ་པོའི་བྱ་བ་ལ་ཎི་ཙའི་ཨཙའི་འོ་ཞེས་པས་ཨཱ་ཉིད། །་ར་ཕ་རོལ། ་ཨརྫ། སརྫ་སྒྲུབ་པ་།
