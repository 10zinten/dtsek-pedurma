བསྟན་འགྱུར། སྒྲ་མདོ། ་རིནི༑་ཨཱ་ཤཱ་ལཱ་ཡ་ན། ་ཨེ་ཌ་ཀྭི་ཡ་ན། ལོན་ཌ་ལ་སོགས་པ་རྣམས་ལས་ཆ་ཀའོ་ཞེས་པས་རྙཀའོ། །ཨཱ་ཡ་ཎ་ལ་སོགས་པའི་མདོས། ཕའི་ཨ་ཡ་ཎ་ར་་་བསྒྱུར་བར་བྱའོ། །།ཀིཏ་ལའང་བུ་རྒྱུད་ལ་སོགས་པ་ལས་ཨ་ཙ་རྣམས་ཀྱི་དང་པོའི་འོ་ཞེས་པས་ཉི་ཏི་ཀི་ཏི་ཙ་སརྦ་ཏྲ་དེ་ར་ཛཱ་དི་ར་ཙ་བྷ་བ་ཏི། ཡཉ་ནི་དཔེར་ན། གརྒའི་བུའི་རྒྱུད་ལ་སོགས་པ་ལ། གཱརྒྱུ། བཏྠའི་བུའི་བུ་རྒྱུད་ལ་སོགས་པ་ལ། བཱཉྩ། །གརྒ་ལ་སོགས་པ་རྣམས་ལས་ཡཉའོ་ཞེས་པས་ཡཉའོ། །ཀ་ནི་དཔེར་ན། འདུལ་བྱེད་མའི་བུ་རྒྱུད་ནི། བི་ན་ཏེ་ཡ། འདབ་བཟང་གི་བུ་རྒྱུད་ནི། སོ་པཎེ་ཡ། གངྒཱའི་བུ་རྒྱུད་ནི། གངྒཱ་ཡ། རྒྱུན་ཤེས་ཀྱི་བུ་རྒྱུད་ནི༑ཨཱ་ཏྲེ་ཡ། བཱཙྪེ་ཡ། ཨོ་ཏོ་ནི་ཉ་ཞེས་ཌྷ་ཀའོ། །།ཤུ་བྷའི་བུ་རྒྱུད་ནི། ཤོ་བྷེ་ཡ། སྐྱེས་བུ་ཁྱབ་འཇུག་གི་བུ་རྒྱུད་ནི། བི་པུ་རུ་ཥ་ཡ། །ཤུ་བྷ་ལ་སོགས་རྣམས་ལ་སོགས་སོ་ཞེས་པས་རྙཀའོ། །གཞོན་ནུ་མའི་བུ་རྒྱུད་ནི། ཐ་ཀོ་མཱ་རེ་ཡ། ཡོ་བ་ཏེ་ཡ། ཀཱ་མཎྜ་ལེ་ཡ། ཌྷཀ་ཅེས་པ་རྗེས་སུ་འཇུག་བཞིན་པ་ལ། དཱི་ཨཱསྟི་ཨུད་ལས་སོ་ཞེས་པས་རྙཀའོ། །ཨེ་རཀ་ནི། བདཔེར་ན། ཙ་ཊ་ཀའི་བུ་རྒྱུད་ནི། ཙཱ་ཊ་ཀི་རཿ། ཙཊ་ལས་ཨེ་རཀའོ་ཞེས་པས་ཨེ་རཀའོ། །ཨི་རི་ཀ་ནི། དཔེར་ན། ས་འཛིན་མའི་བུ་རྒྱུད་ནི། གོ་དྷེ་ར། །གོ་དྷ་ལས་སོ་ཞེས་པ་རྗེས་སུ་འཇུག་བཞིན་པ་ལ། ཡང་ན་ཨེ་རཀ་ཅེས་པ་འདི་ཡིས་སོ། །ཨཱ་རཀ་ནི་དཔེར་ན། ས་འཛིན་མའི་བུ་རྒྱུད་ནི། གོ་དྷ་ར། ཛཱཎྜ་ལས་ཨཱ་་་རཀའོ་ཞེས་པས་རྗེས་སུ་འཇུག་བཞིན་པ་ལ། གོ་དྷ་ལས་སོ་ཞེས་པས་ཨཱ་རགའོ། །གྷ་ནི་དཔེར་ན། ཀླུ་ཏྲ་ཡ་ཡི་བུ་རྒྱུད་ནི། ཀླུ་ཏྲི་ཡ། ཨཱ་ཙིཡན་ལ་སོགས་པའི་མདོས་གྷ་ནི་ཨཱི་ཡ་རུ་བསྒྱུར་བའོ། །ཀླུ་ཏྲ་ལས་རིགས་ལ་གྷའོ། །ཁ་ནི་དཔེར་ན། རིགས་ཀྱི་བུ་རྒྱུད་ནི། ཀུ་ལཱི་ན། ཚིག་མཐའ་ལས་ཀྱང་ཁའོ་ཞེས་པས་ཁའོ། །—།
