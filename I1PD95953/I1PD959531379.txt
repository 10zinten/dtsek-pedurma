བསྟན་འགྱུར། སྒྲ་མདོ། རེནས་ཨུ༑་ཕེན། ་དྷུ་མ་ལས་སྟེང་དུ་འཕྱུར་བ་ལའོ། །།ཀྱང་ཡང་ན་འགྱུར་ཏེ། རླངས་པ་གྱེན་དུ་འཕྱུར་བ་ལ། བཱབྷཱ་ཡ་ཏེ་སྦྲ་ལཱི། ་ཨུཏྟཱ་ཡ་ཏེ་བྷ་ཏྟཾ། ཕེ་ནཱ་ཡ་ཏེ་ས་མུ་དྲ། །དྷུ་མཱ་ཡ་ཏེ་བནྣི། ཀུན་སྤྱོད་ལའང་ཀྱང་:སྟེ། དྷུ་མཱ་ཡ་ཏེ་དྷཱ་ལི་པ་ཊ་ཋལཾ། སུ་ཁ་སོགས་ལས་ཡང་དག་པར་རིག་པ་ལའོ། །ཉེང༌། ཀྱང་ཡང་ན་འགྱུར་ཏེ། བདེ་བ་ཡང་དག་པར་རིག་པ་ལ་སུ་ཁཱ་ཡ་ཏེ་དུ། ཁཱ་ཡ་ཏེ། སུ་ཁི། དུ། ཁ། ཏྲིཔའི། ཀྲྀཙྪ། ཨ་:པྲ་ཨ་ལཱི་ཀེ། ཀྲི་པ་ཎ༑ཀ་རུ་ཎ། སོ་ལྷ། པ་ཏཱི་པ། །མ་དྲ། ཡང་དག་པར་རིག་པ་ནི་མངོན་སུམ་དུ་བྱེད་པ་སྟེ། ཚོར་བ་ཙམ་ནི་མ་ཡིན་ནོ། །དེས་ན་རྗེས་སུ་དཔག་པས་ཚོར་བ་ནི་བདེ་བ་ཡང་དག་པར་རིག་པ་མ་ཡིན་ནོ། །ཆེ། རབ་ཏུ་སྒྲུབ་པ་པོ་་་ལྷས་བྱིན་གྱི་སྟེ། གཞན་གྱི་རྒྱུད་དུ་གཏོགས་པའི་བདེ་བ་ནི་ལྐོག་ཏུ་གྱུར་པའི་་་ཕྱིར་རོ། །ཤརྡ་སོགས་ལས་བྱེད་པ་ལའོ། །ཀྱང་ཡང་ན་འགྱུར་ཏེ། སྒྲ་བྱེད་དོ་ལ་ཤབྷཱ་ཡ་ཏེ། བི་ོརཱ་ཡ་ཏེ། ༽ཀ་ལ་ཧཱ་ཡ་ཏེ། །༽ཨ་བྷིཡ་ཏེ། ཀཎཱ་ཡ་ཏེ། མེ་གྷཱ་ཡ་ཏེ། སུ་དི་ནཱ་ཡ་ཏེ། ཅི་ལྟར་འོ་ན་བེ་ར་ཡ་ཏི་ཞེ་ན། རྣམ་པར་གཞག་པ་རྣམ་པར་བཤད་པ་ལས་ཎི་ཙ་ཡང་བསྟན་པའི་ཕྱིར་རོ། །།ཏ་བས་ལས་ཀྱ་ཨེཙའོ། །བྱེད་པ་ལ་ཡང་འགྱུར་ཏེ། དཀའ་ཐུབ་བྱེད་:དོ། །ལ་པ་ཏ་པ་སྱ་ཏི། ཙི་ཏྲང་ལས་ངོ་མཚར་བ་ལའོ། །འཇུག་པའི་བྱེད་པ་ལ་ཀྱ་ཨིཙ་ཡང་ན་འགྱུར་ཏེ། བཀྲ་བར་བྱེད་པ་ནི། ཙིཏཱི་ཡ་ཏི། ་དེའི་ཕྱིར་འདི་ངོ་མཚར་བ་ནི་གསེར་དང་དངུལ་ལས་བྱས་པའི་རི་དྭགས་ལྟ་བུའོ་ཞེས་ཇོ་བོ་ཟེར་རོ། །ང་ཡིག་ནི་ཏང་གི་དོན་ཏོ། །ངོ་མཚར་ཞེས་པ་ཅི་ཞེ་ན། ཙི་ཏྲ་ཀ་རོ་:ཏཾ། ཧ་ལས་དང་:བ་རི་བས་ལས་མཆོད་པ་དང་ཡངས་སུ་སྤྱོད་པ་དག་ལའོ། །ཀྱ་མི་ཙ་ཡང་ན་འགྱུར་ཏེ། ན་མསྐ་རོ་ཏི། །ན་མ་སྱ་ཏི། ན་མ་སྱ་ཏི་དེ་བཱན། །
