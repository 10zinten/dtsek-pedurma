བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིཨུ་གྲ། དྲག་ཤུལ་ལམ་མི་བཟང་བ། །བྷེ་ར། རྔ་བོ་ཆེ། །གོ་ར་ལ་སོགས་པ་རྣམས་ལས་དྲི་ཥའོ༑། བྷེ་རཱི། རྔ་བོ་ཆེ། །ཨ་གྲོ། མཆོག་། ཚིག་ཕྲད་ལས་ཨོ་ཉིད། །གོ་ར། དཀར་པོ། ཤུ་ཙའི་ཀུ་ཉིད་དང་ཨཏ་ཨེང་མེད་དོ། །ཤུ་ཀྲྀ། ཁུ་བ་ཡང་ན་བཟའ། བྲིདྡྷེ། རྒན་པོ། །ཨརྟིའི་ཛུག་ཨཱ་ག་མའོ། །ར་ཛ། འདྲེན་པ། བི་སྔོན་མའི་པྲ་ཏིའི་ཨཱ་དབྱིའོ། །པི་པྲ། བྲམ་ཟེ། ་ཀུ་པྲ། མདའ། ཙུ་པའི་ན་དབྱིའོ། །ཙུ་བྲོ། དགའ་བར་བྱེད་པ། ཀླུ་ར། སྤུ་གྲི། བུ་རི། རྨིག་པ། །ཨི་རཱ། ས། མ་ཧཱི༑ས། དེ་བཞིན་དུ་གཞན་ཡང་བལྟ་བར་བྱའོ། །ཙ་ཏི། ཀ་ཊི་ཤུ་བྲྀཉ་ལས་ཊ་ར་ཙའོ། །ཙ་ཏེ་སློང་བ་ལའོ། །ཀ་ཊི་སྒྲིབ་པ་ལའོ། །ཨེ་ཡིག་ནི་ཧལ་དང་པོའི་མཐའ་ཉེའི་འོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །༌༌། ::ཤྲཱི་འཚེ་བ་ལའོ། ་། དྲྀཉ་སྒྲིབ་པ་ལའོ། །ཊ་ཡིག་དཱིཔ་ཀྱི་དོན་ཏོ། །ཙ་ཡིག་དབྱངས་དོན། ཙ་ཏཱ་རཾ། ཨོཾ་གྷ་ཎཾ། ཀ་ཏཱ་ར། ཞོའི་རྣམ་འགྱུར། ཨཏ་ཨེང་དང་ར་ཕ་རོལ། ་ཤརྦ་ར། ལྷ་་ཆེན་པོ། ཊིཏ་ཞྭ་ཨན་ཨཉ་ཞེས་པ་ལ་སོགས་པས་དཱིཔའོ། །ཤརྦ་རཱི། མཚན་མོ། པརྦ་རཱི། སྐྲ་ལི་བ། པནྠ། སྒྲའི། རཾ། རྡོ་འགྱུར། པཱི་བ་ར་ལ་སོགས་པ་རྣམས་སོ། ་། པཱ་བཏུང་བ་ལའོ། །ཙི་སོགས་པ་ལའོ། །ཏཱཾ་རྐྱལ་པ་ལའོ། །ཎི་རབ་ཏུ་ཐོབ་པ་ལའོ། །ཎའི་ནའོ། །གཱ་ཧུ་རྣམ་པར་དཀྲུགས་པ་ལའོ། །ཙྪི་་་དིར་རྣམ་པ་གཉིས་སུ་བྱེད་པ་ལའོ། །ཙྪ་ད་སྒྲིབ་པ་ལའོ། །ཊུ་དྷཱཉ་འཛིན་པ་་་ལའོ༑། ཏུ་མི་ཉ་འཕེན་པ་ལའོ། །ཌུ་ཡིག་ཏུ་འགྲོ་བ་ལ་ཀྲིའོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །ཡ་ཏིའི་རབ་ཏུ་འབད་པ་ལའོ། །སམ་སྔོན། འདི་རྣམས་ལ་ཊ་ར་ཙ་ཕ་རོལ་ཏུ་ཚིག་ཕྲད་མཐའོ། །་། དེ་ལ་པཱི་བ་ཏིའི་ཨཱི་ཉིད། །པཱི་བ་རཾ། སྦོམ་པོ། ཙཱ་བ་ར། ཆོས་གོས། ཏ་རི་ཏིའི་ཨཱི་ཉིད་དང༌། ར་ཕ་རོལ་ཉིད་མེད་པ་ཚིག་ཕྲད། ཏཱི་བ་ར། ཀླ་ཀློ། །ཚིག་ཕྲད་ལས་ཨཏ་ཨེང་མེད་དོ། །ནཱི་བ་རཾ། ཁྱིམ། གཱ་་་་།
