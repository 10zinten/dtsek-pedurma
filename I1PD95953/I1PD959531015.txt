།བསྟན་འགྱུར། སྒྲ་མདོ། རིའི་ཊི་ལ་ཉེ་བར་སློང་བ་དང་། རྩོམ་པ་ལ་ཉེ་བར་རྩོམ་པ་དང་། ཁམཆོད་པ་ལ་ཉེ་བར་མཆོད་པ་དང༌། ཟིན་པ་ལ་ཉེ་བར་ཟིན་པ་དང་། ཞི་ཁྲོ། ལྟུང་བ་ལ་ཉེ་བར་ལྷུང་བ་དང༌། །ལགསོད་པ་ལ་ཉེ་བར་གསོད་པ་དང་། དགའ་བ་ལ་ཉེ་བར་དགའ་བའོ༄༅། །གྲོ་། ཉེ་བར་སྒྱུར་བ་ཉི་ཤུ་པའི་འགྲེལ་པ་སློབ་དཔོན་ཆེན་པོ་ཙནྡྲ་གོ་མིས་མཛད་པ་རྫོགས་སོ། །ཀ་། དྲ་བ་ཤ་ལ་ཁ་སྤྱོབ་པ་དཀོ་ཁལ་ཡོད་པ་ཀྱི་སྒྲ་པ་ཆེན་པོ་པཎྜི་ཏ་:ཛི་ཏ་ཀརྞའི་ཞལ་སྔ་ནས་ལེགས་པར་མཉན་ཏེ། ལོ་ཙཱ་བ་ཤཱཀྱའི་དགེ་སློང་ཉི་མ་རྒྱལ་མཚན་དཔལ་བཟང་པོས་དཔལ་ཐར་པ་གླིང་གི་གཙུག་ལག་ཁང་དུ་ལེགས་པར་བསྒྱུར་བའོ། :མ་ལ། །སའི་སྟེང་དུ་ཉི་ཟླ་ལྟར་གྱུར་ཅིག། :སྒྲ་སོགས་ཤེས་བྱའི་ཆོས་ནི་ཕལ་ཆེ་བ། ྋ། སྐྱེ་བ་གཞན་དུ་སྦྱངས་པས་་་འདིར་མཁས་ཀྱང་། །བོད་ཡུལ་བྱིས་པའི་ནང་དུ་བདག་སྐྱེས་པས། །མཁས་་་པའི་མཁས་པ་ཡིན་པར་སུས་ཤེས་ཀྱེ། །དམ་འདོ། །ཡི་ཀ་ག་ངོག་ན་ཡོད་པ་ཀ་ཏོ་དབང་དུ་ਧਣਵਾਲਾਟਜਨਮਾਧਵਪਰਥਬਾਲਕਦਮੀਸੜਸੜདང༌། ཁོ་། དབང་པོག་པད་དཀརྫོའམ་ལྟ་བུར་རང་གྲོལ་ལ། །ཁྲོ་ང་འོད་ལ་གས་བསྡུར་མཆན། ངས་བ་ལམ་ཡོངས་སྤྱིའི་མཚན་བྱ:ང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང༌། «པེ་»«སྣར་ལ་ཉེར་བསྒྱུར་ཉི་ཤུ་པའི་འགྲེལ་པ་ཙནྡྲ་གོ་མིས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་»བི། «པེ་»«སྣར་»སངྒ། «པེ་»«སྣར་»བསྒྱུར། /«པེ་»«སྣར་པྲ་དང་། པ་རཱ་དང་། ཨ་བ་དང་། སམ་དང་། ཨ་ནུ་དང་། ཨ་བ་དང༌། ནིར་་་་དང༌། དུར་དང༌། བི་དང༌། ཨཱདྲ་དང་། ནི་དང་། ཨདྷི་དང་། ཨ་པི་དང༌། ཨ་ཏི་དང༌། སུ་དང་། ཨུཏྲ་དང༌། ཨ་བྷི་དང་། པྲ་ཏི་དང་། པ་རི་དང་། ཨུ་པ་དག་གི། «པེ་»«སྣར་»བསྒྱུར། @«པེ་»«སྣར་པྲ། «པེ་»«སྣར་ལའོ། །པྲ་ཞེས། «པེ་»«སྣར་_པ། «པེ་»«སྣར་གང་། ྋe«པེ་»«སྣར་»པ། «པེ་+པའི། 《སྣར་+ཞི་བའི། «པེ་»«སྣར་»སྦོམས། «པེ་»པ་རཱ། «སྣར་པ་ར། «པེ་»«སྣར་པ་རཱ་ཞེས། «པེ་»«སྣར་བསྒྱུར། ཁྱེ«པེ་»«སྣར་»ཨ་པ། ་་«པེ་»«སྣར་ཨ་པ། «པེ་»«སྣར་»བསྒྱུར། འ«པེ་»«སྣར་»འདི། «པེ་»«སྣར་»ལ། «པེ་»—།
