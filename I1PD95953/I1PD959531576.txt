ཡིན་པ་ལ་རྩེ་བ་ལ་འདི་མི་སྲིད་པའི་ཕྱིར་རོ། །དངོས་པོ་ལ་གྷཉ་ལས། དངོས་པོ་གྷཉའི་མཐའ་དང་པོའི་མཐའ་ལས་འདིའི་ཞེས་པ་བདུན་པའི་དོན་ལ་ཎ་འགྱུར་ཏེ། ཁྲ་ལྷུང་བ་འདི་ལ་ཞུགས་པ་ནི། བྱི་ནཾ་པཱ་ཏཱ། ཏེ་ལཾིཔཱ་ཏཱ། ཀྲྀཏཱ། ཁྲ་གཏིལ་དག་ལ་:ལྷུང་བ་ལ་ཎ་ལས་ཞེས་པས་མུ་མ་ཨཱ་ག་མའོ། །དངོས་པོ་ལ་ཞེས་པ་ཅི་ཞེ་ན། འདི་ཡིད་དང་བ་ནི་:པྲ་སཱ་དང་། ་དེ་འདིའི་་་གྲོང་ཁྱེར་རོ། །གྷཉ་ལས་ཞེས་པ་ཅི་ཞེ་ན། བྱེ་ན་པ་ཏན་མ་སྶཾ་བཏྟ་:རེ། བརྩེ་བ་ལ་ཞེས་པ་ལྡོག་པ་ཐུན་མོང་བ་ཉིད་ཀྱི་སྒྲུབ་པའོ། །དེས་ན་དཎྜ་པཱ་ཏ། ཨཱ་ཏི་ཐིམོཥ་ལ་བཱ་ཏཱ་མིཞེས་པ་ལ་སོགས་པ་འགྲུབ་བོ། །གང་རིག་བྱེད་ཀློག་པ་ལ་ཡང་ནའོ། ཋ། དེ་ལས་གཉིས་པའི་མཐའ་ལས་ཨཎ་འགྱུར་ཏེ། སྡེབ་སྦྱོར་་་རམ་རིག་བྱེད་ཀློག་པ་ནི། བི་ད། ཙྪིནྡ་ས། བི་ཡཱ་ཀ་:ཎ། ཚོ་ཏི་ཥ:»། ཀ༌ནི་མིནཏྟཉྩི། །ཡང་ན་སྨོས་པ་ནི་འདི་དག་སོ་སོར་སྒྲུབ་པའི་དོན་དུ་རབ་ཏུ་སྒྲུབ་པ་ཉེ་བར་གཟུང་བའི་དོན་དུ་ཡང་ངོ༌། །ཀྲ་:ཏུ་ཨཀྵ་སོགས་རྣམས་ལས་ཋཀའོ། །ཨུཏྤ་སོགས་རྣམས་ལས་མཆོད་སྦྱིན་བརྗོད་པ་དང་རིག་བྱེད་ཀློག་པའི་དོན་ལ་ཋཀ་འགྱུར་ཏེ། ཨཎ་གྱི་དམིགས་བསལ་ལོ། མཆོད་སྦྱིན་ལས་ནི། མེའི་མཆོད་སྦྱིན་ནམ། རིག་བྱེད་པ་བཀླགས་པ་ནི། ཨཱགཱིཏྠོམི་ཀ། བཱ་ཛཱ་པ་ཡི་ཀ། དེའི་ཆེད་དུ་དེ་མངོན་པར་བརྗོད་པའི་གཞུང་ལ་དེའི་སྒྲས་བརྗོད་དེཚེ། གཞན་དུ་རིག་བྱེད་ཀློག་པ་དག་དང་མི་འགྱུར་ཞེ་ན། ཡང་ན་སྨོས་པས་རབ་ཏུ་གྲུབ་པ་ཉེ་བར་གཟུང་བའི་དོན་གྱི་ཕྱིར་རོ། །དེ་བཞིན་དུ་ཡཱ་ཛཱིཀ། མ་དྷཱི་ཏི༄༅། །ཨུ་ཀྵ་ལོ་ཀཱ་ཡ་ཏ། །རྩྭ་ཡ་:རྩྭ་ས་པུ་ན་རུཀྟ། ཡཛྷ། ཙརྩྭ་༈ཀྲ་མེ་ཏ་ར། ཤཱཀྵ། །སཾ་ཧི་ཏཱ། པ་ད་ཀྲ་མ་སཾ་གྷ་ཊ། བྲྀཏྟི''སཾ་གྲ་ཧ། ག་ཎ་ཨཱ་ཡུརྦད། དྲི་པ་དཱ། ཚོ་ཏི་ཥ་ཨ་ནུ་བ་ད། །ཨ་ནུ་:སཱུ་ཨཱ་ལི་།
