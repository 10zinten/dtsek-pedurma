བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིབ་ཙ་ནཱི་ཡོ། བཙུ། ཨུཀྟ་བཱ། དྷརྨ་ཤིཊྛོ་གུ་རུ་ཎཱ། ཤཱ་སི་ཏ་བྱསོ། ཤཱ་ས་ནི་ཡ། ཤིཊྚ་བཱ་དྷརྨོ་མཱ་ན་བ་ཀ་མུ་པཱ་དྷཱ་ཡེ་ན། ཨུ་པཱ་དྷ་ཡ་སྐྱེ་བཱ། །བོ་དྷ་ཡི་ཏ་བྱ། བོ་དྷ་བྱ། །བོ་དྷ་ནཱ་ཡོ། བོ་དྷྱ། བོ་དྷི་ཏོ་བཱ། མཱ་ན་བ་ཀོ། དྷརྨ་མུ་པཱ་དྷཱ་ཡེ་ན། བོ་དྷ་ཡི་ཏ་བྱོ། བོ་དྷི་ནི་ཡོ། བོ་དྷོ། བོ་དྷི་ཏོ། །མཱ་ཎ་པ་ཀནྡྷརྨ། ཨུ་པཱ་དྷ་ཡ་ན། བ་ད་ནི་ཡོ། བེ་དི་ཏ། ཤ་སྤྲོ་བུ་ཏྲ་པི་ཏཱ་་བེ་དི་ཡ་ཏ་བྱ། བེ་ད་ནཱི་ཡཾ། བེ་ཧུཾ་བེ་དི་ཏཾ། ཤ་ཙྰ་པུ་ཏྲ་མྲི་ཏཱ་ཛྙཱ་པ་ཡི་་ཏ་བྱོ་ཙཱ་པ་ནི་ཡོ་ཙཱ་པཱུ་ཛྙཱ་པི་ཏ། ཤ་ཧཱུཾ་པུ་ཏྲ་པུ་ཏཱ། ་ཙཱ་པ་ཡ་ཏ་བྱོ། ཙཱ་པ་ནི་ཡོ། ཙཱ་པ་ཏོ་བཱ་ཤཱ་སྟགླུ་ཏྲི་པི་ཏཱ། བྷོ་ཛུ་ཡི་ཏ་བྱོ། བྷོ་ཛ་ནཱི་ཡོ། །བྷོ་ཕྱོ། །བྷོ་ཛི་ཏཱ། ་བྲཱཧྥ་ཎ་བྷོ་ཛ་ཡི་ཏ་བྱོ། བྷོ་ཛ་ནི་ཡོ། །བྷོ་ཕྱོ། ༽བྷོ་ཛཾ་ཏོ། བྲཱཧྥ་ཎཱ་ནོ་ད་ནཛྲ་ཧསྦྲེན། ཨཱ་སི་ཡ་ཏ་བྱོ། ཨཱ་སྱ། ཨཱ་སི་ཏཱ། བྲཱཧྥ་ཎཱ་ནོ་ད་ནདྡྷཾ། ཧསྦེ་ན། ཨཱ་ས་ཡི་ཏཱ་བྱ། ཨཱ་ས་ནཱི་ཡི། ཨ་སྱ། ཨཱ་སི་ཏཱ། བྲཱཧྨ་ཎཱ་ནོ་ད་ནོ་དེ་བ་དཏྟེ་ན། པཱི་ཥཱ་ཡ་ཏི་བྱ། པཱ་ཋ་ནཱི་ཡ་པཱ་ཉྩ། པཱ་ཋི་ཏོ། མཱ་ན་བ་ཀོ་བེ་ད་མུ་པཱ་དྷ་ཡེ་ན། པཱ་ཋ་ཡི་ཏ་བྱ། པཱ་ཋ་ནཱི་ཡ། པཱ་ཉྩ། པཱ་ཋི་ཏོ། མཱ་ན་པ་ཀཾ་བེ་ད་ཨུ་པཱ་དྷཱ་ཡེ་ན། ཨ་དྷཱ་པ་ཡི་ཏ་བྱོན། དྷཱ་པ་ཏིཤི་ཤཱ་སྔ་མུ་་བ་དྷཱ་ཡེ་ན། ་ཨ་དྷི་པི་ཏ་བྱ་མ་རྒྱ་ནི་ཡ་མ་དྷ་པ་མ་དྷཱཔ་ཏི། ཎི། ཤཱ་ཥ་ཡི་ཏ་བྱཤཱ་་ས་ནཱི་ཡཤཱ། སི་ཥ་ཏ། པ་ཏི་མཱ་སམྦྷནཱ་སཱ་ཥ་ཡི་ཏ་བྱ། །སཱ་ས་ནཱི་ཡ། སཱ་ས། སཱ་ཥི་ཏ། ས་ཏི་མཱ་སོ་བྷཱན། ཡོ། བྷརྨ་ཡཱ་བཱ། ཨཱ་ས་ཡི་ཏ་བྱ༑ཨཱ་ས་ཡི་ནཱ་ཡ། ཨ་སྱ། ཨ་སི་ཏི། པ་ཏི་མཱ་སམྦྷནཱ། ཨཱ་ས་ཡི་ཏ་བྱ། ཨཱ་སི་ནཱ་ཡ། ཨཱ་སྱ། ཨཱ་སི་ཏ། པ་ཏིམྨཱ་སོ་བྷཱནཱ་ཡཱ་ནཱ་ཡི་ཏ་བྱོ། ནཱ་ཡ་ནཱི་ཡོ་ནཱ་སྤྱོ། ནཱ་ཡི་ཏོ་གྲ་མནྡེ་བ་དཏྟེ་ན། —ཡ་ཛྷ་དཏྟེ་ན། ཨ་ད་ཡི་ཏ་བྱཨད་ནི་ཡ་ཨཱ་ཏྱ། ཨ་དི་ཏི། ཨོ་ནམཱ་ལ་བིཏྟཱ''མི་ཏྲི་ཎ། བཱ་ཧ་ཡི་ཏ་བྱོ། བཱ་།
