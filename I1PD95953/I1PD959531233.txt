བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིཤཱསྔ་སྨུ་ཏྲ་པི་ཏཱ། བེ་རྱ་ཏེ་ཤཱསྡུ་སྨུ་ཏྲ་པི་ཏཱ། བྷོ་ཛ་ཏི་བྲཱཧྥ་ཎ་ནོ་ད་ནི་གྲོ་ཧ་སྦྱ། བྷོ་ཙུནྟེ་བཱཧྥ་ཎ'ིན་ཨོ་ད་ནི་གྲི་ཧ་སྦེ་ན། བྷོ་ཛ་ཏེ་བྲཱཧྥ་ཎ་ཨོ་ད་ནོ་གྲོ་ཧ་སྦེ་ན། པཱ་ཐ་ཡ་ཏི་མཱ་ན་བྷ་ཀཾ་བེ་བྱ་མུ་པཱ་དྷཱ་ཡ། །པཱ་ཉྩ་ཏེ་མཱ་ན་བ་ཀོ་བེ་ད་མུ་པཱ་དྷཱ་ཡེ་ན། པཱ་ཉྩ་ཏེ་མཱ་ན་བ་ཀམྨེ་ད་ཨུ་པཱ་དྷ་ཡེ་ན། ཨཱ་ཤ་ཡ་ཏེ་མཱིསནྡེ་བ་དཏྟཾ་བི་མི་ཏྲ། ཨཱ་སྱ་ཏེ་མཱ་སོ་:དེ་བ་དཏྟཾཡཛྙཱ་དཏྟེ་ན། ཨཱ་སྱ་ཏེ་མཱ་སཾ་དེ་བ་དཏྟོ་བིབྷཱུ་མི་ཏྲེ་ཎ། སཱ་ཡ་ཡ་ཏི་པ་ཏི་པ་ཏི་མ་ཡཾ་བྷཱརྨ། སཱ་ཡུ་ཏེ་བ་ཏིརྨ་སཾ་བྷཱརྨ། སཱ་ཡུ་ཏེ་པི་ཏཾ་མཱ་སོ་བྷཱཎཱ། རྣམས་ཏེ་སྦྱོར་བ་པོའི་བྱ་བ་ལ་ཎའོ། །བྱེད་པ་དང་ལས་དང་དངོས་པོ་དག་ལ་དངོས་པོ་དང་ལས་དག་ལའོ་ཞེས་པས་ཏངའོ། །ན་ཡ་ཏེ་དེ་བ་དཏྟོ་གྲྭ་མངྒནཱ་པ་ཡི་ཏ་དེ་བ་དཏྟ་ན་སཾ་གྲྭ་མ་ཡཛྙཱ་དཏྟ། ཁཱ་ད་ཡ་ཏི་ཨ་སྭ་བྷཱ་སཾ། ཁཱ་ད་ཡ་ཏི་ཨ་སྡེ་ན་མཱ་སཾ་དེ་བ་དཏྟཨིནྟེ་དེ་བ་དཏྟཔཱ་ཡ་སཾ། བཨཱད་ཡ་ཏེ་དེ་བ་་་དཏྟ་ན་པཱ་ཡ་སཾ་ཡཛྙཱ་དཏྟ། རྣམས་ཏེ། འགྲོ་བ་དང་རྟོགས་པ་དང་ཟས་དང་སྒྲའི་དོན་དང་ལས་མེད་པ་རྣམས་ཀྱིས་སྦྱར་བྱ་ལའོ་ཞེས་པས་གཉིས་པ་ཐོབ་ལ། ནཱ་ཁཱ་རྱ་དི་ཧྭ་ཤཔ་ཡ་ཀྲནྡའི་མ་ཡིན་ནོ་ཞེས་པའི་དགག་པ་ལའོ། །མཐར་ནི་ཎིའི་མཐའ་ལས་གཡོ་བ་དང་ཟས་ཀྱི་དོན་ལས་སོ་ཞེས་པས་ཏིང་དང་ཨཱ་ན་དག་དགག་་པ་ཐོབ་པ་ལ། འབྲས་བུ་ལྡན་པ་དང་བྱེད་པ་པོ་ལ། ན་པཱ་ད་མ་ཡ་མཱ་ཡས་པ་རི་མུ་ཧཱ་ཏི་རུ་ཙ་ཏྱནྟི་དྷེ༔་ད་པ་ས་ཞེས་པའི་དགག་པ་ལས་འབྲས་བུ་ལྡན་པ་ལ་་་་༌་ཞེས་པ་ཏངའོ། །བ་ཧནྟི་བྷཱ་རཾ་བི་ལཱི་བནྡྷབ་ཧ་ཡ་ཏི་བ་ལཱི་བདྷེནྡྷ་རཾ་དེ་བ་དཏྟསྦྱར་བྱའི་བྱེད་པ་པོ་ལ། བ་ཧ་ར་ནི་ཡནྟ་ཀི་ཞེས་པས་གཉིས་པ་དགག་པའོ། །ཨ་ནི་ཡནཱི་ཀའི་ཞེས་པས་གཉིས་པ་དགག་པའོ། །ཨ་ནི་ཡནྟ་ཀའི་ཞེས་པ་ཅི་ཞེ་ན༑བ་ཧནྟི་བ་ལི་བནྡྷ་བྷཱ་རཾ། བཱིཧ་ཡནྟི་བ་ལི་བནྡྷ་བྷཱ་རཾ། ར་ཐི་ཀབྷཀྵ་ཡ་ཏི་པཎྜི་དེ་བ་དཏྟ། དེ་ལས་ཙུར་སོགས་རྣམས་ལས་ཎི་ཙའོ། །གང་ཡང་་་།
