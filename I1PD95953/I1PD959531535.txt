བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིགྲགས་ཀྱི་མཐར་མ་གུས་པས་རབ་ཏུ་འགྲོ་ཞེས་པའི་དོན་ཏོ། །སཱ་དྷུ་དང་ནི་པུ་ཎ་དག་མཆོད་པ་ལའོ། །@སཱ་དྷ་རྨ་ཏ་རི། པི་ཏ་རི་ནི་པུ་ཎ། ཨ་སཱ་དྷུ་མཱ་ཏ་རི། བྷཏྟ། པི་ཏ་རི། ཀུ་ཊུལླེ་ཥཱ་བཏྠ་ལ་ཞེས་པ་ཡུལ་ལ་བདུན་པའོ། །དེ་བཞིན་དུ་ཏ་ཐཱ་ཀེ་ཤེ་ཥཱ་པྲ་སི་ཏི། ་རབ་ཏུ་ཆགས་ཞེས་པའི་དོན་ཏོ། །ཀེ་ཤི༑པྲ་བི་ཏ་ཞེས་པ་བྱེད་པ་ལ་གསུམ་པའོ། །ཡང་ན་བྱེད་པ་པོ་ལའོ། །བྱ་བ་ཞེས་པ་འདིའི་ཚིག་རྟོགས་པར་བྱེད་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་ཀེ་ཤེ་ཥཱཉྩི་ཀ༑ཀེ་ཤི་ཏུཏྨ་ཀ། པུཥྱ་ཎ་པཱ་ཡ་ས་ཧི་ཡཱ་ཏ། པུ་སྐྱེ་ཎ་པཱ་ཡ་ས་མ་བྷི་ཡ་ཏ། བྱེད་པའི་དབང་དུ་བྱས་པ་འཆད་པར་འདོད་དོ། །ཨཱ་ཡུཀྟ་དང་ཀུ་ཤ་ལ་ཡང་ཀུན་དུ་བསྟེན་པ་ལའོ། ་། དྲུག་པ་ཞེས་པ་རྗེས་སུ་འཇུག་གོ། །ཨཱ་ཡུཀྟ་དང་ཀུ་ཤ་ལ་དག་སྦྱོར་བ་ལ་ཀུན་དུ་བསྟེན་པ་རྟོགས་པར་བྱེད་པ་ལ་ང་ཨོས་སུཔ་ཏིའགྱུར་ཏེ། དྲུག་པའང་ངོ༌། །ཨཱ་ཡུཀྟ། ཀ་ཊ་ཀ་ར་ཎེ། པཨཱ་ཡུཀྟ། ཀ་ཊ་ཀ་ར་ཎ་སྱ། ཀུ་ཤ་ལ་ཀ་ཊ། ཀ་ར་ཎ་སྱེ་ཨཱ་སེ་ཅིབ་ཏཱཏ་པརྨི། ལྔ་པ་དང་བདུན་པ་དག་ལ་བྱེད་པ་པོ་བར་མ་དང་དུས་དང་ལམ་དག་གོ། །བྱེད་པ་པོ་བར་མ་དང་ལམ་དག་ལ་བདུན་པ་དང་ལྔ་པ་དག་འགྱུར་ཏེ། ཨ་བྱ་བྷུཀྟ་དེ་བ་དཏྟོ་དུ་ཧེ་བྷོཏྟཱ། །དུ་ཧཱ་བྷོ་ཏྟཱིཀཱ། ཨི་ཧསྦོ་ཡ་མི་ཥ་ས། ཀྲོ་ཤེ་ལཀྵ་བི་དྷུ་ཏི། ཀྲོ་ཤཱཝཀྐཾ་བི་དྷུ་ཏི། ནུས་པ་དང་བྱེད་པ་པོ་རྫས་མ་ཡིན་ཞེས་པའི་ནུས་་པའི་བྱེ་བྲག་ལས་དུས་དང་ལམ་དག་གི་བར་མ་ཉིད་རྟོགས་པར་བྱེད་དོ། །ཨཏ་་ལས་བྷིས་ཀྱི་ཨིསའོ། །ཨ་ཡིག་མཐའ་ལས་ཨངྒའི་ཕྱི་མའི་བྷིས་ནི་ཨེས་སུ་བསྒྱུར་བར་འགྱུར་ཏེ། ཤྲཱི་གླེ། །ཨཏ་ལས་ཞེས་པ་ཅི་ཞེ་ན། ཌཨགྣི་བྷི། ཏ་ཕ་་་རོལ་དུ་བྱེད་པ་ཅི་ཞེ་ན། མཱ་ལཱ་བྷི། ཅི་ལྟར་ཨ་ཏི་ཛ་ར་སིཞེ་ན། ཚིག་ཕྲད་ཀྱི་མཚན་ཉིད་ཀྱི་སྒྲུབ་པ་རྒྱུ་མཚན་མེད་པས་དེ་རྣམ་པར་བཅོམ་ཞེས་པ་འདི་མི་་—།
