བསྟན་འགྱུར། །སྒྲ་མདོ། ༼རི་ཨ་ཛ། བ་ཛ། ཨོཾ་བྲ་ཛ་འགྲོ་བ་ལའོ། ༈། ཤོ་དྲི་སྐེམས་པ་ལའོ། །ཡོ་ཛྲ་འབྲེལ་པ་ལའོ། །མེ་དྲིལ། སླེ་དྲི་སྨྱོ་བར་བྱེད་པ་ལའོ། z་། ཀ་ཊེ་སྒྲིབ་པ་ལའོ། །ར་ཊ་ཡོངས་སུ་བཤད་པ་ལའོ། །ལ་ཊ་བྱིས་པ་ཉིད་ལའང་ངོ༌། །ཤ་དྲིའཚེ་བ་ལའོ༑། བ་ཊ་བསྐོར་བ་ལའོ། །།ཁི་ཊ་སྐྲག་པ་ལའོ། །ཁི་ཊ། ཏི་ཊ་མ་གུས་པ་ལའོ། །ཛ་ཊ། ཛྷ་ཊ། པི་ཊ་འདུས་པ་ལའོ། ་། བྷ་ཊ་གླ་ལའོ། །ཏི་ཊ་བསྒྲེང་བ་ལའོ། །ཁ་ཊ་འདོད་པ་ལའོ། །ན་ཊ་གར་ལའོ། །ཧ་ཋ་འབར་བ་་་ལའོ༑། །ཥ་ཊ་ཆ་ཤས་ལའོ། །ལུ་ཊ་དཀྲུགས་པ་ལའོ། །།བི་ཊ་སྤྲིན་སྐུལ་ལའོ། །བི་ཊ་སྨོད་པ་ལའོ། །།ཨེ་ཋ། ཧེ་ཋ་རྣམ་པར་གནོད་པ་ལའོ། »། ཨ་ཊེ། ཨི་ཊ། བ་ཊ། ཀི་ཊ། ཀ་ཊཱི། ཨི་འགྲོ་བ་ལའོ། །མ་ཌི་རྒྱན་ལའོ། »། ཀུ་ཌི་མ་ཚང་བ་ལའོ། །ཡུ་ཌི་རབ་ཏུ་ཉེད་པ་ལའོ། །ཙུ་ཌི་ཆུང་ངུར་གྱུར་པ་ལའོ། །མུ་ཌི་དུམ་བུར་བྱེད་པ་ལའོ། །བ་ཊི་རྣམ་པར་འགོད་པ་ལའོ། །།རུ་ཊི། ལུ་ཋི་རྐུན་མོ་ལའོ། །སྤ་ཊ། སྤུ་ཊ་འཚེ་བ་ལའོ། །པ་ཋ་བརྗོད་པ་ལའོ། །བ་ཋ་སྦོམ་པོ་ལའོ། །མ་ཋ་ངེས་པར་གནས་པ་ལའོ། །ཀ་ཋ་ཚིགས་ཀྱིས་འཚོ་བ་ལའོ༑། ཧ་ཋ་དབང་ཆེ་བྱེད་པ་ལའོ། །རུ་ཋ། ལུ་ཋ་ཉེ་བར་འཇོམས་པ་ལའོ། །བི་ཥ་འཚེ་བ་ལའོ། །ཤ་ཥ་འཚེ་བ་ལ་ཡང་ངོ༌། །ཀུ་ཋི། ཤུ་ཋི་སྐེམས་པ་ལའོ། །ལུ་ཟི་ལེ་ལོ་ལའོ། །རུ་ཟི་འགྲོ་བ་ལའོ། །ཙུདྨ་སྒེག་པའི་རྣམ་འགྱུར་ལའོ། །ཨཏྤ་མངོན་པར་སྦྱོར་བ་ལའོ། །ཏཱཾ་ཏཱཾ་རྩེ་བ་ལའོ། །རུ་ཊཾ་ཨེཐུ་བ་ལའོ། །ཧུ་ཊེ༄༅། །ཧོ་ཊཱའགྲོ་བ་ལའོ། །རོ་༉*མ་གུས་པ་ལའོ། །ལོ་ཌི་ྋསྨྱོ་བར་བྱེད་པ་ལའོ། །ཨ་ཌ་རྩོམ་པ་ལའོ། །ལ་ཌ་རོལ་པ་ལའོ། །ཀ་ཌ་རྒྱགས་པ་ལའོ། །།ཀདྨ་སྲ་ཅིབ་ལའོ། །ག་ཊི་གདོང་གི་ཕྱོགས་གཅིག་ལའོ། །གུ་པུ་སྲུང་བ་ལའོ། །བྷུ་བ་ཀུན་དུ་གདུང་བ་ལའོ། །ར་པ། ལ་ས། ཛ་པ། འོཛམྦ་བརྗོད་པ་ལའོ། །—།
