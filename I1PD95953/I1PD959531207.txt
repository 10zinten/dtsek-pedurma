བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིནི། དཔེར་ན། ཁ་བ་བཟོད་པར་བྱེད་པ་ནི་ཧི་མེ་ལུའོ། །ཧི་མ་ལ་བཟོད་པ་ལ་ཙེ་ལུའོ་ཞེས་པས་ཙེ་ལུའོ། །ཙུ་ལ་ནི། དཔེར་ན། སྟོབས་བཟོད་པར་བྱེད་པ་ལ། བ་ལུ་ལཱ། རླུང་བཟོད་པར་བྱེད་པ་ལ། བཱ་ཏུ་ལཱ། བ་ལ་དང་བཱ་ཏ་ལ་ཙུ་ལའོ་ཞེས་པས་རྩྭ་ལའོ། །ཨཎ་ནི། དཔེར་ན། གནས་འཇོག་གི་བུ་རྒྱུད་ནི། བཱ་ཤཱིཥྛ། ཉི་མའི་བུ་རྒྱུད་ནི། ཨཱངྒི་རི་ས། ：ངན་སྤོང་གི་བུ་རྒྱུད་ནི། བྷཱཏྨ་བ། །ཨནྡྷ་ཀའི་བུ་རྒྱུད་ནི། །ཨནྡྷ་ཀ་ལ། བྲྀའི་བུ་རྒྱུད་ནི། བཱཙྪ་བ། །སྒྲ་ངན་པའི་བུ་རྒྱུད་ནི། ཀོ་ར་བ། རིགས་མེད་ཀྱི་བུ་རྒྱུད་ནི། ཌ་ནཱ་ཀུ་ལ། །ནོར་ལྟའི་བུ་རྒྱུད་ནི། བཱ་སུ་དེ་བ། །སྭ་ཕལྟའི་བུ་རྒྱུད་ནི། །སྭ་ཕ། །དྲང་སྲོང་ཀུ་རུ་བྲྀ། ཨནྡྷ་ཀ་ལས་སོ་ཞེས་པས་ཨཎའོ། །ཨུ་པ་གུའི་བུ་རྒྱུད་ནི། ཨོ་པ་ག་ཝ། ཀུ་པ་ཊུའི་བུ་རྒྱུད་ནི། ཀཱ་བ་ཊ་བ། ཛི་ཏ་ལས་གོང་དུ་ཨཎའོ་ཞེས་པ་དེའི་བུ་རྒྱུད་ལའོ། །ཨུའི་ལོཏ་ཞེས་པས། ཆ་ཨུ་ཡིག་གི་ཨོ་ཉིད་དོ། །ཨེ་ཙའི་ཨ་ཡ་ཨཱཝ་ཨཱ་ཡ་ཨཱཝ་རྣམས་སོ། །ཀིཏ་ལའང་བུ་་རྒྱུད་ལ་དང་པོའི་ཨ་ཙའི་ཨཱཏའོ་ཞེས་པས་ཨཱཏ་ཨིཙའོ། །།རྟ་བདག་གི་བུ་རྒྱུད་ནི། ཨཱ་ཤཱ་བ་ཏ། ཕྱུག་བདག་གི་བུ་རྒྱུད་ནི། པཱ་ཤུ་པ་ཏ། པ་ཏི་ལས་ཨ་ཤྭ་ལ་སོགས་པ་མ་ཡིན་པ་དང་པོ་ལས་སོ་ཞེས་པས་ཨ་ཤུ་ལ་སོགས་པའི་བདག་པོ་ལ་ཨུ་དཱ། ཤངྐ་ར། སོ་ར། །འདིའི་ལྷ་ཁྱབ་འཇུག་ཡིན་པ་ནི། བིཥྛ་བཨེནྡྲ། དེ་བ་ལས་སོ་ཞེས་པས་ཨཎའོ། །འདི་སངས་རྒྱས་ལ་གུས་པས། བོདྡྷ། །ཞེས་ལ་སོགས་པ་རྣམས་ལ། དེ་ལ་གུས་པས་ཞེས་འདིའི་དོན་གྱི་ཁྱད་པར་ཡིན་པས། ཛི་ཏ་ལས་གོང་དུ་ཨཎའོ། །དུར་སྨྲིག་གིས་བཙོས་པའི་གོས་ནི། །་ཀཱ་ཥཱ་ཡཾ། གུར་གུམ་གྱིས་བཙོས་པའི་གོས་ནི། ཀོརྐུ་མ། དེ་བཞིན་དུ་ལེ་བརྒན་གྱིས་བཙོས་པ་ནི། ཀོ་སུ་དྡྷཾ། ཡུང་བས་བཙོས་པའི་གོས་ནི། ཧཱ་རི་དྲི། དེ་།
