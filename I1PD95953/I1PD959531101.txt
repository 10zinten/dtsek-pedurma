བསྟན་འགྱུར། སྒྲ་མདོ། རིཞེས་པ་འདིའི་སྔ་མ་བཞིན་ཏེ། ཨ་ནྱ། ཨ་དྷོ། ཨ་མྱེ། ཨ་ཧཱུཾ། ཨ་སྐྱོ། ཨ་ནཱན། ཨ་མྱེ་ན། ཨ་ནྡྷ་ཧཱུཾ། ཨ་ནྱི། ཨ་ནྱ་སྨི། །ཨ་ནཱ་ཙིཏྟཾ། ཨ་རྩེ་༑ཨ་ནྱ་སྨཏ། ཨ་ནྱ་ཧཱུཾ། ཨ་ནྱ་། ཨ་རྒྱ་སྱ། ཨ་ནྱ་ཡོ། ཨ་མྱེ་ཥཱམ༑ཨ་ནྱ་སྨིན། ་ཨ་རྩེ་ཡོ། 《ཨ་ནྱ་ཥཱ། བུད་མེད་ལ། ཨ་ཅུ། ཨ་མྱེ། ཨ་ཅཱ། ཨ་ཧཱུཾ། ཨ་མྱེ། ཨ་དྷཱ། ཨ་ནྱ་ཡཱ། ཨ་རྩྭ་བྷཱ། ཨ་ནྟཱ་བྷི༄༅། །ཨ་ནྱ་སི། ཨཱ་ན་༔། །ཨ་རྩྭ་། །:ཨཱ་ནཱུ་པཿ། ཨ་རྩྭ་ཧཱུཾ། :ཨ་ནྱ་p། ཨ་རྒྱ་སྱ། །ཨ་རྩེ་ཡོ། ཨ་ནྱ་ཙེསཾ། ཨ་ནྱ་སིཾ?། །ཨ་ནྱ་ཡོ༑ཨ་ཅུ་སུ། མ་ནིང་ལ་ཌ་ཏ་ར་ལ་སོགས་པ་:སྔ་མ་ལྔ་རྣམས་ལས་ཨེ་ཀ་ཏ་ར་མ་ཡིན་པ་ལས་ཏའོ་ཞེས་པ་སུ་ཨམ་དག་གི་ཏ་ཡིག་གོ། །ཨ་:ནྱཏ། ཨ་རྩེ། ཨ་རྩྭ་ནི། ཨ་ནྱཏ། ཨ་ནྱ། །༼ཨ་ཅུ་ནི། ཊ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཨཀཙ༑ལ་ཨ་ནྱ་ཀ། ཨ་ནྱ་ཀོ། ཨ་ནྱ་ཀེ། བུད་མེད་ལ། ཨ་རྩ་ཀཱ། ཨ་ནྱ་ཀེ། ཨ་ནྱ་ཀཱ། །མ་ནིང་ལ། ཨ་ནྱ་ཀཾ། ཨ་ནྱ་ཀེ། ཨ་ནྱ་ཀཱ་ནི། །འདི་ལྟར། ཨ་རྱ་ཏ་རེ། ༼ཨ་ནྱ་ཏ་རོ། ཨ་ནྱ་ཏ་རེ། བུད་མེད་ལ། ཨ་ནྱ་ཏ་རཱ། ཨ་ནྱ་ཏ་རེ། ཨ་ནྱ་ཏ་རཱ། །མ་ནིང་ལ། 《ཨ་ནྱ་ཏ་རཏྲ། 《ཨ་རྱ་ཏ་རེ༑ཨ་ནྱ་ཏ་རཱ་ཎི༄༅། ་ཨཀཙ། ཨ་ནྱ་ཏ་ར་ཀཿ། ཨ་ནྱ་ཏ་ར་ཀོ། ཨ་ནྱ་ཏ་རེ་ཀེ། བུད་མེད་ལ། ཨ་ནྱ་ཏ་ར་ཀཱ། ཨ་ནྱ་ཏ་ར་ཀེ། ཀླུཨ་ནྱ་ཏ་ར་ཀཱ། མ་ནིང་ལ། ཨ་ནྱ་ཏ་ར་ཀཾ། ་ཨ་ནྱ་ཏ་ར་ཀེ། །ཨ་ནྱ་ཏ་ར་ཀཱ་ཎི། ཏཱ་ཞེས་པ་འདིའི་སརླ་བཞིན་ཏེ། ཏཱ། ཏཱཾ། །ཏེ། བུད་མེད་ལ། ཏཱ། ཏྲེ། ཏྭཱི༑། མ་ནིང་ལ། །ཏྭཾ། །ཏྭཾ། །ཏྭཱ་ནི། ཏུད་ནི་དྲི་ཤ་དང་འདྲ་སྟེ། ཏུད༑ཏོ་ཨེདོ། ཏཱ་ད། ཌ་ནེ་མ་ཞེས་པ་འདིའི་སརྦ་བཞིན་ཏེ། ནེ་མ། ནེ་མོ། ནེ་མེ། བུད་མེད་ལ། ནེ་མཱ། ནེ་མེ། །ནེ་མཱ། མ་ནིང་ལ། །
