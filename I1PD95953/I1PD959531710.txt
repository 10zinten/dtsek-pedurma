མ་དྷཱ་ར། ་ཨཱ་མ། ཤཱི་ཏ། ཨུཥྛ། ཛ་ཌ། །བ་དྷི་ར། །མཱུ་:ཀ། མཱུཎ། པཎྜི་ཏ། །བི་ལས་ཡཱ་ཏ་ལཱ་བྷ་ཤཱ་ར་ད་ལས་སོ། །དེའི་དངོས་པོ་ཞེས་པ་འདིའི་་དོན་ལ་ཉ་འགྱུར་ཞིང་ཨི་མཎ་ཡང་ངོ༌། །འདི་འགལ་བར་:འགྱུར་བ་ྋརྟོག་པ་ནི། བཱི་ཡཱ་ཏ། །དེའི་དངོས་པོ་ནི། བེ་ཡཱ་ཏྱཾ། བི་ཡཱ་ཏི་མཱ། བི་ལཱ་བྷེ། བི་ལཱ་བྷི་མཱི། ་བི་ཤྭ་ར་:ཏྱཾ། བི་ཤཱ་ར་དི་མཱ། སམ་ལས་ཀྱང་མ་ཏི་མ་ནས་་ལས་སོ། །སཾ་དང་བི་ལས་ཀྱང་ཕ་རོལ་གྱི་མ་ཏི་དང་མ་ནས་ཀྱི་སྒྲ་དག་ལ་དེའི་དངོས་པོ་ཞེས་པ་འདིའི་དོན་ཉ་འགྱུར་ཞིང་ཨི་མན་ཡང་ངོ༌། །ཡང་དག་པའི་་བློ་གྲོས་ཀྱི་དངོས་པོ་ནི། སཱམྨ་ཏུཾ། །བི་མ་ཏཾ། སམྨ་ཏི་མཱ། བི་མ་ཏི་མཱ:：སྭཱ་མ་ན་སྤྱིི། :སཾ་མ་ནི་མཱ། བི་མ་ནི་མཱ། །ཡོན་ཏན་བརྗོད་པ་དང་བྲཱཧྥ་ཎ་་ལ་སོགས་པ་རྣམས་ལས་ལའང་ངོ༌། །ཡོན་ཏན་བཤད་པ་ནི། །ཡོན་ཏན་བརྗོད་པ་སྟེ། དེ་རྣམས་དང་བྲཱཧྥ་ཎ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་དངོས་པོ་དང་ལས་ལ་ཉ་འགྱུར་ཞིང་ཨི་མན་ཡང་ངོ༌། །སེམས་པའི་ལས་སོ། །དངོས་པོ་ནི༑ཛཱ་ཌྷ། ཛ་ཌི་མཱ། ཛ་ཌ་ཏྭཾ། ཛ་ཌ་ཏཱི། མོ་རྒྱུ། བ་དྷི་རྨོ། །བྲཱཧྥ་། བྲཱདྨ། :ནི་ལ། བཱ་ལི་ཤ། དུཧྥུ་རུ་ཥ། ྋརཱ་ཛ་ན། ག་དུ་ལ། དཱ་ཡཱ་ད། བི་ཥསྟིི། བི་ཤ་སྱ། ཙ་ཏུརྦ་ཎ་སོགས་རྣམས་ལས་་་རང་དོན་ལའོ། །:དང་པོའི་མཐའ་རྣམས་རང་དོན་ལའོ། །།དང་པོའི་མཐའ་རྣམས་ཉ་འགྱུར་ཏེ་རིགས་བཞི་རྣམས་ཉིད་ནི། ཙཱ་ཏུརྦ་ཙྩོཙ་ཏུ་ར་བ་མྱི། ཏྲི་ལོ་ཀྱི། ཏྲི་ཀ་ལཾ། རྭ་རཱུ་པཾ། ཙཱ་ཏུ་རཱུ་པཾ། ཏྲི་གུ་ཧཱུཾ། ཋརྒྱུ་ཙྩཾ། སྡེ་ཉིད་ནི། སི་དྡྷཾ། བི་ཌ་ལྟ་བུ་ནི་བི་ཤྩ། བེ་དའི་མཐའ་ལས་ནི་ཚིག་གཉིས་ཀ་ལ་འཕེལ་བ་སྟེ། ཙཱ་ཏུ་རྗེ་བྱོ༄༅། །སཱརྦ་བི་ཏྱཾ། པཱུ་རོ་ཧི་ཏ་དང་པ་ཏིའི་མཐའ་ལས་སོ། །དངོས་པོ་དང་ལས་དག་ལ་ཥཱཉ་ཡང་ན་འགྱུར་ཏེ། པོ་རོ་།
