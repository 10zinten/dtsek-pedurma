བསྟན་འགྱུར། སྒྲ་མདོ། ༼རི༽ས། «ཅོ་»ར། ་«ཅོ་»ཨས། ཀྱང:ཕྱིན་པ། །ཀུན་ཏུ་ད་ཀྱི་ཨེ«ཅོ་»ཤ@«ཅོ་པྲི། «ཅོ་»ཨ། «ཅོ་»རཱི། «ཅོ་»ཕྲི། «ཅོ་»གླི། *«ཅོ་»རི། སྲི། «ཅོ་»བྷིཉ། «ཅོ་»ཤི། ཀྱི«ཅོ་»ཎི་ཧཱི། «ཅོ་»དད། @«ཅོ་»ཛིཥ། །«ཅོ་»མཐིའི། ︽«ཅོ་»ཏྲི། «ཅོ་»དྲི། @«ཅོ་བྲི། «ཅོ་»དད། ཉི«ཅོ་»ཤ@«ཅོ་»བྷི་སུ་བྷམཱུ། ཚེ«ཅོ་»ཀྲི། «ཅོ་»ཏྲི། «ཅོ་»ཧྲི། ཉི«ཅོ་བསྒྲེངས་པ། «ཅོ་»ཀྱི་དྲི། «ཅོ་»གི། «ཅོ་»ལྟུང་། ཚེ«ཅོ་»དཱི་དུ། @«ཅོ་ལའོ། «ཅོ་བྲི་ད། «ཅོ་བྲི། ཚེ«ཅོ་»ཤི། ཨེ«ཅོ་སྲི། ཉེ«ཅོ་»པའོ། ཅེ«ཅོ་»མི། རོ་«ཅོ་»ཊུ། ཚེ«ཅོ་»། «ཅོ་»ཀྲི། «ཅོ་»བྲིའྲ། «ཅོ་»ཊུ། «ཅོ་»ཕྲིལླི། ཨ«ཙོ་»། «ཅོ་»ཀླུ། «ཅོ་»འི། ཚེ«ཅོ་»རི། གྱི«ཅོ་ཀྲི། «ཅོ་། «ཅོ་རི། «ཅོ་»རི། «ཅོ་»ཏྲི། «ཅོ་»ཏྲིན། «ཅོ་དྲི། ཨུ«ཅོ་»དྲིན། «ཅོ་དྲི། ཨེ«ཅོ་»ཙཱི། «ཅོ་མི་ཊ། «ཅོ་»བྲི། ཚེ《ཅོ་»གླིང་། ཀྱེ«ཅོ་»སྣ། «ཅོ་»བྲི། ཉེ«ཅོ་»ཏྲི། «ཅོ་»ལྕི། ༼ཉི«ཅོ་»གྱུར་ལའོ། །གྲུ། ཨེ«ཅོ་»ཊ། །ཏིའོ«ཅོ་»ཀྲི། «ཅོ་»ཊ། «ཅོ་»མུ། ༄༅«ཅོ་»ཊ། «ཅོ་སྲིང་། «ཅོ་མིང་། «ཅོ་»རི། @«ཅོ་»ནི། ཀྱི«ཅོ་»ཀྲི། གེ«ཅོ་གི། «ཅོ་»དྲིང༌། ཉེ«ཅོ་»དྷིང་། ཅི«ཅོ་སྲི། །«ཅོ་»སྐྱི། ཀྱེ«ཅོ་»མི། ་ཁྱེ«ཅོ་»འཇིགས། («ཅོ་»དྷི། རྒྱུད་། ་«ཅོ་»ཏྲི། «ཅོ་»གཤད། «ཅོ་»ཀྲི། ®«ཅོ་»ཏཱི། གྱི«ཅོ་»བྲི། «ཅོ་»ཊིནུ། ཚེ«ཅོ་»གླུ། «ཅོ་»ཀྲིཉ། @«ཅོ་»སྟི། »«ཅོ་»ཀཱི། ས་«ཅོ་»ཎཱི། «ཅོ་»འཇིགས། «ཅོ་»ཊ། མི«ཅོ་»ཀྲི། «ཅོ་»དྲི། «ཅོ་»ཊ། ཚེ«ཅོ་»ཊི། «ཅོ་»ཊ། ཀྱེ«ཅོ་»སྤྱི། «ཅོ་»ཊི། ཨེ«ཅོ་»ཊི། ཨི«ཅོ་»ཊ། e«ཅོ་»ཀྲི། «ཅོ་»སྒྲ། «ཅོ་»ཡན། «ཅོ་»སྲིད། «ཅོ་»—དྲཟླ་པ་རི། «ཅོ་»གཎ། «ཅོ་»དི། «ཅོ་»གི། ཚ«ཅོ་»མི། ཚུ«ཅོ་»+པ། །དོན་::ཀྱིངབདཔ། ་ཀྱི་ཕྱོགས་ཀྱིལན་(། ་་་་་་་་་་་་། །མང་ཆེ་ཙོང་།
