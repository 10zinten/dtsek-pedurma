བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིདཔལ་ལྡན་ཆོས་ཀྱི་དུང་དང་གཎྜི་དང་། །ཟབ་ཡང་སྒྲ་སྒྲོགས་ཆོས་ཀྱི་རྔ་བོ་ཆེ། །སྒྲོགས་མཁས་ངག་དབང་གོ་འཕང་མྱུར་ཐོབ་ཤོག། །།།ཀ་ཏm.སླར་ཡང་རྒྱ་དཔེ་རྙིང་པ་ལ་གཏུགས་ཏེ་འགྱུར་བཅོས་དང་ཞུས་དག་་་ཅུང་ཟད་གྲུབ་པོ། །།།།འཁཚོ་ཀླུབ་པ། །ཀ་ར་ཎ་ཁབ་འདེབས་སོགས་ཀཀ་ར་བའི་ལྷོ་ཏཎྜ་ཀ། ཊ་འག། །བར་གྱིབསྡུར་མཆན། གཏེད། རྒྱུ་ཞིག་ཁྱེཚོ་ཀཱ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«པེ་»«སྣར་»ལ་བཞུགས་མེད་པས་«ཅོ་»དང་དཔེ་བསྡུར་བྱས་ཡོད། «ཅོ་»པར། «ཅོ་»སྒྲུབ། །«ཅོ་»ཧཱུྃ། «ཅོ་སད། ལ་ལ་ལ་ལ་་་་་ལ་། འལ་(«ཅོ་»མེ། Q«ཅོ་»བིཥྛ། («ཅོ་»ཏིཙྩཾ། «ཅོ་»དྲི། «ཅོ་པ་ཙུ། @«ཅོ་»ཀྲི། «ཅོ་»ནེ། («ཅོ་»དེར། «ཅོ་»ཀན«ཅོ་»རྟ། ཚེ«ཅོ་རྣམས། གྱི«ཅོ་»པར། «ཅོ་སུ། ཨ«ཅོ་»ཀེ། «ཅོ་»འགྱུར་བའོ། ཎ་ཟླ:«ཅོ་»+དད། «ཅོ་»པཱ། «ཅོ་»མ། «ཅོ་»ཀེ། མི«ཅོ་»མི། «ཅོ་»ག་མཛྷ། ༄༅«ཅོ་»པའི། ཉི«ཅོ་»བཱ། «ཅོ་»ཏ། «ཅོ་པྲ། «ཅོ་»པྲི། «ཅོ་»པྲ། ཅི«ཅོ་»ས། །«ཅོ་»དེ། «ཅོ་»g་ཏོ་ཙཱ། W«ཅོ་»པ། «ཅོ་»ཀཱ་ཎཱ། མ«ཅོ་»ཕྲ། @«ཅོ་»ཏེ། ཚ«ཅོ་»བློག་ཀུ། །(«ཅོ་»གྲིཙྪཱ། «ཅོ་»ན། «ཅོ་»ནི། «ཅོ་»བྲི། མུ«ཅོ་»འཇིག«ཅོ་»ཀཎྜི། ཀྱི«ཅོ་»ཀཏྲི། ཉི«ཅོ་ཤཱང་། ཚི«ཅོ་»—ཡུལ་ཅན་དང༌། «ཅོ་»ཏ། མ«ཅོ་»ཌི། 《་(ཀྱི«ཅོ་»འདི། ཉི«ཅོ་»+ཕྱིར། ་ཚ@@«ཅོ་»ཁྱོཏ། «ཅོ་»ནི། ཀ་ཁ་སློག:«ཅོ་»ན། «ཅོ་»ཀོ། འ«ཅོ་»—ལ། («ཅོ་པར། «ཅོ་»ཀྲིཙྪ། ༄༅«རྩ་རྐརགས་འདྲ་བ། དཔལ་«ཅོ་»ཏོ། «ཅོ་»བ། ་«ཅོ་»འདུན། །ཐུབློ་ཝཁང་::ཀྱིཀྱིའི«ཅོ་»མ་ཡིན་པ། ཉི«ཅོ་»པ། «ཅོ་རྣམས། བ་ལ་ས་བ་ག་དར:གལ་(«ཅོ་»ང་། «ཅོ་རྣམས། ཚ«ཅོ་»ཊ། «ཅོ་»པས། O«ཅོ་བནྡྷ་ཛི་པི་ཀཾ«ཅོ་»དང༌། «ཅོ་»ཀྲི། «ཅོ་»དཱ། «ཅོ་»ཎ། །
