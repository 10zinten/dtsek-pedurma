ཙ་ཀྲ། པྲཀྵ་སོགས་སོ། །།:：ཨུ་རྐ་ར་སོགས་ལས་ཚའོ། །དོན་བཞི་པོ་རྣམས་ལ་འགྱུར་རོ། །ལ་། ཨུཏྟ་རཱི་ཡཾ། ཤ་ཕ་རི་ཡཾ། ཨུཀྐ་ར། ཤ་ཕ་ར་པིཔྤལ། པིལླི་ལཱི་མཱུ་ལ། ཨ་སྨན། ་་:ཨཀི། །ཨིཎྜ། བྲྀཀྵ། །ན་ཌ་སོགས་ལས་:：ཀུཉ་འང་ངོ་། །ན་ཌ་སོགས་ལས་དོན་བཞི་པོ་རྣམས་ཚ་འགྱུར་ཏེ། ཀུ་ཀ་ཨཱ་ག་མའོ། །ན་ཌ་ཀཱི་ཨིཡཾ། པྲཀཱ་ཀཱི་ཡཾ། :ཀུཉྩ་ལས་ཐུང་ངུ་ཉིད་ཀྱང་ངོ༌། །ཀྲུཉྩ་ཀྱི་ཝིཡཾ། ན་ཌ་སླཀྵ། བ་ར་:བིཝ་བེ་ནིུ། དྷེ་ནུ། ཨིཀྵ།  ོཀཱཥྛ། ཀ་པོ་ཏ། བེ་ཏ་ས། ་ཏ་ཀུན་གྱི་ན་དབྱི་བའང་ངོ༌། །ཛ་ན་པ་ཉིང་ལ་འཇིག་གོ། །དེའི་མིང་སྟོན་པ་ལ་དོན་བཞི་པོ་ལ་བྱས་པ་དེ་ཡུལ་གྱི་ཁྱད་པར་:ཛ་ན་པ་ད་་་མངོན་པར་བརྗོད་པ་ལ་འཇིག་པར་འགྱུར་ཏེ། ་གྲོང་གི་སྤྱི་ནི་ཛི་ན་པ་ད་སྟེ། ལྔ་ལེན་པ་རྣམས་གནས་པའི་ཛ་ན་པ་ད་ནི། བཉྩ་ལ། ཀུ་ར་བ། མཉྩ། :：ཨངྒཱ༑ཀཱ་ལིངྒྷ། གནྡྷ་ར་ཡ། པུཎྞ། པུ?། འདིར་ཅི་ལ་མི་་འགྱུར་ཞེ་ན། འདིར:：ཨུ་དམ་ཨེཔར་ཡོད་པ་ནི། །ཨོ་དུམྦ་རོ་ཛ་ན་བ་དེ། དེ་ནི་མིང་ལ་འཇུག་གོ། #། འདིར་འཇིག་པའི་མཐའ་དེ་མིང་དུ་མི་འགྱུར་རོ། །བ་ར་ཎ་སོགས་ལས་སོ། །བ་ར་ཎ་སོགས་རྣམས་ལས་མཆོག་གི་དོན་ལ་འཇིག་པར་འགྱུར་ཏེ། ཛ་ན་:བ་དའི་དོན་མི་རྩོམ་པའོ། །བ་ར་ཎ་ཞེས་བྱ་བ་རིང་བར་གྱུར་པའི་གྲོང་ཁྱེར་ནི། བ་ར་ཎཿཤིངྒཱི། ཤཱརྨ་ལ་ཡཔུ་:རྡོ་གོ་དོ། ཨ་ནིནྡྷ་ཡ་ན་པལླི། །ས་ཀུཎྜི་ཛ་ལ་པ་དཱ། མ་ཐུ་རཱཏ། །ཨུཛྫ་ཡ་ནཱི། ཤ་ཡཱ། ཏགླུ་ཤི་ལ། དབྱིབས་ཀྱི་ཚོགས་འདིའོ། །དེས་ན་ཚ་ཅན་དང༌། རྒྱ་ཤུག་ཅན་དང་མི་རིང་བའི་གྲོང་ནི། ཀ་ཊུ་ཀ་བ་ད་རཱི། ཤརྐ་ར་ལས་ཡང་ནའོ། །ཤརྐ་རའི་སྒྲ་ལས་སྐྱེས་པའམ་དོན་བཞི་པོ་རྣམས་ལ་འཇིག་པར་འགྱུར་ཏེ། ཤརྐ་རཱ། ་ཀུ་མུ་ད་སོགས་ཀྱི་ཕྱིར་ཋ་ནའོ། །།ཤརྐ་རི་ཀཾ། བ་རཱ་ཧ་སོགས་ཀྱི་་།
