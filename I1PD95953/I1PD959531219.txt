བསྟན་འགྱུར། སྒྲ་མདོ། *རིའིརྣམས་ལས། །ཨསྠཱ་ཏི་ཞེས་པས། ་ཨསྠཱ་ཏིའི་ཆོ་ག་སྟེ། །པཱུརྦ་ཨ་དྷ་ར་དག་གི་ས་ཡང་པུ་ར་ཨ་དྷ་དག་ཀྱང་ངོ་ཞེས་པས་པུ་སྒྲ་ནི་པུ་ར་རུ་བསྒྱུར་རོ། ་ད། བྱས་ནི་དཔེར་ན། ས་མཱ་ནེ་ཧ་ནི་ས། ས་ཏྱ་ས་རེ་བ་བྱ་བའོ་ཞེས་པའི་ཚིག་ཕྲད་ལས། ས་མཱ་ནའི་ས་རུ་བསྒྱུར་ཏེ། རྒྱའི་སྒྲུབ་པའོ། །ཨུཏ་ནི་དཔེར་ན། ལོ་སྔ་མ་ལ་པ་རུ་ཏའོ། །ཨ་རི་ནི་དཔེར་ན། པུཊྚ་ཏ་རའི་སྒྲ་ངག་ལ་རིམ་པ་བཞིན་དུ་ཨུཏ་དང་ཨ་རི་དག་བསྒྲུབ་པར་འགྱུར་ཞིང་། པ་ར་རུ་བསྒྱུར་བ་ཡང་སྟེ། ལོ་མངོན་པར་བརྗོད་པ་ལའོ། །པ་རུ་ཏ། པ་རཱ་རི། བི་རཱནྣ་ཞེས་པའི་ཚིག་ལས་སོ། །ས་མ་སན་ནི། དཔེར་ན། ལོ་འདི་ལ་ཞེས་པ་ལ། ཨི་ཥ་མ། ཨི་ད་མ། ས་མ་སནའི་བསྒྲུབ་པ་ལོ་མངོན་པར་བརྗོད་པ་ལའོ། །ཨི་དཾ་ཨི་རུ་བསྒྱུར་བའོ། །ཨི་ཥ་མོ་ཧ་སྲོ་སོ་བེ་ཏི་ན་བཱི་ཏ་ནཱཏ། །ཀིཏ་ལའང་ཞེས་པས་ཨཏ་ཨེ་ཙའོ། །ཀུའི་ཡང་ངོ་ཞེས་པ་ལ་སོགས་པའི་མདོས་ཥ་ཉིད་དོ། །ཨེ་དྲ་བི་ནི། །དཔེར་ན། པ་རསྨིནྣ་ཧ་ནི། པ་རེ་བྱ་བ། ཏནྟི། ས་ཏྱ། །པ་རེ་བྱ་བིའོ་ཞེས་པའི་ཕྲད་ལས། པ་རིའི་སྒྲ་ལས་ཨེ་བྱ་བའི་ཆོ་གའོ། ་། བྱ་ནི། དཔེར་ན། ཨ་སྨིན་ནི་ཧ་ནི། ཨ་དྱ། །ཨི་དམ་ནི་ཨ་རུ་བསྒྱུར་ཞིང་། རྒྱའི་སྒྲུབ་པའོ། །ཉི་མ་མངོན་པར་བརྗོད་པ་སྟེ། ཤརྡ་བྱ་ས་ནེ་ནཱ་བྱ་བྱུགཉྩཤུ་ཞེས་ཚིག་ཕྲད་ལས་སོ། །ཨེ་རླུས་ནི། དཔེར་ན། ཉི་མ་སྔ་མ་ལ། པཱུབྦེ་རླུ། ཉི་མ་གཞན་ལ། །ཨ་རྩེ་གླུ། །ཆེས་གཞན་པའི་ཉི་མ་ལ། །ཨ་ནྱ་ཏ་རེ་རླུ། །ཅིག་ཤོས་ཉི་མ་ནི། ཨི་ཏ་རེ་རླུ། ཉི་མ་གཞན་ནི། ཨ་པ་རེ་རླུ། །ཉི་མ་འོག་མ་ནི། ཨ་དྷ་རེ་ཟླུ། ཉི་མ་ཕྱི་མ་ནི། །ཨུཏྟ་རེ་རླུ། །རྣམས་ཏེ། པུཏྤ་ཨ་ནྱ། །ཨ་ནྱ་ཏ་ར། ཨི་ཏ་ར། །ཨ་པ་ར། ཨ་དྷ་ར། ཨུཏྟ་ར་ལ་ཨེ་རླུསའོ་ཞེས་པས་དེ་རྣམས་ལ་ཨེ་་་ཡུས་ཀྱི་ཆོ་གའོ། །།ཉི་མ་མངོན་པར་བརྗོད་པ་ལ་གཉི་གའི་ཉི་མ་ལ། ཨུ་བྷ་ཡ་རླུ། །
