ཏ་མོ་ནཱ་མ་བྲྀ་ཥ་ལོ་ཡཾ་བྷ་བཱན་ཡཱ་ཛ་ཡེ་ཏ། །ཡཱ་ཛ་ལི་བྱ་ཏི། སྨད་པ་ཞེས་་པ་ཉིད་དེ། ་ཀོ་ནཱ་:མ་བྲཱཧྥ་ཎོ་ཡཾ་ཏ་ཏྲ་བྷ་བཱན། ཡ་ཛ་ཡ་ཏི། མི་བཟོད་པ་དང་མི་སྲིད་པ་དག་ལའོ། ཝཱ། །ཡུལ་གང་གི་དོན་དེའི་དོན་གྱི་བྱིངས་ལས་ལིང་དང་་ལིཊ་དག་འགྱུར་ཏེ། དང་པོའི་ལ་ཡིག་ཐམས་ཅད་དམིགས་བསལ་ལོ། ༈། ན་་་མན་ཡཱ་མི་དུཛྫ་ན། སམྤ་དཾ་:པདྡྷ་ཡ་ཏེ། ཡང་ན་པྲ་དྨ་ཏི། ནཱ་བ་ཀལྤ་ཡཱ་མི་ན་:ཤད་དྷེ་ན་སཾ་བྷཱ་བ་ཡཱམི་མནྡ་བྷཱ་གྱི། པུ་བཎ་ལ་བྷེ་ཏ། ཡང་ན་ལ་དྨ་ཏེ། ཡང་ན་བྱུང་བ་བྱ་བ་ཤོར་བ་ལ་ཕྲེངའོ། །བྷཱ་བི་ནི་ཏུ་ནི་ཧཱུཾ། ན་མན་ཡཱ་མི་ཏ་ཏྲ་བྷ་བཱན། བོདྡྷཾ་མནྟ་མ་:པཱ་ཋ་ཡི་ཏི། ཀིཾ་ཀ་ལ་དང་། ཡོད་་་པའི་དོན་དག་ལ་ཀྲྀཊའོ། །ཀིམ་ཀི་ལའི་སྒྲ་དང་པོ་ཡོད་པའི་དོན་དག་སྦྱོར་བ་་ན་མི་བཟོད་པ་དང་མི་སྲིད་པའི་ཡུལ་ལ་ད་ལྟར་བའི་བྱིངས་ལས་མྲྀཊ་ཉིད་འགྱུར་་གྱི་ལིང་ནི་མ་ཡིན་ནོ། །ཀིམ་ཀི་ལ་ཏ་ཏྲ་བྷ་བཱན་པྲྀ་ཥ་ལཾ་ཡཱ་ཛ་ཡི་ཏི། ཀིཾ་ཀྲ་མིཥྛ་ཏི་ཀི་ལེ་ལེ་ཥ་བཱ་མ་ན། ན་གྷོ་མེ་ན་མརྟ་ཡཱ་མི། ཚོན་སནྣཱ་བ་ཡཱ་མི། ན་:ཤད་དྷེ། ཁོ་ཨ་སྟི། བྷ་བ་ཏི། བི་བྱ་ཏེ། ནཱ་མ་ཏ་ཏྲ་བྷཱ་:བཱན་བྲི་ཥ་ལཾ་ཡཱ་ཛ་ཡེ་ཥྛ་ཏི། ཡད་ཡ་ད་ཡ་དཱ། ་ཛཱ་ཏུ་རྣམས་ལ་ལིངའོ། །ཡད་ལ་སོགས་པ་་་རྣམས་སྦྱོར་བ་ན་མི་བཟོད་པ་དང་མི་སྲིད་པ་དག་ལ་ལང་ཉིད་འགྱུར་གྱི་ཁྲིཊ་མ་་་ཡིན་ནོ། །་། ཡཏྟ་ཏྲ་བྷ་བཱན་པྲྀ་ཥ་ལཾ་ཡཱ་ཛ་ཡེ་ཏེ། ཡ་དི་ཏ་ཏྲ་བྷ་བཱན་པྲྀ་ཥ་ལཾ་ཡཱ་ཛ་ཡེ་ཏེ། ་ཡ་དཱ་ཏ་ཏྲ་བྷ་བཱན་བྲྀ་ཥ་ལོ་ཡཱ་ཛ་ཡེ་ཏ། ཛཱ་ཏུ་ཏ་ཏྲ་བྷ་བཱན་བྲྀ་ཥ་༌ལོ་ཡཱ་ཛ་ཡེ་ཏ། ་ཚུན་སམྦྷ་བ་ཡཱ་མི། ན་མནྟ་ཡཱ་མི། །ོན་ཤརྡ་དེ། བྱ་བ་ཤོར་བ་ལ་བྱུང་བ་ལ་སྲིང་ཡང་ནའོ། །།བྷཱ་བི་ནི:;ཏུ་ནི་ཏྱཾ། ན་སམྦྷ་བ་ཡཱ་མི། ཡ་ཏུ་ཏྲ་བྷ་བཱན་བྲི་ཚེཥ་:ལཾ་མཱ་ཡ་ཛ་མི་ྋཏ། །ཡཙྪ་དང་ཡ་ཏྲ་དག་ལས་སྨད་་པ་:ལའང་ངོ་མེ། །ཡཙྪ་དང་ཡ་ཏྲའི་སྒྲ་དག་སྦྱོར་བ་ན་སྨད་པ་ལ་མི་བཟོད་པ་དང་།
