དང་བ་ཧ་སོགས་ཀྱི་གཙོ། །དུ་ཧ་སོགས་ཀྱིས་སྒྲུབ་བྱེད་དེ། །ཨིན་མཐའ་རྣམས་ཀྱི་ལས་མཁས་བླངས། །བརྗོད་བྱ་སྦྱོར་བ་པོ་བྱེད་པོར། །བརྗོད་པར་བྱ་བ་ཡོད་མིན་ལ། །ནི། གཞན་ཡང་ལས་ལས་སྐྱེས་པར་གསུངས། །ལས་གཞན་་་ཡང་དག་ཡོད་པ་ན། །ཁ། དུས་སོགས་མ་བརྗོད་ན་ཡང་ངོ༌། །གལ་ཏེ་གང་ཞིག་་་ཁྱད་པར་ཅན། །ཏི་སོགས་:ཀྲྀ་ཏས་བརྗོད་ཉིད་ཅིང་། །བྱ་བ་ཁྱད་པར་གྱིས་མ་བརྗོད། །འ་། དེ་ཡང་བརྗོད་པར་འགྱུར་ཞེས་པའོ། །བརྗོད་པ་ཉིད་ནི་ཚིག་གཞན་གྱིས། ་། བརྗོད་པ་ཉིད་དུ་ལ་ལར་འགྱུར། །ལ་ལར་བརྗོད་པ་མ་བརྗོད་ཉིད། །།བྱེ་། ལ་ལར་མ་བརྗོད་མ་བརྗོད་འགྱུར། །ཞེས་པ་འབྲེལ་པ་མདོར་བསྟན་པ་ཐུན་མོང་དུ་བསྟན་པ་སྟེ་དང་པོའོ། །ཏོ་ལ་། ལུས་དང་། ད་བར་ཁ་ཡ་ན་ཡོད། ཡཌ་ཉིད་::ཀ་བྱེད་པ་ནི། འབགཤིས། ཁལ་འོད་ཌབློའི་ལྟ་དང་། ཀན་་རང་ཁྲུ་པ་ཁོ་ཀ་པ་རསྨི་བྱེད་པ་པོ་བརྗོད། །ཨཱད་མ་ནེ་ཚིག་དོན་གསུམ་རྣམས། །བྱིངས་ཀུན་པ་ལ་ཡ་ཎ་འདི། །ཨ་། དངོས་པོ་ལས་བསྒྲུབ་པ་ལ་འགྱུར། །བྱིངས་ཀུན་པ་ལ་བྱེད་པ་པོ། །བསྒྲུབ་ལ་རྣམ་འགྱུར་ཐམས་ཅད་རྣམས། །ཨ་བྱ་ཏ་ནཱིའི་རྣམ་དབྱེ་ལ། །སིཙ་སཎ་བྱེད་པོ་ལས་ལ་འགྱུར། །ཙཎ་ཨཎ་བྱེད་པ་པོ་ལ་འགྱུར། །ཨིཙ་ཕལ་ཆེར་དངོས་ལས་ལ། །ཤཱསྟ་ནི་སྱ་སིཙ་སྨོན་ལ། །ཨིཊ་ཨིཙ་བཞིན་དུ་དངོས་ལས་ལ། །།ཧ་ནི་གྲ་ཧི་:ཏྟཾ་ཤ་དབྱངས་མཐའ་ལས། །ཚིགས་བཅད་ཕྱེད་ནི་བྱེ་བྲག་བཤད་པར་བྱ། །ཨ་ཏི་པད་ཏི་པ་རོ། །ཨ་བྱ་ཏ་:ནི་རྒྱ་སྟ་ནིའི། །བཞི་རྣམས་འདས་པའི་དུས་ལ་འགྱུར། །བ་། བརྟ་མཱ་ན་ད་ལྟར་ལའོ། །བྷ་བི་བྱེ་ཏི་སྨོན་པ་ཡང་། ཁོ་། དེ་ལས་འབྱུང་འགྱུར་ལ་འགྱུར་རོ། ལོ་། སང་ལ་ཤཱ་:སྟ་ནི་རྟག་ཏུའོ། །དེ་ཉིད་གཞན་དུའང་བྱེ་བྲག་བཤད། །ཆོ་ག་སོགས་ལ་བདུན་པ་དང་། །ལྔ་པའོ་བསྒོ་དང་སྨོན་ལམའང་དེ། ཀྱི་། སྒྲ་གཞན་སྦྱོར་ལ་།
