ཨཏྲའི་མ་ཡིན་ནོ། །པྲ་ལས་བྷ་ཧའི་ཞྭ་ལའོ། །ནཤྲལ་ལས་ཤུ་ཙི་དང་ཨཱི་ཤྭ་ར་དང་། གླུ་ཛྷ་དང་ཀུ་ཤ་ལ་དང་ནི་པུ་ཎ་རྣམས་ཀྱིའོ། །ཧ་ནའི་ཏ་ཙི་ཎལ་མ་ཡིན་པ་ལའོ། །ཨཏྲའི་ཡུཀྲ་ཎཁྲ་མ་ཡིན་པ་ལའོ། །མའི་ཨཊར་བཅས་པའི་བ་མི་དང་ཨ་མི་དང་ཀཱ་མ་དང་ཨཱ་ཙ་མ་དང་བི་ཤ་མ་མ་ཡིན་པའོ། །མིའི་ཎལ་ལ་་་རུང༌། །ཤེ་དང་རི་དང་བློ་དང་ཧྲི་་དང་ཀླུ་ཡིཀྵ་ལཱི་དང་ཨཏྲ་རྣམས་ཀྱི་པུཀ་ཎི་ལའོ༑། ཤཱ་དང་ཙྪ་དང་སཱ་དང་ཧྭ་དང་བྱ་དང་བི་དང་པ་བཱ་རྣམས་ཀྱི་ཡུ་ཀའོ། །བྷེའི་ོརྣམ་པར་གཡོ་བ་ཛུ་ཀའོ། །དྷུཉྫ་དང་བྲྀཏྲ་དག་གི་ནུ་ཀའོ། །།།ལིའི་གཉེན་པོ་དང་རྣམ་པར་ཟློས་པ་ལ་རུང་ངོ་། །།ལ་བའི་ལུགྲའོ། །པ་ཏིའོ། །བསྒོ་བ་དང་པོ་ལ་ས་བྷིའི་ཥཱ་ཀའོ། །བྷ་ཡའི་བའོ། ་། ཤ་དིའི་འགྲོ་བ་མ་ཡིན་པ་ལཏྲའོ༑། སཏྱ་དང་ཨངྒ་དང་པོ་དྭ་རྣམས་ཀྱི་སུཀྲའོ། ་། མ་ཕྱི་བ་རྣམས་ཀྱི་ཐུ་་་བའོ༑་་། ཙི་ན་དང་ཎ་མ་དག་ལ་རིང་པོ་ཡང་ངོ༌། །ཚ་དའི་གྷ་ལའོ། །པ་ལ་སོགས་པ་ལ་གཅིག་ལའོ། །ཨིས་དང་མན་དང་ཧྲན་དང་ཀིན་རྣམས་ལའོ། །ཙང་ལ་ཉེ་བའི་མཐའིའོ། །ཨཀྲ་ཕྱི་བ་ཅན་དང་གྲྭ་སུ་དང་རིཏྲ་ཕྱི་བ་རྣམས་ཀྱི་མ་ཡིན་ནོ། ་་། བྷ་ཛ་དང་བྷས་དང་བྷ་ཥ་དང་དཱིས་དང་ཛ་བ་དང་ནཱི་ལ་དང་བི་ཌ་་་རྣམས་ཀྱི་རུང་ངོ༌། །ཀ་ཎ་ལ་སོགས་པ་རྣམས་ཀྱིའོ། །རའི་རིཏྲའོ། །།གླུ་ཨིཏྲའོ༑། སྦྲའིའོ། །བ་པའི་བི་སྱའོ། །དེ་ངའི་དི་ཀ་ལེཊ་ལའོ། །ཁམས་མ་ཡིན་པའི་ཀ་ལའོ། །།ཨིཏྲ་ཨིཏྲའི་སུས་མ་ཡིན་པ་ལས་ཨཔྲ་ལའོ། །ཡ་དང་ཀ་དག་ལས་ཨ་བའི་རྐྱཀྲ་དང་ཏཔྲ་མ་ཡིན་པའི་རུང་ངོ་། །བྷ་སྟ་དང་ཨེ་ཥ་དང་ཨཛྲ་དང་ཛཱི་དང་ཧྥ་དང་སཱ་རྣམས་ཀྱིའོ། །སྐྱེས་བརྗོད་མ་ཡིན་པ་ལས་ཨཏྲ་ཡང་ངོ༌། །བརྟ་ཀཱ་བྱ་ལའོ། །སཱུཏྲ་ཀ་དང་སུཏྲ་ཀ་དང་བྲི་ན་ར་ཀ་རྣམས་སོ། །ན་རི་ཀའོ། །ཡཏྲ་དང་ཏཏྲ་དག་གི་མ་ཡིན་ནོ། །ཤིས་པ་ལའོ། ་༌། ཀཱི་པྲ་པ་ལ་སོགས་པ་——།
