དྷཾ༑?གླུ་དྷོ། གླུ་དྷ། །ཀླུ་དྷཱ། བྷུདྡྷཾ། གླུ། གླུ་དྷེ། གླུདྡྷཾ། གླུདྡྷ། གླུ་དྷ། །ཀུདྡྷཾ། ཀླུདྡྷ། ཀླུ་དྷ། གླུ་དྷོ། གླུ་དྡྷཾ། གླུ་དྷི། གླུ་བྷོ། གླུp༑ཚོ་ཧེ་གྱུཏ། །ཧེ་ཀླུ་དྷོ། ཧེ་གླུ་དྷ། སཾ་པད་ཅེས་པ་འདིའི། །སུའི་འོ༑། རྫོགས་པ་ལའམ། ཁར་ལ་ཛྷལའི་ཙརའོ། །རྣམ་པར་གཅད་པའོ། །སཾ་པཏ། སཾ་པད། སཾ་པ་དོ། །སཾ་པ་ད། ：སཾ་པ་དཾ། །སཾ་པ་དོ། སཾ་པ་ད༑ལྔ་སཾ་པ་དཱ། སཾ་པདྡྷཾ། སཾ་པདྷི། སཾ་པ་དེ། སཾ་པདྡྷཾ། སཾ་པདྡྷ། སཾ་པ་དེ༅། སཾ་པདྡྷཾ། །སཾ་པདྡྷ། སཾ་པ་ད། སཾ་པ་དོ། སཾ་པ་དཾ། སཾ་པ་དི། །པསཾ་པ་དོ། སཾ་པཏུ། ཧེ་སོ་པད། ཧེ་སཾ་པ་དོ། ཧེ་སཾ་པ་ད། དེ་ལྟར་ཤ་རད། ་ཤ་ར་དོ། ཤ་ར་ད། ཡོ་ཥིཏ་ཅེས་པ་འདིའི། སུའི་འོ། །ཛྷལའི་ཛ་ཤའོ། །རྫོགས་པ་ལ་ཡང་ནའོ། །ཁར་ལ་ཛྷལའི་ཙརའོ། །རྣམ་པར་གཅད་པའོ། །ཡོ་ཥིཏ། ༼ཡོ་ཥི་ཏོ། ཡོ་ཥི་ཏ། ཡོ་ཥི་ཏཾ། །ཡོ་ཥི་ཏོ། ཡོ་ཥི་ཏ། ཡོ་ཥི་ཏཱ། ༼ཡོ་ཥིཏྟཾ། ཡོ་ཥིདྡྷི༄༅། ཡོ་ཥི་ཏེ། ཡོ་ཥིདྡྷཾ། ཡོ་ཏིདྡྷ༑། ོཡོ་ཥི་ཏ། ཡོ་ཊིདྡྷཾ། ཡོ་བིདྡྷ། །ཡོ་ཥི་ཏ། ཡོ་ཥི་ཏོ། །་ཡོ་PHIBaresaPa^ཡོ་ཥི་ཏ། ཏི་ནི་ཤཱ་ཞེས་པ་འདིའི་:དཱིའི་ཨཱ་པའི་རིང་པོ་ལས་སོ་ཞེས་པ་སུ་དབྱིའོ། །ཨཱཔ་ལས་ཨོཏའི་ཤཱིའོ། །།ཨ་ལས་ཨཏ་ཨེངའོ། །དང་པོ་དག་གི་ཨཏ་ལའོ། །ཨམ་ལས་སྔ་མ་ལའོ། ་། ཤས་ལ་སོགས་པ་ཨཙ་དང་པོ་ཅན་རྣམས་ལ། པད་ནི་ཤ་མཱ་ས། ::：ཧྲིད་སུ་ཤན་དོ་ཥན་ཡང་ན་ཤས་ལ་སོགས་པའོ་ཞེས་པ་ནི་ཤཱའི་་་་སྒྲའི་བསྒྱུར་བ་ནི་ཤ་ཞེས་པ་འདི་ཡིན་ནོ། །ཕྱོགས་ལ་ཁ་ཌ་བཞིན་ནོ། །རྣམ་་་པར་གཅད་པའོ། །།ནི་ཤཱ། ནི་ཤེ། ནི་ཤཱ། ནི་ཧཱུཾ་ནི་ཤེ། ནི་ཤ། ནི་ཤཱི། ནི་ཤདྡྷཾ། ནི་ཤ་བྷི། ནི་ཤེ། ནི་ཤ་ཧཱུཾ། ནི་ཤ་བྷུ། ནི་ཤ*། །
