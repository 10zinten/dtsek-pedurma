བྱེད་པ་ཉིད་བརྗོད་པར་འདོད་པ་ལ་ལས་བཞིན་དུ་བྱ་བ་རྣམས་འགྱུར་ཏེ། ཡཀ་དང་ཨཱཏྨ་ནེའི་ཚིག་དང་ཙིཎའོ། །ཙིཎ་བཞིན་གྱི་དངོས་པོ་རབ་ཏུ་སྦྱོར་རོ། །པཱཛཱ་སྱཱཙྪི་བྱ་ཏེ་ཀྲྀཏྟཱ་སཱ་ཡ་མེ་བ། བྷི་བྱ་ཏེ་ཀུ་ཤུ་ལ་སཱ་མེ་བ། ་འཅི་ལྟར་ཀུ་ཤུ་ལ་ཡིས་འཇོམས་པར་བྱེད་ཅེ་ན། རྣམ་པར་འཇོམས་པའི་དོན་གྱི་ལས་ཉིད་ཀྱིས་ལས་བྱེད་པ་པོའི་མཚན་ཉིད་ལྡན་པའི་ཕྱིར་རོ། ་། སྭ་ཡཾ་གྱི་སྒྲ་ནི་འདིར་བདག་་སྟེ་རང་གི་དོན་ལ་འཇུག་པར་བྱེད་པས་ལས་བྱེད་པ་པོ་ཉིད་གསལ་བར་བྱེད་དོ། །བྱ་བའི་གཞི་ལས་ལ་ཞེས་པ་ཅི་ཞེ་ན། གཙྪ་ཏི་གྲྭ་མཾ་དེ་བ་དཏྟི། ཀུ་ཤ་དང་རཉྫི་ལས་ཤན་དང་པ་རསྨིའི་ཚིག་ཡང་ནའོ། །ཀུ་ཥི་དང་རཉྫི་ལས་རབ་ཏུ་སྦྱོར་བའི་ལས་ལ་བྱེད་པ་པོ་ཉིད་དང་བརྗོད་པར་འདོད་པ་ལ་ཐུན་དང་པ་རསྨིའི་ཚིག་་ཡང་ན་འགྱུར་:ཏེ། ཤ་ཥྛ་ཏི་པཱ་དང་སཱ་ཡ་མེ་བ། ་་ཡང་ན་ཀུ་ཊྛ་ཏེ། །རཚུ་་་ཏི་:བསྭོ་སྭཱ་ཡ་མེ་བ། ཡང་ན་རཛུ་ཏེ་ཀུ་ཉྩནྟི་ཛ་གྷཱ་ཤུན་ལ་རྟག་ཏུ་:ནུམ་ལོ་དུ་ཧ་ལས་ཡ་ཀ་མ་ཡིན་ནོ། །དུ་ཧི་ལས་ལས་ལ་བྱེད་པ་པོ་བརྗོད་པར་འདོད་པ་ལ་ཡ་ཀ་མི་འགྱུར་ཏེ། དུགྒེ་གོ། སཱ་ཡ་མེ་:བ། །བྲུ་དང་ཀི་ར་ཏི་སྣུ། །་་ན་མ༑རྒྱན་གྱི་དོན་དང་ཤི་ཤཱནྡྲི་ཎི། སན་རྣམས་:ལ་ཙིཎ་འང་ངོ༌། །།།འདི་་་རྣམས་ཎིརབ་ཏུ་སྦྱོར་བའི་བྱེད་པ་པོ་ལ་ཙིཎ་མི་འགྱུར་ཞིང༌། འང་ཡིག་གིས་ཡཀ་ཀྱང་ངོ༌། །བྷཱུ་ཏེ་དྷརྨ། སྭ་ཡ་མེ་བ། ཨ་བོ་ཙཏ། ཨ་བ་ཀི་ར་ཏེ་ཧསྟི་སྭ་ཡ་མེ་བ། ་ཨ་བཱ་:ཀིནྡྷ་པྲ་སྐུ་ྋཏེ་གོ། །སཱ་ཡ་མེ་བ། པྲསྣོབྦཙེ། ན་མ་ཏེ་དཎྜ་ཧཱུཾ་ཡ་མེ་བ།  ོ་ཨ་ནཾ་ལྟ་ཨ་ལཾ་ཀུ་རུ་ཏེ་ཀ་རྩྭ་སྭ་ཡ་མེ་བ། ཨ་ལམ་ཀྲྀཏ། ཨུཙྪཱ་ཝིཡ་ཏེ་དཎྞ། སྭ་ཡ་མེ་བ། ་ཨུ་ད་ཤི་ཤི་ཡཏ། ཤ་བྷི་ཏེ་མཱ་ལཱ་སཱ་ཡ་མེ་བས། །ཨ་བནྡྷིཤྩ། ཀཱ་ར་ཡ་ཏེ་ཀ་ཊ། སྭ་ཡ་མེ་བ། ཨ་ཙ་ཀཱིནྡྷ། །ལས་མ་ཡིན་པའི་ཏང་གི་མཐའ་ལས་ཅན་མ་ཡིན་པའི་ཕྱིར་དང་ལས་དང་བྱེད་པ་པོ་མེད་་།
