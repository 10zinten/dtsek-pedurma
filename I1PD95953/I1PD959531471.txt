བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིཝིམ་ཀརྦ༑ཁྲི་ཀ་ཥ་ལ་ས་ལས་སོ། །འདི་རྣམས་བི་སྔོན་མ་ལས་ངང་ཚུལ་སོགས་ལ་་གྷི་ནུ་ཎ་འགྱུར་ཏེ། བི་པྲམྦྷི། བི་ཀཱི། ཀ་ཥ་འཚེ་བ་ལའོ། །བི་ཀ་ཊཱི། ལས་འབྱར་བ་དང་རྩེ་བ་དག་ལའོ། །བི་ལཱ་སཱི། པྲ་ལས་བ་ད། བ་ཅེས་ལ་པ་མ་ཐ། སྲི་དྲུ་རྣམས་ལས་སོ། །པྲ་སྔོན་མའི་བ་ད་སོགས་དྲུག་པོ་ལ་ངང་ཚུལ་སོགས་ལ་གྷི་ནུ་ཎ་འགྱུར་ཏེ། སྲ་བཱ་དཱི། པྲ་ལཱ་:པི། པྲ་མ་ཤྲཱི། པྲ་པཱ་རཱི། པྲ་ད་བཱི། བ་ས་ནི་བ་ས་ཞེས་པ་འདི་གཟུང་ངོ༌། །།བ་ས་ཀུན་དུ་འགེབས་པ་་ལའོ་ཞེས་པ་འདི་ནི་མ་ཡིན་ཏེ། བི་ཀ་ར་ཎ་འཇིག་པའི་ཕྱིར་རོ། །ནིནྡ། ཧིནྣི༑ཀླུ་ཤ། ཁཱ་ད། བི་ནཱ་ཤ། བྱཱ་བྷཱ་ཥ། ཨ་སཱུ་ཡ་ལས་བུཉའོ། །ནིན་སོགས་བདུན་པོ་ལས་ངང་ཚུལ་སོགས་ལ་བུཉ་འགྱུར་ཏེ། ལནིནྡ་ཀ། ཧིནྟ་ཀ། གླེ་ཤ་ཀ། ཁཱ་ད་ཀ། བི་ནཱ་ཎིཤ་ཀ། བྱཱ་བྷཱ་ཥ་ཀ། ཨ་སུ་ཡ་ཀ། འདི་ནི་ཁྲོན་གྱི་དམིགས་བསལ་ལོ། །པ་རི་ལས་བཱ་དི། །ཀྱི་པ། ར་ཊ་ལས་སོ། །ངང་ཚུལ་སོགས་ལ་བུཉ་འགྱུར་ཏེ། པ་རི་བཱ་ཙེད་ཀ། །པ་རི་ཀྱེ་པ་ཅིཀ། པ་རི་རཱ་ཊ་ཀ། ཉེར་བསྒྱུར་ལས་དེ་བི་ཀྲུ་ཤ་ལས་སོ། །ངང་་་ཚུལ་སོགས་ལ་བུཉ་འགྱུར་ཏེ། ཨཱ་དེ་བ་ཀ། པ་རི་དེ་བ་ཀཿ། །བི་ཀྲོ་ཤ་ཀ། ཉེར་བསྒྱུར་ལས་ཞེས་པ་ཅི་ཞེ་ན། དེ་བ་ན་ཤཱི་ལི། དེ་བ་ཡི་ཏཱ། ༔ཀྲོ། གཡོ་བ་དང་སྒྲའི་དོན་ལས་ལས་མེད་པ་ལས་ཡུཙའོ། །གཡོ་བ་དང་སྒྲའི་དོན་རྣམས་ལས་ངང་ཚུལ་སོགས་ལ་ཡུཙ་འགྱུར་ཏེ། ཙ་ལ་ན། བེ་པ་ནི། ཙེཊྚ་ན༑ཤརྡ་ནི། ར་བ་ཎ། ་་:ཀལྤ་ན། ལས་མེད་པ་ལས་ཞེས་པ་ཅི་་ཞེ་ན། པ་ཋི་ཏཱ་བི་དྱེ། རིཏ་ལས་ཧལ་དང་པོ་ལས་སོ། །བིཏ་ལས་ཧལ་དང་པོ་ལས་བྱིངས་ལས་ངང་ཚུལ་སོགས་ལ་ཡུཙ་འགྱུར་ཏེ། བརྟནི། ཞུ་:བནྡྷ་ནེ། དེ་བ་ནི། བི་:ཀཏྠ་ནི། ངཏ་ལས་ཞེས་པ་ཅི་ཞེ་ན། བྷ་བི་ཏཱ། ཧལ་—།
