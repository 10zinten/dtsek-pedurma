བསྟན་འགྱུར། སྒྲ་མདོ། ?འིཏཱི་བྷུ། བྷ་ཝ་ཏུ། འོ། བྷ་ཝ་ཏྱཾ། བྷ་ཝ་ཏཱི་ནཱཾ། བྷ་ཝ་ཧཱུཾ། །བྷ་ཝ་སྐྱོ། བྷ་:བ་ཏི་ཥཱ། ཧེ་བྷ་ཝ་ཏི། ཧེ་བྷ་ཝ་ཏོ། །་ཧེ་བྷ་ཝ་ཏྱ། ཋ ། མ་ནིང་ལ་བྷ་ཝ་་ཏ་ཅེས་པ་འདིའི། །སུ་ཨམ་དག་གི་མ་ནིང་ལས་སོ། །ཛྷལའི་ཛ་ཤའོ། །རྫོགས་པ་ལ་ཡང་ནའོ། །མ་ནིང་ལས་སོ། །ཛས་ཤས་དག་གི་ཤིའོ། །ཡམ་མིན་པའི་ཤེ་ལའོ། ་་། ཁར་ལ་ཛྷལའི་ཙརའོ། །།རྣམ་པར་གཅད་པའོ། །བྷ་ཝད༑ཌ་བྷ་ཝ་ཏཱི། བྷ་ཝནྟི། བྷ་ཝཏ། ་བྷ་ཝ་ཏཱི། ཤྩབྷ་ཝནྟིཡེ། གསུམ་པ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཀིམ་ཞེས་པ་འདིའི། །ཀིམའི་ཀའོ་ཞེས་པ་ཀ་ཡིག་དག་ཏུ་བསྒྱུར་རོ། །ལྷག་མ་རྣམས་སརྦ་བཞིན་ནོ། །།ཀ། ཀོ། ཀི༈། ཀཾ། ཀོ། ཀཱན། ཀེ་ན། ཀཱ་ཧཱུཾ། ཀི། །ཀསྨི། ་ཀ་ཧཱུཾ། ཀེ་ཧྥུ། ཀ་སྨཏ། ཀཱ་ཧཱུཾ། ཀེ་ཤྩ། ཀ་སྱ། ཀ་ཡོ། ཀེཊཱ། ཀ་ྋསྨིན། །ཀ་ཡོ། །ཀེ། བུད་མེད་ལ། ཀཱ། ཀེ། ཀཱ། །ཧཱུཾ། ཀེ། །ཀཱ། ཀ་ཡཱ། <ཀཱ་ཧཱུཾ། ལན་ཀཱ་བྷི། ཀ་ཀྱེེ། སྐྱུང་ཀཱ་ཧཱུཾ། ཞེ་ཀཱ་ཤྩ། ཀཱ་སྣུ༑ཀཱ་ཧཱུཾ། ་པཀཱ་ཤྩ། ་ཀཱ་y། །འི་ཀཱིཡོ། ཧེཀཱ་སིཾ། ཊ་ཀཱ་ཧཱུཾ། །ལོ་ཀ་ཡོ། ་ཀཱ་སུ། །མ་ནིང་ལ། ཕྱིས་པ་མིན་པ་ལའོ། །ཀཾ། ཀེ༑ཀཱ་ནི། ཀཾ། ཀེ། ཀཱ་ནི། ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །རྣམ་དབྱེའི་ཚིག་ལེའུར་བྱས་པ་རྫོགས་སོ། །ལ་ཁོp། །@འོ། །ཏོག་གཟུང་བ། སྤྱི་དེབ་:::ཡོན་ཏན་མཐའ་ཡས་པ་དང་ལྡན་པ་བླ་མ་དམ་པ་ཆོས་ཀྱི་རྒྱལ་པོའི་བཀའ་ལུང་གིས། བྷོ་ཊ་པཎྜི་ཏས་ེབསྐུལ་བ་བཞིན་དུ་བལ་པོའི་པཎྜི་ཏ། །དཔལ་ས་དབང་བཟང་པོའི་ཞལ་སྔ་ནས། ་བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཤོང་བློ་བརྟན་གྱིས་བལ་པོའི་མཐིལ་དུ་བསྒྱུར་བའོ་། :། འདི་ཡིས་འགྲོ་བ་ཐམས་ཅད་ལ། །གསལ་བའི་ཚིག་དང་དགོས་པའི་དོན། །བདེ་བའི་ལྕེ་དང་སྙན་པའི་ངག། བདེ་བླག་།
