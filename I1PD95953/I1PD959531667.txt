བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིཟའདབ་དྷ་ནུ་རཱཎྜི་ཀ། བསྣ་ཀྲ་ཡ། བི་ཀྲ་ཡ་ལས་ཋནའོ། །དེ་ཡིས་འཚོ་བའི་དོན་:འདི་ལ་འགྱུར་ཏེ། བསྣ་ཡིས་འཚོ་བར་བྱེད་པ་ནི། །བརྨི་ཀ། ཀྲ་ཡི་ཀ། བི་ཀྲ་ཡི་ཀ། ཀྲ་ཡི། ཨཱ་ཡུ་དྷ་ལས་ཚའང་ངོ༌། །ཨཱ་ཡུ་དྷའི་སྒྲ་ལས་ཚ་འགྱུར་ཏེ། འང་ཡིག་ལས་ཋནའོ། །མཚོན་ཆས་འཚོ་བ་ནི། ཨཱ་ཡུ་དྷི་ཀ། ཨཱ་ཡུ་དྷི་ཡ། བཱ་ཏ་ལས་ཁཉའོ། །རིགས་སྣ་ཚོགས་པའི་མ་ངེས་པའི་སྤྱོད་པ་་དང་ལུས་ཀྱི་སྒྱུ་མ་དང་འཚོ་བའི་འདུས་པ་གང་ཡིན་པ་དེ་རྣམས་ནི་བཱ་ཏ་སྟེ་དེ་་་དང་འབྲེལ་པ་ཡང་བྲ་ོཏའོ། །དེས་འཚོ་བའི་དོན་ལ་ཁཉ་འགྱུར་ཏེ། འ་བརྟུལ་ཞུགས་ཀྱིས་འཚོ་བ་ནི། བཱ་ཏཱི་:ན། ཧ་རི་ཏི་ལ་ཨུཏྟཾ་ཅིག་སོགས་ལས་སོ། །ཨུཏྠཾ་ག་སོགས་ལས་བྱེད་པ་ལས་འཕྲོག་པར་བྱེད་པའི་དོན་ཅིཋཀ་འགྱུར་ཏེ། པང་པས་འཕྲོག་པ་ནི། ཨོཏྟཾགི་ཀ། ཨོ་འུ་པི་:ཀ། བྷ་རི་ཀ། །ནཱ་བི་ཀ། ཨུཏྤངྒ་། ཨུ་ཊུ་པ། བྷཱ་ར་ནོ། ཨུཏྟ་ོཔ། ཨུཏྤ་ཨིཊ་བི་ཊ་ཀ། པི་ཊཱ་ཀ། བྷས་སོགས་ལས་ཥྛནའོ། །བྱེད་པ་ལས་འཕྲོག་པའི་དོན་ལ་འགྱུར་ཏེ། བྷ་སྔ་ཡིས་འཕྲོག་པ་ནི། བྷ་པཱི་:ཀཱི། བྷསི་ཀི༄༅། བྷ་ར་ཊི་ཀ་ོ། བྷ་ར་ཊི་ཀཱི། བྷ་སྠཱ། བྷ་ར་ཊ། བྷ་ར་ཎ། ཤིནྟ་བྷ་ར། ཨོཾ་ཤ་བྷཱ་ར། བི་བ་བྷཚེ། བཱི་བ་དྷ་ལས་ཡང་ནའོ། །གསུམ་པའི་མཐའ་ལས་འཕྲོག་པའི་དོན་ལ་སྐྲུན་འགྱུར་ཏེ། ཕྱོགས་གཅིག་ཙལ་ཋཀའོ། ་། བི་བ་དྷ་ཡིས་འཕྲོག་པ་ནི། བི་བ་དྷི་ཀ། བི་བ་དྷི་ཀཱི། :：བེ་བ་དྷ་ཀི། བི་བ་དྷི་ཀཱི། བི་བ་དྷ་དང་བ་ཅིབ་དྷའི་སྒྲ་དག་དོན་མཚུངས་པ་ཉེ་བར་སྡུད་པ་འཇུག་གོ། །ཀུ་ཊི་ལ་ཀཱ་ལས་ཨཎའོ༑་་། བྱེད་པ་ལས་འཕྲོག་པའི་དོན་ལ་འགྱུར་ཏེ། གྱ་གྱུའི་འགྲོས་ཀྱིས་རྔོན་པ་འཕྲོག་པར་བྱེད་པ་ནི། ཀོ་:ཀྲི་ལི་ཀ༄༅། །རི་དྭགས་སོ། །།::ཀོ་ཊི་ལི་ཀཱིསི། ཚོགས་ཅན་མའོ། །ཨཀྵ་རླུ་ཏ་ལས་ཟིན་པ་ལའོ། །བྱེད་པ་ལ་ཋཀ་ཙེའགྱུར་ཏེ། །
