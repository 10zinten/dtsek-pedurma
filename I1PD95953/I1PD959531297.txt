བསྟན་འགྱུར། སྒྲ་མདོ། རེ། རྣམས་ལ་ཌ་ཕ་རོལ་དང་ཚིག་ཕྲད་མཐའོ། ལ། དེའི་ཚིག་ཕྲད་ལས་ཨཏ་ཨེང་དགག་གོ༑། ཀུཏྟཾ། མེ་ཐབ། མུཏྟཾ། མགོ་ཟླུམ། ཛུཎྜི?། ནགས་ཚལ། །ཏུཏྟཾ། མཚུལ་པའམ་མཆུ། དེ་བཞིན་དུ་གཞན་ཡང་བལྟ་བར་བྱའོ། །ཤ་མི་ལས་ཟྭའོ། །ཤ་མུ་ད་མུ་ཉེ་བར་ཞི་བ་ལའོ། །ཞྭ་ཕ་རོལ་རྗེས་སུ་ང་རོ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །ཤམྦྷ༑ཡལྷའི་ཁྱུ་མཆོག་། ཤ་ཀི་ལས་ཨནྟའོ། །ཤྲཱི་ནུས་པ་ལའོ། རྗེ། ཨུནྟ་ཕ་རོལ། གདོང་པ། ཤ་ཀུནྟ། བྱའོ། །ཛྲ་བི་ཤ་དག་ལས་ཨནྟ་ཙའོ། །ཛྲ་རྒས་པ་ལའོ། །བི་ཤ་རབ་ཏུ་འཇུག་པ་ལའོ། །།འདི་དག་ལས་ཨནྟ་ཙ་ཕ་རོལ། །ཡ། ཨཏ་ཨེང་དང་ར་ཕ་རོལ་ཉིད་དོ། །ཛ་རནྟ། རྒས་པ། ཡང་བ་མཐའ་ཉེ་ཞེས་པས་ཨེ་ཉིད་དོ། །བེ་ཤནྟ། ཕྱེ་མ་ལེབ། རུ་ཧི། ནནྡིཉྩི། ཛཱི་བ་ལས་ཥི་ཏའོ། །རུ་ཧ་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཏུ་ཎ་དི་ཀུན་དུ་འབྱོར་པ་ལའོ། །ཛི་བ་སྲོག་འཛིན་པ་ལའོ། །འདི་རྣམས་ལས་ཨནྟ་ཙ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་ཊི་ཏ་བཞིན་ནོ། །ཡང་བ་མཐའ་ཉེའི་་ཞེས་པས་ཨོ་ཉིད། །རོ་ཧཡཤིང༌། ཥིཏ་ལས་དྲི་ཥའོ། །ཞེས་པས་དྲི་ཥའོ། །ང་ཡིག་དཱི་ལའོ་ཞེས་པའི་རྟགས་དོན། ཥ་ཡིག་ནི་དཱིཔ་དང་དཱི་ཥ་ཞེས་པ་ཐུན་མོང་དུ་བཟུང་བའི་དོན་ཏོ། །དྲི་ལའོ་ཞེས་པས་ཨ་ཡིག་དབྱིའོ། །ནནྡནྟི། །ཡགྲོགས་མོ། ཛཱི་བནྟཱི། སྨན། ྋ་། ཛི། བ་སི། །བ་ཧི། །སཱ་དྷི། བ་སི། བྷཱ་སི། ག་ཊི། མཎྜི། ཧེ་མི་རྣམས་ལས་སོ། །བྷུ་ཡོད་པ་ལའོ། །ཇི་རྒྱལ་བ་ལའོ། །བ་ས་ངེས་པར་གནས་པ་ལའོ། །བ་ཧ་ཐོབ་པ་ལའོ། །རཱ་དྷ། སཱ་དྷ༑ཡང་དག་པར་གྲུབ་པ་ལའོ། །ཏུ་བཔ་ས་བོན་འདེབས་པ་ལའོ། །།བྷཱ་སྲི་འབར་བ་ལའོ། །ག་ཌ་འཁྲུད་པ་ལའོ། །།།ཙུ་ར་སོགས་ལས་ཎི་ཙའོ། །ཥ་ཧ་བཟོད་པ་ལའོ། །མ་ཌི་རྒྱན་ལའོ། །སྦྱོར་བ་པོའི་བྱ་བ་ལ་ཎི་ཙའོ། 《བ། ཙ་ཡིག་དབྱངས་དོན། ཎ་ཡིག་ཊྛིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། ་ཧེ་མ་ཞེས་པ་།
