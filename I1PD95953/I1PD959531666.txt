གཉིས་པ་ལས་ཋནའོ། །ནོའི་སྒྲ་ལས་ཨ་ཙ་གཉིས་པའི་ཨངྒ་ལས་སྒྲོལ་བའི་དོན་ལ་བྱེད་པ་ལས་ཋན་འགྱུར་ཏེ། གྲུ་ཡིས་སྒྲོལ་ལོ་ལ། ནཱ་བི་ཀ། གྷ་ཊི་ཀ། བླ་བི་ཀ། བཱ་ཧུ་ཀ། ཏཱིརྫི་ཀ། །བུད་མེད་ལ། ནཱ་བི་:ཀྲ། །ཏཱིརྨི་ཀཱ་སྤྱོད་དོ། །མངོན་པར་རྗོད་པ་ལ་བྱེད་པ་ལས་ཋཀ་འགྱུར་ཏེ། སྤྱོད་པ་ནི་ཟ་་བ་དང་འགྲོ་བ་ལ་འཇུག་གོ། །ཞོ་ཡིས་སྤྱོད་པ་ནི། དྭ་དྷི་ཀ། ་ཤངྒ་བེ་རི་ཀ་ལྔ། རྟས་སྤྱོད་པ་ནི། ་་་:ཨ་ཤཱི་ཀི། ཀ་ར་ཎ་ལས་ཞེས་པ་ཉིད་དེ། ས་ཧ་པུ་ཏྲེ་ཎཙ་ར་ཏི། ཅི་ལྟར་འོ་ན་:ག་ཊི་ཀ། ཅི་ཞེ་ན། :བྱེད་པའི་དོན་དུ་བརྗོད་པར་འདོད་པ་ལས་འགྱུར་རོ། །པརྦ་སོགས་ལས་ཥྛནའོ། །བྱེད་པ་ལས་སྤྱོད་པའི་དོན་དུ་འགྱུར་ཏེ། པ་ཀ། པཆི་ཀཱི། ཨ་ཤཱི་ཀཾ?། །ཨ་ཤཱི་ཀཱི?། ཨ་ཀ་ཎི་ཀ། cཨ་ཀ་ཎི་ཀཱི། ཨཱ་:ཀ་ཥ་ཞེས་བྱ་བ་ནི་གསེར་བརྟགས་པའི་དོན་དུ་ྋརྡོལ་བ་དར་བ་ལ་བརྗོད་དོ། །རྐང་པ་དག་གིས་སྤྱོད་པ་ནི༑བ་དི་ཀཱི། པ་དི་ཀཱི། ཧི་མ་ཧ་ཏི་ཀཱ་ཤིཥྛན་ལ་པ་ད་:ཞེས་པས་དཱ་ན་ཡུལ་ལོ། །པཉྫ་ཨ་ཤཱ། ཨ་ཤཱཏྠ། ར་ཐ། ཛཱ་ལ། བྱཱ་ས། བྱཱ་ལ། པཱ་ད། ཨཱ་ཀ་ཥ། ་་ཤ་ག་ཎ་ལས་ཡང་ནའོ། །།སྤྱོད་པར་བྱེད་པའི་དོན་ལ་ཊྛན་ཡང་ན་འགྱུར་ཏེ། །ཤཱ་ཙེག་ཎི་ཀ། ཤཱ་ག་ཎི་ཀཱི། །ཤཱ་ག་ཎི:;། ཤཱ་ག་་ཎི་ཀཱ། །བེ་ཏ་ན་སོགས་ལས་འཚོ་བར་བྱེད་དོ། །བྱེད་པ་ལས་ཨིཋཀ་འགྱུར་ཏེ། །བེ་ཏ་ན་ཡིས་འཚོ་བར་བྱེད་པ་ནི། །བི་ཏ་ནི་ཀ། བཱ་ཧི་ཀ། ་བེ་ཏ་ན། བཱ་ཧ། ཨཌྷ་བཱ་ཧ། ་བི་ཤ། ཨུ་པ་བེ་ཤ། སྲེ་ཥ་ཎ། ཎསུ་ཁ། ཤ་ཡ། ཨུ་པ་དེ་ཤ། ཨུ་པ་བཱ་ཧ། ཨུ་པ་ཧསྟ། ཨུ་པ་སྦྲ་ན། ་ཨུ་པ་སྒྲི། བཱ་ད། །བཎ༑ཤཀྟི། ཨུ་པ་ནི་ཥད། སྤྱི་ཀ། ཛཱ་ལ། །བཱ་ལ། དྷ་ནུ་དང་དཎྜ་ལས་འདུས་པ་རྣམ་པར་གཟུང་བའི་ཕྱིར་རོ། །དྷ་ནུཥ། །དཎྜི་ཀ། —།
