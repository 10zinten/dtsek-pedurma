བསྟན་འགྱུར། སྒྲ་མདོ། རེཨཱུའི་ཞེས་པ་ཐུང་ངུའོ། །ཨེང་ཐུང་དུ་ལས་ཏ་མ་ཡིན་པའི་:ཀུན་དུ་བོད་པ་ལའོ། །རྣམ་པར་གཅད་པའོ། །བ་དྷཱ། བདྷོ། བ་དྷཱི། བ་ནཱཾ། །བ་དྷོ། བ་དྷུ། བ་:དྷཱ༑། བ་དུ་། བ་དྷུ་བྷི། བ་དྷཱི། བ་དྷུ་བྷཾ། །བ་དྷུ་བྷྱ། བ་དྷཱ། ་བ་དྷུ་བྷཾ། བ་དྷུ་བྷུ། བ་དྷཱ། བ་དྷོ། བ་དྷཱ་ནཾ། བ་དྷཱི། བ་དྷོ། བ་དྷུ་ཥཱ། ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། དེ་ལྟར། ཀ་དུ། ཀ་དྲོ། །ཀ་དྭ::། ཤྲཱི་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། །ཀླུ་བྱིངས་བྷཱུ་རྣམས་ཀྱི་ཨཱི་ཨཱུ་དག་གི་ཨི་ཡ་ཨུཝ་དག་གོ་ཞེས་པ་ཨི་ཡ་རུ་བསྒྱུར་རོ། །ལོ། དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །།ཨེཙ་ལའོ། །ཨཀའི་རིང་པོ་་ཨཀ་ལའོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་:ནུཊ་ཡང་ནའོ། *། ཚིག་གཅིག་་ལ་རཱ་ཥ་ལས་ནའི་ཎའོ། །ངའི་ཨཱམ་དེ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །རྣམ་པར་:གཅད་པའོ། །།ེཝི། ཤི་ཡོ། ཤི་ཡ། ཤི་ཡཾ། ཤི་ཡོ། ཤི་ཡ། ཤི་ཡཱ། ཤྲཱི་ཧཱུཾ། ་ཤྲཱི་བྷི། ཤི་ཡི། ཤི་ཡེ། ཤྲཱི་ཧཱུཾ། ཤྲཱི་བྷུ། ཤི་ཡཱ། ཤི་ཡ། ་ཤྲཱི་ཧཱུཾ། ཤྲཱི་བྷུ། ཤི་ཡི། ཤི་ཡཱི། ཤི་ཡོ། ཤི་ནཾ། ཤི་ཡཾ། ཤི་ཡཾ། ་ཤི་ཡི༄༅། ཤི་ཡོ། ཤྲཱི་ཥཱ། ཧེ་བྷི། ཧེ་ཤི་ཡོ། ཧེ་བྷི་ཡ། དེ་ལྟར། སི། སྦི་ཡོ། སྦི་ཡ། ་སུ་དྷཱི་ཞེས་པ་འདིའི་ཤཱིའི་སྒྲ་བཞིན་ནོ། །སུ་དྷཱི། སུདྷི་ཡོ། སུ་དྷི་ཡ། ་སུ་དྷི་ཡཾ། སུ་དྷི་ཡོ། སུ་དྷི་ཡ། བྷཱུ་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། །།།ཟླ་བྱིངས་བྷཱུ་རྣམས་ཀྱི་ཨིཡ་ཨུཝ་དག་གོ་ཞེས་པ་ཨུ་ཝ་རུ་བསྒྱུར་རོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །ཨེཙ་ལའོ། །ཨཀའི་ཨཀ་ལ་རིང་པོའོ། །དེ་ཨིཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་ནུཊ་ཡང་ནའོ། །ཚིག་གཅིག་ལ་ར་ཥ་ལས་ནའི་་་ཎའོ༑། དེ་ལ་ཇིའི་ཨཱམའོ། །ཨཀའི་ཨཀ་ལ་རིང་པོའོ། །རྣམ་པར་གཅད་ཀྱི་།
