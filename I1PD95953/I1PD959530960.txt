།བརྩི་ས་ལས་སོ། ་། སུ་ཏྲ་ལས་ཚ་ཡང་ངོ༌། །པྲི་ཐི་བི་དང་སརྦ་བྷུ་མི་ལས་ཨཎཱ་དང་ཉ་དག་གོ། །དབང་ཕྱུག་ལའོ། །དེར་རྟོགས་པ་ལའོ། །ལོ་ཀ་དང་སརྦ་ལོ་ཀ་ལས་སོ། །དེ་དེར་འདིར་ལ་འཕེལ་བ་དང་འདུ་བ་དང་རྙེད་པ་དང་ཁྲལ་་་དང་ཨུཔྲ་དང་སྦྱིན་ཏོ། །གང་བ་ལས་ཨངྷ་ལས་ཥཱནའོ། །བྷ་ག་ལས་ཀྱང་ངོ༌། །དེ་འདིའི་ཡོངས་སུ་འཇལ་བའོ། །པཉྩ་དང་ད་ཤ་ད་སྟེ་ལ་རུང་ངོ༌། །ཚོགས་པ་ལ་ཌཊའོ། །དྲི་ཤད་དང་ཙཏཱ་རི་ཤ་དག་ལས་བྲམ་ཟའི་མིང་ལ་དྷཎཱའོ། །།བླ་ངན་དང་གོས་དང་ཆས་རྣམས་སོ། །།དེ་འཚེད་པ་བྲེ་བོ་ཆེ་ལས་ཡང་ངོ༌། །ཡིན་པ་དང་སྟོབས་པ་དང་ཡང་ངོ༌། །།པཏྲ་དང་ཨིརྩ་ཏ་དང་ཨཱཌྷ་ཀ་ལ་ཁ་རུང་ངོ༌། །གྲངས་དང་པོ་ལས་ཥཱན་མིང་ངོ་། །བེ་ཤི་ལ་སོགས་པ་རྣམས་ལ་སྟོར་བ་དང་། ཁུར་བ་དང་ཁོར་བ་ཁུར་བུ་ལས་སོ། །དྲ་བྱ་དང་བསྣ་ལས་ཀྲན་དང་ཟན་དག་གོ། འོས་པའོ། ལ། ཙྪེ་ད་ལ་སོགས་པ་རྣམས་ལས་རྟག་པར་རོ། ་། ཤིནྟ་ཙྪེ་ད་ལས་ཡཏྲ་ཡང་ངོ་། །ཡཛྙཱ་ལས་གྷའོ། །པྲ་དྲ་ལས་ཡང་ངོ༌། །།དཎྜ་ལ་སོགས་པ་རྣམས་ལས་སོ། །དཀྵཎ་དང་ཀཌངྒ་ར་དང་སྦྲུ་ལི་བ་ལ་ལས་ཚ་རུང་ངོ༌། ལ་། ཨརྟི་ཛིནའོ༑། མི་ཕོད་པ་དོན་མིན་དག་ལས་ཥཱ་ལིན་དང་ཀོ་པིན་དག་གོ། །།པར་ཡཏྲ་དང་རུར་ཡཏྲ་དང་ཙནྟ་དང་ཡ་ནར་འཇུག་པར་བྱེད་དོ། །ས་ཤ་ཡར་ཐོབ་པའོ༑། ཡོ་ཛ་ནར་འགྲོའོ། །ཀྲོ་ཤ་དང་ཡོ་ཛ་ན་དང་པོ་ལས་ཤ་ཏ་ལས་མངོན་པར་འགྲོ་བ་ལ་ཡང་ངོ༌། །བ་པཱ་ལས་སོ། །ཎ་པནྡྲ་ཡང་རྟག་པར་རོ། །ཨ་ཛ་དང་སཾ་ཀུཊ་ཏ་ར་དང་བ་རི་དང་ཛངྒ་ལ་ཀནྟར་དང་པོ་ཡིས་སྟེར་བ་ལ་ཡང་ངོ༌། །སྦྲ་ལ་སོགས་པ་ཡིས་སོ། །མ་དྷུ་ཀ་དང་མ་རི་ཙ་དགལ་ཨནའོ༄༅། །དུས་ལས་སོ། །དེ་ཟིན་པའོ། །།དེའི་ཕྱིར་གང་ལ་དང་ཞུས་པའོ། །དེར་གྱུར་པ་དང་མ་གྱུར་་་་པའོ༑། མཱ་ས་ལས་ག་ཚོད་ལ་ཡཏྲ་དང་བཏྲ་དག་གོ། །གྲང་པོ་ལས་ཡང་ངོའོ། །—།
