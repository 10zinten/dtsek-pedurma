བསྟན་འགྱུར། སྒྲ་མདོ། ༼རི?ཨིནྡྲ་བ་རུ་ཎ་བྷ་བ་ཤརྔ་རུ་:དྲ་མི་ཌ་ོརྣམས་ཀྱི་:ཨ་ནུཀ་ཡང་ངོ༌། །ལ་ཊ། ཨཱ་ནུཀ་ཡང་ཞེས་པ་ཡ་བ་ན་ལས་ལི་པི་ལའོ་ཞེས་པ་ཇི་སྲིད་པར་སྤྱིར་བཏང་བའོ། །ཀྟ་་ལས་ཆུང་དུ་བརྗོད་པ་ལའོ༄༅། །ལྟ་ལས་ཞེས་པ་ཇི་སྲིད་པར་སྤྱིར་བཏང་བའོ། །རིགས་ལས་ཨཱཙྪ་ད་མ་ཡིན་པ་ལས་ཡང་ནའོ། །ཡང་ན་ཞེས་པ་པུཙྪ་ལས་སོ་ཞེས་པ་ཇི་སྲིད་པར་སྤྱིར་བཏང་བའོ། །རང་གི་ཡན་ལག་ལས་གཙོ་བོ་མ་ཡིན་པ་ལས་སོ་ཞེས་པ་ས་ཁྲི་ཨ་ཤཱི་ཧོའོ་ཞེས་པ་ཇི་སྲིད་པར་སྤྱིར་བཏང་བའོ། །དི་བ་ཡཛཱི་སོ་ཙི་་དྲྀ་ཀྱི་:སཱརྱ་མུ་གྲི་ཀཱནྡྷེ་བི་དྷི་རྣམས་ཀྱི་ཡང་ནའོ། །ཡང་ནའི་སྒྲ་འདི་ཉིད་སུཔ་ལ་འཇུག་པ་འགོག་བྱེད་མེད་ན་ནི། །རྟེན་ཐ་དད་པ་དྲུག་པ་ལས་ཏསའོ་ཞེས་པ་ལས་་གོང་དུ་སྤྱིར་བཏང་བའོ། །གཉིས་པའི་རྐང་པ་གསུམ་པ་རྫོགས་སོ། །ཏིང་ཡེ། །འགལ་བ་ག་ལའི་ཡ་སཱི་བ་ལ་སོགས་གང་། །བློEལས་ལ་། ཛི་ཏ་ལས་གོང་དུ་ཨཎའོ། །དེའི་བུ་རྒྱུད་དོ་ཞེས་པ་རྐང་པ་ཡོངས་སུ་རྫོགས་པ་ཇི་སྲིད་དུ་སྤྱིར་བཏང་བའོ། །བཱ་ཧུ་ལ་སོགས་པ་རྣམས་ལས་ཆེརིགས་ཀྱི་དང་པོ་རྣམས་ལས་སོ། །རིགས་ཀྱི་དང་པོ་རྣམས་ལས་ཞེས་པ། ཀཏྲ་ོལས་ཞེས་པ་ལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །སིཏ་སམྦྷ་བེ་བེ་ཤེ་ཥ་ཎཾ། དྲང་སྲོང་གི་བུའི་བུ་ལ་སོགས་པ་ལའོ། །བུའི་བུ་ལ་སོགས་པ་ཞེས་པ་དྲོཎ་ེལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །གརྒ་ལ་སོགས་པ་རྣམས་ལས་:ཡཉའོ། །ཡཉ་ཞེས་པ་ཕཉ་ལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །བུད་མེད་མང་པོ་རྣམས་ལས་:ཕཀའོ། །ཕཀ་ཞེས་པ་ཨཎ་ལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །ཤི་བ་སྐྱེལ་སོགས་པ་རྣམས་ལས་:ཨཎའོ། །ཨཎ་ཞེས་པ་རྙཀ་ལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །ཆ་ཀའོ་ཞེས་པ་:ཙཊ་ཀ་གྱིལས་ཨི་རཀའོ་ཞེས་པ་ལས་གོང་དུ་སྤྱིར་བཏང་བའོ། །རེ་བ་ཏཱི་ཨེལ་སོགས་པ་་་རྣམས་ལས་:ཋཀའོ། །ཋཀ་ཅེས་སེཔ་ཕི་ལས་ཆ་ཡང་ངོ་ཞེས་པ་ལས་གོང་དུ་སྤྱིར་—།
