ནས་བསྟན་འགྱུར། སྒྲ་མདོ། །་རི་ཆ་ཡོན། ལས་ཀྱཔ་འགྱུར་ཏེ། སོགས་པར་བྱེད་པས་ན་ཙི་:ཏྱ་སྟེ་མེའོ། ་། མེ་སོག་པས་་ཨགྣི་ཙི་ཏུའོ། །རཾ་དང་ཧལ་ལས་ཙྪཏའོ། །རི་ཡིག་དང་ཧལ་མཐའི་བྱིངས་་་ལས་ཅུཏ་འགྱུར་ཏེ། ཀཱéཔ། །:·ཧཱམཾ་པཱཧུཾ། ཞེ་པཱལཾ། ཇི་ལྟར་ལ་ཨ་བྷུ་ཅིས་་མ་སྱ་ཞེ་ན། མིང་སྔོན་མའི་ཆོ་ག་མི་རྟག་གོ་ཞེས་པས་འཕེལ་བ་མེད་པའི་ཕྱིར་རོ། །པཱ་ཎི་དང་། སཾ་ཨ་བ་དག་ལས་སྲིཛ་ལས་སོ། །པཱ་ཎི་ྋསྔོན་མ་དང་སཾ་དང་ཨ་མ་སྔོན་མའི་སྲོཛ་ལེལས་ཀྱང་ཅུར་འགྱུར་ཏེ། ོཔཱ་ཎི་:ཧཱུཾ་སྔཏྠ་བྱ། པཱ་ཎི་སརྒྱུ་ར། ས་མ་བ་སརྒྱུ། པཱ་ཎི་དང་སམ་ཨ་བ་དག་ལ་ཞེས་པ་ཅི་ཞེ་ན། སྲི་ཚིགས་སྟེ། རྀཏ་མཐའ་ཉེ་ལས་:ཀྱི་པའི་པྲི་ཏྟཱཏ་མ་ཡིན་པ་ལས་ཞེས་པས་ཀྱཔའོ༄༅། །ཨུ་ལས་གདོན་མི་ཟ་བ་ལའོ། །ངེས་པའི་དངོས་པོ་ནི་ཨ་པ་བྱ་ཀཾ་སྟེ་དེ་རྟོགས་པར་བྱེད་པ་ལ་ཨུའི་ཡི་གེའི་མཐའ་ལས་ཅུ་འགྱུར་ཏེ། ལོ་ལཱ་པཾ། ཨ་བ་བྱ་ལཱ་པཾ། གཞན་ལ་ནི་ལ་ཧུཾ། ཨ་སུ། ཡུ། བ་པ། ::ར་པི་ལ་པི། ཏྲ་པི། ད་མི། ཙ་མི། ད་བྷ་ལས་སོ། །དགུ་པོ་འདི་རྣམས་ལ་ཙྪཏ་འགྱུར་ཏེ། །ཡཏ་ཀྱི་དམིགས་བསལ་ལོ། །ཨཱ་སཱ་ཏྱཾ། ཡཱ་ཏྱཾ། བཱ་སྱེ། རཱ་སྱ། ལཱ་སྱ༄༅། ཨ་པ་ཏཱ་ཧཱུཾ། ནཱ་ཏྱཾ། ཨཱ་ཙཱ་ཏྱཾ། ཨ་བ་དཱ་ཧཱུཾ། ཨ་མ་མྱོབ་ས་ལས་གཞི་སྱེལ་ཡང་ནའོ༑། །ཨ་མ་སྔོན་མའི་བ་ས་ཏི་ལས་གཞི་ལ་ཅུཏ་ཡང་ན་འགྱུར་ཏེ། འདིར་་ཟླ་བ་དང་ཉི་མ་དག་ལྷན་ཅིག་གནས་པ་ནི། ཨ་མ་བཱ་སུཾ་:ཨ་མ་བ་སྙའོས། །༄ཏ་ཨ་པ་བཤད་པ་ལ་གང་གྱུར་པའོ། །པྲ་ཎཱ་སྤྱ། ཡང་དག་པར་མི་འདོད་་པ་ལའོ། །།ཚིག་ཕྲད་དེ། པྲ་ཎཱ་ཡུ། ཆོམ་རྐུན་ཏེ་ཡང་དག་པར་མི་འདོད་་་ཅེས་པའི་དོན་ཏོ། །པྲ་སྔོན་མའི་ན་ཡ་ཏི་ལས་ཎཱཏ་དང་ཨཱ་ཡར་བསྒྱུར་བ་ཚིག་ཕྲད་དོ། །ཡང་དག་པར་མི་འདོད་པ་ཞེས་པ་ཅི་ཞེ་ན། པྲ་ཎེ་ཡ། ལེགས་པའོ། །ཡང་དག་པར་འདོད་པ་ལའང་མཐོང་ངོ༌། །ཛེཊྛཱ༄༅། ཡ་པུ་ཏཱ་ཡ་།
