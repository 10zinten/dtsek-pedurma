རབསྟན་འགྱུར། སྒྲ་མདོ། རེའིཆུང་རྣམས་རབ་ཏུ་སྦྱོར་བ་ན་བྱུག་འདོད་པ་རྟོགས་པར་བྱེད་པའི་འབྱུང་འགྱུར་ལ་ལཊ་ཡང་ན་:འགྱུར་ཏེ། ལབྷཱ་མིཙྪཱ་ལིཔྤེ། དོན་དུ་གཉེར་བ་:ནི་མངོན་པར་འདོད་པའོ། ::། ཀིཾ་བྷ་བནྟོ་བྷོ་ཛ་ཡནྟི། །ཀཾ་བྷ་བནྟོ་བྷོ་ཛ་ཡི་ཥཱནྟི། ཀ་ཏ་རོ་་་་བྷིཀཾ་མེད་དཱ་ཏི། །ཡང་ན་དཱ་སྱ་ཏི། ཀ་ཏོ་མ་བྷིཀྵ་ད་དཱ་ཏི། ཟཡང་ན་དཱ་སྱ་ཏི༑བྱུག་འདོད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཀཔཱ་ཊ་ལི་པུ་ཏྲ་ཡཱ་སྱ་ཏི། འབྲས་བུ་བརྗོད་པ་ལའོ། །བྱུག་པར་བྱེད་པ་གང་གི་འབྲས་བུ་མཐོ་རིས་ལ་སོགས་པ་ཅན་དེ་བརྗོད་པ་འབྱུང་འགྱུར་གྱི་དུས་ལ་ལཊ་ཡང་ན་འགྱུར་ཏེ། ཨ་གི་ལ་་་སོགས་ཀྱི་དོན་ནི་རྩོམ་པའོ། །།ཡོ་:བྷ་གྷོ་ད་དྷ་ཏི་ས་སཱགྷཾ་ཡ་ཏི། ཡོ་བྷཏྟཾ་དཱ་སྱ་ཏི་ས་སཱགྷཾ་ཡཱ་སྱ་ཏི། ་ལོཊའི་དོན་མཚན་ཉིད་དོ། །ལོཊའི་དོན་གྱི་སྒྲུབ་པ་་་སོགས་དེ་མཚོན་:པ་དེ་བྱིངས་ཀྱི་དོན་གང་གིས་ལོཊ་དང་བཅས་པའི་དོན་མཚོན་པ་དེ་བྱིངས་ཀྱི་དོན་ཏེེ། དེ་ལ་འཇུག་པའི་བྱིངས་ལས་འབྱུང་བར་འགྱུར་བའི་དུས་ལ་ཡཊ་ཡང་ན་འགྱུར་ཏེ། ཨུ་པཱ་ཙྪེདྷ་ཡཕྱེ་དཱ་གཙྪ་ཏི། ཨ་ཐ་ཏཱ་:ཙྪརྗོ་དྷི་ཥ། ཨུ་པཱ་དྷ་ཡཉྩེ་དཱ་ག་མི་ཥྛ་ཏི། ་ཨ་ཐ་ཏྭཾ་ྋཙྪརྗོཉ་དྷི་ཥ། མཁན་པོའི་ཐད་དུ་འགྲོ་བ་ཀློག་པའི་ཆོ་གའི་མཚན་ཉིད་དོ། །ཨུཛྷ་མོ་ཧཱནྟི་ཀ་ལ་ལིང་འང་ངོ༌། །ཨུཛྙཱ་དང་མོ་ཧུནྟི་ཀ་ལ་འབྱུང་བར་འགྱུར་བ་ལོཊའི་དོན་མཚོན་པར་བྱེད་པ་ན་བྱིངས་དོན་ལ་ད་ལྟར་བ་ལས་ལིང་ཡང་ན་འགྱུར་ཏེ། འང་ཡིག་ལས་ལཊ་འང་ངོ༌། །ཨུཛྷ་མོ་ཧཱུཏྟཱ་དུ་པཱ་དྷཱ་ཡཉྩེ་དཱ་གཙྪེ་ཏ། ཡང་ན་ཨཱ་གཙྪ་ཏི། :ཨུ་པཱ་དྷ་ཝིཡཉྩེ་དཱ་ག་མི་ཥྛ་ཏི། ཨཱ་གནྟཱ། ཨ་ཐ་ཏྭཾ་བྱཱ་ཀ་ར་ཎ་མ་དྷི་ཥི། མཱང་ལ་ལུངའོ། །མཱང་སྦྱོར་བ་ལ་འབྱུང་འགྱུར་གྱི་དོན་ལ་ད་ལྟར་བ་ལས་ལུང་འགྱུར་ཏེ། མཱ་ཀཱརཱི། ང་ཡིག་ནི་ཀིཏའོ། །མཱ་ཀ་རོ་ཨེཏུ། མཱ་ཀ་རི་ཥྛ་ཏི། སྨྲ་ཕ་རོལ་ལ་ལང་:ངོ༌། །མང་ལ་ཞེས་པ་ཉིད་དེ་སྨ་ཕ་རོལ་ལ་མཱང་སྦྱོར་བ་ན་འབྱུང་འགྱུར་གྱི་དོན་ལ་ད་་་།
