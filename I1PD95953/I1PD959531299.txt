བསྟན་འགྱུར། སྒྲ་མདོ། *རིཛི་རྒྱལ་བ་ལའོ། །།ཨརྦ་པརྦ་འགྲོ་བ་ལའོ། །ཡ་ཛ་ལྷ་མཆོད་པ་ལའོ། །ས་ཙ་་གསལ་པོར་བྱེད་པ་ལའོ། །།ཡ་མུ་ཉེ་བར་དགའ་བ་ལའོ། །ཎ་མ་རབ་ཏུ་འདུད་པ་དང་སྒྲ་ལའོ། །ཎའི་ནའོ། །ཧ་ཡ། ཧརླ་འགྲོ་བ་ལའོ། །ཁ་ལ་ཀུན་དུ་་་གཡོ་བ་ལའོ། །འདི་རྣམས་ལ་ཨ་ཏ་ཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །།ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །བྷ་ར་ཏ། གར། །དན་ཏ། ཟླ་བ། ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཉིད་དང༌། ཨ་ཡ་རུ་བསྒྱུར་རོ། །།ཛ་ཡ་ཏ། མེ། པརྦ་ཏ། ཡརི་བོ། :པ་ཙ་ཏ། བདེ་བའི་ཡོ་བྱད། ཛ་ཡ་ཏ། མེ། ཡ་མ་ཏ༑ཡནང༌། ན་མ་ཏི། ཡཆུག་པ། ཧ་ཡ་ཏ། གནོད་སྦྱིན། ཁ་ལ་ཏ། །སྐྱེ་བོ་ངན་པ། སྲི་ཊི་རཉྫི་དག་ལས་ཀིཏའོ། །སྲྀ་ཥཱ་པྲྀ་ཥཱ་མྲ་གུ་བཀྲུ་བ་ལའོ། ་། རཉྫ་ཚོས་པ་ལའོ། །འདི་དག་ལས་ཨཏ་ཙ་ཕ་རོལ་ཏུ་འགྱུར་ཏེ། ཀིཏ་བཞིན་ནོ། །ཀིཏ་ཀྱི་ཕྱིར་ཨཏ་ཨེང་མེད་དོ། །སྲྀ་ཥ་ཏ། ཀིཏ་ལའོ། །ཞེས་པས་རཉྫའི་ན་དབྱིའོ། །།ར་ཛ་ཏ། དངུལ། །ཌམྲ། ཤྲཱི། བཱ། ཧ་སི། ཎ་མི། ་ད་མི། །་ལུ། པཱུ། བྷཇི་རྣམས་ལས་ཏནའོ། །ཚིགསང་སྲོག་གཏོང་བ་ལའོ། །གི་ངེས་པར་གནོན་པ་་་ལའོ༑། ཨི་ཎ་བི་བཱ་འགྲོ་བ་ལའོ། །ཧ་ཡེ་དགོད་པ་ལའོ། །ཨི་ཎ་འགྲོ་བ་ལའོ། །ཨ་མ་ནད་ལའོ། །ཤ་མུ་ད་མུ་ཉེ་བར་ཞི་བ་ལའོ། །ལུ་གཅོད་པ་ལའོ། །པུང་དག་བྱེད་པའོ། །དྷུའྲི་འཚེ་བ་ལའོ། །འདི་རྣམས་ལས་ཏན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨཏ་ཨེང་དང་ར་ཕ་རོལ་ཉིད་དོ། །མརྟ། མི་ཡུལ་ལམ་འཇིག་རྟེན། གཏྟ། ཡདོང༌། བཱ་ཏ། རླུང་། ཧསྟ། ལག་པ། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཉིད་དོ། །ཨེ་ཏ༑གྱར། །ཧལ་ལ་མའི་འོ་ཞེས་པས་རྗེས་སུ་ང་རོ། རྗེས་སུ་ར་རོ་ནི་ཡ་ཡ་ལ་ཡམའོ་ཞེས་སྡེ་པ་སོ་སོ་དང་ཉེ་བའི་ན་ཡིག་གོ། ་། ཨནྟ། མཐའ། དནྟ། སོ། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨོ་ཉིད་དོ། །ལོ་ཏ། མཆི་མ་ལྷུང་བ། པོ་ཏ། བྱིས་པ། །
