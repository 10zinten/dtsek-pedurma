ད་ལ་ཧལ་ལ་བ་ཕྱི་བ་མ་ཡིན་པ་ལའོ། །ཤ་ཧི་ལའོ། །ཧ་ནའི་ཛའོ། ::། ལིཊ་་་་དང་ཨི་ཤིར་ལིང་ཏིང་ཤ་ཕྱི་བ་མ་ཡིན་པ་ལའོ། །བྷ་རྫའི་བྷ་རྫ་རུང་ངོ༌། །དབྱི་བ་ཨཏྲའིའོ། །ཡཀྲ་ལའོ། །ཡའི་ཧ་ལ་ལས་སོ། །།ཀྱསྲའི་རུང་ངོ༌། །ཎའི་ཨིཊ་མ་ཡིན་པ་ལའོ། །ཏ་དང་ཏ་བཏྲ་ལ་ལིཊ་ལའོ། །ཨ་ཡ་ཨཱམ་དང་ཨནྟེ་དང་ཨཱ་ལུ་དང་། ལྟ་ཨསྦྱ་དང་ཨུནུ་རྣམས་ལའོ། །ཁྱཔྲ་ལ་ཡ་ང་བ་ལས་སོ། །ཨསྲ་ལས་རུ་བའོ། །ཀླུའི་ཀླུའོ། ཀ། གདམས་ངག་ལ་ཨཙཱ་དང་ཧ་ན་དང་གྲ་ཧ་དང་གྲ་ཤ་རྣམས་ལ་སྱ་དང་སཱི་ཙ་དང་སའི་ཡུཊ་དང་དེ་རྣམས་ཀྱི་བྱ་ཙ་དང་ལས་དག་ལ་ཙི་ཎ་བཞིན་དུ་ཨིཊ་རུང་ངོ་། །དཱི་ངའི་ལིཊ་ལ་ཡུཀྲའོ། །དབྱི་བ་ལ་ཨ་ཙ་ལ་ཀ་ད་ཕྱི་བ་ལ་ཡང་བ་ལ་ཡང་ཨཏྲའོ། །ཨིཏ་ཡཏྲ་ལའོ། །མ་དང་སྦྲ་དང་ས་དང་ག་དང་སི་བྲ་ཧཱཀྲ་དང་དཱ་དང་དྷ་རྣམས་ཀྱི་ཧ་ལ་ལའོ། །ལིང་ལ་ཨེའོ། །སྦྱར་བ་དང་པོའི་སྦྲ་མ་ཡིན་པའིའོ། །ཁྱབྲ་ལ་མ་ཡིན་ནོ། །མེ་དའི་ཨིཏྲ་རུང་ངོ༌། །ལུང་དང་མ་དང་ང་རྣམས་ལ་ཨཊ་མའི་སྦྱར་བ་མ་ཡིན་པ་ལའོ། །ཨཛྲ་ལ་ཟླ་དང་ཁམས་དང་བྲུ་རྣམས་ཀྱི་ཨི་དང་ཨུ་དག་གི་ཨིཡ྄་ཨུལྔ་དག་གོ། །གཉིས་ཉིད་ལ་སྔ་མའི་མ་མཉམ་པ་ལའོ། །སྡིའིའོ། །ཨམ་དང་ཤསྲ་དག་ལ་རུང་ངོ༌། །ཨིཎའི་ཡནྡྲའོ། །ཨའི་སྦྱར་བ་མ་ཡིན་པ་ལས་ཨཛྲ་དུ་མའིའོ། །བྱེད་པ་དང་གྲངས་མེད་པ་ལས་ཨུའི་ཡོང་སུ་པྲ་ལས་སུ་དྷི་མ་ཡིན་པ་ལའོ། །།བ་ཥ་དང་དྲིན་དང་པུ་ན་དང་ཀ་ར་ལས་བྷུའིའོ། །ཧུ་དང་ལྷ་དག་གི་ལིཊ་མ་ཡིན་པ་ལའོ། ::། བྷུའི་བུཀ་ལུང་དང་ལིཊ་དག་ལའོ། །།ཨུཏྲ་གུ་ཧའི་ཨཛྲའིའོ། །དུ་ཥའི་ཎིའོ། །ཙིཏྟ་ལ་རུང་ངོ་། །ཡི་:། གམ་དང་ཛན་དང་གྷས་རྣམས་ཀྱི་ལ་ལ་དབྱི་བ་ཕ་ཕྱི་བ་མ་་་ཡིན་པ་ལའོ། །།ཀ་ཕྱི་བ་ལ་ཡང་ཧ་ནའིའོ། །ཧུ་དང་:ཛྷལ་ལས་ཨིཊ་མ་ཡིན་་་པའི་ཧིའི་དྷིས་སོ། །།ཨཏྲ་ལས་དབྱིའོ། །ཨུཏྲ་ལས་སྦྱར་བ་མ་ཡིན་པ་ལ་ཁམས་།
