བསྟན་འགྱུར། སྒྲ་མདོ། རེཚཐཥཡ་བ་གཱ་ཞེས་པ་འདིར་ཀྲྀཏ་རྐྱེན་འཇིག་པའི་མཚན་ཉིད་ལས་དཱིཔ་མི་འགྱུར་རོ། །ཡི་གེ་ལ་བརྟེན་པའི་སྒྲུབ་པ་རྐྱེན་དབྱི་བའི་མཚན་ཉིད་མེད་པས་བཀག་པའོ། །བུ་རྒྱུད་ལ་ཡཉ་ལས་ཥ་ཨ་བ་ཊ་ལས་མ་ཡིན་ནོ། །།གང་བུ་རྒྱུད་ལ་བྱས་པའི་ཡཉ་དེའི་མཐའ་ལས་བུད་མེད་ལ་དཱིཔ་འགྱུར་ཏེ། གལ་ཏེ་ཥ་ཡིག་དང་ཨ་བ་ཊའི་སྒྲ་ཕ་རོལ་ཏུ་མ་གྱུར་པའོ། །གནཱི། བཱཏི། ཧལ་ལས་དེ་ཕན་གྱི་ཞེས་:པས་ཡང་ན་དབྱིའོ། །བུ་རྒྱུད་ཀྱི་ཡཉ་ལས་ཞེས་པ་ཅི་ཞེ་ན། ་་:དཱི་པེ་བྷ་བཱ་དཱི་པཱིུ། ཥ་ཨ་བ་ཊ་ལས་མ་ཡིན་ཞེས་པ་ཅི་ཞེ་ན། :ཤརྐ་རག། པོ་ཏི་མུ་ཉྩ། ཨ་བ་ཀྱུ། ཥཱ་ཡ་ན་ཡང་ནའོ། །།བུ་རྒྱུད་ཀྱི་ཡཉའི་མཐའ་ལས་བུད་མེད་ལ་ཥཱ་ཡ་ན་ཡང་ན་འགྱུར་ཏེ། ཤཱརྐ་རཱ་གླུ། ཤཱརྐ་རཱ་ག་ཡ་ཎི། པོ་ཏི་མཱ་:;། །པོ་ཏི་མཱ་ཉྩ་ཡ་ཎཱི། ཨཱ་བ་ཁྱུ། ཨཱ་བ་ཁྱུ་ཡ་ནཱི། ཥ་འགྲོ་བ་ཉིད་ལས་རཱིཔའོ། །འགའ་ཞིག་ཏུ་དམིགས་བསལ་གྱི་ཡུལ་ལ་ཡང་སྤྱིར་བཏང་འཇུག་གོ། །དེས་ན། :：གཱཧཱུྃ༑གརྒྱུ་ཡ་ཎྜི། བཱཏི། བཱཉྩ་ཡ་ནི༄༅། །ཤ་ཀ་ལའི་མཐའ་ལས་ལོ་ཧི་ཏ་སོགས་ལས་སོ། །བུ་རྒྱུད་ཀྱི་ཡཉའི་མཐའ་ལས་:ཥ་ཡང་ན་འགྱུར་ཏེ། ལོ་ཧི་ཏྱཱ་ཡ་ནཱི། ཧཱུཾ་སི་ཏྱཱ་ཡ་ནཱི། བཱ་བཱ་བྷ་ཡ་ནཱི། མ་ན་སཱིའི་སྒྲ་ལས་མཱ་ནཱ་ཡཱ་ཡ་ནཱི། བྷ་ཏ་དཱི་ཏ་ཞེས་པ་སྐྱེས་པ་བཞིན་གྱི་དངོས་པོར་བརྗོད་པའི་ནུས་པ་་་ལས་མི་འགྱུར་རོ། །འོ་ན་ཅི་ལྟར་མཱ་ན་བྱཱ་ཡ་ནཱི་ཞེ་ན། མ་ནུའི་སྒྲ་ཡང་བརྗོད་པའི་ཕྱིར་རོ། །བ་དཎྜའི་སྒྲ་ལས་ཨཱངྒཱི་ར་ས་མ་ཡིན་པ་སེལ་སྟེ། བཱ་ཏཙྪཱ་ཡ་ནཱི། ཤི་བ་སོགས་སུ་བརྗོད་པའི་ཕྱིར་། བཱ་ཏིཙྪིའང་ངོ༌། །ཨཱངྒཱི་ར་ས་ལ་ནི་ཡཉ་ལ་ཉིད་དེ་། དེའི་བུད་མེད་ལ་འཇིག་པ་རིགས་ལས་ཞེས་པ་ལ་སོགས་པས་ཇིཔའོས། །བ་ཏཎྜི། ཀ་པའི་སྒྲ་ལས་ཨཱངྒཱི་ར་ས་ལ་ཡཉ་ཉིད་དོ་། །ཀཱ་བྷ་ཡ་ནཱི། ཨཱངྒཱི་ར་ས་མ་ཡིན་པ་ལ་ནི་ཀཱ་པེ་ཡི། ཀཱ་ྋཏུ་ཡ་ནཱི། ཤཱ་ཀ་ལྱཱ་ཡ་ནཱི། ཤ་ཀ་ལའི་—།
