དཀར་བསྟན་འགྱུར། སྒྲ་མདོ། ༼རི། aཔR། ཨདྷཛྷ་བི་ཤ་ཏི་ཀཱི་ནཾ། ཁ་རཱི་དང་ཀ་ཀཱ་ཎིའི་མཐའ་ལས་ཀྱང་ཨི་ཀནའོ། །ཁཱ་རཱི་དང་ཀཱ་ཀ་ཎིའི་མཐའ་བ་གཉིས་པ་ལས་དང་ཀྱང་ཡིག་ལས་འབའ་ཞིག་པ་ལས་ཀྱང་ཨི་ཀན་འགྱུར་ཏེ། དྲི་ཁཱ་རཱི་ཀཾ། ཨདྷཛྷ་ཁཱ་རཱི་ཀཾ། དྲི་ཀཱ་ཀ་ཎི་ཀཾ། ཨདྷཛྷ་ཀཱ་ཀ་ཎི་ཀཾ། འབའ་ཞིག་པ་ལས་ཀྱང་། ཁཱ་རཱི་ཀཾ། ཀཱ་ཀ་ཎཱི་ཀ། ཁ་ཅིག་ནི་ཀཱ་ཀ་ཎིའི་གནས་སུ་ཀཱ་ཀ་ནཱི་ཞེས་ཀློག་པར་བྱེད་དོ། །པ་ཎ་པཱ་ད་མ་ས་ལས་ཡཏའོ། །པ་ཎ་སོགས་ཀྱི་མཐའ་ལས་བ་གཉིས་པ་ལས་འོས་པའི་མཐའ་རྣམས་ལ་ཡཏ་འགྱུར་ཏེ། དྭི་པ་ཙྩཾ། ཨདྷཛྷ་པ་དྡྷཾ། དྲི་མཱ་སིཾ། ཨདྷ་པཱ་ཏྱཾ། དྲི་མཱ་ཧཱུཾ། ཨདྷཌྷ་མཱ་སྱ། སྲོག་ཆགས་ཀྱི་ཡན་ལག་མ་ཡིན་པའི་ཕྱིར་པ་དར་བསྒྱུར་བ་མེད་དོ། །ཤ་ཏ་ལས་ཡང་ནའོ། །ཤ་ཏའི་མཐའ་ལས་བ་གཉིས་པ་ལས་འཇིགས་པ་ལ་ཡཏ་་འགྱུར་ཏེ། དཱི་ཤ་ཏུཾ། ཨངྒཱདྷ་ཤ་ཏུཾ། དཱི་ཤ་ཏཾ། ཨདྷཛྷ་ཤ་ཏཾ། ཋཉའི་་་ཕྱོགས་ལ་འཇིག་གོ། །ཤཱ་ཎ་ལས་སོ། །ཤཱཎའི་མཐའ་ལས་བ་གཉིས་པ་ལས་འཇིག་པ་ལ་ཡཏ་ཡང་ན་འགྱུར་ཏེ། དྲི་ཤཱ་ཙྩཾ། །དཱི་ཤཱ་ཎཾ། ཨདྷཛྷ་ཤ་ཙྩཾ། ཨངྒཱདྷ་ཤ་ཎཾ། དྲི་ཏི་ལས་ཨཎའང་ངོ༌། །དཱི་དང་ཏི་སྔོན་མའི་ཤཱ་ཎ་ལས་ཨཎའང་འགྱུར་ཏེ། ཡཏ་ཀྱང་ངོ༌། །དྲི་ཤཱ་ཙྩཾ། དཱི་ཤཱ་ཎཾ། ཏྲི་ཤཱ་ཙྩཾ། ཏྲེ་ཤཱ་ཎཾ། དྲི་ཤཱ་ཎཾ། ཏྲི་ཤཱ་ནཾ། ཊ་ལས་ཉོས་པ་ལ་རིན་ལས་སོ། །རིན་ཐང་བརྗོད་པ་གསུམ་པའི་མཐའ་ལས་ཉོས་པ་ཞེས་པ་འདིའི་དོན་ལ་ཆོ་ག་ཇི་ལྟ་བ་་་བཞིན་དུ་ཋཀ་འགྱུར་ཏེ། བདུན་ཅུས་ཉོས་པ་ནི། ་སཱཥ་ཏི་ཀཾ། ཨ་ཤཱི་ཏི་ཀཾ། ཀཾ་སི་ཀཾ། ཨཱདྷི་ཀཾ། སོཧཾ། སཱ་ཧ་སྲ། པཉྩ་ཀ། རིན་ལས་ཞེས་པ་ཅི་་ཞེ་ན། དེ་བ་དཏྟེ་ན་ཀྲི་ཏཾ། པ་ཎི་ནཱ་ཀྲི་ཏཾ། འགྲེལ་པར་གྲངས་ཀྱི་ཁྱད་པར་བརྗོད་པའི་ཤུགས་ལས་གཉིས་ཚིག་དང་མང་ཚིག་ལ་མི་འགྱུར་རོ། །པྲ་བྷཱ་བྷཱི་ཀཱི་ཏཾ། པྲཉྫིཀཱི་ཏཾ། ནུས་པ་ཡོད་པ་ལས་འགྱུར་བ་ཉིད་དེ། དྲི་ཀཾ། ཏྲི་ཀཾ། པཉྩ་—།
