བསྟན་འགྱུར། སྒྲ་མདོ། ?རིའིཉིད་དོ། །ཛྷིའི་ཛུསའོ། །།ཛ་ཡིག་ནི་ཛུས་སུཀ་དག་ལའོ་ཞེས་པ་རྟགས་ཀྱི་དོན་:ཏོ༑། མ་པའི་ཨམའོ། །ཝས་མས་དག་གི་དབྱིའོ། །རྣམ་པར་གཅད་པའོ། །ཨ་པཱཀྵི་ཏ། ཨ་པཱ་:ཀྵ། ཨ་པཱཁྱིུ། ཨ་པཱཀྵིgི། ཨ་པཏྟཾ། ཨ་པཱཀྟི། ཨ་པཱཀྵ། ཨ་པཱཀྵ། ཨ་པཱ། ལུང་གི་ཨ་ཏང་རྫོགས་སོ། །།་་་་མི་དེ་ཚོ་ལRད་ན། ::ཀ་ཁ་ག་nཥཱ་བ་དག་སྐྱེ་ཚོ་། །ལ་ཏུ་བྱ་བ་ཏང་བརྗོད་པར་འདོད་ན། པནྡྲ་ཞེས་ཀྱི་གནས་པ་ལ། འབྱུང་བ་ལ་་་ལུང་ངོ༌། །ཨུ་ཡིག་ང་ཡིག་དག་དབྱི་བ་སྔར་བཞིན་ནོ། ཕྱི་ལའི་གནས་སུ་ཏ་ལ་་སོགས་པའོ། །སྔར་བཞིན་ཨ་ཊའོ། །ཊ་ཡིག་དབྱི་བའང་ངོ་། །།ལུང་ལ་སིཙའོ། །ཨི་ཡིག་དང་ཙ་ཡིག་དབྱི་བ་སྔར་ཅིབཞིན་ནོ། །གློ་། །ཙུའི་ཀུའོ། རྟེ། ཀུ་ལས་ཀྱང་་་་ཞེས་པས་ཥ་ཉིད་དོ། །ཛྷལ་ལས་ཛྷལའོ། །ཨཏ་མ་ཡིན་པ་ལས་ཏང་ལའོ་ཞེས་པས་ཨཏ་རུ་བསྒྱུར་བའོ། །དང་། ཨ་པལྟ། །ཨ་པཀྵ་ཏཾ། ཚོ་ཨ་པཀྵཏྲ། ཨ་:པཀྵ་ཐP། ཨ་པཀྵ་ཛཾ། ཨ་པཀྵ་དྡྷཾ། ཨ་པཀྵི། ཨ་པགླུ་ཧི། ཨ་པཀྵ་ཧི། ལུང་ཏང་རྫོགས་སོ། །ག་ཏེ། ཞེ་ཆེན་ཞེབུ། །ཀ་ཀ་ལ་དང་། དེན་ལ་། འོ་་། གང་དག་ཉེ་བ་འོ་དབང་ཁའཁཞི་རཚོ། སྟོད་ཀཆཀྐ:རྟག་པཙ་ཞེས་གནས་པ་ལ། ལིང་ལས་ཀྱི་བྱ་བ་ཤོར་བ་ལརྀངའོ། །རི་འདི་ཡིག་ནི་མྱ་ཏཱས་དག་ལྡི་ལུཊ་དག་ལའོ་ཞེས་པ་རྟགས་ཀྱི་དོན་ཏོ། །དེ་ཉིད་ཀྱི་སྱའོ༑:། ང་ཡིག་དབྱི་བ་སྔར་བཞིན་ནོ། །ལའི་གནས་སུ་ཏིཔ་ལ་སོགས་པར་བསྒྱུར་བར་འགྱུར་རོ། །སྔར་བཞིན་ཨ་ཊ་ཨཱ་ག་མའོ།  ོ། ཊ་ཡིག་དབྱི་བ་སྔར་་བཞིན་ནོ། །ཙུའི་ཀུའོ། ་། ཀུ་ལས་:ཀྱང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཨིཏའི་ཨ་ཏང་ལའོ། །ཏས་ཐས་ཐ་རྣམས་ཀྱི་ཏཾ་ཏཾ་ཏ་རྣམས་ཏེ་ང་འགྲོ་བའི་འང་ངོ༌། །ཛྷའི་ཨནྟའོ། །ཨ་ཏའི་ཨཏ་ཨེང་ལའོ། །བརྩེགས་པའི་ཚིག་གིའོ་ཞེས་པས་ཙིཏ་།
