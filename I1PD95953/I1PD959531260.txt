ཡིག་ནི་སྦེ་སྲུང་ཨུ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ནོ། །ཧ་ན་འཚེ་བ་ལའོ། །མ་ན་ཤེས་པ་ལའོ། །ན་ཡིག་གི་ཨ་ཡིག་ཀློག་པའི་དོན་ཏོ། །ཨི་དི་མཆོག་གི་དབང་ཕྱུག་ལའོ༑། ཨི་ཡིག་ནི་ཨི་དིཏ་ལས་ནུམ་མོ་ཞེས་པའི་རྟགས་དོན་ནོ། །འདི་ཉིད་ཀྱིས་ནུམ་མོ། །ཨུ་ཡིག་བརྗོད་པའི་དོན། མ་ཡིག་ནི་མི་ཏ་ཨཙ་མཐའ་ལས་ཕ་རོལ་ཞེས་པའི་རྟགས་དོན་ནོ། །ཀ་དི་ཀྲ་དི་ཀླུ་དི་འབོད་པ་ལའོ། །སྔ་མ་བཞིན་དུ་ཨི་ཡིག་་དབྱི་སྟེ། ནུམ་ཡེང་། བནྡྷ་འཆིང་བ་ལའོ། །ཨ་ཡིག་བརྗོད་དོན་ཞེས་པ་བྱིངས་འདི་རྣམས་ལས་ཨུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །སུའི་རུ་ཉིད་དང་རྣམ་པར་བཅད་པའོ། །པ་ཊུ༑ཡམཁས་པའམ་ཤེས་རབ་ཅན། ཨ་སུ། ཡགྲོགས། བ་སུ། ཡནོར་རམ་རྫས། ཏྲ་སུ༈ཡཞ་ཉེ། ཧ་ནུ། འགྲམ་པའམ་གདོང་གི་ཕྱོགས་གཅིག། མ་ནུ། ཡསྐྱེ་དགུའི་བདག་པོ། :ཨིནྡྲ། ཡཟླ་བ། ཀནྡྷ། འཚེད་པའི་གནས། །རང་གི་སྐྱེ་བོའོ། །བནྡྷ། ཡགཉེན་ནམ་རྩ་ལག། །བ་ཧི་ར་ཧི་པ་ཤི་ལས་རིང་པོ་ཡང་ངོ༌། །བ་ཧ་ཐོབ་པ་ལའོ། །ར་ཧ་གཏོང་བ་ལའོ། །ཧའི་ཨ་ཡིག་ཀློག་པའི་དོན། ::པ་ཤི་མ་པི་ཡོངས་སུ་འཇལ་བ་ལའོ། །ཨི་ཡིག་ནི་ཨ་དིཏ་་ལས་ནུམའོ་ཞེས་པའི་རྟགས་དོན་ནོ། །དེས་ན་་་ནུམའོ༑། ཨུ་ཡིག་བརྗོད་དོན་ནོ། །མ་ཡིག་ནི། མིཏ་ཨཙ་མཐའ་ལས་ཕ་་་རོལ་ལོ་ཞེས་པའི་རྟགས་དོན་ཏོ་ཞེས་པ་འདི་རྣམས་ལས་ཨུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །རིང་པོ་ཡང་ངོ་ཞེས་པས་ཨཱ་ཉིད་དོ། །བཱ་ཧུཡདཔུང་པའམ་ལག་པ། རཱ་ཧུ། སྒྲ་་་གཅན། པཱི་ཤུ། ཡརྡུལ། ན་མི་མ་ནི་ཛན་རྣམས་ཀྱི་ནཱ་ཀི་དྷ་ཡང་ངོ༌། །ཎམ་རབ་་ཏུ་འདུད་པ་དང་སྒྲ་ལའོ། །ཎའི་ན་འོས་ན་ཡིག་གོ། །།མན་ཤེས་པ་ལའོ། །།མ་ཡིག་དང་ན་ཡིག་དག་གི་ཨ་ཡིག་བརྗོད་དོན་ནོ། །ཛ་ནཱི་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཨཱི་ཡིག་ནི་ཤཱི་ཨཱི་དིཏ་ལས་ཏ་ཏེ་བཏ་དག་གིའོ་ཞེས་པའི་རྟགས་དོན་ནོ། །ཞེས་པ་དེ་རྣམས་ལས་ཨུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །འདི་རྣམས་ཀྱི་གནས་སུ་ནཱ་ཀི་དང་དྷ་་་།
