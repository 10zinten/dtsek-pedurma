བསྟན་འགྱུར། སྒྲ་མདོ། རེལས་སོ། །དྲུག་པོ་འདི་རྣམས་ལས་ཀྱཔ་འགྱུར་ཏེ། ཨི་ཏི། །སྐུ་ཚེ། ཌ་ཤི་༑དྲྀ་ཏྱ། སྲིང་ལས་ནི་བཱན། །དུས་མཆོད་དོ། །ཨཱ་མྲྀ་ཏྱ། :ཛུ་ཥྱི། ཎ་ཡིག་ཅི་ཞེ་ན། ཨུ་པེ་ཡ། །ཨ་བ་བྱ་སཱུ་ཏྱ་ཞེས་:པར་ཀྱང་འདོད་པའི་ཕྱིར་རོ། །ཤཔ་དང་པྲཉ་ལས་ཨཱ་དང་ནི་སྔོན་མ་ལས་མི་འདོད་དོ། །དེས་ན་ཨཱ་ཤཱ་སྱ་མ་་་ནྱཏ༑ཡང་བརྗོད་པར་གྱུར་པ། ་ཨ་ནི་བཱརྨོ་ག་:ཛི་ར་རྩེ༄༅། སྭ་བྷཱ་བ་ཨི་བ་་དེ་ཧི་ཧཱུཾ་ཞེས་གྲུབ་བོ། །རིཏ་མཐའ་ཉེ་ལས་:ཀྲི་པི་ཝཾ་ཏ་མ་ཡིན་པ་ལས་སོ། །རྀཏ་མཐའ་ཉེའི་བྱིངས་ལས་ཀྱཔ་འགྱུར་ཏེ། འཇུག་པར་བྱ་བ་ནི་བྲི་ཏྱ། ཤྲཱི་ཏུཾ། ཀྲྀ་ཏྱཾ། ::ཀླུ་པ་ཙཱི་ཏ་མ་ཡིན་པ་ལས་ཞེས་པ་ཅི་ཞེ་ན། ལསཾ་:ཀ་ལྱཱི། བི་ཙནྟུ། །་ཅི་ལྟར་ཚིགས་སྩོལ། པུཾ་སུཾ་ཞེ་ན། ཀྱཔ་ཉིད་འདོད་པའི་ཕྱིར། ཁན་ལས་ཨིཏ་འང་ངོ༌། །ཁ་ནིའི་བྱིངས་ལས་ཀྱཔ་འགྱུར་ཏེ། མཐའ་ཨི་ཡིག་་:ཏུའང་བསྒྱུར་རོ། །ཁེ་ཡཾ། བྷོཉ་ལས་མིང་མ་ཡིན་པ་ལའོ། །ཀྱཔ་འགྱུར་ཏེ། གཟུང་བར་བྱ་བ་ལ་ཤྲིp། །ལས་བྱེད་པའོ། །མིང་མ་ཡིན་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། བྷཱན། མིང་ལ་ཡང་མ་ཡིན་ནམ་ཞེ་ན། བུད་མེད་ལ་ཀྱཙ་འགྱུར་བ་སྟེ། དེ་ནི་དངོས་པོ་ལ་སྒྲུབ་པའི་ཕྱིར་རོ། ་། ཀྲྀ། དྲྀ::ཎི། ཡ་ཛྲ་ཛི། ཤྲི་སི། སཾ་ཤྲཱི༑དུ་ཧི། གུ་ཧ་ལས་ཡང་ནའོ། །ཀྱཔ་འགྱུར་ཏེ། ::ཀྲྀ་ཏྱཾ། པཀཱཎཱཾ། དྲྀ་ཏྱཾ། བཙྩཾ། ཤྲཱིཙྩཾ། །མཱཧུཾ། ཤ་ཧཱུཾ། སཾ་བྷཾ་ཏཾ༄༅། སཾ་བྷཱརླ། དུ་:པཾ། །དོ་ཧཾ༄༅། གུ་ཧཱུཾ། གོ་ཧཱུཾ། རཱ་ཛ་སྭཱ་ཡ། སཱུ་རྱ། རུ་ཙུ། ཀྲྀཥྚ་པ་ཙྪ། ཨ་བྱ་བྱ་རྣམས་སོ། །ལྔ་པོ་འདི་རྣམས་ཀྱཔའི་ཚིག་ཕྲད་དེ། རྒྱལ་པོས་འཁྱུད་ཅིང་བཏུང་བར་བྱ་བས་ཡང་ན་རཱ་ཛ་སྭཱ་ཡ་ཏེ་ཞེས་སོ། །རཱ་ཛ་སཱུ་ཡཚེ། མཆོད་སྦྱིན་ནོ། །རུག་མེད་པ་དང་རིང་པོ་ཉིད་ཚིག་ཕྲད་དེ། རྒྱུ་བར་བྱེད་པས་་ན༑་སཱུརླའོ། །རོ་ཙ་ཙིཏི་ལས་སོ་རུ་ཙུའོ། །།རང་ཉིད་ཀྱིས་རྨོས་ཤིང་་་།
