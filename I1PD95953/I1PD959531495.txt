བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིནི་ཀ། ཨཱ་ཁ་ནི་ཀ་བ་ག་ཞེས་པ་ཀའི་སྒྲ་དང་བསྡུ་བའོ། །ཨཱ་ཁཞེས་པ་ཌ་ཡིས་གྲུབ་བོ། །ཨཱ་ཁ་ནཞེས་པ་གྷཉ་ཡིས་སོ། །ཨཱི་:ཥ་དེ། དུ། སུ་རྣམས་ལས་ཁལའོ། །བྱིངས་ལས་འགྱུར་ཏེ། ཨཱིཥ་ཏ་ཀ་ར། ཀ་ཊོ་བྷ་བ་ཏཱ། :དུཀྐ་རཿ། སུ་ཀ་ར། ཨི་ཥལླ་ཞེས་པ་མང་པོ་ལས་སོ། །པུ་ཤ་ཀི་ཞེས་པ་ལ་སོགས་པས་ཡཏའོམ། །བྱེད་པ་པོ་དང་ལས་དག་ལ་བྷུ༄༅། ཀྲྀཉ་ལས་སོ། ཨེ། ཨི་ཥད་ལ་སོགས་པ་རྣམས་ལས་རོལ་བྱེད་པ་པོ་དང་ལས་དག་ལས་ཕ་རོལ་བྷཱུ་་་ཀྲྀཉ་དག་ལས་ཁལ་འགྱུར་ཏེ། ཨཱི་ཥ་དཱ་:ཆུ་སྱ་བྷ་བ་ནཾ། ཨཱི་ཥ་དཱ་རྒྱུ་བྷ་ལིབ་མ་སུ་ནཱ། དུ་རཱ་ཧུཾ་བྷ་ཧཾ། སཱ་དྷཾ་བྷ་ཐཾ། ཅུང་ཟད་ཕྱུག་པར་བྱེད་པ་ནི། ཨཱི་ཥ་ད་ེདྷཾ་ཀ་ར། དུ་ར་ཙྩཾ་ཀ་རི། པྲ་ཏྱཾ་ཀ་ར། ཤི་་སཱ་ཡཱ། གང་གི་ཚེ། ཅུང་ཟད་ཕྱུག་པ་ཉིད་འགའ་ཞིག་བྱེད་པ་དེའི་ཚེ་ཅི་ལྟར་མི་འགྱུར་ཞེ་ན། །ཙཱིའི་་དོན་མངོན་པར་བརྗོད་པའི་ཕྱིར་རོ། །ཤཱ་པི་ཡུ་དྷི། །དི་ཤི། ཛཱ་ཏི། ཨཱཏ་ལས་ཡུཙའོ། །ཨཱི་ཥ་སོགས་ལས་ཕ་རོལ་དང་ཤི་ཥྱ་སོགས་རྣམས་དང་ཨཱཏ་མཐའ་ལས་ཀྱང་ཡུཙ་འགྱུར་ཏེ། ཨཱི་ཥཙྪཱས་ཅིན། །དུཤཱས་ནས། སུ་ཤཱས་་ན༑དུརྨོ་དྷ་ན། སུ་ཡོ་དྷ་ན། །དུརྙན་ན། ལསུམནྟ་ཎ། ཨཱི་ཥཏྟཱ་ཙིན། དུཥཱ་ན། སུ་པཱ་ན། ཐུན་མོང་གི་ཚོགས་ཀྱང་མང་པ་ལ་སྤྱིར་བཏང་བའི་ཕྱིར་དངོས་པོ་དང་ལས་དག་ཉིད་ལ་ཡུཙ་འགྱུར་ཏེ། བྱེད་པ་པོ་ལ་་་ཞེས་པ་འདི་མཐའ་རྒྱས་པའོ། །ཨ་ལཾིཁ་ལུ་ལས་དགག་པ་ལ་ལྟ་ཡང་ནའོ། །།།དགག་པ་ལ་འཇུག་པའི་ཨ་ལོ་དང་ཁ་ལུའི་སྒྲ་དག་སྦྱོར་བ་ལ་བྱིངས་ལས་དངོས་པོ་ལ་ལྟ་ཡང་ན་འགྱུར་ཏེ། ཨ་ལཾ་ཀྲྀ་ཏཱ། ཁ་ལུ་ཀྲྀ་ཏཱ། ཨ་ལཾ་ཀྲྀ་ཏེ་ན། བྱེད་་པ་དང་མི་སྦྱོར་བ་ན་མ་ཏུ་པ་མི་བྱའོ་ཞེས་པ་ཇི་སྲིད་པའོ། །འབྲས་བུ་མེད་པ་ལས་བྱུང་བ་སྔོན་མ་ཅན་འདིར་བཟློག་པ་མཚོན་ནོ། །དགག་པ་ལ་ཞེས་པ་ཅི་ཞེ་—།
