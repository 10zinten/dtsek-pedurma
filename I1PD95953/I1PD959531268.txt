ནས་སྦྱོར་བ་ལ་ཎི་ཙའོ། །ཎ་ཡིག་ནི་ཉྩིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། མང་པོའི་ཚིག་ལས་ཉྩིཏ་ལའོ་ཞེས་པས་ཨཱ་ཉིད་དུ་མི་འགྱུར་རོ་ཞེས་པ་འདི་རྣམས་ལས་་ཎིའི་མཐའ་ལས་ཨིནྡྲ་ཙ་ཕ་རོལ་དུ་འགྱུར་རོ། ་། ཙ་ཡིག་ཙ་ཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། ཨཱམ་ཨན་ཨཱ་ལུ། ་ཨཱ་ཡ་ཨིནུ་རྣམས་ལ་ཨ་ཡའོ་ཞེས་པས་ཎིའི་ཨ་་ཡ་རུ་འགྱུར་རོ། །སྟ་ན་ཡིནུ། སྤྲིན། ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨ་ཉིད། ར་ཕ་རོལ་ལོ། །ཧནྟ་ཡིནྡྷཡགསེར། ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨོ་ཉིད་དོ། །པོ་ཥ། གསོས་པ། ཡིནུ། ཁུ་བྱུག། ག་ད་ཡིནུ། མང་དུ་སྨྲ་བ། མ་ད་ཡིནུ། ཆང༌། ཧྲ་ཀྲྀ་དག་ལས་ཨེརྟུའོ། །ཧྲཉ་འཕྲོག་པ་ལའོ། །།ཉ་ཡིག་ནི་ཧྲིཉ་འགྲོ་བའི་ངང་ཚུལ་ལའོ་ཞེས་པའི་རྟགས་ཀྱི་དོན་ཏོ། ཌ་་། ཌཱ་ཀྲྀཉ་བྱེད་པ་ལའོ། །ཌཱ་ཡིག་ནི་འུ་་འགྲོ་བ་ལས་ཀྲིའོ་ཞེས་པའི་རྟགས་ཀྱི་དོན། ཉ་ཡིག་ནི་ཀྲྀཉ་ལས་སུཊ་མེད་པ་ལས་སོ་ཞེས་པའི་རྟགས་ཀྱི་དོན་ཞེས་པ་འདི་དག་ལས་ཨེ་ཎུ་ཕ་རོལ་ཏུ་་་འགྱུར་རོ། །ཨི་ཀའི་ཨཏ་ཨེང་ངོ་ཞེས་པས་ཨ་ཉིད། ར་ཕ་རོལ་ཉིད། ཧ་རེ་ཎུ༑དྲིའི་རྫས། ་ཀ་རེ་ཎུ། གླང་མོ། །དཱ་བྷཱ་དག་ལས་ནུའོ། །ཌཱ་དཱ་སྦྱིན་པ་ལའོ༑། །རྔུ་ཡིག་ནི་རུ་འགྲོ་བ་ལས་ཀྲིའོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །བྷཱ་འབར་་བ་ལའོ། །ཞེས་པ་འདི་དག་ལས་ནུ་ཕ་རོལ་དུ་འགྱུར་རོ། །དཱ་ནུ། ཡསྦྱིན་པ་པོ། བྷཱ་ནུ། ཡཉི་མ། རྀ་བྲོཉ་དག་ལས་ནི་ཏའོ། །ར་བྲི་འགྲོ་བ་ལའོ། །བྲོཉ་སྒྲིབ་་པ་ལའོ། །ཉ་ཡིག་ནི་བྲོཉ་སྒྲིབ་པ་ལའོ་ཞེས་པའི་རྟགས་དོན་ཞེས་པ་འདི་ལས་ནུ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་། ནི་ཏའོ་ཞེས་པས་ཎ་ཡིག་ཕྱིས་པ་བཞིན་ཏེ། །དེས་ན་ནི་ཏ་ཉིད་ཀྱི་ཕྱིར་དབྱངས་སོ། །ཨི་ཀའི་ཨཏ་ཨེངའོ་ཞེས་པས་ཨེ་ཉིད། ར་ཕ་རོལ་ཉིད་དོ། །ཚིག་གཅིག་ལས་ར་ཥ་ལས་ནའི་ཎའོ་ཞེས་པས་ཎ་ཉིད། རེ་རྟུ༑ཡརྡུལ། །ཝཎྞ། སྒྲའོ། །སཱིུ། བི་ཥི་དག་ལས་ཀིཏའོ། །།ཁྱུ་དེད་པ་།
