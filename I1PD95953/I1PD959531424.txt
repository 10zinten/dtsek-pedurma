བྱེད་ཅེས་པའི་དོན་ཏོ། །པྲཛྙཱ་ཤཱ་སྣ་ཎི་ཀྲ་མནྟེ། །རྒྱས་པར་གྱུར་ཞེས་པའི་དོན་ཏོ༑་། ཨུ་པ་པ་རཱ་དག་ལས་སོ། །ཉེར་བསྒྱུར་ཨུ་པ་དང་པ་རཱ་དག་ཉིད་ལས་ཀྲ་མ་ཏི་ལས་བྲྀཏྟི་སོགས་ཀྱི་དོན་རྣམས་ལ་ཏ་སོགས་པ་འགྱུར་ཏེ། །ཨུ་པ་ཀྲ་མ་ཏེ། །པ་རཱ་ཀྲ་མ་ཏེ། བཤད་མ་ཐག་པས་གྲུབ་པ་ལ་ངེས་པར་བྱ་བའི་དོན་དུ་བརྗོད་པའོ། །དེས་ན་མི་འགྱུར་ཏེ། ཨ་རྒྱ་ཡ་ནཱ་ཡ་སཾ་ཀཱ་མ་ཏི། བི་ལས་རྐང་པ་དག་གིས་སོ། །བི་སྔོན་མའི་ཀྲ་མ་ཏི་ལས་རྐང་པ་དག་གིས་འགྲོ་བ་ལ་ཏ་ལ་སོགས་པ་འགྱུར་ཏེ། །སཱ་དྷ་བི་ཀྲ་མ་ཏེ། མདོ་ལས་རྟགས་བརྗོད་པ་ནི་རྒྱུད་་་ཉིད་ཀྱི་ཕྱིར་རོ། །སཱ་དྷུ་བི་ཀྲ་མནྟེན་ཤཱ། ་རྐང་པ་དག་གིས་ཞེས་པ་ཅི་ཞེ་ན། ཨ་ཤྭ་ན་བི་ཀཱ་མ་ཏི། །པྲ་ཨུ་པ་ལས་རྩོམ་པ་ལའོ། །བྱ་བའི་སྐད་ཅིག་མང་པོ་་་་རྣམས་ཀྱི་དང་པོའི་བྱ་བ་སྒྲུབ་པ་ནི་རྩོམ་པ་སྟེ། དེ་ཡང་ཟས་ཀྱི་བྱ་བ་སོགས་ཀྱི་་དོན་དུ་བྱེད་པ་ཡང་ངོ༌། །དེ་ལ་འཇུག་པའི་ཀྲ་མ་ལས་པྲ་དང་ཨུ་པ་ལས་ཕ་རོལ་ཏུ་ཏ་ལ་སོགས་པ་འགྱུར་ཏེ། པྲ་ཀ་མ་ཏེ་བྷོཏྟཾ། ་་་ཨུ་པ་ཀྲ་མ་ཏེ་ཐོ་ཏུཾ། རྩོམ་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། པཱུརྦ་རླུ། པྲ་ཀྲ་མ་ཏི། ཨ་པ་རེ་རླུ་རུ་པ་ཀཱ་མ་ཏི། སྔ་ཉིན་རབ་ཏུ་འགྲོ་བ་དང༌། ཕྱི་ཉིན་:ཉེ་བར་འགྲོ་བ་ཞེས་པའི་དོན་ཏོ། །ངོ་ཏི། ཨུངྒ་ཏི་ལ་ཨང་ལས་སོ། །ངོ་ཏི་ཨུངྒ་ཏི་ལ་བགྲོད་པར་བྱེད་པ་ལ་ཨང་། སྔོན་མའི་ཀྲ་མ་ཏི་ལས་ཏ་ལ་སོགས་པ་འགྱུར་ཏེ། ཨཱ་ཀྲ་མ་ཏེ་སཱུརྞ། ཚོ་ཏི་ཞེས་པ་ཅི་ཞེ་ན། །ཨཱ་ཀྲ་མ་ཏི་དྷཱ་མ། ཨུངྒ་ཏི་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨགི་རིནྡྷཱ་ན་མཱ་ཀཱ་མ་ཏི། ་འདི་སྟེང་དུ་བགྲོད་པས་འཕགས་པར་དགོངས་པའོ། །ཉེར་བསྒྱུར་མེད་པ་ལས་ཡང་ནའོ། ་། ཀྲ་མ་ཏི་ལས་ཏ་ལ་སོགས་པ་འགྱུར་ཏེ། ཀྲ་མ་ཏེ། ཀཱ་མ་ཏི། ཉེར་བསྒྱུར་མེད་པ་ལས་ཞེས་པ་ཅི་ཞེ་ན། སཾ་ཀཱ་མ་ཏི། ཨ་པ་ལས་གི་ར་ཏི་ལས་སོ། །ཆ། །ཏ་ལ་སོགས་པ་འགྱུར་ཏེ། ཨ་བ་གི་ར་ཏེ། ཨ་—།
