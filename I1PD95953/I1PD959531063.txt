བསྟན་འགྱུར། སྒྲ་མདོ། ༼རི་«སྣར་པར། དེ་«པེ་»«སྣར་»སམྲའི་སུཊ། «པེ་»«སྣར་»ནྣཾ། «པེ་»«སྣར་དང་པོ། «པེ་»«སྣར་ཨིཎཱ། «པེ་»«སྣར་སྲིད་ཧཱུཾ་ཛའི་ལའོ་ཞེས་པ་ཇི་སྲིད་པར་སྤྱིར་བཏང་བའོ། །བསྡུ་བ་ལ་མ་ཕྱི་མ་ཡིན་་་་་པའིའོ། །བསྡུ་བ་ལ། «པེ་»«སྣར་»+ལ། («པེ་»«སྣར་»སིང༌། «པེ་»«སྣར་»ལིཊ། ཚེ«པེ་»:འི་ནཱིར། 《སྣར་»ནཱིར། འགྱེ«པེ་»«སྣར་སྲིང་སཱཉྫའི་ལིཊའོ། 《པེ་»«སྣར་ནའོ། ཙེ«པེ་»ད《སྣར་»ཀབྷཱ་ཧཱུཾ། སྐྱེ«པེ་»«སྣར་»—པོ། སྐྱེ«པེ་»«སྣར་ཀུ། ཀྱེ«པེ་»«སྣར་ཨཙཱ། (ལོ«པེ་»སྣར་»:::པའོ། «པེ་»«སྣར་སུཔྲ་ལ་ཨཙཱ། ཀྱེ«པེ་»«སྣར་ཨཙཱ། དགྱེ«པེ་»«སྣར་»བྷ། འགྱེ«པེ་»«སྣར་སྐུ་་པ་ཡི་བེཔྲ། འབུ«པེ་»«སྣར་ཀྱི་དང་:ལ་རེད། @«སྣར་»ཉམ། མི«པེ་»«སྣར་ཨཙཱ། ®«པེ་»«སྣར་། །།«པེ་»«སྣར་»+ཡང་། @«པེ་»::«སྣར་+འགྲོ་བ་དུ་མ་ལ་ཕན་པར་གྱུར་ཅིག། །ད་དེ་དེད་དུ་ད་དུང་ཤེས་རྒྱ་བླ་གྲོང་ན། ཀ་ཀཔའི:ལ་ན་བསྡུར་འབྲས་རེའུ་མིགཚེ་སྲིད་པ་ག་ལ་ཡེ་ད་སྒྱུར་བྱེད་་་་་་་་་་་པ་ན་སྒྲ་མདོ། (བྱེ་ན་བཙན་པོ་དེའི་ན་གཞིར། ཚོང་། སྣར་ཐང་། ::ཅོ་ནེ། །ཚེ་དཔེ་གཞི། །སྡེ་ཚན། །པོད་རྟགས། །སྡེ་དགེ། སྒྲ་མདོ། པེ་ཅིན། ཙནྡྲ་པའི་མདོ། མདོ། སྒྲ་མདོ། །པ་ལེན་ཚན་གྲངས། སསྐབས། །ཕྲེང་། ཚེག་ཁྱིམ། ནན། ཉན་_བར་ན་_བ་ཤན་ན་། །།།།ཡ། དེ་ནས་དགག་འགྲོ་བ་མ་ལུས་པ་རྒྱུ་མེད་་ད་ནི་དེ་བས་ཟླ་བ་དགུ། །ཁ་ར་བ་བློ་། ཏུན་དང་དཀར་སྟོང་ལཀྱི་ལྟ་བདེ་ན་ཏེ་ག་ག་་་་ཏཱི—།
