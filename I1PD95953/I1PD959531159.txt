བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིརྱ་ཛ་ཉམས་པ་ལའོ། །ཥཉྩ་འགྲོགས་པ་ལའོ། །་། ::ཛྲ་ཤིར་བལྟ་བ་ལའོ། །་། དན་མཆེ་བ་ལའོ། །ཀྲྀ་ཥ་རྣམ་པར་རྐོ་བ་ལའོ། །།ད་ཧ་ཐལ་བར་བྱེད་པ་ལའོ། །མི་ཧ་འཁྲུད་པ་ལའོ། །ཀི་ཏ་ངེས་པར་གནས་པ་ལའོ། །ཏང་ཨཱ་ན་མ་ཡིན་པ་རྣམས་སོ། །ཨེ་དྷ་འཕེལ་བ་ལའོ། །སྤནྣཱ་དགའ་བ་ལའོ། །།གཱ་དྷི་རབ་ཏུ་གནས་པ་ལའོ། །བཱ་དྷཾ་དཀྲུགས་པ་ལའོ། ལ། ད་དྷ་འཛིན་པ་ལའོ། །ཀ་དི་ཀུན་དུ་འགྲོ་བ་ལའོ། །སཱི་དི་དཀར་བ་ཉིད་ལའོ། །བ་དི་གསོང་པོར་སྨྲ་བ་ལའོ། །བྷ་དི་དགེ་ལེགས་ལའོ། །མ་དི་བླུན་པོ་ལའོ། །སྤ་དི་ཅུང་ཟད་གཡོ་བ་ལའོ། །མུ་ད་དགའ་བ་ལའོ། །ད་ད་སྦྱིན་པ་ལའོ། །སཱ་ད། །སཱ་ད། སཱརྡ་རོ་མྱང་བ་ལའོ། །ཨཱུརྞཱ་ཚད་མ་ལའོ། །།ཀུ་རྡ། ཁུ་རྡ་གུ་ད་བརྩེ་བ་ལའོ། །ཥཱ་ད་འགྱུར་བར་བྱེད་པ་ལའོ། །ཧཱ་ད་སྒྲ་ལའོ། །སརྡ་སྨད་པ་ལའོ། །ཡ་ཏཱི་རབ་ཏུ་འབད་པ་ལའོ། །ཡུ::དྲྀ་ཛུ་ཏྲ་བཤད་པ་ལའོ། །ནཱ་དྷོ། ་་:ནཱ་ཤྲཱི། བི་ཤྲིལ། བེ་ཤྲཱི་སློང་བ་ལའོ། །ཤ་ཐི་བསྙེན་པ་ཉིད་ལའོ། །ཀནྡྲ་བསྔགས་པ་ལའོ། །:：ཤཱ་ཀྲིའཁྲུད་པ་ལའོ། །ལོ་ཀྲྀལྟ་བ་ལའོ། །བློ་ཀྲྀ་ཀུན་དུ་ཚོགས་པ་ལའོ། །ད་དྲེ་ཀྲྀ། སྡེ་ཀྲྀ་འཕེལ་བ་ལའོ། །རེ་ཀྲྀ་དགོས་པ་ལའོ། །ཤེ་ཀྲྀ། གླེ་ཀྲི། སྲ་ཀྲྀ། ཟླ་ཀྲྀ་འགྲོ་བའི་དོན་ལའོ། །ཤ་ཀི་དོགས་པ་ལའོ། །ཨ་ཀི་མཚོན་བྱེད་ལའོ༑། བ་ཀི་གྱ་གྱུ་ལའོ། །མ་ཀི་རྒྱན་ལའོ། །གྲ་སུ། གླ་སུ་ཟ་བ་ལའོ། །ཨཱི་ཧ་བྱེད་པ་ལའོ། །བ་ཧི་མ་ཧི་འཕེལ་བ་ལའོ། །ཨ་ཧི་འགྲོ་བ་ལའོ། །གརྡ། གལྔ་སྨད་པ་ལའོ། །བརྡ། བལྔ་གཙོ་བོ་ལའོ། །མི་ཧ་འགྲོ་བ་ལའོ། །བེ་ཧྲ༑་ཛེ་ཧྲི། བྲ་གྲོགས་རབ་ཏུ་འབད་པ་ལའོ། །དཱ་ཧྲི་འཇོག་པ་ལའོ། །ཀཱ་ཤི་འབར་བ་ལའོ། །།།ཨཱུ་ཧ་རྣམ་པར་རྟོག་པ་ལའོ། །གཱ་ཧུ་སྲུབ་པ་ལའོ། །གླུ་ཥིར་བྱེད་པ་ལའོ། །སྨིད་བཞད་གད་ལའོ། །གུང་མི་གསལ་བའི་སྒྲ་ལའོ། །།
