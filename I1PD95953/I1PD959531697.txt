བསྟན་འགྱུར། སྒྲ་མདོ། རེཡོ་པཱ་དྲ། ཨ་ཙི་ཏ། ཨཱ་དྷ་ཀ་ལས་ཁ་ཡང་ནའོ། །གཉིས་པའི་མཐའ་ལས་ཀུན་འབྱུང་བ་ལ་སོགས་པ་རྣམས་ལ་འགྱུར་ཏེ། བོ་སྣོད་དུ་ཀུན་འབྱུང་བའམ་ཟ་བའམ་འཚེད་པ་ནི། པྲ་ཏཱི་ཎཱ། པྲ་ཏཱི་ཀཱི། ཨཱ་ཙི་ཏཱི་ནཱ། ཨ་ཙི་ཏཱི་ཀཱི། འི་ཨཌྷ་ཀཱི་ནཱ༑ཨཱཌྷ་ཀཱི། བ་གཉིས་པ་ལས་ཋཊ་འང་ངོ༌། ང་། པཱ་ཏྲ་ལ་སོགས་པའི་མཐའ་ལས་ཀུན་འབྱུང་བ་ལ་སོགས་པ་རྣམས་ལ་ཋཉ་འགྱུར་ཞིང་ཁ་ཡང་ཡང་ནའོ། །སྣོད་གཉིས་སུ་ཀུན་དུ་འབྱུང་བའམ་ཟ་བའམ་འཚེད་པ་ནི། དྲི་པཱ་ཏཱི་ཀཱི། དྲི་པཱ་ཏི་ཎ། དཱི་པཱ་ཏི། ་ཡོངས་སུ་འཇལ་བ་ལས། །ཀཱ་ལ་བསྟི། ག་ཨ་ཙི་ཏ། ཀམྤ་ལྱ། སཾ་ཁྱ་ལས་འཇིག་པ་ལའོ་ཞེས་པས་ངཔའོ། །དྷུ་ཙི་ཏ་ཀཱི། དྷུ་ཙི་ཏཱི་ནཱ། བྱཱ་ཙི་ཏཱི། ཙཱུ་ཌྷ་ཀི་ཀཱི། བྷཱུ་ཌྷ་ཀཱི། ནཱདྷཱུ་ཌྷ་ཀཱི། ་འང་ཡིག་གིས་རྐྱེན་གཉིས་གྲུབ་པ་ལ་འམ་གྱི་སྒྲ་རྗེས་སུ་འཇུག་པ་དང་ཋཉ་འཇིག་པའི་རང་བཞིན་ཡས་གསལ་བའི་དོན་ཏོ། །ཀུ་ལི་ཛི་ལས་ཡང་ནའོ། །ཀུ་ལི་ཛའི་མཐའ་ལས་འབྱུང་བ་ལ་སོགས་པ་རྣམས་ལ་ཁ་ཡང་ན་འགྱུར་ཏེ། ཁ་ཡང་ཡང་ནའོ། །རིགས་གཉིས་ལས་སྐྱེས་པར་ཀུན་འབྱུང་བའམ་ཟ་བའམ་འཚེད་པ་ནི། །དྲི་ཀུ་ལི་ཛི་ཀཱ། དྲི་ཀུ་ལི་ཛཱི་ནཱ། དཱི་ཀུ་ལི་ཛཱ། དཱི་ཀུ་ལི་ཛི་ཀི། །འམ་གྱི་སྒྲ་རྗེས་སུ་འཇུག་པ་ལ་བི་ཀ་ར་ཎ། ཋ་རྣམ་པར་བརྟག་པའི་དོན་ཏོ། །དེས་ན་ཙཱ་རཱུ་པཾ་སམྤ་རྱ་ཏེ། བི་ཤ་སོགས་ལས་བྷ་ར་ལས་ཁུར་ཕྲོགས་པ་དང་འདྲེན་པ་དང་ཀུན་དུ་འདྲེན་པའོ། །ཐཾ་ཤ་སོགས་ལས་ཕ་རོལ་བྷ་ར་ལས་འཕྲོག་པ་སོགས་ཀྱི་དོན་རྣམས་ལ་ཟཉ་འགྱུར་ཏེ། འོད་མའི་ཁུར་འཕྲོག་པའམ་འདྲེན་པའམ་ཀུན་དུ་འདྲེན་པའམ། །ངབཱ་ཤ་བྷཱ་རི་་ཀཀོ་ཏྲ་ཛ་བྷཱ་རི་ཀ། ་ཐཾ་ཤ་སོགས་ལས་ཞེས་པ་ཅི་ཞེ་ན། ལ་དྷ་ནྱ། བྷཱ་རཾ་བ་ཧ་ཏི། ་ཡང་ན་བི་ཤ་སོགས་ལས་ཁུར་འབྱུང་བ་ལས་སྒྲུབ་པའོ། །བི་ཤ་ཧི་ར་ཏི། ཀོ་ཏྲི་ཛི་ཀ། བྷ་ར་ལས་ཞེས་པ་ཅི་ཞེ་ན། ཐཾ་ཤཾ་ཧཾ་ར་ཏི། བི་ཤ། །།
