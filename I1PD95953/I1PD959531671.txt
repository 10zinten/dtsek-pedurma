བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིརྒན་པོའི་ནོར་གྱི་བར་དུ་བྱིན་པ་དེས་ལས་ཉིས་འགྱུར་སུམ་འགྱུར་དུ་འཛིན་པར་བྱེད་པ་ནི། །མལླ་ཏེ་སྨད་པའོ། །འདི་ལྟ་བུ་འཛིན་པར་བྱེད་པའི་བསྟན་བཅོས་སུ་བཀག་པའི་ཕྱིར་རོ། ་། དྲྀདྷི་ལས་སྲི་དྷ་ཥ་ལས་སོ། །ཋཀ་སྦྱོར་བར་་འགྱུར་ཏེ། ་འཕེལ་བ་འཛིན་པ་ནི། བཱདྷུ་ཥི་ཀ། བྲྀདྷུ་ཥིའི་སྒྲ་ཡིས་གྲུབ་པ་བརྗོད་པ་ནི་བིདྡྷི་ཐོས་པ་བཟློག་པའི་དོན་ཏོ། །ད་ཤི་ཀ་ད་ཤ་ཀུ་ཤྲཱི་ར་ལས་ཋཊའོ༑། འདི་དག་ལ་སྨད་པ་འཛིན་ནོ་ལ་ཁཊ་འགྱུར་ཏེ། ད་ཤི་ཀ་ད་་ཤ་ནི་བཅུ་གཅིག་པའི་དོན་མངོན་པར་རྗོད་ཀྱི་བྱེད་དེ་འདི་ཉིད་ཚིག་ཕྲད་པ་ལས་ལེགས་པའོ། །ད་ཤི་:ཀཱ་ད་ཤི་ཙིཀ། ::ཀོ་ཤི་ཀཱ་ད་ཤ་ཀི། ཀོ་ཤཱི་དི་ཀ། ཤིན་ཏུ་འཕེལ་བ་དང་ཐ་མའི་བུ་ལོན་སྦྱིན་པར་མི་ནུས་:པའི་ཀུ་ཤི་སྐྱིད་སྟེ་བུ་ལོན་་་བྱེད་པ་ཉིད་ཀུ་ཤི་དར་མངོན་པར་བརྗོད་:དོ། །དེ་འཛིན་པར་བྱེད་པ་ནི་མལླ་སྟེ་སྨད་པའོ། །དྷརྨ་ཨ་དྷརྨ་ལས་སྤྱོད་དོ། །།མངོན་པར་བརྗོད་པ་གཉིས་་པའི་མཐའ་ལས་ཋཀ་འགྱུར་ཏེ། :ཆོས་སྤྱོད་པས་བསྟེན་པར་བྱེད་པ་ནི། དྷཱརྨི་ཀ༑་ཨཱ་དྷརྨི་ཀ། པྲ་ཏི་པ་ལས་ཡཏ་ལ་ཁན་འང་ངོ༌། །ཋཀ་འགྱུར་ཏེ། འང་ཡིག་ལས་:ཋན་འང་ངོ༌། །ལམ་སོ་སོ་ཞེས་པ་ལ་པྲ་ཏི་:པ་ཟི་ཀ། པྲ་ཏི་པ་ཋི་ཨེཀཿ། ་འདུས་པའི་དོན་ལས་འདུ་བར་བྱེད་དོའི་དོན་ལ་གཉིས་པའི་མཐའ་ལས་ཋཀ་འགྱུར་ཏེ། འདུས་པ་འདུ་བར་བྱེད་དོ་ལ། སཱ་མཱ་ཛི་ཀ། སཱ་མཱ་:བཱ་ཡིཀ། ས་མུ་ཧི་ཀ། ས་མུ་དཱ་:ཡ་ཀི། འདུ་བར་བྱེད་དོ་ཞེས་་པ་ནི་དེའི་ཕྱོགས་གཅིག་ཁྱིལ་འགྱུར་རོ་ཞེས་པའི་དོན་ཏོ། །པ་རི་ཥད་ལས་ཅུའོ། །འདུ་བར་བྱེད་དོ་ལ་འགྱུར་ཏེ། འཁོར་འདུ་བར་བྱེད་དོ་ལ། པཱ་རི་ཥ་། ཕྱོགས་གཅིག་རྣམ་པར་འགྱུར་བ་ནི་གཞན་མ་ཡིན་པ་བཞིན་ནོ་ཞེས་པའི་རིགས་ཀྱི་པས་པནྟད་ལས་ཀྱང་འགྱུར་ཏེ། པཱནྟ་༄༅། །སེ་ནཱ་ལས་ཡང་ནའོ། །སེ་ནཱའི་—།
