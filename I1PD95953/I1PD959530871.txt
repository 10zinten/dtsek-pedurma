སྒྲ་མདོ། རིའིདང་། ཡོད་ཀྱང་། ཁ་ཊི་ག་བ་ག་གེ་མོ་གཡོ་བ་ལྟ་བྱེད་དུ་ཁ་། ་་། །བ་ཚུལ་ཁ། །ཎི་ཛི་ཏ་ལས་གོང་དུ་ཨཎའོ༄༅། །།དི་ཏི། ཌཨ་དི་ཏི། ཨཱ་དི་ཏྱ། ལ་ཡ་མ་ལས་ཅུའོ། །།།པ་ཏི་ལས་ཨ་ཤྭ་ལ་སོགས་པ་མ་ཡིན་པ་དང་པོ་ལས་སོ། །།ལ། སྦུ་མན་ལས་ཨའོ། །།ལོ་མན་ལས་བུ་རྒྱུད་རྣམས་ལའོ། །།པྲི་ཐི་ཝཱི་ལས་ཉའོ། །།ཨུཏྤ་ལ་སོགས་པ་རྣམས་ལས་ཨཉའོ། །དེ་བ་ལས་སོ། །ཡཉའོ། །།།བ་ཧིས་ལས་ཐི་ཀཀ་ཡང་ངོ༌། །བུ་རྒྱུད་མ་ཡིན་པ་ལ་གྲངས་དང་པོ་ལས་བགྲང་བྱ་ལས་ཨཙ་ལ་སོགས་པའི་གཉིས་མ་ཡིན་པ་འཇིག་གོ། །བཏི་ལས་གོང་དུ་ཨགྣི་ཀ་ལི་དག་ལས་ཆ་ཀའོ། །སི༑སུཾ་སཱ་དག་ལས་ནཉ་སྣཉ་དག་གོ། །དངོས་པོ་ལ་ཡང་ནའོ། །གོ་ལས་ཨཙ་ལས་ཡཏ་དོ། །དེའི་བུ་རྒྱུད་དོ། །དང་ཉིད་ལས་སོ། ་། བུད་མེད་མ་ཡིན་པ་ལ་གུ་རུ་ཨཱ་ཡཏྟ་ལ་བུའི་བུ་ལ་སོགས་པ་ལས་སོ། ་། ཨཏ་ལས་ཨིཉའོ། །བཱ་ཧུ་ལ་སོགས་པ་རྣམས་ལས་རིགས་ཀྱི་དང་པོ་ལས་སོ། །བུ་ས་ལ་སོགས་པ་རྣམས་ཀྱི་ཨ་ཀང་ཡང་ངོ༌། །བི་ད་ལ་སོགས་པ་རྣམས་ལས་ཨཉའོ། །དྲང་སྲོང་བུའི་བུ་ལ་སོགས་པ་ལའོ། །།གརྒ་ལ་སོགས་པ་རྣམས་ལས་ཕཉའོ། /། མ་དྷུ་ལས་བྲམ་ཟེ་ལའོ། །བ་བླུ་ལས་ཀོ་ཤི་ཀ་ལའོ། །ཀ་པི་ལས་ཨོཾ་གི་ར་ས་ལའོ། །བྷོ་དྷ་ལས་སོ། །།།་་་ཏཎྜ་ལས་སོ། ་། བུད་མེད་ལ་འཇིག་གོ། །ཨཤྭ་ལ་སོགས་པ་རྣམས་ལས་ཕཉའོ། །བྷརྒ་ལས་གྲོང་ཁྱེར་གསུམ་པ་ལའོ། །ཀུཉྫ་སོགས་རྣམས་ལས་ཕྱཉའོ། ་། བུད་མེད་དང་མང་པོ་རྣམས་ལ་ཕཀའོ། །ན་ཌ་ལ་སོགས་པ་རྣམས་ལས་སོ། །ཧ་རི་ཏ་ལ་སོགས་པ་རྣམས་ལ་ཨཉ་ལས་སོ། །ཡཉ་ཨིཉ་ལས་སོ། །ལོ་། ཤ་ར་དྭཏེ། ཤུ་ན་ཀ། དབྷ་ལས་དང་བྷརྒ་བ། བཱཉྩ། ཨ་གྲྭ་ཡཎ་རྣམས་ལའོ། ༈། སཊྚ་ཏ། ཛཱི་བནྟ་ལས་ཡང་ནའོ། །དྲོཎ་ལས་སོ། ཆུ། ཤི་བ་ལ་སོགས་པ་རྣམས་ལ་ལས་ཨཎའོ༑། །ཆུ་བོ་དང་མིའི་མིང་ལས་ཨཙ་དང་པོ་ཨཱཏ་ཨིཙ་མ་ཡིན་པ་ལས་སོ། །—།
