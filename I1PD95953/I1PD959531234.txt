སྦྱོར་བ་པོའི་བྱ་བ་ལ་ནི་ཙའོ་ཞེས་སྒྲུབ་པར་བྱེད་པ་དེ་ལ་ནི་འདི་ལྟར་འགྱུར་ཏེ། བྷཀྵ་ཡ་ཏི་པཎྜིནྡེ་པ་དཏྟེ་ན་ཡཛྙཱ་དཏྟ། སྔ་མའི་ཎི་ཙོ་ཎ་ར་ཎི་ཊི་ཞེས་པས་དབྱིའོ། །ཟས་ཀྱི་དོན་ལས་གཉིས་པ་ཐོབ་པ་ལ། ཐ་ཀླུ་ར་ཧི་ན་སཱ་ཡཱམ། ཞེས་པའི་དགག་པ་ལས་ཏཾ་ལས་ནི་དགག་པ་མེད་དོ། །བྷཀྵ་ཡནྟི་པ་ལི་བནྡྷན་དེ་བ་དཏྟདེ་ལྟ་བུ་རྣམས་ནི་བཤད་པའི་བྱེད་པ་པོའི་རབ་ཏུ་སྦྱོར་བའོ། །མ་བཤད་པའི་ལས་་ཀྱང་བརྗོད་པར་བྱ་སྟེ། ནཱ་མྱ་ཏེ་གོ་གྲྭ་མནྡེ་བ་དཏྟོ། གོསྒྲ་མ་དེ་བ་དཏྟེན་ཡཛྷ་དཏྟ༑་པཁཱ་རྱ་ཏ་ཨ་སྦེ་ན་གྷ་སཾ་དེ་བ་དཏྟེ་ན། ཨཱ་རྱ་ཏེ་ཨོ་ད་ནཾ་དེ་བ་དཏྟ་ན་་་ཡཛྙཱ་དཏྟ། །ཡོ་བཱ་ཉྩ་ཏེ་གྲྭ་མཾ་དེ་བ་དཏྟེ་ན་ཡཛྷ་དཏྟ། །བླ་བྱ་ཏེ་བྷ་རོ་པ་ལཱི་བརྙེ་གྲྭ་མོ་དེ་བ་དཏྟེ་ན། བྷ་ཏྱནྟེ་བྷཱ་རཾ་བ་ལི་བནྡྷ་དེ་བ་དཏྟེ་ན། བྷཀྵནྟེ་བིཊྚཾ་དེ་བ་དཏྟེ་ན་ཡཛྷ་དཏྟེ་ན། ༼ཧི་སཱ་ཡཾ་པུ་ནབྷཀྵནྟེ་བ་ལི་བནྡྷན། ་ཡ་བ་དེ་བ་དཏྟེ་ན། ཞེས་པ་རྣམས་ཏིང་གི་རབ་ཏུ་སྦྱོར་བ་རྣམས་བཤད་ཟིན་ཏོ། །ཨ་ཛཱ་ནི་ཏ་བྱཱ་ན་ཡ་ནཱི་ཏཱ་བཱ་དེ་བ་དཏྟེ་ན། དེ་བ་དཏྟ་སྱ་བཱ། །བྷ་རོ་བོ་ཌྷ་བྱོ་ཧཱི་ཡོ་བཱ་ཧྱ་ཨུ་རྣོ་བཱ་་་གྲྭ་:མཾ་ཙི་ཙིཏྲ་ཎ། །ཀུནྡྷོ་ལེཧརྟ་བྱ་ཧ་ར་ཎི་ཡོ་ཧཱརྨོ་ཧྲི་ཏོ་བཱ་གྲྭ་མམྨིཥྛ་ྋམི་ཊེ་ཎ། ཨ་ཀནྡྷུ་རཱ་ཛཱ་ཨཱ་ཀརླུ་ཨཱ་ཀནྟ་ཎི་ཡ། ཨཱ་ཀྲྀཊོ་བཱ། གྲྭ་མོ་མ་ནུ་སྐྱེཎ། ་ཡཛྷ་་་དཏྟོ་ཛེ་ཏ་བྱོ་ཛ་ཡ་ནཱི་ཡོ་ཛེ་ཏོ་བཱ་ཤ་ཏནྟེ་བ་དཏྟེ་ན། གནཱཎྜ་ཡ་ཏི་བྱོ། དཎྜ་ཡི་ཏཱ་དཎྜ་ནི་ཡོ། །དཎྞཎྜི་མུ་བཱ་ས་ཏཾ་པྲཧྨ་དཏྟེ་ན། དོགྷ་བྱོ། དོ་ཧ་ནཱི་ཡོ། དུ་ཧཱ། དུ་བྷཱ་བཱནྟི། གོ་ཀཱ་ཡཱ་སི་གོ་པཱ་ལ་ཀེན། ཡཱ་ཙི་ཏ་བྱོ། །ཡཱ་ཙ་ནི་ཡོ། ཡ་སྦྱོ༄༅། ཡཱ་ཙི་ཏཱ་པཱ། ཀམྤ་ལ་མཱི་ཤཱ་རོ་བྲཱཧྥ་ཎེ་ན། རོདྡྷ་བྱོ། རོ་དྷ་ནཱི་ཡོ། རོ་དྷཱ། རུདྡྷ་ྋབ། ་བྲ་ཛོ་གོ་པཱ་ལ་ཀེན?། པྲཉྫ་བྱོ། ཕྲིབྦོ་བཱ། དྷརྨ་ཤི་བྱེ་ཎ་གུ་རུ། བྷཀྵི་ཏ་བྷོ། བྷིཀྵི་ཎི་ཡོ། བྷིནྡྷོ་ཏོ། བཱ་བཱཧྨ་ཎེ་ན་བསྟི་ཎི། ཙེ་ཏ་བྱཉྩ་ཡ་ནཱི་ཡཱཉྩ་ཡཱཉྩི་ཏོ་བཱ། བྲྀཀྵ། ཕ་ལཱ་ནི་དེ་བ་དཏྟེ་ན། བལྟོ་བྱོ། —།
