བསྟན་འགྱུར། སྒྲ་མདོ། རེ། ཆ་བླང། མ་འཆར་བ་ལ་སོགས་པའི་སྣང་བས་ཕྱོགས་དང་མཚམས་ལ་སོགས་པ་རྣམ་པར་་་་འབྱེད་པར་བྱེད་པ་རྣམ་པར་གཞག་གོ། །གཞན་དུ་ན། དཀྱི་ཎ་ཨི་མེ་ག:：ཐ་ཀཱ༑། རབ་ཏུ་སྒྲོགས་ཞེས་པའི་དོན་ཏོ། །ཨུཏྟཱ་རཱཉྩི། ཀུར་བ་ཞེས་པ་མིང་ལ་བཀག་པའི་ཕྱིར་སརྦ་ནཱ་མའི་བྱ་བ་མི་འཇུག་གོ། །།སཱ་ནི་བདག་དང་། བདག་གི་ཚེ་ནིའི་སྒྲ་ལས་གཞན་ལ་ལྟོས་པ་མེད་པར་མིང་གི་ངོ་བོས་ཤེས་པ་ན་ནོར་ལ་ཡང་འཇུག་གོ། །དེའི་ཚེ་སྭཱ་ཡ་ཏིཡ་ཏཾ། ཤེས་པའི་ཆེད་དུ་ཞེས་པའི་དོན་ཏོ། །དྷ་ནེ་པྲ་བྷཱུ་ཏཱ། སཱ་ན་དཱི་ཡནྟེ་པྲ་བྷཱུ་:ཏཱ། སཱ་ཨིན་བྷུ་ཛཱནྟེ། ཨན་རཾ་ནི་ཕྱི་རོལ་དང་སྦྱོར་བ་དང་འགེབས་པ་དག་ལའོ། །ཕྱི་རོལ་ཞེས་པ་མ་་་བསྒྲིབས་པའི་ཕྱོགས་བརྗོད་པའོ། །དེས་ན་སྦྱོར་བ་དང་ཕྱི་རོལ་སྦྱོར་བ་དེ་ཡང་མ་བསྒྲིབས་པའི་ཕྱི་རོལ་གྱི་དངོས་པོར་འགྱུར་རོ། །ཨུ་པ་རི་སནྡྲི་ཡ་ཏེ་ཞེས་པ་ཉེ་བར་འགེགས་པའོ། །འདི་དག་གི་དོན་ལ་ཨནྟ་རའི་སྒྲ་སརྦ་སོགས་སོ་དཔེར་ན། ཨནྟིརེ་གཾ་ཧཱ། ཨནྟ་ར་ཤྲཱི་ཧཱ། ་གྲོང་ཁྱེར་གྱི་ཕྱི་རོལ་གྱི་གཏུམ་པོ་ལ་སོགས་པའི་ཁྱིམ་བརྗོད་པའོ། །།ཨནྟ་རེ་ཤཱ་ཊ་ཀཱ། ཨནྟ་རཱ། 《ཤཱ་ཊ་ཀཱ། ཕོ་བྲང་ཉིད་བརྗོད་པའོ། །ན་པྲ་བ་ར་ཎཱི་ཡཾེ། ཕྱི་རོལ་སྦྱོར་བ་དང་འགེབས་པ་དག་་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨ་ན་:ཡོ་རྒྱུ་མ་ཡོ་རནྟ་ཏེ་ཙིཏཱ་པ་ས། པྲ་ཏི་བ་ཁྱིས་ཏི། ཨ་པུ་རི་ཏི་:པཀྟ་བྱི། ཨནྟ་རཱ་ཡུང་ཀེཔུ་རི་བ་ས་ཏི། ཏད། :ཏད། ཡད༑ཨི་དམ། ཅིཨེ་ཏད། ཨེ་ཀ། དྲི། ཡུད། ཨསྨད། བྷ་་་བནྟ༑ཀིམ། རིའི་སྨིནའོ། །སརྦ་ནཱ་མ་རྣམས་ལས་འགྱུར་ཏེ། སརྦ་སྨིན། ཨཏ་ལས་ཞེས་པ་ཉིད་དུ་འགྱུར་ཏེ༄༅། ཛས་ཀྱི་ཤིའོ། །སརྦ་ནཱ་མ་ལས་ཕྱི་་་་མའི་ཛས་ནི་ཤཱིའི་དངོས་པོར་འགྱུར་ཏེ། སཟེ། བི་ཤུ། ཨཏ་ལས་ཞེས་པ་ཉིད་དེ། སརྦ་རྣམས་སོ། །གཙོ་བོ་མ་ཡིན་པ་དང་མིང་ལས་གཞན་ཡང་མ་—།
