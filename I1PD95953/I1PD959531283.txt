བསྟན་འགྱུར། སྒྲ་མདོ། རིའིརྣམས་ལས་ནིའོ། །ཐུ་ཐོབ་པ་ལའོ། །བྷ་ཛ། ཤི་བསྟེན་པ་ལའོ། །ལོ་། ཡུ་བསྲེ་བ་ལའོ། །།བ་ཧ་ཐོབ་པ་ལའོ། །འདི་རྣམས་ལས་ནི་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨོ་ཉིད་དང་ཨེ་ཉིད། ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །ཐུ་ཎིཡརྐེད་པའམ་ཕྱོགས། ཐེ་ཎི། ཡཕྲེང་བ། ཡོ་ཎི། མོ་མཚན་ནམ་སྐྱེ་གནས། བསྟྲཱི། མེའོ། །པཎྜི་ལ་སོགས་པ་རྣམས་སོ། །པྲྀ་ཥཱ་པྲྀ་ཥཱ་བཀྲུ་བ་ལའོ། ་་། སྲི་འགྲོ་བ་ལའོ༑། ང་། བྷཱ་ཡོད་པ་ལའོ། །བྷོཉ་འགེངས་པ་ལའོ། །ཙུ་ར་རྐུ་བ་ལའོ། །།ཏུ་ར་མྱུར་བར་བྱེད་པ་ལའོ། །ཛཾ་ཥ་རྒས་པ་ལའོ། །ཀོ་། འདི་རྣམས་ལས་ནི་ཕ་་་རོལ་དང་ཚིག་ཕྲད་མཐའོ། །དེ་ལ་པྲ་ཏིའི་ཨཱ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །པཎྜི༑ཡརྟིང་པ། བྲྀ་ཏིའི་ཨཏ་ཨེང་དགག་པ་ཚིག་ཕྲད་དོ། །བྲྀཏྟི། དགྲ་བོ། ཤྲཱི་ཎི། ཡའོད་ཟེར། སྲྀ་ཎི། ཡལྕགས་ཀྱུ། བྷུ་ནི། ཡསའོ།  ོ། བི་བྷ་རྟིའི་སྤུ་ཉིད་་་དང༌། ར་ཕ་རོལ། བྷུཎི། སའི་བདག་པོ། ཙུ་རིའི་རིང་པོ་ཉིད། ཙུཎི། ཡཞིབ་མོ། ཏུ་རིའི་རིང་པོ་ཉིད། ཏཱུརྞཱི། མྱུར་བ། །ཛི་ཞེས་པ་འདིའི་ཨུ་ཉིད་དང་ར་ཕ་རོལ་ཉིད། བྱིངས་ཀྱི་རབ་དག་ལ་ཨ་ཙ་མེད་པ་དག་ལ་ཨི་ཀའི་རིང་པོ་ཉིད། ཛཱཎཱི། གཏུན་ཤིང༌། དེ་བཞིན་དུ་སྒྲ་གཞན་ཡང་ཤེས་པར་བྱའོ། །བྲི་དྲྀ་དག་ལས་བིནའོ། །བྲོཉ་སྒྲིབ་པ་ལའོ། ཊ། ཉ་ཡིག་ནི་ཉིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། དྲིང་མ་གུས་པ་ལའོ། །།ང་ཡིག་ནི་ཏང་ལྡན་པ་ལས་ཞེས་པའི་རྟགས་དོན་ཏོ། །དེ་དག་ལས་བིན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་དབྱངས་དོན༑ཨི་ཀའི་ཨེང་ཞེས་པས་ཨ་ཉིད་དོ། །ར་ཕ་རོལ། །བརྗི། ཤིང་རྟའོ། །དངི། སཱརྡུ༄༅། ཛཱ་གྲོ་ལས་ཀྭིནའོ། །ཛ་གྲི་གཉིད་ཟད་པ་ལའོ། །ཨཀ་ཕྱིས་པ་དང་ཤཱ་སུ་རི་དིཏ་རྣམས་ཀྱི་མ་ཡིན་ནོ་ཞེས་པས་རི་@ཡིག་དབྱི་བར་་་མི་འགྱུར་རོ། །དཱི་ཙྪཱཏི་བེ་བཱི་དཱ་རི་དྭ་ཨུཎུ་དང་ཛཱ་གརྟིའི་འང་ཙི་རེ་ཛི་ཨ་ཙའི་་།
