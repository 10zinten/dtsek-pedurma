རྀ་འགྲོ་བ་ལའོ། །འདི་ལས་པི་སན་ཕ་རོལ། ན་ཡིག་དབྱངས་དོན། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དོ།  ོ། པཇི་སཾ། ཤའི་མཆོག། བྲི་བྷཾ། བ་མི། ཀུ་་རྣམས་ལས་ཤཀའོ། །བྲོཉ་སྒྲིབ་པ་ལའོ། །ཤྲཱི་འཛིན་པ་ལའོ། །རྔུ་བ་མ་སྐྱུག་པ་ལའོ། །ཀུ་སྒྲ་ལའོ། །འདི་རྣམས་ལས་ཤཀ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང༌། །ཀ་ཡིག་ཨཏ་ཨེང་དགག་པའི་དོན་ཏོ། །ཐཾ་ཐཾ། མ་ལ་ཀཾ། བྷཾ་ཤཾ། མྱུར་བ། མ་། ཡིག་གི་རྗེས་སུ་ང་རོ། །ཁཾ་ཤ། ཡའོད་མ། །ཀུ་ཤ། རྩྭ་དྲུར་བ། ཀཱི་ནཱ་ཤ། དཱ་ཤ། ཨོཾ་ཀུ་ཤ་རྣམས་སོ། །ཀྲྀ་ཥ་རྣམ་པར་རྐོ་བ་ལའོ། །དཾ་ཤ་མཆེ་བ་ལའོ། །ཨ་ཀི་མཚོན་བྱེད་ལའོ། །ཨི་དབྱི་བ་དང་ཨི་དིཏ་ཀྱི་ནུམའོ། །འདི་རྣམས་ལས་ཤཀ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་ཚིག་ཕྲད་ལས་ཨི་ཉིད་དང་ཥ་ཡིག་གིའང་ན་རུ་བསྒྱུར་རོ། །ཀཱི་ནཱ་ཤ། རལ་གྲི། ཀྲྀ་པ་ཎ། བརྩེ་བ། དན་ཤའི་ཤ་ཡིག་དང་ན་དབྱི་བ་དང༌། དའི་ཨཱ་ཉིད་དོ། །དཱ་ཤ། ཡཉ་པ། ་ཨངྒི་ལས་ཨུཀ་ཨཱ་ག་མའོ། །ཀ་ཡིག་་་་ཀིཏ་བྱ་བའི་དོན། ཨརྐུ་ཤ། ལྕགས་ཀྱུའམ་གླང་པོ་ཆེ་བསྐུལ་བ། རཾ། མཉྫི། པཱི་ཡི། ཧ་ནི། །ཨངྒི་རྣམས་ལས་ཨུ་ཥནའོ། །རྀ་འགྲོ་བ་ལའོ། །མཉྫི། པཱི་ཡི་མདོའི་བྱིངས་སོ། །།ཧ་ན་འཚེ་བ་ལའོ། །ཨ་གི་འགྲོ་བའི་དོན་རྣམས་སོ། །འདི་རྣམས་ལས་ཨུ་ཥན་ཕ་རོལ། ན་ཡིག་དབྱངས་དོན། :ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །ཨ་རུ་ཥ། ཙནྡྲ་མ། ཆ་པའོ། །མཉྫ་ཥཱ། ཤིང་སྒམ།  ིཔཱི་ཡུ་ཥ། བདུད་རྩི། ཧ་ནུ་ཥ། ཡསྟག་། ཨདྷུ་ཥ། གླང་པོ་སྲུང་བ། སྲྀ་ལས་ཀུ་ཥནའོ། །རྀ་སྐྱོང་བ་ལའོ། །འདི་ལས་ཀུ་ཥན་ཕ་རོལ། ན་དབྱངས་དོན་ཏོ། །ཀ་ཡིག་ཀིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། མཆུ་འབྱུང་ལས་ཨུཏའོ་ཞེས་པས་ཨུ་ཉིད་དོ། །རཀའི་ཨན་ལ་ར་ལ་ཞེས་པས་ར་ཉིད་དོ། །པུ་རུ་ཥ། སྐྱེས་བུ། ཀྲྀ། ཀྲྀ་དག་ལས་ཨཱི་ཥནའོ། །ཀྲྀ་རྣམ་པར་གཡེང་བ་ལའོ། །།
