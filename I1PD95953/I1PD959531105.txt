བསྟན་འགྱུར། སྒྲ་མདོ། རེལའོ༑། ཤི་སུཊ་ལའོ། །གསུམ་པ་ལ་སོགས་པ་ལ་སྐྱེས་པ་དང་མཚུངས་སོ། །:：ཏད། ཏཏི། ཏེ། །ཏཱ་ནི། :：ཏུད། །ཏུཏ། ཏེ། ཏྭཱ་ནི། ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ་། :། འདི་ལྟར། །ས། ཏོ། »ཏེ། ཏཾ། ཏོ། ཏཱན༑ཏེ་ན། ཏཱ་ཧཱུཾ། ཏི། ཏསྨི། ཏཱ་ཤྩ། ཏེ་། ༼ཏསྨཱཏ། ཏཱ་ཧཱུཾ། ཏེ་བྷྱ། །ཏི་སྱ། ཏེ་ཡོ། ཏེ་ཊཾ། །ཏ་སྨིན། ཏ་ཡོ། །ཏེ་ཥཱ། བུད་མེད་ལ། སཱ། ཏེ། ཏཱ། ཏྭཾ། ཏེ། ཏཱ། །མ་ནིང་ལ། ཏད། ཏེ༑ཏཱ་ནི། །ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །འདི་ལྟར། །ཡ། ཡོ༑ཌཡེ། ཡཾ། ཡོ། ཡེ། ཡེ་ན། ཡཱ་ཧཱུཾ། །ཡི། ཡསྨི། ཡཱ་ཧཱུཾ། ཡེ་ཤྩ། ་ཡསྨདི། ༄༅། ཡཱ་ཧཱུཾ༄༅། །ཡེ་ཧྥུ། »ཡ་སྱ། །ཡ་ཡོ། ་ཡེ་ཧཱུཾ། ཡ་་་སྨིན། །ཡ་ཡོ། ཡེ་ཏེ། བུད་མེད་ལ། "ཡཱ། ཡེ། ཡཱ། ཡཾ། ཡེ། ཡཱ༑། མ་ནིང་ལ། །ཡཏ། ཡེ། ཡཱ་ནི། ་ཡཏ། ཡེ། ་ཡཱ་ནི། །ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །འདི་ལྟར། ཨེ་ཏད་ཅེས་པ་འདིའི་སྔ་མ་བཞིན། །ཨེ་ཥ། ཨེ་ཏོ། ཨེ་ཏེ། ཨེ་ཏམི། ཨེ་ཏོ། །ཨེ་ཏཱན། ཨེ་ཏེ་ན། ཨེ་ཏཱ་ཧཱུཾ། ཨེ་ཏི། །ཨེ་ཏསྨི། ཨེ་ཏཱ་ཧཱུཾ། ཨེ་ཏེ་། ཨེ་ཏ་སྨད། །ཨེ་ཏཱ་ཧཱུཾ། ཨེ་ཏེ་། ཨེ་ཏ་སྱ། ཨེ་ཏ་ཡོ། །ཨེ་ཏེ་ཚྭ། ཨེ་ཏ་སྨིན། །ཨེ་ཏ་ཡོ༑ཨེ་ཏེ་ཞུ། བུད་མེད་ལ། ཨེ་ཏྭཾ། ཨེ་ཏེ། ཨེ་ཏཱ། ཨེ་ཏྭཾ། ཨེ་ཏེ། ཨེ་ཏཱཙེ། མ་ནིང་ལ། ཨེ་ཏད། ཨེ་ཏེ། ཨེ་ཏཱ་ནི། ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཨ་དས་ཞེས་པ་འདིའི། ཁྱད་རྣམས་ཀྱི་ཏས་ལ་སོགས་པ་རྣམས་ལའང་དྲིའི་བར་གྱི་ཨའོ། །ཨཏའི་ཨཏ་ཨེང་ལའོ་ཞེས་པ་ཕ་རོལ་གྱི་གཟུགས་སོ། ༈། ཨ་སོ། ་ཨ་སུ་ཀ། །ཨ་སུ་ཀོའོཞེས་པ་ཚིག་ཕྲད་དོ། །ཨེཙ་ལའོ། ཚེ་། ཛས་ཀྱི་ཤཱིའོ༄༅། །ཨ་ལས་ཨཏ་ཨེངའོ། །ཨམ་ལ་སྔ་མའོ། །།།།
