བསྟན་འགྱུར། སྒྲ་མདོ། རེས་ས་:ཛུ་ཁའི་རུའོ་ཞེས་པས་རུ་ཉིད་དོ། །ཨཏ་ེལས་ཧ་ཤ་ལའང་རུའི་འོ་ཞེས་པ་ཨུ་ཉིད་དོ། །ཨ་ལས་:ཨཏ་ཨེངའོ་ཞེས་པ་ཨོ་ེཉིད་དོ། །ཤར་ལ་ཡང་ནའོ་ཞེས་པ་ས་ཡིག་གོ། ?། རྣམ་པར་གཅད་པའོ། ཌི། ཙནྡྲ་མཱ། ཙནྡྲ་མ་སོ། ཙནྡྲ་མ་ས། །ཙནྡ་མ་སཾ། །ཙནྡ་མ་སོ། ཙནྡ་མ་ས། །ཙནྡ་མ་སཱ། ཙནྡྲ་་མོ་དྡྷཾ། ཙནྡྲ་མོ་བྷི། ་ཙནྡྲ་མ་སེ། ་ཙནྡྲ་མོ་ཧཱུཾ། ཙནྡྲ་མོ་བྷྱ། །ཙནྡྲ་མ་ས། ཙནྡྲ་མོ་ཧཱུཾ། ཙནྡྲ་མོ་བྱཱ། ཙནྡྲ་མ་སི། ཙ་མ་སོ། ཙནྡྲ་མ་ཧཱུཾ། ཙ་མ་སི། ཙནྡ་མ་སོ། ཌིཙནྡྲ་:མ་སུ། ཟ་ཧེ་ཙནྡྲ་མ། ཧེ་ཙནྡ་མ་སོ། །ཧེ་ཙནྡྲ་མ་ས། འདི་ལྟར། པུ་མ་ནཱ། ཞེ་པུ་མ་ན་སོ། །།པུ་ོམ་ན་ས། ཁ་དྭས་ཞེས་པ་འདིའི་སྐྱེས་པའི:;སུཊ་ལ་ཨུཀ་ཏིའི་འོ་ཞེས་པ་སྔོན་དུ་ནུམའོེ། །མ་ཡིག་ནི་:མི་ཏ་ཨཙའི་མཐའ་ལས་ཕ་རོལ་གྱིའོ་ཞེས་པ་རྟགས་ཀྱི་དོན་ཏོ། །ཨུ་ཡིག་ནི་བརྗོད་པའི་དོན་ཏོ། །ཀུན་དུ་བོད་པ་མ་ཡིན་པའི་སུ་ལའོ། །སུའི་འོ། །བརྩེགས་པའི་ཚིག་གིའོ། །ས་མ་ཧད་དག་གི་:ནུམ་ལའོལ། ་། རིང་པོ་ཉིད་དོ། །ཤས་ལ་སོགས་པ་ཨཙ་དང་པོ་ཅན་རྣམས་ལ་ཝསའི་ཝའི་ཨུ་ཉིད་དོ་ཞེས་པ་ལ་ཡིག་གི་ཨུ་ཉིད་དོ། །:ཡཎ་ཨིཀའི་འོ་ཞེས་པ་ཕ་རོལ་ཏུ་སྔ་མ་ཉིད་དོ། །ཀུ་ལས་ཀྱང་ཞེས་པ་ལ་སོགས་པ་རྣམས་ཀྱི་ལོཥ་ཉིད་དོ། །བ་སུ་:སྲམ་སུ་དྷཱས་ཏེརྣམས་ཀྱི་སའི་འོ་ཞེས་པ་ས་ཡིག་གི་སྐྱོད་ཡིག་གོ། །།ཨཙ་མ་ཡིན་པ་ལ་ཞེས་པ་་་་གཉིས་ཉིད་དོ། །བརྩེགས་པའི་ཚིག་གིའོ་ཞེས་པ་གཅིག་གི་:དབྱི་བའོ། །ཁར་ལ་ཛྷལའི་ཙརའོ། །རྣམ་པར་གཅད་པའོ། །བི་:དཱན། །བི་དྷཱནྡྷཾ། བི་རཱམྨ། བི་རཱཎཾ། བི་དྷཱནྡྷཾ། བི་དུ་ཥ། བི་དུ་ཥཱི། བི་དྭ་དྡྷཾ། 《་བི་དྭ་དྷི། བི་དུ་ཁེ། བི་རཱུདྡྷཾ། བི་དྭིདྡྷ། བི་དུ་ཥ། བི་རཱུདྡྷཾ། བི་དདྡྷ། བི་དུ་ཥ༑བི་དུ་ཁོ། བི་དུ་ཏཱཾ། བི་དུ་ཏི། བི་དུ་བྐ,། :བི་དཱནྟུ། ཧེ་།
