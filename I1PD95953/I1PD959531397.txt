བསྟན་འགྱུར། སྒྲ་མདོ། རེ། ཆགྷསྟནྡྷཱི༑སུམྦྷ། སྐནྡྷ། །སྐུམྦྷ། སྐུ་རྣམས་ལས་ཟླ་འང་ངོ༌། །།སྟནྡྷོ་ལ་སོགས་པ་བཞི་པོ་རྣམས་མདོ་ལས་བྱུང་བའོ། །སྐུ་ཀུན་དུ་འགྲོ་བ་ལའོ། །འདི་རྣམས་ལས་བྷ་རྐྱེན་དུ་འགྱུར་ཞིང་་འང་ངོ༌། །སྟ་བྷཱ་ཏི། སྟ་བྷཱུ་ཏི། སྐུ་ཏི། སུ་བྷཱུ་ཏི། སྐ་བྷཱུ་ཏི། སྐུ་བྷཱུ་ཏི། སྐུ་ནཱ་ཏི། སྐུ་ནོ་ཏི། ་མདོ་་་ལས་བྱུང་བ་:རྣམས་ལ་བྱུང་བའི་རྐྱེན་བྱེད་པ་ནི་ཡུལ་ཐམས་ཅད་ཉིད་བཤད་པའི་དོན་:ཏེ། སྟབྷཱ་སྟ་ཎྜི་ཏཱ་སྟབྷཱ། ཧལ་ལས་ཧི་ལ་ཤཱན་ཙའོ། །ཧལ་མཐའ་ལས་ཧི་ཕ་རོལ་ལ:༔བྷའི་ཡུལ་ལ་ཤཱན་ཙ་འགྱུར་ཏེ། པུ་ཥ་ཎ། རྣགྲོ་ཧཱ་ཎ། ཧལ་ལས་ཞེས་པ་ཅི་ཞེ་ན། ཀཱི་*ཌི་ཧི། །ཧི་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨ་བྷཱ་ཏི། ཞེས་པ་སཱ་དྷ་ཀཱིརྟིའི་ྋའཇམ་དཔལ་གྱི་སྒྲ་བརྡ་སྤྲོད་པ་དཔལ་ལྡན་ཀ་ལིངྒའི་་་དབང་ཕྱུག་བྱང་ཆུབ་སེམས་དཔའི་ཡུལ་གྱི་སློབ་དཔོན་རྒྱལ་པོའི་ལྷས་རྣམ་པར་བཀོད་པའི་འགྲེལ་པ་སྟོང་ཕྲག་བརྒྱད་པ་ལས་སྐབས་དང་པོའི་རྐང་པ་གཉིས་་་་པ་རྫོགས་སོ། །ཁ། །་་་་་་དང་པོ་ཞེས་ཚུར་་བདེ་བ་དོན་་་་:། །ཀུནས་་དང་ཟླ་བ་ལོ་བྱེད་པ་བྱུང་བ་ལ་ལུངའོ༄༅། །བྱུང་བའི་དོན་ལྟར་ཅིབ་ལ་བྱིངས་ལས་ལུང་འགྱུར་ཏེ། ཨ་ཀྲྀ་ཏ། ཨ་ཀྲྀ་ཥཱིཏཱ། ཨ་ཀནྟི་ཏ། འདི་ནི་འདས་པའི་དུས་བརྗོད་པ་སྟེ། བྱུང་བའི་སྒྲ་ནི་ས་ལ་སོགས་པ་བརྗོད་པའོ། །བྱིངས་ལས་ཞེས་པ་སྤྱིར་བཏང་བའི་ཕྱིར་དང་དེ་དགག་པ་འཇུག་པ་སོགས་ཀྱི་དབང་དུ་ཡང་བྱས་སོ། །དེ་ཡང་བྱ་བ་ཉིད་ཡིན་གཞན་མ་ཡིན་ཏེ་འདས་པ་ལ་སོགས་པ་སྒྲུབ་པའོ། །དེ་ཡང་དེ་དང་ལྡན་པས་གཞན་མ་ཡིན་ཞེས་པ་མཆོག་གི་དོན་ཏོ། །ཐུ། །སད། བས་ལས་ལིཊ་ཡང་ནའོ། །ད་ལྟར་བ་འཇུག་པའི་ཐི་ཎོ་ཏི་སོགས་ལས་ལིཊ་ཡང་ན་འགྱུར་ཏེ། ཨུ་པ་ཤུ་ཤྲ་བ། ཨུ་པཱ་སིཁྲོ་ཏཱི་ཏ། །ཨུ་:པཱ་ས་ད། ཨུ་པཱ་ས་ད་:ཏི། །
