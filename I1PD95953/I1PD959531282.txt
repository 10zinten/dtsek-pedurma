རོལ་ཉིད་དོ། །ཨ་རལླི། གླང་པོ་ཆེ། ཨ་ཛི་ལས་ཨ་ལི་ཙའོ། །ཨཉྫ་གསལ་བ་་ལའོ༑ག། ཨུ་ཡིག་ནི་སྦྲི་སྲུང་ཨུ་ངཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། འདི་ལས་ཨ་ལི་ཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཙ་ཡིག་དབྱངས་དོན། ནའི་ཡང་ངོ༌། །།ནའི་་་་ཞེས་པས་རྗེས་སུ་ང་རོའོ། །དེའི་ཡམ་དུ་བསྒྱུར་བས་ཉ་ཡིག་གོ། །ཨཉྫ་ལི༑ལག་པ་ཁ་སྦྱར་བ། རྀ་འྀ་སྲི་ཌྷ་དྷ་མི་ཨ་ཤི་ཨ་བི་ཀྲྀ་ཏི་གྲ་ཧ་རྣམས་ལས་ཨིནའོ། །རེ་འགྲོ་བ་ལའོ༄༅། །ཏཱཾ་རྒྱལ་བ་ལའོ། །སྲེ་འགྲོ་བ་ལའོ། །རྗེ་འཛིན་པ་ལའོ། །དྷ་མ་ཞེས་པ་མདོའི་བྱིངས་སོ། །ཨ་ཤ་ཟ་བ་ལའོ། །ཨ་བ་རཀྵ་སྐྱོང་བ་ལའོ། །།དྲྀ་ཏུ་འཇུག་པ་ལའོ། །གྲ་ཧ་ཉེ་བར་ལེན་པ་ལའོ། །།འདི་རྣམས་ལས་ཨ་ནི་ཕ་་རོལ་ཏུ་འགྱུར་རོ། །།ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དང་། ར་ཕ་རོལ་ཉིད་དོ། །ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །ཨ་ར་ཎི། མེའི་སྲུབ་ཤིང་། །ཏ་ར་་ཎི༑ཡརྒྱ་མཚོ། ་ས་ར་ཎི། ཡལམ། དྷ་ར་ཎི། འཛིན་མའམ་ས་གཞི། དྷ་མ་ཎི༑རྩ། ཨ་ཤ་ནི། ཐོག་རྡོ་རྗེ། ཨ་བ་ནི། སའོ། །ཡང་བ་མཐའ་ཉེའི་འོ་་་་ཞེས་པས་ཨ་ཉིད་དང༌། ར་ཕ་རོལ་ཉིད་དོ། །བརྟ་ནི། གཅོད་པར་བྱེད་པའི་རྫས། གྲ་ཧ་ཎི༑ཡཕྲེང་བར་གནས་པ། ཤཀྱི་པི་ལས་ཀིཏའོ། །།ཀླུ་པ་དེད་པ་ལའོ་ཞེས་པ་འདི་ལས་ཨ་ནི་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་ཀིཏ་བཞིན་ནོ། །དེས་ན་ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨཏ་ཨེང་དུ་མི་འགྱུར་རོ། །།།ར་ཥ་ལ་ཞེས་པས་ཎ་ཉིད། སྤྱི་པ་ཎི༑ཡརླུང་། ཤ་ཀི་ལས་ཨུ་ནིའོ། །ཤཀྑཾ་ནུས་པ་ལའོ། །ལཾ་སྔ་མ་བཞིན་དབྱིའོ། །འདི་ལས་ཨུ་ནི་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཤ་ཀུ་ནི། བྱའོ། །ཨ་གི་ལས་ནིའོ། །ཨ་ཀ། ཨ་ག་གྱ་གྱུར་འགྲོ་བ་ལའོ། །འདི་ལས་ནི་ཕ་རོལ་ལོ། །ཨགྣི®། མེའོ། །བཱི་ལས་ནིའོ། ན་། ཨིཎ་བཱི་བཱ་འགྲོ་བ་ལའོ། །འདི་ལས་ནི་ཕ་རོལ། ཎ་ཉིད་ཚིག་ཕྲད་དོ། །བེ་ཎི། ལྷས་མའམ་སྐྲ་བཅིངས་པ། ཐུ། །ཤི། །ཡུ། བ་ཧ། —།
