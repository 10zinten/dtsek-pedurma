བསྟན་འགྱུར། སྒྲ་མདོ། རིའིཡིབསྔལ་ཞི་བར་རབ་ཏུ་སྦྱོར་བར་ཤོག་་ཝེ། བརྡ་སྤྲོད་པ་ཙནྡྲ་པའི་ཡི་གེའི་མདོའི་་་འགྲེལ་པ་ཤློ་ཀ་བརྒྱ་བཅུ་དགུ་པ་རྫོགས་སོ། །།ཁཚོ། །མ་ཀུ་ཚོལ་བ་དག་ཁ་ཎི་ཡ་གལ་བོ་སློབ་དཔོན་ཆོས་སྐྱོང་གིས་མཛད་པའོ། །གནས་ཆེན་པོ་ཉེ་བའི་འཐུང་གཅོད་བལ་པོ་ཡམ་བུའི་གྲོང་ཁྱེར་གྱི་འཁོར་ཆེན་པོ་ཡུལ་སྐྲ་ཁར་སྒྲའི་མཁན་པོ་་་ཆེན་པོ་ཛེ་ཏེ་ཀརྞའི་ཞལ་སྔ་ནས་ལེགས་པར་མཉན་ནས། རྩ་བ་གུང་ཐངགི་ས་ཆ་སློབ་དཔོན་ལྷའི་སྐུ་མཁར་གདུང་ཕུགས་སུ་བསྒྱུར་ཅིང༌། མཁས་པ་ཆེན་པོའི་ཆེན་པོ་རྣམས་ཀྱི་གདན་ས་གཙུག་ལག་ཁང་ཆེན་པོ་དཔལ་ཐར་པ་གླིང་དུ་ཤཱཀྱའི་དགེ་སློང་ཉི་མ་རྒྱལ་མཚན་དཔལ་བཟང་པོས་ལེགས་པར་བསྒྱུར་བའོ། །བར། །ཞི། །།ཀ་ཀོ་བ་ལས་བྱེད་པ་སྦྱོར་བ་ལ་ཁའི་དུར་ཁཚེ་དབང་རབ་བོད་རིག་པ་གཏད་ཀྱི་་་་་ལ་ཚེ་དབང་། བསྡུར་མཆན། བག་མེད་ལ་རང་ཁ། ཛྫ་ན་ཁ་མ་ཀ་དབང་ལྔ་འོ། བྲི་ལ་དགེ་བ་ལ། ན་ན་ན་ཤཚོདྲིgsཁྲོ༌ཆག་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་ཡི་གེའི་མདོའི་འགྲེལ་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་»པཏྟི། «པེ་»«སྣར་གིས། «པེ་»«སྣར་»གེའི། ཀྱི«སྣར་ལ། «པེ་»«སྣར་རིམ། «པེ་»«སྣར་གི། ིན@«པེ་»«སྣར་བལྟོས། «པེ་(«པེ་»«སྣར་བཅད། @«པེ་»«སྣར་འབྱུང་། «པེ་»«སྣར་»བཅད། «པེ་»«སྣར་རྣམ་་་པར་བཅད་པ་རྣམས་ནི་ཚེག་དྲག་གོ། །དེ་རྣམས་མགྲིན་པའི་གནས་ནས་ཀུན་ཏུ་འབྱུང་བར་རྣམ་པར་ཤེས་སོ། །མགྲིན་པ་དང་ཀན་ལས་ཨིཏྲ་དང་ཨེཏྲ་དང༌། །ཨིཏྲ་རྣམས་སོ། །རྐུན་ལ་འབྱུང་བ་ནི་རྐན་ཅན་ནོ། །མགྲིན་པ་དང་རྐན་དག་ནི་མགྲིན་པ་དང་རྐན་གྱི་གནས་སོ། །དེ་རྣམས་ཨེཏྲ་ནི་ཨེ་ཡིག་གོ། ཚུ«པེ་»«སྣར་ཨེཏྲ་དང་ཨིཏྲའོ། །«པེ་»«སྣར་ཨུཏྲ་དང་ཨོཏྲ་དང་ཨོཏྲ། Q«པེ་»«སྣར་»ཨུཏྲ། «པེ་»«སྣར་»ཨོཏྲ། «པེ་»«སྣར་ཨོཏྲ། «པེ་ཨུད་དང་ཨུཏྲ་དང་ཨོཏྲ། 《སྣར་»ཨུཏྲ་དང་ཨུད་ཨོཏྲ། @«པེ་»«སྣར་+ཏེ། ༄༅«པེ་»«སྣར་»རི། «པེ་»«སྣར་»རི། «པེ་»«སྣར་»དང༌། ཙེ«པེ་»«སྣར་དྲི་ལྕེ། «པེ་»«སྣར་ཨཏྲ་ཞེས་པས་ཨཏྲའོ། །ཨགི་ཚྭ་ཏ་མི་ཏྱཱ་ཏ། ཏེ«པེ་»«སྣར་ཏྲེ་ར་ཐནྟར། «པེ་»«སྣར་རོ། །།
