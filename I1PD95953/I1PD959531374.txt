མུ་ཥ་གྲ་ཧེ་སཱ་པི་པྲཙྪ་ལས་སོ། །ཕ་རོལ་ཨིཊ་དང་བཅས་པའི་ལྟ་ལ་སན་དང་་་ཀིཏ་དག་འགྱུར་ཏེ། རུ་དི་ཏཱ། རུ་རུ་དི་ཥ་ཏི། བི་དི་ཏཱ། བི་བི་དི་ཥ་ཏི། མུ་ཥི་ཏཱ། ཁོ་མུ་མུ་ཎི་ཥ་ཏི། གྲོ་ཧི་ཏཱ། ཛི་གྷཾ་ག་ཏི། སུ་སུདྨ་ཏི། པི་པྲ་ཙྪི་ཥ་ཏི། ཀི་ར་སོགས་དང་ཨརྟི་སྨི། པུང་། ཨཉྫས། ཨ་ཤ་ལས་སན་ལས་ཞེས་པས་ཨིཊའོ། །་། སཱ་པི་པཙྪ་ལས་སན་ནི་དོན་བརྗོད་པའོ། །།ཀིཏ་ཉིད་ལས་་ཧི་ཥཱ་ཨིཊ་མེད་པའི་ཕྱིར་རོ། འ། ཨི་ཀ་ལས་ཨིཊ་མེད་པ་ལས་སོ། །ཨིཀ་མཐའི་་་བྱིངས་ལས་ཕ་རོལ་ཨིཊ་མེད་པའི་སན་གྱི་རྐྱེན་ལས་ཀིཏ་འགྱུར་ཏེ། ཙི་ཙཱ་ཥ་ཏི། ནི་ནཱི་ཥ་ཏི། ཏུནྟུ་ཥ་ཏི། ཙི་ཀཱིརྱ་ཏི། ཨིཏ་མེད་པ་ཞེས་པ་ཅི་ཞེ་ན། །ཤི་ཤ་ཡི་ཥ་ཏི། དེའི་མཐའ་ཉེ་ལས་སོ། །ཕྱི་མར་ཨཏ་མེད་པའི་་སན་ལས་ཀིཏ་འགྱུར་ཏེ། བི་བྷིནྡྲ་ཏི་བུ་:བྷུརྨ་ཏེ། །རི་སྐྱེལས་ཀྱང་:ལིང་ཕྱི་སྐྱེ་ཙ་དག་ཏང་ེལས་སོ། །ཨིཀ་མཐའ་ཉེ་དང་རི་ཡིག་མཐའ་ལས་ཀྱང་ཕ་རོལ་གང་ལིང་སི་:ཙ་དག་ཏང་དང་འབྲེལ་པ་དག་ལ་ཀིར་ཅེའགྱུར་ཏེ། བྷིསྭཱཧྥི། བྷུནཱི། འཨ་བྷིནྟཾ། ཨ་བུདྡྷ། ཧྲ་ཏཱིཊྚེ། རུ་ཨ་ཀྲྀ་ཏ། ཨ་བྲྀ་ཏ། ཏཱིརྟིབྦ༈ན་དཱི། ཨ་:ཏཱིནྡྷ། དྷི་དང་སྦྲུ་ལ་ཨིཏ་དང་སིའང་ངོ༌། །དྷིའི་མིང་ཅན་རྣམས་དང་ཏིཥྛ་ཏི་ལས་ཕ་རོལ་སི་དང་ཀིཏ་འགྱུར་ཏེ་མཐའ་ཨིཏ་དུ་་་བསྒྱུར་སེབའང་ངོ༌། །::ཨ་དི་ཏ། ཨ་དི་:ཥ་ཏཾ། ཨ་དྷི་ཏི། ཨ་དྷི་ཥ:：ཏཾ། པྲ་སྦི་ཏ་པྲ་སྦི་ཥ་ཏཾ། ཞེས་པ་སཱ་དྷུ་ཀཱིརྟིའི་འཇམ་དཔལ་གྱི་བརྡ་སྤྲོད་པ་དཔལ་ལྡན་ཀ་ལིངྒའི་དབང་ཕྱུག་བྱང་ཆུབ་སེམས་དཔའི་ཡུལ་གྱི་སློབ་དཔོན་རྒྱལ་པོའི་ལྷས་རྣམ་པར་བཀོད་པའི་འགྲེལ་པ་སྟོང་ཕྲག་བརྒྱད་པ་ལས་སྐབས་དང་པོའི་་་རྐང་པ་དང་པོ་རྫོགས་སོ། །།ཚིག་། གསལ་བའི་ལ་ས་བདག་འབད་ན་ཏ་བློ་བལྟ་བ་དེ། ਸੰਨਲੇਸਮਤਲੈਲੀਸ਼ਾਨਬਾਲਿਗਲਾੜy།
