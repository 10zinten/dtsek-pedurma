བསྟན་འགྱུར། སྒྲ་མདོ། འིཆུ་བྱ་དེ་ན༑གེ་ཧ་མ་ནུ་པྲ་བི་བྱ་གྷོ་ལྟེ། ཡང་ཡང་ཞེས་པས་གྲུབ་པ་ལ་ཡང་ཡང་སྨོས་པ་ནི་བསྡུ་བ་རྣམ་པར་བརྟག་པའི་དོན་ཏོ། །དུས་ལས་ཨ་སུ་དང་ཀྲྀཥ་ལས་བར་ཆོད་པ་ལའོ། །ཨ་སྱ་ཏི་དང་ཏྲི་ཥྛ་ཏི་ལས་ཀྱང་གཉིས་པའི་མཐའ་ཅན་དུས་བརྗོད་པ་ཕ་རོལ་ལས་བར་ཆོད་པ་རྟོགས་པར་བྱེད་པ་ལ་ཎམ་འགྱུར་ཏེ། དྲུ་ཧཱ་ཏྱ་སཾ་གཱ། བཱ་ཡ་ཡ་ཏི། དྲུ་ཧ་མ་ཏི་ཀྲ་མྱ་འཐུང་དུ་འཇུག་ཅེས་པའི་དོན་ཏོ། །དུ་ཧ་ཏ་རྟ་གཱ། བ་ཡ་ཡ་ཏི། ཨ་སུ་ཚིགསཥ་ཞེས་པ་ཅི་ཞེ་ན། རཱུ་ཧ་མུདྲཱ་སྱ་བྷུ་ལྟེ། དུས་ལས་ཞེས་པ་ཅི་ཞེ་ན། པཉྩ་པཎྜི་ན་ཏྱཱ་སྱ་ཛ་ནི་པི་བ་ཏི། བར་ཆོད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨ་ཧ་ར་ཏྱ་སྐྱེ་ཥྛ་ཨིན་ག་ཏ། ཉི་མ་མངོན་པར་ཁྱབ་པའི་མདའ་རྣམས་འཕངས་ཏེ་སོང་ཞེས་པའི་དོན་ཏོ། །མིང་ལས་གྲྭ་ཧ་དི་ཤ་ལས་སོ། །མིང་ལས་གཉིས་པའི་མཐའ་ལས་ཕ་རོལ་གྲ་ཧ་དང་དི་ཤ་ལས་་་ཀྱང་ཎམ་འགྱུར་ཏེ། མིང་ནས་སྨོས་ཏེ་སྦྱིན་པ་ནི། །ནཱ་མཱ་ནི་གྲྭ་ཧ། མིང་བསྟན་ཏེ་སྦྱིན་པ་ནི། ནཱ་མཱིནཱ་དེ་ཤཾ། མི་ཟད་པ་ལས་མི་འདོད་པ་བརྗོད་པ་ལ་ཀྲྀཉ་ལས་ཀྟ་འང་ངོ༌། །མི་ཟད་པ་ལས་ཕ་རོལ་ཀྲྀཉ་ལས་མི་འདོད་པ་བཤད་པ་ལ་ཎམ་འགྱུར་ཏེ། ཀཱ ིའང་ངོ༌། །བཱཧྥ་ཎ་སུ་དྲ་སྟེ་ཛཱ་ཏ། ཅི་ལྟར་དམངས་རིགས་ཤེ་ཨིན། ནཱི་ཙི། ཀྲྀ་:ཏྱཱ་ཙིཋ། །ནཱི་ཙི། :ཀྲྀ་ཏྭཱ་ནི་ཙི། :ཀཱ་རཾ། ཨུཙྪི་རྣ་མ་པི་ཝིཡ་མཱ་ཁྱེ་ཡཾ། །ཏིརླཉྩ་ལས་ཨ་པ་བརྒ་ལའོ། །ཏིརླཀ་གི་སྒྲ་ལས་ཕ་རོལ་ཀྲྀཉ་ལས་ཀ་དང་ཎམ་དག་འགྱུར་ཏེ། ཏིརླཀ་ཀྲྀ་ཏྱཱ་ག་ཏ། ཏིརྨ་ཀླཙིཏཱ། ཏིརྨ་ཀཱ་རོ་ཡོངས་སུ་རྫོགས་པར་རྟོགས་པ་ཞེས་པའི་དོན་ཏོ། །ཨ་་པ་བརྒ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཏི་རྨ:ཀྑཾཏཱ་ཀཱ་ཎཾ་ག་ཏ། ་རང་གི་ཡན་ལག་ལས་ཏས་ལས་བྷུ་ལས་ཀྱང་ངོ༌། །ལྟ་དང་ནམ་དག་འགྱུར་ཏེ། མུ་ཁ་ཏ། ཀྲྀ་ཏྱཱ་ཡེ་ག་ཏ། །མུ་ཁ་ཏ། ཨིཀཱ་རཾ་མུ་ཁ་ཏོ་བྷཱ་ཡ། མུ་ཁ་ཏོ་བྷཱ་བ། ཁམུ་ཁ་ཏོ་བྷ་ལ་—།
