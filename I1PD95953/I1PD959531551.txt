བསྟན་འགྱུར། སྒྲ་མདོ། རེཁྱོཊའབྲེལ་པ་མེད་དོ། །འཇིག་པ་ལ་ཞེས་པའི་ཚིག་ལས་བ་གཉིས་པའི་ཀཱ་ལ་སོགས་པ་བོར་བའི་ཚད་ལས་འཇིག་པ་ལ་བུད་མེད་ལ་དཱིཔ་འགྱུར་ཏེ། དྲི་ཀུ་ཌ་བཱི། དཱི་:པྲ་སྒྲི༄༅། ཚད་ལས་ཞེས་པ་ཅི་ཞེ་ན། པཉྩ་བི། འཇིག་པ་ཞེས་པ་ཅི་ཞེ་ན༑དྭི་པ་ཅུ། འཇིག་པ་ཞེས་པ་ཡང་དེ་ཕན་ཉིད་ཡིན་གྱི་ཀྲྀཏ་མ་ཡིན་ལ། སུཔ་འཇིག་པ་ཐམས་ཅད་དུ་འབྱུང་བས་རྣམ་གཅད་མེད་པའི་ཕྱིར་རོ། ཞེ། ཀཱ་ལ་སོགས་དགག་པ་ཅི་ཞེ་ན། ལོ་གཉིས་སུ་གྱུར་པ་@ནི། དཱི་བནཱ། དཱི་བིསྟཱ། དྷཱ་ཙི་ཏཱ། ཧྲི་ཀམྤ་ཝཱ། དྲི་ཤཏཱ། སུ་བཎ་བི་སྟོ། ཧེ་:མོན་གྷཾ། ཨཱ་ཙི་ཏོ་ད་ཤ་བྷིནྡྷ་རི། ཤ་ཏིའི་ཤ་སྟི་བྷིནྡྷ་ར། པ་ལཱ་ནཾ་ཀམྤ་རྩ་མཱུརྞ་པ་ལ་ཤ་ཏཾ། ཀཱཎྜ་ལས་ཞིང་མ་ཡིན་པ་ལའོ། །:ཚད་བརྗོད་པ་ཀཱཎྜའི་སྒྲའི་མཐའ་ལས་བ་གཉིས་པ་ལས་ཞིང་མ་ཡིན་པ་འཇིག་པ་ཉིད་ལ་བུད་མེད་ལ་དཱིཔ་འགྱུར་ཏེ། ཁོ་ཌ་ཤ་ཧ་སྟ་པ་རི་མཱ་ཎཾ་ཀུཎྜ*། འདི་མདའ་གཉིས་ཀྱི་ཚད་ཡིན་པ་ནི། དྲི་ཀཱཎྜི་རཛྫཱ། ཞིང་མ་ཡིན་པ་ཞེས་པ་ཅི་ཞེ་ན། དཱི་ཀཱཎྜ་ཀྵེ་དྲ་ྋབྷཏྟི། ཚད་ལས་ཞེས་པ་ཉིད་དེ། མདའ་གཉིས་ཀྱིས་རྩེ་བ་ནི། ༼དཱི་ཀཱཎཱ། པུ་རུ་ཥ་ལས་་ཡང་ནའོ། །བ་གཉིས་པ་སྐྱེས་པའི་མཐའ་ལས་དཱིཔ་ཡང་ན་འགྱུར་ཏེ། ་ཙ་་་ཏུརྙསྟ༑པུ་རུ་ཥ། འདི་སྐྱེས་བུ་གཉིས་ཀྱི་ཚད་ཡིན་པ་ནི། དྭི་པུ་རུ་ཥཱལ། དཱི་པུ་རུ་ཥཱཙྪཱ་ཡཱ། ཏྲི་པུ་རུ་ཏི། ཏྲི་པུ་རུ་ཥཱ་པ་རི་ཁཱ། ཚད་ལས་ཞེས་པ་ཉིད་དེ༑། སྐྱེས་བུ་ལྔ་རྣམས་རྩེ་བ་ནི། པཉྩ་པུ་རུ་ཥ་ཙེཧསྟི་ནཱི། ཀེ་བ་ལ། མཱ་མ་ཀ། བྷཱ་ག་དྷེ་ཡ། པཱ་པ། ཨ་:པ་རཱི། ས་མཱ་ཝེན། :ཨཱརྨ་ཀྲྀ་ཏི། །སུ་མཾ་ག་ལ། བྷི་ཥ་ཛུ་ལས་མིང་ལའོ། །དགུ་པོ་རྣམས་ལས་མིང་ལ་དཱིཔ་འགྱུར་ཏེ། །ཀེ་བ་ལཱི་ནཱ་མ་ག་ཎི་ཏ་:བི་དྱི། མཱ་མ་ཀཱི་དེ་བ་ཏཱ། བྷ་ག་དྷེ་ཡཱི། པཱི''པཱི། ཨ་བ་རཱི། ས་མཱ་ནཱི། ཨཱརྱ་:ཀྲི་ཏཱིཙེ། སུ་མངྒ་ལཱི། བྷི་ཥ་ཛཱི། —།
