བསྟན་འགྱུར། སྒྲ་མདོ། རིཨཀRརྣམས་ལས་བུ་རྒྱུད་ལ་ཨཉ་°འགྱུར་ཏེ། །འདི་ཨི་ཉའི་དམིགས་བསལ་ལོ། །བཱ་ཤིཋ། བི་ཤྭ་མི་ཏྲ་ན་ཀུ་ལ། ཀུ་རུའི་རིགས་སོ། དུ། དེའི་བུ་རྒྱུད་ནི་ནཱ་ཀུ་ལ༑བ་སུ་དེ་བ། དྲྀའི་རིགས་སོ། །དེའི་བུ་རྒྱུད་ནི་བཱ་སུ་དེ་བ། །ཤྭ་ཕལྟ་:ན་ཨིནྡྲ་ཀའི་རིགས་ལས་འབྱུང་བའོ། །དེའི་བུ་རྒྱུད་ནི་ཤཱ་ཙིཕལྒ། དྲང་སྲོང་ཛཏ་སེ་ནཱ་དང་ཧི་མ་སེན་ནི་ཀུ་རུའི་འོ། །:བི་ཤྭ་ཀེ་ན་ནི་བྲྀཏྟིའི་འོ། །ཨུ་གྲ་སེན་ནི་ཨནྡྷ་ཀའི་འོ། །དེ་ལ་སེནའི་མཐའི་མཚན་ཉིད་ལ་རྒྱ་དང་ཉ་དག་འགྱུར་རོ། །བར་མ་ནི་དམིགས་བསལ་ལོ། །སྔ་མའི་སྒྲུབ་པ་བཀག་པ་ལས་ཕྱི་མའོ་ཞེས་པའི་རིགས་པ་ལས་སོ། །ཅི་ལྟར་:མ་ཏྲེ་ཡི། ཞེ་ན། ཨིཊྚ་ཉིད་ཀྱི་ཕྱིར་ཆ་ཀའོ། །གྲངས་དང་སཾ་དང་བྷ་སོགས་ལས་མཱ་ཏྲི་ལས་ཨུ་ཨང་འང་ངོ༌། །གྲངས་ལ་སོགས་པ་སྔོན་མའི་མཱ་ཏྲི་ལས་བུ་རྒྱུད་ལས་ཨིན་འགྱུར་ཞིང་ཨུ་ཨང་དུ་བསྒྱུར་བའང་ངོ༌། །དྲི་མཱུ་རུ་རཿ། ཏྲི་མཱ་རུ་ར། ཥཎཱ་རུ་རཿ༑སཾ་མཱ་རུ་ར། བྷ་དྲ་མཱ་རུ་ར། གྲངས་དང་སཾ་དང་བྷ་དཱ་སོགས་དང་པོ་ལས་ཅི་ཞེ་ན། བི་མཱ་ཏྲིའི་བུ་རྒྱུད་ནི། བི་མཱ་ཏྲི་ཡ། མཱ་ཏྲིའི་སྒྲ་ནི་་་སྐྱེད་བྱེད་མ་བརྗོད་པ་ཉིད་ལ་གཟུང་ངོ༌། །མཱན་ཀརྟི་ལས་ནི་མ་ཡིན་ཏེ། དེ་ལ་སཱི་མཱ་ཏྲ། ཀཅུ་ལས་ཀ་ནཱི་ན་ཡང་ངོ༌། ྋ། ཀ་རཱུའི་སྒྲ་ལས་བུ་རྒྱུད་ལས་ཨཎ་འགྱུར་ཞིང་ཀ་ནཱི་ན་རུ་བསྒྱུར་བའང་ངོ༌། ་། ཀཱ་ནཱི་ན། རྒྱས་པའོ། །།དེ་ཉིད་ཀྱི་ཕྱིར་ཡོངས་སུ་མི་འགྱུར་ཡང་ཀ་རཱུའི་སྒྲ་ཡིས་བསྟན་པར་བྱའོ། <། ནཱ་ཀླུ་ཏ་ཡོ་ནི་རེ་བ། ཤུངྒཙྪ་ག་ལ། བི་ཀརྨ་ལས་བྷཱ་ར་རཱ་ཛ། བཱཏྨ་ཨ་ཏྲེ་ཡ་རྣམས་ལའོ། །ཨཎ་འགྱུར་ཏེ། ཤུངྒའི་བུ་རྒྱུད་ནི། ཤོདྷི། གལ་ཏེ་བྷཱ་ར་དྷཱ་ཛ་ཡིན་ནའོ། །གཞན་ལ་ཤོདྷི། རྟགས་ཀྱི་ཁྱད་པར་སྨོས་པས་ཤུངྒའི་སྒྲ་ལས་ཀྱང་འགྱུར་རོ། །དེའི་ཚེ་གཞན་ལ་ག་ཤོངྒེ་ཡ། ཙྪ་ག་ལཙོ། གལ་ཏེ་བཏྟའི་བུ་ཡིན་ནའོ། །—།
