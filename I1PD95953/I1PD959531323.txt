བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིབ་ར། སྐྱུར་པོ། ཎིའི་ཨིཊ་མ་ཡིན་པའོ་ཞེས་པས་ཎི་དབྱིའོ། །བཱ་ས་ར། ཉིན་མོ། ཨ་གི། མ་དི?། མནྡི། ཀ་ཊི་ལས་ཨཱ་རནའོ། །ཨ་གི་འགྲོ་བའི་དོན་རྣམས་སོ། །ཨི་ཡིག་ནི་ཨི་དིཏ་ཀྱི་ནུམའོ་ཞེས་པའི་རྟགས་དོན། འདིས་ནུམའོ། །རྗེས་སུ་ང་རོ་ཡམ་དུ་བསྒྱུར་བས་ང་ཡིག་གོ། །མ་དཱི་དགའ་བ་ལའོ། །མ་དི་བླུན་པ་ལའོ། །ཨི་ཡིག་དབྱི་བ་དང་ནུམ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །ཀ་ཊ་་རྒྱགས་པ་ལའོ། །འདི་རྣམས་ལས་ཨཱ་རན་འགྱུར་ཏེ་ན་ཡིག་དབྱངས་དོན། ཨངྒཱ་ར། སོལ་བ། ་མནྡཱ་ར། ནོར་བུའི་བྱེ་བྲག། མནྡཱ་ར། ཤིང་གི་རིགས། །ཞུཀ་ཌཱ་རཿ༑དམར་པོ། ཐེངྒི། ་བྷངྒི། མྲྀ་ཛི། ་ཀརྫི་ལས་ཙི་ཏའོ། ་། ཐིདྡྷི། བྷངྒི་དག་གཉིས་པའི་གཅིག་གི་ཚིག་གོ། །སྒེག་པར་བྱེད་དོ། །།བྷིངྒི་བྱེད་དོ་ཞེས་པས་སུཔ་ལས་བྱིངས་དོན་ལ་མང་པོར་འདོད་པ་བཞིན། ཡང་ངོ་ཞེས་པས་ཎི་ཙའོ། །ཎ་ཡིག་དང་ཙ་ཡིག་དག་སྔ་མ་བཞིན་དབྱིའོ། །།འདོད་པ་བཞིན་ཡང་ངོ་ཞེས་པས་ཨ་ཡིག་དབྱིའོ། །ཁྲིདྡྷི། བྷིངྒི། ཚྭ་ཛུ་དག་པ་ལའོ། །།ཀཎྜ་ཞེས་པ་མདོའི་བྱིངས་སོ། །འདི་རྣམས་ལས་ཨཱ་རན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཅི་ཏ་བཞིན་ནོ། །དེས་ན་ཙི་དབྱངས་སུ་འགྱུར་རོ། །ཎིའི་ཨིཊ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཨི་ཡིག་དབྱིའོ། །ཁྲིདྷཱ་རཾ། སྒེག་པའམ་ལྟ་ན་སྡུག་པ། བྲྀངྒཱ་རཾ། གསེར་གྱི་རིལ་བ། ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨ་ཉིད་དང༌། ར་ཕ་རོལ་ཉིད་དོ། །ཚིགས་ཛིའི་ཨཱཏའོ་ཞེས་པས་ཨཱ་ཉིད་དོ། ་། མཱཙཱ་ར། བྱི་ལ། ཀསྠཱ་རཿ། ཡརྨ་བྱ། ཀ་་་མའི་ཨཏ་ཀྱི་ཨུཏ་ཀྱང་ངོ༌། །ཀ་མུ་མཛེས་པ་ལའོ། །།འདི་ལས་ཨཱ་རན་ཕ་རོལ་་ཏུ་འགྱུར་ཞིང་ཀའི་ཨཏ་ཀྱི་ཨུ་ཉིད་དོ། །ཀུ་མཱ་ར། གཞོན་ནུའམ་བྱིས་པའི་སྐྱེ་བོ། ཞྭ་ལས་ཀ་རཎའོ། །ཤྲཱི་འཚེ་བ་ལའོ། །འདི་ལས་ཀ་རན་འགྱུར་རོ། F། ན་ཡིག་དབྱངས་དོན། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད། རཾ་ཀའི་ཨན་ལས་ར་ལ་—།
