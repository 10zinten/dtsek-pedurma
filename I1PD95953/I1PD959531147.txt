བསྟན་འགྱུར། སྒྲ་མདོ། རིའིརང་གི་ནི། <། སྤྱོད་ཡུལ་རྣམས་ལས་གཞན་དུའང་འགྱུར། །ཞེས་པ་ཏི་སོགས་བསྟན་པ་སྟེ་གཉིས་པའོ། །།།་ཏུ་འགྲོ་མ་གྲགནའཕྲཋཀ་བྱེད་ཚུགས་། ནི་ཀ། །ཉེ་བ་ཟ་ཁRང་པོ། །ཁཎི་ཀ་ག་དངོས་ལས་ལ་ཀྲྀ་ཏ་རྣམས་འགྱུར། ཞ། ལས་དང་བྱེད་པོར་ཀ་ལི་མའོ། །ལ་ལར་གཞན་ལའང་ཀྲྀ་ཏྱ་རྣམས། །ཁ་ལའི་དོན་དག་དངོས་ལས་ལ། །ཨཱ་ཤི་ཏ་ཉེ་ཚིག་བླུ་ལས། །།དངོས་དང་བྱེད་ལ་ཁ་འགྱུར་རོ། །བྱེད་ལ་ཀྲྀཏྲ་ལས་ཁྱུཊའོ༑། ལས་ལ་:ཊཀྲ་སཀ་ཀྭིཔྲ་ཀྱང་བལྟ། །ལྟ་འགྲོ་བའི་དོན་ལས་བྱུང་ལ། །དངོས་ལས་བཞི་བྱེད་པོ་ལ་འགྱུར། །།གལ་ཏེ་ལས་མེད་ལས་བྱེད་པོའམ། །གཞི་དང་དངོས་དག་ལཀྟ་འགྱུར། །དངོས་པོ་གཞི་དང་ལས་རྣམས་ལ། །ཟ་་་བའི་དོན་ལས་ལྟ་རབ་གྲགས། །ལས་བརྗོད་འདོད་པ་མེད་པ་ཡི། །བྱིངས་ལས་དངོས་པོ་ཙམ་ལའོ། །།ཐིཥ་ཤིཉ་བས་ཛ྄ཥ་ཨསྠཱ་དང་། །།ཛ་ན་རུ་ཧ་ལས་ཕ་་རོལ༑། ཀྟ་གསུམ་ལ་འགྱུར་དེ་བཞིན། །རྩོམ་ལ་འོ་ལྷག་ལས་ལ་བཤད། །འདི་ཡི་དེ་མ་ཐག་བརྗོད་པ། །ཐམས་ཅད་ལས་ཀྱང་བྱེད་པོ་ལའོ། །དངོས་ལ་མང་པོ་གྷ་སོགས་རྣམས། །དེ་རྣམས་འགའ་ཞིག་བྱེད་དྲུག་ལ། །རྫས་བཞིན་ཀྲྀཏའི་དངོས་པོར་འགྱུར། །:དེས་ན་གཉིས་ཚིག་ལ་སོགས་ཀྱང་། །མང་། པོར་ཡོད་ལ་འང་ལས་ནི། །དངོས་དོན་དུ་བྱས་ཀྲྀཏ་རྣམས་སོ། །གལ་ཏེ་བྱེད་པོ་གཅིག་བྱ་བ། །།གཅིག་མིན་དེ་ཚེ་སྔོན་དུས་ལས། །དངོས་ལ་ཀླུ་སོགས་སོ་འདི་ལ། །བྱ་བ་ཕ་རོལ་དེ་དོན་ཏུམ། །:：ཝུཎ་ཏཱིཙ་ཨེལ་སོགས་ཐམས་ཅད་རྣམས། །བྱེད་པ་པོ་ནི་བརྗོད་པར་བསྟན། །ཤན་ཏྲང་ཨཱན་དང་ཀན་སུ་ཡང༌། །བྱ་བ་གཞན་དང་ལྟོས་བཅས་འགྱུར། །བྱིངས་དོན་འབྲེལ་ལ་ཀྲྀཏ་རྣམས་ནི། །དུས་གཞན་རྗོད་པར་བྱེད་པར་འགྱུར། །སྔ་མ་ཕྱེད་དང་ལྷན་ཅིག་གོ། ་། ཞེས་་།
