པ་ཀྲྀཏ་མདོར་བསྟན་པ་སྟེ་གསུམ་པའོ། །།།།།བཀའ་དང་བློ་ཎ་ཁབཀྱི་གཞསྤུ་ར། །ཎ་ཀེ། ཁ་ལོ་་ཁ་ཁ་དོག་སྔོན། ཁྱད་པར་དུ། དམ་ཚིག་དོན་གང་ལ་ནི་ཏི་སོགས་སོགས། །དེ་རྣམས་སྒྲུབ་པར་མི་བྱེད་དེ། །།མ་བརྗོད་པ་དེར་ལ་སོགས་ལ། །ཇི་ལྟ་ཇི་ལྟར་གཉིས་པ་སོགས། །ལས་ལ་གཉིས་པ་རབ་བརྗོད་དེ། །བྱེད་པོ་བྱེད་དྲུག་ལ་གསུམ་པ། །བྱེད་པ་ལ་ཡང་དེ་ཉིད་་་འགྱུར། ཞ་། བཞི་པ་ཡང་དག་རབ་སྦྱིན་ལས། །འབྱུང་ཁུངས་ལ་ནི་ལྔ་པ་སྟེ། །འབྲེལ་པ་ཙམ་ལ་དྲུག་པ་སྟེ། །ལས་དང་བྱེད་པོ་ཀྲྀཏ་སྦྱོར་ལ། །གཉིས་དག་་་་ཐོབ་ན་བྱེད་པོར་མིན། །གྷཉ་སོགས་ཀྱི་ཡང་ནའོ་ཨ་ཀ་དང་། ཁ། ཨ་ཡིག་ལའི་གཉིས་ཀ་ལ་ཡང་འགྱུར། །ཀྲྀ་ཏ་རྣམས་ཀྱི་འདི་འདྲར་གཉིས་ཀ་ལའང་། །མ་ཡིན་བྱེད་པོ་འབའ་ཞིག་ལ་ཡང་ནའོ། །།།ནིཥྛ་ཨུཏ་མཐའ་མི་ཟད་ཨུ་ཀ་དྲིན། །ཤན་ཊཾ་ཨཱ་ན་ཀཱན་སུ་འབྱུང་འགྱུར་གྱི། །ཨིན་:ཝ་ཆུ་དེར་ཚུལ་ཀི་་བུ་ལོན་ལ། །ཎི་ནི་རྣམས་སྦྱོར་བ་ལས་དེ་མ་ཡིན། །དངོས་ལ་ལྟའི་བྱེད་པོར་ཡང་ན་སྟེ། །ཤན་་ཀྲྀང་འིས་བརྗོད་དཱི་ཥའི་ལས་ལ་ཡང་། །ཀྲྀ་ཉའི་འབད་ལ་ཀྲྀཏ་དང་ཀུན་བཤད་ལའོ། །ཧྲཞྭ་གྱིའང་རྗེས་སུ་བྱེད་ལ་དེ་བཞིན་ནོ། །དྲན་དང་འཚེ་དོན་དག་གི་དེ་བཞིན་ཏེ། །ཚིམ་པའི་དོན་གྱི་བྱེད་ལའང་ཡང་ན་འོ། །བཞི་ལ་འཇུག་བཞིན་:པ་ཡི་ལྟ་ཡི། །བྱེད་པ་པོ་དང་ལས་དག་:ལ་དེའོ། །བཞི་ལ་བདུན་པར་ཤེས་པར་བྱ། །ལྟ་ཨིན་རང་བཞིན་གྱི་ལས་ལ། །བྱ་གང་བྱ་བའི་རྟགས་ལ་འང་། །དེ་ལས་མི་གུས་ལ་དྲུག་:པ་ལའང་ཀྱི། །ངེས་པར་དགར་བ་ལ་དེ་བཞིན། །ལས་རབ་བརྗོད་དང་ཨ་ཀྱི་ན་ལ། འ། །གཉིས་པའོ་ས་མ་ཡ་སོགས་དང༌། །ཨུ་པ་རི་སོགས་ཨ་བྱ་སྟེ། །ཨུ་བྷ་:ཡོ་སོགས་ཏས་མཐའ་ཡིས། པ། གཙོ་འདྲ་ཡོངས་སུ་བརྗེ་བ་ལ། །པྲ་ཏིས་སྦྱོར་བ་་་ལས་ལྔ་པ། །ཨ་དྷིས་ཇོ་བོ་ལ་བདུན་པ། །ཨུ་པས་ལྷག་པ་ལས་ཀྱང་དེ། །ཕྱི་།
