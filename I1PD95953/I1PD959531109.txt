རབསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིཨེ་ཀ་སྨི། །ཨེ་ཀ་ཏ། །ཨེ་ཀ་སྱ། །ཨེ་ཀ་སྨིན། གང་དོན་གཞན་གྱི་ཚིག་ལ་མང་པོའི་ཚིག་གི་ཡུལ་ཉིད་ལས་མང་པོའི་ཚིག་དེ་ངེས་པའོ། །ཨེ་ཀེ། ཨེ་ཀ་ན། ཨེ་ཀི། ཨེ་ཀེ་ཧཱུྃ། ཨེ་ཀེ་ཧྥུ། ཨེ་ཀེ་ཏཱཾ། ཨེ་ཀེ་ཥཱ::། ཀེ་བru། །དཱི་ཞེས་པ་འདིའི་དོན་གཉིས་མངོན་པར་བརྗོད་པ་ཉིད་ལས་གཉིས་ཀྱི་ཚིག་སྟེ། །ཀ/ཏུད་རྣམས་ཀྱི་ཏས་ལ་སོགས་པ་རྣམས་ལའང་དྲིའི་བར་གྱི་ཨའོ། །།ཨཏའི་ཨཏ་:ཨེང་ལའོ༄༅། །།ལྷག་མ་ནི་ཨུ་བྷ་ཡ་བཞིན་ནོ། །དྲོ། །དྭཱ་ཧཱུཾ། རྭ་ཡོ༑། བུད་མེད་ལ། དྭེ། །དྭཱ་ཧཱུཾ། དྭ་ཡོ། མ་ནིང་ལ། དྭེ། དྭཱ་ཧཱུཾ། དྭ་ཡོ། གྲངས་ཀྱི་ཚིག་རྣམས་ལ་ཨེ་ཀའི་སྒྲ་དང༌། དཱིའི་སྒྲ་དང་། དྲི་ལ་སོགས་པ་རྣམས་ལའང་རྟགས་ཀྱི་སྒྲ་གསལ་བར་བྱ་སྟེ། །དོན་མངོན་པར་བརྗོད་་པའི་ཁྱད་པར་ལས་རྟགས་གསུམ་པོ་རྣམས་་གཟུགས་བལྟའོ། །ཏྲི་ཞེས་པ་འདིའི་དོན་མང་པོ་མངོན་པར་བརྗོད་པ་ཉིད་ལས་མང་པོའི་ཚིག་སྟེ། །།ཛས་ལའོ། །ཨེཙའི་ཨཡ་ཨཝ་ཨཱཡ་ཨཱཝའོ། ་་། དང་པོ་དག་གི་:ཨཙ་ལའོ། ལྔ། །དེ་ལ་ཤས་ཀྱི་སྐྱེས་པ་ལ་ནའོ། །།ཏྲ་ཡཱ་ཎཾ། ཀུ་ལས་ཀྱང་ཞེས་པའོ། །ཏྲ་ཡཱ། ཏཱིན། ཏྲི་བྷི། ཏྲི་ཤྩ། ཏྲ་ཡཱ་ཎཾ། ཏྲི་ཏེ། བུད་མེད་ལ། ཏྲི་ཞེས་པ་འདིའི། བུད་མེད་ལ་ཏྲི་ཙ་ཏུར་དག་གི་དྲི་སྲི་ཙ་ཏ་སྲོའོ་ཞེས་པ་ཏི་སྲི་འབྱུང་ངོ༌། *། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ། །གྲངས་ལ་ཨཏ་མེད་པ་ལས་སོ་ཞེས་པ་ནུཊའོ། །ར་ཥ་ལས་་་ནའི་ཎའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། ཝ། ཏི་སྲ། །ཏི་སཾ་བྷི། ཏི་སཾ་བྷུ། ཏི་སྭ་ཏྭཾ། ཏི་སྲྀ་། མ་ནིང་ལ། ཏྲི་ཞེས་པ་འདིའི་ཛས་ཤས་དག་གི་ཤིའོ། །ཡམ་མིན་པའི་ཤི་ལའོ། །ཤི་དང་སུཊ་ལའོ། །ར་ཥ་ལ་ལས་ནའི་ཎའོ། །ཏྲ་ཡཱ་f། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །ཏཱི་ཎི། ཏྲི་བྷི། ཏྲི་བྷུ། ཏྲ་ཡཱ་ཎཾ། ཏྲི་ཥཱ། ཨཙ་ཏུར་ཞེས་པ་འདིའི་ཙ་ཏུར་ཨ་ན་རུཧ་དག་གི་ཨཱམའོ་ཞེས་པ་ཨཱམ་སྔོན་།
