བསྟན་འགྱུར། སྒྲ་མདོ། ༼རིའིབ་དྷ་ཀ། ཤན་པ། ཧའི་གཉིས་ཀྱང་ངོ༌། །།།ཨོ་ཧཱཀ་གཏོང་བ་ལའོ། །འདི་ལས་ཀུན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །གཉིས་ཉིད་ཀྱང་ངོ༌། ཁོངོ། ཐུང་ངུ་ཞེས་པས་ཐུང་ངུའོ། །ཨ་ཙ་ལ་ཀིཏ་ལའང་ཨཱ་ཙིཏའི་དབྱིའོ་ཞེས་པས་ཨཱ་དབྱིའོ། །ཀུ་ཧ་དག་གི་ཙུའོ་ཞེས་པས་ཀུ་ཉིད་ཀྱི་ཧ་ཡིག་གི་ཛྷ་ཡིག་གོ། །ཛྷལའི་ཛས་སོ་ཞེས་་པས་ཛས་ཉིད་ཀྱི་ཛྷ་ཡིག་གི་ཛ་ཡིག་གོ། །ཛ་ཧ་ཀ། ཡདུས། ་་ཀྲྀ་ཊི་ལས་ཨཙའི་འང་ཨཱཏ་ཡང་ནའོ། །ཀྲྀ་ཥ་རྐོ་བ་ལའོ། །འདི་ལས་ཀུན་ཕ་རོལ་ཏུ་འགྱུར་རོ༑། ཨཙའི་གནས་ལ་ཨཱ་ཉིད་དུ་ཡང་ན་འགྱུར་རོ། །རཾ་ཀའི་ཨཎ་ལ་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད་དོ། །ཀཱཎྜ་ཀབྲན་བཟའ། ཀྲྀ་ཥ་ཀ། ཡབྲན་བཟའ། བྲཉི་མུ་ཁི་ལས་ཀྱང་ཀི་ཀནའོ། །ཨོ་བྲཤྩ་གཅོད་པ་ལའོ། །མུ་ཥ་དུམ་བུ་ལའོ། །ཀྱང་ཡིག་ལས་ཀྲྀ་ཥ་རྐོ་བ་ལའོ། །འདི་རྣམས་ལས་ཀི་ཀན་ཕ་་་རོལ་ཏུ་འགྱུར་རོ། ་། ན་ཡིག་དབྱངས་དོན། ཀི་ཡིག་ལ་ཨི་ཡིག་གཟུང་སྟེ། ཀ་ཡིག་ཀིཏ་ལའོ་ཞེས་པའི་རྟགས་ཀྱི་དོན་ཏོ། །ཀིཏ་ལའང་དེ་རྣམས་ཀྱིའོ་ཞེས་་་པས་ཎ་ཨི་ཀའོ། །།ཡཎའི་ཨི་ཀའོ་ཞེས་པས་ར་ཕ་རོལ་སྔོན་མ་ཉིད་དོ། །།།བྲརྩི་ཀ༑ཡསྡིག་པ། མུ་ཏི་ཀ། བྱི་བ། ཀྲྀ་ཊི་ཀབྲན་བཟའ། ཨཱང་ལས་པ་ཎི་ལ་་ཏི་དག་ལས་སོ། །།པཎ་གཙུག་ལག་ལའོ། །པ་ཊཱ་ལམ་ལ་འགྲོ་བ་ལའོ། །འང་། འདི་དག་ཨཱང་སྔོན་མ་ཅན་ལ་ཀི་ཀན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ང་ཡིག་ཨཱང་མཱང་དག་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །ཨཱ་པ་ཎི་ཀ། ཚོང་པ། ཨཱ་པ་ཏི་ཀ། བྱིས་པ། སྱ་མུའི་ཡའི་ཨཱིཏ་:ཀྱང་ངོ་ཉི། །སྱ་མུ་སཱ་ན་དྷཱ་ན་སྒྲ་ལའོ། །འདི་དག་ལས་ཀི་་ཀན་ཕ་རོལ། ན་དབྱངས་དོན། ་ཡའི་ཨཱི་ཡིག་གོ། །སི་མི་ཀ། ཡཤིང༌། བྷི་ཤི་དག་ལས་ཨན་ཀའོ། །བྷི་འཇིགས་པ་ལའོ། །ཤིང་ཉལ་བ་ལའོ། །འདི་དག་ལས་ཨཱ་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཡིག་གོ། །ཨེ་ཙའི་—།
