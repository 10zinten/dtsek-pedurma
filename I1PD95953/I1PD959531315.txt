བསྟན་འགྱུར། སྒྲ་མདོ། ་རིཡའོ་ཞེས་པས་ཨ་ཡ་རུ་བསྒྱུར་རོ། །སྤྲོ་ཧ་ཡཱ་ཡུ། མར། མང་པོ་ལའོ་ཞེས་སྤྱིར་བཏང་བའི་ཕྱིར་གཞན་ལའང་འགྱུར་ཏེ། ཤདྡྷ་ཡཱཡྻ། དད་པ། །ཞེས་བྱ་བ་ལ་་སོགས་པའོ། །བྲིང་ལས་ཨེ་ཙྩའོ། །བྲིང་ཡང་དག་པར་གུས་པ་ལའོ། །ང་ཡིག་ནི་ཏ་ལྡན་པ་ཧལ་དང་པོ་ལས་ང་མེད་པ་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །འདི་ལས་ཨ་ཅུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨིཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད། རཾ་ཀའི་ཨཎ་ལ་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད་དོ། །།པ་རེ་ཎྱ། གཙོ་བོའམ་་་མཆོག་ལྡན། ཨརྟི་ལས་ཨ་རྒྱ་ཙའོ། །རྀ་འགྲོ་བ་ལའོ། །འདི་ལས་ཨ་ཅུ་ཙ་ཕ་རོལ། ཙ་དབྱངས་དོན། སྔ་མ་བཞིན་ཨ་ཉིད་དང་ར་ཕ་རོལ། ་ཨ་ར་ཙྩཾ། ཡདགོན་པའམ་ནགས་ཚལ། ཧཱའི་ཧིར་ཡང་ངོ༌། །ཨོ་ཧཱཀ་གཏོང་བ་ལའོ། །འདི་ལས་ཨ་རྱ་ཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །།ཧིར་དུ་བསྒྱུར་བ་ཡང་ངོ༌། །ཧིར་ཙྩཾ། ཡདབྱིག་གམ་གསེར། པརྫ་རྒྱའོ། །པྲ་རྫོགས་པ་ལའོ། །འདི་ལས་ཨརྱ་ཙ་འགྱུར་ཞིང་འཇུག་གོ། །ཨ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །པརྫ་ནྱ། བརྒྱ་བྱིན། པུ་ཎི་ལས་ཀྱནའོ། །པུ་ཎ་དགེ་བ་ལའོ། །འདི་ལས་ཀྱན་ཕ་རོལ། ཀ་ཡིག་ཨཏ་ཨེང་དགག་པའི་དོན། །པུ་ཙྩཾ། བསོད་ནམས་སམ་དགེ་བའི་གནས། ཤཀྱཱ། དྷིཤྩའོ། །སྒྲ་འདི་དག་ཚིག་ཕྲད་དོ། །ཤ་ཀྵི་ནུས་པ་ལའོ། །ཏུ་དྷཱཉ་འཛིན་པ་ལའོ། །འདི་དག་ལས་ཀྱན་ཕ་རོལ་ཏུ་འགྱུར་རོ༑། ཤ་ཀའི་ཨཏ་ཀྱི་ཨི། ཤི་ཀྱི། དྲ་བ་ས། ད་དྷཱ་ཏིའི་ཨ་ཡིག་གི་ཨི་ཉིད་ཥཱཀ་དང་ནུཊའངའོ༑། ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད། དྷིཤྩ། ཡཁྱིམ། ཨུཎ་ལ་སོགས་པའི་འགྲེལ་པ་ལས་རྐང་པ་གཉིས་པ་རྫོགས་སོ། །།ན། །ཀ་པེ་༼སྤྱི་ལོ་::མནྡི། ཨོཾ་ཀི། བཱ་སི། མ་ཐི། ཙ་ཏི་རྣམས་ལས་ཨུ་ར་ཙའོ། །མ་དི་རྒྱགས་པ་ལའོ། །ཨི་དཏ་ཀྱི་ནུམའོ། །ཨུ་ཡིག་དང་མ་ཡིག་དབྱི། ་ན་ཡིག་།
