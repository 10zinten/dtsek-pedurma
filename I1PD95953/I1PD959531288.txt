ཨ་ཡ་ཨ་བ་ཨཱ་ཡ་ཨཱ་བ་རྣམས་སོ་ཞེས་པས་ཨ་ཡ་རུ་བསྒྱུར་རོ། །བྷ་ཡཱན་ཀ། འཇིགས་རུང་ངམ་ནགས་འཐིབས་པོ། ༽ཤ་ཡཱ་ན་ཀ། གཉིད་ལོག་པ། སི་གྷི་ལས་ཨཱཎ་ཀའོ། །སི་གྷི་སྣོམ་པ་ལའོ། ག་། ཨི་ཡིག་ནི་ཨི་དིཏ་ལས་ནུམའོ་ཞེས་པའི་རྟགས་དོན། ནུམ་དང་རྗེས་སུ་ང་རོ་ལ་སོགས་པ་སྔར་བཞིན་ནོ། །འདི་ལས་ཨཱན་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། ཊ། སཾ་གྷ་ཎ་ཀསྣབས་སོ། ཀྲྀ་ཏི་བྷི་དིར་ལ་ཏི་ལས་སྟེ་ཀནའོ། །ཀྲྀ་ཏཱི་གཅོད་པ་ལའོ། །བྷི་དིར་རྣམ་པར་འཇོམས་པ་ལའོ། །ལ་ཏ་ཞེས་པ་མདོའི་བྱིངས་སོ། །འདི་དག་ལ་ཀིཏ་ཀན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ནི་ཀིཏ་བྱ་་་བའི་དོན། ཆ་པའོ། །ན་ཡིག་དབྱངས་དོན་ནོ། །ཀྲྀཏི་ཀཱ། སྨིན་དྲུག་ཁར་ལ་ཛྷ་ལའི་ཙརའོ་ཞེས་པས་ད་ཡིག་གི་ཏ་ཡིག་གོ། །།བྷིརྟི་ཀཱ། བརྩིགས་པ། ལརྟི་ཀཱ༑འཁྲི་ཤིང་། ཨི་ཏི་ལས་ལྟ་ཀནའོ། །ཨི་ཀྵ་འདོད་པ་ལའོ། །འདི་ལས་ལྟ་ཀན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཚ་དིའི་ད་ཡིག་དབྱིའོ། །།ཚ་བི། ཡཔགས་པ། །ཏིཥྛ་་་་ཏིའི་ཐུང་བ། སྦི་བི། ཐག་པ། ཀ་ཀའི་ན་དབྱི། ཀའི་ཡིག་ནི་ཨི་ཉིད། བི་ཡིག་གི་ཨི་ཡིག་བཟུང་ནས་བ་ཡིག་དབྱི་བ་ཚིག་ཕྲད་དོ། །ཀི་ཀི། བྱ། :：དི་་་བིའི་ཅིབ་ཡིག་དབྱི། དི་བི། ཡཉི་མ། བྱིངས་དེ་ཉིད་གཉིས་ཉིད་དོ། །དི་བིའི་ཧལ་དང་པོ་མ་ཡིན་པའི་འོ་ཞེས་པས་བ་དབྱིའོ། །།རིང་པོ་ཉིད་ཀྱང་ངོ༌། །ཕྱི་མའི་ཆོ་ག་ལ་མ་ཡིག་དབྱིའོ། །དཱི་དི་བི། མཐོ་རིས། ཀྲྀ་བི། ཁྲོན་པ། བྲྀཏྟི། རིམས་ནད། ཇི་བའི་བ་དབྱིའོ། །ཛཱི་བི། ཡཨོ་ཥ་དྷི། ནཱི་བི། ཡཐོབ་པ། ་དེ་བཞིན་དུ་སྒྲ་གཞན་ཡང་བལྟ་བར་བྱའོ། །དྲོ་བ་སི་དག་ལས་ཀིནྟའོ། །དྲང་མ་གུས་པ་ལའོ། །།བས་གནས་པ་ལའོ། །འདི་དག་ལ་སྟིན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ཨཏ་ཨེང་དགག་པའི་དོན། ན་ཡིག་དབྱངས་དོན། དྲྀ་ཏི། པགས་པ། བསྟི། ཡཆུ་བོ། པཱ་ཏི་ལས་:ཌི་ཏིའོ༄༅། །པཱ་སྲུང་བ་ལའོ། ང། ཌ་ཏི་ཕ་རོལ། །
