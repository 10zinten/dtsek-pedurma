བསྟན་འགྱུར། སྒྲ་མདོ། ༼རི༽ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ལ། །ཡ་ཀ་པ། ལབSཏ། །སྐྱོ་པོ་ཞིག་ད་ཏེང་ཚགཞིའདིས་འགྲོ་བ་དཔག་ཏུ་མེད་པ་ལ་ཕན་པར་གྱུར་ཅིག་ན། འདིར་འབད་་་དགེ་བས་འགྲོ་བ་ཀུན། །གཉིས་མེད་གསུང་བས་བསྟན་པ་ཡི། །བརྗོད་མེད་དེ་ཉིད་རབ་རྟོགས་ནས། །ངག་གི་དབང་ཕྱུག་ཉིད་གྱུར་ཅིག། །།།མདངབདག་དམ་ལ་འདེད༌རྒྱལ་བ་ཀློག་པའམ་བག་ཚགཤེད། །ཁམས་གཉིའི་ནད་དུག་ག་་་་་་་མ། ཡུ་བ་བསྐམས་ཟླ་བ་དེའི་ལས་ལ་ལ་ལ་ལ་་ཆ་་ཎཐོལོ་ལ་ཁ། །།བསྡུར་མཆན། ཤོ་དེ་དང་། ཟླ་བ་དང་འཁྲབ་པ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་འགྲེལ་པ་མདོར་བསྟན་པ་ཙཱ་ཀཱ་དས་མཛད། ཞེས་འཁོད། «པེ་»«སྣར་»སཾ་བྷོ་དྷེ། @«པེ་»འགྲེལ། «པེ་»《སྣར་»གིས། «པེ་»«སྣར་»དཔའི། «པེ་»«སྣར་»གྱི། ཨ«པེ་»ཅས། «སྣར་»ཅེས། «པེ་»«སྣར་བར། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»གྱིས། @«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་ལན། «པེ་»«སྣར་གྱུར། ཚེ«སྣར་»+གི་@«པེ་»«སྣར་»དང༌། ཨ«པེ་»བརྟེན། 《སྣར་»བསྟེན། «པེ་»«སྣར་ཡང་། }@«པེ་»«སྣར་ཧྲི། ཀྱི«པེ་»«སྣར་»—དི། ཀྱེ«པེ་»«སྣར་»ཨ་དི་ཤབྷཱ་ཡ། །ཀཱན་དའི། ཏེ«པེ་»«སྣར་»དེ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ཁས། «ཅོ་»པོས། «པེ་»《སྣར་»ཀྲྀཏྲའི་ས། «པེ་»«སྣར་»ག༄༅«པེ་»«སྣར་»ལས། «པེ་»«སྣར་»ནི། ཉ«པེ་»«སྣར་ཤསྟ། «པེ་»«སྣར་»དྲྀ་ཤི། ཀྱི«པེ་»«སྣར་»ནི་རྒྱལྟ་ནི། གྱི«པེ་»«སྣར་»སྟི་ནི། «སྣར་དུ་ལ། ཉེ«པེ་»«སྣར་»པའང་། །ཨེ«པེ་»«སྣར་»ཏུ། ཉི«པེ་»«སྣར་»ཨ། «པེ་»«སྣར་»ང་ཉ། «པེ་»«སྣར་ཊཀྲ་ས། «པེ་»«སྣར་གཞི། ༄༅«པེ་»«སྣར་+པ། «པེ་»ཛི་ཥ་ཨསྲ་སྒྲ། «སྣར་»ཛཱིཥཨསུ་སྦྲ། །ཟུང«པེ་»«སྣར་»ཏུ། «པེ་»«སྣར་»བཅོམ། «པེ་»«སྣར་»+ལས། གྱི«སྣར་»ཀྲིཏྲའི། ཅི«པེ་»«སྣར་པོ། ཅེ«པེ་»«སྣར་དེ་ནས། ཁྱེ«པེ་»«སྣར་»པོ། ཚེ«པེ་»«སྣར་ལཀྟ། ཚེ«པེ་»«སྣར་»སོ། ཁ«པེ་»«སྣར་»བུཎ་ཏྲིཛྲ། «པེ་»«སྣར་»བལྟོས། ༽རའིཚེ«པེ་»«སྣར་»བསྒྲུབ། «པེ་»《སྣར་»ལ་དེ། «པེ་»«སྣར་ཝུ་ཎ། «པེ་»«སྣར་»+ངང་། «པེ་»«སྣར་»གིས། @«སྣར་ཀྲིཏྲ། ཨ«པེ་»«སྣར་»ད། «པེ་»«སྣར་པའི་ལྟའི། «པེ་»«སྣར་»ལས་དེའོ། གཞི། ཁ«པེ་»«སྣར་པའང༌། «པེ་»«སྣར་»ཨེ། ཚེ«པེ་»«སྣར་»ཡ་་་།
