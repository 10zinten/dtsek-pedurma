གཞན་ལ་ནི་ཙྪ་ག་ལི། བི་ཀཎྞི། ་ཚགལ་ཏེ་རྒྱུན་ཤེས་ཀྱི་བུ་ཡིན་ནའོ། །གཞན་ལ་ནི་:བི་ཀཎི། པཱི་ལ་ལས་ཡང་ནའོ། །བུ་རྒྱུད་ལས་ཨཎ་འགྱུར་ཏེ། པི་ལ། པི་ལེ་ཡ། མཎྜཱུ་ཀ་ལས་ཀ་འང་ངོ༌། །མཎྜཱའི་ཀའི་སྒྲ་ལས་ཨཎ་་་ཡང་འགྱུར་ཞིང་ཆ་ཀ་འང་ངོ༌། །མཱཏྟཱ་ཀི། །མཱཌཱ་ཀི་ཡཱ། མཎྚ་ཀི། བུད་མེད་རྣམས་སོ། །བུད་མེད་དུ་བྱས་པ་རྣམས་ལ་དྷཀ་འགྱུར་ཏེ། ཨཎ་གྱི་དམིགས་བསལ་ལོ། །།སོ་པརྗེ་ཡ། བི་ན་ཏྲི་ཡ། ཀཱ་མཎྜ་ལེ་ཡ། ཀོ་མཱ་རེ་ཡ། །ཡོ་བ་ཏེ་ཡ། བ་ཌ་བ་ལས་བྲ་ཥ་ལ་ལའོ། །བྷཀ་འགྱུར་ཏེ། བ་་ཌ་བའི་བུ་རྒྱུད་ནི། བཱ་ཌ་བེ་ཡ། པྲྀ་ཥ་ནི་འདིར་མངལ་འཇོག་པའི་ནུས་པ་ལ་བརྗོད་དོ། <། ཨ་ཙ་གཉིས་པ་བུད་མེད་ཀྱི་རྐྱེན་མཐའ་ལས་རྙཀ་འགྱུར་ཞིང་། དཱཏྟེ་ཨིཡ། ཏགོསྟ་ཡ། །བྲྀདྡྷ་མ་ཡིན་པ་ལས་ཆུ་བོ་དང་མིའི་མིང་ལས་སོ་ཞེས་་པས་ཨཎ་འགོག་པའི་དོན་ཏོ། །མིའི་མིང་ཉིད་ལས་:ཨཎ་ལ་སོགས་ཅིཔ་འདི་ཆུ་་བོའི་མིང་ལས་མ་ཡིན་ཞེས་པས་འགའ་ཞིག་ཏུ་མ་ཡིན་ཏེ། ཤི་པྲ། ཨིཏ་ལས་ཨིཉ་མ་ཡིན་པ་ལས་སོ། །ཨི་ཡིག་མཐའ་ཅན་ཨིཉ་མ་ཡིན་པ་ལས་བུ་རྒྱུད་ལ་ཆ་ཀ་འགྱུར་ཏེ། ཨཱ''ཏྲེ་ཡ། བི་དྷེ་ཡ། ཀཱ་བི་ཡ། ཨིཉ་མ་ཡིན་པ་ལས་ཞེས་པ་ཅི་ཞེ་ན། དཱགྷཱ་ཡ་ནི། ཨ་ཙ་གཉིས་:ཞེས་པ་དེ་ཉིད་དེ། མཱ་རཱི་ཙ། ཤུ་བྷ་སོགས་རྣམས་ལས་སོ། །།བུ་རྒྱུད་ལ་ཌྷཀ་འགྱུར་ཏེ། སྦྱོར་བ་ཇི་ལྟ་བར་ཨིཉ་ལ་སོགས་པ་རྣམས་ཀྱི་དམིགས་བསལ་ལོ། །ཤོ་བྷ་ཡ། བིཥཱ་པུ་རེ་ཡ། ཤུ་བྷེ། བིཥཱ་པུ་ར། བྲཧྨ་བྲིཏི། ཤ་ཏ་དཱ་ར། ཤཱ་ལཱ་སྦྲ་ལ། ཤཱ་ལཱ་ཀ་བྷུག། ལེ་ཁ་བྷཱུ། བི་དྷ་བཱ། བིཀ་ཤཱ། ་རོ་ཧི་ཎི། །རུ་ཁི་ཎི། །བརྨི་ཎི། དྷཱཙྪཱཤི། ཝེཤཱ་ལཱུ་ཀ། །ཨ་ཛ་བསྟི། ཤ་དཱི། །ཤ་ཀ་དྲི་ལཀྵ་ཎ། ཤུ་མ་དག་ལ་བཱ་ལིཤིཥྛ་ལའོ། །ལག་ཅིཎའི་བུ་རྒྱུད་ནི། :：ལཀྵ་ཎེ་སཔ། གལ་ཏེ་གནས་་་།
