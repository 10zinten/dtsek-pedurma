བསྟན་འགྱུར། སྒྲ་མདོ། རེའིཞེས་པས་ཨཱ་ཉིད་དོ། །པཱ་ཏཱ་ལཾ། ས་འོག་། ཙཎྜ་ལ། དམན་པའི་རིགས། ཀུ་ཎ། པཱི་དག་ལས་ཀཱ་ལནའོ། །ཀུ་ཎ་སྒྲ་ལའོ། །བིང་བཏུང་བ་ལའོ། །འདི་དག་ལས་ཀཱ་ལན་ཕ་རོལ་ཏུ་འགྱུར་རོ། ་། ན་ཡིག་དབྱངས་དོན། ཀ་ཡིག་ཨཏ་ཨེང་་དགག་པའི་དོན། ཀུ་ཎཱ་མ། བྱའོ། །ཨ་ཙ་ལ་ཟླ་བྱིངས་བྷུ་རྣམས་ཀྱི་ཨཱི་ཨཱུ་དག་གི་ཨི་ཡ་ཨུ་བ་དག་གོ་ཞེས་པས་ཨི་ཡར་བསྒྱུར་རོ། །།པི་ཡཱ་ལ། ཡཤིང༌། ཤིང་ལས་ཕཱ་ལན་བ་ལཉ་དག་གོ། །ཤིང་ཉལ་བ་ལའོ། །འདི་ལས་ཕྲ་ལན་པ་ལཉ་་་ཞེས་པ་འདི་དག་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་དབྱངས་དོན། ཨི་ཀའི་ཨཏ་་ཨེང་ཞེས་པས་ཨེ་ཉིད་དོ། །ཤེ་ཕ་ལ། ཆུའི་རྩྭ། ཉ་ཡིག་ཨཱཏ་ཨི་ཙའི་དོན་ནོ། །ཨ་ཡང་ངོ་ཞེས་པས་ཨེ་ཡིག་གོ། །ཤི་བ་ལཾ། ཆུའི་རྩྭ། མ་གི་ལས་ཨ་:ལ་ཙའོ། །མ་གི་འགྲོ་བའི་དོན་རྣམས་སོ། །གང་ལ། ཨི་ཡིག་དབྱི་བ་དང་ནུམ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །འདི་ལས་ཨ་ལ་ཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཙ་ཡིག་དབྱངས་ཀྱི་དོན། མངྒ་ལཾ། བཀྲ་ཤིས་སོ། །མཱ་ལཱ། །ཨི་ལཱ་ལ། པ་ལཱ་ལ། ཙ་ཥཱ་ལ། ཤི་ཐི་ལས། ཤུཀླ༑ཏཎྜ་ལ་རྣམས་སོ། །འདི་རྣམས་ལས་ཨ་ལ་ཙ་ཕ་རོལ་ཏུ་ཚིག་ཕྲད་དུ་་བྱའོ༑། མཱང་འཇལ་བ་ལའོ། །ཨི་ལ་འགྲོ་བ་ལའོ། །པ་ལྔ་ཞེས་པ་མདོའི་་་བྱིངས་སོ། །ཙ་ཥ་ཟ་བ་ལའོ། །ཤཱནྡྲ་སྤེལ་བ་ལའོ། །ཤུ་ཙ་མྱ་ངན་ལའོ། །ཏ་ཌི་བསྣུན་པ་ལའོ། །འདི་རྣམས་ལས་ཨ་ལ་ཙ་ཕ་རོལ་དུ་འགྱུར་རོ། ས་ཆ་པ། མཱ་ལཱ། ཕྲེང་བ། ཨི་ལི་ལས་བུག་ཨཱ་ག་མའོ། །ཨི་ལཱ་ལ། ཡསྐར་མ། །པ་ལཱ་ལ༑ཡལ་འདབ། ཙ་ཏིའི་ཨའི་ཨཱའོ། །ཙ་ཥཱ་ལ། མཆོད་སྦྱིན་གྱི་ཡོ་བྱད། ཤནྡྲའི་ན་དང་ར་དབྱི་ཞིང་ཨའི་ཨི་ཉིད་ཚིག་ཕྲད་དོ། །ཤི་ཐི་ལཡལྷོད་པའམ་དམ་པོ་མ་ཡིན་པ་ཤིག་ཤིག་པོ། ཤུ་ཙིའི་ཀུ་ཉིད་དང་ཨ་ཡིག་གིའང་དབྱི། །ཤུ་ཀྑཾ། དཀར་པོ། ཏཎྜིའི་ཌའི་ཨུ་ཉིད་དོ། །ཏཎྜ་ལ། ཡའབྲས་དྲུས་མ་སོ་བ་ལས་བྱུང་བ། འོ་ཨརྟི་ལས་པི་སནའོ། །—།
