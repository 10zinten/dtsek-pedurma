*སྒྲ་མདོ། རིབཅས་པར་རུང་ངོ༌། །སྟི་ལས་སོ། །།ཨཔྲ་ལས་ཡཊའོ། ''། སྨིན་ལྡན་པ་ལས་སྣ་ཊཱ་ཨཊ་ཡང་ངོ༌། །ཏཱི་དཱི་ཡང་དང་ཏཱི་ཏི་ཡ་ལས་རུང་ངོ༌། །ོ། ངའི་ཨམ་དེ་ལའོ། །ནི་ལས་སོ། །ཨིཏྲ་དང་ཨུཏྲ་དག་ལས་ཨོཏྲའོ། །ཨོདྲ་ལཱིས་ཨཏྲ་ཡང་ངོ༌། །ཌའི་བུད་མེད་ལ་ནའོ། །རིཏྲའི་ང་ལས་སུ་ལ་ཨཏྲའོ། །སྦྱར་བ་དང་པོའི་ལིཌ་ལའོ། །ཀྲྀཧྲའིའོ། །།:རྀཏ་དང་རིཙྪ་དང་རེ་ྋརྣམས་ཀྱིའོ། །རཾ་དང་ཤྲཱི་དང་དྲི་ཤའི་ཨདྲ་ལའོ། །ཨས་དང་པ་ཏ་དང་པ་ཙ་རྣམས་ཀྱི་ཐུཀྲ་དང་། །སུམ་་་དང་མུམ་རྣམས་ཀྱིའོ། །ཀ་ལ་ཨཎྜའི་ཐུང་ངོ༌། །ཀ་པ་ལ་མ་ཡིན་ནོ། །ཨ་པའི་རུང་ངོ༌། །ཤིངའི་ཨཏྲ་ལིར་མ་ཡིན་པ་ལའོ། ཀྵ་། ཡལ་ཀ་ལ་ང་ཕྱི་བ་ལ་ཨ་་ཡའོ༑། །པྲ་ལ་སོགས་པ་རྣམས་ཀྱི་སྤུ་ཧའི་ཐུང་ངོ༌། །ལིང་ལ་ཨིཊའོ། །ཤིས་པ་ལ་རུང་ངོ་། །ཙཱི་དང་ཡཀྲ་དང་ཀྱ་རྣམས་ལས་སོ། །རིང་རིཏའི་ཡ་ལ་ཡང་ངོ༌། །རིང་ཤ་དང་ཡཀྲ་དང་ཨ་ཤིཛྲ་ལའོ། །རྀ་དང་སྦྱར་བ་དག་གི་ཨཏྲའོ༑། ཡང་ལའོ། །ཧ་ནའི་གླི་འཚེ་བ་ལའོ། །ཨཱི་གྲྭ་དང་དྷ་དག་གིའོ། །ཨའི་ཙཱི་ལའོ། །།ཀྱཚྭ་ལའོ། །ལྟོགས་པ་ལ་ཨ་ཤནའི་མ་ཡིན་ནོ། །དྷ་ནའི་སྲེད་པ་ལའོ། །ནྱའོ། ་། བྲི་ཤ་དང་ཨ་ཤ་དག་གི་འཁྲིག་པ་ལ་སུཀའོ། །ཨ་སུཀྲ་ཡང་ཟ་བའི་ཆེད་དུའོ། །དོ་དང་ཤོ་དང་མཱ་དང་ས་རྣམས་ཀྱི་ཨིཏྲ་ཏ་ལ་ཀ་ཕྱི་བ་ལའོ། །ཚའི་རུང་ངོ༌། །དྷཉྫའི་ཧིའི་ཏཀྲའི་ཏཱ་ལའོ། །དའི་དཏྲ་པྲ་ལ་སོགས་པའི་ཨཙཱའི་ཏའོ། །ཨབྲའི་བྷ་ལའོ། །ས་ལ་སའི་ལི་ད་ཨཏྲ་ང་ལའོ། །:ཏས་་སུའི་ར་ལ་ཡང་དབྱིའོ། །ཨཏྲ་ལ་ཧའོ། །ཀྱང་ལ་རུང་ངོ༌། །ཌ། ཨོ་རྫས་དང་ཨས་ས་རས་དག་གིའོ། །ཡ་དང་ཨིའི་ཡི་གེ་ལ་དཱི་དྷི་དང་པོ་བཱི་དག་གིའོ། །ཨཙཱ་ལ་ཡཎྜའོ། །།མི་དང་མཱི་དང་མཱ་དང་ར་བྷ་དང་ལ་བྷ་དང་ཤི་ཀ་དང་པ་ཏ་་་་དང་པ་ད་དང་ས་དཱ་དང་དྷ་རྣམས་ཀྱི་ཨཔྲའི་སན་ལ་ཨིས་སོ། །ར་དྷའི་ཚེ་བ་ལའོ། །—།
