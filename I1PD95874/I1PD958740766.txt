བསྡམས་པ་བཅིངས་ཏེ། ཨོ་གུ་རྒྱ་ར་ཏི་བཛྲ་ཝ་ཤཾ་ཀ་ར་སིདྡྷི་སིདྡྷི་ཧཱུྃ་ཞེས་བྱ་བ་འདི་དང༌། ་མཚན་ཐོག་ཐག་བྱས་ན་དེས་འགྲུབ་པར་འགྱུར་རོ། །རྡོ་རྗེ་སྲུང་བ་དང༌། །གནོད་སྦྱིན་དང་། ཁྲོ་བོ་སེམས་དཔས་ཡང་དག་པར་བསྲུངས་ལ། ཡི་གེ་བརྒྱ་པས་བརྟན་པར་བྱའོ། །མིག་ལ་མིག་སྨན་བསྐུ་བར་བྱའོ། །དེ་སྐད་དུ་ཡང་། རང་གི་རིགས་ཀྱི་ལྷ་གང་ཡིན། །དེ་ནི་ཡིད་འོང་བུད་མེད་བཤད། །མིག་སྨན་ཚོགས་ལ་དེས་བཟླས་ནས། །པདྨའི་སྟེང་དུ་གཞག་པར་བྱའོ། །བྷ་ག་པདྨ་ཞེས་སུ་བཤད། །།བཅུག་ལ་བཞག་པར་བརྗོད་པ་ཡིན། །བྷ་གའི་སྦྱོར་བ་འདི་ཡིས་ནི། །མིག་སྨན་ལ་སོགས་བསྒྲུབ་པར་བྱ། །ཞེས་གསུངས་སོ། འ་། ལྡོང་རོས་བུ་གར་བཅུག་ནས་ནི། །ཞེས་བྱ་བ་ནི། །དེ་བཞིན་དུ་ལྡོང་རོས་དང༌། གོ་རོ་ཙ་ན་དང་། གུར་གུམ་དང་། ཡིད་དུ་འོང་བའི་བུད་མེད་དང་ལྷན་ཅིག་ཡང་དག་པར་སྦྱར་རོ་ཞེས་ཤེས་པར་བྱའོ། །ལྡོང་རོས་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་སྔ་མ་བཞིན་ནོ། །རྡོ་རྗེ་བསྡམས་པས་དེ་གཡོགས་ཏེ། །ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་་་དབྱིངས་ཀྱི་དཀྱིལ་འཁོར་ལས་གསུངས་པའི་རྡོ་རྗེ་བསྡམས་པ་བཅིངས་ལ། །དེ་་་བཟུང་ནས་དེ་ཉིད་དང་མཚན་ཐོག་ཐག་བཟླས་པར་བྱའོ། །མཚན་མ་རྣམ་པ་་་བཞི་པོས་ནི། །དངོས་གྲུབ་རྣམ་པ་བཞིར་འགྱུར་རོ། །ཞེས་བྱ་བ་ནི་དེ་ལ་་་མཚན་མ་བཞི་ནི་དྲོ་བ་དང༌། དུ་བ་དང་། རྡུལ་བ་དང་། འབར་བའོ། །དྲོ་བས་ནི་དབང་དུ་བྱེད་དོ། །དུ་བས་ནི་རྐང་མགྱོགས་སོ། །རྡུལ་བས་ནི་མི་སྣང་བའོ། །འབར་བས་ནི་ནམ་མཁའ་ལ་འགྲོ་བའོ། །མིག་སྨན་བྱས་པ་དང་རྐང་པ་ལ་བྱུག་པ་དང་། ཡན་ལག་ལ་བྱུག་ཡིཔ་དང་། དཔྲལ་བར་ཐིག་ལེ་་་བྱས་པས་འགྲུབ་པར་འགྱུར་རོ། །གོ་རོ་ཙ་ན་བུ་གར་བཅུག། ཅེས་བྱ་བ་ནི་སྔ་མ་བཞིན་ནོ། །་ལ། གསང་བའི་ཁུ་ཚུར་གྱིས་བཙིར་ན། ་། ཞེས་བྱ་བ་ནི་འཇིག་།
