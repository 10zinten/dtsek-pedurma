བྷ་ནི་སུཾ་བྷ་ཧཱུྃ། །:གྲིཉྩ་གྲིཉྩ་ཧཱུྃ། །གྲིདྡྷཱ་པ་ཡ་གྲིདྡྷཱ་པ་ཡ་ཧཱུྃ། ཨཱ་ན་ཡ་ཧོ་བྷ་ག་བཱན། བཛྲ་ཧཱུཾ་ཕཊ་ཧཱུྃ་ཞེས་བྱ་བ་ཡི་གེ་ཧཱུྃ་གི་མཐའ་ཅན་བཅོམ་ལྡན་འདས་་རྡོ་རྗེ་ཧཱུྃ་མཛད་ཀྱིས་བསད་པའི་སྦྱོར་བ་བྱའོ། །ཨོཾ་བཛྲ་ཨོ་གུ་ཤ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་་ཀརྟ་ཡ་སརྦ་ས་མ་ཡ་:ཧཱུྃ་ཛཛ%ཞེས་ཡི་གེ་ཛ?དང་བཅས་པའི་ཁྲོ་བོ་རྡོ་རྗེ་ལྕགས་གྱུར་དགུག་པའི་སྦྱོར་བ་བྱའོ། །ཨོཾ་བཛྲ་དྷཱ་ཏུ་ཨོ་ཞེས་བྱ་བ་རྒྱས་པ་ལ་བཅོམ་ལྡན་འདས་རིན་ཆེན་འབྱུང་ལྡན་ཡི་གེ་ཨོཾ་གྱི་མཐའ་ཅན་གྱི་སྦྱོར་བ་ཡང་བྱའོ། །དབང་དུ་བྱ་བ་ལ་ཨོ་བཛྲ་རཱ་ག་ཧོ་ཞེས་བྱ་བ་དང་། ཡི་གེ་ཧོ་དང་བཅས་པ་དང་། རྡོ་རྗེ་ཆགས་པས་སྦྱོར་བར་བྱའོ། །།པྥཀྐ་དང་། ཨུ་དུམྨཱ་ར་དང༌། ནྱ་གྲོ་དྷ་དང་། །མྱ་ངན་མེད་པ་དང༌། ཙུ་ཏ་དང་། ཤ་མི་དང་། པ་ལ་ཤ་དང༌། ཞི་ཀསྨརྫུ་ལ་སོགས་པ་མཆོད་སྦྱིན་གྱི་ཤིང་འོ་མ་ཅན་རྣམས་སམ། །གཞན་དག་གི་ལྗོན་པ་རློན་པའི་རྩེ་མོའི་ཆ་ལས་བྱུང་བ་མཐོ་གང་གི་ཚད་ཙམ་རྩེ་མོ་གཉི་ག་ྋསྦྲང་རྩི་དང་འོ་མ་དང་མར་ལ་སྦགས་ལ། ཙནྡན་དཀར་པོའི་ཆུར་ཡོངས་སུ་སྦྱངས་པའི་ཡམ་ཤིང་དང་། ་ནས་དང༌། འོ་མ་དང་། ཞོ་དང་། འབྲས་ཡོས་དང༌། དྷུར་བ་དང་། །ཀུ་ཤའི་མྱུ་གུ་དང་། ཙནྡན་དཀར་པོ་དང་། ཙུ་ཏའི་ལོ་མ་དང་། ཞཔྲི་ཡང་ཀུའི་མེ་ཏོག་དང༌། །འབྲས་ཆན་དང་། མེ་ཏོག་དཀར་པོ་དྲི་ཞིམ་པ་དང་། ནཱ་ག་གེ་སར་དང་། སྤོས་ཀྱི་རོང་བུ་་དང༌། ཀྲྀ་སྦྲང་རྩི་དང་། མར་དང་། །ཇི་ྋལྟར་རྙེད་པའི་རྫས་རྣམས་དང༌། །ཞི་བའི་ལས་ལ་སྦྱིན་སྲེག་བྱ་སྟེ། །མངར་གསུམ་དང་ནི་སྦྱར་བ་ཡི། །ཟས་མཆོག་གི་ནི་གོང་བུ་དང༌། །སྦྲང་རྩི་མར་དང་ཞོ་སྦྱར་ལ། །ཏིལ་ནག་ཞོ་དང་མར་ནག་དང༌། །ཚེ། ཤ་ཏ་པུ་བིལ་བ་དང༌། །:ཤཱ་ལིའི་འབྲས་་་་དང་འབྲས་དཀར་དང༌། །ནཱ་ག་གེ་སར་པདྨ་དང་། །ཀཎི་ཨེཀ་ཡི་མར་དང་—།
