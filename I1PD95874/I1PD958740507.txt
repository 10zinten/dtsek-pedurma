ཝ་ཁ་ཏི་ཀ་བསྤྲིན་གྱི་རྒྱ་མཚོའི་ལྷ་མོ་རྣམས་སུ་གྱུར་ཏེ་བྱུང་ནས། སངས་རྒྱས་ཀྱི་རྫུ་འཕྲུལ་དང་། །ཆོ་འཕྲུལ་ཐམས་ཅད་བསྟན་ནས་བཞུགས་སོ་ཞེས་བྱ་བར་སྦྱར་རོ། །མཆོད་པའི་སྤྲིན་གྱི་རྒྱ་མཚོའི་ལྷ་མོ་དེ་རྣམས་ཀྱང་ཁྱབ་པར་ཇི་ལྟ་བུ་ཞིག་ཅེ་ན། ཆོས་ཀྱི་དབྱིངས་ཐམས་ཅད་དུ་ཁྱབ་པ་ཞེས་བྱ་བ་གསུངས་ཏེ། །ཆོས་ཀྱི་དབྱིངས་ཐམས་ཅད་ནི་སྟོང་པ་ཉིད་དོ། །་། དེར་ཁྱབ་པ་ནི་སྤྲོས་པ་ཁྱབ་པར་གྱུར་པའོ། །དེ་བཞིན་དུ། ནམ་མཁའི་དབྱིངས་ཀུན་དུ་རྒྱས་པ། །ཞེས་བྱ་:བ་ནི། འདི་ནམ་མཁའི་དབྱིངས་ཐམས་ཅད་དུ་ཀུན་ནས་རྒྱས་ཤིང་ཁྱབ་པས་ནམ་མཁའི་དབྱིངས་ཀུན་དུ་རྒྱས་པ་ཞེས་བྱ་སྟེ། །འདི་ནི་འཇིག་རྟེན་དུ་གྲགས་པས་བཤད་པ་ཡིན་ནོ། །དེ་ལྟ་བུ་ྋསུ་ཞིག་ཡིན་ཞེ་ན། །རྣམ་པ་སྣ་ཚོགས་པའི་མཆོད་པ་ཐམས་ཅད་ཅེས་བྱ་བ་གསུངས་ཏེ། པརབ་ཏུ་དགའ་ཞིང་བདེ་བ་མ་ལ་སོགས་པའི་རང་བཞིན་ནོ། །དེ་རྣམས་ཀྱི་བཀོད་པ་ནི་ཚོགས་རྣམས་སོ། །དེ་རྣམས་ཀྱི་ཆོ་ག་རྒྱས་པ་ནི་རྣམ་པར་རྒྱས་པའོ། །དེའི་རབ་འབྱམ་ནི་རྒྱུན་ནོ། །དེའི་སྤྲིན་ནི་་དེ་རྣམས་ཀྱི་ཕུང་པོའོ། །དེའི་རྒྱ་མཚོ་ནི་དེ་རྣམས་ཀྱི་ཡང་ཚོགས་ཏེ། དེ་ལྟ་་་བུའི་ཚད་ཀྱི་ལྷ་མོ་རྣམས་སུ་གྱུར་ཏེ་བྱུང་བ་ནི་མཐོན་པོའོ། ་ཌ། བྱུང་ནས་ཅི་ཞིག་མཛད་ཅེ་ན། འཇིག་རྟེན་གྱི་ཁམས་ཐམས་ཅད་ཀྱི་རབ་འབྱམ་གྱི་སྤྲིན་གྱི་རྒྱ་མཚོ་རྣམས་སུ་ཞེས་བྱ་བ་གསུངས་ཏེ། འཇིག་རྟེན་གྱི་ཁམས་ནི་སྟོང་གསུམ་ལ་སོགས་པའོ། །དེ་རྣམས་ཀྱི་རབ་འབྱམ་གྱི་སྤྲིན་གྱི་རྒྱ་མཚོ་དེ་དག་ནི་དེ་བཞིན་གཤེགས་པ་དེ་དག་ཐམས་ཅད་ཀྱི་དཀྱིལ་འཁོར་གང་དག་བཞུགས་པའི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་དཀྱིལ་འཁོར་གྱི་དཀྱིལ་འཁོར་དུ་དེ་ཐམས་ཅད་བསྟན་ནས་བཞུགས་སོ། །ཅི་ཞིག་བསྟན་ཞེ་ན། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བླ་ན་།
