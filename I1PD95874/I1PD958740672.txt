སུ་དྲན་བྱས་ན། །ཞེས་བྱ་བ་ནི། །ཨོ་ཙིཏྟ་པྲ་:ཏི་བེ་དྷཾ་?ཀ་རོ་མི་ཞེས་བྱ་བ་ནས། ཨོ་ཡ་ཐཱ་སཏྠ་ཏ་ཐཱ་ག་ཏསྟ་ཐ་ཨ་ཧཾ་ཞེས་བྱ་བའི་བར་དུ་བྱས་ནས། མི་རྡོ་རྗེ་་་་་དབྱིངས་ཀྱི་སྦྱོར་བ་གང་ཡིན་པ་དེས། །སངས་རྒྱས་རྗེས་སུ་དྲན་པ་བསྒོམས་པས་རྣམ་པར་སྣང་མཛད་བསྒོམ་པར་བྱའོ། །རྣལ་འབྱོར་པ་གང་གིས་སེམས་ཅན་ལ་ཕན་པའི་ཕྱིར་བསྒོམ་པ་བྱས་པ་དེ་ཉིད་ཀྱིས་དེས་ནི་སངས་རྒྱས་ཐོབ་པར་འགྱུར་་་་རོ༑། /། དབང་ཕྱུག་དང་ནི་ཨུ་མ་ཡང་། །ཞེས་བྱ་བ་ལ་སོགས་པས་ནི། །བཅོམ་ལྡན་འདས་རྡོ་རྗེ་ཧཱུྃ་མཛད་ཀྱི་ཕྱག་རྒྱ་ཆེན་པོ་བསྟན་ཏོ། །ས་ལ་བྲིས་ཏེ་དེ་ལྟར་མནན། །ཁ། ཞེས་བྱ་བ་ནི་རས་སམ། སྤང་ལེབ་བམ། རྩིག་པ་ལ་བྲིས་པའམ། ཤིང་ངམ་རྫས་ལས་བླུགས་པའི་གཟུགས་བརྙན་བྱས་པ་ས་ལ་བཞག་་་སྟེ༑ཇི་ལྟ་བའི་བཅོམ་ལྡན་འདས་རྡོ་རྗེ་ཧཱུྃ་མཛད་:ཀྱི་ཧཱུྃ་ཞེས་བྱ་བ་དང་། ཁམས་གསུམ་རྣམ་པར་རྒྱལ་བའི་ཕྱག་རྒྱ་བཅིངས་ཏེ། ཁྲོ་བོ་འདོད་པའི་རྒྱལ་་་པོས་བཀུག་པ་དང་། །བསད་པ་དང༌། །ཚེ་སྦྱིན་པའི་ཕྱག་རྒྱས་གསོ་བ་དང་། ཁྲོ་བོའི་ཞབས་ཀྱིས་འདྲེན་པས་ཨོཾ་བཛྲ་ཨཾ་ཀུ་ཤ་ཧོ%ཞེས་བྱ་བས་དགུག་པ་དང་། ཨོ་བཛྲ་ཨ་བྷི་ཤ་ཧཱུཾ་ན་བ་:ཀྲྀ་ཏྲཏ་ཅེས་བྱ་བས་གཟིར་བ་དང༌། །ཨོཾ་བུདྡྷ་མི་ཏྲི་་་བཛྲ་རཀྵ་ཧཾ་ཞེས་བྱ་བས་བབས་པའི་སྡུག་བསྔལ་བ་དང༌། ཨོཾ་བཛྲ་ས་མ་ཡ་་་:གྲོཙྪ་བནྡྷ་ས་མ་ཡ་ཞེས་བྱ་བ་ལ་སོགས་པའི་སྔགས་ཀྱིས་ཁྲོ་བོ་ཏི་རཱནྟི་རིས་བཅིངས་ཏེ། དམ་ཚིག་ལ་འཇུག་པ་དང་། རྡོ་རྗེ་འདི་ནི་རྡོ་རྗེ་ཆེ། །་། སངས་རྒྱས་ཀུན་གྱི་བྱིན་གྱིས་བརླབས། །རྡོ་རྗེ་འདི་ལས་འདས་ན་ནི། །རིགས་་་་རྣམས་མྱུར་དུ་ཐལ་བར་རློགཚེ། ཨོ་ཧ་ན་ས་མ་ཡ་ཧཱུྃ་ཕཊ་ཅེས་བྱ་བ་ནི། དམ་་་ཚིག་གི་ཕྱག་རྒྱ་གཉིས་པ་བསྟན་ནས། །འ་ལྷ་ལ་སོགས་པ་དགུག་པ་དང༌། རྡོ་རྗེ་ཧེ་རུ་ཀའི་ཕྱག་རྒྱས་རང་གི་ཆུང་མ་དང་བཅས་པའི་ལས་ལ་སྦྱར་ལ། དེ་བཞིན་་།
