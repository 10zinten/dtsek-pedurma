ཞེས་བྱ་བ་ནི་ཇི་ལྟར་རྡོ་རྗེ་དབྱིངས་ཀྱི་གཟུངས་ཀྱི་དཀྱིལ་འཁོར་དུ་གསང་བའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་ལས་བྱུང་བའི་ལས་ཀྱི་ཕྱག་རྒྱ་དུས་ཀུན་དུ་བཅིང་བར་མི་བྱ་བ་དེ་་་བཞིན་དུ་འདིར་ཡང་གསང་བའི་གསང་བ་ཁུ་ཚུར་ལས་བྱུང་བའི་ལས་ཀྱི་ཕྱག་རྒྱ་དུས་ཐམས་ཅད་དུ་བཅིང་བར་མི་བྱ་སྟེ། གསང་:བའི་གསང་བ་ཅིདམ་ཚིག་གི་་་ཕྱག་རྒྱ་མི་བཅིང་ངོ༌། །དགག་པ་དེས་དམ་ཚིག་གི་ཕྱག་རྒྱ་མེད་ན་དམ་ཚིག་གི་་ཕྱག་རྒྱ་དེས་ལྷ་བསྒྲུབ་པར་མི་བྱའོ་ཞེས་སྔར་བསྟན་ཟིན་ཏོ། །དེ་ནས་ཞེས་བྱ་བ་ནི་དེའི་རྗེས་ལ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཧཱུྃ་ནི་རྣམ་པར་སྣང་མཛད་ཀྱི་ལྕགས་ཀྱུ་ལ་ཆོས་ཀྱི་ཕྱག་རྒྱར་བསམ་པར་བྱའོ། །ཏྲེ་ནི་དོན་ཀུན་གྲུབ་པ་སྦྱིན་པ་དང༌། ནོར་བུ་རིན་ཆེན་འགུགས་མ་དང་། །ནོར་བུ་དམ་ཚིག་གི་གཟུངས་ཀྱིའོ། །བྷེ་་ནི་ནོར་བུའི་དམ་ཚིག་གི་རིན་ཆེན་ལྕགས་ཀྱུ་མའིའོ། །ཧྲེ་ནི་ནོར་བུ་ཆགས་པའི་་དམ་ཚིག་མའིའོ། །སྭེ་ནི་ནོར་བུ་ལེགས་པ་མའི་འོ། །ཀྱེ་ནི་རྡོ་རྗེ་ནོར་བུ་རིན་ཆེན་སྤྱན་དང༌། རྡོ་རྗེ་རིན་ཆེན་ཕྲེང་བ་མ་དང་། །ནོར་བུ་རིན་ཆེན་ཕྲེང་་་་བའི་དབང་བསྐུར་བའི་གཟུངས་རྣམས་ཀྱིའོ། །ཁཾ་ནི་ནོར་བུ་རིན་ཆེན་ཉི་མའིའོ། །དྷི་ནི་ནོར་བུ་ཟླ་བ་རྒྱལ་མཚན་གྱི་རྩེ་:མོ་མའི་འོ། །བྱེ་ནི་ནོར་བུ་བཞད་པ་མའི་འོ༑@། དྲི་ནི་ནོར་བུ་རིན་ཆེན་པདྨ་མོ་དང་། ནོར་བུ་རིན་ཆེན་གཏོང་བའི་དམ་ཚིག་མ་དང་། །ནོར་བུ་ཏིང་ངེ་འཛིན་པདྨ་རྣམས་ཀྱིའོ། །བྱི་ནི་ནོར་བུ་་དམ་ཚིག་རལ་གྲི་མའིའོ། །ཀི་ནི་ནོར་བུ་དམ་ཚིག་འཁོར་ལོ་མའིའོ། @ས། །བི་ནི་ནོར་བུ་:གསུང་བ་མཆོག་མའིའོ། །ཀྲུ་ནི་ནོར་བུ་རིན་ཆེན་ཆར་འབེབས་མ་དང༌། མཆོད་པ་ཆེན་མོའི་དམ་ཚིག་མ་དང་། ནོར་བུ་རིན་ཆེན་ཆར་གྱི་རྒྱུན་མ་རྣམས་སོ། །རི་ནི་ནོར་བུ་དམ་ཚིག་བསྲུང་བའིའོ། །།སྤུ་ནི་ནོར་བུ་མཆེ་བ་གཙིགས་མའིའོ། །མཱི་ནི་ནོར་བུ་དམ་ཚིག་ཁུ་ཚུར་གྱི་གཟུངས་ཏེ། ལྕེ་རྣམས་ལ་བསམ་པར་བྱའོ། །།
