ཡེ་དྷརྨ་ཧེ་ཏུ་པྲ་བྷ་བ་ཧེ་ཏུནྟེ་ཥཱནྟ་ཐཱ་ག་ཏོ་ཧྱ་བ་དཏ། དྷ་ཨེ་ཝཾ་བ་དི་མ་ཧཱ་ཤྲ་མ་ཎི། དགེའོ། །།།ཏེ་ཥཱཉྩ་ཡོ་ནི་རོ་བ་བསྡུར་མཆན། དགེ་བའི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་སྟོད་ཆ་ཐོ་དུམ་པའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་ཆོས་་་མངོན་པ་མཛོད་ཀྱི་གནས་བཞི་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»ཅིག། ཨེ«པེ་»ཡང་། «པེ་»«སྣར་ལྷ། «པེ་དགེ་སྐྱེའོ། །།«པེ་ལུས། @«པེ་+རོ། །«པེ་»—དེ། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་»རིམ། @«པེ་»༼སྣར་»ཀྱི། «པེ་»«སྣར་»གིས། «པེ་རྟེན། @«པེ་»ཀྱིས། ་་་་་་(«པེ་»«སྣར་ཅེས། ཉི«པེ་»དི། «པེ་བརྟེན། ཁོའི༽«པེ་»«སྣར་རིགས། «པེ་»«སྣར་»རྗོད། ཉི«པེ་»རེག་ཚ«པེ་»ཆི། «སྣར་»མཆི། །«པེ་»ཉ་པའི་ཁྱིའུ། «སྣར་»ཉེ་བའི་ཁྱིའུ། ཨི«པེ་»«སྣར་»མེད། འི། །«པེ་»«སྣར་བླ། «པེ་»གསུམ། །ད་«པེ་པའམ། «པེ་»«སྣར་གྱི་སྒྲུབ། «པེ་»«སྣར་ཅེས། །དེ་བས་་་་་་འི་སྒོ་(«སྣར་»མར་བ། «པེ་»«སྣར་»ལེན། «པེ་»«སྣར་»ཏེ། (ཆོས་ཚན་འདིའི་ནང་དུ་«པེ་»«སྣར་ལ་ལྷག་བཅས་ཀྱི་ཕྲད་སྐབས་སྐབས་སུ་སྒྲའི་སྦྱོར་ཚུལ་དང་མི་མཐུན་པ་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་་་་མེད༑ཆོ«པེ་པ་ལ། «པེ་»«སྣར་མའི། བཞི། །ཁྱོ་བསེར་«པེ་»«སྣར་བརྟག། «པེ་»པའི། «པེ་»བྱི། «སྣར་»ཐོས། ་«པེ་»«སྣར་ན། ༄༅«པེ་»འདོན། །དོན་དང་། བ་མརམེ་ལོང་། «པེ་»«སྣར་ལ་རག®«པེ་»«སྣར་གྱི། ::ས་པོ་པདེཏེ། «པེ་»«སྣར་བརྟགས། ཉི«པེ་»«སྣར་པ་རིགས་ཞེ། «པེ་བཏག་་(«པེ་»«སྣར་»གི། «པེ་»«སྣར་»ཀྱིས། «པེ་»སྣར་ཀྱི། :་ན་་ན་་«སྣར་»སྨྲོས། ཉི«པེ་»«སྣར་བསམ། «པེ་»«སྣར་གྱི། «པེ་»«སྣར་བ། ཚུ«པེ་རིག། །ལ«པེ་»《སྣར་པ། @«པེ་»《སྣར་དེ། འ་ཀྱི་དབང་དང་། འོད་དང་། དེ་«སྣར་»ཅིག«པེ་»«སྣར་»ཅིག«པེ་རྗེས། «པེ་»«སྣར་»སྟེན། ་་་་་་་«པེ་«སྣར་»འགྲིབས། ཉི«པེ་»«སྣར་»+ཅིགQ«པེ་དབྱི། «པེ་»«སྣར་པ། «པེ་»::«སྣར་»དོ། །པོ«པེ་པོའོ། །།
