བསྟན་འགྱུར། རྒྱུད། ངའི:་བདེ་བ་ཁྱོད་དང་མི་ཁྱུང་དང་ས་ག་ཟླ་བ་ཡང་། ད་ཟབ་ਉਸਨਾਲਸਰਦੀਤੋਂਇਸਨੂੰਸਮਝਾਉਸਦਾਮਨਿਲਓਪਰਵਾਰ:ཀ། ཁཆེན་བལ་ཚོ་བ། བསྡུར་མཆན། །ཁའི་་«པེ་»«སྣར་»༄༅། །ཀྱེ་རྡོ་རྗེའི་དེ་ཁོ་ན་ཉིད་རྣམ་པར་ཕྱེ་བ་བཞུགས། «པེ་»«སྣར་»ནོར་བུ། །«པེ་»«སྣར་དངོས། ་ ེ«པེ་»«སྣར་འབར། «པེ་»«སྣར་བཟང་། མི་«པེ་»«སྣར་ཁཱ། Q«པེ་»«སྣར་»པར། སྔགས་པ་དཀློང་ལ་ངོ་ ེ«པེ་»«སྣར་མ། ཉི«པེ་»«སྣར་འདོད། །(«པེ་»«སྣར་»དང༌། «པེ་»«སྣར་»པ། «པེ་»«སྣར་བས། «པེ་»«སྣར་ལ། འི་ལོ་བས་བབས་ཀྱི་སྨིན་པརྒྱུད། （ར་་གནས་པ་དཔེ་གཞི། །སྡེ་ཚན། །དང་ནང་དང་། །འ། །།སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། རྒྱུད། རྒྱུད་འགྲེལ། སྤྱོད། རྒྱུད། པོད་རྟགས། བ་བརཚན་གྲངས། ལྡེབ། །བ་བར། ནབར། བནས། ཕྲེང་། །ཚེག་ཁྱིམ། ཡོང་བ་ལ། དུ་མ་དང་པ་རབ་ན་འབད་པས་ན་དང་དེ་དག་ལ་ལ་ལ་ལ་་་་་བ་དང་། བོད་འད། །
