དུར་ཁྲོད་བརྒྱད་ཀྱི་བཤད་པ་ཞེས་བྱ་བ། བྱ། །སློབ་དཔོན་རྣལ་འབྱོར་པས་མཛད་པའི་དུར་ཁྲོད་བརྒྱད་ཀྱི་བཤད་པ་་་་་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་བུདྡྷ་ཀ་ར་བཛྲ་དང་། བོད་ཀྱི་ལོ་:ཙཱ་བ་འགོས་ཀྱིས་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»༄༅། །དུར་ཁྲོད་བརྒྱད་པ་བཞུགས། །«སྣར་»༄༅། །དུར་ཁྲོད་བརྒྱད་པ་བཞུགས་སོ། །︽«པེ་《སྣར་ཤ་མ་ན་ཤཱ་ནཱ་མ། ཚེ«པེ་»སྣར་—ཞེས་བྱ་བ། «པེ་»«སྣར་»དབང་པོ། མཁས་«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་གཞོན། «པེ་»«སྣར་»བསྡིགས། «པེ་»«སྣར་པདྨ། «པེ་»«སྣར་»མདོང༌། @«པེ་»«སྣར་སྟན། ཚེ«པེ་»«སྣར་»དུམྤ། «པེ་»«སྣར་བསྡིགས། «པེ་»ལ། Q«ཅོ་གཏུང་། «པེ་»«སྣར་»དབང༌། «སྣར་»པར། ཨེ«པེ་»«སྣར་བུདྡྷཱ། Q«པེ་»«སྣར་»ཙྪ་བ་མགོས། «པེ་»«སྣར་+མངྒ་ལཾ། །པར་བདག་ལ་སོ་་་བ་པ་ལ་བསཔ་ལ་བར་བྱེད་པདང་། མ་་ལ་བོད་པ་དང་། དུས་དབང་དང་མན་བ་དང་། །
