ལ་བར་ན་་་བསྟན་འགྱུར། རྒྱུད། ཇིའིཔཎ་་་གྱི་སྐུ་པར། བླ་མ་ཡེ་ཤེས་དཔལ་གྱི་ཞབས་བཏུད་ནས། །རྗེ་བཙུན་བདེ་མཆོག་སྐུ་ལ་་དད་པ་ཡིས། །བྱིན་རླབས་མངོན་སུམ་ལམ་རྣམས་གང་ཐོབ་པ། །རྒྱུད་རྒྱལ་་་དོན་རྣམས་ཕྱིན་ཅི་མ་ལོག་རྟོགས། །གཞན་གྱིས་བསྐུལ་ཕྱིར་རང་གིས་རྟོགས་་་་བཀོད་པ། །འགྲོ་བ་འདི་ཀུན་མངོན་སུམ་ལམ་ཞུགས་ནས། །རྣམ་ཤེས་ཡེ་ཤེས་གོ་འཕང་ཐོབ་ནས་ཀྱང་། །གཉིས་མེད་བདེ་མཆོག་ངང་དུ་འགྲོ་བར་ཤོག་། ་གཤརྒྱུད་ཐམས་ཅད་ཀྱི་གླེང་གཞི་དང་གསང་ཆེན་དཔལ་ཀུན་དུ་ཁ་སྦྱོར་ལས་་་བྱུང་བའི་རྒྱ་ཆེར་བཤད་པ་རིན་པོ་ཆེའི་ཕྲེང་བ་ཞེས་བྱ་བ། རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་དཔའ་བོ་རྡོ་རྗེའི་ཞལ་སྔ་ནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །བ་ཀསུ་་་བར་་སྟེ། ་བ་ལོ་:བྱབསྟར་མཆན། ，པལ་བཨ་ཞེ་་ཞུ། གྲོ་མ་པ་དུ་(མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད། Q«པེ་»«སྣར་»+དགོས་པ། «པེ་»«སྣར་»རྒྱུད། «པེ་»«སྣར་»གཟིགས་ཏེ། ཀྱི«པེ་»«སྣར་པས། «པེ་»«སྣར་»དང་པོ། སྐྱེ་བ་:། «པེ་»«སྣར་»གསུམ། «པེ་»«སྣར་»པ་དང་། ཉི«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་བསྟེན། «པེ་»«སྣར་»བརྡའི། @«པེ་»«སྣར་»ཚོགས་འཁོར་གྱི། ༄༅«པེ་»«སྣར་པ། «པེ་»«སྣར་»ཏོ། ན«པེ་»«སྣར་བརྒྱད། །«པེ་»«སྣར་—ལྷག་མ་གསུམ་གྱིས་ནི་འབྲས་བུའི་རྒྱུད་བསྟན་ཏོ། །«པེ་»«སྣར་»སྟོན་ཏོ། «པེ་»«སྣར་»ལ་སོགས་པ། «པེ་»«སྣར་བཟངས། «པེ་»«སྣར་པོའི། «པེ་»«སྣར་»མ། «པེ་»«སྣར་»དུས། «པེ་»«སྣར་»པོ། «པེ་»«སྣར་»ཞུགས། །«པེ་»«སྣར་—པ། @«པེ་《སྣར་ལས། «པེ་»«སྣར་—བཞི། :བདུན་«པེ་»«སྣར་»+མ། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་+བ། «པེ་»«སྣར་»གི། སུ«པེ་»«སྣར་»ནི་བརྗོད། །
