བསྟན་འགྱུར། རྒྱུད། ཇི་། ན། །་་་་་་་་་་། དང་mཆོ་་ཀོབ་པ་་གལོ་ਨੀਮੈਲਵਮਾਲਕਾਂਨੂੰਛੁਟਣੋਂਨਹਨੂੰਸp'ਨੂੰ། །ཅན་ངེ་བསྡུར་མཆན། སྤྲོ་བ་ལ། །པའིད་ལ་རེ་ཚེ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»༄༅། །དུར་ཁྲོད་བརྒྱད་ཅེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»གླ། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་གཏུན་མོ་མཆོག་ག་ཧ་ར་ཉིད། ཁེའིཀྱི་ཚེ«པེ་»«སྣར་ཡང་། ་་་་་་པོ་པ་ཊ་ཁ་ལ་ਅੱਜਤੋਂਲਈਗਰਾਂਝਵਾਲਤਲਵਾਰਵਰਤੋਂਲਤཁ་བ་དང་བསྡུར་འབྲས་རེའུ་མིག་ཏུ་ང་ཚོ། ས་ག་ཟླ་བ། ས་ལ་པ་དང་། ཕྱོགས་གང་དག་ཚེ་རྒྱུད། （རེ། ་ཌ་ད་བ་བྱ་བ་ལ་སྣར་ཐང་། རྒྱུད། །གང་རྒྱུད། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྡེ་ཚན། །རྒྱུད་འགྲེལ། ་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན་བ། ན་བ་ཕྲེང་། །ཚེག་ཁྱིམ། །།བན་། ནབ། ་དང་། ད་་་་་་བ་པད་མ་འཚོལ་བ་གང་ཉིད་དང་། ལ་བབ་པ་ཞེས་པ་ལ་དེ་དག་ང་ལ་བག་ན་བ་དང་། དབང་ལྔ་་་་ལའི་།
