་ཡོ༌་་་༌་་པ། །དོ། ་ཡོད་ལ་ཡི་ག་ལྔའི་དབང་པོ། །ཚེ་མ་ཀ་ལ་ཚེ་དཔལ་ལྔ་པ་ལ། །ཁྱད་པར་ཁ་རྒྱཁ་། ཏི། །ཀ་aག་ང་། །ཁ་ཁ་ཏུ་ཎི་ཀྱི་ཆཏཚོ། །ཁ་ཁ་ལོ་བ་ཚོད་བྱ་བདེ་བ་དང་ཁོਨਾਸਨਵਲੋਂਟਲੀਬਠਿੰਨਸਾਲਦਾ༄༅། །རབ་ཏུ་མི་གནས་པ་གསལ་བར་ਲਇਸਨੂੰཔ་ਟgeསྟོན་པ་ཞེས་བྱ་བ། ਚਨਸਨੂੰਸਲਲਾ,ਊਲਲਲਲਉਸਨੂੰਨਾਸਰੋ'ਝਲਓ।ਉਸਨੇਸ਼ਰਨਾਮਾਵਿੱਚਵਸਤੂਸਰਸਕਮਾਓਸਪਰਸ਼ਕੋਡਨੂੰਸਰਵੋ।ལ་ཁ་༄༅། །རྒྱ་གར་སྐད་དུ། ཨ་པྲ་ས་ཧ་པྲ་ཀ་ས་ནཱ་མ། བོད་སྐད་དུ། རབ་ཏུ་མི་གནས་པ་གསལ་བར་སྟོན་པ་ཞེས་བྱ་བ། འཇམ་དཔལ་གཞོན་ནུར་་་་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །ཤེས་པ་རབ་ཏུ་མི་གནས་པ། །སངས་རྒྱས་ཕ་ནོར་ཐམས་ཅད་འདོད། །འོན་ཀྱང་འབད་མེད་སྦྱོར་བ་ཡིས། །གང་ཚེ་སེམས་ཅན་དོན་བྱེད་ནའོ། །གང་ཕྱིར་གཞན་སེལ་སྒྲུབ་ལས་སྐྱེས། །དེ་ནི་རྟག་དང་ཆད་་་ལ་གནས། །རྟག་ཏུ་མ་སྐྱེས་མི་འགག་ཉིད། །སྐྱེ་དང་འགག་པ་བྱོལ་སོང་ཚིག། གལ་ཏེ་རང་རིག་ཚད་མ་ནི། །རིག་དེ་སེམས་ཅན་འདོད་མ་ཡིན། །སྣང་བ་ཐམས་ཅད་:སྤང་བའི་ཕྱིར། །སེམས་ཅན་བདེན་པ་མ་ཡིན་ནོ། །རྗེས་ལ་་་་་སེམས་ཅན་འདོད་ཅེ་ན། །དེ་ཚེ་སེམས་ཅན་རྫས་ཡོད་མིན། །དམ་བཅས་་་་་གཅོད་པ་གནོད་པའི་དོན། །རྗེས་སུ་རྟག་པ་ཞེས་བརྗོད་དོ། །དངོས་གཞི་ལ་ཡང་རིག་པ་ཡོད། །རྗེས་ལ་དེ་ལྟར་སྣང་བའི་ཕྱིར། །དང་པོ་མི་རྟོག་པར་་—།
