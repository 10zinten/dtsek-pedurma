བསྟན་འགྱུར། རྒྱུད། ཝི། རྒྱ་མཚོར་ནུབ། །སྔགས་ཀྱི་དེ་ཉིད་རྣལ་འབྱོར་པས། །དོགས་མེད་དོན་ལ་རབ་ཏུ་གནས། །སྲིད་པ་ཡང་འདི་བླ་མར་བྱས། །དོགས་པ་མེད་པའི་ཡུལ་གྱིས་སྟོན། །ལྷན་ཅིག་སྐྱེས་པ་དྲུག་པ་བླ་མ་མི་ཏྲི་པས་མཛད་པ་རྫོགས་སོ། །།།བླ་མ་བཛྲ་པཱ་ཎི་དང་། མཚུར་གྱིས་བསྒྱུར་བའོ། །།།།།།ལ༌། །:ཀྱི་ཁྱད། བསྡུར་མཆན། (མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་ལྷན་ཅིག་སྐྱེས་པ་དྲུག་པ། འ«པེ་»«སྣར་ཕྱུགས། «པེ་»«སྣར་+རྣམས། ®«པེ་»«སྣར་»—པ། ད་པ་ཆ་ར་ཎ། །བོ་བསྡུར་འབྲས་རེའུ་མིགའོ་འོལ་ཁ་བ་འི་nཀྱི་བྱཀ་བའམ་ཁབོ་རྒྱུད། （ལ་འཁ་པའི། །ལ་་ད་ཀཡོདཔེ་གa,། དཔེ་གཞི། །རྨ.སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །ག་ཚརྒྱུད། ཀརྒྱུད་འགྲེལ། །རལ་རྒྱུད། རྒྱུད། ལས་ཀ་པོད་རྟགས། །ཚན་གྲངས། །།ལྡེབ། ན་—ན་། བན་བ་ནན་_ན་ཕྲེང་། །༈ཚིག་རྐང་། ཚེག་ཁྱིམ། །ནསའིརབབབབཔལSནི། ་་ཁྱབ། །ཀུན་ཏུ་དཀཚོ་དང་ཀཚོ་ན། །ཟད་པ་དང་ལ་ལ་།
