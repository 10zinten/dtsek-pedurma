པ་ལྷའི་བླ་མ་ལ། །རླུང་ཆུ་ཚོད་གཉིས། མེ་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་་་གཉིས། ས་ཐུན་ཚོད་:གཅིག། ཆ་། ཆུ་ཆུ་ཚོད་གཉིས། ་རླུང་ཆུ་ཚོད་གཉིས་སོཌཌ་། མཚན་མོ་མེ་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ས་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས་སོ། །་་་ཎཀོཁོ་གཟའ་པ་སངས་ལ། མེ་ཐུན་ཚོད་གཅིག། ས་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས། ་ས་ཆུ་ཚོད་གཉིས་སོ༑ཥ། མཚན་མོ་ས་ཐུན་ཚོད་གཅིག་དང་། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས༑མེ་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ས་ཆུ་ཚོད་གཉིས་སོ། །ན། གཟའ་སྤེན་པ་ཤར་བ་ན། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས། མེ་ཐུན་ཚོད་གཅིག། ས་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ཆུ་ཆུ་ཚོད་གཉིས་སོ། །མཚན་མོ་རླུང་ཆུ་ཚོད་གཉིས། །མེ་ཐུན་ཚོད་གཅིག། ས་ཐུན་ཚོད་གཅིག། ཆུ་ཆུ་ཚོད་གཉིས་སོ། །ན་གཟའ་ཉི་མ་ཤར་བ་ན། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས། མེ་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་གཉིས། ས་ཐུན་ཚོད་གཅིག། ཆུ་ཆུ་ཚོད་་་་་་གཉིས་སོ། །།མཚན་མོ་རླུང་ཆུ་ཚོད་གཉིས། མེ་ཐུན་ཚོད་གཅིག་དང་ཆུ་ཚོད་་་གཉིས༑ས་ཐུན་ཚོད་གཅིག། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས་སོ། །གཟའ་ཟླ་བ་ཤར་བ་ན། མེ་ཐུན་ཚོད་གཅིག། ས་ཐུན་ཚོད་གཅིག་་་་དང་ཆུ་ཚོད་གཉིས། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས། ་་:མེ་ཆུ་ཚོད་་་གཉིས་སོ། །མཚན་མོ་ས་ཐུན་ཚོད་གཅིག། ཆུ་ཆུ་ཚོད་གཉིས། རླུང་ཆུ་ཚོད་གཉིས། :：མེ་ཐུན་ཚོད་གཅིག། ས་ཐུན་ཚོད་གཅིག་གོ། །རྣལ་འབྱོར་པས་འདི་ལྟར་བྱ་སྟེ། རླུང་དང་མེ་གཉིས་སྤང་བར་བྱའོ། །།
