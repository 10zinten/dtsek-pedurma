བསྟན་འགྱུར། ་རྒྱུད། ནི་ས་ག་གྱིས། @«པེ་»«སྣར་»ཉེན། གྱེ«པེ་»«སྣར་»པའི། དགྱེཙེ«པེ་»སྣར་པ། ་《བ་རྒྱལ་བ་«པེ་»«སྣར་གིས། «པེ་»«སྣར་»ལོཙྪ། །ར་། གོ་བ་ཚེ་སྐྱེལ་བ་ཡིན་ན་བདག་མེད་པ་དང་། །ཞེས་བློ་དབང་སྒྱུར་བ་མེད་དེ། ལ་ཎ་ཊ་ཀ་པ་ག་བ་དང་། དེ་བརྒྱ་བདུན་བར་དུ། འམ་འབད་ཀྱང་ཏེ་ཁྱེད་::ཀྱིབ་དང་། །བ་བསྡུར་འབྲས་རེའུ་མིག་ན། །ཁྱོད༌ཀྱི་་་་་་་་་་་་་་་་་རྒྱུད། (ང་ཚོས་ཁྱོད་ཀྱི་དྲ་བ། དཔེ་གཞི། ཅོ་ནེ། །སྡེ་དགེ། རྒྱུད། །པེ་ཅིན། །རྒྱུད་འགྲེལ། སྣར་ཐང་། རྒྱུད། སྡེ་ཚན། །།།པོད་རྟགས། ཚན་གྲངས། །་་ལྡེབ། བ?ན། ནན་[བན། [བ?ན་ཕྲེང་། །[ཚེག་ཁྱིམ། །།།རེ་བ་་་་་་དཔེ་་་་་,། །པ་དང་ལེ::། ་་་་ས་ག་ལ་བགས་ལེགས་དང་བཞི་བ་བྱ་བ་ཡོད། །་་་ཆུ། སྐྱོ་དྲན་གདུང་བ་ན། ད་ག་འི་ལན་བའི་དག་པ་དྲན་པ་རྒྱུན་མེད། ངས་དེའི་དཔེ་དེ་ནས་ད་བར། ཁ་བ་ལ། ཕྱི་ཟླ་དགུ་བར་ཁྱུ། །།ཆོ་ག་་་། གང་ན་བ་དང་། །ངག་ཡིད་དུ་འོང་བ་དེནང་བ་ཁོ་ནར་ཁ་དོནང་་་་་:། དང་ཀྱི་ཀྱིད/བ་པ་དེ་དང་། ད་དེ་ག་ང་བ་མ་འ་ར་ལ་ས། ::ཁེར་རྒྱུ་བ་དང་། དབང་པོ་དང་། ཞིབ་དང་ཞི་བ་། —།
