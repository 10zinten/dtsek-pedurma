དངོས་པོ་གསལ་བའི་རྗེས་སུ་འགྲོ་བའི་དེ་ཁོ་ན་ཉིད་གྲུབ་པ། །རྣམ་པ་གསུམ་དུ་བཤད་། :། དེ་ཡི་གཞི་ནི་བཤད་པར་བྱ། །ཕྲ་བའི་གཟུགས་ཅན་རླུང་དང་ནི། །ཡེ་ཤེས་དག་ནི་འདྲེས་གྱུར་པ། །དབང་པོ་ལམ་ནས་བྱུང་ནས་ནི། །ཡུལ་རྣམས་ལ་ནི་སྨིན་པར་འགྱུར། །གང་ཚེ་སྣང་བ་ལྡན་:འགྱུར་བའོ། །རླུང་ནི་བཞོན་པར་གྱུར་པ་ཡིན། །དེ་ཚེ་རང་བཞིན་དེ་ཐམས་ཅད། །མ་ལུས་ལུས་པ་མེད་པར་འབྱུང༌། །རླུང་ནི་གང་དང་གར་གནས་པ། ་། རང་བཞིན་དེ་དང་དེ་རྗེས་སྤྱོད། །ཅེས་དགོངས་པ་ལུང་སྟོན་པས་སོ། །།དཀྱིལ། །ཅིག་་། ན་epཏོ། ་་ས་ཆ་ཡང་ན་རིམ་པ་ལྔ། ངས་རྒྱལ་བ་ན་རྒྱལ་བ་སྐྱོ་བ་འགས་གཡག་ལ། །མ་རྩ་བ་ཡང་སྐྱེ་ན:བཀླུབ་ཀྱི་ཀྱི་གྱི་བསྡུར་མཆན། །ཆ། ང་། པེ་གང་བྱུང་། ་འི། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དངོས་པོ་གསལ་། བའི་རྗེས་སུ་འགྲོ་བའི་དེ་ཁོ་ན་ཉིད་གྲུབ་པ། «པེ་»«སྣར་»ཏ་ཏནྟཱ། «ཅོ་»ཊནྟཱ། Q«པེ་»«སྣར་»གི། {He«པེ་»«སྣར་རིངས་སུ། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»ཀྱི། །«པེ་»«སྣར་»དུ། Q«པེ་»«སྣར་པས། «པེ་»«སྣར་»ནས། «པེ་»«སྣར་»འཐུག་པོར་བྱུགས། «པེ་»«སྣར་བཀྲུས། «པེ་»«སྣར་»མཆོད། ་«པེ་»—དུ། འ«པེ་»«སྣར་»མོ། @«པེ་»«སྣར་སྒྲོག་ ེ«པེ་»«སྣར་གྱིས། «པེ་»«སྣར་»+སྤྱོད་པས། ཉེ«པེ་»«སྣར་»ལྟ། །ཉེ«པེ་»«སྣར་»ལག་«པེ་»«སྣར་»སྐྱོང་། «པེ་»«སྣར་གྱུར་པ། «པེ་»«སྣར་མེད། ཉི«པེ་»«སྣར་མི། «པེ་»«སྣར་»གོ། «པེ་»«སྣར་གྱི། «པེ་»«སྣར་བཅིངས་ཏེ། «པེ་»«སྣར་»པས། :«པེ་»«སྣར་»གྱུར་པའི། «པེ་»«སྣར་»པས། «པེ་»«སྣར་ཀྱི་ཐིགས་པ། འགྱེ«པེ་»«སྣར་རྟོགས། སྟེ«པེ་»«སྣར་བསྐྱེད། སྟེ«པེ་»«སྣར་འབྱུང་། «པེ་»«སྣར་ཀྱི། ལུ«པེ་»«སྣར་»+ཀྱི། ཨེ«པེ་»«སྣར་»གི། «པེ་»«སྣར་གྱི། «པེ་ཀྱིས། «པེ་»«སྣར་»ཏུ་བཞག«པེ་»«སྣར་»«ཅོ་»ཚིག«པེ་»«སྣར་»ལྟས། ་«པེ་»«སྣར་+ཁ། «པེ་»«སྣར་བུ།  ི«པེ་»«སྣར་བརྒྱུད། «པེ་»«སྣར་»—ཚོར་བའི། །«པེ་»«སྣར་བར། །«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་»—དེ་ལྟར། «པེ་»«སྣར་དྲག«པེ་»«སྣར་རླབས། །
