བསྟན་འགྱུར། *རྒྱུད། ་༼ཞིབ་གསལ། ་། ཧ་ཡིས་བདེ་བ་འཕེལ་བར་བྱེད། །z། །ཧིཀ་གིས་བདེ་བ་གནས་ནས་་འདྲེན༑། །ཕྱི་ནས་འབྱུང་བའི་དགྲ་གཤེད་དང་། །།ནང་ནས་འབྱུང་བའི་རོ་སྙོམས་དང་། །ཕྱི་ཡི་དུས་ཚོད་གཡོན་ནས་གཟུང་། ཊ། ནང་གི་དུས་ཚོད་གཡས་ནས་གཟུང་། །ལོ་ཡི་མགོ་བོ་བྱི་བ་ཡིན། ་། ཟླ་བའི་མགོ་བོ་སྟག་ཡིན་པས། །ལྷོ་དང་བྱང་དང་ཤར་དང་ནུབ། །འབྱུང་བ་བཞི་རུ་ཤེས་པར་བྱའོ། ་། ཤེས་ན་་ཐམས་ཅད་འགྲུབ་པར་ངེས། །ཚིག་གཅིག་བསྟན་ོཔས་སངས་རྒྱས་དང་། །རྫས་གཅིག་རྙེད་པས་དོན་ཡོད་འགྲུབ། །སྔགས་གཅིག་བཟླས་པས་ནུས་པ་་་་འབྱུང༌། །མཉམ་ཉིད་གཞིའི་སྒྲོན་མ་རྫོགས་སོ། །།།འོ་ཚེ་ཡང་བཏང་ਲਾਲਸਬੰਧਈ।ਸਰਸਮਲਵੋਨੌਓਲਸਰབྱེ་། ཆུ་ཚོ། ཀ་ཨ་། །བྱ་ཀ། ང་། །ཅ་ཏཚོ་དད་ཅན། །བ་མ་འ། །བབན་བསྡུར་མཆན། དེ་ལ་དང་པད་མ་བདག་ཏ་དེ་དུས་དང་ཀ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། མཉམ་ཉིད་་་་གཞིའི་སྒྲོན་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །༼«པེ་《སྣར་»པཱ། z་%/། q*j?«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་»ཀྱིས། །འ«པེ་»«སྣར་»ར་ལ་ལཾ། ®«པེ་»«སྣར་ཧཾ། ས<«པེ་»«སྣར་པར། པོ«སྣར་ཚིགས། «པེ་»«སྣར་»གེ། «པེ་»«སྣར་»ཙ། «པེ་»《སྣར་བྱུང་། །དགྱེ«པེ་»«སྣར་ཡིས། འགྱེ«པེ་《སྣར་ཧཱུྃ་གི་འོག་གིས། ཏནྟི«པེ་»«སྣར་སྡུས། «པེ་»«སྣར་བཟུང་། ་«པེ་»«སྣར་བྱས། «པེ་»«སྣར་སྟན། ཁོ«པེ་《སྣར་»སྔགRਇਮਲੈਕੇਨtoਊਡੀਈਸੀ।་་ཁག་་ཉནམས་། །ཙཎྜ་ཁཚོ་ཏཁོང་མ་:::དག་བཏང་ཏེ་ན་ཚ་ལ་ལེན་ཏེ། །ཆད་པ་:::་ག:བྱིས། འོ་ན་བ། བྱ་བ་སྤྱི་ལུ་། :༈། །གྱི་ལོ་:ཀྱི། །འབའ་དོན་རྒྱུ་ཁོ་ན་ཉི། ཞི་ལ་། #/Sཡོད།
