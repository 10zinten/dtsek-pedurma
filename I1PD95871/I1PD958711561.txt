བསྟན་འགྱུར། རྒྱུད། །བཞི་ཐུབ་པས་བྱིན་གྱིས་འགྲེལ་མཛད་པའི། །བདག་གི་བསོད་ནམས་གང་དེ་ཡིས། །སྐྱེས་བུ་དམ་པ་རྒྱལ་གྱུར་ཅིག། སྤྱོད་པའི་གླུའི་མཛོད་གྲུབ་པ་རྣམས་ཀྱི་རྡོ་རྗེའི་གླུ་དེ་ཁོ་ན་ཉིད་གསལ་བར་ཤེས་པའི་སྐྱེ་བོ་རྣམས་ལ་སློབ་དཔོན་ཐུབ་པས་བྱིན་གྱིས་་འགྲེལ་པ་མཛད་པ་ལས་གྲུབ་པ་རྣམས་ཀྱི་རྡོ་རྗེའི་གླུ་འགྲེལ་པ་དང་བཅས་པ་རྫོགས་སོ༑། །དག་པ་ཞི་བ་ལ་སུན་པ་མེད། གོ་སོགས་ན་ཚེ་ཀུན་གང་པཎྜི་ཏ་ཆེན་པོ་ཀཱིརྟི་ཙནྡའི་ཞལ་སྔ་ནས་གདམས་པ་དང་བཅས་པ་ལེགས་པར་ཞུས་ཤིང་། ཆོས་ཀྱི་རྗེ་དཔལ་ལྡན་ས་སྐྱ་པ་ཁུ་དབོན་རྣམས་ཀྱི་སྐུ་དྲིན་་་་བརྟེན་ནས་བལ་ཡུལ་ཡམ་བུའི་གྲོང་ཁྱེར་དུ་བོད་ཀྱི་ལོ་ཙཱ་བ་བནྡེ་གྲགས་པ་རྒྱལ་མཚན་གྱིས་བསྒྱུར་རོ། །ཆུ་བྱ་མོ། ་ཀླུ་མོ་བར་དུ་ཁང་པ་དེ་ཁྱེཊ། །ਉਧਲੜਮਸਲਕਰਕਬੱਲਟਕਿਸਾਸང་ལ་ཡོད་པ་རེད། བསྡུར་མཆན། བབས་དུ་ཡོད་པའི་ཁ་ནི་ལང་པ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»༄༅། །སྤྱོད་པའི་གླུའི་མཛོད་ཀྱི་འགྲེལ་པ་བཞུགས་སོ། །«པེ་《སྣར་»ཀོཏྟཱ་པཱ་ཎི་བྲིཏྟི། «པེ་»«སྣར་»བཅིངས། «པེ་»«སྣར་ཡཱི། འ«པེ་»«སྣར་»ཐིག«པེ་»«སྣར་པའི་་་་་པ་དགའ་བས། ་(«པེ་»«སྣར་དབྱིངས། །«པེ་《སྣར་པ། ༄༅«པེ་»«སྣར་»ཡི། («པེ་«སྣར་ཀྱི་བཞིའི། །«པེ་»《སྣར་འདིའི། «པེ་»«སྣར་»ཡིས། འུ«པེ་»«སྣར་»འགྲོ་བ། «པེ་»«སྣར་+ཀྱང་། ས《སྣར་རིམས། «པེ་»«སྣར་ཤ། །ང་གཏོ། །ར//ཀྱི་ལ«པེ་»«སྣར་ཡིས། «པེ་»«སྣར་ནས། «པེ་»《སྣར་»ཀྱི། ན་«པེ་»«སྣར་»—ལ། «པེ་»«སྣར་བསྙུང༌། །«པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་+འདོད་ཡོན་ལྔ་ནི་སྤང་མི་བྱ། །«པེ་»《སྣར་»—འདོད་ཡོན་ལྔ་ནི་སྤང་མི་བྱ། །«པེ་»«སྣར་»འཛག(«པེ་»«སྣར་»ཡི། («པེ་»«སྣར་»གི་བསམ་གཏན། ག་བ་ད་ལ་་་་་་་་་་་། «པེ་»«སྣར་མདུན། «པེ་»«སྣར་རིགའ«པེ་《སྣར་»རྒྱུག«པེ་»«སྣར་འཇིགས། ས«པེ་»«སྣར་རུ། «པེ་»«སྣར་»ནས། ་«པེ་»«སྣར་»ཀྱི་གཉིས། «པེ་《སྣར་ལ། །
