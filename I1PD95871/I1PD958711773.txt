ཁ་ར། བསྟན་འགྱུར། རྒྱུད། ་༼ཞི་ཁ་། ན་ཁཆ། །བདག་ཁ་ལ་རྒྱ་བཏོ་ལ་ཁ་དང་། །ཁྱབ་དེལ་ཡག་ས་འདང་བ་དང་། ས་ཁད་ཆ་དག་བྱད་ཀཚོ་ཀ་་ཀ་ར་ལོ་བསྡུར་མཆན། །གང་ལ་ཁ་ཏོར་ཚེ་ལས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །་«པེ་»«སྣར་»བསྒོམས། «པེ་《སྣར་བུདྡྷི། «པེ་»«སྣར་མི། «པེ་»«སྣར་སྔགས། «པེ་»«སྣར་རླངས་་་་པའི་བསྐྱུར། སུ《པེ་»«སྣར་»ཤིས། ་་ཚེསའོ་ཡ་དེཡེ༌་ཟླ་། ཀཱ་ལང་་ས་་་་་་ལ་ལ་ཟླ་དུ་བའང་འག་ཏོགནས་པ་ན། འདྲ་ཊ། ཎ་REARਨਨਨਓਨਰਸབསྡུར་འབྲས་རེའུ་མི། ང་་་། :ད་རྒྱུད། (ཁརེ་ལ་དེ་ཤ་བྱ་བ་དང་། །སྣར་ཐང་། ཅོ་ནེ། ཅུད། དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྡེ་ཚན། །་རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བནབན་བ་ན་བན?ཕྲེང་། །ཚིག་རྐང༌། །།།ཚེག་ཁྱིམ། ་། ་། བར་དབུས་རྒྱུན་ནས་ཕྱི་པའི་ལ། །བང་ར་བ་དེ་ཡི་དང་ར་བ། །དམ་ཚིག་དེ་མབས་དཔལ་དེ་ལ་འདུད་དང་བགྲོད། །བླ་མ་དང་ཡི་།
