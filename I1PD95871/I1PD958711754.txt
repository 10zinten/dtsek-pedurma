དོན་ལ་མཚོན་དུ་ཅི་ཡང་མེད། །།ཐ་སྙད་བློ་ལ་རྒྱུ་འབྲས་:དཱི་པཾ་གཉིས། །ཞེན་པ་བཟློག་ཕྱིར་བསྟན་པ་ལས། །དོན་དམ་བརྗོད་དུ་མེད་པས་ན། །རྒྱལ་བས་་་ཚིག་གཅིག་གསུངས་པ་མེད། །མཐར་ཐུག་འབྲས་བུའི་སྒྲོན་མ་ཞེས་བྱ་བ་རྗེ་་་་་བཙུན་ཀ་མ་ལ་:ཤཱི་ལས་མཛད་པ་རྫོགས་སོ། །པས། །འགྲོ་ལ་བུ་བརྟེན་ག་དུས་པ། དུ་མ་ལས་དང་འགྲོ་བར་བྱ། བདེ་བའི་ངོས་ལ་བྱེ་བ། བབསྡུར་མཆན། ས་སློབ་གྲྭ་བདུན། ལབ། །་དེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»༄༅། མཐར་ཐུག་་་་འབྲས་བུའི་སྒྲོན་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»པྲ། «པེ་»«སྣར་པ། «པེ་«སྣར་རྟོགསུ«པེ་»«སྣར་མི་སྟོར། «པེ་»«སྣར་རྟོག«པེ་»«སྣར་»དཱི་པཾ། «པེ་»«སྣར་ཤིས། པ་དང་ཚོང་པ་དེ། །ཟླ་བ་ཁྱདེཡེས་རྒྱུ། ཚོཊ་ཏེ། སྤྲོ་ན་བཟློ་་་་་་:བཙོ། ཚོང་ཀ་། ང་གང་། །ཁྱི་ཁྲོ། ང་བ་དེ་དྲང་བདེ་བསྡུར་འབྲས་རེའུ་མིགལ་ཁུ་དཀྱིལ་ལ་བརྒྱུད། (བྱས་པ་ཁ་འབྱར་བ། །དེ་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། །ཅོ་ནེ། སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། །པོད་རྟགས། <ཞི་དང་ཚན་གྲངས། ་ལྡེབ། བ་བབ་ན། ན་རབ་བ་—བ་ཕྲེང༌། ཚིག་རྐང་། ཚེག་ཁྱིམ། :།
