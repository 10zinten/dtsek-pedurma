བསྟན་འགྱུར། །རྒྱུད། ༼ཞིཟད་པར་བྱས་ནས་བྱང་ཆུབ་ཆེན་པོ་མངོན་དུ་བྱེད་དོ་ཞེས་བྱ་བའི་དོན་ཏོ། ལ་། ཀ་ནི་ཀཱ་མ་ན་ན་གཅེར་བུ་སྟེ་ལྷན་ཅིག་སྐྱེས་པའི་སེམས་ཏེ་རྟོག་པ་དང་། ཚོ་སྡིག་་་སྒྲིབ་དང་། ཁ་དོག་དང་དབྱིབས་དང༌། ཡན་ལག་དང་། །ཆ་དང་བྲལ་བས་ན་གཅེར་བུའོ། །དེ་ལྟ་བུའི་གཅེར་བུ་དེས་འཁོར་བ་དང་འཁྲུལ་པ་ཐམས་ཅད་བསླུས་པ་སྟེ། །བསླུས་པ་ནི་དེ་རྣམས་ཀྱི་སྐྱོན་གྱིས་མ་གོས་ཤིང་ཕྱག་རྒྱ་ཆེན་པོའི་་དངོས་གྲུབ་འཐོབ་པའོ། །རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཐམས་ཅད་ཀྱི་གཙུག་གི་། ནོར་བུ་ལྟ་བུ་དཔལ་ས་ར་ཧས་མཛད་པའི་ཀ་ཁའི་དོ་ཧའི་བཤད་པ་བྲིས་པ་འདི། རྒྱ་གར་ལྷོ་ཕྱོགས་ཡུལ་ཀོ་ས་ལར་སྐུ་འཁྲུངས་པའི་རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་ཤྲཱི་བི་རོ་ཙ་ན་བཛྲའི་ཞལ་སྔ་ནས་རང་འགྱུར་དུ་མཛད་པའོ། །།ཎ་། །པོ་རྒྱལ་པདེ། གྲོ་ལ་ལུ་དང་ད་ལོ་སུ་ལོ་པ་དེ་ཚོ་ཁ་ས་ཁཀྱི་བ་དང་དེ་དག་ལ་བགྲོ་བ་དང་། ཀྱང་། །དང་ཅ་ད་ཀ་ལ་དཀའ་བསྡུར་མཆན། ཀློག་པ་དང་བདག་ལ་གནང་། ད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»༄༅། །ཀ་ཁའི་དོ་ཧའི་བཤད་པ་བཞུགས་སོ། །་«པེ་»«སྣར་པ་ལ། «པེ་»«སྣར་»ན་གནས། «པེ་»«སྣར་»ཀཱ། («པེ་»«སྣར་+པ། «པེ་»«སྣར་»+དེ་ལྟ་བུའི་གཅེར་བུའི་དོན་ཏེ་རྟོགས་པར་གྱུར་ན། ཁ་ནི་ཁཱ་ཧི་པ་སྟེ་ཟ་བའོ། །རྣལ་འབྱོར་པས་ཟ་བ་དང་འཐུང་བ་དང༌། འདོད་པའི་ཡོན་ཏན་ལ་ལོངས་སྤྱོད་ཀྱང་། དེ་ལ་་་་་གནོད་པར་མི་འགྱུར་ཏེ། མི་གནས་པའི་མྱ་ངན་ལས་འདས་པའི་དོན་མངོན་དུ་བྱས་པའོ། །ཁ་ནི་ཁ་མ་ན་སྟེ་་་་ངོས་གཅེར་བུའོ། །«པེ་»«སྣར་གྱིས། «པེ་»«སྣར་+ན། །དྲན་པ་དང་། པགས་པ་བླ་ན་ཀཱ་«པེ་»«སྣར་ཨ། («པེ་»«སྣར་»ན། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་+ན། («པེ་»«སྣར་པ། །«པེ་»《སྣར་»ཏིར། ་«པེ་»«སྣར་»སོའི། །«པེ་»«སྣར་+ཐིག་ལེ་བཞི་སྟེ། །སྐྱེ«པེ་»«སྣར་+ན། །«པེ་»«སྣར་»བཞག«པེ་《སྣར་»ཛ་ཛ། (《སྣར་»གང༌། «པེ་《སྣར་བརྟན། «པེ་»«སྣར་»ཎ། «པེ་»《སྣར་ན། །།
