རྗེ་ཕག་མོའི་སྒྲུབ་ཐབས་རྗེ་བཙུན་ཀ་མ་ལ་:ཤཱི་ལས་མཛད་པ་རྫོགས་སོ། །ལུ་མ་དང་། །བསྡུར་མཆན། བོད་པ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»༄༅། རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་ཧི་སཱ། («པེ་»«སྣར་བྱིན། «པེ་»སྣར་»གྱིས་ཆང་། «པེ་»ཏྲེ་མོ། ཧྲི་ཧྲི། «སྣར་བྲི་མོ། ཧྲིབྲྀ། «པེ་»«སྣར་དུབྦཾ། །(«པེ་»«སྣར་»སརྦ་དྷརྨཱ། «པེ་»«སྣར་»—ལྷའི། «པེ་»«སྣར་ཤུདྡྷསརྦ་དྷརྨ། «པེ་»«སྣར་»—ཨོ་ཨོ། ་«པེ་»«སྣར་»—སྭཱ་ཧཱ། པ«པེ་《སྣར་»ཤུདྡྷེ་སརྦ་དྷརྨ། «པེ་»«སྣར་»+པས༑(«པེ་»སྣར་»ག«པེ་»«སྣར་»+ཀྱི། (ཁྱེ«པེ་»«སྣར་»བརྐྱང་བས། ་«པེ་»«སྣར་བྱ། «པེ་»«སྣར་»བཏང་། <«པེ་»《སྣར་»ཀལེ«པེ་»«སྣར་»ཁཱ་ཧི། །«པེ་»«སྣར་»ཀྱིས། དེ་་«པེ་»«སྣར་»བྲིས། ཀ་ཁ་ག་མ་ག་ལ་བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (པེ་ཅིན། །དཔེ་གཞི། །སྡེ་དགེ། སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། པོད་རྟགས། ཚན་གྲངས། འ། ཀྱི་ལྡེབ། ན་བ་བ་ན་ནན་ན་—བ་ཕྲེང་། །།ཚེག་ཁྱིམ། ་ਖੁਝਕੋਲਮਦਰਾਸਨਲਨੂੰਇਕਓਨ།
