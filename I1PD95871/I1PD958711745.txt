བསྟན་འགྱུར། ་་རྒྱུད། བཞི་། ཆ། ཆེ་ཁྱཊ་་ཎ། །ཀ་ཀའོ་ཨ་ཏི་རི། ་་ཀཚོ། ཁ་ད་། འོ་ཁཚོ་ཆུ་ਸਾਊਥਨਾਲਈਸਾਨਵੀਰਾਨੂੰਨਸਨਨੇ་ན། ཀྱི། །ཡོལ་བསྡུར་མཆན། ཀ་ཏའི་ལངཁ། ཚེ་ཀ་ཀཱ་བབབབ། དབང་། དཔལ༌བ་། །༼་བྱན་དང་ལ་མང་་༽ར་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཡང་དག་ལྟ་་བའི་སྒྲོལ་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་《སྣར་»ཤོར། «པེ་《སྣར་བལྟ། («པེ་»«སྣར་རྟོག«པེ་»«སྣར་པར། །རང་འབྱུང་། 《སྣར་སྲུབས། «པེ་»འཁར་པ། 《སྣར་མཁར་པ། «པེ་《སྣར་»སོ་ནམ། «པེ་»«སྣར་བལྟ། །ਇਹੋਰਲਿੰਕਰਫ਼ੀਨੂੰਨਹੀਂདབང་ལ་ལ་་ང་། ཀའོ། །བྱ། །ཀ་ཎང་། དུ་བ། ཞལྟ་ཀ། བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (ਸੀਸਰਨ॥ཊ་ཋ་འཁཔེ་ཅིན། ཅོ་ནེ། སྡེ་ཚན། རྒྱུང། དཔེ་གཞི། སྡེ་དགེ། །སྣར་ཐང་། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། པོད་རྟགས། ཞི་ཙི་བ་ཙི་དོ་ཞི་ཚན་གྲངས། བར། བྱ་བ་དང་:ལྡེབ། བ—ན། —ན་བ—ན?བ—ན། ཕྲེང་། །ཚིག་རྐང་། །འོག་ལ་་། ཚེག་ཁྱིམ། །།བ། ་། པ་ལ་བདེ་བྱེ་བའི་དམ་པ་ཁོ་ན་ཡིན། །དངོས་ས་མ་འབག་པ་ཡི་མི་ལ་སྐྱེ་ནས། །ཡེ་ཤེས་མིག་གིས་མ་མེད་པ་ལ། །དྲི་མེད་མཐོང་ཞེས་ལྟ་བ་ཡིན། ལ། ད། བྱ་རུམ་ལ་བ་བ་བ་བགཏད་མེད། །།
