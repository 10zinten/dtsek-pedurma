བསྟན་འགྱུར། རྒྱུད། བཞི། ལ་ན་་ལ་། །་་་་་་་་་་་་་་། ངན་པས་བསྡུར་མཆན། །དུད་། ག་ང་?བློ་། །ཀ་བ་འདྲ་བ། །བ་ཀ་ཟླ། གང་ལ་ལ་་་་་ལ་་་གད་པ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། བསྒོམ་རིམ་དྲུག་པ་ཞེས་བྱ་བ་ཕྱག་ན་རྡོ་རྗེས་མཛད་པ་བཞུགས་སོ། །«སྣར་»༄༅། །བསྒོམ་རིམ་བཞུགས། འ«པེ་«སྣར་བྷ་ཝཱ་ན་ཀྲ་མ་ཥཊ། (ན«པེ་»«སྣར་ནི། «པེ་»《སྣར་བརྟགས། («པེ་»«སྣར་ཅོད། 《སྣར་རྣམས། ཟླ«པེ་»«སྣར་»རིགའ«པེ་»«སྣར་སྦྱང་། «པེ་»«སྣར་—དང་། «པེ་»«སྣར་གྲོལ། «པེ་»«སྣར་གྱིས། «པེ་»《སྣར་»ལོཙྪ་བ་ནགས། ད་ལྟ་བར་དེ་ལ་ཡོ་བ། །::ཀ་གལ་ཆེ་ལ་ཆུ་བྱཁ་བྱེ༌བ། ་་་ཏཱ་ཏྲ་ཀ་ਕਰਨਨੂੰਲਮੈਨੂੰਮਾਰਲੈ'ਵਾਲཔར་བྱ་དང་བསྡུར་འབྲས་རེའུ་མིག་དུར་་་་་་་ཀྱི་འགྲན་བསམ། རྒྱུད། （་། ག་ག་གལ་ཏེ། དག་ས་། ད་ཅིདཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །།སྣར་ཐང་། Sཅོ་ནེ། །༽སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །རྒྱུད། པོད་རྟགས། །ཞི་ཚན་གྲངས། ལྡེབ། ན་—བ་ནབས། བནན་བ་ཕྲེང༌། །ཚིག་རྐང༌། །བ། ཚིག་ཁྱིམ། ་་་། ལོ་ལུ་བ་དང་དབང་པོ་ལ་ཉེས་བྱེད་མང་དེད་ན་བོད་པ་དཔག་པར་བྱ། ་དང་བའི་དུས་ལས་ཤེས། པ།
