ནས་དེརའུ་ང་ཀོ་ཏུ་ཎསུ་ན། རྣལ་འབྱོར་མ་ཀང་ཀ་རའི་ཞལ་ནས། གཉིས་མེད་ཤེས་པ་ཆུ་ལ་ཆུ་བཞག་འདྲ། །རང་བཞིན་མ་བསླད་གླང་པོ་ལྟ་བུར་ཞོག། ཉམས་མྱོང་སྤྲེའུ་གང་་་ལྟར་བྱུང་ཡང་རུང༌། །།རྟོགས་པའི་ནམ་མཁའ་ཇི་ལྟར་ཁྱབ་པར་བྱ། །ཞེས་་གསུང་ངོ༌། །།།་ཁ་གས་ག། དུ། །།་ཁོ་ད་ཀཁག་ཏུ་ཁ་དཔལ་:ཨུ་རྒྱན་དུ་ཚོགས་འཁོར་བྱས་པའི་དུས་སུ་རྣལ་འབྱོར་པ་གྲུབ་པ་ཐོབ་པ་བཞི་བཅུས་རྡོ་རྗེའི་མགུར་བཞེངས་པ་ཉམས་ཀྱི་མན་ངག་ཐིག་ལེ་གསེར་གྱི་ཕྲེང་བ་ཞེས་བྱ་བ་རྫོགས་སོ། །་། །།།གཏཁཀོ་ཀ་ཁ་ག་ང་ਵਨੂੰਸਮਝਲਉਤੋਂREਪਾਅਇਧਰਖ਼ਲਸਨ।བབབཔདེ། ཀོ་བ་ཀུ་ལ། །འཁོརའདི་དག་ལ་བསྡུར་མཆན། བྱ། །ཀ་ཊ་དུ་ཁ་ཏོ་ལ་ཁ་ཕྱོགས་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»«སྣར་—༄༅། །«པེ་»《སྣར་»ཀྱིས། ཚེ་བ། /ལ་། རྐྱེན་མེད། །«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་མྱོང་། ，(«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་དམན། ས«པེ་»«སྣར་རས། «པེ་»«སྣར་ཐུག«པེ་»«སྣར་»ཏི། «པེ་»«སྣར་»ཏི། «པེ་《སྣར་བཏགས། «པེ་མེད་དོ། །ཞེས། 《སྣར་»མེད་དོ་ཞེས། «པེ་《སྣར་བརྒྱལ། «པེ་བསགས། མ«པེ་»«སྣར་+ཞེས། «པེ་»«སྣར་པོར། «པེ་»«སྣར་»+ཅེས། «པེ་»«སྣར་»བདེའི། «པེ་»«སྣར་»ར། ས«པེ་»«སྣར་»+ཞེས། «པེ་»«སྣར་བསྐལ་པ། «པེ་»«སྣར་ཏུ། «པེ་《སྣར་»+ཅེས། «པེ་»«སྣར་»+ཞེས། «པེ་བསྒོམས། །འ«པེ་»«སྣར་+ཞེས། «པེ་»«སྣར་»བརླབས། ས«པེ་»«སྣར་»+ཞེས། «པེ་»«སྣར་»ལགླི། «པེ་»«སྣར་»+ཞེས། «པེ་»《སྣར་»རེམ། ་ཡངཁྲི་ཞུ་བ་དང་«པེ་»«སྣར་»+ཞེས། «པེ་»«སྣར་»ར། «པེ་»«སྣར་བརླབས། «པེ་»«སྣར་རྟོགས། ས«པེ་»«སྣར་»བརྟེན། «པེ་»«སྣར་»ལྐུག«པེ་»«སྣར་+ཅེས། «པེ་»«སྣར་བརླབས། «པེ་»«སྣར་»+ཞེས། «པེ་»«སྣར་»འཐོབ། ཞེས་པ་ལ། འདུ་པ། —།
