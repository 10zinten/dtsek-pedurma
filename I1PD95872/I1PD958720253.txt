ང། ་་་་ཏ། མ་་ས་ག་ད་། དང་། དེ་རྗེས་དེད་ད་ཀ་ལན་ཏེ་བྲི། །ཞི་པ་དེ་ལྟ་ན་ཏེ་མེ་དུས་མེད། མིང་་་་་་་་་་པ་པ་ཡི་ར་ག་དུ་མ་མཁྲགས་སྔགས་རྫོགས་པ་ལ་བ། །དུ་མ་ལ། ཁྱོད་ཀྱི་ང་བདག་བདག་ལ་ལུ་ལུགས་པ། སྤེལ། ས་མེ་རུ་ཞུགས་བ་ལ་མི་ནུས་ཞུས་པ་བརྒྱད་ཉིན་རྒྱབ་ལ་སྦྱོར་གཉིས་པ་ནི། །ན་མོ་ལ་ད་དུ་ཕྱི་ལོ། ་དང། །ལྔ་བ་ནང་དུས་༄༅། །ཕྱི་ནང་གི་དུག་སེལ་གྱི་རིམ་པ་ཞེས་བྱ་བ། བ། །།༄༅། །རྒྱ་གར་སྐད་དུ། བི་ཥ་ཨནྡ་ར་བཱ་རྒྱ་ནི་པྲ་ཏི་བ་བྷ་ན་ཀྲ་མོ། བོད་སྐད་དུ། ཕྱི་ནང་གི་དུག་སེལ་གྱི་རིམ་པ་ཞེས་བྱ་བ། དཔལ་ཧེ་རུ་ཀ་ལ་་་ཕྱག་འཚལ་ལོ། །ཡིད་དང་མཐུན་པའི་སྒྲུབ་གནས་སུ། །ཕྱི་ནང་དུག་ནི་སེལ་་བ་ཡི། །གསང་བའི་རྣལ་འབྱོར་སྒྲུབ་པའི་ཐབས། །ཏི་ལོ་བདག་གིས་བཤད་པར་བྱ། །རྣལ་འབྱོར་བསྐྱེད་པའི་རིམ་པར་ལྡན། །རུ་ཙོག་སྦྲུལ་ལ་སོགས་པ་ཡི། །དུག་ཆེན་འཇོམས་པའི་ཨོ་ཨཱཧཱུཾ། །ལན་གསུམ་བརྗོད་ཅིང་འདི་ལྟར་བསམ། །རྩེ་སོར་ེབཅུ་གཉིས་ཚད་དུ་ནི། །སྤྱི་བོ་ཚངས་པའི་བུ་ག་ལ། །དབུས་སུ་ཨོ་བསམ་བདུད་རྩི་འབབ། །སྦྱོར་བ་དེ་ཡི་རྣལ་འབྱོར་ནི། །སྐད་ཅིག་གཅིག་ལ་་མདངས་ལྡན་ནོ། །དེ་ནི་ཕྱི་ཡི་དུག་སེལ་ཏོ་། :། ནང་གི་འདི་ལྟར་བསྒོམ་པར་བྱ༑། རྡོ་རྗེ་ཆེན་པོའི་སྙིང་གའི་དབུས། །སྐུ་མདོག་དཀར་པོ་བསིལ་བའི་འོད། །ཚེས་གཉིས་ཟླ་བར་སྤྲིན་ཆེན་པོ། །བསྒོམས་ཤིང་འཕྲོས་འགྱུར་རྣམ་པར་བསྡུ། །།
