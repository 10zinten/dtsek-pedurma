པས་དུག་གསུམ་རྟ་རྒོད་ཐུལ། །སློབ་དཔོན་ཀཾ་པ་ལས་མཛད་པ་ཞེན་མེད་ཀྱི་ལྟ་བ་ཞེས་བྱ་བ་རྫོགས་སོ། །སློབ་། བར་པ་ལ་ད་ལོ་ང་ལ་མ་དང་། འདོད་བློ་ཡག་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། ཞེན་མེད་ཀྱི་ལྟ་བ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་སམྦྷནྡྷ། «པེ་《སྣར་དཀྱུས་ས། «པེ་»«སྣར་»གིས། སུ«པེ་»《སྣར་རྒྱུགས། «པེ་»«སྣར་གཏད། «པེ་»«སྣར་རྒྱུགས། «པེ་»བསོགས་གཅིག«སྣར་བསོས་གཅིག«པེ་《སྣར་རྒྱུགས། («པེ་»《སྣར་རྒྱུགས། ་་་་་ཚོ་ཁ་:་། ཙཡ་ཀ་གདེ་་་་་་་་ཚོགས་ਨਾਲਉਸབོད་ཀྱི་ཁ་ན་བ་། བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (༄༅། །ཀའ་འདཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། རྒྱུད་འགྲེལ། །སྣར་ཐང་། རྒྱུད། །།སྡེ་ཚན། རྒྱུད། རྒྱུད། པོད་རྟགས། ཟིཚིཚན་གྲངས། །ནབ་། ན་བ་། བ—ནནབ?། ལྡེབ། ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ་ཡོད། ་་་།
