འདི་ལྟའི་དཀྱིལ་འཁོར་གླུ་རུ་བླངས། །གར་བྱེད་དཔལ་ནི་ཨ་ཙིན་ཐོད་པ་ཅན། །ནམ་མཁར་རྔའུ་ཆུང་ས་ལ་དྲེའུ་༔ཆུང་བརྡུང་། །མུ་སྟེགས་ཀྱི་གཏུམ་མོ་ཞེས་བྱ་བ་སློབ་དཔོན་ཨ་ཙིནྟས་མཛད་པ་རྫོགས་སོ། །།།ན་ཉེ་ཡོད་ཀྱིན་ཡོད་དེ་དང་མཉམ། །ས་ཏྲེ་སང་ཉིན་ད་པ་བྱེད་དང་ཡིན། །ལ་ལ་ན་བསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། མུ་སྟེགས་ཀྱི་གཏུམ་མོ་ཞེས་བྱ་བ་བཞུགས་སོ། །།«པེ་»«སྣར་ཙཎྜ་ལི། «པེ་»མོ། ་«པེ་»«སྣར་ལྕེས། །«པེ་»«སྣར་»ཙན། «པེ་»«སྣར་རྗེ། «པེ་《སྣར་བཞུགས། དུ«པེ་»«སྣར་རྗེ། «པེ་»«སྣར་རྩེན། «པེ་»«སྣར་རེ། ་་་་(«པེ་《སྣར་ཙེན། །(པེ་»«སྣར་»རྗེ། ས་བྱ། །ཨ་པ། ངག་ལ་བ་ཀྱི་ནང་དུ་འོ་མ་ལ་ཀ་་་་་། །།ཡུག། ཚེ་་་བསྡུར་འབྲས་རེའུ་མིག་་་:ཞུགས་བློ་གྲོས་ཏེ་ཀ་། བ། ང་དང་ན་བརྒྱུད། (ང་ཁ་འདོ། དཔེ་གཞི། །སྡེ་དགེ། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། རྒྱུད། གྲོལ་པོད་རྟགས། པེ་ཅིན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུ་ཚན་གྲངས། الدیادیལྡེབ། །བན་བ་ན། བ་—ན་། བ—ནja::དུས་ཚིག་རྐང༌། ཚེག་ཁྱིམ། །།
