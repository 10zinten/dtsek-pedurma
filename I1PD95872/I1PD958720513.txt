རྒྱུད། ཝིཏི་་། ཀ་ལ་ཁྱོད། པའི་ངང༌། །བསྒོམ་བློ་འདུན་པ་དྲི་མ་ཡིན། །ཞེས་གསུང་ངོ་། །ད་ཎཊ། མོས་གུས་ཀྱིས་རྒྱུད་སྨིན་པ་ཨཱརྱ་སིངྒའི་ཞལ་སྔ་ནས། ཆོས་རྣམས་མ་་་་ལུས་རྣམ་ཀུན་ནི། །རྗོད་བྱེད་ཚིག་དང་བརྗོད་བྱའི་དོན། །རྣམ་པ་གཉིས་ཀྱིས་སྟོང་:འགྱུར་བས་ཅི། །གཞི་རྩ་མེད་པའི་བདག་ཉིད་ཅན། །ཞེས་གསུང་ངོ༌། །བཟོད་པའི་སྟོབས་ལྡན་མ་ན་ཏ་རའི་ཞལ་སྔ་ནས། །ཚོགས་དྲུག་ཡུལ་གྱི་ས་སྣང་བ་ནི། །སྣ་ཚོགས་རྐྱེན་གྱི་སྣང་བ་སྟེ། །རྐྱེན་སྣང་སྒྱུ་མ་ལས་མ་འདས། །རང་བཞིན་མེད་དེ་ངོ་མཚར་ཡུལ། །ཞེས་གསུང་ངོ༌། །།འདོ་བ་རྣམས། དངོས་རྣམས་བདེན་མེད་དུ་གཟིགས་པ་རཏྣ་བཛྲའི་ཞལ་སྔ་ནས། ཇི་སྲིད་འབྲས་བུ་འདོད་པའི་བློ། །མ་གྲོལ་མྱ་ངན་ལས་མི་འདའ། །གང་གི་མཐར་་་་ཐུག་ངོས་ཟིན་པའི། །བློ་དེ་ཉིད་ནི་རྣམ་རྟོག་གོ། །ཞེས་གསུང་ངོ༌། ཁཚོ། ཡང་དཔལ་གྱི་འོད་ཟེར་ཤིངྒཱ་ནའི་ཞལ་སྔ་ནས། མཆོག་ཏུ་འཛིན་པའི་དྲི་མ་དེ༑། ལམ་གྱི་བདུད་དུ་ཤེས་པར་བྱ། །རེ་བ་མེད་པའི་སྙིང་དཀྱིལ་ན། །བླ་མེད་ནོར་བུ་རིན་ཆེན་གནས། །ཞེས་གསུང་ངོ༌། །།།།།ན་ཚོགས་ཀྱི་་གཉེན་པོའི་འཕྲང་ཆོད་པ་ཛ་ར་ནའི་ཞལ་སྔ་ནས། །དཔའ་བོ་བློ་གྲོས་ལྡན་པ་ཡིས། །ཆོས་ཀྱི་རང་བཞིན་ཞི་བ་རུ། །ཤེས་ཚེ་རིང་དུ་མི་ཐོགས་པར། །སྲིད་པའི་འཁོར་ལོ་འཇིག་པར་བྱེད། །ཅེས་གསུང་ངོ༌། །གྲགས་པ། །གྲོང་པ། བདེ་བའི་མདངས་ལྡན་མ་ཤྲཱི་ཀ་སུ་ཏའི་ཞལ་སྔ་ནས། འདས་དང་མ་་་་་འོངས་ད་ལྟར་གྱི། །ཐུབ་པའི་གསུང་གི་སྡེ་སྣོད་རྣམས། །རྣམ་གྲངས་ཐ་དད་་དོན་གཅིག་པར། །བླ་མའི་བརྡ་ལས་རབ་ཏུ་:བླང༌། །ཞེས་གསུང་ངོ་། །།འགོ་ཡུལ་རྗེས་ལ་:འཁྱེལ་བ་ེར་ཏྲི་ཏའི་ཞལ་སྔ་ནས། ཡིད་ལ་མ་བྱས་་་་་དངས་མའི་ངང་། །།གང་ཡང་མ་སྐྱེས་གཉུག་མའི་གཤིས། །མ་བཅོས་རྣལ་མ་་།
