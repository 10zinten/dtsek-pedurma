རང་བཞིན་གྲུབ་པ་ཞེས་བྱ་བ། ག་ཉིད་དུ་རྗེ་ཀུན་དུ་ཁྱབ། །དེ་བ་ན་ས་སེམས་ཉིད་ཡིན། དེ་དག་མ་དུ་བྱ་བ། །ཐོང་། །འདིར། བསྡར་མཆན། ，ན་པ་བར་ཀང་དེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་འདི། དེ་དག་ལ་«པེ་»སྣར་ངོ༌། །ཀྱི«པེ་《སྣར་《། «པེ་»«སྣར་གྱིས། «པེ《སྣར་»དྲངས། །བར་གྱི་་་པ་ལ། རང་ལ་བྱ་དཉིད་རྟོགས། །མེ་དེ་བདག་དེ་ལ་ཡང་དེ། །རྣལ་འབྱོར་བསྡུར་འབྲས་རེའུ་མིག་ནང་འགོ་::མས་པ་སློབ་དཔོན་མེ་དེ་རྒྱུད། (बारཟིདཔེ་གཞི། སྡེ་དགེ། ང་པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། མ་རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། ན་རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བ་ན། བནད། བ་_ན་བ་_ན༴༼ཁ༽ཕྲེང་། _ཨཀ། ཚིག་རྐང༌། །།ཚེག་ཁྱིམ། །༼ང་ཀྱི་ནང་་། ཊ་། བབབབ། ད། འོལ་ཀ་ཁ་ཁ་བ་ཁ་ཆཆ་ཁ་ལ། །ཀ་མ་ལ། །ཀུཊ་ཁ་ག་ང་ཀློག(ཎྜ་gਸਲ'ਨਇਸਸਾਲਉਸੌਣ,ਸੌਣਸਰੋਂਸਤਰਸਉਨਉਨਮਕਸੌਣ། འི་ཊི་ཀ་ཏ་ཁ་ཁ་ཁ་བ་ཆ། ཀོ་ལོ་ཀྱ་མ་། ཏཥཱ་ཀཱ་མ་ལ་།
