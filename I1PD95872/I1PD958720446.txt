རྒྱས་འདེབས་རྡོ་རྗེ་འཛིན། །བྱིན་བརླབས་ཤར་བས་དབྱེར་མེད་རྫོགས་སངས་་་རྒྱས༑། གསུང་ངོ་། །།ན་དང། ::སློབ་དཔོན་བྷ་ཏི་བའི་ཞལ་ནས། ཆོས་རྣམས་ཀུན་ཀྱང་རང་གི་སེམས་དང་གཉིས་མེད་སྦྱོར། །རྟོགས་པའི་ཉམས་མྱོང་ངེས་དོན་གདམས་པའི་དོན་བྲལ་ན༑། ོ། ནོར་བུ་མཆིང་བུ་དག་ལ་བདེ་ཆེན་མིན། །ཆོས་དེ་རྣམས་དང་ལྡན་ན་རེ་དོགས་མེད་པར་རང་སར་ཞོག། གསུང་ངོ༌། །།།།།།སློབ་དཔོན་ལུས་མེད་རྡོ་རྗེའི་ཞལ་ནས། སེམས་མ་གདུངས་ན་བདེ་བ་ཆེ། །།ཞེ་འདོད་མེད་ན་རོ་སྙོམས་ཡིན། །འབད་དང་བྲལ་ན་བསྒོམ་པའི་མཆོག། །།གཉིས་སྣང་བྲལ་ན་རྫོགས་སངས་རྒྱས། །གསུང་ངོ༌། །ལ་ལ། ཀ་ཁའི་ཁ་ཟར་རྒྱལ་པོ་ཨིནྡྲ་བོ་དྷིའི་ཞལ་ནས། དུས་ཀྱི་སྣ་རྩེ་བདག་བྱིན་བརླབ་དང་མ་ལྡན་ན། །།།ཕྱི་དང་ནང་དུ་སྤྱོད་པ་བྱས་ཀྱང་སངས་མི་རྒྱ། །བདེ་བ་རང་ཆས་་་སྣང་བ་དག་དང་ཁྱད་པར་མེད། །ཞེན་པ་བཏང་བས་ཉམས་སུ་མྱོང་བ་མཆོག་་་ཐོབ་འགྱུར། །གསུང་ངོ་། །་་ཚུ་བྱིས་དུས། །པ་སློབ་དཔོན་པདྨ་བཛྲའི་ཞལ་ནས། གར་ཡང་མ་བལྟས་དབུས་སུ་ལྟོས། །མཆོག་དང་བྲལ་ན་དགའ་བ་ཆེ། །སྔགས་རིག་གསང་བ་ལྟོས་སྤྱོད་ན། །རྣལ་་འབྱོར་རྟགས་ཀྱིས་བསྒྱུར་བར་བྱ། །གསུང་ངོ༌། ་་་་། །གང་བ་བྱ་བ་བྱེད་པ་ལོ་། :ལྷ་ལྕམ་ལགྣི་ཀ་རའི་ཞལ་ནས། །དང་པོ་བློ་ལྡན་རྟོགས་པ་བསྐྱེད། །ཟ་། གཉིས་པ་ནགས་སུ་སྣ་རྩེ་བསྒོམ། །ཉམས་སུ་མྱོང་བ་སྐྱེས་པ་ན། །སྤྱོད་པའི་ཁྱད་པར་རྟོགས་པར་བྱ། །གསུང་ངོ༌། །དེ་། ང་ཚོ་ཚེ་:ཀན་ས་དར་བ། :ཀྱིསློབ་དཔོན་པདྨོ་ཞབས་ཀྱི་ཞལ་ནས། གང་གང་ལྷན་ཅིག་སྐྱེས་པ་དེ། །གླང་པོ་ཆེ་ཡི་མིག་བཞིན་དུ། ་། །གཟེ་རེག་སྐྱེས་ན་ཟུག་རྔུར་འགྱུར། །།།གློད་ན་་།
