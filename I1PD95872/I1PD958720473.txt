འ་བསྟན་འགྱུར། རྒྱུད། ཝིཏིཀྱི་། ཀ་ང་ད་ན་བ་མ་འགྲོ་བ་དང་། ལས་ཁ་ཆ་བ། བྱ་མ་ལ་ཡི་ཆད་ཀྱི་ཁྲོ་བ་དང་ਦਾਸੰਸਕਾਰਉਸਮਾਨਸ਼ਮਸਰੂਪབ་ཡི་ཁ་ཊ་ལ་བསྡུར་མཆན། ལྟ་བ། བྱ་བ་མ་པ་ཤང་ཀའི་གཡོདེ་འབད་དང་། །།ས་ག། (མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཕྱག་རྒྱ་ཆེན་པོ་་རིན་པོ་ཆེ་དབང་གི་འཁོར་ལོ་ཞེས་བྱ་བ་དུར་ཁྲོད་རྒྱུ་སྐར་དཔལ་གྱི་རོལ་པས་བསྟན་པ་བཞུགས་སོ། །«པེ་《སྣར་ད། འ«པེ་»《སྣར་དཀའ། ཚེ«པེ་»«སྣར་འབབས། མ«པེ་《སྣར་»འཕྲ། པོ«ཅོ་»ན༑@«པེ་《སྣར་ཆེ། «པེ་»«སྣར་བརྩོད། «པེ་《སྣར་མི། «པེ་»«སྣར་བརྒྱུད། «པེ་»«སྣར་»སོགས། «པེ་»«སྣར་»འཐེབ་མཛུབ་འཐེབ། «པེ་»«སྣར་འཛུབ། །«པེ་»«སྣར་བཤད། «པེ་《སྣར་»སྙན། «པེ་《སྣར་གྲལ། (Q«པེ་《སྣར་»གྲོས། «པེ་»«སྣར་མཉན། ེ«པེ་»«སྣར་འབྱུང་། དང་མར་མ་དང་་«པེ་»«སྣར་བརླབས། «པེ་»«སྣར་དབྱེར། «པེ་»«སྣར་བསྲུང་། «པེ་«སྣར་»ལ་བབས། «སྣར་ཧ། །@«པེ་»«སྣར་དབུལ་མོའི། ༄༅«པེ་»«སྣར་གོན། ལང་ཏཁྲི་ལ(《སྣར་ཅེས། «པེ་»«སྣར་ཁུར། «པེ་»«སྣར་»གདད། «པེ་»«སྣར་ཀྱིས། མ«པེ་»«སྣར་བཟུང་། «པེ་《སྣར་»བཟང་། @«པེ་»«སྣར་བརྩོག་པའི་ཆོས་ཟང༌། «པེ་»《སྣར་མི། ང་:ས་ག་ལ་ཡིག་དེ་ལ་བག་མི་པལནིཏེ་«པེ་»《སྣར་བརྟེན། «པེ་»«སྣར་བྱུང༌། ：«སྣར་»—པའི། ཚེ«པེ་»«སྣར་ཟབ། «པེ་»《སྣར་བསྐོང་། «པེ་《སྣར་གདུགས། །་«པེ་»«སྣར་ཟླ་བརྟེན། 《«པེ་《སྣར་། ང་ཡོདཁོར་འགས་ਕਹੁਨਾਲਅਸ਼ਲੀਲབ་འདུ་བ་་་་་། ལ་ག་དང་ལམ་བདེ་མཚོན། པ་པད་མ་འབྱུང་དུ་སོང་བ་ངོ་མ་ཚེ། བ་ཕྱིན་པ་ལ་བྱེད་དུས་ཁྱོད་ག་ན་པོ་ཆེ། འབའ་པ་་་་་་་་་་་་་་་་འོ་དེ་འདྲ་མང་བ་ན། གདུགས་དེ་བས་འདོད་པའི་བ་འགྲོ། པ་བྲག་ལམ་ག་ར་།
