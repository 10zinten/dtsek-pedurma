རྒྱུད། ༼ཝིབདག་མེད་ཆོས་ཀྱང་རང་བཞིན་ལ། །ཐོག་མཐའི་རང་བཞིན་དེ་ཉིད་དོ། །བདེ་གཤེགས་ལྟ་བ་དེ་ཉིད་དེ། །སྟོང་ཉིད་དོན་ཡང་ཤེས་པར་བྱ། །དངོས་མེད་དྲན་་་་མེད་བློ་འདས་སྨྲ། །དཔེར་ན་ཉི་མ་ཟླ་བ་དང་། །ནམ་མཁའ་སྐར་མའི་འཁོར་ལོ་་་དང་། །ས་དང་ཆུ་དང་རྒྱ་མཚོ་དང་། །ཕྱོགས་དང་མཚམས་དང་དེ་ཡིན་ལ། །རི་རབ་ལ་སོགས་འཁྲུལ་སྣང་ཡིན། །ཨིནྡྲ་:བྷཱུ་ཏི་སྨྲ་བར་བྱེད། །དེ་རྣམས་གང་ཡང་གྲུབ་པ་མེད། །གཉིས་མེད་དོན་ལ་ཉམས་ལེན་གྱིས། །དེ་ཉིད་བརྒྱད་ཀྱི་ལྟ་བ་ཞེས་བྱ་བ་སློབ་དཔོན་ཨིནྡྲ་:བྷཱུ་ཏིས་མཛད་པ་རྫོགས་སོ། །།།།བསྡུར་མཆན། ས་བརམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་བརྟེན། «པེ་»《སྣར་»བོ་དྷི། «པེ་»རྟོགས། འ«པེ་»«སྣར་»བོ་དྷིས། །ཡང་ན་། དབྱར་«པེ་»«སྣར་བོ་དྷི། «པེ་»«སྣར་»བོ་དྷིས། ਪਲਊਸਨੂੰਸੁਨਣਲੜngyA"ਉਸཁ་ན་བ་ཏི་ར་བ། ཤིང་བས་བདེའི་ཀཀོ་ས་ཀཡ་ཀྱང་། ད་བསྡུར་འབྲས་རེའུ་མིགਨੂੰਮਨਵਸਾརྒྱུད། (::གོ་པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། ངབགས་རྒྱུད་འགྲེལ། རྒྱུད། དཔེ་གཞི། སྡེ་དགེ། །སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནབ་ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ་་། བྱལོར་གྱི། །བ་—ན། ནབའི་བ་ནཀྱི་ཀ། ༼—།
