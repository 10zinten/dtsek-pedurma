སྙིང་རྗེ་ཌ་མ་རུ། །སྒྲ་སྒྲོགས་པ་ནི་ངོ་མཚར་ཆེ། །ཨཱརྱ་དེ་བ་དམིགས་མེད་པའི། །རྒྱལ་པོའི་སྲིད་ལ་ངོམས་པ་མེད། །ཨཱརྱ་དེ་བ་ཐམས་ཅད་སྤྱད། །སྤྱད་པས་རང་་་་བཞིན་དམིགས་མེད་འགྱུར། །ཀྱེ་མ་སྙིང་རྗེ་ཌ་མ་རུ། །སྒྲ་སྒྲོགས་པ་ནི་ངོ་མཚར་ཆེ། །ཨཱརྱ་དེ་བ་དམིགས་མེད་པའི། །རྒྱལ་པོའི་སྲིད་ལ་ངོམས་པ་མེད། །སྟོང་ཉིད་སྙིང་་རྗེའི་ལྟ་བ་ཞེས་བྱ་བ་སློབ་དཔོན་:ཆེན་པོ་ཨཱརྱ་དེ་བ་ནས་བརྒྱུད་པ་སློབ་དཔོན་ཙ་ཏ་པས་མཛད་པ་རྫོགས་སོ། །ཏི་། ་ཁྱཊ། བར་བྱེད་པ་གཅིག་ཀྱང་གང་ཡིནཔ། བར་བ་དང་ཉེན་བ། སུམ་ན་གཅིག་པུ་ཡིན་པ་དེ་དང་། བག་མི་ཤེས་::ཡུལ་དྲ་བས་བཅིངས། བསྡུར་མཆན། ལོ་བདེན་པ་གསུམ་ན་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ཀ། ་«པེ་《སྣར་—ཆེན་པོ། །ར་ཏིངང་ཀ་ཀྱི། །ਸਮਰਾApnghKਨੂੰਦਾਣਵਾ॥ਏਨਨ'ਮਨਨੂੰཀ་ཊ་མ། །ཅཎཱ་བསྡུར་འབྲས་རེའུ་མིག་ཁར་ག་ཁ་ཁ་ཌཚོཊ་ཁཛིད་་ཎཚ་ཁོ་པ། ཀླ་ཚོ་རྒྱུད། (ལོག་པའ/ང་ཁ་ཆེ་ཀ་ཏུ་ཡང་ཅོ་ནེ། རྒྱུད་འགྲེལ། རྒྱུད། ཟིདཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། །སྣར་ཐང་། །སྡེ་ཚན། རྒྱུད། ཊརྒྱུད། པའི་པོད་རྟགས། ལྔ། ཟི། འ་ནི་ཚི་ཀ་ཀ་ཀཱ་། ཚི་ལང་པ། །ཟི་ར་ས་ལ་ཚན་གྲངས། ༽ལྡེབ། །བ་_བ། བ་ན། ན—ན། བ། _བ། ཕྲེང་། ལོ°དེཀྱིཚིག་རྐང༌། ཡཱ་་ཡི་︽ལཚེག་ཁྱིམ། །་དང༌པས་གཏམ། —།
