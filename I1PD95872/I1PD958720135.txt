དོ་རཀོ་བ་། །།།ཁ་བ་་༌་་། ཀ་ཏཁ། ཁོ་ཎ་ཌཏུག་ਸਨੂੰਸਲਾਥਸ਼।pਠਤੋਂਸਲੋਂਵਾਨਰ'ਸ਼ਰ་་་༌་་་་་ཁོ་རེ། བསྡུར་མཆན། །ཀ་ཆང་ཚོ། །།ལ་ར་་ཀ་ལན་ད་དང་། །ང་དs@་ཀལ་པསཔསབསལ་དྲོ་ཚོང་ང༽༼ཁ༽། p། །ལzད་བགས། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཐིག་ལེ་བཅུ་དྲུག་་གི་ཉམས་ལེན་ཞེས་བྱ་བ་བཞུགས། «པེ་»«སྣར་བྷ་བཱ། «པེ་བཅུ། «པེ་»«སྣར་པདྨ། ཀྱང་། «པེ་»«ཅོ་རིག«སྣར་རིགས། བློ་བག་ག་ཚཛིན་ཀན་ཡི་ཀ་ཁ་«པེ་《སྣར་བསྲེག«པེ་»«སྣར་»ཀྱི། ན་མི་ཚེ་། ཀའོ་ཀཀ་ཀར་བ་དང་། ཚེ་རོགਸਗੋਂਸ਼ਲਾਨਨਾਲੋਂਸਲਾਮਹਨ། །ངོ་Sང་ཁྱ:;ཏུ་བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (སྡེ་དགེ། །རྒྱུད། ?། ཟིལྡེབ། །པེ་ཅིན། །སྣར་ཐང་། །སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །།ལོ་ལུ་བབ་བ་བ—ནནབ། བ—བའི་ཕྲེང༌། །།།།ཚིག་རྐང་། ཚེག་ཁྱིམ། །ད་བ་བ་ན་མ་ལ་ཡོད་པ་རབ་ཏུ་འབར། །བརྒྱད་ཟིལ་བན་བདེ་ལ་དབང་བ་བདེ། །བ་ཉིས་ཀྱི་ཕྱེད་ནི་བོད་པ་མེད་མ། །བྲལ་མེད་ད་ཕྲ་བས་ཡིན་ནོག་ཆུ་ལ་བཙུགས་ནས་མ་ཡི་བ། ར། །རེག་ད་དེ་ལ་གཡུང་དང་། ཧཡི་བར། །ཡི་ལེབ་མིང་དུ་—།
