བསྟན་འགྱུར། རྒྱུད། ཝིབརྙན་མེད། །མཁའ་ལ་བྱ་འཕུར་རྟེན་ས་མེད། ་། གསུངས་སོ། །ཁྱབ། ཞེ་མཁའ་འགྲོ་མ་བརྡའི་མགུར་ཆུང་གསང་མཛོད་འཕྲུལ་གྱི་རྒྱ་ཁྱད་པར་དུ་་་་གྱུར་པའི་བརྒྱུད་པ་ཞ་མ་ལོ་ཙཱ་བས་བསྒྱུར་བའོ། །།།ཅེཀ་སྟོང་། བའ་ཞིག་ਮਰਨਸਲਾਬਤਾਬਸਮੂਹਵਿਲਸਲਾਉਸਸਮਰਥਕཔ། ཀོ་ས་ལ། བྱ་བ་ཤིན་ཏུ་གཟ། ང་་་་ན་བསྡུར་མཆན། རྒྱལ་nཚེ་ཆ་ཁོང་། །བྱཏམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། དཔལ་རྡོ་རྗེ་མཁའ་འགྲོ་མའི་མགུར་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»ཅིག«པེ་»«སྣར་ལ༑སོ«པེ་»«སྣར་»གཞེངས། @«པེ་»«སྣར་»—པོ། «པེ་»«སྣར་སྒྲོགས་ལས་བྲོས། «པེ་»«སྣར་»སྒྲ། ་«པེ་»«སྣར་སྐྱོན། «པེ་»«སྣར་གསུངས། །«པེ་»«སྣར་སྒྲུབ། །®«པེ་《སྣར་གསུང་ངོ༌། «པེ་»སྣར་བཀའ། @«པེ་»«སྣར་མིས། «པེ་《སྣར་བཤིན་ལ། «པེ་»《སྣར་བརྟན། «པེ་»«སྣར་གསུང་། །ཁྱེ«པེ་»«སྣར་བྲན། («པེ་»སྣར་བརྙད། སྟེ《པེ་ཚེ། 《སྣར་འཚེ། «པེ་ཟན་གསོ་གསུངས། །«སྣར་ཟན་གསོ་གསུང་། །(«པེ་»«སྣར་ལུགས་སུ། /Q«པེ་»«སྣར་བཟུང་། («པེ་»«སྣར་བཅིང་། &«པེ་»«སྣར་གསུང་། །ས«པེ་བཟུང༌། «པེ་»«སྣར་རྨང་ནས་སྒུལ། «པེ་»«སྣར་»ལྡོག«པེ་»《སྣར་+ཞེས། «པེ་»«སྣར་ཤ། /«པེ་»«སྣར་འགྲིབས། «པེ་»«སྣར་»ཀྱི། «པེ་»གསུངས། །«སྣར་»གསུང་། ེ«པེ་»«སྣར་ཨ། ང་འ་སྟེ་འོང་ངོ། («པེ་»སྣར་»མཁའ། «པེ་»«སྣར་གསུང༌། །«པེ་»«སྣར་»གི། ®«པེ་»«སྣར་»+ཅེས། «པེ་»《སྣར་»། «པེ་མེད། «སྣར་མེན། «པེ་»«སྣར་གཅེར་གྱིས་བལྟས་པ། «པེ་»::ཀྱིབདེདེ«སྣར་རྩེའི། «པེ་»«སྣར་»བླག(དགུ«པེ་»«སྣར་»ལྐུག«པེ་»«སྣར་ཞེས་གསུང་ངོ༌། །«པེ་ད་པ《སྣར་»ཀྱི། «པེ་»སྣར་ཡིས། «པེ་《སྣར་བུ། «པེ་»«སྣར་བརླབས། །ཡེ«པེ་《སྣར་»བལྟ། །་«པེ་»«སྣར་ཞེས་གསུང་ངོ་། །«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་ཞེས་གསུང་ངོ༌། །ང«པེ་»སྣར་མ། «པེ་»«སྣར་»སྐྱལ། @«པེ་»«སྣར་བསྡིགས། ི«པེ་»«སྣར་»གསུང་ངོ༌། །«པེ་《སྣར་»མཁའི། «པེ་»«སྣར་»འགྱུར། འགྱེ«པེ་»«སྣར་ཅེས་གསུང་ངོ༌། །«པེ་»མཁའི་བཙད། «སྣར་མཁའ་བཙད། «པེ་»«སྣར་»འཕུང་། «པེ་»«སྣར་བཅིང་བ། —.།
