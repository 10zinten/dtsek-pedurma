བསྟན་འགྱུར། རྒྱུད། ཟིརིམ་པ་ནི། །སྐལ་ལྡན་རྣམས་ཀྱི་དོན་ཕྱིར་བརྩམས། །དེ་ཡིས་དགེ་བ་གང་ཐོབ་པ། །འགྲོ་བ་བྱང་ཆུབ་ཐོབ་ཕྱིར་བསྔོ། །ཎ་། ལྷ་ཀུན་གྱི་རྫོགས་པའི་རིམ་་པ་གྷ་དྷ་རིས་མཛད་པ་རྫོགས་སོ། །་། །།pཁ་ཟླ་ཁ་ལ་ངོ་། ཚེ་ག་ག་་བོ་དང༌། ཟླ་བ་་ལ་དང་། །བ་ཚོར་བར་་མབ་མར་གདམར་དང་། བརྒྱ་དང་། །ཀ་ལ། །།པས༌དང་། ད་ལ་གོད་བྱེད་བསྡུར་མཆན། ་་་པའི་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། ལྷ་ཀུན་གྱི་རྫོགས་་་པའི་ལམ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»གསང་། འོ། །ཎ་། །སྤྱང་ཀ་བལ་(«པེ་»《སྣར་བསགས། «པེ་»«སྣར་བསྐྲད། «པེ་»«སྣར་»སྙིང་། «པེ་»《སྣར་སྒོམ། ས«པེ་»«སྣར་བརྟན། «པེ་དཀན་གྱི། 《སྣར་དཀན་གྱིས། «པེ་»«སྣར་འབབ་བསམ། །«པེ་ཡི། །།།་་བ་་་:། ཁ་དེ་བས་བསག་པའི་བསྡར་འབྲས་རེའུ་མིགཀྱིརྒྱུརྒྱུད་བ་བྱ་རྒྱུད། (བསྒྱུར་འབས་རེ་མིག༌ལ་སྤྲོ་བབ་ཏཎལ་དབང་བའི་ལོ་ལ། །ཀེ་ལ་རྒྱུ་དེ་དོབདས་པས་སྣར་ཐང༌། སྡེ་དགེ། རྒྱུད། །།པེ་ཅིན། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། དཔེ་གཞི། །སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། པsལྡེབ། །ན—བ། བ་བ། ནནནབ་ཕྲེང༌། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ནང། །མི་བདག་གང་ན་གནས་པ་དང་། བྱ་ཚེ། བོད་།
