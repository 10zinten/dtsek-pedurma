རྒྱུད། ཝིཀྱི་་བརྗོད་པ། ཀྱེ་ཧོ་བུ་རྣམས་སེང་གེའི་འོ་མ་འཐུང༌། །ཁྲི་ཀ་པ། ཕན་པའི་སྨན་མར་ཕྱི་རུ་མ་སྐྱུག་ཅིག། །སྙིང་གི་དབུས་སུ་ཉི་མ་ཤར་ནས་སུ། །།།མུན་པའི་ཚོགས་རྣམས་རབ་ཏུ་འཇོམས་པར་འགྱུར། །།གསུང་ངོ༌། །།།།འཁོ༌་དབང་། ན་ཕྱག་རྒྱ་ཆེན་པོ་བརྡའི་མན་ངག་ལས་ལས་རྣམ་པར་གྲོལ་བའི་དུམ་བུ་མཁའ་འགྲོ་མ་དཔལ་ལྡན་པདྨའི་སྤྱན་གྱིས་བསྡུས་པ་སྟེ་དུམ་བུ་དྲུག་པའོ། །ཁ་བོ། །བ་ཞེ་ལེ་ཕྲན་ལྔ་བཅུ་རྩ་ལྔ་པ་ཕྱག་རྒྱ་རིན་པོ་ཆེ་བརྡའི་མན་ངག་རྗེ་དམ་པ་ལ་་་མཁའ་འགྲོ་མས་གཏད་དེ་ཞ་མ་ལོ་ཙཱ་བས་བསྒྱུར་བའོ། །ཞ་། །གུར་གྱི་ਮਾਸਤੋਂਖੋਮਲਿੰਕਬਿਲਾਇੰਸਪਨੇਸਰਨਵਾਬਕਿਉਂਝ॥། བ། ལན་ཆ་བྱ་བ། །ག་ཏེ་བོད་ཀྱི་འཚོ་བ། ལ་གཏད་དོ། །བསྡུར་མཆན། གེ་རྒྱལ་ཚབ་ཏུ་དགའ་བ་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །ཕྱག་རྒྱ་ཆེན་པོ་རིན་ཆེན་བརྡའི་མན་ངག་ཅེས་བྱ་བ་བཞུགས་སོ། །«པེ་»《སྣར་ད། «པེ་»«སྣར་»ཚན་་་་དོའི། ®«པེ་»«སྣར་»དཾ་བ། །«པེ་»«སྣར་ལས། @«པེ་»«སྣར་ཡའི། «པེ་»«སྣར་ལ། «པེ་《སྣར་རྫིངས། པེ་《སྣར་སྡང་། «པེ་»《སྣར་ཐར། «པེ་»«སྣར་»བཅད། «པེ་»«སྣར་»སྒྲུབས། ང«པེ་»«སྣར་ཕུར། «ཅོ་ཕུབ། ་ ོ«པེ་»«སྣར་པོ། @«པེ་»«སྣར་ཁྱག(ན«པེ་»«སྣར་དྷ། «པེ་»«སྣར་དྲོད་པར་རྟོལ། («པེ་»«སྣར་»ཤཱ། ་ཉེ«པེ་»«སྣར་»བརྟེན། འུ«པེ་»«སྣར་གཟག ེ«པེ་»«སྣར་»ནས། །«པེ་»«སྣར་མཚོན། །«པེ་»«སྣར་མའི། «པེ་»«སྣར་»སྦྲབ། «པེ་»«སྣར་ལྕེ། ས«པེ་»«སྣར་རྩེར། «པེ་»«སྣར་ཅིག་གིས། «ཙོ་»གཅིག་གི། «པེ་»«སྣར་ཀུན་དྷ། «པེ་»«སྣར་»མདོངས། «སྣར་རྟོགས། ཁྱེ«པེ་»«སྣར་»སྟོར། ་འཀྱེ«པེ་»སྣར་འཇོལ། «པེ་»«སྣར་བལྟ། ་«པེ་»«སྣར་»སྟོངས། ཁྱེ«པེ་»«སྣར་སོ། །དེ་ལས་«པེ་»«སྣར་དུམ། «པེ་»«སྣར་འཕོང་། འ«པེ་»«སྣར་ལ། «པེ་»«སྣར་»ཅིགགོ«པེ་»«སྣར་མྱུད། པོ«པེ་»སྣར་»—གསུང་ངོ༌། །།།«པེ་»«སྣར་སྔང༌། «པེ་»«སྣར་ལྐུགའ«པེ་»«སྣར་»འཇོལ། ཏོ«པེ་»«སྣར་ལྕེ། _།
