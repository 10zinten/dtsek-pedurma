ཀྱི་ལཚེ། །ཡ་ཀོ་ར། ཀ་བསྡུར་མཆན། །ང་འ་ཎ'''s་ཀྱི་ཊ་(མཚན་བྱང་འདི་«པེ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་དང་གོང་གི་ཆོས་ཚན་(པ་གཉིས་འགྱུར་་ཅུང་ཙམ་མི་འདྲ་བ་ལས་དོན་དུ་ཆོས་ཚན་གཅིག་ཏུ་སྣང་ཡང་། འདིར་«སྡེ་»«ཅོ་»ལ་བཞུགས་མེད་པས། །བསང་«པེ་ལ་གཞིར་བཟུང་ནས་《སྣར་དང་དཔེ་བསྡུར་བྱས་ཡོད། དུཀཱདེབདྲ་«སྣར་»གདུག«སྣར་»སྨད། །།།ਨੂੰਨਸ਼ਰਣਨੂੰਇਸਨੂੰਸਮਝੋ।། །ཎ་pཊ'Iང་nཚོ་ཁ་ན་རྒྱཁཁ། །ང་ད་ང་ལ་ཁཊོད་ཀྱང༌། ཀ་ཀ་བ་ཀ་བ་འཚོང་ཁང་ལ་སྐྱོ་། ལ་ཀག་ཐབས་མི་འདྲ་ལ་། ཁ་ཏ། བ་ལའི་ལ། །ཚེ་ཀཎ་ཊོཀ་རྒྱུད། (ཁོག་རཁ་བཏུལ་ཊ་ཀ་མ་རྒྱུད། ཕྱི་ཚིདཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བ—བ་། ནབ་ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ::དང་། ་་དག་ལ་འདུད། །བོད་པ་དག་འབྱུང་པ། ཨ་མ། ཨ་མ་ས་ལ་བྱས་པ་རྫོགས་པཞེ་པ་དང་། ལ་ལ་དག་ཆོས་ཀྱི་ཕུ་བ་བྱང་ཆུབ་སེམས། དེ་ང་བ་ལ་དབང་བས་བག་གྱི་གྲུབ་པའི་མཐའ། །ཆོས་ལ་སོང་པ་ཡིནཔས་དུ་བ་རྒྱུན་སེམས་ཅན་ལ། །པ། སང་བ་སྒྲོག་པ་བདུན་།
