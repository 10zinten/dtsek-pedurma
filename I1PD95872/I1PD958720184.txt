གནས་པར་བྱེད། །རི་སྦྱོང་བ་ཡི་གནས་དག་ཏུ། །སྐྱེ་རྒའི་རྒྱུན་ནི་བཅད་ནས་གནས༑བྱེ། བདག་ནི་མཚམས་སུ་ཕུང་བར་འགྱུར། །བྷནྡྷ་པ་ཡིས་མཚམས་སུ་བཅོམ༑། བྷནྡྷ་མཁས་པ་ཆེན་པོ་དེས། །སྣ་ཚོགས་སེམས་ནི་གར་བྱེད་པ། །དེ་ཡི་རང་བཞིན་དབང་དུ་འདུས། །བདག་ནི་མཚམས་སུ་ཕུང་བར་འགྱུར། །བྷནྡཱ་པ་ཡིས་མཚམས་སུ་བཅོམ། །།རྩ་རླུང་གི་སྒོ་ནས་རྣལ་འབྱོར་གྱི་སྤྱོད་པ་ཞེས་་བྱ་བ་སློབ་དཔོན་ཌ་བི་པས་མཛད་པ་རྫོགས་སོ། །།།།།བསྡུར་མཆན། ད་པ་འི་བློ་དག་ན་ཀུ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། Q«པེ་»«སྣར་»དྭ། «པེ་»《སྣར་སྦྱོང་། «པེ་»«སྣར་»འཕུང་། «པེ་»«སྣར་འཕུང༌། ལ་ཟག་་་་་་་་་་་་འོ་ཀྱི་ཁ་ན་མ་/ཅཎཱ་ཎའ་དྲིན་ཟླ་བ་དར་རླུང་བ་ན་བྱིན་རལ་ཎD::བསྡུར་འབྲས་རེའུ་མིག་ཏུ། །ལ་ཚེ། ཀཀཱཀའ་ཁྲི། ཏན་ངལཱལ་སུ་ཏ། རྒྱུད། (ང་ལ་འ་དང་། པེ་ཅིན། །༽སྣར་ཐང༌། ནེ། །རྒྱུད། རྒྱུད། ང་། ལ་དཔེ་གཞི། སྡེ་དགེ། །སྡེ་ཚན། ལོ་རྒྱུད། ་ཀཱརྒྱུད་འགྲེལ། པོད་རྟགས། ཟིཚན་གྲངས། །ལྡེབ། [ན། —ན། བ—བརཚིན་—ན། ནན། དེ་བས་ཚིག་རྐང་། དང་། བྱང་ཚེག་ཁྱིམ། །།།
