རྒྱུད། ཏིགསུམ་གནས་སུ་ལོངས་སྤྱོད་ཀྱང་། །དེ་ཉིད་གནས་ནི་རྟོགས་པ་མེད། །བདག་གིས་གཞན་ལ་ཇི་ལྟར་དྲི། །ཆུ་གཏིང་ཟླ་བའི་གཟུགས་བརྙན་ནི། །དོན་དམ་སྒྱུ་མར་རྟོགས་པ་མེད། །ལུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །འདི་ལྟ་བུ་ནི་ངོ་མཚར་ཆེ། །ཁམས་གསུམ་གནས་སུ་ལོངས་སྤྱོད་ཀྱང་། །དེ་ཉིད་གནས་ནི་རྟོགས་པ་མེད། །ལུ་ཡི་པ་ནི་སྨྲ་བར་བྱེད། །བདག་གིས་གཞན་ལ་ཇི་ལྟར་དྲི། །དེ་ཡི་རང་བཞིན་རྟོགས་པ་མེད། །རང་བཞིན་གྱི་དོ་ཧ་མཛོད་ཀྱི་གླུའི་ལྟ་བ་ཞེས་བྱ་བ་སློབ་དཔོན་ལུ་ཡི་པས་མཛད་པ་རྫོགས་སོ། །།།བསྡུར་མཆན།  ེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཙེ«པེ་»«སྣར་»ཏཏྟཾ། «པེ་»«སྣར་བ། «སྣར་»སྒོ། ཡོ་པ་བསམ་བསྡུར་འབྲས་རེའུ་མིགདང་། ན་བ་ལ། །བྱང་ལ་བྱ་ਸਲੇਨੂੰਨਾਲਲੋਟਲརྒྱུད། ་(སྡེ་དགེ། །སྣར་ཐང་། །ཅོ་ནེ། པེ་ཅིན། རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། རྒྱུད། །བྲིདཔེ་གཞི། །སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང་། །ན—བར། བ——བ་བ—ན། ན?—བའོ་ལ་ཕྱོགསཚིག་རྐང་། ཚེག་ཁྱིམ། ཀྱི་། །
