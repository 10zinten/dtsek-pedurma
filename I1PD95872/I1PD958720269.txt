བསྟན་འགྱུར། རྒྱུད། ཝིཏིསྒྲོག་པར་བྱ། རྗེབ་། མར་སར་བླངས་ནས་ཐོ་རངས་བཟའ། །ཞག་ནི་བདུན་་ནས་རྟགས་འབྱུང་སྟེ། ལྟ། བདུན་གཉིས་གྱུར་ནས་ཉི་འོད་ལྡན། དཔོ། བདུན་་་་་གསུམ་ཉི་ཤུ་རྩ་གཅིག་ནས། །འབྲས་བུ་ནད་སེལ་ལུས་ཡང་ངོ༌། །ལོ་་་་དྲུག་ཏུ་ནི་བཟའ་བཏུང་གིས། །རྣམ་རྟོག་བྲལ་ཞིང་ཚེ་འགྲུབ་བོ། །ཐོ་རས་མུ་ཟི་སྲང་ཕྱེད་ནི། །ཡུངས་ཀར་མར་ཁུས་བཏུལ་ནས་བཟའ། །ཤུ་བའི་་་ནད་སེལ་ཚེ་འགྲུབ་འགྱུར། ལ། སྤྲིན་ནག་ཞོ་གཉིས་སྦྲུལ་མིག་པའི། ༄༅། ཁུ་་་བས་མཉེས་ནས་གང་བར་བྱ། །ཞུ་རྗེས་དྲུག་ཅུའི་འབྲས་ཆན་བཏང་། །འོ་མ་བཏུང་ཞིང་ཚ་སྔོ་སྤང་། ང། །ལོ་གཅིག་སྦྱོར་བ་འགྲུབ་པར་འགྱུར། །:ཤྲཱིང་ག་ར་ཛའི་ཕྱེ་ལ་ནི། །མར་དང་སྦྲང་གི་རྩི་སྦྱར་ནས། །ལྟོ་སྟོང་དག་ལ་་་བཟའ་བར་བྱ། །སྐྱུར་རྩང་བཅུད་ཅན་ཁ་ཟས་བཏང༌། །ཟླ་བ་དྲུག་ཏུ་ཟོས་གྱུར་ན། །ནད་རྣམས་སེལ་ཅིང་ཉིས་བརྒྱ་ཐུབ། ཁ་། འབྲས་བུ་གསུམ་་དང་ལྕགས་ཕྱེ་དང་། །དེ་བཞིན་བླིང་ག་ར་ཛ་དང་། །རུ་རྟ་དང་ནི་ཆ་མཉམ་སྟེ། །:རྡིང་གའི་ཁུ་བས་ལྡེ་གུར་སྦྱར། །གྲིབ་སྐམས་ཕྱེ་མ་བཟའ་བར་བྱ། ཤོ་། རེག་པ་དང་ནི་སྔོ་སྐྱུར་སྤང་། ཊ་། ལོ་གཉིས་ཀྱིས་ནི་འགྲུབ་འགྱུར་རོ༑། ལྕགས་ཀྱི་ལྷུང་བཟེད་མར་གྱིས་བཀང་། །ལྷ་ཤིང་རྩེ་བའི་ཤུ་་་་་ལྤགས་ནི། །ཞིབ་བཏགས་སྦྱར་ལ་ལོ་གཅིག་བཞག། །གཟའ་སྐར་བཟང་་་པོའི་དུས་སུ་བཏོན། །ཡི་དམ་ལྷ་ཡི་བསྙེན་པ་བསྐྱལ། །།ར་ཡི་འོ་མ་བཏུང་བར་བྱ། །ཟླ་བ་གསུམ་ན་གཉེར་མ་དངས། །ཟླ་བ་དྲུག་ནས་རླུང་ཤུགས་་ལྡན། །ལོ་གཅིག་སྦྱོར་བས་ཚེ་འགྲུབ་བོ། །འབྲས་བུ་གསུམ་དང་:བྷིང་་་གའི་ཕྱེ། །མར་དང་སྦྲང་རྩི་སྦྱར་ལ་བཟའ། རྒྱུ་ལ། སྨན་ཉིད་ཞུ་བའི་དུས་སུ་ནི༑། འོ་མ་འབྲས་བསྐོལ་བཟའ་བར་བྱ། ་ཁ། མིག་རྣོ་ནད་མེད་སུམ་བརྒྱ་་་།
