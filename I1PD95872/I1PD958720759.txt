རྒྱུད། ༼ཝིསྟོང་པ་ལ་ཐོག་མ་དང་ཐ་མ་དང་དབུས་སུ་རྟོག་པ་དང་། དད་ལྡན་གྱི་སེམས་་་སུན་འབྱིན་པ་དང་། ཇི་ལྟར་རྙེད་པའི་དམ་ཚིག་མི་བསྟེན་པ་དང་། བུད་མེད་ལ་སྨོད་པ་སྟེ་རྩ་བའི་ལྟུང་བ་བཅུ་བཞིའོ། །དེ་ལ་སྟོབས་ཀྱིས་རྣལ་འབྱོར་མ་ལེན་པ་དང་སྟོབས་ཀྱིས་དེའི་བྱང་ཆུབ་ཀྱི་་སེམས་ལེན་པ་དང༌། ཚོགསང་བ་མི་སྡོད་པ་དང་། དམ་ཚིག་ལ་རྩོད་པ་དང༌། དད་པ་དང་ལྡན་པ་ལ་ཕྱིན་ཅི་ལོག་གི་ཆོས་སྟོན་པ་དང་། ཉན་ཐོས་ཀྱི་ཐེག་པ་པ་དང་ལྷན་ཅིག་ཞག་བདུན་དུ་གནས་པ་དང༌། རྣལ་འབྱོར་མི་ཤེས་་པར་རྣལ་འབྱོར་པར་མངོན་པའི་ང་རྒྱལ་བྱེད་པ་དང་། མི་དད་པ་ལ་ཆོས་སྟོན་པ་སྟེ་སྦོམ་པོའི་ལྟུང་བ་བརྒྱད་དོ། །རྡོ་རྗེ་སེམས་དཔའི་བསླབ་པ་ལ་གནས་པས་རྣམ་པ་ཐམས་ཅད་དུ་ཡོངས་སུ་སྤང་བར་བྱའོ། །།།རྡོ་རྗེ་ཐེག་པའི་ལྟུང་བའི་སྙེ་མ་ཞེས་བྱ་བ་མཁས་པ་ཆེན་པོ་ཨ་བྱཱ་ཀ་ར་གུསྟས་མཛད་པ་རྫོགས་སོ། །མ་ག་དྷཱའི་༔མཁས་པ་ཆེན་པོ་དེ་ཉིད་དང་། །མང་དུ་ཐོས་པའི་དགེ་སློང་བསོད་སྙོམས་པ་སངས་རྒྱས་གྲགས་པས་ནཱ་ལེནྡྲར་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་་་ཕབ་པའོ། །།།ཀ་ཁ་ཁ་ཌ་ཁ་ཙ་ཆ་ཟླ་བ་ཌོ། །བོད་པ་དང་། ཀ་ཆུང། ་་ར་ཁ་བླང་བྱ་མ་ཎན་ཊོ་ཁོ་ཀ་བ་ལྔ་པོ་ཀ་ཀུ་ཏ་ཀਨਵੇਂਸਾਲ'ਸਟਰੋਲਛੁਟਵਰਤੋਂབཀའ་ཁྱཀཱ་ཀཾ། ་ཁང་། བློ་ལ་ཀུན་ལ་ཁཎྜི་ཏ་ཀབ་niཚོ། ཨ་མ་པ་ལ་ང་ལས་བྱ་ཁྱབ་ཀཚོ། ་བར་་་་ལ་བལ་ཀ་ཁ་ལོའི་ཀ་ཁ་ག་ལ་ཀ་ཁ་ཁལ་ཁ་ཁ་ཆཚོ་ཊཱ་ཀཊ་དང་། ་། ཆཎྜ་རར་འཚེ་བའི་བ་དང་། མུཥཱ་ཏཚོ་ཏི་ཏཁན་༌ར་ཎ་ཥ་ལྔ་། ཚོ་ཁཛོ་ཀྑཡ་ཀ་ཙན་།
