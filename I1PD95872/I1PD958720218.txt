བདག་ཉིད་སྙིང་རྗེ་ཀུན་དུ་ཁྱབ། །ནམ་མཁའ་ལྟ་བུར་འགྱུར་བ་མེད། །གཉིས་ལ་མི་གནས་སེམས་ཉིད་ཡིན། །མེ་དྷི་བདག་ནི་སྨྲ་བར་བྱེད། །རང་རིག་མ་བཅོས་བྱ་བ་ཐོང་། །འཛིན་མེད་དྲན་མེད་སེམས་ཉིད་དེ། །ཆད་པར་མི་་འགྱུར་རང་རིག་རྟོགས། །རྟག་པར་མི་འགྱུར་སྒྲོ་འདོགས་བཅད། །སྟོང་པའི་རང་བཞིན་རྟག་ཆད་མེད། །མེ་དྷི་བདག་ནི་སྨྲ་བར་བྱེད། །ལྟ་བ་བསྒོམ་མེད་འབྲས་བུ་མེད། །འp། སྣང་ལ་རང་བཞིན་དངོས་པོ་མེད། །རྣམ་རྟོག་སྦྲུལ་གྱི་མདུད་པ་ལྟར། །རང་གྲོལ་ཟུང་འཇུག་ཐ་སྙད་མེད། །དེ་ལྟར་བསྒོམས་པས་་་སྟོང་ཉིད་རྟོགས། །མེ་དྷི་བདག་ནི་སྨྲ་བར་བྱེད། །ལྷན་སྐྱེས་ཟུང་འཇུག་སྟོང་་ཉིད་དེ། །རྣལ་འབྱོར་བདག་གིས་རྟོགས་པའོ། །ལྷན་སྐྱེས་ཟུང་འཇུག་གི་་གདམས་པ་སློབ་དཔོན་མེ་དྷི་ནིས་མཛད་པ་རྫོགས་སོ། །།།།།རྒྱ་གར་སྐད་དུ། 《ང་བོད་ར་བ་འད་ལས་ས་བདག་རེད་བ་སྐྱེ་འགྲོ་མ་བཞི། །རྫོགས་པ་་དེ། མ་ལེ་དང་བའི་། བདེ་བ་བདུན་པ་པ་དང་དག་སྦྱོར་བ་ཡིས། ་ལ། མ་དག་དེ། །བམ་གྱི་རྣལ་འབྱོར་གྱིས། །རྒྱུ་བ་ཀུན་ལ་གལ་གནད་ས་དེ་དག་མ་རྟོགས་པ་ཡིན། བླ་ན་མེད་པའི་བར་ན་ས་བདུན་འོད་གསལ་ལ་ཤེས་རབ་དང་། །སེམས་ཉིད་པ་ད་བ་ས་དང་དེ་ལ་བདག་བདེ་བ་།
