རྒྱུད། ཝི#དེ་ར་སྒྲ་དང་། །དེ་རྗེས་ཏེ་ཏ་ད་ཀ་རན་ཏེ་བྲི། །ནི་པ་རི་བྷ་ན་ཏེ་མེ་ོནུས་མེད། །ཨོ་སིདྡྷི་ནི་གཱརྫུ་ན་པ་སྟེ། །པ་ཡི་ར་གསུམ་མཐར་སྔགས་རྫོགས་པར་བཟླ། །ལན་གསུམ་གཅོད་ཅིང་ཡུངས་ཀར་མར་ཁུ་གཏོར། །ལྕགས་བསྲེགས་མེ་རུ་ཞུགས་ཀྱང་སྲེག་མི་ནུས། །ནུས་པ་མཐར་ཕྱིན་སྔགས་སྦྱོར་གཉིས་པ་ནི། ལ་། ན་མོ་ལ་་བྷི་ཤྭ་མི་ཏྲ་ཡ་ཏེ། །ཧཱུ་ཏེ་བྷ་ཧ་ཧ་ཧཱུཾ་ཧཱུཾ་བྷ་ཡ་མི། །ཙི་ལི་སྭཱ་ཧཱ་མེད་ནུས་མེད་པར་གྱིས། །ད་ལྟར་བྱང་སེམས་ལྡན་པའི་རྣལ་འབྱོར་གྱིས། །རིམ་བཞིན་བསྒྲུབས་ན་འགྲུབ་འགྱུར་ཐེ་ཚོམ་མེད། །འཇིགས་པ་བརྒྱད་རྨོངས་པར་བྱེད་པའི་ཐབས་ཞེས་བྱ་བ་སློབ་དཔོན་:བྷི་ཥ་ནས་མཛད་པ་རྫོགས་སོ། །ཁཱ་ཌ། p། །ཏོ་ཅཆེ་ཀའོ། ཆ་ཙི་ཀོ་ཎ་ཎཁ་པཎྜི་ཎ་ཚོ། །ཀུ་ཀ་ཁ་ཁ་ལ་ཟ་རྒྱུ་ཚེགས་ཀ་ਤੋਂਸਿੰRTAN&।ਹਿੰਸਨ'ਰਸੋਇགང་ན་ང་་་་ཆུ། ད་་་བསྡུར་མཆན། །དད་པ་བྱེད་རྒྱུརྒྱུནཱ་མ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅འཇིགས་པ་བརྒྱད་ང་རྨོངས་པར་བྱེད་པའི་ཐབས་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་《སྣར་»ཎཾ། «པེ་《སྣར་པ་ཡཾ། «པེ་»«སྣར་ཡིས། །«པེ་»«སྣར་»པོ། _ི«པེ་»«སྣར་»བཅས། «པེ་»«སྣར་འབྱུང་། «པེ་»«སྣར་ཨཱ། «པེ་»«སྣར་»སྐན་གྱིས། །«པེ་»«སྣར་ཀྱིས། «སྣར་སྐ། །《པེ་《སྣར་བྱས། «པེ་《སྣར་བོ། ༼ན«པེ་《སྣར་»སྨདྨ། «པེ་»«སྣར་ཐུནྟ། @«པེ་»«སྣར་+། ཉེ«པེ་»«སྣར་+། །«པེ་»«སྣར་བལ། དགྱེ«པེ་»«སྣར་»འཛམ། «པེ་»«སྣར་»བསྲེག«པེ་»«སྣར་»སིདྡྷི། «པེ་»《སྣར་»རི་ཏ། ཕྱི་::ཀྱིང་::འགྲོ་འདྲ། །(«པེ་《སྣར་»མི། «པེ་《སྣར་བསྲེག«པེ་»མི། 《སྣར་»མེ། «པེ་»སྣར་»«ཅོ་དེ། །«པེ་»«སྣར་བྷ་ཁ། བ་ཏུ་ག་ལ་བ་ཀུ་ཊ་་ཏེ། །ང་ཏའི་ཟླ་བ་དང་ཁ་ཀྱི་ངང་ལ་ཁྱབསློབ་མ་ཎཱ་ཏྲ་་ཀྱི་འཚོ་བ་ཡོད་མེད། ད་ནི་ང་ਨਸਵੀਮੁਲਕਉਲੁਨਖੁਸ਼།
