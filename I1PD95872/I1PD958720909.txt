ཁ་ཏཁང་། ཟ་ཁང་ཁོ་། །ལ་ལོ་རྒྱུས་ཀྱཊི་། །pད་། །དུཡང་ལ་ཀྱང་། ལ་ལ་ཆུ་ཀཡ་ར་བཅཚོ་ཁ། །(་ལ་གལོཔ་དང་། །ལྟོ༌བསྡུར་མཆན། ་་་་་བར་་ཟླ་ལ། །།།བརརདརརརངརདངརདཔ། །་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»རྡོ་རྗེ་སེམས་དཔའི་སྒྲུབ་་ཐབས་ཞེས་བྱ་བ་བཞུགས་སོ། །།།/p?། ལ་་ད་ ོ་དེ་ལས་ཀྱང་། «པེ་»«སྣར་འཐེབ། «པེ་»«སྣར་»གིས། «པེ་ཡིག«པེ་»«སྣར་»འཐེབ། སུ«པེ་«སྣར་བཏོལ། @«པེ་»«སྣར་»འཐིལ། «པེ་»«སྣར་»སལ། ཉི«པེ་»«སྣར་»གྱིས། ག་༼S/་ཡོན་་ཡི་ཡིད་དུ་ཡོད་ལ། /་་༼ཀ་ག་ང་། །ཏང་ག། བ་དང་དཔ། ནགསརཔར་དེ་ང་ཚོ་ད་༼པན། འདས་ཆ་ག་གས་ས་ཡོད་ལ་ཡེདེ། ཙ་བའི་ཆ་ལ་ངོ་བསྐྱབས་རྗེ་མི་ལ་ལ་ཁ་ཀ་ཁ་ག་། /?ཚོ་E་རྒྱུད། (ཀ་ཀ་ཁག་ཊ་། །རྒྱ་རྒྱུད། རྒྱུད། नारकार། དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། པོད་རྟགས། པ། ཚན་གྲངས། /:ཟླ་་ལྡེབ། ན—བ་བ—བ་བ—ན་བ་བར་་། ཚེག་ཁྱིམ། །ལ་ཚུལ་དང་། །༼རྫཔ་བླ་མ་་ག་ཊ་བ་ཁ་བ་བླ་ཊ་བློས་བཀྲ་རྒྱལ་ਪਾਰਕਿਨਸਨਪਰདང་པོ་སུམ་པ་ཞེས་དང་། དཔེ་བྱད་འབྲལ་པོ་བདུད་ཀྱི་བ་མ་ཡབ་རྗེ་བདུད་འཛིན་པ། བཡོན་པས་དྲིལ་བུ་།
