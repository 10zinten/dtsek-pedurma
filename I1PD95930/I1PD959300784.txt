ས་ཁོ་བ་མེད་དོ། །གླ་རྔན་བྱིན་པ་ལ་ལྟུང་བ་མེད་དོ། །ལས་དང་པོ་པ་དང་། སྨྱོས་པ་དང་། སེམས་འཁྲུགས་པ་དང༌། ཚོར་བ་དག་གིས་ཉེན་པ་ལ་ལྟུང་བ་མེད་དོ། །བསླབ་པར་བྱ་བ་ནི་དགེ་སློང་གིས་ཡུལ་ཐམས་ཅད་ལ། གནས་་་སྐབས་ཐམས་ཅད་དུ་འཐག་པ་བསྐྱེད་པའི་སྤང་བར་འགྱུར་བ་རྣམ་པ་ཐམས་ཅད་མི་བསྟེན་པར་བསླབ་པར་བྱའོ། །ཕྱིར་བཅོས་པ་ནི་སྔ་མ་བཞིན་ནོ། །འཐག་པ་བསྐྱེད་པའི་སྤང་བ་སྟེ་ཉི་ཤུ་བཞི་པ་བཤད་ཟིན་ཏོ། །།།བོད་པ་དེ་དག་གང་ਪੀਸਬਨਬਤਾਪབསྡུར་མཆན། དེ་བ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། འ«པེ་»«སྣར་»སྤོ། «པེ་»《སྣར་ཞན། («པེ་»«སྣར་བ། «པེ་»«སྣར་རྩོམ། «པེ་»«སྣར་བརྟོལ། «པེ་»«སྣར་»མདོ་སྡེའི། «པེ་《སྣར་བར། །(«པེ་ན། «པེ་»«སྣར་»སོ། (ཆོས་ཚན་འདིའི་ནང་དུ་སོ་སོར་ཞེས་པ་རྣམས་«པེ་»«སྣར་ལ་སོ་སོ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་»«ཅོ་བར། ︽«པེ་»《སྣར་བར། «པེ་»《སྣར་»—འདི། «པེ་»«སྣར་བསྡུ་བ། དབང་དུ་འདུབར་བྱ་བ་་་་་་་་་་(«པེ་»«སྣར་ལས། «པེ་»«སྣར་བྱ་བར། «པེ་»«སྣར་བ། །«པེ་དལྟར་ན། «པེ་དསྐུར་དབ། དེའི་«པེ་«མྱུར་དཔལ«པེ་»«སྣར་པ། ་«པེ་»«སྣར་པའི། «པེ་»«སྣར་»+ཇི། ༄༅«པེ་»«སྣར་»སྲུང་། (མ«པེ་»《སྣར་བར། «པེ་»«སྣར་བསྟན། ཚེ«པེ་»«སྣར་ཟབ། «སྣར་»འབྱམས། «པེ་»«སྣར་»ལ༑(«པེ་»«སྣར་»ཀྱིས། །«པེ་«སྣར་དྲུངས། «པེ་»«སྣར»ཏུ། ཉི«པེ་»«སྣར་པ། ཚེ«པེ་»«སྣར་བརྒལ། །«པེ་»«སྣར་+སྙམ། «པེ་»ཉིད། «པེ་《སྣར་»གེ། «སྣར་པ། «པེ་»«སྣར་»ཀྱིས། འ«པེ་»«སྣར་ཇི། «པེ་»«སྣར་དགོས་པའི། «པེ་《སྣར་བ་སྟེ། «པེ་»«སྣར་—དང་། ་་་བན་གབ་པའི་།
