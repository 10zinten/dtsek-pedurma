«སྣར་ཟུར་དུ་འགྲེལ། «པེ་»《སྣར་བ། «པེ་»«སྣར་ཨ་ལི་ཉིས་གྱུར། «པེ་»«སྣར་པ། «པེ་»«སྣར་»ཟླ། ཀྱེ«པེ་«སྣར་»ཨ་ལི་ཐིག་ལེ་ངོ་བོ་གནས། འབུ«པེ་»བསམ་ནི་ཧཾ་ཨཱ་ཨོཾ་ཧཱུྃ། «སྣར་»བསམ་ནི་ཧཾ་ཨཱ་ཨོ་ཧཱུྃ། །«པེ་»«སྣར་མོ། «པེ་»«སྣར་»བཅུ། «པེ་»«སྣར་»ལེ། «པེ་»«སྣར་»སྒོམ། (ག«པེ་ཡི། པོ«སྣར་+པ། ༄༅«པེ་»«སྣར་»ལེ། །«པེ་»«སྣར་བཏགས། «པེ་»«སྣར་»སྐྱེས་པ། «པེ་»«སྣར་»རླུང་ཐིག་ལེ་ལ། ︽«པེ་»སྣར་»རྗེ༑«པེ་»«སྣར་བ། «པེ་»«སྣར་རྟེན། «པེ་《སྣར་»ལྔ་ལོ་ཟླ་ཕྱག་རྒྱའོ། །ན«པེ་མཁའ་སྣང་སྟགས། «སྣར་མཁའ་སྣང་རྟགས། Ppp། «སྣར་»དག«པེ་»«སྣར་མོ། «པེ་»«སྣར་ཆོས། «པེ་»«སྣར་པ། «པེ་»«སྣར་རིགཔོ《སྣར་རྟོགས། @«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»པོ། «པེ་»«སྣར་བརྩམ། །«པེ་»«སྣར་གརྫུ། དང་། །།བདག་བློ། འ་འདྲ། ཏུ་བ་། བདེན་དང་། དེ་དག་ལ་བ་དེར་ལོཏོག་དང་འདྲ་སྟེ། །་། བསྡུར་འབྲས་རེའུ་མིག་ཁྲོན་པ་བར། །་་༌་དུ་ཞུ་བ་རྒྱུད། （ལྟ་རཏ་ཏུ་གནས་བརྒྱ་བསད་སྣར་ཐང་། ཅོ་ནེ། སྡེ་དགེ། zrརྒྱུད། པེ་ཅིན། :རྒྱུད་འགྲེལ། །རྒྱུད། T། དཔེ་གཞི། སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང༌། ཕྱིའི་ན་_ན། ནབ་བབ། བ_ན་ན་་:，ཚེག་ཁྱིམ། །འོང་ཆུ་སྐྱར། ས་བབ་ཀྱི་དམ་པ་རྒྱུད་པ་ལ། །འ་པ། ::ཀྱི་ཊི་ན། n:ཀ་ལོན་པ་དང་། དེ་ལྟར་བཀོལ་པསཁོ་ཚོ་ཁའི་གནད་ཀ་ཤེས་པ་ཁ་ཏོ་::་་ལོ་་་་པ་ལའོ་ཏུག་ཚོས་དཀའ་འཁབ་ལ་དང་བ་དན་ལ་སོགས་།
