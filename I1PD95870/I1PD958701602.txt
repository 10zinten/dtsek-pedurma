ན་ཉིད་ཉི་ཟླ་སྣང་བ་ཞེས་བྱ་བའི་སྒྲུབ་ཐབས་ཆེན་པོ་དགེ་སློང་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།པཎྜི་ཏ་གུ་རུ་པདྨ་འབྱུང་གནས་དང་། :ལོ་ཙཱ་བ་གླང་དཔལ་གྱི་སེང་གེས་བསྒྱུར་རོ། :། སླད་ཀྱིས་ཀྱང་པཎྜི་ཏས་རྒྱ་གར་ཕྱོགས་ནས་སྒྲ་རང་འགྱུར་དུ་བསྒྱུར་རོ། །བི་ལ་བཛྲ་ལས་བླ་མ་སྨན་བཙུན་ཁྲོས་པ་ལ་གནང༌། དེས་དངོས་ལ་གནང་བ་ལགས་སོ། །།།སོ། །དེར་འདུ་བ་ཡི། །ལ་ཏུ་ནང་བ་དར་བ་། །ཁས་། །དེ་ན་ཡོད། ་མ་དང་དེ་ནས་བན་དེ་བ་ལ་བསྡུར་མཆན། ཚོར་བ་བདེ་བ་ལ། །བདེ་བའི་། དབྱངས་སྒྱུ(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་《སྣར་དེ་ཁོ་ན་ཉིད་་སྒྲོན་མ་དགེ་སློང་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་བཞུགས། །«པེ་»«སྣར་+ས་ན་ཏ། ང་ཟླ«པེ་《སྣར་ཨ། «པེ་»«སྣར་+ཨ་ཏི། །(«པེ་»«སྣར་»མཛད། «པེ་»«སྣར་»ཐམས་ཅད། «པེ་»《སྣར་གྲུབ། །«པེ་»«སྣར་+ལམ། «པེ་»«སྣར་དང་། «པེ་»«སྣར་སྒྲོག། («པེ་»《སྣར་ཞིང་བརྩོན་འགྲུས་དགའ། («པེ་»«སྣར་»བོ། །:«པེ་《སྣར་རྫས། ®«པེ་»«སྣར་བསྲུང་། (ས«པེ་»«སྣར་རླབས། @«པེ་»«སྣར་བསད། «པེ་»སྣར་པ། «པེ་»ཀ་«སྣར་»གཏུབ། དགྱེ«པེ་»«སྣར་ན། །ཁ་བ་དང་། བྱེད་པ་བ་མ། །ར་(«པེ་»«སྣར་»བཙལ། :«པེ་»«སྣར་གཟུང་། །སྐྱེ«པེ་《སྣར་བདེར། ®«པེ་《སྣར་»དྲངས། («པེ་《སྣར་»ལ། པ«པེ་»«སྣར་གཟུང་། («པེ་»«སྣར་ཡི། «པེ་»«སྣར་»བསྡུ་བ། །«པེ་»«སྣར་པདྨ། །«པེ་»《སྣར་པར། ་་་་་་་་་། ་ཀྱི«པེ་»«སྣར་»ཡི། («པེ་»«སྣར་»སྲེད། «པེ་»《སྣར་+ལ། «པེ་《སྣར་ལྷ། «པེ་»«སྣར་མ། «པེ་»སྣར་བྱ། ་«པེ་»«སྣར་»གོང་། («པེ་»«སྣར་+དང་། «པེ་»ལཀྱིས《སྣར་དང༌། «པེ་»«སྣར་ཧཱུཾ། །ཏའི། །ཁ་སྐྱེས་པ། དུད་པ་A(«པེ་»«སྣར་ལ། «པེ་»«སྣར་ཡི། («པེ་»«སྣར་»ལྔ། «པེ་»«སྣར་»འགྲིལ། །«པེ་»《སྣར་འཁྲང་། »«པེ་»སྨད་གོས་སྔོན་ཤམ་ཐབས་ཅན། །«སྣར་»སྟོད་གོས་སྔོན་སྟོད་གཡོགས་ཅན། །«པེ་»«སྣར་»འཁྲིས། (ན«པེ་»《སྣར་བཅུ། །་(@«པེ་»«སྣར་+མཆོད། ་༌«པེ་»«སྣར་»—ནི། པ《«པེ་»«སྣར་པོ། སྟེ«པེ་»«སྣར་བདེར། ་(སྟེ«པེ་《སྣར་རློབས། «པེ་»«སྣར་»གེ། །
