ནས་། །བས་མ་གས་་་་་་་་མ་་་་འིཀྱི་ཀ་བ་གླ་། དང་རྒྱུདཔ། །ཁ་་་་་་་་། མདའིཚེ། ཚེ་ཁ། དང་། དwགས་བའི། །རྒྱུད་དང་་་་། །སྤོ་བའི་་ལ་ལམ་ཀྱི་སལར་ཌ་ཀྱིས་སྟེ། རྒྱུ་ན། ང་ཁག་དེ་ལས་:::༄༅། །དམ་ཚིག་རྒྱན་གྱི་རྣམ་པར་། འཁོར་བའི་ཚོང་བ་བཤད་པ་ཞེས་བྱ་བ། །པ་ལ། ང་དང་བྱ་བའི་ཁ་ལོ་བ་དང་། བཀའི་༄༅། །རྒྱ་གར་སྐད་དུ། ས་མ་ཡ་ཨཱ་ལཾ་ཀཱ་ར་ནཱ་མ་ཊཱི་ཀཱ། བོད་་སྐད་དུ། དམ་ཚིག་རྒྱན་གྱི་རྣམ་པར་བཤད་པ་ཞེས་བྱ་བ། སྟོན་ཀའི་ཟླ་བ་དྲི་མེད་འོད། །མ་སྐྱེས་རྡོ་རྗེ་ཡེ་ཤེས་ཉིད། །རྡོ་རྗེ་ཅན་ལ་ཕྱག་འཚལ་ལོ། །ས་་གཞི་དང་ནི་ཡོངས་བསྐྱེད་ལ། །འབྲས་བུ་འབྱིན་པའི་ནུས་པའོ། །ཏིང་འཛིན་རྒྱུ་དང་རབ་སྦྱོར་བའི། །བདག་པོ་གྲོགས་ལ་སོགས་པ་ཡིས། །འདི་ཡང་འབྲས་བུ་འབྱིན་པའི་གྲོགས། །རྟེན་མེད་འབྲས་བུ་འབྱིན་འདོད་པ། །མི་འབྱུང་ས་བོན་མེས་བསྲེགས་བཞིན། །དེ་ན་འདི་ཡི་དོན་འདི་བྲི། །ཞེས་པའོ། །:དེ་ནས༑མི་དམ་ཚིག་མེད་དང་འཚལ་བར་གཏོང༌། །སོམ་ཉི་ལ་སོགས་བསལ་་་བའི་ཕྱིར། །དམ་ཚིག་རྣམས་ནི་མདོར་བསྡུས་ནས། །གསང་བའི་བདག་་་་་་བཏུད་དེ་ནི་བཤད། །ཅེས་པ་ནི་རྩ་བའི་དོན་བཤད་པའི་དོན་དགོངས་སོ། །དེ་ལ་དམ་ཚིག་མེད་པ་ནི་བསྒོམ་པ་དང་། བསྒྲུབ་པ་དང་། སྦྱིན་སྲེག་ལ་་་་་།
