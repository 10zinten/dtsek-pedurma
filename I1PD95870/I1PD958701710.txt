ཀྱིས༑། འགྲོ་བ་མ་ལུས་ཐམས་ཅད་ཀུན། །དེ་ཉིད་དོན་ནི་རབ་རྟོགས་ཤོག། བདག་ཀྱང་འགྲོ་དོན་བྱེད་པར་ཤོག། ཀླུའི་བྱང་ཆུབ་ཀྱི་སྒྲུབ་ཐབས་ཀྱི་རྣམ་པར་བཤད་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །།དེ་ནས་དེ་ན་སློབ་དཔོན་ནཱ་གཱརྫུ་ན་ཞེས་ཀྱང་བྱ། །ཀླུ་སྒྲུབ་སྙིང་པོ་ཞེས་ཀྱང་བྱ། །ཀླུ་ཡི་བྱང་ཆུབ་ཞེས་ཀྱང་བྱ། །ཀླུ་སྒྲུབ་སྙིང་པོས་མཛད་པ་རྫོགས་སོ། །།།ནས་ཚེས་ལ་ང་། ང་ཚོ་བབཀྱི་བ། བཀར་པ་དང་། ་བསྡུར་མཆན། གཏེན་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»གོས་སྔོན་ཅན་སྒྲུབ་ཐབས་ཀྱི་མན་ངག་གི་རྒྱ་ཆེར་བཤད་པ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་བཞུགས་སོ། །«པེ་»སྣར་པ་ཎི་སཏྭ། «པེ་»«སྣར་—ཊི་ཀ«པེ་»«སྣར་བཞི། ་་་ང་ས་«པེ་»«སྣར་གྱི། «པེ་»«སྣར་»བོ། «ཅོ»གསུངས། ︽«པེ་»«སྣར་»འཐོབས། «པེ་»སྒྲོན་མ། «སྣར་སྒྲོལ་མ། «པེ་»«སྣར་རིས། «པེ་《སྣར་»དང༌། «པེ་»«སྣར་»སྒོམ་པར། «པེ་»«སྣར་»དང༌། (གྱེ«པེ་»«སྣར་»པ། «པེ་»«སྣར་མ། ཉེ«པེ་»«སྣར་འདུས། རབ་ཀྱི། །ལོ་མའི་རྩེ༽(«པེ་»«སྣར་ཆེ། Q«པེ་»《སྣར་པའི། «པེ་《སྣར་གསོག«པེ་»«སྣར་ལ། (ན«པེ་«སྣར་»རྒྱུ། ཚེ«པེ་»«སྣར་»བསྐྱེད། @«པེ་»«སྣར་»བློ་གི། འ«པེ་»«སྣར་»རླབས། «པེ་»《སྣར་»སྒོམ། («པེ་»«སྣར་སྡང་། །::ཀ་བ་༼ང་«པེ་»«སྣར་སོང་། «པེ་《སྣར་བློ་གི། «པེ་»«སྣར་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་སྭཱ་ཧཱུཾ་ཧ་ཧཱུྃ་ཨ་ཧཱུཾ། «པེ་»《སྣར་»དབུལ། །«པེ་»«སྣར་དང་། 《«པེ་《སྣར་གྲུབ། @«པེ་»«སྣར་—པ། ་«པེ་»《སྣར་ཨཱ། །སྙིང་རྗེ། «པེ་»«སྣར་གསོ་བ། (《སྣར་བཞི། ཀྱེ«པེ་《སྣར་བསད། («པེ་»ཧཱུྃ་བ། «སྣར་ཧཱུྃ་བ། ་«སྣར་»+བ། ་《སྣར་པོའི། །ཚེ«པེ་»《སྣར་»ཕྱི། «པེ་»མེད་དེ་《སྣར་»དོན་ནོ། །།ཏཀྱི་ཚོས་ཀྱི་མ་དོའི་དེ་ལས་ཀྱི་Dཎ(«པེ་»«སྣར་མི་འདའ་ཡང་། («པེ་»«སྣར་»པའོ། །«པེ་»«སྣར་»+ཐམས་ཅད། «པེ་»《སྣར་»དང་། སུ«པེ་»པ་ཆོས། 《སྣར་པའི་ཆོས། དེདག་ན། །«པེ་»«སྣར་འབྱུང་། («པེ་»«སྣར་»སྟེ་ཧཱུཾ། «པེ་»«སྣར་»བཞགཆོ«པེ་»«སྣར་»+པའོ། །
