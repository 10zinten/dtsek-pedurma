རེངས་པར་འགྱུར་རོ། །སྒོ་རྩར་སྦས་ན་འབྱེར་བར་འགྱུར་རོ། །རལ་གྲི་ལ་་བསྔགས་ནས་གཡུལ་དུ་ཞུགས་ན་རྒྱལ་བར་བྱེད་དོ། །ཁྲོས་པས་སྙིང་གར་་་་་བསྣུན་ན་འཆི་བར་འགྱུར་རོ། །མེའི་དུར་དུ་སྦས་ན་:རིམས་ཀྱིས་འཆི་བར་་་་འགྱུར་རོ། །རབས་ཆད་ཀྱི་ཐོད་པ་ལ་མིང་དང་སྔགས་བྲིས་ནས་རབས་ཆད་ཀྱི་་་དུར་དུ་སྦས་ན་རབས་ཆད་པར་འགྱུར་རོ། །མངོན་པར་བསྔགས་ལ་དུར་ཁྲོད་ཀྱི་ཐལ་བའི་སྟེང་དུ་བཞག་ན་འབྱེར་བར་བྱེད་དོ། །ཁྱི་ནག་པོའི་ཐོད་པ་ལ་་་་་སྔགས་དང་མིང་བཅས་པ་བྲིས་ནས་ཀླུའི་གནས་སུ་སྦས་ན་མཛེས་ཟིན་པར་་་་་འགྱུར་རོ། །དབང་མ་ཐོབ་ཅིང་མན་ངག་མེད། །ཚད་དང་མི་ལྡན་འདི་མི་བྱ། །འདི་ལ་ཉེས་པ་ཡོད་པས་ན། །ཤིན་ཏུ་གསང་ཞིང་སྤེལ་མི་བྱ། །སློབ་དཔོན་་་་ལས་ཀྱི་རྡོ་རྗེས་མཛད་པའི་གསང་བ་ཐུགས་ཀྱི་ཕུར་གཅིག་མའོ། །།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ཕྱག་ན་རྡོ་རྗེའི་སྒྲུབ་་ཐབས་སློབ་དཔོན་ལས་ཀྱི་རྡོ་རྗེས་མཛད་པ་བཞུགས། 《སྣར་ཕྱག་ན་རྡོ་རྗེའི་པ་བཞུགས་སོ། །«པེ་»《སྣར་»ཤི། «པེ་»«སྣར་»ན། «པེ་»«སྣར་བའི་རྒྱན། «པེ་»«སྣར་»—དང་རྒྱན། «པེ་»«སྣར་»རྒྱ་གྲམ་གྱི། «པེ་»«སྣར་»ཀྲྀ་ཧྲིw། «སྣར་»ཙནྡྲ། «པེ་»«སྣར་བསྲེག«པེ་»«སྣར་བསྲུང༌། (མ«སྣར་»ཙན། «པེ་»གྲོསྣ་གྲོསྣ། «སྣར་གྲིཉྩ་གྲིཉྩ། «པེ་»«སྣར་»། །ན«པེ་»«སྣར་»རྭ། «པེ་»«སྣར་ག་འགྱེ«སྣར་དྲུས། «པེ་»«སྣར་»ལིངྐ། ཉེ«པེ་»«སྣར་»པྲ་ཚིལ་ལ། སྟེ«པེ་»«སྣར་མའི། «པེ་»«སྣར་འཇིག«པེ་»«སྣར་ལིངྐ་ལ། «པེ་»«སྣར་དབྱུག་པར། ཁྱེ«པེ་»«སྣར་»བཞགམར་«པེ་»«སྣར་རྩ། «སྣར་»ཙནྟ། «པེ་»«སྣར་ཧ་ཧཱུྃ་ཕཊ། མུ་ག«པེ་»«སྣར་བཏང་། «པེ་»«སྣར་»ཏྭཾ། «པེ་»«སྣར་མཾ། @«པེ་»«སྣར་»ཏི། ཏེ«པེ་»《སྣར་བག་གླེ། «པེ་»།
