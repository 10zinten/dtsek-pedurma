གྲོ། །མ་ནིང་འ་མ་ཁང་། ཚོད་བདེ་བ་ਵੀਇਸਧਰਟੌਲਝੂਲਿਥਾਵਾਂਨੂੰਨਾਲਲਲਤੋਂਸ਼ਾਮਲਈKਲਇਕਇਸਦੇਨਾਲਸਰਸ਼ਾਸਲੇ'ਝਸੀ༄༅། །ཕྱག་ན་རྡོ་རྗེའི་ལམ་ཡན་ལག་པོབརྒྱད་པ་ཞེས་བྱ་བ། །ਬਠਿੰਨਾਂਨੂੰਲਲਕਈਲgBIਲਾਲੜਕੀཀ་ལན་ད་ཀྱི་སྒྲ་མ་དག་པ་ལས། །མ་དུ་བལ་པོའི་པ་དག་ལ་དགའ་བ། BRਹੀਮਨfiਸਨੇཀྱིཆ༄༅། །།རྒྱ་གར་སྐད་དུ། བཛྲ་པཱ་:ཎི་མཱནཱཏྟཾ་བི། བོད་སྐད་དུ། ཕྱག་ན་རྡོ་རྗེའི་ལམ་ཡན་ལག་བརྒྱད་པ། འཕགས་པ་ལག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ༑། རིགས་ལྔ་ཁྲོ་བོ་རིགས་གཅིག་པས། །གཉིས་པོ་དབྱེར་མེད་རྣམ་བཞི་་་འདྲས༑། བཅོམ་ལྡན་དེ་བཏུད་སྒྲུབ་པའི་ཐབས། །གཏུམ་པོ་ཁྲོས་པའི་གཞུང་བཞིན་བྱ། །དེ་ལ་དུར་ཁྲོད་ལ་སོགས་པའི་གནས་སུ་གདོང་ལ་སོགས་པ་དག་པར་བྱས་ཏེ་དྲི་ཆུའི་ཆག་ཆག་བཏབ་ལ། རྣལ་འབྱོར་པ་དབང་ཐོབ་པས་སྟན་བདེ་བ་ལ་འདུག་སྟེ་རང་གི་སྙིང་གར་ཧཱུཾ་སྔོན་པོ་བསམ། །དེའི་འོད་ཟེར་གྱིས་བླ་མ་དང་ལྷ་དང་རྒྱལ་བ་སྲས་དང་བཅས་པ་རྣམས་ཐུགས་བསྐུལ་ཏེ་བསོད་ནམས་ཀྱི་ཚོགས་བསགས་པ་ནི། སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་དང༌། །བྱང་ཆུབ་སེམས་་་་དཔའི་ཚོགས་རྣམས་དང་། །བཅོམ་ལྡན་ཕྱག་ན་རྡོ་རྗེ་ལ། །བདག་ནི་གུས་་་་པས་ཕྱག་འཚལ་ལོ། །འདོད་ཡོན་ལྔ་དང་རིན་ཆེན་བདུན། །བཀྲ་ཤིས་རྫས་་་ഇം།
