སྡུག་བསྔལ་མ་ལུས་རབ་ཞི་ནས། །གོས་སྔོན་དངོས་གྲུབ་ཐོབ་པར་ཤོག། འཕགས་པ་གོས་སྔོན་པོ་ཅན་གྱི་རྩ་བའི་རྒྱུད་ཀྱི་འགྲེལ་པ་རྫོགས་སོ། །སྐྱེ་བ་ལྔ་བརྒྱར་བློ་རབ་སྦྱངས། །སྔགས་ཀྱི་ཐེག་པའི་བསྟན་བཅོས་འདི། །དབུ་མའི་ལྟ་:བ་ཡང་དག་སྟེ། །མཁས་བཙུན་རྣམས་ཀྱིས་ཡོངས་སུ་གཟུང་། །སྒྲོལ་མ་གསལ་བ་ཞེས་བྱ་བའི་འགྲེལ་པ་དགེ་སློང་ཀླུའི་བྱང་ཆུབ་ཀྱིས་མཛད་པ་་་་་རྫོགས་སོ། །།།།།ནས་ཀྱི་མ་ཟླ་བ་ལ། ན་ཡེ། :་་་་་་བསྡུར་མཆན། :མ་ག་་བ་དང་། པ་དང་། །བདུད་དེརམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»གོས་སྔོན་ཅན་གྱི་རྒྱུད་ཀྱི་རྣམ་བཤད་དགེ་སློང་ཀླུའི་བྱང་ཆུབ་ཀྱིས་མཛད་པ་བཞུགས། །ཉོ«པེ་»«སྣར་པ། །«པེ་»《སྣར་བ། «པེ་»«སྣར་ན། །«པེ་»«སྣར་»དམན་གྱུར། ::«པེ་»«སྣར་བསྟན་བཅོས། འ«པེ་»«སྣར་»«ཅོ་»བསྔོས། །འིཚེ«སྣར་བརྒྱུད། «པེ་»«སྣར་ཡཱ། ཚེ«པེ་»«སྣར་»པ། ་«པེ་»«སྣར་ཕྱི། «པེ་»«སྣར་»ཚོགས་ཏེ། ཚེ«པེ་»«སྣར་ཕྱི། ཀྱེ«པེ་»«སྣར་དམ། «པེ་»«སྣར་»—པས། «པེ་»དེབད་ཁ《སྣར་གཟིང་། («པེ་»«སྣར་ཡ། «པེ་»«སྣར་པ། :«པེ་»«སྣར་+བ། (ན«པེ་»«སྣར་»+སྤྱི། ཆེ«པེ་»«སྣར་»ཀྱིས། ::ཀྱིང«པེ་»《སྣར་པ། ་པོ«པེ་»«སྣར་ཤམ། ༄༅«པེ་《སྣར་—པ། ་་་(«པེ་»«སྣར་»རྟེན། «པེ་»«སྣར་མི་བལྟོས། «པེ་»«སྣར་པ་དེ། །«པེ་»«སྣར་»ང+པས་རྒྱུ་མེད། «པེ་»«སྣར་»ལྟ། @«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་བདག«པེ་»«སྣར་»ཀྱི། ་«པེ་»«སྣར་»+དང༌། ཀྱེ«པེ་»«སྣར་པ། ་་་་་་ཞེས་པ་«པེ་»«སྣར་། །Q«པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་»+དུ། &«པེ་»«སྣར་ནོ། སོ«པེ་»«སྣར་»འདི། @«པེ་»«སྣར་པ། །«པེ་《སྣར་ཝ་ར་ཎཱ་སེར། «པེ་»«སྣར་བར། །«པེ་»«སྣར་པའོ། །དེ་ལྟར་བདག་བྱེད་པའི། །ན་དང་བར་མ། བྱེད་སྒྲོན་དེ་«པེ་《སྣར་»ལས། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་»དང་། «པེ་»«སྣར་སྟོང་། ལ་«པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་»གཉིས་སུ། «པེ་»«སྣར་དེང་། «པེ་»«སྣར་»གྱི། ཊ་།
