བསྟན་འགྱུར། རྒྱུད། 《༼ཙིཏིབསྟན། པ«པེ་»རྟོགས་བཟློགས། 《སྣར་རྟོག་བཟློགས། «པེ་»《སྣར་བརྟན། @«པེ་»«སྣར་»འགྲུབ། «པེ་»«སྣར་གྱིས། «པེ་»«སྣར་རིམས། «པེ་»«སྣར་པས་རྫོགས་པའི། ཚེ«པེ་»«སྣར་»སྤྱིར། «པེ་»«སྣར་»:::། ་་་་ལན། ་ཚེ@«པེ་»«སྣར་»སྟོན། ༄༅«སྣར་བསྟོན། ་། །།«པེ་»«སྣར་ཡིས། «པེ་»སྣར་རིམས་ཀྱི། «པེ་»«སྣར་»བཞག«པེ་»«སྣར་སྤོས། «པེ་»«སྣར་སྲེགས་འཁོར་ཚོགས། «པེ་»«སྣར་—སྣ། །༽(ི«ཅོ་»ཀྱི།  ེ«པེ་»སྣར་རིམས་ཀྱིས། །«པེ་»«སྣར་རྒྱུད། ་ཚེ«སྣར་པར། («པེ་»《སྣར་»ཡི། 《སྣར་བརླབས། ཚེ«པེ་»«སྣར་»འཇིག«པེ་»«སྣར་»ཡི། «པེ་»སྣར་སྐྱེད། ཁྱེ«པེ་»«སྣར་རབ། ལ་ལས་ཚེ་བྱི«ཅོ་གར། ་«པེ་»《སྣར་»འཆོར། «པེ་བརྟག་པའི། 《སྣར་བརྟགས་པར། ་ཀ?@འི་ཀྱེའི«པེ་»«སྣར་བརྩལ། «སྣར་གཙུགས། ®«པེ་ནི། «པེ་»«སྣར་པའོ། བྱེད་པ::ཀྱིང་ཚོ། འབའ་ཡི་ཚོ་༽ན་ཚོ་ཁོ་སོགས་ལ་ཁོ་ཚོས་གདོཀ་ར་འོ་མ་། དེ་ཁྱེད་ཏུ་བསྡུར་འབྲས་རེའུ་མིགན་རྒྱུད། (ས་པ་ལས་པར། མང་་་་་དཔེ་གཞི། སྡེ་ཚན། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། །རྒྱུད། །རྒྱུད་འགྲེལ། ཚེ། ཕིང་། ཕྱི་རྒྱུད། རྒྱུད། །པས། པོད་རྟགས། ཚིབྱ་ལས་ཚན་གྲངས། ན_བ*ན?བན་ནན་—བཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། །།ལ་དུ་ལ་བ། ਕਵਾਲੀਸੰs། ོ་ང་《་ཀྱི་ལ་བདེ་བ་། །དགེ་བ་དང་། །ཕྱག་ན་པ་དང་དེ་བའི་མི། །
