བསྟན་འགྱུར། རྒྱུད། ༼ཛིཏི་པ་དང་། པར་བཅུག་ལ། མཆོད་རྟེན་གྱི་འོག་ཏུ་མནན་ནོ། །སྐྲང་འབུར་བྱུང་བ་ལ་ཡང་གདབ་བོ། །བདུད་དང་མ་མོ་དང་བཙན་དང་། ་འབྱུང་པོ་ཐམས་ཅད་བཟློག་པའི་རིམ་པའོ། །།།མཚོདཀྱིལ་བཅོམ་ལྡན་འདས་རྡོ་རྗེ་འཛིན་ལ་ཕྱག་འཚལ་ལོ། །ཀུན་དུ་བསྲུང་བ་འདི་ཁྱད་པར་དུ་རྒྱལ་པོ་ལ་བྱའོ། །བསྙེན་པ་གོང་དུ་བྱས་ལ། མཎྜལ་དང་། །ལྷ་བསྐྱེད་པ་དང་། །མཆོད་བསྟོད་བྱས་ལ། མདུན་དུ་དོང་གྲུ་གསུམ་དུ། ངར་ཟན་ལས་རྒྱལ་པོའི་གཟུགས་བྱས་ལ་བཅུག་ལ་ཚོན་སྐུད་ཀྱིས་སྐེ་ནས་གདགས་སོ། །དེ་ནས་ཕུར་པ་དང༌། ཉུངས་ཀར་དང་། །ཉུངས་ནག་དང་། སྐྱེ་ཚེ་དང་། ཐལ་ཆེན་ལ་སྔགས་འདི་བཟླས་པ་མང་དུ་བྱའོ། །སྐྲ་སྤྱི་བོར་བཅིངས་ལ། ཉི་་མ་ས་བླ་ནའི་རྟིང་ལ། ཨེ་པ་ཀ་ར་ཙེ། དི་ནཱ་ག་བར་མ་པ་ལ་སེ་ཏོ་ཡེ་ཞེས་་་བཟླས་ལ། གཟུགས་ལ་ཡུངས་ཀར་ལ་སོགས་པའི་ཐུན་བརྡེག་གོ། །སྙིང་གར་ཕུར་པ་བཏབ་པ་ཐོད་པར་བཅུག་ལ། ཅེ་ཏིའི་འོག་ཏུ་མནན་ནོ། །རྒྱལ་པོ་་་བཟློག་པའི་རིམ་པའོ། །།།་་་་་་ཁོ་བ་ག་བཅོལ་ལྡན་འདས་རྡོ་རྗེ་འཛིན་ལ་ཕྱག་འཚལ་ལོ། །ཀུན་དུ་སྲུང་བར་བྱེད་ན༑བཅའ་གཞི་གོང་དང་འདྲ། །སྐྲ་སྤྱི་བོར་བཅིངས་ལ། །སྔགས་འདི་་་་བཟླས། ཉི་མ་ས་བླ་ནའི་རྟིང་ལ། ཨེ་པ་ཀ་ར་ཙེ། དི་ནཱ་ག་བར་མ། ་པ་ལ་སེ་ཏོ་ཡེ། ཤིག་བྱེར་ནན། གོ་ནན། །ཧ་ན་ཧ་ན་མཱ་ར་ཡ་རྦད་ནན་ཞེས་མང་དུ་བཟླས་ལ། ཐུན་དང་མནན་པ་ཀུན་གོང་བཞིན་བྱའོ། །ཐམས་ཅད་་་་བསྲུང་བའི་རིམ་:པའོ། །།།ཚེ་ཏན་ཏོ། །།
