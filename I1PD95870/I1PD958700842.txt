།མྱོས་དང་རྨོངས་པར་གྱུར་ནས་ནི། །ཚ་བ་ཆེན་པོས་གཟིར་བར་བྱེད། །དེ་་་་བས་དུས་རྣམས་ཐམས་ཅད་དུ། །དམ་ཚིག་འདི་ནི་རྟག་ཏུ་སྲུངས། །ཨོ་ནི་ལོ་བ་ར་དྷ་ར་ས་མ་ཡ་ནཱ་ག་:རཀྵ་རཀྵ་ཧཱུྃ་ཕཊ། དེ་འདྲའི་མཆོད་པ་གསུམ་བྱས་ཏེ། །དཀར་གསུམ་མངར་གསུམ་གཏོར་མ་ནི། །ཆོ་གའི་རིམ་པས་གཏང་བར་བྱ། །དེ་ནི་ཕྱི་ཡི་གསོ་བ་ཡིན། །ཉམས་དགའ་བ་ཡི་གཞལ་ཡས་ཁང་། །འཛམ་གླིང་ཕོ་བྲང་ཆེན་པོ་སྟེ། །རྟེན་ཡོད་གནས་སུ་དམ་དུ་སྦ། །ཐམས་ཅད་ཞི་བར་་་་་་འགྱུར་བ་ཡིན། །གསོ་བའི་དེ་ནི་རིམ་པ་སྟེ། །ཆོ་ག་འདི་ནི་ཤིན་ཏུ་ལེགས། །བདག་གིས་མན་ངག་རིམ་པ་འདི། །གང་ཞིག་འགའ་ལ་ཕན་ཕྱིར་བྱས། །ཟླ་བ་ལྟ་བུ་གང་ཐོབ་དེས། །འགྲོ་རྣམས་རྟོག་པའི་ནད་གྲོལ་ཤོག། བསད་་་་་་གསོ་བའི་ཆོ་ག་རྫོགས་སོ། །།།།་མཚུངས་པ་ཞེས་བྱ་བ། ཕྱོངས་བྱས། ས་འཛིན་བསྡུར་མཆན། དོ་ལུང་དུ་འཇུག་པ་ཁྱེར། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་གསད་པ་བདུད་རྩི་གསོ་་བ་ཞེས་བྱ་བ་བཞུགས། «པེ་རུ་ཚྭ། «སྣར་རུ་མི། «པེ་»«སྣར་གསད། «པེ་»«སྣར་རྩི། «པེ་»«སྣར་གསད། «པེ་»«སྣར་»བསྐུས། @«པེ་»ཡིས། ཉི«པེ་»«སྣར་»བཞག«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་»གཙོ། «པེ་»«སྣར་བཞག«སྣར་»སྟིམ། «པེ་»«སྣར་ཕུལ། «པེ་»«སྣར་ན། «པེ་»ཅིང་བཏུབས། «སྣར་»ཅིང་་་་གཏུབས། «པེ་»«སྣར་བར། །«པེ་»ཅིང་བཏུབ། «སྣར་ཅིང་གཏུབས། «པེ་»«སྣར་བརླབས། «པེ་»«སྣར་»མི། «པེ་»«སྣར་»ཀླུ་བྱད། །«པེ་»«སྣར་»གསོལ། «པེ་»«སྣར་ཕུ། «པེ་བསྒྲགས། «པེ་»«སྣར་»—ཧཱུྃ། «པེ་»«སྣར་བཅད། །«པེ་»«སྣར་ཡཀྵ་ཡཀྵ། «པེ་»«སྣར་བཏང་། «པེ་»«སྣར་གསད། «པེ་»«སྣར་»+པཎྜི་ཏ་དེ་བ་པུརྞ་མ་ཏི་དང་ལོ་ཙཱ་བ་ཆོས་ཀྱི་བཟང་པོས་བསྒྱུར་བའོ། །།།།
