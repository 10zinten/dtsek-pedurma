རྒྱུད། ༼ཙིཏི་འགྱུར་སེམས། །རྟག་ཏུ་བྱེད་བྲལ་གཏི་མུག་མེད། །ཁྲོ་ཁྲོ་བག་ཆགས་:གཏན་བཅོམ་པ། །ཞི་བའི་དམ་པ་མྱ་ངན་འདས། །གཏེར་ཆེན་རང་ལ་ཡོད་ཤེས་་ཀྱིས། །འཇིག་རྟེན་སྒྱུ་མའི་ལས་ཀྱིས་བསྒྲུབས། །སྒྱུ་མ་སྒྱུ་མའི་ཡིད་ཀྱིས་ཏེ༑། འདི་ཀུན་སྒྱུ་མ་ཉིད་ཤེས་ན། །སྒྱུ་མ་ལྟ་བུར་རྣམ་པར་གྲོལ། །སྒྱུ་མར་རྟོགས་པས་བདེན་རྟོག་འཇོམས། །རྟོག་མེད་སྒྱུ་མ་ལ་མི་གནས། །མི་གནས་འགྲོ་ལ་ཕན་བརྩོན་པས། །ཨེ་མ་ནོར་བུ་རིན་ཆེན་རྙེད། །རྟོགས་པ་བརྗོད་པ་སུམ་ཅུ་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །རལ། །རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་མི་ཏྲ་ཛོ་ཀི་དེ་ཉིད་ཀྱི་སྤྱན་སྔར། གནུབས་་ལོ་ཙཱ་བ་བྱམས་པའི་དཔལ་གྱིས་བསྒྱུར་བའོ། །བླ་རེ། ད་ཁྱབ་པས་ན་མ་མ་གནང་བ། བཀོད་པ་སོགས་དབྱེ་བ་བསྒྲུབ་རྒྱུ། ཏཏྟ་ག་ཏེ་ਸਾਕਾਵੀਰੋਇਸਲਈਉਸਨੂੰ། མ་ན་ཀོ་ཏོ་བསྡུར་མཆན། །ཀ་ང་ད་ཚུལ་ལ་རྒྱབ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་རྟོགས་པ་བརྗོད་པ་སུམ་ཅུ་པ་བཞུགས་སོ། །Q«པེ་»«སྣར་ཤསྐྱུ་བ་བྷོ། «པེ་»«སྣར་གཡོ་བའི། «པེ་»«སྣར་»མཁའི༑«པེ་»«སྣར་ན། ན་པ་་། ཀྱེ«སྣར་པ། «པེ་»«སྣར་»མའོ། «པེ་»《སྣར་ཡོངས། «པེ་»«སྣར་བསལ། «པེ་»«སྣར་མཐིང་། །«པེ་»«སྣར་»བུ། «པེ་»«སྣར་འགགས། ༄༅«སྣར་བསྟན། །«པེ་»«སྣར་»སྙམ་དེའང༌། པོ«པེ་»«སྣར་བ། «པེ་《སྣར་རླབས། «པེ་»མཁའ་དེངས། «སྣར་»མཁའ་དངོས། «པེ་»གྱིས། འཁོ«སྣར་»དགའཀྱེ«པེ་»མཆོད། མེ«པེ་»«སྣར་ཤེས། «པེ་»«སྣར་ཁོང་། «པེ་»བརྟན་བཅོམ། «སྣར་བརྟན་བཅོས། «པེ་»«སྣར་རྟོགས། ན་«པེ་»རྟོག«སྣར་རྟག་་་བ། ལོར་ཡུལ་ལ། ཁ་བར་བ་། དེ་ལས་མང་། དང་ལ་ཊི་་འདཙ་། ཁྱོRཁ་ਵਾਂਸਰੀਰਨੂੰਸੁਣਾਕੇਸਾਂ।ਪਰ—།
