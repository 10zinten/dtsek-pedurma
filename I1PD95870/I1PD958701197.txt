བསྟན་འགྱུར། རྒྱུད། །འཛིཏི་ཐམས་ཅད་རྫོགས་པའི་མན་ངག་ཚང་ེ། །བླ་ན་མེད་པའི་:མན་ངག་གི་སྦྱོར་ཀ་སྟེ་བཅུ་གཅིག་པའོ། །ཏ་ད། །པ་པོ། རྫོགས་སྤྱོད་བྱེད་པའི་ཁ་ཟླ་གཏོར་མའི་མན་ངག་སྦྱོར་ཀ་རྣམས། །མན་ངག་རིམ་པ་བཀོད་པ་ཡིས། །བདག་དང་མཐའ་ཡས་སེམས་ཅན་རྣམས། །བླ་མེད་བྱང་ཆུབ་ཐོབ་པར་ཤོག། སློབ་དཔོན་བྷ་བས་མཛད་པ་རྫོགས་སོ། །།།ཁ་པེ། ཀ་ཊ་ཀ་བ་:：ལོ་ཙཱ་བ་དགེ་སློང་ཚུལ་ཁྲིམས་གཞོན་ནུས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་་་་་ཕབ་པ། །།།།པོ་ཤེས་ལས་མེད། ཁྱད་པར་བ། གཏོང་མ་བན་པ་བཏབ་ཏེ་བསྡུར་མཆན། མི་སོགས་རྒྱུ་ཡིན་ལ། པ་རྣབློ་བྱ»བྱ་རྒྱུ། མས་སུ་པར་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་གཏོར་མའི་ཆོ་ག་བཞུགས་སོ། །འ«པེ་»«སྣར་ལཾ་བཱ། ན«པེ་»«སྣར་—། «པེ་»«སྣར་»+པོ། །«པེ་»«སྣར་རྣལ་འབྱོར། «པེ་»«སྣར་»བཅད། འ«པེ་»«སྣར་»བསད། «པེ་《སྣར་»འཚོགས། །་་་་་ཆུ། ས། ་ག་«པེ་»«སྣར་»སྟེངས། «པེ་»«སྣར་བཅད། «པེ་»«སྣར་སྦྱོངས། ཆོ«པེ་»«སྣར་བྲུ༑ས«པེ་»«སྣར་ལ། «པེ་»སྣར་བབ། «པེ་»«སྣར་བསྐུལ་བྱེད་འོད་དུ། «པེ་»《སྣར་བསྐྱེད། །ཛི་ཧཱུྃ། «པེ་»《སྣར་དབབ། བཀྱེ«པེ་པ། འབུ<<པེ་»སྣར་»གསལ། «པེ་»«སྣར་སྦྱོར་ཀ«པེ་»《སྣར་དག་དང་མཐུན། 《པེ་»«སྣར་བས་བྱས་ལ། «པེ་»《སྣར་གྱིས། ས«པེ་»«སྣར་པ། «པེ་»«སྣར་»—ཏ། «པེ་»«སྣར་གསུམ་མམ། «པེ་བྷུ་ཏ་ཊཱ་ཀི་ནི། 《སྣར་བུདྡྷ་ཌཱ་ཀི་ནི། བུ«པེ་»«སྣར་ནི། ང༼(སྟེ«པེ་»«སྣར་»མ། «པེ་»«སྣར་ནི། ལེ«པེ་»«སྣར་ནིཨི་མཾ། «པེ་»«སྣར་»ཀྱི། འབྱེ«པེ་ཨཱ། «སྣར་ཨཱ་ཀ«པེ་»«སྣར་»འཕྲིན། «པེ་»«སྣར་དགུག་པའི་གཏོར་མའི། ་«པེ་»«སྣར་»ཀར་བལྟ། «པེ་»«སྣར་ལ། ས«པེ་»སྣར་ཡང་། ལོ«པེ་»«སྣར་»ལྔ། «པེ་»«སྣར་བསད། :«པེ་»«སྣར་»བཞི། །(«པེ་»《སྣར་སྲེག«པེ་»《སྣར་»བཅིངས་པ་ཡིས། «པེ་»«སྣར་»བཅིངས་པར་བྱ། །བཅིངས་པའི། «པེ་»«སྣར་»ལྔ་འབྲུ་ལྔ། །«པེ་»རྟོགས། 《སྣར་»རྟོགཔ«པེ་»«སྣར་བསམ་སྟེ། «པེ་»—།
