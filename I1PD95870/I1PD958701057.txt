རྒྱུད། ཝོ་འཚིགཆ་ཎོར་་ཎ་ས་གཅོད། སྤྱན་དྲངས་ནས། །ནམ་མཁར་བཞུགས་པ་དེ་ཉིད་ལ། །མཆོད་བསྟོད་བྱས་་་ནས་དམ་ཚིག་གི། །དཀྱིལ་འཁོར་དག་ལ་བསྟིམ་པར་བྱ། །དེ་ནས་བདག་་་་ཉིད་ཤར་སྒོ་རུ། །དང་པོ་ཉིད་ནས་མེས་པོའི་ཚུལ། །བ། །ཤར་སྒོ་རུ་ནི་གནས་པ་འམ། ཞི། །ཡང་ན་ཕྱི་ནས་སྲས་ཀྱི་ཚུལ། །ཉ་། །དེ་ནས་བྱིན་རླབས་དབང་བསྐུར་བྱ། །མཆོད་དང་བསྟོད་པ་རྫོགས་བྱས་ལ། །ཏུག་། སྒྲུབ་པའི་བཟླས་བརྗོད་ཐིམ་་པ་དང༌། །སློབ་མ་དཀྱིལ་འཁོར་གཞུག་པར་བྱ། །།འོ། །ཚོར་བའི་དེ་ནས་ཡོལ་བས་བཅད་བར་དུ། །ལས་ཀྱི་ཆུ་བོ་ྋགཏོར་བ་དང༌། །གདོང་གཡོགས་བཅིང་བ་དང་། ཡོན་ཕུལ་ནས་ནི་མཐོང་བ་དྲི། །ཆུ་བླུད་བཀའ་་་བསྒོ་བྱས་ནས་མེ་ཏོག་དོར། ཁ། མིག་དབྱེ་ལྷ་ངོ་དེ་ནི་རིམ་བཞིན་བསྟན། །ག་མ་དེ་ནས་དབང་བསྐུར་རིམ་བཞིན་བྱ། ཁ། ཡབ་ཡུམ་སྙོམས་ཞུགས་འོད་ཟེར་ལས༑་། །བདེ་གཤེགས་འོད་བསྐུལ་བུམ་པར་ཐིམ། །མ་འདྲེས་པས་ནི་འོད་་ཞུ་ནས། །།སྤྱི་བོར་ཆུ་ཡི་དབང་བསྐུར་རོ། །ལ་། ཚིགས་བཅད་འདིས་ནི་སྦྱིན་པར་བྱ། །སྐུ་གསུང་དམ་ཚིག་ལས་བྱུང་བའི། །ཡེ་ཤེས་བདུད་རྩི་དྲི་མ་མེད། །ལྷ་ཡི་ཆུ་ནི་དག་པ་ཡིས། །མི་བསྐྱོད་པ་ཡི་དབང་བསྐུར་རོ། །ཨོཾ་བཛྲ་ཨཱ་དན་་ཛྙཱ་ན་ཨ་བྷི་ཥིཉྩ་ཞེས་པས་སོ། །དབུ་རྒྱན་མན་ངག་ཤེས་བྱས་ལ། »། འདི་སྐད་དོ། །སངས་རྒྱས་ཀུན་གྱི་:རྒྱལ་སྲིད་དེ། །དབུ་རྒྱན་རིན་ཆེན་དམ་པ་སྟེ། །དབུ་་་་རྒྱན་རིན་ཆེན་མཆོག་འགྲོ་བས། ཞ་། རྫོགས་སངས་རྒྱས་ལ་ཕྱག་འཚལ་ལོ། །ཨོ་ས་མནྟ་ཛྷཱ་ན་རཏྣ་ནཱ་མ་ཧཱུཾ་ཏིཥྛ་ཨ་བྷི་ཥིཉྩ་ཞེས་པས་སོ། །ཚོ། :རྡོ་རྗེའི་མན་ངག་ཤེས་བྱས་ལ། །སངས་རྒྱས་ཀུན་གྱི་རྡོ་རྗེའི་དབང༌། །མི་ཕྱེད་རྡོ་རྗེ་འོད་གསལ་བ། །དམ་ཆོས་འབྱུང་བའི་སྒོར་གྱུར་པ། །རྡོ་རྗེ་ཆོས་ཀྱི་དབང་བསྐུར་རོ། །།།ཨོ་པྲ་ཏ་ཨ་བྷི་ཥེ་ཀ་ཛྙཱ་ན་བཛྲ་ཨ་བྷི་ཥིཉྩ། །དྲིལ་བུའི་མན་ངག་ཤེས་བྱས་ལ། །ཐབས་་་།
