རྒྱུད། ཅིའི་ཁ་ངོ་་། །ཀྱི་ལ:། ད་བ་་༌་་་་་་་་་་་་་་ལ་བ་ਰਕੇਉਸਨਾਲਇਲਾਕਿਆਂਤੋਂབསྡུར་མཆན། མ་ལ་་sམཚན་བྱང་འདི་《སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ཀུ་ལ་ལོ་ཀ་ནཱ་ཐའི་བྱིན་རླབས་དང་རྗེས་གནང་བྱ་བའི་རིམ་པ་ཁ་ཆེའི་པཎྜི་ཏ་བི་མ་ལ་ཤཱིའི་ཞལ་སྔ་ནས་མཛད་པ་བཞུགས། «པེ་»«སྣར་ཀྲུངས་སམ། །«པེ་»«སྣར་བ། «སྣར་པཾ། «པེ་»《སྣར་»—གྱི། །ད་བ་ཡང་་«པེ་»གཏང་། «པེ་»«སྣར་པ། «པེ་《སྣར་»དང་། «པེ་»ཀྱི། «པེ་»«སྣར་ཀྱང་། «པེ་»«སྣར་»+གྱི། །«པེ་»«སྣར་བསམ། «པེ་»སྣར་»ཤུ་ནྱ་ཏྱ། «པེ་པཾ་ལས་པདྨ་ར་་་ལས་ཉི་མའི། 《སྣར་པཾ་ལས་པདྨ་རོ་ལས་ཉི་མའི། དགྱེ«པེ་»«སྣར་»—སུ། «པེ་»«སྣར་—འདུས། («སྣར་»པོ། ་ ེ«སྣར་ཀྲུངས་ཀྱིས། །«པེ་»«སྣར་དྲངས། «པེ་»མཾ། «སྣར་»མཾ། ས«པེ་»《སྣར་བསམ། «པེ་»《སྣར་པའི། ་་ག་ཚོད་༼༽་དུ། །ལྔ་པ་ལ་པ་(«པེ་»«སྣར་»རིམ་པ། «སྣར་»སྐྱེད། «པེ་»«སྣར་བཞག(«པེ་»«སྣར་+ཀྱིས། «པེ་»«སྣར་»—གྱིས། ་འ«པེ་»«སྣར་»ཙ། «པེ་»«སྣར་»སུ། «པེ་»«སྣར་»བརླབས། «པེ་»«སྣར་»པ་ཡི། ਨਸਨੂੰਉਸਵਲབསྡུར་འབྲས་རེའུ་མིགརྒྱུད། ་(། །RAZEZEPTསྡེ་ཚན། །དཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། །སྣར་ཐང་། །ཅོ་ནེ། རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །ལ་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན?བར། ནvན་བ?བ་ན་—ན་ཕྲེང་། །།།་་། །ཚེག་ཁྱིམ། །།།།—།
