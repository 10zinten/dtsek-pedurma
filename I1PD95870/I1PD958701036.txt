ཡེ་ཤེས་ལྷ་ལ་མཆོད་བྱས་ལ། །གསོལ་བ་བཏབ་སྟེ་གཉིས་སུ་བསྲེ། །དེ་ནས་་་བསྲུང་བའི་སྔགས་བཟླས་བྱ། །སོ་:སོའི་རྩ་བའི་སྔགས་འདི་ལ། །བར་དུ་མིང་རྣམས་བཅུག་ནས་ནི། །དེ་ནས་བསྲུང་བའི་སྔགས་གདགས་བྱ། །སྔགས་་་་རྣམས་ཀྱང་ནི་བསྟན་བྱ་སྟེ། /། ཨོཾ་ན་མོ་ནཱི་ལཾ་བ་ར་དྷ་ར་བཛྲ་པཱ་ཎི་མ་ཧཱ་ཡཀྵ་སེ་ནཱ་པ་ཏ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཏྲི་ལོཀྱ་བི་ཛ་ཡ་ཧཱུྃ་ཕཊ། ༀ་མ་ཧཱ་ཀཱ་ལ་རོ་ཥ་ཎ་ཧཱུྃ་་་ཕཊ༑ཨོ་བི་ཤ་བ་ཎ་ཡེ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི་ཧ་ན་ཧ་ན་ཧཱུཾ་ཕཊ། །ཨོཾ་བཛྲ་ཡཀྵ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བེ་ཏ་ལི་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀཱི་ཡ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་བྷི་ར་ཡ་ཧཱུྃ་ཕཊ། པཨོ་བིད་ཏ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་མ་མོ་ེཧ་ར་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་ས་མནྟ་བྷ་དྲི་ཡ་ཧཱུྃ་ཕཊ། ཨོ་:ཤལྟ་གཱི་ཏི་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་པྲ་ཏི་ས་ཏ་མི་ཡ་ཧཱུྃ་ཕཊ༑ཨོཾ་ཤོ་ས་ག་པ་ཡེ་ཙྪཱཧཱུྃ་ཕཊ། ་ཨོ་ར་ཚིག་ཏ་པྲི་ཏ་ཡ་ཧཱུྃ་ཕཊ། ༀ་མ་ས་བྷཀྱི་ནི་ཧཱུཾ་ཕཊ། ཨོཾ་ནི་ལཾ་བ་ར་དྷ་ར་བཛྲ་པཱ་ཎི་ཧཱུྃ་ཕཊ། ཅེས་པ་ནི་དྲག་་་པོའི་སྔགས་བསྡུས་པའོ། །སྭཱ་ཧཱ་ཞེས་པ་ནི་ཞི་བའིའོ། །ཨོཾ་ཞེས་པ་ནི་རྒྱས་པའི་འོ། །ཧོ་ཞེས་པ་ནི་དབང་གིའོ། །རཀྵ་རཀྟ་ཞེས་པ་ནི་བསྲུང་བའི་འོ། །བསྲུང་བའི་སྔགས་ནི་འདི་དག་ལ། །བར་དུ་མིང་ནི་གཞུག་པར་བྱ། །སྲུང་བའི་འཁོར་ལོ་དེ་དག་ཀྱང་། ཎ། བཞི་ཚན་བཞི་ཡིས་བསྲུང་བ་དང་། །གཙོ་བོ་་་ཉིད་དང་བཅས་ཏེ་བསྲུང་། །ཡང་ན་སྡེ་ཚན་སོ་སོར་བསྲུང༌། །གཙོ་བོ་གཅིག་པུས་བསྲུང་བ་ཡིན། །ཡང་ན་སོ་སོར་རེ་རེས་ཀྱང་། །བསྲུང་བའི་ཆོ་ག་ཤེས་་་པར་བྱ། ཡོ་། ཐམས་ཅད་ཀུན་གྱིས་བསྲུང་བ་ཡི། །འཁོར་ལོ་བཅའ་བའི་ཐབས་་་བསྟན་པ། །དང་པོ་འཁོར་ལོ་བརྒྱད་པ་ལ། །རིམ་པ་གཉིས་པ་བཅུ་གཉིས་་་་དང་། །དེ་འོག་རིམ་པ་བཅུ་དྲུག་པའོ། །མཐའ་མ་མུ་ཁྱུད་གཉིས་དང་ལྡན། །དང་པོ་གཙོ་བོའི་རིམ་པས་དགོད། །དེ་འོག་སྡེ་ཚན་གཉིས་པ་དགོད། །ཐ་མ་།
