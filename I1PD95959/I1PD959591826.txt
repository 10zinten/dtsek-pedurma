བསྟན་འགྱུར། སྣ་ཚོགས། ༼༽དཔང་ཀླུ་དབང་ནོར་བུ་ལྟར། །ཉམས་སུ་དོན་བཞེས་དཀོར་མཛོད་དུ་གཞག་་་གསོལ། །ཅེས་པ་དང་བཅས་པ་འདི་དག་ཕག་ལོ་ཟླ་བ་དང་པོའི་ཚེས་གཅིག་་གི་ཉིན། ཐོས་པ་ཅུང་ཟད་འཛིན་པ་མ་ཏི་བྷ་དྲས་གསོལ་པ་ཕྱོགས་ཀུན་དུ་བདེ་་ལེགས་ཀྱིས་གང་བར་གྱུར་ཅིག། ་་་་་། །ཏུ་ཆད་པ་དེ་ཚོ་ཁ་ས་བདག་བ་དང་། །བདག་དང་། ཁོ་པ་ཚོ་ਸਾਥਸੁਕਸਨਪਰੋਸਿਸਨਾਲਉਸਦਾਕਤਲ་བསྡུར་མཆན། ਕੌਸ਼ਨਲਓਟਨਬੂਲਸਨਸਲམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཚུ«པེ་ཡཱ་མ་ཤཱ་ཡུརྦ་དྷ་སིདྷ། «སྣར་»ཡཱ་མ...སིང་ཧ། «པེ་»«སྣར་»སྤྱི། «པེ་»«སྣར་ཡིས། ༼«པེ་»«སྣར་»གཞོན། «པེ་»«སྣར་བྱད། ཚུ«པེ་»«སྣར་»རྟེའུའི། །«པེ་»«སྣར་»མདོངས། «པེ་»«སྣར་»སངས། «པེ་»«སྣར་»ཚོང་། «པེ་»«སྣར་སྐ། །ཚེ«པེ་»«སྣར་»འདྲི། @«པེ་»«སྣར་»བཏུས་པ་ཞེས། @«པེ་»«སྣར་»དམུ། «པེ་»སྐྱེག་བུ། །《སྣར་»སྐྱེགས་བུ། །འ་དང་འགལ་ལོ་ཞེ་བརྒྱད་འབར་བ་དང་འདྲ་བར་མ་/«པེ་»«སྣར་»ཁའི། «པེ་»«སྣར་»སྔ། «པེ་»«སྣར་»མདངས། «པེ་»སྐྲང་བ། ཚེ«པེ་»«སྣར་»རླག@«པེ་»«སྣར་»ཆགས། ༄༅«པེ་»«སྣར་»རྐན། ཉི«པེ་»ཏཾ། ལན་«པེ་»ལ་ཀ་གསོ་བ། 《སྣར་»ལག་གསོ་བ། «པེ་»«སྣར་བཀོལ། «པེ་»«སྣར་»ཚའི། །«པེ་»«སྣར་»འཆང་། ཚུ«པེ་»«སྣར་»སོགས། «སྣར་བསྐྲངས། «པེ་»«སྣར་སྔ། «པེ་»«སྣར་»བཅོས། «པེ་»གདུལ་ཉེས་བཞོན། «སྣར་»གདུལ་ཉེས་གཞོན། ཚེ«པེ་»«སྣར་»བསྣད། ཀྱེ«པེ་»སློན། 《སྣར་»སྨོན། /:ལུ་«སྣར་»དབྱེན། «པེ་»«སྣར་མོ། «པེ་»«སྣར་ལྕོང་། «པེ་»«སྣར་»དམུ། «པེ་»《སྣར་»འགེགས་པ་ཅན། «པེ་»སྣད་མིག་ནད། «སྣར་»སྣད་མིག་སྣད། ཚ«པེ་»སྣར་»ལྟོ། «པེ་»«སྣར་»ཉམས། «པེ་»«སྣར་»མུཉྫ། «པེ་»«སྣར་»མུཉྫའི། ཀྱི《སྣར་ཐོགས། བྱེ«སྣར་»ནས། «པེ་»«སྣར་»ཟེའུ། «པེ་»«སྣར་»སྦོ་དང་བྲང་ནད་ཅེས་བྱ་བ། །«པེ་»«སྣར་»པ། ཚེ«སྣར་ལུགཁྱེ«སྣར་ཟས། ཉེ«པེ་»«སྣར་»—ལྔ་པའི་གནས་སུ་ངས་་་་བཤད་ཅིང་། །ཚེ«པེ་»«སྣར་»ཁངས། །ཡོ་བའི—།
