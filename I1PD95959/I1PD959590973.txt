སྔོ་སྤང་ངོ༌། །ཕྱེ་དང་ཁུར་:བ་ཡང་སྤང་?ལ། ར་ཏི་བཞི་བཞི་ཟླ་བ་དྲུག་ཏུ་་ཟོས་ན། སྐྲ་དཀར་དང་གཉེར་མ་སྤོང་ཞིང་། ཚེ་ཉི་ཟླ་དང་མཉམ་མོ། །མཚོན་གྱིས་མི་ཆོད་ཅིང་བུད་མེད་ལ་ཁུ་བ་འཕྲོ་བར་མི་འགྱུར་རོ། །ཟླ་བ་་དྲུག་གི་བར་དུ་གྲུ་ལ་ཕྱིན་ན་:འབྱིངས་པ་བསྲུང་ངོ༌། །དབང་ཕྱུག་གིས་་བསྟན་པའི་རིན་པོ་ཆེའི་བཅུད་ལེན་གྲུབ་པ་རྫོགས་སོ། །།།ཧ་རི་རྡོ་བར་གྱི་རྣལ་འབྱོར་པ་ཤི་བ་དཱ་ཤ་དང༌། ཨུ་རྒྱན་པས་པུ་ོཏྲར་་་་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། 《སྣར་རྒྱ་ཆ་ཀྱིམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་《སྣར་སེ་ཡ་ན་རོ་ཀ་ཧ་རི་པ་རི་ར་པུཊྚ་ཀ«པེ་《སྣར་»གི། ༄༅། «པེ་»«སྣར་»བཏགས་ཤིང་། @«པེ་»«སྣར་སྐབས། ཚེ«པེ་»སྣར་»ཀཎྜའི། «པེ་»«སྣར་»མ་རིའི། །«པེ་»«སྣར་+བ། @«པེ་»«སྣར་+དང༌། «པེ་»«སྣར་བའི། «པེ་»«སྣར་»—ཆུ། །«པེ་»《སྣར་བཤའ། «པེ་»«སྣར་བསད། «པེ་»《སྣར་བསྲེག«པེ་《སྣར་ཀྱང༌། «པེ་»«སྣར་བསད། ཚེ«པེ་»《སྣར་པོས། («སྣར་»། །མུ«པེ་»«སྣར་བསད། «པེ་»«སྣར་»ཀནྡྷའི། «པེ་»«སྣར་»ཚནྡྷེ། ་«པེ་»«སྣར་»འབྱོངས། ་《སྣར་། །«པེ་»《སྣར་»བི། ེ«པེ་»«སྣར་བསད། འབྱེ«པེ་»«སྣར་»བྷི། དགྱེ«པེ་»«སྣར་བསད། ཉེ«པེ་»«སྣར་སྦྱང་། འ«པེ་»«སྣར་»+ཁུ། གྱི«པེ་»«སྣར་སྦྱང་བ། «པེ་»《སྣར་པོར། ན«པེ་»《སྣར་»ཐམས་ཅད། ་«པེ་»སྣར་བའང་སྤངས། «པེ་»«སྣར་»འཕོ། «པེ་»«སྣར་འབྱིང་བ། «པེ་»«སྣར་བྷུ། །
