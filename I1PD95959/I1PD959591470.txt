བསྟན་འགྱུར། སྣ་ཚོགས། །༼། སྲེག་མཆོད་སྦྱིན། འབུ«པེ་»ཤིས་པར་བརྗོད། 《སྣར་»ཤིང་པར་བརྗོད། «པེ་»«སྣར་རྟོག་བྱེད་དེ། «པེ་»«སྣར་བསྒོ། ཨུ«པེ་»«སྣར་»པོ། ལོ་ཚེ«པེ་ཕྱེད་སྟེན། །«སྣར་བྱེད་སྟེན། «པེ་«སྣར་འདྲའི། «པེ་»«སྣར་»མཁན། དེའི་ཀ་མད་ས་ཞོ། ཆ་ལ་བབ་སྐོར་མེད་པ། «པེ་»«སྣར་»ཞིང་། 《«པེ་»ཀྱིས། «པེ་»«སྣར་པོས། ︽«པེ་»«སྣར་བསྟེན། 《སྣར་པ། ་ ོ«པེ་»«སྣར་»རྒོད། ཚེ«པེ་《སྣར་»གཞོན། «པེ་»«སྣར་བྱང་ཞིང་མཁས། @«པེ་»«སྣར་»འབྱུང་གྲིམས་ཤིང་བཤོར། དགྱེ«སྣར་རིགས། :(ཀྱེ«པེ་»«སྣར་བའི། («པེ་»«སྣར་ཡི། སྟེ«པེ་»«སྣར་»—ལོང་བ་འཕྱེ་བོ་ཐེང་པོ....སྡེབ་སྦྱོར་མཆོག་ཤེས་པ། །«པེ་»《སྣར་»བསྲུང༌། ཡེ«སྣར་»བརྟན། «སྣར་ཐོབ། (ཀྱེ«པེ་»«སྣར་»ཏུ། ་འབུ«པེ་»«སྣར་»སྟེ། གྱི«པེ་»«སྣར་ཞེས་ས། (འགྱེ«པེ་»«སྣར་བརྫངས། ནེ«པེ་»«སྣར་»བདེར། འགྱི«པེ་»«སྣར་»མ་སུ་རཀྵས་....བསྟན་བཅོས་ལས། ཞ«པེ་»«སྣར་»། །དགྱེ«པེ་»«སྣར་+མ་སུ་རཀྵས་བྱས་པ། ཁུལ། «པེ་《སྣར་ཨ་ཙརྱ་དྷརྨ་ཤྲཱི་བཛཱ། Q«པེ་»«སྣར་—ཞུས་ཏེ། །བློན་དང་། ་དང་། ་་་ནང་་ག་ད་བ་པདང་པ་དེདེ་ནང་ན། ཏ་ཁ་ཁ་དང་། །།བསྡུར་འབྲས་རེའུ་མིགཁ་ཚུབ། །བྱེདསྣ་ཚོགས། (ཀྱི། འོ་་༌་་་་པའི་། སྡེ་དགེ། སྣ་ཚོགས། །།པེ་ཅིན། བྱེ་བྲག་རྟོགས། སྣར་ཐང༌། མདོ། ཅོ་ནེ། སྣ་ཚོགས། །སྟེན། །ཡེ་། དཔེ་གཞི། སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ་:། སྟེ། ལེའུ། ལྡེབ། །ཕྲེང༌། །ཚིག་རྐང་། །བ—ན་། བབབ། ན། ན—ན། ཚེག་ཁྱིམ། མ་་་:དུས་དེ་ན། །ཚོགས་དང་། གབ། །
