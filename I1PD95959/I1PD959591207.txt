བསྒྲུབ་པར་བྱ། །དུས་ཀྱི་ལྕེ་ལ་གཞན་སྡུད་བྱ། །སེམས་ཅན་ཐམས་ཅད་་བསད་ལ་བསྒྲུབ། །ཅི་ལ་བབ་ཀྱང་བདག་གིས་ནི། །ྋབིཊྚིའི་གདོང་ནི་སྤང་་བར་བྱ། །དེ་ནས་བིཌིའི་ལོངས་སྤྱོད་བཤད། །བིཏྟིའི་ཆུ་ཚོད་སུམ་ཅུ་སྟེ། །བིཏྟིའི་གདོང་ལ་ཆུ་ཚོད་ལྔ། །བར་གྱི་ཆ་ལ་བཅུ་བཞི་སྟེ། །ཐ་མ་ལ་ནི་བརྒྱད་དུ་བརྗོད། །གཞུག་མ་ལ་ཡང་གསུམ་དུ་དྲན། །མཚན་མོ་གདོང་དགེ་་་མཇུག་མ་ངན། །མཇུག་དགེ་གདོང་གདུག་ཉིན་མོར་དྲནི། །བར་གྱི་ཆ་ཉི་ལ་ནོར་ཉམས་ཤིང༌། །ཐ་མ་ལ་ནི་སྲོག་འཕྲོག་བྱེད། །ྋབདུན་བརྒྱད་ེཉ་དང་བཅུ་གཉིས་ཁྱིལ། །ཉིན་མོའི་བིཊྚི་ཞེས་བྱ་སྟེ༄༅། །གསུམ་བཞི་བཅུ་་དང་བཅུ་གཅིག་པའི། །བིཤྲཱི་མཚན་མོར་ཤེས་པར་བྱ། །།དེ་ལྟར་ཆོ་ག་དབང་་་་ཕྱུག་མ། །བིབྷིའི་འཁོར་ལོར་ཡོངས་སུ་གྲགས། །:：གླུ་ཛ་ཊེ་ཎ་དཀར་ཕྱོགས་ལ༑། :：གུཙྪེ་ཏེ་ཝིདྷ་ནག་ཕྱོགས་བརྗོད། །གསལ་བྱེད་ཚེས་སུ་ཤེས་:བྱ་སྟེ། །དེ་བཞིན་དབྱངས་ནི་ཐུན་མཚམས་ཉིད། །བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལས༈། དེ་ལྟར་བྱས་ན་གཡུལ་ལས་རྒྱལ། །མིང་གི་ཡི་གེ་དང་པོ་ནི། །ཕྱོགས་ཀྱི་དུས་ལ་ཅིའཆར་བར་དྲན། །དབྱངས་སྡེ་མཐར་ནི་སྡེ་ཚན་ལྔ། །ཡ་ར་ལ་ཝ་དེ་བཞིན་ཏེ། །ཤ་ཥ་ས་ཧའི་མཐར་ཐུག་པ། །ཕྱོགས་མཚམས་བརྒྱད་ལ་རིམ་བཞིན་དགོད། །མིང་གི་ཡི་གེ་དབྱངས་:ལས་སོ། །གཅིག་དང་དགུ་ལ་རང་ཕྱོགས་དགེ། །ཀ་སོགས་རང་གི་ཕྱོགས་ལ་ནི། །གཉིས་དང་བཅུ་ལ་དགེ་བར་དྲན། །ཙ་སོགས་རང་གིཕྱོགས་ལ་ཡང་ལ། །གསུམ་དང་བཅུ་གཅིག་དག་ལ་འཛོམི། །ཊ་སྡེ་རང་གི་ཕྱོགས་གནས་པ། །ཁེབཞི་དང་བཅུ་་གཉིས་ཚེས་ལ་འཆར། །ཊ་སྡེ་རང་གི་ཕྱོགས་ལ་ཡང་། །ལྔ་དང་བཅུ་གསུམ་པ་ཡི་ཚེའོ། །རང་གི་ཕྱོགས་སུ་པ་སྡེ་རྣམས། །ཚེས་ནི་དྲུག་དང་བཅུ་བཞི་ལི། །།
