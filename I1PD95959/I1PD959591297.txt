འཁྲུབ། @«སྣར་»སྡོང་རས། ཚེ«སྣར་»མེ་མི། «སྣར་»ཏེ། །«སྣར་»སྐྱེད། @«སྣར་བས་ན། ®«སྣར་བསམས། «སྣར་»འདུག«སྣར་+བཟང་་་པོར། «སྣར་»ཀྱི། ཚ«སྣར་+བུའི། «སྣར་ཐོབ། «སྣར་»+ཞིགཚེ«སྣར་»དགོས། ལོ་《སྣར་དང་། ཉེ《སྣར་»སྟོན་པའང་། །«སྣར་གང་འདྲི། «སྣར་»—སྟེ། 《སྣར་»+དོན་འདི་ནི་འགྲུབ། ®«སྣར་»དོན་འདི། @《སྣར་»pའགྱུར། ༄༅«སྣར་»པས་གསང་། ༄༅«སྣར་»+དགྲ། «སྣར་»ཀྱིན། «སྣར་»དུང་། ་«སྣར་»རྐྱང་བར་འགྲུབ་པ་ནི་དཀར་སྦྱིན། @«སྣར་»ལ། «སྣར་»ལས། ཚ«སྣར་»དྲག«སྣར་»བོས། ༄༅«སྣར་»+དུ། ༄༅«སྣར་»+མྱུར་དུ། «སྣར་»པར། «སྣར་»ལ། «སྣར་»བླངས། «སྣར་ངན། «སྣར་བཞིན་བྱེད་པར་གདའ། གྱི་བ་ཕྱི་མ་ད་པའི་གཏེར་དལོ་བ་བྱ་བ་ལ། བསྡུར་འབྲས་རེའ་མིག་སྣ་ཚོགས། (དཔེ་གཞི། །སྡེ་ཚན། :པོད་རྟགས། སྡེ་དགེ། eས་བ། པེ་ཅིན། །བྱེ་བྲག་རྟོགས། ་གོསྣར་ཐང་། །མདོ། འ། །།།།གོ་བ་དང་། ད་བར་། ཚན་གྲངས། །ང་ཚན་གྲངས། །ལྡེབ། །ནབ་། ན?ན་ཚེག་ཁྱིམ། །།སུ་ཚེག་ཁྱིམ། ཁྲོན་པ་་། །པ་དག་དང་ཆེད་ཀྱི་ལས་བཟ:ཁ་བབ་བཞི། བཀའ་བབ་ལ་དབང་བ་ཞེས་ਅਸਲਵਧੇਲਾਲਈਲਊਨੂੰਲਊਨਾਮਤੋਂ'ਸ਼ਸ਼ਲਵਸ'ਸੰਨਤੋਂਉਨਹਾਸA'ਸਿੰਸਲਲਚ།
