རག་མ་བྱུང་གང་ལ་་་་་བསྡུར་མཆན། བམ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། 《སྣར་ལ་དངུལ་ཆུ་སྒྲུབ་པའི་་ཁ་ན། བསྟན་བཅོས་ཞེས་བྱ་བ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་»ཤཱ་ཏཱ། «སྣར་པོ། 《སྣར་»ཀྱི། «སྣར་»སྤྱོད་པ། @«སྣར་བསྐྱེད། «ཅོ་»སྐྱོད། «སྣར་»ཀྱི་བསྒྲུབས། ཉི«སྣར་»བསྐམས། 《སྣར་པར། «སྣར་»གཉིས། །《སྣར་བཟངས། «སྣར་»སྐྲུད། «སྣར་»གྱུར་པ། གྱི«སྣར་བས་ནི་་་་:མདངས་པར། @«སྣར་»སྟེངས་འདི་ནི། །༄༅«སྣར་»གཙོས། «སྣར་བྱེད་པའང་། @«སྣར་གསེར། དེཚེ«སྣར་ལས། འགྱེ«སྣར་བཞུགས། སྟེ《སྣར་བྷི། ཙེ«སྣར་»ཚལ། །«སྣར་»སྦྱངས་པར། «སྣར་»ཀང་ཀུ་ཤི་ཡིས། «སྣར་»སྦྱངས། «སྣར་»བཟངས། «སྣར་»ལས། «སྣར་རྟ། ༄༅«སྣར་པ། «སྣར་»འདུ། «སྣར་སྲེགས། 《སྣར་»པོ། ཨེ《སྣར་»གཟའ། ཚེ«སྣར་མཁར། ཉེ«སྣར་»བརྡད། ལེ«སྣར་»གྱིས། ཨེ«སྣར་»ཀུན་དྷ། །«སྣར་»དབུག་ཅིང་། ཉི«སྣར་»སྤྱད། «སྣར་»སྦྱང་ཞིང་། ཚ«སྣར་»སྐྱར། «སྣར་»ལྟར་བུ་འཡིས་སྲིམ། «སྣར་རྣོན། ༄༅«སྣར་»བསྲེ། ཉི«སྣར་»དུ། @«སྣར་སེ་བརྡར། «སྣར་»རྣར། ཀྱེ«སྣར་»མར། «སྣར་བསྲེག«སྣར་སྲེགཨེ«སྣར་བའི། «སྣར་པས། ཚེ《སྣར་»ལ་སྦྱང་བ། «སྣར་»འཁྱར། «སྣར་»བླུག་ཅིང་། གྱེ«སྣར་»ཡིས། ནགེ《སྣར་»བཞུགཀྱེ«སྣར་»ལོར་ཞི་ངེ་། འཚེ《སྣར་མཁར། འདུ«སྣར་»སྦྱངས་པའི། བདེ་བ་《སྣར་»ངེ་། «སྣར་»རྣམས། «སྣར་ཤ་རི་ཁ་དྷ། «སྣར་»བཞུག«སྣར་བྱེད། «སྣར་»ངེ་བྱིན་ལ་རྟག«སྣར་»ངེ་། «སྣར་གཤཚེ«སྣར་»གྱི། ཀྱེ《སྣར་»སྦམ། «སྣར་»ཡིས། །ལ་ད་འུ«སྣར་»ངེ་། ཚེ«སྣར་བསྲེས། ཚེ«སྣར་ཚེ་སྦྱར། རྟགས། ་ཨེ«སྣར་»བཟུང༌། ཁྱེ《སྣར་བརྟགས། །ཚེ«སྣར་གཏིང་། །ཚེ«སྣར་»དགའ། །ནགེ《སྣར་བཏུམ། «སྣར་»ཕྱི། «སྣར་དཀད་བར། 《སྣར་»+བ། «སྣར་གནང་། «སྣར་གསལ་འཚམ། 《སྣར་»ཚུལ། «སྣར་བཟུང་། «སྣར་»སླ། 《སྣར་བཟུང་། (《སྣར་»དཔྱད། «སྣར་»།
