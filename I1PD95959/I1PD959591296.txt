བསྟན་འགྱུར། སྣ་ཚོགས། བ་ན་དེ་ལ་ལོ་ང་་་་་ਇਏ"ਉਸਸਿਖਰਉਤਸ਼ਾਖਲཀ་བསྡུར་མཆན། །དཀའི་ཎམཚན་བྱང་འདི་«པེ་»«སྣར་»ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«སྡེ་»«ཅོ་»ལ་མི་བཞུགས་པས་«པེ་»ལ་གཞི་བཟུང་ནས་《སྣར་»དང་དཔེ་བསྡུར་བྱས་ཡོད། «སྣར་»དུ། «སྣར་»གྱི། «སྣར་»+འགྲུབ་པར། «སྣར་ཡི་ཕ་རོལ་ཀྱི་དཔུང་། «སྣར་+དགྲ་མང་ནའང་། «སྣར་»+བྱེ་བ་ལ། «སྣར་སོམ། «སྣར་»གེས། ཨ«སྣར་»མི། O«སྣར་»འཐད། «སྣར་»ཀྱི། ཚེ«སྣར་»འདུགཀྱེ«སྣར་»གཤེད། «སྣར་»འགྲུབ། །«སྣར་»+ལོ། །«སྣར་»ཐ། «སྣར་བརྟགས་གཞིགས། «སྣར་»+སྔར། ཚེ«སྣར་གྲུབ་བོ། «སྣར་»ཀྱི། «སྣར་»ན། «སྣར་»+རྣམ་རྟོག་མ་བྱེད། @«སྣར་»བསམས་ཏེ། འཁྱེ《སྣར་»+མིང་། ཀྱེ«སྣར་»གྱིས། ཉེ«སྣར་»འགའ། ཉེཙེ«སྣར་ཤོལ། ཡོ«སྣར་གཉིས། ཨེ«སྣར་+ན། ཨེ«སྣར་»+ནོར། «སྣར་»ཀྱི། ལ་«སྣར་»བརྟན། @«སྣར་»འགོག«སྣར་»མི་འགྲུབ། «སྣར་»ཀྱི། «སྣར་བསམས། «སྣར་»+ཤིག@«སྣར་བསམས། «སྣར་»—པར། «སྣར་བསམས། ཁྱེ«སྣར་ཨ། ཀྱེ«སྣར་»དྲི་བ་པོ་ཉོན་ཅིགཉེ«སྣར་»འཐོབ། ཉེ«སྣར་»འགྲུབ་ཀྱི། །«སྣར་»རྟས་བགྲོད། «སྣར་»བགྲོད་ཐབས་མེད། ཚེ«སྣར་སྲིད། «སྣར་»+པའི། @«སྣར་»འོང་། «སྣར་»+ལན། ༄༅«སྣར་+བཟང་། ཉི«སྣར་»+ཡོན་ཏན། «སྣར་»ཤིགཅེ«སྣར་»དོན་འགྲུབ་པའི། ཀྱེ«སྣར་»+པས། ཚེ《སྣར་»གྱིས། 《སྣར་»+འགྱུར། ཚེ«སྣར་བྱ་བ། ཚེ《སྣར་»+གཅིག་ཏུ་མ་བྱེད་ཡང་ན་ག་ཤི་ག་དུ། ཚེ«སྣར་»ཞིག་ཁྱེ«སྣར་བར་སྐྱེད་པ་དང་སེམས་ཁྲལ། བྱ་«སྣར་»+ངོ༌། «སྣར་གཟུང་ཞིག«སྣར་»མཆོག«སྣར་ན། ་མ«སྣར་»འགྲུབ། «སྣར་»+ནས། «སྣར་»གི། ཨི«སྣར་»ཀྱི། «སྣར་»+པའི། «སྣར་»དུམ་བ། ཚེ«སྣར་»ཅུང་། @«སྣར་»བསྔལ། @«སྣར་»ཀྱི། O«སྣར་+ཞིག་བསྒྲུབས་པ་ཐད། «སྣར་»ན། ཚེ《སྣར་»འདི། «སྣར་»+ལས་ཆེ། «སྣར་»ཀྱི། @«སྣར་བསམས། «སྣར་»འགྲུབ། «སྣར་+མ། @«སྣར་»མི་འགྲོ། «སྣར་འཇུག«སྣར་»ནའང་། «སྣར་»ཀྱི། ཚེ«སྣར་»ཕྱིད་ཀྱི་ཡོན་ཏན་ཕུན་་་་་ཚོགས་སྐྱེ་རྒྱལ་ཞིང་། གེ«སྣར་»དང༌། ཚེ«སྣར་གེགས། །«སྣར་ཝ། «སྣར་»+བྱེད་ཀྱིན། མི«སྣར་+བདེན་རྫུན། «སྣར་»གྱི། «སྣར་མཐུ་ལས། «སྣར་»བྱེད། «སྣར་»+ཡ། W«སྣར་»ཁོ། «སྣར་ལ། @«སྣར་»བ། ༄༅«སྣར་»+—།
