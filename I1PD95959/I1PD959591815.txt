གི་ཚད་བྱས་པ། །ཆེ་ཚད་ཁོ་ན་ཡིན་པར་བཤད། །རྟ་ལ་སྔ་དྲོ་བླུད་ནས་ནི། །ཕྱི་དྲོ་ཁ་ཟས་སྦྱིན་བྱ་སྟེ། །ཤིན་ཏུ་སྣུམ་ཞིང་:གཡོས་སུ་བྱ། །ཤ་ནི་དེ་ལ་སྦྱིན་པར་བྱ། །ཆུ་ནི་ཉེ་བར་བསྐོལ་བ་དང་། །ལྗང་པ་སྐམ་པོ་སྦྱིན་པ་བསྔགས། །རྟ་ནི་ཆེན་པོ་ཉིད་ལ་ནི། །མཐར་གྱིས་དྲུག་ཅུ་རྩ་བཞི་སྦྱིན། །བར་མ་ལ་ནི་་བཞི་བཅུ་བརྒྱད། །ཉེ་བར་སྦྱིན་པ་ཉིད་དུ་བྱ། །ཆུང་ངུ་ལ་ནི་སྲང་སུམ་ཅུ། །རྩ་གཉིས་ཀྱི་ནི་ཚད་དུ་སྦྱིན། །འདི་ནི་རྟ་ཡི་ནད་གཞི་གསུམ། །འཁྲུགས་པར་གྱུར་ལ་བསྔགས་པ་ཡིན། །ཚིགས་སྐྲངས་པ་དང་སྐྲན་དང་ནི། །དེ་བཞིན་་རྨ་ཡི་ནད་རྣམས་དང་། །།ལུས་རེངས་ཡན་ལག་སྐྲངས་པ་དང༌། །སྐྲངས་དང་གཞང་ཁའི་ནད་རྣམས་དང་། །།ལྟོ་བ་ཡི་ནི་ནད་རྣམས་དང་། །དེ་བཞིན་བད་ཀན་ནད་རྣམས་དང་། །འཐེང་བའི་ནད་ནི་ཐམས་ཅད་:དང༌། ་། མཛེ་ནད་བད་ཀན་ལས་གྱུར་དང་། །རྟ་གང་མེ་དྲོད་ཉམས་པ་དང་། །དེ་བཞིན་འབྱུང་བ་མི་སྙོམས་དང་། །།མཚན་དང་རླིག་པ་སྐྲངས་པ་དང་། །ནད་རྣམས་སྦྱོང་་་་བར་བྱེད་པ་ཡིན། །རྒོད་མ་རྟེའུ་ཤོར་མངལ་མི་ཆགས། །གང་ཞིག་རྟག་ཏུ་བུ་འཆི་དང་། མ། །གང་ཞིག་རྟག་ཏུ་རླུང་གིས་གནོད། པ། རྒོད་མ་དེ་ལའང་སྨན་པས་སྦྱིན། །ལུས་ཀྱི་སྟོབས་དང་གོམས་པ་དང༌། །མེ་དྲོད་ཆེ་ཆུང་་་ཤེས་བྱས་ལ། དེ་། :ཉམས་སྟོབས་ལྡན་ཞིང་ྋགཟུགས་དྲེགས་བདོ། །ན། ཁུར་་་དང་སྡུག་བསྔལ་:བསྟེན་ནུས་སྤེལ། །བྱིན་ན་བདེ་བ་ཉིད་འགྱུར་གྱི། །།།གཞན་ལ་བཅུད་ཀྱིས་ལེན་མི་འགྱུར། །ཤ་ལི་ཧོ་ཏྲའི་གསང་བའི་གནས་ཏེ་་་་བརྒྱད་པ་ལས་གུ་གུལ་གྱི་ཆོ་གའ:ོ། ་དང་། །ཏེང་ཞུ་ཏོཏུ་འབབ་ཀྱིན་། ཡ་བྱེད། ཁ་ཏོ། ན་ས་ག་ཟླ་བ་ཡང་། ད་བར་དུ་ནི་ག་ལ་ཤ་ལི་ཧོ་ཏྲ་དྲང་སྲོང་མཆོག། ལེགས་ཐོས་ཀྱིས་ནི་ཡང་དྲིས་པ། །།
