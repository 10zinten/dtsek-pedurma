བསྟན་འགྱུར། གསོ་བ་རིག་པ། །༼གི༽དམ་པ་གླན་ཞིང་དཀྲི་བར་བྱའོ། །ལེགས་སྨིན་སྐྲངས་པ་ཕྱོགས་གཅིག་འདུས། །:བཙིར་བས་ཉེ་བར་བཙིར་གྱུར་པ། །དྲལ་བར་འོས་ལ་དྲལ་བྱ་ཞིང་། །ཤིན་ཏུ་འཇམ་པོར་གྱུར་པའང་དྲལ། །ཞེས་བྱ་བ་ལ། སྐྲངས་པ་དེ་སྨིན་ཅིང་། རྣགས་པར་གྱུར་པ་དང་། ཕྱོགས་གཅིག་ཏུ་འདུས་ཤིང་བསྒོངས་པར་འགྱུར་ལ༑བཙིར་བའི་སྨན་གྱིས་ཉེ་བར་བཙིར་ཞིང་། །བསྡུས་པར་གྱུར་ནས་དྲལ་བར་འོས་པ་ནི་དྲལ་བར་བྱས་ལ། དེ་བཞིན་དུ་ཤིན་ཏུ་འཇམ་པོར་གྱུར་པ་ཡང་དྲལ་བར་བྱའོ། །གུ་གུལ་ཟར་མ་བ་ལངས་སོ། ་། ཁྲོན་བུ་ཕུག་རོན་རྟུག་པ་་དང༌། །ཐལ་སྨན་འཕྲོད་དང་ཐལ་སྨན་དེ། །སྨིན་པའི་སྨན་གྱིས་བརྟོལ་བར་བྱ། །ཞེས་བྱ་བ་ལ། གུ་གུལ་ལ་སོགས་པའི་སྨན་དང༌། ཐལ་སྨན་གྱི་འཕྲོད་ཀྱི་སྨན་དང་ཐལ་སྨན་ཉིད་དེ། དེ་ལྟར་སྨིན་པའི་སྨན་དག་གིས་ཀྱང་སྨིན་པའི་རྣག་དེ་་བརྟོལ་བར་བྱའོ། །ནང་ན་རྣག་འབྱུང་སྒོ་ཕྲ་ཞིང་། །མཐོ་བར་གྱུར་དང་། གནད་ལ་གནས། །སྣུམ་མེད་བཙིར་བའི་སྨན་གྱིས་ནི། །ཀུན་ནས་རབ་ཏུ་་་བཙིར་བར་བྱ། །ཞེས་བྱ་བ་ལ། ནང་ན་རྣག་ཡོད་པ་དང༌། འབྱུང་བའི་སྒོ་ཕྲ་ཞིང་ཁ་ཆུང་བ་དང༌། རྨའི་ཁ་མཐོ་བར་གྱུར་པ་དང༌། གནད་ཀྱི་གནས་ན་གནས་པའི་རྨ་ལ་ནི་བཙིར་བའི་སྨན་གྱིས་ཐམས་ཅད་ནས་བཙིར་བར་བྱའོ། །བསྐུ་བ་སྐམས་ན་བཙིར་བ་ཡི། །ལས་ལ་བཏང་སྙོམས་:ཉིད་དུ་བྱའོ། །ཞེས་བྱ་བ་ལ། ལྡེ་གུ་སྐམས་པར་གྱུར་ན་ནི་བཙིར་བའི་སྦྱོར་བ་དེ་བཏང་སྙོམས་ཉིད་དུ་་བྱ་ཞིང་བཤོལ་བར་བྱའོ། །དེ་ཡི་རྨ་སྒོར་མི་བསྐུ་སྟེ། །དེ་འདྲས་ནད་ནི་དབྱུང་ཉི་མི་འགྱུར། །ཞེས་བྱ་བ་ལ། རྨ་འདིའི་རྨ་ཁའི་སྟེང་དུ་ནི་བསྐུ་བར་མི་བྱ་སྟེ། དེ་ལྟར་བྱས་ན་རྣག་ཅེས་བྱ་བའི་ནད་དེ་འབྱུང་བར་མི་འགྱུར་རོ། །སྲན་མ་ནས་དང་གྲོ་དང་ནི། །མ་ཤ་མུད་ག་བྲ་བོ་དང་། །སྨན་ནི་སྤྱི་བག་ཅན་དག་གིས། །།
