བསྟན་འགྱུར། སྣ་ཚོགས། ཛིཕྱིན་:ནས་སུ། །དེར་ཡང་ནགས་འདབ་དབེན་པ་རུ། །སྨྲ་བ་བསྙུངས་ཏེ་གནས་པར་བྱ། །དཀར་པོའི་ཆོས་རྣམས་ཡིད་ལ་མི་འཇོག་སྐྱེ་བོ་རྣམས་ལ་ནི། །མི་ཆོས་ཕུལ་དུ་བྱུང་བ་ཡིད་ལ་འཇོག་པ་:ཉིད་ལ་ནི། །ཅུང་ཞིག་དགེ་བར་འགྱུར་སྙམ་ནས་ནི་:འདི་རྣམས་བྲིས་པ་ཡིན། །དགེ་བས་བླུན་པོ་མ་ལུས་གྲོལ་ཞིང་ཡུན་དུ་གནས་པར་ཤོག། ལུགས་ཀྱི་བསྟན་བཅོས་སྐྱེ་བོ་གསོ་བའི་ཐིགས་པ་་་ཞེས་བྱ་བ། སློབ་དཔོན་ྋཀླུ་སྒྲུབ་ཀྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །རྒྱ་གར་གྱི་མཁན་པོ:༔ཤཱི་ལནྡྲ་བོ་དྷི་དང་། ་་:ཞུ་ཆེན་གྱི་ལོ་ཙཱ'ིབ་བན་དེ་ཡེ་ཤེས་སྡེས་བསྒྱུར་བའོ། །།།ཁོ་བློ་ਸਬਰਸੰਮੇਲਨਤਨਸਕਉਬਰਮਿੰਟབསྡུར་མཆན། Pਨਤੋਂਉਸਨੂੰམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་ལུགས་ཀྱི་བསྟན་་བཅོས་སྐྱེ་བོ་གསོ་བའི་ཐིགས་པ་ཞེས་བྱ་བ་བཞུགས། ཞེས་འཁོད། «པེ་»«སྣར་»ན་པུ་ཛ་ན་ཏུ་བྷི་དྷ། @«སྣར་»ཤཱ། «པེ་»«སྣར་ཆོས། «པེ་»«སྣར་»གྱིས། «པེ་»སྣར་པར། @«པེ་»གྱིས། «སྣར་»གྱི། «པེ་»«སྣར་མ་ལྟ་བུ། «པེ་»《སྣར་»ལ་སྙན་སྔགས། ཚེ«སྣར་»དུ་དག་ཚེ«པེ་»«སྣར་»སུ། «པེ་»སྣར་»ན། @«པེ་»«སྣར་»འཕུང་། ཆེ«པེ་»ལན། 《སྣར་»ལན། གྱི«སྣར་»མི། ཁོ@«པེ་»«སྣར་»སྨྲས། «པེ་»«སྣར་»བསྟན་བཅོས། «པེ་»«སྣར་»ཚིག་རྩུབ། །«པེ་»«སྣར་བྱ་སྟེ། འགྱེ«པེ་»«སྣར་»རྫུན། ཅི«པེ་»རྫུན། ་ཚེ«པེ་»«སྣར་»རྫུན། ཁྱེ«པེ་»《སྣར་པ་བྱ་བ་མིན། ཚེ«པེ་»«སྣར་ཡིས། ཨེ«པེ་»«སྣར་ཡ། ལེ«པེ་《སྣར་»ཤེས། ཚེ«སྣར་»སྙན། ནགེ«པེ་»ཤེས་ངན། «སྣར་»ཤེས་ཉན། ནགེ«པེ་»«སྣར་དེ། འཉེ«པེ་»སྣར་གཡོ། ཞཉེ《སྣར་»ཉན། དཔེ«པེ་»«སྣར་»སྤྱང་ཀི། 《སྣར་བལྟོས། གེ«པེ་»«སྣར་ན། འགུ«པེ་»«སྣར་»བཞིན། «པེ་»«སྣར་བ། །«པེ་»«སྣར་»ངན། «སྣར་པོ། W«སྣར་སྡོས། «པེ་པདྨ། ཚུ«པེ་པས་ནི་རྒུན་ནས་་་གསོས། 《སྣར་པ་ནི་རྒྱུན་ནས་གསོས། «པེ་»«སྣར་འབྲས་བུ། ༄༅«པེ་»«སྣར་»ཡོད། «པེ་»«སྣར་»ཡི། «སྣར་»གྱིས། «པེ་»«སྣར་»བསླངས། ཀྱེ«པེ་《སྣར་»རྩ། །ཚི«པེ་»«སྣར་»།
