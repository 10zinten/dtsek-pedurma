བསྟན་འགྱུར། སྣ་ཚོགས། ཛི༽སྐྱེས་བུ་མཁས་པ་འདུས་ནང་དུ། །མགུལ་གྱི་མུ་ཏིག་རྒྱན་བཞིན་མཛེས་པ་ཡིན། །:བརྟགས་ནས་རྒྱལ་སྲིད་རྣམ་སྤོང་བའི། །རྒྱལ་པོ་རིན་ཆེན་ཕྲེང་བ་འདི། །དོན་ཡོད་འཆར་གྱིས་བྱས་པ་སྟེ། །བློ་བཟད་དམ་པའི་རྒྱན་ཡིན་ནོ། །དྲི་མ་་མེད་པའི་དྲིས་ལན་རིན་པོ་ཆེའི་ཕྲེང་བ་ཞེས་བྱ་བ། རྒྱལ་པོ་ཆེན་པོ་སྙན་ངག་་མཁན་གྱི་དམ་པ་སློབ་དཔོན་དོན་ཡོད་འཆར་གྱིས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་ཀ་མ་ལ་གུཔྟ་དང་། ཞུ་ཆེན་གྱི་:ལོ་ཙཱ་བ་དགེ་སློང་་་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པོ། །།།།བསྡུར་མཆན། :མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་དྲི་མ་མེད་པའི་དྲིས་ལན་རིན་པོ་ཆེའི་ཕྲེང་བ་ཞེས་བྱ་བ་བཞུགས། ཞེས་འཁོད། ཉི«པེ་»«སྣར་»བྷི། «པེ་»«སྣར་»ཤོད་ཏྲ་ར་རཏྣ་མ་མ་ལེ། ®«སྣར་»འབྲིས། «པེ་»«སྣར་བདག་གིས། @«པེ་»«སྣར་»འགྱུར། @«པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་གྱུར་པ། «པེ་»ཁས་པས་རིག《སྣར་མཁས་པས་རིགཨེ«སྣར་»གིས། «པེ་»«སྣར་»དུག«པེ་»«སྣར་»བསྲུན། ཚེ«སྣར་སྲོད། ཚེ«པེ་»«སྣར་ན། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»གསང་། «པེ་»«སྣར་»ཟད། «པེ་»«སྣར་»དཔྱོད། (ཚེ«པེ་»བསྒྲིམས། 《སྣར་བསྒྲིམ། ཀྱེ«པེ་»«སྣར་གྲལ། ཉེ«པེ་»«སྣར་»པ་་་ཉིད། འཉེ«པེ་»«སྣར་ཡང་། «པེ་《སྣར་བལྟོས། ཨེ«པེ་»འཇིགས། «པེ་»«སྣར་ཟེར། ཉི«པེ་»«སྣར་»འདོགས། ཉི«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ལེགས་པར་བྱིན་པ་གང་...བྱ་བ་མ་ཡིན་ལ་དགའ་བའོ། །«པེ་»«སྣར་»ལྐུག«པེ་»«སྣར་»བུའི་འཆིང་། @«པེ་»«སྣར་»དུར། «པེ་»«སྣར་»བལྟོས། «པེ་»«སྣར་བསྲུན། །«པེ་»«སྣར་»པའི་སྐྱོན་ནི་ཞན། «པེ་»«སྣར་»ཡི་བཟོ། ཉི«པེ་«སྣར་»བརྩོམ། «པེ་»《སྣར་»འདི་དག«སྣར་པར། @«པེ་»«སྣར་བྱས། «པེ་»«སྣར་བརྩེ། «པེ་»«སྣར་»ཐོབ་བྱའི། «པེ་»«སྣར་»ཡོད་བཞིན། ཆེ«པེ་»«སྣར་»ཉམས། ་ཀྱེ«པེ་»«སྣར་»བུ། ཏེ«པེ་»«སྣར་པར་ལྟ། ཉེ«པེ་»«སྣར་»དྲི་ལྡན། ཨེ«པེ་»«སྣར་»འདི། ཨུ«པེ་»«སྣར་»འགྱུར། །
