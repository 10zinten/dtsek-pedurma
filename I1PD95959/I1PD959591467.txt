རྒྱ་གར་གྱི་མཁན་པོ་:ཨཱ་ཙཱརྱ་དྷརྨ་ཤྲཱི་བྷ་དྲ་དང་། ལོ་ཙཱ་བ་དགེ་སློང་ཤཱཀྱ་བློ་གྲོས་ལ་སོགས་པས་བསྒྱུར་ཅིང་:ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།:ཀཁབདེ་བ་དང་བསྡུར་མཆན། བའི་གས་་:བདེབདསྐྱེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་ལུགས་ཀྱི་་::ཀྱིངབལབསྟན་བཅོས་བཞུགས། ཞེས་འཁོད། «པེ་»«སྣར་»ཏཱི་ཤཱ་ཁཱ། «པེ་»«སྣར་དག«པེ་»::ཀྱི《སྣར་»པའི། «པེ་ལེགས་བསྐྱེད་ཤེས་པ་བསྒྲག་ཕྱིར་དང༌། །ཤེས་པ་བསླབ་ཕྱིར། 《སྣར་ལེགས་་::ཀྱིང་བསྐྱེད་ཤེས་པ་བསྒྲག་ཕྱིར། ཤེས་པ་བསླབ་ཕྱིར། «པེ་»«སྣར་བསྒྲུབ། @«པེ་»«སྣར་བསྐྱེད་ཕྱིར་་བོ་ནི་ཚིག་བཤད། ཉི«ཅོ་སྤྱོད། «པེ་»«སྣར་སྨྲ། «པེ་»《སྣར་བ་ལང་། («པེ་»《སྣར་བོ། །«པེ་»«སྣར་བརྙས་མི། «པེ་《སྣར་བགོ། «པེ་《སྣར་ངགས་མཉན་ཞིང་། «པེ་»། ཞེ《སྣར་»ལ། «པེ་»«སྣར་»གླ། «པེ་»«སྣར་མན་ངག་མཉན། @«པེ་»«སྣར་»འདྲས། འ«པེ་»«སྣར་»+དྲུས། 《སྣར་བརྒལ། ེ«པེ་»《སྣར་འཐུང༌། ཀྱེ«པེ་»«སྣར་མཛེག::ཀ་ལ་(ཉེ«པེ་»«སྣར་»དེ་སྤང་། «པེ་»«སྣར་»ཀུན་ལ་ཡིད་པར་བརྟག། སྐྱེ«པེ་»«སྣར་»ཁ། ཅལེ«པེ་«སྣར་བསྡུད་སྐྱེས་བུས། «པེ་»«སྣར་གསོང་། །བྱེ«པེ་»«སྣར་»མིན་ཚིག་མི་སྨྲ། «པེ་»ཏེ《སྣར་»ལྷགས། ཚེ«པེ་»«སྣར་གོས་ལ་གུས། དགེ«པེ་»«སྣར་མེན། ཀྱེ«པེ་»«སྣར་བཟའ། །«པེ་»«སྣར་»གི། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»བཞིན། «པེ་»«སྣར་»པར་བྱ། («པེ་»«སྣར་»འཕོག@«པེ་»«སྣར་བསྒྲག་སྟེ། @«པེ་»«སྣར་»ཤེས། «པེ་»«སྣར་»ནས། «པེ་»«སྣར་པར། དགྱེ«སྣར་»སློབས། «པེ་»«སྣར་སྣོད། «པེ་»«སྣར་བྱས་ཏེ། །རང་གི་དོན་རྣམས། གྱི«པེ་»«སྣར་བར། «པེ་»«སྣར་བསམས། ་«པེ་»«སྣར་»ལའི། །་«པེ་»«སྣར་བ། ཁྱེ«པེ་»«སྣར་»དོན་བཞིན་དུ་བརྡུལ། ་ ེ«པེ་»«སྣར་»གེ་ཅིག ེ«པེ་»«སྣར་བཞི། ནགེ«པེ་»«སྣར་བརྩམས། གྱི«པེ་»«སྣར་ཅིག་འཉེ«པེ་»«སྣར་རྟག། «པེ་»«སྣར་མ། ་«པེ་»«སྣར་»བར། ་.«པེ་»«སྣར་བསྐྱོད། «པེ་»«སྣར་»ཤུར། །ཚེ«པེ་»«སྣར་མི་ཐུབ། «པེ་»«སྣར་»མ་སུ་རཀྵས་...བསྟན་བཅོས་ལས། «པེ་»«སྣར་»:བཞིན༑*«པེ་»《སྣར་འདོན། «པེ་»«སྣར་བཏང༌། (ཚེ«པེ་»«སྣར་བྱམས། ་དགྱེ«པེ་»:::ང《སྣར་»འཚེ་བར་བྱེད་པ། («པེ་»«སྣར་ངེས། འ«པེ་»«སྣར་»བལྟ། «པེ་»«སྣར་»སྤྱད། །འཕོངས། ཨུ«སྣར་བསྲུངས། «པེ་»«སྣར་བྱ་བ་ཉམས། «པེ་»《སྣར་པའི་འོ་མ། །།
