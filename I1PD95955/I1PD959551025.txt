བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཨི༽་ངོ༌། །ཨེ་ལ་ཀ་ཀོ་ལའོ། །།ནི་ཊི་ཧུ་ལ། ཚང་མང་ཅན་ནོ། །པུཀྵ་ལ་་་སོགས་པ་ལྔ་ནི་སུག་སྨེལ་གྱི་མིང་ངོ༌། །་་་་་་པ་བྱཱ་དྷི་ལས་སོགས་པ་དྲུག་ནི། གུ་ཊ་ཞེས་གྲགས་པའི་དྲི་བཟང་པོའི་མིང་སྟེ། ཀུཊྚ་རུ་ཏའོ། །དེ་ཉིད་ལ་ཀུ་ཊར་གྲགས་པའོ། །བྱཱ་དྷི་ཞེས་པ་ནད་ཀྱི་མིང་་་་མངོན་པར་བརྗོད་པའོ། །།ལ་ད། ལོ་དུ་མ་པོ་ད་འཁོ་ད་དུང་ཅན་མ་ལ་སོགས་པ་གསུམ་ནི། ཙོ་ར་ཀུ་ལ་ཞེས་གྲགས་པའི་མིང་ངོ་། །རུནྡ་ལ་སོགས་པ་དྲུག་ནི། དགའ་བྱེད་ཤིང་གི་མིང་སྟེ། འདི་འོ་མ་དང་་ལྡན་པ་ཤིང་ཨ་ཤཱཏྠའི་རྣམ་པ་ཅན་ཏེ། །ཀཨ་ཤཱཏྠ་དང་རྣམ་པ་མཚུངས་སོ། །དེ་ནི་དགའ་བྱེད་ཤིང་དུ་བརྗོད། ཅེས་འབྱུང་ངོ༌། །སྲིན་མོ་ཞེས་བྱ་བ་ལ་སོགས་པ་དྲུག་ནི། ཙོ་རའི་མིང་ཅན་དྲིའི་རྫས་ཀྱི་མིང་ངོ་། །སྦྲུལ་གྱི་མཚོན་ཆ་ལ་སོགས་པ་བཞི་ནི། སྟག་གི་སེན་པོ་ཞེས་བྱ་བའི་རྫས་ཀྱི་མིང་ངོ༌། ་། ཤུ་ཏི་ར་ལ་སོགས་པ་ལྔ་ནི། སྟོང་བུའི་མིང་ཅན་གྱི་དྲིའི་རྫས་ཀྱི་མིང་ངོ་། །།ཅི་དྲ་ཚོ་འཚོ་དྷམ་ནི་ལ་སོགས་པ་ལྔ་ནི། ་ཧ་སྒྲ་རྣམ་པར་སྒེག་པའི་མིང་ཅན་གྱི་དྲིའི་་་རྫས་སོ། །ཤུཀྟི་ལ་སོགས་པ་ལྔ་ནི་ན་གའི་མིང་ངོ་། །ཤུཀྟི་ཉ་ཕྱེས་འདྲ་བའོ། །ཤང་དུང་འདྲ་བའོ། །ཁུར་རྨིག་པ་འདྲའོ། །ཀོ་ལད་ལ། །ཏུ་ཤུག་གི་འདབ་་མ་ལྟ་བུ་ཞེས་པའི་དོན་ཏོ། །ན་ཁ་སེན་མོ་འདྲ་བའོ། །ལོ། །དང་། །ཀོད། ཀུ་ཊ་སྣ་ཊ་ལ་སོགས་པ་བརྒྱད་ནི། །ལི་ག་དུར་གྱི་མིང་སྟེ། ཀུན་ཌན་ཊ། ལི་གདུར་སྟེ། འཁྱོག་ཅིང་གར་བྱེད་ཅེས་བྱའོ། །དཤ་པཱུ་རཾ།  ི་་གང་བ་བཅུ་་་པའོ༑། བ་ནེ་ཡེ་ནག་སྐྱེས་སོ། །པ་རི་པི་ལ་བ་དང་། སླ་བ་ག་ནི་རྒྱལ་བྱེད་དེ། རྒྱལ་འགྲེའོ། །ཀི་བིནྟི། རྟ་བ་མོའམ། རྒྱལ་ཆེན་མའོ། །་དེ། །གྲིནྡྲ་པརྞ། ལ་སོགས་པ་དྲུག་ནི། གནྡྲི་བ་ནི་མིང་ངོ༌། །མ་རུ་ཏ། །
