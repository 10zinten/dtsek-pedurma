བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཤེས་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །།དེ་ཚེ་དེ་དང་ཆགནས། འདི་ཡི་གེ་བཅུ་གཉིས་པའི་ཚིགས་བཅད་བཅུ་གཉིས་པ་ཡ་མུ་ན་:ཡན་་་་ཆད་༑ཤོང་སྟོན་ལོ་ཙཱ་བ་ཆེན་པོ་རྡོ་རྗེ་རྒྱལ་མཚན་གྱིས་བསྒྱུར་ཏེ་འཕྲོ་ལ་ལུས་པ་ལས་ཤོང་སྟོན་དེ་ཉིད་ཀྱི་བརྒྱུད་པའི་སློབ་མ་དཔང་ལོ་ཙཱ་བ་དགེ་སློང་དཔལ་ལྡན་་བློ་གྲོས་བརྟན་པས། སློབ་དཔོན་ཤཱཀྱ་རཀྵི་ཏས་མཛད་པའི་འགྲེལ་པ་ལ་བརྟེན་ནས། ཡོངས་སུ་རྫོགས་པར་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །་་། ་ཀྱི་དབྱེPaལ་ཅཊ་ཁག་ལ་དབྱེ། དེས་འུ་ཚོ་ལྔ་དཔལ་ཆེན་དབང་བྱེད་བཟང་། ཆུབསྡུར་མཆན། ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»«སྣར་བོས། འི«པེ་»«སྣར་»ལླི་ཀཱའི། «པེ་»«སྣར་»འོས་པ། །«པེ་»«སྣར་རབ་ཏུ། «པེ་»«སྣར་»མངོན་པར། «པེ་»«སྣར་—བ། །«པེ་»《སྣར་»རྩིའི་བདུད། འུ«པེ་»«སྣར་»ཡི། ན«པེ་»«སྣར་»ཡིད། །༄༅«པེ་»«སྣར་བས། དང་། «པེ་»«སྣར་གཅོད། དང་དང་། ནང་བ། ནཔ། །ཚེ་དང་བར་བྱ་«པེ་»«སྣར་—ངེས་པར། ཉེ«པེ་»«སྣར་བྱ་བར་འགྱུར། «པེ་»«སྣར་»ཡིས། བ«པེ་»«སྣར་»ཙུ། ་ས«པེ་»《སྣར་»དག་ནི། *«པེ་»«སྣར་»རྒྱབ། «པེ་»«སྣར་རྗེ། «པེ་»《སྣར་བསྒྲུབ། །་་་་་་:་ནས་དང་པོ་དང་། པའི༼རབཀར«པེ་»«སྣར་རྗེ། འི«པེ་»«སྣར་»ཤིན་ཏུ། «པེ་»«སྣར་»ལས། «པེ་«སྣར་»པ། ས«སྣར་»བསྒོམ། «པེ་»«སྣར་སྤང་། «པེ་»«སྣར་»བོས། །«པེ་»«སྣར་»—ཡང༌། «ཅོ་»དང་། །«པེ་»«སྣར་པའི། ་་སྤྱི། ་«པེ་»«སྣར་པའི།  ེ«པེ་»«སྣར་»བུས་བཀུར། ི«པེ་»«སྣར་ནི། «པེ་»«སྣར་»བཟངས། «པེ་»«སྣར་བཅའ། «པེ་»«སྣར་འཁུར། «པེ་»«སྣར་པ། «པེ་བོས། «སྣར་བས། «པེ་»«སྣར་»གི། དོ«པེ་»«སྣར་»གི། ལོ་ངོ་ཀྱི་/«ཅོ་ཡངས། «པེ་«སྣར་ལ། «པེ་ལ་བསྟེན། «སྣར་ལ་བརྟེན། «པེ་»«སྣར་རིང༌། «པེ་»སྣར་»ཀྱི། «སྣར་བརྟེན། «སྣར་བསྒྲུབ། «པེ་»«སྣར་རྣམ། «པེ་»«སྣར་»ཡི། ་དགྱེ«པེ་»«སྣར་པ། །
