ཁ་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཚ་སྐྱོབ་སནྟ་ཏིརྒོ་ཏྲ་ཛ་ན་ན་ཀུ་ལཱ་རྱ་བྷི་ཛ་ནཱནྡྲ་ཡོ། །ཐཾ་ཤོཉནྟ་བཱ་ཡ༔རྒྱུད་དང་རིགས་དང་ཆོ་འབྲང་དང༌། །རུས་དང་རྩ་ལག་རྗེས་སྐྱེས་དང༌། :། ཐཾ་ཤ་རྩ་བྲིད་བརྒྱུད་སནྣཱ་ནོ་བཎྞསཱུརྦདྨ་ཎཱ་ད་ཡ། །རཱ་ཛ་བཱི་ཛཱི་ར་ཛ་ཁཾ་ཤཾ་བཱི་ཛུསྡུ་ཀུ་ལ་སམྦྷ་བ། །པའོ༑། བཎ་བྲམ་ཟེ་ལ་སོགས་པ། །རྒྱལ་པོའི་ས་བོན་རྒྱལ་པོའི་རྒྱུད། །རིགས་ལས་བྱུང་བ་ས་བོན་ནོ། །མ་ཧཱ་ཀུ་ལ་ཀུ་ལཱི་ནཱརླ་ས་བྱ་སཛྫ་ན་ས་དྷ་བ། །བྲ་ཧྥ་ཙཱ་རིགས་ཆེན་རིགས་བཟང་འཕགས་རིགས་དང༌། །མཆོག་དང་སྐྱེས་བུ་དམ་པ་ལེགས། །ཚངས་པར་སྤྱོད་རཱི་གྲོ་ཧཱི་བཱ་ན་པྲསྦོ་བྷི་གླུ་ཉྩ་ཏུཥཱ་ཡེ། །ཨཱ་ཐ་མོཉ་དཱི་ཛཱ་ཏྱ་དང་ཁྱིམ་གནས་དང༌། །ནགས་ན་གནས་དང་དགེ་སློང་དང་། །རྣམ་བཞི་ལ་གནས་མོ་མིན་ནོ། །གྲ་ཛཾ་མ་བྷཱུ་དེ་བ་བཱ་ཌ་བཱ། །བི་པྲ༔་བཱསྒྲ་ཎཉསོ་ཥཥརྨོ་གཉིས་སྐྱེས་སྔོན་སྐྱེས་ས་ཡི་ལྷ། །ཚངས་པའི་བུ་དང་བྲམ་ཟེ་ཚངས། །འདི་དག་ལས་ནི་དྲུག་ལྡན་པ། །ཡཱ་གཱ་དི་བྷརླུ་ཏ། །བི་དྷཱརྨི་པྲཕྱིརྡོ་ཥ་ཛྷས་ན་སུ་མཆོད་སྦྱིན་ལ་སོགས་ནུས་པའོ། །རིག་ལྡན་རྣམ་གསལ་ཉེས་པ་ཤེས། །མཛེས་ལྡན་བློ་བཟངས་ཀུན་དྷཱི་ཀོཉབི་དོ་བུ་དྷ། །དྷི་རོ་མ་ནཱི་ཏཱི་ཛྷ་བྲ་ཛྷསཾ་ཥཱ་བཱ་ན་པཎྜི་ཏརིག་དང་། །རྟོགས་ལྡན་བརྟན་པ་སེམས་ཤེས་དང་། །ཤེས་ལྡན་གྲངས་ཅན་ཤེས་རབ་ཅན། །པཎྜི་ཏ་ཀ་བི། །དྷཱི་མཱན་རིཀྲྀ་ཏཱི་ཀྲྀཏྟིརླ་བྷ་བཙྪེབི་ཙ་ཀ་ཎ། ?། དྷུ་ར་དནི་དཱིགྷ་དང་ཉེས་པ་ཤེས། །བློ་ལྡན་དཔའ་བོ་གོ་བ་ཅན། །མཁས་པ་གྲགས་ཅན་རྣམ་པར་དངས། །རིང་པོ་དདྷཱི་ཤོ་ཏི་ཡཙླི་ད་སོ་ས་མོ། །ཨུ་པཱ་དྷ་ཡོཉདྷ་ཡ་ཀསྱཱཧཱནྣི་མཐོང་དང་རིང་པོ་ཤེས། །ཐོས་ལྡན་སྡེབ་སྦྱོར་ཅན་དག་མཚུངས། །ཨུ་པ་དྷཱ་ཡ་མཁན་པོའོ། །དབང་ཥ་ཀཱ་དི་ཀྲྀདྷུ་རུ། །མནྡྲ་བྱཱ་ཀྵ་ཀྲྀ་དཱ་ཙཱརྨ་ཨཱ་དེབྦ་ཏུ་དྷཱ་རེ་བྲ་ཏཱི། །སོགས་བྱེད་པ་བླ་མའོ། །སྔགས་འཆང་བྱེད་པ་སློབ་དཔོན་ཡིན། །ཨཱ་དེབྦེ་འཛིན་བརྟུལ་ཞུགས་ཅན། །ཡབྷྱཱ་ཙ་ཡ་ཛ་མ་ན་སཱུ་ས་སོ་མ་བ་ཏིདི་་ཏ། །ཨི་ཛཱ་ཤཱི་ཨོ་ཡཱ་ཡ་ཛ་མ་ན་མཆོད་སྦྱིན་ལྡན། །ཟླ་བ་བཏུས་པའི་དིཥི་ཏ། །མཆོད་སྦྱིན་མང་བྱས་མཆོད་སྦྱིན་ལྡན། །།
