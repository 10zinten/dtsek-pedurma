་་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་ཎ་ཡོན་ཏཅོ་ཀ༌ཝཏྟཀ་པིཀ་ཁཝོ་ཌའབྱུང་ལིང་ལ། །ཀཱི་ཎཱི་ཡཱཏྲ། ཀཱི་ཎཱི་ཡཱ་ཏཱ། ཀཱི་ཎཱི་ཡུསོགས་དང་། ༈ཤིས་་་་་བྱེད་ལོཊ་ལ། ཀཱི་ཎཱ་ཏུ། །ཌཀཱི་ཎཱི་ཏཱཏྲི། ་ཀཱི་ཎཱི་ཏཱཾ། ཀཱི་ཎནིུ། ཀྲི་ཎཱི་ཧི། ཀཱི་ཎཱི་ཏཱཏྲེ། ཀཱི་ཎཱི་ཏཱཾ། ཀཱི་ཎཱི་ཏ། ཀཱི་ཎཱ་ནི། །ཀཱི་ཎཱ་བ། ཀཱི་ཎཱ་མ། ༈དེ་རིང་མ་ཡིན་པ་ལདྲ་ལ། ཨ་ཀཱི་ཎཏྲི། ་ཨ་ཀཱི་ཎཱི་ཏཱཾ། ཨ་་ཎན། ཨ་ཀཱི་ཎཱིཨ་ཀཱི་ཎཱི་ཏཾ། ཨ་ཀཱི་རྟི་ཏ། ཨ་ཀཱི་ཏཱཾ། ཨ་ཀཱི་ཎཱི་བ། ཨ་ཀཱི་ཎཱི་མ། །ཁྲོ་། e༈ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལས་བརྟ་མ་ན་ལཊ་ལ། ཀཱི་ཎཱི་ཏེ། ཀཱི་ཎཱ་ྋཏེ། ཀཱི་ཎ་ཏེ། ཀཱི་ཎཱི་ཥ། ཀཱི་ཎཱ་ཐེ། ཀཱི་ཎཱི་ཏེ། ཀཱི་ཎེ། ཀཱི་ཎཱི་བ་ཧེ། ཀཱི་ཎཱི་མ་ཧེ། ༈ཆོག་:འབྱུང་ལིང་ལ། ཤྲཱི་ཎཱི་ཏ། ཀྲྀ་ཎི་ཡཱ་ཏཱ། ཀཱི་ཎཱི་རན་སོགས་དང་། ༈ཤིས་བྱེད་ལོཊ་ཅེལ། ཀཱི་ཎཱི་ཏཱཾ། ཀཱི་ཎཱ་ཏཱ། ཀཱི་ཎ་ཏྭཾ། ཀཱི་རྟི་། ཀཱི་ཎཱ་ནྡྷེཐཾ། ཀཱི་ཎཱི་དཾ། ཀཱི་ཎི༄༅། །ཀཱི་ཎཱ་ཙེབ་ཧི། ཀཱི་ཎཱ་མ་ཧི། །ད་བ་ས་ཀྱི་ཁ་ཚ་ཏུ་ཁ་ལོ་དེ༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་ཀྲྀ་ཎི་ཏ། །ཨ་ཀཱི་ཎཱ་ཏྭཾ། ཞུ་ཨ་ཀྲྀ་ཎ་ཏ༑ཨ་ཀཱི་ཎཱི་ཐ། །ཨ་ཀཱི་ཎཱ་ཐཾ། །ཨ་ཀཱི་ཎཱི་ཧཱུཾ། ཨ་ཀཱི་ཎི། །ཨ་ཀཱི་ཎཱི་བ་ཧི། ཨ་ཀྲི་ཎཱི་མ་ཧི། མྱ་ལོ་ཀ་བ་པ། ཁོས་ཁོ་ཚོའི་ཁོང་ཁྲོ། །ཡང་ཁོ་༈དེ་བཞིན་དུ་གྲཧྲ་ཉེ་བར་ལེན་པ་ལའོ་ཞེས་པའི་བྱིངས་ལ་པ་རསྨི་ོཔ་ད་རྣམས་སྦྱོར་བ་ལས། ༈གྲ་ཧཱུྃ་ཀྲི་ཏི་ཙ། །གྲ་ཧ་རྣམས་ཀྱི་ཀྲ་ང་འགྲོ་བ་ལའང་ངོ༌། །བྱིངས་གྲཧྲི། ཛཱ། བྱེའྲི། དེའྲ། བྱདྷ། བདྲ། བས། བཙཱ། པྲཚྭ། བྷསྔ། བཱ་ལ་སོགས་པ་རྣམས་ཀྱི་ཀྲ་ཡིག་དང་དྲ་ཡིག་འགྲོ་བའི་མིང་ཅན་ཕ་་་རོལ་ལ་སཾ་པྲ་ས་ར་ཎ་སྟེ་སླར་ཡང་དག་པར་བསྒྱུར་བར་བྱ་ལ། ལྔ་དེ་ཡང་ཡ་ཡིག་ནི་ཨི་ཡིག། བྲ་ཡིག་ནི་ཨུ་ཡིག། རྔ་ཡིག་ནི་ཛཾ་ཡིག། ཁྲ་ཡིག་ནི་ལ་ཡིག་་་རྣམས་སུ་འགྱུར་ཏེ། ནཱ་བི་ཀཱ་ར་ཎ་སྦྱར། གྲ་ཧའི་ར་ཡིག་དབྱངས་བཅས་རེ་ཡིག་ཏུ་བསྒྱུར་བས། གྲོ་བྷྱ་ཞེས་པ་ལ་བརྟ་མ་ན་སོགས་སྦྱོར་བའི་ཆོ་ག་རྣམས་་་—།
