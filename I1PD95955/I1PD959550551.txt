མཁ་བས་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་ཀཀ་ཡི་ཏོ༑། ཨི་རཱ་བྷུ་བཱཀླུ་རཱསླུ་སྙནྟནི་ནི་དཱ་པྲ་མཱི་ལ་ཡོ། །དྷཱ་ཏཱི་སཱུ་དུ་པ་མཱ་ཏཱ་འགྲོ་བའོ། །ཨི་རཱ་ས་དང་ངག་ཆང་ཆུ། །ཏནྡྲི་གཉིད་དང་རྨུགས་པའོ། །དྷཱ་ཏཱི་ཉེ་བའི་མ་མ་པི་ཀྱི་ཏི་ར་པཱུ་མ་ལ་ཀྱ་པི། །ལ། ཀླུ་དཱ་བྱངྒཱ་ན་ཊཱི་བི་བྱ་ས་ར་གྷ་ཀཎྜ་དང༌། །ས་དང་སྐྱུ་རུ་ར་ལ་ཡང༌། །ཀླུ་དཱ་ཡོན་པོ་གར་མཁན་མ། །སྨད་འཚོང་དང་ནི་སྦྲང་རྩི་ལ། །ཀཎྜ་ཀཱ་རི་ཀཱ། །ཏྲི་ཥཱ་ཀཱུ་རེཉདྷ་མེཉ་ལྕེ་པི་ཀླུ་ཏྲ་མཱ་ཏྲ་པ་རིཙྪ་དེ། །ཨལྤ་ཙ་པ་རི་མཱ་ཎེ་སོ་མཱ་ཀཱ་རི་ཀའོ་གསུམ། །ཐང་ན་སྨོད་ཉུང་གླུ་དྲའོ། །མཱ་ཏྲ་ཡོ་བྱད་ཉུང་དང་ཚད། །མཱ་ཏྲ་ཀུན་ཊཾ་ཀཱནྟེཉབ་ཉྫཱདྷཱ་ར་ཎེ། །ཨཱ་ལེ་ཁྱུཉྩརླ་ཡོཕྱི་ཏྲ་ཀ་ལ་ཏྲ་དབྱི་མགོ་ཟེར་ཐོ་ཎི་བྷཱརྨ་ཡོ། །ཡོ་གྱ་དང་ངེས་བཟུང་ལ། །ཙ་ཏྲ་ཀུན་བྲིས་ངོ་མཚར་ཅན། །ཀ་ལ་ཏྲ་ནི་ཆུང་མ་ཆུ། །རིགས་དང་སྣོད་ནི་བྷཱ་ཛ་ན་ཡོཔཱ་ཊཾ་པ་ཏཱཾ་བཱ་ཧ་ཎ་པཀྵ་ཡོ། །ནིརྡེ་ཤ་གྲནྠ་ཡོཤཱམྲྀ་ཤསྔ་མཱ་ཡུ་དྷ་པཱ་ཏྲ་ངོ༌། །པ་ཏྲ་བཞོན་པ་འདབ་མའོ། །ཤཱ་སྔོ་ངེས་བསྟན་དང་ནི་གཞུང་། །།ཤ་སྔ་མཚོན་དང་ལྤགས་ལོ་ཧ་ཡོ། །སཛྫ་ཊཾ་ཤུ་ཀ་ཡོནེ་ཏྲཀྵེ་ཏྲ་པཏྟཱིཤ་རཱི་ར་ཡོ། །མུ་ཁཱ་གྲེ་ཀྲོ་ཌ་ཧ་ལ་ཡོདག་ལའོ། །རྩ་ཕྲན་གོས་མིག་ནེ་ཏྲའོ། །ཀྵེ་ཏྲ་ཆུང་མ་ལུས་དག་གོ། །།ཕག་སྣ་གོས་བཤོལ་པ་ཏྲའོ། །པོ་ཏྭཾ་གོ་ཏཱཉྩ་ནཱམི་ཙ། །སཏྟཱ་མཱཙྪ་ད་ནེ་ཡ་ཛྙཱ་ས་དཱ་དཱ་ནེ་བ་ནེ་བི་ཙ། །གོ་ཊཾ་རིགས་དང་མིང་ལ་ཡང༌། །སཾ་དྲི་གོས་དང་མཆོད་སྦྱིན་དང་། །རྟག་ཏུ་སྦྱིན་དང་ནགས་ཚལ་ལོ། །ཨ་ཛི་རཾ་བི་ཥ་ཡེ་ཀཱ་ཡེ་སྱམྦ་རཾ་བྱོནྣི་བཱ་ས་སི། །ཙཀྲྀ་རཱབྦེ་པྱཀྵ་རནྟུ་མོ་གླུ་ཨ་ཛི་རོ་ནི་ཡུལ་དང་དབུས། །ཨམྦ་རཾ་ནི་ནམ་མཁའ་གོས། །ཙ་ཀྲྀ་ཕ་རོལ་དམག་དང་ཡུལ། །ཨཀླུ་པི་གླི་ར་མསླུ་ཙ། །སུ་བནྡྷེ་པི་ཧྥུ་རི་ཙནྡོ་དྲོ་དྷཱ་ར་མ་ཏྲེ་པི་གོ་པུ་རཾ། རྗེ་། ར་ནི་གྲོལ་བའོ། །ཀཱི་ར་ཆུ་དང་འོ་མ་ལའོ། །གསེར་ཀྱང་ཧྥུ་རི་ཙནྡོ་གཉིས། །སྒོ་སྲུང་དང་ནི་བ་གླང་ཁྱིམ། །གུ་ཧཱ་ད་ག་ཧཱ་རེ་དྭེ་ར་ཧོནྟི་ཀ་མུ་པ་ཧཱ་རེ། །པུ་རོཉདྷི་ཀ་མུ་བརླ་གྲྭ་ཕུག་དང་ང་རྒྱལ་ག་ཧཱ་རེ། །དབེན་གནས་ཨུ་པ་ཧཱ་རེའོ། །སྔོན་ལྷག་སྟེང་མདུན་ཨ་གྲྭའོ། །ཏྱཱ་གཱ་རེ་ན་ག་རེ་ཕུ་རཾ། །མནྡི་རཾ་ཙཱ་ཐ་རཱཉྫཉསྟི་བི་ཥ་ཡེ་སྭཱ་དུ་ས་དྲ་བེ། །ཁྱིམ་དང་གྲོང་ཁྱེར་པུ་རཾ་ངོ༌། །འདུ་བ་དེ་ནས་ཡུལ་འཁོར་ནི། །མོ་མིན་ཡུལ་དང་བར་དུ་གཅོད། །།
