་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཨི༽སྒྲོལ་ཌའི་ཚོང་དང་ཁ་འཐབ་པོ། །།་ཎ་ལན་གནང་བ་གང་ཞི་རྒྱུའང་། ད་ལུང་། ་་་་་་་དེ་ནས་ཚིགས་བཅད་ཕྱེད་ཀྱི་དམྱལ་བ་སྤྱིའི་མིང་སྟེ། ནཱ་རཀ་དང་། ནི་ར་ག་དང་། ནི་ར་པ་རྣམས་དམྱལ་བའམ་འདོད་བྲལ་ལོ། །དུརྒ་ཏིང་ན་འགྲོའོ། །དམྱལ་བ་དེའི་དབྱེ་བ་ནི། ཏ་པ་ན་ཚ་བའོ། །།།ཨ་བཱི་ཙི་ནི་མནར་མེད་དོ། །རོ་རབ་དང་། །མ་ཧཱ་རོ་རབ་ནི། ངུ་འབོད་དང་ངུ་འབོད་ཆེན་པོའོ། །།ཀཱ་ལ་བུ་དྲ་ཐིག་ནག་པ་ཅན་ནོ། །།།།ཞི་བ་ལོ་ཟླ་བ་དྲུག་དང་། །གཏེར་པའི་སྡུག་བསྔལ་གྱི་མིང་ལ། ཡཱ་དྲ་ན། །བརྣག་དཀའ་བའོ། །།ཏི་བྲ་བེ་ཏན། མི་ཟད་པའི་ཚོར་བ་ཅན་ནམ། ཚོར་བ་དྲག་པའོ། །།པཱི་ཊ། གཟིར་བའམ། ཟུག་ངུའོ། །བཱདྷ་ཟུག་ཟེར་རོ། །བྱཱ་ཐ། ན་བའམ། སྡུག་བསྔལ་ལོ། །།དུཁཾ་སྡུག་བསྔལ་ལོ། །ཨཱ་མཱན་སུཾ་ཡིད་གྲོགས་སོ། །བྲ་པཱུ་ཏ་ཛཾ། བུ་སྐྱེས་་་་པའི་སྡུག་བསྔལ་ལོ། །ཀཊཾ་དང་གྲོཙྪཱ་དཀའ་བའམ་སྡུག་པའོ། །ཨཱ་བྷཱ་ལཾ། འཇིགས་པའི་ལེན་ཞེས་བྱའོ། །ཀྐ་། །བྱེ་བ་ཁོ་ན་བྱང་དེ་ནས་ཚིགས་སུ་བཅད་པ་ཕྱེད་དང་གཉིས་ཀྱིས་རྒྱ་མཚོའི་མིང་སྟེ། །དེ་ལ་ས་མུ་དྲ་ནི་རྒྱ་མཚོ་སྟེ། སམ་སྔོན་མ་ཅན་གྱི་ཨུནྟི་ལ། །དྲུ་ར་ལ་སོགས་པ་རྣམས་ལས་སོ་ཞེས་པས་ར་ཀ་རྐྱེན་དུ་བྱས་པའོ། །ཨ་བྷིཆུ་གཏེར་ཏེ། །ཨཔ་སྔོན་པོའི་དྷཱ་ལ་ཀི་རྐྱེན་དུ་བྱས་པའོ། །།ཨཀླུ་པཱ་རཿཟླ་བས་འཕེལ་ཞེས་བྱའོ། །སཱ་རཱ་བཱ་རཿམཐའ་མི་མངོན་ཞེས་བྱ་སྟེ། ཨ་དྲྀ་ཤྱ་པཱ་ར་ཨི་ཏྱརྦ། ཕ་མཐའ་མི་མཐོང་བ་་་་ཞེས་པའི་དོན་ནོ་ཞེས་སོ། །པ་རིརཱཏི། ཆུ་ཀླུང་བདག་གོ། །ཨདནྣཱན། ཆུ་ལྡན་ཏེ་མ་རུས་རྐྱེན་དུ་བྱས་པའོ། །ཨུ་ད་དྷི། །ཆུ་གཏེར་རོ། །སིནྡྷ་རྒྱུན་ལྡན་ནོ༑། པར་སཱན་ཆུ་ལྡན་ནོ། །སག་ར་དུ་གཅན་པའོ། ་། ཨརྣ་བ་བརྒྱ་མཚོའོ། །རཏྭཱ་ཀ་ར། རིན་ཆེན་འབྱུང་གནས་སོ། །ཛ་ལ་ནི་དྷི། །ཆུ་གཏེར་རོ། །ཡཱ་ཏེ—།
