པ་འགྷོཥ་ཎཡོ། སྟུ་ཏསྐ་ར་མོ་ཥ་ཀཱ། །པྲ་ཏི་རོ་དྷི་བ་རཱསྐནྡྲ་པཱ་ཊ༔་ར་མ་ལིམྦུ་ཙཱ། །ཙོ་རི་རྐུན་འཇབ་བུ་པ། །ཨར་བ་གཞན་དཊུ་རྐུན་མ་ཇག། རྐུན་ནོར་རྐུན་རྫས་ཆོམ་དང་ནི། །རྐུན་རྫས་ཀཱསྟི་ནྱ་ཙོཇེ་ཙསྟེ་ཡཾ་ལོyནྟུ་ཏདྡྷཱ་ནཾ། ཀྵ་། བཱི་ཏཾ་སམྱུ་པ་ཀ་ར་ཎཾ་བནྡྷ་ནེ་མྲྀ་ག་ལོག་ནོར་དེ་ནགས་ན། །རྒྱ་དང་རྙི་དང་འཆིང་བྱ་བ། །རི་དྭགས་དང་ནི་བྱ་རྣམས་སོ། །འབྲིང་ཀྱི་ཀཱི་ཎཾ། །ཨུཎཱ་ཐཀཱུ་ཊ་ཡནྟཾསཱདྷཱ་གུ་རཱ་མྲྀ་ག་བནྡྷ་ནི། །ཤུཝཾ་བ་རཱ་ཊ་ཀསི་ཏུ་རཛྫ་ཟས་ངན་པའི་འཁྲུལ་འཁོར་རོ། །རི་དྭགས་འཛིན་དང་རི་དྭགས་འཆིང་། །ཞགས་པ་སྒྲིལ་མ་མོ་དང་ཐག། ལྡི་ཥ་བ་ཊཱི་གུ་ཎ། །ཨུདྡྷཱ་ཊ་ནཾ་གྷ་ཊཱི་ཡནྟཾ་ས་ལི་ལོདྷཱ་ཧ་ཎཾ་པྲ་གསུམ་རྣམས་ཡིན་ཏེ་སྲད་བུའོ། །ཟོ་ཆུན་ཁྱུད་དང་བུམ་འཁྲུལ་འཁོར། །ཆུ་འབབ་ཆུ་ཡི་འཁྲུལ་ཧ༑། པུཾ་སི་བེ་མཱ་བཱ་པ་དཎྞསཱུ་ཏཱ་ཎི་ན་རི་ཏཾ་ཏ་བ། ྋ། བཱ་ཎི་བྷཱུ་ཏིརྫི་:ཡོ་འཁོར་རོ། །སྐྱེས་བུ་ཐགས་བཟང་རྟག་མའོ། །སྣལ་མ་སྲད་བུ་ཏཾ་ཏ་བ། །ཐགས་འཐག་བྱཱ་ཏི་མོ་སྟུ་ལྕེ་པཾ་་ལེ་པཱུ་དི་ཀརྨ་ཎི། །པཝཱ་ལི་ཀཱ་པུ་ཏྲེ་ཀ་༴སྔ་དག་མཚུངས། །བྱུག་པ་ཞལ་བ་ལ་སོགས་ལས། །གཟུགས་བརྙན་དང་ནི་འབག་ཡིན་ཏེ། །དངོས་དན་དི་བྷིཀྲྀ་ཏཱ། །པི་ཊ་ཀ་པེ་ཊ་ཀཔེ་ཌཱ་མཾ་ཛུ་ཥཱཉཐ་བི་ཧངྒཱི་ཀཱ། །པོ་བ་སོ་སོགས་ལས་བྱས། །སྒྲོམ་དང་མཛོད་དང་སྣོད་བང་བ། །དེ་ནས་ཁུར་ཤིང་གཉའ་ཤིང་ངོ༌། །བྷཱ་ར་པཏྟིསྟ་དཱ་ལམྦཤི་ཀཾ་ཀ་ཙོཉཐ་པཱ་དུ་ཀཱ། །བཱ་རླུ་རུ་པཱ་ནནྟི་སི་བཱ་དེ་འཕྱང་ཐག་པ་ཀ་ཙ་འོ། །དེ་ནས་ལྷམ་དང་ཆགས་བཅག་ལྷམ། །ཞབས་སྟེགས་དང་ནི་རྐང་ནུ་པ་དཱི་ན་པཱ་དཱ་པ་ཏཱ། །ནདྡྷི་བདྡྷི་བ་ར་ཏཱ་སཱུ་ད་རེ་ད་ཌ་ཎཱི་ཀ་ཤཱ། །ཙཱཎཱ་སྟེགས་སོ། །འབྲེང་ཐག་འབྲེང་ཞགས་ཀོ་ཐག་གོ། །རྟ་སོགས་ལྕགས་ནི་ཀ་ཤཱའོ། །གཏུམ་པོ་ལཱི་ཀཱ་ཏུ་ཀཎྜ་ལཱ་བཱི་ནཱ་ཙཎྜ་ལ་བལ་ཀཱི། །ནཱ་རཱ་ཙི་བྱཱ་དེ་ཥ་ནི་ཀཱ་ཤཱ་ཎ་སཱུ་ཎི་ཥ་ཀཿརྣམས་ཀྱི་པི་ཝང་ནི། །ཙཎྜ་ལ་དང་བལྦ་ཀཱི། །ནཱ་རཱ་ཙི་དང་ཞོ་ཆའོ། །གསེར་སྤྱད་རྡོ་ནི་ཀ་ཀ་ཥ། །པྲཤྩ་ཎཔ་ཏྲ་པ་ར་ཤུ་རཱི་ཏི་ཀ་ཀཱུ་ལཱི་ཀཱ་ས་མེ། །ཏེ་ཛ་སཱ་བནྟ་ནཱི་མཱུ་ཏྟཱ་ཥའོ། །ཆན་གྲི་པ་ཏྲ་པ་ར་ཤུ། །རཱིཥི་ཏུ་ལི་ཀཱ་དག་མཚུངས། །འོད་དང་ལུགས་ཀོང་མཱུ་ཥཱའོ། །།
