འབྲེལ༈རྔུ་བ་ཙཥ་འཚེད་པ་ལ། ཌབདག་གི་འདོད་པ་ལོ་ས་འོས། རྐྱེན་ས། གཉིས་བརྗོད། སྔོན་མའི་ཧ་ས་དང་པོ་ལྷག་པོའོས། རྩྭ་གོང་མ་ཕྱིས། ་་་་་པཛྲ་ས་ཞེས་གནས་པ་ལ། །ཀུ་བ་ཀ་བ་དང་འདྲ། །།འཛུང་༈ཡསེ?། ས་ལ་ཨ་ཨཱིའོ། །རྐྱེན་ས་ཕ་རོལ་ལ་བྱིངས་གཉིས་བརྗོད་པའི་དང་པོའི་ཨ་ཡིག་ནི་ཨི་ཡིག་ཏུ་འགྱུར་ཏེ། ཨ་ཨི། ཙུའི་ཀུ་དང་ཀུ་ཨི་ལ་་་ཞེས་པས། པི་པཀླུ་ཞེས་འགྲུབ་པ་ལ་ཨཱམ་སོགས་སྦྱར་བས་པི་པཀྵ་ཙ་ཀཱ་ར། པི་བཀྵ་མཱ་ས། །པི་བཤཾ་བ་བླུ་བ་སོགས་སོ། །།།ན་བའི་༈པཎ་རིགས་བརྗོད་པ་ལའོ་ཞེས་་པ་ལ། ཙུ་ར་སོགས་ཀྱིས་ཉི། ཀཱ་་ས་ཞེས་པས་ཨཱམ་ཨི་ལ་ཡོན་ཏན་སོགས་ཀྱིས། བཎ་ཡཾ་ཙ་ཀཱ་ར། བཎ་ཡཱ་མཱ་ས། བཎ་ཡཾ་བ་བྷཱ་བ་སོགས་སོ། །་་་་་ཕྱོགས་ཁ་བ་ནས་ཟླ་:་་ལ་ལ་༈དེ་བཞིན་དུ་བ་དྷ་འཆིང་བ་ལའོ་ཞེས་པ་ལ་ཉི་ཨཱམ་སྦྱོར་བ་སོགས་ཀྱིས། བ་དྷ་ཡཾ་ཙཀྲེ། བ་དྷ་ཡཱ་མཱ་སེ། བ་དྷ་ཡཱ་བ་བྷཱ་བེ་སོགས་དང་ཡང་བ་དྷ་ལ་་་བདག་གི་འདོད་པའི་དོན་གྱིས་སྦྱར། གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། བི་བྷནཾ་ཙཀྲེ། བི་:བྷཏྟཱ་མཱ་སེ། བི་བྷཏྟཾ་བ་བྷཱ་བ་སོགས་སོ། །༈བྱིངས་བྷི་ཧུ་བྷཾ་ཧཱུྃ་རྣམས་ཀྱི་རྐྱེན་ཨཱམ་སོགས་ཕ་རོལ་ལ་གཉིས་སུ་་་བརྗོད་པ་དང་། ལྔ་བྱིངས་དེ་རྣམས་ཉིད་ལ་ཎཔྲ་སོགས་སྦྱོར་བ་འདམ་ཀ་ཅན་དུ་་་བཤད་དེ། ཉི་བྷི་འཇིག་པ་ལ་ཨཱམ་སོགས་སྦྱར་ནས་གཉིས་སུ་བརྗོད་པ་དང༌། བྱིངས་ཉིད་ལ་ཎཔྲ་སོགས་སྦྱོར་བ་ཡང་ན་བར་བྱས་པས། །ཨཱམ་སྦྱར་བའི་་་ཕྱོགས་ལ། །བི་བྷ་ཡཱ་ཙ་ཀཱ་ར། །བི་བྷ་ཡཱོམཱ་ས། བི་བྷ་ཡཱ་ཎིབ་བྷཱ་བ་སོགས་དང་ཨཱམ་སོགས་མ་སྦྱར་བའི་ཕྱོགས་ལི། བི་བྷཱ་ཡ། བི་བྷུ་ཏུ་བིབྷུ་བི་བྷ་ཡི་ཐ། བི་བྷེ་ཐ། བི་བྷུ་ཐུབི་བྷཱུ། བི་བྷཱ་ཡ། །བི་བྷ་ཡ། བི་:བྱི་བའི། །།
