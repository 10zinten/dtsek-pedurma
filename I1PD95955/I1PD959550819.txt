ར་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་ཀ་ཀ་ནི་ཀ་ཨ་ཌ་ཌྷ་ནི་ཆུ་སྟོད་ཟླ་བའམ་དབྱར་ཟླ་འབྲིང་པོའོ། །ཤཱ་བ་ཎ་ནི། ་ཐོས་ལྡན་ཏེ། འདི་ལ་ཐོས་པ་མང་བས་སོ། །ན་བྷའི་སྒྲ་ནི་ནམ་མཁའ་སྟེ་དེ་ཉིད་ཟླ་བ་འདིའི་མིང་ངོ༌། །སྐྱབང་ཁོང་ནི་རྒྱུ་ལ་དེ་ཚེ་ཀ་ཏཚོར་ལ་ཁབ་ལཌ་ཀ་བ་ཁ་བཤྭ་བ་ཎི་ཀ་ནི་གྲོ་ཞུན་ཟླ་བའམ་དབྱར་ཟླ་ཐ་ཆུང་ངོ༌། །བྱསསྐྱབུ་སྐྱེ་བ། སྲོཥྛ་པ་དང་ཁྱུ་མཆོག་རྐང་པ་དང༌། བྷ་དྲ་དང་། བྷ་དྲ་པ་དང་། ཡང་བ་ལང་རྐང་པ་དང་། །ཁྲུམས་ཟླའམ་སྟོན་ཟླ་ར་བའོ། །ཊོ་ཀཎཎྜ་ཀ། Pལ་ཨཱ་ཤཱི་ན་ཐ་སྐར་ཟླ་བའམ་སྟོན་ཟླ་འབྲིང་པོའོ། །།ཨི་ཥ་ནི་ལྟད་མོ་སྟེ་ཨི་ཥ་འགྲོ་བ་ལའོ་ཞེས་པའི་ཁམས་སོ། །ན་བའོག་ལ་་ཏེ་ཟླ་བ་དེ་གླང་དུ་གཏོར་ཆེན་ངེས་ཀཱརྟི་ཀ་དང་ཀཱརྟི་ཀི་ཀ་ནི་སྨིན་དྲུག་ཅན་དང་སྨིན་དྲུག་ཟླ་བའམ་སྟོན་ཟླ་ཐ་ཆུང་ངོ༌། །བ་ཧུ་ལ་ནི་མང་པོ་འཛིན་ནོ། །དེ་རྣམས་ནི་ཟླ་བ་སོ་སོའི་མིང་ངོ༌། །ད་ཞདུས་དྲུག་གི་མིང་ལ། ཧེ་མནྟ་ནི་དགུན་སྟོད་དོ། །ཤི་ཤི་ར་ནི་དགུན་་་་སྨད་དོ། །བ་སནྟ་ནི་དཔྱིད་དོ། །པུཥཱ་ས་མ་ཡ་དང་། ་་སུ་ར་བྷི་ནི། ་མེ་ཏོག་ལྡན་དུས་དང༌། བདེ་བྱེད་དེ་དཔྱིད་དོ། །གཱི་ནི་སོས་ཀ་དང་། ཨུ་ཀ་ཚ་བའི་དུས་དང་། ལནི་དཱ་གྷ་ངེས་པར་བསྲེག་པ་དང༌། སྐྱོ་ཨུདྨོ་པ་ག་མ་དྲོད་ཕེབས་པ་དང་། ཨུ་དྲོད་དང་། ་ཨུཏྟཱ་ག་མ་དྲོད་འོངས་པ་དང༌། མ་ཏ་པ་ཚ་བའི་དུས་ཏེ་དེ་རྣམས་སོས་ཀའི་མིང་ངོ༌། །བླ་བྲིཊ་དབྱར་གྱི་དུས་སོ། །ཤ་ར་་ཏ་སྟོན་ཀའོ། །ང་ད་ག་ལོ་འདབ་ནི་༼ལོའི་མིང་ཚིགས་བཅད་ཕྱེད་ལ་སམྦརྨ་ར་ལོ་དང་། །བརྨ་ར། །བགྲང་བྱ་དང་། ཨརྡ་འགྲོ་འོང་དང༌། །ཧཱ་ཡ་ན་འདོར་བྱེད་རྣམས་སོ།  ོ། མི་རྣམས་ཀྱི་ལོ་དང་ཡང་དག་པར་འབྲེལ་པའི་ཟླ་བ་གཅིག་གི་ཡི་དྭགས་རྣམས་ཀྱི་ཉིན་ཞག་གཅིག་གོ། །དེ་བཞིན་དུ་མི་དང་ཡང་དག་པར་འབྲེལ་པའི་ལོ་གཅིག་གིས་ལྷ་་་—།
