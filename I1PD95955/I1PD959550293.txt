ོ་ཁཉཁྱོཀ་ཎཥ་རོལ་བློ་ལ་དེ་བ༈ཧན་འཚེ་བ་ལ། །ས་གཉིས་བརྗོད། །ཧན་གྱི་གྷ། ཀུ་ཧ་དག་གི་་་ཙུར་བསྒྱུར་བས། །ཛི་གྷན། ས་ཞེས་གནས་པ་ལ། ༈ཧནྟེསེ་ཎིཏྲ། ཧན་ལས་ས་ཎ་འགྲོ་བའོ། །ཧན་ལས་ཕ་རོལ་གྱི་འདོད་པའི་རྐྱེན་ས་ལཎ་འགྲོ་བའི་་་་མིང་ཅན་ཞེས་བྱ་ལ་དགོས་པ་ནི་འཕེལ་བ་བྱེད་པའི་དོན་ཏེ། <འཕེལ་བ་བྱས་་་པས། ཛི་གྷཾས་ཏི། ལཊ། ཛི་གྷཾ་སེཏྲ། ལིང་། ཛི་གྷོ་ས་ཏུ། ལོཊ། ཛི་གྷོ་ས་ཏཱཏྲ། ཨ་ཛི་གྷཾ་སཏཱ་ལང་སོགས་སོ། །པདྲོ་ར་ན། བུ་ཡིས་འོ་། །སྐྱེ་ཁང། ཊ། ་མིག། པ་དེ་ནི་༈རྔུ་ཀྲྀཏྲ་བྱེད་པ་ལ་ས་སྦྱར། །རྀཏྲ་ཨིར། རྀཏྲ་ཨིནྡྲའོ། །བྱིངས་་མཐའི་ར་ཡིག་ནི་རྐྱེན་ས་དང་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨིར་་དུ་འགྱུར་ཏེ། རཾ་ཨིར་དུ་བསྒྱུར། གཉིས་བརྗོད་སྔོན་མའི་ཧ་ས་དང་པོ་ལྷག་པོ། ཀུ་ཧ་དག་གི་ཙུ། ཕྲ་བྲ་སོགས་ཀྱིས་ཨི་ཕྱི་མ་རིང་པོར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཙི་ཀིརཱ་ཏི། ལཊ། ཙིཀིརྟེཏྲེ། ལིང་། ཙི་ཀིརྟ་ཏུ། ལོཊ། ཙིཀཱིརྟ་ཏཱཏྲ། །ཨི་ཙི་ཀཱིརྙཏྲ་ལང་སོགས་དང་། ༈དེ་བཞིན་དུ་ཧྲིན་འཕྲོག་པ་ལ། ཛི་ཧཱིནྡྲ་ཏི། ལཊ། ཛི་ཧཱིཙྩེཏྲ། ལིང་། །ཇི་ཧཱིཎྜ་ཏུ། ལོཊ། ཛི་ཧཱིརྟ་ཏཱཏྲ། ཨ་ཛི་ཧཱིརཱཏྲ་ལང་སོགས་དང༌། ༈ཙཱ་རྒྱལ་བ་ལ། ཏི་ཏཱིརྱ་ཏི། ལཊ། ཏི་ཏཱིརྙེཏྲ། ལིང་། །ཏི་ཏཱིན་ཏུ། ལོཊ། ཏི་ཏཱིརྱ་ཏཱཏྲ། ཨ་ཏི་ཏཱིནྡཏྲ་ལང་སོགས་སོ། །ཤིང་། གང་ཉིད་ན་བློར་འདས་ལོ། །ལ་༈རྔུ་ཀྲོའྲ་བྱེད་པ་ལ་ས་ར་ཨིར་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་ཙིཀཱིར། ས་ཞེས་གནས་པ་ལ། བཞི་པ་ལ་ཡཀྲ་འོས་དངོས་པོ་དང་ལས་ཀྱི་དོན་གྱི་རྐྱེན་ཡཀྲ་སྦྱར༑ཀྱིས་ཀུ་བ་བ་ན་ཤེས་ཡོན། །ཏ་ས་ཆེ་པོ་་་་་་ཁྲི་་གི་ན། །ཚེ་༈ཡཏ་ཡའི་ཨཏྲའོ། །རྐྱེན་ཡའི་ཨཏྲ་ནི་ཨཔྲ་བི་ཀཱ་ར་ཎ་མེད་པའི་་ཡུལ་ལ་དབྱི་ཞིང་འདོད་པའི་དོན་གྱི་རྐྱེན་སའི་ཨ་ཡང་དངོས་པོ་དང་ལས་ཀྱི་དོན་་ཅན་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། སྐབས་འདིར་སའི་ཨ་ཕྱིས་པས། །ཙི་ཀཱི་་་།
