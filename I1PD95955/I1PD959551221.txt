བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽དང་ན་དང་ན་དང་ཡང་བ་དང་ཀ་དང་སུ་དང་ཛ་དང་ཛ་དང་བ་རྣམས་ཡིན་པ་དེ་ལྟ་བུའི་ནོར་དང་མྱོས་བྱེད་དང་མེ་དང་ཐུབ་པ་རྣམས་ཀྱིས་ངལ་བསོ་བར་གྱུར་པ་དེའི་ཚེ་ལག་འགྲོ་རྣམ་་བསྐྱིངས་སོ། །མོ་གྲོ་ནོ་ཀླུ་སུརྫོ་བཱཙྪེསདྨ་སུ་མ་ད་ན་ད་ཧ་ན་མུ་ནི་བྷིནྡྷ་:ཛཾ་ག་བི་ཛིམྷི་ཏཾ། tArམ་ན་ནུ་དང་ན་ནུ་རྣམ་གཉིས་ས་ག་ཝ་ཞེས་བྱ་རྣམས་དགུ་དང་རོ་རོ་བྱེད་པར་ངལ་བསོ་་བཟློག་ཕྱོགས་འབབ་ཏུ་འགྱུར། །།ནུ་ཞེས་པའི་ཨུ་ཡིག་་ནི་ངལ་བསོ་་གསལ་བར་བྱེད་པའོ། །རྐང་པ་སོ་སོ་ལ་མ་དང་ན་དང་ནུ་དང་། ན་ནུ་ན་ནུ་དང༌། ་ས་དང་ལྕི་བ་དང་། བ་ཞེས་པ་འདི་ལྟ་བུའི་གང་ལ་དགུ་པ་དང་རོ་དང་རོ་དང་དབང་པོ་རྣམས་ཀྱིས་ངལ་བསོ་བ་ནི་བཟློག་ཕྱོགས་འབབ་ཏུ་འགྱུར་རོ། །རྣོ་:ནུརཱིན་ནུ་ས་ག་་བི་ཏི་ན་བ་ར་ས་ར་ས་ཀ་ར་ཎ་བི་ར་ཏི་ར་ས་བཱ་ཧསཱུཏ། འདི་ཉིད་དུ་ནི་རྒྱུན་ཆགས་དང་པོ་གཏུམ་པའི་ཆར་འབེབས་དག་ཏུ་འགྱུར་ཏེ་ན་གཉིས་དང་ནི་དེ་ནས་ར་བདུན་རྣམས་ཀྱིས་སོ། །འདི་ཉིད་དུ་ཞེས་པ་ནི་མཉམ་པའི་་བྲྀཏྟ་རྣམས་སུ་རྒྱུན་ཆགས་བཤད་པར་:བྱ་བའོ། །དེ་ལ་རྒྱུན་ཆགས་དང་པོ་ནི་་གཏུམ་པའི་ཆར་འབེབས་ཞེས་བྱ་བ་སྟེ། དེ་ཡང་རྐང་པ་སོ་སོ་ལ་ན་གཉིས་དང༌། དེའི་བར་མ་ཆད་དུ་རེ་ཕ་བདུན་རྣམས་སུ་གྱུར་པ་ནའོ། །པྲ་ཐ་མ་ཀ་ཨི་ཧ་་་དཎྜ་ཀཉྩཎྜ་བྲྀཏྟི་པྲ་ཡཱ་ཏོ་བྷ་བེནྣ་དྭ་ཡེ་ནཱ་ཐ་རིསཔ་བྷི། I། ་་འདིར་ནི་རྐང་པ་སོ་སོར་རེ་ཕ་འཕེལ་བར་་་གྱུར་རྣམས་ཆུ་རྒྱུན་རྒྱ་མཚོ་སྦྲུལ་དང་ཆར་སྤྲིན་རོལ་པའི་འབྱུང་གནས་ཤིན་ཏུ་་་། བཀླུབས་པ་དུང་སོགས་སོ།  ོ། འདིར་རྒྱུན་ཆགས་ལ་ཆུ་རྒྱུན་ལ་སོགས་པར་ཕྱེ་བ་རྣམས་ནི་རྐང་པ་སོ་སོ་ལ་ར་ཡིག་རེ་རེ་འཕེལ་བས་སྔ་མ་སྔ་མ་ལས་ཐ་དད་དུ་་་on།
