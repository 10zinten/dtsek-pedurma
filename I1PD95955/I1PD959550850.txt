།།པ་འཇོ་བ་ལས་སྐབས་དང་པོ་རྫོགས་སོ། །ཀ་ཊ། །་་་་་་་། དེ་ན། །བའི་འང་བླ་ན་། དཔ་ཡིདེ་ནས་སའི་སྡེ་ཚན་བཤད་པར་བྱ་སྟེ། ཚས་ཡི་སྡེ་ཚན་གྲོང་ཁྱེར་དང་། །རི་བོ་དང་ནི་ནགས་ཚལ་དང་། །སྨན་དང་རི་དྭགས་ལ་སོགས་རྣམས། །།།མི་་་དང་བྲམ་ཟེ་རྒྱལ་རིགས་དང་། །།རྗེའུ་རིགས་དམངས་རིགས་ཡན་ལག་བཅས། །ཉེ་བའི་ཡན་ལག་བཅས་འདིར་བརྗོད། །ཅེས་པ་ལ་འདིར་སའི་སྐབས་ཡན་ལག་དང་བཅས་པ་ཉེ་བའི་ཡན་ལག་ལས་ས་ལ་སོགས་པ་བཅུ་རྣམས་ཀྱིས་སའི་སྡེ་ཚན་་ཏེ་རྣམ་པ་བཅུར་བཤད་དོ་ཞེས་པ་བསྡུས་པའོ། །རི་དྭགས་ལ་སོགས་པ་རྣམས་་་ཞེས་པས་ནི་སེང་གེ་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པའི་དོན་ཏོ། །གང་ལ་ལོཎཚིགས་སུ་བཅད་པ་གཉིས་ཀྱིས་སའི་མིང་སྟེ། །བྷུས་དང༌། བྷུ་མི་སྐྲུན་བྱེད་དེ། བྷུ་ཡོད་པ་ལ། སྔ་མ་ལ་ཀཱིབ་ང་དང་། ཕྱི་མ་ལ་ཨུ་རྨི་བྷུ་མི་སོགས་ཀྱིས་མིའི་རྐྱེན་སྦྱར་བའོ། །ཨ་ཙ་ལ།  ོམི་གཡོ་བ་སྟེ་སའོ། ་། ཨ་ནནྟ། འདི་ལ་མཐའ་མེད་པས་མཐའ་ཡས་ཞེས་བྱའོ། །ར་ས། །བཅུད་ལྡན་ནོ། །བི་ཤཱམྦྷ་རཱ་༔སྣ་ཚོགས་འགེངས་སོ། །སྦི་རཱ་བརྟན་པ་སྟེ། །སྦྲ་གནས་པ་ལ་ཀི་ར་ཙ་རྐྱེན་ནོ། །དྷཱ་ར་འཛིན་མའོ། །དྷ་ར་ཏི། ་རྟེན་བྱེད་མའོ། །དྷ་ར་ཎིའཆང་་བྱེད་མའོ། །ཀོ་ཎྜི། སྒྲ་སྒྲོགས་མ་སྟེ། ཀླུ་སྒྲ་ལ། །ཨི་ནིན་རྐྱེན་ནོ། །།ཚུ། རྒྱལ་བྱེད་དོ། །༼། ཀཱ་བྱ་བཱི། འོད་སྲུངས་བྱིན་ཞེས་བྱ་སྟེ། །རྒྱལ་པོ་དགའ་བྱེད་་དགྲ་སྟ་ཅན་གྱིས་དྲང་སྲོང་འོད་སྲུངས་ལ་རྟ་གཙང་གི་མཆོད་སྦྱིན་གྱི་མཐར་ཡོན་དུ་བྱིན་ཞེས་གྲག་གོ། །ཀཱི་ཏེ། གནས་ཏེ། ཀྱི་ངེས་པར་གནས་པ་ལ་ཏྟིའོ། །སརྦ་ས་ཧ། ཀུན་བཟོད་དེ། ཥ་ཧ་བཟོད་པ་ལ་ཁ་ཤ་རྐྱེན་ནོ། །བ་སུ་མ་ཏཱི། ནོར་ལྡན་མ་སྟེ། གསེར་དང་དངུལ་དང་། ཧཱི་ར་སྟེ་རྡོ་རྗེ་ལ་སོགས་པའི་རིན་།
