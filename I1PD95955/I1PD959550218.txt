རསྨི་པ་ད་རྣམས་སྦྱོར་ཏེ། བྱིངས་ཀྱི་རྔུ་ཡིག་ཏུ་འགྲོ་བའི་དོན། རྦ་གཉིས་ཀའི་ཚིག་ཅན་གྱི་དོན། ཨཔྲ་འཇིག། གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས་:བརྟ་མ་ནའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ད་ཧཱ་ཏི། གཉིས་ཚིག་ལ། ད་ཧཱི། ཏས་ཞེས་གནས་པ་ལ། ༈དཱ་དེདཱ་ལ་སོགས་པའི་འོ། །རྣམ་དབྱེ་དྲ་འགྲོ་བའི་མིང་ཅན་རྣམས་ཕ་རོལ་ལ་བྱིངས་ཌུ་དཱརྦ་ལ་སོགས་པ་གཉིས་བརྗོད་པའི་མཐའི་ཨཱ་ཡིག་ནི་དབྱི་བར་བྱ་སྟེ་ཨཱ་ཕྱིས་པས། དཏྟ། མང་ཚིག་ལ། ད་ད་ཏི། དེ་བཞིན་དུ་བསྒྲུབས་པས་དབུས་མ་གསུམ་ལ། ད་དཱ་སི༄༅། དཏྠདཏྠ༑ད་དཱ་མི། ད་དྭད། པོ། ཚཚ། ༈ཆོག་འབྱུང་ལིང་ལ། ད་ཏུཏྲ། ད་ཏྭ་ཏྭཾ། ད་དྷུད་རཿད་ཅུ་ཏཾ༑ད་དྷུ་ཏི། ད་། ད་དྷྱ་པ། ད་དྲུ་མ། ༈ཤིས་བྱེད་ལོཊ་ལ། ད་དཱ་ཏུ། དཏྟཱཏྲ༑དཏྟཾ། ད་ད་ཏུ། །སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། །ད་དཱ། ཧི་ཞེས་གནས་པ་ལ། ༈དཾ་ཧོ། དཱ་རྣམས་ཀྱི་ཧི་ལའོ། །བྱིངས་དཱ་ལ་སོགས་པ་རྣམས་ཀྱི་རྣམ་དབྱེ་ཧི་ཕ་རོལ་ལ་བྱིངས་གཉིས་བརྗོད་པའི་དང་པོ་དབྱི་ཞིང་མཐའི་ཨ་ཡིག་ཀྱང་ཨེ་རུ་འགྱུར་ཏེ། དེ་ཧི། དཏྟཏྲ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། དཏྟཾ༑དཏྟ། ཕྱི་མ་གསུམ་ལ། ད་དྭ་ནི། ད་དཱ་པ། ད་དཱ་མ། ༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་ད་དཱཏྲེ། ཨ་དཏྟཾ། ཨ་ད་དུ༔ཨ་ད་དཱ་ཨ་དཏྟཾ། ཨ་དཏྟ། ཨ་ད་དཾ། ཨ་ད་དྭ། ཨ་དདྨ། དེ་བཞིན་དུ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱོར་བ་ལས་བརྟ་མ་ན་ལཊ་ལ། དཏྟེ། ད་དཱ་ཏེ། ད་ད་ཏེ། དཏྠེ། ད་དྲ་ཐེ། དདྡྷེ། ད་དེ། ད་དྭ་ཧེ། དདྨ་ཧེ། །ཙ༈་ཆོག་འབྱུང་ལིང་ལ། ད་དཱི་ཏ། ད་དཱི་ཡཱ་ཏྭཾ། ད་དཱི་རན། སོགས་སོ། །བདགའོ། །ཏཱ—།
