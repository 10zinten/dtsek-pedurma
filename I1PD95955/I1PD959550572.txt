།ཡོ་ནི་རོ་དྷ་ཨེ་བཾ་བཱ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎ?། །།ཁྱོད་དང་པོ་ཚོགས་པའི་གཤེགས་པ་ཉིད་ཀྱིས་བཀའ་སྩལ་ཏེ། །དེ་སྐད་གསུང་བའི་ཚུལ་ཅན་དགེ་སྦྱོང་ཆེ། །།།པཎྜི་ཏ་ཆེན་པོ་ཀཱིརྟི་ཙནྡྲའི་ཞལ་སྔ་ནས། ཡར་ལུང་པ་གྲགས་པ་རྒྱལ་མཚན་གྱིས་བལ་ཡུལ་ཡམ་་བུའི་གྲོང་ཁྱེར་དུ་བསྒྱུར་བའོ། །ཤུ་བྷཾ་མསྟུསརྦ་ཛ་ག་ཏཱཾ། སླད་ནས་འགོས་ལོཙྪ་བ་དགེ་སློང་བྷི་དྷརྨ་པཱ་ལ་བྷ་དྲ་ཞེས་བགྱི་བས། །རྒྱ་དཔེ་དང་འགྱུར་རྙིང་ལ་གཏུགས་ཏེ། དོན་ལ་ཅི་ནུས་དཔྱད་དེ་འགྱུར་བདེ་བར་བགྱིས་པའོ། །སྣ་ཚོགས་དངོས་པོ་རྣམས་ཀྱི་མངོན་བརྗོད་པ། །རིན་ཆེན་འབྱུང་གནས་ལ་འགྲན་བསྟན་བཅོས་འདི། །སྔོན་གྱི་པཎྜི་ཏ་དང་ལོ་ཙཱ་བས། །གངས་ཅན་སྐད་དུ་བསྒྱུར་བའི་རྗེས་སུ་ཡང༌། །མ་དག་དཀའ་བ་རྣམས། །གཙུག་ལག་དབང་པོ་བགྱིས་ཏེ་ཅི་ནུས་བཅོས། །སླད་ནས་འཇིག་རྟེན་མིག་མཆོག་རྣམས་ཀྱིས་ཀྱང་། །འགྲེལ་དང་བསྟུན་ནས་བྱི་དོར་རྣམ་དག་ཞུ། །སརྦ་མངྒ་ལཾ། །།།།།ཊི་ཚེ་ཚོ་དུ། བུ་ལ་བྱེད་་་་་་་་་་་་་ནས་། རྫོར་སྤྱི་བདུན་ཨཁན། །ས་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་དང་«པེ་»«སྣར་»གཉིས་འགྱུར་མི་འདྲ་་་བས་འདིར་《སྡེ་ལ་གཞིར་བཟུང་ནས་«ཅོ་»དང་དཔེ་བསྡུར་བྱས་ཡོད། དེ་དག་འགྲོ་བ་«ཅོ་»དྲི། བྱེ«ཅོ་མ། ཚེ་ཞེསཔ་དེ་«ཅོ་»སྡུག«ཅོ་»ཚེ། ::་་་་་་་ལོ་«ཅོ་»སྐྲ། འ«ཅོ་»ཤཱཐེ། ་«ཅོ་»བཱ། «ཅོ་»ཀཔའི་《ཅོ་»ཤིའི། «ཅོ་»ཆོགོ། «ཅོ་རྒུ། «ཅོ་»ཀཱ། །«ཅོ་»ཏེ། «ཅོ་ཙ། :«ཅོ་»འཇོམ། །«ཅོ་»འཇོམ་པ། «ཅོ་»བརྡོ། ་ཚེའོ། །«ཅོ་བ། «ཅོ་»དྷཱ་བཱ། ་་་་་་་་འོ་ཚ། ལྷ་«ཅོ་»ཏིདྨ། «ཅོ་—སྟ། «ཅོ་»བི། བདེ་བ་དང་། ི«ཅོ་བེ། «ཅོ་»ཏགས། «ཅོ་ཟི། «ཅོ་»ཅིཥྛ། «ཅོ་བུ། །ནས་དང་། ཚོ། །«ཅོ་»བྷཱ། །བར་དོ་དུས། །ེ«ཅོ་»སྐྱོ། «ཅོ་ཨིནྡ། སྐྱོན་འདམ་གཞུང་། སྐྱེ་—།
