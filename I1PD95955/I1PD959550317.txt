ཙམ་བསྟན་འགྱུར། །སྒྲ་མདོ། ༼སེ་འོ་མཚོ་འཁྱོ་ཀ་བློ•ལྔགློ་ཝnབའོ༑། རྐྱེན་པ་འགྲོ་བ་ཕ་རོལ་ལ་མི་ཛའི་ཨུ་པ་དྷཱ་ནི་འཕེལ་བར་འགྱུར་ཞིང་ང་འགྲོ་བའི་མིང་ཅན་དབྱངས་དང་པོ་ཕ་རོལ་ལ་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། །མཱནྟི། ?མྲྀཋ༑མཱ་ལྡནྟི། ཚྭ་ཛནྟི་སོགས་སོ། །ཀླུ་བྱ་ཕྱི་བ་ལ་ལ་ད་ད་ལྟ་བ་ཏུ༈ནི་ཛཾ་གུ་ཎཿནི་ཛ་རྣམས་ཀྱི་ཡོན་ཏན་ནོ། །ཨཔྲ་བི་ཀཱ་ར་ཎ་འཇིག་པ་ལས་ནི་ཛ་ལ་སོགས་པ་རྣམས་ཀྱི་གཉིས་བརྗོད་པའི་དང་པོ་དེ་ཡོན་ཏན་དུ་་་འགྱུར་ཏེ། ཉེར་བསྒྱུར་ནི་ར་སྔོན་མའི་ཎི་ཛི་དག་བྱེད་ལ་ཨཔྲ་འཇེག་ཡག་བ། །ཛུ་ཧ་སོགས་ཀྱིས་གཉིས་བརྗོད། སྔོན་མའི་ཧ་ས་དང་པོ་ལྷག་པོ། །ནིཛ་སོགས་་་ཀྱིས་སྔོན་མ་ཡོན་ཏན། དེ་ནས་པ་འགྲོ་བ་རྣམས་ལ་ཡོན་ཏན་བྱེད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ནིཎེ་ནེཀྟི། ནིཎེནིཀྐནིཎེ་ནི་ཛ་ཏི། ནིརྗེ་ནི་ཀི་སོགས་དང་། ཨ་བ་ནེ་ནིཀྐེ་སོགས་ཏེ་གཉིས་ཀའི་ཚིག་ཅན་ནོ། །དབང་པོ་༈དེ་བཞིན་དུ། བི་ཛིར་ཐ་དད་དུ་གནས་པ་ལ། བེ་བེསྟི། བེ་བིཀྐབེ་བི་ཛ་ཏི་སོགས་དང་། ༈བི་ཊྚི་ཁྱབ་པ་ལ། བེ་བེའྲི། བེ་བིཊྚ། བེ་བི་ཥ་ཏི་སོགས་སོ། །རྒྱན་ཆ་དང་ནང་བ་གང་ཞེ་ན་ཁྱོ་གས་པོ་འདྲ་བ་༈བི་ཨཱང་སྔོན་མའི་ཚིགས་སྲུང་བ་དང་གང་བ་ལ། རཾ་པྲ་རིཔཱུརྦ་སྱ། སྔོན་མའི་རི་མྲྀ་དག་གི་ཨིའོ། ་། ཨཔྲ་འཇིག་པ་ལས་བྱིངས་གཉིས་བརྗོད་པའི་སྔོན་མའི་རྀ་འགྲོ་བ་དང་དྲི་དག་གི་རཾ་ཡིག་ནི་ཨི་རུ་འགྱུར་ཏེ། ཨཔྲ་འཇིག་་། གཉིས་བརྗོད། རཾ་སྔོན་མ་ཨི། དེ་ནས་པ་འགྲོ་བ་ལ་ཡོན་ཏན་སོགས་ཀྱིས་་་བསྒྲུབས་པས། བྱཱ་པི་པརྟི། ལཊ། བྱཱ་པི་པྲཏལཊ། བྱཱ་པི་པྲི་ཏི། ལིང་། བྱཱ་པི་པྲ་ཡཱ་ཏྲ། བྱཱ་པི་པརྟུ། ལོཊ། བྱཱ་པི་པྲ་ཏཱཏྲ་སོགས་དང་། །རཾ་དང་སྲི་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་དིཔྲ་དང་སིཔྲ་ལ་ཨཊ་ཨཱ་ག་མ་སྟེར་བ་དང་མི་སྟེར་བ་འདམ་ཀ་ཅན་དུ་བཤད་པས། བྱཱ་པི་པ་རཏྲ། དཔ། བྱཱ་པི་པབྱཱ་པི་པ་རཿསིཔ། བྱཱ་པི་པ།
