ཟླ་་ཚོཀ་མཚོ་ཨ་བྷི་དྷཾ་ག་སྤྲོ་ཧཱ་ཡཱཉྩ་ག་བྷསྟོ་ཙ་རུ་སི་ཡཱམ། ་། ་་་་་་པ། མངོན་འཁྱུད་དང་ནི་འདོད་པ་དང་། །འོད་ལ་རུ་ཙི་བུད་མེད་དོ། །ཙ་མཐའ་རྣམས་སོ། །།།ཀེ་ཀི་ཏཱརྒྱུ་བ་ཧི་ཧྥུ་ཇོ་དནྟ་བི་པཱཎྜ་ཛཱ་དཱི་ཛཱ། །ཨ་ཛཱ་བི་ཧ་རཙྪ་གཱ་གོ་ཨ་ཧི་བྷུ་ཛ་རྨ་བྱ་ཁྱུང་སྤྲུལ་ཟ། །དྲི་ཛ་བྲམ་ཟེའི་སྒོ་ང་སོ། །ཨ་ཛཱ་ཁྱབ་འཇུག་དབང་ཕྱུག་ར། །དྷ་ནི་བ་ཧཱ་བྲ་ཛཱ། །དྷརྨ་ར་ཇོ་ཛི་ན་ཡ་མོ་ཀུཊྚ་དནྟེ་པི་ནསྟི་ཡཱ། །བ་ལ་ཛེ་ཀྵེ་ཏྲ་འབྲོག་ལྷས་ལམ་དང་ཚོགས་བྲ་ཛཱ། །ཆོས་རྒྱལ་རྒྱལ་བ་གཤིན་རྗེའོ། །ཀུཉྫ་སོ་དང་ཕུག་པའོ། །བ་ལ་ཛ་པཱུརྞ་རེ་བ་ལ་ཛཱ་བལྷ་དན་ནཱ། །ས་མེ་གྷཾཤེ་ར་ཎེ་པཱུ་ཛིཔྲ་ཛཱ་སྙཏཱནྟ་ཏོ་ཛ་ནེ། །ནི་ཞིང་གྲོང་སྒོ། །བ་ལ་ཛ་ནི་ལེགས་པར་མཐོང་། །ས་མཉམ་གཡུལ་ལ་ཨཱ་ཛིའོ། །པྲ་ཛཱ་བུ་རྒྱུད་སྐྱེ་དགུའོ། །ཨཊྚི་ཤཾ་ཁ་ཤ་ཤཱི་ཀོ་ཙ་སྭཱ་ཀེ་ནི་ཏྱ་ནི་ཛཾ་ཏྲི་ཥཱ། །།།ཨ་ཌ་དུང་དང་རི་དྭགས་མཚན། །ནི་ཛི་བདག་པོ་རྟག་པ་གསུམ། །ཛ་མཐའོ། །འོ་ན་ཀར་་ཁ་ཁོ་པཾ་པདྨ་ནི་པྲ་བཱི་ཎེ་ཥཱ་ཀྵེ་ཏྲ་ཛྙཱ་བཱ་ཙྩ་ལི་ག་ཀཿ། །སཾ་ཛྙཱ་སཱཙེ་ཏེ་ནཱ་ནཱ་མ་ལ་སྐྱེས་བུ་བདག་དང་ལེགས་རྟོགས་པ། །ཀྵེ་ཏྲ་ཛྷཱའོ་བརྗོད་བྱའི་རྟགས། །སཾ་ཛྙཱ་འདུ་ཤེས་མིང་དང་ནི། །ཧཙཱ་ཝཱནྡེ་སཱུ་ཙ་ནཱ། །ཉཱནྟ། །དགའལག་སོགས་དོན་བྱེད་བརྡ་ལའོ། །ཉའི་མཐའོ། །ཀཱ་ཀེ་བྷ་གཎྜ་ཀ་ར་ཊོ་ག་ཛ་གཎྜཀ་ཊི་ཀ་ཊོ། །ཤི་བི་པིཋསྐུ་ཁ་ལ་ཏོ་བྱ་རོག་གླང་འགྲམ་ཀ་ར་ཊ། །གླང་པོའི་འགྲམ་དང་རྐེད་ཀ་ཊོ། །ཤི་པི་པིཤྩ་ཉམས་པ་དང་། །དུཉྩརྨ་ནི་མ་ཧེ་ཤྭ་རེ། །དེ་བ་ཤིཝིནྱ་པི་ཏཱཥཱ་དིཝཾ་དེ་བེ་བི་ན་དྭ་ཡོ། །པགས་པ་ངན་དང་དབང་ཕྱུག་ཆེ། །ལྷ་ཡི་བཟོ་ནི་ཏཱཋའོ། །དིཊཾ་སྐལ་བ་གཉིས་འདུས་ལ། །ར་སེ་ཀ་ཊུཀ་ཊ་ཀཱརྨེ་ཏྲི་ཥཱ་མཏཱ་ར་ཏཱི་ཡོ། །རིཊཾ་ཀྵེ་མཱ་ཤུ་བྷཱ་བྷཱ་བེ་རོ་བྲོ་ཀ་ཊུཀ་ཊུ་ནི། །དོན་མེད་གསུམ་མོ་སེར་སྣ་རྣོ། ལ། རིཤྩ་བདེ་དང་མི་དགེ་མེད། །།ཥཱ་རིབྦེ་ཏུ་ཤུ་བྷཱ་ཤུ་བྷེ། །མཱ་ཡཱ་ནི༔་ལ་ཡནྟེ་ཥཱ་ཀེ་ཏ་བཱ་ནཾ་ཏ་རཱ་ཤི་ཥཱ། །ཨ་རིབྦེ་ནི་དགེ་མི་དགེ། །སྒྱུ་མ་མི་གཡོ་འཁྲུལ་འཁོར་དང༌། །གཡོ་སྒྱུ་མི་བདེན་ཕུང་བ་དང་། །ཉི་»jབྱཟླ་། pདན་་་་་་། །་ཟེ་,ས་པ་I—།
