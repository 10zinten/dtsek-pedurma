ཥཁ་འ་ཌg་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་ཀ་ལ་ཝལ་ཅཡོeཅུ་ཐེ་ཡ་སཱི་སུ་ཀྲྀ་ཏཾ་པྲྀ་ཥ། །མུངྒཱི་ཏིཔྲ་མ་དོ་ཧནྟ་པྲ་མོ་ད་མོ་ད་སམྨ་དགེ་བ་ལེགས་བྱས་ཁྱུ་མཆོག་གོ། །དགའ་མགུ་རངས་དང་སྤྲོ་བ་དང་། །ཡི་རངས་དགྱེས་དང་མཉེས་པ་དཱ༈། །བོ། སཱུ་དཱ་ནཾ་ད་ཐུ་རཱ་ནཾ་དསརྨ་སཱ་ཏ་སུ་ཁཱ་ནི་ཙ། །སཱསྲེ་ཡ་སཾ་དང༌། །ཉམས་དགའ་སྡིག་ཟད་ཞི་བ་དང་། །མི་དགེ་སྤངས་དང་ལེགས་སྒོའོ། །།དགེ་དང་བསྔགས་འོས་ཤི་ཐཾ་བྷ་དྲངྐ་ལྱཱ་ཎམྨཾ་ག་ལ་སུ་བྷཾ། །བྷ་བུ་ཀམྦྷ་བི་ཀམྦྷ་ཧུཾ་ཀུ་ཤ་ལཾ་ཀྵེ་ཞི་བ་དང་། །བཟང་པོ་ཕན་ཐོབ་བཀྲ་ཤིས་ལེགས། །སྐལ་པ་སྐལ་ཐོབ་འདོད་ཐོབ་དང༌། །ངན་སེལ་མ་མསྦི་ཡཱ་མ། །སསྟཉྩ་ཐ་ཏྲི་ཥཱ་དྲ་བྱེ་བཱ་པཾ་པུཙྩཾ་སུ་ཁཱ་དི་ཙ། །བདེ་ལེགས་མོ་མ་ཡིན། །།བསྟོད་འོས་དེ་ནས་གསུམ་དངོས་པོ། །སྡིག་དང་བསོད་ནམས་བདེ་ལ་སོགས། །མ་ཏལླི་ཀཱ་མ་ཙརྩི་ཀཱ་པྲ་ཀཱཎྜ་མུདྷཱ་ཏལྤ་ཇོ། །པྲ་སསྟ་བཱ་ཙ་ཀཱ་ནྱ་ཤིན་ཏུ་བཟང་དང་ཕུལ་བྱུང་དང་། །རབ་དང་མཆོག་དང་ལྟད་མོར་ཆེ། །འདི་རྣམས་མཉམ་པ་ཅན་ཞེས་སཱུ་རྒྱ་ཡ་ཤུ་བྷཱ་བ་ཧོ་བི་དྷི། །དེ་བནིཊཾ་བྷཱ་ག་དྷེ་ཡཾ་བ་གྱི་བརྗོད། །སྐལ་ལྡན་དགེ་བ་བསགས་རྒྱབ་ཅན། །ལྷས་བསྐོས་ལྷས་བསྟན་སྐལ་བ་དང་། །མོའོ་སྔོན་བྱས་་སི་ནི་ཡ་ཏིརྦ་དྷི། །ཧེ་ཏུནྟཱ་ཀཱ་ར་ཎཾ་བཱི་ཛ་ནི་དཱ་ནཾ་ཏཱ་དི་ཀཱ་ར་ཎ། །:ཀསྔོན་གྱི་ལས། །རྒྱུ་དང་བྱེད་པོ་ས་བོན་དང་། །གླེང་གཞི་ཐོག་མའི་བྱེད་པ་པོ། །།།ཀུན་ཏུ་༈ཀྵེ་ཏྲ་ཛྷ་ཨཱཏྨ་པུ་རུ་ཥཔྲ་དྷཱ་ནཾ་པྲ་ཀྲྀ་ཏིཔྲི་ཡཾ། །བི་ཤེ་ཥཀཱ་ལི་ཀོཉ་༈ཞིང་ཤེས་བདག་དང་སྐྱེས་བུ་སྟེ། །གཙོ་བོ་རང་བཞིན་མོ་ཡིན་ནོ། །ཁྱད་པར་དུས་ཅན་གནས་བསྠཱ་གུ་ཎཱ་ས་ཏྭཾ་ར་ཛསྟ་མ། །ཛ་ནུརྫ་ན་ན་ཛན་ནི་ཛ་ནི་རུཏྠཏྟི། སྐབས་སོ། །ཡོན་ཏན་སྙིང་སྟོབས་རྡུལ་དང་མུན། །སྐྱེ་བ་སྐྱེད་པོ་སྐྱེས་པ་སྐྱེ། །ཉེར་བསྐྲུན་ཉེ་བར་རུདྡྷབ། །པཱ་ཎཱི་ཏུ་ཙེཏྟ་ནོ་ཛནཱི་ཛཅུ་ཛ་ཅུ་ཤ་རཱི་རི་ཎ། །ཙིནྟནྟུ་ཙེ་འབྱུང་རྣམས་སོ། །སྲོག་ཅན་སེམས་ཅན་སྐྱེ་ལྡན་དང༌། །སྐྱེ་བོ་སྐྱེ་དགུ་ལུས་ཅན་ནོ། །སེམས་དང་སེམས་ཏོ་ཧྲི་ད་ཡཾ་སཱནྟཾ་ཤྲཱིནྡྲ་ན་སསྨྲ་ན། །བུདྡྷིརྨ་ནཱི་ཥ་དྷི་ཥ་ཎཱ་དྷཱིཔྲཛྙཱ་པ་སྙིང་དང་ནི། །སྒྲོགས་བྱེད་འཕྲོག་བྱེད་ཡིད་བྱེད་ཡིད། །བློ་གྲོས་ཡིད་འདོད་སྤོབས་པ་ཅན། །།བློ་—。。།
