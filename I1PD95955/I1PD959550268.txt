ཕྱོགས་ལ། ཨ་རྒྱ་གཱི། ཨ་རྒྱ་གཱི་ཥཱ་ཏྭཾ། །ཨངྒ་གཱི་ཥ་ཏ་སོགས་དང་གཱི་རུ་མ་་་བསྒྱུར་བའི་ཕྱོགས་ལ། དིཔྲ་སོགས་ཀྱི་ཨཊའི་སྔོན་དུ་དབྱངས་དང་པོའི་ཨཊ་་་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་དྷིཥྛ། ་་ཨ་དྷི་ཥཱ་ཏྭཾ། ཨ་དྷི་ཥ་ཏ། ཨདྷིཥྛསོགས་སོ། །རུ་སྒྲ་ལའོ་ཞེས་པ་ལ་ཨིཊ་འཇུག་ཅིང་། པ་ལ་ཎ་འགྲོ་བ་འོས། ་ལ་འཕེལ་བ་བྱས་པས། ཨ་རཱ་བཱིཏྲ། ཨ་རཱ་བིཏྟཾ། ཨ་རཱ་བི་ཥཱ། ཨ་རཱ་བཱིསོགས་སོ། །ཌུ་པ་ཙཥ་འཚེད་པ་ལ། འ་པ་རསྨི་པ་དེའི་སྐྱེས་བུ་དང་པོའི་་་གཅིག་ཚིག་ལ། །ཨཱིཊ་བྱིན། །་འཕེལ་བ་བྱས་པས། །ཨ་བཱཀཱིཏྲ། གཉིས་ཚིག་ལ། ཨ་བཛྲ་ས་ཏྭཾ་ཞེས་གནས་པ་ལ། ༈ཛྷ་སཱཏྲ། ཛྷ་ས་ལས་སོ། །བྱིངས་ཛྷ་ས་སོ་སོར་བསྡུ་བའི་མཐའ་ལས་ཕ་རོལ་གྱི་སི་ནི་རྣམ་དབྱེ་ཛྷ་ས་སོ་སོར་བསྡུ་བའི་ཡི་གེ་དང་པོ་ཅན་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། སི་ཕྱིས། འཕེལ་བ། །ཙུའི་ཀུ་་་སོགས་ཀྱིས། ཨ་པཱཀཾ། མང་ཚིག་ལ་ཨན་ཨུས་སུ་བསྒྱུར་བས། ཨ་པཱཀྵདེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། །ཨ་པཱཀྵིཨ་པཱཏྟཾ། ཨ་པཱཏྟ། ཨ་:པཱཀྵ༑་ཨ་པཱཀྵ། ཨ་པཱ། དེ་བཞིན་དུ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། །ཨ་པཏྟ༑འོ་ཨ་པཀྵ་ཏྭཾ། །ཨ་པཀྵ་ཏ། ཨ་པཀྵཨ་པཀྵ་ཛཾ། ཨ་པཤྩ། ཨ་པཀྵི། ཨ་པཀྵ་ཧི། ཨ་པཀྵཧི། གང་ཉིད་གདན་ས་བ་མང་བ་་ང་དེཊ༈པྲ་ཚ་ཤེས་པ་ལ།  ོཛྷ་ས་དང་པོ་རྣམས་ལ་སི་དབྱི་བ་དང་ཚྭཤ་ཥ་་་སོགས་ཀྱིས་ཚྭ་ཥ་དེ་ཀྲ་ཀུ་ཨི་ལ་སོགས་ཀྱིས་བསྒྲུབས་པས། །ཨ་པཱཀྵིཏྲ། ཨ་་པཱཎྞཾ༑ཨ་པྲ་གླུ། ཨ་པཱཀྐིཨ་པཱཊཾ། ཨ་པཱཋ། ཨ་པཱཏྟཾ། ཨ་པཱ། ཨ་པཱ། བྱིངས་མཐའ་ཛྷ་སའི་ཡི་གེ་མ་ཡིན་ན་སི་མི་དབྱི་སྟེ་ཨེ་དྷ་འཕེལ་བ་ལ་སི་ས་ཏཱ་་་སོགས་ཀྱིས་ཨིཊ་བྱིན་པས། ཨི་དྷིཤྩ། ཨ་དྷིཥྛ'ཏྭཾ། ་ཨེ་དྷིཥ་ཏ། ཨི་དྷི་ཨི་དྷིཥཱ་ཐཱ། ཨི་དྷིདྡྷཾ། ཨི་དྷི་ཥི། ཨེ་དྷི་ཥ་ཧི། ཨེ་དྷི་ཧྥ་ཧི། དེའི་སྟེང་།
