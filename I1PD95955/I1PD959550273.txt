ཁ་བ། བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་ཀྱི་ཚོཚེ་ཀ་ཁ་སྦྱོར་བ་ཁྲོ་ཙ། བའི་དཔེ་ནི། སུ་ཁ་མ་ནུ་བ་བྷ་བ་དེ་བ་དཏྟེ་ན། ལྷས་བྱིན་གྱིས་བདེ་བ་རྗེས་སུ་མྱོང་བར་གྱུར་ཅིག། ༈ཌུ་པ་ཙཥཱ་འཚེད་པ་ལའང་གཉིས་བརྗོད། སྔོན་མའི་ཧ་ས་དང་པོ་་་ལྷག་པོ་དང་། གྲྭ་པཛྲ་རྣམས་ཀྱི་ཀ་འགྲོ་བ་ལ་དབྱི་སྟེ་ཨའི་ཡང་ཨེའོ་སོགས་ཀྱིས་བསྒྲུབས་པས། པེ་ཅེ། པེ་ཙཱ་ཏེ། པེ་ཙི་རེ། པེ་ཙི་ཏེ། པེ་ཙཱ་་ཐེ། པེ་ཙི་དྷེ། པེ་ཙ། པེ་ཙི་བ་ཧེ་པེ་ཙི་མ་ཧེ། ལོ་དང་པོ་བ་རགསྨ་ལགས་ན་ད་ཤཀྱི། །བ་འདུ་::༈དེ་བཞིན་དུ་ཏ་ནུ་ཡང་བ་ལ་ཏེ་ནེ། ཏེ་ནཱ་ཏེ། ཏེ་ནི་རེ་སོགས་དང་། ༈མན་ལ། མེ་ནེ། མེ་ནཱ་ཏེ། །མེ་ནི་རེ་སོགས་སོ། །ཀླད། ཀ་་་་ཎ་འོད། །།༈ཌུ་བཱཉ་་སྦྱིན་པ་ལ་པཛྲ་སོགས་མིན་པས་ཨ་ཨེར་བསྒྱུར་བ་སོགས་མི་་་་བྱེད་པས། ད་དེ། ད་དཱ་ཏེ། ད་དཱི་རེ་སོགས་སོ། །།ཨ་མུ་ཙ་སོགས་། ༈ཥཱ་གཙང་བར་བྱེད་པ་ལ། །སསྣེ། སནྣཏེ། སསྣེ་རེ་སོགས་ཏེ། ཚིག་གྲོགས་དང་སྦྱར་བའི་དཔེ་ནི། པཾ་པཱ་ས་ར་སི་རཱ་མེ་ཎ་ཡཏྠཾ་ཏཾ་ཤཱིཏ་ཡཱ་ས་ཧ། །ཡ་དི་ཛཱ་ནཱ་སི་ཏ་ཏཱ་༔ད་སསྣེ་ཧཾ་ས་བི་ལཱ་ས་ཡཱ། པདྨའི་མཚོ་ལ་དགའ་བྱེད་དང་། །གང་དུ་ཤཱི་ཏ་་ལྷན་ཅིག་རྩེ། །གཙང་བྱེད་གང་དང་གང་ཡིན་པ། །དེ་དེར་་་ངང་པ་རྣམ་པར་རོལ། །།།བཤེས་གལ། བར་སྐུ་གང་ལ་ཁཁ་ཁྱེབ་ཀཁྲཔདེད༈གླི་དང་གླུ་དགའ་བ་ཟད་:པ་ལའང་མཚམས་སྦྱོར་གྱི་ཡི་གེ་རྣམས་ཀྱི་ཨཱའོས་ཨེ་ཨཱ། གཉིས་བརྗོད་ཐུང་ངུའོས་ཐུང་ངུ་དང་། སྔོན་མའི་ཧ་ས་དང་པོ་་ལྷག་པོ་ཀུཧ་དག་གིས་ཙུའོ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཛགླེ། ཛ་ཏེ། ཛགླི་རེ་སོགས་དང་། དེ་བཞིན་དུ། ༈མམྨེ། །མ་ཏེ། མལླི་རེ་སོགས་སོ། །བྱིངས་དབྱངས་ཀྱི་མཐའ་ཅན་རྣམས་དང་ཧན་དང་གྲཧྲ་དང་དྲི་ཤ་ལ་སོགས་པ་་་རྣམས་ལ་དངོས་པོ་དང་ལས་ཀྱི་དོན་ལ། སི་ས་ཏཱ་སཱི། ཌ་སྱཔྲ་རྣམས་ཨིཊ་དང་་བཅས་པ་རྣམས་ཎ་འགྲོ་བའི་མིང་ཅན་དུ་བྱེད་པ་དང་མི་བྱེད་པ་འདམ་ཀ་ཅན་དུ་།
