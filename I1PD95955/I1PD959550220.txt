ད་དྷེ། ་ད་དྷ་ཧེ། དདྨ་ཧེ། པ་ལ། །ཁྱེད་ལ་འབད་པ་ལ་བྱེ་པ་༈ཆོག་འབྱུང་ལིང་ལ། ད་དྷཱི་ཏི། ད་དྷཱི་ཡཱ་ཏྭཾ། ད་དྷི་རན་སོགས་དང་། །དང་༈། ཤིས་བྱེད་ལོཊ་ལ། དྷཏྟཾ། ད་དྷཱ་ཏྭཾ། ད་དྷ་བྷཱུཏྭཾ། དྷཧཱ། ད་དྷཱ་ཤྲཱི། དྷདྡྷཾ། ད་དྷི། ད་དྷཱ་བ་ཧི། ད་དྷཱ་མ་ཧི། ཁ་ལ་རག་༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། །ཨ་དྷཏྟ། ཨ་ད་དྷཱ་ཏཱ། ཨ་ད་དྷ་ཏ། ཨ་དྷངྒྷཨ་ད་དྷཱ་ཐཱ། ཨ་དྷདྡྷཾ། ཨ་ད་དྷི། ཨ་ད་དྷཱ་ཧི། ཨ་དངྒཱ་ཧི། སོགས་ཏེ། དེ་རྣམས་ཀྱི་རིགས་འགྲེ་དེད་དེ་ཤེས་པར་བྱའོ། །ལ་འདེ། ༈ད་ནི་བྱིངས་དིབྲ་ལ་སོགས་པ་ཨཔྲ་ཀྱི་ཡུལ་ལ་ཡ་བི་ཀཱ་ར་ཎ་འཇུག་པ་རྣམས་ཀྱི་བྱ་བ་རྣམས་བརྗོད་པར་བྱ་སྟེ། དེ་ཡང་དེ་བུ་རྩེ་བ་དང༌། །མངོན་པར་འདོད་པ་དང་། ཐ་སྙད་དང་། སྡུག་གུའམ་འདུ་འགོད་དང༌། གསལ་བ་དང་། བསྟོད་པ་དང༌། རྨི་ལམ་དང་། མཛེས་པ་དང༌། དགའ་བ་དང་། །ཌ་མྱོས་པ་དང༌། འགྲོ་བ་རྣམས་ལའོ། །ཞེས་པའི་བྱིངས་ལ་བརྟ་མ་ན་སོགས་ཀྱི་པ་རསྨི་པ་ད་རྣམས་སྦྱོར་ཏེ། བརྟ་མ་ནའི་སྐྱེས་བུ་་དང་པོའི་གཅིག་ཚིག་ལ། །དིཔྲ། ཏིཔྲ་ཅེས་གནས་པ་ལ། ༈དི་་་བཱ་དེདིཔྲ་ལ་སོགས་པ་ལ་ཡའོ། །བྱིངས་དིཔྲ་ལ་སོགས་པའི་སྡེ་་ཚན་རྣམས་ལ་བྱེད་པ་པོའི་དོན་གྱི་ཨཔྲ་བི་ཀཱ་ར་ཎ་འི་ཡུལ་ལ་ཡ་བི་ཀཱ་ར་ཎ་་འཇུག་སྟེ་ཡ་བི་ཀཱ་ར་ཎ་སྦྱར། །ཨི་ཨུ་སོགས་ཀྱི་མདོས་རིང་པོར་བསྒྱུར་བས། དཱི་བྱ་ཏི། དེ་བཞིན་དུ་བསྒྲུབས་པས་གཉིས་ཚིག་དང་མང་ཚིག་ལ། དཱི་བྱ་ཏདྲི་བྱནྟི། སྐྱེས་བུ་དབུས་མ་དང་ཕྱི་མ་ལ། དཱི་བྱ་སི། དཱི་བྱ་ཐདཱི་བྱ་ཐ། དཱི་བྱཱ་མི། དཱི་བྱཱ་བདྲི་བྱཱ་མ། །་་ན་༈དེ་བཞིན་དུ་བསྒྲུབས་པས་ཆོག་འབྱུང་ལིང་ལ། དཱི་བྱེཏྲ། དྲི་བྱེ་ཏྭཾ། —།
