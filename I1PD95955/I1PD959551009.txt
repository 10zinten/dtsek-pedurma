བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཨི༽་མ་ཎའི་བོ་དྷི་དྲུ་མ། བྱང་ཆུབ་ཤིང་སྟེ། འདིའི་རྩ་བར་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཀྱི་བྱང་ཆུབ་མངོན་དུ་མཛད་པས་སོ། །ཙ་ལ་ད་ལ། འདབ་བསྐྱོད་ཅེས་བྱ་སྟེ། འདིའི་འདབ་མ་རྟག་ཏུ་གཡོ་བས་སོ། །སིས་ལ་ཡང་བྱང་ཆུབ་ཤིང་སྟེ། ཌརིན་་ཆེན་མཛོད་དུ། སིས་ལ་དང་བྱང་ཆུབ་ཤིང་དང་ཨ་ཤཱཏྠ་རྣམས་རྣམ་གྲངས་སོ། །ཞེས་སོ། །ཀུཉྫ་ར་ས་ནི། གླང་པོའི་ཟས་ཏེ། གླང་པོ་ཆེ་རྣམས་ཟ་བར་བྱེད་པས་སོ། །ཨ་ཤཱཤྩ་རྟ་གནས་སམ། ཁྱད་པར་གནས་ཏེ། ཨ་ཤྭ་ཏིཥྛ་ཏི་རྟ་ལ་གནས་པར་བྱེད་པས་སུས་ལས་པོ་ཞེས་པས་ཀའི་རྐྱེན་དང་། པྲྀ་ཥོ་དར་ལ་སོགས་པའོ། །ན། ང་དང་འདཀའ་ལ་ཡིག་མཁན་དང་བདེ་དེ་ནས་དནྟ་ཤ་ཟའི་བར་ནི་ཤིང་ཀ་པིཏྠའི་མིང་སྟེ། དེ་ལ་ཀ་སདྡྷི། སྤྲེའུ་ལ་གནས་པས་སྤྲེའུ་གནས་ཏེ། ཀ་སོ་ཏིཥྛ་ཏ་ཀ་ཏིསྦྲ་ཞེས་སོ། ་། ད་དྷི་ཏྠ་ཞོ་བཞིན་གནས་པས་ཞོ་གནས་ཏེ། ད་དྷི་བ་ཏིཥྛ་ཏི་ཏིཥྛ་ཞེས་སོ། །་། གྲྭ་ཧཱི་དང་། མམྨཐ་ཞེས་པ་ཡང་དེ། །འཛིན་མ་དང་། ཡིད་སྲུབ། :ཅེས་བྱའོ། །དདྷི་ཕ་ལ་ཞོ་འབྲས་ཏེ། འདིའི་འབྲས་བུ་ཞོ་ཞིམ་པ་ལྟར་ཡོད་པས་སོ། །།པུཧྥ་ཕ་ལ། མེ་ཏོག་འབྲས་ཏེ། མེ་ཏོག་ལས་འདིའི་འབྲས་བུ་འབྱུང་བས་སོ། །དནྟ་ཤ་ཐ། སོ་རྩེ་ཉམས་བྱེད་དོ། །།འདིའི་ཆུ་སོ་ལ་གནོད་པས་སོ། །་ཥ་། །ཏིན་ཏུ་ན། །དང་དེ་ནས་ཤི་བི་ཨུ་དུམ་གང་ལ་མིང་བཞི་སྟེ། ཨུ་དུམྦ་ར་ཞེས་པ། "་ས་ཏ་་་སྟེའི་གསུམ་པ་ཅན་ཉིད་ཡང་ན། ཨུ་ཊམྦར་ཞེས་པ་ཡང་འགྱུར་ཏེ། །དེ་ནི་་་བདུད་འཇོམ་སཾ་ཞེས་བརྗོད་དོ། །འདྲ་བའི་སྒྲ་ནི་ངེས་པར་བསྟན་པའི་དོན་གྱི་་་ཕྱིར་རོ། །ཛན་པ་ལ། ཛརྫུའི་འབྲས་བུའོ། །ཡ་ཛྷཱངྒ། མཆོད་སྦྱིན་ཡ་་་་ལག་གོ། །ཧེ་མ་རུབྷ་ཀ། གསེར་གྱི་འོ་མ་ཅན་ཏེ། གསེར་གྱི་མདོག་ཅན་གྱི་འོ་མ་འབྱུང་བས་སོ། །སྒོ་། །འ་དང་ཀླུ་ཞི་གཡོ། ཊ་འོ། —།
