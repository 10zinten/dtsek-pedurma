།པའོ། ༄༅། །དེ་ཡང་ཨུ་མཐའི་བྱ་བ་ནི་རྟག་པའི་ཆོ་ག་དང་རཾ་མཐའི་ཆོ་ག་ནི་མི་རྟག་པ་བཅོས་མའི་ཆོ་ག་ཡིན་པས་རྟག་ཏུ་རྟག་པའི་ཆོ་ག་སྟོབས་ཆེ་བས་དྲུག་པའི་མང་ཚིག་ལ་ཨུ་མཐའི་ཆོ་ག་ཁོ་ན་བྱ་སྟེ། ཀྲོཏྟཱ་ནཾ། ཨཱམ། ཟླ་བདུན་པ་གསུམ་ལ། ཀྲོཋ་རི། ང་། :ཀྲོཌོ། ང་། ་ཀྲོཌོཨོསྲ། ཀྲོཌོཡཨོས། ཀྲོབྷུ་ཥཱ། སུཔྲ། བོད་ཚིག་གསུམ་ལ། ཧེ་ཀྲོཌོ། སི། ཥ་ཧེ་ཀྲོཋ་རོ། ཧེ་ཀྲོ༴་རཿདེ་རྣམས་རྣམས་ཀྱི་སྐྱེས་པའི་རྟགས་རཾ་ཡིག་མཐའ་ཅན་རྣམས་སོ། །ཀ་བའོ། །།ཆུ་ཡི་ཤི་གཅཛཱ་ཡིག་དང་ལོ་ཡིག་དང་ཨེ་ཡིག་རྣམས་ཀྱི་མཐའ་ཅན་ནི་རབ་ཏུ་མ་གྲུབ་བོ༑་༈། སྐྱེས་པའི་རྟགས་ཨི་ཡིག་མཐའ་ཅན་ལས་སུ་རིའི་་ཞེས་པའི་སྒྲ་ལ་མཚོན་ན༑དང་པོའི་གཅིག་ཚིག་ལ། རི་སྙི། རིའི་ས་དང་བྷ་ལའོ། །རེ་ལྟ་བུའི་ཨེ་ཡིག་མཐའ་ནི་（རྣམ་དབྱེ་ས་ཡིག་དང་བྷ་ཡིག་དང་པོ་རྣམས་ཕ་རོལ་ལ་ཨཱ་ཡིག་ཏུ་འགྱུར་ཏེ༑། ༈སུ་རཱནོར་ཅན། ཌ་དབྱངས་ཐོག་མ་ཅན་ཐམས་ཅད་ལ་ཨཱཡ་རུ་བསྒྱུར་་་བས་གཉིས་ཚིག་དང་མང་ཚིག་ལ། སུ་རཱ་ཡོ། ཨོ། སུ་རཱ་ཡཛས། གཉིས་པ་གསུམ་ལ། སུ་རཱ་ཡཾ། ཨམྤ། སུ་རཱ་ཡོ། ཨོ། །སུ་རཱ་ཡ༔ཤེས། གསུམ་པ་གསུམ་ལ། སུ་རཱ་ཡཱ། ཊཱ། སུ་རཱ་ཧཱུཾ། སུ་རཱ་བྷིསོགས་སོ། །།།p། སྐྱེས་པའི་རྟགས་ཨོ་ཡིག་མཐའ་ཅན་གོའི་སྒྲ་ལ་མཚོན་ན། ཨོ་རོ། ཨོ་ཨོའོ༑། །སི་སོགས་རྣམ་དབྱེ་ཐོག་མ་ལྔ་པོ་གང་རུང་ཕ་རོལ་ལ་རྟགས་མཐའ་་ཨོ་ཡིག་ནི་ཨོ་'ཡིག་ཏུ་འགྱུར་ཏེ། དང་པོ་གསུམ་ལ། ༈་བ་་ལང༌། གོསི། ཁགཱ་བོཙེ། ཨོ། གཱ་བཛསྲི། གཉིས་པའི་གཅིག་ཚིག་ལ༑་པཨམྱ་སི། ཨམ་ཤས་ལའོ། །ཨོ་ཡིག་མཐའ་ནི་རྣམ་དབྱེ་ཨམ་དང་ཤས་ཕ་རོལ་ལ་ཨཱ་ཡིག་ཏུ་འགྱུར་ཏེ། ཧཱུྃ། ཨམ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། གཱ་བོ༑ཡཨོ་:གཱཤས། གསུམ་པ་གསུམ་ལ། །ག་བཱ། ཊ། གོ་ཧཱུྃ། །གོ་བྷི། །
