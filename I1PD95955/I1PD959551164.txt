ཤཱ་ཁ་ལུ་ཡ་ཏྲ་ཏོ་ཛ་རོ། །པུ་ཏྲ་ཨི་ཧ་བ་སུ་བེ་དཱ་ནོ་ག་བོཟོ། །རཾ་ཏཱུ་ཛ་སུ་ངེས་པར་དབང་པོའི་རྒྱུད་འགྱུར་རོ། །ན་དག་ག་བ་ར་བ་ནོར་རིག་བྱེད་འདིར་སྦྱོར་བའོ། །དུས་དག་ཛ་སཱུ་ཛ་ལོདྡྷ་ཏ་ག་ཏི། །ཨ་ཐ་ཏོ་ཊ་ཀ་མདྡྷི་ས་ཀཱ་ར་ཀྲྀ་ཏཾ། །དྲུ་ཏ་བི་ལཾ་བི་ལ་ནི་ཛ་སུ་ཛ་སུ་ཆུ་འཐོར་འགྲོ། །ས་ཡིག་ཆུ་གཏེར་བྱས་པ་དེ་ནས་འགྲོ་བྱེད་ཡིན། །ན་དང་བྷ་དང་བྷ་ཏ་མཱ་ཧ་ན་བྷོ་བྷ་རོ། །བྷུ་ཛངྒ་པྲ་ཡཱ་ཏཾ་ཙ་ཏུ་བྷིརྨ་ཀཱ་རི། །ཀཱིརྟི་ཏི་ཥཱ་ཙ་ཏུ་རེ་ར་དལ་མྱུར་ཡིན་པར་སྨྲ། །།ཡ་ཡིག་བཞི་རྣམས་ཀྱིས་ནི་ལག་འགྲོ་ལྟུང་བ་ཡིན། །ར་ཡིག་བཞི་པ་དེ་ནི་མེ་ཕི་ཀཱ་པྲ་ཤྲཱི་ཎཱི། །པྲ་མི་ཏཱཀྵ་རཱ་ས་ཛ་ས་སིཀ་ཐི་ཏཱ། །མ་་རེ་ཕོ་བཱ་བི་ཤཱ་དེ་བཱི་ཏོག་ཕྲེང་ཅན་གྲགས། །ས་ཛ་ས་ས་རྣམས་ཀྱི་ཡི་གེ་ཉུང་ངུར་གྲགས། །མ་ག་ཝ་ར་ར་ཝ་མདའ་རྟས་སྣ་ཤ་ར་ཤྲཱི། །མོ་བ་རྒྱ་གཱ་ཛ་ལ་དྷ་ར་མཱ་ལཱ་བྦྷངྒི། །།ན་ན་རར་གྷ་ཊི་ཏཱ་ཙ་མནྡཱ་ཚོགས་ལྷ་མོའོ། །མ་བ་ན་ཡ་ག་ཝ་རྒྱ་མཚོ་ཡན་ལག་ཆུ་འཛིན་ཕྲེང་། །།ན་ན་ར་ར་འདུས་པ་དལ་གྱིས་ཀི་ནཱི། །བ་ད་ཏི་མུ་ནིརྨ་མུ་ནཾ་ན་ཇོ་ཛ་རོ། །ཨི་ཧ་ན་བ་མཱ་ལི་ནཱི་ཡ་དི་ན་ཇོ་བློ། །འབབ་པ་ཡིན། །ན་ཛ་ཛ་ར་ཡ་མུ་ནཱ་རུ་ཐུབ་པ་སྨྲ། །གལ་ཏེ་འདིར་ནི་ན་ཛ་བྷ་ཡ་ཕྲེང་བ་གསར་པའོ། །ཀླུ་ཤཱ་བྷིརྨུ་ན་ཛ་ར་ཝཱ༔པྲ་ཧནྟི་ཎཱི་སཱུཏི། ་། ཛ་བོ་ན་ཛོ་རི་ཏི་རུ་ཙི་རཱ་ཙ་མུ་ན་ཛ་ར་ཝ་རྣམས་གསུམ་དང་ཕྱོགས་ཀྱིས་རབ་དགའ་ཅན། །ཛ་ཝ་ན་ཛ་ར་ཞེས་བཞི་དང་གཟའ་ཡིས་ཏུརྦ་ཧི། །མོ་བོ་བྷཏྟཱསི་བ་ཡ་ཏི་རྨཏྟ་མ་ཡུ་རཿ། །ས་ཛ་སཱས་གོ་མཛེས་མ་ཡིན། །མ་བ་བྷ་ཏ་ཡ་རྣམས་དེ་ཉིད་ངལ་བསོ་རྨ་བྱ་མྱོས། ་། ས་ཛ་ས་དང་ས་ག་རྣམས་ནི་ཙ་ཀ་ཐི་ཏཀ་ལ་ཧཾ་ས། །ས་ཛ་སཱ་ཛ་གོ་ཙ་ཡ་དི་མཉྫ་བྷཱ་ཎི་ཎི། །མཱ་གཱ་བཱ་ནོ་སྙན་པའི་ངང་པར་བཤད། །གལ་ཏེ་ས་ཛ་ས་རྣམས་དང་ནི་ཛ་ག་འཇམ་པའི་ངག། མ་ག་ཝ་དང་ན་མཱཤ་ར་ན་བ་བྷི་ར་སཾ་བཱ་དྷཱ། །ན་ན་བི་ཛ་ཛ་བི་ཙ་སཱ་རི་ར་པ་གཉིས་མ་རྣམས་མདའ་དང་དགུ་ཡིས་གནོད་མེད་དོ། །ན་ན་བ་དང་ཛ་ཛ་ཝ་རྣམས་དབྱངས་ཀྱིས་གཞན་རཱ་ཛི་ཏཱ། ་། པྲ་ཧ་ར་ཎ་ཀ་ལི་ཏཱ་ན་ན་བི་ན་ན་བི་ཏི། །།ཛྷེ་ཡཾ་བ་སནྟ་ཏི་ལ་གྱི་མི་ཐུབ་པ། །ན་ན་ཝ་དང་ན་ན་ཝ་རྣམས་དེ་ཡིས་མཚོན་ཆ་ཁ་འབུས་སོ། །ཏ་བྷ་ཛ་ཛ་ག་ག་དཔྱིད་ཀྱི་—།
