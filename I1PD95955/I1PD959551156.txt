ཀ་གང་ཞིག་ཐུགས་དགོངས་རྣམ་པར་དག་པ་ཡིས། །འདི་ལ་བསྐུལ་ཞིང་་་་མཐུན་རྐྱེན་བསྒྲུབས་པ་དང༌། །བདག་ཅག་འབད་ལས་བསོད་ནམས་གང་་་་བསྒྲུབས་པས། །འཁོར་བ་ལས་རྒལ་སྐུ་གསུམ་ཐོབ་པར་ཤོག། བྱང་གི་བདག་པོའི་གསུང་བཟང་ལས། །སྙན་ངག་འདི་ནི་བོད་སྐད་དུ། །ཁ་ཆེ་པཎ་ཆེན་་་དང་ལྷན་ཅིག། དགེ་སློང་བྱང་ཆུབ་རྩེ་མོས་བསྒྱུར། །འདིས་འགྲོ་བ་དཔག་ཏུ་མེད་པ་ལ་ཕན་ཐོགས་ནས་བཀྲ་ཤིས་ཤིང་བདེ་ལེགས་སུ་གྱུར་ཅིག། །།ཁཚོ་n@་་་ལ། ་ལོ་ཉིས་བར་བྱོའི་གནས། །ལ་ན་དེ་གྱུར་གོད། དཔེ་གསར་པ་ན། །ང་ད་ན་བ་མ་དེ་མ་ཀྱན་དང་། ཁ་འཁང་ར་བ་བ་ལྟ་འི་《མེ་:::ཀྱིངབདར་་། །།ཁ་《དྲ་བ་བཤད་པའི་རྒྱ་ཆེ་བསྡུར་མཆན། དེར་བ། །དབང་་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། :«པེ་ལ་སྤྲིན་གྱི་ཕོ་ཉ་་་་བཞུགས། «སྣར་»༄༅། །སྤྲིན་གྱི་ཉ་བཞུགས། ཞེས་འཁོད། «པེ་»མེ་གྷ་དྷུ་ཏ་ན་ནཱ་«སྣར་མ་དང་གྷ་དྷུ་ཏ་ན་ན། «པེ་»«སྣར་པས། །ཚེ«པེ་»«སྣར་»སྤོང༌། «པེ་»«སྣར་ཆུ་བོ། «པེ་»མི་འི་《སྣར་རཱ་མའི་རི། /a}་སྐབས་བར་དང་། དེར་་«པེ་»«སྣར་སྦྱིན། «པེ་»སྣར་»འབྲང་། «པེ་»«སྣར་»འཆང་བ། །དེ། ®«པེ་»པ། 《སྣར་»ཀུ་ཊ་ཛ། །«པེ་»«སྣར་»ནི་དེ་ལ། «པེ་»«སྣར་བས། @«པེ་»«སྣར་»ལ། «པེ་»་ཀྱི་«སྣར་»གི། (@«པེ་»«སྣར་»གཉེར། སྐྱེ«པེ་»«སྣར་»མདུན། «པེ་»«སྣར་མ་ལ། །གཏན། (ཉེ«པེ་»«སྣར་ཆེས། ང་ལ་དོསའིབདག་«པེ་»«སྣར་»ཚལ། ::(«པེ་»སྣར་བཟང་། «པེ་»«སྣར་»—བསྟི་གནས་ཆེན་པོ། «པེ་»་ཁ《《སྣར་»མཁས། མ«པེ་»«སྣར་»ལན། «པེ་»«སྣར་བྲལ། @«པེ་»«སྣར་»མཛེས་མའི། པ«པེ་»«སྣར་པ། «པེ་»«སྣར་རྗེས། «པེ་»«སྣར་»བསྟོད་པའི། ཀྱེ«པེ་»«སྣར་སྒྲོག(མེ«པེ་»«སྣར་»བསྟེན། (《སྣར་»བརྗོད། རྩེ«པེ་»«སྣར་»སྐྱེས། «པེ་»«སྣར་»འཕྲལ། /《ཨེ«པེ་»«སྣར་»ཀྱིས་འབྲུག་སྒྲ་བསྒྲགས་པ་དེ་ནི་བསྐལ། །།«པེ་»«སྣར་»ར་གྱུའི། «པེ་»«སྣར་»ལོང་། «པེ་»«སྣར་བསལ། &«པེ་»«སྣར་མ། དང་། རི«པེ་»«སྣར་»འཇལ། «པེ་»«སྣར་»ཀཚེ«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་ཕྲིན། 《«པེ་»«སྣར་»སྐྱེས་ཁྱོད་ཀྱི་རྣ་བས། །(ཁྱེ«པེ་《སྣར་བས། ཀྱེ«པེ་»«སྣར་»ལོང་། སྟེ«པེ་»།
