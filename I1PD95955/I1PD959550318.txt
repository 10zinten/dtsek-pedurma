སོགས་སོ། །།་བས། དམ་པ་་་་་་་་་་་བས་བ་ག་པའི་དབང་བQ༈རྀ་འགྲོ་བ་ལ། གཉིས་བརྗོད། རོ་དང་པོ་ཨི་འདྲ་བ་མ་ཡིན་པའི་་་དབྱངས་ཕ་རོལ་ལ་སྔོན་མའི་ཨི་ནི་ཨིཡ་ཞེས་པར་བསྒྱུར་བར་བཤད་པས། ཨི་ཡརྟི༑/ཨི་ཡཾ་ཏ། ཨི་ཡནྟི། །ལིང་། ཨི་ཛཾ་ཡཱཏྲ། ཨི་ཡཞུ། ལོཊ། ཨི་ཡཾ་ཏཱཏྲ། ་ཨཊ། ཨི་ཡ་རཏྲ། དིཔྲ། ཨེ་ཡསོགས་སོ། །ཨིཎ་འགྲོ་བ་ལ། ཨེ་ཏི། ཨི་ཏ་ཡནྟི། ཨི་ཡཱཏྲ། ཨི་ཡཱཏཱི། ཨི་ཡུ། ཨི་ཏུ། ཨི་ཏཱཏྲ། ཨི་ཏཱཾ། ཡནྟུ། ཨེཏྲ། ཨེ་ཏྭཾ། ཨཱ་ཡན་སོགས་དང༌། དངོས་པོ་དང་ལས་ཀྱི་དོན་ལ། བཞི་པོ་ལ་ཡཀྲ་འོས། ཡཀྲ་བྱིན། ཡ་ལའོས་རིང་པོར་བསྒྱུར་བས། ཨཱི་ཡ་ཏེ་སོགས་དང་། ལྐོག་གྱུར་ལིཊ་ལ། ཨཱི་ཡཱ་ཡ། ཨཱི་ཡ་ཏུཨཱི་ཡུསོགས་སོ། །ཨི་ཏུ་འདོད་པ་ལ། ལྐོག་གྱུར་ལིཊ་རྣམས་སྦྱར། གཉིས་བརྗོད། སླར་ཡང་་ཨི་ཕྱི་མ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨི་ཡེ་ཥ། ཨཱི་ཥ་ཏུ། ཨཱི་སོགས་སོ། །།དྷཱ་ཏུ་ནཱ་མ་ནནྟ་ཏཱ་ཏྲ་བི་ཤཱརྦ་ཏཱཉྩ་སརྦ་ཐཱ། ཨ་བྷི་དྷཱ་ཏུ་མ་་ཤཀྱ་མི་ཏྱ་ལ་མཱ་ཁྭ་ཏ་ཁཱ་པ་ནེ། དེ་ལྟར་བྱིངས་རྣམས་ཀྱང་ནི་མཐའ་ཡས་ཉིད། །སྣ་ཚོགས་དོན་ཉིད་ཀྱང་ནི་ཐམས་ཅད་དུ། །བྱིང་རྣམས་འདི་ཞེས་མངོན་གསལ་མི་ནུས་ཀྱང༌། །ཀུན་ནས་བཤད་པ་རྫོགས་པར་བཤད་ཟིན་ཏོ། །ཨི་ཏི་སྭ་ར་སྭསྟི་བྱུ་ཀཱ་ཀེར་ཎ་སྱ་སཱུ་ཏཱ་དཱ་ཀྵ་ཏ་པྲ་ཀྲི་ཡ་ས་མཱབྷཱ། ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཀུན་ནས་བཤད་པའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ། །ལ་། དུ། བོ་ཨ་ཙ་ད། ཁོང་ཀོ:པའི་ནང་ལ་་བབབ་ང་དེ་ནས་བྱིངས་ལས་མིང་དུ་བསྒྲུབ་པ་ཀྲྀཏ་སོགས་ཀྱི་ཆོ་ག་རྣམས་བཤད་་་་པར་བྱ་སྟེ། ༈ཀྲྀཥ་རྟ་རི། །བྱེད་པ་པོ་ལ་ཀྲྀཏྲའོ། །འཆད་འགྱུར་ཏྲི་ལ་་་སོགས་པའི་རྐྱེན་རྣམས་ལ་ཀྲྀཏྲ་ྋརྐྱེན་ཞེས་བྱ་ལ། །རྐྱེན་དེ་རྣམས་གཙོ་བོར་བྱེད་།
