།།ཁ་ཁ་པ། བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིའིཡི། ལ་་ལ་སྒྲོགས་ཁོང་དན་ཤ་ཏཱཾ་བཱཏྟཱ་མྲ་ལོ་ཎི་ཀཱ། །ས་ཧ་སྲ་བེ་དྷི་ཙུ་ཀྲོཉམ་བེ་ཏ་སཤ་ཏ་བི་དྷུ་པི། །སོ་རྩེ་མོ་དང་སྐྱུར་མོ་དང་། །སྟོང་འབིགས་དང་ནི་ཙུ་ཀྲོ་དང༌། །སྐྱུར་མོ་དང་ནི་བརྒྱ་འབིགས་ཀྱང་། །ཏ་མསྐུ་རཱི་གཎྜ་ཀཱ་ལཱི་ས་མཾ་གཱ་ཁ་ད་རཱི་ཏྱ་པི། །ཛཱི་བནྟཱི་ཛཱི་བ་ནཱི་ཛཱི་བཱ་ཛཱི་ཕྱག་འཚལ་དང་ནི་གཎྜ་ཀཱ་ལཱི། །ལུས་མཉམ་དང་ནི་སེང་ལྡེང་ངོ༌། །འཚོ་བྱེད་ཛཱི་བ་ནཱི་ཛཱི་བཱ། །ཛཱི་བ་ནཱི་ཡཱ་མ་དྷ་སཱ། །ཀཱུརྦ་ཤཱནྟེ་མ་དྷ་ར་ཀཿཤྲངྒ་ཧྲ་སཱངྒ་ཛཱི་བ་ཀཱ། །ཀི་རཱ་ཏ་ཏིཀོ་བ་ནི་ཡ་མངར་མོ་ཡང་། །ཀུརྦ་ཤིནྟེ་མ་རྒྱུ་རཀ། རྭ་ཅན་ལུས་ཐུང་ཛཱི་བ་ཀཱ། །ཀི་རཱ་ཏ་ཏིཏྟ་བྷཱུ་ནིམྲྀཞནཱརྨ་ཏིཀྟཉཐ་སཔྟ་ལཱ། ་། བི་མ་ལཱ་ཤཱི་ཏ་ལཱ་བྷཱ་རི་ཕེ་ནཱི་ཙརྨ་ཀ་ཤེ་རྐྱ་པ། །སའི་ནིམྤ། །ཨ་ན་མ་ཏིཀྟ་དེ་ནས་བདུན། །དྲི་མེད་བསིལ་འཛིན་བྷུ་རི་དང༌། །ཕེ་ནཱི་ཙརྨ་ཀ་ཤེ་ཏི། །བཱ་ཡ་སོ་ལཱི་སཱ་དུ་ར་སཱ་པ་ཡསྠཱ་ཐ་མུ་ཀཱུ་ལ་ཀ། །ནི་ཀུནྡྷོ་དནྟི་ཀཱ་བྲ་ཏྱཱཀྵ་རྒྱུ་ན་ཚོད་འཛིན་དང་མངར་བ་དང༌། །ན་ཚོད་གནས་དང་མུ་ཀུ་ལ། །སོ་ལྡན་ནི་ཀུནྡྷ་པྲ་ཏྱཀ། ཐེ་ཏུ་ཨུ་དུམྦ་ར་བནྡྷུ་པི། །ཨ་ཛ་མོ་དཱ་ཏུ་གྲ་གནྣཱ་བྲ་རྨ་གནྡྷ་ཡ་མཱ་ནི་ཀཱ། །མཱུ་ལེ་པོཥ་དུམ་བཱ་རའི་འདབ། །ར་དགའ་མོ་མ་དྲག་པོའི་དྲི། །ཚངས་པའི་སྙིང་པོ་ཐུན་ཚོད་ཅན། །རྩ་བ་རྫིང་ར་ཀཱམཱི་ར་པདྨ་པཏྟཱ་ཎི་པོཥ་རེ། །ཨ་བྱ་ཐཱ་ཏི་ཙ་རཱ་པ་ཙ་ར་ཊི་པདྨ་ཙཱ་རི་ཎཱི། བུ་ཁ་ཆེ་དང་། །པདྨའི་འདབ་དང་རྫོང་ལས་སྐྱེས། །ནད་མེད་ཨ་ཏི་ཙ་རཱ་དང་། །པདྨ་ཙཎྜི་པདྨ་སྤྱོད། །ཀམྲྀལྤཀཀྵ་ཤ༔ནྡོ་རཀཾ་གོ་རོ་ཙ་ནཱི་ཏུ་པི། །པྲ་པུནྣཱ་ཌསྟེ་ཌ་ག་འདར་བྱེད་བཙུན་མོ་ཟླ་བ་དང་། །ཡན་ལག་དམར་དང་མཛེས་ལྡན་ཀྱང་། །པྲ་པུནྣཱ་ཌསྟེ་ཌ་གླང་། །ཇོ་ཏེ་དྲུ་བྷཱཉྩ་ཀྲ་མརྡ་ཀ༄༅། །པདྨ་ཊ་ཨུ་ར་ཎཱ་ཁྱ༔་པ་ལཱཎྜ་སྟེ་སུ་ཀཾ་ད་ཀ། །ཏེ་དྲུ་འཇོམས་དང་འཁོར་ལོ་འཇོམས། །པདྨ་ཊ་དང་ཨུ་ར་ཎ་བཤད། །མེ་མོ་དང་ནི་སྒེའུ་དང་། །ལ་ནཱརྐ་དུ་དྲུ་མོ་ཏ་ཏྲ་ཧ་རི་ཏེ་ཐ་མ་ཧོ་ཥ་དྷཾ། །ལ་ཤུ་ནཾ་གཱཉྫ་ནཱ་རིཊྚ་མ་ཧཱ་ཀནྡ་ར་སོ་བཙོང་དང་ཀུན་དོང་སྒོག་སྔོན་དང་། །དེ་ནས་སྨན་ཆེན་སྒོག་སྐྱ་དང་། ྋ། སྒོག་པ་ཨ་རིཥྛ་ཆེན་པོ། །ཟ། ན་ཀཱ། །པུ་ནནྣཱ་བཱ་ཏུ་སོ་ཐ་བྷིཔི་ཏུནྣཾ་སུ་ནི་ཥནྣ་ཀཾ། །སྡུདྡྷཱ་ཏ་ཀཤཱི་ཏ་ལོཉཔ་རོ་གཅིག་པུ་ནརྣ་བ་དང༌། །།སོ་ཐཙྪི་དང་བི་ཏུནྣི། །སུ་ནི་ཥསྣ་ཀཾ་དྷཱ་ཏ་ཀ། བསིལ་འཛིན་གཞན་གྱིས་།
