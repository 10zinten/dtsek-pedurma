བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི། ་པའི་ཀཚོས་ཁགatཀ་བློ་ཟmཁ་ཚོད་ཆེན་པོབོད་ཞུ་ཏིབ། ཀ་བ་ཀ་ཏོ་རྗེ་ལུ་་བུ། །ཁ་ན་འཚོ་བ། ཀླུ་བཏབློདང་ཁ་ཕྱོགས་ལ་ངངཏེ་འཚོ་བ་ལ་ཡོན་པཎ་ཊི་་བོ་་་་་་ཁོང་ཁོང་རྒྱལ་ཞེ་ཆུ་བ་དེ་དག་ད་ཆ་མང་བ་དང་བསྡུར་མཆན། ད་བ་ལ་ན་ཚུ་བཏོན་ཏེ་བ་དང་བད་ཀན་ཀཱ། :ལྟ་ཀློག:ནོ། ཀ་རྒྱ་དག། ::་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད། «པེ་»«སྣར་»ལ་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་དབྱངས་ཅན་བྱཱ་ཀ་ར་ཎའི་འགྲེལ་པ་རབ་ཏུ་བྱ་བ་གསལ་ལྡན་ཞེས་བྱ་བ་བཞུགས་ཞེས་འཁོད། «ཅོ་»མཛད། «སྣར་བྱ། ::ཀྱི《སྣར་»རི། ་འ《སྣར་ནི། «སྣར་+ཏུ། «པེ་བདེ་བས་འཚམས། 《སྣར་འདེ་བས་འཚམས། ས«ཅོ་ལི། པ«པེ་»«སྣར་བཎ། «སྣར་»དུར། ཏེ«སྣར་»ཏཾ། །.«པེ་»«སྣར་»འཚམས། །«པེ་»«སྣར་»མཐའ། བ«པེ་»ལཾ། པེ་མིའི། «སྣར་རྣམས། 《སྣར་བཟུང་། «པེ་བྲིདྡྷི། ་་་ན་་«པེ་ལ། «པེ་»«སྣར་»འཚམས། «སྣར་ཡ། ཆེ«པེ་སེརྦ་ཧྥ། སུ«པེ་»«སྣར་»«ཅོ་དྷ། །་«པེ་»«སྣར་པའམ། «པེ་པེ་རེ། «སྣར་»བེ་རེ། ་.«སྣར་རོག་«ཅོ་»པས་དྲངས་ར། །ས«པེ་»«སྣར་ཨུཝའོ། «པེ་»«སྣར་»ཏྲཟླ། «ཅོ་ལ། .«པེ་»«སྣར་»ཨེ་ཨ་ཡ། ::ཀྱི«པེ་»«སྣར་བ། ༌«ཅོ་ཨཱ། ：:ན«པེ་»«སྣར་»འཚམས། སུ«པེ་ཨཱཡ། «པེ་བྲ། །«པེ་»—དང་། ཉེ«པེ་»«སྣར་»གཱ། «པེ་»«སྣར་»འཚམས། ཀྱི«པེ་»«སྣར་ལ་སླར་འཚམས། «སྣར་ཁ། །།«པེ་ཧེ། ཝ«པེ་»དཾ། ས«སྣར་པ། 《སྣར་»ཏི། 《སྣར་»དབཤོལ། ར《སྣར་ནས།  ི«པེ་»«སྣར་ནཱི། «ཅོ་རཾ། «པེ་པཱ་ཎཾ། «སྣར་»དགྲངས། །«པེ་ཨེ། ་བ«སྣར་ཨེ། ས«སྣར་»བེ། «ཅོ་»བོ། «པེ་»«སྣར་ཨོཏྟཱིཏོ། «པེ་»«སྣར་»འཚམས། སྐལ་::ཀྱི«པེ་»«སྣར་»འཚམས། ་འ«སྣར་ཐོབས། «ཅོ་ཨེ། «པེ་»ཎི། །«པེ་»«སྣར་»འཚམས༑«པེ་»ག་ཁྱོ་ཤུགས། «སྣར་ན་ཁྱོད་ཤུགས། «པེ་»«སྣར་པ། རུ«པེ་»«སྣར་»འཚམས༑«ཅོ་»སམྦྷི། «པེ་»«སྣར་ཉེ། ::《སྣར་ཅེ། འ«པེ་»«སྣར་»ཊ། «པེ་»ཥརྨི། །་སུ«པེ་»«སྣར་»«ཅོ་»ཚ། པོ«ཅོ་པཱག། «པེ་»«སྣར་»ར། ར«སྣར་ཧ། «སྣར་»སྔན། ཁྱེ«པེ་»བ། ཀྱེ«ཅོ་»བཱཀ«སྣར་»དབསྲེག«སྣར་»སྟེ། «པེ་»«སྣར་ཀས། «ཅོ་གས། «པེ་»«སྣར་»ཀས། ཨེ«ཅོ་»ཏ་ཏ། འབྱེ«པེ་»«སྣར་»ཏནྟཱི། ནོབ། གྲུབ་བམ། ཚེ་དབང་(«པེ་—«པེ་»«སྣར་»ཏ། །«སྣར་»ཏེཏི། «སྣར་སྡོའི། «ཅོ་རནྟ། «པེ་»«སྣར་»སཀ། ༄༅«སྣར་»ལ། ཉི«པེ་»སཀྲའོ། «སྣར་»ཀའོ། «སྣར་ས་སར། «པེ་»ཏྲེ། པའིག་འཀྱེ«པེ་»«སྣར་»«ཅོ་»ཏི། «པེ་»«སྣར་ཙཀ། ཉོ་སྤྱོད་པ་ཡིན་ཏེ—།
