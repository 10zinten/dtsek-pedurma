།ཏདྨ་ཏེ་ག་ལྱཱ་སཱུ་ར་པ་རསྤ་རཱ། །སཱ་ཀ་ལྱཱ་སཾ་ག་བཉྩ་ནེ་པཱ་ར་ཡ་ཎ་པ་རཱ་བ་དེ་དག་ཏུ། །འགྲོ་བའོ་ཕན་ཚུན་མིན་པ་རྣམས། །ཚོགས་པ་ཟླ་ཡི་ཚིག་ལ་ནི། །ཕ་རོལ་འགྲོ་དང་ཕ་ཡ་ཎེ། །ཡ་དྲྀཙྪཱ་སཱི་རི་ཏཱ་ཧེ་ཏུ་ཤུ་ནྱ་ཏཱ་སྭཱ་བི་ལ་ག་ཎཾ། །ཤ་མ་རོལ་བརྟེན། །དབང་དུ་བྱེད་དང་རང་དབང་བྱེད། །རྒྱུ་སྟོང་གནས་དང་མཚན་ཉིད་བྲལ། །ཞི་དང་ཐསྐུ་ཤ་མ༔ཤཱནྟིརཱནྟི་སྐུ་ད་མ་ཐོ་ད་མ། །ཨ་བ་དཱ་ཎཾ་ཀརྨ་བྲྀཏྟཾ་ཀཱ་མྱ་དུལ་དང་དེས་པའོ། །བཟོད་དང་ནུས་དང་ཐུབ་པའོ། །རྟོགས་བརྗོད་སྔར་གྱི་ལས་འཇུག་གོ། །འདོད་དཱ་ཎ་པྲ་བཱ་ར་ཎཾ། །བ་ཤ་ཀྲི་ཡཱ་སམྦ་ད་ནཾ་མཱུ་ལ་ཀརྨ་ཏུ་ཀརྨ་ཎཾ། །བི་དྷུ་བ་ཎཾ་བི་པ་སྦྱིན་པ་རབ་མཆོག་གོ། །དབང་བྱེད་དབང་དུ་སྡུད་པ་དང་། །།རྩ་བའི་ལས་དང་ལས་སུ་བྱ། །གཡོ་བ་དུ་ན་ནཾ་ཏཔ་ཎཾ་པཱི་ཎ་ནཱ་བ་ནཾ། །པནཱསྟིསཱུཏྤ་རི་ཏཱ་ཎཾ་ཧསྟ་བཱ་ར་ཎ་མི་རྱ་ཝེཔི། །དང་ནི་འདར་བར་བྱེད། །ངོམས་པ་དང་ནི་ཚིམ་པའོ། །ཆོག་པ་ཡོངས་བསྲུངས་ལག་པ་འགེབས། །སེ་བ་ཎཾ་སཱི་བ་ཎཾ་མྱུ་བིཟེ་ད་རསྤུ་ཊ་ནཾ་བྷི་དཱ། །ཨཱ་ཀྲོ་ཤཱ་ན་མ་བྷི་ཎཾ་འཚེམ་པ་ཚེམ་བུ་སྦྱོར་བ་དང༌། །འདྲ་བ་གཅོད་པ་གཏུམ་པའོ། །ཁྲོ་བ་དང་ནི་མངོན་པར་ཁྲོ། །གསེ་བོ་དྲོ་བེ་ད་ནཱ་ན་ནཱ། ་། ས་མཱུཙྪ་ན་མ་བྷི་བྷཱུབྷི་རཱནྟཱ་བྷི་ཀྵརྦ་ནཱཉཌྷ་ཚོར་བ་མྱོང་བ་ཉམས་སུ་མྱོང་། །བརྒྱལ་བ་མངོན་པར་ཁྱབ་པ་དང་། །སློང་བ་བསོད་སྙོམས་འདོད་དོན་ནཱ༑། བཛྷ་ནཾ་ཙྪེ་ད་ནེཤཐ་དྭེ་ཨཱ་ནནྡ་ན་ས་བྷཱ་ཛ་ནེ། །ཨཱ་པྲིཙྪ་ཎ་མ་སློང་། །གཅོད་པ་དང་ནི་གསེ་བའོ། །དེ་ནས་གཉིས་པོ་དྲི་རྨད་དང་། །ཁམས་བདེ་མི་བདེ་འདྲི་ཐཱསྣ་ཡ༔སཾ་པྲ་དཱ་ཡ༔ཀྵ་ཡེ་ཀྱི་ཡཱ། །གྲ་ཧེ་གྲྭ་ཧོ་བ་ཤཀཱནྟོ་རབྷྱཱསྔ་བའོ༑། དེ་ནས་མན་ངག་སྡོམ་པའོ། །ཟད་པ་དང་ནི་རྫོགས་པའོ། །ལེན་པ་དང་ནི་བླངས་པའོ། །ཎེ་ར་ཎཀཱ་ཎེ་རྒྱུ་དང་མི་གགལ་སྨྲའོ། །།བྱ་དྷོ་བེ་དྷི་པ་ཙཱ་པཱ་ཀེ་ཧ་བོ་ཧུ་ཏོ་མདངས་ཅན་དང་ནི་འཚེར་བའོ། །བསྲུང་བ་དང་ནི་སྐྱབས་པའོ། །འབུགས་པ་འབིགས་པ་ཚོས་པ་བ་རོ་བྲྀ་ཏོ། །ལ་། ཨོ་ཥབློ་ཥ་ན་ཡོ་ནཱ་ཡེ་ཛྷ་ནིཙླིཙྩོ་བྷ་སྨིན། །འབད་པ་དང་ནི་སྒྲོན་པའོ། །མཆོག་དང་རབ་དང་སྲེག་དང་ཚིག། མདུན་མ་གྲོས་དང་།
