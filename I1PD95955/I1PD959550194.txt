ཏོག༈ཡང་ཨ་ཏི་ཤ་ཡེ་ན་དཱིགྷཆེ་ཆེར་རིང་བའམ་ཤིན་ཏུ་རིང་བ། ཞེས་པའི་དོན་ལ། དཱིགྷ་དཱགྷ། ཆེས་ཆེར་རིང་བའམ་ཤིན་ཏུ་རིང་བ། ཞེས་པར་བསྒྱུར་བས། དྷཱངྒཱི་ཡཱནི། ཤིན་ཏུ་རིང་བ། དཱགྷིཥྛཨ་ཏི་ཤ་ཡེ་ན་པྲ་ཤཡཤིན་ཏུ་གཙོ་ཆེ་བའམ་གཙོ་བོ་ཕུལ་དུ་བྱུང་བ། ཁ་ཞེས་པའི་དོན་ལ། པྲ་ཤ་ཤ་ཞེས་པར་བསྒྱུར་བས། ཐེ་ཡཱན། ཤིན་ཏུ་གཙོ་ཆེ་བའམ་གཙོ་བོ་་ཕུལ་དུ་བྱུང་བ། ཤེཥྛསོགས་སོ། །འདུ་། ཁ་ན་མ་༈བ་ཧོ་རིཥྛ་ཡིབ་ཧུའི་ཨིཥྛ་ལ་ཡིའོ། །རྐྱེན་ཨིཥྛ་ཕ་རོལ་ལ་བ་་ཧུའི་སྒྲ་ནི་བྷུ་ཞེས་པར་འགྱུར་ཞིང་ཨིཥྛའི་ཨི་ཡང་ཡི་ཞེས་པར་འགྱུར་ཏེ། ཨ་ཏི་ཤ་ཡེ་ན་བ་ཧུ…མང་བའི་ཕུལ་དུ་བྱུང་བའམ་ཤིན་ཏུ་མང་བ། ཞེས་པའི་དོན་ལ། བྷཱུ་ཡིཥྛམིང་བའི་ཕུལ་དུ་བྱུང་བའམ་ཤིན་ཏུ་མང་བ། །ག་ད་བ་ས་དྲག་ལ། གླུ་བ་གཙོ་ཀེ་ཁྱེད་ལ་བཏོ། ༈ཀིམ་དང་། མི་ཟད་པ་དང་ཀུན་བཤད་ཀྱིས་བསྒྲུབས་པའི་ཚིག་མཐའ་རྣམས་ལ་རྐྱེན་ཏར་དང་ཏ་མ་དག་སྦྱོར་བའི་མཐར་ཨཱམ་ཨཱ་ག་མ་སྟེར་བར་་བཤད་དེ། ཀིམ་ལ་ཏ་མ་སྦྱར་དེའི་མཐར་ཨཱམ་ཨཱ་ག་མ་བྱིན་པས། ཀིཾ་ཏ་མཾ། ཅི་ཡིན། དེ་བཞིན་དུ། ཀུ་ཏས་ཏར་རཾ། ཨཱམྲ། པ་ར་མཱ་ཎ་བམཆོག་ཏུ་ཕྲ་བ་དེ་རྣམས་ཆུང་དུ་ཉིད་དོ། །ཀུ་ཏས་མཱི་ཏེ་ཊཱ་ཨཱ་རམྦྷ་ཀ་ཏྭཾ། ཏ་ཨཱ། ཨུཙྪི་སྟ་རཱ་གཱ་ཡནྟི། སྐད་གསང་མཐོན་པོར་སྨྲ་བར་བྱེད་དམ་སྐད་ཆེན་པོས་གླུ་ལེན། ཨུཙྪི་སྟ་ཨཾ་གཱ་ཡནྟི། བ་ཙ་ཏི་ཏ་རཱེ། ཤིན་ཏུ་འཚེད་པར་བྱེད། །པ་ཋ་ཏི་ཏ་ནཱཾ། ཆེས་ཀློག་པར་བྱེད། །མ་ཀྱི་བ། །ཆབ་ཟླ:ཁྲུག་::ཀློ::༈པ་རི་མཱ་ཎེ་དགླུ་ད་ཡཚད་ལ་ད་ལ་སོགས་པའོ། །གང་ལའང་ཆེ་ཆུང་མཐོ་དམའ་སོགས་ཀྱི་ཚད་བརྗོད་པའི་དོན་ལ། །ད་ངྷ་དང་:མཱ་ཏྲ་དང༌། དྭ་ཡས་རྣམས་འགྱུར་ཏེ། ཛཱ་སི་ནུ་པ་རི་མཱ་ཎཾ་ཛ་ལཾ། སི། ཆུ་བོ་པུས་མོ་ནུབ་ཙམ། ཞེས་པའི་དོན་ལ། :པུས་ནུབ་ཙམ། ཛཱ་ནུ་དདྡྷཾ། སི། དེ་བཞིན་དུ། པུ་རུ་ཥ་མཱ་སི་་ཨམྤ། ཀྲྀ་བྲྀཀྵསྐྱེས་བུ་དང་མཉམ་པའི་ཤིང༌། ཤི་རོ་དྭ་ཡ་སཾ་སི་དཎྞ། མགོ་བོའི་ཚད་དང་མཉམ་པའི་—།
