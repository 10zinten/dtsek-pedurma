འཆི་མེད་མཛོད་ཀྱི་རྒྱ་ཆེར་འགྲེལ་པ་འདོད་འཇོའི་བ་མོ་ཞེས་བྱ་བ་བཞུགས་སོ། །སོགས་པའོ། །།།།།།ཆུ་ས་ཏཁའེ་ཌར་ཁྱའི་བློ་ཚོ་ཀཀ། ཆ་ཆུ་ཁ་/pདེ་ནས་ཏ་ན་དང༌། ཀུ་ཙ། རུ་མའི་མིང་ངོ༌། །་བ་ཡི། །ཊ་ཀའི་ཡཚེཕྲག་པ་ལ་མིང་གསུམ་སྟེ། སྐནྡྷ། ཕྲག་པ་སྟེ། ཀཾ་མ་ཏ་དཾ་ཀཾད་དྷ་རི་ཀན༑ཞེས་པས་ཀཾ། མགོ་བོ་ཡིན་ལ་འཛིན་པར་བྱེད་པས་ཕྲགས་སྟེ། པར་སྐ་ལ་སོགས་པས་སུཊ་ཨཱ་གམ་སྦྱར་བའོ། །བྷུ་ཛ་ཤ་རིདཔུང་པ་མགོ་བོ། ཨོཾ་ས་ཕྲག་པའོ། །ཀེའི། །ན་བ་མ་འ། །ཁོ་ཚོ་ཆེ་བ་ཤ་ཆེ་བ་བཏང་བ་ཚོ་ཡང་། པ་དང་། དེ་ནས་ལག་པའི་མིང་ལ། བྷུ་ཛ་ལག་པ་སྟེ། སྤྱོད་པར་བྱེད་པའོ། །བ་ཧུ་འབབ་པར་བྱེད་པས་ལག་པའོ། །ཕྲ་བེཊྚ། །་འཇུག་པར་བྱེད་པས་ལག་་་པའོ༑། དོཞི་བར་བྱེད་པས་ལག་པ་སྟེ། ད་མུ་ཉེ་བར་ཞི་བར་བྱེད་པ་ལ། ཊོས་རྐྱེན་ནོ། །པཉྩ་ཤཱ་ཁ། ཡལ་ག་ལྔ་པ་སྟེ། ཤིང་ཡལ་ཁ་ལྔ་པ་འདྲ་བས་སོ། །ཤ་ཡ། ལག་པ་སྟེ། ཉལ་བྱེད་དོ། །བཱ་ཎཱི། ལག་པ་སྟེ། ཐ་སྙད་སྟོན་་་པར་བྱེད་པས་སོ། །དེ་ནས། །ཀུརྦ་ར། ་ཞེས་པ་གྲུ་མོ་ཡིན་ལ། དེའི་གོང་་དུ་གནས་པ་བྲ་གཎྜ་བར་ལྟ་བུའོ། །།གྲུ་མོ་ལས་འོག་ཏུ། པྲཀྵ། ལག་ངོར། དེ་དང་ལག་མགོའི་བར། མ་ཎི་བནྡྷ།  ོ་ནོར་བུ་འཆིང་བ་ཞེས་བྱ་སྟེ། འཁྲིག་མའོ༑། །དེ་ནས་འཐེའུ་ཆུང་གི་བར། ཀ་ར་བྷ་ཞེས་བྱའོ། ལྟ་། ཎ་ཞི་བར་་བན་:འཛུབ་མོའི་མིང་ལ། ཏརྫ་ནི། སྡིག་འཛུབ་དང༌། པྲ་ཏ་ཤཱི་ནི། རབ་ཏུ་བསྟོན་བྱེད་ཅེས་བྱའོ། །།།སོར་མོ་ལ། །སྒྲུང་ལཱི་དང་། སྒྲུང་རཱི། སོར་མོ་དང༌། །་ཀ་ར་ཤ་ཁ། །ལག་པའི་ཡལ་ག་ཞེས་བྱའོ། །ན། །བྱ་བྱིའོ། །པ་::འཐེ་བོང་ལ་སོགས་པ་སོ་སོའི་མིང་ལ། སྒྲུང་ཥཱ་འཐེ་བོང་སྟེ། སྒྲུང་ཏིཥྛ་ཏཾ་ཞེས་པས། །ལུས་ལ་གནས་པས་འཐེ་བོང་ངོ༌། །འཛུབ་མོ་ལ། པྲ་ཏི་ཤྲཱི་ནཱི། སླར་བཤད་པ་སྟེ། འཕྲལ་གྱི་ཐ་སྙད་རབ་ཏུ་སྟོན་ཞིང་གོ་བར་བྱེད་པས་སོ། །།
