བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་་[། པ་དེའི་ཚེ་འཕགས་མ་ཀུན་ནས་གཡོ་བའོ། །།ཀུ་རུ་མ་རྒྱ་བརྟི་་ནོ་ཙེཇི་ཏཱི''ཡ་ཏུརྨོ་ག་ཎོ་ཛ་ཀཱ་རཱ་ཁྱོ། ཨཌྷ་་དྭ་ཡེ་བྷ་བེ་ཏཱཾ་བྷ་བཏྟ་དཱ་སརྦ་ཏཅུ་པ་སྟེ། ལཱ༑I[། Lལགལ་ཏེ་སྔ་ཕྱེད་འབའ་ཞིག་ཉིད། །གཡོ་བ་ཡི་ནི་མཚན་་་'ཉིད་གྱུར། །སླར་ཡང་ཕྱི་ཕྱེད་ལ་གྱུར་མིན། །།དེ་ཚེ་དེ་སྒོ་གཡོ་བར་གྱུར། །གཡོ་བའི་མཚན་ཉིད་ནི། གཉིས་པ་དང་བཞི་པ་དག་ལྕི་བ་དང་དབུས་ན་གནས་པའི་ཛ་དག་སྟེ། ་དེ་གལ་ཏེ་ཕྱེད་སྔ་མ་ཉིད་དུ་གྱུར་ཅིང༌། ཕྱེད་ཕྱི་མ་ལ་མ་ཡིན་པ་དེའི་ཚེ་དེ་འཕགས་མ་སྒོ་གཡོ་བའོ། །པཱུརྦདྷ་ཨེ་བ་སོཙེཥེ་སེབ་ལེ་བྷ་བེལླ་ཎཾ་ཏུ་ཙ་པ་ལཱ་ཡཱ། ན་སཀླུ་ན་ར་པ་རཱནྟེ་མུ་ཁ་ཙ་པ་ལཱ་སོ་་ཏ་་་དཱ་བྷ་་བ་ཏི། LTIགང་ལ་བསྟན་པའི་་་，གཡོ་མཚན་ཉིད། །སྔ་མ་ཕྱེད་ལ་སྤངས་ནས་ཀྱང༌། །ཕྱི་ཕྱེད་གཅིག་པུར་བརྟེན་ནས་ཞུགས། །དེ་ནི་ལུས་རྟེན་། གཡོ་བའོ། །འཕགས་མ་གང་ལ་ཕྱེད་སྔ་མ་དོར་ནས་ཕྱེད་ཕྱི་མ་འབའ་ཞིག་ལ་བརྟེན་ནས་གཡོ་བའི་མཚན་ཉིད་སྔར་་་་བརྗོད་པ་ཡོད་པ་དེ་ལུས་རྟེན་གཡོ་བའོ། །ནིདྷིཤྩ་ཙ་པ་་ལཱ་ཡཱ་ལཀྵ་ཎ་མ་ེཔི་ཧཱ་ཡ་ཡ་ེཏྲ་པཱུཧཱདྷཾ། བཌྷཌྷ་མེ་ཀ་མཱ་ཤི་རྐྱ་བརྟ་ཏེ་ཛ་གྷ་ན་ཙ་པ་ལཱ་སཱ། I*་གང་དུ་འདབ་མ་གཉིས་ཀ་ཡང་། ༽། དྲུག་ཚོགས་ཛའམ་ཁ་ཡིག་གྱུར། །ཕྱེད་སྔ་བཞིན་དང་ལྷག་མ་ཡང་། །སོ་སོ་TAI། །
