ཨཌྷ་དྭ་ཡེ་བྷ་བེ་ཏཾ་བྷ་བེཏྟ་དཱ་སརྦ་ཏ་པ་ལཱ། །པཱུརྞནྟ་ཨེ་བ་ཙེ་ཏེ་བ་ལེ་བྷ་བེལླཀྵ་ཕྱེད་པ་གཉིས་ལ་གྱུར་དེ་ཚེ། །ཀུན་ནས་གཡོ་བར་གྱུར་པ་ཡིན། །གལ་ཏེ་སྔ་ཕྱེད་འབའ་ཞིག་ཉིད། །ཎཾ་ཏུ་ཙ་བ་ལཱ་ཡཱ། །ན་སཱུཏྤ་ན་ར་བ་རཱདྷེ་མུ་ཁ་ཙ་པ་ལཱ་སོ་ཏི་དཱ་བྷ་བ་ཏི། །ནིརྟི་གཡོ་བ་ཡི་ནི་མཚན་ཉིད་གྱུར། །སླར་ཡང་ཕྱི་ཕྱེད་ལ་གྱུར་མིན། །དེ་ཚེ་དེ་སྒོ་གཡོ་བར་འགྱུར། །གང་ལ་ཐཾ་ཙ་བ་ལཱ་ཡཱ་ལཀྵ་ཎ་མ་པི་ཧཱ་ཡ་ཡ་ཏྲ་པུརྭ་རྣོ། །པཤྩ་ཎྷ་མེ་ཀ་མཱ་ཤི་ཏུ་བརྟ་བསྟན་པའི་གཡོ་མཚན་ཉིད། །སྔ་མ་ཕྱེད་ལ་སྤངས་ནས་ཀྱང༌། །ཕྱི་ཕྱེད་གཅིག་པུར་བརྟེན་ནས་ཞུགས། །ཏེ་ཛ་གྷ་ན་ཙ་པ་ལཱ་སཱ། །སཱུ་དུ་བྷ་ཡོ་ར་པི་ད་ལ་ཡོཥཥཱ་ག་ཎ་ཡ་ཏྲ་ཛཁ་ཀཱ་རོ་བཱ། དེ་ནི་ལུས་རྟེན་གཡོ་བའོ། །གང་དུ་འདབ་མ་གཉིས་ཀ་ཡང་། །དྲུག་ཚོགས་ཛའམ་ཁ་ཉིད་ཡིན། །ཕྱེད་པཱུ་རཱཛྷ་བཙྩ་ཤེ་ཥཔྲ་ཏི་ནི་ཡ་མོ་བྷ་བ་ཏི་བྷ་བ་ཏི་སཱ་གཱི་ཏི། །ཡ་སཱུ་ཀ་ཨེ་བ་ད་ལ་ཡོ་སྔ་བཞིན་དང་ལྷག་མ་ཡང་། །སོ་སོ་ངེས་གྱུར་དེ་དབྱངས་འགྱུར། །གང་གི་འདབ་མ་གཉི་གའི་ཡང་། །རུ་བྷ་ཡོ་ར་པི་བྷ་བ་ཏི་ཥཥཱ་ག་ཎ། །ཨ་པ་རཱནྡྷ་བཙྩ་ཤེ་ཊོ་ནི་ཡ་མ་མསྟ་ཡཱ་ཧུ་རུ་དྲུག་པའི་ཚོགས་ནི་ཀ་ཉིད་འགྱུར། །ཕྱེད་ཕྱི་བཞིན་དང་ལྷག་ངེས་པ། །དེ་ནི་ཉེ་བའི་དབྱངས་སུ་སྨྲ། །པ་གཱི་ཏི། །པཱུཏྲཱནྟེ་ཥཥཱསཱཀྐོ་ཡ་ཏྲ་ཛ་ཁོ་རུ་པཉྩནྟེ། །ཤེ་ཥཔྲ་ཏི་ནི་ཡ་མོ་ང་ལ་བི་གང་དུ་སྔ་ཕྱེད་དྲུག་གྱུར་ཀ། ཕྱི་ཕྱེད་ལ་ནི་ཛའམ་ཁ་དག། ལྷག་ངེས་འདབ་མ་ལོག་པ་ལས། །གྱུརཔ་རྱ་ཡདྡྷ་བ་ཏི་བྷ་བ་ཏི་ས་དྷཱི་ཏི། །།ཡ་སྭཱ་ད་ལེ་ད་ལེཉཏྠོ་ཙཱ་ཋཱཥཊྛོ་ག་ཎསུ་པ་དེ་ནི་མཆོག་དབྱངས་ཡིན། །གང་འདབ་འདབ་མར་བརྒྱུད་རྣམས་ཙ་ཞེས་བྱ། །དྲུག་པའི་ཚོགས་ནི་ཛོ་བཱ་ཁོ་བཱ། །པཱུཊཱདྷ་བཙྩ་ཤེ་ཥབྲ་ཏི་ནི་ཡ་མོ་བྷ་བ་ཏི་བྷ་བ་ཏི་སཱནཱ་ཛའམ་ཡང་ན་ཁ། །ཕྱེད་སྔ་བཞིན་དང་ལྷག་མ་སོ་སོར་ངེས། །གྱུར་པ་དེ་ནི་འཕགས་མའི་དབྱངས་སུ་གཱི་ཏི། །ཨཱརྱ་ཛཱ་ཏ་ཡ། །ཏད། །ཁ་འགྱུར། །འཕགས་མའི་ཛ་ཏི་རྣམས་སོ། །།།ད་ལ། བབ་གིའོ་ཊ་ཌ་པ། ::::ང་བ་དང་། །ཚེ་ད་ཏྲ་ཡ་མ་ཡུ་ཛོཏྨ་ཁེ་ཡུ་ཛོརྡ་ཙ་ཏུཏྟཾ་ར་ལ་ག་སྟེ་པྲྀཏྟཱ་ཏ། །བེ་ཏཱ་ལཱི་ཡཾ་ཟུང་མིན་དང་པོར་ད་གསུམ་ཟུང་ད་བཞི། །ཕྱི་མ་ལ་ནི་ར་ལ་ག་རྣམས་ཏེ། །ཟུང་ལ་ཡང་དྲུག་པར་།
