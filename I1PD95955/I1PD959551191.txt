བསྟན་འགྱུར། སྒྲ་མདོ། གི་ལ། མཱ་ཀྲོ་ད། བི་ཥ་མེ་,ཛ་ར་ཧི་ཏཱ་ཉྩནྡ་ལེ་ད་ལོཉེབྷཾ་ག་ཎཾ་བྷ་བིཉྩ་རླུ། གཉིས་ད། །གལ་ཏེ་འདབ་མ་མདབ་མར་ཚོགས་བརྒྱད་རྣམས། །མི་མཉམ་རྣམས་ལ་ཛ་སྤངས་འཕགས་སྔམར་འགྱུར། །ཕྱི་མ་ཕྱེད་ལ་དྲུག་པ་ཀའོ་གཉིས་ཀ་ཡི། །བརྒྱད་པ་ད་སྟེ་ལྷག་མ་ཙ་རྣམས་སོ། །ཕྱེད་སྔར་པུཏྲཱནྟེ་ཥཌོ་ཛི༔ཁོ་བཱ་ཁེ་ཏཱ་བྱི་ལ་གླུ་ནི་བྷ་བ་ཏི་ཡ་ཏི། ཥ་ཁ་པ་དྲུག་པ་ཛའམ་ཁའོ་ཁ་ལ་ནི། །དང་པོ་ལ་ནི་ཡང་བའི་ངལ་བར་འགྱུར། །ཁ་ཕ་རོལ་ལ་དྲུག་པའི་མཐར་། རོཉནྟ་ཡ་ཏི་སྙཚོ་བྱ་ཐཾ་བྷ་བ་ཏི་ཙ་རཾ་མ་ད་ལོ། །ཙིནྟི་བྷི་ར་ཡུདྨ་པཱ་དོ་ཡོ་། ངལ་བསོ། །འདབ་མ་ཕྱི་མར་བཞི་པའང་དེ་ནས་འགྱུར། །ཙ་གསུམ་རྣམས་ཀྱིས་ཟུང་མིན་རྐང་པ་དག། དཱ་ས་མཱ་བྷོ་ཏ་དཱ་བྷ་བེཏྠ་། ཨེ་ཀོ་པི་ནིཊྛི་ཏོ་ན་ཡ་དི་ཏི་ར་ཡུགྒསྟ་དཱ་གང་ཚེ་རྫོགས་པ་དེ་ཚེ་འཕྲོད་མར་འགྱུར། །གལ་ཏེ་ཟུང་མིན་ཅིག་ཀྱང་རྫོགས་མིན་པ། །དེ་རྣམས་ཀྱིས་ནི་་བི་པུ་ལཱ། །གུ་རུ་མ་དྷ་བནྟི་ནོ་ཙིདྡྷི་ཏི་ཡཾ་ཏུརྨོ་ག་ཎོ་ཛ་ཀཱ་རཱ་ཁྱོ། །དེ་ཚེཡངས་མ་ཡིན། །གལ་ཏེ་གཉིས་པ་བཞི་པའི་ཚོགས། །ལྕི་གཉིས་དབུས་ཞུགས་ཛ་དག་བཤད། །ཨནྡྷ་ཧཱ་ཡ་བྷ་བི་དྷྭཾ་བྷ་བིཏྟ་དཱ་སརྦ་ཏ་པ་ལཱ། །པུཧཱདྷཾ་ཨེ་ཐེ་ཙིཏྟཾ་བ་ལེ་བྷ་བ་ཕྱེད་ལ་གཉིས་ལ་གྱུར་དེ་ཚེ། །ཀུན་ནས་གཡོ་བར་གྱུར་པ་ཡིན། །གལ་ཏེ་སྔ་ཕྱེད་འབའ་ཞིག་ཉིད། །ལླཀྵ་ཎཾ་ཏུ་ཙ་པ་ལཱ་ཡཱ། ན་སཱུཏ་བུ་ན་ར་བ་རཱནྟེ་མུ་ཁ་ཙ་པ་ལཱ་སོ་ཏ་དཱ་བྷ་བ་ཏི། གཡོ་བ་ཡི་ནི་མཚན་ཉིད་གྱུར། །སླར་ཡང་ཕྱི་ཕྱེད་ལ་གྱུར་མིན། །དེ་ཚེ་དེ་སྒོ་གཡོ་བར་འགྱུར། །གང་ལ་ནིཇི་ཏྠཾ་ཙ་བོ་ལཱ་ཡཱ་ལཀྵ་ཎཾ་མ་པ་ཧཱུཾ་ཡ་ཡ་ཏྲ་པཱུཧཱ་དྷམ། པ༔་རྒྱུ་མེ་ཀ་ལཱ་ཐི་སྐྱ་བརྟན་པའི་གཡོ་མཚན་ཉིད། །སྔ་མ་ཕྱེད་ལ་སྤངས་ནས་ཀྱང༌། །ཕྱི་ཕྱེད་གཅིག་པུར་བརྟེན་ནས་བོནྟཱ་ཏེ་ཛ་གྷ་ན་ཙ་པ་ལཱ་སཱ། །སྟ་དུ་བྷ་ཡོ་ར་པི་དེ་ལ་ཡོ་ཥཥཱ་ག་ཚོ་ཡཾ་ཏཱ་ཛཱི༔ཞུགས། །དེ་ནི་ལུས་རྟེན་གཡོ་བའོ། །གང་དུ་འདབ་མ་གཉིས་ཀ་ཡང་། །དྲུག་ཚོགས་ཛའམ་ཁ་ཉིད་ཡིན། །ཁ་ཀཱ་རོ་བཱ། པཱུཧཱནྟེ་བཙྪ་ཤེ་ཥཔྲ་ཏི་ནི་ཡ་མོ་བྷ་བ་ཏི་བྷ་བ་ཏི་སཱ་གཱི་ཏི། །ཕྱེད་སྔ་བཞིན་དང་ལྷག་མ་ཡང་། །སོ་སོར་ངེས་འགྱུར་དེ་དབྱངས་འགྱུར། །ཏི། །
