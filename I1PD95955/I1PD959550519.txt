བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ལནྟ་ར་ལོ་ཙི་བ་པཱ་རི་བླ་བ་པ་རི་བླ་བེ། །ཨ་ཏི་རིཀྐས་མཱ་དྷི་ཀོ་དྲྀ་ཌྷ་ཤག་ཤིག་སག་སིག་དང་། །ཡོངས་སུ་འགུལ་བ་ཞེས་པའོ། །སྟོང་པ་དང་ནི་ཏིང་ངེ་འཛིན། །བརྟན་སནྟིསུ་སཾ་ཧ་ཏ། །ཀཀྑ་ཊཾ་ཀ་ཟི་ནཾ་ཀཱུ་རཾ་ཀ་ཁོ་རོ་ནིཥྛ་རཾ་དྲྀ་ཌྷ། །ཛ་ཁ་རཾ་པོ་དང་ནི་མི་འགྱུར་བ། །སྲ་མཁྲེགས་མི་ཤིགས་ཀ་ཊོ་རཾ། །མི་འཇིགས་མི་ཕེད་སྲན་དང་ནི། །གོང་བུ་མཱནྟི་མནྣཱནྟཾ་པྲ་ཧྲིདྡྷཾ་པྲ་ཌྷ་མེ་དྷི་ཏཾ། །པུ་རཱ་ཎ་པྲ་ཏ་ན་པྲཏྣ་པུ་རཱ་ཏ་ན་ཙི་དང་ནི་གོར་མའོ། །འཕེལ་བ་རྒན་པོ་ཆེན་པོ་དང་། །རྙིང་པ་སྔོན་སྐྱེས་དང་པོ་པ། །རྙིང་མ་དང་ནི་ཡུན་རནྟ་ནཾ། །པྲ་ཏྱ་གྲོའབྷི་ན་བོ་ན་བྱོ་ན་བཱི་ནོ་ནཱ་ཏེ་ནོ་ན་བ། །།ནུཐུ་སུ་ཀུ་རིང་མ། །གསར་པ་སོ་མ་ཐོག་མ་དང༌། །ཡག་མ་གཞོན་ནུ་གསར་པའོ། །རབ་གཞོན་ཤིན་ཏུ་གཞོན་མཱ་རནུ་ཀོ་མ་ལཾ་མྲ་དུ་ལཾ་་དུ། ་། ཨན་གནྡྷ་ག་མ་ནུ་གོཉནུ་པ་པ་དང༌། །འཇམ་པ་དང་ནི་མཉེན་དང་འབོལ། །།རྗེས་འགྲོ་ཕྱི་འབྲང་རྗེས་འཇུག་དང་། །ཏིང་དཾ་ཀཱི་བ་མ་བྱ་ཡཾ། །པྲ་ཏྱཱ་ཀྵ་སྭཱ་དིནྡྲི་ཡ་ཀ་མ་:ཏྲ་དྷ་ཀ་མནྟིནྡྲི་ཡཾ། །འབྲང་མ་ནིང་མི་ཟད་པ། །མངོན་སུམ་དང་ནི་དབང་པོའི་ཡུལ། །ལྐོག་ཏུ་གྱུར་དང་མངོན་སུམ་མིན། །ཨེ་ཀ་ཏཱ་ནོཉན་ནྱ་བྲྀཏྟི་རེ་ཀཱ་གྲི་ཀཱ་ཡ་ནཱ་བ་པི། །ཨ་པེ་ཀ་སརྒ་གཅིག་སྔ་གཅིག་སྔོན་གཅིག་རྩེ་དང་། །གཅིག་སོང་གཅིག་མཐོ་གཅིག་དང་པོ། །གཅིག་རྩེ་སྔོན་དུ་གཅིག་ཨེ་ཀཱ་ཧཱུྃ་པེ་ཀཱ་ཡ་ན་ག་ཏོ་པི་ཙ། །པུཾ་སྣ་དིཔཱུརྦ་པོ་ར་སྐྱ་པྲ་ཐ་མཱ་དྱ་ཨ་ཐ་སོང་བ། །སྐྱེས་བུ་ལ་སོགས་དང་པོ་དང༌། །ཐོག་མ་དང་པོ་ལ་སོགས་སོ། །དེ་ནས་མོ་དང་མཐའ་མ་སྦི་ཡཱ། །ཨནྟོ་ཛ་གྷ་ནྱ་ཙ་ར་མ་མཾ་ོཏྲ་པཱཉྩ་ཏྱ་པཉྩི་མཾ། །མེ་གྷོ་ནི་རརྦ་ཀཾསྤ་ཊཾ་དང༌། །།ཕྱི་མ་རྟིང་མ་རྗེས་མ་དང་། །གཤམ་མ་དང་ནི་ཕྱི་ཤོས་སོ། །དོན་མེད་པ་དང་དོན་སྟོང་དང་། །སྤུ་ཊ་པྲ་བྱལྟ་མུ་ལཱ་ཎཾ། །སཱ་དྷཱ་ར་ཎནྟུ་སཱ་མཱ་ཏྱཱ་མེ་ཀཱ་ཀཱི་ཏྲེ་ཀ་ཨེ་ཀ་ཀ། །བྷིནཟླ་གསལ་དང་རབ་གསལ་དྲི་མ་མེད། །གཞི་འཛིན་དང་ནི་སྤྱི་ཡིན་ནོ། །རེ་རེ་རྐྱང་རྐྱང་གཅིག་པུའོ། །ཐ་ཀཱ་ཨ་ནྱ་ཏ་ར་ཨེ་ཀ་་རྩེ་ཏ་རཱ་བ་པི། ཚོ། ཨུཙྪཱ་བ་ཙོ་ནི་ཀ་བྷེ་ད་དད་དོན་དང་གཞན་དང་ནི། །གཅིག་གནས་གཞན་པ་སོ་སོ་བ། །ཚིག་མང་གཅིག་དབྱི་མིན་པ་དང༌། །།
