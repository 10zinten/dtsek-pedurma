བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽་ཎགpི་ཁསྨོན་པ་ཁོ་ན་ཨུ་བྷོ་ཏུ་དྲི་ཡ་བི་ཡཱ་བློ་དྭེ་བ་ཙམྨ་ཤྲཱི་ཏེ་ར་ཐེ། །པཱཎྜ་ཀཾ་བ་ལ་སམྦྷི་ཏསྱནྡནཔཱཌཱ་ཀཾ་གཉིས་པོ་སྟན་གཡོགས་སྟག་ཅན་དང་། །སྟག་གིས་གཡོགས་པའི་ཤིང་རྟའོ། །།ལ་བ་དཀར་པོས་གཡོགས་བ་ལཱི། ཊི་། ར་ཐེ་ཀཱམྦྷ་ལ་བཱབྷྱཱ་༴ཀཧྥ་ལཱ་དི་བྷི་རཱ་དྲྀ་ཏེ། །པ་དང༌། །སྤྱནྡན་དང་ལ་བོ་དཀར། །ཤིང་རྟ་ལ་ཅན་གོས་ལ་སོགས། །ལ་བ་ལ་སོགས་གཡོགས་པའོ། །ཏྲི་ཥཱ་དཱི་པཱ་ད་ཡོ་ར་བྷ་ར་ཐ་ཀ་དྲྭ་ར་ཐ་བྲ་ཛེ། །དྷུསི་ཀི་བེ་ཡཱ་ན་གསུམ་པོ་གཉིས་འཐུང་ལ་སོགས་ལ། །ར་བྷ་ཤིང་རྟའི་ཚོགས་དང་ལམ། །དྷཱམོ་མ་ནིང་ཤིང་རྟའི་མུ་ཁཾ་ས་དྲ་ཐཱངྒ་མ་པ་སྐ་རཿ། །ཙ་ཀྲྀ་ར་ཐཱངྒནྟ་སྱཱནྟོ་ནེ་མིནྡྷི་སཏྠ་དྷིཔུ་གདོང་། །ཤིང་རྟའི་ཁོང་པ་མ་ཐ་སྐར། །འཁོར་ལོའི་ཤིང་རྟའི་ཡན་ལག་དང༌། །དེ་ཡི་མཐའ་དང་མུ་མཱན། །:པཎྜི་ཀཱ་ནཱ་བྷི་རཀྵ་གྲ་ཀཱི་ལ་ཀེ་ཏུ་དྭ་ཡོ་ཁྱུད་དང༌། །མོ་མིན་བྲ་དྷི་ཕོ་ཡིན་ནོ། །འཁོར་ལོའི་ལྟེ་བ་འཁོར་ལོ་ནི། །བསྲུང་བའི་གཟེར་གཉིས་ར་ཎི། །ར་ཐ་གུསྟི་བ་རཱུ་ཐོ་ནཱ་ཀཱུ་ཊ་རསྟུ་ཡུ་གནྡྷ་ར། །ཨ་ནུ་རཀོ་དཱཉྫ་ཨ་ཎིའོ། །ཤིང་རྟ་སྒྲིབ་དང་འཐབ་རའོ། ་། ཡུ་གན་དྷ་ར་གཉའ་ཤིང་འཛིན། །རྗེས་སུ་འདྲེན་དང་དཱརྦ་དྷཎྜཾ་པྲ་སཾ་གེཤ་ནོ་ཡུ་མཱ་ཛུ་ག། ་། སབྦཾ་སྱཱདྷཱ་ཧ་ནཾ་ཡཱ་ནཾ་ཡུ་གྱི་པཏྟཾ་འདྷ༑། གཉའ་ཤིང་འཆིང་ཞིང་ཡུ་ག་དང་། །ཐམས་ཅད་བཞོན་པ་ཐེག་པ་དང༌། །།ཁྱོགས་དང་འཁྲུལ་ཙ་དྷོ་ར་ཎཾ། །།པ་རལྦ་རཱ་བཱ་ཧ་ནཾ་ཡཏྟདྲི་ནཱི་ཏ་ཀ་མ་ལཾ། །འཁོར་བགྲོད་བྱེད་དེ། །།ཕན་ཚུན་འབབ་དང་ལེན་པ་དང༌། །མོ་མིན་བཞོན་པ་གླང་པོ་པ། །ཨཱ་དྷོ་ར་ཎཱ་ཧསྟི་པ་ཀ་ཧམྦྷ་རོ་ཧ་ནི་ཥཱ་དི་ན། །ནི་ཡནྡཱ་པཱ་ཛི་ཏཱ་ཡནྟཱ་གླང་པོའི་རྒྱབ་ཏུ་གནས་པ་དང༌། །གླང་པོ་ཞོན་དང་གནས་པའོ། །ཁ་ལོ་བ་དང་མགོ་འདྲེན་དང༌། །སཱུ་ཏགྷཏྟཱ་བ་ས་ར་ཐི། །ས་བྱེཥྛ་དཀྵི་ཎ་ཌོ་ཙ་སཛྷ་ར་ཐ་ཀུ་ཊུམྦི་ན། །སྣ་ཐག་སྣ་ཁྲིད་ཁ་ལོ་སྒྱུར། །གཡས་སྡོད་ལྷོ་ཕྱོགས་གནས་པ་དག་། ཤིང་རྟའི་གཉེན་ཞེས་བྱ་བའོ། །ར་ཐི་ནསྱནྡ་ནཱ་རོ་ཧཱ་ཨ་ཤཱ་རོ་ཧཱསུ་སཱ་དི་ན། །ང། འི། ཀ་དུ་ར་ཐི་ན་ནི་ཤིང་རྟ་པ། །ཨ་ཤྭ་བ་ཧ་རྟ་པའོ། །::། ་ག་p། ལན། དང་ཕྱོགས། སྟག་ལྷ། །
