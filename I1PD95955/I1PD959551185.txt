བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་་ཤ་ར་སྭཱ་ར་ཏོ་ར་ཎི་བི་རཱ་ཝཱ། པའི། ཀྲོ༄༅པ་དཱ་བྷ་གཱབྷ་ག་བཱ་ནོལྟ་ན་ན་བོ་སྒྲོལ་བྱེད་ངལ་བསོ་ལུས་ཕྲར་འགྱུར། །བྷ་ག་ཝ་བྷ་ག་ཝ་ན་དག་ལ་ཀ་ན་ན་བ་རྣམས་མདའ་མདའ་ནོར་དང་ཨི་དུ་ཤ་རཾ་བ་སུ་མུ་ནི་བི་ར་ཏི། པའོ། མོ་གྷོ་ནོ་ལྷོ་སུ་རྫོགས་བཱིསྙཛྲ་སུ་ཐུབ་པར་ངལ་བསོ་ཁྲུང་ཁྲུང་རྐང་། །མ་དག་ག་ཝ་ན་དག་ལ་ཀ་སུ་དང་ཛ་དག་ཝ་རྣམས་ནོར་ལྷ་མྱོས་མ་ད་ན་ད་ཧ་ན་མུ་ནི་བྷིབྷུ་ཛཾ་ག་བི་ཛཾཏཾ། པའོ། སྨོནུ་རིནཾ་ནུ་ས་ག་བི་བྱེད་མེ་དང་ཐུབ་པ་རྣམས་ཀྱིས་ལག་འགྲོ་རྣམ་བསྒྱིངས་སོ། །མ་ན་ནུ་དང་ན་ནུ་རྣམ་གཉིས་པ་ག་ཝ་ཞེས་ཏི་ན་བ་ར་ས་ར་སཾ་ཀ་ར་ཎ་བི་ར་ཏི་ར་བ་བྷ་ཧིསཱུཏི། པའོ། པྲ་ཐ་མ་བྱ་རྣམས་དགུ་དང་རོ་རོ་བྱེད་པར་ངལ་བསོ་བཟློག་ཕྱོགས་འབབ་ཏུ་འགྱུར། །འདི་ཉིད་དུ་ནི་རྒྱུན་ཆགས་ཀཾ་ཨི་ཧ་ད༔རྐ༔༔་བྲིབྦཾ་པྲ་ཡཱ་ཏོ་བྷ་བནྡྷ་ནཱ་ཡེ་སྭཱ་ཧྥ་རིསཾབྷ་བྷི། པྲ་ཏི་དང་པོ་གཏུམ་པའི་ཆར་འབེབས་དག་ཏུ་འགྱུར་ཏེ་ན་གཉིས་དང་ནི་དེ་ནས་ར་བདུན་རྣམས་ཀྱིས་སོ། །འདིར་པ：ད་མི་ཧ་རོཕ་ཤྲཱིསྤྱི་་ཙཱ་བ་བྱ་ལ་རྗེ་མཱ་ཏ་མཱི་ཝཱ་ཀཾ་རོནྡ་མ་ནི་རྐང་པ་སོ་སོར་རེ་ཕ་འཕེལ་བར་གྱུར་རྣམས་ཆུ་རྒྱུན་རྒྱ་མཚོ་སྦྲུལ་དང་ཆར་སྤྲིན་རོལ་པའི་འབྱུང་ཤངྐཱ་ད་ཡོཔྲ་ཙི་ཏཾ་ཨི་ཧཱ་ན་ཡཾ་ཧཱÉསྟ་ཐཱ་ཙརྩྭ་ཤྲཱི་པྲ་ཡཱ་ཏཱ། གནས་ཤིན་ཏུ་ཀླུབས་དང་དུ་བསོགས་སོ། །འདིར་ནི་ན་ཟུང་དག་ལས་ཡ་རྣམས་ཀྱིས་ནི་རབ་བསགས་གཏུམ་ད་ཡོ་ཡིརྨ་ཐཱ་དརྩྭ་ཀོ་རི། ས་མ་བྲྀཏྟཱ་ནི། །།།པའི་ཆར་འབེབས་སོགས་བཞིན་ར་ཡིས་རྒྱུན་ཆགས་བཞིན་དུ་ཡ་ཡིས་སོ། །མཉམ་པའི་བྲྀཏྟ་རྣམས་སོ། །།།རy*oo༈ཨུ་བ་ཙི་ཏྲ་ཀ་མཱ་ཧུ་ར་ཡཾཀྵ་ཡཾཀྴི་ས་ལ་གཱ་ཡ་དི་བྷ་ཏྲི་ཏ་ཡཱངྒཱ། གལ་ཏེ་ཟུང་མིན་པ་དག་པ་དང་ལ་ག་དག་དང༌། །ཟུང་ལ་བྷ་གསུམ་ཅན་ལས་ག་དག་ཉེར་བཀྲར་སྨྲ། །བྷ་ཏྲི་ཏཾ་ཡཾབླ་དི་གོ་དྲུ་ཏཾ་མ་དྷ་ཡཾ་དི་པུ་ན་ར་བ་བྷ་བནྟི་ན་ཇོ་བློ། ཡ་དི་ས་གལ་ཏེ་བྷ་གསུམ་ཅན་ལས་ག་དག་གྱུར་པ་ལ། །གལ་ཏེ་སླར་ཉིད་ན་ཛ་ཛ་ཡ་མྱུར་བའི་དབུས། །གལ་ཏེ་ས་།
