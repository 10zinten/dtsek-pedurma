ཀ་ཀ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་ཀམ་ཎི་ཀ་ཝིཚོགས་པ་དང་ལྡན་པའོ། །ར་མ་ཎི་དང༌། །རཱ་མཱ། དགའ་བྱེད་མའོ། །དེ་ནས་ཀོ་པཱ་ནཱ་ཁྲོ་བྱེད་མ་སྟེ། དེ་ལྟ་བུ་ལ་སོགས་པ་བུད་མེད་དང་འབྲེལ་པའི་མིང་ཇི་སྙེད་པ་རྟོགས་པར་བྱའོ། །་་་་་་་་་་་་པ་ལ། ལ་རཱ་ཛ་སཱ་ལཱ་ལ་སོགས་པ་བརྒྱད་ནི། ་རྡུལ་ལྡན་མ་སྟེ་བུད་མེད་ཟླ་མཚན་་་་ཅན་མའི་མིང་ངོ༌། །ཁེ་བཞི་པ་བ་འབད་བའི་སྤྱཚོ་ལམ་དུ་མ་ན་བུད་མེད་སྦྲུམ་མ་ལ་མིང་བཞི་སྟེ། ཨཱ་པནྣ་ས་ཏཱ། སེམས་ཅན་ཞུགས་་་མ་དང༌། གུརྦ་ཎི;༽ལུས་ལྕི་མ་དང༌། ཨནྟ་རྦཏི། །ནང་ལྡན་མ་དང་། །གབྷི་ཎཱི། སྙིང་པོ་ལྡན་མའོ། །་་་་་་་་་་ང་ཚོ་ཞུ་པོ་དག། །ལ། བུ་ལ་མིང་ལྔ་སྟེ། །ཨཱཏྨ་ཛ། བདག་སྐྱེས་དང༌། ཏ་ན་ཡ། ལུས་་་་་འཕེལ་དང༌། སཱུ་ནུསུ་ཏ། པུ་ཏྲ་རྣམས་ཀྱང་བུ་སྟེ་དེ་རྣམས་སོ། །ཀན། ཕ་ལ་མིང་གསུམ་སྟེ། ཏཱ་ཏ། ཡབ་བམ་ཕའམ་སྐྱོང་བྱེད་དོ། །ཛ་ན་ཀ། །སྐྱེད་བྱེད་དོ། །པི་ཏྲ། ་ཕ་སྟེ་སྲུང་བྱེད་དོ། །དང་པོ་ནི། སྟེས་ཀྱི་ལྟ་::མ་ལ་མིང་བཞི་སྟེ། ཛ་ན་ཡ་ཏཱི། ཡུམ་མམ་སྐྱེད་བྱེད་མ་དང༌། པྲ་སཱུ། རབ་སྐྱེད་མ་དང་། མཱ་ཏཱ། འཚོ་མ་དང་། ཛ་ན་ནི། སྐྱེད་བྱེད་མའོ། །སྲིང་མོ་ལ་གཉིས་ཏེ། བྷ་གི་ནཱི། སྲིང་མོ་འམ་ཆེ་ཞེ་དང་། །སཱ་སཱ་་་་སྲིང་མོའོ། །།དུ་འཚོ་ཞིང་པ་ཚུ་བྱ་ཕྲ་ཚ་བོའི་མིང་ལ་གཉིས་ཏེ། ན་ཏྲི་དང་། པོ་ཏཱི། བུའི་བུའོ། །ཉོ་ཚོ་ཉེ་དུ་སྤྱིའི་མིང་ལ་དྲུག་སྟེ། ས་གོ་ཏྲ། རུས་མཚུངས་པ་དང་། བཱནྟ་་་བ༑འབྲེལ་པ་དང་། ཛཱ་ཏི། མཛའ་བཤེས་དང༌། བནྡྷརྩ་ལག་་དང༌། ་སྭགཉེན་དང་། །སཱ་ཛ་ན། ་་རང་རིགས་སམ་རང་གི་སྐྱེ་བོའམ་གཉེན་གྱི་སྐྱེ་བོའོ། །ཁ་བྱང་དུ་ཚེ་འགྱུར། །དེ་ཚོ། །ཏུ་ཏུ་—།
