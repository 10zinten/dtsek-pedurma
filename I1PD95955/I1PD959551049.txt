ཆོ་ག་ཁ་བསྟན་འགྱུར། སྒྲ་མདོ། །༼སི་ཤོཆ་ལྔའི་ཚིག་དང་ཀ་ཀོ་ལ་ལ་མིང་གསུམ་སྟེ། །ཀོ་ཀ་ཀཾ་དང་། ་ཀཀྐོ་ལ་ཀཾ་དང་། ཀོ་ཤ་་ཕ་ལ། སྦུབས་འབྲས་སོ། །།།དེ་ནས། ཁོ་ཁྲིད་ཁོ་དག་བུར་ལ་མིང་ལྔ་སྟེ། ཀཇུ་རནུས་པ་དང་ལྡན་པས་ག་བུར་སྟེ། གྲོ་བུ། ནུས་པ་ལ། ་ཨུར་རྐྱེན་དུ་བྱས་པའོ། །གྷ་ན་པཱ་ར། སྤྲིན་གྱི་སྙིང་་་་པོའོ༑། ཙནྡྲ་སཾ་ཙཱ། ཟླ་བའི་མིང་ཅན་ཏེ། ཟླ་བ་དང་ག་བུར་ཕན་ཚུན། མིང་གི་རྣམས་གྲངས་སོ། །སི་ཏཱ་བྷཱ། འོད་དཀར་ཅན། །ཧི་བ་ལུ་ཀཱ་ཁ་བའི་ཕྱེ་མའོ། །:། ལ་ལ་ལ་ལ་་་་་། ཀ་ཏུ་ཏ་ཀོབ། པ་ཚོ་དེ་ནས་ཚིགས་བཅད་ཕྱེད་ཀྱིས་ཙན་དན་སྤྱིའི་མིང་སྟེ། གནྡྷ་པཱ་ར། དྲིའི་སྙིང་པོ་དང་། མ་ལ་ཡ་ཛ། མ་ལ་ཡ་སྐྱེས་ཏེ། རི་བོ་མ་ལ་ཡ་ལས་བྱུང་བས་སོ། །བྷ་དྭ་ཤྲཱི། དཔལ་བཟང་པོ། ཙན་དན་ན། །ཙནྟ་ཞེས་གྲགས་པ་སྟེ། །ཚིམ་བྱེད་ཅེས་བྱ་སྟེ། ཅད་ཚིམ་པ་ལ་ལྷུརྔ་རྐྱེན་ནོ། །་བ། བྱེའོཙན་དན་དཀར་པོ་ལ་མིང་གསུམ་སྟེ། ཏི་ལ་པནྟི་ཀ། ཏིལ་མར་འདབ་ཅན་ཞེས་པའི་ཙནྡ་རི་བོ་མ་ལ་ཡའི་ཕྱོག་གཅིག་ནས་འབྱུང་བའོ། །གོ་ཤྲིནྟ། ས་མཆོག་གམ། བ་ལང་གོ་ཞེས་བྱ་སྟེ། བ་ལང་བོག་བའི་རྣམ་པ་ཅན། །མ་་ལ་ཡའི་ཕྱོགས་སུ་སྐྱེ་བའོ། །ཧ་རི་ཐུན་ན། ཞེས་པ་ལ། ཧ་རི་སྦལ་བ་སྟེ། དེའི་དབྱིབས་ཅན་གྱི་རི་བོ་ཞིག་ལས་སྐྱེས་པས་ཧ་རི་ཙནྡན་ཞེས་བྱའོ། །།ཀ་བ་དེ་ནས་ཙནྡན་དམར་པོའི་མིང་སྟེ། ཏི་ལ་པཎཱི། ཏི་ལ་མར་འདབ་ཅན་ནི། པ་དྲངྒེ། འདབ་མའི་ལུས་ཅན་ཏེ། དབུས་དམར་བའི་འདབ་མ་ཅན་ནོ༑། རཉྫ་ནམ། ཚོས་བྱེད་དོ། །རཀྟ་ཙནྡན། ཙནྡན་དམར་པོའོ། །ཀུ་ཙནྡན་ནི། ཙནྡན་ངན་པའི་མིང་ངོ༌། །།ཞ་དང་ཞིང་གཏུ་རྫོགས་པ་དེ་ནས་ཛ་ཏིའི་མིང་ལ། ཛཱ་ཏི་སྣ་མའམ། སྲི་ཤིང་སྣ་མ་དང་། ཀོ་ཥ། །
