སརལ་དབྱེ་སོགས་ཕྱིས། །ལ་ལར་ཨམ་ལ་སོགས་པའི་མཐའ་ཅན་ཕ་རོལ་ཏུ་སྤོ་བར་་བཤད་པས། ཨ་ཧན་རྗེས་སུ་སྤོས། ཊ་བྱིན། ༈ནོ་བཱ། ནའི་ཡང་ནའོ། །ན་མཐའ་ཅན་གྱི་ཊི་ནི་ཡ་དང་དབྱངས་རྐྱེན་ཕ་རོལ་ལ་དབྱི་ཞིང་སྐབས་ལ་ལར་ཡང་ན་ཨུ་པ་དྷཱ་དབྱི་བར་བྱ་སྟེ། ཊི་དང་ཨུ་པ་དྷཱ་དབྱི་བ་ཡང་ན་བར་བྱས་པས། སཱ་ཡཱནྣ®ཉིན་མཚན། སཱ་ཡཱ་ཏཿདེ་བཞིན་དུ་བསྒྲུབས་པས། ་ཨནྡྷངསི། ཨ་་པ་རཾསི། ཉིན་མོ་ལས་གཞན་ནོ། །ཨ་པ་རཱནྣིཉིན་མོ་ལས་གཞན་ནོ། །ཨ་པ་རཱ་ཧ་ཉིན་མོའི་གུང་། ཨནྣངསི། མ་དྷོ། སི། མ་དྷནྣཉིན་གུང་། མ་དྷ་ཧཡང་ལ་ལར་སྔ་ཕྱི་མི་བརྗེ་བ་ཡང་ཡོད་དེ། ཀ་བཱི་ཨམྲ་ནཱ་རཱ་ཛཱ། སི། སྙན་ངག་མཁན་གྱི་རྒྱལ་པོ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། སྔ་ཕྱི་མི་བསྒྱུར་ཊ་བྱིན་པས། ང་སྙན་ངག་མཁན་གྱི་རྒྱལ་པོ། ཀ་བཱི་རཱ་ཊ་ཛསི། ཀསྐབས་འདིའི་རྐྱེན་ཊ་ལ་ནི་ཊ་འགྲོ་བའི་ཆོ་ག་་་མཐའ་གཅིག་ཏུ་འཇུག་པ་ནི་མ་ཡིན་ནོ། །མོ་རྟགས་སུ་བྱེད་ན་ཨཱིཔྲ་བྱིན་པས། ཀ་བི་རཱ་ཛཱི། སྙན་ངག་མཁན་གྱི་རྒྱལ་མོ། །པད་འབྱུང་བ་དེ་དག་༈རྐྱེན་གཉིས་པ་ཨ་ཡིག་གི་དཔེ་ནི། རཱ་ཙྪཱཨཱ་མ་པཱུསི། །་རྒྱལ་པོའི་གྲོང་ཁྱེར། ཞེས་པའི་ཚིག་སྦྱོར་ལ་རྣམ་དབྱེ་སོགས་ཕྱིས། རྐྱེན་ཨ་བྱིན་པས། རཱ་ཛ་་པུ་རརྒྱལ་པོའི་གྲོང་ཁྱེར། དེ་བཞིན་དུ། །བཱཉྩ་སི་མ་ན༔ཡསི། ངག་དང་ཡིད་ཀྱང་ངོ༌། །བཱདྨ་ན་སངག་ཡིད། ༽ཡཱཊ་ཡདཀྵི་ཎ་སུཊ་སྩོ་དི་ཤི་ཇི་པནྡྷ་སི། ཡུལ་ལྷོ་ཕྱོགས་ཀྱི་ལམ། དཀླུ་ཎཱ་པ་ཐལྷོ་ཕྱོགས་ཀྱི་ལམ། ཨ་ཧསི་ཧྥ་རཱ་ཏྲི། ཉིན་མོ་དང་མཚན་མོ་ཡང་ངོ༌། །སི། །ཨ་ཧོ་རཱ་ཏྲི། ཉིན་མཚན། ་་། །ལྷ་དང་ཀུན། །འབད་བྱེད་ན་ལག་པའི་རྒྱུ་་་་ལ་་་་༌ཁཙོ་། ༈རྐྱེན་གསུམ་པ་ཌའི་དཔེ་ནི། ཨོ་རྡོ་ཙ་ཏྲ་ཡ༄༅། ཛས། གཉིས་དང་གསུམ་ཡང༌། ཞེས་པའི་ཚིག་སྦྱོར་ལ་རྣམ་དབྱེ་སོགས་ཕྱིས། རྐྱེན་ཌ་བྱིན། །ཌིཏྲ་ལ་ཊི་ཕྱིས་པས༑དྲི་ཏྲགཉིས་གསུམ། དེ་བཞིན་དུ། ལྔ་དང་དྲུག་ཡང་། པཉྩ་ཛས། ཙ་ཥ༄༅། ཛེས་—།
