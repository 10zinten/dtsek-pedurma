འོ་མཚོ་ཀཅོག་མའི་བློ་ཟངས་༈དེ་བཞིན་དུ་ཤིང་ཉལ་བ་ལ། ཨ་ཤ་ཡིཤྩ། ཨ་ཤ་ཡི་ཥཱ་ཏྭཾ། ཨ་ཤ་ཡི་ཥ་ཏི། ཨ་ཤ་ཡིཥྛ་ཨ་ཤ་ཡི་ཥཱ་ཐཱ། ཨ་ཤ་ཡི་དྷཾ་སོགས་སོ། །ཁོང་ལ་ནི། ཊ་ཤ་༈རུ་དྷིར་སྒྲིབ་པ་ལ་པ་རསྨི་པ་ད་རྣམས་སྦྱོར་བ་ལས། ༈ཨི་རི་ཏོ་བཱ། ཨི་རུ་ལ་ཡང་ནའོ། །བྱིངས་ཨིར་རྗེས་འབྲེལ་འགྲོ་བ་ལ་བྱུང་ཟིན་གྱི་དོན་བར་་་རྐྱེན་སིའི་ཡུལ་ལ་ང་སྦྱོར་བ་དང་ཡང་ན་སི་ཉིད་སྦྱོར་བ་འདམ་ཁ་ཅན་དུ་བྱ་སྟེ། ང་སྦྱར་བའི་ཕྱོགས་ལ། རྐྱེན་གྱི་ང་བར་མཚམས་ེབཟུང་བའི་དོན། །དྷ་ཨ་ལ་འདྲེན་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་རུ་དྷཏྲ། ཨ་རུ་དྷ་ཏྭཾ། ཨ་རུ་དྷན། ཨ་རུ་དྷཨ་རུ་དྷ་ཏཾ། ཨ་རུ་དྷ་ཏེ། ཨ་རུ་དྷཾ། ཨ་རུ་དྷཱ་བ། །ཨ་རུ་དྷཱ་མ། སི་ཉིད་སྦྱིན་པའི་ཕྱོགས་ལ། །དང་བའི་ཕྱིར་ན་མོ་ལ་ཆུ་ཡི་ས་ལ་འབད་པ་པ༈ཨ་ནི་ཊོ་ནཱ་མི་བ་ཏཨིཊ་མེད་པའི་ནཱ་མི་ལས་འཕེལ་བའོ། །བྱིངས་ཨིཊ་མི་འཇུག་པའི་དང་པོའི་དབྱངས་དེ་ནཱ་མིའི་ཡི་གེ་ཡིན་ན་སི་ཕ་རོལ་ལ་འཕེལ་བར་འགྱུར་ཏེ། བྱིངས་ལ་འཕེལ་བ། །སི་དབྱི་བ་དང་ཏ་ཐ་དྷར་བསྒྱུར་བ་སོགས་སོ་སོའི་ཆོ་ག་ཐོབ་ལྟར་བསྒྲུབས་པས། ཨ་རོཏིཏྲ། ཨ་རོ་དྡྷཾ། ཨ་རོདྡྷ། །ཨ་རོཧི། ཨ་རོ་དྷཾ། །ཨ་རོདྡྷ། ཨ་རོཏྟཾ། ཨ་རོཧཱ། ཨ་རོp། ཟླ་བ་དེ་ལ། སྤྲུལ་༈དེ་བཞིན་དུ་རྫི་དིར་གཉིས་སུ་བྱེད་པ་ལའང་། ང་དང་སི་ཡང་ན་་་་བར་བྱས་པས། ཨཙྪི་དཏྲ། ཨཙྪེ་ད་ཏྭཾ། ཨཙྪི་དན་སོགས་དང༌། ：ཨཏིཏྲ། ཨཙྪིཏྟཾ༑། ཨཙྪེpསོགས་སོ། །ན་ཏན་བཞི་དང་། བ། ༈དི་བུ་རྩེད་མོ་ལའོ་ཞེས་པ་ལ། ཨིཊ་འཇུག་པས་ཨིཊ་བྱིན། བྱིངས་ལ་ཡོན་ཏན་སོགས་ཀྱིས་བསྒྲུབས་པས། །ོཨ་དེ་བཱིཏྲ། ཨ་དེ་བཱིཏྟཾ། ཨ་དེ་བི་༑ཨ་དེ་བཱིསོགས་དང་དེ་བཞིན་དུ། ༈སི་བུ་བསྙེན་པར་བྱེད་པ་ལ། ཨ་ས་བཱིཏྲ། ་ཨ་སེ་བིཏྟཾ། ཨ་སེ་བི་ཥཱསོགས་སོ། །།ཤི་བ་གི་བྱིའི་ཁྲི་།
