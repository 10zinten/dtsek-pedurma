ཁ། ཏ་ཐོཎཊ་མཱ་དྷཱི་ཀ་མ་དྭ་ཡོ། །མི་རེ་ཡ་མཱ་ས་བཤྲཱི་དྷུམྨེ་ད་ཀོ་ཛ་ག་ལིས་མོ། །སནྡྷཱ་ནཾ་སྦྲང་གི་ཆ་ནི་གཉིས། །མི་རེ་ཡ་དང་མེ་ཏོག་ཆང་། །བུར་ཆང་མེ་ད་ཀ་དང་ནི། །སྦྱར་བའི་ཆང་དག་སྣ་ད་བྷི་ཥ་བཀིཙྩཾ་པུཾ་སི་ཏུ་ན་གཱུ་ཧུ། །ཀཱ་རོཏྟ་རཿསུ་རཱ་མཎྜ་ཨཱ་པཱ་ཎི་མཚུངས་པའོ། །སནྡྷཱ་ནཾ་ནི་ཆང་གཡོས་སོ། །ཕབས་དང་ཕོ་ནི་ཆང་རྩིའོ། །སྙིང་ཁུ་བདུད་རྩིའི་སྙིང་པོ་དང་། །པཱ་ནཾ་གོ་ཀོ། །བ་ཥ་ཀོཉསི་པཱ་ན་པ་ཏྲ་ས་ར་ཀོ་པྱ་ནུ་ཏ་ཎཾ། །ཆང་ར་ཆང་འཐུངས་གནས་དང་ནི། །ཆང་སྐྱོགས་མོ་མིན་ཆང་སྣོད་དང༌། །ཁམ་ཕོར་དང་ནི་ཚིམ་བྱེད་དོ། །དྷཱ་རྟོཞཀྵ་དེ་བཱི་ཏོ་བོཉག་དྷཏྟཾ་དྷུ་ཏ་ཀྲྀཏྠ་མཱ། །སཱུརླ་ག་ཀཿབྲ་ཏི་བྷཱ་བས་བྷི་ཆོ་ལོ་རྒྱུན་པོ་ཤོ་དང་ནི། །ཟ་རྒྱན་རྩེད་རྒྱན་མཚུངས་པའོ། །ཁས་ལེན་དང་ནི་གཉའ་བོའོ། །ཀཱརླུ་ཏ་ཀཱ་ར་ཀཱ། །རླུ་ཏོཉསྦི་ཡཱ་མཀྵ་བ་ཏཱི་ཀེ་ཏ་ཐཾ་པ་ན་ཨི་ཏུ་མགོ་འཐུ་བ་དང་ཆ་འཛིན་ནོ། །ཤིང་རྩེ་མོ་མིན་རྒྱན་འགྱེད་དང༌། །ཀེ་ཏ་བ་དང་པ་ན་པི༑། པ་ཎཉཀྵེ་ཥཱ་གླ་ཧོཉཀྐོསྐུ་དེ་བ་ནཱབཱ་ཤ་ཀཱཉྩ་ཏེ། །པ་རི་ཎཱ་ཡསྐུ་ཤཱ་རཱི་ཎཾ་ས་ཞེས། །ཐ་སྙད་ཤོ་དང་རྒྱལ་ཞེས་པ། །ཤོ་དང་ཆོ་ལོ་པཱ་ཤ་ཀཱ། །ཡོངས་འདྲེན་བརྐྱངས་དང་ཁྱི་མནྟཱནྣ་ཡ་ནེཉལྡི་ཡོ། །ཨཏྟཱ་པ་ད་ཤཱ་རི་ཕ་ལཾ་པྲ་ཎི་དྷཱུ་ཏཾ་ས་མཱ་ཧཱ་ཡ། །ཨུཀྟ་དང་དྷོ། །མོ་མིན་རྐང་བརྒྱད་ཤོ་ཡ་སྟན། །སྲོག་ཆགས་སྐུགས་དང་ཤོ་རྒྱལ་ལོ། །རབ་ཏུ་མང་པོ་ལ་བྷུ་རི་པྲ་ཡ་ག་ཏཱ་དེ་ཀསྨི་རྐྱེཉཏྲ་ཡོ་གི་ཀཱ། །ཏདྷརྨ་ད་ནྱ་ཏོ་བྲྀཏྟཱ་བུ་ཧུ་སྦྱོར་ཕྱིར། །གཅིག་ཏུ་བཤད་པས་འདིར་གང་སྦྱོར། །དེ་ལས་ཆོས་གཞན་ལ་ཡང་སྦྱོར། །རྟགས་ཀྱི་ལིངྒན་རེ་པི་ཏེ། །ཤུ་དྲ་བརྒ། །།མཐའ་ལས་བསྒྲུབ་པར་བྱ། །དམངས་རིགས་ཀྱི་སྡེ་ཚན་ནོ། །་ད༈ཨི་ཏྲ་མ་ར་སི་ཧ་ཀྲྀ་ཏོ་ནཱ་མ་ལིངྒ་ནུ་ཤཱ་ས་ནེ་དཱི་ཏཱི་ཡོ་བྷྱཱ་མི་ཀཱཎྜི་ཡཾ་སཱངྒ་༈འདིར་འཆི་བ་མེད་པའི་སེང་གེས་སྦྱར་བའི་མིང་དང་རྟགས་རྗེས་སུ་བཤད་པ་ལས། ས་གཞིའི་ཨེ་བ་ས་མརྗི་ཏཱ། །།།ང་ཚོ་ཁའོ། ཁོ་། །སྐྱེ་ན་དངོས་ལ། ཁསྐབས་ཏེ་གཉིས་པ་ཉིད་དོ། །།རྒྱུ་འཚེ་བཏོའི་སྒྱུ་རྒྱལཀྱིཀྱིའི།
