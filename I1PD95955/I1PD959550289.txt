འོ་་ཁཚོ་ཉཀ་ཚརཝ་ཌ་བློ་ཌན་ནི༑བ་རི་བནྟི། བ་རཱི་བནྟི། བཛྲཏྠབ་རི་བྲྀཏྠབ་རཱི་བྲྀཏྠབ་བྲྀཏྠ། ་བ་རི་བྲྀཏྠ། །བ་རཱི་བྲྀཏྠ། །བཛྲ་ཏཱི་མི། །བ་རི་བྲྀ་ཏཱི་མི། བ་རཱི་ཤྲཱི་ཏཱི་མི། བརྦ་རྨི། །བ་རི་བཛྫི། ་བ་རཱི་བཛྫི། བཛྲ་ཏྭབ་རི་བྲྀ་ཏཱ་བ་རཱི་ཤྲཱི་ཏཱབཛྲpཔ་རི་བྲྀཏྠབ་རཱི་བྲིp། འདྲེ་རི་བ་ཏི། །ས་ང་ལ་:ཞེ་༼༈ཆོག་འབྱུང་ལིང་ལ། བཛྲ་ཏྭཱཏྲ། བ་རི་བྲྀ་ཏྱཱཏྲ། བ་རཱི་བཾ་ཏུཏྲ་་་སོགས་སོ། །ཤིས་བྱེད་ལོཊ་ལ། བཛྲ་ཏཱི་ཏུ། བ་རི་བྲྀ་ཏཱི་ཏུ། བ་རཱི་བྲྀ་ཏཱི་ཏུ། བརྦརྟུ༑བ་རི་བརྟུ། ་བ་རཱི་བརྟུ། བཛྲ་ཏཱཏྲ། བ་རི་བྲྀ་ཏཱཏྲེ། བ་རཱི་ཤྲཱི་ཏཱཏྲ་སོགས་དང་། དེ་རིང་མ་ཡིན་པ་ལང་ལ། ་ཨ་བཛྲ་ཏཱིཏྲ། ཨ་བ་རི་ཤྲཱི་ཏཱིཏྲ༑་ཨ་བ་རཱི་བྲྀ་ཏཱིཏྲ། ་ཨ་བཛྲ་ཏྲ། ཨ་བ་རི་བྲྀཏྲ། ཨ་བ་རཱི་བྲྀཏྲ། ཨ་བཛྲ་ཏཱཾ། ཨ་བ་རི་བྲྀཏྟཾ། ཨ་བ་རིབྲྀཏྟཾ། ཨ་བཛྲ་ཏུཨ་བ་རི་བྲྀ་ཏུ། ཨ་བ་རཱི་བྲི་ཏུསོགས་སོ། །ཨི་ཏི་སྭ་ར་སྭ་སྟི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་དུརླུ་གནྟ་པྲ་ཀྲི་ཡ་ས་མཱདྷཱི། ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཡང་འཇིག་པའི་་མཐའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ། །ན་། ་བ་མ། །ཁྲོ་ཊ་ཀྱི་རྗེ་ཀྱི་ང་བ་དེ་རིགས་ལ་བརྟེན་པ་བ་ཡང་། །བ། །Rཚོ་ཁ་བ་ད། །ཁ་༈དེ་ནས་བྱེད་པའི་དོན་སོགས་བྱིངས་བྱིངས་སུ་སྒྲུབ་པ་དང༌། །མིང་་བྱིངས་སུ་སྒྲུབ་པའི་རྐྱེན་རྣམས་ཀྱི་བྱ་བ་རྣམས་བཤད་པར་བྱ་སྟེ། དེ་ལ་རྐྱེན་དང་པོ་ཡང་གི་འཇུག་ཚུལ་ནི། ༈ཀརྟུ་ཡང་། བྱེད་པ་པོ་ལ་ཡང་ངོ༌། །།ངན་པ་ལ་བཟང་པོར་རློམ་པ་སོགས་ཉེ་བའི་ང་རྒྱལ་ཀུན་ཏུ་སྤྱོད་པའི་བྱེད་པ་པོའི་དོན་ལ་་་མིང་དང་བྱིངས་ལས་བྱིངས་སུ་སྒྲུབ་པའི་རྐྱེན་ཡང་འགྱུར་ཏེ། །ཤེ་ན་ཞེས་པའི་ས་མིང་ལ་བྱིངས་སུ་སྒྲུབ་པའི་རྐྱེན་ཡང་སྦྱར། ཡ་ལའོས་མིང་མཐའ་རིང་པོ་དེ་ནས་ཨཔྲ་བི་ཀཱིར་ཎ་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཤེ་ན་ཨི་བ་ཨཱ་ཙ་ར་ཏཱི་ཏི་ཀཱ་།
