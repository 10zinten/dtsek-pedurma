་ཁཁོ་ས་ཏཏྟཀ་ཀོ་བི་བྷ་བ་སུ་གྲ་ཧ་པ་ཏིསི་ཥསྨྲ་ཏི་ར་ཧཔ་ཏི། ་་། བྷཱ་ནུསའོད་ཀྱི་ནོར་ཅན་གཟའ་ཡི་བདག། འོད་ཟེར་བདག་པོ་སྣང་བྱེད་བདག་། འོད་ཅན་འཇོམས་བྱེད་འོད་ས་ཧ་སྭཱ་ཤུས་པ་ནིས་ཐཾ་ཏཱ་ར་བི། །མཱ་ཋ་ར་པི་ག་ལོ་དཎྜ་ཧྥཎྞཾ་ཤོསྟོང་པ། །གདུང་བྱེད་འོད་འབྱིན་འོད་ཟེར་གཟུགས། །གནས་བྱེད་དམར་སེར་དབྱུག་ཅན་གྱིས། །ཉི་མའི་པཱ་རི་པཱནཱི་ཀཱ། །སཱུ་ར་སཱུ་ཏོཉརུ་ཎཉནཱ་ཀཱ་སྱ་པིརྒ་རུ་ཌཱ་གྲ་ཛ། །པ་རི་འོད་ལ་ཡོངས་སུ་བསྐོལ། །ཉི་མའི་ཁ་ལོ་སྐྱ་རེངས་དང༌། །བརླ་མེད་འོད་སྲུང་ཁྱུད་སྔོན་སྐྱེས། །ཡོངས་བེ་ཥསྡུ་པ་རི་དྷི་རུ་པ་སཱུརླ་ཀ་མཎྜ་ལེ། །ཀི་ར་ཎོ་པྲ་མ་ཡུ་ཁཾ་ཤུ་བསྐོར་ཡོངས་སུ་དཀྲིས་པ་དང་། །ཉེ་བའི་ཉི་མའི་དཀྱིལ་འཁོར་རོ། །འདེད་བྱེད་འོད་བྱེད་ནམ་མཁའ་ག་བྷསྟི་ཤྲཱི་ཎི་དྷཾཥྛ་ཡ། །བྷཱ་ནུཀ་རོ་མ་རཱི་ཙིཙླི་སུཾ་ས་ཡོཇི་འཇལ། །ཕྱོགས་འགྲོ་འོད་འགྲོ་ཆུ་འཐུང་ཡོད། །གསལ་བ་འོད་བྱེད་སྲིན་བུ་འཇོམས། །མོ་དང་སྐྱེས་དྷི་ཏིཏྲི་ཡཾ། །སླུཔྲ་བྷཱ་རུ་གུ་ཙི་ཏཱིབྷཱ་བྷཱསྦ་བི་དྷུ་ཏི་དཱིཥ་ཡ། །རོ་བུ་གསལ་བྱེད་མོ། །འོད་དང་གསལ་བ་སྣང་བ་སྒྲོན། །སྣང་གསལ་འོད་དང་འོད་ཟེར་དང༌། །འོད་ཙིཤོ་ཙི་རུ་བྷེ་ཀི་བེ་པྲ་ཀཱ་ཤོ་བྱོ་ཏ་ཨཱ་ཏ་པ། །ཀོཎྞཾ་ཀ་བོཋམྨོ་དོད་མདངས་གསལ་སྒྲོན་མ་ནིང་ངོ༌། །མུན་སེལ་མུན་འཇོམས་མཚན་མོ་སེལ། །ཅུང་དྲོ་བག་དྲོ་དྲོ་ཙམ་དང་། དྲོཎྞནཱི་ཥཱ་ཏངྒ་ཏི། །ཏིགྷཱནྟི་ཧཱུཾ་ཁ་རནྟདྨ་ཎྞཾ་ག་ཀྲྀཏྟཱ་མ་རཱི་ཙི་ཀཱ། བོ། ་བ་དྲོ་བ་དེ་ལྡན་གསུམ་ཡིན་ནོ། །ཚ་བ་རབ་ཚ་ཚ་དྲག་གོ། །དེ་བཞིན་སྨིག་རྒྱུ་རི་དྭགས་སྲེད། །ར། ༈ཀཱ་ལོ་དིཏྠོ་པྱ་ནེ་ཧཱ་པི་ས་མ་ཡོ་བྱ་ཐ་བ་ཏི། །པྲ་ཏི་བདྡེ་༈ལོའི་མིང༌། བགྲང་བྱ་དུས་དང་སླར་བྱུང་དང༌། །ཉེ་བར་འགྲོ་དང་དེ་ནས་ནི། །ཕྱོགས་རྩ་འཕེལ་ཨི་མཙླི་ཏྲེ་ཏ་རཱ་སྟི་ཐ་ཡརྡ་ཡོ། །གྷ་སྲོ་དི་ནཱ་ཧ་ནི་བ་འགྲིབ་འདི་གཉིས་མོ། །དེ་སོགས་ཚེས་བདག་གཉིས་དག་གོ། །མུན་བསལ་ཉིན་མོ་བྱ་སྤངས་བསལ། །རུ་ཀཱི་བེ་དི་བཱ་ས་རོ། ་། པྲ་ཏུ་ཁོཉཧརྨུ་ཁཾ་ཀ་ལྱ་མུ་ཥཔྲ་ཏུ་ཥཔཱི་ཨ་ཡང་ན་མ་ནིང་རྩེ་ནས་འཇོམས། །མུན་དངས་ཉིན་མོ་རབ་སད་སྲེད། །།ནམ་ལངས་པ་དང་རབ་སྔ་—།
