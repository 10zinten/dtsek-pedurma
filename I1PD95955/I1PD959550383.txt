།བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི། ཋཀ་སྤྱི་སྒྲོལ་ཎའོན་སཱ་ཉྩ་ཏུ་དཱཉྩ་ག་ན་དེ་བ་ཏཱ། ༈། བི་རཱུ་དྷ་རོ་ཉཔ་རོ་ཡཀྵ་རཱཀྵ་ས་པ་དྲག་པོ་སྟེ། །དེ་རྣམས་ཚོགས་པའི་ལྷ་རྣམས་སོ། །རིག་འཛིན་ཆུ་སྐྱེས་གནོད་སྦྱིན་དང་། །སྲིན་པོ་གནྟབྷ་ཀིནྣ་རཱ། །པི་ཤཱ་ཙོ་གུ་རྒྱ་ཀསིདྡྷི་བྷུ་ཏེ་ཉམཱི་དེ་བ་ཡོ་ན་ཡ། །ཊོ་ལི་དྲི་ཟ་མི་འམ་ཅི། །ཤ་ཟ་གསང་བ་གྲུབ་པ་དང༌། །འབྱུང་པོ་འདི་རྣམས་ལྷ་ལས་བྱུང་། ་། །ཆཞུཥ་ཎ་ཊ་༈ཨ་སུ་རཱ་དེ་ཏུ་དི་ཏེ་ཡ་དྷ་ནུ་ཛེན་རི་དཱ་ན་བཱ། ཙཱ། ཤུ་༈ལྷ་མིན་གྱི། ལྷ་མིན་སྦྱིན་སྐྱེས་བཅུད་མི་ལྡན། །གཞུ་སྐྱེས་སྦྱིན་བྱུང་དབང་པོའི་དགྲ། །པ་ཀྲ་ཤི་རྒྱུ་དི་ཏི་ཤུ་ཏཱཔཱཉྫ་དེ་བཱསུ་ར་དྭི་ཥ། །བློ་ང་་ང་སྒོ་སངས་སློབ་མ་སྦྱིན་བྱེད་བུ། །སྔོན་གྱི་པུསྟ་སུ་ཏྲ་ཤི་ཤུ་པཱ་ལ་རྣམས་བུའི་བྱིས་པ་ལྷ་དང་ལྷ་ལ་སྡང༌། །།།དེ་༈སརྔ་ཙཱུསུ་ག་ཏོ་ཤུདྡྷོ་དྷརྨ་རཱ་ཛསྟ་ཐཱ་ག་ཏ། ཏི་ར་། ༈སངས་རྒྱས་སྤྱིའི། ཀུན་མཁྱེན་བདེ་གཤེགས་སངས་རྒྱས་དང་། །ཆོས་ཀྱི་རྒྱལ་པོ་དེ་བཞིན་གཤེགས། །ས་མན་བྷ་དྲོ་བྷ་ག་བཱནྣཱ་ར་ཛིཀློ་ཀ་ཛིཛྫི་ན། །ཥ་ཌ་བྷི་ཛྙཱ་ཀུན་ཏུ་བཟང་པོ་བཅོམ་ལྡན་འདས། །བདུད་འདུལ་རྒྱལ་བ་འཇིག་རྟེན་འདུལ། །མངོན་ཤེས་དྲུག་ལྡན་ད་ཤ་བ་ལོཉདྭ་ཡ་བཱ་དཱི་བི་ནཱ་ཡ་ཀཿ། །།བྱེ་བྲག་ཏུ་ཐུབཔའིམུ་ནཱིནྡྲཤྲཱི་གྷ་ན་སྟོབས་བཅུ་པ། །གཉིས་མེད་གསུང་བ་རྣམ་པར་འདྲེན། །ཐུབ་པའི་དབང་པོ་དཔལ་སྟུག་དང༌། །སྟོན་པཱ་ཏཱ་མུ་ནིཤཱཀྱ་མུ་ནི་ཏུ་ཡ། །ས་ཤཱཀྱ་སིཾ་ཧསནྡྷརྦ་སིདྡྷ་སོ་རྣོ་ད་ནི་ས། །པ་ཐུབ་པ་ཉིད་རྣམས་སོ། །ཤཱཀྱ་ཐུབ་དང་དོན་ཀུན་གྲུབ། །ཤཱཀྱ་སེངྒེ་ཟས་གཙང་སྲས། །གོ་ཏ་མཤྩཎྜ་བཛྫཱཤྩ་མཱ་ཡཱ་དེ་བཱི་སུ་ཏཉྩ་ས། །་ཌ་། ་་་ཀློབ་བྱ་ག་གོ་ཏ་མ་དང་ཉི་མའི་གཉེན། །ལྷ་མོ་སྒྱུ་འཕྲུལ་སྲས་རྣམས་སོ། །་ཎ། །བྱ་བ་ཀཚོ། ཟླ་བའི་ཁཚོན༈བྲགྷཱཏྨ་བྷཱུསུ་ར་ཛེཥྛ་པ་ར་མེཊཱི་བི་ཏཱ་མ་ཧ། ཧི་ར་ཎྱ་གབྷོ་༈ཚངས་པ་བདག་སྐྱེས་ལྷ་ལས་རྒན། །མཆོག་ལ་གནས་དང་མེས་པོ་དང་། །གསེར་གྱི་མངལ་དང་ལོ་ཀེ་ཤསྭ་ཡཾ་བྷུཉྩ་ཏུ་རཱ་ཎ་ན། །དྷཱ་ཏཱབྱཱ་ཡོ་ནིརྡུ་ཧི་ཎོ་བི་རིཉྩིཀ་འཇིག་རྟེན་དབང་། །རང་བྱུང་གདོང་བཞི་འཛིན་བྱེད་དང༌། །པདྨའི་སྐྱེ་གནས་ཉེས་ཅན་དང་། །སྣ་ཚོགས་ཤཱསེAJ—།
