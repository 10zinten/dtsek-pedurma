འཆི་མེད་མཛོད་ཀྱི་རྒྱ་ཆེར་འགྲེལ་པ་འདོད་འཇོའི་བ་མོ་ཞེས་བྱ་བ་བཞུགས་སོ། །སྲི་ཥོ་ད་ར་ལ་སོགས་པས། །་ཌ་ཨུ་ལུ་ཀ་རབ་སྒྱུར་བའོ། །བཱ་ཡ་པཱ་ར་ཏེ། བྱ་་་་རོག་དགྲ་སྟེ། བཱ་ཡ་པ་བྱ་རོག་དང་། །་ཨ་རཱ་ཏི། དགྲའོ། །སེ་ཙ་ཀ། འུག་པའོ། །།།་་་བས་འདུ་དུ་མ་ཞཔ། རིན་གོང་ན་ཁང་བདུར་བྱ་ལ་མིང་གཉིས་སྟེ། ལོ་ཉཔྲཥྛ། རྒྱབ་དམར་དང་། ཀང་ཀའོ། །ལོ་དེ་ནས་ཙཱ་ཤ་དང་། ་་ཀེ་ཀི་དི་བི་གཉིས་ནི། སྦུའི་མགོ། བི་ཏུམ་་ལྟར་དྭངས་ཤིང་བཤོག་པ་བཀྲ་བའི་བྱ། གསེར་གྱི་གཙུག་ཕུད་ཅེས་གྲགས་པའི་་མིང་ངོ༌། །ཨ་ཏོ། །ཞི་བའི་གོ་བ་ཁང་དུ་འཚོ་ཞུ་བ་གསན་ཏmཏོག་ག་ན་ཀློ། ང་ཚོ་བམ་དདེ་ནས་བྱ་ཤིང་རྟ་མོའི་མིང་གཉིས་སྟེ། །ཤ་ཏ་པ་དྲ། འདབ་མ་བརྒྱ་པ་དང༌། དཱནཱ་གྷ་ཊ། ཤིང་འཇོམས་ཤིང་དགྲའམ་ཤིང་རྟ་མོ་སྟེ། དཱརུ་ཤིང་་་་་དང་། ཨགྷ་ཊ་འཇོམས་པའོ། །།་དུ། །ཚེ་དེ་ཤེས་སུ་ཕྱིན་ཏེ་ཀ་ཏོ། དེ་ནས་ཁུག་རྟ་ལ་མིང་གསུམ་སྟེ། ལོ་པཱ་རང། ཁུང་རྟ་དང་། སྟོ་ཀ་ཀ། ཆུ་ཉུང་སྟེ། སྟོང་ཀ་ཉུང་ངུ་དང་ཀ་ཆུ་སྟེ། །ས་ཡིག་ཕྱིས་པའོ།  ོ། ཙཱ་ཏ་ཀ་་རྣམས་སོ། །།།གང་ལ་། བལག་དམ། ཡཏོང་་ཁ་ག་ལ་ལའང་། ས་པ་དང་། ཁྱིམ་བྱ་ལ་མིང་བཞི་སྟེ། ཀྲི་ཀ་བཱ་ཀུ་དང༌། ཀུཀླུ་ཊ་ཁྱིམ་བྱ་སྟེ། ཞེས་སྒྲོགས་པའོ། །དྲ་མ་ཙུ་ཊ། ཟངས་ཀྱི་གཙུག་ཕུད་ཅན་ནོ། །ལ་། /ནོ། །ང་། ཌ་ལ་གྲོང་གི་བྱིའུ་ལ་མིང་གཉིས་སྟེ། ་ཙ་ཌ་ཀབ། །མཆིལ་བ་དང༌། ཀ་ལ་་བིངྒའོ༑། འོ། །་། །་དག་ལ། བྱི་ཏ་ལ་ག་ལྔའི་སྒྲུབ། ཀ། ཀ་པ། ཁགངཀ་བ། །ན། ཁཁུ་ཕྱུག་ལ་མིང་བཞི་སྟེ། །བ་ན་པྲ་ཡ། ནགས་ན་དགའ་དང༌། པ་རཾ་བྷ་ཏ། །གཞན་གྱི་གསོས། །ཅེས་བྱ་སྟེ། བྱ་རོག་གི་བཟུང་ནས་བསོ་གར་བྱེད་པས་སོ། །ཀོ་ཀི་ལ་དང་། པི་ཀ། ཁུ་ཕྱུག་གོ། ཤོག་། དྲི་པ་ཏབྱ་རོག་གི་མིང་ལ་ཚིགས་སུ་བཅད་པ་གཅིག་སྟེ། ཀཱ་ཞེས་པའི་སྒྲ་སྒྲོགས་་II།
