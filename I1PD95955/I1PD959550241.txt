།ཤ་ཀ་ལྟ་ཀ་ཎ། བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་ཁཚོ་ཀཀ་ཎཐཱ་ག་ཚོ་ལྔ་པར་བྱ་སྟེ། ༈ཨིཙྪ་ཡཱ་མཱཏྨ་ནསབདག་གི་འདོད་པ་:ལ་སའོ། །བྱིངས་་ལ་བདག་ཉིད་ཀྱི་འདོད་པའི་དོན་ལ་ས་རྐྱེན་དུ་འགྱུར་ཏེ། ་རྔུ་ཀྲྀཉ་བྱེད་པ་ལ་་ས་སྦྱར་བ་ལ། ཀྲྀ། ས། །དང་། ༈རྀཏྲ་ཨིར། ་རྀཏྲ་ཨིརྔའོ། ་། བྱིངས་མཐའི་རི་ཡིག་ནི་ཀྲ་ཡིག་དང་དྲ་ཡིག་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨིར་དུ་འགྱུར་ཏེ། ་རོ་ཨིར་དུ་བསྒྱུར་་་བས། ཀིརྔ་ས། ཞེས་པ་ལ། ༈སཱཏྲ་ོདྭིཤྩ། ས་ལས་ཀྱང་གཉིས་སོ། །བདག་གི་འདོད་པའི་དོན་གྱི་ས་ཕ་རོལ་ལས་ཀྱང་བྱིངས་གཉིས་སུ་འགྱུར་ཏེ། གཉིས་བརྗོད། ་ཀིར་ཀིར། ས་ཞེས་གནས་པ་ལ། ༈པུརྦ་ཧ་སཱ་དིཤེ་ཥསྔོན་མའི་ཧ་ས་དང་པོ་ལྷག་པོའོ། །བྱིངས་གཉིས་བརྗོད་པའི་སྔོན་མའི་ཧ་ས་་་དང་པོ་དེ་ལྷག་པོ་སྟེ་གནས་ལ་ཧ་ས་གཞན་རྣམས་དབྱི་བར་བྱ་སྟེ། །རྔ་གོང་མ་་་ཕྱིས༑ཀུ་དང་ཧའི་ཙུའོས་ཀ་ཙ། ཨིནྡྲ་སོགས་ཀྱིས་རིང་པོ། །ཀུ་ཨི་ལ་ཞེས་པས་ས་ཥར་བསྒྱུར་བས། ཙི་ཀཱིརྟ་ཞེས་གནས་པ་ལ། ཀཱས་ཞེས་པས་ཨཱམ་་་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཙིཀཱིནཱཾ་ཙ་ཀཱ་ར། བདག་གི་འདོད་པས་བྱེད་དོ་སྔོན་བྱས་སོ། །ཙི་ཀཱིནཱ་མཱ་ས། ཙི་ཀཱི་ནཱཾ་བ་བྷཱ་བ་སོགས་སོ། །ཆོ་ག་དེ་བཞིན་དུ་བྱས་་་པས༑བྷུཛྲ་སྐྱོང་བ་ལ་སྟེ་ཟས་གཡོ་བ་ལའང་ངོ་ཞེས་པ་ལ། བུ་བྷཀྵ་ཙ་ཀཱ་ར། བུ་བྷགྷཱ་མཱ་ས། བུ་བྷཀྵ་བ་བྷཱ་བ་སོགས་སོ། །རྒྱུ་༈ནཱམོ་ཡ:༔ཨཱི་ཙཱ་སྱ། །མིང་ལ་ཡ་སྟེ་ཨའི་ཡང་ཨཱིའོ། །མིང་ལས་་་བདག་གི་འདོད་པའི་དོན་ལ་ཡ་རྐྱེན་དུ་འགྱུར་ཞིང་ཡ་ཕ་རོལ་ལ་མིང་མཐའི་ཨ་་ཡང་ཨཱི་རུ་འགྱུར་ལ་ཨཱམ་སོགས་ཀྱང་སྦྱར་བར་བྱ་སྟེ། །པུ་ཏྲ་ལ། ཡ་མིང་་་མཐའི་ཨ་ཨཱི་:དེ་ལ་ཨཱམ་སོགས་སྦྱར་བས། པུ་ཏཱི་ཡཾ་ཙ་ཀཱ་ར། ་པུ་ཏཱི་ཡཱ་མཱ་་་ས༑སྔོན་བྱུང་བའི་བུ། པུ་ཏཱི་ཡཾ་བ་བྷཱུ་བ་སོགས་སོ། །ཀ་མ་ལྔ་པ་་་་་འོ་ཚོའི་།
