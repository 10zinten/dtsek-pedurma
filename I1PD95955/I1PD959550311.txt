གིཎན་བསྟན་འགྱུར། སྒྲ་མདོ། གི་པའི་ཚོ་ཁ་ཁ་ཡིན་ལ་བློ་ཁའི་པ་༈རུ་དཱ་དེ་ཙ་ཏུནྟཾ་ཧྭ་སཱ་དེརུད་སོགས་ལས་བཞི་པོ་རྣམས་ཀྱི་ཧ་དང་བ་ས་ལའོ། །བྱིངས་རུད་ལ་སོགས་པ་ལས་ཕ་རོལ་:གྱི་རྣམ་དབྱེ་བྱིངས་ཐམས་ཅད་པའི་མིང་ཅན་བཞི་པོ་རྣམས་ཀྱི་རྐྱེན་དང་རྣམ་དབྱེ་ཧ་ཡིག་དང་བ་སའི་ཡི་གེ་དང་པོ་ལ་ཨིཊ་ཨཱ་ག་མ་སྟེར་བར་བྱ་སྟེ། ༼༈རུ་དིར་མཆི་མ་འཛག་པ་ལ་ཨདྲ་སོགས་ཀྱི་ཨཔྲ་འཇིག། ཧ་དང་བ་སའི་ཡི་གེ་དང་པོ་ཅན་ལ་ཨིཊ་སྟེར་བ་སོགས་ཀྱིས་་་བསྒྲུབས་པས། རོ་དེ་ཏི། རུ་དི་ཏརུ་དནྟི། རོ་དི་ཥ། རུ་དི་ཐརུ་དི་ཐ། རོ་དི་མི། རུ་དི་བརུ་དི་མཉྩསོགས་དང་། དེ་རིང་མ་ཡིན་པ་ལང་གི་དིཔྲ་དང་སིཔྲ་ལ་ཨཱིཊ་དང་ཨཊ་འདམ་ཀ་ཅན་དུ་བྱས་པས། ངཨ་རོ་དཱིཏྲ། ཨ་་རོ་དཏྲ། ཨ་རོ་དཱིཨ་རོ་དསོགས་སོ། །སྦྱོ་རཀ་ཁ་བོ་ཤ?། དེ་བཞིན་དུ། ཤཱས་སྲོག་ལ།  ེཤཱ་སི་ཏི། ལཊ། ཨ་ཤཱ་སཱི་ཏ། ལང༌། ཨ་ཤཱ་སཏྲ་སོགས་དང༌། ༈ཉི་ཥ་པ་ཉལ་བ་ལ། སཱ་པ་ཏི། ལཊ། ཨ་སཱ་་་པཱིཏྲ༑ལང་། ཨ་སཱ་པཏྲ་སོགས་དང་། ༈པྲཎྜ་རྩེད་ོམོ་ལ། པྲ་ཎི་ཏི། ལཊ། ཨ་པཱ་ཎྜིཏྲ། ལཊ། ཨ་པཱ་ཎཏྲ་ལང་སོགས་དང་། ༈ཛཀྟ་དགོད་པ་ལ། །ཛཀྱི་ཏི། ལཊ། ཨ་ཛཀཱིཏྲ། ལང་། ཨ་ཛཀཏྲ་སོགས་དང་། ཡང་ཛཀྟ་སོགས་ལ་ཨནྟ་རྣམས་ཀྱི་ན་དབྱི་བ་དང་ཨན་ཨུས་སུ་བསྒྱུར་བར་བཤད་པས། ལཊ། ཛཀཱ་ཏི། ཨནྟི། ལང་ཨ་ཛgཨན་སོགས་སོ། །དེ་རྣམས་ནི་རུད་ལ་སོགས་པ་སྟེ་ཇི་སྐད་དུ། སྭ་ལཊ་:པ་ཏི་ཙི་བ་ལཊ་ཤཱ་སི་ཏི་པཱ་ཎི་ཝེཏི་ཏ་ལཊ་ཐཱ། ཛ:ཀཱིཏི་ཙི་ེབ་བིདྡྷེ་ཡོ་རུ་དཱ་དིཔཉྩ་ཀོ་སྟེག་ཎ། འདི་དང་འདྲ་བར་སཱ་པི་ཏི་ཡང་སྟེ། དེ་བཞིན་ཤཱ་སི་ཏི་དང་པཱ་ཎི་ཏི། །ཛ་ཀྱི་ཏི་ཡང་འདི་འདྲར་ཤེས་བྱ་སྟེ། རུད་ལ་སོགས་པ་ལྔ་པོའི་ཚོགས་རྣམས་སོ། །པ་༈ཡང་ཨ་ཉ་དང་ཨན་བསྒྱུར་བའི་ཛཀླུའི་སོགས་ཁོངས་ནི། ཛཀྟ། ཛཱ་གྲི༑ད་རི་དཱ། ཤཱས། དི་དྷི། བེལྕི། ཙ་ཀཱ་སྭ་སྟེ་བྱིངས་བདུན་པོ་དེ་་་།
