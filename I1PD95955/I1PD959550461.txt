ཁ་ཁ་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེཀྱིཔའི་ཀྱི་རྫོང་ཁོཊཔཝཱནྟི་ཏལྤ་ཀ་ཊཱཉྩཀཱི་བེ་ཏུ་ཛ་གྷ་ནཾ་པུ་ར། །།ཀཱུ་པ་ཀོ་ཏུ་བུད་མེད་རྣམས་ཀྱི་འཕོངས་ཀྱི་ནི། །ཕྱི་མའི་ཆ་ལ་ནི་ཏམྦ། །མ་ནི་ཛགྷ་ནཾ་མདུན་ཆ། །ཀུ་པ་ཀཽ་ཏུ་ནི་ཏམྲ་བློ་དྭ་ཡ་ཧཱི་ནེ་ཀུ་ཧཱུཾ་དུ་རེ། །སྡི་ཡཾ་སྤྱི་ཙོ་ཀ་ཊི་པྲ་ཐཱ་བུ་ནི་ཏམྦ། །ཀུ་ཧཱུྃ་དུ་རེ་གནས་གཉིས་དམན། །མོ་ནི་རྐུབ་ཙོས་ཉིད་དང་ནི། །ཀ་ཊི་དང་ནི་པྲ་ཐཱའོ། །པ་སྦོ་བ་མཱ་ཎ་ཡོ། །བྷ་ཧཾ་ཡོ་ནིརྡ་ཡོsཤིསྣོ་མེ་དྲོ་མེ་ཧ་ན་ཤེ་ཕ་མོ་མཚན་འཆད་པར་འགྱུར་བའོ། །བྷ་གཾ་སྐྱེས་གནས་གཉིས་དག་གོ། །རླིག་པ་སྒྲོ་བ་མེ་ཧ་ན། །ཤེ་ཕ་སཱི༑། མུདྡྷོཉཎྜ་ཀོ་ཥོ་བྲྀ་ཥ་ཎདྲྀཏྟཱ་པཾ་ཤཱ་དྷ་རེ་ཏྲི་ཀཾ། །པི་ཙིཎྜ་ཀུཀྟི་ཛ་སཱི་དང་རླིག་སྒོང་དང༌། །ཤུབས་དང་སྦུབས་སོ་རྒྱབ་གཞུང་དང༌། །དྲུག་མདོ་བ་དང་གསུས་པ་དང༌། །ཐ་རོ་ད་ར་ཏུནྡོ་སྟ་ནོ་ཀུ་ཅོ། །ཙུ་ཙུ་ཀརྟུ་ཀུ་ཙཱ་གྲོ་སྙནྣ་ནཱ་ཀྲོ་ཌ་བྷུ་ཛཱནྟི་རཾ། །ཨུ་རོ་དཀུ་དང་ལྟོ་བ་བྲང་ཏུནྡ། །ནུ་མ་ནུ་རྡོག་ནུ་མའི་རྩ། །ན་ནཱ་ཀྲོ་ཌི་ལག་པའི་བར། །བྲང་དང་ནང་བཏྠཾ་ཙ་བ་ཀཉྩ་པྲཥྛནྟུ་ཙ་ར་མནྟ་ནོ། །སྐནྡྷོ་བྷུ་ཛ་སི་རོཉདྷོ་མི་ས་ཚོམས་བྱང་ཁོག་སྟོང་། །་། ཕྲག་པ་དང་ནི་ལག་མགོ་དང་། །།དཔུང་པ་རྣམས་ནི་མོ་མིན་ནོ། །དེ་ཡི་ནི་ཏ་སྤྱི་བ་ཛ་རུ་ཎཱི། །བཱ་ཧུ་མཱུ་ལེ་ཨུ་བྷོ་ཀཀྐོ་པཱནཱ་མ་སི་ཏ་ཡོ་ར་དྷ། །མ་མཚམས་ནི་ཛ་ཏུ་ཎྜི། །ལག་རྩ་གཉིས་པོ་མཆན་ཁུང་ངོ༌། །རྩིབ་ལོགས་མོ་མིན་དེ་དག་འོག། དབུས་རྒྱ་མཉྩ་བ་ལཱགྷཝཱ་མ་དྷོཉསྟི་དྲོ་པ་རོ་དྭ་ཡོ། །བྷུ་ཛ་བཱ་ཧུ་པྲ་བེཊོ་དང་རྐེད་པ་བར་མ་ནི། །མོ་མིན་གཉིས་པོའི་ཕ་རོལ་གཉིས། །སྤྱོད་བྱེད་འབབ་བྱེད་འཇུག་བྱེད་ལག། དོསཥ་ཕོ་ཎིཅུ་ཀུལླི་རཿ། །ཨ་སྱ་པ་རི་པྲ་གཎྞསཱཏྨ་ཀོཥྛ་ཏ་སྱ་ཙཱ་སྱ་དྷ། །མ་ཎི་བར་ཚིགས་དང་ནི་གྲུ་མོའོ། །དེ་ཡི་སྟེང་དུ་བར་ལྷུ་ཡིན། །ལག་ངར་དེ་ཡི་འོག་ཏུའོ། །མཁྲིག་མ་ལག་བནྡཱ་དཱ་ཀ་ནིཏྟཾ་ཀ་ར་སྱ་ཀ་ར་བྷོ་བ་ཧི། །པཉྩ་ཤཱ་ཁཱིཤ་ཡཔཱ་ཎིསྟརྫ་ནི་སྙཏྠ་པའི་མཐེ་ཆུང་བར། །ཀ་ར་བྷོ་དང་ལག་བཞག་གོ། །བྱེད་པོ་ལག་པ་ཡལ་ག་ལྔ། །སྡིགས་མཛུབ་རབ་དེ་ཤི་ནཱི། །ༀ་གུ་ལྱཱཀ་ར་སཱ་ཋཱyཔཾ་སྱངྒཱཥྛཔྲ་དེ་ཤི་ནཱི། །མ་རྒྱ་མཱཉནཱ་མི་ཀཱ་ཏུ་སྟོན་བྱེད་དོ། །སོར་མོ་ལག་པའི་ཡལ་གའོ། །།སྐྱེས་བུ་མཐེ་བོང་སྟོན་བྱེད་དང༌། །གུང་མོ་མིང་མེད་།
