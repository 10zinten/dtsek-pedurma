།།ཀོ་ཀཀྐརྒྱས་ཀ་ལ་ཀ་ཎ། །བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི། ,་པ་ཚོལ་བ། ཀ་ཀ་ནས་ཁྲོ་ཟ་ཅེས་གནས་པ་ལ། ༈ཏ་ཐེ། ཐ་ལ་ཏའོ། །རྣམ་དབྱེ་ཐ་ཡིག་དང་པོ་ཅན་ཕ་རོལ་ལ་བྲུ་འགྱུར་བའི་:ཨཱཧྲའི་ཧྲ་ཡིག་ནི་ཏ་ཡིག་ཏུ་འགྱུར་ཏེ། ཧྲ་ཏྲར་བསྒྱུར་བས། ཨཏྠ༑གཉིས་ཚིག་ལ། ཐསི། ཨ་ཐུས་སུ་བསྒྱུར་བས། ཨཱ་ཧ་ཐུརྣམས་སོ། །གལ་བ། ༈་གོང་མ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས་ཆོག་འབྱུང་ལིང་གི་པ་རསྨི་་པ་ད་རྣམས་ལ། བྷཱུ་ཡཱཏྲ། བྲུ་ཡཱ་ཏཱ། བྷཱུ་ཡུལ་བྷཱུ་ཡཱབྷཱུ་ཡཱིཏྭཾ། བྷུ་ཡཱ་ཏ། བྲུ་ཡཾ། བྷཱུ་ཡཱ་བ། བྷཱུ་ཡཱ་མ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། བྲུ་བཱི་ཏ། བྲུ་བཱི་ཡཱ་ཏྭཾ། བྲུ་བཱི་རན། བྲུ་བཱི་ཐི་བྲུ་བཱི་ཡཱ་ཤྩ། བྲུ་བཱི་ཧཱུཾ།  ོ་བྲུ་ཤྲཱི་ཡ། བྲུ་བཱི་བ་ཧི། བྲུ་ཤྲཱི་མ་ཧི། ་ཤིས་བྱེད་ལོཊའི་པ་རསྨི་པ་ད་རྣམས་ལ། །བྲ་བཱི་ཏུ། བྷུ་ཏཱཏྲ། བྲུ་ཏྭཾ། བྲུ་བནྟུ། བྷཱུ་ཧི། བྷུ་ཏཱཏྲ། བྷུ་ཏཾ། བྷུ་ཏ། བྲ་བཱ་ནི༑འབྲ་བཱ་བ། བྲ་བཱ་མ། ས་འཁ༈ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། བྷཱུ་ཏཱཾ། ཌབྲུ་བཱ་ཏཱ། བྲུ་བ་ཏཱ། བྷཱུ་ཏེ། བྲུ་བཱ་ཐཱ། བྷུ་བྷཾ། བྲ་བོ། བྲ་བཱ་བ་ཧི། བྲ་བཱ་མ་ཧི། ས་འབད་ནད༈དེ་རིང་མ་ཡིན་པ་ལང་གི་པ་རསྨི་པ་ད་རྣམས་ལ། ཨ་བྲ་བཱིཏྲ། ཨ་བྷུ་ཏཱཾ། ཨ་བྲུ་བན། ཨ་བྲ་བཱིཨ་བྷུ་ཏཱ། ཨ་བྷཱུ་ཏ། ཨ་བྲ་ཁཾ། ཨ་བྲུ་བ། ཨ་བྲུ་མ། ''དེ་རྒྱུ་བ་མཚོ་ལ་གྲོགས་བྱེད་དུ་p:པལོ་ཀ་བ། ཁྱི་སྐྱེས༈ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ཨ་བུ་ཏ། ཨ་བུ་བཱ་ཏཱཾ། ཨ་བྲུ་བ་ཏ། ཨ་བྲུ་ཐཱཨ་བུ་བཱ་ཐཱ། ཨ་བཱ་ནཱཾ། ཨ་བྲུ་བི། ཨ་བྲུ་བ་ཧི། ཨ་བྲུ་མ་ེཧི། དེ་རྣམས་ནི་ཨཔྲ་བི་ཀཱ་ར་ཎ་འཇིག་པ་རྣམས་ཀྱི་ཆོ་ག་དཔེར་མཚོན་ཙམ་བརྗོད་པ་ཡིན་ལ། །བྱིངས་ཧུ་སོགས་ལ་ནི་ཨཔྲ་བི་ཀཱ་ར་ཎ་འཇིག་པར་མ་ཟད་ཁྱད་པར་་གཞན་ཡང་ཡོད་དེ། ༈ཧྭ་དེདྭིཤྩ། ཧུ་ལ་སོགས་པའི་གཉིས་ཀྱང་ངོ༌། །རྣམ་དབྱེ་ཏིཔྲ་ལ་སོགས་པ་ཕ་རོལ་ལ་ཡོད་ཅིང་ཨཔྲ་བི་ཀཱ་ར་ཎ་འཇིག་པ་ལས་이—།
