རྒྱུག་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེབས་ཌསློཎ་ཀྱིར་ཛནྣཱཀྐི་ཤི་ཁ་བ་ར་བཱ་བྷཱི་ཀ་བཱི་ཏ་ནཾ། །རཀྟ་སངྒཱ་ཙ་པི་ཤུ་ན་དྷཱི་ར་རྩེ་མོ་མཆོག་ལྡན་དང་། །བ་དྷཱི་ཀ་དང་སེར་པོའོ། །དམར་པོ་ཀུན་འཁུམས་བི་ཤུ་ན། །བརྟན་པ་ལོ་ཧི་ཏཱ་ཙནྡ་ནཾ། །ལཱཀྐཱ་རཱ་ཛ་ཏུ་ཀླུ་བེ་ཡ་བོ་ལ་ལྟོ་དྲུ་མཱ་མ་ཡ། །ལ་བངྒ་ཙནྡན་དམར་པོའོ། །།རྒྱ་སྐྱེགས་སྲུང་སྐྱེ་མ་ནིང་སྟེ། །ཡ་བ་རྒྱ་ཚོས་ཤིང་གི་ནད། །ལི་ཤི་ལྷ་ཡི་མེ་རྗེ་བ་ཀུ་སུ་མཾ་ཤྲཱི་སཾ་ཛྷ་མ་ཐ་ཛ་ཡ་ཀཾ། །ཀཱ་ལཱི་པ་ཀཉྩ་ཀཱ་ལཱ་ནུ་སཱརླཉྩ་ཐ་ཏོག་དཔལ། །མིང་ཅན་དེ་ནས་ཛཱ་ཡ་ཀཾ། །ཀ་ལཱི་པ་ཀཾ་དུས་རྗེས་འབྲང་། །དེ་ནས་ནུས་ལྡན་རྒྱུད་ས་མཱརྦ་ཀཾ། །ཁཾ་ཤི་ཀཱཉགུ་རུ་རཱ་ཛཱརྟ་ལོ་ཧ་ཀྲི་མི་ཛ་ཛོདྨ་ཀཾ། །ཀཱ་ལཱ་གུཊྚ་གུ་རུ་ཅན་དང༌། །ཨ་ག་རུ་དང་རྒྱལ་མཉེས་བྱེད། །དམར་པོ་སྲིན་སྐྱེས་ཛོ་ག་ཀཾ། །ཨ་ག་རུ་ནག་ཨ་ག་སྙཏྟཧྥངྒ་ལྟ་མལླི་གནྟི་ཡཏ། །ཡཀྵ་དྷུ་པ༔སརྫ་ར་སོ་རཱ་ལ་མརྦ་ར་ཤཱ་བ་རུ། །བཀྲ་ཤིས་ཕྲེང་བའི་དྲི་དང་ནི། །མཆོད་སྦྱིན་སྤོས་དང་པྲ་རྩི་ཤིང་། །སྤོས་དཀར་ཁུ་བ་ཀུན་ལྡན་ཡི༑། བ་ཧུ་རླུ་པོཉཔྱ་ཐ་པྲ་ཀ་དྷུ་བ་ཀྲྀ་ཏྲི་མ་དྷུ་བ་ཀོ། །།ཏུ་རུ་ཀཿཔིཎྜ་ཀསི་བློ་དང༌། །གཟུགས་མང་པོ་དང་དེ་ནས་ནི། །བྲི་ཀ་དྷུ་བ་སྦྱར་སྤོས་སོ། །ཏུ་རུསྐ་དང་པིཎྜ་ཀ། སི་ཧྥ་པཱ་བ་ནོ་པྱ་ཐ་པཱ་པ་ཤ། །ཤྲཱི་བ་སོ་ཐཾ་ཀ་དྷུ་པོ་ཤྲཱི་པིཋ་ཤ་ར་པཱ་བ་ན་ཡང་ངོ༌། །དེ་ནས་པཱ་ཡ་ཤ་དང་ནི། །དཔལ་གོས་ཐཾ་ཀ་དྷུ་པ་དང༌། །ཤྲཱི་པིཥྛ་དང་ལ་དྲ་པོ། །མྲྀ་ག་ནཱ་བྷིནྡཾ་ག་མ་དཀསཱུ་རཱི་ཙཱ་ཐ་ཀོ་ལ་ཀཾ། །ཀཀྐོ་ལ་ཀཾ་ཀོ་ཥ་ཐང་ཆུའོ། །རི་དྭགས་ལྟེ་བ་རི་དྭགས་ཆང༌། །གླ་རྩི་དེ་ནས་ཀོ་ལ་ཀཾ། །ཀཀྐོ་ལ་དང་སྦུབས་ཀྱི་ཕ་ལ་མ་ཐ་ཀརྩྭ་ར་མ་ཏྲི་ཡཾ། །གྷ་ན་སཱ་རབྷཾ་དྲ་སཾ་ཛྷསི་ཏཱ་བྷཱ་ཧི་མ་བཱ་ལུ་ཀཱ། །འབྲས། །དེ་ནས་ག་པུར་མོ་མིན་ནོ། །གྷ་ནའི་སྙིང་པོ་ཟླ་བའི་མིང༌། །འོད་དཀར་ཁ་བའི་བྱེ་མའོ། །གནྟཱ་སཱ་རོ་མ་ལ་ཡ་ཛེ་བྷ་དྲ་ཐིཉྩ་ནྡ་ནོ་ཞཔྲི་ཡཾ། །ཏེ་ལ་པཎྜི་ཀ་གོ་ཤྲིནྟོ་ཧ་རི་དྲི་སྙིང་མ་ལ་ཡ་སྐྱེས་དང༌། །བཟང་དཔལ་ཙནྡན་མོ་མིན་ནོ། །ཏེ་ལ་པཎྜི་ས་མཆོག་དང༌། །ཧ་རི་ཉན་ན་མཏྲི་ཡཾ། །ཏི་ལ་པཎྜི་ཏུ་པ་ཏཱངྒཾ་རཉྫ་ནཾ་རཀྟ་ཙནྡ་ནཾ། །ཀུ་ཙནྡ་ནཉྩཱ་ཐ་ཛ་ཙནྡན་མོ་མ་ཡིན། །ཏི་ལ་པཎྜི་པ་ཏཱངྒ་། །ཚོས་བྱེད་ཙནྡན་དམར་པོའོ། །ཙནྡན་ངན་པ་དེ་ནས་ནི། །polokཏཱ།
