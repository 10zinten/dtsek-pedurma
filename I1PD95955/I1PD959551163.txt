བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིངྷ་ཡ་ཤུདྡྷ་བི་རཱརྨ་སོ་ཛ་གོ། །རྣོ་ཡཱངྒཁ་ལུ་པ་ཎ་བཀྟ་ཏ། །སྙརྨ་ཡུ་ར་མ་ས་ཛ་ག་དག་པའི་སྣང་བར་ཤེས་པར་བྱ། །མ་ན་ཡ་ལས་ལྕི་ངེས་འཁར་རྔར་བཤད། །ར་ཛ་ར་དང་ག་སཱ་རི་ཎཱི་ར་ཛོ་རྒོ། །རུ་བ་ཏཱི་སཱ་ཡ་ཏྲ་བྷ་མོ་སྒོ། ་། ཛྷེ་ཡཱ་མཏྟཱ་མ་བྷ་ས་ག་ནི་རྨ་བྱའི་འགྲོས་ཡིན། གང་ལ་བྷ་མ་ས་ག་དེ་ནི་གསེར་ལྡན། མ་བྷ་ས་གས་བསྐྱེད་པ་མྱོས་མར་ཤེས་པར་པྲི་ཏྟཱ། །ཏཱཛྫོ་ག་གི་ཏཱི་ཡ་མུ་པ་སྦི་ཏཱ་སྟེ་ཏྲ། །སྭཱ་དིནྡ་བཛཱ་ཡ་དི་ཏོ་ཛ་གོ་ག། །བྱ། །ཏ་ལས་ཛ་གཉིས་ག་ག་ཞེས་འདི་ཉེར་འཁོད་འགྱུར། །གལ་ཏེ་དབང་པོ་རྡོ་རྗེ་ཏ་གཉིས་ཛ་ལྕི་གཉིས། །ཨུ་པེནྡ་བཛྲ་ཧི་ཛ་ཏོ་ཛ་གོ་ག། །ཨ་ནནྟ་རོདྡྷཱ་བི་ཏ་ལཀྵ་ཎཱ་དྱེ། །པཱ་དོ་ཛ་ཏ་ཛ་ལྕི་གཉིས་ནི་ཉེ་དབང་རྡོ་རྗེ་སྟེ། །དེ་མ་ཐག་ཏུ་བརྗོད་ཟིན་ལ་སོགས་མཚན་ཉིད་དག། སྣ་ཚོགས་བྷ་བེ་ཏྭཾ་བི་བི་དྷིཇི་ཀལྤ། །།ཡཱ་སཱ་མུ་བྷོ་ཤ་བྱ་ཡ་ཏི་པྲ་པཉྙོ། ་། སྤྱི་རྣམ་རྟོག་རྣམས་ཀྱིས་རྐང་པ་དག་ཏུ་འགྱུར། །གང་དེ་གཉིས་ཀ་མཉན་འོས་ངལ་བསོའི་སྤྲོས་པ་ཅན། །དེ་ཏཱསྨྲ་ཏཱི་ཤི་རུ་པ་ཛཱ་ཏ་ཡསྣ། །དོ་དྷ་ཀ་མིཙྪ་ཏི་བྷ་ཏྲི་ཏི་ཡཱརྒོ། །ཌ། རྣམས་ཉེར་བསྐྱེད་རྣམས་སུ་དྲན་པའི་དབང་པོས་བཤད། །བྷ་གསུམ་ལས་ནི་ག་དག་འཇོ་བ་པོར་འདོད་དོ། །མོ་བཱ་རོ་བཱ་ཤཱ་ལི་ནཱི་བེ་ད་ལོ་ཀི། །མོ་བཱསྲོ་བཱ་ཡ་དི་བཱ་ཏོརྨི་མཱ་ལཱ། ་། མོ་བཱ་ནོ་མ་བར་གཉིས་པ་རིག་འཇིག་རྟེན་ཤཱ་ལི་ནི། །གལ་ཏེ་མ་བ་ས་ར་བ་ནི་ཆུ་རླབས་ཕྲེང་། །མ་ཝ་ན་དག་ཝ་བཱ་བྷ་མ་ར་བི་ལ་སི་ཏཱ། །རཱཏྤ་རེ་ན་ར་ལ་གཱ་ར་ཐོདྷཱ་ཏཱ། །སཱ་ག་ཏཱ་ར་ན་བྷ་གི་རྒུ་རུ་ནི་བུང་བ་རོལ་པའོ། །ར་ལ་ས་ཕ་རོལ་ན་ར་ལ་ག་ཤིང་རྟ་དྲེགས། །ར་ན་བྷ་ག་རྣམས་དང་ལྕི་བས་ལེགས་ཎཱ་ཙ། །ན་ན་ས་ག་གི་ཏི་ཙ་བྷ་བེབྲྀཏྟཱ། །ན་ན་ར་ལ་གི་ཏི་ཙི་ཝ་བྷ་དྲི་ཀཱ། །འོངས་སོ། །ན་ན་ས་ག་ག་ཞེས་བྱ་བ་ཟླུམ་པོར་འགྱུར། །ན་ན་ར་ལ་ག་ཞེས་བྱ་བ་བཟང་ལྡན་ཉིད། །ར་བྱརླུ་དཱི་རི་ཏཱ་ར་ཛོ་ར་ལོ་གུ་རུ། །ཛ་སོ་ཙ་ཏ་ག་ག་ཕྱོགས་པསྦི་ཏི་སཱུ་ཏ། །ཙནྡ་བཏྨ་ཛ་ར་ལ་དང་ནི་ལྕི་བ་དཀར་མོར་བརྗོད། །ཛ་ས་ཏ་དང་ག་ག་རྣམས་ནི་ཉེར་གནས་འགྱུར། །ར་ན་བྷ་སས་ཙ་ར་ཡཎྜི་ར་ན་བྷ་སི། །ཨུ་ཤནྟི་པཾ་ཤ་སྦྲ་མི་དཾ་ཛ་ཏོ་ཛ་རོ། ན་། ས་དིནྡྲ་བོ་རྒྱུ་ཞིང་འཁོར་བ་ཟླ་བའི་ལམ། །ཛ་ཏ་ཛ་ར་འདི་ནི་རིག་གནས་ཉིད་དུ་འདོད། །གང་ལ་ཏ་གཉིས་ཛ་ར་—།
