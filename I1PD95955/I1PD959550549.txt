འི་ལོ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེའིཎཏི་ཀ་ལས་ཛྫ་ནི་རཱ་མ་ཡོ། །ཨཱཏྨ་བཱ་ན་ནུ་པེ་ཏོཆ་ད་པུ་ཙྩཾ་ཏུ་ཙཱརྦ་པི། །རཱུ་པྱཔྲ་མེད་ཡོན་ཏན་ཅན། །བདག་ལྡན་ནོར་བྲལ་ཨརྒྱུའོ། །པུ་རྒྱུ་ཡིད་དུ་འོང་བ་ལའང༌། །རཱུ་བྱ་དགའ་ཤསུ་རཱུ་པེ་པི་བ་དཱ་ནྡྷོ་བལྷ་བཱ་ག་པི། །རྩྭ་ཡ་པི་མ་དྷསོ་མྱྀ་ཏུ་སུནྡ་རེ་ཤོ་མ་དང་གཟུགས་ལའོ། །།བ་དྭ་ཅོ་ནི་ངག་མཉེན་པ། །མ་རྒྱ་བར་མ་རིགས་པ་ལ། །སོ་མྱི་གདུ་བུ་ཟླ་བ་དེ་བ་ཏི། །སརྦ་ཛྷ་བྷི་ཥ་ཛོ་བི་དྷུ་བཱཏྨ་ཀཱ་མཉྩ་ཧྲིཙྪ་ཡོ། །ཕ་ལ་ཀ་ལྱཱ་ཎ་ཡོདྷ་བྱི་ལྷ། །བི་དྱ་ཀུན་མཁྱེན་སྨན་པ་ལ། །བདག་དང་འདོད་པ་བྲཱཙྪ་ཡོ། །བྷ་བྱ་དགེ་དང་འབྲས་བུ་ལ། །ཡོགྱ་སཾ་པྲ་ཏི་ཀེ་ཏྲི་བུ། །ལ་ལ་མོ་ཊ་་་་་་ཀྱི་བཡོ་གྱ་ཤིན་ཏུ་རིགས་པ་ལ། །ཡ་མཐའོ། །་་་་། ཚོང་རྒྱུ་། ་:ཀ་ཏེནནི་བ་ཧཱ་བ་ས་རོ་བཱ་རོ་སཾ་སྟ་རོ་པྲ་སྟ་རཱ་དྷཱ་རོ། །གུ་རུ་གིཥྛ་ཏི་པི་ཏཱ་ཏྱཾ་དཱ་པ་ཚོགས་དང་རེ་མོས་བཱ་རའོ། །སཾ་སྟ་རོ་ནི་མཆོད་སྦྱིན་ལམ། །གུ་རུ་ཕུར་བུ་ཕ་ལ་སོགས། །ད་པ་རོ་ཡུ་ག་སཾ་ཤ་ཡོ། །པྲ་ཀཱ་རོ་བྷེ་དཱ་སཱ་དྲི་བྱེ་ཨཱ་ཀཱ་རཱ་་བི་གི་ཏ་ཀྲྀ་ཏཱི། །ཀཽ་ཤཱ་རཱུ་རོ་ནི་དུས་ཐེ་ཚོམ། དུ། པྲ་ཀཱ་ར་ནི་དབྱེ་བ་དཔེ། །ཨཱ་ཀཱ་ར་ནི་ལུས་ཚུལ་བྱད། །ཀི་ཤཱ་རུ་ནི་གྲ་དྷཱ་ནྱ་ཤུ་ཀེ་ཥཱ་མ་རུ་དྷ་ནཱ་དྷ་རོ། །ཨ་དྲ་ཡོ་དྲུ་མ་ཤི་ལཱཎཱསི་སྟེ་ནཱ་བྷོ་པ་ཡོ་ེདྷ་རོ། །མ་མདའ་སྙེ་མ། །མ་རུ་མྱ་ངམ་ཐང་དང་རི། །ཨ་དྲ་ཤིང་དང་རི་ཉི་མ། །པ་ཡོ་དྷ་ར་ནུ་མ་སྤྲིན། །དྷཱནྟཱ་རི་དཱ་ཎ་བཱ་བྲི་ཏཱ་བཱ་ལི་ཧནྟཾ་ཤ་བཀ་རཱ། །པྲ་དོ་རཱ་བྷངྒ་ནཱ་རཱི་རུབྷཱ་ཎཱ་དྷཱནྟཱ་རི་ནི་ལྷ་མིན་འཁོར། །དཔྱ་སྡུད་ལག་པ་འོད་ཀ་རཱ། །པྲ་ད་ར་ནི་འཇོམས་བྱེད་དང་། །བུད་མེད་རྐེད་ཨ་སྭཱཀ་ཙཱ་ཨ་པི། །ཨ་ཛཱ་ཏ་ཐེངྒེ་གོཀཱ་ལེ་པྱ་སྨ་ཐུཊཱ་ཏུ་བ་རོ། །ནད་དང་ནི་མདའ། །ཨ་པྲ་སྐྲ་དང་ཟུར་ཡིན་ནོ། །རྭ་མེད་བ་ལང་དུས་དང་ནི། །སྨ་ར་མེད་པ་ཏུ་པ་རོ། །སུ་བརྩེ་བི་རཱཔ་རི་ཀ་རཿཔཎཾ་ཀ་པ་རི་བཱ་ར་ཡོ། །མུཀྟ་ཤུདྡྷོ་ཏཱ་རཿསཱུཏྟཱ་རོ་བཱ་ཡོ་རཱ་ནི་ནོར་ཏེ་གསེར་ལ་ཡང་། །པ་རི་ཀ་ར་སྐྱིལ་ཀྲུང་འཁོར། །མུ་ཏིག་དག་པ་ཏཱ་ར་ཡིན། །ཁྲ་བོ་རླུང་གསུམ་པ་ཏྲི་ཥཱ། །ཀཅུ་རེཞཐ་བྲ་ཏི་ཙཱ་ཛི་སམྤི་དཱ་པཏྨ་སཾ་ག་རཿ། །།བེ་ད་བྷེ་དེ་ཀཅུ་ར། །དེ་ནས་དམ་འཆའ་གཡུལ་དང་ནི། །རིག་པ་གནོན་པ་སཾ་ག་ར། །རིག་བྱེད་བྱེ་བ་དག་དང་ནི། །།
