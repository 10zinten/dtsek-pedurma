རྨ་ཀམི་ཞིག་ཨུ་ལ་ནི་འབར་བ་སྟེ། ལྟ་ན་སྡུག་པའོ། །དེ་དག་སྒེག་པའི་བྱ་བའི་བདག་་་་ཉིད་བརྗོད་པའོ། །དེ་སྐད་དུ་སློབ་དཔོན་བྱཱ་ཊིས། གཙང་དང་འབར་བ་སྒེག་པ་སྟེ། །དཔའ་བ་སྤྲོ་བ་འཕེལ་བ་ཡིན། །ཞེས་སྟེ། སྦྱིན་པ་དང་ཆོས་གཡུལ་གྱི་འཐབ་མོ་ལ་སྤྲོ་བར་བྱེད་པ་ནི། དཔའ་བའི་ཉམས་སོ། །ཀཱ་རུ་ཅུ་དང་ཀ་རུ་ཎ་སྙིང་རྗེ་སྟེ། །སྔ་མ་ནི་གླུ་ཎའི་རྐྱེན་ཅན་ནོ། །གཽཎཱ་བརྩེ་བ། །བློ་འཁྲུད་པ་་་ལའོ། །ར་སྒྲ་ལ་སོགས་པ་རྣམས་སོ། །ཞེས་པས་ན་ཀ་རྐྱེན་ནོ། །ཀྲི་པཱ་དང༌། ད་ཡཱ་དང་། ཨ་ནུ་ཀམྤ་རྣམས་ཀྱང་བརྩེ་བའོ། །དེ་ནས་ཧ་ས་དང་། །ཧ་་་གསུམ་བཞད་གད་གད་མོ་ཞེས་བྱའོ། །བཱི་བྷཏྠ་དང་། བི་ཀྲྀ་ཏ་རྣམས་དང་་་་འབྱུང་དང་མི་སྡུག་པའོ། །བསྨི་ཡ་ངོ་མཚར་རོ། །ཨ་དྷུ་ཏ་རྨད་བྱུང་ངོ༌། །ཨཱ་རྱ་མཚར་ཆེའོ། །ཙི་དྲི་ཕུལ་ཕྱུང་ངོ༌། །བྷེར་བོ་འཇིག་བྱེད་དང་། དཱ་རུ་ཎཾ་མི་བཟད་པ་དང་། བྷི་ཥ་ཎཾ་འཇིགས་པར་བྱེད་པ་དང་། བྷི་མ་དང་བྷིཥྛ་་་འཇིག་པ་དང་། གྷོ་རོ་དྲག་པོ་འམ་སྐྲག་བྱེད་དང་། བྷ་ཡ་ན་ཀཾ་འཇིག་རུང་་་དང་། བྷ་ཡཾ་ཀ་ར་དང་། །པྲ་ཏི་བྷ་ཡཾ། འཇིགས་བྱེད་དང༌། སོ་སོར་་་འཇིགས་སོ། །་་་་འ། །ཡུ་བ་རྒྱུ་བ་དང་། ཁྲིག་པ་ཉིད་དུ་ག་ལ་འབྲེལ་སྒྲིགདེ་ནས་བསམ་པའི་མིང་ལ། ཨིཙྪ་དང་། ཀཱཀླུ་དང་སྤྱི་ཧ་དང༌། དྲིཊ་དང༌། བྷཉྩ་དང་། ལིསྶ་རྣམས་ནི། འདོད་པའི་བསམ་པའི་མིང་གི་རྣམ་གྲངས་ཏེ། རེ་བ་དང་འདོད་པ་དང་བསམ་པ་ལ་སོགས་པ་ཅི་རིགས་པའོ། །མ་ནོར་ཐ་་ནི་ཡིད་ཀྱི་ཤིང་རྟ་བའམ་པའམ། ཀ་མ་འདོད་པ་དང༌། ཨཱ་བྷི་ལཱ་ཥ་མངོན་འདོད་དང༌། ཏནྟ་འདོད་པ་དང་། མ་ཧཱ་ན་ལ་ས་འདོད་ཆེན་པོ། །།།དགའ་ལ་ཀན་གཉིད་ཀྱི་མིང་ལ། ནི་དྲ་གཉིད་དང་། ཤ་ཡ་ན་ཉལ་བ་དང་། སྡེ་པ་དང་སྭ་སྣ་གཉིད་ལོག་པ་འམ་རྨི་ལམ་དང་། སམྦས། ལེགས་བབ་ཅེས་བྱའོ། །་ཏེ་སྒྲུབ། །བ་—།
