བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་མིག་མ་དག་དང༌། །ལང་ཚོ་ཅན་བཞིན་དགའ་དང་གྲགས་པ་ཐོབ་པར་གྱུར། །:：སྙན་ངག་མེ་ལོང་ལས་ལེའུ་སྟེ་གསུམ་པའོ། །་ཎ་ཁ། །།ལྟ་བ་བཀའ་བློན་པ། ་པོ་ཏ། ཅ་བ་སློབ་དཔོན་དབྱུག་པ་ཅན་གྱིས་མཛད་:པའི་སྙན་ངག་མེ་ལོང་གི་གཞུང་་་འདི་ཕུལ་དུ་བྱུང་བའི་ཡོན་ཏན་དཔག་ཏུ་མེད་པས་སྤྲས་པ་བླ་མ་དམ་པ་ཆོས་ཀྱི་རྒྱལ་པོ་རིན་པོ་ཆེ་བ་དང་། དཔོན་ཆེན་ཤཱཀྱ་བཟང་པོའི་བཀའ་ལུང་གིས་རྒྱ་གར་་་གྱི་མཁན་པོ་སྙན་ངག་མཁན་ཆེན་པོ་བི་ལགཱི་ཀ་ར་དང༌། བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཤོང་སྟོན་གྱིས་དཔལ་ས་སྐྱའི་གཙུག་ལག་ཁང་ཆེན་པོར་:བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། དེ་རྗེས་དཔང་ལོ་ཙཱ་བ་བློ་གྲོས་བརྟན་པས་གོ་དཀའ་བའི་ཆ་ཤས་སུ་གྱུར་པ་རྣམས་དང་དོན་ཆེ་བ་རྣམས་ལ་འགྱུར་བཅོས་བགྱིས་པ་ལ་ཕྱིས་སྙེ་་ཐང་བློ་བརྟན་བཞི་པས་ལེགས་པར་འགྱུར་བཅོས་བགྱིས་པ་ལགས་སོ། །ཁ་བཏབ་པཞེར། །ཟླ་འཚོ་ཞི་ཁག་ལ་ཞི་བ་འཚོ་དང་་་་་་ར། དེའི་དབང་དེབ་ཀྱི་ཚོ་ཁག་གང་ན་འཁnsཔ་དང་། དཔལ་འབྱོར་བ་ལ་ཕྱིར་མཚན་དུ་བློ་ལ་བསྔགས་པ་དང་ད་ལྟ་ཁ་«པེ་»སྣར་»མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་དབང་ལ་༄༅། །ཚིག་གི་རྒྱུན་གྱི་བསྟན་བཅོས་སྙན་དངགས་མེ་ལོང་བཞུགས་སོ། །«པེ་»«སྣར་»ཀན«པེ་»«སྣར་འཕགས་པ། «པེ་»«སྣར་དག་གི། ས《པེ་»«སྣར་»ལྷག་མ་ནི། ཕྱི«པེ་»«སྣར་»སྒྲས་བོད་སྣང་བ་ཡི། །འཁོར་བར་གསལ་བར་མ་བྱས། །(«པེ་»«སྣར་»གོ་གོ་འདོད་པ་འཇོ་བར་དྲན། «པེ་»«སྣར་བ། »«པེ་»སྣར་བ་པོ་ནི་གོ་ཉིད་རྗོད། «པེ་»«སྣར་རྣམ་པར་བསྐྱེད་ལ་མངོན་བལྟས། སུ«པེ་»«སྣར་དག་གིས་ནི་སྙན་དངགས་ཀྱི། །@«པེ་»«སྣར་རྣམས། «པེ་»«སྣར་ལ་ཡང་། ར«པེ་»«སྣར་རྒལ། /«པེ་»«སྣར་»པ་དེ་འདྲ་ཡིས། ཀྱེ«སྣར་»སྔགས། །ལ་། «པེ་»«སྣར་སླུ་བ་དང་ནི། «པེ་»«སྣར་པ་ན། །འདིར། །ཉི«པེ་དག་ཀྱང་བསྔགས། 《སྣར་»དག་ཀྱང་སྔགས། ®«པེ་»«སྣར་ལ། །«པེ་»«སྣར་»སྐྱེད། «པེ་»«སྣར་རྗོད། «པེ་»པ་དངོ«སྣར་འདྲེན། «པེ་»«སྣར་འདྲ། «པེ་»«སྣར་བརྟན། དགེ«པེ་»«སྣར་ཨཱ། རི་།
