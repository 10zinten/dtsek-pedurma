བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་ ོ་བའི་མཚོ་སྨན་ཏནྟིཀ༌ལྔ་བློ་ཀྱིས་བསྒྲུབས་པས། མྱིང་སྲོག་གཏོང་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ། མི་®ཡ་ཏེ། ་༈དྲྀ་འགྲོ་ེབ་ལའོ་ཞེས་པའི་བྱིངས་ལ། བྲཱཧྥ་ཎེ་ན་དཱ་ན་མཱ་དྲི་ཡ་ཏེ། བྲམ་ཟེ་སྦྱིན་པ་ལོང་། ཧྲཞྭ་འཕྲོག་པ་ལའོ་ཞེས་པའི་བྱིངས་ལ། ་་:ཧྲི་ཡ་ཏེ་དྷ་ན་་་ཏསྐ་རེ་ཎ། འཕྲོག་པར་བྱེད་དོ་ནོར་ཆོམ་རྐུན་གྱིས། དེའི་ཚོད། :ཁ་བ་དང་བརྒྱ་བ་སོགས་བྱེད་དུ་ཡོད། །ཡོད་ཀྱང་ད་བར་དེ་༈ཛཱཏ་ཨིར། ་་:རྀཏྲ་ཨིནྡྲའོ། ་། རྐྱེན་ཡ་ཡིག་ཕ་རོལ་ལ་བྱིངས་མཐའི་རྀཏྲ་ནི་ཨིར་དུ་འགྱུར་ཏེ། ཏཱཾ་མཆོད་པ་དང་:རྐྱལ་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ་ཡཀྲ༑རཱཏྲ་ཨིནྡྲ་དུ་བསྒྱུར་བས། ཏིར་ཡ་ཏེ་ཞེས་གནས་པ་ལ། ༈ཡེ། ཡ་ལའོ༑། རྐྱེན་ཡ་ཕ་རོལ་ལ་དབྱངས་སྔོན་མ་དེ་རིང་པོར་འགྱུར་ཏེ། །༼ཨི་རིང་་པོར་བསྒྱུར་བས། ཏཱིརླ་ཏེ། ག་ག་ན་ས་གནས་དང་། དམག་དཔོན་ལ་བ་སུ་དེ་བཞིན་དུ། །་ཙིའྲ་གསོག་པ་ལའོ་ཞེས་པའི་བྱིངས་ལ། ཨིཊྚི་ཀཱ་ལྕི་ཡ་་་ཏེ་སོ་ཕག་རྣམས་གསོག་པར་བྱེད། ཧུ་སྦྱིན་པ་དང་ཟ་བ་ལའོ་ཞེས་པའི་བྱིངས་ལི། ཧུ་ཡ་་་ཏེ་:ཨགྣི་བནྡྷ། མེ་ལ་ཟས་སྦྱིན་སྲེག་བྱེད། བདག་༈དཱ་དེ་རིེ། དཱ་སོགས་ཀྱི་ཨིའོ། །བྱིངས་ཌུ་དཱའྲ་ལ་སོགས་པའི་ཨཱ་ཡིག་ནི་རྐྱེན་ཀྲ་དང་དྲ་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨི་རུ་འགྱུར་ཏེ། །ཊུ་དརྦ་་་སྦྱིན་པ་ལའོ་ཞེས་པའི་བྱིངས་ལ་ཡཀྲ། །ཨཱ་ཨི། ཡ་ལ་འོས་རིང་པོར་བསྒྱུར་བས། དཱི་ཡ་ཏེ། སྦྱིན་པར་བྱེད། བེ་ད་བི་དེ་ཨཱ་དོ་དཱི་ཡ་ཏྭཾ་དྷ་ནཾ། རིག་བྱེད་རིག་པ་ལ་ནོར་བྱིན་ཅིག། དེ་བཞིན་དུ། ་ཌཱ་དྷཱཞྭ་འཛིན་པ་དང་གསོ་ལེབ་ལའོ་ཞེས་པའི་བྱིངས་ལ། ནི་དྷི་ཡ་ཏེ་དྷཱ་ཏྱཾ་བྷུ་མོ༄༅། གཏེར་དུ་སྦོས་ཤིག་འབྲུ་རྣམས། མཱང་ཚད་འཇལ་བ་ལའོ་ཞེས་པའི་་་བྱིངས་ལ། མཱི་ཡ་ཏེ། ཚད་འཇལ་བར་བྱེད། ད་ལ། །ན། ནི་ངོ་ཚའི་བདག་ཞེས་་ན་འ༈སནྡྷ་ཀཱ་རཱ་ཎཱ་མཱ། མཚམས་སྦྱོར་གྱི་ཡི་གེ་རྣམས་ཀྱི་ཨཱའོ། །བྱིངས་་་མཐའི་མཚམས་སྦྱོར་གྱི་ཡི་གེ་རྣམས་ནི་དངོས་པོ་དང་ལས་ཀྱི་དོན་གྱི་ཨཱཏྨ་ནེ་པ་ད་—།
