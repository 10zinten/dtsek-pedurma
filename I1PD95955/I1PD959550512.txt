།རབི༑འ་ཇི་། ཨ་དྷཱི་ནོ་ནི་བྷྱཱ་ཨཱ་ཡཏྟོ་སཱ་ཙྪནྡོ་ཤྲི་རྒྱ་ཀོ་པྱ་སོ། །ཁ་ལ་དབང་། །རང་དབང་མེད་དག་བདག་དབང་མིན། །།གཞན་དབང་དབང་མེད་འཛིན་པ་ཅན། །ཕྱགས་པུས་ཧུ་ཀ་རོ་དཱིགྷ་སཱུ་ཏྲཉྫི་ར་ཀྲི་ཡ། །ཛཱཡྻོཉས་མཱི་ཀྵ་ཀ་རཱི་ཀྐུདྡྷོ་མ་དང་ནི་མང་པོས་བྱེད། །ཤིན་ཏུ་ཐོགས་དང་ཡུན་རིང་བྱེད། །འདྲེས་མར་སྤྱོད་དང་ཅི་གྲུབ་བྱེད། །མཎྜོ་མནྡ་ཀྲི་ཡཱ་སུ་ཡ། །ཀརྨཀྵ་མོཉལཾ་ཀརྨི་ཎཀྲྀ་ཡཱ་བཱན་ཀརྨ་སཱུ་རྱ་ཏཱ། །ས་ཀཱམྨདམན་དང་ངན་དང་བྱེད་འཁོས་ཆུང་། །ལས་ནུས་ལས་ཤོགས་ཆེ་བ་དང་། །བྱེད་མཁས་ལས་པའི་ངང་ཀརྨ་ཤཱི་ལོ་ཡ་ཀརྨ་ཤུ་རསྟུ་ཀརྨ་ཋ། །བྷ་ར་ཏྱཱ་བྷུ་ཀརྨ་ཀ་རཀརྨ་ཀཱ་རསྟུ་ཏཌི་ཡ། །ཚུལ་ཅན། །ལས་ལ་དཔའ་དང་ལས་མགོ་ཐོན། །གསོ་དང་ཟ་བའི་ལས་བྱེད་པ། །ལས་མི་དེ་ཡི་བྱེད་པ་ཅན། །ཨ་པ་སྣ་ཏོ་མྲྀ་ཏ་སྣ་ཏ་ཨཱ་མི་ཏཱ་ཤཱི་ཏུ་ཤོཥ་ལ། །བུ་བྷུཀྟི་ཏ་:ས་དྷི་ཏོ་ཛིཧཱུཾ་སྐྱེ་ཁྲུས་དང་ནི་ཤི་ཁྲུས་སོ། །ཀྲུམས་དང་ཤ་ཡ་ཟས་དང་ནི། །།ལྟོགས་དང་བཀྲེས་དང་སུ་ར་ཤ་ནཱ་ཡི་ཏ། །པ་རཱནྣཔ་ར་པིཎྜ་དོ་བྷཀ་ཀོ་བྷསྨ་རོཉརྨ་རཿ། །ཨ་ཛུ་ནབྲུ་བ་ཚ། །།ཟས་འདོད་གཞན་ཟས་ཕ་རོལ་ཟས། །ལྟོ་ཆེ་གྲོད་ཆེ་དང་ག་བདེ། །སྙོམ་ལས་དང་ནི་སྐྱིད་སྣ་དོ་ད་རི་ཀོ་བི་ཛི་ཨཱི་ཥཱ་བི་བརྫི་ཏེ། །ཨུ་བྷོ་ཏཱཏྠམྦྷ་རིཀུཀྟིམྦྷ་རིསཱ་ད་ར་པཱུ་ར་ཀོ། སྐྱུར་དང་། །སྐྱིད་སྙོམ་དང་ནི་འཇས་པོ་གཉིས། །བདག་འགེངས་ལྟོ་འགེངས་རང་དྲོད་འགེངས། །སནྣཱནྟི་ནསྡུ་སབྷཱརྨ་བྷོ་ཛཱི་གྲོསྐུ་གཌྷ་ན། ཨ་། ལུདྡྷོཉབྷི་ལཱ་ཏུ་ཀ་ཀུན་ཟ་ཀུན་གྱི་འཚོ་བའོ། །།འདོད་ཅན་རྗེ་ཅན་ཞེན་ཅན་དང༌། །།མངོན་འདོད་དང་ནི་སྲེད་ཅན་སྔོཎྜ་ཀ་མོ་ལོ་ལུ་པ་ལོ་ལུ་བྷོ། ་། སཱུརྞ་དསྣུམྨ་དིཥྛ་སྭཱ་ད་བི་ནཱི་ཏ་ས་མུངྒ་ཏ། །དང༌། །འདྲ་བ་འདོད་ཆེན་ཡང་ཡང་འདོད། །མྱོས་པ་དང་ནི་ར་རོ་བ། །མ་དུལ་བ་དང་མ་ཞི་བ། །མཏྟེ་ཤོཎྜིཀྐ་ཊཀཱི་བཱཀཱ་མུ་ཀཀཱ་མི་ཏཱཉ་ནུ་ཀཿ། །ཀ་མཿཀཱ་མ་པི་ཏཱ་བྷི་རྒྱགས་པ་རྒོད་པ་གཡོ་ལྡན་རེངས། །ཆགས་པ་འདོད་ཅན་སྲིད་པ་ཅན། །རྗེ་ཅན་འདོད་ལྡན་སྲེད་ལྡན་ཀཀ་མ་ནཀཱ་མ་ནོཞབྷི་ཀ། །བི་དྷེ་ཡོ་བི་ཎ་ཡ་གྲྭ་ཧཱི་བ་ཙ་ནེསྦི་ཏ་ཨཱ་ཤ་བ། །དང༌། །མངོན་འདོད་སྒྲུབ་ཅན་དུལ་བ་འཛིན། །ཚིག་ལ་གནས་དང་ཨཱ་ཐ་བ། །དབང་དང་རེང་།
