།པའོ། ལ་། ཤ་ཀུ་ལཱ་ནྡྷ་ག་ནི། །ཤ་ཀུ་ལའི་ཕྲུ་གུ་དང་འདྲ་བས་ཤ་ཀུ་ལའི་བུའོ། །ས་ཧ་པྲ་དཏྟཾ། ཌམཆེ་བ་སྟོང་ལྡན་དང༌། པཱ་ཐི་ན། ཉ་ཟན་ཏེ། ཉ་ཏི་མི་་ཞེས་བྱ་བ་དང་། ཉ་དེ་མིད་ཅེས་ཀྱང་བྱའོ། །འདི་ལ་འགའ་ཞིག། སཱ་ཐ་་་ཞེས་ཀྱང་ཟེར་རོ། །ཨ་ལུ་པ་སྟེང་དུ་འཕར་རོ་དང་། ཤི་གུ་ཀ་བྱིས་པ་བསོད་་དང་དབྱི་བ་འདྲ་བའི་ཉའི་ཁྱད་པར་རོ། །ན་ལ་མཱིན་ནི། །སྟོབས་ལྡན་ཉ་སྟེ། །ཏ་ལ་མཱི་ན་ཞེས་ཀྱང་ཁ་ཅིག་གོ། །པོཎྜི་ནི་དྲོ་བའི་ཉ་སྟེ། །ཨུ་ཥ་སྲེག་པའོ། །ཐན་རྐྱེན་དུ་བྱས་ཏེ། རབ་ཏུ་སྲེག་པར་བྱེད་ཅེས་པས་ཚིག་ཕྲད་ལས་སྲོཎྜིའོ། །ཤ་དྷ་རི། ལོགས་ཉལ་ཏེ་ལོགས་ཀྱང་ཉལ་བར་བྱེད་པའོ། །རྒྱུ་། ་བ་། དེ་ནས་རུ་སྦལ་གྱི་མིང་ལ། ཀུརྨ། རུས་སྦལ་སྟེ། སྨད་པའི་ཤུགས་དང་ལྡན་པས་སོ། །ཀ་མཐ། རུས་སྦལ་ཏེ་འགྲོ་བ་བུལ་བས་སོ། །།ཀོཙྪ་པ་བློས་་་་འཐུང་སྟེ་རུས་སྦལ་ལོ། །ང་ཁྲུ། །ཁ་བ། །བ་དེ་ནས་ཚིགས་བཅད་ཕྱེད་ཀྱི་སྲིན་བུ་པད་མོའི་མིང་སྟེ། །རཀྟ་པཱ་ནི་ཁྲག་་འཐུང་ངོ༌། །ཛ་ལོ་ཀཱ། ཡང་ཁྲག་འཐུང་སྟེ། ཨུ་ཙ་ཚོགས་པ་ལའོ། །བྱེད་པ་ལ་ལྷ་ཉ་སྟེ། དེ་ལྟར་ཡང་། ་བརྡ་སྤྲོད་པའི་རྒྱུད་ཉུང་དུ་ཀཱ་རྒྱ་ཡ་ནའི་མདོ་་ལས་པ་དང༌། །དེ་རུ་ཛ། བི་ཤ། པྲ་ཤ་ཨུ་ཙུ་རྣམས་ལས་གྷ་ཉའོ། །ཞེས་་་བཤད་དོ། །ཛ་ལོ་ཀཱ་པ་ཡང་དེ་ཉིད་དོ། །ཁ། ན། ་་་་་དེ་ནས་ཉ་ཕྱེས་ཀྱི་མིང་ལ། ་མུཀྟི་སྨོ་ཊ། མུ་ཏིག་སྐྱེས་གནས་དང་། །ཤུཀྟིཉ་ཕྱེས་སོ། །ཀླུ་དང་། །བུ་དང་། ཁོ་ཚོ་ཚོ། ལྔལྔ་པེ་ལྟ་། ལགས་ཀྱི་དུང་གི་མིང་ལ་ཤཾ་ཁ་དང་ཀརྨུའོ། །དེ་ནས་སྦལ་བའི་མིང་ལ། བྷེ་ཀ་་་འཆོང་འགྲོ་དང་། བྷ་བྷ་བྷུ། དབྱར་སྐྱེས་སམ། ཆར་སྐྱེས་དང༌། ཤ་ལུར་དང་། སླ་བ་སྐྱལ་མཁན་དང་། ད་རྡུ་ར་རྣམས་སོ། །ཀ་ར་བ། ཏེ་གནམ་གཤིབ་ཏུ་—།
