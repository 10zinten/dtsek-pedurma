ཁ་པ་ལས་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཡོད་པ་ཅི་བྱེད་པོའོ། །་་། །འདུས་ལ་བབ། །རྐང་པ་གཉིས་ཀྱིས་བདག་གི་རྣམ་གྲངས་ཏེ། །གླེ་ཏྲ་ཛྷ་ནི་ཞིང་ཤེས་ཏེ། ཞིང་ནི་ལུས་སོ། །།ཨཱཏྨ་ནི། བདག་སྟེ། ཨ་ཏ་འགྲོ་བ་ལ། བྱོ་མ་ལ་སོགས་པ་རྣམས་སོ། །ཤེས་པས་མ་ནཾ་ན་རྐྱེན་དང་རིང་བའོ། །པུ་རུ་ཥ་ནི་སྐྱེས་བུ་སྟེ། པྲ་སྐྱོངས་བ་ལ། ་ཀུ་ཥན་རྐྱེན་དང་། པུར་དུ་བསྒྱུར་བའོ། །ལོ་འགའ། གཏིཚིག་བཅད་ཕྱེད་ཀྱི་སྲོག་ཆགས་ཀྱི་མིང་སྟེ། པྲ་ཎི། སྲོག་ཆགས་ཏེ། སྲོག་རྣམས་ལུས་ལ་འཛིན་ཅིང་ལྡན་པས་སོ། །ཙ་ཏ་ན་སེམས་ཅན་ནོ། །ཛིན་སྐྱི་ལྡན་ནོ། །ཛནྟུ་དང་ཛནུ་སྐྱེ་བོའི་སྐྱེ་དགུའོ། །་་་་། །་ཀྱི་ཁ་ན་བཞི་ཞེས་པ་ཡང་ཚིག་བཅད་ཕྱེད་ཀྱི་སེམས་ཀྱི་མིང་སྟེ། ཙིཏྟ་ནི་སེམས་པར་བྱེད་པའི་་་་སེམས་སོ། །ཙེ་ཏ་སེམས་དཔའོ། །ཧྲི་ད་ཡཾ་སྙིང་ངོ༌། །སཱནྟ་ཡིད་དོ། །ཧྲི་་ད་ཡ་སྙིང་པོའོ། །མཱ་ན་སཾ་ཡིད་བྱེད། །མ་ན་ཡིད་དོ། །སྐྱིད་ཚེ་། ལྟ་བས་ཀྱང་ཚིག་བཅད་གཅིག་གི་བློའི་མིང་སྟེ། བི་བུ་དྷ་ནི་བློ་སྟེ། །ཡིད་བུམ་པ་སྟེ། ༈་བུ་དང་ཤིང་རྟ་ལ་སོགས་པ་ལ་རྟགས་པར་བྱེད་པས་སོ། །མ་ནཱི་ཥ་ཡིད་འདོད་དམ༑ཡིད་འགྲོའོ། །དྷི་ཥ་ཎ། སྦོབས་བ་ཅན་ནོ། །སེམས་པར་བྱེད་པའི་བློའོ། །ཤེ་མུ། ཏཱི་ཉལ་སློང་ངོ༌། །མ་ཏི་བློ་གྲོས་སོ། ་། པྲཀྵེ། ་་རབ་་་ཏུ་བལྟ་བའམ་རྟོག་པའོ། །ཨུ་བ་ལནྟི་ཉེ་བར་དམིགས་པའོ། ་། ཅི་ཏ་སེམས་སོ། །སམྦིད་ཡང་དག་རིག་གོ། །པྲི་ཏ་པྲ་ཛཱ་སྟེ། སོ་སོར་རྟོགས་པའོ། །ཙེ་ཏ་ནཱ་་་སེམས་པའོ། །དྷཱི༔མི་རྗེད་པ་དང༌། དྷར་ན་བ་ཏི། གཟུངས་ཀྱི་ནུས་པ་དང་ལྡན་པའོ། །མེ་དྷཱ་ནི། ཡིད་གཞུངས་པ་དང་། སཾ་ཀལྤ་ནི། ཡང་དག་པའི་རྟོགས་པ་སྟེ་དེ་དག་བློ་ཁྱད་པར་ཅན་ནོ། །།།།་་དུ་སློབ་མ:ཡིད་ཀྱི་མིང་ལ། ་་ཀརྨ་མན་སཾ། ཡིད་ཀྱི་ལས་དང༌། །ཙིཏྟཱ་བྷོ་ག་སེམས་།
