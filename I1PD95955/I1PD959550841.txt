ལ་ལ་ནི་ར་ད་རུ་སྐྱེས་པའོ། །ཡུལ་དེ་ལ་འོད་བཟང་ཞེས་བྱ་བར་ཡང་རབ་ཏུ་གྲགས་སོ། །བརྨ་ནཱ་བྷ་ནི་བེའུའི་ལྟེ་བ་ཞེས་བྱ་སྟེ་སིནྡྷཱ་བཱ་རའི་འདབ་མ་འདྲ་བ་བེའུའི་ལྟེ་བའི་་་དབྱིབས་ཅན་དེ་དང་ཉེ་བའི་ཤིང་རྣམས་མི་འཕེལ་བར་བྱེད་པའོ། །་ཌ་ཎ་ཊིདུག་འཇོམས་པའི་མིང་ལ། བི་ཥ་བི་ཏྱ། །དུག་གི་སྨན་པའོ། །ཛདྷུལི། དུག་སེལ་ལོ། །བྱཱ་ལ་གྲྭ་ཏི། སྦྲུལ་འཛིན་ནོ། །ཨ་ཧི་་་་རུཎྜི་ཀ:། སྦྲུལ་དང་ཁ་འཐབ་པའོ། །གསུམ་པ་ཚལ་ཏཎ་ར་ཊི། དེ་ནས་ཚིགས་བཅད་ཕྱེད་ཀྱིས་དམྱལ་བ་སྤྱིའི་མིང་སྟེ། ན་ཀ་ར་དང་། ནི་ར་ཡ་རྣམས་དམྱལ་བའམ་འདོད་བྲལ་ལོ། །དུརྒ་ཏི་ངན་འགྲོའོ། ལེན་ཊི་། དམྱལ་བ་དེ་ཡི་དབྱེ་བ་ནི། ཏ་པ་ན་ཚ་བའོ། །/། ཨ་བི་ཙི་ནི་མནར་མེད་དོ། །རོ་ར་བ་དང་མ་ཧཱ་རོ་ར་བ་ནི་ངུ་འབོད་དང་ངུ་འབོད་ཆེན་པོའོ། །ཀཱ་ལ་པུ་ཏྲ་ཐིག་ནག་པོ་ཅན་ནོ། ་། ་་་་་་་། །པ་ཊིབ་ཏ། ཀུང་ཟ་ཁྱིལཁསྡུག་བསྔལ་གྱི་མིང་ལ། ཡཱ་ཏ་ནཱ་བརྣག་དཀའོ། །།ཏཱི་བྲ་བེ་ད་ན། མི་བཟད་པའི་ཚོར་བ་ཅན་ནམ་ཚོར་བ་དྲག་པའོ། ་། པཱི་ཌ། །གཟིར་བའམ་ཟུག་རྔུའོ། །བཱ་དྷ། །ཟུག་གཟེར་རོ། ་་། བྱཱ་ཐཱ། ན་བའི་སྡུག་བསྔལ་ལོ༑། དུཁཾ། ་སྡུག་བསྔལ་ལོ། །ཨཱ་མཱ་ན་ཧཱུཾ་ཡིད་འགྲོགས་སོ། །པྲ་པུ་ཏི་ཛཾ་བུ་སྐྱེས་པའི་སྡུག་བསྔལ་ལོ། །ཀཊཾ་དང་ཀྲྀཙྪ་དཀའ་བའམ་སྡུག་པའོ། །ཨ་ཧྥི་ལམ། འཇིགས་པ་ལེན་ཞེས་བྱ་བའོ། །་་་་་བློ་དེ་ནས་དེ་ནས་ཚིགས་སུ་བཅད་པ་ཕྱེད་དང་གཉིས་ཀྱིས་རྒྱ་མཚོའི་མིང་སྟེ། དེ་་་ལས་ས་མུ་དྲ་ནི་རྒྱ་མཚོ་སྟེ། །སམ་སྔོན་མ་ཅན་གྱི་ཨུནྡྲི་ལ། ཏདུ་ར་ལ་སོགས་པ་རྣམས་ལས་སོ་ཞེས་པས་རཀ་རྐྱེན་དུ་བྱས་པའོ། ་་། ཨབྷིཆུ་གཏེར་ཏེ། ཨཔ་སྔོན་མའི་དྷཱ་ལ་ཀི་རྐྱེན་དུ་བྱས་པའོ། །ཨ་ཀཱུ་པཱ་རཿཟླ་བས་འཕེལ་ཞེས་བྱའོ། །།
