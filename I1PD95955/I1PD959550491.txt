ཁམས་ཀྱི་ཁཟློཥ་ཎ་ཁོ་ན་ཏ་པྲ་མཱི་ཏོ་ཏྲི་ཥྛ་ཏེ་ཙི་ཏཱ་ཙི་ཏྟཾ་ཙི་ཏིརྫི་ཡ། །ཀ་བནྡྷོཉནི་ཀྲི་ཡཱ་ཡུལྟ་མ་དང་ཚེ་འདས་གསུམ་ཞེས་སོ། །།རོ་ཐབ་སྒང་ཕུག་ཙི་ཏི་མོ། །ཀ་བན་དྷ་ནི་མོ་མིན་ཏེ། །བྱ་ལྡན་པ་མཱནྟ་ཀ་ལེ་བ་རཾ། །ཟླ་ཤཱ་ནཾ་སཱུཏྲི་དྲྀ་བ་ནཾ་ཀུ་ཎཱ་པཤ་བ་མཉྫི་ཡཾ། །པྲ་གྲ་མགོ་མེད་རོ་ཡིན་ནོ། །དུར་ཁྲོད་དང་ནི་གཤིན་གྱི་ནགས། །རོ་དང་ཤི་བ་མོ་མིན་ནོ། །གདོན་དང་ཧོ་པ་གྲ་ཧོ་ཐཾ་སུཾ་ཀཱ་ར་སཱུཏྤནྟ་ནཱ་ལ་ཡེ། །་སི་བྷཱུརྞ་ས་བཔཱ་ཎཱཉྩི་པཾ་འབྱུང་པོ་འཆིང་བྱེད་དོ། །ཀཱ་ར་སྟད་ནི་འཆིང་བྱེད་གནས། །སྐྱེས་བུས་སྲོག་བཱ་ཎའོ། །དེ་བཞིན་ཛཱི་བོཉསུ་དྷཱ་ར་ཎཾ། །ཨཱ་ཡུརྫི་བི་ཏ་ཀཱ་ལོ་ནཱ་ཛཱི་བཱ་ཏུ་ཛཱི་བི་ཏོ་ཥ་དྷཾ། །འཚོ་དང་གསོན་པ་དང༌། །ཚེ་ནི་ཇི་སྲིད་འཚོ་བའི་དུས། །སྐྱེས་བུ་འཚོ་བྱེད་འཚོ་བའི་སྨན། །ཤ་ཏྲི་ཡ་བརྒ། །རྒྱལ་རིགས་ཀྱི་སྡེའོ། །༈ཨུ་རུ་བྱུ་སྤུ་རུ་ཛ་ཨཱཆུ་བི་སྭཱ་བྷཱ་༔མི་སྤྱི་ཤོ་བི་ཤ། །ཨཱ་ཛཱི་བོ་ཛཱི་༈བརླས་བྱུང་བརླ་སྐྱེས་འཕགས་པ་དང་། །རྗེ་རིགས་ས་འཛིན་རྗེའུའོ། །ཀུན་འཚོ་འཚོ་བ་བ་རྟ་བི་ཀཱ་བནྡྷ་བྲྀཏྟི་རྦཏྟ་ན་ཛཱི་བ་ནེ། །པྲི་ཡཾ་ཀྲྀ་ཎིཔཱ་ཤུ་པཱ་ལྕི་བཱ་ནི་་ཙེ་དང༌། །གསོན་བྱེད་སོ་ནི་འཚོ་ཚིས་སོ། །གསུམ་ཡིན་རྨོ་རྐོ་དུད་འགྲོ་སྐྱོང་། །ཚོང་དང་འཚོ་བའི་ཏི་བྲྀཏྟ་ཡ། །སེ་བཱཉྩ་བྲྀཏྟི་ར་ནཾ་ཏཾ་ཀྲྀ་ཎི་རུཉྩ་ཤི་ལཾ་ཀྲྀ་ཏཾ། །དྭེ་ཡཱ་ཙི་ཏཱ་ཡཱ་ཚིས་ཞེས་བྱ། །ཞབས་ཐོག་པ་ནི་ཁྱིའི་འཚོ། །རྨོ་རྐོ་དང་ནི་ཕྲུ་སློག་དང་། །སྟོན་ཐོག་དང་ནི་ལོ་ཙི་ཏ་ཡོདྨ་ཐཱ་སཾ་ཁཾ་མྲྀ་ཏཱ་མྲྀ་ཏེ། །ས་ཏྭཱ་ནཱཾ་ཏཾ་བ་ཎིག་བསཛྫཾ་ཎཾ་ཡརླུ་ད་ཐོག་གོ། །བྱ་ཚོལ་དང་ནི་སློང་མོ་པ། །ཇི་ལྟར་བདེན་དང་ཚོང་དངོས་པོ། །བུན་དང་སྐྱེག་དང་ནཾ༑། ཨུདྡྷཱ་རོཉརྦ་པྲ་ཡོ་གསྟུ་ཀུ་ཤྲཱི་དཾ་བྲིདྡྷི་ཛཱི་བི་ཀཱ། །ཡཱཙཱ་ཡཱཊྚ་ཡཱ་ཙི་ཏ་ཀཾ་ནི་བདེ་འཚོ་དང་། །དོན་སྦྱོར་འཚོང་ངན་འཚོ་བ་ཅན། །བསྐྱི་བ་དང་ནི་གཡར་བ་དང༌། །བརྗེ་བ་མ་ཡཱ་དཱ་པ་མི་ཏྱཱ་ཀཾ། །ཨུཏྟ་མཙྪཱ་དྷརྨནྡྷོ་དྷོ་པྲ་ཡོ་གྷཾ་གྲ་ཧ་ཀོ་ཀྲ་མཱཏ། །དང་ནི་བསྒྱུར་ཟློག་གོ། །ནོར་གྱི་བདག་དང་བུ་ལོན་པ། །གཉིས་པོ་རབ་སྦྱོར་ལེན་མཁན་རིམ། །_—།
