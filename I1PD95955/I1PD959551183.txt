བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽བཱ་བཱ་ནོ་ལིཾ༔ཤར་ན་བ་བྷི་ར་སཾ་བཱདྷ། །ན་ན་བི་ཛ་ཛ་བི་སྭ་རི་ར་པ་ག་ཝ་དང་ན་གཉིས་མ་རྣམས་མདའ་དང་དགུ་ཡིས་གནོད་མེད་དོ། །ན་ན་ཝ་དང་ཛ་ཛ་བ་རྣམས་དབྱངས་ཀྱིས་རཱ་ཛི་ཏཱ། །པྲ་ཧ་ར་ཎ་ཀོ་ལིཀཱན་ན་བི་ནི་ནི་བི་ཏི། ཙྪཱ་ཡཾབ་སན་ཏིལ་གཞན་གྱིས་མི་ཐུབ་པ། །ན་ན་ཝ་དང་ན་ན་ཝ་རྣམས་དེ་ཡིས་མཚོན་ཆ་ཁ་འབུས་སོ། །ཏ་བྷ་ཛ་ཛ་ག་ག་ཀཾ་ཏཾབྷ་ཛཛ་གོ་ག། པའོ། བ་སུ་དྷཱརྨ་བོ་ན་པི་ཝཾ་རཾཙ་པཉྩ་གྲ་ཧི། རངཇིཔའོ། དཔྱིད་ཀྱི་ཐིག་ལེར་ཤེས་པར་བྱ། །ས་ལ་ཝ་ན་གཞན་ཡང་ར་དག་ལྔ་དང་གཟའ་ཡིས་ནོར་འཛིན་ནོ། །གུ་རུ་ནི་དྷ་ན་མ་ནུ་ལ་གྷ་རི་ཧ་ཤ་ཤི་ཀེ་ལཱ། པའོ། སྲ་གི་ཡཾ་མ་པི་ཙ་ར་ས་ན་ལྕི་བ་ཐ་མ་ཡང་བ་མ་ནུས་འདིར་ནི་བོང་ཅན་གྱི་ཆ། །འདི་ནི་གཞན་ཡང་རོ་དང་དགུ་ཡིས་ངལ་བསོར་གྱུར་བ་ཀ་བི་ར་མ་ཎེ། །བ་སུ་མུ་ནི་ཡཾ་ཏི་རི་ཏི་མ་ཎི་གུ་ཎ་ནི་ཀ་ར། པའོ། ན་ཕྲེང་བའོ། །དེ་ལྟ་བུ་ལ་ནོར་དང་ཐུབ་པར་ངལ་བསོ་ཡོན་ཏན་ནོར་བུའི་ཚོགས། ་། ན་ན་ག་ཝ་ར་ར་་་ན་ན་གོ་བ་ར་ར་བཱས་ལི་ནིཏཱ༔་ལོ་ཀི། པའི། ན་ཛཾ་བྷ་ཛཙཱ༔རཱ་ཛི་ཏེ་མི་ཝ་རྣམས་བརྒྱད་དང་འཇིག་རྟེན་ཕྲེང་ལྡན་མ། ཀ། ན་ཛ་བྷ་ཛ་ར་རྣམས་རྣམ་པར་སྤེལ་བ་འདི་དཾ་ས་མུ་དྲ་ཀི། ,པའོ། ཧྥཀཾ་ན་སཱ་མུ་ནེ༔ཁ་སྤྲི་ཀཾ་བྷ་ག་ཛ་བི་ལ་སི་ཏོ། པའོ། ནི་རྒྱ་མཚོ་ཅན། །བྷ་ར་ཀ་ནད་ན་ས་ཐུབ་ལས་མཁའ་ལ་ཁྱུ་མཆོག་གླང་ཆེན་རྩེན། །ན་ཛཾ་བྷ་ཛ་རཱ་གུ་ཏུཉྩ་ཡོ་དི་ཏེཏྟཾ་དཱ་ཀུ་སུམྦྷ། པའི། ཡཾ་ཧཱུཾ་ཧཱུཾ་སྭཱ་བྷཱཝཱ་ར་ས་ཧ་གལ་ཏེ་ན་ཛ་བྷ་ཛ་ར་ལྕི་དེ་ནི་དེ་ཚེ་ལེ་བརྒན་རྩི། །ཡ་མུ་ན་ས་བྷ་ལ་ཝ་རྣམས་རོ་དང་འཕྲོག་བྱེད་ངལ་་ར་བི་རཱ་མ་ཞི་ཁེ་རི་ཎཱི། པའ། ཛཾ་སཾ་ཀླུ་ན་ར་རྭ་བ་སཾ་གྲ་ཧ་ཡ་ཏིཔཕྲ་བྷཱུ་མི་བསོ་གཙུག་ཕུད་ཅན། །ཛ་ས་ལ་ཝ་ན་ར་ར་རྣམས་ནོར་དང་གཟའ་ལ་སྡོད་པ་ཡངས་ལྡན་་་ཏེ། ནཔའིདི༔་ནི་ཝེ་ཤ་པ་ཏྲ་པིཏཾ་ཏཾ་བྷ་ར་ན་བ་ན་ན་བི། ན་སུ་མ། །བྷ་ར་ན་བ་ན་ན་ཝ་རྣམས་ཕྱོགས་དང་ཐུབ་པར་འོད་མའི་འདབ་མ་ལྟུང༌། །།ན་སུ་མ་བ་ཛཛཱ་བིསཱདཱབྷཱི་སཱརཱདྷ་རི་ཎཱི་མ་ཏཱ། པའོ། མནྡ་ཤཱནྟཱ་མ་བ་ན་སུ་ར་མ་ཝ་ཛ་ཛ་ཝ་རྣམས་རོ་ཆུ་དབྱངས་ལ་རི་དྭགས་མོ་རུ་འདོད། །མ་ཝ་ན་སུ་ར་ར་ཝ་རྣམས་མཚོ་དུས་།
