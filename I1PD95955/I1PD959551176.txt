ས་མོ་པ་རི་ཀླུ་པི། འི་། ཀྲ་མ་ཁཟི་རུཏྠཱ་ནི་བྲྀཏྟ་མ་ལ་ཧྥེ་ཀ་ལ་གྷ་དི། །བསླང་། །མཉམ་པ་སྟེང་དུ་བཅུག་པ་ཡིས། །ལངས་པ་རིམ་པས་ཐུང་དུ་སྟེ། ༈། བྲྀཏྟ་ཡང་མེད་གཅིག་ཡང་པིངྒ་ལཱི་ལ་གླུ་ཀྲི་ཡཱ། །ཨུཙྪི་སྱ་རཱུ་པ་མཱ་རྱ་མ་ཀྲྀ་ཏེཔི་སཾ་ཊུ་ལི་ཁེཏྟ་སོགས། །དམར་སེར་ཅན་གྱི་ཡང་བའི་བྱ་བའོ། །གཟུགས་ནི་དང་པོར་མ་བྱས་ཀྱང་། །སྟེང་གཞུག་ཏོནདྷསྦི༑། སཱ་སཱ་དྷ་ར་ས་མ་མི་སྲཔྲྀཏྟ་མ་ལ་བྷེ་ཀ་ལ་གྷ་དི། །གྲངས་བྲི་དེ་ནས་འོག། གནས་པའི་རང་རང་འོག་མཉམ་བསྲེ། །བྲྀཏྟ་ཡང་མེད་གཅིག་ཡང་སོགས། །ཨ་ཡ་མ་སྨ་དཱི་ཡ་ལ་གླུ་ཀྲི་ཡཱ། །ཀཱ་སཾ་ཁྱ་བྲྀཏྟཱ་ནཱ་མ་ཀླུ་ར་བཱ་རཱ་ཎི་ཡ་མ་གུ་ཎི་རཱུ་འདི་ནི་ཁོ་བོའི་ཡང་བའི་བྱ་བའོ། །བྲྀཏྟཱ་རྣམས་ཀྱི་གྲངས་ནི་གཟུགས། །ཟུང་གི་ཡི་གེའི་གྲངས་སུ་པཾ། །ཨུ་དྷིཥྛ་དེ་ཥཱནྟོ་དཱི་གུ་ཎོ་ལ་གླུ་ཀརྨ་པིཎྜ། །སཾ་ཁྱི་བ་བཎྜ་གུ་ཎི་ཏཱ་བསྒྱུར། །ཨུདྡིཥཱ་ཡི་ཨངྐའི་མཐའ། །ཉིས་འགྱུར་ཡང་བའི་བྱ་བ་བསྡུ། གྲངས་ཉིད་ཝཎྜས་བསྒྱུར་ཏ་ཏསྟ་དནྟཾ་ཏ་ཏོཉསླུ་བྷ་ཡ་པཎྞ། །ཨཱ་དོ་ཝཎྞ་མ་་གུ་ར་བོ་ལ་གྷ་བལྟ་ད་ནུ་དེ་ནས། །དེ་ཕྱེད་དེ་ནས་གཉིས་ཀའང་སྡོམ། །དང་པོར་ཡི་གེ་དབུས་ལྕི་ཡང་། །དེ་རྗེས་ཕྱི་མོ་རྣམས་མཱ་ཏཱ། །ཨ་དྷཱ་ཏིར་ག་དྷ༔་དྲི་གུ་ཎི་ཀོ་ནཱ་ནི་བརྞ་བྲྀཏྟཱ་ནི། །ཏུ་ལྕ་ཧི་གུ་རུ་ལ་ཡིན་ནོ། །ཡི་གེ་བྲྀཏྟ་གཉིས་ཀྱིས་བསྒྱུར། །གཅིག་དམན་ཨ་དྷཱ་འཕྲེད་དང་འོག། ལྕི་ཡང་རྣམས་ཀྱི་ས་གླུ་ནཾ་བྷུ་མིས་ཏཱནྟ་རཱ་ཎཱཉྩ། །ཨ་ནུ་པཱུ་ཏྲི་པྲལྟ་རོ་ནཎོདྡི་ལ་གྲུ་ཀྲི་ཡཱ་ཙི་བ་སཾ་ཀྵ། དང་ནི། །དེ་ལ་བར་ཡང་མཚུངས་པའོ། །བྲསྨཱ་ར་དང་ནཋ་ཨུདྡི། །ཡང་བའི་བྱ་བ་གྲངས་དང་ཨ་དྷ་སྟེ། །ཏ་ཐཱ་དྷཱ་ཡོ་གོ་བྲྀཏྟེ་ཥཱ་པྲ་ཏྱཱ་ཡ། །ཥ་ཌི་མོ་ལ་གྷུ་བྷིརྦ་རྩྭ་ར་ཧི་ཏཱ་ལ་གྲུ་ར་དེ་ལྟར་དྲུག་པོ་དེ་རྣམས་རིམ་པ་བཞིན། །བྲྀཏྟ་རྣམས་ལ་རྟོགས་བྱེད་རྣམས་ཡིན་ནོ། །ཡང་བ་རྣམས་ཀྱི་ཡི་ཧི་ཏ་ནཾ་ད་ལཉྩ་མ་ཏཱ་ཎཱི་མཱ་ཏཱ། །བཎ་ར་ཧི་ཏཱ་ཡ་ཐི་བ་ཏཱ་བནྟ་ཨི་ཧ་གུ་གེ་དོར། །ཕྱི་མོའི་ཡང་བ་དོར་འདབ་མའང་། །ཡི་གེ་དོར་བའི་ཕྱི་མོ་ཡང་། །དེ་རྣམས་ཇི་སྙེད་འདིར་ར་བ། །དྲི་གུ་ཎ་གུ་རུ་ཎཱ་ཙ་མཱ་ཏཱ་མཱ་ཏྲ་བྷིརྦ་ར་ཧི་ཏཱ་དཱི་གུ་ཎ་བཎཱ། །གུ་རུ་བི་ར་ལྕི་བ། །ལྕི་བ་ཉིས་འགྱུར་གྱི་ཕྱི་མོ། །ཡི་གེ་ཉིས་འགྱུར་ཕྱི་མོས་སྤངས། །ལྕི་བ་བྲལ་བའི་ཡི་གེ་།
