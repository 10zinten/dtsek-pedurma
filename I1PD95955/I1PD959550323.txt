ཡོལ་བ་ཁྱོད་ལ་ངོ་ལྟ་ཕྱོགས་ག་མའང་ཡང་ན་སྟེར་བར་བྱ་སྟེ། ནམ་མཁའ་ལ་འགྲོ། བི་ཧཾ་གཌསི། ནམ་མཁའ་ལ་་་འགྲོ། བི་ཧ་གཡཌསི། །ཞི་བ་དང་དྲིས་བྱེད། ༈ཨ་ཊོ། ,ཨ་ཊ་དག་གོ། །བྱིངས་ལས་སྔོན་ན་མིང་ཡོད་ཅིང་མིང་དེ་་དང་འབྲེལ་བའི་ལས་དང་སྦྱོར་བའི་བྱེད་པའི་དོན་ལ་རྐྱེན་ཨ་དང་ཊ་དག་འགྱུར་ཏེ༑ཨམྦི་སྔོན་མའི་གྲོགས་འཕྲོག་པ་ལ་ཨ་ཡོན་ཏན་བྱས་པས། ཨ་སི། ཨསྦི་ཧ་རཿཤཱ༑ཡརུས་པ་ཁྱེར་རོ་ཁྱིས། །དེ་བཞིན་དུ། གོ་ཆ་ཕྲོགས་སོ་རྒྱལ་རིགས་ཀྱིས། ཀ་པ་ཙ་ཧ་་་རཨསི། ཀྵ་ཏྲི་ཡཨསི། བྷ་རཾ་ཧ་རཡ་ཁུར་ཕྲོགས་སོ། །ཨསི། དྷནུ་དྷརཿཀུ་མཱ་རཿགཞུ་འཛིན་ནོ་གཞོན་ནུས་སོགས་དང༌། ༈ཅ་ར་གཡོ་བ་ལཊི། ཊ་ཡིག་མོ་རྐྱེན་ཨཱིཔྲ་་་འཇུག་པའི་དོན། ཊ་སི། ཀུ་རུ་ཙ་རཿཡུལ་སྒྲ་ངན་དུ་སྤྱོད། ་ཊ་ཨིབ། ཀུ་རུ་ཙ་རཱི་ཡུལ་སྒྲ་ངན་དུ་སྤྱོད་མ་སོགས་སོ། །ན་ད་ན་དེ་ལ་སྐྱོ་ཁོ་རྗེ་དང་གང་ལ་བཏོ༈ཨི་ཁ་ཁི། ཨི་ཁ་ཁིའོ། །མིང་སྔོན་མའི་བྱིངས་ལས་རྐྱེན་ཨི་དང་ཁ་དང་ཁི་རྣམས་འགྱུར་ཏེ། སྟམྦ་སྔོན་མའི་ཀྲྀཏྲ་བྱེད་པ་ལ་རྐྱེན་ཨི། སྟམྦ་ཀ་རི་བྲྀ་་ཧི༑འབྲུ་འདེབས་པར་བྱེད། དེ་བ་སྔོན་མའི་ཨཔྤེ་ཁྱབ་པ་ལ། དེ་བཱ་པིལྷས་ཁྱབ་པོ། །སོགས་སོ། །བ་ད་ལྔ་ལ་ད་་་ཁྱཁ་ཡ་ལ་ཤེན་༈ཁི་ཏི་པཱུརྦ་པ་ད་སྱ་ཁ་འགྲོ་བ་ལ་སྔོན་མའི་ཚིག་གིའོ། <། རྐྱེན་ཁ་ཡིག་་་རྗེས་འབྲེལ་འགྲོ་བ་ཕ་རོལ་ལ་སྔོན་མའི་ཚིག་དེ་ལ་མུམ་ཨཱ་ག་མ་སྟེར་བར་བྱ་སྟེ། གླུ་མ་སྔོན་མའི་ཀྲྀ་ཨེལ། ཁ་ཚིག་སྔོན་མར་མུམ་ཨཱ་ག་མ་མ་གནས་དོན། ཨུ་་་བརྗོད་དོན། རྐྱེན་གྱི་ཁྲ་ཡིག་དེའི་དོན་ཡོན་ཏན་བྱས་པས། ་ཀླུ་མོ་ཀ་རཿབདེ་བར་བྱེད། དེ་བཞིན་དུ། པྲི་ཡཾ་བ་དངཿདགའ་བའི་གཏམ། ་ཨཱཏྨ་བྷ་ར། བདག་ཉིད་རྒྱས་པ། ཨུ་ད་རཾ་བྷ་རིལྟོ་བ་ཁེངས་པ་སོགས་སོ། །།།ཅུ་ཊ་བི་ཀོ་ཊ་འི་པ་༈ཨེ་ཛཾ་ཁཤ་། ཨེ་ཛ་རྣམས་ཀྱི་ཁཤའོ། །བྱིངས་ཨེ་ཛ་ལ་སོགས་པ་་་།
