ནོ་ནཾ་ཐ་ཤལ་དབྱེན། །ཧཱི་ནོ་མ་གང་དམན་ཐ་ཤལ། །བང་དང་དཔའ་བཏར་སཱི་ནོ། ་། ཨབྷི་པརྣོ་ཉེས་བྱས་དང་། །དགྲ་དབང་སོང་དང་དོན་མེད་བྱས། །ཀལཱ་ས་ནི་རྒྱན་དང་ནི། །རྨ་བྱའི་མདོངས་དང་ཚོགས་པ་ལའང༌། །།པ་རཱི་བཱ་ས་ཡོངས་གཅོད་དང་། །བ་ཅུསྟོ་དང་ཆུ་གནས་པ། །འཇོ་བ་ལ་བདག་གོ་པོ། །དབང་ཁྱབ་འཇུག་བྲི་ཏྟཱ་པཱི། །བཱ་ཋཱ་བླངས་པ་མིག་ཆུའོ། །ཀ་ཤི་པཱུ་ནི་ཟས་གོས་གཉིས། །ཏཝཾཉལ་བ་ཁང་སྟེང་ཆུང་མ། །བི་ཊ་པོ་ནི་ཚང་ཚིང་ངོ༌། །མོ་མིན་པྲ་ཏི་རཱུ་པ་ས་རཱུ་པ། །ཨི་རཱུ་པ་ནི། །མཁས་པ་དང་ནི་ཡིད་འོང་བ། །དབྱེ་བ་རྟགས་འདི་རྣམས་ལ་ནི། །རུ་སྦལ་མོ་དང་དབྱངས་ཅན་མའི། །པི་ཝཾ་ཀཙྪ་པི་་ཡིན་ནོ། །བར་དོ་བར་སྲིད་སེམས་ཅན་དང་། །རྟ་དང་ལྷ་ཡཾ་་གནྡྷརྦ། །ཀརྨུ་རྒྱན་མངས་དུང་ལའོ། །ལྕེ་གཉིས་སྦྲུལ་དང་སྒྱུ་མའོ། །པཱུརྦ་གཞན་གྱི་རྟགས་ཅན་ཏེ། །དང་པོ་མང་ཚིག་སྔོན་ལ་སྐྱེས། །།ཀུན་བྷོ་བུམ་པ་གླང་ཆེན་མགོ༑:། ཌི་བྷོ་བྱིས་པ། །།སྟམ་བྷོ་སྐ་བ་རེངས་བྱེད་ལ། །ཤཾ་བྷུ་ཚངས་དང་་སྤྱན་གསུམ་པ། །ར་གྲྭ་མང་ལ་དང་སེམས་ཅན་བྱིས། ?། བི་:ཤྲཱི་བྷ་་ནི་ཡིད་ཆེས་མཐུན༑། རྔ་ནི་དུསྡུ་བྷིསྐྱེས་སོ། །ཚེས་ལོ་དུསྡུ་བྷི་སྟེ་མོ། །ཚོན་ཆེན་མ་་ནིང་ཀུ་སུཾ་བྷོ། །སྤྱི་བླུགས་ལ་ནི་སྐྱེས་བུའོ། །རྒྱལ་རིགས་ལྟེ་བ་ན་བྷི་སྐྱེས། །སུ་ར་བྷི་ནི་བ་ལང་མོ། །ས་བྷ་ཚོགས་དང་ཡིད་ཆེས་ཆོས་སོ། །གསུམ་བདག་་་ཡིད་ཆེས་བསྨྲ་བྷ། །བྷའི་འོ། །།འོད་ཟེར་དྲ་བ་ནི་རབྷཱིའོ། །སྤྲེའུ་སྦལ་བ་་མཆོད་འགྲོའོ། །།འདོད་དང་ཡིད་སྐྱེས་ཀཱ་མོའོ། །པ་རཱ་ཀྲ་མོ་དཔལ་པ་་་ལ་་གནོན༑། དྷརྨ་བསོད་ནམས་གཤིན་རྗེ་རིགས། །རང་བཞིན་སྤྱོད་དང་ཟླ་བའི་བཏུང་། །ཐབས་དང་སྔོན་མ་རྩོམ་པ་དང༌། །གོ་རིམ་ཨུ་པ་ཀྲ་མིའོ། །ཚོང་་ལམ་གྲོང་ཁྱེར་རིག་བྱེད་དང༌། །ནི་ག་མའོ། །ཞི་དང་བཟོད་པ་ག་མའོ། །—།
