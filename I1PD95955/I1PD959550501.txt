:ས་ལ་སྐུ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིཀྱི་་ལ། ལྟ་ཕྱོགས༌སྐོར་མ་ལོན་པ་ལཾ་ཀརྞ་ཙ་ཏུཥཱ་ཡཾ། །སུ་བནྡྷཱ་བིསྟོ་ཧེ་མོ་ཀྵ་ཀུ་རུ་བིསྟསྟུ་ཏཏྤ་ལེ། །ཞོ་བཞི་སྲང་ངོ་མོ་མིན་ནོ། །གསེར་འཇམ་ཞོ་ཆ་གསེར་མིག་གོ། །ཀུ་རུ་བ་སྟེ་དེ་ཡི་སྲང་། །ཏུ་ལཱ་སྟི་ཡཱ་པ་ལ་ཤ་ཏཾ་བྷཱ་ར་པཱ་ཤ་ཏིསྡུ་ལཱ། །ཨཱ་བི་ཏཾ་ཟེར་ཨཱ་ཙི་ཏཾ་ད་ཤ་བྷ་རཱ༔མཉམ་བྱེད་མོ་མིན་སྲང་བརྒྱ་ལ། །ཁུར་ཡིན་ཉི་ཤུ་ཐམ་པ་ནི། །སྲང་ངོ་ཨཱ་ཙི་ཏཾ་ཁུར་བཅུ། །སྩིསཱ་ཀ་ཊོ་བྷ་ར་ཨཱ་ཙི་ཏ། །ཀནཱ་པ་ཎཀརྟི་ཀསཱུཏྟཱ་རྟི་ཀེ་ཏཱ་ལིཾ་ཀེ་པ་ཎ། །ཤིང་རྟ་ཁུར་རྟ་ཨཱ་ཙི་ཏ། །ཀར་ཥཱ་པ་ཎ་དོང་ཚེ་ཡིན། །ཟངས་མའི་དོང་ཚེ་ཀཱནྟི་ཀི། །མོ་མིན་བྲེ་ཨམྦི་ཡཱ་མཱ་ཌྷ་ཀ་དྲོ་ཎོ་ཁཱི་རཱི་བ་ཧོ་ནི་ཀུཉྩ་ཀ། །།ཀུ་ཌ་བསྲསྦྲ་ཨི་ཏྭཱ་རཱུ་པ་རི་བོ་ཨཱ་ཌྷ་ཀ། བྲེ་དང་ཁཱ་རི་བཱ་ཧ་ནི། །ནི་ཀུཉྩ་ཀ་ཞེས་པའོ། །ཀུ་ཌ་བ་ནི་བྲེ་ལ་སོགས། །ཡོངས་སུ་མཱ་ཎཱརྦ་ཀཱཔྲ་ཐཀ། པཱ་དསྐུ་རཱི་ཡོ་བྷ་གསཱུ་དཾ་ཤ་བྷཱ་གོ་ཏུ་བཎྜ་ཀེ། །དྲ་བྱོ་བིཏྟ་འཇལ་བྱེད་སོ་སོའོ། །རྐང་པ་དང་ནི་གཅིག་ཆ་ཡིན། །ཆ་དང་ཆ་ཤས་འགོད་པའོ། །རྫས་དང་ཡོ་བྱད་སྭཱ་ཏེ་ཡཾ་རཾཀྵ་མཀྵ་དྷ་ནཾ་བ་ཙ་སུ། ཞྭ་། ཧི་ར་ཙྩཾ་དྲ་བེ་ན་རླུསྣ་ནོར་དང་ནི། །འབྱོར་པ་དངོས་པོ་དཀོར་དང་རྫས། །དབྱིག་དང་ཟང་ཟིང་ལོངས་སྤྱོད་དོན། །མརྦ་རེ་བི་བྷ་བཱ་ཨ་པི། །སཱཀྐོ་ཥཉྩ་ཧི་ར་ཎྱཤྩ་ཧེ་མ་རུ་པྲ་ཀྲྀ་ཏཱ་ཀྲྀ་ནོར་དང་དངོས་པོ་ཉིད་ཀྱང་ངོ༌། །མཛོད་དང་རིན་ཆེན་གསེར་གཟུགས་དང་། །སྒྲོམ་དང་སྣོད་ཅེས་བྱ་ཏེ༑། ་། ཏཱ་བྷཱུ་མ་ཡ་དུ་ནྱ་ཏུཀྐུ་་ཏདྲྭ་ཡ་མཱ་ཧ་ཏཾ། །།གཱ་རུདྨ་བའོ། །།དེ་དང་གང་ཞིག་གཞན་མཛོད་དང་། །དངུལ་དང་དེ་ྋགཉིས་མ་ཧ་ཏཾ། །མཁའ་ལྡིང་གཤོག་ཏཾ་མ་ར་ཀ་ཏ་མསྨྲ་གཏྟཾ་ཧ་རིམྨ་ཎི། །སོ་ཎ་ར༑་ལོ་ཧི་ཏ་ཀཔདྨ་རཱ་པ་མར་གད་དང་། །རྡོ་ཡི་སྙིང་པོ་ནོར་བུ་ལྗང་། །རིན་ཆེན་དམར་པོ་ལོ་ཧི་ཏ། །པདྨ་རཱ་ག་དེ་གོཐ་མོཀྟི་ཀཾ། །མུཀྟཉཐ་བི་དྲུ་མཔཾ་སཾ་པྲ་བཱ་ལཾ་བུནྣ་པུརྨ་ཀཾ། །རཏྟཾ་ནས་ནི། །མུ་ཏིག་དང་ནི་མུཀྟའོ། །དེ་ནས་བྱི་རུ་སྐྱེས་བུ་སྟེ། །བྱི་རུ་ཕོ་དང་མ་ནིང་ངོ༌། །རིན་མེ་ཎིརྡ་ཡོ་རསྨྲ་ཛཱ་ཏོ་མུཀྵ་དི་ཀེ་པི་ཙ། །སཱཎྞ་སུ་བནྡྷཾ་ཀ་ན་ཀཾ་ཧི་ར་ཧཱུཾ་ཧེ་ཆེན་ནོར་བུ་གཉིས་ཡིན་ཏེ། །རྡོ་སྐྱེས་མུ་ཏིག་ལ་སོགས་དང་། །གསེར་དང་མདོག་བཟང་ཀ་ན་ཀཾ། །།
