ཡ་སཱུཀ་ཨེ་བ་ད་ལ་ཡོ་རུ་བྷ་ཡོ་ར་པི་བྷ་བ་ཏིཥཥཱ་ག་ཎ༔ཨ་པ་རཱནྡྷ་བཙྪ་ཤེ་གང་གི་འདབ་མ་གཉིས་ཀའི་ཡང༌། །དྲུག་པའི་ཚོགས་ནི་ཀ་ཉིད་གྱུར། །ཕྱེད་ཕྱི་བཞིན་དང་ལྷག་ཁོ་ནི་ཡ་མསྟེ་མཱ་ཧེ་ཏུ་པ་གཱི་ཏི། །པཱཧཱནྟེ་ཥàསྟདྨོ་ཡཾ་ཏྲ་ཛ་ཁོ་ཏུ་པཉྩནྟདེ། ངེས་པ། །དེ་ནི་ཉེ་བའི་དབྱངས་སུ་སྨྲ། །གང་དུ་སྔ་ཕྱེད་དྲུག་གྱུར་ཀ། ཕྱི་ཕྱེད་ལ་ནི་ཛ་ད་ག། ཤེ་ཥཏྲ་ཏི་ནི་ཡཾ་མོ་ང་ལ་བི་པཎཾ་ཡཱདྷཱ་བ་ཏི་བྷ་བ་ཏི་སངྒཱི་ཏི༔ཡ་ཧཱུཾ་ད་ལྷག་ངེས་མདབ་མ་ལག་པ་ལས། །གྱུར་པ་དེ་ནི་མཆོག་དབྱངས་ཡིན། །གང་མདབ་མདབ་མར་བརྒྱད་ལེ་ད་ལེཉཏྠོ་ཙ་ཥབློ་ག་ཎེསྡུ་ཛོ་བཱ་ཁོ་བཱ། པཱུརྦཛྷ་བཙྩ་ཤེ་ཥཔྲ་ཏི་ནི་ཡ་རྣམས་ཙ་ཞེས་བྱ། །དྲུག་པའི་ཚོགས་སུ་ཛའམ་ཡང་ན་ཁ། །བྱེད་སྔ་བཞིན་དང་ལྷག་མ་སོ་སོར་ངེས། །མོ་བྷ་བ་ཏི་བྷ་བ་ཏི་སཱནྟཾ་གཱི་ཏི། ཨཱརྱ་ཛཱ་ཏ་ཡ། །།།རིགས་ཁོགྱུར་པ་དེ་ནི་འཕགས་པའི་དབྱངས་སུ་འགྱུར། །འཕགས་མའི་ཛ་ཏི་རྣམས་སོ། །།།ག་ང་བ་མ་ཚཥ་ན་ལྔ་བ[g::ཀ་་་། །ཟླ་བ་ཁ་ཟ་:ད་ཏྲ་ཡ་མ་ཡུ་ཛྭཆུ་ཁེ་ཡུ་ཛོདྨ་ཙཱ་ཏུཏྟཾ་ར་ལ་གཱ་སྐུ་པཱིཥྛ་ཏ། །བྱི་ཏཱ་ལཱི་ཡཾ་ཏཱ་ཟུང་མིན་དང་པོར་ད་གསུམ་ཟུང་ད་བཞི། །ཕྱི་མ་ལ་ནི་ར་ལ་ག་རྣམས་ཏེ། །ཟུང་ལ་ཡང་དྲུག་པར་དུ་ཙ་ཏེ་ལུ་གུ་ཥཱཀྵ་ན་ནི་རཱནྟ་རཾ་ཡུ་ཛེ། །བེ་ཏཱ་ལཱི་ཡོ་པ་མཾ་མུ་ཁེ་ཙེ་དོ་པ་ཙྪནྡ་མ་ཆད་དུ་མིན། །དེ་ནི་རོ་ལངས་ཅན་དུ་བརྗོད་པར་བྱ། །གལ་ཏེ་དང་པོ་རོ་ལངས་ཅན་ལྟ་བུ། །ཕྱི་མར་སི་ཀཾ་ར་ཡོ་ཏུ་བྲྀཥྛ། །ཨོཾ་པཱ་ཏ་ལི་ཀཾ་བྷ་ག་གནྟཱ་མུ་ཁ་མི་ཧ་བེ་ཏཱ་ལཱི་ཡ་ས་མཱ་ཡ་དག་ནི་ཉེ་སྡེབ་སྦྱོར། །འདིར་ནི་དང་པོ་རོ་ལངས་ཅན་དང་མཚུངས། །བྷ་ག་ག་མཐའ་ཀུན་ནས་སྲབ་ནཾ། ཞ། ཡུདྨ་མུ་ཁེ་དཱནྡི་རོ་ཛ་དོ་པཱ་ཙ་བྲྀཏྟི་ར་བ་ཤིཥཱ་མཱ་དྱ་བཏ། །ཨུ་དཱི་ཙྪ་པ་ཡིན། །ད་ལས་ཕ་རོལ་ཛ་ད་ཟུང་དང་པོ། །ལྷག་མ་དང་པོ་ལྟ་བུ་སྔ་འཕེལ་ཡིན། །གང་ཚེ་ཟུང་མིན་བྲྀཏྟི་རྫ་དོ་ཡཾ་དཱ་མུ་ཁ་མ་ཡུ་ཛོཔ་རི་ཤིཥྛ་མཱ་བྱ་བཏ། །ཨུ་དཱི་ཙ་བྲྀཏྟེ་རཱ་དང་པོ་ཛ་ད་དག། ལྷག་མ་དང་པོ་ལྟ་བུ་ཕྱི་འཕེལ་ཡིན། །ཟུང་མིན་ཆ་ནི་ཕྱི་འཕེལ་ལས། །ཟུང་ཆ་སྔ་།
