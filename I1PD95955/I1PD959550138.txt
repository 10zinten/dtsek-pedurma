བྷིངེད་རྣམས་ཀྱིས། དེ་ན་ང་ཚོ་ལ་ཀམ་བE། ཚེ་ལྔ། ཏེ། ༈བཞི་པའི་གཅིག་ཚིག་ལ། གཅིག་ཉིད་ལ་ཞེས་པས། ཏཱདྲ། ངེ་། །མདྲ༑ངེ་ཞེས་གནས་པ་ལ། ་ཏུ་:བྷཾ་མ་ཧུཾ་ད་ཡཱ། ངེ་ཡི་ས་ཏུ་ཧཱུྃ་མ་ཧཱུཾ་ངོ༌། །ཡུད་དང་ཨསྨད་དག་ནི་རྣམ་དབྱེ་ངེ་ཕ་རོལ་ལ་རིམ་པ་བཞིན་རྟགས་དང་རྣམ་་་དབྱེ་ལྷན་ཅིག་པར་ཏུ་ཧཱུཾ་དང་མ་ཧུཾ་ཞེས་པར་འགྱུར་ཏེ། ཁྱོད་ཀྱི་ཆེད་དུ། ཏུཧཱུཾ། ངེ༌། བདག་གི་ཆེད་དུ། མ་ཧཱུྃ། ངེ་། །གཉིས་ཚིག་ལ། ཡུ་བཱ་ཧཱུཾ། ཁྱེད་དག་གི་ཆེད་དུ། ཨཱ་བཱ་ཧཱུཾ། ངེད་དག་གི་ཆེད་དུ། མང་ཚིག་ལ། བྷུས་ཤྩ། །བྱས་ཙྩོམོ། །ཡུཥྛདྲ་ཨ་་སྨདྲ་དག་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་བཞི་པའི་བྱུས་ནི་ཞེས་པར་འགྱུར་ཞིང་ཤ་ཡིག་ནི་བྱས་ཡོངས་རྫོགས་སྒྱུར་བ་དང་བྷ་ཡིག་ཐོག་མའི་བྱ་བ་རྣམས་འགོག་པའི་དོན་ཏེ། ཁྱད་སོགས་ཀྱི་སི་སོགས་ལ་ཊི་ཨ། ་བྱས། བྷཾ། ཤ་ཕྱིས་པས། ཡུཏྨ་ཧཱུཾ། ཁྱེད་རྣམས་ཀྱི་ཆེད་དུ། ཨ་སྨ་ཧཱུྃ། ངེད་རྣམས་ཀྱི་ཆེད་དུ། ལོ་ཟླའི་ནང་། འད། བོད་ཟླ་སྐྱེ་འགྲོ༈ལྔ་པའི་གཅིག་ཚིག་ལ། །གཅིག་ཉིད་ལ་ཞེས་པས། ཏཱདྲ། མདྲ། ཅེས་བསྒྱུར་བ་ལ། ་ཁྱད་སོགས་ཀྱི་སི་སོགས་ལ་ཊི་ཨར་བསྒྱུར་བས་ཏཱ། ང་སི། མ།  ིང་སི་ཞེས་གནས་པ་ལ། །ང་སི་བྱ་སོgསིང་སི་ཧྥུས་དག་གི་ཤུའོ། །ཡུཧྨདྲ་དང་ཨསྨད་དག་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་ལྔ་པའི་ང་སི་དང་ཧྥུས་དག་ནི་ཤུ་ཞེས་པར་འགྱུར་ཞིང་ཨུ་བརྗོད་དོན་དང་ཤ་ཡིག་ནི་ཐམས་ཅད་འགྱུར་བའི་དོན་ཏེ། ཏུཏྲ༑ཡཁྱེད་ལས། །མཏྲ། ངེད་ལས། གཉིས་ཚིག་ལ། ཡུ་བཱ་ཧཱུཾ། ཁྱེད་དག་ལས། ཨཱ་བཱ་ཡངེད་དག་ལས། མང་ཚིག་ལ། ་བྷས་བྱུར་བསྒྱུར་བས། །ཡུཥྛཏྲ། ཁྱེད་རྣམས་ལས། ཨསྨཏྲ། ངེད་རྣམས་ལས། །འདོད་པ་དང་བློ་ལུ་ར་འག་ལོ་:་ཡོ༈དྲུག་པའི་གཅིག་ཚིག་ལ། གཅིག་ཉིད་ལ་ཞེས་པས། །ཏཱདྲི། ངས། མདྲ། ངས། ཞེས་གནས་པ་ལ། །ཏ་བ་མ་མ་ང་སཱ། ངས་ཀྱིས་—།
