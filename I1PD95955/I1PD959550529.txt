ནུ་ཥཱ་ཀཾ་ནི་ཎཾ། །གྲྭ་མ་ཏཱ་ཛ་ན་ཏཱ་དྷཱ་མྱ་པཱ་ཐཱ་གཱ་ལ་པྲ་ཐཤྲཱི་ཐཀ། ཨ་པི་ཚོམས་མི་རྣམས་ཀྱི། །གྲོང་ཚོམས་སྐྱེ་ཚོམས་དུ་བ་མནོམ། །ཞག་ཚོམས་ག་ཅུ་སོ་སོའོ། །སྟོང་ཚོགས་ཤཱ་ཧ་ཤ་ཀཱ་རཱི་ཥ་བཱམྨ་ཎཱ་ཐརྦ་ཎཱ་དི་ཀཾ། །སཾ་ཀཱིཙྩཱ་བནྡྷ། །གུ་བྱུང་བའིདང་ནི་ལྕི་མེ་ཚོགས། །གོ་ཆའི་ཚོགས་དང་རིག་བྱེད་སོགས། །འདྲེས་མའི་སྡེ་ཚན་ནོ། །འཁ་། ་འཚོ༈*ནཱ་ནཱཅཱཀེ་པི་ཀཱནྟཱ་དི་བརྒེ་ཊེ་བཱ་ཏྲ་ཀཱིརྟི་ཏཱ། །ཧྥུ་རི་པྲ་ཡོ་*འདིར་ནི་སྣ་ཚོགས་དོན་གྱི་སྡེར། །བྱེད་པོ་རྣམས་ལ་དང་པོ་གསུམ། །དེ་ཡི་འཇུག་ཡུལ་རྣམས་ལ་བཞི། །སྦྱར་བས་། རྟོགས་པར་བྱ་བའོ། ༈། སྣ་ཚོགས་དོན་ལ་ཀ་མཐའ་སོགས། །སྡེ་ཚན་གྱིས་ནི་འདིར་གྲགས་སོ། །མང་པོ་ལ་སྦྱོར་གཱ་ཡེ་ཡེ་ཥཱ་པཎཱ་ཡེ་ཥཱ་པི་ཏེ་ཥཱ་ཏེ། །ཨཱ་ཀ་ཤེ་ཏྲི་དི་བེ་ནཱ་ཀོ་ལོ་ཀསྐུ་བྷུ་བ་ནེ་ཛ་གང་གང་ལ། །རྣམ་གྲངས་ལ་སྦྱོར་དེ་དང་དེ། །ནཱ་ཀ་ནམ་མཁའ་སྐབས་གསུམ་ལ། །ལོ་ཀ་ས་གཞི་སྐྱེ་བོ་ནེ༑། པ་བྱེ་ཡ་ཤ་སི་ཙ་བློ་ཀཤ་རེཁརྒེ་ཙ་ས་ཡ་ཀ། །ཛཾ་བུ་ཀོ་ཀྲོཥཱ་བ་ལ༑་། བློ་ཀ་རྐང་བཅད་གྲགས་པ་ལ། །མདའ་དང་རལ་གྲི་སཱ་ཡ་ཀ། ཅེ་སྤྱང་ཆུ་བདག་ཛམ་བུ་རུ་ཎོ་པྲི་ཐུ་ཀོ་ཙི་པི་ཌཌྷ་ཀོ། །ཨཱ་ལོ་ཀོ་དན་ནོ་བྱོ་ཏོ་བྷེ་རཱི་པ་ཊ་ཧ་མཱ་ན་ཀོ། །ཀ། སྣ་ཉེག་བྱིས་པ་སྲི་ཐུ་ཀ། མཐོང་འོད་དག་ལ་ཨཱ་ལོ་ཀ། རྔ་ཆེན་རྔ་རིངས་མཱ་ན་ཀ་། ཨུཏྠཾ་ག་ཙི་ཙྪེན་ཡོ་རོ་ཀཀ་ལཾ་ཀོཉངྒཱ་པ་བཱ་ད་ཡོ། །ཏག་ཀོ་ནཱ་ག་བརྙ་ཀྱོ་རཀཿཧྥ་ཊི་ཕང་པ་རྟགས་དག་ཨཾ་ཀའོ། །རྟགས་མཚན་ངན་སྨྲ་ཀ་ལཾ་ཀ། ཀླུ་དང་ཤིང་འཇོག་ཏཀྵ་ཀ་། ཨརྐ་ཤེལ་ཀ་སཱུརྱ་ཡོ། །མཱ་རུ་ཏེ་བེ་དྷ་སི་བྲངྒེ་པུཾ་ཕ་ཤི་ཀཀཾ་ཤི་རོཉམྦུ་ནོ། །སྩཏྤ་ལཱ་ཀyཙྪ་དང་ཉི་མ་ལའོ། །།ཀནི་རླུང་དང་ཚངས་ཉི་མ། །ཀཾ་ནི་མགོ་དང་ཆུ་དག་ལ། །པུ་ལཱ་ཀ་ནི་སྤུན་སྟོང་དང༌། །དྷ་རྩེ་སཾ་ཀྵེ་པེ་བྷཀྵ་སིཀྵ་ཀེ། །ཨུ་ལཱུ་ཀེ་ཀ་རི་ཎཔུཙྪ་མཱུ་ལོ་པཱནྟེ་ཙ་སེ་ཙ་ཀ། །བསྡུས་དང་ཟན་གྱི་སྙིགས་མ་ལའོ། །པེ་ཙ་ཀ་ནི་འུག་པ་དང་། །གླང་པོའི་མཇུག་མའི་མཐའ་ཉེ་ལའོ། །།ཀ་མཎྜ་ལོ་ཙ་ཀ་ར་ཀསུ་ག་ཏེ་ཙ་བི་ནཱ་ཡ་ཀཿ། །ཀིཥྛ་ཧསྟེ་བི་ཏསྟེ་ཙ་ཤུ་ཀ་ཀཱི་སྤྱི་བླུགས་སེར་བ་ཀ་ར་ཀ། བདེ་གཤེགས་ལོག་འདྲེན་ནཱ་ཡ་ཀ་། ཀིརྟུ་ཁྲུ་དང་མཐོ་ལའོ། །ཤུ་ཀ་སྲིན་—།
