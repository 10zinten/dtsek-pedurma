བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽་loloteདང་ར་དང་བ་དང་ན་དང་ན་དང་བ་དང་ཏ་དང་སུ་རྣམས་ཡིན་པ་དེའི་ཚེ་་་བདུན་དང་རྟ་དང་རོ་རྣམས་ཀྱིས་ངལ་བསོ་་བ་ནི་ཁ་མཛེས་སོ། །སམྦྷ་ཤཱ་སཱ་ད་བྷིནྡྷ་མ་ར་བ་ན་ན་བ་ཏཱཔུཉྩེདྨ་བ་ད་ནཱ། tTIIIཀཾ༈ར་ཛ་ར་ཛ་ར་ཛ་ལྕི་བ་ཡང་བ་འདི་ལྟ་བུ་རྣམས་ཟླུམ་པོ་ཞེས་བྱར་་་་འདོད༑། གལ་ཏེ་རྐང་པ་སོ་སོ་ལ་ར་ཛ་དག་ལ་ན་གསུམ་དང་ལྕི་བ་དང་ཡང་བ་ཡིན་པ་དེའི་ཚེ་འདི་ལྟ་བུ་ཉིད་ནི་ཟླུམ་པོའི་མིང་ཅན་ནོ། །ཟླུམ་པོ་ཞེས་བྱ་བ་ནི་ཟླུམ་པོ་ཞེས་པའི་་དོན་ཏོ། །བྲྀཏྟ་མཱི་ཏྲ་ཤནྟུ་ནཱ་མ་ཏོ་ར་ཛོ་ར་ཇོ་ར་ཇོ་གུ་རུ་་་གྷ། །མ་ར་ཝ་ན་ན་ཝ་ར་ར་ཝ་རྣམས་བདུན་པ་གསུམ་གྱིས་སྡོད་པ་ཕྲེང་འཛིན་ཞེས་བྱར་འགྱུར། །གལ་ཏེ་རྐང་པ་སོ་སོ་ལ་མ་ར་བ་རྣམས་དང་ན་ན་བ་ར་ར་བ་རྣམས་ཏེ་འདི་ལྟ་བུའི་བདུན་པ་ལ་སྡོད་པ་གསུམ་གྱིས་ཡིན་པ་དེའི་ཚེ་ཕྲེང་བ་འཛིན་པར་འགྱུར་རོ། །་ཟད། ལྟ། ::མོ་བོ་ནོ་བཉྩ་རོ་བསྟི་བྷི་རི་ཏི་བི་ར་ཏིསཔྟ་ཀིསྲསྒྱུ་ར་་སྙཏ། ༽IIགང་ལ་བྷ་ར་ན་ཝ་ཛ་ས་ཛ་སུ་ཕྱོགས་དང་ཉི་མར་ངལ་བསོ་ལོ། འདི་ནི་ཟབ་མོའི་དབྱངས་སོ། ང་། གང་རྐང་པ་སོ་སོ་ལ་བྷ་དང་ར་དང་ན་དང་བ་་དང་ཛ་དང་ས་དང་ཛ་དང་ས་རྣམས་ཡིན་ཞིང་བཅུ་དང་ཉི་མ་རྣམས་ཀྱིས་ངལ་་་བསོ་་བ་འདི་ནི་ཟབ་མོའི་དབྱངས་སོ། །ཡ་ཏྲ་བྷ་རོ་ན་བོ་ཛ་ས་ཛ་སྭཱ་ད་ཤོ་གུ་བི་རཱ་མི་མ་དྲ་ཀ་མི་དཾ། །༈ན་ཛ་བྷ་དང་ལ་ཝ་ཛ་ས་དག་དང་ཛ་སུའང་འཕྲོག་བྱེད་ཉི་མར་ཐ་དད་པ་ནི་རྟ་རོལ་པ། །གལ་ཏེ་རྐང་པ་སོ་སོ་ལ་ན་དང་ཛ་དང་བྷ་དང་ཡང་བ་དང་བ་དང་ཛ་དང་ས་དང་ཛ་དང་སུ་ཡིན་ཞིང་འཕྲོག་བྱེད་དང་ཉི་མ་རྣམས་ཀྱིས་ངལ་བསོ་སྟེབ་དེའི་ཚེ་རྟ་རོལ་པའོ། །ཧ་ར་ར་བི་བྷིནྡ་མ་ཤུ་ལ་ལི་ཏཾ་ན་ཇོ་ཙ་བྷ་ལ་པཱ་ཛ་སོ་ཙ་ཛ་སུ་ཙ། །
