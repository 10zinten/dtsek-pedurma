སྩརྨོ་ཧུནྟི་ཀ་མོ་ཧུཏྟ་ཛྷཱ་ནི་ཀཱཎྜ་ཏི་ཀཱ་ཨ་པི། །ཏཱནི་ཀོ་ཛྙཱ་ཏ་སི་དྷཱནྟསནྣི་ཙམ་ཐང་ཅིག་དང་། །དུས་ཤེས་པ་དང་རྩིས་པའོ། །རྒྱུད་དང་གྲུབ་མཐའ་ཤེས་པ་དང༌། །སྦྱིན་བདག་ཧྲ་ཧ་པ་ཏི་ས་མོ། །ལི་པི་ཀ་རོཉཀྵ་ར་ཙ་ནོཉཀཱ་ར་ཙུཉྩ༔་ལེ་ཁ་ཀེ། །ལི་ཁྱིམ་བདག་མཚུངས་པའོ། །།བྲིས་མཁན་པ་དང་ཡིག་མཁན་དང༌། །ཡི་གེ་པ་དང་ལེ་ཁ་ཀེ། །ཡི་ཁི་ཏཱ་ཀཱ་ར་སཾ་སྦྲ་ནེ་ལི་པིསྨིཟི་རུ་བྷིནྡི་ཡོ། །སpནེ་ཤ་ཧ་རོ་དྷུ་ཏོ་དྷུ་ཏུནྟ་གེའི་གཞི་དང་ཡི་གེའི་གནས། །ཡི་གེའི་རི་མོ་གཉིས་ནི་མོ། །སྐྱས་ཁ་བ་དང་ཕོ་ཉ་བ། །དྷུ་ཏྲི་དེ་ཡི་བྷཱ་བ་ཀརྨ་ཎི། །།།ཨ་དྷ་ནཱི་ནོཞདྷཱ་གོ་དྷ་ནྱཔཱཔ་ཐི་ཀ་ཨི་ཏྱ་པ། །སྣ་མྱ་ངོ་བོའི་ལས། །ལམ་པ་ཚོང་པ་འགྲོན་པོ་དང༌། །ལམ་འགྲུལ་དང་ནི་འགྲོན་ལམ་དང་། །བདག་པོ་མཱ་ཏུསུ་བྲྀཥོ་ཥོ་རཱཥཱ་དྷུ་བ་ལ་ནི་ཙ། །རཱ་ཛྙཱ་གཱ་ནི་པྲ་ཀྲྀ་བློན་པོ་ཐུགས་གཅེས་དང༌། །མཛོད་དང་ཡུལ་འཁོར་རྫོང་དང་དཔུང་། །རྒྱལ་སྲིད་ཡན་ལག་རྣམས་ཏ་ཡབོ་རཱ་ཎཱ་ཐེ་ཡོ་བི་ཏཾ་ཙ། །སནྟིརྣ་ ིབི་གྲ་ཧོ་ཡཱ་ན་མཱ་ས་རང་བཞིན། །ཁ་བ་དང་ཕོ་བྲང་གཅེས་པ་ཡང་། །དུམས་དང་འཐབ་དང་ཆས་པ་དང༌། །གནས་ན་དྲི་དྷ་མཱ་ཐ་ཡ། །།ཥདྷུ་ནཱཤཥ་ཡསྟི་ཤྲཔྲ་བ་བློཏ་ཧ་དང་ནོར་གཏོང་གཞན་བརྟེན་རྣམས། །ཡོན་ཏན་དྲུག་དང་ནུས་པ་གསུམ། །མཐུ་དང་སྤྲོ་བ་སྔགས་མནྡྷ་ཛ། །ཀླུ་ཡསྠཱ་ནཉྩ་ཤྲིདྷིཉྩ་ཏྲི་བསྒོ་ནཱི་ཏི་བེ་དི་ནཾ། ་། ས་བྲ་ཏཱ་ཡ༔བཟླས་དང་། །ཟད་དང་གནས་དང་འཕེལ་བ་དང་། །སྡེ་ཚན་གསུམ་པོ་རིག་པའོ། །དཔུང་དང་མཐུ་སྭ་བྷཱ་བཧྥ་ཡཏྟེ་ཛཀོ་ཥ་དཎྜ་ཛཾ། །བྷེ་དོ་དཎྜསཱ་མ་དཱཎྜན་མི་བརྗིད་ཅན་དང་ནི། །བང་མཛོད་དཔྱ་ཡིས་གང་བ་དང༌། །དབྱེས་དང་སྡུད་ཁ་དུམས་དང་ནི། །དཔྱ་ཏུ་པཱ་ཡ་ཙ་ཏུཥཱ་ཡཾ། ་། སཱ་ཧ་སཾ་ཏུ་ད་མོ་དཎྜསཱ་མ་སཾ་ཏུ་མ་ཐོ་ས་མོ། །འཇལ་ཐབས་ནི་རྣམ་པ་བཞི། །ཁྲལ་དང་སྡུད་དང་དཔྱ་དང་ནི། །ཕྲ་མ་སྐྱོ་མ་དེ་ནས་མཚུངས། །བྷེ་དོ་པ་ཛཱ་པཱ་བུ་ཛཱ་དྷཱ་དྷམྨོ་བྱིརླཏྲ་རིཀྵ་ཎཾ། ་། པཉྩ་ཏྲི་ཥ་ཥ་ཌ་བྷཱི་དབྱེན་དང་ཕྲ་མ་ཨུ་ཛཱ་དྷཱ། །ཆོས་སོགས་གང་ཞིག་ཡོངས་བལྟ་བའོ། །ལྔ་ནི་གསུམ་མོ་མིག་དྲུག་།
