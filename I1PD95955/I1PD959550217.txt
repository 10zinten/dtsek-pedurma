།ཀལ་ཚབསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ་་བའི་ཞཆོལ་ཁajལ་བ་རྫོགས་ལ་གཏོང་བ་ལའོ། །ཞེས་པའི་བྱིངས་ལ་རྗེས་འབྲེལ་རྣམས་ཕྱིས་རྣམ་དབྱེ་བདག་དོན་དང་གཞན་དོན་གཉིས་ཀ་སྦྱོར་བ་ལས། དང་པོ་པ་རསྨི་པ་ད་རྣམས་སྦྱར་ཆོ་ག་སྔོན་མ་ལྟར་བྱས་པས། ༈བརྟ་མ་ན་ལ། ཛ་ཧཱ་ཏི། ཛ་ཧཱི་ཏཊཱིཛ་ཧི་ཏི། ཛ་ཧཱ་སི། ཛ་ཧཱ་ཐཛ་ཧཱི་ཐ། ཛ་ཧཱ་མི། ཛ་ཧཱི་བ་ཛ་ཧཱི་མ། ་ཏོ༈ཆོག་འབྱུང་ལིང་ལ། །ཡེ་ཛ་ཧཱའི་ཨཱ་ཡིག་ཡཱ་ཕ་རོལ་ལ་དབྱི་བར་་བཤད་པས། ་ཨཱ་ཕྱིས་ནས་ཡཱ་ལ་དྲངས་པས། ཛ་ཧཱུཏྲ། ?ཛ་ཧྭ་ཏྭཾ། ཛ་བྷུསོགས་སོ། །ལ་ན་མ་ལོ་ལ་ར་ས་་་ཁཁོ་ར་ཁྲིཌ་༈ཤིས་བྱེད་ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །་ཛ་ཧཱ་ཏུ། ཛ་ཧཱི་ཏཱཏྲ། ཛ་ཧཱི་ཏཱཾ། ཛ་ཧ་ཏུ། །སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ༈ཨཱིརྦ་ཧོ། ཧི་ལ་ཨི་དང་ཨཱི་ཡང་ནའོ། །རྣམ་དབྱེ་ཧི་ཕ་རོལ་ལ་ཨོ་ཧཱཀའི་མཐའི་ཨཱ་ནི་ཨི་རུ་་བསྒྱུར་བ་དང་ཨཱི་རུ་བསྒྱུར་བ་ཡང་ན་བར་བྱ་སྟེ། ་ཛ་ཧི་ཧི། ་ཛ་ཧཱི་ཧི། ཛ་ཧཱི་ཏཱཏྲ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཛ་ཧཱི་ཏཾ། ཛ་ཧཱི་ཏ། ཕྱི་མ་་གསུམ་ལ། ཛ་ཧཱ་ནི། ཛ་ཧཱ་བ། ཛ་ཧཱ་མ། བཞི་རྒྱུ་༈དེ་རིང་མ་ཡིན་པ་ལང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ༑ཨ་ཛ་ཧཱཏྲ། ཨ་ཛ་ཧཱི་ཏཱཾ། མང་ཚིག་ལ་ཨན་ཨུས་སུ་བསྒྱུར་བས་ཨ་ཛ་ཧཱ༑ཨུསྲ་ཞེས་གནས་པ་ལ། ༈ཨུ་སཱུ་ལོ་པ། ཨུསྲ་ལ་ཨཱ་དབྱིའོ། །རྣམ་་་དབྱེ་ཨན་བསྒྱུར་བའི་ཨུསྲ་ཕ་རོལ་ལ་བྱིངས་ཨོ་ཧཱཀའི་མཐའི་ཨཱ་ནི་དབྱི་བར་བྱ་སྟེ་ཨཱ་ཕྱིས་པས། ཨ་ཛ་ཧུསྐྱེས་བུ་དབུས་མ་དང་ཕྱི་མ་ལ། །ཨ་ཛ་ཧཱཨ་ཛ་ཧཱི་ཏཾ། ཨ་ཛ་ཧཱི་ཏ། ཨ་ཛ་ཧཾ། ཨ་ཛ་ཧཱི་བ། ཨ་ཛ་ཧཱི་མ། སོགས་་་དང་དེ་རྣམས་ཀྱི་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ཀྱང་རིགས་འགྲེ་སྟེ་ཤེས་པར་བྱའོ། །་་༈དེ་བཞིན་དུ་འུ་དཱཧྲ་སྦྱིན་པ་ལའོ་ཞེས་པའི་བྱིངས་ལ་ཡང་ཐོག་མར་་་་—།
