ཚས་ཀྱི་སྒོར་ལོབ་རཎྜ་བཅིངས་པ་ཀི་ཎོ་གླུ་ཎ། །ཛྲ་པགས་པ་ཏི་སཱི་མནྟཧ་རི་ཏོ་རོ་མནྡདྷི་ཐ་བུདྨ་དཱ། །ལོ་མ་བསྡམས་པ་ཤིང་སྲིན་བུ། །ཆུ་རྐྱལ་སྐྲ་མཚམས་ལྗང་གུ་དང༌། །སྐྱུགས་ལྟར་སྒྲོགས་པ་ཆུ་བུར་དང་། །ཀཱ་སྨན་བྱེ་བྲག་ས་མརྡོཉརྡ་ནི་ཀུནྡམེ་ཏོག་ཕེ་ན་སྐུ་པོ་ས་ཟན་པུ་བ་ཀོ། །ཨཱ་ཏ་པཀྵ་ཏྲི་ཡེ་ནཱ་འཁྲི་ཤིང་འགྲོ་སློང་བཟོད་དང་ནི། །སྦུ་བ་མཆོད་རྟེན་གཏིང་ཚིག་དང༌། །ཉི་འོད་རྒྱལ་རིགས་ལྟེ་བ་དང༌། །བྷིཀ་ཎ་ཡ་གླུ་ར་ཀེ་ད་རཿ། །པཱུ་ར་ཁུ་ར་པྲ་ཙུཀཱཉྩ་གོ་ལ་རྡོག་པོར་བྱས་ཧིངྒྷལ་པུངྒ་ལཱ། །མདའ་དང་སྤུ་གྲི་གསེ་བ་དང་། །ཆུ་འཕེལ་རྨིག་པ་སྐྱུར་མོ་དང༌། །བྱི་ཕྲུག་མཚལ་དང་གང་ཟག་དང༌། །བེ་ཏཱ་ལ་བྷལླ་མཚོན་མལླི་པུ་རོ་ཌ་ཤོཉཐ་པཎྜི་སམཚོན་ཆ། །ཀུལླ་ཁོ་ར་བྷ་སཉྩི་བ་ཤ་ཀཱ་རོ་ལངས་གྱད་དང་སྟ་རི་དང༌། །མཆོད་སྦྱིན་དེ་ནས་པཎྜི་ཤ། རྩབ་མོ་དགའ་བ་དེ་བཞིན་དུ། །ཤིང་རྟ་ཊཱ་ཟན་ཟངས་ཧཔ་ཁྲུས་ཟངས་ཏཾ་གྲ་ཧ། །དྲི་ཧྲི་ནེཉརྱ་ཙ་ཁཱ་ར་ཎྱ་པཎྜ་ཤྭ་བྷཱ་ཧི་མོ་ད་ཀཾ། །བཤད་པ་སོ་སོར་འདོད། །གཉིས་དམན་ལའོ་གཞན་ཡང་མཁའ། །དགོན་པ་འདབ་མ་དོང་དང་ནི། །ཤྲཱི་ཏོ་ཚྭ་ས་རུ་དྷི་ར་མུ་ཁཱ་ཀྱི་དྲ་ནོར་དང་དཔུང་བི་ཎཾ་བ་ལ་ནོར་དང་དཔུང༌། །ཧ་ལ་ཧེ་མ་ཤུ་ཁ་བ་ཆུ་དང་གྲང་དྲོ་དང་། །ཤ་ཁྲག་ཁ་མིག་ཁུ་བ་དང་། །སྟོབས་དང་ཤོལ་དང་གསེར་ཟངས་ལྕགས། །ལཱ་ལོ་ཧ་སུ་ཁ་དུཁ་ཤུ་བྷཱ་ཤུ་བྷཾ། །ཛ་ལ་པུཥཱ་ཎི་ལ་བ་ཎ་བྱཉྫ་ནཱ་རྱ་ནུ་ལེ་པ་ནཾ། །བདེ་སྡུག་དགེ་དང་མི་དགེ་དང་། །ཆུ་དང་མེ་ཏོག་རྣམས་དང་ནི། །ལན་ཚྭ་རླུང་བྱན་ཡབ་རྗེས་བྱུག་པ། །ཀོ་ཥྛཤ་ཏཱ་དི་སཾ་ཁྱནྟཱ་བཱ་ལ་ཀཱ་ནི་ཡུ་ཏཉྩ་ཏཏ། །༑་མ་སི་སུས་བྱེ་བ་བརྒྱ་ལ་སོགས་པའི་གྲངས། །གཞན་ཡང་འབུམ་དང་ས་ཡ་དང་། །ཨ་ཙ་གཉིས་པ་ཨས་ཨི་ཨུས་ཨན་རྣ་ཡ་ད་ནཱན་མ་ཀཎྜ་རི། །ཏཱནྟཾ་ས་ལོ་པ་དྷཾ་ཤི་བྷཾ་ར་ཏྲ་པྲ་ཀ་མཐའ། །གང་ཞིག་བྱེད་པོ་མིང་ལ་ཨ་ནའི་མཐའ། །ཏྲ་མཐའ་ས་ལ་ཉེ་སྔོན་ལྷག་མ་དང་། །ཐོག་མར་སཾ་ཁྱུ་ཡཱཉནཱི་ཏཾ། །བཱ་ཏཱ་དྱ་དནྟི་རེ་ཀཱཟོདྲི་གྲངས་དང་ལྡན་པའི་རཱ་ཏྲ་དང་། །སྣོད་ལ་སོགས་པ་ཨ་ཏ་མཐའ་རྣམས་ཀྱི་དོན། །གཅིག་པའི་བ་གཉིས་གུརླ་ཀྵ་ནུ་སཱ་ར་ཏ། །ཏཱཾ་དྲི་ཀ་ཏྭཱ་བྱཱ་ཡི་བྷཱ་བོ་བ་ཐཱ༔སཾ་བྷྱཱཉ་བྱ་མཚོན་བྱའི་རྗེས་འབྲང་ནས། །ཟླས་དབྱེའི་གཅིག་ཉིད་མི་ཟད་གྱུར་པ་དག། མི་ཟད་གྲངས་ལས་ཕ་རོལ་—།
