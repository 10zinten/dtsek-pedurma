ི་ཁཚོ་ཞཚོལ་བ། ཁ་ན་ཌ་ཁྲོ་སྨན་འཇུག་སྟེ། ་བྱིངས་མཐའི་ཨ་སཱ་:རི་ཏ་སྟེ་དེའི་དོན། ཙུའི་ཀུའོས། ::ཛ་གར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། བྷོཀྵ་ཏི། བྷོ་གྱུ་རཿབྷོ་གྷཱནྟི་སོགས་་དང༌། བྷོཀྵ་ཏེ། བྷིཀྵེ་ོཏེ། ་བྷོpནྟེ་སོགས་སོ། །།ཨི་ཏི་སྭ་ར་སྭསྟི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱདྡྷ་བི་ཥྛནྟིཔྲ་ཏྱཱ་ཡ་སྱ་པྲ་ཀྲི་ཡ་ས་མཱཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་འབྱུང་འགྱུར་གྱི་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ། །ང་བས་། པ། དྲ་བ་ལ་ད་མཇུག་ན་ར་ལ་ལ་་མདྲ་ལ་ཊ་སྤུ་ག་བ། །ཀུན་ཏོ་རྗེ་༈དེ་ནས་བྱ་བ་ཤོར་བའམ་མ་འཛོམ་པའི་དོན་གྱི་རྣམ་དབྱེ་རྣམས་ཀྱི་ཆོ་ག་རྣམས་བཤད་པར་བྱ་སྟེ། ༈སྱཔ་ཀྲི་ཡཱ་ཏི་ཀྲ་མེ། བྱ་བ་ཤོར་བ་ལ་སྱཔྲའོ། །བྱ་བ་ཤོར་བ་སྟེ་དུས་ལ་མ་ཁེལ་བའམ་རྒྱུ་ཚོགས་མ་འཛོམ་པ་དང་། བྱ་བ་མ་་་རྫོགས་པར་འཕྲོ་ལུས་པ་དང་། བྱ་བ་ཟིན་མ་ཐག་པའི་གྲུབ་ཟིན་གསར་བ་རྣམས་ཀྱི་བྱ་བའི་དོན་ལ་རྣམ་དབྱེ་དིཔྲ་ལ་སོགས་པ་བཅོ་བརྒྱད་པོ་འཇུག་ཅིང་བར་རྐྱེན་་སྱཔྲ་ཀྱང་སྦྱིན་པར་བྱ་སྟེ། །པཱ་ཎི་པའི་ལུགས་ལ་ལྡིང་ཞེས་ཟེར་རོ། །དེ་ལྟ་ན་བྷུ་ཡོད་པ་ལ། དིཔྲི། །་བར་རྐྱེན་སྱཔྲ། དིཔྲ་སོགས་ལ་ཨཊ། སི་ས་ཏཱ་སཱི་་་་སྱཔ་རྣམས་ལ་ཨིཊ་འོས། ཨིཊ་ཨཱ་ག་མ་ཡོན་ཏན་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་བྷ་བིཏྲི། དེ་བཞིན་དུ་ལྷག་མ་རྣམས་ལ། །ཨ་བྷ་བི་ཏཱཾ། ཨ་བྷ་བི་་་ན། ཨ་བྷ་བི་ཥྱ་ཨ་བྷ་བི་ཥྛ་ཏཾ། ཨ་བྷ་བི་ཥྛ་ཏ། ཨ་བྷ་བི་ཧཱུྃ། ཨ་་བྷ་བི་བྷཱ་བ། ཨ་བྷ་བི་བྷཱུ་མ། ཉེ་བའི་ཚིག་དང་སྦྱར་བའི་དཔེ་ནི། །གང་དུ་ཆར་་་བབས་ཤིང་རྒྱལ་པོ་བདེ་བར་གྱུར། ཡཱ་དྨ་སུ་བྲྀཏྟིསུ་རཱ་ཛཾ་ཙཱ་བྷ་བི་ཏྲ། གྲུབ་ཟིན་གསར་བའི་དཔེ། དེ་དག་ཏུ་སློང་མོ་བཟང་པོ་ཐབ་པར་འགྱུར། ཏ་དཱ་སུ་བྷི་ཀླུ་མ། བྱ་བ་མ་རྫོགས་པའི་དཔེ། བྷ་བི་ཏྲ། པ་དང་༈དེ་བཞིན་དུ་བྷུཛྲ་སྐྱོང་བ་དང་ཟ་བ་ལ། ་ཨ་བྷོ་གྱུཏྲ། ཨ་བྷོ་ཏྭཾ། ཨ་བྷོན་སོགས་སོ། །བླ་མ་དག་ལ་ཤེ་ཡོན་ཏན་ལ་བློ་བག་ག་ས་ཕྱོགས་ཏེ་ཡི་གེ་བྱུང་།
