ཁ་ཥ་་་ཀྱི་ཎ་་བསྟན་འགྱུར། །སྒྲ་མདོ། ༼སེ༽པ་ཁ་ཏོ། ཀ་ཀ་བྷོ་ཏྟཥ་བ་ཕ་རོལ་ལ་ཨོ་རུ་བསྒྱུར་བ་དང་ཡང་ན་ཨོ་རུ་འགྱུར་ཞིང་ཨོ་རུ་བསྒྱུར་བའི་ཕྱོགས་ལ་ཨཱིར་ཨཱ་ག་མ་སྟེར་བར་བྱ་སྟེ། རྦ་རྗེས་འབྲེལ། །ཨོ་དང་ཨོ་འདམ་ཀ་ཅན་་་དུ་བྱས་པས། ༈ཡུ་སྤེལ་བ་ལ། ཡོ་ཏི་སོགས་སམ། ཡ་བཱི་ཏི་སོགས་སོ། །་་::༈དེ་བཞིན་དུ་ནུ་དེད་པ་ལ། ནོ་ཏི་སོགས་སམ། ན་བཱི་ཏི་སོགས་སོ། །་:༈རུ་སྒྲ་ལ། རོ་ཏི་སོགས་སམ། ར་བཱི་ཏི་སོགས་སོ། །།།།་སློབ་མ་བློ་༈བྷུཉྫ་བསྟོད་པ་ལ། སྟོ་ཏི་སོགས་སམ། །སྟ་བཱི་ཏི་སོགས་སོ། །ང་འགྲོ་བ་རྣམས་ལ་བྱ་བ་ལྷག་པོ་མེད་པས། ཡུ་ཏི་ཡུ་བནྟི་སོགས་སོ། །ཟབ་༈ལོ་པ་སྣ་ནུ་དཱཏྟ་ཏ་ནཾ། ཏན་རྣམས་ཀྱི་ཨ་ནུ་དཱཏྟ་དབྱི་བ་ཉིད་དོ། །བྱིངས་ཏ་ནུ་ལ་སོགས་པ་རྣམས་ཀྱི་མཐའི་ཨ་ནུ་དཱཏྟ་སྟེ་ཉམ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་རྣམས་ནི་ཀྲ་དྲ་འགྲོ་བའི་མིང་ཅན་ཛྷ་སའི་ཡི་གེ་དང་པོ་ཕ་རོལ་ལ་དབྱི་བར་བྱ་བ་་་ཉིད་དེ། ༈ཧན་འཚེ་བ་ལ། ཧནྟི། ཧ་་སྨྲ་ཏི་སོགས་སོ། །༈ཨ་། ད་ལ། ༈ན་མནམ་ལའོ། །།ནམ་བི་ཀཱ་ར་ཎ་ཕ་རོལ་ལའང་བྱིངས་ཀྱི་ཉ་མའི་ཡི་གེ་རྣམས་དབྱི་བར་བྱ་སྟེ། ༈ཧིདྨ་འཚེ་བ་ལ། རུནྡྷ་སོགས་ཀྱི་ནམ་སྦྱར་དྲ་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཧན་སི། ཧིVཧིདྨནྟི་སོགས་སོ། །།ཁ་༈བྷཛྲ་བསྙེན་པ་ལ། ནམ་བི་ཀཱ་ར་ཎ་སྦྱར་བས། །བྷ་ནཀྟི། བྷཾཥྛགྷོ་་་KềKTTKXITRUNGQUỐCGIATINH༈ཨ་པི་རཉྫ་དཾ་ཤ་ཁཾ་ཛ་བྷཾ་ཛཾ། ཨཔྲ་ལ་རཉྫ་དཾ་ཤ་ཥཌུ་ཥཱཉྫ་རྣམས་ཀྱིའོ། །ཨཔྲ་བི་ཀཱ་ར་ཎ་ཕ་རོལ་ལ་བྱིངས་རཉྫ་ལ་སོགས་པ་འདི་རྣམས་ཀྱི་ཨུ་པ་དྷཱའི་ན་ཡིག་ནི་དབྱི་བར་བྱ་སྟེ། རཉྫ་ལ། ར་ཛ་ཏི། ར་ཛ་ཏར་ཛནྟི་སོགས་དང་། ང་ཚོ། །རབ་༈དེ་བཞིན་དུ། ད་ཤ་ཏི། ས་ཛ་ཏི། སྭ་ཛ་ཏི་སོགས་སོ། བི་དོན་བཱ་ནཾ་ཏ་དཱི་ནཱ་ཎ་བཱ་དིནཱ། ཚ་བིང་ལས་ཏིཔྲ་ལ་སོགས་པ་དགུ་རྣམས་ནི་ཎཔྲ་ལ་སོགས་，—།
