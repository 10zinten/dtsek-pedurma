པཉྩནི་ཁ་གླིང་དང་། བི་ཌ:དྲི་ཆེན་ནོ། །ལོ། ཀ། ད་དུང་ས་ཁ་མཐོ་པ་སེལ་ཏེ། བོད་ཀྱི་དེ་ནས་ཐོད་པའི་མིང་ལ། ཀ་རོ་ཊི་དང༌། ཀརྨ་ཊ་དང་ཀ་པཱ་ལ་་་རྣམས་སོ། །ལྟ་བ་དེ་ཉིད་ཀྱི་འཁྲིག་པའི་དཀྱིལ་དུ་ཁ/རུས་པ་ཙམ་ལ་མིང་གསུམ་སྟེ། ཀཱི་ཀ་སཾ་དང་། ཀུ་ལྕི། གདུང་ངམ་རུས་དང་། ཨསྦི། རུས་པའོ། །།ཚེ་བ་ཞུ། །ཕྱིཀ་ལེ་བ་ར་ལ་སོགས་པ་བཅུ་གཉིས་ནི་ལུས་སྤྱིའི་མིང་ངོ༌། །།དེ་ལ་ཀ་ལེ་བ་ར་འམ་ཀ་རེ་བ་ར་ནི་ཁོག་པའམ་ལུས་སོ། །གཱ་ཊཾ?གོང་བུའམ་ལུས་དང༌། བ་སུ་འདུས་པ་དང༌། སཾ་ཧ་ན་ནཾ། ཚོགས་པ་དང༌། ཤ་རཱི་རཾ། ཐོས་བྱེད་དང་། །བརྞ། གཟུགས་པོ་དང་། བི་གྲ་ཧ། རྣམ་འཛིན་དང༌། ཀཱ་ཡ་ལུས་དང༌། དེ་ཧ། །ཁྱབ་པ་དང་། མཱུརྟི། ལུས་དང་། ཏ་ནུ་དང་། ཏ་ནུ་ཡང་ལུས་ཏེ། སྔ་མ་ནི་ཏ་ནུ་རྒྱས་པ་ལ། །མེ་རུའི་སྒྲ་བཞིན། ཨུ་རྐྱེན་དང་། །ཕྱི་མ་ནི། ཙ་མི་ཏ་ནི་ཕ་དྷི་རྣམས་ལས་ཨུའོ་ཞེས་པས་ཨུ་རྐྱེན་དུ་བྱས་པའོ། །གལ་ཏེ་ཁྱེད་ཀྱི་ཚལརྐང་པ་ལ་མིང་བཞི་སྟེ། པཱ་ད། བ་ད། །འགྲོ་བར་བྱེད་པས་རྐང་པ་དང་། ཙ་ར་ཎ། ་རྒྱུ་བྱེད་དང་། ཨོཾ་ཤྲཱི་འགྲོ་བྱེད་དེ། ཨ་ཧི་འགྲོ་བ་ལ༑མང་པོ་ལས་སོ་ཞེས་པས་ཀྲིནའོ། །འཇིག་རྟེན་དབང་ཕྱུག་གི་བསྟོད་པ་བརྒྱ་པར། ཨོཾ་ཧྲི་ཞེས་ཧ་ཡིག་གི་རྗེས་ཁྲིད་ཡོད་དེ། གླེ་ཤ་བྷུ་ཧཱ་ཧི་ཏེ་ཧཱ་ཧུ་ཏི་བི་ཧི་ཏ་ཤྲཱི་ཧ་ཏྲེ་ཧི་དུཥྱགི་དཱ་ཧཱ་བཱ་ཧཱི་ནཱི་སྣདྨ་ཧུ་བི་ཧི་ཏ་ཧི་ཏཱ་པདྨ་ཧ་ཙཱ་བྲི་བྷ་བ༔ཞེས་འབྱུང་ངོ༌། །་་་་་པ། བུད་མེད་ཀྱི་རྟགས་ལ་མིང་གསུམ་སྟེ། ་ཨུ་པ། ཉེར་གནས་དང་། བྷ་གཆ་བའམ་སྐལ་བ་དང་། ཡོ་ནི་སྐྱེ་གནས་སོ། །ཀཥ། ཤོ་ངོ་། །_།
