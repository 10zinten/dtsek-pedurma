།ཀ། །ཚད་མས་དོན་ལ་བལྟ་བ་ལ་མིང་གསུམ་ཏེ། ཙརྱ་སྤྱོད་པ་དང༌། ས་མ་ཁྱབ་བཀྲང་བ་དང་། བི་ཙ་ར་ཎ། །རྣམ་པར་དཔྱོད་པའོ། །།།།ཀ་ལྷག་པར་བརྗོད་པ་ལ་རྐང་པ་གཅིག་སྟེ། །ཨ་དྷ་དྷཱ་ར་ལྷག་པར་བརྗོད་པ་སྟེ། མ་ཚངས་པ་ཁ་བསྐང་བའི་ཕྱིར་དུ་ལྷག་པའི་ཚིག་བརྗོད་པའོ། །དརྐ་ནི་རྟོག་སྟེ། །ཀྲྀ་ཏཱི་གཅོད་པ་ལའོ། །ཞེས་པའི་ཁམས་ལ་གྷ་ཉ་རྐྱེན་དང་། ཡི་གེ་བཟློག་པས་གྲུབ་བོ། །ཨཱུ་ཏ་ནི་རྟོག་པའོ། །་། །།བ་པ་ཐེ་ཚོམས་ལ། མིང་བཞི་སྟེ། བི་ཙི་ཀིཏྲ། ཐེ་ཚོམ་དང་། སཾ་ཤ་ཡ། སོམ་ཉི་དང་། སཾ་ཧེ་ན། ནེམ་ནུར་དང༌། དྭ་ས་ར། ཡིད་གཉིས་སོ། །།གང་ཐེ་ཚོམ་བྲལ་བའི་ཡིད་གཉིས་ལ། ནིརླ་ཡ་ནི་གཏན་ལ་འབེབས་པ་དང་། མ་ཐ་སྭཱ་ཎོསྦྲ། རི་ཏི་པྲ་རཱི། ཏི་ནི་རླ་ཡ་ནཾ། ཞེས་བྱ་སྟེ། ཇི་ལྟར་གནས་པ་དང༌། གནས་པ་ཉིད་དུ་ཡིད་ཆེས་པས་གཞི་ངེས་པའོ། །ནི་ཡ་ནི་ངེས་པ་སྟེ། ངེས་བཟུང་གི་བདག་ཉིད་དུ་ཡིད་ཆེས་པའོ། །ས། ལ་དེ་ལ་ཡང་། །།:ལས་དང་འབྲས་བུ་ལ་སྐུར་བ་འདེབས་པའི་ལྟ་བ་ལས་མིང་གཉིས་ཏེ། །མི་བྷ་ཏྲཊྚི་སོགས་ལོག་པར་ལྟ་བ་སྟེ། བདེན་པ་མ་ཡིན་པ་ཉེ་བར་དམིགས་པའོ། །ནཱསྟི་ཀ་ཏཱ། མེད་པ་སྟེ་སྦྱིན་པ་མེད་དོ། །སྦྱིན་བསྲེགས་མེད་དོ། །མཐོ་རིས་མེད་དོ། །ཐར་བ་མེད་དོ། །ཅེས་འདོད་པའོ། །བྱཱ་པཱ་ད་གནོད་སེམས་དང་། །ཏྲེ་ཧ་ཙནྟ་ནམ་འཚེ་བའི་བསམ་པ་འམ་འཁུ་བའི་བསམ་པ་སྟེ་སེམས་སོ། །དང་། འཁྲུལ་པ་ལ་མིང་གསུམ་སྟེ། བྷཱནྟི་འཁྲུལ་པ་སྟེ། ཐག་པ་ལ་སྦྲུལ་དུ་ཤེས་པ་ལྟར་བུའོ། ཚེ་འ། བྷ་མུ་གཡོ་བ་ལ་ལྟེན་དང་རིང་བའོ། །མི་ཤྩ་ལོག་པ་དང་། ཨ་ཏི་བླ་མ། ་ཤིན་ཏུ་འཁོར་བའོ། །།ལྟས་དེ་དག་གང་ཞེ་ན། GC།
