ས་རི་ཏིསི་ཡཱམ། །བི་དྷཱ་བི་དྷོ་པྲ་ཀ་རེ་ཙ་སཱ་དྷུ་རམྨེ་པི་ཙ་ཏྲི་ཏུ། །བ་དྷཱཛཱ་ཡཱསྐུ་ཆ་ཕྲན་ལ་འོ་མོ། ཟི། བི་དྷཱ་བི་དྷི་རྣམ་པའོ། །སཱ་དྷུ་དགའ་དང་ལེགས་པ་ལ། །བ་དྷུ་བུད་མེད་ཥཱསི་ཙ་སུ་དྷཱ་ལེ་པོཉམྲྀ་ཏཾ་སྦུ་ཧཱི། །སཾ་དྷཱ་པྲ་ཏི་ཙྪཱ་མནྡཱ་དཱ་ཤུདྡྷ་སཾ་པྲ་མནའ་མ་ཆུང༌། །སུ་དྷཱ་བྱུག་པ་བདུད་རྩི་སྔོན། །སཾ་དྷཱ་དམ་འཆའ་ཡུལ་ཆོས་སོ། །ཤཱདྷཱ་བློ་དང་ཏྱ་ཡསྤྲོ་ཧཱ། །མ་དྷུ་མ་བྱེ་པུཏྤ་ར་སེ་ཀློ་དྲེཉནྡྷནུ་ཏ་མ་སྱ་པི། །ཨ་ཏལྡི་ཥཱ་འདོད་པའོ། །མ་དྷུ་ཆང་དང་མེ་ཏོག་བཅུད། །སྦྲང་རྩིའོ་ཨནྡྷ་ལོང་བ་མུན། །འདི་ནས་གསུམ་མ་མུནྣདྷོ་པཎྜི་ཏཾ་མ་རྱ་གཟི་ཏོ། །བྲཧྨ་བནྡྷ་ར་དྷི་ཀྵེ་པེ་ནིརྡེ་ཤེཉཐཱ་བ་ལམྤི་ཡིན་ས་མུནྣ་དྷོ། །མཁས་པའི་ང་རྒྱལ་ཁེངས་པའོ། །བྲཧྨ་བན་དྷུ་གཡོ་ངེས་བསྟན། །བརྟེན་དང་ཏ༑། ཨ་བི་དྲུ་རོ་པྱ་བཋབྷོ་པྲ་སིདྡྷོ་ཥཱ་ཏ་བྷུ་ཥི་ཏོ། །དེ་ད་ལྗོལ་རིང་མིན་ཨ་བཧྥདྡྷཾ། །པྲ་སིད་དྷོ་ནི་གྲགས་དང་རྒྱན། །དྷ་མཐའོ། །ལེ་ལྔ་དང་བམ་ཚོད་དང་བརྒྱ་བར་སཱུརླ་བ་དྷཱི་ཙི་ཏྲ་བྷཱ་ནུ་ས་ཧ་སྲ་རསྨི་དེ་བཱ་ཀ་རོ། །བྷུ་ཏཱཏྟཱ་ནོ་:དྷ་ཚིགས་དེ་ཙི་ཏྲ་བྷཱ་ནུ་ཉི་མ་མེ། །འོད་སྟོང་ཅན་དང་ཉིན་མོར་བྱེད། །བྷཱུ་ཏཱཏྨ་ནོ་གསོ་བྱེད་ལུས། །ཧོ་མུཎ་ནཱི་ཙོ་པྲི་ཐསྔ་ནོ། །གྲྭ་བ་ཎོ་ཤི་ལ་པཱ་ཥཱ་ནོ་ས་ར་པཀྱི་པ་ཏྲི་ཌོ། །ཏ་རུ་ཤི་ལོ་ཐནྡྲ་ན་རྨོངས་དང་དམན། །གྲྭ་བཱ་ཎོ་ནི་རི་དང་རྡོ། །བྱ་དང་མདའ་ལ་པ་ཏྲི་ཌོ། །ཤི་ཁ་རི་ཎོ་ཤིང་དང་ཤི་ཁ་རི་ཎོ་ཤི་ཁི་ཎོ་བ་སྦི་བརྙི་ཎོ། །པྲ་ཏི་ཡཏྟཱ་བུ་བྷོ་ལིམྨོ་པ་གྲ་ཧཱ་པ་ཐ་སཱ་དི་ནོ། །བྲག་། མེ་དང་རྨ་བྱ་ཤི་ཁི་ཎོ། །པྲ་ཏི་ཡཏྠོ་བརྣབ་སེམས་དང་། །ཉེར་འཛིན་དེ་ནས་སཱ་དི་ནོ། །རྡོ་སཱ་ར་ཐི་ཧ་ཡཱ་རོ་ཧོ་བཱ་ཛི་ནོ་ཤཱི་ཏུ་པཀྵི་ཎ། །ཀུ་ལེ་པྱ་བྷི་ཛ་ནོ་ཛརྨ་བྷཱུ་ཁ་ལོ་སྒྱུར་དང་རྟ་པའོ། །བཱ་ཛི་ནོ་རྟ་མདའ་དང་བྱ། །ཨ་བྷི་ཛ་ན་རི་དྭགས་ལ་ཡང་། །སྐྱེ་བ་དང་ནི་སྐྱེ་མཱ་བ་པྱ་ཐ་ཧཱ་ཡ་ནཱ། །བནཱ་རྩི་ཤྲཱི་ཧི་བྷེ་དཱ༔་ཙནྡྲ་ལྕཀ་བི་རོ་ཙ་ནཱ། །ཀེ་ཤེ་བའི་ས། །དེ་ནས་ཧཱ་ཡ་ན་ཞེས་པ། །ལོ་དང་འོད་ཟེར་འབྲུའི་བྱེ་བྲག། བི་རོ་ཙ་ན་ཟླ་མེ་ཉེ། །སྐྲ་འཁྱིལ་ཤྲཱི་བི་ཤྲཱི་ཛི་ནོ་བི་ཤྭ་ཀརྞརྐ་སུ་ར་ཤིལླི་ནོ། །རེ་། ཨཱཏྨ་ཡཏྟཱ་ཛྷ་ཏིཇུདྷིཛི་ན་ཡིན་ནོ། །བི་ཤྭ་ཀརྨཱན་ཉི་མ་དང༌། །ལྷ་ཡི་བཟོ་བོ་དག་ཡིན་ནོ། །ཨཱཏྨན་འབད་པ་འཛིན་དང་བློ། །།
