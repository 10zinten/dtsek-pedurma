ཨཱ་ར་(ཏུ་ཁསྒྲོལ་ཎ་ལོ་མ་ཏྲཱཉྩ་བྷཱ་ཛ་ནཾ། །དབྦིཀམྦཁ་ཛཱ་ཀཱ་སཏྟརྡརྡ་རུ་ཧསྟ་ཀ། །ཨནྡྲི་ཤཱ་ཀཾ་ཧ་རི་ཏ་སྣོད་དང་ཨ་མ་ཏྲ་སྣོད་སྤྱད། །སྐྱ་བ་སྲུབ་བྱེད་ཁ་ཛཱ་ཀཱ། །གཟར་བུ་དང་ནི་ལག་སྐྱོགས་སོ། །།མོ་མིན་ཀཾ་ཤི་གྲ་ར་སྱ་ནཱ་ལི་ཀཱ། །ཀ་ཎམྤ༔་ཀ་ལམྤ༔་བེ་ཤ་བཱ་ར་ཨུ་པསྐ་ར། །ཏིནྟི་ཊཱི་ཀཉྩ་ལོ་མ་སྔོ་ལོ་མ། ་། ་ངར་བ་སྡོང་བུ་ཆུ་བའོ། །སྤྱོད་དང་ཁ་དོག་སྒྱུར་བྱེད་དོ། །ཏིནྟི་ལ་ཀ་བསེ་ཡབ་ཙུ་ཀྲྀ་ཙ་བྲྀཀྵ་སླ་མ་ཐ་བེལླ་ཛཾ། །མ་རི་ཙི་ཀོ་ལ་ཀཾ་ཀྲྀཥཱ་མུ་ཥ་ནཾ་དྷརྨ་པཏྟ་ནཾ། །ཛཱི་དང༌། །ཤིང་སྐྱུར་དེ་ནས་ཕོ་བ་རིས། །མ་རི་ཙམ་དང་ཀོ་ལ་ཀཾ། །ནག་པོ་དྲོད་སྨན་ཆོས་ཀྱི་སྨན། །ཛཱི་ར་ཀོ་ཛ་ར་ཉཛ་ཛཱི་ཀ་ནཱ་ཀྲྀཥྛ་ཏུ་ཛཱི་ར་ཀེ། །སུ་ཥ་བཱི་ཀཱ་ར་ཤྲཱི་པྲྀཏྟི་པྲ་ཐུཀཱ་ལོ་པ་ར་ཛི་ར་ཨ་ཛ་ཛཱི། །ཟེགས་མ་ཟི་ར་པུཤྲཱིའོ། །ཀཱ་ར་བཱི་དང་ས་དང་ནི། །རྒྱ་ཆེན་ནག་པོ་ཟི་ར་ནག། ཀྲི་ཀཱ། །ཨརྔ་ཀཾ་ཐིངྒ་བེ་རཾ་སྭཱ་ད་ཐ་ཙྪཏྟཱ་བི་ཏུནྣ་ཀཾ། །ཀུསུམྦུ་རུ་ཙ་དྷ་ནྱ་ཀ་མ་ཐ་སྒེའུ་གཤེར་དང་ནི་རྭ་ཅན་ནོ། །དེ་ནས་གདུགས་ཅན་འུ་སུ་དང་། །།།ཀུམཱུམྦུ་རུ་སོ་བ་ཅན། །དེ་ནས་སྒ་ཤུཏྟཱི་མ་ཧོ་ཥ་དྷཾ། །:·སྣེ་ན་པུརྨ་ཀ་ཡོ་བི་ཤྭ་ན་ག་རཾ་བི་ཤྭ་བྷི་ཥ་ཛཾ། །ཨཱ་ར་ཎཱ་ལ་ཀ་དང་སྨན་ཆེན་དང་། །མོ་དང་མ་ནིང་ཐམས་ཅད་ལ། །གྲོང་སྐྱེས་མ་ནིང་བི་ཤྭ་སྨན། །།རྩབས་མོ་སོ་བཱི་ར་ཀུལླེ་ཥ་བྷི་ཥཱ་ཏཱ་ནི་ཙ། །ཨ་བནྟི་སོ་ཥ་བཱ་རཱུམ་ཀུཉྫ་ལཱ་ནི་ཙ་ཀིཾ་ཛི་ཀཾ། །ས་སྐྱུར་མོ་ཀུལླེ་ཥ། །ཐུག་སྐྱུར་སྐྱུར་རྩི་ཨམ་ལ་དང༌། །སྐྱུར་སྦྱར་དང་ནི་ཀཾ་ཛི་ཀཾ། །སྟོང་འབིགས་ཧ་ཤ་བེ་དྷཱི་ཛ་ཏུ་ཀཾ་བཱ་བྷཱི་ཀཾ་ཧིདྨ་རཱ་མ་ཎཾ། །ཏཏྲ་ཏཱི་ཀཱ་ར་བཱི་པྲྀཏྟི་བཱཐི་ཀཱ་ཀ་ཤིང་ཀུན་མེ་ལྡན་དང་། །ཧིགླུ་དང་ནི་ཤིང་ཀུན་ནོ། །དེ་ཡི་ལོ་མ་ཀཱ་ར་བཱི། །ཟི་ར་ནག་པོ་བཤྲཱི་ཀཱ། །ཀ་ར་བཱི་པྲ་ཐུ། །ནི་ཤཱ་ཧཱ་ཀཱཉྩ་ནཱི་པཱི་ཏཱ་ཧ་རི་དཱ་བ་ར་བཎྜི་ནཱི། །།།སཱ་མུ་དྲི་ར་བཱི་དང་རྒྱ་ཆེན་ནོ། །མཚན་མོའི་མིང་ཅན་ཡུང་བ་སེར། །ཧ་རི་དཱ་དང་ཡང་བའོ། །རྒྱ་མཚོ་གང་ཡཏྟ་ལ་བ་ཎ་མ་ཀཱི་པཾ་བ་སི་རཉྩ་ཏཏ། །སིནྡྷ་བོནཎི་སཱི་ཏ་ཤི་ཐཾ་མཱ་ཎི་མནྡཱ་ཉྩ་ཞིག་ལན་ཚྭ་ཚ། །དཔྱིད་དུས་ཆགས་པ་དེ་དང་དེ། །མཚོ་སྐྱེས་མོ་མིན་དཀར་ཞིང་ཞི། །མཱ་ཎི་མནྡྲ་སིནྡྷ་ཛེ། །རོ་མ་ཀེ་བ་སུ་ཀ་པཱ་ཤཾ་བི་ཌཉྩ་ཀྲྀ་ཏ་ཀེ་དྭ་ཡཾ། །སོ་བརྩ་ལེཉཀྟ་རླུ་ཙ་ཀེ་སིནྡྷཱ་སྐྱེས། །རོ་མ་ཀེ་དང་སྦྱར་ཚའོ། །སྨིན་ཚ་ཁ་རུ་སྦྱར་ཚ་གཉིས། །ཚ་ནི་མི་ཟད་རཱུ་ཙ་ཀེ། །—།
