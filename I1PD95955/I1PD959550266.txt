།དབྱི་བར་བྱ་སྟེ། །སྲ་ཕྱིས་པ་ལ་ཎ་འགྲོ་བ་འོས་འཕེལ་བ་བྱས་པས། ཨ་བཱ་དཱིཏྲ། དེ་བཞིན་དུ་སིཔ་ལ་ཨིཊ་དང་ཨཱིཊའི་བར་གྱི་ས་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ལྷག་མ་རྣམས་ལ། །ཨ་བཱ་དིཏྟཾ། ཨ་བཱ་དི་ཥཱཨ་བཱ་དཱི་ཨ་བཱ་དིཝཾ། ཨ་་བཱ་དི། ཨ་བཱ་དི་ཎཾལ། །ཨ་བཱ་དི་ཥཱི། ཨ་བཱ་དི། ཀློཁ་བའི་ཁྲོད། འོས། ༈ཉེར་བསྒྱུར་པྲ་སྔོན་མའི་པྲ་ཛྲ་རྐུ་བ་ལའང་དེ་དང་འདྲ་བར་བསྒྲུབས་་་པས༑། པྲ་བཱ་:ཛཱི་ཏྲ། པྲ་བཱ་ཛིཙྪཱ། པྲ་བཱ་ཛི་པྲ་བཱ་ཛཱི%སོགས་སོ། །ན་པ། ༈གྲཧྲ་ཉེ་བར་ལེན་པ་ལ། །ཆོ་ག་གཞན་རྣམས་སྔ་མ་དང་འདྲ་བ་ལ་་་བྱིངས་ཧ་ཡིག་དང་མ་ཡིག་དང་ཡ་ཡིག་རྣམས་ཀྱི་མཐའ་དང་། །ཀླུ་ཎ། ཤཱས། ཛ་གྲི། ཧས་རྣམས་ལ་སི་ཕ་རོལ་ལ་འཕེལ་བ་མི་བྱེད་པར་བཤད་པས། འཕེལ་བ་མི་བྱེད། ཨ་གྲ་ཧཱིཏྲ། ཨ་གྲ་ཧིཏྟཾ། ཨ་གྲ་ཧི་ཥཱ། ཨ་གྲ་ཧཱིསོགས་སོ། །པོ་བཏང༌། ༈ཥཧྲ་ཡིད་དགའ་བ་ལ། ཨ་ས་ཧཱིཏྲ། ཨ་ས་ཧིཏྟཾ། ཨ་ས་ཧི་ཥཱ། ཨ་ས་ཧཱིསོགས་དང༌། ཨ་ས་ཧི། ཨ་ས་:ཧི་ཥཱིཏྟཾ། ཨ་ས་ཧི་ཥཏ་སོགས་སོ། །མ་ད༈ཤ་མ་དང་དྭ་མ་ཉེ་བར་ཞི་བ་ལ། ཨ་ཤ་མཱིཏྲ། ཨ་ཤ་མིཏྟཾ། ཨ་ཤ་མི་སོགས་དང་། ཨ་ད་མཱིཏྲ། ?ཨ་ད་མིཏྟཾ། ཨ་ད་མི་སོགས་སོ། །ཁྱ༈ཀྲ་མུ་རྐང་པ་བསྐྱོད་པ་ལ། ཨ་ཀྲ་མཱིཏྲ། ཨ་ཀྲ་མིཏྟཾ། ཨ་ཀྲ་མི་ཎཿསོགས་སོ། །ཤེ་བ་/ངོ་ལོག་པ། ༈ད་ཡ་འགྲོ་བ་ལ། ཨ་ད་ཡིཏྲ། ཨ་ད་ཡིཏྟཾ། ཨ་ད་ཡི་སོགས་སོ། །༈ཀྱི་ཎ་འཚེ་བ་ལ། ཨཀྵི་ཎཱིཏྲ། ཨཀྵི་ཎི་ཧཱུྃ། ཨཀྵི་ཎི་སོགས་སོ། །ཕྱིན་པ། ༈ཤཱས་སྲོག་འཛིན་པ་ལ། །ཨ་ཤཱ་སཱིཏྲ། ་ཨ་ཤཱ་སིདྡྷཾ། ཨ་ཤཱ་སི་ཥཱསོགས་སོ། །ང་ཚོ་ལསྤྱིའི་འདོ། །ཏེ་དེ་ས་ཕྱོགས། རྒྱུ༈ཛཱ་གྲོ་ྋགཉིད་སངས་པ་ལ། ཨ་ཛཱ་ག་རཱིཏྲ། ཨ་ཛཱ་ག་རིཏྟཾ། ཨ་ཛཱ་ག་།
