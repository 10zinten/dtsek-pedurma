ལ་ཎྜི་ཎཁོisཀོ་ལཾ་ག་ལཾ་ཧ་ལཾ། །གོ་དཱ་ར་ཎཉྩ་ཤྲཱི་ར༔་ཤ་མཱ་སི་ཡུ་ག་ཀཱི་ལ་ཀ། ་། ཨི་ཤ་གཤོལ་དང་ནི། །གླང་འཇོམས་དང་ནི་རྩེ་མོ་ཅན། །གཉའ་ཤིང་གཉའ་བའི་ཕུར་བ་དང༌། །མདའ་དང་ལཱངྒ་ལ་དཎྞསཱུཏི་ཏཱ་ལཱངྒ་ལ་པ་དྭ་ཏི། །པུ་སི་མེ་དྷིཁ་ལེ་དཱ་རུ་ཎྱསྟེ་གཤོལ་གྱི་དབྱིག་པའོ། །ཐོང་ལྕགས་དང་ནི་གཤོལ་ལྕགས་སོ། །།སྐྱེས་བུ་རྟོད་ཕུར་འདོགས་ཐག་དང༌། །ཡཏྲ་ཤུ་པཾ་དྷ་ནེ། །ཨཱཤུ་བཱི་ཧིཔཱ་ཊ་ལ་སཱུཏི་ཏ་སུ་ཀ་ཡ་བོ་ས་མོ། །གང་ཞིག་ཕྱུགས་རྣམས་བཏགས་པའོ། །འབྲས་དཀར་འབྲུ་དང་པཱ་ཊ་ལ། །།ན་དང་ཡ་བ་མཉམ་པའོ། །སྟོཀླསྟུ་ཏ་ཏྲ་ཧ་རི་ཏེ་ཀ་ལཱ་བསྡུ་ས་ཏཱི་ན་ཀ། །ཧ་རེཎྜ་ཁཎྜི་ཀོ་ཙ་སྨིནོ་ར་དྷུ་ཥ་སྐུ་མ་སྨིན་པ་ནི་ལྗང་བའོ། །།སྲན་མ་ས་ཏཱི་ན་ཀ་དང་། །ཧ་རེནྡ་དང་ཁཎྜི་ཀ། ཀོ་ར་དྷུ་ཥ་ཀོ་དྲ་བ། །མངྒ་ལ་ཀོ་མ་སཱུ་རོཉཐ་མུཀྟཌ་ཀ་མ་པསྔ་ཀོ། །བ་ན་མུངྒེ་སརྟ་པེ་ཀོ་དྲ་བ། །བཀྲ་ཤིས་དང་ནི་སྲན་ཆུང་ངོ༌། །མོན་སྲན་གྲེའུ་བརྒྱད་པའོ། །ནགས་སྲན་ཡུངས་ཀར་ཡུངས་ཏུརྡོཏུནྟུ་བྷ་ཀ་དམྦ་ཀོ། །སིདྷཱརྦསྟེ་ཨེཥ་དྷ་བ་ལོ་གོ་དུ་མ༔སུ་མ་ན་ས་ནག་དང༌། །པལ་ཁང་དང་ནི་ཀ་དམྤ་ཀའོ། །དོན་གྲུབ་ཡུངས་ཀར་ཟས་གཙང་དང་། །གྲོ་དང་ཡིད་ལེགས་མོ། །སདྨ་བ་ཀསྐུ་ཀུལླེ་ཥཉྩ་ན་ཀོ་ཧ་རི་མནྡྲ་ཛ། །རྡོ་ཏི་ལེ་ཏི་ལ་པེ་*ཛ་ཏི་འདྲ་བའོ། །ནས་དང་ཀུལླཱ་ཥ་ཞེས་སོ། །སྲན་མ་ཧ་རི་མནྠ་སྐྱེས། །གཉིས་ནི་ཏིལ་དང་ཏིལ་སྐྱེས་དང་། །པིཉྫཉྩ་ནིཥྛ་ལེ། །ཀླུ་བཀླུ་དྷཱ་བྷི་ཛ་ན་ནོ་རཱ་ཛི་ཀཱ་ཀྲྀཎྜི་ཀཱཉསུ་རཱི། །སྟི་ཡོ་ཀགླུ་བྲི་ཏི་ལ་པིཉྩ་འབྲས་མེད་དོ། །སྐེ་ཚ་གླུ་དྷཱ་མངོན་སྐྱེས་དང༌། །རྒྱལ་པོ་ནག་པོ་ལྷ་མིན་ནོ། །བྲེ་དང་དྲི་ཡཾ་གུ་དཱི་ཨ་ཏ་སཱི་སྣ་དུ་མཱ་ཀླུ་མཱ། །མཱ་ཏུ་ལ་ནཱི་ཏུ་བྷཾ་གཱ་ཡཱམྦྷི་ཧི་བྷེ་དསྟུཎྜཔུ་མཱན། །ཡཾ་ཀུ་ནི་གཉིས། །ཨུ་མ་ཟར་མ་ཨ་ཏ་སི། །མཱ་ཏུ་ལ་ནཱི་བྷང་གཱ་ཡཱི། །འབྲུ་ཡི་བྱེ་བྲག་ཨཎྜ་ཕོ། །ཀིཾ་ཤཱ་རུས་སྱ་སཱུ་ཀཾ་སྙཀྐ་ནི་སཾ་ས་སྐྱ་མཉྫ་རཱི། །།དྷཱ་ཏྱཾ་ཤྲཱི་ཧིསྟ་བ་ཀ་རིསྟམྦོ་གུཙྪ་ཀི་ཤཱ་རུ་དང་སྙེ་མ་དང༌། །སྙེ་མ་ལོ་ཐོག་དོག་པའོ། །སོ་བ་སྐོགས་ཅན་གངས་མའོ། །སྟམྦོ་གུཏྠ་དྲྀ་ཎཱ་དི་ན། །ནཱ་ཛཱི་ནཱ་ལཉྫ་ཀཱཎྜི་སྱ་པ་ལ་ལོཉསི་ས་ནིཟླ་ལ། །ཀ་ཎངྒ་རོ་པུ་རྩྭ་སོགས་འབྲུ། །མགུལ་པ་ངར་པ་ཀཱཎྜའོ། །སོག་མ་མོ་མིན་འབྲས་བུ་མེད། །བུན་སྟོང་ཕུབ་མ་མ་Iང།
