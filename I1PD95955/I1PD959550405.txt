།སྦི་ཡཱ་མྲྀཀྵ་མ་ཡ་ཛུ་ཥི་ཨི་ཏི་བེ་ངག། གསུམ་པོ་དེ་ནི་ཆོས་སུ་སྒྲུབ། །སྙན་དངགས་ངེས་བརྗོད་མཆོད་སྦྱིན་ལོ། །འདི་གསུམ་རིག་བྱེད་དཱསྔ་ཡསྔ་ལཱི། །ཤིཀྵེ་ཏྭཱ་དི་ཐུ་ཏེ་རངྒ་མ་ར་པྲ་གསུམ་པ་ཡིན། །བསླབ་སོགས་བསླབ་པ་རྟོག་པ། ངེས་ཚིག་། བརྡ་སྤྲོད། སྡེབ་སྦྱོར། སྐར་རྩིས་ཐོས་པའི་ཎ་བོ་ས་མོ། བ། ཨི་ཏི་ཧཱ་སཔུ་རཱ་ཛྲཏྟ་མུ་དཱཏྟཱ་དྷཱསྔ་ཡན་ལག་སྟེ། །ཨོ་ཡིག་པྲ་ཎ་བ་དག་མཚུངས། །སྔོན་རབས་སྔོན་དུ་བྱུང་བའོ། །ཨུ་དཱ་ཏྟཱ་སོགས་ཡ་སཱ་རཱ། །ཨཱ་ནཱི་ཀྱི་ཀཱི་དཎྜ་ནཱི་ཏཱི་ཏརྐ་བི་དྷྱཟླ་ཤཱསྔ་ཡོ། །ཨཱ་གསུམ་དབྱངས་རྣམས། །རྩོད་པ་དང་ནི་ཆད་པའི་ལུགས། །རྟོག་གེ་རིག་དོན་བསྟན་བཅོས་དག། གཙོ་ཁྱུ་ཡི་ཀོ་བ་ལབྷཱརྦ་པུ་རཱ་ཎཾ་བཉྩ་ལཀྵ་ཎཾ། །པྲ་བནྡྷ་ཀལྤ་ནཱ་ཀ་ཐཱ་པྲ་བོའི་ངག་ནི་བཀའ་ལུང་སྟེ། །སྔོན་རབས་མཚན་ཉིད་ལྔ་ཡིན་ནོ། །རབ་སྦྱར་རྟོག་པ་གཏམ་རྣམས་དང་། །བཉྩི་ཀཱ་པྲ་ཧེ་ལི་ཀཱ། །སྨྲ་ཏིསྐུ་དྷརྨ་སཾ་ཧི་ཏ་ས་མཱ་ཧྲ་ཏིསྐུ་སཾ་གྲ་ཧ། ་། བརྡ་ཡིས་སྟོན་དང་གོ་དཀའོ། །དྲན་པ་ཆོས་དང་བཅས་པ་རྣམས། །ལེགས་པར་བསྡུས་དང་བསྡུས་པའོ། །ས་མ་སཱུ་ཏུ་ས་མ་སཱརྦ་ཀིཾ་བ་དནྟི་ཛ་ན་ཐུ་ཏི། །བཱཏྟཱ་པྲ་བྲྀཏྟཱིབིཏྟཱནྟ་ཚིག་སྡུད་པ་དང་བསྡུས་པའི་དོན། །གང་ཞིག་བརྗོད་དང་སྐྱེ་བོས་ཐོས། །གཏམ་དང་ལོ་རྒྱུས་ཁ་སྐོང་ཨུ་དན་སྨཱ་ད་ཐཱ་ཧཱ་ཡ། །ཨཱ་ཁྱཀྵེ་ཨ་བྷི་དྷཱ་ནཾ་ཙ་ནཱ་མ་དྷེ་ཡཉྩ་དང་། །བཀའ་མཆིད་ཡིན་ཏེ་དེ་ནས་ནི། །མིང་དང་ཞེས་དང་མཚན་དང་ནི། །མངོན་པར་བརྗོད་དང་ནཱ་མ་ཙ། །ཧུ་ཏི་རཱ་ཀཱ་ར་ཎཱ་ཧཱ་ནཾ་སཾ་ཧུ་ཏིརྦ་ཧུ༔བྷི་ཀྲྀ་ཏཱ། །བི་བཱ་མིང་དུ་བརྗོད། །འབོད་པ་དང་ནི་སྐད་གཏོང་དང་། །ཡང་དག་འབོད་པ་མང་པོས་བོས། །རྣམ་པར་དོ་བྱ་བ་ཧཱ་ར༔ས་དུ་བ་རྩྭ་སསྐུ་བཱདྷུ་ཁཾ། །ཨུ་པོདྷཱ་ཏ་ཨུ་དཱ་ཧཱ་རཿརྩོད་དང་ཐ་སྙད་དང་། །ཉེ་བར་འགོད་དང་སྨྲ་བའི་སྒོ། །ཉེ་བར་འཇལ་དང་དཔེར་བརྗོད་དོ། །མནའ་ཤ་བ་ནཾ་ཤ་པ་ཐཔུ་མཱན། །པྲ་བྷོཉ་ནུ་ཡོ་གཔྲ་ཙྪཱ་ཙ་པྲ་ཏི་བཱཀྱོཏྟ་རེ་ས་མེ། །དང་གྲོས་འདོན་སྐྱེས་བུའོ། །འདྲི་དང་རྗེས་སྦྱོར་ཤེས་འདོད་དོ། །སླར་སྨྲ་བ་དང་ལྷན་ཅིག་མཚུངས། །།
