རྐྱེན་དུ་འགྱུར་ཏེ། ་ཀཱ་སྲི་ངན་དུ་སྨྲ་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ་རྐྱེན་ཨཱམ། དེ་ལ་ཀྲྀ། ཨས། ་བྷུ་རྣམས་ཎཔྲ་སོགས་ཀྱི་མཐའ་ཅན་དང་སྦྱར་བས། །ཀཱ་སྭཱ་ཙཀྲེ༑ཀཱ་སཱ་མཱ་སེ། ཀཱ་སཾ་བ་བྷུ་བ། །ཁཊ་༈དེ་བཞིན་དུ། ཙ་ཀཱ་སྭ་སྣང་བྱེད་ལའོ་ཞེས་པའི་བྱིངས་ལ། ཙ་ཀཱ་སྭཱ་ཙ་ཀཱ་ར། །ཙ་ཀཱ་སཱ་མཱ་ས། ཙ་ཀཱ་སཱ་བ་བྷུ་བ། ལོ་གསུམ་བྱ་བར་འབད་པ་གནང་བ་དང་ཞེས་། ༈ཛ::གྲོ་གཉིད་ཟད་པ་ལའོ་ཞེས་པ་ལ། ཨཱམ་བྱིངས་མཐའི་ཡོན་ཏན་བྱས་པས། ་ཛ་ག་རཱི་ཙ་ཀཱ་ར། ་ཛ་ག་རཱ་མཱ་ས། །ཛ་ག་རཱ་བ་བྷཱུ་བ་སོགས་དང་། ༈ཨེདྷ་འཕེལ་བ་ལ། ཨེ་དཾ་ཙ་ཀྲེ། ཨེ་དྷཱ་མཱ་སེ། ཨེ་དཾ་བ་བྷ་བེ། །ས་ཊ༈ཀ་ལ་ཆ་ཤས་ལའོ་ཞེས་པ་ལ། ཙུ་ར་ལ་སོགས་ལ་ཉི་འོས་རྐྱེན་ཉིའི་མཐའ་ཅན་ལ། ཨཱམ་སོགས་སྦྱར་ཡོན་ཏན་བྱས་པས། ་ཀ་ལ་ཡཾ་ཙ་ཀཱ་ར། ཀ་་ལ་ཡཱ་མཱ་ས། ཀ་ལ་ཡཾ་བ་བྷཱ་བ་སོགས་དང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ཀ་ལ་ཡཱ་ཙ་ཀྲེ། ཀ་ལ་ཡཱ་མཱ་སེ། ཀ་ལ་ཡཱ་བ་བྷཱུ་བེ་སོགས་སོ། །དག་པ་དང་། ས་༈བིད་ཤེས་པ་ལའོ་ཞེས་པ་ལ། ་ཨཱ་མི་བི་དེརྣ་གུ་ཎ་ཨཱམ་ལ་བིད་་ཀྱི་ཡོན་ཏན་མ་ཡིན་ནོ། །ཨཱམ་ལེཕ་རོལ་ལ་བྱིངས་བིད་ལ་ཡོན་ཏན་འབྱུང་བ་མ་ཡིན་ཏེ། ཡོན་ཏན་མི་བྱེད། བི་དཱ་ཙ་ཀཱ་ར། །བི་དཱ་མཱ་སེ། །བི་དཱ་བ་བྷཱུ་བ་་སོགས་སོ། །ཕྱིས་པ་ལ་ས་དང་པོ་པ་ཁུ་ཁྲོ་ཁོ་དབལ་པ། དམ་ལ། །༈ཨཱིཀཱ་ལྟ་བ་དང་མཚན་མ་ལའོ་ཞེས་པ་ལ། ་ཨཱིཀྵ་ཙ་ཀྲེ། ཨཱིཀཱ་མཱ་སེ?། ཨཱིཀཾ་བ་བྷུ་བ། ཨུཏྤ་རྣམ་པར་བརྟགས་པ་ལ། །ཨུ་ཧཱུྃ་ཙ་ཀྲེ། ཨུ་ཧཱ་མཱ་སེ། ཨཱ་ཧཱུྃ་བ་བྷཱ་བེ་སོགས་སོ། །དགེ་བ་འདི་ད་བ་འདྲ། ༈ཨིཧ་རྩེད་མོ་ལའོ་ཞེས་པ་ལ། །ཨི་ཧཱུྃ་ཙ་ཀྲེ། ཨི་ཧཱ་མཱ་སེ། །:ཨི་ཧཱུྃ་བ་བྷུ་བེ་ྋསོགས་སོ། །དེ་ནས་རྐྱེན་ས་དང་བཅས་པའི་ཨཱམ་གྱི་བྱ་བ་རྣམས་བཤད་།
