ཀ་ཀ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཀུ༌ལའི་ཁ་གླིང་མྱུར་བ་དང་མགྱོགས་པ་དང་ལ་གོར་དང་སྐྱེན་པ་དང་ཡུད་ཙམ་དང་ཅིག་ཅར་དང་མི་ཐོགས་པ་དང་དེ་མ་ཐག་པ་དང་སྐད་ཅིག་མ་དང་མི་འགོར་བ་དང་དར་ཙམ་མོ་ཞེས་པ་འདི་དག་གིས་མྱུར་བའི་སྤྱི་ཙམ་བསྟན་ནོ། །ཧ་རི་འགྲོ་བ་ལའོ། །ཨ་་་སུནའོ༑། རཾ་ཧ། ཏ་རནྟི་ལ་འདིས་ཨ་སུནའོ། །ར་ཡ་འགྲོ་བ་ལ་གྷའོ། །ཡང་ནི་ར་ཉན་པ་ལའོ། །མྱ་དའི་རྒྱུག་པ་ལ་ཞེས་པས་གྷཉ་ལ་ཚིག་ཕྲད་ལས་་་སྱནྡའི་སྱ་དའོ། །་n:། ཛུ་ཞེས་པ་མདོའི་བྱིངས་ལ་ཤུགས་བརྗོད་པ་ལ་ཛུ་ཙམ་ཀྲ་མྱ་་་ཞེས་འདིར་བཀླགས་ཏེ། ་རྀཏ་ཀྱི་ཨུཏ་ཀྱི་ཨསའོ། །ཨ་ཤུའི་མཐར་ཐུག་པ་ནི་་་མྱུར་བའི་མིང་སྟེ་མྱུར་བ་དང་མགྱོགས་པའི་དབྱེ་བའོ། ཚོ། །ཞུགས་དང་བཅས་པས་འགྲོ་བ་བརྗོད་པ་ནི་མགྱོགས་པ་ལ་སོགས་པའོ། །མྱུར་བ་ལ་སོགས་པ་ནི་ཆོས་་་བརྗོད་པའོ། །།དེ་ཡང་ཤཱི་གྷོ་པ་ཙ་ཏི། ཤྲཱི་གྷོ་གཙྪ་ཏི་ཞེས་པ་རབ་ཏུ་སྦྱོར་བར་འགྱུར་གྱི། །།ཛ་བོ་གཙྪ་ཏི། ཛ་ཐཾ་པ་ཙ་ཏི་ཞེས་པ་མ་ཡིན་ནོ། །བྷ་དྲ་ལ་་་སོགས་པ་རྣམས་སོ་ཞེས་པས་ཤིང་ལས་རཀ་དང༌། །གླུ་ཀ་ཧཱུཾ་ཧྰུཾ་ཨཱ་ག་མ་ཡང་ངོ༌། །ཡང་ན་ཤིང་ལས་ཀྭིབ་དང་། མལ་དུ་ཉལ་བ་ལ་ཛི་གྷ་ཏི་ཞེས་པས་ཀའོ། །ཏུ་་ར་ཀུན་དུ་འཁྲུལ་པ་ལའོ། །།འགྲོ་བའི་དོན་དང་ལས་མེད་པ་ལ་ཞེས་པས་མང་པོ་ལས་དུས་དང་ཐུན་མོང་བ་ལ་སཱ་རི་ཏ་ཀྟའོ། །རཱུཊྚ་ཧྲི་ཥཱ་མ་ར་ཏཱ་ར་སཾ་གྷོ་ཥཱ་ལས་སྲ་ན་ལས་སོ་ཞེས་པས་ཡང་ན་ཨཊའོ། །ཤ་ཀུ་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པས་ར་ཤྲཱི་ལས་ཀུའོ། །ལ་བྷུཀ་བི་རི་ཀ་ལ་སོགས་པ་རྣམས་སོ། །།རུ་དྲ་བཞིན་དུ་ཀླུ་པ་ལ་རཀའོ༑། རཾ་སྲི་འགྲོ་བ་ལའོ། །ཨི་ཡནྟི་ཞེས་པས་ཨཙའོ། །ལྷན་ས་ཧ་ཅིག་་་མྱུར་བར་འཇུག་པས་ས་ཏཱ་རཾ་སྟེ། ཅིག་ཅར། ལས་ཧའི་ས་རུ་བསྒྱུར་ཏེ་གཞན་དོན་གྱི་བསྡུ་བའོ། །ཚུ་པ་དལ་བར་འགྲོ་བ་ལའོ། །བྲི་ཥ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཀ་ལའོ། །མང་དུ་ཕྱི་པོ་ལས་ཨུཏ་ཀྱི་ཨཏ་ཉིད་ཀྱང་ངོ༌། །ཟ་བ་ལ་མི་ཐོགས་པ། ཛ་ར་ཏཱ་རཞེས་་་།
