བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཎ༑། བཀྑཾ་པ་ཏི་ཀཱིརྟི་ཏཾ། །ཛེ་ནཱ་ཡུ་ཛོརུ་དཱ་ཤྲཱི་ཏཱ། །བི་པ་རཱི་ཏཱ་དི་ཏེ་ནི༑། ཁ་སྒོ་ཕན་པ་ཞེས་པར་གྲགས་སོ། །ཛ་ཡིས་ཟུང་མིན་བརྗོད་པ་འདི་ནི། །བཟློག་པ་དང་པོ་ཅན་ཥསྨིན༑། བཀྲྀ་ཏ་དེ་བ་ཙ་པ་ལཱ། །ན་ཀཱ་རེ་ཎཱ་ཡུ་ཛོ་ཛྷེ་ཡཱ། །ལ་གླུ་སཔ་མ་གྱི་དེའོ། །ཁ་སྒོ་དེ་ཉིད་གྱ་གྱུ་ཅན་དུ། ་། ན་ཡིག་ཟུང་མིན་ལས་ཤེས་བྱའོ། །ཟུང་ལ་བདུན་པ་ཡང་བ་ཡུ་གྷཱ་ཡཱ། །བི་པུ་ལཱ་སོ་མ་ཏཱ་མུ་ནི། །སེ་བ་ཏ་སྱ་སཱ་སརྦ་ཏ། །བྷེ་ནཱམྦུ་དྷེ་ཡིན་པས། །ཐུབ་པ་དེ་ནི་ཡངས་མར་འདོད་དོ། ་། ཐམས་ཅད་ནས་དེ་དེ་ཉིད་དེ་ཡི། །མཚོ་ལས་བྷ་སྟདྲི་པུ་ལཱ། །རེ་ཕེ་ཎ་ཙི་ཐཾ་ཏ་ཏོ་ཧུཾ། །ན་ཀཱ་རེ་ཎཱ་པི་ཙ་ཏ་ཐཱ། །ཡིས་དེ་ཡི་ཡངས་མ། །རེ་ཕས་དེ་ལས་གཞན་ཡང་དེ་བཞིན། ་། ན་ཡིག་གིས་ཀྱང་དེ་བཞིན་ཡང་ངོ༌། །ཨེ་བི་ཏ་ཀཱ་རེ་ཎ་པི་བཱ། །དྲྀཏྟཱ་མ་ཀཱ་རེ་ཎཱ་པེ་པཾ། །ཨ་ནྱ་ད་ཏོ་ཧི་བི་ཏཱ་ནཾ། །ཏ་ཡིག་གིས་ཀྱང་དེ་བཞིན་ཉིད་དོ། །མ་ཡིག་གིས་ཀྱང་དེ་ལྟར་བལྟའོ། །དེ་ལས་གཞན་ནི་བླ་བྲེ་ཡིན་ནོ། །ན་རཱུནྡྷོ་སརྦ་པཱ་དེ་ཏུ། །།ཇོཉམྦུ་དྷེཆུ་གླ་པཱ་ད་ཡོ། ་། ཨ་ཡུགླ་ཡོདྨ་ཐཱ་རྐང་པ་ཀུན་ལས་ཐོག་མ་ལས་ས། །ན་མིན་རྐང་ཟུང་རྒྱ་མཚོ་ལས་ཛ། །ཟུང་མིན་དག་གི་ཇི་ལྟར་ཤོ་བྷ། །མི་ཏི་པྲ་ཡབྲ་ཡོ་ག་ཛྲ་ཀྲ། ?། བཀྲུ་པྲ་བྷེ་དཱ། ་། ་་་་་མཛེས་པས། །སྦྱོར་བ་ཕལ་ཆེར་བལྟ་བར་བྱའོ། །ཁ་སྒོའི་བྱེ་བྲག་རྣམས་སོ། །ལྟ་བ་ཀློང་ན། པྲ་ཐ་མོཉཏྟཱཀྟ་རཔཱ་ད༔་ཏུབྷི་ར་ཏི་བརྗེ་ཏཱཔ་རེཉམུ་རྐང་པ་དང་པོ་ཡི་གེ་བརྒྱད་ཡིན་ནོ། །འདི་ལས་ཕྱི་མ་རིམ་བཞིན་བཞི་རྣམས་མངོན་པར་འཕེལ་བྷཱུཏྲ། །ཀྲ་མ་ཤོ་ཡ་སྱཱ་ཏ་དེ་ཏནྟྲ་ནཱ་བི་རྩྭ་ས་སཾ་ཤོ་བྷི། །ས་ལ་ལི་ཏ་བ། །དེ་དང་དེ་རྣམས་སྣ་ཚོགས་རྣམ་པར་བཀོད་པས་ཡང་དག་མཛེས་པ་ཅན་ལ། །རོལ་སྒེག་དང་པ་ད་བ་ད་བཱི་ཀཾ་པ་ད་ཙ་ཏུ་རུ་ཚྭ་བུ་དྷཱ་པཱ་ཧུ། །ཨི་ཧ་པ་ད་ཙ་ཏུ་བཅས་རྐང་པ་རྐང་པར་ལྡན་པ་དག་ན་རྐང་བཞི་འཕེལ་དུ་མཁས་རྣམས་སྨྲའོ། །འདི་ནི་རྐང་པ་བཞི་ཀ་རུ་སྣེ། ག་ག་ནི་དྷ་ན་ནི་ཁི་ལ་ལ་གྷ་པཱ་ད། །ཡ་དེ་མུ་ད་མུ་པ་ན་ཡ་འཕེལ་ལ། ག་དང་ག་ཡི་མཐའ་ཅན་རྐང་པ་མ་ལུས་ཡང་བ། །རོལ་སྒེག་དང་བཅས་ཉེ་བར་ཕན་པ་ངལ་__།
