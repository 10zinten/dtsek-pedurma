ཎཀི་ལྔ་སྐྱོན་ཆ་ཡོད་ཆ་ཛཱ་ཤི་ཛི་ནཱི་གུ་ཎ། ་། སཱུཏྤ་ཏྱཱ་ལི་ཌྷ་མཱ་ལཱི་ཌྷ་མི་ཏྱ་དིསྦྲ་ན་པཉྩ་ཀཾ། འི་། སྲད་བུ་རྒྱུད་ཡོན་ཏན། །གཡོན་བརྐྱང་གཡས་བརྐྱང་ལ་སོགས་པའི། །གནས་ནི་རྣམ་པ་ལྔ་ཞེས་སོ། །ལསྶཾ་ལཀཾ་ཤཱར་བྱཉྩ་ཤ་རཱ་བྱ་ས་ཨུ་པཱ་ས་®ནཾ། །།པྲྀ་ཥཥ་་བཱ་ཎ་བི་ཤི་ཁཱ་ཨ་མདའ་འཕེན་མཚོན་ཆ་ལཀྵ་ངོ༌། །འཕོང་སློབ་ཨུ་པ་ས་ནཾ་འོ། །མདའ་དང་བཱ་ཎ་རྩེ་མོ་ཅན། །དྲང་ཛི་ཧྥ་ག་ཁ་གཱ་ཤུ་གཱ། ་་། ཀ་ལཾ་བ་མཱརྒ་ཎ་ཤ་རཱཔཏྟི་རོ་པ་འགྲོ་མཁའ་འགྲོ་མྱུར་འགྲོ་དང༌། །ཉག་ཕྲན་ལམ་ལྡན་འབིགས་བྱེད་དང༌། །ཤོག་ལྡན་འདེབས་བྱེད་ཨི་ཥཱརྡ་ཡོ། །པྲཀྵ་ཌནཱསྟུ་ནཱ་རཱ་ཙཱཔཀྵ་བཱ་ཛམྦྷི་ཧཱུཏྟ་རེ། །ནི་རསྟམདའ་ནི་གཉིས། །ལྕགས་མདའ་ནཱ་རཱ་ཙ་ཡིན་ནོ། །སྒྲོ་དང་འདབ་མ་གསུམ་ནི་ཕྱིས། །ནི་རསྟ་པྲ་ཧི་ཏེ་བཱ་ཎེ་བི་ཥཱཀྟ་དགྷ་ལིསྟ་ཀོ། ་། ཏཱུ་ཎ་པཱ་སཾ་ག་ཏཱུ་ཎཱི་ར་ཎི་ཥ་ནི་མདའ་འཕངས་པ། །དུག་ཅན་མདའ་ནི་ལི་སྟ་ཀོ། །དོང་པ་མདའ་ཤུབས་ཏུ་ཎྷི་ར། །བརྩོད་རྩེ་གཱ་ཨི་ཥཱ་དྷིནྡྲ་ཡོ། །ཏུ་ཏྟཾ་ཁནེ་ཏུ་ནིསྡིདྷ་ཙནྡ་ནཾ་སཱ་སི་རྀཏྠ་ཡ། ་་། མདའ་འཛིན་གཉིས་དོང་པ། །རལ་གྲི་གཅོད་བྱེད་མདའ་འཛུམས་དང༌།  ི་། ཨ་སི་རལ་གྲི་རྡེག་ཆ་དང་། །ཀོཀྵ་ཡ་ཀོ་མཎྜ་ལཱ་གྲཀ་ར་བཱ་ལཊཾ་པཱ་ཎ་བཏ། །ཏྠ་རུཁནྡྲ་དི་མུཊོ་སྙཏྠེ་ཁ་ལཱ་མཎྜ་ལ་རྩེ་ལག་པ་བསྐྱོད། །ཀྲྀ་པཱཎ་འདྲ་བསེ་རུའོ། །རལ་གྲིའི་ཡུ་བ་ཆང་ཟུང་ངོ༌། །དེ་ཡི་འཆིང་ཏནྣ་བནྡྷ་ནཾ། །ཕ་ལ་ཀོཉམྦྷི་ཕ་ལཾ་ཙརྨ་སཾ་གྲ་ཧོ་མུཉྫི་ར་སྱ་ཡ། །དྲུ་གྷ་ཐག་སྐེ་རགས་སོ། །ཕུབ་ནི་མོ་མིན་ཕ་ལའོ། །ཙར་མ་བསྡུས་ནས་ལག་འཛིན་པ། །ཐོབ་འཇོམས་ཎོ་མུངྒ་ར་གྷ་ཎོ་སཱ་དཱི་ལཱི་ཀ་ར་ཕ་ལི་ཀཱ། །བྷིནྡི་པཱ་ལསྲི་གསཱུ་ལོ་ྋཔ་རི་གྷཔ་རི་བྱེད་གྷ་ནའོ། །སྐེད་གྲི་ཀ་ར་ཕ་ལི་ཀཱ། །འབིགས་སྐྱོང་མདུང་དང་མཉམ་པའོ། །དབྱུག་པ་ལྕགས་གྷཱ་ཏ་ན། །དྭ་ཡོཀུ་ཋཱ་རཿསཱ་དྷི་ཏིཔ་ར་ཤུཉྩ་པ་ར་ཤཱ་དྷ། །སཱཙྪ་སྟི་ཙཱ་སི་པུ་ཏཱི་ཙ་བཅིངས་པ་རི་གྷ། །།གཉིས་པ་སྟ་རེ་པ་ར་ཤུ། །དགྲ་སྟ་དང་ནི་སྟ་མོའོ། །ཆུ་གྲི་གྲི་ཐུང་ཚུ་རི་ཀ། ཚུ་རི་ཀཱ་ཙཱ་སི་དྷེ་ནུ་ཀཱ། །བ་་སི་ཤ་ཙཾ་སརྐུན་སརྦ་ལཱ་ཏོ་མ་རོཉཧྲི་ཡཾ། །པཱ་ཤསྐུ་རལ་གྲི་ཕུ་གུའོ་འཕང་མདུང་དང་། །ཤ་ལ་དང་ནི་གསེར་མདུང་ངོ༌། །མདའ་བོ་ཆེ་དང་ཏོ་མ་ར། །—།
