རྣམས་ཀྱི་མང་ཚིག་ལ། །ཨངྒྷཨངྒྷཨ་པཾ། ཨཱམ། ཨ་སླུ། །ལྷ་ཚུལ་སྣ་ཤ་ཡིག་མཐའ་ཅན་དིཤ་ཞེས་པའི་སྒྲ་ལས། དང་པོའི་གཅིག་ཚིག་་ལ༑་དི་ཀཾ་ཨཀདིཤ་རྣམས་ཀྱི་:ཀའོ། །དིཤ། ཕྱོགས། དྲིཤ། ལྟ་བ། སྤྲོཤརེག་པ། མྲྀཤ། ནརྫུ། མཉམ་པར་ལྟ་བ། ས་དྲི་ཤ་རྣམས་ཀྱི་མཐའ་ནི་ར་སའི་ཡི་གེ་ཕ་རོལ་ན་ཡོད་པ་དང་ཚིག་མཐའ་ལས་ཀྱང་ཀ་ཡིག་ཏུ་འགྱུར་ཏེ། ཤ་ཀྲ། གནས་མེད་པ་དང༌། ཛཱིབྷ་ལ་ཛ་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ༈ཕྱོགས། དིཀྲ༄༅། སི། དགྲ། སི། དི་ཤོ། ཨོ། དི་ཡཛཐ༄༅། ཧེ་དིཀྲི། སི། ཧེ་དིག། སི། ཇི་ཤཾ། ཨམྦི། དི་ཤོ། ཨོ། དི་ཤཤསྲི། དི་ཤཱ། ཐཱ། དི་དྡྷཾ། དིཀྐིསོགས་དང་བདུན་པའི་མང་ཚིག་ལ། དིཀླུ། སུཔ། ལེ་༈ཥ་ཡིག་མཐའ་ཅན་ཏཱིཥ་ཞེས་པའི་སྒྲ་ལ། ཥའི་ཌའོ་དང་གནས་མེད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ༈མཛེས་པའམ་བཀོད་པ་ལེགས་པ། ཏཱིཌ། སི། ཏཱིཊ༄༅། སི། ཏཱིཎོ༑ཨོ། ཏཱི་ཥཛགྲཅི། ཧེ་ཊི། སི། ཧེ་ཏཱིཊ༄༅། སི། ཏཱི་ཎཾ། ཨམྲ། ཏཱི་ཁོ། མོཏཱི་ཥཤགྲ། ཏཱི་ཥཱ། །ཏཱིཙྪཱ། ཏཱིཙྪིསོགས་དང་བདུན་པའི་མང་ཚིག་ལ། ཏིཀཱ༄༅། གུཔྲ། ལོ་སུ༈ཥ་༌༌ཡིག་མཐའ་ཅན་ཨཱ་ཤིཥཱ་ཞེས་པའི་སྒྲ་ལས། སི་ལ་ར་ཉིད་དང་ལྷག་མ་ར་སའི་ཡི་གེ་རྣམས་ལ། དོཥཱ་ཙིརྣམས་ཀྱི་ར་འོས་དྲ་དང་། ཥའི་ཌའོས་ཊར་སྒྱུར་བ་འདམ་ཁ་ཅན་དུ་བྱས། རའི་ཕྱོགས་ལ། ས་ཛུཥ༌ིསོགས་ཀྱིས་རིང་པོར་བསྒྱུར་བས། ༈མཛའ་བའམ་སྙིང་ཉེ་བཨཱ་ཤཱིསི། ཨཱ་ཤིཎོ། མོ། ཨཱ་ཤིཥཛགྲི། :：ཧེ་ཨཱ་ཙྪིཤཱིསི། ཨཱ་ཤི་ཎཾ། ཨམྲ། ཨཱ་ཤི་ཌོ། ཨོ། ཨཱ་ཤི་ཥཤགྲ། ཨཱ་ཤི་ཥཱ། ཊ། ཨཱ་ཤཱིདྡྷཾ། ཨཱ་བྷིནྡྷ། ཨཱ་ཤྲིབྷིེ༔ཨཱ་བྷིནྡྷོསོགས་དང་བདུན་པའི་མང་ཚིག་ལ། །ཨཱ་ཤཱིནྡྷ། ཨཱ་བྷིgི། །ཀློག་ཚེ་ད་ཆ་ལོ་བོདིA—།
