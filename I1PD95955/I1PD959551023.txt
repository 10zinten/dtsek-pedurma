ལོར་འདིའི་དྲྀ་ཤྲཱི་ཀ་ལ་སོགས་པ་བཞི་ནི། ཀ་ཀོ་ལོའི་མིང་སྟེ། པྲ་བྷི་ཀ་ཊི་། ས་ཅན་མའོ། །ཙནྡྲ་བཱ་ལ། ཟླ་བའི་བྱེ་མའམ། ཟླ་བ་གཞོན་ནུའོ། །བནྡྷ་ཀ། རྩ་ལག་ཅན་དང༌། རཀྟ་དམར་པོའོ། །འོ་ན་པྲ་ཏི་ཧ་ས་ལ་སོགས་པ་ལྔ་ནི། ཀ་ར་བི་རིའི་མིང་སྟེ། པྲ་ཏི་ཏཱ་ས། ：སོ་སོར་གཞོད། ཤ་ཏཔྲས། བརྒྱར་གྱེས་དང་། ཙནྡྷུ་ཏ། གཏུམ་པོ་དང་། ཧ་ཡ་མ་ར་ཀ། རྟ་གསོད་པ་དང༌། ཀ་ར་བཱིར། དཔའ་བོའི་ལག་པའོ། ལུ། མེཀ་མི་ང་དེང་སྨྱོ་བྱེད་ལ་སོགས་པ་བདུན་ནི། བྷཱུ་རཱུ་རའི་མིང་སྟེ། ཨུནྣཏྟ། སྨྱོ་བྱེད་དང༌། ཀ་ན་ཀཱ་ཧཱ་ཡ། གསེར་མིང་ཅན་དང་། མ་ད་ན། མྱོས་བྱེད་དོ། །དཔལཱི་ལ་དེ་ནས་བཞི་ནི་བཱི་ཛ་པཱུ་རའི་མིང་སྟེ། བཱི་ཛ་པཱུར། ལྟ་ས་བོན་གྱི་གང་བ་་་དང༌། ལ་ཁ་བུར། འབྲས་བུ་གང་བ་དང༌། །རུ་ཙ་ཀ། མདོག་མཛེས་དང་། མཱ་ཧུ་ལུངྒ་བ་ལུང་ཁ་ལུང་ངོ༌། །་་དེ་བགྱི་བ་གམ་བལྟཁ་དོག་ནགདེ་ནས་གསུམ་ནི། ཅི་དྲ་ཀའི་མིང་སྟེ། པཱ་ཋི་དང་། ཅི་དྲ་ཀ། ཁྲ་བོ་དང་། བནྡྷི་སཾ་ཛྷ། མིའི་མིང་ཅན་རྣམས་སོ། །»ཚོ་བ་ཏུ། །ལོ་ཡཚེཆ་དེ་ནས་བདུན་ནི། ཨརྐ་བཙྪའི་མིང་སྟེ། ཨན་ཧཱ། །ཉི་མའི་མིང་ཅན་དང་། ཨརྐ་པརྞ་ཉི་མའི་འདབ་མ་ལ་སོགས་པའོ། །ཅ་། །ད་ན། །དེ་ནས་ཧོང་ལེན་ལ་མིང་བཞི་སྟེ། བནྡྷ། ཧོང་ལེན་དང་། བྲྀཀཱ་དནཱི། ཤིང་ལ་ཟ་བ་དང་། བྲྀཀྵ་རུ་ཧ། ཤིང་སྐྱེ་སྟེ། ཤིང་གི་སྟེང་དུ་སྐྱེ་བའོ། །ཛཱི་བ་ནི་ཀཱ། འཚོ་བྱེད་རྣམས་སོ། །།།བདག་ལ་ཁ་བུ་ཟ་ལ་སོགས་པ། སླེ་ཁྲེས་ཀྱི་མིང་སྟེ། བིནྡད་ནི། བུ་ཟའོ། །ཙྪི་་རྣརུ་ཧ། བཅད་སྐྱེ་སྟེ། བཅད་ཀྱང་སྐྱེ་བས་སོ། །ག་ཀླུ་ཙི། སླེ་དྲེས་སོ། །ཏནྡྲི་ཀ། རྒྱུད་ཅན་ཏེ་དེ་ལ་སོགས་པ་རྣམས་སོ། །རི་མོ་དང་ཉེ་བའི་རི་མོ་ལ་་—ནྡྲི།
