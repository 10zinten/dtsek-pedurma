བསྟན་འགྱུར། སྒྲ་མདོ། སེ། བྱེ་བྲག་གོ། །གལ་ཏེ་ས་གསུམ་ཅན་ལས་ལ་དང་ག་ཡིན་ཞིང་། །།ན་བྷ་བྷ་ར་་རྣམས་ཀྱང་དེ་ཚེ་རི་དྭགས་:འཕྱོང་ངོ་། །གལ་ཏེ་རྐང་པ་ཟུང་མིན་ས་གསུམ་་ལས་ཕ་རོལ་ཏུ་ཡང་བ་དང་ལྕི་བར་གྱུར་ཅིང་། །ཟུང་གི་རྐང་པ་ནི་ན་དང་བྷ་དང་བྷ་དང་ར་རྣམས་ཡིན་པ་དེའི་ཚེ་རི་དྭགས་འཕྱོང་ངོ༌། །ཡ་དི་ས་ཏྲི་ཏ་ཡཱདྷ་བ་ཏོ་ལ་གོ་ན་བྷ་བྷ་རཱཉྩ་ཏ་དཱ་ཧ་རི་ཎ་པཱུ་ཏཱ། Tགལ་ཏེ་ན་ན་ར་དང་ལ་དང་ག་ཡིན་ཞིང་། །གལ་ཏེ་ན་ཛ་ཛ་ར་དེ་ཚེ་གཞན་སྒོར་འདོད་དོ། །གལ་ཏེ་རྐང་པ་ཟུང་མིན་ན་དང་ན་དང་ར་དང་ཡང་བ་དང་ལྕི་བ་ཡིན་ཞིང་། །ཟུང་གི་རྐང་པ་ཡང་གལ་ཏེ་ན་དང་ཛ་དང་ཛ་དང་ར་ཡིན་པ་དེའི་ཚེ་གཞན་སྒོར་འདོད་དོ། །ཡ་དི་ན་ན་ར་ལ་གཱ་ན་ཇོ་ཛ་རོ་ཡ་དི་ཙ་ཏ་དཱཉཔ་ར་བཀྟ་མི་བྱ་ཏེ། TILགལ་ཏེ་འདིར་ནི་ཟུང་མིན་ན་དག་ར་ཡའོ་ཟུང་ལ། ་། ན་ཛ་ཛ་ར་ག་རྣམས་མེ་ཏོག་རྒྱས་པའི་རྩེ་མོར་བརྗོད། །འདིར་ཞེས་པ་ཕྱེད་མཉམ་པ་རྣམས་ལ་སྟེ། གལ་ཏེ་རྐང་པ་ཟུང་མ་ཡིན་པ་ན་དང་ན་དང་ར་ཡ་རྣམས་ཡིན་ཞིང༌། ་ཀཟུང་་གི་རྐང་པ་ནི་ན་དང་ཛ་དང་ཛ་དང་ར་དང་ལྕི་བ་རྣམས་ཡིན་པ་དེའི་ཚེ་མེ་ཏོག་རྒྱས་པའི་་རྩེ་མོར་བརྗོད་དོ། །ཁང་། ཨ་ཡུ་གི་ཧ་ཡ་དི་ནོ་ར་ཡོ༔་་ཡུསྶཾ་ན་ཛ་ཛར་གཱཀ་ཐ་ཡནྟི་་་པུཥི་ཏཱ་ནཱཾ། /ལ་འདི་གཉིས་རོ་ལངས་ཅན་གྱི་བྱེ་བྲག་གོ། །།ར་ཛ་ར་ཛ་དེ་བཞིན་ཛ་ར་ཛ་ར་ག་རྣམས། །ཡ་བའི་སྒྲ་ལས་མ་ཏུས་ཡིན་ཏེ་བུད་མེད་མཚུངས་པར་བརྗོད། །གལ་ཏེ་རྐང་པ་ཟུང་མ་ཡིན་པ་ར་ཛ་གཉིས་དང༌། །ཟུང་གི་རྐང་པ་ནི་ཛ་ར་ཛ་ར་ལྕི་བ་རྣམས་་་ཡིན་པ་དེའི་ཚེ་ནས་ལྡན་མའོ། །་། ཡ་ཝའི་སྒྲ་ལས་མ་ཏུས་ཀྱི་རྐྱེན་གང་ཡིན་པ་དེ་བརྗོད་བྱ་ལས་བུད་མེད་ཉིད་མཚུངས་པར་བསྟན་པ་ནི་ནས་ལྡན་མ་ཞེས་པའི་དོན་།
