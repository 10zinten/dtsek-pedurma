སཾ་ལས་ཕ་རོལ་གྱི་དྲི་ཤིར་ལྟ་བ། གཙླི་འགྲོ་བ་ཐུ་ཐོས་པ། བོད་་་ཤེས་པ། པྲིཙྪ་ཤེས་པར་འདོད་པ་རྣམས་ལའང་རྣམ་དབྱེ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་་་འཇུག་སྟེ། པྲཏྱ་བ་ཏིཥྛ་ཏེ། སོ་སོ་ཀུན་ནས་གནས། སཾ་ཏིཥྛ་ཏེ། ཡང་དག་པར་གནས། སཾ་པ་་ཏེ། ཡང་དག་པར་ལྟ་བར་བྱེད། སཾ་གཙྪ་ཏེ། ཀུན་ཏུ་འགྲོ་བར་བྱེད། སཾ་ཀྲྀ་ཊུ་ཏེ། ཡང་དག་པར་ཉན་པར་བྱེད། ་སཾ་བིཏྟེ། ཀུན་ཏུ་ཤེས་པར་བྱེད། སཾ་པྲ་ཙྪ་ཏེ། ཀུན་ནས་འདྲི་བར་བྱེད། ཉེར་བསྒྱུར་ཨཱང་ལས་ཕ་རོལ་གྱི་པྲཙྪ་དང་ཏ་ནུ་ཡང་བ་ལའང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། ཨཱ་པྲིཙྪ་ཏེ། ཀུན་ཏུ་འདྲི་བར་བྱེད། ཨཱ་ཏ་ནུ་ཏེ། ཀུན་ནས་ཡང་བར་བྱེད། སོགས་སོ། །ཏ་ན་ག་ལོ་མ། །ཁ་ར་ཏཁཎྜ་ཟླ་་ཟླ་ཁོ་༈ཨཱ་ངོ་དོཉ་ནཱ་སྱ་བི་ཧ་ར་ཎེ་། ཨཱང་ལས་ང་ལ་ཁ་གདངས་པ་མ་ཡིན་པ་ལའོ། །ཉེར་བསྒྱུར་ཨཱརྱ་ལས་ཕ་རོལ་གྱི་དཱ་ལ་ཁ་གདངས་པའི་དོན་མ་ཡིན་པ་ལ་ཨཱཏྨ་ནེ་པ་ད་ྋརྣམས་འཇུག་སྟེ། ཨཱ་དཏྟེ་ཀུན་ཏུ་སྦྱིན་པར་བྱེད་སོགས་དང༌། ཁ་གདངས་པའི་དོན་ཡིན་ན་པ་རསྨི་པ་ད་རྣམས་སྦྱོར་བས་། ལ་བྱཱ་ད་དཱ་ཏི་ཙཊ་ཀཿམུ་ཁཾ། བྱེའུ་ཙ་ཏ་ཀའི་ཁ་ལ་སྦྱིན་པར་བྱེད། །གང་ལ་ནཁོ་བ་ད་ན་བ་མ་དག་ཏུ་ཆོ་ལྟ་ཀྲྀ་ཌཉནུ་སཾ་པ་རི་བྷུཉྩ། །ཨ་ནུ་སཾ་པ་རི་རྣམས་ལས་ཀྱང་ཀཱི་ཌ་ལའོ། །ཉེར་བསྒྱུར་ཨ་ནུ་སཾ་པ་རི་རྣམས་དང་ཨཱང་ལས་ཀྱང་ཕ་རོལ་གྱི་ཤྲཱིརྔ་་་ལ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། །ཨ་ནུ་ཀཱིཌ་ཏེ། མཐུན་པར་རྩེ་བར་བྱེད། སཾ་ཀཱི་ཌ་ཏེ། ལེགས་པར་རྩེ་་བར་བྱེད། པ་རི་ཀྲིཌ་ཏེ། ཡོངས་སུ་རྩེ་བར་བྱེད། ཨཱ་ཀཱི་ཌ་ཏེ། ཀུན་ཏུ་རྩེ་བར་བྱེད། ད་བར་དུ། ཀ་ཁ༈ས་མ་བ་པྲ་བ་བིབྷུསཿསརྦ་ཨ་བ་པྲ་ཨུ་པ་བི་རྣམས་ལས་སྦྲུ་ལའོ། ཡ་། ཉེར་བསྒྱུར་སཾ་དང་ཨ་ེབ་དང་:པྲ་དང་ཨུ་པ་དང་བི་རྣམས་ལས་ཕ་རོལ་གྱི་ཏྟཱ་་་་འགྲོ་བ་ལས་ལྡོག་པ་ལའང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། སཾ་ཏིཥྛ་ཏེ། ཡང་དག་།
