ཡུལ་གྱི་ས་ཚིང་བ་རྐང་པ་གཉིས་ཀྱིས་བདག་གི་རྣམ་གྲངས་ཏེ། ཀྵེ་ཏྲ་ཛྷཱ་ནི་ཞིང་ཤེས་ཏེ། ཞིང་ནི་ལུས་སོ། །ཨཱཏྨ་ནི་བདག་སྟེ། ཨཏ་འགྲོ་བ་ལ། བྱོ་མ་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པས་མ་ནིན་རྐྱེན་དང་རིང་པོའོ། །པུ་རུ་ཥ་ནི་སྐྱེས་བུ་སྟེ། སྲིད་སྐྱོང་བ་ལ། ཀུ་ཥན་རྐྱེན་དང་། པུར་དུ་བསྒྱུར་བའོ། །འདུས་སྒྲཊ། ཚིགས་བཅད་ཕྱེད་ཀྱིས་སྲོག་ཆགས་ཀྱི་མིང་སྟེ། པྲ་ཎི། །སྲོག་ཆགས་ཏེ་སྲོག་རྣམས་ལུས་ལ་འཛིན་ཅིང་ལྡན་པས་སོ། །ཙེ་ཏ་ན་སེམས་ཅན་ནོ། །ཛནྟཱི་སྐྱེ་ལྡན་ནོ། །ཛནུ་དང་ཛ་ཅུ། སྐྱེ་བོའམ་སྐྱེ་དགུའོ། །::། ཤ་རཱི་རི་ཎ་ལུས་ཅན་ནོ། །འོན་ཁ་ཏོན་ལོག་ཁྱིཚིགས་བཅད་ཕྱེད་ཀྱིས་སེམས་ཀྱི་མིང་སྟེ། །ཙིཏྟ་ནི་སེམས་པར་བྱེད་པས་སེམས་སོ། །ཙེ་ཏ་སེམས་པའོ། །ཧྲི་ད་ཡཾ། སྙིང་ངོ༌། །སྟནྟཾ་ཡིད་དོ། །།།ཧྲི་ད་སྙིང་པོའོ། །མཱ་ན་སཾ་ཡིད་བྱེད། མ་ན་ཡིད་དོ། །།པ་ཏོ་ཀན་ཁ་ལ། ཚིགས་བཅད་གཅིག་གིས་བློའི་མིང་སྟེ། བུདྷི་ནི་བློ་སྟེ། །ཡིད་བུམ་པ་དང་སྣམ་བུ་དང་ཤིང་རྟ་ལ་སོགས་པ་ལ་རྟོགས་པར་བྱེད་པའོ། །།མ་ནཱི་ཥ། ཡིད་འདོད་དམ། ཡིད་འགྲོའོ། །དྷི་ཥ་ཎ། སྤོབས་པ་ཅན་ནོ། །དིསེམས་པར་བྱེད་པས་བློའོ། །ཤེ་མུ་ཏཱི། ཉལ་སློང་ངོ༌། །མ་ཏི་བློ་གྲོས་སོ། །པྲ་ཀྵ། རབ་ཏུ་ལྟ་བའམ་རྟོགས་པའོ། །ཨུ་བ་ལབྷི་ཉེ་བར་དམིགས་པའོ། །།ཙེ་ཏ་་་སེམས་སོ། །སམྦི་ད། །ཡང་དག་རིག་གོ། །པྲ་ཏི་པྲ་ཛྷསྟི། སོ་སོར་རྟོག་པའོ། །ཙེ་ཏ་ན། སེམས་པའོ། །དྷཱམི་བརྗེད་པ་དང༌། དྷཱ་ར་ཎ་བ་ཏཱི། གཟུངས་ཀྱི་ནུས་པ་དང་ལྡན་པ་དང་། མེ་དྷཱ་ནི་ཡིད་གཞུངས་པ་དང་། །སཾ་ཀལྤ་ནི་ཡང་དག་པའི་རྟོག་པ་སྟེ་དེ་དག་བློ་ཁྱད་པར་ཅན་ནོ། །ལོ་བ། ་་་་་ཁྱག་ཁ་ན་བ། ལ་ཡིད་བྱེད་ཀྱི་མིང་ལ། །ཀརྨ་མཱ་ན་སཾ། ་ཡིད་ཀྱི་ལས་དང༌། །ཙིཏྟ་བྷོ་ག་—།
