ཏྲི་ཏཾ་ཡཾ་གུ་རུ་ཡུལྟོ་བོ་ག་བ་ཏྭཱ་ཡཾ་ཛཾ་བྷ་ཏིཏཾ་ཡབློ། ,ཡི་ཏ་ཛ་རེ་ཕ་གསུམ་ཅན་ལས་ལྕི་བ་དང་ལྡན་ཞིང་། །གལ་ཏེ་བྷ་གསུམ་ཅན་ལས་ག་དག་ཤུགས་ལྡན་མ། །གང་ལ་ཟུང་ཨ་ཡུཧྥེ་ཡུསྶཾབྷ་དྲ་བི་རཱརྨ་སོ་ཛ་གོ་ག། བོ:མི་སཾ་མི་ཡ་དེས་ཛོ་སྒོཀེ་ཏུ་མིན་ཏ་ཛ་རེ་ཕ་ལྕི་བ། །ཟུང་ལ་མ་ས་ཛ་ལྕི་ག་ནི་བཟང་པོའི་སྣང་། །གལ་ཏེ་མི་མཉམ་ས་ཛ་ས་དང་ལྕི་བ། །མ་ཏཱི་བྷ་རེ་ཕ་ན་ག་ཐཱཤྩ། ,། ཨཱ་ཁྱུ་ནི་ཀོ། དྨ་དི་ཏཾ་ཛ་གོ་ག་སྟེ་ཡུཏྤ་དེ་ཡུཏྟཾ་ཛ་ཏོ། མཉམ་པ་བྷ་ར་ན་ག་ལྕི་བ་ཏོག་ལྡན་མ། །གལ་ཏེ་རྐང་པ་ཟུང་མིན་ཏ་དག་ཛ་ལྕི་བ། །ཟུང་ནི་ཛ་ཏ་ཛ་ག་ལྕི་ཛ་གོ་གོ། ཛ་ཏོ་ཛ་གོ་གླུ་ཏྲ་ཙ་ཏོ་ཛ་གོ་གཱ་ཨཱི་ཁྱུ་ནི་ཀཱ་སོ་བི་བ་རཱིཏ་པུཊྚེ། ཀུན་ཀུན་བསྒྲགས། །འདིར་ཡང་ཛ་ཏ་ཛ་ག་ག་ཏ་དག་དང་ཛ། །ག་ལྕི་དེ་ནི་ཀུན་བསྒྲགས་བཟློག་པ་སྔ་མ་ཅན། །བསྒྲགས་གཉིས་ནི་ཉེར་བསྐྱེད་བདུན་པ་དང་མཐུན། ཡ་གལ་ཏེ་ས་གསུམ་ཅན་ལས་ལ་དང་ག་ཡིན་ཅིང་། །ན་བྷ་བྷ་ར་ཉྩ་ཏ་བཱ་ཧ་རི་ཎ་པཱུ་ཏཱ། འདི་རོ་ལངས་ཅན་ཡང་ཡིན། ཡོ་དི་ན་ན་ར་ལ་གཱ་ན་ཇོ་ཛ་རོ་ཡི་རྣམས་ཀྱང་དེ་ཚེ་རི་དྭགས་འཕྱོང་། །གལ་ཏེ་ན་ན་ར་དང་ལ་དང་ག་ཡིན་ཅིང་། །དི་ཙ་ཏ་དཱཉཔ་ར་བོཀྟ་མི་ཏེ། འདི་རོ་ལངས་ཅན་གྱི་བྱེ་བྲག་ཉེ་བའི་སྡེབ་སྦྱོར་ཡང་ཡིན། ཨ་ཡུ་གལ་ཏེན་ཛ་ཛ་ར་དེ་ཚེ་གཞན་སྒོར་འདོད། །གལ་ཏེ་འདིར་ནི་ཟུང་མིན་ན་དགར་གི་ཧ་ཡ་དི་ནི་ར་ཡཾ་ཙཡུསྶཾ་ན་ཛཛ་ར་གཱཀཾ་ཧྥ་ཡིནྣཾ་པུཊྛིཏཱགྲོ། ,རེ་ཕ་ཛོར་ཡའོ་ཟུང་བ། །ན་ཛ་ཛ་ར་ག་རྣམས་མེ་ཏོག་རྒྱས་པའི་རྩེ་མོར་བརྗོད། །ར་ཛ་ར་ཛ་དེ་བཞིན་ཛི་ཏ་༔ཛརཾརྫཀཾ༔་ཡ་བརྒྱན༔རྟཡཱ་ས་རྨབྷི༔་ནི། ,ད་་ད་ཐ་བ་ས་ཛ་ར་ཛ་ར་ག་རྣམས། །ཡ་བའི་སྒྲ་ལས་མ་ཏུ་ས་ཡིན་ཏེ་བུད་མེད་མཚུངས་པར་བརྗོད། །གལ་ཏེ་མི་མཉམ་་ལ་སྒྱུ་ཙ་ཡ་དེ་བི་ཥ་མ་མི་ཏ་ར་ད་བི་ཙ་ད་ཤ་ད་ཤ་ད་ཤ་ལ་སྒོ་བཅུ་དང་བཅུ་དང་ནོར་ལྡན་དག་ནི་ཡང་བ་ཡིན་ལ་གཅིག་ཤོས་དག་ཀྱང་བཅུ་དང་བཅུ་དང་བཅུ་རྣམས་ཡོང་། ཀཾ་གུ་རུ་ནི་དྷ་ན་མུ་བྷ་ཡ་མ་པི་པི་ཧི་ཏ་ལ་ལི་ཏ་པ་ད་བི་སྲི་ཏི་ཡ་དི་བྷ་བ་བ་ཡིན་ཞིང༌། །གལ་ཏེ་ཐ་མ་གཉིས་ཀ་དག་ཀྱང་ལྕི་བར་བྱས་པ་རོལ་པའི་རྐང་པ་རྣམ་པར་སྤྲོ་བར་གྱུར་པ་་།
