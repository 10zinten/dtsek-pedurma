བསྟན་འགྱུར། སྒྲ་མདོ། སེ་ཡུསྶཾ་བྷ་གྲུ་ཙ་བྲྀཏྟི་ཡུ་ག་བྷཱ་ཀྲི་བྲྀཏྟཾ་ཀཾ། །སདྨ་བྲྀཏྟ་ཀ་ཡུ་ཛཱཉབ་རཱནྟི་ཀཱ། །ཨ་ཡུསྶཾ་འཕེལ་རབ་འཕེལ་ལོ། །རབ་འཕེལ་ཟུང་ནི་གཞན་གྱི་མཐའ་ཅན་འགྱུར། །ཟུང་མིན་རྐང་པས་དིཤུ་རུ་ཧཱ་སི་ནཱི། །བེ་ཏཱ་ལཱི་ཡ་ཛ་ཏ་ཡ། །་ཊ༌ཀ་ལང་ར་ཎི། ཤོམཛེས་རྒོད་མའོ། །རོ་ལངས་ཅན་གྱི་ཛཱ་ཏི་རྣམས་སོ། །།།ལྟ་སྐུལྷ། ཝ་འ་ལ་འཚེ་ཚུད་པ»ཡིབ་ལས་ཡངསཨ་གུ་རུ་ར་ཙ་ལ་དྷཱི་ཏི་རུ་ད་དྷི་ཙི་པ་དི་པ་དི། །གནྟེ་མཱ་ཏཱ་ས་མ་ཀཾ་ལི་ད་། རྐང་པ་རྐང་པར་ཆུ་གཏེར་ཙ་སྟེ་ལྕི་མེད་མི་གཡོ་བརྟན། །མཐར་ལྕི་འདབ་ལས་ཡང་བ་ཕྱི་མོ་ལཱཏ། །བྷོ་ཡ་དི་ཙ་དཱི་ཏ་ཡེ་དུ་བ་ཙི་ཏཱ། །ཙཱཏྠ་ཛ་ཨེ་བ་བཱ་བི་བློ་ཀ། །ཁོ་མཉམ། །གལ་ཏེ་ཙ་གཉིས་ལས་བྷ་ཉེར་བཀྲ་ཡིན། །ཙ་ལས་ཁའམ་ཛ་ནི་ཁྱད་ཚིགས་བཅད། །ཕྱེད་ཛོ་བཱ་བཱ་ན་བཱ་སི་ཀཱནྟཱ་ཏྲ། །ཙ་ཁྱདྨ་དི་བ་དྲི་ཏཾ་ཡཾ་ཙི་ཏཱ། །ས་མ་ནེན་ར་ནི་ག་དི་ལས་ཁའམ་ཛ་ནི་ནགས་ཁྲོད་གནས། །ཙ་ཞེས་བྱ་ལས་གལ་ཏེ་ས་གཉིས་བཀྲའོ། །བར་མ་ཆད་དུ་བཤད་ཏ་ལཀྵ་ཎ་བྷཱཏ། མཱ་ཏྲ་ས་མཱ་དི་པཱ་དཱ་ཀུ་ལ་ཀཾ། །རྐང་པ་དང་པོ་གཉིས་ཁྱད་པར་ཚིགས་བཅད། པའི་མཚན་ཉིད་བསྟན། །ཕྱི་མོ་མཉམ་སོགས་རྐང་པ་དཀྲུགས་པ་ལས། །ཨ་བི་ཏི་ཏེ་བྲྀཏྟ་པ་རི་མཱ་པ་རི་ག་ཏཾ། ་། གསུམ་པ་ནག་ཁྲོད་གནས། མུ་ནི་ནཱཉཉྩ་དྷཱ་པི་བཱ་བྲྀཏྟའི་ཚད་རྣམས་རྟོགས་པ་མེད་གྱུར་པ། །ཐུབ་པས་རྐང་འཁྲུགས་སུ་ནི་མངོན་ཏབཞི་པ་ཁྱད་པར་ཚིགས་བཅད། དཱ་ཀུ་ལ་ཀཾ། །གྲངས་ཡིག་གི་བརྡ་བསྒྲིག་རྒྱུ། པྲ་ཏི་ད་ལ་མུ་པ་ཏི་ཏང་པར་བརྗོད། །འདབ་མ་སོ་སོར་བཅུ་བཅུ་བཅུ་ནམས་ཡང་བ་བཀོད་མཐར་ལྕེ་བྱས་ཤ་ད་ཤ་ལ་ར་བ་སི་ཏི་བི་ར་ཙ་ཏེ་གུ་རུ་ར་ཏི་རུ་ཙི་རཱ། །ཨ་་བཏྠོ་ཙཱདྷེ་ཡ་སཱུ་?ཤིན་ཏུ་གསལ་མཛེས་སོ། །གང་གི་ཕྱེད་ལ་བརྒྱུད་བརྒྱད་དག་ནི་གང༌། །གཞན་ཡང་གཱསྱཱསོཀྟཉནཾ་ག་ཀཱི་ཎཱ། ས་ཀ་ལ་མ་པ་ར་མ་བི་ས་ལི་ལ་ནི་དྷི་གུ་ཎི་ཏ་བ་སུ་ལ་མཐའ་དག་ཆུ་གཏེར་ནོར་གྱིས་བསྒྱུར། །ཡང་བ་ཕྱི་མར་བཀོད་པའི་རྐང་པ་སྤྲོ། །དེ་ནི་ལུས་སྐལ་།
