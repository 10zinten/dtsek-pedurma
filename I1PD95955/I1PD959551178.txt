དུ་དུ། ས་གུ་རུ། མ་དེ་རྒྱ་དེ་ལ་ལུ་བ། ད་དེ་སུ་་སྤྱི་བ། ས་ཡུལ་ཕྱི་བ། མ་ཀུན་བུ་མང་བ། བྱ་བ། ཡང་། བདག་མ། ་ལ་རྒྱུས་ལ་བསྡུར་མཆན། གང་དུ་སྐད་:ཁྱད་དེ་གའི་དམག་པ། ག་པ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཡང་གོ་པོཡོད་པ་་ད་ཡ། དྲི་བ་དེ་དག་མ་ནུས་ལྡན་ལེགསན་སྦྱོང་བ་དང་བསྡུར་འབྲས་རེའུ་མིག་་མག་པསྒྲ་མདོ། (བ་བ་དཔེ་གཞི། 《སྡེ་དགེ། ::པེ་ཅིན། སྣར་ཐང༌། པ། ཅོ་ནེ། །སྡེ་ཚན། སྒྲ་མདོ། སེཔོད་རྟགས། ཚན་གྲངས། །ལྡེབ། བ_བ་ན?ན། :ཚེག་ཁྱིམ། :གཏེར། ན་མ་སོང་། མི་ན་བ་གནས་ནི། ཡབ་གསང་བ་བདེ་བ་དང་། :ཕྱི་པ་ཡིན་ལ། དེ་ཉིན་ས་ར་ཐུལ། ང་དང་། །ཐོད་པའི་བྱ་བ། ་ཊ་་་་ལ་ལྟར་གས་པ་ཀཉི་ཤོ་ད་ན་ཉན། ཨཁ་ན་བ། ལུ་བ། །ཁོ་བོ་བསམ་ਲਵਜੋਸਲਮਰਉਸ'ਸੋਉਹਛਕਕਾਰਦਿੱਤਾ।ਉਹ_།
