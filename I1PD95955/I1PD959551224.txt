ཟུང་ལ་ནི། །གལ་ཏེ་བྷ་གསུམ་ཅན་ལས་ཕ་རོལ་ཏུ་ལྕི་བ་དག་ཡིན་པ་དེའི་ཚེ་་་ཤུགས་ལྡན་མའོ། ཀྵ་། ཡ་དི་ས་ཏྲི་ཏ་ཡཾ་གུ་རུ་ཡུཀྟ་བེ་ག་བ་ཏཱི་ཡ་དི་བྷ་ཏྲི་ཏ་ཡཱངྒོ། གང་ལ་ཟུང་མིན་ཏ་ཛ་རེ་ཕ་ལྕི་བ། །ཟུང་ལ་མ་ས་ཛ་ལྕི་ག་ནི་བཟང་པོའི་སྣང་། ཥཌ། གང་ལ་རྐང་པ་ཟུང་མིན་ཏ་དང་། ཛ་དང་ར་དང་ག་ཡིན་ཞིང༌། །རྐང་པ་ཟུང་ནི་མ་དང་ས་དང་ཛ་དང་ག་དང་ག་ཡིན་པ་དེ་བཟང་པོའི་སྣང་བའོ། །ཡཧཱུཾ་ཏ་ཛ་རེ་ཕ་གཱ་་ཨ་ཡུགློ་ཡུསྶཾ་བྷ་དྲ་བི་་་རཱརྨ་་སོ་ཛ་གོ་ག། ：L་གལ་ཏེ་མི་མཉམ་ས་ཛ་ས་དང་ལྕི་བ། ལ། །མཉམ་པ་བྷ་ར་ན་ག་ལྕི་བ་ཏོག་ལྡན་མ། །གལ་ཏེ་རྐང་པ་ཟུང་མིན་ས་དང་ཛ་དང་ས་དང་ག་ཡིན་ཞིང་། །རྐང་པ་ཟུང་ནི་བྷ་དང་ར་དང་ན་དང་ལྕི་བ་དང་ག་ཡིན་པ་དེ་ཏོག་ལྡན་མའོ། །བི་ཥ་མཾ་ས་མཾ་ཡ་དི་ས་་ཛོསྒོ་ཀེ་ཏུ་བ་་ཏཱི་བྷ་རེ་ཕ་ན་ག་གཱ། གལ་ཏེ་ཟུང་མིན་རྐང་པ་ཏ་ཏ་་དག་ཛ་ལྕི་ག། ཟུང་ནི་ཛ་ཏ་ཛ་ག་ལྕི་རྣམས་ཀུན་བསྒྲགས་སོ། །གལ་ཏེ་རྐང་པ་ཟུང་མིན་ཏ་དག་དང་ཛ་དང་ལྕི་བ་་དང་ག་ཡིན་ཞིང་། ཚོ་ཟུང་གི་རྐང་པ་ལ་ཛ་དང་ཏ་དང་ཛ་དང་ག་དང་ལྕི་བ་ཡིན་པ་དེའི་ཚེ་ཀུན་བསྒྲགས་སོ། །ཨཱ་ལྱཱ་ནི་ཀཱ་སཱུརྱ་དི་ཏོ་ཛ་གོ་:གསཱུ་ཡུགྲ་དོ་ཡུཥ་ཛ་ཏོ་ཛ་གོ་ག། །/འདིར་ཡང་ཛ་ཏ་ཛ་ག་ག། ཏ་དག་དང་ཛ་ག་ལྕི་བ་དེ་ནི་ཀུན་བསྒྲགས་བཟློག་པ་སྔ་མ་ཅན། །གང་ལ་ཟུང་མིན་རྐང་པ་ཛ་དང་ཏ་དང་ཛ་དང་ག་དང་ག་ཡིན་ཞིང༌། །།ཟུང་གི་རྐང་པ་ནི་ཏ་དག་དང་ཛ་དང་ག་དང་ལྕི་བ་ཡིན་པ་དེ་ནི་བཟློག་པའི་ཀུན་བསྒྲགས་ཀྱི་སོ༑། ཛ་ཏོ་ཛ་གོགྲུ་ཏྲ་ཙ་ཏོ་ཛ་གོ་གཱ་ཨཱ་ལྱཱ་ནི་ཀཱ་ྋསོ་བི་པ་རཱི་ཏ་པུཏྟཱ། L་པ་ཀུན་བསྒྲགས་གཉིས་ནི་ཉེར་བསྐྱེད་ཀྱི་་།
