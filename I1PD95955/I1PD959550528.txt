ཧུ་ཡ་ཏ་ཏྲ་སྱཱཏྠ་ཏི་ཤཱ་ས་ནཾ། །ས་སཾ་ལྟ་བཀྲ་ཏུ་ཥཱ་ཡཱསྐུ་སྟི་བྷཱུ་མིརཱི་ཛརྨ་ནཾ། །སླར་ཡང་བསྒོ་བའོ། །མཆོད་སྦྱིན་དང་ནི་མཆོད་པ་དང་། །བསྟོད་པའི་ས་གཞི་གཉིས་སྐྱེས་རྣམས། །ནི་དྷཱ་ཡ་ཏགླུ་ཏེ་ཡ་ཏྲ་ཀཱཥཱ་ཀཱཏྟཾ་ས་ཨུདྷ་ན། །སྟམྦསྨཱ་སཱུསྟམྦ་གྷ་ན༔ས་སྟཝོ་ཤིང་སྟན་ཤིང་སྟེགས་གང་ལ་ནི། །ཤིང་སྟན་ཤིང་ནི་ཨུདྷ་ན། །སྡོང་པོ་འཇོམས་དང་སྡོང་འཇོམས་སོ། །ཡེ་ན་ནི་ཧ་ནྱ་ཏེ། །ཨཱ་བི་དྷོ་བི་དྷ་ཏེ་ཡེ་ན་ཏ་ཏྲ་བི་ཤྭ་ཀ་ས་མེ་ནི་གྷ། །དེ་སྡོང་གང་གིས་ངེས་འཇོམས་པ། །ཀུན་བརྡོལ་མཐོ་བ་གང་དེ་ལ། །སྣ་ཚོགས་ཅན་མཉམ་ནི་གྷའོ། །ཨུཏྟཱ་རཱཉྩ་ནི་ཀཱ་ར༔་དྲོ་དྷཱ་རྩོཎི་པ་ནཱརྦ་ཀོ། །ནི་གཱ་རོདྡཱ་ར་བི་ཀཱ་བོདྡཱ་ཧཱསྐུ་ག་བྱུར་བུ་དང་ནི་ཕུང་པོ་དང་། །སོ་བ་སྟེང་དུ་འདྲེན་པའི་དོན། །སྒྲིགས་པ་དང་ནི་གྱེན་བཟློག་དང་། །ར་ཎཱ་དི་ཞུ། །ཨཱ་རཱ་ཏྱཱ་བ་ར་ཏི་བི་ར་ཏ་ཡ་ཨུ་པ་རཱ་མེ་བཱསྟི་ཡཱནྟུ་ནེཥཱ་བ། །གསུས་པ་སྒྲེགས་པ་ལ་སོགས་སོ། །ཉེར་འགྲོ་བར་ཏི་བི་རྟ་ཡ། །ཨུ་བ་ར་མེ་མོ་མིན་ནོ། །གློ་ལུ་དང་ནི་ལུད་པ་ལུ། །ནིབྷཱུ་ཏིནིཥྛ་བ་ན་ནི་ཊཱི་བ་ན་མི་ཏྲ་བྷིནྡྲ་ནི། །ཛ་ནེ་བེ་་ཏིས་ཏི་སྣ་བ་སཱ་གློ་འཁོགས་དང་ནི་ལུད་པ་འོག། འཁོགས་པ་ཞེས་པ་ཐ་དད་མིན། །སྐྱུགས་སྐྱུ་པ་ཏི་རིག་བྱེད་དོ། །ནཾ་སྭཱ་ད་ཐ་ཛཱ་རོ་ཛཱནྟི། །ཨུ་ད་ཛསྐུ་པ་ཤུ་པྲ་ར་ཎ་མ་ཀ་ར་ཎི་རི་ཏཱ་ད་ཡཐཱ་པཱ། །དེ་ནས་རིམས་དང་ཚད་པ་དང༌། །ཨུ་ད་ཛ་ན་ཕྱུགས་བསྐུལ་བ། །བྱ་མིན་ཞེས་སོགས་སྨོད་མོའོ། །གོ་ཏཱནྟེ་བྱཱསྟ་སྱབྷནྡ་མི་ཏོ་པ་ག་བ་ཀཱ་དི་ཀཾ། །ཨཱ་པཱུ་པི་ཀཾ་ཤཱསྐུ་ལི་ཀ་མེ་རིགས་མཐར་དེ་ཡི་ལོ་ཚིགས་ལ། །ཨོ་པ་ག་པ་ཀཱ་ལ་སོགས། །བེམས་པོ་སེམས་མེད་རིགས་མེད་དང་། །དེ་བ་མཱ་རྱ་མ་ཙེ་ཏ་སཾ། །མཱ་ན་བཱ་ནཾ་ཏུ་མཱ་ཎ་ཧུཾ་ས་ཧཱ་ཡཱ་ཏཱཾ་ས་ཧཱ་ཡ་ཏཱ། །བཞིན་སེམས་མེད་ལ་སོགས་པ། ་། བྱིས་པ་དང་ནི་བུ་ཚ་བའོ། །ལྷན་ཅིག་འགྲོ་དང་དུས་གཅིག་འགྲོ། །ཧ་ཝཱ་ཧ་ལཱ་ནཾ་བཱ་སྒྲ་ཏྱཱ་བཱ་ཌ་བྱེ་ཏུ་དཱི་ཛནྣཱ་ནཾ། །དྲི་པརུ་ཀཱ་ཐོང་གཤོལ་ཅན་དང་གཤོལ་ཚིགས་སོ། །བྲམ་ཟེ་སྨྲ་པོ་གཉིས་སྐྱེས་རྣམས། །གཉིས་པོ་རྩིབ་ལོགས་ནཾ་པྲྀཏྟཱ་ཎཱ་པཱནྟཾ་པྲྀཥྛ་མ་ནུ་ཀྲ་མཱཏ། །ཁ་ལཱ་ནཾ་ཁ་ལི་ནཱི་ཁ་ལྕེ་པྲ་ཐ་མཱ་རྒྱབ་རྣམས་ཀྱི། །པཱཧཱུྃ་ཕྲིཥྛ་གོ་རིམས་བཞིན། །གཡུལ་ས་གཡུལ་ཐང་གཡུལ་ཁའོ། །དེ་ནས་མི་—།
