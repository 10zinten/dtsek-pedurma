།མ་ལཱ་ས་ན། །སྲཏྟཱ་པྲ་ཛཱ་ས་ཏིབྦེ་དྷཱ་བི་དྷཱ་ཏཱ་བི་སྭ་སྲིཤྲཱི་དྷི། །།།།།་ཌོ་ལ་སྒྲུབ་བྱེད་པདྨའི་གདན། །སྦྱིན་བྱེད་སྐྱེ་དགུའི་བདག་པོ་དང་། །རིག་བྱེད་རྣམ་བྱེད་སྣ་ཚོགས་བྱེད། །།།༽༈བིཥྛནཱ་རཱ་ཡ་ཎཀྲྀཥཱ་བི་ཀུཎོ་བིཏྟཱ་ར་ཤ་བཱཨ་ཤཱཏྨ་ཐོས། ་། དཱ་མོ་ད་རོ་ཧྲ་ཏཱི་། །༈ཁྱབ་འཇུག་ནག་པོ་སྲེད་མེད་བུ། །རྟུལ་མོའི་བུ་དང་རྣ་བ་ལེགས། །དབང་པོའི་དབང་ཕྱུག་མེ་ཏོག་ཀེ་ཤཀེ་ཤ་བོ་མཱ་དྷ་བསྭ་བྷཱུ། །དི་ཏཱ་རི་པུཎྜ་རཱི་ཀཱ་ཀློ་གོ་བིནྡོ་ག་ལྟོ༑:། སྒྲ་ཅན་གསོད་དང་སྦྲང་རྩིའི་བུ། །རང་སྐྱེས་སྦྱིན་དགྲ་པད་དཀར་མིག། མཁའ་ལྡིང་རྒྱལ་རུ་ཌ་དྷ་ཛ། །པཱི་ཏཱལྤ་རོཉཙུ་ཏ་སཾ་གཱི་བི་ཤཱཀེ་ནོ་ཛ་ནཱཌྷ་ན། ་། ཨུ་མཚན་གླང་རྫིར་གྲགས། །མི་འཛག་གོས་སེར་རྭ་གཞུ་ཅན། །སྣ་ཚོགས་སྡེ་དང་འགྲོ་བ་འཇོམས། །སྟེང་པེནྡ་ཨིན་བ་ར་ཛཉྩཀྲ་པཱ་ཎི་ཏུརྒྱུ་ཛ། །པདྨ་ནཱ་བྷོ་མ་དྷུ་རི་གི་དབང་པོ་དབང་གཞན་སྐྱེས། །ལག་པ་བཞི་པ་འཁོར་ལོའི་ལག། པདྨའི་ལྟེ་བ་སྦྲང་རྩིའི་དགྲ། །པུཏྲཱ་སུ་དེ་བསྟི་བི་ཀྲ་མ། ་། དེ་བ་ཀཱི་ནནྡ་ན་སོ་རིཤྲཱི་པ་ཏིཔུ་རུ་འདུག་གནས་ལྷ་དང་གོམ་གསུམ་གནོན། །ལྷ་མོ་དགའ་བྱེད་དཔའ་བོའི་རྒྱུད། །དཔལ་གྱི་བདག་པོ་སྐྱེས་ཊོཏྟ་མ། །བ་ན་མཱ་ལཱི་བ་ལི་དྷཱནི་ཀཾ་སཱ་རཱ་ཏི་ར་དྷོཉཀྵ་ཛ། །བི་ཤྭ་བུ་མཆོག། མེ་ཏོག་ཕྲེང་ལྡན་སྟོབས་ལྡན་འཇོམས། །འཁར་བའི་དགྲ་དང་འོག་དབང་སྐྱེས། །ཀུན་བྷ་རཀི་ཊ་བྷ་ཛིདྲི་དྷུབྷཾ་བཏྠ་ལཉྩ་ན། །བ་ཅེསུ་དེ་བོཉམྱ་ཛ་ན་ཀས་འགེངས་ལྷ་མིན་འཇོམས་པ་དང༌། །སྡེབ་བྱེད་དཔལ་གྱི་བེའུས་མཚན། །འདི་ཡི་ཕ་ནི་ནོར་ལྷ་སྟེ། །ཨེ་བཱཉན་ག་དུནུ་བྷི། ་། བ་ལ་བྷ་དྲཔྲ་ལམ་གློ་བ་ལ་དེ་བོཉཙུ་ཏཱ་གྲ་ཛ། །རྔ་བོ་ཆེ་ནི་ཨ་ན་ཀ་ཁ། སྟོབས་བཟངས་དང་ནི་རབ་འཕྱང་གསོད། །སྟོབས་ལྷ་མི་འཛག་སྔོན་སྐྱེས་དང༌། །རེ་བ་ཏཱི་ར་མ་ནོ་རཱ་མཀཱ་མ་པཱ་ལོ་ཧ་ལཱ་ཡུ་དྷ། །ནཱི་ལཱལྤ་རོ་རོ་ཧི་ནམ་གྲུའི་བདག་པོ་དགའ་བྱེད་དང་། །འདོད་སྐྱོང་གཤོལ་གྱི་མཚོན་ཆ་དང་། །གོས་སྔོན་ཅན་དང་སྣར་མའི་ཙྩེ་ཡལྟ་ལོ་ཀོ་མུ་ཥ་ལཱི་ཧ་ལཱི། །སངྐརྟ་ཎཔཱི་ར་པཱ་ཎིཀཱ་ལིནི་བུ། ཏཱ་ལས་མཚན་དང་གཏུན་ཤིང་ཅན། །ཐོང་ལྡན་ལེགས་འགུགས་ལག་ན་གཤོལ། །ཀླུ་འཇོམས་—།
