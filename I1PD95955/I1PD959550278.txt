བྱིངས་ཧ་ས་སོ་སོར་བསྡུ་བའི་ཡི་གེ་དང་པོ་ཅན་ལ་ཕུལ་དུ་བྱུང་བ་དང་ཡང་ཡང་གི་དོན་ལ་བྱིངས་བྱིངས་སུ་སྒྲུབ་པའི་རྐྱེན་ཡང་འགྱུར་ཞིང་། རྐྱེན་ཡང་ཕ་རོལ་ལ་་་བྱིངས་དེའང་གཉིས་སུ་འགྱུར་ཏེ། བྷཱུ་ལ་ཡང་ང་ཨཱཏྨ་ན་པ་ད་སྦྱོར་བའི་དོན། གཉིས་བརྗོད། ཛྷ་བ་རྣམས་ཀྱི་ཛ་བ་དང་ཙ་བ་འོས། བྷབྲ། བུ་བྷཱ་ཡ་ཞེས་་་གནས་པ་ལ། ༈ཡ་ང་། ཡང་ལའོ། །བྱིངས་ཀྱི་དང་པོ་ནཱ་མིའི་དབྱངས་ནི་རྐྱེན་ཡང་ཕ་རོལ་ལ་ཡོན་ཏན་དུ་འགྱུར་ཏེ། ཡོན་ཏན་བྱས་པས། བོ་བྷྱཱ་ཡ། །དེ༈སདྷཱ་ཏཿདེ་བྱིངས་སོ། །ཡང་ལ་སོགས་པའི་རྐྱེན་སྦྱར་བ་དེ་ལའང་་་བྱིངས་ཞེས་ཟེར་རོ། །དེས་ན་བྱིངས་དེ་ཚིག་ཏུ་བསྒྲུབ་་པའི་ཕྱིར་རྣམ་དབྱེ་་་རྣམས་སྦྱོར་བ་ལས། ཨཔྲ་བི་ཀ་ར་ཎ་སྦྱོར་བ་སོགས་གོང་ལྟར་བསྒྲུབས་པས། །བརྟ་མ་ན་ལཊ་ལ། བོ་ངྷ་ཡ་ཏེ། བོ་དྷུ་ཡེ་ཏེ། བོ་ངྷ་ཡནྟེ་སོགས་དང་། ཆོག་འབྱུང་ལིང་ལ། བོ་དྷྱ་ཡེ་ཏ། བོ་ངྷ་ཡེ་ཡཱ་ཏྭཾ། བོ་དྷྭ་ཡེ་རན་སོགས་དང༌། ཤིས་བྱེད་ལོཊ་ལ། བོ་བྷྱཱ་ཡ་ཏྭཾ། བོ་ངྷ་ཡེ་ཏྭཾ། བོ་དྷཱ་ཡནྡྷཾ་སོགས་དང༌། དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་བོ་བྷྱཱ་ཡ་ཏ། ཨ་བོ་བྷུ་ཡེ་ཏྭཾ། ཨ་བོ་ངྷ་ཡནྟ་་་སོགས་སོ། །དེ་བཞིན་དུ་བྷུཛྲ་སྐྱོང་བ་ལ་། བོ་དྷྭ་ཛུ་ཏེ། ལཊ། ་བོ་བྷུ་ཛེ་ཏ༑ཡལིང༌། ་བོ་བྷུ་ཛ་ཏྭཾ། ལོཊ། ཨ་བོ་བྷུ་ཛུ་ཏ་ལང་སོགས་སོ། །༈བྷཱཥཱ་རྒྱན་ལ། བོ་བྷཱ་ཥྱ་ཏེ། ལཊ། བོ་བྷ་ཀྵེ་ཏ། ལིང༌། བོ་བྷ་བྱ་་་་ཏྭཾ། ཡལོཊ། ཨ་བོ་བྷུ་ཥྱ་ཏ་ལང་། སོགས་སོ། །ཡ༈མུ་ཧ་སེམས་ཉམས་པ་ལ། མོ་མུ་རྒྱ་ཏེ། ལཊ། མོ་མུ་བྱེ་ཏ། ལིང་། མོ་མུ་རྒྱ་ཏྭཾ། ལོཊ། ཨ་མོ་མུ་ནྱ་ཏ་ལང་སོགས་སོ། །བདག་༈ལི་ཧ་རོ་མྱང་བ་ལ། ལེ་ལི་བྱ་ཏེ། ལཊ། ལེ་ལི་བྱེ་ཏ། …ལང་། ལེ་ལི་ཧྥུ་ཏཱ༑ཡལོཊ། ཨ་ལེ་ལི་རྒྱ་ཏ་ལང་སོགས་སོ། །དེ་ལ་པའི་བྷཀྱིས་occ།
