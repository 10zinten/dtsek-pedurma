བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽yསོཀྵཉནཾ་ག་ཀྲྀ་ཌཱ། །ས་ཀ་ལ་མ་པ་ར་མ་བི་ས་ལི་ལ་ནི་དྷི་གུ་ཎི་ཏ་བ་སུ་ལ་སྒྱུ་གཞན་ཡང་མཐའ་དག་ཆུ་གཏེར་ནོར་གྱིས་བསྒྱུར། །ཡང་བ་ཕྱི་མར་བཀོད་པའི་རྐང་པ་སྤྲོ། །དེ་ནི་ལུས་ཀ་མུ་པ་རི་གྷ་ཊི་ཏ་པ་ད་བི་སྲི་ཏི། །པྲ་ཏི་ད་ལ་མུ་པ་ཧི་ཏ་ད་ཤ་ད་ཤ་ད་ཤ་ལ་གླུ་ར་མེད་རྩེན་དུ་བརྗོད་པ་ཡིན། །འདབ་མ་སོ་སོར་བཅུ་བཅུ་བཅུ་རྣམས་ཡང་བ་བཀོད་མར་ལྕི་བ་སི་ཏི་བི་ར་ཙི་ཏ་གུ་རུ་ར་ཏི་རུ་ཙི་རཱ། །ཙ་ད་ཀ་ད་ཀ་ཏྲི་ཙ་ཛ་གི་ཏི་ཁཎྜ་མ་ཛོ་བྱས་ཤིན་ཏུ་གསལ་མཛེས་སོ། །ཛ་མེད་རང་གི་དང་པོ་ཡིན་པ་བྷ་བྲལ་གང་ཞིག་ཙ་ད་ཀ་ད་ཀ་དང་བྷ་ར་ཧི་ཏ་མ་ནཱ་དི་མེ། །མཱ་ཏཱ་ས་མ་ཀ་ཛཱ་ཏཱ་ཡ། །ཁྲོ། ལྷ་ས་དྲ་བ་ཁ་ལ་སོགས་པ། ཙ་གཞེས་དུམ་བུ། །ཕྱི་མོ་མཉམ་པའི་ཛཱ་ཏི་རྣམས་སོ། །།།ཁྱོད་ལ་འགྲོ་བ་ཀུན། མཛའ་བ་དང་བདག་བྱེད་པ་ནི་བག་དང་བདེ་སརྦ་གུ་རུ་བྲྀཏྟ་མཱ་དོ་ཚེཥཱ་གུ་རོ་རྨ་་ར་པ་ར་ས་དྲི་ཤཱ། །པཱུརྦ་གཱཔཱ་བྲྀཏྟཱའི་དང་པོ་ཐམས་ཅད་ལྕི། །ལྕི་གཙོའི་འོག་ཡང་ཕྱི་རྣམས་འདྲ། །སྔ་མར་ལྕི་འདི་པཱ་ལྟ་ར། །སྟ་རེ་ཏཱ་བ་དི་མཱ་སརྦ་ལ་གྷ་ཡཱ་བ་ཏྲ། །པེངྒ་ལ་པྲ་སྣ་རཿ། །གུ་རཱུ་པ་རི་ལ་དྷཱཔ་ཇི་སྲིད་ཐམས་ཅད་ཡང་དེ་སྲིད། །དམར་སེར་ཅན་གྱི་པཱ་སྣ་རའོ། །སྟེང་ལྕི་འོག་ཡང་པཱ་ལྟ་ར་ལ་ཕྲེང༌། །ལྟ་རེ་པཾཏྟི། །རྡི་གུ་ཎཱ་ཡ་ཏཱ་པྲ་ཏི་ཙྪནྡ། །གུ་རུ་ལ་གླུ་བྷིནྡྲ་གུ་ཎ་ག་ཎི། །རེ་ཀི་ཀ་ཉིས་འགྱུར་རིང་པོ་སྡེབ་སྦྱོར་སོ་སོ་སྟེ། །ལྕི་དང་ཡང་བའི་ཉིས་འགྱུར་ཚོགས་རྣམས་ཀྱིས། །རིམ་པ་བཞིན་བདྷ་ཏེ་ཀྲ་མ་ཤ། །ཨ་ཡ་མསྨྲ་དཱི་ཡ་པཱ་ལྟ་ར། །ཨེ་ཏཱ་བ་ཏི་དྷཾ་ཀི་མི་ཏི་ནྱ་སི་ཏ་དཱ་དུ་རེ་རེ་སྤེལ་བ་བྱ། །འདི་ནི་ཁོ་བོའི་པྲ་ཙཱ་རའོ། །འདི་འདྲ་འདི་ལྟ་ཞེས་བྱ་ལྕི། །དེ་བཀོད་འབྲས་ཕ་ལཱདྡལ་ཡེཏྲ། །བི་ཥ་མཾ་སེ་ཀཾ་ད་ལ་ཡངྒེ་ཨི་ཧ་ས་མེ་ལཕལ་ཏི་ནé། །བྲྀཏྟ་ལས་འདབ་མ་བྱ། །མི་མཉམ་གཅིག་བཅས་འདབ་མ་ལྕི། །།འདིར་མཉམ་ཡང་འབྲས་ནབྦའོ། །བྲྀཏྟཱ་འདི་མི་དཾ་ཀ་ཏི་ཐ་མི་ཧེ་ཏུ་པ་རི་ལི་ཁི་ཀཱ་དྱ་ནུ་ཀྲ་མ་དྲི་གུ་ཎཾ། ་། ཏ་ཏྲ་ཏུ་ནི་དུ་པ་ཡིན་ཞེས་འདིར། །སྟེང་དུ་གཅིག་སོགས་རིམ་པས་ཉིས་འགྱུར་བྲི། །དེར་ནི་ཡང་བའི་སྟེང་ལ་གྷ་རོ་པིཎེ་སི་ཀེ་ཏཱ་བ་ཏི་ཐ་མུདྡིཝཾ། །རཱུ་པཾ་ནྱ་སྤྱོཏྟི་ཥེར་ཀླུ་ར་སཾ་ཁྱིབསྡོམས་གཅིག་དང་བཅས། །འདི་ལྟ་ཉིད་ཡིན་ཞེས་པ་ཨུདྡིཊྛའོ། །གཟུགས་བཀོད་ཡི་གེའི་གྲངས་ཀྱིས་om།
