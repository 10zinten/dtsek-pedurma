བ་དནྟི། །ཏཙྪཱདྷ་པི་རཱཎྜི་ཤེ་ཥ་ཎཾ། །པྲ་ཙུ་པི་ཏ་ག་ཎ་ས་མ་མ་ས་དང་ཚོགས་དེ་དག། མཚན་པར་གྱུར་པ་དག་པ་སྣང་ལྡན་གྱི། །ཁྱད་པར་རབ་མཆོག་ལྡན་དུ་དེ་ནི་རཾ་ཀྲྀ་ཏ་ཡཉྩེཏྲ། །པྲ་ཙུ་པི་ཏ་པྲ་བྷེ་དཱ། བི་ཥ་མ་བྲྀཏྟཱ་ནི། །།།།ཀྱི། ཁ་ན་རྗོད་པར་བྱེད་པ་ཡིན། །དལ་འགྲོའི་བྱེ་བྲག་རྣམས་ཏེ། མི་མཉམ་པའི་བྲྀཏྟཱ་རྣམས་སོ། །::ཀུན་::ཀ་ཨུབྷཱ་བྲྀཏྟ་བི་ཀལྤ་ཨི་ཏི་ཨཱདྷཾ་ཛཱ་ཏ་ཡོནབྷི་དྷཱ་ཏ་བྷཱུ། །ན་བྱཉྫ་པ། བྲྀཏྟཱའི་རྣམ་པར་རྟོག་པ་བཤད་ཟིན་ནས། །འདི་ལས་ཕྱི་ནས་ཛཱ་ཏི་བརྗོད་པར་བྱ། །ཡང་བ་ཕྱི་མོ་ནེ་ན་མཱ་ཏཱ་ལ་གླུ་ནི་ཀ་བྷ་བ་ཏི་གུ་རུ་ཎཱ་དྭེ། །པཉྩ་བྷ་ཛ་སྣ་གུ་རུ་ཡུ་གཾ་ལ་གཅིག་སྟེ་ལྕི་བ་ནི། །ཕྱི་མོ་གཉིས་ཏེ་གསལ་བྱེད་དག་གིས་མིན། །བྷ་དང་ཛ་ས་ལྕི་ཟུང་ཡང་བ་བཞི་ཚོགས་གླུ་ཀ་ཙ་ཏུ་ཀཾག་ཎཱཉྩ་ཏུརྨ་ཏཱ། །ཏེ་ཙཱཁོ་པི་ཙ་ཏུརླཀ་ཨེ་ཀ་མཱ་ཀྲོ་དྲི་ལས། །ཕྱི་མོ་བཞི་པ་དེ་ལྔ་རྣམས་ནི་ཙ་ཞེས་བྱ། །ཡང་བ་བཞི་ནི་ཁ་ཡང་ཡིན། །ཕྱི་མོ་གཅིག་པ་ཀ་སྟེ་མཱ་ཀྲོ་ད། །བི་ཥ་མེ་ཥཱ་ཛ་ར་ཧི་ཏ་རྩེ་རྡ་ལེ་ད་ལེཉཊོ་ག་ཎྞཾ་བྷ་ཕྱི་གཉིས་ད། །གལ་ཏེ་འདབ་མ་འདབ་མར་ཚོགས་བརྒྱད་རྣམས། །མི་མཉམ་རྣམས་ལ་ཛ་སྤངས་འཕགས་བཏྟཱནཱ། ཞུ་། བབྷྱཱདྷེ་ཀཿཥཊྛོ་དོ་ཟླ་མ་ཨུ་བྷ་ཡ་ཏྲ་ཙཱཤེ་ཥཱ། །པུཊཱདྷེ་མར་གྱུར། །ཕྱི་མ་ཕྱེད་ལ་དྲུག་པ་ཀའོ་གཉིས་ཀ་ཡི། །བརྒྱད་པ་ད་སྟེ་ལྷག་མ་ཙ་རྣམས་སོ། །ཕྱེད་སྔར་དྲུག་ཥཊྛོ་ཛཁོ་བཱ་ཁེ་ཏཱ་རྱ་ལ་གླུ་ནི་བྷ་བ་ཏི་ཡ་ཏི། །ཥཊྛཁ་པ་རོཞནྟ་ཡ་ཏི་པ་ཛའམ་ཁའོ་ཁ་ལ་ནི། །དང་པོ་ལ་ནི་ཡང་བའི་ངལ་བསོར་འགྱུར། །ཁ་ཕ་རོལ་ལ་དྲུག་པའི་མཐར་ངལ་བསོ། །སྟུརྨོ་སྱ་ཐ་བྷ་བ་ཏི་ཙ་ར་མ་ད་ལེ། །ཙིནྡི་བྷི་ར་ཡུདྨ་པཱ་དོ་ཡ་དཱ་ས་མཱ་བྷོ་འདབ་མ་ཕྱི་མར་བཞི་པའང་དེ་ནས་འགྱུར། །ཙ་གསུམ་རྣམས་ཀྱི་ཟུང་མིན་རྐང་པ་དག། གང་ཚེ་ཏ་དཱ་བྷ་བེཏྲ་བྱ། །ཨེ་ཀོཔི་ནིཏྟི་ཏོ་ན་ཡ་དི་ཏི་ར་ཡུགླསྟ་་་་རྫོགས་པ་དེ་ཚེ་འཕྲོད་མར་འགྱུར། །གལ་ཏེ་ཟུང་མིན་གཅིག་ཀྱང་རྫོགས་མིན་པར། །དེ་རྣམས་ཀྱིས་དཱ་བི་པུ་ལཱ། །གུ་རུ་མ་དྷ་བརྟི་ནོ་ཙེདྡྷི་ཏཱི་ཡ་ཏུརྨོ་ག་ཎོ་ཛ་ཀཱ་རཱ་ཁྱོ། །ནི་དེ་ཚེ་ཡངས་མ་ཡིན། །གལ་ཏེ་གཉིས་པ་བཞི་པའི་ཚོགས། །ལྕི་གཉིས་དབུས་ཞུགས་ཛ་དག་བཤད། །།
