།ཉོ་pཏ་ཀ་སྐུ་བློ་ཎ་"བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་མི་ཉཙོ་ཎ་ཀ་ཊི་ལ་ཁ་ལོང༈སཾ་ཁཱ་ཡཱཔྲ་ཀཱ་རེ་དྷཱ། གྲངས་རྣམས་ཀྱི་རབ་ཏུ་བྱེད་པ་ལ་དྷཱའོ། །གཉིས་ཚན་ལྟ་བུ་གྲངས་ཀྱི་རབ་ཏུ་བྱེད་པའི་དོན་ལ་རྐྱེན་དྷཱ་འགྱུར་ཏེ། དྲི་དྷཱ༑ཡཉིས་ཚན། ་་:ཏྲི་དྷཱ?། གསུམ་ཚན། ཙ་ཏུནྟཱ། བཞི་ཚན། སོགས་སོ། །ཟ་ཁ་༈གུ་ཎོཉཎ་བཱ། དེ་ཡོན་ཏན་ལ་ཨཎ་ཡང་ངོ༌། །གཞན་ལ་སྦྱིན་པ་གཏོང་བ་སོགས་ཀྱི་དོན་དང་འབྲེལ་པའི་ཚན་པར་བཅད་པའི་དོན་དུ་བྱིན་པའི་རྐྱེན་དྷ་ཕ་རོལ་ལ་དབྱངས་དང་པོ་དེ་ཡོན་ཏན་དུ་འགྱུར་ཞིང་དབྱངས་དང་པོ་དེ་ཡོན་ཏན་བྱས་པ་ན་ངེས་པར་རྐྱེན་ཨརྱ་ཡང་སྟེར་དགོས་ཏེ། དྲི་ལ་དྷཱ། ཡོན་ཏན་བྱས། དྭེ་དྷཱ། །དེ་ལ་ཨཎ་སྦྱར་བས། །ཨི་ཨ་སོགས་སི་ཨམ། :：དྲི་དྷ། ཉིས་ཚན། དེ་བཞིན་དུ༑ཏྲེ་ྋདྷཾ། གསུམ་ཚན། ཙཱ་ཏུནྟཾ། བཞི་ཚན། སོགས་སོ། །ཚོད་::༈ཀྲྀ་ཡཱ་ཡཱ་ཨཱ་པྲྀཏྟོཀྲྀ་ཏུས། བྱ་བའི་ཆེད་དུ་ཀུན་ནས་འཇུག་པ་ལ་ཀྲྀ་་ཏུས་སོ། །ཟས་དུས་ཡང་ཡང་བྱེད་པ་ལྟ་བུ་ལན་མང་དུ་བསྐྱར་བའི་བྱ་བའི་ཆེད་དུ་ཀུན་ནས་འཇུག་པའི་དོན་ལ་རྐྱེན་:ཀྲྀ་ཏཱས་འགྱུར་ཏེ། པཉྩ་ཎིབཱ་རཾ་་་བྷུཀྟི༑ལན་ལྔ་ཟ། །ཞེས་པའི་དོན་ལ། པཉྩ་ཀྲྀཏྲཿལན་ལྔ་ཟ། དེ་བཞིན་དུ། ཥཊྚི་ཏཱལན་དྲུག་ཟ། སོགས་སོ། །དེར་བར། དང་་ཕྲུཔའིའདོད། ༈དྲི་ཏྲི་ཧཱུཾ་སུ་དཱི་ཏྲི་དག་ལ་སུའོ། །དྲི་དང་ཁྲི་དག་ལ་བྱ་བའི་ཆེད་་དུ་ཀུན་ནས་འཇུག་པའི་དོན་ལ་རྐྱེན་སུ་འགྱུར་ཏེ། ༼ཨུ་བརྗོད་དོན་སྲ་རྣམ་བཅད་བྱས་པས། དྭིཏྲིཞེས་གྲུབ་པ་ཨུཀྟདང་སྦྱར་བས། དྲི་རུཀྟལན་གཉིས་སྨྲ། ཏྲི་རུཀྟལན་གསུམ་སྨྲ། །གདོ། །༈བ་ཧཱ་དེཤས། བ་ཧུ་སོགས་ལ་ཤསྲའོ། །བ་ཧུའི་སྒྲ་ལས་བྱ་བའི་ཆེད་དུ་ཀུན་ནས་འཇུག་པ་དང་གཞན་ཡང་བསྐྱར་བ་དང་རྗེས་སུ་འཇུག་པའི་དོན་རྣམས་ལ་རྐྱེན་ཤསྲ་འགྱུར་ཏེ། བ་ཧུ་བཱ་རཱནྟཱི་ཏི། འདི་ལན་མང་བའམ་མང་བའི་མཚན་ཉིད་ཅན། _—།
