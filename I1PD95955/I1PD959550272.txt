།བསྒྲུབས་པས། ཨ་དྷུ་ཀླུཏྲ། དེ་བཞིན་དུ། ཨ་དྷུཀཱ་ཏཱ། ཨ་དྷུཀན་སོགས་སོ། །གཙཊོ༈ལི་ཧཱ་རོ་མྱང་བ་ལའང་སྔོན་མ་ལྟར་བསྒྲུབས་པས། ཨ་ལི་ཀཱཏྲ། ཨ་ལི་ཀཱ་ཏཱཾ། ཨ་ལིཀྐན་སོགས་སོ། །ན་ཚོད་དུ་། །མེད་དུ་་་་་་ཚོཏུ། ༈བིཤ་འཇུག་པ་ལ། ཨ་བིཀཏྲ། ཨ་བིཀྵ་ཏྭཾ། ཨ་བིཀྐན་སོགས་དང་། །༈ཀྲྀཥཱ་འགུགས་པའམ་གཤོར་གཏོད་པ་ལ། ཨ་ཀྲྀཀྵཏྲ། ་ཨ་ཀྲྀཀྵ་ཏྭཾ། ཨ་ཀྲྀཁྲུན་སོགས་སོ། ༈། ཨི་ཏི་སྭ་ར་སྭ་སྟི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་བྷཱུ་ཏ་པྲ་ཏྱཱ་ཡ་པྲ་ཀྲི་ཡ་ས་མཱབྷཱ༑ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་བྱུང་ཟིན་གྱི་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ། །།ཀ་ལ་་། ་་་་བ་ཁ་ཡི་དམ་དཔའ་་ཊ་་དུ་ང་། །ཏོ། ལ་ལར། ཀ་ར། ང་ལ་Pཚོ༈དེ་ནས་དངོས་པོ་དང་ལས་ཀྱི་དོན་སོགས་ཀྱི་རྣམ་དབྱེ་འཇུག་ཚུལ་་་རྣམས་རབ་ཏུ་བརྗོད་པར་བྱ་སྟེ། སྔར་བཤད་ཟིན་གྱི་དངོས་པོ་དང་ལས་ཀྱི་དོན་གྱི་འཇུག་ཚུལ་དང་ཆོ་ག་རྣམས་ནི་བྱིངས་ཐམས་ཅད་པ་རྣམས་ཀྱི་ཆོ་ག་ཡིན་ལ། འདིར་བྱིངས་ཕྱེད་པ་རྣམས་ཀྱི་དངོས་པོ་དང་ལས་ཀྱི་འཇུག་ཚུལ་ཏེ། དེ་ཡང་གོང་བྱིངས་ཐམས་ཅད་པའི་སྐབས་སུ་བཤད་པ་ལྟར་འདིར་ཡང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་་་འཇུག་པ་ལས། །ཉེར་བསྒྱུར་ཨ་ནུ་སྔོན་མའི་བྷུ་ཡོད་པ་་ལ། ལྐོག་གྱུར་ལིཊའི་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱར། གཉིས་ཀྱང་ངོས་གཉིས་བརྗོད། །ཎཔྲ་སོགས་ལ་བྷཱུ་ཨ་དག་གི་ཨཱ་འོས། ཨུ་ཨཱ། ཐུང་ངུའོས། དེ་ཐུང་ངུ་། ཛྷ་བ་སོགས་དང༌། བྷཱུ་ལ་བུ་ཀྲའོས། པུཀྲ་ཨཱ་ག་མ། ཁྲོ་སོགས་ལས་ཎཔྲ་སོགས་ལའོས། བ་ས་དང་པོ་ཅན་ལ་ཨིཊ་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ནུ་བ་བྷ་བེ། ཨ་ནུ་བ་བྷཱ་བཱ་ཏེ། །ཨ་ནུ་བ་བྷཱུ་བི་རི། ཨ་ནུ་བ་བྷུ་བི་ཥ། ཨ་ནུ་བ་བྷཱུ་བཱ་ཐེ། ཨ་ནུ་བ་བྷཱུ་བི་ལྷེ། ཨ་ནུ་བ་བྷུ་བེ། ཨ་ནུ་བ་བྷཱུ་བི་བ་ཧེ། །ཨ་ནུ་བ་བྷཱུ་བ་མ་ཧེ། ཉེ་བའི་ཚིག་དང་སྦྱར་It།
