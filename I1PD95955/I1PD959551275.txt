བསྟན་འགྱུར། །སྒྲ་མདོ། ༼སེ༽༈ཡོਭ་བ་ལ་སུན་པོ་དག་དང་གཙང་མ་ཉིད་དུ་བྱེད་ཅིང་། ང་བབ་ཏུ་གསལ་འོད་དུ་གྲུབ། ས་པ་དག་ནི་བསྡུར་འབྲས་རེའུ་མིག་རི་རྩེ་ལས་ཀྱང་དལ་གྱི་སྒྲ་མདོ། (་ཀྱི་དག། །།།།།།།།ਨੇ''ਨਵਨਲਸਲਰਨ'(ਛੋਨੂੰਸ਼ਰਨསྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། སྐྱོཅོ་ནེ། །སྒྲ་མདོ། །མངོན་བརྗོད། །མདོ། །རྒྱུ་ཡི་སྒྲ་མདོ། །་ར་ཉེ། དེང་ਰདཔེ་གཞི། སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། བན། ན?བ་། ན?ན་ན—ན། ཕྲེང་། །ཚེག་ཁྱིམ། །།།རྒྱ་གར་དུ་འགྲོ་བ་དྲག་པ་ཀཁོང་ཕྱི་ལུས་ཀྱི་ཕྱི་སྐྱོད། ཚེ་ལན་པ་སམ་རྗེ་བ་ག་བ་ག་བ་ག་བ་ག་བ་ཀྱི་ཟླ་བོ། །ངའི། །ཆང་ཁང་། དྲ་་ལ་ནང་ས་འ་ཚུ་པ་ཤོ་ཀ་བ་ཤོགཔ་བློ་ར། ཀན་པ་ཏ། ཆེ་སྐུ་ཚེ་བསྒྱུར་ཀྱིས་ང་།
