:ཀ་ཀ་བ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཚེཎས་བཀབྱོལ་ཎ་ཡོད། ཡ་གོ་མཱ་ཡུ་མྲྀ་ག་དྷཱཏྟ་ཀཱ། །ཐཾ་གཱ་ལ་བཉྩ་ཀ་ཀྲོལྦུ་ཕེ་རུ་ཕེ་ར་བ་ཛམྦུ་ཀ། །།།ཅེ་སྤྱང་རི་དྭགས་མ་རུངས་པ། །ཝ་དང་གཡོན་ཅན་ཁྲོ་བ་ཅན། །ངན་གཡོ་བ་དང་ཛམྦུ་ཀ། ：ག་སྐྱོ། །༈ཨོ་ཏུའི་ཌཱ་ལོ་མཱཛཱ་རོ་པྲྀ་ཥ་དཾ་ས་ཀ་ཨཱ་ཁུ་བྷུཀ། ཏྲ་ཏེ་ལོ་བཞེས་པར་བསྒྱུར་བའང་འདུག་ནི། ༈ཞུམ་བུ་བྱི་ལ་ཁྲུས་མཁན་དང༌། །བྱི་བ་གསོད་དང་བྱི་བའི་ཟས། །གསུམ་པོ་ད་བྱིད་གོ་དྷཱ་ར། །ཡོ་གོ་དྷེ་ར་གོ་དྷཱ་ར་གོ་དྷེ་ཡཱ་གོ་དྷི་ཀཱཏྨ་ཛེ། །ཤཱ་བིཏྟ་ས་མྱ་ཏམྦྷོ་སྣེ་ཤ་ལ་ལི་ས་ལ་ལཾ་ཤ་གོ་དྷེ་ཡ་ད་བདག་སྐྱེས་སོ། །གཟུགས་མོ་ཇུ་ཐུར་ཤ་ཁྱའོ། །དེ་ཡི་སྤུ་ནི་ཤ་ལ་ལི། །ས་ལ་ལོ་དང་ཤ་ལོ༑། བཱ་ཏ་པྲ་མཱིནཱ་ཏ་མྲྀ་གཀོ་ཀ་ཨཱི་ཧཱ་མྲ་གོ་ཧྲི་ཀི། །མྲྀ་གེ་ཀུ་རངྒ་བཱ་ཏ་ལོའོ། །བཱ་ཏ་པྲ་མཱི་ཞྭ་ཏའོ། །འཕར་བ་གནས་ཁྱི་རི་དྭགས་གསོད། །རི་དྭགས་ས་ལུས་རླུང་གི་སྲོག། ཡུ་ཧ་རི་ཎཱ་ཛི་ན་ཡོ་ན་ཡ། །ཨི་ཎེ་ཡ་མེ་ཙྪཱཉྩམྨཱ་ཏྲ་མེ་རྩེ་སྱེ་ཎ་མུ་བྷེ་ཏྲི་ཥཱ། །རི་དྭགས་པགས་པའི་སྐྱེ་གནས་རྣམས། །ཨི་ཎེ་ཡ་མེ་པགས་པ་སོགས། །འདི་དག་གཉིས་ཀ་གསུམ་ཡིན་ནོ། །ཀ་ད་ལཱི་ཀནྡ་ལི་ཙཱ་ནཉྩ་མཱུ་རུ་བྲི་ཡ་ཀཱ་བ་བི། །ས་མཱུ་རུཉྩེ་ཏི་ཧ་རི་ཎཱ་ཨ་མཱི་ཨ་ཛི་ཀ་ད་ལཱི་དང་ཀནྡ་ལཱི། །ཙཱི་ན་མཱུ་རུ་པྲ་ཡ་ཀཱ། །ས་མཱུ་རུ་དང་ཧ་རི་ཎཱ། །འདི་རྣམས་པགས་པའི་སྐྱེ་ན་ཡོ་ན་ཡ། །ཀྲྀ་ཤཱ་ར་རུ་རུ་ཧཱུཾ་ཀུ་རངྷ་སམྦ་ར་རོ་ཧི་ཥཱ། །གོ་ཀཥྛ་དྲྀ་ཥ་ཏི་གནས་སོ། །ཁྲི་སྙན་ས་ལེ་རུ་རུ་དང༌། །རུཾ་ཀུ་རརྐུ་སྡོམ་པ་སྐྱེས། །བ་ལང་རྣ་བ་སྲྀ་ཥཊ་དང༌། །ཨེ་ཎར་ཎཅུ་རོ་ཧི་ཏཱཉྩ་མ་རོ་མྲ་གཱ། །གནྡྷབྷས་ར་བྷོ་རཱ་མ༔སྲྀ་མ་རོ་ཅུ་དང་རོ་ཧི་ཏ། །གཡག་དང་རི་དྭགས་དྲི་ཟ་དང༌། །ཤ་ར་བྷ་དང་རཱ་མ་དང༌། །བ་མེན་རི་ག་བ་ཡཤ་ཤ། །ཨི་ཏུ་ད་ཡོ་ཚིགས་གཎཱ་ར་ག་བཱ་པ་ཤུ་ཛཱ་ཏ་ཡ། །ཨུན་བོང་སྲི་མ་ར། །འདི་སོགས་རི་དྭགས་དབང་པོ་སོགས། །གླང་སོགས་ཕྱུགས་ཀྱི་རིགས་རྣམས་སོ། །བྱི་བ་རུམྱུ་ཏི་ཀོ་བྱཱ་ཁུརྒ་རི་ཀཱ་བཱ་ལ་མཱུ་ཎི་ཀཱ། ་། ས་ར་ཊཀྲྀ་ཀ་ལཱ་སོཉཐ་མུ་ཀུན་པོ་འབིགས་བྱེད་དང་། །རི་བྱི་བྱི་བ་གཞོན་ནུ་དང༌། །ཚངས་པ་ཀྲྀ་ཀ་ལཱ་སའོ། །དེ་ནས་ས་ལཱི་གྲོ་ཧ་གོ་དྷི་ཀཱ། །།ལཱུ་ཏཱ་སི་ཏཾ་ཏུ་བཱ་ཡོཎྞ་ནཱ་བྷ་མརྐ་ཊ་ཀཱས་མཱ། །རྩངས་ཕྲན་ཁྱིམ་གྱི་རྩངས། །སྡོམ་ནི་མོ་དང་ཐགས་མཁན་ནོ། །ཨུཎ་ནཱ་བྷ་སྤྲེའུ་བལ་གྱི་ལྟེ་བ་མཚུངས། །།
