སྐྱ་བ་ལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽ཏིཀ་འཚོ་ཏ་ཞིགཔྲསྟ་ར། །སྦད་བྱེད་དོ། །གྲ་བ་ཞེས་པ་གོར་མ་སྟེ། །སྔར་རི་བོ་ལ་ཡང་བཤད་པའོ༑། ཨུ་པ་ལ། ཙོག་པོའི་རྡོ་བ་སྟེ། ཨུ་པ་ཉེ་བ་དང་། ལ་འཛིན་པར་བྱེད་པའོ། །ཨཧྥ་ན། ཇུ་པོའོ། །ཤི་ལ་རྟུན་བུའི་རྡོ་བའོ། །དྲི་ཤད:་རྡོའོ། ཊདེ་ནས་རྐང་པ་གཅིག་གིས་རྭའི་མིང་སྟེ། ཀུ་ཊ：བརྩེགས་པའོ། །ཤི་ཁ་ར་ཞེས་པ་གཙུག་སྐྱེས་སམ། །རྭའི་རྩེ་མོའོ། །ཤི་ཁ་གཙུག་དང༌། ཌ་རུ་ཧ་སྐྱེ་བ་སྟེ། ཤི་ཁཱི་རོ་ཧ་ཏི་ཞེས་པས་རུ་ཧལ་ཌ་རྐྱེན་དུ་བྱས་པའོ། །ཐིངྒ། རྭ་རྩེའོ། །།ཀན་དེ་ནས་རྐང་པ་གཅིག་གིས་རི་གཟར་གྱི་མིང་སྟེ། །པྲ་པཱཏ། རབ་ལྟུང་སྟེ། གཡང་སའོ། །།ཏ་ཊ་རི་འགྲམ་སྟེ། རི་གཟར་གྱི་འགྲམ་མམ་ངོས་སོ། །ཤྲཱི་གུ་རི་གཟར་ཏེ། །བྷསྔ་ལ་ཀུ་རྐྱེན་དང་། །ས་དབྱི་བ་དང་། ཡན་ཨ་ཀི་རུ་་བསྒྱུར་བའོ། །།དེ་ནས་ཀཌ་ཀ། རི་ཞོལ་དང་། །ནི་ཏམྦུ་རི་ཤམ་མོ། །།ནི་ཀུཉྫ་ནགས་དང་། ཀུཉྫ་ནགས་ཚལ་དང་། ཁྱི་ལ་ཏཱ་པི་ཧི་ཏཱ། འཁྲི་ཤིང་། ཁེབས་ཞེས་བྱ་སྟེ། ནགས་ཚལ་ལོ། །རིའི་སྡེའོ། །གཞི་ཁགབཟླ་བའི་ਨੇਵਟਸਵਾਆਲਕੇਉਸ'ਚਸਰਕਲਕਨਵੇਸਣདེ་ནས་ནགས་ཀྱི་སྡེ་ལ། ཨཱ་ཊ་བི། ཨ་རི་ཧཱུྃ་དགོན་པའོ། །བཱི་པི་ནཾ། དགོན་པ་སྟོང་པའོ། །ག་ཧ་ནཾ། །ནགས་འཐིབས་པོའོ། །ཀཱ་ན་ནཾ་:ནགས་་་ཚལ་ལོ། །བ་ནཾ:ནགས་སམ་ཚལ་ལོ། །གྲི་ཧཱ་རཱ་མ། ཁྱིམ་གྱི་ཀུན་དགའ་ར་བ་སྟེ། ཁྱིམ་དང་ཉེ་བའི་ཀུན་དགའ་ར་བའོ། །ནིཥྛ་ཊ་ཋངེས་འགྲོའོ། །ཨུ་པ་བ་ནཾ། ཉེ་བའི་ཚལ་ལོ། །ཀྲྀ་ཊཾ་མཾ་བ་ནཾ། །་བཅོས་མའི་ཚལ་ལོ། །ཀྲྀ་ཌ:རྩེད་མོའི་སའོ། །ཨུཏྤ་ན། སྐྱེད་ཚལ་ལོ། །མ་ཁ་ཡ་གེ་ཁྲོ་ལང་ཀ་ཀ་ལ་དེ་ནས་ཕྲེད་དུ་བྱས་པའི་མིང་ལ། བི་བྱ་ཕྲེང་བ་དང༌། ཨཱ་ལི་དང་། ཨ་བ་ལི་ཡང་ཕྲེང་བ་དང་གྲལ་དང་། པཾ་སྟི། ཚར་དུ་དངར་བའི་ཕྲེང་བ་དང་། —།
