སྡེབ་སྦྱོར་རིན་ཆེན་འབྱུང་གནས་ཞེས་བྱ་བ། །ཏ་དུ་ནྱ་ཏེ་ལ་གླུ་ཥཀྵ་ན་ནི་རནྟ་རཾ་ཡུ་ཛེ། ཡེ་། བེ་ཏཱ་ལི་ཡོ་པ་མཾ་མུ་ཁེ་ཙེ་དོ་པ་མ་ཆད་དུ་མིན། །དེ་ནི་རོ་ལངས་ཅན་དུ་བརྗོད་པར་བྱ། །གལ་ཏེ་དང་པོ་རོ་ལངས་ཅན་ལྟ་བུ། །ཕྱི་མར་ཙྪནྡ་སི་ཀཾ་ར་ཡོ་ཏུ་བྲྀཏྟེ། །ཨཱ་པཱ་ཏ་ལི་ཀཱ་བྷ་ག་གནྟཱ་མུ་ཁ་མི་ཧ་བེ་ཏཱ་ལཱི་ཡ་ས་ཡ་དག་ནི་ཉེ་སྡེབ་སྦྱོར། །འདིར་ནི་དང་པོ་རོ་ལངས་ཅན་དང་མཚུངས། །།བྷ་ག་ག་མཐའ་ཀུན་ནས་སྲབ་མཱ་ནཾ། །ཡུགླ་མུ་ཁེ་དཱརྡ་རོ་ཛ་དོ་པྲ་ཙྩ་བྲྀཏྟི་ར་བ་ཤིཋ་མཱ་རྱ་བཏྲ། །ཨུ་དཱི་ཙྪ་བྲྀཏྟི། པ་ཡིན། །ད་ལས་ཕ་རོལ་ཛ་ད་ཟུང་དང་པོ། །ལྷག་མ་དང་པོ་ལྟ་བུ་རབ་འཕེལ་ཡིན། །གང་ཚེ་ཟུང་རྫ་དོ་ཡ་དཱ་མུ་ཁ་མ་ཡུ་ཛོཔ་རི་ཤིཥཱ་མཱ་རྱ་བཏྲ། །ཨུ་དཱི་ཙ་བྲྀཏྟེ་རཱ་ཡུརྨ་བྷ་ཀྵ་མིན་དང་པོ་ཛ་ད་དག་། ལྷག་མ་དང་པོ་ལྟ་བུ་ཕྱི་འཕེལ་ཡིན། ཟུང་མིན་ཆ་ནི་ཕྱི་འཕེལ་ལས། །ཟུང་ཙྩ་བྲྀཏྟི་ཡུག་བྷ་ཀྵ་མྲྀཏྟ་ཀཾ། །སདྨ་བྲྀཏྟ་ཀ་ཡུ་ཛཱཉབ་རཱནྟི་ཧཱུྃ། །ཨ་ཡུཀྲ་ཆ་རབ་འཕེལ་ཀུན་འཕེལ་ལོ། །ཀུན་འཕེལ་ཟུང་ནི་གཞན་གྱི་མཐའ་ཅན་འགྱུར། །ཟུང་མིན་རྐང་དི་རུ་ཧཱ་སི་ནཱི། །བེ་ཏཱ་ལཱི་ཡ་ཛཱ་ཏ་ཡ། །།།པས་མཛེས་རྒོད་མའོ། །རོ་ལངས་ཅན་གྱི་ཛཱ་ཏི་རྣམས་སོ། །།།ཨ་གུ་རུ་ར་ཙ་ལ་དྷཱི་ཏི་རུ་ད་དྷི་ཙི་པ་དི་པ་དི། །གནྟེ་མཱ་ཏཱས་མ་ཀཾ་ལི་ད་ལཱཏྲ། །རྐང་པ་རྐང་པ་ཆུ་གཏེར་ཙ་སྟེ་ལྕི་མེད་མི་གཡོ་བརྟན། །མཐར་ལྕི་འདབ་ལས་ཡང་བ་ཕྱི་མོ་མཉམ། །བྷོ་ཡ་དི་ཙ་ཏྲི་ཏ་ཡཱ་དུ་བ་ཙི་ཏཱ། །ཙཱཏྠ་ཛ་ཨེ་བ་བཱ་བི་བློ་ཀཿ། །ཁོ་ཛོ་གལ་ཏེ་ཙ་གཉིས་ལས་བྷ་ཉེར་བཀྲ་ཡིན། །ཙ་ལས་ཁའམ་ཛ་ཉིད་ཁྱད་པར་ཚིགས་བཅད། །ཕྱེད་ལས་བཱ་བཱ་ན་བཱ་སི་ཀཱདྷཱཏྲ། །ཙཱ་ཥྛདྨ་དི་ས་དྲི་ཏ་ཡཾ་ཙི་ཏཱ། །ས་མ་ནན་ར་ནི་བ་དི་ཏ་ཁའམ་ཛ་ནི་ནགས་ཁྲོད་གནས། །ཙ་ཞེས་བྱ་ལས་གལ་ཏེ་ས་གཉིས་བཀྲའོ། །བར་མ་ཆད་དུ་བཤད་པའི་ལཀྵ་ཎ་བྷཱཏ། །མཱ་ཏཱ་ས་མཱ་དི་པཱ་དཱ་ཀུ་ལ་ཀཾ། ཨ་བི་དི་ཏ་བྲྀཏྟ་པ་རི་མཱ་པ་རི་མཚན་ཉིད་བསྟན། །ཕྱི་མོ་མཉམ་སོགས་རྐང་པ་དཀྲུགས་པ་ལ། །བྲྀཏྟའི་ཚད་རྣམས་རྟོགས་པ་མེད་གྱུར་ག་ཏཾ། །མུ་ནི་ནཱཉབྱཱ་དྷཱ་པི་བཱ་དཱ་ཀུ་ལ་ཀཾ། །ཨཊྚ་བཏྠོ་ཙཱཌྷ་ཡ་སྭཱ་གཱ། །པ། །ཐུབ་པས་རྐང་འཁྲུགས་སུ་ནི་ངེས་པར་བརྗོད། །གང་གི་ཕྱེད་ལ་བརྒྱད་བརྒྱད་དག་ནི་ག། །
