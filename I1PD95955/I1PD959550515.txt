ཀ་ཀ་ཀ་ཎ་ཡོནབི་ཀྲྀ་ཏོ་ཙ་བི་པྲ་ལབྷཱསྐུ་བཉྩི་ཏ། །མ་ན་ཧ་ཏཔྲ་ཏི་ཧ་ཏཔྲ་ཏི་བདྷོ་ཧ་ཏཉྩ་ས། །ཨ་དྷི་ཀྱི་ལྟཔྲ་ཏི་སྟོ་བདྷེ་ཀཱི་ལི་ཏ་སཾ་ཡ་ཏོ། །ཨ་པཱསྣ་ཨཱ་བཏྰYཚིག་རྡེག་དང་ནི་སླར་སྨྲ་བ། །འཆིང་བྱེད་ཕུར་བུ་འགོག་བྱེད་དོ། །དོན་མེད་པ་དང་སྟོང་པ་ཐོབ། །སྙཥནི་ཤི་ཀོ་བྷ་ཡ་དྲུ་ཏ། །ཨཱ་ཀཱ་རི་ཏཀཱ་རི་ཏོཞབྷི་ཤསྟེ་སཾ་ཀ་སུ་ཀོཉཧྥི་རེ། །ཉམ་ང་བ་དང་བག་ཚ་བ། །བྱི་བཤོར་སུན་འབྱིན་ཀཱ་རི་ཏ། །གཡོ་བ་དང་ནི་སེམས་མི་གནས། །བྱ་ས་ནཱཉྙོ་པཀྟ་རྡོ་བི་ཧསྟ་བྱུ་ཀུ་ལ་ལོ་ས་མོ། །བི་ཀླ་བོ་བི་ཧཱ་ལམནར་དང་གཞན་གྱིས་འཚིར་བ་གཉིས། །ལག་སྟོང་དང་ནི་གཡོ་བ་འདྲ། །བག་ཚ་བ་དང་འཁྱོར་བའོ། །སཱུཏྟ་བི་བ་ཤོཉརིཊྚ་དུཥཱ་དྷི། །ཀ་བྱཱཀ་ཤཱནྟ་སནྣདྷེ་ཏཱ་ཏ་ཏཱ་པཱི་བ་བྷོ་བྱ་ཏེ། །འཆི་ལྟས་སུན་འབྱིན་བློ་གྲོས་ཅན། །ལྕག་གིས་གཞུས་དང་འབྲབ་པར་བྱེད། །ལྕག་གཞུས་རྟ་ཡི་གསང་བྱ་དང་། །དྭེ་བྱེ་ཏུཀྱི་ག་ཏོ་བ་དྷུབྷིནྡ་ཙྪེ་དྱ་ཨི་མོ་ས་མོ། །བི་བྷྱོ་བི་ཥ་ཎ་ཡོ་བ་དེ་མུ་ཤ་ཞེ་སྡང་དང་ནི་འཁྲུག་པའོ། །གསད་བྱ་མགོ་གཅོད་འདི་དག་མཚུངས། །དུག་སྟེར་དུག་གིས་གསད་བྱ་བ། །གཏུན་ཤིང་གཏུན་ལོ་མུ་ས་ལེ་ན་ཡ། །ཤི་ཤྲཱི་དཱ་ནཀྲྀཥྛ་ཀ~་པ་ལཉྩི་ཀུ་ར་ས་མོ། །གྱིས་བརྡུང་བ་དང་། །སྡིག་སྤྱོད་དང་ནི་ནག་པོའི་ལས། །སོ་མི་བསོད་དང་རྣམ་གཡེངས་འདྲ། །དོ་ཥ་ཀ་དྲི་གླུ་རོ་བྷཱ་གཱི་ནི་ཀྲྀ་ཏསྣ་ཚྭ་ཛུཤ་ཋ། །ཀནྡྷེ་ཛ་པསཱུ་ཙ་ཀསཏྲི་སྐྱོན་འཛིན་ཕ་རོལ་ཉམས་པ་རྟོག། སྒྱུ་ཅན་དང་ནི་གཡོ་ཅན་ཀྱོག། རྣ་བར་ཟློས་དང་ཕྲ་མ་དང༌། །ཕྲ་ཤུ་ནོ་དུརྫ་ནཁ་ལ། །རྫོགས་སཾ་སོ་གྷཱཏྟ་ཀཀཱུ་ར་པཱ་པོ་དྷཱཎྜསྐུ་བཉྩ་མ་དབྱེས་དང་གཅུགས་མཁན་ཞེས། །རྣམ་འཚེ་འཇོམས་བྱེད་ཁྲོ་བ་སྡིག། མ་རུངས་པ་དང་མ་སྒྲུངས་ཀ༑། ཨཛཱ་མུ་ཊ་ཡ་ཐཱ་ཛཱ་ཏ་མཱུཎ་བི་དྷེ་ཡ་བཱ་ལི་བྷུ། །ཀ་དརྗེ་ཀྲྀ་པ་ན་ཀླུ་དྲ་ཀིམ་པ། །མི་ཤེས་རྨོངས་པ་གླེན་པ་དང་། །འཐོམ་པ་མི་རིག་མི་མཚོན་པ། །འཆུམས་པ་ཆ་ཕྲ་སེར་སྣ་ཅན། །ཙ་ན་མི་ཏམྦ་ཙཱ། །ནི@སཱསྐུ་དུབི་དྷོ་དཱི་ནོ་ད་རི་དྲོ་དུརྒ་ཏོ་པ་ཞིབ་མོ་མཛངས་པ་སྟེན་ཆོས་པ། །ནོར་མེད་སྒྲུབ་ངན་བཀྲེན་པོ་དང༌། །།དབུལ་པོ་ངན་འགྲོ་ཕོངས་ས༑། བ་ནཱི་ཡ་ཀོ་ཡ་ཙ་ན་ཀོ་མཱངྒཱ་ཎོ་མཱ་ཙ་ཀཱཎྜི་ནོ། །ཨ་ཧཾ་ཀཱ་ར་བཱ་ན་ཧཾ་ཡུཔའོ། །སློང་མོ་སྤྲང་པོ་ལམ་པ་དང༌། །སློང་མོ་པ་དང་དོན་འདོད་པ། །ང་རྒྱལ་ཅན་དང་ཧཾ་ཡུའོ། །།
