དpgཀ་ཁ་སུ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་འཁ་ཎ་ཊ་སི་ཀུ་སུ་མེ་པི་ཙ། །བིཥཱ་བ་པྱ་ཛི་ཏཱ་ཏྱལྟོསཱུ་ཏསྱཱཋ་རི་སཱ་ར་ཐོ། །བྱཀྟཔཱ་རཾ་ཏུ་བུད་མེད་མེ་ཏོག་དུས། །ཁྱབ་འཇུག་མ་ཕམ་ཏུཏྟོའོ། །སཱུ་ཏི་བཟོ་བོའི་ཁ་ལོ་སྒྱུར། །བྱཀྟ་མཁསདྷེ་པི་དྲྀཥྛནྟཱ་བུ་བྷེ་ཤཱསྔ་ནི་དན་ནེ། །།གཏྟཱ་སཱུཏྟཱ་ར་ཐོ་རཿསྦེ་གྷོ་ཌི་ཡཱ་དང་གསལ་བའོ། །དཔེ་དང་བསྟན་བཅོས་དྲྀཥྛནྟ།  ོ། ཀཱཏྟཱ་ཁ་ལོ་སྒྱུར་བ་དང་། །སྒོ་སྐྱོང་རྒྱལ་ཡཱཉྩ་ཤུ་དྲ་ཛེ། །བྲྀཏྟཱནྟི་སཱུཏྲ་ཀ་ར་ཎེ་པྲ་ཀཱ་རེ་ཀནྡྷཱ་བཱནྟ་ཡོ། །ཨཱ་རིགས་དམངས་འདྲེས་བུ། །བྲྀཏྟཱནྟ་ནི་རབ་བྱེད་དང༌། །བྱེ་བྲག་ཚོགས་པ་གཏམ་རྣམས་ལའོ། ་། ཨཱ་ནནྟ་ས་མ་རེ་ཀྲྀ་ཏྱ་སྠཱ་ན་ནཱི་ཤྲདྲི་ཤེ་ཥ་ཡོ། །ཀྲྀ་ཏཱནྟོ་ཡ་མ་སིདྡྷན་ནནྟ་ནི་གཡུལ་དང་ནི། །གར་གནས་སྐྱེས་མཐའི་བྱེ་བྲག་ལ། །ཀྲྀ་ཏཱནྟ་ནི་གཤིན་རྗེ་དང༌། །དེ་བཱ་ཀུ་ཤ་ལ་ཀརྨ་སུ། །ཐེགྷཱ་དི་ར་ས་རཀྵདི་མ་ཧཱ་བྷཱུ་ཏཱ་ནི་ཏཾ་གུ་གྲུབ་མཐའ་དབང་ཐང་མི་དགེའི་ལས། །བད་ཀན་ལ་སོགས་རོ་དང་ནི། །ཁྲག་སོགས་འབྱུང་བ་ཆེན་པོ་ཎ༑། ཨིནྡྲི་ཡཱ་ཅུསྨྲ་བི་ཀྲྀ་ཏི་ཤཔ་ཡོ་སྦི་དྷཱ་ཏ་བ། །ཀ་ཤཱནྟ་དང༌། །དེ་ཡི་ཡོན་ཏན་དབང་པོ་རྣམས། །རྡོ་ཡི་རྣམ་འགྱུར་སྒྲའི་སྐྱེ་གནས། །དྷཱ་ཏུ་ཡིན་ཏེ་ཁམས་རེ་པི་སུདྡྷན་ཚྭ་པ་སཱུ་སརྔ་གོ་ཙ་རེ། ་ཀཱ་སུ་པཱ་མརྒྱུ་ཡོཞེས་སོ། །ཀ་གནྟ་རེ་བཙུན་མོའི་འཁོར། །མི་དབང་ཀུན་གྱི་སྤྱོད་ཡུལ་མིན། །མཚོན་དང་ནུས་པ་ཤཏྟིམཱུཏྟིཀཱ་ཟི་རྩ་ཀཱ་ཡ་ཡོ། །བིསྟཱ་ར་བལྕོཔྲ་ཏ་ཏིརྦ་ས་ཏི་རཱ་ཏྲི་བེཧྥ་ནོ། །ཤཏྟིའོ། །མཱུརྟི་སྲ་དང་ལུས་ལའོ། །ཡངས་པ་འཁྲི་ཤིང་པྲ་ཏ་ཏི། །།བ་ས་ཏི་ནི་མཚན་མོ་ཁྱིམ། །ཀྵ་ཡཱརྩ་ཡོ་ར་པ་ཙི་ཏིསཱ་ཏིཌྷ་ནཱ་བ་སཱ་ན་ཡོ། །ཨནྟི་པཱི་ཌཱ་དྷ་ནུཀོ་ཁྱོཛཱ་ཏིཟད་མཆོད་ཨ་པ་ཙི་ཏིའོ། །སཱ་ཏི་སྦྱིན་དང་གནས་དག་ལོ། །ཨརྟི་ཟུག་རྔུ་གཞུ་ཡི་མཆོག། ཛཱ་ཏི་སྤྱི་སཱ་མཱ་རྱ་ཛརྨ་ནོ། །པྲ་ཙཱ་ར་སྱནྡ་ཡོ་རཱི་ཏི་རཱི་ཏི་ཌི་སྨྲ་པྲ་བཱ་ས་ཡོ། །ཨུ་ད་དང་སྐྱེ་བ་དག། རབ་སྐྱོད་བསྐྱོད་པ་ཨཱི་ཏིའོ། །རཱི་ཏི་མཆོད་དང་གནས་དག་གོ། །ཤར་དང་ཐོབ་པ་ཡོཤདྷི་ག་མེ་པཱསྟི་སྦེ་ཏཱ་ཏཱཀྐི་ཏྲ་ཡེ་ཡུ་གེ། །པི་ནཱ་བྷེ་དེ་པི་མ་ཧ་ཏཱི་བྷཱུ་ཏབྷསྨ་ནི་སམྤ་པཱསྟིའོ། །སྦེ་ཏཱ་མེ་གསུམ་གསུམ་ལྡན་དུས། །པི་ཝང་དབྱེ་བ་མ་ཧ་ཏི། །བྷཱུ་ཏི་ཐལ་བ་ཕུན་སུམ་།
