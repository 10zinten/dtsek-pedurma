བསྟན་འགྱུར། །སྒྲ་མདོ། །སེ?པའི་ནང་་་ཀ་ཁ་ཙེ༑ཡངེ༌། གྲྭ་མ་ཎི་ནེ། ངེ༌། གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་ཎི་?ལྔ་པ་གསུམ་ལ། གྲྭ་མ་ཎྱཡང་སི། གྲྭ་མ་ཎི་ནང་སི། གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་ཎི་བྷུདྲུག་པ་གསུམ་ལ། གྲྭ་མ་ཎྱངསྲ༅། །གྲྭ་མ་:ཎི་ནངས། གྲྭ་མ་ཙྩོཨོས། གྲྭ་མ་ཎི་ནོཨོས། གྲྭ་མ་ཧཱུཾ། ཨཱ་ན། གྲྭ་མ་ཎཱི་ཧཱུྃ། ཨཱམ། །བདུན་པ་གསུམ་ལ། གྲྭ་མ་ཧཱུཾ། ང་། གྲྭ་མ་ཎི་ནི། ང་། »གྲྭ་མ་ཙྩོཨོསྲ། གྲྭ་མ་ཎི་ནོཨསྲ། གྲྭ་མ་ཎི་ཥཱ༑ཡསུཔ། བོད་པ་གསུམ་ལ། །ཧེ་གྲྭ་མ་ཎི། སི། ཧེ་གྲྭ་མ་ཎི་ནཱི། ཨོ་་ཧེ་གྲྭ་མ་ཎི་ནི། ཛསྲ༄༅། དེ་བཞིན་དུ་མ་ནིང་གི་རྟགས་ཨཱི་རིང༌། མཐའ་ཅན་ཐམས་ཅད་འདི་དང་འདྲའོ། །ན་འ། །མ་ནིང་གི་རྟགས་ཨཱ་ཡིག་མཐའ་ཅན་ལ་སོ་མ་པཱ་ཞེས་པ་ལ་དཔེར་བརྗོད་ན༑དེ་ཡང་མ་ནིང་གི་ཐུང་ངུ་འོས་ཐུང་ངུར་བསྒྱུར་བས་ཆོ་ག་ཐམས་ཅད་ཨ་ཐུང་མཐའ་ཅན་ཀུ་ལའི་སྐབས་དང་འདྲ་སྟེ། ༈སི། སོ་མ་པཾ། ཞི་བ་འཐུང༌། སོ་མ་པེ༑ཡཨོསོ་མ་པཱ་ནི། ཡཛསྲ། སླར་ཡང་སོ་མ་པོ། ཨམྲ། སོ་མ་པེ། ཨོ། སོ་མ་པཱ་ནི༑ཤསྲ༄༅། སོ་མ་པེ་ན། ཊཱ། སོ་མ་པཱ་ཤྩ། སོ་མ་པིཉིས། སོ་མ་པཱ་ཡ༑ཡངེ་། སོ་མ་པཱ་ཤྩ། སོ་མ་པེ་སོགས་སོ། །།།མི་ཁ་། །མ་ནིང་གི་རྟགས་ལ་ེཨུ་ཡིག་མཐའ་ཅན་ལ་མ་དྷུ་ཞེས་པའི་སྒྲ་ལ་མཚོན་ན། དེ་ཡང་ནཱ་མའི་དབྱངས་ལ་ནུམ་སྟེར་བ་ཁོ་ན་སྟེ། སྦྲང་རྩི། མ་དྷུ། སི། མ་དྷུ་ནཱི། ཨོ། མ་དྷཱ་ནི། ཛསྲ༄༅། །སླར་ཡང་། མ་དྷུ། ཨམྲ། མདྷུ་ནཱི། ཨོ། མ་དྷཱ་ནི། ཤསྲི། མ་དྷུ་ནཱ། ཊ། ༼མ་བྷུ་བྷཾ། མ་དྷུ་བྷིམ་དྷུ་ནེ། ངེ་། མ་དྷུ་བྷཾ། ་མ་དྷུ་་སོགས་སོ། །།།ནད་མ་་ས་་m/ཆཎྜི་ཏ་ཚོ་ཨུལ། འཚོམ་ནིང་གི་རྟགས་ཨེ་ཡིག་མཐའ་ཅན་ལ་ཨ་ཏི་རི་ཞེས་པའི་རྟགས་ལ། མ་ནིང་ལ་ཐུང་ངུ་འོས་ཐུང་ངུར་བསྒྱུར་ལ་དེ་ཡང་ཐུང་ངུར་བསྒྱུར་བ་ལ་མཚམས་།
