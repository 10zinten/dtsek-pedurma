བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽། འབྲས་ནཉྩའོ། །གང་གི་ཚེ་མི་:མཉམ་པའི་བྲྀཏྟ་གཅིག་པུ་དེ་དྲི་བར་འདོད་པ་པ་དེའི་ཚེ་ཆོ་ག་འདིའོ། །འདི་འདྲ་བའི་རྫོགས་པ་འདི་ལྟ་ཡིན་པ་ནི་དཔེར་ན། ཡི་གེ་བཞི་པའི་སྡེབ་སྦྱོར་ལ་བྲྀཏྟ་བཅུ་པ་ཇི་ལྟ་བུ་ཞེས་འདྲི་ན། །རྒྱུ་དེ་ཉིད་དུ་བཅུ་་བྲིས་ནས་ཇི་སྲིད་འབྲས་བུ་ནཋར་གྱུར་པ་དེ་སྲིད་དུ་འདབ་མ་བྱའོ། །འདབ་མ་་ནི་ཕྱེད་དུ་བྱེད་པའོ། །།དེར་གང་མི་མཉམ་པ་དེ་ལ་གཅིག་བསྣན་པ་ལས་མཉམ་པར་བྱས་ནས་འདབ་མ་བྱའོ། །འདིར་ལྕི་བ་ཞེས་:བྱ་བ་་ནི་མི་མཉམ་པའི་འདབ་མ་ལ་ལྕི་བ་ཐོབ་པོ་་ཞེས་པའི་དོན་ཏོ། །མཉམ་ཡང་ཞེས་པ་ནི་མཉམ་པའི་་་འདབ་མ་ལ་ཡང་བ་ཐོབ་:བོ་ཞེས་་པའི་དོན་ཏོ། །གང་དང་གང་ཐོབ་པ་དེ་དང་དེ་ཉིད་སོ་སོར་འཕྲེད་ལ་ཕྲེང་དུ་བྲི་བར་བྱའོ། ༈། དེ་ལྟ་བུར་གྱུར་པ་ནི་འབྲས་བུ་ནཋའོ༑། །ན༔་ནི་འདྲེན་བྱེད་དོ། །བྲྀཏྟ་འདི་དུ་པ་ཡིན་ཞེས་པ་འདིར། །སྟེང་དུ་གཅིག་སོགས་རིམ་པས་ཉིས་འགྱུར་བྲི། །དེར་ནི་ཡང་བའི་སྟེང་བསྡོམས་གཅིག་དང་བཅས། །འདི་ལྟ་ཉིད་ཡིན་ཞེས་པ་ཨུདྡིཊྚ། །གང་གི་ཚེ་བྲྀཏྟ་་་རྣམས་་རང་གི་ངོ་བོས་ནི་དུ་པ་ཡིན་ཞེས་ཤེས་པར་བྱ་བ་ལ་འདི་ལྟ་བུ་མ་ཤེས་པ་དེ་ཤེས་པའི་དོན་དུ་ནི་ཨུངྒིཊྚའི་ཆོ་གའོ། །དཔེར་ན་ཡི་གེ་བཞི་པའི་སྡེབ་སྦྱོར་ལ་ལྕི་བ་གཉིས་དབུས་ན་ཡོད་པའི་བྲྀཏྟ་དུ་པ་ཡིན་ཞེས་འདྲི་ན། ཚོ་བྲྀཏྟ་དེ་ཉིད་ཀྱི་རི་མོ་ཐོར་ལོང་དུ་བྲིས་ནས། རི་མོ་རྣམས་ཀྱི་སྟེང་དུ་གཅིག་ལ་སོགས་པ་རིམ་གྱིས་་ཉིས་འགྱུར་དུ་སོང་བ་རྣམས་རིམ་པས་:བྲི་སྟེ། དེ་ལྟ་བུས་དེ་གསལ་བར་བརྗོད་པར་བྱའོ། །དེ་ལ་ཡང་བ་རྣམས་ཀྱི་སྟེང་གི་ཨངྐ་གང་ཡིན་པ་དེ་རྣམས་ནི་ཡང་་་བའི་སྟེང་སྟེ། དེ་རྣམས་བསྡོམས་ལ་གཅིག་དང་བཅས་པ་བྱའོ། །དེ་ལྟར་བྱས་པས་གྲངས་གང་རྙེད་པ་དེ་འདི་ལྟ་བུ་ཡིན་ནོ་ཞེས་པ་ནི་ཨུངྒིཊྚར་རིག་པར་བྱའོ། །འདིར་ཡང་བའི་སྟེང་དུ་གཅིག་དང་བརྒྱད་དོ། །དེ་དག་བསྡོམས་པས་་་།
