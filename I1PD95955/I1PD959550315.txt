ོཎཚོ་ཁ་ལ་ཀ་ཀ་ལྟོ་ཆ་ཁ་ཨཔྲ་འཇིགཨེ་དྷི། ལྷག་མ་རྣམས་ལ། ཧཱུཾ། སྟ། ཨ་སཱ་ནི། ཨ་སཱ་བ། ་ཨ་སཱ་མ། དེ་རིང་མ་ཡིན་པ་ལང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ༈ཨ་སྟེ་རིཊ། ཨསྟི་ལས་ཨཱིཊའོ། །བྱིངས་ཨས་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་དིཔྲ་དང་སིཔྲ་ལ་ཨཱིཊ་ཨཱ་ག་མ་སྟེར་བར་བྱ་སྟེ། འིཨཱིཊ་བྱིན་པས། ཨཱ་སཱིཏྲ། ལྷག་མ་རྣམས་ལ་་་སིཔྲ་ལ་ཨཱིཊ། ཨ་མིཔྲ་ལ་ཨ་མི་དབྱི། ང་འགྲོ་བ་རྣམས་ལ་ཨ་དབྱི་བར་བྱེད་་་ཀྱང་དཔྲ་སོགས་ཀྱི་ཨཊ་དང་། དབྱངས་དང་པོའི་ཨཊ་མཚམས་སྦྱར་བས་་་གཟུགས་འདྲ་སྟེ། ཨཱ་ཙཱ། ཨཱ་སན། ཨཱ་སཱིཨཱ་ཙཱ། ཨཱསྟ། ་་་་:ཨཱ་སཾ། ཨཱ་སཱི། ཨཱསྨ། །སུ་ཞུ་ད། ན་ཏོ། ཞི་ལྷ། མི་མྨ་ཞུའམ་གྱི་བགླིང་། ཞོ་ཁའི་མ་༈ལིཏྲ་རོ་ཀུན་ཏུ་མྱང་བ་ལ། བརྟ་མ་ནའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ༑ཧའི་རྣའོས། ཧྲརྗེ། ཏ་ཐ་དག་གི་དྷའོས་ཏ་དྷ། །ཝུ་སོགས་ཀྱིས་དྲྭ་ཟླ། པ་འགྲོ་བའི་ཡོན་ཏན་བྱས་པས། ལེནྡྲ། རྙི་ཞེས་གནས་པ་ལ། ༈རྨི་ལྷོ་ལོ་པོ་་་དཱིགྷ༈ཌྷའི་ལྷ་ལ་དབྱི་ཞིང་རིང་པོ་ཡང་ངོ༌། །།ཚུ་རོལ་གྱི་མཐའི་ཌྷ་ཡིག་ནི་་ཌྷ་ཡིག་ཕ་རོལ་ལ་དབྱི་ཞིང་། དབྱངས་སྔོན་མ་དེ་ཐུང་ངུ་ཡིན་ན་རིང་པོར་་་འགྱུར་ཏེ། ཌྷ་སྔོན་མ་ཕྱིས་པས། ལེ་རྨི། དེ་བཞིན་དུ་བསྒྲུབས་པས། ལཱི་ཌྷལི་ཧི་ནྟི། ལེཀཱི། ལཱི་:ཌྷལཱི་ཌྷ། ལཉྩི། ལི་ཧཱ། །ལིཧྥ། ལས་བ། ༈ཆོག་འབྱུང་ལིང་ལ། །ལི་ཧཱུཏྲ། ལི་ཧྥ་ཏྭཾ། ལི་ཧྥུསོགས་སོ། །༈ཤིས་བྱེད་ལོཊ་ལ། །ལེ་ཟླུ། ལཱི་ཎཱཏྲ། ལཱི་ཏཱཾ། ལི་ཧནྟུ་སོགས་་་དང༌། །༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་ལེཊ། ཌཨ་ལཱི་ཌྷ། །ཨ་ལཱི་ཧན་་་སོགས་སོ། །།།འོ་་་་་་་། །བཀའ་གདམས་གར་ག་འི། ཁྱུང་ལ་༈བི་ཨཱདྲ་སྔོན་མའི་ཙཀྐིང་གསལ་བར་བརྗོད་པ་ལ། སྒོ་རྭ་ངོ། ས་ཀ་སོགས་ཀྱི་ཡང་ངོ༌། །ཁ་། ཛྷ་ས་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཕ་རོལ་ལ་བྱིངས་མཐའི་སྲ་།
