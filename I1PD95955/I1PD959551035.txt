།བསྟན་འགྱུར། ་སྒྲ་མདོ། ༼སེ༽། སྐོན་ལཙམི་དེ། འདབ་མའི་ཤིང་རྟའོ། །ཨཎ་ཛ་སྒོང་སྐྱེས་སོ། །ན་གོ་ཀ། ཤིང་གནས་ཏེ། ནག་འགྲོ་མེད་ཅེས་པ་ཤིང་གི་མིང་ངོ༌། །བཱ་ཛི་ཀི་ར། སྲིན་བུ་ཟའོ། །བིབྦ་་བྱའོ༑། བིཀི་ར། འདབ་ཆགས་སོ། །།ནཱི་ཊདྡྷ་པ། ་ཚང་སྐྱེས་སོ། །ག་་་ཏུརླནྟ། གཤོག་ལྡན་ནོ། །པིཏྤན། འཕུར་འོད་དོད། ནབྷ་སངྒ་མ། མཁའ་ལ་སྤྱོད་དོ། །།བ་སྤུ་བ། དེ་ནས་ཧཱ་རི་ཏ་ལ་སོགས་པ་བྱའི་ཁྱད་པར་གྱི་མིང་སྟེ། །ཙི་ཏིཏྟར་སོགས་པ་ནི་སྲེག་པའི་མིང་ངོ༌། །ཛཱ་བ་ཛཱི་བ། ལ་སོགས་པ་ཤང་ཤང་ཏིའུའི་མིང་སྟེ། ཛཱི་་་༌་བཉྫ་བ་ནི་འཚོ་བའམ་འཚོ་བྱེད་ཅེས་བྱ་སྟེ། དུག་གི་འཆི་བ་འཚོ་བར་བྱེད་པས་སོ། །གརླུད་ཅེས་པ་བྱ་ལ་སོགས་པ་བྱ་དང་འབྲེལ་བའི་མིང་རྣམས་སོ། །ཐུན་༌་། །པ་མཱུ་ཧ་ལ་སོགས་ཚིགས་སུ་བཅད་པ་གཉིས་ཀྱིས་ཚོགས་པའི་མིང་སྟེ། ཌ་པ་མཱུ་ཧི། ཨཚིགས་པ་དང་། །ནི་བ་ཧ། །འདུས་པ་སྤུ་དང་། །བྷཱུ་ཧ། ༼ཕུང་པོ་དང་། པནྟོ་ཧ། །ཚོགས་དང་། སཾགྷཱ་ཏ། འདུས་པ་དང་། སཾ་ཙ་ཡ་ལ་སོགས་པ་རྣམས་སོ། །སེང་གེ་ལ་སོགས་པའི་སྡེའོ། །།གང་དུ་འང་། །།ག། ་་ཚེ་འདི་ནས་ང་ད་ན་བ་ལ་ཕྱི་མ་དེའོ། །དང་དེ་ནས་མིའི་སྡེ་ལས། མི་སྤྱིའི་མིང་ལ། །མ་བུཉྩ། མི་དང་། མ་བུཥ། ཡིད་སྐྱེས་དང༌། མཅུ། ཤེས་ལྡན་དང༌། མ་བུ་ཛ། མ་ནུའི་བུ་དང༌། མ་ན་བ། ཡིད་ལྡན་དང༌། །ན་ར། མི་དང་། པུ་མཱརྨ། །སྐྱེས་པ་དང་། པཉྩ་ཛན་ལྔ་སྐྱེས་ཏེ། ས་ལ་སོགས་པའི་བྱུག་པ་ལྔ་རྣམས་ལས་སྐྱེས་པའོ། །་། པུ་རུཥ་དང༌། སུ་རུཥ། སྐྱེས་བུའོ། །བྲི་མི་རྣམས་སོ། །།ཝི། ལ་བུད་མེད་ཀྱི་མིང་ལ། །སྡེ་བུད་མེད་དང༌། ཡོ་གཏི་བཙུན་མོ་དང་། ཨ་བ་ལ། སྟོབས་ཆེ་མེད་མ་དང༌། བཱ་རི་མི་མ་དང་། པཱི་མནྟི་ནཱི། །
