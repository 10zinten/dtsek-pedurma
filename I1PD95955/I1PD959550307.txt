ཁོ་པ་ཚོ་ལ་བློན་སྤྲོ་སྐལཔར་གནས་པར་བྱེད། ་ཨ་བ་ཏིཥྛ་ཏེ། ཀུན་ཏུ་གནས་པར་བྱེད། པོ་པྲ་ཏི་ཥྛ་ཏེ། རབ་ཏུ་གནས་པར་་་བྱེད། པཨུ་པ་ཏིཥྛ་ཏེ། …ཉེ་བར་གནས་པར་བྱེད། བི་ཏིཥྛ་ཏེ་རྣམ་པར་གནས་པར་བྱེད་སོགས་སོ། །སོན། ༈ཨུ་དཱི་བྷི་ཙིཏ་བ། ཨུ་ཏྲ་བི་དག་ལས་ཏ་པ་ལའོ། །ཉེར་བསྒྱུར་ཨུཏྲ་དང་བི་དག་ལས་ཕ་རོལ་གྱི་ཏ་པ་ལའང་ཨཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། །ཨུཏྟ་པ་་་ཏེ༑ཤིན་ཏུ་གདུང་བར་བྱེད། པི་ཏ་པ་ཏེ་རྣམ་པར་གདུང་བར་བྱེད་སོགས་སོ། །ཊ་ཏུ། །ལ་འཁྱི་༈ཨཱ་ངོ་ཡ་མ་ཧ་ནཨཱཛ྄་ལས་ཡམ་ཧན་ལའོ། །ཉེར་བསྒྱུར་ཨཱང་ལས་ཕ་རོལ་གྱི་ཡམ་དང་ཧན་དག་ལའང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། ཡམ་ལ། ཨཱ་ཡཙྪ་ཏེ་ཀུན་ཏུ་འགོག་པར་བྱེད་སོགས་དང་། ཨ་སྔོན་མའི་ཧན་འཚེ་ཅིབ་ལ། ༈ལོ་པསྡུ་ནུ་དཏྟ་ནཾ། ་་ཨ་ནུ་དཱཏྟ་རྣམས་དབྱི་བ་ཉིད་དོ། །བྱིངས་མཐའི་ཨ་ནུ་དཱཏྟ་སྟེ་སྐད་གསང་དམའ་བར་འདོན་པ་ཉ་མ་སོ་སོར་བསྡུ་བ་རྣམས་ནི་ཛྷ་སའི་ཡི་གེ་ཀ་ང་འགྲོ་བ་རྣམས་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། །ཨཱ་ཧ་ཏེ། ཀུན་ཏུ་སྣུན་པར་བྱེད། ཨཱཀྵ་ཏེ། ཨ་ཏེ། ཨཱ་ཧ་སེ། ཨཱབྷཱ་ཐེ། ཨཱ་ཧ་དྷེ། ཨཱངྒེ། ཨཱ་ཧཾ་བ་ཧེ། ཨཱ་ཧཾ་མ་ཧེ་སོགས་སོ། །བདེ་བ་ག་ཎ་ས་རི་བ། མ་ཎེ་ཁལོ་ཀ་ཀོ་ལ། ༈ཨུ་ཏཧྥ་རནྡྷཱ་ྋགེ། ཨུཏྲ་ལས་ཙ་ར་ལ་འདོན་པ་ལའོ། །ཉེར་བསྒྱུར་་་་ཨུཏྲ་ལས་ཕ་རོལ་གྱི་ཙ་ར་གཡོ་བ་ལས་འདོན་པ་དང་བཅས་པའི་དོན་ལ་ཨཱཏྨ་ནེ་། པ་ད་རྣམས་འཇུག་སྟེ། ཨུཙྩར་ཏེ་སཱུ་ཏྲ། …སྤྱོད་ཅིང་གཞུང་འདོན་པར་བྱེད། རྣམ་དབྱེ་་གསུམ་པའི་མཐའ་ཅན་དང་སྦྱོར་བའི་སཾ་སྔོན་མའི་ཙ་ར་ལའང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་པར་བཤད་དེ། །ཨ་ཤུ་ན་སཾ་ཙར་ཏེ། རྟ་ཡིས་ཀུན་ཏུ་སྤྱོད་པར་བྱེད། ་ ོ་རཱ་ཐེ་ན་སཾ་ཙ་ར་ཏེ། ཤིང་རྟས་ཀུན་ཏུ་སྤྱོད་པར་བྱེད། །བྱཱ་བ་པ་རི་བྷུཀྲི་ཡབི་ཨ་བ་པ་རི་རྣམས་ལས་ཀྲི་ལའོ། །ཉེར་བསྒྱུར་བ་དང་ཨ་བ་དང་པ་རི་རྣམས་ལས་ཕ་རོལ་གྱི་ཀཱི་ལ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་འཇུག་སྟེ། བི་ཀཱི་ཎཱི་ཏེ། རྣམ་པར་འཚོང་བར་བྱེད། ལྷོ་ཨ་བ་ཀཱི་ཎཱི་།
