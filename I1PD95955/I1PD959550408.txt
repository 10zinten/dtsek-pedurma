ཤི་ཏཾ་རུ་ཏཾ། ཌ། སྟི་པྲ་ཏི་ཐུཏྠ་ཏི་དྷཱ་ནེ་གཱི་ཏཾ་གཱ་ན་མི་མེ་ས་མེ། །ནི་ཥཱ་དནྟ་བཱ་ཤི་ཏཾ། །མོ་ནི་སྒྲ་བརྙན་བྲག་ཅའོ། །གླུ་དང་གླུ་དབྱངས་འདི་དག་མཚུངས། །འཁོར་ཉན་དྲང་བྷ་གཱནྟཱ་ར་ཥ་ཛཱ་མ་དྷ་མ་དྷི་བ་ཏཱ། །པ་མལྕེ་ཏུ་མཱི་སསྟ་ཏ་ནི་སྲོང་ས་འཛིན་དང༌། །དྲུག་སྐྱེས་བར་མ་བློ་གསལ་དང༌། །ལྔ་པ་ཞེས་བྱ་བདུན་པོ་འདི། །རྒྱུད་དང་ཀཎྜིཏྠི་ཏཱསཱ་རཱ། །ཀཱ་ཀ་ལཾ་ཏུ་ཀ་ལེ་སཱུམྨེ་དྷཱ་ནོ་ཏུ་མ་དྷུ་རབྷུ་ཊེ། །ཀ་ལོ་མནྡཱསྐུ་མགྲིན་ལས་སྐྱེས་པའི་སྒྲ། །ཀཱ་ཀ་ལཱི་ནི་སྒྲ་ཆ་གཉིས། །མི་གསལ་སྙན་པ་དྷཱ་ནིའོ། །སྔགས་ནི་ཟབ་གཙླི་རེ་ཏཱ་རོ་ཏུཙེ་ཏྲ་ཡསྡི་། །ས་མ་ནཱི་ཏ་ལ་ཡསྟེ་ཀ་ཏཱ་ལོ་བཱི་ཎཱ་ཏུ་མོའི་དབྱངས་ཡིན་ནོ། །ཏཱ་ར་མཐོན་པོ་གསུམ་པ་གསུམ། །ས་མ་ནི་ནི་ཆ་ལང་གཅིག། ཕེག་དང་པི་ཏུ་བལྤ་ཀཱི། །བི་པཝཱི་སཱུ་ཏུ་ཏནི་བྷིསཔྟ་བྷིབ་རི་བཱ་དི་ནི། །།ཏ་ཏམཱི་ཎཱ་དི་ཝང་བསྨྲ་ཀཱི། །རྣམ་པར་སྤྲོ་དང་ཞན་པ་ནི། །རྒྱུད་བདུན་ཡོངས་སུ་བསྒྲགས་པའི་སྒྲ། །།པི་ཝང་ལ་ཀམ་བྱ་མཱ་ན་དྭ་མུ་ར་ཛཱ་དི་ཀཾ། །ཐཾ་ཤཱ་དི་ཀནྡྷཱ་ཤུ་ཏི་རཾ་ཀཾ་སྱ་ཏཱ་ལཱ་དི་ཀ་སོགས་རྒྱུད་ཀྱི་སྒྲ། །རྔ་ལ་སོགས་པ་སྦུབས་ཀྱི་སྒྲ། །གླིང་བུ་ལ་སོགས་བུ་གའི་སྒྲ། །སིལ་སྙན་སྒྲ་ནི་ཆ་གྷ་ནཾ། ལྟ་། ཙ་ཏུརྦ་དྷ་མི་དཝཱ་པཾ་བཱ་དི་ཏཱ་ཏོ་ཏྲ་ནཱ་མ་ཀཾ། །ཛྲ་དཾ་གཱ་མུ་ར་ཛཱ་བྷེ་དཱ་ལང་སོགས། །རྣམ་པ་འདི་བཞི་རོལ་མོའི་སྒྲ། །བཱ་དི་ཏཱ་ཏོ་ཞེས་བྱའོ། །།རྫ་རྔ་རྔ་ཟླུམ་དབྱེ་བ་ལས། །དཾ་ཀཱ་ལིངྒཱ་ཀཱསྔ་ཡ། །སྙདྨ་ཤཔ་ཊོ་ཧོ་ཊཀཱ་བྷེནཱ་མཱ་ན་ཀ་དུནུ་བྷི། །རྔ་རིངས་ཨུངྒཱ་ཀ་གསུམ་དང༌། །པ་ཊ་ཀ་དང་ཊག་རྔར་གྲགས། །འཇིགས་བྱེད་རྔ་དང་རྔ་བོ་ཆེ། །ཨཱ་ན་ཀཔ་ཊ་ཧོཞསྟི་ས་རྐོ་ཎོ་བཱི་ཎཱ་དིབཱ་ད་ནཾ། །བཱི་ཎཱ་དཎྜཔྲ་བཱ་ལ་སྙཏྟ་ཀུཤྲཱི་ཏུ་པྲ་ཨཱན་ཀརྔ་མོ་མ་ཡིན། །ཀོ་ཎོ་པི་ཝང་ལ་སོགས་སྒྲ། །པི་ཝང་དབྱུག་པ་པྲ་བཱ་ལ། །རྩ་བའི་ཤིང་འཁྱོག་སེ་བ་ཀ། ཀོ་ལམྦ་ཀསྟུ་ཀཱ་ཡོཞསྭཱ་ཨུ་པ་ནཱ་ཧོ་ནི་བནྡྷ་ན། །བཱ་རྱ་པྲ་བྷེ་དཱ་ཌ་སེ་བ་ཀ། །ཏམྦུ་ར་ཟེར་ཀོ་ལམྤ་ཀ་ཁོག་པའོ། །ཐག་པའི་གནས་དང་འཆིང་བའོ། །རོལ་མོའི་བྱེ་བྲག་ཌ་མ་རུ་མ་ཊུ་ཌིཎྜི་མཛྷཛྷ་རཱ། །མརྡ་ལཔ་ཎ་:བ་རྩེ་ཙ་ནཎྜ་ཀཱི་ལཱ་སི་ཀ་ས་ཅང་ཏེའུ་མ་རུ། །མ་ཊུ་ཌིཎྜི་མ་རྫ་རྔ། །རྔ་དང་ཁར་རྔ་གཞན་ཡང་ངོ༌། །གར་མ་དང་ནི་རོལ་རྩེད་—།
