བསྟན་འགྱུར། སྒྲ་མདོ། སྒེར་་བ་#ཀྱང་བྱའོ། མ་ཧ་མི་ཀཱ་ཏུ་སཱ་སཱུཏྤ་རསྤ་རཾ་ཡོ་བྷ་བ་ཏུ་ཧཾ་ཀཱ་ར། །དྲ་བི་ཎན་རས་ཧོ་བ་ཞེས་སྨྲ་ཕན་ཚུན་འགྲན། །།གང་ཞིག་ང་རྒྱལ་བྱེད་པའོ། །འཚམ་མེད་ཐལ་འགྲོ་རྟུལ་ཕོད་དང༌། །ལ་སོརཱ་ཎི་་མ་ཤུཉྩ། །སཀྟིཔ་རཱ་ཀྲ་མ་པྲ་ཎོ་བི་ཀྲ་མསཱུ་ཏི་ཤཏྟི་ཏཱ། ཁོ། སྟོབས་ལྡན་དཔའ་དང་ཉམ་ང་མེད། །ནུས་ལྡན་ཕ་རོལ་གནོན་པའོ། །གྱད་དང་ཤིན་ཏུ་དཔའ་བའོ། །བཱི་ར་པཱ་ནནྟི་ཡཏྟཱ་ནཾ་ཤྲཏྟེ་བྷ་བི་ནི་བཱ་ར་ཎེ། ་ཊ། ཡུབྷཾ་མཱ་ཡོ་དྷ་དཔའ་བོའི་བཏུང་བ་གང་ཞིག་གི། །ཆང་འཐུངས་གཡུལ་གྱི་སྔ་ཕྱི་རུ། །འཁྲུག་པ་འགྱེད་པ་རྒྱལ་ནཾ་ཛ་ཧཱུྃ་པྲ་དྷ་ནཾ་པྲ་བི་དཱ་ར་ཎཾ། །ཛྲ་དྷ་མཱ་སྐནྡ་ནཾ་སཾ་ཏུཾ་ས་མཱི་ཀཾ་སཱམྤ་ར་བྱེད་པ། །འཇོམས་པ་རབ་ཏུ་འཇོམས་བྱེད་པ། །མྲྀ་ཌྷ་སྐེམ་བྱེད་སཾ་ཁྱ་དང༌། །གཡུལ་དང་སཾ་པ་རཱ་ཡི་ཀཾ། ལྟེ་། ཨཔྲི་ཡཾ་ས་མ་རཱ་ཎཱི་ཀ་ར་ཎཱ་སྐ་ལ་ཧ་བི་གྲ་ཧོ། །སཾ་པྲ་ཧཱ་ཡི་ཀཾ། །མཚོན་འཛིངས་མཉམ་ཤི་ཨ་ཎཱི་ཀ། །གཡུལ་དུ་འཐབ་མོ་བི་གྲ་ཧོ། །འཕྲོག་འཇོམས་དགེ་རཱ་བྷི་སམྦུ་ཏ་ཀ་ལི་སཾ་སྤོ་ཊ་སཾ་ཡུ་གཱ། །ཨ་བྷི་མཛྲ་ས་མཱ་གྷ་ཏ་སཾ་གཱ་མཱ་བྲལ་རྩོད་པ་དང༌། །ཤིན་ཏུ་འདྲལ་དང་སཾ་ཡུ་ག་། མངོན་པར་ཉེད་དང་རྡེག་གསོད་དང༌། །ས་མུ་བྷཱ་ག་སཱ་ཧ་བཱ། །ས་མུ་དཱ་ཡསྦི་ཡསཾ་ཡཏྠ་མི་ཏཱ་ཛི་ས་མིརླུ་དྷ། །ནི་ཡུདྡྷཾ་ད་ཡ་འཐབ་འཁྲུག་ལོང་། །མི་བདེ་གཡུལ་འགྱེད་ཨ་ཛི་དང་། །ས་མི་རླུ་དྷ་ཞེས་པའོ། །མཚོན་བྲལ་བཱ་ཧུ་ཡུ་དྷེ་སཱུཏྟ་མུ་ལཾ་ར་ཎ་སཾ་ཀུ་ལཾ། །ཀྱེ་་ཏུ་སིཾ་ཧ་ནཱ་དསཥ་རི་ཎཱ་གྱད་ཀྱི་དཔུང་འཛིན་དང་། །ཏ་མུ་ལཾ་ནི་གཡུལ་དུ་གསོད། །།ཅ་ཅོ་སྒྲོགས་པ་སེང་གེའི་སྒྲ། །གླང་པོའི་གྷ་ཊི་ནཱ་ྋག་ཊཱ། ་། ཀྲནྡ་ནཾ་ཡོདྡྷཾ་སཾ་རཱ་བེ་བྲྀ་ཧི་ཏཾ་ཀ་རི་གརྫི་ཏཾ། །བིས་བུམ་པའི་སྒྲ་དང་ནི། །དུད་མོ་ཅ་ཅོ་དར་དིར་དང༌། །འབྲུ་མང་པོ་དང་གླང་པོའི་སྒྲ། ཟ། བི་སྭ་ར་ནི་རོ་དྷ་ནུ་ཥསྭཱ་ནཔ་ཊ་ཧཱ་ཌམྤ་རོ་ས་མོ། །པྲ་ས་བྷསྐུ་བཱ་ལཱཏྟ་རོ་ཧ་ཅེཞདྷཱསྐྱ་གཞུ་ཡི་སྒྲ། །འཁྲུག་རྔ་རྔམ་པ་རོ་དག་མཚུངས། །གཡོ་སྒྱུ་ངན་ཐབས་ཧ་ཟའོ། །དེ་ནས་འཁྲུལ་དང་ལི་ཏི་ཀ་ལཾ། །ཨ་ཛ་རུཾ་ཀཱི་བ་མུངྒུ་ཏ་ཨུ་པ་སངྒྷས་མཾ་ཏྲ་ཡཾ། ་་། མུཙྪ་ནོར་པ་དང༌། །མི་ཤེས་མ་ནིང་ལྟས་བྱུང་དང༌། །༈། ཉེར་བསྒྱུར་དང་ནི་མཚུངས་པ་གསུམ། :། བརྒྱལ་—།
