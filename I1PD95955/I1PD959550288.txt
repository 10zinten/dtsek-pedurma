དག་གི་ཙུ། །རཾ་ཨ་རུ་བསྒྱུར་བས། ཙ་ཀྲྀཏི། ཙ་རི་ཀྲི། ཙ་རཱི་ཤྲཱི་ཞེས་གྲུབ་པ་ལས་རྣམ་དབྱེ་རྣམས་ཆོ་ག་སྔོན་མ་བཞིན་སྦྱོར་བ་ལས། བརྟ་མ་ན་ལཊ་ལ། ཙརྐ་རཱི་ཏི། ཙ་རི་ཀ་རཱི་ཏི། ཙ་རཱི་ཀ་རཱི་ཏི། ཙ་རྐརྟི། ཙ་རཱི་ཀརྟི། ཙ་རཱི་ཀརྟི༄༅། ཙ་ཀྲྀཏཙ་རི་ཀྲྀ་ྋཏཙ་རཱི་ཀྲྀ་ཏ༔ཙཀྲ་ཏི། ཙ་རི་ཀྲ་ཏི། ཙ་རཱི་ཀྲ་ཏི། ཙརྐ་རཱི་ཏི། ཙིཀ་རཱི་ཏི། ཙ་རི་ཀ་རཱི་ཏི། ཙ་རཱི་ཀ་རཱི་R། ཙརྐ་རྟི། ཙ་རི་ཀརྟི། ཙ་རི་ཀརྟི། ཙ་ཀྲྀ་ཐཙ་རི་ཀྲྀ་ཐཙ་རི་ཀྲྀ་ཐ། ཙཀྲྀ་ཐ། ཙ་རི་ཀྲྀ་ཐ། ཙ་རཱི་ཀྲྀ་ཐ། ཙཀ་རཱི་མི། ཙ་རི་ཀ་རཱི་མི། ཙ་་་རཱི་:ཀཱ་རི་མི། ཙརྐ་:རྨི། །ཙ་རི་ཀརྨི། ཙ་:རི་ཀརྨི། ཙཀྲྀ་བཙ་རི་ཀྲྀ་བ། ཙ་རི་ཀྲྀ་བཙཀྲྀ་མཙ་རི་ཀྲྀ་ྋམཙ་རི་ཀྲྀ་མ། །་༈ཆོག་འབྱུང་ལིང་ལ། ཙཀྲྀ་ཡཱཏྲ། ཙ་རི་ཀྲྀ་ཡཱཏྲ། ཙ་རི་ཀྲྀ་ཡཱཏྲ་་་སོགས་དང་། ༈ཤིས་བྱེད་ལོཊ་ལ། ཙཀ་རཱིཏུ། ཙ་རི་ཀ་རཱི་ཏུ། ཙ་རི་ཀ་་རཱི་ཏུ། ཙརྐ་ཆུ། ཙ་རི་ཀཆུ། ཙ་རཱི་ཀརྟུ། ཙཀྲྀ་ཙེཏཱཏྲ། ཙ་རི་ཀྲྀ་ཏཱཏྲ། ཙ་རི་ཀྲྀ་ཝེཏཱ་ཏྲ། ཙཀྲྀ་ཏྭཾ། ཙ་རི་ཀྲྀ་ཏྭཾ། ཙ་རཱི་ཀྲྀ་ཏྭཾ། ཙཀྲ་ཏུ། ཙ་རི་ཀྲ་ཏུ། ཙ་རི་ཀྲ་ཏུ་སོགས་དང་། ༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་ཙརྐ་རཱིཏྲ། ཨ་ཙ་:རི་ཀིརཱིཏྲ། ཨ་ཙ་རི་ཀ་རཱིཏྲ། ་ཨ་ཙརྐཙ་་ཨ་ཙ་རི་ཀཨ་ཙ་རཱི་ཀཨ་ཙཀྲྀ་ཏྭཾ། ཨ་ཙ་རཱི་ཀྲྀ་ཏཱཾ། ཨ་ཙ་རཱི་ཀྲྀ་ཏྭཾ། ཨ་ཙརྐུཨ་ཙ་རི་ཀུཨ་ཙ་རཱི་ཀྲུསོགས་སོ། །ཤོ་ལ་པ་ར་མི་བྱའོ། །།ཚེ་ཀྱི་ཁཀྱང་ལ་ལ། པ་བ་༈ཆོ་ག་རྣམས་དེ་བཞིན་དུ་བྱས་པས་གྲོགས་ཏུ་འཇུག་པས། བརྟ་མ་ན་ལཊ་,ལ༑བཛྲ་ཏཱི་ཏི། བ་རི་བྲྀ་ཏཱི་ཏི། བ་རཱི་པྲྀ་ཏཱི་ཏི། བརྦ་ཙེ། །བ་རི་བནྟི། བ་རཱི་བནྟི། བཛྲ་ཏྟ་བ་རི་བྲྀཏྟབ་རཱི་བྲྀཏྟབཛྲ་ཏ་ཏི། བ་རི་བྲི་་ཏ་ཏི། བ་རཱི་དྲྀ་ཏ་ཏི། བརྦ་རཱིཏི། བ་རི་བརཱིཏི། བ་རཱི་བཏཱིཏི། །བརྦ་་།
