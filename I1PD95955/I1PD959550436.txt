།སུ་ཀཱ་སྤོ་ཏ་ག་ཎ་རཱུ་པ་བི་ཀཱི་ར་ཎ། །མན་རཉྩརྐ་པཎྞེཉཏྲ་ཤུཀླཉལ་སུ་ཀཱ། །མི། སྤོ་ཏ་ཚོགས་གཟུགས་རྣམ་པར་འཐོར། །མནྡ་ར་དང་ཨརྐའི་འདབ། །།འདི་དཀར་ཨ་རྐ་ཏ་པ་སོ། །ཤི་བ་མལླི་པཱ་ཤུ་པཱ་ཏི་ཨེ་ཀཱཊྛི་ལོ་བ་ཀོ་བ་སུ། །།།བན་བྲོག་ལརྐ་ཏས་པ། །ཞི་བའི་མེ་ཏོག་ཕྱུགས་བདག་དགའ། །ཤིང་གཅིག་སྤྲ་བ་བ་སུའོ། །བན་ཧོང་ལེན་ཤིང་ད་ནི་བྲྀཀྵ་རུ་ཧཱ་ཛཱི་བནྟི་ཀེ་ཏུ་པི། །བཏྟཱ་ད་ནཱིཙྪིནྡ་རུ་ཧཱ་གུ་རླུ་ཙཱི་ལ་ཟ་བ་དང༌། །ཤིང་སྐྱེས་འཚོ་བྱེད་ཉིད་ཀྱང་ངོ༌། །བུ་ཟས་བཅད་སྐྱེ་སླེ་ཏྲེས་དང༌། །རྒྱུད་ལྡན་ཏནཱི་ཀཱ་མྲྀ་ཏཱ། །ཛཱི་པཾ་ཏི་ཀཱ་སོ་མ་བལླི་བི་ས་ཝཱ་མ་དྷུ་པདྡྷཱ་པི། །མཱུསྠཱ་དེ་བཱི་བདུད་རྩི་འཚོ་བྱེད་དང་། །ཟླ་བའི་མགུལ་བ་ཡངས་པ་ཅན། །སྦྲང་གི་ལོ་མ་ཉིད་ཀྱང་ངོ༌། །མཱུརྭ་ལྷ་མོ་མ་དྷུ་ར་སཱ་མོ་ར་ཊཱ་ཏེ་ཛ་ནཱི་:སུ་བཱི། །མ་དྷཱ་ལི་ཀཱ་དྷ་ནུའེ་རྟི་གོ་པཎྜི་པཱི་ལུ་པཎྜུ་པི། །སྦྲང་གི་རོ། །མོ་ར་ཊ་དང་འོང་ལྡན་ཐོས། །མ་དྷུ་ལི་ཀ་གཞུ་ཤིང་དང་། །བྲི་ཎི་ས་འདབ་པཱི་ལུའི་འདབ། །པཱ་ཋམྲཏྟཱ་བི་དྭ་ཀཙྪིསྦྲ་བ་ནི་ཐེ་ཡ་ཤཱི་ར་སཱ། །ཨི་ཀཱཙྪི་ལཱ་པཱ་བ་ཙེ་པཱ་ཋཱ་མ་ནུ་སྤོས་དཀར་དང༌། །བི་དྭ་ཀཎི་གནས་ཅན་དང་། །རྟེན་དང་རོ་ལྡན་སྡོང་བུ་གཅིག། ལཱི་པཱ་ཙཱི་ནཱ་བ་ན་ཏིཀྟི་ཀཱ། །ཀ་ཊུ་ཀ་ཌམྦྷ་རཱཥྚཤོ་ཀ་རོ་ཧི་ཎཱི་ཀ་ཊུ་རོ་ཧི་སྡིག་པའི་བྲན་མོ་པཱ་ཙཱི་ནཱ། །ནགས་ཀྱི་ཏིག་ཚ་བ་དང༌། ་། ཚ་བས་གང་དང་མྱ་ངན་མེད། །སྤོར་དང་ཎྜི༑། མཉྩ་བིཏྟཱ་ཀྲྀཥྛ་བྷེ་དཱི་ཙཀཱངྒཱི་ཤ་ཀུ་ལཱ་ད་ནི། །ཨཱཏྨ་གུཔྟ་ཛ་ཌཱཉངྒྷཎཱ་ཚ་སྤོར་ཉ་མཁྲིས་དང༌། །ནག་དབྱེ་འཁོར་ལོའི་ཡན་ལག་དང་། །བྱ་ཡི་ཟས་དང་བདག་སྦེད་དང་། །ཀཎྜ་རཱ་པྲ་བྲྀ་ཥཱ་ཡ་ཎྜི། །།རཾ་་བྲོཀྟ་ཤུ་ཀ་ཤིཝིཀ་པཱི་ཀཙྪཱ་ཛ་ཌཱ་དང་ནི་ཨ་དྷཎྜ། །གཡའ་ཤིང་བླ་བྲྀ་ཥཱ་ཡ་ཎི། །།ར་བྱ་པྲཀཱ་ཤུ་ཀ་ཤིམྤི། །ཀ་བཱི་རུས་སྦལ་སྤྲེའུ་དང་། །ཕྱེ་མརྐ་ཊཱི། །ཙི་དྲོ་པ་ཙི་ཏཱ་ནྱ་གྲོ་དྷཱི་དྲ་བནྟི་ཤཾ་བ་རཱི་པྲྀ་ཥཱ། །པྲ་ཏྱཱཀ་རི་མོ་ཉེ་བའི་རི་མོ་ཅན། །ནྱ་གྲོ་དྷི་དང་ཁུ་བ་ལྡན། །སྡོམ་པ་དང་ནི་ཆར་འབབ་དང་། །པྲ་ཏྱཱཀ་ཤེ་ནཱི་སུ་ཏ་ཤེ་ཎཱི་རཎཱ་མཱུཏི་ཀ་པཎྜུ་པི། །ཨ་པཱ་མཱརྒཤེ་ཁ་རི་ཀོ་དྷཱ་མཱརྒ་བ་མ་ཡུ་ཤེ་ནཱི་སུ་ཏ་ཤེ་ཊཱི། རཎྜ་བྱི་བའི་རྣ་བའོ། །ཨ་པཱ་མཱརྒ་ཐོད་ཅན་དང༌། །དྷཱ་མཱརྒ་དང་རྨ་བྱ་—།
