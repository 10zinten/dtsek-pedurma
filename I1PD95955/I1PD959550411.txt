pgཀ་ཀ་ལ་སྐུ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་་ཚུ་འབྱོར་ཎའོ་དོན་ཥཱ་རོ་པོ་གུ་ཎེ་ཥཱ་པི། །བི་རལླི་རོ་དྷོ་བི་དྭེ་ཏོ་མ་རྒྱུ་ཤོ་ཀོ་ཏུ་ཤུ་ཀ་པྲི་ཡཾ། །ཉེས་བསྒོ་ཨཱ་སཱུ་ཡ། །འགྲས་དང་འགལ་དང་རྣམ་པར་འགོག། ཡིད་གདུང་མྱ་ངན་གདུང་བ་མོ། །པཝཱཏྟཱ་པོཉནུ་ཏཱ་པཉྩ་བི་པྲ་ཏཱི་སཱ་ར་ཨི་ཏུ་པི། །།ཀོ་པ་ཀྲོ་དྷོཉམནྟ་རོ་ཕྱི་ནས་གདུང་བ་རྫས་གདུང་དང༌། །རྣམ་པར་བསྙིལ་བ་འགྱོད་པའོ། །ཁྲོ་བ་འཁྲུག་པ་འཚིག་དང་ཁྲོས། །ཥ་བྲ་ཏི་ག་རུ་ཥ་ཀྲུ་རྡོ་སྦི་ཡོ།  ོ། ཤུ་ཙོ་ཏུ་ཙ་རི་ཏེ་ཤི་ལ་མུནྣཱ་དཕྱིཏྟ་བི་བྷཱ་མ། །མི་གནས་གཞན་སྣུན་ཁོང་ཁྲོ་མོ། །གཙང་བར་སྤྱོད་པ་ཚུལ་ཁྲིམས་སོ། །སྨྱོ་བ་སེམས་ནི་རྣམ་པར་འཁྲུལ། །སྲེ་མཱ་ནཱ་པྲ་ཡ་ཏཱ་ཧཱཌྷ་བྲེ་མསྣེ་ཧོཉཐ་དོ་ཧ་དགའ་དང་མོས་དང་ཡིད་མཐུན་དང༌། །སེམས་སྣུམ་དགའ་དང་དེ་ནས་ནི། །འཇོ་དང་འདོད་དང་དཾ༑། ཨིཙྪཱ་ཀཾ་སྤྲོ་ཧེ་ཧཱ་ཊཾ་ཊ་བཱཉྙཱ་་ལིསྨཱ་མ་ནོ་ར་ཐ། །ཀཱ་མངོན་འདོད་དང༌། །བརྐམ་དང་རེ་བ་འདོད་སྤྲོ་དང་། །མངོན་འདོད་ཡིད་ཀྱི་ཤིང་རྟ་དང་། །འདོད་མོཉབྷི་ལཱ་ཥ་སྟརྟ་ས་མ་ཧཱནཱ་ལ་སཱ་དྭ་ཡོ། །ཨུ་པ་དྷིནཱ་དྷརྨ་ཙིནྟཱ་དང་མངོན་ཞེན་སྲེད་པ་དང༌། །ཞེན་པ་དང་ནི་མཚུངས་པ་གཉིས། །ཨུ་བ་དྷི་ནི་ཆོས་སེམས་སྐྱེས། །སུ་སཱུ་དྷིཛྙཱ་ན་སཱི་བྱ་ཐཱ། །སཱུཙྪིན་སྨྲ་ཏི་རཱ་དྷཱ་ན་མུཀྐཀྐོཥ་ལི་ཀེ་ས་ཨ་དྷི་ཡིད་ཀྱི་གདུང་བའོ། །སེམས་དང་དྲན་བྱེད་བསམ་མནོའོ། །ཡིད་གཅགས་ཤིན་ཏུ་སེམས་པ་མེ༑་། ཨུཏྟཱ་ཧོཉརྒྱ་པ་སཱ་ཡསཱུཏ་ས་བཱིརླ་མ་ཏི་ཤཏྟི་བྷཱཀ། ཀ་པ་ཊཉསྟི་བྱཱ་མཚུངས། །སྤྲོ་བ་དང་དེ་ཉིདགའ་བའོ། ༈། དེ་བརྩོན་ཤིན་ཏུ་ནུས་པའི་ཆ། །ངན་གཡོ་ལོ་མིན་སྒྱུ་ཛ་དམྦྷ་ད་དྷ་ཡརྨ་ཀི་ཏ་བེ། །ཀུ་སྲྀ་ཏཱིནི་ཀྲྀ་ཏིསཱ་ཐཾ་པྲ་མཱ་དོཉན་དང་བརྐུ། ་། ཁྲམ་དང་ངན་ཐབས་ཟོག་དང་ནི། །ཟོལ་དང་ངན་གཡོ་བཅོས་མའོ། །བག་མེད་པ་དང་བ་དྷཱ་ན་ཏཱ། །ཀོ་ཏུ་ཧ་ལཾ་ཀོ་ཏུ་ཀཉྩ་ཀུ་ཏུ་ཀཉྩ་ཀུ་ཏུ་ཧ་ལཾ། །སི་ཎཱམྦི་ཤེས་བཞིན་མེད། །ཡ་མཚན་དང་ནི་ངོ་མཚར་དང༌། །ལྟད་མོར་ཆེ་དང་མཚར་ཆེ་བ། །བུད་མེད་ལཱ་ས་བིཙྩོ་ཀ་བི་བྷཱ་མཱ་ལ་ལི་ཏནྟ་ཐཱ། །ཧེ་ལཱ་ལཱི་ལེ་ཏུ་མཱི་ཧཱ་བཱ༔རྣམས་ཀྱི་འཇོ་སྒེག་དང་། །རྣམ་འགྱུར་རྣམ་འཕྲུལ་རོལ་པ་དང་། །དེ་བཞིན་སྐུལ་དང་ཟུར་གྱིས་སྟོན། །—།
