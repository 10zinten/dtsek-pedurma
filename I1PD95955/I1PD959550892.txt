ན་ཏེ་ད་རྒྱུ་ཏོག་སྐྱེ་བ་བརྒྱུད་དང་ཀ་ཀྑལུས་ཀྱི་རྩ་ལ་མིང་གསུམ་སྟེ། ནཱ་ཌི་དང་། དྷ་མ་ནི་དང༌། ཤི་རཱ་་་་རྣམས་ཏེ། བུ་ག་དང་ལྡན་ཞིང་རླུང་འབབ་པ་དང་། ཟསྲད་བུ་བཞིན་རུས་པ་་འཆིང་བར་བྱེད་པ་དང༌། ཀློ་མ་དང༌། མནྟི་ཏྟཾ། ཀླད་རྒྱས་སམ་ཀླད་པ་སྟེ། མགོ་བོ་ལས་བྱུང་བ་མར་གྱི་གོང་བུ་ལྟ་བུ་གྷི་བྷི་ཞེས་གྲགས་པའོ། །ནང་ན་གནས་པའི་ཟས་ལས་བྱུང་བའི་དྲི་མ་འབབ་པ་ནི་སྣབས་ལ་སོགས་པ་རྣམ་པ་བཅུ་གཉིས་ཏེ། དེ་ལྟར་ཡང་། ཞག་དང་ཁུ་བ་རྐང་དང་ཁྲག་། རྣ་སྤབས་བཤང་གཅི་སྣབས་་་མཚན་མ།  ོ། བད་ཀན་དང་ནི་མཆི་མ་རྡུལ། །མི་ཡི་དྲི་མ་བཅུ་གཉིས་སོ། །ཞེས་སོ། །ལས་དེ་ལ་ཚོང་དེ་ལ་ཁ། ལ་ལ་དམནཔའཚེ་བྱེད་པ༌ཊ་་བ་གསུམ་གྱིས་མཆིལ་མའི་མིང་སྟེ། སྲྀ་ཎི་ཀ། ཁ་ཆུ་དང་། ཙནྡྲ་ནཱི་དང་། ལཱ་ལཱ། ཉྙོམཆིལ་མའོ། །་་་་་་་། ངའོ། བློན་པོ་ge། ང་། བྱ་པ། པད་མ་གཉིས་ཀྱིས་དྲི་ཆུའི་མིང་སྟེ། མཱུ་ཏྲ། དྲི་ཆུའམ་གཅིན་དང་། པྲ་སྲ་བ། གཅི་བའམ་རྒྱུན་འབབ་པོ། །དམ་ཚིག་འོ་ཏེ་ལྔ་ཊ་པའི་ཡ:ཨུཙྪཱ་ར་ལ་སོགས་པ་བཤང་བའི་མིང་སྟེ། ཨུཙྪཱ་ར་དོར་བྱའོ། །ཨ་བསྐ་ར། ༼ཁྱབ་ལྡན་ནོ། །ཤམ་ལ། །འདྲེས་མའོ། །ཤ་ཀྲྀཏ། འགྲོ་བྱེད་དོ། །པུ་རི་ཎཾ། བཤང་བའོ། །།གུ་ཐ། ཕྱི་སའོ། །བརྩ་སྐོ། དྲི་མའོ། །བིཏྟཱ་—།
