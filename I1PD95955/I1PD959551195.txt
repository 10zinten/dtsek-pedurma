བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽བྲྀཏྟ་མི་དཾ་ཀ་ཏི་ཐ་མི་ཧེ་ཏུ་པ་རི་ལི་ཁེ་ཀཱ་དྱ་ནུ་ཀྲ་མ་དཱི་གུ་ཎཾ། ཏ་ཀྲྀ་ཏུ་ལ་བྲྀཏྟ་འདི་ནི་དུ་པ་ཡིན་ཞེས་འདིར། །སྟེང་དུ་གཅིག་སོགས་རིམ་པས་ཉིས་འགྱུར་བྲི། །དེར་ནི་ཡང་བའི་བྷནྡྷ་པིཌེ་སེ་ཀེ་ཏཱ་ཏི་ཐ་མུདྲིཏྟཾ༑，རཱུ་པཾ་ནྱ་སྐྱོ་ཏི་ཙྪེད་ཀླུ་སྟེང་སྡོམ་གཅིག་དང་བཅས། །འདི་ལྟ་ཉིད་ཡིན་ཞེས་པ་ཨུདྡིབྦའོ། །གཟུགས་བཀོད་ཡི་གེའི་གྲངས་ཀྱིས་ར་སཾ་ཁྱིས་མོ་པ་རི་ཀླུ་པིཀྲ་མ་ཁཆེ་རུཏྟཱ་ནི་བསླང་། །མཉམ་པ་སྟེང་དུ་བཅུག་པ་ཡིས། །ལངས་པ་རིམ་པས་ཐུང་དུ་སྟེ། །བྲྀཏྟ་ཡང་མེད་གཅིག་ཡང་བྲྀཏྟ་མ་ལ་གྷ་ཀ་ལ་བྷ་དི། :། པི་ག་ལཱི་ལ་གྲུ་ཀྲི་ཡཱ། །ཨུཀྐི་སྱ་རཱུ་པ་སོགས། །དམར་སེར་གྱི་ཡང་བའི་བྱ་བའོ། །གཟུགས་ནི་དང་པོར་མ་བྱས་ཀྱང་། །སྟེང་གཞུག་མཱ་དྷྱ་མ་ཀྲི་ཏེ་པི་སཾ་ཁཾ་ལི་ཁེཏྟ་ཏོཉངྒྷསྦྱིསྭ་སཱ་དྷ་ར་ས་མ་མི་སྲི་ཧྲི་ཏྟ་མ་ལ་གྲངས་བྲི་དེ་ནས་འོག། གནས་པའི་རང་རང་འོག་མཉམ་བསྲེ། །བྲྀཏྟ་ཡང་མེད་གཅིག་ཡང་ཧྥེ་ཀ་ལ་གྷ་དི། །T། ཨ་ཡ་མ་སྨ་དཱི་ཡཱ་ལ་གྲུ་ཀྲི་ཡཱ། །།།སོགས། །འདི་ནི་ཁོ་བོའི་ཡང་བའི་བྱ་བའོ། །ཀཱ་སཾ་ཀྵ་བྲྀཏྟཱ་ནཱ་མ་ཀཱ་ར་བཱ་རཱ་ཎི་ཡ་མ་གུ་ཎཾ་རཱུ་པཾ། ཨུདྷིཥྛ་དེ་ཥཱནྟོ་དཱི་གུ་བྲྀཏྟ་རྣམས་ཀྱི་གྲངས་ཅི་གཟུགས། །ཟུང་གིས་ཡི་གེའི་གྲངས་སུ་བསྒྱུར། །ཨུདྡིཊྚ་ཡི་ཨངྐའི་མཐའ། །ཎོ་ལ་གུ་ཀརྨ་པིཎྜ། །སཾ་ཁྱི་བ་བཙྪཱ་གུ་ཎི་ཏཱ་ཏ་ཏསྟ་དདྷཾ་ཏ་ཏོ་ཕྱྭ་བྷ་ཡ་པིཎྞ། །ཉིས་འགྱུར་གཉིས་ཡང་བའི་བྱ་བ་བསྡུ། །གྲངས་ཉིད་བཎས་བསྒྱུར་དེ་ནས། །དེ་ཕྱེད་དེ་ནས་གཉིས་ཀའང་སྡོམ། །ཨཱ་དོ་བཙྪཱ་མ་དྷཱ་གུ་ར་བོ་ལ་གྷ་བསྟ་ད་ནུ་མཱ་ཏཱ། །ཨ་དྷཱ་ཏིདྨ་ག་དྷཉྩ་དྲི་གུ་དང་པོ་ཡི་གེ་དབུས་ལྕི་ཡང་། །དེ་རྗེས་ཕྱི་མོ་རྣམས་ཡིན་ནོ། །ཡི་གེ་བྲྀཏྟ་གཉིས་ཀྱིས་བསྒྱུར། །གཅིག་དམན་ཨ་དྷཱ་ཎི་ཀོ་ནཱ་ནི་བཙྩཱ་བྲྀཏྟཱ་ནི། །ཏུ་མྱ་ཧི་རུ་ལ་གྷ་ནཾ་བྷཱུ་མིསྟ་ཏཱནྟ་རཱ་ཎཱཉྩ། །འཕྲེད་དང་འོག། ལྕི་ཡང་རྣམས་ཀྱི་ས་དང་ནི། །དེ་ལ་བར་ཡང་མཚུངས་པའོ། །པྲསཱུ་ར་དང་ནཋ་ཨུདྡི། །ཨ་ནུ་པཱུ་ཐཾ་པྲསྨཱ་རོ་ནཊོདྷིཥྛ་ལ་གླུ་ཀྲི་ཡཱ་ཙི་བ་སཾ་ཀྵ། །ཏ་ཐཱ་དྷཱ་ཡོ་གོ་བྲྀཏྟེ་ཏུ་ཡང་བའི་བྱ་བ་གྲངས་དང་ཨ་ཧྥུ་སྟེ། །དེ་ལྟར་དྲུག་པོ་དེ་རྣམས་རིམ་པ་བཞིན། །བྲྀཏྟ་རྣམས་ལ་རྟོགས་བྱེད་།
