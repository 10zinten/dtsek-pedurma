ཏ:་བ་ཨཁོ་བསྟན་འགྱུར། སྒྲ་མདོ། སི་འོ་མཚོ་ཁ་བདའ་ཏིལ་ཌ་ཎཥ་་་་༈ག་མྱི་དང་སྲི་བློ་འགྲོ་བ་ལའང་པ་རསྨི་པ་ད་རྣམས་སྦྱོར་བ་ལས། །༈ལིཏྠུ་ཥཱ་དེང༌། ལོ་འགྲོ་བ་དང་པུ་ཥ་སོགས་:ལ་ངའོ། །བྱིངས་ལ་ྋཡིག་་་རྗེས་འབྲེལ་འགྲོ་བ་རྣམས་དང་པུ་ཥ་ལ་སོགས་པ་རྣམས་ལ་བྱུང་ཟིན་གྱི་རྐྱེན་སིའི་་དོན་ལ་ང་འགྱུར་ཏེ། རྐྱེན་ང་སྦྱར། ང་ཡིག་ཡོན་ཏན་འགོག་པའི་དོན་ཡིན་པས། དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ག་མཏྲ། ཨ་ག་མ་ཏཱཾ། ཨ་ག་མན། ཨ་་ག་མཨ་ག་མ་ཏཾ། ཨ་ག་མ་ཏ། ཨ་ག་མཾ། ཨ་ག་མཱ་བ། ཨ་ག་མཱ་མ། བ་ཏི༈དེ་བཞིན་དུ། སྲི་བློ་ལ། །ཨ་སྲི་པཏྲ། ཨ་སྲི་པ་ཏྭཾ། །ཨ་སྲི་པན་་་སོགས་སོ། །རྒྱུ་ཎ་པ་༈མུ་ཙཱ། གྲོལ་བ་ལ། ཨ་མུ་ཙཀྲ། ཨ་མུ་ཙཏྭཾ། ཨ་མུ་ཙན་སོགས་བolc༈བུཥཱ་རྒྱས་པ་ལ། ཨ་པུ་ཥཏྲ། ཨ་པུ་ཥ་ཏྭཾ། ཨ་པུ་ཥན་སོགས་སོ། །༈དྷཱུཏྲ་སྣང་བྱེད་ལ། ཨ་དྷུ་ཏཏྲ། ཨ་དཱུཏྲ་ཏྭཾ། ཨ་དྷུ་ཏན་སོགས་སོ། །བཛྲ་ཡོངས་སུ་བརྗོད་པ་ལ། རྐྱེན་ང་ཕ་རོལ་ལས། བཛྲའི་དབྱངས་ལས་ཀྲ་ཨཱ་ག་མ་སྟེར་བར་བཤད་པས། ་ཙ་དང་བྲའི་བར་དུ་སཀྲ། རྗེས་འབྲེལ་ཕྱིས། ས་་རྣམ་བཅད་དེ་ཨུར་བསྒྱུར་བ་སོགས་ཀྱིས། ་ཨ་པོ་ཙཏྲེ། ཨ་པོ་ཙ་ཏྭཾ། ཨ་པོ་ཙན་སོགས་སོ། །ཌ་་་་་་ང་ལ་ཕྱི་མ་དག་པ་དང་། །ཊ་ཏེ། །༈དུ་ཧ་རབ་ཏུ་འགེང་བ་ལ་ང་དང་དིཔྲ་སྦྱར་བས། ཨ་དུཧྲ། །ཨདྲ་་་ཅེས་གནས་པ་ལ། ༈ཧ་ཤ་ཏཱཾ་ཏཱཏྨཀྲ། ཧ་ཤ་ཥ་རྣམས་ཀྱི་མཐའ་ལས་ས་ཀྲའོ། །ང་དང་བཅས་པའི་དིཔྲ་སོགས་ཕ་རོལ་ལ་བྱིངས་ཧ་ཡིག་དང་། ཤ་ཡིག་དང་ཥ་་་ཡིག་རྣམས་ཀྱི་མཐའ་ཅན་ལ་སཀྲ་ཨཱ་ག་མ་སྦྱིན་པར་བྱ་སྟེ། སཀྲ་བྱིན། ཀ་གནས་དོན་ཨ་བརྗོད་དོན། ཧྲནྡྲ། །དྲདྷཱ། རྗཀྲ། ས་ཥར་བསྒྱུར་བ་སོགས་ཀྱིས་་་།
