བསྟན་འགྱུར། ཙྪཱསྒྲ་མདོ། ༼སི་ཀཾ་ཏ་བྷ་ཛཱ་ཛ་གོ་ག། བ་སུ་དྷཱ་སླ་བོ་ན་པི་ཙ་རོ་ཙ་པཉྩ་གྲ་ཧི། །གུ་རུ་ནི་ཐིག་ལེར་ཤེས་པར་བྱ། །ས་ལ་ཝ་ན་གཞན་ཡང་ར་དག་ལྔ་དང་གཟའ་ཡིས་ནོར་འཛིན་ནོ། །ལྕི་བ་ཐ་མ་དྷ་ན་མ་ནུ་ལ་གླུ་རི་ཧ་ཤ་ཤི་ཀ་ལཱ། །པྲ་གི་ཡ་མ་པི་ཙ་ར་ས་ན་བ་ཀ་བི་ར་ཡང་བ་མ་ནུས་འདིར་ནི་རི་བོང་ཅན་གྱི་ཆ། །འདིར་ནི་གཞན་ཡང་ར་དང་དགུ་ཡི་ངལ་བསོར་གྱུར་ན་མ་ཎཱ། །བ་སུ་མུ་ནི་ཡ་ཏི་རི་ཏི་མ་ཎི་གུ་ཎ་ནི་ཀ་ར། ཡ། ན་ན་ག་བ་ར་ར་ཕྲེང་བའོ། །དེ་ལྟ་བུ་ལ་ནོར་དང་ཐུབ་པར་ངལ་གསོ་ཡོན་ཏན་ནོར་བུའི་ཚོགས། །ན་ན་ག་ཝ་ར་ར་ཝ་བཱསླུ་རྨ་ལི་ནཱི་ཏཱཏྠ་ལོ་ཀི། ན་ཛ་བྷ་ཛ་རིཟི་རཱ་ཛི་ཏ་མི་དཾ་ས་མཱུ་དྲ་ཀཾ། བྷཀ་ན་རྣམས་བརྒྱད་དང་འཇིག་རྟེན་ཕྲེང་ལྡན་མ། །ན་ཛ་བྷ་ཛ་ར་རྣམས་རྣམ་པར་སྤེལ་བ་འདི་ནི་རྒྱ་མཚོ་ཅན། །སཱ་མུ་ནེཁ་བྲྀ་ཥ་བྷ་ག་ཛ་བི་ལ་སི་ཏཾ། །།ན་ཛ་བྷ་ཛ་རཱ་གུ་རུ་བྷྱ་ཡ་དི་ཏཏྟ་དཱ་ཀུ་བྷ་ར་ཀ་ན་ན་ས་ཐུབ་ལས་མཁའ་ལ་ཁྱུ་མཆོག་གླང་ཆེན་རྩེན། །གལ་ཏེ་ན་ཛ་བྷ་ཛ་ར་ལྕི་དེ་ནི་དེ་ཚེ་ལེ་སུཎྞཾ༑། ཡ་མཱུ་ནཱ་སཱ་བྷཱ་ཝཱ་ར་ས་ཧ་ར་བི་རཱ་མཱ་ཤི་ཥ་རི་ཎི། །ཛ་སཱ་ལླ་ན་ར་བརྒན་རྩི། །ཡ་མུ་ན་སུ་བྷ་ལ་ཝ་རྣམས་རོ་དང་འཕྲོག་བྱེད་ངལ་བསོ་གཙུག་ཕུད་ཅན། །ཛ་ས་ལ་ཝ་ན་ར་རཱ་ཝ་སུ་གྲ་ཧ་ཡ་ཏིསུ་པྲ་ཤྲཱི་མ་ཏཱ། །དིདྨ་ནི་ཐཾ་ཤ་པ་ཏཱ་པ་ཏི་ཏཾ་བྷ་ར་ན་བ་ར་རྣམས་ནོར་དང་གཟའ་ལ་སྡོད་པ་ཡངས་ལྡན་མ། །བྷ་ར་ན་ཝ་ན་ན་ཝ་རྣམས་ཕྱོགས་དང་ཐུབ་པར་འོད་ན་ན་བི། །ན་སུ་མ་ཝ་ཛ་ཛཱ་བཱསཱ་དཱབྷི་སྭ་རིན་རི་ཎཱི་མ་ཏཱ། །མན་ཀྲནྟཱ་མའི་འདབ་མ་ལྟུང་། །ན་སུ་མ་ཝ་ཛ་ཛ་ཝ་རྣམས་རོ་ཆུ་དབྱངས་ལ་རི་དྭགས་མོ་རུ་འདོད། །མ་ཝ་ན་སུ་ར་མ་བ་ན་སུ་ར་རཱ་བས་མུ་དྲཆུ་ལོ་ཀི། །ན་ཛ་བྷ་ཛ་ཛཱ་ལ་གོ་ཙ་ཡ་དི་ནརྐུ་ར་ཝ་རྣམས་མཚོ་དུས་འཇིག་རྟེན་དལ་གྱིས་གནོན་པ་ཡིན། །གལ་ཏེ་ན་ཛ་བྷ་ཛ་ཛ་ལ་ག་རྣམས་དེ་ཚེ་ཊ་ཀནྟུ་ཏ་དཱ། །ན་ཛ་ཁཾ་ན་བྷ་བྷ་བོ་མུ་ནི་ར་སོ་ད་དྷི་ཀོ་ཀི་ལ་ཀཾ། །མ་གཱནཱ་རོ་བུམ་པ་ཅན་དུ་འདོད། །ན་ཛ་ཝ་ན་བྷཱུ་དང་བྷ་ཝ་ཐུབ་པ་རོ་དང་ཆུ་གཏེར་ཁུ་བྱུག་ཅན། །མ་ག་བ་ན་བཀུསུ་མི་ཏ་ལ་ཏཱ་བེལླི་ཏ་ཀརྟུ་ལོ་ཀི། །།ཨི་ཧ་ན་ན་ར་ཙ་སུ་དང་ར་གཉིས་པ་རྣམས་དབང་དུས་འཇིག་རྟེན་མེ་ཏོག་གཡོ་བའི་འཁྲི་ཤིང་ངོ༌། །འདིར་ནི་ན་གཉིས་ར་།
