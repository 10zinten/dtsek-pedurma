ཤས་དག་གི་ཨའོ། །སྐྱེས་པ་ལ་སྲའི་ནའོ། །ཤས་ལའོ། །ཊཱ་ཨི་ནའོ། །བྷ་ལ་་ཨཱའོ། །བྷིའི་བྷ་ཨའོ། །དེ་ལ་ཨཀྲའོ། །བྷ་ས་མང་པོ་ཉིད་ལ་ཨེའོ། །ང་སི་ཨཏྲའོ། །ངས་སྱའོ། །ཨོས་ལའོ། །ཨཱམ་ལ་ནུཊའོམ། །ནཱམ་ལའོ། །ཀུ་ཨི་ལ་ལ་ཥ་རུ་བྱེད་དེ་སའོ། །ཀུན་ཏུ་བོད་པ་ལ་སི་དྷིའོ། །བྱིངས་མ་ཡིན་པའི་མཚུངས་པ་ལ་དྷི་དབྱིའོ། །ཛས་ཨཱིའོ། །ཥ་ར་རི་ལས་མཐའ་མ་ཡིན་པའི་ནའི་ཎའོ། །ཨ་བ་ཀུ་པུ་བར་ན་ཡོད་ཀྱང་ངོ་། །སརྦ་སོགས་ལ་སྨཊའོ། །ཨཏྲ་ལས་སོ། །ཨཱམ་ལ་སུཊའོ། །ཇི་སྨིནའོ། །མཱ་སའི་དབྱི་བ་ཡང་ནའོ། །།བྱིངས་ཀྱི་ཨཱཏྲ་དབྱིའོ། །ཨོ་ནི་ཨཱི་ཨཱུ་དག་གོ། །ཨེ་ཨོ་ཛས་ལའོ། །དྷི་ལའོ། ::། བུད་མེད་མ་ཡིན་པ་ལ་ཊཱ་ནཱའོ། །ང་འགྲོ་བ་ལའོ། །ང་སི་ངས་དག་གི་ཨའོ། །ང་ཨོའོ། །དེ་ཡང་ཌིཏྲའོ། །རིཌ་ལ་ཊིའོ། །དྷི་མ་ཡིན་པའི་སི་ཌའོ། །ས་ཁིའི་ཨིའོ། །ས་ཁི་པ་ཏི་དག་ལ་ཨཱིཀྲའོ། །ང་ལ་རྀཀྲའོ༄༅། །རཾཏྲ་ལས་ངའི་ཨུའོ། །ཏད་སོགས་ཀྱི་སི་སོགས་ལ་ཊི་ཨའོ། །ཏྲིའི་ཨ་ཡང་ངོ་། །།བྱིངས་ཀྱི་ཨཱི་ཨཱུ་དག་གི་ཨི་ཕྲ་ཨུ་བྲ་དག་དབྱངས་ལའོ། །ཡ་ལྔ་དག་ཡང་ནའོ། །ངའི་ཨཱམའོ། །སིའི་ཨཱའོ། །ལྔ་ལ་ཨརྱའོ། །དྷིའི་ཨརྱའོ། །།ཇི་ལའོ། །ལྕིའི་ཨཱརྱའོ། །རིའི་ས་དང་བྷ་ལའོ། །།ཨོ་ཨོའོ། །ཨམ་ཤས་ལའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་སྐྱེས་པའི་རྟགས་དབྱངས་ཀྱི་མཐའ་ཅན་རྣམས་རྫོགས་སོ། །འོ་ཁྲ་༎་ཏ་ལ་ནས་ས་འཊོ་ཊཀ་ལ་ཨFམི་ཡིན་ནོ། ་མ། ཉིས་ཆོ་གྲོ་བ་ལྟ་ཏཆ་ཁ་ག་བུད་མེད་ལ་ཨ་ལ་ཨཱཛ྄འོ། །ཨཱཔྲ་ལས་སོ། །ཨོཾ་ཨཱིའོ། །དྷི་ཨིའོ། །ཊཱ་ཨོས་དག་ལ་ཨེའོ། །།རིཏྲ་རྣམས་ཀྱི་ཡཊའོ། །ཡཊ་ལས་ཨཏྲ་ཡང་ངོ༌། །ཨིཏྲ་ཨུཏྲ་དག་ལས་སོ། །།བུད་མེད་ཀྱི་ཨི་ཨུ་དག་ལའོ། །ཧ་ས་ཨཱིཔྲ་ལ་སི་དབྱིའོ། །དྷི་ལ་ཐུང་ངུའོ། །།ངཏྲ་རྣམས་ཀྱི་:སྤྲཊའོ། །ཎི་བྷཱའི་འོ། །ཨམ་ཤས་དག་ལ་།
