ཁ་ས་ཁ་ཤོག་ཁ་ད་བ་ས་རསཱུ་ཏྲ་བེབྦ་ནོ། །པྲ་ཛ་ནས་དུ་པ་ས་རཔྲ་ས་རཿཔྲ་ཎ་ཡས་མོ། །དྷི་ནི་སྐབས་ཡིན་ནོ། །ཐགས་སྤུན་སོན་པ་སྣལ་མ་དཀྲི། །ཤེ་སྦྱོར་འཁྲིག་དང་ཚོགས་པ་མཚུངས། །བློ་ཤཏྟིནིཤྩ་མོཉལྡི་ཏུ་སཾ་ཀྲ་མོ་དུརྒ་སཉྩ་ར། །པྲ་ཏུཏྠ་མ༔པྲ་ཡོ་གཱརྦཔྲ་ཀྲ་ནུས་དང་ནི་ཆུ་སྲོལ་ལོ། །མོ་མིན་ཟམ་པ་བགྲོད་དཀའ་ཟམ། །འཁྲུག་ཕྱིར་འཇུག་དང་རབ་སྦྱོར་དོན། །མས་དུ་པྲ་ཀྲ་མ། །རྩྭ་ད་ས་དཱ་ན་མུ་ཏ་ཨཱ་རཾ་བྷསམྦྷ་མསཱུ་རཱ། །པྲ་ཏི་བནྡྷཁོམ་པ་དང་ནི་མགོ་འཇུ་བ། །འཇུ་བ་རྩོམ་པ་ཨཱ་རཾ་བྷ། །ཏམ་ཏམ་ཏུར་ཏུར་བར་ཆད་གེགས། །པྲ་ཏིཋནྡྷོཉབ་ནཱ་ཡསྐུ་ནི་པཱ་ཏ་ནཾ། །ཨུ་པ་ལནྡྷས་ནུ་བྷ་བས་མཱ་ལནྡྷོ་བེ་ལེ་པ་ནཾ། །འོག་མིན་འོག་ཏུ་ལྟ་བའོ། །སྔོན་ཤེས་སླར་ཡང་འཕེལ་བའོ། །བསྐུ་བྱུག་དང་ནི་བྱུག་པའོ། །བི་པྲ་ལམྦྷོ་བི་པྲ་ཡོ་གོ་བི་མམྦྷསྦ་ཏི་སཛྫ་ནཾ། །བི་ཤཱ་བསྟུ་པྲ་བི་ཁྱཱ་ཏི་ར་བེ་ཀླུ་རྗེས་ཆགས་དང་ནི་སྦྱོར་བའོ། །སྦྱིན་སྟེར་རྣམ་གྲགས་རབ་ཏུ་གྲགས། །མེལ་ཙེ་བ་དང་སོ་པའོ། །ངེས་ཀློག་པྲ་ཏི་ཛཱ་ག་ར། །ནི་པཱ་ཋ་ནི་པ་ཊོ་པཱ་ཟེ་ཏེ་མ་སྟེ་མོ་ས་མུནྡ་ནེ། །ཨཱ་དཱི་ན་བཱ་ཤ་བོ་ཀླུ་ངེས་བརྗོད་འདོན་པའོ། །བངས་དང་གཤེར་བངས་རློན་པའོ། །།ཀུན་ནས་དམན་དང་ཟག་ཉོན་ཤེ་མེ་ལ་ཀེ་སཾ་ག་སཾ་ག་མོ། །སཾ་བཱི་ཀྵ་ཎ་བི་ཙ་ཡ་ནཾ་མཱངྒཱ་ཎཾ་མཾ་ག་ཎཱ་མྲ་ག། །པ་རི་ར་མོངས། །ཚོགས་པ་འདུས་པ་འཕུངས་པའོ། །།ལྟ་བ་འཚོལ་བ་མྱུལ་འཚོལ་ལྟ། །།ཡོངས་འཁྱུད་མྦྷཔ་རི་ཧཱུྃ་གསཾ་གླེ་ཥ་ཨུ་པ་ཧཱུཾ་ཧ་ནཾ། །ནིརྦཎྜ'ིནཾ་ཏུ་ནཾ་དྷཱ་ནཾ་དན་ནཱ་ལོ་ཀ་ནི་ཀླུ་ཎཾ། །ཡོངས་འབྱར་ཡང་དག་རེག། ཉེར་འཛིན་ཉེར་ལྟ་རྟོག་དང་གཟིར། །སྣང་བར་བྱེད་དང་ལྟ་བའོ། །པྲ་ཏཱ་ལྱཱ་ནཾ་ནི་ར་ས་ནཾ་པྲ་ཏྱཱ་དེ་ཤོ་ནི་རཱ་ཀྲི་ཏི། །ཨུ་པ་སཱ་ཡོ་བི་ཤྭ་ཡ༔་པནཱ་ཡ་ཕྱིར་རྒོལ་ཟློག་པ་འགོག་དང་འགེགས། བན་པ། རྣམ་གྲངས་མིང་གཞན་མིང་རིམ་ཤ་ཡ་ནཱརྦ་ཀོ། །ཨཎྜ་ནཾ་ཙ་ཛྷ་ཏཱི་ཡཱ་ཙ་ཧྲི་ཎཱི་ཡཱ་ཙ་ཧྥེ་ཏོརྦ་ཀེ། །སཱརྱ་ཏཱ་སོ་བི་པཎཱ་སོ་བྱ་དོན། །སྐྱུག་བྲོ་བཙོག་པ་སྐྱུག་བྲོ་དོན། སྤྱོད་པ་འཚོགས་ན་། བཟློག་པ་ཕྱིར་བཟློག་རྣམ་བཟློག་་་་ཏྱཱ་ཡཐུ་བི་པདྨ་ཡེ། །པརླ་ཡོཉཏི་ཀྲ་མལྟ་སྨིནྣ་ཏི་པཱ་ཏ་ཨུ་པ་ཏྱཱ་ཡ། །སྲེ་ཥ་ཎ་ཡཏྠ་མཱ་ལོག། འགོངས་པ་འཕྱངས་པ་རྒལ་དང་འཕར། །སྲེ་ཥ་ཎ་ནི་གང་བོས་ནས། །དེ་ལ་—།
