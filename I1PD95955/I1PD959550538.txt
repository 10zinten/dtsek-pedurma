ཏི། །ན་དཱི་ན་གརློནཱ་ག་ཏཱཾ་བྷོ་ག་བ་ཏྱ་ཐ་སཾ་ག་རེ། །སཾ་གེ་ས་བྷཱ་ཡཱ་ས་མི་ཏིཚོགས། །གཙང་པོ་ཀླུ་ཡི་གྲོང་ཁྱེར་རྣམས། །བྷོ་ག་བ་ཏི་དེ་ནས་གཡུལ། །གྲོགས་དང་འཕོང་ས་ས་མི་ཏི། །གྷ་ཡ་བཱ་སཱ་བ་པི་ཀྱི་ཏི། །ར་པེ་རརྩིཉྩ་ཤསྔཉྩ་བཙཱི་ཛཱ་ལཱ་ཙ་ཧེ་ཏ་ཡ། །ཛ་ག་ཏཱི་ཟད་དང་གནས་ལ་ཀྲི་ཏིའོ། །ཉི་མའི་འོད་ཟེར་མཚོན་ཆ་དང་། །མེ་ལྕེ་འབར་ལ་ཧེ་ཏིའོ། །ཛ་ག་ཏཱི་ཛ་ག་ཏི་ཙྪནྡོ་བི་ཤེ་ཥ་པི་ཀྲི་ཏཱ་བ་པི། ་། པཾཀྟི་སྨནྡོ་པི་ད་ཤ་མཾ་སཱུཏྤ་བྷཱ་ནི་འགྲོ་བ་དང༌། །སྡེབ་སྦྱོར་བྱེ་བྲག་ས་གཞི་ལའང་། །རྀཏྟི་གྲལ་དང་ཡིག་བཅུའི་སྡེབ། །འོད་དང༌བེ་པི་ཙཱ་ཡ་ཏི། །པཏྟིརྒ་ཏོ་ཙ་མུ་ལེ་པི་བཀྵ་ཏིཔཀྵ་བྷེ་ད་ཡོ། །པྲ་ཀྲྀ་ཏིརྨོ་ནི་ཆགས་པ་ཨཱ་ཡ་ཏི། །བཏྟི་འགྲོ་དང་རྐང་ཐང་ངོ༌། །རྩ་བ་གཤོག་དབྱེ་པཀྵ་ཏི། ་། པྲ་ཀྲྀ་ཏི་ནི་སྐྱེ་ལྀ་གེ་ཙ་ཀོ་ཤི་ཀཱ་ཏྱཱཉྩ་བྲྀཏྟ་ཡ། །སི་ཀ་ཏཱསྩརྭ་ལུ་ཀཱ་པི་བེ་དེ་ཐ་བ་སི་ཙ་ཐུ་ཏི། །གནས་རྟགས། ཎ། གར་གྱིས་འཚོ་བ་བྲྀཏྟིའོ། །།སི་ཀ་ཏཱ་ནི་བྱེ་གྱོ་དུམ། །རིག་བྱེད་ཐོས་པ་ཐུ་ཏིའོ། །བ་ནི་ཏཱ་ཛ་ནི་ཏཱ་ཏྱརྦ་ནུ་རཱ་གཱ་ཡཾ་ཙ་ཡོ་ཥི་ཏི། ་། གུཔཱིཀྵི་ཏི་བྱུ་དཱ་སེ་པི་:ཛཱ་ཏིནྡྷ་བ་ནི་ཏཱ་དང་ཛ་ནི་ཏ། །མ་དང་རྗེས་ཆགས་བུད་མེད་ལ། །གུསྟི་དོང་འབྲུ་བུག་བཅས་གྲུ། །དྡོ་ཏི་ར་ཎ་དྷེརྨ་ཡོ། །པྲ་ཧ་ཏཱི་ཀླུ་དྲ་བཱཏྟ་ཀཱི་ཙྪནྡོ་བྷེ་དེ་མ་ཧ་ཏུ་པི། །བཱ་སི་འཛིན་དང་བློ་གྲོས་སོ། །བྲི་ཧ་ཏཱི་ནི་བུམ་སྨན་དང་། །སྡེབ་སྦྱོར་དབྱེ་བ་ཆེན་པོ་ལའང་། །བུད་མེད་ཏཱཙྪི་ཀ་ར་ཙྩོཉྩ་བཱཏྟཱ་བྲྀཏྟོ་ཛ་ན་ཐུ་ཏོ། །བཱནྟཾ་ཕལྷུ་ནྱ་རོ་གེ་ཏུ་ཏྲི་པཱུ་ཙ་གླང་མོ་བཱ་སི་ཏཱ། །བཱནྟཱ་འཚོ་བ་སྐྱེ་བོའི་གཏམ། །བཱཏྟཾ་ནད་མེད་སྙིང་པོ་མེད། །ཞུན་མར་མར་གྷཾ་ཏཱ་མྲྀ་ཏེ། །ཀ་ལ་དྷོ་ཏཾ་རཱུ་པྱ་ཧེ་སྣོནི་མིཏྟཾ་ཧེ་ཏུ་ལཀྵ་ཎོ། །ཐུ་ཏཾ་ཤཱསྨཱ་ནི་ཨ་མྲྀ་ཏ། །ཀ་ལ་གྷོ་ཏཾ་དངུལ་དང་གསེར། །ནི་མིཏྟ་ནི་མཚན་ཉིད་རྒྱུ། །ཐུ་ཏཾ་བསྟན་བཅོས་བ་དྷཾ་ཏེ་ཡོཆུ་ག་པནཔ་ཡོཀྲྀ་ཏཾ། །ཨ་ཏྭཱ་ཧི་ཧཾ་མ་ཧཱ་བྷཱི་ཏིཀརྨ་ཐོས་པ་འཛིན། །ཀྲྀ་ཏཾ་རྫོགས་ལྡན་ཡོངས་སུ་ཐོབ། །ཨ་ཏྭཱ་ཧི་ཏཾ་འཇིགས་ཆེན་དང༌། །འཚོ་བ་ཛཱི་བཱ་ན་པེ་ྋཀྱི་ཙ། །ཡུལྟེཀྵ་དཱ་ཤྲཱི་ཏེ་བྷཱུ་ཏཾ་པྲ་ཏྱཱ་ཏཱི་ཏེ་ས་མེ་ཏྲི་ཏུ། །བྲྀཏྟཾ་ལ་ནི་ལྟོས་པའི་ལས། །རིགས་པས་སོགས་དུས་བྷཱུ་ཏཾ། །སྲོག་ཆགས་འདས་དུས་འདྲ་བ་གསུམ། །བྲིཏྟཾ་—།
