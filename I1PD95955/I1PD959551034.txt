སྲིན་བུ་མེ་ཁྱེར་ལ། ་ཁ་རྒྱ་སྟེ། མཁའ་སྲུང་དང་། ཛ་ཏི། དཀར་འོད་དང༌། །༼ཨིང་ག་ཎ། །ནམ་མཁའི་འོད་རྣམས་སོ། །ད་ལྟ་། །ལས་ཀྱི་བཀའ་ཁཝཱ། བུང་བའི་མིང་ལ་ཚིགས་སུ་བཅད་པ་གཅིག་སྟེ། མ་དྷུ་བྲཏ། སྦྲང་རྩིའི་བརྟུལ་ཞུགས་དང་། མ་དྷུ་ཀ་ར། །སྦྲང་རྩིའི་བྱེད་པ་དང༌། མ་དྷུ་ལེགས་་་་སྦྲང་རྩི་མྱང་དང་། ་མ་དྷུ་ལི་ཏ། །སྦྲང་རྩི་སྐྱོང་དང་། །དྷཱི་རེ་ཕ། །ར་ཡིག་་་གཉིས་པ་སྟེ་འདིའི་མིང་སྦྲ་མ་ར་ལ་ར་ཡིག་གཉིས་པས་སོ། །སཱུཏྤ་ལི་ཊ། མེ་་ཏོག་མྱང་དང་། བྷིངྒ། ་བུར་བ་དང༌། བྷཊཱཏ། ྋརྐང་དྲུག་པ་དང༌། སྦྲ་མ་དང་འཁོར་ལོ་དང༌། ཨ་ལི་སྦྲང་བུ་རྣམས་སོ། །པ་ཚོ། ང་ཚོ་ལ་ཙུས་དེ་ནས་རྨ་བྱའི་མིང་ལ་ཚིགས་སུ་བཅད་པ་གཅིག་སྟེ།  ོ་མ་ཡུ་ར། ས་་་་འགྲོ་དང༌། བརྩི་ཎ། མདོང་ཅན་དང༌། བརྩི་མང་པོས་ལྡན་དང༌། །ནཱི་ལ་ཀཎྜ་མགྲིན་སྔོན་དང༌། བྷུ་ཛངྒ་བྷུག་ལག་འགྲོ་ཟ་དང༌། ཤི་ཁམ། གཙུག་་་ཕུད་ཅན་དང་། །ཤི་ཁཱི་ཙེ་མོ་ཅན་དང་ཀེ་ཀཱི། ༼སྒྲ་ཅན་དང་། །མ་གྷ་ནཱ་དཱ་བུ་ལ་སི། འབྲུག་སྒྲའི་རྗེས་སུ་སྒེག་པ་རྣམས་སོ། །རྨ་བྱའི་སྒྲ་སྐད་ལ་ཀི་ཀཱི་ཞེས་བྱ་ཞིང༌། འཇུག་མ་ལ་གནས་པ་མིག་གི་དབྱིབས་ལྟ་བུའི་ཟླ་ཁམ་ནི། མེ་ཙག་ཏེ་མདོངས་ཞེས་བྱ་ལ། མེ་ཙ་ཀ་ནག་པོ་ལ་ཡང་སྔར་བཤད་དོ། །༄༅། །རྗེས་མ། :ཀ་དེ་ནས་ཚིགས་སུ་བཅད་པ་ཕྱེད་དང་གསུམ་གྱིས་བྱ་སྤྱིའི་མིང་སྟེ། །ཁ་ག། མཁའ་འགྲོའོ། །བི་ཧངྒ། བི་ཧད། བི་ཧཱ་ཡ་ལ། ནམ་མཁའ་འགྲོའའམ། མཁའ་རྒྱུའམ། མཁའ་སྤྱོད་སོགས་སོ། །ཤ་ཀུནྟི་བྱའོ། །པཀྵི། འདབ་་་ཆགས་སོ། །ཤ་ཀུ་ནི། །ཤ་ཀུ་ན། །ཤ་ཀུན། རྣམས་ཀྱང་བྱ་འབུས་ལྡན་ནོ། །དྲི་ཛ། གཉིས་སྐྱེས་སོ། །པ་དྲ་ཏི། །འདབ་ལྡན་ནོ། །པ་དྲི། /སྒྲོ་ཅན་ནོ། །པ་ཏག་། འཕུར་བྱེད་དོ། །པ་ས་ཏཏ། །འཕུར་འགྲོ། པ་དྲ་རཱ་ཐ། །
