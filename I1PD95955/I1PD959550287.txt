ཀ་ཀླུ་ཏ་ཎཡབསྟན་འགྱུར། སྒྲ་མདོ། ༼སེ༽འོ་ཁཚོ་ཀཚོ་བ་གཏིད་ཀཚོ་ཁྲITOUTབོ་ལུང༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ས་མ་ཛ་གྷ་ཊྛིཏྲ། ་ས་མ་ཛཱ་གྷཊ། ས་མ་ཛཱ་གྷཊཾ། ས་མ་ཛཱ་གྷ་ཊུ། ས་མ་ཛཱ་གྷཊཱི། ས་མ་ཛཱ་གྷཊ། ས་མ་ཛཱ་གྷཊཾ། ས་མ་ཛཱ་གྷཊྚ། ས་མ་ཛཱ་གྷ་ཊཾ། ་ས་མ་ཛཱ་གྷཊ། ས་མ་ཛཱ་གྷརྨ་སོགས་ཏེ། ཨ་ཏི་ཤ་ཡེ་ན་གྷ་ཊ་ཏེ་ཕུལ་དུ་བྱུང་བའི་ཞེས་པའི་དོན་ལ། སཾ་ཛཱ་གྷ་ཊཱི་ཏི། རྩེད་མོ་བྱེད། །ལས། ༈ཉེར་བསྒྱུར་སུཾ་སྔོན་མའི་བྷཱུ་ཡོད་པ་ལས་ཀྱང་བརྟ་མ་ན་ཁྲཊ་ལ། སཾ་བོ་བྷ་བཱི་ཏི། སཾ་བོ་བྷོ་ཏི། སཾ་བོ་དྷཱ་ཏཿསཾ་བོ་བྷ་བ་ཏི། སཾ་བོ་བྷ་བཱི་ཏི། །སཾ་བོ་བྷོ་ཏི། སཾ་བོ་བྷུ་ཐསཾ་བོ་བྷྱཱ་ཐ། སཾ་བོ་བྷ་བཱི་མི། སཾ་བོ་བྷོ་མི། སཾ་བོ་བྷཱ་བསཾ་བོ་བྷ་མ། ་བ་ལེ་.༈ཆོག་་འབྱུང་ལིང་ལ། སཾ་བོ་བྷཱུ་ཡཱ་ཏཱ། སཾ་བོ་བྷཱུ་ཡཱ་ཏཱ། སཾ་བོ་བྷཱུ་ཡུསོགས་སོ། །ཀ་ཁ་ག་གེ་འབར་བ་མ་ས་འོ། ༈ཤིས་བྱེད་ལོཊ་ལ། ་་སཾ་བོ་བྷ་བཱི་ཏུ། སཾ་བོ་བྷོ་ཏུ། སཾ་བོ་བྷཱུ་ཏཱཏྲ། སཾ་བོ་དྷུ་*ཏྭཾ། །སཾ་བོ་བླུ་བ་ཏུ། །སཾ་བོ་དྷཱ་ཧི། སཾ་བོ་བྷཱུ་ཏཱཏྲ། སཾ་བོ་བྷུ་ཏཾ། :：སཾ་བོ་བྷཱུ་ཏི། སཾ་བོ་བྷ་བཱ་ནི། སཾ་བོ་བྷ་བཱ་བ། སཾ་བོ་བྷ་བཱ་མ། ་་འཚོ་ལ། པ་འ། །༈དེ་རིང་མ་ཡིན་པ་ལང་ལ། ས་མ་བོ་བྷ་བཱིཏྲ། ?ས་མ་བོ་བྷོཏྲ། ་ས་མ་བོ་བྷཱུ་ཏྭཾ། ས་མ་བོ་བྷ་བུ། ས་མ་བོ་བྷ་བཱིས་མ་བོ་བྷོས་མ་བོ་བྷཱུ་ཏཾ། ས་མ་བོ་བྷཱུ་ཏ། །་ས་མ་བོ་བྷ་ཐཾ། ས་མ་བོ་བྷཱ་བ། ས་མ་བོ་བྷུ་མ། ཝཱ་ཎཱ། །ཀྱི་དུ་༈རྔུ་ཀྲྀཏྲ་བྱེད་པ་ལ་ཡང་། གཉིས་བརྗོད། ཁྲི་ཀྲྀ་ཡ་ཞེས་གནས་པ་ལ། ཡང་འཇིག། བྱིངས་རཾ་ཡིག་མཐའ་ཅན་དང་རི་ཡིག་ཨུ་པ་དྷཱ་ཅན་ལ་ཡང་་་འཇིག་པ་ལས་གཉིས་བརྗོད་པའི་དང་པོ་དེ་ལ། ་རུཀྲ་ཨཱ་ག་མ་དང་རིཀྲ་ཨཱ་ག་་་མ་དང་རཱི་ཨཱ་ྋག་མ་གསུམ་པོ་གང་རུང་ཞིག་ྋསྟེར་བར་བཤད་པས། ཨཱ་ག་མ་་་གསུམ་པོ་བྱིན། ཀྲ་གནས་དོན། རུའི་ཨུ་བརྗོད་དོན་ཡིན་པའི་ཕྱིས། །ཀུ་ཧ་—།
