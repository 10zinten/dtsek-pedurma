བསྟན་འགྱུར། སྒྲ་མདོ། སི་བ་སྦྱར་སྤོས་སོ། །ཏུ་རུ་ཀྵ་དང་པི་ཤུ་ཀ་། པི་ལ་ཡཱ་བ་ན་ཡང་ངོ༌། །།།དེ་ནས་པཱ་ཡ་ས་དང་ནི། །དཔལ་གོས་བྲི་ཀ་རཱུ་པ་དང༌། །བཱི་ཡི་ཌི་དང་ཐང་ཆུའོ། །རི་དགས་ལྟེ་བ་རི་དགས་ཚང་། །གླ་བ་དེ་ནས་ཀོ་ལ་ཀ། ཀ་ཀྐོ་ལ་ཀ་སྦུབས་ཀྱི་་་འབྲས། །།དེ་ནས་ག་པུར་མོ་མིན་ནོ། །གྷ་ནའི་སྙིང་པོ་ཟླ་བའི་མིན། །འོད་དཀར་ཁ་བའི་བྱེ་མའོ།  ོ། དྲི་ཡི་སྙིང་པོ་མལ་ཡ་སྐྱེས་དང་། ལ་། དཔལ་བཟང་ཙནྡན་མོ་མིན་ནོ། །ཏིལ་པརྩི་ཙནྡན་དང་། །བ་ལང་མགོ་དང་ཧ་རི་ཙནྡན་་དང༌། །མོ་མིན་ནོ། །ཏཱིལ་ལནྡྲི་པ་ཊཾ་ག། ཆོས་བྱེད་ཙནྡན་དམར་པོ་དང༌། །ཙནྡན་ངན་པ་དེ་ནས་ནི། །།ཛཱ་ཏི་སྦུབས་ཅན་ཛཱ་ཏཱི་འབྲས་བུ་མཉམ། །ག་པུར་ཨ་ག་རུ་དང་གླ་བ་དང་། »། ཀཀྐོ་ལ་རྣམས་ལ་མཆོད་སྦྱིན་བྱུག། ལུས་ཀྱི་བྱུག་པ་བརྟིའོ༑། བཎ་ཀཾ་ནི་ཁ་དོག་བྱུག། ཕྱེ་མ་གོས་དང་སྦྱར་བ་དང༌། །དྲི་འཛིན་གོས་ནི་གསུམ་ཡིན་ནོ། །དྲི་དང་ཕྲེང་སོགས་ལེགས་སྦྱར་གོས། །ལྷག་པའི་གོས་དང་མེ་ཏོག་ཕྲེང་། །མེ་ཏོག་ཕྲེང་བ་སྤྱི་བོའི་སྟེ། །སྒྲ་ཡི་དབུས་སུ་གནྡྷ་ཨིཀ། སྟེང་དུ་འཕྲང་དང་རྩེ་མོང་འཕྱང་། །མདུན་:དུ་ལ་ལཱ་མ་ཀཾ་ངོ༌། །རབ་འཕྱང་དྲང་པོར་མགུལ་དུ་འཕྱང༌། །།མེ་ཏོག་ག་ཤལ་སེ་རལ་ཁ། །གང་ཞིག་འཕྲེང་་་བཅིངས་བྲང་ལའོ། །ཤཱི་ཁཱ་སཱ་ནཱི་སྤྱི་བོར་བཞག། བཀོད་པ་དང་ནི་ཆས་ཙོམ་པ། །ཨཱ་བྷོ་ག་ནི་ཡོངས་སུ་རྫོགས། །ཨུ་པ་དྷཱ་ནཾ་རྒྱས་ཡིན་ནོ། །ཉལ་བ་གཟིམས་པ་་ཤ་ཡི་ནཾ། །ཁྲི་དང་ཉལ་ཁྲི་གཟིམས་ཁྲི་དང་། །ཁ་གསུམ་ཡཱ་ནི་མཚུངས་པའོ། །སྡེར་འབོལ་ཆུང་དུ་ཀརྫུ་ཀོ། :། མར་མེ་དང་རབ་འབར་བ་དང༌། །པི་ཐཱ་དང་ནི་སྟན་ཡིན་ནོ། །ཟ་མ་ཏོག་དང་གའུ་དང་། །པྲ་ཏི་གྲ་ཧ་ཆིལ་མའི་སྣོད། །།།སྐྲ་ཤང་དང་ནི་སོ་མངས་དང་། །མེ་ལོང་མ་གུ་ར་དང་དཔེ། །རླུང་ཡབ་བསིལ་ཡབ་ཅེས་པའོ། །མིའི་སྡེ་ཚན་ནོ། །།།་། ཡིག་ཚ་བ་ག་ཞུན། ཚོ—།
