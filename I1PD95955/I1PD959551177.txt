བསྟན་འགྱུར། སྒྲ་མདོ། ༼སི་་ཧི་ཏཱཤྩ་ཝརྞ་ཡ་ཐི་བ་ཏཱ་བནྟ་ཨི་ཧ་ལ་གྷ་བ། །གུ་རུ་ལ་གླུ་པིཊོ་ཡཱ་བཱརཱ་ཏཱ་གུ་རུ་བི་རྣམས། །དེ་རྣམས་ཇི་སྙེད་འདིར་ཡང་བ། །ལྕི་ཡང་སྡོམ་པ་ཇི་སྙེད་དེ། །ཕྱི་མོ་ལྕི་ར་ཧི་ཏཱཉྩ་ཡཱ་བཏྠ། །མཱ་ཏཱ་ལ་གླུ་པིཎྜ་ད་ལོ་ཡཱ་བཏྟཱ་བནྟ་ཨི་ཧ་ཝརྞ། །ལ་གྷ་བོ་བྲལ་ཇི་སྙེད་དེ། །ཇི་སྙེད་ཕྱི་མོ་ཡང་སྡོམ་འདག། དེ་སྙེད་འདིར་ནི་ཡི་གེ་རྣམས། །ལྕི་བ་ཉིས་དཱི་གུ་ཎ་གུ་རུ་ཡུ་ཏཱ་དཱི་གུ་ཎཱ་ཝ་ཌཱ་བི་བརྫི་ཏ་ལ་གྷུ་བྷི། །གུ་རུ་མི་ཤི་ཏཱ་༔་ཝཎཱ་ཡ་འགྱུར་ཡང་བ་བསྲེ། །ཡི་གེ་ཉིས་འགྱུར་ཡང་རྣམས་དོར། །ལྕི་བ་བསྲེས་པའི་ཡི་གེ་རྣམས། །དེ་ཐི་བ་ཏཱ་བ་རྱ་ཨི་ཧ་མཱ་ཏཱ། །བུ་ཅུ་མེ་ཛཱ་ཡ་ཏ་ཡམྨེ་ཛི་ན་སྱ་བི་དྱ་མི་ནཱཾ་པྲ་རྣམས་ཇི་སྙེད་འདིར་ཕྱི་མོ། །རྒྱལ་བའི་རིག་པ་འདི་ལ་བདག་དད་པའི། །བསོད་ནམས་གང་ཞིག་བདག་ལ་སཱ་ད་ཡ་ཏ། །ཏཏྤ་ར་མཉྫིཔ་ར་མཾ་ཛ་ག་ཏོ་བི་དྷཾ་པྲ་སཱ་ད་ཡནྟུ། །ཤུ་བྷ་རབ་སྐྱེས་པ། །དེ་ཡིས་འགྲོ་ཀུན་དམ་པའི་དོན་དང་ནི། །རིག་པ་མཆོག་ལ་རབ་ཏུ་དད་གྱུར་ཅིག། དགེ་བར་མསྡུ་སརྦ་ཛ་ག་ཏྭཾ། །ཀྲི་ཏི་རི་ཡཾ་ཀ་ལི་ཀཱ་ལ་སརྦ་ཛྙཱ་རསྨཱ་ཀ་ར་ཤཱནྟི་་་་གྱུར་ཅིག་འགྲོ་བ་ཐམས་ཅད་ལའོ། །འདི་ནི་རྩོད་པའི་དུས་ཀྱི་ཐམས་ཅད་མཁྱེན་པ་རིན་ཆེན་འབྱུང་པཱ་དཱ་ནཾ། །ལི་ཁི་ཏ་མི་དཾ་ཤཱཀྱ་བྷིཀྵ་ལོ་ཀ་ཙཀྵ་གྲ་རཏྣ་་་་་གནས་ཞི་བའི་ཞལ་སྔ་ནས་མཛད་པའོ། །འདི་ནི་ཤཱཀྱའི་དགེ་སློང་འཇིག་རྟེན་མིག་མཆོག་རིན་ཆེན་དྷརྨ་པཱ་ལ་བྷ་དྲེ་ཎ། །།ཆོས་སྐྱོང་བཟང་པོས་བྲིས་པའོ། །སླད་ཀྱི་ས་གསུམ་ན་འགྲན་ཟླ་བྲལ་བའི་ཐམས་ཅད་མཁྱེན་པ་ཆེན་པོ་དོན་གྱི་སླད་དུ་མཚན་སྨོས་པ་བྱང་ཆུབ་རྩེ་མོའི་་་ཞལ་སྔ་ནས་བསྒྱུར་བ་ལས། སླར་ཡང་དེ་ཉིད་ཀྱི་བཀའི་རྗེས་སུ་གནང་བ་ལ་བརྟེན་ནས། བླ་མ་དེ་ཉིད་དང་སྐྱེས་བུ་དམ་པ་རྣམས་ཀྱི་བཀའ་དྲིན་གྱིས་བསྐྱངས་པའི་དགེ་སློང་ནམ་མཁའ་བཟང་པོས་བསྒྱུར་བ་ལས། སླར་ཡང་ཞ་ལུ་ལོ་ཙཱ་བ་ཆོས་་སྐྱོང་བཟང་པོས་དག་པར་བཅོས་པའོ། །འདིས་ཀྱང་སངས་རྒྱས་ཀྱི་བསྟན་པ་རིན་པོ་ཆེ་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཕན་ཐོགས་པར་གྱུར་ཅིག། །ཁ་ལོ་བ་ཚོའི་ཡིད་སྐྱོའི། །ཡ་མཚོ་ཁ། ཞེ་ལ་བཞི་ནསཟླ་བ་བ་བ་བླ་མ་སོང་ན་ཁ་ལོ་རྣའི་ཀྱངཔ་བསག་འད། །ག་ང་བ་མ་འ། །བ་ཡིན། ག་བ་ཡོག་ཁྲི་བཅུ་པ་དང་། —།
