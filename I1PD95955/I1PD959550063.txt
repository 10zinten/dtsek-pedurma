་སྐུn:::ཀ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼སེའིཡོ་ག་ལ་བློ་ཡུལ་:ཀ། དེ་ནས་གསལ་བྱེད་ཀྱི་བྱ་བ་རྣམས་བརྗོད་པར་བྱ་སྟེ། ཙ་པཱ་ཨ་བེ་ཛ་བཱཙ་པའི་ཨ་པ་ལ་ཛ་བའོ། །།ཚུ་རོལ་ཚིག་མཐར་གནས་པའི་ཙ་པ་རྣམས་ནི་ཨ་བ་་ཕ་རོལ་ལ་གྲངས་སམ་སྡེ་པ་གཅིག་པའི་ཛ་བ་རྣམས་སུ་འགྱུར་ཏེ། །་ཥཊ་ཨ་ཏྲཿཥ་ཌ་ཏྲཿདྲུག་འདིར། །བཱཀ་ཡ་ཐཱ། བཱ་གྱ་ཐཱ། ཇི་ལྟ་བའི་ངག་སྟེ་བདེན་པའི་ཚིག། ཉ་མེ་ཉ་མཱ་བཱ། ཉ་མ་ལ་ཉ་མ་ཡང་ནའོ། །ཚིག་མཐར་གནས་པའི་ཙ་པ་རྣམས་་ནི་ཉ་མ་ཕ་རོལ་ལ་གྲངས་སམ་སྡེ་པ་གཅིག་པའི་ཉ་མ་རྣམས་སུ་ཡང་ན་འགྱུར་ཏེ། བཱཀྲ་མ་ཏྲ། བཱདྨ་ཏྲ། ཚིག་ཙམ་ཞེས་པ་སྟེ་དོན་ལ་མི་གནས་པ། བཱཏྨ་ཏྲ། ཥ་ཌི། མ་མ། ཥརྨ་མ། མ་དྲུག་པ་སྟེ་གཞོན་ནུ་གདོང་དྲུག་གོ། །ཥཟླ་མ། ཙ་བཱ་ཚ་ཤ། ཙ་་པ་ལས་ཤ་ནི་ཚའོ། །ཙ་པ་སྔོན་མ་ལས་ཤ་ཡིག་ནི་ཆ་ཡིག་ཏུ་ཡང་ན་འགྱུར་ཏེ། བཱཀྲི༑་ཤུ་རཿབཱཉྩ་རཿཡདཔའ་བོའི་ངག་སྟེ་རྫུན་གཡོ་ཟོལ་མེད་པའི་ངག་གོ། བཱཀཱུ་ར་ཧོ་ཛྷ་བྷཱཧའི་ཛྷ་བྷ་རྣམས་སོ། །ཙ་པ་སྔོན་མ་ལས་ཧ་ཡིག་ནི་ཛྷ་བྷ་རྣམས་སུ་ཡང་ན་འགྱུར་ལ། དེ་ཡང་སྔོན་མ་དེའི་སྡེ་པའི་བཞི་པ་གང་ཡིན་པ་དེར་་འགྱུར་ཏེ། ་ཏཏྲ། ཧ་བིཥྛཏདྡྷ་བིདེ་སྲེག་བླུགས། ཏདྷ་བིབཱཀྲ་ཧ་རིབསྒྱུ་རི…སྨྲ་བའི་སེང་གེའམ་འཕྲོག་བྱེད་ཀྱི་ངག་གམ་སེང་གེའི་སྐད། གྷ་རིཨཙཱ། ཧ་ལོ། ཨ་ཛྫཱ་ལོ། དབྱངས་གསལ་དག། ཨཛྷ་ལོ། ཥཊ་ཧ་ལཱ་ནི། ཥཌྷ་ལཱ་ནི། རོལ་དྲུག་རྨོས་སོ། །ཡཥ་ཌྷ་ལཱ་ནི། བྲྀཏྟཱ་པྲ། ཧུ་ཏཾ། ཀྲྀཥྛ་བྷཱུ་ཏཾ། གསུམ་འདུས་སྦྱིན་སྲེག་བྱེད། ཀྲྀཥྛབྷཱུ་ཏཾ། སྟོཥྛ་བྷིཤྩས་ཏུའི་ཤ་ཙུ་རྣམས་ལ་ཤ་ཙུའོ། །ས་ཡིག་དང་ཏ་སྡེ་ཉི་རྣམས་ནི་ཕ་རོལ་ཤ་ཡིག་དང་ཙ་སྡེ་རྣམས་ལ་སྦྱོར་བ་ལ་ཤ་ཡིག་དང་ཙ་སྡེ་གྲངས་མཉམ་པ་རྣམས་སུ་འགྱུར་ཏེ། ཀས་ཙ་ར་ཏི། ཀཉྩ་ར་ཏི། གང་དུ་སྤྱོད། ༼ཀས་ཤུ་རཿཀབྷཱུརཿདཔའ་བོ་གང་། ཏཏྲ་ཙི་ཏྲ། ཏཙི་ཏྲ། དེ་རི་མོ་བྱེད། ཏཏྲ་ཤཱ་༑༑ཏཙྪཱ་ྋཧཱུྃ། བསྟན་བཅོས་དེ། ན་ཤཱཏྲ། ཤ་ལས་མ་ཡིན་ནོ། །ཤ་ཡིག་་—.།
