བསྟན་འགྱུར། སྒྲ་མདོ། ༼སིའི་ཆཚོ་ཞཙོ་ག་ཝཱཎ་ཊིཀཱ༌ལྔ་བློ་པཱ་མིཏྠ་ཏ་ལང་སོགས་དང་། ༈རྔུ་ལ་བྷ་ཐོབ་པ་ལ། ས་རྐྱེན་སྦྱར། །གཉིས་བརྗོད། ཁ་པ་སོགས་ཀྱིས་བྷ་པ། ས་ལ་ཨིས་འོས། ༼ལའི་ཨ་ཨིས་རུ་བསྒྱུར། གཉིས་བརྗོད་པའི་སྔོན་མ་ཕྱིས། སྲ་ཀྲ་དག་གི་སྔོན་མའོས། ཨིས་རུ་བསྒྱུར་བའི་ས་ཡང་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ལིཏྠ་ཏེ། ལཊ། »ལིཥྛ་ཏ། ལིང༌། ལིཏྠ་ཀཾ། ལོཊ། ཨ་ལིཏྠ་ཏ་ལང་སོགས་དང་། ༈དེ་བཞིན་དུ་ར་བྷ་རྩེད་མོ་ལ། རིབ་ཏེ། ལཊ། རིམྨེ་ཏ། ལིང་། རིཙྪ་ཏྭཾ། ལོཊ། ཨ་རིམྨ་ཏ་ལང་སོགས་དང་། ༈བ་ཏྠི་ོལྟུང་བ་ལ། པིཏྠ་ཏི། ལཊ། །པིཏྟེཏྲ་ལིང༌། །པིཏྠ་ཏུ། ལོཊ། པིདྨ་ཏཱཏྲ། ཨ་པིཏྟཏྲ་ལང་སོགས་དང་། ༈ཨཱ་པྲི་ཁྱབ་པ་ལས་གཉིས་བརྗོད་སྔོན་མའི་ཧ་ས་་་དང་པོ་ཕྱིས་པས། ཨཱ་ཨཱཔྲ། ས་ཞེས་གནས་པ་ལ། ༈ཨཱརྣོ་ཏེ་རཱི། ཨཱ་་་ཏིའི་ཨཱིའོ། །།ཨཱ་ཡོ་ཏི་སྟེ་བྱིངས་ཨཱཔྲ་ཀྱི་ཨུ་པ་དྷཱའི་ཨཱ་ནི་ཨཱི་རུ་འགྱུར་ཞིང་བྱིངས་་གཉིས་བརྗོད་པའི་སྔོན་མ་དེ་ཡང་དབྱི་བར་བྱ་སྟེ། ཨཱ་ཕྱི་མ་ཨཱིར་བསྒྱུར་ཨ་སྔོན་མ་ཕྱིས་པས། ཨཱིཙྪ་ཏི། ལཊ། ༼ཨཱིཔྤེཏྲ། ལིང༌། འཨཱིཔྤ་ཏུ། ལོཊ། ཡཨིམྨ་ཏཱཏྲ། ཨིཙྪཏཱ་ལང་སོགས་སོ། །ཏུགཁོག་བཞིན་དཀར་པོ་ཁྲོ་བ། །མཚོན་བ༈ཨ་ཤ་ཟ་བ་ལ། ཨ་ཤེ་ར་ན་ཡོབཱ། ཨ་ཤ་ལ་ཨ་ནཱ་ཡ་ཡང་ནའོ། །བྱིངས་ཨ་ཤ་ལ་འདོད་པའི་དོན་གྱི་རྐྱེན་སའི་ཡུལ་ལ་རྐྱེན་ཨ་ནཱ་ཡ་ཡང་ན་འགྱུར་ཏེ། ཨ་ནཱ་ཡ་སྦྱར་བའི་ཕྱོགས་ལ། ཨ་ཤ་ནཱ་ཡ་ཏི། ལཊ། ཨ་ཤ་ནཱ་ཡེཏྲ། ལིང་། 《ཨ་ཤ་ནཱ་ཡ་ཏུ། ལོཊ། ཨ་ཤ་ནཱ་ཡ་ཏཱཏྲ། ཨཱ་ཤ་ནཱ་ཡཏྲ་ལང་། སོགས་དང༌། ༈ས་་་སྦྱར་བའི་ཕྱོགས་ལ། གཉིས་བརྗོད། དབྱངས་དང་པོའི་ཕ་རོལ་ལོ་ཞེས་པས་ཤ་གོང་མ་མི་དབྱི། ས་ལ་ཨིས་འོས། ཨ་ཕྱི་མ་ཨིས་སུ་བསྐྱུར། །སྲ་ཀྲ་དག་གི་་སྔོན་མའོས། སྲ་ཕྱིས། །ཨིཊ་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཤི་ཤི་ཥ་ཏི*༑ལཊ། ཨ་ཤི་ཤི་ཥཏྲ། ལིང་། །ཨ་ཤི་ཤི་ཥ་ཏུ། ལོཊ། ཨ་ཤི་ཤི་ཥ་ཏཱཏྲ། ་ཨཱ་—།
