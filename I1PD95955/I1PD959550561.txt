ཎཨ་ནུ་བློ་ལ་ཡོད་ཏེ་ཁེ་དེ་ཧི་ཧེ་ཏཱ་བ་བ་དྷཱ་ར་ཎེ། །ཡ་ནེ་ཀཱརྦ་བསྒྲ། །།།་དབང་། ས་མུ་ཏུ་ཧི་ནི་རྒྱུ་མཚན་དམིགས་འཛིན་ནོ། །དོན་སྣ་ཚོགས་ཀྱི་སྡེ་ཚན་ནོ། །།ལ་། བ་ཀུ་ལི་ལ་འབད་ན་ཡིབ། །ཡོན་ཙེ་རཱ་ཡ་ཙི་ར་རཱ་ཏཱ་ཡ་ཙི་ར་སཱུ་ཏྲཱཉྩི་རཱརྦ་ཀཱ། །མུ་ཧུཔུ་ནཔུ་ནརྒྱུན་རིང་ཡུན་རིང་རིང་པོ་དང་། །དུས་རིང་པོ་ལ་སོགས་པའི་དོན། །ཡང་དང་སླར་དང་སླད་དང་ཤ་ཤཱ་ད་བྷིཥྛ་མ་ས་ཀྲྀཏྠ་མཱ། །སཱཀྟ་་གི་ཏྱཱཀྵ་སཱ་དྷཱ་ཡ་ས་ཡ་དི་བཾ་ནི། །རྟག་ཏུ་ཡང་ཡང་གཅིག་མིན་མཚུངས། །མགྱོགས་མྱུར་སྐྱོན་དང་ཡུད་ཙམ་དང༌། །དེ་མ་ཐག་མཀྵ་ཙ་དྲ་ཏེ། །བ་ལ་བཏཱུབྷྱཱ་ཀི། ཊི་བ། ལྔ་ནི། བད་ཁོག་དང་ཁ་ཏོ། ?ལ་དབང་པོ་ཀ་། །གཏི་དང་མི་སྡོད་མྱུར། །ཤིན་ཏུ་ནན་ཏར་ཧ་ལས་དང་། །།།་གྲྭའི་སགས་མུ་ཏ་སྭཱ་རྱ་ཏཱི་བ་ཙ་ནི་བྷ་ཁྱད་འཕགས་ལའོ་རེ། །པྲི་ཐཤྲཱི་ནཱནྟ་རེ་རྨ་རི་ཏྲེ་ཎརྟེ་ཏུ་ཧི་རུཀྲ་སུ་་དང་ཨ་ཏི་ཤིན་ཏུའི་དོན། །སོ་སོ་རེ་རེ་མཐར་བྱས་དང་། །ཏུ་ཧི་རུག་དང་ནཱ་ནཱ་བོར། །ཡད་ནཱ་ནཱ་ཙ་བརྫ་ནེ། །ཡཏྟདྨ་ཏཔ་ཏོ་ཧེ་ཏཱ་བ་སཱ་ཀ་ལྕེ་ཏུ་ཀིཉྩི་ཏ། །ཀ་དཱ་ཙི་ཛྫཏུ་ཏད་གང་ལས་དེ་ལས་རྒྱུ། །ཐམས་ཅད་མ་ཡིན་ཀིཉྩི་ཏ། །ཀ་དཱ་ཙིཏ་ནི་ནམ་ཞིག་གོ། །ལྷན་ཅིག་དུས་སཱརྦནུ་སཱ་ཀཾ་ས་ཏྭཾ་ས་མཾ་ས་ཧ། །ཨཱ་ནུ་ཀུ་ལྕརྦ་ཀཾ་པཱདྷཱི་བྱརྦ་ཀེ་ཏུ་བྲིཐཱ་མུ་དྷཱ། །གཅིག་མཉམ་མཚུངས་ལྡན། །མཐུན་ཕྱོགས་དོན་དང་མཐུན་པའོ། །དོན་མེད་བྱར་མེད་དོན་མེད་དོ། །ཨཱ་ཧོ་ཨུ་ཏཱ་ཧོ་ཀི་མུ་ཏཱ་བི་ཀ་ལྕེ་ཀིཾ་ཀི་མཱུ་ཏ་ཙ། །ཏུ་ཧི་ཙསྨྲ་ཧ་བི་རྟོག་པ་རྣམ་རྟོག་བརྟག་པ་དང་། །འདམ་ཁ་ཅི་ཞིག་རྣམ་པར་རྟོག། ཏུ་ཧི་ཙསྨྲ་ཧ་དང་བི། །ཚིག་གི་ཁ་པཱ་ད་པཱུ་ར་ཎེ་པུ་ཛ་ནེ་སཱ་ཏི། །དི་བཱ་བྷཱི་ཏྱ་ཐ་དོ་ཥོ་ཙ་ནཥཉྩ་ར་ཛ་ནཱ་་་སྐོང་དོན་ཡིན་ནོ། །སུ་དང་ཨ་ཏི་མཆོད་པའི་བྱ། །ཉིན་མོ་ཐུན་དང་དེ་ནས་ནི། །ཉིས་མཚན་ནམ་ཞེས་བི་ཏི། །ཏིརྨ་གརྗེ་འཁྱོག་པོ་སཱ་ཙེ་ཏ་རོ་པྱ་ཐ་སཾ་བྷོ་དྷ་ནཱརྦ་ཀཱ། །སྩཐཱ་ཊཾ་ག་བྱ་བའོ། །དུད་འགྲོའི་དོན་དང་སཱ་ཙི་ཕྱུགས། །དེ་ནས་བོད་པའི་དོན་ཡིན་ནོ། །སཿསཊ་དང་པཱ་ཊཾ་ག། —།
