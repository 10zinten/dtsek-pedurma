་་་ཁ་སྤྲོད་ཅའོ་དྷིདྷཱ༑། ་པ་ལ། །།།ཎ་ཤ་ལ་་ཁན་ས་ཏཱ་ར་སྒྲུའི་རིགས་ལ་ལྡན༑། བལ། །།།།།ཀྱི་གཏིཀ་མ་༼པ༽/ལ་ལཱyཆུ་ཅུ་ལྟས་ཀpཏ་ཀ་༈ཨུ་མཱ་ཀཱ་ཏྱཱ་ཡ་ནཱི་གོ་རཱི་ཀཱ་ལཱི་ཧེ་མ་བ་ཏཱི་ཤྭ་ར། །༈ཨུ་མའི་མིང༌། དཀའ་ཟློག་ཀ་ཏྱའི་བུ་མོ་དང༌། །དཀར་སེར་སྐྱ་མོ་ནག་མོ་གངས་ལྡན་མ། །ཤི་བཱ་བྷ་བཱ་ནཱི་རུ་དཱ་ཎཱི་ཤནྡྷ་ཎཱི་སརྦ་མཾ་ག་ལཱ། །ཨ་པཎྞཱ་པཱབྦ་ཏཱི་དུངྒཱ་མྲྀ་ཌཱ་དབང་མོ་ཞི་མ་སྲིད་མ་དང༌། །དྲག་མོ་གསོད་མ་ཀུན་ལ་ཤིས། །རི་སྐྱེས་རྫོང་གནས་ལོ་མ་མེད། །ཚིམ་ནཱི་ཙཎྜི་ཀཱམྦི་ཀཱ། །ཀྱི། །མེ་ཏོངཏི༌ཏེ་དེ་ཚེ་བྱེད་གཏུམ་མོ་འགྲོ་བའི་མ། །།།ས་དགའ་སྐྱོའི་ཆུ་ར་འདར་བྱེད་པ། །།ཚར་གྱི་ཁྱུ། ༈བི་ནཱ་ཡ་ཀེ་བ་བྷྱཱ་རཱ་ཛ་དྭེ་མཱ་ཏུ་རུ་ག་ཎ་:དྷ་པཱ༄༅། །༈ཚོགས་བདག་གི་མིང་། :ལོག་འདྲེན་བགེགས་ཀྱི་རྒྱལ་པོ་དང་། །མ་གཉིས་པ་དང་ཚོགས་ཀྱི་བདག། ཨ་སྱ་ཀ་དནྟ་ཧེ་རམ་ལམྦོ་ད་ར་ག་ཛཱ་ན་ནཱ། ་ཡོ། །ལ་ལོ། །ཆགས་ཏོག་མཆེ་བ་གཅིག་པ་ཧ་ལ་གནས། །ལྟོ་བ་འཕྱང་དང་གླང་པོ་གདོང་། །་ཙ། ཁྱོ་ཆ@p། །།ཟག་༈ཀཱརྟིཀེ་ཡོ་མ་ཧཱ་སེ་ནས་ར་ཛནྣཱ་ཥ་ཎཱ་ན་ན། ལྔ་། ༈གཞོན་ནུ་སྨིན་དྲུག་གི་བུའི་མིང་། སྨིན་དྲུག་བུ་དང་སྡེ་ཆེན་དང་། །འདམ་བུའི་ཚལ་སྐྱེས་གདོང་དྲུག་པ། །པཧྥ་ཏཱི་ནཾ་ད་ནསྐནྡསེ་ནཱ་ནཱི་རགི་བྷཱརྒུ་ཧ། །བཱ་ཧུ་ལེ་ཡསྟ་ར་ཀ་ཛི་རི་སྐྱེས་དགའ་བྱེད་སྐེམ་བྱེད་དང༌། །སྡེ་དཔོན་མེ་སྐྱེས་སྦེད་བྱེད་དང༌། །མང་པོ་ལས་སྐྱེས་སྒྲོལ་བྱེད་དྲི་ཤཱ་ཁཤི་ཁི་བཱ་ཧ་ན། །ཥཱཎ་ཏུ་རཤཀྟི་དྷ་ར༔ཀུ་མཱ་ར༔འདུལ། ༄༅། །གཙུག་ཕུད་ཅན་ཞོན་ས་ག་སྐྱེས། །མ་དྲུག་པ་དང་མདུང་ཐོགས་དང༌། །གཞོན་ནུ་དང་ནི་ཕྱི་ཀྲོལྦ་དཱ་ར་ཎ་། །ཚ་། །ཏུ་ཏུ་ཀྱི་ཊ་ཋཌོ་ཎ་ཊིཏ་་ཏ་་་ཁྲུང་ཁྲུང་འཇོམས། །་། /:། དེའི་ནང་་་་ཆོ་ཀཀཀྱིཁ་n"དང་ལྔཏུཾ་ཏ་༈ཨིནྡོ་མ་རུ་ཏཱརྨ་གྷ་བཱ་བི་ཌོ་ཛཱ༔པཱ་ཀ་སཱ་ས་ན། །བྲི་༈བརྒྱ་བྱིན་གྱི་མིང་། དབང་པོ་ལྔ་ལྡན་མཆོད་བྱེད་དང༌། །ཁྱབ་སྟོབས་དང་ནི་སྨིན་པ་འཇོམས། །རྒན་—།
