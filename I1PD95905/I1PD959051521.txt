བསྟན་འགྱུར། དབུ་མ། ཡ?ས་ཏེ་བ་བཞི་བརྒྱ་པའི་འགྲེལ་པ་སློབ་དཔོན་ཟླ་བ་གྲགས་པའི་ཞལ་སྔ་ནས་ཀྱིས་་་མཛད་པ་རྫོགས་སོ། །ལ། །་དཔལ་གྲོང་ཁྱེར་དཔེ་མེད་ཀྱི་དབུས་རིན་ཆེན་སྦས་པའི་ཀུན་དགའ་ར་བར། རྒྱ་གར་གྱི་མཁན་པོ་སུམྨ་ཛཱ་ན་ཞེས་བྱ་བ་གདུང་རབས་གྲངས་མེད་པར་པཎྜི་ཏ་བརྒྱུད་པའི་རིགས་སུ་སྐུ་འཁྲུངས་པ་བྲམ་ཟེ་ཆེན་པོ་རིན་ཆེན་རྡོ་རྗེའི་དབོན་པོ་བྲམ་ཟེ་ཆེན་པོ་སད་ཛཱ་ནའི་སྲས་གཅིག་ཏུ་བདེ་བར་གཤེགས་པའི་བསྟན་པ་ལ་་་གཅེས་སྤྲས་སུ་མཛད་པ་བདག་དང་གཞན་གྱི་གཞུང་ལུགས་རྒྱ་མཚོའི་ཕ་རོལ་ཏུ་སོན་པའི་ཞལ་སྔ་ནས་དང༌། བོད་ཀྱི་སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་པ་ཚབ་ཉི་མ་གྲགས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པོ། །གས། །བདེ་དང་། ནས་ཀྱི་དབང་དང་། དེན་:::ཀ་བསྡུར་མཆན། ::་་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་《སྣར་ལ་༄༅། །བཞི་བརྒྱ་པའི་འགྲེལ་པ་སློབ་དཔོན་ཟླ་བ་གྲགས་པས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད།  ི«པེ་»«སྣར་»ཙཱརྱ། «པེ་»«སྣར་»ཀ«པེ་»«སྣར་—པ། «སྣར་»—པོ། «པེ་《སྣར་+ནི། «པེ་»«སྣར་བཟུང་། །«སྣར་»དགོད། (འ«པེ་《སྣར་»རྣམ། ་.«པེ་»«སྣར་»བྱུང༌། &«པེ་»«སྣར་བརྒྱུད། ས«པེ་《སྣར་»གྱུར་ཅིང་། «པེ་ཆོས། «པེ་»«སྣར་རྟོག«པེ་»«སྣར་འགག་ཀྱི«པེ་»«སྣར་»ཁྱེན། «པེ་»«སྣར་སྣང་། «པེ་»སྣར་»ནི། ︽«པེ་»གནན། «སྣར་པས། «སྣར་བུར། ཚེ«པེ་»སྣར་ཅིང་། «པེ་»«སྣར་+དག«པེ་»«སྣར་+གྱུར་པར། («པེ་»«སྣར་»དེའི་ཕྱིར་སྤྱོ། ཀྱེ«པེ་»—པ། «སྣར་ཆོས། «པེ་»«སྣར་»དེ། ན་གན་«པེ་»«སྣར་»སྤྱོ། «པེ་»«སྣར་མ། «པེ་»«སྣར་»དུག«པེ་»ཆགས། མ«པེ་»«སྣར་སྐྱོན། «པེ་»«སྣར་པོ། ཚེ«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་ཀྱིས། «པེ་»།
