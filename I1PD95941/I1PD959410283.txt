རྟེན་ཀུན་ན་ཟླ་བྲལ་སྐྱེས་མཆོག་དེ་ཉིད་ཀྱི། །ཕྲིན་ལས་སྒྲུབ་པ་:སྙེ་བའི་ཐུགས་དགོངས་རྣམ་དག་དང་། །མི་རྗེ་ལྷ་སྲས་གུང་ཐང་རྒྱལ་མོ་ཡུམ་སྲས་དང་། །ཆོས་ལ་གཅིག་ཏུ་དཀར་བའི་ལྷ་གཅིག་ཀུན་དགའ་འབུམ། །མང་ཡུལ་སྐྱི་རོང་བཟང་མོ་ཡོན་ཏན་སྐྱིད་ལ་སོགས། །ཆོས་ལྡན་དམ་པ་རྣམས་ཀྱིས་སྦྱིན་བདག་་་བགྱིས་པའི་ངོར། །འཕགས་པ་རྟ་དབྱངས་ཀྱིས་མཛད་ཐུབ་པའི་མཛད་པ་འདི། །ས་དབང་བཟང་པོ་དང་ནི་བློ་གྲོས་རྒྱལ་པོས་བསྒྱུར། །འདི་བསྒྱུར་བསོད་ནམས་རྒྱལ་པོ་གང་དེས་ཕ་མ་དང༌། །འགྲོ་རྣམས་ལམ་གྱི་རྒྱལ་པོ་:དམ་པ་འདིར་ཞུགས་ཏེ། །ཉེས་ཚོགས་ཚང་ཚིང་ཀུན་དང་བདུད་བཞིའི་རྒྱལ་པོ་རྣམས། །བཅོམ་ནས་ཆོས་ཀྱི་རྒྱལ་པོ་:དམ་པ་ཐོབ་པར་ཤོག། །བསྡུར་མཆན། (མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»《སྣར་»ཏྲ། (འ«ཅོ་»གཱ། །(«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་»ཡང་། འ«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»བལྟམས། «པེ་»《སྣར་»རྩིབས། །(ི«པེ་»«སྣར་»འབྱུང་བར་འགྱུར། «པེ་»«སྣར་འཕོས། «པེ་»«སྣར་ལ། ཆོ«པེ་»«སྣར་»ཕྲོགས༑་«པེ་《སྣར་པས། «པེ་»«སྣར་གྱི། («པེ་»«སྣར་»ད། «པེ་»«སྣར་བས། ན«པེ་»«སྣར་»བལྟ། «པེ་»«སྣར་»དང་། «པེ་»«སྣར་རྒྱུད། «པེ་»«སྣར་»གྱུར། ༄༅«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ཡིད། «པེ་《སྣར་ལྟ། །«སྣར་པ་གར། «སྣར་»འདིར། «སྣར་དག®«པེ་»«སྣར་»སྟེང་དུ། །«པེ་»«སྣར་»སྟེང་དུ། «པེ་»«སྣར་»དུང་། @«པེ་»«སྣར་འཛུམ་མི། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»དག། (ཀྱེ«པེ་»«སྣར་»ཀྱི། །(«པེ་»«སྣར་མིན། «པེ་»«སྣར་»གིས། ︽«པེ་»《སྣར་བ། «པེ་»«སྣར་ཀྱི། 《སྣར་»—།
