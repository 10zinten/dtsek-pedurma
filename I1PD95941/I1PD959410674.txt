ཟད་ས་ལ་ལྷུང་བ་ནི། །འདབ་ཆགས་བྱ་ནི་མཁའ་ལ་བཞིན། །ལས་ཀྱི་ཤུགས་ཀྱིས་འགྲོ་བ་འདི། །གང་ཕྱིར་འཁོར་ལོ་བཞིན་དུ་འཁོར། །དེ་བས་ཆགས་པ་ཡོངས་སྤངས་ཤིང་། །ཐར་པ་ལ་ནི་བློ་ཞེན་བྱ། །མྱ་ངན་བསལ་པ་ཞེས་བྱ་བ། །སློབ་དཔོན་རྟ་དབྱངས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།ཁ་པ། རྒྱ་གར་གྱི་མཁན་པོ་ཨ་ཛི་ཏ་ཤྲཱི་བྷ་དྲ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཤཱཀྱ་འོད་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།Pਕੋਈ"ਸਮਕRAਲਬབ་བྱ་::བསྡུར་མཆན། དེ་ལ་བྱེད་་་་་་། ཁན་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་ལ་མྱ་ངན་བསལ་་་་་་བཞུགས་སོ། །ཞེས་དང་། 《སྣར་»ལ་རྟ་དབྱངས་ཀྱིས་མཛད་པའི་མྱ་ངན་བསལ་བ། མི་དགེ་བ་བཅུའི་ལམ་བསྟན་པ། དྲན་པ་ཉེར་གཞག་གི་ཚིག་ལེའུར་བྱས་པ་འགྲོ་བ་དྲུག་གི་ཚིག་ལེའུར་བྱས་པ། བཙུན་པ་ཆོས་ལྡན་རབ་་་་འབྱོར་དབྱངས་ཀྱིས་མཛད་པ། འདོད་ཡོན་ལྔའི་ཉེས་དམིགས་སློབ་དཔོན་དབྱིག་གཉེན་གྱིས་མཛད་པ། དགེ་སློང་གཞོན་ནུ་རབ་གསལ་ལ་སྤྲིང་པ་རྣམས་བཞུགས། ཞེས་འཁོད། «པེ་»«སྣར་»བཱི། ཉི«པེ་»«སྣར་»བསྒྲིམས། «པེ་»«སྣར་སྲིང་ཞིང་། མི«པེ་»«སྣར་»སྒྲོན་མ། ཚེ«པེ་»«སྣར་པོ། ༄༅«པེ་»«སྣར་»འཕོ། «པེ་»«སྣར་གདུང་བ། ནང་བ་པ། ནང་བ་བ། «པེ་»«སྣར་»གྱིས། ཚེ«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་སྐྱབསྲིང་། ︽«པེ་»«སྣར་»འགྲེ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»བའམ། ༄༅«པེ་»«སྣར་»བསྲེག«པེ་»«སྣར་རྣམ། «པེ་»«སྣར་མྱོང་བ། «པེ་»«སྣར་»གིས། ཀྱེ«པེ་»«སྣར་»ལ། ཚེ«པེ་»«སྣར་བསྲེགས། «པེ་»«སྣར་»སྐམས་གྱུར། །«པེ་»«སྣར་»གྱི། Q«པེ་»«སྣར་ཕྲོགས། མི«པེ་»«སྣར་»འཆི། «པེ་»«སྣར་»འབབ། ཚུ«པེ་»འགའ། «པེ་»«སྣར་»བླངས་པོར་བ་གངྒཱའི། «པེ་»«སྣར་སྙེད། ཚེ«པེ་»«སྣར་»འཐུངས་པའི། །«པེ་»«སྣར་ན། ཁེ«པེ་»«སྣར་བརྗེད། ཀྱེཉེ«པེ་»«སྣར་བཏགས། ཉེ«པེ་»«སྣར་»ཆད། ཨེ«པེ་»«སྣར་རྣམ།  ི«པེ་»«སྣར་»བཞུན། «པེ་»«སྣར་ནི། «པེ་»«སྣར་»ཡིས། ཚེ«པེ་»«སྣར་»ཡི། ཁོ«པེ་»«སྣར་མ། @«པེ་»«སྣར་བུར། ཚེ«པེ་»«སྣར་པ། །ས་ལུས། @«པེ་»—།
