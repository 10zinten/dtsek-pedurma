བཞིན་བྱེད་ཅིང་། །སྐྱེ་བོ་སྨྲ་བ་ལ་བརྟེན་འཚོ་བར་བྱེད། །རྩོད་པའི་དུས་ཀྱི་གཏམ་་་སློབ་དཔོན་:མ་ཏི་ཙི་ཏྲས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་ཨ་ཛི་ཏ་ཤྲཱི་བྷ་དྲ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཤཱཀྱ་འོད་ཀྱིས་བསྒྱུར་བའོ། །།།ਲਉਸਰਬਸੁਲਤਾਨਉਰੂਲਨੂੰབསྡུར་མཆན། ་དུ་དེ་དག་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་རྩོད་དུས་་་་་གཏམ་མཱ་ཏི་ཙི་ཏྲས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་»ཀཱ། Q«པེ་»«སྣར་»དགོད། «པེ་»སྤྱོད་པ་ཡི། «སྣར་སྤང་བ་ཡི། «པེ་»«སྣར་»དཔས། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»+དེ་དང་། ཚེ«པེ་»«སྣར་»བཏང་། «པེ་»«སྣར་བསྐམས། «པེ་»ཀྱི་བཅོམ་བྱས་པ། «སྣར་»ཀྱི་བཅོམ་བྱས། @«པེ་»«སྣར་རྣམས། @«པེ་»«སྣར་»ནི། «པེ་ནུས། «པེ་»«སྣར་»ཀུན། «པེ་»«སྣར་བརླན། @«པེ་»«སྣར་གྱུར། ༄༅«པེ་»«སྣར་»བཅུགས། «པེ་»«སྣར་»གྱི། «སྣར་»སྐྱེད། «སྣར་»ཡིས། «སྣར་གྱུར། ཨོ«པེ་»«སྣར་སྒོད། «པེ་»«སྣར་»མཱ་ཏྲི་ཙི་ཏས། བསྡུར་འབྲས་རེའུ་མིགསྐྱེས་རབས། (དེའི་ཁ་བྱ་::gམདོ། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། །སྐྱེས་རབས། །།སྤྲིངས་ཡིགསྐྱེས་རབས། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནན་བན་བན་ན༽ན་ཕྲེང་། །།།།ཚིག་རྐང་། །།།ཚེག་ཁྱིམ། །རྒྱ་—།
