རྟག་ཏུ་ནམ་མཁའ་ལ་རྒྱུ་ཞིང་། །གཟུགས་ནི་ཤིན་ཏུ་ཡིད་འོང་བའི། །:བུམ་པ་དྲི་ཡིས་ཉེས་པ་ཡིས། །ངེས་པར་འཆི་བར་འགྱུར་བ་ཡིན། །རི་དྭགས་ཉིད་ནི་སྒྲ་ཙམ་གྱིས། །རེག་པས་གླང་པོ་བརླག་པར་འགྱུར། །གཟུགས་ཉིད་ཀྱིས་ནི་ཕྱེ་ལེབ་སྟེ། །རོ་ཡིས་ཉ་ནི་རྣམ་པར་རླག། བུང་བ་དྲི་ལ་ཆགས་པ་སྟེ། །འདི་དག་རྒྱུ་ནི་རེ་རེ་ཡིས། །སྐྱེས་བུ་དག་ནི་གཅིག་ལ་ཡང་། །ལྔ་པོ་འདི་དག་རྟག་ལྡན་ཞིང་། །ཉིན་མཚན་དུ་ནི་རབ་འཇོམས་ན། །བདེ་བ་ཅིགནས་ཇི་ཐོབ་་་་པར་འགྱུར། །འདོད་པའི་ཡོན་ཏན་ལྔའི་ཉེས་དམིགས་བཤད་པ་སློབ་དཔོན་་་དབྱིག་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་དྷརྨ་ཤྲཱི་བྷ་དྲ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་རིན་ཆེན་་བཟང་པོས་:བསྒྱུར་ཅིང་ྋཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་དབུའི་མཚན་སྨོས་པ་ལྟར་བཀོད། @«པེ་»«སྣར་བརྟེན། «པེ་»«སྣར་»སྲུངས། «པེ་»«སྣར་»འགྲན། («པེ་»«སྣར་»འདྲའི། «པེ་»«སྣར་»རིགཚེ«པེ་»«སྣར་གསད་པར། «པེ་»«སྣར་མི། «པེ་»«སྣར་»ཡི། །རོ་མྱག་པ། ཏེ«སྣར་»ཡིས། །«པེ་»«སྣར་»བུང་བ་དྲི་ཡི། ཨི«པེ་»«སྣར་བརླག«པེ་»«སྣར་»+སྐྱེས་བུ་དག་ནི་རེ་རེ་ཡིས། །«སྣར་»བ། «པེ་»«སྣར་»+ལྟར། @«པེ་»«སྣར་»+དགེ་སློང་། «པེ་»«སྣར་»བསྒྱུར་ཅིང་། །
