བི་དྱ་ཀུ་མུ་དི་ནཱི་བི་ཀཱ་ཤ་ར་སི་ཀས་དཱ། །ཀ་ལཱ་བཱནྟིརྨ་ལ་ཏུ་ཙིརླཔུཎྞནུ་ཀུན་གྱི་ཀུ་མུད་ཅན། །རྟག་ཏུ་རྣམ་རྒྱས་བྱེད་ལ་ཆགས། །དྲི་མེད་མཛེས་པ་སྒྱུ་རྩལ་ཅན། །གང་བའི་རི་བཱ་བ་བྷོ། །བ་བྷཱུ་བ་ཏ་སྱ་བྷཱུ་བཏྟཱནྟཱ་པ་ཏིནྟཱ་མྱ་ནན་ར། །མཱ་ནཱི་ཟླ་བ་ལྟ་བུར་མཛེས། །ས་བདག་དེ་དང་ས་གཞི་ནི། །བར་མེད་ས་བདག་བྱུང་གྱུར་པ། །ཁེངས་ལྡན་མ་ཧེནྡྲ་སེ་ནཱ་ཀྵཔྲ་ཀྵ་ཏ་པྲ་ཐུ་བི་ཀྲ་མ། །ཏེ་ན་དུས་ཧ་དཎེ་ན་དབང་ཆེན་སྡེ་ཞེས་པ། །རྒྱ་ཆེ་རྣམ་པར་གནོན་པར་གྲགས། །དུས་ཡིན་དུས་སུ་བརྟག་བྱ་ཞིང་། །སརྦ་སཏྠ་སཱ་ཧཱ་རི་ཎཱ། ་། ཨ་ཀཱ་ལ་ཀཱ་ལ་ཀལྤ་ན་བཱི་ཌ་ཏཱཤུ་ཤུ་ཙུཀུན་གྱི་བདོག་ཀུན་འཕྲོག་བྱེད་པ། །བཟོད་དཀའི་དཔྱ་དང་ལྡན་པ་དེས། །གཟིར་བའི་སྐྱེ་རྒུས་མྱ་ངན་པྲ་ཛཱ། །ཏ་ས་དྷརྨ་པྲ་བྲྀཏྟ་སྱ་ནི་བྲྀཏྟ་སུ་ཀྲྀ་ཏོཏྲ་བེ། །ཏཱི་བྲོ་པ་ཏཱ་པཾ་ཏཱསྟེ་བྱས༑། ཆོས་མིན་ལ་ནི་ཞུགས་པ་དེའི། །གྲོང་ནི་ལེགས་བྱས་སྤྲོ་བ་དག། ལོག་ཅིང་གདུང་བ་དྲག་པོ་ཥཱ་སུ་རེ་ན་བ་བྲྀ་ཥཱནྟཱ་ནཱ། །པྲ་ཏི་ཀུ་ལེ་མ་ཧཱ་པཱ་ལེ་ཏ་ཏྲ་དུབྷི་ཀྵ་པི་བླ་ཡིས། །གདུང་བར་སྤྲིན་གྱི་ཆར་མ་ཕབ། །ས་སྐྱོང་རྗེས་སུ་མི་མཐུན་པ། །དེར་ནི་མུ་གེའི་ཉེན་འཚེ་པ༑། བ་བྷུ་བ་བྱ་ས་ནེ་ཥཱ་བ་བྷ་བནྟི་བྱ་པ་ནོ་ད་ཡཿ། །ཏ་ཏགླེ་ཤ་བི་བྱུང༌། །སྡུག་བསྔལ་ཅན་རྣམས་ཁོ་ན་ལ། །སྡུག་བསྔལ་རྒྱས་པར་གྱུར་པ་ཡིན། །དེ་ནས་ཉོན་མོངས་ཀྱིས་ཤེ་ཥཱནྟ༔སབྦེ་པུ་ར་ནི་བཱ་སི་ན། །ཀུ་པ་ཏི་བ་ས་ནོདྲེ་གཱ་དེ་བཱི་བྷཱུ་ཏཱ་ཁྱད་གཟིར། །བདག་ངན་གྱིས་གདུངས་ཡིད་བྱུང་ཞིང་། །གྲོང་ནང་གནས་པ་ཐམས་ཅད་ནི། །གཅིག་བྱ་ཙིནྟ་ཡན། །རཱ་ཛྙཱ་དོ་ཥ་ཀ་རེ་ཎེ་ཡཾ་ཇ་ཎེ་ན་ཛ་ན་ཏུ་འཚོགས་ནས་རྣམ་པར་བསམས། །རྒྱལ་བླུན་མཚན་མོར་བྱེད་པ་ནི། །གསར་པའི་དཔྱ་ཟེར་ཏ་པ་རཾ། །པྲ་ཏྱ་གྲ་ཀ་ར་པ་ཏེ་ན་ན་ལི་ནཱི་བ་ནི་མི་ལི་ཏཱ། །ཨེ་ཥ་འབེབས་པ་ཡིས། །མཆོག་ཏུ་སྐྱེ་བོའི་ཚོགས་པ་ནི། །པདྨོ་ཅན་བཞིན་ཟུམ་པར་གྱུར། །རྟག་ཏུ་ངན་དུཅུ་ས་ནཱི་ནི་ཏྲ་མ་ས་ཏྭཱ་མ་ཎྱ་སརྨ་ཏ། །ཨསྨཱསྣ་པཱི་རྒྱུ་པུབྷྱཱ་ཏི་པར་ཞེན་པ་འདིས། །བདེན་མེད་བློན་པོའི་གྲོས་དག་གིས། །བདག་ཅག་བཙིར་ནས་གཡོ་ཅན་དང་། །།
