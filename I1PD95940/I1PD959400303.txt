རཾ་བྷོ་ཚུ་ཐུ་ཡ་ཏེ་ཀཱནྟཾ་རཾ་བྷ་གཌྷ་ས་མུདྡྷ་བཾ། །ཏ་ཏ་པྲ་བྷ་བ་བི་ནཋ་ཤེ་མཛེས་མ་བྱིན་པ་ཆུ་ཤིང་འདྲ། ་། ཆུ་ཤིང་ནང་ནས་བྱུང་བར་ཐོས། །གཟི་ལྡན་སྙིང་སྡུག་དུ་འོས་དེས། །སྟེ་ཛ་སཱི་པྲ་ཎ་ཡོ་ཙི་ཏཱ། །སཱ་ཏིན་དྷུ་ཥི་ཏཱ་རྱ་པི་མཱ་ཏཾ་གི་རི་བ་པདྨེ་ནི། །དེ་མཐུས་རེ་ེཉམས་དེ་དག་གིས། །ད་ལྟ་ཡང་ནི་སུ་®མ་ཕྱིན། །པདྨ་ལྡན་པ་གླང་པོས་བཞིན། །ཤྲ་བ་ཎཱ་དེ་བ་སཱ་ནནྡ་མ་པཆུ་ཥི་ཏ་ཀོ་ཏུ་ཀཾ། །ན་ཀ་རོ་ཏི་མ་ནཀ་སྱ་བུད་མེད་རིན་ཅེན་སྐྱེ་གནས་ལས། །མ་སྐྱེས་པ་དེ་ཐོས་ཉིད་ལས། །ཡིད་ནི་ཀུན་དགར་བཅས་པ་ཡིས། །ཏཏ་སི་རཏྣ་མ་ཡོ་ནི་ཛཾ། །ཨ་བྷི་ལ་ཥི་མ་ནས་ཧཱུཾ་ཛ་ཏཾ་མེ་ས་ཧ་ཙཀླུ་ཥཱ། །དགེ་མཚན་མི་གཞན་སུ་མི་བྱེད། །བདག་གི་མིག་དང་ལྷན་ཅིག་ཏུ། །ཡིད་ནི་དེ་ལ་མངོན་པར་ཆགས། །ཏད་གུ་ཎ་ཐུ་ཏི་དྷ་རྩྭ་ཡ་ཐོ་ཏཱ་ཡ་སྤྲོ་ཧ་ཡཱ་མྱ་ཧཾ། །ཨི་ཏུལྟེ་བྷཱུ་མི་པ་ཏི་ནཱ་དེ་ཡི་ཡོན་ཏན་ཐོས་པ་ཡི། །སྐལ་ལྡན་རྣ་བར་བདག་དགའ་འོ། །ས་ཡི་བདག་པོས་འདི་བརྗོད་ཚེ། །གོ་པསྟ་པྲ་ཏྱ་བྷ་ཥ་ཏ། །།བྷཱུ་ཛ་ག་ག་ཎ་སཾ་རུདྡྷས་རཱ་ཛ་ན་མཱརྨ་ཐོ་ནི་ས་འཚོས་དེ་ལ་ལན་སྨྲས་པ། །རྒྱལ་པོ་འདོད་ཆགས་གཏེར་དེ་ནི། །ལག་འགྲོའི་ཚོགས་ཀྱིས་ཡང་དག་དྷི༈། ཨ་ཏྱལླས་ལི་ཏཾ་པྲ་སྱ་དུས་ཧཱ་པ་ཏ་དུན་མ། །།ཨེ་ཥ་ཏེ་བི་ཥ་བཀག། ཤིན་ཏུ་ཆུང་བ་འཁྲུལ་གྱུར་ན། །བཟོད་དཀའི་གཡང་ས་ངན་པར་འགྲོ། །མི་བཟད་ལམ་འདི་མ༔པནྠ་དནི་ཏོ་བི་ཥ་མེ་ཏུ་ཎ། །ལ་བྷ་ཏི་སཱ་ནི་རྒཧཱུྃ་ན་ཡུཏྟཾ་ཁྱོད་ལ་ནི། །མི་བཟད་མདའ་ཆེན་དག་ྋགིས་བསྟན། །དེ་ནི་བླང་དུ་འཐོབ་པ་མིན། །ཁྱོད་ཀྱང་འགྲོ་ག་མ་ནཾ་ཙ་ཏེ། །ཀི་མསྨིན་བི་ར་ཏོ་པཱ་ཡེ་བི་དཱ་མྱུ་བྷ་ཡ་སཾ་ཤ་ཡེ། །ཨི་ཏུཀྟསྟེ་བར་འོས་པ་མིན། །གཉིས་ཀ་ཐེ་ཚོམ་ཐབས་དང་བྲལ། །འདི་ལ་བདག་ནི་ཅི་ཞིག་སྨྲ། །དེ་ཡིས་དེ་ན་ཚིགས་པ་ཏིརྣོཥཎྜ་གྲ་མ་ཏུ་ཛ་ཏ། །བི་དཾ་སོ་སྟུ་ཙི་ཏྭཾ་ནི་ཏྭཾ་བརྗོད་མི་བདག་གིས། །འདོད་པའི་གདོན་ནི་མ་གཏང་ྋསྟེ། །འདོད་པས་གཟིར་ན་རིགས་པའི་ལུགས། །ན་སྨ་རནྟི་སྨ་རཱ་ཏུ་རཱ། །བི་ཤ་ལི་ཀ་པུ་རཱི་ག་ཏོ་གོ་བེ་ན་ས་ཧི་ཏོ་ཐ་པ། །།མཁས་པ་རྣམས་ཀྱང་མི་དྲན་ནོ། །དེ་ནས་ས་འཚོ་ལྷན་ཅིག་དེ། །གྲོང་ཁྱེར་ཡངས་པ་ཅན་སོང་ནས། །།
