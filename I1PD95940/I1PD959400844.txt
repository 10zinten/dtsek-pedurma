ལང་ལཀེ་ཎ་ཤྭ་བྷཱ་བཱ་སི་ནཱ། །རཀྵི་ཏ༔་ཏི་མཱ་སཱ་རྱ་པཀྱི་བསྣ་བྱ་པ་བྱ་གཡང་སར་གནས་པས་གཞོན་ནུ་ཡང་། །བསྲུངས་ཏེ་འདབ་ཆགས་བཞིན་དུ་ནི། །ས་ལ་ཕྱིན་ཅིང་ཤི་མ་ཏ༑། སུ་དྷརྨཱ་བི་པ་ཏི་ཏྭཾ་ཐུ་ཏྭཱ་པུ་ཏྲ་སཱ་ནི་དྷ་ནེ་ཏི་ཎི། །སུ་ཏཱ་ཏེ་ཛཱི་བ་གྱུར། །བུ་ནི་ལྷུང་ཐོས་ཆོས་བཟང་དག། རང་ཉིད་འཆི་བ་འཚོལ་བ་ནི། །ལྷ་མོ་རྣམས་ཀྱིས་ཁྱོད་ཏཱི་ཏྱཱ་ཀྵ་རཀྵི་ཏཱ་དི་བྱ་ཡོ་ཥི་ཏཱ། །ཀུ་མཱ་རོ་པི་ཁ་ལ་བྱཱ་གྷ་ཀླུ་ར་གྱུརྞ་ཤི་ལཱ་ཀྱི་བུ། །འཚོ་ཞེས་བརྗོད་ནས་རབ་ཏུ་བསྲུངས། །གཞོན་ནུའང་སྟག་རྨིག་རྣོན་པོ་ཡིས། །རྡོ་བའི་གཞི་ནི་ཏ་ལཾ། །ག་ཛཱ་སྲིདྨཏྟ་ཤཱརྡུ་ལ་དཱ་རུ་ཎཾ་བ་ན་མཱ་བི་ཤཏ། །ཏ་ཏྲ་ཉམས་བྱས་ཤིང་། །གླང་པོའི་ཁྲག་གིས་མྱོས་པ་ཡིས། །མི་བཟད་སྟག་ལྡན་ནགས་སུ་ཞུགས། །དེ་ནི་སཾ་ག་ལ་ཀ་ཀྱི་ན་ལུདྡྷ་ཀེ་ན་ནི་བེ་དི་ཏཾ། །མཱརྒ་མཱ་སཱ་རྱ་པུ་རུ་ཎཾ་ཀྲྀ་ཏྟ་རྔོན་པ་མིག་ཅེསེར་ཅན། །ཞེས་པ་ཡིས་ནི་རབ་བཤད་དེ། །ལམ་ནི་ཐོབ་པའི་སྐྱེས་བུ་དག། ལུས་ཀུན་གཱ་ཊཾ་བྱ་ལོ་ཀ་ཡ་ཏ། །བི་ལོ་ཀྱ་ཛཏ་ཀ་རུ་ཎ་སྟེ་ས་པྲ་ཙྪ་ཛྲ་པཱཏྨ་ཛ། །ཀེ་ནི་བཅད་པ་མཐོང་བར་གྱུར། །དེ་མཐོང་སྙིང་རྗེ་སྐྱེས་པ་ཡི། །མི་བདག་བུ་ཡིས་རབ་ཏུ་དྲིས། །ཁྱོད་ཀྱི་ཚྭ་བི་ཤ་བ་སྦེ་ནཱི་ཏོ་པི་བི་ཛ་ནེ་བ་ནེ། །སོ་བྲུ་བཱིསྣ་ཏི་་རེ་ཏྲ་སྡུག་བསྔལ་གནས་སྐབས་འདི། །མི་མེད་ནགས་སུ་ཅི་ཡིས་ཐོབ། །དེས་སྨྲས་རིང་པོ་མིན་འདི་ན། །ཙཱཎཱ་ལཉྩཎྜ་ཙེཊྚི་ཏ། །ནི་བ་ས་ཏྱནྟ་ཀཿཔཾ་སཱ་སུ་དཱ་སོ་ནཱ་མ་དུས་ཧ། །།གཏུམ་པོ་གཏུམ་པོའི་སྤྱོད་ཚུལ་ཅན། །བཟོད་དཀའ་སུ་དཱ་ས་ཞེས་པ། །སྐྱེས་བུ་རྣམས་ཀྱི་མཐར་བྱེད་གནས། །ཏ་སྱ་ཤཾ་ཁ་མུ་ཁོ་ནཱ་མ་པཱ་ར་མེ་ཡོསྟི་བྷི་ཥ་ཎ། །པནྣཱ་སྦི་ས་ཀ་ལ་ཀཱིཙྪཱ་དེ་ལ་དུང་གི་ཁ་ཞེས་པ། །ཁྱི་ནི་འཇིགས་སུ་རུང་བ་སོདི། །ལམ་གནས་མཐའ་དག་གང་མཐོང་ཡེ་ནི་ཏཱ་བི་ཧི་ཏཱ་དི་ཤ། །ཏངྒཱ་ཙ་རཙུ་ཏ་ཀྱེ་ཡ་མཾ་གཙྪེ་ད་ས་མུདྡྷ་བ+༑། གདོན་མི་ཟ་བར་རྣམ་འཇོམས་པ། །དེ་ཡི་སྤྱོད་ཚུལ་ལྟུང་གྱུར་པ། །ཡུད་ཙམ་འཚོ་བའི་ལྷག་+ཚིག་རྐང་གཉིས་པོ་འདི་བོད་སྐད་དུ་མ་བསྒྱུར་འདུག་པ་ཟ་ཧོར་བན་དེས་བསྒྱུར་བ་མཁས་པ་རྣམས་ཀྱིས་དཔྱད་པར་ཞུ། བཱ༑། མཱུ་ཧཱུརྟ་ཤི་ཥ་ཛཱི་བ་མྱ་བྱ་ཐཱ་མེ་མརྨ་ཤཱ་ཡི་ནཱི། །ཀཱདྷཤཾ་ཁ་མ་ཅན། །བདག་གི་གནད་བསྣུན་ཟུག་རྔུ་འདི། །ཡན་ལག་བཅད་ལས་ཡང་དག་སྐྱེས། །གཏུམ་པོ་ཉིན་—།
