མཱ་ཐ་ཡནྟི། །ཏ་ཧཱུྃ་པྲ་ཡཉྫ་ནེ་བ་རཱཌི་ཏཱ་ཡཱ་མ་པ་རྱ་ཏཱ་ཏཾ་པྲ་ཎ་ཡོ་པི་དྲུ་རེ། །པུ་བདག་ལ་ཡོད། །དེ་ནི་འབད་པས་མག་པ་ལ་བྱིན་ཚེ། །བུ་ཚ་ཉིད་དང་དེ་ལ་བྱམས་པའང་རིང་། །པཱ་ཎོ་པྲ་དཱིལྟ་མི་བ་དྲི་བནྟི་དྷཚུ་ན་ཤཀོ་ཏི་ཛ་ན་སྭ་ཀ་ནྱ། །ནྱ་པཱཟླ་ལག་པར་སྒྲོན་མེའི་སྡོང་བུ་རབ་འབར་བཞིན། །སྐྱེ་བོས་རང་གི་བུ་མོ་བཟུང་མི་ནུས། །གཏམས་པའི་ཏུ་ལྕ་པ་རཱམ་ཎཱི་ཡཱ་ཙིནྟཱ་ཕ་ལཱ་ཨེ་བ་ཀུ་ལ་སྱཱ་ཀ་ནྱ། །ན་བྷཾརྱ་དེ་ཡཱ་ནོར་བཞིན་གཞན་ལ་གཏད་བྱ་བ། །རིགས་ཀྱི་བུ་མོ་བསམ་པའི་འབྲས་བུ་ཉིད། །གཡོག་ལ་སྦྱིན་མིན་ན་ཙ་པོ་ར་ཡོ་གྷ་དྷུ་རོ་ཙི་ཏཱ་ཨེ་བ་ན་རེནྡྲ་ཀ་ནཱ། །ཨ་ཛྷཱ་ཏ་མི་ཏྱཱ་བཱ་ཙི་ཏ་སྱ་རཱུ་གྲོང་པ་ལ་འོས་མིན། །མི་དབང་བུ་མོ་རིང་པོ་ཉིད་ལ་འོས། །དེ་ལྟར་ཀུན་སྤྱད་རིང་པོ་མི་ཤེས་པས། །ར་མྲྀ་ཏ་སྱ་ཡཱ་ཏ་སྱ་ཙ་ཀོ་ནི་ཤེ་ཥ། །ཏསཏྠ་ཡནྡྷེ་ན་མ་མ་ས་ཀཉྩིཛྫཱ་མཱ་ཏཱ་བྷཱ་བེ་གུ་ཤི་དང་སོང་བ་དག་ལ་བྱེ་བྲག་ཅི། །དེ་སླད་གང་ཞིག་བུ་དངོས་བདག་ཉིད་ཅན། །རང་ཡུལ་རབ་སྤངས་ཎེ་བཱཎྚི་ཡོ་ཛུ། །སཱ་དེ་ཤ་མུཏྠཾ་ཛུ་མ་མི་བ་དེ་ཤེ་བི་བྷནྟི་ཡཔུ་ཏྲ་བཱ་བདག་གི་ཡུལ་འཛིན་པ། །བདག་གི་མག་པའི་དངོས་གྱུར་འགའ་ཞིག་དེ། །ཡོན་ཏན་ལྡན་པ་འབད་པས་དྷི་པ་ཏྱཾ། །ཐུ་ཏ་ཡཱ་ཛ་སུ་ཏཱ་པྲ་བཱ་ཧ་པུ་ཏྟཱ་པ་དཱ་ཏེ་པུ་ལི་ནོ་པ་ཀཎེ། །བཙལ་བར་བྱ། །བདག་གིས་ཐོས་པ་གངྒཱ་དག་གི་རྒྱུན། །བསོད་ནམས་དག་པའི་ངོགས་དང་ཉེ་བ་ན། །ས་ཧཱཀྵ་ལཱི་ནཱནྟི་ཏ་པོ་བ་ནཱནྟི་རཱ་ཛནྟི་རཱསྟེ་ཀི་ལ་ཀཱ་བྱ་སཱ་ཎྱ། །ཏ་སྱ་ཙུ་སྙིམ་པར་བཅས་ན་དཀའ་ཐུབ་ནགས་མཐར་ནི། །རྒྱལ་པོའི་དྲང་སྲོང་ཀཱ་ཤུ་པ་གནས་ལོ། །དེ་ཡི་ས་ཏཾ་པྲ་ཤྲ་བ་ཎེ་ན་བཱིརྨ་དེ་བོ་ད་ཡཱ་ད་ནི་སནྣི་ཏཱ་གྲེ། །ཏོ་ཡསྤྲོ་ཧཱཎྜ་བོན་ལྕི་བས་ལྷུང་གྱུར་པ། །ལྷ་ཡིས་བསྐྱེད་ལས་རྡོ་སྟེང་དམའ་བར་ཆགས། །ཆུ་འདོད་ཀྱིས་གཟིར་ཀྱེ་ཧ་རི་ཎཱི་ནི་བཱི་ཡ་སུ་བརྞ་བརྞ་སུ་པྲ་བེ་ཀུ་མཱ་རཾ། །བི་ན་མྲྀ་གཱསྟ་རི་དྭགས་མོ་ཡིས་འཐུངས། །གསེར་གྱི་མདོག་ཅན་གཞོན་ནུ་བཙས་པར་གྱུར། །དེ་ཡི་ཕས་བཟུང་སྟེསུ་བི་བདྷཱི་ཏོ་སོ་པི་ཏཱ་གྲོ་ཧི་ཏོ་པི་ཧི་ཏ་བྲ་ཏ༄༅། །ནཱ་སྣ་པྲ་སིདྷནགས་སུ་རི་དྭགས་དེ། །རྣམ་པར་བསྐྱེད་ཅིང་བརྟུལ་ཞུགས་དག་ཀྱང་བསྒྲུབས། །སོར་ཚད་རྭ་ཡིས་ཡང་—།
