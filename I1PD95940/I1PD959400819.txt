བ་དྷོ་མེ་ཏཱཏྲ་ཏིཔཱ་པེ་ཧཱི་ན་་སམྤནྣཱ་ལཛྫ་ཡཱ། །མུ་ཁཾ་ཛྲཐཾ་ན་ཤཀོ་མི་འབྲེལ་དམན་ངོ་ཚས་སྡིག་ཅན་མ། །ཁྱོད་ཁྱོ་བདག་གིས་གསད་པར་བྱ། །ཁྱོད་སློང་ལ་འོངས་ལྷ་རྣམས་ཀྱི། །ཏཱཀྵཙྪཱཋ་དི་བོ་ཀ་སིཾ། །ས་ཏྱ་མུཏྟཱ་ཧཱ་ཡུཀྟ་སྱ་ཀུ་ལ་མྱོསྣ་ཏི་ཤཱ་བཞིན་ནི་བདག་གིས་ལྟ་མི་ཕོད། །བདེན་པར་སྤྲོ་བ་དང་ལྡན་ཞིང་། །མཐོན་པོར་གྱུར་པའི་རིགས་ལི་ན། །ཛ་རེ་བི་ཀི་ལ་ཀཱ་ལ་ཀ་རྩྭ་སཾ་ཀོ་ཙ་ཀཱ་རི་ཎཾ། །ཨི་ཏི་བཱ་བྷནྟི་དག་ནི། །བུ་མོས་འཁུམས་པར་བྱེད་པ་སྟེ། །དུས་ནི་ངེས་པར་དགའ་བས་བཞིན། །དེ་སྐད་ཕ་ཡིས་ཏཱ་པི་ཏཱ་ཏ་མུ་བཱ་ཙ་ན་ཏཱ་ན་ནཱ། །ཨ་སཱུ་ཏྲ་ཧཱ་རཾ་ཀུནྡྷ་ཎྞཾ་ཀུ་ཙ་ཡོ༴བླ་སྨད་གྱུར་དེས། །མཆི་མའི་ཐིགས་པ་ནུ་མ་ལ། །དོ་ཤལ་སྐུད་མེད་སྒྲིག་བྱེད་ཅིང་། །བཞིན་བཏུད་དེ་བིནུ་བྷི། །ཀོ་པ་ད་ནུ་ཙི་ཏཾ་ཏཱ་ཏ་ནི་ཏནཾ་བཀྱུ་མན་སི། །ན་ཤུ་ལ་རབ་སྨྲས་པ། །ཡབ་ལགས་ཁྲོ་བར་མི་འོས་ཨེཔ། །དེ་དག་བདག་ལ་གསུང་འོས་མིན། །མཐུ་ཡིས་ཡནྟེ་པྲ་བྷཱུ་པེ་ན་ཀིནྣ་རཱཀིནྣ་རཱ་དྷི་ཀ། །ག་རུ་ཌ་སཱུ་པི་དུརྀ་རྒྱུ་མི་མི་འམ་ཅི་དག་ལས། །ལྷག་པའི་མི་མ་ཐོས་སམ་ཅི། །མཁའ་ལྡིང་གིས་ཀྱང་བགྲོད་དཀའ་བ། །ས་མཱ་མུ་རྒྱུ་ཡཀྴི་ཏི། །པྲ་བྷཱ་བ་བྷུ་མི་རཱ་ཡཱ་ཏི་མ་ནུ་ས་ཀ་ཐ་བྷ་བེཏ། །འདི་གང་གིས་བགྲོད་གྱུར་ནས། །འོངས་པ་མཐུ་ཡི་ས་གཞི་དེ། །མི་ཉིད་ཡིན་པར་ཅི་ལྟར་འགྱུར། །གུ་ཎ་སཾ་བཱ་དི་ནི་མཱུཏྟིཌྷ་བ་ཏྱ་ཡ་ཤ་རཱི་རཱ་ཎྞཾ། །ཀ་རོ་ཏྭཱ་ནནྡ་སཾམ་དཾ་ལུས་ཅན་རྣམས་ཀྱི་ལུས་དག་ནི། །ཡོན་ཏན་ཡང་དག་རྗོད་བྱེད་ཉིད། །བསིལ་ཟེར་ཅན་གྱི་འོད་ཟེར་ནི། །དྷུ་ཏི་རེ་བ་ཧི་མ་ཏཱི་ཥ། །ཛཱ་ཏྭཱ་ཀིཾ་ཀྲྀ་ཡ་ཏེ་ཏཱ་ཏ་སྭཱ་བྷཱ་བཱ་ནུ་གུ་ཎཱ་གུ་ཀུན་དགའ་སྐྱེད་པར་བྱེད་པ་ཡིན། །ཡབ་གཅིག་རིགས་ཀྱིས་ཅི་ཞིག་བྱ། །ཡོན་ཏན་རང་བཞིན་རྗེས་སུ་ཎཱ༑། པུཎེནུ་ར་མྲྀ་ཏ་སྱནི་ཀཱ་ལ་ཀུཊ་སྱ་སོ་ད་རཿ། །བྷ་ཐཾ་ཏནྟརྒུ་ཎ་འབྲང་། །ཟླ་རྒྱས་བདུད་རྩི་ཟག་པ་ནི། །ནག་པོ་བརྩེགས་པའི་སྤུན་ཟླ་ཉིད། །ཁ་ཅིག་ནང་ན་ཡོན་ཀེ་ཙིཙྪནྣཱ་དོ་ཥཱ་སྟ་ཐཱ་པ་རེ། །ཨ་པ་རཱི་རྒྱུ་ན་ཀརྟ་བྱ་མ་ཎྜི་ཏན་ལྡན། །དེ་བཞིན་གཞན་ནི་སྐྱོན་བསྒྲིབས་པས། །ནོར་བུ་རྣམས་ཀྱི་རིན་དག་ནི། །མ་བརྟགས་།
