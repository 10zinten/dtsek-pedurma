ས་བན་སྟ་ཌི་ཏ་ཨི་བ་ནིརྨེ་པ་ཙ་པ་ལཱ། །བ་སུསཧྰ་པཱ་ཡི༔རྣམས་རྫོགས་པར་འཆིང་བའི་ཀློག་ཀྱི་བཞིན་རེག་པ་མེད་ཅིང་རབ་ཏུ་གཡོ། །ལུས་ནི་གནོད་པ་ཀུན་དང་ཀྵ་ཡ་བྷ་ཡ་ནི་ཀཱ་ཡི༔པ་རི་ག་ཏཾ་ཛ་ར་རོ་གོདྲེ་གིསྟ་ད་བི་ན་བི་འཇིག་དང་འཇིགས་པའི་ཚོགས་དང་རྒ་དང་ན་བ་དང་། །སྡུག་བསྔལ་དག་གིས་ཡོངས་སུ་འགྲོ་སྟེ་དེ་ལྟ་ན་རཱ་གས་ནུ་ཤྲཱི་ཏཾ། །ཤི་ཡེ་སྭསྟི་ས་མལྟ་ཡི་སཱ་ཛ་ན་སཱུ་ཡ་མཉྫ་ལི། །ཡང་ལུས་འཛིན་ཡིད་མི་འབྱུང༌། །དཔལ་འབྱོར་མཐའ་དག་ཤིས་གྱུར་ཅིག། ཐལ་མོ་འདི་སྦྱར་ནང་མི་ལ། །དཱཀྵི་ཅུ་ཀྵ་མ་ཏཾ་སྦི་ཥཱ་པྲ་ཛཱི་བ་པྲ་ཡཱ་མ་མ། །ཨི་ཏི་་ཏྭཱ་ས་པི་ཏ་བུད་མེད་རྣམས་ལ་ཡོངས་བཟོད་མཛོད། །བདག་ནི་རབ་བྱུང་ཁོ་ན་དགའ། །འདི་བསམ་དེ་ཡི་ཕ་མ་རོ་ས་མཱ་ཤཱ་སྟསྟ་ལོ་ཙ་ནོ། །བུདྡྷོ་དྷརྨ་པ་ཐེ་ཤུདྡྷ་ཤ་མ་དྷཱམི་རྒྱ་བེ་དག། །ཡང་དག་དབུགས་ཕྱུང་མིག་དག་ཐོབ། །ཆོས་ལམ་དག་ཅིང་ཞི་བ་ཡི། །གནས་ལ་བློ་ནི་བཀོད་ཤ་ཡཏ། །སཱརྦ་བྷཱཋཤྲཱི་རཱ་ཡཱ་ཏཀྲྀ་ཤོ་པི་སུ་ཛ་ན་སྱ་ཙ། །ཨ་ལུལྟ་པར་གྱུར། །དོན་མཐུན་ཉམས་གྱུར་ཡུན་རིང་ན། །་འོང་ཞིང་རིད་ཀྱང་སྙིང་སྟོབས་འབྱོར། །མ་ཞུམ་ས་ཏྭཱ་བི་བྷ་བནྣ་ཀྲྀ་པཱ་སྤ་ད་ཏྭཾ་ཡ་ཡོ། །ཨ་ནུ་ཀཾ་སྱ་ཡ་བྱེ་ཥ་སཾ་པས་ན་རང་མི་ཡང་། །བརྩེ་བའི་གནས་ཉིད་དུ་མ་གྱུར། །འཁོར་བའི་ཉོན་མོངས་ཀྱིས་རྣམས་འཁྲུགས། །སཱ་ར་ཀླུ་ཤ་བི་ཧྭ་ལཾ། །ས་ཏཀ་སཱུ་ནུ་ཀཾ་སྭཱ་སྟེ་སམྤ་ཏ་སམྤརྐ་ནི་སྤྲོ་ཧཱ། །པ་དེ་དག་གལ་ཏེ་བརྩེ་བ་ན། །དམ་པ་སུ་ལ་རྗེས་མི་བརྩེ། །དེ་དག་ཕུན་ཚོགས་ཐོབ་མི་འདོད། །ཕྱུགས་ཤུ་པཱ་ལ་ཀ་པི་པྲ་སྟི་སཾ་དེ་ཤ་དི་ཡ་ཐོ་དི་ཏཾ། །ནི་ག་བྱ་དེ་བྱཱཀ་ན་སྐྱོང་བྲམ་ཟེ་བུད་མེད་ལ། །གསེར་ཐོབ་མངོན་སུམ་མཚན་ཉིད་ཅན། །འཕྲིན་སོགས་ཇི་ལྟར་བརྗོད་པ་ཀ་པཱསྟི་པྲ་ཏྱ་ཡ་ལཀྵ་ཎཾ། །ཤཱནྟཀཱ་ཏྱཱ་ཡ་ནཾ་པྲ་སྱ་བྲ་བྷཾ་ས་ས་མཱ་ད་དེ། །བཞིན། །དེ་དག་རྣམས་ལ་བཟླས་བྱས་ཏེ། །ཀཱ་ཏྱཱ་ཡ་ན་དང་ཕྲད་ནས། །ཞི་བ་དེ་ཡིས་རབ་བྱུང་བླངས། །མུབྷཱ་ནཾ་ཡདྲི་ཥཱ་དཱ་ཡ་ཏནྡྲ་སཱ་ད་ཡ་དྷི་མ་ཏྭཾ། །ས་ས་མཱ་སཱ་བྱ་བི་ཥ་དཾ་ཤོ་བླུན་པོ་རྣམས་ཀྱི་སྒྱིད་ལུག་རྒྱུ། །དེ་ནི་བློ་ལྡན་རབ་དང་རྒྱུ། ་། དེ་ནས་དེ་ཡི་རྒྱུན་ཞུགས་ཀྱི། །འབྲས་།
