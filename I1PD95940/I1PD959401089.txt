ཤ་ཡ་སཾ་ཏརྟ་༑་དྲྀབྷྱཱ་ནརྡ་བྷུ་པ་ཏི། །ནི་ནནི་མན་པུ་ཅུ་ཏྭཾ་བདག་དགའ་བོ་ནི། །དྲག་པོར་འགྱོད་པས་རབ་གདུངས་པ། །བག་མེད་ལས་བྱུང་བདག་ཉིད་ཀྱིས། །པྲ་མཱ་དོདྷཱ་ཏ་མཱདྨན། །ས་བྷིཧཱུྃ་པ་དཱ་པ་ཏི་ཏཔྲ་སཱ་དྷཱཙྪཱ་བྱ་བཱ་བསོད་ནམས་དམན་པ་ཉིད་ལ་སྨད། །དེ་ཡིས་དགེ་སློང་ཞབས་ལ་གཏུགས། །དང་བྱས་གོས་ཀྱིས་གཡོགས་ས་སཱ། །ཨུ་བཱ་ཙཱ་ཙཱ་ན་ཛཾ་དོ་ཎཾ་མ་མཱརྨ་ཀྵཀྱུ་མཱན་སི། །ཏ་དེ་བ་ནས་སྨྲས། །མི་ཤེས་ལས་སྐྱེས་བདག་གི་སྐྱོན། །འཕགས་པ་ཁྱོད་ཀྱིས་བཟོད་པར་འོས། །གཉིས་སྐྱེས་དེ་བཏྠ་ཀཱ་སཉྫི་ལ་བྷཱ་ཏེ་ན་དཱི་ཛརྨ་ནཱ། །ཏཱ༴ནྡྷ་ནཾ་ཏུ་པཱ་པ་སྱ་མོ་ཧེ་ན་ཡིས་དེ་ཚེ་ཉིད། །བེའུ་དག་ནི་ཐམས་ཅད་ཐོབ། །སྡིག་ཅན་བདག་ནི་གཏི་མུག་གིས། །ཁྱོད་ཉིད་མ་མ་བིསྨིཾ་ཏཾ། །།པ་རི་ཛཱ་ཏས་བྷཱ་ཡཾ་ཏཱ་མནྡ་བྷཱ་གྱིན་ཀེ་ན་ཙིཏ། །བཅིངས་པ་བརྗེད་པར་གྱུར། །མདུན་སར་ཁྱོད་ནི་སུ་ཡིས་ཀྱང༌། །སྐལ་བ་ཞན་པས་ཡོངས་མ་ཤེས། །དཎྜཔ་ཏ་ཏི་ཤུདྡྷེ་ཥཱ་པཱསྟ་པཱ་ཏེ་མ་ཧཱི་པ་ཏོ། །ཨི་ཏུཀྟི་ཀྲི་ཏི་ནཱ་ཐེ་ན་ས་བདག་ལྷུང་བ་ཐོབ་པ་ན། །དག་པ་རྣམས་ལ་ཆད་པ་ལྟུང་ཚེ། །ས་ཡི་མགོན་པོས་འདི་བརྗོད་ཚེ། །གྷཱནྟ་པྲ་གྷཱ་ལི་ཏཱ་ཤ་ཡ། །ཏཾ་རི་བ་ཏོ་བ་དམྨནྟརྨ་མ་རཱ་ཛརྣ་བི་དྱ་ཏེ། །ན་ཏཱ་བཟོད་པས་བསམ་པ་རབ་བཀྲུས་པ། །རི་བ་ཏ་ཡིས་དེ་ལ་སྨྲས། །རྒྱལ་པོ་བདག་ལ་ཁྲོ་བ་མེད། །ཅི་ཡང་པཱ་པ་ཀྲྀ་ཏཾ་ཀིཉྩིདྲི་པུ་ལ་ཀླུ་ཤ་བཱ་ཏི་ན། །ཏདྨ་མོ་པ་ན་ཏཾ་པཱ་ཀེ་སྭཱ་ཁྱོད་ཀྱིས་གནོད་མ་བྱས། །ཉོན་མོངས་རྒྱ་ཆེར་ལྟུང་གྱུར་པ། །བདག་ལ་རང་ལས་མཚུངས་པའི་འབྲས། །ཀརྨ་ས་དྲྀ་ཤཾ་ཕ་ལཾ། །ཡཏྠོཋཌཱ་ཏ་ཡི་བ་སརྦ་བི་པ་དཀུཊྚནྟི་ཀཎེ་གྲ་ཧཾ་སྨིན་པ་དེ་ནི་ཉེ་བར་གནས། །གང་ཞིག་འདོད་ལྡན་ཉིད་ཀྱིས་རྒུད་པ་ཀུན་གྱི་མགྲིན་ནས་འཛིན་པར་བྱེད་སབྷྱཱ་ག་པྲ་ས་བྷོ་ག་བྷོ་ག་སུ་བྷ་གཱཀྱི་བྱནྟི་ཡཏཱལྤ་ད། །པ་དང༌། །གང་ཞིག་ཕུན་ཚོགས་ཡན་ལག་ཐམས་ཅད་ལྷག་པའི་ལོངས་སྤྱོད་སྐལ་བཟང་ཉམས་བྱེད་དང་། །ཡནྡྷེཏཱཙྪནྟ་སུ་ཁཱསྤ་དཾ་བི་ཧ་ར་ཎཾ་དཱིགྷཉྩ་ཡ༴ནྡྷ་ནཾ་ཏཧཱུྃ་ཧཱུཾ་ནི་ཛ་ཀརྨ་གང་ཞིག་རང་འདོད་བདེ་བའི་གནས་སུ་རྣམ་པར་རྩེ་དང་གང་ཞིག་རིང་དུ་འཆིང་བྱེད་པ། །དེ་དག་སྐྱེས་བུ་།
