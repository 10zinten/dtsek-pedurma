ཡོ་ཧི་ནི་ར་པ་ཏྲ་པཱ། །ཨི་ཎྱཔྲ་ཡ་མ་བི་ཤུ་ཏྭཱ་རཱ་ཛཱ་ཏདྲ་ཙ་ནཾ་ར་ཧ། །མེད་རྣམས་ལ་ཁྲེལ་ཡོད་མིན། །ཞེས་པ་དེ་ཚིག་སྙན་མིན་ཡང་། །དབེན་པར་རྒྱལ་པོས་ཐོས་གྱུར་ནས། །པི་ཏྲ་བི་ཤ་ཤ་པཱ་པོ་གྲ་དུཁ་སནྟཱ་བ་མ་ཏྱ་ཛཏ། །པྲ་ཀུཊྚནྟུ་ལྟ་དྲེ་རུ་ད་ཡ་ཕ་བསད་སྡིག་པ་དྲག་པོ་ཡིས། །སྡུག་བསྔལ་དག་གིས་གདུང་བ་བཏང་། །ཤར་གྱི་རི་བོས་ནུབ་ཀྱི་རི་གི་རི་ཎཱ་བྷེ་ཥ་ཀ་ལ་ནཾ་ཀཱ་ཎཱ་ཏ་ཀོ་ཎཱི་གྷཱ་བྷིདྡི་གྷ་ཊ་ན་བི་ནོ་དཾ་བི་ད་དྷི་ཏི། །ལ་སྦྱོར་བའི་བྱ་བ་རབ་ཏུ་བྱེད། །སྐད་ཅིག་གིས་ནི་ས་དང་ས་འཛིན་བྲལ་བའི་རོལ་རྩེད་རྣམ་པར་བསྒྲུབས། །སྲི་ཛནྟེ་ཏཱ་བཙཱི་ས་ཡ་དི་ས་ལི་ལཱཏྟཙཱ་ད་ཧ་ནཱ་ད་སཱ་དྷཾ་ནཱ་རཱི་ཏཱཾ་ན་ཧི་བྷ་འདི་དག་འཕྲལ་ལ་མེ་ནི་ཆུ་ལས་རྣམ་སྤྲོ་དེ་ཡང་སྲེག་བྱེད་ལས། །།སྲིད་གསུམ་དག་ན་བུད་མེད་རྣམས་ཀྱི་བ་ཏི་ཀིཝིཏཱི་བྷུ་བ་ནེ། །ཨ་ཐཱ་སོཉདྷདྨ་དྷེ་ནི་བ་ཤ་ལ་ཏུ་ཕྱེ་ན་བཱི་ཌི་ཏ། །ཚིགས་བསྒྲུབ་བྱ་མིན་པ་ཅི་ཡང་མེད། །དེ་ནས་དགྲ་བཅོམ་བསད་པ་ཉིད། །ཟུག་རྔུ་ལྟ་བུས་གཟིར་བ་ཡི། །མི་པས་པྲཙྪ་དྷམྨ་ཛྙཱ་ནྣིཀྲྀ་ཏི་ཏ་སྱ་ཀརྨ་ཎ། །ཏ་ཏ་སྟོ་དུཥཱ་ས་ཙི་བོ་བདག་དེ་ཡིས་ལས་དེ་ནི། །ཕྱིར་བཅོས་ཆོས་ཤེས་རྣམས་ལ་དྲིས། །དེ་ནས་བློན་ངན་དེ་དག་གིས། །ཏི་ཥྛ་པུ་༔་ཁྱ་ཙི་ཏྱ་ཡོ། །མཱཛྫ་ར་པོ་ཏོ་བྷཾ་ཏཱནྟས་ཀཱ་བཱ་མཱི་ཥ་ཤི་ཀླུ་ལ། །རྒྱལ་དང་སྐར་རྒྱལ་མཆོད་རྟེན་གྱི། །ནང་དུ་བྱི་ལ་ཆུང་ངུ་གཉིས། །ཤ་ཡི་བསླབས་པས་ཆགས་བྱས་ནས། །ནི་ཏིདྷཱ་བ་པི་དྷནྡྷེ་ན་པྲ་བི་བྱ་ཚྭ་པ་ཏེས་བྷཾ། །ཏ་མཱུ་ཙ་ཏུབཀག་པར་གྱུར་ཀྱང་སྤྱི་བརྟོལ་གྱིས། །མི་བདག་མདུན་སར་རབ་ཞུགས་ཏེ། །གདུང་བ་དྲག་པོས་ཏཱི་པྲ་ཏཱ་པ་སནྟཱ་བ་པྲ་ཤ་མཱནི་ནཾ། །དེ་བ་མི་ཕྱི་བ་བྷ་བ་ཏཱ་ཙིཏྟ་མཱ་ཡོངས་གདུངས་པ། །རབ་ཞི་དོན་གཉེར་དེ་ལ་སྨྲས། །ལྷ་གཅིག་ཁྱོད་ནི་རྫུན་ཉིད་ཀྱིས། །སེམས་ནི་ཡཱ་སྱ་ཏེ་བྷཾ་ཤཾ། །སརྔ་ཀ་ལྱཱ་ཎཱ་ལོ་ཀེ་སྨིན་ནཱཉནྟསནྟི་ཏེ་ཏ། །མཆོག་ཏུ་གདུང་བར་བྱེད། །འཇིག་རྟེན་འདི་ན་དེ་ཉིད་དུ། །ཐམས་ཅད་དགེ་བའི་དགྲ་བཅོམ་མེད། །ཡ་དི་ས་ཏྱཾ་བྷ་བེ་ཡུསྟེ་ན་བྷ་སཱ་རཱ་ཛ་ཧཾ་ས་པ་ཏ། །རིདྡྷཱ་ཡཱནྟི་ཀ་ཐཾ་ཏེ་ཥཱ་མ་གལ་ཏེ་དེ་དག་བདེན་པ་ན། །མཁའ་ལ་ངང་པའི་རྒྱལ་པོ་བཞིན། །དེ་དག་རྫུ་འཕྲུལ་གྱིས་བགྲོད་པས། །—།
