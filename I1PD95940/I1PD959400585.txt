ཀྲྀ་ཏྭཾ་ཛ་ལི། ་། ཨ་ཡཾ་གུ་ཎ་ག་ཎོ་པེ་ཏཀུ་མཱ་རཿཤྲཱི་མཱ་ཏཱ་བ་ར། །ཧེ་ནས་རྒྱལ་པོས་དྲིས། །བཅོམ་ལྡན་ལས་ནི་ཅི་ཞིག་གིས། །གཞོན་ནུ་དཔལ་ལྡན་རྣམས་ཀྱི་མཆོག། ཡོན་མེ་ན་དནྟི་ནཱ་ཡཱ་ཏྭཾ་བྷ་ག་བནྟེ་ན་ཀརྨ་ཎཱ། །ཨི་ཡཾ་ཙཱ་བ་ར་ཀ་རྩྭ་ཏན་ཚོགས་ནི་རྒྱས་པ་འདི། །གསེར་གྱི་གླང་པོས་འགྲོ་བ་ལགས། །འདི་ཡི་བག་གསར་བདག་གི་ནི། །ཙ་མཉྩ་ཏ་སྱ་ན་བཱ་བནྟུ། །།ཀེ་ན་པུ་ཏྟཱ་བི་པཱ་ཀེ་ན་ཛཱི་བི་ཏཱ་ད་དྷི་པལླ་བྷཱ། །བུ་མོ་གོས་ཀྱིས་མཛེས་མ་འདི། །བསོད་ནམས་གང་གི་རྣམ་སྨིན་གྱིས། །སྲོག་ལས་ལྷག་པར་སྙིང་དུ་སྡུག། ཨི་ཏི་པྲཋཀྱི་ཏཱི་ཤེ་ན་སནྣཱ་བིདྡྷ་ག་བཱན་ཛི་ན། །ཏ་མུ་ཙེ་བྷཱུ་པ་ཏེ་པཾ་སྭཱ་ས་ཡི་བདག་པོས་འདི་དྲིས་ཚེ། །ཀུན་རིག་བཅོམ་ལྡན་རྒྱལ་བ་ཡིས། །དེར་གསུངས་སྐྱེས་བུའི་འབྱོར་པ་པུ་ཙྩོདྷུ་ཏཱ་བི་བྷཱུ་ཏ་ཡ། །ཡ་དུ་དཱ་རཾ་ཡ་དུ་ཙི་ཏེ་ཡདྷཱ་རྣམས༑། ས་བདག་བསོད་ནམས་དག་ལས་བྱུང་། །འཇིག་རྟེན་ན་ནི་འདོད་བྱ་གང་། ་། ་རྒྱ་ཆེ་གང་ཛིཝཱུ་མདྡྷ་ཏྭཾ། །སྤྲོ་ཧ་ཊཱི་ཡཾ་ཙ་ཡཏྠོ་ཀེཏྟཀྲུཅུ་ས་མུབྷཱུ་ཁཾ། །དང་འོས་པ་གང་། །®མཛེས་པ་གང་དང་རྨད་བྱུང་གང་། །དེ་དེ་བསོད་ནམས་ཉིད་ལས་འབྱུང༌། །བི་པ་བྱི་བྷ་ག་བཱན་པུབྷཾ་སུ་ག་ཏས་ཧ་བྷིཀྵ་བྷི། །ཙ་ཙཱ་ར་ལོ་བཅོམ་ལྡན་བདེ་གཤེགས་རྣམ་པར་གཟིགས། །དགེ་སློང་དག་དང་ལྷན་ཅིག་སྔོན། །རྒྱལ་པོ་གཉེན་ལྡན་ཀ་ཀྲྀ་པ་ཡཱ་རཱ་ཛྙཱ་བཛྫཱ་མ་ཏཔུ་རེ། །ཏསྨིནྣ་བ་ས་རེ་ཏ་ཏྲ་ཀུ་མཱ་རྩྭ་ས་ཧ་གྲོང་ཁྱེར་དུ། །འཇིག་རྟེན་ལ་ནི་བརྩེ་བས་རྒྱུ། །གནས་སྐབས་དེར་ནི་ཁྱེའུ་ཞིག། གཞོན་ནུ་མ་དང་དཱ་ར་ཀཿ། །བི་ཀཱི་ཌཱ་བནྡྷ་ནི་པུ་རཿཀྲྀ་ཏྭཱ་དཱ་རུ་མ་ཡཾ་ག་ཛཾ། །ཏོ་བི་ལོ་ཀྱ་ས་འགྲོགས་ནས་ནི། །ཤིང་གི་རང་བཞིན་གླང་པོ་བྱས། །ལམ་གྱི་མདུན་དེར་རྣམ་པར་རྩེ། །བཙོ་མ་ཛམྦུ་ན་མཱ་ཡཱནྟཾ་དྷཱ་ཏ་ཛཾ་བྷུ་ན་ད་དྷུ་ཏི། །ཕུལླ་པདྨ་ད་ལཱ་ཀཱ་ར་ཀ་རུ་ཎཱ་སྣེབྷཱུ་དའི་འོད། །པད་འདབ་རྒྱས་པའི་རྣམ་པ་ཅན། །ཡིད་འོང་སྙིང་རྗེའི་སྤྱན་ལྡན་པ། །བཅོམ་ལྡན་བྱོན་པ་ལོ་ཙ་ནཾ། །།བྷ་ག་བནྟཾ་ས་མུདྡྷཱ་ཏ་ཏདྡྷཏྟི་པ་ར་སོརྨུ་ཁོ། །ཀཱི་ཌཱ་ག་ཛཾ་ནི་དེ་དག་གིས། ་། མཐོང་ནས་དེ་ལ་གུས་པ་རྒྱས། །རྗེས་སུ་ཆགས་པས་མངོན་ཕྱོགས་ཏེ། །དེ་ལ་རྩེ་ཆ་གླང་།
