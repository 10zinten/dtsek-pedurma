ག་ཛི་ཏསྟེ་ན་ས་མུ་དཱ་ཙཱ་ར་ཤིཀྵ་ཡཱ། །ཨ་དྷི་པི་ཏཔ་ནི་ཁི་ལོ་དྷརྨཱ་ག་མ་ཙ་ཏུཊྚ་ཕྱུང་སྟེ། །:ཀུན་སྤྱོད་དང་ནི་བསླབ་པ་ལ། །ཆོས་ཀྱི་ལུང་ནི་བཞི་པོ་དག། མ་ལུས་དེ་ཡིས་ཁ་ཏོན་ཡཾ༑་་། ཏ་ཏཀ་དཱ་ཙིདྨ་ཎི་ཛཾ་མནྡཱ་ཎཱ་ཏུ་ལྱཱ་ཛནྡ་ནཾ། །ཤ་ཏི༔པཉྩ་བྷི་སྦྱངས། །དེ་ནས་ནམ་ཞིག་ཚོང་པ་ནི། །མཚུངས་པར་སྐྱེས་པའི་གྲོགས་པོ་དག། བརྒྱ་ཕྲག་ལྔ་ཡིས་ཆུ་རཾ་བྷོ་དྷི་ག་མ་ནེ་ན་ས་ཧཱ་ནཱི་། །ཨཱ་རུ་རོ་ཧ་པྲ་བ་ཧ་ཎཾ་ཏེ་གཏེར་ལི། །།ལྷན་ཅིག་འགྲོ་བར་གསོལ་བཏབ་སྟེ། །དེ་དག་དགེ་བ་སེམས་བྱེད་པས། །འཇིགས་པ་ཏཱཾ་ཀུ་ཤ་ལ་ཙིནྟ་ཀ། །བྷ་ཡེ་ཥ་དྷིརླ་ཛ་ན་ནཾ་གྲ་ཧི་ཏཱ་ཤཱ་ས་ནཾ་གུ་རོ། །ཨ་དག་ལ་བརྟན་བསྐྱེད་པ། །བླ་མའི་བཀའ་ནི་བཟུང་བྱས་ཏེ། །གཟིངས་ལ་རབ་ཏུ་ཞོན་པར་གྱུར། །དེ་ཐཱབྷི་མ་དེ་སཾ་རུབྷེ་ཏསྨིནྡྲ་བ་ཧ་ཎི་ཀ་ཎཾ། །སཾ་ཀྲནྟེ་བ་ཎི་ཛཱི་སཱརྗེ་ནས་གཟིངས་དེ་ཆུ་གཏེར་གྱི། །དབུས་སུ་ཡང་དག་འགགས་གྱུར་ཅིང་། །ཚོང་པའི་དོན་མཐུན་ཆོ་ངེ་སྲོཙཱ་ཙཱ་ར་བ་ཙོ་བྷ་ས། ་། ཡ་དི་པྲ་བ་ཧེ་ཎ་སྭཱ་སྱ་མུཀྟིརྨུཥཱ་ཀ་མིཔི་དག། སྒྲོགས་ཚེ་ཆུ་ལས་ཚིག་དྲག་པ། །གལ་ཏེ་གཟིངས་འདི་ཁྱེད་ཅག་རྣམས། །གྲོལ་བར་འདོད་པ་དེ་ཏཱ༑་་། ཏ་དེ་ཥཀྱི་པྱ་ཏྭཾ་ཀླུ་ཛྲ་ས་ལི་ལེ་སཾ་གྷ་ར་ཀྵི་ཏཱ། །ཐུ་ཏཱི་ཏ་ཏ་པཱ་ཎ་སནྡེ་ལྟ་ན། །དགེ་འདུན་སྲུང་འདི་ཆུ་ལ་ནི། །མྱུར་བ་ཉིད་དུ་འཕང་བར་མཛོད། །དེ་ཐོས་སྲོག་ལ་ཐེ་ཚོམ་ཚེ། །ཧེ་ཏེ་ཏྟཱ་མེ་ཀཱ་བྷ་བརྨ་ཏི། །བ་རཾ་ནི་དྷ་ན་མེ་བསྡུ་ན་ཏུ་སཱ་དྷུ་སུ་ཀྲྀདྨ་དྷ། །དེ་དག་རྣམས་ལ་བློ་འདི་བྱུང༌། ་། ཤི་བར་གྱུར་པ་མཆོག་ཡིན་གྱི། །དམ་པའི་གྲོགས་པོ་གསོད་པ་མིན། །བི་ཥ་མེ་སཾ་ཤ་ཡེ་ཏེ་ཏཱཾ་ཀྲྀ་པ་ཡཱ་རཀྵ་ཎོདྨ་ཏ། །བི་ཀྵེ་པ་སཱ་ཡ་མཱནྡྷཱ་ནཾ་དེ་དག་མི་བཟད་ཐེ་ཚོམ་ལ། །བརྩེ་བ་ཡིས་ནི་བསྲུང་བརྩོན་པས། །ཆུ་གཏེར་འཁྲུགས་པར་དགེ་འདུན་གླུབྷྱཱབྷོ་སཾ་གྷ་ར་ཏ། །ས་ནཱ་ག་ནི་ལ་ཡཾ་པཱསྟསྟ་ཏྲ་ནཱ་གི་སྲུང་། །རང་གིས་བདག་ཉིད་རབ་ཏུ་འཕངས། །དེ་ནས་ཀླུ་ཡི་གནས་སུ་ཕྱིན། །དེ་ནི་ཀླུ་ཡིས་རཞྭ་ནཱི་ཏ། །བི་དྷ་ཡ་པཱུརྦ་སཾ་བུདྡྷ་ཛཾ་དྷ་ཙེ་ཏྭཱ་པི་བནྡ་ནཾ། །གསོལ་བཏབ་དེས། །རྫོགས་སངས་རྒྱས་ཀྱི་མཆོད་རྟེན་ལ། །མངོན་པར་ཕྱག་འཚལ་མངོན་བསྒྲུབས་ནས། །།
