ཏ་སྱ་བ་ཏ་ར་ཏོ་བྷུ་ཁཾ། །བི་མཱ་ན་ཀཱ་ན་ནཱ་ཀཱཎྜི་བ་བྷཱུ་བ་བྷཱུ་བ་ནཾ་ག་ཎ། །མེད་རྣམས་འབྲངས་པས། །གཞལ་ཡས་ཁང་གི་ཚོགས་ཀྱིས་ནི། །ས་གཞི་མཉམ་པ་ཁེངས་པར་གྱུར། །ཏ་སྱ་དཾ་ཏཾ་ཤུ་སཾ་ཏཱ་ནི་རུ་པ་དེ་ཤ་མི་བཱ་ཙི་ཏཾ། །ཛ་གྲྭ་ཧ་ཙནྡྲ་རུ་ཙི་རཾ་ཟླ་བའི་འོད་ལྡན་རྔ་ཡབ་ནི། །དེ་ཡི་ཚེམས་ཀྱི་འོད་རྒྱུན་གྱིས། །རྒྱས་པའི་ཉེ་བར་བསྟན་པ་བཞིན། །ཙ་མ་རཾ་ཙ་ཏུ་རཱ་ན་ན། །།ཙྪ་ཏྭཾ་ཤ་ཏ་ཤ་ལཱ་ཀཾ་ཀ་མ་ནཾ་ཀེ་ནཾ་ཀ་གདོང་བཞི་པ་ཡིས་བླངས་པར་གྱུར། ་། ཐུར་མ་བརྒྱ་ཡིས་མཚན་པའི་གདུགས། །མཚན་མེད་རི་དྭགས་པཱ་ཎུ་རཾ། །པྲ་སཱ་ད་མི་བ་སཱ་ཀཱ་ར་མཱ་ད་དེ་སྱ་པུ་རཾ་ད་ར། །ཤཾ་མཚན་ལྟར་དཀར། །དེ་ཡི་བཀའ་དྲིན་རྣམ་བཅས་བཞིན། །བརྒྱ་བྱིན་གྱིས་ནི་བླངས་པར་གྱུར། །གྲོང་ཀ་ཤུ་ན་ག་རོ་པཱནྟ་ཀཱ་ན་ནོ་དུམྦ་རཱི་ཏི་ཀེ། །ཨ་བ་ཏཱིཎཾ་སུ་ཀྲྀ་ཏྲི་ནསཱ་ཁྱེར་ཤང་ཀཱ་ཤུ་ཡི་འགྲམ། །ཨུ་དུམ་བཱ་ཅིརའི་ནགས་དྲུང་དུ། །བབས་པ་དེ་ལ་ལེགས་བྱས་ལྡན། །ཀུན་ནན་སྟི་བ་བནྡི་རི། །ཏསྨིནྣཏྲ་ལ་པཎཱ་ཏྱཱ་བྷིཀཱ་ཀཱི་ཛ་ན་སཾ་ག་མེ། །དགའ་བཅས་པས་ཕྱག་འཚལ་གྱུར། །སྐྱེ་བོའི་ཚོགས་དེར་དགེ་སློང་མ། །ཨུཏྤལ་མདོག་མ་ཞེས་པ་ཡིས། །ཨ་ལབྷཱ་བ་ས་རཱ་ཏིཧཱུྃ་ཚྭ་པ་རཱུ་པ་མུ་པ་ད་དེ། །པྲ་དཱིལྟ་རཏྣ་མུ་བལྟ་བའི་གནས་སྐབས་མ་ཐོབ་ནས། །མི་བདག་གཟུགས་ནི་ཉེ་བར་བླངས། །རིན་ཆེན་ཅོད་པན་རབ་ཀུ་ཊཾ་གཎྜ་ཏཱཎྜ་བི་ཀུཎྜ་ལཾ། །ཏྭཾ་བཱ་སྱ་ན་ཐཾ་རཱུ་པཾ་ཛ་གསལ་ཞིང་། །རྣ་རྒྱན་འགྲམ་པར་རབ་མཛེས་པ། །དེ་ཡི་གཟུགས་གསར་མཐོང་གྱུར་ནས། །གཙུག་ཧཱ་སོཙྪི་ཥ་པལྤ་བ། །སཱ་ཙིནྟ་ཡནྟི་བཱ་ཡཾ་ཛ་ནིནིཆི་བ་རསུ་ཏོར་ཡལ་འདབ་དགོད་པ་བཞིན། །དེ་ཡིས་བསམ་པ་ཤིན་ཏུ་ཡང་། །སྐྱེ་བོའི་མདུན་འདི་བར་མཚམས་ར༔༈། ཨནྟ་རཾ་ཀྲྀ་པ་རཱུ་པཾ་མེ་ཏྭཾ༔་དཱ་སྱ་ཏི་སཱ་ད་ར། །ཨ་མེད། །བདག་གི་རྒྱལ་པོའི་གཟུགས་མཐོང་ནས། །གུས་བཅས་སྐབས་ནི་ྋསྟེར་བར་འགྱུར། །འདི་ཏོ་ནྱ་ཐཱ་ཏུབྷ་ག་བཏ་པྲ་ཎ་ཏིརྨ་མ་དུརླ་བྷ། །ན་གུ་ཏཱཾ་གོ་ར་བ་ལས་གཞན་ཡང་བཅོམ་ལྡན་ལི། །ཕྱག་འཚལ་བདག་གིས་རྙེད་པར་དཀའ། །དབང་ཕྱུག་ལ་དགའ་སྐྱེ་།
