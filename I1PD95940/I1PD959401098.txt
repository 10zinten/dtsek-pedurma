བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེ་ཀྲྀཙྪེ་ཎ་ལ་བྱཱ་ཏེ། །ན་བྷཀྟ་བི་ར་ཧེ་བྷུཀྟ་སཱུཀྟི་ན་ཀེ་ན་ཙིཏ། །ནི་ཚོགས་ཆེན་དག་གིས་འཐོབ། །ཟས་དང་བྲལ་ན་སུ་ཞིག་གིས། །ལེགས་བཤད་དག་ནི་བཟའ་བར་ནུས། །ཨི་ཏི་ཐུ་ཏྭཱ་གྲོ་ཧ་པ་ཏིསྣེ་ཧ་དུལྟསྨྲ་ཧཱི་བྷུ་ཛཱ། །སཱུཀྟ་ལཱ་བྷེ་ན་མ་ཞེས་པ་ས་བདག་མཛའ་བ་ཡིས། །བརྗོད་པ་ཁྱིམ་བདག་གིས་ཐོས་ནས། །ལེགས་བཤད་རྙེད་པ་ཆེན་པོ་ཧ་ཏ་སུ་ཏུཥསྟ་མ་བྷཱ་ཥ་ཏ། །རཱ་ཛf་ཀྲི་ཡ་ཏེ་རནྣིརྨནྣ་རཀྵ་བི་ནཱ་ཤི་ཡིས༑། ཚིམ་པས་དེ་ལ་རབ་སྨྲས་པ། །གང་གིས་སྦྲུལ་བཞིན་སྐྱེས་བུ་ལ། །ཆགས་སྡང་དུག་ནི་སྐྱེད་བྷི༑། སནྡྷཱ་ཎཱ་མི་བ་ཡི༔པཾ་སྭཱ་རཱ་ག་དྭེ་ཥ་བི་ཥོ་ད་ཡ། །བི་མ་ཁྱོ་བྱེད་པ། །རྒྱལ་པོ་འབད་བསྲུངས་ཉམས་པ་ཡི། །རིན་ཆེན་དག་གིས་ཅི་ཞིག་བྱ། །ལེགས་བཤད་ནོར་བ་ཙི་ཏྟཾ་རུ་ཙི་ར་ཙ་ཡ་ཏསནྣཱརྒ་དཱི་པོ་པ་ཚྭ་བྷ་བྱཱ་ནཾ་ཏྲཾ་ད་ཡེ་སུ་བྷཱ་ཥི་ཏ་བུ་ཨེལེགས་ལྡན་རྣམས་ཀྱི་སྙིང་གར་ཡིབརྟན་པའི་ཡོན་ཏན་གྱིས་བརྟགས་ཤིང་། །དམ་པའི་ལམ་གྱི་སྒྲོན་མ་ཎེནཱ་དྷེ་གུ་ཎཱ་ལལླི་ན། །པཱ་ཀོཀྟིཙྪཱ་སུ་བཙྪཱ་སུནྡ་མ་དང་མཚུངས་དྲི་མེད་རྒྱས་པའི་འོད་ཟེར་སྤྲོ་བྱེད་པ། །འཚེད་པ་ལས་ཐར་གསེར་བཟང་མཛེས་པའི་ར་པ་ད་པཱསྟ་པྲ་བནྡྷསྦི་ཏིམཱུ་ལཾ་སཱ་ག་ར་མེ་ཁ་ལཱ་བ་སུ་མ་ཏི་པཱ་དཱི་ཤ་ཀོ་གནས་ལ་རབ་སྦྱར་གནས་པ་ཐོབ་པ་ཡི། །རིན་དེ་ནོར་གྱི་བློ་གྲོས་རྒྱ་མཚོའི་སྐེ་རགས་དང་ལྡན་རྐང་པའི་པྱ་བྱ་ཀཾ། །ཨི་ཏི་ཤྲཱི་མཛྲ་ཧ་པ་ཏིརྨི་ག་ཎྱ་ཀུ་ཤ་ལོ་རྱ་ཏ། །སུ་བཎྜ་ཆ་ལའང་ཆུང་། །ཞེས་ཏེ་དཔལ་ལྡན་ཁྱིམ་བདག་ནི། །དགེ་ལ་བརྩོན་ལྡན་ཕྱིར་ཐོས་ནས། །གསེར་གྱི་པ་ཏྲ་ལི་ཁི་ཏཾ་སཱུཏྟཾ་དིཀླུ་རྒྱ་བེ་ཤ་ཡཏ། །ཤུདྡྷོ་ད་ནོ་གཾ་ཧ་པ་ཏིསོ་ཧ་མེ་གླེགས་བུ་ལ་བྲིས་ཏེ། །ལེགས་བཤད་ཕྱོགས་རྣམས་དག་ཏུ་སྤྲིངས། །དེ་ཡི་ཚེ་ན་ང་ཉིད་ནི། །ཁྱིམ་བདག་བ་ཏ་དཱ་བྷ་བ། ་། སཱུཀྵེཔུ་ཙྩོ་པ་དེ་ཤཱཀྲི་སཧྥེ་ན་ནཱ་བི་པཱ་ད་ཟས་གཙང་དེར་གྱུར་ཏེ། །ལེགས་བཤད་བསོད་ནམས་ཉེར་སྟོན་པའི། །དོན་ལ་བདོག་ཀུན་བཏང་བས་ར༑། ཤཱ་རི་པུ་ཏྲ་པྱ་ཡཾ་བྷིཀྵསཱ་བྷུ་པ་ཏི་སུ་ཏཱ་བྷ་པཏ། །ཨི་ཏུ་བཱ་ཙཱ་ཁི་ལ་གུས༑། དགེ་སློང་ཤཱ་རིའི་བུ་འདི་ཡང་། །ས་བདག་བུ་མོ་དེར་གྱུར་ཏོ། །ཞེས་པ་དེ་བཞིན་གཤེགས་པ་།
