བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁིན་བ་ཤ་ཤེ་ཥ་མ་ནེ་ཀ་ཡུ་གོ་པ་མཾ། །ཨ་ཐ་སཾ་དྷཱ་བ་དྷུ་སཾ་གཱ་དི་བ་སཾ་ཀཱནྟ་ཀྱི་ཀུ་རྟོག་ཅིང་རང་གི་གནས་སུ་སོང་། །དེ་ནས་ཐུན་མཚམས་བུ་མོ་དང་། །འགྲོགས་པས་གུར་གུམ་འཕོས་པ་མ༈། ཛ་གཱ་མ་ག་ག་ནོ་ཏྱཱ་ནེ་ཏིགཽ་ཤུ་རཱ་ཊ་རཱ་ག་ཏྭཾ། །དི་ནཱནྟེ་པརྨི་བཞིན། །ནམ་མཁའ་ཡི་ནི་སྐྱེད་ཚལ་དུ། །ཉི་མ་ཆགས་རྟག་ལྡན་པར་གྱུར། །པདྨ་ཅན་དགའ་ཉིན་ནཱི་ཀཱནྟི་བི་ཤནྟེསྟ་ཏ་ཀྲྀ་ཏ་ར། །བཱ་ད་སཾ་བཱ་ཧ་ནཱ་སཀྵ་སནྣཱ་ཏ་སྙནྟི་ཀེ་མོའི་མཐར། །ནུབ་རིའི་ངོས་ལ་འཁོད་པ་ན། །དེ་ཡི་མཇུག་ཏུ་ཐུན་མཚམས་ནི། །རྐང་པ་བསྟེན་པ་ལ་བ་བྷོ། །ཏ་ཏ་པཉྩི་ཚྭ་བྷོ་དྷི་པྲ་བིབྦེ་བཱ་ས་རེ་ཤྭ་རེ། །ཁ་མ་ཆགས་མཛེས། །དེ་ནས་ནུབ་ཕྱོགས་ཆུ་གཏེར་དུ། །ཉིན་མོའི་དབང་ཕྱུག་རབ་ལྷུང་ཚེ། །དེ་ལས་པཱུརྨ་ཏ་ན་ཀ་ཏྲེསྟ་དུཏྠི་རི་བ་ཤཱི་ཀ་རེ། །ཤ་ནིསྟོ་ཀ་ཏ་འཐོར་བས་ཐིགས་པས་ཉེབཞིན། །སྐར་མ་རྣམས་ཀྱིས་ནམ་མཁའ་གང་། །དལ་གྱིས་མུན་པ་ཆུང་དུ་མཥཱ་མཱ་ནི་ཤ་བྷུ་བ་ན་བྷཱ་ཛ་ནཱཏ། །སནྡྷ་རཱ་གཱ་ས་ཁཾ་པཱི་ཏཱ་ཀཱི་རེ་གནག། མཚན་མོས་སྲིད་པའི་སྣོད་དག་ལས། །ཐུན་མཚམས་དམར་བའི་ཆང་འཐུངས་ནས། །མྱོས་བཱ་གྷཱཎྜི་ཏེ་ཀཱ་ཎཾ། །ཏ་ཏཔྲ་ཙཱི་ཤ་ཙཱི་ཀཱནྟ་བི་ལཱ་སི་བ་ས་ཏིཀ་ཀུ་པ། །པ་བཞིན་དུ་སྐད་ཅིག་འཁྱམས། །དེ་ནས་བདེ་སོགས་མཛེས་པ་ཡི། །རྩེ་དགའ་གནས་པ་ཤར་གྱི་ཕྱོགས། །ཨཱ་སནྣེརླུ་ད་ཡཱ་ལོ་ཀ་ཙན་ནེ་ཀིཾ་ཀི་ཏཱ་བ་བྷོ། །ཨ་ཐོདྨ་ཡོ་སུ་དྷཱ་སཱུ་ཏི་ཟླ་འཆར་སྣང་བ་ཉེ་བ་ན། །ཙནྡན་དག་གིས་མཚན་བཞིན་མཛེས། །དེ་ནས་བདུད་རྩི་སྐྱེད་པ་ནི། །ཤུ་མཱ་མུ་ཁ་བི་ཤེ་ཥ་ཀ། །བྷོ་གི་ཚྭ་བྷོ་ག་སཾ་བྷཱ་གྱ་ལཱི་ལཱ་མུ་ཁ་བི་ཤེ་ཥ་ལོངས་སྤྱོད་ཅན་གྱི་ལོངས་སྤྱོད་ཀྱི། །སྐལ་བཟང་རྩེ་དགའ་ཁྱད་པར་ཅན། །སྔོ་བསངས་གདོང་གི་ཁྱད་ཀ༈། བི་ལཱ་ས་ཧཱ་སཱ་བྷི་མུ་ཁཱི་མི་དོཀཱིཀྵ་ཀུ་མུདྲ་ཏཱི། །མ་ཅུ་ཅན་ཤར། །།ཟླ་བའི་རྩེ་དགར་མངོན་ཕྱོགས་པའི། །ཀུ་མུད་ལྡན་པ་མཐོང་གྱུར་ནས། །ཁྲོ་བས་ཞུམ་ལཱི་ནེ་བ་ན་ལི་ནཱི་ནི་མི་མཱི་ལ་ག་ལདྷུ་ཏི། །པྲ་ཏྱ་གྲ་ཙནྡྲ་ཏི་ལ་ཀཱ་ཏཱ་རཱ་ནི་བཞིན་པདྨ་ཅན། །གཟི་བྱིན་ཉམས་ཤིང་ཟུམ་པར་གྱུར། །ཟླ་བའི་ཐིག་ལེ་གསར་པ་དང་། །སྐར་མའི་།
