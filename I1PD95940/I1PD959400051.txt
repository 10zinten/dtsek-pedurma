ཅུ་སོ་ར་བྷ་ནིབྷ་རེ། །ར་མནྟེ་སུ་ཀྲྀ་ཏོདྷཱ་ནེ་ནནྡ་ནེ་པུ་ར་བཱ་སི་ན། །ནམས་དྲི་བཟང་རབ་ཏུ་རྒྱས། །ལེགས་བྱས་སྐྱེད་ཚལ་དགའ་སྟེར་དུ། །གྲོང་ཁྱེར་བ་རྣམས་རབ་ཏུ་རྩེན། །ཏ་ཏྲ་བྷུད་བྷཱུ་པ་ཏི་ཧྥུ་རི་གུ་ཎ་རཏྣ་མ་ཧོད་དྷི། །བྷུ་ཏིརྱ་ཤཤ་ཤཱི་ཀ་སྱ་དེར་ནི་ས་བདག་ཆུ་གཏེར་ཆེ། །རབ་མང་ཡོན་ཏན་རིན་ཆེན་དང༌། །གྲགས་པའི་རི་བོང་མཚན་མ་བསྐྱེད། །ཧེ་མ་ཙཱུ་ཌ་ཨི་ཏི་ཐུ་ཏ། །ས་དཱ་ས་དཱ་ཤ་ཡཱནྟེ་ཎ་ཀ་ལི་ཀཱ་ལཱ་པ་ཧཱ་གསེར་གྱི་གཙུག་ཕུད་ཅེས་གྲགས་བྱུང་། །རྟག་ཏུ་དམ་པས་བསྟེན་འོས་ཤིང་། །རྩོད་དུས་རབ་ཏུ་འཕྲོག་རི་ཎཱ། །ཀྲྀ་ཏཀྲྀ་ཏ་ཡུ་ག་ནི་བ་ཡེ་ནདྷརྨ་དྷ་རོ་ཛ་ན། །མ་ཧཱི་པ་ཏིཀྵ་བྱེད་པ། །རྫོགས་ལྡན་ལྟ་བུ་གང་གིས་ནི། །སྐྱེ་བོ་ཆོས་རྣམས་འཛིན་པར་བྱས། །ས་བདག་བཟོད་ལ་ཀུན་མ་སཀྟཤྲཱི་བྲྀ་ཏ་ཀ་རུ་ཎཱ་ར་ཏ། །བལྤ་བྷོ་བྷཱུཏ་པྲ་ཛཱ་ནཾ་ཡཔྲ་ཁྱུ་ཆགས་ཤིང་། །དཔལ་གྱིས་དཀྲིས་གྱུར་སྙིང་རྗེ་དགའ། །རབ་གྲགས་ལ་ཡང་དབང་ཐུལ་གང་། །སྐྱེ་རྒུ་ཏོ་པི་ཛི་ཏེནྡྲི་ཡ། །ཨ་མཱ་རཱ་ས་བ་སཾ་པཱུརྩྭ་མ་ཧི་སཱ་ས་ཏྲ་དཱིཀྱི་རྣམས་ཀྱི་དགའ་བྱེད་གྱུར། །འཚེ་མེད་མཆོག་་སྦྱིན་བསླབ་ལྡན་པ། །གང་གི་་བསོད་ནམས་མི་འཇིགས་ཏ༑། ད་དོ་ཡ༔སརྦ་བྷཱུ་ཏཱ་ནཾ་པུ་ཎྱ་མ་བྷ་ཡ་དཀི་ཎཾ། །ནིརྨ་དོ་ཡཔྲ་ཡོན༑། །མི་གསོད་སྒྲ་དབྱངས་ཡོངས་རྫོགས་པ། །འབྱུང་པོ་ཀུན་ལ་བྱིན་པར་གྱུར། །མཐུ་དང་ལྡན་ཡང་བྷཱ་བེ་པི་བི་བྷ་བེ་པི་དྲི་ཡཾ་བ་ད། །གྱུ་མཱ་ཤཱི་ལཔྲ་བྷཱུ་ཏྲེ་པི་ཡོ་བ་གང་མ་མྱོས། །རྣམ་པར་འབྱོར་ཡང་སྙན་པར་སྨྲ། །རྗེ་ཉིད་ཡིན་ཡང་བཟོད་ལྡན་ཚུལ། །ལང་ཚོ་ལྡན་ནི་པི་ཛི་ཏེནྡྲི་ཡ། །གཾ་བྷི་རེ་ཎོནྣ་ཏི་ག་ཏཱ་ཤུ་རེ་ཎ་ཤ་ཤཱི་ཀཱནྟི་ནཱ། །སཏྠཀེ་ཡང་དབང་པོ་ཐུལ། །མཐོན་པོར་གྱུར་པའི་ཟབ་མོ་དང༌། །ཟླ་བའི་འོད་ཟེར་དཔའ་བོ་དང༌། །ས་གཞི་ཎ་ཀྱི་ཏི་ཐཱ་ཏཱ་ཡེནཱ་ཏ་ནྱ་ཏ་བིསྨྲ་ཡ། །རཱ་ཛྷཏ་སྭཱ་དཱི་ཏཱི་ཡ་སྱ་བ་འཛིན་པ་དམ་པའི་ཕྱོགས། །གང་གི་ཡ་མཚན་བྱས་པར་གྱུར། །རྒྱལ་པོ་དེ་ལ་ཟླ་མེད་པའི། །རྒྱན་བྷུ་བཱ་བྷ་ར་ཎ་དྭ་ཡཾ། །ཏྭཱ་ག་པུཎཾ་ཙ་ཀཱ་རུ་ཙྩཾ་ཏཱ་རུ་ཙྩཾ་སུ་ཀྲྀ་ཏ་ཤི་ཡ། །ནི་གཉིས་ཤིག་བྱུང་གྱུར་ཏེ། །གཏོང་བ་རྫོགས་པའི་སྙིང་རྗེ་དང་། །ལེགས་བྱས་དཔལ་ནི་དར་བབ་པའོ། །—།
