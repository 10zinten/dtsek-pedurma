སུ་རེ་ཤྭ་ར། །པྲ་ས་ཏྱ་ཤྲཱི་པ་ཏི་ཡཱ་ཏཱ་དྲིཋ་ས་ཏྲེ་ཤ་ཏ་ཀྲ་ཏོ། །ཨཙིནྟ་རབ་ཏུ་བསྔགས། །བརྒྱ་བྱིན་སྙིང་སྟོབས་ལྡན་པ་ཡིས། །མི་བདག་ལ་ནི་དང་བྱས་སོ། །དེ་ཡིས་ཡ,་དྷཱ་སྭ་དཾ་བྷུ་པ་ཏིསྟརྒྱུ་བྷཱ་ཥི་ཏཾ། །ཤི་བིརྣ་རེ་ཤཱ་རསོ་ཧ་མ་ལེགས་བཤད་བདུད་རྩི་ཡི། །ཉམས་མྱོང་ས་ཡི་བདག་པོས་བསམས། །ང་ཉིད་སྔོན་གྱི་སྐྱེ་བ་ལ། །མི་བྷ་ཐཾ་པཱུཏྤ་ཛརྨ་ནི། །ཨི་ཏི་པྲ་ཎ་བ་ནེ་ནཱ་བྷུདྷཱལྤ་བྷོ་མེ་སུ་བྷཱ་ཥི་ཏཾ། །དབང་ཤི་བི་དེར་གྱུར་ནས། །དེ་ལྟར་སྲོག་གི་ཟོང་གིས་བདག། ལེགས་བཤད་ལ་ནི་དགའ་བར་གྱུར། །ཏ་ཐཱ་ག་ཏེ་བ་ཀ་ཐི་ཏཱ་ཛརྨནྡཱ་ར་ཀ་བྷཾ་ནི་ཛཾ། །ཨཱ་ཀནྡྷུ་བྷིཀྵ་བ༔སལྦེ་པྲ་རང་གི་སྐྱེ་བ་གཞན་གྱི་གཏམ། །དེ་བཞིན་གཤེགས་པས་གསུངས་པ་དག། ཐོས་ནས་ཡ་མཚན་ལྡན་པ་ཡི། །ཤཾ་སུས་བི་སྨྲ་ཡཱ། །དྲྀཥྛ་དྲྀཥཱ་སུ་ཁེ་སུ་བྷཱ་ཥི་ཏ་རུ་ཙིམནཱ༔པྲ་བྷཱ་དཱི་དགེ་སློང་ཀུན་གྱིས་རབ་ཏུ་བསྔགས། །མཐོང་དང་མ་མཐོང་བདེ་ལ་ལམ་ནི་ལེགས་བཤད་ཀྱིས་གསལ་སྒྲོན་པ་ཡཏ་སཾ་བཱ་དེ་ན་སུ་བྷཱ་ཏི་ཏཱ་མྲ་ཏ་ར་སཔཱི་ཏི་པ་རཾ་བརྟ་མ་ལྟ་བུར་སྤྱོད། །ལེགས་བཤད་བདུད་རྩིའི་རོ་ནི་ཉམས་མྱོང་ལྡན་པས་མཆོག་ཏུ་དགའ་བའི་ཆར་དག་ཏི༑། སརྦ་དྷོ་པི་སུ་བྷཱ་ཥི་ཏཱརྦ་བི་ཥ་ཡེ་བྱརྦ་ནི་ཡརྦ་ཚྭ་ཏསནྣཱསྟི་སུ་བྷཱ་ཥི་འབེབས། །ལེགས་བཤད་དོན་གཉེར་ཡུལ་ལ་གཉེར་བྱ་ཀུན་མཁྱེན་དག་ཀྱང་དོན་གཉེར་ཉིད་དུ་འགྱུར། །ཏེ་ན་ས་དྲྀ་ཤཾ་ཀིཾ་ཙིཏྠ་ཀཾ་བལྤ་བྷམ། །ཨི་ཏི་ཀྵེ་དེ་སླད་དམ་པ་རྣམས་ལ་ལེགས་བཤད་དང་མཚུངས་དགའ་བ་ཅི་ཡང་ཡོད་མ་ཡིན། །ཞེས་པ་དགེ་བའི་མེནྡ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་ས་ཏཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་ཤི་བི་སུ་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ཤི་བིའི་ལེགས་བྷཱ་ཥི་ཏཱ་བ་དཱ་ན་མེ་ཀ་ན་བ་ཏི་ཏ་མཔལླ་བ། །།།།བཤད་ཀྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་དགུ་བཅུ་རྩ་གཅིག་པའོ། །::གུ་རཱུ་ཎ་མ་སྱ་གྲེ་གུ་རུ་ཏ་ར་པ་དེ་ས་བ་མ་ཧ་ཏཱི་ས་དཱ་སདྡྷིཔཱུ་ཛྙཱ་བླ་མ་རྣམས་ཀྱི་མདུན་དག་ཏུ་ཡང་ཤིན་ཏུ་བླ་མའི་གནས་སུ་ཆེ། །འབྱུང་པོ་རྣམས་ནི་འཛིན་བྱེད་—།
