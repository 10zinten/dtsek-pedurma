ཤ་ལྱཱ་བིདྡྷ་ཨི་བ་ཀྱི་ཕྲོ་བྲིདྡྷ་བྷིཀྵ་ར་ཙིནྟ་ཡཏ། །ལཙྪཱ་ཡི་ཀི་མི་དཾ་བྷུཏྟཾ་མྱུར་དུ་ཟུག་རྔུས་ཕུག་པ་བཞིན། །དགེ་སློང་རྒན་པོས་རབ་བསམས་པ། །ངོ་ཚའི་སླད་དུ་སྡུག་བསྔལ་དུཁཱནྟ་མ་ཤ་ནམྨ་ཡཱ། །ཨ་པི་གཱ་ཐཱ་ཙ་ཏུབྷྱཱ་གཾ་ན་ཛཱ་ནཱ་མི་ནི་མཐའ། །ཟས་འདི་བདག་གིས་ཅི་སླད་ཟོས། །ཚིགས་སུ་བཅད་པ་བཞི་ཆ་ཡང༌། །བདག་གིས་མི་རཀྵ་ར། །ཀིཾ་ཀ་རོ་མི་ས་ཏྭཾ་མ་ཎཾ་ཡ་དི་པྲཙྪེནྣཾ་ཧཱི་པ་ཏི། །།ཏཀྟི་ཤེས་ཡི་གེ་མེད། །དམ་པའི་དབུས་སུ་ས་བདག་གིས། །གལ་ཏེ་དྲིས་ན་བདག་ཅི་བྱེད། །སྐྱེ་བོ་འགའ་བསྒྱུ་ཏེ་མཱ་མེ་ཥ་མཱུ་ཀཾ་ཧཱ་སྱ་ར་ཏིཛྫ་ན། །ཀཱི་ཊིཀོ་ཊ་ར་ཀཱ་རི་བྷིརྦ་གྷ་ཞིག་དགོད་ཅེའདི་དག། ལྐུགས་པ་བདག་ལ་ཅི་བརྗོད་འགྱུར། །སྲིན་བུ་ཁོང་སྟོང་བྱེད་པ་རྣམས་ཀྱིས་དལ་ཊི་ཏ་སྦནྡ་པྲ་བནྡྷཤ་ནི་རནྟསུསྟ་ཀྲྀ་ཤཱ་ནུ་དྷུ་མ་མ་ལི་ན་ཤྭ་བྷེ་པི་དྷ་གྱིས་སྡོང་པོའི་རྒྱུན་ནི་གཞིགས་གྱུར་ཅིང་། །ནང་དུ་མེ་དག་ཉལ་ཞིང་དུ་བས་དྲི་མར་གྱུར་པ་གཡང་སའི་རྒྱལྟ་རུ། །མཱུརྞཔཎྜི་ཏ་ཁཎྜི་ཏཱ་ན་ན་རུ་ཙིནྡི་ལཀྵ་ལཱི་ན་སྦི་ཏེ་མཱུ་ཀཱནྟ་པྲ་ཏི་ཤིང་ཡང་སླ། །མཁས་པ་བཞིན་རས་མཛེས་པས་བཅོམ་པའི་རྨོངས་པ་སྐྱེངས་ཤིང་ཞུམ་པས་གནས་གྱུར་པ། །མ༔པྲ་མཱ་ད་བ་པ་ཏི་མསྡུ་སཱ་པྲ་གྷ་ན། །ཨི་ཏི་ཙིནྟཱ་པ་རི་ལྐུགས་ཤིང་ལོང་བའི་གཟུགས་ཅན་གྱི་བག་མེད་དུ་གནས་སྐྱེ་བོ་བདག་འདྲར་མ་གྱུར་ཅིག། དེ་ལྟར་བསམས་ཙི་ཏེ་ཏ་ནི་ཤཱ་ས་ནིསུ་ཁཾ། །བུདྡྷ་པྲ་སཱ་དི་ནི་དེ་བཱི་མ་བྷེ་པ་ཡོངས་འདྲིས་ཤིང་། །ཚ་བའི་དབུགས་འབྱིན་བདེ་མེད་པ། །དེ་ལ་ལྷ་མོས་སངས་རྒྱས་ལ། །རབ་དང་ཏྱཱ་ཛ་གཱ་ད་ཏཾ། །པྲཀྵ་ཏི་ཏཱཾ་ཡ་དཱ་རཱ་ཛཱ་བཀྟ་ཧུཾ་བྷ་བ་ཏཱ་ཏ་དཱ། །བིསྟི་མངོན་དུ་ཕྱོགས་ཏེ་གནས། །གང་ཚེ་ཁྱོད་ལ་རྒྱལ་པོ་འདྲི། །དེ་ཚེ་ཁྱོད་ཀྱིས་འདི་བརྗོད་བྱ། །ཤིན་ཏུ་ཎྞ་ཏ་ར་དྷནྣཱ་སྱ་སཾཀེ་བཤུ་ཡ་ཏཱ་མི་ཏི། །ཀིཾ་ཙིནྟཱ་ཏྲ་དྷ་ན་མུ་རྒྱ་ཆེ་བདེན་པའི་ཆོས། །མདོར་བསྡུས་པ་དག་ཉན་ཅེཔར་མཛོད། །ནོར་ནི་ཅུང་ཟད་ཙམ་ཞིག་ཕན་པ་ས་ཀྲྀ་ཏོ་ཀྲྀ་ཏྱཱ་མ་ཏམྦྷ་མེ་བ་པཱ་ཎཱ་དྷཱ་རཱ་ཏ་ནུ་ཏ་ར་མ་ཏི་སྭཱ་ད་ནིརྨུལྟ་མནྣཾ། །བྱེད་ཅིང་བྱ་བ་ཤིན་ཏུ་ཉུང་བ་ཉིད། །སྲོག་གི་ཟུངས་ནི་ཤིན་ཏུ་ཆུང་ཞིང་རོ་དང་བྲལ་བའི་ཟས་ཀྱིས་ཀྱང་། །—།
