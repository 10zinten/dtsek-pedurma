ཎ་པྲ་བཏྟནྟེ་བྷ་བདྭེ་དྷཱ། །ཡ་ཐི་བཱ་ཡཾ་ཏ་ཐི་བཱ་ཧཾ་ཀོ་བི་ཤེ་ཁས་བཱ་བ་ཡོ། །སྒྱུ་ལྡན་ཆགས་པས་འཇུག་པ་མིན། །ཇི་ལྟར་འདི་ཉིད་དེ་བཞིན་བདག། ཁྱེད་སྐྱེལ་ངེད་གཉིས་བྱེ་བྲག་ཅི། །སརྦ་བྷཱུ་ཏ་ས་མ༔སརྙཀྲྀ་སནྣི་ཀ་ཏྲ་ཀུཉྫ་ཏེ། །ཨི་ཏུལྟེ་ཏེ་ན་ཚྭ་པ་ཏིརཱི་དམ་པ་འབྱུང་པོ་ཀུན་ལ་སྙོམས། །གཅིག་ལ་བརྩེ་བ་བྱེད་པ་མིན། །དེས་འདི་སྨྲས་ཚེ་མི་བདག་གི། ནཾ་ཀ་ཀ་ན་རཱ་བཱི་ཎཱ། ་། ནེ་བྷ་ཏ་བྱ་མི་ཏཱི་བཱ་ཧ་ཁ་ཀཾ་པྲཙྪ་བྱ་པཱ་ཎི་ལག་པ་གདུབ་བུའི་སྒྲ་ལྡན་པས། །འཇིགས་པར་མི་བྱ་ཞེས་སྨྲ་བཞིན། །བྱ་ནི་ཕྱག་གིས་རབ་བསྒྲིབས་ནཱ༑། ས་ཏ་ཏསྡིགྷ་ཛཱི་མཱུ་ཏ་གྷོ་ཥ་གམྦྷི་ར་ཡ་མི་བ། །ཨུ་བཱ་ཙ་སརྦ་ཏེ། །དེ་ནས་སེམས་ཅན་གཟིར་བ་ཀུན། །ཡོངས་བསྐྱབ་དུས་ནི་བྱས་པ་ཡི། །ཆར་སྤྲིན་འཇམ་པའི་སཏྭཱནྟི་པ་རི་ཏཱ་ཎ་ཀྲྀ་ཏ་གྷ་ཎ། །མ་ཀྲྀ་ཐ་ཎ་ཊཾ་སྐྱནི་བི་ཥ་མཾ་སྒྲ་དབྱངས་ལྟར། །ཟབ་པའི་ཚིག་གིས་རབ་སྨྲས་པ། །སྐད་ཅིག་ཚིམ་པ་དོན་གཉེར་བས། །མི་བཟད་པྲ་ཎི་བི་ཤ་མ། །ཏུ་མྱ་བྱ་ཐཱ་བི་ཀཱ་རོ་ཡཾ་པཱ་ཎ་སྣེ་ཧཤ་རཱི་རི་ཎཾ། །སྲོག་ཆགས་གསོད་མ་བྱེད། །ལུས་ཅན་སྲོག་ལ་ཆགས་པ་ཡི། །གདུང་བའི་རྣམ་འགྱུར་མཚུངས་པ་ཉིད། །པ་ར་སྭཱ་ཎཱ་པ་ཧ་རེ་ཎ་ཡཱ་བྲྀཏྟིཔ་རི་ཀལྤ་ཏེ། །ནི་བྲྀཏྟི༔བེ་ཡ་སཾ་མ་གཞན་གྱི་སྲོག་ནི་ཕྲོགས་པ་ཡིས། །གང་ཞིག་ཡོངས་སུ་འཚོ་བྱེད་པ། །དེ་ཡི་དགེ་ལེགས་ལྡོག་འགྱུར་ཏེ། །ཧི་པཱ་པ་ཏཱ་པཾ་པྲ་ཡཙྪ་ཏེ། །ཨ་དྷུ་ནི་ཝཱ་པ་རིཙྪིནྡྷེ་ནི་ཛེཙྪ་སམྨ་ཏཱ་ད་རཾ། །གང་ཕྱིར་སྡིག་པའི་གདུང་བ་སྟེར། །ད་ལྟ་ཉིད་དུ་མ་ཆད་པར། །རང་འདོད་ཡང་དག་བཞེད་ཅིང་གུས། །གྲོ་བྱ་ཏཱ་མ་ཤ་ནཾ་ཡརླརྨ་དརྦ་མུ་པ་ཀམྤི་ཏཾ། །ཐུ་ཏྲེ་ཏི་རཱ་ཛ་བ་ཙ་ནཾ་བདག་གི་དོན་དུ་ཟས་གང་གང་། །ཉེ་བར་བསྒྲུབས་པ་བླང་བར་མཛོད། །ཅེས་པ་རྒྱལ་པོའི་ཚིག་ཐོས་པ་རི་མ་ནཱ་ན་ནཱཤཱ་པ་ན། །བ་རོ་བ་དྷོ་ག་བི་མ་ནཱ༔པྲ་ཏུ་ནས། །ཡོངས་སུ་བཞིན་ལོག་ཤུགས་རིངས་ཅན། །ལོངས་སྤྱོད་མཆོག་ལ་ཡིད་བྱུང་བའི། །རྔོན་པས་བྷཱ་ཥ་ཏ་ལུབྷཱ་ཀ། །ན་པ་ཡཾ་རཱ་ཛ་བྷོ་གཱ་ནཾ་ར་སཱ་ཛྙཱ་བ་ན་བཱ་སི་ན། །ལན་ནི་རབ་སྨྲས་པ། །བདག་ཅག་ནགས་ན་གནས་རྣམས་ཀྱིས། །རྒྱལ་པོའི་ལོངས་སྤྱོད་རོ་མི་ཤེས། །—།
