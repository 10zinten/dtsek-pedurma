ཚྭ་པདྨ་ཛ། །ཡསྨི་པུ་རཱ་ཏྭཾ་གུ་རུ་ཎཱ་བ་ཙ་སཱ་པྲ་ཏི་ཡ་དི་ཏཱ། །ས་ཨེ་བཱ་ཧཾ་ཡ་བྱིན་གྱུར་པ། །སྐལ་བ་བཟང་མོ་རྒྱལ་པོའི་བུ། །དགེ་བྱེད་དེ་ནི་ང་ཉིད་ཡིན། །དེ་ཉིད་བདག་ཡིན་མིག་དི་པ་རཾ་ནིཤྲཱི་རཿཔཱ་ཊ་ནེ་དྲི་ཤོ། །ཏེ་ན་ས་ཏྲེ་ན་ན་ཡ་ནཾ་སྭ་སྦེ་བྷ་བ་ཏུ་ཕྱུང་ལ། །གལ་ཏེ་ངེས་པར་ཁོན་མེད་ན། །བདེན་པ་དེ་ཡིས་བདག་གི་མིག། གཞན་པ་རྣལ་དུ་གནས་མེཉཔ་རཾ། །ས་ཏོ་པ་ཡཱ་ཙ་ནེ་ནེ་ཏི་ས་ཧ་སི་བཱ་སྱ་ལོ་ཙ་ནཾ། །དྲི་ཏཱི་ཡ་མ་པི་བི་གྱུར་ཅིག། ཅེས་པའི་བདེན་པ་བརྗོད་པ་ལས། །འཕྲལ་ལ་ཉིད་དུ་དེ་ཡི་མིག། གཉིས་པའང་སེམས་དང་མ་མྱ་མ་པཱལྟ་ས་ཧ་ཙེ་ཏ་སཱ། །ཏ་ཏོ་བི་དི་ཏ་བྲྀཏྟེ་ན་པུ་ཅུ་སེ་ལྷན་ཅིག་ཏུ། །དྲི་མ་དག་དང་བྲལ་བ་ཐོབ། །དེ་ནས་སྤྱོད་ཚུལ་རིག་པ་ཡིས། །ས་ལ་སྤྱོད་པ་ནེ་ན་བྷུ་བྷུ་ཛཱ། །ཀྲྀ་ཏ་སྭཱ་ཧཱ་ཡ་ཀཿཔྲ་ས་སཱ་རཱ་ཛྲ་ས་པྲ་ཡཱ་ས་ཁ། །ཀ་བསོད་ནམས་སྡེས། །སྟོང་གྲོགས་བྱས་ཏེ་རང་གིས་ནི། །རྒྱལ་སྲིད་མཛེས་མ་ལྡན་དེས་ཐོབ། །གང་ལྟ་ཎ་ཀཱ་རཱི་ཡསོ་ཧཾ་དེ་བ་དཏྟཱས་ཙཱ་ནུ་ཛ། །ཏེ་ན་པཱུསྠཱ་ནུ་བྷཱ་བེ་ན་ཏདྲི་དྷོ་ཨ་ཞིག་དགེ་བྱེད་དེ་ནི་ང་། །ནུ་བོ་དེ་ཡང་ལྷ་སྦྱིན་ཏེ། །སྔོན་གྱི་མཐུ་དེས་ད་ལྟ་ཡང་། །རྣམ་པ་དག་ནི་ཏུ་པི་བརྟུ་ཏེ། །ཨི་ཏུ་དཱ་ར་མུ་པ་ཀཱ་ར་ནིརྨ་ལཾ་བོ་དྷི་ས་ཏྭ་ཙ་རི་ཏཾ་ནི་ཤམྱ་ཏེ། །དེ་ལྟར་གྱུར། །དེ་ལྟར་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་ནི། །དྲི་མེད་རྒྱ་ཆེར་ཕན་དང་མི་བསྲུན་པའི། །བྷིཀྵ་བ༔ཁ་ལ་བི་ཙེཋ་ཏཉྩ་ཏཧཱུ་ལྱཱ་མ་པྲ་ཏི་མ་བིསྨྲ་ཡཾ་ཡ་ཡུ། །ཨི་ཏི་སྤྱོད་ཚུལ་ངན་པ་ཐོས་ནས་དགེ་སློང་རྣམས། །དེ་ལ་མཚུངས་པར་མཚུངས་མེད་ཡ་མཚན་གྱུར། །ཅེས་པ་ཀྵེ་མེམྲ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་ཀ་ལྱ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་དགེ་ཎ་ཀཱ་རྩ་བ་དཱ་ནཾ་དྲྭ་ཚིགས་ཤཔལྤ་ཝ། །།།།བྱེད་ཀྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་སུམ་ཅུ་རྩ་གཉིས་པའོ། །།་་བཱ་མཱསཛྫ་ན་བཱ་མཱབཱ་ཡེ་ཎ་བྷ་བནྟི་ནི་ཙ་རཱ་གི་ཅུ། །སྐྱེ་བོ་དམ་པའི་@གཡོན་ཕྱོགས་གཡོན་པ་རྣམས། །ཕལ་ཆེར་དམན་ལ་ཆགས་དང་ལྡན་པར་གྱུར། །—།
