རྒྱལ་སི་ཏི་ཡཀཱ་ཡ་ཤ་ཏེ་ཥཱ་པཾ་སཀསྠཱ་རི་ཀཱ་མོད་ཨི་དཾ་ཤུ་ཀཱ་སྱ། །ཛི་ན་སཏྠ་ནམས་ཉེ་བར་རྫོགས་པའི་མཐུ་ཆེན་གང་། །ཅི་ཡང་དེ་ནི་ལུས་བརྒྱར་མེད་མི་འགྱུར། །གྲོང་ཁྱེར་རྒྱལ་པོའི་ཛ་གདྡཱནྟི་སནྟཱ་པ་ཀ་རུ་ཎཱ་ཧྲ་དེ། །པུ་རེ་རཱ་ཛ་གཾ་ཧེ་བེ་ཅུ་བ་ནཱནྟ་ར་ཁབ་ཀྱི་ནི། །འོད་མའི་ཚལ་གྱི་ནགས་ནང་ན། །སྲོག་ཆགས་འགྲོ་ཀུན་ཡོངས་གདུང་ལ། །རྒྱལ་བ་ཐུགས་བི་ཧཱ་རི་ཎི། །ཤྲཱི་མཱ་ན་ཧྥུ་ད་བྷུ་པ་ཏིཉྩམཱ་ཡཾ་པོ་ཏ་ལཱ་བྷི་དྷ། །ཛ་རྗེའི་མཚོ་བཞུགས་ཚེ། །གང་གི་ནོར་གྱིས་ནོར་སྦྱིན་ནི། །ནོར་གྱི་དྲེགས་པས་ལོང་བ་འཕྲོག། དཔལ་ཧཱ་ར་དྷ་ན་དནྣཱཙྩཾ་དྷ་ནེ་ན་དྷ་མ་ད་སྐྱ་ཡ། །ཨ་ཛཱ་ཡ་ཏ་སུ་ཏསྟ་སྱ་མ་ནོ་ར་ལྡན་གྲུ་འཛིན་ཞེས་བྱ་བ། །ས་བདག་ཙམ་པར་བྱུང་བར་གྱུར། །དེ་ཡི་བུ་ནི་བཙས་གྱུར་པ། །ཡིད་ལ་ཐ་ཤ་ཏཱ་ནཱི་ཏ། །དྷ་ནཱ་ནཱ་མཱིསིཏཱ་བཱསྱཱ་སུ་ཁ་ས་ཁྱོ་བི་བྷཱུ་ཏ་ཡ། །ཐ་རེ་བ་བརྒྱ་ལྡན་པའོ། །ནོར་རྣམས་འདོད་པས་ཐོབ་པ་ཡི༄༅། །རྣམ་པར་འབྱོར་པ་བདེ་བའི་གྲོགས། །གྲོ་བ་ཎཱ་ཀཱ་ཎེ་ཛཱ་ཏ་སྱ་ས་སཱུ་ནོ་སྟ་སྱ་ཛརྨ་ནི། །པཱི་ཏྱཱ་ད་དོ་ད་རི་དྲ་སྱ་བཞིན་དག་ལས་སྐྱེས་གྱུར་ཞིངེ། །བུ་སྐྱེས་པ་ཡི་ཚེ་ན་དེ༄༅། །དགའ་བས་བྱེ་བ་ཉི་ཤུ་ནི། །ཀོ་ཊཱི་ནཱ་བི་ཤ་ཏི་ཡ་དཱ། །ས་ཐོ་ཎ་ཀོ་ཊི་བི་ཤཱ་ཁྱཤི་ཤུ་ཁྭ་ཏསྟ་དཱ་བྷ་བཏ། །བི་བྷ་དབུལ་པོ་ལ་བྱིན་གང་གི་ཚེ། །གྲོ་བཞིན་བྱེ་བ་ཉི་ཤུ་ཞེས། །དེ་ཚེ་བྱིས་པ་དེ་གྲགས་གྱུར། །གང་གི་བ༔སུ་ཀྲྀ་ཏེ་ནི་བ་ཡེ་ན་ཁཾ་ཤ་བི་བྷུ་ཥི་ཏ། །པཱཥ་བི་དྱས་སནྣཱརིགས་ནི་རྣམ་བརྒྱན་པ། །ལེགས་བྱས་ཀྱིས་ནི་འབྱོར་པ་བཞིན། །ནོར་རྣམས་ཀུན་དང་ལྷན་ཅིག་ཏུ། །ས་ཧ་ཧྲིདྡྷི་མུ་པ་གཱ་ཏ། །ཙ་ཀཱ་ར་སུ་ཁ་བི་ཤཱནྟི་བྱ་བ་ཧཱ་ར་དྷཱ་རཔི་ཏུ། །རིག་པ་ཐོབ་ཅིང་འཕེལ་བར་གྱུར། །ཕ་ཡི་ལུགས་རྣམས་བཟུང་བ་ཡིས། །བདེ་བ་ཉིད་དུ་ངལ་གསོ་བྱས། །ས་ཀ་དཱ་ཙིཏ་པྲ་བྷཱ་པུཎྜ་མི་བ་སཱུརྱ་སྱ་མཎྜ་ལཱཏ། །ཨ་བ་ཏཱིཙྩཱ་པུ་ར་པྲ་སྟི་ནམ་ཞིག་ཉི་མའི་དཀྱིལ་འཁོར་ལས། །ཉི་མའི་ཚོགས་བཞིན་མདུན་དུ་ནི། །་མོ་གལ་གྱི་བུ་བབས་མོངྒ་ལྱཱ་ཡ་ན་མ་བྲ་བཱིཏ། །ཀོ་བྷ་བཱ་ནརྐ་སཾ་ཀཱ་ཤཔྲ་ཀཱ་ཤི་ཏ་དི་གནྟ་རཿ། །ནས༑། འོངས་པ་ལ་ནི་དེ་ཡིས་སྨྲས། །ཕྱོགས་མཚམས་རབ་ཏུ་གསལ་བྱེད་པ། །ཉི་མ་འདྲ་བ་ཁྱོད་ནི་སུ། །—།
