དྷི་ན། །ཏ་ཏྲ་ཏེ་ན་ཀྲྀ་ཏཱ་ཏི་ཐུ་ནི་བི་རཱུསྨི་ཏ་དདྡྷཱ་ཏཾ། །ཨ་ནཱ་ཐ་པཎྜི་དཾ་ཛ྄ཧཱུྃ་པར་ཕྱིན། །དེས་དེར་མགྲོན་གྱི་བྱ་བ་བྱས། །དེ་ལ་རྨད་བྱུང་དེ་དག་བཤད། །མགོན་མེད་ཟས་སྦྱིན་བལྟ་ཏེ་ནི་བ་ས་ཧི་ཏ་ཡ་ཡུ། །ཐ་བཙྪཱནྟེ་ས་མཱ་སཱ་བྱ་པཱུ་ཛི་ཏཱསྟེ་བའི་སླད། །དེ་དང་ལྷན་ཅིག་ཉིད་དུ་སོང་། །མཉན་ཡོད་དུ་ནི་ཡང་དག་ཕྱིན། །དེ་རྣམས་དེ་ཡིས་ན་ས་ད་རཾ། །ཨསྨི་རྒྱ་བེ་ད་ཡན་སཐཾ་ཡ་ཐཱ་པྲྀཏྟཾ་ཡ་ཐཱ་ཐུ་ཏཾ། །ས་ཏཱ་ན་གུས་བཅས་མཆོད། །དེ་ལ་ཇི་ལྟར་མཐོང་བ་དང་། །ཇི་ལྟར་ཐོས་པ་ཐམས་ཅད་བཤད། །བརྟུལ་ཞུགས་པྲ་ཏཟི་ནསརཱན་སུ་ཏྲཾ་དཾ་སུ་དྷ་ནཉྩ་ཏཾ། །ནི་ནཱ་ཡ་དྷརྨ་ས་བི་ཙདོན་གཉེར་དེ་ཀུན་དང་། །གྲོགས་པོ་ནོར་བཟང་དེ་ཡང་ནི། །ཆོས་ཀྱི་གྲོགས་དང་མཛའ་བ་ཡིས། །བཱི་ཏི་བྷ་ག་བ་ཏོནྟི་ཀཾ། །བྷ་ག་བཱ་ན་པི་ཏདྷཱ་ཀཱཙྩཀྲེ་ཏ་ཥཱ་མ་ནུ་གྲ་ཧཾ། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་ཁྲིད། །དེ་ཡི་ཚིག་གིས་དེ་དག་རྣམས། །བཅོམ་ལྡན་གྱིས་ཀྱང་རྗེས་སུ་བཟུང་། །ས་ཏྱ་དན་ན་སཾ་བུདྡྷ་ཡེ་ན་ཏེ་སུ་ག་ཏི་ཡ་ཡུ། །ཏེ་ཥཱ་ཡཱ་ཏེ་ཥཱ་གང་གིས་དེ་དག་བདེན་པ་མཐོང་། །ཡང་དག་རྟོགས་ཤིང་བདེར་བགྲོད་གྱུར། །དེ་དག་སོང་ཚེ་བཅོམ་སུ་དྷ་ནཾ་པཀྵ་པཱ་ཏཱརྔ་ཡ་དྲྀ་ཤ། བི་ལོ་ཀྱ་བྷ་ག་བཱན་སམྱཀ་བི་ལྡན་གྱིས། །ཕྱོགས་ལྷུང་བརླན་པའི་སྤྱན་གྱིས་ནི། །ནོར་བཟང་ཉིད་ལ་རྣམ་གཟིགས་ནས། །ཡང་དག་ད་དྷེ་ཛྙཱ་ན་དྷཱ་ཛ་ནཾ། །ས་ཏྱ་སཾ་དན་ནཱ་བཱཥ་བི་ཤེ་ཥ་ཀུ་ཤ་ལོ་ད་ཡ། །ག་ཏྭཱ་ཡེ་ཤེས་ལྡན་པར་མཛད། །བདེན་པ་ཡང་དག་མཐོང་བ་ཐོབ། །ཁྱད་པར་དགེ་བ་རྒྱས་པ་དེ། །སོང་ནས་ཛི་ནཱ་ཡ་ཀོ་ཧཱུྃ་ཧཱུཾ་ས་བི་ཧཱ་ར་མ་ཀཱ་ར་ཡཏ། །ཡསྣདྡྷ་ག་བ་ཏཱ་དིཥཱ་རྒྱལ་བའི་གཙུག་ལག་ཁང་། །ཀོའུ་ཤཱམ་བཱིར་བྱེད་དུ་བཅུག། གང་ཕྱིར་བཅོམ་ལྡན་གྱིས་བསྐོས་པས། །ཏ་སྨིན་ཡཱ་ཏས་ཧཱ་ཡ་ཏྭཾ། །བྷིཀྵཧྲུན་བྷི་དྷསྟ་སྨཱཏ་སོ་བྷུཙྩནྡ་བི་ཧཱ་ར་དེར་ནི་གྲོགས་ལ་འོངས་གྱུར་པ། །དགེ་སློང་སྐུལ་བྱེད་དེ་ཕྱིན་དེས། །སྐུལ་བྱེད་གཙུག་ལག་ཁང་ཞེས་བྷུ༑། རཱ་དྷཱ་བྷི་དྷཱ་ད་དངྒཱ་སཱི་བི་ཧཱ་ར་པ་རི་ཙཱ་ར་ཀཱ། །ད་ཡ་ཡཱ་བྷ་ག་གྱུར། །གཙུག་ལག་ཁང་སྐྱོང་འབངས་མོ་ནི། །མགུ་བྱེད་ཅེས་པས་ཕུལ་བ་ཡི། །གོས་ཧྲུལ་དེ་ལ་བརྩེ་།
