ཡཱ་ཏིཙཱ་ར་ཐ་མཱ་རོ་སྱ་གྷཌཱ་ཤནྡ་ད་བྷི་དུ་ཏཱན། །ས་ཧ་ཏྭཱ་བཱི་ར་པུ་རུ་དེ་ནི་ཤིང་རྟ་ལ་བསྐྱོན་སོང༌། །དྲིལ་སྒྲ་ལས་ནི་མངོན་འདེད་པའི། །སྐྱེས་བུ་དཔའ་རྣམས་དེས་བཅོམ་ཥཱ་ན་བཱ་པ་ཚྭ་པ་ཏེ༔པུ་རཾ། །ཏ་མཱུ་ཙ་དེ་བ་ཀ་ནྱ་ཡཾ་པྲསྨཱ་བི་ནི་མ་ནས། །མི་ཡི་བདག་པོའི་གྲོང་ཁྱེར་ཕྱིན། །དེ་ལ་དེས་སྨྲས་མཛེས་མ་འདི། །བདག་གི་བསམ་ཡཱཏྨ་ཡཱ། །པྲ་ཏྲི་ཧཾ་ཏྲཱ་སུ་ཏོ་རྒྱ་སཱུ་ནི་མིཏྟ་ཛྙཱ་ན་མཱུ་ཙི་ཏ། །ཏསྣ་དེ་ཥ་པ་མིན་པ་ཐོབ། །འདི་ཡི་བུ་ཡིས་ཕ་གསོད་ཅེས། །གང་ཕྱིར་མཚན་མཁན་རྣམས་ཀྱིས་སྨྲས། །དེ་སླད་མི་ཡི་ན་ར་པ་ཏེ་མ་ཧི་ཥཱི་ན་ཏ་བོ་ཙི་ཏཱ། །ཏྭཱ་ཡི་ཛཱི་བ་ཏི་ཛཱི་བནྟི་པྲ་ཛཱ་ནཾ་སཏྠ་བདག་པོ་ནི། །ཁྱོད་ཀྱི་བཙུན་མོར་འོས་པ་མིན། །ཁྱོད་ནི་འཚོ་ན་སྐྱེ་རྒུ་ཡི། །ཕུན་སུམ་ཚོགས་པ་སཾ་པ་ད། །ཨི་ཏུཏྟསྟེ་ན་ཏྭཾ་པྲབྷཱ་ཏུཀྟ་ནི་བ་ཤ་ཤཱ་ཀ་ས། །ནི་རུདྡྷ་ཐམས་ཅད་འཚོ། །དེ་ཡིས་འདི་བརྗོད་ཚེ་ན་དེས། །དེ་མཐོང་གཏང་བར་ཡོངས་ྋམ་ནུས། །དེ་གདོང་ཀརྨ་ན་བྱེ་བ་ཏམྦུ་ཁཱ་ལེ་ཁྱ་ལེ་ཁ་ཡཱ། །སོ་བ་ད་ཏ་ཀཱ་ཀ་དཱ་ཛྲཟླཔུ་ཏྲེ་ཎ་ལ་བྲིས་རི་མོ་ཡིས། །ལས་ཀྱི་སླད་ཉིད་བཅིངས་པ་བཞིན། །དེ་ཡིས་སྨྲས་པ་བུ་ཡི་ཕ། །གསོད་པ་གང་ནི་ཧ་ཏཔི་ཏཱ། །སྭ་ཡཾ་མ་ཡཱ་བྷི་ཥིཀྟ་བྱསུ་ཏོ་ཡ་དི་བྷ་བི་ཥྛ་ཏི། །ཨི་ཏུཏྟཱ་དུ་གང་ཚེ་མཐོང་། །གལ་ཏེ་བུ་ནི་བྱུང་གྱུར་ན། །རང་ཉིད་བདག་གིས་དབང་བསྐུར་བྱ། །དེ་སྐད་ཚྭ་པ་ཏིཀ་རྩྭ་ས་རི་ཎཱི་ཡཱ་བྷ་བཏ་མུ་ཁཱི། །ཀྲྀ་ཏཀསྨོརྨི་ནི་རྨ་ཎིབརྗོད་ནས་མི་ཡི་བདག། མཛེས་མ་ལེན་ལ་དགའ་བར་གྱུར། །ལས་ཀྱི་རླབས་ཀྱིས་སྤྲུལ་པ་ཡིས། །པྲ་བྷ་བནྟི་ན་བུདྡྷ་ཡ། །བྷོ་གི་ནལྟ་སྱ་ཀཱ་ལེ་ན་ཏ་ཙཱ་སུ་ནུ་ར་ཛ་ཡ་ཏ། །ཕྲོགས་པའི་བློ་ལ་མཐུ་ཡོད་མིན། །དུས་ཀྱིས་ལོངས་སྤྱོད་ལྡན་པ་དེའི། །བུ་ནི་དེ་ལས་སྐྱེས་གྱུར་ཏེ། །ངོ་ཏིཥ་ཙ་རི་ཏེ་ཡ་སྱ་བྲྀཏྟཾ་མུཀྟ་པི་ཊཾ་དུ་ཧ། །ཏ་པོ་པ་ན་མྲྀ་ག་གྷ་གང་གི་སྤྱོད་ཚུལ་བརྩིས་པ་ཡིས། །ཕ་ནི་གསོད་པར་བརྗོད་གྱུར་པའོ། །དཀའ་ཐུབ་ནགས་ཀྱི་རི་དྭགས་ན་ཚིགས་ག་ཡ་བྱ་ས་ནེ་བ་ནེ། །ཨེ་བཾ་བི་དྷོ་རྒྱ་བྷཱུཏྟ་སྱ་མུ་ནི་ཤ་པསུ་རྣམས༑། བསད་ཕྱིར་ནགས་སུ་རྔོན་ཉེལ་ཞེན། །དེ་ལ་ཐུབ་པའི་དམོད་པ་ནི། །བུ་ཡི་དབྱིབས་ཅན་།
