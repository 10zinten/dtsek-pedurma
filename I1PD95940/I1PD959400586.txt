ལོར་བེ་རྨི་པྲ་ཎ་ཏོཏསྦྲ་ཏུཔུ་ར། །བྷ་ག་བཱ་ན་པི་སརྦ་ཛྷསྟ་ཡོཛྙཱ་པོ་ནི། །ཕུལ་ནས་མདུན་དུ་བཏུད་དེ་གནས། །བཅོམ་ལྡན་ཐམས་ཅད་མཁྱེན་པས་ཀྱང་། །དེ་དག་ཏྭཱ་མ་ནོ་ར་ཐཾ། །ད་ཡ་ཡཱ་ཙ་ར་ཎ་སྤནྣཾ་བི་ད་དྷེ་དཱ་རུ་དནྟི་ན། །སམྱཀྱིཏྟ་བསམ་པ་མཁྱེན་གྱུར་ནས། །བརྩེ་བས་ཤིང་གི་གླང་པོ་ལ། །ཞབས་པད་དག་གིས་རག་པར་མཛད། །དེ་པྲ་སཱ་དེ་ན་དྲྀཏྠོ་བྷ་ག་བ་ཏོཉཐ་ཏོ། །།པྲ་ཎི་དྷ་ནི་བི་བཱ་ཧཱ་ནས་དེ་དག་བཅོམ་ལྡན་གྱིས། །ཡང་དག་དང་བའི་ཐུགས་ཀྱིས་གཟིགས། །ཕན་ཚུན་ཁྱེའུ་བུ་མོ་དག། ཡ་ཙཀྲ་ཏུརྭ་ར་ཀོ་མི་ཐ། །ཀུ་ལ་པྲ་བྷཱ་བ་བི་བྷ་བིནྡྷཱ་ཡཱཛྫརྨ་མ་ཡོ་ཙི་ཏ། །ཁྱོ་ཤུག་ཏུ་ནི་སྨོན་ལམ་བྱས། །གསེར་གྱི་གླང་པོའི་བཞོན་པ་དང་། །རིགས་དང་མཐུ་དང་འབྱོར་པ་ཡིས། །བཱ་ཧ་ནཾ་ཧེ་མ་དནྟི་ཙ་ཀུ་མཱ་ར་སྱེ་རྱ་བྷཱུཏྠ་ཏི། །དྲྀབྷཱ་བྷ་ག་བ་ཏཀ་ནྱ་བདག་གི་སྐྱེ་བ་འོས་གྱུར་ཅིག། ཅེས་པར་གཞོན་ནུའི་བློ་གྲོས་གྱུར། །བུ་མོས་བཅོམ་ལྡན་ཆོས་གོས་ནི། །སཾ་སཱིཀྟ་ཙཱ་རུ་ཙཱ་བ་རེ། །ཛརྨ་ཙཱ་བ་ར་ཡུཥཱ་ཡ་མ་མ་སཱི་ད་ཏྱ་ཙིནྟ་མཛེས་པ་དང་ལྡན་མཐོང་གྱུར་ནས། །བདག་གི་སྐྱེ་བ་ཆོས་གོས་དང་། །ལྡན་པར་གྱུར་ཅིག་ཅེས་པ་ཡཏྲ༑། ས་ཨེ་ཥ་པྲ་ཎི་དྷཱ་ནེ་ན་ཛཱ་ཏ་སྟེ་ནེ་བ་ཧས་ཀཿ། །ཨི་ཡཾ་ཙཱ་བ་ར་ཀ་རྩྭ་བསམས། །དེ་ནི་སྨོན་ལམ་ད་དག་གིས། །གླང་པོ་ཅན་འདི་ཉིད་དུ་སྐྱེས། །ཆོས་གོས་བུ་མོ་འདི་ཡང་ཙ་ཏ་ནུ་ཙཱི་བ་ར་ལག་ཎཱ། །ཨི་ཏི་ཐུ་ཏྭཱ་ཀཱི་ཏི་པ་ཏིསྟཛྲཏྟཾ་སུ་ག་ནི། །ལུས་ནི་ཆོས་གོས་ཀྱིས་མཚན་ནོ། །བདེ་གཤེགས་ཀྱིས་གསུངས་བྱུང་ཚུལ་དེ། །དེ་ལྟར་ས་བདག་ཏོ་དི་ཏཾ། །མུ་ཀུ་ཊ་ཀྲྀཥྛ་ཏཏྟཱ་ད་པདྨསྭ་བྷ་བ་ནཾ་ཡ་ཡོ། །ཛ་ཏེ་ས་བིསྨ་གིས་ཐོས་ནས། །ཅོད་པན་དེ་ཡི་ཞབས་པད་ལ། །།རེག་ནས་རང་གི་ཁང་པར་སོང་། །རྒྱལ་པོ་ཡ་མཚན་ཡ་རཱ་ཛཱི་ཀུ་མཱ་ར༔ས་ཧ་ཛ་ཡ་ཡཱ། །ཀ་ཐ་མཱ་ནཾ་བྷ་ག་བ་ཏཱ་དྷམྨོ་བཅས་ཡིསོང་ཚེ། །གཞོན་ནུ་ཆུང་མར་བཅས་པ་ཡིས། །བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ཡི། །ཆོས་ཐུ་ཐཱ་བ་ཤུདྡྷདྷི། །ཏ་ཏ་སྟོ་ཛཱ་ཏ་བེ་རཱ་གྱོ་ཀཱི་ཎ་སཾ་སཱ་ར་བཱ་ས་ནོ། །ནི་དག་པའི་བློ་ཡིས་ཐོས། །དེ་ནས་དེ་དག་ཆགས་བྲལ་སྐྱེས། །འཁོར་བའི་བག་ཆགས་ཉམས་གྱུར་ཅིང༌། །—།
