བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀི་བ་ཡནྟདྲི་ཥ་ནི་ཤཱ་སིནྡི་ད་བྷཱ་དྷཱ་ན་ཡོ་གི་ན། །བི་བཙྩཱ་བ་ད་ནཙྪཱ་དེ་ཡི་དུག་གི་དབུགས་ཀྱིས་ནི། །བསམ་གཏན་སྦྱོར་བ་བདག་ཅག་བསྲེགས། །བཞིན་མཛེས་ཁ་དོག་བྲལ་ཡཱཀི་བ་ལཾ་ཀྲྀ་ཤ་ཏྭཾ་ག་ཏཱ། །ཨི་ཏི་ཏིཀ་ཐི་ཏེ་ཤསྭཱ་བྷི་ཀྵ་སི་བ་དང་། །རིད་པ་འབའ་ཞིག་ཉིད་དུ་གྱུར། །ཅེས་དེ་བཤད་ཚེ་དགེ་སློང་གི། །ཚོགས་ཀྱི་མདུན་གཟི་ཏཔུ་ར། །ནཱ་ག་ཡོརྡ་མ་ནེ་ཡོ་གྱི་མོངྒ་ལྱཱ་ཡ་ན་མཱ་དི་ཤཏ། །ས་སུ་མེ་རུ་བཞུགས་སྟོན་པ་ཡིས། །ཀླུ་དག་འདུལ་བ་ལ་འོས་པར། །མོད་གལ་གྱི་བུ་ལ་བསྟན། །སྦྱོར་བས་མི་ས་མཱ་སཱ་བྱ་ཤིངྒི་རཱ་ལིདྷ་ཏྭཾམྲ་རཾ། །སུསྟོད་དན་ནཱ་གིནྡོ་ཡོ་སྣང་དབྱིབས་ཅན་དེས། །ལྕོག་གིས་མཁའ་ལ་འཁྱུད་པ་ཡི། །རི་རབ་ལྷུན་པོར་ཡང་དག་ཕྱིན། །ཀླུ་གེ་ནཱནྟཆི་ཏཱ་ཤཱི་ཏི། །བོ་རྒྱ་མ་ནོ་ཤ་ནིསྟེ་ན་བུ་བུ་དྷཱ་ཏེ་ཡ་དཱ་ན་ཏོ། །ཏ་དཱ་དབང་དག་ནི་ཉལ་བ་མཐོང་། །དེ་ཡིས་དལ་བུས་བསླང་བ་ལས། །གང་ཚེ་དེ་དག་མ་སད་པ། །དེ་ཚེ་མ་ཧཱ་ནཱ་ག་བ་པུབྷཱུ་ཏཱ་ས་ས་མ་བེབྱཱ་ཡཏ། །པྲ་བུདྡྷོ་སི་ཌི་ཏོ་ཏེ་དེ་ནི་ཀླུ་ཆེན་གྱི། །ལུས་སུ་གྱུར་ནས་ཡང་དག་བཅིངས། །དེ་ཡིས་བསྡམས་པས་རབ་སད་དེ། །ན་དྲྀབྷྱཱ་བྷི་ཥ་ཎཱ་ཀྲི་ཏི། །བི་དྲུ་ཏོ་ན་ར་རཱུ་པ་ཎེ་ཏ་སྦྲ་ཏུབྷ་འཇིགས་རུང་དབྱིབས་དེ་མཐོང་གྱུར་ནས། །འཇིགས་པས་རྣམ་པར་འཁྲུགས་གྱུར་ཏེ། །མི་ཡི་གཟུགས་ཡ་བི་ཧྭ་ལོ། །ནཱ་ག་རཱུ་པ་པ་རི་ཏྱཱ་ཚུ་ཀྲྀ་ཏཱ་རཱུ་པཾ་སྭཱ་ཀཾ་ཏ་ཏ། །པ་ལཱ་ཡ་མཱ་ཀྱིས་རབ་ཏུ་བྲོས། །དེ་ནས་ཕན་ཚུན་འབྲོས་བྱེད་པ། །དེ་ལ་མོ་གལ་གྱི་བུས། །ཀླུ་ཡི་གཟུགས་ནི་ནཱ་བ་སྐྱོ་ནྱ་ཏོ་མོངྒ་ལྟ་ཡ་ནོ་བ་དཏ། ་། ནཱ་གོ་ཀཱ་ག་མྱ་ཏེ་ཏུཎྞཾ་བྷ་ཡཾ་ཡོངས་བཏང་ནས། །རང་གི་གཟུགས་སུ་བྱས་ཏེ་སྨྲས། །ཀླུ་དག་མགྱོགས་པར་གང་དུ་འགྲོ། །འཇིགས་སཾ་རྱ་ཛ་ཏཱ་མི་དཾ། །ན་ས་ནཱ་གསྦི་ཏ་སྟ་ཏྲ་ཡེ་ན་བི་དྲ་བི་ཏོ་ཡུ་བཾ། །སརྦ་པ་འདི་ནི་ཡང་དག་ཐོང་། །གང་གིས་ཁྱེད་ཅག་འཇིགས་བྱེད་པའི། །ཀླུ་དེ་དེ་ན་མི་གནས་སོ། །དེ་ཡི་ཐཱ་ཡ་དི་ནཱསྐྱེ་པ་ཏདྡྷ་ཡ་ཏ་ཡུ་བ་ཡོཛྷ་ཏི། །ཀྲྀ་ཡ་ཏེ་ཀིཾ་ཀ་ར་ཏྱཱ་སྱ་འཇིགས་པས་གལ་ཏེ་ཁྱོད། །རྣམ་ཀུན་བརྟན་པ་མེད་ཉིད་ན། །སྐྱབས་སུ་འོས་པ་སངས་རྒྱས་ལ། །།
