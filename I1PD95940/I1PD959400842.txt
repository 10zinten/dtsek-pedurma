ཀཱ་ཏྱཾ་མ་ཧཱ་མཱ་ཏྱཾ་བཤྩཙཱ་རཾ་བི་ཀྲྀཥཱ་བཱ་ན། །སཱུ་ཌ་མ་ཎི་པྲ་བྷཱ་བེ་ན་སཾ་སུཾ་བྷི་དེ་ནི་བཙལ་བའི་སླད། །པ་ཏ་ཀ་ཞེས་ཉན་རྣ་བཏང་། །དེ་ནས་གཙུག་གི་ནོར་བུ་ཡི། །མཐུ་ཡིས་ཆུ་ཏ་ཛ་ལཾ་ཏ་ཏ། །ནཱ་གཀུ་མཱ་ར་མཱ་ཤཱསྱ་སྦི་ཡ་ཏཱ་མི་ཏ་བྷཱ་ཥ་རྟ། །བི་སྣ་ནི་རེངས་བྱས་ཤིང་། །ཀླུ་ཡིས་ཀྱི་གཞོན་ནུ་དབུགས་ཕྱུང་ནས། །འདུག་ཅིག་ཅེས་པ་རབ་ཏུ་སྨྲས། །རྒྱལ་ཀྱ་རཱ་ཛ་པུ་ཏཱན་པཱ་ད་མུ་དཱ་བ་ཏཱི་བྷུ་ཁཾ། །སྦི་ཏཾ་ནཱ་ག་སྱ་བྷ་བ་ནེ་པ་ད་ཀསྟ་སྲས་ལ་འོས་རྐང་པ་ཡིས། །རྒྱ་ལྡན་ས་གཞི་མཐོང་གྱུར་ནས། །ཀླུ་ཡི་གནས་ན་གནས་པ་དག། པ་ད་ཀ་སུ་ཙ་ཡ་ཏ། །ཨ་ཐ་ནཱ་གེནྡྲ་བྷ་བ་ནཾ་པ་རི་བཱསླུ་ས་མནྟ་ཏ། །ཨ་ཐཱ་ཡིས་དེ་ལ་བརྗོད། །དེ་ནས་ཀླུ་ཡི་དབང་པོའི་གནས། །ཀུན་ནས་ཡོངས་སུ་བསྐོར་ནས་ནི། །བློན་པོ་བ་ཡརྨ་ཧཱ་མཱ་ཏྱཔཱཟི་བཱ་དྷཱི་ཕ་ཎཱི་ཤཱ་རཾ། །ཏཱཾ་ཤུ་བྷིཔཱུ་ར་ཡ་མྱེ་ཥ་ཆེན་པོས་ས་བདག་བཀའ། །གདེངས་ཅན་དབང་ལ་རབ་ཏུ་བསྒྲགས། །ལག་འགྲོ་ཁྱོད་ཀྱི་ཁང་པ་ནི། །བྷ་བ་ནཾ་ཏེ་བྷུ་ཛཾ་ག་མ། །ཛ་ལོ་སྦྲ་ལཾ་སྦྲ་ལོ་ཤྭ་བྷཾ་ཀ་རོ་ཏི་ཀུ་པི་ཏཔྲ་བྷུ། །བདག་གིས་རྡུལ་གྱིས་ཁེངས་པར་བྱ༄༅། །ཆུ་ནི་ཇོ་བོ་ཁྲོས་པ་ཡིས། །ཐད་དང་གཅོང་རོང་དག་ཏུ་བྱེད། །བྷུ་ཛི་གཱི་བྷོ་ག་བི་ཙྪེ་དཾ་ན་ཙེ་དིཙྪ་སི་ཏྲ་སྭ་ཡཾ། །ཨ་ཧི་ཏ་རཱ་ཛ་རཱ་ཛ་སྱ་རང་ཉིད་ལག་འགྲོའི་ལོངས་སྤྱོད་དབང་། །གལ་ཏེ་འཆད་པར་མི་འདོད་ན། །རྒྱལ་པོའི་རྒྱལ་པོ་མི་ཕན་པ། །རཱ་ཛ་པུ་ཏྲི་པ་རི་ཏྱ་ཛ། །ཨི་ཏི་སནྟརྫི་ཏསྟེ་ན་ནཱ་ག་སྣུཎྜ་ཀླུ་ཡཱ་ཀཱ་ཎེ། །རྒྱལ་པོའི་བུ་ནི་ཡོངས་སུ་ཐོང་། །དེ་སྐད་དེ་ཡིས་བསྡིགས་པ་ནི། །མཚན་མོའི་དུས་སུ་ཀླུ་ཡིས་ནི། །ཏ་ཏྭཱ་ཛ་རཱ་ཛ་ན་བྷ་ཐཾ་བྷ་ཡ་བྷོ་གྷ་ཧི་ཛཾ་ཛ་ན། །པྲཙྪནྣཾ་རཱ་ཛ་དུ་ཏྲི་མྱུར་བར་རྒྱལ་པོའི་བུ་བཏང་སྟེ། །སྲོག་ཆགས་རྣམས་ནི་འཇིགས་དབང་གྱུར། །དེ་ནས་རྒྱལ་སྲས་རབ་ཐ་ར་ཛ་ཀཱ་བ་པ་ཐཱསྦི་ཏ། །ཙ་རཱ་ཎ་རྩྭ་ས་མུ་དཱ་བྷིཔ་ཏ་ཀེ་ནོ་བསྒྲིབས་ཏེ། །འཁྲུད་མཁན་དག་གི་ཁང་པར་གནས། །རྐང་པ་བཀོད་པའི་རྒྱ་དག་གིས། །པ་ཏ་ཀ་ཡིས་བུ་པ་ལའི་ཏ། །ཏ་ཏཔཱསྟེ་མ་ཧཱ་མཱ་ཏེ་ར་ཛ་ཀེ་ནཱ་བི་ཏདྷ་ཡཱཏ། །ཉེ་བར་མཚོན། །དེ་ནས་བློན་ཆེན་འོངས་པའི་ཚེ། །དེས་འཇིགས་འཁྲུད་མཁན་དག་གིས་ཀྱང་། །—།
