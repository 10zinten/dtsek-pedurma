མ་ཧོཏྲ་བེ་མི་ཏྲ་ས་མཱ་ག་མེ་ན། །བཱ་ལཱ་ཏ་པིསིདྡྷ་པུ་རེ་བ་བྷཱུ་བ་སིནྡྷ་ར་པཱུ་རི་རི་ཡོངས་སུ་རྒྱས་ཚེ་གྲོགས་དང་ཕྲད་པ་ཡིས། །ཉི་འོད་གཞོན་ནུས་ལི་ཁྲིའི་ཚོགས་ཀྱིས་བཞིན། །གྲུབ་པའི་བ་པོ་ར་ཀེ་ལི། །ཤ་ནི༔པྲ་ཡཱ་ཏེ་ཥཱ་དི་ནེ་ཥཱ་ཥགླུ་མ་ཧོཏྠ་བ་སདྡྷཏ་ཧཅུ་གྲོང་དུ་གྲོང་བའི་རྩེ་དགའ་བྱུང་། །དགའ་སྟོན་ཆེན་པོ་རྨད་བྱུང་དགའ་བའི་ས། །ཉི་མ་དྲུག་ནི་དལ་གྱིས་བྷུ་མེ། །གི་རེས་ཊེ་སཔ་མ་བཱ་ས་རཱ་ཀྲེ་ཙ་ཙཱ་ར་བི་དྷཱུ་དྷ་ར་རཱ་ཛ་སཱུ་ནུ། །།ད་རབ་འདས་ཏེ། །ཉི་མ་བདུན་གྱི་དང་པོ་རི་ངོས་ལ། །རིག་འཛིན་རྒྱལ་པོའི་བུ་ནི་རྣམ་པར་རྒྱུ། །དེར་དན་ཏ་ཏྲ་སྤུ་ཊ་རབི་ཛཱ་ལ་ཕ་ཎཱ་མ་ཎི་སི་ཏ་ཀྲྀ་ཏ་པྲ་ཀཱ་ཤཾ། །མཱ་ཏཱ་ནུ་ནི་གདེངས་ཀའི་ནོར་བུའི་འོད་ཟེར་དག། རབ་གསལ་དྲ་བའི་སྣང་བ་རྒྱས་བྱེད་པ། །ཟླ་བའི་དཀྱིལ་ཡཱ་ཏཾ་པ་རི་པཱུཙྪཱ་ཙནྡྲ་བིམྦྷ་ན་ནཾ་ནཱ་ག་ཀུ་མཱ་ར་མ་གྲེ། །ཏ་yའཁོར་ཡོངས་རྫོགས་བཞིན་རས་ཅན། །།ཀླུ་གཞོན་མ་དང་བཅས་པ་མདུན་དུ་མཐོང་། །མཆི་མ་སྟུག་པོར་པྲ་ས༔དྷ་ན་བཱཟླ་པིནྟྲ་མཱ་ལཱ་བ་ཀཱིཙྪཱསྟ་ན་མཎྜ་ལཱ་ཡཱ། ་། ཨ་བྷུ་གྲ་ཤོ་ཀཱ་རབ་འབབ་ཐིགས་པ་ཡིི། །ཕྲེང་བས་ནུ་མའི་དཀྱིལ་འཁོར་དཀྲིགས་གྱུར་ཅིང་། །རབ་དྲག་མྱ་ངན་གྱིས་ཧ་ཏི་ཀམྤི་ཏཱ་ཡཱཤུ་ཤཱ་བ་ཏཱ་རཾ་ཀ་རུ་ཎ་པྲ་ལཱ་པཾ། །ཧཱ་བཏྠ་པཱ་ཏཱ་ལ་མ་ཎི་བཅོམ་འདར་བ་དེའི། །སྙིང་རྗེའི་ཆོ་ངེ་རབ་ཏུ་གསང་བ་ཐོས། །ཀྱི་ཧུད་མཉེས་གཤིན་ས་འོག་ནོར་པྲ་དཱི་པ་ས་མཱི་ས་མཱསྟེ་སི་ཀ་ཐཾ་ཀྵ་ཡ་སྱ། །དྲགྲུ་མི་ཀཱནྟཾ་ཀཱ་པུ་ནསྟ་བེ་ད་བུའི་སྒྲོན། །ཇི་ལྟར་ཁྱོད་ནི་བརླག་ལ་ཉེ་བ་ཐོབ། །ཁྱོད་གདོང་ཆུ་སྐྱེས་ཀུན་དགའི་ཚོགས་བཞིན་མཛེས། །མཱ་ནནྡ་སནྡོ་ཧ་མི་བཱ་ན་ནཱ་ཛཾ། །ཨ་སྨིནྡྲི་ཡཱ་པཱི་ཏི་ལ་ཏཱ་བ་སནྟེ་ཀི་ཡོ་འདི་ནི་བདག་གིས་ཕྱི་ནས་གང་དུ་མཐོང་། །ཀྱི་ཧུད་དགའ་མའི་དགའ་བའི་འཁྲི་ཤིང་དཔྱིད། །ཡིད་བ་ནེ་མརྨ་ཐ་སནྟི་རཱུ་ཏེ། །ཧཱ་བགྱུ་སེ་བཱནྟཱ་བ་ཛཱི་བ་བྷཱུ་ཏ་ཀཱ་ལ་ཏྲི་པཱ་སྲུབས་མཚམས་སྦྱོར་ཕོ་ཉ་ལང་ཚོ་འདིར། །ཇི་ག་གཉེན་གྱི་སྲོག་གྱུར་ཁྱོད་ཟ་འགྱུར༄༅། །དུས་ཀྱི་གཉིས་ཀྲནྟ་ཨི་བཱ་སི་ཙཱུ་ཏ། །ཐུ་ཏྲེ་ཏི་ཏ་སྩཀ་རུ་ཎ་པྲ་ལཱ་པ་མནྟས་མུཏྟཱིཙྩཱ་བི་ཥཱ་ད་ཤ་འཐུང་གིས་བཅོམ་ཙུ་ཏ་བཞིན། །ཞེས་པ་དེ་ཡི་སྙིང་རྗེའི་སྒྲ་ཐོས་ནས། །མྱ་ངན་ཟུག་རྔུ་ནང་དུ་རབ་—།
