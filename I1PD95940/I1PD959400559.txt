དརྦ་ན་མཱ་རྩི་བ་ཛི་ན་སུ་བི་ན་ཡེ་ན་ཙ། །མཱ་ནི་ནོ་ཙྰ་པ་ཏི་རནྟརྨ་རྙཔས་ཁེངས་པ་དང༌། །རྒྱལ་བ་ལ་ནི་གུས་པས་ཀྱང༌། །ཁེངས་ལྡན་མི་བདག་ཁྲོ་བ་ནི། །ནང་དུ་རབ་ཏུ་བྲཙྪནྣ་ཏྭཾ་ཡ་ཡོ། །ས་སཾ་ཛྷ་ཡཱ་མཱ་དི་ཤུ་ནི་ཛ་པ་རི་ཛ་ནཾ་པུ་ར། །ག་མ་ནེ་བསྒྲིབས་པར་གྱུར། །དེ་ཡིས་རང་གི་འཁོར་ལ་ནི། །འགྲོ་ཚེ་སྔོན་དུ་དེ་དག་རྣམས། །ཚར་བཅད་བརྡ་ནི་གྲ་ཧཾ་ཏེ་ཏྭཾ་ནིཤྲཱི་ཀཱ་ར་ཨི་བ་བྷ་བཏ། །།བྷ་ག་བཱ་ན་བི་སཧྥ་ཡིས་བསྐོས་བྱས་ནས། །རྣམ་འགྱུར་མེད་པ་བཞིན་དུ་གྱུར། །བཅོམ་ལྡན་ཐམས་ཅད་མཁྱེན་པས་ཀྱང་། །ཛྷསྟ་སྱ་ནིཉྩ་ཡཾ་ཛྷ་ཏོ། །བྱམྨོ་པ་དེ་ཤ་པརླནྟེ་པྲ་བཱ་ཙ་ར་ཙི་ཏསྨི་ཏ། །དེ་ཡི་ངེས་པ་མཁྱེན་གྱུར་ནས། །་ཆོས་ནི་ཉེ་བར་བསྟན་པ་ཡི། །མཇུག་ཏུ་འཛུམ་མཛད་རབ་གསུངས་པ། །ན་བི་དྭེ་ཥ་ར་ཛཔུཎྞ་མ་ནོ་མ་ལི་ན་དམྤ་ཎེ། །བྷ་ཏི་དྷརྨོ་པ་དེ་ཤ་སྱ་པྲ་ཡིད་ཀྱི་མེ་ལོང་དྲི་མ་ཅན། །སྡང་བའི་རྡུལ་གྱིས་གང་བ་ལ། །ཆོས་ནི་ཉེ་བར་བསྟན་པ་ཡི། །གཟུགས་ཏི་བིམྦ་པྲ་ཏི་གྲ་ཧ། །།ཨ་བི་ཏ་ས་མ་ཏཱ་ནཾ་ཀོ་ས་མོ་ཧ་ཧ་ཏཱ་ནཾ་བརྙན་འཛིན་པ་མཛེས་མ་ཡིན། །མཉམ་པ་ཉིད་དང་བྲལ་ཞིང་ཁྲོ་དང་རྨོངས་པས་བཅོམ་རྣམས་ལ། །ཀྲྀ་ཤ་མ་པི་ཀུ་ཤ་ལཱ་ཤཾ་ནོ་བ་དེ་ཤཀ་རོ་ཏི། །ན་ཧི་བ་ཧུ་ཏ་ར་དོ་ཥ་ཤུདྡྷི་ཧཱི་ཉེར་བསྟན་གྱིས་ནི་དགེ་བའི་ཆ་ཤས་ཆུང་དུ་ཡོངས་མི་བྱེད། །གང་ཕྱིར་བློ་གྲོས་བླུན་པོ་རྣམས་ཀྱི་རབ་མང་ནེ་ཤ་རཱི་རེ་བྲ་ཛ་པྲ་མ་ཏཱི་ནཾ་བྷེ་ཥ་ཛཾ་བྷེ་ཥ་ཛ་ཏཱམ། །ཨི་ཏི་ཡུཀྟ་བྷ་ག་བ་སྐྱོན་ལྡན་ལུས། །དག་པས་དམན་པ་དག་ལ་སྨན་ནི་སྨན་ཉིད་མི་འགྱུར་རོ། །ཞེས་པ་བཅོམ་ལྡན་རིགས་པ་ནཱ་ཧི་ཏ་མུལྟསྨྲ་ཧི་བ་ཏི། །ཐུ་ཏྭཱཔི་ན་ཙ་ཏ་ཏྭཱ་ཛ་ནཱ་གེ་ཥཱ་བི་མ་ཡིས། །ཕན་པ་གསུང་བ་ཐོས་གྱུར་ཀྱང་། །ཀླུ་རྣམས་ལ་ནི་སྡང་བའི་ཡིདི། །ས་ཡི་བདག་པོས་བཏང་ནསྐ་ཏམ། །བྷ་ག་བནྟཾ་པྲ་ཎ་མཱ་ཐ་པྲ་ཡཱ་ཏེ་སཱ་པ་དཾ་ནཱི་པཾ། །ནཱ་གཱསྟཌི་ནི་ཀཱ་མ་གྱུར། །དེ་ནས་བཅོམ་ལྡན་ལ་བཏུད་ནས། །མི་བདག་རང་གི་གནས་སོང་ཚེ། །དེ་ཡི་དཔུང་གིས་ལམ་བདྷ་མཱརྒ་བྱོ་མ་ས་ཐཱ་ཡཱསྟ། །ཏེ་བ་ཙིནྟཾ་སྭ་བྷ་བ་ནེཀྵ་སཀྵ་ཡ་བཀག་པས། །ཀླུ་རྣམས་མཁའ་ཡི་ལམ་ནས་སོང་། །ཁྱིམ་དུ་དེ་དག་གིས་བསམས་ཏེ། །ས་བདག་གཞོམ་—།
