སརྦ་སིདྡྷཱཀི་མ་ཏྲ་ཀྲི་ཡ་ཏེ་བི་དྷཱ་ཏུ། །མི་བྱི་བ་ནིཀཱ་ར་ཎ་བཏྠ་ལ་ཏྭཾ་ན་ཀརྟུ་གྲུབ། །འདི་ལ་བྱེད་པོ་ལ་ནི་ཅི་ཞིག་བྱ། །རྒྱུ་མེད་མཉེས་གཤིན་དག་ལས་གནས་མིན་འདིར། །ཁྱོད་མརྟ་སྱ་པ་དེ་ཉཏྲ་ཙིནྟཾ། །ཨི་དཾ་གུ་ཎཱ་ལཾ་ཀ་ར་ཎཾ་བ་སུ་སྟེ་སཾ་པུརྞ་སོ་ཛ་རྒྱ་སུ་ནི་ལོག་པའི་བསམ་པ་བྱར་མི་འོས། །ཁྱོད་ཀྱི་ལུས་ནི་ཡོན་ཏན་རྒྱན་དང་ལྡན། །ཚུལ་བཟང་ཡོངས་སུ་དྷཱ་ནི་དྷཱ་ནཾ། །ཏྲི་ལོ་ཀྱ་ཛཱི་བི་ར་པི་རཀྵ་ཎཱི་ཡཾ་ཏྭཱ་ཛྙཱ་ཊཾ་ཎ་སི་བ་ཀ་རྫོགས་པ་བདུད་རྩིའི་གཏེར། །།སྲིད་གསུམ་འཚོ་བ་ཡིས་ཀྱང་བསྲུང་བྱ་འདི། །ཚྭ་མཚུངས་བདག་དོན་ཇི་ཐཾ་མ་མཱརྗེ། །བྷ་བནྟི་ནི་ཎཾ་བྷ་བ་ཀཱ་ན་ནེསྨིནྟསྨ་དྲི་དྷཀཱ་ཤ་པ་ལཱ་ཤ་ལྟར་གཏང་®བར་འོས། །།སྲིད་པའི་ནགས་ཚལ་འདིར་ནི་བདག་ལྟ་བུ། །ཀ་ཤིའི་འདབ་མ་དང་མཚུངས་ཏུ་ལྕ། །བྷ་བདྲི་དྷ་སཱུ་མྲྀ་ཏ་སོ་ད་ར་སྱ་ན་པཱ་རི་ཛཱ་ཏ་སྱ་པུ་ནཔྲ་སཱུ་ཏི། །རྟག་ཏུ་འབྱུང༌། །བདུད་རྩིའི་སྤུན་ཟླར་གྱུར་པ་ཡོངས་འདུའི་ཤིང་། །ཁྱོད་ལྟ་བུ་ནི་ཕྱི་ནས་འཁྲུངས་མ་ཡིན། །པྲ་སཱི་ད་ཛརྞནྟ་ར་པུ་ཎྱ་པ་ཙྪི་དྲྀཥོ་སི་སོ་ཛ་རྒྱ་སུ་དྷཱ་སུ་དཾ་ཤུ། །སྐྱེ་གཞན་བསོད་ནམས་ཟོང་གི་ཚུལ་བཟང་ཁྱོད། །བདུད་རྩི་བདུད་རྩིའི་ཟེར་མཐོང་དང་བར་མཛོད། །ཨསྨདྲི་ཥཱ་དེ་ན་མ་ནཔྲ་ཡཱ་སསདྡྷཱཏྨ་ནཱ་ཡཾ་བྷ་བ་ཏཱ་ན་ཀཱཙྪ། །ག་བདག་གི་ཐུགས་ཁྲལ་དག་གིས་ཡིད་གདུང་འདི། །བདག་ཉིད་ཀུན་གྱིས་ཁྱོད་ནི་ཡོངས་མ་མཛད། །རྒྱ་ཏཱཎྚ་བཱ་བྱཱཎྜ་ནི་བིཥཱ་མུནྟི་གོ་ཀཎྜ་ཤཐཾ་པྲ་ཎི་པ་རྱ་ཏུཎྞཾ། །པཱ་ཏཱ་ལ་གེ་ཧཾ་ཛ་མཚོའི་དྲུང་དུ་ལུས་བཀོད་གོ་ཀཎ། །ཤ་སྦར་མྱུར་དུ་ཕྱིན་ནས་ཕྱག་འཚལ་ཏེ། །མ་ནི་ས་འོག་ཁང་པར་ནི་ཚྭ་བི་སྲ་ཚུ་གཙྪཱ་ཏི་ཏཱ་ཏརྒྱུ་ཤི་ལ་མ་ཁིནྣ། །ཨུཥཱ་ཏི་ནཱ་གཔྲ་ཎི་པ་ཏྱ་མུདྡྷཱ་ཛཱི་རྣམ་སྤྲིངས་ནས། །མཁའ་ལྡིང་རྡོ་དེར་སྐྱོ་མེད་བདག་མཆིའོ། །ཞེས་བརྗོད་མགོ་ཡིས་སྤྲིན་གྱི་བཞོན་པ་ལ། །མཱུ་ཏ་བཱ་ཧཾ་ཛ་ན་ནི་ས་ཧཱ་ཡ། ་། ཛ་གཱ་མ་གོ་ཀཎྜ་ཏ་ཊཾ་མ་ཧཱ་བི་ཏ་རཾ་ག་ཧསྟའི་རབ་ཏུ་བཏུད་ནས་ཀླུ་ནི་ཡུམ་དང་བཅས། །རྦ་རླབས་ཆེན་པོའི་ལག་པས་དབུ་བའི་ཕྲེང་། །རབ་བཀོད་ཏ་ཕེ་ཎ་མཱ་ལྱ། །བིདྡྷ་དྷ་རཱ་ཏི་ཤ་སུ་ཏ་སཱུ་ཏ་ཏྲ་ཏཏྤ་ཎི་སཾ་རཀྵ་ཎ་ནིཉྩ་གོ་ཀཎའི་ངོས་སུ་སོང༌། །རིག་འཛིན་བདག་པོ་དག་གི་བུ་ཡང་དེར། །དེ་ཡི་སྲོག་ནི་རབ་བསྲུངས་ངེས་—།
