ཀཾ༑་། ཧར་པཱུརྞསཱ་ན་ག་རཾ་པྲ་ཏ་སྦེ་པཱརྦ་བཱཏྨ་ཛ། །ས་ཧསྟི་ནཱ་པུ་རཾ་པ་བྱ་ནི་བཀང་། །དགའ་བ་རྒྱས་པའི་ས་བདག་སྲས། །རང་གི་གྲོང་ཁྱེར་དག་ཏུ་སོང་། །ཧསྟི་ན་ཡི་གྲོང་ཕྱིན་ཏེ། །བེ་ཏྱ་སཱ་ཀ་ཐཱ་པི་ཏུ། །ཧནཱངྒིསྨྲ་ཡི་ནཱི་ཏེ་ན་བི་བཱ་ཧེ་གི་ཧི་ཏོཏྠ་བ། །རང་གི་གཏམ་ནི་ཕ་ལ་བཤད། །དགའ་བས་ཡ་མཚན་གྱུར་པ་དེས། །བག་མ་ཡི་ནི་དགའ་སྟོན་བསྒྲུབས། །སུ་ཀྲྀ་ཏིབྷོ་གྱ་ཏཱ་ཡཱ་ཏཱཾ་མཱུཏྟ་མི་བ་ཤ་ཤི་ཤི་ཡཾ། །ཨནྟཔུ་ར་པྲ་ཎ་ཡི་ནཱི་ཙཀྲེ་གཟུགས་ཅན་རི་བོང་ཅན་གྱི་དཔལ། །དགེ་བས་སྤྱད་བྱར་གྱུར་པ་བཞིན། །མི་འམ་ཅི་ཡི་བུ་མོ་ནི། །ཕོ་ཀིནྣ་ར་ཀ་ནྱ་གཾ། །ད་དྷ་ཏཱ་མ་དྷུ་པེ་ནེ་པ་ཏེ་ནཱ་དྷ་ར་མ་དྷུ་སྤྲོ་ཧཾ། །བྲང་འཁོར་དུ་གནས་པར་བྱས། །མཆུ་ཡི་སྦྲང་རྩིར་ཆགས་ལྡན་པ། །སྦྲང་རྩི་འཐུངས་པས་བཞིན་དུ་དེས། །སྤྲོཏྟཱན་ཏ་མུ་ཁཾ་བྷོ་ཛཱ་ཙ་ཀ་མྨེ་ན་ལཱི་མཱི་བ་ས། །མོ་ནེ་པ་ཀ་ཐི་ཏོཀྐཎྞཾ་མུ་རེག་པས་བཞིན་གྱི་ཆུ་སྐྱེས་ནི། །དུད་ཅིང་པདྨོ་ཅན་དེ་འདར། །སྨྲ་བ་བཅད་ཀྱང་སྲེད་པ་འཆད། །གཡོ་ཧུཀམྨེ་པི་ནིཧྥ་ལཾ། །བི་ལགྱུ་བི་སྤུ་ཊ་ལཀྵ་ཏ་སྱཔཱི་ཏྭཾ་ཏ་ཏཱ་ན་བ་མེད་ཀྱང་ཡང་ཡང་འདར། །སྐྱེངས་པར་གྱུར་ཀྱང་དཔལ་མདངས་གསལ། །དེ་ཡི་དགའ་བ་དེ་ཡིས་སཱ༑། ས་ཤ་ནི་ར་དྷཱ་རཱ་སཱ་དེ་ད་ཏཱ་དནྟ་བི་བྷི་ཡི་ཀཾ། །བི་ནི་དཱཌ་བསྒྲུབས། །དལ་གྱིས་མཆུ་ཡི་རོ་མྱང་ཞིང་ཚེ། །སོ་ཡི་འཇིགས་པ་བྱིན་ནས་ནི། །ཆུ་སྐྱེས་རྒྱས་པའི་དྲྀ་ཤསྟ་སྱ་མོ་ན་མུ་དཱ་མ་བཱ་ར་ཡཏ། །ནཱི་བཱི་མོ་ཀྵེ་ནི་ཥ་དྷེ་ཙ་ད་མ་མིག་ཅན་དེའི༄༅། །སྨྲ་བཅད་རྒྱ་ནི་དེ་ཡིས་ཕྲལ། །གོས་མདུད་འགྲོལ་དང་དགག་པ་ལ། །ཁྱོ་ཤུག་བྷོཔཱ་ཎི་པདྨ་ཡོ། །བི་བཱ་ད་ཨི་བ་སོཥམྤ་ཀཾ་ཀ་ཎ་སཱ་ན་ཡོ་ར་བྷཱུཏ། །ས་ལག་པའི་པདྨ་དག། རབ་ཏུ་འདར་ལྡན་གདུ་བུའི་སྒྲ། །རྩོད་པ་བྱེད་པར་གྱུར་པ་བཞིན། །རྩེ་རཱ་ག་བལྤ་བསྟ་སྱ་བི་མཱ་པ་སྨི་ཏ་པུཐི་། །ཀཱན་ཀུ་ཙ་ཕ་ལཱ་ཀོ་དགའ་འཛུམ་གྱི་མེ་ཏོག་རྒྱས། །མཛེས་པའི་ནུ་མའི་འབྲས་བུ་མཚན། །ཆགས་བཅས་ཡལ་འདབ་བྷཱུདྡྷ་གྱསཾ་བྷོ་ག་བཱ་ད་པ། །ཨ་ཏཱན་རེ་ད་ཎཱ་ཏོ་བི་སྲོ་ཀ་ལྡན་པ་དེའི། །ལོངས་སྤྱོད་རྐང་འཐུང་སྤྱད་བྱར་གྱུར། །གནས་སྐབས་དེར་ནི་ལྷོ་ཕྱོགས་པ། །བྲམ་ཟེ་།
