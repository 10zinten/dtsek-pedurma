རུ་ཙྪཱ་བྷ་ར་ཎོ་པ་བྷོ་ག་ལ་ཧ་རི་ཏྱཱཀྟ་ཀྲྀ་ཎ་ཀྲི་ཌ་ཡཱ། །པྲ་ཎཱ་ཏཱ་ཎ་དར་བབ་རྒྱུན་གྱི་ཉེ་བར་ལོངས་སྤྱོད་རྦ་རླབས་རྩྭ་ཡི་རྩེ་དགའ་བཏང་བྱས་ནས། །གུས་བཅས་གང་དག་བི་དྷོ་པ་ར་སྱ་ཀྲྀ་པ་ཡཱ་ཀུཊྚནྟི་ཡི་སཱ་ད་རཱ་ནིཏྟཱ་ཛཱ་ནི་ཛ་དེ་ཧ་དཱ་ན་མ་ཙ་ལཱལྟ་བརྩེ་བས་གཞན་གྱི་སྲོག་ནི་བསྐྱབ་པར་བསྒྲུབ་ལ་རང་གི་ལུས། །ཟོལ་མེད་སྦྱིན་པར་བྱེད་ལ་གཡོ་བ་མེད་པ་ནེ་བ་བན་མ་ཧི། །བ་བྷུ་བ་ཀཱཉྩ་ན་པུ་རེ་ཤྲཱི་མཙླི་བྱ་དྷ་རེ་ཤྭ་དེ་ལ་བདག་ཅག་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་རིག་པ་འཛིན་པའི་དབང་། །སྤྲིན་གྱི་ཏོག་ནི་སྤྲིན་བཞིན་ར༑། ཛཱི་མཱུ་ཏ་ཀེ་ཏུཙླི་མཱུ་ཏ་ཨི་བ་ཏཱ་པ་ཧ་རོཟི་ནཾ། །ཡ་སྱ་ཀལྤ་དུ། །སློང་བའི་གདུང་བ་འཕྲོག་པ་དག། གྲོང་ཁྱེར་གསེར་དུ་བྱུང་བར་གྱུར། །གང་གི་དཔག་བསམ་དྲུ་མོདྷཱ་ཏཱཔ་དཱ་ན་བ་ན་བཱཤི་ཡ། །ཙ་ཀཱ་ཤི་རེ་ཡེ་ཤཔུབྷྱཱཔུཎྱ་ཤིང་ལས་འཁྲུངས། །རྟག་ཏུ་གསར་པ་གསར་པའི་དཔལ། །གྲགས་པའི་མེ་ཏོག་བསོད་ནམས་ཀྱི། །དྲི་སོ་ར་བྷཱ་ནིབྷ་རཱ། །ཛཱི་མཱུ་ཏ་བཱ་ཧ་ནས་སྱ་པུ་ཏྲཱཉྩནྡྲ་ཨི་བོ་ད་དྷེ། །ཨ་བཟང་རྒྱས་པ་རྣམ་པར་མཛེས། །ཆུ་གཏེར་ལ་ནི་ཟླ་བ་བཞིན། །སྤྲིན་གྱི་བཞོན་པ་དེ་ཡི་སྲས། །ཤིན་བྷུ་ད་ཏུ་གྲ་པུ་ཏྟཱ་ནཱི་པྲ་ཏྱ་གྲ་ཨི་བ་སཉྩ་ཡ། །གུ་ཎ་བཱརྫི་ན་ཡེ་ནེ་བ་ཏུ་ལྷག་པའི་བསོད་ནམས་རྣམས། །གསར་དུ་ཚོགས་པ་བཞིན་དུ་གྱུར། །དམ་པའི་སྐྱེ་བོ་ལེགས་བྱས་དང་། །ཏུ་གེ་ནེ་བ་བི་བྷཱུ་ཏི་མཱན། །ས་བ་བྷོ་ཏེ་ན་པུ་ཏྲེ་ཎ་སུ་ཀྲྀ་ཏ་ནེ་ཡ་སཛྫ་ན། །ཡོན་ཏན་ལྡན་པ་དུལ་བས་®དང་། །འབྱོར་པ་ལྡན་པ་གཏོང་བས་བཞིན། །དེ་ནི་སྲས་པོ་དེ་ཡིས་མཛེས། །ཀལྤ་དྲུ་མཾ་ས་སཱ་མ་རཱ་ཛ་དུ་ཏཱ་ཡ་པྲ་ཏི་པཱ་བྱ་ས། །ཏ་པ་སེ་ཤཱནྟི་ནི་དེ་ཡིས་བུ་ལ་དཔག་བསམ་ཤིང་། །རྒྱལ་སྲིད་ཆེ་དང་བཅས་བཏང་ཨིནས། །དཀའ་ཐུབ་སླད་དུ་ཞི་བའི་ལ་ཡཾ་བི་བེ་ཀཱ་མ་ལ་ཡཾ་ཡ་ཡོ། །ཛཱ་ཡཱ་ས་ཁེ་ཤི་ཡཾ་ཏྲཿཀཱ་ཡཱ་ཏེ་ཏཱ་ཏེ་ཏ་གནས། །མ་ལ་ཡར་ནི་དབེན་པས་སོང་། །ཡབ་ནི་བཙུན་མོ་དང་འགྲོགས་ཏེ། །དཔལ་བཏང་དཀའ་པོ་བ་ནཾ། །ཛཱི་མཱུ་ཏ་བཱ་ཧ་ནཔཱལྟ་བི་བྷ་བས་མ་ཙིན་ཏ་ཡཏ། །གུ་རུ་སེ་ཐུབ་ནགས་སོང་ཚེ། །སྤྲིན་གྱི་བཞོན་པ་འབྱོར་པ་དག། ཐོབ་པ་ཡིས་ནི་རབ་བསམས་པ། །བླ་མ་བསྟན་།
