བ་པུཉྩནྡ་ན་ཙཙྩི་ཏཾ། །ཏེ་ན་ཧཱ་ར་ཏུ་ཥཱ་རོ་གྷ་ཤ་ཤི་ཤི་ཏོ་པ་ཧཱ་སི་ནཱ། །གུས་པ་ཡིས་ནི་བྱུགས་པ་བྱས། །དོ་ཤལ་དང་ནི་ཁ་བའི་ཚོགས། །ཟླ་བ་བསིལ་བ་དམན་བྱེད་པ། །སཱ】སཱ་ཀཱིརྟི་ཤུ་བྷེ་ཎ་ཙནྡ་ནེ་ནཱ་བྷ་བཛྫ་ན། །ཏི་ན་བུ་ཏྟཱ་ནུ་བྷཱ་བེ་ན་པུ་ཎྱ་གྲགས་པ་ལྟར་དཀར་ཙནྡན་དེས། །རྒྱལ་བ་རྣལ་དུ་གནས་པར་གྱུར། །དེ་ཡི་བསོད་ནམས་མཐུ་ཡིས་ནི། །སེ་ནཤ་ཤི་པྲ་བྷ། །ཤྲཱི་མཱརྦ་བྷུ་བ་ས་ག་པོདྷུ་ཏཱདྷྱ་ཏ་སོ་དཔལ་དང་ལྡན་པ་བསོད་ནམས་སྡེ། །ཟླ་བའི་འོད་དང་ཡན་ལག་ཀུན། །རྨད་བྱུང་དྲི་བཟང་ལྡན་ར་བྷ། །ཏ་སྱཙནྡ་ན་རཏྣ་དི་ཙུ་ཏ་ཡཱཏྟཱ་ག་རཱ་བྷ་སི། །ཏpབྦཾ་སྭཱ་ཧྲ་ཧོ་པཱཏྟ་པར་གྱུར། །དེ་ཡི་ཙནྡན་རིན་ཆེན་སོགས། །རྒྱ་མཚོའི་ཆུ་ལ་གང་ལྷུང་བ། །དེ་དག་ཐམས་ཅད་རང་ཁྱིམ་ཀུ་པ་མ་དྷཱཏྠ་མུངྒ་ཏཾ། །ནི་ཛཾ་ནི་ཛ་ས་ད་ཏྭཱ་ཐ་བ་ཎི་ཛ་ཙནྡ་ནཱ་དང༌། །ཉེ་བར་ཁྲོན་པའི་ནང་ནས་བྱུང་། །ཚོང་པ་རྣམས་ལ་ཙནྡན་སོགས། །རང་རང་དག་ལ་དེས་དི་ཀཾ། །བྷོ་ཛུ་རལྤ་རིརྒེ་ཧཾ་སཾ་བྷི་གླུ་སཾ་གྷ་པཱུ་ཛ་ཡཏ། །ཨ་ཐ་བྱིན་ནས། །ཟས་དང་རིན་ཆེན་གོས་རྣམས་ཀྱིས། །ཁྱིམ་དུ་དགེ་སློང་དགེ་འདུན་མཆོད། །དེ་ནས་སཱརྦ་པ་ཏེ་དི་བྱ་པྲ་བྷཱ་བཱ་བྷ་ར་ཎཱ་ཤི་ཡཾ། །ཛཱབྷྱཱ་པ་སྦྲ་ཡ་ཀེ་ནཱ་གྲེ་པྲཥཱ་དེད་དཔོན་དག་གིས་ནི། །མཆོག་གི་མཐུ་ཡིས་བརྒྱན་པའི་དཔལ། །མཐོང་ནས་མངོན་དུ་ཉེར་གནས་ཀྱིས། །ཏཏུ་ཀྲྀ་ཏཾ་ཛི་ན། །ཧེ་ཏུཏྟ་མོ་ཐ་བྷ་ག་བཱན་ས་རྨི་ཏསྟ་མ་བྷཱ་དེ་ཡིས་ལེགས་བྱས་རྒྱལ་ལ་དྲིས། །དེ་ནས་བཅོམ་ལྡན་རྒྱུ་ཡི་མཆོག། འཛུམ་དང་བཅས་པས་དེ་ལ་ཥ་ཏ། །པཱསྟ་བཿསརྦ་བ་ཧེ་ནཱ་ལཾ་པ་ཏཱ་ཙནྡ་ནེ་ན་ཚྭ། །ཀྲྀ་ཏཾ་སུ་གསུངས༑། དེད་དཔོན་གྱིས་ནི་ཙནྡན་གྱིས། །ས་ལ་བྱུགས་པས་དཔལ་འབྱོར་ཐོབ། །བསོད་ནམས་ཀྲྀ་ཏཾ་ཤཱི་ལེ་ནཱ་པུ་ཎྱ་སེ་ནེ་ན་ཀརྨ་ཏཏ། །ཨ་མྲྀ་ཏ་ཏྭཱ་མ་བཱབྷ་སྱ་ཡ་སྤྱི་སྡེས་ནི་ལེགས་བྱས་ཀྱི། །ངང་ཚུལ་དག་གིས་ལས་དེ་བྱས། །བདུད་རྩི་ཉིད་ནི་རབ་ཐོབ་པ། །གང་ཏཱ་ལེ་ཤཱ་བི་སྲུ་ཥ༅། །པཱ་ཀེ་ཀུ་ཤ་ལ་མཱུ་ལཱ་ནཱ་མེ་ཥཛརྨ་ནི་པཉྩི་མེ། །པྲ་བྲ་གི་འདི་དག་ཆ་ཡི་ཐིགས། །དགེ་བའི་རྩ་བ་སྨིན་པ་ན། །སྐྱེ་བ་ཐ་མ་དག་ལ་འདི། །རབ་བྱུང་བླ་མེད་།
