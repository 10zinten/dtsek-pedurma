སརྦ་ཏཱ། །ན་ཉྩ་ཏདྲི་དྷཱ་ཡཱ་སཱ་མ་པཱ་རཱ་བཱ་རི་སམྤ་ད། །ཨེ་ཏ་དུཏྟིརྨ་ནི་ཁི་ཏུ་ཁེངས། །དེ་ལྟའི་ཆུ་ཀླུང་དག་ཀྱང་སྟེ། །·ཆུ་ཡི་ཕུན་ཚོགས་ཕ་མཐའ་མེད། །དེ་དག་མ་ལུས་རབ་ལཾ་པུ་ཙྪིཔ་ར་ཧི་ཏོདྨ་ཏ། །ཨ་རོ་ཧ་ཏི་སུ་དྷཱ་ཤི་ལ་ཐཾ་གི་རཱ་ལཾ་བརྒལ་ཏེ། །གཞན་ཕན་བརྩོན་པ་བསོད་ནམས་ཀྱིས། །རྩེ་མོ་མཁའ་ལ་རེག་གྱུར་པ། །དཀར་བའི་རི་ལ་གི་ཏཱམྦ་རཾ། །ཏ་ཏསྟ་སྣ་པ་རེ་པནེ་ཀལྤ་བྲྀཀྵ་པ་ཤོ་བྷི་ཏཾ། །བུ་རཾ་རོ་ཧི་ཏ་འཛེག་པར་འགྱུར། །དེ་ནས་དེ་ཡི་ངོས་གཞན་ལ། །རོ་ཧི་ཏ་ཀི་ཞེས་གྲོང་ཁྱེར། །དཔག་བསམ་ཤིང་ཀཾ་ནཱ་མ་དྲི་བྱ་ཏེ་སྭརྒ་སནི་བྷཾ། །ཨསྟི་ཏ་ཏྲ་མ་གྷོ་ནཱ་མ་མ་གིས་ཉེ་བར་མཛེས། །མཐོ་རིས་དང་མཚུངས་མཐོང་བར་འགྱུར། །དེ་ན་དེད་དཔོན་སྙིང་སྟོབས་ནི། གྷ་པཱ་ནི་བ་བི་ཤུ་ཏ། །སརྦ་བཱ་ཧོ་མ་ཧཱ་ས་ཏོསརྦ་ས་ཏྭཱ་ཧི་ཏ་རེ་ཏ། །ཆེན་པོ་མ་གྷ་ཞེས་པ་ཡོད། །སེམས་ཅན་ཀུན་ལ་ཕན་པར་དགྱེས། །བརྒྱ་བྱིན་བཞིན་དུ་རྣམ་པར་གྲགས། །བ་ད་ར་དྷཱི་པ་ཡཱ་ཏཱ་ཡཱ་མུདྨ་ཏ་ས་ནཱ་བ་བྱ་དྷཱི། །།མཱརྒོ་པ་དེ་ཤཾ་དེ་ཤ་ཛྷཤེས་བཟང་བློ་གྲོས་སྐྱོན་མེད་དེས། །བ་ད་ར་ཡི་གླིང་དག་ཏུ། །འགྲོ་བར་བརྩོན་པ་ཁྱོད་ཀྱིས་ནི། །ས་ཏེ་སཐཾ་ཀ་རི་་ཏི། །ཨིཀཱུ་ཀྟིནཱ་རྒྱ་བ་ཧུ་ཤསུ་བྷཱ་དྲི་རི་ལམ་གྱི་གདམས་ངག་ཀུན་བྱེད་འགྱུར། །འདི་བརྗོད་བདུད་རྩིའི་རླན་གྱིས་བཞིན། །འོས་པའི་ཚིག་གིས་བ་སུ་པྲ་ཡཾ། །བ་ཙོ་ཏི་ཏུ་ཙི་ཏིརྗེ་བཱི་ས་ཧཱ་སཱནྟ་ར་དྷཱི་ཡ་ཏ། པྲ་བུདྡྷསུ་པྲ་ཡཱཤིན་ཏུ་དགའ། །མང་དུ་སྤྲོ་བ་བསྐྱེད་བྱས་ནས། །ལྷ་མོ་འཕྲལ་ལ་མི་སྣང་གྱུར། །ཤིན་ཏུ་དགའ་ནི་རབ་སཐཾ་ཏཏྟཱ་ཐེ་ཏི་བི་ཙིནྟ་ཡན། །པྲ་ཏསྦི་ས་ཏཱ་མཱ་རཱུ་རྒྱ་ནི་ཛོཏྟཱ་ཧ་སད་ནས། །ཐམས་ཅད་དེ་དེ་བཞིན་ཞེས་བསམས། །རང་གི་སྤྲོ་བ་སྔོན་བཏང་ནས། །སྙིང་སྟོབས་ལ་ནི་པུ་རས་རཿ། །ས་བྲ་ཛན་བི་ཛི་ཏཱ་ཡཱ་སསྟེ་ན་ནིརྡེཎྜ་བནྡྷ་ནཱ། །པུཎིཛྲ་ད་ཞོན་ཏེ་ཆས། །དེ་ཡིས་བསྟན་པའི་ལམ་ལས་ནི། །ངལ་བ་རྣམ་བཅོམ་དེས་བགྲོད་པས། །ལོ་ནི་བཅུ་ཤ་བྷི་རྦརྟིཔྲ་ས་རོ་ཧི་ཏ་ཀཾ་པུ་རཾ། །ཨ་དཱནྟ་རེ་ཀརྨ་ཡོ་གཱཏྟ་ཏྲ་གཉིས་རྫོགས་པ་ན། །རོ་ཧི་ཏ་ཀའི་གྲོང་ཁྱེར་སླེབ། །སྐབས་འདིར་དེར་ནི་དོན་མཐུན་བདག། མ་གྷ་།
