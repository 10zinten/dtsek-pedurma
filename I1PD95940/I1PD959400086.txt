ལོ་ས་སརྫ་དུསྨྱུ་ཁོ་ནཱ་མ་མུ་ནིཤཱ་པ་ཛ་ལོ་དི་ནི། །པྲ་ཧ་སནྣ་ཐ་ཏཾ་པྲ་ཧ་བཞིན་ངན་ཞེས་བྱ་ཐུབ་པ་ཡིས། །དམོད་པའི་ཆུ་ནི་མཁའ་ལ་གཏོར། །དེ་ནས་སྡེ་དཔོན་ཡོངས་འདྲེན་སེ་ནཱ་ནཱི་པ་རི་ཎཱ་ཡ་ཀཿ། །མ་ཧརྩེ་སཾ་ཧ་ར་རུ་ཎཾ་མཱ་ཀྲྀ་ཐསྟ་པ་སཀྵ་གྱིས། །རབ་ཏུ་བཞད་ཅིང་དེ་ལ་སྨྲས། །དྲང་སྲོང་ཆེན་པོ་ཁྲོ་བ་ཐོང་། །དཀའ་ཐུབ་ཟད་པར་ཡོངས་མ་ཡཾ༑། བི་ཕ་ཁྱཉལ་ཛཾ་ཤཱ་བོ་ཡཾ་ཡཱ་སྱ་ཏུ་གྲེ་མ་ཧཱི་པ་ཏེ། །ན་ཏེ་བ་མཛད། །ས་བདག་མདུན་དུ་དམོད་པ་འདི། །འབྲས་མེད་ངོ་ཚར་འགྲོ་བར་འགྱུར། །ཁྱོད་ཀྱིས་གང་ཡཾ་ཁ་ག་ཡེ་ཊཱ་ཡུ་ཡཾ་པཀཱ་ཡ་བྱ་མཱ། ་་། ཨི་ཏུཀེ་སེ་ནྱ་པ་ཏི་ན་་་གིས་འདབ་གཤོག་ཟད། །བཟོད་པའི་བྱ་དེ་བདག་ཅག་མིན། །དེ་སྐད་སྡེ་དཔོན་གྱིས་སྨྲས་ཚེ། ཤཱ་པ་སྟབྷཱ་མ་ནཱི་ཀི་ནཱི། །དྲྀཥྛ་གྲེ་བིརྨ་མཱ་རླུ་ཙེ་ཀི་མི་ཏ་དི་ཏི་བྷུ་་དམོད་པས་བཟུང་བའི་དཔུང་ཚོགས་ལ། །མདུན་དུ་ས་བདག་གིས་བལྟས་ནས། །ཡ་མཚན་པས་འདི་ཅི་པ་ཏི། །སཾ་རབྷོ་ཐ་ས་མ་ཧྥེ་ཏྱ་སེ་ནཱ་པ་ཏི་རུ་བ་ཙ་ཏཾ། །ཏེ་ཏཱཾ་དེ་་་ཞེས་གསུངས། །དེ་ནས་སྡེ་དཔོན་ཁྲོ་ལྡན་པས། །དེ་ལ་ཡང་དག་མངོན་ཕྱོགས་སྨྲས། །ལྷ་གཅིག་དྲང་བ་མ་ཧརཱི་ཎཾ་ཤཱ་པཱ་སི་དྡྷཾ་ན་སརྦ་ཏི། །ཨི་དཉྩ་ཙཀྲ་ར༑་ཏེ་་་སྲོང་དེ་དག་གིས། །དམོད་པས་དཔུང་འདི་མི་བགྲོད་དོ། །ཁྱོད་ཀྱི་འཁོར་ལོ་རིན་ཆེན་འདིའང་། །བྱོནྣི་ཤཱ་པ་བི་བྷྱཱཙྪི་ཏཾ། །དྷེཏྟ་ཛ་ལ་ད་སཾ་རུདྷཱ་ཏིསྨྲ་དཱི་དྷི་ཏི་་་མཁའ་ལ་དམོད་པས་རྣམ་པར་འཁྱམ༄༅། །སྤྲིན་གྱིས་ཡང་དག་བགགས་པ་ཡི། །ཚ་ཟེར་ཅན་དང་ཏུ་ལ་ཏྭཾ། །ཨེ་ཏ་དཱ་ཀརྞ་ནི་པ་ཏིཛྲཏྟཱ་ཙཱ་གྲེ་ཏ་ཐི་བ་ཏཏ། །དྲྀ་་་མཚུངས་པར་བཟུང་། །དེ་དག་ཐོས་ཤིང་མི་བདག་གིས། །མདུན་དུ་དེ་ལྟར་མཐོང་ནས་ཀྱང་། །མིག་ཤི་བ་བི་ད་དྷེ་ཤཱ་་བི་ཕ་ལོཙྪཱཎྜ་བི་བླ་བཻ། །དེ་ཧ་ཀྵ་ཡཾ་མ་་་ཉིད་ཀྱིས་བསྒྲུབས་དམོད་པ་དག། འབྲས་མེད་རབ་གཏུམ་བསླད་པར་རོ། །སྙིང་རྗེའི་ཚོགས་ཀྱིས་ལུས་ཧརཱི་ཎཾ་པ་རི་རཀྵན་ཀྲྀ་པཱ་ཀུ་ལཱ། །ཛ་ཊཱ་ནྱ་པ་ཏི་ཡ་ཏ་བྷུ་མོ་ས་་འཇིག་པས། །ཡོངས་བསྲུངས་དེ་ཡི་ཀྱལ་ཀའི་བཀས། །དམན་པའི་དྲང་སྲོང་ཆེ་རྣམས་ཀྱི། །རལ་—།
