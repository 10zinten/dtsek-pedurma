ལ་ན་ཡ་ནཾ་ཤ་ནི། །ཨུཏྟཱ་ཁྱོཏྟཱ་ཊུ་ཏུནྟེ་ན་པྲཏྠ་སརྫ་པུ་ན༔པུ་ན། །འགྲོ་བའི་མཁར་བགྲོད་ཀྱིས། །མཆུ་ཡིས་ཕྱུང་ཞིང་རབ་ཕྱུང་ནས། །ཡང་དང་ཡང་དུ་རབ་ཏུ་བཏང་། །དྷེཆེ་ནིཉྩ་ལ་སབྷཾ་གས་དྲྀབྷྱཱ་པི་སྨི་ཏཾ་ཁ་གཾ། །།ཨུ་བཱ་ཙ་བྷུ་ཀཱ་ནིཤཾ་ཀཾ་བརྟན་ཞིང་ཡན་ལག་ཀུན་མི་གཡོ། །དེ་ཡིས་ཡ་མཚན་བྱ་མཐོང་ནས། །སྨྲས་པ་དོགས་མེད་ཉིད་དུ་ཟོ། །ནི་བ་ཏཱཾ་དྷཱ་ར་ཡཱ་མྱ་ཧཾ། །ནིསཱ་ར་བི་ར་སཀཱ་ཡ༔སཱ་པཱ་ཡོཉཡཾ་གྷ་ཎ་ཀླུ་ང་ཡིས་ཁྱོད་ནི་ཡོངས་མི་འཛིན། །རོ་བྲལ་སྙིང་པོ་མེད་པའི་ལུས། །གནོད་བཅས་སྐད་ཅིག་འཇིག་པ་ཡི༑། པ་རོ་པ་ཀཱ་ར་ལེ་ཤེ་ན་ཡཱ་ཏི་སཾ་སཱ་ར་པཱ་ར་ཏྭཾ། །གླེ་ད་སྱནི་འདི། །གཞན་ལ་ཕན་པའི་ཆ་ཤས་ཀྱིས། །འཁོར་བའི་སྙིང་པོ་ཉིད་དུ་གྱུར། །མི་རྣམས་ལུས་ནི་མི་ནི་ནིནྡི་ཏེ་པྲ་ཏི་པ་དཾ་ཤཱ་སཀྵཎ་སནྡྲི་ནི་སྣེ་ཧཀོཉཡ་མ་པཱ་ཡ་དྷཱ་རྨི་མ་ལ་ནེ་མི་གཙང་ཟག་ཅིང་སོ་སོའི་གནས་སུ་སྨད་གྱུར་དྲི་མ་ཅན། །གནོད་པའི་གནས་སུ་གྱུར་ཅིང་དབུགས་དུས་གཡོ་བྱཱ་ཤ་རཱི་རེ་ཤྲཱི་ཎཾ། །ཨེ་ཀི་བ་སྤྲོ་ཧ་ཎཱི་ཡ་ཏཱ་སྱ་ཡ་དི་དམ་པུ་ཙྪིཀཱཙིཏྟ་ལ་ལོག་པར་ཆགས་པ་འདི་ནི་ཅི། །གང་ཞིག་བསོད་ནམས་རྣམས་ཀྱི་ལེའགའ་ཞིག་དག་ཏུ་འགའ་ལ་འགའ་སྱ་ཙིཏ་ཀིཉྩ་ད་བཱི་གྱུ་ཀ་དཱ་ཙི་དཱནྟི་ཤ་མ་ན་ཏཱ་ཎཱ་ཡ་སནྣ་ཧུ་ཏི། །ཚེ་འགའ་མཐོང་ནས། །གདུང་བ་ཞི་བྱེད་སྐྱོབ་ལ་ཞུགས་གྱུར་འདི་ཉིད་གཅིག་པུ་འདི་ཡི་དགའ་བྱ་ཉིད། །ཨི་ཏི་ཏ་སྱ་བྲུ་བཱ་ཎ་སྱ་གྱུཎྜ་མིཔཀྴི་བྷིཀ་ཎཱཏ། །བྷ་གྲུ་མཱ་ཎ་སྱ་བི་ཀྱི་སྱ་ནཱི་ཏཅེས་པ་སྨྲ་བ་དེ་ཡི་ལུས། །བཀྲེས་པས་ཉེན་པའི་བྱ་རྣམས་ཀྱིས། །ཟ་ཞིང་འཕངས་པའི་སྐད་ཅིག་ལ། །ཀཱ་ཡོ་ད་སྦི་ཤེ་ཥ་ཏྭཾ། །མ་ཧཱ་ཤཱ་ལ་སྱཱ་ཀཱ་ལེ་ན་བྲཱཧྥ་ཎ་ཏ་སྱ་པུ་ཏྲ་ཏྭཾ། །ཡཱ་རུས་པའི་ལྷག་མ་ཉིད་དུ་གྱུར། །དུས་ཀྱིས་དེ་ནི་བྲམ་ཟེ་དག། ས་ཡུལ་ཆེན་པོའི་བུར་སྐྱེས་ཏེ། །བདེན་ཏས་ཏྱ་པྲ་ཏོ་ནཱ་མ་བ་བྷཱུ་བ་ཛ་ན་སརྨ་ཏ། །ཨ་བཱསྱཱ་ཁི་ལ་བི་བྱ་སྱ་པའི་བརྟུལ་ཞུགས་ཞེས་བྱ་བ། །སྐྱེ་བོ་དག་གིས་བཀུར་བར་གྱུར། །རིག་པ་མ་ལུས་ཐོབ་གྱུར་ཅིང་། །ཀ་རུ་ཎཱ་སཀྟ་ཙེ་ཏ་ས། །ཏ་སྱ་ཤཱནྟི་ར་ཏ་སྭཱ་བྷུདྲི་བཱ་ཧ་བི་མུ་ཁམྨ་ན། །སྙིང་རྗེ་དག་ལ་ཆགས་པའི་སེམས། །ཞི་བར་དགའ་བ་དེ་ཡི་ཡིད། །ཁྱིམ་ཐབ་ལས་ནི་ཕྱིར་ཕྱོགས་གྱུར། །—།
