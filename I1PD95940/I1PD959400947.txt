གྲི་ཧཾ། །།ར་ཧསཾ་བྷོ་ག་སནྣཱདྷཾ་བིགྷ་མཱ་ས་སཱ་ད་ཏྭཾ། །།ཏེ་ཏ་སྤྱི་སོང་ནས། །མཁས་ཤིང་ལོངས་སྤྱོད་ལ་ཆས་མ། །ད་ནི་དབེན་པར་ཡང་དག་ཐོབ། །དེ་ལ་རང་གིས་པྲ་ཀྲྀ་ཏཾ་པཱ་པ་མཱ་ད་རཱ་ཡ་ནི་བེ་ད་ཡཏ། ན། ཏེ་ནི་བ་མཱུཎཱཟླ་རྒྱུནྟེ་ཡེ་ནི་བཱ་བྱས་པའི་སྡིག། གུས་པའི་སླད་དུ་དེ་ཡིས་བཤད། །གང་གིས་སྨད་པ་ཉིད་འགྱུར་བ། །དེ་ཉིད་ཀྱིས་ཡནྟི་ནི་རྱ་ཏྭཾ། །ཏ་ས་མཱ་ཏྲི་བ་དྷ་ཀཱུ་ར་པཱ་པ་ཤ་བ་ཎ་ཀམྤི་ཏཱ། །སཱ་པཱ་ཀ་ནི་རྨོངས་པ་བསྔགས། །མ་བསད་སྡིག་པ་དྲག་པོ་ནི། །ཐོས་ནས་འདར་བར་གྱུར་པ་དེ། །གཏུམ་པོ་ལ་སྐྱེ་བ་སཾ་སྤནེ་བི་མུ་ཁཱ་སཱ་བྷ་བ་པ་རཾ། །ཧརྩྭ་དྷི་རི་ཧ་ཎ་བྱཱ་ཛཱ་ཏ་སཱ་ཛ་གཱ་བཞིན་དེ་ལ་ནི། །རེག་ལ་ཤིན་ཏུ་ཕྱིར་ཕྱོགས་གྱུར། །དེ་ཡི་གནས་ནས་ཁང་བཟང་ལ། །འཛེག་པའི་མ་ཏེ་དནྟི་ཀཱཏི། །སཱ་བྷི་ལཱ་ཥོ་པི་པཱ་པཱ་ནཾ་ད་རེ་ཎ་བི་ར་ཛུ་ཏེ། །ཨུཙཱ་ཟོལ་གྱིས་དེ་སོང་སྟེ། །སྡིག་པ་རྣམས་ཀྱིས་གུས་པ་ཡིས། །སྲེད་ལྡན་དག་ཀྱང་ཆགས་བྲལ་གྱུར༄༅། །སྡིག་ཧམྱ་ས་མཱ་རུ་ཙྪཱ་ཀཱནྟཱ་པཱ་པ་པ་རཱརླུ་ཁཱི། །ཙོ་རཉྷོ་རོ་ཡ་མི་ཏུཙྩཤྩ་པ་ལས་ནི་ཕྱིར་ཕྱོགས་མ། །དེ་ཡིས་བག་མ་མཐོན་པོར་འཛེགས། །འདི་ནི་ཆོམ་རྐུན་ཆོམ་རྐུན་ཞེས། །ཀྲོ་ཤ་ཙ་ཀི་ཏེ་བ་སཱ། ་། པྲ་དྷཱ་བི་ཏེ་པ་རི་ཛ་ནེ་བྷ་ཡ་བྷགྷ་ས་བི་འཇིགས་པ་བཞིན་དུ་སྐད་ཆེར་བསྒྲགས། །འཁོར་གྱི་སྐྱེ་བོས་རྒྱུགས་པའི་ཚེ། །འཇིགས་པས་ཉེན་པ་དེ་དྲུ་ཏེ། །ཡ་ཡོ་ཡཱ་བདྡྷི་ཛ་གྲི་ཧཾ་ཀྲྀ་ཏ་ཀ་ལ་ཧ་ལཤྭ་བྷི། །ཧ་ཏ་མེ་ཛ་ན་ནཱི་རབ་བྲོས། །ཇི་སྲིད་རང་གི་ཁྱིམ་དུ་སོང་། །ཁྱི་རྣམས་ཀྱིས་ནི་ཅ་ཅོ་བྱས། །བདག་གི་མ་ནི་ཆོམ་རྐུན་ཙོ་རི་རི་ཏི་ཏཱནྡྲ་ལཱ་པ་བཱན། །པྲ་བྷཱ་ཏེ་བི་ད་དྷེ་མཱ་ཏུས་ཤ་རཱི་རཱ་དི་སཎི་ཡཱ། །གྱིས། ྋ། བསད་ཅེས་སྐད་ཆེར་རབ་སྨྲ་ཞིང་། །སྔ་དྲོ་དེ་ཡིས་མ་ཡི་ནི། །།ལུས་ལ་སོགས་པ་སྦྱང་བ་བྱས། །པྲཙྪནྣ་པཱ་པ་སན་དསྒྱུ་མཱ་ནཔ་ཙེ་ཏེ་ཡན། །སུ་ཏཱི་བྲ་ན་ར་ཀཱ་པཱ་རབ་བསྒྲིབས་སྡིག་པའི་གདུང་བ་ཡིས། །སྲེག་ཅིང་ཤིན་ཏུ་མི་བཟད་པའི། །དམྱལ་བར་ལྷུང་བ་དེ་་ཏཾ་བ་བྷཱ་མ་བྷུ་བི་བྷུ་ཏ་བཏ། །པྲ་ཡཤྲཱི་ཏཾ་དཱི་ཛསྟེ་ན་པྲཋཔཱ་ཏ་བསམས་ཏེ། །།འབྱུང་པོ་བཞིན་དུ་ས་ལ་འཁྱམས། །སྡིག་པ་སྨྲ་བྱེད་དེ་ཡིས་ནི། །ཕྱིར་བཅོས་དྲིས་ལ་།
