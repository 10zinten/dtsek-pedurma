ཡོ་མུ་ཁསཾ་གྷསྟསྨཱདྷུབྷ་བྷ་ར་ཙ་ཏྭཾ། །སཟློ་བ་བློ་ག་ས་མ་གི་གཤེགས་ལ། །།སོགས་པའི་དགེ་འདུན་མངོན་པར་མཆོད། །ཁྱོད་ལ་ལོངས་སྤྱོད་དག་གི་ཚོགས། །མ་ཧཾ་སཾ་པཱ་ད་ཡཱ་མི་བ། །ཁ། ཨི་ཏི་ཏེ་པྲ་རི་ཏཱསྟེ་ན་སཾ་བྷཱ་རི་ཏ་དུ་པཱ་ཐམས་ཅད་བདག་གིས་ཕུན་ཚོགས་བགྱི། །།ཞེས་པ་དེས་བསྐུལ་དེ་དག་གིས། །དེ་ཡི་®ཉེར་བསྒྲུབས་ཚོགས་ཀྲྀ་ཏི། །ས་སཾ་གྷ་མ་ཧཱ་སཤཱནྟི་བྷ་ག་བནྟ་མ་པུ་ཛ་ཡཏ། ཊ། ཏ་ཏ་སྟེ་ཀྲྀ་རྣམས་ཀྱིས། །བཅོམ་ལྡན་དགེ་འདུན་དང་བཅས་པ། །སྡིག་པ་ཞི་བའི་སླད་དུ་མཆོད། །དེ་ནས་དེ་ཚེ་ཏ་ཀ་ལྱཱ་ཎཾ་པདྨ་ཎའི་ཎ་ཀཏྠ་ཥཱ། །ད་དྲྀ་ཤུསཱ་ཐཾ་ཧཾ་དཱི་པྲིས་པུཎྞཾ་རཏྭཱ་རཱ་དགེ་བ་བྱས། །སྡིག་པ་ཟད་པ་དེ་དག་གིས། །རང་ཁྱིམ་རབ་འབར་རིན་ཆེན་གྱི། །ཕུང་པོས་ཀུན་ཏུ་ཤི་བྷི། ་། ཏ་ཏཔྲ་བུདྡྷ་བི་རཱ་གྱཔྲ་ཤཱནྟིམ་དྷུ་ར་སཱ་ར། །པྲ་བྲ་གང་བ་མཐོང་། །དེ་ནས་ཆགས་བྲལ་སད་པ་ཡི། །བློ་དང་ལྡན་པ་སྦྲང་རྩིའི་དབྱངས། །།བཅོམ་ལྡན་ཛཱ་མཱ་ད་དེ་དྷི་མཱན་ག་ཏཱ་བྷ་ག་བ་ཏོནྟི་ཀཾ། །ས་ཤཱyཤཱ་སཱ་ནཱཉུཏྟཱ་འདས་ཀྱི་དྲུང་སོང་ནས། །རབ་ཏུ་ཞི་སླད་རབ་བྱུང་བླངས། །སྟོན་པའི་བཀའ་ཡིས་དུལ་བ་ཡི། །ཤ་བཙཱི་ནི་ཡ་ཏ་བྲ་ཏ། །ཛ་གཱ་མ་ཛ་ན་པརླནྟ་བི་ཧཱ་རཾ་ཀཱཧྥ་ཊ་ཤ་ཊཱ་བརྟུལ་ཞུགས་ཅན་གྱིས་མཉན་ཡོད་བཏང་། །སྐྱེ་བོའི་མཐའ་ཡི་གཙུག་ལག་ཁང་། །རི་བྲག་དག་ལ་བརྟེན་ཤ་ཡཾ། ཌ། དཏྟ་ཤིཀྵ་པ་དསྟེ་ན་ཏ་ཏྲ་ཀཱཉྫ་ཊི་ཀོ་ཛ་ན། །རཏྣ་ཏྲ་ཡཾ་ཀླུ་པར་སོང་། །དེར་དེས་བསླབ་པའི་གནས་བྱིན་པ། །རི་བྲག་པ་ཡི་སྐྱེ་བོ་རྣམས། །ཉོན་མོངས་དུག་ནི་ཤ་བི་ཥ་པྲ་ཤཱནྟི་ཤ་ར་ཎཾ་ཡ་ཡོ། །།ཨསྨིནྣ་བ་ས་རེ་ཙོ་རཀཱ་ན་ནཾ་རབ་ཞིའི་སླད། །དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་སོང་། །གནས་སྐབས་དེར་ནི་ཆོམ་རྐུན་པ། །ནགས་ཏནི་བཱ་སི་ན། །ཨུཏྟཾ་དུརྒོ་པ་ཧཱ་ཡ་ན་ར་མསྡེ་མུ་བྱ་ཡུ། །ཏེ་ཀྱི་མཐའ་ན་གནས་པ་རྣམས། །།དུར་གའི་ཡ་སྟགས་ཁས་བླངས་པའི། །མི་ནི་ཚོལ་བ་དག་ལ་ཞུགས། །ཕྲ་ཏཾ་བི་ཧཱ་ར་མཱ་ས་ཏྱ་པི་ཤུ་ནེ་ན་པྲ་དནྟི་ཏཾ། ་། བ་བནྡྷབྷིཀྵ་སཾ་གྷཱ་ཏཾ་མ་ཅན་གྱིས་རབ་བསྟན་པའི། །གཙུག་ལག་ཁང་དེར་དེ་དག་འོངས། ་། གཎྜིའི་སྒྲ་ལ་འདུས་པ་ཡི། །—།
