ལཱ་བྱཱསྭ་བྷཱ་བ་ཏ། །ན་མནྣི་བྷོ་ན་མསི་བྱོ་ན་མསི་བྷོ་ན་མོ་བཞིན་ཉིད་ཀྱིས་ཚུལ་ཁྲིམས་འཆལ། །བུད་མེད་ལ་འདུད་བུད་མེད་འདུད། །བུད་མེད་ལ་འདུད་འདུད་དོ་ན་མ། །དྲྀཏྠ་སི་བྲྀཏྟ་དོ་ཥཱ་ཡ་ཏཙིནྟཱ་བྱ་ཐི་ཏཱཏྨ་ནེ། །ཨ་སྦེ་འདུད། །བུད་མེད་སྤྱོད་ཚུལ་སྐྱོན་མཐོང་བ། །དེ་བསམས་གདུང་བའི་བདག་ཉིད་ཅན། །བདག་ནི་ཥ་པྲ་ཐི་བཱི་མྱ་ཧཱུྃ་ར་པཱུཙྪཱ་ན་རོ་ཙ་ཏེ། །ན་ག་ལཾ་ག་བ་དྷུ་རིན་ཅེན་གྱིས་གང་བའི། །ས་གཞི་འདི་ལའང་ཡོངས་མི་སྲིད། །རི་ཡི་རི་དྭགས་ཆུང་མ་ལྟར་རྨོངས་ཕ་མུཏྟཱ་སི་བྷཱཔ་རཾ་པ་ར་པཉྩནེ་ཏ་ནུ་བི་ཏ་ར་ཎེ་སཱཀྵཔཾ་སཾ་ཧ་རནྟི་ཙ་ཛཱི་བི་ཏཾ། །ད་རོལ་བསླུ་ལ་མཆོག་ཏུ་སྨིན། །ལུས་ནི་རྣམ་པར་སྦྱིན་ལ་ཆགས་ཤིང་སྐྱེས་བུ་རྣམས་ཀྱི་སྲོག་ཀྱང་འཕྲོག། མེ་པ་ཏི་ཙ་བྷ་ཡཾ་པུབྷྱཱ་ཡཱ་ཏེ་བི་ཤནྟི་ཙ་སཱ་བ་ཀཾ་ས་ར་ལ་ཀུ་ཊི་ལཱ་ནི་བ་ཏོག་ལྷུང་ལ་འཇིགས་པར་བྱེད་ཅིང་མ་ཡི་ནང་དུའང་རྣམ་པར་འཇུག། དྲང་ཞིང་གྱ་གྱུ་བུད་མེད་རྣམས་ཙཱ་ཏཱ་བི་ཙཱ་ར་ཤ་ཏི་སྦི་ཡ། །ཏ་ཐ་བི་ཡ་དི་ནིརྦ་རྣལྟ་བ་པ་བྱཱ་མི་ཀཾ་པྲི་ཡཾ། །ནི་རྣམ་པར་དཔྱད་པ་བརྒྱས་མི་ཤེས། །དེ་ལྟ་ན་ཡང་གལ་ཏེ་ཁྱོད། །འབད་ན་བུད་མེད་དེ་བལྟ་བྱ། །ཨི་ཏྱཱཀྟ་ཧརྨ་མཱ་རུ་བྱ་ཏྭཾ་ད་དན་ན་རེ་ཤྭ་ར། །ཏཱ་མེ་བ་ས་པ་རི་ཙཱ་ཡ་ཞེས་བརྗོད་མི་ཡི་དབང་པོ་ཡིས། །ཡང་ཐོག་འཛེགས་ནས་དེ་ལ་བལྟས། །ཡན་ལག་བཅད་པའི་གྲོགས་ལྡན་པཱ་པཾ་ཀྲྀཏྟཾ་ག་སཇི་ནཾ། །ས་ཙི་བེ་བྱཱཀྐི་ཏི་པ་ཏི་སྟབྲྀཏྟཱནྟཾ་ནྱ་བེ་ད་མ༑། སྡིག་ཅན་མ་དེ་ཡོངས་ཤེས་ནས། །བློན་པོ་རྣམས་ལ་ས་བདག་དེས། །དེ་ཡི་སྤྱོད་མཐའ་ངེས་པར་ཡཏ༑། ནཱི་པཾ་སཱ་བི་པ་རི་ཛཱ་ཡ་པཱ་པཱ་ཀྵ་ཎ་མ་དྷོ་མུ་ཁཱི། །ནི་རལྟ་པྲ་ཡ་ཡོ་བཤད། །དེས་ཀྱང་མི་བདག་ཡོངས་ཤེས་ནས། །སྐད་ཅིག་གདོང་དུད་སྡིག་ཅན་མ། །སྐྱེ་བོ་རྣ་བ་བསྒྲིབས་ཏུཙྪཱ་བི་ཧི་ཏ་ཤྲ་བ་ཎིཛ་ནི། །བི་ཤྭ་ཁ་ནཱ་མཱ་ན་ར་ནཱ་ཐ་སཱུ་ནུསོ་ཧཾ་བ་དྷུསཱ་སྱ་རྣམས་ཀྱིས། །མྱུར་དུ་བསྐྲད་དེ་རབ་ཏུ་སོང་། །ཁྱད་པར་ལོ་མ་ཞེས་པའི་མི་དབང་སྲས། །དེ་བདག་དེ་ཡི་ཆུང་ཙ་དེ་བ་དཏྟ། །ཐུ་ཏྭེ་ཏི་བྲྀཏྟ་ཀ་ཐི་ཏཾ་ཛི་ནེ་ན་བྷིཀྵ་པྲ་ཛ་སྟཙྩ་རི་ཏཾ་ནི་ནིནྡ། །མ་དེ་ལྷ་སྦྱིན། །ཞེས་པའི་བྱུང་ཚུལ་རྒྱལ་བས་བཤད་ཐོས་ནས། །དགེ་སློང་རྣམས་ཀྱིས་དེ་ཡི་སྤྱོད་པ་སྨད། །།
