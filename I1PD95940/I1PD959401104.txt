རབསྟན་འགྱུར། འཁྲི་ཤིང་། ཁིབྱ་བི་ཡུལྟ་སྱ་མ་མ་ནིསུ་ཁ་ཏྭཾ་ག་ཏཱ། །ཨི་ཡཾ་བྷིནེ་ཏྲ་ཧཱི་ན་སྱ་ཙི་ཏྲག་ལེ་བ་པ་དང་བྲལ་བ། །བདག་ལ་དཔལ་ནི་བདེ་མེད་གྱུར། །མིག་གིས་དམན་པ་དག་ལ་ནི། །རི་མོའི་གནས་ནིཕ་ལཱ། །ཏཱ་ཏ་སྱ་པཱ་ད་པ་ཏ་ནེ་ན་ཁ་རབླ་མཱ་ལཱ་བི་བྷཱ་ཛ་མཱན་དག་འབྲས་ེམེད་བཞིན། །ཡབ་ཀྱི་ཞབས་ལ་གཏུགས་ཚེ་སེན་མོའི་འོད་ཀྱི་འཕྲེང་བ་ཡིས། །རྣམ་པར་མུ་ཀུ་ཊ་སྱ་ཡ་ཐཱ་པུ་རཱ་བྷཱུཏ། །ཏཙྪཱ་ས་ན་ཤ་པ་ཎ་ཀུཎྜ་ལ་མཎྜི་མཛེས་པའི་ཅོད་པན་དང་ལྡན་དེ་ཡི་བཀའ་ཉན་པའི། །རྣ་རྒྱན་གྱིས་བརྒྱན་དཔལ་ནི་ཇི་ལྟར་བདག་ལ་ཏ་སྱ་ནོ་ཙ་ཀྲ་བནྟི་བི་བྷ་བེ་བྱ་ཏ་ཐཱ་མ་མ་ཤྲཱི། །ཨི་ཏ་སཾ་ཙི་ཏྱ་ས་སྔོན་གྱུར་པ། །ད་ལྟ་འཁོར་ལོས་སྒྱུར་བའི་འབྱོར་པ་དག་ལ་དེ་ལྟ་མེད། །ཅེས་བསམས་དེ་ཡི་འཇིག་རྟེན་ནི་ཛཾཀྲྀ་ཏཱ་ཀ་ན་ཀ་བརྩི་ཎཾ། །སཏྠ་ལོ་ཀོ་པ་གཱ་རཱ་ཡ་སནྣཱདྷཾ་ཀལྤ་ནི། །ཀུན་ལ་ཕན་སླད་རང་ཉིད་ཀྱིས། །དཔག་བསམ་རྐང་འཐུང་གསེར་གྱི་ཆར། །འབེབས་པ་དག་པཱ་ད་པཾ། །པཱ་ཛྲ་སཱམྦྷ་ཛུ་མུཏྤ་ཚུ་པྲ་ཡ་ཡོ་པི་ཏུ་རཱ་ཐ་མ། །ཏྲེ་ལོཀྱ་ཏུ་བསྟར་བྱས་ནས། །རྒྱལ་སྲིད་ཆེན་པོ་རབ་བཏང་སྟེ། །ཡབ་ཀྱི་བསྟི་གནས་དག་ཏུ་སོང་། །འཇིག་རྟེན་སཱ་ར་མི་ཤཱརྨ་ཙཱ་ཎཾ་བི་པུ་ལ་ཙེ་ཏ་ཧཱུཾ། །ཏསྨིནྣཱསྣ་ཚུ་མུཎྜི་ཚུ་པྲ་ཡཱ་ཏེ་གསུམ་གྱི་སྙིང་པོ་ཡི། །དབང་ཕྱུག་སེམས་ཡངས་རྣམས་ལ་རྩ༄༅། །རྒྱལ་སྲིད་ཆེན་པོ་བཏང་ནས་དེ། །མ་མ་ལ་ཡཱ་ཙ་ལཾ། །ཧེསྨཱ་ཀལྤ་དྲུ་མཔྲ་ཤྲཱི་པ་རི་པཱུམ་དི་པཾ་ཡ་ཡོ། །ལ་ཡ་ཡི་རིར་སོང་ཚེ། །དཔག་བསམ་ལྗོན་པས་ས་གཞི་ནི། །གསེར་གྱིས་ཡོངས་བཀང་མཁའ་ལ་སོང་། །གི་རཾ་ཛི་མཱུ་ཏ་བཱ་ཧ་ན༔པཱ་སྱ་ཤྲཱི་ཁཎྜ་མཎྜ་ནཾ། །བི་ཡོ་ག་ཏཱ་པ་ཏ་ཏཱ་ཛི་བི་སྤྲིན་གྱི་བཞོན་པས་རི་བོ་ནི། །དཔལ་གྱི་དུམ་བུས་བརྒྱན་པར་ཕྱིན། །ཕ་མའི་ཞབས་པད་བསྟེན་པ་ཡིས། །དྲོཔཱ་དཱ༴་སེ་བ་ཡཱ། ་། ཨ་ཏཱན་རེ་ཙནྡ་ན་བཝ་རཱི་ཎཾ་དི་དེ་ཤ་བཱ་འབྲལ་བའི་གདུང་བ་རབ་ཏུ་བཏང་། །སྐབས་དེར་ཙནྡན་འཁྲི་ཤིང་རླུང་གཞོན་གྱིས། །བསྐྱོད་ཅིང་ལཱ་ནི་ལ་ལོ་ལ་ཏཱི་ནཾ། །ཨུཙྪཱ་བི་ནི་ནཱ་མ་བྷི་ལཱ་ཥ་བྲྀཏྟཾ་ཛི་བྷཱ་ཛུ་ཤུགས་རིང་གླལ་བ་བསྟེན་རྣམས་ལ། །འདོད་པའི་གྲོགས་པོར་གྱུར་པ་དཔྱིད་ཀྱིས་ནི། །མངོན་པར་I།
