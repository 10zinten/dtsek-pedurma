རཱ་ཛ་པྲ་ཙྪནྣ་ཀཱ་མུ་ཀ། །སཔ་རཱ་ཏྲ་མཱ་ནཱ་ལོ་ཀྱེ་ཏསྦོ་ཏད་བྷ་བ་ནེ་རྒྱལ་པོ་འདོད་ལྡན་རབ་བསྒྲིབས་པ། །ལྟ་བར་མི་འོས་དེ་ཡི་ཁྱིམ། །དབེན་པར་ཞག་བདུན་གནས་ར་ཧ། །ལ་ཏེ་བ་པུཥི་ཏཱ་ཀཱ་ལེ་ཏ་མངྒབྷ་མ་བཱ་སྱ་སཱ། །ཙ་ཀྲེ་བི་དི་པར་གྱུར། །འཁྲི་ཤིང་མེ་ཏོག་ལྡན་པ་བཞིན། །དུས་སུ་དེ་ལས་མངལ་ཐོབ་དེས། །ངོ་ཚས་དམན་ཞིང་ཏ་བྲྀཏྟཱནྟཾ་ཏཾ་ལཛྫཱ་བ་ན་ཏཱ་ཤ་ནི། །ཨཱ་སནྣེ་བེདྨ་བི་ཙ་ཡེ་ད་ཏཱསྨི་ཚྭ་པ་ཏིརྨ་དལ་བུ་ཡིས། །དེ་ལ་སྤྱོད་མཐའ་རིག་པར་བྱས། །ཁྱིམ་ནི་སྤྱོད་ལ་ཉེ་བ་ན། །བྱུང་ན་བུ་ནི་ཡོངས་ཤེས་ཡོ༑། བྷཱ་བི་པུ་ཏྲ་པ་རི་ཛྙཱ་ན་པྲ་ཏྱ་ཡ་དཾ་གུ་ལཱི་ཡ་ཀ། ཡཱ་ཏེ་བྷཱ་སོཏ་བ་པུ་པའི། །ཡིད་ཆེས་སླད་དུ་སོར་གདུབ་དག། དེ་ལ་བྱིན་ནས་མི་བདག་སོང་། །མི་བདག་འོད་ལྡན་ལུས་ཥི་ནི་པ་ཏོ་སམྨ་ཏེ་ལོ་ཙ་ནཱ་ནཾ་ས་པོ་བྱདྲི་ར་ཧ་ཏི་མི་ནཀྟི་ཅན་མི་རྣམས་དག་གིས་བཀུར་བ་གཤེགས་གྱུར་ཚེ། །འཕྲལ་ལ་རབ་སྐྱེས་འབྲལ་བའི་རབ་རིབ་ཀྱིས་གཏིབས་བ་མི་ལརྨུ་ཁཱ༴། །སཱ་དྷུཏ་སཱ་ཡནྟ་ན་ཏ་ནུ་ཏ་རཱ་པཱ་ར་བཱ་ཏཱ་བྷི་བྷུ་བཞིན་གྱི་པདྨ་ཟུམ། །སྲོད་ལ་ཆར་དང་འཇམ་པའི་རླུང་གི་ཟིལ་གྱིས་མནན་པའི་པད་ལྡན་བཞིན། །ཏཱ་ས་ཐོཙྪ་ས་བྱ་ཏི་ཀ་རཱ་བ་ཏཱི་ཧནྟ་ཧཱི་ནཱཀྟི་ནཱི་བ། །ཀ་པོ་ལཾ་པཱ་ཎི་ཏ་མཆི་མ་དང་བཅས་ཤུགས་རིངས་རབ་ལྡན་དེ་ནི་དགའ་བ་དམན་པར་གྱུར། །ལག་པའི་སྟན་གྱིས་འགྲམ་ལེ་ན་སཾ་ཀལྤ་ན་མ་ཧཱི་པ་ཏི། །ན་ཐཾ་ཏཱ་ན་ར་མཾ་གེ་ན་བ་ཧནི་པ་དང་། །བརྟག་པ་ཡིས་ནི་ས་བདག་དང་། །ལུས་ཀྱི་གསར་དུ་ཉམ་ཆུང་ཉིད། །འཛིན་ཅིང་དེ་ནི་མི་མཱི་ལ་སཱ། །ཏ་ཏཿཀཱ་ལེ་ན་ཀ་ལྱཱ་ཎི་པྲ་ཏི་བིམྨོ་པ་མཾ་པི་ཏུ། ་། ཨ་ཛཱི་ཛ་ནི་ངེས་པར་དམན། །དེ་ནས་དུས་ཀྱིས་དགེ་ལྡན་མ། །དེ་ལས་ཀྱི་བུ་ནི་བཙས་གྱུར་ཏེ། །ཕ་ཡི་གཟུགས་ནཏ་སཱ་ཏ་ན་ཡཾ་བི་ན་ཡཾ་སཱ་དྷུ་དྷི་རི་བ། །བཌྷ་མཱ་ནེ་ཤ་ནི་སྟསྨིན་བརྙན་དག་དང་མཚུངས། །ལེགས་པའི་བློ་བཞིན་རྣམ་དུལ་བའོ། །དེ་ནི་ཟླ་བའི་དཀྱིལ་འཁོར་བཞིན། །ཀཱ་ལེ་བིམྲ་ཨི་བརྗེ་བེ། །བིམྱི་སཱ་ར་སྱ་པུ་ཏྲ་ཡ་མི་ཏི་ལོ་ཀེཥཱ་པ་པྲ་རིམ་གྱིས་བསྐྱེད་པར་གྱུར་པའི་ཚེ། །འདི་ནི་གཟུགས་ཅན་སྙིང་པོ་ཡི། །བུ་ཞེས་འཇིག་རྟེན་རྣམས་སུ་།
