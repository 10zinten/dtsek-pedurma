བསྟན་འགྱུར། འཁྲི་ཤིང༌། ཀིཡོ་སྣ་ཏུཾ་ན་དཱི་པ་ཊཾ། །སཱ་པི་པྲ་ཏེ་ཀ་བུདྡྷཱ་ཡ་ཏྭཾ་པལྟ་ཡ་ཏ་དན་ལ་ཀླུང་གི་ངོགས་སུ་སོང་། །སྐབས་དེར་མ་དག་འོངས་པ་ཡིས། །ཕྱག་ན་ལྷུང་བཟེད་རང་སངས་རེ༑། པྲ་ཎ་ཏཱ་པཱ་ཏྲ་ཧསྟ་ཡ་ཏ་དེབཱ་བྷི་མ་ཏཾ་ད་དོ། ་། ཨ་ཐ་བཱ་སྣ་རྒྱས༑། དེ་ཉིད་མངོན་པར་བཞེད་གྱུར་པ། །དེས་ཀྱང་རབ་ཏུ་བཏུང་ནས་བྱིན། །དེ་ནས་དེ་ཡི་བུ་ས་མ་ཡཱ་ཏནྟེ་ན་བུ་ཏསྣ་རི་ཏཾ་ཀླུ་དྷཱ། །ཛ་ན་ནཱི་བྷོ་ཛ་ནཱ་རཾ་བྷེ་ཡ་ཡཱ་ཙེ་ནི་བཀྲེས་ནས། །རིངས་པར་ཁྲུས་བྱས་ཡང་དག་འོངས། །ཟས་ནི་རྩོམ་པར་བྱེད་པའི་ཚེ། །མ་ལ་རང་གིས་ཛ་མཱུ་ལ་ཀཾ། །པུ་ཙྩཾ་པུ་ཏཱ་ནུ་མོ་ད་པཱ་ཏརྨ་ཡཱ་ཏི་ཐ་ཡེཉམི་ཏཾ། །ཨི་ཏི་མཱ་ཏུརྦ་ལ་ཕུག་བླངས། །བདག་གིས་མགྲོན་ལ་དེ་བྱིན་བུ། །དགེ་ལ་རྗེས་སུ་ཡི་རང་བྱོས། །དེ་སྐད་མ་ཡི་ཚིག་ཐུ་ཏཱ་སོ་ཧྥུ་བྲིདྷ་ཨི་བེ་ཥཱ་ཎ། །ས་དྷོ་བི་སཱུ་ཙ་ཀཱརྟ་སྱ་མནྡཱ་ལ་ཀ་ཐོས་ནས། །དེ་ནི་མདའ་ཡིས་ཕུག་བཞིན་གྱུར། །མགྲོན་ཁྱོད་འཕྲལ་ལ་མ་ཞུ་བས། །གཟིར་ཅིང་དཀུ་མ་ནལླི་ཀཾ། །ཀུ་ཧཱུཾ་བྷི་ཏཱ་བི་ནིམཱ་ཏུ་པཱ་ཎེས་ཧ་ཏཱ་བཱ་ཏི་ཐེ། །།ཨི་ཏི་ནི་ཕུག་གྱུར་ནས། །མི་ཆུང་བདག་གི་ལ་ཕུག་དག། སྲོག་དང་ལྷན་ཅིག་ཕྱིར་འབྱུང་ཤོག། དེ་སྐད་ཏ་སྩལྟ་པཱ་པ་སྱ་བཱཀཱ་ཏུ་སྐྱེ་ཎེ་བྷི་ཡ་སཱ། །བི་སྐུ་ཙི་ཀི་བ་པརླནྟེ་བ་བྷུ་བཱ་པ་སྡིག་པ་ཐོབ་པ་དེའི། །ཤིན་ཏུ་རྩུབ་པའི་ཚིག་དེ་ཡིས། །མཇུག་ཏུ་མ་ཞུའི་ནད་ཉིད་ནི། །སྐྱེ་བ་གཞན་པ་ར་ཛརྨ་སུ། །པཱཀླུ་ཅུནྟ་ར་པཱ་ཀེ་ན་ས་ཨེ་བཱ་རྱ་པྲ་པེ་ན་ཛིཏ། །བི་པུ་རྣམས་སུ་གྱུར། །དེ་ཉིད་ད་ལྟ་གསལ་རྒྱལ་ཏེ། །སྔ་མའི་བསོད་ནམས་གཞན་སྨིན་པས། །རྒྱལ་སྲིད་ལཾ་རཱ་ཛུ་མཱ་པཱ་རྱ་ཏ་ཡི་བཱནྟེ་ཀྵ་ཡཾ་ག་ཏ། །སཾ་སཱ་ར་པ་ཐ་པནྟྲ་ན་མེ་བ་ཡངས་པ་ཐོབ་གྱུར་ནས། །མཇུག་ཏུ་དེ་ཉིད་ཀྱིས་ཞིག་གྱུར། །འཁོར་བའི་ལམ་གྱི་མགྲོན་པོ་ལ། །དགེ་ཀརྨ་་ཤུ་བྷཱ་ཤུ་བྷཾ། །སཱ་ཐེ་ཡ་མེ་བ་ཧསྟ་ཙྭ་བྷོ་གཱ་ཡི་བོ་བ་བ་བྱ་དང་མི་དགེའི་ལས་ཁོ་ན། །ལོངས་སྤྱོད་སླད་དུ་ཉེར་བསྒྲུབས་པའི། །ལམ་བརྒྱགས་བཞིན་དུ་ལག་ན་ཏེ༑། ཨི་ཏི་ཐུ་ཏྭཱ་བྷ་ག་བ་ཏསྟ་é་པ་ཐུཉྩ་ཏདྨ་ཙ། །ཨེ་བ་མེ་གནས༑། དེ་སྐད་བཅོམ་ལྡན་དེ་ཡི་གསུང་ཚེ། །ཐམས་ཅད་བདེན་པ་ཐོས་ནས་ནི། །དེ་དེ་བཞིན་།
