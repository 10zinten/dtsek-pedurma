ལ་ཡི་ཏཱ་པྲ་སསྣ་དྷཱི། ་་། ཡཱ་ཏ་ཀཱ་ལེ་ན་ཀཱ་ལུ་་མ་དྷནྣཱ་བྷི་ར་ཆོས་ཀྱིས་སྐྱེ་རྒུ་བསྐྱངས་བྱས་ནས། །དུས་ཀྱིས་ཆོས་མིན་མངོན་འདོད་ཅིང་། །རྙོག་པ་དག་དང་ལྡན་ཏོཤབྷ་བཏ། །ཀ་ལུ་ཎཾ་ཀཱཉྩ་ན་རུ་ཙ་བྱ་སྟི་བིརླུདྲི་ལཱ་སི་ནཱི། །ན་ཀ་པར་གྱུར། །གསེར་འོད་གསལ་བའི་གློག་དག་ནི། །རྣམ་པར་རོལ་པའི་སྤྲིན་འཕྲེང་བཞིན། །ཕུན་སུམ་སྱ་ཀུ་རུ་དེ་ལགླིརྨེ་གྷ་མཱ་ལེ་བ་མཱ་ན་སཾ། ་། ཨ་ནཱ་ཡཏྟཾ་མ་ཧཱ་མ་ཏྱནྟེ་རུ་ཀོ་བྷི་ཚོགས་པས་སུ་ཡི་ཡིད། །དྲི་མ་ཅན་དུ་བྱེད་མ་ཡིན། །ཧེ་རུ་ཀ་དང་བྷི་རུ་ཀ། བློན་ཆེན་དག་གིས་ཆོས་རུ་ཀ༔་ཏཾ། །ཨ་དྷརྨ་ཀརྨ་ནི་ར་ཏཾ་ཀྲུདྡྷཾ་ཏ་ཏྱཱ་ཛ་ཏུཔྲ་ཧཱུཾ། །ཏཏ་པ་དེ་བི་ཧི་ཏོ་མིན་གྱི། །ལས་ལ་དགའ་ཞིང་དབང་མ་གྱུར། །ཁྲོ་ལྡན་ཇོ་བོ་དེ་བཏང་གྱུར། །དེ་ནས་གནས་སུ་རྒྱལ་པོ་རཱ་ཛཱ་ས་ཙི་བོ་དཎྜ་མུངྒ་རོ། །ཙིཏྟཱ་ནུ་བྲྀཏྟ་ཀུ་ཤ་ལོ་སཱི་རཾ་སཏྟཾ་ཏ་ཡིས། །དབྱུག་པ་ཐོ་བ་བློན་པོར་བཅུག། སེམས་ཀྱི་རྗེས་འཇུག་མཁས་པ་ཡིས། །དམན་པར་ཆགས་པ་མུ་ཙ་ཏུ། །།པྲ་ཛཱ་རཉྫ་ན་སཾ་སཀྵརཱ་ཛ་དོརྫ་རྩ་བཱ་དི་ན། །དེ་ལ་སྨྲས། །སྐྱེ་རྒུ་རྣམས་ལ་མཛའ་བས་ཡང་དག་ཆགས། །རྒྱལ་པོ་གཞུང་ངན་དག་ཏུ་སྨྲ་བར་བྱེད། །སྭ་ཡ་ཤཥྛ་ས་ནཱ་ཡི་བ་ཛཱ་ཡནྟེ་དྷཱཏཱ་མནཱི་ཎ། །ན་དྷརྨ་གཡོན་ཅན་བློན་པོ་རྣམས་ཀྱིས་རང་ཉིད་ཀྱི། །གྲགས་པ་རབ་ཏུ་སྒྲོགས་པར་བྱེད་པ་ཡིན། །གང་ཞིག་རྗེ་ན་སུ་ཁཾ་ནཱཐཾ་ན་ཀིཏྟ་ཡེ་ན་ཛཱི་བི་ཏཾ། །ག་ཎ་ཡནྟི་པྲ་བྷོ་རརྗེ་ཏེ་པྲྀ་ཏྭཱ་བྷ་བོའི་དོན་དུ་ནི། །ཆོས་ལ་མི་རྩི་བདེ་ལ་མིན། །དོན་མིན་གྲགས་མིན་འཚོ་བ་མིན། །དེ་དག་མཆོག་ཏུ་བྱ་བྷཀྵ་ཡ། །ནཱ་ཁཎྜི་ཏཱ་ཀླུ་ཡི་ཏཱ་ན་ཏཔཱ་ནཱ་སྱ་པཱི་ཌི་ཏཱ། །ཀུཊྚནྡྷཟི་གུས་པའི་གཡོག། ངེས་པར་ཏིལ་མཚུངས་སྐྱེ་རྒུ་རྣམས། །རྒྱལ་པོའི་དོན་བྱེད་མ་བཅག་པས། །མ་ཡིན་ཀྲི་ཡཱ་རཱ་ཛྙཱསྟི་ལ་ཏུ་ལྕཀི་ལ་པྲ་ཛཱ། །ཨི་ཏི་བྲུ་བཱ་ཎོ་ཏོ་རཱ་མ་ཉམས་པས་ཀྱང་མིན། །མ་གདུངས་པས་མིན་མ་གཟིར་མིན། །ཞེས་སྨྲས་དེ་དག་རྒྱལ་སྲིད་ཀྱི། །ཙཱ་ནི་ཡུལྟོ་རཱ་ཛ་ཙིནྟ་ནེ། །ལོ་བྷཏ་པྲ་བྲྀཏྟོ་དུམྲྀ་ཏོ་ཧནྟཾ་ནིཤ་ར་ཎཔྲ་བསམ་པ་ལ་ནི་རྒྱལ་པོས་བསྐོས། །།ཆགས་ལས་སྤྱོད་ཚུལ་ངན་པ་ཡིས། །སྐྱབས་མེད་སྐྱེ་རྒུ་གཞོམ་པར་—།
