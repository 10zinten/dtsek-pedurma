བསྟན་འགྱུར། །འཁྲི་ཤིང་། ཀི་ཎིསྔ་ནཾ། །ཏེ་མ་ཧོ་ད་དྷི་མུཏྟཱི་བྱ་པཱཙྪཱ་བྷ་ག་བ་ཏོནྟི་ཀཾ། །ཏགླུཀྟ་བཏང་བར་གྱུར། །དེ་དག་ཆུ་གཏེར་ཆེ་སྒྲོལ་ནས། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་ཕྱིན། །དེ་ཡིས་མུ་རཏྣ་ནི་ཀ་ར་པྲ་ཎ་མྱ་སྨི་བྱ་བེ་ད་ཡན། །བ་ཎིབྷིཀ་ཐི་ཏྭཾ་ཐུ་ཏྭཱ་ཏཥ་ཐཾ་ཏིག་རིན་ཆེན་ཚོགས། །ཕྱག་འཚལ་ནས་ནི་དེ་ལ་ཕུལ། །དེར་ནི་ཚོང་བས་དེ་ཡི་གཏམ། །བཤད་པ་ཐོས་ཏ་ཏྲ་བྷི་ཎཱ། །ཨཱ་ནནྡ་ནཱམཱ་པྲཎོ་ཐ་བ་བྷཱ་ཥ་བྷ་ག་བཱན་ཛི་ན། ་། ཡཱ་ནས་དགེ་སློང་ནི། །ཀུན་དགའ་བོ་ཞེས་བྱ་བས་དྲིས། །དེ་ནས་བཅོམ་ལྡན་རྒྱལ་བས་གསུངས། །གང་སོ་པུ་རཱ་རོ་ཧི་ཏཱ་ལྱཱ་དཱ་སི་ཤཱཀྟ་གྲོ་ཧ་བྷ་བཏ། །སེ་བ་མུཀྐཱ་ལ་ཏཱ་ཛཱ་ཏཱ་སཏྟརྨ་འདི་ཤཱཀྱའི་ཁྱིམ་དུ་སྔོན། །རོ་ཧི་ཏ་ཞེས་འབངས་མོར་གྱུར། །དེ་ཉིད་མུ་ཏིག་འཁྲི་ཤིང་དུ། །སྨོན་ལམ་དམ་པྲ་ཎི་དྷཱ་ན་ཏ། །ཨ་བྷུད་མ་ཧཱ་དྷ་ནོ་ནཱ་མ་བཱ་རཱ་ཎ་སིཾ་བ་ཎི་ཀ་པུ་རཱ། །ཏ་པའི་ལས་ཀྱིས་སྐྱེས། །ནོར་ཆེན་ཞེས་པའི་ཚོང་པ་ནི། །བཱ་རཱ་ཎ་སཱིར་སྔོན་བྱུང་གྱུར། །དེ་ཡི་ཁྱིམ་ཐབ་སྱ་རཏྭཱ་པ་ཏཱི་ནཱ་མ་པདྡྷི་པུ་སྐྱོ་ཙི་ཏཱ་བྷ་བཏ། །སཱ་པ་ཏོ་པཉྩ་ཏཱ་ཡཱ་ཏེ་ནིཔུ་ཏཱ་བསོད་ནམས་ཅན། །རིན་ཆེན་ལྡན་ཞེས་བྱ་བར་གྱུར། །བདག་པོ་ལྔ་པ་ཉིད་གྱུར་ཚེ། །བུད་མེད་དེ་སྐུ་པ་ཤེ་ཁ་རེ། །པཱུ་ཛཾ་ཀྲྀ་ཏྭཱ་མ་ཧཱ་ཧཱ་རཾ་བྷཏྟི་ཡུཀྐཱནྱ་བེ་ད་ཡཏ། །ཏེ་ཡིས་མཆོད་རྟེན་རྩེར། །དོ་ཤལ་ཆེན་པོ་གུས་ལྡན་པས། །མཆོད་པ་བྱས་ནས་ཕུལ་བར་གྱུར། །བསོད་ན་པུ་ཎྱ་བི་པཱ་ཀེ་ན་སཾ་ཧ་ལཱ་དྷི་པ་ཏེསུ་ཏཱ། །ཛཱ་ཏཱ་མུཀྐཱ་ལ་ཏཱ་སེ་བ་པཱམྦྷ་ནམས་རྣམ་སྨིན་དེས་དེ་ཉིད། །སིང་ག་ལ་ཡི་བདག་པོ་ཡི། །བུ་མོ་མུ་ཏིག་འཁྲི་ཤིང་གྱུར། །ཡོངས་སུ་ཙ་པ་རི་ནིབྲྀ་ཏི། །སེ་བ་ཛརྨ་ནི་ཙཱ་ནྱསྨིནྣི་ཤཱརྱ་མ་ད་མོ་ཧི་ཏཱ། །མྱ་ངན་འདས་པའང་ཐོབ། །དེ་ཉིད་སྐྱེ་བ་གཞན་ལ་ནི། །ཕྱུག་པས་རྒྱགས་ཤིང་རྨོངས་གྱུར་ནས། །པཱུ་ཛྙཱ་དྷི་ཀྵ་པ་ད་ཁཱ་བྷཱུཏ་དཱ་སཱི་ཏེ་ནཱ་ཏི་བཙྩ་རཾ། །ཛརྨ་ཏེ་ཛ་ནེ་ནོཙོ་ཡརླད་མཆོད་ལ་སྨོད་ཅིང་སྒྲིན་གྱུར་པ། །དེ་ཡིས་ཡུན་རིང་@འབངས་མོར་གྱུར། །དགེ་དང་མི་དགེའི་ལས་གང་ཀརྨ་ཤུ་བྷཱ་ཤུ་བྷཾ། །ཏ་སྱ་ཏ་སྱ་ས་ཏད་རཱུ་པཾ་ཧཱུཾ་ཀེ་པ་རི་ཎ་ཏི་ཕ་ལཾ། །གང་། །སྐྱེ་བའི་ཞིང་ལ་སྐྱེ་བོས་བཏབ། །དེ་དང་དེ་ཡི་རང་བཞིན་དེ། །ཡོངས་སུ་སྨིན་པའི་འབྲས་བུ་སྤྱོད། །—།
