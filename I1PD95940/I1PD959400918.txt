གྲ་ཧེ་ན་ཛ་ར་ཡཱ་བཱཏྨ་ལྕེ་ནེ་བ་བཱརྨ་ཏེ། །ཀིཾ་ཙིཏྟཾ་ཀོ་ཙ་ཀུ་ཊི་ལནས་བཟུང་བྱས་ཏེ། །མཉེས་གཤིན་གྱིས་ནི་ཟློག་པ་བཞིན། །རྒན་པོ་དལ་གྱིས་ཅུང་ཟད་ནི། །པྲ་ཡཱ་ཏིསྦ་བི་ར་ཤ་ནི། །ཧཱ་རི་ཏཾ་ཡོ་བ་ན་མཎི་བཱིཀྐཱ་མཱ་ཎ་ཨི་པཱ་བ་ནོ། །འཁུམས་ཤིང་སྐུ་བོར་རབ་གྱུར་པ། །ས་ལ་ལང་ཚོའི་ནོར་བུ་ནི། །སྟོར་ཉིབ་འཚོལ་ཡིབར་བྱེད་པ་བཞིན། །བྲིདྡྷེ་ན་པ་ར་ལོ་ཀཱརྦ་ཛཱ་ཡཱ་ནཱི་ཏཱ་ད་དྷཱི་མ་ཏཱ། །པ་ར་བྷོ་ག་པྲ་ཎ་ཡི་ཎཱི་ཏཏྟ་རྒན་པོ་བློ་ལྡན་མ་ཡིན་པ། །འཇིག་རྟེན་ཕ་རོལ་དོན་ཤེས་པས། །གཞན་གྱི་ལོངས་སྤྱོད་ལ་སྦྱར་གང་། །དེ་བ་ཀ་རོ་མྱ་ཧཾ། །ཨནྟ་ཀྲྀ་ཧ་མ་ཏེ་ཏཱསྨིནྣ་ཤཀྱི་ཙོདྨ་ཀཱ་མི་བྷི། །དེ་ཉིད་བདག་གིས་རབ་ཏུ་བྱ། །འདི་ནི་ཁྱིམ་ན་གནས་པའི་ཚེ། །རྐུན་པོའི་འདོད་ལྡན་དང་མཛའ་བས། །པྲ་མ་ནིརྡ་ཡ་སཾ་བྷོ་ག་ནི་རརྒ་ལ་སུ་ཁམྨ་ཡཱ། །ཨི་ཏི་སཾ་ཙིནྟ་པ་བྷེ་ཏྱ་བརྩེ་མེད་ལོངས་སྤྱོད་མ་བསྡམས་པ། །བདེ་བ་དག་གིས་སྤྱོད་མི་ནུས། །ཞེས་བསམས་རིགས་བཟང་ཤ་ནིཔ་ཏི་མཱ་བྷཱ་ཥཊ། །ལཛྫ་མཱ་ནེ་བ་བི་ན་ཡཱ་དཱ་བྷི་ཛཱ་ཏྱ་ནུ་རྗེས་མཐུན་ཞིག། ངོ་ཚ་ལྡན་པ་བཞིན་དུ་དེས། །དུལ་བ་ཡིས་ནི་མངོན་ཕྱོགས་ཏེ། །དལ་གྱིས་བདག་ཀཱ་རི་ཎཾ། །ཛཾ་ཧ་པལྟེ་ན་བྷ་བ་ཏཱ་ནིཙྪཱ་པཱ་ར་སཱུ་ཁི་ཎི་ཎཱ། །ཏསྟེ་པོ་ལ་སྨྲས་པ། །བྱ་མེད་བདེ་བ་འཚོལ་བྱེད་པ། །ཁྱོད་ནི་ཁྱིམ་དུ་ཆགས་པ་ཡིས། །བཟོད་པར་ནཱ་ཀྲྀ་ཥྱ་དཱ་རི་དཱ་མཱ་ནཱི་ཏཾ་ཏ་བ་དུས་ཧཾ། །སཾ་བྷོ་ག་དྭེ་ཏི་ནསྟི་བྲ་མཱ་དགའ་ཉེབའི་དབུལ་པོ་ཉིད། །ཁྱོད་ཀྱི་ལག་པས་དྲངས་ནས་བླངས། །གང་ཞིག་རྩོམ་པ་ལ་སྡང་ཞིང་། །ལ་ཙྩཾ་ཡ་སྱ་བ་ལུ་བྷཾ། །བ་ཧུ་བྱ་ཡ་པྲ་བཱ་ཧཱདྷ་བི་བཱ་ཧཾ་པ་ཀ་རོ་དྲག་པོའི་སྙོམ་ེལས་དག་ལ་དགའ། །རབ་མང་འགོད་པའི་བརྒྱུད་ལྡན་པའི། །ཁྱིམ་ན་གནས་པ་དེས་ཏི་ཀཾ། །ཡ་ཏྲ་ལ་སྩཛཾ་ཧ་པ་ཏིཀྲྀ་ཧ་ཀོ་ཎཾ་ན་མུཉྩ་ཏི། །དྷརྨ་ཛཱ་ནཱ་ཅི་བྱེད། །གང་དུ་ཁྱིམ་བདག་སྙོམ་ལས་ལས། །ཁྱིམ་གྱི་ཟུར་ནི་མིག་གཏོང་བ། །དེར་ནི་བུད་མེད་ཡ་ནིརཱནྟུ་མུབྷཱ་སྟ་ཏྲ་ཀི་མཾ་ག་ནཱ། །སོཏྟཱ་ཧཔུ་རུ་ཥོ་ཡ་ཏྲ་བྱ་བ་ཧཱ་ར་ར་རྨོངས་པ་རྣམས། །ནོར་སྒྲུབ་སླད་དུ་འབྱུང་ངམ་ཅི། །གང་ན་སྐྱེས་པ་སྤྲོ་ལྡན་པས། །ཕྱི་ཡི་ཐ་སྙད་ལ་།
