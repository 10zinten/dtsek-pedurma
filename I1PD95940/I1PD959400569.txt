མ་བྷུ་བ་ནནྡ་བི་ཧྭ་ལ། །།མརྟུ་ཛཆ་་ཀ་ཐཾ་ཏེ་ན་སཱཝཱ་ག་མ་ན་ཀཱ་བརྗོད་ཅིང་ཀུན་དགའ་འཁྲུགས་པར་གྱུར། །མི་ཡི་སྐྱེ་བའི་གཏམ་དང་ནི། །རང་ཉིད་འོངས་པའི་རྒྱུ་ར་ཎཾ། །ཐུ་ཏྭཱ་ནི་བེ་དི་ཏཾ་ནཱ་ག་བ་རི་ཥ་ཛུ་ཛ་གཱ་ད་ཏཾ། །ཙིནྟ་མ་ཎི་ར་དག་ཀྱང་། །དེ་ཡིས་བཤད་པ་ཐོས་གྱུར་ནས། །ཀླུ་ཡིས་ཡོངས་འཁྱུད་དེ་ལ་སྨྲས། །བུ་གཅིག་ཡིད་བཞིན་ཡཾ་པུ་ཏྲ་མ་མ་མོ་ལི་བི་བྷུ་ཥ་ཎྞཾ། །གྲོ་བྱ་ཏྭཾ་ཏ་བ་སཾ་ཀ་ན་བནྟཾ་ཀརྟུ་ནོར་བུ་འདི། །བདག་གི་མགོ་བོ་ཉིད་ཀྱི་རྒྱན། །ལོང་ཞིག་ཁྱོད་ཀྱི་བརྟག་པ་ནི། །འབྲས་མེད་བྱེད་པར་མུཏྠ་ཧེ། །དེ་ཡཀྲྀ་ཏ་ཛ་གཏྠཾ་ོ་མ་མེ་བཱ་ཡཾ་སུ་ནསཱུ་ཡཱ། །ཨི་ཏུཀྵསྨེ་ད་དོ་མི་སྤྲོའོ། །འགྲོ་བའི་བྱ་བ་བྱས་ནས་འདི། །ཁྱོད་ཀྱིས་བདག་ལ་སླར་སྦྱིན་བྱ། །ཞེས་བརྗོད་གཙུག་གི་རིན་དི་བྱ་ཙཱུ་ཌ་རཎཾ་བི་མུ་ཙྪ་ས། །ཙི་ནྟཱ་མ་ཎི་ས་མ་དཱ་ཡ་ཀུ་མཱ་རསྟ་ར་ཆེན་མཆོག། དེ་ཡིས་བཀྲོལ་ནས་དེ་ལ་བྱིན། །ཡིད་བཞིན་ནོར་བུ་ཉི་མའི་འོད། །ཡང་དག་བླངས་ནས་ཎི་པྲ་བྷི། །ཧྲཋཔྲ་ཎ་མྱ་ནཱ་གེནྡྲ་ཡ་ཡོ་པྲ་བ་ཧ་ཎཱནྟི་ཀཾ། །སམུ་དྲ་གཞོན་ནུ་ནི། །དགའ་བས་ཀླུ་དབང་ལ་བཏུད་དེ། །གཟིངས་དང་ཉེ་བ་དག་ཏུ་སོང་། །དེ་ཡི་གཏམ་དེ་བ་ཏཱ་ཏ་ཏྲ་ཏཾ་དྲྀཏྟཱ་ཐུ་ཏ་ཏཥ་ཐཱ། །ཨུ་བཱ་ཙ་ཀཱི་ཛཾ་ཤཿསཱ་དྷོ་བཱསྟ་སྦྱིན་ཐོས་རྒྱ་མཚོ་ཡི། །ལྷ་ཡིས་དེར་ནི་དེ་མཐོང་ནས། །དགེ་བ་ཁྱོད་ཀྱིས་ཐོབ་པ་ཡིས། །ཡིད་བཞིན་ནོར་མ་ཎི་པཱུ་ཡཱ། *། ཏེ་ནེ་ཡཾ་ཀཱ་ཤ་བཱཀྱ་མེ་དྷཱ་མི་ཐུ་རི་ཛེ་ཏེ་ན་ཙ། །པདྨ་པཱ་ཎེ*：དེ་ཡིས་གསལ་བར་འདི་བརྗོད་ཚེ་སོགས་བློ་ཀ་གཉིས་ཀྱི་རྒྱ་སྐད་ཆད་པ་ཟ་ཧོར་གྱི་བརྗེས་སྡེབ་སྦྱོར་ཁ་སྒོ་ཏ་ཡང་མར་བཀོད་པ་སླར་རྒྱ་དཔེ་རྙེད་ན་ཞུས་དག་བྱེད། །བུ་ཅི་འདྲ་སྨྲོས། །དེ་ཡིས་གསལ་བར་འདི་བརྗོད་ཚེ། །ཟོལ་མེད་བསམ་པ་དྲང་པོ་དེས། །ལག་པའི་པདྨ་པྲ་ནཱ་སྱ་ཏྭཾ་དེ་ཤ་ཙིནྟཱ་ཏོ་མཱ་ཎི་ཀཾ། །ཏྲཏྟེཛྲཏྠ་རདྡྷོཏྟཱ་མ་ཏཱིཙྩཾ་རབ་གདེངས་ནས། །ཡིད་བཞིན་གྱི་ནི་ནོར་བུ་བསྟན། །རིན་ཆེན་མཆོག་དེ་དེས་མཐོང་ནས། །ཕྲག་གནྣཱསཾ་ཀལྤ་ཙ། །ཏྱ་ཡོནྡྷ་ཛ་ཤ་ཡེ་ནཱཉྫ་ལྟ་སྨཱ་ཏྭཾ་ཏ་རན་པ་ཏུཾ། །དོག་དྲི་མའི་བསམ་པ་ཡིས། །དེ་ཡི་ལག་པ་ལག་པ་ཡིས། །ཐལ་མོ་བསྣུན་ནས་ཆུ་གཏེར་དོར། །།
