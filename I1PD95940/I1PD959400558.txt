ལོའི་ཙ་ཏྭཱ་རསུ་ག་ཏོ་དིཎྞཾ་སདྷཱརྨ་ས་ར་མཱ་མྲྀ་ཏཾ། །ཨ་བྷི་ར་ཎྱ་ཁྱུ་སྟོབས་དང་ཤིན་ཏུ་སྟོབས་དང་ནི། །དབུགས་དང་དབུགས་ཆེན་ཞེས་བྱ་བ། །སྲིང་མོ་མངོན་དགའ་ཞེས་ཡཱ་སཱ་པཱ་པྲ་རི་ཏཱཐོ་ཏུ་མཱ་ས་བྱ། །ཏེ་བ་ལཱ་ཏི་བ་ལཱ་ཤཱ་ས་མ་ཧཱ་ཤཱ་པ་ཡིས། །བསྐུལ་ནས་དེ་དག་མཉམ་དུ་ནི། །བདེ་གཤེགས་ཀྱིས་གསུངས་དམ་པའི་ཆོས། །།མཆོག་གི་སཱ་བྷི་དྷཱ༔ས་མཾ། །ཀྲ་ཀུཉྩ་ནྡ་སྱ་སུ་དྷི་ཡཀཱ་ལེ་བྷ་ག་བ་ཏབུ་རཱ། །བདུད་རྩི་ཉན་དུ་འོངས། །སྔོན་གྱི་དུས་ན་བཅོམ་ལྡན་འདས། །བློ་བཟང་འཁོར་བ་འཇིག་དང་ནི། །ཀ་ན་ཀཱ་ཏྱཱ་སྱ་ཙ་མུ་ནེཀཱ་བྱ་པ་སྱ་ཙ་ཡེརྟ་ཏ། །ནཱ་ཛགླུཤྲཱི་སུ་ཁཱ་ས་ཀཿགསེར་ཐུབ་ཅེས་ནི་བྱ་བ་དང་། །དེ་བཞིན་དགྲ་བཅོམ་འོད་སྲུངས་ལ། །ཉན་དུ་སྲིང་མོས་བསྐུལ་གྱུར་ཐོ་ཏུཾ་ཡ་སཱ་ཏཱ་ཨ་པི། །ཏེ་པུ་ཅུ་པ་རི་ཎཱ་མེ་ན་པྲསྟཀྱང་། །དཔལ་གྱི་བདེར་ཆགས་འོངས་མ་གྱུར། །དེ་དག་བསོད་ནམས་ཡོངས་སྨིན་ནས། །ཤཱཀྱ་ཐུབ་ཤཱཀྱ་མུ་ནཔུ་རཱ། །ཏེ་ཥཱ་པྲ་ཎ་མྱ་ཤཱསྟཱ་རཾ་ཙ་ར་ཎཱ་ལཱི་ན་མོ་ལི་ཏུ། །བི་དྷཱ་ཡ་པའི་སྔོན་དུ་འོངས། །དེ་དག་མི་ཡི་ལུས་བསྒྲུབས་ཏེ། །སྟོན་པ་ལ་ནི་ཕྱག་བཙལ་ནས། །ཞབས་ལ་མགོ་མཱ་ནུ་༈་རཱུ་་མུ་ས་བིཝེ་པརྟ་དི། །སདྷརྨ་མ་ཡ་ཡོ་ཐོ་ཏུཾ་ཀོ་ཡིས་གཏུགས་བྱས་ཏེ། །འཁོར་དེར་ཉེ་བར་འདུག་པའི་ཚེ། །གསལ་རྒྱལ་ཀོ་ས་ལ་ཡི་དབང་། །དཔལ་ཤ་ལེནྡྲཔྲ་སེ་ན་ཛིཏ། །ལགླི་མ་དརྨི་ཏཙྪ་ཡཾ་ནི་བཱམྨ་ཙཱ་ཏྲ་ཙཱ་མ་འབྱོར་རྒྱགས་འཛུམ་མཛེས་པ་ཡི། །གདུགས་དང་རྔ་ཡབ་བཟློག་ནས་ནི། །དམ་ཆོས་ཉན་དུ་ཡང་དག་རཾ༑། ཤཱyཔཱ་ད་པྲ་ཎཱ་མཱ་ཡ་བི་ཤ་ཏསྟ་སྱ་སཾ་ས་དི། །ཨ་བ་ཀཱ་ཤཾ་ན་ཏཱཉྩ་འོངས། །སྟོན་པའི་ཞབས་ལ་ཕྱག་འཚལ་སླད། །ཚོགས་སུ་ཞུགས་པ་དེ་ལ་ནི། །མི་ཡི་བདག་ལ་གུས་པ་ཀྲུ༔སཟློ་ཚིགས་པ་ཏི་གོ་ར་བཱཏ། །ཏ་སྭཱ་བྷི་ནཾ་བྱ་མཱ་ན་སྱ་བཙྪཱ་ཐ་མ་ལས། །ཀུན་གྱིས་བཏུད་ཅིང་གོ་སྐབས་ཕྱེ། །རིག་གནས་བླ་མ་མི་རྣམས་ཀྱིས། །མངོན་པར་དགའ་གུ་རོརྫི་བྷི། །ནཱ་ག་རཱ་ཛ་སུ་ཏཱཉྩ་ཀུནྣཱ་བ་ཀཱ་ཤཾ་ན་སཏྠཾ་ཏཾ། *། ཏ་ཡ་*：དེ་དག་དྲེགས་པས་ཁེངས་པ་དང་ཞེས་པའི་སྐད་དོད་ཆད་པ་འདར་ལོས་བཅུག་པ་རྒྱ་དཔེ་རྙེད་ན་ཞུས་དག་བྱེད། བར་བྱས་དེ་ལ། །ཀླུ་ཡི་རྒྱལ་པོའི་བུ་རྣམས་ཀྱིས། །གོ་སྐབས་བཀུར་སྟི་བྱས་མ་གྱུར། །དེ་དག་དྲེགས་—།
