པུ་ཊཾ་བུ་རསྦི་ཏཾ། །ཏ་ཏཀྲྀ་ཏ་པྲ་ཎ་མ་སྱ་ཏ་སྭཱ་བི་བྷ་ག་བཱན་པུ་མདུན་ན་གནས་པའི་བུ་མ་མཐོང་། །དེ་ནས་སླར་ཡང་བཅོམ་ལྡན་གྱིས། །རབ་ཏུ་བཏུད་བྱས་དེ་ཡི་ན༑། དྷརྨྱ་ཡ་ཀ་ཐ་ཡ་མོ་ཧཾ་ཏཱི་ཥ་ཏ་མ་ཨི་བཱ་ཧ་རཏྲ། །ཏཱི་བྲ་མོ་ཧསྟ་ཡང་། །རྨོངས་པ་ཆོས་གཏམ་རིགས་པ་ཡས། །འོད་ཀྱི་མུན་པ་བཞིན་དུ་ཕྲོགས། །དེ་ནས་རྨོངས་བྲལ་ཏཔུ་ཏྲ་ས་དྲིསྠཱ་བི་མ་ལ་གྱུ་ཏི། །ཤཱསྟུཉྩ་ཀཱ་རཱ་ནུ་མ་ཏཔྲ་ཎ་ཡཱ་ད་དྷི་དེ་ཡིས་བུ། །དྲི་མེད་འོད་ལྡན་མཐོང་གྱུར་ནས། །རྗེས་སུ་འདོད་ཅིང་གུས་པ་ཡིས། །སྟོན་པ་ལྷག་པར་བཱ་ས་ནཾ། །བྷ་ག་བ་ན་ཐ་ཏཛཾ་ཧེ་ཀྲྀ་ཏྭཱ་པཱུ་ཛཱ་པྲ་ཏི་གྲ་ཧཾ། །ཏཾ་པདྡྷི་གནས་པ་བྱས། །དེ་ཡི་ཁྱིམ་དུ་བཅོམ་ལྡན་གྱིས། །མཆོད་པ་རབ་ཏུ་བླངས་མཛད་ནས། །དེ་ནི་ཆུང་ས་ཧི་ཏཾ་ཙཀྲེ་ཤུདྷ་ཤིཀྵ་པ་དོ་ཛཱ་ལཾ། །ཏ་ཏོ་ཡ་ཤོ་ད་ས་ཅིབཱཉྩ་མར་བཅས་པ་ལ། །་དག་པའི་བསླབ་ཚིགནས་རབ་འབར་མཛད། །དེ་ནས་གྲགས་སྦྱིན་གྲོགས་པོ་ནི། །ཏཱ་ར་ཤྲཱི་མནྟཾ་བཱ་རཱ། །བི་མ་ལཱ་ལྱསུ་བཱ་ཧུཤྱ་པཱུརྞ་ཀོ་ཐ་ག་བཱ་པ་ཏི། །དཔལ་དང་ལྡན་པ་རྣམས་ཀྱི་མཆོག། དྲི་མེད་གང་པོ་ལག་བཟང་དང་། །བ་གླང་བདག་ཅེས་བྱ་བ་བཞིས། །ཡ་ཤཔྲ་ཀཱ་ཤི་ཏ་དི་ཤཾ་ཡ་ཤོ་དཾ་བྷ་ག་བདྲུ་ར། །བྲཧྥ་ཙརླ་བྲ་ཏཱ་བཅོམ་ལྡན་མདུན་ན་གྲགས་སྦྱིན་ནི། །ཚངས་སྤྱོད་བརྟུལ་ཞུགས་ལ་ཆགས་པའི། །གྲགས་པ་ཕྱོགས་སུ་སཏྟཾ་ཐུ་་ཏཾ་དེ་ཤ་གཱ་ཡ་ཡུ། །ཏེ་་ཀུ་ཤ་ལཱ་པཱ་ཀེ་ན་པཱཔ་རབ་གསལ་བ། །ཐོས་ནས་གནས་དེར་འོངས་པར་གྱུར། །དགེ་བ་སྨིན་པས་ཐོབ་གྱུར་པ། །དེ་དག་ནཾ་བྷ་ག་བཱནྡྲ་ན། །བི་ད་དྷེ་དྷརྨ་བི་ན་ཡེ་ཤཱ་ས་ནཾ་ཤུདྡྷ་ཤཱ་ས་ན། །རྣམས་ལ་བཅོམ་ལྡན་འདས། །དག་པའི་བཀའ་ཡིས་སླར་ཡང་ནི། །འདུལ་བའི་ཆོས་ལ་བསྟན་པ་བསྒྲུབས། །ཡ་ཤོ་དསྟེ་ཙ་ཙ་ཏཱ་རོ་བྷིཀྵ་བ༔པཉྩ་ཙཱ་པ་རེ། །ཨརྙཏྠ་དཾ་ད་ཤ་པཱ་པུསྟ་བཞི་པོ་དེ་དག་གྲགས་སྦྱིན་དང༌། །དགེ་སློང་གཞན་པ་ལྔ་དག་སྟེ། །བཅུ་ཡིས་བཅོམ་ལྡན་དྲུང་དུ་ནི། །ད་བྷ་ག་བ་ཏོནྟི་ཀེ། །ཐུ་ཏྭཱ་ཡ་ཤོ་ད་ཐྲཏྟཾ་ཏཾ་དྷ་སྟཏྠ་ཙི་བཱ་དགྲ་བཅོམ་གནས་དག་ཐོབ་པར་གྱུར། །གྲགས་སྦྱིན་སྤྱོད་ཚུལ་ཐོས་གྱུར་ནས། །དེ་ཡི་གྲོགས་པོ་སྐལ་།
