བ་ད་ནསྟལྟ་བི་ཤཱ་ས་ཤུ་ཏ་སཾ་ཀ－ཝཱི་ཀྲྀ་ཏ་ན་ཏ་ཏ་ནུརཱ་ཏི་ཡཱསྨཱསྣ་སེམས་ལ་འཕྲལ་ལ་རབ་ཏུ་གསལ་བ་གང་། །དགེ་ལྡན་ཡོངས་སུ་སྐྱོབ་གཉེན་བསྔགས་འོས་གཅིག་པུ་དེ་ཉིད་ཙཱ༑། དྷིསྔརྨ་ཀཱ་ར་སིདྡྷོཔ་ཐི་ཀ་ཛ་ན་པྲ་ཐུཙྪཱ་ས་ནི་རྡགྷ་བྲྀདྷེ་ལྟ་པ་འཚོ་བ་ཡིན། །རང་ཉིད་རབ་བསྲུངས་གདུང་བའི་ཉམས་མྱོང་ཤེས་ཤིང་རི་བྱིའི་ལྟོ་བའི་ཕུག་ཏུ་བླུགས་བྱས་ཡོཉཟི་བྲ་ཛཱ་ནཱ་ཧ་ར་ཏི་ར་བི་ཥ་མཾ་ཏཱི་བྲ་ཀྲྀཥྛ་ས་ཏེ༑། སྨིན་བྱེད་མེ་ཡིས་རབ་ཏུ་གདུང་བའི་བྱེད་པས་ཁྱབ་ཅིང་འཚེད་པའི་བྱ་བ་ཡོངས་སུ་འདྲིས་གྱུར་མུཏྟཱ༑། ཛཱ་ཏཱ་སཱ་དོ་པྱ་གཙོ་ད་ར་ཀུ་ཧ་ར་ལུ་ཋཛྫ་ཋ་རགྷི་པྲ་ཏཱ་པ་བྱཱ་པཱ་ར་ཀྱང་། །སློང་བའི་ཚོགས་ཀྱི་སྲེད་པ་དྲག་པོས་བསྐྱེད་པའི་མི་བཟད་གདུང་བ་འཕྲོག་པ་མིན་པ་གང༌། །བྱུ་སྱ་མ་ན་ཀཱ་ཐ་ན་པ་རི་ཙི་ཏ་སྭ་པྲ་མཱ་ཐ་བྱ་ཐཱ་ནཱཾ། །ཨི་སྐྱེ་བོ་འགོན་པོའི་ཤུགས་རིངས་ཆེ་བ་བསྐོལ་ཞིང་འཕེལ་བར་འགྱུར་བྱེད་རྒྱ་མཚོའི་སྐྱེ་བ་སྨད། །དེ་ཏི་སཾ་ཙིནྟ་ནཱི་པ་ཏིཉཀྲྀ་ཏཱ་ཏི་ཕ་ལཾ་བུ་ནཱ། །ཏ་མུ་བཱ་ཙཱ་བྲི་ཡཱ་ཁྱ་ན་བྷི་ཏ་ལྟར་སྐད་ཅིག་བསམས་བྱས་ནས། །ཆུ་དང་འབྲས་བུ་མགྲོན་བྱས་ཏེ། །མི་སྙན་སྨྲ་བས་འཇིགས་འཇིགས་དྷཱི་ཏ་ཨི་བཱཀྵ་ཎ། །བྲཧྥསྣ་དཱི་ན་པུ་ཙྩོཧཾ་རཱ་ཛ་ཤ་ཏུ་བ་དྷོ་བཞིན༑། མི་ཡི་བདག་པོས་དེ་ལ་སྨྲས། །བསོད་ནམས་མི་དམན་རྒྱལ་པོ་དང་། །བྲམ་ཟེ་དགྲ་གསོད་བྱ་མེ། །ཧི་སཱ་བི་རཀྟསཾ་རྱ་ཛ་རཱ་ཛྲ་བི་ཛ་ན་མཱ་ཤི་ཏ། །ཧི་མཱ་སྦྱོར་བའི་ཚེ། །འཚེ་ལས་ཆགས་བྲལ་རྒྱལ་སྲིད་ནི། །ཡོངས་སུ་བཏང་ནས་དབེན་པ་བསྟེན། །ས་སྐྱོང་ཕ་ལ་མ་ཧཱི་པཱ་ལ་བྷོ་གཾ་བྷུ་བྷཾ་ག་གྷཾ་གུ་རཾ། །པྲ་ཏྱ་གྲ་རུ་དྷི་རཱ་དིབྷཾ་ཀྲ་བྱཱ་ལོངས་སྤྱོད་འཚེ་བའི་འབྲས། །སྨིན་འཁྱོག་ལྟ་བུར་མི་བརྟན་པོ། །གསར་པའི་ཁྲག་གིས་སྦགས་པ་ནི། །ད་ཨི་བ་བྷུཉྫ་ཏེ། །ཀིཾ་ཀ་རུམུ་པ་ད་བློ་ཧ་མ་ཀཱ་ལེ་ཏཱ་མུ་སཱ་ག་སྲིན་པོ་བཞིན་དུ་ལོངས་སྤྱོད་བྱེད། །གནས་མིན་འདུག་དེས་ཅི་ཞིག་བྱེ། །ཁྱོད་ནི་དུས་མིན་ཉེ་བར་ཏ༑། ཡནྟུ་ཤ་ཀྱི་མ་ཡཱ་ཀི་ཙིནྟ་ད་ཡཾ་ཏྲི་ཏ་མུ་ཙུ་ཏྭཾ། །ཨི་ཏི་རཱ་ཛ་བ་ཙ་འོངས། །ཅུང་ཟད་བདག་གིས་ནུས་པ་གང་། །དེ་ནི་བསྡམས་པ་མེད་པར་བརྗོད། །ཅེས་པ་རྒྱལ་པོའི་—།
