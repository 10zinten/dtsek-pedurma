མེ་བ་བྲུ་ཧི་བྷ་ག་བན་ཡ་དི་ཡུཀྟ་མི་དཾ་ཀྲྀ་ཏཾ། ་། ཨི་ཏི་ཏ་སྣ་བ་ཙཐུ་ཏཱ་ཏེ་རིགས་པས་འདི་བྱས་ན། །བཅོམ་ལྡན་ཁྱོད་ཉིད་བདག་གིས་གསུངས། །ཞེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །ཀིཉྩིལླཛྫཱ་ན་ཏཱ་ན་ན། །།ཏཱ་མུ་བཱ་ཙ་མུ་ནིརྨུབྷེ་ཏ་བི་ཥཱ་བྷ་བི་ཏ་བྱ་ཏཱ། །ཅུང་ཟད་ངོ་ཚས་བཞིན་བཏུད་དེ། །ཐུབ་པས་དེར་སྨྲས་བཞིན་བཟང་མ། །འདི་ནི་ཁྱོད་ཀྱིས་མྱོང་བྱ་ཉིད། །ཨ་མོ་གྷ་པཱ་ཤས་སཱུཙཱི་ཏྱ་ཙཱ་ཏྭཱ་ཀ་ཐི་ཏམྨ་ཡ། །ཨུ་པ་ལ་བྱཱ་ཀ་ཐཾ་དྷཱཏྟཙཱ་དེ་ལ་ཞགས་པ་དོན་ཡོད་ནི། །ཡོད་ཅེས་བདག་གིས་མ་ཤེས་བཤད། །གཡོན་ཅན་རྔོན་པས་དེ་ཡི་གཏམ། །པ་བནྡྷ་པ་ལུབྷཱ་ཀཿ། །དུབྷྱཱཏྨ་ཚྭ་ན་ཛཱ་ནཱི་མཀོ་ཊི་ལཾ་ཀཱ་ར་ཙེ་ཏ་ཤེས་ནས་ཁྱོད་ནི་བཅིངས་པར་གྱུར། །བདེན་པར་སྨྲ་བའི་མུ་ཅོར་ཅན། །རང་བཞིན་དྲང་བ་བདག་ཅག་སཱ༑། པ་ཏྱ་པྲ་བཱ་ད་མུ་ཁ་རཱསཏ་བྱ་བ་པ་ར་ལཱ་བ་ཡཾ། །ཨི་ཏུཏྟཱ་མུ་མི་ནཱ་གིས༑། མ་རུངས་བདག་ཉིད་གདུག་པ་ཡི། །སེམས་ཀྱི་གྱ་གྱུ་ཡོངས་མི་ཤེས། །ཞེས་པ་ཐུབ་པས་བརྗོད་ཏ་ནཱི་པྲ་ཎེ་ཡཱཏྟ་མ་བྷཱ་ཥ་ཏ། །བྷ་ག་བནྡྷ་མྱ་ཏཱ་མེ་ཏདྷཱ་ལཱ་བ་ཙ་ན་དེ་ལ། །གཟུགས་ཅན་མ་ཡིས་གུས་པས་སྨྲས། །བུ་མོའི་ཚིག་ནི་གཡོ་བ་འདི། །བཅོམ་ལྡན་གྱིས་ནི་ཙཱ་པ་ལཾ། །ཨི་དནྟཾ་བྷ་བ་ཏ་མ་གྲེ་ཡནྡྲ་ཡཱ་ཀིཾ་ཙི་དུ་ཙ་ཏེ། །ལ་ལཱ་བཟོད་པར་མཛོད། །ཁྱེད་ཀྱི་མདུན་དུ་བདག་གིས་ནི། །གང་འདི་ཅུང་ཟད་བརྗོད་གྱུར་པ། །དམ་པའི་སུ་བྷ་གསོ་ཡཾ་པཱ་དཱ་ཙཱ་ར་བྱ་ཏི་ཀྲ་མ། །དུཁོདྷཱ་ར་ཎ་སནྣདྷཱསྤྱོད་ལས་འགལ་བ་འདི། །བུད་མེད་རྣམས་ཀྱི་སྐལ་བ་བཟང་། །སྡུག་བསྔལ་འདོན་ལ་ཆས་གྱུར་ཅིང་། །སནྟབྷཱ་ནཾ་ད་ཡ་ལ་བ། །ཀཱརླནྟ་ར་གཱཔཱ་ཡེ་ཎ་བྷ་བི་དྱ་ནུ་ཙི་ཏེ་ཥཱ་པི། །ཡོངས་སུ་གདུང་ལ་བརྩེ་ལྡན་རྣམས། །འོས་པ་མིན་པ་རྣམས་ཀྱི་ཡང་། །ཕལ་ཆེར་བྱ་བ་སྒྲུབ་མཛད་ཡིན། །པཱ་ཤ་བནྣཱདྲི་མུཀཱ་ཧཾ་ལུབྦྷ་ཀེ་ན་པྲ་ལཱ་སི་ནཾ། །ནི་བདྷཱ་རཱ་ཛ་ཆོ་ངེ་དང་ནི་ལྡན་པ་བདག། རྔོན་པའི་ཞགས་པས་བཅིངས་ལས་གྲོལ། །འཕྲོག་བྱེད་མཛའ་བའི་པུ་ཏྲེ་ཎ་སྣེ་ཧ་པཱ་ཤེ་ན་ཧ་རི་ཎཱ། །སུ་བྷ་གསུ་དྷ་ནཱ་ཁྱོ་སོ་མདྲི་ཞགས་པ་ཡིས། །རྒྱལ་པོའི་བུ་ཡིས་ངེས་པར་བཅིངས། །ནོར་བཟང་ཞེས་པ་སྐལ་བ་བཟང་། །བདག་།
