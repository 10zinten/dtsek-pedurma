ས་ར་སྱ་ན་བ་ཏལྟ་ཁྱེ་རཉྫ་སོ་སཱ་ན་སཉྩ་ཡེ། །ཧེ་མཱཛྫ་བུཉྫ་ཀཱིཉྫལྟ་རྫིང་བུ་གདུང་མེད་ཅེས་བྱ་བ། །རིན་ཅེན་ཚོགས་ཀྱི་ཐེམ་སྐས་ཅན། །གསེར་གྱི་པད་ཚོགས་གེ་སར་གྱིས། །པིལྔ་རཱི་ཀྲྀ་ཏ་ཥགླ་དེ། །པ་ས་ནསྦྱོ་བྷ་ག་བཱན་བྷིཀྵ་བྷིཔ་རི་རྐང་དྲུག་སེར་པོར་བྱེད་པ་ཡི། །པདྨའི་ཁྲིར་གནས་བཅོམ་ལྡན་འདས། །དགེ་སློང་རྣམས་ཀྱིས་བ་རི་ཏ། །སཧྥ་ཛྷཀརྨ་ཏརྩྭ་སྱ་དན་ཡནྟ་བི་ལཾ་རྒྱུ་ཏྭཾ། །ཡོངས་བསྐོར་བ། །ཐམས་ཅད་མཁྱེན་པ་ལས་ཀྱི་རྒྱུད། །འགོམ་བྱ་མིན་པར་སྟོན་མཛད་ཅིང་། །སྭ་ཀརྨ་པཱུ་ཏི་བ་ཙི་ཏྱཾ་བཀྟི་ས་མུ་པ་ཙ་ཀྲ་མེ། །ཀ~་བྱཱ་ན་ཀྵ་ཎེ་རང་ལས་སྤྱོད་པ་རྣམ་པར་བཀྲ། །གསུང་བར་ཡང་དག་བརྩོན་པར་གྱུར། །ལས་ནི་འཆད་པའི་གནས་ཏསྨིན་བྷ་ག་བཱནྟཱཀྟ་བཏྠ་ལ། །ཤཱ་རི་པུ་ཏྲ་ས་མཱ་ཧཱ་ནེ་མོངྒ་ལྟ་སྐབས་དེར། །བཅོམ་ལྡན་གུས་ལ་མཉེས་གཤིན་པས། །ཤཱ་རིའི་བུ་ནི་འབོད་པ་དག། ་མོ་གལ་ཡ་ན་མཱ་དི་ཤཏྲ། །ཤུ་ཙཱ་སཱུ་ཏྲེ་ཎ་སཾ་གྷ་ཊཱི་ཀུནྡཱ་ཎཾ་ར་ཙ་ནཱ་ཙི་ཏྭཾ། །ས་ཤཱ་རི་བུ་ལ་རབ་ཏུ་བསྐོས། །བྱ་རྒོད་ཕུང་པོའི་རི་གནས་ལ། །ཁབ་དང་སྐུད་པས་སྦྱར་མ་ནི། །བཀྲ་ཤིས་པུ་ཏྲེ་མཱ་སཱ་ཎྱ་གྲོ་དྷ་ཀཱུ་ཊཱ་ཙ་ལཱ་ཐ་མེ། །ༀ་གུ་རཱི་སཉྩ་ཀོ་ནི་བ་པཱུ་ཙཱི་ཀརྨ་རྫོགས་པར་བྱེད་པ་ཡིན། །ཤཱ་རིའི་བུ་ཅན་དེ་ཡིས་ཕྱིན། །སོར་མོ་རངས་བྱས་འདྲུབ་པ་ཡིས། །པྲ་བྷཱ་བ་བཱན། །བི་ལམྦ་བྷཱི་ཏ་ཀྲྀ་ཏྭཱ་སྱ་ཏ་མ་བྷ་ཥ་ཏ་ས་ཏཱ་ར། །ལས་ཀྱི་མཐུ་ཉིད་མ་ཡིན་པ། །བྱ་བ་འགོར་བས་འཇིགས་པ་ཡིས། །མྱུར་དང་བཅས་པ་དེ་ལ་སྨྲས། །ཨེ་བྱེ་ཧི་ཏུནྟཾ་བྷ་ག་བཱན་ས་ར་སྱ་ན་བ་ཏཔ་ཀི། །ཀརྨོ་པ་དེ་ཤཾ་བྷི་ཥཱ་ཎཾ་ཚུར་ཤོག་ཚུར་ཤོག་མགྱོགས་པར་རོ། །བཅོམ་ལྡན་ཀུན་མཁྱེན་རྫིང་བུ་ནི། །གདུང་བ་མེད་པར་དགེ་སློང་སཏྠ་ཛྷཀཀྟ་མུ་ཏྱ་ཏ། །ཀརྨ་བྱ་གྲ་ཡཱ་་པྲ་བི་ལལྤ་ཝི་ད་དྷཱ་ལ༑། ལས་ནི་ཉེ་བར་སྟོན་པར་བརྩོན། །ལས་ལ་ཆགས་ནས་མྱུར་མིན་ཞིང་། །ཐོགས་པར་གལ་ཏེ་ཁྱོད་སི་ཙེ་ཏ། །ཏརྨ་ཧརྒྱུ་ན་ཡཱ་མ་ཏྭཾ་བི་པུ་ལ་པ་བྱ་མེ་བ་ལ་མ། །ཨི་ཏི་ཏ་བསྒྲུབས་ན། །རྫུ་འཕྲུལ་ཆེན་པོས་ཁྱོད་བླང་བྱ། །བདག་གི་སྟོབས་ཡང་ཅིབལྟ་བར་བྱོས། །ཞེས་པ་དེ་—།
