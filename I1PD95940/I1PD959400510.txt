རྱ་ཨེ་བ་པྲ་རཱ་དྷི་ན། །ག་ཏཾ་ཤོ་ཙ་ཏི་ཀཽ་རཱ་ཛཱ་ཡཏྠཾ་ཏཾ་ཏ་མེ་བ་ཏཏ། །གནོད་པ་ཉིད་དང་ལྡན་པར་འགྱུར། །རྒྱལ་པོ་འདས་ལ་ཅི་ཞིག་འགྱོད། །གང་བྱས་དེ་དག་བྱས་པ་ཉིད། །ཨུ་པེ་ཀ་ཏེ་ཏཱ་ཡཱ་དེ་བི་ཀསྨཙིནྟཱ་ཀྲྀ་ཤ་སུ་ཏ། །ཏཱ་བྱ་མི་ཏུ་དི་ཏཾ་ཐུ་ཏྟཱ་ལྷ་མོ་བསམ་པས་རིང་བའི་བུ། །ཁྱོད་ཀྱིས་ཅི་སླད་བཏང་སྙོམས་བྱེད། །དེ་དག་གིས་ནི་བརྗོད་ཐོས་ནས། །ས་རཱ་ཛ་ཛ་ན་ནཱི་ཤ་ནི། །ཨཱ་ཙེ་ཏ་ར་ལི་ཀཱ་ནཱ་མ་ཏདྷཱ་ཀྱ་བི་ཧི་ཏཱ་ད་རཱ། །ཨཱ་དེ་དག་ཚིག་ལ་གུས་པས་བསྒྲུབས། །ཏ་ར་ལི་ཀ་ཞེས་པ་ཡི། །རྒྱལ་པོའི་མ་དེས་དལ་བུས་སྨྲས། །ཁྱོད་ནནྟརྱ་མི་དཾ་ཀརྨ་དྭ་ཡཾ་ན་ར་ཀ་པཱ་ཏ་ཀཾ། །ཡུམྨེརྨ་ཏཱ་དུ་པ་ན་ཏཾ་རཱ་ཀྱི་གྲོས་ལ་ཉེར་བསྐྲུན་པའི། །རྒྱལ་པོས་སྔོན་ནི་བྱས་པ་ཡིས། །དམྱལ་བར་ལྟུང་བར་བྱེད་པའི་ལས། །ཛྙཱ་པཱུ་ཧྥ་ཀྲྀ་ཏེ་ན་བཱ། །ཨ་ཧཾ་ཏུ་བཱ་ར་ཡཱ་མྱ་སྱ་ཤོ་ཀཾ་པི་ཏྲ་བ་འདི་ནི་མཚམས་མེད་གཉིས་སུ་འགྱུར། །འདི་ཡི་ཕ་བསད་ལས་སྐྱེས་པའི། །མྱ་ངན་བདག་གིས་བཟློག་དྷོདྷ་པཾ། །ཨཉདྲ་བྷོདྷུ་ཐཾ་དུཁཾ་བྷ་བནྡྷ་མ་པི་བཱརྱ་ཏྭཾ། །ཨི་ཏི་པར་བགྱི། །དགྲ་བཅོམ་བསད་སྐྱེས་སྡུག་བསྔལ་ནི། །ཁྱེད་གཉིས་ཀྱིས་ཀྱང་ཟློག་པར་མཛོད། །ཅེས་པ་ཏོ་སཱི་ར་མཱ་དི་བྱ་སཱ་ག་ཏཱ་པཟི་བཱནྟི་ཀཾ། །ཏ་མུ་བཱ་ཙ་ཤུ་ཙཱཀཱནྟཾ་པ་རི་ཀྵི་ཎ་མི་དལ་གྱིས་དེ་ལ་བསྟན། །ས་བདག་དྲུང་དུ་དེ་སོང་ནས། །མྱ་ངན་གྱིས་ནོན་ཟླ་བ་ནི། །ཡོངས་ཟད་ལྟ་བུ་བོ་ཌཱ་པཾ། །དྷརྨ་དྷརྨ་མ་ཡཾ་དུ་ཏྲ་རཱ་ཛྭ་རཱ་ཛྙཱ་བ་ཧུཙྪ་ལཾ། །པཱ་པཱ་ཤཾ་དེ་ལ་སྨྲས། །ཆོས་དང་ཆོས་མིན་རང་བཞིན་གྱིས། །རྒྱལ་པོའི་རྒྱལ་སྲིད་ཉེས་པ་མང་། །དེ་ལ་སྡིག་ཀ་ཡཱ་ཏསྨིན་ཀིཾ་ཤུ་ཙཱ་པ་རི་ཤུ་ཥཱ་སི། །པི་ཏུརྦ་དྷཱ་ཏ་པྲ་ཏཔྟ་སི་ཡ་དི་ནཱ་མ་དོགས་མྱ་ངན་གྱིས། །བུ་ཁྱོད་ཅི་ལ་ཡོངས་སུ་གདུང་། །གལ་ཏེ་ངེས་པར་ཕ་བསད་ལས། །བླ་མ་ལ་གུ་རུ་པྲ་ཡ། །ཏ་ཏཱ་རྱ་ཏེ་ས་མུཎྜི་ཚུ་ལཛྫ་ཏཱདྷུཁ་སཾ་ཀ་ཊེ། །སཱི་རཾ་བྱམས་ཁྱོད་གདུང་ན། །ཁྱོད་ཀྱི་སྡུག་བསྔལ་མི་བཟད་དེར། །དེ་རིང་ཁྱོད་ལ་ངོ་ཚ་གཏོང་། །ཁྱོད་ཀྱི་ཛཱ་ཏསྭཱ་མ་རྩེ་ན་ན་ས་ཏ་དྷརྨ་ཏཔི་ཏཱ། །སཱིཙྪཱ་ཧཱ་ར་སུ་ཁ་པུ་ཏྲ་ཆོས་ཉིད་ཕ་དེ་མིན། །ཁྱོད་ནི་གསང་བ་གཞན་ལས་སྐྱེས། །རང་འདོད་ཟས་དང་བདེ་བས་བུ། །བུད་—།
