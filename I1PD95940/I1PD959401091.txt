དྷི་པྲ་དརླུ་པྲ་ཡ་ཡོ་རི་བ་ཏས་ཧ་བྷིཀྵ་བྷི། །ཨ་ས་ཏུ་དོ་ཥ་རྫུ་འཕྲུལ་བསྟན་ནས་རི་བ་ཏ། །དགེ་སློང་རྣམས་དང་ལྷན་ཅིག་སོང་། །བདེན་མེད་སྐྱོན་གྱིས་བྱེ་ཎ་བི་ཥ་ཡུཀྱུ་པྲ་ཏུཀྵ་ག་ཏ་ལཀ་ཎེ་ན། །ཨ་ལཾ་ཀི་ཏསཱ་དྷུ་ཛ་བྲག་རིག་པ་ནི། །མངོན་སུམ་དུ་མཚོན་མི་ཉམས་མཚན་ཉིད་ཀྱིས། །བརླང་པོས་ཟིལ་མནན་སྐྱེ་བོ་དམ་ན༔ཁ་ལེ་ན་ཀིཾ་བཀྟི་བི་ཀྲྀ་ཏ་ཨི་ཐཾ་ཏྱ་དེ་ཤ། །ཨི་ཏི་ཀེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་པ་ནི། །ཡུལ་གཞན་དག་ཏུ་བཙོངས་བཞིན་ཅི་ཞིག་བརྗོད། །ཅེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཡཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་རི་བ་ཏཱ་བ་དཱ་ནཾ་པཉྙོཏྟ་ར་ཤ་ཏ་ཏ་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་རི་བ་ཏའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་མཔལླ་བ། །།།།འདབ་སྟེ་བརྒྱ་རྩ་ལྔ་པའོ། །།།A。。བི་ཛི་གཱི་ཥ་བ་ལ་ཨི་བ་ཏ་ར་སཱ་དུས་ཧསཾ་སཱ་ར་ཤ་ཏུཾ་བ་ལཾ་ཧ་རྒྱལ་བར་འདོད་རྣམས་བཞིན་དུ་སྟོབས་ཀྱིས་ནི། །བཟོད་དཀའ་འཁོར་བའི་དགྲ་ཡི་སྟོབས་བཅོམ་ཏྭཱ༑། རི་པུ་ཨཱ་སཱ་ད་ཡནྟི་དྷི་རཱཔྲ་ཛཱི་སདྡྷརྨ་སཱསྣ་ཚོ། །ནས། །དམ་པའི་ཆོས་ཀྱི་རྒྱལ་སྲིད་ཆེན་པོ་ནི། །རྒྱས་པ་བརྟན་པ་རྣམས་ཀྱིས་ཐོབ་པར་འགྱུར། །ཡ་དཱ་སཱུ་ཏྭཾ་བྷ་ག་བ་ཏཱ་བྷཱ་ཥཱ་ཏཾ་ན་ག་རོ་པ་མཾ། །ཏ་དཱ་ས་ཀོ་ཏུ་ཀཱཎྚིཊོ་གང་ཚེ་གྲོང་ཁྱེར་ལྟ་བུའི་མདོ། །བཅོམ་ལྡན་གྱིས་གསུངས་དེ་ཡི་ཚེ། །དགེ་སློང་རྣམས་ཀྱིས་དགེ་མཚན་བྷིཀྵ་བྷིལྟ་ན་བྷ་ཥ་ཏ། །ཨ་བྷུ་ཏ་ཀ་ན་ཀ་བ་ཏྭཱ་ཀྵ་ཛ་ན་སཱརྒ་པྲ་བྷཱ་ལས། །དྲིས་པ་དེ་ཡིས་རབ་གསུངས་པ། །གྲོང་ཁྱེར་གསེར་ཅན་ཞེས་པ་ནི། །སྐྱེ་བོ་མཐོ་རིས་འོད་པུ་རཱི། །ཏ་སྭཱ་མཱ་སཱིནྡ་ར་བ་ཏིརཱུ་ཏི་མཱཎ་ན་ཀཱབྷི་དྷ། །ཏ་སྱ་པྲ་ལྡན་བྱུང༌། །དེར་ནི་མི་བདག་འོད་དང་ལྡན། །གསེར་ཞེས་བྱ་བ་བྱུང་བར་གྱུར། །དེ་ཡི་དགའ་བ་བནྡཱ་མཱ་ནཱ་ཡཔཱི་ཏེ་རཱ་ཡ་ཏ་ནཾ་པ་རཾ། །པུ་ཏྲཀ་ན་ཀ་བམྨཱ་བྷཱུཏུ་ཏཱ་བ་ཀ་ན་ཀ་པྲ་མཆོག་ཏུ་ནི། །རབ་ཏུ་རྒྱས་པ་དག་གི་གནས། །བུ་ནི་གསེར་གྱི་གོ་ཆ་དང་། །བུ་མོ་གསེར་འོད་དག་ཏུ་—།
