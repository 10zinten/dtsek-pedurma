ལ་ཨ་ཧོ་པི་ཧི་ཏ་མངྒེ་ཎ་གཙཱ་ཏཱ་མ་སྱ་ཙེ་ཏ་སཾ། །ཛ་ལཱ་ནཾ་ཀརྨ་སཾ་སིདྷིཏྲི་ཀྱེ་མ་སེམས་མེད་ཆུ་རྣམས་ཀྱང་། །བསྒྲུབས་པའི་ལམ་ནས་བགྲོད་པ་ཡིས། །ལས་ནི་ཡང་དག་སྒྲུབ་མཐོང་བྱེ་ཏེ་ན་ཏུ་དེ་ཧི་ནཾ། །སཾ་བ་ཏེ་ཧི་པྲ་ཛསྣ་གྲེ་ཛྲཏྟཱ་སྟེ༑། ལུས་ཅན་རྣམས་ལ་མ་ཡིན་ནོ། །ཞེས་བསམས་བགྲོད་པས་མདུན་དུ་ནི། །མདའ་མཁན་དག་གིས་སྱཊཱི་ཤྲཱི་རཾ། །ས་ཏ་པ་མི་ཥཱ་ཀ་རེ་ཎ་པྲ་ད་གྷོ་བྷི་མ་ཏཱཝ་ར། །མདའ་བསྲོས་ཤིང་། །།དྲང་པོར་བྱེད་པ་མཐོང་གྱུར་ནས། །བློ་ལྡན་མཆོག་དེས་རབ་ཏུ་བསམས། །ཏཱ་སཏྲ་གུ་ཎ་ཏཱ་མེ་ཏེ་ཡཱནྟི་ནིལྕེ་ཏ་ནཱཤ་ར། །ན་ཏུ་སཾ་སཱ་མདའ་ནི་སེམས་མེད་འདི་དག་རྣམས། །་གདུང་བས་དྲང་པོ་ཉིད་འགྱུར་ཏེ། ཡང་། ལུས་ཅན་འཁྱོག་པོ་ར་སཾ་ཏལྟ་མ་པི་བཤ་རི་རི་ཎ། །ཨི་ཏི་གཱ་ཡ་ན་བི་ལོ་ཀྱ་པྲ་ཏ་རྣམས་དག་ནི། །འཁོར་བའི་གདུང་བས་ཀྱང་མ་ཡིན། །ཞེས་སེམས་མདུན་དུ་ཤིང་མཁན་གྱིས། །་། ཤིང་ཛྙཱ་ཤ་ཀ་ཊ་ཙཀྲ་ནཾ། །ནཱི་ཏཱ་ནི་ཚིགས་ཌྷ་རཱུ་པཱ་ཎི་པུ་ནིཉྩིཏྟཾ་ས་མཱ་རྟའི་འཁོར་ལོ་ཉིད་བྱས་པའི། །བརྟན་པའི་གཟུགས་རྣམས་མཐོང་གྱུར་ནས། །སླར་ཡང་ཡང་དག་སེམས་ཡ་ཡོ། །ཨ་ཧོ་ནུ་བྱ་ཌྷ་ནཱ་ཡོ་གཱཏ་ཡཱནྟི་ཀམྨཱ་རྱ་ཏྭཾ་ག་ཎཱཏ། །ནིཉྩེ་ཏ་ནཱ་ནི་དཱ་པར་གྱུར། །ཀྱེ་མ་སེམས་མེད་ཤིང་རྣམས་ནི། །བཟོ་ཡི་སྦྱོར་བས་སྐད་ཅིག་ལས། །ལས་སུ་རུང་བ་ཉིད་རཱུ་ཎི་ན་ཙིཏྟཱ་ནི་ཤ་རཱི་རི་ཎཾ། །ཨི་ཏི་སཾ་ཙི་ཏྱ་སཾ་ཡཱ་ཏསུ་དྷརྨ་འགྱུར་ཏེ། །ལུས་ཅན་རྣམས་ཀྱི་སེམས་མ་ཡིན། །འདི་དག་བསམས་ཤིང་ཡང་དག་སོང་། །ཆོས་ནི་ཡ་མཱ་ད་ར། །བཏྠ་ལཾ་པི་ཏ་ར་པུ་ཏྲ་ཨི་བཱ་ཙཱརྨ་མུ་བཱ་ཙ་ས། །བཟང་ངེས་པ་ལ་གུས་པ། །དེ་ཡིས་སློབ་དཔོན་ལ་སྨྲས་པ། །མཉེས་གཤིན་ཕ་ལ་བུ་ཡིས་བཞིན། །ཨཱརླ་ཨེ་བ་པྲ་ཡཱ་ཏཱ་བྱ་པཎྜ་པཱི་ཏཱ་ཡེ་མཀྵ་ཏེ། །ཨ་ཧཾ་རུ་བྷ་བ་འཕགས་པ་ཉིད་ནི་བསོད་སྙོམས་ལ། །དེ་རིང་བདག་གི་ཆེད་དུ་གཤེགས། །ཁྱོད་ཀྱི་བསྟན་པ་རང་ཏཱ་དིཝཾ་ཙནྟཾ་ཡཱམི་ནི་ཛ་བྲ་ཏཾ། །ཨི་ཏུ་པཱ་དྷ་ཡ་བྱཱརྒྱ་བྷལྟ་ཀྲྀ་ཏྭཱ་ཉིད་ཀྱི། །བརྟུལ་ཞུགས་བདག་གིས་བསམ་པར་བགྱི། །ཞེས་ཏེ་ཟས་ཀྱི་བྱ་བ་ནི། །མཁན་པོ་ལ་གསོལ་།
