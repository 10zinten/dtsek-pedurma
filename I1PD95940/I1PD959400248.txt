ར་ཧ་གླེ་ཤྲཱི་དུ༔ས་ཧཾ་ས་ཧ་སེ་ཀ་ཐཾ། །ཏཏྟཾ་ན་ཤཀྱ་ཏེ་ཀིཤྲཱི་ད་དི་བའི་ཉོན་མོངས་ནི། །བཟོད་པར་དཀའ་བ་ཇི་ལྟར་བཟོད། །རང་གི་ལུས་བཞིན་ཡུལ་དག་ནི། །སུས་ཤོ་དེ་ཧ་ཨི་བ་སཱ་ཀ། །།བྷཱ་རཾ་དྲ་བི་ཎཾ་སཾ་བྷཱ་རཾ་བེཏྟི་གནྡྲི་གུ་ཎ་གུ་ཎཾ། །ཀྱང་གཏང་ཀྱི་བར་མི་ནུས་སོ། །རང་ཡུལ་བྲལ་བའི་སྐྱེ་བོ་རྣམས། །ནོར་གྱི་ཚོགས་ནི་ཁུར་པོ་དང་། །བྷོ་ག་ནི་རུ་པ་ཡོ་གཉྩ་སྭ་དེ་ཤ་བི་ར་ཧཱི་ཛ་ན། །ཨི་ཏུཀྵ་ས་གུ་ཧཱ་གེ་ལོངས་སྤྱོད་ཉེ་བར་ལོངས་སྤྱོད་མིན། །ཡོན་ཏན་ཐག་པའི་མདུད་པར་རིག། ཅེས་སྨྲས་ཕུག་གི་ཁང་པ་ཧཱ་དུཏྟཱ་རྱ་བི་པུ་ལཱ་ཤི་ལཾ། །ཀྲྀ་ཏཱ་བྷཱུ་ཏ་ག་མེ་ནཱ་ཤུ་ཛ་ན་ཀེ་ན་ཡ་ཡོ་ནས། །རྡོ་བ་ཆེན་པོ་དེས་བསལ་ནས། །ཁས་བླངས་པ་ཡི་ཕ་དང་ནི། །ལྷན་ཅིག་མགྱོགས་པར་སོང་ས་ཧཱ། །པྲ་ཡཱ་ཏ་ཡོyཎཾ་ས་མ་བྷེ་ཏྭཱ་ཐ་ཀིནྣ་རཱི། །ཤུ་དྡྷཾ་དྲྀབྷཱ་གུ་ཧ་བར་གྱུར། །དེ་དག་མྱུར་དུ་སོང་དེ་ནས། །མི་འམ་ཅི་མོ་མངོན་པར་འོངས། །ཕུག་ཁྱིམ་སྟོང་པ་མཐོང་གེ་ཧཾ་ནིརྦ་དཱ་དི་ཏྱ་ཙིནྟ་ཡཏ། །ཨ་ཧོནེ་བི་སྨྲ་ཏ་སྣེ་ཧས་ག་ཏཀཱ་པི་གྱུར་ནས། །སྙིང་མི་དགའ་སྟེ་འདི་ལྟར་བསམས། །ཀྱེ་མ་བདག་ལ་གཅུགས་པ་བརྗེད། །སྐྱེ་བོ་ངན་པ་དེ་དུརྫ་ན། །དཱི་ཛི་ཧཱ་ནཾ་བྷུ་ཛཾ་གཱ་ནཾ་ཀོ་ཊི་ལཾ་བཱ་ཀི་མ་ད་བྷུ་ཏཾ། །ན་ར་མནྟེ་པ་གར་སོང་། །ལྕེ་གཉིས་པ་དང་ལག་འགྲོ་ཡིས༅། །གྱ་གྱུ་ཉིད་ལའང་རྨད་བྱུང་ཅི། །བདེ་ལ་ཆགས་ལྡན་ལཱ་ཡནྟེ་སརླནྟེ་སུ་ཁ་རཱ་གི་ཎ། །ཙི་རསྦེ་ཨ་པི་ནི༔སྣེ་ཧཱ༔ཤུ་ཀ་མཇུག་ཏུ་ནི། །དགའ་བར་མི་བྱེད་འགྲོས་པར་བྱེད། །རིང་དུ་གནས་ཀྱང་མཛའ་གཅུགས་མེད། །སྐྱེ་ཨི་བ་དྲི་ཛ་ཏ་ཡ། །ཨི་ཏི་སཾ་ཙིནྟཱ་སཱ་པ་ཏུནི་ཀཱ་རཱ་ཏ་པཱི་ཏི་མ་ཏུ་ཛཏ། །གཉིས་ལྡན་པ་ནེ་ཙོ་བཞིན། །དེ་ལྟར་བསམས་ནས་དེ་ཡིས་ནི། །གནོད་ལས་ཁྱོ་ལ་དགའ་བ་བཏང༌། །པུཏྠོ་ས་མཱ་ནི་སྲེ་མཱ་ཎི་ན་ས་ཧནྟེ་ཀ་དརྦ་ཏཱཾ། །བི་བྱཱ་གུ་ཎེ་ན་ཀི་ནཱ་སོ་མེ་ཏོག་ལྟ་བུ་མཛའ་བ་ཡིས། །གནོད་པ་བཟོད་པ་མ་ཡིན་ནོ། །བདག་གི་བུ་ནི་རིག་བྱེད་དང་། །པུ་ཏྲེ་མི་བྷུ་བི་ཛི་བ་ཏི། །ཨ་ཏི་དྷཱ་ཏཱ་ས་ཁཱི་ཧསྟེ་ཏསྨེ་བཱི་ཎཱ་དི་དེ་ཤ་ས། །ཡོན་ཏན་གང་གི་ྋས་ལ་འཚོ། །ཞེས་བསམས་གྲོགས་མོའི་ལག་ཏུ་ནི། །དེ་ལ་དེ་ཡི་རྒྱུད་མང་སྤྲིངས། །།
