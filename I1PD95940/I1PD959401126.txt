བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁིརྒྱ་རྒྱ་བྷུ་ཛཾ་ག་ཨི་ཏི་ཏ་ཏྲ་བི་ཧཾ་ག་རཱ་ཛསཾ་སཥ་བིསྨ་ཡ་བི་ནི་ཤྩ་ཡ་ནི་ཧྥུ་ལོ་ཧྥུ་འདི་ནི་སུ་ཞེས་དེར་ནི་འདབ་ཆགས་རྒྱལ་པོ་དག། ཡ་མཚན་ལ་ཆགས་རྣམ་པར་ངེས་སླད་གཡོ་བ་མེད་པར་ཏ༑། ཨ་ཐ་པུ་ན་ར་བྷི་པ་ཁྱོ༔ཎྜ་མཱཎྷཎྜ་མཱ་ལཱ་ཀ་པི་ལ་བི་པུ་ལ་གྱུར། །དེ་ནས་འདབ་ཆགས་བདག་པོ་རབ་དྲག་ཉི་མའི་ཕྲེང་བ་བཞིན། །གཟི་བྱིན་སེར་སྐྱ་རྒྱ་ཆེན་ཏེ་ཛཔུཎྜ་སི་གི་ཀྲྀ་ཏཱ་མ། །ཁ་ག་པ་ཏི་ར་རུ་ཚྭ་ཤུ་བྱཀྟ་རཀྟ་པྲ་བཱ་ཧ་པྲ་ཀ་ཚོགས་ཀྱིས་ཕྱོགས་ནི་སེར་བྱེད་པས། །སླར་ཡང་མངོན་བཏབ་མྱུར་དུ་དེ་ཡི་མགོ་ཡི་དུམ་བུ་དག། ནོར་བུ་ཊ་མ་ཎི་མ་ཧཱནཱིཀྵཎྜ་མ་སཱུ་ཤུ་མཱུཙྪཱ། །ཨསྨིནྣཾ་བ་ས་རེ་བཱ་ལ་ཧ་འོད་དམར་རབ་འབར་ཁྲག་རྒྱུན་ལྟ་བུར་རབ་གསལ་བླངས། །གནས་སྐབས་དེར་ནི་གཞོན་ནུ་ཡིས། །ཧ་རི་ཙནྡན་ཀཱ་ན་ནེ། །ཛཱི་མཱུ་ཏ་ཀེ་ཏུ་རཱ་སཱི་ནཔཏྟཱ་སཱནྡྷ་སྐུ་ཥཱ་ནཱི་ཏ། །པུ་རི་ཙནྡན་ནགས་ཚལ་དུ། །སྤྲིན་གྱི་ཏོག་ནི་བཙུན་མོ་དང་། །ལྷན་ཅིག་མནའ་མ་དང་ལྡན་གནས། །ཆུ་ཏྲ་དན་ན་སོཥ་ཎྜཉྩ་ནྡོདྷུ་ཀ་ཨི་བ་ད་དྷི། །ཤཾ་ཀཱ་བི་ཥ་ཥྛ་དྲྀ་ད་གཏེར་ཟླ་བ་འདོད་པ་བཞིན། །བུ་ནི་བལྟ་བར་འདོད་གྱུར་ཅིང་། །སྙིང་ལ་དོགས་པ་གནས་པ་ཡིས། །ཡ་ཤྲིནྟཱ་ཀཱནྟ་ཨི་བཱ་བྲ་བཱིཏ། །ཨ་ཧོ་བ་ཏ་ཀི་མ་རཱུ་པི་བཏྠོ་ཛཱི་མཱུ་ཏ་བཱ་ཧ་ན། །ན་བསམ་པས་ནོན་པ་བཞིན་སྨྲས་པ། །ཀྱེ་མ་ཅི་ཞིག་ད་དུང་ཡང་། །བུ་ནི་སྤྲིན་གྱི་བཞོན་པ་དག། རི་དྲྀ་ཤྱ་གི་རི་བ་རཱ་ཙ་ར་སཱནྟཱ་ལོ་ཀ་ན་ཀོ་ཏུ་ཀཱི། །དཱིཔཱ་ག་རུ་ཌ་བེ་ལེ་ཡཾ་མཆོག་མཐའ་ལ་བལྟ་བ་ཡི། །དགེ་མཚན་ཅན་དེ་མ་མཐོང་ངོ༌། །རི་འདིའི་ངོས་ལ་མཁའ་ལྡིང་གིས། །པ་བྲྀཏྟཱ་སྱ་གི་རེ་སྟ་ཊེ། །ཡ་ཡཱ་བི་བྷཱ་བྱ་ཏེ་བྱོ་མ་དི་གྲྭ་ཧོངྒཱ་ར་དཱ་དུས་འདི་འབར་བ་རབ་ཏུ་ཞུགས། །གང་གིས་ནམ་མཁའ་མི་བཟད་པའི། །ཕྱོགས་སྲེག་འབར་བ་ལྟ་རུ་ཎཾ། །བྷ་ཡ་བྷསྨཱ་ཤ་བྷུ་ཛཾ་ག་གྲྭ་ས་གྲོ་དྷཱཕ་ཎི་དཱི་ཥ། །བུར་རྟོགས། །འཇིགས་པས་ལུས་ཉམས་ལག་འགྲོ་གང་། །ཟ་འདོད་གདེངས་ཅན་དགྲ་ཡིས་ནི། །ཨཱ་བིདྷཱ་པཱ་ཏ་ནི་རྒྱུ་ཏ་གྷོ་རོ་ཡོ་བརྟ་ཏེ་ཀྵ་ཎ། །ཨི་ཏུམྨེ་ཏི་ཎི་སཱ་ཀུན་ཏུ་འབིགས་བྱེད་བསྙེན་ཨེཔའི་སྒྲ། །འཇིགས་སུ་རུང་བ་དུས་འདིར་འཇུག། དེ་ལྟར་ཐེ་ཚོམ་རབ་།
