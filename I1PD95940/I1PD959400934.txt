བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེཔྲ་སཱ་དཱ་ནནི་རཱི་ཙཀྲེ་བྱཀྟ་ས་་པ་ཐཾ་ཀ་ཐཾ། །ཏཱི་དཎྜ་ཀུཎྜི་བདེན་པའི་ལམ་ནི་གསལ་བའི་གཏམ། །རབ་དང་ཀུན་དགའ་ལྡན་པར་མཛད། །དབྱིག་པ་གསུམ་དང་སཾ་བྷ་རཾ་ཏྱཱཀྐཱ་བྷཱ་ར་བི་བཱ་དྷ་ག། །པྲ་བྲ་ཛྭ་ཤྭ་ས་ནེ་ཤཱསྟུསུ་བྷ་རིལ་བའི་ཚོགས། །ལམ་འགྲོན་གྱིས་ནི་ཁུར་བཞིན་བཏང་། །སྟོན་པའི་བཀའ་ལས་རབ་བྱུང་ནི། །དེ་དྲོ་བྷྱཱ་ད་དེ་ཏ་ཏ། ་། ཨཆ་ཏེ་པ་ད་མ་ཐཱ་སཱ་བྱ་སནྡྷ་ཀྵེ་ཤ་སཾ་ཀཱ་ནས་རབ་བཟང་དག་གིས་བླངས། །དེ་ནས་ཉོན་མོངས་ཀུན་བཅོམ་པ། །དགྲ་བཅོམ་གནས་ནི་ཐོབ་གྱུར་ཡཱཏྲ༑། ཨ་ཙིནྟ་ཡ་ད་ཧོ་ནེ་ཡ་མུ་ཙི་ཏཱ་ཤི་ཥྱ་ཏཱ་མ་མ། །པ་བྷཱུ་མི་པ་རི་ནིཌྷ་ཎ་ནས། །བསམས་པ་ཀྱེ་མ་བདག་གི་ནི། །སློབ་མ་ཉིད་འདི་འོས་པ་མིན། །བདག་གི་མདུན་དུ་བཅོམ་མ་གྲེ་བྷ་ག་བ་ཏཀ་ཐཾ། །ཏསྨཱཀྐུཧྥ་མ་ཧཾ་དེ་ཧཾ་བི་ཧཱ་ཡ་སྭཱ་ལྡན་འདས། །མྱ་ངན་འདའ་བ་ཇི་ལྟར་བལྟ། །དེ་སླད་སྔོན་དུ་ལུས་བཏང་ནས། །བདག་ནི་རང་གི་ས་དཾ་བྷ་ཛེ། །ཨི་ཏི་སཾ་ཙིནྟ་ས་ཤ་ནིཔཉྩ་དྷིཥཱ་ན་སཾ་ཤ་ཡ། །གནས་དག་བསྟེན། །ཞེས་བསམས་བྱིན་བརླབས་ལྔ་དག་ནི། །བསྟེན་པ་དེ་ཡིས་དལ་བུ་ཡིསོ། །ཕ་ཎི་ནོརྨ་ཀ་ཀ་ལ་ཡཱ་པ་རི་ནིབྲྀ་ཏི་མཱསྟ་བཱན། །ཏ་སྭཱ་ཐ་དེ་ཧ་སཏྟཱ་གདེངས་ཅན་ཤུན་པ་དོར་བ་ལྟར། །ཡོངས་སུ་མྱ་ངན་འདས་པ་ཐོབ། །དེ་ནས་དེ་ཡི་ལུས་སྦྱོང་ལ། །རེ་བྷི་ཀླུ་ཎཾ་ས་ཧ་ཏཱིརྦ་ཀི། །ཀ་སྱ་སཾ་བྲཧྨ་ཙཱ་རཱི་བཱན་ནཱ་མ་བི་བཱ་ད་སཾ་ཛ་ཏ། །དགེ་སློང་རྣམས་དང་མུ་སྟེགས་དག། སུ་ཡི་ཚངས་སྤྱོད་དང་ལྡན་ཞེས། །རྣམ་པར་རྩོད་པ་ཡང་དག་སྐྱེས། །ཏ་སྭཱ་ཐ་བི་ཤི་ཀཾ་བོཊྚ་མ་སཏྟེ་ཥཱ་ཀྲི་ཡཱ་བི་དྷོ། །ཏཱིནི་ཀེ་ཥཱ་པྲ་ཡཱ་ཏེ་ཥཱ་དེ་ནས་མུ་སྟེགས་བྱེད་རྣམས་ཀྱིས། །དེ་ཡི་སྒྲུབ་བྱེད་ཆོ་ག་ལ། །ཁྱོགས་ཁྱེར་མ་ནུས་སོང་བའི་ཚེ། །སཧྥ་བྷིཀྵ་ག་ཎོ་བྱཱ་དྷཱཏ། །ཤ་རཱི་ར་ནི་ཙ་ཡསྟུ་སཾ་ཀྲྀ་ཏྭཱ་ཏ་སྣ་ཐ་དགེ་སློང་ཚོགས་ཀུན་དག་གིས་བསྒྲུབས། །དེ་ནས་དེ་ཡི་གདུང་གི་ཚོགས། །མཆོད་རྟེན་བྱས་ནས་བྷིཀྵ་བ། །ས་པྲཙྪསུ་ག་ཏཾ་པཱུཏྤ་པ་རི་ནིཝཱ་ཎ་ཀཱ་ར་ཎཾ། །ས་ཏཱ་དགེ་སློང་གི། །སྔོན་ལས་མྱ་ངན་འདའ་བ་ཡི། །རྒྱུ་མཚན་བདེ་བར་གཤེགས་ལ་དྲིས། །དེ་དག་—།
