མི་ཥཱ་གྷཱ་ཎ་ནིབྷཾ་ཎ། །ནི་ཤཾ་ཀི་རརྫ་ཏེ་བིཏྟ་མརྫི་ཏཉྩ་བི་བྷ་ཛ་ཏེ། །ནོར་མེད་རྣམས་ནི་འཆི་བར་འགྱུར། །མི་བདག་ཆོས་ཀྱི་ནོར་ལྡན་ན། །དོགས་པ་མེད་པས་སྐྱེ་བོ་རྣམས། །བི་བྷཏྟཾ་བྷུ་ཛ་ཏེ་སྤཎྞཾ་ཛ་ནིཛྷརྨ་དྷ་ནེ་ཚྭ་པི། །ཨུ་པ་པནྣཾ་ནོར་བསྒྲུབ་བསྒྲུབས་པའང་འགོད་བྱེད་ཅིང་། །བགོས་པ་གསལ་བར་ལོངས་སྤྱོད་བྱེད། །བློ་ལྡན་དེ་ཡི་བ་ཙཐུ་ཏཱ་དྷཱི་མ་ཏསྟ་སྱ་བྷུ་པ་ཏི། །ཏ་མཱུ་ཙེ་ད་ཤ་ནོ་ངོ་ཏིཔྲ་ས་ཚིག་དག་ནི། །ཕུན་ཚོགས་ས་བདག་གིས་ཐོས་ནས། །སོ་ཡི་འོད་ཀྱིས་ཅིབཀའ་དྲིན་ནི། །གསལ་བར་དཾ་དན་ཡནྟི་བ། །མ་ཏི་མཱན་ཀཱརྱ་ས་ཙི་བ་རྩྭ་མེ་བྷ་བི་ཏུ་མཆ་སི། །དྷི་དྷུ་རྨི་བྱེད་བཞིན་དེ་ལ་སྨྲས། །བདག་གི་བྱ་བའི་བློན་པོར་ནི། །བློ་གྲོས་ལྡན་པ་ཁྱོད་འགྱུར་འོས། །བློ་སྟོབས་སཱུདྲིདྷི་རེ་བ་དྷཱཟླ་ཏེ་དྷ་ར་ཎཱི་བྷ་ར། །ཨི་ཏི་རཱ་ཛ་བ་ཙཐུ་ཏཱ་ལྡན་པ་ཁྱོད་ལྟ་བུས། །འཛིན་བྱེད་ཁུར་ནི་ཨེའཛིན་པར་བྱེད། །དེ་སྐད་རྒྱལ་པོའི་ཚིག་ཐོས་ནས། །ནོར་སུ་དྷ་ནསྟ་མ་བྷཱ་ཥ་ཏ། །ན་བ་ཡཾ་རཱ་ཛ་སེ་བཱ་སུ་ས་བྷཱ་མཎྜ་ན་པཎྜི་ཏཱ། །བཟང་གིས་ནི་དེ་ལ་སྨྲས། །བདག་ནི་རྒྱལ་པོའི་གཡོག་འཁོར་ལ། །མདུན་སར་མཛེས་པའི་མཁས་པ་མིན། །སཱཙྪཙྩོདྷུ་ན་བི་ཙྪེ་དསུ་ནི་དྲ་སུ་ཁ་བི་ཀྲ་ཡ། །སེ་བ་ཧི་པཾ་སཾ་སཾ་གཡོག་ནི་སྐྱེས་བུ་རྣམས་དག་གི། །རང་དབང་སྐྱེད་ཚལ་རྣམ་པར་གཅོད། །བདེ་བའི་གཉིད་དང་བདེ་སཱ་ར་དུཁ་དེ་རྒྱ་བྷ་ཡ་ཀྲ་ཡ། །ཨཱི་ཤྭ་རིརྨསྟ་ཀ་རྒྱསྟ་ཙ་བ་འཚོང་། །འཁོར་བའི་སྡུག་བསྔལ་བཀྲེན་འཇིགས་ཉོ། །གཡོག་པོ་དག་ནི་རྐང་སྟེགས་བཞིན། །རྟག་ར་ཎཀྲྀ་ཏ་ཀྲྀ་ཏུ་ཏྭཾ། །པཱ་ད་པ་ཋ་ཨི་བཱ་ཡཱ་ཏི་སེ་བ་ཀོཉཧ་སྦི་ཏས་དཱ། །སེ་ཏུ་འོག་ན་གནས་པ་ཡི། །མགོ་ལ་དབང་ཕྱུག་ཞབས་བཀོད་ན། །བྱ་བ་བྱས་པ་ཉིད་དུ་འགྱུར། །གཡོག་བཱ་ཡ་ས་པྲ་ཡཱ་སེ་ན་པཱབྷྱཱ་ན་མ་པི་སཾ་པ་དཾ། །སཾ་བྷོ་གཔི་ཤུ་ན་གི་ཉོན་མོངས་ངལ་དུབ་ཀྱིས། །ཕུན་ཚོགས་ཐོབ་པའང་ལོངས་སྤྱོད་ཀྱང༌། །ཕྲ་མ་ཅན་དང་འགྲོགས་པ་སཾ་གི་པྲ་བྷུ་བྷཱུ་བྷངྒ་བྷདྷུ་ར། །ཨེ་ཏཱ་སཱ་ནཱ་བ་ཏིཀྵནྟེ་པྲ་ཡི༑། རྗེ་བོའི་སྨིན་མ་འཁྱོག་པོས་གཡོ། །གཟུང་དཀའི་གདོན་བཞིན་མི་བདག་གི། ཕུན་—།
