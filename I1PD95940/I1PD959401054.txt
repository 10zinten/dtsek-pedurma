བསྟན་འགྱུར། འཁྲི་ཤིང་། [ཁི་བུ་རེ། །ཏ་ཏསྟོ་བ་དྷ་མཱ་ལཾ་ཀོ་ནཱི་ལ་རཀྵམྲ་ར་ཤྲཱི་ཏོ། །ཕྱེ་བར་འགྱུར། །དེ་ནས་གསོད་པའི་འཕྲེང་བས་མཚན། །སྔོན་དམར་གོས་ཀྱིས་གཡོགས་དེ་དག། ལྟ་རུ་ག་ག་ཏ་ཛ་ནཱ་ཀཱིཎྜོ་བ་དྷ་སྣ་ན་མུ་པཱ་ག་ཏོ། །ཛྙཱ་ཏཱ་ད་ཡཱ་བཱདྷབྦ་ཛྷའོངས་པའི་སྐྱེ་བོས་བསྐོརེ། །གསོད་པའི་གནས་སུ་ཉེ་བར་སོང་། །བརྩེ་ལྡན་ཀུན་མཁྱེན་གྱིས་མཁྱེན་ནས། །སཱ་ཡ་མ་བྱེ་རྱ་ཏྭཾ་བྷུ་ཁཾ། །པྲ་སཱ་དཱ་ནུ་གྲ་ཧེ་ཎི་བ་ཙཀྲེ་བི་ག་ཏ་རང་ཉིད་ས་དེར་མངོན་ཕྱོགས་ཏེ། །རྗེས་སུ་བཟུང་བའི་བཀའ་དྲིན་གྱིས། །བཅིངས་པ་དག་དང་བྲལ་བནྡྷ་ནོ། །སརྦ་ཛྙཱ་ཛྷ་ཡཱ་རཱ་ཛྷཱ་ཏོ་ཏྱཱཀྟ་མུཀྟ་པཱ་པ་ཏ་ཀོ། །པྲ་བྲ་བར་མཛད། །ཀུན་མཁྱེན་བཀའ་ཡིས་དེ་དག་ནི། །རྒྱལ་པོས་བཏང་ཞིང་སྡིག་པ་གྲོལ། །བཅོམ་ལྡན་ཛ་ཡཱ་བྷ་ག་བ་ཏཤཱ་ས་ནེན་ཏཱ་མཱ་པ་ཏུ། །ཛ་ན་ནཱ་སི་ཧི་ཏོ་དྲྀབྷཱུ་ཏོ་ཛྷཱ་ན་འདས་ཀྱི་བསྟན་པ་ལ། །རབ་ཏུ་བྱུང་ནས་དགྲ་བཅོམ་ཐོབ། །ཡེ་ཤེས་འབྱོར་འབར་དེ་དག་ནི། །མ་བི་བྷ་བོ་ཛཱ་ལོ། །ཏཥ་ཐཾ་ཀོ་ཏུ་ཀཱཀྲྀབློ་བྷིཀྵ་བྷིསུ་ག་ཏོ་བྱ་དང་བཅས་པ་མཐོང་གྱུར་ནས། །དགེ་སློང་གིས་དྲིས་དགེ་མཚན་ལས། །དེ་གཏམ་བདེ་བར་གཤེགས་པས་དྷཱཏ༑། མ་ཡེ་བ་རཀྴི་ཏཱ་བེ་ཏོ་པཱུབྦསྨིནྣི་པི་ཛརྨ་ནི། །ཨེ་ཏ་ཡོརྫ་གསུངས༑། སྔོན་གྱི་སྐྱེ་བ་དག་ལ་ཡང་། །འདི་དག་བདག་གིས་རབ་ཏུ་བསྲུངས། །དེ་ཚེ་འདི་དག་ཉིད་ན་ནཱི་བྱཱ་གྷི་གྷོ་ར་རཱུ་པཱ་ཏ་དཱ་བྷ་བཏྲ། །ཏསྨིན་ཡོ་ཀྲྀ་པཱ་བནྡྷབྦོ་དྷི་ས་ཏཱ་ཀྱི་མ། །སྟག་མོ་འཇིགས་རུང་གཟུགས་ལྡན་གྱུར། །དེ་དུས་བྱང་ཆུབ་སེམས་དཔའ་ནི། །བརྩེ་བས་ལུས་ཤ་རཱི་རི་ཏཱཾ། །ཨ་ཧཾ་ཀ་རུ་ཎ་རེ་ཎ་ཁྱཥི་ཏི་པཱ་ལ་སུ་ཏོ་བྷ་ཐཾ། །ཀ་དཱ་ཙི་ཅན་རྣམས་ཀྱི་གཉེན། །སྙིང་རྗེའི་རྡུལ་ཞེས་བྱ་བ་ཡི། །ས་སྐྱོང་བུར་ནི་ང་ཉིད་གྱུར། །ནམ་ཞིག་བཀྲེས་དེ་ཏོཀླུཎྜ་མཱ་པོ་ཏ་ཀོ་བྷོལཱུ་མུ་རྱ་ཏཱ། །སཱ་ཤ་རཱི་རསྨྲ་ཡཱ་ད་ཏྭཱ་བྷཱ་ཤྲཱི་སཱ་ཏུ་ནི་པས་ཉེན་པ་ཡིས། །ཕྲུག་གུ་འདི་དག་ཟ་རྩོམ་པའི། །སྟག་མོ་དེ་ནི་རང་གི་ལུས། །བྱིན་ནས་བདག་གིས་བཱ་རི་ཏཱ། །ཨ་བྱ་ཏཱ་བེ་བ་ཙོ་ར་ཏྭཾ་ཀརྨ་ཤེ་ཥཱ་དུ་སཱ་ག་ཏོ། །མ་ཡཱ་སཾ་རྣམ་པར་བཟློག། ལས་ཀྱི་ལྷག་མས་སྐྱིད་ལྟ་ཡང་། །འདི་དག་ཆོམ་རྐུན་ཉིད་དུ་གྱུར། །སྟག་མོ་དེ་ནི་མ་—།
