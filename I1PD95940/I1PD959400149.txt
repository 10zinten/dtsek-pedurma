དིགྷ་ར་སཾ་སབྦཾ་བྷོ་ཛུ་ཏེ་ཀལླི་ཏསྨྲ་ཡཱ། །ཨ་ཧོ་མ་མི་བ་སཾ་ཀཱནྟཾ་ཟས་དང་བཏུང་བ་ཀུན། །བདག་གིས་དུག་གིས་བསྒོས་པ་བྱས། །ཕྱི་ནས་གདུང་བའི་རང་བཞིན་དུག། པཉྩཏྟཱ་པ་མ་ཡཾ་བི་ཎཾ། །ཨི་ཏི་ཤུ་པཱ་ཚྭ་ཤྲཱི་གུཔུཾ་སྭཱ་ཐུ་ནེ་ཏྲ་ཀྲྀ་བཱ་ནི་ཀྱེ་མ་བདག་ཉིད་རང་ལ་འཕོ། །འདི་བརྗོད་དཔལ་སྦས་མཆི་མ་དང་། །བཅས་པའི་མིག་ལ་སྙིང་རྗེའི་དྷི༈། དྲྀཟླ་བ་བྷཱ་ཥ་བྷ་ག་བཱན་བྷིཀྵ་སཾ་གྷ་སྱ་ཤི་ཎ་ཏ། །གཏེར། །བཅོམ་ལྡན་འདས་ཀྱིས་གཟིགས་ནས་གསུངས། །དགེ་སློང་དགེ་འདུན་ཉན་པར་བྱེད། །བི་ཥཱ་དམྨཱ་ཀྲྀཐ་སཱ་དྷོ་ན་བ་ཡཾ་བི་མུ་ཁཱ་སྣ་ཡི། །གྷོ་ར་བི་ར་བི་ཥ་ཏུ་གཱཏ་ལེགས་པ་ཡི་མུག་མ་བྱེད་ཅིག། བདག་ཅག་ཁྱོད་ལ་ཕྱིར་མ་ཕྱོགས། །འཇིགས་རུང་ཁོན་གྱི་དུག་སྤངས་ན་བཱསྨན་ཏཱ་པ་ཏེ་བི་ཎི། །བཱ་རཱ་ཎ་ཧཱུཾ་བུ་རཱ་ཤཱི་མཱན་བྲ་དྨ་དཏྟོ་བྷ་བཏ་པས། །ངེད་ལ་དུག་གིས་གནོད་པ་མེད། །ཁྱོད་ནི་བཱ་རཱ་ཎ་སཱིར་སྔོན། །དཔལ་ལྡན་མི་བདག་ཚངས་ཚྭ་པ། །ཨ་བྷུ་ད་ནུ་པ་མཱ་ནཱ་མ་ཏ་སྦྲ་པྲ་ཎ་ས་མཱ་ཤི་ཡཱ། །སུ་བཎྜ་བྷཱ་ས་སཾ་ཛྷ་བྱིན་གྱུར། །དེ་ཡི་སྲོག་དང་མཉམ་པ་ཡི། །དགའ་མ་དཔེ་མེད་མ་ཞེས་པའོ། །དེ་ནས་དེ་ཡི་གྲོང་ཁྱེར་སྱ་ཏ་ཏ་པུ་རཱནྟ་བ་ན་སྦི་ཏེ། །ར་བསྨྲ་ཡཱ་ར་རཱ་ཛ་སྱ་སཱ་ཀ་དཱ་ཐཱ་ཤྲཱི་མཐའི། །ནགས་གནས་རྨ་བྱའི་རྒྱལ་མོ་ནི། །གསེར་དུ་སྣང་བ་ཞེས་པའི་སྒྲ། །དེ་ཡིས་ནམ་ཞིག་ཐོས་ཎོ་ཏ། །སཱ་ཏ་སྐྱ་ཤསྟ་མཱ་ཀཉྩ་བི་རེ་བཱི་ཎཱ་སྲ་ནོ་པ་མཾ། །ཀི་མེ་ཏ་དི་ཏི་ས་པྲཙྪ་པར་གྱུར། །གླིང་བུ་པི་ཝང་གདངས་ལྟ་བུ། །དེ་ཡི་སྒྲ་ནི་ཐོས་གྱུར་ནས། །དགེ་མཚན་ལྡན་པས་འདི་ཅི་ན་ར་ནཱ་ཐི་ས་ཀོ་ཏུ་ཀཱ། །རཱ་ཛ་བདྷ་ནཱསྟསྨིནྣསྟི་རཏྭཱཙྪཱ་དཾ་ཞེས༑། མི་ཡི་མགོན་ལ་དེ་ཡིས་དྲིས། །རྒྱལ་པོས་སྨྲས་པ་ནགས་མཐའ་འདིར། །རིན་ཆེན་སྒྲ་ལྡན་རྨ་ཤི་ཁཱི། །མ་དྷུ་རཾ་བྷུ་བ་ན་བྱཱ་པི་ཡ་སྤྱི་ཏཏ་ཀཌཱ་ཀཱུ་ཛི་ཏཾ། །ཨི་ཏི་བྱ་ཡོད། །གང་གིས་མགྲིན་ནས་བསྒྲགས་པ་འདིས། །འཇིག་རྟེན་རྣམས་ནི་སྙན་པས་ཁྱབ། །དེ་སྐད་ཐུ་བཱ་ཎོ་ཚིགས་པ་ཏིསྟཏྠནྡན་ན་མརྨི་ཏ། །སྲེ་མ་པྲ་ཡཙྪིཔྲ་མ་སཱུ་པྲ་སྨྲས་པའི་མི་བདག་གིས། །ཤིན་ཏུ་མཛའ་སྡུག་རབ་འབད་པས། །དེ་མཐོང་དོན་དུ་གཉེར་བ་ལ། །རབ་།
