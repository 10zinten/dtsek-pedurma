རཾ་བི་དྷཱ་ཡ་བི་དཏྟེ་བི་ཥ་ཡཱ་བྷི་མུ་ཁཾ་མ་ན། །ཏ་མ་བོ་ཙེ་ཏ་ས་ནི་ཏྱ་ཤིང་བསྒྲུབས་ནས་གཞོན་ནུའི་ཡིད། །ཡུལ་ལ་མངོན་པར་ཕྱོགས་པར་བྱས། །རྟག་ཏུ་འདོད་པ་རབ་སྤྲོས་ཀྲྀ་ཏྭཱ་ཀཱ་མ་སུ་ཏཱ་ར་ཎཾ། །ཨསྟི་བཏྠ་ཏཱ་དུ་ཙི་ཏཱ་ཀཱ་ཙིཏྠོ་ཥི་ཏ་བྷནྡོ་ག། །ནས། །དེ་ལ་དེ་ཡིས་རབ་སྨྲས་པ། །བུ་ཁྱོད་ལ་འོས་འགའ་ཞིག་ནི། །བདག་པོ་ཡུལ་གཞན་སོང་བ་ཡོད། །ཤུ་རྩེ་བེཐ་ནེ་ནིརཱི་པེ་རཱ་ཊོ་མོ་ནཱབ་ལམྦི་ནཱི། །ལཛྫ་བ་ཏཱི་སཱ་སུ་ཏ་རཾ་ཏཱ་ཡཱ་ཁང་སྟོང་སྒྲོན་མ་མེད་པར་ནི། །མཚན་མོ་མི་སྨྲ་ལ་བརྟེན་ཏེ། །ཤིན་ཏུ་ངོ་ཚ་ལྡན་མ་དེ། །ཁྱོད་དང་སཾ་ག་མ་མིཙྪ་ཏི། ་། ཨི་ཏི་ཏ་སྱ་བ་ཙཐུ་ཏྭཱ་སཱ་བྷི་ལཱ་ཌོ་བ་ཎི་ཀ་སུ་ཏ། །ཡང་དག་འགྲོགས་པར་འདོད། །ཅེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །མངོན་པར་འདོད་ལྡན་ཚོང་པའི་བུས། །ས་དཱ་གུསྟ་ཀྲྀ་ཧེ་བྷེ་ཛེ་ཛ་ན་རྩྭ་ར་ཏི་སཾ་ག་མཾ། །ས་ཏ་ཏཾ་སེ་བ་མཱ་ནཱ་ཡཿརྟག་ཏུ་སྦས་པའི་ཁྱིམ་དུ་ནི། །མ་དང་དགའ་མགུར་འགྲོགས་པ་བསྟེན། །རྟག་ཏུ་རབ་བསྒྲིབས་ཁྱིམ་དུ་ནི། །པྲཙྪནྣ་བྷ་བ་ནེ་ར་ཏིཾ། །བཛྷ་མཱ་ནཱ༔ས་རི་ཏ་སཱུ་རཱ་གཱཀྐིཔྲ་ཤ་མ་ཡ་ཡོ། །དགའ་མགུར་སྤྱོད་པ་སྟེན་བྱེད་པ། །དེ་ལ་ཆགས་པའི་མ་དག་ནི། །རབ་ཏུ་འཕེལ་ཞིང་ཞི་མ་གྱུར། །སཱ་ཙིནྟ་ཡ་དི་མཱི་ནི་ཧུཾ་པྲཙྪཱ་ད་ན་ཀ་དརྦ་ནཾ། །ན་སེ་ཧི་རཱུ་པ་བྷོ་ག་སྐྱ་དེ་ཡི་བསམས་པ་རྟག་ཏུ་ནི། །རབ་ཏུ་བསྒྲིབས་པའི་སྡུག་བསྔལ་འདི། །མི་བཟོད་གཟུགས་ཀྱི་ལོངས་པྲ་ཀཱ་ཤཀི་ལ་ཛཱི་བི་ཏཾ། །ཨ་སྐྱོ་ནྱ་བ་ད་ནཾ་བྷ་བི་ལོ་ཀ་ན་ར་སཾ་བི་ནཱ། །སྤྱོད་ལ། །སྣང་བ་ཉིད་ནི་འཚོ་བ་ལོ། །ཕན་ཚུན་བཞིན་གྱི་ཆུ་སྐྱེས་ལ། །རྣམ་པར་ལྟ་བའི་རོ་བྲལ་བའི། །ན་ཙུསྨྲ་ནེ་ན་སུ་ར་ཏེ་ཀ་ད་སྟི་སུ་ཁོཏྠ་པ། །ཏསྨཱཏྲཱཙྪཱ་ད་འོ་ཡི་®དགའ་མགུར་སྤྱོད་པ་ལ། །བདེ་བའི་དགའ་སྟོན་འགའ་ཡོད་མིན། །དེ་བསྒྲིབས་པ་ཡི་ཉོན་ནཱ་ཡཱ་པ་ཀུ་མཱ་ར་སྱ་པ་ཡནྣཱ་ཏ། །བྷཾཀྐཱ་པྲ་ས་བྷ་ས་བྷོ་གེ་པྲ་དི་ཤཱ་མོངས་ནི། །རབ་ཏུ་འབད་པས་བཅོམ་བྱས་ཏེ། །འཕྲལ་ལ་ལོངས་སྤྱོད་གཞོན་ནུ་མ། །གསལ་བར་མི་པྲ་གམྦྷ་ཏྭཾ། །།ཨི་ཏི་སཾ་ཙི་ན་སཱ་རཱ་ཏྲ་པྲ་བྷཱ་ཏེ་བྱཀྟ་བི་གྲ་ཧ། །བདག་གིས་བསྟན་པར་བྱ། །ཞེས་པ་མཚན་མོ་རབ་བསམས་ཏེ། །རབ་གསལ་ལེལུས་དེས་གོས་དག་ནི། —།
