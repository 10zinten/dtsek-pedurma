ཡནྡྲ་མནྡྲ་ཏནྣཱ་དི་ཏཏྲ་རཾ། །ཨཱ་ཧཱུ་ཡ་བི་ནི་བེ་སྨི་ནི་ཛཾ་སྨྲ་འཁོར་དང་། །སྔགས་དང་རྒྱུད་སོགས་དེ་སྦྱོར་མ། །བོས་ནས་དེ་ལ་རང་ཉིད་ནི། །ཆགས་པའི་ཟིལ་གྱིས་ར་པ་རཱ་བྷ་བོ། །ཡ་ཡཱ་ཏེ་དྷ་ན་སཾ་ཏུཊཱི་རཱ་གཱཎཱ་པྲི་ཡཾ་སཾ་ག་མཾ། །ཨ་མནན་པ་བཤད། །ནོར་གྱིས་ཚིམ་བྱས་ཆགས་པ་ཡིས། །གཟིར་མས་དགའ་དང་འགྲོགས་པ་ཞུས། །དེ་ཐ་མནྣཱ་བ་ལཱÉཋསྟ་ཡ་ས་ཡ་དི་པདྨ་ཀ། །བི་བ་ཤ༈་ས་མ་ཐེ་ནས་དེ་ཡིས་འཕྲལ་ལ་ནི། །སྔགས་སྟོབས་ཀྱིས་དྲངས་པདྨ་ཅན། །དབང་མེད་དེ་ལ་མངོན་ཕྱོགས་ཏེ། །ཏྱཱ་ཀིཾ་ཀ་རོ་མཱི་ཏྱ་བྷཱ་ཥཊ། །སཱ་ཏ་མཱུ་ཙེ་ཤ་ཤི་མུ་ཁཱི་བྷ་ཛ་པྲ་ཎ་ཡི་ནི་མི་མཱི། །བདག་གིས་ཅི་ཞིག་བྱ་ཞེས་སྨྲས། །དེ་ལ་དེས་སྨྲས་བསོད་ནམས་ཀྱིས། །ཉེར་གནས་མཛེས་མ་ཟླ་བཞིན་མ། །པུ་ལྕི་རུ་པ་ན་ཏྭཾ་ཀཱནྟཾ་དཱིལྟ་བཱ་བནྡི་མཱ་བི་ཤཏ། །པདྨ་ཀོ་བྱ་བ་ད་དྲྀཥཱ་ཛཱ་མཛའ་མོ་འདི་ནི་རབ་ཏུ་བསྟེན། །ཡང་ན་འབར་བའི་མེ་ལ་ཞུགས། །པདྨོ་ཅན་གྱི་མདུན་དུ་ནི། །མེ་ལི་ཏཾ་པཱ་བ་ཀཾ་པུ་ར། །པཱ་ཏ་སཛྫས་མུངྒཱི་ཚུ་བ་ས་ཏཱ་སི་ཏྭཾ་འབར་མཐོང་ནས་གོས་བཞག་སྟེ། །སྙིང་སྟོབས་དཀར་བའི་གོས་ཅན་གྱིས། །མཆོང་བར་ཆས་ནས་རབ་ཤུ་ཀ། །བི་ཤཱ་མི་སཱ་བ་ཀཾ་དཱིཔ་རཱ་གཱཀྐིནྣ་ཏྭཱ་པཱ་བ་ཀཾ། །པཾ་སྨྲས་པ། །གཙང་བྱེད་འབར་བར་བདག་འཇུག་གི། །མི་གཙང་ཆགས་པའི་མེར་མ་ཡིན། །ཆགས་སཱསྨྲ་ད་ན་དབྷཱ་ནཾ་དཱ་ཧོ་ཛརྨ་ཤ་ཏཱ་ནུ་ག། །ཨི་ཏི་ཏ་སྱ་བ་ཙ་པས་བསྲེགས་པའི་སྐྱེས་བུ་རྣམས། །སྐྱེ་བ་བརྒྱ་ཡི་རྗེས་འབྲང་སྲེག། ཅེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །ཐུ་ཏྭཱ་ཙཎྜ་ལཱི་ཁཎྜི་ཏཱ་ཤ་ཡ། །ཨ་ཀཱརྨ་ཀ་ར་ནོརླིག་པཉྩཏྟཱ་པི་ས་མཱ་ཡ་ཡོ། །ས་གཏུམ་མོ་རེ་བ་ཉམས་གྱུར་ཅིང༌། །བྱ་མིན་བྱས་པས་ཡིད་བྱུང་སྟེ། །རྗེས་སུ་གདུང་བ་དག་ཏུ་གྱུར། །དེ་ཏ་ཡཱདྷྱནྟི་ནི་ལ་ཡཔྲ་ཎ་ཡེ་ན་པྲ་སཱ་དི་ཏ། །བེ་བྱ་ཡཱ་ཙ་བི་རཱ་གྱ་ཡི་བཟོད་པའི་གནས་དེ་ལ། །གུས་པ་ཡིས་ནི་དང་བར་བྱས། །ཆགས་བྲལ་ཞི་ཞིང་བསིལ་བ་ཡིས། །བློ་ཤ་མ་ཤཱི་ཏ་ལ་ཡཱ་དྷི་ཡཱ། །བི་ད་དྷེ་དྷརྨ་ས་ཤུདྡྷོ་ཏ་ཡོ༔སདྷརྨ་དེ་ཤ་ནཱཾ། །ལྡན་སྨད་འཚོང་མ་དག་ཀྱང་། །ཆོས་ཀྱིས་ཀུན་ཏུ་དག་པའི་སླད། །དེ་དག་ལ་ནི་ཆོས་བསྟན་མཛད། །།
