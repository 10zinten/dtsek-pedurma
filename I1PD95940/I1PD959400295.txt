ཀི་ཏཱ། །ཨ་བི་ཤེ་ཥ་ཛྷ་པ་ལཱ་བྷཱུ་པཱ་ལཿཀཱ་ཀ་ཤཾ་ཀི་ན། །པྲ་བྷོཇི་རཀྟ་པ་མེད། །བྱེ་བྲག་མི་ཤེས་གཡོ་བ་ཡིས། །ས་སྐྱོང་བྱ་རོག་དོགས་པ་ཅན། །ངོ་བོ་ཆགས་བྲལ་རྟགས་ལིངྒ་ནི་བི་ལོ་ཀྱ་མཱ་ཏྱ་སཾ་ག་བ། །སྭ་སུ་ཏོ་གོ་པ་སཾ་ཧཱ་ཁྱོ་སཱ་ཤཾ་ཀཿརྣམས་ནི། །རྣམ་གཟིགས་བློན་པོའི་གང་ཟག་གིས། །རང་སྲས་ས་འཚོ་སེང་གེ་ཞེས། །བྱ་ལ་དོགས་སི་ར་མ་བྲ་བཱིཏ། །དྷཱཏྟིམྨེ་བི་ར་ཏཱ་ཤཱ་སཔི་ཤུ་ནེཛྲཔ་ཏེཀྲི་ཏ། །པྲ་ཏུ་བཅས་དལ་བུས་སྨྲས། །གཡོ་ཅན་ཕྲ་མས་མི་ཡི་བདག། བདག་ལ་ཡིད་བརྟན་བྲལ་བར་བྱས། །སྙིང་ནི་ཡཾ་ནི་ཏི་ཧྲི་ད་ཡེ་བི་དཱནཱ་པི་བྲ་དན་ཏེ། །བི་རཀྐསྦྲ་གི་ཏཱ་ལཱ་པ་དན་རྣམ་པར་ཕྱེ་བསྟན་ཀྱང་ཚེ། །གཅུགས་པ་ཉིད་དུ་འགྲོ་མ་ཡིན། །ཆགས་དང་བྲལ་བའི་ཇོ་བོ་ནི། །ཀུན་ན་ཤྲ་བ་ཎ་པྲ་བྷུ། །པ་ཕཱ་ཨི་བ་དྲྀདྷ་སྱ་ཡ་ཏཤི་ཐི་ལ་ཏྭཾ་མ་མ། །བི་བརྗོད་མཐོང་ཐོས་བོར་བ་དེ། །བདག་ལ་ལྷོད་པ་ཉིད་གྱུར་ཏེ། །རྒན་པོ་ཡི་ནི་ལུས་པོ་བཞིན། །ཕྲ་ཤུ་ནོདྷ་བ་བྷིནྡ་སྱ་པྲཎྞསནྟི་རྣ་བི་བྱ་ཏེ། །ན་མ་ཎིཐི་བྱ་ཏི་པུ་ནམས་ཐ་དད་རབ་བསྐྱེད་པའི། །མཛའ་བ་འདུམ་པར་མི་འགྱུར་ཏེ། །རྡོ་ཡིས་ེདུམ་བུར་བྱས་པ་ཡི། །པ་ཥ་ཎ་ཤ་ཀ་ལི་ཀྲྀ་ཏ། །དྲི་ཛི་ཧྭ་ཀུ་ཊི་ལ་ཀྲནྟཔྲ་བྷུཉྩནྡ་ན་པཱ་ད་པ། །ནོར་བུ་སླར་ཡང་འབྱར་བ་མིན། །ལྕེ་གཉིས་གྱ་གྱུས་ནོན་པ་ཡིས། །ཇོ་བོ་ཙན་དན་རྐང་འཐུང་ནི། །པཱ་ཡཱ་ཏྲརྦ་ཀྲི་ཡཱ་ཀཱ་རཱི་གུ་ཎ་བཱ་ན་མྱ་སེ་བྱ་ཏྭཾ། །ཀ་ཐཾ་སྲི་པ་ནི་དྷཱ་ནཱཎྜི་དོན་བྱ་བྱེད་ཅིང་ཡོན་ཏན་དང༌། །ལྡན་ཡང་བསྟེན་བྱ་ཉིད་མི་འགྱུར། །མི་བདག་གཏེར་ནི་དོན་གཉེར་བ། །ཀུ་ཤ་ལཾ་བྷ་ཛ་ཏེ་ན་ར། །གྷོ་ར་དྭེ་ཥ་བི་ཥཱ་བི་ཤ་ཛ་ཥ་ཧཱ་གྷ་ཏཱ་འཇིགས་རུང་སྡོང་བའི་དུག་འབེབས་པའི། །ལྕེ་གཉིས་པས་བཅོམ་རྣམ་འཁྲུགས་པའི། །མི་ནི་ཇི་ལྟར་བི་ཧྭ་ལ། །ཏམདྨ་ཛ་མསཾ་ཏྱཱ་ཚུ་དྭེ་ཥ་དོ་ཥ་ཎ་བྷཱུ་བ་ཏེ། །ཤཾ་ཀཱ་དགེ་བ་བསྟེན། །དེ་སླད་ཡང་དག་བཏང་ནས་འགྲོ། །་ས་བདག་སྡོང་བའི་སྐྱོན་གྱིས་ནི། །གང་ན་ཤ་ལྱཱ་མ་ཡི་བྲྀཏྟེརྨ་སྨིན་དེ་ཤེ་ན་ཏེ་ན་ཀིམ། །དཀྵ་རཀྵ་ཀླུ་མ༔ཤུ་རཿདོགས་པ་ཟུག་རྔུ་ཡི། །རང་བཞིན་གྱིས་འཚོ་ཡུལ་དེས་ཅི། །མཁས་ཤིང་རྟུལ་ཕོད་བསྲུང་བར་བཟོད། །།
