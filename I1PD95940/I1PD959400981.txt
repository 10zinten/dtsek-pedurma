སཾ་བྷོ་ག་བ་བརྫི་ཏཱ། །ཏཱརྡན་ནཱ་ཤ་ཡི་བཱ་ཧཾ་བྷཾ་ཏཱ་ས་ཁྱེ་བ་ནིསུ་ལོངས་སྤྱོད་དག་ནི་རབ་སྤངས་པ། །བདེ་མེད་བདག་ནི་གྲོགས་ཀྱིས་བཞིན། །ཁྱོད་མཐོང་རེ་བ་ཉིད་ཀྱིས་ཁཱ༑། ཡ་ད་སྱ་ནུ་ཙི་ཏེ་བཀྟི་དུ༔ཁཱཙྪཱ་དི་ད་མུ་ཙ་ཏེ། །བ་དནྟི་བཟུང་། །གང་ཡང་བརྗོད་པར་མི་འོས་པ། །འདི་ནི་སྡུག་བསྔལ་དག་གིས་བརྗོད། །མི་སྨྲ་འདོད་མོ་ན་ཀཱ་མེ་ཡཾ་ཛ་ནེ་ཛི་ཧཱ་པ་ལཛྫ་ཏེ། །ནཱ་ནཾ་གཾ་ཧཱ་དྷི་པ་ཏི་ནཱ་པཉྩཱཤཱི་ལ་ཀྱང་བརྗོད་པ་འདི། །ལྕེ་ཡང་རབ་ཏུ་ངོ་ཚར་ཤེས། །ཁྱིམ་བདག་གིས་ནི་ཆུང་མ་ཡི། །ཚུལ་ཁྲིམས་བསླད་སྱ་བི་སླ་བ། །དྲིཌོ་སྱ་ཛཱ་ཏཾ་བི་རཱ་གྱི་མ་ཀཱ་ལེ་ཀ་ཐ་མ་ནྱ་ཐཱ། །ཨི་ཏི་བིསྨི་མཐོང་དེ་ལ་ནི། །ཆགས་བྲལ་སྐྱེས་པ་མ་ཡིན་ནམ། །གཞན་དུ་དུས་མིན་ཇི་ལྟར་འགྱུར། །ཞེས་པ་གཞན་གྱི་ཏམུ་ཁསཥྚཔ་རསྤ་ར་བི་ལོ་ཀ་ནེ། །བ་དཏྱ་ནྱ་གུ་ཎགླུ་ནི་ག་ཎཱ་ན་ཡོན་ཏན་ནི། །ཉམས་པ་བགྲང་བ་ལ་མཁས་པའི། །བཞིན་འཛུམ་སྐྱེ་བོ་ཐམས་ཅད་ནི། །ཕན་ཚུན་ནི་སུ་ནོ་ཛ་ན། །བི་ཡོ་ག་ཏཔཾ་ཚྭ་ལོ་ཀསཱསྦྲ་ཙིཏྟེ་ཏི་མ་རྣམ་པར་ལྟ་ཞིང་སྨྲ། །འབྲལ་བས་གདུངས་པ་བདག་ལ་ནི། །འཇིག་རྟེན་སེམས་སོས་ཞེས་པར་ནྱ་ཏེ། །གླེ་ཤེ་་གླིཥཱ་རཱུ་པཱ་ཧཾ་ཏཱ་རུ་ཎྱ་སྱ་ཀ་རོ་མི་ཀིཾ། །ཤྲཱི་ཧེ་འཛིན། །དར་མ་ཉོན་མོངས་དག་ལ་ནི། །མ་ཉམས་གཟུགས་དག་ཅི་ཞིག་བྱ། །ཁྱིམ་དུ་གཡོག་པ་རི་ཛ་ནཉྩ་ཡེ་མ་མོཎྞཾ་ཁ་ལ་ཏྭཾ་ག་ཏ། །ཨ་སཱ་མི་ཀཱ་ནཱཾ་ནཱ་རི་ཎཱ་འཁོར་འདི་དག་ཀྱང་། །བདག་ལ་ཤིན་ཏུ་མ་བསྡམས་གྱུར། །བུད་མེད་བདག་པོ་མེད་རྣམས་ཀྱི། །མ་གྲེ་ཤཱ་པི་པྲ་གཝཱ་ཏེ། །དཱ་ས་སཱུ་ནུརྨ་མེ་བཱ་ཡ་མུདྲྀཥཱ་ན་བི་བནྟི་ཏ། །མདུན་དུ་ཁྱི་ཡང་རབ་ཏུ་དགོད། །བདག་གི་ལྷག་མ་ཉིད་ཀྱིས་ནི། །རབ་ཏུ་བསོས་པ་འབངས་ཕྲུག་འདི། །པ་ཤྩ་ཙཱ་དཱ་ན་ཀོ་པཱངྒེ་ཀེ་ཤཱ་ཀརྟ་ཎ་མཱི་ཧ་ཏེ། །ཨི་ཏྱཱཀྟ་ཀྲྀ་ཏ་བཀའ་བསྒོས་ཁྲོས་པས་བདག་གི་ནི། །སྐྲ་ནས་འདྲེན་པར་འདོད་ལ་ལྟོས། །ཞེས་པ་བརྗོད་ནས་བརྡ་སཾ་ཀེ་ཏཱ་ནི་ཛ་དཱ་ས་པ་རཱ་བྷ་ཐཾ། །བྷནྡྷུཤྲཱིཏྟ་བི་ཀཱ་རཱ་ཡ་པྲ་པྲ་ཏུ་ཀླུ་བྱས་ཏེ། །བདག་པོའི་སེམས་ནི་རྣམ་འགྱུར་སླད། །རང་གི་འབངས་ཀྱིས་ཟིལ་གནོན་པ། །དེ་ཡིས་།
