ཧ་སན་བུ་ན་ར་བྲ་བཱིཏ། །དན་ནི་དུརླ་བྷཾ་མུངྒེ་ཏཏ་བི་དྷཱཏ་བྷཱུ་ཏ་རཱུ་པི་ཎ། །ཏུ་དགོད་ཅིང་ཡང་སྨྲས་པ། །དེ་ལྟའི་ངོ་མཚར་གཟུགས་ཅན་རྣམས། །མཛེས་མ་མཐོང་བ་རྙེད་པ་དགའི། །ཏ་ཐཱ་པི་ཡ་དི་ནིརྦནྡྷཀྲྀ་ཡ་ཏེ་ཏཏ་པ་རི་ཤྲ་མ། །ཨི་ཏུཏྟཱ་ཧྲི་པ་ཏིསྟ་སྱ་གྲ་ཧ་དེ་ལྟ་ན་ཡང་འབད་པ་ན། །དེ་ལ་ཡོངས་སུ་ངལ་བ་བྱ། །དེ་སྐད་བརྗོད་ནས་མི་བདག་གིས། །རྒྱ་ཡིས་ཎ་ཛཱ་ལཱ་ཛི་བི་ན། །བྱ་སྲི་ཛནྟཊཾ་པྲསྟེ་བི་དྷཱ་ཡ་བ་དྷ་སམྦི་དི། །བ་འཚོ་བ་དེ་གཟུང་ཕྱིར། །མ་ཐོབ་ན་ནི་གསོད་པ་དག། ཡང་དག་རིག་པར་བྱས་ནས་བཏང་ཚེ། །ཡོངས་ཤི་ཀྲྀ་ཏོ་ན་བེ་ཥཱ་བ་མོ་ཧ་དཀིཾ་པ་རཱིཀྵ་ཡཱ། །ཨ་ནུ་རཱ་གཱ་ཧ་ཏམི་བྷི་སུ་སྤྱོད་པའི་མིག་རྨོངས་པས། །དབང་དུ་བྱས་པ་མི་ཤེས་ཉིད། །བུད་མེད་རྗེས་ཆགས་ཀྱིས་བཅོམ་པས། །ར་ཀཱམཱ་ཅུ་པི་ཀཱ་རྱ་ཏེ། །པཱདྷཱ་ནཾ་པྲ་ཎ་ཡཱཏ་བཏྤཔོ་དྷཱ་ཡཱཔཱ་ད་བཱི་ཥ་ཏྭཾ། །བྱ་བ་མིན་པ་རྣམས་ཀྱང་བྱེད། །མཛའ་ལས་མ་རུངས་ཁྱིམ་ཐབ་ཀྱི། །རྐང་པའི་སྟེགས་སུ་གྱུར་རྣམས་ཀྱི། ཨཱིརྟ་ཡི་བ་བི་ན་ཤནྟི་དྷཱི་བྷཾ་ཏི་སྨྲ་ཏི་ཀཱིརྟ་ཡ། །ཏ་ཏཤཱ་ཀུ་བློ་དང་དགའ་བྲན་གྲགས་པ་རྣམས། །ཕྲག་དོག་གིས་བཞིན་རྣམ་པར་ཉམས། །དེ་འོག་བྱ་བས་གནས་ནི་ཀི་ནྱཔཱ་ཤ་བནྡྷཔ་དེ་པ་དེ། །པྲ་བྷ་བོདྡདྷི་རཱ་ཛ་སྱ་བྱ་ཤྲཱིརླན་གནས་སུ། །ཞགས་པས་བཅིང་བ་བཀོད་པ་རྣམས། །རྨ་བྱའི་རྒྱལ་མོའི་མཐུ་ཡིས་ནི། །ཀུན་ཏུ་རྣམ་པ་སཾ་ཏཏ། །དུཁི་ཏཱ་ན་ཡནྣཱ་བི་ཕ་ལྕཏ་བི་ཏཱནྣཾ་པ་ཏི་ཤ་བ་པར་ཆད་པར་གྱུར། །འབངས་འབྲས་མེད་ལས་སྡུག་བསྔལ་ཞིང་། །མི་བདག་བཀའ་ཡིས་འཇིགས་གྱུར་ནཱ་ཏ། །མ་ཡུ་ར་རཱ་ཛསྟན་པྲ་ཀ་རུ་ཎཱ་ཀུ་ལ་ཏཾ་ཡ་ཡོ། །སེ་ཙིནྟ་ཡ་ད་ཧོ་བྷི་ཏཱཏེ། །མཐོང་ནས་རྨ་བྱའི་རྒྱལ་མོ་ནི། །སྙིང་རྗེ་འཁྲུགས་པ་ཉིད་དུ་གྱུར། །དེ་ཡིས་བསམས་པ་ཁོ་མོ་གླུ་པ་ཏེཀཱ་ར་ཤཱ་ས་ནཱཏ། །མད་བནྡྷ་ན་བི་སམྦྷ་དཱདྲ་ར་ཀཱ་ཛཱ་ལཱ་དག། འཆིངས་པར་མ་གྱུར་ས་བདག་གི། བཀའ་ནི་དྲག་པོའི་འཇིགས་གྱུར་པ། །ཀྱེ་མ་སྡུག་པ་རྒྱ་ཛཱི་བི་ན། །ཨི་ཏི་སཾ་ཙིནྟ་ཀྲྀ་པ་ཡཱ་སྤཥཱ་བཱ་དཱི་བི་སྭ་ཛུ་ཏཱ་ན། །ཚིགས་པ་：ཡིས་འཚོ། །དེ་ལྟར་བསམས་ནས་སྙིང་རྗེ་ཡིས། །གསལ་བས་སྨྲས་ཏེ་དེ་བཏང་ནས། །མི་བདག་—།
