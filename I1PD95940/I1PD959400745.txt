ཡཾ་བྷོ་དྷི་ས་ཏཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏ་ཡཾ་ཀུ་ཎཱ་ལཱ་བ་དཱ་ན་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ཀུ་ལཱ་ལའི་རྟོགས་པ་བརྗོད་པའི་ཏཔཟླ་བ། །།།ཡལ་འདབ་སྟེ་ལྔ་བཅུ་རྩ་དགུ་པའོ། །།།།མེ་ཀོནྣ་ཏཨི་ཧ་ཀ་ཥ་ཏི་ཤ་རཱི་རཾ་གླེ་ཤ་རཱ་ཤིརྣ་རཱ་ཎཾ་ད་ཧ་ཏི་ཙ་པ་ར་ལོ་ཀེ་འདི་ནི་མི་རྣམས་ལུས་དག་ཉོན་མོངས་ཕུང་པོས་རབ་ཏུ་གདུངས། །འཇིག་རྟེན་ཕ་རོལ་ཏུ་ཡང་ནཱ་ར་ཀཀྲུ་ར་བ་ནྣི། །ཤ་ར་ཎ་ག་མ་ན་པུ་ནྱ་པཱསྟི་ཤིཀྵ་པ་དཱ་ནཾ་མི་བཟད་དམྱལ་བའི་མེ་ཡིས་བསྲེགས། །སྐྱབས་འགྲོའི་བསོད་ནམས་དང་ནི་བསླབ་པའི་གནས་དག་ཐོབ་པྲ་བྷ་བ་ཏི་ན་ཏུ་དེ་ཧེ་དུ༔ཁ་དཱ་ཧཀ་དཱ་ཙིཏ། །དྷ་ན་ནཱ་མཱ་ས་རྣམས་ཀྱི། །ལུས་ལ་ནམ་ཡང་སྡུག་བསྔལ་དག་གིས་སྲེག་པའི་མཐུ་ཡོད་མིན། །རྒྱ་མཚོའི་མཐའ་@ནི་ནོར་མུ་དཱནྟེ་ནཱ་གོ་བྷུདྡ་ཧུ་བཱནྟ་བ། །ཕ་ཎ་རཏྟཱི་ཛཱ་ལཱ་ལོ་ཀ་ཀ་ལི་ཏ་ཞེས་པའི། །ཀླུ་ནི་གཉེན་མང་ལྡན་པ་བྱུང་། །གདེངས་ཀའི་རིན་ཅེན་@རབ་འབར་གྱི༄༅། །སྣང་བས་པཱུཏྤ་བ་ས་ར། །པ་པཱ་ཏཱ་ཧནི་ཤཾ་ཏ་སྱ་བྷ་བ་ནེ་ཏལྟ་བ་ལུ་ཀཱ། །ཡ་སྔོན་མེད་ཉིན་མོར་བྱས། །དེ་ཡི་ཁང་པར་ཉིན་མཚན་དུ། །བྱེ་ཚན་དག་ནི་བབས་གྱུར་ཏེ། །གང་ཡཾ་གེ་ཥཱ་བྷུ་ཛཾ་གཱ་མཱ་ནཾ་ཏཱི་པྲ་ཏཱ་པ་བྱ་ཤྲཱི་བྷ་བ་ཏ། །ཀ་དྭ་ཙིཏྨ་དྷ་ནོ་ནཱ་མ་པུ་ཏྲཿགིས་ལག་འགྲོ་རྣམས་ཀྱི་ལུས། །གདུང་བ་དྲག་པོས་ཉེན་པར་གྱུར། །ནོར་བཟང་ཞེས་བྱ་གཅེས་པའི་བུ། །པྲ་ཏྭཾ་པྲི་ཡ། །སུ་ཀུ་མཱ་རཿཔྲ་ཀྲྀ་ཏེ་པ་བཱ་ལུ་ཀཱ་པ་རི་པཱི་ཌི་ཏ། །རང་བཞིན་ཉིད་ཀྱིས་རབ་གཞོན་པ། །བྱེ་མས་ཡོངས་སུ་གཟེར་གྱུར་པས། །ནམ་ཞིག་དེ་ལ་རབ་ཏུ་དྲིས། །ཀརྨ་ད་མ་ནི་ཡནྟ་ཏེ་བཱ་པ་ཏེ་ཏཔ་བཱ་ལུ་ཀཱ། །མནྟྲ་མཱུ་ལ་པྲ་ཡོ་གེ་ན་ཀེ་ཡབ་གཅིག་བྱེ་མ་ཚ་བ་འདིས། །བདག་ཅག་ཅི་སླད་གདུང་བར་བྱེད། །སྔགས་དང་རྩ་བའི་སྦྱོར་བ་དག། ནེ་ཡ་མུ་པ་ཤཱ་མྱ་ཏི། ་། ཨསྨྲ་ད་བྱཱ་དྷི་ཀཱཀེ་ཙི་དསྨཏྤ་ཏྱ་བ་རཱཔ་ཅི་ཡིས་ཉེ་བར་ཞི་བར་འགྱུརམེ། །འགའ་ཞིག་བདག་ལས་ལྷག་པ་དང་། །གཞན་ནི་བདག་ལས་དམན་པ་།
