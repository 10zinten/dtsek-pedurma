སརྦ་ཛྷ་མཱ་ནི་ན། །ན་སེ་ཧི་རེ་ཏ་དུཏྟཱནྟཾ་པྲ་ཀཱ་ཤ་མི་པ་ཀོ་ཤི་ཀཱ། །དུག་གིས་རབ་གདུངས་པས། །དེ་ཡི་ཁྱད་འཕགས་མ་བཟོད་དེ། ཀོའུ་ཤི་ཀས་སྣང་བ་བཞིན། །མ་ལི་ནི༔སཱ་བི་ནི་ཤཱ་ཡ་པ་ར་བྷཱ་གོ་དི་ཏིཔ་དཱ། །ཀྲི་ཡ་ཏེ་བཱ་ས་ར་སྤཌྷ་ནུབ་མོར་གྱུར་པའི་རབ་རིབ་ཚོགས། །དྲི་མ་ཅན་ནི་ཤར་ཆ་ནས། །འཆར་བར་རྟག་ཏུ་ཉིན་པར་ལ། །ཤརྦ་རིསི་མི་རོཥ་རི། །མཀ་རཱི་སཾ་ཛ་ཡི་བི་རཱ་ད་ཛི་ཏཀ་ཀུ་དལྟ་ཐཱ། །རང་ཉམས་བྱ་སླད་འགྲན་པར་བྱེད། །གནག་ལྷས་བུ་དང་ཡང་དག་རྒྱལ། །མི་ཕམ་ནོག་ཅན་དེ་བཞིན་དུ། །པཱུ་ར་ཎ་ཛྷ་ཏི་པུ་ཏྲ་བྱ་མཱཎཱ་ཀྵ་པ་ཎ་ཀཱས་རེ། །ཨུ་ཙུནྡོ་པ་ཏི་མ་བྱེ་རྱ་རྫོགས་བྱེད་གཉེན་གྱི་བུ་ལ་སོགས། །རྨོངས་པ་ཟད་བྱེད་གཞན་དག་ཀྱང་། །བདུད་ཀྱི་སྒྱུ་མས་རྨོངས་བྱས་མཱ་ར་མཱ་ཡཱ་བི་མོ་ཧི་ཏ། །སཾ་གྷ་དྭེ་ཥ་དོ་ཥ་ཎ་དྷཱ་མེ་ནེ་བཱནྡྷ་ཤིང་། །འགྲན་འདོད་ཞེ་སྡང་སྐྱོན་གྱིས་ནི། །དུ་བས་བཞིན་དུ་མུན་གཏིབས་པས། །མི་བདག་ལ་མངོན་ཀཱ་རི་ཏཱ། །།ཨེ་ཥ་སརྦ་ཛྙཱ་ཏཱ་མཱ་ན་བ་ནེ་ཡ༔ཤ་མ་ཎསྦི་ཏ། །རིདྡྷི་ཕྱོགས་ནས་སྨྲས། །ནགས་ན་འདུག་པའི་དགེ་སྦྱོང་གང་། །འདི་ནི་ཀུན་མཁྱེན་ང་རྒྱལ་ཅན། །ཁྱོད་ཀྱིས་པྲ་བྷཱ་བོ་བྷ་བ་ཏཱ་ཏས་ཀཾ་ཙ་དྲྀ་ཤྱ་ཏྭཾ། །རཾདྷི་པྲ་བྷཱ་བཙྩ་ཡཏྟི ིཏ་ཛ་ན་དེ་དང་ངེད་རྣམས་ཀྱི། །རྫུ་འཕྲུལ་མཐུ་ནི་བལྟ་བར་བྱོས། །རྫུ་འཕྲུལ་མཐུ་ལས་གང་ཅུང་ཟད། །སྐྱེ་བོ་བྱཱ་བཱ་ཛ་ནོརྫི་ཏཾ། །དརྱ་ཏེ་མ་ཧ་དཱཉྩནྡྷཾ་པྲ་ཏི་ཧཱརྱ་ཏ་དུ་ཙ་ཏེ། །ཤཀྟིམགུ་བར་@རབ་སྤེལ་བ། །ངོ་མཚར་ཆེ་བ་མཐོང་གྱུར་པ། །དེ་དག་ཆོ་འཕྲུལ་ཞེས་སུ་བརྗོད། ་། རྒྱལ་པོ་སཾ་ས་དི་ཡ་སཱུསྟི་པྲ་ཏི་ཧཱརྱ་སྱ་དན་ནེ། །ཨསྨ་ཀཾ་ཏ་སྱ་བཱ་རཱ་ཛན་པཱུ་བདག་ཅག་གམ་དང་ཨདེ། །མདུན་སར་ཆོ་འཕྲུལ་སྟོན་པ་ཡི། །ནུས་པ་གང་ལ་ཡོད་གྱུར་དེ། །འགྲོ་བ་ཛུ་སསྡུ་ཛ་གཏྟཱ་ཡེ། །ཨི་ཏི་ཏེ་ཏཱཾ་བ་ཙ་ཐུ་ཏྭཱ་ཏདྡརྩ་བི་མུ་ཁོ་ཚྭ་པ། །གསུམ་དུ་མཆོད་འོས་འགྱུར། །ཞེས་པ་དེ་དག་ཚིག་ཐོས་ནས། །མི་བདག་དེ་དྲེགས་ལས་ཕྱིར་ཕྱོགས། །ཨུ་བཱ་ཙ་བཱིཙྪཱ་ཀེ་ཡཾ་བཔ་ཤུ་ནཱཾ་གི་རི་ལཾ་གྷ་ནེ། །ཨ་པ་མཾ་ཛ་པ་མི་བི་ཏཏ་སྨྲས་པ་ཁྱེད་བསམ་འདི་ཅི་ཞིག། འཕྱེ་བོ་རི་ལ་འཛེག་པར་འདོད། །འདི་ནི་རིགས་མིན་ཕྱེ་མ་ལེབ། །།
