མ༔་ཐ་སཾ་ག་མོ་ཙེ་ཏ་ཏ་ནོསཱ་སཉྩ་བྷི་ཏི་པྲ་དཱ་ལཱི་ལཱ་སྱས་ཧ་སཱ་བི་འགྲོགས་འོས་ལུས་ཅན་འདི་ཡི་འཇོ་སྒེག་འགྲན་ཟླ་དག་ལ་འཇིགས་སྟེར་བས། །དགའ་མའི་རོལ་རྩེད་རྦ་ལཱ་ས་ལ་ཧ་རཱི་ཤོ་ཥ་བི་དྷཏྟེ་ར་ཏེ། །ཨུཛྙཱ་ཏེ་ཀ་ཟི་ཎེ་ས་ཡོ་དྷ་ར་ཡུ་གེ་ན་རླབས་དག་ནི་འཕྲལ་ལ་ཉིད་དུ་སྐེམས་པར་བྱེད་དམ་སྙམ། །ཀྱེ་མ་གང་གི་འོ་མ་འཛིན་པ་ཟུང་ནི་རབ་ཏུ་དྲི་བེ་ཀེ་ཙ་རཾ་ཡ་སཱུ་དོ་ཥ་མ་ཡེ་སྱ་ཧོ་གུ་ཎ་བཱ་ཏཱ་ཧཱ་རེ་ཎ་བདྷཱ་སྲ་མཁྲེགས་དབེན་པ་ཉམས། །སྐྱོན་གྱི་རང་བཞིན་ཡིན་ཡང་ཡོན་ཏན་ལྡན་པའི་དོ་ཤལ་དག་གིས་གནས་སྦི་ཏི། །ཨ་སྱསྨིནྣ་བ་ལ་བ་ཏེཞསླུ་ཛ་དྷི་ཡཱ་རོ་ལམྤ་རེ་ཁཱ་མུ་ཁེ་ལོ་ལཱ་གླུཔར་བྱས། །བཞིན་འདི་ལ་ཡང་བུང་བའི་འཕྲིང་®རྣམས་ཆུ་སྐྱེས་བློ་ཡིས་ཉེ་བར་བརྟན་པ་དང་། །གཡོ་ཀི་མ་པི་པྲ་ཤཱནྟི་ན་ཡ་ནཾ་ལཱི་ནཾ་མུ་ནཱི་ནཱརྨ་ན། ་། ཨི་ཏི་ཙིནྟ་ཡ་ཏསྟ་ལྡན་མིག་གི་འདྲེན་བྱེད་ཐུབ་པའི་ཡིད་ཀྱང་ཅི་ཡང་ཆགས་ཤིང་ཉམས་པར་བྱེད། །དེ་ལྟར་སེམས་པ་དེ་ཡི་སྱ་བ་པུཔུཋ་ཤ་རོ་པ་མཾ། །བི་ལོཀྱ་བྷཱུ་པ་ཏི་སུ་ཏཱ་བ་བྷུ་བཱཉྩརྨ་ནིཉྩ་ལུས༑། མེ་ཏོག་མདའ་ཅན་ལྟ་བུ་དག། མཐོང་ནས་ས་བདག་བུ་མོ་ནི། །ངོ་མཚར་གཡོ་བ་མེད་པར་ལཱ། །ཧཱུྃ་ཏེ་ལཛྙཱ་ཤུ་ཀེ་ན་མཿསྨ་རེ་ཎ་སྨི་ཏཀཱ་རི་ཏཱ། །པྲ་ཏྱ་གྲ་པུ་ལ་ཀཱ་ཀཱིརྞ་གྱུར། །དྲན་པ་འཛུམ་པར་བྱེད་པ་ཡིས། །དེ་ཡི་ངོ་ཚའི་གོས་ཕྲོགས་ཏེ། །གསར་དུ་སྤུ་ལོང་རྒྱས་པ་ཡིས། །བ་པུསྤཥཱ་ན་པྲ་བྱ་ཏ། །རུདྡྷཱ་ན་བཱ་བྷི་ལཱ་ཥ་ཎ་བེ་ལ་ཎ་ནི་ལུས་ནི་གསལ་བར་མཐོང་མ་གྱུར། །མངོན་འདོད་གསར་པས་བཀག་གྱུར་ཀྱང་། །སྐྱེངས་པ་ཡིས་ནི་ཕྱིར་བནྟི་ཏ། །མ་ནས་དྲི་བ་ནི་ཀྱི་པྱ་སཱ་ཤུ་ནྱ་པ་ཤ་ནིརླ་ཡོ། །།རཱ་ཛ་དྷ་ནཱི་ས་མཱ་བཟློག་སྟེ། །ཡིད་ནི་དེ་ཉིད་ལ་བཞག་ནས། །དེ་ནི་སྟོང་བཞིན་དལ་གྱིས་སོང་། །རྒྱལ་པོའི་ཕོ་བྲང་ཕྱིན་ས་བྱ་ལཛྫ་བིསྨྲ་ཡ་མརྨ་ཐི། །།མཱི་ལི་ཏེ་བ་ནི་ལཱི་ནེ་བ་པ་པྲ་ཎི་ཏེ་བ་གྱུར་ནས། །ངོ་ཚ་ཡ་མཚན་ཡིད་སྲུབས་ཀྱིས། །ཟུམ་པ་བཞིན་དང་ཞུམ་པ་བཞིན། །བཅོམ་པ་བཞིན་དུ་བྷུ་བ་སཱ། །ཀུ་མཱ་རོ་པི་སྭ་བྷ་བ་ནེ་ས་མུདྡྷཱ་ཏ་མ་ནོ་བྷ་བ། །ཏཱ་མེ་བེནུ་མུ་ཁྲི་དེ་གྱུར་ཏོ། །གཞོན་ནུ་ཡང་ནི་རང་ཁྱིམ་དུ། །ཡིད་ལས་བྱུང་བ་ཡང་དག་རྒྱས། །མདུན་དུ་ཟླ་བཞིན་མ་—།
