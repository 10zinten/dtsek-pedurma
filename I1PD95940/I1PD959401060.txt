བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེབྱ་བ་ཀྟ་ཎཾ། །མནྣཾ་སཾ་ཧཱ་ར་ཀཱ་མཱ་སྟེ་ས་ཚྭ་ཧནཱུ་མུརླ་ཡུ། །ཏ་ལོག་ཚེ། །བདག་ཤ་ཟ་འདོད་དེ་དག་རྣམས། །ཐམས་ཅད་བདག་ནི་གསོད་པར་བརྩོན། །རྡོ་ཡི་ཆར་དུéཥྛབླ་བྲིཉྩ་བི་མ་ཧ་ཏུ་པིཎྜི་ཏཱ་ཀྲྀ་ཏེ། །ཀཱ་ཡེ་དྲྀ་ཌྷ་ཀ་བཱ་ཊ་སྱ་ན་མེ་ཆེན་རབ་སྤྲོས་ཀྱང་། །ཁོག་པ་རབ་ཏུ་བརྟན་ལྡན་ཞིང་། །གོང་བུའི་དབྱིབས་ཅན་བདག་གི་ནི། །ལུས་ཀཱ་ཙི་ད་བྷཱུཏ་ཏི། །ཏ་ཏསྟནྣཱཏྲ་རི་ཀྵ་མཱཚཱབྷྱཱ་ཧཾ་ཀ་རུ་ཎ་ནི་ནམ་ཡང་ེཉམས་མ་གྱུར། །དེ་ནས་བཀྲེས་པས་ཡོངས་ཉེན་དེ༄༅། །མཐོང་ནས་སྙིང་རྗེས་བདག་ཀུ་ལ། །སྭ་ཡཾ་ཏེ་བྷེསྟ་ཚུ་ད་ཏྭཱ་པྲ་ཡཱ་ཏཀྲྀ་ཏ་ཀྲྀ་ཏུ་ཏྭཾ། །ཏ་ཨེ་ཏེ་འཁྲུགས་པས། །རང་གིས་དེ་དག་རྣམས་ལ་ལུས། །བྱིན་ནས་བྱ་བ་བྱས་ཉིད་གྱུར། །སྔོན་གྱི་ཚོང་པ་བ་ཎི་ཛཔུཏྤ་མ་རྱ་ཏཱ་བ་སཏྭཾ་ག་ཏཱ། །ཀྲྀ་ཏཱ་ག་པོ་པི་ཀཱ་རུ་ཙྪཱཎཱ་བིཏཀུ་དེ་དག་དང་། །དཀའ་ཐུབ་ཅན་ནི་འདི་དག་གྱུར། །གནོད་པ་བྱས་ཀྱང་སྙིང་རྗེ་ཡིས། །བདག་གིས་དགེ་ཤ་ལཾ་མ་ཡཱ། །ཏེ་ཤཱyཤཱ་ས་ནེ་པུཤྲཱི་ཀཱ་ཤུ་པ་སྱ་མ་ཧཱཏྨ་ན། །པྲ་པྲ་ཛྙཱ་བ་དག་ལ་བཀོད། །དེ་དག་སྔོན་ཚེ་བདག་ཉིད་ཆེ། །སྟོན་པ་འོད་སྲུངས་བསྟན་པ་ལ། །རབ་བྱུང་ཐོབ་པཱ་སྱ་སཾ་ཛཱ་ཏཱསྟཏྤ་ལ་སཱུ་རྱ་བྷཱ་གི་ན། །ཏ་ཐཱ་ག་ཏེ་ན་ཀ་ཐི་ཏཾ་ཐུ་ཏྲེ་དཾ་ནས་དེ་ཡི་འབྲས། །ད་ལྟ་སྐལ་ལྡན་དག་ཏུ་སྐྱེས། །དེ་བཞིན་གཤེགས་པས་གསུངས་པ་འདི། །ཐོས་ནས་ཏདྷི་ཀྵ་བག་ཚྭ། །པྲ་ཤ་ཀཾ་པུཔ་རཾ་ཏ་སྱ་ཀ་མཱ་མི་བྷ་རཀྵ་མཾ། །ཡེ་དགེ་སློང་རྣམས་ཀྱིས་ནི། །དེ་ཡི་བཟོད་པ་ས་གཞི་བཞིན། །ཁུར་དག་བཟོད་པ་རབ་ཏུ་བསྔགས། །གང་ཊཱ་ནིརྨ་ལ་ཤཱི་ལ་ཤཱི་ཏ་ལ་ཛ་ལཱ་སཏྭཱ་ཤ་ཡཱ་ཤཱ་སི་ནཱི་ནི་ཧུཾ་བི་ར་ར་ཛཔྲ་ཞིག་ཚུལ་ཁྲིམས་དྲི་མེད་བསིལ་བའི་ཆུ་ལྡན་སེམས་ཅན་བསམ་པ་དབུགས་འབྱིན་བྱེད། །ཁོན་གྱི་རྡུལ་ནི་མཱརྫ་ན་ན་དཱི་གྷནྟིསྦི་ཏཱ་ཙེ་ཏ་སཾ། །དུབྷྱཱ་རཱ་རི་ནི་ཀཱ་ར་ཀོ་པ་ད་ཧ་རབ་ཏུ་དག་བྱེད་བཟོད་པའི་ཆུ་བོ་སེམས་ལ་རྟག་གནས་པ། །དེ་དག་ཟློག་དཀའ་དགྲ་ཡི་གནོད་ལ་ཁྲོ་བའི་ན་ཛཱ་ལཱ་བ་ལཱི་བི་བླ་བ་བློ་ཥོདྷཱ་ཏ་པྲ་ཐུ་བྱ་ཐཱ་བ་རཾ་ཙི་ཏྭཾ་ནཱ་ཡཱནྟི་མེ་འབར་འགྲེང་བས་རྣམ་བསླད་པས། །བསྲེག་པར་རབ་བསྐྱེད་རྒྱ་ཆེར་གདུང་བས་ཡོངས་འདྲིས་—།
