སཾ་པཱ་དྱ་ཏ་ཏཱ་སྱཔྲ་ར་སཔ་རཱ། །སཾ་བྷོ་ག་སུ་བྷ་གསདྲང་ཡོ་དཀཱི་ལྷ་མོ་གཞན། །སུམ་ཅུ་རྩ་གཉིས་ལོངས་སྤྱོད་ཀྱིས། །སྐལ་བཟང་གིས་ནི་ཡང་དག་ཐོབ། །དེ་བཞིན་ལྷོ་ཎ་བརྨ་ནཱ། །ས་དྲ་ཛརྣ་ག་རཾ་པྲ་པ་དུརྒ་པྲ་ཡ་མ་ཡོ་མ་མཾ། །ཡི་ལམ་ནས་སོང་། །དེ་ནི་སོང་བས་མི་བསྲུན་པའི། །གྲོགས་བཞིན་ཉོན་མོངས་ཚོགས་ཀྱི་གནས། །པ་རུ་ཎཾ་ཁ་ལ་པོ་ཧཱརྡ་མི་བཱ་ཡཱ་པ་ཙ་ཡཱ་ཐ་ཡཾ། །ཏ་སྱ་ཏ་ཏྲ་པྲ་བིཥཱ་ལྕགས་ཀྱི་རང་བཞིན་རྩུབ་པོ་ཡི། །གྲོང་ཁྱེར་ངན་འགྲོ་མང་པོ་ཐོབ། །དེར་དེ་ནང་དུ་ཞུགས་པ་ན། །སྱ་ཀྲ་མཱདྷཱ་ར་ཎི་བྷཱུ་མི་ཥཱ། །ཀེ་ནཱ་པི་པི་ཧི་ཏཱ་ནི་བ་ཡ་ཡུནི་སནྟི་བནྟ་རིམ་པས་ས་ལ་སྒོ་རྣམས་ནི། །སུ་ཞིག་དག་གིས་བསྒྲིབས་པ་བཞིན། །མཚམས་མེད་བསྡམས་པ་ཉིད་དུ་ནཾ༑། ཏ་དན་རེ་མ་ཧཱ་ཀཱ་ཡཾ་སོ་པ་བྱཧྥུ་རུ་ཥ་སྦི་ཏཾ། །ས་ཀ་ལ་ཀླ་ཤ་དོ་ཥཱ་གྱུར། །དེ་ཡི་ནང་ན་སྐྱེས་བུ་ནི། །ལུས་ཆེན་གནས་པ་དེ་ཡིས་མཐོང་། །མཐའ་དག་ཉོན་མོངས་སྐྱོན་ཎྞཾ་བིསྟིཥྛ་མི་བ་བྷཱ་ཛ་ན། །ཛཱ་ལཱ་བི་ལ་ཡི་ཏཾ་ཙཀྲྀ་ཏ་ས་མཱུཙྪི་པ་རི་རྣམས་ཀྱི། །སྣོད་ནི་ཤིན་ཏུ་ཆེ་བ་བཞིན། །བཅིངས་ཤིང་རྒྱུན་དུ་ཁྲག་རྒྱུན་འབབ། །དེ་ཡི་སྤྱི་བོར་བླ་མཏ། །ཏཱིཁཾ་ད་དན་བདྷ་སྱ་སཏྟཾ་རཏྟཾ་སྲ་བཱ་ཧི་ཎ། །སཱ་ཤི་འཁོར་ལོ་ནི། །རྣོན་པོ་འབར་བས་དཀྲིགས་པ་དག། ཡོངས་སུ་འཁོར་བ་མཐོང་བར་གྱུར། །རང་གི་རཔྲ་སུ་ཏིཔཱུ་ཡ་ཤོ་ཎི་ཏིཇི་ཧི་ཏཱ་ས་ནཾ། །ཏཾ་དྲྀབྷྱཱ་ཏཱི་བྲ་ཎིཟེ་དཱ་མགོ་ལས་ཟགས་པ་ཡི། །རྣག་ཁྲག་གིས་ནི་ཟས་བསྒྲུབས་པ། །དེ་མཐོང་དྲག་པོའི་ཡིད་བྱུང་ནས། །ད་བ་དམཻ་ཏྲ་ཀ་ནྱ་ཀ། །ཀནྟཾ་ཀཱ་ར་བྱ་ཐཱ་ཀཱནྟཀ་སྱེ་དཾ་ཀརྨ་ཎཕ་ལཾ། །མཛའ་བོའི་བུ་མོས་རབ་སྨྲས་པ། །གདུང་བ་དྲག་པོས་ནོན་ཁྱོད་སུ། །འདི་ནི་ལས་དག་ཅི་ཡི་འབྲས། །པྲཋསྟེ་ནེ་ཏི་ཀ་རུ་ཙྪཱཏ་པུ་རུ་ཥསྟ་མ་བྷཱ་ཥ་ཏ། །མཱ་ཏུཀྲྀ་ཏཱ་པ་ཀཱ་རོ་ཧཾ་ཞེས་དེས་སྙིང་རྗེས་དྲིས་པ་ན། །སྐྱེས་བུས་དེ་ལ་རབ་སྨྲས་པ། །བདག་ནི་ཁྱིམ་ལ་གནོད་བྱེད་དོ། །ཏ་སྐྱེ་དཾ་ཀརྨ་ཎཕ་ལཾ། ་། ཨེ་ཏ་དཱ་ཀཎྞཾ་སོ་པ་བཱིཏྟཱ་པ་སམྦད་ཤཾ་ཀི་ཏ། །འདི་ནི་ལས་དེ་ཉིད་ཀྱི་འབྲས། །དེ་དག་ཐོས་ནས་དེ་ཡང་ནི། །སྡིག་པ་དྲན་ནས་དོགས་པར་གྱུར། །དེ་།
