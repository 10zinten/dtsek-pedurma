ནུ་ཥི། །ས་ཀ་དཱ་ཙིཛྫ་ལཱ་ནཱ་ཏི་ཡུ་ཐཱ་ནཱ་ཐས་ཧཱ་ནུ་གི། །ཏ་ཊི་རཿསྐད་འབྱུང་། །ནམ་ཞིག་ཆུ་ནི་འཚོལ་བྱེད་ཅིང་། །ཁྱུ་ཡི་མགོན་དེ་རྗེས་འབྲང་བཅས། །ཝུ་བའི་ཕྲེང་བེ་ན་མཱ་ལི་ནྱཔུ་ལི་ནཾ་ས་མུ་པཱ་ཡ་ཡོ། །ཏ་ཏཱ་པ་ཤུཏ་པྲ་བཱ་ཧ་ཎ་ཀྲི་ཡ་ལྡན་ཆུ་ཀླུང་གི། །འཇུག་ངོགས་དག་ཏུ་སོང་བར་གྱུར། །དེར་ནི་དགྲ་རྣམས་ཀྱིས་བཅིངས་ནས། །ཆུར་མཱ་ཎ་མ་ཧཱི་ཡ་སཱ། །ས་ཤ་ཏུ་བྷིནྡྷ་ལེ་བདྷ་ཀླུབྷྱཱ་མ་ཀྲནི་ནཾ་ན་རཾ། །འཕངས་ཆོ་ངེ་འདེབས་པའི་མི། །ཤིན་ཏུ་ཆེ་བའི་ཆུ་རྒྱུན་གྱིས། །ཁྱེར་ནས་འགྲོ་བ་དེ་ཡིས་མཐོང་། །ཏ་སྱ་ཏ་ར་ཏ་རཾ་ཐུ་ཏཱ་དཱིགྷམཱ་ཀྲནྡ་ནི་སྭཱ་ནཾ། །ཧ་རི་ཎི་དུ་དྲུ་བུ༔སཧྥེ་གཱི་དེ་ཡི་ཆོ་ངེའི་སྒྲ་དག་ནི། །ཤིན་ཏུ་ཆེ་ཞིང་རིང་བ་ཐོས། །རི་དྭགས་ཐམས་ཅད་མགྲིན་པ་ནི། །འཁྱོག་བཱ་བ་ལ་ན་བི་བྷྱཱ་མི། །སུ་བཎྞ་པཱནཱསྟུ་ཏ་དཱ་ཀྲྀ་པཱ་པཱ་ཤ་བ་ཤཱི་ཀྲྀ་ཏ། །པའི་རྣམ་འཕྲུལ་ལྡན་པས་བྲོས། །གསེར་གྱི་ལོགས་ནི་དེ་ཡི་ཚེ། །བརྩེ་བའི་ཞགས་པས་དབང་དུ་བྱས། །ཏ་ཏྲི་བ་ནིཕྱི་ལསྟ་སྦོ་མརྨ་བིདྷ་ཨི་བེ་ཥཱ་ཎཱ། །ཏ་དུདྷ་ར་ཎ་སནྣདྷཾ་མདའ་ཡིས་གནད་དུ་ཕུག་པ་བཞིན། །དེ་ཉིད་དུ་ནི་གཡོ་མེད་གནས། །དེ་ནི་གདོན་པར་ཆས་པ་དག། དཱིགྷ་པྲྀཏྟི་བི་ལོ་ཀྱ་ཏྭཾ། །ཀཱ་ཀཔྲ་བཱ་ཙ་ན་ས་ཁེ་ཡུཀྟཉཡནྟེ་ས་མུདྨ་མ། །བྱ་རོག་རིང་དུ་མཐོ་བ་ནི། །མཐོང་ནས་དེ་ལ་སྨྲས་པ་གྲོགས། །ཁྱོད་ཀྱི་བརྩོན་པ་འདི་མི་རིགས། །པུབློ་པ་མཱ་བི་པཏྟཱ་ལེ་ཀྲྀ་ཏཱཞཱཀུ་ལི་ཤོ་པ་མཱ། །ཀྲྀ་ཏ་མེ་ཏེ་ན་མ་རྒྱནྟེ་སྭཱ་རྒུད་པའི་དུས་ན་མེ་ཏོག་མཚུངས། །དོན་བྱས་པ་ན་རྡོ་རྗེ་འདྲ། །མི་བསྲུན་རང་གི་ལུས་ཀྱི་གྲོགས། །ཀཱ་ཡ་སུ་ཧྲི་དཁ་ལཱ། །ཨི་ཏྱ་སོ་བཱརྨ་མཱ་ཎ་པི་ཀཱ་ཀེ་ན་ཀ་རུ་ཎཱ་ཀུ་ལ། །འདི་དག་བྱས་པ་མི་བསམ་མོ། །དེ་དག་བྱ་རོག་གིས་བརྗོད་ཀྱང་། །སྙིང་རྗེས་རྒྱས་ཤིང་དྲང་བ་དེ། །ཨ་བ་ཏིརཱ་ཤུ་པ་རི་ཏཾ་ས་ར་ལས་མ་ཏཱ་ར་ཡཏ། །བི་མུ་ཙུ་བནྡྷ་ནཱ་རྱ་སྱ་ས་ཆུ་བོ་ལ་ནི་མྱུར་བ་རུ། །ཞུགས་ནས་དེ་ནི་ཐར་པར་བྱས། །དོགས་མེད་དེ་ཡི་རུ་དག་གིས། །དེ་ཡི་ཤྲངྒཱ་བྷཱུ་མ་ཤཌི་ཏ། །ཏཾ་པཱ་ད་པ་ཏི་ཏཾ་དཱི་ན་མ་བ་དངྒནྟུ་མུདྨ་ཏཾ། །བཅིངས་པ་རྣམ་བཀྲོལ་ནས། །ཞན་ཅིང་རྐང་པར་གཏུགས་བྱས་ཏེ། །འགྲོ་བར་བརྩོན་པ་དེ་ལ་སྨྲས། །།
