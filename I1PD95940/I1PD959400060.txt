རཀྟ་ན་རཱཀྵ་སེ་ན་ཀྱི་ཏི་པྲ་བྷོ། །བྷཀྵ་མཱ་ཎེ་ཥཱ་མཱ་སེ་ཥཱ་རཀ་ནཾ་ཀྵི་ཏི་ར་ཀམྤ་འཐུངས་པའི། །སྲིན་པོས་ས་ཡི་བདག་པོ་ཡི། །ཤ་རྣམས་རབ་ཏུ་ཟ་བའི་ཚེ། །སྐད་ཅིག་ས་ནི་རབ་ཏུ་ཏ༑། ཏ་ཏཔདྨ་བ་ཏཱི་དེ་བཱི་པ་ཏི་ཛྭབྷཱ་ཏ་ཐཱ་ག་ཏཾ། །བི་ལ་པནྟི་ནི་པ་ཏི་འདར༑། དེ་ནས་ལྷ་མོ་པདྨ་ལྡན། །བདག་པོ་དེ་ལྟར་གྱུར་མཐོང་ནས། །སྨྲེ་སྔགས་འདོན་ཅིང་རབ་ཏུ་ཏཱ་མོ་ཧཱ་མཱུཙྪཱ་ཀུ་ལཱ་བྷ་བཏ། །མ་ནུ་ཛེནྡྲ་སྱ་དེ་བེནྣཱསྟ་ཏྲ་དྲྀཟླ་འགྱེལ༑། རྨོངས་ཤིང་བརྒྱལ་བས་དཀྲིས་པར་གྱུར། །མི་ཡི་དབང་པོའི་སྙིང་སྟོབས་ནི། །རབ་རྒྱས་དེ་སཏྭཱ་མཱུརྫི་ཏཾ། །རཀྵ་རཱུ་པཾ་པ་རི་ཏྱ་ཙུ་ཏ་མུ་བཱ་ཙ་ཀྲྀ་ཏཱཉྫ་ལི། །ཨ་ཧོ་མཐོང་ལྷ་དབང་གིས། །སྲིན་པོའི་གཟུགས་ནི་ཡོངས་བཏང་ནས། །ཐལ་མོ་སྦྱར་ནས་དེ་ལ་སྨྲས། །ཨེ་མ་ནུ་ཀརྨ་ཎཱ་རཱ་ཛ་ན་དུཀྐ་རེ་ཎ་ཏ་བཱ་མུ་ནཱ། །རོ་མཱ་ཙ་ཀཾ་ཙུ་ཀཱ་ཀཱིཎཀཱ་ཧོ་རྒྱལ་པོ་ཁྱོད་ཀྱི་ནི། །བྱ་བ་དཀའ་བའི་ལས་འདི་ཡིས། །བ་སྤུ་ལངས་པ་གཏམས་པའི་གོས། །ལུས་ཡཀ་སྱ་ན་ཛཱ་ཡ་ཏེ། །ཨ་ཧོ་པུ་ཅུ་མ་སཱ་མཱ་ནྱ་མ་ཧོ་ས་ཏྭ་མ་ནུ་ཏ་རཾ། །ཨ་ཧོ་ལ་ཅི་ཕྱིར་འབྱུང་མི་འགྱུར། །རྒྱལ་པོ་རྡུལ་བྲལ་ཁྱོད་ཀྱི་ནི། །ཨེ་མ་བསོད་ནམས་ཐུན་མོང་མིན། །ཨེ་མ་དྷིརླ་མ་མརཱ་དཾ་རཱ་ཛན་བི་ར་ཛ་ས་སྟ་བ། །དུཁི་ཏ༔པ་ར་དུཁེ་ཥཱ་སྙིང་སྟོབས་བླ་ན་མེད། །ཨེ་མ་བརྟན་པ་མཐའ་དང་བྲལ། །ཕ་རོལ་སྡུག་ལས་སྡུག་བསྔལ་ཞིང་། །རྙེད་ནིརྨོ་བྷ་དུརླ་བྷེ་ཥཱ་ཙ། །བི་པཀྵེ་ཏུ་གུ་མཱ་བནྟཔནྟ༔སུ་ཀྲྀ་ཏ་པེ་ཏ་བ། །དཀའ་ལ་ཡང་ཆགས་དང་བྲལ། །མི་མཐུན་ཕྱོགས་ལ་བཟོད་ལྡན་པ། །དམ་པ་རྣམས་ནི་ལེགས་བྱས་ལམ། །ས་མུརྨི་ཥ་ཏི་ཀོ་：་ཥ་ས་ཏོནཱ་ཧོ་མ་ཧཱ་ནཾ། །ཏྲེ་ལོཀྱི་ཀ་རུ་གང་གིས་འཇིག་རྟེན་གསུམ་པོ་ལ། །རྗེས་སུ་བརྩེར་གྱུར་བདག་ཉིད་ཆེ། །སྙིང་རྗེས་བརླན་པའི་སྙིང་ཎཱརྔ་ཎཾ་ཡེན་ཡཱ་ཏྱཱ་ནུ་ཀརྨྱ་ཏྭཾ། །ཨུཀྟི་ཏི་དི་བྱོ་ཥ་དྷི་བྷིནྟཾ་ཀྲྀ་ཏྭཱ་སཱསྦྲ་སྟོབས་ཀྱི། །སྤྲོ་བ་འདི་ནི་ཡང་དག་གསལ། །འདི་བརྗོད་ལྷ་ཡི་སྨན་རྣམས་ཀྱིས། །དེ་ལུས་རྣལ་དུ་བི་གྲ་ཧཾ། །པྲ་སཱ་རྱ་ལཛྫཱ་བ་ན་ཏཤ་ཀྲསཱ་ནི་ལ་ཡཾ་ཡ་ཡོ། །ཏ་ཏས་མཱསྟེ་གནས་པར་བྱས། །རབ་ཏུ་དང་ཞིང་ངོ་ཚས་དམན། །བརྒྱ་བྱིན་རང་གི་གནས་སུ་སོང་། །དེ་ནས་མཆོད་།
