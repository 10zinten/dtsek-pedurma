རྒྱུ་བེ་ཏཱ་ལཤ་རི་པུ་ཏྲེ་ཎ་ཀཱི་ལི་ཏ། ་། སྲེ་རི་ཏོ་མནྣཱ་བཱིརྨེ་ཎ་རོ་ལངས་ལི། །ཤཱ་རིའི་བུ་ཡི་ཕུར་བུས་བཏབ། །སྔགས་ཀྱི་བརྩོན་འགྲུས་ཀྱིས་བསྐུལ་ནས། །མིག་རཀཱཀྑཾ་ཧནྟུ་མུདྨ་ཡོ། །།བེ་ཏཱ་ལཱ་བྷི་ཧ་ཏསྨཱ་སན་ན་ཤྱནྟྲ་ན་མ་ད་ཛཱ་ར། །དམར་གསོད་དུ་རབ་ཏུ་སོང༌། །རོ་ལངས་ཀྱིས་བཅོམ་སྐྲག་པ་ལསེ། །ང་རྒྱལ་རྒྱགས་པའི་རི་ཞི་ནས། །ཤ་ར་ཎཾ་པཱ་ད་པ་ཏི་ཏ༔ཤཱ་རི་པུ་ཏྲ་ཛ་གཱ་མ་ས། །རཀྟག་ཏེ་ན་བྷཾ་གེ་ན་དེ་ནི་ཞབས་ལ་གཏུགས་བྱས་ཏེ། །ཤཱ་རིའི་བུ་ལ་སྐྱབས་སུ་སོང་། །།བཅོམ་པ་དེ་ཡི་མིག་དམར་དེ། །ཤ་ར་ཙྩཾ་ཤ་ར་ཎཾ་ག་ཏ། ་། པྲ་བྲ་ཚྭ་ཡཾ་བཱི་ཏ་རཱ་ག་ས་ཤུདྡྷཱ་བོ་དྷི་མཱསྟ་སྐྱབས་འོས་དག་ལ་སྐྱབས་སོང་ནས། །རབ་ཏུ་བྱུང་བ་འདོད་ཆགས་བྲལ། །།དག་པའི་བྱང་ཆུབ་ཐོབ་བཱན༑། ཏཱིརྒྱུ་སྣ་རྩེ་པ་རི་དྭེ་ཥ་ཀྲོ་དྷ་པཱ་ར་ཏི་ཏཱན་བ། །ཏ་ཏྲ་པར་གྱུར། །མུ་སྟེགས་གཞན་རྣམས་ཡོངས་སྡང་དང་། །ཁྲོ་བའི་ཕ་རོལ་བསྒྲལ་འདོད་པའི། །དགེ་སློང་ཀརྨཀ་ར་བྱཱ་ཛཱ་ཏསྦྲུབྷིཀྵ་བ་དྷོདྷ་ཏཱ། །ཏེ་ཐ་དྷརྨ་བྲུ་ཧཀཱ་ལེ་ཤཱ་དག་ལ་གནོད་བརྩོན་ཞིང་། །དེ་ན་ལས་བྱེད་ཟོལ་གྱིས་གནས། །དེ་ནས་དེ་དག་ཤཱ་རིའི་བུས། །དུས་རི་པུ་ཏྲེ་ཎ་ལཀྵི་ཏཱ། །ཏཛྫིཊཱི་བཱ་ཏ་མཱ་ཏྲེ་ཎ་བ་བྷུ་པུརྨི་ཏྲ་མཱ་ན་མཿ། །སུ་དེ་ལས་འཁུ་ཤེས་ནས། །ཁྱེདེ་ཡི་སྤྱན་ཟུར་ཙམ་ཉིད་ཀྱིས། །ཡིད་ནི་བྱམས་དང་ལྡན་པར་གྱུར། །ཨཱ་ཤ་ཡཱ་ནུ་ཤ་ཡ་དྷཱ་ཏུཾ་པྲ་ཀྲྀ་ཏི་ཙ་བི་ཙཱ་རྱ་ས། །དྷརྨ་དེ་ཤ་ན་ཡཱ་ཏེ་ཊཱ་དི་ཁམས་དང་བསམ་པ་བག་ལ་ཉལ། །རང་བཞིན་དག་ཀྱང་སྤྱད་ནས་དེ། །ཆོས་ནི་བསྟན་པས་དེ་དག་དེ་ཤཱ་ནུཏྟ་རཾ་ད་ཧཱུཾ། །ཨ་ཐ་ཏ་སྱ་བི་ཧཱ་ར་སྱ་ནིཤྲཱིསྨྲ་རྣམས། །བླ་མེད་གནས་སྐབས་དག་ལ་བཀོད། །དེ་ནས་གཙུག་ལག་ཁང་དེའི་ལས། །བརྩམས་པ་རཾ་བྷ་ཀརྨ་ཎི། །།ཨ་ནཱ་ཐ་པྲཎྜ་དཾ་པྲ་ཧ་ཤཱ་རི་པུ་ཏྲཿསྨི་ཏཱ་ན་ན། །བགེགས་དང་བྲལ་བའི་ཚེ། །འཛུམ་པའི་བཞིན་ལྡན་ཤཱ་རིའི་བུས། །མགོན་མེད་ཟས་སྦྱིན་ལ་སྨྲས་པ། །བི་ཧཱ་ར་སཱུ་ཏྲ་པཱ་ཏ་སྱ་ཏུ་ལ་ཨེ་བ་ཀླུ་ཎེ་མ་ཧཱན། །ཧེ་མོ་བི་ཧཱ་རཿསཾ་བྲྀཏྟསྐུ་གཙུག་ལག་ཁང་གི་ཐིག་བཏབ་པའི། །དུས་མཚུངས་ཉིད་དུ་དགའ་ལྡན་གྱི། །ལྷ་ཡི་གནས་སུ་གསེར་གྱི་ནི། །།
