ཡཱཏྟཾ་བྱ་ཛི་ཛྷ་པཏ། །བྷ་ག་བཏ་པྲ་ཎི་དྷཱ་ནེ་ན་ཏཏྟཏྤ་ཏྱཱ་ནུ་བྷཱ་བ་ཏ། །པ་ཡིས་ནི་དེར་གསོལ་པ། །བསོད་ནམས་དེ་དེའི་མཐུ་དག་ལས། །བཅོམ་ལྡན་འདས་ཀྱི་སྨོན་ལམ་གྱིས། །ན་ག་སྟེ་ནུཏྟ་རཱ་ས་མྱཀཾ་བོ་དྷིགྷ་ག་བ་ཏཆི་ཏཱ། །བྷ་བཏ་པྲ་སཱ་ད་པྲ་ཎ་ཡ་བླ་མེད་ཡང་དག་རྫོགས་བྱང་ཆུབ། །ཁྱོད་ཀྱིས་སུ་ལ་སྟེར་མི་བགྱིད། །ཁྱོད་ཀྱི་བཀའ་དྲིན་བརྩེ་བ་ལས། །པྲཔ་མིཙྪཱ་མི་ཏཱ་མ་ཧཾ། ་་། ནི་ཀལྤ་ཕ་ལཱ་བསྤྱི་སེ་བྱནྟེ་ཀལྤ་པཱ་དེ་ནི་ཐོབ་པར་བདག་འཚལ་ལོ། །རྣམ་པར་མི་རྟོག་འབྲས་ཐོབ་སླད། །དཔག་བསམ་རྐང་འཐུང་བསྟེན་དཱ་པ། །ཨི་ཏི་རཱ་ཛཱ་པ་ཙ་ཐུ་ཏྭཱ་བྷ་ག་བཱན་སཱ་ཡ་མ་བྷ་ཥཊ། །དུརྨ་བྷཱ་པ་ལགས། །དེ་སྐད་རྒྱལ་པོའི་ཚིག་ཐོས་ནས། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། །བླ་མེད་ཡང་ནུཏྟ་རཱ་སམྱཀ་སཾ་བོ་དྷིཔྲི་ཐི་བཱི་པ་ཏྲེ། །སུབྷཱ་མྲྀ་ཎཱ་ལ་ཏནུ་བློ་གི་རི་བློ་དག་རྫོགས་པ་ཡི། །བྱང་ཆུབ་ས་བདག་རྙེད་པར་དཀའ། །པདྨའི་སྐུད་པ་བས་ཀྱང་ཕྲ། །རི་བོ་ལས་པི་ག་རི་མ་པཱི། །ས་མུ་དྲེ་བློ་པི་གཾ་བྷི་རཱ་སཱ་སུ་ཁེ་ན་ན་ལབྷུ་ཏེ། །ན་ད་ནིརྦ་ཀྱང་ཤིན་ཏུ་བརླིང་། །རྒྱ་མཚོ་བས་ཀྱང་ཟབ་པ་སྟེ༄༅། །བདེ་བླག་ཏུ་ནི་འཐོབ་མ་ཡིན། །བདག་གིས་སྐྱེ་ཧུ་བྷིརླབྷ་མ་ཡི་བཱ་ནྱ་ཏུ་ཛརྨ་སུ། །ཙིཏྟ་པྲ་སཱ་ད་བི་ཥ་དཾ་ཛྙཱ་ནཾ་ཏཏྟཱ་ར་ཎཾ་བ་གཞན་རྣམས་སུ། །སྦྱིན་པ་མང་པོས་ཐོབ་མ་གྱུར། །སེམས་ནི་རབ་དང་དྲི་མེད་ཀྱི། །ཡེ་ཤེས་དེ་ཡི་རྒྱུ་ཛ་གུ། །ཙ་ཏུཏྲི་པ་དྷི་པ་ཏྲེ་ན་མ་ཡཱ་མནྡཱ་ཏྲ་ཛརྨ་ནི། །ཙི་རཾ་དཱ་ན་ཕ་ལཾ་རུ་གསུངས། །ང་ལས་ནུ་ཡི་སྐྱེ་བ་ལ། །བདག་གི་གླིང་བཞིའི་བདག་པོ་ཡིས། །ཡུན་རིང་སྦྱིན་པའི་འབྲས་བྷུཏྟཾ་བོ་དྷིན་པི་ག་ཏཱ་ཏུ་པཱ། །དཱ་ནེ་ན་ཙཀྲ་བརྟི་ཤྲསཱ་སུ་དན་ཛརྨ་བུ་སྤྱད། །བྱང་ཆུབ་དེ་ནི་ཐོབ་མ་གྱུར། །བལྟ་སྡུག་སྐྱེ་བར་སྦྱིན་པ་ཡིས། །འཁོར་ལོས་སྒྱུར་བའི་དཔལ་ནི༑། བྷུཀྟ་མ་ཡཱ་མ་ཧ་ཏཱི་བོ་དྷིན་པི་གཱ་ཏཱ་ཏུ་པཱ། །པུ་རཱ་ད་ཏྭཱ་ག་ཛཱ་སྐྱོཊོ་དེ་ནི། །ཤིན་ཏུ་ཆེ་བ་བདག་གིས་སྤྱད། །བྱང་ཆུབ་དེ་ནི་ཐོབ་མ་གྱུར། །གཉིས་སྐྱེས་སྐྱེ་བ་བེ་ལ་མར། །བེ་ལཱ་མ་དཱི་ཛ་ཛརྨ་ནི། །མ་ཡཱ་པཱཔསྨྲ་ཧའུ་ཙྩཾ་བོ་དྷིནྡྲ་པི་ག་ཏཱ་ཏུ་པཱ། །།གླང་པོ་བརྒྱད་ནི་སྔོན་བྱིན་པས། །།བསོད་ནམས་ཆེན་པོ་བདག་གིས་ཐོབ། །བྱང་ཆུབ་དེ་ནི་ཐོབ་མ་གྱུར། །།
