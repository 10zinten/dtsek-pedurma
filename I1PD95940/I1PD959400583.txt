ཞཱནྟ་མཱ་ཡནྟཾ་མ་ཧཱི་བ་ཏི། །བི་ལོཉྩརྨ་བི་བྷ་བ་མེ་ནེ་པུ་ཎྱ་བ་ཏྭཾ་བ་འོང་བ་དེ་ནི་ས་བདག་གིས། །མཐོང་ནས་ངོ་མཚར་འབྱོར་པ་ཅན། །བསོད་ནམས་ལྡན་པའི་མཆོག་ཏུ་རཾ༑། ཀོ་ཏུ་ཀཱ་ད་ཐ་བྷུ་ས་ལནྟཾ་ག་ཛཾ་ཧེ་མ་བི་གྲ་ཧཾ། །ཨཱ་དུ་རོ་ཧ་མ་ཧོཏྟཱ་ཤེས། །དེ་ནས་རྒྱལ་པོས་དགེ་མཚན་ལས། །གླང་པོ་གསེར་ལུས་ཅན་དེ་ལ། །སྤྲོ་བ་ཆེན་པོས་ལྷུན་པོ་སུ་མེ་རུ་མི་བ་བཛྲ་བྷིཏ། ་། ཨཱ་ཏུ་རྣོ་པྲི་ཐི་བཱི་པཱ་ལེ་ན་ཙ་ཙཱ་ལ་ས་ཀུཉྫ་ར། །ལ། :། རྡོ་རྗེ་འཛིན་པ་བཞིན་དུ་ཞོན། །ས་སྐྱོང་གིས་ནི་ཞོན་པའི་ཚེ། །གླང་པོ་དེ་ནི་མི་གཡོ་ཞིང༌། །པྲ་ས་སཏྠ་ཀུ་མཱ་རེ་ཎ་པུ་ནཝཱ་ལཾ་ཀྲྀ་ཏཱ་ས་ན། །ཏསྨྲ་ཏྭཱ་ནཾ་པ་ཏི་རེ་སྟན་ནི་གཞོན་ནུས་བརྒྱན་པའི་ཚེ། །སླར་ཡང་རབ་ཏུ་གཡོ་བར་གྱུར། །མི་བདག་གིས་དེ་ལྟར་ཤེས་ཐཾ་ཏཏྲ་བྷཱ་བེ་ན་བིརྨི་ཏ། །དྷ་:མི་ཏི་བ་དན་རྩྭ་ད་དོ་ཏསྨི་སྨ་ཤིང་། །དེ་ཡི་མཐུ་ཡིས་ཡ་མཚན་པས། །བདག་ནི་སྐལ་བཟང་ཞེས་བརྗོད་ཅིང་། །བུ་མོ་འདོད་དཔལ་ར་ཤི་ཡཾ། །ཨ་བྱཱརྱ་ཀ་རྩྭ་རནྡྷེ་ན་ཚིགས་པ་ཏིཔུ་རུ་ཊོཏྟ་མཾ། །ཧནྟོཏྠ་བ་ས་མུདྡྷ་དེ་ལ་བྱིན། །རིན་ཆེན་བུ་མོས་སྐྱེས་བུ་མཆོག། མ་ནོར་མཆོད་ནས་མི་ཡི་བདག། སྤྲོ་དགའ་ཡང་དག་ཏསུ་དྷཱ་སིན་དྷུ་རི་བཱ་བ་བྷོ། །ཏ་ཏཀུ་མཱ་ར་ད་ཡི་ཏཱ་མཱ་དཱ་ཡ་རྒྱས་གྱུར་པ། །བདུད་རྩིའི་རྒྱ་མཚོ་བཞིན་དུ་མཛེས། །དེ་ནས་གཞོན་ནུས་མཛེས་མ་ནི། །།བླངས་ནས་སྭ་ཧུཾ་ཧཾ་ག་ཏེ། ་། ས་ཕ་ལོ་ཧྥུ་ད་ནཾ་ག་སྱ་ཀཱརྨུ་ཀ་ཀརྞ་ཎ་ཐ་མ། །ན་བེ་རང་གི་ཁྱིམ་སོང་སྟེ། །ལུས་མེད་གཞུ་ནི་དྲངས་པ་ཡིས། །ངལ་བ་འབྲས་བུ་ཡོད་པར་གྱུར། །ལང་ཚོ་བ་ཡ་སཾ་བྷོ་གཱརྟེ་ན་བ་ཀཱནྟཱ་ས་མཱ་ག་མོ། །ཏ་སཱ་བྷཱུདྲི་བྷ་བོ་དཱ་རགསར་པ་ལོངས་སྤྱོད་འོས། །མཛེས་མ་གསར་པ་དང་འགྲོགས་ཏེ། །འབྱོར་པ་རྒྱ་ཆེས་རྟག་ཏུ་དེས། ས་དཱ་ན་བ་ན་བཏྠོ་བ། །ཏ་ཏཿཀ་དཱ་ཙིདྡྷཱ་བཱ་ལཱཀྲྀ་ཏ་ཀྲྀ་ཏྲཿཔྲ་སེ་ན་ཛིཏ། །གསར་པ་གསར་པ་དགའ་བར་འགྱུར། །དེ་ནས་ནམ་ཞིག་ས་སྐྱོང་ནི། །བྱ་བ་བྱས་པ་གསལ་རྒྱལ་གྱིས། །པུ་རྱ་པྲ་བྷཱ་ཁཾ་ཛཱ་མཱ་ཏུཀ་ལ་ཡཎྞ་མ་ཙིནྟ་ཡཏ། །ཨ་ཧོ་དི་བྱཔྲ་བྷཱ་བོ་སོ་བསོད་ནམས་མཐུ་ལྡན་མག་པ་དག། ཡིད་ལ་གཅགས་ཤིང་རབ་བསམས་པ། །ཀྱེ་མ་ལྷ་ཡི་མཐུ་འདི་ནི། །—།
