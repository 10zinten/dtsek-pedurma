སྱ་མེ་བྲཧྨནྣརླི་ནཱ་མརྦ་སིདྡྷཱ་ཡེ། །ནི་ཀལྤ་པ་ཀ་ར་ཎཾ་བྱ་ཡཾ་ཡཱ་ཏི་ལེགས་འཚོ་དག། སློང་བ་རྣམས་ཀྱི་དོན་གྲུབ་སླད། །རྣམ་པར་མི་རྟོག་ཡོ་བྱད་དུ། །ཟད་འགྱུར་བྲམ་ཟེ་སུ་ཛཱི་བི་ཏཾ། །ཀ་དཱ་པཱ་ཎཱཔ་རཱརྗེ་མེ་པྲ་ཡཱནྟི་མ་ནོ་ར་ཐ། །ཀི་མེ་ཏཱ་བདག་ལེགས་སོ། །བདག་སྲོག་ཕ་རོལ་དོན་དུ་ནི། །ནམ་ཞིག་འགྱུར་ཞིང་རེ་བ་ཡོད། །གལ་ཏེ་ཁྱོད་ནི་ན་པུ་ཙྪཱ་ནི་པཱརྒྱུནྟེ་ཏེ་ཡ་དི་ཏཱ་ཡཱ། །ཨ་ཧོ་པ་ཀ་ར་ཎ་སིདྡྷི་ཀྱིས་དེ་བསླངས་པ། །འདི་ནི་བསོད་ནམས་ཅིས་མ་ཡིན། །འཇིག་རྟེན་འདིར་ནི་སློང་རྣམས་ལ། །བྷཱཤྩ་མེ་གྲོ་རྒྱ་ཏྭཾ་ཤི་ར། །ཏཏ་ཏ་དེ་བ་སྦི་ར་ལོ་ཀེ་ཡད་ཡ་དཀྲི་པ་མཆི་གང་གང་བྱིན་པ་དེ་དེ་བརྟན། །གྲུབ་པའི་ཡོ་བྱད་བདག་གི་མགོ། །བསྔགས་པར་འོས་པ་གཟུང་བར་་་་ཏཾ། །ཨི་ཏུཀྟ་ཧརྟ་ཡུལྟེ་ན་བྷ་བྷུ་ཛཱ་ས་ཏཱ་ཤཱ་ལི་ནཱ། །ཏ་མཱུ་ཙ་ཏསྨྲ་མཛོད། །ས་སྤྱོད་སྙིང་སྟོབས་ལེགས་ལྡན་པ། །རབ་དགའ་ལྡན་པས་འདི་བརྗོད་ཚེ། །བློན་པོ་ཆེན་པོ་ཧཱ་མཱ་ཏྱཾ་མ་ཧཱ་ཙནྡྲ་མ་ཧཱི་དྷ་རོ། །ནི་ཛ་ཛཱི་བི་ཏ་རཀྵེ་བ་དྷརྨསྟེ་པྲ་ཐ་མཔྲ་ཟླ་ཆེན་དང་། །ས་འཛིན་དག་གིས་དེ་ལ་སྨྲས། །རྗེ་བོ་ཁྱོད་ཀྱི་ཆོས་དང་པོ། །རང་གི་སྲོག་ནི་བསྲུང་བ་བྷོ༑། ཏཱ་ཡི་ཛཱི་བ་ཏི་ཛཱི་བནྟི་སརྦ་ཛ་ག་ཏེ་ཛན་བ། །ནཱ་དཱ་ཏུ་མན་ཉིད། །འགྲོ་བར་ཁྱོད་ནི་འཚོ་བ་ན། །སྲོག་ཆགས་ཐམས་ཅད་འཚོ་བར་འགྱུར། །གང་ཕྱིར་ཁྱོད་ལུས་སི་ཤི་རསརྦ་དྷཱ་ར་ཧི་ཏེ་བ་སུ། །དཱི་ཡ་ཏཾ་བྲཧྨ་ཎཱ་ཡཱསྨེ་ཧེ་མ་རཏྭཱ་མ་ཀུན་གྱི་གཞི། །མགོ་བོ་སྦྱིན་པར་མི་འོས་སོ། །རིན་ཆེན་གསེར་གྱི་རང་བཞིན་མགོ། །བྲམ་ཟེ་འདི་ལ་ཡཾ་ཤི་རཿ། །སནྡཱརྗེ་རརྦ་སཱརྦ་ནཾ་པཱུརྨནྟེ་ཡིརྨ་ནོ་ར་ཐཱ། །།ཏེ་ཊཱ་སཾ་སྦྱིན་པར་མཛོད། །སློང་བའི་དོན་མཐུན་རེ་བ་དག། གང་གི་ཡོདོན་ཀུན་གྱིས་རྫོགས་པ། །དེ་དག་ཡང་རཀྵ་ཎེ་ནི་བ་སཐཾ་བྷ་བ་ཏི་རཀྟི་ཏཾ། །སཾ་ཀ་ཡཾ་དྲི་ཛ་སྭཱ་སྱ་དག་བསྲུངས་པ་ཡིས། །ཐམས་ཅད་བསྲུངས་པ་ཉིད་དུ་འགྱུར། །སྡིག་པའི་སེམས་ལྡན་གཉིས་སྐྱེས་འདིའི། །ཀཱ་རཿཀ་ལུ་ཥ་ཙེ་ཏ་ས། །མཱུ་ལོཙྪེ་དོ་པ་ཛཱི་བྱོ་ཧི་ན་ཀལྤ་ཏ་རུ་རརྨི་རྣམ་རྟོག་འདི་ནི་འཇིགས་སུ་རུང་། །གང་ཕྱིར་སློང་བའི་དཔག་བསམ་ཤིང་། །རྩ་བཅད་ཉེ་བར་འཚོ་བྱ་—།
