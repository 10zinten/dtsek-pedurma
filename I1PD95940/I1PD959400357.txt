སྱ་པྲ་བྷཱ་བ་ཏ། །གོ་ར་བཱདྷཱ་ཏ་སཾ་པཱYཔྲ་ཎ་ཏཾ་པྲ་ཎ་ནཱ་མ་ཏི། །པ་དེས་མཐོང་ནས། །གུས་པས་ཉེརྨད་བྱུང་ཡང་དག་ཐོབ། །རབ་ཏུ་བཏུད་ནས་དེར་ཕྱག་འཚལ། །ཏ་ཏས་ས་ཧི་ཏབི་ཏཱ་ན་ག་ར་གཾ་ཏུ་མུདྨ་ཏ། །ཨ་པ་ཀླུ་ར་པརླནྟེ་དེ་ནས་ཡབ་དང་ལྷན་ཅིག་དེ། །།གྲོང་ཁྱེར་འགྲོ་བར་བརྩོན་པ་དེ། །གྲོང་ཁྱེར་མཐའ་ཡི་དུར་ཁྲོད་ནི། །ཤཱ་ནི་ཤ་བ་སཾ་ཀུ་ལཾ། །ས་དྲིནྡྷ་ཀུ་ཎ་པཱ་ཀཱིཥྛ་མ་ཤི་བོ་ཤི་བ་ཀཱ་ན་ནཾ། །རོ་ཡིས་ཡོངས་སུ་གང་བ་མཐོང་། །ཞི་བའི་ནགས་ནི་མི་ཤིས་པ། །རོ་ཡིས་ཁེངས་མཐོང་ཆགས་བྲལ་ཅན། །སོདྲེ་གསཱ་ར་ཐི་པྲ་ཧསྦ་གི་ཏ་སྤྱནྡ་ནཾ་ཀྵ་ཎི། །སཱ་ར་ཐེ་པ་བྱ་ཛཾ་ཏུ་ཚྭ་དེ་ཡི་®སྐད་ཅིག་ཤིང་རྟ་ནི། །བཟུང་ནས་ཁ་ལོ་སྒྱུར་ལ་སྨྲས། །ཁ་ལོ་པ་ལྟོས་སྲོག་ཆགས་ཀྱིས་། །ཀཱ་ཡཱ་པཱ་ཡ་མ་ཡིརྡ་ཧཱུཾ། །དྲྀཥཱ་ད་མ་པི་རཱ་གཱ༴་མ་ནཱ་མོ་ཧ་གནོད་པའི་རང་བཞིན་ལུས་ཀྱི་སྐབས། །རྨོངས་པས་བག་མེད་རྣམས་ཀྱི་ཡིད། །འདི་ལྟར་མཐོང་ཡང་པྲ་མཱ་དཱི་ནཾ། ་། པ་ར་སི་དན་ནཱ་ཊཾ་་ནེ་ཏྲ་མཱ་སཱ་ཏྱ་སཱ་ར་རཾ། །ཨ་ཆགས་པས་བརླན། །གཞན་གྱི་བུད་མེད་བལྟས་མ་ངོམས། །མིག་ནི་གུས་དང་བཅས་པར་མྱང་། །འདི་སྱ་སཏྱ་བ་ཏཱི་ཛི་ཧཱ་པ་བྱ་ཀཱ་ཀོ་ན་ཀྲྀ་ཥྱ་ཏེ། །ཨ་སྱསྟ་ན་མུ་ཁ་ནྱ་སྟ་ན་ཡི་བདེན་ལྡན་མིན་པའི་ལྕེ། །བྱ་རོག་གིས་ནི་གཅོད་ལ་ལྟོས། །འདི་ཡི་ནུ་མའི་སྟེང་དུ་ནི། །སེན་ཁོམྨེ་ཁ་སུ་ཁསྦི་ཏི། །ཁཎྜ་ཡ་ཏུ་དྷ་རཾ་གྲོ་དྷཀཱ་མཱི་བ་མ་ད་དམུགས་དགོད་ཉེཅིང་བདེ་བར་གནས། །འདོད་པས་རྒྱས་པའི་བྱ་རྒོད་ཀྱིས། །མཆུ་གཅོད་འདོད་ལྡན་ནི་བྷ་ར། །ཨེ་ཏེ་དྲི་ནི་ཥཀྟ་བཱ་ཡ་ས་ཤ་ཀྲྀ་ན་ནིཙྪི་བི་ནདག་གིས་བཞིན། །རྐང་འཐུང་འདི་དག་གཡོ་བའི་བྱ་རོག་ཆགས་པའི་ལྕི་ཁྱིབས་སྐྱུག་ཅིང་བརྒྱལ་བ་དང་། །པཱ་ད་པཱ་མུཙྪནི་བ་པི་པཱ་ཀ་པཱུ་ཡ་ཀུ་ན་པཱ་གྷཱ་ཎེ་ན་ནིཀཱ་ཎི་ཏཱ། །དྲྀབྷཱ་ཁཾ་གྲྭ་རྣམ་པར་གྱུར་ཅིང་རྣགས་པའི་རོ་ཡི་དྲི་ངས་ངེས་པར་སྒྲ་ངན་སྒྲོགས་པ་བཞིན། །ལན་ཅིག་མ་ཡིན་བྱ་རྒོད་བི་དཱརྱ་མཱ་ཎ་མ་ཤ་ཀྲྀཏྟཱཎྞརྔ་ཏནྟཾ་ཤ་ཐཾ་བཀྑཾ་བཱ་ཏ་བི་ལོz་པལླ་བ་ཀ་ཀྱིས་འཇོམས་རླན་ལྡན་རྒྱུ་མ་གཟགས་པའི་རོ་མཐོང་ནས། །རླུང་གིས་རྣམ་པར་བསྐྱོད་པའི་ཡལ་འདབ་།
