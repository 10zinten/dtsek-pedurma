བཱ་ན་སྱཤཱིཙྪཱ་བཧཱུྃ་ས་མ་གྲ་ཧཱིཏ། །ཏཱ་དཱ་སཱི་སྣ་མཱི་ཏི་ཤདྷཱ་པྲ་བ་ཡིསོ། །བཅོམ་ལྡན་འདས་ཀྱིས་ཡང་དག་བླངས། །དེ་ཡིས་འབངས་མོར་བདག་གྱུར་ཅེས། །དད་ཎི་དྷཱ་ནཱབི་ཏསྟ་ཡཱ། །ས་ཙཱི་བ་རེ་བྷ་ག་བ་ཏཔྲ་ཡ་ཡཱ་བེ་ཀ་བཎ་ཏྭཾ། །པའི་སྨོན་ལམ་གྱིས་ཕུལ་དེ། །བཅོམ་ལྡན་འདས་ཀྱི་ཆོས་གོས་ལ། །ཁ་དོག་གཅིག་པ་ཉིད་དུ་གྱུར། །སུ་དྷ་ན་སྩོཛྫཱ་ལཾ་དྲྀབྷྱཱ་པུཎྱ་སམྦྷ་ར་མདྡྷཱ་ཏཾ། །བྷིཀྵ་བྷིབྷ་ག་བཱན་པྲཊ་ནོར་བཟང་གི་ནི་བསོད་ནམས་ཚོགས། །རྨད་བྱུང་རབ་འབར་མཐོང་གྱུར་ནས། །དེ་ཡི་སྔོན་བྱུང་དགེ་སྟཤྩབློ་ད་ཡ་མ་བྱཱ་དྷཱཏ། །སུན་ནཱ་ཁྱོ་གྲོ་ཧ་པ་ཏི་བཱ་རཱ་སློང་གིས། །ཞུས་ཤིང་བཅོམ་ལྡན་གྱིས་ཀྱང་གསུངས། །﹙བཱ་རཱ་ཎ་སཱིར་ཁྱིམ་བདག་སྔོན། །གཟུགས་ཎ་སཱུ་མ་བྷཱུཏ་པུ་རཱ། །ནོ་དཱ་ར་ཀུཉྫ་ར་སྭཱ་བྷཱུ་ཏ་ཡ་སཱུ་དཱ་ན་པ་རི་ཀཱ་ཡ། །ཨ་ནཱ་བྱེད་ཅེས་ནི་བྱ་བར་གྱུར། །རྒྱ་ཆེན་གླང་པོ་གང་ཞིག་ནི། །སྦྱིན་པ་ཡོངས་སུ་ཟད་མ་གྱུར། །ལོ་ནི་བྲྀཤྲཱི་ཧ་ཏེ་ཀ་ལེ་ཏ་སྱ་རཱ་ད་ཤཱ་བཱཙྩཾ་ཀེ། ཨ་བཱ་རི་ཏ་མ་བྷཱུཙྪ་ཏྲ་བཅུ་གཉིས་ཆར་མེད་ཀྱིས། །ཉམས་པའི་དུས་སུ་དེ་ཡིས་ནི། །སྦྱིན་ཁང་མི་ཟློག་སློང་རྣམས་ལ། །མ་བིཙྪིནྡྷསྣ་མཟི་ནཾ། །ཏ་སྐུ་པདྨ་ཀ་ར་ནཱ་མ་ཀོཏྟཱ་ག་ར་པ་ཏིཀྲྀ་ཧེ། །ཟས་ནི་རྒྱུན་ཆད་མེད་པར་གྱུར། །དེ་ཡི་ཁྱིམ་དུ་མཛོད་ཁང་བདག། པདྨའི་འབྱུང་གནས་ཞེས་བྱ་བས། །དཱ་ན་སཱ་ཧཱ་ཡ་ཀཉྩ་ཀྲེ་ས་ཧཱ་ཡསྠཱ་ཧི་སིདྡྷཱ་ཡ། །པྲ་ཏེ་ཀ་བུདྡྷ་སཾ་གྷ་སྱ་སྦྱིན་པ་ཡི་ནི་གྲོགས་དག་བྱས། །གང་ཕྱིར་གྲོགས་དང་ལྡན་པས་འགྲུབཏེ། །དེ་ཡི་རང་རྒྱལ་ཚོགས་དག་གི། །བྷཏྟ་ཀཱ་ལ་ནི་བེ་ད་ཀ། །དྷརྨ་དྷུ་ཏཱ་བྷི་དྷསྟ་སྱ་སནྡྲ་དྷི་མཱ་ནུ་པ་སྦི་ཏ། །ཀརྨ་ཟས་ཀྱི་དུས་ནི་རིག་བྱེད་པ། །ལྷག་པའི་བློ་ལྡན་ཉེ་བར་གནས། །ཆོས་ཀྱི་ཕོ་ཉ་ཞེས་པར་གྱུར། །ནམ་བྷཥཱ་པ་ཏསྟ་སྱ་ཛ་ཏེ་ཀཱ་ལ་བྱ་ཏི་ཀྲ་མེ། །ཀ་དཱ་ཙིཏྟུ་ཀུ་རསྟེ་ཊཱ་ཀཱ་ལ་སཾ་ཛྙཱ་ཞིག་དེ་ནི་ལས་ཀྱིས་བྲེལ༄༅། །དུས་ནི་ཡོལ་བར་གྱུར་པའི་ཚེ། །དེ་དག་རྣམས་ཀྱི་མདུན་དུ་ནི། །ཁྱི་ཡིས་དུས་བྱ་དྷཱ་ཏ་པུ་རཿ། །སནྡྷ་ནོརླས་ཨེ་བཱ་ཧཾ་ཀོཊྛི་ཀོ་ནཱ་ཐ་པིཎྜ་ད། །སཾ་གྷཾ་སྱ་ཀྱི་བརྡ་དག་བྱས། །གཟུགས་བྱེད་གང་དེ་ང་ཉིད་དེ། །མཛོད་པ་མགོན་མེད་ཟས་སྦྱིན་ནོ། །ཚོགས་ཀྱི་ཆོས་།
