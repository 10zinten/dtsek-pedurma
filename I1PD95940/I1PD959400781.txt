རེཀ་ཐི་ཏཾ་གཱ་ཌྷ་ཙཱ་རི་བྷི། །དུདྷིཀྵ་ཁིརཱ་ཡ་དི་ཏེ་ཡཱ་ཏཱ་མ་མ་རི་པོgཔུ་རཾ། །གསང་ནས་རྒྱུ་བ་དག་གིས་སྨྲས། །གལ་ཏེ་བདག་གི་དགྲ་གྲོང་དུ། །དེ་དག་མུ་གེས་ཉེན་འགྲོ་ན། །ཏpཏྠ་ཏྲ་བྷ་ཊཾ་ཏྲེ་བ་པརཱ་ཡེཌྷ་བ་བི་སླ་བཱི། །ཨ་ཐ་བཱ་རཱ་ཛ་དོ་ཥ་ཎ་གནམ་གྱི་ཉེར་འཚེ་ཐམས་ཅད་དུ། །རེ་མོས་ཀྱིས་ནི་འབྱུང་བ་ཉིད། །ཡང་ན་རྒྱལ་པོའི་སྐྱོན་གྱིས་ནི། །ག་ཏསྟེ་སུ་ཁ་བཱིཙྪ་ཡཱ། །བེཥཱ་དཎྜ་ཀ་རོརྨུཀྟཀ་སྱ་རཱ་ཛྷཱཔུ་རེ་ཛ་ན། །བདེ་བ་འདོད་པས་དེ་དག་སོང༌། །རྒྱལ་པོ་སུ་ཡིས་གྲོང་ཁྱེར་གྱི། །སྐྱེ་བོ་ལས་དང་དཔྱ་ལས་གྲོལ། །པྲ་ཡཔ་རི་ཙི་ཏ་དྭེ་ཥཱཏྨ་དཱ་ན་བ་ན་བི་ཏི་ཎ། །དྷུ་རསྦྲ་ཨེ་བ་སཏྠ་སྱ་ཕལ་ཆེར་ཡོངས་སུ་འདྲིས་ལ་སྡང་། །རྟག་ཏུ་གསར་པ་གསར་པ་འཚོལ༄༅། །རིང་ན་གནས་པ་ཐམས་ཅད་སརྦ་བྷ་བ་ཏི་བསྨྲ་བྷ། །་། ཨསྨྲ་ད་བྱཱ་དྷི་ཀཿཀོ་སོ་ཏ་སྱ་བྷུ་མི་པ་ཏེརྒུ་ནི༑། ཀུན་གྱི་ཡིད་དུ་འོང་བ་ཡིན། །གང་གིས་གཞན་གྱི་སྐྱེ་རྒུ་ནི། །རྨོངས་པ་ཆུང་མ་བཞིན་དུ་ཎ༑། པ་རེ་ཊཱ་ཧ་ར་ཏེ་ཡེ་ན་མུདྡྷཱ་ཛཱ་ཡཱ་ཨི་བ་པྲ་ཛཱ། །ཨུ་པཱ་ཡཐིནྱ་འཕྲོག། ས་བདག་དེ་ལ་ཡོན་ཏན་ནི། །བདག་ལས་ལྷག་པ་ཅི་ཞིག་ཡོད། །རེ་ཞིག་དེ་ཡི་གྲགས་པ་ནི། །ཏྭཾ་ཏཱ་བཏྟ་སྱ་དསྨྲ་པྲ་ཤཱནྟ་ཡེ། །ས་སྲྀདྷི་ཀཱ་ར་ཎཾ་ཡཙྪ་ཏདྲི་གྷཱ་ཏོ་བི་དྷི་རབ་གཞིལ་ཐབས་དག་བསམ་པར་མཛོད། །གང་ཡང་ཕུན་སུམ་ཚོགས་པའི་རྒྱུ། །དེ་དག་གཞོམ་པ་བསྒྲུབ་ཡ་ཏྭཾ། །ཨི་ཏི་རཱ་ཛ་བ་ཙཐུ་ཏྭཱ་མ་ཧཱ་མཱ་ཏྱ་བ་བྷཱ་ཏི་རེ། །ཤུ་ཡ་ཏཱརྗེ་བ་པར་མཛོད། །ཅེས་པ་རྒྱལ་པོའི་ཚིག་ཐོས་ནས། །བློན་པོ་ཆེན་པོས་རབ་སྨྲས་པ། །ལྷ་གཅིག་གང་གི་ཡེ་ནཱ་སོ་རཱ་ཛ་དྷ་ན་ཛ་ནོརྫི་ཏ། །བི་ཥ་ཡེ་ནཱ་ག་རཱ་ཛོསྟི་ཙི་ཏྲ་ནཱ་མ་བ་རྒྱལ་པོ་ནི། །ནོར་དང་སྐྱེ་བོ་དར་བ་ཉོན། །ས་བདག་དེ་ཡི་དར་བ་ནི། །གཟུགས་ཅན་བསོད་ནམས་ཧུ་ད་ཀ། །ཏ་སྱཱཀྐི་ཏི་པ་ཏེ་མཱུཏྟཔུ་ཎྱ་བ་དྷ་ཨི་བོ་ད་ཡ། །ཨ་ཀཱ་ལེ་ས་སྐྱ་ཀྱིས་བཏགས་བཞིན། །ཡུལ་ན་ཀླུ་ཡི་རྒྱལ་པོ་ནི། །ཆུ་མང་ཙི་ཏྲ་ཞེས་པ་ཡོད། །དེ་ཡི་མཐུ་ཡིས་དུས་མིན་ནིཋཏྟིསྟིཏྠ་བྷཱ་བེ་ནེ་ཛཱ་ཡ་ཏེ། །ཀྲྀ་ཏི་སཾ་པཏྟི་མཱུ་ལཉྩ་བྷུ་བྷུ་ཛཱ་སརྦ་དུ། །འབྲུ་ནི་ཕུན་སུམ་ཚོགས་པར་འགྱུརཚེ། །ས་བདག་ཕུན་སུམ་ཚོགས་ཀུན་གྱིས། །རྩ་བ་ཞིང་ནི་།
