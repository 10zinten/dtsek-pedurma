བསྟན་འགྱུར། འཁྲི་ཤིང༌། ཀིའིཡུཏྟཾ་ཡུལྟ་མུ་བྱ་ཏེ། །དྷརྨོ་པ་དེ་ཤ་ཀུ་ཤ་ལཾ་བཀྟི་བྷ་ག་བ་ཏི་སཱ་འདུ་བྱེད་མི་རྟག་ཡང་དག་ལྡན། །ཆོས་ཀྱི་གདམས་ངག་དགེ་བ་ནི། །རིགས་པར་གསུང་བར་བརྩོན་པའི་ཡཾ༑། ཨེ་ཀ་ཤཱཀྱ་བ་དྷུཔ་ཏྲ་རཱུ་པ་སོ་བྷཱ་གྱི་གཟི་ཏཱ། །སྦི་ཏཱ་ཤི་ཤ་བ་ཏཱ་རུ་ཎྱ་ཚེ། །དེར་ནི་ཤཱཀྱའི་བུ་མོ་གཅིག། བུ་མོ་གཞོན་ནུ་མ་ཡི་མཚམས། །བཟོད་དཀའི་ལང་ཚོ་ལ་གནས་པ། །བནྟོ་བ་ཡ་སི་དུས་ཧེ། །མུཀྟ་ཧཱ་རཾ་སྟ་ན་ཏ་ཌེ་ལོ་ལ་པཾ་གིརྨུ་ཧུརྨུ་ཧུ། །གཟུགས་དང་སྐལ་བཟང་ཉིད་ཀྱིས་དྲེགས། །ནུ་མའི་ངོས་སུ་མུ་ཏིག་གི། དོ་ཤལ་དགའ་མའི་བདག་པོ་ཡི། །ཨཱ་ལུ་ལོ་ཀེ་ཡ་ཤ་སྭ་ར་སཱ་རཾ་ར་ཏི་པ་ཏི་རི་བ། །ཧཱ་ར་བ་ལོ་ཀི་ནཱི་ཛཾཥྚ་གྲགས་པའི་སྙིང་པོ་གསལ་འདྲ་ལ། །ཟུར་མིག་གཡོ་བས་ཡང་ཡང་བལྟས། །མིང་ཆེན་ཆུང་མ་ཟླ་འོད་ཀྱིས། །ཏཱ་མ་ནེ་ཀཱ་གྲ་མཱ་ན་སཱ། །ཡ་ཙིནྟ་ཡད་བི་རཀྟ་བ་མ་ཧཏ་པདྡྷི་ཤ་ཤི་པྲ་རྩེ་གཅིག་མིན་པའི་སེམས་ལྡན་དེ། །དོ་ཤལ་རྣམ་པར་བལྟ་བ་དག། མཐོང་ནས་ཆགས་བྲལ་བཞིན་དུ་བྷ༑། ཨི་ཡཾ་དྷརྨོ་པ་དེ་ཤེ་པི་ཙ་པ་ལཱ་ཧཱ་ར་མིགྱུ་ཏེ། །བྷཱ་བཱ་ནཾ་ན་ཤི་ཎོ་ཏེ་བསམས། །ཆོས་ནི་ཉེ་བར་སྟོན་ཚེ་ཡང་། །གཡོ་བའི་དོ་ཤལ་འདི་ལ་ལྟ། །དངོས་པོ་རྣམས་ཀྱི་སྐད་ཅིག་བ་མཱུ་ཌྷ་ཀཱ་ཎི་ཀ་ནཱ་མི་ཚྭ། །སྭཱ་ཧཱ་དན་ཡི་ཏཱ་སྱ་ཧཱ་རོན་ཧཾ་ཧ་ར་མྱ་མ༑། འདི་དག་རྨོངས་པས་མི་ཉན་ཉིད། །རང་གི་དོ་ཤལ་བསྟན་ནས་འདི། །དོ་ཤལ་སྤྲོ་བ་བདག་གིས་ཧཾ༑། ཨ་དྷི་ཀ་ལོ་ཀ་དེ་ནི་བ་དཤ་མྱ་ཏི་དེ་ཧི་ནཾ། །ཨི་ཏི་སཾ་ཙིནྟ་དབྲོག། ལྷག་པ་མཐོང་བ་ཁོ་ན་ཡིས། །ལུས་ཅན་དྲེགས་པ་ཞི་བར་གྱུར། །འདི་བསམས་དེ་ཡིས་དཱ་སི་རོ་ཧི་ཏཱ་ཁྱ་མ་བྷ་ཥཏ། །རོ་ཧི་ཀེ་གཙྪ་མེ་ཧཱ་རཾ་ཁཾ་ཧཱ་ཏ་ས་ཏཱ་མཱ་ཧ་འབངས་མོ་ནི། །རོ་ཧི་ཏ་ཞེས་བྱ་ལ་སྨྲས། །རོ་ཧི་ཏ་སོང་བདག་གི་ནི། །དོ་ཤལ་ཁྱིམ་ནས་མགྱོགས་སུ་ར༑། ཨི་ཏྱཱཀྟ་སཱ་ཏ་ཡ་ཏ་ཏྲ་པྲ་བྲྀཏྟེ་དྷརྨ་སཾ་ཤྲཱི་བེ། །ཨ་ཀཱ་ལ་ག་ལོངས༈། དེ་ཡིས་དེ་སྐད་བརྗོད་པས་དེ། །དེར་ནི་ཆོས་ཉན་ཡོལ་ཞུགས་པ། །དུས་མིན་འགྲོ་བས་མ་ནོ་བབྷྱཱ་ནི་ཤཱ་ཙི་ཏཾ་ཡ་ཏ་ཀླུ་ཎཾ། །ཨ་ཧོ་བ་ཏན་རོ་ཡོ་ཡཾ་སཾ་ཡིད་བྱུང་ནས། །ཤུགས་དང་བཅས་པས་སྐད་ཅིག་བསམས། །ཀྱེ་མ་བདག་གི་དགེ་བ་འདིར། །བར་—།
