ལ་ནཾ༑། ཨུ་པ་བི་ཤུ་གྲ་ཏསྟརྨི་བོ་ར་དུཁཾ་རྒྱ་བེ་ད་ཡཏ། །།བྷ་ག་ཝཱ་ན་བི་བི་ཛྷཱ་ཡ་ཏེ། །མདུན་དུ་འཁོད་ནས་དེ་ལ་ནི། །གྲོང་ཁྱེར་པ་ཡི་སྡུག་བསྔལ་སྨྲས། །གྲོང་ཁྱེར་པ་ཡི་བརྒྱུད་དག་ནི། །པོ་རཱ་ཎཱ་སནྟ་ཏིཀྵ་ཡཾ། །ཙིནྟཱ་ནིཉྩ་ལ་ཏྭཾ་ཀྱི་དྲྀ་མུ་བ་ཧཱ་ཀ་རུ་ཎ་ནི་ཉམས་པ་མཁྱེན་ནས་སྙིང་རྗེའི་གཏེར། །བཅོམ་ལྡན་འདས་ཀྱང་མྱུར་བར་ནི། །ཐུགས་ནི་གཡོ་བ་མེད་ཉིད་དྷི༑། ས་བི་དྲི་ཚུ་ཛ་གདྡནྡྷས་ཚིགས་པཾ་སོ་ར་མཎྜ་ལཾ། །པཱ་ཏྲ་འཛིན། །འགྲོ་བའི་དཀྱིལ་འཁོར་མི་བདག་དང་། །བཅས་པས་བསྐུལ་ནས་འགྲོ་བའི་གཉེན་། །དེ་ནི་ཙཱ་བ་ར་མ་དཱ་ཡ་ཡ་ཡོ་ཡཀྵི་ཎི་ཀེ་ཏ་ནཾ། །ཏ་ཡཱ་བི་ར་ཧི་ཏཾ་པྲ་པྱ་ལྷུང་བཟེད་ཆོས་གོས་བཟུང་། །གནོད་སྦྱིན་མོ་ཡི་ཁང་པར་གཤེགས། །དེ་ཡིས་དབེན་པ་དེ་ཡི་ནི། །ཁྱིམ་ཏཛཾ་ཧཾ་བྷ་ག་བཱན་ཛི་ན། །པྲི་ཡཾ་ཀ་རཱ་ཁྱ་ཏཏུ་ཏྲ་ནི་ནཱ་ཡི་ཀ་མ་དན་ནོ། །དུ་བཅོམ་ལྡན་རྒྱལ་བས་བྱོན། །དགའ་བྱེད་ཅེས་པ་དེ་ཡི་བུ། །གཅིག་ནི་སྣང་བ་མིན་པར་མཛད། །ཡཱ་ཏེ་བྷ་ག་བ་ཏེ་སྤྱི་པྲ་ཡཀྵ་སྭཱ་ཁཾ་ཧ་མ་ག་ཏཱ། །པྲ་བྷཱུ་ཏ་པུ་ཏཱ་ནཱ་པ་བཅོམ་ལྡན་གཤེགས་ནས་གནོད་སྦྱིན་མོ། །མྱུར་བར་རང་ཁྱིམ་འོངས་པ་ཡིས། །རབ་མང་བུ་ལས་དགའ་བྱ་པྲ་ཡཾ་པུ་ཏྲ་པྲ་ཡཾ་ཀ་རཾ། །ཏ་མཱི་ཀྵ་མཱ་ཎཱ་པཱ་བེ་ཤཱ་ཀྲྀ་ཏ་བཏྲེ་བའི་བུ། །དགའ་བྱེད་དག་ནི་མཐོང་མ་གྱུར། །དེ་ནི་འཚོལ་ཞིང་རྨོངས་བཅས་པས། །དྲུས་མ་བེའུ་བ་དྷེ་ནུ་ཀཱ། །བ་བྷཱ་མ་སཾ་བྷ་མོཏྟཱནཱ་སཱ་ཛ་ནེ་ཥཱ་པ་ནེ་ཥཱ་ཙ། །ཧཱ་སྟོར་བ་བཞིན། །འཇིག་རྟེན་དང་ནི་ནགས་རྣམས་ཀྱང་། །རིངས་པ་རབ་འཁྲུལ་དེ་ཡིས་བསྐོར། །ཀྱེ་པྲི་ཡཾ་ཀ་ར་ཧཱ་པུ་ཏྲ་ཀཱ་ནུ་པ་ཤྩ་མི་ཏེ་མུ་ཁཾ། །ཨི་ཏི་པྲ་ལཱ་པི་ནཱི་ཏཱ་རཾ་ནིཧུད་དགའ་བྱེད་ཀྱེ་ཡིཧུད་བུ། །ཁྱོད་ཀྱི་བཞིན་ནི་གར་མཐོང་ན། །ཞེས་པ་ཆེ་བར་རབ་སྒྲོགས་ཤིང༌། །ཤེ་ཥསཱ་ཡ་ཡོ་དི་ཤ། །སཱ་པི་ཙིནྟཱ་ཤུ་སརྦ་སཱ་ནི་ར་ཤཱ་པུ་ཏྲ་དན་ནེ། །དེ་ནི་མ་ལུས་ཕྱོགས་སུ་སོང་། །དེ་ཡིས་ཕྱོགས་ཀུན་མགྱོགས་བཙལ་ནས། །བུ་ནི་མཐོང་རེ་མེད་པ་ཡིས། །ཀྲེ་ཤནི་སརྦ་ཏ་དཱི་པ་ས་མུ་དྲ་བ་ལ་ཡཾ་ཡ་ཡོ། །མཎྜ་བྷུ་མི་མ་ཏི་ཀྲ་མྱ་མྱ་ངན་བྱེད་ཅིང་རི་དང་གླིང་། །རྒྱ་མཚོའི་དཀྱིལ་འཁོར་དག་ཏུ་སོང་། །མི་ཡི་ས་ལས་འདས་ནས་ནི། །—།
