ཀཎཱ་མྲྀ་ཏ་མི་དཾ་དྷུ་ར་ཡཎཱ་ཧཱརྨི་མ་ཎི་པྲ་ད། །སཾ་བཱསྟཱ་ནཾ་པུ་ནས་ཏྲ་མཱ་གང་འདི་རྒྱ་མཚོར་ནོར་བུ་སྟེར། །རིང་ནས་རྣ་བའི་བདུད་རྩི་སྟེ། །སློང་བ་ཡང་དག་འོངས་རྣམས་ཀྱིས། །ན་ཏེ་མ་སྱ་ནྱ་ཏོཟིགས་ནཾ། །རཀྵ་སརྦ་པྲ་ཡནྣེ་ན་པ་རེ་ཥཱ་མ་པི་ཛཱི་བི་ཏེ། །དེར་ནི་བཏུང་བའང་གཞན་ལས་སོ། །རྒྱལ་པོ་སྐྱེ་རྒུ་རྣམས་ཀྱི་སྲོག། སློང་བ་རྣམས་ཀྱི་ཡིད་བཞིན་ནོར། །པྲ་ཛཱ་ནཾ་ཛཱི་བི་ཏཾ་རཱ་ཛན་ནཏྲི་ཙིནྟཱ་མ་ཎིར་བྷ་བཱན། །པྲ་སཱི་ད་ད་ཡ་ཁྱོད་ནི་འབད་པ་ཐམས་ཅད་ཀྱིས། །གཞན་སྲོག་གིས་ཀྱང་བསྲུང་བར་འོས། །ལྷ་གཅིག་བབ་ཅོལ་མ་ཡཱསྣ་སུ་དེ་བ་མཱ་སཱ་ཧ་སཾ་ཀྲྀ་ཐཱ། །ན་ཀཱ་ཙ་ས་ཀ་ལ་སཱུརྗེ་ཀྲི་ཡ་ཏེ་མཛད་པར། །བདག་ལ་དང་ཞིང་བརྩེ་བར་མཛོད། །མཆིང་བུ་དུམ་བུའི་སླད་དུ་ནི། །རིན་ཆེན་བཙོང་ཙཱཏྨ་བི་ཀྲ་ཡ། །ཨི་ཏུཏྟཱ་པཱ་ད་ཡོཔ་ཏུཔ་པཱ་ཏཱ་མཱ་ཏྱ་པཾ་ག་བ། །ཤ་བར་མ་མཛད་ཅིག། འདི་སྨྲས་རྗེ་ཡི་ཞབས་དག་ལ། །བློན་པོའི་ཉེགང་ཟག་འགྱེལ་གྱུར་ཀྱང༌། །ས་རཱི་ར་དཱ་ན་སཾ་ཀ་ནུ༔་ཙཱ་ལ་ཙ་བྷཱུ་པ་ཏི། །སོ་བ་དཏ་པྲ་ཎ་ཡསྨེ་བདག་ལུས་ནི་སྦྱིན་པ་ཡི། །རྣམ་པར་བརྟག་ལས་གཡོས་མ་གྱུར། །དགའ་བའི་འཛུམ་གྱིས་ཕྱེ་བ་ཡི། །ར་བི་ཀ་སནྡ་ཤ་ནདྨ་ཏི། །ཛཱི་བི་ཏ་སྣེ་ཧ་སཾ་མོ་ཧ་ཏ་མཉཔ་རི་ཧ་རནྣི་བྷ། །སོ་ཡི་འོད་ཀྱིས་སྲོག་ལ་ནི། །ཆགས་པ་རྨོངས་པའི་མུན་པ་དག། ཡོངས་སུ་ཕྲོགས་བཞིན་དེ་ཡིས་སྨྲས། །ཀེ་བ་ལཾ་བྷཏྟི་སཾ་ཡུཀྟ་མུཀྟ་བྱཀྟ་མི་ད་ཏཱ་ཡཱ། །ན་ས་ཧེཉཧཾ་མ་ཧཱ་མཱ་ཏྱ་བློན་པོ་ཆེན་པོ་ཁྱོད་ཀྱིས་ནི༄༅། །གུས་དང་བཅས་པ་འབའ་ཞིག་ཏུ། །གསལ་བར་བརྗོད་དེ་བྲམ་ཟེའི་སྲོག། བི་པྲ་སྱ་པཱ་ཎ་སཾ་ཤ་ཡཾ། །ཧཱ་རི་སྐུ་ཥཱ་རཿཀ་མ་ལེམྲྀ་ཎཱ་ལི་རིནྡྲ་ཙནྡ་ནི། །ནི་བརྟ་ཐེ་ཚོམ་ང་ཡིས་མི་བཟོད་དོ། །དོ་ཤལ་གྲང་རེག་པདྨོ་དང་། །པད་རྩ་ཟླ་བ་ཙནྡན་གྱིས། །ནང་གི་གདུང་ཏེནྟས་ཏཱ་པོ་ནཱརྨི་བ་མུ་ཁྱ་དུཁ་ཛ། །སརྔ་ཐཱ་སརྦ་དུཁཱཏཾ་བཟློག་སློང་བ་ནི། །།ཕྱིར་ཕྱོགས་སྡུག་བསྔལ་ལས་སྐྱེས་མིན། །སྡུག་བསྔལ་དག་གིས་གཟིར་བ་ཀུན། །ཧ་ར་ཎོང་ཏ་ཙེ་ཏ་ས། །ན་བོ་དྷེ་རནྟ་རཱ་ཡཾ་མེ་སུ་མ་ཏེ་གཞུ་མརྡ་སི། །ཀུན་ནས་འཕྲོག་སེམས་སྤྲོ་ལྡན་པ། །བདག་གི་བྱང་ཆུབ་བར་ཆད་ནི། །བློ་བཟང་བགྱི་བར་མི་འོས་སོ། །།
