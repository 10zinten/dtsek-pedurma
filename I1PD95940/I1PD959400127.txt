ན་མཱཙྪི་བ་ཏིཐཾ་སུ་ཙི་མཱ་ན་སཱ་ནཾ། །ཨེ་ཐཾ་བི་དྷོ་ཡཾ་ནམས་བརྩེ་བ་ཉིད། །ཡིད་ནི་གཙང་བ་རྣམས་ཀྱི་འཇུག་ངོགས་ཀྱི་བདག་ཉིད་དོ། །སྐྱེས་བུ་རྣམས་ཀྱི་རྣམ་གུ་ཎ་སནྣི་བེ་ཤཤཱི་ལེ་ན་བི་མ་ལ་མུ་པི་པཾ་ཧཱུཾ། །སད་རནྣ་མུཀྵ་ནི་པ་དེ་ལྟའི་ཡོན་ཏན་དག། ལེགས་གནས་འདི་ནི་ཁྲིམས་ཀྱི་དྲི་མ་མེད་པར་གྱུར། །རིན་ཆེན་མུ་ཏིག་དམ་ཀ་རཱ་ཏི་རིཀྟ་ཤཱི་ལཾ་ས་ཏཱ་མཱ་བྷ་ར་ཎཾ་བ་དནྟི། །ཨི་ཏུལྟ་པའི་ཚོགས་ལས་ཤིན་ཏུ་ལྷག། ཚུལ་ཁྲིམས་དག་ནི་དམ་པ་རྣམས་ཀྱི་རྒྱན་དུ་བརྗོད། །སེམས་ཅན་དོན་མཱ་ཀཙྩ་གུ་ཎཱ་ནུ་རཱུ་ཐཾ་ས་ཏྭཱཟི་ནཱ་ཛི་ཏེནྡྲི་ཡེ་ཎ། །ཏུ་ཏྟཱསྟ་གཉེར་དབང་པོ་ཐུལ་བ་དེ་ཡིས་ནི། །ཡོན་ཏན་རྗེས་མཐུན་དེ་སྐད་བརྗོད་པ་ཐོས་གྱུར་ནས། །ཚིམ་པར་མཱུ་ཙུནྡྷ་བི་ཙནྡ་ལོ་ཀཾ་ཏཱ་ཀོ་ཏུ་ཀཱ་ཡི་བ་མུ་ཁེ༔སྲྀ་རྀ་ཏྱ། །མ་ཎི་གྱུར་ཅིང་ལྟད་མོའི་དོན་བཞིན་མཁའ་ལས་ནི། །ཟླ་བ་འཇིག་རྟེན་ས་ལ་སྤོ་ཞིང་དེ་ཡིས་སྨྲས། །ནོར་བུ་རི་བཱ་ནགྷ་གུ་ཎོཛྫ་ལ་སྱ་པྲྀ་བ་སཱ་དྷོ་རུ་ཙི་ཙཱ་རུ་ཙིསྟེ། །ཡ་བཞིན་དུ་ཡོན་ཏན་རབ་འབར་རིན་ཐང་མེད། །དགེ་བ་ཁྱོད་ཀྱི་མཛེས་པའི་འོད་ནི་མཐོང་གྱུར་པ། །གང་ཡི་བ་མོ་ལོ་ཧྲི་ད་ཡེ་ཐུ་ཏོ་ཙ་སདྷི་ས་དེ་བཱནྡ་ར་ཎི་ཀྲྀ་ཏོ་སི། །ཞིག་མགོ་དང་རྣ་བ་སྙིང་ག་ཉིད་ལ་ཡང་། །རྟག་པར་དམ་པ་རྣམས་ཀྱི་རྒྱན་དུ་ཁྱོད་ཉིད་བགྱིད། །མ་ཎིརྨ་ཧཱདྷཔྲ་ཐི་ཏ་བྲ་བ་བཔྲ་གྲོ་རྒྱ་ཏཱ་མཱདྷྱ་ས་མསྡུ་བདག་ཉིད་དང་མཉམ་གྲགས་པའི་མཐུ་ལྡན་རིན་ཐང་ཆེ། །ནོར་བུ་འདི་ནི་ཁྱོད་ཀྱིས་རབ་ཏུ་བཟུང་བར་ཡཱ་ཡཾ། །དྷཱ་ཛཱམི་ཏོ་བཆུ་ཏི་ཡོ་ཛ་ནཱ་ནཾ་ས་ཧ་སྲ་མེ་བནི་ས་མ་མཛོད། །གང་ཞིག་རྒྱལ་མཚན་ལ་བཞག་དཔག་ཚད་སྟོང་ཕྲག་ཏུ། །སློང་བ་རྣམས་ཀྱི་འདོད་པ་ཆར་དུ་ཧི་ཏཾ་ཡ། །ཨུཀེ་ཏི་ར་པྲ་བ་རཾ་ཏ་རུ་ཅུསྟརྨི་ད་དུ་ཌྷཎྜ་བབ་པར་འགྱུར། །དེ་སྐད་བརྗོད་པས་རིན་ཆེན་རབ་མཆོག་གཞོན་ནུ་མས། །རབ་ཏུ་དང་བ་གཟུགས་མི་བ་པྲ་སཱ་དཾ། །ཨཱ་དཱ་ཡ་ཏཙྪ་པྲ་ཎ་ཡོ་པ་ཙཱ་ར་སོ་ཅན་བཞིན་དུ་དེ་ལ་བྱིན། །དེས་ཀྱང་དེ་ནི་མཛའ་བའི་ཉེ་བར་སྤྱོད་པས་བླངས། །དངུལ་ལས་བྱས་པའི་།
