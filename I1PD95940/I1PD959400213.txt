ག་བ་ཏཱ་ཏཏྟ་ནི་ཛ་ཀེ་ཤ་ན་ཁཾ་ཤ་ཀཾ། །ས་ཛ་གཱ་མཱ་ས་མཱ་མཱ་དཱ་ཡ་བཱ་ཀུ་ཌི་ཀྱི་རྫུ་འཕྲུལ་གྱིས། །རང་གི་དབུ་སྐྲ་སེན་ཆ་བྱིན། །དེ་ཡིས་བླངས་ནས་བ་ཧུ་ཌ། །ཞེས་པའི་དཀྱིལ་ནཱ་མ་མཎྜ་ལཾ། །ཏ་ཏྲ་པྲ་ཛྙཱ་པྲ་བྷཱ་བེ་ན་སོརློཏྭཱ་ཧ་གུ་ཎེ་ན་ཙ། །ས་ཕྲུ་པ་འཁོར་དག་ཏུ་སོང་། །དེར་ནི་ཤེས་རབ་མཐུ་དག་དང་། །རྟུལ་ཕོད་སྤྲོ་བ་ཡོན་ཏན་གྱིས། །དེ་ཡི་རྒྱལ་རཱཛྲ་དྷཱ་རཱ་ཎཾ་སརྦ་ཏྲ་སུ་ལཱ་བྷཤི་ཡ། །དཀྵ་ཎཱི་ལཀྵ་ཎཾ་ལགཱི༔ས་ཧ་སྲིད་ཐོབ་སྟེ་དཔལ། །བློ་ལྡན་རྣམས་ལ་ཀུན་ཏུ་མོད། །དཔལ་འབྱོར་གྲུང་པོའི་མཚན་ཉིད་དེ། །གྲགས་ཛཾ་བི་དུ་ཀྵ་ཡ་ཤ། །ནྱ་བ་སཱ་ཡ་ས་ཧཱ་ཡཱ་ཀཾ་ཀ་ལ་ཧུཾ་སརྦ་སིདྡྷ་ཡ། །ཏ་པ་མཁས་པའི་ལྷན་སྐྱེས་ཡིན། །གྲུབ་པ་ཐམས་ཅད་ཆུང་མ་ནི། །རྟག་འགྲུས་ལྡན་པ་རྣམས་ཀྱི་ཡིན། །དེ་ཏྲསྦི་ཏོ་བྷ་ག་བ་ཏསོ་ཐ་ཀེ་ཤ་ན་ཁཾ་ཤ་ཀེ། །སྐུ་པ་པྲ་ཏིཥྛ་མ་ཀ་རོད་ནས་དེར་ནི་དེར་གནས་ཏེ། །བཅོམ་ལྡན་དབུ་སྐྲ་སེན་ཆ་ལ། །རིན་ཆེན་མཆོག་གིས་རབ་མཛེས་པའི། །བ་ར་རནྣཱ་བི་རཱ་ཛི་ཏྭཾ། །བི་རཱུ་ཌྷ་ཀི་པི་ཤཱཀྵ་ནཾ་བི་ར་པཱ་ར་མཆོད་རྟེན་རབ་ཏུ་གནས་པར་བྱས། །འཕགས་སྐྱེས་པོའང་ཁོན་གྱིས་ནི། །ཕ་རོལ་བསྒྲལ་འདོད་སླད་ཏཱིར་ཡཱ། །སུ་ནརླབྱ་པུ་རཾ་དྷཱ་ར་བྷེ་དེ་ན་ཧ་སཱ་བི་ཤཏ། །ཧ་ཏཱ་ཏ་ཏྲ་ས་ཧ་པྲ་ནས་ཡང་། །སྒོ་དབྱེའི་རིག་པས་འཕྲལ་ཉིད་ལ། །ཤཱཀྱ་རྣམས་ཀྱི་གྲོང་དུ་ཞུགས། །དེར་ནི་ཤཱཀྱ་སྟོང་ཕྲག་ཎི་ཤཱཀཱ་ནཾ་སསྟ་སཔ་ཏི། །བདྷ་ཀ་རྩྭ་ཀུ་མཱ་རཱ་ཎྞཾ་ས་ཧ་ས་མ་དག། བདུན་ཅུ་རྩ་བདུན་རབ་ཏུ་བསད། །དེ་ནས་བུ་མོ་གཞོན་ནུ་རྣམས། །སྟོང་ཕྲག་དེ་ཡིས་ཐཱ་ཧ་རཏ། །ཤ་ཏཱ་ནི་པཉྩ་ཤཱཀྱ་ནཱ་ག་ཛིརློ་ཧེཐུ་མརྡ་ནི། །སཾ་པྲ་བཅིངས་ནས་ཁྱེར། །ཤཱཀྱ་བརྒྱ་ཕྲག་ལྔ་རྣམས་ནི། །ལྕགས་དང་གླང་པོའི་འཇོམས་བྱེད་ཀྱིས། །རབ་ཏུ་མྲ་ཛེ་པུ་རཱི་ཙཀྲེ་ཀྲྀ་ཏཱནྟ་ན་ག་རཱི་མི་བ། །བྷ་ག་བཱན་པི་ཤཱཀྱཱ་ནཾ་བཅོམ་ནས་གྲོང་ཁྱེར་ནི། །མཐར་བྱེད་གྲོང་ཁྱེར་བཞིན་དུ་བྱས། །བཅོམ་ལྡན་གྱིས་ཀྱང་ཤཱཀྱ་རྣམས། །ཤ་ཏུ་ཚྭ་བྷ་དེ་ནཾ་ཀྲྀ་ཏཾ། །ཀམཱུ་ནུ་བབྷཾ་བི་ཙྪཱ་ཡ་བ་བྷཱུ་བ་བི་དགྲ་རྣམས་ཀྱིས་ནི་བརླག་བྱས་པ། །ལས་ཀྱིས་རྗེས་སུ་བཅིངས་®མཁྱེན་ནས། །སྐད་ཅིག་ཐུགས་ནི་—།
