གཎྜི་ཤསྔ་ས་མཱ་ག་ཏཾ། །ཨེ་ཀེ་ཨེ་བོ་པ་ཧཱ་རཱ་ཡ་བྷིཥྛ་རསྣ་ཀ་དགེ་སློང་དགེ་འདུན་ཚོགས་རྣམས་བཅིངས། །ཡ་སྟགས་སླད་དུ་དགེ་སློང་ནི། །གཅིག་ཉིད་ཁོ་ན་བདག་མིཔི་ཏ། །ཨི་ཏུཀྟ་ཙོ་ར་ཙཀྲེ་ཎ་ཛ་ག་དུབྷིཀྵ་བཀྲ་མཱཏྲ། །མ་མཱ་ཅག་འདོད། །ཆོམ་རྐུན་ཚོགས་ཀྱིས་འདི་བརྗེད་ཚེ། །དགེ་སློང་རྣམས་ཀྱིས་རིམ་པས་སྨྲས། །ཡ་སྟགས་དཱ་ཡོ་མ་ཧཱ་རཱཐཾ་མུ་ཙྩ་་སརྦ་བྷིཀྵ་བ། །ཨུཀེ་བྲོདྷེཀྲ་མེ་ཎེ་ཏི་དོན་དུ་བདག་ཟུང་ལ། །དགེ་སློང་ཐམས་ཅད་བཏང་བར་མཛོད། །ཅེས་པ་རྒན་རིམ་གྱིས་བརྗོད་ཅིང་། །པྲ་བཱ་ཙ་མ་དྷུ་ར་སཱ་ར། །ཨ་ཧ་མེ་བོ་པ་ཧཱ་རམཿསཾ་གྷསརྦ་བི་མུ་ཙུ་སྦྲང་རྩིའི་དབྱངས་ཀྱིས་རབ་སྨྲས་པ། །ང་ཉིད་ཁོ་ན་ཡ་སྟགས་དོན། །དགེ་འདུན་ཐམས་ཅད་རྣམ་པར་ཏྭཾ༑། ཨི་ཏྱ་ཀརྞ་བ་རཱ་ཀཱ་བ་རཾ་ཏེ་ས་མཱ་དཱ་མ་ཏཾ་ཡ། །བདྡྷམ་མ་བ་ཐོང་། །ཞེས་པ་ཐོས་ནས་དེ་དག་གིས། །རྣམ་པ་ཅིམཆོག་ལྡན་དེ་བཟུང་སོང་། །གསོད་པར་ཆས་པ་དེ་དྷ་སནྣདྷེ་ནཱི་ཏསྟེནིཆི་ཀཱ་ར་དྷཱི། །དུནཱ་ཡ་ཏ་ན་མ་ཎྱ་གྲོ་ད་དན་དག་གིས། །བཅིངས་ཁྱེར་རྣམ་འགྱུར་བློ་མེད་དེས། །དུ་རྒའི་གནས་ནི་ནགས་ནང་དུ། །ཤིན་ཏུ་ག་ཧ་ནོ་ད་རེ། །བ་ལི་སཏྟཱི་ཀྲྀ་ཏེ༔སྦྲུ་ལ་ཤ་ཀ་ལེཎཱ་ཧི་ཥིཉྩ་ཏཏ། །འཇིགས་སུ་རུང་བ་མཐོང་། །དེ་ཡང་དུས་མཐའི་སྤྲིན་གྱིས་བཞིན། །གཏོར་མ་ལ་ནི་བསྟར་བྱས་པའི། །ཀཝནྟ་མེ་གྷིསམྦ་ཏོ་པ་ཧཱ་རི་རི་བ་པཱུ་ཛི་ཏཾ། །བྱཱཏྟཾ་པཱ་ད་ཤི་ལཱ་སཀྟ་རུ་རུ་མ་ཧེ་སྲེ་བོ་རགས་པ་ཡི། །རྟག་ཏུ་ཡ་སྟགས་དག་གིས་གང་། །དཔའ་བོས་བན་དུ་ཛི་བ་ཡི། །རང་བཞིན་རཀྟཙྪ་ཊཱ་ཤ་ཏི། །སུགྷ་མ་བྷིནྡྷ་ཊིཏྤཋ་བནྡྷཱ་ཛཱི་བ་མ་ཡི་རིབ། །གྷཎཱ་མེ་ཏོག་འཕྲེང་སྤོས་བཞིན། །རུ་རུའི་ཁྲག་གི་ཟེར་མ་བརྒྱ། །ཞབས་ཀྱི་རྡོ་ལ་ཆགས་པས་ཁྱབ། །གཤིན་ཤོལམྤི་བྷིནྡྷི་ར་ཤི་རོ་བྷིཔ་རི་བཱ་རི་ཏཾ། །ཀྲྀ་ཏ་དཱ་རཱཙྪ་ནཾ་ཕུལླ་ཀ་རྗེས་མེ་ཏོག་རྒྱས་པ་ཡི༄༅། །སྒོ་ལ་མཆོད་པ་བྱས་པ་བཞིན། །དྲིལ་བུའི་སྔོན་དུ་འཕྱང་བ་ཡི། །དཔའ་མ་ལེ་རི་བ་ཚྭ་སྐྱུ་ནཱ། །པྲ་ཏྱ་གྲ་ན་ར་རཀྵ་རཀྟིསོ་པཱ་ནི་ར་རུ་ཎ་པྲ་བྷོ། །བོའི་མགོ་ཡིས་ཡོངས་སུ་བསྐོར། །རི་ཁྲོད་མ་ཡི་རྐང་བཀོད་པས། །སེན་རྩི་རབ་ཏུ་ཆགས་པ་བཞིན། །—།
