ལུ་ཏྭཱ་ཧ་སནི་བ་དྷ་བ་བྷ་མཾ། །ག་ཏེ་ཏསྨིན་མུ་ཁ་ཤ་ཤི་པྲ་ཀཱ་ལ། །དགོད་པ་བཞིན་དུ་དེ་རབ་སོང་། །བཞིན་རས་ནམ་མཁའི་ཟླ་བའི་ཚིགཞི། །རབ་གསལ་ཤི་ཏ་ན་བྷ་ཏ་ལེ། །བྷ་ག་བཱན་བྷིཀྵ་བྷིཔྲཋསྟབྲྀཏྟཱན་མ་བྷཱ་ཥ་ཏ། །པུཊྚ་བྱས་ནས་དེ་སོང་ཚེ། །དགེ་སློང་གིས་དྲིས་བཅོམ་ལྡན་གྱིས། །དེ་ཡི་བྱུང་བའི་མཐའ་བཀའ་སྩལ། །སྔོན་ཀཡན་རེ་ཤཱསྟུཀཱ་བྱ་པཱ་ཁྱ་སྱ་ཐཱ་ས་ནེ། །ཨ་བྷཱུཏྤ་བྲ་ཛི་ཏོ་པེ་ཥ་དུས་གཞན་ལ་སྟོན་པ་ནི། །འོད་སྲུངས་ཞེས་པའི་བསྟན་པ་ལ། །རབ་ཏུ་བྱུང་བར་གྱུར་ཀྱང་འདི། །ཤིཀཱ་པ་ད་ནི་རཱ་ད་ར། །ཨ་ནཱ་ད་རཱཏ་ཀུཉྫ་ར་ཏ་བྷོ་གས་གྷོ་པ་སེ་བ་ནཏ། །བསླབ་པའི་གནས་ལ་གུས་མ་གྱུར། །མ་གུས་པ་ལས་གླང་པོ་ཡི། །ལོངས་སྤྱོད་ཉིད་དང་དགེ་འདུན་ནི། །ས་ཏྱ་དྲིཤྲཱི་བ་ལེ་ནཱནྟི་སཾ་པྲཤཱ་ས་ན་གྲ་ཧཾ། །པཱསྔཆ་་བི་ཧི་ཏཾ་བསྟེན་ལས་བདེན་པའི་ལྟ་སྟོབས་ཀྱིས། །་མཇུག་ཏུ་བསྟན་པ་འཛིན་པ་ཐོབ། །སྐྱེ་བ་སྔ་མར་བསྒྲུབས་ཀསྨྲ་ཀ་སྱ་ཙིནྡ་ནི་བརྟ་ཏ། །ཀརྨོ་པ་དིཋ་སམྦནྡྷ་བྷཏྟཾ་བྷོ་གྱི༔ས་པའི་ལས། །སུ་ཞིག་གིས་ཀྱང་མི་བཟློག་སྟེ། །གུས་པར་སྤྱད་བྱས་སེམས་ལྡན་ལ། །ལས་ཀྱི་ཉེ་བར་ཙེ་ཏ་ས། །ཏསྨིན་བྱ་ཏི་ཀ་རེ་གྷོ་རེ་སམྦསྱཱཀྟིརྨི་བྷིཀྵ་བྷི། །ནབསྟན་ནས་འབྲེལ། །འཇིགས་རུང་གནོད་པ་དེ་ལ་ནི། །དགེ་སློང་ཀུན་གྱིས་བདག་བཏང་ལ། །དེར་ཏཱ་ནནྡ་ན་ཏ་ཏཱ་པི་ཐུ་ཡ་ཏྭཾ་པཱུརྦ་སཾ་ག་ཧི། །ཤ་ཤཱངྐ་ཤཱི་ཏི་ས་ར་སི་བྷཱ་ཡང་ཀུན་དགའ་བོས་མ་ཡིན། །སྔོན་ཚེ་འགྲོགས་པ་མཉན་པར་གྱིས། །རི་བོང་མཚན་མ་ལྟར་བསིལ་མཚོར། །ཏ་རོ་རུ་ཙི་རོ་པུ་རཱ། །པཱུརྞམུ་ཁལྕེ་ཏི་རཱ་ཛ་ཧཾ་སོ་བ་བྷཱུ་བ་ཏུ། །གང་བ་གང་བའི་བཞིན་ཞེས་པའི། །ངང་པའི་རྒྱལ་པོ་སྤུན་དག་ནི། །ཡིད་དུ་འོང་བར་སྔོན་གྱུར་ཏོ། །ཀ་དཱ་ཙིདྨ་དྨ་དཏྟ་སྱ་བཱ་རཱ་ཎ་སིཾ་མ་ཧི་པ་ཏི། །བྲཧྨ་མ་ཏཱི་པུཀྐ་རི་ཎཱི་ནམ་ཞིག་ཝཱ་རཱ་ཎ་སཱི་ཨིན། །ས་ཡི་བདག་པོ་ཚངས་བྱིན་གྱིས། །ཚངས་པ་ལྡན་པའི་རྫིང་བུ་ནི། །ར་ཨཾ་པཱུཙྪཱ་མུ་ཁོ་ཡ་ཡོ། །ས་ཏ་སུཾ་ལོ་ལ་ཀ་མ་ལཱ་ཀིཉྫལྟ་པ་རི་པཉྫ་དགའ་བར་གང་བའི་བཞིན་གྱིས་ཕྱིན། །གཡོ་བའི་པདྨས་དཀྲིགས་པ་ཡི། །ཟེ་ཁྱེའབྲུས་ཡོངས་སུ་སེར་བ་།
