མཱརྱ་ཙ་རི་ཏ་སྱ་ཛ་གདྡུ་ཛཱ་ཏ་ཧཎ་སྨི་ཏི་རི་བ་ན་ཁཾ་ཤུ་བྷི་རུལླི་ཁནྟཾ། །སེན་མོའི་འོད་ཟེར་དག་གིས་འགྲོ་བ་ལ། །འཕགས་པའི་སྤྱོད་པའི་ངོ་མཚར་རབ་ཏུ་འབྲི་བར་བྱེད་པ་བཞིན། །མེ་ཏཱི་བ་ས་ལི་ཏྭཾཀྵ་མེ་ཎ་ཀུ་ཀྲྀ་ཏྭཾ་པྲ་ཛྷེ་བ་ཙིནྟི་ཙ་ཡཏ་དུཁནུབྱམས་པས་འཁྲུལ་བཞིན་བཟོད་པས་ཉེས་བྱས་བཞིན་དང་ཤེས་རབ་ཀྱིས་ནི་བསམ་ཚོགས་བཞིན། །སྡུག་ས་ཧ་བི་སླ་བ་བྷཾ་ཏི་རི་བ་གླེ་ཐཾ་ཏ་པ་ཤཱི་རི་བ། །ཏ་བསྔལ་བཟོད་དཀའི་བསླད་པ་བརྟན་པས་བཞིན་དང་ཉོན་མོངས་དཀའ་ཐུབ་དཔལ་གྱིས་བཞིན། །བརྩེ་སྱ་བྱཱ་གྲྭ་བ་དྷཱ་ནི་པཱ་ཏ་བི་ཥ་མ་ཀཱུ་རཱ་བྷི་གྷ་ཏོལྤ་ནི་སེ་ཧེ་མཱནྟི་ར་ཙཉྩ་ལེ་བ་བས་དེ་ཡི་ཉིལུས་ནི་སྙིང་སྟོབས་ས་གཞི་གཡོ་བ་མེད་པ་དེ་ཉིད་ཀྱིས། །སྟག་གི་ཆུང་མས་བཏབ་ཅིང་མི་ད་ཡ་ཡཱ་སཱ་ས་ཏཱ་བྷཱུ་མིཌྷ་རཾ། །བྱཱ་བྷཱི་ན་ཁཱ་བ་ལི་བི་ལཱ་ས་བི་བཟད་དྲག་པོར་མངོན་འཇོམས་གསལ་བའི་ཁུར་དག་བཟོད། །སྟག་མོའི་སྡེར་མོ་འཕྲེང་ཉིབའི་རོལ་པས་ལུ་པྱ་ས་ནཱ་ཙཀྵེསྦི་ལཱི་ཀྵ་ཎ་མ་ལག་་ཏ་བི་ཀཱ་ཏཱ་སྱ། །རེ་མཱཉྩ་རྣམ་པར་བཅོམ་གྱུར་པ། །དེ་ཡི་སྙིང་གའི་གཞི་ནི་རྣམ་པར་ཉམས་པ་སྐད་ཅིག་མཚོན། །ལུས་ནི་སྤུ་ལོང་ཙཙླི་ཏ་ཏ་ནོསྡུ་ཧི་ཚྭ་ཤུ་ཤུ་བྷ་ས་ཏྭཱ་པྲ་ཀཱ་ཤ་ཀི་ར་ཎྞཾ་ཀུ་ར་གིས་བྱུགས་བསིལ་ཟེར་ཅན་ལྟར་དཀར་བ་ཡིས། །སྙིང་སྟོབས་རབ་གསལ་འོད་ཟེར་མྱུ་གུས་ཁེངས་པར་པཱུ་རི་ཏེ་བ། །ཏ་སཱུ་མི་ཥ་ཧ་ར་ཎ་ཤོ་ཎི་ཏ་པཱ་ན་མཏྟཾ་བྱཱ་གྷི་ས་ཧརྟ་གྱུར་པ་བཞིན། །ཤ་དག་ཟ་ཞིང་ཁྲག་འཐུང་མྱོས་པར་གྱུར་པའི་སྟག་མོ་ནི། །དགའ་དང་བཅས་པ་མཐོང་མ་བ་ལོ་ཀྱ་ཡ་ཏཉྩ་ཀཱ་ར། །དིནྡྷ་པྲ་པཱ་ས་སཱ་མ་ཡཱ་ཀུ་ལི་ཏཱ་མུ་ཧཱུཎྜ་ཀནྟཱ་ནས་འགྲོ་བ་དེ་ཡི་རང་གི་སྲོག། འཚོ་བ་དག་ནི་རིང་དུ་འགྲོ་བའི་དུས་སུ་བྲལ་གྱུར་པ། །ཡུད་ཙམ་ཞིག་བ་ལམྦ་ན་དྷཾ་ཏི་ནི་ཛ་ཛཱི་བ་བྲྀཏྟི། །ཀྲྀཥཱ་པྲ་དཀྵི་ཎ་བི་ལཱ་ས་ག་ཏཱ་ན་མགྲིན་པ་ཉིད་ལ་བརྟེན་ཏེ་བརྟན་པར་བྱས། །སྲེད་པའི་རྗེས་མཐུན་རྩེ་དགའི་འོད་དང་འགྲོ་བ་དག་ག་ཏེ་ན་ལཛྫ་བ་ཤཱ་དི་བ་ཤྲཱི་ཐཾ་བི་ན་ཏཱ་ན་ནེ་བ། །ཏ་སྱཀ་རོ་གིས་ནི། །ངོ་ཚའི་དབང་ལས་བཞིན་ནི་མཆོག་ཏུ་རྣམ་པར་དུད་@པ་བཞིན། །ཁྱིམ་ཐབ་ལས་ནི་ཕྱིར་—།
