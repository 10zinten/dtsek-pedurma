བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀིསཱ་པ་རཱ་བྷ་བ། །ཨི་ཏི་ཐུ་ཏྭཱ་བྷ་ག་བ་ཏཔྲ་ཤ་མ་བྷི་མ་ཏཾ་བ་བརྟན་པས་རང་ཟིལ་གནོན། །དེ་སྐད་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས། །རབ་ཞི་མངོན་པར་བཞེད་ཐོས་ཙ༑། བྷཾ་ཤ་མ་བྷརྦ་ཚྭ་རཱ་ཛཱ་ཙ་ཀཱ་རཱཉྩརྱ་དན་ནེ། །ཏ་ཏཀྲྀཙྪཱཏ་ནས༑། རྒྱལ་པོས་ངོ་མཚར་བསྟན་པ་ལ། །མཆོག་ཏུ་མངོན་པར་ཞུ་བ་བྱས། །དེ་ནས་ཚོགས་ཀྱི་བྷ་ག་བ་ཏཱ་ཀྲྀ་ཏཱདྷུ་པ་ག་མོ་ཚྭ་པ། །རཱ་ཛ་དྷཱ་ནི་ཡ་ཡོ་ཧྲཋསཔ་བཅོམ་ལྡན་གྱིས། །ཞལ་བཞེས་མཛད་པས་མི་ཡི་བདག། མགུ་བས་ཞག་བདུན་དུས་བཏབ་སྟེ། །རྒྱལ་ཧཱ་བ་དྷི་སཾ་བི་དཱ། །ཨསྨིནྣབ་ས་ར་བྷཱ་ཏཱ་བྷུ་མི་བྷཞུ་ར་པོ་ད་ར། །ཙ་པོའི་ཕོ་བྲང་སོང་བར་གྱུར། །སྐབས་དེར་ས་ཡི་བདག་པོ་ཡིས། །མ་གཅིག་མིན་པའི་སྤུན་ཟླ་ཞིག། ཕོ་ཙཱ་རཱནྟཔུ་རོ་པཱནྟེ་པྲ་སཱ་ད་ཏ་ལ་བརྞ་ནཱ། །ས་ལཱི་ལ་བྲ་ཛ་ཏསྟ་སྱ་བྲང་འཁོར་དང་ཉེ་བ་ཡི། །ཁང་བཟང་གཞི་ཡི་ལམ་ནས་རྒྱུས། །ལས་ཀྱི་རླུང་གིས་བསྐྱོད་པ་བཞིན། །ཀརྨ་བཱ་ཏི་རི་བེ་རི་ཏཱ། །ཀུ་སུ་མ་སྲཀྵ་བྷཾ་ཏཾ་ཤེ་རཱ་ཛ་པཏྟཱི་ཀ་རསླུ་རྒྱལ་པོའི་བཙུན་མོའི་ལག་ནས་འཕོས། །རྩེ་དགའ་དང་བཅས་འགྲོ་བ་དེའི། །ཕྲག་པར་མེ་ཏོག་འཕྲེང་བ་ཏྭཾ༑། ཏ་སྱ་བི་ཛཱ་ཏ་དོ་ཥ་སྱ་དོ་ཎཾ་སཾ་བྷཱ་བྱཱ་པཱཀྵི་བྷི། །པི་ཤུ་ནཱཀིཾ་བ་ལྷུང་། །སྐྱོན་ནི་རིག་པར་གྱུར་པ་དེའི། །སྐྱོན་རྣམས་དབང་པོས་བསལ་བྱས་ནས། །ཕྲ་མ་མཁན་གྱིས་དནྟི་ཏྭཾ་ཙ་ཀྲི་རེ་རཱ་ཛ་གཱ་མི་ནཱ། །ཙྪི་ཏྲ་མལླ་མ་པི་པཱུ་པྱ་ཀླུ་དཱ༔སརྦ་པ་གཏམ་དེ་དག། རྒྱལ་པོ་ཡིས་ནི་རྟོགས་པར་བྱས། །རྗེ་བོའི་བསམ་པ་སྟོང་པར་ནི། །ཆུང་ཡང་བུ་ག་ཀཱ་རི་ཎ། །དྲི་ཛི་ཧྭཔི་ཤུན་པྲ་བི་ཤན་པྲ་བྷཱུ་ནཾ་ཤུ་རྩ་མཱ་ཤ་ཡཾ། །པི་ཤུ་ཎ་པྲ་རྙེད་པར་གྱུརཡི། །གནོད་པ་ཀུན་བྱེད་ལྕེ་གཉིས་པ། །ཕྲ་མ་རབ་ཏུ་འཇུག་པ་ཉིད། །རྒྱལ་པོ་ཕྲ་མས་རི་ཏོ་རཱ་ཛཱ་བྷཱ་ཏུ་རཱི་བི་ཁོ་ལཱ་ཎ། །ཙྪེ་དཾ་པཱ་ལཱ་དི་ཤ་ཤུ་པཱ་ཎི་པཱ་ད་སྐྱ་བསྐུལ་བ་ཡིས། །ཕྲག་དོག་དུག་རྒྱས་རྒྱལ་གྱུར་པས། །སྤུན་ཟླའི་ལག་པ་རྐང་པ་དག། མྱུར་དུ་ཆོད་ཅེས་མཱུཙྩི་ཏ། །ནི་ཀྲྀཏྟ་པཱ་ཎི་ཙ་ར་ཎཀུ་མཱ་རཿཀརྨ་བི་བླ་བཱཏ། །པ་ཡང་དག་བསྒོས། །ལག་པ་རྐང་པ་རྣམས་ཀྱི་བཅད་པའི། །གཞོན་ནུ་ལས་ཀྱི་རྣམ་བསླད་པས། །གསོད་།
