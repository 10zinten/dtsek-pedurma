བ་ད་ན་དྷ་ན་ནི་ལ། །ནི་ཤྲཱིསྣ་ནཱ་ཡ་སཾ་ཏུལྟསེརྩྭ་ཅན༑། །བཞིན་ལ་བསམ་གཏན་གཡོ་མེད་ཀྱིས། །གཉིད་ཀྱིས་ཕྲག་དོག་ཅན་གྱིས་བཞིན། །གཏང་ཡེ་བ་ན་ནི་དྲ་ཡཱ། །ཀ་རྩྭ་དན་ན་བྲྀཏྟཱནྟཾ་ཏ་ཏསྟེ་ན་ནི་བེ་དི་ཏཾ། །ཐུ་བ་དེ་ཡིས་མཚན་མོ་འདས། །མཛེས་མ་མཐོང་བར་གྱུར་པའི་མཐའ། །དེ་ནི་དེ་ཡིས་བཤད་པ་དག། ཕ་ཏཱ་པི་ཏྱཱ་སྱ་སཾ་ཀཱནྟ་ཙིནྟཱ་པ་རི་ཙི་ཏོན་བྷ་བཏ། །ས་ཏ་མུ་ཙ་བ་ཡཾ་པུ་ཏྲ་རཱ་ཡིས་ཐོས་ནས་དེ་ཡིས་ནི། །བསམ་པ་འཕོས་ཤིང་ཡོངས་འདྲིས་གྱུར། །དེ་ལ་དེ་སྨྲས་བདག་ཅག་བུ། །དྷོ་པྲ་པུ་ར་བཱ་སི་ན། །ས་ཀ་ཐཾ་ཧེ་དུ་ཧི་ཏ་རཾ་ཙཀྲ་བརྟི་པྲ་དཱ་སྱ་ཏི། །རྒྱལ་པོའི་གྲོང་ཁྱེར་གནས་པའང་མིན། །འཁོར་ལོས་སྒྱུར་དེ་བུ་མོ་ནི། །ཁྱོད་ལ་ཇི་ལྟར་སྟེར་བར་འགྱུར། །ཨཱ་ཤཱཀཾ་ནི་བ་ཀུཊྚནྟི་ས་མཱི་ཧནྟེ་ན་དུརླ་བྷོ། །ཨ་སཾ་བྷ་ཧུཾ་ན་བྷཱ་ཥསྟ་མཱ་མཁས་པ་བཀུར་བ་འདོད་པ་རྣམས། །མི་ནུས་པ་ཉིད་བྱེད་མ་ཡིན། །རྙེད་པར་དཀའ་བ་འདོད་མ་ཡིན། ན་ཀཱ་མཱ་མ་ནཱི་ཏི་ཎ། །ཙུ་ཏ་ཙམྤ་ཀམྤི་ཥཱ་སཱ་དྷཱི་ན་པུ་ནི་རཱ་ད་རཿ། །ཙིནྟ་མི་བདེན་པ་ནི་སྨྲ་མ་ཡིན། །འཁྲི་ཤིང་ཙུ་ཏ་ཙམ་པ་ཀ། རང་དབང་རྣམས་ལ་མི་གུས་པར། །ཡོངས་ཡན་རི་ཛ་ཏ་སྱ་ལ་ཏྭཾ་ཤུ་་ཏིཥབླ་ད། །ཏ་བ་ཏ་སྱཱཉྩ་སམྲཛྙཱའདུའི་འཁྲི་ཤིང་སེམས་པ་ན། །རྐང་དྲུག་པ་ཉིད་བསྐམ་པར་འགྱུར། །གལ་ཏེ་ཁྱོད་དང་དེ་གཉིས་ཀྱང་། །པཱསྔརྨ་བི་ཧི་ཏོ་ཡ་དི། །ཏ་ད་བ་ཐཾ་བྷ་བ་ཏྲེ་བ་ནིཔྲ་ཡཏྣ་ཕ་ལོ་ད་ཡ། །སྐྱེ་བ་སྔ་མར་འབྲེལ་བསྒྲུབས་ན། །དེ་ཡི་འབྲས་བུ་འབད་མེད་དུ། །ངེས་པ་ཁོ་ནར་རྒྱས་པར་འགྱུར། །ཨཱ་ཤཱ་སཱ་ཤི་ར་ནཱ་ཀྲྀཝཾ་བི་ཙཱ་རི་ར་ཀ་ཧཏྲི་ཏཾ། །པྲ་ཡཱནྡྷ་བྷ་རི་ར་ཤཱནྟཾ་རེ་བའི་ཞགས་པས་མ་བཀུག་ཅིང་། །རྟོག་དཔྱོད་དག་གིས་མ་གཟིར་ལ། །འབད་པའི་ཁུར་གྱིས་མ་ངལ་བི་དྷཏྟེ་བྷ་བི་ཏ་བྱ་ཏཱ། །ཨི་ཏྭཱ་ཀཙྩོ་པི་ཏུཐཱ་ཀྱི་ཏཏྟ་ཐེ་ཏི་བི་བར། །ངེས་པར་མྱོང་བྱ་གྲུབ་པ་ཉིད། །ཅེས་པ་ཕ་ཡི་ཚིག་ཐོས་ནས། །དེ་དེ་བཞིན་ཞེས་རྣམ་ཙིནྟ་ཡན། །ན་ཙེ་ཏཀཱནྟ་ཀཱ་ནཱི་ཏཾ་པྲ་ཏནེ་ཏུཾ་ཤ་ཤཱ་ཀ་ས། །ས་ག་བསམས་ཏེ། །སེམས་ནི་མཛེས་མས་བཟུང་བ་དག། སླར་ཡང་ཟློག་པ་དེས་མ་ནུས། །དེ་སོང་།
