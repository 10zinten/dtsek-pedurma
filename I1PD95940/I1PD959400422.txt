ལོ་ནཾ་སུ་ཁ་སམྤ་དཾ། །ཨི་ཏུཏྟཱ་ས་པི་ཏུ་པྲ་སྱཤཱ་ས་ནཾ་ཙ་ར་ཎཱ་ན་ཏ། །ཕུན་ཚོགས་སྐྲག་པ་སྐྱེད། །ཅེས་བརྗོད་དེ་ཡིས་ཡབ་ཀྱི་བཀའ། །ཐོབ་ནས་ཞབས་ལ་ཕྱག་འཚལ་ཏེ། །ཛ་ག་ཧེ་ཛ་ལ་དྷི་ལོ་ལ་ཀལྤ་ལཱ་ལིངྒཱི་ཏཱམྦ་རཾ། །བྷཱ་ཏཱ་ཏཾ་མ་ནུ་བ་བཱ་ཛ་སོ་ཆ་རླབས་གཡོ་བས་ནམ་མཁའ་ལ། །འཁྱུད་པ་ཆུ་ཡི་གཏེར་དུ་ཞུགས། །ཡོན་ཏན་མེད་པ་ཡོན་ཏན་ལ། །ཛ་རྒྱ་བྱ་ཛ་ནོཉནུ་ཛ། །ནིརྒུ་ཎས་གུ་ཅུ་དྭེ་ཥནྡོ་ཧ་མནྟཤྲཱི་ཙིནྟ་སྡང་བས་ནང་དུ་འཁུ་སེམས་ཤིང་ཅི། །གཞུང་བཟང་དུ་སྟོན་རྗེས་སྐྱེས་པའི། །སྤུན་ཟླ་དེ་ནི་དེར་རྗེས་ཡན༑། ཛེཥྛཏམ་བྲ་བཱིདྡཏྠ་ས་མུ་དྲེ་ཀརྨ་བི་སླ་བཏ། །ས་ཧཾ་གྲྭ་རྒྱ་དོགས་འབྲང༌། །ཐུ་བོས་དེ་ལ་སྨྲས་པ་བུ། །རྒྱ་མཚོར་ལས་ཀྱིས་རྣམ་བསླད་པས། །གཟིངས་ནི་ཞིག་པར་གྱུར་ཡཱསྐནྟི་ཛཱ་ཏེ་པྲ་བ་ཧ་ཎ་ཀཱ་ཡེ། །ཨི་ཏི་བྷཱ་ཏཱ་ཀྲྀ་ཏཱ་ཤྭ་སསྟ་ཐེ་ཏུ་པ་ན། །ཁྱོད་ཀྱི་ཕྲག་པར་བདག་བླང་བྱ། །ཞེས་པས་སྤུན་ཟླ་ཡིད་བརྟན་བྱས། །དེ་བཞིན་ཞེས་ཙེ་ས་ཏཾ་ཤ་ཋ། །དོ་ཥ་བྱ་ཏཔྲ་ཎ་ཡི་ཏཱཾ་ཁ་ལས་མ་མ་ལམྦ་ཏེ། །ཏ་ཏདེས་དེ་ལ་སྨྲས། །སྐྱོན་རྩོམ་གཡོ་ཅན་མི་བསྲུན་པ། །རབ་དུད་ཉིད་ལ་ཡང་དག་བརྟེན། །དེ་ནས་པྲ་བ་ཧ་ཎཱ་རུ་ཎྷས་པ་ནི་ར་ནུ་ཀུ་ལ་ཏྭཾ། །ཡཱ་ཏིཔུ་ཙྩེ་རི་བ་པྲ་སྱ་གཟིངས་ལ་ཞོན་བྱས་ཏེ། །བསོད་ནམས་བཞིན་དུ་རྗེས་མཐུན་པར། །འགྲོ་བའི་རླུང་གིས་རིན་ཆེན་གླིང་། །རxདྲི་པཾ་ཛྲ་པཱཏྨ་ཛ། །དི་བྱ་རཏྣ་ནི་སཾ་པྲ་པྱ་ཏྭཱ་བྲྀཏྟེ་ཏ་ཏཤ་མི་ཡི་བདག་པོའི་སྲས་ཀྱིས་ཐོབ། །རིན་ཆེན་མཆོག་རྣམས་རབ་རྙེད་ནས། །རྒྱལ་པོའི་སྲས་དག་དལ་བུ་ནི༑། རཱ་ཛ་པུ་ཏྲེ་པྲ་བ་ཧ་ཎི་དཱ་ག་བ་བྷཉྫ་པྲ་བྷཉྫ་ན། །བྷངྒེ་པྲ་པ་ཧ་ཅེ་ཏ་སྨིན་ཡིས༑། སླར་ལོག་ཚེ་ན་གཟིངས་དག་ནི། །རླུང་གི་སྟོབས་ཀྱིས་མྱུར་དུ་ཞིག། གཟིངས་ནི་ཞིག་པ་དེ་ཡི་བྷཱ་ཏ་རཾ་པཱུཏྤ་སཝཱི་དཱ། །ས་ཛ་གྲྭ་ཧ་ཤ་ཋཀཋེ་ཏཾ་བྷུ་ཛཾ་ག་ཨི་བཱ་ནུ་ཚེ། །སྔོན་གྱི་གྲོས་ཀྱི་སྤུན་ཟླ་དེའི། །མགྲིན་པར་ལག་འགྲོ་ལྟ་བུ་ཡི། །རྗེས་སྐྱེས་གཡོ་ཅན་དེ་ཡིས་ཛ༑། ཀརྨ་བཱ་ཏེ་རི་ཏསྱཱཎྞཾ་ཀཱུ་ལཾ་པཱ་དྱ་ཚྭ་པདྨ་ཛ། །ཨ་བཱ་པ་ས་ཧ་ས་བཟུང་། །ལས་ཀྱི་རླུང་གིས་མྱུར་བསྐྱོད་པས། །རྒྱལ་པོའི་བུ་ཡི་ངོགས་ཕྱིན་ནས། །ལོང་བའི་ཕོ་ཉ་དང་།
