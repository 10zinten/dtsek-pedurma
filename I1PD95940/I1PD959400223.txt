གྷོ་རེ་ཥཱ་ན་ག་རེ་ཥཱ་ས། ་། སྭརྒོདྡེ་ཤེ་ཥཱ་ཤི་ཀེ་ཥཱ་བི་མཱ་ནོདྷཱ་ན་ཤཱ་ལི་འཇིགས་སུ་རུང་བའི་གྲོང་ཁྱེར་དང༌། །མཐོ་རིས་ཕྱོགས་ནི་མ་ལུས་པ། །གཞལ་ཡས་ཁང་དང་སྐྱེད་ཚལ་ཥཱ༑། ཤཱནྟ་ཀཱ་ཙིནྡ་བི་ཤཱན་ཡ་ཏི་ནཱི་པྲ་ཎི་ཡཱ་ཏི་ནཱི། །པུ་གནས། །འཇིག་རྟེན་སྐྱོང་བའི་གྲོང་ཁྱེར་དུའང་ཚེ། །ངལ་ཞིང་གང་དུའང་ངལ་མི་གསོ། །ཕྱོགས་ཏྲ་མ་ནཱི་ཥྱ་ནཱ་པ་བྱཱ་ཏ་ལོ་ཀ་པཱ་ལ་པུ་རེ་ཥཱ་ཙ། །ཀུ་བེ་ར་སཱུ་ཐ་པ་ཙ་སཱ་ག་ནི་འཚོལ་ཞིང་རབ་སོང་སྟེ། །བཙལ་ཡང་བུ་ནི་རྙེད་མ་གྱུར། །དེ་ནས་ལུས་ངན་ཚིག་གིས་ནི། །བདེ་ཏྭཱ་ཙ་སུ་ག་ཏཱ་ཐ་མཾ། །བྷ་ག་བནྟཾ་བི་ཡོ་གཏྟ་ཤ་ར་ཎྞཾ་ཤ་ར་ཎཾ་ཡ་བར་གཤེགས་པའི་གནས་སོང་སྟེ། །བཅོམ་ལྡན་སྐྱབས་སུ་འོས་པ་ལ། །འབྲལ་བས་གཟིར་བ་སྐྱབས་སུ་ཡོ༑། ཏ་ཡཱ་ཏད་དུཁ་བྲྀཏྟཾ་ཏཾ་ས་ནི་ཤ་མྱ་ནི་བི་དི་ཏཾ། །ཏཱ་མ་བོ་སོང་། །དེར་ནི་དེ་ཡིས་རང་སྤྱོད་མཐའ། །ངེས་པར་བཤད་པ་དེས་ཡིགསན་ནས། །ཅུང་ཟད་འཛུམ་ཙ་ཏ་ཤོ་ཙིནྟཾ་ཀིཾ་ཙིཏ་སྨི་ཏ་སི་ཏཱ་དྷ་ར། །ཧཱ་རཱི་ཏི་ཏ་བཔུ་ཏཱ་ཎཱ་སནྟི་གྱི་དཀར་བའི་མཆུས། །མྱ་ངན་ཅན་མ་དེར་བཀའ་སྩལ། །འཕྲོག་མ་ཁྱོད་ལ་བུ་དག་ནི། །ཀྱེ་མ་བརྒྱ་པཉྩ་ཤ་ཏཱ་རྱ་ཧོ། །ཨི་ཡ་ཏི་ཀིཾ་བི་ཡོ་གཱརྟི་རེ་ཀསྨིནྡྷེ་བ་ཧཱ་རི་ཏི། ་། ཨི་ཏི་ཏེ་ཕྲག་ལྔ་དག་ཡོད། །གཅིག་ཉིད་ཁོ་ན་སྟོར་བ་ལ། །འདི་སྙེད་བྲལ་བའི་སྡུག་བསྔལ་ཅི། །ཞེས་དེས་ནོཀྟ་མཱ་ཀཙྪཱ་ཡཀྱི་དུཁཱ་་ཏཱ་བ་དཏ། །པུ་ཏྲ་ལཀྵ་པི་བྷ་ག་བན་བཀའ་སྩལ་སྡུག་བསྔལ་གྱིས། །ཉམས་པ་གནོད་སྦྱིན་མོས་ཐོས་སྨྲས། །བུ་འབུམ་ཡོད་ཀྱང་བཅོམ་ལྡན་ས་ཏྭཱ་ནི་ཀ་པུ་ཏྲ་ཀྵཏི། །པུ་ཏཱ་ཏ་པི་ཡ་ཏ་རཾ་ནཱ་རྒྱཏ་ཀིཾ་དུའདས། །བུ་གཅིག་བརླག་པ་ཡོངས་མི་བཟོད། །བུ་ལས་མཆོག་དགའ་གཞན་ན་མེད། །བུ་ཆགས་ཁཾ་ཏཏྟཱ་ཡཱ་ཏ་བ་རཾ། །པུ་ཏྲ་བཱ་ནེ་བ་ཛཱ་ནཱ་ཏི་པུ་ཏྲ་སྣེ་ཧ་པི་ཥ་བྱ་ཐཱ། །ས་ཧ་དུག་གིས་གདུང་བ་ནི། །དེ་བརླག་ལས་གཞན་སྐྱེསྡུག་བསྔལ་ཅི། །བུ་དང་ལྡན་པ་ཁོ་ནས་ཤེས། །རྒྱུ་མེད་ཛི་བ་སུ་ཏ་པཱི་ཏི་ར་ཀ་ར་ཎ་ཎི་བནྡྷ་ནཱ། །མ་ལི་ནོ་བི་ཀ་ལ་ཀཱི་ཎཀ་སྱ་ནེནུ་ངེས་པར་གདམས་པ་ཡི། །བུ་ཡིས་དགའ་ཉིབ་ལྷན་སྐྱེས་ཉིད། །མ་ཚང་རིད་དང་དྲི་མ་ཅན། །སུ་ཡི་བུ་ནི་།
