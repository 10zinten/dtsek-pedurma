བྱེ་ཏི་ས་མེ་བ་སུ་དྷཱ་སུ་དྷཾ་ཤུ། །ཐུ་ཏྲེ་ཏི་ཏེ་ན་ཀཱ་ཐི་ཏྭཾ་བྱ་དྷི་ཏདྲི་བདུད་རྩིའི་ཟེར་དེས་བདག་གི་གདུང་བ་འཕྲོག་པར་འགྱུར། །ཞེས་པ་དེ་ཡིས་བཤད་ཐོས་ནས། །ས་ཡི་ཐི་བཱི་པ་ཏི། །སཾ་ཀཱན་ནིཤྩ་ཏི་ཀ་ར་སཾ་ཏཱ་པས་མ་ཙིནྟ་ཡཏ། །བདག་པོ་གཟིར་བར་གྱུར། །བཅོས་ཐབས་མེད་པའི་གདུང་བ་དག། འཕོས་པ་ཡིས་ནི་ཡང་དག་བསམས། །ཨ་ཧོཉནུ་ཏུཀྟ་རཱ་ཛུཉཧ་མ་ཀཱ་ལེ་སྨིནི་ཛཆ་་ནཱ། །ཙིནྟ་ཏཀྵཏྠ་ཀྱེ་མ་བདག་གིས་རྒྱལ་སྲིད་བཏང་། །དུས་མིན་འདིར་ནི་གཉིས་སྐྱེས་ཀྱིས། །བསམས་ཏེ་བཀྲེས་པས་རཱི་ཏེ་ན་ཤུཀྵ་བྲྀཀྵ་ཨི་བཱ་དྷཱ་ནི། །དྷུ་རཱ་དྷཱ་ཤྲ་མ་བི་ཕ་མྱ་སནྟ་པ་པྲ་ཉེན་པ་ལས། །ལམ་གྱི་ཤིང་ནི་སྐམ་པོ་བཞིན། །སློང་བ་རྣམས་ལ་ལམ་རིང་གིས། །།ངལ་བ་འབྲས་ད་མཟི་ནཾ། །སཾ་མོ་ཧ་ཛ་ན་ནཾ་དྷིངྒཾ་མྲྀ་ག་ཊཾཥཱ་ཛ་ལོ་པ་མོ། །མུ་ཁཱཧྥ་མེད་གདུང་བ་སྟེར༄༅། །རི་དྭགས་སྲེད་སྐྱེད་ཆུ་ལྟ་བུ། །རྨོངས་པ་སྐྱེད་བྱེད་བདག་སྨད་དོ། །ཁ་ལ་རྡོ་པཱ་ཏཱ་ཏུ་ལྕེ་ན་མཱུར་ཙྪི་ཏཱ་ནཱ་མི་བཱཟི་ནཾ། །ཨཱ་ཤཱ་བྷཾ་གེ་ན་གུ་རུ་ཏྭཾ་བསྣུན་མཚུངས་པ་ཡིས། །བརྒྱལ་བར་གྱུར་པཞིན་དོན་གཉེར་རྣམས། །རེ་བ་ཉམས་པས་ལམ་དུ་ནི། །པྲ་ཡཱ་ཏྱཱ་དྷཱ་པ་རི་ཤྲ་མ། །ཨ་ཧཾ་ས་རཱ་ཛཱ་སམྦུཀྟ་རཱ་ཛུཀཱ་ན་ན་ཤི་ཏ། །ཡོངས་ངལ་ལྕི་བ་ཉིད་དུ་གྱུར༄༅། །རྒྱལ་པོ་དེ་ནི་བདག་ཡིན་ཏེ། །རྒྱལ་སྲིད་བཏང་ནས་ནགས་ཚལ་བསྟེན། །ཨི་ཏི་ཐུ་ཏཱི་པ་བི་པྲ་ཡཾ་ས་བྱསྐྱ་ཛ་ཏི་ཛཱི་བི་ཏཾ། །རཱ་ཛདྷཱ་ནཱི་མི་ཏོ་ག་ཏཱ་ཞེས་པ་ཐོས་ནས་བྲམ་ཟེ་འདི། །འཕྲལ་ལ་སྲོག་ནི་གཏོང་བར་འགྱུར། །འདི་ནས་རྒྱལ་པོའི་གནས་ཕྱིན་མཱ་མ་པྲ་སྱ་བི་ཙིནྟི་ཏཾ། །ཀཾ་ཀ་རི་ཎྱ་ཏི་སནྣཱ་པཱ་ད་ཡཾ་བྷསྨཱ་མ་ནོ་ར་ཏེ། །བསམ་པ་དག་ནི་མ་ཐོབ་ནི། །ཡིད་ལ་རེ་བ་ཉམས་པ་འདི། །གདུང་བས་ཅི་ཞིག་བྱེད་པར་ཐ༑། ཛཱི་བ་ཏེ་ཀཿཀི་ལ་ས་ཀུ་ཤ་ལི་གླེ་ཤ་ཀཱ་ལེ་ན་རཱ་ཎཾ་བྷཱ་ཤྲཱིཏྟ་འགྱུར། །སློང་རྣམས་ཕྱིར་ཟློག་བཞིན་ལེགས་ཚ་བའི་ཤུགས་རིངས་དག་གིས་གདུང་བྱེད་ཅིང་། །སྤུ་ར་ཏི་ས་ཧ་སཱ་ཡཔ་རི་ཏཱ་ཎ་བནྡྷ། །པྲ་ཏྭཱ་ཁྱུ་ནཱརྨ་ལི་ན་ཀུན་ཏུ་རྟོགས་པས་ལུས་དམན་ཆུང་དུར་བྱས་ཤིང་གང་ལས་མི་འགྲོ་བ། །།ཉོན་མོངས་དུས་སུ་མི་རྣམས་—།
