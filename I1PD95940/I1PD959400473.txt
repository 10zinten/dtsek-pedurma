སུ་ཀྲྀ་ཏཱི་ཛཱི་བ་ཏི་ཤ་བས་པཱ་པསྐུ་མྲྀ་ཏོཉམྲྀ་ཏ། །པྲཀྲུ་ཅུ་བྷགྱུ་ཏེ་མཱུ་ལཱཏ་ས་ལེགས་བྱས་དག་གི་རོ་ཡང་གསོན། །སྡིག་ལྡན་ཤི་བ་མིན་ཡང་ཤི། །བསོད་ནམས་སྔ་མའི་རྩ་བ་སྤྱོད། །དྷ་ནི,་རྒྱ་རཱ་ཤི་བཏ། །ཨི་ཏྭཱ་ཤ་ཡཱརྨ་ཡཱ་ཏ་དུཏྟཾ་བ་ཙ་ནཾ་པྲ་ནོར་ལྡན་རྣམས་ཀྱི་འབྲུ་ཕུང་བཞིན། །ཞེས་པའི་བསམ་པས་བདག་གིས་ནི། །ཚིག་ནི་གསུམ་པོ་དེ་དག་ཡོ༑། ཨི་ཏི་ཐུ་ཏྭཱ་ཀཱི་ཏི་པ་ཏིཏཾ་པ་རི་ཥ་ཛུ་སཱ་ད་ར། །ཨུ་ཙི་ཏཾ་ཀྲི་བརྗོད། །ཅེས་པ་ཐོས་ནས་ས་ཡི་བདག། དེ་ལ་ཡོངས་འཁྱུད་གུས་བཅས་པས། །བུ་གཅིག་དགེ་བའི་ཡ་ཏྭཾ་པུ་ཏྲ་ཀུ་ཤ་ལཱ་ཡེ་རྱ་བྷཱ་ཥ་ཏ། །ཏ་ཏས་པི་ཏཱ་ནུ་ཙཱ་ཏསཱ་ཐུ་ནེ་ཏྲེ་སླད་དུ་ནི། །རིགས་པ་མཛོད་ཅེས་རབ་ཏུ་སྨྲས། །དེ་ནས་ཡབ་ནི་མིག་ཆུ་དང༌། །བཅས་པས་རྗེས་ཎ་ཀཱ་ན་ནཾ། ་། པྲ་ཡ་ཡོ་མནྟི་པུ་ཏཱ་ཎཱ་ས་ཧི་ཏཔཉྩ་བྷིཤ་ཏི། །མ་ཧནྟོ་གནང་ནགས་ཚལ་དུ། །བློན་པོའི་བུ་ནི་ལྔ་བརྒྱ་དང༌། །ལྷན་ཅིག་རབ་ཏུ་སོང་བར་གྱུར། །དེར་ནི་དྲང་རནྟི་ཀེ་ཏ་ཏྲ་པྲ་བྲ་བྷཾ་པཱ་པ་ས་ནུ་ག། །ཏེ་ཊཱ་ཀཱ་ལེ་ན་སོ་པ་བྱ་སྲོང་ཆེ་དྲུང་དུ། །རྗེས་འབྲང་བཅས་པས་རབ་བྱུང་ཐོབ། །དུས་ཀྱིས་དེ་དག་རིལ་བ་དང་། །ཤིང་ཤུན་ཀུཎྜ་བལྟ་ལ་སཉྩ་ཡཾ། །ཏ་ཏས་སཾ་ཙ་ཡ་དྭེ་ཏཱི་ཏ་ད་དན་ན་སཾ་བི་དཱ། །ཨེ་བསགས་པ་དེ་ཡིས་མཐོང་། །དེ་ནས་བསགས་པ་ལ་སྡང་དེས། །དེ་དག་མི་བལྟར་ངེས་བྱས་ནས། །བློ་ཀཱ་ཀཱི་བི་ཛ་ནེ་ཏ་སྦོ་ཀིཏ་ཀཱ་ལཾ་མ་ཧཱ་མ་ཏི། །དན་ནཱ་བྷཱ་ཥ་ཎེ་བདྷ་ནི་གྲོས་ཆེན་པོས་ཇི་ཙམ་དུས། །སྐྱེ་བོས་དབེན་པར་གཅིག་པུར་ེབསྡད། །མཐོང་དང་སྨྲ་བ་ངེས་པ་ཡིས། །ཡ་མོ་པི་ཡ་ཛྲཙྪཱ་ཡཱ། །པྲཏྟཾ་སུ་ག་ཏ་མི་ཏྱཱཀྟ་པ་པྲཙྪཀུ་ཤ་ལཾ་བསྡམས་ཀྱང་འདོད་དགར་འོངས་པ་ཡི། །རི་དྭགས་ལ་ནི་དགེ་བ་དག། ལེགས་པར་འོངས་ཞེས་སྨྲས་་གཾ། །པུ་ནཉྩ་པཱུ་ཛི་ཏཾ་དྲྀབྷྱཱ་མུ་ནིནྟེ་ན་མཾ་ག་བྲ་ཏཾ། །ཨ་མཱ་ཏྱ་ནས་དྲིས། །སླར་ཡང་ཐུབ་པ་རི་དྭགས་ཀྱི། །བརྟུལ་ཞུགས་དེ་ཡིས་མཆོད་མཐོང་ནས། །བློན་པོའི་བུ་ཏ་ན་ཡཱསཊྚེ་བི་ལཀྵས་མ་ཙིནྟ་ཡན། །མྲ་གོ་མྲ་ག་བྲ་ཏཝཱ་ཡཾ་པཱུ་རྣམས་ཐམས་ཅད་ནི། །སྐྱེངས་པར་གྱུར་པས་ཡང་དག་བསམས། །ནགས་སུ་ཡོངས་འཛིན་མེད་པ་ཡི། །།
