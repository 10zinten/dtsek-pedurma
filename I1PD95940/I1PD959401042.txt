བསྟན་འགྱུར། འཁྲི་ཤིང་། །ཁེལུ་ཨ་ཡ་མཱ་ཙཱ་ཏ་ཀོཎྜ་སྐྱོ་བྷིཀྵ་རཀྵཎྜཱ་དཱི་དྷི་ཏི། །ཨཱ་ཡ་ཏནྟྲ་ཀྲ་མེ་ཎཱ་ཐ་འོད་ཟེར་དམན་པ་མིན་པ་འདི། །ཉི་མ་དག་དང་མཚུངས་པར་མཐོང་། །དེ་ནས་རིམ་གྱིས་ཤིང་རྟ་ཡིས། །ར་ཐེ་ཥཱ་ཤྭ་ཤུ་རཱ་ད་ཡ། །ཀི་མ་ཡཾ་ཀི་མ་ཡཾ་བྷ་དྲེ་བྷ་ག་བཱ་ནི་ཏི་ཧཱུྃ་ཛ་གུ། །བྱོན་པའི་ཚེ་ན་ཆོས་སོགས་ཀྱིས། །བཟང་མོ་བཅོམ་ལྡན་འདི་འམ་ཅི། །འདི་འམ་ཅི་ཞེས་རབ་ཏུ་སྨྲས། །སཱ་བྲ་བཱིནྣི་བ་ཤཱསྟ་ཡ་མེ་ཏེ་ཏཙཱ་ས་ནོ་ཙི་ཏཱ། །བྷིཀྵ་བཔྲ་ཤ་མ་ཟླ་དེས་སྨྲས་འདི་ནི་སྟོན་པ་མིན། །འདི་དག་དེ་ཡི་བཀའ་ལ་འོས། །དགེ་སློང་རབ་ཞི་བསྔགས་པར་འོས། །གྷཱསྟ་པོ་དཱིཥ་ཏ་ར་ཏཱི་ཥ། །ཡཀཱནྟ་ཧེ་མ་དྲུ་མ་ར་མྱ་ཤི་ལ་ཤཾ་གཱ་བི་དྷུ་ཆ་པུ་ར་དཀའ་ཐུབ་རབ་འབར་འོད་ཟེར་ཉིཅན། །གསེར་ལྗོན་མཛེས་པས་དགའ་བ་རི་ཡི་ནི། །རྩེ་མོ་ལ་ཞོན་སྔོན་ཏོ་བྷཱུ་པི་ཏི། །ཨཱཉྩརྨ་ཀྲྀནྣཱཀྟ་ཨི་བ་པྲ་བྷཱ་བཤྲཱི་མཱརྨ་ཧཱ་ཀཱ་ཤཱ་པ་ཨེ་ཥ་ལ་བྱོན་པ་གང་། །ངོ་མཚར་བྱེད་པ་མཐུ་ནི་གཟུགས་ཅན་བཞིན། །དཔལ་ལྡན་དགེ་སློང་འདི་ནི་འོད་བྷིཀྵ༑། པཉྩ་ན་ན་སུནྡ་ན་མཱ་སྦི་ཏོ་ཡཔ་ཏོ་ཡ་ཛཱི་མཱུ་ཏ་སཾ་བྷི་ར་གྷོ་ཥ། །།སྲུངས་ཚེ། །ཆུ་ལྡན་ཆར་སྤྲིན་ཟབ་མོའི་འབྲུག་སྒྲ་ལྡནི། །གདོང་ལྔ་ཤིང་རྟ་དག་ལ་གནས་པ་གང་། །བྱོསྣ་ས་མ་བྱེ་ཏི་གུ་ཎིཔྲ་ཎྜི་ཏས་ཨེ་ཥ་བྷིཀྵ་བྷུ་བི་ཤཱ་རི་པུ་ཏྲ། །།ཨཱ་རུ་མཁའ་ལ་སྔོན་དུ་བྱོན་པའི་དགེ་སློང་འདི། །ས་ལ་ཡོན་ཏན་གྱིས་བསྒྲགས་ཤཱ་རིའི་བུ། །གཉིས་འཐུང་སོ་རྒྱ་ཀི་ལ་ཤ་མི་བཱ་ཏི་ཤུ་བྷཾ་དཱི་པཾ་ཙ་ཏུརྡནྟ་མ་ནནྟ་ཀཱནྟི། ་། ཨཱ་ཡཱ་ཏི་ཡཔུ་བཞི་ཀེ་ལ་ཤ་བཞིན་དུ། །རབ་དཀར་ལ་ཞོན་མཐའ་ཡས་མཛེས་པ་གང༌། །བྱོན་པའི་དགེ་སློང་དེ་ནི་ཅུ་བ་ཏྭཾ་ཛ་གཏྠ་མོངྒ་ལྱཱ་ནཱ་མཱ་པ་ར་ཨེ་ཥ་བྷིཀྵུ། །བཻ་ཌཱུརྨ་ནཱ་ལཾ་ཀ་ན་ཀཱ་ར་བིནྡ་འགྲོ་རྣམས་ན། །བསོད་ནམས་ལྡན་མཆོག་མོ་གལ་བུ་ཞེས་བྱ། །གསེར་པད་བི་ཌཱ་རྱ་ཡི་ཡུ་བ་ཅན། །མཱ་རུ་རྒྱ་རཏྟཾ་ཀུ་ར་ཀེ་ཤ་རཾ་ཀཾ། །ཨུ་པེ་ཏི་ཡསོ་ར་བྷཱ་པཱུ་རི་ཏཱ་ཤས་ཨེ་ཥ་བྷིཀྵརིན་ཆེན་མྱུ་གུ་གེ་སར་གྱིས་མཚན་ཞོན། །ཕྱོགས་ནི་དྲི་བཟང་གིས་འགེངས་བྱོན་པ་གང་། །དགེ་སློང་དེ་པྲ་ཐི་ཏོ་ནི་རུ་དྷཱ། །པནྟཱ་མ་རཱ་གྲོ་ག་རུ་ཌཱ་དྷི་རུ་ཌྷཔཀྵ་ནི་ལོཏྲ་རི་ཉིད་མ་འགགས་པ་ཞེས་གྲགས། །གང་ཞིག་མཁའ་ལྡིང་ལ་ཞོན་གཤོག་རླུང་གིས། །།ཆུ་འཛིན་རབ་ཏུ་—།
