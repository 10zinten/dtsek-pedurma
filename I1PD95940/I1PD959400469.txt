ད་ཡཾ། །རཱ་ཛ་པུ་ཏྲཔྲ་བྲིདྡྷོ་སི་ནོ་ད་ཏིཥྛསྣ་ཙཱ་བ་དཏ། །དྲྀཥྛསྟ་ཏཥི་ཏཱི་ཤེ་ན་བི་ཐོབ་ཚེ། །།རྒྱལ་སྲས་རབ་ཏུ་ཆེར་གྱུར་ཀྱང་། །ལྡང་བ་མ་ཡིན་སྨྲ་བ་མིན། །དེ་ནས་དེ་ཡི་ལུས་སྐྱོན་གྱི?༑། སྟ་དངྒོདོ་ཥ་བྷེ་ཥ་ཛཾ། །ཨ་བ་ད་ན་བི་ཀ་ལྱཱ་རཱ་ཛ་སཱུ་ནོརྣ་དྲི་བྱ་ཏེ། །ཨ་བྷཱུ་སྨན་ནི་ས་ཡི་དབང་པོས་དྲིས། །སྨན་པས་སྨྲས་པ་རྒྱལ་པོ་ཡི། །སྲས་ལ་མ་ཚང་མཐོང་བ་མེད། །གལ་ཏེ་སཱ་རྱ་དི་ཛཱ་ཏོཉ་སྱ་དོ་ཁོཉ་ཡི་སུ་ཁ་སེ་བིན། །ཏ་དེ་ཥ་བྷ་ཡ་སཾ་བེ་གཱ་བདེ་བ་བསྟན་པ་འདིའི། །སྐྱོན་ནི་གོམས་ལས་སྐྱེས་གྱུར་པ། །དེ་ལྟ་ན་འདི་འཇིགས་པ་ཡི། །ཤུགས་དུཏྟིཥྛ་ཏི་ཙ་བསྟི་ཙ། །ཨི་ཏི་བི་དྱ་ར་བྷི་ཧི་ཏ་ཏ་ཐེ་ཏུཏྟཱ་ཀི་ཏཱི་ཤྭ་ར། །ལས་ལྡང་འགྱུར་སྨྲ་བར་འགྱུར། །ཞེས་པ་སྨན་པས་སྨྲས་པའི་ཚེ། །དེ་བཞིན་ཞེས་སྨྲས་ས་དབང་གི། །མི་ཕྱིརྦ་རྒྱ་བ་སུ་དྷཾ་བྷ་ཡཱ་ཡ་བྱ་སིཾ་ཛ་ཏ་སུ་ཏཾ། །ས་བྷཉྩ་མཱ་ནཔུ་རུ་ཥ་སྟ་བརྫུན་གྱིས་གསོད་པའི་ས་རུ་ནི། །འཇིགས་པ་བགྱི་སླད་བུ་བཏང་གྱུར། །སྨོད་བྱེད་སྐྱེས་བུ་དེ་དག་ལ། །ནུ་བཱ་ཙ་ར་ཐ་སྦི་ཏཾ། །ཨ་པི་ཀདྨ་ས་ཏྱཱ་སིཾ་བཱ་རཱ་ཎ་སྩོན་ན་བཱ་ཛ་ན། །ཨི་ཏི་ཤིང་རྟ་ལ་གནས་དེས་སྨྲས་པ། །བཱ་རཱ་ཎ་སཱི་འདི་ན་ནི། །སྐྱེ་བོ་འགའ་ཞིག་ཡོད་དམ་མེད། །ཅེས་པ་ཏདྲ་ཙ་ནཾ་ཐུ་ཏཱ་ཏིནྡྲི་ཏས་ནཱཾ་བནྟི་ཀཾ། །ཏ་ཏྲ་པི་ཏཱརྒྱུ་མཱ་ནོཔི་མཱུ་ཀ་དེ་ཡི་ཚིག་ཐོས་ནས། །དེས་དེ་མི་བདག་དྲུང་དུ་ཁྱེར། །དེར་ནི་ཡབ་ཀྱིས་གསོལ་བཏབ་ཀྱང་། །སླར་ཨེ་བཱ་བྷ་བཏ་པུ་ན། །པུ་ནརྦ་དྷ་བྷུ་ཁཾ་ནཱི་ཏཤ་བོ་དྲྀཏྟཱ་ཛ་གཱ་ད་ས། །ཨ་ཡང་ལྐུགས་པ་ཉིད་དུ་གྱུར། །སླར་ཡང་གསོད་སར་ཁྲིད་པ་ན། །རོ་ཞིག་མཐོང་ནས་དེས་སྨྲས་པ། །རོ་སྐྱེ་ཥ་ཛཱི་བ་ཏི་ཤ་བཀིམ་སནྣཱཏྨ་ནཾ་མྲྀ་ཏ། །ཐུ་ཏྭེ་ཏཏྟིཔི་ཏུཔཱནྟེནྱསྟོ་འདི་འཚོ་བ་ཡིན་ནམ་ནི། །བདག་ཉིད་ཀུན་གྱི་ཤི་བ་ཡིན། །དེ་ཐོས་ཡབ་ཀྱི་ངོགས་སུ་དེས། །བཀོད་མོ་ནི་བྱ་དྷཱཉྩི་ན། །པུ་ནཊྚདྷ་བྷ་ཡཱ་ཡི་བ་ནི་ཏཿཔྲ་བཱ་ཙ་ཏཱན་པ་ཐི། །ཚེ་སླར་ཡང་སྨྲ་བཅད་བྱས། །སླར་ཡང་གསོད་འཇིགས་ཉིད་སླད་ཁྱེར། །དེ་དག་ལ་ནི་ལམ་དུ་སྨྲས། །རཱ་ཤིརླ་ཨེ་ཥ་དྷ་ནྱ་སྱ་ས་ཧི་ཧཱུཾཀྟཉནུ་བྷཱུ་ཛ་ཏེ། །ཨི་ཏུལྟ་བཱ་ཀྱི་པི་པུ་ནརྣོ་ཙེ་གང་ཞིག་འབྲུ་ཡི་ཕུང་པོ་འདི། །གང་ཕྱིར་དེ་ཟོས་རྗེས་སུ་ཟ། །ཞེས་པའི་ཚིག་འདི་བརྗོད་ཀྱང་སླར། །—།
