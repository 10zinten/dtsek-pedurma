བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེ་ཡ་ཏྱ་ནི་ཤ་མདྡྷ་ཏ་རཱུ་པ་མེ་བ་ཀཱ་ལེནྡ་ཛཱ་ལ་ཀ་བ་དྷུབྷ་བི་ཏ་བྱ་པར་མྱོང་བྱ་དུས་ནི་མིག་འཕྲུལ་བྱེད་པའི་བུད་མེད་འདི། །རྟག་ཏུ་རྨད་བྱུང་རང་བཞིན་ཉིད་ནི་རབ་ཏུ་སྟོན་ཏེརྨ༑། ཨི་ཏི་ཙིནྟ་ཡ་ཏསྟ་སྱ་མོ་ནཱཀྐུདྷེ་དྷི་ཀཾ་དཱི་ཛ། །མཱུརྞཱི་པཱ་པ་མི་པར་བྱེད། །ཅེས་སེམས་དེ་ནི་མི་སྨྲ་བས། གཉིས་སྐྱེས་ལྷག་པར་ཁྲོས་གྱུར་ཏེ། །སྡིག་པ་བཞིན་དུ་སྦོམ་བསྒྱུ་ལཾ་ལ་གུ་ཌི་ས་མ་པཱ་ཏ་ཡཏ། །ཏཾ་བནྡྷ་ས་བྷཱུ་པ་ཏི་ས་བྷནྡི་ཏཱ་རཀྟ་པ་ཡི། །དབྱུག་པ་མགོ་ལ་ཡང་དག་བསྣུན། །ཁྲག་གིས་བསྒོས་ཏེ་སྔོན་སྐྱེས་དེ༄༅། །བཅིངས་ནས་ས་མ་གྲ་ཛ། །ཏ༴ཏྠ་ཚྭ་སཾ་ཤརྡརྱ་ཙཀྲེ་ཀཱ་རཱ་ཁཾ་ཧཱ་ཏི་ཐི། །བདག་མདུན་སར་ཁྲིད། །དེ་བཞིན་ཤ་དང་བཅས་པ་བསྟན། །བཙོན་ཁང་དག་གི་མགྲོན་དུ་བྱས། །ནིརྡོ་ཥ་ཀླུ་ཤ་མཐཱ་ཏི་ཛིནྡྷ་ཏེ་གུཔྟ་པཱ་ཏ་ཀ། །ཀཀ་སྱ་ཤུདྷི་ཛཱ་ནཱ་ཏི་སྐྱོན་མེད་དག་ནི་ཉོན་མོངས་སྤྱོད། །སྡིག་པ་སྦས་པ་དར་གྱུར་པ། །དབང་མེད་མཚར་བའི་རྣམ་པ་ལ། །ཙི་ཏཱ་ཀཱ་རེ་ཥཱ་སཱ་ཥཱ། །ནི་བདྷེ་བནྡྷ་ནཱ་གཱ་རེ་ཏསྨིནྟཱ་ལེ་ན་བིམྲྀ་ཏེ། །སུ་ཡི་དག་པ་སུ་ཡིས་ཤེས། །བཙོན་ཁང་དག་ཏུ་བཅིངས་པ་®དེ། །དུས་ཀྱིས་བརྗེད་པར་གྱུར་པ་ན། །པཱུ་པྱ་བཏྟཱ་དཱི་ཛཀིཉྩིནྣོ་བེ་དོ་རྒྱ་ལཛྫ་ཡཱ། །ཨ་ཐ་དྭཱ་ད་ཤ་བྷིནྡནྟི་བློ་མ་བེའུ་ཐོབ་ནས་གཉིས་སྐྱེས་ནི། །ཚུལ་ངན་ངོ་ཚས་ཅང་མི་སྨྲ། །དེ་ནས་ལོ་ནི་བཅུ་གཉིས་ན། །ནམ་བཱ་ཎཱི་བོ་བྷི་ཏ། །མུ་མོ་ཙ་བནྣཱ་ནཱདྷིཧཱུྃ་ཏཙྪི་ཥྛ་སྣ་རི་ཏོ་ཚིགས་མཁའི་སྐད་ཀྱིས་རྟོགས་བྱས་ཤིང་། །དེས་སློབ་ཀྱིས་སྦྲན་མི་བདག་གིས། །དགེ་སློང་བཅིངས་པ་དག་པ༑། ཀཱ་རཱ་གཱ་རཱཏྨ་ཤཱ་ཀཱ་རཿཀཾ་ཀཱ་ལ་ཨི་བ་དྷུ་ས་ར། །ཨཱདྷ་ཀེ་ལས་བཀྲོལ། །བཙོན་ར་དག་ལ་ཀེང་རུས་བཞིན། །རིད་པའི་རྣམ་པ་མདོག་མི་སྡུག། གོས་མད་སྒྲ་ཤོ་བི་བ་ས་ནས་པྲ་ཏ་ཨི་བ་ནིརླ་ཡོ། །ཨ་ཧོ་གཱ་བཱ་གྲ་ལེ་ཁེ་ན་ནི༔་ལཱ་ཀརྨ་སཾནྟ་ནི་གྱེན་དུ་ལངས། །དེ་ནི་ཡི་དྭགས་བཞིན་དུ་འཐོན། །ཀྱེ་མ་རྡོ་སྟེང་རི་མོ་བཞིན། །ལས་ཀྱི་རྒྱུན་ནི་གཡོ་ཏི༑། པཱབྷྱཱ་བྷི་དྷོ་པི་ཡཏྠ་པ་ཀླུ་ར་ཀླུ་ཤ་ཀ་དརྦ་ཏྭཾ། །ཏཱི་གྲྭ་ནུ་བ་མེད། །གང་ཕྱིར་མངོན་ཤེས་ཐོབ་པས་ཀྱང༌། །དྲག་པོའི་ཉོན་མོངས་སྡུག་བསྔལ་ཐོབ། །དེ་མཐོང་ས་—།
