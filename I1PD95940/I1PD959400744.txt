ཨ་ཐཱ་རྱ་ཛརྨ་རྱ་པི་བཱ་ལ་ཀོ་སོ་མུདྡྷཱ་བྷི་དྷཤེཊཱི་སུ་ཏཔྲ་མོ་ཧཱཏ། །ཙ་ཀཱ་ར་ཙི་དེ་ནས་སྐྱེ་གཞན་ལ་ཡང་ཚོང་དཔོན་བུ། །མུག་དྷ་ཞེས་ནི་བྱིས་པ་རབ་རྨོངས་ལས། །མཆོད་རྟེན་གྱི་ནི་ཏྱ་པྲ་ཏི་མ་ཛཾ་ན་མྱ་ཤ་སྦེ་ཎ་ལོ་ཙ་ནཱ་མཱ་ན་ནཱ༴། །ས་ཛཱ་ཏ་སཾ་ཛྷ་རྒྱལ་བའི་སྐུ་གཟུགས་ཀྱི། །བཞིན་རས་པདྨ་མཚོན་གྱིས་མིག་མེད་བྱས། །འདུ་ཤེས་བསྐྱེད་དེས་དེ་ཡི་ན་བེནྡྲ་ནཱི་ལཾ་མ་ཡཾ་བྱ་དྷཱ་དའི་ཡུ་གཾ་ས་ཏ་སྱ། །ཏ་ཏོ་རྒྱ་ཛམྦ་རྒྱ་པི་བི་ཙི་ཏེ་སཾ་སྤྱན་ཟུར་དག། ཨིནྡྲ་ནཱི་ལའི་རང་བཞིན་གསར་དུ་བསྒྲུབས། །དེ་ནས་སྐྱེ་གཞན་ལ་ཡང་མཆོད་རྟེན་སྐུ་ར་པཱུ་ཛཱས་སུ་ནཉྩ་ཀཱ་ར། །ནེ་ཏཱ་པ་ཧཱ་རེ་ཎ་བ་ནེ་ཝཾ་གཱ་ཎཾ་བཱ་ལྕེ་ནི༑། ཞིག་པ་གསོ་དང་མཆོད་པ་དག་ཀྱང་བྱས། །ནགས་སུ་རི་དྭགས་རྣམས་ཀྱི་མིག་ཕྲོགས་ཤིང་། །བ་ཙི་ཏྱ་པྲ་ཏེ་མཱ་ཀྱི་ལོ་པཱ་ཏ། །ཨ་བཱསྟ་བཱ་ནེ་ཥ་བི་ནཱ་ཤ་མ་ཚོརྫནྣཱནྟ་བྱིས་པས་མཆོད་རྟེན་སྐུ་གཟུགས་སྤྱན་ཕྱིས་པས། །རྒྱལ་པོའི་བུ་འདིས་སྐྱེ་བ་གཞན་རྣམས་དང་། །ད་ལྟ་རེ་ཥ་བྱེ་བ་རཱ་ཛ་པུ་ཏྲ། །ས་རནྡ་ནེ་ཏྲ་པྲ་ཏི་མཱཉྩ་ཀྲྀཏྭཱ་པྲཉྫི་བི་ནཱཏྟཾ་པུ་མིག་དག་རྣམ་པར་ཉམས་པ་ཐོབ། །རིན་ཆེན་མིག་ལྡན་སྐུ་གཟུགས་བྱས་པ་ཡིས། །རྣམ་པར་ཉམས་པའི་ན་རཱ་ས་པཱ་ད། །བི་བྷིནྡྷཱ་ཙི་ཏྱ་པྲ་ཏི་པཱུ་ར་ནེ་ན་པྲ་སཱ་དི་ཀཀཱིནྟཾ་མ་ཡ༔་མིག་ནི་སླར་ཡང་ཐོབ། །མཆོད་རྟེན་ཞིག་པ་རབ་ཏུ་རྒྱས་བྱས་པས། །རབ་ཏུ་དང་ལྡན་མཛེས་པའི་རང་ཛཱ་ཏ། །ཐོ་ཏཔཱཔི་ཕ་ལ་པྲ་དྲྀཏྟ་བི་མ་ལཱ་ལོ་ཀ་ཀྲ་མེ་ཎཱ་མུ་ནཱ་བཞིན་གྱུར། །རྒྱུན་ཞུགས་འབྲས་བུར་རབ་འཇུག་དྲི་མ་དང་བྲལ་སྣང་བའི་རིམ་པ་འདི་ཡིས་ནི། །བི་རཱ་གྱོ་ཛཱ་ལ་པ་ཏྲ་དན་ན་བི་དྷོ་ལ་ཐཱུ་དྷི་ཀཱ་རསྦི་ཏི། །སམྱཀླུ་ཅུ་ཆགས་བྲལ་རབ་འབར་བདེན་པ་མཐོང་བའི་སྒྲུབ་བྱེད་དག་ལ་གནས་པའི་དབང་གྱུར་ཐོབ། །དུས་ཀྱི་ཡང་བ་ཤཱདྷ་བི་ཥྛ་ཏི་ཤ་ནིཀཱ་ལེ་ན་སཾ་བུདྡྷཱ་ཏཱ་མི་ཏུཏྟཾསྦ་བི་རེ་ཎ་བྷིཀྵ་ནི་བ་དག་བསོད་ནམས་དབང་ལས་དལ་གྱི་རྫོགས་པའི་སངས་རྒྱས་ཉིད་དུ་འགྱུར། །ཞེས་པ་གནས་བརྟན་གྱིས་ཧཐུ་ཏྭཱ་བྷ་བདིསྨི་ཏ། །ཨི་ཏི་ཀྵེ་མེམྲ་བི་ར་ཙི་ཏཱ་གསུངས་དགེ་སློང་ཚོགས་ཀྱིས་ཐོས་ནས་ཡ་མཚན་ལྡན་པར་གྱུར། །ཞེས་ཨེཔ་དགེ་བའི་དབང་པོས་བྱས་པའི་།
