ཏམྦྷཏྟ་སྱ་པ་ཧཱ་སཱ་ཡ་ཀྲྀ་ཡ་ཏཾ་ཀཱ་པི་བཉྩ་ཏྭཾ། །མཱ་ཡཱ་མོ་ཧི་ཏཱ་དྷཱཎཱ་ནཾ་དེས་ན་ཉེ་བར་བགམ་པའི་སླད། །དེ་ལ་སླུ་བ་འགའ་ཞིག་བགྱི། །གཡོ་སྒྱུ་ཅན་རྣམས་སྒྱུ་མ་ཡིས། །པ་རོ་པི་ཏུ་་ཏི། །ཨི་ཏི་ཏེ་ནོཀྟ་མཱ་ཀཙྩ་ཤྲཱི་གུསྟཀརྨ་རྨོངས་ནས་གཞན་ཡང་ཡོངས་སམ་འགྱུར། །དེ་ཡིས་དེ་སྐད་བརྗོད་ཐོས་ནས། །ལས་ཀྱིས་རྨོངས་མོ་ཧི་ཏ། །པ་ཏི་ཏུཾ་པཱ་ག་ཤྭ་ཧྥེ་ཡུ་ཀྵ་ཏ་དུ་པ་དིཥཱ་ཡཱ། །པྲ་པའི་དཔལ་སྦས་ནི། །སྡིག་པའི་གཡང་ལ་མཆོང་བའི་སླད། །དེ་ཡིས་བསྟན་པའི་རིགས་པ་ཡིས། །ཁྱིམ་དཱིལྟ་ཁ་དི་རཾ་གཱ་རཱ་པུཎཾ་གཱ་ཌྷ་ཁ་དཾ་ཤྲཱི་ཧེ། །ཀྲྀ་ཏྭཱ་པ་བི་ཥ་མ་ལུཉྩ་ཡ་དུ་གསང་དོང་སེང་ལྡེང་གིས། །སོལ་བ་རབ་འབར་གྱིས་ཁེངས་དང་། །དུག་ལྡན་ཟས་ཀྱང་བྱས་ནས་ནི། །ཡོ་བྷ་ག་བ་ཏོནྟི་ཀཾ། ཏེ་ན་མི་བྱཱ་བིཧི་ཏ་ཡ་བྷཀྵ་བྷོ་ཀྱུ་ནི་མཎྜི་བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་སོང་། །བརྫུན་གྱིས་བསྐྲུན་པའི་གུས་པ་ཡིས། །དེ་ཡི་ཟས་ལ་མགྲོན་དུ་ཏ༑། བི་ཛཱ་ཡ་སཐཾ་སརྦ་ཛྙཱསྟ་ཐེ་ཏི་པྲ་ཧ་ས་སྨི་ཏ། །མི་ཋཱཀྐི་སོང་། །ཐམས་ཅད་རྣམ་རིག་ཀུན་མཁྱེན་གྱིས། །དེ་བཞིན་ནོ་ཞེས་འཛུམ་བཅས་གསུངས། །དུག་དང་ཡོ་ག་ཀུ་པི་ཏྭཾ་པཏྟཱི་སདྡྷརྨ་བཱ་དཱི་ནཱི། །གྲི་ཧེ་བ་བནྡྷ་ཤྲཱི་གུརཿཤཾ་ཀི་ཏོ་མེ་སྦྱོར་ལ་ཁྲོས་པའི། །ཁྱིམ་ཐབ་ཆོས་ནི་སྨྲ་བ་མོ། །གྲོས་ནི་ཟགས་སུ་དོགས་པ་ཡིས། །དཔལ་སྦས་ཀྱི་མནྡཱ་བི་ཤ་བཱཏ། །ཨ་ཐ་བི་ཛཱ་ཏ་པྲི་ཏོ་པི་བྷ་ག་བཱན་སཱ་ཡ་མཱ་ཡ་ཡོ། །ནི་ཁྱིམ་དུ་བཅིངས། །དེ་ནས་སྤྱོད་ཡུལ་མཁྱེན་གྱུར་ཀྱང་། །འགྲོ་བས་ཕྱག་འོས་གདོང་བཞི་སོགས། །བརྩྭ་མཱ་ནོ་ཛ་ག་བཤྩ་ཏུརྨུ་ཁ་མུ་ཁིསུ་རི། །ཤྲཱི་གུལྟ་ལྷ་རྣམས་ཀྱིས་ནི་ཕྱག་འཚལ་ཞིང་། །བཅོམ་ལྡན་རང་ཉིད་གཤེགས་པར་གྱུར། །དཔལ་སྦས་ཀྱིས་ནི་སྱ་ཏ་མཱ་རཾ་བྷཾ་བི་བེ་ད་ན་ག་རེ་ཛ་ན། །དིཀླུ་དྷཱ་བ་ཏི་པཱ་པཱ་ནཾ་སུ་གུསྟ་མ་བརྩམས་པ་དེ༄༅། །གྲོང་ཁྱེར་དག་ཏུ་སྐྱེ་བོས་རིག། སྡིག་ཅན་རྣམས་ཀྱི་སྡིག་པ་ནི། །ཤིན་ཏུ་སྦས་ཀྱང་པི་པཱ་ཏ་ཀིཾ། །ཏ་ཏཀཉྩིཏ་ས་མ་ཧྥེ་ཏ་བྷ་ག་བནྟ་མུ་པཱ་སི་ཀ། །ཨུ་ཕྱོགས་རྣམས་རྒྱུག། དེ་ནས་དགེ་བསྙེན་དག་ཅིག་ནི། །བཅོམ་ལྡན་འདས་ལ་མངོན་ཕྱོགས་ཏེ། །མེ་།
