བསྟན་འགྱུར། འཁྲི་ཤིང་། །ཁི་སབྷཱ་ཏུ་ར་བྱཱ་དྷི་ཧཱ་རི་ཡོ་བྷཱུཏྠ་ཏོ་ཥ་དྷ་བྷི་དྷ། ང་། སཊྚ་ད་དོ་ནད་ཀྱིས་གཟིར་བ་ཀུན་འཕྲོག་གང་། །སེམས་ཅན་སྨན་ཞེས་བརྗོད་གྱུར་དང་། །འདབ་ཆགས་བསྐྱབ་བྷ་བEཉྩ་ཁ་ག་ཏཱ་ན་སཱ་དེ་ཧ་ད། །ཙ་ཀྲི་གོ་པཱ་ལ་ནཱ་ག་སྱ་ཤཱནྟི་ལ་རང་ལུས་སྟེར། །ཐམས་ཅད་སྟེར་བར་གང་གྱུར་དང་། །གང་གིས་ཀླུ་ནི་བ་གླང་སྐྱོང༌། །གཟིགས་པ་ཀ་ར་དྲི་ཤི་བ་ཡ། །ཡཏྠ་བྷཱ་བེ་ན་བ་སུ་དྷཱ་རིནྡ་ཙི་ཏྟཾ་ཀི་ཏཱ་བྷ་བཏ། །ཉིད་ཀྱིས་ཞི་བར་མཛད། །གང་གིས་མཐུ་ཡིས་ནོར་འཛིན་ནི། །རིན་ཆེན་མཆོད་རྟེན་གྱིས་མཚན་ཨཱ་སཱིཀཱུ་ཙེ་བ་ལོ་ཡཉྩ་རོ་གཱནྟེ་ཏཱ་ཎ་བནྣཱ་བ། །ཀུ་ཎཱ་ལཤཱི་གྱུར༑། ནད་ཀྱིས་གཟིར་བ་སྐྱོབ་པའི་གཉེན། །བསོད་ནམས་སྟོབས་སུ་གང་གྱུར་དང་། །ཚུལ་ལྡན་ཀུ་ལ་བཱཙྩོ་བྷཱུཏ་ནཱ་ག་ཤནྟི་བྱ་དྷཱ་ཙཱ་ཡ། །ཀར་ཀཤྲཱི་པ་དོ་ཡོ་བྷཱུཏ་ཡ་ལ་ལར་གང་གྱུར། །གང་གིས་ཀླུ་ནི་ཞི་བར་མཛད། །གང་ཞིག་ཞིང་པ་བཀྲ་ཤིས་གྱུར། །ཞི་ཤོ་དཱ་ཁྱཉྩ་ཤནྟི་མཱན། །ཡ༔ཀཱ་བྱ་པཱ་ནུ་གྲ་ཧ་ཀྲྀཏ་པྲལྟ་བཱརླཉྩ་ལྡན་གྲགས་སྦྱིན་ཞེས་པ་དང༌། །གང་གིས་འོད་སྲུངས་རྗེས་བཟུང་མཛད། །གང་གིས་མི་འམ་ཀིནྣ་རཱི། །མུ་ནིརྨཉྩི་ཀ་ཤྲཱི་གི་བྷཱུཏ་ཀ་བི༔་ཀྱི་ཏི་པཱཏྨ་ཛ། །ཅི་མོ་ཐོབ། །་རྭ་གཅིག་ཐུབ་པར་གང་གྱུར་དང༌། །ས་སྐྱོང་བུ་ནི་སྙན་དངགས་མཁན། །ཡ༔སཾ་གྷ་ར་ཀི་ཏ་བྷི་ཧཱུྃ་ཙ་ཀཱ་རཱདྷ་ཏ་པ་དོ་དི་ཏཾ། །པཱ་ཧ་པདྨ་གང་ཞིག་དགེ་སློང་དགེ་འདུན་སྲུངས། །དགྲ་བཅོམ་གནས་སུ་བསྐྱེད་མཛད་དང་། །པདྨ་ཅན་གྱི་བ་ཏཱི་བྲྀཏྟཾ་དྷརྨ་རཱ་ཛསྦི་ཏཾ་ཙ་ཡ། །ཀབྷི་རེ་ཥཱ་བི་ཧཱ་རཱཙྩཾ་ཡརྨི་ཚུལ་རབ་གསུངས། །གང་ཞིག་ཆོས་རྒྱལ་རབ་ཏུ་གནས། །གང་ཕྱིར་ཉི་མ་གུད་པ་ཡིས། །ཁ་ཆེར་མཱ་དྷནྟི་ཀོ་བྱ་དྷཱཏ། །ཤཱ་ཎ་པཱ་སཱི་བི་ཧཱ་རཾ་ཡོ་མ་ཐུ་རཱ་ཡཾ་བི་ནིརྨ་གཙུག་ལག་ཁང་མཆོད་བསྒྲུབས། །ཤཱ་ནའི་གོས་ཀྱིས་བཅོམ་བརླག་ཏུ། །གང་ཞིག་གཙུག་ལག་ཁང་སྤྲུལ་མེ༑། ཨུ་པ་གུཔྟཱཿཧྥ་ཡ་སྭཱནྟཾ་བ་བྷཱུ་བཱ་ལཀྵ་ཎོ་ཛི་ན། །ཡཎྞཾ་ཏྭཱ་ཤོ་ཀ་དང༌། །གང་གི་མཇུག་ཏུ་ཉེར་སྦས་ནི། །བདུད་ལས་རྒྱལ་བར་གྱུར་པ་དང༌། །མི་བདག་མྱ་ངན་དྷ།
