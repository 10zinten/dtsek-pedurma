བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀི་བི་ཀྵེ་པ་ཤ་ཤ་ཀསྟ་ནཾ། །བཱརྟ་མཱ་ནོ་པི་ཡནྟི་ན་པྲ་ཎ་ཡཱཏ་མུ་གི་མེར། །ཐུབ་པས་ཀྱི་བྱམས་པས་ཡང་ཡང་དུ། །འབད་པ་ཡིས་ནི་བཟློག་གྱུར་ཀྱང་། །རི་བོང་གི་ནི་ནི་ནཱ་མུ་ཧུ། །ཏ་སྱ་ས་ཏཱ་པྲ་བྷཱ་བེ་ན་ཛཱ་ལ་ཛཱ་ལཱ་ཀུ་ལོ་ན་ལ། །པྲ་ལུས་པོ་འཕངས། །དེ་ཡི་སྙིང་སྟོབས་མཐུ་ཡིས་ནི། །འབར་བའི་ཚོགས་ཀྱིས་དཀྲིགས་པའི་མེ། །འཇམ་ཡ་ཡོ་མཉྫ་ཤི་ཛཱ་ན་བླ་མ་རཱ་བྷོ་ཛ་ཁཎྜ་ཏྭཾ། །སོ་པི་དི་བྱ་བ་པུ་སྟ་ཏྲ་ཀ་མ་པའི་སྒྲ་སྒྲོགས་བུང་བ་ཅན། །པདྨོའི་རྫིང་བུར་རབ་ཏུ་གྱུར། །དེ་ཡང་ལྷ་ཡི་ལུས་ཀྱིས་དེར། །པདྨ་ཆེན་པོ་ལེ་མ་ཧ་ཏི་སྦི་ཏ། །པྲ་ཎ་མྱ་མཱ་ནོ་མུ་ནི་ནི་བི་ད་དྷེ་དྷརྨ་དེ་ཤ་ནཾ། །ཨི་ཏི་བོ་དྷི་དག་ལ་གནས། །ཐུབ་པས་ཕྱག་འཚལ་ལྡན་པ་ལ། །ཆོས་རྣམས་རབ་ཏུ་བསྟན་པ་བྱས། །དེ་ལྟར་བྱང་པྲ་དྲྀཏྟཱ་ནཾ་ན་བ་དྷེརྣ་བི་ཥཱཏ་བ་ཡཾ། །བྷ་ག་བཱན་ཀ་ཐ་ཡི་ཏྲེ་ཏི་ཆུབ་ཞུགས་པ་རྣམས། །འཇིགས་པ་མེས་མིན་དུག་གིས་མིན། །དེ་སྐད་གསུངས་ནས་བཅོམ་ལྡན་འདས། །ཤྲཱི་གུཔ་བྷ་བ་ནཾ་ཡ་ཡོ། །ཏ་ཏྲ་ཏེ་ན་པྲ་བི་བྱི་བ་བི་ནི་ཀློསྟེ་དཀྵི་ཎེ་པ་དེ། །དཔལ་སྦས་ཀྱི་ནི་ཁང་པར་གཤེགས། །དེས་ནི་དེ་རུ་ཞུགས་ཉིད་ན། །ཞབས་ནི་གཡས་པ་བཞག་པ་ཡིས། །བ་བྷཱུ་བཱཀྐི་ཁ་དཱ་མཎྜཱུ་ཁཾ་ཛཏ་བྷཾ་ག་ས་ར་ཛ་ནཱི། །དྲྀཏྠོ་བ་བིབྷཾ་ཤྲཱིགུཔྟ་མེ་ཡི་དོང་ནི་མཚོ་སྐྱེས་ཅན། །བུང་བ་འཇམ་སྒྲོགས་ལྡན་པར་གྱུར། །མཚོ་སྐྱེས་ཁྲི་ལ་བཞུགས་པ་དེ། །སྦི་ས་རོ་རུ་ཧ་བསྟི་རེ། །ཏད་དྲྀཊི་ནཥཱ་ཀཱ་ལུ་སྲོ་བཱ་ཙ་ཙ་ར་ཎཱ་ན་ཏ། །།དཔལ་སྦས་ཀྱིས་ནི་མཐོང་གྱུར་ནས། །དེ་མཐོང་སྡིག་པ་ཉམས་པ་ཡིས། །ཞབས་ལ་བཏུད་དེ་རབ་ཏུ་སྨྲས། །བྷ་ག་བཱན་མ་མ་པཱ་པ་སྱ་ཀུནྟ་བྱོ་ཡཾ་བྱ་ཏི་ཀྲ་མ། །མོ་ཧཱནྟེ་པ་ཏི་བཅོམ་ལྡན་བདག་ནི་སྡིག་ཅན་གྱིས། །རིམ་ལས་འདས་འདི་བཟོད་པར་བགྱི། །གཏི་མུག་གིས་ལོང་ལྟུང་ཏཾ་རུ་ཙུ་ཀཱ་རུ་ཅུ་མ་དྷི་ཀཾ་ས་ཏྭཾ། །མ་མཱ་ཀ་ལྱཱ་ཎ་མི་ཏྲེ་ཎ་ཡོ་ཡཾ་པཱ་ཆགས་ལ། །དམ་པ་རྣམས་ནི་ཐུགས་རྗེས་ལྷག། བདག་ནི་མི་དགེའི་བཤེས་གཉེན་དག། རབ་ཏུ་པ་པ་ཐེ་ཀྲྀ་ཏ། །ཨུ་པ་དེ་ཤཔྲ་མོ་ཧེ་ཎ་ཏ་ཏྲ་ཏཱ་ཎཾ་བྷ་བཏ་སྨྲི་ཏི། །བི་ཥ་རྨོངས་པས་སྡིག་པ་ཡི། །ལམ་ལ་ཉེ་བར་གདམས་པ་བྱས། །དེ་ལ་ཁྱོད་ཉིད་དྲན་པ་སྐྱབས། །ཁྱོད་ཀྱི་—།
