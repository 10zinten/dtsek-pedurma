ཡ་པཎྜི་ཏ། །ཏསྨིན་ཡཱ་ཏི་ཏ་དཱ་དིཝཾ་བི་ཧཱ་ར་གི་རི་མཱ་བི་ཤཏ། །ཏ་ཏྲ་སྱཱཏྟཱི་མཁས་པ་ནི། །དེ་སོང་ཚེ་ན་དེས་བསྟན་པའི། །གཙུག་ལག་ཁང་གི་ཁང་པར་ཞུགས། །དེ་ནི་གཡོ་མེད་ཀྲྀ་ཏཾ་ཏ་ནུཀྲྀ་ཏྭཱ་པྲ་ཏི་མུ་ཏཱཾ་སྨྲ་ཏི། །ས་ཏྲ་ད་གྷོ་ནི་ཛདྷམྨོ་བདྡྷ་པརླངྐ་སྐྱིལ་ཀྲུང་བཅས། །ལུས་ནི་དྲང་པོ་ཉིད་དུ་བྱས། །དྲན་པ་སྔོན་དུ་བྱས་ནས་ནི། །དེ་ཡིས་རང་གི་ཆོས་ནི་ཡ། །ཏསྨི་རྨ་མཱ་དྷེ་སནྣདྷེ་བ་སུ་དྷཱ་ས་དྷ་རཱ་དྷཱ་རཱ། །བི་ཙ་པཱ་ལ་བསམས། །དེ་ནི་ཏིང་འཛིན་ལ་ཞུགས་ཚེ། །ནོར་འཛིན་འཛིན་མ་འཛིན་བྱེད་མིབཅས། །མ་ལུས་ཆུ་ལཱ་ཁི་ལཾ་བྷོ་དྷི་ཛ་ལ་ལོ་ལ་དུ་ཀུ་ལི་ནཱི། །ཤ་ཀྲསྦི་ནི་ཡ་ཏ་དྷཱ་ནཱཾ་གཏེར་ཆུ་རླབས་ཀྱིས། །དུ་ཀུ་ལ་ཡི་གོས་ཅན་གཡོས། །ངེས་པའི་བསམ་གཏན་དེ་ཤེས་ནས། །གེགས་ཏཛྫཱ་ཏྭཱ་ནིཐིམྨཱ་སིདྡྷ་ཡེ། །དི་དེ་ཤ་དིཀྵ་རཀྵ་ཡི་དིཀཱ་ལཱནྡྷེརྩྭ་བྷཱ་སྐ་རན། །མེད་གྲུབ་སླད་བརྒྱ་བྱིན་གྱིས། །ཕྱོགས་སུ་བསྲུང་སླད་ཕྱོགས་སྐྱོང་ནས། །ཟླ་བ་ཉི་མར་བཅས་པ་བསྐོས། །བྷ་ག་བཱ་ན་ཐ་སརྦ་ཛྷསྟ་སྱ་སིདྷི་མཱུ་བསྦི་ནཾ། །བཱ་ཀཱཀྐུ་ཤ་ལ་ལུ་མཱ་མཱ་དེ་ནས་བཅོམ་ལྡན་ཀུན་མཁྱེན་གྱིས། །དེ་ཡི་དགེ་རྩ་སྨིན་པ་ལས། །གྲུབ་པ་ཉེ་བར་གནས་པ་དག། ཛཱ་ཏྭཱ་ག་ཎ་མ་ཙིནྟ་ཡཏ། །ཨཱ་སནྣཱཉཏྲ་ད་སྭཱ་སྱ་ཤཱ་རི་བུ་ཏྲ་ས་མེ་ཏྱ་མཁྱེན་ནས་སྐད་ཅིག་ཡང་དག་བསམས། །འདི་ཡིས་དགྲ་བཅོམ་ཉེ་བ་ལ། །གལ་ཏེ་ཤཱ་རིའི་བུ་འོངས་ཙཏ༑། དཱ་ར་མུདྡྷཱ་ཊ་ཡ་དྷེ་བི་གྷཱ་རེ་ཥ་ན་ཤཾ་ས་ཡ། །ཏསྨཱ་དཱ་ག་ཏ་སྱ་ཏ་སྱ་ནས། །སྒོ་ནི་རབ་ཏུ་བརྡུངས་པ་ན། །བར་དུ་བགེགས་འདི་ཐེ་ཚོམ་མེད། །དེ་ནས་དེ་ནི་འོང་བ་ཡིས། །ག་ཏཱ་སཱ་ཡ་མ་ཧཾ་པུ་ར། །ཀ་རོ་མི་ཀཱ་ལ་ཧཱ་ར་ཡ་ནཱ་ནཱ་པྲབྷཱ་ཏྲ་ཡཱ༔སྔོན་དུ་བདག་རང་ཉིད་སོང་སྟེ། །སྣ་ཚོགས་འདྲི་བ་ལ་བརྟེན་པའི། །གཏམ་གྱི་དུས་ནི་དབྲོག་ཀ་ཐཱ། །ཨི་ཏི་སཾ་ཙི་ཏྲ་བྷ་ག་བཱན་སཱ་ཡཾ་ཏཏྟེ་ཤ་མཱ་ག་ཏ། །བྷི་ཀྵ་རཱ་པར་བྱ། །དེ་ལྟར་དགོངས་ནས་བཅོམ་ལྡན་འདས། །རང་ཉིད་ཕྱོགས་དེར་བྱོན་ནས་ནི། །དགེ་སློང་དེ་གཙྪ་ཏ་ཏེ་སྱ་བི་ལམྦ་ཀ་ཐ་ཡཱ་ཀ་རོཏ། །སུ་ར་པྲ་བྷཱ་བཱནྣིཤསྡེ་ནི་ནི་འོང་བ་དག། གཏམ་གྱིས་ཐོགས་པ་ཉིད་དུ་བྱས། །།ལྷ་ཡི་མཐུ་ལས་མཁར་བགྲོད་རྣམས། །སྒྲ་མེད་།
