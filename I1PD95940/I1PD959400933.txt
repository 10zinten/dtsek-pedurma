སཱུ་རྱ་བྷ་བ་ཏཱ་ནི་རཱ་ལོ་ཀ་སྱ་ཀཱ་ག་ཏི། །ཀོ་ནཱ་མ་ཏཱལླི་ནཱ་མི་ཊཾ་ཛྙཱ་ནཱ་ལོ་ཀིབཏང་གྱུར་པ། །སྣང་མེད་དག་གི་འགྲོ་བ་ནི། །བཤེས་གཉེན་ཁྱོད་དང་བྲལ་བ་ན། །སུ་ཡི་ཡེ་ཤེས་ཀ་རི་ཥྛ་ཏི། །མོ་ཧ་ནི་དལྟ་མུ་དཱ་སྱ་ཛ་ག་ཏ་པདྨ་སྱ་བོ་དྷ་ནཾ། །ཀྲྀ་ཏྭ་ས་མལྟ་སྣང་བ་ཡིས། །འགྲོ་བའི་པདྨ་རྨོངས་པ་ཡི། །གཉིད་ཀྱི་རྒྱ་ལྡན་རྒྱས་བྱེད་འགྱུར། །མཐའ་དག་འགྲོ་ལ་རྒྱུ་ཛ་ག་ཏཱ་མུ་ལ་ཀཱ་ར་ཎཾ། །པྲ་ཡཱ་སི་སརྦ་ཛྷ་སུ་ལ་བྷཀ་ཐཾ་དུརླ་བྷ་མེད་དུ། །ཕན་པ་དག་ནི་རབ་མཛད་ནས། །ཀུན་མཁྱེན་རྙེད་པ་སླ་བ་ཁྱོད། །བདག་ལ་རྙེད་དཀའ་ཇི་ཏཱསྨྲ་མ། །ཨི་ཏུཀྟཔྲ་ཎ་ཡཱཏྟེ་ན་ས་མུནྟཱ་མྱ་སུ་ཁཱམཱུ་ཛཾ། །པྲ་སནྣ་ལྟར་གྱུར། །ཞེས་པ་དེ་ཡིས་གུས་པས་གསོལ། །དེ་ལ་བཅོམ་ལྡན་འཛུམ་ལྡན་ཞིང་། །བཞིན་གྱི་ཆུ་སྐྱེས་ཀཱནྟི་བྷ་ག་བཱན་སརྨི་ཏསྟ་མ་བྷཱ་ཥ་ཏ། །།ཨི་མཱཝཏྠ་ལ་བེཏི་ཏྭཾ་གཟེངས་བསྟོད་དེ། །རབ་དངས་མཛེས་པས་བཀའ་སྩལ་པ། །མཉེས་གཤིན་རྟག་ཏུ་མི་རྟག་ཉིད། །ཀིནྡྲ་ནི་ཏྱ་མ་ནི་ཏྱ་ཏྭཾ། །གླུ་ཎི་ཀཱསརྦ་སཾ་སྐུ་རཱཀོ་ཡནེ་ཧ་གྲ་ཧཱ་གྲ་ཧ། །འདི་ཁྱོད་རིག་པ་མིན་ནམ་ཅི། །འདུ་བྱེད་ཐམས་ཅད་སྐད་ཅིག་ལ། །ལུས་ཀྱི་གདོན་ནི་སུ་ཞིག་འཛིན། །དྲུ་རཾ་པྲ་ཡཱནྟི་དི་ན་ཀྲྀཏྟཱ་མ་ནིན་ཏཱ་ནི་ཡཱ་ནི་ག་ཡཱ་ཤ་བ་ལི་ཏཱ་ནི་ས་དཱ་དི་ནཱ་ནི། །གང་དག་རྟག་ཏུ་ཉིན་བྱེད་འཁོར་བས་རབ་བཟླས་པའི། །མཚན་མོས་བཀྲ་བའི་ཉིན་པར་རིང་དུ་རབ་འགྲོ་ཨེ་ཏཱ་ནི་ཏཱ་ནི་ནི་པ་ཏིལླཾ་ཐུ་ཀཱ་ལ་ཙཀྲ་སཾ་ཤ་ཏི་ཏཱ་ནི་ཀུ་ཤ་ལཱ་ནི་ཚྭ་ཛཱི་བི་ཏཱ་ནཾ། །བ། །མི་ཡི་འཚོ་བ་རྣམས་ཀྱི་དགེ་བ་འདི་དག་ནི། །རྒྱ་ཆེ་དུས་ཀྱི་འཁོར་ལོ་ལྷུང་བས་གཅོད་པ་ཉིད། །ད་ལོ་ཀྱེ་ཏེ་ཤི་ལཱ་ཛ་ལ་དྷི་བ་ལ་ཡཤུ་ཉྩ་ཏི་ས་རཾ་ས་ཧོདྡྷཱ་ཏིརྒྱུ་ཏེ་བྷ་བན་ཤ་ར་ཎི་སུ་རི་འདི་སྙིལ་འགྱུར་ཆུ་གཏེར་འཇིང་®ནི་མཆོག་ཏུ་བསྐམ། །རབ་སྐྱེས་འབྱུང་པོ་དང་བཅས་སྲིད་པའི་ལམ་སྱ་ཏི་ཏ་རཱི། །ཨ་ས་རེ་སཾ་སཱ་རེ་ཙ་ཀི་ཏ་ཧ་རི་ཎི་ལོ་ཙ་ན་ཙ་ལེ་སྦྲ་སཱི་ནཱ་ཡཾ་བ་ཧུ་པ་རབ་འཇིག། རི་དྭགས་སྐྲག་མིག་ལྟར་གཡོ་སྙིང་པོ་མེད་འཁོར་བར། །མང་དུ་ཡོངས་འདྲིས་ལུས་ཀྱི་རི་ཙ་ཡཀཱ་ཡ་ནི་ཙ་ཡ། །ཨ་བྷི་དྷཱ་ཡེ་ཏི་བྷ་ག་བཱན་སུ་བྷ་དྲ་སྱ་པྲ་ཤཱནྟ་ཡེ། །ཚོགས་ནི་བརྟན་གནས་མིན། །ཞེས་པ་གསུངས་ནས་བཅོམ་ལྡན་གྱིས། །རབ་ཏུ་ཞི་སླད་རབ་བཟང་ལ། །—།
