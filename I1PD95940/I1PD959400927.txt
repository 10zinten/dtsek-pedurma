ཨ་སྨནྣཱ་དུམྲ་ར་པ་ནེ་ནཱ་རྒྱ་ཐཱ་མུ་ཀུ་ལ་བྷི་ཡ། །མ་མི་བ་པུ་ལྕི་འབྱུང་ཞིང་འཁོར་ལོས་སྒྱུར་བ་དག། འབྱུང་ན་ཡང་ངོ་གཞན་མེདུ་མིན། །ཡང་ན་བདག་གི་བསོད་་ར་ཐ་བཱ་སཾ་ཛཱ་ཏ་མི་ད་ཡདྷུ་ཏཾ། །།སུ་བྷ་དྲཕྱིནྟ་ཡི་ཏྲེ་ཏི་བ་བྷུ་བཱ་ནམས་ཀྱིས། །རྨད་བྱུང་འདི་ནི་ཡང་དག་སྐྱེས། །ཞེས་པ་བསམས་ནས་རབ་བཟང་ནི། །ཀུན་དགའ་ནནྡ་ནིབྷ་ར། *། ཨ་ཐཱ་མརྱ་པ་ཏི་ག་ཏཱ་སུ་པྲི་ཡཾ་པུ་ན་ར་བྲ་བཱིཏ། །་། རབ་ཏུ་རྒྱས་པར་གྱུར། །དེ་ནས་སླར་ཡང་འཆི་མེད་བདག། སོང་ནས་རབ་དགའ་དག་ལ་སྨྲས། །ནི་ཛ་ཀཱན་དྷི་གོབློ་གཱཔ་ར་ཀཱརྨེ་ཥ་ས་དྷ་བ། ་། ཨེ་ཧི་སུ་པྲ་ཡ་གཙྪ་བ༔དམ་པ་རང་གི་བྱ་བ་བས། །གཞན་གྱི་བྱ་བར་བརྩོན་པ་ལྷག། ས་གཞི་ཟླ་བ་བདེ་གཤེགས་ཀྱིས། །ཀྱི་ཏི་ཀཱི་ཏི་ཏ་ལེནྡྲ་ནཱ། །སནྡེ་ཤ་ནཱི་ཤུ་བྷིནྡྲ་ཤཱ་སུ་ག་ཏེ་ན་པྲ་ཀཱ་ཤི་ཏཾ། །བསྟན་པ་འོད་ཀྱིས་ཐམས་ཅད་ནི། །།རབ་ཏུ་གསལ་བའི་ས་གཞི་ལ། །རབ་དགའ་ཚུར་ཤོག་འུ་ཅག་ཏཾ་པུ་པྲ་ཡོ་བ་དེརྡ་བི་པཱི་ཎཱ་ཡཾ་གཱི་ཏི་སཱ་རེ་ཎཱ། ་། ཡོ་ཛི་ཏོ་ཡམྨ་ཡཱ་འདོང༌། །དེ་ལ་རབ་དགས་སྨྲས་པ་ལྷ། །རྒྱུད་མང་ལ་ནི་གླུ་དབྱངས་རྒྱུན། །བདག་གིས་རབ་ཏུ་སྦྱར་པཧྥུ་ཀ་རཱ་ཎ་བྷ་བི་བྱེ་ཏི། །ཨི་ཏྭཱ་ཀནྡྷོ་མ་ར་པ་ཏིརླ་ཡོ་མ་ན་མ་ནོ་ར་ཐ། །བ་འདི། །ཕྱི་ནས་བྱ་བ་སླ་མི་འགྱུར། །ཞེས་ཐོས་འཆི་མེད་བདག་པོ་ནི། །ཡིད་ལ་རེ་བ་ཉམས་པར་གྱུར། །ཨ་ཡོནྣཱ་པ་ན་ཏཾ་ཧཱི་ན་པུ་ཙྩཔི་བནྟི་ནཱ་མི་ཏཾ། །ཨུཀཱུལླ་འབད་མེད་བདུད་རྩི་ཉེར་གནས་པ། །བསོད་ནམས་དམན་རྣམས་འཐུང་བ་མིན། །དུས་ཀྱིས་ཨུ་དུམ་ཀུ་སུ་མཱ་མོ་ད་སཾ་པཱུ་རི་ཏ་དི་གནྟ་རཾ། །ཀཱ་ལེ་ནོ་དྲུམྦ་ར་བ་ནཾ་སུ་བྷ་དྲོ་པི་བཱ་རའི་ནགས། །མེ་ཏོག་རབ་རྒྱས་དྲི་བཟང་གིས། །ཕྱོགས་མཚམས་ཀུན་ཏུ་ཁེངས་གྱུར་པཆེ། །རབ་བྱ་ལོ་ཀ་ཡཏ། །སོ་ཙིནྟ་ཡདྷ་བ་ཏེ་ཏདྲི་ཀཱ་སི་ཀུ་སུ་མཾ་ད་ནཾ། །བཟང་གིས་ཀྱང་མཐོང་བར་གྱུར། །དེ་ཡིས་ཀྱི་བསམས་པ་མེ་ཏོག་གི། །ཚལ་ནི་རྣམ་པར་རྒྱས་པ་འདི། །དྷརྨ་པྲ་བཏྟེ་ན་ཤུyཛྫ་ཡེ་ནཱ་ཙཀྲ་བརྟི་ན། །མཏྤ་ཏྱཱ་ནཾ་པྲ་བྷ་བ་ཏྭཱ་ནི་སྟོན་པ་དམ་ཆོས་སྟོན་པ་དང་། །འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་ལ་འགྱུར། །ཡང་ན་བདག་གི་བསོད་ནམས་།
