མི་རྩྭ་ས་ཏྲིཋསྤཋ་མརྨ་ཐ། །ཡ་སྱ་སཾ་དན་ནེ་ནི་བ་སཱ་བྷ་མོ་ཡིས། །ཡིད་སྲུབས་ཀྱི་གསལ་བ་དེ་མཐོང་གྱུར། །གང་ཞིག་མཐོང་བ་ཉིད་ཀྱིས་དེ། །དྲན་པ་རྣམ་པར་བཏ་བི་ཀྲྀ་ཏ་སྨྲ་ཏཱི། །ཏ་སྤྱི་བན་བཱ་བྷི་ལཱ་ཌི་ཎ་ཛ་ན་ཀེ་ནེ་བ་སརྦ་ཏཱ། །ཀཱན་བརྗེད་པར་གྱུར། །གསར་དུ་མངོན་པར་འདོད་པ་ཡིས། །ཕས་བཞིན་དེ་ལ་དེ་གཏད་ནས། །མཛེས་ཀཱནྟི་མ་ཡཱི་ནཱ་མ་བི་ཛ་ཧཱ་ར་གུ་ཧཱ་ཤྲཱི་ཧེ། །ཏ་ཏཱ་བྷ་ར་ཎ་རནྣཾ་ཤུལྡན་ཞེས་པའི་མཛེས་མ་དང་། །ཕུག་གི་ཁང་པར་རྣམ་པར་རོལ། །རིན་ཆེན་རྒྱན་གྱི་འོད་ཟེར་གྱིས། །པཱི་ཏ་ཏི་མི་རོཏྟ་རེ། །ར་མ་མཱ་ཎཱ་ཙི་རནྟེ་ན་ཀཱ་ལེ་པུ་ཏྲ་མ་བཱ་པ་པཱ། །བ་ལ་རབ་རིབ་ཚོགས་ནི་དེས་བསལ་ཏེ། །དེ་དང་ཡུན་རིང་ཚེས་པ་ལས། །དུས་སུ་བུ་ནི་དེ་ཡིས་ཐོབ། །བུ་དེ་བཱན་མ་རུ་ཏ་ཛ་བས་བཱ་སྐྱོ་པི་ཡ་དཱ་ཤི་ཤུ། །ཏ་དཱ་སྱ་མཱ་ཏཱ་སཾ་ཛྙཱ་ཤཱི་གྲུ་ག་བྱིས་པ་ཉིད་ལ་ཡང་། །གང་ཚེ་སྟོབས་ལྡན་རླུང་ལྟར་མགྱོགས། །དེ་ཚེ་མ་ཡིས་དེ་ཡི་མིང་། །མྱུར་འགྲོ་ཨི་ཏྱ་སཱ་དྷ་བྷུ་ཏ། །སཱ་ནིཟིམྦྷེ་སཾ་བྷོ་ག་སུ་ཁཱ་ཏྲཔ་གུ་ཧཱནྟ་རེ། །ཞེས་པར་བསྒྲུབས་པར་གྱུར། །བགེགས་མེད་ལོངས་སྤྱོད་བདེ་བ་ཡིས། །ངོམས་མེད་དེ་ཡང་ཕུག་གཞན་དུ། །པྲི་ཡི་བྷཾ་ཏྭཱ་བ་དཱ་ཡཱ་ཏི་པི་དྷཱ་ཡ་ཤི་ལ་ཡཱ་ཁཾ་ཧཾ། །ཀ་དཱ་ཙི་ད་ཐ་བྲྀཏྟཱནྟཾ་མཛའ་བོ་བཞག་ནས་རྡོ་བ་ཡིས། །ཕུག་ནི་བསྒྲིབས་ཤིང་རྟག་ཏུ་འགྲོ། །དེ་ནས་ནམ་ཞིག་ཕ་ཡིས་ནི། །ནི་ཛི་པི་ཏྠཱ་ནི་བི་དེ་ཏཾ། །ཨཱ་ཀརྞ་ཤཱི་་གཤྲིནྟ་བིསྣ་ཡཱ་ཀུ་ལི་ཏོ་བ་ད་རང་གི་སྤྱོད་པའི་མཐའ་རྣམས་བཤད། །མྱུར་འགྲོ་ཡིས་ཐོས་བསམ་པ་ནི། །ཡ་མཚན་དག་གིས་རྒྱས་པས་ཏ༑། ཤི་ལཱ་ནི་བདྡྷ་དྷཱ་རེ་སྨིནྣ་ཐ་ཀ་སྱེ་མ་གཾ་ཧཱནྟ་རེ། །ཨ་ཧོ་སྣེ་ཧོ་པེ་ཡི་སྨྲས། །ཁྱིམ་གཞན་སྒོ་ནི་རྡོ་བ་ཡིས། །ལོང་བ་བཞིན་དུ་བསྡམས་པ་འདིར། །ཀྱེ་མ་ཡབ་ཁྱོད་མཛེས་ཏཱ་ཏ་ཏ་བ་བན་ན་ཏྭཾ་ག་ཏ། །ཨེ་ཧི་བཱ་རཱ་ཎ་སཱི་མེ་བ་གཙཱ་བསྟེ་ནི་པས་ཀྱང་། །བཅིངས་པ་འདི་ཉིད་དག་ཏུ་གྱུར། །ཚུར་ཤོག་ཁྱོད་ཀྱི་རང་གི་གནས། །བཱ་རཱ་ཎ་སཱི་ཛཱ་སྤ་དཾ། །བི་སུ་ལཱ་མ་པྱ་ཡཾ་མེ་ན་ཤི་ལཱ་མུཏྟཱ་ར་ཡཱ་མྱ་ཧཾ། །སྭ་དེ་ཤ་བི་ཉིད་དུ་འདོང་། །ཆེ་ཡང་འབད་པ་མེད་པར་ནི། །རྡོ་བ་བདག་གིས་བསལ་བར་བྱསི། །རང་ཡུལ་བྲལ་།
