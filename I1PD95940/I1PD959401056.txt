འཁྲི་ཤིང་། །ཁེ་בהOlotsནཱིནྡྷ་མ་ཅུ་པ་རཱ་ཡ་ཎ། །ས་ཙཀྲེ་པ་ཎི་ཙ་ར་ཎ་ཙྪེ་དཱ་སྲིཀཾ་ཀ་ཤཱ་ཡི་དགའ་མར་མིག་བཀོད་དེ་དག་རྣམས། །རྐང་ལག་བཅད་པའི་ཁྲག་གིས་ནི། །འདམ་ལ་ཉལ་བ་དག་ཏུ་ན༑། ཏཱ་ནཱནྟ་ནཱ་ད་ནསྟི་བྲ་མརྨ་ཙྪ་ད་བྱ་ཐཱ་ཏུ་རཱན། །ད་དནོ་བྷ་བྱས། །མི་བཟད་གནད་བཅད་ཆོ་ངེ་ཡིས། །གཟིར་ཅིང་ཆོ་ངེ་ལྡན་དེ་དག། སངས་རྒྱས་བཅོམ་ལྡན་ག་བཱནྟུདྡྷཀ་རུ་ཎཱ་སྣེགྷ་ལོ་ཙ་ན། །ཏཛྲཏྟཱ་མྲྀ་ཏ་ཀྲྀཥྞ་བ་སྤྲཏྟཱསྟེ་པྲྀ་ཏ་ཐུགས་རྗེ་ཡིས། །སྣུམ་པའི་སྤྱན་གྱིས་རབ་ཏུ་གཟིགས། །དེ་ཡི་གཟིགས་པ་བདུད་རྩི་ཡི། །ཆར་ལྟར་རེག་ཛཱི་བི་ཏཱ། །སཾ་བྷཱིཋ་པཱ་ཎི་ཙ་ར་ཎཿས་མུཏྟསྦུག་ཏ་བྱཱ་ཐཱ། །པས་དེ་དག་རྣམས། །སྲོག་བཟུང་རྐང་ལག་ལེགས་པར་འབྱར། །གདུང་བ་དང་བྲལ་ཡང་དག་ལངས། །ཏ་ཏསྟེ་ཏྭཾ་པྲ་སནྣཱ་ནཾ་བྷ་ག་བཱནྣཱརྨ་དེ་ཤ་ནཾ། །བྱ་དྷཱ་ཏྲ་ཡཱ་ཤུ་ཏེ་དེ་ནས་རབ་རྫོགས་དེ་དག་ལ། །བཅོམ་ལྡན་འདས་ཀྱིས་ཆོས་བསྟན་མཛད། །གང་གིས་མྱུར་དུ་དེ་དག་པྲ་པུ་ར་ཎཱ་གཱ་མི་ཕ་ལོ་ད་ཡཾ། །ཏཛྲབྱཱ་བྷིཀྵ་བྷི༔དྲྀཋསཱཉྩཆེཌྷ་ག་རྣམས། །ཕྱིར་མི་འོང་འབྲས་དར་བ་ཐོབ། །དེ་མཐོང་དགེ་སློང་མཚར་བཅས་པས། །དྲིས་ཤིང་བཅོམ་བཀྲི་ན། །ཛ་གཱ་ད་པཱུཧྥ་མ་ཀྱེ་ཏེ་ད་ཡ་ཡི་བ་མ་ཡོདྡྷཾ་ཏཱ། །ལྡན་རྒྱལ་བ་ཡིས། །གསུངས་པ་སྔོན་ཡང་འདི་དག་རྣམས། །བདག་གིས་བརྩེ་བ་ཉིད་ཀྱིས་བཟུང་། །བཱ་རཱ་ཎ་ཧཱུཾ་པུ་རུ་རཱ་ཛ་བྲཧྨ་ད་ན་མཎྜི་ཏཱཾ། །ཤ་ཏཱ་ནི་པཉྩ་བྷིནྡྷཱ་ནཾ་ནི་བཱ་རཱ་ཎ་སཱི་དག་ཏུ་སྔོན། །རྒྱལ་པོ་ཚངས་བྱིན་དག་གིས་ནི། །འབྱེད་བྱེད་སྡིག་པ་བྱས་པ་ཡིས། །རསྠཱ་ནི་ཀྲྀ་ཏཱ་ག་སཾ། །ཏེ་རཱ་ཛ་བྷི་ཏྭཱ་སནྟཔ་མ་རུ་མཱརྒེ་པྲ་བཱ་སི་བློན་པོ་ལྔ་བརྒྱ་ངེས་པར་བསྐྲད། །རྒྱལ་པོའི་འཇིགས་པས་ཡང་དག་གདུང་། །མྱ་ངམ་ལམ་དུ་བགྲོད་དེ་ན༈། གྷིམྨེ་ནི་རུ་ད་ཀཙྪཱ་ཡ་ནེ་པེ་ཏུ་ལྕི་བྲ་ཀྲྀཥཱ་ཡཱ། །བི་ལོ་ཀྱ་བོ་དྷི་ས་དག། སོ་ག་གྲིབ་མ་ཆུ་མེད་པར། །མི་བཟད་སྐོམ་པས་ལྷུང་བར་གྱུར། །བྱང་ཆུབ་སེམས་དཔའ་གཉིས་ཏཱ་ལྟ་ན་བྷ་དྲོ་ནཱ་མ་མ་ཧཱ་དཱི་པ། །ཏཏྠ་ལཱ་བེ་ཥཱ་ནིབྷིནྡཀཱ་རུ་འཐུང་ཆེ། །བཟང་པོ་ཞེས་པས་དེ་དག་མཐོང་། །དེ་དག་ཆོ་ངེའི་མདས་ཕུག་ནས། །སྙིང་རྗེས་གདུང་།
