རུ་དྷཱི་རཾ་གི་ར་ཏྭཾ་ག་ཏཾ། །ཨངྒ་ཙྪེ་དེ་པྱ་ཀ་ལུ་ཎི་བ་ཤུ་བ་ཡ་དི་མེ་མ་ན། །འོ་མ་ཉིད་དུ་གྱུར་ལ་ལྟོས། །ཡན་ལག་བཅད་ཀྱང་བདག་གི་ཡིད། །གལ་ཏེ་སྡིག་མེད་གྱུར་པ་ན། །པ་ཏེ་ནེ་ཏེ་ན་ཥཱ་ནི་ཏཱ་རྩེ་བཱངྒཱ་ནི་སནྟུ་མེ། །ཨི་ཏི་ཤུདྡྷ་དྷི་ཡསྟ་སྱ་ཏཱི་བདེན་པ་དེ་ཡིས་བདག་གི་ནི། །ཡན་ལག་དེ་དག་འབྱར་བར་ཤོག། ཅེས་པ་དག་པའི་བློ་ལྡན་དེསི། །བྲ་ས་ཏོ་བ་ཡཱ་ཙ་ནཱཏ། །ཐི་ཏྲཱ་ཏྱངྒཱ་ནི་ཏཱ་ནྱ་བ་ས་ཧ་སཱ་སཱསྦྱམཱ་ཡ་ཡུ། །མཆོག་ཏུ་བདེན་པ་བརྗོད་པ་ལས། །འཕྲལ་ལ་ཡན་ལག་དེ་དག་ཉིད། །འབྱར་ནས་རྣལ་དུ་གནས་པ་ཐོབ། །མུ་ཀུ་ཊཀྲྀཥཱ་ཙ་ར་ཎས་མུ་བཱ་ནཱཾ་པསྟ་ཏ། །མུ་ནེ་མ་ཧཱ་པྲ་བྷཱ་བོ་སི་ཏ་པ་སཱ་དེ་ནས་མི་བདག་ཅོད་པན་ནི། །ཞབས་ལ་རེག་པས་དེ་ལ་སྨྲས། །ཐུབ་པ་ཁྱོད་ནི་མཐུ་ཆེན་པོ། །དེ་ལས་ཏ་ཏ་ཀི་མིཙྪ་སི། །པུ་ཎྱ་ཧལྟ་བ་ལཝེ་ན་མོ་ཧཱནྟཾ་ཀ་རུ་ཎཱ་ནི་དྷེ། །དཀའ་ཐུབ་ཀྱིས་ཅི་བཞེད། །རྨོངས་པས་ལོང་བ་སྡིག་པ་ཡིས། །གཡང་ལ་ལྷུང་བར་གྱུར་པ་བདག། པཱ་པཱ་བ་སཱ་ཏ་པ་ཏི་ཏཾ་ཚྭ་ཏུ་མུདྡྷཎྜ་མརྟ་སི། །ཨི་ཏུཟི་ཏསྙིང་རྗེའི་གཏེར་ཁྱོད་བསོད་ནམས་ཀྱིས། །ལག་པའི་འཆེལ་སས་གདོན་པར་འོས། །ས་དབང་གིས་དེར་ཀྱི་ཏཱི་ཤེ་ན་པྲ་ཏྱ་བྷཱ་ཥ་ཏ་ཏཾ་མུ་ནི། །སནྡྷ་ར་ཎཱ་ཡ་མདྡྷཱ་ནཱཾ་བནྡྷ་གསོལ་བཏབ་པའི། །ཐུབ་པས་དེ་ལ་ལན་སྨྲས་པ། །བྱིང་བ་རྣམས་ནི་ལེགས་བསྒྲལ་དང་། །བཅིངས་པ་ནཱ་མ་པི་མུཀྟ་ཡེ། །ཨ་ཤཱ་ས་ནཱ་ཡ་བྷཱི་ཏཱ་ནཾ་ནིཝཱ་ཎཱ་ཡ་བི་མུ་རྣམས་ཀྱང་དགྲོལ་བ་དང༌། །འཇིགས་པ་རྣམས་ནི་དབུགས་དབྱུང་དང་། །རྣམ་པར་རྨོངས་རྣམས་ཐར་བྱ་ཏྭཾ། །རཱ་ཛརྣ་ནུཏྟ་རཾ་ས་མྱཀ་སཾ་བོ་དྷི་མ་ཧཱ་མརྦ་ཡེ། །ཡ་དཱ་ཏུ་བྱའི་སླད། །རྒྱལ་པོ་བླ་མེད་ཡང་དག་པར། །རྫོགས་པའི་བྱང་ཆུབ་བདག་དོན་གཉེར། །བླ་མེད་ཡང་ས་མྱཀཾ་བོ་དྷིནཱ་མ་བཱསཱུ་ནྱ་ནུཏྟ་རཾ། །མོ་ཧ་ཙྪེ་དཾ་ཀ་རི་ཎྱ་སི་ཏ་དཱ་དག་རྫོགས་པ་ཡི། །བྱང་ཆུབ་གང་ཚེ་བདག་གིས་ཐོབ། །དེ་ཚེ་ཡེ་ཤེས་རལ་གྲི་ཡིས། །ཁྱོད་ཀྱིས་རྨོངས་ཙཱ་ནཱ་སི་ནཱ་ཏ་བ། །པྲ་ཡ་ཡོ་མུ་ནི་རི་ཏུཏྟཱ་ཏ་མཱ་མནྡྲ་སཱ་མཱ་ཐ་མཾ། །ཏ་མེ་པ་གཅད་པར་བྱ། །ཐུབ་པས་དེ་བརྗོད་དེ་ལ་ནི། །ཁམས་བདེ་བྱས་ནས་རང་གནས་སོང་། །རྒྱལ་པོའི་—།
