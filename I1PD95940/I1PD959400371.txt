བི་ཤྭ་ཀྱི་ཀཱ་ན་ནཱནྟ་རཾ། །པཱ་ཡ་སཱ་མྲྀ་ཏ་བྷ་གེ་ན་ལབྷཱ་དི་བྱ་བ་ལོ་ད་ཡ། །དེ་དག་ལ་ནི་གླེངས་ནས་གཤེགས། །འོ་ཐུག་བདུད་རྩིའི་ཆ་དག་གིས། །སྟོབས་ནི་མཆོག་ཏུ་རྒྱས་པ་ཐོབ། །པྲཙྪཱ་ཡ་ཏ་རུ་སུཙྪཱ་ཡ་མཱ་རུ་རོ་ཧ་མ་ཧཱི་དྷ་ར། །པརླངྐ་བནྣཱ་བསིལ་གྲིབ་མཆོག་ལྡན་ཤིང་དག་གིས། །རབ་མཛེས་ས་འཛིན་དག་ལ་འཛེགས། །དེར་ནི་སྐྱིལ་ཀྲུང་བཱ་པཱ་ཡ་ཏ་ཏྲ་ཏསྨིནྡྷ་ཁཾ་སྦི་ཏེ། །ཨ་ཧཾ་ཀཱ་ར་ཨི་བཱ་ཏྱཱཙྩ་ཤི་རཱ（བཅིངས་པ་མཛད། །དེ་ཡིས་བདེ་བར་བཞུགས་པའི་ཚེ། །ང་རྒྱལ་བཞིན་དུ་རྩེ་མོ་ནི། །རབ་ཏུ་མཐོ་སོ་དྲིཅུ་གིརླཏ། །བི་བྷིཥྛ་བྷཱུ་དྷ་རེ་ཏསྨིམྨ་པྲ་དབློ་བི་ཥ་དྷི། །ས་པཀྵ་བའི་རི་དེ་ཞིག། ས་འཛིན་རྣམ་པར་ཞིག་པ་དེར། །ཐུགས་ནི་བྱུང་བཞིན་དེ་ཡིས་བསམས། །ཀྱེ་མ་དྲི་ལ་ནི་ཀམྨཱ་ཎི་མ་ཡཱ་ཀཱ་ནི་ཀྲྀ་ཏཱ་རྱ་ཧོ། །ཨི་ཏི་ཙིནྟཱ་ཤ་ཏོཙྪཱ་སཾ་ཏ་མེད་མ་ཡིན་པའི། །ལས་འདི་བདག་གིས་ཅི་ཞིག་བྱས། །དེ་ལྟར་བརྒྱ་བསམ་དབུགས་རིང་བ། །དེ་མཱུ་ཙུནྡྷོ་མ་དེ་བ་ཏཱ། །ན་ཏཱ་ཡཱ་བི་ཧི་ཏཾ་སཱ་དྷོ་ཀརྨ་ཀིཉྩི་ད་སཾ་པྲ་ཏཾ། །ཨལ་ནམ་མཁའ་ལྷ་ཡིས་སྨྲས། །དམ་པ་ཁྱོད་ཀྱིས་མི་འོས་པའི། །ལས་ནི་ཅི་ཡང་བསྒྲུབས་པ་མེད། །རྒྱུན་ཙྪིནྡོཏྟལྟ་ཀུ་ལ་ཤྲཱི་དྷནྡྷཾ་ན་གྷ་མ་ཏེ་ཀི་ཏི། །སཏྭཱ་མུཏྟསྟ་ཀུ་ཤ་ལཆད་མེད་པར་དགེ་བསྒྲུབས་པ། །ས་ཡིས་འཛིན་པར་བཟོད་མ་གྱུར། །སེམས་ཅན་དགེ་བ་རབ་བསྒྲུབས་པྲལྕེ་ཤེ་ལ་ཤ་ཏཱདྷུ་རུ། །ནི་རཉྫ་ནི་ས་མུཏྟཱི༔་ས་རི་ཏཾ་བྲ་ཛ་ནིཤྩ་ལཾ། །སིདྡྷི་པ། །རབ་མཐོ་རི་བོ་བརྒྱ་བས་ལྕི། །ཆུ་བོ་ནི་རཉྫ་ན་ནི། །ཡང་དག་སྒྲོལ་ལ་གཡོ་མེད་ཡུལ། །བྱང་དཾ་བོ་དྷི་སཏྭཱ་ནཾ་དེ་ཤ་བཛཱ་སཏྠ་ལི་ལཱ་ཀུ་ལཱ། །དེ་བ་ཏཱ་བིཥྛ་མཱཧྥེ་ཎ་ཆུབ་སེམས་ལ་གྲུབ་སྟེར་བ། །རྡོ་རྗེ་གདན་ཞེས་བརྗོད་པར་སོང་། །ལྷ་ཡིས་བསྟན་པའི་ལམ་ནས་དེ། །པྲསྦི་ཏ་སྱ་དྷཱབྷཱུ་ཏ་ལེ། །།པཱ་ད་ནཱ་སི་རབྷཱུཏྟ་སྱ་ཧེ་མ་པདྨ་པ་རཾ་པ་རཱ། །རབ་གཤེགས་ལམ་གྱི་ས་གཞི་ལ། །ཞབས་ནི་བཀོད་པས་གསེར་དག་གི། །པདྨ་བརྒྱུད་པ་བྱུང་བར་གྱུར། །པྲི་ཐི་བཱི་བྲ་ཛ་ཏསྟ་སྱ་པྲལྤ་སཏྠ་ལི་ལཱ་ཀུ་ལཱ། །ར་ཎནི་ཀཾ་སྱ་པཱ་ཏི་བ་དེ་ནི་གཤེགས་པའི་ས་གཞི་ལ། །ཆུ་རྣམས་རབ་ཏུ་རྡོལ་བས་དཀྲིགས། །འཁར་བའི་སྣོད་བཞིན་སྒྲ་ལྡན་།
