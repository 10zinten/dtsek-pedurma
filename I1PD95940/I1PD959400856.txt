ལང་བྷཀྟ་ཀཱ་ལེ་ཀ་ཡིཀསྣ་དྡྷསྨཱ་ཀ་མ་ཡ་མུངྒ་ཏ། །ཏ་མེ་བོ་ཙ་ཏེསྨཱ་བྷིརྦ་ཟས་ཀྱི་དུས་སུ་རྩོད་པ་འདི། །ཁྱེད་ཅག་རྣམས་ལ་ཅི་སླད་བྱུང་། །དེ་ལ་དེས་སྨྲས་བདག་ཅག་གིས། །ཧཱ་རེ་པཱུཏྤ་ཛན་ནི། །བྷཀྟ་ཀཱ་ལེ་ཀྲྀ་ཏཾ་ཡུདྡྷཾ་ཏ་སྤྱི་ཏཥརྨ་ཎཕ་ལཾ། །ཨེ་སྐྱེ་བ་སྔ་མར་ཟས་དུས་སུ། །གཙུག་ལག་ཁང་དུ་རྩོད་པ་བྱས། །ལས་དེའི་འབྲས་བུ་འདི་དག་གོ། །དེ་ཏ་དཱ་ཀཙྩོ་ས་ཏ་ཏཔྲ་ཡཱ་ཏོ་ནྱ་ཏྲ་ཏདྲི་དྷཾ། །ར་མྱལླི་ཧཱ་ར་མ་པ་རཾ་དག་ཐོས་ནས་དེ་ནས་དེ། །སོང་བས་གཞན་དུ་དེ་ལྟ་ཡི། །གཙུག་ལག་ཁང་གཞན་དགའ་བ་ནི། །རབ་པྲ་དྲུགྷུ་ཏཾ་ལོ་ཀ་ཡཏ། །ཨ་ཐ་ཏ་ཏཱ་པི་བྷཥྞ་ཎཱ་མ་ཤ་ན་བྷོ་ཛ་ནཥ་ཎེ། །ཏུ་འཁྲུངས་པར་གྱུར་པ་མཐོང་། །དེ་ནས་དེར་ཡང་ཟས་ཀྱི་ཚེ། །དགེ་སློང་རྣམས་ཀྱི་མདུན་དུ་ཟས། །ཨ་ཡོ་ར་ས་ཏྭཾ་སཾ་པྲཏྟཾ་ཛྲ་པཱིཙྪཏྠ་ཏཱནུ་ར། ཏེ་ཏཾ་བ་བྷཱ་ཥི་རེ་མོ་ཧཱ་དཱ་ཡཾ་ལྕགས་ཀྱི་ཁུ་བ་ཉིད་གྱུར་པ། །མཐོང་ནས་དེ་དག་ལ་དེས་དྲིས། །དེ་ལ་དེས་སྨྲས་རྨོངས་པ་ལས། །དགེ་ཏུ་ནཾ་དུ་རཱཏྨ་བྷི། །བྷཀྟ་བི་བྷྱཱཀྲྀ་ཏོསྨྲ་བྷིབྷིཀྵ་ཎཾ་བྷིཀྵ་བྷིསློང་འོངས་པའི་ཟས་ཀྱི་གེགས། །དགེ་སློང་ངན་པའི་བདག་ཉིད་ཅན། །བདག་ཅག་རྣམས་ཀྱིས་སྔོན་པུ་རཱ། །ཨེ་ཏ་དཱ་ཀནྡྷཱ་པ་ཏ་ཏོ་ག་ཏཱ་པ་བྱཱནྟི་པཾ་པུ་ར། །ཀྲྀ་ཏཱི་ཡི་རུ་ཙི་ཚེ་བྱས། །དེ་ཐོས་དེ་ནས་སོང་གྱུར་པ། །དེ་ཡིས་གསེར་ཞིང་མཛེས་པའི་གནས། །གཙུག་ལག་ཁང་ནི་རཱ་བཱ་སཾ་བི་ཧཱ་རཾ་བྷིཥྛ་བྷི་ཀྲྀ་ཏཾ། །བྷི་ཀྵ་བྷཀྵ་ཀ་ནེ་ཏཾ་ཙ་ནི་དགྷཾ་པུ་གསུམ་པ་དག། དགེ་སློང་གིས་གང་མདུན་དུ་སོང་། །དགེ་སློང་རྣམས་ཀྱི་ཟས་ཀྱི་ཚེ། །དེ་ཡང་ཚིག་ཅིང་ན་རུངྒ་ཏཾ། །དྲྀབྷཱུ་ཏཾ་པིསྨི་ཏསཾ་གྷ་མ་པྲིཙྪདཱ་ཧ་ཀཱ་ར་ཎཾ། །ཏེ་ཏ་སླར་ཡང་བྱུང་། །དེ་མཐོང་ཡ་མཚན་གྱུར་པ་ཡིས། །འཚིག་པའི་རྒྱུ་ནི་ཚོགས་ལ་དྲིས། །དེ་ལ་དེས་མཱུ་ཙུཔུ་རཱསྣ་བྷིབྷིཀྵ་མཏྠ་རཱཏྲ་བི་ཧཱ་ར། །ཨཱ་དཱི་པི་ཏཀྲོརླ་ར་ཏིཟི་སྨྲས་དགེ་སློང་ལ། །ཕྲག་དོག་ལས་སྔོན་གཙུག་ལག་ཁང་། །སྔོན་གྱི་སྐྱེ་བར་གདུག་པ་ལ། །དགའ་བ་ཧཱ་རཿཔཱུཏྤ་ཛརྨ་ནི། །ཐུ་ཏཱི་ཏཏྠ་ཏ་ཏོ་ག་ཏཱ་ད་དནཱ་རྱ་ཏྲ་མིཉྩ་ལཱན། །བདག་ཅག་རྣམས་ཀྱིས་བསྲེགས། །དེ་ཐོབ་དེ་ནས་སོང་བ་དེས། །གཞན་དུ་གཡོ་མེད་ཀ་བ་དང་། །།
