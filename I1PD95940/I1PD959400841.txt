དྷི་ནི་ཡ་ཏ་བྱཀྵ་བྷཏྟི་པྲ་ཏཱ་ཀི་ལ་མཎྜི་ཎ། །ཨ་བྷི་མ་ཏ་ཕ་ལ་པཱམྦྷ་ས་དྷཱཚུལ་ལྡན་གུས་པའི་བརྟུལ་ཞུགས་དག་ནི་ངེས་པར་འཛིན། །མངོན་འདོད་འབྲས་བུ་ཐོབ་པས་གྲུབ་པ་དག་པྲ་དནི་ཏ་སིདྡྷ་ཡ༔ཤུ་ཙི་པ་རི་ཙ་ཡོ་དཱ་རཱ་པུ་ཙྪིབྷ་བནྟི་མ་ཧཱི་བྷུ་ཛཱ། །ནི་འཕྲལ་ལ་རབ་ཏུ་སྟོན། །ཡོངས་འདྲིས་ཀྱི་དག་ཅིང་རྒྱ་ཆེ་ས་བདག་རྣམས་ཀྱི་བསོད་ནམས་དག་གིས་ཐོབ། །ཏུཎྞཾ་གུ་རུ་ཏ་རཱ་རཾ་བྷིནྡཱ་ར་ཀོ་སོ་ནི་དགླུ་ཏྭཾ། །ཀཱ་ལ་ཧཱི་ནཔྲ་ཡཙྩོ་མྱུར་དུ་རབ་ལྕི་རྩོམ་པ་ཡིས། །བྱིས་པ་འདི་ནི་རྣམ་པར་ཆོམ། །དུས་དང་བྲལ་བའི་འབད་པ་ནི། །ཏི་པཉྩཏྭཱ་པཱ་ཀེ་བ་ལཾ། །ཨི་ཏི་རཱ་ཛྙཱ་སཱ་མཱ་དིཋཔཱུཟློ་པེ་ཀྵ་བི་ལཀྵ་རྗེས་སུ་གདུང་བྱེད་འབའ་ཞིག་ཡིན། །ཞེས་པ་རྒྱལ་པོས་ཡང་དག་བསྟན། །སྔོན་གྱི་བཏང་སྙོམས་སྐྱེངས་དྷཱི༑། ག་ཛ་བྷ་ཛི་ར་ཐཱ་ནཱི་ཀི་ར་མཱ་ཏྱ་པ་ཧི་ཏོ་ཡ་ཡོ། །ཨ་ཏཱནྟ་རེ་པའི་བློས། །བློན་པོ་གླང་པོ་རྟ་དང་ནི། །ཤིང་རྟའི་ཚོགས་དང་ལྷན་ཅིག་སོང་། །ཆོས་བཟང་གིས་ཀྱང་སུ་དྷརྨཱ་བི་ཀཱུ་ཌྷ་མཱ་ཧུ་ཡ་པུ་ཏྲ་ཀཾ། །ཏཆ་ནཱི་ཎི་ནི་པེ་ཏྭ་སྨི་ག་མྱ་ཏཱ་མི་ཏུ་གནས་སྐབས་དེར། །བུ་ནི་གསང་ནས་བོས་བྱས་ཏེ། །གྲོས་དེ་དེ་ལ་རྣམ་པར་བཤད། །འགྲོ་བར་མཛོད་བཱ་ཙ་ཏྭཾ། །ད་ཏཱ་ཙཱུ་ཌཱ་མ་ཎིམྨཱ་ཏཱ་ས་བི་ཀྲྀཏྠ་སཱུ་རཱ་ཀུ་ལ། །བྲ་ཛན་ཅེས་དེ་ལ་སྨྲས། །མ་ཡིས་གཙུག་གི་ནོར་བྱིན་ནས། །བཏང་བ་དེ་ནི་རིངས་པས་འཁྲུགས། །རིན་ཅེན་དྷུ་རཱ་ད་མ་ཏེ་ན་དྲིཊོ་རཏྣ་བི་བྷུ་ཥི་ཏ། །རཱ་ཛ་པུ་ཏྲས་ཨེ་བཱ་ཡཾ་ནཱ་ན་གྱིས་བརྒྱན་འགྲོ་བ་དག། བློན་པོ་ཡིས་ནི་རིང་ནས་མཐོང་། །རྒྱལ་པོའི་བུ་དེ་མ་གསད་ནས་ནི། །བོས་པ་གཱུ་ཌྷ་པ་ལཱ་ཡ་ཏེ། །ཨི་ཏུཏྟཱ་སྱ་བ་དྷཱ་ཡོ་གཱན་ས་སེ་རྩྭ་གྲྭ་ན་ཙོ་དེ་ཡཏྲ། །འདི་ཉིད་མ་ཡིན་ནམ། །ཞེས་བརྗོད་དེ་ཡིས་དེ་གསད་སླད། །དཔུང་གི་སྔོན་འགྲོ་དྲག་པོ་བསྐུལ། །ཛ་བེ་ན་དྲ་བ་ཏྭཾ་ཏེ་ཛཾ་མཾ་ག་བེ་ས་དྷུ་ར་ག། །ཙམྤ་ཀཱ་ཏྱཱ་སྱ་ནཱ་ག་སྐྱ་མགྱོགས་པས་དེ་དག་ལ་འབྲོས་ཤིང་། །རི་དྭགས་ལྟར་མགྱོགས་རིང་བགྲོད་དེ། །ཙམ་པ་ཀ་ཞེས་བྱ་བ་ཡི། །མཛྫ་ཏ་བྷ་བ་ནཾ་བྷ་སི། །དྲྀཏྠ་ནབྦེ་ཏ་ཏསྟསྨི་སྟནྡྷཏྟཾ་པྲ་ཡཏྟཱ་བཱ་ན། །པ་ད་ཀླུ་ཡི་གནས་ཀྱི་ཆུ་ལ་བྱིང་། །དེ་ནས་མཐོང་ཉམས་དེ་ཡི་ཚེ། །བློན་པོ་ཆེན་པོ་འབད་ལྡན་པས། །རྗེས་སུ་།
