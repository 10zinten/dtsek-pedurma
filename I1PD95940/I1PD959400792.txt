བ་ཀོ་ཏུ་ཀཱ་ཀྲྀཥཱསྟཏྠ་མཱི་པཾ་ཀྲྀ་པཱ་ནི་དྷི། །སོ་ཏེ་ཏྱ་སངྒཱ་ཎཱ་དནསྟན་དན་མཚན་གྱིས་དྲངས་ཏེ། །དེ་དང་ཉེ་བ་བརྩེ་བའི་གཏེར། །ཡོན་ཏན་དམ་པའི་མེ་ལོང་དེ། །སོང་བས་རི་མཾ་གཱི་ཏཱཾ་བྷཾ། །ཡཱ་ཙ་མཱ་ནཱ་བ་རི་ཏཱ་ཎཾ་ལུབྷཱ་ཀཾ་སཱ་སྲུ་ལོ་ཙ་ནཾ། །དྭགས་མིག་དེ་མཐོང་། །མིག་ཚུར་བཅས་པས་རྔོན་པ་ལ། །ཡོངས་སུ་བསྐྱབ་པ་ཞུ་བྱེད་ཅིང་། །བ་ནེ་ཙ་ར་ཀྵ་ཡོདྲི་གྲྭ་མི་བ་ཀཱ་ན་ན་དེ་བ་ཏྭཾ། །ཨ་ནེཧཱུྃ་ལུབྷཱ་ནགས་རྒྱུ་ཟད་པའི་སྡུག་བསྔལ་དང༌། །ལྡན་པའི་ནགས་ཚལ་ལྷ་མོ་བཞིན། །རི་དྭགས་མཚན་པའི་ཚེཀཱ་ཀྲྀཏྟཾ་སྭ་མཾ་ཀ་ཚིགས་ག་མཱ་ག་ཏྭཾ། །ཁིནྣཱམི་བ་བ་ན་བྷཱཙྪཱ་ལཤྲཱི་བ་དཔལ་མོ་ནི། །རབ་བརྟགས་རི་དྭགས་རྔོན་པ་ཡིས། །དྲངས་པ་འཚོལ་དུ་འོངས་གྱུར་པ། །ནགས་ཧ་རི་ཎ་ལགླུ་ཎ། །ཏྭཾ་བི་ལོ་ཀྱ་ས་ཨཱཉྩརྨ་རཱུ་པཱ་ཏི་ཤ་ཡ་བིསྨི་ཏ། །སུ་འཁྱམ་ཞིང་སྐྱོ་བ་བཞིན། །དེ་མཐོང་ཕུལ་དུ་བྱུང་བའི་གཟུགས། །ངོ་མཚར་བཅས་པས་ཡ་མཚན་ཏེ། །ཨ་བྷི་ལཱ་ཥ་པ་ཊེ་ཀྱི་ཕྲོ་ཙི་ཏྲ་རྒྱལྟ་ཨི་བཱ་བྷ་བཏ། །སོ་ཙིནྟ་ཡ་ད་ཧོ་ར་མྱ་མངོན་པར་འདོད་པའི་རས་ལ་ནི། །མྱུར་དུ་རི་མོ་བཀོད་བཞིན་གྱུར། །དེ་བསམས་ཀྱེ་མ་འདི་ཡི་བཞིན། །ནིམྨཱ་ཎཱ་བྷཱ་ས་ཀཱ་རི་ཎ། །ཨསྨིནྡྲ་ཁ་པ་མུམྨེ་ཁེ་རེ་ཁཱ་པ་རི་ཎ་ཏི་ཧྲིདྷི། །ལེགས་པར་འབྲི་ལ་མཛེས་པ་ནི། །རྣམ་པར་སྤྲུལ་བྱེད་གོམས་བྱས་པའི། །བྱེད་པོ་རི་མོ་ཡོངས་སུ་སྨིན། །དུརྨ་བྷཱ་བྷོ་གི་བྷ་བ་ནེ་མརྱ་ལོ་ཀེ་ཀ་ཐི་བ་ཀཱ། །མ་རྩེ་ལཱ་བ་ཏྱཱ་མུ་དྲེ་གདེངས་ཅན་གནས་ན་རྙེད་པར་དཀའ། །མི་ཡི་ཡུལ་ན་གཏམ་ཉིད་ཅི། །མཛེས་སྡུག་རྒྱ་འདི་མཐོ་རིས་ཡཾ་སྭརྒེ་ཥཱ་བྷི་ན་བོ་དི་ཏཱ། །ཏཱ་རུ་ཎྱ་ན་ནི་པཱི་ཏ་ཤེ་ཤ་བ་ཏ་ཡཱ་སུའང་། །གསར་དུ་མངོན་པར་འཁྲུངས་སམ་སྙམ། །གཞོན་ནུ་མ་ཉིད་དར་ལ་བབ་པས་འཐུངས་པས་སཱ་ནཾ་ག་ཐཾ་གཱ་རི་ཎཱི་ཏནྟཾ་གྷཱ༔སཱ་ཀ་ལཾ་ག་སཾ་ག་མ་ས་ཁཱི་བྷཾ་གི་རྣ་ཊཱ་ལུས་མེད་དང་བཅས་སྒེག་ལྡན་ཞིང་། །ལུས་ཕྲ་མཉེན་ལྕེལྕག་གསར་དུ་བསྟར་བྱས་ཡན་ལག་ཀུན་འགྲོགས་གཱ་ཀྲྀ་ཏཱ། །ནིསཾ་རཾ་བྷ་པ་རཱ་ཀྲ་མཔྲ་ཐུ་ཏ་རཱ་རཾ་བྷཱ་བྷི་ཡོ་གཾ་བི་གྲོགས་ཅན་གང་གིས་ནི། །འབད་མེད་ཕ་རོལ་གནོན་ལ་ཤིན་ཏུ་རྒྱ་ཆེར་རྩོམ་པའི་མངོན་སྦྱོར་དང་བྲལ་བར། །།
