ཛ་པུ་ཏྲ་ཛི་ཏེནྡྲི་ཡ། །ཨེ་ཏཱསྟི་ཡ་ཨི་བ་ཀྱི་ཛྲ་ས་མཱ་ག་སུ་ཁཱཤི་ཡ། །པོའི་སྲས་ནི་དབང་པོ་ཐུལ། །བུད་མེད་བདེ་དང་དཔལ་འབྱོར་འདི། །མྱུར་དུ་ཡང་དག་འགྲོགས་པ་བཞིན། །ཀྲྀ་ཡ་ཏྭཾ་ཀུ་ཤ་ལཱ་ཡི་བ་བྲཧྨ་ཙརྨ་པ་རི་གྲ་ཧ། །རྱ་ཛ་ཏཱ་མེ་ཥ་ནིསཱ་ར་གྲོ་དགེ་བ་ཉིད་སླད་ཚངས་པ་ཡི། །སྤྱོད་པ་ཡོངས་སུ་བཟུང་བར་མཛོད། །འཁོར་བ་སྙིང་པོ་མེད་པའི་ཁྱིམ། །ཧ་སཾ་སཱ་ར་དུརྒྱ་ཧ། །ཨི་ཏི་ཐུ་ཏཱ་བྷ་ག་བ་ཏཀ་རུ་ཎཱ་བྷ་ར་ཎཾ་གདོན་ངན་འདི་ནི་གཏང་བར་མཛོད། །བཅོམ་ལྡན་གསུང་ནི་སྙིང་རྗེ་ཡིས། །བརྒྱན་པ་འདི་སྐད་ཐོས་བ་ཙ། །པྲ་ཏྱཱ་བྷཱ་ཥ་ཏ་ཏནྣནྡཔཱུ་ཛུ་པྲ་ཎ་ཡ་ཡནཱི་ཏ། །ས་དཱ་གྱུར་ནས། །དོན་གཉེར་མཆོད་འོས་ཀྱིས་གདམས་པའི། །དགའ་བོས་དེ་ལ་ལན་གསོལ་པ། །རྟག་ཏུ་བྷ་བ་དུ་པཱ་ཡི་བ་པྲ་བྲ་ཚུ་བྷ་ག་བཱརྨ་མ། །བྷི་ཀླུ་སཾ་གྷོ་པ་ཀཱ་རཱཉེ་གྲོ་ཁྱོད་ཉིད་བསྙེན་པ་ཉིད། །བཅོམ་ལྡན་བདག་གི་རབ་བྱུང་ཡིན། །དགེ་སློང་དགེ་འདུན་རིམ་གྲོར་ནི། །ཧ་ཨེ་བཱ་ད་ར་པ་རཾ། །ཨི་ཏུ་ཏྟཱབྷ་ག་བ་བཱཀྱ་མ་ཏི་ཀཱནྟུ་མ་ནཱི་ཤྭ་འོས་པའི་ཁྱིམ་ལ་མཆོག་ཏུ་གུས། །དེ་སྐད་སྨྲས་ནས་བཅོམ་ལྡན་གསུང་། །བསྒོམ་བྱའི་དབང་ཕྱུག་མིན་པ་རཿ༑། ཀྲི་ཤྩ་མཱ་ཎཔྲ་ཡཱ་པྲསླ་སོ་བྷུདྡོ་ལཱ་ཀུ་ལཱ་ཤ་ཡ། །པུ་ན་:པུ་ནཉྩ་དང་། །དགའ་མ་ལ་དགས་དྲངས་པས་འདིའི། །བསམ་པ་ཁྱོགས་ལ་ཕྱང་མོ་ཡུག། ཡང་དང་ཡང་དུ་ཀཱ་རཱ་སྱ་བྷ་ག་བཱན་བྲ་ཏ་དེ་ཤ་ནཾ། །ཨུ་པ་ཀཱ་རོདྨ་ཏཱསཾ་ཏཐིནྟ་བཅོམ་ལྡན་གྱིས། །འདི་ལ་བརྟུལ་ཞུགས་ཉེར་བསྟན་མཛད། །ཕན་པར་བརྩོན་པའི་དམ་པ་རྣམས། །མི་ཡནྟཾ་ན་ཡོ་གྱ་ཏྭཾ། །ཡ་དཱ་ནེཙྪ་ཏི་ནནྡཙཱ་པྲ་བྲཀྵ་མ་ཛེ་ཏིནྡྲི་ཡ། །ཏ་དཱ་སྱ་འོས་པ་ཉིད་མི་བསམ་མོ། །དབང་པོ་མ་ཐུལ་དགའ་བོ་ནི། །གང་ཚེ་རབ་བྱུང་མི་འདོད་པ། །དེ་ཚེ་དེ་བྷ་ག་བདྷཱ་ཀྱ་ས་པ་ཏཏ་བ་པུ་ཥི་སྭ་ཡཾ། །མུཎྞཀཱ་ཥཱ་ཡ་པཎྞ་ནཔཱ་ཡིས་ལུས་ལ་དེ། །བཅོམ་ལྡན་གསུང་གིས་རང་ཉིད་ལྷུང་། །དེར་ནི་དེ་ཚེ་ངུར་སྨྲིག་གོས། །མགོ་ཏྲ་བཱ་ཎཏ་ཏ་གྷ་ཎ། །བ་བྷོ་ཙ་མཱི་ཀ་ར་རུ་ཙིརྨ་ཧཱ་པུ་རུ་ཥ་ལཀ་ཎི། །ཟླུམ་ལག་ན་ལྷུང་བཟེད་ཅན། །སྐྱེས་བུ་ཆེན་པོའི་མཚན་རྣམས་ཀྱིས། །གསེར་གྱི་འོད་ཟེར་ལྟ་བུར་མཛེས། །།
