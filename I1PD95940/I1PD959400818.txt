་ཏཱི་ཙཀྲནྟསྟོ། །ཏ་ཏཔྲཙྪནྣཱ་བྲྀཏྟཱནྟཾ་ལཛྫ་མཱ་ན་མ་ནོ་ཧ་རཱ། །བི་གཉིས་དག་གིས་བྱས། །དེ་རྗེས་སྤྱོད་ཚུལ་བསྒྲིབས་པ་ནི། །ངོ་ཚ་ལྡན་པའི་ཡིད་འཕྲོག་མས། །བཤད་བེ་ཏྭ་དན་ཡཏྲི་ཌོཔ་ཏི་བྷཱུ་མི་མ་ནོ་བྷ་ཐཾ། །ཏ་ཏཀིནྣཱ་རཱ་རཱ་ཛཙཱ་ཀོ་པ་པྲ་སྤུ་རི་ནས་ཕ་མ་དག་ལ་ནི། །བདག་པོ་ས་ཡི་ཡིད་འབྱུང་བསྟན། །དེ་ནས་མི་འམ་ཅི་ཡི་རྒྱལ། །དེ་ལ་ཁྲོས་པས་ཏཱ་དྷ་ར། །ཨུ་བཱ་ཙ་དན་ན་པ་ཐི་པ་རིབྲྀ་ཏྭཱཏྨ་ཛཱ་བ་ཏེ། །།ཨ་ཧོ་པྲ་མཆུ་བསྐྱོད་ཅིང་། །བུ་མོའི་བདག་ནི་མཐོང་བའི་ལམ། །ཡོངས་སུ་བཏང་ནས་རབ་སྨྲས་པ། །ཀྱེ་མ་མི་མཱ་དཱ་པ་ཏི་ཏཱ་དེབ་དཱ་ནུ་ཙི་ཏེ་ཛ་ནེ། །ན་བི་མུཉྩ་པི་དུབྲྀཏྟེ་ཀྵ་ལ་མཱ་ནཱ་འོས་སྐྱེ་བོ་ལ། །ཆ་ཡིས་བག་མེད་ལྷུང་གྱུར་པ། །སྤྱོད་ཚུལ་ངན་པ་བཀྲུས་ཀྱང་ནི། །ཆགས་པ་ཉིད་ནི་པི་རཀྟ་ཏྭཾ། །ཡོ་བ་ནཱ་བ་ནི་ལཱ་བ་ཙྩཾ་ཀྲྀ་ཧ་ཎཱི་ཡཱ་དི་བོ་ཀ་སཾ། །བ་ཏ་པྲ་ཡཱ་ཏ་ཁྱོད་མི་གཏོང་། །ཁྱོད་ཀྱི་མཛེས་སྡུག་ལང་ཚོ་ལ། །ལྷ་རྣམས་ཀྱིས་ནི་རེག་བྱ་འདི། །ཀྱེ་མ་མི་དང་མཛའ་མ་ཏཏེ་མརྐྱ་སྣེ་ཧེ་ན་ཤོ་ཙ་ཏྭཾ། །ཨུ་ད་གྲ་གོ་ཏྲི་པྲ་བྷ་བཱ་གྷ་ན་ཡོ་བ་བ་ཡིས། །མྱ་ངན་བྱ་འོས་ཉིད་དུ་གྱུར། །མཐོན་པོའི་རིགས་ལས་སྐྱེས་གྱུར་ཅིང་། །སྟུག་པོའི་ལང་ཚོ་ན་ནིབྷ་རཱ། །ནིསྣ་གེ་ཀྵ་བྷ་བི་བྷཏྟཱ་ཡཱ་ཏཱ་སི་སུ་ཏ་རཱ་མ་དྷ། །བིདྨ་དུངྒེ་རབ་རྒྱས་པ། །ཐུར་འགྲོ་འཁྲུགས་པས་རྣམ་ཉམས་པ། །ཁྱོད་ནི་ཤིན་ཏུ་འོག་ཏུ་འགྲོ། །མཁས་པ་ཆགས་ག་ཛ་ན་ནཱི་ཀུ་ལ་བ་ལཀྱུ་ཀཱ་རི་ཎཱི། །མ་ལི་ན་ཁ་ལ་བི་དྱེ་བ་སསྨྲ་བྲལ་སྐྱེད་བྱེད་ཅིང་། །རིགས་ནི་སྐྱེངས་པར་བྱེད་པ་མོ། །མི་བསྲུན་རིགས་མིབཞིན་དྲི་མ་ཅན། །ཁྱོད་ཏཱ་པི་ན་ཀ་སྱ་ཙིཏ། །རླུ་པ་མཱ་ཏྲེ་ཎ་ཏུཏྟཱ་ཐ་ཡ་དི་མཅུ་བ་ཤཾ་ག་ཏཱ། །ཏཥིནྣ་ནི་སུ་ཡང་མི་འདོད་དོ། །གཟུགས་ཙམ་གྱིས་ནི་ཁྱོད་ཚིམ་ནས། །གལ་ཏེ་མི་ཡི་དབང་གྱུར་ན། །གསེར་ར་མཏྠ་ཧེ་མ་ནིརྨ་ཎ་པུ་རུ་ཥ་ཏཱི་ཥ། ་། པྲ་བྷཱ་བ་གུ་ཎ་ཧཱི་ནཱ་སྱ་ལས་སྤྲུལ་པ་སྐྱེས་བུ་ཡི། །གཟུགས་དང་རྩེ་དགའ་ཅིས་མི་བྱེད། །མཐུ་དང་ཡོན་ཏན་གྱིས་དམན་པ། །པུཾ་སཾཉྩ་ཏུ་ཏ་ར་ཀྲྀ་ཏེ། །ཨཱ་ལེ་ཁྱུ་པུ་རུ་ཥ་སྤྱི་བ་སོནྡམཾ་བྷིཏྟི་རཾ་ཛ་ནཾ། །སྐྱེས་བུ་རབ་མཛེས་དབྱིབས་ཅན་གྱི། །མཛེས་སྡུག་རི་མོར་བྲིས་པ་ཡི༄༅། །སྐྱེས་བུ་ཡི་བཞིན་རྩིག་པ་མཛེས། །།
