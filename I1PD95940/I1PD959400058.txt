ཀ་ལ་ནཱ་ཀལྤ་པཱ་ད་པ། །སྨྲ་ཏྭཱ་མུ་ནེརྦ་ཙཀཱ་ལེ་ཀརྟུ་བི་པུ་ལ་བགྱིད་པའི་དཔག་བསམ་རྐང་འཐུང་ནི། །ས་བདག་བདག་གིས་ཐུབ་པའི་ཚིག། དུས་སུ་བྱ་བར་དྲན་དཀི་ཎཾ། །ཨ་ཧི་སཱ་བ་སུ་སཾ་པུཎཾ་མཱ་ཛ་ཧཱ་ར་མ་ཧཱི་པ་ཏི། །ཏསྨིན་ཡ་དྷེ་གྱུར་ནས། །མི་འཚེ་ནོར་ནི་ཡོངས་རྫོགས་ཤིང་། །ཡོན་ནི་ཡངས་པའི་མཆོད་སྦྱིན་བྱས། །བྷཾ་གུའི་བུ་ས་མཱ་ཛགླུ༔སརྔ་ཀཱ་མིཆི་རརྒ་ལེ། །མུ་ན་ཡོ་བྷན་བ་མུ་ཁཱ་སོགས་ཐུབ་པ་དང་། །མི་བདག་རབ་ཏུ་བཟོད་དཀའ་སོགས། །འདོད་པ་ཀུན་གྱིས་མ་བསྡམས་པའི། །ཚིགས་པཱ་དུཀྵ་ས་ཧཱ་ད་ཡ། །བརྟ་མཱ་ནེ་མ་ཁེ་ཏསྨིན་ནིསཾ་ཁྱ་བ་སུ་མཆོད་སྦྱིན་དེར་ནི་ཡང་དག་འོངས། །གྲངས་མེད་ནོར་གྱི་ཆར་འབེབས་པའི། །སྦྱིན་སྲེག་མངོན་དུ་གྱུར་བརྩི་ཎི། །རཀྵ་རཱུ་པ་ས་མུཏྟསྦྱོ་བ་སྟི་མ་དྷཏ་སུ་རེ་ཤྭ་ར། །ཨུ་པ་སྲི་པ་དེའི། །མེ་ཡི་དབུས་ནས་ལྷ་ཡི་དབང་། །སྲིན་པོའི་གཟུགས་སུ་ཡང་དག་ལངས། །རིད་ཅིང་རྣམ་ཏྱ་ས་བྷཱུ་པཱ་ལཾ་ཀྲྀ་ཤོ་བི་ཀྲྀ་ཏ་བི་གྲ་ཧ། །གླུཏྲི་པཱ་སཇི་ཏོམི་ཏི་ཡ་ཡཱ་འགྱུར་ལུས་ཅན་དེ། །ས་སྐྱོང་ལ་ནི་ཉེར་བཏུད་དེ། །བདག་ནི་བཀྲེས་སོ་སྐོམ་མོ་ཞེས། །བཟའ་དང་ཙེ་པཱ་ན་བྷོ་ཛ་ནཾ། །ཤཱ་ས་ནཱ་ད་ཐ་བྷུ་བྷརྟུསྟསྨེ་བི་བི་དྷ་བྷོ་ཛ་ནཾ། །བཏུང་བ་སློང་བར་གྱུར། །དེ་ནས་ས་ཡི་བདག་པོའི་བཀས། །རིམ་གྲོ་བ་ཡི་འཁོར་རྣམས་ཀྱིས། །ཨུ་པ་ནི་རླུཔ་རི་ཙི་ཏཱཔཱ་ནཾ་ཙ་པ་རི་ཙཱ་ར་ཀཱ། ་། ཏ་ཏཀིཉིད་བི་ཧ་སྤྱི་བཟའ་བ་རྣམ་པ་སྣ་ཚོགས་དང༌། །བཏུང་བ་དག་ཀྱང་ཉེ་བར་བསྟབས། །དེ་ནས་སྲིན་པོས་ས་བདག་ལ། །བ་ཀྱི་ཏི་པཾ་ས་ཧ་རཱཀྵ་ས། །ནེ་ད་མསྨཏ་པྲི་ཡཾ་རཱ་ཛན་བ་ཡཾ་ཧི་པི་ཤི་ཏཱ་ཅུང་ཟད་དགོད་བཞིན་རབ་ཏུ་སྨྲས། །རྒྱལ་པོ་འདི་ལ་བདག་མི་དགའ། །གང་ཕྱིར་བདག་ནི་ཤ་ཡིས་ཤ་ནཱ། །ས་བྱོ་ཧ་ཏ་སྱ་ཚྭ་སེ་ན་རུ་དྷི་རེ་ཎ་ཙ་བྷཱ་ཡ་པཱ། །ཏྲཔཱི་རུཏྤ་བྱ་ཏེ་འཚོ། །འཕྲལ་ལ་བསད་པའི་ཤ་དང་ནི། །ཤིན་ཏུ་མང་བའི་ཁྲག་དག་གིས། །བདག་ཅག་ཚིམ་པ་རབ་སྣ་ཀཾ་དཱི་ཡ་ཏྭཾ་ཡ་ད་བྷིསི་ཏཾ། ་། སརྦ་ཀཱ་མ་པྲ་དོ་སཱི་ཏི་ཏྭཱ་མ་ཧཾ་ས་ཏུ་སྐྱེ། །གང་ཞིག་འདོད་པ་གནང་བར་མཛོད། །འདོད་ཀུན་རབ་ཏུ་སྟེར་རོ་ཞེས། །ཁྱོད་ཅན་བདག་ནི་།
