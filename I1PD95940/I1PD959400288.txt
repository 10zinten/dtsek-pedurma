བི་མ་ལ་ནུ་གྲ་ཧཾ་དི་ཧཱུཾ། །ཏ་ཏསྣ་ཏྭཱ་ས་མཱ་ཡཱ་ཏསྣུཎྞཾ་པ་ཏི་ར་རྗེས་སུ་བཟུང་བ་མཛད་བཞིན་གཤེགས། །དེ་ནས་བདག་གི་བདག་པོ་འདི། །ཁྲུས་བྱས་མགྱོགས་པར་ཡང་ཡསྨྲ་མ། །ཏཏྲིཎྜ་པཱ་ཏ་མཱ་ཀཉྩ་ཕྲ་མོ་ད་ཡ་མ་ཡོ་དི་ཏཾ། །ཀོ་པཱ་དག་འོངས། །རབ་ཏུ་དགའ་སླད་བདག་གིས་སྨྲས། །བསོད་སྙོམས་ཕུལ་བ་དེ་ཐོས་ནས། །ཁྲོ་ལས་དུ་བཱ་ཙ་མཱི་ཀསྨཱ་དྲི་ཤི་ཁཤྲ་མ་ཎཤ་ཋ། །ཨ་པཱུ་ཛི་ཏེ་ཥཱ་པཱུ་ཛཱི་ཥཱ་བདག་སྨྲས་མཆོད་འོས་རྣམས། །མཆོད་པ་མེད་པར་དགེ་སྦྱོང་ནི། །སྒྲ་མེད་གཡོ་ལྡན་གྲ་ཕུབ་འོས། །བུ་ཥཱརྡཔཱུ་ཛི་ཏནྡྲ་ཡཱ། །ཨི་ཏི་མོ་ཧཱ་ད་ནེ་ནོལྟེ་པུ་ཊོ་པེ་ཥ་ཛ་གཱ་ད་ནཱཾ། །ཁྱོད་ཀྱིས་ཅི་ཡི་ཕྱིར་དུ་མཆོད། །རྨོངས་ལས་འདི་ཡིས་དེ་བརྗོད་ཚེ། །དེ་བཞིན་བུས་ཀྱང་བདག་ལ་སྨྲས། །པཱ་ཀེ་པཱུནྡཱ་ཤ་ན་ཡོ་གྱས་ཀིནྣཱབྷཱཏྱཱ་ཡོ་གུ་ཌཱ་ཎ། །ཨི་ཡཾ་སྐུ་ཥཱ་མེ་ས་ཏ་ཏཾ་ཚོས་པ་ཐོག་མར་ཟ་འོས་ཏེ། །ལྕགས་ཀྱི་ཐོ་ལུམ་ཅིས་མི་ཟ། །བདག་གི་མནའ་མ་འདི་རྟག་ཏུ། །སྔོན་པུཏྠ་བྷ་གྷཱ་བ་ལོ་ཀི་ནཱི། །མ་ཡོཀྟ་ཤ་པ་ཐོ་ཙ་ཀྲེ་སཱ་མཱ་སཱ་ད་ན་པཱ་དི་ནི། །ཨི་ལ་ཟས་ལ་ལོངས་སྤྱོད་ཅིང་། །བདག་གིས་བརྗོད་ལ་མནའ་བྱེད་པ། །རང་ཤ་ཟ་ཞེས་སྨྲ་བ་མོ། །འབང་ཡཾ་ཙ་དཱ་སཱི་བྷོདྨ་ཎཾ་ཙོནཱཏ་བྱ་ཡ་ཀཱ་རི་ཎཱི། །ཨཱ་ཀཱིཔ་ས་ཏྱཾ་ན་ཀ་རོ་ཏ་པཱུ་ཡ་མོ་འདི་ཡང་ཟས་དག་ལ། །རྐུ་ཞིང་ཉམས་པ་དག་ཏུ་བྱེད། །སྨད་ཚེ་བདེན་པར་བྱེད་པ་ནི། །རྣག་ཁྲག་ཤོ་ཎི་ཏ་བཱ་དི་ནཾ། །ཏ་ཏྲ་ཏེ་པྲ་ཏ་ཏྭཾ་ཡཱ་ཏཱསཱ་བཱཀྱ་ས་དྲི་ཤཱ་ས་ནཱ། །ཨ་ཧཾ་ཟ་ཞེས་སྨྲ་བྱེད་ཡིན། །དེ་དག་ཡི་དྭགས་འདི་དག་གྱུར། །རང་གི་ངག་དང་མཚུངས་པ་ཟ། །བདག་ནི་ཏཱརྱ་པྲ་སཱ་དེ་ན་དི་བྱ་བྷོ་གོ་ས་བྷོ་གི་ནི། །ཏྭཱ་ཡཱ་སཱ་དེ་ཤེ་ནཱསྟེ་ན་འཕགས་པའི་བཀའ་དྲིན་གྱིས། །ལྷ་ཡི་ལོངས་སྤྱོད་ཉེ་བར་སྤྱོད། །ཁྱོད་ཀྱིས་རང་ཡུལ་ཕྱིན་ནས་ནི། །བལྟ་བྱ་དུ་ཧི་ཏཱམ་མ། །སནྟི་ཧེ་མ་ནི་དཱ་ནཱ་ནི་ཧུཾ་ཧེ་ཙ་ཏྲ་རི་ཏེ་པི་ཏུ། །ཏ་བདག་གི་བུ་མོ་དག་ལ་སྨྲོས། །ཁྱོད་ཀྱི་ཕ་ཡི་ཁང་པ་ན། །གསེར་གྱི་གཏེར་ནི་རྣམ་བཞི་ཡོད། །དེ་ཅུ་ཛྙཱ་ཏྱ་ཡ་ཐཱ་ཡོ་གཾ་བྷ་ཛ་སྭཱ་སཱ་ཛཆ་སྦི་ཏི། །པཱུ་ཛ་ནཱི་ཡཔི་ཏུནྟཱ་ཏུ་དག་ཐོན་ལ་ཅི་རིགས་པར། །རང་གི་སྐྱེ་བ་གནས་པར་བྱོས། །ཕ་དང་མིང་པོའི་མིང་དག་གིས། །ཀཱ་།
