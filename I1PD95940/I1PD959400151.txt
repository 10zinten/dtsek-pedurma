མཱ་ནཱ་ཡྻ་ཏད་བེཧྥ་ཏེ་ནི་བ་ས་ཧི་ཏོ་ཡ་ཡོ། །ས་ཏ་ཏཱནྟསུ་རེ་ནི་ཏྱཾ་ས་བཀུག་སྟེ་དེའི་ཁྱིམ་དུ། །དེ་ཉིད་དང་ནི་ལྷན་ཅིག་སོང་། །ཕོ་བྲང་ནང་དེར་རྟག་ཏུ་དེ། །ས་བདག་བྷཱརྨེ་ཎ་མཾ་ཧཱི་བྷུ་ཛཱ། །པཱུ་ཛུ་མཱ་ན་པ་རི་ཙ་ཡཱ་ཨུ་བཱ་ས་བི་ཀྲྀ་ཏཱ་ད་ར། །བཙུན་མོར་®བཅས་པ་ཡིས། །མཆོད་ཅིང་ཡོངས་སུ་འདྲིས་པ་ལས། །གུས་པ་རྣམ་པར་སྤྲོ་ཞིང་གནས། །སྣེགྷ་སཱུ་ཚྭ་བུ་ད་ཏེ་ཥཱ་པཱ་ལཱི་ལ་མ་ཎི་བེཧྥ་སུ། །ཙི་ཏྲ་པ་ཏྲ་རུ་ཙཱ་ཙཀྲེ་མཐོན་མཐིང་ནོར་བུའི་ཁང་པ་ནི། །འཇམ་ལྡན་ཆུ་འཛིན་འོད་དེ་ལི། །འདབ་མ་མཛེས་བཀྲ་དབང་སཾ་སཏྟེན་ཡུ་དྷ་བྷ་མཾ། །ཨ་ཐ་དིག་ཛ་ཡ་ཡཱ་ཏྲ་ཡཾ་ཀ་དཱ་ཙིད་བ་སུ་དྷཱ་དྷི་པོའི་གཞུ། །ཆགས་པ་ལྟ་བུར་འཁྲུལ་པ་བྱས། །དེ་ནས་དེ་ཡི་རིམ་གྲོ་ལ། །གུས་དང་བཅས་པས་ལྷ་མོ་པ༑། ཡ་ཡོ་ཏ་དུ་པ་ཙཱ་རཱ་ཡ་དེ་བཱི་མཱ་དི་བྱ་སཱ་ད་ར། །ཏ་ཏཝཱ་ནུ་པ་མཱ་དེ་བཱི་བསྐོས། །ཕྱོགས་ལས་རྒྱལ་བའི་བགྲོད་པ་ལ། །ནམ་ཞིག་ས་ཡི་བདག་པོ་སོང་། །དེ་ནས་ལྷ་མོ་དཔེ་མེད་མ། །པ་ཁྱོ་ཡཱ་ཏེ་བྲ་མཱ་དི་ནི། །རཱུ་པ་ཡོ་བ་ན་དབཱནྟཱ་ནཱ་ལུ་ལོ་ཀེ་ཀུ་བདག་པོ་སོང་བས་བག་མེད་ཅིང་། །གཟུགས་དང་ལང་ཚོའི་དྲེགས་པས་ལོང་། །རིགས་ཀྱི་ལུགས་ནི་ལ་སྦི་ཏི། ་། ཏ་རུ་ཎཾ་པྲ་སྒྱུ་རཱ་གི་ཙྪཱསྟ་སྱཱཀཾ་དརྦ་བི་བླ་བེ། །བྷཱུ་ཡ་མཐོང་མ་གྱུར། །གཞོན་ནུ་འབངས་ལ་ཆགས་ལྡན་མ། །འདོད་པས་རྣམ་པར་བསླད་མ་དེའི། །ངོ་ཚ་ཤིན་པ་ལཾ་བྷ་བྷི་ཏེ་བ་ལཙྪཱ་དྷུ་ར་ཏ་རཾ་ཡ་ཡོ། །མ་ལི་ནཀུ་ཌི་ལཙཱིཀྐུཀཎ་ཏུ་རིང་སོང་སྟེ། །ཡང་ཡང་དམིགས་པས་འཇིགས་པ་བཞིན། །དྲི་མ་ལྡན་ཞིང་གྱ་གྱུ་རྣོ། །རྣ་བ་དག་སཾ་སྤན་ནོ་ཙི་ཏ། །ཙ་བ་ལ༔་པ་ལཱཀྵི་ཎཾ་སུ་དྲི་ཤཱ་ས་དྲྀ་ཤཀྲ་མ། །བི་བི་ལ་རེག་པར་བཟོད། །གཡོ་བ་གཡོ་བའི་མིག་ཅན་དང་། །མིག་ནི་ལེགས་པ་རིམ་པ་མཚུངས། །འཁོར་བ་དྷོདྨ་ད་ཀཱ་རི་ཎྱསཾ་སཱ་ར་མ་ཀ་རཱ་ཀ་རེ། །ཙ་རནྟི་པྲ་ཎ་ཧཱ་རི་ཎྱཀཱ་ལ་ཆུ་སྲིན་འབྱུང་གནས་སུ། །རྣམ་པ་མང་པོས་མྱོས་བྱེད་ཅིང་། །སྲོག་འཕྲོག་ནག་པོ་བརྩེགས་པ་ཡི། །རི་ཀཱུ་ཌ་ཙྪ་དཱསྟི་ཡ། །ཀུ་སུ་མཱཏ་སུ་ཀུ་མཱ་ར་སྱ་ཀྲུ་ར་སྱ་ཀྲ་ཀ་ཙཱ་ད་པ། །ཀོ་མོ་བུད་མེད་རྣམ་པ་རྒྱུ། །མེ་ཏོག་པས་ཀྱང་ཤིན་ཏུ་འཇམ། །སོག་ལེ་བས་ཀྱང་ཤིན་ཏུ་རྩུབ། །བུད་།
