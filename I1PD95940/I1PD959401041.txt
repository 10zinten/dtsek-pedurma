ཅུ་ན་བྷ་བོ་ཧི་དུས་ཧ། །ཨ་ནཱ་དི་ཀཱ་ལོ་པ་ཙི་ཏཱཀླུ་ཤ་ཛ་ཛ་རཱ་ད་དམན་པའི་དངོས་པོ་བཟོད་པར་དཀའ། །ཐོག་མེད་དུས་སུ་བསགས་པ་ཡི། །ཉོན་མོངས་སྐྱེ་དང་རྒ་ཡ༑། ཏནུ་པ་ཡཤཱཀྱནྟེ་པྲལྟ་ཚིགསདྷི་པ་ད་ཀི་ཡཏ། །ཨི་ཏི་ཙིན་བ་སོགས། །འབད་པ་ཅན་གྱིས་གཞོམ་ནུས་ན། །རྫུ་འཕྲུལ་གནས་ནི་ཐོབ་ཅེལ་ཅི། །ཆགས་བྲལ་དྲག་ཡ་ཏ་སྱ་ཏཱི་བྲ་སཾ་བེ་ག་ཡཱ་དྷི་ཡཱ། །པཱ་དུ་རཱ་སཱིཥྛ་ཎེ་ནི་བ་མ་ཧཱདྷི༔ཤུདྡྷ་པོའི་བློ་གྲོས་ཀྱིས། །དེ་ལྟར་སེམས་ཤིང་དག་པ་ཡིསི། །སེམས་ལྡན་དེ་ལ་སྐད་ཅིག་གིས། །རྫུ་འཕྲུལ་ཙེ་ཏ་ས། །ཨ་ཐ་རཱ་ཧཱུྃ་བྱ་ཏཱི་ཏཱ་ཡཾ་པྲ་བྷཱ་ཏེ་སརྦ་བྷིཀྵ་བ། །ནཱ་ནཱ་ཏཱི་ཆེན་པོ་རབ་ཏུ་སྐྱེས། །དེ་ནས་མཚན་མོ་འདས་གྱུར་ཏེ། །སྔ་དྲོ་དགེ་སློང་ཐམས་ཅད་ནི། །སྣ་ཚོགས་ལྷ་ད་ཤ་བེ་ཤེ་ན་བི་མཱ་ནིརྣ་བྷ་པཱ་ཡ་ཡུ། །ཨ་ཏཱན་རེ་མ་ཧཱ་རཾ་བྷ་ཡི་ཆས་ལྡན་པའི། །གཞལ་ཡས་ཁང་ནས་མཁའ་ལ་གཤེགས། །སྐབས་དེར་རྩོམ་པ་ཆེན་པོ་ལ། །སཾ་བྷཱ་ར་པ་རི་པཱུ་ར་ཏེ། །སུ་མཱ་ག་དྷཱ་བྷནྡཾ་གཾ་ཧེ་བྷ་ག་བརྡན་ནོཏུ་ཚོགས་ཀྱིས་ཡོངས་སུ་གང་གྱུར་པ། །བདག་པོའི་ཁྱིམ་དུ་བཅོམ་ལྡན་འདས། །ལྟ་འདོད་མ་ག་དྷ་བཟང་ཀཱ༑། ས་ཧ་པྲ་པཱ་ད་མཱ་རུ་བྱ་ཤྭ་ཤུ་ར་བྷ་ནཱཾ་བྷི། །ཏསྦོ་ཀུ་སུ་མོ༑། མེ་ཏོག་བདུག་པའི་མཆོད་པ་ཡི། །བཀོད་པ་འཛིན་ལ་མངོན་ཕྱོགས་ཤིང་། །སྒྱུག་མོ་གྱོས་པོ་ཁྱོ་མ་དྷ་པ་ཙཱ་ར་ཙ་ནཾ་སཾ་གྲ་ཧོ༔་ཁཱི། །བྷིཀྵ་རཱ་ཛཱ་ཏ་ཀོཎྜི་རྒྱསྟ་ཏོ་རྣམས་དང༌། །ལྷན་ཅིག་ཁང་བཟང་ལ་འཛེགས་འཁོད། །དེ་ནས་རྟ་ཡི་ཤིང་རྟར་གནས། །དགེ་ཤཱ་ར་ཥ་མ་སྦི་ཏ། །དི་བྱཌྷ་བི་བི་བྷཱརྨ་པྲ་ཐ་མ་པྲ་ཏྱ་པྲྀ་ཏ། །ཏཾ་སློང་ཀུན་ཤེས་ཀཽཎྜི་ནྱ། །རྣམ་མང་ངོ་མཚར་རྫུ་འཕྲུལ་ནི། །མཆོག་ལྡན་དང་པོ་རབ་ཏུ་མཐོང་། །ཉི་དྲྀབྷྱཱ་སཾ་ཀཱ་་བིསྨི་ཏཱཤཱ་ཤུ་རཱ་ད་ཡ། །ཨུ་ཙུསུ་མཱ་ག་དྷཱ་པཱི་མ་དང་མཚུངས་དེ་མཐོང་ནས། །།ཡ་མཚན་གྱིས་སོགས་དགའ་བ་ཡིས། །མ་ག་དྷ་བཟང་ལ་སྨྲས་པ། །ཏྭཱ་ཀི་མེ་ཥ་བྷ་ག་བཱ་ནི་ཏི། །སཱ་བ་དབྷ་ག་བཱནྟཱ་ཡཾ་ཛྲ་བྱ་ཏེ་ཏ་ར་ཎི་པྲ་བྷ། །བཅོམ་ལྡན་ཞེས་པ་འདི་འམ་ཅི། །དེས་སྨྲས་འདི་ནི་བཅོམ་ལྡན་མིན། །དགེ་སློང་ཀུན་ཤེས་ཀཽཎྜི་ནྱ། །།
