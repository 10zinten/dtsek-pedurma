རསྨི་སཱརྦ་ཤ་ར་ཎི༔སཱ་ཀཱ་པི་བཀྵརླུ་ཏིཔུནྡརྨ་ཀ་ཐ་མེ་ཏི་ཙི་ཏྲ་པཱ་ད་བཱི་སྐྱོན་མེད་བདུད་རྩིའི་ཟེར་གྱི་དོན་མཐུན་དག་གི་ལམ། །མཛེས་མ་ལུས་ནི་ངོ་མཚར་སྙིང་པོ་རི་མོའི་མཉྩརྱ་པཱ་རཾ་བ་པུ། །ཨི་ཏེ་ས་ཙིནྟ་པ་ཤ་ནི་ར་ལི་ལེ་ཁ་ཤི་ལཱ་ཏ་ལམ་དུ་ཇི་ལྟར་འགྲོ་བར་འགྱུར། །འདི་བསམས་དེ་ཡིས་རྡོ་གཞི་ལ། །མཆི་མས་བཀྲུས་ཤིང་འདར་བ་ལེ༑། པུཎྜ་རཱི་མ་ཀ་ཐ་མ་སྱ་ཤུ་པ་ཏ་ཀཾ་པཾ་ཀུ་ལོ་གུ་ལི། །ས་སཾ་ཀལྤ་ཡིས། །འཁྲུགས་པའི་སོར་མོས་ཇི་ལྟར་ཡང་། །མཛེས་མ་དག་ནི་དལ་བུས་བྲིས། །དེ་ཡིས་རྟོག་པས་ས་མུད་གཱིཙྪཱ་པྲ་ཏི་བིསྨྲ་མི་བཱ་ཤི་ཏྭཾ། །ཀྲྀ་ཏྲ་པུ་རཿཔྲ་ཡ་ཏ་མཱ་མ་ཙཱ་གསལ་བྱས་པའི། །གཟུགས་བརྙན་ལྟ་བུ་ལ་བརྟེན་པར། །སྔོན་དུ་བྱས་ནས་མཛེས་མ་ལ། །མཆི་མས་དཝཱཋ་གངྒ་ད། །པྲ་ཡཱ་མཱ་ལི་ཁྱ་ཧཾ་ནི་ཁི་ལ་སུ་ཁ་བྲོཋི་ན་ཡ་ན་ཡོན་འཛེར་འཛེར་སྐད་དུ་སྨྲས། །མིག་གཉིས་མ་ལུས་བདེ་བའི་ཆར་རྒྱུན་སྟོན་ཀའི་ཟླ་བའི་བཞིན་རས་ཅན། །པ་བྱ་མྱུདྷཱཀྵཀྵ་ཎ་མ་པི་ཤ་ར༔ནྡ་བ་ད་ནཾ། །ཨ་ཡཾནྡྷཱ་ནཾ་ཏ་ནཱི་མཛེས་མ་བདག་གིས་བྲིས་ཏེ་མིག་ཆུས་རྒྱས་པས་སྐད་ཅིག་ཀྱང་མ་མཐོང༌། །གཟུགས་ཅན་མ་དང་བྲལ་ལ་བི་ར་ཧ་ན་པེ་གླུ་བྲ་ཏ་བ་ཏསྤུ་རནྟ་བཤ་པཀི་མ་བི་མ་མ་པཱ་བཱ་དུལྟོས་མེད་བརྟུལ་ཞུགས་ལྡན་པ་བདག་ལ་ནི། །སྡིག་ལས་ཅི་ཡང་གདུང་བ་རབ་བསྐྱད་དམོད་པ་ཟུག་པ་པ་ན་ཏ། །ཕུ་ལཾ་བྷོ་ཛ་བ་ན་ཏཱི་ཥཱ་སྤྲོ་ཧ་དཱ་ཤཱ་སཀྵ་སུ་ཏཱ་ཡཾ་བ་པུསྟཀྐ་ལོ་མ་ཡིན་ནམ། །རྒྱས་པའི་ཆུ་སྐྱེས་ནགས་ཀྱི་འོད་འདྲ་དགའ་བའི་མིག་གིས་རྗེས་སུ་ཆགས་པའི་ལུས། །དེ་ཡི་པ་ག་ཏཱནྟ་རཱ་ཡ་ཛ་ནི་ཏཀོ་པས་མུཊཾ་ཛ་ཊཱ། །ཧཾ་ཧོ་བུནྡ་རི་དེ་ཧི་མེ་པྲ་ཏི་དུས་ན་ཉེ་བར་གྱུར་པའི་བར་ཆད་ཀྱིས་ཉེབསྐྲུན་ཁྲོ་བ་ཡང་དག་ཐོང་། །ཀྭ་ཡེ་མཛེས་མ་བདག་ལ་ཚིག་ལན་བ་ཙཏ་ཀིཾ་མོ་ན་མཱ་ལནྡྲ་སེ་ས་ཏྱཾ་ཏཱཆ་་ཡ་རཱ་ག་ཙཱི་བ་ར་མི་ཐཾ་ཙིནྟ་བྱིན་ཅིག་མི་སྨྲ་བར་ནི་ཅི་ལ་འདུག། བདེན་པར་བདག་ཀྱང་ཁྱོད་ལ་ཆགས་པའི་ཆོས་གོས་འདི་བསམ་བྲ་ཏཾ་མེ་བྲ་ཏཾ། །ཨི་ཏི་བྲུ་བ་ཎཾ་ཏཾ་དྲྀབྷྱཱ་དྷུ་རཱ་དཱ་ལི་ཁི་ཏཾ་མ་ནཾ། །：བརྟུལ་ཞུགས་བརྟུལ་ཞུགས་སོ། །དེ་སྐད་སྨྲ་ཞིང་བུད་མེད་དག། བྲིས་པ་རིང་ནས་མཐོང་གྱུར་ནས། །—།
