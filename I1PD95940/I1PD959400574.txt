རི་ཏཱ་བྷོ་ག་བྷུ་ཛསྟམྦྷ་བི་བྷུ་ཥ་ཎཾ། །ལེ་བྷེ་མ་ནོ་བྷ་བཱ་ར་བྷ་བྷ་བ་ནཾ་ཁོ་ལག་རྫོགས། །དཔུང་བའི་ཀ་བས་རྣམ་པར་བརྒྱན། །ཡིད་ལས་བྱུང་བ་རྩོམ་པའི་གནས། །ལང་ཚོ་ན་བ་ཡོ་བ་ནཾ། །ཀ་དཱ་ཙི་ད་ཐ་བྷུ་བརྟུས་པྲ་སེ་ན་ཛི་ཏསུ་ཏྭཾ། །ཏ་ནུ་ཙི་གསར་པ་ཐོབ་པར་གྱུར། །དེ་ནས་ནམ་ཞིག་དེ་ཡིས་ནི། །ས་ཡི་བདག་པོ་གསལ་རྒྱལ་གྱི། །བུ་མོ་ལུས་བ་ར་ཙི་སྣེ་ན་ས་ཧ་ཛེ་ན་བི་རཱ་ཛི་ཏྭཾ། །ཀ་རྩྭ་ཙཱ་བ་ར་ཀ་རྩྭ་ཏྭཾ་ལཱ་བ་རྱ་ནི་ཆོས་གོས་ཀྱི། །རྟགས་དང་ལྷན་སྐྱེས་རྣམ་མཛེས་མ། །ཆོས་གོས་བུ་མོ་ཞེས་བྱ་བ། །མཛེས་ཤིང་ལ་ལི་ཏཱ་ན་ནཱཾ། །ཨུ་ཏྭཱ་ན་དན་ནཱ་ཡཱ་ཏྭཾ་ད་དནཱ་ཡ་ཏ་ལོ་ཙ་ནཱ། །ཨ་ཡིད་འོང་བཞིན་རས་ཅན། །མིག་ཡངས་གཞོན་ནུ་མ་དག་ནི། །སྐྱེད་ཚལ་ལྟ་རུ་འོང་བ་མཐོང་། །ཉམས་ཀླུཀྵ་རཱུ་པ་མཱ་ལོཀྱ་ཏཱ་མ་ཡ་པརླུ་ཥིཏ་ཏྭ་ཏི། །བིསྨྲ་ཡསྨ་ར་ཡོསྟེ་རྩྭ་ས་པ་མེད་པའི་གཟུགས་ལྡན་ཞིང་། །མཛེས་པ་མི་དམན་དེ་མཐོང་ནས། །ད་ནི་འཕྲལ་ལ་ཡ་མཚན་དང་། །ཡ་ཡོ་ས་ཧ་སཱ་བ་ཐཾ། །སོ་ཙིནྟ་ཡཏ་ཨ་ཧོ་ཀཱནྟ་མི་དཾ་ན་ཏུདྷཱ་ཏཾ་འདོད་པའི་དབང་དུ་མཉམ་པར་གྱུར། །དེས་བསམས་ཀྱེ་མ་ལུས་འདི་ནི། །ཤིན་ཏུ་རྨད་བྱུང་མཛེས་བ་པུ། །ཡ་སྨིནྡྷཱ་ཏི་མུ་ཁ་བྱཱ་ཛཱ་ད་དོ་ཥ་བི་ཤ་དཤ་ཤཱི། །ལཱ་བ་ཏྱཱ་མ་པྲཏི་པ་སྟེ། །སྐྱོན་བྲལ་དྲི་མེད་རི་བོང་ཅན། །གང་དུ་བཞིན་གྱི་ཟོལ་གྱིས་མཛེས། །གཟུགས་ཅན་མཆུ་ནི་བནྡྷ་མ་མེ་ཥ་བི་བྷནྟི་ཏག་བནྣཱ་ཀ་བནྣཱ་ར་དྷ་རོ་མ་དུ་ར་སྭ་བྷཱ་བ། །ཀ་གཉེན་ཡིད་འོང་རང་བཞིན་ཅན། །འདི་ནི་དོ་ཟླ་མེད་པའི་མཛེས་སྡུག་རྣམ་པར་འཛིན་བྱེད་ཅིང༌། །དྲོ་ཧོ་བྱ་ཏས་ར་ས་བི་དྲུ་མ་པལྤ་བཱ་ནཾ་བིམྤ་པྲ་བྷཱ་པྲ་ས་བྷ་བི་རླན་ལྡན་བྱུ་རུའི་ཡལ་འདབ་དག་ལ་འཁུ་བར་བརྩོན་པ་དང་། །བིམ་པའི་འོད་ཀྱིས་རྣམ་འཕྲུལ་དག་ནི་བྷ་མ་བནྡྷ་ཀཱ་རཿ། །བནྡྷཾ་ན་ཀྵ་མ་ཏེ་མ་དཾ་ཤ་ཤ་བྷཾ་ཏཀྱིབྷཱ་ཏི་ཀཱནྟིསུ་འཕྲལ་ལ་འབྲས་མེད་བྱེད། །རི་བོང་འཛིན་པའི་རྒྱགས་པ་བཞིན་གྱི་མི་བཟོད་མཛེས་པས་བདུད་རྩི་དམན་དྷཱ་མུཏུཏྲ་ལ་ཀཱ་ན་ན་སྱ་ཀུ་རུ་ཏེ་དྲྀཋཔྲ་བྷཱ་བྷནྡྷ་ནཾ། །མ་རྩེ་བྱེད་ཅིང་། །ཨུཏྤལ་རབ་ཏུ་རྒྱས་པའི་ནགས་ཚལ་འོད་ནི་མིག་གིས་རབ་ཏུ་ཉམས་པར་བྱེད། །ཡིད་སྲུབས་—།
