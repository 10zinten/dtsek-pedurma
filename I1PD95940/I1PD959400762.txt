རི་ཤཱལྟ་ཡ་ཤོ་ད་ད་ཡ་ཡཱ་སཱ་ཡཾ། །པུ་ར་ན་yཔ་ར་པཱ་རཾ་པཱཙླི་ས་མུ་ནི། །གྲགས་སྦྱིན་དག་ལ་བརྩེ་བ་ཡིས། །གྲོང་ཁྱེར་ཀླུང་འགྲམ་ཕ་རོལ་ཏུ། །དེ་འཐོབ་སླད་དུ་ཡང་པཱ་ཡ་ཡོ། །ཡ་ཤོ་དེ་པི་དི་ནཱ་ཡ་ས་ཏུ་ལ་ཡཱ་ཀ་ལ་ཡཎྜཱུ་ཧུ། །སཾ་སཱ་དག་གཤེགས། །གྲགས་སྦྱིན་གྱིས་ཀྱང་ཉིན་པར་གྱིས། །ཉོན་མོངས་མཚུངས་པ་ཡང་ཡང་དུ། །འཁོར་བ་རཱ་སཱ་ར་ཏ་མེ་བ་ཤཡཱ་བྷ་བ་ན་མཱ་བེ་ཤཏ། །བི་ཆུ་བཱི་ཎཱ་མྲ་དཾ་གཱ་དི་བི་ནོ་སྙིང་པོ་མེད་པ་ཉིད། །སེམས་ཤིང་ཉལ་བའི་ཁང་པར་ཞུགས། །གླིང་བུ་རྒྱུད་མང་རྫ་རྔ་སོགས། །：རྩེ་དིརྨ་ད་ནིཧྥ་ར། །ཏ་ཏཱ་བ་བྱ༴་ལ་ལཱཐ་མ་ནི་དཱ་ནི་མཱི་ལི་དགའ་ལ་ཞིང་གཉིད་ལོག་པ། །མིག་ཟུམ་རྒྱགས་པས་འཁྲུགས་པ་ཡི། །བུད་མེད་རྣམས་དེར་དེ་ཡིས་ཏཱ༑། བཱི་ཎཱ་བི་སྟ་བ་ད་ནཱ་མྲྀ་དཾ་གཱམྤི་ཏ་དོརྨ་ཏཱ། །ཀྵ་ཎི་ཀ་མཐོང་། །ཁ་ནི་རྒྱུད་མང་ལ་བཀོད་ཅིང་། །ལག་པའི་འཁྲི་ཤིང་རྫ་རྔར་ཅིགཏང་། །བདེ་བ་སྐད་ཅིག་ཏྭཱ་སུ་ཁ་ཕྱེ་བ་དུཁ་དྷཱ་ན་མུ་པ་ག་ཏ། །བི་ལོ་ཀྱ་ཐལྟ་བ་ད་ནཱབྷྱཱ་མྲྀ་ཏཱ་པ་ཉིད་ཀྱིས། །སྡུག་བསྔལ་བསམ་གཏན་ལ་ཞུགས་བཞིན། །གོས་རལ་ཤི་བ་བཞིན་དུ་ནི། །གཡོ་མེད་ཨི་བ་ནིཧྥ་ལཿ། །སཾ་ཛཱ་ཏཱ་དྷི་ཀ་བི་རཱ་གྱས་ཙི་རཾ་ས་མ་ཙིནྟ་ཡཏ། །དེ་དག་མཐོང་གྱུར་ནས། །ལྷག་པར་ཆགས་བྲལ་ཡང་དག་སྐྱེས། །ཡུན་རིང་དེ་ཡིས་ཡང་དང་བསམས། །ཨ་ཧོ་བ་ཏཱམིཙླི་ཥ་ཡེ་པརླན་བར་པེ་བ་རཾ། །ཨེ་ཐཾ་བི་དྷེ་བ་་ཀྱེ་མ་ཡུལ་ནི་འདི་དག་རྣམས། །མཇུག་ཏུ་མཆོག་ཏུ་རོ་དང་བྲལ། །བུད་མེད་ཅེས་པ་འདི་ལྟ་ལ། །ནཱསྣེ་མུདྡྷཱ་ནཾ་བྷི་ཤ་མཱ་ད་ར། །བི་རྒྱུ་དཱི་ལ་ཥི་ཏཱ་ནཱ་སུ་མ་སཏུ་ཁ་གྷ་ནོ་ད་རྨོངས་པ་རྣམས་ནི་མཆོག་ཏུ་གུས། །མཆོག་མིན་བདེ་བའི་སྤྲིན་རྒྱས་ལ། །གློག་གི་རྩེ་དགའ་འདི་དག་ཡེ༑། སུཔཱ་ནཾ་ཙ་མྲྀ་ཏཱ་ནཾཉྩ་ཀུཌི་ཛཾ་ཀཱུ་ཙ་བཾ་བྷ་མ། །ཤེཏེ་ཀཱ་ཙི་ད་དྷོ་རྣམས། །གཉིད་ལོག་ཤི་བ་དག་ལ་ཡང་། །རྣམ་འཕྲུལ་གང་ན་འཛུམ་གང་ན། །འགའ་ཞིག་འོག་ཏུ་ཁ་མུ་ཁཱི་པཾ་བ་ལི་ཏཱ་པྲཊྛེ་ཙྩ་ཏཱ་སྱཔ་རཱ་ཡཱ་ཏཱ་ནཱུ་པི་བི་ཀཱ་ཤི་ཏཱ་སྱ་བི་བ་རྭ་སབྷཾ་ག་བལྟས་ཉལ་ཞིང་དེ་ཡི་རྒྱབ་ནི་གཞན་དུ་བཟློག་སྟེ་ལྷུང་། །གཞན་ཡང་གན་ཀྱལ་ཡན་ལག་ཀུན་བརྐྱངས་།
