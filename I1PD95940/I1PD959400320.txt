ཏ་སྱ་མུ་ཁེ་ན་ནཱ་ནཾ། །སཟོ་ཧི་ནཱ་མ་སྭཱ་ཧི་ཏཱ་བྷཱི་ལཱ་ཏི་དྷཏྟེ་པ་རཱཟོ་ཀི་བརྗེད་པ་མ་ཡིན་ནམ། །གང་ཕྱིར་ཐམས་ཅད་ངེས་པར་རང་ཕན་འདོད། །གཞན་གྱི་དོན་ལ་བསིལ་ལ་ཤཱི་ཏ་ལ་ཏྭཾ། །བི་ལོ་ཀ་ནེ་ནི་ཏྱ་ནི་ཥིཉྩ་ཏུཎཾ་པཱི་ཡུ་ཥ་པཱུ་རེ་ཎ་མ་བ་ཉིད་འཛིན་ལོ། །ཆགས་མེད་འདི་ནི་ཡུད་ཙམ་ངལ་གསོ་མཛོད། །རྟག་ཏུ་བདུད་རྩིས་ཁེངས་པའི་ཨཾ་ག་སདྡྷཾ། །ནིསཾ་ག་ཏི་ཥ་ཀྵ་མ་ཏཱཾ་མཱུ་ཧཱུཏྟཾ་ད་ཡཱ་བི་དྷེ་ཥཀུ་རུ་བནྡྷ་རྣམ་གཟིགས་ཀྱིས། །མྱུར་དུ་བདག་གི་ལུས་ལ་ཆགས་གྱུར་པ། །བརྩེ་བའི་དབང་གིས་གཉེན་གྱི་དོན་ཀཱཎཾ། །ཨེ་ཏེ་ཏ་དཱ་ཀརྞ་ཀ་ཐོ་ས་ཀུམཱཏ་ཀཱ་ཎཾ་བི་ལཝཾ་མ་མ་དན་ནེ་པི། །ཏཙེ་འདི་མཛོད། །ཅེས་པ་འདི་ཐོས་བདག་དང་ཕྲད་པ་ལ། །སྐད་ཅིག་ཀྱང་དེ་ཕྱི་བཤོལ་ཇི་ལྟར་མཛད། །དེ་ཏ་སཔལླ་བ་པེ་ཤ་ལ་སྱ་ན་ཧི་སྭ་བྷ་བ༔པྲ་ཎ་ཡཱ་བི་བྷཾ་ག། །མ་ནོ་ར་ཡི་ཐུགས་ནི་ཡལ་འདབ་མཉེན་པ་ལ། །རང་བཞིན་མཛའ་གཅུགས་རྣམ་རྒྱུན་མེད་དམ་ཅི། །ཡིད་ལ་རེ་ཐེ་ནེ་ཏི་པུ་རཿཔྲ་ཡཱ་ཏེ་ཏདྡན་ནཱ་ཡེ་བ་དྷ་རཱ་དྷི་ནཱ་ཐེ། །པྲ་བྲ་ཛུ་ཡ་བྱཎྜི་ཏ་ཏཏྠ་བ་འདི་ཡིས་ས་ཡི་བདག། དེ་བལྟའི་སླད་བཞིན་མདུན་དུ་སོང་བའི་ཚེ། །དེ་ཡི་དྲིན་གྱིས་རབ་བྱུང་གསལ་ས་དས་མ་ཡ་ཡོ་ཧརྟ་ར་ས་དུ་དཱ་ཡཱི། །དྲྀབྷྱཱ་ཏ་མཱ་ནནྡ་བི་པུཎ་མཱ་ས་བྱས་པའི། །དགའ་བའི་རོ་ལྡན་འཆར་ཀ་ཡང་དག་འོངས། །རབ་ཏུ་བྱུང་བ་དེ་དང་མཚུངས་པའི་མཐུ། །པྲ་བྲ་ཛ་ཡཱ་ཏཏྠ་དྲྀ་ཤཱ་ནུ་བྷཱ་ཐཾ། །ཨུཥཎྜི་ཏཀུཙྪི་ཏ་དྷིརྨ་བྲྀཏྟིཀུན་དགའ་རྣམ་པར་ཁྱབ་པ་དེ་མཐོང་ནས། །བརྟན་པའི་སྤྱོད་ཚུལ་ཉམས་ཤིང་འདོད་ལྡན་པས། །སཾ་མོ་ཧ་མུ་ཙྪནཱ་པ་ཏིཔྲ་པ་དེ། །ས་ལབྦྷ་སཾ་ཛྙཱཤི་ཤི་རིཔ་ཡེ་བྷིས་པྲཙྪ་མི་བདག་ཀུན་ཏུ་རྨོངས་པས་ཀྱིབརྒྱལ་བར་གྱུར། །བསིལ་བའི་ཆུས་བྲན་འདུ་ཤེས་རྙེད་པ་དེས། །དེ་ལ་ཅི་ཏཾ་ཀིནྣཱ་ས་མེ་་ཏཱི་ཏི། །སོ་སྐྱ་བྲ་བཱིདེ་བ་དི་ནིགྷ་བནྟཾ་ས་སཾ་བྷཾ་སླད་བྱོན་པ་མིན་ཞེས་དྲིས། །ལྷ་གཅིག་ཁྱོད་ཅན་ཉིན་ཞག་ཚོགས་ཀྱིས་དེསི། །གུས་བཅས་འབྱོན་ཏིསཱ་ད་ར་མེ་ཥྱ་ཏཱི་ཏི། །ཏ་ཏཔྲ་ཡཱ་ཏེ་ཥཱ་དི་ནེ་ཥཱ་བྱོ་སྣ་ཤ་ནིབྷི་གླུ་ག་པར་འགྱུར་ཞེས་དེས་ཀྱང་སྨྲས། །དེ་ནས་ཉིན་ཞག་རྣམས་ནི་རབ་འདས་ཚེ། །གཞོན་ནུ་དོན་ཀུན་གྲུབ་—།
