སཱཙྪ་ཀ་ན་ཀ་བྷཱ་ཛ་ནེ། །བ་རོ་པ་ཙཱ་ར་མ་དཱ་ཡ་མ་དྷུ་རཾ་སཱ་མི་བ་ཤ་ཡཾ། །རང་གི་གསེར་གྱི་སྣོད་དུ་ནི། །མཆོག་གི་ཉེར་སྤྱད་དངར་པོ་དག། རང་གི་བསམ་པ་བཞིན་དུ་བླངས་། །པཉྩེདྨ་ཛ་ན་མཱ་ལོ་ཀྱ་བྷཀྵ་བྷ་ག་བ་ཏཔ་ཏི། །ན་རེགྱུ་ཏ་ཀ་ཊཱཀྵེ་ཎ་བཅོམ་ལྡན་འདས་ཀྱི་རྗེས་བཞིན་དུ། །གུས་པས་བདག་པོ་བགྲོད་མཐོང་ནས། །འབྲལ་བ་མི་བཟོད་མཛེས་པུནྡ་རི་བི་ར་ཧཱ་ས་ཧ། །གུ་རཱུ་ཎཱ་མ་གྲེ་སཱ་ས་ར་ལ་ཏ་ར་ལཱ་ལོ་ཙ་མ་ཡིས། །ཟུར་མིག་གིས་ནི་ངེས་པར་བལྟས། །བླ་མ་རྣམས་མདུན་མིག་གི་འགྲོས་ནི་དྲང་ཞིང་རབ་ཏུ་གཡོ་ན་ག་ཏི་བི་ཧཱ་ཡེ་བ་ཏྲ་ཧཱུྃ་ཙི་ཏ་མུ་ཀུ་ལི་ཏཱ་བྷཱི་པྲི་ཡ་ཏ་མཾ། །ནི་རཱིག་བ་དག། ཡངས་པ་ཉིད་ཀྱིས་ཟུར་མིག་རབ་མཛེས་ཅུང་ཟད་བྱེ་བས་མཛེས་པའི་མཆོག། བལྟ་བྱ་བལྟས་ལ་ཧཾ་ཡ་ཏ་གྷ་ཎ་མ་བ་ན་ཏཱ་བྷཱུཏྟ་ད་དྷི་ཀཾ་ན་གན་དྷཾ་ནཱ་ཐེ་ཏྱ་བ་ད་དི་བ་མོ་ནཱ་ནས་གང་ཞིག་སྐད་ཅིག་ཏུ་གྱུར་དེ་ཕྱིར་ལྷག་པར་ནི། །འགྲོ་བར་མི་བྱ་མགོན་པོ་ཞེས་ནི་ཐུབ་པ་ཚིག་མེད་ད་བ་ཙ་ནཾ། །ནནྡཔྲ་ཎ་ཡཾ་ནི་དྲྀཥྛ་སོཙྪཱཙཾ་ཙ་ལི་ཏཱ་ན་ནཱ། །གསུངས་པ་བཞིན། །མཛའ་མོ་ཤུགས་རིངས་ལྡན་པ་ལ། །དགའ་བོ་བཞིན་ནི་ཕྱིར་ཕྱོགས་ནས། །ཨ་ཡཱ་མཱ་ལ་ཏ་ཨེ་བཱ་ཧ་མི་བི་ཙཱ་རཱ་དི་ཏྱ་བྷཱ་ཥཏ། །ཏ་ཏསཱ་མཱ་ཐ་མཾ་པྲཏྟཾ་བདག་ནི་རིང་པོར་མི་ཐོགས་པར། །འོང་བ་འདི་ཉིད་ཅེས་རབ་སྨྲས། །དེ་ནས་བཅོམ་ལྡན་རང་ཉིད་ཀྱི། །བྷ་བནྟཾ་ཀྲྀ་ཏཱལྔ་ལི། །བྲ་ཛཱ་མི་སཱ་ཤྲ་ཧནྟཱ་བ་ད་ཏྭཱ་ཧ་བི་ར་ཧཱ་ས་ཧ། །གནས་སུ་བྱོན་ལ་ཐལ་སྦྱར་ནས། །རེ་ཞིག་རང་ཁྱིམ་མཆིའོ་ཞེས། །འབྲལ་བ་བཟོད་པ་མེད་པས་སྨྲས། །ཏ་ཏས་མཱ་ས་ནཱ་སཱི་ནཔྲ་ཎ་ཏཾ་བྷ་ག་བཱན་པུ་ར། །ཨུ་བཱ་ཙ་ས་རྨི་ཏ་ཀེ་ཡཾ་ག་དེ་ནས་ཁྲིར་བཞུགས་བཅོམ་ལྡན་གྱིི། །མདུན་དུ་རབ་ཏུ་བཏུད་དེ་ལ། །འཛུམ་དང་བཅས་གསུངས་ད་མ་ནེ་སཾ་པྲ་ཏི་ཏུ་ཡཱ། །བི་ཥ་ཡཱ་སྭཱ་ད་སོ་ཧཱརྡ་སཾ་མོ་ཧཱ་དྷི་ལྟ་ཉིད། །མྱུར་བར་འགྲོ་བ་འདི་ཅི་ཞིག། ཡུལ་གྱི་རོ་ལ་གཅགས་པ་ཡི། །ཀུན་རྨོངས་ཀྱིས་གཟིར་ཙེ་ཏ་སཾ། །ཨ་ཧོ་ཧྲ་ཧོ་སུ་ཁ་སྦེ་བ་ནིརྗེ་ད་བི་མུ་ཁཱ་མ་ཏི། །གུ་ཎཱ་བྷ་སེམས་ལྡན་རྣམས། །ཀྱེ་མ་ཁྱིམ་གྱི་བདེ་ཉིད་ལ། །ཡིད་འབྱུང་ལས་ནི་ཕྱིར་ཕྱོགས་བློ། །ཚེ་ཡི་རྒྱན་ནི་།
