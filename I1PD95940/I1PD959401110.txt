རརབསྟན་འགྱུར། །འཁྲི་ཤིང༌། །ཁིལ། ཎཱ་ཏི་བ་ཏ་ནི་བི་ལཀྵ་ལཱི་ནསྦི་ཏིརྣོ་ཛཱ་ནེ་ཀ་ཏ་མེ་ན་སཱ་མ་མ་པ་ཐཱ་ཙོ་རཱི་ཏུ་བསྣུན་ཚེ་སྐྱེངས་པ་ཡི་ནི་ཆགས་ཤིང་གནས། །ལུས་ཕྲ་ཆོམ་རྐུན་མ་དེ་བདག་ཡིད་ལ་ཞུགས་ལམ་གང་པྲ་བིཏྟཱ་མ་ན། །ཙི་རཾ་བི་ཙི་ཏེ་ཏི་ཤི་ལཱ་ཏ་ལ】ས་མརྨ་ཐཱ་ཛྙཱ་བི་ཥ་ཡཾ་པྲ་ནས་ནི་ཡོངས་མི་ཤེས། །ཞེས་པ་ཡུན་རིང་བསམས་ནས་རྡོ་ལ་འཁོད། །ཡིད་སྲུབས་བཀའ་ཡི་ཡུལ་དུ་ཡཱ་ཏ། །སཾ་ཀལྤ་བནྡྷ་པུ་ར་ཏོ་བྷི་ལི་ཁྱ་མུ་ཧུ་ལཾ་གཱ་ཀིཾ་སྟི་མི་ཏེ་ག་ཎོ་བྷཱུཏ། །གྱུར་པ་དེས། །ཀུན་རྟོག་པིར་གྱིས་མདུན་དུ་རི་དྭགས་མིག། ཡང་ཡང་མངོན་བྲིས་མིག་ནི་གཡོ་མེད་གྱུར། །ཨ་ཐཱ་ཡ་ཡོ་པུཥྛ་རཱ་ག་རེ་ཎུ་ནཱ་པུཎྞེ་བ་ནེ་པཱ་ད་པ་རོ་ཛ་མུ་དཱ། །ཙཀྲ་དྷཱ་དེ་ནས་མེ་ཏོག་རྡུལ་དང་རྡུལ་གྱིས་གང་། །ནགས་ཚལ་དབེན་པར་རྐང་པའི་མཚོ་སྐྱེས་རྒྱ། །འཁོར་ལོ་རྒྱལ་ཛཱ་ཀ་བི་ཛ་ནེ་ནུ་སྲི་ཏྱ་ཏ་སཱུནྟི་ཀཾ་ནརྨ་སུ་ཀྲྀཥྛ་བནྡྷ། །ས་ཏཾ་བི་མཚན་གྱིས་མཚན་རྗེས་འབྲངས་ནས། །དེ་ཡི་དྲུང་དུ་རྩེད་གྲོགས་གཉེན་བཟང་འོངས། །མངོན་འདོད་ལོ་ཀྵ་བྷི་ན་བཱ་བྷི་ལཱ་ཥ་བི་ཤེ་ཥ་ཙིནྟཱ་པྲ་ཏི་བདྷ་ཙིཏྟཾ། །བི་ཀཱི་ཏ་གསར་པའི་ཁྱད་པར་བསམས་པ་ཡིས། །སེམས་ནི་རབ་ཏུ་བཀག་དེ་དེས་མཐོང་ནས། །དྲན་པའི་བཀའ་དྷི་སྨ་ར་ཤཱ་ས་ནེ་ན་པ་པྲཙྪཱ་སཱརྨ་བི་ཀཱ་ར་ཧེ་ཏུཾ། །ཀོ་ཡཾ་ས་ཡིས་བརྟན་པ་བཙོངས་པ་ལ། །ངོ་མཚར་བཅས་པའི་རྣམ་འགྱུར་རྒྱུ་དག་དྲིས། །།གྲོགས་པོ་བརྟན་པའི་ཁྱི་དྷིརྨ་ནི་དྷེསྟ་བཱ་པི་ནི་ཏནྟི་སནྟཱ་པ་པ་དོ་པ་དེཋ། །ཙིནྟ་ཐ་ཡནྣི་གཏེར་གྱུར་ཁྱོད་ལ་ཡང་། །ཤིན་ཏུ་གདུང་བའི་གནས་ནི་ཉེར་བསྟན་ཅིང་། །བསམ་པ་ལ་བརྟེན་མིག་ནི་ལ་ལོ་ཙ་ན་སྱ་བི་བྷཱ་བཱ་ཏེ་ཀོ་པྱ་དྷཾ་ཏི་པྲ་ཀཱ་རཿ། །ཨི་ཏི་པྲ་ཡནྡྲ་པྲ་ཎ་ཡེ་ན་པྲཊོ་བི་གཡོ་མེད་པ། །བརྟན་མེད་རྣམ་པ་ཅི་ཞིག་ཉམས་སུ་མྱོང་། །ཞེས་པ་ཡིད་གཅུགས་གནས་ནི་གྲོགས་པོ་དག། ཤྲཱི་བྷ་དྷཱརྨཱ་སུ་ཧྲི་དཱ་ཀུ་མཱ་ར། །ཏ་མ་བྲ་པཱིནྡི་ཤཱ་སི་ཏཱ་ནུ་མེ་ཡ་ས་པཀྵ་བཱ་རབ་ཏུ་འབད་ཅིང་གུས་པས་དྲིས་པ་ནི། །ཤུགས་རིངས་མཐུན་ཕྱོགས་རླུང་གིས་རྗེས་དཔག་བྱ། །དྲན་ཏ་སྨ་ར་བཱ་ཎ་ནཱ་ཏ། །ས་ཁེ་ས་ཁེ་ད་སྱ་པུ་རཱ་ཎ་སརྒ་བརྒེ་བི་དྷརཱུ་ཏ་ན་རཱུ་པའི་མདའ་ལྷུང་གཞོན་ནུས་དེ་ལ་སྨྲས། །གྲོགས་པོ་སྔོན་སྤྲོས་སྡེ་ལ་སྐྱོ་ལྡན་པའི། །བྱེད་པོ་ཡིས་ནི་གཟུགས་།
