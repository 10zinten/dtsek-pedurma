ཨ་བཱ་པ་དེ་བ་ནི་ལ་ཡཾ་མ་ཏི་ན་ར་ཀཾ་ཡ་ཡོ། །སོ་ཧཾ་སུ་མ་ཏི་རེ་དགའ་ལྡན་ཞེས་པ་ཐོབ་གྱུར་ཏེ། །བློ་གྲོས་དག་ནི་དམྱལ་བར་སོང་། །ད་ལྟ་བློ་བཟང་དེ་ཉིད་ང་། །བློ་བཱ་དྱ་མ་ཏིསུ་མ་ཏི་བི་སླ་བཱཏ། །ཨ་ཡཾ་དྷརྨ་རུ་ཙིལླིཋསུནྡ་རཱི་སཱ་ཡ་ཤོ་དྷ་རཱ། །གྲོས་བློ་གྲོས་རྣམ་བསླད་པས། །ཉམས་པར་གྱུར་པ་ཆོས་སྲེད་འདི། །མཛེས་མ་དེ་ནི་གྲགས་འཛིན་མའོ། །ཏྲ་ཏཱི་ཡེ་བྷ་ག་བཱཎལྕེ་ཀྲ་ཀུཙྪནསྟ་ཐཱ་ག་ཏ། །ཨ་ནུཏྟ་ར་ཙཱ་གསུམ་པའི་བསྐལ་པར་བཅོམ་ལྡན་འདས། །དེ་བཞིན་གཤེགས་པ་འཁོར་བ་འཇིག། བླ་ན་མེད་པའི་ཡེ་ཤེས་ནི་ནི་དྷིཛཾ་བཱ་དཱི་པ་པུ་རཱ་བྷ་བཱཏ། །བ་བྷཱུ་བ་ས་མ་ཡེ་ཏསྨིནྣཱཛྫ་ཡ་ཧཱུྃ་མ་ཧཱ་གཏེར། །འཛམ་བུའི་གླིང་དུ་སྔོན་བྱུང་གྱུར། །དེ་ཡི་དུས་ན་འཕགས་རྒྱལ་དུ། །ཚོང་པ་ཙནྡན་བྱིན་དྷ་ན། །བ་ནི་ཀུནྡ་ན་དཏྟཱ་ཁྱཔྲ་ཀྵ་ཏ་ཀྲ་ཡ་བི་ཀྲ་ཡ། །ཏ་སྱ་ཀཱ་ཞེས་པ། །ནོར་ཆེན་ཉོ་ཚོང་དག་གིས་ནི། །རབ་ཏུ་གྲགས་པ་བྱུང་བར་གྱུར། །དེ་ཡི་ཆུང་མ་འདོད་མ་བ་ལཱ་ཀྵ་ཡཾ་ཛཱ་ཡཾ་ཡཾ་མ་ད་ན་དྷུ་ཏི། ་། ཨ་ཤྭ་དཏྟཱ་བྷི་དྷཱ་ནོ་བྷཱུཏ་པུ་ཏྲཀཱ་པའི་སྟོབས། །ཞེས་ལ་མྱོས་བྱེད་ལྟར་མཛེས་བུ། །རྟས་བྱིན་ཞེས་ནི་བྱ་བ་དག། ལུས་བཞིན་དུ་ནི་དགའ་ཡ་ཨི་བ་དྲི་ཡ། །ཨཌྷནི་ས་ཛ་གཱ་མཱདྷཱི་པཏྟཱི་ནྱལྟ་གྲོ་ཧ་སྦི་ཏི། །དྷ་ནེ་བ་བྱུང་། །ཁྱིམ་ལུགས་ཆུང་མ་ལ་བཀོད་ནས། །ནོར་དོན་གཉེར་དེ་རྒྱ་མཚོར་སོང་། །ལན་ཚྭ་ཡིས་ནཱརྦ་ནི་ནཾ་ཀྲྀ？་ལ་བ་དེ་ནི་བ་བདྷ་ཏེ། །ཏཏྠཏི་པྲ་ཏི་ཏེ་བ་ཁྱོ་ཡོ་བ་བཞིན་ནོར་ཅན་རྣམས། །ནོར་དག་གིས་ནི་སྲེད་པ་འཕེལ། །དེ་ཡི་ཆུང་མ་ཁྱོ་སོང་ཚེ། །ལང་ཚོས་མྱོས་ནོནཱ་ད་མོ་ཧི་ཏཱ། །གྲོ་ཧ་ཙིནྟཱ་པ་རི་ཎྱ་ཛ་ཀཱ་མ་དྷཱ་ན་པ་རཱ་བྷ་བཏ། །།ཀུ་ཤིང་རྨོངས་པ་ཡིས། །ཁྱིམ་གྱི་བསམ་པ་ཡོངས་བཏང་ནས། །འདོད་པའི་བསམ་གཏན་ལ་ཆགས་གྱུར། །མཱ་རཱ་མ་ཤཱ་དཏྟཾ་ས་ནི་དྷཱ་ཡ་དྷ་ན་ཀརྨ་ཎི། །ས་དཱ་པྲ་ས་ད་ཧམྱ་དེ་ཡིས་ནོར་གྱི་ལས་ལ་ནི། །གཞོན་ནུ་རྟས་བྱིན་རྣམ་བཀོད་ནས། །རྟག་ཏུ་ཁང་བཟང་བ་གམ་དུ། །སྠཱ་རཱ་ཛ་མཱནཾ་བྱ་ལོ་ཀ་ཡཏ། །སྭཱ་རཱ་ར་ཀཱ་ཤ་ར་ཧི་ཏཱ་བ་ཧུ་དྲྀ་ཏྱཱ་ཛནེ་གནས་ཏེ་རྒྱལ་པོའི་ལམ་ལ་བལྟས། །གཡོག་འཁོར་མང་པོའི་ཁྱིམ་དུ་ནི། །རང་དབང་གནས་སྐབས་བྲལ་—།
