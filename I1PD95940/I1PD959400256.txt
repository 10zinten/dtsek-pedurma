ཀུ་ཤ་ལི་ཁི་ཎ། །ཝརྨ་སཾ་ཤོ་དྷ་ནཾ་ཙཀྲུ་རུ་པ་ཙཱ་ར་པ་དོརྨ་ཏཱ། །དེ་ཤཾ་བྷུ་ལ་རབ་བརྩོན་པ། །དགེ་བ་འཚོལ་བ་དེ་དག་གིས། །ལམ་ནི་ལེགས་པར་སྦྱང་བ་བྱས། །རྩྭ་དང་ཚེར་ཥ་ཡ་ཏྭཾ་ཏེ་ཏཱཾ་སཾ་སིཏྟཾ་ཙནྡ་ནོ་ད་ཀི། །།ཀྲྀ་ཎ་ཀཎྜ་ཀ་པཱ་ཥཱ་ཎ་སརྐ་མ་རྡོ་བ་དང༌། །གསེག་མ་རྡུལ་དག་བསལ་བྱས་ནས། །ཙནྡན་ཆུ་ཡིས་ལེགས་བྲན་ཏེ། །ཡུལ་ནི་རཱ་རེ་ཎུ་བརྫི་ཏཾ། །མ་དྲི་ས་མཱ་ཡ་ཡོ་བྷུ་མི་ནི་མགྷཱ་མ་ཧ་ཏཱི་ཤི་ལཱ། །མཛེས་པ་དེ་དག་གིས། །དབུས་སུ་འབིགས་བྱེད་རྒུད་གྱུར་པའི། །ཆུང་མ་གཏིང་དུ་ཞུམ་པ་བཞིན། །ཨ་བ་སནྣཱ་བི་སནྣ་སྱ་བ་དྷུ་བི་ཧུཾ་གི་རེ་རི་བ། །ཏཱ་མུཏྟཱ་ཏ་ཡ་ཏྭཾ་ཏེ་ཊཱ་གཏིང་དུ་བྱིང་བའི་རྡོ་ཆེན་ནི། །ས་ལས་ཡང་དག་བྱུང་བར་གྱུར། །འབྱོར་དང་དཔུང་པ་ཐག་པ་ཡིས། །ཀུདྡ་ལ་བྷཱུ་ཛ་རཛྫ་བྷི། །མཱ་སོ་ཛ་གཱ་མ་ན་ཏཱ་སྱཱས་ཧ་པྲྀ་ཤེ་སྱ་བྷཱུ་ཏ་ཁཱ་ཏི། །ཨ་ཐ་དེ་ནི་སོལ་ལ་དེ་དག་རྣམས། །ཟླ་བ་སོང་ཡང་དེ་ཡིས་ནི། །སྟོང་ཕྲག་ཆ་ཡང་ཟད་མ་གྱུར། །དེ་ནས་སཾ་སཱ་ར་སཾ་ཏཱ་པ་པྲ་ཤ་མཱ་མྲྀ་ཏ་དཱི་དྷི་ཏི། །ཨཱ་ཡ་ཡོ་བྷ་ག་བཱན་སརྦ་འཁོར་བའི་ཡོངས་གདུངས་དག། རབ་ཏུ་ཞི་བྱེད་བདུད་རྩིའི་ཟེར། །ཀུན་གྱི་ཡིད་ནི་དགའ་བའི་གཉེན། །མཱ་ན་སོ༈་ས་བནྣཱ་བ། །གྷ་ནཱནྡྷ་ཀཱ་ར་བི་ར་ཏི་བྱཀྟ༔ས་ཕ་ལ་བཅོམ་ལྡན་འདས་ནི་བྱོན་པར་གྱུར། །སྟུག་པོའི་མུན་པ་དག་དང་བྲལ། །གསལ་ཞིང་མཐོང་བ་འབྲས་དན་ན། །པྲ་སཱ་ད་སཾ་བི་བྷཏྟཱ་ཤཔྲ་ཀཱ་ཤ་ཨི་བ་ཤཱ་ར་ད། །ས་ཏཱན་དང་བཅས། །རབ་དང་ཕྱོགས་ནི་རྣམ་པར་ཕྱེ། །རབ་ཏུ་གསལ་བ་སྟོན་ཉིད་ཀྱི་བཞིན། །འབྲས་མེད་དྲྀབྷྱཱ་པ་རི་ཤནྟཱན་བི་ཕ་ལ་ཀླ་ཤ་པཱི་ཌི་ཏཱན། །ཐུ་ཏྭཱ་ཙ་ཏདྱ་བ་སི་ཉོན་མོངས་ཀྱིས་གདུངས་པ། ་། ཡོངས་ངལ་དེ་དག་དེས་གཟིགས་ཤིང་། །དེ་དག་གདུངས་གྱུར་ཐོས་ནས་ཏཾ་ཏརྦ་ཙེ་པྲ་ཎ་ཏཱ་ན་ནཱན། ་། ཨ་ཧོ་གླེ་ཤ་ཕ་ལཱ་རཾ་བྷས་ཡཱ་ས་ཀྱང་། །ཕྱག་འཚལ་བཞིན་ལྡན་དེར་བཀའ་སྩལ། །ངལ་དུབ་འབད་རྩོལ་ལྡན་པ་ཁྱེད། །འཁོར་བ་ཡིས་ཏེ་བྱ་བ་སཱ་ཡི་ནཾ། །སཾ་སཱ་ར་ཀརྨ་ཎཱི་ཙསྨིན་བྱཱ་པཱ་རེ་བས་མུདྨ་མ། །པྲ་ར་བྷེ་ནི་ལས་ལ་བཞིན། །བྱ་བ་འདི་ལ་རབ་བརྩོན་པ། །ཀྱེ་མ་ཉོན་མོངས་འབྲས་བུ་རྩོམ། །བརྩམས་ན་མི་།
