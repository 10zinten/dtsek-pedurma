རྒྱལ་བ་རུ་ཎ་རཱ་གཱནྟཾ་ག་རཱ་གཱ་བ་བྷོ་པཱ་ཏི་མི་ར་ཤ་བ་ལ་སནྣཱ་བོལྤ་སདྡྷཱ་པ་དྷུ་བའི་གྲགས་པ་ཟླ་བའི་གཟུགས་ནི་དང་པོ་བཞིན། །ལན་བུ་ཐིག་ལེའི་རི་མོ་ས་འཇོ་སྒེག་ལྡན་པར་རབ་སྤྲས་མི༈། མ་ན་སི་ཛ་ཛ་ཡ་ཀིཏྟིཾ་མུནྟི་མཱ་ཏྱཱ་མི་བེནྡོ་ར་ལ་ཀ་ཏིལ་ཀ་ལེ་ཁཾ་ལཱི་ཤིང་། །རབ་འཕྲོ་བདུག་པའི་དུ་བས་རབ་རིབ་ཐུན་མཚམས་བཞིན་བཀྲ་བ། །ཆགས་པ་དར་བྱས་ཡན་ལ་ཡཱ་སཱུ་ཏྲ་ཡནྟི། །ཏ་ཏཔྲ་བི་བྱ་ཏཱ་རི་ཏཱ་དཱ་སཱི་མ་ཀཱ་རི་ཀཱ་བྷི་དྷཱ། །ལག་དང་བཅས་ཆགས་པ་དེ་རབ་མཛེས། །དེ་ནས་གཡོག་མོ་ཆུ་སྲིན་མ། །ཞེས་པ་མྱུར་བར་རབ་ཞུགས་ཏེ། །ཏ་མུ་ཙེ་་ཎ་སཾ་གཱཎཱི་བ་ཧི་རཱསྟེ་ཡུ་བཱ་ན་བ། །ཀཱནཱ་པ་ཎཱ་ནཾ་ཀྲྀ་ཎ་དེར་སྨྲས་སྐད་ཅིག་འགྲོགས་འདོད་པའི། །གཞོན་ནུ་གསར་པ་ཕྱི་ན་སྡོད། །ཀཱནཱ་པ་ཎི་བརྒྱ་ཕྲག་ལྔ། །བདྲི་ཏཱིརྨ་ཤ་ཏ་པཉྩ་ཀཾ། །པྲ་བི་བྱེ་བ་བི་ནིརཱ་ཏེ་སོ་ཡནྣི་དྷི་ར་ས་ཏ། ་། རྩྭ་བཞིན་དུ་ནི་སྟེར་བར་བྱེད། །འདི་ནི་ཁྱོད་ཀྱི་གཏེར་ཞིང་འོངས། །ནང་ཞུགས་ཉིད་ནི་ཕྱི་རོལ་ཐོང་། །པྲ་བྷཱུ་ཏ་དྲ་བི་ཎ་སྭཱ་གཱི་བྱ་གྲ་ཏཱ་ད་ཙི་རསྦི་ཏི། །ཙྪནྣ་ཀཱ་མཀྵ་མཱི་རབ་མང་ནོར་ལྡན་གཏོང་ཕོད་ཅན། །བྲེལ་བས་ཡུན་རིང་མི་སྡོད་པ། །འདོད་བསྒྲིབས་འདོད་ལྡན་ཀཱ་མཱི་སུ་བྷ་གེ་ལ་བྱ་ཏེ་ཀུ་ཏ། །ཨི་ཏི་ཏ་སྱེབ་ཙ་ཐུ་ཏྭཱ་བྷ་དཱ་པྲ་བཱ་བཟོད་པ་ཅན། །སྐལ་བ་བཟང་མོ་ག་ལ་ཐོབ། །ཅེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །ཡོན་དང་ཆགས་པའིཙ་ས་སྨི་ཏཱ། ་། ཀྵ་ཎཾ་དོ་ལཱ་ཡ་མཱ་ནེ་བ་མ་དྨེ་དཱཀྐི་ཅུ་ལོ་བྷ་ཡ། །ཀ་བར་དུ་ནི། །སྐད་ཅིག་འཕྱང་མོ་ཡུག་པ་བཞིན། །བཟང་མོས་འཛུམ་བཅས་རབ་སྨྲས་པ། །གཞན་ཐཾ་ར་བྷཾ་ག་ནེ་བཱ་ཧ་མ་ནྱ་སྣ་དཱ་ཏུ་བེ་ཏ་ནཱ། །གཙྪ་མི་པྲཆི་ཏཱ་ནྱ་སྱ་ལས་གླ་ནི་བླང་བ་ཁྱེདག། ལམ་སྲང་དག་གི་བུད་མེད་བཞིན། །དེ་དུས་དོན་གཉེར་གཞན་ལ་ནི། །ལག་ཏདྨ་ནོཏྟཱ་ན་པཱ་ཎི་ཏཾ། །པྲ་ཛཱ་ནཱ་མི་བ་སརྦ་སྱ་སཱ་དྷཱི་ནཱ་ནཱ་མ་པི་ཀླུ་ཎཾ། །བཟེད་ཉིད་དུ་ཇི་ལྟར་འགྱུར། །སྐྱེ་རྒུ་རྣམས་བཞིན་ཐམས་ཅད་ཀྱི། །དབང་དུ་གྱུར་ཀྱང་སྐད་ཅིག་ནི། །པཱུཏྠ་མེ་བཱ་ཤྲཱི་ཏཱ་ཡེ་ན་ས་སྭཱ་མཱི་བེ་ཤ་ཡོ་ཥི་ཏྭཾ། །ཨེ་ཀི་བེ་ཡཾ་མྲྀ་ཎཱ་གང་གིས་སྔོན་ཉིད་འཐམས་པ་དེ། །སྨད་འཚོང་བུད་མེད་རྣམས་ཀྱིས་བདག། མཚན་མོ་གཅིག་པུ་འདི་—།
