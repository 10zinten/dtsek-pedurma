རི་བརྟ་ཏེ། །བ་ཧུསྟེ་ཤ་ཡ་ནཱ་ལམཱི་བི་ཏྱཾ་སཾ་རཀྟི་ཏོ་མ་ཡ། །སུ་རུང་བ་ཡོངས་སུ་རྒྱུ། །ཁྱོད་ཀྱི་ལག་པ་མལ་སྟན་ལས། །འཕྱང་®བ་འཇིགས་ནས་བདག་གིས་བསྲུངས། །ཨེ་ཏཏྠ་ཏུཏྟཱ་ཙཐུ་ཏཱ་ཏེ་ཐཾ་ཀཱ་ཛ་ག་ད་ས། །དིཤྩ་ཏུབ་ཏུཤཱི་ལ་མྱ་ཁྱོ་ཡི་ཚིག་འདི་ཐོས་གྱུར་ནས། །དོགས་བཏང་དགའ་བ་དེས་སྨྲས་པ། །བདེན་པའི་ངང་ཚུལ་ཁྱོད་ཀྱི་བློ། །ན་ཀཱ་མ་མ་ལི་ན་མ་ཏི། །སམྤ༔སཱ་དྷུཊྚ་རཱ་ཀོ་ཡཾ་བྷ་ཡཾ་རཱ་གོ་ར་གཱཏྠ་འདོད་པས་དྲི་མར་བྱས་པ་མེད། །གདུག་པའི་སྦྲུལ་ནི་ཁེགས་པ་སྟེ། །ཆགས་པའི་ལྟོ་འཕྱེས་མཆོག་ཏུ་རཾ། །ཨེ་ཀཾ་ཧནྟི་ཏི་ནཾ་སམྦཀཱ་མཀཱ་ཡ་ཤ་ཏྭཾ་ཏ་ཀྲྀ་ཏ། །རཀྵ་འཇིགས། །སྦྲུལ་གྱིས་ལུས་གཅིག་འཇོམས་པ་སྟེ། །འདོད་པས་ལུས་ནི་བརྒྱ་མཐར་བྱེད། །དེ་སླད་ཏམིདྨ་ཀཱ་རོ་ཡཾ་མི་ཏྱཱཀྟ་བི་ར་རཱ་མ་སཱ། །ཏཱི་བྲོ་ཙ་པྲ་ཤ་ཤྲཱི་སཱ་སཱུསཾ་རྣམ་འགྱུར་འདི་དག་བསྲུངས། །ཞེས་སྨྲས་དེ་ནི་ཆགས་དང་བྲལ། །དེ་ཡི་ཤིན་ཏུ་དུལ་བ་ཡང་། །པིསྤ་ཡ་མཾ་པིསྤ་ལཱ་ཡ་ན། །ནྱ་གྲོ་དྷ་ཀཝེ་ཀཱ་ལེ་ན་ས་བྷཱརྨེ་ཏྲ་དི་པཾ་གེ་ཏེ། །པྲ་ལ་སྐྱེས་དག་གིས་བསྔགས། །དུས་ཀྱི་རྒྱ་གྲོ་དྷ་ཡི་ཏོག། ཆུང་མར་བཅས་པ་མཐོ་རིས་སོང་། །རབ་བྷུ་ཏ་ཀྲྀ་ཧ་སཾ་བྷཱ་ར་ཙིནྟཱ་བྷཱ་ར་མུ་བཱ་ཧཱ་ས། །ཀ་དཱ་ཙིཏྟི་ལ་པཱ་ནཱ་ཡ་པྲྀ་ཥཱ་མང་ཁྱིམ་གྱི་ཚོགས་དག་གིས། །བསམ་པའི་ཁུར་ནི་དེ་ཡིས་བླངས། །ནམ་ཞིག་ཏིལ་མར་བཏུང་བའི་སླད། །ཎྞཾ་ཏི་ལ་པཱི་ཌ་ཎི། །ཏེ་ན་བྷ་དྷཱ་ས་མཱ་དིཥྛ་རྒྱ་ཡུཏྟཾ་པ་རི་བ་རི་ཁྱུ་མཆོག་རྣམས་ཀྱི་ཏིལ་འཚིར་ལ། །བཟང་མོས་ཡོངས་སུ་བསྟན་པ་ཡིས། །འཁོར་རྣམས་དག་ནི་ཡོངས་ཀཱ༑། ཏཱབི་ཌི་ཏ་ཏི་ལཱསྟེ་ལ་ཧཱུྃ་བྷེ་ཥཱ་ཀྲྀ་མི་སཾ་ཙ་ཡཾ། །ཙྩཾ་ཏཾ་སུ་བསྐོས། །ཏིལ་ནི་འཚིར་བ་དེ་དག་གིས། །ཏིལ་མར་བུམ་པར་སྐྱེསྲོག་ཆགས་ཚོགས། །་ལྷུང་ཞིང་སྟེབི་ལོ་ཀྱ་པཱི་དན་མུ་ཙུས་ཀ་རུ་ཎ་མི་ཐ། །ཨ་ཧོ་ནུ་པཱ་བ་མསྣ་ཀཾ་བ་ཧུ་ཉམས་པ་མཐོང་གྱུར་ནས། །སྙིང་རྗེ་བཅས་པས་ཕན་ཚུན་སྨྲས། །སྲོག་ཆགས་མང་པོ་ཡོངས་བརླག་པས། །པཱ་ཎི་པ་རི་ཥཱ་ཡཏ། །བྷ་དཱ་ཡཱ༔སརྦ་མེ་ཏདྷཱ་པཱ་ཏ་ཀཾ་ཡངྒི་རཱ་ཀྲྀ་ཏཾ། །ཀྱེ་མ་བདག་ཀྱང་སྡིག་པ་ཅན། །ཡང་ན་གང་གི་ཚིག་གིས་བྱས། །སྡིག་པ་འདི་ཀུན་བཟང་མོ་ལའོ། །—།
