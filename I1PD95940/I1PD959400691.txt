ག་ལ་ནི་དྷི༔ཀུ་ཤ་ལཾ་ཏ་ནོ་ཏི། །བྷ་ག་བརྗིཐི་ལཱ་ནཱ་མ་བི་ནཱི་ཡཀྵ་ཤིས་གཏེར་གྱིས་འཇིག་རྟེན་དག་གི་དགེ་བ་རྒྱས་པར་བྱེད། །མེ་ཏོག་ཅན་མ་ཞེས་པ་ཡི། །མཚན་རྒྱུ་ཎ་་ཙ་རིམྲ། །བི་ཛ་ཧཱ་ར་ཧ་རི་བ་ཏ་ནིཐཾ་ཀ་ཧ་རི་ཎ་བཏུལ་ནས་བཅོམ་ལྡན་འདས། །རི་དྭགས་འཕྲོག་བྱེད་བརྟུལ་ཞུགས་ནི། །དོགས་མེད་ལྡན་པའི་ཚལ་ན་བ་ནེ། །ཏ་ཏཱ་ནུ་ཡཱ་ཡི་ན་པྲ་ཋཔུ་ནཤཀྲེ་ཎ་ས་སྨི་ཏ། །ས་པུཏྤ་ཙ་བཞུགས། །དེར་ནི་རྗེས་སོང་བརྒྱ་བྱིན་གྱིས། །སླར་ཡང་འཛུམ་ལྡན་དག་ལ་དྲིས། །དེ་ཡི་མངོན་སྤྱོད་རི་ཏཾ་སྨྲ་ཏྭཱ་རྨི་ཏ་ཀཱ་ར་ཎ་མ་བྱཱ་དྷཱཏྲ། །དྷཱ་ས་ཏི་ས་ཧ་བྷཱབཱ་དིནྟཱ་ཡུ་ཥི་ཛ་དྲན་གྱུར་ནས། །འཛུམ་པའི་རྒྱུ་ནི་རབ་ཏུ་གསུངས། །ལོ་ནི་སྟོང་ཕྲག་བདུན་ཅུ་གཉིས། །སྐྱེ་བོ་ཚེ་རིང་ནེ་པུ་རཱ། །ཨ་བྷུདྨ་ཧེན་བ་ཏྭཱ་ཥཱ་ཛི་ཏ་སྭརྒོཏྲ་བཱ་པུ་རཱི། །མ་ཧེནྡྲ་སེ་གྱུར་ཚེ་སྔོན། །གྲོང་ཁྱེར་དབང་ཆེན་ལྡན་ཞེས་པ། །མཐོ་རིས་དགའ་བ་ཐུལ་བར་གྱུར། །དེར་ནི་ནོར་གྱི་ན་ཨི་ཏྱ་སཱིཏྟ་སྱ་བ་སུ་མ་ཏཱི་པ་ཏི། །ཡཉྩ་ཀྲེ་ཀཱིརྟི་ཀཆུ་ར་བ་ཏྭཱ་བི་ཏི་མི་ར་བློ་གྲོས་བདག། དབང་ཆེན་སྡེ་ཞེས་བྱ་བ་བྱུང་། །གང་གི་གྲགས་པ་ག་པུར་གྱི། །སྡོང་བུས་ཕྱོགས་ནི་དི་ཤ། །རི་པུ་དསྨྲ་ཛཱ་ར་ཧ་རཀྲྀཙྪཱ་ཧྲཇུརྡ་ཤཱ་ཛུ་ཏྭཾ། །བྱ་མུན་བྲལ་བྱས། །གང་ཞིག་སྨན་པ་དམ་པ་བཞིན། །དགྲ་ཡི་གྲགས་པའི་རིམས་བཅོམ་ཞིང་། །སྐབས་དྷཏྟཾཥྛ་ས་ཧ་སྭསྠཱ་ཡ༔སདྡྷིརླ་ཨི་བ་པྲ་ཛཱ། །ཏ་སྭཱ་བྷཱུ་ཏ་པུ་ཎྱ་སཾ་བྷཱ་ངན་བསྟེན་པའི་ནད་ཕྲོགས་ཏེ། །སྐྱེ་རྒུ་སྲེད་བྲལ་རྣལ་གནས་བསྒྲུབས། །བསོད་ནམས་ཚོགས་ནི་རྣམ་ར་ཨི་བ་སཱ་ཀཱ་རཾ་ག་ཏ། །པུ་ཏྲས་ཏོ་ཥ་དྷཾ་ནཱ་མ་སརྦ་ས་ཏྭཱ་ཧི་ཏོ་བཅས་ཉིད། །གྱུར་པ་བཞིན་དུ་དེ་ཡི་བུ། །སེམས་ཅན་ཀུན་ལ་ཕན་བརྩོན་པ། །སེམས་ཅན་སྨན་ཞེས་བྱ་བྱ་ཏ། །ས་བྷ་དྲ་ཀཝི་ཀ་བོ་དྷི་སཏཱ༔ས་ཏྭཱ་བི་བྷུ་ཥི་ཏ། །ས་ཀ་བར་གྱུར། །དེ་ནི་བསྐལ་པ་བཟང་པོ་པའི། །བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔས་བརྒྱན། །སྙིང་རྗེ་རུ་ཎཱ་མུ་དི་ཏོ་བེ་ཀྵ་མི་ཏྲི་ཏཱཾ་བསྨྲ་བྷོ་བྷ་བཏ། །པུ་ར་གྲྭ་མ་བ་ནཱནྟ་བྷོ་དགའ་དང་བཏང་སྙོམས་དང་། །བྱམས་པ་རྣམས་ཀྱི་གྲོགས་སུ་གྱུར། །གྲོང་དང་གྲོང་ཁྱེར་ནགས་མཐའ་—།
