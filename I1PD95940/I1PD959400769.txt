མུཉྩི་ཀ་ཀ་སྱཝཾ་བྷ་བ་བནྡྷ་ནཾ། །བཱཋ་སྱ་བྱ་ས་ཏ་ཏ་ས་ནེ་ཧོ་མ་དྷུ་མེནས་ནི། །སྲིད་པའི་འཆིང་བ་སུ་ཞིག་འདོད། །དང་པོར་སྦྱིན་སྲེག་དུ་བས་མཆི་མ་རྟག་ཏུ་འབབ་ལ་རབ་པྲ་པྲཏྟིཔ་ཏུ་གཙླིབྱ་པ་ན་ཤ་ར་ཎོ་ཏུ་ལ་ཧ་སྟཧྥ་ཎེ་ན། །སཾ་ཏུ་འཇུག། མཚུངས་པར་ལག་པ་གཏད་པས་གདུང་བའི་ལམ་ལ་བདེན་པར་མདུད་པ་མཛད། །འཕྲེང་སཱ་རཱ་ཛཱ་ས་མ་ཡ་ཙ་ལ་ནེ་བནྡྷ་ནཾ་མཱ་ལེ་དཱམ་མོ་ཧཱ་རོ་ཧོ་པ་ཧ་ཏ་མ་ན་ཧཱུཾ་ཧནྟ་བའི་ཐག་པས་འཁོར་བའི་བཀའ་ཡི་དམ་ཚིག་དག་ལ་གཡོ་བ་འཆང་། །ཁྱིམ་ཐབ་ཡིད་ནི་རྨོངས་པ་གནས་ཧེ་ཏུདྲི་བཱ་ཧ། །ཡི་ཏུ་ནི་ལ་ལོ་ལ་བཱ་ལ་ལ་ལ་ན་བལྤ་བི་ལཱ་པས་ཉེར་བཅོམ་རྣམས་ཀྱི་དགའ་བའི་རྒྱུ། །ན་ཆུང་གཞོན་ནུའི་འཁྲི་ཤིང་གར་གྱི་རླུང་གིས་བསྐྱོད་པའི་རྩེ་ས་ནུ་གཱ་བཱི་ཎཱ་པཾ་ཤ་ར་བཱ་བི་བཱ་ཧ་ས་མ་ཡེ་པྲྀཏྟཱ་ཧི་ཏིརྣ་ཐུ་ཏ། །དགའི་རྗེས་འབྲང་བའི། །རྒྱུད་མང་གླིང་བུའི་སྒྲ་ནི་ཁྱིམ་ཐབ་དུས་སུ་སྤྲོ་ལྡན་གང་གིས་མ་མཉན་པ། །ཧ་པུ་ཏྲེ་ཏི་བ་པུ་པྲ་ལཱ་པ་ཀ་ལ་ན་ཀཱི་བཱ་ན་ཏེ་ཏཱཾ་ཤི་ཏཱ་ཐཱ་ཡནྟི་གུ་རུ་བཱཋ་གངྒ་དེ་དག་ལ་བརྟེན་ཀྱི་ཧུད་བུ་ཞེས་ཆུང་མས་ཆོ་ངེ་སྨྲ་ཞིང་དམན་གྱུར་པ། །མཆི་མ་ལྕི་བར་ཐིགས་ཐིགས་ད་ག་ལངྒ་ཊ་བྷི་ཡོ་གཱ་གི་རི། །ཨི་ཏྱཱཀྐཱ་ཏི་བྲ་ནི་དྷནྟཾ་པི་ཏ་རཾ་ཛ་ན་འབབ་པས་མངོན་པར་སྦྱར་བའི་སྒྲ་གྲགས་ཐོས་མི་འགྱུར། །ཞེས་པ་འབད་པ་དྲག་པོ་ཡིས། །ཕ་དང་མ་ནཱི་ཙ་པ། །ཏ་མ་ཀ་ཙྩཾ་ས་མཱ་དཱ་ཡ་ཀུ་ཤ་ལིཤིཝཾ་བྷིཀྲྀ་ཏྭཾ། །སཾ་ལ་དེས་སྨྲས་ནས། །མཛེས་པས་དྲན་པ་དགེ་མཚན་གྱིས༄༅། །མྱུ་གུ་ཡང་དག་སྐྱེས་པ་བཞིན། །བཟོ་ཛཱཏྟཾ་ཀྲནྟི་པིཀྟ་སྱ་ཀོ་ཏུ་ཀ་སྐྱེ་བ་ཀནྡ་ལིཾ། །ཀུལ་བི་ཙྪེ་ད་ཙིཏྟཱཏྟཾ་དྷཱི་རཿབོ་མཁས་པས་བྱས་པ་ཡི། །གསེར་གྱི་བུ་མོ་བཟུང་ནས་ནི། །རིགས་ཆད་བསམས་པས་གཟིར་དེ་ལ། །པུ་ན་རུ་བཱ་ཙ་ཏི། །ཏུ་ལུ་ལཱ་བཱཙྩཱ་བཎཱ་སྱ་ལ་བྱ་ཏེ་ཡ་དི་ཀ་ནྱ་ཀཱ། །བརྟེན་པས་སླར་ཡང་རབ་སྨྲས་པ། །ཁ་དོག་མཛེས་སྡུག་འདི་མཚུངས་པའི། །བུ་མོ་གལ་ཏེ་རྙེད་གྱུར་ན། །ཀ་རོ་མི་ཏཱིདྷི་རཱ་ཏཱ་ཏ་ཏ་ད་ཧཾ་དཱ་ར་སཾ་གྲ་ཧཾ། །ཨི་ཏི་བུ་ཏྲ་བ་ཡབ་གཅིག་ཁྱོད་ཀྱི་ཚིག་གིས་ནི། །བདག་གི་བུད་མེད་བཟུང་བར་བགྱི། །ཞེས་པ་བུ་ཡི་ཚིག་ཐོས་—།
