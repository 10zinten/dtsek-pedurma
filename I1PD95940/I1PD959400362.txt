པྲ་ཐི་ཏཾ་སཏྱ་ཐེ་ནེ་བ་ནི་ཛ་མེ་ནི་མ་ནོ་ར་ཐཾ། །ནིབྲྀཏྟེནྣཱཀྵ་ཎཾ་ཛཾཟླ་རང་གི་འདོད་པ་ལེགས་པ་ཡིསི། །ལམ་ནས་སོང་བ་བཞིན་དུ་སེམས། །དེ་ཡི་མྱང་འདས་མཚན་ཉིད་ས་སྭ༈་ཀཱ་ཎ་ནི་དྲ་ཡཱ། །ཨ་ནུཏྟ་ར་ཛྙཱ་ན་ནི་དྷི་པྲ་ཏྭཱ་སནྣ་མ་ཀྱིས། །རྨི་ལམ་སྐད་ཅིག་གཉིད་ཀྱིས་མཐོང་། །བླ་ན་མེད་པའི་ཡེ་ཤེས་ཀྱིས། །གཏེར་ལ་རབ་ཏུ་ཉེ་མ་རྱ་ཏ། །ཏ་ཏཔྲ་བུདྷ་ས་ཧ་སཱ་ཏྲ་ལྟ་དེ་བི་ཡ་ཤོ་དྷ་རཱ། །ཏ་ཏ་ཀཱ་ལོ་བར་བསམ། །དེ་ནས་ལྷ་མོ་གྲགས་འཛིན་མ། །འཕྲལ་ལ་རབ་སད་སྐྲག་པ་ཡིས། །དེ་དུས་མཐོང་པ་ན་ཏཾ་སཱནྣཾ་ད་ཡི་ཏཱ་པ་ནྱ་བེ་ད་ཡཏ། །པÉངྒཱ་བྷ་ར་ཎཱ་གཱ་ནི་སྭསྣེ་བྷསྨཱ་བའི་རྨི་ལམ་ནི། །བདག་པོ་ཉིད་ལ་རབ་ཏུ་བཤད། །ཇོ་བོ་བདག་གི་ཁྲི་རྒྱན་ལུས། །ཆག་དང་དཔལ་ན་མེ་བི་དྷོ། །ཤྲཱི་རྦ་ཛནི་མ་ཡཱ་ཛྲཟླ་ཙནྡཱཀྐོ་ཙ་ཏི་རོ་ཧི་ཏོ། །ཨི་ཏུ་ཀནྡྷུ་ནི་འགྲོ་བ་དང་། །ཉི་ཟླ་དག་ཀྱང་བསྒྲིབས་གྱུར་པ། །རྨི་ལམ་དུ་ནི་བདག་གིས་མཐོང་། །དེ་ཐོས་དེ་ཡིས་སཱ་ཏཱ་མཱུ་ཙེ་མུངྒེ་ས་ཏྱཱ་བི་བཛྫི་ཏ། །སཾ་སཱ་ར་ཨེ་བ་སཱམོ་ཡཾ་སྭསྣེ་སཱསྒོ་དེར་སྨྲས་པ། །མཛེས་མ་བདེན་སྤངས་འཁོར་བ་འདི། །རྨི་ལམ་ཉིད་ནི་རྨི་ལམ་གྱིའང་། །རྨི་ལམ་དེ་པི་ཀཱི་ཛཾ་ཤ། །སཱསྣེ་དྲ་ནཱ་བྷི་སཾ་ཛཱ་ཏ་ལ་ཏཱ་བྷཱཥ་བི་ཧཱ་ས་སཱ། ་། མེ་རཱུ་པ་དྷཱ་ནི་ཅི་ཞིག་འདྲ། །དེ་རིང་རྨི་ལམ་ལྟེ་བ་ནས། །འཁྲི་ཤིང་སྐྱེས་པས་ནམ་མཁའ་ཁྱབ། །ལྷུན་པོ་མགོ་ཡི་ན་ཤི་ར་སཱ་པཱུབྦ་པཉྩི་མ་བཱ་རི་དྷི། །བྷུ་ཛཱ་ཧཱུཾ་ཙ་ར་ཎཱ་བྷྱཱཉྩ་དཀྵི་ཎཱབྷིམྨ་སྔས་སུ་བྱས། །ཤར་དང་ནུབ་ཀྱི་ཆུ་གཏེར་གཉིས། །ལག་པ་གཉིས་དང་རྐང་པས་ཀྱང་། །ལྷོ་ཡི་རྒྱ་མཚོ་ཡཱ་ཤྲཱི་ཏ། །བྷ་དྲེ་སྭརཿཤུ་བྷོ་ཡཾ་མེདྷཱི་ཎཾ་བྷཏྲི་ཤུ་བྷཾ་ཤུ་བྷམ། །ཨི་ཏུལྟེ་ང་ཡིས་བཟུང་། །བཟང་མོ་བདག་རྨིས་འདི་ནི་དགེ། །བུད་མེད་བདག་པོ་དགེ་བས་དགེ། །བྱང་ཆུབ་བོ་དྷི་སཏྭེ་ན་ནོ་ཙེ་ཀིཉྩརླ་ཤོ་དྷ་རཱ། །པུ་ན་ནི་དཱ་བྷི་མུ་ཁེ་བ་སེམས་དཔས་དེ་བརྗོད་ཚེ། །ཅི་ཡང་མི་སྨྲ་གྲགས་འཛིན་མ། །སླར་ཡང་གཉིད་ལ་མངོན་ཕྱོགས་ཏེ། །བྷཱ་བཱ་མི་ལི་ཏེ་ཀ་ཎ། །ཤ་ཀྲ་བྲཧྨ་མུ་ཁ་སདྡྷེ་ས་མེ་ཏུ་ཐ་སུ་དྷ་བྷུ་ཛ། །མིག་ནི་ཟུམ་པ་ཉིད་དུ་གྱུར། །དེ་ནས་བརྒྱ་བྱིན་ཚངས་ལ་སོགས། །བདུད་རྩི་སྤྱོད་རྣམས་ཡང་དག་འོངས། །།
