བསྟན་འགྱུར། འཁྲི་ཤིང་། །ཁི་ཏེ༑། སུ་ཏཱ་ཀཱཉྩ་ན་མཱལཱ་ཥཱ་ཀ་ཙཀཾཉྩ་ན་མཱ་ལི་ཀཱ། །ཀཱ་ཤུ་པཱ་ཀྵ་ཡི། །བུ་མོ་གསེར་འཕྲེང་ཅན་ཞེས་པ། །གསེར་འཕྲེང་སྐྲ་ཉིདང་ལྡན་པ་བྱུང་། །སྟོན་པ་འོད་སྲུངས་སྱ་ཤyས་ས་ཏ་ཏཾ་བྷཏྟཾ་ཤཱ་ལི་ནི། །པ་རི་ཙན་བྱ་དྷཱཏྟཱདྷཾ་ས་ཁཱི་ནཾ་པཉྩ་བྷིཞེས་པ་ལ། །རྟག་ཏུ་གུས་དང་ལྡན་པ་དེས། །གྲོགས་མོ་བརྒྱ་ཕྲག་ལྔ་དང་ནི། །ལྷན་ཅིག་དག་ཏུ་རིམ་ཤ་ཏི། །ས་ཀ་དཱ་ཙིནྡ་ར་པ་ཏིཥཱི་ཀྲྀ་ཏ་སྭཱབྷ་དན་ནཱཏ། །བྷ་ཡ་སཾ་གྲོ་བསྒྲུབས། །ནམ་ཞིག་ས་ཡི་བདག་པོ་དེས། །རྣམ་འགྱུར་རྨི་ལམ་མཐོང་བ་ལས། །འཇིགས་པའི་ཐེ་ཤ་ཡ་སཾ་བྷཱནྟཔ་པྲཙྪ་ཕ་ལ་ཀོ་བི་དྷཱན། །ཏེ་ཏཾ་རཱ་ཛ་སུ་ཏཱ་དྭེ་ཥཱནི་མིཏྟ་ཛྷཱ་ཚོམ་ཀུན་འཁྲུལ་བས། །འབྲས་བུ་མཁས་པ་དག་ལ་དྲིས། །རྒྱལ་པོའི་བུ་མོ་ལ་སྡང་བས། །མཚན་ཤེས་དེ་བ་བྷཱ་ཥ་རེ། །ཨ་ཏི་པྲ་ཡ་མྱ་ཧྲི་ད་ཡཾ་ཏུ་ཏཱསྒོ་ལ་བྱ་ཏེ་ཤི་ཐཾ། །ཨི་ཏི་ཥ་དག་གིས་དེར་སྨྲས། །ཤིན་ཏུ་མཛའ་བ་དག་གི་སྙིང་། །མེ་ལ་བསྲེགས་ན་ཞི་བ་ཐོབ། །ཅེས་པ་དེ་དག་ཥཱ་མ་ནཱ་དྲྀ་ཏྱ་པ་ཙ་ཀཱུ་ར་ཏ་རཾ་ནི་པ། ་། བྷ་ག་བནྟཾ་ཡ་ཡོ་བྲྀཧཱུྃ་ཀཱ་བྱ་པཾ་རྣམས་ཀྱི་ཚིག། རབ་གདུག་མ་བཟུང་མི་བདག་ནི། །བུ་མོའི་ཚིག་གིས་བཅོམ་ལྡན་འདས། །འོད་སྲུངས་དུ་ཧི་ཏུནི་རཱ། །ས་ཏ་མ་བྱེ་ཏུ་པ་དཏྟཱས་བི་ཀཱ་རཿཔ་རམྨ་ཡཱ། །དག་ལ་བལྟ་®རུ་སོང་། །དེ་ལ་ཕྱོགས་ཏེ་དེས་གསོལ་བ། །རྨི་ལམ་མཆོག་ཏུ་རྣམ་འགྱུར་ཅན། །དྲྀཏྠོ་བྱསཐཾ་སཧྥ་ཛྷ་ཏཏྠལཾ་བཀྱུ་མན་སི། བ། བཱ་ཏཱ་ཡ་ནེ་བདག་གིས་དིང་མཐོང་དེ་འབྲས་ཀུན། །ཐམས་ཅད་མཁྱེན་པས་གསུང་བར་འོས། །གླང་ཆེན་རྒྱ་སྐར་ནིརྒཙྪཱནྟུདྡྷཱ་སུཙོ་མ་ཡཱ་ག་ཛ། །ཀྲྀ་ཥི་ཏ་སྱ་ཏ་ཐཱ་པཉྩཏ་པཱ་བོ་དྷཱ་པཎྚི་ལོ་ནས་འཐོན་པ། །རྔ་མ་དག་ལ་ཐོགས་པ་དང༌། །དེ་བཞིན་སྐོམ་པའི་ཕྱིར་རྗེས་ནས། །ཁྲོན་པ་རྒྱུག་པ་ཀི་ཏ། །དྲྀཋཧྥ་སཀྱུ་པྲསྦེ་ན་མོཀྟི་ཀ་པྲསྦྲ་བི་ཀྲ་ཡ། །ས་མཱི་ཀྲྀ་བདག་གིས་མཐོང་། །ཕྱེ་ནི་བྲེ་གང་དག་ལ་ཡང་། །མུ་ཏིག་བྲེ་གང་འཚོང་བ་མཐོང་། །ངན་པའི་ཤིང་ཏཱ་ནི་པྲཏྟཱ་ནི་ཀུ་དཱ་རུ་ཎི་ཙ་ཙནྡ་ནི། །ཀ་ལ་བྷེ་ན་མ་ཧཱ་ཧསྟི་ས་མཱ་ཧུ་དང་ཙན་དན་ནི། །མཉམ་པ་དག་ཏུ་བྱེད་པ་མཐོང་། །དེ་བཞིན་གཡུལ་དུ་གླང་ཕྲུག་གིས། །གླང་པོ་།
