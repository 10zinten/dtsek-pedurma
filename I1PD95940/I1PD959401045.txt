ད་ཤ་རཱ་ར་པཐ་པུ་ར་ཏཏྠཧཱུ་ལ་མབྷྱཱ་དེ་ཤ་མཱུནྟི་བི་བ། །པྲ་བི་བྱ་སྒོ་ལམ་བཅོ་བརྒྱད་ལྡན་པ་དེར། །མཚུངས་པའི་སྐུ་ནི་བཅོ་བརྒྱད་དང་བཅས་པས། །རབ་ཞུགས་མ་ག་ཙ་ཀྲེ་ཤ་ཤི་ཀཱནྟ་རཏྣ་ཤི་ལ་མ་ཡཾ་བེཐ་སུ་མ་ག་དྷཱ་ཡཱ། །ཨ་བྱཱཙཱི་ཏཾ་ཏ་ཏྲ་བ་ཧུ་པྲ་དྷ་ནི་བཟང་མོའི་ཁྱིམ། །ཟླ་ཤེལ་རིན་ཆེན་རྡོ་ཡི་རང་བཞིན་བྱས། །དེར་ནི་ཡོ་བྱད་རྫོགས་པ་རྣམ་མང་གིས། །ཀཱ་རིཔུཎྞོ་པ་ཙཱ་རིཔྲ་ཎི་པ་ཏྱ་སམྦི། །ཨཱ་པཱུ་ཛ་ཡཊོ་ར་ཛས་མཱ་ཤྲཱི་ཧིཉྩ་མངོན་པར་མཆོད་ཅིང་ཀུན་གྱིས་ཕྱག་འཚལ་ཏེ། །གྲོང་པའི་སྐྱེ་བོ་དག་གིས་ཀུན་ནས་མཆོད། །རྩིག་པར་བྷིཏྟི་པྲ་ཏི་བིམྤི་བྷཱནྟ། །སུ་མཱ་ག་དྷཱ་ཡཱ་ད་ཡ་ཡཱ་ད་ཡཱ་ལུཔཱུ་ཛི་ཐཾ་ཧཱི་ཏཱ་གཟུགས་བརྙན་སྣང་དེ་ཕྱི་ནས་ཀྱང་། །བཅོམ་ལྡན་བརྩེ་ལྡན་དགེ་འདུན་དང་བཅས་པས། །བརྩེ་བས་བྷ་ག་བཱརྨ་སཾ་གྷ། །ཨ་ནུ་གྲ་ཧཱ་ལོ་ཀ་ན་སམྦི་བྷ་གིས་བནྟ་མཆོད་པ་བླངས་ནས་རྗེས་བཟུང་ནི། །གཟིགས་པའི་ཆ་ཡིས་མ་ག་དྷ་བཟང་མོ། །གཉེན་དང་བཅས་པ་བཱ་ཡཱ་བི་ད་དྷེ་པྲ་སཱ་དཾ། །སུ་མཱ་ག་དྷཱ་ས་ཤཱ་ཤུ་རཱ་དི་བརྒ་ས་ཧཱ་པ་རིབོ་ར་ཛ་རྣམས་ལ་བཀའ་དྲིན་བསྒྲུབས། །མ་ག་དྷ་བཟང་གོས་སོགས་སྡེ་དང་བཅས། །གྲོང་པའི་སྐྱེ་བོ་ཀུན་དང་ན༔་སཉྫ། །ཤཱyསྟ་དཱ་དེ་ཤ་ན་ཡཱ་བ་བྷཱུ་བཤུདྡྷཱ་ཤ་ཡཾཏཏྟཱ་ཎ་དྲྀཥཱ་སཏྱ། །བཅས་པ་ཡང༌། །དེ་ཚེ་སྟོན་པའི་བསྟན་པ་དག་གིས་ནི། །དེ་དུས་བདེན་མཐོང་བསམ་པ་དག་པར་གྱུར། །སུ་མཱ་ག་དྷཱ་ཡཿཀུ་ཤ་ལཱ་ནུ་བདྷཾ་པུ་ཅུ་པྲ་བྷཱ་ཁཾ་བི་པུ་ལཾ་བི་ལོཀྱ། །ཏེ་བྷི་དགེ་བ་རྗེས་འབྲེལ་མ་ག་དྷ་བཟང་གི། །བསོད་ནམས་མཐུ་ནི་ཡངས་པ་མཐོང་གྱུར་ནས། །དགེ་སློང་དེ་བསྟ་ཏྲ་ཀུ་ཏུ་ཧ་ལེ་ན་པ་པྲནྡྷུ་ར་སཱུ་ཛི་ན་མཱ་དི་བྲྀཏྟཾ། །པྲཉྫས་ཏིསཾ་ས་དི་དག་གིས་དེར་དགེ་མཚན་ལས། །དེ་ཡི་སྔོན་བྱུང་རྒྱལ་བ་དག་ལ་དྲིས། །མ་ག་དྷ་བཟང་དགེ་བའི་རྒྱུ་དག་སཏྠ་དནི་སུ་མཱ་ག་དྷཱ་ཡཿཀུ་ཤ་ལ་སྱ་ཧེ་ཏུཾ། །དནྟ་པྲ་བྷཱ་བྷིཀ་ཀུ་བྷཱ་མུ་ཁེ་ཞུ་ནི༑། ཚོགས་སུ་དེ་དག་གིས་དྲིས་བཅོམ་ལྡན་འདས། །ཀུན་གཟིགས་དེ་ཡི་ཚེམས་ཀྱི་འོད་རྣམས་ཀྱིས། །དི་ཤནྡྲ་ཀཱ་ཤ་བྷ་ག་བཱནྡྲ་བྷཱ་ཥ། །བཱ་རཱ་ཎ་ས་མ་བྷུཏྟཱཐཾ་ཀྲི་ཀྲིཀཱནྟ་སྱ་བྷཱུ་བ་ཕྱོགས་སྒོར་སྣང་བ་བྱིན་ཅིང་རབ་གསུངས་པ། །བཱ་རཱ་ཎ་སཱི་དག་ཏུ་སྔོན། །ས་བདག་མཛེས་པ་ཀྲི་ཀྲི་—།
