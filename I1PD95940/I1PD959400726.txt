ཡེ་པི་ས་དྭེ་ཤ་ད་ཤཱ་མ་བཱསྟ། །སཏྟིསྟ་ནོཏྟཾ་ག་ནི་ཥཎྜ་མཱདྷཱ་ཀཱ་འོས་རང་བཞིན་ཆགས་བྲལ་གྱིས་སྨོད་ཅིང༌། །བཙུན་མོའི་ནུ་མའི་བར་དུ་མགོ་བཞག་པའི། །སྐད་མ་་ན་གླུ་པ་ཏི་རཱ་ཙ་ཙཀྵེ། །བི་དྱཀི་མ་རཱུ་པི་ནི་བྲྀཏྟ་བི་བྱིབྱ་ཐཱ་ནི་མིཏྟིགདངས་ཞན་པ་ས་བདག་གིས་སྨྲས་པ། །རིག་བྱ་རྫོགས་པའི་སྨན་པས་དིང་ཡང་ཅི། །གདུང་བ་སྐྱེད་ཀི་མ་ཏ་ཐ་པ་ཕྱི། །ཨ་མན་ཀརྨོ་པ་ནི་པཱི་ཌི་ཏཱ་ནཾ་དྷརྨོ་པ་དེ་ཤ་བྱེད་བདེན་མེད་དཔྱད་ཀྱིས་ཅི། །བདེ་ཉམེད་ལས་ཀྱིས་ཡོངས་སུ་གཟིར་བ་རྣམས། །གསོ་བྱེད་ཆོས་ཀྱིས་པྲ་ཎ་ཡཙྪི་ཀིཏྟཱ། །ཀཱ་ཡ༔པྲ་ཡཱ་ཏོ་ཡ་མ་པཱ་ཡ་བྷཱུ་མི་ཤ་ལྱཱ་ཡ་ཏེ་བྷོ་ཉེར་བསྟན་ལ་ཆགས་ཡིན། །ལུས་འདི་གནོད་པའི་སར་སོང་དཔྱད་མི་འོས། །ལོངས་སྤྱོད་ཚོགས་ཀྱང་ག་ག་ཎོ་བྱ་བྷོ་གྱ། །ཨནྡྷ་བྱ་ལཱ་བ་ཏྱཱ་བ་ཏཱི་བ་ཀཱནྟཱ་བྷོ་གོངྷི་ཏཱ་ཤྲཱིགྷ་ན་ཨེ་བ་ཟུག་རྔུ་ལྟ་བུར་སྤྱོད། །ལོང་བའི་ཆུང་མ་མཛེས་སྡུག་ལྡན་པ་བཞིན། །སྤྱོད་སྤངས་དཔལ་ནི་སྨོད་པ་སྟུག་ཤཱ་པ། །ཨ་ཏྱནྟ་མནྟཱཀྐི་པ་རི་པྲ་སཏྟ་པྲ་དཱམཿཤཱ་ཀཱ་ན་ལ་ད་རྒྱ་མཱ་ན། །པོ་ཉིད། །མེ་དག་ཤིན་ཏུ་དམན་ཡང་རྒྱུན་དུ་ནི། །རབ་འབར་མྱ་ངན་མེ་ཡིས་སྲོག་བྱེད་ཅིང་། །པྲ་དྲྀདྡྷ་དྲྀཊྛོ་བྱ་ན་པེ་ཏ་ཛཱ་ཊུསུ་ཁཱ་ག་ཏཱ་ཤུན་ཏུ་དིགྷ་རོ་གཱི། །པྲཙྪསྣ་གདུང་བ་རྒྱས་ཀྱང་རེངས་དང་མི་འབྲལ་བའི། །རིང་པོའི་ནད་ལྡན་མ་ཡིན་སྲོག་སོང་བདེ། །སྡིག་པ་རབ་མནྟཔ་རི་པནྟི་པ་པཾ་ནཱི་ཙཱ་བ་མཱ་ནཀ་ལ་ཧཱ་ནུ་བིནྡྷ། །བྱཱ་དྷི་སྦེ་བསྒྲིབས་ནང་དུ་ཡོངས་གནས་དང་། །དམན་པ་བརྙས་ཤིང་རྩོད་པ་རྗེས་འབྲེལ་དང་། །ནད་ཀྱི་ཉེབསྟན་རཱ་རཾ་བྷ་ཛུ་གུཔྤི་ཏ་དཱིདྷཱགཱི་ཏཱ་སེ་ན་ས་མཾ་པྲ་ཡཱནྟི། །ཀུ་ཀརྨ་ཎཱ་མེ་ཥ་པར་ཟིལ་མནན་སྨད་པ་དང་། །རབ་འབར་མེ་ཡི་གདུང་བ་ཞི་བར་འགྱུར། །ལས་ངན་རྩེ་དགའ་ཉོན་བི་ཙི་ཏྲ་རཱུ་པ་བི་པརླ་པ་ས་མ་ཡོ་བི་ལཱ་ས། །དཱ་རི་ཎྱ་ཀཝཾ་ཡ་ད་མོངས་རང་བཞིན་འདི། །རྣམ་པར་བཀྲ་བའི་གཟུགས་ནི་ཟློག་ཅེཔ་སྟེ། །ནད་མེད་བསྟེན་གང་དབུལ་བས་རོ་ག་བྷཱ་ཛྭ་ལ་གླི་བ་ཏཾ་ཡཙྪ་ས་དེ་བ་རོ་ག། །བཙྩཾ་ཛརྨ་ཤ་རཱི་རི་ཎཱ་བི་ར་ཧི་སྡུག་བསྔལ་ཞིང་། །ཕུན་ཚོགས་ལྡན་གང་རྟག་ཏུ་ནད་ཅན་ནོ། །རྣམ་དཔྱོད་རྒྱས་པའི་བློ་དང་བྲལ་བའི་།
