མཾ་ཏ་མཁ་ལ་སཾ་ག་མ། །པྲ་ཏྱཾ་ཀ་བུདྡྷཱཙཱ་རཱ་ཛན་ཀཱ་ལེ་ན་ཀཱི་ཎ་：བསྐྱེད་དེ་མུན་པ་དང་ནི་མི་བསྲུན་གྲོགས། །རྒྱལ་པོ་དལ་བུ་དལ་བུ་ཡིས། །དུས་ཀྱིས་སྡིག་པ་ཟད་ཀིཝི་ཥ། །བྷ་བི་ཥཱ་པི་བི་བེ་ཀེ་ན་ཀྲྀ་ཏཱ་ལོ་ཀཤ་ནིཤ་ནི། །ཨི་ཏི་ཏ་གྱུར་ནས། །དབེན་པས་སྣང་བར་བྱས་པ་ཡི། །རང་སངས་རྒྱས་སུ་ཁྱོད་འགྱུར་རོ། །ཞེས་པ་བཅོམ་སྱ་ད་ཡཱ་སྭཱ་སཾཉྩ་ཀཱ་ར་བྷ་ག་བཱཎྜི་ན། །པ་ཏི་ཏེ་ཥ་དྷི་ཀ་སནྟལྡན་རྒྱལ་བ་ཡི། །བརྩེ་བས་དེ་ནི་དབུགས་དབྱུང་མཛད། །ལྷུང་ལས་ལྷག་པར་དམ་པ་རྣམས། །ཀ་རུ་ཎཱ་སྣེགྷ་ལོ་ཙ་ནཱ། །ཏ་ཏཔྲ་ཎ་མྱ་སུ་ག་ཏཾ་པྲ་ཡཱ་ཏསྭཱ་པ་དཾ་ཀྲྀ་སྙིང་རྗེ་བྱམས་པའི་སྤྱན་དང་ལྡན། །དེ་ནས་བདེ་གཤེགས་ལ་བཏུད་དེ། །མི་བདག་རང་གི་གནས་སུ་པ༑། མ་ཧ་ཏིཔཱ་པ་བྷཱ་ར་སྱ་བི་བེ་དེ་ལ་གྷུ་ཏཱ་མི་བ། །ཏསྨིནྡྲ་ཡཱ་ཏེ་སརྦ་སོང་། །སྡིག་པའི་ཁུར་པོ་ཆེན་པོ་ནི། །ཡང་བ་ཉིད་བཞིན་རྣམ་པར་རིག། དགེ་སློང་རྣམས་ཀྱིས་དེ་སོང་ཛྷཔྲཋསྟ་སྱ་ཀརྨ་ཀོ་ཏུ་ཀཱཏ། །བྷིཀྵ་བྷིཀྵི་ཏཾ་པཱ་ལ་སྱ་པཱུཏྤ་བྲྀཏྟ་མ་ཚེ། །དེ་ཡི་ལས་ཀྱི་དགེ་མཚན་ལས། །ས་སྐྱོང་གི་ནི་སྔོན་བྱུང་མཐའ། །དྲིས་ཤིང་བཅོམ་ལྡན་གྱིས་བྷཱ་ཥ་ཏ། །བཱ་རཱ་ཎ་ཧཱུཾ་ནི་རཱ་ཡཱ་ས་བི་ལཱ་ས་བྱ་བ་སཱ་ཡི་ན། །ཙ་ཏྭཱ་རཿཐེཊྛི་བཀའ་སྩལ། །་བཱ་རཱ་ཎ་སཱིར་ཚོང་དཔོན་གྱི། །བུ་བཞི་དག་ནི་བྱུང་གྱུར་ཏེ། །དཔལ་ལྡན་བསྡམས་པ་ཏ་ན་ཡཱབ་བྲུ་བུ༔ཤི་བི་ཛྲ་ཁ་ལ། །ཏེ་ཀ་དཱ་ཙིཏྠ་ཁ་ཀཱི་བཱ་མ་ཐིཀ་དག་བྲལ་ཞིང་། །གདུང་མེད་རྩེ་དགའ་ལ་བརྩོན་པའོ། །ནམ་ཞིག་བདེ་བས་མྱོས་གྱུར་ཅིང་། །ཕན་ཚུན་ལི་ཀ་ཐཱསྦི་ཏཱ། །པྲ་ཏྲེ་ཀ་བུདྡྷཱ་མཱ་ཡཱནྟཾ་ད་དྲྀ་ཤུརྨོ་བ་ནོདྷཱ་ཏཱ། །ཀྱལ་ཀའི་གཏམ་གྱིས་གནས། །ལང་ཚོས་འཕྱར་བ་དེ་དག་གིས། །རང་སངས་རྒྱས་ནི་བྱོན་པ་མཐོང་། །ཏཾ་པྲབྷཱ་ཛཱ་ཏ་བི་དྭེ་ཥཤ་མ་ས་ཡ་མ་ནིནྡ་ཀཱ། །ཛིཊྛ་སུ་པུནྡ་ར་ཀོ་ནཱ་མ་དྷཱ་དེ་མཐོང་རྣམ་པར་སྣང་བ་སྐྱེས། །ཞི་དུལ་སྨོད་བྱེད་ཕུ་བོ་ནི། །སུན་ད་ར་ཀ་ཞེས་པ་ཡིས། །འཛུམ་ཀྲྀཥྚ་བཱ་ཙ་ས་སྨི་ཏ། །ཨ་ཡཾ་ཙི་བ་ར་པཱ་ཊཾ་ཀཔཱ་ནེ་ན་ག་ཏ་ཛཱི་བི་བཅས་སྤུན་ཟླ་རྣམས་ལ་སྨྲས། །བཏུང་བས་ཉེས་@པས་དགེ་སློང་ནི། །ཆོས་གོས་ལྷུང་བཟེད་རྟགས་ཅན་—།
