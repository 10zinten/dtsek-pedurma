བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀི་་བྷགྷ་བྷཱ་གྱུ། །ཁ་ར་མ་རུ་པ་རི་ཏཱ་པེ་ཀེ་ཙི་དུཏྠལྟ་པུ་ཎྱས་ཧ་ཛ་བ་ཉིད་འཛིན་ཅིང་། །བསོད་ནམས་རྒྱས་པ་ཁ་ཅིག་མྱ་ངམ་དྲག་པོས་ཡོངས་གདུང་ལ། །ལྷན་སྐྱེས་ཆུ་ས་ལི་ལ་སཱ་རིཙླི་བྲ་ཀྲྀཏྟཾ་ཏྱཱ་ཛནྟི། །ཤ་བཙྩཾ་པུ་ཎྱ་མི་ཏྲ་སྱ་སཱུ་ནུཀྲྀ་ཧ་བ་ཏེ་ཡི་ཚོགས་ཀྱི་མི་བཟད་སྲེད་པ་གཏོང་བར་བྱེད། །མཉན་ཡོད་དུ་ནི་ཁྱིམ་གྱི་བདག། བསོད་ནམས་བཤེས་ར་བྷུ་ཏྲ། །ཡ་ཤོ་མི་ཏྲ་ཨི་ཏི་ཥཱ་ཏ་པྲ་ཏི་མཱ་ནཾ་ཡ་ཤ་སཱི་ནཾ། །ཏ་སྱ་གཉེན་དག་གི་བུ། །གྲགས་པའི་བཤེས་གཉེན་ཞེས་བྱ་བ། །གྲགས་པ་ཅན་གྱི་ཚད་གྱུར་བྱུང༌། །བདུད་ཛརྨ་ཀླུ་ཎེ་པུ་ཙྪིབེ་ཡུ་ཥ་ཀི་ར་ཎ་ཏཱི་ཥ། །ཨ་བྲྀཏྲི་པྲ་བྷ་བཱ་བཱ་བྷིཤ་རྩིའི་ཟེར་གྱི་འོད་ཟེར་ཅན། །དེ་སྐྱེས་ཚེ་ན་བསོད་ནམས་ཀྱི། །ཆུ་ཡི་ཨེཆར་མེད་ལས་སྐྱེས་པ། །མི་ཤཱ་མ་བི་ཥ་མཱ་བི་བཏ། །པྲ་ས་བི་བ་བ་ཤཱི་ག་ཏཱ་བྷ་བ་བྷོ་ག་པ་རཱརླུ་ཁ། །བཟད་རྒུད་པ་ཞི་བར་གྱུར། །གཞོན་ནུ་ཉིད་ལ་དབང་གྱུར་ཅིང་། །སྲིད་པའི་ལོངས་སྤྱོད་ལས་ཕྱིར་ཕྱོགས། །ཛི་ནཾ་ཛེ་ན་བ་ནཱ་སཱི་ནཾ་སི་ཥ་བེ་ཀུ་ཤ་ལོ་རྱ་ཏ། །དྷརྨ་དེ་ཤ་ན་ཡཱ་ཤyདགེ་ལ་བརྩོན་པ་དེ་སོང་ནས། །རྒྱལ་བ་རྒྱལ་བྱེད་ཚལ་བཞུགས་བསྟེན། །སྟོན་པས་ཆོས་ནི་བསྟན་པ་ཡིས། །པྲ་བྲ་ཛུརྒཏྲ་དཾ་ཤི་ཏ། །ཏུ་མྱ་ཀཱཉྩ་ན་བཱ་ཥཱ་ཎསོ་བྷཱུཏུཀྟ་སྲི་རབ་བྱུང་དགྲ་བཅོམ་གནས་ལ་བརྟེན། །དེ་ནི་གསེར་དང་རྡོ་མཚུངས་ཤིང༌། །དགའ་དང་མི་དགའ་ཡཱ་བྲི་ཡ། །དཾབྷཱ་ཧཱུཾ་ས་ར་སཾ་ཏ་སྱ་དྲུ་ཏ་སྤ་ཊི་ཀ་ནིརྨ་ལཾ། །ཤུ་ཤྭ་བ་ཙཱ་རི་ཡེ་བཏང་བར་གྱུར། །དེ་ཡི་མཆེ་བ་དག་ལས་ཆུ། །རོ་ལྡན་དྲི་མེད་ཤེལ་གྱི་ནི། །ཁུ་བ་ཟགས་པ་གང་ཞིག་ནཱ་སོ་བཱི་ཏསྔཋས་དཱ་བྷ་བཏ། །ཀ་དཱ་ཙིདྡྷིཀྵ་པསརྦ་བྷ་ག་བནྟཾ་ཀུ་གིས༑། དེ་ནི་རྟག་ཏུ་སྲེད་བྲལ་གྱུར། །ནམ་ཞིག་དགེ་སློང་ཐམས་ཅད་ཀྱིས། །དགེ་མཚན་ལས་ནི་ཏུ་ཧ་ལཱཏ། །ཀྲྀཥཱ་བྱུ་པ་ར་མེ་ཏ་སྱ་བ་པྲཙྪཔུ་ཎྱ་ཀཱ་ར་ཎཾ། །སོ་བྲ་བཱིཏྟཱ་བཅོམ་ལྡན་ལ། །དེ་ཡི་སྲེད་@པ་བྲལ་བ་ཡི། །བསོད་ནམས་རྒྱུ་ནི་རབ་ཏུ་དྲིས། །དེས་གསུངས་ཀཱ་ཤིའི་ཤུ་པཱ་ཏྱཱ་སྱ་ཤཱyཀཱ་ཤི་པུ་རེ་པུ་རཱ། །ཨ་བྷཱུཏྲ་ཛི་ཏཐེཊྛི་སུ་ཏསུ་ན་ད་ར་ཀཱ་གྲོང་དུ་སྔོན། །མཛེས་པ་ཞེས་བྱ་ཚོང་དཔོན་བུ། །སྟོན་པ་འོད་སྲུངས་ཞེས་པ་ཡི། །རབ་ཏུ་བྱུང་བ་ཉིད་—།
