ཎཤ་ཤཱི་སྦ་བི་རཿས་རཱ་ཛུ་བྷོལཱུ་མིཙྪ་ཏི། ་། ཙཱ་བ་རོདྲིསྒ་གཱ་ཏྲ་སྱ་བ་བརྟན་རི་བོང་ཅན་ཉམས་བཞིན། །དེ་ནི་རྒྱལ་སྲིད་སྤྱོད་པར་འདོད། །ལུས་ནི་ཆོས་གོས་ཀྱིས་སྐྱོ་ཞིང་། །ར་བསྨཱ་བྷི་ལཱ་ཏི་ཎ། །ཛཱ་ཏཱ་སྱ་མུཎྜི་ཏེ་མཱུདྡྷི་རཏྭཾ་ཀ་མུ་ཀུ་ཊ་གོས་མཆོག་མངོན་པར་འདོད་ལྡན་འདི། །མགོ་ཟླུམ་ལ་ནི་རིན་ཆེན་གྱིས། །མཚན་པའི་ཅོད་པན་འདོད་སྤྲོ་ཧཱ། །རཏྭཱ་ཧརྩེ་ཥཱ་ན་བ་ཏཱ་སཉྙོ་ག་བི་བྷ་བོདྷ་བ། །ཏྱཱཀྟ་པར་གྱུར། །རིན་ཆེན་ཁང་པར་གསར་པ་ཡི། །ལོངས་སྤྱོད་འབྱོར་བས་རྣམ་བསྐྲུན་པའི། །རྩེ་བི་ལཱ་སཾ་སཱ་ཡཱ་སཾ་པ་ན་བཱ་པཾ་ས་ཧེ་ཏ་། །མྲ་དུ་ཤ་་ཡ་ན་སུ་དགའ་བཏང་ནས་ནགས་ཀྱི་གནས། །ངལ་དུབ་ལྡན་པ་སུ་ཡིས་བཟོད། །རི་དྭགས་རྨིག་པས་རྩུབ་ཅིང་ཁཱནྟཱ་ཡེ་ཀ་ཐཾ་ཤེ་ར་ཏེ་ཏེ་ཧ་རི་ཎ་ཁུ་རཁ་རོཀྐཎྜ་ཀཱ་སུ་སྦྲ་ལཱི་ཥཱ། །ཚེར་མ་བལྟམས་པའི་ཐང་ཁ་ལ། །གང་ཞིག་འཇམ་པའི་མལ་སྟན་བདེར་འོས་དེ་དག་ཇི་ལྟར་ཉལ། །མ་དྷུ་བི་དྷུ་ཀ་ར་ཤྲཱི་ཏཾ་ཡནྣི་པཱི་ཏཾ་ཀ་ཐནྟེ་བ་ན་ཛ་ག་ཛ་མ་དོཎྞཾ་ཏིཀྟ་མ་གང་གིས་ཟླ་ཟེར་ལྟར་བསིལ་ཆང་འཐུངས་དེ་དག་ནགས་སྐྱེས་ཀྱི། །གླང་པོའི་ཆང་གིས་དྲོ་ཞིང་ཁ་བའི་ཆུ་ནི་མྦྷཔི་བནྟི། །ཨ་དྷུ་ནི་བ་ཏ་བ་སནྣ་པྲ་བ་ས་བི་ཥ་མ་སྦི་ཏ། །ཨཱ་རྱམྨ་ཏཾ་ཇི་ལྟར་འཐུང་། །ཁྱོད་ལ་མི་བཟང་བགྲོད་པ་ནི། །ད་ལྟ་ཉིད་དུ་ཉེ་བར་གནས། །ལུགས་ཤེས་རྣམས་ན་ཡ་བི་དཱ་མ་རཱ་ཛ་པུ་ཏྲ་ནི་པཱ་ཏ་ནཾ། །ཏསྨཱ་ད་ནཱ་ག་ཏོ་རཱ་ཛཱ་པུཏྤ་བ་སྟ་ཀྱི་དང་པོའི་གྲོས། །རྒྱལ་སྲས་ངེས་པར་ལྟུང་བྱེད་པའོ། །དེ་སླད་རྒྱལ་པོ་མ་འོངས་པའི། །སྔོན་དུ་ཇོ་བོ་བ་པྲ་བྷོ། །དཱི་པཾ་ཧནྟི་པ་ཏཾ་གོ་ཧི་ན་དབྷེལྕེ་ཏ་ས་མཱ་བ་ཏན། །ཏ་ཡོ་རི་ཏི་ཁྱོད་ཀྱིས་གསོད། །ཕྱེ་མ་ལེབ་ནི་འབབ་པ་དག། གལ་ཏེ་མ་བསྲེགས་མར་མེ་འཇོམས། །ཞེས་པ་དེ་དག་གི་རཱ་ཀྱི་པྲ་མ་བྷུདྡྷཱ་པ་ཏི་རཱ་ཀུ་ལ། །ཁ་ལ་མེ་བྷིཀ་ལུ་ཥ་ཏྭཾ་ནཱི་ཏཾ་ཀ་ཏྱཱ་ན་ཚིག་གིས་ནི། །ས་བདག་མྱུར་དུ་འཁྲུགས་པར་གྱུར། །མི་བསྲུན་སྤྲིན་གྱིས་སུ་ཡི་ཡིད། །དྲི་མ་ལྡན་པར་མཱ་ན་སཾ། །ས་ཏོ་བ་བྷཱ་ཥ་སཱ་ཤི་ཀཀྲ་ཀ་ཙ་ཀླུ་ར་ཏྭཾ་ག་ཏ། །བཱ་དྷབྱེད་མི་འགྱུར། །དོགས་བཅས་སོག་ལེ་བཞིན་དུ་ནི། །རྩུབ་མོར་གྱུར་དེས་དེར་སྨྲས་པ། །བདག་དང་།
