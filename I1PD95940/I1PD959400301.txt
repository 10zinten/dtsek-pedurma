ལྗི་ནི་ཡ་མ་ད་རཱཏ། །ཏ་ཏས་རཏྣ་བྷ་བ་ནེ་བ་རཱ་བྷ་ར་ཎ་བྷུ་ཥི་ཏ། །པས་ངེས་པ་བརྟན་པོ་བྱས། །དེ་ནས་རིན་ཅེན་ཁང་པར་དེ། །མཆོག་གི་རྒྱན་གྱིས་རབ་བརྒྱན་ཅིང༌། །ཧེ་མ་ཧརྨ་ས་མཱ་རཱུ་ཌྷ་དི་དེ་ཤ་དི་ན་ཙནྡཾ་ཀཾ། །ཏ་ཏཔཱ་ཎི་ཀྲྀ་ཏཿཀཱ་མཱི་གསེར་གྱི་བ་གམ་ཁྲི་སྟན་ལ། །འཛེགས་ནས་ཉིན་པར་ཟླ་འོད་བྱས། །དེ་ནས་འདོད་ལྡན་གང་གང་གིས། །ཡོ་ཡཙཱ་ས་མུ་པཱ་ཡ་ཡོ། །ཏ་སྱ་ཏ་སྭཱ་བྷ་བཏྟ་སྱཔྲ་བྷ་བེ་ནོ་ཛ་སཿཀཱ་ཡ། །དྲབྷུ་ཟོང་བྱིན་དེ་ཅན་ཉེ་བར་སོང་། །དེ་དང་དེ་ཡི་གཟི་བརྗིད་ནི། །དེ་ཡི་མཐུ་ཡིས་རབ་ཏུ་ཉམས། །བལྟ་བ་མེ་བ་ན་ཤེ་ཀུསྟེ་ཀི་གླུ་ནསྤྲཏྟཱ་མཱ་ཀུ་ལཱ། །བྷུ་ཛཾ་ག་བྷོ་ག་སཾ་རུདྡྷཾ་ཏུཾ། ཉིད་དུའང་མི་ནུས་ན། །རེག་པས་འཁྲུགས་®པ་སླར་ཡང་ཅི། །དེ་ནི་ཙན་དན་འཁྲི་ཤིང་དག། ལག་ཙནྡ་ན་ལ་ཏ་མི་བ། །ཏ་ཏས་པུནྡ་རཱི་བྷེ་ཛེ་ཡོ་བ་ན་སུ་བི་ཡོ་བ་ནཾ། །འགྲོས་ལོངས་སྤྱོད་བགགས་པ་བཞིན། །དེ་ནས་མཛེས་མ་དེ་ཡིས་ནི། །ནུ་མ་རྒྱས་པའི་ཁུར་®དག་གིས། །གུ་རུ་ནསྟ་ན་བྷཱ་རེ་ཎ་མ་དྷ་བྷཾ་ག་བྷ་ཡ་པྲ་དཾ། །སྨ་ར་སཾ་བྷོ་ག་ར་ཧི་ཏཾ་རྐེད་འཁྱོག་འཇིགས་པ་རབ་སྟེར་བ། །ལང་ཚོ་ཡི་ཡང་ལང་ཚོ་བསྟེན། །དེ་ཡི་གཟུགས་ནི་རྨད་བྱུང་དེ། །ཏརྟ་སཱུ་རཱུ་པ་མརྒྱུ་ཏཾ། །ཤྭ་བྷ་ཧེ་མ་ལ་ཏཱ་པུཋ་མི་བ་ནིཧྥ་ལ་འདོད་ཆགས་རྫོགས་ཤིང་ལོངས་སྤྱོད་བྲལ། །གཡང་སའི་གསེར་གྱི་འཁྲི་ཤིང་གི༄༅། །མེ་ཏོག་བཞིན་དུ་ཏྭཾ་ཡ་ཡོ། །ཀོ་ཏུ་ཀཱ་ཤཱ་བི་ནོ་དཱ་ལ་ནཱ་ནཱ་དེ་ཤཱནྟ་རཱ་ག་ཏི། །ཨ་ཀཱ་འབྲས་མེད་གྱུར། །ལྟད་མོའི་བསམ་པས་རྩེད་མོའི་སླད། །ཐ་དད་ཡུལ་གཞན་ནས་འོངས་པའི། །རི་མོ་རཙི་ཏྲ་ཀ་རེནྡྷ་པ་ལ་པྲ་ཏི་མ་ཧྲ་ཧེ། །བྷི་དྷ་ཡ་ཙི་ཏྲ་ལི་ཁི་ཏཱན་པཱ་ཀྲ་མཁན་གྱིས་《ས་སྐྱོང་གི། །གཟུགས་ནི་དེ་ཡི་ཁང་པར་བྲིས། །མི་དབང་རྣམས་ནི་རི་མོ་ལ། །བྲིས་ཤིང་མེ་ཎ་ན་རེ་ཤཱ་རཱན། །ད་དན་བིམི་སཱ་ར་སྱ་དྲིཤྩ་ར་ཏི་པ་ཏེ་རི་བལྟས་པས་རིམ་གྱིས་དེ། །གཟུགས་ཅན་སྙིང་པོའི་གཟུགས་དག་ནི། །དགའ་མའི་བདག་པོ་བཞིན་དུ་བ༑། ཏ་མཱ་ལོ་ཀྱི་བ་ས་ཧ་ས་ས་མུདྡྷཱ་ཏ་མ་ནོ་བྷ་བཱ། །ས་ཡེ་ན་ལི་ཁི་མཐོང༌། །དེ་མཐོང་ཉིད་ན་འཕྲལ་ལ་ནི། །འདོད་ཆགས་ཡང་དག་ལངས་པར་གྱུར། །གང་ཞིག་དེ་ནི་།
