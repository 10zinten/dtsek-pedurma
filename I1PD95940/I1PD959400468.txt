ག་ཏཱ་ཀཱ་ལེ་སུ་ཥཱ་བེ་དི་བྱ་ལཀྵ་ཎཾ། །ཨུ་ད་ཀཱ་ཊྛས་བ་ལོཞབྷཱུཏ་སཾ་ཛཱ་ཏས་ལི་ལྟ་བུའི་བུ། །མཚན་མཆོག་ལྡན་པ་བཙས་པར་གྱུར། །ཆུ་ནང་དུ་སྐྱེས་བྱིས་པ་དེ། །ཆུ་སྐྱེས་ཞེས་ནི་བྱ་ལཱནྟ་རེ། །བཛྲ་མཱ་ཎཔི་ཏུསྡུ་ལོ་ཡོ་བ་རཱ་ཛུ་མ་ནོ་ར་ཐི། །ཤ་ཏཱ་ནི་པ་མཱ་བར་གྱུར། །ཡབ་ཀྱི་རྒྱལ་ཚབ་བཞེད་པ་དང༌། །མཚུངས་པ་ཉིད་དུ་ཆེ་བར་སྐྱེས། །དེ་ནི་སྐྱེས་པའི་ཉིན་ཏྭཱ་ནཾ་ཏ་སྱ་ཛཆ་་དི་ནཾ་ས་མཾ། །ཨ་བཱ་པུསྟུ་མྱ་རླུ་པ་ཎཱ་པུ་ཏྟཱ་ཎཾ་པར་ལ། །བློན་པོ་ལྔ་བརྒྱས་མཚུངས་པར་ནི། །གཟུགས་ནི་མཚུངས་པའི་བུ་རྣམས་དག། ལྔ་བརྒྱ་རབ་ཤ་ཏ་པཉྩ་ཀཾ། །ནི་ཛཾ་ཛཱ་ཏི་སྨ་རཿསྨྲ་ཏྭཱ་པཱབྲྀཏྟཾ་ཤི་ཤུཤ་ནི། །ཨ་ཙིནྟ་ཏུ་ཐོབ་པར་གྱུར། །རང་གི་སྐྱེ་དྲན་བྱིས་པ་དེས། །རིམ་གྱིས་སྔོན་བྱུང་དྲན་གྱུར་ནས། །བདག་ཉིད་ཡཏ་པཱལྟ་ཀཱ་ལཾ་ཧི་ཏཾ་སུ་ཀྲྀ་ཏ་མཱཏྨ་ན། །ཥཥཱ་བནཱ་ཎི་ཀྲྀ་ཏྭཱ་ཏུ་ཡོ་བ་ལ་ཕན་ལེགས་བྱས་ཀྱི། །དུས་ལ་བབ་པར་ཡང་དག་བསམས། །ལོ་ནི་དྲུག་ཅུ་ཐམ་པར་སྔོན། །བདག་རཱ་ཛུ་མ་ཧཱ་པུ་རཱ། །ཨ་བྷ་ཐཾ་ཀྲྀཙཱ་སཾ་ཏཔཕྱི་རཾ་ན་ར་ཀ་སཾ་ཀ་ཊེ། །གིས་རྒྱལ་ཚབ་བྱས་པ་ན། །མི་བཟད་དམྱལ་བར་ཡུན་རིང་དུ། །ཉམ་ངའི་སྡུག་བསྔལ་གདུང་བ་ཐོབ། །ཛརྨ་སྨིནྣ་པི་བུ་ནརྨོ་བ་རཱ་ཛུ་མུ་པ་སྦི་ཏཾ། །སརྦ་ཐཱ་བཱརྒྱ་མ་ཎོ་པི་ན་སླད་ནས་སྐྱེ་བ་འདི་ལ་ཡང་། །རྒྱལ་ཚབ་དག་ནི་ཉེ་བར་གནས། །རྣམ་པ་ཀུན་ཏུ་གསོལ་བཏབ་ཀྱང་། །ཀ་རི་ཎྱ་མི་བཱ་ཏ་ཀཾ། །ཨི་ཏི་ས ི་ས་ཙི་རཾ་རཱ་ཛུ་བྷོ་ག་པ་རཱརླུ་བདག་ནི་སྡིག་པ་མི་བྱེད་དོ། །ཞེས་པ་དེ་ཡིས་ཡུན་རིང་བསམས། །རྒྱལ་སྲིད་ལོངས་སྤྱོད་ལས་ཕྱིར་ཁ༑། པི་ཏུ་རུདྡེ་ག་ཛན་ནི་མ་གྲ་ཧཱི་ཏ་པཾ་གུ་མཱུ་ཀ་ཏྭཾ། །སརྦ་ལ་ཀླུ་ཕྱོགས། །ཕ་ནི་ཡིད་འབྱུང་སྐྱེད་བྱེད་པའི། །ལྐུགས་པ་འཕྱེ་བོ་བཟུང་བར་གྱུར། །མཚན་ཉིད་ཐམས་ཎ་ཡུཀྟ་པི་རཱ་ཛ་བྲྀཏྟེ་ར་བྷཱ་ཛ་ནཾ། །ས་མཱུ་ཀ་པཾ་གུ་ནཱ་མ་བྷཱུད་ཅད་ལྡན་མོད་ཀྱང༌། །རྒྱལ་པོའི་ལུགས་ཀྱི་སྣོད་མ་གྱུར། །གཉེན་རྣམས་སྡུག་བསྔལ་སྐྱེད་བྱེད་དེ། །བནྡྷ་ནཾ་དུཁ་བཌྷ་ན། །པཱསྟེ་ཥཱ་མནྟི་སུ་ཏེ་་ཤསྔ་ཤཱསྔ་བ་ལོ་ལྐུགས་པའི་འཕྱེ་བོ་ཞེས་པར་གྱུར། །མཚོན་དང་བསྟན་བཅོས་སྟོབས་རྒྱས་པ། །བློན་པོའི་བུ་རྣམས་ཀྱིས་—།
