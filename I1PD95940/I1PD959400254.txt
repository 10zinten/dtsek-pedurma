པ། སྭཱ་སྟི་རེ་ནི་རམྥ་ར། །ཤུ་ཤོཙ་ཧཱུྃ་ཛ་ནི་རཱ་ཐུ་ཛ་ལེ་སཱ་མ་ལི་ནས་ནཱི། །ཏ་དུ་གོས་དང་བྲལ་བ་དེས། །མིག་སྨན་དང་བཅས་མཆི་མའི་ཆུས། །ནུ་མ་རྣག་ཅིང་མྱ་ངན་བྱས། །དེ་སྨིན་ཀླུ་ཎེ་མུ་ཁཱ་ཤ་ཀྵ་མཾ་ས་མུཏྠཾ་ཛུ་ཛཾ་བུ་ཀེ། །ཡཱ་ཏེ་ཛ་ལོཥྛ་ཏཾ་མཎྞཾ་ཏ་ད་ཡི་དུས་ན་ཝ་སྐྱེས་ཤིག། ཁ་ཡི་ཤ་ནི་རབ་བཏང་ནས། །ཆུ་ལ་འཕར་བའི་ཉ་ལ་སོང་། །དེ་ནི་མཁའ་ཧཱནདྲི་ཧཾ་ག་མ། །མཉྩ་ནི་ལགི་ས་ཧ་ས་ཁ་གེ་ན་པི་སི་ཏེ་ཧཱུྃ་ཏེ། །ས་བ་བྷུ་ལ་འགྲོ་བས་ཁྱེར། །ཉ་ནི་འཕྲལ་ལ་གཏིང་དུ་སོང་། །ཤ་ནི་བྱ་ཡིས་ཕྲོགས་པའི་ཚེ། །གཉིས་ཀ་ེཉམས་བོ་བྷ་ཡ་ཤྲཱི་སཱཙྪིནྡྲ་ནི་ལ་ལོ་ཙ་ན། ཞེ། ཏ་སྩལྟ་བཱིགྱུ་དུ༔ཁེ་པི་མུ་ཁེ་པས་དེ་ཡིས་ནི། །བསམས་པ་མིག་ནི་གཡོ་མེད་གྱུར། །དེ་མཐོང་དེ་ཡིས་བཞིན་ལ་ནི། །སྡུག་ཀྱང་སྨི་ཏ་མ་དྲྀ་ཤྱ་ཏ། །ཧཱ་སས་རེ་ཏུ་སྐྲ་ཡི་ཏེ་ཥཱ་སཱུརྨོ་བ་ར་ཛཱ་ཡཱ་ཏེ། །འཛུམ་པ་མཐོང་བར་གྱུར། །གཞན་དག་གི་ནི་འཁྲུལ་པ་ལ། །འཆིར་ཉེ་རྣམས་ཀྱང་དགོད་པ་ཡིན། །ས་ཏྭཾ་བི་ལཀླུ་ཀུ་པི་ཏཿཔྲ་བ་ཙཱ་ནུ་ཙི་ཏ་སཱི་ཏཾ། །ཨ་ཧོ་ཧ་ས་སི་མཱ་ལོ་ཀེ་སྐྱེངས་པས་ཁྲོས་པ་དེ་ཡིས་ནི། །མི་འོས་འཛུམ་པ་དེ་ལ་སྨྲས། །ཀྱེ་མ་འཇིག་རྟེན་གད་མོ་ཡི། །ཧཱ་སྭཱ་ཡ་ཏ་ན་ཏྭཾ་ག་ཏྭཾ། །ཛྲ་པ་ཏི་ཏྱཱཀྟ་ག་སྡུནྡྷ་ཏྱཱཀྐཱནྣཱ་ཙོ་ར་མཱ་གནས་གྱུར་ཁྱོད་ཉིད་ཀྱི་བདག་ལ་དགོད། །མི་བདག་བཏང་ནས་ལོང་བར་ཕྱིན། །ལོང་བ་བཏང་ནས་ཚོམ་ཤི་ཏཱ། །ཏ་བཱ་ཧ་མུ་བྷ་ཡ་བྷཱཝཾ་ཏྲེ་བྷཏྟཱ་ཡཱསྨི་ཏཱསྤ་དཾ། །།ཨཱ༑་རྐུན་བསྟེན། །གསུམ་ཆར་ཉམས་པ་ཁྱོད་ཀྱིས་ནི། །འཛུམ་པའི་གནས་ནི་གཉིས་ཉམ་ང་། །ཡོངས་སུ་བཱ་པ་རི་ཧཱ་སོ་ཡཾ་ཏཾ་ཡུཀྵ་ཀ་རོ་མི་ཏེ། །ཁ་ལཱསྟེ་བི་ཥ་མ་སྠཱ་ནཾ་བགད་པ་འདི་ནི་ཆོགི། བདག་ནི་རིགས་པས་ཁྱོད་ཕན་བསྒྲུབ། །མི་བཟད་གནས་ལ་གང་ཞིག་ནི། །ཡེ་བི་ཌཝ་ན་པཎྜི་ཏཱ། །ཨི་ཏུཀྟ་ན་ག་རཾ་ག་ཏཱ་ས་ནཱཾ་པཱ་ཡ་ནྱ་བེ་ད་ཡ་ཏཱ། །ཅོ་འདྲིའི་མཁས་པ་དེ་མི་བསྲུན། །འདི་བརྗོད་གྲོང་ཁྱེར་སོང་ནས་ནི། །ཀླུང་གི་ངོགས་ན་སོ་སུམ་བཱ། །སོ་ཤུམཱ་སྟེ་ན་དཱི་ཏཱི་རེ་ཏ་སོ་ཡུཀྐི་ཏི་སརྨ་ཏི། ་། ཨ་ཐ་ནི་ནཱ་ཡ་ཏྭཾ། དཀའ་ཐུབ་རིགས་པས་ཅིགནས་སོ་ཞེས། །བློ་གྲོས་བཟང་མོས་རྒྱལ་ལ་སྨྲས། །དེ་ནས་གོས་དང་རྒྱན་།
