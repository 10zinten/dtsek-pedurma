སཛྫ་བྲྀ་ད་ཡཱ་ད་མྲ་ཏ་སྭ་བྷཱ་ཐཾ། །པུ་རེ་པུ་ར་རཱ་ཛ་གྲོ་ཧེ་བྷ་ག་མཚོ་སྙིང་ལས་བདུད་རྩིའི་རང་བཞིན་རབ་ཏུ་གཏོང་བ་མིན། །སྔོན་ཚེ་གྲོང་ཁྱེར་རྒྱལ་པོའི་ཁབ། །ཀ་བཱན་བེ་ཅུ་ཀ་ན་ནེ། །ཀ་ལནྡ་ཀ་ནི་བཱ་ས་ཁྱེ་བི་ཛ་ཧཱ་ར་མ་ནོ་ཧ་རེ། །ལན་ད་ཀའི་གནས་ཞེས་པ། །འོད་མའི་ཚལ་ནི་ཡིད་འཕྲོག་ན། །བཅོམ་ལྡན་འདས་ནི་རྣམ་པར་བཞུགས། །ཏ་ཏྲ་བི་ཏྲ་སི་ཏཱ་ནེ་ཀི་ཤ་ཅུ་ནིཙླི་ཤ་བཱནྟ་བ། །ཨ་ཛཱ་ཏཱ་ཤ་ཏུ་ནཱ་མ་དེར་ནི་རྒྱལ་པོ་དགྲ་མང་རྣམས། །རྣམ་པར་སྐྲག་བྱེད་རལ་གྲིའི་གཉེན། །གཟུགས་ཅན་སྙིང་པོ་དག་གི་བྷཱུཏ་བིལླི་སཱ་ར་སུ་ཏོ་ཚིགས་པ། །ཤཱཀྱ་ཐཾ་ཤསཱ་བྲྀཏྟ་སྱ་དེ་བ་དཏྟ་བྷི་ཏོཞབྷ་བ་བུ། །མ་སྐྱེས་དགྲ་ཞེས་བྱ་བར་གྱུར། །ལྷས་བྱིན་ཞེས་པ་ཤཱཀྱའི་རིགས། །དེ་ཡི་སྙིང་གི་གྲོགས་པོར་ཏ༑། ཀླུ་དྲ་མནྡེ་ཎ་ཡ་སྭཱ་སཱིཏ་ས་བེ་ཏཱ་ལ་ཨི་བོཥ་ཊ། །ས་ཀ་དཱ་ཙིཏ་སུ་གྱུར། །གང་གིས་ངན་པའི་སྔགས་ཀྱིས་དེ། །རོ་ལངས་བཞིན་དུ་དྲག་པོར་གྱུར། །དེ་ཡིས་ནམ་ཞིག་ཁཱ་སཱི་ནཾ་རཧཔྲ་ཧ་མ་ཧཱི་པ་ཏི། །ན་རཱ་ཛན་ཕ་ལི་ཏོཉརྱ་པི་མ་མཱ་པི་ཏཱད་ས་ཡི་བདག། མདུན་སར་བདེ་བར་འཁོད་ཨེལ་སྨྲས། །རྒྱལ་པོ་བདག་གིས་ཁྱེད་བསྟེན་པའི། །འབྲས་ནི་ས་མཱ་ཐཱ་ཡ། །མི་ཐོ་པཱ་ཙ་ར་ར་ཧི་ཏསུ་ཁ་ནིརླནྡ་ཏནྡྲ་ཡོ། །མི་ད་དུང་གྲུབ་པ་མེད། །བདེ་བར་ཡིད་གཅུགས་གྲོགས་དག་ནི། །ཕན་ཚུན་ཡིད་ལ་རེ་བ་སྐོང་། །རྫུན་ཐོ་མ་ནོ་ར་ཐ་ཏཱ་ཎརྨི་ཏྲ་ཤདཿཔྲ་བནྟ་ཏེ། །ཡ་ཨེ་ཥ་ཤཱཀྱ་ཤྲ་མ་ཎཿགྱིས་རིམ་གྲོ་བྲལ་བ་ལ། །གྲོགས་ཀྱི་སྒྲ་ནི་རབ་ཏུ་འཇུག། གང་འདི་འོད་མའི་ནགས་ཚལ་ན། །སུ་ཁ་བེ་ཎུ་བ་ནེསྦི་ཏ། །ཏཾ་ཧ་ཏྭཱ་པཱལྟ་མིཙྪཱ་མི་ཏཏྠ་དརྗེ་བ་བ་ན་དི་ཏཾ། །ཤཱཀྱའི་དགེ་སྦྱོང་བདེར་གནས་པ། །དེ་བསད་ལྷ་ཡིས་ཕྱག་བྱས་པ། །དེ་ཡི་གནས་ནི་ཐོབ་པར་འདོད། །ཀཱི་ཡ་ཏེ་ན་ཡ་ཡཱ་ཤ་ཏུརླ་བྱཱ་ཏེ་ན་ཡ་ཡཱ་ཡ་ཤ། །།བཌྷ་ཏེ་ན་ཡ་ཡཱ་མཱ་ནགང་གིས་དགྲ་བོ་མི་ཆོམས་དང་། །གང་གིས་གྲགས་པ་མི་འཐོབ་དང་། །གང་གིས་ཁེངས་པ་མི་འཕེལ་ཀིནྟ་ཡཱ་མི་ཏྲ་སམྤ་ད། །མ་ཧཱ་དྷ་ནཱ་བྷི་དྷ་ནེ་ན་ཏེཏ་སོ་རེ་ཎ་ནི་བ། །གྲོགས་ཀྱི་ཕུན་ཚོགས་དེ་ཡིས་ཅི། །ནོར་ཅན་ཞེས་པའི་གྲོང་ཁྱེར་བས། །དེ་ནི་ཟས་ལ་མགྲོན་།
