ར་པ་རི་ཤཱནྟཾ་ཏ་ཏདྲ་བ་ཧ་ཎཾ་མ་ཧཏ། །ཨ་བྷ་ཛཱ་ཏ་མ་ཧཱ་བཱ་ཏི་རི་གཟིངས་ཆེན་རིན་ཆེན་གྱི། །ཁུར་གྱིས་ཡོངས་སུ་ངལ་བ་དེ། །རླུང་ཆེན་དག་གིས་ཞིག་གྱུར་ཏེ། །ཚུལ་ཤཱརྨ་མི་བ་དུཙྪཱ་ཡེ། །ཏ་ཏཕ་ལ་ཧ་ཀཱ་བཱནྟཾ་ཀྲྀ་ཏ་ཛྷཱ༔པཱལྟ་ཇི་བི་ཏ། །ངན་རྣམས་ཀྱི་དབང་ཕྱུག་བཞིན། །དེ་ནས་ཤིང་དུམ་བཟུང་བ་ཡིས། །བྱས་ཤེས་ཀྱིས་ནི་འཚོ་བ་ཐོབ། །ཨ་ཀྲྀ་ཏ་ཛྙཱ་ནི་མཛྫནྟཾ་པྲཏྟེ་ན་ས་མ་ཏ་ར་ཡེཏ། །ཏཱ་རི་ཏཀྲྀ་པ་ཡཱ་རྒྱབ་ཀྱིས་བྱས་མི་ཤེས་དག་ནི། །བྱིངས་པར་གྱུར་པ་ཡང་དག་བསྒྲལ། །འཇིགས་རུང་ཆུ་སྲིན་འབྱུང་བྷཱ་ཏཱ་ས་གྷོ་ར་མ་ཀ་རཱ་ཀ་རཱཏ། ་། ཨ་ས་གླུ་ད་ཙ་ལེ་ཏ་སྱ་རུ་ཙི་རཾ་གནས་ལས། །བརྩེ་བས་བསྒྲལ་བའི་སྤུན་ཟླ་དེས། །དེ་ཡི་ཐུ་བོ་ཉིདག་ལ་ནི། །རིན་ཆེན་ཡིད་འོང་རཏྣ་སཉྩ་ཡོ། །ས་ཏ་སྱ་རཏྣ་ལོ་བྷེ་ན་དྭེ་ཥ་ཙ་བ་ཤཱི་ཀྲྀ་ཏ། །ས་མུ་བསགས་པ་མཐོང་། །དེ་ཡི་རིན་ཆེན་ལ་ཆགས་དང་། །སྡང་བའི་དབང་དུ་གྱུར་པ་དེས། །རྒྱ་མཚོའི་དྲ་ཏཱི་རེ་ཤཱནྟ་སྱ་བྷཏྲི་དྲོ་ཧ་མ་ཙིནྟ་ཡཏ། །ཏ་སྱ་ནི་དཱ་ནི་ལཱི་ན་སྱ་ཤསྡེ་འགྲམ་དུ་ངལ་བ་ཡི། །སྤྱན་ཡ་འཁུ་བ་བསམས་པར་གྱུར། །དེ་ནི་གཉིད་དུ་སོང་བ་ལ། །མིག་དག་ནོཏྟཱ་བྱ་ལོ་ཙ་ནེ། །གྲི་ཧཱི་ཏཱ་རཏྣ་ནི་ཙ་ཡཾ་ཀྲྀ་ཏ་བྷནས་ཡེ་ཡོ་ཛ་བཱཏ། །ཀྲུ་མཚོན་གྱིས་རབ་ཕྱུང་ནས། །རིན་པོ་ཆེ་ཡི་ཚོགས་བཟུང་སྟེ། །བྱས་མི་གཟོ་དེ་མགྱོགས་པར་སོང་། །སྒྲ་རེ་ཎཱནྟཱི་ཀྲྀ་ཏསྟེ་ན་རཱ་ཧུ་ཎེ་བ་དི་བ་ཀ་ར། །ལོ་ཀོ་པ་ཀཱ་ར་བི་ར་ཧཏ་དུགཅན་གྱིས་ནི་ཉིན་བྱེད་མིབཞིན། །གདུག་པས་ལོང་བར་བྱས་པ་དེས། །འཇིག་རྟེན་ཕན་པ་བྲལ་བ་ལས། །ཁི་ཏསོ་སྱ་ཙིནྟ་ཡཏ། །ཨ་དྷུ་ནཱཏྲི་པྲ་དཱ་ནེརྗེ་བྱནི་བྷཱུ་ཏེ་མ་ནོ་ར་ཐེ། །སྡུག་བསྔལ་ལྡན་པ་དེས་ཀྱང་བསམས། །ད་ནི་སློང་ལ་སྦྱིན་དོན་དུ། །ཡིད་ལ་རེ་བ་དོན་མེད་གྱུར། །ཀིམྨ་མཱནྟ་སྱ་བནྡྷེ་ཛཱི་བི་ཏེ་ན་པྲ་ཡོ་ཛ་ནཾ། །ཨ་པཱཥ་བི་དྷ་ཡ༔པཱ་ཎཱ་ལོང་བ་བདག་གི་འཚོ་བ་ནི། །འབྲས་མེད་དག་གིས་དགོས་པ་ཅི། །གལ་ཏེ་དུས་ལ་མ་བབ་པས། །སྲོག་ན་པྲ་ཡཱནྟི་ཡ་དིཀྵ་ཡཾ། །ཏ་ད་སཾ་ག་ཏ་པོ་ཡོ་གལླེ་ཤ་ཤཱནྟི་གྷ་ཡ་ཀྵ་མ། །ནི་འཇིག་པར་མི་འགྱུར་ན། །དེ་ལ་གྲོགས་མེད་དཀའ་ཐུབ་ཀྱི། །སྦྱོར་བ་ཉོན་མོངས་ཞི་འཇིག་བཟོད། །།
