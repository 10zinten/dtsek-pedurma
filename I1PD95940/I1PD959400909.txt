རརན་པྲ་ཡཱ་སྱ་ཏི་མ་ནུ་ཥཱ་ཏྭཾ། །ཤཱསྟུ་རུཏྟ་རཾ་སཾ་ཛྷ་སྱ་ཏ་ཏཀྵ་ཡི་ཏ་དུས་ཀྱིས་ནི། །མི་ཉིད་རབ་ཏུ་ཐོབ་པར་འགྱུར། །དེ་ནས་སྡིག་པ་ཟད་གྱུར་ཏེ། །ཉོན་མོངས་མེ་ཆེན་ཀིལླི་ཥ། ་། དཱིགྷཱ་ཀླུ་ཤ་ན་ལོདྲིསྨཱ༔ཤཱ་ས་ནེ་པྲ་བྲ་ཛི་ཥྛ་ཏི། །ཨུ་ཏྱཱ་ནེ་དག་གིས་ནི། །ཡིད་བྱུང་སྟོན་པ་ཨུད་ཏ་ར། །ཞེས་པའི་བསྟན་ལ་རབ་འབྱུང་འགྱུར། །དེ་ནི་ཀཱ་ཤིའི་ཀཱ་ཤི་རཱ་ཛ་སྱ་ཀཱནྟཱ་བྷིཔཱུ་ཛི་ཏསྟ་ཏ། །ཨཱིནྡྷ་ཀོ་པེ་ན་ཤ་རཱི་ར་ལ་པཱཙྪ་རྒྱལ་པོ་ཡི། །སྐྱེད་མོས་ཚལ་དུ་མཛེས་མས་མཆོད། །ཕྲག་དོག་ཁྲོ་བས་ལུས་དག་ནི། །ཆད་པ་དྲག་པོས་ཁྱུ་གྲ་ནི་གྲ་ཧཾ། །ཏ་ཏོ་སྱ་ཏཱི་པྲ་དུཁཱགྷཾ་བ་བ་པཱ་ནཾ་བྷ་བི་ཥྛ་ཏི། །ཨི་ཏི་གཅོད་པར་འགྱུར། །དེ་རྗེས་འདི་ཡི་སྡུག་བསྔལ་མེ། །མི་བཟད་རབ་ཏུ་ཉམས་པར་འགྱུར། །ཞེས་པ་ཤཱསྟུཏྤ་ཙཐུ་ཏྭཱ་སརྦ་བི་རཱ་གྱ་མ་ཡ་ཐུ། །ཨུངྒིབྷཱ་ནཱ་མ་ཐ་བྷ་བ་བྷ་ཡ་སྟོན་པའི་གསུང་ཐོས་ནས། །ཐམས་ཅད་ཆགས་བྲལ་ལྡན་པར་གྱུར། །དེ་ནས་དེར་ནི་སྟོན་པའི་གསུང་གི་གླེ་ཤ་ཙིན་ཏུ་རཱ་ཎཱ་ཤཱསྟུཔུཙྪེ་རཱ་མྲ་ཏ་བ་ས་རེནྣཱདྨ་ཡེ་སྟ་རང་བཞིན་བདུད་རྩིའི་རྒྱུན་རྫོགས་པས། །ཆགས་བྲལ་སྲིད་པའི་འཇིགས་པའི་ཉོན་མོངས་བསམ་པས་གཟིར་ཏྲ་ཏེ་ཏཱཾ། །སཀྵ་དྷྱ་དིཀུ་ཤ་ལ་པ་ལི་ལིཔྲ་བྱཱ་མཱ་ནཱ་ཤ་ཡཱ་ནཾ་ཏཱ་བབ་དེ་དག་རྣམས། །དགའ་བ་དང་ལྡན་དགེ་བའི་ཆུ་ཡིས་བསམ་པ་རབ་ཏུ་རྫོགས་གྱུར་ཅིང་། །རབ་ཞི་མེ་ཀཱ་པི་པྲ་ཤ་མཀུ་སུ་ནཱ་རཱ་མ་ཐཱ་ཛཾ་ཛ་གཱ་མ། །ཨི་ཏི་གླེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཱ་བོ་དྷི་ཏོག་དགའ་ཚལ་ལྡན་པ་རྣམས་ཀྱི་གདུང་བ་ག་ཤེད་སོང་། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་བི་དུ་རཱ་པ་དཱ་ནཾ་ཥཊ་སསྟ་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་བི་དུ་རའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་་ཏི་ཏ་མཉཔལླ་ཝ། །།།འདབ་སྟེ་བདུན་ཅུ་རྩ་དྲུག་པའོ། །དེཁ་ལོཥནེ་ཎ་སཏྠ་སྱ་གུ་ཎ་ལསྨཱ་ནི༔པྲ་ཛཱ་ཡ་ཏེ། །བྷ་བ་ཏུ་སྒྱུ་མི་བསྲུན་®རབ་ཏུ་རྒྱས་པ་ཡིས། །ཀུན་གྱི་ཡོན་ཏན་ཉམས་པར་བྱེད། །བདག་ཉིད་ཆེན་པོའི་་་།
