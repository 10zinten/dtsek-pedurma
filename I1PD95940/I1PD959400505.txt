ཏྱཱ་ཤི་ཁཎྜི་ནཾ། །ཨི་ཏུཏྟསྟེ་ན་སཱ་ནནྡསཱ་དེ་ཤཾ་ཤ་ན་ཀིཏྠ་ཎིཀ། ག་ཏྭཱ་ཏུ་དགོད་པར་བྱ། །ཞེས་པ་དེ་བརྗོད་ཀུན་དགའ་ལྡན། །ཚོང་པ་རིམ་གྱིས་རང་ཡུལ་དུ། །སོང་ནས་པྲ་ཛཱ་ནཾ་བི་ད་དྷེ་སྭ་ར་མཱ་ཤཱ་ས་ནཾ་པུ་ར། །།པྲ་བཱ་དེ་པྲ་སྲི་ཏ་ཏསྨི་སྔོན་དུ་སྐྱེ་རྒུ་རྣམས། །དལ་བུས་དབུགས་ནི་དབྱུང་བ་བསྒྲུབས། །གཏམ་དེ་རབ་ཏུ་གྲགས་པའི་ཚེ། །མཱ་ཏོ་དཎྜ་མུངྒ་རོ། །ཨ་ཏཱི་ཏ་བྷཱུ་པཱ་ལ་ག་མ་ཎ་ཏྲསྟོནཾ་པ་ཏི་མཱུ་ཙ་ཏུ། །སརྦ་བློན་པོ་དབྱུག་པ་ཐོ་བ་དག། འདས་པའི་ས་སྐྱོང་འོང་བ་ཡིས། །སྐྲག་སྟེ་མི་ཡི་བདག་ལ་སྨྲས། །ལེགས་ཏྲ་ཐུ་ཡ་ཏེ་དེ་བ་པྲ་བཱ་དསཱ་དྷུ་ནནི་ཏ། །བྲོདྷཔྲ་བྲ་ཛི་ཏོ་རཱ་ཛཱ་པ་རྣམས་ཀྱིས་སྨད་པའི་གཏམ། །རྒྱལ་པོ་རྒན་པོ་རབ་བྱུང་བ། །རྒྱལ་སྲིད་འབད་པས་དོན་གཉེར་ལྡན། །རཱ་ཛ་ནི་ཡནྣ་བཱ་ནི་ཏི། །ཏཱི་བྲ་བྲ་ཏ་པ་རི་ཀྱིཋསཾ་བྷོ་གཱ་བྷི་མུ་ཁཱ་ཞེས་པ་ལྷ་གཅིག་ཀུན་ཏུ་ཐོས། །བརྟུལ་ཞུགས་དྲག་པོས་ཡོངས་སུ་དུབ། །གུས་པས་ལོངས་སྤྱོད་ལ་ད་ར། །ལÉ་པྲ་བྲ་ཛ་ཡཱ་སཱཌྷ་ཏྱཱཀྟ་ས་པུ་ན་རེ་བྱཱ་ཏི། །རཱ་ཛསྣ་མངོན་ཕྱོགས། །རབ་བྱུང་ངོ་ཚ་ཐབས་ཅིག་ཏུ། །བཏང་ནས་དེ་སླར་འོང་བར་འགྱུར། །རྒྱལ་པོ་ས་ཀཱ་བི་རཱ་སྐྱ་ཛནྟི་ས་ཧ་སི་བ་ཡཏ། །ཏཉྩནྡྲ་བྱཱ་དྷི་ཀཾ་ཏེ་ཏཱཾ་བྲ་ཡཱ་ཏི་ཆགས་བྲལ་མ་སྨིན་པས། །འཕྲལ་ལ་ཉིད་དུ་གང་བཏང་བ། །དེ་རྣམས་སླར་ཡང་དེ་དག་ལ། །སྔོན་ལས་པྲི་ཡ་ཏྭཾ་པུ་ན། །ལོ་ཀ་སྦི་ཏི་བི་རུདྷེ་ཥ་བི་ཥ་ཡེ་ཥ་བི་ཤེ་ཥ་ཏ། །ལྷག་པར་དགའ་བར་འགྱུར། །འཇིག་རྟེན་ལུགས་དང་འགལ་བ་ཡིས། །ཡུལ་རྣམས་ལ་ནི་ཁྱད་པར་དུ། །སྤྲོ་ཧཱ་སཾ་ཛཱ་ཡ་ཏེ་ཛནོ་ར་པ་བྱེ་ཊི་བ་རོ་གི་ཎ། །སུ་ཁོཥཱ་ཀཱཏ་པ་རི་སྲོག་ཆགས་ཡང་དག་འདོད་གྱུར་ཏེ། །གནོད་པའི་ཟས་ལ་ནད་ཅན་བཞིན། །བདེ་བ་རབ་རྒྱས་ལས་ཏྭཱཏྟཾ་པ་ར་ཧིསྟ་མུ་པ་ག་ཏཾ། །པྲ་ཡ་སཊཾ་བྷ་བ་བྱེ་བ་ཛ་ཌ་སཱུ་མྲ་མི་ཡོངས་བཏང་ཚེ། །གཞན་གྱི་ལག་ཏུ་ཉེ་བར་སོང་། །བླུན་པོ་དག་གི་ཨ་མ་བཞིན། །ཕལ་ཆེར་ཐམས་ཅད་བ་པྲ་ཡཾ། །ཏསྣདྡྷ་བནྟ་མུཏྟཱརྨ་པྲ་ཏཱ་པ་ནི་དྷི་མཱ་ས་ནཱཏ། །་དགའ་བར་འགྱུར། །དེ་སླད་གཟི་བྱིན་ཆེན་པོའི་གཏེར། །ཁྱོད་ནི་ཁྲི་ལས་རབ་བསྐྱོད་ནས། །གནས་—།
