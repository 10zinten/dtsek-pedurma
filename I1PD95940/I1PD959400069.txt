སཾ་ཏོ་ཥ་བཱ་སཾ་ཀོ་ནཱ་པ་བྷ་བ་ཏཱ་ན་བི་མོ་ཧི་ཏ། །ཨི་ཏི་བཱ་དི་ནི་བྷཱུ་པཱ་ལེ་རབ་ཏུ་སིམ་པར་འདོད་པ་རྣམས། །སུ་ཞིག་ཁྱོད་ཀྱིས་རྣམ་མི་རྨོངས། །ས་སྐྱོང་དེ་སྐད་བརྗོད་པ་ཡིས། །ས་ཧ་སཱནྟ་རི་ཏི་སྨ་རེ། ་། བ་བྷཱ་བ་བི་ཀླ་བཱ་དེ་བཱི་ཏསྨཱ་བི་ར་ཧ་བ་ནྣི་འཕྲལ་ལ་འདོད་པ་མི་སྣང་གྱུར། །འབྲལ་བའི་མེ་ཡིས་གདུངས་པ་ཡིས། །ལྷ་མོ་ཤིན་ཏུ་རྣམ་པར་སྣ། །དུཁི་ཏཱནྣཱ་མན་དུཁཱ་པ་ཏི་བྷོ་ག་པི་ཡོ་གི་ནཱི། །ཨུ་བཱ་ཙཱ་འཁྲུགས། །ཡིད་བྱུང་ལས་རྒྱལ་རྒྱལ་པོ་ཡི། །བཙུན་མོ་བདག་པོ་ལོངས་སྤྱོད་བྲལ། །སྡུག་བསྔལ་མི་ཤཱ་ས་ཡ་ནཱ་ཛཱ་ཛཱ་ཡཾ་ཛི་ཏ་མ་ནོ་བྷ་བ། །དེ་བ་དྷརྨ་ཤྲཱི་ཡ་ཡུཏྟཱ་ན་ཤོ་ཀཾ་འོས་སྡུག་བསྔལ་ལྡན། །དེ་ལ་དབུགས་ནི་འབྱིན་ཞིང་སྨྲས། །ལྷ་མོ་ཆོས་ནི་བྱ་བར་རིགས། །མྱ་ངན་བྱ་ཀརྟུ་མརྟ་སི། །དུཁཱ་པ་པ་ནི་བི་ར་སསརྦ་ཡཾ་སུ་ཁ་བི་བྷ་མ། །དེ་ཧི་བར་མི་འོས་སོ། །བདེ་བས་རྣམ་པར་འཁྲུལ་པ་ཡིས། །རོ་བྲལ་མཇུག་ནི་སྡུག་བསྔལ་ཡིན། །ལུས་ཅན་ནཾ་ད་ཡི་ཏཱ་སཾ་གཱསྟ་རཾ་ག་ཏ་ར་ལཱ་ཡུ་ཏྭཾ། །ལོ་ལ་པདྨ་པ་ལཱ་ཤཱ་གྲ་སྐྱ་ལཛྫ་མ་ཚེ་ནི་རྒྱུག་ཆུ་འདྲ། །གཡོ་བ་རྣམས་ཀྱི་སྙིང་སྡུག་གྲོགས། །པདྨ་གཡོ་བའི་འདབ་རྩེ་ན། །འབྲེང་བའི་ཆུ་ལ་བཱ་ཀུ་ལཱ། །ཨི་མཱ་མཱུ་ཧཱུརྟ་ནཱརྟ་ཀཱལ་མེ་གྷ་ཏ་ཌིལླ་ཏཱ། །སཾ་ཐིགས་ལྟ་བུ་ཡིན། །དཔལ་འདི་ཡུད་ཙམ་གར་མཁན་དང་། །དུས་ཀྱི་སྤྲིན་གློག་འཁྱུག་པ་དང་། །འཁོར་སཱ་ར་སམྦུ་ར་ས་ནཱ་བི་ལཱ་ས་ཙ་པ་ལཱ་ཤི་ཡ། །བྷོ་ག་ཀླུ་ཎོ་ནི་བ་བི་ཡོ་ག་རོ་གཱབའི་སྦྲུལ་གྱི་རོ་འཛིན་ནི། །རྣམ་པར་རྩེ་བཞིན་རབ་ཏུ་གཡོ། །ལོངས་སྤྱོད་སྐད་ཅིག་ཁོ་ནས་འབྲལ་བི་བྷཱུ་ཏ་ཡསཱསྣ་བི་བཱ་ཧ་ཏུ་ལྕ། །བཱ་ཏཱ་ཧ་ཏཱ་དཱི་པ་ཤི་ཁཱ་སུ་ཁ་ཤྲཱི་བའི་ནད། །འབྱོར་པ་རྨི་ལམ་བག་མ་ལེན་དང་མཚུངས། །བདེ་བའི་དཔལ་ནི་རླུང་བསྣུན་མར་མེ་རྩེ། ཏུནྣཏྟ་ཀྲྀཏྟཾ་བྷ་བ་མྲྀཏྟ་མེ་ཏཏ། ་། སཟོ་པ་ཛཱི་བྱཱ་ཀ་རུ་ཎ་ན་ལགླི་སདྷརྨ་སྲིད་པའི་སྤྱོད་ཚུལ་འདི་དག་སྨྱོན་པའི་གར། །སྙིང་རྗེ་ཀུན་གྱིས་བསྟེན་བྱ་དཔལ་འབྱོར་མིན། །དམ་ཆོས་པྲ་ཀཱ་ཤས་ཏན་ན་དཱི་པཱ། །།ཡ་ཤྲཱི་སི་ར་མྱུ་ནི་ན་ཡོ་བ་ནཱ་ནི་སྦི་རཱ་ཎི་རྟག་ཏུ་གསལ་བྱེད་ཀྱི་མར་མེ་མིན། །གྲགས་སྙན་རྣམས་ཀྱིས་དགའ་ལྡན་ན་ཆུང་མིན། །བསོད་ནམས་དག་།
