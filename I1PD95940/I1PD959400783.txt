དིགྷནྡྷ་མཱ་བད་དེ། །དིགྷནྡྷེ་བི་ཧི་ཏི་ཏེ་ན་པྲ་ཏེ་ན་ཏུ་གྲ་ཏེ་བསྒྲུབ་སླད་ཕྱོགས་ནི་བཅིང་བ་བྱས། །བརྟུལ་ཞུགས་རབ་དྲག་གཟི་ལྡན་དེས། །ཕྱོགས་ནི་བཅིངས་པ་ཛ་ས། །ཤི་རོ་བྱ་ཐཱ་ནཱ་གཱ་བ་ཏེ་ར་བྷཱུཏྟལྟ་པ་ཌཱ་མ་ཎེ། །ཨ་ཐཱ་དྲི་བསྒྲུབ་པ་ནི། །ཀླུ་བདག་གདེངས་ཀའི་ནོར་བུ་ནི། །གདུང་ཞིང་མགོ་བོ་གཟིར་བར་གྱུར། །དེ་ནས་མ་ཤཱ་ས་མུཏྟཱ་ཡ་ཏཾ་ཏྲཱབྷྱཱ་མནྡྲ་སཱ་དྷ་ཀཾ། །ཀསྨྲ་མཱ་ནཕ་ཎི་པ་ཏིཏྲཱནྟ་མཐོང་ཡང་དག་ལངས། །སྔགས་ནི་བསྒྲུབ་པ་དེ་མཐོང་ནས། །འདར་ཞིང་གདེངས་ཅན་བདག་པོ་ནི། །ཏཱ་ས་ད་ཙིནྟ་ཡཏ། །བ་བྷུདྡྷཱཟླ་ཐུ་ས་པ་ལ་སྟ་ཌིཀྐ་པི་ལ་ལོ་ཙ་ན། །་བཅིངས་པས་སྐྲག་པས་རབ་བསམས་པ། །ཁྲོ་བོ་སྨ་ར་སྨིན་མ་སེར། །ལྟ་བྱེད་སེར་སྐྱ་གློག་ལྟ་བུ། །ཨ་ཀཱ་ལེ་ཀཱ་ལ་སཾ་ཀཱ་ཤཱཀུ་ཏཀཱ་རོ་པ་མཱ་ག་ཏ། །བ་ནེ་སཾ་པྲ་ཏི་དུས་མིན་དུས་དང་མཚུངས་པ་ཡི། །གདུག་པ་འདི་ནི་ག་ལས་འོངས། །ད་ནི་གདུག་པའི་བདག་ཉིད་དིནྡྷཀྲྀ་ཏོ་ནེ་ན་དུ་རཱཏྨ་ནཱ། །བ་དྷཱ་ཏེ་ཥ་ན་ཡཱ་བf་ཏཱ་བཏྱཱཀྟི་འདིས། །ནགས་ཀྱི་ཕྱོགས་ནི་བཅིངས་པ་བྱས། །ཇི་སྲིད་འདིས་བདག་མ་བཅིངས་པ། །དེ་སྲིད་ཀྲ་མཀྵ་མ། །ཏི་རོ་བཱནྟེ་བ་ས་ཏུསྨིརྨ་ཧནྟིཏྠ་ལཱ་ཡ་རིགས་པའི་རིམ་པ་བཟོད། །དྲང་སྲོང་ཆེན་པོ་ཤིན་ཤུན་ཅན། །འཇུག་ངོགས་ཉེ་བའི་མཐའ་འདིར་ན༑། ཛཱ་ནེ་སོ་སི་ནཱ་མ་སཱ་དྷུ་པ་རི་ཏཱ་ཎེ་པྲ་གལྷ་ཏེ། །ཏ་སཱུ་ཐ་མ་གནས། །བདག་ཡོངས་བསྐྱབ་པ་དམ་པ་ནི། །དེས་ཀྱང་ནུས་པ་མིན་པར་ཤེས། །གང་འདི་དེ་ཡི་བསྟི་ཡ་དེ་ཡོ་སོ་པ་རི་ཙནྡཱ་ས་རས་དཱ། །ལུབྷཱ་ཀཔདྨ་ཀོ་ནཱ་མ་ས་མེ་སཾ་རཀ་གནས་ན། །རྟག་ཏུ་རིམ་གྲོ་ཡོངས་བསྒྲུབ་པ། །རྔོན་པ་པདྨ་ཅན་ཞེས་པསོ། །དེ་ཡིས་བདག་ནི་ཎ་ཀྱུ་མ། །ཨི་ཏི་ནིཕྱི་ཏྱ་མ་ན་སཱ་ག་ཏཱ་ལུབྷ་ཀཱནྟི་ཀཾ། །སཱ་བྲྀཏྟཱནྟཾ་བསྲུང་བར་བཟོད། །ཅེས་པ་ཡིད་ཀྱིས་ངེས་བྱས་ནས། །རྔོན་པ་ག་ལ་བ་དེར་སོང་། །རང་གི་སྤྱོད་ནི་བིཅུ་རྨི་བནྡྷ་རཀྵ་མ་ཡཱ་ཙ་ཏ། །སཱ་དྷ་ཀ་སྱ་བ་དྷཱ་ཡི་བ་ནཱ་གཱ་རཱ་ཛེ་ན་ཚུལ་བཤད་ནས་ནི། །དེ་ལ་འཆི་བ་བསྲུང་བར་ཞུས། །ཀླུ་ཡི་རྒྱལ་པོས་སྒྲུབ་པ་པོ། །གསད་ཅེཔའི་སླད་།
