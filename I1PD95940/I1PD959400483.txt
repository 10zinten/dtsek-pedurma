ཡཱ་སདྷུ་རུ་བྷུཔཱབྷསྨཱ་བྷིརྦ་བུ་དྷ་བི་ཛ་ཡི་ཀོ་པི་བི་ཧཱུཾ་ཤ་ལེ་ཤ། །ཨ་ཧཱུཾ་བླ་མ་དམ་པ་དག། བསྟེན་ལས་མཁས་པ་རྒྱལ་བྱེད་རིག་པའི་ཆ་ཡི་ཆ་ཤས་ཅི་ཡང་ཐོབ། །ཡིད་འོང་བི་དཱད་ཀ་མ་ལ་བྷ་ར་སོ་མྱ་པྲ་བྷཱ་ས་བྷཱ་ཡཱ་ཏ་སྤྱོཥདྷཾ་ཀ་ཏི་པ་ཡ་པ་མཁས་པའི་པདྨའི་ཚོགས་ཀྱི་འོད་®དང་ལྡན་པའི་མདུན་ས་འདིར། །དེ་ཡི་ཚིག་ནི་ཕུལ་དུ་བྱུང་བའི་ཡིད་དཾ་པྲ་ཏྱ་ཡཾ་དན་ཡ་མ། །ནི་ཛ་གུ་ཎ་ག་ཎ་ནེ་དྷཱིརླཛྫ་ཏེ་སཛྫ་ནཱ་ནཾ་མུ་ཁ་ར་ཡ་ཆེས་འགའ་ཞིག་བདག་གིས་བསྟན། །སྐྱེ་བོ་དམ་པའི་བློ་ནི་རང་གི་ཡོན་ཏན་བརྩིར་ངོ་ཚ། །དེ་ལྟ་ན་ཡང་ཏི་ཏ་ཐཱ་བི་དྲྀ་ཌྷ་བཱ་དཱ་བྷི་ལཱ་ཥ། །ཨི་ཡ་ཏི་ཛ་ག་ཏི་རཱ་ཛན་ཀྱི་ཕྲ་མ་ནཱི་སྨྲ་འདོད་དྲག་པོས་མུ་ཅོར་སྒྲོག་ཏུ་བཅུག། རྒྱལ་པོ་འགྲོ་འདི་དག་ན་བདག་གི་ཚིག་ལན་ལ་སྲེད་པའི། །ཥཱ་ཏྭཾ་མེ་པྲ་ཏི་བ་ཙ་ན་རུ་ཙིཉྩེ་དསྟི་ཀདྲི་པཉྩིཏ། །སནྡདྷརྒཌྷ་གམྦྷི་ར་མི་ཏི་གལ་ཏེ་མཁས་པ་ཡོད་ན་མྱུར་བ་ཉིད་དུ་བརྟག་པར་གྱིས། །ཞེས་པ་དེ་ཡི་གཞུང་གི་ཚིག། དྲག་པོར་ཏ་སྩོཥ་ཊསྨྲ་ཙ། །ཐུ་ཏྭཱ་ཀྱི་ཏི་པ་ཏིཥི་པྲ་བི་ལཀྵས་མ་ཙིནྟ་ཡད། །སྙེམས་ཤིང་ཟབ་པ་དག། ཐོས་ནས་ས་ཡི་བདག་པོ་ནི། །སྐྱེངས་པས་མྱུར་བར་ཡང་དག་བསམས། །ཨ་པྲལྟ་པྲ་ཏི་མཀློ་ཡཾ་ཡ་དི་ཡཱ་ཡཏ་མ་དོདྷ་ཏ། །ཏ་དེ་ཥ་མ་མ་དེ་ཤ་གལ་ཏེ་ལན་ནི་མ་ཐོབ་པར། །རྒྱགས་ཤིང་འཕྱང་འདི་སོང་གྱུར་ནི། །དེ་ལྟ་ན་ནི་བདག་ཡུལ་གྱི། །སྱ་ཡ་ཤཁཎྜ་ན་ཌིཎྜི་མ། །གུ་ཎཱ་བ་མཱ་ན་ཀྲྀདྨ་ཏྲ་མཱུཌོ་བྷ་བ་ཏི་བྷཱུ་པ་གྲགས་པ་མིན་པ་སྒྲོག་པའི་རྔ། །གང་ན་ས་བདག་བླུན་པོར་གྱུར། །ཡོན་ཏན་ཅན་རྣམས་དམན་པར་ཏི༑། ན་ཀ་རོ་ཏི་ཛ་ནས་ཏྲ་བི་དྷཱཛྫ་ན་པ་རི་ཤ་མོ། །བི་བེ་ཀ་བི་མ་ལ་ལོ་ཀེ་བྱེད། །དེ་ན་སྐྱེ་བོ་རིག་བྱེད་ལ། །ཡོངས་སུ་ངལ་བ་བྱེད་མ་ཡིན། །ས་བདག་ཆོས་ཀྱི་ཀུན་དགའ་ར། །དྷརྨཱ་རཱ་མེ་མ་ཧི་པ་ཏོ། །ལོ་ཀེ་བི་རཱུཔྲ་བཏྟནྟེ་ས་དཱ་ཙཱ་ར་ཀྲི་ཡ་ཨི་བ། །དྲི་མེད་དབེན་པའི་སྣང་བ་ཅན། །འཇིག་རྟེན་རིག་བྱེད་ལ་འཇུག་སྟེ། །དམ་པའི་བྱ་བའི་སྤྱོད་ལ་བཞིན། །ཏསྨཱ་ད་སྱ་པྲ་ཡཉྫཱ་ན་ཀལྟ་བྱོ་མ་ད་ནི་གྲ་ཧ། །བི་་ད་རི་དྲ་ཏཱ་དེ་ཤེ་དེ་སླད་འདི་ཡི་རྒྱགས་པ་ནི། །རབ་ཏུ་འབད་པས་ཚར་གཅད་བྱ། །ཡུལ་ནི་རིག་བྱེད་ཀྱིས་དབུལ་བ། །།
