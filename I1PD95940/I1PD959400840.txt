བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི། བྷཀྵ་སརྨ་ཏ། །དེ་བི་ཀ་བནྟ་པ་ད་ནེ་མ་ཡཱ་དྲྀཥཱཀུ་མཱ་ར་ཀ། །ཡ་མངོན་པར་འདོད་པས་སྨྲས། །ལྷ་གཅིག་ཉ་པའི་ཁང་པར་ནི། །གང་གིས་ཁྱོད་ཀྱི་རྒྱལ་སྲིད་དང་། །སྲོག་ཨེ་བ་ཀ་ཐི་ཏ་པུཤྲཱི་རཱ་ཛྲ་པཱ་ཎཱ་པ་ཧ་སྟ་བ། །ཐུ་ཏྲེ་ཏི་ནཱཾ་པ་ཏིཀྲོ་དག་འཕྲོག་པར་སྔར་བརྗོད་པའི། །གཞོན་ནུ་བདག་གིས་མཐོང་བར་གྱུར། །འདི་ཐོས་མི་བདག་ཁྲོས་པ་པཱནྡྲིབྷཱུརྞཱ་པ་ར་མཱ་ཏ་རཾ། །གོ་བི་ཥ་ཎི་ས་མཱ་ཧུ་ཡ་མ་ཧཱ་མཱ་ཏ་མ་བྷཱ་ཥ་ཏ། །ཡིས། །གཞན་པའི་མ་ལ་རབ་སྨད་དེ། །བློན་པོ་ཆེན་པོ་བ་གླང་རྭ། །ཡང་དག་བོས་ནས་རབ་སྨྲས་པ། །ཨ་ཧོ་བ་ཏཱ་བ་ཏཱ་པེ་ན་བྷ་བ་ཏཱ་མ་མ་སཾ་ཤ་ཡེ། །རཱ་བྷཱི་ཀརྞཱ་དྷཱ་རེ་ཎ་ནོ་རི་བ་ཀྱེ་མ་བདག་གི་ཐེ་ཚོམ་ལ། །རྣ་བ་འཛིན་པ་རབ་གཡེལ་བ། །ཁྱོད་ཀྱི་རྒྱལ་སྲིད་ཆུ་གཏེར་ལ། །གྲུ་བཞིན་ཤྲཱི་རུ་ཕེ་ཀི་ཏཱ། །ཏྭཱསྨྲ་ཏི་ནྱ་སྟ་ཙིཏྟ་སྱ་སུ་ཁཾ་ནི་དཱ་བྷ་བརྨ་མ། །དཔལ་ལ་བཏང་སྙོམས་བྱས། །ཁྱོད་ཀྱི་བློ་ལ་སེམས་བཀོད་པ། །བདག་ནི་བདེ་བར་གཉིད་གྱུར་ཏེ། །སི་བཱ་རྱ་པྲ་ཎ་སཾ་དེ་ཧ་ཛཱ་ར་ཏནྟི་པ་དཾ་ཤྲཱི་ཏྭཾ། །གཱ་ཌྷ་གབྷཾ་པ་རི་ཏུ་དེ་རིང་སྲོག་ལ་ཐེ་ཚོམ་གྱི། །རིམ་གྱི་དབང་གི་གནས་དག་བསྟེན། །བདག་གི་མཐར་བྱེད་སྦས་པའི་ཙུ་ཀི་བནྟེ་ཥཱ་མ་དནྟ་ཀཾ། །པ་བྲྀཏྟཱ་པ་ར་མཱ་ཏཱ་མེ་ག་ཎ་ཡ་ཏེ་བ་མངལ། །ཉ་པ་དག་ལ་ཡོངས་གཏད་ནས། །རབ་ཏུ་དགའ་བས་གཞན་པའི་མ། །ངེས་པར་ཉི་མ་བཱ་ས་རཱན། །ཨ་བྱཱ་པི་ཏདྨ་དྷོ་པཱ་ཡེ་ཏཱ་ཡཱ་ཡུཀྟིཐི་དྷི་ཡ་ཏྭཾ། །ནབགྲང་བར་བྱེད། །ད་དུང་དེ་ནི་གསད་པའི་ཐབས། །རིགས་པ་ཁྱོད་ཀྱིས་སྒྲུབ་པར་མཛོད། །སེན་ཁ་ཙྪེ་བྱི་ཀུ་ཥཱ་རོ་ཀཱ་ལེ་ནཱ་ཡཱ་ཏི་ཀུཎྜ་ཏྭཾ། །རཀྵ་ཏི་སྭཱ་མི་རཱཥྛརྦ་མོས་ཆོད་པའི་དུས་ཀྱི་ནི། །སྟ་རེ་ཡིས་ཀྱང་ནུས་མི་འགྱུར། །རྗེ་བོ་ཡུལ་འཁོར་ནོར་དང་རྫོང་། །དུརྒ་མི་ཏྲཱ་བ་ལོ་ད་ལཾ། །ཨ་མཱ་ཏྱ་བྲ་ཀྲྀ་ཏིསྟམཏྲ་ཀྲྀ་ཏི་བློ་ག་རི་ཡ་གྲོགས་དང་དཔུང་རྒྱས་སྲུང་བྱེད་པ། །དེ་ཡི་སླད་དུ་རང་བཞིན་ལས། །བློན་པོའི་རང་བཞིན་ཤིན་ཏུ་སཱི༑། བྱ་ས་ན་ཤ་མ་ན་རྒྱུ་ནཱ་སཀྵས་དཱ་ཧི་ཏ་པཱ་ཡི་ནཔྲ་ཎ་བརླིང་། །བློན་པོ་གདུང་བ་ཞི་བྱེད་བསམ་གཏན་ལ་ཆགས་རྟག་ཏུ་ཕན་པ་བློསི། །ཉན་རྣ་ཨེདུལ་བའི་།
