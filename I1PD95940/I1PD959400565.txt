།བྷུ་བ་ན་བྱཱ་སི་བི་བུ་དྷཱ་བྱཙྩཾ་ཏཾ་ཡ་། །ས་ཀ་དཱ་ཙིདྡ་རོ་བྱ་ནཾ་སྱནྡ་ནེ་མཐའ་དག་ཁྱབ་པ་ནི། །ཆོས་ཀྱི་བཞིན་དུ་ཡང་དག་རྒྱས། །སྐྱེད་ཚལ་མཆོག་ཏུ་དར་མ་དེ། །་ནམ་ཞིག་ན་ཡུ་བཱ་བྲ་ཛན། །ད་དན་བྲིདྷི་པུ་རུ་ཥ་དེ་བ་ཏཱ་ནིརྨི་ཏ་བུ་ར། །ཏཾ་བི་ཤིང་རྟས་འགྲོ་བ་ན། ་། མདུན་དུ་ལྷ་རྣམས་ཀྱིས་སྤྲུལ་པའི། །སྐྱེས་བུ་རྒན་པོ་མཐོང་བར་གྱུར། །རྒ་བས་ལོ་ཀྱ་ཛ་རཱ་ཛཱིཙྩཾ་ཛཱ་ཏ་བི་རཱ་གྱ་བཱ་ས་ན། །སཾ་སཱ་ར་མི་བ་ནི་སཱ་རཾ་ས་འཁོགས་པ་དེ་མཐོང་ནས། །ཆགས་བྲལ་གྱི་ནི་བག་ཆགས་སད། །འཁོར་བ་བཞིན་དུ་ལུས་དག་ནི། །ཤ་རཱི་ར་མ་མ་ནྱ་ཏ། །ཨུཏྱཱ་ནཾ་ཡཱ་ཏྲ་བི་ར་སཤ་ནི༔པྲ་ཏི་ནི་བྲི་ཏུ་སྙིང་པོ་མེད་པར་དེ་ཡིས་བསམས། །སྐྱེད་ཚལ་བལྟ་བའི་ཆགས་བྲལ་ཏེ། །དལ་གྱིས་ལོག་ནས་འོང་བ་ཀ༈་། དཱ་རི་དྲ་བི་དྲུ་ཏཙྪཱ་ཡ་ན་དཱཀྵཀྵ་པ་ཎཱརླ་ཐི། །དྲྀབྷཾ་ཏཱ་མ་དེས། །ལམ་དུ་དབུལ་བ་ཉིད་ཀྱིས་ནི། །མཛེས་བྲལ་ཕོངས་པར་གྱུར་པ་མཐོང་། །སྙིང་རྗེས་དྲངས་སུ་ཁ་གླིཋ་ནུ་ཀ་རུ་ཎཱ་ཀྲྀཥཱ་མཱ་ན་ས། ་། ཨ་ཙིནྟ་ཡ་ད་ཧོ་དུཁཾ་ས་པའི་ཡིད་ཅན་དེས། །བདེ་མེད་ཉོན་མོངས་དེ་མཐོང་ནས། །བསམས་པ་ས་ལ་དབུལ་པོ་རྣམས། །ཀྱེ་མ་ཧནྟེ་བྷ་བི་དུརྒ་ཏཱ། །ཨ་དཱ་ན་པྲ་བྷ་བ་དུཁམྦ་དནྟེ་ཏི་བི་སངྒ་ཏཱ། །སྡུག་བསྔལ་བཟོད་པར་བྱེད། །རིན་ཆེན་གྱིས་གང་ས་གཞི་ན། །གཞན་ལ་བསླངས་པས་ཉེར་འཚོ་ཞིང༌། །པྲི་ཐི་ཧཱུཾ་རཏྣ་པཱུཙྪ་ཡཾ་པ་ར་པིཎྜོ་པ་ཛཱི་བི་ན། །ཨི་ད་མེ་བཱ་བི་སཾ་བཱ་དི་ཕོངས་གྱུར་འདི་དག་སྡུག་བསྔལ་ནི། །སྦྱིན་པ་མིན་ལས་སྐྱེས་པར་སྨྲ། །སྐྱེས་བུ་བཀྲེན་པ་འདི་དག་རྣམས། །ཙིནྟཾ་ཀ་ལུ་ཥ་ཀརྨ་ཎཾ། །དཱི་ནཱ་ཡ་དེ་ཏེ་ཡ་ཙནྟེ་པུ་རུ་ཎཾ་པུ་རུ་ཥཱཔ་རཾ། །།སྐྱེས་བུ་གཞན་ལ་སློང་བ་གང་། །འདི་ཉིད་ཁོ་ན་མི་བསླུ་བར། །སྡིག་པའི་ལས་རྣམས་དག་གི་རྟགས། །ཨ་ཧོ་དུཀྲྀ་ཏ་མེ་ཏེ་ཥཱ་མ་བ་བྷཾ་ཏཱཔ་དེ་པ་དེ། །ཡ་དེ་ཏེ་མཱརྒ་གནས་དང་གནས་སུ་སྨད་གྱུར་ཅིང་། །འདི་དག་སློང་བས་ཡིད་བྱུང་ལ། །བསླངས་པར་གྱུར་ཀྱང་ཎོ་དཱིགྷཱ་བྷིཀྵི་ཏཱ་པི་བུ་བྷུ་ཥི་ཏཱ། །ཨི་ཏི་སཾ་ཙིནྟ་པུ་ཙི་ར་བི་ཤ་ཀྵེ་ཤ་བཀྲེས་པ་ན། །ཀྱེ་མ་འདི་དག་ཉེས་བྱས་ཉིད། །དེ་ལྟར་ཡུན་རིང་བསམས་བྱས་ནས། །ཀུན་གྱི་ཉོན་།
