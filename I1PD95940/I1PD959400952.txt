ཆུ་ཀཛ་ནཤུ་བྷེ་སྱསྨིནུ་ཀྲྀ་ཏ་ཙི་རི་ཏེ་དུཀྲྀ་ཏ་ཀ་ཎཾ་ནི་དྷཏྟེ་ཡེ་ན་སོ་མ་ལི་སྐྱེ་བོ་ལེགས་བྱས་སྤྱོད་ཅིང་དཀར་བ་འདིར་ཡང་ཉེས་བྱས་དག་གི་ཆ། །འཛིན་འདི་གང་གི་བཞིན་ན་བ་ད་ནཤོ་ཙ་ཏི་ཏ་རཾ། །སྤྲོ་ཧཱ་ནི་དྷཱ་ནཾ་ཀྵ་ཎི་བྷ་བ་སུ་ཁེ་དུཁ་ནི་རས་དྲི་མ་ཅན་འདི་མཆོག་ཏུ་གདུང་བར་བྱེད། །སྐད་ཅིག་སྲིད་པའི་བདེ་བ་སྡུག་བསྔལ་ཚོགས་ལ་ཆགས་ཙ་ཡེ་པྲ་བི་ཤཱནྟཤོ་ཥ་དི་ཤ་ཏི་ཤ་ཏ་ཤཱ་ཁོ་རྒྱ་ནུ་ཤ་ཡ། ་ ོ། གཙྪ་པས་འཕངས་རྣམས་ལ། །།ལོ་འདབ་བརྒྱ་ལྡན་ཕྲ་རྒྱས་ནང་དུ་ཞུགས་ཏེ་གདུང་བ་སྟེར་བར་བྱེད། །མི་མི་ཚྭ་བ་ཏེ་ར་གྲེ་པཱ་པ་མཱ་དེ་བ་ཡཱ་མྱ་ཧཾ།  ོ། རཱ་ཛ་ཤ་སཱ་ན་ཤུདྡྷཱ་ནཱ་མ་ཚྭ་བདག་སྔོན་དུ་བདག་ཕྱིན་ལ། །སྡིག་པ་བདག་གིས་བརྗོད་པར་བྱ། །རྒྱལ་པོའི་བཀའ་ཡིས་དག་པ་རྣམས། །ཎོ་དྷརྨ་པཉྩ་ཡ། ༄༅། །ཤཱཙྩཾ་དྷརྨ་པ་ཐྲཏྟཱ་ནཾ་རོ་གཱཙྪཱ་ནཾ་ཙི་བུ་ལོན་མེད་པའི་ཆོས་ཀྱི་ཚོགས། །ནད་ཀྱིས་གཟིར་རྣམས་སྨན་པས་གསོ། །།།ཆོས་ལ་ཞུགས་རྣམས་བསྟན་ཀིཏྠ་ཀ། །བཙྪཱ་ཤ་མ་གུ་རུན་ཏཱ་ཚིགས་པ་ཏིཔཱ་པ་ཀརྨ་ཏཱཾ། ཌ་། ས་བཅོས་ཀྱིས། །།སྡིག་ལས་ཅན་རྣམས་མི་བདག་ནི། །རིགས་གནས་ལུགས་ལྡན་བླ་མ་ཡི། །ཞེས་བི་ཙིནྟི་ཏི་སནྟཱ་པཱཏ་བྷཱུ་པ་ཏེ་བྷཱ་ཏུ་རནྟེ་ཀཾ། །ག་ཏཱ་ནྱ་བེ་ད་ཡཏྟཱཎྞཾ་མ་དཏྟཱ་བསམས་ཡོངས་སུ་གདུང་བ་ཡིས། །ས་བདག་སྤུན་ཟླའི་དྲུང་དུ་ནི། །ཕྱིན་ནས་དེ་ཡིས་མྱུར་བར་ནི། །མ་དཱ་ན་པཱ་ཏ་ཀཾ། །ཛུཏྟཾ་བྷཱ་ཏེ་ར་མ་བྱཱཙཱ་ཏཾ་བྷཱུ་མི་པ་ཏི་ར་བྲ་བཱིཏྲ། ?། ཨཱ་བྱིན་བླངས་པའི་སྡིག་པ་སྨྲས། །སྤུན་ཟླ་ཐུ་བོ་མངོན་མཆོད་དེ། །དེ་ལ་ས་བདག་གིས་སྨྲས་པ། །སྐྱེས་ཛརྨ་ཤུདྡྷ་བྷཱ་བ་སྐྱ་ན་པཱ་པཾ་བི་དྱ་ཏེ་ཏ་བ། །སུ་སཱུ་ཀྵ་མ་པི་ཡཀྲྀ་ཏཱ་དེ་ནས་དག་པའི་དངོས་གྱུར་པ། །ཁྱོད་ལ་སྡིག་པ་ཡོད་མ་ཡིན། །།རིགས་མིན་ཤིན་ཏུ་ཕྲ་བ་ཡང་། །།ཇི་ཞིག་བཱིགྷཾ་ཙི་ཏཱ་སཾ་པྲ་ཏཾ། །པཉྩཏྟཱ་པཾ་བ་ཧ་ཏྱནྟཱ་ཏ་སྱ་སེ་བ་བི་ཤུདྡྷ་ཏ། །ཨི་ཏི་ཀྱི་ལྟར་སྟེས་གང་བྱས་པ། །ནང་དུ་རྗེས་སུ་གདུང་བ་འཛིན། །དེ་ཉིད་དེ་ཡི་རྣམ་དག་ཉིད། །ཅེས་པ་ས་ཏི་པ་ཏེཝཱ་དྱ་ཐུ་ཏྭཱ་སཱུརླསྟ་མ་བྱཱ་དྷཱཏྲ། །ཀ་ལཾ་ཀ་ལེ་ཤི་ན་ས་ཧེ་སདྷརྨཱ་དྷ་བ་བདག་ཚིག་ཐོས་ནས། །ཉི་མས་དེ་ལ་རབ་སྨྲས་པ། །དམ་པའི་ཆོས་ནི་དཀར་བ་ཡི། །གོས་ལ་དྲི་མའི་ཆ་—།
