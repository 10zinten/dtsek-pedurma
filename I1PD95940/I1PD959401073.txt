མཱ་ན་པོ་ཏོ། །སྣེ་ཧཱདྡྷ་བཱ་ཎ་བུ་པ་སྲྀ་ཏུ་ར་ཛཱ་ཛ་གཱ་ད་ཤཱ་སཱ་་བ་མཱི་ཧ་མཱན། །འདོད་ལྡན་བུ་ཞེས་པ། །སྨྲ་བ་དེ་དག་ལ་ནི་ཉེར་བཏུད་དེ། །དམོད་པས་ཉམས་པ་མི་འདོད་རྒྱལ་པོས་སྨྲས། །ན་པཱ་པ་བྷ་གི་ཡུ་བ་ཡོསུ་ཏོ་ཧཾ་སོ་ཛ་རྒྱ་བལླི་ཀ་ཐི་ནཀུ་ཥཱ་ར། །མཱ་གླང་ཆེན་མྱོས་པའི་ཚུལ་གྱིས་ཨ་མའི་ཤིང་། །འཇོམས་བཞིན་གང་གིས་གཞོན་ནུ་རབ་བཅོམ་པ། །སྐྱེས་ཏཾ་ག་བྲིཏྟཱ་མ་ཏ་རུ་ནི་པ་ཏུ་མནྟེ་ན་ཡོ་ཐི་ཏཀུ་མཱ་ར། །མཧཱ་ཡ་མཆོག་འཁྲི་ཤིང་དག་གི་སྟ་གྲི་རྩུབ། །སྡིག་པའི་སྐལ་ལྡན་བདག་ནི་ཁྱེད་སྲས་མིན། །བདག་གི་མདའ་ཀ་སཱུ་ཏ་ནུས་ཤེ་ཏེ་ཡུརྒྱུ་ཏཀྲྀཙྪཱ་ག་ཏསྟ་པནྟེ། །ནི་དྷཱི་ཡ་ཏི་མུདྡྷེ་མ་ཡིས་ལུས་བཅོམ་ཁྱོད་ཀྱི་སྲས། །དེ་ནི་སྡུག་བསྔལ་གྱུར་ནས་ཆུ་འགྲམ་ཉལ། །སྡིག་པའི་གདུང་བ་ཆེ་ལས་ཚ་མོ་གྲ་ཤཱ་པས་ཤི་ཏ་ལོབྷཱ་པྲ་ཐུ་པཱ་པ་ཏཱ་པཱཏ། །ཨེ་ཏུཀྟ་མཱ་ཏྲེ་བ་སུ་དྷཱ་དྷི་གྲང་དང་། །བཅས་པའི་དམོད་དྲག་བདག་གི་སྤྱི་བོར་དོར། །ནོར་འཛིན་བདག་པོས་དེ་སྐད་བརྗོད་ཙམ་པེ་ན་ཏོ་བཛྲ་བྷསྨཱ་བི་བ་བྷསྒ་དྷིརྨོ། །བྷི་ཏེ་བ་ཤོ་ཀ་སྱ་སུ་དུལ། །རྡོ་རྗེས་བཅོམ་བཞིན་བརྟན་པ་ཉམས་དེ་དག། ཤིན་ཏུ་བཟོད་དཀའི་མྱ་ངན་ལ་འཇིགས་ས་ཧ་སྱ་མ་ཧཱི་ཡཱ་སི་མོ་ཧ་གུ་ཧཱ་པྲ་བིཏྠོ། །སིཏྟོ་ཏ་ཏཤྲཱི་ཏ་ཛ་ལེ་ན་རཱ་ཛྷཱ་ཏོ་ལབྷཱ་བཞིན། །རབ་ཏུ་ཆེ་བ་རྨོངས་པའི་ཕུག་ནང་ཞུགས། །དེ་ནས་རྒྱལ་པོས་ཆུ་གྲང་གཏོར་དེ་དག། ཇི་སཾ་ཙཱ་བ་ཡ་བོ་ཀ་ཐཾ་ཙིཏ། །པུ་ཏཱནྟི་ཀཾ་བྷཱུ་པ་ན་ཡ་ཏཱ་མཱ་བཱ་མི་ཏུཙ་ཏུཏྠོ་ཀ་ཞིག་ལྟར་སྟེས་འདུ་ཤེས་ཆ་ཤས་ཐོབ། །ས་བདག་ཁྱོད་ཀྱིས་བདག་ཅག་བུ་དྲུང་དུ། །ཁྲིད་ཅེས་སྙིང་རྗེ་རུ་ཎ་པྲ་ལཱ་པོ། །ནཱི་ཏོ་ཏ་ཏསྟེ་ན་པ་རིཏྠ་མཱི་པ་མ་དཱི་པ་དཱི་པེ་བྱ་བ་ནེ་ནི་མསྒོ། །རབ་སྨྲ་དེ་དག་སྨྲ། །དེ་ནས་དེ་ཡིས་ཆུ་ཀླུང་ཉེ་བར་ཁྲིད། །གླིང་དང་སྒྲོན་མེ་ཉེ་བར་བྱིང་གྱུར་པོ། ཧསྟེ་ན་སཾ་སྤྲོ་བྱ་སུ་ཏཾ་ཤ་རཱཝཾ་ཏདྷཱ་ཎ་བིདྡྷ་དི་པ་བི་ཏ་ཏུསྟོ། །ཀསྣ་ལག་པས་མདའ་བསྣུན་བུ་ལ་རེག་བྱས་ནས། །དེ་ཡི་མདའ་ཡིས་ཕུག་བཞིན་དེ་དག་འགྱེལ། །བུ་གཅིག་ད་ཀསྣརྒྱུ་རུ་བཏྠ་ལོ་བི་བྲྀདྷཱ་བ་ནཱ་ཐོ་བི་ཏ་རོ་ཏཱ་མནྟོ། །ཏྲཀྵ་དི་པཾ་གཙྪ་བླ་མ་དག་ལ་མཉེས་གཤིན་ཡང་། །ཕ་མ་རྒན་ལོང་མགོན་མེད་བཏང་ནས་ནི། །ཅི་སླད་གློ་བུར་མཐོ་།
