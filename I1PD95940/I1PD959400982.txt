མ་དན་ཡཏྲ། །སྤྲོ་ཧཱ་མ་གླིཋ་རཱུ་པེ་ཏི་བ་ཀླ་ཐཾ་ནིསུ་ཁེ་ཏི་མངོན་སུམ་ཉིད་དུ་བསྟན། །མ་ཉམས་གཟུགས་ཞེས་ཆགས་པ་དང༌། །བདེ་བ་མེད་ཅེས་འཁྲུགས་པ་ཙ༑། ཨཱིཙྩཾ་ཛཱ་ཏཱ་པ་བཱ་དེ་ཏི་ཀྲུ་དྷཾ་དཱ་པཱཇི་ཏེ་ཏྱ་པི། །ས་ཏྭཾ་བི་དང༌། །སྨད་པར་གྱུར་ཅེས་ཕྲག་དོག་དང༌། །འབངས་ཀྱིས་གཟིར་ཞེས་ཁྲོ་བ་ཡང་། །ཐོབ་ནས་དེ་ལ་ལོ་ཀ་ཡནྡཱ་སྱ་སཾ་ཏྱལྟ་བྲ་ཏ་ཡནྡྲ་ཎ། །ཏུ་་སཾ་སཱ་ར་དོ་ཥཱ་ཎཾ་སཧྥེ་ལྟ་བྱེད་ཅིང་། །བརྟུལ་ཞུགས་ཀྱིས་བསྡམས་བཏང་བ་དེ། །མཚུངས་པར་འཁོར་བའི་སྐྱོན་རྣམས་ནི། །ཏཱཾ་བ་ཤ་མཱ་ཡ་ཡོ། །པཱ་ཏྲ་ཙཱི་བ་ར་མུབྷཾ་ཛུ་ཁརྒ་ཨཱ་ཀྲྀཀྵ་བྷི་ཐམས་ཅད་དག་གི་དབང་དུ་འགྱུར། །ལྷུང་བཟེད་ཆོས་གོས་རབ་བཏང་ཞིང་། །འཇིགས་རུང་རལ་གྲི་ཥ་ཎཾ། །པྲ་ཡཱ་ནི་ཀཱ་ར་ཀྲོརྨེ་ཎ་ས་དཱ་སཾ་ཧེ་ཏུ་མུརླ་ཡོ། །པྲ་ཎ་ཕྱུང་ནས་ནི། །དགའ་མའི་གནོད་པ་ལ་ཁྲོས་པས། །དེ་ཡིས་འབངས་ནི་གསོད་པར་ཞུགས། །འབངས་ཡཱདྷཱ་རི་ཏསཏྟཱ་པ་དཱ་ས་བ་དྷ་བི་ག་སཱཏ། །ལཱི་ནཾ་པུ་ནཀྲྀ་ཧ་སུ་ཁཾ་བྷེ་ཛེ་ནི་གསོད་པའི་སྡིག་པ་ལས། །ཆུང་མས་འབད་ནས་ཟློག་ཅིཔ་དེས། །དགྲ་ལས་རྒྱལ་འདོད་ཁྱིམ་གྱིས་ནི། །ཛ་ན་ཛི་གཱིཆུ་ཡཱ། །སོ་བྷཱུབྷྱཱ་ག་པ་རི་གྲ་ཧེ་ཎ་ལ་ལ་ནཱ་སཾ་བྷོ་བདེ་བ་དམན་པ་སླར་ཡང་བསྟེན། །དེ་ནི་ལོངས་སྤྱོད་ཡོངས་སུ་བཟུང་བས་བུད་མེད་ལོངས་སྤྱོད་སྐལ་ག་སོ་བྷཱ་གྱ་བྷཱུརཱ་ནཱ་ཀཱ་ར་ནི་ཀཱ་ར་ཀཱ་ར་ཎ་ཀ་ཐཱ་ཏི་ཏེ་ཥཱ་ཀོ་པཱ་བཟང་ས་གཞིར་གྱུརཚེ། །རྣམ་པ་སྣ་ཚོགས་གནོད་པའི་གཏམ་གྱི་རྒྱུ་ལས་གཡོག་འཁོར་རྣམས་ལ་ཁྲོ་བས་ཀུ་ལ། །ཀཱ་མ༔ཀརྟུ་ཏི་སཾ་ཡ་མཾ་ཀཱ་པ་ཡ་ཏི་ཀྲོ་དྷཀྵ་ཎེ་ན་ཀླུ་མཱི་ནོ་མི་འཁྲུགས། །འདོད་པས་དུལ་བ་ཉམས་བྱེད་ཁྲོ་བས་སྐད་ཅིག་ཉིད་ཀྱིས་བཟོད་པ་རླག་པར་བྱེད། །རྫུན་ཐཱ་བྲ་ཏ་ཤོ་ཏི་ཏེ་ན་མ་ན་སུ་བི་རཱ་གྲ་མཱ་ས་བྱ་ཏེ། །ཨ་ཐ་ཧསྟི་གྱི་བརྟུལ་ཞུགས་དག་གིས་གདུང་བའི་ཡིད་ཀྱིས་ཆགས་བྲལ་ཉིད་ནི་འཐོབ་ཀྱི་མ་ཡིན། །དེ་ནས་རྒྱལ་པོས་མ་ཧཱ་མཱ་ཏྲ་པ་དཾ་རཱ་ཛཱམི་ཏ་པུ་ན། །ཨཱ་སཱ་རྱ་བི་བྷ་བོ་དཱ་རཿསོ་བྷུ་གླང་ཆེན་གྱི༅། །སྣ་ཆེན་གནས་ནི་སླར་ཡང་བྱིན། །འབྱོར་པ་རྒྱ་ཆེན་ཐོབ་གྱུར་ནས། །དེ་ནས་ལྷག་།
