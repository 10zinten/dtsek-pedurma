དྲ་ས་མུ་དྲ་མེ་བ་ཡེ་ནེནྡྲི་ཡཱ་ཙྪཱ་ཤུ་པ་རི་ཤ་བན་ཏི། །དྷ་རྩྭ་སྦྲ་ཨེ་བ་ས་མ་ས་བུག་པའི་མཚོ་ཉིད་དེ། །གང་གིས་མྱུར་དུ་དབང་པོ་ཡོངས་ངལ་བྱེད། །ཚུལ་ཁྲིམས་མཐུ་ལས་གང་དག་ཏཱ་དྷཱི་རཿཤཱི་ལ་པྲ་བྷཱ་བཱརྨ་བྷ་བནྟི་ཡེ་ཡཱཾ། །ལཱ་བ་ཅུ་པཱ་ནེ་ན་བི་ཤེ་རྣམས་ཀྱི་མིག། མཛེས་པའི་བཏུང་བས་ཁྱད་པར་དུ་ངོམས་ཤིང་། །རྣམ་པར་གཡོ་བ་མིན་པ་དེ་དག་ཥ་ཏྲིཔ་བི་བྷུཎྜ་མཱ་ནཱ་ནི་བི་ལོ་ཙ་ནཱ་ནི། །ཨི་ཏྲ་དི་ཏ་སྱ་པྲ་ཤ་པ་པནྣ་ཉིད། །སྙིང་སྟོབས་བརྟན་པ་དང་ལྡན་སྐལ་བ་བཟང༌། །ཞེས་སོགས་དེ་ཡི་ཚིག་ནི་རབ་ཞི་བ། །ཉེ་མཱ་ཀཎྚ་བཱ་ཀྱནྣ་ར་ནཱ་ཐ་སུ་ནུ། །ཏ་ཐེ་ཏི་ཀྲྀ་ཏྭཱ་མ་ན་སི་པྲ་སནེ་ཀྲྀ་ཏ་པྲ་ཎ་མསྭཱ་བར་རྫོགས་པ་མི་བདག་བུས་ཐོས་ནས། །དེ་བཞིན་ཞེས་དེ་ྋརབ་དང་ཡིད་ལ་བྱས། །ཕྱག་འཚལ་ནས་ནི་པ་དྨ་ཛ་གཱ་མ། །ཨ་ཐ་ཡ་ཡོ་བྷཾངྒ་ག་ཎོ་པ་གཱི་ཏཾ་སི་དྷུ་ར་པཱུ་རཱ་པི་ཏ་ཀིཾ་ཤུ་ཀ་ཤཱི། །རང་གི་གནས་སུ་སོང་། །དེ་ནས་བུང་བའི་ཚོགས་ནི་གླུ་ལེན་ཅིང་། །ཀི་ཤུ་ཀ་དཔལ་ལི་ཁྲིས་རྒྱས་པ་འདྲ། །མ་ན་སཱི་ནཱི་མཱ་ན་བ་དྷཱ་ནུ་བནྡྷི་མ་དྷུ་ས་རཏོ་ར་བྷ་མཏྟ་ནཱ་ག། །ཨུངྒཱ་ཁེངས་ལྡན་མ་ཡི་ཁེངས་པ་འཇོམས་ཆས་པས། །སྟོན་གྱི་དྲིས་མྱོས་གླང་པོ་ཅིག་འོངས་གྱུར། །སྐྱེད་ན་བལླི་ན་བ་པལྤ་བཱ་ནཾ་བི་ཡོ་གི་ནཱི་ཏཱ་པ་བི་པལྦ་བཱ་ནཱི། །བ་བྷཱུ་ཚལ་འཁྲི་ཤིང་གསར་པའི་ཡལ་འདབ་དང༌། །འབྲལ་བའི་གདུང་བ་རྒུད་པའི་ཆ་ཤས་དག། གསར་དུ་བདྲྀདྷས་ཧ་སཱ་ས་ཧི་བ་བྲ་ཏྱ་གྲ་རཱ་གངྒཱ་མ་དུས་ཧེ་བ། །བཱ་ཏེ་རི་ཆགས་པས་རབ་བསྐྱེད་བཟོད་དཀའ་ཉིད། །འཕྲལ་ལ་ལྷན་ཅིག་ཉིད་དུ་རྒས་པར་གྱུར། །ཙམ་པ་ཀ་ཏིཉྩམྤ་ཀ་པ་ཏྲ་ལེ་ཁིཙླི་ཏཱི་པ་མཱ་སཱུ་ཏྲ་ཡ་ཏསྨྲ་རེ་ཎ། །ས་པ་སྲརྗེ་འདབ་རླུང་གིས་བསྐྱོད་པ་ཡིས༄༅། །སྤྲིང་ཡིག་གིས་ནི་བྱམས་པ་སྐྱེད་བྱེད་ཅིང་། །འདོད་དང་ལྷན་ཅིག་དིཀྐུབྷཾ་ཏི་པྲ་མཱ་ཤྲཱི་ཙི་ཏྲ་སྱ་ཛེ་ཏྲཿཔྲ་ཐ་མཱ་བྷི་ཡོ་ག། །ལ་སཏྰ་པུལླེ་དཔྱིད་ཀྱི་དང་པོ་ཡི༄༅། །མངོན་འབྱོར་རྒྱལ་བྱེད་བདག་འཇོམས་ཕྱོགས་སུ་གྲགས། །མེ་ཏོག་རབ་རྒྱས་ཥཱ་པི་ཏེ་ཥཱ་ཏྲེ་ཥཱ་བ་བྷཱུ་བ་བྷུ་སྣ་མ་རཱུ་བནྡྷ་བ་སྱ། །བྷཾངྒ་སཱ་ནིཧཱུཾ་ཏ་ཡ་ཤ་པྲ་ཀཱ་དེ་དང་དེ་དག་ལས། །།དཔྱིད་ཀྱི་གཉེན་ལ་མཆོག་ཏུ་ཕན་པ་ནི། །གྲགས་པའི་རྣམ་པ་བུང་བའི་སྒྲ་སྒྲོགས་—།
