ཚྭ་པ་ཏིརྣ་གེ་ཥཱ་ཛྷཱ་ནྱ་བི་ཤ་ཡཏ། །པྲི་ཐི་བཱི་པྲ་དཱ་ནཾ་ཡཔྲ་ཧ་པྲ་ཏཱི་མེད་པ་ཡིས། །གང་དྲན་ཀླུ་ལ་བཀའ་བཀོད་དང་། །གང་གིས་ས་གཞིའི་སྦྱིན་པ་གསུངས། །དེ་ནས་ཁྱོཏཱ་ད་མ་པྱ་ཐ། །བེ་དུ་ར་ཀླུ་ཤ་མུཏྟིཉྩ་ཀ་ཤ་ལེ་ཡཾ་ཙ་ཤ་ཤཱ་ས་ཡ། །རྟེན་ཅིང་འབྲེལ་འབྱུང་ཡང་། །བེ་དུ་ར་ཡི་ཉོན་མོངས་བཀྲོལ། །གང་གིས་ཀེ་ནེ་ཡ་ལ་བསྟན། །པྲ་ཏྭཱ་སནྣཱབྷི་ཏེཉྩ་ཀྲེ་ཡཤཀྲ་སྱཱ་དྷི་ཏི་པུ་ན། །མ་ཧེནྡཱ་སེ་ནཔྲ་ད་དོ་གང་གིས་བརྒྱ་བྱིན་ལྷང་བ་ལ། །ཉེ་བར་སླར་ཡང་བསྟན་པ་མཛད། །གང་ཞིག་ཀོ་ཤི་ཀ་ཁྱིལ་ནི། །ཀོ་ཤི་ཀཱ་ཡེཀྐི་ཏཾ་ཙ་ཡ། །ཨ་ནུ་གྲ་ཧཾ་སུ་བྷ་དྲ་སྱ་ཡཔརླན་ཡ་དབང་ཆེན་སྡེ་ཡིས་འདོད་པ་བྱིན། །གང་གིས་མཐར་ཐུག་གྱུར་པ་ན། །རབ་བཟང་གི་ནི་རྗེས་ཏེཅུ་དྷཱཏ། །ཧེ་ཏུཏྟ་མཱ་ཁྱོ་བྷག་བཱནྡྷོ་བྷུཀྐུརྦ་ཏ་ཐཱ་ག་ཏ། །ན་ར་བཟུང་མཛད། །སྔོན་གྱི་བཅོམ་ལྡན་དེ་བཞིན་གཤེགས། །རྒྱུ་ཡི་མཆོག་ཏུ་གང་གྱུར་དང་། །དམྱལ་ཀེ་ཀླ་ཤ་ཤཱནྟི་ཡཉྩ་ཀཱ་ར་བྷ་བ་བརྨ་ཎ། །པཱུརྦ་ཀརྨ་པཱུ་ཏི་སཱུ་ནོབར་སྲིད་པའི་གོ་ཆ་ཡིས། །ཉོན་མོངས་གང་གིས་ཞི་བར་མཛད། །གང་ཞིག་སྲས་ནི་སྒྲ་གཅན་འཛིན། །པྲ་ཧ་རཱ་ཧུ་ལ་ཀ་སྱ་ཡ། །གྷོ་རཱཏྟཾ་ཏཱ་རི་ཏོ་ཡེ་ན་སཾ་སཱ་རཱརྨ་སྔོན་གྱི་ལས་དག་སྤྱོད་གསུངས་དང་། །གང་གིས་འཇིགས་རུང་འཁོར་བ་ལས། །སྦྲང་རྩིའི་དབྱངས་དྷ་ར་སྭཱ་རཿ། །པ་གྷུ་ཧི་ཏེ་ཥ་ཡཔྲ་དཱཏ་ཛེཊྛོ་བྷུངྑཀ་པིལྔ་ལ། །པདྨ་ནི་ཡང་དག་བསྒྲལ། ར། ཕན་ཚོལ་གང་གིས་ཕན་པ་བྱིན། །གང་ཞིག་བ་མེན་རྒན་པོར་གྱུར། །པད་ཀཤཱི་ལ་བཱནྡྷོ་བྷཱུཏ་ཙི་ཏྲ་ཙཱ་ནུ་ཤ་ཤཱ་ས་ཡ། །ཨ་ནུ་གྲ་ཧཾ་མཉྩཏ་ནོམ་ཅན་ཁྲིམས་ལྡན་གང་གྱུར་དང་། །ནག་པ་གང་གིས་རྗེས་སུ་བསྐྱངས། རར། ཉ་ཡི་བུ་ནི་ཆོས་སྲེད་ལ། །དྷརྨ་རུ་ཙེཅུ་དྷཱརླ། ་། བི་ད་དྷེ་དྷ་ནི་ཀིཾ་དྷ་ཙྩཾ་སཀྱ་སནྡན་ནེ་གང་གིས་རྗེས་སུ་གཟུང་བ་མཛད། །གང་ཞིག་བདེན་པ་མཐོང་བ་ཡིས། །ནོར་ལྡན་བདག་ྋནི་སྐལ་ན་ཡ། །སཱ་ནཱཾ་སྭཱ་སྲྀཀྵ་དཱ་ནེ་ན་ཡ༔ཤི་བི་སུཀྟ་མ་གྲ་ཧཱིཏ། །མཱ་ཏུཇི་བཟང་བསྒྲུབས། །གང་ཞིག་ཤི་བ་རང་ཉིད་ཀྱིས། །ཤ་ཁྲག་བྱིན་ནས་ལེགས་བཤད་བླངས། །གང་ཞིག་།
