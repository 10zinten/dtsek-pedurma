་དྷ་ནཾ་ག་ཏེ། །ཨེ་ཀ་པུ་ཏྲི་བ་ཛ་ན་ནཱི་ནི་དྷི་ཏུ་ལྕི་ཙ་རཀྵ་ཧཱུྃ། །ཀུ་ལ་ཏྲ་པའི་ཚེ། །བུ་གཅིག་པ་ཉིད་མ་ཡིས་ནི། །གཏེར་དང་མཚུངས་པ་དག་ཏུ་བསྲུངས། །རིགས་ཀྱི་རིམ་པས་མཱ་ག་ཏཱ་ཏ་སྱ་དྲི་པ་ཡཱ་ཏྭཾ་ནི་བརྗེ་ས། །ཨཱ་དི་དེ་ཤ་སཱ་དེ་ཤ་ནཱཾ་སཱལྤ་བི་ཀྲ་ཡེ་འོངས་པ་ཡིས། །གླིང་དུ་འགྲོ་བ་བཟློག་ནས་དེས། །དེ་ལ་འཚོ་ཐབས་རང་ཡུལ་དུ། །འོས་པས་ཆུང་ཛཱི་བི་ཀཾ། །ཀཱཆུ་པ་ཎ་ཙ་ཏུཋ་པཱུབྦཾ་ཡཾ་བི་ཀྲ་ཡོདྨནྣཾ། །ཏ་ཏོ་དྲི་དུ་འཚོང་བ་བསྟན། །དང་པོར་ཚོང་ལས་སྐྱེས་པ་ནི། །ཀརྟ་པ་ཎ་བཞི་དག་དང༌། །དེ་ནས་ཉི་མ་གུ་ཎ་མ་ནྱསྨིན་རྡི་ནེ་པ་ཐ་ཙ་ཏུརྒུ་ཎཾ། །ཏ་ཏཝཱཋ་གུ་ཎཾ་ལཱ་བྷོ་སé་མྨཱགཞན་ལ་ནི། །དེ་ལས་ཉིས་འགྱུར་བཞི་འགྱུར་དང་། །དེ་ནས་བརྒྱད་འགྱུར་དག་ཀྱང་རྙེད། །ཐམས་ཏྲ་རྒྱ་བེ་ད་ཡཏ། །ཀུ་ལོ་ཙི་ཏཱ་དཱི་པ་ཡཱ་ཏྭཾ་ཐུ་ཏྭཱ་བྷུ་ད་ཐ་སོཎྜཱུ་ཀཿ། །ས་ཅད་དེ་ཡིས་མ་ལ་ཕུལ། །དེ་ནས་གླིང་བགྲོད་རིགས་ལ་ནི། །འོས་པ་ཐོས་ནས་འདོད་ལྡན་གྱུར། །རྒྱ་མུ་དྲ་ག་མ་ནཱ་རཾ་བྷ་སཾ་བྷཱ་ར་ར་པ་པཱ་ད་ར། །ས་མཱ་དཱ་བརྨ་མཱ་ཎོ་མཚོར་འགྲོ་བ་རྩོམ་པ་ཡི། །ཚོགས་ལ་ཆགས་ཤིང་གུས་ལྡན་དེ། །མ་ཡིས་བཟློག་པར་གྱུར་མོད་ཀྱང་། །པི་ནཱ་མ་ནྱ་ཏ་ནི་བཏྟཱ་ནཾ། །ཏ་ཏབ་པཱ་ད་བ་ཏི་ཏཱ་ཛ་ན་ནཱི་ཤོ་ཀ་བི་ཧྭ་ལོ། །བཟློག་ཨེཔ་ཉིད་དུ་ཁས་མ་བླངས། །དེ་ནས་མ་ནི་མྱ་ངན་གྱིས། །འཁྲུགས་ཤིང་རྐང་པ་ལ་འགྱེལ་བས། །དུཋཤི་ཡ་མི་བོཏྠརྨ་ཙ་ར་ཎེ་ན་བི་ནིརྨ་ཡོ། །པེ་ཡུ་ཥས་ལི་ཏཱདྷ་གདུག་པ་དེ་ཡི་དཔལ་བཞིན་དུ། །རྐང་པས་རབ་ཏུ་སྤངས་ནས་སོང༌། །དང་པོར་བྱིས་པ་འཁྲུལ་ཞིང་བརྞ་ལ་ལི་ཏཱ་ལཱ་བི་ཤི་ཤུཏྠཆུ་ཏི་པཱི་ཏི་སི་ཏ་མ་ནོ་ར་ཐོཏྠ་བ་ཤ་ཡི་གེ་ཕྱེད་བརྗོད་ཡིད་འོང་བདུད་རྩིའི་ཆར་དག་འབེབས། །དེ་ནས་དུལ་བའི་ཡིད་ལ་རེ་བའི་དགའ་སྟོན་ཏིཛྷཏྟེ་བ་ནི་ཡསྟ་ཏ། །པཉྩཏྤཥ་ཊ་ཡོ་བ་ནོ་བི་ཀ་ཊཱ་ཊངྒཱ་ར་བཱི་དཱི་པ་ཊུ་དུརྒྱས་པ་བརྒྱ་ཡིས་དགའ་བ་འཛིན། །ཕྱི་ནས་མི་བཟད་ལང་ཚོའི་ཚ་བ་དྲག་པོས་བརླང་པོ་དག་ཏུ་སྨྲ་བྱེད་པུ་ཏྲསྤུ་ཊ་ཀཱ་ལ་ཀཱུ་ཊ་ཀ་ཏུ་ཀཀཥྛ་ཀ་རོ་ཏི་བྱ་ཐཱ། །མཁས། །གསལ་བར་ནག་པོ་བརྩེགས་པ་ལྟར་རྩུབ་ངན་པའི་བུ་ཡིས་སྡུག་བསྔལ་གདུང་བར་བྱེད། །།
