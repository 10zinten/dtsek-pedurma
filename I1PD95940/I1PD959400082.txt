པ། པུ་ཎྱ་དྷཱམི་ཏ་བོ་བ་ནེ། །ཏ་པསྭཱ་དྷ་ཡ་ཤཥཱ་ནཾ་མུ་ནཱི་ཏཱཾ་དཱི་ཥ་ཏེ་ཐུབ་པ་གཟི་བརྗིད་འབར་བ་ནི། །ལྔ་བརྒྱ་བསོད་ནམས་གནས་གྱུར་པའི། །དཀའ་ཐུབ་ནགས་ཚལ་འདི་ཛ་ཧཱུཾ། །ཏེ་ཥཱ་མ་དྷ་ཡ་ན་དྷ་ན་ཛ་པ་བི་སྠཱ་བི་དྷཱ་ཡི་ན། །ཨེ་ཏེ་ཀོ་ལཱ་ན་ཡོད། །དེ་དག་གི་ཀློག་བསམ་གཏན་དང་། །ཟློས་པའི་བགེགས་ནི་སྒྲུབ་བྱེད་པ། །ཤིང་གི་ནགས་ལ་ཧ་ལཾ་ཙ་ཀྲུ་ཁ་གསྟ་རུ་བ་ནེ་ས་དཱ། །ཏསྨི་བི་ཧ་ག་སཾ་གྷཱ་ཡ་ཀཎཾ་ཀཱ་ཡ་པ་བྱ་འདི་རྣམས། །རྟག་ཏུ་ཅ་ཅོ་སྒྲོགས་པར་གྱུར། །རྣ་བར་མཆོག་ཏུ་གནོད་པ་དག། ཤིན་ཏུ་ཡང་དག་རཾ་ཐུ་ཏེ། །ཨ་བྷི་སཾ་བནྡྷ་མཱ་ནཱ་ཡ་ཙུ་ཀོ་པ་མུ་ནི་མཎྜ་ལ། །ཏ་དུདྡྷཱ་ཏ་རྒྱས་བྱེད་པ། །བྱ་ཡི་འདུས་པ་དེ་དག་ལ། །ཐུབ་པའི་དཀྱིལ་འཁོར་ཁྲོས་པར་གྱུར། །དེས་བསྐྲུན་ཆེན་མ་ཧཱ་ཤཱ་པ་ཏཱ་པ་བླ་ཥ་ཎ་སཧྥ་ཏ། །ག་ཎེ་ན་པཀཱི་ཎཱ་པཀྵ་བྱ་པོའི་དམོད་པ་ནི། །སྲེག་བྱེད་གདུང་བ་ཐམས་ཅད་དུ། །གནོད་པ་བྱས་པའི་བྱ་རྣམས་ཀྱི། །འདབ་སཱིརྨནྟ་ཀྲྀ་ཏཱ་ཤ་ཧཱུཾ། །ཏ་ཨེ་ཏེ་བི་ཧ་གཱ༔པཀྵ་ར་ཧི་ཏཱ་ཀྲྀ༔་བརྟི་་་གཤོག་སྐད་ཅིག་གིས་ཟགས་གྱུར། །བྱ་ནི་འདི་རྣམས་དེ་དག་སྟེ། །སྡུག་བསྔལ་གྱིས་འཚོ་འདབ་གཤོག་ན༑། སཱཏ་བི་བཀྟ་ཨི་བ་བ་ནེ་ཤཱནྟཧྥ་ར་ཎ་ཙཱ་རི་ཎ། །མ་ཧཱ་མཱ་ཏེ་ན་བྲལ། །ཁྱོད་ཀྱི་མི་མཐུན་ཕྱོགས་བཞིན་དུ། །ངལ་ཞིང་ནགས་སུ་རྐང་པས་རྒྱུ། །བློན་པོ་ཆེན་པོས་ཀ་ཐི་ཏཾ་ནི་ཤ་མྱེ་ཏད་མ་ཧཱི་བ་ཏི། །ཨུ་བཱ་ཙ་ཀ་རུ་ཎཱ་ཀཱནྟ་སྟཔ་ཤཱ་པེ་བརྗོད་དེ་དག། ཐོས་ནས་ས་ཡི་བདག་པོ་ནི། །བྱ་རྣམས་དམོད་པས་གདུངས་པ་ལ། །སྙིང་རྗེས་རབ་ཏུ་ན་པཀྴི་ཏཱཾ། །ཨ་ཧོ་ཏེ་ཛཔ་རི་ཎ་ཏཾ་ཤཱནྟཱ་ནཱ་མ་པི་ཀཱ་ན་ནེ། །ཨོཾ་གཱ་རཱ་ཎཱ་ནོན་པས་སྨྲས། །ཀྱེ་མ་ཐུབ་པ་སོལ་བ་རྣམས། །ཞི་བར་གྱུར་ཀྱང་ནགས་སུ་ནི། །ཡོངས་སུ་གྱུར་པའི་མུ་ནཱི་ཏཱཾ་ཙ་ད་ཧ་ཏེ་བཱ་ནི་བཱ་རི་ཏཾ། །མི་བྱཱ་ཏ་པ་སཱི་བྷིཀིཾ་ཏིསཱ་སུ་མེ་དག་གིས། །མི་ཟློག་སྲེག་པར་བྱེད་པ་ཉིད། །རང་ཉིད་བདེ་ཕྱིར་གང་གི་ཡིད། །ཁྲོ་བས་གདུང་ལ་ཁཱ་ཡ་ན་ཡེཀྲྀཏ། །མ་ན་ས་ཀོ་པ་ཏལྟ་སྱ་པ་རི་ཥ་ཀཀྐ་ཚྭ་བྷུ་བྷི། །པྲ་སནྣཱ་བྷིམྨ་བཟོད་པ་ཡི། །ཆུ་ཡིས་ཡོངས་སུ་བྲན་མ་བྱས། །ལོག་པའི་དཀའ་ཐུབ་ཅན་དེས་ཅི། །རབ་དང་བློ་དང་།
