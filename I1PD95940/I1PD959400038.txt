དཀྵི་ཎཱ་བྷི་མ་ཏཱ་གུ་རོ། །རཱ་ཛན་ཛ་ཡ་པྲ་བྷཱ་དེ་བཱི་དཱི་ཡ་ཏྭཾ་ཡ་དི་ཤཱ་ཀྱ་སློབ་དཔོན་མངོན་པར་འདོད་པའི་ཡོན། །རྒྱལ་པོ་ལྷ་མོ་རྒྱལ་བའི་འོད། །གལ་ཏེ་ཕོད་ན་གནང་བར་ཏེ༑། ཨི་ཏཱུཀྵེ་མུ་ནི་ཤི་བྱེ་ཎ་ས་ཧ་སེ་བ་མ་ཧཱི་པ་ཏེ། །སྣེ་ཧ་དཱ་མཛོད། །ཐུབ་པའི་སློབ་མས་དེ་བརྗོད་པའི། །འཕྲལ་ཉིད་ས་ཡི་བདག་པོ་ལ། །མཛའ་དང་སྦྱིན་ན་ར་སུ་བིདྷ་དཱི་དྷཱ་བྷཱུ་ཏ་མ་བྷུདྨ་ན། །ས་ཛ་གཱ་ད་དཱི་ཛཾ་དནྟ་པའི་རོར་གཅགས་པ། །རྣམ་པ་གཉིས་ཤིག་®བྱུང་བར་གྱུར། །སོ་ཡི་འོད་ཟེར་མདུན་དཀྲིགས་པས། །ཚོ་ཏི་ཥཱ་གྲ་བི་པཱ་རི་ཎཱ། །ཀྲྀ་་ཏྭཾ་ད་ཡི་ནཾ་སཱཙྪ་བ་ས་སཱ་ཙྪ་ད་སྙིང་དུ་སྡུག་པའི་བཙུན་མོ་དག། དཀར་བའི་གོས་ཀྱིས་གཡོགས་པ་བཞིན། །དེ་ཡིས་གཉིས་སྐྱེས་@དག་ཡིནྣ་བ། །ཨ་བི་ཙཱརྱ་མ་ཡཱ་དེ་བ་ཡ་མཱིསི་ཏཾ་ཏ་བ་ཡད་གུ་རོ། །བི་ཡོ་ག་ལ་སྨྲས། །འབྲལ་བས་འཇིགས་པའི་སེམས་ལ་ཡང་། །།ང་ཡིས་བདེན་པར་མི་རྩི་ཞིང་། །རྣམ་པར་མི་ཙ་ཀི་ཏཾ་ཙེ་ཏས་ཏྱཾ་ན་ག་ཎ་ཡཱ་མྱ་ཧཾ། ་། ཨི་ཏུཏྟཱ་ཧྲི་ད་ཡ་ད་ཡི་ཏཱཾ་རཱ་ཛཱ་དཔྱད་ཁྱོད་ཀྱི་ནི། །བླ་མ་གང་བཞེད་བདག་གིས་སྦྱིན། །དེ་སྐད་བརྗོད་ནས་རྒྱལ་བ་ཡིས། །བཙུན་མོ་རཱ་ཛཱི་བ་ལོ་ཙ་ནཾ། །ས་དཱ་ཧྲི་ད་ཡ་སཾ་སཀྵ་ཛཱི་བ་བྲྀཏྟི་མི་བཱ་པ་རཱི། །ནཱི་བཱ་པདྨའི་སྤྱན་ལྡན་མ། །རྟག་ཏུ་སྙིང་ལ་ཡང་དག་གཅགས། །འཚོ་སྲོག་གཞན་པ་ལྟ་བུ་བོས། །འབྲལ་རི་ཏེ་པི་གུ་རུ་ཎཱ་པི་ཡོ་ག་བྱ་པ་ནཱཀྐི་ནཱ། །ནི་ཥིདྡྷོ་སྐྱ་ཏི་སྲྀདྷེ་ན་བས་གདུང་བའི་མེ་དག་ནི། །བླ་མར་གྱུར་པས་བཟློག་གྱུར་ཅིང༌། །མཛའ་བ་ཤིན་ཏུ་འཕེལ་བ་ཡི། །སྣེ་ཧེ་ན་སྨ་ར་བནྡྷ་ནཱ། །པྲ་ད་དོ་མུ་ནི་ཤི་ཥཱ་ཡ་ས་ཧ་སཱ་ཧ་རི་ཎི་མི་འདོད་པའི་གཉེན་གྱིས་བཀག་གྱུར་ཀྱང༌། །ཐུབ་པའི་སློབ་ལ་རབ་ཏུ་བྱིན། །འཕྲལ་ལ་རི་དྭགས་མོ་བཞིན་བ༑། ཀི་མེ་ཏ་དི་ཏི་ས་ཡཱ་ས་སཾ་ཏཱ་ས་ཏ་ར་ལེཀ་ཏྭཾ། །དཏྟཱ་ཡཱམྦྷ་ག་ཤཱི་ལི་ན་དུ། །འདི་ནི་ཅི་ཞེས་ངལ་བ་དང་། །སྐྲག་བཅས་མིག་ནི་རིག་རིག་གྱུར། །ས་སྤྱོད་གཏོང་བའི་ངང་ཅན་པྲ་ཡཱ་ཡཱ༔པྲ་ཐི་བཱི་བྷུ་ཛཱ། །ཙ་ཀམྨེ་ཏུ་ག་བྷི་ཏེ་བ་བྷུ་མི་ལོ་ལཱབྷི་གྱིས། །དགའ་མ་བཏང་བར་གྱུར་པ་ལ། །གཏོང་བས་འཇིགས་པ་བཞིན་དུ་ས། །ཆུ་གཏེར་སྐེ་རགས་—།
