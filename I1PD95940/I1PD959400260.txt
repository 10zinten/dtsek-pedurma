ལུ་ལཱ་བྷི་ཀཱ་མ། །སཾ་སཱ་ར་བཱ་མསུ་ཀྲྀ་ཏཱ་བྷི་རཱ་མ༔མ་ནོ་མ་ལོཟི་དང་ལེགས་བྱས་ལ། །མངོན་པར་དགའ་ཞིང་ཡིད་ཀྱི་དྲི་མ་རྣམས་ཀྱི་དག་གི། །ཁོན་གྱི་རྡུལ་དང་བྲལ་བ་ར་ར་ཛོ་བི་རཱ་མ། །པུ་རཱ་ནཾ་གཱ་མ་ནུཏྟིརྨ་ནཱ་གཱ་ནཾ་ཕ་ཎ་སེ་ཏུ་ནཱ། །དགེ་ལེགས་བསྐྱེད་པ་ཡིན། །སྔོན་ཚེ་ཀླུ་རྣམས་གདེངས་ཀ་ཡིས། །ཟམ་པས་གངྒཱ་ལེགས་པར་བསྒྲལ། །བྷ་ག་བཱ་ན་སུ་ག་ཏ་སུ་པྲ་པ་བྷིཀྵ་ན་བྷ་ཥ་ཏ། །ཨེ་ཧ་རཏྭཱ་མ་བཅོམ་ལྡན་བདེ་གཤེགས་ཕ་རོལ་ཏུ། །བྱོན་ནས་དགེ་སློང་རྣམས་ལ་གསུངས། །འདིར་སྔོན་རིན་ཆེན་རང་ཡ་པཱུཐཾ་ཡུ་པོ་བྷུ་དངྒྷ་ཏདྷུ་ཏི། །དན་ཡཱ་མི་ཏ་མ་ཏྲི་བ་ཏྲཤྩ་བབཞིན་གྱི། །མཆོད་སྡོང་རྨད་བྱུང་མཛེས་པ་བྱུང་། །གལ་ཏེ་དགེ་མཚན་ཁྱེད་བལྟ་ན། །འདི་ཉིད་དུ་ཀོ་ཏུ་ཀཾ་ཡ་དི། །ཨི་ཏྱཱཀྟ་བྷ་ག་བཱན་བྷཱུ་མོ་པཱ་ཎི་ནཱ་དི་བྱ་ལཀྵ་ཎཱ། །དེ་བདག་གིས་བསྟན། །ཞེས་གསུངས་ས་ལ་བཅོམ་ལྡན་གྱི། །མཚན་མཆོག་ལྡན་པའི་ཕྱག་དག་གིས། །སྤྲོཌཱ་ནཱ་ག་ག་ཎོ་ཏ་ཀྱི༈་རཏྭཱ་ཡུ་པྲ་མ་དན་ཡཏ། །ཏཾ་དྲྀབྷྱཱ་བྷིཀྵ་རེག་པས་ཀླུ་ཡི་ཚོགས་རྣམས་ཀྱིས། །རིན་ཅེན་ེམཆོད་སྡོང་བླངས་ནས་བསྟན། །དགེ་སློང་ཐམས་ཅད་པསརྦ་ནནི་ནིརྨི་མེ་ཀྵ་ཆ་ཤྲཱི་རཾ། །བ་བྷཱུ་པུཉྩི་ཏྲ་ལི་ཁི་ཏ་ཨི་བ་ནི་དེ་མཐོང་ནས། །ཡུན་རིང་མིག་ནི་མི་འཛུམ་ཞིང་། །ལུས་ནི་གཡོ་བ་མེད་པ་དག། རི་མོར་བྲིས་པ་ལ་ནི་གྲ་ཧ། །ཏཥ་ཐཱ་མ་ཐཱ་ཏིཔྲཋཔྲ་བཱ་ཙ་བྷ་ག་བཱན་པུ་ན། །ཨ་ཏཱི་བཞིན་དུ་གྱུར། །དེ་དག་གིས་དྲིས་བཅོམ་ལྡན་གྱིས། །དེ་གཏམ་སླར་ཡང་རབ་གསུངས་ཏེ། །འདས་ཏཱ་ནཱ་ག་ཏ་ཙཱ་ནཾ་དནྟཱ་ལོ་ཀིཀི་རནྟི་པ། །དེ་བ་པུ་ཏྲ་པུ་རཱ་ཀཉྩི་ཏ་དང་མ་འོངས་མཁྱེན་པ་ནི། །ཚེམས་ཀྱི་སྣང་བས་འགྲེམས་པ་བཞིན། །སྔོན་ཚེ་ལྷ་ཡི་བུ་འགའ་ཞིག། ཀཱ་ལེ་སཱརྒ་པ་རི་ཙུ་ཏ། །མ་ཧཱ་པྲ་ཎཱ་ད་ནཱ་མཱ་བྷཱུ་ཏ་ནི་པ་ཏིཤ་ཀྲ་ཤཱ་པ་དུས་ཀྱི་མཐོ་རིས་ལས་ཡོངས་འཕོས། །བརྒྱ་བྱིན་བཀའ་ལས་མི་ཡི་བདག། མང་པོས་བཀུར་བ་ཞེས་པར་ནཱཏ༑། དྷརྨ་བྲྀཏྟཱ་ནུ་ས་ར་ཎཱ་སྨ་ར་ཎ་ཡ་མ་ཧཱི་ཏ་ལེ། །ཨུ་ཙི་ཏཾ་ལཀྵ་ཎཾ་ཀིཉྩི་ཏ་གྱུར། །ས་ཡི་གཞི་ལ་ཆོས་ཀྱི་ནི། །སྤྱོད་པ་རྗེས་འཇུག་དྲན་པའི་སླད། །དེ་ཡིས་མཆོད་སྦྱིན་བརྒྱ་པ་ལ། །།
