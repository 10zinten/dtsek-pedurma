ཛཱ་ཤྲཱི་ཧཱབྷི་ཁྱེ་བྷ་ག་བཱནྡྷ་པྲྀ་ཏཔུ་རཱ། །ཀ་ཊ་ཀེ་གྲོ་དྷ་ཀུ་ཊ་སྱ་བི་ཛ་པོའི་ཁབ་ཅེས་པར། །ས་འཛིན་བྱ་རྒོད་ཕུང་པོ་ཡི། །ངོས་ལ་སྔོན་ཚེ་བཅོམ་ལྡན་འདས། །དེ་བཞིན་ཧཱ་ར་ཏཐཱ་ག་ཏ། ་། ཏསྨིནྣ་བ་ས་རེ་རཱ་ཛཱ་བིམྦི་ས་རཿཔུ་ཏ་པྲ་ཡ། །གཤེགས་པ་བཞུགས་པར་གྱུར། །སྐབས་དེར་བུ་ལ་བྱམས་པ་ཡི། །རྒྱལ་པོ་གཟུགས་ཅན་སྙིང་པོ་ཉིད། །ཨ་ཛཱ་ཏ་ཤ་ཏུ་ཎ་ཏ་ཏྲ་པུ་ཧྲེ་ཎ་ཀྲུ་ར་ཀཱ་རི་ཎཱ། །སུ་ཧྲི་དཔཱ་པ་ཀ་ཕྱེ་བ་དེ་དེ་ཡི་བུ་ནི་མ་སྐྱེས་དགྲ། །གདུག་པ་བྱེད་པར་གྱུར་པ་ཡིས། །གྲོགས་པོ་སྡིག་པ་དང་ལྡན་པ། །ལྷ་སྦྱིན་བ་དཏྟ་སྱ་སམྨ་ཏཾ། །གྷོ་རཱནྡྷ་བནྡྷ་ནཱ་གཱ་རཾ་ནིསཾ་ཙཱ་རཾ་པྲ་བེ་ཤི་ཏ། །སནྡྷགྱི་ནི་གྲོས་བཞིན་དུ། །འཇིགས་རུང་མུན་ཅན་བཙོན་ཁང་ནི། །རྒྱུ་བ་མེད་པར་རབ་ཏུ་བཅུག། བཀྲེས་པྲ་བེ་ཤི་ཏཾ་ཏ་སྱ་བནྟ་ནེ་གཱ་ཌྷ་བྷོ་ཛ་ནཾ། །ཛཱ་ཏྭཱ་ས་ཏཥྛ་ཡཱ་ཀཾ་ཀཱི་པས་ཉེན་པ་དེ་ལ་ནི། །བཙུན་མོས་ཟས་དག་བཙོན་ཁང་དུ། །གསང་ནས་རབ་ཏུ་འཇུག་པར་བྱས། །གླུདྨ་མ་སྱ་རྩ་བཱ་ར་ཡཏ། །།རླུ་གྷཱཀྲྀ་ཤོ་ནི་མ་ལི་ནསོ་བྷ་བཛྲ་ཐི་བཱི་པ་ཏི། །：དེས་གསད་འདོད་དེས་ཤེས་ནས་བཟློག། ས་ཡི་བདག་དེ་ཤིན་ཏུ་རིད། །སྔོ་ཞིང་དྲི་མ་ལྡན་པར་གྱུར། །ཨ་ཀ་ལ་ཀཱ་ལ་མེ་གྷཎྜཀྲྀཀྵ་ས་ག་ཨི་བེནུ་ཡ། །ས་བ་ས་སི་དུས་མིན་སྤྲིན་ནག་གིས་གཟིར་བའི། །ནག་པོའི་ཕྱོགས་ཀྱི་ཟླ་བ་བཞིན། །ཕལ་ཆེར་མཉེས་གཤིན་སེམས་ཏཱ་བཔྲ་ཡཔེ་ཤ་ལ་ཙེ་ཏ་ཧཱུཾ། །ཀ་རོ་ཏཱ་ལི་གཱ་ནཾ་པྲ་ཝཱ་གཱ་ཌྷ་པྲ་ཎ་ལྡན་ལ། །མཆོག་ཏུ་རྒུད་པའི་མཛའ་མོ་ནི། །གདུང་བཅས་སྲེད་དང་ལྡན་བཞིན་དུ། །ཤིན་ཏུ་དམ་ཡི་ནཱི་བ་མཏ། །ས་ས་མུདྲི་བྱ་ཤོ་ཀཱཎྜསུ་ག་ཏཱ་དྷུ་ཥི་ཏྭཾ་དི་ཤཾ། །པར་འཁྱུད་པར་བྱེད། །བདེ་གཤེགས་བཞུགས་པའི་ཕྱོགས་སུ་ནི། །མྱ་ངན་གྱིས་གཟིར་དེར་ཕྱོགས་ཏེ། ཀྲྀ་ཏཱཉྫ་ལིརྣ་ཏ་ཤི་རཱཀྵ་མ་སཱ་ར་མ་བྷཱ་ཥ་ཏ། །ན་མསྟུ་ཧུཾ་བྷ་ག་བ་ཐལ་མོ་སྦྱར་ཏེ་སྤྱིས་བཏུད་ནས། །སྐད་གདངས་ཞན་པར་རབ་སྨྲས་པ། །དམན་པ་འདོན་པ་ལ་བརྩོན་ཏེ་མ་ཧཱཧཱ་ཡ་མ་ཧཱདྷ་ཏེ། །དཱི་ནོདྷ་ར་ཎ་ས་ནཱདྷ་སམྱཀཾ་བོ་དྷི་པ། །ཡང་དག་རྫོགས་པའི་བྱང་ཆུབ་སེམས། །རིན་ཆེན་དགྲ་བཅོམ་ཆེན་པོ་ཉིད། །བཅོམ་ལྡན་ཁྱོད་ལ་།
