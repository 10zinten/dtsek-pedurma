།ནཾ་སཔྟ་ཏི་ཏ་མཔལླ་ཝ། །་་་་་་་་་་་་་་ཀྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་དྲུག་ཅུ་རྩ་དགུ་པའོ། །།།བྷ་ཏྟི་པྲ་བནྟི་ཏ་ཛཾ་ནོ་དི་ཏ་བཱ་པ་ནཱ་ནཾ་ཏེ་ཊཱ་ཛ་ཡ་ཏུ་བྷི་མ་ཏསུ་གང་གི་གྲགས་པའི་མཚན་ཉིད་ཁྱད་པར་རྣམ་པར་བཀོད་པ་ཡིས། །ས་གཞི་བསོད་ནམས་མཆོག་གི་ཀྲྀ་ཏཱ་བྷི་ཡོ་ག། །པུཙྪཱ་ཏི་པུ་ཅུ་ཏ་ར་ཏཱ་མུ་པ་ཡ་ཏི་པྲ་བྷཾ། བསོད་ནམས་ལྡན་པར་ཉེར་བསྐྲུན་ཅིང་། །རྒྱལ་བ་ལ་ནི་གུས་པ་རབ་བསྒྲུབས་བག་ཆགས་བསྐྱེད་བྱས་སྩོལ། །ཡཏིནྟི་ལཀྵ་ཎ་བི་ཤེ་ཥ་ནི་བེ་ཤ་ནེ་ན། །མ་དྷནྟི་ཀཱ་བྷི་དྷ་བྷིཀྵ་རཱ་ནནྟ་དེ་དག་རྣམས་ཀྱི་མངོན་འདོད་ལེགས་བྱས་སྦྱོར་བ་རྒྱལ་གྱུར་ཅིག། དགེ་སློང་ཉི་གུང་ཞེས་བྱ་བ། །བླ་མ་སྱ་གུ་རོ་ག་རཱ། །བུདྡྷ་ཤཱ་ས་ན་མཱ་དྷཱནྟཾ་ཡ་ཡོ་ཀཤྲཱི་ར་མཎྜ་ལཾ། །ཏནྡྲ་དགའ་བོའི་ཚིག་གིས་ནི། །ཁ་ཆེའི་དཀྱིལ་འཁོར་དག་ཏུ་ནི། །སངས་རྒྱས་བསྟན་པ་འཛིན་དུ་སོང་། །ཀླུ་གཱ་དྷཎྜི་ཏཾ་ཙྪཱ་ཏྭཱ་བྷཱི་ར་ས་ཏཱས་མཱ་དྷི་ནཱ། །ས་སཾ་ཀོ་བྷཾ་བི་ད་དྷེ་ཡིས་བྱིན་བརླབས་དེ་ཤེས་ནས། །ས་གཞི་འདར་བྱེད་དེ་ཡིས་ནི། །སྙིང་སྟོབས་བརྟན་ལྡན་ཏིང་འཛིན་ན་ནཱ་གཱ་ནཾ་ཀཾ་བི་ཏཱ་བ་ནི། །ཤསྨཱཀྐི་བྲྀཥྛ་བུཏྲིཥྛ་ཏི་ར་མན་བི་གྱིས། །ཀླུ་རྣམས་ཀུན་ཏུ་བསྐྱོད་པ་བསྒྲུབས། །རྣམ་འགྱུར་ཁྲོ་ལྡན་དེ་དག་གིས། །མཚོན་དང་མེ་ཡི་ཀཱ་རི་བྷི། །པྲ་བྷཱ་བེ་ན་བྷ་བཏྟ་སྱ་མཱདྷི་པཏྲ་ལཱ་བ་ལཱི། །ཏ་ཏནྟཾ་ཛ་ཆར་སྤྲོས་པ། །མཐུ་ཡིས་དེ་ཡི་སྤྱི་བོར་ནི། །པདྨ་ཨུཏྤལ་འཕྲེང་བར་གྱུར། །དེ་ནས་དེ་ཡི་བརྩོན་ག་དུནཱ་གཱསྟིདྷཱརྨོ་རྐཆུ་བི་སྨི་ཏཱ། །ཡཱ་བརླམཾ་ཀ་བནྡྷོ་ཏྲ་དེ་ཤཔྟ་བཛྲ་འགྲུས་ཀྱིས། །རབ་ཏུ་ཡ་མཚན་ཀླུས་དེ་ལ། །སྨྲས་པ་ཇི་སྲིད་སྐྱིལ་ཀྲུང་བཅས། །དེ་སྲིད་ཡུལ་འདིར་ཤེ་ཏ་བ། །ཨུཀེ་ཏི་ཏསྨི་པÉ་ཀ་བནྡྷ་ཏུ་ལཾ་པྲ་མཱ་ཎ་ཏ། །ན་བ་ཁྱོད་ཀྱི་གནས། །ཞེས་པ་བརྗོད་ནས་དེ་ལ་ནི། །སྐྱིལ་ཀྲུང་ཡངས་པར་བཅིངས་པ་ཡི། །ཚད་དང་དྲོ་ཎ་མུ་ཁཱ་ཡཱ་ལཾ་པྲ་ད་དུཤུ་མཎྜ་ལཾ། །ས་ཏ་ཏྲ་ན་ག་ར་གྲྭ་མ་པནི་མཚུངས་པ་སྟོང་པ་ཡི། །དཀྱིལ་འཁོར་དག་ནི་རབ་ཏུ་བྱིན། །དེར་ནི་དེ་ཡི་གྲོང་ཁྱེར་དང་། །གྲོང་གི་།
