ལོ་ཀེ། ་། དཎྜ་མེ་བཱནྟཱ་ཡེ་རཱ་ཛན་པུ་བི་ཙཱརྨ་བི་དྷི་ཡ་ཏྭཾ། པཱུ་ཡ་མི་བཟོད། །།རྒྱལ་པོ་ཆད་པ་བདག་འཚལ་ལོ། །ཤིན་ཏུ་དཔྱད་ནས་བསྒྲུབ་པར་མཛོད། །དག་བྱེད་ཏེ་པཱ་བཱ་ཀེ་ནེ་བ་རཱ་ཛ་དཎེ་ན་ཀིཝི་ཏཱི། ཎི། ཨི་ཏཱ་གྲ་བྱེ་ཎ་བ་ཧུ་ཀྱིས་བཞིན་རྒྱལ་པོ་ཡི། །།ཆད་པས་སྡིག་ཅན་གཙང་བར་འགྱུར། །ཞེས་དེ་འབད་པས་ལན་མང་དུ། །ཤོ་ཡཱ་ཙ་མཱ་ནཾ་སཱ་ཤཱ་ས་ནཾ། ་། ཏ་མུ་བཱ་ཙཱ་ནུ་ཛོ་རཱ་ཛོ་པྲ་ཏཀཏྟཱསྨི་ཡཀྵ་མཾ། ཌོ། རང་གི་ཆད་པ་དོན་གཉེར་བ། །དེ་ལ་ནུ་བོ་རྒྱལ་པོས་སྨྲས། །ནང་པར་བདག་གིས་གང་བཟོད་བྱ། །ཨསྨིན་པྲ་ཏཱི་་ཏཱ་བདྷཱ་བཱ་ནུ་དྱ་ནུ་ཀཱ་ན་ནེ། ་ཌ། ཨི་ཏྱཱཀྟ་པོ་ར་ཀཱནཱ་ཡ་རེ་ཞིག་ཁྱོད་ནི་སྐྱེད་ཚལ་གྱི་ཕྱེ། །ནགས་འདི་ཉིད་དུ་གནས་པར་མཛོད། །ཅེས་བརྗོད་གྲོང་པའི་བྱ་བ་ལ། །ནིརླ་ཡོ་བྲི་ཐི་བཱི་པ་ཏི། །ཏ་ཏྲ་པྲ་ཏཱི་ཀ་མཱ་ཎེ་ཐ་མུ་ནོ་ནི་ཡ་མཱ་ནི༔་ལེ། །རཱ་ས་ཡི་བདག་པོ་ཕྱི་རོལ་སོང་། །དེ་ནས་དེར་ནི་ཐུབ་པ་དག། ངེས་པར་གཡོ་མེད་གནས་པའི་ཚེ། །རྒྱལ་ཛཱ་བིསྨྲ་ཏ་ཏཎྚི་ཏྱཥཉྷི་ར་བྷཱུ་ཡ་ཡོ་གི་ནི། །ཀྲྀ་ཤཱ་ཀཱ་རནྣི་རཱ་ཧཱ་ར་མ་པོ་དེ་ཡི་བྱ་བ་བརྗེདི། །ཉི་མ་དྲུག་ནས་མངོན་པར་འོངས། །དེ་ནས་རྒྱལ་པོ་ཕུ་བོ་ནི། །།ཟས་མེད་ཐ་དྲྀཏྟཱ་གྲ་ཛི་ནཱཾ་པ། ཀྵ། ནིནྡནཱ་བུདྷི་སམྨོ་ཧཾ་ཙཀྲ་ཏ་སྱ་པྲ་པཱ་ད་ནཾ། །།བཱཏྨ་རིད་པའི་རྣམ་པ་མཐོང༌། །ཀུན་རྨོངས་རང་གི་བློ་སྨོད་ཅིང་། །དེ་ལ་རབ་ཏུ་དང་བར་བྱ། །དེ་ནི་ཎིས་ཧི་ཏ་ཀྲྀ་ཏཱ་ས་ཏཏྟཱ་པ་པྲ་མཱཛྫ་ནཾ། །ཥཎྜི་ན་ཀླུ་ཤ་དཱ་ནེ་ན་ཡ་ཏེ་བྲམ་ཟེ་དང་བཅས་པས། །དེ་ཡི་སྡིག་པ་དག་བྱས་ནས། །ཉིན་དྲུག་ཉོན་མོངས་བྱིན་པ་ཡིས། །།དེ་སོང་ཏསྨིནྣ་ཏེ་སྱ་ཏི། ་། ཨ་ཧཾ་ས་སཱུརྨོ་རཱ་ཛཱནྟིཉྩནྡོ་རཱ་ཛཱ་པྱ་ཡཾ་ཤི་བུ། ་། ཛཱ་ཏསཾ་ཚེ་ན་རབ་ཏུ་གདུངས། །ང་ཉིད་རྒྱལ་པོའི་དྲང་སྲོང་ཉི༄༅། །བྱིས་པ་འདི་ཡང་རྒྱལ་པོ་ཟླ། །ཉིན་དྲུག་བཏྠ་རིཥབྷིཥི་ཌ་ན་གླེ་ཤ་ཀྲྀརླུ་ན། །ཨཱ་བཱི་ར་བྷཱནྟ་བྷ་དཱ་ཥཱ་ཧ་རི་ཐུབ་པ་ཉོན་མོངས་པར། ྋ། བྱས་པས་ལོ་དྲུག་དག་ནས་སྐྱེས། །།བཟང་མོ་ཞེས་བྱ་བ་རྫི་མོ། །རྒན་མོ་ཅུ་སུ་ཏ་ཡཱ་ས་ཧ། །ན་ག་རཱ་བྷི་མུ་ཁྲི་དྲྀདྷཱ་ཏ་ཀྲྀ་བི་ཀྲེ་ཏུ་མཱ་ཡ་ཡོ། །ཏཏུ་ཏཱ་གྲོང་ཁྱེར་ལ་ཕྱོགས་ཏེ། །བུ་མོ་རི་དྭགས་མོ་དག་དང་། །འགྲོགས་ནས་དར་བ་འཚོང་དུ་སོང་། །དེ་ཡི་བུ་།
