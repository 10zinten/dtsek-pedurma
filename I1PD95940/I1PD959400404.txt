ད་སན་པ་ཤ་མ་ནཾ་བ་ནཾ། །ཨ་བི་ཤཱནྟ་སྱ་སཾ་སཱ་རཱ་པ་ཐ་པནྡྲ་སྱ་ཡོངས་གདུངས་ཞི་བའི་ནགས་སུ་འགྲོ། །རྟག་ཏུ་གནོད་པའི་ལུས་འདི་ཡང་། །།ངལ་གསོ་མེད་པ་དུཊྚ་ཧ། །ཀཱ་ཡོ་སྱ་ཡཾ་ས་ད་པཱ་ཡཀཾ་པུ་ནསྲི་ཐི་བཱི་བྷ་ར། །ཨི་ཏི་པུ་ཏྲ་འཁོར་བ་ཡི། །ལམ་གྱི་འགྲོན་པོས་བསྐུར་དཀའ་ནི། །ས་ཡི་ཁུར་ལྟ་སླར་ཡང་ཅི། །ཞེས་པ་མཆོག་བ་ཙཐུ་ཏྭཱ་བྷཱ་པ་ཏིཛྲ་ཤ་མ་སྲི་ཡཾ། །པྲ་པྲ་ཛུ་ཤལྟ་ཙ་ཀི་ཏསོདྲེ་གསྟ་མ་ཏུ་མི་སྙན་པ། །བུ་ཡི་ཚིག་ཐོས་ས་ཡི་བདག། རབ་ཏུ་འབྱུང་བའི་སྒྲ་ཡིས་འཇིགས། །མྱ་ངན་ལྡན་པས་བྷཱ་ཥ་ཏ། །ཨ་སྱ་ཐཾ་ཤ་སྱ་མ་ཧཱ་ཏསཱ་མཱ་ཛུ་སུ་ཙ་བྲིདྡྷཱ་ཡེ། །ཨཱ་ཤཱ་དེ་ལ་སྨྲས། །ཆེན་པོར་གྱུར་པའི་རིགས་འདི་དང༌། །ཡང་དག་རྒྱལ་སྲིད་འཕེལ་བའི་སླད། །རྒན་པོ་ནི་བདྷཱ་བྲིདྷེ་ན་ཏཱ་ཡི་པུ་ཏྲ་མ་ཡཱ་པ་རཾ། །བཏྲ་སཾ་ཀལྤ་བྷངྒོ་མེ་ན་བདག་གིས་བུ་ཁྱོད་ལ། །རེ་བ་མཆོག་ཏུ་རྣམ་པར་བཅིངས། །མཉེས་གཤིན་བདག་གིས་བརྟག་པ་ནི། །ཀཱ་ལེ་ཀརྟུ་མ་དྷ་སི། །ཨི་དནྟ་བ་ནཙྪཱ་ཡཾ་ཡོ་བ་ནཾ་ན་བ་ནོ་ཙི་ཏཾ། །དུས་མིན་ཉམས་པར་བྱ་མི་འོས། །ནགས་སུ་ཁྱོད་ཀྱི་ལང་ཚོ་འདི། །གསར་དུ་མཛེས་པ་འོས་མ་ཡིན། །སམྨན་བྷྱཱ་ས་ཡུཏྟཱ་ནཾ་ཤཁཱ་ནཱ་སཱ་དྷུ་དན་ནེ། །ཛི་ཏེནྡྲི་ཡཱ་ཎཾ་སརྦ་ཏྲ་ཚིགས་དམ་པའི་སྔགས་ཀྱི་གོམས་པར་ལྡན། །ལེགས་པའི་ལྟ་བ་ལ་ཆགས་པ། །དབང་པོ་ཐུལ་བའི་མི་བདག་ནི། །པཱ་ནཱ་མ་བ་ནཾ་ཏ་པ། །སཱ་པ་དེ་པི་ས་རོ་ཛ་སྱ་ནིསངྒ་ས་ལི་ལ་སྦི་ཏེ། །ཀུན་ཏུ་ནགས་མེད་དཀའ་ཐུབ་ཡིན། །རང་གནས་ཡིན་ཡང་མཚོ་སྐྱེས་ནི། །གོས་པ་མེད་པར་ཆུ་ལ་གནས། །དྲྀཥྛ་བ་ནེ་སྱ་ཤོ་ཀ་སྱ་ལ་ལ་ནཱ་ཙ་རཱ་ཎཱ་ཧ་ཏི། །སྭ་གེ་ཧ་སུ་ལ་བྷིནྡོ་ནགས་ནང་ཡིན་ཡང་མྱ་ངན་མེད། །བུད་མེད་རྐང་པས་བསྣུན་པ་མཐོང་། །རང་ཁྱིམ་རྙེད་སླའི་ལོངས་གིརཱ་བཛྲཏྟི་བི་སཱུ་ཙི་ཀཱ། །ཏཱ་བ་དེ་ཏེ་པ་རི་ཀཾ་ཤཀྱནྟེ་བི་ཥ་སྤྱོད་ཀྱིས། །ཇི་སྲིད་མིག་ནི་རྨུས་ཆུང་ཁེབ། །དེ་སྲིད་ཡུལ་ནི་འདི་དག་རྣམས། །སྐད་ཅིག་ཡོངས་སུ་ཡཱཀྵ་ཎཾ། །སུ་ཁ་མནྟཔ་རི་ཏུ་ཛྲ་སཱ་ཛ་ན་ཁཾ་ཧ་ནི་རྒ་ཏ། །ཨ་བྱཱས་གཏང་བར་ནུས། །རང་གི་སྐྱེ་བོ་ནང་གི་ནི། །བདེ་བ་ཡོངས་བཏང་ཁྱིམ་ལས་འཐོན། །ཉམས་མྱོང་།
