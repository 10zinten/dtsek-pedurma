ནཱ་ཐུ་ཀ་ཎོཏིརྩྭ་ལ་ཏེ་བ་བྷུ་ཏ་ཥཀྵ་དཱ། །ཨ་ཐ་པྲ་བ་ཧ་ནཾ་བྷེ་ཛེ་རྐང་དྲུག་པས་བརྒྱན་བཞིན། །མིག་དམན་དང་བཅས་མཆི་ཐིགས་རྒྱས། །དེ་ནས་དེ་དང་ལྷན་ཅིག་ཏུ། །སཱཟླ་ནཱ་ཐསྟ་ཡཱ་པ་ཧ། །ཏཱི་བྲ་ཛྲཏྟཱ་ཁཾ་ཧཱི་ཏཱ་ནཾ་ཧསྟ་པཱ་ཏྲ་མ་ཧོ་དོན་མཐུན་མགོན་གྱི་གྲུ་གཟིངས་བསྟེན། །སྲེད་པ་དྲག་པོ་འཛིན་རྣམས་ཀྱིས། །ཆུ་གཏེར་ཆེན་པོ་ལག་ད་དྷི། །ཏ་སྱ་ཛཱ་ཡཱ་ས་ཁ་སྣ་སོ་ཀམྨཱ་བཱ་ཏཱ་ནུ་བཏྟིཾ་ན། །ཨ་བྷ་ཙུ་ཏ་པའི་སྣོད། །ལས་ཀྱི་རླུང་གི་རྗེས་ཞུགས་པ། །ཆུང་མའི་གྲོགས་ལྡན་དེ་ཡི་གཟིངས། །ཡིད་ལ་རེ་བ་པྲ་བ་ཧ་ཎཾ་ས་མ་ནོ་ར་ཐ་ཛཱི་བི་ཏཾ། །ཏ་ཏཕ་ལ་ཀཱ་བཱསྟཱ་བྷ་ག་ཤེ་ཥཱཙྩ་སྲོག་དང་བཅས། །ཆུ་གཏེར་དུ་ནི་ཞིག་པར་གྱུར། །དེ་ནས་ལས་ཀྱི་ལོངས་སྤྱོད་ཀྱི། །ལྷག་མས་ཤིང་དུམ་ཀརྨ་ཎ། །ཀ་ཤེ་རུ་དཱི་པི་མཱ་སཱ་རྱ་ཏཏྠཏི་ན་བྱ་བ་ཏ། །ཨ་ནཱ་ཐཱ་ཏ་ཏྲ་ཤོ་བཟུང་གྱུར་པས། །ཀ་སེ་རུ་ཡི་གླིང་ཕྱིན་ནས། །དེ་ཡི་ཆུང་མ་ཤི་མ་གྱུར། །དེར་ནི་བསམ་པ་ཉམས་ཙནྟི་བི་ཧཾ་གཔུ་རུ་ཥཱ་ཀྲྀ་ཏི། །ཏཱ་མཱ་པ་པཱ་ད་དཱིཎ་ཀཾ་སུ་པཎ་གྱུར་ཅིང་། །མགོན་མེད་མྱ་ངན་བྱེད་དེ་དང་། །འདབ་བཟང་རིགས་ལས་སྐྱེས་པའི་བྱ། །སྐྱེས་བུའི་ཀུ་ལ་སཾ་བྷ་བ། །ས་ཀཱནྟཾ་སུ་མུ་ཁོ་ནཱ་མ་ཏཱ་མུ་བ་ཙ་རུ་ཙཱ་ཀྲྀ་ཏ། །དབྱིབས་ཅན་ཕྲད་པར་གྱུར། །མཛེས་པས་ཕྲོགས་པ་བཞིན་བཟང་མིང་། །དེ་ཡིས་མཛེས་མ་དེ་ལ་སྨྲས། །ས་མཱ་ཤཱ་སི་ཧི་ལོ་ལཱ་ཀྐི་ནིརྡ་ཡོ་ཡཾ་ཏ་པཱ་ཤ་ཡ། །དེ་བྱེ་ཡཾ་སུ་བྷ་གཱ་བྷཱུ་མིག་ནི་གཡོ་མ་ཡིད་བརྟན་མཛོད། །གནས་འདིར་ཁྱོད་ལ་འཇིགས་པ་མེད། །འདི་ནི་སྐལ་བཟང་མཆོག་མི་བ་ཡཾ་ཏཱཏ་པྲ་ཎ་ཡི་ཥི་ཎ། །པུ་ན་ཏཱ་སི་ཀ་ལྟ་ཎི་གྷོ་རོ་གི་ས། །བདག་ནི་ཁྱོད་ཀྱི་འདོད་པ་འཚོལ༄༅། །ཆུ་སྲིན་འབྱུང་གནས་འདི་འཇིགས་རུང་། །དགེ་མ་ཁྱོད་ཡཾ་མ་ཀ་རཱ་ཀ་ར། །ཨི་ཏྱཱཀྵཏེ་ན་ཤ་ན་ཀིརཱི་ཏཱ་རཏྣ་ལ་ཡཾ་ཁཾ་ཧཾ། །སཾ་ནི་ཡང་སྐྱེས་སོ། །དེ་ཡིས་འདི་བརྗོད་དལ་བུ་ཡིས། །རིན་ཆེན་རང་བཞིན་ཁང་པར་ཕྱིན། །མངལ་ནི་པུཎ་ག་ཏ་ན་ཡཾ་ཙཱ་རུ་རཱུ་པ་མ་པཱུ་ཏ་སཱ། །བཛྷ་མཱ་ན་ཤི་ཤོ་ཏ་སྨིན་སཱ་ཡོངས་སུ་རྫོགས་དེ་ལ། །གཟུགས་མཛེས་བུ་དག་བཙས་པར་གྱུར། །བྱིས་པ་དེ་ནི་ཆེར་སྐྱེས་ཚེ། །རིམ་།
