རྩྭ་སཾ་དྷཱ་ཥི་ཎཾ་སཱ་དྷོ་ན་ཡུཏྟཾ་བྲ་ཧྥ་ཙཱ་རི་ཎ། །ཏ་བ་པུཥྛ་པྲ་དཱ་ནེ་ན་པ་ཚངས་སྤྱོད་བུ་མོ་ལ། །ཕེབས་པར་སྨྲ་བའི་རིགས་མ་ཡིན། །ཁྱོད་ལ་མེ་ཏོག་རབ་བྱིན་པས། །སྐྱེ་བྷཱརཱམཱ་མ་ནྱ་ཛརྨ་སུ། །ཨི་ཏྲེ་བ་པྲ་ཎི་དྷཱ་ནེ་ན་ད་དཱ་མྱེ་ཏཱ་ནི་ནཱ་རྱ་ཐཱ། །བ་གཞན་དུ་ཆུང་མར་བདག། གྱུར་ཅིག་ཅེས་པའི་སྨོན་ལམ་གྱིས། །འདི་དག་སྟེར་གྱི་གཞན་དུ་མིན། །ཨེ་བ་མནྟི་ཏི་ཏེ་ནོ་ཀེ་པཉྩ་པདྨ་ནི་སཱ་ད་དོ། །ཏསྨི་སྭ་ཡཾ་ཙ་པཱུ་ཛཱ་ཡི་ཛ་གྲྭ་དེ་ལྟར་གྱུར་ཅེས་དེས་བརྗོད་ཚེ། །པདྨ་ལྔ་ནི་དེ་ལ་བྱིན། །རང་ཡང་མཆོད་ཕྱིར་དེ་ཡིས་ནི། །པདྨོ་གཉིས་ཧ་ཀ་མ་ལ་དྭ་ཡཾ། །ཨ་ཐ་དཱི་པ་ཀྱི་ཏི་པ་ཏིབཱ་པ་བཉྩ་མ་ཧཱི་པ་ཏི། ་། བྷ་ག་དག་རབ་ཏུ་བཟུང་། །དེ་ནས་ས་བདག་མར་མེ་དང༌། །ས་བདག་ནོར་ལྷའི་བུ་དག་ཀྱང༌། །བཅོམ་ལྡན་བནྟཾ་ས་མཱ་ཡནྟཾ་བྷཀྟ་བ་ཏྱ་བྷི་ཛགླ་ཏུ། །ཏ་སྱ་རནྣཱ་ཤ་ལོ་ཀཾ་ལེགས་པར་བྱོན་པ་ཅན། །གུས་པས་མངོན་དུ་འོངས་པར་གྱུར། །དེ་ནས་མར་མེས་དེ་ཡི་གདུགས། །ཀཾ་དཱི་པཉྩ་ཏྲ་མ་ཐཱ་གྲ་ཧཱིཏ། །ཙཱ་མ་རཾ་ཙནྡྲ་དྷ་བ་ལཾ་བཱ་ས་བོ་བ་སུ་དྷཱ་དྷི་པ། །རིན་ཆེན་ཐུར་མས་མཚན་པ་བཟུང་། །ནོར་འཛིན་བདག་པོ་ནོར་ལྷའི་བུས། །ཟླ་བ་ལྟར་དཀར་རྔ་ཡབ་བོ། །སུ་ར་སིདྷནྟི་གནྡྷབྷེཔཱུ་ཛྲ་མཱ་ན་སྱ་སརྦ་ཏ། །པུ་ར་པྲ་བེ་ཤེ་ཏ་ལྷ་དང་གྲུབ་པ་དྲང་སྲོང་དང་། །དྲི་ཟ་མཆོད་དང་གྲོང་ཁྱེར་དུ། །རབ་ཏུ་བཞུགས་ཚེ་སེམས་ཅན་རྩྭ་སཱི་ཏ་སརྦ་སཏྭཱ་སུ་ཁཏྟོ་བ། །ནི་བི་ཌེ་ཛ་ན་སམཱ་ཏེ་ཛི་ན་སནྡན་ནོཏུ་ཀུན། །ཀུན་ཏུ་བདེ་བའི་དགའ་སྟོན་རྒྱས། །སྐྱེ་བོའི་ཚོགས་ནི་སྟུག་པོ་ལ། །རྒྱལ་བ་ལྟ་བར་འདོད་པ་ཀ༈། སུ་མ་ཏིསྟནྡྲ་བྷཱ་བེ་ན་ཀ་ཐཾ་ཙིཏྟཱརྒ་མཱལྟ་བཱན། །ཀི་ཏུ་ཡིས༄༅། །བློ་གྲོས་བཟང་གི་དེ་མཐུ་ལས། །ཇི་ཞིག་ལྟར་སྟེས་ལམ་ཐོབ་གྱུར། །སྟོན་པའི་ཞབས་ལ་པདྨ་ཤཱསྟུཉྩ་ར་ཎེ་ཀ་མ་ལཱ་ནི་ཀྲྀ་ཏཱལྔ་ལི། །ས་ཏཾ་བ་བནྡེ་སཱ་ནནྡཀྴི་ཏོ་ཀཱིཙྪཱ་ཛ་དག། རབ་ཏུ་འཕངས་ནས་ཐལ་སྦྱར་ཏེ། །ཀུན་དགའ་ཅན་དེས་རལ་པའི་ཁུར། །ས་ལ་བཀྲམ་ནས་ཊཱ་བྷཱ་ར། །ཏ་སྱ་པདྨ་ནི་བི་པུ་་སུ་ཀྲྀ་ཏཱ་ནཱི་བ་ཏཏྟཱ་ཎཏ། །ཨ་བཱ་པ་དེར་ཕྱག་འཚལ། །དེ་ཡི་མདུན་ན་ཡངས་པ་ཡི། །ལེགས་བྱས་བཞིན་དུ་པདྨོ་རྣམས། །དེ་མདུན་ཞལ་གྱི་།
