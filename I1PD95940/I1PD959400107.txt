ཥོ་པྱ་མོ་ཧཉྩ་མ་ཧཱདྨ་ནཾ། །དཱ་ན་ཤཱི་ལ་ཀྵ་མ་བཱིརྨ་དྷ་ཡིས། །བདག་ཉིད་ཆེན་པོ་གང་དག་དང་། །སྦྱིན་དང་ཚུལ་ཁྲིམས་བཟོད་བརྩོན་འགྲུས། །བསམ་ན་པྲཛྙཱ་ཛུ་ཏཾ་ས་དཱ། །ཨུ་པཱ་ཡ་པྲ་ཎི་དྷི་ཛྙཱ་ན་བ་ལི་རཱ་ཐི་ཏ་ཙེ་ཏ་གཏན་ཤེས་རབ་རྟག་བསྟེན་ཞིང༌། །ཐབས་དང་སྨོན་ལམ་ཡེ་ཤེས་དང་། །སྟོབས་ཀྱི་རབ་ཏུ་བསྟེན་པའི་སྭཱ༑། པ་རི་ཏཱ་ཎི་ཀ་བཱི་རཱ་ནཾ་ཡ་དེ་པཱ་དྭ་ཡ་བ་དི་ནི། །བི་བྱཱ་ཐུགས། །ཡོངས་སྐྱོབས་དཔའ་བོ་གཅིག་པུ་དང་། །རྟག་ཏུ་གཉིས་མེད་གསུང་ཉིད་དང་། །རིག་པ་ཏྲ་ཡ་པྲ་དཱིདྷཱ་ནཾ་ཙ་ཏུཤྲཱི་མ་ལ་ཤཱ་ལི་ནཱཾ། །པཉྩ་སྐནྡྷ་བི་མུཀྟ་ཚྭ་ཥ་ཎཱ་གསུམ་གྱི་གསལ་བ་དང་། །དྲི་མེད་བཞི་དང་ལེགས་པར་ལྡན། །ཕུང་པོ་ལྔ་ལས་རྣམ་གྲོལ་ཞིང་། །སྐྱེ་ཡ་ཏ་ན་བྷེ་དི་ནཾ། །སསྟ་བོ་་ག་ཡུཀྵ་ནཱ་མཱརཱཏྟཾ་གོ་པ་དེ་་་་མཆེད་དྲུག་ནི་འབྱེད་པ་དང་། །བྱང་ཆུབ་ཡན་ལག་བདུན་དང་ལྡན། །འཕགས་ལམ་ཡན་ལག་བརྒྱད་ཤི་ནཾ། །ན་བ་སཾ་ཡོ་ག་ཧཱི་ནཱ་ནཱཾ་ཏེ་ཏཱཾ་ད་ཤ་བ་ལཱཏྨ་ནཾ། །ཀིཾ་བཱསྐྱ་ཉེར་སྟོན། །ཀུན་སྦྱོར་དགུས་དམན་རྒྱལ་བ་ནི། །སྟོབས་བཅུའི་བདག་ཉིད་ལྡན་དག་གིས། །འཇིག་རྟེན་བི་དི་ཏཾ་ལོ་ཀེ་ཛི་ནཱ་ནཾ་ཛ་ནེ་ཙེཤྲཱི་ཏཾ། །ཏ་ཏསྟེ་ཥཱ་ཏི་ཀ་རུ་ཙྪཱཙྪ་ར་ཎཱ་སྐྱེ་བོའི་སྤྱོད་ཚུལ་ལ། །མི་མཁྱེན་པ་ནི་ཅི་ཞིག་ཡོད། །དེ་ནས་ཤིན་ཏུ་སྙིང་རྗེ་ཡིས། །ཞབས་ལ་མགོས་ལཱི་ན་མཱུདྷ་སུ། །ཏ་ཐེཀླུ་བཱ་ཙ་བྷ་ག་བཱི་སྟཉྙོཛྫཱ་པ་ནི་མནྡཱ་ཎེ། །ཏིསཎཾ། གཏུགས་དེ་དག་ལ། །དེ་བཞིན་ནོ་ཞེས་བཅོམ་ལྡན་གྱིས། །དེ་ཡི་ཟས་ལ་ཞལ་གྱིས་བཞེས། །དེ་མཐོང་དན་ན་གླི་ཎཱ་ཀིཝི་ཥས་མ་མཟི་ཏཾ། །བྷིཀྵ་སཾ་བྷཾདྲྀ་ཏོབྷོ་ཛི་བི་དྷི་སྡིག་པ་ཟད་གྱུར་པ། །དེ་དག་གིས་ནི་མཉམ་དུ་ཕུལ། །དགེ་སློང་ཚོགས་ཀྱིས་བསྐོར་བ་ཡིས། །ཟས་ཀུན་བཏྠརྦ་མཱ་ད་དེ། །ཏ་ཏསྟད་པྲ་ཎི་དྷཱ་ནེ་ན་ཛྷཱ་ན་ལོ་ཀ་ཤ་ལཱ་ཀ་ཡཱ། །ཏེ་ས་མུནྡྲི་ཆོ་ག་བཞིན་དུ་བླངས། །དེ་འོག་དེ་ཡི་སྨོན་ལམ་ནི། །ཡེ་ཤེས་སྣང་བའི་ཐུར་མ་ཡིས། །དེ་དག་མིག་ནི་ལི་ཏ་པྲ་ཤཔྲ་ཀཱ་ཤཾ་ད་དྲྀ་ཤུཔ་དཾ། །ཏེ་པ་བྱསྟི་བྲ་བི་རཱ་གྱི་པ་རི་པ་ཡང་དག་ཕྱེ། །རབ་ཏུ་གསལ་བའི་གནས་མཐོང་གྱུར། །དེ་དག་འཕྲལ་ལ་དྲག་པོར་ནི། །ཆགས་བྲལ་།
