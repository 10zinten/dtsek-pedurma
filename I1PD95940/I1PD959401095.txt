ཡཾ་ཀ་ན་ཀ་བནྡཱ་བ་དཱ་ན་ཥ་ཌ་དྷི་ཀ་ཤ་ཏ་ཏ་མཔལླ་བ། །།།འཁྲི་ཤིང་ལས་གསེར་གྱི་གོ་ཆའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་བརྒྱ་རྩ་དྲུག་པའོ། །།།གུ་ཎོ་སྤྲོ་ཧཱ་ན་དྲ་པི་ཎེ་ཀ་དཱ་ཙིབློ་གེ་བྷི་ཡོ་གཔྲ་ས་གྷོ་ན་བྷོ་གེ། །བསོད་ནམས་རབ་འབར་བསམ་པ་དྲི་མེད་རྣམས། །ཡོན་ཏན་ལ་ཆགས་ནམ་ཡང་ནོར་ལ་མིན། །པུ་སྐྱོ་ཛྭ་ལཱ་ནཾ་བི་ཤ་དཱ་ཤ་ཡཱ་ནི་ར་ཇོ་བི་རཱ་གཱ་བྷི་ར་ཏིརྣ་ཀཱ་མེ། །པུ་རྣལ་འབྱོར་ལ་སྦྱོར་ལོངས་སྤྱོད་དག་ལ་མིན། །རྡུལ་གྱི་ཆགས་བྲལ་ལ་དགའ་འདོད་ལ་མིན། །བསོད་ཅུ་པྲ་བྷཱ་ཐཾ་བྷ་ག་བཱ་་ཤི་ལསྟ་ཐཱ་ག་ཏ། །བ་བྷཱ་ཥ་བྷིཀྵ་བྷིནམས་རི་བོ་བཅོམ་ལྡན་འདས། །དེ་བཞིན་གཤེགས་པ་བསོད་ནམས་མཐུ། །དེ་ཡི་བསོད་ནམས་ལ་གུས་དྲྀཏྠསྟཧྥུ་ཙྪཱ་ད་ར་བིསྨི་ཏེ། །བཱ་རཱ་ཎ་སིཾ་མ་ཧཱི་རཱ་ཛཱི་བྲཧྨ་དཏྟེ་པྲ་ཤཱ་པས། །ཡ་མཚན་དགེ་སློང་གིས་དྲིས་གསུངས། །བཱ་རཱ་ཎ་སཱིར་རྒྱལ་པོ་ནི། །ཚངས་བྱིན་དག་གིས་ས་ས་ཏི། །ཤུདྡྷོ་ད་ནེ་གྲོ་ཧ་པ་ཏིབྦ་བྷཱུ་བ་དྷཱ་ན་དོ་པ་མ། །ཏ་སྣ་བསྲུངས་ཚེ། །ཁྱིམ་བདག་ཟས་ནི་གཙང་མ་དག། ནོར་སྦྱིན་དང་ནི་མཚུངས་པ་བྱུང་། །རྒྱ་མཚོར་རྨི་ཀལྤ་བྲྀཀྵ་སྱ་ས་མུ་དྲ་ག་མ་ནཱརྫི་ཏེ། །མ་ཎི་མོཀྟི་ཀ་སཾ་བྷཱ་རིའགྲོ་བས་རབ་བསྒྲུབས་པ། །ནོར་བུ་མུ་ཏིག་ཚོགས་རྣམས་ཀྱིས། །སློང་བ་དག་གི་དཔག་བསམ་ཤིང་། །པ་རི་པཱུརྞ་མ་བྷུདྡཾ་ཧཾ། །རཏྣ་དྷཱི་པཱནྟ་རཱ་ཡཱ་ཏས་ཀ་དཱ་ཙིནྡྲ་ཧཱི་པ་ཏི། །དེ་ཡི་ཁྱིམ་ནི་ཡོངས་གང་གྱུར། །ནམ་ཞིག་རིན་ཆེན་གླིང་གཞན་ནས། །འོངས་པ་དེ་ནི་ས་བདག་ལ། །དྲཧཱུཾ་ཡ་ཡོ་ས་མཱ་དཱ་ཡ་མུཀྐཱཧཱ་ར་མུ་པཱ་ལ་ནཾ། །པྲ་ཏྱངྒ་ཏེ་ན་པྲ་ཎ་ཡཱརྨ་ནི་ཏསྐྱེས་སུ་མུ་ཏིག་དོ་ཤལ་དག། ཁྱེར་ནས་བལྟ་རུ་སོང་བར་གྱུར། །ས་བདག་སྔོན་དུ་ལངས་བྱས་ནས། །ས་མ་ཧཱི་བྷུ་ཛཱ། །ཏཱ་རཾ་ཧཱ་རཾ་ད་དོ་ཏསྨི་ལཱི་ལཱ་ཧཱ་ས་མི་བ་ཤི་ཡ། །གུས་པས་མཆོད་པ་དེ་ཡིས་ནི། །དཔལ་གྱི་རྩེ་དགའ་དགོད་པ་བཞིན། །གསལ་བའི་དོ་ཤལ་དེ་ལ་ཕུལ། །ས་ར་སྨི་སཱུ་ཏྲ་སཾ་སཏྟ་ཏཱ་ར་ཀཱ་ནི་ཀ་རོ་པ་མ། །བྷུ་མི་བྷཌྷརྒུ་ཎཱི་ཧཱ་འོད་ལྡན་སྐུད་པར་ཡང་དག་ཆགས། །རྒྱུ་སྐར་གྱི་ནི་ཚོགས་ལྟ་བུ། །ཡོན་ཏན་ལྡན་པའི་དོ་ཤལ་གྱིས། །—།
