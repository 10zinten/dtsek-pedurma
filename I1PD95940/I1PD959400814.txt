སཱི་པི་པི་ཏི་ཥ་ཎྜི། །པ་རི་པྲཏྟེ་ན་སཾ་ཙྪནྣཱ་པ་ཚྭ་ས་མྲ་ག་ཙསྨྲ་རྒོད་གཟུགས་ཅན་ཡང་དག་མཐོང་། །ཤ་ལྡན་རི་དྭགས་པགས་པ་ནི། །ཡོངས་སུ་བསྒྱུར་བས་ཀུན་དུ་ཎཱ༑། ས་པ་ད་མཱུ་ལེ་ཏ་སཱུ་དྲེས་སྦོ་ནིཤྩ་ལ་བི་གྲ་ཧ། །མཾ་ས་ལུཔྟ་ཏ་མུཎི་གཡོགས། །རི་བོ་དེ་ཡི་ཞབས་ཀྱི་རྩརི། །དེ་ཡི་ལུས་ནི་གཡོ་མེད་འཁོད། །མཚན་རྒྱུ་བྱ་རྒོད་གཟུགས་མྱ་ཤྲཱི་དྷ་རཱུ་པཱ་ནི་ཤཱ་ཙ་རཱི། ་། ནི་ད་དྷེ་ཤི་ཁ་ར་ས་གྲེ་བྷོཀྟབྷི་ཅན་ནི། །ཤ་ལ་ཆགས་པས་དེ་བླངས་ནས། །འཇིགས་རུང་གཟུགས་ཀྱིས་བཟའ་བའི་ཕྱིར། །རི་རྩེའི་ཥ་ཎ་བི་གྲ་ཧཱ། །མྲྀ་ག་ཙརྨ་སྟ་མུཙླི་ཚུ་ཏཱ་ནི་ཧ་ཏྱ་པ་བཱིརྨ་བཱན། །སྟེང་དུ་རྣམ་པར་བཀོད། །བརྩོན་འགྲུས་ཅན་གྱི་རི་དྭགས་ཀྱི། །པགས་པ་བཏང་ནས་དེ་བསད་དེ། །ནཱི་རནྟ་ཁ་དི་ར་ཀཱིཙྩཾ་པྲ་པ་ཁ་དི་ར་བྷཱུ་དྷ་རཾ། །པྲ་བི་བྱ་ཏདུ་ཧཱུྃ་ལེ་བྷེ་སེང་ལྡེང་སྟུག་པོས་ཁེངས་པ་ཡི། །སེང་ལྡེང་ས་འཛིན་དག་ཏུ་ཕྱིན། །དེ་ཡི་ཕུག་ཞུགས་རྡོ་བསལ་བས། །ཤི་ལཾ་བྱ་སྱ་མ་ཧོ་ཥ་དྷི། །ཤཱི་ཏཱ་ཏ་པ་ཏ་མ༔སརྦ་རཱཀྵ་སཱ་དི་བྷ་ཡཱ་པ་ཧཾ། །གྲང་དང་ཚ་དང་མུན་པ་སྤྲུལ། །སྲིན་པོ་ལ་སོགས་འཇིགས་པ་དག། འཕྲོག་པའི་སྨན་ནི་ཆེན་པོ་ཐོབ། །ས་ཡནྡྲ་སཧྥ་ཏོ་པྲ་པྱ་པཾ་གྷཎྜི༔པྲ་ཎ་ཧ་རི་ཎོ། །ཡནྡྲ་ཀཱི་ལཾ་ཤ་རཱ་འཁྲུལ་འཁོར་རི་གཉིས་རབ་བསྡེབས་པས། །སྲོག་ནི་འཕྲོག་པར་དེ་ཡིས་སླེབ། །འཁྲུལ་འཁོར་ཕུར་པ་གྲེ་ཎ་ཙྪི་ཏྭཱ་ཙཀྲེ་བི་ནི༔་ལོ། །ཡནྣཱ་ཀཱི་ལ་ས་མུཙྪེ་དིཆནྣཱ་དྷཱ་མདའ་རྩེ་ཡིས། །བཅད་ནས་གཡོ་བ་མེད་པར་བྱས། །འཁྲུལ་འཁོར་ཕུར་པ་བཅད་རྣམས་ཀྱིས། །འཁྲུལ་རཾ་བི་དརླ་པ། །ཙྪེ་ད་ནཾ་ཡནྣཱ་ཙཀྲ་སྱ་ཡན་ཡུཀྟ་ཏ་ཐཱ་ཡ་འཁོར་སྒོ་ནི་དེས་བཅོམ་ཞིང་། །འཁྲུལ་འཁོར་འཁོར་ལོ་བཅད་དེ་བཞིན། །ལྕགས་ཀྱི་འཁྲུལ་འཁོར་སྦྱར་སོ༑། ཏཱི་བྲ་སྲ་ཧཱ་རོ་པུ་རུ་ཥོ་ཡནྡྲ་མ་ཎོ་ཙ་དུས་ཧོ། །ཡནྡོ་གྲ་བའོ། །དྲག་ཏུ་སྣུན་བྱེད་སྐྱེས་བུ་གཉིས། །བཟོད་དཀའ་ལུག་གི་འཁྲུལ་འཁོར་གཉིས། །དེ་བཞིན་ཆུ་དནྟ་ནི་ེ་ཡོ་ཏ་ཐཱ་མ་ཀ་ར་རཱ་ཀྵ་སོ། །གྷོ་རཱནྡྷ་ཀཱ་ར་གཾ་བྷི་གུ་སྲིན་སྲིན་པོའི་སོའི། །འཁྲུལ་འཁོར་དྲག་པོས་འཇོམས་པ་གཉིས། །འཇིགས་རུང་མུན་པ་ཟབ་པའི་ཕུག། །
