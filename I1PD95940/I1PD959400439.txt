ན་བུདྡྷ་སྭ་བྷི་བནྡ་ནཾ། །ཨི་ཏི་ཏེ་ནོ་ཀཱ་མཱ་ཀནྡྷུ་བི་ན་ཡ་ཏྟོ་མཱུ་ཙ་ཏུ། །མངོན་པར་ཕྱག་འཚལ་ཅི་མི་བྱེད། །ཅེས་དེས་བརྗོད་ཐོས་དུལ་བ་ཡིས། །དེ་དག་གིས་ནི་དེ་ལ་སྨྲས། །པྲ་ས་དཀྲི་ཡ་ཏཱ་མཱརྨ་བྷ་ག་བརྡན་ནེ་ན་ནོ། །ཨི་ཏི་བྲུ་བཱ་ཎོ་ནཱ་འཕགས་པ་བཅོམ་ལྡན་བསྟན་པ་ཡིས། །བདག་ཅག་ལ་ནི་བཀའ་དྲིན་མཛོད། །ཅེས་སྨྲ་ཀླུ་ཡི་དབང་པོ་གེནྡོ་ནཱི་ཏཱ་བྷ་ག་བནྟོནྟི་ཀཾ། །ས་ཏཛྲ་ཏྟཱན་མ་བེ་རྱ་པྲ་ཎ་མྱ་ཏ་མུ་པཱ་དག། བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་ཁྱེར་ནས། །དེ་དག་ལོ་རྒྱུས་བཤད་ནས་དེ། །དེ་ལ་ཕྱག་འཚལ་ཉེ་བར་བི་ཤཏ། །བྷ་ག་བཱ་ན་ཐ་ནཱ་གནྡོ་བ་བྷཱ་ཥ་ཤ་ར་ཎ་ག་ཏོ། །པྲ་ཎཱ་མཉྩ་འཁོད། །དེ་ནས་ཀླུ་དབང་སྐྱབ་སོང་ཞིང་། །རིན་ཅེན་འོད་ཀྱིས་ས་གཞི་དག། བརྒྱན་པས་ཕྱག་ཀྲ་ཏུཎོ་རཏྣ་ཏྲ་བྷ་བྷུ་ཥི་ཏ་བྷུ་ཏ་ལོ། །ཤི་ཀྵ་པ་དཱ་ནི་སཾ་པྲ་སྱ་འཚལ་བྱས་པ་ལ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། །བསླབ་པའི་གནས་ནི་ཡང་དག་ཐོབ། །སཉྫ་བྷཱུ་ཏཱ་བྷཱ་ཡ་པྲ་དོ། །ཤ་ར་ཎ་པཱསྟི་སམྦ་པཱ་ད་དྷུ་ནཱ་ནིཌྷ་ཡོ་ཡུ་ཏཱཾ། །འབྱུང་པོ་ཀུན་ལ་འཇིགས་མེད་སྟེར། །སྐྱབས་ཀྱི་གྲོགས་ནི་ཐོབ་པ་ལས། །ད་ནི་ཁྱེད་ཅག་འཇིགས་པ་མེད། །ཨི་ཏྭཱ་ལོ་ཀ་ན་མ་ཏྲེ་ཎ་དོ་ཥ་དྭེ་ཥ་བི་བརྫི་ཏོ། །ཀྲོ་ཏོ་བྷ་ག་བ་ཏཱ་ཅེས་ནི་ཡང་དག་གཟིགས་ཙམ་གྱིས། །དེ་དག་སྐྱོན་དང་སྡང་བ་དང་། །བྲལ་བ་བཅོམ་ལྡན་འདས་ས་མྱཀྵགླ་ཏུཙྩཾ་པྲ་ཎ་མྱ་ཏོ། །སཾ་དན་ནེ་ནི་བ་མ་ཧཱ་ཤ་ཡཱ་ནཾ། ཀྱིས་མཛད། །དེ་ལ་ཕྱག་འཚལ་སོང་བར་གྱུར། །བསམ་པ་ཆེ་རྣམས་ཡན་ལག་མཐོང་ཉིད་ན། །སྡང་པྲ་བྷཱ་ས་དེ་ཤེ་ན་ཤ་རཱི་ར་ལགྣི། །ཧི་པྲ་ཨ་པི་དྭེ་ཥ་བི་ཏོཥ་མ་ཏཱབྷཱཤ་མ་མྲྀ་ཏིབའི་དུག་ཚན་གདུང་བས་གཙེས་པ་ཡང་། །འོད་ཀྱི་ཟོལ་གྱིས་ལུས་ལ་ཆགས་པ་ཡིས། །ཞི་བའི་བདུད་རྩི་ཤྲཱི་ཏ་ལ་ཀཾ་པྲ་ཛནྟི། །ཏཁྲུབྦ་ཛརྨ་བྲྀཏྟཱནྟཾ་བིཀྵ་བྷིནྡྷ་ག་བཱམྦི་ནཱ། །པྲ་བསིལ་བ་ཉིད་དུ་འགྱུར། །དེ་ཡི་སྔོན་གྱི་བྱུང་བའི་མཐའ། །མཐུ་ཡི་ཡ་མཚན་དགེ་སློང་གིས། །དྲིས་བྷཱ་བ་བིསྨྲ་ཡཱཏ་པྲཏྟཱ་སབྷཱ་དནི་ཛ་གཱ་ད་ཏཱན། །ཀྲི་ཀིརྩྭ་མ་པུ་རཱ་ཤིང་རྒྱལ་བ་བཅོམ་ལྡན་འདས། །ཐམས་ཅད་གཟིགས་པས་དེ་ལ་གསུངས། །བཱ་རཱ་ཎ་སཱིར་མི་ཡི་དབང་། །།
