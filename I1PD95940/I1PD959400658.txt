ཏཏྟཱ་པཾ་བ་ཧུ་ཛརྨ་སུ། །བྷུཀྟ་ཅུ་ཀོ་ཊ་ར་ཡ་བཱ་ཧཱ་རཿཤེ་ཥ་ཎ་ཀརྨ་ཎ། །།ཡ་དཱ་སྐྱེ་བ་མང་པོར་དེ་ཡི་སྡིག། སྤྱད་ཅིང་ད་ལྟ་ལས་ལྷག་གིས། །ཀོ་ཊ་ར་ཡི་ནས་ཀྱི་ཟས། །གང་ཚེ་བདག་ཧ་མ་བྷ་ཊཾ་པུཏྤ་མུཏྟ་རོ་ནཱ་མ་མཱ་ཎ་བ། །པྲ་ཀུ་ལ་སཱུ་པ་བཱ་དེ་ན་མ་ཡཱསྟ་མ་ཤུ་སྔོན་བྲམ་ཟེ་ཡི། །ཁྱེའུ་ཨུད་ཏ་ཕྱིར་ཞེས་གྱུར། །དེའི་ཚེ་བདག་གི་རིགས་མཆོག་ལ། །སྨད་པ་ཡིས་ནི་གྷཾ་ཏ་དཱ། །ཥཊནཱ་ཎྱ་མུ་ནཱ་ཏེ་ན་ཀརྨ་ཎཱ་དུཥ་རམྨ་ཡཱ། །ཙ་རི་ཏཱཾ་རཀྵ་མི་དགེ་ཐོབ། །ལས་དེ་ད་ལྟ་ལོ་དྲུག་ཏུ། །དཀའ་བ་སྤྱད་པ་བདག་གིས་བསྒྲུབས། །དེ་ལས་ལྷག་པ་ཎཱདྲོ་དྷེན་པྲཏྟཾ་ཏཱ་དྷི་ཀཾ་ཏ་ཏ། །པུ་རཱ་བྷ་བཛྲ་ཧ་པ་ཏིཌྷ་ན་བཱརཱ་མ་བྱང་ཆུབ་ཀྱིས། །བསྲུངས་པ་ཡིས་ནི་ཐོབ་མ་གྱུར། །ཁྱིམ་བདག་ནོར་ལྡན་ཞེས་བྱ་བ། །ཀར་པ་ཊར་ནི་ཀཏྤ་ཊེ། །བཱི་མཱནྟཱ་མ་སུ་ཏསྟ་སྱ་བ་བྷུ་བཱ་སཱསྦྲ་བི་ཀྲ་ཧ། །བི་བྱ་སྟིཀཱ་མུ་སྔོན་བྱུང་སྟེ། །དཔལ་ལྡན་ཞེས་བྱ་དེ་ཡི་བུ། །ལུས་ནི་རྣལ་གནས་མིན་པར་གྱུར། །དེ་ནི་སྨད་པ་ཁ་བའི་ཁོ་ནཱ་མ་ཏཾ་བ་ཧུ་དྲ་བི་ན་ཤ་ཡཱ། །སཱ་ྩནྡྲི་ནཱ་ཡ་ཏཏྲི་ཏཱ་དཏྟཾ་ཡཱམྨེ་ན་ཀིཉྩ་ན། །བཞིན། །ཞེས་པ་ནོར་མང་རེ་བ་ཡིས། །རྣལ་གནས་བྱས་ཀྱང་དེ་ཡི་ཕས། །དེ་ལ་ཅི་ཡང་བྱིན་མ་གྱུར། །ཀཱ་ལེ་ནཱ་ཏུ་ར་་ཡཱ་ཏསཱཙྭ་ཏེ་ན་པུ་ན༔པུ་ན། །ས་ཏཱི་ཏསྟཌི་ཏུདུས་ཀྱིས་གཟིར་བ་ཉིད་གྱུར་ཏེ། །ཡང་ཡང་དེ་ཡིས་བོས་པར་བྱས། །དེ་ཡིས་ཕ་ལས་ཅུང་ཟད་ཀྱང་། །ཀིཉྩབྷཱ༈་ནདྷཱནྟི་བྷཱ་ཥི་ཏཾ། །ཨ་མརྟ་ཛ་ར་སནྟསྟབྷྱཱ་ཏ་ར་ལི་ཏཁས་བླང་སྨྲས་པ་ཐོབ་མ་གྱུར། །སྨན་པ་ཁྲོ་བའི་རིམས་ཀྱིས་གདུངས། །སྲེད་པས་གཡོ་བའི་ཤུགས་སཱ་པན། །སོ་ཙིནྡྲ་ཡདྷི་ཥགྲུཁཱ་དཱ་ཏུ་རེ་ཎཱ་པི་ལོ་གྷི་ཏ། ་། ཨ་ཧོ་རིངས་ཅན། །ནད་པ་ཡིས་ཀྱང་འགོངས་པ་དེས། །སྡུག་བསྔལ་ལས་ནི་རབ་བསམས་པ། །ཀྱེ་མ་བ་ཏཱ་ཧཾ་དྷུརྟེ་ན་བཱ་ཧི་ཏས་ར་ལཱ་ཤ་ཡ། །།ཀི་ཀ་རོ་མྱ་དྷ་ནཱ་ཧ་ཙཱ་བསམ་པ་དྲང་པོ་དག། གཡོ་དང་ལྡན་པ་བདག་གིས་ཀྱེབསླུས། །བདག་གི་ལག་ནས་ནད་པ་ཡིས། །ག་ཏོ་མེ་ནི་དྷི་རཱ་ཏུ་ར། །ཀ་ཊུ་ཀོ་ཥ་དྷི་བདྲི་པུཏྠ་མཱནྟ་སྱ་སམྨ་ཏཱ། ་། གཏེར་འདས་ད་ནི་ཅི་ཞིག་བྱ། །ཚ་བའི་སྨན་བཞིན་སྨན་པ་ནི། །ནད་པ་རྣམས་ཀྱིས་སྔོན་དུ་བསྐུར་ཏེ། །—།
