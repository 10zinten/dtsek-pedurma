བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་ཎསྭཱ༑། ལོ་ཀེ་ཡ་དཱ་ལོ་ཙ་ན་ལོ་བྷ་ལོ་བྷཱ་ཏ་སཾ་རཀྵ་ཏེ་ཙཀླུ་ར་ཏི་པྲ་ཡཏྟཱཏ། །གནས༑། གང་ཞིག་བལྟ་ཐོབ་ལ་ཆགས་འཇིག་རྟེན་ན། །ཤིན་ཏུ་རབ་འབད་མིག་ནི་ཡང་དག་བསྲུངས། །རཱུ་པཾ་ཏ་དེ་ཏཏྟ་ར་ནེནྡྲ་ཛཱ་ལ་སཱསྣ་བ་ལཱི་མི་ཏྲ་མ་བྷིཏྟི་ཙི་ཏྲ། །ཙི་རཾ་དེ་དག་གཟུགས་དེ་རབ་གཡོ་མིག་འཕྲུལ་དང་། །རྨི་ལམ་འཕྲེང་གྲོགས་རྟེན་མེད་རི་མོའོ། །ཡུན་རིང་བི་ཙིལླེ་ཏི་ན་རེནྡྲ་སཱུ་ནུསྟ་སྨིནྣ་ནི་ཙཱ་བི་མུ་ཁེ་སི་རཱ་ཛཱ། །ནིཧཱརཱ་མཱ་ནེ་པི་འདི་སྤྱད་མི་དབང་སྲས་པོ་ཡིས། །རྒྱལ་པོ་དེ་ནི་བཞེད་མིན་ཕྱིར་ཕྱོགས་ཤིང་། །སྐྱེ་བོ་མཆི་མར་བཅས་ཛ་ནིཔ་བཱà་ར་ཀཱ་ར་ཡཏྟ་ཏྲ་བིགྷཱ་ཏ་མཎྜོ། །ཧེ་མ་པྲ་དཀཱུ་ར་ཏ་རེ་ཎ་པས་བཟློག་གྱུར་ཀྱང་། །དེར་ནི་མིག་དག་ཉམས་པར་བྱེད་དུ་བཅུག། གདུག་པའི་སྐྱེས་བུ་ནོར་ལ་བརྐམ་་སཱ་ས་བིཏྟ་ལུབྷེ་ན་ས་མུདྡྷཾ་ཏཱཀྵ། །དུབྷྱཱ་ར་མཱ་ཏཾ་ག་ཀྲྀ་ཏཱསྔ་པུཎྜ་པ་ཡིས། །གསེར་བྱིན་མིག་ནི་རབ་ཏུ་ཕྱུང་བ་དེས། །མི་ཟློག་གླང་པོས་ཆུ་སྐྱེས་ཚོགས་བཅོམ་པའི། །པདྨ་ཀ་རཱ་ཀཱ་ར་ཏུ་ལཱ་མ་བ་པཱ། །ཛ་ཡོ་བྱ་མེ་ཏ་ཏྲ་ས་ཧོ་པ་ཡཱ་ཏཱ་པྲ་མོ་པདྨའི་འབྱུང་གནས་རྣམ་པར་མཚུངས་པ་ཐོབ། །དེ་ཡིས་མཛའ་བོས་གསེར་གྱི་འཕྲེང་ལྡན་མ། །རྒྱལ་ཙི་ཏཱ་ཀཱཉྩ་ན་མཱ་ལི་ཀཱ་སྱ། །ཏཾ་དེ་ཤ་མ་བྱེ་ཀྱ་བི་ནབྱཱ་ནེ་ཏྲ་པྲཎྜི་བ་བར་སྦྱོར་ཚེ་ལྷན་ཅིག་དེར་འོངས་པས། །གནས་དེར་མངོན་ཕྱོགས་མིག་ནི་རྣམ་ཉམས་དེ། །མཐོང་བ་ཏཾ་མོ་ཧ་ཧ་ཏཱ་པ་པཱ་ཏ། །ཨ་བཱལྟ་སཾ་ཛྷི་བི་པུ་ལ་པྲ་ལ་ཡཱ་ཏཾ་ནེ་ཏྲ་ལཱ་ཉིད་ན་རྨོངས་པས་བཅོམ་ནས་འགྱེལ། །འདུ་ཤེས་ཐོབ་ཅིང་རྒྱ་ཆེར་རབ་གདུང་བ། །མིག་མཛེས་རྩེ་བ་ནྱ་བི་ལཱ་པ་ལུདྡྷཾ། །ཨ་ནི་ཏྱཱ་ཏཱ་ཙིནྟ་ན་དྲྀཏྠ་ས་ཏྱསྲོ་ཏཕ་ལཱམྦྷ་ས་དགའ་ལ་ཆགས་དེ་ལ་ནི། །མི་རྟག་ཉིད་བསམས་བདེན་པ་མཐོང་གྱུར་ཅིང༌། །རྒྱུན་ཞུགས་འབྲས་ཐོབ་ཛ་གཱ་ད་དྷི་ར། །མུམྨེ་ཧཱུྃ་ཏི་སཾ་ཤ་ཡ་བིཀླ་བ་ཏཱརྨ་མོ་ཧ་དེ་སྐྱོ་ད་ཡ་བརྟན་པ་དེས་སྨྲས་པ། །མཛེས་མ་འཁྲུགས་པ་ཉིད་ལས་བརྟན་པ་བསྟེན། །རྨོངས་དང་དམན་རྒྱས་སྐྲག་ཀ་ཏ་རཱ་བྷཱུ། །ཨ་བ་བྱ་བྷོ་གྷ་ནི་བྷ་བནྟི་བྷི་རུ་སྭ་ཀརྨ་ཎཱ་མེ་བ་ཕ་ལཱ་པར་མ་བྱེད་ཅིག། སྲོག་ཆགས་རྣམས་ཀྱིས་རང་ལས་འབྲས་བུ་ནི། །འཇིགས་པ་ངེས་པར་སྤྱད་པར་བྱ་—།
