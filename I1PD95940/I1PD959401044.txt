བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་བས་ཐོ་ཎ་ཀོ་ཊི་ཐུ་ཏ་ཨེ་ཥ་བྷིཀྵ་རཀྵཎྜ་ལགྷི་ཀྲྀ་ཧ་ནིཅུ་པེཀྵ། །ཡཉྩ་ཀྲ་མི་དམན་ཁྱིམ་གྱི་དཔལ་ལ་ལྟོས་མེད་པའི། །དགེ་སློང་འདི་ནི་རྣ་བ་བྱེ་བར་གྲགས། །གསེར་གྱི་འོད་ཀྱིས་བནྟི་དི་བི་བྷཱ་ཏི་སོ་ཡཾ་ཤཱསྟུསུ་ཏོ་རཱ་ཧུ་ལ་ཀཱ་བྷི་དྷ་ན། །ཧེ་མ་ཕྱོགས་ཆ་རྣམ་པར་བརྒྱནི། །ལྷུན་པོའི་གདུགས་གཞན་བཞིན་དུ་ལེགས་པར་མཚོན། །འཁོར་ལོས་པྲ་བྷཱ་བྷུ་ཥི་ཏ་དིགཱི་བྷཱ་གསཾ་ལགྱུ་ཏེ་མེ་རུ་རི་བཱ་སྱ་རཱུ་པ། །ཨེ་ཏེ་གི་རི་སྒྱུར་གང་མཁའ་ལ་མཛེས་པ་འདི། །སྟོན་པའི་སྲས་པོ་སྒྲ་གཅན་འཛིན་ཞེས་བརྗོད། །དེ་ནས་རི་དང་སྦྱོ་ཐ་དི་གེན་རེ་བྱཱཀྵ་མཎྜ་ལཱདྡྷོ་མ་ཏ་ཊཱ་ཏ་ར། །ཨཱ་ཡཱ་ཏུ་ཕྱོགས་མཚམས་རྣམས་དང་ནི། །ས་ཡི་དཀྱིལ་འཁོར་མཁའ་ངོས་ཁོང་ལས་ཀྱང་། །རྣམ་བཀྲ་རིན་ཆེན་སཾ་བྷཱུདྡྷཱ་ཏ་བྷིཀྵ་སཾ་གྷཱ་པི་ཙི་ཏྲ་རཏྭཱ་ས་ན་བཱ་ཧ་ནསྠཱ། །ཏ་ཡཱ་ཀྲ་མེ་ནེ་ཁྲི་དང་བཞོན་པར་གནས། །གྲངས་མེད་རྨད་བྱུང་དགེ་སློང་ཚོགས་འདི་བྱོན། །དེ་ལྟར་དེ་ཡིས་རིམ་ཏི་ནི་བེ་བྱ་མཱ་ནཾ་ཏེ་བྷི་སཾ་བྷཾ་བི་མུ་ཁཾ་བི་ལོ་ཀྱ། །ཡ་ཡུཔྲ་པས་བཤད་གྱུར་པའི། །དགེ་སློང་དགེ་འདུན་ཁྱད་འཕགས་བཞིན་མཐོང་ནས། །གཞན་དག་མཚོན་མིན་ཧནཱདྷཱ་ཏ་སཾ་བྷ་མཱ་ཎཱ་བི་དྷེ་ཡ་ཏཱ་ཏུ་ལུ་མ་ན་རྩ་ལགྷཱ། །ཨ་ཐ་ཛ་ལངྐཱཉྩ་དེ་དག་མཚུངས་པར་ནི། །དགའ་དང་རྨད་བྱུང་གུས་པའི་དབང་ཉིད་གྱུར། །དེ་ནས་རབ་འབར་གསེར་ན་ཙུནྡྷུ་བསྡུ་ཛ་གདྡ་བྷུ་བཱརྐ་ཤ་ཏ་པྲ་གཱ་ཤཾ། །ཨ་ཤེ་ཥ་སཾ་ཏཱ་བ་བི་ཤེ་ཥ་ཧཱུཾ་ཏྭཱ་གྱི་ཕྱེ་མའི་གོས། །ཉི་མ་བརྒྱ་ལྟར་གསལ་བ་འགྲོ་ལ་བྱུང་། །གདུང་བའི་ཁྱད་པར་མ་ལུས་ཞི་བྱེད་པས། །ཤཱི་ཏྟཾ་ཤུ་མཱ་ལཱ་ཤ་ཏ་ཤཱི་ཏ་ལཉྩ། །ཨ་ཐ་དྷ་ན་པ་ཏི་ཤ་ཀྲ་བྲཧྨ་མུ་ཁྱི་ར་མཱ་ཏིའྲི་བསིལ་ཟེར་ཕྲེང་བ་བརྒྱ་ལྟར་བསིལ་བ་ཡང་། །དེ་ནས་ནོར་བདག་བརྒྱ་བྱིན་ཚངས་སོགས་བློན་པོ་རྣམས་ཀྱི་ཕུ་ལ་ག་ག་ན་ཡཱ་ཏཱ་དཏྟ་སེ་བཱ་ནུ་ཡཱཏ། །ཨ་མ་ར་པུ་ར་པུ་ནི། །རྒྱ་ཆེ་ཨིནམ་མཁའ་བགྲོད་པ་བསྟེན་ཅིང་རྗེས་སུ་བགྲོད་པ་ཕུལ། །འཆི་མེད་གྲོང་གི་བུ་མོས་མེ་རནྡྲི་ཀཱིརྞཱ་པུཏྤ་པྲ་བྷཱ་བཱནྡྲ་ཡ་ན་པ་ཐ་མ་ཡཱ་ཤཱིཀཱུ་རྱ་བྷཱ་ཛཾ་ཛི་ནེརྫུ། །ཨཏྟཱ་ཏོག་ཚོགས་ནི་གཏོར་བ་ལས། །རྒྱལ་བའི་ཟླ་བ་བསོད་ནམས་ལྡན་པ་རྣམས་ཀྱི་མིག་ལམ་གྱུར། །གྲོང་ཁྱེར་།
