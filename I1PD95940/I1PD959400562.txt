བྱ་བ་པ་རོ་པ་ཙཱ་རི་རུ་ཙི་རེ་རཙྩི་ཏེ་ཀཱཉྩ་ནཱ་ས་ནེ། །སུ་ཁཱ་ཡ་བིཐཾ་སྲོ་བཱ་ཙ་མཆོག་ཏུ་མཛེས་པའི་རིམ་གྲོ་ཡིས། །མཆོད་དེ་གསེར་གྱི་ཁྲི་ལ་ནི། །བདེར་བཞུགས་དེ་བཞིན་གཤེགས་པ་ན་ར་ནཱ་ཐསྟ་ཐཱ་ག་ཏཾ། །།བྷ་ག་བརྨེ་བྷ་བནྣཱཀྟི་བི་བྷཀྟ་སུ་ཀྲྀ་ཏ་ལ། །མི་ཡི་མགོན་པོས་རབ་སྨྲས་པ། །བཅོམ་ལྡན་ཁྱེད་ལ་གུས་པ་ཡིས། །ལེགས་བྱས་དཔལ་ནི་རྣམ་ཤི་མ། །ཙ་ཡཀུ་ཤ་ལ་མཱུ་ལཱ་ནཱ་མ་སི་མུཀྟི་བྷ་བི་ཥྛ་ཏི། །བི་ན་ཡཱཏྟཱཟི་རྒྱས་པ། །དགེ་བའི་རྩ་བ་བསགས་པ་ཡིས། །བདག་ནི་གྲོལ་འགྱུར་སྲིད་ལགས་སམ། །བསོད་ནམས་བི་ནེ་ཏི་པུཙྪཱ་པུ་ཙྪཱ་བྷི་མཱ་ནི་ནཱ། །པྲཋསྨི་ཏ་སི་ཏཱ་ལོ་ཀཾ་ཛ་རྫོགས་པའི་ང་རྒྱལ་ཅན། །ས་ཡི་བདག་པོ་གུས་པས་འདི། །དྲིས་ལ་བདེ་བར་གཤེགས་པ་ཡིས། །གཱ་ད་སུ་ག་ཏཔྲ་ཛན། །རཱ་ཛཱ་ན་སཾ་སཱ་རཱ་མཱསྒྲ་ཡ་མ་ནཱ་དི་ནི་དྷ་ན་འཛུམ་དཀར་སྣང་བ་སྤྲོ་ཞིང་གསུངས། །རྒྱལ་པོ་འཁོར་བ་དག་གི་ལམ། །ཐོག་མཐའ་མེད་པར་འཇུག་པ་བ༑། ཧེ་ལཱ་ལཾ་རྒྱུཀ་ཐཾ་་སཱ་མ་པྲལྟ་ཀླ་ཤ་སཾ་ཀྵ་ཡ། །ཙི་ར་འདི། །ཉོན་མོངས་ཀུན་ཟད་མ་ཐོབ་པར། །སྐྱེས་བུས་བདེ་བར་ཇི་ལྟར་འགོངས། །ཡུན་རིང་མབ་རི་ཙི་ཏིཉྩ་ཀཱ་བནྟེ་ར་སཀྟ་ག་ཏཱ་ག་ཏིཔྲ་ཀྲྀ་ཏི་ག་ཧ་ན༔སཾ་སཱ་རོ་ཡོངས་འདྲིས་འཁོར་ལོ་བསྐོར་མིབཞིན་རྒྱུན་དུ་འོང་དང་འགྲོ་བ་ཡིས། །རང་བཞིན་གྱིས་ཟབ་འཁོར་བ་འདི་ཡཾ་སུ་ཁེ་ན་ན་ལཾ་གྷ་ཏེ། །ཨ་པ་ཏི་ཧི་བི་ནཱ་ཡོ་གྷ་བྷཱ་སཾ་ཀཱ་ཡ་ཀི་ལ་ཀརྨ་ལ་བདེ་བླག་འགོངས་པར་མི་ནུས་ཏེ། །སྦྱོར་བ་གོམས་པ་མེད་ཅིང་ལས་རྣམས་ངེས་པར་ཟད་པ་མིན་པ་ཉིད་ཎཱ་སྤུ་ཊ་ཕ་ལ་ཏཾ་ཏི་དྷརྨེ་སྱ་སྨིནྣཾ་བནྡྷ་ན་ཏྭཾ་གཏ། །སརྦ་ཏོ་བི་ནི་བྲྀཏྟ་སྱ་ནི། །རྒྱ་ཆེ་འབྲས་བུ་གསལ་བའི་ཆོས་ཀྱང་འདིར་ནི་འཆིང་བ་ཉིད་དུ་གྱུར། །སྦྱིན་ལ་གོམས་པ་ཆེན་པོ་དཱ་ནཱ་བྷྱཱ་སེ་ན་བྷ་ཡ་སཱ། །མ་མཱ་པི་དྷརྨ་སཾ་སཱ་རོ་བ་བྷཱུ་བ་བྷུ་བི་ཛརྨ་ཡིས། །ཀུན་ལས་རྣམ་པར་ལོག་གྱུར་པ། །བདག་ཀྱང་ས་ལ་སྐྱེ་བ་བཟུང༌། །ཆོས་ཀྱིས་འཁོར་བ་ཉིད་དུ་ཀྲྀཏ༑། དྷ་ནི་ཀོ་ནཱ་མ་དྷ་ན་བཱན་བཱ་རཱ་ཎསཱུ་མ་བྷུཉྫི་ར། །ཏ་པོ་ས་ཧ་ཕ་གྱུར། །ནོར་ཅན་ཞེས་པ་ནོར་དང་ལྡན། །་བཱ་རཱ་ཎ་སཱིར་སྔོན་བྱུང་གྱུར། །གདུང་བ་འཕྲོག་ཅིང་།
