ད་ན་ཏཱི་ཥཱ། །གྲི་ཧཱི་ཏཱ་བྷཱ་ར་ཎ་ནི་ཏཱཾ་ཧ་ཡཉྩ་བྲ་ཛ་མནྡ་རམ། །ནེ་དཱ་ནཱི་མསྟི་པ་ལ་སྨྲས་པ། །ཁྱོད་ནི་རྟ་དང་རྒྱན་རྣམས་དག། བཟུང་ལ་ཁང་པ་དག་ཏུ་སོང་། །སྒྱུ་མའི་འཆིང་བ་མེ་ཀྲྀ་ཏྱ་མེ་ཏིརྨ་ཡ་ནི་བནྡྷ་ནི། །བ་ནེ་སྨིནྣ་ཧ་མེ་ཀཱ་ཀཱི་ཤ་མ་སནྟོ་འདི་དག་གིས། །ད་ནི་བདག་ལ་དགོས་པ་མེད། །ནགས་འདིར་བདག་རང་གཅིག་པུ་ཉིད། །ཞི་བ་ཆོག་ཥ་བཱནྟ་བ། །ཨེ་ཀ་སཾ་ཛཱ་ཡེ་ཏ་ཛནུ་རེ་ཀ་ཨེ་བ་བི་པ་བྱ་ཏེ། །བི་ཤེས་གཉེན་དང་ལྡན། །སྐྱེ་བོ་གཅིག་པུར་ཡང་དག་སྐྱེ། །གཅིག་པུ་ཉིད་དུ་རྣམ་པར་ལྷུང་ཚེ། །མི་ཥ་མ་བི་ཥ་ཡ་ཡོ་གཾབྷོ་ག་མུཏྠཾ་ཚུ་རོ་ག་ས་ར་ས་ར་ཏི་བི་ཤེ་ཥ་ཀླ་ཤ་བཟད་ཡུལ་དང་སྦྱོར་བའི་ལོངས་སྤྱོད་དག་གི་ནད་བཏང་ཞིང་། །རོ་ལྡན་དགའ་བའི་ཁྱད་པར་ཉོན་མོངས་ཤོ་ཥ་པྲ་དྲྀ་ཏཱ། །པ་རི་བྷ་བ་བྷུ་བ་ནེ་སྨིནྣེ་ཥ་ན༔སེནྣི་བེ་ཤཤ་མི་སྐེམས་པར་རབ་ཞུགས་པ། །ཡོངས་སུ་ཟིལ་གནོན་སྲིད་པ་འདི་ནི་འདི་ཉིད་བདག་གི་གནས། །འདོད་ཏ་མ་ད་ན་ཀཱནྟིཤཱནྟི་མེ་བ་ཐ་ཡཱ་མི། །ཨི་ཏུཏྟཱ་བྷ་ར་ཎཱ་ནྱ་སྱ་དཱིཏྟཾ་སུཾ་ཀེ་མུ་པའི་མཛེས་པ་ཞི་བ་དག་ནི་ཞི་བ་ཁོ་ན་བསྟེན། །ཞེས་སྨྲས་དེ་ཡིས་འབར་བ་ཡི། །རྒྱན་རྣམས་དེ་ཡི་མོ་བ་ས། །ཏྱལྟ་ཤོ་ཀཱ་ནི་ཧཱ་ནཱི་བ་མུཀྟ་སཥ་ཎ་པཉྩ་ཡེ། ་། ཙུ་ཌི་ལག་ཏུ་གཏདི། །མུ་ཏིག་མཆི་མའི་ཐིགས་ཚོགས་ཀྱིཚེ། །དེ་དག་མྱ་ངན་གཏོང་བ་བཞིན། །རལ་གྲིས་ནིཀྲྀ་ཏྲ་ཁནེ་ན་ས་བིཀྵེ་ས་ན་བྷསྟ་ལེ། །ཤཀྲ༔་ཏཾ་ས་མཱ་དཱ་ཡ་ནི་གཙུག་ཕུད་བཅད་ནས་ནི། །ནམ་མཁའི་གཞི་ལ་དེ་ཡིས་འཕངས། །བརྒྱ་བྱིན་གྱིས་ཀྱང་དེ་བླངས་ནས། །ནཱ་ཡ་དི་བ་མཱ་ད་རཏ། །ཀེ་ཤགླེ་ཤ་ཨི་བོཎྚིཏྟ་ཡ་ཏྲ་ཏེ་ན་མ་ཧཱཏྠ་ནཱ། །གུས་པས་ལྷ་ཡི་གནས་སུ་ཁྱེར། །གང་དུ་བདག་ཉིད་ཆེན་པོ་དེས། །ཉོན་མོངས་བཞིན་དུ་སྐྲ་བཅད་པ། །ཀེ་ཤ་པྲ་ཏི་གྲ་ཧཾ་ཙེ་ཏཾ་སདྷི་སྟ་ཏྲ་ནི་བེ་ཤི་ཏཾ། །ཙྪནྡ་ཀོ་པྱ་ཤཱ་མཱ་དཱ་དེར་ནི་དམ་པ་རྣམས་དག་གིས། །དབུ་སྐྲ་བླངས་པའི་མཆོད་རྟེན་བརྩིགས། །འདུན་པ་ཡིས་ཀྱང་རྟ་ཡ་པྲ་ཡཱ་ཏ༔སསྟ་བྷིརྩི་ནི། །ཤ་ནིཔྲ་བྱ་པུ་རོ་པཱནྟཾ་ཤོ་ཀཱནྟས་མ་བླངས་ནས། །རབ་ཏུ་སོང་སྟེ་ཉིན་བདུན་གྱིས། །གྲོང་ཁྱེར་ཉེ་བར་དལ་བུས་ཕྱིན། །མྱ་ངན་དག་གིས་—།
