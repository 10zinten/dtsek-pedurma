ཀིནྣ་རཱི་ན་དུརླ་བྷཱ། །ཨི་ཏི་ཏ་སྱ་བ་ཙཐུ་ཏྭཱ་ཀཱུ་རཾ་པཱ་ཏ་ཀ་ཀཱུ་ཎི་ཏ། །ཡོད་པས་མི་འམ་ཅི་མི་དཀོན། །ཞེས་པ་དེ་ཚིག་མ་རུངས་པ། །ཐོས་ནས་སྡིག་པས་རྨུས་ཕྱིན་ཅིང༌། །ཚྭ་ཤཾ་པ་བྲྀཏྟ་སཾ་ཏྲསྟསྟསྟ་མ་བྷཱ་ཥབྷཱུ་པ་ཏི། །ནི་ཛ་ཛི་བི་ཏ་རཀྵ་ཡི་གཏུམ་པོའི་སྤྱོད་པས་སྐྲག་གྱུར་པ། །ས་ཡི་བདག་པོས་དེ་ལ་སྨྲས། །རང་གི་སྲོག་ནི་བསྲུང་སླད་དིུ། །ཀ་ཐཾ་སི་བ་དྷ་མུཏྲ་ཧེ། །ཀིནྣ་རཱི་བི་ར་ཧེ་བ་é་ན་ཙ་ཛཱི་བ་ཏི་མེ་སུ་ཏ། །བུད་མེད་གསོད་པར་ཇི་ལྟར་སྤྲོ། །མི་འམ་ཅི་མོ་དང་བྲལ་ན། །ངེས་པར་བདག་གི་བུ་མི་འཚོ། །ཨི་ཏི་བྷུ་མི་བྷུ་ཛཱ་ཏ་སྱ་པྲ་ཏྭཱ་ཀྵ་ཏེ་ས་མཱི་ཧི་ཏེ། །ཏཾ་པཱ་པཱ་བྷི་ནི་བེ་ཤེ་ན་སུ་ཞེས་ཏེ་ས་ལ་སྤྱོད་པས་དེའི། །འདོད་པ་ཕྱིར་ལ་བཟློག་པ་ན། །མདུན་ན་འདོན་ནི་སྡིག་པ་ལ། །མངོན་ནཔཱ་ཧ་པུ་རོ་ཧི་ཧ། །ཨ་ཧོ་རཱ་ཛནྣ་ཛཱ་ནཱི་ཥ་དྷི་མཱ་ན་བི་ཛ་ནསྦི་ཏཾ། །ཏྲི་པར་ཞེན་པས་ཡང་དེར་སྨྲས། །ཀྱེ་མ་རྒྱལ་པོ་བློ་གྲོས་དང་། །ལྡན་ཡང་སྐྱེ་བའི་ལུགས་མི་ཤེས། །སྡེ་བརྒ་སཱ་དྷ་ནཾ་ཏྭཱ་རྫུ་ན་ར་ཛུ་ན་ཙ་ཛཱི་བི་ཏཾ། །ཨཟླ་ཨི་བ་སྭཱ་ཛ་ན་གསུམ་སྒྲུབ་བྱེད་རྒྱལ་སྲིད་དང་། །སྲོག་ཀྱང་གཏང་བར་བྱ་བ་མིན། །ནོར་བཞིན་ནང་མི་གྲོགས་པོ་མི་ཏྲ་ཀ་ལ་ཏྲ་པུ་ཏཱནཋསྦི་ཏ་སྱ་པུ་རུ་ཥ་སྱ་པུ་ནཌྷ་བནྟི། །ཨུཙྪཱ་ས་མཱ་ཏྲ་ཆུང་མ་བུ་དག་ནི། །ཉམས་པ་སྐྱེས་བུ་གནས་ན་སླར་ཡང་འབྱུང་བར་འགྱུར། །དབུགས་ཙམ་བྲལ་ཞིང་བི་ར་ཧེ་ག་ཏ་ཛཱི་བི་ཏ་སྱ་ཏཏྟཱ་ལ་སནྣི་ཧིཏམ་སྱ་ས་དེ་བ་སབྦཾ། །ཏྭཱ་ཛནྟི་ཛཱི་བི་སྲོག་ནི་སོང་བར་གྱུར་པ་ལས༅། །དེ་དུས་ཉེ་བར་གནས་ཀྱང་ཐམས་ཅད་མེད་པ་ཉིད། །འཚོ་བའི་དོན་དུ་ཏ་སྩཟེ་ནི་ཛ་དེ་ཤ་པྲ་ཡཱཏྨ་ཛ། །ཛཱི་བི་ཏཱ་ད་པ་རཾ་རཱ་ཛ་ན་ཛཱི་བི་ལོ་རང་ཡུལ་དང་། །གཅེས་པའི་བུ་ཡང་གཏོང་བར་བྱེད། །རྒྱལ་པོ་འཚོ་བའི་འཇིག་རྟེན་ན། །སྲོག་ལས་ཀེཅི་ན་པྲི་ཡཾ། །ཨི་ཏི་ཛཱི་བི་ཏ་ལོ་བྷ་ཡ་ཏེ་ན་ནཱ་ནཱ་ནི་དན་ནི། །ཤ་ནིགཞན་ནི་གཅེས་པ་མེད། །དེ་ལྟར་དེ་ཡི་སྣ་ཚོགས་ཀྱི། །དཔེ་ཡིས་དལ་གྱིས་བྲིད་པ་ན། །རྒྱལ་པོ་པྲ་ཏཱ་རི་ཏེསཱ་རེ་རཱ་ཛཱ་ཡུལྟ་མ་མ་ནྱ་ཏ། །ཏ་ཏ༔སུ་སཾ་བྷཾ་ཏཱ་རཾ་སྲོག་ལ་ཆགས་པ་ཡི། །སྡིག་ལ་རིགས་པར་ཁ་ཡིས་བླངས། །དེ་ནས་ཕུན་ཚོགས་རྩོམ་པ་ཡིས། །།
