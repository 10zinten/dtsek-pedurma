སུ་བྷཱ་ཥི་ཏཾ། །ཐུ་ཏྲེ་ཏ་མཱ་ཏྱཱ་སྱ་བ་ཙཀྱི་ཏི་ཤས་ཏཾ་ས་མུཏྤ་ཚུ་བི་པྲ་ཛུ་ཡིས་ལེགས་བཤད་བླངས། །ཞེས་པ་བློན་པོའི་ཚིག་ཐོས་ས་བདག་གིས། །མདུན་ས་པ་བཀྱེ་མདུན་ས་རབ་ས་ཏྱན། །ཨནྟཔུ་རཾ་གུསྟ་མུ་པེ་ཏུ་རཱུ་པཾ་ཙ་ཀཱ་ར་ས་མཱ་རྱ་ཛ་ནཱ་ནུ་རཱུ་པཱི། །བཏང་ནས། །སྦས་པའི་གཟུགས་ཀྱིས་ཕོ་བྲང་འཁོར་སོང་སྟེ། །ཐུན་མོང་སྐྱེ་བོའི་རྗེས་སུ་མཐུན་པར་བྱས། །པ་རསྤ་རཾ་ཤུ་པྲ་ཏི་བདྷཱ་ཏཱ་རཾ་བི་སྭ་ར་ཏཱ་རཱ་ནི་ཀ་རཱ་བྷི་རཱ་མཾ། །ཧཱ་རཾ་སྐར་ཚོགས་རྒྱས་པ་ལྟ་བུར་མངོན་དགའ་ཞིང་། །ཕན་ཚུན་འོད་ཟེར་རབ་ཏུ་གསལ་བྱེད་པའི། །དོ་ཤལ་པ་མཱ་ད་ཡ་སུ་བྷཱ་ཥི་ཏཱཟིཙྪ་ཡཱ་དཱི་ཏཱི་ཡས་ཡ་ཡོ་བ་ནཱནྟཾ། །ས་ཏ་བཟུང་ནས་ལེགས་བཤད་དོན་གཉེར་དེ། །གྲིབ་མ་གཉིས་པ་དང་བཅས་ནགས་མཐར་སོང་། །དེ་ནི་ཏྲ་བཱ་ལཱ་ནི་ལ་ཀཱིཎ་པུà་བྲྀཀྵིཀྲྀ་ཏཱ་ཏི་ཐ་ཨི་བཱ་བ་ན་མྱི། །ཨ་ནི་ངེས་པར་དུད་པའི་ཤིང་རྣམས་ཀྱིས། །མེ་ཏོག་རླུང་གཞོན་གྱིས་བཀྲམ་མགྲོན་བྱས་བཞིན། །དེ་ཡིས་ཥཱ་ཡཏྭཱཏ་ཀྲྀ་ག་ཡཱ་པྲ་སཏྟཾ་གི་རེཔ་བེ་ལུབྷ་ཀ་མཱ་ས་ནཱ་ད། །ཙཱ་མེ་ན་བཱ་འབད་པས་བཙལ་བས་རྔོན་པ་ནི། །རྔོན་སེལ་རབ་ཆགས་རི་ཡི་ངོས་སུ་རྙེད། །གླང་པོ་རྣམས་ཀྱིས་བདེ་མཾ་ཀ་རི་ཎཱ་སུ་ཁཱ་ནཾ་བི་བདྨ་དཱིཀྵ་ར་སི་ཀཾ་མྲྀ་གི་ཎཾ། །བ་ཀྲ་ཀྲ་མ་ཀླུ་ར་ཏ་བའི་མི་མཐུན་ཕྱོགས། །རི་དྭགས་རྣམས་ལ་ཡུག་སས་འདེབས་བྱེད་མཁས། །འཁྱོག་པོའི་རིམ་པ་བཞིན་རཾ་ཀ་རེ་ཎ་ཙཱ་པཾ་སྭ་ཙིཏྟཾ་པ་མ་མུདྷ་ཧ་སྟི། །བ་དྷི་ཀ་ད་ཀེ་ཎ་བ་དུ་མ་རུངས་གཞུ། །རང་སེམས་ལྟ་བུ་ལག་པ་གཡོན་པས་འཛིན། །རྟག་ཏུ་ནགས་ན་གནས་པའི་མི་དཀྵི་ཎེ་ན་ན་ནོ་ཀ་སཾ་ནི་ཏྱ་མ་དཀྵི་ཎ་ན། །ཧསྟེ་ན་བི་ནྱལྟ་ས་མལྟ་ཧསྟི་མཐུན་ཕྱོགས། །གསོད་པ་གཅིག་པུ་ལ་མཁས་ལག་གཡས་ཀྱིས། །ལག་ལྡན་སྡེ་རྣམས་མཐའ་དག་བྱང་གྲོལ་བནྡྷ་པ་བརྒ་བི་ཤི་ཁཾ་ད་དྷཱ་ནཾ། །མཱ་ཡུ་ར་པཀྱི་ར་ནི་ནཱ་བ་ཧེ་མཱ་ཏ་རཾ་ལ། །རྣམ་པར་འགོད་པའི་མདའ་ནི་རབ་ཏུ་འཛིན། །རྩེ་མོ་རླུང་གིས་དལ་གྱིས་བསྐྱོད་པ་ཡི། །རྨ་བྱའི་གི་ཏཱ་གིཟི་ཧི་ཏོཏྟ་རི་ཡཾ། །ནེ་ཌིཛྲ་གཱི་ཎཾ་པ་ཏི་ཛཱི་བ་རཀཾ་སཾ་ཏཱ་ས་ལོ་ལ་རི་སྒོ་ཡི་སྟོད་གཡོགས་བསྒྲུབས་གྱུར་པ། །སྐྲག་པ་གཡོ་བ་རི་དྭགས་རྣམས་ཀྱི་མིག། བདག་པོ་འཚོ་བ་—།
