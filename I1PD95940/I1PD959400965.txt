སཱ་དཱ་རཱ་ཛ་ཤི་རོདྡྷ་བཾ་ཏསྨི་ད་དོ་སཱ་ཤོ་ཏཾ། །རཀྟ་ན་སཱུསྠཱ་མཱ་ཡཱ་ཏཿཥདྷིམ་རྒྱལ་པོས་མགོ་ལས་བྱུང་བ་ཡི། །རང་ཁྲག་རྟག་ཏུ་དེ་ལ་བྱིན། །དེ་ནས་ཟླ་བ་དྲུག་གི་ནི། །ཁྲག་གིས་སིར་ཐཱ་ཏུ་ར། །ཛ་གཱ་མ་པ་རི་པཱུཙྪཱ་ཤཀྲྀ་ཤ་མཱ་མནྟ་བྷུ་པ་ཏི། །ཏ་ནད་པ་བསོས་པར་གྱུར། །ས་བདག་རིད་པ་ལ་ཞུས་ནས། །བསམ་པ་ཡོངས་སུ་རྫོགས་ཏེ་སོང་། །དེ་ཏཀ་དཱ་ཙིདྡྷཱ་དཱ་ལ་སཱུ་ནོ༔་ཡཱ་མ་ཤཱི་ལ་ནཱཏ། །པཱནཱ་ཤོ་ཥ་གྷ་ཡ་སཾ་སཱི་པྲ་ནས་ནམ་ཞིག་ས་སྐྱོང་བུ། །ཕན་པ་ཚོལ་ལ་ངལ་དུབ་ནི། །བསྟེན་ལས་ཉམས་པ་གསལ་བྱེད་པ། །དུ་རཱ་སཱིདྷི་ཏི་ཥི་ཎ། །བི་པ་རཱི་ཏ་སྭ་བྷཱ་བོ་ཡཾ་བི་པཱ་ཀཔཱུཏྠ་：གཞོགས་སྐམ་ཉིནད་ནི་བྱུང་བར་གྱུར། །སྔོན་གྱི་ལས་ཀྱི་རྣམ་སྨིན་ལས། །རྣམ་པར་བཟློག་པའི་རང་ཀརྨ་ཎཾ། །སཱསྠཱ་ནཱ་མ་ཤ་ནཾ་ནཱསྟི་ཤྲཱི་མ་ཏྭཾ་རོ་ག་སཾ་ག་ཏི། །ཨུ་པ་བཞིན་འདི། །།ནད་མེད་རྣམས་ལ་ཁ་ཟས་མེད། །དཔལ་ལྡན་རྣམས་ནི་ནད་དང་འགྲོགས། །དེ་ནས་དིཊཾ་ཏ་ཏསྟ་སྱ་སཱབྷྱཱ་ཡ་བྷི་ཥ་ཛཱཝ་རི། །ས་ཧ་སྲ་པཱ་ཀཾ་སཾ་བིདྷཾ་སྨན་པ་མཆོག་རྣམས་ཀྱིས། །སྨན་པ་སྟོང་ྋགིས་ལེགས་བསྒྲུབས་པ། །ཀུན་གྱི་སྙིང་པོ་ཞེས་པའི་མར། །སརྦ་པཱ་རཱ་བྷི་དྷི་གཱི་ཏཾ། །སཉྫ་ཏྲ་བྷཱུ་ཏ་སཱ་རེ་ཎསྨིན་ཛཱི་པཾ་ཛི་བ་སྱ་པཀྵི་ཎ། །དེ་ནི་བསོས་སླད་ཉེ་བར་བསྟེན། །རྫས་ཀུན་གྱི་ནི་སྙིང་པོ་བླངས། །འདབ་ཆགས་ཤང་ཤང་དག་གི་ནི། །སྣ་ནཾལྦུ་ཀ་ཐ་ཡོ་གེ་ན་བྷཾ་ཏཾ་ཡ་བྷཱུ་ཡ་སཱ། །སི་དྡྷཾ་དྭཱ་ད་ཤ་བྷིརྦཀུ་ཁྲུས་ཀྱི་ཆུ་ཡི་ཐང་དང་ལྡན། །འབད་པ་ཆེན་པོས་ཞུན་མར་ནི། །བཅུ་གཉིས་ལོ་ཡིས་ལེགས་པར་གྲུབ༄༅། །མཱ་རེ་པཱ་ཏུ་མུརླ། །པྲ་ཏྲེ་ཀ་བུདྡྷཤོ་ཀཱརྟསྟ་ཏྭ་ཙི་ཏུ་མ་ཐཱ་ཡ་གཞོན་ནུ་འཐུང་བར་བརྩོན་པ་ནི། །ནད་ཀྱིས་གཟིར་བའི་རང་སངས་རྒྱས། །ཅི་སྟེ་དེ་ལ་སློང་དུ་ཡོ༑། སཎྚིརྣ་བཱརྨ་མཱ་ཎོ་པི་ཀུ་མཱ་ར༔ས་ཏྭཱ་སཱ་ག་ར། །བྷཀྵ་ཏརྨི་འོངས། །ཀུན་གྱིས་བཟློག་པར་གྱུར་མོད་ཀྱང་། །གཞོན་ནུ་སྙིང་སྟོབས་རྒྱ་མཚོ་ཡིས། །ཀུན་གྱི་སྙིང་པོ་ད་དོ་ཏསྨི་སཤྲཱི་སཱ་ར་ཤྲཱི་ཏཾ་ཧི་ཏཾ། །སཱཔྲ་ཏྱཾ་ཀ་བུདྡྷོཉཐ་པྲ་སནྣཔྲ་ཎི་ཕན་པའི་མར། །ཐམས་ཅད་དེ་ལ་གུས་པས་བྱིན། །དེ་ནས་རང་སངས་རྒྱས་བསོས་ཏེ། །རབ་དང་སྨོན་།
