ཙུ་ཌ་མ་ཎི་བབྷྱཱ་གཙྪཱ་ཏུཙྪཱ་བི་ཧཱ་ཡ་པཱ། །ཛཾ་བ་པཱ་པ་པྲ་ཐྲཏྟོ་གཤིན་གཙུག་གི་ནོར་བུ་ཆིངས། །ནམ་མཁའ་དག་ལ་མགྱོགས་པར་སོང་། །མི་བདག་འདི་ནི་སྡིག་དང་ཡཾ་ན་པ་དཱ་ཙ་ར་མཱི་ཀླུ་ཏེ། །ཡཙྪཱ་བྷཱུ་མི་ཏཱ་ལཱ་ག་ཏཱ་གནྟ་བཾ་པོ་མ་ལྡན༑། དམ་པའི་སྤྱོད་ལ་མི་ལྟའོ། །ཁྱོད་ནི་མཆོད་སྦྱིན་སར་སོང་ནས། །ནམ་མཁའི་ལམ་ནས་འགྲོ་བརྞ་ནཱ། །ཨ་ནྱ་ཐཱ་ཏྭཱ་མ་པོ་བེ་ཧི་གུ་ཌྷ་ནྱལྟ་མ་ཡཱ་ཀྭ་ཙིཏ། །ཨི་ཏི་བར་བྱ། །གཞན་ནི་ཁྱོད་ནི་འགའ་®ཞིག་ཏུ། །བདག་གིས་སྦས་བཀོད་འདི་ཡིས་རིག། ཅེས་པའི་བྷནྡཾ་པྲ་བཱ་སཱནྟཱ་བ་ཙཐུ་ཏྭཱ་མ་ནོ་ཧ་རཱ། །ཏཏྟཾ་ག་མཱ་ཡ་རཀྵ་ནི་ཚིག་ཐོས་ཡིད་འཕྲོག་མ། །ཁྱོ་དང་བྲལ་བས་གདུངས་གྱུར་པས། །འབད་པས་དེ་དང་ཕྲད་པའི་སླད། །ཡཏྟཱཏྲི་ཡ་ཏ་རཾ་པ་པུ། །ཤྭ་སཱ་དཏྟཾ་ས་མ་དྭ་ཡ་བདྷཱ་ནཱདྷི་ཤི་ཁཱ་མ་རབ་ཏུ་གཅེས་པས་ལུས་བསྲུངས་ཤིང་། །གཙུག་གི་ནོར་བུ་སྒྱུག་མོ་ཡིས། །བྱིན་པ་བླངས་ནས་སྤྱི་བོར་ཎི༑། སྣུ་ཥཱ་ཧཱུཾ་ཏཱ་ཀྲ་ཏུ་ཀྵེ་ཏྲ་ག་ཏཱ་བྱོ་མ་བྱ་གཱ་ཧ་ཏ། །རཱ་ཛནི་ཏཏྟ་བཅིངས། །མནའ་མ་བོས་པ་མཆོད་སྦྱིན་གྱི༄༅། །ཞིང་དུ་སོང་ནས་མཁའ་ལ་འགྲོ། །རྒྱལ་པོ་བུ་ནི་བ་ས་མུ་ཙ་ཏཾ་ཡཏྲི་ཡ་སཱུ་དྷི་སཱུ་ནོབ་དྷ་པཙླི་ནི་ཛ་པ་ད་མི་ཡཾ་སྭསྟི་ཏུ་གཅེས་ཀྱང་མནའ་མ་གསད་@པར་བྱ་བ་གང་འདི་ནི། །ཁྱོད་ཀྱིས་འོས་པ་མ་ཡིན་ཁྱོད་ནི་ཤིས་གྱུར་བདག་ཧཱུྃ་ག་ཏཱ་ཧཾ། །རཝཱ་པོ་པྲ་ཏི་ཧ་ཏ་བྷཾ་ཏིསྨྲ་ཤྲཱི་ཡོ་གེ་ཀུ་མཱ་རཿནི་རང་གནས་སོང་། །བདག་བྲལ་བརྟན་པ་རབ་ཏུ་ཉམས་པའི་གཞོན་ནུ་འདི་ཡང་བསྲུང་ཀྱི་བར་གྱིས། །ཏ་ཏྲེ་ཏུཏྟཱ་ཏ་ནུ་ཏ་ར་ཏ་ཌིདྨ་བྷ་མེསཱ་ཛ་གཱ་མ། །ཏ་ཧཱུཾ་ག་ཏཱ་ཡཱ་ནཱི་པ་དེར་དེ་བརྗོད་ནས་རབ་ཕྲ་གློག་གི་རྣམ་འཕྲུལ་དག་གིས་དེ་སོང་གྱུར། །དེ་སོང་ཚེ་ན་མི་བདག་གི༄༅། །ཏིརྨ་ཙཱ་བིཧྥེ་ཐཾ་ཀིཏ། །པུ་རོ་ཧི་ཏ་སྟ་མ་བ་ད་བ་མཱ་སཾ་ཤ་ཡཾ་མཆོད་སྦྱིན་བགེགས་ཀྱིས་དོགས་པར་གྱུར། །མདུན་ན་འདོན་གྱིས་དེ་ལ་སྨྲས། །ས་བདག་ཐེ་ཚོམ་ཡོངས་ཀྲྀ་ཐཱ། །མནྟིརྨ་ཡཱ་པ་མཱ་ཀྲྀཥཱཀཱུ་རཱ་ཁྱོ་བྲཧྨ་རཱཀྵ་ས། །ནིàསྨཱ་ཏེ་མ་བྱེད། །གདུག་པ་ཞེས་བྱ་ཚངས་པ་ཡི༄༅། །སྲིན་པོ་བདག་གིས་སྔགས་ཀྱིས་དྲངས། །མི་འམ་ཅི་མོ་—།
