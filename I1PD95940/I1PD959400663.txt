ས་ཏྱཱ་ན་ས་ཧ་སཱ་སྟི་ཏྭཾ་ཏཱ་བནྡི་བཏྟ་ཏྭཾ། །ཨི་ཏུཏྟཱ་མཱ་ཏྲེ་སི་རཱུ་པ་སཱ་ཏྱཱཀ་པ་དེ་ཡིས་རེ་ཞིག་འཕྲལ། །བུད་མེད་ཉིད་ནི་ལྡོག་གྱུར་ཅིག། དེ་བརྗོད་ཙམ་ལ་བདེ་ལྡན་མ། །དེ་ཡི་པ་ཏྱ་སཱ་ལི་ནཱི། །སབྦེ་ལ་ཀྵ་ཎ་སམྤནྣཾ་པུ་རུ་ཥ་ཏྭཾ་ས་མཱ་ཡ་ཡོ། །བུད་མེད་གཟུགས་བཏང་ནས། །མཚན་ཉིད་ཐམས་ཅད་ཕུན་ཚོགས་པའི། །སྐྱེས་པ་ཉིད་དུ་ཡང་དག་གྱུར། །ན་གརཱ་མུཏྤ་ལཱ་བ་ཏྭཱ་མསྨིནྣ་བ་ས་རེ་ཧྲ་བ། །ཨུཏྟཱ་ལཱ་ཎཿས་མཱདྷཱ་ཡུརྞ་གྲོང་ཁྱེར་ཨུཏྤལ་ལྡན་པར་ནི། །མི་བདག་ཨུཏྤལ་ཞེས་བྱ་བ། །གནས་སྐབས་དེར་ནི་ཚེ་རྫོགས་ཤིང༌། །དྷི་ཡོ་གཱདྨ་པ་རྱ་ཏ། །ལ་ཀ་ཎ་ཛྷེ་ར་ཐཱ་བྷེ་ཏ་པྲ་བ་རི༴དྡྷ་ནད་ཀྱི་སྦྱོར་བས་ལྷུང་བར་གྱུར། །དེ་ནས་རབ་མཆོག་རྒན་དང་བློན། །མཚན་ཉིད་ཤེས་རྣམས་མངོན་མནཱི་བྷི། །ས་ཏྲཿཔྲསྟོ་སོ་རུདྨ་པཱ་ན་བྱཱ་ཥི་ཙ་ཏ། །ས་རཱ་ཛཾཕྱོགས་ཏེ། །འཕྲལ་ལ་སྐྱེས་བུ་ཉིད་ཐོབ་པ། །གསེར་ལྡན་དེ་ནི་མངོན་དབང་བསྐུར། །ཆོས་ལྡན་རཱ་ཛཱ་མཱ་ན་ཤྲཱིཀྲྀ་ཏྭཱ་དྷརྨ་དྷཱ་ནཤྲཱི་རཾ། །ཏ་ནཾ་ཏ་ཏྭཱ་ཛ་ཀཱ་ལེ་ནསྒྲ་མཛེས་པའི་དཔལ་དེ་ཡིས། །རྒྱལ་སྲིད་དག་ནི་ཡུན་རིང་བྱས། །དུས་ཀྱིས་ལུས་ནི་བཏང་གྱུར་ཏེ། །ལུས་ཡི་ནོ་ན་ཧི་དེ་ཧི་ན། །ཏ་ཏྲི་བ་ཐེཊྛི་ནསུ་ནུསོ་བྷཱུཏྠ་ཏཱ་བ་རཱ་བྷི་དྷ། །ནི༔ཅན་དག་ནི་བརྟན་པ་མེད། །དེ་ཉིད་དུ་ཚེ་ཚོང་དཔོན་བུ། །སྙིང་སྟོབས་མཆོག་ཅེས་བྱ་བར་གྱུར། །ཟོལ་ཇོརྫི་ཏ་དཱ་ནེ་ཥཱ་ཛམ་བྱཱསྟེ་ཥཱ་སཱ་ད་ར། །སམྦ་བྷཱུ་ཏཱནྟི་ཙིཏྟཱ་ས་ས་ས་མེད་བསྒྲུབ་པའི་སྦྱིན་པ་ནི། །སྐྱེ་བ་གོམས་ལ་གུས་བཅས་པའོ། །འབྱུང་པོ་ཀུན་གཟིར་བསམ་པ་ལ། །དཱ་ནྱསྟ་མཱ་ན་ས། །དུས་ཧམྲཀྐི་ཎཱ་མེ་བ་གླུཇུཁེ་ས་མ་ཙིཏྟ་རྟག་ཏུ་དེ་ཡིས་ཡིད་བཀོད་དེ། །བཟོད་དཀའ་འདབ་ཆགས་རྣམས་ཉིད་ཀྱི། །བཀྲེས་པའི་སྡུག་བསྔལ་ཡང་ཡཏ༑ས་ཧྥ་ཤཱ་ན་བ་ནཾ་ག་ཏཱ་གླུ་རེ་ཎོལླི་ཁྱ་བི་གྲ་ཧཾ། །ཨུཏྟཱ་ན་ཤཱ་དག་བསམས། །དེ་ནས་དུར་ཁྲོད་ནགས་སུ་དེ། །སོང་ནས་སྤུ་གྲིས་ལུས་བཀས་ཏེ། །་གན་ཀྱལ་ཉལ་ཡི་པྲ་ད་དོ་ཏ་ནཾ་ཀྲ་བྱཱ་ད་པ་ཀཱི་ཎ། །ཨཱུརྞཱིརྒ་ཏི་བི་ཧཾ་གཉཔྱ་དཀྵི་ཎཾ་ནས་ཤ་ཟ་བའི། །འདབ་ཆགས་རྣམས་ལ་རབ་ཏུ་བྱིན། །རིམ་གྱིས་དེ་ཡི་མིག་གཡས་ནི། །མཐོན་པོར་—།
