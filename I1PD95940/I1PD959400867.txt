རལི་ཀཱ། །མྲྀ་གི་གཏྟཱ་ས་མུངྒུ་ཏ་ཤཱཎྜི་ལྱཱ་སྱ་སུ་ཏཱ་མུ་ནེ། །སཱ་ག་ཏཾ་ཏེ་མུ་ནི་ཞེས་བྱ། །རི་དྭགས་མངལ་སྐྱེས་ཐུབ་པ་ནི། །ཤཱི་ཊི་ལྱཱ་ཡི་བུ་མོ་ཡིན། །ཐུབ་པའི་སྲས་ཁྱོད་ལེགས་པར་སུ་ཏ་པྲ་ཡཾ་མེ་ཏ་བ་དན་ནཾ། །མ་ནོ་ཧ་ར་ཏ་རཾ་ཙི་ཏྲ་བལྒ་ལཾ་ཀི་མི་འོངས། །བདག་ནི་ཁྱོད་མཐོང་རབ་ཏུ་དགའ། །ཤིན་ཏུ་ཡིད་འཕྲོག་བཀྲ་བ་ཡི། །ཤིང་ཤུན་བརྟུལ་དཾ་བྲ་ཏཾ། །བི་བྷུ་ཥི་ཏོ་ཡམྨཱ་ཡུ་རཿཤི་ཁཎྜི་རི་བ་པཉྩ་བྷི། །སུ་ཞུགས་འདི་ནི་ཅི། །རྨ་བྱའི་གཙུག་ཕུད་ལྔ་དག་གིས། །བརྒྱན་པ་བཞིན་དུ་ཁྱོད་ཀྱི་ནི། །རལ་ཚོགས་རཱཙྪཱ་ཀཱིརྞ་ཀུ་སུ་མཤོ་བྷ་ཏེ་ཏེ་ཛ་ཊཱ་བྷ་བ། །ཨི་ཡ་མཱ་མ་ལ་ཀསྠཱ་ལ་འདི་དག་ལྷ་ལས་ཀྱང་། །མེ་ཏོག་དག་གིས་རྒྱས་པར་མཛེས། །སྐྱུ་རུ་ར་ནི་རགས་པ་ཡི། །འབྲས་བུ་ཁ་པྲ་ལེ་ཡ་ཕ་ལ་ཀོ་ཛྭ་ལ། །དཱིནྡྷ་ཀྵ་སཱུ་ཏྲ་མཱ་ལ་བྷི་ཀནྡྷེ་བ་ཧུ་གུ་ཎཾ་ཏ་བ། །ཨཱ་བ་རབ་འབར་འདྲ། །བགྲང་འཕྲེང་འཕྲེང་®བ་རིང་པོ་ཡིས། །ཁྱོད་ཀྱི་མགྲིན་ལ་ཡོན་ཏན་མང་། །འོད་སཱུ་ཏྲ་ཡ་ཏི་ཏེ་པཱ་ཎིཉྩི་ཏྲིཀུ་ཤ་པ་བི་ཏྲ་ཀི། །ཀུ་ཊི་མེ་བེ་ཎུ་དཎེ་སྨིན་མའི་དབྱུག་པ་འཁྱོག་པོ་འདིར། །ཡལ་འདབ་གསར་པའི་འཕྲེང་བ་ལ། །ཁྱོད་ཀྱི་ལག་པ་ཀུ་ཤ་ཡིས། །བྱ་ལ་པལླ་བ་མཱ་ལི་ཀཾ། །ར་མ་ཎི་ཡཱ་བྲ་ཏ་བ་ཏཀ་ཧཱུཾ་ཏཱཾ་ཀཱ་ཏ་བཱ་ཐ་མ། །དག་བྱེད་བཀྲ་བས་ཀུན་ཏུ་སྤྲས། །ལེགས་པ་བདག་ཡིད་རི་དྭགས་ནི། །གང་དུ་ངལ་ནི་བསེས་གྱུར་པ། །མ་ནོ་མཾ་གཱ་ཛི་ན་ཤཱནྟི་མེ་ཏི་མེ་ཡ་ཏྲ་སཱ་དྷུ་ཙ། །ཨི་ཏི་མུབྷཱ་བ་ཙདགའ་བར་རྨ་བྱའི་བརྟུལ་ཞུགས་ཅན། །ཁྱོད་གནས་གང་ཉེན་ཡོད་པ་སྨྲོས། །ཞེས་པ་བཞིན་བཟང་མ་སྣ་དུ་སུ་དྷཱཛྲ་བ་སུ་དྷ་དྷི་པ། །ཨཱ་སཱ་རྱ་ད་ཏཱ་པཱ་ཧྥེ་ཡ་མོ་ད་ཀཾ་ཏ་ཡི་ཚིག་སྙན་ཞིང་བདུད་རྩིས་བརླན་པ་དག། ས་བདག་གིས་མྱང་ལམ་བརྒྱགས་ཀྱི། །ལ་དུ་བྱིན་མ་བྷཱ་ཥ་ཏ། །ནི་བཱསྨིནྡྲ་ཙི་ཏཾ་སུ་བྷཱུ་དགྷ་སུ་ཙཱི་ར་ཡཱ་ཙི་ཏེ། །སུ་ཀུ་ནས་དེ་ལ་སྨྲས། །སྨིན་ལེགས་ཁྱོད་ལུས་ཤིན་ཏུ་གཞོན། །ཤིང་དང་རྩྭ་ནི་བསྐམས་གྱུར་ཅིང་། །ཀུ་ཤའི་མཱ་རཾ་ཏ་བ་བ་པུ༔ཤུཏྟ་རུ་ཏྲི་ཎེ་བ་ནི། ་། ཨི་ཏསཾ་བྷོ་ག་ཡོ་ཁབ་ཀྱི་ཚོགས་གཏམས་པའི། །།ནགས་ཚལ་འདིར་ནི་འོས་པ་མིན། །འདི་ནས་ཧ་ཅང་མི་རིང་ན། །—།
