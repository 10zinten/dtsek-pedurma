ཏ་ཏྲ་ཀྲྀ་ཏྭཱ་ཐ་བྷ་ག་བཱན་བཱ་སཱ་ད་སྱ་པྲ་ཏི་གྲ་ཧཾ། །པུ་ནརྗེ་ཏ་བ་ནཾ་གཙྪ་དེ་ནས་དེར་ནི་བཅོམ་ལྡན་གྱིས། །ཁང་བཟང་རབ་ཏུ་བླངས་པ་མཛད། །དགེ་སློང་དང་འགྲོགས་རྒྱལ་མུརླ་ཡོ་ས་ཧ་བྷིཀྐུ་བྷི། །བྲ་ཛནྣཱ་རཱི་ཙཱི་ལོ་ཀ་་མོདྷ་ལྟ་ཡ་བྱེད་ཀྱི། །ཚལ་དུ་སླར་ཡང་གཤེགས་པར་བརྩོན། །གཤེགས་ཤིང་མོ་གལ་བུ་ཡི་མ། །འཇིག་རྟེན་འོད་ཎཱ་མཱ་ཏ་རཾ། །།སངྒཱི་ར་རྱ་པ་ཏྭཱ་ཧ་ཏ་དྷརྨ་མཱཧྥེ་ནྱ་བེ་ཤ་ཡཏ། །ཨ་ཟེར་ཅན་གནས་པ། །ཚིག་གིས་འཕགས་པའི་བདེན་པ་ཡི༄༅། །གཞོམ་མེད་ཆོས་ཀྱི་ལམ་ལ་བཀོད། །དེ་ཐ་ཛེ་ཏ་བ་ནཾ་པྲཏྟཾ་བྷ་ག་བཱནྟཾ་ས་བིསྨྲ་ཡཱ། །པཱུརྞ་སྱ་པུཙྩཾ་པ་པྲཛྙཱ་ནས་བཅོམ་ལྡན་རྒྱལ་བྱེད་ཀྱི། །ཚལ་གཤེགས་གང་པོའི་བསོད་ནམས་ལ། །ཡ་མཚན་ལྡན་པའི་དགེ་སློང་བྷི་ཀྵ་བ༔སོ་བྱ་བྷཱ་ཥ་ཏ། །ཀཱ་ཤུ་པ་སྱ་པུ་རཱ་སམྱག་སཾ་བུདྡྷཱ་སཱུ་ནྱ་གིས། །དྲིས་ཤིང་དེས་ཀྱང་བཀའ་སྩལ་པ། །སྔོན་གྱི་ཡང་དག་རྫོགས་སངས་རྒྱས། །འོད་སྲུངས་ཀྱི་ནི་ཛདྨ་ནི། །བི་ཧཱ་རཱ་བི་ཀྲྀ་ཏཔུཎྞསཾ་གྷོ་པ་སྦྲ་ཡ་ཀོཉབྷ་བཏ། །ས་ཀ་དཱ་ཙི་དགེ་འདུན་གྱི། །གཙུག་ལག་ཁང་སྐྱོང་ཉེ་གནས་སུ། །སྐྱེ་བ་གཞན་ལ་གང་པོ་གྱུར། །ནམ་ཞིག་དེ་ཡིས་ད་སཾ་ཀྲྀཏཱཾ་ཛྲཏྟཱ་བི་ནཱ་རི་ཀིཾ་བྷུ་ཁཾ། །པཱ་ཧ་པྲ་བྲ་ཛི་ཏཾ་ཏཱི་ཛྲ་ཀྲོ་དྷཱ་དུ་གཙུག་ལག་ཁང་། །ས་གཞི་བྱི་དོར་མ་བྱས་མཐོང་། །རེ་མོས་འབབ་པའི་རབ་བྱུང་ལ། །ཁྲོ་བ་དྲག་པ་དྷི་བཱ་རི་ཀཾ། །དྲྀཏ་ཀྱོ་པ་དྷི་བ་རོཞབྱ་བི་ཧཱ་རོཉསྨིནྣ་མཱཛྫི་ཏེ། །ཀ་པོས་རབ་ཏུ་སྨྲས། །དེང་འདིར་གཙུག་ལག་ཁང་མ་ཕྱགས། །དྲེགས་པ་ལྡན་པ་འབངས་མོའི་བུ། །སུ་སྱ་དཱ་སཱི་སུ་ཏ་ཀྱེ་ཏི་བུ་བཱ་ཎས་མ་བྷནྡྷ་ཡཏ། །ཏེ་ན་པཱ་རུ་ཥཱ་བཱ་པེ་ན་བྷུཏྟཱ་ཡིས་@རེས་ལ་བབ་པ་ཞེས། །བརྗོད་ཅིང་དེ་ལ་རབ་ཏུ་སྨད། །ཚིག་རྩུབ་སྡིག་པ་དེ་ཡིས་ནི། །དམྱལ་ན་ར་ཀ་དུརྒ་ཏི། །དཱ་སཱི་སུ་ཏོཉབྷ་བཏྟཱཎྞཔཉྩ་ཛརྨ་ཤ་ཏཱ་ནི་ས། །སཾ་བའི་ངན་འགྲོར་སྡུག་བསྔལ་སྤྱད། །སྐྱེ་བ་ལྔ་བརྒྱར་གང་པོ་དེ་༅། །འབངས་མོ་ཡི་ནི་བུ་རུ་གྱུར། །དགེ་གྷོཔཱ་སེ་ན་མེ་བཱ་སྱ་པུ་ཏྟཱ་ཡ་བྷཱུརྞ་ཧི་ཡ་སཱ། །ཨརྒ་ཏྭཾ་ཡེ་ན་ནིཤེ་ཥ་འདུན་བསྙེན་པ་ཉིད་འདི་ཡི། །བསོད་ནམས་ཆེན་པོའི་སླད་གྱུར་ཏེ། །གང་གིས་སྲིད་པའི་ཉོན་མོངས་—།
