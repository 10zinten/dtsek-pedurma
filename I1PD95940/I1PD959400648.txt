་ནཱ་ནཱ་ཡུཀྟི་ནི་དན་ནི། །ཤ་ནི༔སཱ་ནི་དེ་བྷེ་པ་ཏྱཱཔཱ་ཏ་ཀཱ་བྷི་མུ་ཁམྨ་སྣ་ཚོགས་རིགས་པ་ཉེར་བསྟན་ནས། །དལ་གྱིས་སྡིག་པ་དག་ལ་ནི། དེ་ཡིད་མངོན་པར་ཕྱོགས་པར་ན༑། བནྟི་ཏཱནྡྲ་ཧུ་བྷསྣེ་ཧིས་ཧ་ཛམྦུདྡྷཱ་ཛཱ་ནི་བ། །སྨིཾ་ཏྭཱ་ཧ་རནྟི་བྱས། །མཛའ་སྣུམ་མང་པོས་བསྐྱེད་པ་ཡིས། །ལྷན་སྐྱེས་སྤྱི་བོར་སྐྱེས་པ་བཞིན། །བུད་མེད་སྤུ་གྲིའི་ས་ཧསཱ་ཀླུ་ར་དྷ་རཱ་ཁ་རཱསྟི་ཡ། ་། བ་ཀཱ་པ་རཾ་ཀཱ་ར་ཏ་རཱ་ཀྲི་ཡཱ་སོ་རྣོན་གྱིས། །འཕྲལ་ལ་བཅད་ནས་འདོར་བར་བྱེད། །རྨོངས་པས་བཅོམ་པའི་བློ་དང་བུད་མེད་ནི། །སུ་པྲ་བནྟ་ནཱ་ཡི་བ་དྲིནྟཱ་བྷི་ཡོ་ག། སཱ་པཱ་ནི་བཱ་ཏཱ་ཡ་བྷ་བ་ཏུ་ཤུ་མོ་ཧཱ་ཧ་ཏཱ་ནཾ་འཁྱོག་ཅིང་གཞན་ལ་རབ་གདུག་བྱ་མིན་ལ། །འཇུག་པའི་སླད་དུ་བརྟན་པ་སྔོན་ེསྦྱོར་ཞིང་། །སྡིག་པ་ཉིདཡུ་བ་ཏིརྨ་ཏི། །བནྡྷ་མི་ཏྲ་བི་རཀཱ་ནཾ་སྭ་སུ་ཁ་་བ་ཙེ་ཏ་ངེས་པར་ལྷུང་བར་བྱེད་པ་ཉིད། །གཉེན་དང་གྲོགས་ལ་ཆགས་བྲལ་ཞིང་། །སེམས་ནི་རབ་བདེས་མྱོས་སཾ༑། ཤྲཱི་ཡུ་ཏཱ་ནཱ་མ་བ་ནཱ་ཏཎི་ཛི་ནཱ་ནཱཾ་མ་ཏིཀུ་ཏ། །།ཨ་ཐ་བྷཱ་ཏར་གྱུར་པ། །དཔལ་ལྡན་རྣམས་ལ་དུད་པ་བཞིན། །བུད་མེད་རྒྱལ་ལ་བློ་གྲོས་གང་། །དེ་ནས་དེ་ཡིས་མཱ་ཧུ་ཡ་ནཱི་ཏཱ་པུཊྚཙཱ་ཡཱ་ཡ་ས། །ནྱ་བ་དྷཱི་ད་ཐུ་ཏཱ་ཀྲན་མ་སྨ་ནཱ་བི་ཛ་ནེ་སྤུན་ཟླ་ནི། །མེ་ཏོག་བཏུ་སླད་ལ་དྲངས་ནས། །མི་མེད་ནགས་སུ་རྡོ་བ་ཡིས། །ཆོ་ངེ་ཐོས་པ་མེད་པར་བ་ནེ། །ཨ་ཧ་མེ་བ་ཏཏྟཱ་པཾ་བྷུཀྟ་པཱུབྦེ་ཥཱ་ཛརྨ་སུ། །བ་ཧཱ་མྱ་རཱུ་པི་ཤེ་ཊཱ་བསད། །དེ་ཉིད་ང་ཡིན་སྡིག་པ་དེ། །སྐྱེ་བ་སྔ་མ་རྣམས་སྤྱད་ནས། །མཐེ་བོང་སྣད་པའི་མཚན་ཉིད་ཤ་མཾ་གུཥཱ་ཀྵ་ཏ་ལཀྵ་ཎཾ། །སཱརྦ་བཱ་ཧོརྦ་དཏྟ་ཁྱུཔུཎྜ་པྲ་བ་ཧ་ཎཅན། །ད་ལྟའང་ལྷག་མའི་ཆ་ཤས་འཛིན། །དེད་དཔོན་དོན་བྱིན་ཞེས་བྱ་བ། །གཟིངས་ནི་གང་བར་པུ་རཱ། །ས་བ་ན་སཱུ་ནུ་ཀུ་ལེ་ན་རཏྣ་དཱི་པཱཏྟཱ་མཱ་ཡ་ཡོ། །དྲི་ཏཱི་ཡ་གྱུར་པ་སྔོན། །རྗེས་སུ་མཐུན་པའི་རླུང་དག་གིས། །རིན་ཆེན་གླིང་ནས་ཡང་དག་འོངས། །དེ་ནས་དེད་སཱཟླ་བ་ཧོ་ཐ་ནཥྛརྦ་སྟེ་ས་མཱ་ཤྲཱི་ཏ། །དྭེ་ཥཱཉྫ་བ་ཧ་ཎེ་ཙྪི་ཏྲ་པྲཙྪསྣ་དཔོན་གཉིས་པ་ནི། །དོན་ཉམས་དེ་ལ་བརྟེན་གྱུར་པ། །སྡང་བས་གཟིངས་ལ་སྦས་པ་ཡི། །འུག་པ་—།
