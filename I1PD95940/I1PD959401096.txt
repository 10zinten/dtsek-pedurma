བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེ་རཉྩ་ཀཱ་ར་ཤྲཱི་ད་ཡ་གྲ་ཧཾ། །ཨསྨིནྣ་བ་ས་རེ་རཱ་ཛྷསུ་ཏ་བཱ་ལ་ས་ར་སཱཏཱི། །ས་བདག་སྙིང་ནི་བཟུང་བ་བྱས། །གནས་སྐབས་དེར་ནི་རྒྱལ་པོ་ཡི། །བུ་མོ་གཞོན་ནུ་དབྱངས་ཅན་མ། །སུ་བྷཱ་ཏི་ཏཾ་བྷ་བྱ་གཾ་རཱ་ཧརྨོ་སཾ་ག་ཏཱ་ཛ་གོ། །སཱུཏྟཾ་གྲོ་ཧ་པ་བ་གམ་ཅན་གྱི་ནང་སོང་བས། །ལེགས་ལྡན་ཚིག་གིས་ལེགས་བཤད་བླངས། །རྣ་བའི་བདུད་རྩི་འདོད་པ་ཏིཐུ་ཏྭཱ་ཤྲ་བ་ཎཱ་མྲྀ་ཏ་མཱིཔི་ཏཾ། །པྲ་ལ་ཀཾ་ཀུ་རི་ཏཱ་ཀཱ་རསྟསྨརྨ་ཡ༔སཱ་གྲོ་ཡི། །ལེགས་བཤད་ཁྱིམ་བདག་གིས་ཐོས་ནས། །སྤུ་ལོང་མྱུ་གུའི་རྣམ་པ་ཅན། །དེ་ཡི་རང་བཞིན་ཧཾ་ཡ་ཡོ། །ཏྱཱཀྟ་ཧཱ་ར་བི་ཧཱ་རོ་ཐ་སུ་བྷཱ་ཥི་ཏ་ཧྲི་ཏཱ་ཤ་ཡ། །རང་ཁྱིམ་སོང་། །དེ་ནས་ཟས་དང་རྩེ་དགའ་བཏང་། །ལེགས་བཤད་ཀྱིས་ནི་བསམ་པ་ཕྲོགས། །ནི་དཱ་ད་རི་དྲསཱ་ཤྲཱི་ཧེ་བ་ཙི་རཾ་མ་ཙིནྟ་ཡཏ། །ཨ་ཧོ་ནུ་རཱ་ཛ་པུ་གཉིད་ཀྱིས་དབུལ་བ་དེ་ཡིས་ནི། །རང་གི་ཁྱིམ་དུ་ཡུན་རིང་བསམས། །ཀྱེ་མ་རྒྱལ་པོའི་བུ་མོ་དེས། །ཏུཏྟ་གཱི་ཏཾ་སཱ་དྷུ་སུ་བྷཱ་ཥི་ཏཾ། །ཨ་པརཱ་ལོ་ཙི་ཏེ་ནཱ་པི་ཡེ་ནཱ་མྲྀ་ཏ་མི་བཱམྦི་ལེགས་པའི་ལེགས་བཤད་གླུར་བླངས་པ། །གང་དེ་ཡོངས་སུ་མ་སྤྱད་ཀྱང་། །བདུད་རྩི་དག་ནི་བྱིན་པ་ཏཾ༑། མ་ཡཱ་དྷཱ་ད་ཤ་བྷིརྦརྟོཆི་པུ་ལོ་རཏྭཱ་སཉྩ་ཡ། །སཾ་པཱབཞིན། །བདག་གིས་ལོ་ནི་བཅུ་གཉིས་ཀྱིས། །རིན་ཆེན་ཚོགས་ནི་ཡངས་པ་དག། ཐོབ་དེ་ལེགས་སཱུཀྟ་རཏྟཾ་ཏུ་ན་ཀུ་ཏཤྲཱི་དུ་པཱརྫི་ཏཾ། །རཀྐིཀི་མ་གུ་རུ་བྷིཔྲ་བཤད་རིན་ཆེན་ནི། །འགའ་ཞིག་ལས་ཀྱང་ཉེར་མ་བསྒྲུབས། །འབད་པ་བརྒྱ་ཡིས་བསགས་པ་ཡི། །རྡོ་ཡཏྣ་གུ་རུ་སཾ་ཙི་ཏི། །སནྣཱརྒ་ལོ་ཀ་ནེ་རནྣཾ་ཏདྲི་དྷནྟུ་སུ་བྷཱ་ཥི་ཏཾ། །ལྕི་རིན་ཆེན་རྣམས་ཀྱིས་ཅི༄༅། །དམ་པའི་ལམ་ནི་གསལ་བྱེད་པའི། །ལེགས་བཤད་དེ་ལྟར་རིན་ཆེན་ཡིན། ཨི་ཏི་སཾ་ཙི་ཏུ་པུ་རུ་ཥ་བི་སྭ་ཛུ་ཚྭ་པ་མནྡི་རཾ། །ས་ཙཀྲེ་རཱ་ཛ་དུ་ཞེས་པ་བསམས་ནས་མི་བདག་གི༄༅། །ཁང་པར་སྐྱེས་བུ་རབ་བཏང་སྟེ། །ལེགས་བཤད་རྒྱལ་པོའི་བུ་མོ་ཧིཏུ་སཱུཀྟ་ཡཙྪཱ་མ་ཡཱ་ཙ་ཀ། །སཱ་བ་བྷཱ་ཥསྨི་ཏ་མུ་ཁཱི་སུཏྟཾ་མུ་ལེ་ལས། །སློང་བ་པོ་མིན་དེ་ཡིས་བསླངས། །འཛུམ་པའི་བཞིན་ལྡན་དེས་སྨྲས་པ། །ལེགས་བཤད་རིན་།
