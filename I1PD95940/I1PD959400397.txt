མཱ་ཡ་ཡོ། །སཱ་མཱ་ཐ་མ་པ་དཾ་པྲ་སྟོ་བྷ་ག་བནྟ་མུ་པེ་ཏྱ་ས། །པྲ་ཎ་ལྟ་རུ་འོངས། །རང་གི་བསྟི་གནས་ཕྱིན་གྱུར་པའི། །བཅོམ་ལྡན་འདས་ལ་མངོན་ཕྱོགས་ཏེ། །ནོར་བུའི་ནཱ་མཱརྐ་སཾ་ཀཱ་ཤཾ་པྲ་དཱིལྟ་མ་ཎི་ཀུཎྜ་ལ། །ཏ་སྱ་ཀེ་ཡུ་རེ་མུ་ཀུ་ཊ་པྲ་བྷཱ་རྣ་ཆ་རབ་འབར་བ། །ཉི་མ་འདྲ་བ་དེས་ཕྱག་འཚལ། །དེ་ཡི་དཔུང་རྒྱན་ཅོད་པན་གྱི། །འོད་ཀྱི་ཡལ་པལླ་བ་པུ་རི་ཏཱ། །ཤཀྲ་ཙཱ་སེ་རི་བཱ་བྱཥ་བི་རེ་ཛུརྙན་རཱ་ཛ་ཡ། །བི་ན་ཡཱ་འདབ་རྒྱས་གྱུར་པ། །སྤྲིན་གྱི་ཕྲེང་བ་བརྒྱ་བྱིན་གྱི། །གཞུ་ཡིས་ཁྱབ་པ་བཞིན་དུ་མོསམི། །སྡིག་པ་བྲལ་དུ་བ་བི་བྱ་གྲེ་ས་ཤཱyཤ་སྟི་ཀསྨྲ་ཥ། །ཏཾ་དི་བྱ་པུཥི་རཱ་ཀཱིརྱ་སཏྭཱ་དེས་སྟོན་པ་ཡི། །མདུན་དུ་དུལ་བས་འཁོད་ནས་ནི། །ལྷ་ཡི་མེ་ཏོག་སྙིང་སྟོབས་བཞིན། །དཀར་ཤུབྷི་ར་བྷཱ་ཥཏ། །བྷ་ག་བཱན་བྷ་བ་ཏཔཱ་ད་པདྨ་སཾ་ཨན་ནེ་ན་མེ། །བས་གཏོར་ནས་དེས་སྨྲས་པ། །ཡོངས་སུ་སིམ་པའི་གནས་གྱུར་པ། །བཅོམ་ལྡན་ཁྱོད་ཀྱི་ཞབས་དག་གི། །དུརྡ་ཤ་དུཁ་སནྟ་པཤཱནྟསནྟོ་ཥ་ཤཱ་ལི་ན། ཤ་མ་ཐཱ་གྷ་ཀཱ་པི་བྱ་པདྨོ་ལ་ནི་རེག་པས་བདག། སྐབས་ངན་སྡུག་བསྔལ་གདུང་བ་ཞི། །ཀྱེ་མ་བཅོམ་ལྡན་གཟིགས་པ་གདུང་ས་ན་བི་ཥ་དོ་ཥོ་ཤ་མ་ནཱི་སུ་དྷཱ་བྲྀཏྟཛྲཏྟིརྦ་ཏ་བྷ་ག་བ་ཏསྙིགྷ་མ་དྷུ་རཱ། །བའི་དུག་སྐྱོན་ཚ་བ་ཞི་བར་བྱེད། །ཅི་ཡང་བདུད་རྩིའི་ཆར་པ་ཞི་ཞིང་བསྔགས་འོས་འཇམ་ཞིང་ཡིད་དུ་འོང་། །ཡ་ཡཱ་སྤྲོཥཱ་སྤཝཾ་ཁ་ར་ཏ་ར་བི་ཀཱ་ར་བྱ་ཏི་ཀ་རཾ་བི་མུ་ཙུནྟ༔ཤཱནྟིཤ་ཡ་ཏི་ཧ་གང་གིས་གསལ་བར་རེག་པས་མཆོག་ཏུ་རྩུབ་པའི་རྣམ་འགྱུར་གནོད་པ་དག། རྣམ་པར་བཏང་ནས་རྨོངས་ཏ་མོ་ཧgཔ་ཤུ་ར་པི། ་། ཨི་ཏི་ཏ་སྱ་བུ་བཱ་ཎ་སྱ་བྷ་ག་བཱན་བྷ་བ་པའི་ཕྱུགས་ཀྱང་ནང་དུ་རབ་ཏུ་ཞི་བ་སྟེན། །དེ་སྐད་རྗོད་པར་བྱེད་དེ་ཡིས། །སྲིད་པ་ཞི་སླད་ཤཱནྟ་ཡེ། །ས་་དན་ན་སཾ་ཤུབྷཾ་བི་དེ་དྷེ་དྷརྨ་དེ་ཤ་ནཾ། །བཅོམ་ལྡན་གྱིས། །ཡང་དག་ལྡན་པ་མཐོང་བྱེད་པ། །་དག་པའི་ཆོས་ནི་བསྟན་པར་ཨེམཛད། །མོ་ལི་མུཏྟཾ་ཤུ་ཤུ་བྷེ་ཎ་ཤི་སཱ་ཙ་ར་ཎ་དཱ་ཡཾ། །ས་ཤཱསྟུ༔པྲ་ཡ་ཡོ་ན་དབུ་རྒྱན་མུ་ཏིག་འོད་དཀར་བའི། །མགོ་ཡིས་སྟོན་པའི་ཞབས་དག་ལ། །ཕྱག་འཚལ་སྲིད་པའི་འཁྲུལ་པ་།
