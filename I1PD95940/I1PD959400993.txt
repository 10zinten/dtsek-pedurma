སརྦ་ཏིཙྪཱསྟི་བྲ་ཏ་རཱདྷཱ་ཡཱཏ། །ཐཱ་བསྟི་ར་སཾ་པུརྞ་བི་ཤུདྷིབི་ཐུ་ལས་ཐར། །གྲགས་པའི་དཔལ་ལྡན་རིན་ཆེན་གྱིས། །ཁེངས་ཤིང་དགེ་བ་ལྡན་པ་ཀུན། །མཉན་ཡོད་ཏ་ཤྲཱི་ཡ། །།ཏེ་ག་ཏྭཱ་བྷཀྟ་སཾ་ཡུཀྟ་བྱ་སན་ཏཱ་ཎ་ཀཱ་རི་ཎཾ། །ཛི་ནཾ་དུ་ནི་རྣམ་པར་ཞུགས། །གུས་ལྡན་དེ་དག་སོང་གྱུར་ནས། །གདུང་བ་སྐྱོབ་པར་མཛད་པ་པོ། །རྒྱལ་བ་ཛེ་ཏ་བ་ནཱ་སི་ནཾ་བྷ་ག་བཱནྟཾ་བ་བནྡི་རེ། །ཏེ་ཏ་སྱ་དན་ནེ་ནི་བ་ཙྪིནྡ་སཾ་བཅོམ་ལྡན་རྒྱལ་བྱེད་ཀྱི། །ཚལ་ན་བཞུགས་ལ་ཕྱག་དག་འཚལ། །དེ་དག་དེ་མཐོང་ཉིད་ན་ནི། །འཁོར་སཱ་ར་བནྡྷ་ནཱ། །ས་ཏྲཿཔྲ་སཱ་ད་མཱ་སཱ་རྱ་པྲ་བྲ་ཛྙཱན་ཏཱ་མཱ་ཡ་ཡུ། །བའི་འཆིང་བ་ཆད་གྱུར་ཅིང་། །འཕྲལ་ལ་རབ་ཏུ་དང་བ་ཐོབ། །རབ་ཏུ་བྱུང་ནས་དགྲ་བཅོམ་གྱུར། །ཨ་བྷ་བཏྟཱ་བྱ་པཱ་ཏྱཱ་སྱ་ཤཱyཤཱ་ས་ན་བནྟི་ན། །ཨེ་ཏེ་ཛརྨ་ནི་པཱུཏྠསྨིན་འདི་དག་སྔོན་གྱི་སྐྱེ་བ་ལ། །སྟོན་པ་འོད་སྲུངས་ཞེས་པ་ཡི། །བསྟན་ལ་ཞུགས་པར་གྱུར་ཏོ་ཞེས། །པོ་བཱ་ཙ་བྷ་ག་བཱ་ནི་ཏི། །ཏི་མི་གི་ལ་གི་ལི་སྭཱབྷོ་བིསྟ་རས་ཏཱ་སཾ་བཅོམ་ལྡན་འདས་ཀྱིས་རབ་ཏུ་གསུངས། །རྒྱ་མཚོར་ཉ་མིད་མིད་པ་ཡང་། །སེམས་ཅན་འཇོམས་ལས་དྲྀ་ཏེ། །པཱ་པཀྵ་ཎེ་ན་ཏཱིཀྐཱཀྐིཀྐུརྒྱུ་ཏ་པཱ་ཎ་སཾ་ཀྵ་ཡཾ། །ཀླནྣ་ཀ་ཆགས་བྲལ་ཏེ། །བཀྲེས་པའི་མེ་རྣོན་གྱིས་ཉེན་ཅིང་། །སྐད་ཅིག་གིས་ནི་སྲོག་ཟད་ཐོབ། །དྲི་དང་མི་ལེ་བ་རཾ་ཏ་སྱ་དུ་རཱ་མོ་དཱ་ས་ཧིསྐུ་བྷི། །སོདྲེ་གེརྦ་ལི་བྷིན་གིཀླུབྷཾ་ཛ་བཟོད་ཡིད་འབྱུང་དང་། །བཅས་པའི་ཀླུ་ནི་སྟོབས་ལྡན་གྱིས། །དེ་ཡི་རོ་ནི་མི་གཙང་བ། །ཆུ་གཏེར་ལ་ནི་དྷེསྟ་ཊ། ་། ས་ཤུཥསྦི་མ་ཡཀཱ་ཡཀཱ་ལེ་ན་གྲྭ་ལོ་ཏེམྷི་བྷི། །ཏུ་ཏྟཱ་དག་གི་ངོགས་སུ་འཕངས། །དུས་ཀྱིས་རུས་པ་སྐམ་པོ་ཡི། །རང་བཞིན་ལུས་དེ་ྋཆུས་བཀྲུས་པ། །རི་ར་རཱ་ཤི་སཾ་ཙྪནྣཤེ་ལ་ཀུ་ཊ་ཨི་བཱ་བ་བྷོ། །ཤྭ་བཙྩཾ་བཱཏྨ་ཎ་སཱུ་ཐ་སུ་མནྟོབོ་བརྩེགས་པ་ཁ་བ་ཡི། །ཕུང་པོས་བསྒྲིབས་པ་བཞིན་དུ་མཛེས། །དེ་ནས་མཉན་ཡོད་བྲམ་ཟེ་ནི། །སུ་པུ་ཏྲ་ཏྭཾ་ག་ཏ། །མཱ་ཏུས་ཙཀྲེ་གཌྷ་སྦ་པཱི་བྲོ་གླུཇུཁ་བེ་ད་ནཾ། །མན་ཏུ་ཡི་བུ་ཉིད་གྱུར། །མ་ལ་བཀྲེས་པའི་སྡུག་བསྔལ་ཚོར། །མི་བཟད་མངལ་གནས་དེ་ཡིས་བྱས། །—།
