ཏསྨི་ཡེ་ན་ས་མཱ་ནུ་ག། །མོ་ཧ་ནི་དཱཀླུ་ཡེ་ན་པཱསུ་པྲ་བུདྡྷཔྲ་འདུལ་བའི་ཆོས་ནི་དེ་ལ་བསྟན། །རྨོངས་པའི་གཉིད་སད་ལེགས་པར་ནི། །རབ་སད་རབ་སད་ཉིད་བུདྷཱ་ཏཾ། །ས་ཤཱ་ས་ནཱབྷ་ག་བ་ཏཔུ་ཎྱ་ནི་ཛ་མི་བོསྣ་ཏཾ། །བ་ལོ་ཀཱ་ཐོབ་གྱུར། །བཅོམ་ལྡན་འདས་ཀྱི་བཀའ་ལས་དེས། །རང་གི་བསོད་ནམས་བཞིན་མཐོ་བ། །བཱ་ལོ་ཀཱི་ཡཱ་དྷི་དྷ་སྟུ་པཾ་རཏྣ་དཱི་པཾ་ནྱ་བེ་ད་ཡཏ། །ཌམྦ་ར་ཀཱ་མ་མཱ་སཱ་རྱ་ཤ་ཡ་ཞེས་པ་ཡིས། །མཆོད་རྟེན་རིན་ཆེན་འབར་བ་བྱས། །དེ་ནས་ཡིདལ་གྱིས་དེ་བཞིན་གཤེགས། །ནི་ར་ཐ་ཏ་ཐཱ་ག་ཏ། །བི་ནཱ་ཡ་ཌམྦ་རཾ་ནཱ་མ་ཡཀཾ་ཤི་ཀཱ་པ་ད་པྲ་ད། །ཌམ་བ་ར་ཡི་གྲོང་དུ་བྱོན། །གནོད་སྦྱིན་ཌམ་བ་ར་ཞེས་པ། །བསླབ་གནས་རབ་ཏུ་བྱིན་པས་བཏུལ། །ཙཎྜ་ལ་གྲྭ་མ་མ་བྱེ་རྱ་ཙཎྜ་ལཱི་མ་ལི་ཀཱ་བྷི་དྷཱ། །ས་ཧི་ཏྭཾ་སཥ་བྷིཔུ་ཏྲིཆི་གཏུམ་པོའི་གྲོང་དུ་མངོན་ཕྱོགས་ཏེ། །གཏུམ་མོ་མ་ལི་ཀ་ཞེས་པ། །བུ་བདུན་དག་དང་ལྷན་ཅིག་ཉིད། །ན་ཡོནྣ་ཏྭཾ་བྱ་དྷཱ་ཏ། །ཏེ་ཀརྨ་ཤེ་ཥ་སཾ་པྲཏ་མཱ་ཏཾ་ག་ཀུ་ལ་དྷུ་ཥི་འདུལ་བ་ཡིས་ནི་མཐོ་བར་མཛད། །ལས་ཀྱི་ལྷག་མས་གཏུམ་པོའི་རིགས། །ཐུབ་པས་སུན་ཕྱུང་དེ་དག་ཏ༑། པདྨ་ཀ་ར་ཨི་བཱ་པཱ་པཱུརྦ་མ་མྱ་ཛི་ན་དན་ནཱི་ཏ། །ཨནྡྷ་པྲ་དུ་ཎི་རྣམས། །རྒྱལ་བ་མཐོང་ལས་པདྨ་ཡི༄༅། །འབྱུང་གནས་བཞིན་དུ་དྲི་མེད་གྱུར། །ཐ་ཤལ་སྐྱོན་གྱིས་ཏ་ཏ་རེ་ཥཱ་ཛུ་གུ་པ་ཏེ་ཥཱ་པཱ་པོ་པ་ཏཱ་པ་བི་པུ་ལ་བྱ་ས་ནཱརྟ་རེ་སུན་ཕྱུང་སྨད་པ་དག་ཏུ་འོས་གྱུར་ཅིང་། །སྡིག་པའི་ཉེ་བར་གདུང་བ་ཆེན་པོས་གདུངས་ཤིང་ཉེན་རྣམས་ཥཱ༑། སཾ་ཏ་པྲ་བྲྀཏྟ་ཀ་རུ་ཎཀུ་བི་ཀལྤ་ཧཱི་ན་དཱི་ནཱ་བ་ལསྨྲ་ན་ར་ལ༑། རྣམ་རྟོག་ངན་པ་དང་བྲལ་སྙིང་རྗེས་རབ་ཞུགས་དམ་པ་རྣམས། །དམན་ལ་དམིགས་པའི་རྗེས་སུ་སཱ་ད་དྷི་ཀཾ་བྷ་བནྟི། །ཨ་བཱ་སྱ་པཱ་ཊ་ལ་གྲྭ་མཾ་ས་ནུ་གསུ་ག་ཏསྟ་ཆགས་པས་ལྷག་པར་གྱུར་པ་ཡིན། །དེ་ནས་བདེ་གཤེགས་རྗེས་འབྲང་བཅས། །པཱ་ཊ་ཀྱེལ་ཡི་གྲོང་ཁྱེར་ཏ༑། དྷ་རྩྭ་ཧཱུྃ་ཧ་པ་ཏེཧྥ་ཀྲེ་པོ་ཏ་ལཱ་ཁྱ་སྱ་སཥ་ཐཾ། ཤིཀྵ་བྱོན། །དཔལ་ལྡན་ཁྱིམ་བདག་གྲུ་འཛིན་ཞེས། །བྱ་ལ་ལེགས་པའི་གཏམ་དག་མཛད། །བདེ་བར་—།
