སུ་དྷཱ་བརྩེ་སུ་ཁ་སྤ་ལེ་ནི་ཤཱ་ཀ་རེ། །དིགྲ་དྷུ་བ་ད་ན་དནེ་ཧརྗེ་དགའི་བདུད་རྩིའི་ཆར། །རེག་བདེ་མཚོན་བྱེད་ཕྱོགས་ཀྱི་ནི། །བུ་མོའི་བཞིན་གྱི་མེ་ལོང་དག། དགའ་མཱུ་ཏྟཱ་ཨི་བོ་དི་ཏེ། །ཐོ་ཎ་ཀོ་ཊི་ད་དནཱ་གེ་བི་མཱ་ན་མཱ་ན་ནཾ་དྷུ་ཏེ། །བ་གཟུགས་ཅན་བཞིན་ཤར་ཚེ། །ལྟད་མོ་ལ་ནི་གཞན་གཟུགས་ཀྱིས། །མཐོ་རིས་ས་ལ་འོངས་པ་ཡིསི། །ཀོ་ཏུ་ཀཱ་ད་རྒྱ་རཱུ་པ་ཎ་སཱརྒཏ་བྷུ་བ་མི་བཱ་ག་ཏཾ། །ཏསྨིནྣཱ་པ་བྱཱཏྨ་མ་མཛེས་པའི་གདོང་བཞིན་གཞལ་ཡས་ཁང་། །གྲོ་བཞིན་བྱེ་བས་མདུན་དུ་མཐོང་། །དེར་ནི་ལྷ་ཡི་བུ་མོ་དཱིཤྩ་ཏ་སྲསྐྲྀ་ད་ཤཱི་གཱ་ནཱ། །དི་ཤཉྩནྡོ་ད་ཡཱ་ནནྡ་བི་ཧ་ར་ཡེ་བ་དག། རྒྱགས་པ་དང་བཅས་བཞི་མཐོང་སྟེ། །ཟླ་བ་ཤར་བའི་ཀུན་དགའ་དང༌། །རྣམ་པར་རྩེ་ཕྱིར་སཾ་ག་ཏཱ། །ཏཱ་སྭཱ་མ་དྷེ་བ་རཱ་ཀཱ་རཾ་ར་མ་མཱ་ཎེ་བ་ལོ་ཀ་ཡཏ། །ཏ་འགྲོགས་པ་བཞིན། །དེ་དབུས་མཆོག་གི་རྣམ་པ་ཅན། །དགའ་བ་དང་ལྡན་མཐོང་གྱུར་ཏེ། །གཞོན་རུ་ཎ་པྲ་མ་སཾ་བྷ་ར་མི་བ་སཱ་ཀཱ་ར་ཏྭཾ་ག་ཏཱ། །ནུའི་དགའ་བའི་ཚོགས་རྣམས་དག། རྣམ་བཅས་ཉིད་དུ་གྱུར་པ་བཞིན། །རིན་ཅེན་རྣ་རྒྱན་ཅེས་ནས་བོས་པར་གྱུར་རནྣ་ཀུཎྜ་ལ་ཀེ་ཡཱུ་ར་ཀི་རཱི་ཊི་ཀ་ཙི་རཾ་གྱི་བར་ཚིགས་བཅད་གསུམ་དྷརྨ་པཱ་ལས་འགྱུར་ཆད་བསབས་པའོ། །རིན་ཆེན་རྣ་རྒྱན་དཔུང་རྒྱན་དང༌། །།ཟུར་ཕུད་འོད་ཤུ་བྷི། །ཨཱ་ལི་ཁན་མི་བཉྩརྱ་མ་མནཱ་དཾ་དི་ཤྭ་མུ་ཁེ། །ཏ་་་ཟེར་རིང་པོ་ཡི། །མཚམས་ལས་འདས་པའི་ངོ་མཚར་དག། ཕྱོགས་ཀྱི་གདོང་ལ་འདྲི་བ་བཞིན། །དེར་སྱ་ཏཱ་མདྡྷཱ་ཏཱཾ་ཛྲབྷྱཱ་སཾ་བྷོ་ག་སུ་ཁ་སཾ་པ་དཾ། །མེ་ན་ས་སུ་ཀྲྀ་ཏཱ་ཁྱ་ནི་ལེགས་བྱས་ལས་བྱུང་བའི། །ལྗོན་ཤིང་རྒྱས་པའི་འབྲས་བུའི་དཔལ། །ལོངས་སྤྱོད་བདེ་བ་ཕུན་སུམ་སྱ་ཏ་རོ་སི་ཏྭཾ་ཕ་ལ་ཤི་ཡཾ། །པཱི་ཏྭཱ་ཏེ་ན་ཀྲྀ་ཏཱ་ཏི་སྭཱ་དུ་པཱ་ནཱ་ཤ་ཚོགས། །རྨད་བྱུང་དེ་དག་མཐོང་བར་གྱུར། །དེ་ནས་གྲོ་བཞིན་བྱེ་བ་རྣ༄༅། །དེ་དག་དགའ་བས་ནཱ་དི་བྷི། །ཏྭཾ་ཐོ་ཎ་ཀོ་ཊི་ཀཎྞོ་ཐ་ནི་ནཱ་ཡ་ར་ཛ་ཙཱ་སུ་ཁཾ། །ཏཱ་རཱ་མགྲོན་གཉེར་དེ། ：ཞིམ་པའི་བཟའ་བཏུང་དག་ལ་ནི། །མཚན་མོ་བདེ་བར་བོས་པར་གྱུར། །དེ་ནས་།
