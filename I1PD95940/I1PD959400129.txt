ཁྱ་པྲ་ཏེ་ན་རྒྱ༑་བྷ་སཱཏ་ཀི་རོ་ཎ་ནི་ཀ་རཾ་པུ་ཎྱ་བྷཱ་ཛཱ་དྷཱ་ཛཱ་གྲེ། །བརྟེན་རྣམས་ཀྱིས། །གསོ་སྦྱོང་ཞེས་བྱའི་བརྟུལ་ཞུགས་ངེས་པའི་ཆོ་གས་རྒྱལ་མཚན་ལ་བཀོད་ན། །ཛོ་བཱ་དཱི་པེ་ཤ་ཀ་ལ་ཛ་ན་དྷ་བྷི་བླ་ཏན་བརཱུ་ཏྲ་བསྦི་ར་ཛམྦུའི་གླིང་དུ་མཐའ་དག་འགྲོ་ཚོགས་མངོན་འདོད་དོན་རྣམས་ཆར་ཉིད་འབེབས། །གཞན་དོན་བརྟན་པ་ར་ཧི་ཏ་བྱཱཔ་ཡེ་ཤྲ་ཧུ་ཏྭཾ་ཏཏ། །ཨི་ཏུཏྟཱ་ཏྱཱ་མ་ར་ཏ་ཏུ་ཕ་ལཾ་སཱ་པར་གྱུར་པས་ཁྱབ་པར་བགྱིད་སླད་དེ་ནི་བཟུང་བར་མཛོད། །དེ་ལྟར་མཛེས་མ་རྣམས་ཀྱིས་འཆི་མེད་ད་རཾ་སུན་རཱི་བྷིཔྲ་མདྡྷ་མ་པྲ་ཎ་ཡ་སུ་བྷ་གི་དཏྟ་མཱ་སཱ་ཤིང་འབྲས་ལེགས་བཏུས་གུས་བཅས་ཤིང༌། །ལྷག་པར་མཛའ་སྡུག་ལྡན་པས་སྐལ་བཟང་ནོར་བུ་བྱིན་པ་བྱ་རནྣཾ། །བཱ་ལཱ་ཧཱ་ཊྚཾ་བི་ཛི་ཏ་པ་བ་ནནྟ་པྲ་ཧྲིཊཾ་ཏུ་རཾ་གཾ་སོ་པྟ་ཏུ་རྒྱ་སཱ་ན་བླངས་བྱས་ནས། །མགྱོགས་འགྲོ་རབ་དགའ་སྤྲིན་ཤུགས་ཞེས་བྱ་རླུང་ལས་རྒྱལ་བ་ལ་ཞོན་ཏེ། །དེ་ཡི་ལམ་ག་ར་མ་ག་མལླབྷ་མཱརྒོ་པ་དེ་ཤ། །ཏསྨིན་ཀཱ་ལེ་བི་པུ་ལ་ཀུ་ཤ་གྱི་གདམས་ངག་ཐོབ་པས་དེ་ཡང་རང་གི་གྲོང་ཁྱེར་སོང་། །དུས་ཚོད་དེར་ནི་དྲི་མེད་གྲགས་ལྡན་ས་ཡི་ལེསཱརྒ་མཱནཾ་པྲ་ཡཱ་ཏེ་བཱ་རཱ་ཎ་ཧཱུཾ་བི་ཥ་ད་ཡ་ཤ་སི་ཀླུ་པ་ཏོ་བྲཧྨ་དཏྟ། །བདག་པོ་ཚངས་པས་བྱིན། །ཡངས་པའི་དགེ་བས་མཐོ་རིས་ལམ་དུ་རབ་གཤེགས་བཱ་རཱ་ཎ་སཱི་@རུ། །ཤྲཱི་མཱན་སརྦ་པྲ་ཎ་ཡི་ཕ་ལ་དསུ་པྲ་ཡ༔པོ་ར་མུ་ཁྱིར་ལོ་ཀ་ཏཱ་ཎཱི་དཔལ་ལྡན་འཇིག་རྟེན་བསྐྱབ་བློ་རྣམ་བསྐྲུན་སློང་བ་ཀུན་གྱི་འབྲས་བུ་སྟེར། །ཤིན་ཏུ་དགའ་ནི་གྲོང་ཁྱེར་ཏ་བི་ཧི་ཏ་མ་ཏིདྷརྨ་རཱ་ཛཱ་བྷི་ཥིཏྟ། །ཏ་ཏཤི་རསྣ་ན་བི་དྷི་ཀྲ་མ་ཎ་ཏཏྤཉྩ་ད་གཙོ་བོས་ཆོས་ཀྱི་རྒྱལ་སྲིད་ལ་དབང་བསྐུར། །དེ་ནས་མགོ་བཀྲུ་ཆོ་གའི་རིམ་པ་ཡིས། །གསོ་སྦྱོང་དག་ལ་ཧཱུྃ་དྷཱ་ཛ་མུངྒི་རནྣཾ། །ས་པོ་ཥ་དྷོ་ཡོ་ཥི་ཏ་ཨེ་བ་དྷཱི་ཏཱ་ཙ་ཀཱ་ར་པི་ཤཱི་པ་རི་ཉེར་གནས་བཅོ་ལྔ་ལ། །རིན་པོ་ཆེ་དེ་རྒྱལ་མཚན་རྩེར་བཏགས་ནས། །སྣ་ཚོགས་འདོད་པས་ཡོངས་སུ་པུཎ་ཀཱ་མཾ། །ཀྲྀ་ཏྭཱ་ཡཱ་ཏཱཾ་པ་ར་ཧི་ཏ་ཕ་ལོ་པཏྟཱ་རཱ་ཎཾ་ཤ་ཏེ་ན་སྦི་ཏྭཱ་གང་བར་བྱས། །གཞན་ལ་ཕན་པས་འབྲས་བུ་ལྡན་པར་གཤེགས་བགྲོད་བྱས་ཤིང་ལོ་བརྒྱར་ནི། །རྒྱལ་།
