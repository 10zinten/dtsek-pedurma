ཆུ་འཔི་ནི་བཱ་དྲི་བྱ་ཏ་ཡཱ་ཙ་ཀཿ། །ས་ཀ་དཱ་ཙིཛཾ་ཧཱ་ཡཱ་ཏཾ་པྲ་ཤཱནྟེནཱ་པ་ཐ་སྦི་ཏི། །སློང་བ་པོ་ནི་མཐོང་མ་གྱུར། །ནམ་ཞིག་དགེ་སློང་ཀུན་དགའ་བོ། །རབ་ཞི་འཕགས་པའི་ལམ་གནས་པ། །ཨཱ་ནནྡ་བྷིཀྵ་མ་ལོ་ཀ་པ་པྲཙྪ་པི་ཏ་རཾ་པུ་རི། །ཏཱ་ཏ་བྲ་ཏི་བི་ཤེ་ཥ་སྱ་ཀོཏྟཾ་ཁྱིམ་དུ་བྱོན་པ་དེ་མཐོང་ནས། །མདུན་དུ་ཕ་ལ་རབ་དྲིས་པ། །ཡབ་གཅིག་དྲི་མེད་དགེ་བ་ཅན། །བི་མ་ལ་ཤ་ལི་ན། །།།ཡ་སྱ་སནྡན་ན་ནི་བ་མ་ནཔ་པྲ་པཱ་ད་ཏི། །འདི་ཡི་བརྟུལ་ཞུགས་འདི་ནི་ཅི། །གང་ཞིག་ཡང་དག་མཐོང་ཉིད་ན། །འཕྲལ་ལ་ཡིད་ནི་དང་བར་བྱེད། །པུ་ཏྲ་ཀྱེ་ཏི་བ་ཙཐུ་ཏཱ་སུ་དྷི་རསྟ་མ་བྷཱ་ཥཏ། །པུ་ཏྲ་ས་ཏྭཱ་པྲ་ཀཱ་ཤོ་ཡཾ་ཤཱནྟི་ཅེས་པ་བུ་ཡི་ཚིག་ཐོས་ནས། །ཤིན་ཏུ་བརྟན་པས་དེ་ལ་སྨྲས། །བུ་གཅིག་སྙིང་སྟོབས་རབ་གསལ་འདི། །བྲ་ཏི་པ་རི་གྲ་ཧ། །ཡསེ་ཏུཔ་ར་ལ༔ས་མསྟ་ཛ་ག་ཏཱཾ་སཾ་སཱ་རོ་གྷོ་ཞི་བའི་བརྟུལ་ཞུགས་ཡོངས་སུ་འཛིན། །གང་ཞིག་འཁོར་བའི་མཚོ་ལ་མཐའ་དག་འགྲོ་བའི་ཟམ་སྟེགས་རཱཎྞ་བེ་ཀྲོ་དྷ་བྱ་དྷི་ཙ་ཀིཏྠ་ཀཤ་མ་སུ་དྷཱ་སཱ་རེ་ཎ་ཀྲྀཀྵ་པ་ཧ། ་། དྲང་པོ་དང་། །ཁྲོ་བའི་ནད་ཀྱིས་གསོ་བྱེད་དག་དང་ཞི་བའི་བདུད་རྩིའི་སྙིང་པོས་སྲེད་འཕྲོག་དང་། །དོ་ཥོཏིཥྛ་ཏི་མོ་བི་རཱ་མ་ཏ་ར་ཎིཅུདྡྷཔྲ་བུདྡྷ་དྷུ་ཏིསྟ་སྱ་ཐཱ་བ་ཀ་ཨེ་ཥ་ཤཱནྟ་སྐྱོན་གྱི་ཚོགས་ཀྱི་མུན་པ་ཞི་བྱེད་ཉི་མ་སངས་རྒྱས་འོད་ཟེར་རབ་རྒྱས་པ། །།དེ་ཡི་ཉན་ཐོས་ཞི་བའི་ཡིད་ཅན་མ་ནཱ་སཱ་མཱ་ནནྡ་ནཱ་མཱ་གྲ་ཎི། །ཨ་བྷི་དྷཱ་ནཾ་བྷ་ག་བ་ཏཐུ་ཏཱི་བ་མ་དྷུ་རྣམས་ཀྱི་མཆོག་གྱུར་འདི་ནི་ཀུན་དགའ་བོ། །བཅོམ་ལྡན་འདས་ཀྱི་མངོན་བརྗོད་དག། ཐོས་པ་ཉིད་ནི་ར་ས་ར། །བ་བྷཱ་བོདྡྷཱ་ཏ་རོ་མཱཉྩ་པཱདྡན་ཀུ་ཤ་ལོ་ད་ཡཱཏ། །ཨཱ་ནནྡ་མ་ཐ་སྦྲང་རྩིའི་དབྱངས། །སྔར་སྐྱེ་དགེ་བ་རྒྱས་པ་ལས། །བ་སྤུ་རབ་ཏུ་ལངས་པར་གྱུར། །དེ་ནས་ཀུན་དགའ་སཱ་ནནྡཔྲ་ཎ་མྱ་པྲ་ཎ་ཡོརླུ་ཁ། །ས་བྷིཀྵ་སཾ་གྷི༔ས་ཧི་ཏཾ་སརྦ་དང་བཅས་དེས། །གུས་པས་མངོན་ཕྱོགས་ཕྱག་འཚལ་ཏེ། །དགེ་སློང་ཚོགས་བཅས་ཀུན་དགའ་བོ། །བྷོ་གི་ར་པཱུ་ཛ་ཡཏ། །ཨོཥྛ་ཀཱཏྨ་ཧ་ཏེ་ནི་བ་ག་ཏཱ་ཛེ་ཏ་བ་ནནྟ་ལོངས་སྤྱོད་ཀུན་གྱིས་རབ་ཏུ་མཆོད། །།དེ་ནས་འདུན་པས་དེ་ཉིད་དང་། །ལྷན་ཅིག་རྒྱལ་བྱེད་ཚལ་སང་།
