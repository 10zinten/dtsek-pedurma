ཏཾ༑་། ཨི་ཏི་སྤཋ་གི་རཱ་ཏ་yསྣེ་ཧེ་ན་ཙ་པ་བི་སྨི། །མུ་མོ་ཙ་ལུབྷཱ་ཀཕྲོགས། །ཞེས་པ་དེ་ཡི་གསལ་བ་ཡི། །ཚིག་དང་མཛའ་བས་མཚར་སྐྱེས་པ། །རྔོན་པ་དེ་ནི་བྱམས་པ་པཱི་ཏྭཱ་ཧ་རི་ཎཾ་ཧ་རི་ཎཱི་ས་ཁཾ། །མྲྀ་ག་ཡཱ་ཐ་བ་ཏི་སོ་ཧཾ་མཱ་ནནྡསཱ་ཀུ་རཾ་ཡིས། །རི་དྭགས་རི་དྭགས་མོར་བཅས་བཏང་། །རི་དྭགས་ཁྱུ་ཡི་བདག་དེ་ང་། །རི་དྭགས་མོ་དེ་ཀུན་གི་ཀཱ། །ཨི་ཏི་ཥ་པཱི་ཏི་སམྤནྣཱ༔པྲཤྲཱིཏྟ་མ་ནུ་བརྟ་ཏེ། །ཐུ་ཏྲེ་ཏི་དགའ་བོ། །དེ་ལྟར་མཛའ་བའི་འབྲེལ་པ་འདི། །སྔོན་གྱི་སྤྱོད་པས་རྗེས་སུ་ཞུགས། །ཞེས་པ་བདེ་བཱ་ཀྱི་སུ་ག་ཏ་སྱ་སཉྫ་ལཛྙཱ་ནི་ལཱི་ནི་ཨི་བ་བྷིཀྵ་པ་སྟེ། །ཨཱ་ནནྡ་མཱ་གཤེགས་གསུང་ཐོས་དགེ་སློང་ནི། །དེ་དག་ཐམས་ཅད་ངོ་ཚས་ཞུམ་པ་བཞིན། །འདོད་པས་བརླན་པས་ནརྡ་མུ་ཁཱ་ར་བིརྡ་པྲ་བྷཱ་བྷི་རཱ་མཾ་ད་ཛཾ་ཤུསྤྲོ་ཧཱརྔ། །ཨི་ཏི་ཀུན་དགའ་དང་བཅས་པར། །འོད་ཀྱིས་མངོན་དགའ་ཀུན་དགའི་བཞིན་པད་བལྟས། །ཞེས་པ་དགེ་ཀྱེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་སཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་དྷ་ན་པཱ་ལཱ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ནོར་སྐྱོང་བ་དཱ་ནཾ་ཨུ་ན་བིདྡྷཔལླ་ཝ། །གི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་ཉི་ཤུ་རྩ་དགུ་པའོ། །།ཛ་ཡ་ཏི་ས་སཏཱ་བི་ཤེ་ཥསཏཱ་བ་ཏྭཾ་སརྦ་སཏཱ་སུ་ཁ་ཧེ་ཏུ། །པ་གང་ཞིག་སྙིང་སྟོབས་ལྡན་པའི་སྙིང་སྟོབས་ཀྱི༄༅། །ཁྱད་པར་སེམས་ཅན་ཐམས་ཅད་བདེ་བའི་རྒྱུ། །དེ་ཧ་ད་ལ་ནེ་པི་ཤ་མ་ཡ་ཏི་ཀོ་པཱཙླི་ཤཱནྟི་མུཙྩརླ། །དྷམྨོ་པ་དེ་ལུས་ནི་ཞིག་ཀྱང་ཁྲོ་བའི་མེ་རྣམས་དག། མཆོག་ཏུ་ཞི་བར་འགྱུར་དེ་རྒྱལ་གྱུར་ཅིག། ཆོས་བསྟན་པ་ལ་ཤེ་བྷ་ག་བཱན་ཀོཎྜི་ནྱ་སྣ་གྲ་བནྟི་ན། །བྷི་ཀྵཀ་ཐཱ་པྲ་སཾ་གེ་ན་པྲཊོ་བྷིཀྵ་བ་མངོན་འདུག་པ། །དགེ་སློང་ཀོཎྜི་ནྱ་ཡི། །གཏམ་གྱི་ཞར་ལ་དགེ་སློང་གིས། །དྲིས་ཤིང་བཅོམ་ལྡན་བྷི་ར་བྱ་དྷཱད། །བྲཧྨ་དཏྟ་སྱ་ཏ་ན་ཡོ་བཱ་རཱ་ཎ་སིཾ་མ་ཧཱི་པ་ཏེ། །ཀཱ་ཤི་པུནྡ་གྱིས་བཀའ་སྩལ། ་། ་ཝཱ་རཱ་ཎ་སཱི་དག་ཏུ་ནི། །ས་བདག་ཚངས་པས་བྱིན་གྱི་བུ། །ཀཱ་ཤི་མཛེས་པ་ཞེས་།
