ཛ་ན་ནྡྷཱཉྩ་བྷུ་ཛཾ་ག་མ། །ཨཱ་དཱ་ཡ་དི་བྱ་པུཥཱ་ཎི་པུ་ཎྱ་ཛེ་ཏ་བ་ནཾ་ཡ་ཚིག་དག་ཐོས་ནས་ལག་འགྲོ་ནི། །མེ་ཏོག་མཆོག་དག་ཡོངས་བཟུང་ནས། །དག་པའི་རྒྱལ་བྱེད་ཚལ་དུ་ཡོ༑། སུ་ག་ཏ་ཤྲ་མ་མཱ་སཱ་རྱ་དྷརྨ་ཤྲ་བ་ཎ་སཾ་ག་ཏྭཾ། །ད་དན་སོང་། །ཆོག་ཤེས་སྒོར་ནི་མངོན་ཕྱོགས་དེས། །བདེ་བར་གཤེགས་པའི་གནས་ཕྱིན་ཏེ། །དེར་ནི་ཆོས་པརྟ་དཾ་ཏ་ཏྲ་ས་སནྟོ་ཥ་མུ་ཁྱོརྨུ་ཁ། །ཏ་ཏཱ་པ་བྱ་རྫི་ནཾ་ཀཱནྟ་བ་དག་ཉན་པ་ལ། །ཚོགས་པའི་འཁོར་དག་མཐོང་བར་གྱུར། །བཞིན་རས་མཛེས་ཤིང་སྤྱན་ཡངས་པ། །ད་ནཾ་དཱིགྷ་ལོ་ཙ་ནཾ། །པུཎྞེནུ་པདྨ་བ་ན་ཡོ་རི་བ་མི་ཏཱི་སུ་ཁ་པྲ་དཾ། །ཟླ་བ་ཉ་དང་པད་ཚལ་དག། འགྲོགས་པ་བཞིན་དུ་བདེ་སྟེར་བའི། །རྒྱལ་བ་དེར་ནི་མཐོང་བར་གྱུར། །ཨུ་པ་དེ་ཤ་ཀྲྀ་ཏཱ་བྱཀྟ་མ་དྷ་རེ་ཎཱ་པ་རཱི་ཀྲྀ་ཏྭཾ། །ཏརྫ་ཡནྟ་མི་ཉེ་བར་སྟོན་བྱེད་མཆུ་ཡིས་ནི། །ཆགས་ལྡན་རྣམས་ཀྱི་དམར་བ་ཉིད། །རབ་རྒྱས་དམན་པ་ཉིད་བྱས་ཤིང་། །བོཏྟིཀྟ་རཱ་ག་བརྒ་སྱ་རཀྟ་ཏྭཾ། །ནི་རཱ་བྷ་ར་ཎཱ་ལཱ་བཱ་རྱ་ཀརྞ་པཱ་ཤ་བི་བྷཱུ་གསལ་བར་ཟིལ་གྱིས་གནོན་པ་བཞིན། །རྒྱན་དང་བྲལ་བའི་མཛེས་སྡུག་གིས། །རྣ་བའི་ཞགས་པ་རབ་ཥི་ཏཾ། །དན་ཡནྟཾ་མི་བོཏིཀཾ་ནི་རཱ་བ་ར་ཎ་ཤུ་ནྱ་ཏྭཾ། །དྷརྨ་དཱི་པ་བརྒྱན་ཅིང་། །སྒྲིབ་པ་དང་བྲལ་སྟོང་པ་ཉིད། །རབ་ཏུ་རྒྱས་པ་སྟོན་པ་བཞིན། །དཔུང་པ་ཆོས་ཀྱི་ཀ་རོ་བ་བྷུ་བི་བྷཱ་ཎཾ་དཱ་ན་ཤོ་བྷི་ཏོ། །པྲ་བྷཱ་བ་བྷ་བ་ནསྟམྦྷོ་ཤཱ་གཉིས་འཐུང་ལགི། སྦྱིན་པས་མཛེས་པ་རྣམ་པར་འཛིན། །ནུས་མཐུའི་ཁང་པ་དག་གི་ནི། །ཀ་བ་ཏ་ཀུམྦྷ་མ་ཡཱ་བི་བ། །དི་ཤནྟཾ་ཙ་ར་ཎ་ཙྪཱ་ཡཾ་ཙཱ་བ་རཱ་བྷ་ར་ཎཱ་བྷཱུ་པ། །ལབྷཱ་པྲ་གསེར་གྱི་རང་བཞིན་བཞིན། །ཆོས་གོས་རྒྱན་དང་ལྡན་པ་ཡིས། །ཞབས་ཀྱི་གྲིབ་མ་སྤྲོ་བྱེད་པ། །ས་གཞི་བྷོ་དྷི་བི་ཧི་ཏཱ་མི་བ་རཱ་ཛཱི་བི་ཏེ། །དེ་ཧི་ཀཱནྟི་བི་ཏཱ་ནེ་ན་རྟོགས་པ་ཐོབ་པ་ལ། །པདྨའི་གྲོགས་རྣམས་ཀྱིས་བསྒྲུབས་བཞིན། །ལུས་ཀྱི་མཛེས་པ་རྒྱས་པ་ནི། །ན་ཡ་ནཱ་མྲྀ་ཏ་བནྟི་ཎ། །སཾ་སཱ་ར་ག་ཏ་སཾ་ཏཱ་པ་བཱ་ར་ཡནྟི་པ་སྤྱན་གྱི་བདུད་རྩིའི་ཆར་འབེབས་པས། །དམ་པ་རྣམས་ཀྱིས་འཁོར་བ་ཡི། །མྱ་ངན་ཡོངས་གདུང་།
