བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་བྷཾ་ག་ས་མུ་དྷཾ་ཏ༔སུ་བྷ་དྲསྟམྦྷ་ཥ་ཏ། །ཛཱ་ནཱ་བྱཱ་བ་ས་ར་ནཱཎྞཔྲ་སཱི་པས་རབ་འཁྲུགས་པ། །རབ་བཟང་གིས་ནི་དེ་ལ་སྨྲས། །གཟིར་བ་ཡིས་ནི་སྐབས་མི་ཤེས། །སློང་ལ་ད་ད་ཡཱཏྲི་ན། །སུ་ལ་བྷོ་དན་ཐ་ཛི་ནམྨཱ་མི་བ་བྷ་བ་ནིཀྵ་ར། །ཙྪ་དང་ཞིང་བརྩེ་བར་མཛོད། །རྒྱལ་བ་རྙེད་སླ་རབ་ཏུ་སྟོན། །རྩུབ་པོ་འདི་ལྟ་®ཡོངས་མ་མཛད། །གྲིབ་ཡཱ་ཡཱ་མུ་པ་བི་བྱ་ཤི་ཏ་ལ་ཏ་ཡཱ་ལནྡྷ་པྲ་མོ་དོཏྠ་བི་རཱ་རུ་ཧཱུརྟ་ཕ་ལེས་དཱ་བསིལ་དག་ལ་ཉེ་བར་འཁོད་ཅིང་བསིལ་བས་རབ་དགའི་དགའ་སྟོན་ཐོབ་གྱུར་པ། །རྟག་ཏུ་གསར་པ་ན་བ་ན་བིསཾ་པཱི་ཎི་ཏ་སཱུ་དྷ་གི། །ཛཱ་ནཱ་ཏུ་ཤི་ཏ་བཏཱ་གསར་པའི་ལམ་འགྲོན་འཛེགས་ཤིང་འབྲས་བུ་འཛིན་པས་གཟིར་བའི་ཤིང་། །བསྟན་པ་དག་ལ་མཉེས་ལ་སྱ་སུ་ལབྦྷ་དྷ་སབྷཱཟི་ནཾ་ཤཱ་ཁཱ་བྷཱ་ཛ་ན་མོ་ཊ་ལ་པྲ་ས་ར་ཎ་ཀླུ་གཤིན་སློང་བ་ཀུན་གྱི་གཞི་ནི་རྙེད་སླར་གྱུར་པ་ཡི། །ལོ་མ་དག་ནི་བཅོམ་དང་དཀྲུགས་དང་རབ་ཏུ་དྲངས་ཤཱ་བ་ལི་ཀསྟ་རོ། །ཨི་ཏི་ཡནྟཱཏ་སུ་བྷ་དྲེ་ཎ་ཡཱ་ཙི་ཏོ་པི་པུ་ནཔའི་ཉོན་མོངས་འཕྲེང་བ་སུ་ཡིས་ཤེས། །དེ་ལྟར་འབད་པས་རབ་བཟང་དེས། །ཡང་དང་ཡང་དུ་པུ་ན། །ནི་རཱ་ནན་མ་ནནྡཔྲ་ཏི་ཥ་དྷ་ཤ་ཏིརྒྱུ་དྷཱཏ། །བྷ་ཞུས་གྱུར་ཀྱང་། །དགག་པ་རྒྱ་ཡི་ཀུན་དགའ་བོས། །དེ་ནི་ཀུན་དགའ་མེད་པར་བསྒྲུབས། །བཅོམ་ག་བཱ་ན་སི་སཧྥ་ཛྷཤཡ་མེ་ཏུ་སུ་རཱ་ལ་ཡཱཏ། །ཨུཙླི་རཱ་ནརྡ་ལྡན་ཀུན་མཁྱེན་གཟིམས་མལ་དུ། །ལྷ་ཡི་གནས་ལས་མངོན་ཕྱོགས་ཏེ། །རབ་བཟང་ནང་དུ་ཐོང་ཞིག་མ་བ་དདྷུ་བྷ་དྲཔྲ་བྲི་ཤ་ཏི་ཧི། །ལབྷཱ་པྲ་བེ་ཤསུ་ག་ཏཾ་ཛ྄ཟླ་ཅེས། །ཀུན་དགའ་བོ་ལ་གསུང་ཚར་གསུངས། །འཇུག་པ་ཐོབ་སྟེ་ནོར་མེད་ཀྱིས། །གཏེར་བཞིན་ནི་དྷི་མི་བཱ་དྷ་ན། །ཛ་གཱ་ད་སཱ་སྲུ་བ་ད་ནསུ་བྷ་དྲཔྲ་ཎི་པ་ཏུ་བདེ་གཤེགས་མཐོང་གྱུར་ནས། །རབ་བཟང་གདོང་ནི་མཆི་མ་དང་། །བཅས་པས་ཕྱག་འཚལ་དེར་གསོལ་ཏ༑། བྷ་ག་བནྡྷཱ་བ་ན་མྱ་བྱ་ཀེ་ཡཾ་བྷཱ་ག་ད་རི་དྲ་ཏཱ། ་། སཾ་ཏྱཱཀྟ་པ༑། བཅོམ་ལྡན་དེ་རིང་འཇིག་རྟེན་ནི། །སྐལ་བས་དབུལ་བ་ཉིད་འདི་ཅི། །དེ་རིང་ཁྱོད་ཀྱིས་ཕྱེ་།
