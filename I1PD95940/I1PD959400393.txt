མཎྜི་ཏ། །ས་བུ་རཾ་པྲ་ཏ་རཱ་གནྟཱ་དཱམྷི་ཀས་ཧཱ་བྷིཀྵ་བྷི། །རཱ་ཛ་མཱརྒ་དུ་གཉེར། །ཚུལ་འཆོས་ཅན་དེ་དགེ་སློང་དང་། །ལྷན་ཅིག་ནང་པར་གྲོང་ཁྱེར་འོངས། །རྒྱལ་པོའི་ལམ་པྲ་བི་ཤ་ཏས་ཏ་མྱ་བྱ་ལ་ཀུཉྫར། ་། ཨུཏྠཾ་ཛུ་ཏཱ་མ་བྷི་མུ་ཁཀྲོ་དྷཱནྟོ་དུ་རབ་ཞུགས་པ། །དེ་ལ་གླང་ཆེན་ནོར་སྐྱོང་ནི། །ཁྲོ་བས་ལོང་བཞིན་གཏུམ་པ་དེ། །མངོན་སུམ་ཉིདྷ་ན་པཱ་ལ་ཀཱ། ་། ཨི་ཏུཀྟ་དེ་བ་དཏྟེ་ན་ཚྭ་པ་ཏིརྨི་ཏྲ་བཏྠ་ལ། །རབ་ཏུ་གཏང་བར་བྱོས། །ལྷས་བྱིན་གྱིས་ནི་དེ་སྨྲས་ཚེ། །མི་བདག་གྲོགས་ལ་མཉེས་གཤིན་པས། །བུདྡྷ་པྲ་བྷ་ཐཾ་སིནྡྷ་ནོ་ཙི་ཀི་ན་ད་བཱདྨ་ཁ། །དེ་བ་དཏྟོ་པ་ནི་རྒ་སངས་རྒྱས་མཐུ་ནི་བསམས་བྱས་ནས། །གདོང་སྨད་ཅི་ཡང་མི་སྨྲ་གྱུར། །རྒྱལ་པོའི་གྲོགས་ཀྱིས་ངན་ཏྱཱ་རཱ་ཛ་སོ་ཧཱརྡ་དུརྨ་ད། །པྲ་ཧ་ཧསྟི་མ་ཧཱ་མཱ་ཊཾ་ཧཱ་རཾ་ད་ཏཱ་སྱ་ཏོ་རྒྱགས་པའི། །ལྷས་བྱིན་དག་ཀྱང་ཕྱིར་བྱུང་ནས། །གླང་པོའི་བློན་པོ་ཆེན་པོ་ལ། །དོ་ཤལ་རྔན་པར་བྱིན་ཥ་ཎཾ། །ཐ་མ་ཎབཱ་ཏ་ར་གནྟ་པུ་རཾ་བྷིཀྵ་ཤ་ཏེབྲྀ་ཏ། །བྲེ་རཱ་ནས་སྨྲས། །དགེ་སློང་དགེ་སྦྱོང་བརྒྱས་བསྐོར་ནས། །ནང་པར་གྲོང་ཁྱེར་འོང་བར་འགྱུར། །དེ་ཡུལ་ཎཱི་ཡསཱུ་ཡཱ་ཏསྨི་ག་ཛ་ཨི་ཏྭཱ་ཧ་བྷཱ་པ་ཏི། ་། དེ་བ་དཏྟ་བ་ཙཐུ་ཏྭཱ་ཁྱོད་ཀྱིས་གླང་པོ་ཆེ། །གཏང་བར་བྱོས་ཤིག་ཡིས་བདག་གསུང་། །ལྷས་བྱིན་ཚིག་ཐོས་གཉིས་འཐུང་གི། །ཏ་ཐེ་ཏུཙྪིནྡཾ་པཱ་དྷི་པ། ་། ཐེ་ཎཱི་ཧི་མེ་ཥ་མཱུ་ཎཱ་ཎཱ་མེ་ཀ་ཡཱ་ཏཱ་ནུ་པཱ་ཏི་བདག་པོས་དེ་བཞིན་ཞེས་པ་སྨྲས། །གང་ཕྱིར་ལུག་དང་རྨོངས་པའི་ཕྲེང་། །གཅིག་སོང་རྗེས་སུ་ལྷུང་བ་ནཱི༑། ཚྭ་ཏྭཱ་པི་ཏཱ་མ་བྷི་པྲ་ཡཾ་སཏྠ་ཛྷཱ༔པཱ་བ་ཙེ་ཏ་སཾ། །པྲ་ཏས་ཉིད། །ཐམས་ཅད་མཁྱེན་པས་སྡིག་ཅན་གྱི། །སེམས་ཀྱི་བསམ་པ་མཁྱེན་མོད་ཀྱང་། །དགེ་སློང་ལྔ་མཱ་ཡ་ཡོ་སནྣཾ་བྷི་ཁྱུ་ཚཱ་པཉྩ་བྷིཤ་ཏི། །ཨ་ཐ་ཧསྟི་པ་ཀོཏྠཝཾ་ཀྲྀཥཱ་པྲ་བརྒྱ་དག་དང་ནི། །ཐབས་ཅིག་ཡང་དག་བྱོན་པར་གྱུར། །དེ་ནས་གླང་ཆེན་བདག་པོ་ཡིས། །གནད་དུ་སཱ་ད་པཱ་ད་བ། །ཏ་མ་བྱཱ་དྷཱ་བ་དཱ་བྷིནྡྷཀྲོ་དྷཱནྟཀཱུ་ར་ཀུཉྫར། །བསྣུན་བཀྱེ་གླང་པོ་ཆེ། །དྲག་པོ་ཁང་བཟང་རྐང་འཐུང་འཇོམས། །ཁྲོ་བས་ལོང་བས་དེ་མདུན་རྒྱུགས། །།
