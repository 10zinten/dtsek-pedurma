བྲྀཀྵ་ཡ་ས་ཀ་ལ་ཀླ་ཤ་ཤརྨ་ཎེ། །ཨི་ཏུཀྵེ་ཏེ་ན་ས་ཧ་སཱ་སཾ་མ་བྱེ་བདེ་མཛད་པ། །དགྲ་བཅོམ་ཆེ་ལ་ཕྱག་འཚལ་ལོ། །ཞེས་པ་དེ་ཡིས་ཀྱི་བརྗོད་པའི་ཚེ། །འཕྲལ་ལ་རྒྱ་ཏུདྡྷི་དེ་བ་ཏཱ། །སཱརྦ་པྲ་ཏྭཱ་ག་ཏ་པྲ་ཎཾ་བི་ཀེ་པཱ་ཛ་ལ་དྷེསྟ་ཊེ། །མཚོའི་ལྷ་འོངས་ཏེ། །དོན་མཐུན་སྲོག་ནི་སླར་བྱུང་ཉམས། །ཆུ་གཏེར་གྱི་ནི་ངོགས་སུ་བསྒྲལ། །ཏཔདྨ་ཧཱ་བྷ་ཡཱརྨཱཀྟསྟེ་སཱ་དེ་ཤ་མུ་པཱ་ག་ཏཱ། །ཧེ་ཏུཏྟ་མ་ན་མསྐུ་ར་ནི་ར་འཇིགས་ཆེན་དེ་ལས་གྲོལ་གྱུར་པ། །དེ་དག་རང་གི་ཡུལ་དུ་ཕྱིན། །རྒྱུ་ཡི་མཆོག་ལ་ཕྱག་བྱེད་ལ། །དུས་ཏཱསཏྠ་དཱ་བྷ་བཱན། །ཏསྨནྣ་བ་ས་རེ་ཤཱམྦུཀརྨ་ཤེཥ་ཎ་ཀེ་ན་ཙིཏ། །གཱ་ཏྲེ་ཀུན་དུ་ནི་ཆགས་པར་གྱུར། །གནས་སྐབས་དེར་ནི་སྟོན་པ་དགེ། རྒྱུ་ཡི་མཆོག་གི་སྐུ་ལ་ནི། །ལས་ཀྱི་ཧེ་ཏུཏྟ་མ་ཙྭ་བྷཱུཏི་བྲ་ཏཱ་པ་སཱི་ཏི་ཛ་ར། །ཡེ་སཱསྦྲ་ཏྭཾ་བྷ་བ་ལྷག་མ་འགའ་ཡི་རིམས། །དྲག་པོའི་གདུང་བ་གནས་པར་གྱུར། །གང་དག་གཟིགས་པ་སྲིད་པའི་མ་ཧཱ་བྷ་ཡ་བྷི་ཥ་ཛེ་ན་ཀུཊྚནྟི་སཧྥ་ཛ་ག་ཏཱ་མ་ལོ་ཀ་ནེ་ན། །འཇིགས་པ་ཆེན་པོའི་སྨན་གྱིས་ནི། །འགྲོ་བ་ཐམས་ཅད་ལེགས་པར་བསོས་པ་ཉིད་ནི་བསྒྲུབས་མཛད་པ། །ཏེ་པི་པྲ་སཀྟ་ནི་ཛ་ཀསྨྲ་ལ་ཏཱ་བི་ཤེ་ཥཱ་དྷི་བཱ་ས་ཀ་ལ་ནཾ་ན་བ་རི་དེ་དག་གིས་ཀྱང་རང་ལས་འཁྲི་ཤིང་ཁྱད་པར་རབ་ཆགས་པའི། །ལྷག་མ་ལྷག་པར་གནས་ཤིང་ཆགས་པ་རྱ་ཛིནྟི། །གོ་ཤཱིརྱ་ཙནྡ་ནཾ་ཏ་སྱ་ཏཱི་བྲ་སཾ་ཏཱ་བ་ཤཱནྟ་ཡེ། །།རཱ་ཛཱ་ཙནྡྲ་དག་ནི་གཏོང་མི་འགྱུར། །དེ་ཡི་དྲག་པོར་གདུང་བ་ནི། །ཞི་སླད་ཙནྡན་ས་མཆོག་དག། རྒྱལ་པོ་ཟླ་བ་བ་ལོ་ཀཱ་ཀྵསཧྥ་ཏཱ༉སྐུ་མུརླ་ཡོ། །ལཀྵཧྥ་ཏུབྷིནྡྷ་བེ་ན་ས་སཱཟླ་སྣང་ཞེས་པས། །ཐམས་ཅད་དུ་ནི་བཙལ་བར་བརྩམས། །ས་བདག་གིས་ནི་འབུམ་ཕྲག་བཞིས། །དེད་པ་ཏི་ར་ཏ། །གོ་ཤྲིནྟ་ཙནྡ་ན་པ་ལཾ་ན་ད་དོ་བྷུ་བེ་དུརྨ་བྷོ། །བྷཀྵ་དཔོན་དེ་ལ་དོན་དུ་གཉེར། །ཙནྡན་ས་མཆོག་ས་ལ་ནི། །རྙེད་དཀའ་སྲང་གཅིག་བྱིན་མ་གྱུར། །དེད་སཱ་ཡཾ་ས་མ་བྷེ་ཏེ་སྨྲ་ཏི་མཱ་ཏྲ་པ་ཀཱ་རི་ཎ། །ཤཱyསཱརྨ་པ་ཏིཀྲེ་དཔོན་རང་ཉིད་མངོན་ཕྱོགས་ཏེ། །དྲན་པ་ཙམ་གྱིས་ཕན་མཛད་པ། །སྟོན་པའི་སྐུ་ལ་ཙནྡན་གྱིས། །—།
