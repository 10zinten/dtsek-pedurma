བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེབྷི་རུཥཱ་ཀ་ཀུ་ཏུཾ་བི་ནི་བྷི་རུན་ད་མཱུཙྪཱ་བི་ཥ་བz་རཱི་བྷི། །སི་བྷིམྨ་ཧཱ་མོ་ཧ་རྣམས་ནི་མྱ་ངན་རྒྱས་པའི་གཉེན། །སྨྱོ་ཞིང་བརྒྱལ་བྱེད་དུག་གི་འཁྲི་ཤིང་སྟེ། །རྨོངས་པའི་ཤ་ཟ་ཆེན་མོ་པི་ཤཱ་ཙི་ཀཱ་བྷིསཾ་སྤྲོ་བྱ་མཱ་ནཀུ་ཤ་ལཱི་ན་ལོ་ཀ། །སཱསྦྲུལྟ་ཨེ་ཏོ་ནི་བ་སནྟི་འདི་དག་གིས། །ཀུན་ཏུ་རེག་པའི་འཇིག་རྟེན་དགེ་ལྡན་མིན། །བུད་མེད་མིག་ཟུར་གདུང་བྱེད་ཟུག་རྔུ་ནི། །སནྟསནྟོ་ཥ་ཀཱནྟི་ཥཱ་ཏ་པོ་བ་ནེ་ཥཱ། །སནྟཱ་པ་ནཾ་ནི་ན་ཡ་ནཱནྟ་ཤ་ལོ་ཡེ་ཊཱ་ཤི་ཏཾ་རྣོན་པོ་གང་གིས་སེམས་ལ་མ་ཟུག་པ། །ཆོག་ཤེས་ཀྱིས་མཛེས་དཀའ་ཐུབ་ནགས་ཚལ་དུ། །གནས་པའི་ཙེ་ཏ་བི་ནཱ་བ་སནྣཾ། །ཨི་ཏི་པྲ་ཡཏྟཱཛྫ་ན་ཀེ་ན་ཏེསྟེཇི་བེ་ཀ་བཱ་ཀྱེཔྲ་ཏི་བོ་རྒྱ་དམ་པ་འདི་དེ་རྣལ་དུ་གནས། །ཞེས་པ་དབེན་པའི་ཚིག་ནི་དེ་དང་དེས། །ཕ་ཡི་འབད་པས་རབ་ཏུ་མཱ་ན། །ན་བུ་རྒྱ་ཏ་སོ་མ་ད་ནཱ་ནཱི་ཏེ་ན་ལཱ་བ་ཅུ་པཱ་ནེ་ནཱ་ནི་གྷཱཙྪཱ་རྟོགས་པར་བྱས། །འདོད་པ་དང་ལྡན་མཛེས་སྡུག་བཏུང་བ་ཡིས། །རྣམ་པར་འཕྲུལ་པ་དེ་ཉེན་རྟོགས་མ་མཱ་ན། །ཨ་ཐཱ་ས་རེ་དྡྷཇི་ཤྭ་ཀྲྀ་ཏྱཱ་ཀྲྀ་ཏོ་མུ་ནོ་ཕ་ལེ་དྨ་ཧ་ར་ཎ་ཡ་ཡཱ་ཏེ། །ལཱི་ལཱ་གྱུར། །དེ་ནས་ཉིན་གཞན་སྣ་ཚོགས་བྱ་བ་བྱས། །ཐུབ་པ་འབྲས་བུ་ཡམ་ཤིང་ལེན་དུ་སོང་། །མི་དབང་བ་ཏི་ལོ་བྷ་ཡི་ཏུཾ་ཀུ་མཱ་རཾ་ན་རེནྡྲ་ཀ་རྩྭ་པུ་ན་རཱ་ཛ་གྲྭ་མ། །དཱ་སཱི་ག་ཎེ་ནཱ་ནུ་བུ་མོ་འཇོ་སྒེག་ལྡན་པ་ནི། །གཞོན་ནུ་ཆགས་པར་བྱེད་དུ་སླར་ཡང་འོངས། །འབངས་མོའི་ཚོགས་རྣམས་ག་ཏཱ་ལ་ཏཱ་ཚྭ་པུཐ་མི་ཏཱ་ནཾ་ཤིཡ་མཱ་ཐ་ཡན། །མིཾ་གཾ་ན་རྗེས་འབྲང་འཁྲི་ཤིང་ནི། །མེ་ཏོག་འཛུམ་ལྡན་དག་གིས་དཔལ་བསྟེན་ཅིང་། །ལུས་ཅན་ལུས་མེད་པཱི་ག་མི་བི་ཀ་ཤྲཱི་ག་མཱ་པཱ་རྱ་པྲྀཏྟཱ་བི་བྷ་བོ་ན་ཏྭཾ་གཱི། །དི་བྱོ་ཙི་ཏཾ། གསར་མཚུངས་རྭ་གཅིག་ཅན། །འབྱོར་པས་མཐོ་བའི་ལུས་ལྡན་བརྟན་པས་ཕྱིན། །བདག་གི་བསྟི་གནས་ཀལྤ་ལ་ཏཱ་གྲ་ལཾ་བི་ལ་ཏྭཾ་བྷི་རཱ་མཾ་རུ་ཙི་རཱ་ཐ་མ་མེ། །ཏྲཊུ་ཏཱ་མཱ་ཧེ་ཏྱ་དཔག་བསམ་ཤིང་རྩེ་ལས། །འཕྱང་བའི་འཁྲི་ཤིང་གིས་དགའ་ལྷ་ལ་འོས། །ལྟ་བར་ཁྱོད་ཀྱིས་སྨྲས་ཞེས་བྷི་དྷཱ་ཡ་སཱ་ཏཾ་ནི་ནཱ་ཡ་ཏཱི་རཾ་ཏྲི་ད་ཤཱ་པ་གཱ་ཡ། །ས་ཏ་ཏྲ་རཏྟཱི་ཛཱ་ལ་པི་ཙི་ཏྲ་བརྗོད་ནས་དེས། །དེ་ནི་གང་གཱ་དག་གི་ངོགས་སུ་ཁྲིད། །དེས་དེར་རིན་ཅེན་རབ་འབར་འདབ་མ་བཀྲ། །ནཱ—།
