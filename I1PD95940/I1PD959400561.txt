ཤཱ་ལིསྟམཱཙྪ་ཛཱ་ཡ་ཏེ། །དི་བྱ་སཱ་ནཱརྣ་སམྲཏྟིཕ་ལ་ཏཥ་སྱ་ཀརྨ་སྡོང་པོ་བུར་ཤིང་སྡོང་པོར་འགྱུར། །ལྷ་ཡི་བཟའ་བཏུང་ཕུན་སུམ་ཚོགས། །དེ་དག་ལས་ནི་གང་གི་ཎ༑། ཨི་ཏི་པྲཋཔྲ་ཎ་ཡི་ནཱ་བྷིཀྵ་ཎཱ་བྷ་ག་བཱཎྜི་ན། །ཨུ་བཱ་ཙ་འབྲས། །དགེ་སློང་གུས་པས་དྲིས་པ་འདི། །བཅོམ་ལྡན་རྒྱལ་བས་བཀའ་སྩལ་པ། །རྒྱལ་པོའི་ལོངས་ཐུ་ཡ་ཏྭཾ་རཱ་ཛྷཀཱ་ར་ཎཾ་བྷོ་ག་སམྦ་ཛཱ། །ཀོ་ཤ་མེསྨིལྔ་ན་ས་ད་ཁཌཱ་སྤྱོད་ཕུན་ཚོགས་ཀྱི༄༅། །རྒྱུ་ནི་རབ་ཏུ་མཉེན་པར་མཛོད། །ཀོ་ས་ལ་ཡི་ལྗོངས་འདི་རུ། །བུར་ཤིང་ཁྱུ་གུ་ཌ་ཀཆུ་ཀཿ། །ད་དོ་པྲ་ཏྱཾ་ཀ་བུདྡྷཱ་ཡ་པཱུཏྠ་མིཀཱུ་ར་སོ་དནཾ། །བྷུཀྟི་ནི་གླུ་ཞིང་པ་དུམ་བུ་ཞེས། །བྱ་བས་རང་སངས་རྒྱས་ལ་སྔོན། །བུར་ཆུ་དང་ནི་ཟས་དག་བྱིན། །དེ་ཡི་བསོད་ར་སེནེ་ན་ཏེ་ན་བཱ་ཏ་ག་དཱཎྜི་། པྲ་ཏེ་ཀ་བུདྷསྟཧྥུ་ཙྩེནམས་རྣམས་དང་ནི། །བུར་ཆུ་གཟན་དག་གསོལ་བ་དེས། །རླུང་ནད་ཀྱིས་ཉེན་རང་སངས་རྒྱས། །པྲ་སནྣ་སཱསྦྲ་ཏཱཾ་ཡ་ཡོ། ་། རཱ་ཛཱ་པྲ་སེ་ན་ཛིཏྟཾ་ཡཾ་པུ་ཎྱ་བཱན་གུ་ཌ་ཀརྟ་ཀ། །རབ་དང་དྲི་མེད་ཉིད་དུ་གྱུར། །བུར་ཤིང་ཞིང་པ་བསོད་ནམས་ཅན། །དེ་ནི་རྒྱལ་པོ་གསལ་རྒྱལ་འདི། །ཏེ་ན་པུ་ཎྱ་པྲ་བྷཱ་བེ་ན་བྷོ་ག་བྷཱ་གཱི་བི་རཱ་ཛ་ཏེ། །ཨུ་པ་ཀཱ་ར་ཀྲྀ་ཏ་ཛྷ་ནཾ་བསོད་ནམས་དེ་ཡི་མཐུ་ཡིས་ནི། །ལོངས་སྤྱོད་སྐལ་ལྡན་རྣམ་པར་མཛེས། །བྱས་ཤེས་རྣམས་ལ་ཕན་པ་དང་། །ནི་ཀཱ་རཀཱ་ར་ཙེ་ཏ་སཾ། །སུ་ཀྲྀ་ཏྭཾ་ཤཉྩ་སཱ་དྷུ་ནཱ་མཝོ་པཱུ་ཡཱ་ཏྱ་ནལྤ་གདུག་སེམས་ཅན་ལ་གནོད་པ་དང་། །དམ་པ་རྣམས་ལ་ལེགས་བྱས་ནི། །ཆུང་ཡང་ཆུང་ངུ་ཉིད་མིན་ཏྭཾ༑། སརྦ་ཛྷེ་ནེ་ཏི་ཀ་ཐི་ཏེ་པཱུརྦ་པུ་ཙྩ་མ་ཧཱི་པ་ཏེ། །བ་བྷུ་འགྱུར། །ས་བདག་སྔོན་གྱི་བསོད་ནམས་ནི། །ཀུན་མཁྱེན་གྱིས་ནི་གསུངས་པའི་ཚེ། །ལེགས་བྱས་བ་སུ་ཀྲྀ་ཏོངྐརྟ་བྷིཀྵ་རཱཉྩརྱ་ནི་ལ། །ཨ་ཐ་བྷཉྩ༔བྷ་ག་བ་ཏཁྭ་ཏྭཱ་མཆོག་ཏུ་གྱུར་པ་ལས། །དགེ་སློང་ངོ་མཚར་གཡོ་མེད་གྱུར། །དེ་ནས་བཅོམ་ལྡན་ལ་གུས་པས། །རྒྱལ་རཱ་ཛཱ་དྷི་བཱ་ས་ནཾ། །ཨུ་པཱ་ནི་རྩེ་སྭཱ་ཡནྟཱཎཱ་སུ་རཱགྷཾ་བྷོ་ག་སམྦ་དཾ། །པོས་ལྷག་པར་གནས་བྱས་ནས། །ལོངས་སྤྱོད་ཕུན་ཚོགས་ལྷ་ལ་འོས། །དེ་དེ་རང་གིས་ཉེ་བར་བཀོད། །—།
