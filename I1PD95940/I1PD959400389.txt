བ་ཀྱི་ཏོ། །བཱ་རི་ཏེ་ཥཱ་ཐ་བསྡེ་ཥཱ་དི་བྱ་བ་སྦྲི་དྲྀཏ་མ་ཧཱི། །ཨ་པྲ་འགེབས་པ་ཟློགི། །དེ་ནས་གོས་ནི་བཟློག་པའི་ཚེ། །ལྷ་ཡི་གོས་ཀྱིས་ས་གཞི་གཡོགས། །བསོད་ནམས་ཡནྡྷོ་པ་ཀ་ར་ཎཱསམྲ་དཔུ་ཎྱ་ཤཱ་ལི་ནཾ། །ནི་བཱ་རྱ་དི་བྱ་བསྠཱ་ཎི་ལྡན་པའི་ཉེར་མཁོ་རྣམས། །འབད་པ་མེད་པར་ཕུན་སུམ་ཚོགས། །ལྷ་ཡི་གོས་རྣམས་བཟློག་ནས་ནི། །བྷུ་མོ་ཏེ་ནཱམྦི་ཏེ་པ་དེ། །བི་ཙ་ཙཱ་ལཱ་ཙ་ལཱ་པྲ་བྷི་ས་ཤེ་ལ་བ་ན་སཱ་ག་རཱ། །དེ་ཡིས་ས་ལ་རྐང་བཞག་ཚེ། །རི་ནགས་རྒྱ་མཚོ་དང་བཅས་པ། །གཡོ་མེད་ས་གཞི་རྣམ་པར་གཡོས། །ཏ་ཏས་བྷུ་མི་པ་ཏི་ནཱ་ས་ཧ་པྲ་སྱ་ཛི་ནཱ་ཤྲ་མཾ། །བྷ་ག་བནྟཾ་བི་ལོ་ཀླུ་དེ་ནས་ས་བདག་དང་འགྲོགས་ཏེ། །རྒྱལ་བའི་གནས་སུ་ཕྱིན་པ་དེས། །བཅོམ་ལྡན་འདས་ལ་རབ་སྱ་བི་ད་དྷེ་པཱ་ད་བནྡ་ནཾ། །ཨུ་པ་པིཋ་སྱ་ཏ་སྣ་གྲེ་ཧཱུྃ་སྭཱ་ལོ་ཀ་བལྟས་ཤིང་། །དེ་ཡི་ཞབས་ལ་ཕྱག་འཚལ་བྱས། །དགའ་དང་ཞི་ཞིང་དབེན་ལྡན་པ། །མདུན་ན་གནས་ནཱ་མྲ་ཏི། །ཙཀྲེ་ཤ་མ་བི་བེ་ཀ་སྱ་བྷ་ག་བཱན་བྷི་ཥ་ཙ་ནཾ། །པ་དེ་ལ་ནི། །བཅོམ་ལྡན་འདས་ཀྱིས་གཟིགས་པ་ཡིསི། །བདུད་རྩིས་མངོན་པར་དབང་བསྐུར་མཛད། །ཨཱ་ཤ་ཡཱ་ནུ་ཤ་ཡཾ་དྷཱ་ཏུ་པྲ་ཀྲྀ་ཏིཉྩ་བི་ཙཱརྱ་ས། ་། ས་སྐྱ་སནྡན་ནཱ་ཡཱ་ཁམས་དང་བསམ་པ་བག་ལ་ཉལ། །རང་བཞིན་དག་ཀྱང་སྤྱད་ནས་དེ༄༅། །བདེན་པ་ཡང་དག་རྟོགས་སྱ་བི་ད་དྷེ་དྷརྨ་དེ་ཤ་ནཾ། །སཏྟཱ་ཡ་དྲྀཥི་ཤི་ལོཉམྱ་ཏ་ཡཱ་བི་ཤ་ཏི་ཤྲཱིངྒ་བཱན། །བྱའི་སླད། །དེ་ལ་ཆོས་ནི་བསྟན་པ་མཛད། །དེ་ཡི་འཇིག་ཚོགས་ལྟ་བའི་རི། །རྩེ་མོ་ཉི་ཤུ་ལྡན་པ་ནི། །ཛྙཱ་ན་བཛེ་ཎ་ནིབྷིནྡཤོ་ཏཔཱབྷི་པ་ད་སྤྲོ་ཤ། །པྲ་པྲ་ཛཱ་ཡཾ་ཏ་ནོ་ཏ་སྱ་ཛཱ་ཏཱ་ཡེ་ཤེས་རྡོ་རྗེ་དེ་ཡིས་བཅོམ། །རྒྱུན་དུ་ཞུགས་པའི་གནས་ལ་རེག། དེ་ནས་དེ་ཡི་ལུས་ལ་ནི། །རབ་ཡཾ་ས་ཧ་སཱ་སྭ་ཡཾ། །བྷ་ག་བནྟཾ་པྲ་ཎ་མཱ་ཐ་བིསྨི་ཏཔྲ་ཡ་ཡོ་ཙྰ་པ། །བྱུང་སྐྱེས་ཚེ་མི་ཡི་བདག། ཡ་མཚན་གྱུར་པས་བཅོམ་ལྡན་ལ། །ཕྱག་འཚལ་འཕྲལ་ལ་རང་ཉིད་སོང་། །ཏཱི་བྲ་བྲ་ཏེ་པི་ཐོ་ཎ་སྱཱ་ཀ་དཱ་ཙིཏྠ་མཱ་ཛཱ་ཡ་ཏ། །བཱ་ས་ནཱ་ཤེ་གྲོ་བཞིན་བརྟུལ་ཞུགས་དྲག་ལྡན་ཡང་། །བག་ཆགས་ལྷག་པའི་འདུ་བྱེད་ལས། །ནམ་ཞིག་གཉེན་དང་—།
