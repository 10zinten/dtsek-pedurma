_ནྱ་ཏ། ཊ་། ས་ག་ཏཱ་ཙ་ཀི་ཏ༔པྲ་ཧ་ཛ་ན་ནཱིནྡྷཱ་བ་བརྨ་ཎ། ཚ། སུ་ཧྲི་པར་གྱུར། །གྲོགས་ལ་མཛའ་བས་ཆགས་གྱུར་ཅིང་། །གནོད་པ་འབྱུང་བར་དོགས་པ་ཡིས། །འཇིགས་དསྣེ་ཧ་སཾ་སཏཿཤཾ་ཀ་མཱ་ནཔ་ར་བྷ་ཐཾ། །།སུ་ཏསྟེ་ཀྲྀ་ཏ་སཾ་ཀེ་ཏསྤཊཾ་སུ་བྷ་ཊ་པ་དེ་ནི་སོང་གྱུར་ནས། །སྲིད་པའི་གོ་ཆའི་མ་ལ་སྨྲས། །ཁྱོད་ཀྱི་བུ་དང་དཔའ་བོ་ཡི། །བུད་མེད་གསལ་ཡོ་ཥི་ཏཱ། །ར་ཛ་རྩྭ་མ་བི་ནཱི་ཏོ་བྱ་སྤྱི་ར་ཙ་རཱི་ག་མི་ཥྛ་ཏི། །ཙ་རནྟསཾ་བའི་བརྡ་དག་བྱས། །མཚན་མོ་རྣམ་པར་མ་དུལ་བ། །དམན་མིན་རྒྱུ་ཞིང་འགྲོ་བར་འགྱུར། །མི་ཡི་རི་སཱ་ར་བྱ་ས་ན་ག་ཧེ་ན་སྨིནྣ་ར་མྲྀ་ག་ཛཾ་ཤཾ་བྷོ་ག་བྱ་གྲྭ་ནི་ཁི་ལ་སུ་ཁ་དྭགས་འཁོར་བ་སྡུག་བསྔལ་ནགས་ཚལ་འདིར་ནི་རྒྱུ་བྱེད་ཅིང་། །མཆོག་ཏུ་ལོངས་སྤྱོད་ཀྱིས་འཁྲུགས་མ་ཤ་ཥྱ་པྲ་ཎ་ཡི་ཎ། །་། བི་ནཱ་ཤ་བྱ་པཱ་རི་སྨ་ར་ཤ་བ་ར་པཱ་ཤ་ནི་བ་ལུས་བདེ་བའི་རྩ་ལ་ཆགས་གྱུར་རྣམས། །འཇིག་པའི་བྱ་བ་དང་ལྡན་རྔོན་པ་ཞགས་པ་དག་ཏུ་ལྟུང་གྱུར་ཏེ། །ཏི་ཏཱ་ལ་བྷནྟེ་པརླནྟ་བི་ཥ་ནི་ཤི་ཏ་ཤསྡིདྡི་ཤ་ས་ནཾ། །་་། བཱ་ས་རེ་རཀྱི་མཇུག་ཏུ་དུག་ལྡན་རྣོ་བའི་ཉམཚོན་རྣམས་དག་གིས་བརླག་པ་ཐོབ་པར་འགྱུར། །གཡོ་བའི་བསམ་པ་ཏསྣ་བཏ་མ་ཡཱ་ཡཾ་ཙ་པ་ལཱ་ཤ་ཡ། ་། རཀྵ་ཎཱི་ཡསྣ་ཡཱ་རཱ་ཏཱ་བི་ཏྱཱཀྟ་དང་ལྡན་ཞིང་། །རེ་ཞིག་ཉིན་པར་བདག་གིས་བསྲུང་། །མཚན་མོ་ཁྱོད་ཀྱིས་བསྲུང་བར་བྱ། །ཞེས་པ་ཡ་ཡོ་སུ་ཧྲིཏ། ་། ཨ་ཐཱནྟཀྲྀ་ཧ་སུལྟ་སྱ་པུ་ཏྲ་སྱ་རཱ་ར་ཤཱ་ཡི་ནི། །མཱ་ཏཱ་བརྗོད་ནས་གྲོགས་པོ་སོང་། །དེ་ནས་ཁྱིམ་ནང་དུ་ཉལ་བུ། །སྒོ་རུ་ཉལ་བའི་མ་ཡིས་ནི། །མཚན་མོ་ནི་ཤི་པྲ་བུདྡྷེ་བ་སཾ་རཀྵ་ཎ་པ་རཱ་བྷ་བཏ། །བཱ་རི་ཤོ་ཙཱ་པ་དེ་ཤེ་ན་ཡཱ་ཙ་མཱ་ན་རབ་ཏུ་སད་བཞིན་དུ། །ཀུན་ཏུ་བསྲུང་བ་ལ་ཆགས་གྱུར། །ཆུ་ཡི་དག་བྱེད་ཟོལ་གྱིས་ནི། །ཕྱིར་འབྱུང་སྱ་ནིན་མ། །ན་ནིནཾ་ཏུཾ་ད་དོ་ཏ་སྱ་ཛ་ན་ནཱིསྣེ་ཧ་མོ་ཧི་ཏཱ། །།སོ་ཙིནྟ་ཞུ་བར་བྱེད་པ་ལ། །མཛའ་བས་རྨོངས་པའི་མ་ཡིས་ནི། །་ཕྱིར་ལ་འཐན་པ་བྱིན་ེམ་གྱུར། །དེས་ཡ་ད་ཧོ་ར་མྱ་རཱ་མཱ་ར་ས་མཱ་ག་མ། ༽། ཀེ་ནེ་ཡཾ་བིརྒྱུ་ཛ་ན་ནཱི་ཛ་བསམས་ཀྱེ་མ་དགའ་བ་ཡི། །བུད་མེད་རིན་ཆེན་དང་འགྲོགས་ལ། །གེགས་ནི་སྐྱེད་པར་བྱེད་པ་ཡི། །—།
