བསྟན་འགྱུར། འཁྲི་ཤིང་། ཝིཏི་བི་དྭེ་ཥ་དོ་ཥ་བི་ཥ་དྷུ་ཥི་ཏ་མཱ་ན་སཱ་ཚྭ་ཡ༔སཱ་དྷུ་བཱ་དྷ་ན་བི་དྷོ་ནི་རྣམ་སྡང་སྐྱོན་གྱི་དུག་གིས་ཡིད་ནི་སུན་ཕྱུང་གྱུར་རྣམས་ཀྱིས། །དམ་པ་དག་ལ་གནོད་པ་བསྒྲུབ་ཤི་ཏཔྲ་ཡཏྠཱ། །སྩཙེཏྟ་དུ་ཛཱ་ལ་གུ་ཎ་ནུ་ཀྲྀ་ཏོ་ས་ཨེ་བ་ཏཥོ་བྷ་བེཎྚི་ལ་འབད་པ་རབ་རྣོ་གང་། །དེ་ཉིད་གལ་ཏེ་དེ་ཡི་ཡོན་ཏན་རབ་འབར་རྗེས་བྱེད་པར། །གྱུར་ནས་སྲིད་ཐུ་པ་རཱ་བྷ་བ་བྷནྡྷ་བེསྨིན། །ཛི་ནེ་རཱ་ཛ་གཾ་ཧོ་པཱནྟ་བེ་ཎུ་ཀཱ་ན་ན་ཙཱ་འདིར་སུ་ཞིག་རྒྱ་ཆེར་གནོད་དང་ལྡན་པར་གྱུར། །རྒྱལ་བ་རྒྱལ་པོའི་ཁབ་དང་ཉེ། །འོད་མའི་ཚལ་ན་རྒྱུ་རི་ཎི། །བི་དྭེ་ཥཱདེ་བ་དཏྟ་ན་པྲ་རི་ཏཱལྟ་པ་སཱཔུ་རཱ། །བྷུ་བྷཾ་ག་བ་ན། །ལྷས་སྦྱིན་གྱིས་ནི་སྡང་བས་སྔོན། །རབ་ཏུ་བསྐུལ་བའི་དཀའ་ཐུབ་པ། །དེ་ཡིས་བྱིན་པའི་ཧི་མ་བ་ད་ནཱསྟདྡཏྟ་བི་བི་དྷཱ་ཡུ་དྷཱ། །ཨ་བྷྱཱ་དྲ་བརླུ་སཾ་རབྷཱ་ཀྲོ་དྷ་བི་མཚོན་ཆ་ཅན། །སྨིན་འཁྱོག་འཇིགས་སུ་རུང་བའི་བཞིན། །ཁྲོ་བས་དུལ་བ་ཉམས་གྱུར་པའི། །རྩོམ་པ་ཥསྟ་སཾ་ཡ་མཱ། །པཉྩ་བྷིལྟ་པ་ས་ཤ་ཏིཏེ་ཊཱ་ཀྱིསྟསྟེདྲྀཥཱ་ཡ། །ཆེན་པོས་མངོན་པར་གཅེས། །བརྒྱ་ཕྲག་ལྔ་ཡི་དཀའ་ཐུབ་ཅན། །དེ་དག་གིས་འཕངས་མཚོན་གྱི་ཆར། །ཡ་ཡུཌྷ་ག་བཏཀཱ་ཡ་ཀཱ་ཡ་ཀ་མ་ལོཏྲ་ཛཱ་ལཏྭཾ། །ནི་ཤི་ཏཱ་ནྱ་བི་ཤསྨཱ་ཎི་ཤ་བཅོམ་ལྡན་འདས་ཀྱི་སྐུ་ལ་ནི། །པདྨ་ཨུཏྤལ་ཚོགས་ཉིད་གྱུར། །དམ་པ་རྣམས་ལ་རྣོན་པོ་ཡི། །མཚོན་ཡནྟི་སཏུ་མཱརྡ་ཐཾ། །བི་དྭེ་ཥ་བི་ཥ་དིབྦྷ་ནི་ན་ཏུ་ཙིཏྟཱ་ནི་པཱ་པི་ནཾ། །པྲ་ཡང་འཇམ་པ་སྟེན་བྱེད་དེ། །སྡང་བའི་དུག་གིས་བྱུགས་གྱུར་པ། །སྡིག་པ་ཅན་གྱི་སེམས་ནི་མིན། །སྟོན་དུ་རཱ་སཱིདྨ་ཎི་མ་ཡཾ་ཀཱུ་ཊཱ་གཱ་ར་ན་བྷཔྲ་བྷཾ། །ཤ་རཱི་ར་ཙྪཱ་ད་ནཾ་ཤཱསྟུན་པའི་སྐུ་ནི་སྒྲིབ་བྱེད་པ། །ནོར་བུའི་རང་བཞིན་ཁང་པ་བརྩེགས། །ནམ་མཁའི་འདི་ནི་རབ་འཁྲུངས་ཏེ། །ཏུ་དྲྀཥཱ་ནི་བཱ་ར་ཎཾ། །ཏ་ཏསྟེ་ཏ་པ་སཱ་ཤཱནྟཱ་ལཛྫཱ་བ་ན་མི་ཏཱ་ན་ནཱ། །ལྟ་བ་དག་ནི་བཟློག་མ་གྱུར། །དེ་ནས་དཀའ་ཐུབ་ཅན་དེ་དག། ངལ་ཞིང་ངོ་ཚ་བཞིན་དུད་པས། །ནི་པེ་ཏུཔཱ་ད་ཡོཤཱyཀཱ་མཱ་སིནྡྷོཔྲ་སཱ་དི་ན། །བྷ་ག་བཱན་པི་པུ་བཟོད་པའི་རྒྱ་མཚོ་རབ་དང་བསི། །སྟོན་པ་ཡི་ནི་ཞབས་ལ་གཏུགས། །བཅོམ་ལྡན་གྱིས་ཀྱང་བུ་ལ་—།
