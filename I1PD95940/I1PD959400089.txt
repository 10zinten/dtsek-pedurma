ཡཱ་ཀུ་ལཾ། །ད་དན་ཏ་ཏྲ་ས་མོ་དཾ་སཱ་ཀཱ་ར་མི་བ་ནནྡ་ནཾ། །པུ་རན་རས་་ཚོགས་རྒྱས་པ། །།དགའ་ཚལ་རྣམ་པ་དང་བཅས་བཞིན། །དེར་ནི་དྲི་བཟང་ལྡན་པ་མཐོང་། །དེ་ནས་ས་ཏོ་ཙྪཱ་ཏྭཱ་པཱསྟ་བྷཱུ་མི་པུ་རནྡ་རཾ། །པྲ་ཀྱུརླ་ཡོ་ཕྲ་མུ་དི་ཏས་ཧ་་་་ཡི་དབང་པོ་ནི། །འོངས་པར་ཤེས་ནས་དབང་པོ་ནི། །ལྷ་ཡི་ཚོགས་དང་བཅས་པ་དག། རབ་དགའ་སཎྚིརྨ་རུངྒ་ཎི། ::། པཱུ་ཛི་ཏསུ་ར་རཱ་ཛེ་ན་རཏྟཱ་རཱ་ཛ་བི་རཱ་ཛི་ཏྭཾ། །ལྡན་པས་བསུ་བ་བྱས། །རིན་ཆེན་འཕྲེང་བས་རྣམ་མཛེས་པའི། །མདུན་སའི་ས་གཞི་རྒྱལ་པོ་ཡི། །རཱ་ཛ་རཱ་ཛས་བྷཱ་བྷུ་མི་བྷེ་ཛེ་བི་ར་ཛ་སཱམྦ་ར། །ཏྲི་ད་ཤེ་ཥཱ་པ་བིཥྛ་ཏུ་་རྒྱལ་པོ་ལྷ་ཡི་རྒྱལ་པོ་ཡིས། །མཆོད་པ་རྡུལ་བྲལ་མཆོག་གིས་བསྟེན། །རིན་ཆེན་སྟན་གྱི་འཕྲེང་ཅིབ་ལ། །རཏྭཱ་པཎཾ་ཀ་པཾ་སྟི་། ་་བྱ་། ཨུ་པཱ་བི་ཤནྣཾ་པཤཱི་མཱ་ནཱ་ས་ནཱདྷེ་ཤ་ཏ་་་སྐབས་གསུམ་པ་རྣམས་ཉེར་བཞུགས་ཤིང་། །མཆོད་སྦྱིན་བརྒྱ་པའི་སྟན་ཕྱེད་ལ། །མི་ཡི་བདག་པོ་དཔལ་ཀྲ་ཏོ། །ཨེ་ཀཱ་ས་ན་ཛུ་ཥསྟ་ཏྲ་སུ་རེནྡྲ་མ་ནུ་ཛེནྡྲ་ཡོ། །རཱུ་པ་་ལྡན་བཞུགས། །དེར་ནི་སྟན་གཅིག་ཁྱིལ་འདུག་པའི། །ལྷ་དབང་མི་ཡི་བདག་པོ་དག། ཡོན་ཏན་གུ་ཎ་ག་ཎོ་དཱ་རཾ་ནིརྨི་ཤེ་ཥ་མ་དྲྀ་ཤྱ་ཏ། །ཏ་ཏསརྦ་སུ་རོཏྠཾཀྵ་སྤཏྠ་་་ཚོགས་ནི་རྒྱ་ཆེ་བའི། །གཟུགས་ནི་ཁྱད་པར་མེད་པར་མཐོང་། །དེ་ནས་ལྷ་ཀུན་གྱིས་སྤྲོས་པའི། །རབ་ལོ་ཙ་ན་ཥཟླ་དི། །པཱི་ཡ་མཱ་ན་མུ་ཁཱམྦྷོ་ཛཾ་བྷ་ཛ་ཧཱ་ར་ཧ་རི་ནཾ་ལཾ། །གསལ་མིག་ནི་རྐང་དྲུག་པས། །བཞིན་པད་འཐུང་ལྡན་མི་བདག་ལ། །འཕྲོག་བྱེད་ཀྱིས་ནི་ཡང་དག་སྨྲས། །ཨ་ཧོ་པ་དོ་ཡབྷཱ་གྷསྟེ་ཛ་སཱ་ཏེ་ཛ་ཧཱུཾ་ནི་དྷེ། །བྷ་བ་ཏཱ་བྷཱ་ཥི་ཏཱ་་་ཀྱེ་མ་གཟི་བརྗིད་གཏེར་གྱི་གཟི། །གང་ཞིག་བསྔགས་འོས་མངོན་པར་འཆར། །ཁྱེད་ཀྱིས་ས་ནི་བརྒྱན་བྷུ་མིརྡོ་དེ་བ་ན་བྷཱ་སཱ་ཏཱི། །ཨ་བྷུསྣ་ཏ་པྲ་བྷཱ་བོ་ཡཾ་ལ་སཏ་་་གྱུར་ཅིང་། །འོད་ལྡན་ལྷ་ཡི་@ནམ་མཁའ་ཡང་། །མཐུ་ཡིས་རབ་མཐོ་གྲགས་སྙན་དག། གསལ་བའི་སི་ཏ་ཡཱ་ཤཱི་ཤུ་ཀཾ། །བྷཱ་ཛ་ཏེ་ཏེ་ཏྲི་བྷུ་བ་ནེ་སྭཱ་མཱ་ཛུ་བི་ཛ་ཡ་དྷཱ་ཛ། །ཏཱཏ་གོས་ཅན་ས་གསུམ་དུ། །ཡང་དག་རྒྱལ་པོ་རྣམ་རྒྱལ་བ། །ཁྱོད་ཀྱི་རྒྱལ་མཚན་འདི་མཛེས་སོ། །ཁྱོད་—།
