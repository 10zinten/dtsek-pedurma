སཱ༑། སཱ་ག་ཏཱ་ན་ག་རཾ་པ་ཏུསྣ་མཱནྟི་རི་བིརྗེ་བཱི། །བི་བོ་དྷ་ནི་ཤི་གྱུར༑། །དེ་ཡིས་མཁའ་ལ་ཟླ་བ་ཡི། །གཟུགས་བཞིན་བདག་པོའི་གྲོང་ཁྱེར་ཕྱིན། །མཚན་མོ་ཉལ་བ་སུསྟ་སྱ་ཙཀྲེ་བོ་དྷི་པྲ་ཀཱ་ཤ་ནཾ། །ཡཱ་ཏཱ་ཡ་སྭ་པ་དཾ་ཏ་ཧཱུྃ་པྲ་བྷཱ་ཏེ་བ་བསླངས་ནས་ནི། །བྱང་ཆུབ་རབ་ཏུ་གསལ་བར་བྱས། །དེ་ནི་ཐོ་རངས་རང་གནས་སུ། །སོང་བའི་ཚེ་སུ་དྷཱ་དྷི་པ། །པྲ་བྲ་ཛཱ་བྷི་མུ་ཁཔུ་ཏྲ་མ་བྷི་ཥི་ཙྩ་ཤི་ཁཎྜི་ནཾ། །ན་ནོར་འཛིན་བདག། རབ་བྱུང་མངོན་པར་ཕྱོགས་པས་སུ། །གཙུག་ཕུད་ཅན་ནི་དབང་བསྐུར་ཏེ། །པྲ་ཛཱ་ཏཱ་ཎཱ་ཡ་ནི་ཀཱི་ནྱ་ཏཾ་མ་ཧཱ་མཱ་ཏྱ་ཡོསྟ་བཿ། །ཚིགས་པ་ཏེཟིལླི་སཱ་ར་སྱ་སུ་དེ་ནས་བློན་པོ་དག་ལས་དེ། །སྐྱེ་རྒུ་བསྐྱབ་པའི་ཕྱིར་གཏད་ནས། །དེ་ནི་མི་བདག་གཟུགས་ཅན་སྙིང་། །ཧྲ་དས་ཡ་ཡོ་པུ་རཾ། །བིཝི་སཱ་རསྟ་མཱ་ཡཱ་ཏཾ་བི་ག་ཏཙྪཏྭཱ་ཙཱ་མ་གྲོགས་པོའི་གྲོང་ཁྱེར་དག་ཏུ་སོང་། །གདུགས་དང་རྔ་ཡབ་དང་བྲལ་བ། །དེ་འོངས་གཟུགས་ཅན་སྙིང་པོ་རཾ༈། ནི་པོ་པ་ཙཱ་རཿཔྲ་ཎ་ཏ༔པཱི་ཏི་པཱུ་ཏི་ར་པཱུ་ཛ་ཡ་ཏ། །བི་ཤཱནྟ་ཡིས། །རབ་བཏུད་མི་བདག་ཉེར་སྤྱད་ནི། །མཛེས་ཤིང་གཙང་མ་རྣམས་ཀྱིས་མཆོད། །ངལ་བསོས་མཱ་ས་ནཱ་སི་ནཾ་ས་ཏ་མཱུ་ཙེ་བ་བི་སྨྲ་ཡ། །དྲྀཋསཾ་དན་ནེ་ནཱ་སྱ་ཤྲཱི་བི་ཁྲི་ལ་འཁོད་དེ་ལ། །དེ་དང་ཕྲད་པས་རབ་དགའ་ཞིང་། །དཔལ་དང་བྲལ་བས་སྡུག་བསྔལ་བའི། །།ཡ་ཡོ་གེ་ན་དུ༔ཁི་ཏ། །རཱ་ཛསྣ་ནནྟ་སྭཱ་མནྟ་མོ་ལི་བི་ཤཱནྟ་ཤཱ་ས་ན། །མཚན་ལྡན་པ་དེ་ཡིས་སྨྲས། །མཐའ་ཡས་ཀུན་གྱི་མགོ་བོ་ལ། །རྒྱལ་པོའི་བཀའ་ནི་ངལ་གསོ་བ། །པཱ་ཀ་ཤཱ་ས་ན་ཏུ་མྱ་སིཾ་ཀ་ཐ་མེ་བ་མུ་པཱ་ག་ཏ། །ཨ་བྷིནྡ་བཀྑ་པྲ་ཀྲྀ་བརྒྱ་བྱིན་དག་དང་མཚུངས་པ་ཁྱོད། །འདི་བཞིན་ཉེ་བར་ཇི་ལྟར་གྱུར། །རང་བཞིན་བཞིན་རས་མ་བྲལ་ཏེརྒུཔ་མནྡཱ་སྱ་རཱི་མ་ཏ། །པ་རེ་ཎ་རཱ་ཛུ་ཧ་ར་ཎཾ་བཱི་ར་སཾ་བྷཱ་བྱ་ཏེ་ཞིང་། །བློ་ལྡན་དོན་@ནི་སྦས་པ་ཅན། །དཔའ་བོ་ཁྱོད་ཀྱི་རྒྱལ་སྲིད་ནི། །གཞན་གྱིས་གྲོགས་པོ་སྲིད་ན་ཏེ། །ཨི་ཏི་པྲཉྫས་སུ་ཧྲི་དཱ་སརྨི་ཏསྟ་མ་བྷཱ་ཥ་ཏ། །རཱ་ཛན་བྲྀདྷཱ་མ་ཡིན། །ཞེས་པ་གྲོགས་ཀྱིས་དྲིས་ལ་དེས། །འཛུམ་བཅས་དེ་ལ་རབ་སྨྲས་པ། །རྒྱལ་པོ་འབྱོར་པ་—།
