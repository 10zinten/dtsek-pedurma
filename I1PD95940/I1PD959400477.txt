མ་བྷེ་ཏ་ན་རེ་ཤཱ་རོཉཐ་དྲྀཥཱ་བ་དྷུ་བྷིཔ་རི་བཱ་རི་ཏཾ་ཏཾ། །ཨིནྡྷ་པྲ་མི་དབང་ཕྱོགས་དེར་མངོན་ཕྱོགས་པས། །བུད་མེད་རྣམས་ཀྱིས་ཡོངས་བསྐོར་དེ་མཐོང་ནས། །ཕྲག་དོག་ཀོ་པཱ་ན་ལ་དུནི་རཱིཙིཙྪེ་ད་ཏ་སྭཱ་ཤུ་ས་པཱ་ཎི་པཱ་དཾ། །ཚིནྡྷངྒ་བརྒོ་བི་ས་ནིཅི་ཁྲོ་བའི་མེ་ནི་བལྟ་དཀའ་དེས། །དེ་ཡི་ལག་པ་རྐང་པ་མྱུར་དུ་བཅད། །བརྟན་པར་རྣམ་འགྱུར་མེད་དེས་ཀཱ་ར་ཀོ་པ་བྷུ་པཱ་ཡ་ནཱ་མ་དྷཱི་རཿ། །ནྱ་བཱ་ར་ཡཏ་ཀྲུདྷ་ཏ་རཉྩ་ཏརྨི་ཡན་ལག་ཚོགས། །བཅད་ཀྱང་ས་ཡི་བདག་ལ་ངེས་མ་ཁྲོས། །དྲི་ཟ་གནོད་སྦྱིན་ལྟོ་འཕྱེ་ལྷ་ཚོགས་ཀྱིས། །གནྟབྷ་ཡཀོ་ར་ག་དེ་པ་སཾ་གྷ། །ཏ་ཏཔྲ་ཡཱ་ཏེ་ནཱཾ་པ་ཏོ་པུ་རཾ་སྭ་ས་མེ་དེ་ལ་ཁྲོ་བ་དྲག་པོ་དག་ལས་བཟློག། དེ་ནས་མི་བདག་རང་གི་གྲོང་སོང་ཚེ། །ནགས་ནས་འོངས་པའི་ཏྱ་སཉྫ་མུ་ན་ཡོ་བ་ནེ་བྷྱ། །ཏཾ་ཏ་ཏྲ་ཀྲིཏྟཱ་བ་ཡ་ཐཾ་བི་ལོཀྱ་གྷཱནྟཱ་ཨ་བི་ཀྲོ་དྷ་དྷུ་ཐུབ་པ་ཐམས་ཅད་ཀྱིས། །དེར་ནི་ཡན་ལག་བཅད་པ་དེ་མཐོང་ནས། །བཟོད་ལྡན་དག་ཀྱང་ཁྲོ་བས་ཏཱ་བ་བྷཱུ་བུ། །ཤཱ་ས་པྲ་དཱ་ནཱ་བྷི་མུ་ཁཱན་ནི་བཱརྱ་བྷཱནྟབྱ་མི་སྐྱེ་བ་ས་ཏཱ་འདར་བར་འགྱུར། །དམོད་པ་འདེབས་ལ་མངོན་ཕྱོགས་དེ་དག་ལ། །བཟློག་ནས་བཟོད་པར་བྱ་ཞེས་དེ་ནུ་བཱ་ཙ། །ཀཱ་མཱ་ས་མཱ་ལིངྒི་ཏ་མཱ་ན་སཱ་ནཾ་ཀོ་པ་ཀྲི་ཡཱ་བྷིཡ་ཏེ་ན་སཾ་ག། །ཡིས་སྨྲས། །བཟོད་པས་ཡང་དག་འཁྱུད་པའི་ཡིད་ལྡན་རྣམས། །ཁྲོ་བའི་ལས་ཀྱིས་ཆགས་པར་མི་བྱེད་དོ། །བི་ཀཱ་ར་བེ་གོ་པི་ན་པཱ་ཎི་པཱ་ད་ཙྪེ་དེ་མ་མཱ་བྷཱུརྨ་དི་བཱི་ཏ་མ་ཏྱཾ། །ས་ཏྲེ་ན་ཏེ་ཁྲོ་བྲལ་བདག་གི་རྐང་ལག་བཅད་གྱུར་ཀྱང་། །གལ་ཏེ་རྣམ་འགྱུར་ཤུགས་ལྡན་མ་གྱུར་ན། །བདེན་པ་དེ་ནཱ་ཀྵ་ཏ་དེ་ཧ་ཨེ་བ་སྟ་མི་ཏྱཱ་བ་དཱི་ས་སུ་ནཔྲ་སཱ་དཱི། །ཏ་ཏཁཱ་ཎཱ་ཏ་སངྒ་ཡིས་ཉམས་མེད་ལུས་ལྡན་པར། །བདག་གྱུར་ཅིག་ཅེས་རབ་དང་དེས་སླར་སྨྲས། །དེ་ནས་བདེན་ཚིག་ཏ་པཱ་ཎི་པཱ་དཾ་རཱུ་ཌྷ་བྲ་ཎཾ་བ་ཏྱ་ས་དོ་ད་ཡེ་ན། །ཨ་བུ་ཛ་ཡཏ་གཱན་ཏི་གུ་བརྗོད་པས་སྐད་ཅིག་ལ། །ལག་པ་རྐང་པ་འབྱར་ཞིང་རྨ་དག་གསོས༅། །ལྷ་ཡིས་བཟོད་པའི་ཡོན་ཏན་དེ་ཎཾ་སྟ་བེ་ན་ཏཾ་དེ་བ་ཏཱ་ས་ཏཱ་སི་ཏི་པུཊྚི། །རཱ་ཛཱ་པི་ཏ་ཏ་ཀིཝི་ཥ་ཀཱ་བསྟོད་ཅིང་། །སྙིང་སྟོབས་དཀར་བའི་མེ་ཏོག་རྣམས་ཀྱིས་མཆོད། །རྒྱལ་པོའང་དེ་སྡིག་ནག་པོ་བརྩེགས་།
