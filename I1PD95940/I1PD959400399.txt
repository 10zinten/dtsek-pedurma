རཿ༈། བི་ཛ་ཧཱ་ར་པ་རོ་ཛི་རྒྱ་ཧཾ་སཱ་ནཾ་པཉྩ་ཤ་ཏི། །པཱུཧྥ་པུ་སྐྱེ་ནུ་བྷཱ་དེ། །ངང་པ་དག་ནི་ལྔ་བརྒྱ་དང་། །མཚོ་སྐྱེས་ཅན་དེར་རྣམ་པར་རོལ། །སྔོན་གྱི་བསོད་ནམས་མཐུ་བེ་ན་ཏཾ་རཱུ་པཱ་ཏི་ཤ་ཡོཛྫ་ལཾ། །ད་དན་ཀནཱ་ཏྱཱཏྠཾ་ཚུ་ཛ་ནོ་ཅི་ཤུ་ཡིས་ནི། །ཁྱད་འཕགས་གཟུགས་རབ་འབར་བ་དེ། །བྱ་བ་དོར་བའི་སྐྱེ་བོ་ཡིས། །གཡོ་མེད་མིག་ལ་ལོ་ཙ་ན། །ཏཾ་ཐུ་ཏྭཾ་བྷཱུ་བ་ཏིས་ཏྲ་སྦི་ཏཾ་ཏརྡན་ནོརླུ་ཀ། །ཀུ་ཤ་གིས་མཐོང་བར་གྱུར། །དེ་ནི་དེར་གནས་ས་བདག་གིས། །ཐོས་ནས་དེ་དག་བལྟ་འདོད་པས། །རྒྱ་ཡིས་ལཱན་གྲ་ཧ་ཎེ་ཏ་སྱ་བྱ་སིཾ་ཛ་ཛཱ་ལ་ཛཱི་བི་ན། །ཏསྨིན་གྲྭ་ཧཱི་ཏེ་ན་ལི་ནཱི་ལཱི་འཚོ་བའི་མཁས་པ་དག། དེ་བཟུང་སླད་དུ་བཏང་བར་གྱུར། །པདྨ་ཅན་གྱི་རོལ་པའི་འཛུམ། །དཀར་ལཱསྨི་ཏ་ཤི་ཏ་ཏཱི་ཎི། །ཤ་ཏཱ་ནི་པཉྩ་ཧཾ་སཱ་ནཾ་ཏྱཱཁྭ་ཏཾ་པྲ་ཡ་ཡུརྫ་བཱཏ། །བའི་འོད་ཟེར་དེ་བཟུང་ཚེ། །ངང་པ་ལྔ་བརྒྱ་དག་གིས་ནི། །དེ་བཏང་མགྱོགས་པར་རབ་ཏུ་སོང་། །ཨེ་ཀསྐུ་ཏ་སྱ་སོ་ཛ་རྩྭ་ད་བདྡྷོ་པི་སུ་བདྷ་བཏ། ་། ཏ་དརྦ་གཅིག་ནི་དེ་ལ་གཞུང་བཟང་བས། །བཅིངས་པ་མེད་ཀྱང་བཅིངས་པ་བཞིན། །མཛའ་བའི་ཞགས་ཡུ་བྱ་ཐི་ཏསྟ་སྒྲོ་སྲེ་མ་པཱ་ཤ་བ་ཤཱི་ཀྲྀ་ཏ། །ཏ་ཏསྟེ༔པྲ་པི་ཏཾ་རཱ་ཛ་པས་དབང་བྱས་ནས། །དེ་ཡི་དོན་དུ་གདུང་ཞིང་གནས། །དེ་ནས་རྒྱལ་པོས་ངང་པ་ཡི། །རྒྱལ་པོ་དེ་ཧཾ་སཾ་བི་ལོ་ཀྱ་ཏཾ། ་། སྣེ་ཧ་པདྡྷཾ་དཱི་ཏཱི་ཡཉྩ་བིསྨི་ཏཱལྟ་བ་ལོ་ཙ་ཡཏ། །ཡིས་བཟུང་བ་དང་། །གཉིས་པ་མཛའ་བས་བཅིངས་པ་ཡང་། །མཐོང་ནས་ཡ་མཚན་དེ་དག་བཏང་། །ཧཾ་སཔཱུརྞ་མུ་ཁསོཉཧ་མཱ་ནནྡསྟ་སྱ་ཙཱ་ནུ་ག། །ག་ཏཱསྟ་ད་རྱ་ཙ་ཏུཏྟཱ་ངང་པ་གང་བའི་བཞིན་དེ་བདགི། །དེ་ཡི་རྗེས་འབྲང་ཀུན་དགའ་བོ། །དེས་ན་དེང་ཡང་བདག་ཚྭ་ཧཾ་སཱ་ཨེ་བ་བྷི་ཀྵ་བ། །པཱུཏྤ་སྨིན་ན་བྷ་བཏྟཱ་ལེ་བཱ་རཱ་ཎ་ཧཱུཾ་མ་ཧཱི་བཏང་ནས། །སོང་བའི་དགེ་སློང་ངང་པ་ཉིད། །གང་གི་གྲགས་པ་ཕ་རོལ་གྱིས། །ཡིད་ཀྱི་གླེགས་བུ་ལ་པ་ཏི། །ཨོཊྚ་ནཱ་མ་མ་ནབཉྫ་ལི་ཁི་ཏཾ་ཡདྨ་ཤབ་རི། །ས་ཧ་སྲ་ཡོ་དྷས་བྲིས་པ། ་། ་ཝཱ་རཱ་ཎ་སཱིར་སྔོན་དུས་ན། །ས་བདག་ཨ་ཊ་ཞེས་པ་བྱུང༌། །སྟོང་དང་གཡུལ་འགྱེད་ལྷོ་—།
