ཏ་བཱ་རཱ་ཎ་སིཾ་ན་ར་ཤྭ་ར། །ཡ་ཡུགླི་ཏི་ཤྭ་རསབྦེ་ཡ་སྱ་ལསྒྱུ་དབང་། །དབང་པོའི་སྡེ་ཆེན་ཞེས་པ་བྱུང་། །གང་གི་དཔལ་གྱིས་ས་ཡི་དབང་། །ཐམས་ཅད་སྐྱེངས་པ་བི་ལཀྵ་ཏཾ། །ཙནྡྲ་པྲ་བྷཱ་བྷ་བཏྟ་སྱ་དི་བཱི་ཀཱིརྟི་རི་བ་པྲ་ཡཱ། །ཡ་སྣུས་ཏྱ་པྲ་བྷཱ་ཉིད་དུ་གྱུར། །དེ་ཡི་ལྷ་མོ་གྲགས་པ་བཞིན། །ཡིད་འོང་ཟླ་བའི་འོད་ཅེས་གྱུར། །གང་གི་བདེན་པ་བེ་ན་སཱ་རཿས་ཏྱ་ཏཱ་མཱ་ཡ་ཡུ། །བ་བྷ་བ་ས་མ་ཡེ་ཏསྨིན་མྲྀ་ག་ཡུ་ཐ་པ་མཐུ་ཡིས་ནི། །རྨི་ལམ་བདེན་པ་ཉིད་དུ་གྱུར། །དུས་དེར་ནགས་སུ་རི་དྭགས་ཀྱི། །ཁྱུ་ཡི་བདག་པོ་བྱུང་ཏིརྦ་ནེ། །སུ་བཎྞ་པཱནྡྷ་ཨི་ཏཥ་ནཱ་མ་ཧེ་མ་མ་ཡཙྪ་བི། །ནཱི་ལ་རཏྣ་གྱུར་པ། །གསེར་གྱི་ལོགས་ཞེས་བྱ་བའི་མིང་། །ཐོབ་ཅིང་གསེར་གྱི་རང་བཞིན་སྣང་། །མུ་ཏིག་དོ་ཤལ་ད་ར་སྭ་ར་མུཀྟ་ཧཱ་ར་ནི་བྷ་པྲ་བྷཱ། །བ་བྷཱཛྫཋ་ཙྪ་ཆ་་ཡ་སྱ་བྷུ་ཥ་གསལ་བའི་ཁོངས། །རིན་ཆེན་མཐིང་གིས་སྤྲས་འདྲའི་འོད། །གང་གི་མིག་གི་རྫི་མ་ནི། །ནགས་ཀྱི་ཎཾ་ཀཱ་ན་ན་ཤི་ཡ། །པྲ་བཱ་ལ་བལླི་ཤྲཱིངྒ་སྱ་ཙི་ཏྲ་རཏྣ་ཙི་ཏ་ཏཱ་ཙ། །ཏ་དཔལ་གྱི་རྒྱན་དུ་མཛེས། །བྱུ་རུའི་འཁྲི་ཤིང་རྭ་ཅོ་དང་། །རིན་ཆེན་བཀྲ་བའི་པགས་པས་རྒྱས། །ངོ་སཱུརྱ་སུ་ཏཾ་བྷོ་དྷི་ལ་ཧ་རཱི་རུ་རུ་ཙི་རུ་ཙི། །བོ་དྷི་སཏྭ་བ་ཏ་ར་སྱ་མཚར་བདུད་རྩིའི་ཆུ་གཏེར་གྱི། །རྦ་རླབས་དེ་ཡི་འོད་ཟེར་མཛེས། །བྱང་ཆུབ་སེམས་དཔར་ཞུགས་ཏ་སྱ་ཀཱནྟ་མ་བྷཱུཏྠ་པུ། །པཱུབྷཾ་སུ་ཀྲྀ་ཏ་ཙི་ཏྲ་སྱ་ལཀྵཎཾ་གྱུར་པ། །དེ་ཡི་གཟུགས་ནི་མཛེས་པར་གྱུར། །གང་ཕྱིར་ལེགས་བྱས་རི་མོ་ཡི། །དང་པོའི་མཚན་ཉིད་ཧི་སུ་རཱུ་པ་ཏཱ། །དཱིགྷ་དྲིཊཱི་རྦ་བྷཱུ་བཱ་སྱ་བ་ཡ་མྱོ་བྲྀཏྟཱ་བཱ་ཡ་སྱ། །ལུདྡྷ་གཟུགས་བཟང་ཉིད། །དེ་ཡི་གྲོགས་པོ་བྱ་རོག་ནི། །རྒན་པོ་རིང་དུ་མཐོང་བར་གྱུར། །རྔོན་པ་ཀཱ་ནེ་ཥ་ཎ་ཏཱ་སྱེ་དིགླི་ལོ་ཀ་ན་ཏཏྲ་ར། །ཏོ་ཀ་ཐཱ་བྷིརྨི་ཐཔྲ་ཏྭཱ་འཚོལ་བས་འཇིགས་པ་ཡིས། །དེ་ནི་མཆོག་ཏུ་ཕྱོགས་རྣམས་ལྟི། །དེ་དག་ཕན་ཚུན་དགའ་བ་ཡི། །བི་བིཀྟ་ཥཱ་བི་ཛ་ཧྲ་ཏུ། །པཱཀྲུ་ཙྪིཛཱ་ཡ་ཏེ་བཱ་ཎཱི་ཏི་ར་མ་པི་མཱ་གཏམ་གྱིས་དབེན་པར་རྣམ་པར་རྩེན། །དུད་འགྲོ་ལ་ཡང་སྔ་མ་ཡི། །།བསོད་ནམས་རྣམས་ཀྱིས་མི་—།
