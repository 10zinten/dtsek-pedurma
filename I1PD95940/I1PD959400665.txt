ཀུ་ལེ་ཛ༔་གུ་ཎ་བཱསྟིཆི་བེ་ཀཱ་ལཾ་ཀྲྀ་ཏ་མ་ཏི། །སརྦ་བྷཱུ་ཏ་ད་ཡཱ་རིགས་སུ་སྐྱེས་ཤིང་ཡོན་ཏན་ཐོབ། །བློ་གྲོས་དབེན་པས་རབ་བརྒྱན་ཅིང་། །འབྱུང་པོ་ཀུན་ལ་བརྩེ་ཞིང་མི་ཏི་ལག་ཎཾ་བུ་ཅུ་ཀརྨ་ཎི། །བི་རཱ་ཀྟ་བྷི་ར་ཏིག་ཏྟཱ་ས་ཡུ་བི་བ་བྱམས། །བསོད་ནམས་ལས་ཀྱི་མཚན་ཉིད་ཡིན། །གཞོན་ནུ་དེ་ཉིད་ཆགས་བྲལ་གྱིས། །དགའ་ཞིང་ཏ་པོ་བ་ནཾ། །དཏྟ་བྲ་ཏོ་མ་ཧནྟི་བྷཾ་བྷེ་ཛེ་བི་ཤཱནྟི་མཱ་ཤ་དཀའ་ཐུབ་ནགས་སོང་སྟེ། །དྲང་སྲོང་གཉིས་ཀྱིས་བརྟུལ་ཞུགས་བྱིན། །བསྟི་གནས་སུ་ནི་ངལ་གསོ་མེ༑། ཏ་ཏཀཱ་ལེ་ན་སི་ཏ་ཥ་བི་མ་ལ་ཛྷཱ་ན་ལོ་ཙ་ན། །ཨཱ་སནྣཾ་པྲ་ས་ཏཱཾ་བྱཱ་བསྟེན། །དེ་ནས་དུས་ཀྱི་ཡེ་ཤེས་སྤྱན། །དྲི་མེད་ཤིན་ཏུ་རྣོ་བ་ཡིས། །སྟག་མོ་བཙའ་ལ་ཉེ་བ་དག། གྷི་ས་དྲྀཥྛ་ས་མ་ཙིནྟ་ཡཏ། །པྲ་ས་བོསྟགླུ་དྷཱནྟ་ཡཱ་སཔ་ཧ་ན་བྷ་མཐོང་ནས་དེ་ཡིས་ཡང་དག་བསམས། །བཀྲེས་པ་ཡིས་ནི་གཟིར་བ་འདི། །ཞག་བདུན་དག་ན་བཙའ་བི་ཥྛ་ཏི། །སྤྲི་ཧཱ་ཙིཏཱཉྩ་ཏེ་ཏཱི་བཱ་ནི་ཛ་པོ་ཏ་ཀ་བྷཀྵ་ཎེ། །ཨི་ཏི་སཾ་ཙི་ཏྲ་བར་འགྱུར། །རང་གི་བུ་ནི་ཟ་བ་ལ། །སྐོམ་པ་དྲག་པོའང་འབྱུང་བར་འགྱུར། །ཞེས་བསམས་དེ་ཡིས་ཏཇུ༔ཁཾ་མུ་ནི་བི་བེ་བྱ་ས། །ཙ་ཀཱ་ར་ཏཏྲ་ཏི་ཀཱ་རེ་ཀཱ་རུ་སྡུག་བསྔལ་ནི། །ཐུབ་པ་དག་ལ་བཤད་ནས་དེས། །སྙིང་རྗེ་དེ་ནི་ཕྱིར་བཅོས་པའི། །ཡིད་ལ་རེ་བ་རྩེ་ན་མ་ནོ་ར་ཐཾ། །ཏ་ཏཔྲ་ཡ་ཏེ་སམྨཱ་ཧེ་བྱཱ་གྷི་གབྷ་བྷ་རཱ་ལ་ས། །བ་ཧུ་བྱས་པར་གྱུར། །དེ་ནས་ཞག་བདུན་འདས་པའི་ཚེ། །སྟག་མོ་མངལ་གྱི་ཁུར་གྱིས་དུབ། །ཟས་ཆག་པ་བཱ་ས་སནྟཔཱ་ཀྲྀཙྪེ་ཎ་པཱུ་ཏ་པོ་ཏ་ཀཱན། །ནི་ཛ་ཤོ་ཎི་ཏ་གནྡྷེ་ན་ཛཱ་ཏ་མང་པོས་གདུངས་གྱུར་པ། །དཀའ་ལས་ཆེན་པོས་བུ་ཚ་བཙས། །རང་གི་ཁྲག་གི་དྲི་ཡིས་ནི། །སྲེད་པ་ཏཱི་པྲ་ཏ་ར་སྤྲོ་ཧཱི། །ས་ཏྱ་བྲ་ཏསྟ་མཱ་ལོ་ཀྱ་ད་ཡཱརྔས་མ་ཙིནྟ་ཡཏྲ། །དྲག་པོ་རབ་སྐྱེས་པ། །དེ་མཐོང་བདེན་པའི་བརྟུལ་ཞུགས་ནི། །སྙིང་རྗེས་བརླན་པས་ཡང་དག་བསམས། །ཨི་དཾ་བ་རཱ་ཀཱི་གླུདྡཁཱ་དུ་བྱ་ཏཱ་པོ་ཏ་བྷ་ཀྵ་ཎེ། ་། ཨ་ཧོ་བ་ཏཱ་སྱཱསཱརྗེ་ན་སྡུག་མ་འདི་ནི་བཀྲེས་པ་ཡིས། །སྡུག་བསྔལ་དག་ལས་བུ་ཟར་བརྩོན། །ཀྱེ་མ་འདི་ནི་རང་དོན་གྱིས། །—།
