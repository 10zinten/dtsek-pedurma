དྷོ་ཛྲ་པཱདྨ་ཛ། །ཨ་ཧོ་མ་ཧོ་ད་ཡ་ཏཱངྒེ་པྲ་པྲ་ཛཱ་དུརྦ་བྷཱ་བྷ་བཏ། །དཱ་པོའི་སྲས་དེས་བསམས། །ཀྱེ་མ་དར་བ་ཆེ་ཉིད་ཕྱིར། །བདག་ལ་རབ་བྱུང་དཀོན་པར་གྱུར། །དབུལ་རི་རྒྱ་ད་བི་བེ་ཀཙཱ་ནཱི་ཙཱ་ནཱ་མ་པི་དུརླ་བྷཱ། །ཛཱ་ཡེ་ཡཾ་ཡ་བྱ་དྷ་མ་ཞིང་རྣམ་འབྱེད་མེད་ལས་ནི། །དམན་རྣམས་ཀྱིས་ཀྱང་རྙེད་པར་དཀའ། །དེས་ན་དམན་རིགས་རྣམ་ཀུ་ལེ་ཏསྣ་དརྨི་བི་བེ་ཀ་བཱན། །ས་ཨེ་པ་ཤཱ་རི་པུ་ཏྲི་ཡཾ་ཛཱ་ཏསྟཏ་པྲ་ཎི་དྷ་འབྱེད་དང༌། །ལྡན་པར་བདག་ནི་སྐྱེ་བར་ཤོག། དེ་ཡི་སྨོན་ལམ་དག་ལས་ནི། །དེ་ཉིད་ཤཱ་རིའི་བུ་ན་ཏ། །པྲ་བྲ་ཛི་ཏོ་བྷ་ག་བ་ཏཱ་ཀཱ་བྱ་སེ་ནཱ་རྱ་ཛརྨ་ནི། །ཏེ་ནཱ་འདིར་སྐྱེས། །སྐྱེ་བ་གཞན་དུ་བཅོམ་ལྡན་འདས། །འོད་སྲུངས་ཀྱིས་ནི་རབ་ཏུ་བྱུང་ཏེ། །བདེན་པའི་ཡཾ་ནི་ཡ་མ་སྲི་ཡི་ཎི་བི་ན་ཡི་ས་མྱཀྵ་སཱ་དོ་ད་ཡཱ་དཱི་དིཋཀུ་ཤ་ལཱ་ཡ་གཏེར་དེ་ངེས་པར་དགྱེས་པས་འདུལ་ལྡན་འདི་ནི་དགེ་བ་དག་གི་སླད། །ཡང་དག་རབ་ཏུ་དང་བ་རྒྱས་པས་སཏྱ་ནི་དྷི་ནཱ་པྲཛྙཱ་བ་ཏཱ་མ་གྲ་ཎི། །ཀཱ་ལེ་ཤཱཀྱ་མུ་ནིབྷ་བི་ཥ་ཏ་མ་ཏཤི་ཥྱ་ཏཱ་ཡོ་དུས་སུ་ཤེས་རབ་ལྡན་པ་རྣམས་ཀྱི་མཆོག། ཤཱཀྱ་ཐུབ་པའི་སློབ་མར་འོས་པ་དག་ལས་མཆོག་ཏུ་བཞེད་པར་གཱཏ་པ་རཾ་མོདྷཱ་མྱ་ཡ་ཎ་ཨེ་ཥ་ཙཱ་ཏྲ་ཀ་ཐི་ཏསམྱིརྨ་ཡཱ་ནཾ་བ་འགྱུར་བར་བསྟན། །མོ་གལ་བུ་འདིའང་འདི་རུ་གསུངས་ཞེས་ཡང་དག་རིག་པའི་རང་བཞིན་ལྡན་རྣམས་དཱ༑། ཨ་ནྱ་ཛརྨ་ནི་ད་རི་དཱ་ཀཱམྦི་ཀཀ་ན་ཙིཏ་པ་ད་ཡ་ཡཱ་མ་ཧཎྜི་ཎཱ། །དཏྟ་སྨྲ༑། སྐྱེ་བ་གཞན་དུ་དབུལ་པོ་ལས་བྱེད་ལ། །དྲང་སྲོང་ཆེན་པོ་འགའ་ཞིག་བརྩེ་བ་ཡིས། །ལྷུང་པཱ་ཏྲ་ཙཱི་པ་རོ་བྷ་བརྡནི་ཏརྙི་ར་ཏུ་ལ་པྲ་བྷཱ་བ་བཱན། །ཨི་ཏི་ཀེ་བཟེད་ཆོས་གོས་དག་ནི་བྱིན་གྱུར་ནས། །མཚུངས་མེད་རྫུ་འཕྲུལ་མཐུ་ཡང་རབ་ཏུ་བསྟན། །ཞེས་པ་དགེ་མེརྩྭ་བི་ར་ཙི་ཏཱ་ཡོ་བོ་དྷི་ས་ཏཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་ཤཱ་རི་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ཤཱ་རིའི་པུ་ཏྲ་ཕྲ་བྲ་ཙཱ་བ་དཱ་ནནྟཱ་མོ་ན་བི་ཤ་ཏིཔལླ་བ། །བུ་རབ་ཏུ་བྱུང་བའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་བཅུ་དགུ་པའོ། །།—།
