བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་༑། ས་པྲཙྪ་ཧཱུཾ་དུས་ཧ་དུ༔ཁ་ཧ་ཏུ་མ་བྷེ་ཏྱཱ་བི་རཱུ་དྷ་ར་རཱ་ཛ་པུ་ཏྲ། །ཟུག་པ། །རིག་འཛིན་རྒྱལ་པོའི་བུ་ཡི་མངོན་ཕྱོགས་ཏེ། །དེ་ལ་བཟོད་དཀའི་སྡུག་བསྔལ་རྒྱུ་དག་དྲིས། །ཀ་སྨཏྟ་བཱ་མཾ་ང་ན་དུཁ་པཾ་ཤྲཱི་མཱ་ཏཔྲ་ཛཱ་ཏཔྲ་ས་བྷཾ་པྲ་ལཱ་པ། ་། ཀིཾ་ཏཱི་པྲ་མ་མ་མ་ཁྱོད་ཀྱི་ཆོ་ངེ་དྲག་པོ་འདི། །སྡུག་བསྔལ་དྲག་པོའི་གསལ་བྱེད་ཅི་ལས་སྐྱེས། །དགེ་བའི་གཟུགས་ཀམྤ་ཀ་མ་སྱ་སཱ་དྷོཀ་ལྱཱ་ཎ་མཱནྟེ་ར་ཏི་ཤྲཱི་ཀ་སེ་ཏྭཾ། །བ་པཱུ་ཎི་ཀཱ་མཾ་ཀུ་ཤ་ཅན་དམ་པ་འདི་ལ་ཁྱོད། །དྲག་པོ་ཀུན་ཏུ་འདར་བྱེད་ཅི་ཞིག་དོགས། །ངེས་པར་དགེ་བ་དག་ལ་འོས་ལོ་ཙི་ཏཱ་ནི་སོ་ཛ་རྒྱ་སམྨཱ་ད་སུ་ཁ་པྲ་དཱ་ནི། ་། ཨ་པཾ་བི་དྷཱ་ནི་པྲ་ཏི་བྷཱ་ཏི་ནཱི་ཧཱུཾ་ནི་པའི་ལུས། །སྐྱེ་བོ་བཟང་པོ་དང་འགྲོགས་བདེ་བ་སྟེར། །རྣམ་པ་འདི་ལྟ་བུ་ནི་རབ་འཇོམས་པའི། །རྒུད་བ་སྱ་ད་ཏྭཾ་བི་པ་དཾ་པྲ་ཡཱ་ཏི། །སཾ་ཀཱནྟ་ཏཇུཁ་བི་ཥཱ་ཀུ་ལེ་ན་པྲྀཏྟེ་ཏི་སཱ་པའི་གནས་ཉིད་རབ་ཏུ་མི་འགྱུར་རོ། །ཞེས་པ་དེ་ཡིས་སྡུག་བསྔལ་དུག་འཕོས་འཁྲུགས། །བརྩེ་བའི་ཏེ་ན་ད་ཡཱ་མ་ཡེ་ན། །ཏམ་བྲ་བཱིཏི་ལི་ཏ་ལོ་ཙ་ནེ་བ་བི་ཡོ་ག་བྷི་ཏྭཱ་བ་ད་རང་བཞིན་དེ་ཡིས་དྲིས་ལ་དེས། །འབྲལ་བས་འཇིགས་ནས་བུ་ཡི་བཞིན་ལ་མིག། ཕུར་པ་བཙུགས་བཞིན་ནེ་སུ་ཏ་སྱ། །ཀཾ་ནི་སྤྲ་ཏཱི་ཀཱ་ར་ས་ནུངྒ་མེ་ན་ཤུ་ཏེ་ན་ཏེཞསྨདྨ་ས་ནཱ་ན་ལེ་དེ་ལ་རབ་སྨྲས་པ། །བཅོས་ཐབས་མེད་པར་གྱུར་པ་ངེད་ཀྱི་ནི། །སྡུག་བསྔལ་མེད་དག་ཐོས་པས་ཁྱོད་ལ་ན༑། ཨ་ཀཱ་ལ་པཱ་ཏཱི་མ་མ་དུས་ཧོ་ཡཾ་དུཀརྨ་པཱ་ཀོ་པ་ན་ཏཔྲ་ཀཱ་ར། །ནི@༑:། ལས་ངན་སྨིན་པ་ཉེར་གནས་བཟོད་དཀའ་ཡི། །རྣམ་པ་འདི་ནི་བདག་ལ་དུས་མིན་ལྷུང་། །བི་ཤཱ་ལ་ཀཱིནྡྷཛྫ་ལ་ཤཾ་ཁ་པཱ་ལ་ཀུ་ལཾ་ཀུ་རེ་བཏྠ་ཏ་རེ་མ་མཱསྨིན། །མཱུ་ལཱ་བྷི་གྷ་རྒྱ་ཆེ་གྲགས་འབར་དུང་སྐྱོང་རིགས་ཉིད་ཀྱི། །མྱུ་གུ་བདག་གི་བུ་གཞོན་འདི་ལ་ནི། །བྱེད་པོ་གདུག་པས་ཏཱི་བི་དྷུ་རེ་ཎ་དྷཱ་ཏཱ་སཛྫི་ཀྲྀ་ཏོ་ཡ་ཀ་ཐི་ནཀུ་ཋཱ་ར། །ཨེ་ཀས་དཱ་བྷཀྵ་ཎ་དཱིཀྐི་རྩ་བ་འཇོམས་པ་ཡི། །སྟ་གྲི་རྣོན་པོ་འདི་ནི་ཆས་སུ་བྱས། །རྟག་ཏུ་ཟ་བར་སྟོན་བྱེད་མཁའ་ལྡིང་ལ། །ཏསྨ་ཏཱཎཱ་སྱ་སཉྫ་་ཡ་རཀྵ་ཎཱ་ཡ། །བཱ་རེ་ཎ་ནཱ་གཕ་ཎི་ནཱ་ཡ་ཀེ་ན་བི་སྲ་ཚུ་ཏེ་ཐམས་ཅད་བརླག་པ་བསྲུང་སླད་རེ་མོས་ཀྱི། །གདེངས་ཅན་འདྲེན་པ་དག་གིས་ཀླུ་དེ་ནི། །གོས་དང་སྐྱེ་ཏཱ—།
