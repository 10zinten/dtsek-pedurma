མ་བྲྀཋལྦ་པཉྩ་མེ། །རཏྣ་པྲཋསྟ་ཏཔཾ་ཤུ་བྲིཋཤུ་སཔ་མེ། །ཏ་ཡཱ་ས་བནྡྷ་ཡང་གསེར་གྱི་ཆར། །དེ་ནས་རིན་ཆེན་ཆར་པ་སྟེ། །རྡུལ་གྱི་ཆར་པ་བདུན་པ་ལ། །དེ་ཡིས་ས་བདག་རཱཙྩོ་སོ་ན་བྷ་བི་ཥ་ཏི་བྷུ་བ་ཏི། །ཏམདྡྷ་པ་གནྟ་བྱ་ར་རྩྭ་གཉེན་དང་ནི། །ཡུལ་འཁོར་བཅས་འདི་མེད་པར་འགྱུར། །དེ་སླད་ཁྱེད་གཉིས་རིན་ཆེན་ནི། །མང་པོ་དཱ་ཡ་བྷཱུ་ཡ་སེ། །ཨི་ཏི་ཏངྒ་ཙ་ནཾ་ཐུ་ཏྭཱ་བི་ནི་ཤི་ཏྱ་ཏ་ཐི་ཏི་ཏོ། །ཧེ་རུ་ཀབླངས་ནས་འགྲོ་བར་བྱ། །ཞེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །དེ་བཞིན་ཞེས་ཏེ་དེ་དག་སོང་། །སྔོ་བསངས་བྱ་མ་ཀཾ་པུ་ཊཾ་ཏ་མྱོ་པསྨཱ་ཡ་ཀཾ་བྱ་དྷཱ་ཏ། །བྷི་རུ་ཀ༔་སུ་ཏཱ་ཤྩ་མཱ་བ་ཏཱི་མཱ་དཱ་ཡ་པཱ་ཧེ་རུ་ཀ་ཡི་བུ། །དེ་ཡི་ཉེ་བར་གནས་པ་བྱས། །བྷི་རུ་ཀས་ཀྱང་བུ་མོ་ནི། །སྔོ་བསངས་ཅན་མ་ལག་ཎི་ནཱ། །ཨི་བྷེ་ཏ་བྷིཀྵ་ཀིཾ་ཤི་ལོ་པྲ་ཎ་ཡཱ་དི་དམ་པྲ་བཱིཏ། །ཨཱརྱ་བྷ་བ་ཏུ་པས་ཁྲིད། །དགེ་སློང་མ་ནི་རི་མ་ལ། །མངོན་ཕྱོགས་གུས་པས་འདི་དག་སྨྲས། །འཕགས་མ་བདག་གི་མ་ཏ་ཀཱ་རྩྭ་གྷོ་ཥི་ལ་སྱ་གྲོ་ཧ་པྲ་བྷོ། །ཁཾ་ཧེ་ས་མར་ཎཱི་ཡེ་ཡ་མཱ་སནྣ་པྲ་བུ་མོ་ནི། །རྒུད་པ་ཞི་བར་མཛད་ཁྱེད་ཀྱིས། །ཁྱིམ་གྱི་བདག་པོ་གདངས་ཅན་གྱི། །ཁྱིམ་དུ་ཡང་དག་ཏི་པསྣ་ཡཱ། །ཨེ་བ་མུཀྐཱམི་ཏཱ་པ་ཏྭཱ་བ་མཱ་ཏྱཾ་ཛགླ་ཏུཀྲྀ་ཧཾ། །ཤི་ལྟ་བི་གཞག་པར་བྱ། །ཞེས་སྨྲས་བུ་ཚ་གཏད་ནས་ནི། །བློན་པོ་དག་ནི་རང་ཁྱིམ་སོང༌། །རི་མོ་ལཡིས་ཀ་ནཱ་མཱ་ད་ཡ་བྲ་ཡ་ཡོ་གྷོ་ཥི་ལཱ་ལ་ཡཾ། །ཏ་ཏཀྲ་མེ་ཎ་ཏ་ད་བྷུཏ་ཡ་ཐོཏྟཾ་ཀྱང་བུ་མོ་བླངས། །གདངས་ཅན་གྱི་ནི་གནས་སུ་སོང་། །དེ་ནས་ཇི་ལྟར་དགེ་སློང་གིས། །བརྗོད་དེ་བྷི་ཎཱ་པུ་རེ། །ཛྙཱ་ན་དཱི་པ་བ་ཏི་པྲ་ཛྙཱ་ཡ་ཐཱ་ཏ་ཏྭཱ་ཧི་པ་བྱ་ཏི། །ཥཥྛ་བྷི་རིམ་པས་གྲོང་དུ་བྱུང་། །ཡེ་ཤེས་སྒྲོན་ལྡན་ཤེས་རབ་ཀྱིས། །དེ་ཉིད་ཇི་ལྟ་བཞིན་དུ་མཐོང་། །དེ་ནས་ཉི་རཏྣ་བརྟཉཐ་པ་ཏི་ཏེ་རཏྣ་པཱུ་རི་ཏཱི། །ཡ་ཡ་ཏུནྣ་བ་མཱ་དཱ་ཡ་ཏཱ་བ་མཱ་མ་དྲུག་@པ་ལ། །རིན་ཆེན་ཆར་ནི་བབ་པའི་ཚེ། །བློན་པོ་དེ་དག་རིན་ཆེན་གྱིས། །གང་བའི་གྲུ་ཏྭཱ་བ་ལའི་ཏོ། །ཏོ་དཀྵི་ཎཱ་དི་བྷཾ་ག་ཏཱ་ཙཀྲ་ཏུནྣཾ་ག་ར་དྭ་ཡཾ། །ཧེ་རུ་ཀོ་བཟུང་མ་ཚོར་སོང་། །དེ་དག་ལྷོ་ཡི་ཕྱོགས་སོང་ནས། །གྲོང་ཁྱེར་གཉིས་བྱས་ཧེ་རུ་ཀས། །ཧེ་རུ་ཀ་ཞེས་།
