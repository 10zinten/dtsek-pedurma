ནིཟི་ཀཱ་ར་བ་ལོ་ད་ཡཾ། །ཨུཏྟཱ་ཧ་ཤི་ཁ་རཱ་རཱུ་ཌྷ་མེ་ནེ་རཱ་ཛཱ་ཛ་ཡ་ལྕགས་ཀྱུའི་དབང་གྱུར་དེ་མཐོང་ནས། །སྤྲོ་བ་ཡི་ནི་རྩེར་ཞུགས་པ། །རྒྱལ་པོའི་དཔལ་དུ་རྒྱལ་པོས་ཤི་ཡཾ། །ས་སཾ་ཛཱ་ཏ་པྲ་ཧརྟོརྦ་དཱཀྵ་ཤི་ཀླུ་དི་དྲྀཀྵ་ཡཱ། །ཏ་མཱ་རུ་བསམས། །དེ་ནས་རབ་དགའ་ཡང་དག་སྐྱེས། །མཁས་པར་བསླབས་པ་བལྟ་ཕྱིར་དེ། །འོད་སྟོང་ཤར་རོ་ཧ་སོཏྠ་ཧས་ཧ་སྲྀ་ཤུ་རི་བོ་ད་ཡཾ། །སཾ་ཡ་ཏོ་ཐ་ག་ཛེནྡྲ་མྱ་མནྡྲི་བ་གྱི་རི་ལ་བཞིན། །དེ་ལ་སྤྲོ་བ་ཆེན་པོས་ཞོན། །དེ་ནས་གླང་པོའི་དབང་པོ་དེའི༄༅། །དཀྱིལ་འཁོར་ཀུན་བ་ཤ་བརྟི་ན། །སརྦ་མཎྜ་ལ་སཾ་ཙཱ་ར་ཙཱ་ཏུརྱ་ས་མ་དན་ཡཏ། །ཏུ་ཡང་དག་རྒྱུ། །མཁས་པ་བློན་བཞིན་དབང་སྒྱུར་བ། །ཀུན་ཏུ་བགྲོད་ཀྱིས་ཡང་དག་བརྟན། །ག་ཛ་པྲ་གྷཱ་པྲ་སཾ་གེ་ན་མྲྀ་ག་ཡཱ་ཀེ་ལི་ལཱ་ལ་ས། །རཱ་ཛཱ་ནི་ཛོན་ཧ་མི་བ་རྒྱལ་པོ་རང་གི་སྤྲོ་བ་བཞིན། །གླང་པོ་བལྟ་བའི་ཞར་ལ་ནི། །རྔོན་ལ་རྩེ་ཞིང་དགའ་ལྡན་པས། །ཆེ་བའི་བྱ་གཱ་ཧ་ཏ་བ་ནཾ་མ་ཧཏ། །ས་ཡ་ཡོ་རངྒ་ཀེ་ཡཱུ་ར་ཀི་ར་ཎི་རྡུ་ར་སཇི་བྷི། །ཤལླ་ནགས་སུ་འཁྱམས་པར་གྱུར། །རིན་ཅེན་དཔུང་རྒྱན་འོད་ཟེར་ནི། །རིང་དུ་འཕྲོ་དེ་ཤལླ་ཀའི། །ལོ་ཀཱི་པལྤ་བ་བ་རིཇིགྷ་གཱ་ནཱ་ཧཱ་ཡརྨི་བ། །བྲ་ཛནྟི་ད་དྲ་བྱཱ༑འདབ་མཆོག་གི་ཕྱོགས་ཀྱི་ནི། །གླང་རྣམས་འབོད་པ་ལྟ་བུར་སོང་། །ནགས་སུ་དེ་ནི་འགྲོ་བ་དག། བ་ནེ་བ་ན་དེ་བ་ཏཱ། །པྲ་ཧནྟ་བིསྨྲ་ཡཱ་ཀཱིཎ་ཀཎ་པཱུ་རཱི་ཀྲྀ་ཏེ་ག་ཎཱ། །རབ་དགའ་ཡ་མཚན་ཀུན་རྒྱས་ཤིང་། །མིག་ནི་རྣ་བར་རྫོགས་བྱས་པ། །ནགས་ཀྱི་ལྷ་ཡིས་མཐོང་བར་གྱུར། ཤ་བ་རཱི་ཀ་བ་རཱི་པཱ་ཤི་པུཥྤ་སོ་ར་བྷ་ནིབྷ་རཱ། །བིན་རྒྱུ་བ་སུན་དྷཱ་རཱ་པི་ཤཾ་ནོར་འཛིན་གྱི་བདག་དབང་པོ་དེ། །ཤ་བ་རི་ཡི་སྐྲ་ཚོགས་ཀྱིས། །མེ་ཏོག་དྲི་བཟང་རྫོགས་པའི་རྒྱུས། །མ་རུ་ཏ༑སི་ཥ་བི་རེ། ་། ཨ་ཐ་བི་ནྡྷོ་པ་ཀཥྛ་ཥཱ་སྭ་ཙྪནྡ་སུ་ཁ་ཤཱ་འབིགས་བྱེད་རླུང་གིས་བསྟེན་པར་གྱུར། །དེ་ནས་འབིགས་བྱེད་ཉེ་བ་ནི། །རང་དབང་བདེ་བའི་འདབ་ཁི་ཥཱ། །སྨྲ་ཏཱ་བི་ལཱ་ས་བྲྀཏྟཱནྟཾ་ག་ཛ་སོཋཌཱ་ཏྭཾ་ཡ་ཡོ། །ཀ་རི་ཙྪཱཔྲ་ལྡན་པར། །རྩེ་དགའི་བསམ་པ་དྲན་གྱུར་ཏེ། །གླང་པོ་ཤིན་ཏུ་རྒོད་པར་གྱུར། །དགའ་བས་བཅིངས་།
