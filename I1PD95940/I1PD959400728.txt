བྱ་བི་རཱུཀྟ་ཡ་ཡིནྟ་ནི་ཏྱཾ་གྷཱ་ཡོ་རྱ་ཏཱ་བཱ་པཱ་ར་སཱ་ཏུ་ར། །ཏྭཱ་ཛཾ་ན་རཱ་ཛྲ་སཱ་ཤེས་སྨོད་མི་མཁས་རྗེས་སྨྲ་བས། །ཉིན་ཞག་དང་ནི་ནད་པ་འཕེན་པར་བྱེད། །རང་གི་བུ་ལའང་རྒྱལ་སྲིད་སུ་ཏོ་པི་རཱ་ཛཱན་སྤྱི་ཧཱ་པ་ར་བྷཾ། །ཏྱཱཀྟ་ཙ་ལགཱིཀུ་རུ་ཏེ་ཀྵ་ཎེ་མི་གཏོང་སྟེ། །རྒྱལ་པོ་གཞན་ལ་ཡོད་ཀུན་དགའ་སྐྱེད་བྱེད། །ཕུན་སུམ་ཚོགས་པ་བཏང་ན་སྐད་ཅིག་ན་བི་པཏྠ་ཧ་སྲ་ཛཱ་ལ་ནཱ་ནུ་ཏ་པཾ། །སསུ་ཏཱ་རོ་པི་ཏ་ཤེ་ཁ་རཱ་ཎཱ་གིས༑། རྒུད་པ་སྟོང་ཕྲག་མེ་ཡིས་གདུང་བར་བྱེད། །འཕྲལ་ལ་བུ་ལ་ཟུར་ཕུད་བཀོད་གྱུར་ཅིང་། །དེ་ཏཏྟཱ་ལ་མི་ལཉྫ་བྷུ་གོ་ར་པཱ་ཎཾ། །རཱ་ཛྙཱ་མ་ན་མི་ར་བ་དྷ་རི་ཏཱ་ནི་ཏྲ་ཊཱི་བྷ་བ་སྐྱེ་བ་དུས་རྗེ་བོའི་གུས་པ་ཉམས་གྱུར་པའི། །རྒྱལ་པོའི་བཀའ་ནི་དུད་པ་མིན་རྣམས་ཀྱིས། །ཆ་ཡི་གཅོད་ཧི་ཤཱ་ས་ནཱ་ནི། །ཨི་ཏི་་ཏཱི་ཤ་སྱ་ཛྷ་ཏི་བི་དྷཱ་ནིརྒ་ཏུ་ཏསྣ༴་ཅིང་རྩྭ་དང་མཚུངས་པར་འགྱུར། །ཞེས་པས་ས་དབང་བརྟན་པ་བསྒྲུབས་ནས་ནི། །ཁང་ནང་དེ་ལས་བ་ནཾ་ཏ་རཱཧཱ། །ཨ་ནཱི་ཥྛ་ཏཏྟཱ་ལ་ག་དཱ་བྷི་བྷུ་ཏ་མ་བྷི་རེ་མ་ཀཱནྟ་མ་ཐཱ་ནི་དེ་ནི་ཕྱི་རོལ་སོང་། །བ་རྫི་དེ་མཚུངས་ནད་གྱིས་གཟིར་བ་ཞིག། བཙལ་ནས་དབེན་པའི་གནས་སུ་བོས་ནཱ་ཡ། །ཀཱུ་རཱ་ཤ་ཡ་ཀླུ་ར་དྷི་ཡི་བ་དཱ་སྱ་ཧ་ཏྭཱ་ཏ་མུཏྟཱ་ཊི་ཏ་ནཱ་བྷི་ཀོ་ཤཾ། །བྱས་ཏེ། །གདུག་པའི་བསམ་པས་འབངས་མོ་གདུག་པའི་བློས། །དེ་བསད་ལྟེ་བའི་མཛོད་ནི་རབ་ཕྱེ་སྟེ། །ཏ་སིཾ་ཏྲ་ལགཾ་པ་རུ་ཎཾ་ད་དན་ཤྲཱི་ཎཱ་བི་ཧཱི་ནཱི་བི་ཀྲྀ་ཏ་ཀྲྀ་མི་སཱ། །ཨུཛྙཱ་པྲ་ཙཱ་བརྩེ་བས་སྨན་དེས་དེ་ཡི་རྒྱུ་མར་ནི། །རྣམ་འགྱུར་རྩུབ་མོའི་སྲིན་བུ་ཆགས་པ་མཐོང་། །གྱེན་དུ་རབ་རྒྱུ་རེ་ཎ་ཛ་བཱ་ད་དྷཉྩ་ས་ཀྲྀཏྠཾ་ཛཾ་ཏརྟ་བ་བེ་ཀྵ་ཧནྟཾ། །སཱ་པི་སྤྱི་ལཱི་ཧི་མགྱོགས་པར་ཐུར་དུ་ཡང་། གཅི་བ་སྤྲོ་བྱེད་དེ་མཐོང་གསད་ཕྱིར་དེས། །བྱི་དང་ག་སྦྱར་ཤིང་ཀུན་པི་གུ་བི་ཊཾ་ག་ཡུཏྟཾ་བིཀྵེ་པ་ཏསྨི་མ་རི་ཙཱ་དི་བརྐོ། །ཏིསྟིས་ཀི་ཊས་བི་ཥཉྩ་ཀིཤྲཱི་ཏེ་པི་ལིང་། །ཕོ་བ་རིས་སོགས་སྡེ་ཚན་དེ་ལ་བླུགས། །མི་བཟད་འགྱུར་བྱེད་དྲུག་ལྡན་གང་དག་སྟེ། །དེ་ཀླུ་རེ་ར་ས་ཧིན་མུ་མོ་ཛཱི་པཾ། །པ་ལཱཎྞ་ནཱ་ཥཎྜི་ཏ་གནྡ་ལེ་ན་སཾ་ཀྲྀཏྠ་མཱ་ཏྲཿཔྲ་ལ་དེས་སྲིན་བུ་དེ་ནི་སྲོག་མ་གཏང་། །སྒོག་སྐྱའི་རྩ་བ་དག་ནི་གཏུབས་པ་ཡིས། །རེག་པ་ཙམ་ལས་རབ་ཏུ་།
