ཥི་ཏེ་དེ་བ་པདྨ་ནི། །ཨེ་ཏ་དཱ་ཀཉྩ་ས་ཛཱ་ཏ་པྲ་སཱ་ད་དྲི་གུ་ཎ་ད་གཙུག་ལག་ཁང་ཆེན་ཡང་དག་གྲུབ། །དེ་ཐོས་གུས་པ་ཉིས་འགྱུར་གྱིསི། །རབ་ཏུ་དང་བ་ཡང་དག་རཿ༑ལ། ཧེ་མ་རཏྭཱ་བ་རཱ་གཱ་རཾ་བི་ཧཱ་རཱ་མ་ཀཱ་ར་ཡ་ཏ། །བི་བྷ་བི་ར་ཐ་སྐྱེས། །རིན་ཆེན་གསེར་མཆོག་ཁང་པ་ཡི། །གཙུག་ལག་ཁང་ནི་དེ་ཡིས་བྱས། །དེ་ནས་བཅོམ་ལྡན་རཱ་ཛཱནྟི་པ་ཐི་ཏེ་ནོ་པ་ཀམྤི་ཏི། །བི་ཛཱསྟི་སྦྲི་དེ་བིསཱནྟུ་མཱ་ཡ་ཡོ་བྷ་ག་རྒྱལ་བ་ལ། །གསོལ་ཏེ་ལྷ་དང་ལྷན་ཅིག་ནི། །ལམ་དུ་དེ་ཡིས་ཉེར་བསྒྲུབས་པའི། །རྒྱལ་པོ་ལ་འོས་བཱན་ཛི་ན། །ཏ་དཱ་ག་མ་ནཱ་ཧནྟེ་ཎ་པྲ་སནྣི་བྷུ་བ་ན་ཀྲ་ཡི། །ཨ་ནཱ་ཐ་པིཎྜ་འབྱོར་པས་བྱོན། །དེ་ནི་བྱོན་པ་ལ་དགའ་ནས། །སྲིད་གསུམ་རབ་ཏུ་དང་བའི་ཚེ། །དེ་ལ་མགོན་མེད་དyསྨི་བཱ་རི་དྷཱ་རཱ་མ་པཱ་ཏ་ཡཏ། །ཏ་སྨིན་ཡ་དཱ་ན་པྲ་དེ་ཤེ་བཱ་རི་དྷཱ་ར་པ་པཱ་ཏ་ཟས་སྦྱིན་གྱིས། །ཆུ་ཡི་རྒྱུན་ནི་རབ་ཏུ་བསྒྲེངསས། །གང་གི་ཚེ་ན་ཕྱོགས་དེ་རུ། །ཆུ་ཡི་རྒྱུན་ནི་མ་བབས་སཱ༑། ཏ་དཱ་བྷ་ག་བ་ཏོ་བཱཀྵ་ད་བ་ཏཏྟཾ་ཏ་ཏཱ་ནྱ་ཏ། །ཏཾ་དྲྀབྷྱཱ་ཀོ་པ། །དེ་ཚེ་གཞན་དུ་བཅོམ་ལྡན་གྱི། །གསུང་ལས་རྒྱུན་དང་ལྡན་པར་བབསི། །དེ་མཐོང་མཚར་ནས་ཏུ་ཀཱ་ཏ་བྲྀཏྠོ་བྷིཀྵ་བྷིནྡྷ་ག་བན་པུ་རཿ། །ཨུ་པཱ་ཙ་ཐུ་ཡ་ཏཱ་མེ་ཏད་བཱ་དགེ་སློང་གིས། །ཞུས་ཤིང་བཅོམ་ལྡན་འདས་ཀྱིས་སླར། །བཀའ་སྩལ་ཆུ་འདིར་རེངས་པ་ཡི། །རྒྱུ་རི་སྟམྦྷ་སྱ་ཀཱ་ར་ཎཾ། ་། ཨ་ནེ་ན་པཱརྦ་བུདྡྷ་བྷོཉསྨིནྡྷེ་བེ་ད་མཱསྤ་དཾ། །ནི་རབ་ཏུ་མཉན་པར་གྱིས། །སྔོན་གྱི་སངས་རྒྱས་རྣམས་ལ་འདིས། །གནས་འདི་ཡི་ནི་འདི་ཉིད་དུ། །པྲ་ཏི་པཱ་དི་ཏ་མི་ཏེ་ཥ་བཱ་རི་དྷཱ་རཱ་ནྱ་ཏགྷུ་ཏཱ། །ཨ་ནེ་ནི་བ་སུ་རཱི་ས་མྱཀཾ་རབ་ཏུ་བྱིན་པར་གྱུར་ཏོ་ཞེས། །ཆུ་རྒྱུན་འདི་ནི་གཞན་དུ་ལྷུང༄༅། །སྔོན་གྱི་ཡང་དག་རྫོགས་སངས་རྒྱས། །བུདྡྷ་ཡ་བི་པ་བྱི་ནེ། ་། ཨ་ཡ་མེ་བ་བ་རཱ་རཱ་མ་པྲ་དེ་ཤཔྲ་ཏི་པཱ་དི་ཏ། །རྣམ་པར་གཟིགས་ལ་འདི་ཉིད་ཀྱིས། །མཆོག་གི་ཀུན་དགའ་ར་བ་ཡི། །ས་ཕྱོགས་འདི་ཉིད་རབ་ཏུ་བྱིན། །བུདྡྷཱ་ཡ་ཤི་ཁི་ནེ་པྲ་དཱཏ་པུ་ཎྱ་ཛརྨ་རྒྱ་ཡཾ་པུ་ན། །ཏ་ཏོ་ད་དོ་བི་ཤྭ་བྷུ་བེ་ཛི་ནཱ་རྒྱལ་གྱི་སྐྱེ་བ་དག་ལ་ཡང་། །སངས་རྒྱས་གཙུག་ཏོར་ཅན་ལ་ཕུལ། །དེ་ནས་ར་གླུའི་སྐྱེ་བ་ལ། །རྒྱལ་བ་།
