ཛི་ཏོ་ནིཔ་རི་གྲ་ཧོ། །ཨེ་ཏི་བཱ་ནཱ་ཛི་ནཱ་དཎྜ་སམྦྷ་རཱ་ཌམྦ་རི་དྭགས་རི་དྭགས་བརྟུལ་ཞུགས་འདི། །གོས་དང་དབྱུག་པའི་ཚོགས་དག་གི། །ཕུང་པོ་སྤངས་པ་རོཛྫི་ཏོ། །ཨེ་ཏ་དརྦ་མ་ནེ་ནསྨ་དནི་ན་ནི་ཡ་མཀྲྀ་ཏ། །པྲ་ཏོ་བ་འདི་དག་མཆོད། །དེ་ཡི་དོན་དུ་འདི་ཡིས་ནི། །བདག་ཅག་བལྟ་ལ་ངེས་པ་བྱས། །བརྟུལ་ཞུགས་ཡོ་ཀ་ར་ཎ་བྱ་གཱ་ཏ་ནུ་ན་མ་སཱུ་པི་བཱ་ར་ཡེཏ། །ཨི་ཏི་སནྱ་སé་ཏེ་བྲ་བྱད་བསགས་པ་ན། །འདི་དག་གིས་ཀྱང་བཟློག་གམ་སྙམ། །ཞེས་པ་བསམས་ནས་དེ་དག་གིས། །ཏོ་བསྐུ་ར་སཉྩ་ཡཾ། །ན་༔་པྲ་ཀཱི་པྱ་བཱ་རཱ་ཡཾ་ཡ་ཡུ༔ཤུདྡྷ་སྟ་དནྟི་བརྟུལ་ཞུགས་ཡོ་བྱད་བསགས་པ་ཀུན། །ཆུ་ཡི་ཀླུང་དུ་རབ་སྤངས་ནས། །དག་པས་དེ་དང་ཉེ་བར་ཀཾ༑། ཀྟ་གྲ་ཧ་བྷུ་ཛྭ་ཏེ་ཥཱ་མཱ་ཤ་ཡཱ་ནུ་ཤ་ཡོ་ཙི་ཏྭཾ། །ས་ཕྱིན། །འཛིན་པ་སྤངས་པ་དེ་དག་གིས། །ཁམས་དང་རང་བཞིན་མཁྱེན་ནས་དེའི༄༅། །བསམ་དྷཱ་ཏུཾ་པྲ་ཀྲྀ་ཏི་ཛྙཱ་ཏཱ་བི་དི་དྷེ་དྷརྨ་དེ་ཤ་ནཱཾ། །རཱ་ཛ་པུ་ཏྲས་ཨེ་བཱ་ཧཾ་ཤཱཀྱ་སྟེ་པ་བག་ལ་ཉལ་དག་ལ༄༅། །འོས་པའི་ཆོས་ནི་བསྟན་པ་མཛད། །རྒྱལ་པོའི་བུ་དེ་ང་ཉིད་དེ། །བློན་པོའི་མནྡྲ་སཱུ་ན་བ། །པུ་ནསྨཱ་གོ་པ་དེ་ཤོ་ཡ་མ་ཏྭ་སྱ་ཏཱཾ་མ་ཡཱ་ཀྲྀ་ཏཱ། །ཨི་ཏི་བུ་དག་ཤཱཀྱ་རྣམས། །ད་ལྟ་བདག་གིས་དེ་དག་ལ། །གཏོང་བ་ཉེར་བསྟན་འདི་སླར་བྱས། །ཞེས་པ་ཤཱཀྱ་ཀུ་མཱ་རབྲྀཏྟ་མེ་ཏཥ་ཐི་ཏཾ་བྷིཀྵ་ག་ཎསཱ་ཡ་ཛི་ནེ་ན། །ཨ་བི་ཤཱཀྱ་གཞོན་ནུ་རྣམས་ཀྱི་སྔོན་བྱུང་ནི། །རྒྱལ་བ་ཉིད་ཀྱིས་གསུངས་པ་དེ་དག་ངེས་བཟུང་སྟེ། །རྟེན་བྱེད་དྷཱརྱ་པ་རཱ་མ་པཱུ་ཛ་ཡཏ་ཏྭཾ་ཀཱ་རུ་ཎ་མཱ་ཤི་ཏ་བཏྠ་ལ་སྱ་ཏ་སྱ། །རྣམས་ལ་མཉེས་གཤིན་ལྡན་པར་གྱུར་པ་དེའི། །ཐུགས་རྗེ་དེ་ནི་དགེ་སློང་ཚོགས་ཀྱིས་རབ་ཏུ་མཆོད། །ཨི་ཏི་ཀྵེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཅེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ཏཱ་ཡཾ་མཱུ་ཀཱ་བངྒཱ་བ་དཱ་ན་མཎྜ་ཏྲིདྡྷཔལླ་ཝ། །།།ས་ལས་ལྐུགས་པ་འཕྱེ་བོའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་སུམ་ཅུ་རྩ་བརྒྱད་པའོ། །།།
