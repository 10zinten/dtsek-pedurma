ཏུསྟོ་པྲ་ད་གླི་ཎཾ། །ཏད་དྲྀཥྛ་ས་ཧ་སཱ་བཱལྟ་པྲ་ཏྱ་ཡེ་སྭཱ་ནུ་གེ་ཚིགས་པར་འཐོན་ནས་བསྐོར་བ་བྱས། །མི་བདག་རྗེས་འབྲང་བཅས་པ་ཡིས། །མཐོང་ནས་འཕྲལ་ལ་ཡིད་ཆེས་པེ༑། ཡཱ་ཏེ་དུརྫ་ན་མཱ་ཡི་བ་ཛ་གཛྫ་ཡ་མ་ཧཱི་ཡ་ཡོ། །མུཊོ་པཱ་ཡཾ་ཛྲ་ཥ་དི་ཀ་མ་ཐོབ། །སྐྱེ་བོ་ངན་པའི་སྒྱུ་མ་ཉིད། །ས་སྟེང་འགྲོ་ལས་རྒྱལ་བར་གྱུར། །ཁུ་ཚུར་དུ་ྋརླུང་རྡོ་ལ་པདྨ་ལཾ་ཙི་ཏྲ་མཱ་ཀཱ་ཤ་དེ་ཤེ་ཛི་ཧཱ་གྲེ་ཙ་པྲ་ཙུ་ར་ར་ཙ་ནཱ་སྲིཌི་སཾ་ཧཱ་རཱ་ལཱི་ལཱ། །ནམ་མཁའི་ཕྱོགས་ལ་རི་མོ་དང་། །ལྕེ་རྩེ་ལ་ཡང་སྤྲོ་དང་སྡུད་པའི་རོལ་པ་རབ་མང་བཀོད་པ་དང་། །ཀིཾ་བཱ་ནཱ་རྱ་ཏ་བ་ཤུ་ཤི་ཤུ་དྷི་ཡོ་མོ་ཧཱ་མཱ་ཡརྗེ་ཛཱ་ལཾ་མུནྟཾདྷུཙྪཱཀྵ་ཎ་གཞན་ཡང་ཅི་མིན་ཕྱུགས་དང་བྱིས་པའི་བློ་ལ་རྨོངས་སྒྱུ་མིག་འཕྲུལ་དག། སྐད་ཅིག་ཡོངས་འདྲིས་ཡིད་ཆེས་པ་རི་ཙི་ཏ་པྲ་ཏྱ་ཡཾ་དན་ཡནྟི། །ནིཔྲ་ཏྱ་ཡ་པ་རཾ་རཱ་ཛཱ་ཏ་ཏསོ་གཟུགས་ཅན་གཡོན་ཅན་རྣམས་ཀྱིས་སྟོན་པར་བྱེད། །དེ་ནས་བདེ་གཤེགས་བསྟན་པ་ལ། །རྒྱལ་པོ་མཆོག་ག་ཏ་དན་ནེ། །ཨཱརྱ་ཀཱ་ཏྱཱ་ཡ་ཎཱ་སཱུ་གྲེ་ཤཱདྷཱ་པཱུ་ཛཱ་མ་བཱ་རཱ་ཡཏ། །རཱ་ཛ་དྷཱ་ཏུ་ཡིད་ཆེས་བྲལ། །།འཕགས་པ་ཀཱ་ཏྱཱ་ཡ་ན་ནིི། །མདུན་དུ་དད་པས་མཆོད་པ་བཟློག། དེ་ནས་རྒྱལ་ཚྭ་ནི་ཥོདྡྷོ་ཐ་བ་ཧི་རེ་པ་ས་སཱ་ནུ་ག། །བི་ནེ་ཡ་ཀྲྀ་པ་ཡཱ་ཏ་ཏྲ་ཏསྦོ་པོས་གྲོང་ཁྱེར་བཀག། ཕྱི་རོལ་ཉིད་དེར་གདུལ་བྱ་ལ། །བརྩེ་བས་རྗེས་འབྲང་བཅས་ཏེ་གནས། །དགེ་ཤི་ལཱ་ཙ་བྷི་ཀླུ་ཎཱི། །ཏ་ཏཀཱ་དཱ་ཡཱནྟཾ་པྲ་ཀཱ་ཏྱཱ་ཡ་ནབུ་ར། །ཚིགས་པ་ཏི་ཛ་ན་སཾ་སློང་མ་ནི་རི་མ་ཡང་། །དེ་ནས་ཀཱ་ཏྱཱ་ཡ་ན་ནི། །གང་ཚེ་སྔོན་དུ་མི་ཡི་བདག། འོང་བ་མཐོང་ནས་སྐྱེ་བོ་པཱ་དྷཱ་ད་བ་མཱ་ན་བྷ་ཡཱཀྵ་ཡོ། །པྲ་ཥི་ཏཾ་པཱུཏྠ་མནཱི་བྷཾ་བྲ་ཛནྟ་ཡི། ༈། ཚོགས་པས་སྨོད་པས་འཇིགས་ནས་གཤེགས། །བློན་པོ་སྔ་མ་དག་གིས་བསྐུལ། །དེ་ནི་གཤེགས་མ་བོ་ལོ་ཀྱ་ཏཾ། །དུབྷྱཱ་མཱ་ཏྱཱ་ན་ར་པ་ཏིཾ་དཱིགྷ་བི་རཱ་བ་བོ་ཙ་ཏུ། །རཱ་ཛསྣ་པ་མཐོང་གྱུར་ནས། ་། གདུག་པའི་བློན་པོ་ཁོན་རིང་དང་། །ལྡན་པ་དག་གིས་རྒྱལ་ལ་སྨྲས། །རྒྱལ་པོ་མཾ་ག་ལ་ནི་དྷི་མུཙྩོ་ཡཾ་བི་ཤི་ཁ༔ས་ཐི། །དྲྀབྦོཉ་བྱ་བྷིཀྵ་རསྣ་བྷིནྡ་བཀྲ་ཤིས་མིན་པའི་གཏེར། །དགེ་སྦྱོང་མགོ་རེག་སྐྲ་མེད་དེ། །ལམ་དུ་དེ་རིང་བདག་ཅག་གིས། །—།
