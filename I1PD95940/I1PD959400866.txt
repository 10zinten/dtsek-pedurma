བྱ་ཀཱ་མཾ་ནིརྨ་ལ་ཀཱརྨུ་ཀཱ་ནི་ཏ་ལ་ཏཱ་ནི་རྒྱ་ཅུ་མཱ་ལམྦ་ཏེ། །ཀི་ར་ཏི་བ་པ་ན། །ངེས་པར་འདོད་པའི་གཞུ་ཡི་འཁྲི་ཤིང་དྲི་མེད་ཡོན་ཏན་མེད་པ་ཉིད་ལ་བརྟེན། །འདི་ཡི་བཞིན་ད་ན་བིསྨྲ་ཧརྟ་པེ་ཡུ་ཥ་མ་སཱུ་ལ་ལི་ཏ་སི་ཏ་མ་ཡུ་ཁོལླེ་ཁ་རེ་ཁཱ་ཀྲ་གྱི་དཀྱིལ་འཁོར་ཡིད་འོང་བསིལ་ཟེར་ཅན་དག་གིས། །བཀོད་པའི་རི་མོའི་རིམ་པས་དགའ་བའི་བདུད་རྩི་མེ་ཎ། །ཨི་ཡ་མ་ཏི་ཤ་ཡ་མི་ཏི་ནེ་ཏྲ་ཡོ༔ཕྱོགས་ཏྲ་མཱུ་ལེ་ན་བ་ཀུ་བ་ལ་རྒྱས་པར་བྱེད། །མིག་དག་རྣ་བའི་རྩ་བར་ཤིན་ཏུ་བྱམས་པར་གྱུར་པ་འདི། །ཨུཏྤལ་གསར་པ་རོལ་པའི་ཡ་ལཱི་ལོཏྟཾ་ས་ཀཱནྟིཏ་ནོ་ཏི། །ཨི་ཏི་སཾ་ཙི་ཏྱ་པ་ཤ་ནི་ར་བ་རུ་རྒྱ་ཏུ་རཾ་ག་རྒྱན་གྱིས་མཛེས་པ་རབ་ཏུ་འཛིན། །ཞེས་བསམས་དེ་ཡིས་དལ་གྱིས་ནི། །རྟ་ལས་ཡང་དག་བབས་བྱས་མཱཏ༑། ཏཱ་མུ་བཱ་ཙ་ས་མ་བྱེ་ཏེ་ཀོ་ཏུ་ཀཱ་ལོ་ཀ་ནོརྨུ་ཏཱཾ། །ཨམ་ན་སུ་ཏེ༑། དགེ་མཚན་བལྟ་འལ་མངོན་ཕྱོགས་མ། །དེ་ལ་བཏུད་དེ་རབ་སྨྲས་པ། །ལེགས་བྱས་རྒྱ་ཆེ་མི་ཀྲྀ་ཏོ་དཱ་ར་སུ་རེ་ཀཎྜ་གྲ་ཧོ་ཙི་ཏཱ། །བི་ཛ་ཡེ་མ་ཎི་མཱ་ལེ་བ་ཀཱ་ཏྭཾ་ཀུ་བ་དམན་པའི། །ལྷ་ཡི་མགྲིན་པར་གཟུང་འོས་མ། །རྣམ་རྒྱལ་ནོར་བུའི་འཕྲེང་བ་བཞིན། །ཨུཏྤལ་མིག་ལ་ཡེ་ཀླུ་ནེ། །ཨ་ཡ་མཱ་ནནྡ་སནྡོ་ཧ་ཅནྡྲ་ནཱི་ལ་ལི་ཏཱ་ཏ་བ། །ན་ཀ་སྱ་ཅན་ཁྱོད་ནི་སུ། །ཁྱོད་ཀྱི་མཛེས་སྡུག་དགའ་བ་ཡི། །ཚོགས་ནི་རྒྱས་པར་བྱེད་པ་འདིས། །སུ་ཡི་ཡིད་ནི་ཀུ་རུ་ཏེ་ཀཱནྟིཀོ་ཏུ་ཀཱ་ཀུཉྩི་ཏསྨྲ་ན། །བྲུ་ཧི་ཀ་སྱ་ཏཱ་ཡཱ་བྷཾ་ཤུཤ་རཙྩནྡཱ་དགེ་མཚན་ལ། །འགུགས་པ་ཉིད་དུ་བྱེད་མ་ཡིན། །སུ་ཡི་རིགས་བརྒྱུད་མུ་ཏིག་གི༄༅། །འཁྲི་ཤིང་རབ་ར་དཱ་ཏ་ཡཱ། །ཀཱ་མ་མུཀྟ་ལ་བྷེ་བི་ཤཱསཾ་བྷ་བེ་ན་བི་བྷུ་ཥི་ཏ། །ཏུ་འཐོབ་པ་ཉིདག། སྟོན་ཟླ་ལྟར་དཀར་ཁྱོད་འཁྲུངས་པས། །རྣམ་པར་བརྒྱན་པ་རབ་ཏུ་སྨྲོས། །་་། ཨི་ཏི་ཏེ་ནཱ་ད་རཱཎྚིཏྟཱ་ཛཱ་ཏཱ་ཏཾ་མུ་ནི་པུ་ཏྲ་ཀཾ། །།ཨ་ཛཱ་ཏ་ཀཱ་མ་བྲྀཏྟཱ་པི་སཱ་ཞེས་པ་དེ་ཡིས་གུས་པས་དྲིས། །དེ་ནི་ཐུབ་པའི་བུར་ཤེས་ནས། །འདོད་པའི་སྤྱོད་ཚུལ་མི་ཤེས་ཀྱང་། །བྷི་ལཱ་ཥཱ་ཛ་གཱ་ད་པཱ། །ཨ་ཧོ་པདྷཱ་བ་ཏཱི་ན་མ་པཱ་དོ་ཏྠ་མཱ་མངོན་པར་འདོད་ལྡན་དེ་ཡིས་སྨྲས། །རྐང་པས་པད་འཕྲེང་རབ་བསྐྱེད་པས། །བདག་ནི་པདྨ་ཅན་།
