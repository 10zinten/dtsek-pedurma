ག་ན་བྱཥ་བི་མ་ལ་བྷཱ་ཛཾ་ནིཙྩཱ་ཛཱ་ནནྡ་བྷུ་མིཌྷ་བ་ཏི་བྷ་བ་ཏ་མཉྩེ་དི་གསལ་དྲི་མ་མེད་པ་དག་གིས་བསྟེན། །ཟོལ་མེད་ཀུན་དགའི་ས་གཞི་དེ་ནི་སྲིད་པའི་མུན་པ་གཅོད་བྱེད་ཉིན་ནཱ་སཱ་དི་ན་ཤཱི། །ཨི་ཏི་ཀེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་ས་ཏྭཱ་བ་ད་ན་མོའི་དཔལ་འབྱོར་ཡིན། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་ཀལྤ་ལ་ཏཱ་ཡཱ་མཱ་དན་མུ་ཁཱ་བ་དཱ་ན་མཥྛ་ད་ཤཔལླ་བ། །།།དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་མེ་ལོང་གི་བཞིན་གྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་བཅོ་བརྒྱད་པའོ། །ནེ་དཾ་བནྡྷནོ་སུ་ཧྲིདྡྷོ་ད་ར་བ་ནེ་དམྨཱ་ཏཱ་སི་ཏཱ་བཱ་ཨ་ཀ་རོ་ཏི། །གང་ཡང་ཡེ་ཤེས་སློབ་དཔོན་དགེ་བའི་རྒྱུ། །འཁོར་བའི་ཆུ་གཏེར་ཟམ་པས་གང་བསྒྲུབས་པ། །ཡཊཾ་སཱ་རཱི་བྷོ་དྷི་སེ་ཏུཏྲི་དྷཏྟེ་ཛྙཱ་ནཱ་ཙརྱཀོ་པི་ཀ་ལྱཱ་ཎ་ཧེ་ཏུ། །ཀ་ལནྟ་འདི་ནི་གཉེན་དང་མཛའ་བཤེས་སྤུན་ཟླ་དང་། །མ་དང་ཕ་དག་གིས་ཀྱང་བྱེད་པ་མིན། །རྒྱལ་པོའི་ཁབ་ཀ་ནི་བཱ་པཱ་ཁྱེ་ར་མྱེ་བི་ཆུ་བ་ནཱ་ཐ་མེ། །བྷ་ག་བཱན་བི་ཧ་རན་བུདྷཔུ་ཀྱི་གྲོང་ཁྱེར་གྱིས། །ཀ་ལན་ཏ་ཀའི་གནས་ཞེས་པ། །འོད་མའི་ཚལ་གནས་དགའ་བར་སྔོན། །སངས་རེ་རཱ་ཛཱ་ཤྲཱི་ཧེ་པུ་རཱ། །ཀོ་ལི་ཏཾ་ཙོ་པ་ཏི་ཥྱཉྩ་དོ་པ་རི་པཱུ་ཛ་ཀོ་པུ་རཱ། །རྒྱས་བཅོམ་ལྡན་བཞུགས་པར་གྱུར། །པང་ནས་སྐྱེས་དང་ཉེ་རྒྱལ་ཏེ། །ཀུན་ཏུ་རྒྱུ་ནི་གཉིས་པོ་སྔོན། །པྲ་པརྣོ་བྷི་ཀྵ་བྷཱ་བེ་ན་ཙ་ཀཱ་ར་ཤ་མ་བྲྀ་ཏོ། །ཏ་ཏཧྥ་ཤཱ་རི་པུ་ཏྲ་སྱ་དགེ་སློང་དངོས་པོར་རབ་རྫོགས་པས། །ཞི་བ་དག་ཏུ་ཞུགས་པར་བྱས། །དེ་ནས་ཡང་དག་བསྟན་མཛད་བྷིཀྵསཾ་དེ་ཤ་ནཱི་བྱ་པཱཏ། །ཡ་ཡཱ་བཱ་ཀཱཏ་ཀྲྀ་ཏན་ཏཱ༔སོ་བྷུད་མོཀྵ་ག་ཏི་པས། །དགེ་སློང་ཤཱ་རིའི་བུ་དག་ནི། །གང་གིས་དགྲ་བཅོམ་མངོན་བྱས་ཏེ། །དེ་ནི་ཐར་པར་འགྲོ་གླུ་མ། །ཏ་སྱ་ཏཾ་མད་བྷུ་ཏཾ་པྲྀཏྟཱ་དྷ་ནེ་ཏྭཾ་སཏྠ་བྷིཀྵ་བི། །ས་དྲྭཙྪབཟོད་གྱུར། །དེ་ཡིས་རྨད་བྱུང་བསྔགས་འོས་དེ༄༅། །དགེ་སློང་ཀུན་གྱིས་མཐོང་གྱུར་ནས། །དེ་ལ་པུརྦ་བྲྀཏྟཱནཾ་ས་ཙཱ་ཏདྡྷོ་བྷཱ་ཥི་ཏ། །བཱཧྥ་ཎ་སྱཱཀྐི་མི་ཏྲ་སྐྱ་བྷཱརྱ་སྔོན་གྱི་སྤྱོད་མཐའ་དྲིས། །དེས་ཀྱང་དེ་དག་རྣམས་ལ་གསུངས། །བྲམ་ཟེ་མེ་ཡི་གྲོགས་པོ་ཡི། །ཆུང་མ་།
