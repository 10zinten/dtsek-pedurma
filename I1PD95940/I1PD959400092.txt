ཧ་ཏཱ་མ་པི། །ཀུ་ཤ་ལོནྣཱ་ལ་ནཱ་ཡི་ཙ་ཀིཝི་ཥ་ཀུ་ལི་་་་བཟོད་པར་དཀའི། །རིགས་ནི་ཉམས་པ་ཉིད་ཀྱི་སླད། །སྡིག་པས་འཁྲུགས་པའི་བློ་གྲོས་སོ། །སྡིག་པ་རྣམ་ཏཱ་མ་ཏི། །པཱ་པ་སཾ་ཀལྤ་མཱ་ཏྲེ་ཎ་ཀྱི་ཏོ་ཥི་ཏི་པ་ཏི་ཀཱ་ཎཱཏ། །པཱ་པ་ཏ་བི་བྷཾ་བྷ་ཕ་པར་བརྟགས་ཙམ་གྱིས། །ས་བདག་ས་ལ་སྐད་ཅིག་གིས། །ལྷུང་སྟེ་འབྲས་བུ་རྣམ་ལྷུང་ཞིང༌། །ལཉྩིནྣ་མཱུ་ལ་ཨི་བ་དྲུ་མ། །ཧནྟི་བི་དྷྱ་མ་ན་བྷྱཱ་སཤི་ཡཾ་ཧནྟི་མ་དོ་ད་རྩ་བ་བཅད་པའི་ལྗོན་པ་བཞིན། །མ་གོམས་པ་ཡིས་རིག་པ་འཇོམས། །རྒྱགས་པ་རྒྱས་པས་དཔལ་འབྱོར་ཡ༑། བི་དཱི་ཥསཱ་དྷུ་ཏྭཾ་ཧནྟི་ཧནྟི་ལོ་བྷ་ས་མུནྡི་ཏི། །ཨ་ཧོ་བ་་འཇོམས། །རྣམ་སྣང་གིས་ནི་ལེགས་ཉིད་འཇོམས། །ཆགས་པས་ཡང་དག་མཐོ་བ་འཇོམས། །ཀྱེ་མ་ཀྭ་ཏེ་མ་ཧོཥཆུ་ཤྲཱི་གཱ་རོ་ཧོ་མ་ཧོ་ད་ཡ། །བི་བྷ་བོདྷ་བ་མཏྟཱ་ནཾ་པ་ཡེ་དར་རྒྱས་ཆེ། །ཆེ་ཞིང་མཐོ་བའི་རྩེར་འཛེགས་པ། །འབྱོར་པས་རྒྱགས་པ་སྐྱེས་པ་རྣམས། །འཕྲལ་ཧ་སེ་བ་པ་ཏ་ཏྲ་དྷ། །ཏེ་ན་སརཱ་བི་བྷཱུརྞ་མ་པཱུ་ཛི་ཏཔུཏྤ་ཛརྨ་ནི། །ཏཏྤ་ལ་ཉིད་དུ་ནི་འོག་ཏུ་ལྟུང་། །དེ་ཡི་སྐྱེ་བ་སྔ་མ་ལ། །ཐམས་ཅད་ཟིལ་གནོན་ཞེས་བྱ་མཆོད། །ལྷ་དབང་དཱསྟ་བ་ན་རཱ་ཛཾ་ཀྲྀ་ཧ་ཎཱི་མ་རུཏྠ་ཏེ། །སུ་རཱ་དྷི་པཱ་དྷི་བཀོ་པེ་པྲ་རྒྱལ་སྲིད་བསྟེན་འོས་ནི། །དེ་ཡིས་འབྲས་བུ་ཐོབ་གྱུར་ཅིང་། །གང་ཡང་ལྷ་དབང་ལས་ལྷག་པ། །མཐུ་བྷཱ་བོ་བིསྨྲ་ཡཱ་བ་ཧ། །ཨ་ནལླསྟ་སྭཱ་བྷཱུ་ཏ་པཱ་ཏྲ་དཱ་ནཾ་ཤ་སཾ་བྷ་བ། །བནྡྷ་ནི་ཡ་མཚན་ལྡན་པ་དག། དེ་ཡི་སྣོད་ལ་སྦྱིན་པའི་ཆས། །བསྐྱེད་པ་མི་ཉུང་སྐལ་བར་གྱུར། །གཉེན་མ་ཏུ་བྷི་དྷཱ་ནཱ་ཡཱ་ན་གན་མུ་ཙི་ཏཤུ་ཙི། །བ་ནི་གུཀྐ་རི་ཀོ་ནཱ་མ་སོ་བྷ་བ་ལྡན་ཞེས་ནི་བྱ་བ་ཡི། །གྲོང་ཁྱེར་འོས་ཤིང་གཙང་བ་རུ། །ཚོང་དཔོན་ཨུད་ཀ་རི་ཀ་ཞེས། །དེ་ཡི་སྔོན་ཏུཏྤ་ཛརྨ་ནི། །བི་པ་ཤཱི་ནཱ་མ་བྷིཀྵ་ཡི་སམྱཀཾ་བུདྡྷ་ཏྭཾ་ག་ཏ། །བི་བེ་ཤ་གྱི་སྐྱེ་བར་གྱུར། །ཡང་དག་རྫོགས་པའི་སངས་རྒྱས་ནི། །ཉིད་དུ་གྱུར་ཅིང་སེམས་ཅན་ཀུན། །སྒྲོལ་བ་ལ་ཏཛཾ་ཧཾ་སཏྠ་སཏྭཱ་སཾ་ཏཱ་ར་ཎོདྷཱ་ཏ། །པཱ་ཏྲེ་ཏ་སྱ་ས་ནི་ཀྵེ་པ་་བརྩོན་རྣམ་གཟིགས་མཚན། །བསོད་སྙོམས་སླད་དུ་དེ་ཁྱིམ་ཞུགས། །དེ་ཡི་སྣོད་དུ་དང་བའི་བློས། །།
