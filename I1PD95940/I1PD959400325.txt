ས་ཏ་ཏཾ་པྲ་ཡཱནྟི། །སཾ་ཏོ་ཥ་ཤཱི་ལསྐུ་མ་ནཔྲ་སཱ་ད་ཤུདྡྷཾ་བུ་དྷོ་ཏཱ་བི་མ་ལཱི་བྷ་ཆགས་དང་ལྡན་པར་གྱུར། །ཡིད་ནི་རབ་ཏུ་དང་བའི་ཆུས་བཀྲུས་པའི། །ཆོག་ཤེས་ངང་ཚུལ་དྲི་མ་བྲལ་བནྟི༑། ཐོ་ཏཾ་ཐུ་ཏེ་ནི་བ་ན་ཀུཎྜ་ལེ་ན་དཱ་ནེན་པཱ་ཎི་རྣ་ཏུ་ཀཾ་ཀ་ཎེ་ན། །བར་འགྱུར། །སྙིང་རྗེ་རྒྱས་པ་རྣམས་ཀྱི་ལུས་མཛེས་པ། །གཞན་ཕན་གྱིས་ཡིན་ཙནྡན་བྱུགས་པས་མིན། །བི་བྷཱ་ཏི་ཀཱ་ཡཀ་རུ་ཎཱ་ཀུ་ལཱ་ནཱཾ་པ་རོ་པ་ཀཱ་རེ་ཎ་ན་ཙནྡ་ནེ་ན། །ཨེ་ཏཱ་ནི་མོ་ཧཱ་རྣ་བ་ཐོས་པས་ཡིན་གྱི་རྣ་ཆས་མིན། །ལག་པ་སྦྱིན་པས་ཡིན་གྱི་གདུབ་བུས་མིན། །ས་བདག་གིས་སྤྱད་ཧ་ཏ་བལྦ་བྷཱ་ནི་སཾ་སཀྟ་མུཀཾ་ཤུ་སི་ཏ་སྨི་ཏ་ནི། །ས་ཏཾ་ན་བྷོ་གྷ་ནི་ལྷག་མར་གྱུར་པའི་རྒྱན། །མུ་ཏིག་འོད་ཆགས་དཀར་བའི་འཛུམ་དང་བཅས། །རྨོངས་པས་དགའ་བ་མ་བྷ་བནྟི་བྷུ་བྷཾ་དུཙྪིཥྛ་ཤིཀྵ་ནི་བི་བྷུ་ཥ་ཎཱ་ནི། །རཱ་གཱ་ཏུ་རཱ་ཎི་རི་པུ་བཅོམ་འདི་དག་ནི། །དམ་པ་རྣམས་ཀྱིས་སྤྱད་པར་འོས་པ་མིན། །ཆགས་པས་གཟིར་དང་དགྲ་ཡིས་ཏཱ་པི་ཏཱ་ནཾ་ནི་དཱ་དྷ་ན་དྷཱ་ན་བ་ཏཱནྣ་ནཱ་མ། །ཤ་ཡཱ་སུ་ཁ་སྤན་བ་ཏཱི་ཥཱ་རཱ་གདུངས་གྱུར་པ། །ནོར་གྱི་བསམ་གཏན་ལྡན་པའི་རྒྱལ་པོ་ལ། །རེག་ན་བདེ་བའི་ལམ་ན་གཉིད་ཡོད་ཙཱ་སཏྠ་ཏྲ་ཤཱནྟ༔སུ་ཁ་མེ་བ་ཤེ་ཏེ། །ནིརྨོ་ཀ་ཀཱནྟེ་ན་བ་རཾ་ཤུ་ཀེ་ན་མིན༑། ཞི་བ་རྣམས་ནི་ཀུན་ཏུ་བདེ་བར་གཟིམས། །ལག་འགྲོའི་སྤྱོད་ཚུལ་སྦྲུལ་ཤུན་མཛེས་པ་ཡི། །བྷུ་ཛོ་ག་བྲྀཏྟིརྣ་ཏུ་ཙཱ་བ་རེ་ཎ། །པཱ་ཏྲ་པ་བི་ཏཱ་ཎི་ས་མཱ་པཱུ་ཧཱ་ནི་གོས་བཟང་གིས་མིན་ཆོས་གོས་དག་གིས་ཡིན༄༅། །ལྷུང་བཟེད་ཀྱིས་ནི་ཟས་དག་གཙང་བ་དང༌། །པཱི་ཡུ་ཥ་མི་ཏཱ་ཅུ་ཤ་ནཱ་ནི་སཱུ་ཏེ། །ཙྪ་ཏཱ་ཎི་བཀྵ་ཤྲཱི་ཥ་མ་པྲ་ཀཱ་ཤྲཱི་ཡོངས་རྫོགས་བདུད་རྩིའི་གྲོགས་པོ་ཉིད་དུ་བསྐྱེད། །གདུགས་ཀྱི་བཞིན་ནི་མཆོག་ཏུ་མི་གསལ་ཞིང་། །མ་ཎ་བི་ལོ་ལཾ་བྱ་ཛ་ནཱ་ནི་ལོ་གྷཱ། །སཾ་སཥ་ཛཱ་ཧཱུཾ་ཧྲི་ད་ཡཾ་ཛྲ་པཱ་ཎཾ་ཀུཊྚནྟ་ཧཱ་རཱ་རླུང་ཡབ་རླུང་གིས་ཡིད་ནི་རྣམ་པར་གཡོ། །ཧ་རི་ཙནྡན་གྱིས་བརླན་དོ་ཤལ་གྱིས། །མི་བདག་སྙིང་ལ་ཧ་རི་ཙནྡ་ནཱནྡྷ། །བི་ཡོ་ག་རོ་གཱ་ནུ་ག་ཏཱ་བི་བྷུ་ཏིཀཱནྟཱཀ་ཎནྟཱ་བི་ར་སོ་བི་བསིལ་བ་རེག་པར་བྱེད། །འབྱོར་པ་བྲལ་བའི་ནད་ཀྱི་རྗེས་སུ་འགྲོ། །རྩེ་དགའ་རོ་བྲལ་མཛེས་མ་སྐད་—།
