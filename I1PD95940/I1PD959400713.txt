ལོ་ཀ་བི་བརྫི་ཏ། །སརྦ་ས་ཏཱ་པ་རི་ཏཱ་ཎཾ་སནྣདྷཾ་ཏཱ་མུ་པཱ་ག་ཏ། །སྣང་བ་རྣམ་པར་སྤངས། །སེམས་ཅན་ཐམས་ཅད་ཡོངས་སྐྱོབ་པར། །ཞུགས་པ་ཁྱོད་ཅན་ཉེ་བར་འོངས། །ཨེ་ཏེ་ཏ་བ་པྲ་བྷཱ་བེ་ན་རོ་གི་ཎཿསམྦྷ་མ་ག་ཏ། །ཏཱཥིཏྟི་བ་ད་མུ་ཁ་རཱཉྩ་ཁྱོད་ཀྱི་མཐུ་ཡིས་ནད་པ་ནི། །འདི་དག་ལེགས་པར་གསོས་གྱུར་ནས། །འོད་ལྡན་མཛེས་པས་ཁྱོད་ཀྱི་རནྟི་རུ་ཙི་ར་ཏཱི་ཥ། །ཏཱ་ཏུ་མདྷ་སི་ཚྭ་དེ་བ་དུ་ཁ་དཱི་ནི། །གྲགས་པ་མུ་ཅོར་སྒྲོག་ཅིང་རྒྱུ། །སྡུག་བསྔལ་བཀྲེན་དང་ལོང་བའི་གཉེན། །ལྷ་ཡིས་བདག་ནི་ནཱནྟ་བཱནྟ་བ། །དཀྵི་ཎཾ་ན་ཡ་ནཾ་མ་ཧཱུཾ་དཱི་ཡ་ཏྭཾ་ཡ་དི་ཤཀྱ་ཏེ། །ཨི་ཏུཏྟསྟེ་བསྐྱབ་པར་འོས། །བདག་ལ་མིག་ནི་གཡས་པ་དག། གལ་ཏེ་ཕོད་ན་གནང་བར་མཛོད། །དེས་འདི་ན་ཚིགས་པ་ཏིཔྲ་སསྣ་བ་ད་ན་རྒྱུ་ཏི། །སནྟ་ར་ཎཱ་ཡ་ཛ་ག་ཏྭཾ་བརྗོད་ཚེ་མི་བདག་ནི། །རབ་དང་བཞིན་རས་མཛེས་པ་ཡིི། །འགྲོ་རྣམས་ཡང་དག་བསྒྲལ་བའི་སླད། །སམྱཀཾ་བོ་དྷི་སིདྷི་ཡེ། །བྲ་ཎི་དྷཱ་ནཾ་ས་མཱ་དྷཱ་ཡ་ཤ་སྡེ་ཎཏྠཥྛ་ལོ་རྫོགས་པའི་བྱང་ཆུབ་བསྒྲུབ་པའི་ཕྱིར། །སྨོན་ལམ་རབ་ཏུ་བཏབ་བྱས་ཏེ། །མིག་ནི་མཚོན་གྱིས་རབ་ཙ་ནཾ། །ད་དོ་དྷེརྨེ་ནི་དྷིསྟསྨི་པུà་རཱ་བུ་རི་ཏ༔སུ་རི། །ཏ་སྭཱ་ད་ཕྱུངས་ནས། །བརྟན་པའི་གཏེར་གྱིས་དེ་ལ་བྱིན། །ལྷ་རྣམས་ཀྱིས་ནི་མེ་ཏོག་གཏོར། །དེ་ཡི་རྨད་བྷུ་ཏ་ད་ནེ་ན་བི་སྨྲ་ཡཱ་དི་བ་སཱ་ཙ་ལཱ། །བི་ཙ་ཙཱ་ལ་ཁི་ལ་པྲ་ཤྲཱི་བི་ལོ་བྱུང་སྦྱིན་པ་ཡིས། །ཡ་མཚན་ནས་བཞིན་ས་གཞི་ནི། །ཆུ་གཏེར་གཡོ་བའི་སྐེ་རགས་ཅན། །རི་དང་ལཱཎྜ་བ་མེ་ཁ་ལཱ། །ཛཱ་ཏཱ་ལོ་ཀཾ་ཏ་མཱ་ལོ་ཀྱ་ནེ་ཏྲེ་ནི་ཀེ་ན་བྷུ་པ་ཏི། །ཨི་ཏི་བཅས་པ་མ་ལུས་འདར། །མིག་གཅིག་གིས་ནི་སྣང་བ་དག། སྐྱེས་པ་དེ་མཐོང་ས་ཡི་བདག། སྦྱིན་པའི་དཱ་ན་ར་ས་པཱི་ཏྭཱ་དཱི་ཏཱི་ཡཾ་དཱ་ཏུ་མུརླ་ཡོ། །ཏ་ཏསུ་ར་པ་ཏི་སྟ་སྱ་རོ་ལ་རབ་དགྱེས་པས། །གཉིས་པ་སྦྱིན་པར་བརྩོན་པར་གྱུར། །དེ་ནས་ལྷ་ཡི་བདག་པོ་ནི། །རང་གི་ས་མཱ་སྠཱ་ཡ་ནི་ཛཾ་བ་པུ། །སཱསྦ་པི་དྷཱ་ཡ་ན་ཡ་ནཾ་པྲ་ཤཾ་ཤཾ་སཱ་ཏི་སཏྭ་ཏྭཾ། །ལུས་སུ་ཡང་དག་གནས། །དེ་ཡི་མིག་ནི་བསོས་བྱས་ཏེ། །སྙིང་སྟོབས་མཆོག་ཉིད་རབ་ཏུ་བསྔགས། །—།
