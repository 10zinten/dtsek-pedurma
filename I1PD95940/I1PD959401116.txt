བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་ཤྲ་ཧེ། །ཨུདྨ་ཡུདྡྷཾ་ག་ལ་ལ་ནཱ་སཾ་ག་མཾ་ག་ག་གཱི་ཏ་ཡ། །ཏ་ཏསིདྡྷཱ་ཞུགས་ཚེ། །བུང་བའི་ན་ཆུང་འགྲོགས་པ་ཡིས། །བཀྲ་ཤིས་གླུ་དག་རབ་ཏུ་བསྐྲུན། །དེ་ནས་གྲུབ་པའི་དྷི་ན་ཐ་སྱ་བྷ་བ་ནེ་བྷཱུ་རི་སཾ་པ་ད། །སུ་ཏཱ་པ་རི་ཎ་ཡཱ་རཾ་བྷ་སཾ་བྷཱ་རཿས་བདག་པོ་ནི། །ཕུན་ཚོགས་མང་བའི་ཁྱིམ་དག་ཏུ། །བུ་མོ་བག་མའི་དགའ་སྟོན་དག། རྩོམ་པའི་ཆས་མ་བཏྟཏ། །ཨ་ཐ་སི་དྷ་པུ་རཉྫི་བྷི་ཇི་ཧཱུཾ་ཤུ་ཀ་བི་བྷུ་ཥ་ཎི། །དྲྀཏྟཱ་པྲ་ནི་རབ་ཏུ་བསྟར། །ཅི་སྟེ་གྲུབ་པའི་ན་ཆུང་གིས༅། །ལྷ་ཡི་གོས་དར་རྒྱན་རྣམས་ཀྱིས། །བུ་མོ་བརྒྱན་སཱ་དྷི་ཏྭཾ་ཀ་རྩྭ་མཱུ་ཙུ་ར་པྲ་ར་སོ་མི་ཐ། ་། ཧཱ་རོ་སཿཀུ་ཙ་ཡོནྟཱ་རཿཔ་རི་ཧཱ་རཱཔ་མཐོང་གྱུར་ནས། །ལྷ་མོ་རྣམས་ཀྱིས་ཕན་ཚུན་སྨྲས། །འདི་ནི་ྋནུ་མ་རབ་མཛེས་པ། །དོ་ཤལ་ཁུར་པ་ར་རུ་ཙ། །ལྭ་བ་ཅུ་བྱ་བ་དྷཱ་ནེ་ན་བྷཱུ་ཡ་སཱ་བྷུ་ཥ་ཎེ་ན་ཀིཾ། །ཏ་ཏྭཾ་གྱིས་རབ་ཏུ་བསྒྲིབས། །མཛེས་སྡུག་སྒྲིབ་པར་བྱེད་པ་ཡི། །ཤིན་ཏུ་མང་བའི་རྒྱན་གྱིས་ཅི། །གྲོགས་མོ་ས་ཁི་ཀིཾ་ཀྲི་ཏ་སྟ་ན་ཏ་ཊེ་རཏྭཱ་བ་ལཱི་བྷིནྡྷ་རའྱི་ཊཾ་མཎྜ་ན་པཎྜི་ཏཱ་སི་སུ་ལུས་ཕྲ་ནུ་མའི་ངོས་ལ་རིན་ཆེན་འཕྲེང་བའི་ཁུར་གྱིས་ཅི་ཞིག་བྱེད། །ཁྱོད་ནི་བཀྲ་བར་རྒྱན་ལ་མཁས་ཏེ་དྲྀ་ཤཀིཾ་བཱཀྵ་ནེ་ནཱ་མུ་ནཱ། །ར་རྩྭ་རལྔ་ན་དུརླ་ཧེ་ཎ་བི་ཧི་མིག་བཟང་དག་ལ་མིག་སྨན་འདིས་ཀྱང་ཅི། །བཞིན་རས་ཟླ་བའི་དཀྱིལ་འཁོར་དག་ལ་ཁྱོད་ཀྱིས་འབད་ཏཱ་བཀྑེནུ་བིཝེ་ཏྭཱ་ཡཱ་པ་བྱེ་ཡཾ་ཀུ་རུ་ཏེ་ཀ་ལཾ་ཀ་ཀ་ལ་ཎྞཾ་ཀ་སྐུ་རི་ཀཱ་མཉྫ་ནས་དགའ་བར་བསྒྲུབས་པའི་མཚོནོ། །གླ་རྩི་དག་གི་དོག་པ་རྣམས་ནི་དྲི་མ་ཆགས་པར་བྱེད་པ་འདི་ལ་རཱི༑། ས་ཁཱི་བྷི་རི་ཏི་སཾ་ཀལྤ་ཛ་ཝེ་རཱ་ཀ་ལྤ་བི་བྷཱ་མ། །ཀམྤི་ཏོ་ལྟོས། །དེ་ལྟར་ཀུན་རྟོག་བརྗོད་པ་ཡིས། །གྲོགས་མོས་རྒྱན་གྱི་རྣམ་འཕྲུལ་ནི། །བཀྲ་ཤིས་སླད་དུ་མཾ་ག་ལཱ་ཡི་བ་སཱཡྻ་པ་སཱུ་བྱ་རཱ་ཛ་ཏ། །ཨ་ཐཱ་ཡ་ཡོ་བི་མཱ་ནེ་ན་བརྟགས་གྱུར་པ། །ཤིན་ཏུ་ཆུང་བ་དག་ཀྱང་མཛེས། །དེ་ནས་ནམ་མཁའ་@གཞལ་ཡས་ཁང་། །མ་ཎི་མཱ་ལཾ་ཤུ་མཱ་ལི་ནཱ། །ཧཱ་ར་ཧམྦྷ་ཡ་མ་ནི་ན་བྱོསྣ་ཛཱི་མཱུ་ཏ་བཱ་ཧ་ན། །ནོར་ལ་འཕྲེང་བའི་ཟེར་འཕྲེང་ཅན། །དོ་ཤལ་བ་གམ་ཅན་ལྟ་བུས། །སྤྲིན་གྱི་བཞོན་པ་རབ་ཏུ་འོངས། །—།
