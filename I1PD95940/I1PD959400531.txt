ནཱ་མ་དུརྒ་ཏ། །དྷིཥཱ་བ་ཏཱི་བདཱ་རི་དྷཱཏྭཱ་ནི་ཙཱ་ཏུ་ཀུ་ཤ་ལ་ཀྲི་ཡཾ། །ནས་རབ་བསམས་པ། །གང་ཞིག་སྐལ་བ་དམན་པ་ཡིས། །དགེ་སློང་གཅིག་ཀྱང་མགྲོན་མ་གཉེར། །ནི་ཀོ་པི་མནྡ་བྷཱ་གྱེ་ན་ཡེ་ན་བྷིཀྵནི་མནྡྲི་། །ཏྭཱ་ཛཱ་ཛ་ན་སྱ་ས་ཤིན་ཏུ་དབུལ་ལས་དགེ་བ་ཡི། །བྱ་བ་བཏང་བ་བདག་འདྲ་དམན། །ངག་ཚད་ཚིག་ནི་མཚམས་སྦྱོར་ཀ་ལ་བྱ་བ་ཧཱ་ར་བཱ་ཧཱ་བཱཀྱ་པྲ་མཱ་ཎ་པ་ད་སནྟི་ཥཱ་ནི་བ་ཡོ་གྷ། །ནཥཱ་རྣམས་ལ་རིགས་པ་ཡོད་མིན་ཞིང་། །བྱ་བ་ཉམས་ཤིང་བྱེད་པ་དང་བྲལ་དེ་ཉིད་དམན་གྱུར་པ། །ཐ་སྙད་ཀྲྀ་ཡཱ་བི་ག་ཏ་ཀཱ་ར་ཀ་ཏརྒྱ་ཧཱི་ཎཤཔ་ཨི་བཱཞྭ་ར་ཧ་ཏཱཔུ་རུ་ཥ་བྷ་ཀུན་ལས་ཕྱི་རོལ་གྱུར་ཅིང་དོན་དང་བྲལ་བ་ཡི། །སྒྲ་རྣམས་བཞིན་དུ་སྐྱེས་བུ་རྣམས་ཀྱི་སྐྱེ་བོས་དོན་བྱ་བཱནྟི༑་། ཨི་ཏི་ཙིན་ན་ལཱ་ཀྲྀ་ཏཾ་ནི་དྲྀནྟནྡྷ་ན་ཧཱི་ན་ཏཱ། །ཏཾ་ས་མཱ་ཡིན། །འདི་བསམས་མེ་ཡིས་ཟིལ་གནོན་ཅིང་། །ནོར་གྱིས་དམན་ཞིང་སྨོད་བྱེད་པ། །དེ་ལ་ལེགས་དཱ་ཡ་ཀོ་པེ་བྷེ་ཏ་སུ་ཀྲྀ་ཏ་པྲ་ར་ཀོ་བྱཱ་དྷཏ། །ཀྵི་ཎཱརྦ་ནཱ་པི་བྷ་བ་ཏཱ་ཛནྣཱནྟ་བྱས་སྐུལ་བྱེད་པའི། །ལས་པས་མངོན་དུ་ཕྱོགས་ཏེ་སྨྲས། །ནོར་གྱིས་དམན་པ་ཁྱོད་ཀྱིས་ཀྱང་། །སྐྱེ་བ་ར་ཤུ་བྷཱབྷ་ཡེ། །ཡ་བྷཾ་ཀ་ཐཾ་ཙི་དེ་ཀོ་བི་བྷིཀྐུཀིཾ་ན་ནི་མཎྜི་ཏ། ་། ཨི་གཞན་དུ་དགེ་ཐོབ་སླད། །ཇི་ཞིག་ལྟར་ཡང་དགེ་སློང་ནི། །གཅིག་ཀྱང་ཅི་སླད་མགྲོན་མི་གཉེར། །ཞེས་ཏུཏྟསྟེ་ན་སཾ་སཀྟ་ཤལླཔུ་ན་རི་བཱ་ཧཱ་ཏ། །བྷིཀྵ་བྷོ་ཛན་བི་ཀ་ལྱཱཏ་ས་ཧཱུཾ་པ་དེ་ཡིས་བརྗོད་པ་ན། །ཟུག་རྔུ་ཟུག་པ་སླར་བསྣུན་བཞིན། །དགེ་སློང་ཞལ་ཟས་མ་འབྱོར་བས། །དེ་ཐཾ་བྱ་ཐི་ཏ་བྷ་བཏ། །ཀ་ཐཾ་ཙིཀྵཏྲ་རི་་མས་ག་ཏཱ་ཐེཊྛི་ནི་མཆོག་ཏུ་གདུང་བར་འགྱུར༄༅། །ཚོང་དཔོན་ཁང་པར་ཇི་ཞིག་ལྟར། །བཀྲེས་པས་ཡོངས་ཉེན་དེ་མནྡི་རཾ། །ཡནྡྷེ་ན་པྲ་པ་མཱུ་ལྕི་ཤ་དཱ་རུ་པཱ་ཊ་ན་ཀརྨ་ཎཱ། །ཀྲྀ་ཏྭཱ་ཏ་ཏྲེ་སོང་ནས། །འབད་པས་ཤིང་ནི་གཅོད་པ་ཡི། །ལས་ཀྱིས་རིན་གྱི་ཆ་ཤས་ཐོབ། །དེ་ཡི་ཆུང་མའི་སྐྱེ་བ་ཏདྡྷནཱ་ཤུདྡྷཱ་ཏཎྜ་ལ་ཁཎྜ་ནཾ། །ཏ་དཾ་ཤ་བྷཾ་ཏི་མཱུ་ལྕ་སྟོ་བྷཀྵ་བྷརྗེ་རྒྱ་དག་པའི་འབྲས། །དེ་ཉིད་དུ་ནི་གྲུབ་བྱས་ནས། །དེ་ཡིས་གླ་རྔན་ཆ་ཤས་ཐོབ། །གུས་པས་རྗེ་བོ་—།
