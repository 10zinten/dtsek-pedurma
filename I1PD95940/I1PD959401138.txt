བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁི་ཀྲྀ་ཏོ་ཙི་ཏ་མ་བྷ་དྷཱ་ཏ། །ཡ༔་གྲོ་ཧཱཛེ་ཏ་བ་་་་་་གང་ཞིག་ཨ་མྲ་སྐྱོང་མའི་སྤྱོད། །ལེགས་བྱས་ལ་འོས་ཀྱི་གསུངས་པ་དང་། །གང་ཞིག་རྒྱལ་བྱེད་ཚལ་ནཾ་ཙ་ཀྲོ་པི་ཏཱ་པ་སཾ་ག་མཾ། །པྲ་ད་དེ་པུ་ཏྲ་དཱ་རཱ་དི་ཡོ་བཟུང་དང་། །ཡབ་དང་མཇལ་བ་མཛད་པ་དང་། །ཐམས་ཅད་སྒྲོལ་གྱི་སྐྱེ་བར་གང་། །བུ་དང་བི་ཤཱནྟ་ར་ཛརྨ་ནི། །ཨ་བྷི་ནིསྒྲ་མ་ཎཾ་ཡཉྩ་ཛཱ་ཏཿཤཱཀྱ་ཀུ་ལེ་བྱཱ་དྷཱཏ། །ཆུང་མ་ལ་སོགས་བྱིན། །ཤཱཀྱའི་རིགས་སུ་འཁྲུངས་གྱུར་པ། །གང་གིས་མངོན་པར་འབྱུང་བ་མཛད། །མཱ་ར་བི་དཱ་བ་ཎཾ་ཀྲྀ་ཏྭཱ་ཤཱ་ཀྱོཏྲཏྟི་མུ་བཱ་ཙ་ཡ། །ཡ༔ཐོ་ཎ་བདུད་ནི་རྣམ་པར་བཏུལ་མཛད་ནས། །ཤཱཀྱ་འབྱུང་བར་ཅིགང་གསུངས་དང༌། །སྐྱེ་བ་གཞན་ལ་ཀོ་ཌི་བི་ཤཱ་ཎཿཤྲཱི་མཱཉྫནྣཱནྟ་རེ་བྷ་བཏ། །དྷ་ན་སཱ་ལཱ་བྷི་དྷཱ་ནཾ་ཡཤ་ཤཱ་ས་དཔལ་ལྡན་གང་། །གྲོ་བཞིན་བྱེ་བ་ཉི་ཤུར་གྱུར། །གང་ཞིག་ནོར་སྐྱོང་ཞེས་པ་ཡི། །གླང་ཆེན་དག་མ་ད་ཀུཉྫ་རམ། །ཡཀཱ་ཤི་པུནྡ་རཿཤཱཙྩཾ་ཀྵནྟི་མེ་བ་བྱ་ཙིནྟ་ཡཏ། །ནི་མྱོས་པ་བཏུལ། །གང་ཞིག་ཀཱ'ཤིའི་མཛེས་པ་ཡིས། །ཞི་སླད་བཟོད་པ་རྣམ་པར་བསམས། །སུ་བཎྞ་པཱནཱ་ནཱ་མཱ་སི་སཏྭཱ་ཧི་ཏ་ཀྲྀ་ཏ་མྲ་ག། །ཀ་ལྟ་ཎ་ཀཱ་རི་སེམས་ཅན་ཕན་བྱེད་རི་དྭགས་གང་། །གསེར་གྱི་ལོགས་ཞེས་བྱ་བར་གྱུར། །གནོད་བྱེད་ལ་ཡང་རྣམ་ཡཉྩ་བྷུ་ད་པ་ཀཱ་རེ་པྱ་བི་ཀྲི་ཡ། །ཡོ་བི་ཤྭ་ཁསཱ་ཛཱ་ཡཱ་ཡཱཔཱ་པཱ་ཙཱ་འགྱུར་མེད། །དགེ་བྱེད་དག་ཏུ་གང་གྱུར་དང་། །རང་གི་ཆུང་མའི་སྡིག་སྤྱོད་ལས། །ཁྱད་པར་ལོ་མ་རཱཏྨ་ར་ཛུཏ། །དི་དེ་ཤ་ནཱ་ག་ཡོཤཱནྟི་ཡཧྥ་ནནྡོ་བ་ནནྡ་ཡོ། །ཙཀྲ་པཱུ་གང་ཆགས་བྲལ། །དགའ་བོ་ཉེར་དགའི་ཀླུ་དག་ལ། །གང་གིས་ཞི་བ་བསྟན་པ་དང་། །བརྩེ་བའི་ཧ་སུ་དཏྟ་སྱཡཱ༔པྲ་ས་དཾ་ད་ཡཱ་ནི་དྷི། །ཡོ་བྷ་བབྷོ་ཏི་ལ༔གཏེར་ནི་གང་ཞིག་གིས། །ཁྱིམ་ལེགས་སྦྱིན་ལ་རབ་དང་མཛད། །གང་ཞིག་དཔལ་ལྡན་གདེངས་ཤྲཱི་མཱན་པུཙྪཱཔཱུརྞ་གུ་ཎསྟ་ཐཱ། །བི་ར་གྷཏ་མཱུ་ཀ་པཾ་གུ་ཏྭཾ་ཅན་གྱུར། །དེ་བཞིན་གང་པོ་ཡོན་ཏན་གང་། །ཆགས་བྲལ་དཔལ་ལས་ཕྱིར་ཕྱོགས་པས། །གང་—།
