བསྟན་འགྱུར། འཁྲི་ཤིང༌། ཁེ་ཀྱི་པ་ཀ་ཏུ་པྲ་ལཱ་པི་རཱ་བནྡྷ་ལེ་ཊིརྣ་ཏུ་རཱ་ཛ་ཙེ་ཊི། །ནི་ཧནྟི་མཱ་ནཾ་མྱོས་པའི་ཚོགས་བཅས་ཏེ། །སྨོད་རྩུབ་བརྗོད་པས་གཟུང་བ་འཚོ་མ་ཡིན། །ང་རྒྱལ་ཉམས་མྱོང་®ངོ་བི་ད་དྷཱ་ཏི་ལÉ་ཙྪི་ནཏྟི་ཤརྞ་ཎི་ཏ་ནོ་ཏི་ཏཱ་པཾ། །ཨ་ས་ཧུ་ནིཅི་ད་བི་པདྲི་ཚ་སྒྲུབ་པར་བྱེད། །བདེ་བ་འཇོམས་བྱེད་གདུང་བ་རྒྱས་པར་བྱེད། །མི་བཟོད་སྡུག་བསྔལ་རྒུད་པ་སྐྱེད་དྷཱ་ཡཱ་ན་ནཱི་ཙ་སེ་བཱ་ས་དྲི་ཤོསྟི་ཤོ་ཀ། །མུ་ཧུཆི་ཙི་ཏྱ་ཏི་ས་ནཱི་ཙ་བྱེད་པ། །དམན་པའི་བསྟན་པ་དང་མཚུངས་མྱ་ངན་མེད། །དམན་པའི་ཚིག་དེ་ཡང་ཡང་དེས་བཱ་ཀཾ་ལཱི་ནཱ་བ་མཱ་ན་བྱ་ས་ནཱཀྐི་ཏY། །ནི་ཤཱ་སྱ་ཀཱ་ལ་་པ་ཎཱ་བསམས་ནས། །སྨད་པའི་གདུང་བ་ཚ་བའི་མེས་རེག་པས། །ཤུགས་ཕྱུང་དུས་ནི་འཕང་བར་མངོན་འདོད་བྷི་ཀཾ་ག་ཤ་ནི་ར་གཱ་ལཥ་ལ་ཡརྨི་པལྕི། །མ་ནོནྣཱ༉པྲ་ཐི་ཏ་བི་པས༑། དལ་གྱིས་རྒྱུད་མང་དག་ནི་སྦེང་ཞིང་བླངས། །ཁེངས་པ་རབ་འཇོམས་དྲེགས་དང་ནོར་གྲྭ་བ་བྷཾ་ས་ཧེ་ལོ་པ་ཧ་སིནྡྷིམྨནཱ་དི་རུ་དི་ཏ་ཙ་རི་ཏིཔཱ་ཊ་ནི༔སཱ་དྷ་བཱ་ཉམས་བདེ་བླག་ཏུ་ནི་ཉེ་བར་རྒོད། །ཚུལ་ལུགས་མིན་དང་དུ་བའི་སྤྱོད་དང་འབྱེད་པར་བྱེད་དང་སྐྱོན་ཅན་དེ༑། མརྨ་སྤན་བྱ་ཐ་ན་བི་ཥ་མ་ཀླ་ཤ་ཤ་ལྟ་ཡ་མཱ་ནི་ཧཱ་སཾ་སཱ་དང༌། །གནད་དུ་རེག་པའི་གདུང་བ་མི་བཟོད་ཉོན་མོངས་ཟུག་རྔུ་ལྟར་སྤྱོད་པས། །ཀྱེ་མ་འཁོར་བ་རཎྜ་ལ་ཀ་མ་ན་ཡཱ་ནརྨ་ལཱི་ལཱཀ་རོ་ཏི། །བི་ཙ་ལ་ད་ནི་ལོ་ལོལླ༴ལླི་ད་དག་ནི་མི་བསྲུན་ཚུལ་གྱིས་སྦྱར་བའི་རྩེ་དགའ་བྱེད། །རླུང་གིས་རབ་བསྐྱོད་འཁྲི་ཤིང་རབ་ཏུ་གཡོ་བའི་ལོ་ཙ་ལ་ཙི་ཙ་ལཉྫི་ར་ཏ་ར་མ་ཧོཏྟཱ་ཧཾ་པུཾ་སུཾ་ཀ་རོ་ཏི་བྷ་བ་བྷ་མ། །ཨ་འདབ་རྩེ་ལྟར་མི་བརྟན། །སྲིད་པ་འཁོར་བས་ཀྱེ་མ་སྐྱེས་བུ་རྣམས་ལ་རྨོངས་པ་རབ་བརྟན་བྱེད། །ཆུས་བྷི་ཀ་ཏ་ར་ལཱཙྪཱ་ཏཱ་པེ་ཏཱ་ཛ་ལཱ་བི་ལ་ས་མི་ལ་ན་གྷ་ན་བ་ན་པོ་གང་སྐྱེ་བོའི་སྤྲིན་ཚོགས་ཡང་དག་འདུས་པས་རབ་བརྩམས་གློག་གི་ནི། །རྩེ་དགའི་ཉམས་ལྡན་དཔལ་འབྱོར་བྱངྒི་དྡྷདྲི་ལཱ་ས་ར་ས་ཤི་ཡ། །བི་བྷ་བ་བི་ར་ཧ་གླེ་ཤ་ཀཱནྟ་སུ་ཁཱནྟ་མ་ཧཱ་འདི་དག་དེ་ལ་ཡང་ནི་ལྷག་པར་གཡོ། །སྐྱེས་བུ་རྣམས་ཀྱི་རྒུད་པ་མཐའ་དག་བསྲུང་བའི་རིན་ཆེན་ཚུལ་།
