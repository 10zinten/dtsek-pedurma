ཀཱ༔པྲ་སཱ་དི་ན། །པྲ་བྲ་ཛཱ་ཡོ་ག་མཱ་ས་ཛ་ཛགླུརྫ་ག་ཏི་པཱུ་ཛཱ་ཏཱ། །ཏཏྟེ་ཡོངས་སྨིན་རབ་ཏུ་དང་། །རབ་ཏུ་བྱུང་དང་ལྡན་ཐོབ་ནས། །འགྲོ་བའི་མཆོད་འོས་ཉིད་དུ་གྱུར། །དེ་ཏཱཾ་ཀུ་ཤ་ལོ་དྲྀཥྛ་ས་ཧ་སོ་ས་ན་ཏཾ་བུ་ར། །བ་བྷཱ་ཥ་བྷ་ག་བཱན་སཾཋདག་འཕྲལ་ལ་དགེ་བ་དེ། །སྔོན་དུ་ཉེ་བར་གནས་མཐོང་ནས། །དགེ་སློང་རྣམས་ཀྱིས་འདི་ཅི་ཞེས། །ཀི་མེ་ཏ་དི་ཏི་བྷིཀྵ་བྷི། །ཨེ་ཏེརྨ་མཱ་ཡཾ་སམྲཛྙཱསརྦ་རཀྵ་དྲིས་ཤིང་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས། །འདི་དག་བདག་དང་འབྲེལ་པ་འདི། །དོན་མཐུན་བསྲུངས་ཎ་ནིཋ་ཡི། །དཱི་པ་ཡཱ་ཏཱ་ནུ་ག་ཏ་སྱ་སཱི་ད་རྒྱ་སྨིནྣ་པི་ཛརྨ་ནི། །ཨ་སྟི་བིསྟིཎ་ཤིང་སླུ་བ་ཡིས། །གླིང་དུ་འགྲོ་བའི་སྐྱེ་བ་ནི། །གཞན་དག་ལ་ཡང་གྲུབ་པར་གྱུར། །རྣམ་རྒྱས་སྤྲོ་བ་མཱརྒ་སྱ་སོརྒ་བརྒ་བ་དྷིབི་དྷེ། །པུ་རཱི་བཱ་རཱ་ཎ་སཱི་ནཱ་མ་ཀོ་ཤ་སྒྲུབ་བྱེད་ཀྱིས། །རབ་མཆོག་མཁས་པ་དགའ་བའི་ས། །བཱ་རཱ་ཎ་སཱི་ཞེས་པའི་གྲོང་། ་། མཐོ་རིས་སྡེ་ལོཀྵནྟ་ཧནྟ་པུ། །ཡ་སྭཱ་མ་མ་ལ་ཀལྤ་ལ་བཱ་ཧི་ཎཱི་སུ་ར་བཱ་ཧི་ནི། །ས་དཱ་ཡིས་འཚམས་གྱུར་ཡོད། །གང་ན་དྲི་མེད་ཆུ་རླབས་ནི། །འཛིན་པ་ལྷ་ཡི་རྒྱུན་འཛིན་དག། རྟག་ཏུ་ད་ཡེ་བ་ཏྲཾ་ད་ཡཾ་པྲ་སཱ་ད་ཡ་ཏི་དེ་ཧི་ནཾ། །ཨ་ཧི་སེ་བ་ས་ཏྭཾ་སེ་བྷ་ལུས་ཅན་རྣམས་ཀྱིས་སྙིང་། །བརྩེ་བ་བཞིན་དུ་དང་བར་བགྱིད། །བཟོད་པ་བཞིན་དུ་འབྱུང་པོ་ཀུན། །བི་བྱེ་བ་བི་དུ་ཥཱསྨྲ་ཏྭཾ། །གླེ་མེ་བ་སརྦ་བྷཱུ་ཏཱ་ནཾ་ཡཱ་བི་ཤྲཱི་བྷ་སུ་ཁ་སྦི་བག་ཕབ་བདེ་བར་གནས་པ་གང་། །འཚེ་བ་མེད་བཞིན་དམ་པས་བསྟེན། །རིག་པ་བཞིན་དུ་མཁས་པས་ཏི༑། བྲཧྨ་ཀལེ་ཚྭ་པེ་ཏ་སྭཱ་བི་ཀ་སཏ་ཀ་མ་ལཱ་ཐ་ཡེ། །བྲཧྥ་འདོད། །དེ་ནས་མི་བདག་ཚངས་པར་ནི། །བརྟག་བྱ་རྣམ་རྒྱས་པདྨར་བསྟེན། །ཚངས་བྱིན་དཏྟཱ་བྷེ་དྷེ་ལོ་ཀེ་ཏྲེ་ལོ་ཀྱ་མི་བ་རཀྵ་ཏི། །བྲི་ཡ་སེ་ནཱ་བྷི་དྷ་ནོ་བྷཱུཏ་ཞེས་བྱས་འཇིག་རྟེན་ནི། །འཇིག་རྟེན་གསུམ་པོ་བཞིན་དུ་བསྲུངས། །དེར་ནི་དེད་དཔོན་དགའ་བའི་སྡེ། །ཏ་ཏྲ་བི་ཤྲ་བ་ཎོ་ས་མ། །སཱརྦ་བཱ་ཧོརྦ་སརྦ་ནཾ་སྦྲ་ན་མདྡྷི་རི་དཾ་བྷ་སཾ། །ཞེས་པ་རྣམ་ཐོས་བུ་ལྟ་བུ། །ཆུ་རྣམས་དག་གི་ཆུ་གཏེར་བཞིན། །ནོར་གྱི་དོན་མཐུན་གནས་སུ་གྱུར། །།
