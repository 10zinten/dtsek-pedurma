ནསྣ་ཀཱ་ཏ་ཡ་ནས་དཱ། །སྭ་ཧྥོ་ཎ་ཀོ་ཊཱི་ཀཎ་ཙཱ་གཙྪ་དེ་ཤོ་རྱ་ཛ་ཐ་མཾ། །བརཱ་ཏྱཱ་ཡ་ན་རྟག་ཏུ་མཆོད། །གྲོ་བཞིན་བྱེ་བའི་རྣ་བ་ཁྱོད། །རང་ཡུལ་དུ་སོང་ངལ་བ་ཐོང་། །ཁྱོད་ནི་དྭཱ་ད་ཤ་སཾ་པཱུརྞསཱགྲོ་ཧཱ་ནིརྒ་ཏ་སྱ་ཏེ། །ཨི་ཏྱཱཀྟ་སཱ་མཱ་དི་བྱ་ཏ་རང་ཡུལ་ནས་འཐོན་ནས། །ལོ་ནི་བཅུ་གཉིས་ཡོངས་སུ་རྫོགས། །དེ་སྐད་བརྗོད་ནས་དེ་ཡིས་ནི། །ཡི་སྱ་པྲ་ཏ་ཙ་ཏུཥཱ་ཡཾ། །སུཔ་སྤྱི་བ་མཱུ་ཧཱུནྟེ་ཤཱསྟི་མ་ཀཱ་ར་ཡ་ཏ། །ཨུཏྠི་དྭགས་བཞི་པོ་ལ་བསྒོས་ཏེ། །ཉལ་བ་ཉིད་ན་ཡུད་ཙམ་གྱིས། །དེ་ནི་རང་ཡུལ་ཕྱིན་པར་བྱས། །དེ་ཡང་ཏསོ་པི་ས་ཧ་སཱ་སཱ་དེ་ཤོདྨ་ན་ཀཱ་ན་ནཱཏ། །བི་ཡོ་ག་ཤོ་ཀཱཏ་པི་ཏ་རོ་ཤུ་ཤཱ་འཕྲལ་ལ་རང་ཡུལ་གྱི། །སྐྱེད་མོས་ཚལ་གྱི་ནགས་ལས་ལངས། །ཕ་མ་བྲལ་བའི་མྱ་ངན་གྱིས། །ལོང་ལ་བན་དྷ་མུ་བ་ག་ཏོ། །བྷིཀྵ་དཱི་ཛཱ་ཏི་ཐི་ག་ཎེ་པཱུ་ཛུ་མཱ་ནེ་སུ་ར་ལ་ཡེ། །ཉེ་བར་གྱུར་པ་ཐོས། །དགེ་སློང་གཉིས་སྐྱེས་མགྲོན་པོའི་ཚོགས། །ལྷ་ཡི་གནས་རྣམས་མཆོད་བྱེད་པ། །སྭཱ་ཀཾ་པི་ཏྲ་དི་ནཾ་ཛཾབྷཱ་པ་རཾ་བིསྨྲ་ཡ་མཱ་ཡ་ཡོ། །ནིཉྩ་ཏྱ་སརྦ་བྷཱ་བཱ་ནཾ་རང་གི་ཕ་ཡི་ཁྱིམ་མཐོང་ནས། །མཆོག་ཏུ་ཡ་མཚན་དག་ཏུ་གྱུར། །སྐད་ཅིག་ཉིད་ཕྱིར་དངོས་པོ་ཀུན། །ཥ་ཎི་ཀ་ཏྭཱ་ད་ནི་རྱ་ཏྭཾ། །སྣེ་ཧ་རཱ་གི་ས་མུཏྠཾ་ཙུ་ཏ་དཱསྦས་མ་ཙིཏྟ་མི་རྟག་ཉིད་དུ་ངེས་བྱས་ནས། །མཛའ་དང་ཆགས་པ་ཀུན་བཏང་སྟེ། །དེར་ནི་གནས་པས་ཡང་དག་ཡཏ༑། ཨ་ཧོ་སཾ་མོ་ན་ནི་དྲེ་ཡཾ་ནིརྦ་ཀཱ་ར་མ་དི་བཱ་ནི་ཧཾ། །སཱསྣ་མཱ་ཡཱ་ཁཾ་བསམས། །ཀྱེ་མ་ཉིན་མཚན་མཐའ་མེད་པར། །ཀུན་ཏུ་རྨོངས་པའི་གཉིད་འདི་ནི། །རྨི་ལམ་སྒྱུ་མའི་ལཱ་སི་ཏི་ཀ་རོ་ཏྱདྷཱ་ཏ་བི་བྷ་མ། །ཛརྨ་བནྡྲ་པྲ་པཱ་མ་ཏཱ་བི་ཏཱ་བཱི་རོལ་རྩེད་ཀྱིས། །རྨད་དུ་བྱུང་བའི་རྣམ་འཕྲུལ་བྱེད། །སྐྱེ་བའི་ལམ་རབ་སྟེར་བའི་མ། །ས་བོན་འདེབས་ཛ་ཤ་གྱ་ཏ་ཁ་ག། །བནྡྷ་པཱུ་ཛཱ་ས་ནཾ་ཀཱ་ཡཀོ་ཡཾ་ནི་ཡ་མ་སཾ་ག་མ། །ནུས་བྱ་འདྲའི་ཡིཕ། །མགྲོན་པོའི་མཆོད་དང་སྟན་ལྟའི་ལུས། །འདི་ལ་ངེས་པའི་ཚོགས་ནི་གང་། །ཤི་ཡསཾ་སཱ་རཱ་གྲྭ་བྷ་མ་པ་རི་ཙི་ཏཱ་ཀཾ་ཙ་ན་རུ་ཙི་ཥ་དིགྲྭ་དཔལ་འབྱོར་འཁོར་བའི་སྤྲིན་རུམ་འཁྱམས་པར་ཡོངས་འདྲིས་གསེར་གྱི་སྲེད་པའི་འོད། །རེ་བ་ཕྱོགས་—།
