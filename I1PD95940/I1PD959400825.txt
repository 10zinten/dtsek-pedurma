ནཱ་དྷ་ར་པལྤ་བ་སྟུ། །པྲ་ཏུགྷ་ཕ་ལ་བི་དྲུ་མཱ་གྷོ་བི་བེ་བྱ་ཤོ་དྷཾ་བྷ་འཛུམ་པ་མཆུ་ཡི་ཡལ་འདབ་ལ། །མུ་ཏིག་འབྲས་བུ་རབ་སྤྲས་བྱུ་རུའི་འོད། །མཛེས་མ་ཚེའགོད་ཅིང་ག་བཱན་བ་བྷཱ་ཥ། །ཡ་ཤོ་དྷ་རཱ་བྱེ་བ་བི་ཀཱ་རི་ཡུཏྟཾ་བི་བྷནྟི་ནི་ཏ་མ་བྷི་བཅོམ་ལྡན་གྱིས་གསུངས་པ། །མངོན་འདོད་རྩེ་དགའ་རྣམ་འགྱུར་ལྡན་པ་འདི། །གྲགས་འཛིན་མ་ནི་དིང་ལཱ་ཥ་ལཱི་ལཾ། །ཛནྣཱན་རེ་པི་སྨ་ར་བི་བྷ་མིརྨེ་སཱ་མོ་ད་ཀིརློ་བྷ་བན་མེ་བ་ཉིད་འཛིན་པ་མིན། །སྐྱེ་གཞན་ལ་ཡང་འདོད་པའི་རྣམ་འཕྲུལ་གྱིས། །དེ་ཡི་ལ་དུས་བདག་ལ་ཆགས་པར་ཙཀྲེ༑། ཀཱ་བྱ་པཔུ་རཱ་ཀཱ་ཤི་པུ་རི་ན་རེནྡཱཉྩནྡཱ་བ་དཱ་ཏ་དྷུ་ཏི་ཀཱིརྟི་རཱ་སཱིཏ། །ཨ་བྱས། །ཀ་ཤིའི་གྲོང་དུ་མི་དབང་ཀཱ་བྱ་སྔོན། །ཟླ་བའི་འོད་ལྟར་དཀར་བའི་གྲགས་པ་ཅན། །དགྲ་སུཾ་ཀུ་ཤཤ་ཏྲི་མ་ད་དྲི་པཱ་ནཾ་མ་ཏཱིཥྛ་དྲྀཏྟཔ་ར་ལ་སྭ་བྷཱ་བ། །པུ་ཏྲེཟི་མྱོས་གཉིས་འཐུང་ལྕགས་ཀྱུ་ཡིན་མོད་ཀྱང་། །སྤྱོད་ཚུལ་མི་རྣོ་རང་བཞིན་དྲང་པོར་གྱུར། །དཀའ་ཐུབ་ན་སྟ་སྱ་ཏ་པ༔པྲ་ཡཌི་རེ་ཀི་བ་ཀ་རྩྭ་ན་ལི་ནཱི་བ་བྷུ་བ། །པྲ་ཡ་པྲ་ཛཱ་པཱ་འབད་པས་བུ་དོན་གཉེར་དེ་ལ། །བུ་མོ་པདྨ་ལྡན་ཉིད་གཅིག་པུ་བྱུང༌། །སྐྱེ་རྒུ་སྐྱོང་ལ་གཡེལ་བ་དང་ལ་ན་སཱ་བ་ལེ་པཱསཱཡཱ་ནཱ་ཡ་ཨེ་བ་བྷ་བནྟི་ཤུ་པཱ། །ཨནྟཔུ་རེ་ཏཱ་པ་རི་ཝཌྷ་མཱ་བཅས་པའི། །ས་བདག་ཕལ་ཆེར་རིགས་བརྒྱུད་ཉུང་བར་འགྱུར༄༅། །ཕོ་བྲང་འཁོར་དུ་དེ་ནི་ཡོངས་ནཾ་ཙིཏྟ་ཙ་ཙིནྟཾ་ཚིགས་པ་ཏི་བི་ཙཱར། །ཨཱ་ཧུ་ཡ་བི༴ཏྲ་མུ་ཁཱ་བསྐྱེད་ཅིང་། །སེམས་འབད་བསམས་པའི་མི་བདག་གིས་སྤྱད་ནས། །མཁས་པ་ལ་སོགས་བློན་པོ་བོས་ན་མཱ་ཏཱན་པྲ་ཛཱ་ག་ར་གསྟ་ར་ཏིརྫ་ག་ད། །བི་བྷིཥྛ་ཤཱ་ཁཾ་སྦི་ར་བདྷཱ་མཱུ་ནས་ནི། །གཉིད་བྲལ་དགའ་བ་ཉམས་པས་རབ་སྨྲས་པ། །ལོ་འདབ་རྒྱས་ཤིང་རྩ་བ་བརྟན་པར་ཟུག། ལ་མཏུནྣ་ཏྲ་སཏྠ་ཛ་ནོ་པ་ཛཱི་བྱ། །ཨ་ཐེ་མི་ཏུ་རྩ་པྲ་ས་བཱ་བ་ཧཱི་ནཾ་གྷ་ཎ་ཀ་ཏཾ་སྐྱེ་བོ་ཀུན་གྱི་ཉེར་འཚོ་ཤིན་ཏུ་མཐོ། །མཚུངས་པར་སྐྱེ་བས་དམན་པའི་རྒྱལ་སྲིད་ནི། །སྲིན་བུས་ཟོས་བྲྀཀྵ་མི་བཱ་དྷི་བ་ཧཱུྃ། །ཨེ་ཀི་བ་ཀ་རྩྭ་ན་ལི་ནཱི་མ་མཱསྟི་བ་ཡཔྲ་དཱ་ནོ་ཙི་ཏ་པའི་ཤིང་བཞིན་བདག་གིས་རིག། ལང་ཚོ་རབ་ཏུ་སྦྱིན་འོས་བསྟེན་པ་ཡིསི། །བུ་མོ་པད་ལྡན་གཅིག་—།
