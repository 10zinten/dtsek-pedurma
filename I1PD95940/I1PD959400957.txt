ཏ༈་། །ད་དན་ཏཱི་ཏྭཾ་རཱ་ཤི་བྷ་ག་བནྟཾ་ཏ་ཐཱ་ག་ཏ། ?། ཕུལླ་པདྨ་ནས༑»། འོད་ཟེར་ཕུང་པོ་བཅོམ་ལྡན་འདས། །དེ་བཞིན་གཤེགས་པ་མཐོང་གྱུར་ཏེ། །།པདྨ་འདབ་མ་པ་ལཱ་ཧཱུཾ་དེ་བྱཱ་ལག་ཎ་ལའི་ཏཾ། ཊ། ལཱ་བ་ཅུ་ལ་ལི་ཏཱ་ཀཱ་རཾ་ཧེ་མ་རྒྱས་པའི་སྤྱན། —། མཆོག་གི་མཚན་ཉིད་དག་གིས་མཚན། །།མཛེས་ཤིང་ཡིད་འོང་རྣམ་པ་ཅན། །གསེར་ཏཱ་ལ་མི་བསྣ་ཏཾ། །ལིམྨནྟ་མ་མྲྀ་ཏནེ་བ་ཏཾ་དྲིཥྛ་ཧན་ནིབྷ་ར། །མཱ་ལཱ་གྱི་ཏ་ལ་བཞིན་དུ་མཐོ། །བདུད་རྩིས་བྱུགས་པ་ལྟ་བུ་སྟེ། །མཐོང་ནས་དགའ་བ་རབ་རྒྱས་པ། །དེ་ཡི་སྤྱི་མི་བཱ་ད་དེ་མཱུ་ཛཱི་པ་ཏཊཱི་ད་ན་ཁ་བྱུ་ཏི། །པྲ་ཎ་ཡཱརྦ་ན་ཡཱ་ཏ་སྱ་པཱི་བོར་འཕྲེང་བ་བཞིན། །དེ་ཡི་ཞབས་སེན་འོད་ཟེར་བླངས། །འབད་པས་གསོལ་བ་བཏབ་པ་ཡིས། །ཀྱི་པྲ་ཎཱ་ཡི་བཏྠ་ལ། ཟ་། ག་ཏཱ་ཙ་ཀཱ་ར་བྷ་ག་བཱན་གྱི་ཧེ་བྷོ་གེ་པྲ་ཏི་གུས་ལ་མཉེས་གཤིན་བཅོམ་ལྡན་འདས། ་། ་དེ་སླད་དེ་ཡི་ཁང་པར་བྱོན། །མཆོད་སྟོན་རབ་ཏུ་བཞེས་གྲ་ཧི། །་། ཨ་བྱཱཙཱི་ཏེ་བྷ་ག་བ་ཏི་པྲ་ཡཱ་ཏེ་ཛེ་ཏ་ཀཱ་ན་ནཾ། ཊ་། ཛ་ན་པ་མཛད། ཌ། བཅོམ་ལྡན་འདས་ནི་མངོན་མཆོད་ནས། །རྒྱལ་བྱེད་ཚལ་དུ་གཤེགས་པའི་ཚེ། །སྐྱེ་བོའི་ཏྭཾ་རཏྣ་སཾ་པུརྞ་ཙ་ཀཱ་ར་མ་དྷ་ར་སྭ་ར། ་། ནི་ཛེ་པུ་སྦྱིན་སཱ་ནཱ་རཏྭཱ་རཱ་ཚོགས་ནི་རིན་ཆེན་གྱིས། །སྦྲང་རྩིའི་དབྱངས་ཀྱིས་ཡོངས་རྫོགས་བྱས། །དབུལ་པོ་རྣམས་ཀྱི་ཨཁྱིམ་ཁྱིམ་དུ། །ཤིཀྲྀ་ཧོ་ཤྲ་ཧེ། །ཏདྲི་ཏིཎྞཀྵ་ཎེ་ནི་བ་ཛ་གཱ་ནཱཾ་གཱ་ར་རཱ་ཤི་ཏྭཾ། །ཡི་། རང་གི་བསོད་ནམས་མ་ཡིན་པས། །རིན་ཆེན་ཕུང་པོ་དེས་བྱིན་པ། །སྐད་ཅིག་ཉིད་ཀྱིས་སོལ་ཕུང་གྱུར། །།ཏབྲྀཏྟཾ་ཏ་མ་ཐཱ་ཀམྦྷོ་དུཁ་ཏོ་མ་དྷ་ར་སྭཱ་རཿ། །ནི། ཏཱ་ནུ་བ་ཙ་པུ་རཱ་ནི་པ་དེ་ནས་བྱུང་ཚུལ་དེ་ཐོས་ནས། །སྦྲང་རྩིའི་དབྱངས་ནི་སྡུག་བསྔལ་ཞིང་། །དེ་དག་ལ་སྨྲས་ཁྱོད་ཀྱིས་སྔོན། །བྷ་བདྷིསུ་ཀྲྀ་ཏཾ་ཀྲྀ་ཏཾ། །ཨ་ད་ཏཱ་ད་ཡ་ཡཱ་དཱ་ན་མ་ཀྲྀ་ཏྭཱ་སཾ་གྷ་པཱུ་ཛ་ནོ། །ལེགས་བྱས་དག་ནི་ཡོངས་མ་བྱས། །བརྩེ་བས་སྦྱིན་པ་མ་བྱིན་ཞིང༌། །དགེ་འདུན་མཆོད་པ་མ་བྱས་ལ། །བྷ་ག་བན་མ་ན་བྱ་ཙྩ་ན་ལ་བྱནྟེ་བི་བྷུ་ཏ་ཡ། ཏ། སུ་ག་ཏ་པྲ་བཅོམ་ལྡན་མངོན་པར་མ་མཆོད་པས། །འབྱོར་པ་འདི་དག་ཐོ་ཁྱིམ་ཡིན། །དེ་སླད་ཁྱོད་ཀྱིས་བདེ་།
