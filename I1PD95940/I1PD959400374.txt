པ་ཤ་ཀྲ་དསྟིས་སཾ་སྟ་རཾ། །ཏ་ཏྲི་པ་བི་ཤུ་པརྨ་ཀཾ་བདྷཱ་ནིཉྩ་ལ་ནི་རྩེ་མོ་གཡས་བསྟན་མལ་སྟན་བྱས། །དེར་ནི་སྐྱིལ་ཀྲུང་བཅས་མཛད་ནས། །གཡོ་བ་མེད་པར་ངེས་པར་ཡ༑། མནྣཱ་བ་སཱ་ན་བི་ཤཱནྟས་སུ་དྷཱབྷི་རི་བ་བ་བྷོ། །ཨ་ཞུགསི། །བདུད་རྩིའི་ཆུ་གཏེར་བསྲུབས་པ་ཡི། །མཇུག་ཏུ་ངལ་གསོ་བཞིན་དུ་མཛེས། །བརྟན་སཱ་དྷཱ་རཀྟ་མཱ་དྷ་རས་དྷི་ར་ས་ར་ལཱ་ཀྲྀ་ཏི། །རུ་རུ་ཙེ་ཀཱཉྩ་ན་རུ་ཙིཞིང་དྲང་པོའི་དབྱིབས་ལྡན་དེ། །ཐུན་མོང་མིན་པར་བཟོད་པའི་རྟེན། །མི་གཡོ་གསེར་གྱི་འོད་ལྡན་པ། །ས་རོ་མེ་རུ་རི་བཱ་ཙ་ལ། །ཨཱ་སྭ་བཀྟ་ཡ་པརླན་པÉརྒོ་ཡམྨ་མ་སྦི་རཿ། །ལྷུན་པོ་གཞན་པ་བཞིན་དུ་མཛེས། །ཟག་པ་ཟད་པ་མཐར་ཐུག་པ། །སྐྱིལ་ཀྲུང་འདི་ལ་བདག་གནས་སོ། །བ་བནྟེ་ཏི་ས་སཾ་ཀཝཾ་ཀྲྀ་ཏྭཱ་པྲ་ཏི་མུ་ཏཱཾ་སྨྲ་ཏི། །ཨ་ཏཱནྟ་རེ་ས་མ་བྱེ་ཏེ་མཱ་ཞེས་པའི་བརྟག་པ་བྱས་ནས་དེས། །དྲན་པའི་སྒོ་ནི་རབ་ཏུ་བཅིངས། །དེ་ནས་སྐབས་དེར་བདུད་དག་ནི། །རསཾ་ཡ་མ་མཏྠ་ར། །ལེ་ཁ་ཧཱ་རསྟནྣཱསྱཱཙྪཱ་བོ་དྷི་སཏཱ་མ་བྷཱ་ཥ་ཞི་ལ་ཕྲག་དོག་སྤྲིང་@ཡིག་ཐོགས། །མྱུར་བ་ཉིད་དུ་མངོན་པར་འོངས། །བྱང་ཆུབ་སེམས་དཔའ་ལ་སྨྲས་ཏ༑། ཨ་ཀཱ་མ་ཀཱ་མ་ཧཱ་ཀེ་ཡཾ་ལོ་ཀེ་བནྡྷཱ་བ་དཱ་མ་ཏཱ། །ཨ་ཀཱ་ལ་ཀ་ལི་པ། །འཇིག་རྟེན་གཉེན་ལ་གཡོན་ཉིད་དང་། །འདོད་མིན་འདོད་པ་འདི་ནི་ཅི། །དུས་མིན་མེ་ཏོག་ཀཱ་ཀཱ་ར་མ་ཏིསྟེ་ཀཱ་སྱཱ་བཱ་མ་ནཱ། །ཐཾ་ཧཱི་ཏཾ་ཧ་ཏ་སཾ་ཀེ་ན་དེ་བ་གཞོན་ནུའི་དབྱིབས། །ཁྱོད་ཀྱི་བློ་ཡང་སུ་ཞིག་འདོད། །དོགས་པ་བཅོམ་པའི་ལྷས་བྱིན་གྱིས། །ཁྱོད་ཀྱི་དཏྟ་ཏ་བཔུ་རམ། །ནི་རུདྡྷཱནྟ༔པུ་ར་ཐེ་ཎི་བདྷཤུདྡྷོ་ད་ནོ་ཚྭ་པ། །གྲོང་ཁྱེར་རབ་ཏུ་བླངས། །བཙུན་མོའི་ཚོགས་ཀྱིས་ཕྲེང་བ་བཀག། མི་ཡི་བདག་པོ་ཟས་གཙང་བཅིངས། །ཨི་ཏི་ཐུ་ཏྲེ་བ་བ་ཙ་ནཾ་ཤོ་ཀ་མན་བི་ཥོཙྪཱཏ། །ཨ་ཤི་ཀཱི་ཏ་བི་ཀཱ་རེ་ཎཞེས་པའི་ཚིག་ནི་ཐོས་ཉིད་ན། །མྱ་ངན་ཁྲོ་བའི་དུག་སྤངས་པ། །དེ་ཡི་རྣམ་འགྱུར་མ་བསླབས་པའི། །ཙེ་ཏ་ས་བྱ་ཙིནྟ་ཡཏ། །།ཨ་ཧོ་བ་ཏཱན་ར་ཡཾ་མེ་མཱ་ར༔ཀརྟུ་ས་མུདྨ་སེམས་ཀྱིས་རྣམ་པར་བསམས་པར་གྱུར། །ཀྱེ་མ་བདག་ལ་བདུད་དག་ནི། །བར་ཆད་བྱ་བར་རབ་ཏུ་—།
