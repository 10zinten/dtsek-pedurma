དྨ་ནརྟ་བཱ་དི་ཏེ། །ཨ་ཐ་ཏེ་ན་པ་ཐཱ་ཀཱ་པི་གོ་པཱ་ཀཱནྟཱགྷ་ན་སྟ་ནཱི། །མ་ཐི་ཏཾ་རོལ་མོས་རྣམ་པར་རྩེ། །དེ་ནས་ལམ་དེར་བ་རྫི་ཡི། །མཛེས་མ་སུ་ཞིག་ནུ་རྒྱས་ཤིང་། །རི་དྭགས་མིག་པ་ཅུ་མཱ་དཱ་ཡ་ཧ་རི་ཎཱ་ས་མཱ་དཱ་ཡོ། །གརྡ་བྷ་རཱ་ལ་ས་ག་ཏིཔྲ་ཏུཏུ་ལྡན་དར་བ་དག། ཁྱེར་ནས་འཚོང་དུ་ཡང་དག་འོངས། །རང་བཞིན་གྱིས་ནི་གླང་པོའི་འགྲོས། །མངལ་ག་ཛ་ག་མི་ནི། །པ་ཤ་ནི་རཱུ་པ་སརྦནྟཱི་སམྦི་ཧཾ་ཏཱ་ན་བྱ་ལོ་ཀ་ཡཏ། །ཁུར་ལེ་ལོར་བགྲོད་ལྡན་མ། །དེ་ཡི་དལ་བུས་ཉེར་བགྲོད་ཅིང་། །ཆགས་བཅས་དེ་དག་རྣམས་ལ་བལྟས། །ཏ་སྣ་བ་ན་ཚིགས་གི་མུབྷི་རཱ་བ་དབྷིརྨི་ལོ་ཀ་ནི། །ཨ་སཾ་བྲྀཏྟཱ་ལཱ་ཥཱདྲི་ནགས་ཀྱི་རི་དྭགས་ལྟར་མཛེས་མ། །ཡིད་འོང་མ་དེའི་ལྟ་ཚུལ་ནི། །མངོན་འདོད་མ་བསྒྲིབས་བརླན་པ་པི་སོཥཥྛ་ཏྭཾ་ཡ་ཡུ། །པཱ་ཏིནི་མནྟི་ཏཱ་ཏྲ་མ་ད་ན་་བ་དེས། །དེ་དག་རྣམས་ཀྱང་སྲེད་ལྡན་གྱུར། །འདོད་པས་མྱོས་པ་ཉིད་གྱུར་ཅིང་། །ངང་ཚུལ་འཕྲལ་ལ་ཏྭཾ་ག་ཏཱ། །ཧཱ་རི་ཏྭཾ་ས་ཧཱ་སཱ་ཤཱི་ལཾ་ན་བི་བེ་ད་པྲ་མ་དི་ནི། །ཏ་ཏསྟེ་ཉམས་པ་དག། མ་རིག་བག་མེད་པ་དེ་ནི། །དེ་དག་རྣམས་ཀྱིས་དེར་དོན་གཉེར། །དེ་ནས་དེ་དག་ཥཱ་པྲ་ཡཱ་ཏེ་ཥཱ་ཏ་སཱུ་ར་ཏི་ཤ་མ་ཏ། །པ་པཱ་ཏ་ས་ཧ་དྷིམེ་ཎ་གཏྣ་ཀོ་རབ་སོང་ཚེ། །ཆགས་པས་ངལ་ནས་དེ་ཡི་མངལ། །ཁྲོས་པས་དམར་པོར་གྱུར་པ་བཞིན། །བརྟན་པ་པཱ་དི་བཱ་རུ་ཎ། །ཨ་ཏཱནྟ་རེ་ས་མཱ་ཡཱ་ཏཾ་ཏཁྲུ་ཙིསྟེ་ན་བནྡྷ་ནཱ། །པྲ་དང་ནི་ལྷན་ཅིག་ལྷུང་། །སྐབས་དེར་དེ་ཡི་བསོད་ནམས་ཀྱིས། །ལམ་དེ་ནས་ནི་རང་སངས་རྒྱས། །ལུས་ཏེ་ཀ་བུདྡྷཾ་སཱ་དཱཀྐི་ཏ་ཀཱ་ཡ་ཙིཏྟ་པྲ་སཱ་ད་ནཾ། །སཱསྨི་མ་ཐི་ཏ་མཱུ་ལྕསྟ་དང་སེམས་ནི་རབ་དང་བ། །ལེགས་པར་འོང་བ་དེ་ཡིས་®མཐོང་། །དར་བའི་རིན་གྱིས་ཐོབ་པ་ཡི། །མ་ཝཱ་ཎཾ་ཤ་ཏ་པཉྩ་ཀཾ། །དྷུ་རཱཏ་པྲ་ཎཱ་མ་བི་ན་ཏཱ་མ་ན་སེ་བ་རྒྱ་བེ་ད་ཡཏ། །ཨ་མ་ལྔ་བརྒྱ་དེ་ཡིས་ནི། །རིང་ནས་ཕྱག་འཚལ་རྣམ་ཅིབཏུད་ནས། །ཡིད་ཉིད་ཀྱིས་ནི་དེ་ལ་ཕུལ། །ཏ་ཏྲ་པུ་ཙྪཱ་རིདྡྷི་མ་ཏྱ་སྨིན་ཛཱ་ཏ་ཡཀྵ་ཀུ་ལེ་དྷུ་ནཱ། །ཛཱ་ཏ་མཱ་མཱར་ཎེ་ནཱ་སྱ་པུ་བསོད་ནམས་དེ་ལས་གནོད་སྦྱིན་རིགས། །རྫུ་འཕྲུལ་ལྡན་འདིར་ད་ལྟ་སྐྱེས། །ཨ་མ་ཕུལ་བས་འདི་ལ་བུ། །།
