རཱཀྵ་ས་དྲི་པ་མུཀྐ་ཊཾ། །ཏ་ཐི་བ་བི་ག་ཏ་ཀྲུ་ར་བི་ཀཱ་རི་སྟིས་པཱུ་ཛི་ཏ། །ཉམ་ང་སྲིན་པོའི་གླིང་དུ་ཕྱིན། །རྣམ་འགྱུར་གདུག་བྲལ་དེ་དག་གིས། །དེ་བཞིན་ཉིད་དུ་མཆོད་ནས་དེ། །བྷུཀྟཀྟ་པེ་ཎ་ནི་སྟ་ཎཱན་གེཎྜ་ས་མུ་ནི། །ས་ཏ་ཏྲ་དཱིལྟ་བི་བྷ་དཔུང་པས་བླངས་ཏེ་སྐད་ཅིག་གིས། །ཀླུ་ཡི་དབང་པོའི་ཁང་པར་འཕངས། །ལྷ་ཡི་དགའ་བདེ་ལ་འོས་བེ་དི་བྱོཏྠ་བ་སུ་ཁོ་ཙི་ཏ། །ཨ་ཀྲྀ་ཎོདྡྷིགྷ་དུཁཱཏཱི་ཏ་ཏ་མཱ་རོ་ད་ན་པའི། །སྣང་བ་འབྱོར་ལྡན་དེར་ནི་དེས། །སྡུག་བསྔལ་ཆེན་པོས་གཟིར་བྱས་པའི། །ཆོ་ངེའི་སྒྲ་ནི་ཐོས་གྷ་ནི། །ས་ཏ་མཱ་ཀརྱ་སོདྲེ་གཔྲ་ཀྲྀ་ཏྲེ་བ་ད་ཡཱརྔ་དྷི། །ཀི་མེ་ཏ་དི་པར་གྱུར། །དེ་ཡིས་དེ་ཐོས་ཆགས་བྲལ་ཅན། །རང་བཞིན་བརྩེ་བས་བརླན་པའི་བློས། །མདུན་དུ་ཀླུ་ཡི་ཏི་ས་པྲཙྪ་དྲྀཏྟཱ་གྲེ་ནཱ་ག་ཀ་ནྱ་ཀཾ། །སཱ་ཏཾ་བ་བྷཱ་ཥ་སཾ་སཀྟ་ཤོ་ཀོཟླ་པི་ཤུ་ནིརྨུ་བུ་མོ་དག། མཐོང་ནས་འདི་དག་ཅི་ཞེས་དྲིས། །ཡང་ཡང་མྱ་ངན་ལ་ཆགས་པའི། །དྲོ་བ་གསལ་བྱེད་ཧུཛྙཱ་ཧུ། །མྨཱ་ནི་ན་ཡནྟཾ་ནི་ཤཱ་སིཊྚིམ་དྷ་ར་ད་ལཀྵ་ཏི། །གུ་ཎ་ཤུགས་རིངས་ཀྱིས། །བིམ་པའི་མཆུ་འདབ་མཛེས་པ་དག། དྲི་མར་བྱེད་དེས་དེ་ལ་སྨྲས། །ཡོན་ཏན་བཱརཱ་ག་རཱ་ཛ་སྱ་པུ་ཏྲཀ་མ་ལ་ལོ་ཙ་ན། །ཛེསཉྫརྦ་སིདྡྷཱ་ཁྱབྲི་ཡཾ་པཉྩ་ཏཱ་མ་ག་ལྡན་པ་པདྨོའི་སྤྱན། །ཀླུ་ཡི་རྒྱལ་པོ་དག་གི་བུ། །ཐུ་བོ་དོན་ཀུན་གྲུབ་ཅེས་པ། །གཅེས་པ་ལྔ་པ་ཉིད་དུ་ཏ༑། ཏ་ཏཱསཱུ་ཏ་བི་ག་ཏཱ་ན་ནེ་པི་བྲྀཏྟ་སུ་ཁོདྨ་བེ། །གྷ་ནོ་ནཱ་རོ་གྱུར། །དེ་ཡི་སྡུག་བསྔལ་དག་གིས་ནི། །ཀུན་དགའ་དང་བྲལ་བདེ་དགའ་སྤངས། །ཁང་པ་འདིར་ནི་ད་ནེ་ནཱསྨིནྣ་བདྡྷ་བྷ་བ་ནེསྦི་ཏི། །ཨི་ཏི་ཏ་སྱཱབ་ཐུ་ཏྭཱ་སོནྟཔ་རཾ་ཙི་ཆོ་ངེ་དག། སྟུག་པོ་ཉིད་ཀྱིས་གནས་པར་བྱས། །ཞེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །རང་ཡུལ་མཐོང་བས་ཏྭཾ་བ་ཧན། །སྭ་དེ་ཤ་དན་ན་པཱསྟོ་ནཱ་ག་རཱ་ཛཱནྟི་ཀཾ་ཡ་ཡོ། །ནཱ་ག་རཱ་ཛསྟ་དགའ་བ་དེས། །ནང་དུ་ཡོངས་འདྲིས་འཛིན་བྱེད་ཅིང༌། །ཀླུ་ཡི་རྒྱལ་པོའི་དྲུང་དུ་ཕྱིན། །་ཀླུ་ཡི་རྒྱལ་པོ་མཱ་ཡཱནྟཾ་པ་རི་ཙཱ་ཡ་པྲ་ཡཱ་ས་ཧ། །ཨེ་བྱེ་ཧི་པུ་ཏྲེ་ཏི་བ་དན་དགའ་མ་དང་། །བཅས་པས་དེ་འོངས་ཡོངས་ཤེས་ནས། །བུ་གཅིག་ཚུར་ཤོག་ཚུར་ཤོག་ཅེས། །—།
