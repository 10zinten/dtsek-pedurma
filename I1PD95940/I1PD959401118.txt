བསྟན་འགྱུར། འཁྲི་ཤིང༌། །ཁེ་ཁིནྡེ་བ་པཱི་ཏྭཱ་མ་དྷུ་ཤ་ཏ་པ་ཏྲ། །བི་ཤྭ་ནྟི་མལྟ་དྲི་ཏ་ཊེ་ནི་ཥརྞ་བྷེ་ཛེ་གར་གྱིས་དལ་བཞིན་འདབ་བརྒྱའི་སྦྲང་རྩི་འཐུངས། །།ཚ་ཟེར་ལྡན་པའི་ཟེར་འཕྲེང་ནུབ་རི་ཡི། །ངོས་ཤ་ནི་རཾ་ཤུ་མ་ཏོ་ཤུ་མཱ་ལཱ། །ཏ་ཏ༔ས་མཱ་དཱ་ཡ་ཀ་རཱ་བ་ལགཽ་དི་ན་ཤི་ལ་འཁོད་པས་དལ་གྱིས་ངལ་གསོ་བསྟེན། །དེ་ནས་འོད་ཆགས་ཉིན་མོའི་དཔལ་དང་ནི། །ཆགས་ལྡན་ཡཾ་རཱ་ག་བ་ཏཱི་ཙ་སདྡྷཱ། །ཨུངྒཱ་ན་ལོ་བྷ་དི་བ་ར་སྨི་མཱ་ལཱི་མེ་རོརྫ་གཱ་ཐུན་མཚམས་དག་ཀྱང་ཁྲིད་བྱས་ནས། །འོད་ཟེར་འཕྲེང་བ་སྐྱེད་ཚལ་ལ་ཆགས་བཞིན། །ལྷུན་པོའི་ཕ་མཱ་པ་ར་པཱནྟྲ་བྷུ་མི། །ནཱི་ལཱམྦ་རཱ་ཛྲ་བྱ་ཏ་བཱ་ས་རཱནྟེ་བི་ལོ་ལཱ་ཏཱ་རཱ་ར་ཛ་རོལ་ངོས་ཀྱི་ས་གཞིར་སོང་། །རྒྱུ་སྐར་གཡོ་ལྡན་མཚན་མོ་རི་དྭགས་མིག། ཉིན་མོའི་མཇུག་ཏུ་གོས་ནཱི་མྲ་གཱ་གཱི། །བི་ལོ་ཀཱ་ཡནྟི་བྷ་ཡ་སཾ་བྷ་མེ་ཎ་མུ་ཧུརྗི་གཾ་ཏཱ་ན་བྷི་སྔོན་ལྡན་པ་མཐོང་། །འཇིགས་ཤིང་འཁྲུལ་པས་ཡང་ཡང་ཕྱོགས་རྣམས་སུ། །རྣམ་པར་བལྟ་ཞིང་མངོན་པཱ་རི་ཀེ་བ། །ཨ་ཐོ་ད་ཡཱ་དྲེཤི་ཁ་ར་ཤ་ཤཱི་ཀཀ་རཱ་བ་ལི་སྤུ་ཊི་ཀ་ཛཱ་ལ་པ་ཊཾ། །པར་འགྲོ་བ་བཞིན། །དེ་ནས་ཤར་གྱི་རི་རྩེར་རི་བོང་ཅན། །འོད་ཟེར་འཕྲེང་བ་ཤེལ་གྱི་དྲ་བས་བསྡམས། །མ་ཧོདྨ་བི་སིདྡྷ་པུ་རནྡྲི་ཀྲྀཏྟཾ་པྲྀཧཱུྃ་མ་ཧཱ་ཧརྨ་མི་བཱ་རུ་རོ་ཧ། །པྲ་ཀཱཎྜ་ཏཱ་བ་གམ་ཆེར་འཛེགས་དགའ་སྟོན་ཆེན་པོ་ལ། །གྲུབ་པའི་བུ་མོའི་གར་ལ་བལྟ་བ་བཞིན། །རྒྱུ་སྐར་རྒྱུ་དང་རཱཀྵ་ཏ་ལཱ་ཛ་པུཀྵེ་ཏ་སྨིནྣཾ་ཤཱ་ཙནྡ་མ་སོཎྜི་བཱ་ཧེ། ་། ཨ་བྷཱུརྨ་དྷ་བ་འབྲས་ཡོས་མེ་ཏོག་གཏོར། །མཚན་མོའི་ཟླ་བས་བག་མར་བླངས་པ་དེར། །སྦྲང་རྩིའི་བརྟུལ་ཞུགས་སྦྲང་མ་དྷུ་བྲ་ཏཱ་ནཾ་ཀོ་པི་པྲ་མོ་དཀུ་མུ་དཱ་ཀ་རཱ་ཎཾ། །མ་ཧོཏྠ་བོཏྠ་ཧོ་ར་རྩིའི་མྱོས་པ་ཅན། །ཀུ་མུད་འབྱུང་གནས་ཅི་ཡང་རབ་དགར་གྱུར། །དེ་ནས་ཉམས་དགའ་དགའ་སྟོན་པེ་ཐ་ཏསྨིནྡྲི་ལོ་ལ་ཕེ་ནོ་ཛཱ་ལ་མཱ་ལ་ཧཱ་ར། །ནྲྀ་ཏྱན་བ་བྷོ་སིདྡྷ་པུ་ཆེན་པོ་དེར། །འཕྲེང་བ་དོ་ཤལ་རབ་འབར་དབུ་བ་གཡོ། །གྲུབ་པའི་ན་ཆུང་འཇིག་རྟེན་རྣམ་པར་རནྡྲི་ལོ་ཀཉྩནྡོ་ད་ཡོདྷཱ་ཏ་ཨི་བཱ་ལིཾ་ཏཱབྷི། །ཏ་ཏཔྲ་བྷཱ་ཏེ་པ་རི་བཌྷ་མཱ་ནེ་མཛེས། །ཟླ་ཤར་བདུད་རྩིའི་རྒྱ་མཚོ་འཕྱུར་བ་བཞིན། །དེ་ནས་སྔ་དྲོ་དགའ་སྟོན་ཆེན་པོ་ནི། །—།
