རཱཏྟ་ཏྲ་ས་རཱ་ཛ་པུ་ཏྲ། །དི་ནེ་ཥཱ་ཏ་སྤྲོ་བྱ་ན་ཡོ་བ་ནཱ་བྷ་ས་མུམྨ་སམྨེ་གྷ་མ་བསྟེན་ཅིང་རྒྱལ་སྲས་དེ། །སྤྲིན་ཚོགས་གཞོན་ནུ་གསལ་བ་རབ་རྒྱས་པས། །དྲི་མར་བཅས་པའི་ཉིན་ལཱི་མ་སེ་ཥཱ། །ཨ་ཏཱནྟ་རེ་པུ་ཏྲ་མུ་ཁཱ་ར་བིནྡ་པ་ནིན་ནོཀྐཎྜ་ཏ་མཱ་ན་ས་སྱ། །རྣམས་རབ་ཏུ་གནས། །སྐབས་དེར་བུ་ཡི་བཞིན་རས་པད་མོ་དག། བལྟ་བར་ཡིད་ལ་རབ་ཏུ་འདོད་པ་ཡི། །ཙིནྟཱ་ནུ་བནྡཱ་དི་བ་བྷུ་མི་བྷཅུ་བྱཱ་དྷི་བ་བུ་བྷོ་ད་ར་བདྷཱ་མཱུ་ལ། །བ་ས་བདག་ལྟོ་བར་རྩེ་བ་བཅིངས་པའི་ནད། །བྱུང་བར་གྱུར་ཏེ་བསམས་པས་བཅིངས་ལས་བཞིན། །སྣ་བྱིབྲྀ་ཏོནཔུ་ར་དྷཱ་སྣེ་ནཱ་ནཱ་བྷི་ཥ་ཚུ་ཙིནྟཱ་བ་ཧི་ཏཱ་བ་དྷཱ་ནི། །ཚོགས་སྨན་བསམ་བསམ་གཏན་སྒྲུབ་བྱེད་ཅིང་། །ནད་གསོས་མིན་པར་རྟོགས་པའི་རྣལ་འབྱོར་དང་། །ཨ་སཱ༔་རོ་གཱ་བ་ག་མཱ་བྷི་ཡོ་ག་པརྗེ་ཧ་སནྡནི་ཏ་ཁེད་བཀྲི། །ཨུངྒེ་ཐེ་ཚོམ་གྱིས་ནི་བཞིན་ལོག་མཐོང་གྱུར་པའི། །སྨན་པས་ཕོ་བྲང་ཁང་པར་རབ་ཏུ་བསྐོར། །མྱ་ངན་ག་བྷི་སྐྱེ་བ་ནི་ཥཎྜ་མོ་ན་ཀཾ་ཙཱི་ཀ་ལཱ་པེ་ན་བ་དྷུ་ག་ཎེ་ན། །ཙི་ཏཱདྲི་འཇིགས་པས་བཞིན་དུ་དྲིལ་ཆུང་སྟེ། །སྨྲ་བཅད་དུ་གནས་བཙུན་མོའི་ཚོགས་རྣམས་ནི། །རི་མོ་བྲིས་ཏཱ་ཀཱ་ར་ཏུ་ལཾ་ཤི་ཏེ་ན་ནེ་སནྡ་ནེ་ཏྲེ་ཎ་བི་ལོཀྱ་མཱ་ན། །ཨཱ་སནྣ་ཀཱནྟ་ཀ་ར་པའི་རྣམ་པ་དང་མཚུངས་ལ། །བརྟེན་པའི་གཡོ་མེད་མིག་གིས་ལྟ་བར་བྱེད། །དལ་བུ་དལ་བུར་རྔ་ཡབ་མན་མརྡ་བི་སྱནི་ནཱ་ཙཱ་མ་ར་པལྤ་བ་ན། །ཨུཙྪཱ་ས་བྷཱ་ཛཱ་པ་རི་འདབ་མ་ནི། །ཉེ་འཁོད་མཛེས་པའི་ལག་པས་རྣམ་བསྐྱོད་པ། །མྱ་ངན་གྱིས་འཁྲུགས་ཤུགས་རིངས་པཱཎྜ་རེ་ཎ་ཤོ་ཀཱ་ཀུ་ལེ་ནེ་བ་ས་བཱི་གླུ་མཱ་ན། །ཤཱི་ཏཱཾ་བུ་ངྷི་གཱ་ར་ནི་བིཥཱ་བསྟན་པ་ཡིས། །ཡོངས་སུ་དཀར་བ་ལྟ་བུས་གཡོབ་པ་དེ། །ཆུ་གྲང་ལྡན་པའི་བུམ་པར་མིག་གཏད་དྲྀཊིཀཱ་ཥཱ་ཡཔཱ་ནེ་བི་ཧི་ཏཱ་བ་མཱ་ན། །ནི་དཱ་ནི་ཥ་དྷ་པྲ་ཏི་བནྣ་ཀོ་པ༔ཅིང་། །བསྐ་ཨེབའི་བཏུང་བ་དག་ལ་ཞེན་ལོག་བསྐྱེད། །གཉིད་ནི་དགག་པའི་ཁྲོ་བ་རབ་རྫོགས་ཤིང་། །པ་ཐོ་པ་དེ་ཤ་པྲ་བི་ཤདྲི་ཥ་ད། །ནི་བྱ་མ་ཡོདྲེ་ག་ཛུ་གུསླུ་མཱ་ནཀཱ་ཕན་པའི་ཉེར་བསྟན་དག་ལ་ཆགས་བྲལ་ཞུགས། །ཞེ་སྡང་བཅས་པའི་སྐབས་ཐོབ་ལུས་ལ་ཡང་། །སྨད་—།
