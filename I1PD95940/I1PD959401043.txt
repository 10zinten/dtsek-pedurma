ཏ་བཱ་རི་བཱ་ཧ། །བི་གཱ་ཧ་ཏེ་སྦི་ཏ་ཛ་བ་ན་བྷིཀྵ་མི་ཏཱ་ཡ་ཎཱི་སཱུ་ནུ་ར་ཡི་སགཏོར་བའི་འཆི་མེད་ལམ། ་། མགྱོགས་པ་རྒྱས་པ་རྣམ་པར་གཅོད་བྱེད་པ། །དགེ་སློང་འདི་སྟེ་བྱམས་པུཎྞཾ༑། ཨ་ནནྟ་མཱནྡྲ་ཡ་ནི་བྷནྟ་ཤཱནྟ་མིནྟཱ་པ་ཐཾ་སཏྭཱ་མ་ཧོ་ད་དྷིམྨ། །མའི་བུ་གང་པོ། །སྙིང་སྟོབས་ཆུ་གཏེར་ཆེ་གང་མཆོག་གི་ལམ། །ཤིན་ཏུ་ཞི་བ་མཐའ་མེད་དག་ལ་གནས། །པྲ་བྷཱ་མྲྀ་ཏེ་བ་སམྦ་ཏི་ཏམི་ཏཱ་ཡཿཔྲ་བྷཱ་བ་བཱ་ན་ཤཱ་ཛི་དེ་ཥ་བྷིཀྵུ། །བི་ལོ་ལ་འོད་ཀྱི་བདུད་རྩིས་ཕྱོགས་ཚིམ་རབ་ཏུ་སྤྲོ། །མཐུ་ལྡན་དགེ་སློང་འདི་ནི་རྟ་ཐུལ་ལོ། །གསེར་གྱི་ཏཱ་ལ་བལླི་པ་ལ་ཡཱ་བྷི་རཱ་མཱི་ཥི་ཤཱ་ལ་མཱ་རུ་རྒྱ་སུ་བཙྪཱ་ཏཱ་ལཾ། །ཡ༔པུ་ཎྱ་ཡངས་པའི་འཁྲི་ཤིང་ནི། །རྣམ་པར་གཡོ་བས་དཀྲིགས་པ་མངོན་དགའ་ཞོན། །བསོད་ནམས་གསལ་པཱུརྞ་བྱུ་ཏི་ར་བྷུ་པི་ཏི་ས་ཨེ་ཥ་བྷིཀྵརྨ་ཏི་མཱ་ནུ་པཱ་ལཱི། །ཨཱ་རུ་རྒྱ་བི་རྫོགས་མངོན་པར་འོངས་པ་གང་། །དགེ་སློང་འདི་ནི་བློ་ལྡན་ཉེ་བར་འཁོར། །གསེར་དང་རིན་ཆེན་ཌཱུརྨ་བི་མཱ་ན་ཤི་གཾ་སུ་བཎྜ་རཉྫཱ་ཛ་ལ་བ་ཏྲ་ལེ་ཁཾ། །ལིམྤནྣཾ་བཱ་གཙྪཱ་ཏི་ཡཔྲ་བྷ་བྷིརབ་འབར་པ་ཏྲ་རིས། །བི་ཌཱ་རྱ་ཡི་གཞལ་ཡས་ཁང་རྩེར་ཞོན། །འོད་ཀྱིས་བྱུགས་བཞིན་བྱོན་པ་གང་ཀཱ་ཏྱ་ཡ་ནོ་ནཱ་མ་ས་ཨེ་ཥ་བྷིཀྵ། ་། ཤ་རཱི་རི་ཎ་དྷརྨ་མི་བཱ་དྷི་རུ་བྱ་འདི་ནི། །དགེ་སློང་ཀཱ་ཏྱཱ་ཡ་ན་ཞེས་པའོ། །གང་ཞིག་གཟུགས་དང་ལྡན་པའི་ཆོས་ལ་བཞིན། །ཁྱུ་བི་གཱ་ཧ་ཏེ་ཁཾ་པྲྀ་ཥབཱ་ཧ་ནོ་ཡ། །པཱཋཔྲ་ཏིཥྛ་པ་རི་ནིཊྛི་ཏཱ་ནཾ་ག་རིཥཱ་དྷིམཆོག་ལ་ཞོན་མཁའ་ལ་རྣམ་པར་ལྡིང་། །རབ་གནས་ཡོངས་སུ་གནས་པ་རྣམས་ཀྱི་མཆོག། བློ་བརླིང་ཀོཊྛི་ལ་ཨེ་ཥ་བྷི་གླུ། །།བི་མཱ་ན་ཧཾ་ས་དྷུ་བྷིམུ་ཧུཏྟམི་ཏོརྨི་ར་མྱི་མུ་ཧུ་དགེ་སློང་འདི་ནི་ཀོཊྛི་ལ། །ཡང་ཡང་དད་པ་གཞལ་ཡས་ཁང་འོད་ཀྱིས། །ཡུད་ཙམ་བར་སྣང་འཛུམ་རན་རཱི་ཧཱུྃ། །ཀུཏྤནྣ་མ་ཧྥེ་ཏི་ཏ་པོ་ནི་དྷིམྨཔི་ལིནྡ་བཏྭཱ་ཧཱ་ཡ་ཨེ་ཥ་རླབས་དགའ་ལྡན་པར། །བྱེད་ཅིང་དཀའ་ཐུབ་གཏེར་གང་བྱོན་པ་འདི། །དགེ་སློང་པི་ལིན་ད་པད་སཱ་བྷི༑། ཡོ་ཡཾ་ས་མུཏྤ་ལྤ་ལ་ཏཱ་བི་ཏཱ་ན་བ་ནཱནྟ་རཱ་ལོ་བི་ཧ་རནཱ་པི་ཏི། །ཞེས་བརྗོད། །མེ་ཏོག་རབ་རྒྱས་འཁྲི་ཤིང་རྒྱས་པ་ཡི། །གསེར་ཚལ་ནང་ན་བཞུགས་ཤིང་བྱོན་པ་གང་། །།
