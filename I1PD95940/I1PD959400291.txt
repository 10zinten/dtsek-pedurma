།ཏ་པཱསྟི་ཕ་ལཾ་ཏ་ཏ། །ས་ཊཾ་དཱ་གཱ་མྱ་ན་གཱ་མི་ཕ་ལ་མརྟ་ཕ་ལཾ་ཏ་ཐཱ། །བུ་དྲི་མ་མེད་ཐོབ་ཅིང་། །ལན་ཅིག་ཕྱིར་འོང་ཕྱིར་མི་འོང་། །འབྲས་དང་དེ་བཞིན་དགྲ་བཅོམ་འབྲས། །ཏཱི་དྷཱ་ཏུ་ཀོ་བཱི་ཏ་ར་གས་མ་ལོ་ཐ་ཀཉྩ་ན། །ཨཱ་ཀཱ་ཤ་ཁམས་གསུམ་དག་ལ་ཀྱི་འདོད་ཆགས་བྲལ། །གསེར་དང་བོང་བ་རྡོ་བར་མཚུངས། །ནམ་མཁའ་ལག་པཱ་ཎི་ཏུ་ལོ་བྷུ་ཏཱ་སི་ཙནྡ་ན་ཡོས་མ། ་། ཐཱ་བཙྩཾ་བེ་ཆུ་ག་ཧ་ནེ་མཐིལ་མཚུངས་པ་དང་། །རལ་གྲི་ཙན་དན་མཚུངས་པར་གྱུར། །དེ་ནས་མཉན་ཡོད་རྒྱལ་བྱེད་ཚལ། །ཛི་ནི་ཛེ་ཏ་བ་ནེ་སྦི་ཏཾ། །བྷ་ག་བནྟཾ་ཡ་ཡོ་དྲནྡྷཾ་སོ་ཐ་ཀཱ་ཏྱཱ་ཡ་ནཾ་ག་ཡཱ། །སྨུག་མའི་ནགས་ན་བཞུགས་པ་ཡི། །བཅོམ་ལྡན་རྒྱལ་བ་བལྟ་རུ་དེ། །ཀཱ་ཏ་ཡ་ནའི་བཀའ་ཡིས་སོང་། །པྲ་ཎི་སཱ་ཏཱི་ཀྲྀ་ཏ་ཏི་བྲི་ཏྱ་བྷ་ག་བ་ཏ་སྭ་ཡཾ། །ཏཾ་ཐོ་ཎ་ཀོ་ཊི་ཀརྣོ་ཐ་ཕྱག་ནི་བྱས་ཤིང་བཅོམ་ལྡན་འདས། །རང་ཉིད་དགྱེས་པས་དྲི་བ་བྱས། །དེ་ནས་གྲོ་བཞིན་བྱེ་བ་རྣ། །བ་བྷ་ཥ་ཧནྟ་ནིབྷ་ར། །བྷ་ག་བཱན་དྷརྨ་ཀཱ་ཡེ་ན་དྲིཌོ་སཾ་ཤོ་ཏྲ་བཉྫ་ན། །དགའ་བ་རྒྱས་པས་དེ་ལ་སྨྲས། །བཅོམ་ལྡན་ཆོས་ཀྱི་སྐུ་ཡིས་ནི། །ཐོས་པའི་ལམ་ནས་མཐོང་བར་གྱུར། །ཨ་དྷ་ན་རཱུ་པ་ཀཱ་ཡེ་ན་པུ་ལྕི་ར་ལོ་ཀྱ་སི་མ་ཡཱ། ་། ཨ་ནལྤ་སུ་ཀྲྀ་ཏ་པྲ་ད་ལྟ་གཟུགས་ཀྱི་སྐུ་ཡིས་ནི། །བསོད་ནམས་ཀྱིས་ནི་བདག་གིས་མཐོང་། །མི་ཆུང་བསོད་ནམས་ཀྱིས་ཐོབ་སྱ་མི་དཾ་ཏཱདྷན་ན་མྲ་ཏཾ། ་། པཱི་ཏྭཱ་ན་ཏྲསྟི་མཱ་ཡཱནྟི་བལྕི་ཏ་ཨེ་བྱ། །ཁྱོད་ནི་མཐོང་བའི་བདུད་རྩི་འདི། །འཐུངས་ནས་ངོམས་པར་མ་གྱུར་པ། །དེ་དག་མཆོག་ཏུ་བ་ཏེ་པ་རཾ། །ན་སྦེ་ཧ་སྣ་བི་ཏེ་མུནྟིཀུ་རུ་ཏེ་ཀ་སྱ་ན་སྤྱི་ཧཾ། །ནིརྨེ་བསླུས་པ་ཉིད། །ཆགས་པ་མེད་ཀྱང་ཁྱོད་ཀྱི་གཟུགས། །སུ་ཞིག་ཆགས་པར་མི་བགྱིད་ལགས། །བྱུགས་པ་སཱུ་པི་ཏེ་དྲིཊཱི་ར་ཧོ་ཧརྙེ་ཎ་ལིམ་ཏི། །ཏཱཀྐ་ཐ་ཏཱ་ད་ནུ་དྷཱ་ནཾ་པ་མེད་ཀྱང་ཁྱོད་ཀྱི་སྤྱན། །ཀྱེ་མ་དགའ་བས་བྱུགས་པ་ལགས། །ཁྱོད་ཀྱི་གཏམ་དང་ཁྱོད་རྗེས་དྲན། །ཏཱཏྠཔཱི་སཱུནྣི་ཥ་བ་ནཾ། །ཨེ་ཏཀུ་ཤ་ལ་མཱུ་ལཱ་ནཾ་གཱི་ཏཕ་ལ་ས་སྲྀདྷ་ཡ། །ཁྱོད་དང་ཕྲད་དང་ཁྱོད་བསྟེན་པ། །འདི་དག་དགེ་བའི་རྩ་བ་ཡི། །འབྲས་བུ་ཕུན་ཚོགས་རྒྱས་པ་ཡིན། །།
