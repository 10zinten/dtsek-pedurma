སརྦ་དོསྟི་གཾ་ཧཱ་དྷི་པ། །།སོ་ཅི་ཀེ་ན་མ་ཡཱ་པུཏྠཾ་ཏངྑཾ་ཧཱནྟི་ཀ་བཱ་སི་ནཱ། །ཀུན་གཏེར་ཁྱིམ་གྱི་བདག་པོ་ཡོད། །ཚེམ་བུ་པ་དག་དེ་ཡིས་ནི། །ཁྱིམ་དང་ཉེ་བར་གནས་པས་སྔོན། །བྷུ་ཛ་མུནྡ་མྱ་ཏངྒེ་དནི་ཏཾ་ནི་ཏྱ་མཟི་ནཱམ། །ཏེ་ན་པུ་ཙེ་ན་དེ་བ་ཧཱུྃ་དཔུང་པ་བསླངས་ནས་སློང་རྣམས་ལ། །དེ་ཡི་ཁྱིམ་ནི་རྟག་ཏུ་བསྟེན༄༅། །བསོད་ནམས་དེ་ཡིས་ལྷ་ཉིད་པྲསྟོཉཏྲ་བི་ཧཱ་རཱ་མྱ་ཧཾ། །བཱ་ཧུརྨ་མ་བི་བྷཱ་ཏེ་ཥ་ད་ཎནི། །ཐོབ་ཅིང་ད་ལྟ་བདག་འདིར་གནས། །སློང་བའི་རྗེས་མཐུན་དཔུང་པ་དེས། །བདག་གི་གཡས་པ་སོརྫི་དཀྵི་ཎ། །ཏེ་ཏ་མཱ་མནྡྷཱ་མུ་ན་ཡཔུ་ནསཾ་པྲསྦི་ཏཱ་བ་ནེ། །གླུ་དྷི་ཏསྣེགྷ་འདི་མཛེས་སོ། །ཐུབ་པ་དེ་དག་དེ་ལ་ནི། །ཁམས་བདེ་བྱས་ནས་སླར་ཨིསོང་བ། །ནགས་སུ་བཀྲེས་པས་སཙྪཱ་ཡཾ་ད་དྲྀ་ཤུ༔པཱ་ད་སཾ་པ་རཾ། །ཏ་མ་པཱུཙྩཾ་ར་ཡཱ་ཙན་བྷོ་རྐང་འཐུང་མཆོག། མཛེས་ཤིང་གྲིབ་བསིལ་ལྡན་པ་མཐོང་། །དེ་བཞིན་ཉིད་དུ་དེ་དག་གིས། །དེ་ལ་ཛ་ནོ་ཏདྨ་དེ་བ་ཏེ། །ཨུཙྪ་ཙཱ་ར་ཙ་གམྦྷི་རཱ་བཱ་ཎི་བིསྨྲ་ཡ་ཀཱ་རི་ཎཱི། །ཨ་ཏྲ་ཡང་ནི་བཟའ་བ་བླངས། །ཡ་མཚན་བྱེད་པ་ཟབ་མོ་ཡི། །སྐད་ཀྱང་རབ་ཏུ་བྱུང་གྱུར་ཏེ། །རྫིང་བུ་པུཥ་རི་ཎྜི་ཏཱི་རེ་དྲོ་ཙྩཾ་དི་བྱཱརྣ་བྷོ་ཛ་ནཾ། །སཾ་པཱུཙྩཱ་མནྟི་ཏཾ་ག་ཏཱ་འདི་ཡི་འཇུག་ངོགས་ཀྱི། །འགྲམ་ན་བཟའ་བ་ལྷ་ཡི་ཟས། །ཡོངས་རྫོགས་ཡོད་ཀྱི་དེར་སོང་ལ། །བྷུ་ཛུ་ཏྭཾ་ཡ་དཱ་བྷིཔི་ཏཾ། །ཨི་ཏི་ཏེ་ན་ས་མཱ་དིཝཾ་བྷུཀྟ་ཏེ་དི་བྱ་བྷོ་ཛ་གང་ཞིག་འདོད་པ་གསོལ་བར་མཛོད། །དེ་སྐད་དེས་བསྟན་དེ་དག་གིས། །མཆོག་གི་བཟའ་བ་ཟོས་ནས་ནཾ༑། ཀོ་བྷ་བཱ་ནི་ཏི་བ་པྲཙྪནྟཾ་དི་བྱ་ཏ་རུ་སཾ་ཤ་ཡཾ། །སོ་སྱ་ཙ་ཙཀེ་ཤཱ་བསྐྱ་ནི། །མཆོག་གི་ཤིང་ལ་གནས་དེ་ལ། །ཁྱོད་སུ་ཞེས་པ་རབ་ཏུ་དྲིས། །དེས་ཀྱང་སྨྲས་པ་མཉན་ཡོད་ན། །གྲོ་ཧསྦོཉནཱ་ཐ་པཎྜི་ད། །ཨསྟི་ཏ་སྭཱ་ཧ་མ་བྷ་ཐཾ་བཱཙྪ་ཎཁྱིམ་བདག་ཨེམགོན་མེད་ཟས་སྦྱིན་ཡོད། །དེ་ཡིས་དགེ་འདུན་མཆོད་ལ་བདག། བྲམ་ཟེ་རིམ་གྲོ་ལ་མཁས་སཾ་གྷ་བྷོ་ཛ་ནེ། །ཙ་ཏུ་རཔ་རི་ཙཅུ་ཡཾ་པས༑། ཡོངས་སུ་སྤྱད་པ་བཞི་ཡིས་ནི། །：ཙ་ཏུ་ར་སོགས་ཀྱི་བོད་སྐད་ཚིག་རྐང་གཅིག་པུ་འདི་ཆད་པ་འདར་ལོས་བསྒྱུར། །།
