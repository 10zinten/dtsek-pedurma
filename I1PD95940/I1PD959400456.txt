ཏེརྦ་བྷུ་ཏ་ཡ། ་། བྷཱ་ཏཱི་ཎཾ་སནྟ་ཏོ་བྷེ་དཀ་ཐཾ་ནཱ་མ་ནི་བཏྟ་ཏེ། །ཨ་དྷཱ་པི་བའི་འབྱོར་པ་འོ། །ཆུང་མས་རྟག་ཏུ་མཚན་མོ་ནི། །སྡང་བའི་རིག་བྱེད་བསླབས་པ་ཡི། །སྤུན་ཟླ་བྱེ་ཏྲ་ནཾ་པདྡྷི་དྷིནྡེ་ཥ་བི་དྷྭཾ་ས་དཱ་ནི་ཤི། །ཨུནྣ་ཊཱ་ནཾ་སྭཱ་ཁཾ་ཤཱ་ནཱ་དཱི་དཾ་ཏཱ་བསྣ་བའི་བརྒྱུད་དག་ནི། །ངེས་པ་ཉིད་དུ་ཇི་ལྟར་ཟློག། ཇི་སྟེ་སྟ་རེ་རྣོ་ལྟ་བུའི། །བུད་མེད་ནང་དུ་མ་ཛ་ཡ་ཏེ། །ཡཱ་བཏ་ཀུ་ཋ་ར་དྷཱ་རེ་བ་ཡོ་ཏིངྒཱི་ཤ་ཏི་ནཱནྟ་རཾ། །བྷཱ་ཏ་རརྦ་ན་བཱ་དེ་ཞུགས་པ། །དེ་སྲིད་རང་རིགས་སྨིག་མ་ནི། །མཐོན་པོ་རྣམ་གཉིས་མི་འགྱུར་རོ། །ནོར་ལ་རྩོད་པ་སྤུན་ན་གུ་རུཾ་བ་རུ་ཎྱ་ཀུཏྠ་ཡཱ། །མི་ཏྲ་མེ་ཀཱ་བྷི་ལཱ་ཥ་ཎ་ན་ཡནྟི་དྭེ་ཥ་ཟླ་དང་། །སྨོད་ཅིང་རྩུབ་པས་བླ་མ་དང༌། །གཅིག་ལ་འདོད་པས་གྲོགས་པོ་ལ། །བུད་མེད་ཀྱིས་ནི་སྦི་ཡ། །ཏདྲ་དནྟི་ཧ་སནྟ་བི་བྷུ་བི་ལཱ་སེ་ན་ཡོ་ཥི་ཏ། །ཡཏ་སྡང་དུ་འཇུག། གྲོགས་ཀྱི་མཛའ་གཅུགས་རྩ་བ་ནས། །འབྱིན་པའི་རྒྱུར་ནི་རབ་འགྱུར་གང་། །དེ་ནི་པྲ་ཡཱ་ཏི་སུ་བྲྀཏ་སྣེ་ཧ་མཱུ་ལོནྣཱ་ལ་ན་ཧེ་ཏུ་ཏྭཾ། །ཧི་ཏ་མུཀྵེ་ཏི་པུ་ཏཱ་ཎཾ་བྷཱུ་བུད་མེད་སྨིན་མ་དག། གཡོ་ཞིང་དགོད་ཀྱང་སྨྲ་བར་བྱེད། །བུ་རྣམས་འབྱོར་པས་མངོན་འདོད་པས། །ཏ་ཡེཉབྷི་མ་ཏཾ་བྷ་བ། །ཨ་ནི་ཏྱ་ཏཱ་པ་རི་མུཀྟཀཱ་ལེ་ནི་དྷ་ན་མཱ་ཡ་ཡོ། །ཕན་པ་འདི་བརྗོད་འབྱོར་པ་ནི། །མི་རྟག་པ་ལས་ཡོངས་མ་གྲོལ། །དུས་ཀྱིས་ཕ་རོལ་སོང་བར་གྱུར། །ཨ་བི་བྷཀྵེ་དྷ་ནེ་ཤཀྟ་དེ་ཤཱནྟ་ར་དྷ་ནཱརྫ་ནེ། །ཛེཝཱ་བ་བྷཱུ་བུཔུཎྜ་སྐུ་ནོར་ནི་དགོས་པ་མིན་པའི་ཚེ། །ཕུ་བོ་རྣམས་ནི་ཡུལ་གཞན་དུ། །ནོར་ཉིད་བསྒྲུབ་ལ་ཆགས་གྱུར་ཅིང་། །ཧྲ་ཧེ་བིཏྟ་མ་ཙིནྟ་ཡཏ། །ཀ་ལེ་ན་གྲོ་ཧ་མཱཔ་ནཾ་བསྨཱ་ཤ་ན་བི་བཱ་དི་ནཾ། །གང་པོས་ཁྱིམ་དུ་ནོར་ལ་བསམས། །དུས་ཀྱི་ཁང་པར་འོངས་པར་གྱུར། །གོས་དང་ཟས་ལ་རྩོད་པ་དང་། །ནི་མནྡཱ་དཏྟ་ཀརྞ་ནཾ་བྷེ་དསྟེ་ཥཱ་མ་ཛཱ་ཡ་ཏ། །བིཏྟེ་བི་བྷ་ཛུ་མཱ་ནེ་ཐ་རྣ་བར་བུད་མེད་སྔགས་བྱིན་པས། །དེ་དག་རྣམས་ནི་རྩེ་བར་གྱུར། །དེ་ནས་ནོར་ནི་བཀོད་པ་ན། །ཏིརྦ་དཱི་ཥ་བ་ཤཱི་ཀྲྀ་ཏི། །དཱ་སཱི་སུ་ཏོ་ཡ་མི་ཏྱཱཀྵ་ནཱི་ཏཔུཎྞོ་ནི་རཾ་ཤ་སྡང་བས་དབང་བྱས་དེ་དག་གིས། །འདི་ནི་འབངས་མོའི་བུ་ཞེས་བརྗོད། །གང་པོ་སྐལ་པ་མེད་པར་།
