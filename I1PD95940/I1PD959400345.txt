བ་ནེསྦི་ཏཱ། །ཨ་སཱུ་ཏ་ཏ་ན་ཡརྗེ་བཱི་དི་བཱ་ཀ་ར་མི་བཱ་དི་ཏི། །མཱ་ཏུངྒརྡ་མ་ནགས་གནས་པའི། །ལྷ་མོ་དེ་ལ་སྲས་བལྟམས་ཏེ། །ཨ་དི་ཏི་ལས་ཉིན་བྱེད་བཞིན། །མ་ཡི་མངལ་གྱི་ལཱ་སྤྲོ་ཀུཀྟི་བྷི་ཏྭཱ་ས་ནི་ཏ། །ཏཱཉྩ་ཀྲེ་ཐ་སྭ་བྷཱ་བེ་ན་སྭ་བྷཾ་དྲི་མ་ཡིས། །མ་རེག་དེ་ཡི་དཀུ་བརྟོལ་བྱུང་ཅི། །དེ་ནས་དེ་ཡི་རང་བཞིན་གྱིས། །གདུང་བྲལ་གི་བི་ག་ཏ་བྱེ་ཐཾ། །ནིསྒྲ་ཙྪནྟེ་བ་རུདྷོ་སོ་བ་ལ་ཛི་དྷྭཾ་ས་ཡཱ་ཀཱ་ཎཾ། །ཤ་ལུས་ནི་རྣལ་གནས་གྱུར། །ཕྱིར་བྱུང་ཉིད་ན་བརྒྱ་བྱིན་གྱིས། །སྟོབས་ཤེས་འདོད་ནས་སྐད་ཅིག་ནི། །རྡོ་ཀྲེ་ཎ་བཛྲ་སཱ་རཱ་གས་སིཾ་ཤཱཀྱ་ཏཱ་མཱ་ཡ་ཡོ། །ཛཱ་ཏཱ་མཱ་ཏྲཤི་ཤུརྒ་ཏཱ་སྭ་ཡཾ་རྗེ་སྙིང་པོའི་སྐུ་དེ་བཟུང་། །དེ་ནི་ནུས་མེད་ཉིད་དུ་གྱུར། །བལྟམས་པ་ཙམ་གྱིས་ཁྱེའུ་དེ། །རང་གི་སཔྟ་པ་དཱ་ནི་ས། །དི་ཤ་བི་ལོ་ཀ་ཡནྡྲ་ཙེ་སུ་བྱཀྵ་ཀླུ་ར་ཡཱ་གི་རཱ། །གོམ་པ་བདུན་ཕྱིན་ནས། །ཡི་གེ་རབ་གསལ་ཚིག་གིས་ནི། །ཕྱོགས་སུ་རྣམ་པར་གཟིགས་ཤིང་སྨྲས། །ཨི་ཡཾ་མེ་ནིབྲྀཏྟཔཱུབྷྱཱ་མ་ཏི་ལོ་ཀེ་ཏུ་དཀྵི་ཎ། །བཤྲཱི་མཱ་ཛཱ་ཏི་ར་བྱེ་འདི་ནི་བདག་གི་མྱང་འདས་ཤར། །འཇིག་རྟེན་བློ་གྲོས་རྗེས་མཐུན་ལྷོ། །འདི་ཡང་སྐྱེ་བའི་མཐའ་དང་ཥ་སཾ་སཱ་རཱ་དི་ཡ་མུཏྟ་རཱ། །ཨེ་ཏི་ཏ་སྱ་བྲུ་བཱ་ཎ་སྱ་པྲི་ཐི་བཱི་ས་མ་ཀམྤ་ཏ། །ནུབ། །འདི་ནི་འཁོར་བ་དག་ལས་བྱང་། །དེ་སྐད་དེ་ཡིས་བརྗོད་པ་ན། །ས་ནི་ཀུན་ཏུ་འདར་གྱུར་ཏེ། །ཏ་མ་ས་ཀླ་བ་ལཾ་དྷཱནྟུ་མ་ཤལྟེ་བ་ཛ་གངྒཱ་རཾ། །ཨིནྡི་ཏ་མྱ་ཡ་ཤཤུ་བྷཾ་འགྲོ་བའི་བླ་མ་མི་ཟད་སྟོབས། །དེ་ནི་གཟུང་བར་མ་ནུས་བཞིན། །མཁའ་ཡི་ཆུ་རྒྱུན་གྱིས་བཀྲུས་པ། །སཏྭ་སྨེ་རཾ་ས་ཙཱ་མ་རཾ། །བྱོ་མཱམཱ་དྷ་ར་དྷོ་ཏ་སྱ་ཛ་ཤྲཱི་ཧུསྟཏྲ་དེ་དེ་ཡི་གྲགས་པ་དཀར་བའི་གདུགས། །སྙིང་སྟོབས་འཛུམ་པའི་རྔ་ཡབ་བཅས། །དེར་ནི་ལྷ་རྣམས་དག་བ་ཏཱ། །ཨ་སྨིནྣ་བ་ས་རེ་བྲྀཥཱསཱཐེ་ཡེ་ཎཱ་སི་ཏཱ་བྷི་དྷ། །ན་ར་དེ་ན་པྲ་བྷཾ་པྲྀཏྟཱ་གིས་བཟུང་། །སྐབས་དེར་ནག་པོ་ཞེས་པ་ཡི། །ཐུབ་པ་ཀིཋི་དེ་ཡི་རིར། །གནས་ལ་ཚ་བོ་མི་སྦྱིན་གྱིས། །ཀིཥིན་དྲྀསྦི་ཏོ་མུ་ནི། །ཀ་སྣ་དརྐ་ཤ་ཏཱ་ལོ་ཀ་ཨི་བཱ་ལོ་ཀེ་པྲ་དྲི་བྱ་འོད་ཟེར་མཐོང་ནས་རབ་ཏུ་དྲིས། །ཕུག་དང་ཁང་པར་བཅས་པའི་རི། །གང་གི་མུན་པས་དབུལ་བྱེད་།
