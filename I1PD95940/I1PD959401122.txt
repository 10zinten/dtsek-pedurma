བསྟན་འགྱུར། འཁྲི་ཤིང་། །ཁེཡ༑། བི་ཡུ་ཛུ་མཱ་ནཱ་ཏ་ན་ཡེ་ན་ནེ་ཡཾ་ཛཱི་བ་ཏྱཱ་བ་ཤྱ་ཛི་ན་ནཱི་བི་བཏྟཱ། ་། ཨ་དམན། །བུ་དང་བྲལ་བར་གྱུར་པ་བུ་མེད་ཀྱི། །མ་འདི་ངེས་པ་ཉིད་དུ་འཚོ་མ་ཡིན། །གཞོན་ནུ་ཤིན་ཡཾ་ཀུ་མཱ་རཿསུ་ཀུ་མཱ་ར་མཱུཏྟིརྨ་ཡེ་བ་རགྱུསཱ་ཏ་ནུ་པྲ་དེ་ན། །ཁཱ་ཎཾ་བི་ཙི་ཏེ་ཏུ་གཞོན་པའི་ལུས་ཅན་འདི། །རང་གི་ལུས་བྱིན་བདག་གིས་བསྲུང་བར་བྱ། །སྐད་ཅིག་འདི་བསམས་ཏི་ས་ཏཱ་མུ་བཱ་ཙ་མཱ་ཏརྦ་ཛ་ཏྭཾ་ས་སུ་ཏཱ་སྭ་བྷཱུ་མི། །གནྟཱ་མྱ་ཧཾ་སནྣ་ག་བ་དེ་ཡིས་དེར་སྨྲས་པ། །མ་གཅིག་ཁྱོད་ནི་བུ་བཅས་རང་སར་སོང་། །ཀླུ་དག་གསོད་པའི་གནས་སུ་བདག་རྒྱ་དྷཱ་མ་པྲ་ཡཙྪ་ཤོ་ཏྟཾ་ཤུ་ཀ་ཙིནྡྷ་མེ་ཏཏ། །ཨི་ཏུཀྟ་མཱ་ཏྲེ་པ་ད་ཡེ་ན་ཏེ་ན་མཆི་ཡིས། །གོས་དམར་མཚན་མ་འདི་ནི་བདག་ལ་སྩོལ། །ཞེས་པ་བརྩེ་ལྡན་དེ་ཡིས་བརྗོད་ཙམ་ལ། །སཱ་ཀམྤ་མཱ་ནཾ་ག་ལ་ཏཱ་ཏ་མཱུ་ཙེ། །ན་བཱ་ཙ་མེ་ཏདྡྷ་བ་ཏཱ་བི་རུ་དྷཾ་ཏཱཾ་ཤཾ་ཁ་ཡན་ལག་འཁྲི་ཤིང་འདར་དེས་དེ་ལ་སྨྲས། །འགལ་བ་འདི་ནི་ཁྱོད་ཀྱིས་བརྗོད་མི་བྱ། །ཁྱོད་ནི་དུང་གཙུག་ཙུ་ཌཱ་ད་དྷི་ཀསུ་ཏོ་མེ། །ཨ་ཧཾ་ཏུ་མོ་ཧོརྨི་མ་གཱ་དྷ་སིནྡྷ་མ་ཤརྨ་ཀརྨོ་པ་ན་ལས་ལྷག་བདག་གི་བུ། །བདེ་མེད་ལས་ཀྱི་ཉེར་གནས་གཏིང་མེད་མཚོ། །རྨོངས་པའི་རྦ་་རླབས་ལྡན་ཏ་བི་ཤྭ་མི། །ཨ་ཀཱ་ལ་ཡནྡོ་པ་ལ་དུས་ཧཱ་ནི་དུཁཱ་ནྱ་ལོ་དུཀྲྀ་ཏི་ནཱ་པ་པར་བདག་འཇུག་གོ། །ཉེས་པ་བྱེད་པ་རྣམས་ལ་སྡུག་བསྔལ་ནི། ་། དུས་མིན་ཕྱོགས་ཀྱི་རྡོ་བ་བཟོད་ཏནྟི༑། སཱ་དྷོ་སུ་དྷཱ་སནྣི་དྷི་རཱ་ཤི་ཏཱ་ནཾ་ས་ཏོ་ད་དྷེསཱསྟི་མནྟི་ཏ་ནུ་སྟེ། །དཀའ་ལྷུང་། །དམ་པའི་སྙིང་སྟོབས་ཆུ་གཏེར་ཁྱོད་ཀྱི་ལུས། །འགྲོ་བའི་མིག་དང་སྲོག་གྱུར་བསྐལ་པ་ནི། །བྷཱུ་ཡཱཛྫ་གམྨོ་ཙ་ན་ཛཱི་བ་བྷཱུ་ཏཱ་ཀལྤ་ཀྵ་ཡེ་སྱ་ཀླུ་ཡ་སཱ་ཀྱི་ཎཱི་ཡཾ། །ཨི་འཇིག་ཀྱང་མི་འཇིག་མངོན་སུམ་རྟེན་བྱེད་ཀྱི། །བདུད་རྩི་ཉེ་འདི་ལེགས་པར་ཚེལྡན་གྱུར་ཅིག། ཅེས་ཏུ་ཙ་མཱ་ནས་ཏ་ཡཱ་པྲ་ཡནྣཱདྨ་དྷཱ་ནུ་བནྡྷསཱ་ཏ་ནུ་པྲ་དཱ་ནེ། །ཡ་དཱ་བྷ་པ་དེ་ཡིས་བརྗོད་ཅིང་འབད་པ་དེས། །རྗེས་སུ་བཅིངས་བྱས་རང་ལུས་སྦྱིན་པ་ལ། །གང་ཚེ་སེམས་སྤྱོད་བནྡི་ལ་ཙིཏྟ་པྲྀཏྟི་སྟི་ཤཾ་ཁཱི་ཙཱུ་ཌོ་པ་ཏ་དཱ་ཛ་གཱ་ད། །ཏཱརྒྱུ་སྱ་བྷགླུ་བ་ཡ་མཱ་དི་གཡོ་བ་མེད་གྱུར་པ། །དེ་ཚེ་དུང་གཙུག་གིས་ཀྱང་དེ་ལ་སྨྲས། །མཁའ་ལྡིང་ཟས་སུ་བདག་སྤྲོས་དང་པོར་—།
