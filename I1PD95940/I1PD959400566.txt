ཚ་ཀཡོ་རྱ་ཏཱ། །ཨ་ད་རི་དྲ་ཛ་གཏེ་ཚུ་རཏྭཱརྨི་ཛ་ལ་དྷི་ཡ་ཡོ། །མོངས་གཞོམ་བརྩོན་པས། །འགྲོ་བ་དབུལ་མིན་བྱ་བའི་སླད། །རིན་ཆེན་དོན་གཉེར་རྒྱ་མཚོར་སོང་། །ཀ་ཐལྕི་དི་བ་སཾ་སཔྟཱཿས་པི་ཏཱ་དྲྀ་ཌྷ་ནིཤྩ་ཡ། །ས་མ་རུ་ཏྱ་པྲ་བ་ཧ་ཕ་ཡིས་ཇི་ཞིག་ལྟར་སྟེས་ཉིད། །རྗེས་གནང་ངེས་པ་བརྟན་པ་དེ། །གཟིངས་ལ་ཡང་དག་ཞོན་བྱས་ཎཾ་རཏྣ་དཱི་པ་མ་བཱསྟ་བཱན། །ཏ་ཏྲ་པྲ་བ་ཧ་ཎཱ་རུ་རྟེན་བ་ཎི་ཛིས་ཧ་ནས། །རིན་ཆེན་གླིང་དུ་ཕྱིན་པར་གྱུར། །དེར་ནི་ཚོང་པ་གཟིངས་དག་ལ། །ཞོན་ནས་ལྷན་ཅིག་ཡཱ་ཡི་ན། །སོ་བྲ་བཱི་ཏ་ཀྲྀ་ཡ་ཏྭཾ་ཀཱ་མཾ་ཡུནྣཱ་བྷིམྨ་ཎི་སཾ་གྲ་ཧ། ་། ཨེ་འོངས་རྣམས་ལ། །དེ་ཡིས་སྨྲས་པ་ཁྱེད་ཅག་རྣམས། །རིན་ཆེན་འདོད་པ་བསྡུ་བར་མཛོད། །ཐུན་ཏེསཱ་མཱ་ནྱ་རནྟིཀྱུ་མ་མ་ནཱསྟི་བྲ་ཡོ་ཛ་ནཾ། །ཀ་ཤ་མ་ཧཱནྟི་བྷཱ་སཱནྟི་མོང་རིན་ཆེན་འདི་དག་གིས། །བདག་ལ་དགོས་པ་ཡོད་མ་ཡིན། །བདག་ཅག་མཛོད་ན་རིན་ཆེན་མཆོག། སནྟི་རནྣོཏྟ་མཱ་ནི་ན། །ཀིནྟུ་ཙིནྡྷ་མ་ཎི་པཱསྟི་བི་པུ་ལོ་ཡསྨྲ་མོ་བྱ་མ། །འོད་དང་ལྡན་པ་མང་པོ་ཡོད། །འོན་ཀྱང་ཡིད་བཞིན་ནོར་ཐོབ་སླད། །བདག་གི་རྩོམ་པ་ཆེན་པོ་འདི། །ཏེ་ན་བི་དྲུ་ཏ་རཱ་རི་ཧཱུཾ་ཀརྟུ་མེཙྪཱ་མི་མེ་དི་ནི། །ཐུ་ཏསྨྲ་ཡཱ་ནཱ་ག་རཱ་ཛསཱ་དེ་ཡིས་ས་ནི་དབུལ་བ་ཉིད། །ཉམས་པ་དག་ཏུ་བྱ་བར་འདོད། །རྒྱ་མཚོ་ཞེས་བྱ་ཀླུ་ཡི་རྒྱལ། །ཆུ་ག་རཱ་ཁཱ་མ་ཧོ་ད་དྷོ། །བ་ས་ཏྭསྟི་ཤྲ་ཧེ་ཏ་སྱ་ཙིནྟ་ཏཱརྦ་བྲ་དོ་མ་ཎི། །བི་གཏེར་ཆེ་ལ་རབ་ཏུ་གནས། །བསམ་པའི་དོན་སྟེར་ནོར་བུ་ནི། །དེ་ཁྱིམ་ན་ཡོད་བདག་གིས་ཐོས། །མི་ལཾ་རྒྱུ་བི་ཥ་མནྣཱཌྷ་ཏ་མཱ་དཱ་ཏུཾ་བྲ་ཛཱ་མྱ་ཧཾ། །ནཱསྟི་དྷིརྨ་ས་ཧཱ་ཡཱ་བཟད་ལམ་ནི་བགྲོད་བྱས་ཏེ། །བདག་ནི་དེ་ཉིད་ལེན་དུ་འགྲོ། །བརྟན་པའི་གྲོགས་ལྡན་འབད་རྩོལ་ནཱི་དུརྒ་མི་བྱ་བ་ས་ཡི་ནཾ། །ན་ཙམདྲི་ར་ཧེ་ཀིཾ་ཙིཏ་བྱ་ས་ནཾ་བོ་དང་། །ལྡན་པ་རྣམས་ལ་བགྲོད་དཀའ་མེད། །བདག་གི་ལེགས་བྱས་རྩོམ་པ་འདི། །གལ་ཏེ་འཐོབ་བྷ་བི་ཥྛ་ཏི། །ས་ཏྲ་མེ་བ་པཱ་རཱརྦ་ཡཾ་ཡ་དི་མེ་སུ་ཀྲྀ་ཏོ་བྱ་མ། །ཨི་ན་བདེན་ཉིད་དུ། །བདག་དང་བྲལ་བས་ཁྱེད་ཅག་རྣམས། །ཅི་ཡང་གདུང་བར་མི་འགྱུར་རོ། །ཞེས་—།
