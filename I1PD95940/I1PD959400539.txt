སཏྠ་ཛྷ་མ་བྷཱ་ཥ་ཏ། །ཤ་ར་ཎྱ་ཤ་ར་ཎསྟ་སྱ་བྷ་ག་བཱན་བྷ་བ་ཧ་རི་འཚལ་ཀུན་མཁྱེན་ལ་གསོལ་བ། །བཅོམ་ལྡན་སྐྱབས་ཐོབ་སྐྱབས་སུ་འོས། །ཉོན་མོངས་མ་ལུས་ཉམས་པའི་ཎི༑། ཨ་ཤེ་ཥ་གླེ་ཤ་ནཱ་ཤཱ་ཡ་པྲ་བྲ་ཛྙཱ་མེ་བི་དྷི་ཡཱ་ཏྭཾ། །ཙ་པ་ལཾ་བྲ་སླད། །སྲིད་པ་འཕྲོག་པའི་རབ་བྱུང་ནི། །བདག་ལ་རབ་ཏུ་སྒྲུབ་པར་མཛོད། །སེམས་ཅན་རྣམས་ཀྱི་ཎི་ནཱ་མཱ་ཡུསྟ་ཏོ་པི་ན་བ་ཡོ་བན། །བི་དྷུ་ད་ཙ་ལ་ན་བི་ལཱསྟ་ཏོ་པེ་ཏཱ་བི་ཚེ་སྲོག་གཡོ། །ལང་ཚོ་གཞོན་པ་དེ་བས་ཀྱང་། །གློག་ནི་འཁྱུག་པ་བཞིན་དུ་གཡོ། །དེ་བས་ཀྱང་ནི་བྷཱུ་ཏ་ཡ། །།ཨི་ཏི་ཏ་སྱ་བྲ་བཱ་ཎ་སྱ་སུ་ག་ཏཱ་ནུ་གྲ་ཧོ་དི་ཏྭཾ། །པ་པཱ་ཏ་བཱི་འབྱོར་པ་འདི། །ཞེས་པ་སྨྲ་ཞིང་ཆགས་བྲལ་བི། །དེ་ཡི་ལུས་ལ་བདེ་གཤེགས་ཀྱིས། །རྗེས་སུ་བཟུང་བ་ཏ་ར་ཛ་སཔྲ་བྲ་ཛྙཱ་བ་པུ་ཥི་སཱ་ཡཾ། །རཀྟ་ཙཱ་བ་ར་སུ་བྱཀྟ་བི་བྷཱ་ལས་སྐྱེས་པའི། །རབ་བྱུང་རང་ཉིད་ལྷུང་བར་གྱུར། །ཆོས་གོས་དམར་པོས་རབ་གསལ་བའི། །ཆགས་ཎས་བི་རཀྟ་ནཾ། །པ་ཏྲ་གྲ་ཧེ་ཎ་ཏ་ཏྭཱ་ཛ་པུ་ནསཾ་སཱ་ར་པཱ་ཏྲ་ནཾ། །བྲལ་ཉིད་ནི་དེ་ཡིས་བཟུང་། །ལྷུང་བཟེད་བཟུང་བས་སླར་ཡང་ནི། །འཁོར་བའི་སྣོད་ཉིད་རབ་ཏུ་བཏང་། །ཏ་སྱ་ཏཱ་མ་ད་བྷུ་ཏཾ་སིདྡྷཾ་པྲ་ཏྱངྒཾ་བཱིགྱུ་བྷི་ཀྵ་བ། །ཏཧྥུཏྤ་བྲྀཏྟཾ་པ་པྲཛྙཱབྷ་དགེ་སློང་རྣམས་ཀྱིས་དེ་ཡིས་ནི། །རྨད་བྱུང་གྲུབ་དེ་མངོན་སུམ་མཐོང་། །དེ་ཡི་སྔོན་བྱུང་བཅོམ་ལྡན་ག་བནྟཾ་ས་ཙཱ་བྲ་བཱིཏ། །བཱ་རཱ་ཎ་སྱ་པུ་ར་རཱ་ཛཱ་ཀྲྀ་ཀཱིནཱ་མ་ཏ་ཐཱ་ག་ལ། །དྲིས་ཤིང་དེས་ཀྱང་རབ་གསུངས་པ། །རྒྱལ་པོ་ཀྲྀ་ཀཱི་ཞེས་པ་སྔོན། །:ཝཱ་རཱ་ཎ་སཱིར་དེ་བཞིན་ཏེ༑། ཀཱ་ཤྱ་པཱ་ཁྱེ་བྷ་ག་བ་ཏི་པྲ་ཡཱ་ཏི་པ་རི་ནིབྲྀ་ཏི། །ཤ་རི་ར་མ་གཤེགས། །བཅོམ་ལྡན་འོད་སྲུངས་ཞེས་བྱ་བ། །ཡོངས་སུ་མྱ་ངན་འདས་གྱུར་ཚེ། །དེ་ཡི་གདུང་ནི་སྱ་སཏྠཾ་ཏསྟུ་པཾ་ར་མ་ཡཾ་བྱ་དྷཱཏ། །སྭགྲྭ་བ་གཱ་ཏ་ན་སྲོ་ཌྷ་རབ་སྦྱངས་ནས། །མཆོད་རྟེན་རིན་ཆེན་རང་བཞིན་དག། མཐོ་རིས་སྙོག་ནུས་བསོད་ནམས་ནི། །མཱུནྟི་པུ་ཎྱ་མི་བོནྣ་ཏཾ། །ཏསྨིནཱ་རོ་བྱ་མཱ་ཎཱ་ཡཾ་ཡསྒྱུ་པཱུ་ཛ་པ་རི་གྲ་གཟུགས་ཅན་བཞིན་དུ་མཐོ་བ་བྱས། །དེ་ལ་མཆོད་པ་ཡོངས་འཛིན་ཅིང་། །བྱ་རུ་འཛུགས་པ་བྱེད་པའི་—།
