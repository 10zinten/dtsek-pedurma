པ་ཏིཥྚ་ཙ། །ཨ་བྷུཏྤ་ཏྱཱ་ག་ཏ་པྲ་ཎསུ་དྷཱ་སིལྟ་ཨི་བ་ཥ་ཎཱཏ། །ཏ་དེ་ཐོས་གྱུར་ནས། །བདུད་རྩིས་བྲན་བཞིན་སྐད་ཅིག་ལ། །སྲོག་ནི་སླར་ཡང་བྱུང་བར་གྱུར། །།ཀུན་སྨིནྣཾ་ཧོཏྠ་བཱ་ནནྡྲ་བོ་དྷི་སཏྭ་བདྷུསུ་ཏ། །ཀན་ཡ་ཤོ་དྷ་རཱ་མྲྀ་ཏ་དགའི་དགའ་སྟོན་ཆེན་པོ་དེར། །ཟླ་བ་སྒྲ་གཅན་གྱིས་ཟིན་ཚེ། །བྱང་ཆུབ་སེམས་དཔའི་བཙུན་མོ་ནི། །རཱ་ཧུ་གྲ་སྟེནི་བྱ་ཀ་རེ། །རཱ་ཧུ་ལཱ་ཁྱས་བཱ་ལོ་བི་ཀྲྀ་བ་ཏེཛྫརྨ་གྲགས་འཛིན་མ་ལ་སྲས་མཛེས་བཙསི། །སྒྲ་གཅན་ཞེས་པ་བྱིས་པ་དེའང་། །མི་བདག་དེས་བསྐྱེད་ཤཾ་ཀི་ཏེ། །ཛན་རྩྭ་ས་སི་ལཤུད་དྷི་ནི་སྟམྦྷ་སི་པྲ་རླུ་བི། །བྷ་དོགས་པའི་ཚེ། །མ་ཡིས་དག་སླད་རྡོ་དང་བཅས། །འཕངས་ཚེ་ཆུ་ལས་འཕྱོ་བར་གྱུར༄༅། །བཅོམ་ག་བཱ་ན་པི་སཔྟ་ཧ་སྦི་ཏོ་ནིཉྩ་ལ་བི་གྲ་ཧ། །བཛྲ་པམཾ་ཀ་བནྡྷེ་ན་དེ་ལྡན་འདས་ཀྱང་ཉིན་ཞག་བདུན། །སྐུ་ནི་གཡོ་བ་མེད་པར་གནས། །རྡོ་རྗེ་སྐྱིལ་ཀྲུང་བཅིངས་པ་ཡིས། །བཱ་ནཾ་བིསྨྲ་ཡཾ་བྱ་བཱཏ། །ས་བྲཧྨ་ཀཱ་ཡི་ཀཱ་ཁཱ་བྷི་དེ་བ་ཏཱ་བྷཾ་བི་བོ་དྷི་ཏ། །ལྷ་རྣམས་ཡ་མཚན་ལྡན་པར་བྱས། །ཚངས་པའི་རིགས་ཞེས་བྱ་བ་ཡི། །ལྷ་གཉིས་ཀྱིས་ནི་བསྐུལ་བ་ན། །ཨ་བ་དཏ་པ་ར་མཱ་ནནྡ་སུ་དྷཱ་སནྡོ་ཧ་པུནྡ་ར། །ཨ་ཧོ་བ་ཏ་མ་ཡཱ་ཛྙཱ་ཏཱ་མཆོག་ཏུ་ཀུན་དགའ་བདུད་རྩི་ཡི༄༅། །ཚོགས་ལྟར་མཛེས་པས་བཀའ་སྩལ་པ། །ལྷ་དང་ལྷ་མིན་དབང་པཱུརྦ་ནེ་ཥཱ་སུ་ཁ་སྦི་ཏི། ་། ཡ་ཡཱ་སུ་རཱ་སུ་རེ་ཤྭརླ་སུ་ཁཾ་དུཁ་ག་ཎཱ་ཕྱུག་བདེ། །གང་གིས་སྡུག་བསྔལ་དུ་རྩི་བ། །བདེ་བར་གནས་པ་འདི་ནི་སྔོན། །ཀྱེ་མ་བདག་གིས་ཤེས་ཡ་ཏེ། །ལཱ་པ་ཙྩཾ་བྷསྨཱ་བི་ཏྭཾ་གཱསྟ་རུ་ཎྱཔེ་ཡུ་ཥཱརྔསཱརྒ་སཾ་བྷོ་ག་བརྒ། །མ་གྱུར། །མཛེས་སྡུག་ཆུས་རྒྱས་ལང་ཚོ་ལྡན་པའི་ལུས། །བདུད་རྩིས་བརླན་པ་མཐོ་རིས་ལོངས་སྤྱོད་ཚོགས། །ཨ་སྱཱ་ཤེ་ཥ་ཏྱཱ་ག་ཧེ་ལ་སུ་ཁ་སྱ་སྤནྣཱ་བནྡྷེ་པཾ་ཤུ་ནི༔སཱ་ར་ཨེ་བ། །སཾ་ཏ་སྟོམ་ལུས་བཏང་གྱུར་རྩོལ་མེད་བདེ་འདི་ཡིས། །འགྲན་ཟླ་སྙིང་པོ་མེད་པའི་རྡུལ་ཉིད་དོ། །མི་བཟད་ཡུལ་གྱི་ཧཾ་བི་ཥ་ཡ་བི་ཥ་མ་གླེ་ཤ་སཾ་སཱ་ར་པཱཏྨ་ཀཱནྟཤྭ་ཏྭཱ་བྲི་ཏི་མུ་པ་ག་ཉོན་མོངས་འཁོར་བའི་ལམ་འགྲོན་ཀུན་ཏུ་གདུང་གྱུར་ཞིང་། །ཉོན་མོངས་ཅན་བདག་ཞི་བས་དགའ་བ་—།
