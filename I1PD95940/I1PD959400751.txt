པ་ར་བ་ཤ། །ད་རི་དྭ་སཙྪཱ་སེ་མྲྀ་ཏ་ཨི་ཏི་ན་ཙཱ་དྲསྟི་བི་མ་ཏིར་ད་རི་དྲ་ཉིད་དག་ཏུ་འགྱུར། །དབུལ་པོ་དབུགས་དང་བཅས་ཀྱང་ཤི་ཞེས་འདི་ལ་ཁྱད་པར་བློ་གྲོས་མེད། །སྐྱེ་བི་ཏཱ་ནི་ཛ་ཛ་ན་བི་ཧཱི་ནོ་ད་ཤ་དི་ཤ། །ཨ་བྱཱཙཱ་ཡཱ་བསུ་ག་དབུལ་པོ་ཕྱོགས་བཅུ་འདི་དག་རང་གི་སྐྱེ་བོ་དག་གིས་དམན་པ་བཞིན། །དེ་སླད་དམན་ལ་མཉེས་གཤིན་ཏཾ་ཏེ་སྨཱཀྵ་པ་ཎ་བཏྲ་ལཾ། །ཨ་ན་རཱ་དྷི་ཏ་བུདྷཱ་ནཾམུབྷྱཱ་ནཾ་ཀུ་ཤ་པའི། །བདེ་བར་གཤེགས་པ་མཆོད་པར་བྱ། །སངས་རྒྱས་མཉེས་པར་མ་བྱས་པའི། །རྨོངས་ལ་དགེ་བ་ལཾ་ཀུ་ཏ། །ཨེ་ཥ་པདྨ་ད་ལཱ་ཤཱཀྵཀྱི་པ་ཏུ་པནྣཱ་བཱནྡྷ་བ། །།ཡ་ཏྲ་ཡ་ཏྲག་ལ་ཡོད། །པདྨའི་འདབ་མ་ལྟ་བུའི་སྤྱན། །ཕོངས་པའི་གཉེན་འདིས་གང་གང་དུ། །སྤྱན་ནི་འཕངས་དྲི་ཤྲཱི་ཛཱ་ནེ་ཏཏྲ་ཏ་ཏཱ་ཐ་ཡ་ཤི་ཡ། །ཨི་ཏི་པ་སྐྱུརྦ་རཿཐུ་ཏྭཱ་བཱཏྨ་པ་དེ་དང་དེར། །དཔལ་ནི་གནས་པ་དག་གིས་ཤེས། །དེ་སྐད་བདག་པོའི་ཚིག་ཐོས་ནས། །ཁྱིམ་དུ་ཎི་པཱ་ད་ར་གྲི་ཧེ། །ཤུ་ཙི་པྲ་ཎྜི་ཏ་བི་ད་དྷེ་བྷོ་ཕྱོ་བྷ་ག་བ་ཏཀྲྀ་ཏེ། །གུས་བྱས་བྲམ་ཟེ་མོས། །ཟས་ནི་གཙང་མར་སྦྱར་བ་དག། བཅོམ་ལྡན་འདས་ཀྱི་སླད་དུ་བསྒྲུབས། །བྷ་ག་བཱན་པི་སརྔ་ཛྷསྟ་ཡོཙྪཱ་ཏྭཱ་ས་མཱི་ཧི་ཏཾ། །པྲ་ཎ་ཡཱཐ་བཅོམ་ལྡན་ཐམས་ཅད་མཁྱེན་པས་ཀྱང་། །དེ་དག་བསམ་པ་ཡོངས་མཁྱན་ནས། །གུས་པས་གསོལ་བ་ན་ཡཱ་ཙཀྲེ་ཏ་སྱ་པཱུ་ཛཱ་པྲ་ཏི་གྲ་ཧཾ། །བྱཀྟདཱ་རི་ཎྱ་དུ༔ཁ་སྱ་བྷཱུ་འདེབས་པ་དེའི། །མཆོད་པ་དག་ནི་བླང་བར་མཛད། །དབུལ་བའི་སྡུག་བསྔལ་རབ་བསལ་པ། །བདག་ཡསྨ་བི་བྷ་བོདྷ་བ། །ཙ་ཀཱ་ར་ཛི་ན་པཱུ་ཛཱནྟེ་པྲ་ཎི་དྷཱ་ན་མི་ཏི་དཱི་ཛ། །ལ་འབྱོར་པ་འབྱུང་གྱུར་ཅིག། ཅེས་པའི་སྨོན་ལམ་གཉིས་སྐྱེས་ཀྱིས། །རྒྱལ་བ་མཆོད་པའི་མཇུག་ཏུ་བྱས། །ཨ་ཐ་ག་ཏཱ་ནི་ཛཾ་ཀྵེ་ཏྲ་བི་པྲས་པྱ་ཡ་བཱ་ཀུརཱན། །སརཱ་ན་པ་ཤུདྡྷོ་བཙྪཱན་ས་དེ་ནས་བྲམ་ཟེ་རང་ཞིང་དུ། །སོང་བས་འབྲུ་ནས་མྱུ་གུ་ནི། །ཐམས་ཅད་གསེར་དུ་མཐོང་གྱུར་ནས། །ཧ་སཱ་ཏྱཱཀྟ་དུརྒ་ཏི། །ཏ་སྱ་པུ་ཎྱ་ས་མུདྡྷཱ་ཏཾ་དྲིཝཱ་ཧེ་མ་མ་ཧཱི་པ་འཕྲལ་ལ་ཉིད་དུ་དབུལ་བ་བཏང་། །ས་ཡི་བདག་པོ་གསལ་རྒྱལ་གྱིས། །དེ་ཡི་བསོད་ནམས་ལས་འཁྲུངས་།
