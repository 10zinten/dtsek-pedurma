བ༑། ཏེ་སྐྱེ་བ་དེ་ཤེ་ཥཱ་མ་ནོ་ཧ་རེ་ཎུ་ཏེ་བྱི་བ་པུཏྟཱ་ཀ་ར་བཱ་ས་རེ་ཞུ། །ཨེ་ཀེ་ཡིན། །ཡིད་འཕྲོག་ཡུལ་ནི་དེ་དག་ཁོ་ན་དང་། །མེ་ཏོག་འབྱུང་གནས་ཉིན་པར་དེ་དག་ཉིད། །སུ་ཞིག་ན་ཀེ་ན་པི་བི་ནཱ་ཛ་ན་སྱ་སརྦ་བི་ཥཱ་དཱ་སྤ་ད་ཏཱ་མུ་པི་ཏི། །བི་ཤོ་ཏ་ཏཔྲ་གཅིག་དང་བྲལ་བའི་སྐྱེ་བོ་ལ། །ཐམས་ཅད་ཆགས་བྲལ་གནས་ཉིད་དག་ཏུ་འགྱུར། །དེ་ནས་ས་གཞི་དག་བྲ་ཛི་ཏཱ་ནུ་རཱུ་པཾ་བི་དྷཱ་ཡ་སཱ་བྷཱུ་ཏ་ལ་ཙནྡྲ་རེ་ཁཱ། །སུ་ཁོདྡྷི་ཏ་དིཀླུ་པ་གི་ཟླ་རིས་ཏེ། །རབ་བྱུང་རྗེས་སུ་མཐུན་པའི་ཆས་བཟུང་ནས། །བདེ་བས་བཏང་སྟེ་ཕྱོགས་སུ་ཡོངས་རི་བྷ་མནྟཱི་བཱ་རཱ་ཎ་སཱི་ཤཱནྟི་མི་བཱ་པ་པཱ་ད། །ཏ་ཏཱ་ཐ་རཱ་ཛྙཱ་ཀྲྀ་ཀཱི་ནཱ་མི་འཁྱམ་ཞིང་། །ཞི་བར་མིབཞིན་དུ་བཱ་རཱ་ཎ་སཱིར་ཕྱིན། །དེ་ནས་རྒྱལ་པོ་ཀྲྀ་ཀཱི་འདོད་ལྡན་པས། །གཱ་ཀླུ་པ་ཕྲརྒྱུ་མཱ་ནཱ་སྱ་བྷི་ལཱ་ཥ་བྷཱ་ཛཱ། །ཏ་པཔྲ་དཱིསྟགི་ཤི་ཁེ་བ་ཏ་སྱ་སཾ་དེར་ནི་རི་དྭགས་མིག་དེ་དོན་གཉེར་ཡང་། །དཀའ་ཐུབ་མེ་ལྕེ་འབར་བཞིན་དེ་ཡིས་ནི། །ཇི་ཞིག་ལྟར་སྤན་ཡོ་གྷ་ན་ཀ་ཐོ་ཙི་དཱ་པཱིཏ། །སཱ་རཱ་ཛ་བཏྟཱི་བྷི་ར་ཏི་པྲ་ཡཏྟཱ་ད་བྷཱུ་ཙ་མཱ་ན་ཏྲི་ད་ཡང་རེག་འོས་ཉིད་མ་གྱུར། །རྒྱལ་པོའི་བཙུན་མོ་རྣམས་ཀྱིས་རབ་འབད་ནས། །མངོན་པར་མཆོད་ཅིང་ཧཱུྃ་ག་ནེ་བ། །ཏ་ཏྲ་སུ་མྲྀཏྟཱ་ནི་ཙིན་ཏ་ཡནྟི་པ་ཏི་བྲ་ཏཱ་ཀཱ་ལ་མུ་པཱ་ས་ལྷ་མོ་བཞིན་དུ་དེ། །ལེགས་སྤྱོད་སེམས་ཤིང་བདག་པོའི་བརྟུལ་ཞུགས་ཀྱིས། །དེར་ནི་ཅུང་ཟད་དུས་ཀཾ་ཙིཏ། །ཏཱཾ་བྲཧྨ་དཏྟོ་པི་བི་ཤིཥཱ་ཙཱ་རསྟ་ཏྲ་སྦི་ཏཱ་བྷཱུ་མི་པ་ཏིནྡི་ཤ་མྱ། །ཤིག་ཉེ་བར་གནས། །འབྲལ་བའི་སྡུག་བསྔལ་མོ་ཡིས་སྲེག་བྱེད་ཅིང་། །དེ་ནི་དེ་ན་གནས་པ་ས་ཡི་བདག། བི་ཡོ་ག་དུཁཱ་ན་ལ་ད་རྒྱ་མཱ་ནས་མཱ་ཡ་ཡོ་བྲཧྨ་ཎ་ཀུ་ཊ་བེ་ཤ། །ཨ་ཚངས་བྱིན་གྱིས་ཀྱང་ཉན་རྣ་བསྲིངས་ལས་ཐོས། །བྲམ་ཟེའི་ཆས་སུ་བཅོས་ནས་ཡང་དག་འོངས། །དེ་ཐ་བྲ་ཏཱཙྪཱདི་ཏ་ཙཱ་རུ་བེ་ཧཱུྃ་བི་ལོ་ཀྱ་ཏྭཾ་ཤཱི་ལ་ཡ་ཤཔ་ཏཱ་ཀཾ། །ནས་བརྟུལ་ཞུགས་མཛེས་པའི་ཆས་ཀྱིས་བསྒྲིབས། །ཚུལ་ཁྲིམས་གྲགས་པའི་བ་དན་དེ་མཐོང་ནས། །པྲ་སཱ་ད་ནཱ་ཡ་པྲ་ཎ་ཡཱ་བྷི་པཱ་རཱི་གྲོགས་པ༔སཱ་རཱུ་པཾ་པྲ་ཀ་ཊཱི་ཙ་ཀཱ་ར། །ཏཱི་དང་བའི་སླད་དུ་མི་བདག་གུས་པར་ནི། །མངོན་དུ་ཕྱོགས་པས་རང་གཟུགས་གསལ་བར་བྱས། །ས་—།
