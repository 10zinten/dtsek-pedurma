ཙིཏྟེ་སེ། །ན་མ་སྟེ་གྷོ་ར་སཾ་སཱ་ར་མ་ཀ་རཱ་ཀ་ར་སེ་ཏ་བེ། །ཛི་ནཱ་ཡ་ཛ་ཕྱག་འཚལ་ལོ། །རྒྱལ་བ་སྐྱེ་ཚོགས་སྐྱེ་བ་ཡི། །ཉོན་མོངས་རབ་ཏུ་ཞི་བའི་རྒྱུ། །འཇིགས་རུང་འཁོར་བ་ན་ཏཱ་ཛཆ་་གླེ་ཤ་པྲ་ཤ་མ་ཧེ་ཏ་བེ། །ན་མོ་ནི་ཏྱ་པྲ་བུདྡྷ་ཡ་སཏྠ་རྒྱ་མཚོ་ཡི། །ཟམ་སྟེགས་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་གཟི་བརྗིད་རྣམ་པར་དག། སྙིང་རྗེ་ས་ཏྲེ་ཀ་བནྡྷ་བེ། །བི་ཤུདྡྷ་དྷཱསྣེ་བུདྡྷ་ཡ་ཀ་རུ་ཎཱ་མྲྀ་ཏ་སིནྡྷ་བེ། །ཨི་བདུད་རྩིའི་རྒྱ་མཚོ་སྟེ། །སེམས་ཅན་ཀུན་གྱི་གཉེན་གཅིག་པུ། །རྟག་ཏུ་སད་ལ་ཕྱག་འཚལ་ལོ། །ཞེས་ཏི་བྷཏྟི་སུ་དྷ་སིཀྵ་སུ་ག་ཏ་ཐ་བ་ཌོ་ཙི་ཏྭཾ། །པུ་ཅུ་པུ་པྲ་སཱ་པ་བསོད་ནམས་མེ་ཏོག་ལས། །འཁྲུངས་ཤིང་གུས་པའི་བདུད་རྩིས་བྲན། །བསྟོད་པའི་དོག་པ་བདེ་བ་ནཱི་ས་ཙཀྲ་པཱུ་ཏི་མཾ་ཛ་རཱི་མ། །སརྦ་ཛྷ་སྟ་སྱ་བི་ཛྷ་ཡ་ཀཱ་ཡ་ཀྵེ་ཤ་གཤེགས་ཀྱི༄༅། །སྙན་དུ་འོས་པ་དེ་ཡིས་བྱས། །དེ་ཡི་ཉོན་མོངས་སྡུག་བསྔལ་གྱི། །རང་བཞིན་སྐབས་མ་ཡི་ད་ཤོ། །བནྡྷ་ནཱ་གཱ་ར་བི་བ་རཱ་ལོ་ཀི་རོ་སཱུ་ཡ་ནཾ་བྱ་དྷཱཏ། །ཨ་ཛཱ་མཁྱེན་ཀུན་མཁྱེན་གྱིས། །འཆིང་བའི་ཁང་པའི་བུ་ག་ཡི། །སྣང་བ་དག་གིས་ཚིམ་པར་བྱས། །དེ་ལྟར་ཏ་ཤ་ཏུལྟ་དྲྀཏྟཾཛཱ་ཏཱ་ཤཾ་ཀཱ་ཀུ་ལ༔པི་ཏུ། །ཨ་བཱ་ར་ཡ༴དྷཱ་ཤྲཱི་ཧེ་གྱུར་པ་མ་སྐྱེས་དགྲས། །ཤེས་ནས་དོགས་པས་འཁྲུགས་གྱུར་ཏེ། །ཕ་ཡི་བཙོན་ཁང་དག་གི་ནི། །སུ་སུདྨ་བི་བ་རཱ་ཏྱཱ་པི། །ཏ་ཏསྟ་སྱ་ཏ་དཱ་དེ་ཤཱཏ་ཀྲུ་ར་བནྡྷ་ན་རཀྴིཎ། བུག་པ་ཤིན་ཏུ་ཆུང་བའང་བཀག། དེ་ནས་གདུག་པའི་བཙོན་སྲུང་གིས། །དེ་ཡི་བཀའ་ལས་སྤུ་གྲི་ཡིས། །གླུ་རེ་ཎ་གཱ་རྟ་བནྣཱ་སྱ་པཱ་ད་ཡོསྟདྲི་ཀཎྜ་ནཾ། །ས་ཏི་བྲ་བེ་ཤ་ས་དམ་དུ་བཅིངས་པ་དེ་ཡི་ནི། །རྐང་པ་རྣམ་པར་བཅད་པར་བྱས། །དྲག་པོ་སྡུག་བསྔལ་ཉོན་མོངས་ཀློ་ཤ་བྱ་ཐི་ཏསཱརྨི་བཔ་ར། །ན་མོ་བུབྷྱཱ་ཡ་བུདྡྷཱ་ཡེ་ཏུནྟི་ཀྱིས། །མཆོག་ཏུ་གཟིར་བའི་ས་བདག་དེས། །སངས་རྒྱས་སངས་རྒྱས་ཕྱག་འཚལ་ཞེས། །གདུང་བའི་ས་ཀྲནྡ་བྱ་དྷ་ཏ། །བྷ་ག་བཱ་ན་ཐ་སརྦ་ཛྷཔུ་རཿཔྲ་ཏུ་ཀེ་ཏཾ་ག་ཏ། །ཆོ་ངེ་ལྡན་པས་བརྗོད། །དེ་ནས་བཅོམ་ལྡན་ཐམས་ཅད་མཁྱེན། །མདུན་དུ་མངོན་སུམ་ཉིད་གཤེགས་ཏེ། །—།
