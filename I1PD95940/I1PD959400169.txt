པོནྟེསྟསྨིན་བྷ་བ་ཏི་ས་ཧ་སཱ་པཾ་ནི་ཀཱ་ཤེ་ཡ་ཡཱངྒེ། །རཱ་གོདྷཱ་རཿསྤུ་ཊི་སུ་ཞིག་ཁོང་དེར་འཕྲལ་ལ་འཇུག་སྟེ་དཔེར་ན་བསྲུབས་པའི་མེ་བཞིན་ནོ། །ཤེལ་དུམ་དག་ལ་ཚོན་གྱིས་ཀ་ཤ་ཀ་ལཾ་མེ་གྷ་མནྡྷོ་དྷི་བཱ་ཐཔུཥྛ་མོ་དསྟི་ལ་ཀ་མི་བཱ་ཤ་ནིཀཱརྩ་ནཾ་བཱགཱི་ཏཱ་པ། །རབ་དབྱེ་སྤྲིན་ལ་ཆུ་ཡི་གཏེར་གྱི་ཆུ། །ཏིལ་ལ་མེ་ཏོག་དྲི་བཟང་གསེར་ལ་མེ་ཡིས་གདུངས་བཞིན་དལ་བུ་ཡིས། །སོ་ཡོ་གབྷཾ་བི་ཤ་ཏི་ནི་ཡ་ཏི་ར་སྱ་སཾ་ལགླུ་བྲྀཏྟིཐཱ་ཡུརྒནྟི་རི་བ་བ་དྲི་རྣམས་ཀྱིས་བཞིན་རྣམ་པ་མང་པོའི་ལས་ཀྱིས་བསྒོས་པར་གྱུར་པའི་རླུང༌། །སྤྱོད་ཚུལ་བྱེད་པོ་ཡིས་ཧུ་བི་དྷིཀརྨ་བྷིཧཱ་སྱ་མཱ་ན། །ཏསྨིན་སུཥྨ་ཀྲ་མ་པ་རི་ཎ་ཏིརྨི་ན་ཀྱང་ཡང་®དག་མི་མཚན་འདི་དེ་ནང་དུ་འཇུག། དེར་ནི་ཕྲ་བའི་རིམ་པས་ཡོངས་སུ་འགྱུར་ཞིང་རང་པརླུཥཱ་ཀ་སོ་ནཱ་ནཱ་ཀཱ་ར་པྲ་ཙུ་ར་ར་ཙ་ནཱ་ཙི་ཏྲ་རཱུ་པོ་སྱ་ལ། །བཞིན་གྱིས་བཙིར་འདིར། །ཇི་སྲིད་དུས་སུ་རྣམ་པར་འགྱུར་བ་མེད་པ་བཞིན་དུ་རྣམ་འགྱུར་འཛིན། །ཀཾ་ཙི་ཏ་ཀཱ་ལཾ་བ་ཧ་ཏི་བི་ཀྲྀ་ཏི་ནིརྦ་ཀཱ་རཱ་ཡ་མཱ་ནཱ་མཱ་ཡུརཱཎྜི་ཛ་ལ་མ་ཡ་རྣམ་པ་སྣ་ཚོགས་རབ་མང་བཀོད་པ་བཀྲ་བའི་གཟུགས་ནི་མི་མཚོན་ཏེ། །རྨ་བྱའི་སྒོ་ང་ཆུ་ཡི་རང་བཞིན་ཨི་བ་བྱལྟ་བྷཏྟིརྨ་ཡུ་རཿ། །ཏ་ཏཱ་དྷཱ་ནཱཏ་གྷ་ན་ཀ་ལ་ལ་ཏཱ་ཤ་མྱ་པ་བྱཱ་རྨ་བྱའི་རྣམ་པར་མི་གསལ་བཞིན། །དེར་ནི་མེར་མེར་ནུར་ནུར་ནར་ཨིནར་གོར་གོར་ལ་སོགས་གནས་བྱ་བ་སྣ་ཀྟ་ཏཱཙྭ་ཛ་ཋ་ནི་ཏེ་ནོ་ཎཱ་པ་ཙ་མཱ་ན། །པཱུཙླི་སྐབས་བཟུང་། །དེ་དང་དེ་དག་ལྟོ་ཡིས་བསྐྱེད་པའི་དྲོད་ཀྱིས་སྨིན་ཅིང་བཏང་བྱས་ནས། །ཟླ་བ་མཱ་སེརྣ་བ་བྷི་ར་ཐ་བཱ་བརྟ་ཀརྨ་ཀྲ་མོ་པོ་ཀླུ་ཤཾ་བྷུཀྟི་ཀི་མ་པི་བི་ཁ་མི་རྫོགས་པའམ་ཡང་ན་དགུ་ཡིས་འཁྱིལ་པའི་ལས་ཀྱི་རིམ་པ་འདི། །སྡུག་བསྔལ་འདུ་ཤེས་རབ་ལྡན་ཅི་ཡང་དུ༔ཁ་སཾ་ཙཱ་པྲ་བནྣ། །ཀཱ་ལྱ་བྲྀཏྟཱཏ་སྭཱ་ཡ་མི་བ་ཕ་ལཾ་བི་ཙུ་ཏཾ་ཀརྨ་རུང་བའི་མི་བཟད་ཉོན་མོངས་སྤྱོད། །དུས་སུ་ཡུ་བ་དག་ལས་འབྲས་བུ་རང་ཉིད་ལྟུང་བཞིན་ཉམངལ་སྨིན་པཱ་ཀཱཏ་ཏཏྟཱ་ལོཎྚི་ར་བི་ཧ་ཏ་ཛ་བིསྲེ་རི་ཏ༔བླ་ཏི་བཱ་ཏེ། །ལགྲུ་བྷུ་པས། །དེ་དུས་ལངས་པའི་མ་ཡི་རླུང་ནི་མི་དམན་མགྱོགས་པས་རབ་བསྐུལ་ནས། །མཚོན་ཆ་གོམས་།
