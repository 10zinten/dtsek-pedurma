ཥ་སཾ་སྐུ་རཱཏ་བནྡྷ་བྷོ་ག་སུ་ཁ་སྨྲ་ཏི། །ས་ཏ་མཱ་ཧུ་ཡ་བྷ་ག་བཱན་བི་ལོངས་སྤྱོད་ཀྱིས། །བདེ་བ་དྲན་པ་སྐྱེས་པར་གྱུར། །བཅོམ་ལྡན་འཛུམ་དང་བཅས་པ་དེས། །སྐྱེངས་ལགྷཾ་པྲ་ཧ་ས་སྨི་ཏ། །ཀོ་ཡཾ་པ་རི་བི་ཏརྐོསྟེ་པྲ་ཏི་སཾ་ལཱི་ན་ཙེ་ཏ་ས། །པ་དེ་བོས་བཀའ་སྩལ་པ། །སེམས་ནི་ཀུན་ཏུ་ཡོངས་ཞུམ་པ། །ཁྱོད་ཀྱིས་ཡོངས་སུ་རྟོག་འདི་ཅི་༅། །བི་ཤྲཱིཏྟཱ་ཏྱནྟ་ཀྲྀཏྟཱ་བཱ་ཏནཱི་བྷ་བ་ཏི་བི་སྭཱ་རཱ། །ས་མཱ་མཱ་དྷུ་ཏྲ་མཱ་ཡཱ་ཏི་དྲལ་བའམ་ཤིན་ཏུ་དྲངས་པ་ཡི། །རྒྱུད་མང་དབྱངས་དང་བྲལ་བ་ཡིན། །མཉམ་པ་ལས་ནི་སྒྲ་སྙན་ཏ་སྨཏ་ས་མཾ་ས་མཱ་ཐ་ཡེཏ། །ཨི་ཏྱཱ་དེ་ཤཱཏ་བྷ་ག་བ་ཏསཏྠ་སཱ་མྱ་མུ་འགྱུར། །དེ་སླད་མཉམ་པ་བསྟེན་པར་བྱ། །ཞེས་པ་བཅོམ་ལྡན་འདས་ཀྱི་བཀས། །ཕྱི་ནང་གདུང་བ་པ་ག་ཏ། །།ས་དྲྭ་པ་བི་མ་ལ་ཛྙཱ་ན་པཉྩཱཏྟ་པ་བི་པཌི་ཏ། །།ཏ་སྱ་ཏཱ་མདྡྷཱ་རྣམ་པར་སྤངས། །ཐམས་ཅད་ཞི་བ་ཉེ་བར་གྱུར། །དེ་ཡིས་དྲི་མེད་ཡེ་ཤེས་ཐོབ། །དེ་ཡི་རྨད་བྱུང་ཏྭཾ་སིདྡྷཾ་བི་ལོཀྱ་པྲ་ཐུ་བིསྨ་ཡཱ། །།ས་པྲཙྪབྷིཀྵ་བ་སཉྫ་བྷ་ག་བནྟཾ་ས་ཙཱ་གྲུབ་པ་དེ། །མཐོང་ནས་ཡ་མཚན་ཆེ་བ་ཡི། །དགེ་སློང་ཀུན་གྱིས་བཅོམ་ལྡན་ལ། །དྲིས་ཤིང་དེས་ཀྱང་བྱཱ་དྷཱ་ཏ། །ཐོ་ཎ་སྱ་ཤུ་ཡཾ་ཏྭཾ་ཐེ་ཡཀརྨ་ཛནྣཱ་ན་རཱ་རྫི་ཏཾ། །ན་རྒྱ་བཀའ་སྩལ་པ། །གྲོ་བཞིན་གྱིས་ནི་དགེ་བའི་ལས། །སྐྱེ་བ་གཞན་ལ་བསྒྲུབས་པ་ཉནི། །གང་ཕྱིར་པུ་ཎྱ་ནུ་བྷཱ་བཱ་ནཾ་བྷ་བནྟུདྡྷ་ཏ་སམྲ་ད། །།།བི་བ་ཤི་བྷ་ག་བཱན་བསོད་ནམས་མིན་པ་ཡི། །མཐུ་ལྡན་རྨད་བྱུང་ཕུན་ཚོགས་མིན། །སྔོན་ཚེ་ཡང་དག་རྫོགས་སངས་རྒྱས། །སམྱཀཾ་བུདྷུསུ་ག་ཏཔུ་རཱ། །པུ་རཱིཝནྣཱ་མ་ཏཱི་ནཱ་མ་ཛ་ན་ཙཱ་རི་བཅོམ་ལྡན་བདེ་གཤེགས་རྣམ་པར་གཟིགས། །གཉེན་ལྡན་ཞེས་པའི་གྲོང་ཁྱེར་དུ། །ལྗོངས་ནི་རྒྱུ་ཞིང་ཀ་ཡཱ་ཡ་ཡོ། །ཏ་ཏྲ་བྷཀྵེ་སུ་ཀྲྀ་ཏི་བྷིནྡྷཀྵ་ཡོ་བ་ནི་མཎྜི་ཏ། །བཱ་གཤེགས་པར་གྱུར། །དེར་ནི་ལེགས་བྱས་གུས་པ་ཡིས། །གཟན་ལ་ཉེ་བར་མགྲོན་དུ་གཉེར། །ཉིན་རེ་ཎ་པྲ་ཏྱ་ཧཾ་གེ་ཧཾ་ཡ་ཡོ་ཏེ་ཏཱཾ་ས་ཧཱ་ནུ་ག། །ཏ་ཏོ་ད་རི་དྲ་སཾ་པཱལྟ་རེ་རེས་ཀྱིས་དེ་དག་གི། །ཁྱིམ་དུ་རྗེས་འབྲང་བཅས་པ་གཤེགས། །དེ་ནས་དབང་པོའི་ཟླ་ཞེས་པ། །—།
