པ་ད་མ་མྲྀ་ཏཱ་་སཾ་ཏ་ཏཱཉྙོ་པ་ཀཱ་རི་ཀྲྀ་ཏ་པ་རི་ཀ་ར་བནྟཾ་པུ་ཅུ་གཟིར་བ་ལ་ཕན་བསོད་ནམས་གནས་ལ་ཡོ་བྱད་ནི། །བསྒྲུབས་བྱས་བདུད་རྩི་ཐོབ་པས་ཀྱི་བརྟན་པའི་གནས་དྷཱནྟི་པྲ་ཡཱ་ཏི། །ཨི་ཏུཀྵ་ཛ་ན་ཀོ་སཱ་སྲོ་པུ་རཿཔྲཏྟཾ་ཁ་གེ་ཤཱ་རཾ། །པཉྩ་སུ་འགྲོ་བར་འགྱུར། །ཕ་མ་མཆི་མ་ཅན་ལ་ནི། །བརྗོད་ནས་མདུན་དུ་འདབ་ཆགས་དབང་། །འགྱོད་ཏྭཾ་པེ་ན་མ་ཧ་ཏཱ་ནི༈ནྡྲ་ནི་ཛ་དུཀྲྀ་ཏཾ། །བི་རཱ་གྱ་གུ་རུ་ཎཱ་སཏྭཱ་པ་ཆེན་པོས་རང་གིས་ནི། །ཉེས་བྱས་སྨོད་ཅིང་གནས་པ་དག། དེ་ཡི་སྙིང་སྟོབས་འབད་པ་ནི། །པྲ་ཎ་ཡེ་ན་པྲ་མྲ་རྒྱ་ས། །ཙ་ཀཱ་ར་སྦི་ར་སཾ་ཀཝཾ་ནཱ་གཱ་ཤ་ན་ནི་བྲྀཏྟ་ཡེ། །ཏ་ཆགས་བྲལ་བླ་མས་དག་བྱས་ནས། །ཀླུ་ནི་ཟ་བ་ལས་ཟློག་ཅེལ། །ཀུན་ཏུ་རྟོག་པ་བརྟན་པོ་བྱས། །དེ་ཏཔརླན་བི་ཥ་མ་ཤཱ་ས་པྲསྐྲ་ལི་ཏཱ་ཀཱ་རཿ། །སཏྭཱ་པྲ་ཀཱ་ཤ་ཤེ་ཥོ་བྷུཏྠ་ནི་ནས་མི་བཟད་མཐའ་ཡི་དབུགས། །ཡི་གེ་རབ་ཏུ་འཁྲུལ་པ་ཅན། །མིག་ནི་རབ་ཏུ་ཟུམ་པ་དེ། །སྙིང་མཱི་ལི་ཏ་ལོ་ཙ་ན། །ཨ་ཐོ་ཙི་ཏྭཾ་སུ་ར་ཙི་ཏཱ་པུཏྟཾ་ཤུ་ཀོ་ཙི་ཏྭཾ་ཙི་ཏྭཾ། །སྟེབས་ོགསལ་བའི་ལྷག་མར་གྱུར། །དེ་ནས་སྲེག་ཁང་རབ་སྤྲས་པ། །འོས་ཤིང་མེ་ཏོག་གོས་ཀྱིས་སྤྲས། །པྲ་བེབྷུ་ཏཏྲི་ཡ་བ་ཧྥེ་ར་གྲ་མེ་བྱ་ས་མ་བྱཱ་དྷཱཏ། །བྷཀྵ་མ་ཡཱ་བྷ་ག་མ་ཡི་མདུན་དུ་མངོན་ཕྱོགས་ཏེ། །འཇུག་འདོད་དེ་ཡི་དགའ་མས་སྨྲས། །བདག་གིས་གུས་པས་མཉེས་བ་ཏཱི་ཏོ་ཏི་ཏཱ་བྷ་བ་བལྤ་བྷཱ། །ཨཱ་དི་ཤསྣེ་བ་རཾ་སཐཾ་བིདྷུ་བྱས་པ། །འབྱུང་པོའི་དགའ་མ་བཅོམ་ལྡན་མས། །རིག་འཛིན་ཀུན་གྱི་འཁོར་ལོས་སྒྱུར། །མཆོག་དྷ་ར་ཙཀྲ་བནྟི་ནཾ། །ཏ་དི་དཾ་པི་ཏ་ཐོ་ཀམྨཱཛྫཱ་ཏཾ་མ་མ་པ་ཏཱི་བ་ཙ། །ནི་བདག་ལ་རབ་ཏུ་བྱིན། །གང་ཞིག་ཞག་བདུན་ཉིད་ཀྱིས་བདག། བུ་མོ་ཡུགས་སའི་སྐལ་ལྡན་གྱུར། །ཡ་ད་ཧཾ་སཔ་རཱ་ཏྲི་ཎ་བཱ་ལ་བི་བྷ་བྱ་བྷཱ་གི་ནཱི། །ཛརཱནྟ་རེ་བི་མེ་བྷཱུ་ཡཱ་དག་པའི་ཚིག་འདི་བདག་ལ་ནི། །ཅི་སླད་བདེན་པ་མིན་པར་གྱུར། །བདག་གི་སྐྱེ་བ་གཞན་དུ་ཡང་། །ད་ཡ་མེ་བོ་ཙི་ཏཱཔ་ཏི། །ཨུཀྟི་ཏི་སཱགྷོ་བི་ཀླུ་པ་མཱན་ར་ཀུ་སུ་མཱཉྫ་ལི། །ཏ་འོས་པ་འདི་ཉིད་བདག་གྱུར་ཅིག། ཅེས་བརྗོད་དེ་ཡིས་མནྡ་ཅིརའི། །མེ་ཏོག་སྙིམ་པ་མེ་ལ་གཏོར། །དེ་།
