ཏཾ་བུདྡྷ་བི་ཙཱ་རེདྲ་ཡ་དྷིགྲུབྷཾ་ན་ཀྲྀ་ཏཔྲ་སཱ་དྷ་ན་བ་དྷིརླ་，ཐུ་ལུས་ཅན་རྣམས་ཀྱི་སྐྱེ་བ་འབྲས་མེད་དེ། །གང་གིས་བློ་གྲོས་ཐོས་པས་རྒྱན་གྱི་སྒྲུབ་བྱེད་རབ་འབར་བྱས་པ་ཏེ་ནོཛྫལླ། །ཀིཾ་ཏེ་ན་ཐུ་ཏ་བིསྟ་རེ་ཎ་ན་ག་ཏོ་ནིཇི་ནྱ་ཏཱ་ཡ༔ཤི་ཡཱ་ཀིཾ་ཤྲཱི་བི་མིན་པ་སྨད། །གང་གི་དཔལ་གྱིས་དབུལ་བ་ཉིད་མིན་མ་གྱུར་ཐོས་པ་རྒྱ་ཆེ་དེ་ཡིས་ཅི། །ཤིན་ཏུ་ནད་མེད་བྷ་མ་ཛིམྷི་ཏེ་ན་ནཾ་ཏ་རཱ་མཱ་རོ་གྱ་བྷོ་གྱོ་ན་ཡཏ། །ཨཱ་ནཱི་ཡ་ཏཱ་མེ་ཏཱ་ར་ཡཱ་ཀུ་མཱ་རཿཉིད་ཀྱིས་གང་ཞིག་སྤྱོད་མེད་དཔལ་གྱི་རྣམ་འཕྲུལ་རྒྱས་པས་ཅི། །བདག་གི་གཞོན་ནུ་སྐྱེ་རྒུ་དགའ་བ་ནི། །པྲ་ཛཱ་པྲ་ཡས་ཀ་ཤི་ལ་ནི་ཡུཀྟ། །པ་བྱཱ་མི་ཏསྨིནྡྲི་མ་ལེ་སུ་བྲྀཏྟེ་སཾ་ཀཱནྟརྨ་རྡོ་འཇོག་ཏུ་བསྐོས་མྱུར་དུ་དགུག་པར་བགྱི། །མེ་ལོང་ལ་བཞིན་དྲི་མེད་ལེགས་སྤྱོད་ཟླུམ། །དེ་ལ་རང་དན་ཨི་བ་སཱ་ར་ཚོ། ་ས་མཇི་ཏཱ་ད་གྲ་པི་ཏཱ་ཏ་པ་ཏྲ་ནི་བདྷ་མོ་གི་རྒྱལ་སྲིད་འཕོས་པ་ལྟའི། །ང་ཉིད་གུས་པས་དབུ་རྒྱན་བཅིངས་བྱས་ཤིང་། །གདུགས་དཀར་མཆོག་ལི་པྲ་ན་ཡཱ་ཡི་བ། །པ་ཊུ་ཏུ་ཏཾ་པུཎྱ་ར་པཱ་ཡ་ནི་ན་མཱ་མེ་བ་པོ་རསྟ་རུ་ནི་ཡང་དག་གཏད་པ་དེ། །གྲོང་པས་ང་ཉིད་བསོད་ནམས་བཅུད་ལེན་གྱིས། །གཞོན་ནུ་ཉིད་ཐོབ་བཞིན་ཎ་ཏཱ་མཱཙཾ། །ཨི་ཏྱཱཀྟ་མཱ་ཀནྡྷུ་ན་རེ་ཤཱ་རེ་ཎ་ཏཾ་ཏིཥྛ་རཀྵ་ནི་ཛ་གཱ་ད་ཛཱ་དུ་ལྟ་གྱུར་ཅིག། ཅེས་པ་མི་ཡི་དབང་པོས་སྨྲས་པ་དེ། །མཚུངས་པར་རབ་ཞུགས་འཇིགས་དང་མྱ་ངན་ཡཱ༑་། ཏུ་ལེ་པྲ་བྲྀཏྟེབྷ་ཡ་ཤཱ་ཀ་དི་རྒྱ་མཱཏྠརྨ་མོ་ཧིཔ་རི་པཱུརྨ་མཱ་དང་། །།དམན་དང་ཕྲག་དོག་རྨོངས་པས་ཡོངས་ཁེངས་པའི། །བཙུན་མོ་སྐར་རྒྱལ་སྲུང་བས་ཐོས་ནས་ནཱ༑། ཨ་ཧཾ་མ་ཧཱི་པཱ་ལ་ནི་རཱ་མ་ཡཾ་ཏཱཾ་ཀ་རོ་མི་མེ་པ་བྱ་བི་ཤེ་ཥི་ཡུཀྟི། །སྨྲས། །ས་སྐྱོང་ཁྱོད་ནི་ནད་མེད་བདག་གིས་བགྱི། །བདག་གི་རིག་པའི་ཁྱད་པར་བལྟ་བར་མཛོད། །ཡཾ་ཏཱ་སཱ་སི་ཀྵརྦ་ཀ་དརྦ་ཏཱཞྭ་ཛ་ན་གྷ་ཡཱ་ཡཱདྨ་ཛུ་ཥཀུ་བེ་ཏྲཿ། །མི་ཐོ་རང་གི་བསླབ་དོན་ནོར་ཉམས་སྐྱེ་བོ་རྣམས། །བརླག་སླད་ཚེ་བསྟེན་སྨན་པ་ངན་པ་སོང་། །རྟག་ཏུ་བི་ཥ་དི༔ཐུ་ཏ་གཏྠ་ན་དེསྟ་ཙཱ་པ་བཱ་དེ་ར་བུ་དྷཱ་ནུབཱ་དི། །ནིཟི་བརླག་བརྩོན་སྨན་པ་རིག་བྱེད་བྲལ། །ཕན་ཚུན་ཆགས་བྲལ་ཐོས་དྲེགས་བསྙེམས་པ་དང༌། །དེ་ཉིད་།
