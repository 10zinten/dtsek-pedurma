རྒྱ་ན་ནཱི་མ་མ་ནིརྨི་ཏཱ། །བྷ་གེརྨ་ཛན་ཀསྠཱ་བབློ་བ་ནཾ་ནི་དྷ་ནཾ་ག་ཏ། །བདག་གི་མ་འདི་ཅི་ཡི་སྦྲུལ། །རེ་ཞིག་སྐལ་བས་མ་®ནི་བདག། ལང་ཚོ་ལྡན་ཞིང་འཚོ་བར་གྱུར། །ཨི་ཡཾ་ཏཱ་པུ་ཙྪིཀཾ་ཀཱ་ལཱི་ཀཱ་ལ་བྱཱ་ལཱི་ཧུཾ་ཧེ་སྦི་ཏཱ། །གཙྪ་མེ་ཏྭཾ་ས་མུམྨཾ་བསོད་ནམས་མིན་པས་སྦྲུལ་ནག་ཅནི། །འཇིགས་རུང་མ་ནི་ཁྱིམ་ན་གནས། །གཉིད་དང་བྲལ་མ་འདི་གྱུ་ནི་ཏཱ་བི་ར་ཧི་ཏྭཾ་ཡ་དི། །ཏ་ད་སྱཀ་ལ་ཧཱ་མ་པིརྫ་ནི་པི་པྲ་ལ་ནི། །གལ་ཏེ་བདག་གིས་འགོམས་ཕྱིན་ན། །འདི་ཡི་འཐབ་མོའི་སྨྲ་བརྗོད་ཀྱིས། །སྐྱེ་བོ་རྣམས་ཏི་བུ་དྨ་ཏེ། །པཱ་ཡཀྵ་པཱ་པ་རི་ཀཱི་ཎཱ་དྷུ་རེ་ཀཱནྟཱ་ཀ་རོ་མི་ཀིཾ། ་། སུ་ཀྱང་རབ་སད་གྱུར། །མཚན་མོ་ཕལ་ཆེ་ཡོངས་ཟད་ཅིང་། །མཛེས་མ་རིང་ལ་བདག་ཅི་བྱེད། །སྐལ་ལ་བྷཱ་བྷཱ་གྱི་ཧཱི་ན་སྱ་ཀ་ཐཾ་སཱ་སུ་བྷ་ག་མ་མ། །ཧརྩོཏྠ་ག་ག་ཏཱ་བི་ལོ་ཀ་ན་ར་སརྣ་བ་དམན་པ་བདག་ལ་ནི། །སྐལ་བཟང་མ་དེ་ཇི་ལྟར་མོད། །།།ཁང་བཟང་ནང་དུ་སོན་ཞིང་རྣམ་པར་བལྟ་མཱ་ན་ནོ་བྷོ་རུ་ཧཱ་པྲསྟ་སྱ་ཤྲ་བ་ཎོཏྲ་ལ་སྱན་ཡ་ནཙྪཱ་ཡཱ་དི་ཤཾ་ཏཱི་པ་ཏེ། ཌ་། བའི་རོ་ལས་བཞིན་གྱི་ཆུ་སྐྱེས་དུད། །རྣ་བའི་ཨུཏྤལ་ལྷུང་བར་གྱུར་པས་མིག་གི་མཛེས་པ་རྣ་བ་དག་ལ་སྟེར། །ཏཱ་ཊཾ་ཀ་དྷུ་ཏི་བྷིཀ་པོ་ལ་ཡུ་ག་ལེ་པ་ཏཱ་བ་ལི་བི་བྷ་ཏཱི་བྷཱ་གྱིབྷོ་གྱ་བྷུ་བི་པྲ་རྣ་རྒྱན་འོད་ཟེར་དག་གིས་འགྲམ་པ་ཟུང་ལ་པ་ཏྲའི་འཕྲེང་ཅིབ་རྣམ་པར་འཛིན། །།མཛེས་མ་དེ་ནི་ལེགས་ཡ་ཏི་སུ་བྷ་གཱ་བྷ་བྱ་སྱ་སཱ་སུནྡ་རཱི། །ཨི་ཏི་ཙིནྟཱ་ཏུ་རཾ་དྷཱ་རཾ་ཡཱ་ཙ་ལྡན་དག་གིས་སྐལ་བ་རྣམས་ཀྱིས་སྤྱད་བྱའི་ས་རུ་འགྱུར། །དེ་ལྟར་བསམས་ཏེཔས་གཟིར་ཞིང་སྒོ། །མཱ་ན་པུ་ནཔུ་ན། །ཨ་ཏྲེ་བ་ཀྲི་ཡ་ཏྭཾ་ཤོ་ཙ་མི་ཏི་མཱ་ཏཱ་ཏ་མ་བྲ་བཱིཏ། ་་། ཨ་ཡང་དང་ཡང་དུ་སློང་བྱེད་པ། །དེ་ལ་མ་ཡིས་འདི་ཉིད་དུ། །གཙང་བ་མཛོད་ཅེས་རབ་ཏུ་སྨྲས། །དེ་ཐ་མཱ་ཏུཤི་རཊྚི་ཏྭཱ་ཁནེ་ན་པ་ཡ་ཡོ་ཛ་བཱཏ། །རཱ་གོ་པ་དིཥྛ་པཱ་པཱ་ནཾ་ཀི་ནས་རལ་གྲིས་མ་ཡི་མགོ། །བཅད་ནས་དེ་ནི་མགྱོགས་པར་སོང་། །ཆགས་པ་ཉེར་བསྟན་སྡིག་བདག་མ་ཀཱལཾ་དུ་རཱཏྨ་ནཾ། །རཱ་ཏཱ་བ་ག་ཎཾ་ཏཱ་ཡ་སས་ག་ཏཱ་པུརྡ་རཱི་ཉིད། །ངན་པ་རྣམས་ལ་བྱ་མིན་ཅི། །མཚན་མོ་ངལ་དུབ་མ་རྩིས་པརསེ། །།མཛེས་མའི་ཁང་པར་དེ་།
