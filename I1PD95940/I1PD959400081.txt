མ་ལཱ་ཐ་ཡ། །ཙཀྲ་བརྟི་ས་སུ་ཀྲྀ་ཏིཟིཙྩོ་ཀ་ར་ཨི་བཱ་བ་བྷོ། །ཏྲི་དཔལ་ལ་གནས། །འཁོར་ལོས་སྒྱུར་དེ་ལེགས་བྱས་ཀྱིས། །ཁྱབ་འཇུག་ལག་པ་བཞིན་དུ་མཛེས། །གྲགས་ཛ་གཛྫ་སྣ་བི་ཀཱིརྟེ༔པྲ་བྷཱ་བཱ་བྷ་ར་ཎཱ་ཤི་ཡ། །སོ་ཡཾ་སུ་ཀྲྀ་ཏ་བལླི་ནཱི་པ་འགྲོ་གསུམ་གང་གཱ་དང༌། །མཐུ་ཡིས་རྣམ་པར་བརྒྱན་པའི་དཔལ། །འདི་དེ་བསོད་ནམས་འཁྲི་ཤིང་གི། །པྲ་ཐ་མཀུ་སུ་མོངྒ་མ། །ས་ཀ་དཱ་ཙིཏྤ་ནཱནྟེ་ཥ་བི་ཀཱ་སི་ཀུ་སུ་མ་ཤི་ཡ། །མེ་ཏོག་རབ་འཐོན་དང་པོ་ཡིན། །དེ་ནི་ནམ་ཞིག་ནགས་ཀྱི་མཐར། །བློན་པོ་དག་དང་ལྷན་ཅིག་ཏུ། །རུ་ཙི་རཾ་ས་ཙི་བིསཱདྷམྤི་ཙ་ཙཱ་ར་བི་ལོ་ཀ་ཡན། །ད་དན་ཏ་ཏྲ་ནིཀྵཀུན་མེ་ཏོག་རྣམ་པར་རྒྱས་པའི་དཔལ། །མཛེས་པ་ལྟ་ཞིང་རྒྱུ་བར་གྱུར༄༅། །དེར་ནི་འདབ་མ་མེད་པའི་བྱ། །བི་ཧ་གཱན་པཱ་ད་ཙཱ་རི་ཎ། །བྱོ་མ་མཱརྒ་ག་ཏི་སྨྲ་ཏྭཱ་པྲ་ཡཱ་ཏཱན་ཀྲྀ་ཤ་ཏཱ་མི་རྐང་པས་འགྲོ་རྣམས་མཐོང་གྱུར་ཏེ། །མཁའ་ཡི་ལམ་བགྲོད་དྲན་པ་ཡི། །ཤིན་ཏུ་རིང་བར་གྱུར་པ་བ༑། པཀྵ་ཧཱི་ནཱ་ན་ག་ཏི་ཀཱན་བྲྀཏྟིཾ་ཀྱི་ཎནྣི་རམྦ་རཱན། །ད་རི་་་བཞིན། །འདབ་མས་དམན་ཞིང་བགྲོད་བྱེད་མིན། །འཚོ་བ་ཟད་ཅིང་ལུས་གཡོགས་མེད། །དེ་རྣམས་དཱ་ནི་བ་ཏཱན་བཱིགྱུ་པྲ་བ་ཙ་ཀྲྀ་པ་ཡ་ཚྭ་པ། །ཨ་ཧོ་བཱ་རཱ་ཀི་བི་ཧཱ་གིཀི་དབུལ་པོ་བཞིན་མཐོང་ནས། །མི་བདག་བརྩེ་བས་རབ་སྨྲས་པ། །ཀྱེ་མ་སྡུག་པའི་བྱ་འདི་རྣམས། །ངན་མི་ཏིཀུ་ཀྲྀ་ཏཾ་ཀྲྀ་ཏཾ། །ཡ་དེ་ཏེ་པཀྵ་བི་ཀ་ལཱཀྲྀཙྪཙྩ་ར་ཎ་ཙཱ་རི་ཎ། །ཨི་པའི་བྱ་བ་ཅི་ཞིག་བྱས། །གང་ལས་འདི་དག་སྒྲོ་དང་བྲལ། །དཀའ་ལས་རྐང་པས་རྒྱུ་བར་གྱུར། །སྙིང་ཏུལྟེ་བྷཱུ་མི་པ་ཏི་ནཱ་ཀ་རུ་ཎཱ་ཀུ་ལི་ཏཱཏྨ་ནཱ། །ཕུ་རསྦི་ཏོ་མ་ཧཱ་མཱ་ཏྱརྗེས་འཁྲུགས་པའི་བདག་ཉིད་ཅན། །ས་བདག་གིས་དེ་བརྗོད་པ་ལ། །བློན་པོ་ཆེན་པོ་བདེན་པའི་སྡེ། །ས་ཏྱ་སེ་ནསྦ་མ་བྲ་བཱིཏ། །ཐུ་ཏ་མེ་ཏད་མ་ཡཱ་ད་བ་ཀ་ཏྱཱ་མཱ་ནཾ་བ་ནེ་ཙཱ་རེ། །མདུན་དུ་འདུག་སྟེ་དེ་ལ་སྨྲས། །ལྷ་གཅིག་འདབ་ཆགས་རྣམས་ཀྱི་འདབ། །གང་ལས་བྱི་བར་གྱུར་པའི་རྒྱུ། །ཀཱ་ར་ཎཾ་པཀྵ་པ་ད་ནེ་ཡ་ད་བྷཱུཏྲཤྲཱི་ཎཱ་མི་ཧ། །སནྟི་པཉྩ་ཤ་ཏཱ་རྱ་ཏྲ་ནགས་ན་སྤྱོད་རྣམས་གླེང་གྱུར་པ། །དེ་དག་བདག་གིས་རབ་ཏུ་ཐོས། །དཀའ་ཐུབ་ཁ་ཏོན་ནུས་ལྡན་པ། །—།
