པ་རི་ཀླུ་ཡ་གུ་ཎ་ཀལྤ་ན་ཡ་བྷ་བ་ཏི་སུ་པུ་ཏྱཱ་བི་ཤེ་ཥ། །ཨ་ཐཱ་བྱ་ཏཱི་ཏེ་ཏན་བརྟག་པ་ཡོངས་སུ་མི་ཟད་པས། །བསོད་ནམས་དེ་ནི་ཁྱད་པར་གྱུར་པ་ཡིན། །དེ་ནས་འགའ་ཞིག་ཀསྨིནྟི་ཏ་ཀཱ་ལེ་བྷ་ག་བ་ཏོཉནྟི་ཀེ། །ནནྡོ་པ་ནནྡ་ཡོདྷརྨ་དེ་ཤ་ནཱ་ཡཱ་ཤོ་ཏུ་འདས་དུས་ན། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་ནི། །ཆོས་སྟོན་པ་ལ་དགའ་བོ་དང༌། །ཉེ་དགའ་ཉན་དུ་མཱཥ་ཡོ། །རཱ་ཛཱ་པྲ་སེ་ན་ཛི་ཏྲཧཱུཾ་བྷ་ག་བནྟ་མུ་པཱ་ག་ཏ། །ཏ་བྷྱཱ་མ་ཀྲྀ་་འོངས་པའི་ཚེ། །རྒྱལ་པོ་གསལ་རྒྱལ་བཅོམ་ལྡན་ལ། །བལྟ་བའི་སླད་དུ་ཉེ་བར་འོངས། །དེ་དག་གིས་ནི་ཏ་སཏྟཱ་ར་པྲ་ཎ་མ་ཀོ་ས་མཱ་ཡ་ཡོ། །ས་པྲ་ཎ་མྱ་ཛི་ནཾ་ག་ཏཱ་པྲ་ད་བསྙེན་བཀུར་དང་། །ཕྱག་མ་བྱས་པས་ཐོས་པར་གྱུར། །རྒྱལ་ལ་ཕྱག་འཚལ་དེ་སོང་ནས། །དེ་དག་དྷོ་ནི་གྲ་ཧཾ་ཏ་ཡོ། །ཤསྔ་བྲྀཏྟིཾ་སཾ་ཏྲཾ་ཛ་ཏསྟེ་ཙ་ཏསྨི་བྱོ་སྣ་ག་ཏོ། །ཆད་པས་གཅད་པར་བསམས། །དེ་དག་གིས་ཀྱང་མཁར་སོང་སྟེ། །དེ་ལ་མཚོན་གྱི་ཆར་པ་ཕབ། །སརྦ་ཛྷ་པྲ་ཥི་ཏསྣུཎྞཾ་མོངྒ་ལྱཱ་ཡ་ན༔པྲ་ཏ་ཏྭཾ། །ཤསྔ་བྲྀཏྟཾ་ན་ར་པ་ཏེ་ཀུན་མཁྱེན་གྱིས་བསྐུལ་མགྱོགས་པར་ནི། །་མོ་གལ་གྱི་བུ་སོང་ནས། །མི་བདག་གི་ནི་མཚོན་ཆར་དེ། །ཉཀྲེ་པདྨོཏྲ་ལཱ་པ་ལཾ། །པུ་ནནྣཏྭཱ་བྷ་ག་བ་ཏས་མཱི་པཾ་པྲི་ཐི་བཱི་པ་ཏི། །པདྨ་ཨུཏྤལ་ཕྲེང་བར་བྱས། །སླར་ཡང་ས་ཡི་བདག་པོ་ཡིས། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་ཕྱིན། །སཾ་པྲཏྟིཾ་ཀཱ་མཱ་ཡ་མཱ་ས་ཏ་དཱ་དེ་ཤཱཏ་པ་ཎཱི་ཤྭ་རོ། །ཨ་ཐཱནྡི་ཏ༔སཱརྫི་གདེངས་ཅན་དབང་པོ་འོངས་པ་ལ། །དེ་ཡི་བཀའ་ཡིས་བཟོད་གསོལ་བྱས། །དེ་ནས་ས་བདག་གིས་བེ་ན་བྷ་ག་བཱན་རཱ་ཛཱ་མནི་རཾ། །བྷཀྟི་པཱུ་ཏཾ་ཡ་ཡོ་བྷོཀྟ་བྷཏྟཾ་གསོལ་ཏེ། །རྒྱལ་པོའི་ཁང་པར་བཅོམ་ལྡན་འདས། །དགེ་སློང་ཚོགས་བཅས་གུས་པ་ཡིས། །གཙང་མའི་བྷིཀྵུ་ག་ཎེས་ཧ། །བྷཀྵེ་ཥཱ་པ་ཙྩ་མཱ་ནེ་ཥཱ་རཱ་ཊོ་ཏ་ཏཱཀི་བི་སླ་པ། །བཤོས་ནི་གསོལ་དུ་གཤེགས། །ཁ་ཟས་འཚེད་པར་བྱེད་ཚེ་དེར། །མཚན་མོ་མེ་ཡི་ཉེར་འཚེ་སྐྱེས། །ཛཱ་ཏོ་ཛི་ན་པྲ་བྷཱ་བེ་ན་ས་ཧ་སཱ་ཤཱནྟི་མཱ་ཡ་ཡོ། །བྷཱུཏྟཱ་ག་ཏི་བྷ་ག་བ་ཏི་གི་འཕྲལ་ལ་རྒྱལ་བའི་མཐུ་ཡིས་ནི། །རབ་ཏུ་ཞི་བ་ཉིད་དུ་གྱུར། །བཅོམ་ལྡན་གསོལ་ནས་གཤེགས་པའི་ཚེ། །—།
