ཡི་ཏེ་ཛ་ནེ། །ཡཱ་ཏྱཱ་ཡཱ་ཙ་ཀ་ཏུ་ལ་ཏཱ་པིཎྜཙྪེ་དོརྫི་ནཱ་མ་བྷུཏ། །བྷིཀྵ་ཏྭཾ་སཾ་ཟད་བྱེད་པ། །བསླང་བྱ་སློང་བྱེད་མཚུངས་པ་ཉིད། །སློང་བའི་འཚོ་བ་ཆད་པ་བྱུང་། །དགེ་སློང་རྣམས་ཀ་ཊེ་ཏ་སྨིན་ཀཱ་ལེ་བ་ར་མ་དཱ་རུ་ཎེ། །པཎྜི་ཏ་སུ་ག་ཏཱ་ཧུ་ཏཔྲ་ཏསྦེ་ཛེ་ནི་ཉམ་ང་བ། །མཆོག་ཏུ་མི་བཟད་དུས་དེར་ནི། །བདེ་བར་གཤེགས་པས་མཁས་པ་ནི། །བོས་ནས་རྒྱལ་ཏ་ཀཱ་ན་ནཾ། །ཏཾ་པྲ་བྲ་ཛནྟཾ་ཏུ་རེ་གེ་ན་ཀཱཉྩ་ནཱ་བླ་ན་ཤོ་བྷི་ན། །ཨུ་ཙུརྦ་བྱེད་ཚལ་དུ་ཆས། །གསེར་གྱི་ཕྲེང་བས་སྤྲས་པ་ཡི། །རྟ་ཡིས་བགྲོད་པར་བྱེད་དེ་ལ། །ཡོན་ཏན་རབ་ཊཱ༔ས་མ་བྷེཏུ་གུ་ཊོཏྠ་ཧ་ས་ཧིལྦུ་བ། །ཨརྦ་སརྦརྦ་ན་ཀལྤ་བྲྀཀྵརྒྱས་མི་བཟོད་པའི། །གཡོ་ཅན་སྔོན་དུ་ཕྱོགས་ནས་སྨྲས། །སློང་བའི་དོན་མཐུན་སློང་བ་ཡིན། །དིགླུ་བི་ཐུ་ཏ། །ཤ་ཏཱ་ནི་པཉྩ་སཾ་པྲསྨཱབྷྱཱ་མུདྲི་བྱཱ་ཤ་ཡ་དཔག་བསམ་ཤིང་ཁྱེད་ཕྱོགས་སུ་གྲགས། །བདག་ཅག་ལྔ་བརྒྱ་ཁྱོད་ལ་ནི། །བསམ་པ་གཏད་ལ་ཡང་བ་ཡཾ། །ཨ་ལཾ་ཀཱ་རཾ་ཤུ་ཀ་ཡུ་གཾ་པྲ་ཏེ་ཀ་ནས་མཱི་ཧི་ཏཾ། །ཨ་དྷུ་ནི་བཱ་བི་ལམྨེ་དག་འོངས། །རྒྱན་དང་གོས་ནི་ཟུང་རེ་དག། བདག་ཅག་སོ་སོ་ཡང་དག་འདོད། །གལ་ཏེ་ཡོད་ན་ད་ལྟ་ན་དཱི་ཡ་ཏྭཾ་ཡ་དི་ཤཱཀྱ་ཏེ། །ཨི་ཏྱཱཀྟསྟེས་དཱ་ཙཱ་རཿསོ་བ་ཏཱིརྨ་ཏུ་རཾ་ཉིད། །ཐོགས་པ་མེད་པར་གནང་བར་མཛོད། །དེ་ཡིས་འདི་བརྗོད་དམ་པ་ཡི། །སྤྱོད་ལྡན་དེ་ནི་རྟ་ག་མན། །སཱ་དྷུ་པཱུ་ཛ་བི་དྷཱ་ཡེ་ཊཱ་དཱི་མཱནྟཱ་ཌ་མ་ཙི་ཏ་ཡ་ཏ། །བྷ་ག་ལས་བབས། །དེ་རྣམས་ལེགས་པར་མཆོད་བྱས་ནས། །བློ་དང་ལྡན་པས་སྐད་ཅིག་བསམས། །གལ་ཏེ་བནྟ་མ་དྲྀཥི་བ་གཙྪཱ་མི་སཱ་ཤྲཱི་ཧཾ་ཡ་དི། །ཨཱ་སནྣཱ་མྲྀ་ཏ་པཱ་ན་སུ་ཏཾ་བིགྷཾ་བཅོམ་ལྡན་མ་མཐོང་བར། །རང་གི་ཁྱིམ་དུ་ཕྱིན་པ་ན། །བདུད་རྩི་འཐུང་ལ་ཉེ་བ་ཡིས། །གེགས་དེ་ཀ་ཐ་མུདྨ་ཧེ། །ཨ་ད་ཏཱ་པྲ་ཡ་མཇི་སྣ་བྲ་ཛཱ་མི་ཡ་དི་ནིསྡུ་པ། །ཀ་བདག་གིས་ཇི་ལྟར་བཟོད། །སློང་རྣམས་འདོད་པ་མ་བྱིན་པར། །གལ་ཏེ་ཁྲེལ་མེད་ཕྱིན་པ་ན། །སྦྱིན་ཐཾ་ཀ་རོ་མི་དཱ་ན་སུ་ཏྭཾ་སྭ་ཡཾ་པྲ་ཏ་ཁཎྜ་ནཾ། །ཨི་ཏི་ཙིནྟ་ཡ་ཏསྟ་སྱ་བྷི་ཏཱ་བྷཱུ་པའི་བརྟུལ་ཞུགས་རང་ཉིད་ཀྱིས། །ཉམས་པ་འདི་ཡང་ཇི་ལྟར་བྱ། །དེ་ནི་དེ་ལྟར་སེམས་པ་ན། །ཀླུ་ཡི་།
