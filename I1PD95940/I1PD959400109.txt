ཏ་སྭཱ་བྷཱུཏ་སུ་པྲི་ཡོ་ནཱ་མ་སོ་ཛཱ་ནྱ་ནི་ལ་ཡ༔སུ་ཏ། །པྲ་ཡ་ཡོ་ཡཾ་ས་མཱ་ཤི་དེ་སྲས་ཤིན་ཏུ་དགའ་ཞེས་པ། །སྐྱེ་བོ་ལེགས་པའི་གནས་སུ་གྱུར། །གང་ལ་བརྟེན་པའི་ཡོན་ཏན་ཚོགས། །ཏུ་སཱརྦ་ཀྲྀ་ཏཱརྦ་ཏྭཾ། །དཱ་ནཾ་ཤཱི་ལ་་མཱ་བཱིརྨ་་ན་པྲཛྙཱ་ས་མ་་དོན་བྱས་ཉིད་དུ་རབ་ཏུ་གྱུར། །སྦྱིན་པ་ཚུལ་ཁྲིམས་བཟོད་པ་དང་། །བརྩོན་འགྲུས་བསམ་གཏན་ཤེས་ནཱི་ཏ། །དྷཱ་ཏཱ་བི་ལོ་བྷ་ནཱ་ཡི་བ་ཡ་ཀྲྀ་ཏསུ་ཀྲྀ་ཏ༔ཤི་ཡེ། །ཏཾ་སརྦ་བི་རཱུ་རབ་ལྡན། །བྱེད་པོ་རྣམ་པར་ཆགས་སླད་གང་། །ལེགས་བྱས་དཔལ་དུ་བྱས་པ་བཞིན། །དེ་ནི་དྲི་མེད་བི་ཥ་དཱཀ་ལཱཉྩ་བི་པུ་ལཱ་ཤ་ཡཾ། །བི་བི་ཤུས་རཱ་སོ་དཱ་རཱ་མ་ཧོ་ད་དྷི་མི་རིག་ཀུན་དང་། །སྒྱུ་རྩལ་ཤིན་ཏུ་ཡངས་པར་ཡང་། །རོ་ལྡན་ཡིད་འོང་ཆུ་གཏེར་ཆེར། །ཆུ་བོ་བཞིན་དུ་བཱ་པ་གཱ། །གུ་ཎཱ་ལཾ་ཀྲྀ་ཏ་ཙཱ་རི་ཏྲ་ལཀྵ་ཎཱ་ལཾ་ཀྲྀ་ཏཱ་ཀྲྀ་ཏི། །པུ་རུ་ཞུགས་པར་གྱུར། །སྤྱོད་པ་ཡོན་ཏན་གྱིས་བརྒྱན་ཅིང་། །ལུས་ནི་མཚན་གྱིས་བརྒྱན་པ་གང་། །སྐྱེས་བུ་ཥིཏྟ་མ་ལུབྷེ་བ་ཡཾ་བྷཱ་རྒྱུ་ཤྲཱི་རརྫི་ཤི་ཡཏ། །ཀཱ་ལེ་ན་སུ་ཀྲྀ་ཏ་མཆོག་ལ་ཆགས་པ་བཞིན། །བསྔགས་པར་འོས་པའི་དཔལ་གྱིས་བསྟེན། །བསོད་ནམས་ཀྱིས་མྱོས་མཐོ་ཀྲི་ཏཾ་སི་ཏ་རི་ཏྲི་དི་པཾ་ག་ཏེ། །ཙཀྲེ་སྐནྡྷ་ཏ་ཊེ་ཏ་མྱ་བྱ་བ་ཧཱ་ར་བྷཱ་རིས་སུ། །ཡབ་ནི་དུས་ཀྱི་གཤེགས་པའི་ཚེ། །དེ་ཡི་ཐ་སྙད་ཁུར་དག་ནི། །ཕྲག་པའི་ངོས་ལ་གནས་རསྦི་ཏི། །སོ་ཙིནྟ་ཡ་དི་ཡཾ་ལགླིབི་པུ་ལཱཏྨ་ཀྲ་མཱ་ག་ཏཱ། །ཏ་ཐཱ་པི་པར་བྱས། །དེ་བསམས་བདག་གི་དཔལ་འབྱོར་འདི། །རིམ་པས་འོང་ཞིང་@ཡངས་པ་སྟེ། །དེ་ལྟ་ན་མ་རྩེ་བརཱབྷཱ་ན་སཧཱ་མ་ནོ་ར་ཐེ། །ཀིཾ་ཏཱ་ཡཱ་སུ་མ་ཧཱ་ཏྭ་པི་ཤི་ཡང་སློང་ཀུན་གྱིསི། །རེ་བ་ཡོངས་སུ་མ་རྫོགས་སྙམ། །སྔར་འོངས་སློང་བས་ཟོས་ཉིད་ཚེ། །གང་ཡ་སཏྠུ་རུ་ཥ་སྦྲ་ཡཱ། །པཱུརྦ་ག་ཏཱཟི་བྷུཀྟ་བ་ཡཱ་ཤེ་ཥཱཟི་ཥཱ་ནི་སྤ་ལཱ། །རཏྣ་ཀ་ཞིག་ཕྱིས་སློང་འབྲས་བུ་མེད། །སྐྱེས་བུ་དམ་པ་ལ་གནས་དཔལ། །ཤིན་ཏུ་ཆེ་བ་དེས་ཀྱང་ཅི། །རྟ་གདོང་ར་སྱ་བི་བུ་ལཾ་ནིཋ་ལཾ་བེ་དྷ་སཱ་ཀྲྀ་ཏཾ། །ཨ་རཱུ་པི་པཱུ་རི་ཏོ་ཡེ་ན་སློང་བ་གཅིག་དེ་ཡང་། །གང་གིས་ད་དུང་ཡོངས་ཁྱིམ་རྫོགས། །རིན་ཆེན་འབྱུང་གནས་བྱེད་པོ་ཡིས། །—།
