ན་སཱ་ཛ་ན་ཨི་བ་ལབྷོ་ན་བ་ས་ར། །ཨི་ཏུཏྟཾ་ན་ར་ནཱ་ཐེ་ན་ཧྲི་ད་ཡ་ཚོགས་ལ་ལྷག་པར་བསྟར་བྱས་འདི་འབྲས་བྲལ། །དེ་སྐད་མི་མགོན་གྱིས་བརྗོད་པ། །བློ་ལྡན་སྙིང་སྤྱནི་དྷི་མ་ཏྭཾ། །བ་ཙ་ཐུ་ཏྭཱ་མ་ཧཱ་མཱ་ཏཿཔྲ་ཏྱ་བྷཱ་ཥ་ཏ་བྷཱུ་པ་ཏི། །རཱ་ཛསྣ་ལ་རེག་པའི་ཚིག། ཐོས་ནས་བློན་པོ་ཆེན་པོ་ཡིས། །ས་ཡི་བདག་ལ་ལན་གསོལ་པ། །རྒྱལ་པོ་ཁྱོད་ཀྱི་བྷི་ན་བ་བློ་ཀསྟ་བེ་བ་བྷུ་བ་ན་ཏྲ་ཡེ། །གཱི་ཡ་ཏེ་བི་བུ་དྷཱ་དྷི་ཤཀི་མ་ཚིགས་བཅད་ཉིད། །མངོན་པར་གསར་པ་སྲིད་གསུམ་ན། །སྒྲོགས་ན་མཁས་པའི་དབང་ཕྱུག་དག། སཱུཀྟ་བིསྟ་རི། །ཏཱ་ཡི་བི་，་བི་ནོ་དེསྨིནྣ་དཱ་རྒྱ་བ་ར་པཱ་ལེགས་བཤད་རྒྱས་པ་གཞན་གྱིས་ཅི། །གུས་ལྡན་གཏོང་ཕོད་མཆོག་གྱུར་ཁྱོད། །རིག་པ་དག་ལ་དགྱེས་ད་རེ། །བི་བྱཱ་དྷ་ར་པུ་རཾ་སཏྠ་ཛཱ་ཏ་མེ་ཏརྨ་ཧཱི་ཏ་ལཾ། །ཀ་ལཱ་ཀ་མ་པ་འདིར། །ས་ཡི་མཐིལ་འདི་ཐམས་ཅད་ནི། །རིག་པ་འཛིན་པའི་གྲོང་ཁྱེར་གྱུར། །ཁྱོད་ནི་ཡོན་ཏན་ལི་ནི་ཀནྟེ་མི་ཏྲེ་གུ་ཎ་བ་ཏྭཾ་ཏཱ་ཡི། །ཨུ་དི་ཏེ་ཡཱ་ཏི་པཱ་ལོ་ཀསནྣཱཧྥེ་ལྡན་པའི་གྲོགས། །སྒྱུ་རྩལ་པདྨོ་ལྡན་པས་མཛེས། །ཤར་ཚེ་སྐྱེ་བོ་མཐའ་དག་ནི། །སྣང་བཅས་ལེགས་ཎཱ་ཁི་ལོ་ཛ་ན། །སཱ་སཱ་ཀ་ལཱ་ས་ས་བི་ལཱ་ས་བི་ཤེ་ཥ་བེ་གསྟེ་ཏེ་གུ་ཎ་པའི་ལམ་ནས་འགྲོ། །གང་དང་གང་ལ་རྒྱལ་པོ་རྗེས་སུ་ཆགས་པར་སྐད་ཅིག་ནི། །གུས་པའི་ཆ་ཤས་བྱས་ས་ས་ཛ་ནཉྩ་ར་ཏཾ་ཙ་ཏཏྟཏ། །ལོ་ཀེ་པྲ་ཡ་ཏྲཱ་མ་དྷི་རོ་ཧ་ཏི་ཡ་ཏྲ་ཡ་ཏྲ་རཱ་ཛཱ་པའི་སྒྱུ་རྩལ་དག་ནི་དེ་དེ་དང་། །རྩེ་དགའི་ཆས་ཀྱི་བྱེ་བྲག་དེ་དེ་ཡོན་ཏན་དེ་དེ་དང་། །སྐྱེ་བོ་སྤྱོད་ཚུལ་ཀ་རོ་ཏི་པ་ར་པཀྵ་ཎ་མཱ་ད་རཾ་ཤཾ། །བྷཱུ་པཱ་ལེ་བི་དུ་ཏི་ཤྲཱི་ཤ་ཏྲ་ཏི་ཤ་ཡཾ་བི་རྩྭ་དེ་དེ་འཇིག་རྟེན་དགའ་བ་ཉིད་དུ་འགྱུར། །ས་སྐྱོང་མཁས་ན་རིག་པའི་རྩེ་དགའི་སྤྲོ་ལ་མཆོག་ཏུ་རེག་པརབི་ལ་པོཏྠ་ཐཾ་པཱུ་རེ་སཾ་ག་ར་རཾ་ག་སཾ་ག་མ་རུ་ཙི་གྲོསྨཱ་ཏི་བཱི་ར་བྲ་ཏཾ། །མཱུ་དྷེ་གྱུར། །དཔའ་ན་གཡུལ་གྱི་རོལ་པ་དང་འགྲོགས་བརྟུལ་ཞུགས་བརྟུལ་ཞུགས་འཛིན་པར་འགྱུར། །བླུན་མུ་རྒྱ་ཏི་ཙཉྩ་ལེ་བི་ཙ་ལ་ཏི་ཀླུ་རེ་ནི་གཽ་པཱ་ཡ་ཏེ་ཡདྨདྷཱུ་མི་པ་ཏིཀ་རོ་ཏི་ཀུ་རུ་ཏེ་ན་རྨོངས་ཤིང་གཡོ་ན་རྣམ་པར་གཡོ་ལ་གདུག་ན་མ་རུངས་སྤྱོད། །གང་གང་ས་བདག་བྱེད་པ་དེ་དེ་མཐའ་—།
