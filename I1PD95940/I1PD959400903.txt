བྱེ་བ་བརྒྱ་སྟེར་བ། །གང་གི་ལུགས་ལ་ངེས་པར་གྱུར། །ས་ཏྭཱ་མེ་བཱཀྵ་ཡ་ཏ་རཱཀོ་ཤཀུ་ཤ་ལ་ཤཱ་ལི་ནཾ། །སཱམྦྷ་ཛུ་བྷུ་རི་བི་བྷ་བ༔དགེ་བ་ལྡན་པའི་སྙིང་སྟོབས་ཉིད། །ཤིན་ཏུ་ཟད་པ་མེད་པའི་མཛོད། །རྒྱལ་སྲིད་ཆེ་ལ་འབྱོར་མང་དེས། །ཥཊཱི་ཤདྷིསཾ་བཏྠ་རི། །ཧེརཿཥཎྜ་བ་ཏི་ཀོ་ཊཱི་རྡ་དོ་སཾ་གྷཱ་ཡ་ས་ཏཱདྷཱི། །ཏ་སྙིང་སྟོབས་བློ་ཡིས་དགེ་འདུན་ལ། །ལོ་ནི་སུམ་ཅུ་དྲུག་གི་གསེར། །བྱེ་བ་དགུ་བཅུ་རྩ་དྲུག་ཕུལ། །དེ་ཏཀཱ་ལེ་ན་ནི་པ་ཏི༔་དྷི་ནཱ་གཱ་ནི་མཱ་ཡ་ཡོ། །སྦེ་རཱ་ཎི་སུ་ཀྲྀ་ཏཱ་རྩེ་བ་ནས་དུས་ཀྱིས་མི་བདག་ནི། །ནད་ཀྱིས་ཐེབས་པ་ཉིད་དུ་གྱུར། །ལུས་ཅན་རྣམས་ཀྱི་ལེགས་བྱས་ཉིད། །ན་ཤ་རཱི་རཱ་ཎི་དེ་ཧི་ནཾ། །ཏ་ནུ་མཱ་བསྣ་པརླནཾ་ནིཉྩི་ཏྱ་པྲི་ཐི་བཱི་པ་ཏི། །བརྟན་པ་ཡིན་གྱི་ལུས་ནི་མིན། །ལུས་ནི་མཐའ་ལ་ཉེ་བ་ཉིད། །དེས་ནས་ས་ཡི་བདག་པོ་ཡིས། །བྱ་བྷིཥྛ་བྱཱཀུཀྐུབྷྱཱ་རཱ་མེ་ཏྲ་བི་ཎི་དཱ་ཏུ་མུ་བྱ་ཡོ། །ཏཊོ་ཏྲས་དཱི་གག་ཀུན་རར་དགེ་སློང་ལ། །ནོར་ནི་འབུལ་བར་རྩོམ་པར་གྱུར། །དེ་ཡི་ཚ་བོ་ཕུན་ཚོགས་ཅན། །ནཱ་མ་ལོ་བྷཱརྦསྟ་སྱ་ཤཱ་ས་ནཾ། །དཱ་ནཱ་བུ་ཎྱ་པྲ་བྲྀཏྟ་སྱ་ཀོ་ཤཱ་དྷ་ཀི་ར་བཱ་ཞེས་པས་ཆགས་སླད་མཛོད་སྲུང་སྐྱེལ། །བསྐོས་པས་སྦྱིན་པའི་བསོད་ནམས་ལ། །ཞུགས་པ་དེ་ཡི་བཀའ་ར་ཡཏ། །དཱ་ནེ་ནི་ཥིདྷེ་པོ་ཏྲེ་ཎ་སཾགྷཱ་ཡ་པྲ་ཤྲཱི་བཱི་པ་ཏི། །བྷི་ཥ་ཛྙཱ་དག་བཟློག། ཚ་བོས་སྦྱིན་པ་བཀག་པའི་ཚེ། །དགེ་འདུན་ལ་ནི་ས་བདག་གིས། །རང་སྐལ་ཉིད་དུ་་མ་ལ་ཀ་སཱུདྡྷཾ་ད་དོ་སཱ་བྷཾ་ག་བྷ་བ་ཏཾ། །དྷཱི་མ་ཏསམྨ་ཏྭེ་ནཱ་ཐ་ར་དྷི་གུསྟ་སྱ་མཉྫི་གྱུར་པའི་སྨན། །སྐྱུ་རུ་ར་ཡི་ཕྱེ་དག་ཕུལ། །དེ་ནས་བློན་པོ་བློ་ལྡན་ནི། །མགུ་བྱེད་སྦས་པའི་གྲོས་དག་ཎ༑། ད་དོ་སཾ་གྷཱ་ཡཱ་ནི་ཁི་ལོ་པྲི་ཐི་བཱི་པྲི་ཐི་བཱི་པ་ཏི། །གོ་གཱམྦུ་ཧཱ་ར་རུ་ཙི་རཱ་ཙ་གིས༑། དགེ་འདུན་ལ་ནི་ས་གཞི་དག། མ་ལུས་ས་ཡི་བདག་པོས་ཕུལ། །གང་གཱའི་ཆུ་ཡི་དོ་ཤལ་མཛེས་།
