བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེ་ཏསྟེ༑། སུ་ཁཱ་དུཁཱ་ཡ་གུ་ཎོ་ད་ཡཱ་ཡ༔་དོ་ཥཱ་ཡ་ཤཱ་པཱ་ཡ་བ་དྷཱ་ཡ་ཀཱ་ལེ། །མི་འགྱུར། །སྲིད་པར་སྲོག་ཆགས་རང་གིས་བྱས་པའི་ལས། །འཁྲུལ་མེད་འབྲས་བུ་ཅན་རྣམས་དུས་སུ་ནི། །ཀརྨཱ་ཎྱ་སཾ་བྷནྟ་ཕ་ལ་ནི་ཛནྟོ་སྭ་ཡཾ་ཀྲྀ་ཏཱ་ནྱ་བ་བྷ་བེ་བྷ་བནྟི། །ཡནྣཱནེ་བདེ་དང་སྡུག་བསྔལ་ཡོན་ཏན་རྒྱས་པ་དང་། །སྐྱོན་དང་དམོད་པ་གསོད་པའི་སླད་དུ་འགྱུར། །གང་ཞིག་ཎ་ཤ་ནིརྦ་ཛནི་པ་ཏི་ཏ༔པྲསྣོ་ཏི་བྷ་གཾ་ཏ་ནོརྨཤྭ་བྷཱ་པ་རི་བྷཱ་ལམ་ནས་དལ་གྱིས་འགྲོ་བ་ལྷུང་ཞིང་ལུས་ནི་རབ་ཏུ་ཉམས་པ་ཐོབི། །གང་ཞིག་ལམ་བྲལ་གཡང་སར་མོཏྲ་ཐ་ག་ཏེཤྲཱིདྷཱནཱི་ཤ་ཏུཀྟ་ཏ། །ཡཏི་ད་ཀྱ་ས་ཀྲྀཏྲ་ཡནྣཱ་ཡོངས་སུ་འཁྱམ་ཞིང་རྒྱུག་པ་ཉམས་པ་མེད་པར་འཇུག། གང་ཞིག་ལན་གཅིག་མ་ཡིན་མཁས་པ་ཉམས་ཙ་ཏུ་རཿཤི་སཾ་ཤ་ཡཐཱ་ལ་སསྟདྲི་ཙི་ཏྲ་མ་བྷིཏྟ་ཙི་ཏྲ་ར་ཙ་ནཾ་སནྡྷཏྠ་ནཱི་ཀརྨ་ཤིང་དཔལ་བསྟེན་ལེ་ལོ་ཅན། །ལས་ཀྱི་རི་མོ་རྩིས་པ་མེད་པར་བདག་ཉིད་ཀུན་གྱིས་བཀོད་དེ་རྣམ་པར་ཎྞཾ༑། ཨ་ཡབྷཾ་མསྟེ་ཥཱ་མ་ཚྭ་བུ་ཧཱུྃ་བྷནྡོཎྞཏྟ་ཡོརྫི་བི་ཏ་མེ་བ་པི་ཌོ། །སོ་ཛ་རྒྱ་བཀྲ༑། བདག་གི་ཆུ་བུམ་ཕྱི་མ་མེད་པ་འདི། །ཕ་མ་སྐོམ་པས་གཟིར་ལ་འཚེ་བར་ནི། །གཞུང་བཟང་ལ་མཱ་ལ་དྷི་ཡཱ་ད་ཡཱརྔ་ག་ཏཱ་ཏཱ་ཡཱ་ཏུཎྜ་ཏ་ར་པྲ་དེ་ཡ། །ཨུཀྵེ་ཏི་ཀནྟཱནྟ་བརྟེན་བརྩེ་བས་བརླན་པའི་བློས། །མྱུར་བར་སོང་སྟེ་ཁྱོད་ཀྱིས་སྦྱིན་པར་བྱ། །ཞེས་པ་མགྲིན་པའི་བར་ར་བནྟི་ཛཱི་བཤཱ་ས་པྲ་ཡཱ་ས་སྦ་གི་ཏཱལྤ་བཎྞ། །མོ་ནི་པ་ཙ་ཀྲེ་དུ་སྲོག་རྒྱུ་ཞིང་། །དབུགས་བརྩེགས་ངལ་བས་སྒྲ་བསྒྲིབས་ཆུང་ངུར་བརྗོད། །བཞིན་ནི་ཕྱི་རོལ་ཕྱོགས་ཐུ་ཌི་ཏཱ་ན་ན་སྱ་བི་ལཀྵ་དཱིབྷཱུ་མི་བ་སཱ་ཡ་ཀ་སྱ། །ཧཱུྃ་བྷཾ་ས་མཱ་དཱ་ཡ་ཏ་པའི་མདའ་དག་ལ། །ཐུབ་པ་དེས་ནི་སྐྱེངས་པ་སྟོན་བཞིན་མཛད། །དེ་ནས་ས་དབང་མཆོག་ཏུ་རབ་ཏཀྱི་ཏཱི་ཤསྟ་དཱ་ཤ་མཾ་པཱ་སྱ་ཤྲཱི་བྷཾ་པྲ་ཏཔ། །ཏཱ་བེ་ཀ་པུ་དྲོསྦ་བི་རོ་བི་ལོཀྱ་གདུང་བས། །བུམ་པ་བླངས་ནས་དེ་དག་བསྟི་གནས་ཕྱིན། །རྒན་རྒོན་བུ་གཅིག་ལྡན་པ་དེ་དག་མཐོང༌། །ཏ་དནྟི་ཀ་ནི་བ་ཤ་ཤཱ་ཀ་གནྟཾ། །པུ་ཏྲེ་ཏི་དྷུརཱཏྠ་ད་ཤལྟ་དཏྟ་ཀཎྜོ་ས་མུཀྐཎྜི་ཏ་དེ་དག་གམ་དུ་འགྲོ་བ་ཡོངས་མ་ནུས། །རིང་ནས་རྡོག་སྒྲ་དག་ལ་རྣ་གཏོད་ཅིང་། །མཛའ་བས་རབ་ཏུ་།
