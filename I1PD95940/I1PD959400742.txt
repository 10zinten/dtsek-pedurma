ནྱ་བེ་ད་ཡེ་ལླེ་ཁ་ནི་བནྡྷ་མསྨི་ནི་ཛ་ཤ་ནིནེ་ཏྲ་བི་ནཱ་ཤ་བྲྀཏྟཾ། །ཐུ་ཏྲི་བ་དེ་ལ་དལ་གྱིས་སྤྲིང་ཡིག་དང་འབྲེལ་པའི། །རང་གི་མིག་ནི་རྣམ་པར་ཉམས་ཚུལ་བཤད། །སྤྱོད་ངན་སྤྱོད་ཏཏྟཱི་བྲ་བྷཾ་ཤཾ་ས་བྲྀཏྟཾ་ནི་བྲྀཏྟཾ་བྲྀཏྟེ་ཥ་པི་ན་པྲ་བིཏྟཾ། །ཀུ་ཋཱ་ར་དྷཱ་རཱ་པ་རི་པ་རྣམས་ལའང་མ་སྤྱད་པའི། །གཏུམ་པའི་སྤྱོད་ཚུལ་མི་བཟད་དེ་ཐོས་ནས། །སྟ་རེ་རྣོན་པོས་རྩ་བ་ཡོངས་པུ་རི་བྷུ་ཏ་མཱུ་ལཔ་བཱ་ཏ་ཤཱ་ཁཱི་བ་དྷ་ར། །ས་ལབྷཱ་སཾ་ཛྷཱཀུ་ཊི་ལཾ་བི་ཙིནྟ་བཅད་པའི། །ཡལ་འདབ་ཅན་བཞིན་འཛིན་བྱེད་ཡུ་བ་འགྱེལ། །འདུ་ཤེས་ཐོབ་དེས་སྐར་རྒྱལ་སྲུང་བ་ཡིས། །ཏཾ་ཏི་བྱ་རཀྵ་ར་ཙི་ཏཾ་པྲ་ཡོ་གཾ། །ས་མུ་ཏྱཱ་ཡོ་སི་བ་དྷ་པཱ་ཏ་ཀོ་པ་དྷརྨཱ་ད་རེ་ཎེ་བཀོད་པའི་སྦྱོར་བ་གྱ་གྱུ་དེ་བསམས་ནས། །དེ་ཡི་ཆད་པ་བུད་མེད་གསོད་པ་ཡི། །སྡིག་ལའང་ཆོས་ལ་བ་བི་ནི་གྲ་ཧེ་སྣུ། །ཀོདྨ་པྲ་ཀཱ་རེ་བི་པུ་ལཱ་པ་ཀ་རེ་ཏསྨིནྡྲ་ཏཱི་ཀ་ར་ས་མུ་ཏྲ་ཏཾ་གུས་པ་བཞིན་བརྩོན་གྱུར། །གདུག་པའི་རྣམ་པ་གནོད་པ་རྒྱ་ཆེ་བ། །དེ་ལ་ལེན་སྦྱོར་རབ་ཏུ་བརྩོན་པ་ཏཾ༑། ཨ་བཱ་ར་ཡཇུས་ཧཱ་དུཁ་ཡོ་གཾ་སྭཱ་ཀརྨ་པཱ་ཀེ་བ་ད་ན་ཀུ་མཱ་ར། །དེ། །རང་ལས་སྨིན་པས་བཟོད་དཀའི་སྡུག་བསྔལ་དང༌། །ལྡན་པར་སྨྲ་བའི་གཞོན་ནུས་རབ་ཏུ་བཟློག། ཏ་མ་བྲ་བཱིདྷཱ་མི་པ་ཏིཅུ་ཐཱཎཱཤོ་ཀེ་ན་ཀོ་པེ་ནཱ་བ་ད་རྒྱ་མཱ་ན། །མོ་ཧཱཀྐི་མེ་ས་བདག་གདུང་བས་གཟིར་ཅིང་མྱ་ངན་དང༌། །ཁྲོ་བས་ཚིག་པས་དེ་ལ་རབ་སྨྲས་པ། །གདུག་པ་ལ་ཏྭཾ་ནི་ཤི་ཏི་མ་ནཱརྨ་ཀྲོརྨ་པྲ་སཀྟ་པ་རི་རཀྟ་སིདྷཾ། །དྭེ་ཥོ་བྱ་ཏེ་ཆགས་འཕགས་མིན་གཏུམ་མོ་འདི། །ཁྱོད་ནི་རྨོངས་པས་ཡོངས་སུ་སྲུང་ངམ་ཅི། །སྡང་བར་སྦྱོར་དང་སྣེ་ཧ་ནི་བནྡྷ་ནེ་བཱ་ཏུ་ལོ་མ་ནོ་ཡ་སྱ་ས་ཀིམྨ་ནུ་། །ཡ་སཱུ་པ་ཀཱ་རེསྙི་ན་རོ་ཥ་ལེ་མཛའ་བས་བཅིངས་པ་ལ། །མཚུངས་པའི་ཡིད་ཅན་གང་དེ་མིའམ་ཅི། །གང་ལ་གནོད་ལ་ཁྲོ་བའི་ཆ་མེད་ཤསྟ་སྱ་པ་ཀཱ་ར་པི་ཀ་ཐཾ་པྲ་སཱ་ད། །ཨི་ཏི་ཤཱ་སནྟཾ་པྲ་ལ་སནྟཾ་མནྟཾ་དྷཱི་རཿཀུ་མཱ་རཿཔ། །དེ་ལ་ཕན་ལ་དྲིན་ཡང་ཇི་ལྟར་ཡོད། །ཅེས་པ་རབ་སྨྲ་ཤུགས་འབྱིན་གཟིར་བ་ཡི། །ཡབ་ལ་གཞོན་པི་ཏ་རཾ་ཛ་ག་ད། །ར་ཛསྣ་མེ་དུ༔ཁ་ལ་བོསྟི་ཀཉྩིཏྟི་པཱ་པ་ཀཱ་རོ་བི་ན་ནུ་བརྟན་པས་རབ་སྨྲས་པ། །རྒྱལ་པོ་བདག་ལ་སྡུག་བསྔལ་ཆ་འགའ་མེད། །མི་བཟད་གནོད་ལའང་ཁྲོ་—།
