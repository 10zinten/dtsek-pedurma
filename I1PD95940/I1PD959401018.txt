ཕྱི་གཙྪ་མཀིཾ་བྷ་བི་ཁྱཱ་ཏི། ་། ཨི་ཏི་ཏེ་ས་མ་ཡཾ་ཀྲྀ་ཏྭཾ་བསྲེ་ཎ་ཏ་མ་དྷཱ་ཛན། །ཏཧྥུ་དུ་མི་འགྲོ་ཅི་འབྱུང་འགྱུར། །ཞེས་པའི་དམ་ཚིག་དེ་དག་གིས། །བྱས་ནས་གོས་ཀྱིས་དེ་ལ་མཆོད། །དེ་ཡི་ཅུ་པྲ་ཎི་དྷ་ནེ་ན་དྷ་ནི་ཀོ་ཡཾ་ས་མཱ་མི་ཀ། །ཛཱ་ཏོ་དི་བྱ་པྲ་བྷཱ་བསོད་ནམས་སྨོན་ལམ་གྱིས། །འཕྲེང་རྒྱུད་མཁན་འདི་ནོར་ཅན་འདི། །ཆུང་མ་བུ་དང་མནའ་མར་བཱདྷཔཏི་པུ་ཏྲསྣུ་ཥ་ནཱི་ཏ། །སཏྠཛྷེ་ནི་ཏི་ཀ་ཐི་ཏཾ་ཐུ་ཏྭཱ་སཾ་ས་ལྡན༑། ལྷ་མཐུ་ཕུན་སུམ་ཚོགས་པར་གྱུར། །ཅེས་པ་ཚོགས་སུ་ཀུན་མཁྱེན་གྱིས། །བཤད་པ་ཐོས་ནས་དི་བྷིཀྵ་བ། །མེ་ནི་རེ་ཙིཏྟ་བི་མ་ལ་མཱུ་ལཱ་དཱ་ན་ཕ་ལ་ཤི་ཡཾ། །ཏ་ཏསྟེ་བཱ་དགེ་སློང་རྣམས། །སེམས་ནི་དྲི་མེད་རྩ་བ་དག། སྦྱིན་པའི་འབྲས་བུའི་དཔལ་དུ་སེམས། །དེ་ནས་བྲམ་ཧྥ་ཎསབྦེ་སཾ་བྷུ་ཡ་ཤུ་ཙི་བྷོ་ཛ་ནི། །བྷ་ག་པནྟཾ་ས་མ་བྱཱ་ཙྩ་བ་ཟེ་དེ་དག་རྣམས། །ཀུན་འཚོགས་གཙང་བའི་ཟས་དག་གིས། །བཅོམ་ལྡན་འདས་ནི་ཡང་དག་མཆོད། །བྷུ་བུཀུ་ཤ་ལོ་ཛཱ་ལཱ། །ཧེ་ལཱམྲྀ་ཏཾ་རཏྣ་དྷ་ནཾ་ཀྲྀ་ཎཱ་གྲོ་ཤུདྡྷཱ་བ་ཏཱཎྞཾ་ཀྲྀ་ཎ་དགེ་བ་རབ་ཏུ་འབར་བར་གྱུར། །དམན་བྱིན་རིན་ཆེན་ནོར་ནི་རྩྭ་ཡི་རྩེ། །དད་པས་བྱིན་པའི་རྩྭ་ཡང་མ་པྱནགྷོ། །ཨ་ནེ་ཀ་ཤོ་བྷཱ་ནུ་བྷ་ཐཾ་ན་བིཏྟཾ་ཙིཏྟ་ནི་མིཏྟཾ་ཤུ་བྷཾ་སཾ་བྷ་བཱ་ནཱཾ། །རིན་ཐང་མེད། །མཛེས་པ་དུ་མའི་ཉམས་མྱོང་ནོར་ཡིན་ཏེ། །དགེ་བ་འབྱུང་བའི་རྒྱུ་ནི་སེམས་ཉིད་དོ། །ཨི་ཏི་ཀྵེ་མེརྩྭ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་སཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་དྷ་ནི་ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ཀཱ་བ་དཱ་ནནྡ་བ་ཏི་ཏ་མཔལྤ་ཝ། །།།ལས་ནོར་ཅན་གྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་དགུ་བཅུ་པའོ། །།སུ་བྷཱ་ཥི་ཏཾ་ས་ཏྱ་པ་ཐཱ་པྲ་ཀཱ་ཤ་ནཾ་ཀ་རོ་ཀྱ་ནཱ་ལོ་ཀ་པ་དེ་ཥཱ་དི་སྣང་བ་མེད་པའི་གནས་རྣམས་དག་ཏུ་སྒྲོན་མེ་བཞིན། །ལེགས་པར་བཤད་པ་བདེན་པའི་ལམ་ནི་པ་བཏ། །ས་དོ་པ་ཀཱ་རཱ་ད་ཏ་ཨེ་བ་བལྤ་ཏཾ་བི་བེ་ཀ་ཙ་ཙཱ་སུ་མ་ཧཱཏྨ་གསལ་བར་བྱེད། །དེ་ཉིད་ཕྱིར་ནི་རྟག་ཏུ་ཉེ་བར་མཁོ་བ་ལས། །བདག་ཉིད་ཆེ་རྣམས་དབེན་པ་སྤྱོད་—།
