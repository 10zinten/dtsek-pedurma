བྷཾ་པ་ཏི་ཏ༔སུ་ཁ་དུཁ་ཡོ། །ཏ་སུ་མེ་ཀུ་རུ་ཀཱ་རུ་ཎཾ་ས་ཁེ་ཀ་ལུ་ཥ་ཀ་རི་ཡིས། །བདེ་དང་སྡུག་བསྔལ་དག་ཏུ་དོར?། །བདག་ནི་སྡིག་པ་བྱེད་དེ་ལ། །གྲོགས་པོ་ཡིས་ནི་སྙིང་རྗེ་ཎ༈:། ག་ཏཱ་སཱ་དེ་ཤཾ་མ་ཏ་པུ་ཊཾ་བྲུ་ཧི་མཱཏ་བ་ཙ་སཱ་ར་ཧ། །ཨསྟི་མ་མཛོད། །རང་ཡུལ་སོང་ནས་བདག་ཚིག་གིས། །དབེན་པར་བདག་གི་བུ་ལ་སྨྲོས། །བདག་གི་ཁང་གྲི་ཧ་ཀོ་ནཱནྟེ་ནི་ཁཱ་ཏཾ་ཧེ་མ་བྷཱ་ཛ་ནཾ། །ཏ་དུདྡྷཾ་ཏྱ་པ་རི་ཏྱལྟ་བཱ་པཱ་པྲྀཏྟི་པའི་ཟུར་མཐའ་ན། །གསེར་གྱི་སྣོད་ནི་སྦས་ནས་ཡོད། །དེ་ནི་བྱུང་ལ་སྡིག་པ་དག། ཡོངས་སུ་གཏོང་པི་དྷཱི་ཡ་ཏྭཾ། །ཨཱརྱ་ཀཱ་ཏྱཱ་ཡ་ནོ་ནི་ཏི་པིཎྜ་པཱ་ཏེ་ན་པཱུ་ཛུ་ཏྭཾ། །ཨི་ཏུཏྟསྟེ་བའི་བྱ་བ་སྒྲུབས། །འཕགས་པ་ཀཱ་ཏྱཱ་ཡ་ན་ནི། །རྟག་ཏུ་བསོད་སྙོམས་དག་གིས་མཆོད། །དེ་ནི་དུལ་ན་བི་ནེ་ཡཱཏ་ཏ་ཐོ་ཏྱཱཀཱ་ཛ་ཀཱ་མ་ས། །ས་ད་དན་བྲ་ཛན་དི་བྱ་ཐཾ་མཱ་བས་དེ་བརྗོད་ཚེ། །དེ་ཡིས་དེ་བཞིན་ཞེས་སྨྲས་སོང་། །དེ་ནི་སོང་བས་མདུན་དག་ཏུ། །ལྷ་ཡི་ན་མ་བ་རཱ་བུ་ན། །།རསྣ་པཱ་ཏྲ་ལ་ཏཱ་ཀཱནྟཾ་དཱི་ཏཱི་ཡ་མི་བ་ནནྡ་ནཾ། །གཞལ་ཡས་ཁང་པ་གཞན། །རིན་ཅེན་པདྨ་འཁྲི་ཤིང་མཛེས། །དགའ་ཚལ་གཉིས་པ་ལྟ་བུ་མཐོང་། །ཏསྨིན་སཾ་ག་མི་བཱ་ནཾ་གི་སཾ་ག་ཏཾ་དི་བྱ་ཡོ་ཥི་ཏཱ། །ཨ་ས་དྷཱ་བ་རྭ་རནྟེ་ཥཱ་དེར་ནི་ཉིན་མོ་རྩོམ་པ་ནས། །ལུས་མེད་དང་འགྲོགས་ལུས་ཅན་བཞིན། །ལྷ་མོ་དག་དང་སྐྱེས་བུ་ནི། །པུ་རུ་ཥ་རཏྣ་བྷུ་ཥི་ཏཾ། །ཏེ་ན་སི་ཏུ་དཱ་རེ་ཎ་ཀྲྀ་ཏཱ་ཏི་ཤྱསྟ་ཐི་བ་ས། །རིན་པོ་ཆེ་ཡིས་བརྒྱན་པ་མཐོང༌། །དེ་ཡིས་ཀྱི་བདེ་བའི་ཡོ་བྱད་ཀྱིས། །དེ་བཞིན་ཉིད་དུ་དེར་གུས་བྱས། །དི་ནཾ་ནི་ནཱ་ཡ་དཱིགྷདྷ་གླེ་ཤ་མ་དྷེ་སུ་དྷཱ་མ་ཡཾ། ་། ཨ་ཐ་བྱོ་མ་བི་མཱ་ལམ་རིང་ཉོན་མོངས་དབུས་སུ་ནི། །བདུད་རྩིའི་རང་བཞིན་ཉིན་མོ་བཟླས། །དེ་ནས་ནམ་མཁའི་གཞལ་ནོ་གྲྭ་ཏ་པ་ཏི་ཏེ་པརྨི་ནི་པ་ཏོ། །ཨ་པཱུརུ་ཡ་ཏ་ཛཱ་གཱ་ཏ་གྷོ་རི་མེད་ཁང་། །རྩེ་ནས་པད་ལྡན་བདག་ལྷུང་ཚེ། །སྡུག་བསྔལ་གྱིས་བཞིན་འཇིགས་རུང་གིས། །མུན་རཿཁི་རི་བ་ཏ་མོ་དྷ་རི། །ཏ་ཏཀྵ་པཱ་པ་ཏི་ཛོནྣཾ་བ་མ་ན་ར་ས་ཏཱ་ཛ་ཌ། །།པའི་ཁུར་གྱིས་འགྲོ་བ་ཁྱབི། །དེ་ནས་མཚན་བདག་ཟླ་བའི་འོད། །སྐྱུག་ཅིང་ཉམས་ལྡན་ཉིད་ཀྱིས་བསིལ། །།
