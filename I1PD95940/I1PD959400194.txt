བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀིགྷོ་རན་ན་ར་ཀ་ཀིཾ་ཀ་རཱན། །ཏེ་ཏ་མཱུ་ཙུ་རི་ཡཾ་བྷུ་མིསྟ་པ་ན་ཀུམྷི་ཤ་ཏཱ་ཙི་འཇིགས་རུང་དམྱལ་སྲུང་རྣམས་ལ་དྲིས། །དེ་ལ་དེས་སྨྲས་ཚ་བའི་ས། །ཟངས་ཆེན་རྒྱ་ནི་རྫོགས་པ་ཏཱ༑། ཀམྤི་ཏཱ་རཱ་ཛ་པུ་ཏྲ་སྱ་ནནྡ་སཱུ་ཏནྡ་རཱ་གི་ཎ། །མི་བྱཱ་བྲ་ཏབ་ནཱ་ཅུ་འདི། །ཤིན་ཏུ་ཆགས་ལྡན་རྒྱལ་པོའི་བུ། །དགའ་བོ་ལ་ནི་བྱས་པ་ཡིན། །ཆགས་བྲལ་ཉིད་ནི་ད་དུང་བི་བྷ་ཛ་ཏེ་་ཏ་རཱ་ག་ཏྭཾ། །བྲཧྥ་ཙམཾ་ཙ་ར་ཏེ་བ་སཱརྒ་སི་ཡང་། །བརྟུལ་ཞུགས་ལོག་པ་དེ་མི་སྟེན། །མཐོ་རིས་བུད་མེད་འགྲོགས་བསམ་པས། །ཚངས་པར་པ་ག་མཱ་ཤ་ཡཱ། །མི་བྱཱ་པྲ་ཏཱ་ནཾ་ལུབྷྱཱ་ནཾ་རཱ་ག་དྭེ་ཤ་ཀ་ཥཱ་ཡི་སྤྱོད་@པ་སྤྱོད་པ་ཉིད། །བརྐམ་ཆགས་ལོག་པའི་བརྟུལ་ཞུགས་ཅན། །ཆགས་སྡང་སྙིགས་མ་ཅན་རྣམས་ནཾ༑། ཨེ་ཏོ་སུ་ནི་ཏྱ་ཏཔ་སུ་ཀུམྦྷི་ཥཱ་བཱ་ཀཱ་ཡཀྵ་ཡ། །ཨི་ཏི་ནམཿས་མཱཀྵ་ཀྱི། །རྟག་ཏུ་ཚ་བའི་ཟངས་ཆེན་ནི། །འདི་དག་འཇིག་པ་མེད་པའི་གནས། །དེ་ལྟར་དགའ་བོས་ཡང་ཛཱ་ཏ་རོ་ཙཾ་ཙ་ཀཾ་ཙུ་ཀ། །ཏ་ཏྲ་ཙུ་ཏཱ་མི་བ་ཏ་ནཾ་པབྷཱུཏྟཱ་པཱ་ད་དག་ཐོས། །སྤུ་ལོང་གོས་ལྟར་རྒྱས་གྱུར་ཅིང་། །དེར་ནི་རྗེས་སུ་གདུང་བ་ཡིས། །ལུས་པོ་དུམ་བུར་གྱུར་མ་རྩྭ་ཏ། །ས་མ་བྱེ་རྱ་ཏྱཱཀྟ་རཱ་ག་སཾ་བཱ་ས་བཱ་ས་ན༔སྭ་ཡཾ། །བ་བཞིན་སེམས། །གནས་སུ་མངོན་སོང་འདོད་ཆགས་ཀྱིས། །བག་ཆགས་ཞེན་པ་བཏང་བྱས་ནས། །བླ་བྷཱུ་བཱ་ནུཏྟ་ར་བྲཧྨ་ཙརྨ་པནཱལྟ་སཾ་ཡ་མ། །གྷ་ན་མོ་ཧི་ཥཱ་ཡཱཏྟ་སྱ་མེད་ཚངས་པར་སྤྱོད་པ་ནི། །ཡོངས་སུ་རྫོགས་ཤིང་དུལ་བར་གྱུར། །ནང་དུ་ཐེ་ཚོམ་རྣམ་གྲོལ་ཅིང་། །བི་མུཀྟི་སཾ་ཤ་མནྟ་ཏ། །མ་ན་པྲ་སཱ་ད་མ་པེ་དེ་ཤ་ར་དྲི་བ་ད་དཱི་པ་ཡ། །།སྟུག་པོའི་རྨོངས་པ་ཟད་ལས་དེའི། །ཡིད་ནི་རབ་དང་ཐོབ་གྱུར་ཏེ། །སྟོན་ཀའི་ཀླུང་གི་ཆུ་བཞིན་ནོ། །ནིཀྐཱ་རཿཔྲ་ཤ་མ་པཱy་པ་རཾ་ནིཌཱ་མུ་པཱ་ག་ཏ། །ཤུདྡྷ་དྷཱིས་མ་བྱེ་རྱ་བྷ་འདོད་པ་མེད་ཅིང་རབ་ཞི་ཐོབ། །མཆོག་ཏུ་ངེས་པར་དེར་ཞུགས་པ། །དག་པའི་བློ་ཡིས་བཅོམ་ལྡན་ལ། །ག་བན་མ་བྷཱ་ཥ་ཏ། །ནཱཔྤ་རོ་ཏིརྣ་པུནྡརྨོ་བྷཱ་ག་བན་ཀྲྀ་ཏྱ་མསྟི་མེ། །ཡང་དག་མངོན་དུ་ཕྱོགས་ཏེ་སྨྲས། །བཅོམ་ལྡན་བདག་ལ་བྱ་བ་ནི། །ལྷ་མོས་ཡོད་མིན་མཛེས་མས་མིན། །།
