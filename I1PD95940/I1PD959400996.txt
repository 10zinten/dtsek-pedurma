ལཱསྦི་ཧཱུཾ་ཁ་ལ། །ཐུ་ཏཱི་ཏཏུ་ག་ཏེ་ནོཀྟ་ཀ་ལ་ཡནྣཱ་ཡཾ་ཀལེ་བ་རཾ། །རགས་པ་འདི་ནི་ཁྱོད་ཀྱིའོ། །བདེ་གཤེགས་ཀྱིས་གསུངས་དེ་ཐོས་ནས། །རང་གི་ལུས་ལ་རྟོག་བྱེད་ཅིང༌། །བྷ་བབྷ་མེ་དྷརྨ་རུ་ཙི་པ་རསྣེཇེ་ད་མཱ་ཡ་ཡོ། །སོ་ཙིནྟ་ཡ་ད་ཧོ་མོ་ཧ་ཆོས་སྲེད་སྲིད་པར་འཁྱམ་པ་ལ། །མཆོག་ཏུ་སྐྱོ་བ་ཉིད་དུ་གྱུར། །དེ་བསམས་ཀྱེ་མ་རྨོངས་པ་ཡིས། །མཱངྒེ་སཾ་ས་ར་ཏྭཾ་ཀྲྀ་ཏྭཾ། །གླུ་ཎ་པྲ་ཏི་ཤ་ཡེ་ཀཱ་ཡེ་སནྣཱ་ནཾ་ནི་ཛ་བཱ་ས་ནཱ། །ལམ་ལ་རབ་ཏུ་བགྲོད་པའི་མགྲོན། །མི་རྣམས་སྐད་ཅིག་བསྟེན་པ་ཡི། །ལུས་ལ་རང་གི་ཉིད་དུ་ཆགས། །བི་སསྨཱསྟྲཱི་སཱུ་ནཔྲ་ཀ་ཊ་ད་ཤ་ནཱ་ཐེ་ཎི་བི་ཀ་ཊཔྲ་ཎཋཀཱ་ཡོ་ཡཾ་ཀྲི་ལུས་ནི་རབ་ཉམས་སྲིན་བུའི་ཚོགས་ཀྱིས་རྐང་ནི་བུག་པར་བྱས་གྱུར་ཅིང་། །དབྱིབས་བྲལ་རུས་པའི་ཀ་ལིབ་མི་ཀུ་ལ་ཀྲྀཏྟཙྪི་དྲ་ན་ལ་ཀཿ། །ཨི་མཱབྷྱཱསཱ་པཱ་ཡཿཔྲ་ལ་ཡ་མ་ཡ་རབ་གསལ་སོ་ཡི་འཕྲེང་ཅིབ་འཇིགས་རུང་འདི། །འཇིག་པའི་རང་བཞིན་སྒྱུ་མ་ལ་ཆགས་མི་རྣམས་དག་གི་མཱ་ཡཱ་པྲ་ཎ་ཡི་ཚྭ་ཚྭ་ཚྭ་རོ་ཧནྡྷེ་ཧཱ་སྨ་ར་ས་བི་ཥ་ཡཱ་སྭཱ་ད་ན་བྷ་བཱ། །ཨི་ཏི་སྲེད་འཕེལ་ཞིང་། །ཆགས་བཅས་ཡུལ་གྱི་ཉམས་མྱོང་ས་གཞི་གནོད་པ་དང་བཅས་དེ་དག་གོ། །དེ་ལྟར་སཾ་ཛཱ་ཏ་བི་རཱ་གྱ་སཱ་ཀ་ལི་བ་ར་དན་ནཱཏྲ། །བྱོསྣ་ཛི་ནས་མཱ་དཱ་ཡཱ་ཡ་ཡོ་ཛེ་རང་གི་ལུས་མཐོང་བསི། །ཆགས་བྲལ་ཡང་དག་སྐྱེས་པ་དེ། །རྒྱལ་བས་བླངས་ནས་ནམ་མཁའ་ལས། །ཏ་བ་ནཾ་པུ་ན། །ཧ་ཏཱ་སྱ་ཙཀྲེ་བྷ་ག་བཱནྡ་ཡ་ཡཱ་དེ་ཤ་ནཱ་ཡ་ཡཱ། །སླར་ཡང་རྒྱལ་བྱེད་ཚལ་དུ་སོང་། །དེར་ནི་དེ་ལ་བཅོམ་ལྡན་གྱི། །བརྩེ་བས་བསྟན་མཛད་གང་གིས་ནི། །སྲོ་ཥས་ཀྲྀ་དཱ་གཱ་མི་ན་ཀྲ་མཱ་དདྷ་ཏྭཱ་མཱཔ་བཱན། །ས་གླེ་ཤ་བི་རྒྱུན་དང་ལན་གཅིག་ཕྱིར་མི་འོང་། །རིམ་པས་དགྲ་བཅོམ་ཉིད་ཐོབ་གྱུར། །ཉོན་མོངས་ཀུན་ལས་རྣམ་ནིརྨུཀྟས་པྲ་ཎ་མྱ་ཏ་ཐཱ་ག་ཏཾ། །པཱསྔརྨ་ནི་ཙ་ཡ་སྨྲ་ཏྭཱ་པྲ་ཡ་ཡོ་གྲོལ་དེས། །དེ་བཞིན་གཤེགས་ལ་ཕྱག་འཚལ་ཏེ། །སྔོན་གྱི་སྐྱེ་བའི་ཚོགས་དྲན་ནས། །།བཞིན་ནི་་ས་སྨི་ཏཱ་ན་ན། །ཏ་ཏ༔ཀ་དཱ་ཙི་དཱ་ཡཱ་ཏཾ་ཏཾ་བྷིཀྵ་ག་ཎ་སནྣི་དྷོ། །འཛུམ་པ་དང་བཅས་སོང་། །དེ་ནས་ནམ་ཞིག་བཅོམ་ལྡན་འདས། །ཀུན་མཁྱེན་དགྱེས་པས་དགེ་སློང་གི། །—།
