ལུ་སྱ་ཀཱ་བྱ་པ། །ཨི་ཀྵ་ཀུ་ར་ནཱ་ཡེ་ཏ་སྱ་ཏ་སྱ་ཙཱ་བྷུད་བི་རཱུ་ཌྷ་ཀ། །ཛྭ་ཏྭཱ་གིས་དང་བར་མཛད། །དེ་བརྒྱུད་བུ་རམ་ཤིང་པ་བྱུང་། །དེ་ལས་ཡང་ནི་འཕགས་སྐྱེས་པོ། །དེ་ཡི་ཀཱ་ནི་ཡ་ས་སུ་ནོ་ཕྱོཊྛསྟེ་ན་བི་བཱ་སི་ཏཱ། །།ཨེ་ཀཱི་བྷྱཱ་ཡ་ཏ་ཏསརྦ་སཱ་དེ་བུ་ནི་ནུ་བོ་ལ། །དགའ་བས་ཕུ་བོ་གནས་ནས་བསྐྲད། །དེ་ནས་གཞོན་ནུ་ཐམས་ཅད་ནི། །གཅིག་ཏུ་ཤ་བི་ག་ཏེ་ཀྲྀ་ཧཱ། །ཀུ་མཱ་རཀ་པི་ལཱ་ཏྱཱ་སྱ་མ་ཧནྟི་རཱ་ཐ་མཾ་ཡ་ཡུ། ་། དྷཱ་འདུས་ནས་རང་ཡུལ་ལ། །ཆགས་དང་བྲལ་བས་ཐུབ་པ་ཆེ། །སེར་སྐྱ་ཞེས་པའི་གནས་སུ་སོང་། །བྱིས་ན་ཀཱ་ལཱནྟ་རཱ་ཡཱ་ཎཱ་བཱ་མཱུ་དུཙྩབྲ་ལཱ་པི་ནཾ། ་། སོ་ནྱ་ཏྲ་ནིརྨ་མེ་ཏེ་ཊཱ་པ་སྒྲ་ཆེན་སྒྲོགས་པ་ཡིས། །བསམ་གཏན་དུས་ཀྱི་གེགས་བྱེད་པ། །དེ་དག་གྲོང་ཁྱེར་གཞན་དུ་ནི། །པུ་རཾ་ཀ་པི་ལ་བསྭེཏི། །ཀཱ་ལ་ན་པུ་ཏྲ་བཱཏྨ་ལྕེ་ད་ནུ་ཏཱ་པ་ན་བྷཱུ་པ་ཏི། །སེར་སྐྱའི་གཞི་ཞེས་དེ་ཡིས་སྤྲུལ། །དུས་ཀྱི་ས་བདག་བུ་ལ་ནི། །མཉེས་གཤིན་རྗེས་སུ་གདུང་བ་ཡིས། །ཨཱ་ནཱི་ཡནྟཾ་ཀུ་མ་རཱ་སྟེ་ས་ཙི་བཱ་ནི་ཏ་བྷཱ་ཥ་ཏ། །ཏ་མཱུ་ཙུརྨནྡྲི་ཎསརྦ་བློན་པོ་རྣམས་ལ་གཞོན་ནུ་ནི། །དེ་དག་ཁྱེར་ལ་ཤོག་ཅེས་སྨྲས། །བློན་པོས་དེར་སྨྲས་རྒྱལ་པོ་ལགས། །རཱ་ཛ་ན་པཱལྟ་པུ་རོཏྟ་མ། །པྲ་ཏྭཱ་ནེ་ཏུ་མ་ཤཱཀྱ་སྟེ་ཛཱ་ཏཱ་པ་ཏྭཱ་པྲི་ཐུ་ཤི་ཡ། །དེ་དག་ཀུན་གྱིས་གྲོང་མཆོག་ཐོབ། །བུ་ཚ་སྐྱེས་ཤིང་དཔལ་འབྱོར་ཆེ། །སླར་ཡང་དགུག་པར་མི་ནུས་སོ། །ཨི་ཏི་ཏེ་ཏཱཾ་བི་ཏུསྟ་ཏྲ་ཤཱཀྟ་ཤཱཀྱ་བི་ཙིནྟ་ནེ། །བ་བྷཱུ་བུ༔ཤཱཀྱ་སཾ་ཛྷཱསྟེ་ཚྭ་དེ་དག་རྣམས་ཀྱི་ཕ་ཡིས་དེར། །ནུས་སམ་མི་ནུས་རྣམ་བསམས་ཚེ། །དེ་དག་ཤཱཀྱའི་མིང་ཅན་གྱུར། །།དེ་པུ་ར་སྟེ་ཥཱ་ཁཾ་ཤ་ཀྲྀཏ། །ཏདྲྀ་ཤེ་ཥཱ་པཉྩ་པཉྩ་ས་ཧ་སྲ་ཥཱ་མ་ཧཱི་བྷུ་ཛཱི། །ཨ་རྣམས་རིགས་ལ་མི་ཡི་གྲོང་། །དེ་ཡི་རིགས་ལས་ཕྱིས་སྤྱོད་ནི། །སྟོང་ཕྲག་ལྔ་པ་ལྔ་འདས་ཚེ། །དཔལ་ཏཱི་ཏེ་ཥཱ་ཀྱི་ཏི་པ་ཏིཤྲཱི་མཱནྡ་ཤ་ར་ཐོཉབྷ་བཏ། །ཏ་སྭཱ་ནཱ་ཡེ་སིཾ་ཧ་ཧ་ནུརྦ་བྷཱུ་ལྡན་ས་ཡི་བདག་པོ་ནི། །ཤིང་རྟ་བཅུ་པ་བྱུང་བར་གྱུར། །དེ་ཡི་བརྒྱུད་ལ་སེང་གེ་འགྲམ། །ས་ཡི་བ་པྲི་ཐི་བཱི་པ་ཏི། །ན་ར་ཎེ་སི་ཧ་མི་བ་ཡཾ་སེ་ཧི་རེ་རཱ་ཛ་ཀུཉྫ་ར། །ཛེཊྛཤུདྡྷོ་ད་བདག་པོ་བྱུང་གྱུར་ཏེ། །གང་ཞིག་སེང་གེ་བཞིན་དུ་ནི། །རྒྱལ་པོ་གླང་པོས་གཡུལ་མི་བཟོད། །དེ་སྲས་ཐུ་—།
