ཌྷ་བྱཀྟི་མཱ་ཡཱ་ཏུ་དཱི་རི་ཏཾ། །ཏིཟིསྟ་རེ་ཎ་ཀ་ཐི་ཏཱ་མཱ་ཀརྞ་བྷ་ག་གནས། །བརྗོད་པས་གསལ་བ་ཉིད་དུ་གྱུར། །དེ་དག་གིས་ནི་རྒྱས་བཤད་པའི། །བཅོམ་ལྡན་འདས་ཝ་ཏ་ཀ་ཐཱ། །སོཉབྷ་བཏྟངྒ་ཏ་མ་ནཱ་ཏ་དན་ན་ས་མུཙྪཱ་ཀ། །ས་ཤ་ནིཀྲྀ་ཀྱི་གཏམ་དག་དེས། །ཐོས་ནས་དེ་ལ་ཡིད་སོང་སྟེ། །དེ་ནི་ལྟ་བར་འདོད་པར་གྱུར། །རིམ་གྱིས་ཁྱིམ་ཧ་མཱ་ག་ཎྱ་ཏུཀྟ་སཏྠ་པ་རི་ཙྪ་དཾ། །ཨ་ནཱ་ཐ་པིཎྜ་དཾ་དྲཧཱུྃ་ཤྭ་བ་དུ་འོངས་ནས་དེས། །ཡོ་བྱད་ཐམས་ཅད་ཡོངས་བཏང་སྟེ། །གྲོགས་པོ་མགོན་མེད་ཟས་སྦྱིན་ནི། །ལྟ་རུ་༈་སུ་ཧྲི་དཾ་ཡ་ཡོ། །ཨ་བྷི་ལཱ་ཎཾ་ནི་བེ་བྱསྨེ་པྲ་བྲ་ཛཱ་ཡཾ་ཛི་ཏེནྡྲ་ཡ། །ཛ་མཉན་དུ་ཡོད་པར་སོང་། །རབ་ཏུ་བྱུང་བར་མངོན་འདོད་པ། །དེ་ལ་དབང་པོ་ཐུལ་བས་བཤད། །དེ་གཱ་མ་ས་ཧི་ཏསྟེ་ན་བྷཀྵ་བྷ་ག་བ་ཏོནྟིཀཾ། །དྲྀ་བེ་ཏ་ཏྲ་སརྦ་ཛཱི་དག་ལྷན་ཅིག་གུས་པ་ཡིས། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་སོང་། །གཏི་མུག་མུན་པ་ཉིན་མོར་བྱེད། །མོ་ཧ་དྷཱན་དི་བཱ་ཀ་རཾ། །ཏཏྟཱ་ད་བརྞ་ནེ་ནི་བ་མེ་ནེ་ས་ཀྲྀ་ཏ་ཀྲྀ་ཏ་ཏྭཾ། །ཀུན་མཁྱེན་དེར་ནི་དེས་མཐོང་ནས། །དེ་ཡི་ཞབས་མཐོང་ཉིད་ན་དེས། །བྱ་བ་བྱས་པ་ཉིད་དུ་བསམས། །བི་ཛཱ་ཡ་ཏ་སྱ་བྷ་ག་བཱན་སཾ་ཀལྤནྟ་མ་བྷ་ཥ་ཏ། །དེ་ཤ་ན་ཚོུདྡྷཱ་ཡ་ཀུཉྫན་དེ་ཡི་བརྟག་ཨེཔ་བཅོམ་ལྡན་གྱིས། །རྣམ་པར་མཁྱེན་ནས་དེ་ལ་ནི། །ཕྱོགས་རྣམས་ཚེམས་ཀྱི་ཟླ་འོད་ཀྱིས། །བི་བེ་ཀ་བི་མ་ལཱ་དི་ཤ། །ཨེ་ཧི་བྷིཀྵ་ནི་རཱ་ཀཾ་ཀྵེ་ནིཇི་བཀྵེ་ཀླུ་དྲི་མེད་དབེན་པར་མཛད་ཅིང་གསུངས། །དགེ་སློང་ཚུར་ཤོག་དོགས་མེད་ཅིང་། །མི་མཐུན་ཕྱོགས་བྲལ་ཡོདྡྷི་ཏེ། །ཨཱ་ཀྵ་ཏེ་དྷརྨ་བི་ན་ཡེ་བྲཧྥ་ཙཙཾ་ཙ་རེཔི་ཏཾ། །ཉམས་པ་སྤངས། །ལེགས་པར་གསུངས་པའི་ཆོས་འདུལ་ལ། །ཚངས་པའི་སྤྱོད་པ་སྤྱད་འདོད་པའོ། །ཨི་ཏི་པྲ་སཱ་ད་ཤཱི་ལེ་ན་ཛི་ནེ་ནོ་དཱི་རི་ཏེ་པུ་རཿ། །པྲ་བྲ་ཛུ་ས་ཧ་སི་བཱ་སྱ་ཞེས་པ་རབ་དང་ངང་ཚུལ་གྱིས། །རྒྱལ་བས་སྔོན་དུ་བརྗོད་པའི་ཚེ། །དེ་ཡི་ལུས་ལ་འཕྲལ་ཉིད་དུ། །པ་པཱ་ཏཱ་ལཀྵི་ཏཱ་ཏ་ནོ། །ཏ་ཏས་ཤ་ཏྲེ་མི་ཏྲེ་ཙ་པྲ་ཤ་མཱ་ཏ་ས་མ་ཏཱཾ་བྲི་ཏ། །མཚོན་པ་མེད་པར་རབ་བྱུང་ལྷུང༌། །རབ་ཏུ་ཞི་ལས་དགྲ་དང་ནི། །གྲོགས་ལ་མཚུངས་པར་བསྟེན་པ་དེས། །—།
