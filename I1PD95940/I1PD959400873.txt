མ་ཏི། །བྷོ་གཱནྡྷཔྲ་བྷ་བ༔སྭ་བྷཱ་བ་སུ་ལ་བྷ་ཤཱི་པི་བྷ་མོཊཾ་ཁ་ལཱབར་འགྱུར། །རྗེ་བོ་ལོངས་སྤྱོད་ཀྱིས་ལོང་རང་བཞིན་གྱིས་སྨོད་ཉིདཔལ་གྱི་རྣམ་འཕྲུལ་ལྡན་ཞིང་འཕྱར། །ས་ཏྱ་བྱ་ཏྱ་ཡ་བྷུ་མ་ཡོ་ཡུ་བ་ཏ་ཡཔུ་ཙྪཱ་བྷི་ཙཱ་ར་ཀྲི་ཡཱ། །སཱ་ཙྪནྟུདྡྷ་ན་ཆུང་བདེན་པ་ཉམས་པའི་ས་གཞི་བསོད་ནམས་མངོན་པར་འཇོམས་པའི་བྱ་བ་ཅན། །ཕྲ་མ་ལྡན་པ་དང་ཏ་བཱ་ད་ནཉྩ་པི་ཤུ་ནཱཁེ་ཙི་ཏྲ་ཀརྨོ་བྱ་ཏཱ་ཡ་ཏྲི་ཀ་ཏྲ་བ་སནྟི་ཏ་ཏྲ་པ་ར་འདུན་རྨད་བྱུང་སྨྲ་བྱེད་མཁའ་ལ་རི་མོའི་ལས་སྦྱོར་རྣམས། །གང་དུ་ངེས་པར་རྣམ་པར་གནས་པ་དེར་ལསཱ་དྷུཀ་ཐཾ་ཛཱི་བ་ཏི། །ཨནྟརྩི་ཏ་ཏ་ནོ་དི་བྱཱཐུདྭེ་ཏི་བ་ཙ་སཾ་དྲོ་པ། །ཀུ་ནི་དམ་པ་དྲང་པོ་ཇི་ལྟར་འཚོ། །ལྷ་མོ་ལུས་ནི་མི་སྣང་བའི། །ཚིག་དེ་ཐོས་ནས་ས་བདག་ནི། །ཁྲོས་པི་ཏ་ཏ་སྱ་བྲྀཏྟཾ་ཏཾ་ས་པྲཛྙཱནྟཔུ་རཱ་ག་ནཱ། །ཀྲོ་དྷཱ་ཀཱནྟ་སྱ་ཚྭ་པ་ཏེ་སྟི་བྲ་པས་དེ་ཡི་བྱུང་ཚུལ་དག། བཙུན་མོའི་ཚོགས་ལ་རབ་ཏུ་དྲིས། །ཁྲོ་བས་ནོན་པ་མི་བདག་གི། །བཀའ་ཤཱ་ས་ན་ཤྲཱི་ཀི་ཏཱ། །ཏཱ་ནི་བི་བྱ་ཡ་ཐཱ་བྲྀཏྟཾ་བ་དྷྱ་བུངྷ་ཡ་བི་ཧྭ་ནི་དྲག་པོས་དོགས་གྱུར་པ། །དེ་དག་གིས་ནི་བྱུང་ཚུལ་བཞིན། །བཤད་ནས་འཇིགས་པས་འཁྲུགས་ལཱ༑། ནིརྡོཥ་ད་ཡི་ཏཱཾ་ཛྙཱ་ཏཱ་པ་པཏི་བཉྩི་ཏྭཾ་ཀྲྀ་པ། །བ་དྷ་བྷཱུ་པར་གྱུར། །ཆུན་མས་བསླུས་པའི་བཙུན་མོ་ནི། །གསོད་པར་རབ་ཏུ་བཏང་བ་དག། སྐྱོན་མེད་ཤེས་མི་ས་མུཉྫིཥྛ་ཤུ་ཤོ་ཙཱ་ནུ་ཤ་ཡཱ་ཀུ་ལ། །རཱ་ག་ཀྲོ་དྷོ་ཊཾ་པཱ་ལཙྪཱ་ཤོ་ནས་མི་ཡི་བདག། འགྱོད་པས་གཟིར་བས་མྱ་ངན་བྱས། །མཚུངས་པར་སྟོབས་རྒྱས་ཆགས་དང་ཁྲོ། །ཀིསྐུ་མྱ་བ་ལོ་ད་ཡི། །ས་མུ་ཧུརྨོ་ཧ་པ་ཙི་བིནྡྷ་གི་ཀྲྀ་ཏ་ཨི་བཱ་བྷ་བརྩེ་བས་ངོ་ཚ་མྱ་ངན་དང་། །རྨོངས་པའི་གྲོགས་ཀྱིས་ཡང་ཡང་དུ། །དེ་ནི་བསྐལ་པར་བྱས་བཞིན་བཏ༑། ཧཱ་སྲི་ཡེ་ཧཱི་ན་པུ་ཏྱཱ་སྱ་ཀཱ་པུ་ནསཾ་ག་མསྦྲ་ཡཱ། །མ་མེ་གྱུར། །ཀྱི་ཧུད་དགའ་མ་ཁྱོད་དང་ནི། །ཕྱི་ནས་བསོད་ནམས་དམན་པ་དག། གང་དུ་འགྲོགས་པ་ཏུཏྟཱ་ན་ར་པ་ཏིའྲི་པ་པཱ་ཏ་མ་ཧཱི་ཏ་ལེ། །ཨ་ཐ་སཾ་གཱ་པྲ་བཱ་སཱནྟཾ་མཤྩ་ཎཾ;ཞེས་བརྗོད་ནས། །མི་བདག་ས་ལ་འགྱེལ་བར་གྱུར། །དེ་ནས་གང་གཱའི་རྒྱུན་ཐོབ་པ། །སྒྲོམ་དུ་རྒྱལ་—།
