པ་ཏིརྨ་ཏཾ་མ་ཏཱ་བ་ར། །བྷ་བི་ལོ་བྷ་བ་བྷ་དྲ༔་བྷ་བ་ནནི་ཙ་བི་ཐུ་ཏཱ། །ཀེ་ཏ་ཧཱུཾ་འབྱོར་པ་ཞེས་པ་བྱུང་བར་གྱུར། །དེ་ཡི་ཆུང་མ་ཀེ་ཏ་ཀི། །ཞེས་པ་ལ་ནི་བུ་གསུམ་བྱུང༌། །འབྱོར་ལེན་ཏ་སྱ་ཛཱ་ཡཱ་ཡཾ་བ་བྷཱུ་བུསྟ་ན་ཡསྡུ་ཡ། །རོ་ག་ཡོ་གཱ་དུ་པ་ག་ཏདང་ནི་འབྱོར་བཟང་དང་། །འབྱོར་པ་དགའ་ཞེས་གྲགས་པའོ། །ནམ་ཞིག་དེ་ནི་ནད་ཀྱིས་བཏབ། །ས་ཀ་དཱ་ཙིནྟྲ་མཱུཆུ་ཏྭཾ། །པཱ་རུ་བྷཾ་ཤ་མུངྒིགིཔདྷཱི་པུ་ཏྲི་རུ་བེ་ཀླུ་འཆི་བ་འདོད་ཅིང་ཉེ་བར་གྱུར། །མཆོག་ཏུ་རྩུབ་ཅིང་ཆགས་བྲལ་བས། །ཆུང་མ་བུ་རྣམས་དག་གིས་ཏ༑། བྷཀྵ་དཱ་སི་ཏུ་ཏ་སྤྱི་ཀཱ་པ་རི་ཙརྱ་པ་ར་པ་རཾ། །མལླི་ཀཱ་ཁྱ་བྷ་བནྟཱ་སྤངས། །དེ་ཡི་ཚེའབངས་མོ་ཕྲེང་ལྡན་མ། །ཞེས་པ་གཅིག་གིས་གུས་པ་ཡིས། །ཇི་སྲིད་རྣལ་དུ་གནས་བཏ་ས་ཡ་བཏ་སཱསྦྱ་མ་ཡཡོ། །སྣེ་ཧོ་པ་ཀཱ་ར་པྲ་ཎ་ཏཀྲྀ་ཏ་ཛྷཔ་ཏ་ཡཱ་གྱུར་པ། །དེ་སྲིད་རིམ་གྲོ་མཆོག་ཏུ་བྱས། །མཛའ་ཞིང་ཕན་པས་རབ་བཏུད་པ། །བྱས་ཤེས་དེ་ནི་དེ་ནཱི་ཏ། །རཾ་ཏོ་སཾ་ག་མ་ཧྥེ་རྱ་ཏ་ཏ་སྙཔུ་ཏྲ་མ་ཛཱི་ཛ་ན་ཏ། །ཡ་དཱ་ཏཛྫརྨ་ནི་པི་ཏུདང་ལྡན། །ཟླ་མཚན་དུས་སུ་འགྲོགས་གྱུར་པས། །དེ་ལ་བུ་ནི་བཙས་པར་གྱུར། །གང་ཚེ་དེ་སྐྱེས་ཕ་ཡི་སནྣཱནྜཔུཎ་ཏཾ་ཡ་ཡུ། །ས་ཏ་དཱ་པུཎྞ་ནཱ་མཱ་བྷུདྨ་ལཿཔུཎེནུ་པུནྡ་ར། །ནི། །ནོར་ྋཀུན་གང་བ་ཉིད་དུ་གྱུར། །བྱིས་པ་ཟླ་ཉ་ལྟར་མཛེས་དེ། །དེ་ཚེ་གང་པོ་ཞེས་པར་གྱུར། །ཚེཥྛསྔ་ཡཀྲྀ་ཏོདྷཱ་ཧ་ཡ་ཡུ་རབྷིནྟ་ནཱཏྲི་ན། །པུཎྞསྐུ་བྷཎྜ་ཤཱ་ལཱ་ཡཱ་ཕུ་བོ་གསུམ་གྱིས་ཆུང་མ་བླངས། །རྒྱ་མཚོར་ནོར་དོན་གཉེར་དུ་སོང་། །གང་པོས་ཕ་ཡི་ཚོང་ཁང་དུ། །པི་ཏུཉྩ་ཀྲེ་དྷ་ནཱ་རྫ་ནཾ། །ཏ་ཏཔྲ་ཏི་ནི་བྲྀཏྟཱསྟེ་པཱབྷྱཱསཱ་ག་རཱཏྨ་ན། །འདུག་ནས་ནོར་གྱི་བསྒྲུབ་པ་བྱས། །དེ་ནས་དེ་དག་ནོར་ཐོབ་སྟེ། །རྒྱ་མཚོ་ལས་ཀྱང་གྲོང་ཁྱེར་དུ། །ག་ཎ་བཱི་ཧེ་མ་ལ་ཏཱཾ་ཀྲྀ་ཏྭཱ་སཱ་མུ་དི་རེ་པུ་རཾ། །ས་མུ་དྲ་ག་མ་ནེ་ཏེ་སླར་ལོག་གསེར་ནི་འབུམ་ཕྲག་རྣམས། །བགྲང་བར་བྱས་ཏེ་དགའ་བར་གྱུར། །དེ་དག་རྒྱ་མཚོར་བགྲོད་༈་ཡཱ་བཱ་ནཱ་སཱིདྡྷ་ནཱ་ག་མ། །པུཎྜ་སྱ་སཱ་ཤྲཱི་ཧེ་པ་རྦ་བྷཱུ་བཱ་བྷེ་དྷི་ཀསྟ་ཏ། །པ་ལ། །ནོར་ནི་ཇི་སྙེད་བྱུང་གྱུར་པ། །རང་གི་ཁྱིམ་དུ་གང་པོ་ལ། །དེ་ལས་ལྷག་པ་ཟོང་གིས་བྱུང་། །།
