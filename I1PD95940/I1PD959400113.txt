ཏ་ཐཱ་སསྟ་མ་ཧཱ་ཙ་ལཱན། །སཔྟ་ས་གཱ༔་ནུ་ལོ་མ་པྲ་ཏི་ལོ་མ་བྷི་ཅིང་། །དེ་བཞིན་དུ་ནི་རི་ཆེན་བདུན། །ཆུ་ཀླུང་བདུན་དང་རྗེས་འགྲོ་དང་། །བཟློག་འགྲོ་ཞེས་པའི་དྷོལཱུ་དྷི། །ཨ་ནུ་ཀཱུ་ལཱ་ནི་ལེརླསྨིན་པཱ་ར་མཱརྣོ་ཏི་པུ་ཎྱ་བཱན། །ཏ་ཏསྟཏྟཱ་ཆུ་ཡི་གཏེར། །རྗེས་སུ་མཐུན་པའི་རླུང་གིས་དེར། །ཕ་རོལ་འཐོབ་བོ་བསོད་ནམས་ཅན། །དེ་ནས་དེ་མྱ་ནཱ་མཱ་དྲིཛཱ་ཏིསྟི་མི་ར་མོ་ཧ་ཀྲྀ་ཏ། །ཡ་ཏཱཉྙོཇི་ཤ་ཏི་སསྦྱ་མ་མོ་མཚུངས་མིང་ཅན་རི༄༅། །རླུང་གིས་ཉིརབ་རིབ་རྨོངས་པར་བྱེད། །སྨན་ཆེན་དོན་ཡོད་ཅེས་པ་ཡིས། །གྷཱ་ཥཱ་མ་ཧོ་ཥ་དྷི། །ཨ་ཐཱ་བརྐྱ་བྷི་དྷཾ་བྷོ་དྷི་བི་རནྟེརྨ་ཏྲ་བ་ཡུ་བྷི། །མིག་གཉིས་གང་དུ་རྣལ་གནས་སྟེར། །དེ་ནས་ཀུན་འཁོར་ཞེས་ཆུ་གཏེར། །གང་དུ་རླུང་ནི་བེ་རཱམ་བྷས། །མཛྫ་ནོརྣཛྫ་ནེ་རྫནཱུསསྟ་བརྟེ་ཥཱ་ཏཱརྨ་ཏེ། །ཨ་བརྟ་ཁྱསྟ་འབྱིང་ཞིང་འབྱུང་བས་སྲོག་ཆགས་རྣམས། །ཀུན་འཁོར་བདུན་ལས་སྒྲོལ་བར་བྱེད། །དེ་ནས་ཀུན་འཁོར་ཏཤི་ལཿཤཾ་ཁ་ནཱ་བྷོ་ནི་བཱ་ཙ་ར། །གྷོ་རཔཱ་ཎ་ཧ་རོ་ཡ་ཏྲ་ཏྲི་དེ་ཤ་ཏཱ་ས་ཞེས་པའི་རི། །གང་ན་མཚན་རྒྱུ་དུང་གི་ལྟེ། །འཇིགས་སུ་རུང་ཞིང་སྲོག་འཕྲོག་པ། །སྐབས་གསུམ་པ་ཀྲྀཏ་སྦི་ཏ། །ཀྲྀཥྛ་སམྦྷ་བྲི་ཏཱ་ཏ་ཏྲ་ཤཾ་ཁ་ནཱ་བྷཱི་མ་ཧོ་ཥ་དྷི། །ཏྭཱ་ཡ་ཏེ་པུ་ཡང་སྐྲག་བྱེད་གནས། །དེ་ན་སྨན་ཆེན་དུང་གི་ལྕེ། །སྦྲུལ་ནི་ནག་པོས་དཀྲིས་གྱུར་པ། །མིག་དང་མགོ་ཅུ་སམྤནྣཾ་ནེ་ཏྲེ་ཤི་ར་སི་ཙཀྲི་ཏཾ། །ཨ་ཐ་ནི་ལོ་ད་ནཱ་མཱ་དཱི་རཀྟ་ཀློ་ལ་བཞག་པ་ན། །བསོད་ནམས་ལྡན་པ་ཡོངས་སུ་སྐྱོབ། །དེ་ནས་ཆུ་སྔོན་ཞེས་པའི་མཚོ། །གང་ན་སྲིན་ཡ་ཏྲ་རཱཀྵ་ས། །མ་ཀནཱ་དྷི་བྷུ་ཏྭཾ་བུདྡྷ་བི་དྷུ་བི་རཱན་བ་ཤེ། །པོ་མིག་དམར་ཡོད། །མ་གིར་ཞེས་བྱའི་སྨན་འཛིན་དང་། །སངས་རྒྱས་རིག་བྱེད་རིག་པའི་དབང་། །ཨ་ཐ་ནཱི་ལོ་དཱ་ནཱ་མཱ་དྲི་ནི་ལ་གི་བཀྟ་པཱ་ཙ་རཿ། །པྲ་དཱིཏ་ནེ་དྲོ་ཡ་ཏཱསྟེ་རཀ་དེ་ནས་ཆུ་སྔོན་ཞེས་པའི་རི། །གང་ན་མགྲིན་པ་སྔོན་པོ་ནི། །མཚན་རྒྱུ་མིག་ནི་རབ་འབར་བ། །སྲིན་པོ་ཧཱུཾ་པཉྩ་བྷིཤ་ཏི། །ཏ་ཏྲི་ཥ་དྷི་མ་མོ་གྷཱ་ཏྟཾ་རཀྵ་ཏྭཱ་ཤི་བི་ཥཱས་དཱ། །ལྔ་བརྒྱ་དག་དང་གནས། །དེ་ན་དོན་ཡོད་ཅེས་པའི་སྨན། །མཐོང་དང་དབུགས་དང་རེག་པ་དང༌། །—།
