བར་དྷི་རཱ་ཎཾ་པྲ་ཐུ་ས་ཏྭཱ་ནཾ་ནཱ་སྟི་ལོ་ཀེ་ཏུ་དུརླ་བྷཾ། །ཡ་ཨེ་ཥ་བི་ཐུ་ཏོ་བརྟན་ལྡན་སྙིང་སྟོབས་ཆེ་རྣམས་ལ། །འཇིག་རྟེན་ན་ནི་རྙེད་དཀའ་མེད། །གང་འདི་ཕྱོགས་སུ་གྲགས་གྱུར་དིཀླུ་དེ་ཤེ་སྨིནྡོ་ར་པཱུ་ཛི་ཏ། །ནྱ་གྲོ་དྷ་པཱ་ད་པ་ཏ་སྱ་པཱུ་ཛ་ཡ་སཏྠ་མཱ་ཅིང༌། །ཡུལ་འདིར་གྲོང་ཁྱེར་པས་མཆོད་པ@། །ནྱ་གྲོ་དྷ་ཡི་ཤིང་དེ་ལ། །མཆོད་པས་ཐམས་ཅད་ཐོབ་སྱ་ཏེ། །ཨ་པུ་ཏཱ༔པུ་ཏྲ་ཎཀེ་ཙིནྡིཛྙཱ་ནཱ་དྷ་ནི༔བརེ། །རོ་གི་ཎོ་བྱ་རུ་པར་འགྱུར། །དེ་ལ་ཞུས་པས་འགའ་ཞིག་ནི། །བུ་མེད་བུ་དང་ལྡན་པ་དང་། །གཞན་ནི་ནོར་མེད་ནོར་ཛཝཱ་རྩེ་དྲྀཀྵསྟ་དུ་པ་ཡཱ་ཙི་ཏི། །ས་ཏྲ་ཡཱ་ཙ་ན་ཙི་ཏྭཱ་ཁྱཔོ་ཡཾ་ནྱ་གྲོ་དང་ལྡན། །གཞན་ཡང་ནད་ཅན་ནད་མེད་མཐོང་། །ནྱ་གྲོ་དྷ་ཡི་རྐང་འཐུང་ནི། །མཆོད་རྟེན་སློང་བ་དྷ་པཱ་ད་པ། །ཕ་ལཾ་པུ་ཏཱ་བྷི་དྷནྟུ་ལཾ་དཱ་སྱ་ཀྱུ་ཙི་ཏ་མཙྩི་ཏ། །ཨི་ཏི་ཏེ་བདེན་ཞེས་བྱ། །འོས་པར་མཆོད་ན་ཁྱོད་ལ་ནི། །བུ་ཞེས་བརྗོད་པའི་འབྲས་སྟེར་འགྱུར། །ཞེས་པ་དེ་༈་པ་ཙཐུ་ཏྭཱ་ཏཱ་ནུ་བཱ་ཙ་ས་སསྨི་ཏ། །ཨ་ཧོ་ནུ་མོ་ཧཱ་ཏ་སྣེ་ཧཱ་ཡཱ་དག་ཚིག་ཐོས་ནས། །འཛུམ་བཅས་དེ་དག་ལ་སྨྲས་པ། །ཀྱེ་མ་རྨོངས་པའི་®མཛའ་བ་ལས། །ཁྱེད་ཡཾ་མུཎ་ཏཱ་མཱག་ཏཱ། །སཱ་ཀརྨ་པ་ར་ཏནྟཱ་མི་ན་ལོ་ཀེ་ན་ནི་ཡ་ཏེ་ནི༔་ལེ། །རྣམས་རྨོངས་པ་ཉིད་དུ་གྱུར། །རང་ལས་གཞན་དབང་འཇིག་རྟེན་འདི། །ངེས་པར་གཡོ་བ་མེད་པ་ལ། །སྦི་ཏཾ་ཀ་རོ་ཏི་པུཥྛ་ཏི་ཧ་ར་ཏི་ཧནྟི་ཀ་སྱ་ཀ། །སྭ་ཀརྨོ་པ་ན་ཏཾ་པཱ་དྱ་ཏུ་སུ་ཡིས་སུ་ཡི་བརྟན་པ་དང༌། །རྒྱས་དང་འཕྲོག་དང་འཇོམས་པར་བྱེད། །རང་ལས་ཉེ་བ་ཉིད་ཐོབ་ནས། །བྱ་ཏུ་ནྱ་སྱ་མུགྷ་དྷི། །སཱ་ལཱ་ལཱ་ལ་བ་མཱ་སཱ་རྱ་བྷེ་བ་ཤུཥ་སྱ་ཙམྨ་ཎ། །པྲིཀྵབློ་གྲོས་བླུན་པོ་གཞན་ལ་ཚིམ། །རང་གི་ཁ་ཆུའི་ཆ་མྱངས་ནས། །ཁྱི་ནི་པགས་པ་སྐམ་ལ་བཞིན། །ཤིང་པུ་ཏཱརྦ་དཔཾ་སཱ་མི་ཏྱ་ཏནཱུརྨ་བྷཱ་ཥི་ཏཾ། །དཱ་ཏུཾ་ཤཥས་ནཱ་ཀཱ་ལེ་པ་ཏྲ་གི་བུ་ནོར་སྐྱེས་བུ་ལ། །སྟེར་ཞེས་འདི་ནི་རྨོངས་པའི་གཏམ། །འདབ་མ་ཙམ་ཡང་དུས་མིན་དུ། །དེ་མཱ་ཏྲ་མ་བི་སྭ་ཡཾ། །ཏ་ད་དྷིཥྛ་དྲྀ་དེ་བཝེ་ཏ་པཱུ་ཛཱ་ལུབྷཱཀ་རོ་ཏི་ཏཏ། །པུཎྜོ་ཉིད་རང་ལ་སྦྱིན་ནུས་མེད། །གལ་ཏེ་དེ་ནི་བྱིན་རློབ་ལྷས། །མཆོད་པ་ལ་ཆགས་བྱེད་པ་ན། །བདག་།
