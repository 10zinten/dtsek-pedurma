དྷི་ག་ཏཱ་ཏུ་པཱ། །པུ་རཱ་ནནྡ་སྱ་ཚྭ་པ་ཏེཥྛ་ཏཱ་རཿཔི་ཤུ་ནཱསུ་ཏཱ། །བ་བྷུ་པུ་རཱ་ནི་ཐོབ་མ་གྱུར། །སྔོན་ནི་མི་བདག་དགའ་བོ་ལ། །ཕྲ་མ་ལྡན་པའི་བུ་བཞི་དང་། །ལྔ་པ་དག་ཀྱང་མེ་དན་མུ་ཁཔཉྩ་མ༔་གུ་ཎཱ་དྷི་ཀ། །ཀ་ལེ་ནཱ་པསྣ་པརླནྟས་བྷུ་པ་ཏི་ར་ཙིནྟ་ལོང་བཞིན། །ཡོན་ཏན་ལྷག་པ་བྱུང་བར་གྱུར། །དུས་ཀྱི་མཐར་ནི་ཉེ་བའི་ཚེ། །ས་ཡི་བདག་པོ་དེ་ཡིས་ཡཏ། །ཨེ་ཏེ་མ་དནྟེ་ཙ་ཏཱ་རོ་རཱ་ཛྷཾ་ནཱནྟནྟི་ཀརྐ་ཤ། །པུ་ཏྲེ་མ་མཱ་དན་བསམས། །བདག་གི་མཐར་ནི་རྒྱལ་སྲིད་ལ། །རྩུབ་མོ་འདི་བཞི་འོས་པ་མིན། །བདག་གི་བུ་ནི་མེ་མུ་ཁེ་རཱ་ཛཱ་ཤྲཱིཔྲ་ཏི་བིཝིཏཱ། །སུ་པྲྀཏྟེ་ལ་བྷ་ཏེ་ཤོ་བྷཾ་བྲཛྙཱ་ལོང་བཞིན། །ལེགས་སྤྱོད་ཤེས་རབ་དྲི་མེད་དང༌། །ལྡན་པས་རྒྱལ་སྲིད་དག་གི་དཔལ། །གཟུགས་བི་མ་ལ་ཤཱ་ཡི་ནི། །དྷཱ་ཏྲེ་ཏ་མཱ་ཏྭཱ་ན་སོ་བཱ་དཱིཏ་ས་བྷ་བདྷཱིནཱ་རེ་ཤཱ་རཱ། །བརྙན་མཛེས་པ་ཐོབ་ཅཔ་ཡིན། །འདི་བསམས་དེ་ཡིས་བློན་ལ་སྨྲས། །སྐྱེ་གནས་ཀྱིས་བསླང་བཙུན་མོ་ཡི། །ཀཱ་ཡོནྟ༔སུ་ར་བནེ་ཎ་ཡོ་ཅུཏྟཱ་ནེ་ན་པཱུ་ཛུ་ཏེ། །མོ་ལིན་ཀམྤ་ཏེ་ཡ་ཚོགས་ཀྱིས་མཆོད་པར་འགྱུར་བ་་དང༌། །གང་གི་དབུ་རྒྱན་མི་གཡོ་ལ། །ནོར་བུའི་རྐང་རྗེས་དག་ལའང་སྱ་པ་མེ་བ་མ་ཎི་པ་དུ་ཀེ། །དྭ་ར་དྲུ་མཱ་དྲི་བཱ་པཱི་ཥཱ་ནི་དྷི་ཥཀྵ་ཙ་པ་བྱ་མཉམ། །སྒོ་ཡི་ལྗོན་པའི་བང་སོ་ལ། །གཏེར་དྲུག་མཐོང་བར་གྱུར་པ་ནི། །ཁྱོད་ཀྱིས་མི་ཡི་དབང་པོར་ཏི༑། ཨི་ཏྱཱཀྟ་ཏྲི་དི་པཾ་ཡཱ་ཏེ་ནཱཾ་པ་ཏོ་མཎྜི་ཎཀྲ་མཱཏ། །ཏ་དུཀྟིརླཀྵ་བྱ། །ཞེས་བརྗོད་མི་བདག་མཐོ་རིས་སུ། །གཤེགས་ཚེ་བློན་པོས་མེ་ལོང་བཞིན། །དེ་ཡིས་བརྗོད་པའི་ཎེ་ཐུཀྲུ་རཱ་དན་མུ་ཁ་མཱི་ཤཱ་རཾ། །དྷརྨ་ནིརྞ་ཡ་ཀཱརྩེ་ཥཱ་ཡཾ་པཱ་མཚན་ཉིད་ཀྱིས༅། །རིམ་པ་རྣམས་ཀྱིས་དབང་པོར་བྱས། །ཆོས་ནི་ངེས་པའི་བྱ་བ་ལ། །རྒོལ་དང་དི་པྲ་ཏི་བ་དི་ན། །བི་ལོ་ཀྱི་པ་སཱ་ཡཾ་ཏ་སྦྱརྱ་ཡེ་ཛ་ཡ་མ་པ་ར་ཛ་ཡེ། །ཕྱིར་རྒོལ་རྣམས་ཀྱིས་གང་། །མཐོང་བ་རྣམས་ཀྱི་རང་ཉིད་ནི། །རྒྱལ་ཕམ་རིགས་པ་དག་ལ་གནས། །པུ་རཱ་ནི་རཱ་བྷི་སནྣཱ་ན་པཱལྟ་བི་ཤ་པ་ཀིལྤ་ཎཾ། །བཱཧྨ་ཎཾ་དཎྜི་ནཾ་ནཱ་མ་སྔོན་ནི་རྒྱུ་མཚན་མེད་པ་རུ། །མི་འདོད་སྡིག་པ་ཐོབ་གྱུར་པ། །བྲམ་ཟེ་དབྱུག་ཅན་ཞེས་པ་ལ། །།
