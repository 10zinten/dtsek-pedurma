འོ་ཏཱ་ཀྲྀ་ཏི། །ཨ་ཏཱན་རེ་མ་ཧཱ་མ་བི་ཤ་ལི་ཀ་ག་ཎཱ་གྲ་ཎི། །ཀ་ནྱ་འདི་ལྟར་གྱུར། །སྐབས་དེར་ཆེན་པོ་ཞེས་བྱ་བ། །ཡངས་པ་ཅན་གྱི་ཚོགས་ཀྱི་གཙོས། །ཨ་མྲའི་མཱ་མྲ་བ་ནཱཏ་པྲ་པ་ཀ་ད་ལཱི་སྐནྡྷ་བི་ནིརྒ་ཏྭཾ། །སཱ་ཏ་སྱ་བྷ་བ་ནེཀཱན་བཌྷ་ནགས་ལས་ཆུ་ཤིང་གི༄༅། །སྡོང་པོ་ལས་འཁྲུངས་མཛེས་མ་ཐོབ། །དེ་ཡི་ཁྱིམ་དུ་མཛེས་མ་དེ། །དལ་བུ་མཱ་ནཱ་ཤ་ནི༔ཤ་ནི། །བི་ད་དྷེ་བི་པུ་ལོ་པཱི་ཏི་ད་ན་ཙིནྟཾ་ཙ་ཙེ་ཏ་སི། །དལ་བུས་བསྐྱེད་བསྲིངས་ཏེ། །དགའ་བ་ཆེ་དང་སྦྱིན་པ་ཡི། །བསམ་པ་ཡིས་ཀྱང་སེམས་མ་བཟུང༌། །པྲ་ཎ་ཡཱ་དཱ་མྲ་པཱ་ལཱི་ཏི་བནྡྷ་བྷིསཱ་ཀྲྀ་ཏཱ་བི་དྷཱ། །སུ་ཊུ་ཧཱི་ན་མི་བ་ཏུཏྟཱ་གཉན་གྱི་མཛའ་ལས་དེ་ཡི་མིང་། །ཨ་མྲ་སྐྱོང་མ་ཞེས་སུ་བཏགས། །བུ་མོ་དམན་པ་བཞིན་བཏང་ནས། །བ་ལུ་ཡོ་བ་ན་མ་ད་དེ། །ཏདྲི་བ་ཧོདྨ་ཏ་སྭཱ་ཐ་ན་སེ་ཧེ་ཏཏྲི་ཏུརྒ་ཎ། །།ལང་ཚོ་དག་ནི་བཟུང་བར་གྱུར། །དེ་ནས་དེ་གཏོང་བརྩོན་པའི་ཕ། །མཛེས་མ་ཚོགས་ཀྱི་ཉེར་སྤྱད་བྱ། །ག་ཎོ་པ་བྷོ་གྷ་ཀ་ནྱ་ཏི་ས་མ་ཡ་སྱ་བྱ་ཏི་ཀྲ་མཾ། །པི་ཏ་རཾ་དུཁ་ཞེས་པའི་དམ་ཚིག་འདས་གྱུར་པ། །ཚོགས་པ་དག་གིས་བཟོད་མ་གྱུར། །དེ་ནས་སྡུག་བསྔལ་གྱིས་སནྟ༈་པ་མེ་ཏྱ་ཐ་ཛ་གཱ་ད་སཱ། །བྷ་བཱ་མི་ག་ཎ་བྷོ་གྱི་བ་ཀིནྡྷེ་ཥ་གདུངས་པའིཡེ། །ཕ་ལ་བཏུད་ནས་དེས་སྨྲས་པ། །བདག་ནི་ཚོགས་ཀྱི་སྤྱད་བྱ་ཉིད། །གལ་ཏེ་འོན་ཀྱང་ས་མ་ཡོ་ཡ་དི། །ཨེ་ཀ་་པ་རི་ནཱ་རྱ་སྱ་པྲ་བེ་ཤཧཱུཾ་པ་དེསྦི་ཏི། །པ་ཋཀཱནྟཱ་དམ་ཚིག་འདི། །གཅིག་གི་སྟེང་དུ་གཞན་དག་ནི། །མི་འཇུག་རང་གི་གནས་སུ་སྡོད། །ཉི་མ་རེ་ལ་པ་ཎ་ཤ་ཏིཔཉྩ་བྷིཔྲ་ཏྱ་ཧ་མ་མ། །སམྦྷ་ཧེ་ནི་བ་བི་ཙ་ཡཀཱརྨོ་བེ་ཧྥ་ནི་བདག་གི་ཟོང་། །ཀན་པ་ཎི་བརྒྱ་ཕྲག་ལྔ། །ཞག་བདུན་དག་ནི་ཁྱིམ་དག་ནི། །རྣམ་པར་བལྟ་བྱ་ནཱ་རྱ་དཱ། །ཨི་ཏྱསྨིན་ས་མ་ཡེ་བ་དྡྷསརྦ་སྱ་བ་བྱ་ཏི་ཀྲ་མཱི། །ཨི་ཏི་ཏཏ་གཞན་དུ་མིན། །ཞེས་པའི་དམ་ཚིག་འདི་འདས་ན། །ཀུན་གྱི་སྤྱད་བྱ་ཉིད་འགྱུར་བའོ། །ཞེས་པ་དེ་ས་མ་ཡཾ་ཛྙཱ་ཏཱ་ཏཏྲི་ཏུརྦ་ཙ་ས་ག་ཎ། །ཨ་ཀ་རོ་ད་བཱ་ཌྷ་མི་ཏྱཱཀྟ་ཛྲ་ཡིས་ཀྱི་དམ་ཚིག་དག། ཤེས་ནས་དེ་ཡི་ཕས་སྨྲས་པ༄༅། །ཚོགས་ཀྱིས་རུང་ཞེས་སྨྲས་ནས་ནི། །གུས་།
