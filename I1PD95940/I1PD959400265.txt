པྲ་སཱ་དེ་ན་བྷཏྟ་ས་ར་མ་མརྦ་ཎེ། །ཀུཎྜི་ནྱ་ནི་པ་ཐཱ་ཏཱ་སྱཔཱ་ཏྲེ་བྷིནྡྷཱ་ཀ་རཾ་དྲག་པོ་ཡིས། །འབྲས་ཆན་རབ་མང་ཕུལ་བའི་ཚེ། །མཛེ་མོ་དེ་ཡི་ལག་པ་ཡི། །སོར་མོ་བྲུལ་ནས་སྣོད་དུ་གུ་ལི། །ཏ་ཏཔ་པཱ་ཏ་ཀ་མི་བ་ཏྲཀཱ་ནི་ཏྱཱ་ཀ་ཌེ་བ་རཾ། །དེ་བཱ་ནཱཾ་ཏུ་ཥི་ཏཱ་ཁཱ་ལྷུང་། །དེ་འོག་དེ་ཡིས་སྡིག་པ་བཞིན། །མི་རྟག་ལུས་དག་དོར་ནས་ནི། །དགའ་ལྡན་ཞེས་བྱ་ལྷ་རྣམས་ནཾ་ནི་ལ་ཡེ་ས་མ་ཛཱ་ཡ་ཏེ། །ཤཀྲ་སྟ་དདྡྷཱ་ཏཾ་ཛྙཱ་ཏཱ་དཱ་ན་པུ་དྡྷོ་དི་དི་ཏ་ཀྱི། །ནང་དུ་ཡང་དག་སྐྱེས་པར་གྱུར། །རྨད་བྱུང་དེ་ཤེས་བརྒྱ་བྱིན་ནི། །སྦྱིན་པའི་དགེ་ལ་གུས་རྒྱས་ར༑་། ཡཏྠཏ་ཀཱ་ཤུ་པ་སཏྲ་ཊཾ་སུ་དྷ་ཡ་ས་པ་པུ་ར་ཡཏ། །སུ་དྷཉྫ་པས། །འབད་པའི་འོད་སྲུངས་དམ་པའི་སྣོད། །བདུད་རྩི་དག་གིས་ཡོངས་སུ་བཀང་། །དགེ་སློང་ཎི་པ་སོ་བྷིཀྵནི་སྤྱི་ཧལྡི་ཎ་ལཱི་ལ་ཡ། །པྲ་ཤ་མཱ་མྲྀ་ཏ་སཾ་པུཎ་སྩཀྲེ་རབ་ཞིའི་བདུད་རྩི་ཡིས། །ཁེངས་དེས་བདུད་རྩི་བྱིན་ལ་ཡང་། །དགའ་མེད་རྩྭ་ཡི་རྩེ་མོ་ཡིས། །ལྷུང་པཱ་ཏྲ་མ་དྷོ་མུ་ཁཾ། །བྷ་ཛནྟེ་པྲ་ཎ་ཡ་པཱིཏི་ཀྲི་པ་ཎེ་ཥཱ་ཀྲི་པ་ཀུ་བཟེད་ཐུར་དུ་ཁ་བསྟན་གྱུར། །དམ་པའི་སྙིང་རྗེ་རྒྱས་པ་རྣམས། །དམན་ལ་མཛའ་གཅུགས་དགའ་བ་ལཱ༑། སཾ་ཏ༔སམྤཏ་ས་མཱ་དྷཱ་ཏ་བ་ད་ནེ་མཱི་ལི་ཏཱ་ད་ར། །ཐུ་ཏྭཱ་ཏྭཾ་ཏུ་སྟེན༈། ཕུན་སུམ་ཚོགས་པ་འབར་བ་ཡིས། །བཞིན་ལ་གུས་པ་དམན་པར་གྱུར། །དེ་ནི་དགའ་ལྡན་ཎི་ཏོ་དེ་བ་ནི་ཀཱ་ཡ་ནི་ར་ཏྭཾ་ཀྲྀ་པ། །པྲ་སེ་ན་ཛི་ཏ་བྷ་ག་བ་ཏཉྩ་ཀྲེ་ལྷ་ཚོགས་ན། །གནས་པ་ཐོས་ནས་མི་ཡི་བདག། གསལ་རྒྱལ་གྱིས་ནི་བཅོམ་ལྡན་གྱི། །མཆོད་སྟོན་བྷོ་ཛྙཱ་དྷི་བཱ་ས་ནཾ། །རཱ་ཛྷསྟ་སྱ་ཤྲ་ཧེ་དྲྀསྨཱ་ལགྷི་མཱརྱ་ཀ་རི་ཎ། །།ལྷག་པར་གནས་པར་བྱས། །རྒྱལ་པོ་དེ་ཡི་ཁང་པ་ནི། །དཔལ་འབྱོར་ངོ་མཚར་བྱེད་མཐོང་ནས། །ཨཱཏྱཱ་ནནྡེ་ན་བྷ་ག་བཱན་པྲཏྠསྟ་ཏ་པུ་ཎྱ་མ་བྱཱ་དྷ་ཏ། །པུ་རུ་ཤྲ་ཧ་པ་ཏེའཕགས་པ་དགའ་བོས་བསོད་ནམས་དེ། །དྲིས་ཤིང་བཅོམ་ལྡན་གྱིས་བཀའ་སྩལ། །སྔོན་ནི་ཁྱིམ་བདག་སཱུ་ནུ་རཱ་རེ་དཱདྷཱ་ས་ཏྭཾ་ག་ཏ། །ཀྵེ་ཏྲ་ཀརྨ་ཎི་སཾ་སཀྟཀྵ་ཏ་གཱ་ཅིག་ྋགི་བུ། །དབུལ་བ་ལས་ནི་འབང་ེཉིད་གྱུར། །ཞིང་ལས་དག་ལ་ཡང་དག་ཆགས། །ལྟོགས་ཤིང་།
