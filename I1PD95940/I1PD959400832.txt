བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁིརི་ཎཾ་ག་ནཱ་ནཾ། །བཀྵཀ་ཊཱི་པཱ་ཎི་ག་ལཱ་བ་སཥ་བི་ཙི་ཏྲ་སཱུ་ཏཱ་ཎི་བྷ་དྲེགས་པ་འཕྲོག་པ་མཐོང་། །དེ་ཡི་སྙིང་ག་རྐང་ལག་མགྲིན་པ་ལ། །་བཏགས་པའི་སྐུད་པ་རྣམ་པར་བནྟི་ཏ་སྱ། །ཨིནྡྲ་ཡ་དྷཱ་ཡ་ཏ་ནཾ་བྷཱ་ནི་སནྟི་མ་མཱ་པི་ཀིཾ་ཏཱ་ཏན་ཏ་བི་པཱ་བཀྲ་གྱུར་པ། །དབང་པོའི་མཚོན་ཡངས་དག་དང་མཚུངས་པ་ཡོད། །ཡབ་གཅིག་བདག་ཀྱང་ཅི་སླད་དེ་ནི༑། *ཨ་རཱུ་པི་མེནྟཛྙཱ་ན་ཏཱི་བ་ཏ་སྱ་བཱ་ཎི་ས་ཏྭཾ་པ་སྲ་ཏི་མཱ་པི་ཡ་སྣུ། །ཐུ་ལྟ་མིན། །གང་གི་ཚིག་གི་རྒྱ་ཆེ་མཚུངས་པ་ཡང་། །:ཅི་ཡང་ཡོད་མིན་ཕུལ་དུ་བྱུང་བའི་སྒྲ། །ནམ་*ཅི་ཡང་ཡོད་མིན་སོགས་བོད་སྐད་ཀྱི་རྐང་པ་གཅིག་ཆད་པ་འདར་ལོས་གསར་དུ་བསྒྱུར་བ་སླར་དཔྱད། །ཏཱ་མ་ཡཱ་ཙུ་ཏ་བ་ནེ་ཀ་དཱ་ཙིནྣ་ཀོ་ཀཾ་ལཱ་ནཾ་ན་ཙ་ཥཟླ་ད་ནཾ། །མནྡཱ་ཀི་ནཱི་ཡང་ཙཱུ་ཏའི་ནགས་སུ་བདག་གིས་ནི། །ཐོས་པ་ཁུ་བྱུག་གི་མིན་བུང་བའི་མིན། །དེ་ནི་ད་ལྟའང་བདག་མཛྫ་ན་ནཱི་ལ་ཕེ་ན་སཱ་མཱ་ན་བྷཱ་སཱ་ན་བ་བལྟ་ལེ་ན། །ཏ་ནི་ཡ་པཱི་ཏ་སྱ་བ་པུ་ནང་སྒྲོགས་པ་བཞིན། །དལ་བབ་ལག་བཀྲུས་དབུ་བ་ཆགས་པ་ཡི། །འོད་དང་མཚུངས་པ་ཤིང་ཤུན་བི་བྷཱ་ཏི་ནེ་དཾ་པྲི་ཡཾ་བལྟ་ལ་ཀཾ་མ་མཱ་བྱ། །ནི་བི་ཤུ་བཀྑཾ་བ་ད་ནཱ་ར་བིནྡེ་གསར་པ་ཡིས། །ཤིན་ཏུ་ཕྲ་བ་དེ་ཡི་ལུས་དག་མཛེས། །བདག་གི་ལུས་ནི་ལག་པས་ཡུན་རིང་བསྡམས། །ནི་བི་ཎྱ་དོབྷཾ་སུ་ཙི་རཾ་བ་པུརྨེ། །ཙཀྲ་ཛ་ཡ་པྲ་སྤུ་རི་ཏཱ་དྷ་རོ་སོ་གདོང་གི་པདྨར་བཞིན་རས་རྣམ་པར་བཀོད། །བསླབས་པས་མཆུ་བསྐྱོད་འདི་ཡི་དགའ་བ་ཡིས། །ཧནཱ་མྲྀ་ཏ་སྤུན་བི་ཤེ་ཥ་དི་ཏཱཾ། །ར་མེ་ཀཱ་མཱ་ཀཱནྟ་བི་ལཀྵ་ཎེ་ན་ནི་བ་བདུད་རྩི་རེག་པའི་ཁྱད་པར་བསླབ་པར་བྱས། །མཛེས་པས་ཀྱི་རྣམ་པར་མཚན་པ་དེ་མེད་ན། །བདག་ཀཱ་ཎཾ་ཏེ་ན་བི་ནཱ་ཙ་ནེ་སྨིན། །ཀ་རོ་ཏི་མེ་ཡདྨ་ཏ་ཤཱ་ས་ནེ་ན་དྲི་ཤོགི་བཟོད་པ་ནགས་འདིར་སྐད་ཅིག་མིན། །གང་གིས་བརྟུལ་ཞུགས་བསྟན་པས་བདག་ཉིད་ལ། །པ་དཾ་མོ་ཧ་ས་ཁཱི་ན་ནི་དཱ། །དྲིཌི་སྟ་དྲྭ་ལོ་ཀ་ན་མཱི་ཧ་ཏེ་མེ་ཐུ་ཏིརྦ་རྨོངས་པའི་གྲོགས་པོ་གཉིད་ནི་གནས་མི་བྱེད། །བདག་གི་མིག་ནི་དེ་ལ་ལྟ་བར་འདོད། །དེ་ཚིག་མེད་ན་ནཱ་ཏངྒ་ཙ་ན་ཙཱསྟེ། །ཏཙིནྟ་ཡཱ་ཏཱམྦ་ཏི་དྷཱནི་ཏཱནྟ་མནྟཾ་ན་ཛཱ་ཏི་ཏ་ནུ་བྱཱ་ཡཱ་ཐོས་པ་ཡོངས་མི་གནས། །ལུས་ལ་གནོད་བྱེད་བློ་ནི་ནང་དག་ཏུ། །མི་གསལ་དེ་ཡི་བསམ་པས་མཆོག་ཏུ་—།
