ཡཱ་ཙི་ཏྲ་པརླནྟ་བལྟ་ལ། །དི་བྱ་ཏ་དཱ་ཐ་མོདྡྷཱ་ཏཾ་ཕ་ལ་མཱ་སྭཱ་དི་ཏསྨྲ་རབ་ཏུ་དང་བའི་ཆུ་ལྟར་མཛེས། །གང་གི་འབྲས་བུ་གཞན་གྱི་ཚོགས། །བསྡོམ་པར་ཡོངས་སུ་རྟོག་བྱེད་ཡཱ༑། ཡེ་ནཱ་ནྱ་ཕ་ལ་སཾ་བྷ་རོ་ནི་ཡ་མཔ་རི་ཀཝཾ་ཏ། །གཙཱ་པ། །དེ་ཡི་གནས་སུ་རབ་འཁྲུངས་པའི། །འབྲས་བུ་མཆོག་ནི་བདག་གིས་མྱོང་། །དེ་ཡི་དཀའ་ཐུབ་མི་ཏཱ་ད་ནུ་ཛྙཱ་ཏཱ་ཀཱནྟཾ་ཏ་སྱ་ཏ་པོ་བ་ནཾ། །ཏཏོ་ཛ་ནེ་པྲ་སཏྟ་མེ་ནཱ་རྱ་ཏྲ་ར་ནགས་ཚལ་དུ། །ཁྱོད་ཀྱི་རྗེས་གནང་བདག་མཆིའོ། །དེ་ཡི་གཞུང་བཟང་ལ་ཆགས་པའི། །བདག་བློ་མ་ཏེ་མ་ཏི། ་། ཨི་ཏི་ཏདྲ་ཙ་ནཾ་ཐུ་ཏྭཱ་བིསྨི་ཏསྨ་ར་སཱུ་ཙ་ཀཾ། །ཏཱ་གཞན་ན་དགའ་བ་མེད། །ཅེས་པ་དེ་ཚིག་འདོད་པ་ནི། །གསལ་བྱེད་ཐོས་ནས་ཡ་མཚན་ཞིང་། །ལང་མུ་བཱ་ཙ་མུ་ནིརྨུབྦྷ་ཡོ་བ་ནོནཱ་ད་ཤོ་ཀི་ཏ། །དྲྀཏྠ་སཱུ་ཡཱ་བྷུ་ཛ་གོ་སོ་མ་རྒྱ་ཚོ་སྨྱོན་པར་དོགས་པ་ཡི། །ཐུབ་པས་མཛེས་མ་དེ་ལ་སྨྲས། །ཁྱོད་ཀྱིས་མཐོང་དེ་ལག་འགྲོ་ནི། །རིན་རཏྭཱ་བི་བྷུ་ཥི་ཏ། །བྷ་བནྟི་མུ་ན་ཡཔུ་ཏྲི་ན་དཱི་ཛི་ཧཱ་ན་བྷོ་གི་ན། །ཆེན་གྱིས་བརྒྱན་ཡིན་ནམ་སྙམ། །བུ་མོ་ཐུབ་པ་རྣམ་དག་ནི། །ལོངས་སྤྱོད་ཅན་མིན་ལྕེ་གཉིས་མིན། །ས་སྣེ་ཧརྒུ་རུ་བྷིས་རཱ་པ་རི་ཎ་ཏོ་དུ༔ཁ་པྲ་དེརྨོ་ད་ཀི་རཱ་ཡཱ་ཏོ་བི་ཁ་ཡོ་ཡུལ་བཞིན་ཡོངས་སུ་གྱུར་ཚེ་སྡུག་བསྔལ་རབ་སྟེར་མ་བཅས་ལྕི་བའི་ལ་དུ་ནི། །ཤིན་ཏུ་དངར་ཅེབ་དག་ར་བཱ་ཏི་མ་དྷུ་རིཔྲི་ཏི་ཤྲཱི་ཐཱ་མཱ་ཀྲྀ་ཐ། ་། མུདྡྷེ་ཀཱ་མ་ཕ་ལཱ་ནི་ཏཱ་ནི་པ་ར་སཾ་གིས་འོངས་པ་ལ་ནི་དོན་མེད་དགའ་བར་ཡོངས་མ་བྱེད། །མཛེས་མ་འདོད་པའི་འབྲས་བུར་རོ་ལྡན་ཉོན་རྩྭ་ཡཱ་ས་སིཀྵ་ནྱ་ལཾ་ཡེ་ཥ་མེ་བ་བི་ཥོ་པ་མེ་ན་ཛ་ན་ཏཱ་སྭཱ་དེ་ན་སཾ་མཱུ་མོངས་དག་གིས་ཉེབྲན་པ་གང་དེ་རྣམས། །ཤིན་ཏུ་ངེས་པར་དུག་དང་མཚུངས་པའི་ཉམས་སུ་མྱོང་བ་བསྐྱེད་ཙྪ་ཏི། ་། ཨེ་ཧི་དན་ཡ་མེ་རཱུ་རཱ་ཥལྟ་བཏྠ་མུ་ནེསུ་། །ཨི་པས་རྒྱལ་བར་བྱེད། །ཚུར་ཤོག་རེ་ཤིག་ཐུབ་པའི་སྲས། །དེ་སུ་རིང་ནས་བདག་ལ་སྟོན། །ཞེས་བརྗོད་ཏུཏྟཱ་ཏ་ཊ་ནཱི་ཏཱི་རཾ་ཏ་ཡཱ་ས་ཧ་ཡ་ཡོ་མུ་ནི། །ས་བི་ལོ་ཀྱ་པ་རིཏྟི་ར་བྲ་ཀླུང་གི་ངོགས་སུ་ནི། །དེ་དང་ལྷན་ཅིག་ཐུབ་པ་སོང་། །དེ་ཡིས་ཆུ་ཀླུང་ངོགས་སུ་ནི། །ས་ཡི་བདག།
