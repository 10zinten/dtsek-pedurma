ན་ལ་བྷུ་ཏེ། །ཏ་སྩཌྷ་ཏུ་ལྕི་དྲ་བི་ཎཾ་དཱི་ཡ་ཏྭཾ་ཡ་དི་ཤཱཀྱ་ཏེ། །ཡཱ་བདྷཱ་གྱིས་ཐོབ་པར་འགྱུར། །དེ་ཡི་རིན་དང་མཚུངས་པའི་ནོར། །གལ་ཏེ་ནུས་ན་སྦྱིན་པར་མཛོད། །ཇི་སྲིད་དཤ་བྷིནྡྲརྟོསྟེ་ན་སད་རནྣ་སཾཉྩ་ཡ། །ནི་ཧི་ཏསཏྱ་ཤྲཱི་ལེ་ན་པ་ཙྪཱཉོ་ལོ་ནི་བཅུ་གཉིས་སུ། །བདེན་པའི་ངང་ཚུལ་དེ་ཡིས་ནི། །རིན་ཆེན་དམ་པའི་ཚོགས་བསྒྲུབས་པ། །རིན་སོ་ས་མམྤི་ཏྭཾ། །ཨེ་ཏདྡྷ་པཱ་ལ་དུ་ཧི་ཏུསནྡེ་ཤཾ་དྷུ་ཏ་བྷཱ་ཥི་ཏཾ། །ཐུ་ཏཱ་དུ་འོས་པ་དེ་སྦྱིན་མཛོད། །ས་སྐྱོང་བུ་མོ་དག་གི་འཕྲིན། །འདི་དག་ཕོ་ཉས་བཤད་པ་ནི། །ཐོས་པས་གྲོ་ཧ་པ་ཏིསཉྫ་རཏྭཱ་ནཱ་དཱ་ཡ་སཱ་དཱ་ར། །སཱ་ཡཾ་རཱ་ཛ་ཧུཾ་ཧ་ག་ཏཱ་ཁྱིམ་བདག་གུས་བཅས་པས། །རིན་ཆེན་ཐམས་ཅད་བླངས་བྱས་ནས། །རང་ཉིད་རྒྱལ་པོའི་ཁྱིམ་སོང་སྟེ། །ད་ཏཱ་སོ་ཌི་ཎ་ལཱི་ལ་ཡཱ། །སཱུཀྟ་རནྣཾ་ཏ་དཱ་དཱ་ཡ་པ་པཱ་ཊོཀྐཎྜཱུ་ཏོ་མུ་ཧུ། །དེ་ལ་རྩྭ་ཡིས་རྩེ་དགའ་བྱིན། །དེ་ཚེ་ལེགས་བཤད་རིན་ཆེན་ཐོབ། །དགའ་བས་ཡང་དང་ཡང་དུ་བརྗོད། །ཡདྨཀྟི་ཙིཏུ་ཁ་མ་ནུ་པ་མཾ་བཱ་ཏྱཱ་མཱ་ཧྲུན་རཾ་བཱ་ཡོ་ཡ༔སཏྭཱ་པྲ་ས་པ་བི་ལ་གང་གང་ཕྱི་འམ་ནང་གི་བདེ་བ་དཔེ་མེད་ཆུང་ཟད་དག་དང་ནི། །གང་གང་སྙིང་སྟོབས་ཀྱིས་བསྐྲུན་ཡ་སངྒིརྨ་ཡ༔སིདྡྷི་ཡོ་ག། །ཡེ་ཡེ་ཀྲྀཥཱ་པྲ་ཤ་མ་བི་མ་ལཱ་སནྟ་ཏཱ་ནནྡ་ལཱ་བྷཱམཚན་རྒྱས་པ་གྲུབ་པའི་སྦྱོར་བ་དང་། །གང་གང་སྲེད་པ་རབ་ཞི་དྲི་མེད་རྒྱུན་དུ་དགའ་བ་ཐོབ་པ་དག། པཱ་ཀོཏིལྟ་ས་ས་ཕ་ལ་བྷ་རཔུ་ཎྱ་ཀལྤ་དྲུ་མ་སྱ། །ཨོ་དཱརྨ་ནི་དྷི་ནཱ་དེ་དེས་བསོད་ནམས་དཔག་བསམ་ཤིང་གི་འབྲས་བུ་སྨིན་ཅིང་རྒྱས་པའི་ཁུར། །རྒྱ་ཆེར་གྱུར་པའི་གཏེར་དེ་ཏེ་ན་སཉྫ་ར༔་སུ་བྷཱ་ཥི་ཏཾ། །ཧཱུྃ་ཧཱི་ཏཾ་ཏྲཾ་པ་ཏི་ཐུ་ཏྭཱ་ཏཾ་པ་པྲཙྪ་ས་ཡིས། །རིན་ཆེན་ཀུན་གྱིས་ལེགས་བཤད་ནི། །བླངས་པ་ཐོས་ནས་མི་ཡི་བདག། ཡ་མཚན་བཅས་པིསྨྲ་ཡ། །།རཏྭཱ་ནི་ཀསྨཱ་དུབྷཾ་ཛུ་སཱུཀྟ་མཱ་ཏྲ་མི་དཾ་ཏཱ་ཡཱ། །གྲོ་པས་དེ་ལ་དྲིས། །ཅི་སླད་རིན་ཆེན་རབ་བཏང་ནས། །ལེགས་བཤད་ཙམ་ནི་ཁྱོད་ཀྱིས་བཟུང་། །བུ་ཧཱི་ཏཾ་བཱ་ལི་ཀཱ་བཱ་ཀྱ་ས་ར་ལཿཔྲ་ཏུ་ཡིདྷཱན། །པཱུརྞ་སུ་བྷཱ་ཥི་ཏེནྟུ་མིདྷ་ནཾ་མོ་འདི་@ཡི་ཚིག་དག་ལ། །དྲང་པོ་ཁྱོད་ནི་ཡིད་ཆེས་ལྡན། །ལེགས་བཤད་ཀྱི་ཨེནི་ས་གཞི་གང་། །ནོར་།
