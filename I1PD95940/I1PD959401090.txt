བསྟན་འགྱུར། །འཁྲི་ཤིང་། ཁེ་་པཱ་ཀ་ཤ་བ་ལཾ་སཾ་སཱ་ར་བལླི་ཕ་ལཾ། །ཨི་ཏི་བྲུ་བ་ཎཾ་ཏྲཾ་པ་ཏིསྟ་མ་པ་རྣམས་ཀྱི་རང་ལས་འཁོར་བའི་འཁྲི་ཤིང་སྨིན་པའི་འབྲས་བུ་བཀྲ། །ཞེས་སྨྲ་དེ་ལ་མི་བདག་གིས། །དགེ་བོ་ཙཥོ་ཏུ་ཀཱཏ། །ཏ་བཱ་པི་སུ་མ་ཏེ་ཀ་སྱ་ཕ་ལ་མེ་ཏཇུརྐརྨ་ཎ། །མཚན་དག་ལ་རབ་སྨྲས་པ། །བློ་བཟང་ཁྱོད་ཀྱིས་འདི་ཡང་ནི། །ལས་ངན་ཅི་ཡི་འབྲས་བུ་ཡིན། །སོ་བྲ་བཱི་ད་བྷ་ཐཾ་པཱུཊྚ་བཱ་རཱ་ཎ་ཧཱུཾ་ཀུ་ཐཱ་བྷི་དྷ། །།གོ་ཙོ་རཀཱུ་ར་ཙ་རི་ཏསནྣཾ་ས་དེས་སྨྲས་བཱ་རཱ་ཎ་སཱིར་སྔོན། །ཀུ་ཐ་ཞེས་པ་གདུག་སྤྱོད་ཅན། །བ་ལང་བརྐུ་ཞིང་དེ་ཡི་ཤས། །ཟས་བི་ཧི་ཏཱ་ཤ་ན། །ཀ་དཱ་ཙིངྒོ་བི་གྷཱཏྟ་ཚྭ་སཾ་ཏཱཾ་བྷ་ཡ་བི་དྲུ་ཏཾ། །ནི་བསྒྲུབས་པ་དག་ཏུ་གྱུར། །ནམ་ཞིག་བ་ལང་བསད་པ་ཡི། །ཤ་བཟུང་འཇིགས་ནས་བྲོས་པ་བདག། གོ་ར་ཎསཱ་ནུ་ཙ་རཱཀོ་པཱཏྠ་མ་བྷི་དུ་དྲུ་རུ། ་། ཏ་ཏ་ས་མཱི་པེ་བ་ལང་བསྲུང་བ་འཁོར་བཅས་རྣམས། །ཁྲོས་པ་ཡིས་ནི་མངོན་པར་གཅེས། །དེ་ནས་བདག་གིས་ཉེ་བར་ནི་ཀླུ་སྱ་པྲ་ཏྱ་གྲ་པི་ཤི་ཏམྨ་ཡཱ། ་། པྲ་ཏེ་ཀ་བུདྷེཤྩ་རོ་ཡ་མི་ཏི་ཏེ་ཏཱཾ་པྲ་དནི་ནི༑། གསར་བའི་ཤ་དག་རབ་བཀོད་ནས། །རང་སངས་རྒྱས་ནི་ཆོམ་རྐུན་ནོ། །ཞེས་པ་དེ་དག་རྣམས་ཏཾ༑། ས་ཏིརྒོ་བ་དྷ་པཱ་པོ་གྲ་ཀྲོ་དྷཔི་ཤི་ཏ་དན་ནཱཏྲ། །བྷ་དྷོ་ཡ་ལ་བསྟན། །བ་ལང་གསོད་སྡིག་དྲག་པོས་ཁྲོས། །དེ་དག་གིས་དེ་ཤ་མཐོང་ནས། །འབད་ནས་འདི་ནི་མི་ཏི་ཡནྡྷེ་ན་ནི་བདྷེ་བནྡྷ་ནེ་པྲྀ་ཏ། །པཉྩཱཏྟཱ་པཱ་ད་ཐཱ་བྷེ་ཏྱ་མ་ཡཱ་དྭཱ་ད་ཤ་གསད་བྱ་ཞེས། །ཚོགས་པ་རྣམས་ཀྱིས་བཅིངས་ཏེ་བཟུང་། །དེ་ནས་འགྱོད་པས་བདག་གིས་ནི། །ཉི་མ་བྷིཇི་ནི། །ཀྲྀ་ཏ་པཱ་པོ་ཧ་མི་ཀྟ་ཀཱ་རཱ་གཱ་རཱཏྨ་མོ་ཙི་ཏ། །བཅུ་གཉིས་ན་སོང་སྟེ། །བདག་གིས་སྡིག་བྱས་ཞེས་བརྗོད་ནས། །དེ་ནི་བཙོན་ར་དག་ནས་བཏང་། །ཏཏྟཱ་པ་བ་ཀཾ་ན་ར་ཀེ་ཥཱ་ནུ་བྷཱུ་ཡ་མ་ཡཱ་ཙི་རཾ། །ཛརྨརྨསྨིནྣ་པི་པཱདྷཱསྡིག་པ་དེ་སྨིན་བདག་གིས་ནི། །དམྱལ་བར་ཡུན་རིང་ཉམས་མྱོང་ནས། །སྐྱེ་བ་འདིར་ཡང་ཉོན་མོངས་ནི། །གླེ་ཤོ་རཱ་ད་བཱ་བནྟི་ཀ། །ཨི་ཏུཏྟཱ་ཚྭ་པ་མཱ་མནྡྷ་པྲ་ཎཱ་མཱསྟཱ་སུ་རཱ་སུ་རཱ། །ལོ་ནི་བཅུ་གཉིས་རབ་ཏུ་ཐོབ། །ཅེས་བརྗོད་མི་བདག་ལ་གསོལ་ནས། །ལྷ་དང་ལྷ་མིན་རབ་འདུད་ཐོབ། །—།
