ལ། པྲ་ཌྷ་དལྕི་ཏ་ཙཱ་རུ་ཙཱ་མ་ར་སི་ཏ་ཙྪ་ཏྲ་རྨི་ཏཱསཾ་པ་ད། །གདུགས་དཀར་འཛུམ་པའི་ཕུན་ཚོགས་དག་གིས་འཇིག་རྟེན་ན་ནི་ལེགས་ལྡན་ཐུགས་ནི་རྣམ་མཛེས་དང༌། །ཡཙྪིཀྵཔ་ཏི་ཏཐི་ཏ་ཐུ་ཏི་ཡ་ཤཀསྡུ་ར་པཱུ་རོཛྫཱ་ལཾ་སྭལྤ་དཱ་ནཱ་ཀ་གང་ཡང་གྲགས་སྙན་ག་པུར་རྒྱུན་གྱིས་དྲི་མེད་ཐོས་པ་སིམ་བྱེད་རབ་ཏུ་འཕྲོ་བ་དག། སྦྱིན་པའི་ཆ་ཤས་དེ་ཎ་སྱ་ཏཏྠ་ལ་མ་ཧོ་དཱ་ནཾ་ནི་དཱ་ན་ཤི་ཡ། །ཨ་བྷུ་དུ་པོ་ཥ་དྷཾ་ནཱ་མ་་་འབྲས་ཤིན་ཏུ་ཆུང་ངུའོ་ཨེ་མ་སྦྱིན་པ་དཔལ་གྱི་དང་པོའི་རྒྱུ། །འོ་མའི་མཚོ་ལས་བདུད་རྩི་བཞིན། །བྷུ་བྷཾཀྵ་སྱ་བི་བྷཱ་བ་ཏེ། །བི་བུ་དྷཱ་བྷི་མ་ཏཱ་ཀཱིཏིསུ་དྷཱ་དུབྷོ་ད་མཁས་རྣམས་མངོན་འདོད་གྲགས་པ་ཡིས། །གང་ཞིག་རྣམ་གསལ་ས་འཛིན་ནི། །གསོ་སྦྱོང་འཕགས་ཞེས་དྷེ་རི་བ། །བ་སུ་དྷཱ་མ་བ་ཏོ་ཡ་སྱ་བ་སུ་དྷཱ་མ་བ་ཏཔུ་རཿ། །ན་ནཱ་མ་བྱ་བ་བྱུང་། །ནོར་འཛིན་བསྲུང་བ་ནོར་དང་ནི། །གཟི་བརྗིད་ལྡན་པ་གང་མདུན་དུ། །མི་ཡི་བདག་པྲ་ཎ་ཏོ་ཀ་སྱ་ན་ནཱ་མ་བྲྀ་པ་ཏེཤི་ར། །ཤུདྡྷ་དྷི་རི་བ་དྷརྨེ་ཎ་དཱ་ནེ་ནེ་བ་པོའི་སྤྱི་བོ་དག། ངེས་པར་ཕྱག་ལ་སུས་མ་བཏུད། །ཆོས་ཀྱིས་དག་པའི་བློ་གྲོས་བཞིན། །སྦྱིན་པས་བརྩེ་ད་ཡཱ་ལུ་ཏཱ། །བི་བྷཱུ་ཏེཟི་ན་ཡེ་ནེ་བ་བྷུ་ཥི་ཏཱ་ཡེ་ན་སྐྱུ་ར་བྷཱུཏ། །གུ་བ་ལྡན་ཉིད་བཞིན། །དུལ་བས་རྣམ་པར་འབྱོར་པ་བཞིན། །གང་གིས་ས་ནི་བརྒྱན་པར་གྱུར། །ཡོན་ཎི་ནཔྲ་ཤུ་ཁཾ་ཤ་སྱ་བ་བྷུ་བེརྫུ་དྷུ་ཏེསྦི་ཏ། །ཡ་སྱ་སརྦ་ཏ་པ་ཏྲ་སྱ་མུདྡྷི་ཏན་ཀུན་ལྡན་རིགས་སུ་མཐོ། །ཀུན་གྱི་གདུང་སྐྱོབ་ཟླ་བའི་འོད། །གང་གི་གནས་པའི་སྤྱི་བོར་གྱུར། །ས་སཏྠ་མ་ཧཱི་བྷཾ་ཏཾ། །ཡ་སྐྱེ་ཤྭ་ར་ཤི་རཿསྣ་ཡི་ཤུ་བྷཾ་གཾ་གཱ་ཛ་ལོཛྫ་ལཾ། །ཡི་བདག་པོ་ཀུན་གྱིའོ། །གང་གི་གྲགས་པ་འཇིག་རྟེན་རྒྱན། །དཀར་ཞིང་དབང་ཕྱུག་སྤྱི་བོར་གནས། །བྷ་མེ་ཏུད་བྷཾ་ལོ་ཀེ་ཥཱ་བྷཱུ་བ་ནཱ་བྷ་ར་ཎཾ་ཡ་ཤ། །།ཀརྟུཀྲ་ཏུ་ས་ཧ་སྭཱ་ཎཾ་ས་དྲི་མེད་གང་གཱའི་ཆུ་བོ་ནི། །རབ་མང་འཇིག་རྟེན་རྣམས་སུ་རྒྱུ། །མཆོད་སྦྱིན་སྟོང་གི་བྱེད་པ་པོ། །མིག་ཧ་སྦ་གྷ་དྷི་ཀ་ཤི་ཡ། །ཡ་སྱ་ཥཎྜི་ས་ཧ་སཱ་ཎི་ཀ་ལ་ཏྲ་སུ་དྲྀ་ཤཱ་མ་དྷཱཏ། །ཀ་སྟོང་པ་ལས་ལྷག་པའི་དཔལ། །གང་གི་བཙུན་མོ་ལྷ་མཛེས་ནི། །སྟོང་ཕྲག་དྲུག་ཅུ་བྱུང་བར་གྱུར། །ནམ་—།
