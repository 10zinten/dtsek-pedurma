དྲྀཥཱ་པྲ་བཱ་ཙ་སཏྠ་ཛྷཤ་ནིཔྲ་ཎ་ཡི་བཏྠ་ལ། །ཀཱ་ཧུ་པ་སྱསུ་དྷཱིནཱ་གུས་པས་མཉེས་གཤིན་ཀུན་མཁྱེན་གྱིས། །དལ་བུ་ཡིས་ནི་རབ་ཏུ་གསུངས། །སྟོན་པ་འོད་སྲུངས་དག་མ་ཤཱསྐུ་རཱ་སཱི་དུ་པཱ་ས་ཀ། །ཏཏྲ་སཱ་དཱསྟརྞནྟེ་ཤིཀྵ་པ་ད་ནི་གི་ནི། །དགེ་བསྙེན་བློ་བཟང་ཞེས་བྱ་བ། །དེ་ཡི་དྲིན་གྱིས་བསླབ་གནས་ནི། །ཡོངས་རྫོགས་ཐོབ་ཅིང་བློ་ཥཎཱ་དྷི། །ས་དྲྀཥཱ་གཱི་ཏི་ཀུ་ཤ་ལཾ་གཱདྷབྷི་ཀ་ག་ཊོཏྟ་མཾ། ཌ། པྲ་ཎི་གནས་གྱུར། །དེ་ཡི་དྲི་ཟའི་ཚོགས་ཀྱི་མཆོག། གླུ་དབྱངས་མཁས་པ་མཐོང་གྱུར་ནས། །དྲི་ཟའི་དྷཱ་ནཾ་པྲ་བི་ད་དྷེ་གནྡྷནྡྷ་དྷི་པ་ཏྭཾ་པྲ་ཏི། །ས་ཏེ་ན་པྲ་ཎི་དྷཱ་ནེ་ན་སུ་པྲ་ཡ་ཏྲཱ་བདག་པོ་ཉིད་ཀྱིས་ནི། །ཆེད་དུ་སྨོན་ལམ་རབ་ཏུ་བསྒྲུབས། །དེ་ནི་སྨོན་ལམ་དེ་ཉིད་ཀྱིས། །རབ་དགའ་མུ་པཱ་ག་ཏ། །སཾ་པཱཔ་དྷརྨ་བི་ན་ཡཤི་གླུ་པ་ད་པ་རི་གྲྭ་ཧཱཏ། །ཉིད་དུ་ཉེ་བར་གྱུར། །བསླབ་པའི་གནས་ནི་ཡོངས་བཟུང་ནས། །འདུལ་བའི་ཆོས་ནི་ཡང་དག་ཐོབ། །སུ་བྷ་དྲ་སུ་པྲ་ཡཱ་བེ་ཏོ་ཀྲྀ་ཏོ་ཀུ་ཤ་ལི་ནོ་མ་ཡཱ། །པ་རཱ་བྷི་མ་ཏ་ཀྲྀ་ཏེ་ཥཱ་ནིཤྩ་རབ་བཟང་@རབ་དགའ་འདི་དག་ནི། །བདག་གིས་དགེ་བ་ལྡན་པར་བྱས། །གཞན་གྱི་མངོན་འདོད་བྱས་ཐཱ༔སཧྥ་ཐཱ་བ་ཡཾ། །པུ་ཛི་ཏོ་ཙ་ཏ་སཾ་བྷོ་བྷཱུཙྭ་ག་རེ་སཱརྒ་པ་ལ། །ཀུན་ཏུ་བདག་ནི་གདུང་བ་མེད། །གྲོང་ཁྱེར་མཆོད་འོས་ཞེས་བྱ་བ། །མཐོ་རིས་དག་དང་སནྣི་བྷེ། །ཤ་ཏ་ཀྲ་ཏུ་རི་བ་ཤཱི་མཱན་བི་ཛ་ཡནྟཱ་བྷི་དྷེ་ནཱི་པ། །ཏ་མཚུངས་པར་ནི། །མི་བདག་དཔལ་ལྡན་བརྒྱ་བྱིན་བཞིན། །རྣམ་པར་རྒྱལ་བ་ཞེས་པ་བྱུང༌། །བསོད་སྱ་པུ་ཎྱ་པྲ་བྲྀཏྟ་སྱ་དྷམྨོ་མཱནཱ་ནུ་སཱ་རི་བྷི། །བྲ་ཛདྷཱསྟི་དི་པཾ་པོ་རེནམས་ལ་ནི་རབ་ཞུགས་དེའི། །ཆོས་ལམ་རྗེས་སུ་འབྲང་བ་ཡི། །གྲོང་པ་མཐོ་རིས་བགྲོད་རྣམས་ཀྱིས། །པཱུ་ཛི་ཏཱ་སུ་ར་བྷུ་མ་ཡ། །རཱ་ཛཱ་པུ་ཏྟཱ་ནུ་བྷཱ་བེ་ན་སཱརྒ་མཱ་པཱུ་ར་ཏཾ་ན་རི། །ལྷ་ཡི་ས་གཞི་ཁེངས་པར་གྱུར། །རྒྱལ་པོ་དེ་ཡི་བསོད་ནམས་མཐུས། །མི་ཡི་མཐོ་རིས་ཀུན་གང་བ། །དྲྀཏྟཱ་ཡུཀྟི་ཙ་ཀཱ་ར་སྱ་ས་ཏྭཾ་ཛྷཱ་ཏུ་སུ་རེ་ཤཱ་རཿ། །ཏཔ་སྨིནྣིརྨི་ཏཱཀུ་རས་མཐོང་ནས་ལྷ་ཡི་དབང་པོ་ཡིས། །སྙིང་སྟོབས་ཤེས་འདོད་སྦྱོར་བ་བྱས། །དེ་ནས་བརྒྱ་བྱིན་གྱིས་སྤྲུལ་པའི། །།
