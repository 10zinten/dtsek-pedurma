སུ་ཀྲྀ་ཏཱརྦ་དེ་ཤ་ཀ་ཏ་ཡཱ་ཀིརཱ་བྷི་ན་ས་དཱ་པཱ་ཎཱ་ནཾ་ཛ་ན་ཀཿཀ་ཐོ་ན་ཛ་ན་ཀཿདོན་སྟོན་བྱེད་པས་སློབ་དཔོན་རྟག་ཏུ་སུ་ཡིས་མངོན་པར་དགའ་མི་བྱ། །སྲོག་རྣམས་སྐྱེད་བྱེད་བླ་མ་རྣམས་པཱུ་ཛུ་གུ་རུ་ཎཱ་གུ་རུ། །ཏ་ཏཱ་བི་ཥ་ཎ་གཌྷ་དྷཱ་ར་ཎ་བྷ་ར་ཀླུ་ཤ་སྱ་མཱ་ཏུཀྱི་བླ་མ་ཕ་ནི་ཇི་ལྟར་མཆོད་མི་བྱ། །དེ་བས་ཀྱང་ནི་མཆོག་ཏུ་སྐད་ཅིག་མངལ་གྱི་ཁུར་བཟུང་མ་ཡི་ངལ་དུབ་པ་རཾ་ཀེ་ནཱ་མྲྀ་ཏྱཱ་བི་དྷཱ་ན་པུ་ཏྱཱ་ནི་ཙ་ཡེ་ནཱ་སཱརྱ་ཏེ་ནིབྲྀ་ཏི། །མཱ་ལ༑། བུ་ལོན་དག་པར་སྒྲུབ་བྱེད་བསོད་ནམས་ཚོགས་ནི་ཅི་ཡིས་དག་པ་ཐོབ་པར་འགྱུར། །དབེན་པ་ཏུནྣི་ཀཱ་ར་ཀ་ལ་མ་སྱ་བི་བེ་ཀ་པཱ་ཀ་སཾ་པཱཔ་སཱ་ཏཀ་ཀུ་ལཀི་ལ་ཡཀ་རོ་མ་སྨིན་སྡིག་པའི་ཚོགས་ནི་ཡང་དག་ཐོབ་གྱུར་པ། །གང་གི་མ་ལ་སྨོད་པའི་ཆ་ཡང་ངེས་པར་སྒྲུབ་བྱེད་ཏི༑། ཏ་སྱ་ཛ་ལཛྫ་ལ་ན་ཛཱ་ལ་ཤི་ཁཱ་ཀ་ལཱ་པཔཱ་པ་བྲ་ཏེ་བྷ་བ་ཏི་མཱུདྡྷི་ཛ་ཊཱ་པ། །སྡིག་པའི་བརྟུལ་ེཞུགས་དེ་ཡི་སྤྱི་བོར་རལ་པའི་ཚོགས་དག་ནི། །རབ་འབར་མེ་ཡི་དྲ་བའི་རྩེ་མོའི་ཀ་ཊ་ཡུ། །ནི་ཀཱ་ར་ཀཱ་རི་ཎཱ་མཱ་ཏུརྨ་ཡཱ་སྱ་བ་ར་ཛརྨ་ནི། །པཱ་པ་ཚོགས་པ་དག་ཏུ་འགྱུར། །སྐྱེ་བ་གཞན་དུ་མ་ལ་ནི། །སྨོད་པར་བྱེད་པ་བདག་གིས་ཀྱང་། །སྡིག་དམོད་ཤཱ་པ་མ་ཡནྟ་པ༔པྲYཔརཱཥ་བི་ཤ་པ། །བཱ་རཱ་ཎ་སུཾ་ཐཾ་ཧ་པ་རང་བཞིན་མི་འདོད་པ། །ཡོངས་སུ་རྫོགས་པའི་གདུང་བ་ཐོབ། །བཱ་རཱ་ཎ་སཱི་®དག་ཏུ་སྔོན། །ཁྱིམ་ཏིརྨི་ཀྲོ་ནཱ་མ་པུ་རཱ་བྷ་བཏ། །བ་སུནྡཱ་རཱ་ནཱ་མ་ཏ་སཱུ་ཨི་ཏི་ཧྥུ་ད་ཏི་བལྦ་བྷཱ། །ཛཱ་བདག་མཛའ་བོ་ཞེས་པ་བྱུང་། །དེ་ཡི་ཆུང་མ་ནོར་འཛིན་མ། །ཞེས་པ་ཤིན་ཏུ་ཡིད་འོང་གྱུར། །དེ་ཡི་བུ་ཏ་མཱ་ཏྲེ་ཏུ་པུ་ཏྲེ་ཥཱ་ཏ་སྣུཔཱསྟེ་ཥཱ་པཉྩ་ཏྭཾ། །པཉྩི་མེ་བ་ཡ་པི་ཤཱི་མཱ་ནེ་ཀ་པཱུ་རྣམས་སྐྱེས་ཙམ་ནི། །ལྔ་པ་ཉིད་ནི་ཐོབ་པའི་ཚེ། །ན་ཚོད་མཐའ་རུ་བུ་དག་ནི། །དཔལ་དང་ལྡན་པ་ནུ་ར་ཛཱ་ཡཏྲ། །བི་པ་ཏི་བྷཱི་ཏྱ་ཏ་སཱུ་པི་ཀ་རྩྭ་ལྱཱ་མ་ཀ་རོཌི་ཏཱ། །མི་གཅིག་བཙས་གྱུར། །དེ་ཡང་འཆི་བས་འཇིགས་པ་ཡིས། །ཕ་ཡིས་བུ་མོ་ཞེས་པར་བཏགས། །དེ་ཡི་ཏྲ་ཀ་ནྱ་ཀཱ་ནཱ་མ་སོ་ཏེ་ནཱ་བྷུདྲི་ཐུ་ཏཿཤི་ཤུ། །ས་མུ་དྲ་ག་མ་ནེ་ཏ་སྱ་ཛ་ན་ཀེ་ནི་མཛའ་བོའི་བུ་མོ་ཞེས། །བྱིས་པ་དེ་ནི་གྲགས་པར་གྱུར། །དེ་ཡི་ཕ་ནི་རྒྱ་མཚོ་རུ། །སོང་ཞིང་ཤི་བར་གྱུར་།
