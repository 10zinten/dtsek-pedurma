བསྟན་འགྱུར། །འཁྲི་ཤིང་། །ཁེ་ནི་དྷིསྟི་རཾ་སཱརྦ་ལཀྵ་དཱ་ཡཱ་ནུ་གཱ། །ཏསྨིན་ལེ་ཛ་གངྒཱ་ས་བྱ་གྲཀཱ་ལ་གཉིས་རྗེས་འབྲང་། །ཆུ་གཏེར་དག་གི་ངོགས་སུ་ཕྱིན། །དེ་དུས་དུས་ནི་གཞན་པ་བཞིན། །འགྲོ་བ་ཟ་ཨི་བཱ་ས་ར། ་། ཨ་བྷཱུ་ཏ་ཛ་ག་རོདྡྷཱ་རཿཡཾ་ཝཱ་ཎཿསྭཱ་ར་བི་གྲ་ཧ། །ཏེ་ཏཾ་བ་ལ་ཆགས་པ། །འཇིགས་རུང་ལུས་རྒྱས་སྤྲུལ་ཆེན་ནི། །ཕམ་པ་ཞེས་པ་བྱུང་བར་གྱུར། །སྐྱེ་བོ་ཀུན་སརྦ་ཛ་ན་ཡཱ་སཾ་པཾ་ཏཱ་ས་མི་བ་དུས་ཧཾ། །བི་ནཋ་དྷེརྨ་དྲྀཏྟཱ་བ་བ་བྷཱུ་བུདྷ་གྱི་ཉོན་མོངས་ཀྱི། །སྐྲག་པ་བཞིན་དུ་བཟོད་དཀའ་དེ། །མཐོང་ནས་བརྟན་པ་རྣམ་པར་ཉམས། །དེ་ཊི་ཏཱ་ཨི་པ། །བྷོ་གེ་ན་བྷོ་གི་ནཱ་ཏེ་ན་ས་མནྣཱདྷེ་ཏ་པ་ཐི། །དག་དཀྲུགས་པ་བཞིན་དུ་གྱུར། །གདེངས་ཅན་དེ་ཡི་ལུས་ཀྱིས་ནི། །ལམ་དག་ཀུན་ནས་བསྐོར་བའི་ཚེ། །ནིལྕེཋཀཱ་ལ་བལྭ་གྲ་བནྟི་ནསྟེ་པྲ་བྲ་ཀྲུ་ཐུ། །།ཏེ་ཏཱཾ་མ་ཤེ་ཥ་ཏྲི་ད་ཤ་བསམ་བྲལ་དུས་ཀྱི་ཁ་སྒོར་གནས། །དེ་དག་རྣམས་ཀྱིས་ཆོ་ངེ་བཏོན། །མ་ལུས་ལྷ་རྣམས་བསྟོད་པ་ཡིས། །སྟ་བཿཤ་ར་ཎ་རཱ་བི་ཎཾ། ་། ཨཱ་ཀྲནྡསཏྠ་དིབྦྷ་པཱི་པོརླ་ཡོ་ཀ་རུ་ཎ་སཱ་ན། །སྐྱབས་པའི་སྒྲ་ལྡན་དེ་དག་གིས། །ཆོ་ངེ་ཕྱོགས་ཀུན་ཁྱབ་བྱེད་པ། །སྙིང་རྗེའི་སྒྲ་ནི་རྒྱས་པར་གྱུར། །ཡ་ཤཀེ་ཤ་ར་སཾ་བྷོ་ཐ་སིཾ་ཧཤེ་ལ་གུ་ཧཱ་ཤ་ཡ། །ཏཾ་ཤསྡོ་མནྡ་ར། དེ་ནས་སེང་གེ་རི་ཕུག་གནས། །གྲགས་པའི་རལ་ཅན་ཞེས་པ་དང་། །དེ་ཡི་གྲོགས་པོ་གླང་པོ་ཆེ། །ཐུ་ཏྭཱ་ཏདྲ་ཡ་པཉྩཀུཉྫ་ར། །ཨཱ་སནྣ་ཏཱ་ཎ་སནྣདྷོ་ཀ་རུ་ཎཱ་ཀྲྀཥཱ་མཱ་ན་སོ། །དྷཱི་མན་དྷ་ར་ཡིས་སྒྲ་དེ་ཐོས། །རྒུད་པ་སྐྱོབ་པ་ལ་ཆས་ཤིང་། །ཡིད་ནི་སྙིང་རྗེས་དྲངས་གྱུར་པ། །དོན་རཱ་བཱ་ཛགླ་ཏུསཱརྦ་སཾ་རཀྵ་ཎ་ཀྲྀ་ཏྲ་ཀླུ་མོ་ཎོ། །ཏ་ཏསིཾ་ཧས་མཱ་ཏུ་རྒྱ་ག་ཛེནྡྲ་མཐུན་སྐྱོབ་ལ་དུས་བྱས་པའི། །བརྟན་པ་དེ་དག་འོངས་པར་གྱུར། །དེ་ནས་གླང་ཆེན་རི་བོ་ཡི། །ལུས་གི་རི་བི་ག་ཧཾ། །བེ་གཱནྟེ་པ་ཏྭཱ་ཛ་ག་རཾ་ཙ་ཀཱ་ར་ག་ཏ་ཛཱི་བི་ཏཾ། །ཅན་དག་ལ་སེང་གེས་ཞོན། །ཤུགས་ཀྱིས་བསྣུན་ནས་སྦྲུལ་ཆེན་ནི། །སྲོག་དང་བྲལ་བ་དག་ཏུ་བྱས། །པྲ་ཎ་པྲ་བཱ་ས་ས་མ་ཡཏ་སྱ་ནི་ཤཱ་ས་མ་རུ་ཏ། །ཙ་ཀ་ར་ད་ཧ་ནོ་བྷཱ་རཱི་དེ་ཡི་སྲོག་ནི་འགྲོ་དུས་སུ། །དབུགས་ཀྱི་རླུང་ནི་སྲེག་བྱེད་དག། རབ་ཏུ་འབར་བས་སེང་གེ་དང་། །—།
