བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀིའིསུ་ཁ་ཤྲཱི་ཡ་ཏ་ཡོ་བ་ནཾ། །ཨ་ཐུ་ནཱ་བི་བྷ་བཱ་བྷོ་གེ་བྷོ་ག་ཡོ་གེ་བསྐྱོད། །བདེ་བའི་དཔལ་གྱི་ལང་ཚོ་སོང་། །ད་ནི་འབྱོར་པ་ལོངས་སྤྱོད་རྣམས། །སྤྱོད་དང་ལྡན་ལ་ན་མེ་རུ་ཙི། །དྷ་ནཾ་བྷཱུ་མིཀྲྀ་ཧཾ་དཱ་རཿསུ་ཏཱ་ཤྲཱི་ཏཱ་པ་རིཙྪ་དཱ། །ཨ་ཧོ་ནི་ར་བདག་མི་སྲིད། །ནོར་དང་ཁྱིམ་དང་ཆུང་མ་དང་། །བུ་དང་གཡོག་པོ་ཡོ་བྱད་རྣམས། །མཐའ་མེད་ནད་བ་དྷི༔པཾ་སྭཱ་མཱ་དྷི་བྱཱ་དི་པ་རི་གྲ་ཧ། །ཡ་ཐཱ་ཡ་ཐཱ་བི་བརྙནྟེ་གྷིམྨོབྷཱ་བི་དང་གདུང་བ་སྟེ། །ཀྱེ་མ་སྐྱེས་བུའི་ཡོངས་འཛིན་ནོ། །སོ་གའི་ཚ་བ་མི་བཟད་དཔལ། །ཇི་ལྟ་ཇི་ལྟར་ཥ་མཱཤི་ཡ། །ཏ་ཐཱ་ཏ་ཐཱ་ཛཱ་ལ་ཏྲེ་བ་དྲྀཥྛ་ཏཱ་པཤ་རཱི་རི་ཎཾ། །པྲ་བྲིདྡྷི་རྣམ་འཕེལ་བ། །དེ་ལྟ་དེ་ལྟར་ལུས་ཅན་རྣམས། །སྲེད་པའི་ཡོངས་སུ་གདུང་བ་འཕེལ། །སྐྱེ་བ་ཀུན་ཏུ་ར་པི་བིཏྟོ་གྷི་རཱ་ཛམྦྷོ་པཱརྫི་ཏིཙྰ་ཎཾ། །ལ་བ་ཎནྟི་རི་བ་ཛ་ལེཆི་བྲོཙྩཾ་ཉེར་བསྒྲུབས་པས། །ནོར་གྱི་ཚོགས་པ་རབ་རྒྱས་ཀྱང་། །ལན་ཚྭའི་མཚོ་ཡི་ཆུ་ཡིས་བཞིན། །མི་རྣམས་ནི་བ་ཛཱ་ཡ་ཏེ། །ནཱསྟི་ནཱསྟི་ཏྱ་སཾ་ཏོ་ཥཱད་ཡ་ཨེ་བ་དྷ་ནི་རཱི་ཛ་པ། །པུ་ནབྷ་བེ་སྲེད་བྲལ་ཉིད་མ་ཡིན། །གང་ཞིག་ནོར་ལྡན་ཚོག་ཉམི་ཤེས། །མེད་དོ་མེད་དོ་ཞེས་ཟློས་པ། །གལ་ཏེ་དེ་བྷ་བེཏ་ཀོ་བ་ས་ཨེ་བ་པྲ་ཤ་མཱདྨ་དི། །ཀཾ་བིཏྟི་རྡུནི་མིཏྟེཀ་ལི་ཀ་ལ་ཧ་ཡིས་རབ་ཞི་ན། །ཡང་སྲིད་དག་ཏུ་སུ་ཞིག་འགྱུར། །རྩོད་དང་འཐབ་མོ་རྨོངས་དང་ཆགས་པ་ཆེན་པོ་རྗེས་མོ་ཧ་ལོ་བྷཱ་ནུ་བྲྀཏྟིཀིཾ་བྷོ་གིཁྲི་པ་ཡོ་གིཅུ་ས་ན་ཤ་ཏ་པ་ཏ་ནཱ་བྷཱུ་པ་སུ་འབྲང་བའི་ལྟས་ངན་ནོར་གྱིས་ཅི། །རྒུད་པ་བརྒྱར་ལྟུང་ཡང་དག་ཆགས་པས་ནད་སྐྱེད་ཅེའབྲལ་བར་སཾ་སཀྟ་རོ་གི། །ཀིཾ་བཱ་མི་ཐཱ་བྷི་མཱ་ནིརྣར་པ་ཏི་ས་ད་ན་པྲལྟ་འགྱུར་བའི་ལོངས་སྤྱོད་དག་གིས་ཅི། །ཡིད་ཉམས་གཡོག་ལ་མི་བདག་ཁེངས་@པ་ཐོབ་པའི་མངོན་པའི་ང་སེ་པཱ་བ་མཱ་ནི་རསྨིན་བི་རཱ་གྱི་མེ་བ་ཀླུ་ཡ་ས་མ་ཡ་བྷ་ཡེ་བྷོ་གྱི་མཱ་རོ་གྱ་རྒྱལ་དག་གིས་ཅི། །འདིར་ནི་ཆགས་བྲལ་ཁོ་ན་འཇིག་དུས་འཇིགས་པར་ནད་མེད་ལོངས་སྤྱོད་དག་ཏུ་འོས་ཡོ་གྱི། །ཨ་ཏི་ཀཱནྟེ་ཀཱ་ལེ་སཱ་ཛ་ན་སུ་ཧྲི་དཱ་ལོ་ཀ་བི་མ་ལེ་ས་པ་ཡིན། །རང་གི་སྐྱེ་བོ་སྙིང་སྡུག་དྲི་མེད་སྣང་བའི་དུས་ནི་འདས་གྱུར་ཅིང་། །ལྷག་པར་མི་བསྲུན་—།
