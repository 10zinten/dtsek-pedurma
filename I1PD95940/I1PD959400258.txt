ལོ་ཙ་ཛ་གཏྟཱ་ཡེ། །ཨཱ༄རྱ་ནི་ལ་དྲི་ཤསྟ་ཏོ་མ་བ་ལོཛྫི་ཀྱང་བཞག་པར་མཛད། །དེ་ནས་བཅོམ་ལྡན་བརྩོན་འགྲུས་དང་། །སྟོབས་ནི་རྫོགས་པས་གྱད་རྣམས་ཏཾ༑། བཱིརྱ་བྷ་ག་བ་ཏོ་བཱིགྱུ་པྲ་ཎ་ཏཱཙྩཾ་བ་བྷཱ་ཥི་རེ། །ཨ་ཧོ་མ་ཧཱ་བྷཱ་བོ་ཡཾ་དག། ངོ་མཚར་མིག་ནི་མི་གཡོ་བས། །མཐོང་ནས་ཕྱག་འཚལ་དེར་གསོལ་པ། །ཀྱེ་མ་ཁྱོད་ཀྱི་མཐུ་ཆེན་པ་ལ་བཱི་ད་ཡསྟ་བ། །ནིཉྩ་ཡཱ་དྷི་ག་མེ་ཡ་སྱ་ན་པྲ་གgསུ་རཱ་པོ། །སྟོབས་དང་བརྩོན་འགྲུས་རྒྱས་པ་འདི། །གང་ཞིག་ངེས་པར་རྟོགས་པ་ཡི། །ནུས་པ་ལྷ་རྣམས་ཨ་པི། །ཨ་ནུ་གྲ་ཧ་པྲ་བྲྀཏྟེ་ན་པ་ལེ་ན་གུ་རུ་ཎཱ་ཏ་བ། །ཨ་དྷོ་ག་ཏི་ནི་ལ་ཡང་མེད། །ཁྱེད་ཀྱི་སྟོབས་དང་བླ་མ་ནི། །རྗེས་སུ་འཛིན་ལ་རབ་ཞུགས་པས། །རྡོ་བ་བྱིང་ཞིང་མཀྵེ་ཡཾ་ཛ་ན་ཏེ་བོཛྫུ་ཏཱ་ཤི་ལཱ། །བཱིརྱ་པྲ་ཛྙཱ་བ་ལཱ་དཱི་ནཾ་པྲ་གཏིང་དག་ཏུ། །སོང་འདི་སྐྱེ་བོ་བཞིན་དུ་བཏོན། །ངོ་མཚར་མཆོག་མཛད་ཁྱོད་ཀྱི་ནི། །བརྩོན་མཱ་ཎཱ་བ་དྷི་ནིཉྩ་ཡཾ། །ཨ་པི་ཛཱ་ནཱ་ཏི་ཏེ་ཀཉྩི་དཱཕྱིརྱ་ཏ་ར་ཀཱ་རི་ཎ། །འགྲུས་ཤེས་རབ་སྟོབས་སོགས་ཀྱི། །ཚད་ཀྱི་མཚམས་ནི་ངེས་པ་དག། སུ་ཞིག་གིས་ཀྱང་ཅང་ཤེས་སམ། །ཨི་ཏི་བྲུ་བཱ་ཎཱ་ནཱམྱ་ནཱི་ཐུ་ལཱ་བ་ལོ་ཀྱ་ཏཱན། །ཏ་ཧཱུཾ་ཤི་ལཱ་ཡཱ་མཱ་སཱི་ནདེ་སྐད་སྨྲས་%ཞིང་ངོ་མཚར་ནི། །གཡོ་མེད་དེ་རྣམས་ལ་གཟིགས་ནས། །རྡོ་བ་དེ་ལ་རབ་བཞུགས་ཏེ། །པོ་བཱ་ཙ་བྷ་ག་བཱན་ཛི་ན། །བ་ལཾ་ཡ་ཏེ་ཀཱི་བྷཱུ་ཏཱ་ནཾ་བྷཱུ་ཏཱ་ནཾ་བྷ་བ་ན་ཏྲ་བཅོམ་ལྡན་རྒྱལ་བས་བཀའ་སྩལ་པ། །སྲིད་གསུམ་འབྱུང་པོ་རྣམས་ཀྱི་སྟོབས། །གཅིག་ཉིད་དུ་ནི་གྱུར་པ་ཡ༑། སུ་ག་ཏ་པ་ཏ་དེ་ཀ་སྱ་ལོ་ཀེ་ནི་བ་སཾ་གི་ཀཾ་པ་ལཾ། །ཨོཾ་བྷཾ་པི་ཀུནྟི་གང་། །བདེ་གཤེགས་གཅིག་གི་སྟོབས་དེ་དང༌། །འཇིག་རྟེན་ན་ནི་མཉམ་པ་མིན། །ཆུ་གཏེར་ཆུ་ནི་རམྦྷ་དྷེརྫ་གནྟི་པ་ར་མཱ་ཆུ་བྷི། །ཤཀྱུ་ནྱ་ལཾ་ག་ཎ་ཡི་ཏུཾ་པྲ་བྷཱ་བུམ་པ་དང་། །འགྲོ་རྣམས་ཕྲ་རབ་རྡུལ་དག་གིས། །བགྲང་ཡིབ་རུ་ནི་ཤིན་ཏུ་ནུབའི། །བདེ་གཤེགས་བོ་ར་ཏུ་སོ་ག་ཏ། །སཾ་ཊུ་སུ་མེ་རོརྨོ་བེཏྟི་ཏུ་ལ་མཱ་ནེ་ན་ཏ་ཏཱ་ཏ། །སུ་ག་མཐུ་ནི་མ་ཡིན་ནོ། །གང་གི་སྲང་གི་ཚད་ཀྱིས་ནི། །ལྷུན་པོའི་དེ་ཉིད་གྲངས་རིག་པ། །དེས་ཀྱང་།
