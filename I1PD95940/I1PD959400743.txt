མ་ཅུ་ཏཱ་པ། །མ་ནཔྲ་སནྣཾ་ཡ་དི་མེ་ཛ་ན་ཧཱུཾ་ཡེ་ནོད་ཛྷ་ཏེ་བ་སྭཱ་ཀ་རེ་ཎ་ནེ་བའི་གདུང་བ་མིན། །མི་བདག་རང་གི་ལག་གིས་ཕྱུང་བ་བཞིན། །གལ་ཏེ་མ་ལ་བདག་ཡིད་རབ་དང་ཏྲེ༑། ཏཏྟེ་ན་ས་ཏྲེ་ན་མ་མཱསཱུ་ཏཱ་བནེ་ཏྲ་དྭ་ཡཾ་པཱལྟ་ན་མེ་བ་ས་ཏྲཿ། །ན། །བདེན་པ་དེ་ཡིས་རེ་ཞིག་མིག་གཉིས་ནི། །སྔོན་བཞིན་འཕྲལ་ལ་ཡང་དག་འབྱུང་བར་ཤོག། ཨི་ཏུཀྟ་མ་ཏྲེ་ཧྲི་པ་ནནྡ་ན་སྱ་སཱ་དུབྦ་བྷཱུ་བཱ་ཀྱི་པ་རོ་ཛ་ཡུསྶཾ། །ས་ཏྲ་པྲ་ཅེས་བརྗོད་ཙམ་ལ་དེ་ཡི་དུས་ཉིད་ན། །མི་བདག་བུ་ཡི་སྤྱན་གྱི་མཚོ་སྐྱེས་ཟུང་། །འཇིག་རྟེན་བདེན་ཏ་པྲ་ཏྱ་ཡ་ཀཱ་རི་ལོ་ཀེ་བི་ལོ་བྷ་ནཾ་ཏཀྵ་ཎ་མེ་པ་ལ་སྨྱོ། །ཚིགས་པསུ་ཁོན་པའི་བརྟུལ་ཞུགས་ཡིད་ཆེས་བྱེད། །དཔལ་མོ་རྣམ་པར་ཆགས་བྱེད་རབ་ཏུ་བྱུང༌། །འདྲེན་བྱེད་ཟུང་གིས་ཧཾ་ཀ་རཾ་པྲ་ཛཱ་ནཾ་བཱི་རཱ་ཛ་མཱ་ནཾ་ན་ཡ་ན་དྭ་ཡེ་ན། །ཏཾ་པོ་ར་རཱ་ཛེ་བི་མུ་ཁཾ་བི་དི་རྣམ་པར་མཛེས་གྱུར་ཅིང་། །སྐྱེ་རྒུ་རྣམས་ནི་བདེ་དང་སྤྲོ་བྱེད་པའི། །བུ་དེ་རྒྱལ་ཚབ་ལས་ཕྱིར་ཕྱོགས་ཏཱ་ཏ་མཱཏྨ་ཛཾ་སཾ་པ་དི་ན་རྒྱ་ཡུལྟ། །གྷོ་རཱ་པ་ཙ་རཾ་ས་དྲྀ་ཤཾ་བི་དྷཱ་ཤེས་ནས། །མི་བདག་དེ་ཡིས་ཕུན་ཚོགས་རྣམས་ལ་བསྐོས། །དེ་ནས་ས་བདག་གིས་ནི་བཙུན་མོ་ལ། །ཡ་པཔྲ་ཏི་ཀཱ་ར་མ་ཐ་གྱི་ཏཱི་ཤ། །ཀོ་དྷཱ་ན་ལཾ་ཏ་གྷ་ཤི་ལཱ་དྷི་པེ་དྲག་པོའི་གནོད་ལ་མཚུངས་པའི་ལན་བསྒྲུབས་བཞིན། །རྡོ་འཇོག་དབང་པོ་ལ་ཡང་ཁྲོ་བའི་མེ། །བཟོད་པི་ཏརྨན་ཆ་ཇུས་ཧ་མུཏྠ་སརྫ། །ཏཥོ་ཏུ་ཀཱདྷིཀྵ་ག་ཎེ་ན་པྲཋཔྲ་བཱ་ཙ་དཀའ་དེ་ཡིས་ཅིབཟོད་པས་རབ་ཏུ་བཏང་། །དགེ་མཚན་དེ་ལས་དགེ་སློང་ཚོགས་ཀྱིས་དྲིས། །དགེ་སཾ་གྷ་སྦ་བི་རས་བི་དཱན། །ཛནྣཱནྟ་རེ་ལུབྷཱ་ཀ་ཨེ་ཥ་ཀཱ་ཤི་པུ་རེ་བ་གླུ་བ་ཀི་ཏི་འདུན་གནས་བརྟན་མཁས་པ་དེས་སྨྲས་པ། །ས་སྐྱོང་བུ་འདི་སྐྱེ་བ་གཞན་ལ་ནི། །ཀ་ཤིའི་གྲོང་ཁྱེར་དག་པཱ་ལ་པུ་ཏྲཿ། །གུ་ཧཱ་པྲ་བིཥྚ'ཧི་མ་བཏྟ་ཊཱནྟེ་ལབྷཱ་མཾ་གཱ་ཎཾ་ཤ་ཏ་པཉྩ་ཀཾ་ཏུ་རྔོན་པར་གྱུར། །དེ་ཡིས་གངས་ལྡན་ངོས་ཀྱི་ཕུག་ཞུགས་པའི། །རི་དྭགས་ལྔ་བརྒྱ་དག་ནི་ཐོབ་གྱུར་ས༑། ཨནྟཾ་བི་དྷཱ་ཡཱཀྵི་བི་པཱ་ཊ་ནེ་ན་ཡ་ཐོ་པ་བྱཱཏྟཾ་ནྱ་བ་དྷཱིཏྠ་མེ་ཎ། །ནས། །མིག་དག་ཕྱུང་ནས་ལོང་བར་རབ་བསྒྲུབས་ཏེ། །ཇི་ལྟར་ཉེ་བར་འཁོར་བ་རིམ་གྱིས་བསད། །།
