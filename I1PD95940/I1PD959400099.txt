དྷཾ་ཏི་ན་ལ་བྷ་ཏེ་མ་ན། །ཨི་ཏི་ཐུ་བཱ་ཎཾ་བྷཱུ་པཱ་ལ་པྲ་བཱ་ཙ་སུ་ར་དེ་བ་ཏྭཾ། །ཨནི་སྐྲག་ཅིང་དགའ་བ་མ་ཐོབ་པོ། །འདི་སྨྲ་གྲོང་ཁྱེར་ལྷ་མོ་ལ། །སློང་བ་བཀག་པས་ངོ་ཚ་དག། རབ་སཾ་རོ་དྷ་སཾ་ཛཱ་ཏ་ལཙྪཱ་ཙ་ན་མི་ཏཱ་ན་ན། །དེ་བི་ཡཱཙྪཱ་བྷི་ཡཱ་ཏོ་སོ་པྲ་སྐྱེས་བཞིན་རས་དམན་གྱུར་པའི། །ས་སྐྱོང་གིས་ནི་རབ་ཏུ་སྨྲས། །ལྷ་མོ་སློང་བ་མངོན་འོངས་པ། །བི་ཤ་ཏཱ་ནི་བཱ་རི་ཏ། །དཱིདྷཱ་སཾ་ས་ཧེ་ནཱ་ཧཱ་མཱ་སཱ་བི་ཕ་མྱ་་་འདི་ནི་མ་ཟློག་ནང་དུ་ཐོང་། །སློང་བའི་བསམ་པ་འབྲས་མེད་པའི། །ཤུགས་རིངས་བདག་གིས་ཡོངས་མཟི་ན། །ཡཱཙྪཱ་པྲ་ཎ་ཡི་ནཱ་མརྗེ་པུ་ཎྱ་པཱསྟསྟ་ནུ་བྱ་ཡ། །ཡུ་ག་མི་བཟོད། །སློང་བ་དོན་གཉེར་རྣམས་དོན་དུ། །ལུས་ཉམས་བསོད་ནམས་རབ་ཏུ་ཐོབ། །གང་ཕྱིར་སཾ་ཥཱ་མ་པི་སྦི་ཏཱ་བི་པ་ཏྱནྟེ་ཧི་དེ་ཧི་ན། །ཨེ་ཏ་དེ་བ་སུ་ཛཱ་ཏཱ་ནཾ་ལུས་ནི་དུས་བཞིའི་གྲངས། །གནས་ཀྱང་རྣམ་པར་འཇིག་པར་འགྱུར། །སློང་བ་རེ་བ་ཉམས་པ་དག། པཱུ་ཛཾ་ཛ་ག་ཏི་ཛཱི་བི་ཏཾ། །ཡ་དེ་ཥཱ་མ་གྲ་ཏོ་ཡཱ་ཏི་ནཱནི་བྷསྨཱ་མ་ནོ་ར་འདི་རྣམས་མདུན་དུ་མི་བགྲོད་གང་། །དེ་ཉིད་སྐྱེ་བོ་ལེགས་རྣམས་ཀྱིས། །འཚོ་བ་འགྲོ་བར་མཆོད་པར་ཐ༈། ཀྲི་ཡ་ཏཱ་མཱ་ན་ཀུ་མྱོ་མེ་བྷ་བ་ཏུ་ཀུ་ཤ་ལོ་ཙི་ཏཾ། །ཨཱ་སཱ་བི་གྷཱ་ཏེ་འོས། །བདག་གི་རྗེས་སུ་མཐུན་པ་ནི། །དགེ་བར་འོས་པ་ཁྱོད་ཀྱིས་མཛོད། །དེ་ཡི་བསམ་པ་རྣམ་སཾ་ཏཱ་པསྟ་སྱ་ཏུཎ་ནི་བཱརྨ་ཀཾ་། ཨི་ཏི་བྷཱུ་མི་པྲྀ་ཏཐུ་ཏྭཱ་བ་ཙོ་ནིབཅོམ་པའི། །གདུང་བ་མྱུར་དུ་ཟློག་པར་མཛོད། །དེ་སྐད་ས་བདག་ཚིག་ཐོས་ནས། །གཡོ་བ་མེད་ཉི་ལ་ནིཉྩ་ཡཾ། །ཛ་གཱ་མཱ་དན་ནཾ་དེ་བི་ཙིནྟཱ་སན་པ་མཱ་ན་སཱ། །ཨ་ཐཱ་ཡ་པར་ངེས་པ་ཡིས། །བསམ་པས་ཡིད་ནི་རབ་གདུང་བ། །ལྷ་མོ་སྣང་བ་མུན་པར་གྱུར། །དེ་ནས་རྒྱ་ཡོ་པ་ཀུ་ཊི་ལ་ཁ་ལ་ཀྲ་ཀ་ཙ་ཙེཎྜི་ཏ། །དཱ་རུ་ཎས་ར་ལ་སྤྱི་བ་ཙྪེ་དཱ་གྱུ་ཅན་དེ་འོངས། །མི་བསྲུན་སོག་ལེའི་སྤྱོད་ཚུལ་ཅན། །མི་བཟད་དྲང་པོའི་ཤིང་དག་ནི། །སྤྱོད་སྐྱེལ་ཡ་སཱ་ཡ་མུངྒ་ཏ། །ཏསྨིནྣཾ་པ་གཾ་ཧཾ་པཱསྟེ་བི་ཀྲྀ་ཏ་དཱ་ར་མཟི་ནཾ། །བྷཱུར་བྷུ་རང་ཉིད་སྦྱོར་བ་བཞིན། །སློང་ལ་སྒོ་ནི་རབ་ཕྱེ་བ། །མི་བདག་ཁང་པ་དེར་འོངས་ལ། །ས་བདག་།
