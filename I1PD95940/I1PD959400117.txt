སཱརྦ་པ་ཏི་རྨ་གྷ། །བྱཱ་དྷི་ནཱ་དུཉྩི་ཀིཏྠེ་ན་བ་བྷཱ་བཱ་སཱསྦྲ་བི་གྲ་ཧ། །ཨ་ལབྷཱནྟ༔ལས་ཀྱི་སྦྱོར་བ་ལས། །གསོ་བ་དཀའ་བའི་ནད་ཀྱིས་ནི། །ལུས་ནི་རྣལ་དུ་མི་གནས་གྱུར། །དེ་ནས་རྒྱལ་པྲ་བེ་ཤོརྦ་ཤྲཱི་ཧེ་རཱ་ཛ་ཡཾ་ཧོ་པ་མེ། །ཏ་སྱ་བྱ་ལམྦ་ཏ་དྷཱ་རི་སུ་པྲ་ཡཔོའི་ཁབ་ལྟའི་ཁྱིམ། །ནང་དུ་འཇུག་པ་མ་ཐོབ་ནས། །དེ་ཡི་སྒོ་ན་ཤིན་ཏུ་དགའ། །བྱ་བ་བསྒྲུབ་སླད་ཀརྨ་སིདྡྷཱ་ཡེ། །ཏ་ཏོ་བོ་བྱོ་པ་དེ་ཤེ་ན་ས་པྲ་བེ་ཤ་མ་བཱསྟ་བཱན། །ཨུ་པ་ཡོ་ག་ཀ་འདུག་པར་གྱུར། །དེ་ནས་སྨན་པའི་ཟོལ་གྱིས་ནི། །དེ་ནི་ནང་དུ་འཇུག་པ་ཐོབ། །ཉེ་བར་རིགས་པའི་ཐཱ་པྲ་ཛྙཱ་ནཀ་རྩྭ་ད་ར་བྷུ་མ་ཡ། །ཨཱ་ཡུརྗེ་ད་བི་དྷཱ་ན་ཛྙཱས་ཏ་སཱུ་རིཥཱ་གཏམ་ཤེས་པ། །སུ་ཡི་གུས་པའི་ས་མ་ཡིན། །ཚེ་ཡི་རིག་བྱེད་སྒྲུབ་ཤེས་ཤིང་། །འཆི་བའི་མཚན་ཉིད་ལཀྵཎི༑། ཥཛྙཱ་པ་ཤེ་ཥ་མེ་བཱ་ཡུཛྙཱ་ཏཱ་ཙིནྟཱན་རོ་བྷ་བཱཏ། །ཏ་སྱ་ཤེས་པ་དེས། །ཟླ་དྲུག་མཇུག་ེཉིད་དེ་ཡི་ཚོར། །ཤེས་ནས་བསམ་པ་འཁྲུགས་པར་གྱུར། །དགའ་བྲི་ཡ་ཧི་ཏ་བཱ་པོ་བྷི་ཥ་ཛུ་པ་རི་ཙནྡཱ་ཡ། །ཨ་ཏྱཱཡྻེ་ནི་བ་ཀཱ་ལེ་ན་སུ་ཞིང་ཕན་པའི་སྨན་མང་དང་། །དེ་ཡི་རིམ་གྲོ་ལེགས་བྱས་པ། །དུས་ནི་ཤིན་ཏུ་ཐུང་བ་ཡིས། །ཤིན་པྲི་ཡ༔པྲ་ཡ་ཏྭཾ་ཡ་ཡོ། །བྷི་ཥ་ཛུ་ཡུཀྟིས་ད་པཱི་ཏཱ་ཏ་སྱ་བལླབྷ་ཏྭཾ་ཡ་ཡོ། །ཏུ་དགའ་དང་མཛའ་བོར་གྱུར། །དེ་ལ་མཛའ་བས་སྨན་སྦྱར་རྣམས། །དེ་ཡིས་འདོད་པ་བྱ་བར་གྱུར། །པྲི་ཡོ་པ་ནཱི་ཏཾ་ཡཏ་ཀིཉྩིཏ་ཏཏྠརྦསྨྲ་ན་སཔྲི་ཡཾ། །པྲི་ཡོ་པ་ཙཱ་རིལྟ་སྣུབྱཱ་མཛའ་བོས་ཅུང་ཟད་བྱིན་པ་གང་། །དེ་དག་ཀུན་ལ་ཡིད་དགའ་འོ། །དེ་ནས་བདེ་བའི་རིམ་གྲོས་དེའི། །དྷིཆ་རྔ་བ་མཱ་ཡ་ཡོ། །ཨཱ་དྷཱིཤཱ་རྱ་ཏི་སཏྠཾ་གཱནྟ་ཏོ་བྱཱ་དྷིའི་ཤིརྨ་ཏེ། །ནད་ནི་ཆུང་བ་ཉིད་དུ་གྱུར། །དམ་པའི་གྲོགས་ལས་སྡུག་བསྔལ་ཞི། །དེ་ལས་ནད་དང་བྲལ་བར་གྱུར། །ཏ་ཏསཾ་ཛཱ་ཏ་བི་བྷཾ་གྷཾ་སུ་པྲ་ཡ་པྲ་ཎ་ཡཱརྨ་བྷཾ། །ཙཀྲེ་བི་དི་ཏ་བྲྀཏྟཱནྟཾ་དེ་ནས་མ་གྷ་མཛའ་བ་ལས། །ཡིད་ཕེབས་སྐྱེས་ལ་ཤིན་ཏུ་དགའ། །སླད་ནས་རང་གི་གཏམ་གྱི་ཚེ། །པཉྩནྡྲི་ཛ་ཀ་ཐཱ་ག་ཎེ། །བ་ད་ར་དྭི་པ་ཡཱ་ཏཱ་ཡཾ་ཏ་སྤྱོདྷཱ་ཧཾ་མ་ཧཱཏྨ་ན། །པ་སྤྱོད་ཚུལ་མཐའ་ནི་རིག་པར་བྱས། །བ་ད་ར་ཡི་གླིང་བགྲོད་ལ། །བདག་ཉིད་ཆེ་དེའི་སྤྲོ་བ་དག། གཞན་།
