ལས་རྒྱུ་སཱ་ཏེ་ན་ན་ཧཱུཾ་ནི་ཀཱིདྡྷ་ནཱ་གིནྡྲི་ཏཱ་པྲ་བྷཱ་བ་ཏཱི། །བི་ཀློ་བྷྱཱ་བྷཱ་ཧྲི་དཱ་ཏེ་བྷནྡྷཀྵ་འོད་ལྡན་དེ་ནི་ཆུ་ཀླུང་དུ། །དེས་འཕངས་ཀླུ་རྣམས་དག་གིས་ཐོབ། །རྡོ་རྗེ་ཅན་གྱིས་དེ་དག་ལས། །རླུ་པེ་ཎ་བཛི་ཎཱ། །པྲ་སཱ་དི་བོ་དྷི་སཏོ་ཐ་ཀ་རྩྭ་ཡུ་མཁའ་ལྡིང་གཟུགས་ཀྱིས་བསྐྱོད་ཐོབ་བླངས། །རབ་དང་བྱང་ཆུབ་སེམས་དཔའ་ཡིས། །དེ་ནས་ག་ལ་མཱ་དྷྱ་པཱཏ། །དཱ་ན་སྱ་པྲ་ཎི་དྷཱ་ནེ་ན་བྷ་བ་བྱོཀིཾ་ས་མཱི་ཧི་ཏཾ། །ཏེ་མཛེས་མ་ཟུང་ལ་སྨྲས། །སྦྱིན་པའི་སྨོན་ལམ་དག་གིས་ནི། །ཁྱེད་གཉིས་ཅི་ཞིག་མངོན་པར་འདོད། །དེ་ཏ་མཱུ་ཙ་རུ་རཱ་ནནྡ་ནི་དྷིཤུདྡྷོ་ད་ནཱཏྨ་ཛ། །སནྡྷརྦ་སིདྡྷོཤྭབྷི་མ་ཏལ་དེས་སྨྲས་ཟས་གཙང་སྲས། །མཁས་པ་ཀུན་ཏུ་དགའ་བའི་གཏེར། །གཞོན་ནུ་དོན་ཀུན་གྲུབ་པ་དེ། །ཀུ་མཱ་རཔ་ཏི་རཱ་བ་ཡོ། །ཨུརླ་མ་ཀ་མ་ར་ཧ་ལཱི་ས་ར་བདག་ཅག་བདག་པོར་མངོན་པར་འདོད། །དེ་དག་ཚིག་ནི་ཆགས་ལྡན་ཞིང་། །འདོད་པའི་རྦ་རླབས་སནྟ༴་ཙསྟ་ཡོ། །།ན་ལི་ལིམྦ་མ་ནས་སྱ་པÉ་ད་ལ་མི་བོ་ད་ཀཾ། །ས་ཛ་གཱ་ད་ཀུ་ལྷག་པ་དེས། །དེ་ཡི་ཐུགས་ལ་མ་གོས་ཏེ། །ཆུ་ཡིས་པདྨའི་འདབ་མ་བཞིན། །དེ་ཡིས་སྨྲས་པ་གཞོན་མཱ་རོ་སོ་ན་ཀིཾ་པྲ་བྲ་ཛི་ཏཐུ་ཏ། །ན་ཏ་སྱ་ལོ་ལ་ན་ཡ་ནསྲི་ཡཱཤི་ཡ་ཨི་ནུ་དེ། །རབ་ཏུ་བྱུང་བ་མ་ཐོས་སམ། །དེ་ལ་བུད་མེད་དཔལ་མོ་བཞིན། །མིག་གཡོ་དགའ་བ་མེད་བསྟི་ཡ། །ཨི་ཏྱཱ་ནཱི་པ་ཏ་མཱ་ཀཙྩཾ་དཱིགྷོ་ནི་ཤཱ་སྱ་ཀ་ནྱ་ཀེ། ་། ཨཱུ་ཙ་ཏུརྭ་དམ་ཅི། །ཞེས་པ་འདོད་པ་མིན་ཐོས་ནས། །ཤུགས་རིང་ཕྱུངས་ཏེགཞོན་ནུ་མས། །སྨྲས་པ་ེསྦྱིན་ན་དྷརྨ་ཡཾ་སིདྡྷི་ཏ་སྤྱི་བ་ཛཱ་ཡ་ཏྭཾ། །ཨ་དྲྀཥཱ་སྣེ་ཧ་སཾ་ཤྲཱིཋཔྲ་བིཥཱནྟ༔པའི་ཆོས་འདི་ནི། །དེ་ཡི་གྲུབ་སླད་ཉིད་གྱུར་ཅིག། མ་མཐོང་བྱམས་པ་ཡང་དག་སྦྱར། །ནང་ཞུགས་ཕྱི་པ་རཱརྨུ་ཁེ། །ན་ནཱ་མ་སུ་ཙི་རཱ་བྱཱསྟཔཀྵ་པཱ་ཏོ་ནི་བརྟ་ཏེ། །ཨི་རོལ་ཕྱོགས་ཚེ་ནའང༄༅། །ཡུན་རིང་གོམས་པ་ངེས་པར་ནི། །ཕྱོགས་སུ་ལྷུང་བ་མི་ཟློགས་སོ། །དེ་ཏི་ཏདྨ་ཙ་ནཾ་ཐུ་ཏྭཱ་བོ་དྷི་སཏྭཔྲ་སནྣ་དྷཱི། །པྲ་ཡ་ཏསྟེ་ས་མཱ་མཱནྟ་དག་ཚིག་དེ་ཐོས་ནས་བློ། །རབ་དང་བྱང་ཆུབ་སེམས་དཔའ་ནི། །ནགས་ཀྱི་ནང་དུ་ངལ་གསོའི་སླད། །།
