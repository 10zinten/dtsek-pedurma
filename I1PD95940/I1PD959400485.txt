ཀྵཎཾ་པུ་ཎྱ་ཀརྨ་ཎཾ། །ཀ་པི་ལོ་ནཱ་མ་ས་ཤི་ཤུསྟེ་ཛཔིངྒཾ་ཤི་རོ་རུ་ཧ། །བསོད་ནམས་ལས་ཀྱི་མཚན་ཉིད་ཡིན། །སྐྲ་ནི་སེར་སྐྱ་གཟི་བྱིན་ཅན། །སེར་སྐྱ་ཞེས་བྱ་བྱིས་པ་དེ། ་། བརྡ་མ་ནཱ་མ་ཏིཥཱི་དྭན་པི་ཏུ་ར་བྱཱ་དྷི་ཀོཉབྷ་བཏ། །ཀུ་ལེ་མ་ཧཱ་ཏི་བི་ཆེར་སྐྱེས་ཤིན་ཏུ་མཁས་པ་ནི། །ཕ་ལས་མངོན་པར་ལྷག་པར་གྱུར། །རིགས་བརྒྱུད་ཆེན་པོ་དག་ལ་དུ་ཙྪ་བི་དུབྱེ་བི་བྷ་བོདྷཱ་བ། །བི་བྷ་བེ་སདྷུ་ཏ་ཀོཕ་ལཾ་སུ་ཀྲྀ་མཁས། །མཁས་པ་ལ་ནི་འབྱོར་པ་རྒྱས། །འབྱོར་ལ་ཁྱད་འཕགས་དམ་པའི་བུ། །ལེགས་བྱས་ཡལ་ག་ཏཤ་ཁི་ན། །ཀ་དཱ་ཙིཏ་བྱཱ་དྷི་སཾ་ཡོ་གཱཏ་པྲ་ཏྱཱ་སནྣཏ་ནུ་གྷ་ཡ། །བི་ཛ་ནེ་པུ་ཏྲ་མཱ་ལྡན་པའི་འབྲས། །ནམ་ཞིག་ནད་ཀྱི་སྦྱོར་བ་ཡིས། །ལུས་ནི་འཇིག་ལ་ཉེ་བ་ན། །སྐྱེ་བོ་མེད་སར་བུ་བོས་ཧ་ཡ་སོབ་བཏ་པུ་ཏྲ་བཏྠ་ལ། །བཱ་ལྕེ་གུ་ཎཱརྫ་ནཾ་པུ་ཏྲ་ཏཱ་རུ་ནས༑། བུ་ལ་མཉེས་གཤེན་དེས་སྨྲས་པ། །བདག་གིས་གཞོན་ལ་ཡོན་ཏན་བསྒྲུབས། །དར་ལ་ཙེ་དྲ་བི་ཎཱཛྫ་ནཾ། །མ་ཡ་ཀྲྀ་ཏཾ་ན་ཏུ་པུ་ནཔ་ར་ལོ་ཀ་སུ་ཁཱརྫ་བབས་ཚེ་ནོར་ཟས་བསྒྲུབས། །བུ་གཅིག་འཇིག་རྟེན་ཕ་རོལ་གྱི། །བདེ་བསྒྲུབ་སླར་ཡང་བྱས་པ་ནཾ༑། ཨུཏྟ་མཙྪཱ་ཨི་བ་པྲསྟེ་ཀཱ་ལེ་སུ་ག་ཎི་ཏཱ་བ་དྷོ། །ཨ་དྷུ་ནཱ་མེད༑། མཚམས་ནི་ལེགས་པར་བགྲངས་པའི་དུས། །བུ་ལོན་པ་བཞིན་འོངས་པའི་ཚེ། །ད་ནི་འཁྲུགས་བི་བ་ཤཀཱ་ཧཾ་ཀཱ་སཱ་བི་ཏྱཱ་ཀཱ་ཏདྷ་ནཾ། །གུ་ཎ་པུལླེ་སུ་ཁ་ཕ་ལེ་བདྷ་མུ་ལ་ཏེ་བདག་གང་ན། །རིགས་དེ་གང་ན་ནོར་གང་ན། །ཡོན་ཏན་མེ་ཏོག་བདེ་འབྲས་ལྡན། །ནོར་གྱི་རྩ་དྷ་ནིརྫ་ནེ། །བ་ནེ་བཛྲ་ཨི་བཱ་ཀཱ་ལ་ཀཱ་ལཔ་ཏ་ཏི་དུས་ཧ། །བཅིངས་སྐྱེ་བོ་ཡིས། །ནགས་སུ་དུས་མིན་རྡོ་རྗེ་བཞིན། །བཟོད་དཀའི་དུས་ནི་ལྟུང་བར་འགྱུར། །ཀླུ་པ་ཡ་ཏི་ས་ཀ་ལཱ་བིཛྫརྨ་བི་བྱཱ་ཀ་ལཱ་བྷིཀྵ་ཎི་ཀ་སུ་ཁ་ནི་མིཏྟཾ། མཐའ་དག་རིག་དང་སྒྱུ་རྩལ་རྣམས་ཀྱིས་སྐྱེ་བ་འཕེན་བྱེད་ཅིང་། །སྐད་ཅིག་བདེ་བའི་རྒྱུར་ནི་ནོར་རྣམས་སནྣི་དྷཏྟེ་ཙ་བིཏྟཾ། །པ་ཤུ་ཤི་ཤུ་ཥཱ་མ་ནུཔྲ་ཡ་ཏེ་མོ་ཧ་ཤི་ཥྱཏ་ནུ་བི་རབ་ཏུ་འཛིན་པར་བྱེད། །རྨོངས་པའི་སློབ་མ་མི་རྣམས་ཕྱུགས་དང་བུ་ལ་དགའ་བར་བྱེད། །ལུས་འབྲལ་།
