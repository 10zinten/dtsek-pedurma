ཀ་རོ་ཏི་མཏ་པྲ་སངྒེ་ན་མུ་ཧཱུཏྟི་ནི་བ་ནིÉ་གཾ། །ཏ་སྨཏ་སེ་ནྱ་པྲ་དྷཱ་ན་སྱ་བདག་གི་ཞར་ལ་རི་དྭགས་རྣམས། །ཡུད་ཙམ་ཉིད་ཀྱིས་མེད་པར་བྱེད། །དེ་སླད་དཔུང་གི་གཙོ་བོ་དང་། །གཙྪཱ་མི་སྭ་ཡ་མནྟ་ཀཾ། །ཨེ་ཀ་སྤྱི་བ་བ་བློ་མི་སྐུ་སཏྠ་ཛཱི་བ་ཏཱ་མཱི་མཾ་གཱ། །ཉེ་བར་རང་ཉིད་འགྲོ་བར་བྱ། །བདག་ཉིད་གཅིག་པུ་གསོད་གྱུར་ལ། །རི་དྭགས་འདི་ཀུན་འཚོ་གྱུར་ཅིག། ཨི་ཏི་ཙྪིཤྲཱི་ཏྱ་ས་ཡ་ཡོ་ས་མཱི་པཾ་བྷཱུ་པ་ཏེÉ་ག། །པ་ར་པྲ་ཎ་པ་རི་ཏཱ་ཎེ་དེ་ལྟར་དེས་བྱས་རི་དྭགས་དེ། །ས་བདག་དང་ནི་ཉེ་བར་སོང་། །གཞན་སྲོག་ཡོངས་སུ་སྐྱོབ་པ་ལ། །ཀྲྀ་ཎཾ་པྲ་ཎཱ་མ་ཧp་ནཾ། །ཏ་མཱ་ཡཱནྟཾ་དྲུ་ཏཾ་དྲིཋཀུ་ཊི་ལ་ཀཔུབདག་ཉིད་ཆེ་རྣམས་སྲོག་ནི་རྩྭ། །མདུན་དུ་མགྱོགས་པར་འོང་བ་དེ། །མཐོང་ནས་གྱ་གྱུ་ཅན་དགའ་རཿ༑། སོ་ཡ་མི་ཏྱཱ་ཤུ་པཱ་ཎི་ཧཱུཾ་རཱ་ཛེ་དྲུ་རོ་ད་དན་ཡཏ། །ཏཏྟཱ་ཎེ་དྲོ་ཎ་ཤཱ་པེ་བས། །རྒྱལ་ལ་མགྱོགས་པར་དེ་འདི་ཞེས། །ལག་པ་དག་གིས་རིང་ནས་བསྟན། །དེ་ཚེ་བྱ་རོག་དམོད་ན་བཛྲ་ཎེ་བ་ཎི་པཱ་ཏི་ནཱ། །ཀ་རོ་པ་རི་ཙཱུ་ཏོ་ཏ་སྱ་པཱ་པ་པཱ་ད་པ་པལླ་བོ། །པ་ནི། །རྡོ་རྗེ་ལྷུང་བ་ལྟ་བུ་ཡིས། །དེ་ཡི་ལག་པ་སྡིག་པ་ཡིས། །རྐང་འཐུང་ཡལ་འདབ་ཡོངས་སུ་ལྷུང་ཚེ། །ཏམྦིཏྟཾ་བིསྨི་ཏཤུ་ཏྭཱ་མཾ་གེ་ན་ཀ་ཐི་ཏཾ་ཀྲྀ་པ། །ཨ་བྷཱུཏ་ཀྲྀ་ཏ་བྷཱ་ཙ་རི་ཏེ་རི་དྭགས་ཀྱིས་བཤད་དེ་ཡི་སྤྱོད། །མི་བདག་ཡ་མཚན་ལྡན་པས་ཐོས། །ཞལ་ནས་བྱས་པ་མི་གཟོ་ཡིབའི། །དྷིཀྐཱ་ར་མུ་ཁ་རཱ་ན་ན། །ཏ་ཏཥི་ཏི་པ་ཏི་པཱི་ཏྱཱ་བི་བྷཱུ་ཏྱ་ས་ར་ཡཱ་མྲྀ་ཧཾ། །ཏཾ་སྤྱོད་ཚུལ་སྨད་ཅེས་མུ་ཅོར་གྱུར། །དེ་ནས་ས་བདག་དགྱེས་པ་ཡིས། །རྗེ་ས་ཤིན་ཏུ་ཆེ་བ་དང་། །འབྱོར་ནི་ནཱ་ཡཱ་སྭ་ན་ག་རཱི་གོ་ར་བེ་ན་ག་རཱི་ཡ་སཱ། །རཱ་ཛ་དྷཱ་ནཱི་མ་ཐཱ་ས་བྱ་པ་མཆོག་གི་ཨ་རི་དྭགས་དེ། །རང་གི་གྲོང་ཁྱེར་དག་ཏུ་ཁྱེར། །དེ་ནས་རྒྱལ་པོའི་ཕོ་བྲང་ཕྱིན། །དེ་ཏསྨི་རཏྟཱ་ས་ནཾ་ནཱི་པ། །ད་ཏཱ་སཱནྟཔུ་རཱ་མཱ་ཏཔ་སྟ་གྲི་ས་མུ་པཱ་བི་ཤཏ། །ནས་རིན་ཆེན་ཁྲི་སྟན་ཕུལ། །དེ་ཡི་མདུན་དུ་མི་ཡི་བདག། བཙུན་མོ་བློན་པོའི་ཚོགས་བཅས་འཁོད། །ས་བོ་དྷི་ས་ཏྭཱ་ཧ་རི་ཎསྟ་སྱ་པཆུ་དི་དི་བྱ་དྷཱི། །དྷམྨོ་དི་དེ་ཤ་ཡེ་ནཱ་བྱང་ཆུབ་སེམས་དཔའ་རི་དྭགས་དེའི། །མཆོག་གི་བློ་ཡིས་མདུན་ས་དེར། །ཆོས་བསྟན་གང་གི་སྐྱེ་བོ་—།
