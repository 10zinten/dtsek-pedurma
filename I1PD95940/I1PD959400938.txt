བསྟན་འགྱུར། འཁྲི་ཤིང་། ༼ཁེ་ཏྭཱཙྩཾ་པྲི་ཐི་བཱི་པ་ཏི། །ཨཾ་སཾ་ཤ་རཱི་ར་ནིཀྲྀཏྟཾ་རུ་དྷི་ར་ཙ་ཡ་ཡཱ་ཙི་རེ། །སེམས་ཅན་གདུག་པས་ས་ཡི་བདག། དེ་ལ་ལུས་ནི་བཅད་པ་ཡི། །ཤ་དང་ཁྲག་དག་རབ་ཏུ་བླངས། །ནིབྦ་ཀཱ་ར་མ་ཏིསྟེ་བྷྱས་ད་ཏཱ་བྷི་མ་ཏ་ཀྲྀ་པ། །ཨསྦི་ཤེ་ཊཾ་པྲ་བི་ད་མི་བདག་བློ་གྲོས་རྣམ་འགྱུར་མེད། །དེ་ཡི་མངོན་འདོད་རུས་པའི་ལྷག། དེ་དག་རྣམས་ལ་བྱིན་ནས་ནི། །དྷེ་པྲ་ཎི་དྷཱ་ནཾ་མ་ཧཱ་ཡ་ཤ། །ཤ་རཱི་ར་དཱ་ན་པུ་ཏྟཱ་ན་ཏཱ་ར་ཎཱ་ཡ་ཤ་རཱི་བསམ་པ་ཆེན་པོས་སྨོན་ལམ་བཏབ། །ལུས་ཀྱི་སྦྱིན་པའི་བསོད་ནམས་འདིས། །ལུས་ཅན་རྣམས་ནི་བསྒྲལ་རི་ཎཾ། །ཨེ་ཏེ་ནཱ་ནུ་ཏྟ་རཾ་སམྱཀཾ་བོ་དྷི་མ་ཧཱ་མཱསྣ་ཡཾ། །ཨུ་ཙེ་ཤ་ཀྲལྟ་བའི་སླད། །བླ་མེད་ཡང་དག་རྫོགས་པ་ཡི། །བྱང་ཆུབ་ཆེན་པོ་ཐོབ་པར་ཤོག། དེ་ནས་བརྒྱ་བྱིན་དེར་མ་བྷེ་ཏྭཱ་ཐ་རཱ་ཛམྦུ་བྱ་ཐ་མི་ཏ་སྱ་ཏེ། །ཨ་སི་དེ་ཧ་པྲ་དཱ་ནེ་བྷུདྡི་ཀཱ་རས་ཕྱོགས་སྨྲས། །རྒྱལ་པོ་རབ་གདུང་ཁྱེད་ཀྱིས་ནི། །ལུས་ནི་རབ་ཏུ་བྱིན་པ་ལ། །རྣམ་འགྱུར་བྱུང་ངམ་ཏྱ་མུ་ཙེ་ཏྭཾ། །ཨི་ཏི་པྲཋསུ་རེནྡེ་ཎ་ན་རེནྡཔ་མ་བྷཱ་ཥ་ཏ། །ཨརྦ་བདེན་པར་སྨྲོས། །ཞེས་པ་ལྷ་ཡི་དབང་པོས་དྲིས། །སློང་བ་རྫོགས་པས་རབ་བསྐྱེད་པ། །དགའ་བས་སཾ་སུ་ར་ཎོདྡྷཱ་ཏ་པྲ་ཧརྩོàལྤ་ལོ་ཙ་ན། །བྱ་ཐཱ་ཡཾ་ནིབི་ཀཱ་རོ་ཧཾ་ཡ་དི་མིག་ནི་རབ་རྒྱས་པའི། །མི་ཡི་དབང་པོ་དེ་ལ་སྨྲས། །གལ་ཏེ་བདག་ནི་གདུང་བ་ན། །རྣམ་འགྱུར་ས་ཏེ་ན་ཏེ་ན་མེ། །སཾ་རཱུ་ཌྷ་བྲ་ཎ་སཙྪ་ཡཾ་སྭསྦྲ་མེ་བཱསྟེ་གྲ་ཧཾ། །མེད་ན་བདེན་པ་དེས། །བདག་གི་ལུས་འདི་ྋརྨ་བསོས་ཤིང་། །མཛེས་ལྡན་རྣལ་དུ་གནས་གྱུར་ཅིག། ཨི་ཏུ་དཱི་རི་ཏ་མཱ་ཏྲེ་ཎ་ཀཱནྟི་སྭསྦྲ་མ་བྷུ་བ་སུ། །ཏ་སྱ་མེ་གྷ་བི་ནི་རྨུཏྟཾ་ཅེས་པ་རབ་ཏུ་བརྗོད་ཙམ་གྱིས། །དེ་ལུས་རབ་མཛེས་རྣལ་གནས་གྱུར། །སྤྲིན་དང་རྣམ་པར་བྲལ་བ་ཡི། །བིཝཾ་ཏཱ་རཱ་པ་ཏེ་རི་བ། །བི་ཛ་ཡནྟོ་ཚིགས་པ་སོ་ཧཾ་པ་ར་ཀཱརྨེ་ཥཱ་ནིཅུ་རྒྱུ་སྐར་བདག་པོའི་གཟུགས་བཞིན་ནོ། །མི་བདག་རྣམ་རྒྱལ་དེ་ཉིད་ང་། །གཞན་གྱི་དོན་ལ་གདུང་བ་ཐ༈། ཨི་ཏུཀྟ་བྷ་ག་བཱནྡྷཱ་ན་ནི་ལཱི་ན་ན་ཡ་ནོ་བྷ་བཏ། །ཨ་ཙ་མེད། །ཅེས་པ་བརྗོད་ནས་བཅོམ་ལྡན་འདས། །བསམ་གཏན་ལ་ཆགས་སྤྱན་ལྡན་གྱུར། །འཕྲལ་ལ་རི་—།
