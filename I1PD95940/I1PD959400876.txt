བ་བྷཱུ་བ་ན་མྲ་བད་ནཱ་གྷ་ཎཾ་མོ་ནཱ་བ་ལམི་ནི། །པྲ་ཎཱ་མེ་ནཱ་པི་ཚྭ་པ་སྐད་ཅིག་བཞིན་རས་དུད་པ་ཡིསི། །མི་སྨྲ་བ་ལ་བརྟེན་པར་གྱུར། །མི་བདག་གིས་ནི་ཕྱག་བྱས་ཀྱང་། །ཏེརྣ་སཱ་པཱ་པ་སྲ་པནྣ་ཏྭཾ། །སྲེསྣ་ཤ་ལུ་ཏུ་ལཾ་དྷཏྟེ་མི་བྱཱ་དོ་ཥ་པ་རི་གྲ་དེ་ནི་རབ་དང་ཉིད་མ་གྱུར། །རྫུན་གྱི་སྐྱོན་ནི་ཡོངས་བཟུང་ཉིབ། །མཛའ་བའི་ཟུག་རྔུ་མཚུངས་པར་ཧ༑། སཱ་དྷཱ་ཏུ་རཀྟ་བ་སནཱ་ཛ་གཱ་མ་པི་ཏུ་རཱ་ཐ་མཾ། །བྷུ་ཛཾ་ག་ཀོ་ཊི་འཛིན། །དེ་ནི་དམར་བའི་བཞིན་བཟུང་ནས། །ཡབ་ཀྱི་བསྟི་གནས་དག་ཏུ་སོང་། །ཁེངས་ལྡན་རྣམས་ལོ་མ་མྨཱ་ནི་ནཾ་ས་ཧི་དུ༔ས་ཧ། །ཨཱ་ལཾ་གྱི་མཱ་ནེ་བ་ལ་ཏཱ་ས་ཁཱི་བྷིནྡྷོ་ཀྱི་ཁྲོ་བ་ནི། །ལག་འགྲོ་གྱ་གྱུ་དེ་བཟོད་དཀའ། །བུང་བའི་སྒྲ་ཡིས་དགེ་ལེགས་བརྗོད་པ་ཡི། །འཁྲི་ཤིང་ག་སྭ་ནི༔སཱ་ག་ཏ་བ་དི་ནཱི་བྷི། །སྲེསྣ་མཾ་གཱི་བྷིཔ་རི་བརྨ་མཱ་ཎཱཏ་ཏ་པོ་གྲོགས་མོ་རྣམས་ཀྱིས་འཁྱུད་པ་བཞིན། །རི་དྭགས་རྣམས་ཀྱི་མཛའ་བས་ཡོངས་བསྐོར་ཞིང༌། །ཕ་ཡི་བ་ནཾ་སཱ་པི་ཏུ་རཱ་པ་སཱ་ད། །ཡཱ་ཏི་མུ་ནོ་པུ་ཎྱ་ནི་དྷོ་པྲ་སནྣན་པ་བི་ཤེ་ཥཱརྫི་དཀའ་ཐུབ་ནགས་སུ་དེ་ཕྱིན་གྱུར། །དཀའ་ཐུབ་ཁྱད་པར་གྱིས་བསྒྲུབས་འཇིག་རྟེན་ནི། །དང་བར་ཐུབ་ཏ་མེ་བ་ལོ་ཀཾ། །ཛཾབྷྱཱ་ཐ་མཾ་ཤུ་རྒྱ་མ་དྷཱ་ར་པྲྀཏྟཔདྷཱ་བ་ཏཱི་མོ་ཧ་པ་བསོད་ནམས་གཏེར་སོང་ཚེ། །བསྟི་གནས་སྟོང་མཐོང་འཚོ་བ་མི་བརྟན་པའི། །པདྨོ་ཅན་ནི་རྨོངས་ཧ་ཏཱ་བ་བྷཱུ་བ། །བཱཏྨ་ལྱཱ་མཱ་ཛརྨ་ནི་ལཱི་ན་མནྟ༔སཱསྦྲ་སྭ་བྷཱ་བ་སྱ་པི་ཏུཔས་བཅོམ་པར་གྱུར། །རང་བཞིན་རྣལ་གནས་ཕ་ཡི་མཉེས་ཅིགཤིན་དང༌། །སྐྱེ་བའི་བར་ནི་ནང་དུ་སྨ་རནྟི། །ཤུ་རྩྭ་སྡི་ལོ་ཀི་མི་བ་མ་ནྱམཱ་ནཱ་བི་ཥཱདི་ཏེ་བཱ་ཧི་ཧ་ཏེ་བ་གཅགས་པ་དྲན། །འཇིག་རྟེན་གསུམ་བཞིན་སྟོངས་པར་སེམས་ཤིང་དེ། །དུག་གིས་གཟིར་བཞིན་སྦྲུལ་སཱ་བྷཱུཏ། །ཏ་དེ་བ་ཏ་སྭཱ་མུ་ནི་ནཱ་བི་ཧཱི་ནཾ་ཁཾ་ནཾ་ཛ་གཱ་མ་པྲ་ཡཾ་ཏྭཾ་གྱིས་བཅོམ་བཞིན་གྱུར། །ཐུབ་པས་དམན་པས་ནགས་ཚལ་དེ་ཉིད་ནི། །དེ་ལ་དགའ་བ་དགའ་མིན་ཀཱ་ལེ་ན། །ནིཤེ་ཥ་ནི་པཱི་ཏ་སཱ་ར་བྷ་བནྟི་ར་ཧ་སྭ་བྷཱ་བཱ་བི་ར་ཧ་སྭ་བྷཱ་ཉིད་དུ་གྱུར། །དུས་ཀྱི་སྙིང་པོ་མ་ལུས་འཐུངས་གྱུར་པའི། །དངོས་པོ་རྣམས་ནི་འབྲལ་བའི་རང་བཞིན་—།
