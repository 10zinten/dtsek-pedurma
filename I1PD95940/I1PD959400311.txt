གྲོ་ཧ་པ་ཏི་སྟེ་ན་པོ་བཱ་ཙ་ས་སྨི་ཏ། །ཨ་ཧོ་བ་ཏ་ན་ཛཱ་ནཱི་ཥ་ཤཱསྟཱ་རཾ་བྷཱུ་བ་ཁྱིམ་བདག་གིས། །འཛུམ་དང་བཅས་པས་རབ་སྨྲས་པ། །ཀྱེ་མ་ས་གསུམ་དག་གི་ནི། །སྟོན་པ་ཁྱོད་ཀྱིས་ན་ཏྲ་ཡེ། །སཾ་སཱ་ར་པཱ་ཤ་བྷཱི་ཏཱ་ནཾ་ཤ་ར་ཙྩཾ་ཤ་ར་ཎེ་ཥི་ཏྭཾ། །མི་ཤེས་སམ། །འཁོར་བའི་ཞགས་པས་འཇིགས་ནས་སྐྱབས། །འཚོལ་བའི་སྐྱབས་གྱུར་རྒྱལ་བ་དེ། །ཛི་ནཾ་ཡསྟ་ན་ཛཱ་ནཱ་ཏི་ས་ལོ་ཀེ་བི་ཙི་ཏཔ་རཾ། ་། ཀི་ཏེ་ན་མོ་ཧ་ལཱི་ནེ་ན་འཇིག་རྟེན་ན་ནི་སུ་ཞིག་གིས། །མི་ཤེས་པ་དེ་མཆོག་ཏུ་བསླུས། །སྐྱབས་ཀྱི་ཐབས་ནི་མི་ཤེས་པར། །བི་ཕ་ལི་ཀྲྀ་ཏ་ཛརྨ་ནཱ། །ཨ་ཙཱ་ན་ཤ་ར་ཎོ་བཱ་ཡི་བ་ཡོ་ཡེ་ན་བྱ་ཡི་ཀྲྀ་ཏཾ། །གང་གིས་ན་ཚོད་ཟད་པར་བྱས། །གཏི་མུག་གིས་རེག་སྐྱེ་བ་ནི། །འབྲས་མེད་བྱས་པ་དེ་ཡིས་ཅི། །གོཏྟ་མོ་བྷ་ག་ཝན་བུདྷ་ཤཱཀྱ་རཱ་ཛ་ཀུ་ལོདྷཱ་བ། །སཾ་བུདྷེ་ནུཏྟ་གོའུ་ཏམ་སངས་རྒྱས་བཅོམ་ལྡན་འདས། །ཤཱཀྱའི་རྒྱལ་པོའི་རིགས་ལས་འཁྲུངས། །ཁྱིམ་སྤངས་བླ་མེད་རཾ་སམྱཀ་སཾ་བོ་དྷི་མ་ན་ག་རི་ཀ། །པབྷུཉྫ་བྲ་ཛི་ཏཱ་ནཾ་ཙ་ཏ་སྤྱི་བྱང་ཆུབ་ཏུ། །ཡང་དག་རྫོགས་པར་སངས་རྒྱས་པ། །དེ་ཡིས་མཆོག་ཏུ་རྗེས་བཟུང་བས། །ཕྱི་ནས་བཱ་ནུ་གྲྭ་ཧཱཔ་རཾ། །བྷི་ཁྱུ་ཚཱ་ག་ཏ་རཱ་གཱ་ཎཱ་ས་མུ་ཧསཾ་གྷ་ཨུ་ཙ་ཏེ། །རབ་ཏུ་བྱུང་གྱུར་པ། །འདོད་ཆགས་བྲལ་བའི་དགེ་སློང་གིསི། །ཚོགས་ལ་དགེ་འདུན་ཞེས་སུ་བརྗོད། །ས་ཨེ་ཥ་བུདྡྷ་པྲ་མུ་ཁསཾ་སྭཱ་ཀུ་ཤ་ལེ་བི་ཎཱ། །མ་ཡཱ་པུ་ཎྱ་པ་ནཾ་སངས་རྒྱས་གཙོར་བྱས་དགེ་འདུན་ནི། །བསོད་ནམས་ཟོང་གིས་རྙེད་གྱུར་ནས། །རང་གི་དགེ་བ་འཚོལ་པྲ་སྱ་པྲ་ཎ་ཡེ་ན་ནི་མཾ་ཏྲི་ཏ། །ཨི་ཏི་ཏད་བ་ཙ་ནེ་ཐུ་ཏཱ་ཏཏ་མྲྀ་ཏྱཱ་ན་ཐ་བྱེད་པ། །བདག་གི་ཚིགུས་པས་མགྲོན་དུ་གཉེར། །དེ་སྐད་དེ་ཡི་ཚིག་ཐོས་ནས། །མགོན་མེད་ཟས་སྦྱིན་པིཎྜ་ད། །བུདྡྷ་ལསྨྲ་ན་བྷ་བེ་ན་ནི་ཤི་ནི་དཾ་ས་མཱ་ཡ་ཡོ། །ར་དེ་དྲན་ཞིང་། །སངས་རྒྱས་ལ་དམིགས་བསྒོམ་པ་ཡིས། །མཚན་མོ་གཉིད་དུ་ཡང་དག་སོང་། །མཚན་ཛ་རྩྭ་ཡཱ་མ་ཤེ་ཥཱ་ཡཾ་ས་མཱ་ཀྲྀཥྛ་ཨི་བོཏུཀཿ། །པྲ་བྷཱ་ཏ་མི་ཏི་ས་ཛྷ་ཏྭཱ་མོའི་ལྷག་མའི་ཐུན་ལ་ནི། །འདོད་པས་ཡང་དག་དྲངས་པ་བཞིན། །ཐོ་རངས་ཞེས་ཤེས་ཚེགྲོང་ཁྱེར་གྱི། །།
