དང་ར་ཎཱ་མ་ཙ་ཀཱ་ལ་བྷུཉྩ་བ་བྦྷ་བ་ཏུ། །ཡོ་བ་རཱ་ཛ་བྷ་བེ་ཀཱདྷཀུ་མཱ་ར་ཀཱ་ཤི་པུནྡ་པ་དང་། །དུས་ཀྱི་ས་ཞེས་བྱ་བ་བྱུང༌། །རྒྱལ་ཚབ་དབང་ནི་བསྐུར་འོས་པ། །གཞོན་ནུ་ཀཱ་ཤི་མཛེས་པ་ར༑། དྷནྣཱ་དྷརྨ་མ་ཡཾ་རཱ་ཛྲ་བི་ཙཱརྱ་ཙིནྟ་ཡཙྪི་རཾ། །ཀྵ་ཎ་ཀཱ་ཡིས། །རྒྱལ་སྲིད་ཆོས་དང་ཆོས་མིན་གྱི། །རང་བཞིན་ཡུན་རིང་དཔྱད་ནས་བསམས། །ལང་ཚོ་སྐད་ཡི་ཎི་ཏཱ་རུ་ཙྩེ་ཛཱི་བི་ཏི་ཤྲཱི་ཙི་ཙཉྩ་ལེ། །རཱ་ཚེ་སྭཱ་སྣ་བི་བཱ་ཧེ་སྨིན་མོ་ཧ་མཱུ་ཅིག་འཇིག་པ་དང་། །སྲོག་ནི་ཆུ་རླབས་གཡོ་བའི་ཚེ། །རྒྱལ་སྲིད་རྨི་ལམ་བག་མ་ལེན། །རྨོངས་པའི་རྩ་ལེ་ན་མེ་མ་ཏི། །རཱ་ག་ཕྲ་ལཱ་པ་བ་ཧུ་ལེ་མཱ་ཡཱ་མོ་ཧ་མ་ཧེ་མུ་ཧུ། །འདིར་བདག་བློ་མེད། །ཆགས་པའི་ཆོ་ངེ་རབ་མང་ཞིང་། །ཡང་ཡང་སྒྱུ་དང་རྨོངས་རང་བཞིན། །བེ་བྱ་རུ་ད་ཏ་ནི༔སཱ་རེ་སཾ་སཱ་རེ་ནཱསྟི་ས་ཏྱཱ་ཏཱ། །པྲ་བྲ་ཛཱ་མྱ་ན་གྷཱསྟསྣ་འཁོར་བ་སྙིང་པོ་མེད་པ་ལ། །འཕྱོན་མ་དུ་ཞིང་བདེན་པ་མེད། །དེ་སླད་ཁྱིམ་ནས་ཁྱིམ་མེད་པར། །དཱ་གཱ་རཱ་ད་ནཱ་གཱ་རི་ཀཾ། །ནིསྦི་ཤ་བྲྀཏྟི་སཾ་སཏྟཱ་བྷིརཱ་ག་ཀིཾ་བི་བྷུ་བདག་ནི་སྡིག་མེད་རབ་ཏུ་འབྱུང་། །རལ་གྲིས་གཅོད་ལ་རེག་གྱུར་པའི། །རྣམ་པར་འབྱོར་ལ་ཆགས་ཏི་བྷི། །རཱ་ཛ་སུ་ནུ་བི་ཙིནྟེ་ཏི་བི་བེ་ཀ་བི་མ་ལཱ་ཤ་ཡ། །ཨུ་བཱཙཱ་པས་ཅི། །དེ་ལྟར་རྣམ་བསམས་རྒྱལ་པོའི་བུ། །དྲི་མེད་དབེན་པའི་བསམ་པ་ཅན། །ནགས་སུ་འགྲོ་བྷེ་ཏ་བྷཱུ་པཱ་ལ་མ་ར་ཎྱ་ག་མ་ནོཏུ་ཀ། །མ་མ་སམྦྷ་ག་བསྒོ་ཡཾ་རཱ་བར་འདོད་པ་ཡི། །ས་སྐྱོང་ལ་ནི་མངོན་ཕྱོགས་སྨྲས། །རྒྱལ་པོ་བདག་ལ་ལོངས་སྤྱོད་ཀྱིས། །ཚོགས་ཛསྣེ་བོ་པ་ཡུ་ཙུ་ཏ། །ཡོ་བ་རཱ་ཛཱ་བྷི་ཥ་ཀརྟ་ས་མ་རནྡྷོ་ནི་བཱརྱ་འདི་ཉེ་བར་མཁོ་བ་མེད། །རྒྱལ་ཚབ་དབང་ནི་བསྐུར་འོས་པའི། །ཀུན་ཏུ་རྩོམ་པའི་ཟློག་པར་ཏྭཾ༑། ཀོ་སཱཀྐི་ཏསྟ༔སུ་ཏ་རཱ་མེ་ཏཱབྷྱཱ་ཏ་མ་མེ་མ་ཏཱ། །བ་མཛོད། །ཡབ་ལགས་གསོད་འཆིང་འཇིགས་དང་རྡུང་ཚེ། །སྐྱེད་བྱེད་རྒྱལ་པོའི་ཕུན་སུམ་ཚོགས། །ཁྲོ་དྷ་བནྡྷ་བྷ་ཡཱ་ཡཱ་ས་ཛ་ན་སྐྱོ་རཱ་ཛ་སམྤ་ད། །བྱཱyཀྲུ་ར་ཏ་རཱ་ཙཱ་རིརྫ་བའི་མེ་ཡིས་རབ་གདུངས་པཚེ། །འདི་དག་བདག་ནི་ཡོངས་མི་འདོད། །དུར་ཁྲོད་མེ་ཡི་རྩེ་མོ་བཞིན། །།
