ཏས་སཱུ་ནུ་བྷཱ་བ་ན་ཏ་ཏྲ་བྷུ་པཱ་ལ་ཡོ་སྟ་ཡོ། །ཡ་ཡོ་བི་ར་ར་ཛ་བྷཾ་ཏྭཱ་མི་ཐོ་མོ་ཧ་ནས་དེ་ཡི་མཐུ་ཡིས་ནི། །དེར་ནི་ས་སྐྱོང་དེ་དག་གིས། །ཁོན་གྱི་རྡུལ་ནི་ཞི་བ་ཡིས། །ལོག་པར་རྨོངས་པ་པ་རི་ཀྵ་ཡཾ། །ཀྲྀ་ཏ་སནྟིཔ་རེ་ཎཱ་ཐ་བཱ་ས་བཔྲི་ཐི་བཱི་པ་ཧེ། །བྷ་ཡོངས་ཟད་གྱུར། །དེ་ནས་ས་བདག་ནོར་ལྷའི་བུས། །ཕ་རོལ་དང་ནི་བསྡུམས་བྱས་ནས། །བཅོམ་ག་བནྟཾ་ས་མ་ཧྥེ་ཏྱ་སརྦ་བྷོ་གི་ར་པཱུ་ཛ་ཡ་ཏ། །པྲ་ཎི་དྷཱ་ནཾ་ཙ་བིད་ལྡན་འདས་ལ་མངོན་ཕྱོགས་ཏེ། །ལོངས་སྤྱོད་ཀུན་གྱིས་མཆོད་པ་བྱས། །མཆོད་པའི་མཇུག་ཏུ་བདག་ཕྱག་དྷ་པཱུ་ཛ་ནེ་པྲ་ཎ་མྱ་ཏཾ། །ཨ་ཧཾ་ཀུ་ཤ་ལ་མུ་ལེ་ན་རྩྭ་མེ་ཏེ་ན་མ་ཧཱ་འཚལ། །དགེ་བའི་རྩ་བ་དེ་དག་གིས། །བདག་ནི་ཆེན་པོར་གྱུར་ཅིག་ཅེས། །སྨོན་ལམ་དག་ཀྱང་བྱས་ནི་ཏི། །ཨམི་ནེ་ན་པ་ས་རེ་བྷཱ་པཾ་ཤྲཱི་ཁ་ཤནྟེ་ས་མུངྒ་ཏེ། །པྲ་བཱ་པར་གྱུར། །སྐབས་འདིར་ཤིན་ཏུ་ཆེ་བའི་སྒྲ། །དུང་ལས་ཡང་དག་འཁྲུངས་པའི་ཚེ། །ཕྱག་འཚལ་ཙ་ཏཾ་རཏྣ་ཤི་ཁཱི་པྲ་ཎ་ཏཾ་སུ་ར་ཏ་སྨྲི་ཏཾ། །ཤཾ་ཁོ་ནཱ་མ་མ་ཧི་པཱ་ལཉྩཀྲ་མདུན་དུ་གནས་དེ་ལ། །རིན་ཅེན་གཙུག་གིས་བཀའ་སྩལ་པ། །དུང་ཞེས་བྱ་བའི་ས་སྐྱོང་ནི། །འཁོར་བནྟ་བྷ་བི་ཥཱ་ཏི། །པརྨནྟེ་བོ་དྷི་ཡུཀྟཉྩ་ཀུ་ཤ་ལ་ས་མ་བཱཉྩ་བི། །ལོས་སྒྱུར་བར་ཁྱོད་འགྱུར་ཏེ། །མཇུག་ཏུ་བྱང་ཆུབ་ལྡན་པས་ཀྱང་། །དགེ་བ་ཡང་དག་ཐོབ་པར་འགྱུར༄༅། །ཨེ་ཐཾ་སཏྲ་ཎི་དྷཱ་ན་ཏཥི་ཏི་པ་ཏིཔུ་དྡྷོ་ད་ཡཱ་དཱ་ས་བཤཾ་ཁོ་ནཱ་མ་བྲྀ་པས་དེ་ལྟར་དམ་པའི་སྨོན་ལམ་བསོད་ནམས་རྒྱས་པ་ལ་ནི་ས་བདག་ནོར་ལྷའི་བུ། །མི་ཡི་བདག་པོ་དུང་ཞེས་རཏྭཱ་ཤི་ཁི་ནཱ་དིཋཤི་ཡཾ་སཱཤྩ་ཏི། །མི་ཏྲེ་ཡཔྲ་ཎ་ཡཱཏ་ཀ་རི་ཎྱ་ཏི་བྱ་བས་བློ་གྲོས་ཐོབ་འགྱུར་རིན་ཅེན་གཙུག་གིས་བསྟན། །དེ་བཞིན་བྱང་ཆུབ་ཏུ་ནི་རྣམ་དག་བློ་གྲོས་ཏ་ཐཱ་བོ་དྷོ་བི་ཤུདྡྷཾ་དྷི་ཡེ་ཀ་ལྱཱ་ཎཱ་བྷི་ནི་བེ་ཤ་པུ་ཎྱ་ཤ་ར་ཎེ་རཱ་དྱ་ཧི་བྱམས་པས་བྱམས་ལས་མཛད་པར་གྱུར། །གང་གི་དམ་པའི་གྲོགས་ནི་དགེ་ལ་མངོན་འཇུག་བསོད་སཏྠཾ་ག་མ། །ཨི་ཏི་ཀྵེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ནམས་ལམ་གྱི་དང་པོ་ཡིན། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་/།
