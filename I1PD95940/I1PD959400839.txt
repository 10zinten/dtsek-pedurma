ཏིན་ཛྷཱ་ཡ་ཏེ་ས་རི། །པྲཏྭཱ་ས་ནེ་སུ་དྷརྨཱ་ཡཀཱ་ལེ་ན་སྲ་ས་བཱ་བ་དྷོ། །གཉྷོ་ཡི་མཛའ་བོར་གྱུར། །དུས་ཀྱིས་ཆོས་བཟང་བཙའ་བ་ཡི། །མཚམས་དང་ཉེ་བར་གྱུར་པ་ན། །མཚན་རྦ་པཱནྟཾ་པྲ་ཧ་ནི་མིཏྟི་ཀ་ཛྙཱཔུ་རོ་ཧི་ཏ། །ཨ་ཐ་མནྟི་གི་རཱ་རཱ་ཛཱ་ཛརྨ་མཁན་མདུན་ན་འདོན་དག་གིས། །མངལ་གྱི་མི་བདག་མཐར་བྱེད་སྨྲས། །དེ་ནས་བློན་པོའི་ཚིག་གིས་ནི། །ཁཱ་ཎ་བ་དྷེ་ཤི་ཤོ། །ཏ་དནྟཔུ་ར་རགརྡོ་ན་བི་དེ་ཤོ་རྱ་ཏཱ་པྲ་དྷཱ་ན། །ཏདྲི་བྱིས་པ་སྐྱེས་ཚེ་གསོད་པ་ལ། །དེ་ཡི་ཕོ་བྲང་སྲུང་འོས་པ། །མཚོན་གྱི་སྦྱོར་ལྡན་རྒྱལ་པོས་བསྐོས། །དེ་ཙཱ་ཡ་སུ་དྷ~་པི་སི་ཏྲལྟ་ཤ་ར་ཎཾ་ཡ་ཡོ། །མ་ཧཱ་མཱ་ཏུཾ་བི་དྷཱ་ཏཱ་དག་ཤེས་ནས་ཆོས་བཟང་ཡང་། །བློན་པོ་ཆེན་པོ་བ་གླང་རྭ། །རང་འདོད་བཞིན་དུ་བྱེད་པ་ལས། །ར་མི་བ་སཱ་ཙྪ་ཀཱ་རི་ཎཾ། །པྲ་བྷུ་བྷཱརྨེ་ཏི་སཱ་ཏེ་ན་ནིངྷིཥྛ་ཏ་ན་ཡ་བྷི་ཡཱ། །ཤིན་ཏུ་སྐྲག་ནས་སྐྱབས་སུ་སོང་། །དེ་ཡི་རྗེ་བོའི་ཆུང་མ་ཞེས། །བསྟན་དེས་བུ་ཡི་འཇིགས་པ་ལ། །སཾ་ཛཱ་ཏཾ་སུ་ཏ་མ་ཏཀྵིཀྵེ་བརྟ་ནཱནྣི་ཀེ་ཏ་ནེ། །ཏ་ཏ་ཨེ་བ་ས་མཱ་ནཱི་ཏྭཾ་བུ་ནི་ཡང་དག་སྐྱེས་པ་དག། ཉ་པ་རྣམས་ཀྱི་ཁང་པར་བཞག། དེ་ཉིད་ནས་ནི་བུ་མོ་དག། དེ་ཡིས་ཏ་པཱ་ཀ་ནཱིནྡྲ་དནི་ཏྭཾ། །ཛཱ་ཏྭཱ་ནི་མིཏྟི་ཀ་བ་ཙས་ཏྱ་མྨེ་ནེ་ན་བྷུ་པ་ཏི། །བླངས་ནས་རབ་བསྟན་པ། །ཤེས་ནས་ས་ཡི་བདག་པོ་ཡིས། །མཚན་མཁན་ཚིག་ནི་བདེན་མིན་བཟུང་། །དྷི་མཱན་ཀ་བི་ཀུ་མཱ་རཱ་ཁཱས་ཀེ་བཏྟཾ་ཤྲཱི་ཧེ་ཤི་ཤུ། །ལེ་བྷེ་ཡ་ཏ་སྟ་ཏནྟཾ་ཉ་པའི་ཁྱིམ་དུ་བྱིས་པ་དེས། །རིག་པ་བཟོ་གནས་སྒྱུ་རྩལ་དག། དེ་དེ་ཐོས་པ་གང་དེས་ན། །བློ་བི་དྷུ་ཤིལླཟི་ཏཱཀ་ལཱ། །ས་ཏ་ཏྲ་བཱ་ལ་ཀིསཱཌྷ་ཀཱི་ཌཱ་ན་ག་ར་ཀྲྀ་ཏ་པ་ལྡན་གཞོན་ནུ་སྙན་དངགས་མཁན། །བྱིས་པ་རྣམས་དང་ལྷན་ཅིག་དེར། །ལམ་དུ་རྩེད་མོའི་གྲོང་ཁྱེར་ཐི༑། རཱ་ཛ་ཀ་ལཱི་པ་རི་ཙ་ཡིཅི་ཛ་ཧཱ་ར་མ་ཧཱ་བྷུ་ཛ། །ཡ་དྲིཙྪ་ཡཱ་ག་བྱས། །རྒྱལ་པོའི་རྩེད་འཇོ་ཡོངས་འདྲིས་ལ། །དཔུང་ཆེན་དེ་ནི་རྣམ་པར་རྩེ། །དེར་ནི་མཚན་མཁན་ཏསྟ་ཏྲ་ནི་མིཏྟི་ཀཱ་པུ་རོ་ཧི་ཏ། །ཏཾ་པྲཏྟཱ་ཚིགས་པ་མ་བྱེ་ཏྱ་བ་བྷཱ་ཥ་མདུན་ན་འདོན། །འདོད་དགའ་དག་ཏུ་འོངས་པ་ཡིས། །དེ་མཐོང་མི་བདག་ལ་ཕྱོགས་ཏེ། །གུས་པར་—།
