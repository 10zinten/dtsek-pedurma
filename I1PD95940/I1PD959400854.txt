དྲྀཋ་ཤཱ་ས་ར་ད་སྤན་བི་ཥཱ་ཎྞཾ་ཙིནྟ་ཡ་ཀྲྀ་ཤ། །སུ་ཙ་རི་ཏཱ་བྷི་མ་བསམ་པས་ཕྲ་དེས་མཐོང་དབུགས་དང་། །སོ་རེག་དུག་ལྡན་དེ་དག་ལ། །ལེགས་པའི་སྤྱོད་ཚུལ་མངོན་ཏཱནྟེ་ཏཱཾ་བི་ད་དྷ་དྷརྨ་དེ་ཤ་ནཱཾ། །ཏཱི་བྲ་ར་ཏི་ས་མུངྒིསྭ་དེ་ཤ་ག་འདོད་པའི། །ཆོས་ནི་རབ་ཏུ་བསྟན་པ་མཛད། །ཆགས་བྲལ་དྲག་པོའི་ཡིད་འབྱུང་གིས། །རང་ཡུལ་མ་ནོ་ཀ། །ས་ནཱ་གིརྦ་ཎི་ཛཱ་མེ་བ་རྒྱསྟཔྲ་བ་ཧ་ཎི་ཀཱ་ཎཱཏ། །འགྲོ་བར་འདོད་པ་དེ། །ཀླུ་རྣམས་ཀྱིས་ནི་ཚོང་པ་ཡི། །གཟིངས་ལ་སྐད་ཅིག་ཉིད་ཀྱིས་བཀོད། །ཏེ་པྲ་ཀྲྀཥྛས་མཱ་སཱ་རྱ་པ་ར་ལོ་ཀཱ་དི་བཱ་ག་ཏཾ། །པྲ་ཏྭཱ་བྲྀཏྟ་པྲ་བ་འཇིག་རྟེན་གཞན་ལས་འོངས་པ་བཞིན། །དེ་དག་དགའ་བ་ཡང་དག་ཐོབ། །གཟིངས་ནི་ཕྱིར་ལ་ལོག་ཧ་ཎ་སཱི་རཾ་པྲ་པུརྨ་ཧོ་ད་དྷེ། །སཱ་ཁཾ་ཧོཏྟཱཙྪཱ་ཡ་ཏེ་ཏཱཾ་བྲ་ཛ་ཏཱ་མ་ཏི་གྱུར་ནས། །རྒྱ་མཚོ་ཆེན་པོའི་ངོགས་སུ་ཕྱིན། །རང་གི་འདོད་པས་དེ་དག་རྣམས། །ཤིན་ཏུ་རིངས་པར་སཏྭཱ་རཾ། །ནི་དྭ་ལུསྠཱ་ལུ་ཀཱ་ཀཱུ་ལེ་བི་སྨྲ་ཏསཾ་གྷ་རཀྴི་ཏ། །ག་ཏེ་སཱརྗེ་བགྲོད་པ་ཡིས། །དགེ་འདུན་སྲུང་ནི་བྱེ་མ་ཡི། །ངོགས་སུ་གཉིད་དུ་སོང་བ་བརྗེད། །དེ་ནས་དོན་མཐུན་པྲ་བུདྷོ་ཐ་པྲ་བྷཱ་ཏེ་བི་ཛ་ནཱ་དི་ཤ། །།པ་བྱ་ནཾ་སུ་ཧྲིདྲི་ར་ཧི་ཏཔྲ་ད་དྷོ་སོང་ཚེ་སད། །སྔ་དྲོ་མི་མེད་ཕྱོགས་རྣམས་སུ། །ལྟ་ཞིང་གྲོགས་དང་བྲལ་བ་ཡིས། །ཆགས་བྲལ་བློ་ཅན་ས་བི་ཥཎྜ་རཱི། །ཨ་ཧོ་དྲྀཏྟཱཧྥ་ནཋཉྩ་གནྟབྷ་ན་ག་རོ་པ་མཱ། །བི་མོ་ཧ་དེས་བསམས་པ། །ཀྱེ་མ་མཐོང་དང་ཉམས་པ་ཅན། །དྲི་ཟའི་གྲོང་ཁྱེར་ལྟ་བུ་ཡི། །སྙིང་སྡུག་གཉེན་ཡནྟི་བི་ར་ཧེ་སུ་ཧྲིདྡནྡྷ་ས་མཱ་ག་མ། ་། ཏ་རུ་ཎ་ས་པ་རོདྷཱ་ལཱ་ལོ་དང་འགྲོགས་པ་དག། བྲལ་ན་རྣམ་པར་རྨོངས་པར་བྱེད། །ས་ཁར་ཨེཉ་གཞོན་འཕར་བ་ལྟ་བུར་རབ་ཏུ་ལཱཔ་ར་པྲ་ཡ་སཾ་ག་མཱ་བི་ད་དྷ་ཏི་ནཾ་ཎ་མཱ་སྦོ་མི་ཐུསྦི་ར་སྦི་ཏི་བནྡྷ་གཡོ་བའི་མཛེས་སྡུག་གྲོགས། །མི་རྣམས་དག་ལ་བརྫུན་གྱི་ཡིད་བརྟན་བརྟན་པར་གནས་པའི་འཆིང་བ་ནཱི༑། སཱ་ཛ་ན་ར་ཧི་ཏཤེ་ཏེ་གབྷེ་ཏ་ཐི་བ་བི་པ་བྱ་ཏེ་བ་ཙ་ར་ཏི་སྒྲུབ༈། རང་གི་སྐྱེ་བོ་དང་བྲལ་མལ་དུ་ཉལ་ཞིང་དེ་བཞིན་ངེས་པར་མཆིསི། །ཀུན་ཏུ་ཆགས་པ་དགེ་—།
