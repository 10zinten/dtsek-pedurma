མཀླནྟི་མཱ་ཡ་ཡོ། །ས་ཛ་རྩྭ་མཱ་ནཱི་ཏཱཾ་ནིསྣེ་ཧ་ལ་བ་ཎཾ་ཙི་རཏ། ་། ཀསྨྲ་：རིད་པས་ཀྱི་ངལ་བར་གྱུར། །ཡུན་རིང་ཞིག་ན་མ་ཡིས་ནི། །སྣུམ་དང་ཚ་མེད་གཟན་ཆང་ཞིག། ཁྱེར་ཥ་པིཎྜི་མཱ་སཱ་རྱ་བྷོཏྟཾ་སཱ་ད་ར་མ་ཡ་ཡོ། །དྷོ་ན་ཧསྟ་ཥ་ཎེ་ཏསྨིན་ནས་འོངས་པ་དེས་ཐོབ་ནས། །གུས་དང་བཅས་པས་ཟ་རུ་སོང་། །ལག་པ་བཀྲུས་པ་དེ་ཡི་ཚེ། །ཅི་སཾ་པྲཔ་ཡ་ཡ་ཛཱིཙྪ་ཡཱ། །ད་དོ་པྲ་ཏེ་ག་བུདྡྷ་ཡ་ཏྭཾ་པྲ་སནྣེ་ན་ཙེ་ཏ་ས། །དགར་ཡང་དག་འོངས་པ་ཡི། །རང་སངས་རྒྱས་ལ་དེ་ཉིད་ནི། །རབ་ཏུ་དང་བའི་སེམས་ཀྱིས་བྱིན། །ཛཱ་ཏས་ཨ་བ་ཀཱ་ལེ་ན་བྷུ་བ་ལོ་ཡཾ་པྲ་པེ་ན་ཛིཏ། །ཏ་སྱ་དཱ་ན་ཀ་ཎ་སྱེ་ཡཾ་དེ་ཉིད་དུས་ཀྱི་ས་སྐྱོང་ནི། །གསལ་རྒྱལ་འདི་རུ་སྐྱེས་པར་གྱུར། །འབྱོར་པ་འདི་ནི་སྦྱིན་པ་ཡི། །ཐིགས་བི་བྷུ་ཏིཔྲ་ཐ་མཾ་ཕ་ལཾ། །ཐུ་ཏཱི་ཏི་བྷ་ག་བདྡྷཱ་ཎཾ་བྷི་གྷརྦསྨ་ཡ་མ་ཡ་ཡོ། །པ་དེ་ཡི་དང་པོའི་འབྲས། །དེ་སྐད་བཅོམ་ལྡན་གསུངས་ཐོས་ནས། །དགེ་སློང་དག་ནི་ཡ་མཚན་གྱུར། །རཱ་ཛཱ་པི་པི་པུ་ལོ་བུ་ཛཾ་བི་ད་དྷེ་བྷ་ག་བ་ཏཔུ་ར། །རཱ་ཛཱཉེ་ར་ཁི་ལེགྷོ་གེརྒྱལ་པོ་ཡིས་ཀྱང་ཡངས་པ་ཡིས། །མཆོད་པ་བཅོམ་ལྡན་མདུན་དུ་བྱས། །རྒྱལ་པོས་ལོངས་སྤྱོད་མ་ཀྲྀ་ཏྭཱ་བྷཏྟི་ནི་བེ་ད་ནཾ། །ས་ཀོ་ཊུ་ཏི་ལཾ་ཀུམྦྷ་ནཾ་དཱི་པ་མཱ་ལཱ་མ་ཀལྤ་ཡ་ལུས་པས། །གུས་པས་ཞལ་ཟས་ཕུལ་བྱས་ནས། །འབྲུ་མར་བུམ་པ་བྱེ་བ་ཡིས། །མར་མེ་ཡི་ནི་འཕྲེང་བ་ཏ༑། དཱི་པ་མེ་ཀཾ་ད་དོ་ཏ་ཏྲ་སཱལྤ་ཀཾ་དུརྒ་ཏྭཾ་ག་ནཱ། །སྣེ་ཡཀྵ་ཡཱཏ་པྲ་ཡ་ཏེ་བྱས། །དེར་ནི་དབུལ་མོས་མར་མེ་ནི། །ཤིན་ཏུ་ཆུང་བ་དག་ཅིག་ཕུལ། །ཐམས་ཅད་སྣུམ་ནི་རབ་ཟད་ཏུ་སབྦེ་ཥཱ་ཎ་ཛ་གཱ་མ་ཡ། །ཙེནྟི་ཏཱཾ་བྲ་ཎི་དྷཱ་ནེ་ན་ཏ་ཡཱ་བི་མ་ལ་ཙེ་ནས། །རབ་ཏུ་བཟླས་ཚེ་གང་མ་སོང་། །དེ་ཡི་སེམས་ནི་དྲི་མེད་ཀྱིས། །སྨོན་ལམ་དེ་ཡིས་རྣམ་ཏ་པཱ། །བྷཱ་བི་ནཱི་ཤཱཀྱ་མུ་ནི་ཏྭཾ་སརྦ་དྷོ་སྣུས་མ་བྱཱ་དྷཱཏ། །རཏྣ་དཱི་པཱ་བ་བསམས་པ། །ཤཱཀྱ་ཐུབ་པ་ཉིད་དུ་ནི། །འགྱུར་བར་ཀུན་མཁྱེན་གྱིས་བཀའ་སྩལ། །བཅོམ་ལྡན་མདུན་ལི་ད་ཏྭཱ་རཱ་ཛཱ་བྷ་ག་བ་ཏཔུ་ར། །ཨུ་པ་བི་བྱ་པྲ་ཎ་མྱུ་གྲེ་ཎ་པྲ་ཎ་དུ་རྒྱལ་པོ་ཡིས། །རིན་ཅེན་སྒྲོན་མེའི་འཕྲེང་ཕུལ་ནས། །ཕྱག་འཚལ་ནས་ནི་མདུན་བསྡད་དེཏེ། །གུས་བསe།
