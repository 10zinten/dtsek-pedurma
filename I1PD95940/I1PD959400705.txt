སྨི་ཏ་པྲ་ས་ར་ནིརྫི་ཏ་བྷཱ་ར་ཧཱ་རཱཀླུ་རཾ་ཕ་ཙྪིནྡ་མ་བ་དརྦ་ན་དེ་བའི་འཛུམ་དཀྲིགས་དོ་ཤལ་ཁུར་ལས་ངེས་པར་རྒྱལ་གྱུར་པའི། །ནགས་ཀྱི་ལྷ་ཡིས་གདུག་པའི་གདེངས་ཅན་བ་ཏཱ་སྟ། །བྷོཀཱ་ལ་མེ་གྷ་ནི་ཀྲྀ་ཏི་པ་རི་མུཉྩ་ཀིལྕེནིཥལྤ་ཨེ་དབང་པོ་དེ་ལ་སྨྲས། །ཀྱེ་ལགས་དུས་ཀྱི་སྤྲིན་དག་ཀྱི་རྣམ་འགྱུར་དག་ནི་ཡོངས་སུ་ཐོབ༄༅། །ཁྱོད་ལྟ་བུ་བ་ཀ་ན་ཀཱ་ཙ་ལ་ཨེ་ཥ་ཡ་སྱ། །ཡུ༔་དཱི་དྷཱ༔པྲ་ལ་ཡ་མཱ་རུ་ཏ་བྷ་རྣམས་འཇིགས་པའི་རླུང་གིས་བཅོམ་པར་གྱུར་པ་ན། །གང་གི་རྐེད་སྐབས་ཕུག་གི་རོལ་པས་ྋསྐྱབས་སུ་ཛུ་མཱ་ནཱ་ལཱི་ལཱ་ནི་ཏམྦ་ཀུ་ཧ་རཾ་ཤ་ར་ཎཾ་པྲ་ཡཱནྟི། །ཏ་ཏསརྱ་པ་རི་གྷཱ་ཎ་མ་དའགྲོ་བྱེད་པའི། །གསེར་གྱི་རི་བོ་འདི་ལ་ཅི་ཡང་གཡོ་བ་མེད་པ་ཉིད། །དེ་ནས་འཕྲལ་ལ་རྒྱགས་པས་ནི། །སཾ་ཏྲ་ཀ་བི་ཀྲྀ་ཡ། །ནཱ་གཤཱསྟ་རམ་བྱེ་ཏུ་པྲ་ན་ནཱ་མ་ཀྲྀ་ཏཱཉྫ་ཡོངས་སུ་ཟད་ཅིང་རྣམ་འགྱུར་བཏང་། །ཐལ་མོ་སྦྱར་ནས་མངོན་ཕྱོགས་ཏེ། །ཀླུ་ཡིས་སྟོན་པ་ལ་ཕྱག་ལི༈། ཏ་སྱ་ཛ་ཏ་སྱ་ཤ་ར་ཎཾ་བྷ་ག་བརྐ་རུ་ཎཱ་ནི་དྷི། །ཤིཀྵ་འཚལ༑། བཅོམ་ལྡན་འདས་ནི་སྙིང་རྗེའི་གཏེར། །བསླབ་པའི་གནས་དག་རབ་བྱིན་པས། །སྐྱབས་སུ་པ་ད་པྲ་ད༔་ཀྲེ་ཀུ་ཤ་ལཱ་གྲ་མ་ནུ་གྲ་ཧཾ། །པྲ་ཎ་ཡ་དཎྜི་ཏསྟེ་ན་ཙ་ར་ཎཱ་སོང་བར་གྱུར་པ་དེ། །དགེ་བ་མཆོག་ལ་རྗེས་བཟུང་མཛད། །ཞབས་ལ་མགོ་ཡིས་གཏུགས་པ་དེ། །གུས་པ་ལོ་ན་མོ་ལི་ནཱ། །ས་ཏ་ཏཾ་ཏ་སྱ་བྷ་ག་བཱན་སནྣི་དྷི་བི་ད་དྷེ་ཛི་ན། །ཏསྨིནྣ་ཡིས་ནི་གསོལ་བཏབ་པ། །བཅོམ་ལྡན་རྒྱལ་བས་རྟག་ཏུ་ནི། །དེ་ལ་ཡང་དག་ཉེ་བར་མཛད། །གནས་སྐབས་བ་ས་རེ་ཏ་ཏྲ་པྲ་སཾ་གཱལྟ་སྱ་ཤཱནྟ་ཡེ། །ཡཀྟ་སྱ་ཙཀྲ་བྷ་ག་བཱན་བཛྲ་པ་ཎི་དེར་ནི་དེ་ཉིད་དུ། །གནོད་སྦྱིན་ལག་ན་རྡོ་རྗེ་དག། ཞར་ལ་འོངས་པ་ཞི་བའི་སླད། །བཅོམ་ལྡན་འདས་ར་ནུ་གྲ་ཧཾ། །ལོ་ཀོ་པ་ཏཱ་པ་མ་ས་མ་བི་ནི་བཱརྨ་ཤཱསྟཱ་བིལྟ་རི་ཏསྟུ་ཏི་པ་ཀྱིས་རྗེས་བཟུང་མཛད། །དེ་ནས་འཇིག་རྟེན་གདུང་བ་མི་བཟད་རྣམ་ཟློག་སྟོན་པ་ནི། །རྣམ་རྒྱས་བསྟོད་ད་ད་ཤཱ་ཙྩ་མཱ་ན། །བྷེ་ཛེ་བ་ནཱ་ནི་བི་ཧ་རསྣ་ཐ་པཱུརྦ་བུདྷཔཱ་པའི་ཚིག་གིས་སྐབས་གསུམ་པ་ཡིས་མཆོད་གྱུར་པ། །སྔོན་གྱི་སངས་རྒྱས་རྣམས་ཀྱི་ཞབས་པད་དག་དང་མཛའ་—།
