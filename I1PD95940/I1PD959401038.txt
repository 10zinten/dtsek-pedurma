བསྟན་འགྱུར། འཁྲི་ཤིང་། ཁེ། གི་ནཱ། །རཱ་ག་དྭེ་ཌོ་གྲ་དཾཥཱ་ཎ་བཱ་དྷ་ཏེ་ན་པུ་ནརྫ་ན། །ཨི་ཏི་ཏ་སྱ་བ་པའི་སྦྲུལ། །ཆགས་སྡང་དྲག་པོའི་མཆེ་ལྡན་པས། །སླར་ཡང་སྐྱེ་བོ་ལ་གནོད་མེད། །ཅེས་པ་དེ་ཚིག་སྣ་ཤཱ་ཤུ་ཐུ་ཏྭཱ་ཐོ་ཏྲ་ར་སཱ་ཡ་ནཾ། །ས་པྲ་སཱ་ད་བི་ཥ་དཱ་ཏྭཾ་ཛ་གཱ་ད་ཕྲ་མོ་དི་བ་ཡི། །བཅུད་ལེན་ཐོས་ནས་སྒྱུག་མོ་ནི། །འཕྲལ་ལ་རབ་དང་དྲི་མ་མེད། །དགའ་མགུ་ལྡན་པས་དེ་ལ་ནཱི༑། ཨ་པི་ཏརྡན་ནོ་པཱ་ཡཀཕྱི་དསྟི་བ་རཱ་ན་ནེ། །ཨ་པི་སྨྲས༑། དེ་ནི་བལྟ་བའི་ཐབས་འགའ་ཞིག། མཆོག་གི་བཞིན་ལྡན་ཡོད་དམ་ཅི། །ཁྱོད་ཀྱི་བསོད་ཏཱཀྐུ་ཎྱ་སམྲནྣཱདྲ་ཡ་མ་པ་མྲ་ཏསྤ་དཾ། །ཨི་ཏི་སཱ་ནུ་ན་ཡཾ་ཤྭ་ཐུ་ནམས་འབྲེལ་པ་ལས། །ངེད་ཀྱང་བདུད་རྩིའི་གནས་ཐོབ་བམེ། །ཞེས་པ་སྒྱུག་མོས་གུས་ལྡན་བློ། །ཥཱ་ས་ད་ར་དྷི་ཡཱ་ནཱི་ཏཱ། །ཏཾ་བསནྡན་ཡཱ་མཱི་ཏི་བ་བྷཱ་ཥ་བྷཏྟི་མཱ་ནི་ནཱི། །ཚུལ་དང་མཐུན་པར་དེ་ལ་ཞུས། །བདག་གིས་ཁྱེད་ལ་དེ་བསྟན་ཞེས། །གུས་དང་བཀུར་བ་ལྡན་མས་སྨྲས། །མ་ཧཱ་པྲ་ཏི་ཛྷ་སཾ་བྷཱ་ར་བྷ་ར་ནིབྷ་ཧ་ནི་ཎི་ཎཱི། །སཱ་པི་ཤ་ཡ་ཏུ་ལཱ་དམ་བཅའ་ཆེན་པོའི་ཚོགས་ཀྱིས་ནི། །ཁུར་དག་ཁྱེར་བ་འདོད་པ་ཡི། །སྲོད་ལ་མལ་སྟན་སྲང་ལ་རཱུ་ཌྷ་་ཎཾ་དྷཱ་ན་པ་རཱ་བྷ་བཏ། །ཏ་ཏཔྲ་པཱ་ད་མཱ་རུ་རྒྱ་བྷ་ག་འཛེགས༑། སྐད་ཅིག་བསམ་གཏན་བསྐྱངས་པར་གྱུར། །དེ་ནས་ཁང་བཟང་ལ་འཛེགས་ཏེ། །བཅོམ་བཏྲེ་བ་ཏྭཾ་དི་ཤཾ། །པྲ་ཎི་པ་ཏུ་སྲི་ཛཀླུ་ཛྙཱ་པཱུ་ཛཱདྷ་ཀུ་སུ་མཱཎྜ་ལི། །ལྡན་བསྟན་པའི་ཕྱོགས་སུ་ནི། །ཕྱག་འཚལ་མཆོད་བྱ་མཆོད་འོས་པའི། །མེ་ཏོག་སྙིམ་པ་རབ་ཏུ་གཏོར། །པུ༔་དྷ་པོ་ད་ཀི་རནྡྷཾ་པཱ་ཀྲྀ་ཏྭཱཏྟ་ས་དོརྨུ་ཁཱི། །ཨཱ་ནནྡ་བཱཐ་ཐཾ་རུདྡྷ་དེ་ཡི་གནས་སུ་མངོན་ཕྱོགས་དེས། །མེ་ཏོག་བདུག་པ་ཆུས་མཆོད་ནས། །ཡངས་པའི་མིག་ནི་ཀུན་དགའ་བིཙླིརྞ་ན་ཡ་ནཱ་བ་དཏ། །ཨ་ནུ་ཀཾ་པི་བ་བྷ་ག་བནྡནྣ་ཏྲ་ཡི༑། མཆི་མས་འགགས་པས་རབ་སྨྲས་པ། །བཅོམ་ལྡན་ཐུགས་རྗེ་འོས་པ་བདག། དཀོན་མཆོག་ཡ་བི་བརྫི་ཏཱ། །ཏ་བཱ་ཤྲ་མ་མྲྀ་ཀཱི་བཱ་ཧཾ་དུ་ར་དེ་ཤ་མི་མཾ་ག་ཏཱ། །གསུམ་པོ་རྣམ་སྤངས་ཏེ། །ཁྱོད་ཀྱི་གནས་ཀྱི་རི་དྭགས་བཞིན། །ཤིན་ཏུ་ཐག་རིང་ཡུལ་འདིར་ཕྱིན། །།—།
