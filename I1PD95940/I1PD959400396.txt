ལུ་ལ་མཱ་ཡཾ་ཏཾ་བྷི་ཏཾ་ཀཱ་རུ་ཎྱ་སཱ་ག་ར། །ཤཱསྟཱ་ཀ་རེ་ཎ་ས་སྤྱན་ཙཀྲ་སྭསྟི་ཡི༑། ཞབས་ཀྱི་དྲུང་དུ་འོངས་གྱུར་ཞིང་ཚེ། །འཇིགས་པ་དེ་ལ་འཁོར་ལོ་དང་། །བཀྲ་ཤིས་ཀྱིས་ཀ་ལཀྵ་ཎཱ། །ཀུམྦྷ་བི་སྟ་ཧསྟ༈་པོ་བཱ་ཙ་བྷ་ག་བཱན་ཛི་ན། །པུ་མཚན་ཕྱག་གིས་རེག། བུམ་པར་ཕྱག་བཀོད་དེ་ལ་ནི། །བཅོམ་ལྡན་རྒྱལ་བས་བཀའ་སྩལ་པ། །གནས་ཏྲ་སཱ་ཀརྨ་ཎི་བེ་ཚྭ་སྟི་སི་མ་མི་ནཱ་ད་ཧཱུཾ། །བི་བེ་ཀཱ་ལོ་ཀ་ཛ་ལ་དཀཱ་སྐབས་དམན་པ་འདི་ཉིད་ཉེབུ། །རང་ལས་ཉིད་ཀྱིས་ཐོབ་པ་ཡིན། །ཁྱོད་ལུས་ཤ་ཡི་རི་བོ་ནི། །དབེན་ཡོཉཡཾ་ཏཱཾ་ས་བྷཱུ་དྷ་རཿ། །བྷཱ་རསྟེ་མོ་ཧཱ་སཾ་བྷཱ་རཿཔཱ་པཱ་དུ་ས་ན་ཏཔ་རཾ། །པའི་སྣང་བ་དག་གི་སྤྲིན། །རྨོངས་པའི་ཚོགས་ཀྱིས་ཁུར་གྱུར་པ། །སྡིག་པས་མཆོག་ཏུ་བསྐྲུན་པ་ཡིན། །ཨི་ཏྱཱཀྟི་ཀ་རུ་ཎཱརྗེ་ཎ་བྷི་ཏོ་བྷ་ག་བ་ཏཱ་ག་ཛ། ་། ས་ལབྷ་ཤཱ་ས་བཅོམ་ལྡན་སྙིང་རྗེས་བརྒྱན་པས་འདི། །བརྗོད་ཚེ་འཇིགས་པའི་གླང་པོ་དེ། །ཡིད་བརྟན་ཐོབ་ནས་མཱ་ལཱ་ན་ལི་ནོ་ནི་ལ་ཏྭཾ་ཡ་ཡོ། ་། དེ་བ་དཏྟ་སྱ་སཾ་ཀཝེ་ཀུཉྫ་རེ་གདགས་ཐག་གིས། །བཏགས་བཞིན་གཡོ་མེད་ཉིད་དུ་གྱུར། །ལྷས་བྱིན་གྱི་ནི་བརྟག་པ་དང་། །གླང་པོ་ཙ་མ་ཧོཀྐ་ཊེ། །བྷགྷེ་ནིཟི་་ཧརོ་བྷཱུ་ཏ་ས་མུདྡྷཱ་ཏཱདྷཱ་ཏོ་ཛ་ན། །ཏ་དྲག་པོ་ཆེན་པོ་ཡང་། །ཉམས་ཚེ་གེགས་ེམེད་དགའ་བ་ཡིས། །སྐྱེ་བོ་ཡ་མཚན་སྐྱེས་པར་གྱུར། །དེ་ཏཀྲྀ་ཏྭཱ་གྲོ་ཧ་པ་ཏེཀྲྀ་ཧེ་བྷོ་ཛུ་པཱ་ཏི་གྲ་ཧཾ། །བྷ་ག་བཱན་བྷི་ཀླུ་བྷིནས་ཁྱིམ་བདག་ཁྱིམ་དུ་ནི། །ཟས་ནི་རབ་ཏུ་བླང་མཛད་ནས། །དགེ་སློང་དག་དང་ཐབས་ཅིག་ཏུ། །སཱདྷཾ་ཀཱ་ན་ནཱ་གནྟུ་མུདྨ་ཡོ། །ཨ་བྷི་ཀྲྀ་ཏྱཱ་ག་ཛནྡོ་པི་ཛི་ན་མྱ་ཙ་ར་བཅོམ་ལྡན་ནགས་སུ་གཤེགས་བརྩོན་གྱུར། །གླང་པོའི་དབང་ཡང་མངོན་བཏུད་དེ། །རྒྱལ་བའི་ཞབས་ཆ་སྔ་ཡོ། །།ཀྲྀ་ཏྭཱ་ཀ་རེ་ཎ་སཾ་ཨཱནཾ་བ་བུས་ཏྭཱ་ཛ་ཀོཉྫ་རཾ། །ཙ་ཏུ་རྨ་ཧཱ་རཱ་ཛི་ཀྱི་པདྨ་ལ། །ལག་པས་རེག་པར་བྱས་ནས་ནི། །གླང་པོས་ལུས་ནི་བཏང་བར་གྱུར། །རྒྱལ་ཆེན་བཞི་པའི་ཀེ་བ་དེ་བེ་ཥཱ་བི་ཥ་དདྷུ་ཏི། །ཨུ་པ་པརཿས་ས་ཧཱ་ས་སུ་ག་ཏཾ་ཛི་ལྷ་དག་ཏུ། །ཉེ་བར་འཁྲུངས་ཤིང་མཛེས་པ་ནི། །དྲི་མ་མེད་དེ་འཕྲལ་ཉིད་ལ། །བདེ་བར་གཤེགས་པ་—།
