ཀཾ་ཀ་ཊཾ། །སུ་ཧྲི་ཏ་པྲ་ཏི་ཏ་མཱ་བཱ་ཡ་པྲ་བཱ་ཙ་ས་ཙི་བཱན་ནི་པ། །ཨི་དཾ་ཆེན་གོ་ཆ་ནི། །གྲོགས་ཀྱིས་སྤྲིངས་པ་ལེགས་བླངས་ནས། །རྒྱལ་པོས་བློན་པོ་རྣམས་ལ་སྨྲས། །ས་བདག་མེ་པྲ་ཧི་ཏཾ་ཏེ་ན་སོ་ཛ་རྒྱ་མི་བ་བྷུ་བྷུ་ཛཱ། །སརྦ་རཀྟ་ཀྵ་མཾ་བརྨ་གཱ་ཌྷ་དེ་ཡིས་ལུགས་བཟང་བཞིན། །ལྷག་པར་མཛའ་བ་གོ་བྱེད་ཅིང་། །ཐམས་ཅད་བསྲུང་བར་བཟོད་པ་ཡི། །པྲ་མ་ནི་བི་དེ་ཀཾ། །ན་པ་བྱ་མྱ་སྱ་ས་དྲི་ཤཾ་པྲ་ཏི་དེ་ཡཾ་ཏ་ཏོ་དྷི་ཀཾ། །ཨལྤ་གོ་ཆ་འདི་ནི་བདག་ལ་སྤྲིངས། །འདི་ལ་ལན་དུ་སྦྱིན་འོས་པ། །དེ་ལས་ལྷག་པ་ངས་མ་མཐོང་། །ཕན་པྲ་ཏི་ཀྲི་ཡཱ་ཤ་ལྱ་མུ་པ་ཀཱ་རོ་པ་ཀཱ་ར་ཡོ། །ཨུ་ཙི་ཏཾ་ཙིནྟ་ཊཱ་ཀིཉྩིཏ་པྲ་ཥ་དང་གནོད་པ་དག་ལ་ཡང་། །ལན་ནི་ཆུང་བ་ཟུག་རྔུ་ཡིན། །འདི་ལས་ལྷག་པ་ཅུང་ཟད་ཅིག། བསྐུར་ཎྜི་ཡ་མ་ཏོ་དྷི་ཀཾ། །སཤྲཱིབྷ་བདྷཱི་རི་ཏུཀྵ་ཚིགས་པཕྱིནཱ་ཀུ་འོས་ཁྱེད་རྣམས་ཐམས་ཅད་ཀྱིས། །བསམ་པར་མཛོད་ཅེས་བརྗོད་ནས་ནི། །མི་བདག་བསམ་པ་ལོ་བྷ་བཏ། །ཨ་ཐ་དྷི་མཱན་མ་ཧཱ་མཱ་ཏྱཤྲཱི་རཾ་དྷུ་ཏཱ་ཏ་མ་པྲ་བཱིཏ། །བནྡྷ་ཀཱ་ར་འཁྲུགས་པར་གྱུར། །དེ་ནས་བློ་ལྡན་བློན་པོ་ཆེ། །དབྱར་གྱི་རྣམ་པ་ཞེས་བྲམ་ཟེ། །རིག་ཀུན་ལེགས་བྷི་དྷོ་བི་པྲསརྦ་བི་དྱ་སུ་པཱ་ར་ག། །ཨ་ཏོ་བ་ཧུ་གུ་ཎཾ་རཱ་ཛན་ནི་ཀ་པར་ཕ་རོལ་ཕྱིན། །ཡུན་རིང་བསམས་ནས་དེ་ལ་སྨྲས། །འདི་ལས་ཡོན་ཏན་མང་བའི་སྐྱེས། །རྒྱལ་པོ་མེ་བསྐྱུ་པཱ་ཡ་ནཾ། །ཏ་སྱ་སཾ་པྲ་ཥ་ཎེ་ཡཏ་ནཀྲི་ཡ་ཏྭཾ་ཡཾ་དི་ཤཀྱ་ཏེ། །ཡ་ཨེ་གཅིག་ཀྱང་ཡོད་མ་ཡིན། །གལ་ཏེ་ནུས་ན་འབད་པ་ཡིས། །དེ་ནི་རབ་ཏུ་སྤྲིང་བར་མཛོད། །གང་ཞིག་ཥ་བྷ་ག་བཱན་བུདྷ་སྦི་ཏདྲི་ཥ་ཡཱནྟི་ཀ། དེ་བཱ་ནཱ་མཱ་ད་ར་སྦེ་བཅོམ་ལྡན་སངས་རྒྱས་འདི། །ཁྱོད་ཀྱི་ཡུལ་ན་ཉེ་བར་གནས། །ལྷ་རྣམས་དག་གི་ལུས་གནས་ཀྱི། །ནཾ་པ་ཊསྟཏྠ་ཏི་མངྒཱི་ཏ། །ཨ་ཤེ་ཥ་ལོ་ཀ་ཀ་ལྱཱ་ཎ་ཀ་ལི་ཀཱ་ཀལྤ་པཱ་ད་ཐང་ཀ་དེ་ཡི་གཟུགས་ཀྱིས་མཚན། །མ་ལུས་འཇིག་རྟེན་དགེ་བ་ཡི། །མེ་ཏོག་གསར་བའི་དཔག་བསམ་བ༈། །ཙི་ཏྲེ་སྭསྣེ་ཐ་སཾ་ཀལྤ་སྲི་ཐུ་པུ་ཙྪིས་དྲི་བྱ་ཏེ། །ཨི་ཏི་མནཾ་བ་ཤིང་། །རི་མོ་རྨི་ལམ་ཀུན་རྟོག་ལའང་། །དེ་ནི་བསོད་ནམས་ཆེ་བས་མཐོང་། །ཞེས་པ་བློན་པོའི་ཚིག་།
