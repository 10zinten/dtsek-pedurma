བྷ་ཀྵ་ཛཾ་བུ་དཱི་པ་ག་ཏཾ་ཛ་ནོ། །ཡཉྩི་ཀཾ་བོ་དྷི་སཾ་ཡུཏྟཾ་ཏ་སྱ་པུ་ཙྩཾ་ཏ་ཏོན་དྷི་ཀཾ། །འགྲོ་ཀུན་གུས་པས་གསོལ་བྱེད་དང་། །སུ་ཞིག་བྱང་ཆུབ་ལྡན་ཞིག་ལ། །དེ་ཡི་བསོད་ནམས་དེ་བས་ལྷག། ཨི་ཏི་བཱཀྱི་བྷ་ག་བ་ཏསྟ་ཐཾ་ཐུ་ཏྭཱ་བི་ནརྩྭ་ཙ། །ནི་ཛ་གེ་ཧཾ་ཅེས་པ་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས། །བདེན་པ་ཐོས་ནས་མངོན་དགའ་ཞིང་། །དེ་ལ་ཕྱག་འཚལ་ཐཾ་ཧ་པ་ཏིཔྲ་ཡ་ཡོ་པྲ་ཎི་པ་ཏྱ་ཏཾ། །དཱི་པཾ་ད་ཏྭཱ་པ་ཋ་ན་རཱ་ཊོཏ་ཏྲ་ཁྱིམ་བདག་ནི། །རང་གི་ཁང་པར་རབ་ཏུ་སོང་། །མཚན་མོ་མར་མེ་བཏང་ནས་དེར། །བཅོམ་ལྡན་རྗེས་བུདྡྷཱ་ན་ཤཱ་ས་ནཱཾ། །དཱཎཱ་ཡ་རཱ་ཛ་པུ་རུ་ཥས་ནཱི་ཏོཉགྷི་པྲ་བརྟ་ནཱཏ། །དཎྜ་བསྟན་ཀློག་པ་ན། །མེ་ནི་རབ་བསྒྲུབས་ཆད་པའི་སླད། །དེ་ནི་རྒྱལ་པོའི་སྐྱེས་བུས་བཟུང་། །ཆད་པའི་སྱ་སམྦྷ་བཱཏ་པདྷཾ་བནྟ་ནཱ་གཱ་ར་བནྟི་ནཾ། །ཏ་དྲྀཤྩ་མཱ་ཡ་ཡུ་དེ་བཱ་དངོས་པོ་མེད་པ་ན། །བཙོན་ཁང་དག་ཏུ་བཞག་སྟེ་བཅིངས། །མཚན་མོ་བརྒྱ་བྱིན་ཚངས་པ་སོགས། །རཱ་ཊོ་ཤཀྲ་བྲངྒཱ་ད་ཡ། །ས་ཏིཌྷ་ནཾ་གཾ་ཧཱ་ཎེ་ཏི་པྲརྦ་ཏོ་ནཱ་གྲ་ཧཱདྨ་དཱ། །ལྷ་རྣམས་དེ་ལ་བལྟ་རུ་འོངས། །དེ་དག་གིས་གསོལ་ནོར་ལོང་ཞེས། །གང་ཚེ་དེ་ཡིས་མ་བླངས་པ། །ཏ་དཱ་དྷརྨོ་པ་དེ་ཤཱ་ཡཾ་པྲ་བྲྀཏྟཾ་ཏ་སྱ་མནི་ཡེ། །རཱ་ཛཱ་པི་ཏཏྠ་བྷཱ་བེ་ན་དེ་ཚེ་དེ་ཡི་ཁང་པར་ནི། །ཆོས་བསྟན་པ་ལས་བསགས་པ་འཕེལ། །དེ་ཡི་མཐུ་ཡིས་རྒྱལ་པོས་ཀྱང་། །དྲྀཏྟཱ་བྲ་ཛཱ་ལི་ཏཾ་པུ་རཾ། །མུཀྟ་ཏཾ་བནྡྷ་ནཱ་གཱ་རཱ་ད་དན་ན་ལཾ་ཀཱ་ཙིཏ། །གྲོང་ཁྱེར་རབ་ཏུ་འབར་མཐོང་ནས། །བཙོན་ར་ནས་ནི་དེ་བཏང་ཚེ། །གང་དུའང་མེ་ནི་མཐོང་མ་གྱུར། །ས་ག་ཏསུ་ག་ཏཾ་པྲྀཏྟཾ་ཀ་དཱ་ཙིཏྟཀླུ་རཿསྦི་ཏ། །ནི་ཐཾ་ཛི་ན་པྲ་ཎཱ་མཱ་ཡ་ནམ་ཞིག་བདེ་གཤེགས་བལྟ་རུ་ནི། །ཕྱིན་ནས་དེ་མདུན་འཁོད་པ་དེས། །མི་བདག་རྒྱལ་ལ་ཕྱག་འཚལ་པཱསྟ་པདྷུ་ལོ་ཀ་ཡཏ། །ཨ་གྲེ་བྷ་ག་བ་ཏཐུཀྲེ་པྲ་ཎ་ཡཾ་ན་ས་སླད། །ཕྱི་@ནས་འོངས་པ་མཐོང་བར་གྱུར། །དེ་ཡི་སྔོན་དུ་བཅོམ་ལྡན་ལ། །ཕྱག་བྱས་ས་ཡི་བདག་བྷུ་ཏེ། །ཛ་གཏྠུ་ཛ་སྱ་པུ་ར་ཏཔཱུ་ཛ་མརྟ་ཏི་ནཱ་པ་རཿ། །ཛི་ན་མ་མནྡྷ་ལ་མིན། །འགྲོ་བས་མཆོད་འོས་སྔོན་དུ་ནི། །གཞན་ལ་མཆོད་པར་འོས་པ་མིནི། །མི་བདག་རྒྱལ་ལ་—།
