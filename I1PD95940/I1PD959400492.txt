ན་ནཱི་མ་བ་པྲ་བྲ་ཛྙཱ་ན་ཏུ་ཏཱ་དཱི་ཛ། །ཐ་མ་ཎིརྨེ་ཧྲ་ཏཔུ་ཏྲ་ཨི་ཏི་ས་ཡིས་མ་ཉིད་བཏང་། །རབ་ཏུ་བྱུང་བ་བཏང་མ་གྱུར། །བདག་གི་བུ་ནི་དགེ་སྦྱོང་གིས། །ཕྲོགས་ཞེས་ཆོ་བི་པྲ་ལཱ་པི་ནཱི། །ཨུན་དི་ནི་ཏ་ནཾ་ཏྲཀཱ་པྲ་པེ་དེ་ན་ར་ཀསྦི་ཏི། །ཏ་ཏཀཱ་ངེ་སྨྲ་དེ་ཡང་། །སྨྱོན་མར་གྱུར་ཅིང་ལུས་བཏང་ནས། །དམྱལ་བ་དག་ཏུ་གནས་པ་ཐོབ། །དེ་ནས་དུས་ལེ་ན་ཀ་པི་ལ༔སྭ་ཡཾ་ཀ་ལི་ར་ཀིལླི་ཥ༅། །དེ་ཧཱནྟེ་བཱཀྱ་པཱ་རུ་ཥཱ་དི་ཚྭ་མ་ཀྱིས་སྡིག་པ་དག། རང་གིས་བྱས་པའི་སེར་སྐྱ་ནི། །ལུས་ཀྱི་མཇུག་ཏུ་ཚིག་ཉེརྩུབ་ཀྱིས། །ཆུ་སྲིན་འདི་ཀ་ར་ཏྭཾ་ག་ཏ། །ཏཱ་རྱ་ཏཱ་ནི་མུ་ཁ་ནྱ་སྱ་ཡཱནྟཱ་ཙེ་བྷིཀྵ་བྷནྡ་ནེ། །ཕ་ལཾ་ས་དྲྀ་ཤ་ཉིད་དག་ཏུ་གྱུར། །དགེ་སློང་སྨད་ལ་གང་སྨྲས་པ། །དེ་དག་འདི་ཡི་གདོང་འདི་དག། གང་ཕྱིར་ལས་ཀྱི་རཱུ་པཾ་ཧི་ཀརྨ་བཱི་ཛཱཏ་པྲ་ཛཱ་ཡ་ཏེ། ་། ཨི་ཏྱཱཀྟ་ཏ་ཏྲ་བྷ་ག་བཱན་དྷརྨ་ས་བོན་ལས། །འབྲས་བུ་མཚུངས་པའི་རང་བཞིན་སྐྱེ། །ཞེས་གསུངས་དེར་ནི་བཅོམ་ལྡན་གྱིས། །རྟག་མཱ་དི་བྱ་ཤཱ་ཤྭ་ཏཾ། །ཛ་ན་སཱུ་ནུ་གྲ་ཧཉྩ་ཀྲེ་ནཱ་ནཱ་བོ་དྷི་བི་དྷཱ་ཡ་ཀཾ། །པའི་ཆོས་ནི་བསྟན་མཛད་ནས། །ཐ་དད་བྱང་ཆུབ་སྒྲུབ་བྱེད་པའི། །སྐྱེ་བོ་དག་ལ་རྗེས་བཟུང་མཛད། །ཏ་ཏཔྲ་ཡཱ་ཏེ་སྭ་པ་དཾ་ཛི་ནེ་ཏརྨ་ཡ་མཱ་ན་ས། །མ་ཀ་རཿཔྲཛཱི་ཏཱ་ཧཱ་རyཏྟཱ་དེ་ནས་རྒྱལ་བ་རང་གནས་སུ། །གཤེགས་ཚེ་དེ་ཡི་རང་བཞིན་ཡིད། །ཆུ་སྲིན་ཁ་ཟས་རབ་སྤངས་ནས། །དེ་ཧཾ་དི་པཾ་ཡ་ཡོ། །ཙཱ་ཏུརྨ་ཧཱ་རཱ་ཛི་ཀེ་ཥཱ་དེ་བ་ཥཱ་བི་ཤ་ད་དྷུ་ཏི། །ལུས་བཏང་ལྷ་ཡི་གནས་སུ་སོང་། །དེ་ནི་སྐད་ཅིག་བདེ་གཤེགས་ལ། །སེམས་ནི་རབ་ཏུ་དང་བ་ལས། །ཤྲཱི་མཱན་ས་ཛཱ་ཏསུ་ག་ཏེ་ག་ཎཾ་ཙིཏྟ་པྲ་སཱ་ད་ནཱཏ། །ཏ་ཏཔུཎྞེནྡྲ་བ་ད་ནརྒྱལ་ཆེན་བཞི་པའི་ལྷ་རྣམས་སུ། །དཔལ་ལྡན་དྲི་མེད་མཛེས་པར་སྐྱེས། །དེ་ནས་ཟླ་བ་ཉ་བའི་བཞིན། །སྲགྷི་རུ་ཙི་ར་ཀུཎྜ་ལ། །ས་སཱ་ཀཱ་ར་ཨི་བཱ་ནནྡསུ་ག་ཏཾ་དྲཏྟུ་མཱ་འཕྲེང་བ་དང་ལྡན་རྣ་རྒྱན་མཛེས། །ཀུན་དགའ་རྣམ་པ་དང་བཅས་བཞིན། །དེ་ནི་བདེ་གཤེགས་བལྟ་ཡ་ཡོ། །པྲ་ཀིརྞ་དི་བྱ་ཀུ་སུ་མཀིརཱི་ཊ་ཀྲྀཥྛ་བྷཱུ་ཏ་ལ། །པྲ་བྷཱ་པཱུ་རི་ཏ་རུ་འོངས། །ལྷ་ཡི་མེ་ཏོག་རབ་ཏུ་གཏོར། །ཅོད་པན་གྱིས་ནི་ས་ལ་རེག། འོད་ཀྱིས་ཕྱོགས་ཀྱི་འཁོར་ལོ་།
