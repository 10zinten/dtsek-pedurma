ཏ་ཏཔ་པཱ་ཏཱ་ཁི་ལ་བྷོ་ཛུ་བྲྀཏྠི་རིཥྛ་པྲ་ཛཱ་ནཾ་བྷུ་བི་སཔྟ་རཱ་ཏྭཾ། །དྷཱ་རྩྭ་དི་བྲྀཥྛསྟ་དེ་ནས་ཞག་བདུན་ས་ལ་སྐྱེ་རྒུ་རྣམས། །འདོད་པའི་ཁ་ཟས་མ་ལུས་ཆར་དུ་འབབ། །དེ་ཡི་འོག་ཏུ་ད་ནནྟ་རཾ་ཙ་རཏྟཱ་དི་བྲྀཋཤྩ་ཏ་ཏཀྲ་མེ་ཎ། །ཨི་ཏི་ས་ཀ་ན་འབྲུ་ལ་སོགས་པའི་ཆར། །དེ་ནས་རིམ་གྱིས་རིན་ཆེན་ཚོགས་ཆར་ཡང༌། །དེ་ལྟར་ལེགས་བྱས་ཕུན་ཀ་བཎྜ་ཀླུ་པ་ཏིནྟཱ་བ་ཏཱི་ནཱ་མུ་ཀུ་ཊ་མ་ཎི་རི་བོ་ཙིནྟཱ་ཛམཱ་ན༔པྲ་ཛཱ་ནཾ། །ཚོགས་ལ་དགའ་ས་ཡི་བདག་པོ་དག། གསེར་མདོག་དེ་ནི་ས་བདག་ཀུན་གྱི་ཅོད་པན་ནོར་བུ་བཞིན། །ཨ་ཀྲྀ་ཏ་སུ་ཀྲྀ་ཏ་སམྨཏྤ་ཎི་ནཔཱ་ཎ་རཀྵ་པྲ་བྷ་བ་ཏི་ཧི་པ་རཟེ་སཛྫ་ན་མཆོག་ཏུ་མཛེས་ཤིང་སྐྱེ་རྒུ་རྣམས་ཀྱི་སྲོག་ནི་བསྲུང་བས་མཛེས། །གང་ཕྱིར་སྐྱེ་བོ་དམ་པའི་མཐུ་ནི་གཞན་ནཾ་པྲ་བྷ་བ། །བྷུ་པ་ཏིཀ་ན་ཀ་བཎྜ་ཨེ་ཥ་ཡསོ་ཧ་མེ་བ་བ་པུ་ཥཱཏྨ་ནཱ་དྷ་ནཾ། །དོན་རབ་སྒྲུབ་བྱེད། །གང་ཞིག་ས་ཡི་བདག་པོ་གསེར་མདོག་དེ། །ད་ལྟ་ང་ཉིད་སྐུ་ཡི་བདག་ཉིད་འདི། ཨི་ཏུ་དཱིརྨ་བྷ་ག་བཱཎྜི་ནས་ཏྭཾ་དྷཱི་མ་ཏྭཾ་བྱ་དྷི་ཏ་དྷརྨ་དེ་ཤ་ནཱཾ། །ཨི་ཏི་དེ་སྐད་གསུངས་ནས་བཅོམ་ལྡན་རྒྱལ་བ་ཡི། །བློ་ལྡན་དམ་པ་རྣམས་ལ་ཆོས་བསྟན་མཛད། །ཅེས་པ་ཀྵེ་མེནྡ་བི་ར་ཙི་ཏཱ་ཡཾ་བྷོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་ཀ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ན་ཀཱ་བ་དཱ་ནཾ་ཏྲི་ཙ་ཏཱ་རི་ཤཔལ་ཝ། །།།གསེར་མདོག་གི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་བཞི་བཅུ་རྩ་གསུམ་པའོ། །།།་པ་རོ་བ་ཀཱ་ར་པྲ་ཎ་ཡི་པྲ་བྷཱ་བསཟོ་པ་ཛཱི་བྱ་མ་ཧ་ཏཱི་བི་བྷཱུ་ཏི། །པུ་ཧཱུཾ་གཞན་ལ་ཕན་པ་འཇེབས་པའི་མཐུ་དང་ནི། །ཀུན་གྱི་ཉེར་འཚོ་འབྱོར་བ་ཆེན་པོ་སྟེ། །འབྲས་བུ་ཀུ་རོ་ཧ་སྱ་བྷཱ་ཏྭཾ་བི་ཤཱ་ལ་ཕ་ལཱདྷ་མེ་ཏཏྲ་ཐ་མཾ་ཧི་པུཥྚམ། །ཛི་ན་ཛེ་ཏ་ཆེར་འོས་འདི་དག་དགེ་རྣམས་ཀྱི། །བསོད་ནམས་མྱུ་གུའི་འབྲས་བུ་དང་པོ་ཡིན། །རྒྱལ་བྱེད་ཚལ་གྱི་བ་ནཱ་རཱ་མ་བི་ཧཱ་རཱ་བྷི་མེ་ཏ་པུ་རཱ། །ཤཱ་བཙྩོ་དེ་བ་སེ་ནཱ་ཁྱཤྲཱི་མཱ་ཀུན་རར་སྔོན། །རྒྱལ་བ་བཞུགས་ཤིང་སྔོན་དགྱེས་ཚེ། །དཔལ་ལྡན་ཁྱིམ་བདག་ལྷ་ཡི་སྡེ༄༅། །ཞེས་།
