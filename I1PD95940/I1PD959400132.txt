བསྟན་འགྱུར། འཁྲི་ཤིང་། ཀེབ་རཱ་སྣ་ཀཱ་ཎཾ་ཏ་བ། །ཤུདྡྷོ་ད་ནཀ་རོ་ཏྲེ་བ་རཱ་ཛ་བྷ་ག་བ་ཏོནྟི་ཀེ། །ཚིག་གིས་ཕྱི་དྲོ་ཁྱོད་ཀྱི་སྐབས། །བཅོམ་ལྡན་འདས་ཀྱི་དྲུང་དུ་ནི། །རྒྱལ་པོ་ཟས་གཙང་བྱེད་པ་ཉིད། །ཨི་ཏི་པ་ཏཱུཔྲི་ཡ་གི་རཱ་ཤྲཱི་ཏཱ་ཤཱཀཾ་ག་ནཱ་ག་ཎི། །ཡཱ་ཙ་ན་ཏད་བྷ་ག་བ་ཏདེ་སྐད་བདག་པོའི་ཚིག་སྙན་གྱིས། །ཤཱཀྱའི་བུ་མོའི་ཚོགས་ཀྱིས་བསྐོར། །བཅོམ་ལྡན་འདས་ཀྱི་བསོད་ས་པུཙྩོ་པ་བ་ནཾ་ཡ་ཡོ། །སཱ་སཏྭཱ་ཀུ་སུ་མཾ་ཏ་ཏྲ་ཏཾ་ད་ནམས་ནགས། །གཙང་མ་དེར་ནི་དེ་སོང་གྱུར། །དེར་ནི་སྙིང་རྗེའི་®དཔག་བསམ་ཤིང་། །སྙིང་སྟོབས་དན་མ་ཧཱ་ཕ་ལཾ། །པྲ་ཤ་མཱ་མྲྀ་ཏ་སཾ་སིཀྟི་ཀ་རུ་ཎཱ་ཀལྤ་པཱ་ད་པཾ། །ལ་ཏེ་བ་མེ་ཏོག་འབྲས་བུ་ཆེ། །རབ་ཏུ་ཞི་བའི་བདུད་རྩི་ཡིས། །ཡང་དག་བྲན་པ་དེ་ཡིས་མཐོང་། །རླུང་གིས་དུད་པ་བ་ནཱ་ན་མཱ་ཏཾ་དྷུ་རཱ་གྲེ་ན་ནཱ་མ་སཱ། །ལོ་ཧེ་ནེ་བ་པ་རི་ཏྱཱཀྐཱཙཱུ་ཏ་པའི་འཁྲི་ཤིང་བཞིན། །དེ་ཡིས་རིང་ནས་དེར་ཕྱག་འཚལ། །རྣ་རྒྱན་གཡོ་བའི་ཟོལ་གྱིས་ནི། །རྨོངས་པ་ཀཎཏྲ་ལཱཙྪ་ལཱཏ། །བི་ལོ་ཀྱ་ཀཾ་ཙ་ན་རུ་ཙི་རཏྣ་བྷུ་ཥ་ཎ་བྷུ་ཥི་ཏྭཾ། །ཡོངས་སུ་གཏོང་བ་བཞིན། །གསེར་འོད་རིན་ཆེན་རྒྱན་རྣམས་ཀྱིས། །བརྒྱན་པ་དག་ནི་མཐོང་གྱུར་ནས། །ཨཱ་ནནྡ་ནཱ་མཱ་བྷིཀྵལྟ་མ་བ་ཙ་པྲ་ཧི་ཏོརྨ་ཏ། །བེ་ཤཔྲ་ཤ་མ་ཤུ་རྩོ་ཡཾ་དགེ་སློང་ཀུན་དགའ་བོ་ཞེས་པས། །ཤིན་ཏུ་ཕན་བརྩོན་དེ་ལ་སྨྲས། །མ་མ་ཁྱོད་ཀྱི་ཆ་ལུགས་འདི། །མ་ཏརྨུ་ནི་ཏ་ཕོ་བ་ནེ། །དཟོཏཱཀྟ་ན་ཡུཀྟསྟེ་བི་རཀྟ་ནཱམི་དྲེགས་པས་རྒྱགས་ཤིང་རབ་ཞིས་སྟོང་། །ཐུབ་པའི་དཀའ་ཐུབ་ནགས་སུ་ནི། །མི་རིགས་འདི་ནི་ཆགས་དཾ་པ་དཾ། །གུ་ཎ་སཾ་ཡ་མ་སིཀྟ་ནི་མུ་ཁ་རཱ་བྷ་ར་ཎཱ་ནྱ་ཧོ། །ཧེ་ཧཱསྨ་བྲལ་གནས། །ཡོན་ཏན་གྱིས་གདམས་ཆགས་པ་ཡི། །མུ་ཅོར་རྒྱན་རྣམས་ཀྱེ་མ་འདིར། །བདག་ཅག་ཏ་གྲ་ཧ་ཎཾ་ཡུལྟ་མི་ཏཱི་བོ་པ་དི་ཤནྟི་ཏེ། །ཨི་ཏྱཱཀྵ་ཏེ་ན་སཱ་ཏ་ནཱི་བཟུང་བར་མི་འོས་ཞེསི། །ཁྱོད་ལ་ཉེ་བར་འདོམ་པ་བཞིན། །དེ་ཡིས་འདི་བརྗོད་གཟུགས་བཟང་བི་ལ་ཛུ་བི་ན་ཏཱ་ན་ནཱ། །ཨུརླུཙྩ་བྷ་ར་ཎཾ་སཐཾ་པྲ་ཧི་དེའི༈། སྐྱེངས་པའི་གདོང་ནི་དམན་པ་ཡིས། །རྒྱན་རྣམས་ཐམས་ཅད་བཀྲོལ་ནས་ནི། །རང་གི་ཁྱིམ་—།
