ཡ？་ན་ཛྷ་ཏཱ་ཨ་པི། །དཔོ་གྲ་དུརྒྱ་ཧ་གྲྭ་ཧ་དུརྒ་ཧཱ་ཚིགས་པ་སམྱ་ཚོགས་དྲག་རྟག་གཟུང་དཀའ་བ། །འདི་དག་རབ་ཏུ་འབད་པ་ཡིས། །བཟུང་ཡང་ངེས་པར་མི་གནས་ད༈། ཀྵ་ཎཾ་ན་བ་ན་བ་གླེ་ཥ་བི་གླེ་ཥ་པྲ་ཎ་ཡོདྨ་ཏཱ། །ཨ་བཱ་ར་བཱ་སོ༑། འབྱོར་པ་གསར་པ་གསར་པ་ནི། །འདུ་འབྲལ་མཛའ་གཅུགས་རྩོམ་པ་དག། མི་བཟློག་ྋསྨད་ར་ར་མ་ཎི་ར་མ་ཎི་ཡཱ་བི་བྷཱུ་ཏ་ཡ། །ཨི་ཏུཀྟ་པྲ་ཏི་ཥ་དྷོ་པི་འཚོང་མཛེས་པ་བཞིན། །སྐད་ཅིག་དགའ་བར་བྱ་བ་ཡིན། །ཞེས་སྨྲས་རབ་ཏུ་བཀག་གྱུར་ཀྱང་། །ས་རཱ་ཛྷཱ་ས་ཙི་བཀྲྀ་ཏ། །ཨ་ཏི་ཀཱ་མ་ཏི་ཀོ་ནཱ་མ་པྲ་བྷ་བིཊྛོས་མཱི་ཧི་ཏཾ། །རྒྱལ་པོས་དེ་ནི་བློན་པོར་བྱས། །ཇོ་བོ་མཆོག་ཏུ་བཞེད་པ་ལ། །ངེས་པར་སུ་ཞིག་འདའ་བར་བྱེད། །པྲསྟ་པྲྀ་ཌྷ་པ་དཾ་རཱ་ཛྷཱ་ནཱི་ཏཾ་སནྡྷ་དྷི་ཀཱ་རི་ཏྭཾ། །བི་དྭེ་ཥ་དུ་ཥི་ཏཱ༔སརྦ་རྒྱལ་པོས་ཀུན་གྱི་གཉེར་གཏད་ཅིང་། །ཤིན་ཏུ་ཆེ་བའི་གནས་ཐོབ་སྟེ། །རྣམ་པར་སྣང་བས་སུན་འབྱིན་མནྟི་ཎ་ན༈་སི་ཧི་རེ། །དྷརྨ་ཛི་ཛྙཱ་ས་ཡཱ་རཱ་ཛྙཱ་པི་ཤུ་ན་པྲ་པའི། །བློན་པོ་རྣམས་ཀྱིས་བཟོད་མ་གྱུར། །ཕྲ་མ་ཅན་གྱིས་བསྐུལ་བ་ཡི། །རྒྱལ་པོས་ཆོས་ནི་བརྟག་རི་ཏེ་ན་མཿ། །ནི་པཱ་ཀ་རོ་ད་སཏྟཱརྱ་ནི་ཡུཀྟ་པི་པུ་ནཔུ་ན། །ཨཱ་ག་ཏཾ་པྲ་ཀྱི་འདོད་ནས། །དམ་པའི་བྱ་བ་མིན་པའི་ལས། །ཡང་ཡང་བསྐོས་ཀྱང་དེས་མ་བྱས། །དེ་ནས་མི་བདག་ཁྲོ་ཏསོཐ་མི་བྱཱ་ཀོ་བ་ན་བྷུ་བྷུ་ཛཱ། །ཏ་ཐཱ་པ་དྷརྨ་སཾ་ཡུཏྟཾ་ན་ཤཱ་ས་རྫུན་གྱིས། །དེ་ནི་གསོད་དུ་བསྐུལ་བར་གྱུར། །དེ་ལྟ་ན་ཡང་ཆོས་མིན་དང༌། །ཡང་དག་ལྡན་པའི་ན་མ་ནྱ་ཏཱ། །ཨེ་ཀ་ཛརྨ་སུ་ཁཱ་ཡི་བ་བ་ཧུ་ཛརྨ་ཤ་ཏཱཇི་ཏུཾ། །ན་སཱ་དྷུ་བཀའ་མ་མནོས། །སྐྱེ་བ་གཅིག་ཉིད་བདེ་བའི་སླད། །སྐྱེ་བ་བརྒྱ་མང་ཉམས་འགྱུར་བ། །སྨད་པའི་ལས་ནིནྡ་ཏཾ་ཀརྨ་ཀ་རོ་མཱི་ཏི་ཛཱ་ག་ད་ས། །བྷ་ཡ་དྷརྨོ་པ་དྷཱ་ཤུདྡྷ༔པྲ་ཏི་ནི་དགེ་བ་བདག། མི་བགྱིད་དོ་ཞེས་དེ་ཡིས་སྨྲས། །ཆོས་ཀྱིས་འཇིགས་ཤིང་ཆོ་འབྲང་གཙང་། །དེ་ནི་ས་མུཀྟས་བྷུ་བྷུ་ཛཱ། །དཱ་ན་ཤ་ཏྲི་མ་བིཙྪིནྡ་མ་ཀ་རོ་ད་ཁི་ལཟི་ནཾ། །སརྦ་བདག་གིས་རབ་བཏང་། །སློང་བ་རྣམས་ནི་མ་ལུས་ལ། །རྒྱུན་ཆད་མེད་པའི་སྦྱིན་ཁང་བྱས། །གྲགས་།
