རམརྫི་ཏོ་དྷ་ཏྭཱ་སརྨི་ཏསྣ་མ་བྷ་ཥ་ཏ། །ཨ་ཡཾ་ནཱ་བྷི་མ་ཏཿཀཱ་ལསྟ་སྱཱསནྡན་མས་དོན་གཉེར། །འཛུམ་ལྡན་དེ་ཡིས་དེར་སྨྲས་པ། །དེ་དང་ཕྲད་པའི་དུས་འདི་ནི། །བདག་ལ་ནེ་མ་མ། ་། ཨ་ཐ་དྷུ་ཏྟཾ་པྲ་ཡཱ་ཏཱ་ཡཾ་སོདྲེ་ཤཱ་ག་ཎི་ཀཱ་བྷ་བཏ། །མངོན་པར་འདོད་པ་མེད། །དེ་ནས་ཕོ་ཉ་མོ་འོངས་ཚེ། །སྨད་འཚོང་ཆགས་བྲལ་ལྡན་པར་གྱུརཚེ། །ནཱ་ནུ་རཱ་གེ་བཱི་རཱ་གེ་བཱ་ནི་ཐ་ཏིཅི་ཥ་ཡོ་ཥི་ཏྭཾ། །ཀ་དཱ་ཙིཏྠནི་རེ་ཏསཿརྗེས་སུ་ཆགས་པས་ཆགས་བྲལ་བ། །སྨད་འཚོང་མ་ལ་ངེས་པ་མེད། །ཚོང་དཔོན་བུ་ནི་ལང་ཚོ་ལྡན། །སྦི་ཏེ་ཡུ་ནི་བ་ནིཀླུ་ཏེ། །ན་བ་སཱརྦ་པ་ཏིཤྲཱི་མཱ་ནཱ་ཛ་གཱ་མོཏྟ་ར་པ་ནམ་ཞིག་དེ་ཡི་ཁྱིམ་གནས་ཚེ། །དཔལ་དང་ལྡན་པའི་དོན་མཐུན་བདག། བྱང་གི་ལམ་ནས་གསར་དུ་ཐཱཏ༑། བི་ཀྲྀཀྵེ་རཱ་ཏྲི་བྷོ་ག་ཡི་ཏེ་ན་ཧེམི་པ་ཧཾ་ཤུ་ཀི། །ཛ་ན་རྩྭ་འོངས། །དེ་ཡི་མཚན་མོ་ལོངས་སྤྱད་སླད། །གོས་དང་བཅས་པའི་གསེར་དག་སྤྲིངས། །བརྐམ་ཆགས་པ་ཏི་ཏཱ་ལུབྷྱཱ་ག་ཎི་ཀཱ་ས་མ་ཙིནྟ་ཡེཏྲ། །ཨེ་ཥ་ཏཱ་བཥི་ཏོ་གེ་ཧེ་ལྡན་པའི་སྨད་འཚོང་མ། །མ་དང་བཅས་པས་སྤྱིརབ་བསམས་པ། །ནོར་ཟད་བྱས་པ་ཚོང་པའི་བུ། །བ་ཎི་སཱུ་ནུཀྲྀ་ཏ་བྱ་ཡ། །ཨནྡྲི་མ་ཧཱ་དྷ་ནཌྷ་ནྱཀིཾ་ཀ་རོ་མི་ན་བེརྨི་ཏཏྲ། །འདི་ནི་རེ་ཞིག་ཁྱིམ་ན་གནས། །གཞན་ཡང་དོན་གཉེར་ནོར་ཆེན་ལྡན། །བདག་གིས་ཅི་བྱ་དེ་མ་རིག། བཱནྟ་བིཏྟཔུ་ནཀཱ་མཱི་ན་བྷ་བ་ཏྱ་དྷི་ཀ་པྲ་ད། །ཏེ་ན་པརླུ་ཥི་ཏེ་ནི་འདོད་ལྡན་ནོར་ནི་སྐྱུགས་པ་དག། ཕྱི་ནས་ལྷག་པར་སྟེར་མི་འགྱུར། །འབྲས་བུ་མེད་པར་ཡོངས་རྙིངས་བ་ཀྲི་ཡ་ཏེ་ནིཤྩ་ལེ་ན་ཀིཾ། །ན་བསྣ་བི་ན་བོདྷུ་ཀྵཀྵཧྥ་མུ་ཙ་ཏྱ་པར། །གྱུར་པ་དེ་ཡིས་ཅི་ཞིག་བྱ། །།གསར་པ་མངོན་འདོད་གསར་པ་ལས། །མ་བསླངས་པར་ཡང་ཡཱ་ཙི་ཏ། །ཨ་པྲ་ཡེ་པི་པྲ་ཡཱ་སཱ་དཾ་ཀ་རོ་ཏི་པྲ་ཐ་མཱ་ད་ར། །ཏ་སྣ་ཐམས་ཅད་སྟེར། །དང་པོར་གུས་པས་མི་མཛའ་ཡང་། །མཛའ་བ་ཉམས་སུ་མྱོང་བར་བྱེད། །དེ་སླད་ཏེ་མ་སྱ་ཀྲི་ཡ་ཏཱ་མནྟ༔པཀྟ་སྱ་ཤ་ལུ་བཏྲ། །ཨ་བྷོ་གེ་ན་ན་ཡཱ་ཏེ་ཥ་ཀརྨ་ནང་དུ་ཟུག་རྔུ་བཞིན། །ཞུགས་པ་འདི་ལ་ཅི་ཞིག་བྱ། །ལས་ཀྱིས་འཆིང་བཞིན་རྗེས་འབྲང་འདི། །མ་།
