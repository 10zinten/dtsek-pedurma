ཏྱཱཀྟ་ཏྲནྠ་མ་མ་ཏྱ་པྲ་ཏསྦེ་སྦི་ར་ནིཉྩ་ཡ། །མ་ཧཱ་ཀཾབྷི་ཏི་མཱ་ལམ་བརྗོད་དེ་དག་ལ་བཟླས་ནས། །སྙིང་སྟོབས་ལྡན་པ་རྒྱལ་པོའི་བུ། །ཆེན་པོ་བརྟན་པ་ལ་བརྟེན་ནས། །བྱ་ས་ཏཱ་བཱནྟཱཟི་བཱཏྨ་ཛ། །གུལླ་མཱ་ཏྲེ་ཎ་སབྷ་ཧཾ་ས་ཏྭཱ་ཛ་མ་བརྞ་ནཱ། །ངེས་པ་བརྟན་པས་རབ་ཏུ་ཞུགས། །དེ་ཡི་ཆུ་ཡི་ལམ་དག་ཏུ། །ལོང་བུ་ཙམ་ལ་ཉིན་བདུན་ཕྱིན། །ཛཱ་ནུ་དངྒེ་ན་སམྦྷ་ཧཾ་སབྷཱ་ཧཾ་པོ་རུ་ཥ་ཎ་ཙ། །ཙ་ཏྭཱ་རི་སཥ་དཱ་ཏཱ་ཎི་པུས་ནུབ་ཙམ་ལ་ཉིན་བདུན་དང་། །སྐྱེས་བུ་ཙམ་ལ་ཉིན་བདུན་ཏེ། །དེ་ནས་ཞག་ནི་བདུན་པ་བཞི། །ཏ་པུཥ་རི་ཎཱི་ཛ་ལི། །ག་ཏྭཱ་ཏྲཏྟི་བི་ཥཱཉྙོ་རཱན་ད་དནཤྲཱི་བི་ཥཱརྫུ་གླང་མོའི་ཚད་ཀྱི་ཆུ་དག་ལ། །ཕྱིན་པས་མདུན་དུ་སོ་དུག་ཅན། །འཇིགས་རུང་མཐོང་བ་དུག་ཅན་རཿ༑། མི་ཏི་ཡུལྟེ་ན་མ་ན་སཱ་ཀྲྀ་ཏཱ་ཏཱ་ན་ཐ་ནིརྫི་ཥན། །ཀཱུ་ར་ཀོ་སེཛྲ་མཐོང་། །དེ་ནས་བྱམས་ལྡན་ཡིད་ཀྱིས་ནི། །དེ་དག་དུག་མེད་བྱས་ནས་དེས། །ཁྲོ་བས་འཁྲུགས་པའི་ཏཾ་ཡགྱིརྨ་དཱི་པ་མ་བོ་ས་ས། །ཏ་ཏྲ་མི་ཏྲེ་ཎ་མ་ནཱ་སཱ་བཱི་ཏ་ཀྲོ་གནོད་སྦྱིན་གྱིས། །གང་བའི་གནོད་སྦྱིན་གླིང་དུ་ཕྱིན། །དེར་ནི་བྱམས་ལྡན་ཡིད་དག་གིས། །དེ་དག་ཁྲོ་དྷཱནྡྲ་དྷཱ་ཡ་ཏཱན། །ཤུ་ཕྲ་བ་ཏི་རུ་ཙི་ཀཱི་ཏཾ་བི་སུ་ལོན་ཧ་བིསྨི་ཏི། །ཀུ་མཱ་དང་བྲལ་བར་བསྒྲུབས། །སྤྲོ་བ་ཡངས་པས་ཡ་མཚན་པ། །དེ་དག་གིས་བཤད་ཐོས་གྱུར་ཏེ། །གཞོན་ནུ་ར་སྱ་སཱི་ཏ་ས་ཏྲེ་ན་བཱིརྨོ་ཧ་ན་ཙཱ་མུ་ནཱ། །ན་ག་རཱ་ཛ་སྱ་བྷ་བ་ན་ས་མཱ་ཧི་སྙིང་སྟོབས་རྒྱས་པ་དང༌། །བརྩོན་འགྲུས་སྤྲོ་བ་འདི་ཡིས་ནི། །ཀླུ་ཡི་རྒྱལ་པོའི་ཁང་པ་རུ། །ཁྱོད་ཉིད་ཏ་མ་བཱསྱ་ཏི། །ཀཱ་ལེ་ན་སམྱཀཾ་བུདྡྷསཧྥ་ཛྷཧཱུྃ་བྷ་བི་ཥྱ་བཞེད་པ་འཐོབ་པར་འགྱུར། །དུས་ཀྱིས་ཁྱོད་ནི་ཐམས་ཅད་མཁྱེན། །ཡང་དག་རྫོགས་པའི་སངས་རྒྱས་སི༑། ཤཱ་བ་ཀཱཉྩ་བྷ་བི་ཥྱ་མོ་བ་ཡཾ་ཏཱ་ད་ནུ་ཡཱ་ཡི་ན། །པྲ་ས་འགྱུར། །བདག་ཅག་རྣམས་ཀྱང་ཁྱོད་ཉིད་ཀྱི། །རྗེས་འབྲང་ཉན་ཐོས་དག་ཏུ་འགྱུར། །འདི་དག་བི་རི་ཏི་ཏི་རུཀྟ་མ་བྷི་ནནྡྷཱ་ནཱཾ་པཱཏྨ་ཛ། །རཀྟིརྒྱུ་རཱ་བྲྀ་ཏཾ་པྲ་ས་དང་བས་འདི་བརྗོད་དེ། །མངོན་པར་དགའ་བའི་མི་བདག་བུ། །སྲིན་པོ་རྣམས་ཀྱིས་ཁེངས་གྱུར་པ། །—།
