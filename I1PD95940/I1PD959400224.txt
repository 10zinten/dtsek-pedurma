ཡོ་ས་མ༔སུ་ཏ། །ཨི་ཏི་ཡག་བ་དྷུ་བཱ་ཀྱི་ཐུ་ཏྭཱ་བཏྲ་ལྱཱ་བི་ཧྭ་ལཾ། །ཟླ་མི་མཚུངས། །དེ་སྐད་གནོད་སྦྱིན་ཆུང་མིའི་ཚིག། མཉེས་གཤིན་གྱིས་འཁྲུགས་གསན་གྱུར་ནས། །བྷཱུ་ཏཱ་ནུ་ཀམཱི་བྷ་ག་བཱན་ས་སྨི་ཏསྟ་ཡ་བྷཱ་ཥ་ཏ། །ཤོ་ཀོ་ཡཾ་བ་ཧུ་ཏཱ་ཡཱ་བཅོམ་ལྡན་འབྱུང་པོ་ལ་བརྩེ་བས། །འཛུམ་བཅས་དེ་ལ་བཀའ་སྩལ་པ། །ཁྱོད་ཀྱི་བུ་ནི་མང་པོ་ལས། །ཡ་བྱེ་ཀཱ་བི་ར་ཧེ་ཏ་བ། །བྲྀ་ཏེ་ཏཱ་ཡི་ཀ་བཏྠཱ་ནཾ་པུ་ཏྲ་ཀེ་ཀིཾ་ཛཾ་ཤྲཱི་བྱ་ཐཱ། །གལ་ཏེ་གཅིག་བྲལ་མྱ་ངན་འདི། །བུ་གཅིག་རྣམས་ཀྱི་བུ་ཁྱོད་ཀྱིས། །ཕྲོགས་པའི་གདུང་བ་ཅི་ཞིག་འདྲ། །ཏྭཾ་པྲ་བི་ཤུ་ས་དཱ་གེ་ཧཾ་སྡེ་ནཾ་པུ་ཏྲ་ནཱ་ལཀྵི་ཏཱ། །ཨ་ཤ་ནཱ་སི་པུ་ཏྲ་མཱ་ཏཱ་པི་བྱཱ་རྟག་ཏུ་ཁྱིམ་དུ་མ་མཐོང་བར། །ཁྱོད་ཞུགས་བུད་མེད་རྣམས་ཀྱི་བུ། །བུ་ཡི་མས་ཀྱང་སྟག་མོ་ཡིས། །རི་རྨི་བ་མྲྀ་ག་ཤཱ་བ་ཀཱན། །ཡེ་ན་ཡེ་ན་སཱ་དེ་ཧ་སྱ་དུཁཾ་ཡཱ་ཏུས་བྷོ་ག་ཏྭཾ། །དྭགས་རྣམས་ཀྱི་བུ་བཞིན་ཟོས། །རང་གི་ལུས་ལ་གང་གང་གིས། །སྡུག་བསྔལ་ཉེར་སྤྱོད་ཉིད་འགྱུར་བ། །ན་ཏཏ་པ་ར་སྱ་ཀུཎྜི་ཏ་སཾ་མཱ་ནརྣོ་བྷ་བ་ཤུ་ཙཱ། །ཏྭཾ་བུདྡྷ་དྷརྨ་སཾ་གྷཱ་ནཾ་ཏཱི་མྱ་ངན་ཉམས་མྱོང་མཚུངས་པས་ན། །དེ་ནི་གཞན་ལ་བྱ་བ་མིན། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་གྱི༄༅། །ཎི་ཤི་གྷ་ས་དཱ་ནི་ཙེ་ཏ། །གཽདྡྷཱ་སི་ཧི་སཱ་བི་མུ་ཁཱི་ཏཏྟཱམོ་སི་པྲི་ཡཾ་སུ་བསླབ་པའི་གཞི་གསུམ་གལ་ཏེ་ཁྱོད། །འཛིན་ཅིང་འཚེ་ལས་ཕྱིར་ལྡོག་ན། །དགའ་བའི་བུ་དེ་ཁྱོད་ཀྱིས་ཏཾ༑། ཨི་ཏྱཱཀྵ་སཱ་བྷ་ག་བ་ཏཱ་པྲསྦི་ཤིཀྵ་པ་དཱ་ཏ་ཏ། །ཧི་སཱ་བི་འཐོབ། །དེ་སྐད་བཅོམ་ལྡན་གྱིས་བཀའ་སྩལ། །དེ་ཡིས་བསླབ་གཞི་བཟུང་དེ་ནས། །འཚེ་བ་སྤངས་ར་མཱཏྟངྒནྟཾ་པུ་ཏྲ་པྲ་པ་སྲི་ཡཾ་ཀ་རཾ། །ཏཏྟཱསྔརྨ་བྲྀཏྟཱནྣཱནྟཾ་ཏ་སྱཀརྨ་ལས་དགའ་བྱེད་བུ། །དེ་ཡིས་བྱིན་ནས་ཐོབ་པར་གྱུར། །དེ་ཡི་སྐྱེ་བ་སྔ་མ་ཡི། །སྤྱོད་མཐའ་ཡས་ཀྱི་ཕ་ལཱ་ནཱ་ཡཾ། །བྷི་བྷིནྡྷ་ག་བཱན་པྲཋསྟཛྲཏྟཱནྟ་མ་བྷཱ་ཥ་ཏ། །པུ་རཱསྨིནྡྷེ། རྗེས་འགྲོ་དེ། །དགེ་སློང་གིས་དྲིས་བཅོམ་ལྡན་གྱིས། །སྤྱོད་པའི་མཐའ་དེ་བཀའ་སྩལ་པ། །སྔོན་ཚེ་གྲོང་བ་ན་ག་རེ་པོ་རཀེ་ཕྱུ་པ་བྷོ་གི་ན། །སརྦ་ཏོདྷཱ་ན་ཡཱ་ཏཱ་ཡཾ་བི་ཛ་ཁྱེར་འདི་ཉིད་དུ། །གྲོང་པ་ཁ་ཅིག་ལོངས་སྤྱོད་ཅན། །རི་དང་སྐྱེད་མོས་ཚལ་འགྲོ་ཞིང་། །གར་དང་།
