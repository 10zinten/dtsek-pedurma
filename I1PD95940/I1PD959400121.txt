ཛཱ་ཡ་ཏེ་ཛཱི་བི་ཏ་བྱ་ཡ། །ཨི་ཏི་ཙིནྟཱ་ཀུ་ལཾ་ཏ་ཏྲ་ཏཾ་ཛྙཱ་ཏཱ་ས་ཏྭཱ་སཱ་སྐྱེ་འགྲོ་རྣམས་སུ་མཆོད་པར་འོས། །དེར་ནི་དེ་ལྟར་བསམ་འཁྲུགས་ཏེ། །སྙིང་སྟོབས་རྒྱ་མཚོར་ཤེས་ག་རཾ། །ནི་ལོ་ནཱ་མ་ག་མ་བྷེ་ཏྱཱ་ཡཀྵ་སྭ་ཧཱ་ཙ་ལཱ་ཤ་ཡ། །པཱུརྦ་ཎ་ཡོ་གྱུར་ནས། །རི་ལ་གནས་པའི་གནོད་སྦྱིན་ནི། །སྔོན་པོ་ཞེས་པ་འོངས་ནས་སྨྲས། །ཤར་དུ་དཔག་ཚད་ཛ་ནཾ་ག་ཏཱ་ཏཱི་ཎི་བྷཾ་གཱ་ནི་བྷཱ་ཤྲཱི་ཏ། །བེ་ཏྲ་སོ་པཱ་ན་ནི་ཐེ་ཙྪཱ་ཤཱ་མཱ་ཏུ་རྩྭ་ཕྱིན་གྱུར་ནས། །ས་འཛིན་རྩེ་མོ་གསུམ་པ་ནི། །ས་ལས་བྱས་པའི་ཐེམ་སྐས་ལ། །འཛེགས་ལ་དེ་ནས་ཐ་ག་མྱ་ཏྭཾ། །ཨི་ཏི་ཡཀྵ་བ་དེ་ཤེ་ན་ས་བི་ལཾ་གྲུ་མ་ཧཱ་ཙ་ལཾ། །ད་དན་འགྲོ་བར་གྱིས། །དེ་སྐད་གནོད་སྦྱིན་གདམས་ངག་གིས། །རི་ཆེན་དེ་ཡི་རྣམ་བརྒལ་བས། །རྩེ་མོ་གྲེ་ས་ཡུཏྟཾ་ག་ཤཾ་གཾ་སྤ་ཊི་ཀ་བྷཱུ་དྷ་རཾ། །ཏསྨིནྡྷེ་ཀ་ཤི་ལ་ཐཥྛ་དུརྒ་ཤིན་ཏུ་མཐོ་བ་ཡི། །ཤེལ་གྱི་ས་འཛིན་མདུན་དུ་མཐོང་། །བྱ་རྣམས་ཀྱིས་ཀྱང་བགྲོད་དཀའ་བ། །འཇམ་མེ་པའི་ཎཱ་མ་པི། །མཱུ་ཧཱུཏྟ་མ་བྷ་བཏྟ་སྱ་ནིཅུ་པ་རོ་མ་ནོ་ར་ཐ། །ཨ་བྷུནྣ་ཏཾ་ཤི་པའི་བྲག་ནི་གཅིག་པུ་དེར། །དེ་ཡི་ཡིད་ལ་རེ་བ་ནི། །ཡུད་ཙམ་བྱ་བ་མེད་པར་གྱུར། །ཤིན་ཏུ་མཐོ་ཞིང་རཱ་ཡཝཾ་སྭ་སཾ་ཀལྤ་ཡི་བཱ་ཙ་ལཾ། །ས་ཏཾ་བི་ཙཱརྱ་སུ་ཙི་རཾ་ཙི་ཏྲ་རྒྱསྟ་རྟེན་དང་བྲལ། །རང་གི་རྟོག་བཞིན་གཡོ་མེད་པར། །དེ་ལ་རྣམ་དཔྱད་ཡུན་རིང་དེ། །རི་མོ་བཀོད་པ་ཨི་བ་བྷ་བཏ། །ཨ་ཐ་ཙནྡྲ་པྲ་བྷོ་ནཱ་མ་ཡཀྵཤི་ལ་གུ་ཧཱ་ཤ་ཡ། །ཨ་བྷེ་ཏྱ་སཏྭཱ་བཞིན་དུ་གྱུར། །དེ་ནས་ཟླ་འོད་ཅེས་པ་ཡི། །གནོད་སྦྱིན་བྲག་གི་ཕུག་གནས་པ། །སྙིང་སྟོབས་ཕུན་སམྤནྣཾ་ཏ་མ་བ་ཥ་ཏ་བི་མི་ཏ། །ཀྲོ་ཤ་མཱ་ཏྲ་མི་ཏཱ་ག་ཏཱ་པཱུཟེ་ཚོགས་དེ་ལ་ནི། །མངོན་ཕྱོགས་ཡ་མཚན་ལྡན་པས་སྨྲས། །འདི་ནས་ཤར་དུ་རྒྱང་གྲགས་ཙམ། །སོང་ཎཱ་པཱུརྦ་བི་བྷཱ་མཾ། །ཛྲ་བྱ་ཏེ་ཙནྡ་ན་བ་ནཾ་བཱ་ལཱ་ནི་ལ་ཙ་ལ་སྨྲ་བས་ཙནྡན་ནགས་ཚལ་ནི། །རླུང་བཞོན་གྱིས་བསྐྱོད་འཁྲི་ཤིང་ལྡན། །སྔོན་མེད་རྣམ་འཕྲུལ་མཐོང་བར་ཏཾ༑། ཏ་ཏྲ་ཕྲ་ས་རཱ་ནཱ་མ་གུ་ཧཱ་ལཱི་ནཱ་མ་ཧོ་ཥ་དྷི། །ལ་བྱཱ་ཏེ་དེ་ཧ་ར་འགྱུར། །དེ་ནི་རབ་གསལ་བྱེད་ཅེས་པ། །ཕུག་ན་གཏམས་པའི་སྨན་ཆེན་པོ། །ལུས་ཅན་བསྲུང་ཕྱིར་།
