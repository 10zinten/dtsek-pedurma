མྱོ་ཙི་ཏ་མེ་ཏ་དཾ་གཾ། །པཱ་ཎོ་ཙ་ལི་ལཱ་ཀ་མ་ལཱ་སྱ་དེ་སྨིན་པཱ་ཊཾ་ཀ་ཐཾ་ཏེ་པྲ་ཡ་མ་བྱ་ནི་ཆོས་གོས་ཇི་ལྟར་འོས། །ཁྱོད་ཀྱི་ཕྱག་ནི་པདྨོའི་རོལ་གནས་འདིར། །ད་ལྟ་ལྷུང་བཟེད་དགའ་བ་ཇི་ཛཱ་ཏཾ། །ཨ་ཡཾ་བི་ཧཱ་ར་སྟ་བ་ཀཎྜ་པཱི་ཐསོཥཎཱ་ཀཱནྟ་བྷུ་ཛ་བནྡྷ་ནཱནྟ། །ལྟར་སྐྱེ། །ཁྱོད་ཀྱི་མགྲིན་པ་རྒྱབ་ཀྱི་གནས་འདི་དག། སྲེད་ལྡན་མཛེས་མའི་ལག་པས་བསྡམས་འོས་པ། །སཾ་བྷོ་ག་ལགླུཀྵ་པི་ཏ་པྲ་མོ་དཀ་རོ་ཏྱཱ་ཀཱིཎྜི་པྲ་ཎ་ཡཱ་བ་བྷཾ་གཾ། །ལོངས་སྤྱོད་དཔལ་མོའི་རབ་དགའ་ཉམས་གྱུར་ཅིང་། །གློ་བུར་ཉིད་དུ་མཛའ་གཅུགས་ཉམས་པར་བྱེད། །རཱུ་པཾ་བི་ལགྷི་ཀྲྀ་ཏེ་པུà་ཙཱ་པོ་མལྟེ་བྷ་ཀུནྡྷོཙྩ་ཀུ་ཙཱ་བི་བྷུ་ཏི། །ར་མེ་ཏོག་གཞུ་ལྡན་སྐྱེངས་པར་བྱེད་པའི་གཟུགས། །གླང་ཆེན་མྱོས་བུམ་ལྟར་མཐོའི་ནུ་རྒྱས་འབྱོར། །ལང་ཏེཇི་ལཱ་སོ་པ་བ་ནཾ་བ་ཡཉྩ་ཀེ་ནཱ་ས་མསྟེ་ན་ཀ་ལི་ཏོ་བི་རཱ་ག། །ཐུ་ཏྭེ་ཏི་ཚོ་དགའ་བའི་རྩེ་ས་ཉེ་བའི་ནགས། །ཅི་སླད་ཁྱོད་ཀྱི་མི་བཟད་ཆགས་བྲལ་བྱས། །ཞེས་གསན་ཚུལ་ཏཾ་ཤཱི་ལ་ནི་དྷིརྦ་བྷཱ་ཥ་ཤ་ཤཱི་ཀ་ལེ་ཁཱ་ལ་ལི་ཏ་སྨི་ཏེ་ན། །སཾ་ཀཱནྟ་ནཱ་ནཱ་ཁྲིམས་གཏེར་གྱིས་བཀའ་སྩལ་ཏེ། །ཟླ་བའི་ཆ་བཞིན་རོལ་པའི་འཛུམ་དག་གིས། །མི་བདག་སྣ་ཚོགས་ཚྭ་པ་རཏྣ་རཱ་གཾ་ཀུརྦནྡ་ལཀྵ་མི་བ་རཱ་ཛ་ལགླི་མ། །རཱ་ཛ་ན་ཛ་རཱ་རོ་ག་ཧ་རིན་ཆེན་འོད་འཕྲོས་པའི། །རྒྱལ་པོའི་དཔལ་ནི་མི་སྣང་འཛད་པ་བཞིན། །རྒྱལ་པོ་ཆུ་རྒྱུན་ལྟར་གཡོ་ཏེ་བ་ན་སྣ་ཏུ་རཾ་ག་ལོ་ལཱ་ཡ་དི་ཛཱི་བ་བྲྀཏྟི། །ཏཥ་སྱ་ན་སཱུ་ད་ནི་ཤཾ་པྲ་ཧནྟ་འཚོ་བའི་སྲོག། གལ་ཏེ་རྒ་དང་ན་བས་མི་འཇོམས་ན། །རབ་ཀྱི་དགའ་བདུད་རྩིའི་ཆར་ལྟའི་ཡུལ་རྣམས་ལ། །པཱི་ཡུ་ཥ་བནཱ་བི་ཥ་ཡཱ་བྷི་ལཱ་ཥ། །ཤ་མཱ་མཾ་ཏཱ་སྭཱ་ད་ན་སུ་སྦི་རཱ་ཎཱ་མ་རྟག་ཏུ་སུ་ཞིག་མངོན་པར་འདོད་མི་བགྱི༄༅། །ཞི་བའི་བདུད་རྩི་ཉམས་མྱོང་བརྟན་རྣམས་ཀྱིས། །ནགས་པ་ཏི་ནི་ཤུ་རྩ་བ་ནཱནྟ་བྷཱུ་མི། །བི་བྷུ་ཏི་ལི་ལཱ་མ་ད་བི་ཧཱ་ལཱ་ནཱཾ་ཧམྱ་ནང་སྟོང་པའི་ས་གཞི་གནོད་མེད་ཡིན། །འབྱོར་པས་རྒྱགས་ཤིང་རྩེ་དགས་འཁྲུགས་རྣམས་ཀྱི། །བ་གམ་ཎི་པརླནྟ་ནི་པཱ་ཏ་ནཱ་ནི། །ས་ཀུ་མ་ཀུ་མིནྣཱནྟི་ཧྲ་པཱས་ཡོ་བྷིས་རཱ་ག་ཏཾ་ཡིལྡན་པ་མཐར་ནི་རྣམ་པར་ལྷུང་། །གང་ཞིག་གུར་ཀུམ་ཎིདང་བཅས་ཆུས་བཀྲུས་པའི། །མི་བདག་རྟག་ཏུ་།
