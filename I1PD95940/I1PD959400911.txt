བྷི་ཀྵ་བྷི༈་ཏ་བིསྨྲ་ཡེ། །པཱཀླུ་ཎྱ་པཉྩ་ཡནེ་ཏཱཾ་པྲཏྟིཔྲ་བཱ་ཙ་ལ། །དགེ་སློང་ཡ་མཚན་སྐྱེས་རྣམས་ཀྱིས། །དེ་དག་སྔོན་ཚེ་བསོད་ནམས་ཚོགས། །དྲིས་པ་ཐམས་ཅད་སརྦ་བི་ཏ། །ཨལྤཤཱ་ས་མ་ཧཱ་ཤྭ་སོ་པུ་ར་ནཱ་གོ་བ་བྷ་བ་ཏུ། །ཏ་དྭེ་རིག་པས་གསུངས། །ཀླུ་ནི་དབུགས་ཆུང་དབུགས་ཆེན་དག། སྔོན་བྱུང་དེ་དག་དགྲ་བོ་ནི། །འདབ་ཥ་ནོ་སུ་པཙྩོ་ཙ་ཙཱུ་ཌིརྨིཊ་སྭ་རསྟ་ཐཱ། །ནཱ་གོ་ཀཱ་བྱ་པཱ་ཤཱསྟ་སཾ་པྲ་བཟང་གཙུག་ལྡན་དེ་བཞིན་དུ། །མི་ཌིའི་དབྱངས་ཞེས་བྱ་བའོ། །ཀླུ་ཡིས་སྟོན་པ་འོད་སྲུངས་ཀྱི༄༅། །སྱ་བྷཱུ་ཏ་ཤ་ས་ནཾ། །སུ་པཙྪཱ་ཡོ་ར་ས་བྷཾ་་བ་བྷ་ཏ་་ལ་ཤཱ་བཀའ་ནི་ཡང་དག་ཐོབ་གྱུར་ནས། །སྟོབས་ལྡན་འདབ་བཟང་དག་གིས་ནི། །བརྫི་བར་ནུས་པ་མིན་་་ལི་ནོ། །དྭེ་ཥ་སྤནྡྷ་བྷི་མཱ་ནེ་ན་སུ་པཙྪཱ་བ་པི་ཀཱ་བྱ་པཱ་ཏ། །གཽ་ཧི་པར་གྱུར། །།ཞེ་སྡང་འགྲན་ཞིང་མངོན་ཁེངས་པ། །འདབ་བཟང་གིས་ཀྱང་འོད་སྲུངས་ལ། །བསྟན་ཏྭཱ་ཤཱ་ས་ནཾ་རཏྣ་ཏྲ་ཡཾ་ཤ་ར་ཎ་མཱ་པ་ཏུ། །ཏོ་ནཱ་གཱ་བཱརྨ་ཛཱ་ཏཱི་ཡོ་ཛཾ་པ་རིན་ཆེན་བཟུང་ནས་ནི། །དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་སོང་། །ཀླུ་དེ་དག་ནི་འཕགས་པའི་རིགས། ཏ་རཱཥཱ་བི་རཱུ་ཌྷ་ཀོ། །ཀཱ་རོ་སུ་པཙྩོ་ཏཱ་པེ་ཏོ་བི་རཱུ་པཱ་ཀྵ་དྷ་ནཱ་ཡུལ་འཁོར་སྲུང་དང་འཕགས་སྐྱེས་པོ། །གདུག་པའི་འདབ་བཟང་དེ་དག་ནི། །མིག་མི་བཟང་དང་ནོར་དྷི་པོ། །ཇི་ན་ཀྱེ་ཏི་བྲུ་བ་ཎ་སྱ་པུ་ཙྩཾ་པྲ་བ་ཙ་ནཱ་མྲྀ་ཏཾ། །ཏདྨ་ནོ་པཱནྟ་བདག་འདི། །ཞེས་པ་གསུང་བ་རྒྱལ་བ་ཡི། །བསོད་ནམས་གསུང་རབ་བདུད་རྩི་ནི། །ནགས་ཚལ་དེ་ན་ནི་ལ་ཡཐུ་ཏཱ་ཀེ་ནེ་ཡ་ཀོ་མུ་ནི། །བཱིསྨི་ཏ་སྟཏྠ་བྷཱ་བེ་ན་ཏ་མེ་བ་ཤ་ར་ཎཾ་ག་ཏ། གནས་པ་ཡི། །ཐུབ་པ་ཀེ་ནེ་ཡ་ཀས་ཐོས། །དེ་ཡི་མཐུ་ཡིས་ཡ་མཚན་ཏེ། །དེ་ཉིད་ལ་ནི་སྐྱབས་སུ་སོང་། །ཨ་ནཱ་གཱ་མི་ལཱཙྩཱ་བྷཱུཏ་པ་ཏྱ་དན་ན་ནིབྲྀ་ཏ། ་། ཏདྷཱ་གི་ནེ་ཡ༔ཤི་ལོ་ཕྱིར་མི་འོང་ནི་ཐོབ་གྱུར་ནས། །བདེན་པ་མཐོང་ཞིང་དགའ་བར་གྱུར། །དེ་ཡི་ཚ་བོ་ཤི་ལ་ཡང་། །པི་ཏ་ཐི་བ་བི་ན་ཡཾ་ཤྲི་ཏ། །ཨ་ནཱ་གཱ་མི་ཕ་ལོ་དཱ་རཾ་ས་ཏྱ་དན་ན་མཱཔ་བཱན། །དེ་བཞིན་འདུལ་བ་ལ་བརྟེན་ནས། །བདེན་པ་མཐོང་ཞིང་རྒྱ་ཆེ་བ། །།ཕྱིར་མི་འོང་གི་འབྲས་ཐོབ་འགྱུར། །།
