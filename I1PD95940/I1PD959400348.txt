ལ་ལ་ཙྪཏྟཱ་ཀཱ་ར་ཤི་རཱཤི་ཤུ། །རཱ་ཛསྣ་ཡཾ་ཏེ་ཏ་ན་ཡ༔ཀྲ་བནྣཱ་བྷ་ཏུ་འབར། །རི་མོ་དུང་དང་ཆུ་སྐྱེས་ཕྱག། རྒྱལ་པོ་ཁྱོད་ཀྱི་སྲས་པོ་འདི། །འཁོར་ལོས་སྒྱུར་བའམ་ཀུན་བིཥྛ་ཏི། །སམྱཀཾ་བོ་དྷི་སཾ་བུདྡྷསརྦ་ཛྙཱ་བཱ་ཛ་གདྷུ་རུ། །ཨི་ཏུཏྟཱ་ཏེ་ཏུ་མཁྱེན་ནི། །ཡང་དག་རྫོགས་པའི་བྱང་ཆུབ་ཏུ། །སངས་རྒྱས་འགྲོ་བའི་བླ་མར་འགྱུར། །ཞེས་བརྗོད་དེ་ཛཱ་ཏེ་ཥཱ་ལེ་བྷེ་ཧནྟཾ་མ་ཧཱི་པ་ཏི། །སཥ་བྷིརྗེ་བ་སེཤཱyསྔ་ན་ནཱི་ཏྲི་དག་སོང་བའི་ཚེ། །ས་ཡི་བདག་པོས་དགའ་བ་ཐོབ། །ཉིན་ཞག་བདུན་ན་སྟོན་པ་ཡི། །ཡུམ་ནི་མཐོ་དི་པཾ་ཡ་ཡོ། །ཏ་སྱ་ཛརྨ་ནི་ཤཱཀྱ་ནཱཾ་མུ་ནཱི་ནཱ་མི་བ་ཤཱནྟ་ཏཱ། །དྲྀཥྛ་ཡ་དཱ་རིས་དག་ཏུ་སོང་། །གང་ཚེ་དེ་བལྟམས་ཤཱཀྱ་རྣམས། །ཐུབ་པ་བཞིན་དུ་ཞི་བ་ཉིད། །མཐོང་བ་དེ་ཚེ་ཤཱཀྱ་མུ་ནིནྣཱ་མཱ་བྷུཏྠ་ཏ་དཱ་ཤི་ཤུ། །དེ་བཱ་ནཱ་མ་པི་དེ་བོ་ཡ་མི་ཏི་ནིཉྩི་གཞོན་ནུ་དེ། །ཤཱཀྱ་ཐུབ་པ་ཞེས་པར་གྱུར། །འདི་ནི་ལྷ་རྣམས་དག་གིས་མིཡང་། །ལྷ་ཞེས་གཟི་བརྗིད་ཏྱཱ་ཏེ་ཛ་སཱ། །དེ་བཱ་ཏི་དེ་བ་ཨི་ཏྱཱ་སྱ་ནཱ་མ་ཙཀྲེ་མ་ཧཱི་པ་ཏི། །ནཱ་ར་དེ་ནཱ་ཐ་དག་གིས་ངེས། །ས་ཡི་བདག་པོས་དེ་ཡི་མཚན། །ལྷ་ཡི་ལྷ་ཞེས་བཏགས་པར་གྱུར། །དེ་ནས་མི་སྦྱིན་ས་ཧི་ཏསྟ་ཏྲ་དནྟི་ཏ་བོ་བ་ནཱཏ། །ཏཾ་ས་མ་བྱ་ཡ་ཡོ་དྲཤྩ་མ་ད་ར་དང་འགྲོགས་ཏེ། །ཐུབ་པ་ནག་པོ་དེ་ཉིད་མཐོང་། །གུས་པས་དཀའ་ཐུབ་ནགས་ཚལ་ནས། །དེ་ནི་བལྟ་དཱ་སི་ཏོ་མུ་ནི། །ས་བོ་དྷི་སཏྭཾ་བཱ་ལཱརྐ་མི་བ་ཀལྤ་པྲ་ཀཱ་ཤ་ཀཾ། །རུ་ཡང་དག་འོངས། །བྱང་ཆུབ་སེམས་དཔའ་སྐལ་བ་ནི། །གསལ་བྱེད་ཉི་མ་འཆར་ཁ་བཞིན། །དྲྀཟླ་བི་ཀཱ་ཤི་བཀྵ་བྷིརླ་བྷེ་ཀ་མ་ལ་ཏུ་མྱ་ཏྭཾ། །སོ་བྲ་བཱིདྲི་ཧི་ཏ་ཏི་ཤྲཱི་གྲོགས་མཐོང་ནས་བཞིན་གྱི་དཔལ་རྒྱས་པ། །པདྨོ་ཉིད་དང་མཚུངས་པ་ཐོབ། །ཕྱག་དང་མགྲོན་ཆ་བྱས་པ་ཡིས། །བ་ཏི་པྲ་ཎི་ཏཾ་མུ་ནི། །རཱ་ཛཙྰ་ཎ་ག་ཎེ་ནི་བ་སྤྲོ་ཧ་ཎཱི་ཡོ་པི་པཱུ་ནུ་རྒྱལ་བ་ཐུབ་པ་དེས་སྨྲས་པ། །རྒྱལ་པོ་ཡོན་ཏན་ཚོགས་ཀྱིས་བཞིན། །སྲས་ཀྱི་ཁྱོད་ནི་དགྱེས་པར་ནཱ༑། ཨེ་ཏཱ་ནི་ལཀྵ་ཎཱ་ནྱ་སྱ་མོཀྵ་ལགླི་ས་མཱ་ག་མཾ། །བ་དནྟི་འོས། །འདི་ཡི་མཚན་རྟགས་འདི་དག་ནི། །ཐར་པའི་དཔལ་འབྱོར་ཕུན་ཚོགས་སམ། །འཁོར་ལོས་།
