་བཱ་ཤ་བཱ་ཁྱ་སྱ་བྷཱུ་པ་ཏེ། །སུ་ཧྲི་དཔྲ་ཧི་ཎོཇུ་ཏཾ་བྷ་ག་བཱནྡྲ་བྱ་ཏཱ་མི་ཏི། ་། ས་བདག་ནོར་ལྷའི་བུ་ཞེས་པའི། །གྲོགས་ལ་བཅོམ་ལྡན་ལྟོས་ཤིག་ཅེས། །རྒྱལ་པོ་དེ་ཡི་ཕོ་ཉ་སྤྲིངས། །ཨ་ཏཱནྟ་རེ་པཱ་ས་བེ་ན་རཱ་ཛྙཱ་དྭཱ་ད་ཤ་བཱནྟི་ཀེ། །ཡཛྷ་ས་མཱསྟེ་པི་དྷི་སྐབས་དེར་རྒྱལ་པོ་ནོར་ལྷའི་བིུ། །མཆོད་སྦྱིན་ལོ་ནི་བཅུ་གཉིས་པ། །ཆོ་ག་བཞིན་དུ་རྫོགས་པ་ན། །བཏྟམྦི་ཏཱཔཉྩ་དཀྵི་ཎཾ། །ཏནེ་བ་ཏཱ་བ་དདྡྷཱ་པཾ་བི་པྲ་ཏཱ་མཱ་ག་མི་ཥྛ་ཏ། །ཡོན་ལྔ་དག་གིས་བརྟག་པ་བྱས། །ས་བདག་དེ་ལ་ལྷས་སྨྲས་པ། །ཁྱོད་ཅན་བྲམ་ཟེ་གཉིས་འོང་འགྱུར། །སུ་མ་ཏིཉྩ་མ་ཏི་ནཱ་མ་དཱ་ནཱན་པྲ་ཐ་མལྟ་ཡོ། །ཨ་ཐཱ་ཏྱཱ་ཡ་ན་བརླནྟེ་གུ་ར་བློ་གྲོས་བཟང་དང་བློ་གྲོས་སོ། །དེ་ལས་དང་པོ་ལ་སྦྱིན་འོས། །དེ་ནས་ཁ་ཏོན་མཐར་ཕྱིན་ཏེ། །བླ་མའི་བེ་དཀྵི་ཎཱརྦ་ནོ། །ཡངྷ་ཀྵེ་ཏྲ་ཀྵི་ཏི་པ་ཏེསྟོ་ས་མ་ཛམྦྷཱ་ཏུནྡྲ་ཛོ། །།བཱ་ས་ཡོན་ནི་དོན་གཉེར་བའི། །བྲམ་ཟེ་དེ་དག་ས་བདག་གི། །མཆོད་སྦྱིན་ཞིང་དུ་ཡང་དག་འོངས། །ས་སྐྱོང་བོ་པི་མ་ཧཱི་པཱ་ལསྟནྣཱ་མ་ཤ་བ་ཎཱ་ད་རཱཏྲ། །ད་དོ་བྲྀཏྠསུ་མ་ཏ་ཡེ་ནོར་ལྷའི་བུ་ཡིས་ཀྱང་། །དེ་ཡི་མིང་ཐོས་གུས་པ་ཡིས། །བློ་གྲོས་བཟང་ལ་མཆོད་སྦྱིན་ནིི། །མཇུག་ཡ་ཛཱནྟེ་པཉྩ་དཀི་ཎཱ། །།ཙ་ཏུ་རཏྣ་མ་ཡཾ་ཤཡོ་ཧི་མོ་དཎྜ་ཀ་མཎྜ་ལུ། །ཏུ་ཡོན་ལྔ་དགའ་བས་ཕུལ། །ཁྲི་སྟན་རིན་ཆེན་བཞི་རང་བཞིན། །གསེར་གྱི་དབྱིག་པ་རིལ་བ་དང་། །སུ་བཎྜ་རཏྣ་པཱ་ཏཱིཙྩཾ་ཧི་ར་ཎྱ་ཤ་ཏ་པཉྩ་ཀཾ། །ཨ་ལཾ་ཀྲྀ་ཏཱ་སྭ་ཡཾ་སུ་ཏྭཾ་རིན་ཆེན་གསེར་གྱི་སྣོད་དང་ནི། །གསེར་ནི་བརྒྱ་ཕྲག་ལྔ་དག་གོ། །རང་གི་བུ་མོ་རྒྱན་གྱིས་བརྒྱན། །མི་ཚྭ་པེ་དཱ་ཏུཾ་ས་མུངྒ་ཏེ། །སུ་མ་ཏིརྣ་གྲ་ཧིངྐ་ནྡྷ་བྲཧྨ་ཙརླ་མུ་དཱ་ཧ་རན། །བདག་སྟེར་བར་བརྩོན་པ་ན། །བློ་གྲོས་བཟང་གིས་ཚངས་སྤྱོད་ནི། །བརྗོད་ཅིང་བུ་མོ་ཡོངས་མ་བླངས། །ཀ་རྩྭ་དྷི་སུནྡ་རཱི་ནཱ་མ་ཏཾ་ཛྲཏྟཱ་ཀ་ན་ཀ་དྷུ་ཏཾ། །སཱ་བྷི་ལཱ་ཥཱ་ས་མུཏྠཾ་སྱ་མཛེས་མ་ཞེས་པ་བུ་མོ་དེ། །གསེར་གྱི་འོད་ཅན་དེ་མཐོང་ནས། །མངོན་འདོད་ལྡན་པས་ངོ་ཚ་ནི། །ལཛྙཱ་ཙཀྲེ་ཏ་དརྦ་ནཾ། །པྲ་ཏཱ་ལྱཱ་ཏཱ་ཏ་ཏསྟེ་ན་སཱ་ཀ་རྩྭ་བྲཧྨ་ཙཱ་རི་ཎཱ། །རབ་ཏུ་བཏང་ནས་ཞུ་བ་བྱས། །དེ་ནས་དེ་ཡིས་ཕྱིར་བཟློག་ཚེ། །བུ་མོ་ཚངས་སྤྱོད་ལྡན་མ་དེ། །།།—།
