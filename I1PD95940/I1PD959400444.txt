ཚྭ་བ་ཏིཔྲ་ཎ་ཏསཱ་པུ་རངྒ་ཏ། །བི་བཱ་པ་ནཾ་གཾ་ཧ་པ་ཏེ་རཱ་དི་དེ་རབ་བཏུད་ནས། །རང་གི་གྲོང་ཁྱེར་དག་ཏུ་སོང་། །ཁྱིམ་བདག་རང་གི་གྲོང་ཁྱེར་ལས། །གཞན་དུ་ཤ་ནི་ཛཱཏ་པུ་རཱཏ། །པྲ་སཱ་དི་ནི་སུ་དཏྟ་སྱ་དེ་བ་ཏཱ་ནཱཾ་བ་ཏི་ཏ་ཏ། །སྨྱུགས་ཞེས་བཀའ་ཡིས་བསྒོས། །དེ་ནས་ལེགས་སྦྱིན་དག་ལ་ལྷ། །རབ་ཏུ་དང་བས་མི་བདག་ལ། །གླུ་དྲ་ཛནྡྷ་བྷི་རུཏྠཾàཉྩཀྱེ་དཾ་ཤ་བི་ཥཱ་ཀུ་ལཾ། །ས་ཏྲསྟསྟིརྗེ་པཔYསྲོག་ཆགས་ཕྲ་མོ་རབ་སྤྲོས་ཏེ། །ས་ཡི་དུག་གིས་འཁྲུགས་པར་བྱས། །དེ་དག་གིས་བཏབ་མི་བདག་དེ། །སཱ་མཱ་ཏྱནྟ༔པུ་རཱ་ནུ་ག། །ག་ཏཱ་པྲ་སཱ་ད་ཡཱ་མཱ་ས་སུ་དཏྟཾ་ཛི་ན་ཤཱ་ས་བློན་པོ་བཙུན་མོའི་རྗེས་འབྲང་ཅིབཅས། །ཕྱིན་ནས་རྒྱལ་བའི་བསྟན་པ་ལས། །ལེགས་སྦྱིན་ལ་ནི་དད་ནཱཏེ༑། ཨི་ཏི་ས་ས་ཏ་ཏ་པྲ་ཏྱཱ་སནྡྷ་པ་རཱ་མྲྀ་ཏ་ནིབྷ་རཾ་བྷ་ག་བ་དུ་དི་ཏཾ་པར་བྱས། །དེ་ལྟར་རྟག་ཏུ་ཉེ་བར་གནས་པས་མཆོག་གི་བདུད་རྩི་དག་གི་ཚོགས། །ཁྱིམ་གྱི་བདག་པོ་ལེགས་ཤྲིནྟཾ་བྷེ་ཛེ་སུ་དཏྟ་ཀྲྀ་ཧཱ་དྷི་པ། །ལྭ་མི་བ་ལ་བྷ་ཏེ་བིབྷྱཱ་ཡཱ་ས་པྲ་ཡཱ་སྦྱིན་དེ་ཡིས་བཅོམ་ལྡན་གྱིས་གསུངས་ཞི་བ་བསྟེན། །དྲི་མེད་ཡིད་ལྡན་རྣམས་ཀྱི་མཐའ་རུ་གནས་པས་དབེན་ས་བི་བཛྫི་ཏཾ་བི་མ་ལ་མ་ན་སཱ་མནྟེ་བཱ་སཱི་བི་བེ་ཀ་མ་ཧཱ་ནི་དྷི་མ། །པའི་གཏེར་ཆེན་ནི། །རང་གི་བཞིན་དུ་གེགས་དང་ཉོན་མོངས་ངལ་དུབ་རྣམ་པར་སྤངས་པས་འཐོབ། །ཨི་ཏི་ཀྵེ་མེརྩྭ་བི་ར་ཙི་ཏཱལཾ་བོ་དྷི་སཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཾ་ཅེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་གྲོ་ཧ་པ་ཏི་སུ་དཏྟཱ་བ་དཱ་ནཾ་པཉྩ་ཏྲི་ཤཔལླ་ཝ། །།།ལས་ཁྱིམ་བདག་ལེགས་སྦྱིན་གྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་སུམ་ཅུ་རྩ་ལྔ་པའོ། །།།ཕ་ལཾ་ས་མཱ་ནཾ་ལ་བྷ་ཏ་ས་དཱ་ཏུརཱ་ཏི་ཀཱ་ཎ་དཱ་ན་ས་ཧཱ་ཡ་ཏཾ་ཡ། །སྦྱིན་པའི་གྲོགས་སུ་སྐད་ཅིག་སུ་གྱུར་པ། །དེ་དག་སྦྱིན་བདག་འབྲས་བུ་མཚུངས་པ་འཐོབ། །པ་རོ་པ་ཀཱ་ར་པྲ་ཎ་ཡོདྱ་ཏཱ་ནཾ་ནཱ་པུ་ཎྱ་ཀརྨ་ས་ཙི་བ་ཏུ་མེ་ཏི། །གཞན་ལ་ཕན་པར་བརྩོན་པར་ཞུགས་རྣམས་ཀྱི༄༅། །གྲོགས་སུ་བསོད་ནམས་ལས་མེད་མི་འགྱུར་རོ། །།
