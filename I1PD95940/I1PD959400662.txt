ར་ཟ་བཏྭཾ་ག་ནཾ། །ཡཱ་ཏུ་དྷཱ་ནཱི་མི་བ་སཏ་བ་ལཱ་ཀཾ་བྷོལཱུ་མུཀླུ་ཀཾ། །ཏྭཾ་ཤིང་ལྟོགས་པས་གཟིར། །སྲིན་མོ་བཞིན་དུ་བུ་ཚ་དག། ཟ་བར་འདོད་པ་མཐོང་བར་གྱུར། །དེ་དྲྀབྷཱ་ཀ་རུ་ཎཱ་ཀཱ་སཱ་ཥཱ་ཎཾ་ས་མ་ཙི་ཏ་ཡཏྲ། །ཨ་ཧོ་སཱ་དེ་ཧ་སྣེ་ཧེ་ན་མ་ཏིམཐོང་སྙིང་རྗེ་རྒྱས་གྱུར་པ། །དེ་ཡིས་སྐད་ཅིག་ཡང་དག་བསམས། །ཀྱེ་མ་རང་ལུས་ལ་ཆགས་པས། །བློ་པཱ་བེ་པྲ་བཎྜ་ཏེ། །ཨ་སྱ་བྷོ་ཛ་ན་མཱ་ཧནྟུ་བྲ་ཛཱ་མི་སཱ་ཁཾ་ཧཾ་ཡ་དི། །ཏ་དི་ནི་སྡིག་ལ་རབ་ཏུ་འཇུག། འདི་ཡི་ཟས་ནི་ལེན་དུ་བདག། གལ་ཏེ་རང་གི་ཁྱིམ་སོང་ན། །བཀྲེས་པས་ཡཾ་ཀླུཏྤ་རི་གླུ་མ་བྷ་གྲུ་ཡ་བྱེ་བ་དཱ་ར་ཀཾ། །ཨ་ཐ་བཱ་ཤི་ཤུ་མཱ་དཱ་ཡ་ཡ་དི་ཡོངས་སུ་ཉེན་པ་འདིས། །བུ་ནི་ངེས་པར་ཟ་བར་འགྱུར། །ཡང་ན་བདག་གིས་བུ་ཁྱེར་ནས། །གལ་ཏེ་གཙྪ་མི་མནྡི་རཾ། །ཏ་དེ་ཥཱ་ཀྲྀ་ཤ་ཏྭཾ་ཡཱ་ཏཱ་ས་ཏྱསྐྱ་ཛ་ཏི་ཛཱི་བི་ཏཾ། །ཨི་ཏི་ཁྱིམ་དུ་ཕྱིན་པ་ན། །འདི་ནི་བཀྲེས་པ་ཉིད་གྱུར་ཏེ། །འཕྲལ་ལ་འཚོ་བ་གཏོང་བར་འགྱུར། །དེ་ལྟར་སཾ་ཙིནྟ་ད་ཡཱ་ཡཱ་ཏསྨིནྡྲ་བྷ་ཡ་སཾ་ཤ་ཡེ། །ཛ་གཏྠཾ་ཏཱ་ར་ཎཱ་ཡེ་བྷ་པྲ་ཎི་གཉིས་ཀ་ཐེ་ཚོམ་དང༌། །ལྡན་པ་དེ་ལ་བརྩེ་བས་བསམས། །འགྲོ་བ་བསྒྲལ་བའི་སླད་ཉིད་དུ། །དེ་ཡིས་དྷཱ་ནཾ་བི་དྷཱ་ཡ་སཱ། །ཙི་ཏཱ་ཏ་ཡོ་པ་ནཱི་ཏེ་ན་ཤི་ན་ཤསྡེ་ཎ་ནི༔་ལཱ། །ནི་སྨོན་ལམ་བཏབ་བྱས་ཏེ། །དེ་ཡིས་གཏད་པའི་མཚོན་རྣོན་གྱིས། །གཡོ་བ་མེད་པར་བཅད་ནས་ནི། །དེ་ཛཾ་སྟ་ན་ཡུ་གཾ་ཏ་སྐྱེ་ད་དོ་ཛི་བི་ཏ་དྷཱ་ར་ཎཾ། །ཏྲི་ལོཀྱ་སྤྲི་ཤི་སཱཉྩམེ་ཏ་སྱ་ལ་རང་གི་ནུ་མ་ཟུང་། །འཚོ་བ་ཡི་ནི་ཟུང་ཉིལ་བྱིན། །དེ་ནས་དེ་གྲགས་ངོ་མཚར་ཅན། །འཇིག་རྟེན་ཡ་ཤ་སི་བི་ཐུ་ཏེ། །བི་པྲ་རཱུ་པེ་ཎ་ཤ་ཀྲེ་ཎ་ཏ་ཏཔྲཏྟཱ་ས་མེ་ཏྱ་སཱ། །གསུམ་རེག་གྲགས་པའི་ཚེ། །བརྒྱ་བྱིན་བྲམ་ཟེའི་གཟུགས་ཀྱིས་ནི། །མངོན་དུ་ཕྱོགས་ཏེ་དེ་ལ་དྲིས། །ཨ་བི་ཏེས་ན་དཱ་ནེ་ན་མ་ནོ་བི་ཀྲྀ་ཏི་མཱ་ཡ་ཡོ། །ཨི་ཏི་པྲྀཏྟཱ་པ་ཏཱི་ས་ཏྱཱ་བཱ་དི་ཁྱོད་ནི་ྋནུ་མ་བྱིན་པ་ཡིས། །ཡིད་ལ་རྣམ་འགྱུར་མ་བྱུང་ངམ། །ཞེས་དྲིས་དེ་ལ་དག་པ་དང་། །བདེན་ནཱི་ཏཾ་ཛ་གཱ་ད་སཱ། །ཡ་དི་མེ་ནཱ་བྷ་བཀྐ་ཉིཏ་བི་ཀཱ་རཾ་ཤ་སྟ་ནཱརྦ་ཎེ། །ཏེ་ན་སྨྲས་ལྡན་པ་དེས་སྨྲས་པ། །ནུ་མ་བྱིན་ལ་རྣམ་འགྱུར་ཆ། །གལ་ཏེ་བདག་ལ་འགའ་མ་གྱུར། །བདེན་—།
