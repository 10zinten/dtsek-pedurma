།སཾ་ཁྱི། །ཏྲེ་ཎོ་ད་ནཱ་བྱི་ར་མྲྀ་ཏོ་ད་ནཱ་བྱིཉྩིཏྟ་པྲ་ས་དསུ་མ་ཧཱ་ན་བº། །བ་ཐོབ། བྲེ་བོ་ཟས་སོགས་ཀྱི་བདུད་རྩི་ཟས་སོགས་ཀྱིས། །ཤིན་ཏུ་ཆེ་བའི་སེམས་ནི་དང་བ་ཐོབ། །ཀེ་ཙིདྨ་ཡུཤཱ་བ་ཀ་བོ་དྷི་ཡུཀྟ་པྲ་ཏྱཾ་ཀ་བོ་ངྷ་ནི་ར་ཏཱཉྩ་ཀེ་ཙིཏ། །སམྱཀྟ་ཐཱ་ཁ་ཅིག་ཉན་ཐོས་བྱང་ཆུབ་ལྡན་པར་གྱུར། །ཉི་ཚེའི་བྱང་ཆུབ་རེག་དང་ལ་ལ་དག། དེ་བཞིན་ཡང་དག་ནུཏྟ་ར་བོ་དྷི་སཀྵཔ་རེ་བ་བྷུ་བུརྒ་ག་ན་པྲ་སནྣཱ། །ཐོ་ཏཔ་རི་པཱལྟ་བླ་མེད་བྱང་ཆུབ་ཞུགས། །གཞན་ནི་མཁའ་བཞིན་རབ་ཏུ་དང་བར་གྱུར། །དེ་ནས་གཞན་གྱིས་རྒྱུན་ཕ་ལཾ་ཏ་ཏོ་རྩེ་ས་ས་ཀྲྀ་དཱ་གཱ་མི་ཕ་ལཾ་ཏ་ཐ་ནྱ། །ཨདྱེ་སྱ་ནཱ་གཱ་མི་ཕ་ཞུགས་འབྲས་བུ་ཐོབ། །དེ་བཞིན་གཞན་གྱིས་ལན་གཅིག་ཕྱིར་འོང་དང་། །ཕྱིར་མི་འོང་འབྲས་དེ་ཡང་ལཾ་ཏ་དཱ་པུ་རདྷཏྠ་ལཾ་ཀླུ་ཤ་བི་མུཀྟི་ར་རྩེ། །ཨེ་ཀསྟུ་ཏ་ཏཱ་ཛི་ཏ་པཱ་པ་གཞན་གྱིས་ཏེ། །ཉོན་མོངས་རྣམ་གྲོལ་གཞན་གྱིས་དགྲ་བཅོམ་ཐོབ། །་མདུན་སར་དེ་ནི་ལྷ་སྦྱིན་གཅིག་ཤ་པསྟ་མ༔ས་མུ་ཧོ་པ་ཧི་ཏ་པྲ་མོ་ཧ། །མཱ་ཡོ་ཧ་མི་ཏྭཱ་ཧ་ཧ་སན་ཛ་པུ་ཞིག། མུན་པའི་ཚོགས་ཀྱིས་གཏིབས་ཤིང་རབ་རྨོངས་པསི། །སྡིག་དམོད་མ་ཐུལ་སྐྱེ་བོ་རྣམས་ཀྱི་ན་ནཾ་ས་ཏྱ་སྦི་ཏི་སཾ་ས་དི་དེ་བ་དཏྟ། །ནཱི་པཾ་ཏུ་བཱཏྨ་ལྱ་ནི་ལཱི་ན་མེ་བ་ལུགས། །བདེན་པ་ལ་བཀོད་འདི་ནི་སྒྱུ་ཞེས་སྨྲི། །མི་ཡི་བདག་ཀྱང་མཉེས་གཤིན་གྱིས་ཆགས་ཤིང་། །པུ་ཏྲི་ད་ཡཱཏ་པྲ་ཏྱཱ་ཏ་ཛཱ་ཏ་དམྨཾ། །མོངྒ་ལ་བྷིཀྵ་ཛི་ན་ཤཱ་ས་ནེ་ན་མ་ཧདྷི་སྲས་ནི་དར་ལས་ཟློག་པར་དོགས་ྋསྐྱེས་པ། །རྒྱལ་བའི་བཀའ་ལས་དགེ་སློང་མོ་གལ་བུས། །རྫུ་འཕྲུལ་བྷིཧཱི་ཏ་མ་དཾ་ཙ་ཀཱ་ར། །དྲྀཟླ་ཕི་རཱ་ཛཱ་བྷ་ག་བཏ་པྲ་བྷཱ་པི་ན་ཏུདྡྷ་ཆེན་པོས་རྒྱགས་དང་བྲལ་བར་བྱས། །བཅོམ་ལྡན་མཐུ་ནི་རྒྱལ་པོས་མཐོང་གྱུར་ཀྱང་། །ཤིན་ཏུ་རྨད་ཏཾ་པོ་རུ་ཥ་མེ་བ་མེ་ནེ། །ཨ་ཧྥུ་ས་ལཱི་ནཱ་ནི་ཛ་ན་མྱ་ནུ་ནཾ་སོཏྟནྟ་ཀྲྀ་ཏྭཱ་ནི་ན་བིསྨ་བྱུང་སྐྱེས་བུ་ཉིད་མི་སེམས། །ཕུལ་བྱུང་མཛད་པ་གོམས་པས་རེག་པ་ནི། །ངེས་པར་སྐྱེ་བོའི་ཡ་མཚན་རྒྱུ་ཡཱ་ཡ། །ཨ་ཐ་པ་རེ་དྷབྷ་ག་བཱན་སུ་རེནྡྲ་སཾ་པ་དི་ཏེ་ཧེ་མ་མ་ཧཱ་བི་མཱན། །མ་ཡིན། །དེ་ནས་ཉི་མ་གཞན་ལ་ལྷ་དབང་གིས། །རབ་བསྒྲུབས་གསེར་གྱི་གཞལ་ཡས་ཁང་ཆེན་དུ། །།
