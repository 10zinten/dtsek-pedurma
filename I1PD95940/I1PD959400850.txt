ཀ་རི་པཱུ་རི་ཏ་བྷཾངྒ་སཱནྟྲ། །།སཱ་བནྡྷ་ཀཱི་ཡ་དི་ཀ་རོ་ཏི་ན་ཀུཉྫ་ར་སྱ་གཱ་ཌྷ་ནུ་རཱ་རྨོངས་པའི་བསླབ་པ་མ་བྱིན་ན། །ཡོངས་རྫོགས་སྦྱིན་པས་བུང་བའི་དོན་མཐུན་ཡོངས་སུ་དགྱེས་བྱས་པ། །ག་བི་བ་ཤ་སྱ་བི་མོ་ཧ་དིགྷཾ། །ཏ་ཏསཱི་ར་གྲི་ཧཾ་ཏཱ་རཱ་ཛཱཙྩཾ་ཊ་ཁྱུ་ཡི་མགོན་པོ་ལྷུང་བའི་གཡང་ལ་ངེས་པར་ལྷུང་བ་མིན། །དེ་ནས་དལ་གྱིས་རྒྱལ་ཁབ་ཏུ། །བྲན་ཁོལ་ཀིཀུ་ཊ་ཀཱ་མི་ནི། །ཙྪན་ནུ་བྲྀཏྟི་མཱུ་དྷིཟི་ནཱ་ཤཱ་ཡ་པྲ་བེ་ཤི་ཏཱ། །བླུན་པོ་འདུན་པ་ཡི། །རྗེས་སུ་འཇུག་པས་བརླག་པའི་སླད། །བཅོས་མའི་བུད་མེད་རབ་ཏུ་བཅུག། བི་བིཀྟ་ཏུལྟ་དྷིརྨ་སྱ་ཏ་སྱ་ཧཱུཾ་ཌྷ་བི་ལཱ་ཏི་ཎ། །བ་བྷཱུ་བ་དབེན་པར་བརྟན་པ་བཏང་བྱས་ཤིང་། །གསང་བ་མངོན་པར་འདོད་དེ་ལ། །དུས་ནི་མཛེས་མའི་ཀཱནྟཱ་རཱུ་པེ་ཎ་ཀཱ་ལཀཥྛ་གྲ་ཧོརྨུ་ཁ། །ཏ་ཏསྟ་དཱིརྒྱུ་ནི་དཱ་ཡ་ཤ་གཟུགས་དག་གིས། །མགུལ་ནས་འཛིན་ལ་མངོན་ཕྱོགས་གྱུར། །དེ་ནས་གཉིད་ནི་རིང་པོའི་སླད། །གུས་ཡཱ་མཱ་རཱུ་ཌྷ་མཱ་ད་རཱཏྲ། ་། བི་ཧཱ་ཡ་ནནྟ་ཀཱི་རཱུ་པཾ་ཀུ་མཱ་རཿས་ཧ་སཱ་པྲ་བཱིཏ། །པས་མལ་སྟན་འཛེགས་དེ་ལ། །གཞོན་ནུ་གར་མཁན་མ་ཡི་གཟུགས། །སྤངས་ནས་འཕྲལ་ལ་རབ་སྨྲས་པ། །བྷོ་གཱ་ནུཙྪེ་ན་བྷ་བ་ཏཱ་བྷཱ་ཏྲ་སྣེ་ཧཱ་ན་པེ་ཀྵི་ཎཱི། །ས་ཧ་བྷོ་གྱ་མི་དཾ་རཱ་ཚུ་ཁྱོད་ཀྱི་ལོངས་སྤྱོད་ལ་ཆགས་ཤིང་། །སྤུན་ཟླའི་མཛའ་ལ་ལྟོས་མེད་པས། །ལྷན་ཅིག་སྤྱད་བྱ་རྒྱལ་སྲིད་ནི། །ཀ་ཐ་མེ་ཀེ་ན་བྷུ་ཛ་ཏེ། །ནིརྡོ་ཥཀྵེ་ཤ་ཛ་ལ་དྷོཀིསྟོ་ཧཾ་བི་ཥ་མེ་ཏཱ་ཡཱ། །གཅིག་པུ་ཡིས་ནི་ཇི་ལྟར་སྤྱོད། །མི་བཟད་ཉོན་མོངས་ཆུ་གཏེར་དུ། །སྐྱོན་མེད་ང་ནི་ཁྱོད་ཀྱིས་འཕངས། །སཱ་ཀརྨ་ཡོ་གཱ་དུཏཱིཎྞསྟཏྲ་ཏཱི་ཀཱ་ར་ཙིནྟ་ཀ། །ཨུཥཱ་ཏི་བདྷཱ་ཚིགས་པ་དེ་ཡི་ལེ་ལན་སེམས་བྱེད་བདག། རང་གི་ལས་ནི་སྦྱོར་བས་ཐར། །ཞེས་བརྗོད་དེ་ཡིས་མི་བདག་ཏི་པྲྀ་མྱྀ་རཱ་ཛུ་མ་བ་སྱ་ས། །མུདྡྷ་པྲ་པཾ་ཤ་མ་ཐཾ་ཙཀྲེ་ཛ་ན་ཏཱ་ཤཱ་པ་བཅིངས༑། རྒྱལ་སྲིད་རབ་ཏུ་རྒྱས་པ་ཐོབ། །སྐྱེ་བོའི་ཚོགས་ནི་དབུགས་ཕྱུང་སྟེ། །སྲེད་ལྡན་རབ་ཏུ་ཞི་ནེ་ན་པ། །ས་སཾ་པྲཏྟིཾ་པ་དསྟི་བྲོ་ཙིནྟ་ཡན་སཱ་པ་རཱ་བྷ་ཐཾ། །ཤི་ལོ་བར་བྱས། །གནས་ཐོབ་དེ་ཡི་རང་ཉིད་ཀྱིས། །གནོད་པ་དྲག་པོ་སེམས་བྱེད་ཅིང་། །སྔ་དྲོ་བརྩེ་མེད་།
