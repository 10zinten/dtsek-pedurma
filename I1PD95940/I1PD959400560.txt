ཀྲྀ་ཏ་ཀ་ཎ། །གྷོ་ར་ནིགྷོ་ཏ་མེ་གྷོ་གྷ་གྲསྟ་ལོ་ཀཱས་མ་ཡ་ཡུ། །པའི་དུས་བྱས་ནས། །འཇིགས་རུང་སྒྲ་སྒྲོགས་སྤྲིན་གྱི་ཚོགས། །སྣང་བ་འཇོམས་པས་ཡང་དག་འོངས། །ཏེ་ཊཱ་བྱ་བ་སི་ཏཾ། ཛྙཱ་ཏཱ་སཏྠ་ཛྷསཀྵ་པ་ཏི་ན། :། རཀྵ་ཀཱ་མཾ་ཀཱི་ཏི་དེ་དག་སྩོལ་བ་ཀུན་མཁྱེན་གྱིས། །མཁྱེན་ནས་ས་ཡི་བདག་པོ་ཡི། །ཕྱོགས་སུ་ལྷུང་ཞིང་བསྲུངས་པ་ཏེརྨོངྒ་ལྕེ་ཡ་ན་མཱ་དི་ཤཏྲ། །ཨ་ཐ་ནཱ་ག་ག་ཎཊཾཥྚ་བཛྲ་པྲྀཏྟིརྨ་ཧཱི་བ་ཏི། །བཟོད་པ། ་མོ་གལ་གྱི་བུ་ལ་བསྐོས། །དེ་ནས་ཀླུ་ཡི་ཚོགས་རྣམས་ཀྱིས། །རྡོ་རྗེའི་ཆར་ཆེན་རབ་སྤྲོས་པ། །བྷུབྷརྟུཔཏྠ་བྷཱ་བེ་ན་པྲ་ཡ་ཡོ་པུ་དྲྀཊྛིཏྟཾ། །ཤམྦྷབླ་བྲྀཥིནི་པི་ཌ་པེ་ཏི་དེ་ཡི་མཐུ་ཡིས་ས་བདག་ལ། །མེ་ཏོག་ཆར་པ་ཉིད་དུ་གྱུར། །དེ་ནས་དེ་དག་གིས་མཚོན་དང༌། །རྡོ་ཡི་ར་ཐ་དུས་ཧཱ། །མོངྒ་ལྱཱ་ཡ་ན་སི་ཀལྤཏ་ཡ་ཡོ་རཱ་ཛཱཧ་བྷོ་ཊུ་ཏྭཾ། །ཏཏྠ་བྷཱ་ཆར་དྲག་བཟོད་དཀའ་བ། །མོ་གལ་བུ་ཡི་བརྟག་པ་ལས། །རྒྱལ་པོ་ལ་འོས་ཟས་ཉིད་གྱུར། །དེ་ཡི་བཱཏ་པྲ་ཡཱ་ཏེ་ཥཱ་བསྒོཧ་ཧེ་ཥཱ་བྷོ་གི་ཥཱ། །ག་ཏྭཱ་བ་བནེ་སུ་ག་ཏཾ་ཚིགས་མཐུ་ལས་གདེངས་ཅན་རྣམས། །སྤྲོ་བ་ཉམས་ནས་སོང་བའི་ཚེ། །འཚེ་བ་དང་བྲལ་མི་ཡི་བདག། སོང་པ་ཏིབཱི་ཏ་བི་སླ་བ། །ས་མོངྒ་ལྟ་ཡ་ན་སཱུརྒྱུམུ་ཙི་ཏཾ་བྷོ་ག་སམྱ་ནས་བདེ་གཤེགས་ལ་ཕྱག་འཚལ། །རྒྱལ་བའི་བཀའ་ལས་མོ་གལ་བུ། །དེ་ཡི་ལོངས་སྤྱོད་ཕུན་ཚོགས་དཱ༑། བྷཏྟི་སཾ་སྐུ་ར་སུ་བྷ་གཱ་བི་དེདྷེ་ཛི་ན་ཤཱ་ས་ནཱཏ། །ཏ་ཏ༔སཱརྒོ་ཀྱིས། །གུས་པས་ལེགས་སྦྱར་སྐལ་བ་བཟང་། །རིགས་པ་ཉིད་དུ་མཆོད་པར་བྱས། །དེ་ནས་མཐོ་རིས་ཙིཏྟཾ་བྷིཀྵབི་བྷཱུ་ཏི་བཱིཀྵ་བྷཱུ་པ་ཏེ། །ས་པྲཙྪ་ཀོ་ཏུ་ཀ་བེ་ཤཱཏ་སཏྠ་ལ་འོས་པའི། །ས་བདག་འབྱོར་མཐོང་དགེ་སློང་གིས། །ཐལ་མོ་སྦྱར་ནས་ཀུན་མཁྱེན་ལ། །དགེ་མཚན་ཛྷ་ར་ཙི་ཏཱཉྫ་ལི། །བྷ་ག་བརྐ་སྱ་པུ་ཏྱཱ་སྱ་པྲ་བྷཱ་བེ་ན་པྲ་སེ་ན་ཛིཏ། །།འབེབས་ལས་རབ་ཏུ་ཞུས། །བཅོམ་ལྡན་བསོད་ནམས་ཅི་ཞིག་གི། །མཐུ་ཡིས་རྒྱལ་པོ་གསལ་རྒྱལ་ནི། །སརྦ་བི་རཱ་ཛི་ཏྭཾབྷོ་གེ༔པྲ་ཛུ་སཾ་རཱཛ་མསྣ་ཏེ། །ཨིཀྐཱཙྩཾ་པཱ་ཤི་དེ་བཱ་སྱ་ལོངས་སྤྱོད་ཀུན་གྱིས་རྣམ་མཛེས་པའི། །རབ་རྒྱས་རྒྱལ་སྲིད་ཆེན་པོ་ཐོབ། །འདི་ལ་རྟག་ཏུ་ས་ལི་ཡི། །—།
