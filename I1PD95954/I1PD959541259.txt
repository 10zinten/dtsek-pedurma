བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི༽ཀ་བ་ང་རྟགས་ཀྱི་མིང་ཅན་ལས། སི་ལ་ཨཱ་སྟེ་སི་དབྱི་བའང་ངོ་ཞེས་པས་ཨ་ཡིག་དང་སི་དབྱི་བ་སྟེ། །ཀཎྜ་དང༌། བྱིངས་ལ་ཚིགས་སྒྲའི་ཨཱར་རོ་ཞེས་པ་སྟེ། ཀཏཱ་རོ། ཀཏཱ་ར། ཀཏཱ་རཾ། ཀཏཱ་རོ་དང༌། ཤས་ལ་ཨགི་བཞིན་ནོ་ཞེས་པས་ཨགི་། དང་འདྲ་བར་འགྱུར་ཏེ། །ྋཤས་ཀྱི་ཨ་ཡིག་ཅེས་པ་ལ་སོགས་པས་ཤས་ཀྱི་ཨ་ཡིག་་རེ་ཡིག་ཏུ་ཐོབ་ཅིང་། ས་ཡང་ན་ཉིད་དུ་འགྱུར་ལ་མཚུངས་པ་ལས་རིང་པོ་ཉིད་དང་ཕ་རོལ་དབྱི་བའང་སྟེ། ཀཛིན་དང༌། ་ཊ་ལ་སོགས་པ་ལ་ནི་པི་ཏྲེའི་སྒྲ་དང་འདྲ་སྟེ། །ཀརྟི། ཀརུཾ་ཧཱུཾ། ཀཏྲ་བྷི། ཀརྗེ། ཀནྡཾ་བྷཾ། །ཀཛྲ་བྷུ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ་ནད་ལ་སོགས་པའི་མདོས་ཨཱིའི་རྐྱེན་དང༌། ཝ་ཉིད་དུ་འགྱུར་ཞིང་ན་དཱིའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱ་སྟེ། ཀཏྲི། །ཀཏྱཾ། ཀརྐྱེ༑ཞེས་པ་ལ་སོགས་པའོ། །།མ་ནིང་ལ། ཞིཀནྡོོཀཎྜི་ཎི། གྷཊ་ལ་་ཡང་སཾ་བུདྡྷི་མ་ཡིན་པ་ལ་ཞེས་པས་རིང་པོ་ཉིད་དེ། ཀཚཱ་ཎ། སླར་ཡང༌། ཀཚིགས། །ཀཎྜི་ཎཱི། ཀཛྲ་ཎི། །ཌཱ་ལ་སོགས་པའི་དབྱངས་ལ་ཡང་ན་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་ཏེ། ཀཅུ། ཀཛྲ་ཎཱ། ཀནྡཾ་ཧཱུཾ། ཀཛྲ་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་ཀནྡཾ། ཧེ་ཀཎྜི་ཎི། ཧེ་ཀཝཱ་ཎི། དེ་བཞིན་དུ་ཚིགས་ཛའི་མཐའ་ཅན་དང་དྲྀནའི་མཐའ་ཅན་འང་སྟེ། ཁ་པ་ཊྚཾ། བྷོཊྚཾ། སྲོ་ཚིགས་ལ་སོགས་པ་རྣམས་སོ། །།།།གཡོན་:སུ་མ་ཏྲིའི་སྒྲ་ནི། འདིའི་མ་མཛེས་མ་ཡིན་ཞེས་འབྲུ་མང་པོ་ཅན་ལ་སྐྱེས་པའི་ནི་པི་ཏྲིའི་སྒྲ་དང་འདྲ་བ་སྟེ། །སུ་མཱ་ཏཱ། ་སུ་མཱ་ཏེ་རོ། ལ་སུ་མཱ་ཏ་རི། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ་མཱ་ཏྲཱའི་སྒྲ་དང་འདྲ་བ་སྟེ། སུ་མཱ་ཏཱ་ར། སུ་མཱ་ཏ་རོ། སུ་མཱ་ཏ་ར། ཞེས་པ་ལ་སོགས་པའོ། །མ་ནིང་ལ་ཀནྡོའི་སྒྲ་དང་འདྲ་བ་སྟེ། སུ་མཱ་ཏྲ། སུ་མཱ་ཀྲྀ་ཎཱི། སུ་མཱ་ཀྲྀ་ཎི། །དེ་བཞིན་དུ། སུ་།
