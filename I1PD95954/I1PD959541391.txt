འ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་། ནཱི་ཏེ། ཀུཉ་སྒྲ་ལའོ། ་་། ཀུ་ནཱ་ཏི། ཀུ་ནཱ་ཏེ། མཱིཉ་དྲུཉ་འཚེ་བ་ལའོ། །མཱི་ནཱ་ཏི། མཱི་ནཱི་ཏེ། ཧི་ནུ་མི་ན་ལ་སོགས་པ་རྣམས་ཀྱི་ཞེས་པས་ཎ་ཉིད་དོ། །པྲ་མཱི་ཎཱ་ཏི། པྲ་མཱི་ཎཱི་ཏེ། དྲི་ཎཱ་ཏི། དཱ་ཎྜི་ཏེ་པཱུ་ཉ་དག་བྱེད་ལའོ། །པུ་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ་ཞེས་པས་པུ་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ། །པུ་ནཱ་ཏི། པུ་ནཱི་ཏེ། ལུཉ་གཅོད་པ་ལའོ། །ལུ་ནཱ་ཏི། ་ལུ་ནཱི་ཏེ། སྡིཉ་འགེབས་པ་ལའོ༑། སྟི་ཎཱ་ཏི། སྦི་ཎཱི་ཏེ། ཨ་ཙཱི་ཎཱ་ཏི། ཨཱ་སྟྲཱི་ཎཱི་ཏེ། བི་སྠཱཎཱ་ཏི་ཀཎྜ་པ་:ཊཾ། བི་སྡེ་ྋཎྜི་ཏི། ཀྲྀཉ་འཚེ་བ་ལའོ། །ཀྲྀ་ཎཱ་ཏི། ཀྲྀ་ཎཱི་ཏེ། བྲྀཉ་སྒྲིབ་པ་ལའོ། །ཤྲཱི་ཎཱ་ཏི། བྲྀ་ཎི་ཏེ། དྷུཉ་འགུལ་བ་ལའོ། །དྷ་ནཱ་ཏི། དྷུ་ནཱི་ཏེ། །གྲ་ཧ་ཉེ་བར་ལེན་པའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །གྲ་ཧི་ཙཱ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་ས་ར་ཎའོ། །ར་ཥ་རིའི་ཡི་གེ་རྣམས་ལས་་་་ཞེས་པ་ལ་སོགས་པས་ནའི་ཎ་ཉིད་དོ། །གྲཧྨཱ་ཏི། གྲིདྡྷི་ཏེ། །ཨུ་པ་གྲཧྨཱ་ཏི། ཨུ་པ་གྲིདྡྷི་ཏེ། པ་རི་གྲིབྷཱི་ཏེ། ཨ་ནུ་གྲིཉྩ་ཏི། ཨ་ནུ་གྲིཉྩི་ཏེ། ་ནི་གྲིདྡྷཱ་ཏི། ནི་གྲིབྷཱི་ཏེ། ཨ་བ་གྲོསཱུ་ཏི། ཨ་བ་གྲིཉྩི་ཏེ། རྣམ་པར་བཤད་པ་རྣམས་སོ། །༉་སྦེ་འཚེ་བ་ལའོ། །།ལྷག་མ་ལས་བྱེད་པ་པོ་ལ་པ་རསྨིའི་ཚིག་གོ། །ཤྲཱི་ཎཱ་ཏི། སྦེ་ཎཱ་ཏི། སྲིད་སྐྱོང་བར་བྱེད་པ་དང་རྫོགས་པར་བྱེད་པ་ལའོ། །པྲ་ཎཱ་ཏི། བློ་ཤྲཱི་སྒྲིབ་པ་ལའོ། །བློ་ནཱ་ཏི། ཤྲཱི་ཎཱ་ཏི། བྷཾ་བརྒྱད་བཀག་ལའོ། །བྷཱ་ཎཱ་ཏི། རྀ་རྣམ་པར་འཇོམས་པ་ལའོ། །རྀ་ཎཱ་ཏི། ཛཾ་ལང་ཚོ་ཉམས་པ་ལའོ། །ཕྲོ་ཎཱ་ཏི། །ཚྭ་ཚུལ་ལའོ། ་། ཉྩཱནཾ་ཎཱ་ཏི། གྲྭ་སྒྲ་:ལའོ། །གྲི་ཎཱ་ཏི། གྲོ་ྋཙཱ་་ལང་ཚོ་ཉམས་པ་ལའོ། །གྲ་ཧི་ཙཱ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་སཱ་ར་ཎཱའོ། །དེ་མཐའ་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །དེ་ནས་ཐུང་ངུའོ། །ཛི་ནཱ་ཏི། རྀ་འགྲོ་བ་ལའོ། །རཱི་ཎཱ་ཏི། ཛཱ་འཇུག་པ་ལའོ། ྋ། རཱི་ཎཱ་ཏི། ལཱི་སྦྱོར་བར་་་།
