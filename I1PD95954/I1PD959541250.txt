སུ་མ་ནས་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ཨནྡྷ་དང་ཨས་མཐའི་ཡང་བྱིངས་མ་ཡིན་པའི་སི་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ས་ཡིག་གི་རྣམ་པར་་གཅད་བྱའོ། ?། སུ་མ་ནཱ། སུ་མ་ན་སོ། །སུ་མ་ན་ས། སུ་མ་ན་སཾ། སུ་མ་ན་སོ། ད་སུ་མ་ན་ས། ་སུ་མ་ན་སཱ། ས་ཡིག་གི་རྣམ་པར་གཅད་བྱ་སྟེ། ཨ་དང་སྒྲ་ལྡན་དག་གི་འང་ངོ་ཞེས་པས་ཨུ་ཉིད་དུ་འགྱུར་ཞིང་། །ཨུའི་ཡི་གེ་ལ་ཨོ་སྟེ་ཕ་རོལ་ཡང་དབྱིའོ། ་། སུ་མ་ནོ་ཧཱུཾ། །སུ་མ་ནོ་བྷི། སུ་མ་ན་སེ། སུ་མ་ནོ་ཧཱུཾ། །སུ་མ་ནོ་བྷུ། སུ་མ་ན་ས། སུ་མ་ནོ་ཧཱུཾ། སུ་མ་ནོ་བྷུ། སུ་མ་ན་ས༑པསུ་མ་ན་སོ། སུ་མ་ན་སཾ། སུ་མ་ན་སི། སུ་མ་ན་སོ། སུཔ་ལ་ནི་རྣམ་པར་གཅད་བྱ་སྟེ། །ཤ་འམ་ཥ་འམ་ས་ལ་ཡང་ཕ་རོལ་གྱི་གཟུགས་སུ་ཡང་ནའོ་ཞེས་པས་ཕ་རོལ་གྱི་གཟུགས་ཉིད་དུའོ། །སུ་མ་ནསུ། ཀུན་དུ་བོད་པ་ལ། སཾ་བུདྡྷི་མ་ཡིན་པ་ལའོ་ཞེས་པ་སྤྱིར་བཏང་བ་ལས། ་ཨམ་ཏུ་དང་ཨས་མཐའི་ཡང་བྱིངས་མ་ཡིན་པའི་སི་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དུ་འགྱུར་བ་མ་ཡིན་ནོ། །ཧེ་སུ་མ་ན། ཡ་ཧེ་སུ་མ་ན་སོ། ཧེ་སུ་མ་ན་ས། །དེ་བཞིན་དུ་སུ་ཤོ་ཏས་དང་ཨཔྤ་རས་ཀྱི་སྒྲ་སྟེ། ཕལ་ཆེ་བས་མང་པོའི་ཚིག་གི་ཡུལ་ཅན་ཁོ་ནས་ཨཔྤ་རས་ཞེས་པའོ། །བྷཱས་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། བྷཱ། བྷཱ་སོ། བྷ་ས། བྷ་སཾ། བྷཱ་སོ། བྷཱ་ས། ་བྷཱ་སཱ། ས་ཡིག་གི་རྣམ་པར་གཅད་བྱ་སྟེ། སྒྲ་ལྡན་ལ་་དབྱིའོ་ཞེས་པས་རྣམ་པར་གཅད་བྱ་དབྱིའོ། །བྷུ་བྷཾ། བྷ་བྷི། ཞེས་པ་ལ་་སོགས་པའོ། །སུཔ་ལ་ས་ཡིག་རྣམ་པར་གཅད་བྱར་བྱས་པ་ན། །ཤའམ་ཥའམ་ས་ལའང་ཕ་རོལ་གྱི་གཟུགས་སུ་ཡང་ནའོ་ཞེས་པས་ཕ་རོལ་གྱི་གཟུགས་ཉིད་དོ། །བྷསྨཱ༑སུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་བྷཱ། ཧེ་བྷཱ་སོ། ཧེ་བྷཱ་ས། །།།།ཨཱང་ཤཱ་སུ་འདོད་པ་ལ་སྟེ། ཞུང་ཡིག་ནི་པ་རི་ཨཔ། ཨཱང་སྦྱོར་བ་ལ་ལྔ་།
