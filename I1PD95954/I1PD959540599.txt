་་་བསྐྱོ་བ་ལ་ཕ་རོལ་དུ་ཨིཊའོ། །གཉིས་བརྗོད་དོ། །ཏཱཾ་ཞེས་པ་འདིའི་རྀཏ་ོམཐའ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཡོན་ཏན་ནོ། །འདིས་ཨེ་ཉིད་དང་ཨ་བྷཱུ་®ས་དབྱི་བ་་དག་གོ། །ཡོན་ཏན་མེད་པ་ཕ་རོལ་དང་པ་:རོཉགཱ་ལ། ཨིནྡྷི་ཐནྡྲི་གྲནྡྲི་ཞེས་་པ་ལ་སོགས་པས་ཨ་ནུ་ཎཾ་ག་དབྱིའོ། །ཐལ་ལ་ནི་ཐལ་ལ་ཡང་ཞེས་པ་འདིས་སོ། །འོ་ན་དནྟི་ཨ་ནུ་ཏཾ་ག་ཕྱིས་པ་གྲུབ་པ་ནི་ཨེ་ཉིད་དང་ཨ་བྷཱ་ས་དབྱི་བ་དག་གྲུབ་་བོ་ཞེས་པ་ཅི་ཡིན་ཏེ། །དམྷི་གཟུང་བས་སོ་ཞེ་ན། །བདེན་ཏེ་མཚན་ཉིད་ལས་བྱུང་བ་ཉིད་ལ་གསལ་བྱེད་གཅིག་:པུ་པའི་དབུས་ཉིད་དུ་ཅི་ལྟར་གྲུབ་པར་འགྱུར། ཛཱི་དང་གྲྭ་མི་དང་ཏྲ་སི་དང་ཕ་ཎི་དང་ར་ཛི་དང་བྲཱ་ཛི་དང་བྲུ་སི་དང་:བླ་སི་་་དང་བྱ་མི་དང་སོགས་ྋནི་རྣམས་ཀྱི་ཨ་ཡིག་ཨེ་ཉིད་དུ་ཡང་ན་བཤད་པར་བྱ་སྟེ། ཨ་བྷཱ་ས་དབྱི་བའང་ངོ༌། །ཛཱིག་ཅེབ་ལའོ། །།ཛེ་ར་ཏུ། ཛེ་:ཏུ། ཛེ་རི་ཐི། ཛ་ཛ་ར་ཏུ། ་ཛ་ཛ་ཏིུ། ཛ་ཛ་རི་ཐ། ཛཱཏ་མཐའ་རྣམས་ཀྱི་ཞེས་པས་་་་ཡོན་ཏན་ནོ། །ཕྱོགས་གཅིག་ལ་ཨེ་ཉིད་དང་ཨ་ཧྥུ་ལིས་དབྱི་བ་དག་གོ། །བྷ་མུ་གཡོ་བ་ལའོ། །བྷ་མེ་ཏུ། བྷེ་མུ། །བྷེ་མི་ཐ༄༅། །བ་བླ་:མ་ཏུ?། :བ་བྷ་མུg༑བ་བྷ་མི་ཐ། ཏྲེ་ས་ཏུ། ཏྲེ་སུ། ་ཏྲེ་སི་ཐ། ་ཏ་ཏྲ་ས་ཏུ། ཏ་ཏྲ་སུ། །ཏ་ཏྲ་སི་ཐ། ཕེ་ཎ་ཏུ། །ཕེ་ཎུ། ཕེ་:ཎ་ཐི། པ་ཕ་ཎ་ཏུ། པ་ཕ་ཎཱཉྩི། །་སེཔ་ཕ་ཎི་ཐ། རེ་ཛ་ཏུ། རེ་ཛུ། རེ་ཛི་ཐ། རེ་ཛེ། རེ་ཛཱ་ཏེ། །རེ་ཛི་རེ། ར་རཱ་ཛ་ཏུ། ར་རཱ་ཛུ། ་་:ར་རཱ་ཛི་ཐ། ལྷོར་རཱ་ཛེ། :·རཱ་རཱ་ཙྪཱཛཱ་ཏེ། ར་རཱ་ཛི་རེ། :བྷཱ་ཛི་གསལ་བ་ལའོ། །།བྷེ་ཛེ། བྷེ་ཛཱ་ཏེ། བ་:བྷུ་ཛེ། བ་བྷཱ་ཛཱ་རེ། །བྷ་སྲི་འབར་བ་ལའོ། །བྷེ་སེ། བྷེ་:སཱ་ཏེ། བ་་བྷཱ་སེ། བླ་སྲི་ྋགསལ་བ་ལའོ། །བློ་སེ། བྷེ་སཱ་ཏེ། །བ་:བྷཱ་སེ། བ་བྷཱ་སཱ་ཏེ༑། བྱ་མུ་སཱན་སྒྲ་ལའོ། །བྱེ་མ་ཏུ། །བྱེ་མི་ཐ། ཤ་བྱ་མ་ཏུ། ཤ་བྱུ་མི་—།
