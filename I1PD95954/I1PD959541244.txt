དྷེ་ནུ་བྷུ། དྷེ་ནཱ། བྷེ་ནོ། དྷེ་ནཱི། དྷེ་ནུ་ནཱཾ། ཧེ་ཧཱུྃ། དྷེ་ནོ། དྷེ་ནོ༑ཥ་ཉིད་དོ། །དྷེ་ནུ་ཞུ། ཀུན་དུ་བོད་པ་ལ་ན། ཧེ་དྷེ་ནོ། ཧེ་དྷོ་ནུ༑ཧེ་དྷེ་ན་ཝ། དེ་བཞིན་དུ། ཏ་རུ། ར་ཛུ། ཀརྐུ། ཏརྐུ་ལ་་སོགས་པ་རྣམས་སོ། །ནད། ་:ན་་སྭ་སྲོའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ན་དཱི་ལ་སོགས་པའི་མདོས་རིཏ་མཐའ་ཉིད་ནའང་ཨཱིའི་རྐྱེན་མེད་ཏེ། །རྐྱེན་མེད་ནའང་བུད་མེད་ཉིད་དུ་མངོན་་པར་བརྗོད་པའི་ཕྱིར་རོ། །སི་ལ་ཨཱ་སྟེ་སི་དབྱི་བའང་ངོ་ཞེས་པས་ཨཱ་རུ་འགྱུར་་་་ཞིང་སི་དབྱི་བའང་ངོ༌། །སཱ་སཱ། སཱ་སྲི་ལ་སོགས་པ་རྣམས་ཀྱི་འང་ངོ་ཞེས་པས་ཨཱར་རོ། །སཱ་སཱ་རོ། སཱ་རཱ་ར། སཱ་སཱ་རཾ། སཱ་སཱ་རོ། ཤས་ལ་ཨགི་་བཞིན་ནོ་ཞེས་པས་ཨགྷི་དང་འདྲ་བར་འགྱུར་ཞིང་། །ཤས་ཀྱི་ཨ་ཡིག་སྟེ་སའང་་་བུད་མེད་མ་ཡིན་པ་ལ་ནའོ་ཞེས་པས་བཀག་པའི་ཕྱིར་ན་ཉིད་དུ་མི་འགྱུར་རོ། །སྭ་སཾ། ཁྱད་པར་གཞན་རྣམས་པི་ཏྲིའི་སྒྲ་དང་འདྲའོ། །།སཱ་ས། སྭ་སྭ་ཧཱུཾ། །སྭ་སྲི་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །།།།མཱ་ཏྲཱའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། འདིར་ཡང་སྔ་མ་བཞིན་དུ་ཨཱིའི་རྐྱེན་མི་འགྱུར་རོ། །མཱ་ཏཱ། །ཟླ་སྒུཊ་ལའང་ངོ་ཞེས་པས་ཨར་རོ། ་། མཱ་ཏ་རོ། མཱ་ཏ་ར། མཱ་ཏ་རཾ། མཱ་ཏ་རོ། ་བུད་མེད་མ་ཡིན་པ་ལ་ཞེས་པས་བཀག་པའི་ཕྱིར་འདིར་ཡང་ན་ཡིག་ཏུ་འགྱུར་བ་མ་ཡིན་ནོ། །མཱ་ཏཱཾ། མཱ་ཏཱ། མཱ་་ཀྲྀ་ཧཱུཾ། མཱ་ཏྲ་བྷི། ཞེས་པ་ལ་སོགས་པ་རྣམས་སོ། །དེ་བཞིན་དུ། དུ་ཧི་ཏྲི། ན་ནནྣཾ། ཛཱ་མཱ་ཏྲ། ལ་སོགས་པ་རྣམས་སོ། །།།སྤྱི་ནོའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །ནོ། དབྱངས་ལ་ཨོ་ཨཱ་ཝའོ། །ནཱ་ཝོ། ནཱ་ཝ། ནཱ་ཝཾ། ནཱ་ཝོ། ནཱ་ཝ། ནཱ་ཝ། ནོ་བྷཾ། ནོ་བྷི། །།
