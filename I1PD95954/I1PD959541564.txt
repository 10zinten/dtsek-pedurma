ཏི་ལ་སོགས་པའི་མཐའི་བྱ་བ་རྣམ་པར་དཔྱད་པ་ཞེས་བྱ་བ། ཡང་ན་རིང་པོའོ། །པ་རི་ཥྛ་ཏེ། བ་རཱི་བྱ་ཏེ། པྲ་བ་རི་བྱ་ཏེ། པྲ་བཱ་རཱི་ཥྛ་ཏེ། ཕྱོགས་:ཏྲིང་། །ཏ་རི་ཥྛ་ཏེ། ཊི་ཏ་རཱི་ཥྛ་ཏེ། །ཀྲིཡཾ། ཀ་རི་ཥྛ་ཏེ། ཀ་རཱི་ཥྛ་ཏེ། གྲི༄༅། ངེས་པར་གནོན་པ་ལའོ། །དབྱངས་ལ་ཡང་ནའོ། །ཞེས་པས་རེ་པའི་ལ་ཉིད་དོ། །ནི་ག་ལི་ཉྩ་ཏེ། ནི་ག་ལཱི་ཥྛ་ཏེ། ནི་ག་རི་བྱ་ཏེ། ནི་ག་རཱི་ཥྛ་ཏེ། པྲི། །པ་རི་ཥྛ་ཏེ། པ་རཱི་ཥྛ་ཏེ། རི། །ཨ་རི་ཥྛ་ཏེ། ཨ་རཱི་ཥྛ་ཏེ། །ཀྲི?། ཀ་རི་ཥྛ་ཏི། ཀ་རཱི་ཥྛ་ཏི། ཀ་རཱི་ཥྛ་ཏེ། ཀ་རཱི་ཥྛ་ཏེ། དེ་བཞིན་དུ་བྲི་ཉའོ། །?། ཤཀྱཱ་ཏི། །ཤ་ཀ། ཤཀྵ་ཏེ། ཌཌུ་པ་ཙ་ཀ་ཏེའཚེད་པ་ལའོ། །ཙ་སྡེའི་ཀིའོ། །པགྱུ་ཏི། པདྨ་ཏེ། པ་ཙ་གསལ་བར་བྱེད་པ་ལའོ། །པདྨ་ཏེ། ཌཔ་ཙ། །པཀྵ་ཏི། །བྷུཉ། ་བྷུའི་བ་ཙིའོ། །བགྲུ་ཏི། བསྒྱུ་ཏེ། །བི་ཙི་ར། ནཱ་མི་ཨུ་པ་དྷའི་ཡོན་ཏན་ནོ། །བེ་ཀླུ་ཏི། བེ་གྲུ་ཏེ། སི་ཙ། སེ་གྲུ་ཏེ། རི་ཙིར། །རེགྱུ་ཏི། རེཀྵ་ཏེ། ་མུཙྪི། ་མོཀྵ་ཏི། མོཀྵ་ཏེ། པྲ། །ཚ་ཤ་དག་གི་འང་ངོ༌། །ཞེས་པས་ཥ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ།  ོ། ཞེས་པས་ཀ་ཉིད་དོ། ས་། པྲཀྵ་ཏི། ཡུ་ཛིར། ཙ་སྡེའི་་་་ཀིའོ༑་། ཡོ་གླུ་ཏེ། ཡུ་ཛ། ་ཡོཀྵ་ཏེ། ཞོ་། རུ་ཛོ། རོགྲུ་ཏི་རན་ཛ། རཾགླུ་ཏི། བྷུ་ཛོ། བྷོཥྛ་ཏི། ཊ་་:བྷུ་ཛ། བྷོསྒྱུ་ཏི། ་བྷུ་ཛའི་ཟས་ལ་རུ་ཙ་སོགས་སོ། །བྷཀྵ་ཏེ། བྷ་ཛ། བྷཀྵ་ཏི། བྷནྡྲ་ཡི། གྷོ་གླུ་ཏི། ་སམྦྷ། སཾགླུ་ཏི། ཏྱཱ་ཛ། ཏགླུ་ཏི། བྷསྔལ། སྔ་མ་བཞིན་རྣམ་པར་བཤད་པ་ལ་ཤྲཱི་ཛ་རུ་བསྒྱུར་རོ། །།བྷཾ་ཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །བྷནྡྷུ་ཏི། བྷརྒྱུ་ཏེ། བརྩེགས་པའི་དང་པོ་ས་དབྱིའོ། །།བྷ་སྔ་ལ་བྷཱཛ་ཉེ་བར་མཚོན་པ་ཉིད་ཀྱི་ཕྱིར་་ཥ་ཉིད་དོ། །བྷཀྵ་ཏི། 《བྷཀྵ་ཏེ། །ཡ་ཛ། ཡགྱུ་ཏི། ཡགྱུ་ཏེ།  ོམསྔ། ན་ཡིག་ཨཱ་ག་མ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །མོ་གླུ་ཏི། སུཾ་ཛ། སྲི་ཛི་དྲི་ཤ་།
