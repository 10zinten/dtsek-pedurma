བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི། ཁ་དབྱིའོ། །ཀྲོསྦྱིན། ཀྲོཏྟཱ་ནཱ། ཀྲྀ་ཙ་མཐའི་ཕྱོགས་ལ། ལེའི་ཡི་གེ་རའོ་ཞེས་པས་ར་ཉིད་དོ། །ཀྲོ། གསལ་བྱེད་ཀྱི་ཡུལ་ཉིད་ལས་ནི་ཏུན་གྱི་མཐའ་ཁོ་ན་རབ་ཏུ་སྦྱོར་རོ། །ཀྲོཥྛ་བྷཱུམ། ཀྲོ་བྷི། ་ཀྲོཥཱ་བེ། ::ཀྲོཊྚེ།  ིཀྲོཥྛ་བྷཱུམ། ཀྲོཥྛ་། ང་སི་ངས་དག་གི་ཨ་ཨུའོ་ཞེས་པ་སྤྱིར་བཏང་ནས། རྀཏ་མཐའ་་ལས་སྔ་མ་དང་བཅས་པའོ་ཞེས་པས་དྲ་སི་ངས་དག་གི་ཨ་ཡིག་རཾ་ཡིག་དང་ལྷན་་་ཅིག་ཨུ་རུ་ཐོབ་བོ། །ཀྲོཟླུ། ཀྲོཌོ། ཀྲོཥྛ་བྷཱུམ། ་༼ཀྲོཥྛ་བྷུ། ་《ཀྲོ། ཀྲོgཀྲོཊྚ། ཀྲོཏྟཱ་ནཱམ། ་ཀྲོབྦྷ་ཎཱམ། ཀྲོཏྠོ། ༼ང་ལ་ཨར་རོ་ཞེས་པས་ཨར་རོ། །ཀྲོཥཱ་རི། ཀྲོཊྚ། ཀྲོཊྛོ། ཀྲོཥྛ་ཏུ། །ཀུན་དུ་བོད་པ་ལ་ནི། མ་ནིང་ལས། །སི་ཨམ་དག་གི་དབྱི་བ་དང༌། ཏུན་རྐྱེན་ནོ། །ཧེ་ཀྲོཌོ། ཧེ་ཀྲོཥཱ་རོ། ཧེ་ཀྲོཏྠ་ར། །བུད་མེད་དང་པོ་ལ་ནི་མདོས་ཨཱི་རྐྱེན་ནོ། །།ཀྲོཊྛི། །སྭ་ཡཾ་བྷ་ཝ་ཏི་ཞེས་པ་ལ། གཞན་རྣམས་ཀྱང་མཐོང་ངོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ཀྭིཔྲ་འང་ངོ་ཞེས་པས་ཀྭིཔྲ་འང་སྟེ། ཉེ་བའི་ཚིག་གིས་ཚིག་སྡུད་དོ། །པ་ཡིག་ནི་བྱིངས་ཀྱི་པ་རྗེས་སུ་འབྲེལ་པ་ལ་ཏ་མཐའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ལ། ཝ་ཡིག་ནི་སྦྱར་བ་མ་ཡིན་པའི་ཝི་དབྱིའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ཞིང༌། ཀ་ཡིག་ནི་ཀ་ལ་ཡཎ་བཞིན་ཡང་སྟེ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །རྟགས་ཀྱི་་མིང་ཅན་ལ་སི་ལ་སོགས་པ་རྣམས་ཏེ། སི་ལ་ནི། སྭ་ཡཾ་བྷུ། དབྱངས་ལ་་གཉིས་ཀྱི་ཚིག་ལ་སོགས་པ་ལ་ནི། འོ། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དུ་ཐོབ་མོད། བྷུའི་ཝནྟཱ་བྷུ་དང་པཱུ་ནར་བྷུ་མ་ཡིན་པའོ་ཞེས་པས་བྷུ་ནི་ཨུ་ཝ་རུ་ཐོབ་བོ། ་། སྭ་ཡཾ་བྷུ་བོ། སྭ་ཡཾ་བྷུ་ཝ། སྭ་ཡཾ་བྷུ་ཝམ། སྭ་ཡཾ་བྷུ་ལོ། སྭཱ་ཡཾ་བྷུ་ཝ། །སྭ་ཡཾ་བྷུ་ཝཱ། །སྭ་ཡཾ་བྷུ་བྷཱུམ། སྭ་ཡཾ་བྷུ་བྷི། སྭ་ཡཾ་བྷུ་ཝེ། སྭ་ཡཾ་བྷུ་བྷཱུམ། སྭ་ཡཾ་བྷུ་བྷུ། སྭ་ཡཾ་བྷུ་ཝ། སྭ་ཡཾ་བྷུ་བྷཱུམ། སྭ་ཡཾ་བྷུ་བྷུ། །
