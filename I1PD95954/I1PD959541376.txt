རིཀྵ༑ཀྲི་བི། ཏེ་ཙི་རི། །ཛི་རི། དཱ་ས། དྲུ་ཧ། འཚེ་བ་ལའོ། །ཏིགཽ་ཏི། ཏིགཽ་ཏི། སདྡྷོ་ཏི། རིདྡྷོ་ཏི། ཨིཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ཀྱི་ཕྱིར་ནུ་ཨཱ་ག་མའོ། །དྷི་ནཱི་དང་ཀྲྀ་ཎི་དག་གི་དྷི་དང་ཀྲྀ་ཡང་ངོ་ཞེས་པ་བཤད་པར་བྱའོ་ཞེས་པས་ཀྲྀའི་དངོས་པོའོ། །ར་ཥ་རེའི་ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་་སོགས་པས་ཎ་ཉིད་དོ། །ཀྲྀ་ཎོ་ཏི། ཀྲྀ་ཊུ་ཏ། བི་ཀ་ར་ཎའི་ནུའི་བ་ཡིག་གོ། །ཀྲྀ་ཎཱནྟི། ཀྲྀ་ཎོ་ཏི། ཀྲྀ་ཊུ་ཐ། ཀྲྀ་ཆུ་ཐ། ་ཀྲྀ་ཎོ་མི། ཨུ་ཡིག་བ་མ་དག་ལ་ཡང་ན་དབྱིའོ། །ཀྲྀ་ཎ། ཀྲྀ་ཙྪ། ཙི་རི་ཎོ་ཏི། ཛི་རི་ཎོ་ཏི། དཱ་སྣ་ཏི། དྲུ་སྒོ་ཏི། ཉི་བྷཾ་ཥ་སྤྱི་བརྟོལ་་བ་ལའོ། ''། ཨཱ་ཡིག་ནི་ཨཱཏ་རྗེས་སུ་་་འབྲེལ་པ་ལས་ཀྱང་ངོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །ཥ་ལས་ཊ་སྡེའི་ནའི་ཎ་ཉིད་དོ།  ི། ཎྜི་ཏི། ཛྷཱཉྩ་ཏ། སྔ་མ་བཞིན་པ་ཉིད་དུ་མི་འགྱུར་བར་ཨུཔ་ཏུ་བསྒྱུར་རོ། །ཛྷཥཱ་བནྟི། ཎྜི་ཏི། ཛཱིཙྩོ ིཐ། ཛྷ་ཐ། ཛཱིཏྟོ་མི། སྔ་མ་བརྩེགས་པ་ཉིད་ལས་ཨུ་ཡིག་བ་མ་དག་ལ་ཡང་ན་དབྱི་བ་མེད་དོ། །དྷིཥྛ་བ། དྷིཥྛ་མ། །དནྡྷ་ངན་གཡོ་བ་ལའོ། ་། ཨིཏ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་རྣམས་ཀྱི་ཡོན་ཏན་མེད་པ་ལ་ཨ་ནུ་ཏཾ་ག་དབྱིའོ། །དབྷཱ་ཏི། རཾ་དྷུ་འཕེལ་བ་ལའོ། །རྀ་དྷཱ་ཏི། དྷི་བི་དང་དྲི་བ་མཛའ་བ་ལའོ། །ཨི་ཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་་ན་ཨཱ་ག་མའོ། །དྷི་ནཱི་དང༌། །ཀྲྀ་ཎི་དག་གི་དྷི་དང་ཀྲྀ་ཡང་ངོ་ཞེས་:པ་བཤད་པ་བྱའོ་ཞེས་པས་དྷིའི་དངོས་པོའོ། །།དྷི་ན་ཏི། དྷི་ནུ་ཏ། བི་ཀ་ར་ཎའི་ནུའི་བ་ཡིག་གོ། །།དྷི་ནཱནྟི། དྷི་ནོ་ཏི། དྷི་ནུ་ཐ། དྷི་ནུ་ཐ། དྷི་ནོ་མི། ཨུ་ཡིག་བ་མ་དག་ལ་ཡང་ན་ཕྱེདབྱིའོ། །དྷི་ནཱ། །དྷི་ནུ་:བ། དྷིམྨས། །དྷི་ནུ་མ། ཀྲྀ་པི་དྲ་སོགས་སོ། །ཀླུ་དྲ་སོགས་རྣམས་ཀྱི་འང་ཎ་ཉིད་དུ་མི་་འགྱུར་རོ། །ཏྲསྒོ་ཏི། ཀྲྀ་ཏ། ཀྲྀཔཱུ་མ། པ་རསྨིའི་བཤད་པ་རྣམས་སོ། །།
