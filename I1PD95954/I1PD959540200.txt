སྲ་ཛི་དྲི་ཤོ་རཱ་ག་:མོ་ཀཱ་རསྭཱ་རཱརྦ་རོ་དྷུ་ཊི་གུ་ཎ་ཧྲི་དྷིསྠཱ་ནེ། །སྲི་ཛ་ནི་རྣམ་་་པར་གཏོང་བའོ། །དྲི་ཤིར་ནི་ལྟ་བའོ། །དེ་དག་གི་དབྱངས་ཀྱི་ཕ་རོལ་དུ་ཨ་་་ཐོབ་པར་འགྱུར་བ་སྟེ། གུ་ཎ་དང་བྲི་དྷའི་གནས་ཀྱི་དྷུ་ཊི་ཕ་རོལ་ན་ཡོད་ནའོ། །སཾ་ཛ་ལ་ཤུ་ནི་ཏཱ་སྦྱར་བས་མདོ་འདིས་ཨཱ་ཐོབ་པ་སྟེ། རམ་སྲི་བཎའོ། །ཛས་ཥ་ཐོབ། །ཏས་ཊ་ཐོབ་པོ། །།སྲིཊྚི། དྲྀ་ཤི་ལ་ཏཱ་སྦྱར་བ་གོང་མ་བཞིན། དྲའོ༑། སྲི་ཛ་ལ་རྒྱལྟ་ནཱི་དི་སྦྱར་བས་གོང་དུ་འབྱུང༌། མདོ་འདིས་བར་་དུ་ཨ་ཐོབ་སྟེ། རཾ་རཾ་ཁེབཎྜས་སིཙ་དང་ཨི་སྦྱར་བས་ཨ་བསྲིངས་པའོ། ་། ཨ་སཱ་ཀྱི་ཏིའོ། །དྲི་ཤི་ལ་དི་སྦྱོར་ཏེ་གོང་མ་བཞིན། ཨ་དཱཀྐི་ཏའོ། །དཱི་:དེཉནྟོ་ཡ་ཀསཱ་རཱ་དཱ་བ་གུ་ཎེ། དཱང་ཀྲུ་ཡ་ཞེས་པ་འདིའི་ཕ་རོལ་དབྱངས་ཐོག་མ་ན་ཡོད་པའི་གུ་ན་ཡོད་ན་ཡ་ཐོབ། དི་ལ་ཨུ་པ་གོང་དུ་སྦྱར། ་མཐར་ཨེ་སྦྱར་་་་བས་དི་གཉིས་པ་བྱས། །བར་དུ་ཡ་ཐོབ། ཨེ་སྦྱར། ཨུ་པ་དེ་དྲི་ཡེའོ། །ཨཱ་ལོ་པོཞསཱརྦ་:དྷ་ཏུ་*ཀེ། དྷཱ་ཏུའི་ཨ་དབྱི་བར་བྱ་བ་སྟེ། དབྱངས་ཐོག་མ་ན་་་་་ཡོད་པའི་ཨ་གུ་ཎེ་སཱརྦ་དྷཱ་ཏུ་ཀ་ཕ་རོལ་ན་ཡོད་ནའོ། །པཱ་ལ་ཨ་ཏུས་སྦྱར་བས། པཱ་གཉིས་པ་བྱ་སྟེ་བསྟུངས། འོམདོ་འདིས་པཱ་འོག་མའི་ཨཱ་ཕྱི་སྟེ། །ཨ་ཏུས་ཀྱི་ཨ་དང་སྦྱར། །པ་བ་ཏུའོ༄༅། །དཱ་ལ་ཨ་ཏུས་སྦྱར་བ་གོང་བཞིན། ད་ད་ཏུའོ། །:ཨི་ཊི་ཙི། །་ཨའི་ཕ་རོལ་ན་ཨིཊ་ཡོད་ན་ཨཱ་དབྱི་བར་བྱ་སྟེ། པཱ་ཅེལ་ཐལ་སྦྱར། ར་བར་དུ་ཨིཊ་སྦྱར་བས། གོང་དུ་པ་གཉིས་པ་བྱས། དེ་བསྟུངས། འོག་མའི་ཨ་ཕྱི། ༼ཨི་དང་སྦྱར། པ་པི་ཐའོ། ལ། དཱ་ལ་ཨིན་དང་ཐལ་སྦྱར་བ་གོང་བཞིན། ད་དི་ཐས་སོ། །དཱ་མཱ་གཱ་ཡ་ཏི་པི་བ་ཏིསྦི་སེམྱ་ཏི་ཛ་ཧཱ་དཱི་ཨིན་་མཱི་ཀཱ་རོ་བྱཉྫ་ནཱ་:དོ། །ོ། དཱའི་མིང་ཅན་རྣམས་མཱ་ནི་འཇལ་བའོ། །གོ་ནི་་་སྒྲའོ༑་ས་། པཱ་ནི་བཏུང་བའོ། །ཥ་ནི་འགྲོ་པ་ལ་མི་འཇུག་པ་སྟེ། འདུག་།
