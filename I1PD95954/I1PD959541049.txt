བསྟན་འགྱུར། སྒྲ་མདོ། །ཤིའིཧཱ་ཧ། ཧཱ་ཧི། ཧཱ་ཧཱ་སིུ། ཀུན་དུ་བོད་པ་ལ་ཁྱད་པར་མེད་དེ། ཧེ་ཧཱ་་་ཧཱ༑ཧེ་ཧཱ་ཧོ། ཧེ་ཧཱ་ཧཱདེ་བཞིན་དུ་ཨཱ་ལཿཡིག་གི་མཐའ་ཅན་རྣམས་ཀྱང་་རྗེས་སུ་འབྲངས་ནས་རྟོགས་པར་བྱའོ། །གལ་ཏེ་ཡང་ཚིག་ལེའུར་བྱས་པ་:ལས་ཨ་ཡིག་གི་མཐའ་ཅན་དངོས་སུ་མ་བརྗོད་པ་དེ་ལྟ་ནའང་གཞན་རྣམས་ཀྱང་ཕྱོགས་་འདིས་ཤེས་པའི་ཚིག་ལས་དོན་གྱིས་བརྗོད་པ་ཁོ་ནའོ་ཞེས་བསྟན་ཏོ། །ཨི་ཡིག་གི་མཐའ་ཅན་རྣམས་བརྗོད་པར་བྱའོ། །ཨགྣིའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ༑། དང་པོ་གཅིག་གི་ཚིག་གི་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ། །ཨགྣི༑ཨོ་ཡིག་ནི་སྔ་མའོ་ཞེས་པས་ཨགི་ལས་ཕ་རོལ་གྱི་ཨོ་ཡིག་སྔ་མའི་་དབྱངས་སུ་འགྱུར་རོ། །མཚུངས་པ་འདྲ་བའི་ཡི་གེ་ལ་ཞེས་པ་ལ་སོགས་པས་རིང་པོ་དང་ཕ་རོལ་དབྱི་བ་དག་གོ། །ཨགྣི། ཛས་ལ་ཨི་ཨེཏ་ཞེས་པས་ཨི་ཨེཏའོ། །ཨེ་ཨ་ཡའོ། །ཨགྷ་ཡ། ཨིཏ་ཨུཏ་ཨབྷིའོ་ཞེས་པས་ཨགྣིའི་མིང་སྟེ། ཨགྣི་ལས་ཨམ་གྱི་ཨ་ཡིག་གོ་ཞེས་པས་ཨམ་གྱི་ཨ་ཡིག་དབྱིའོ། །ཨངྒིམ། ཨབྷི། ཤས་ཀྱི་ཨ་ཡིག་སྟེ་སའང་བུད་མེད་མ་ཡིན་པ་ལ་ནའོ་ཞེས་པས་ཨ་ཡིག་སྔ་མའི་་དབྱངས་སུ་འགྱུར་ཞིང་། སའི་ཡང་ན་ཡིག་སྟེ། རིང་པོ་དང་ཕྱི་མ་དབྱི་བའང་ངོ༌། །ཨབྷིན། :：ཊ་ནཱའོ་ཞེས་པས་ཨགི་ལས་ཕ་རོལ་གྱི་ཊཱིནཱ་རུ་འགྱུར་རོ། །ཨགྣི་ནཱ། ཨགྣི་བྷཱུམ། །ཨགྣི་བྷི། དེ་ལའོ་ཞེས་པས་ཨི་ཨེཏའོ། །ཨེ་ཨ་ཡའོ། །ཨག་ཡེ། ཨགྣི་བྷཱུམཎི། ཨགྣི་བྷུ། །ང་ཡི་ངས་དག་གི་ཨ་དབྱི་བའང་ངོ་ཞེས་པས་ཨི་ཨེཏ་སྟེ་ཨ་དབྱི་བའང་ངོ༌། །ཨངྒེ། །ཨགྣི་བྷཱུམ། ་ཨགྣི་བྷེུ། ཨངྒེ༑་ཨིའི་ཡི་གེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཡ་སྟེ། ཕ་རོལའང་དབྱི་བར་བྱ་བ་མ་ཡིན་ནོ། །ཨམྦྷོ། ཨཱམ་ལའང་ནུའོ། །ནུ་དང་བཅས་པའི་ཨཱམ་ལ་་་རིང་པོའོ། །།ཨནྡྲི་ནཱམ། །ང་སྔ་མ་དང་བཅས་པ་ཨོའོ་ཞེས་པས་ཨི་ཡིག་གཉིས་་།
