དང་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ལིའི་ན་ཀྲི་ཡ་ཏི་པ་ཏི་སྦྱར་བས་སྔ་མ་བཞིན། ་ཨ་དྨ་གི་ཥྛ་ཏའོ། །ཡང་ན་ཨིང་ལ་ཏ་་་་སྦྱར༑གོང་དུ་ཨ་དྷི་སྦྱར། ?སི་ཙ་སྦྱར་ཏེ་ཨི་དང་ཙ་ཕྱི། སྭ་ར་ཨ་དྷི་ནམ་གྱི་མདོས༑ཨིས་ཨི་ཐོབ། ཨ་དྷིབྦའོ། །ཨིད་ལ་མྱ་ཏ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་དྷིཥྛ་ཏའོ་། :ལ་པ། །།ཟླ། དང་ཀླུ་ཆུ། ས་ནཱི་ཤི་ངོ་ག་མི། ཨིན་དང་ཨིང་ལ་ག་མ་གཞག་པ་སྟེ། ཕ་རོལ་ན་སན་ཡོད་ནའོ། །ཨིཎ་སན་སྦྱར་བས་ག་མ་ཐོབ་སྟེ་ཨིཊ་དང་སྦྱར། ག་མ་་་གཉིས་པ་བྱས། མ་ཕྱི། པ་རྒྱ་ཝཎ་སྱའི་མདོས། ཨི་ཐོབ། ཌ་གས་ཛའོ། །ཛི་ག་མིཥ་ཏིའོ། །ཨིཎ་ལ་ཡན་སྦྱར་བས་ག་མ་ཐོབ། །གོང་དུ་ཨ་དྷི་བྱས། ག་མ་གཉིས་པ་བྱས་པ་ལ་སོགས་སྔ་མ་བཞིན། །ཨ་བྷི་ཛི་ཀཾ་ྋས་ཏེའོ། །ཁོང་དུ་ཆུད་པའི་དོན་གྱི་ཨིཎ་ལ་ག་མ་མི་འབྱུང་བ་སྟེ། །ཨིཎ་གྱི་གོང་དུ་པྲ་ཏི་བྱས། སན་སྦྱར་བས་སཱ་ར་ཨནྟ་:རྣམས་ཀྱི་མདོས་མི་བསྲིངས་ཏེ། ས་མ་ན་དིགྷ་སཱ་ར་དྲི་ཝིཡ་སའི་མདོས། ས་གཉིས་པ་བྱས། སརྱ་ཝརྟའི་མདོས་ས་གོང་མས་ཨི་ཐོབ། སས་ཥ་ཐོབ། ཨ་ར་ཐན་པྲ་ཏཱི་ཎི་ཥ་ཏིའོ། །ཨ་:ཏེཆུ་ར་སརླུ་དྷཱ་ཏུ་ཀེ། ཨས་བྷུ་ཞེས་པ་འདི་ལ་:བྷུ་གཞག་པ་སྟེ་ཕ་རོལ་ན་ཨ་སདྨ་ཙྪཱདྷཱ་ཏུ་ཀེ་ཡོད་ནའོ། །ཨ་ས་ལ་ཤཱ་:སྟ་ནཱིའི་ཏཱ་སྦྱར་བས་བྷུ་ཐོབ། ཨིཊ་སྦྱར་བས་:བྷུ་བྷོར་གྱུར། །ཨོ་ཨ་ཝ་དང་སྦྱར་རོ། །བྷ་ཝི་ཏའོལ། །ཨས་ལ་ཏི་སྦྱར་བས་བྷུ་ཏི། ཐོབ་པ་ལ་སོགས་་སྔ་མ་བཞིན་ནོ། །བྷ་ཝི་ཌུ་ཏིའོ། །ཨས་ལ་སྱ་ཏ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་་བྷ་ཝི་་ཏའོ། །ཨས་ལ་ཡཱཏ་ྋསྦྱར་བས་:བྷུའོ། །བྷཱུ་ཡཱཏའོས། །འདི་ནི་ཨ་ས་དི་སྦྱར་བས་བྷུ་ཐོབ། གོང་དུ་ཨ་སྦྱར་བའོ་། :ཌཎ། །སལ་ཎ་ཏོ། མ་ཉམ་བྲུ་ཁོ་བ་ཚོ། བྲུཉ་ནི་ཚིག་གསལ་བ་ཞེས་བྱ་བ་འདི་ལ་ཝ་ཙི་གཞག་པ་སྟེ། ཕ་རོལ་ན་ཨ་སརླ་'དྷཱིདྷཱ་ཏུ་ཀེ་ཡོད་ནའོ། །བྲུ་ལ་ཏཱ་སྦྱར་བས། ཝ་ཙ་ཐོབ། —།
