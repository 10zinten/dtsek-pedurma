ཁ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིག་ཀ༌པའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ལ། སྨོན་པར་བྱེད་པས་ན་ཨཱ་ཤི་སྟེ། སཾ་པད་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ཀྭིཔ་འགྱུར་ཞིང་ཤཱ་སི་ཨི་ཏ་ཨུ་པ་དྷཱའི་ཨན་དང་་་གསལ་བྱེད་དག་ལའོ་ཞེས་པས་འདིར་ཨཱང་ཤཱ་སུ་འདོད་པ་ལའོ་ཞེས་པ་འདིའི་ཡང་ཨི་ཉིད་ཏེ། །ཤཱ་སི་བ་སི་གྷ་སཱི་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། ཝ། དེ་ལས་་སི་ལ་སོགས་པ་རྣམས་ནི། རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ། ས་ཛུཥ་དང་ཨཱ་ཤིས་དག་གི་རའོ་ཞེས་པས་ར་ཉིད་དུ་བྱས་པ་ན། །ཨིར་ཨུར་དག་གི་ཨཱིར་ཨུར་་དག་གོ་ཞེས་པས་ཨཱིར་དུ་བསྒྱུར་ཞིང༌། །རེ་ཕ་ནི་རྣམ་པར་གཅད་བྱའོ། །།ཨཱ་ཤཱི། ཨ་ཤཱི་ཤོ། །ཨཱ་ཤི་ཥ། ཨཱ་ཤི་ཎཾ། ཨཱ་ཤི་ཤོ། ཨཱ་ཤི་ཥཱ། ཨཱ་ཤི་ཥཱ། ཨཱ་ཤྲཱི་ཧཱུཾ། ཨཱ་ཤྲིབྷི། ཨཱ་ཤི་ཥ། ཨཱ་ཤིདྡྷཾ། ཨཱ་ཤིནྡྷ། ཨཱ་ཤི་ཥ། ཨཱ་ཤིདྡྷཾ། ཨཱ་ཤིནྡྷ། ཨཱ་ཤི་ཥ། ཨཱ་ཤི་ཌོ། །ཨཱ་ཤི་ཏཱཾ། །#ཨཱ་ཤི་ཏི། »ཨཱ་ཤི་ཌོ། རྣམ་པར་དབྱེ་བའི་གསལ་བྱེད་ལ་རེ་ཕའི་མི་འགྱུར་རོ་ཞེས་གང་བརྗོད་པ་དེ་ནི་་འཁྲུལ་པའི་དོན་ཅན་དུ་འགྱུར་བའོ་ཞེས་ཀྱང་བརྗོད་པར་བྱ་སྟེ། དེས་ན་རེ་ཕའི་རྣམ་པར་གཅད་བྱ་དང་སུཔའི་ཥ་ཉིད་དུ་འགྱུར་ལ། ཤའམ་ཥ་འམ་ས་ལ་ཡང་ན་ཕ་རོལ་གྱི་གཟུགས་ཉིད་དོ། །།ཨཱ་ཤིg། ཨཱ་ཤཱིཥཱ། ཀུན་དུ་བོད་པ་ལ། ་ཧེ་ཨཱ་ཤཱི། ཧེ་ཨཱ་ཤི་ཏོ། ཧེ་ཨཱ་ཤི་ཥ། ཨུཏ་སྔ་མ་ཅན་གྱི་སྣེ་ཧ་ལས། རཾ་ཏཱིག་ལ་སོགས་པ་རྣམས་ཀྱི་ཀྭིཔ་སྟེ། ་ཨུཏ་ཀྱི་ཏ་དབྱི་བ་དང་ཥ་ཉིད་དོ། །།དེ་་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། དྲྀ་ཤ་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ག་ཉིད་དེ། རྫོགས་པ་དང་ཀ་ཡིག་གམ་ག་ཡིག་དག་དང༌། ཎ་ཉིད་དོ། །ཨཎྜིཀ། ཨུཎྜིག། ཨུཎྜི་ཧོ། ཨུཙྪི་ཧ། །ཨུཎྜི་ཧཾ། ཨུཙྪི་ཧོ། ཨུཎྜི་ཧ། ཨུཎྜི་ཧཱ། ཨུཎྜི་ཧཱུཾ། ཨུཙྪི་བྷི། ཞེས་པ་ལ་སོགས་པ་སྟེ། ་བདུན་པའི་མང་པོའི་ཚིག་ལ་ག་ཉིད་དུ་བྱས་པ་ན་སྒྲ་མེད་པ་ལ་དང་པོའོ་ཞེས་པས་ཀ་ཡིག་གོ། །ཨུཙྪི་གླུ། ཀུན་དུ་བོད་།
