ཁོ་ལྟ་བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའིཁ་ན་ཁཙྩཾ༑གྲྭ་མ་སྐྱོ། གྲྭ་མ་ཎི་ཏུ། ཀུན་དུ་བོད་པ་ལ་ནི། །ཧེ་གྲྭ་མ་ཎི། ཧེ་གྲྭ་མ་སྐྱོ། ཧེ་གྲྭ་མ་ཎྱ། དེ་བཞིན་དུ་བུད་མེད་ལའང་ངོ། །།ཨཱིཏ་ཨུཏ་བུད་མེད་བརྗོད་པ་ན་དྲིའོ་ཞེས་ན་དཱིའི་མིང་ཅན་དུ་ནི་མི་འགྱུར་ཏེ། ཀྟི་བུད་མེད་བརྗོད་པ་སྨོས་པ་ནི་རྟག་ཏུ་བུད་མེད་ཀྱི་ཡུལ་ཅན་ཉིད་ཀྱི་ཕྱིར་རོ། །མ་ནིང་ལ་ནི་་དབྱངས་ཐུང་དུ་མ་ནིང་ལའོ་ཞེས་པས་ཐུང་ངུ་ཉིད་དེ། མ་ནིང་ལས་སི་ཨམ་དག་དབྱི་ཞིང་དེ་ལ་བརྗོད་པ་ཡང་མ་ཡིན་ནོ་ཞེས་པས་སི་ཨམ་དག་དབྱིའོ། ཁྱོད། གྲྭ་མ་ཎི་ཀུ་ལཾ། ཨོ་ཨཱི་སྟེ། །ནཱ་མིན་དབྱངས་ལའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ། །གྲྭ་མ་ཎི་ནཱི༑། སླར་ཡང་། གྲྭ་མ་ཎི། གྲྭ་མ་ཎི་ནཱི། གྲྭ་མ་ཎི་ནི། ཊཱ་ལ་སོགས་་པའི་དབྱངས་ལ་སྐྱེས་པ་དང་འདྲ་བའི་དངོས་པོ་སྟེ། དེས་ན་ཐུང་དུར་མི་འགྱུར་རོ༑། གྲྭ་མ་ཙྪཱ། གང་གི་སྐྱེས་པ་དང་འདྲ་བར་མི་འགྱུར་བ་དེའི་ཚེ་ཐུང་ངུ་་དང་ནུ་ཨཱ་ག་མ་འང་ངོ༌། །གྲྭ་མ་ཎི་ནཱ། གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་ཎི་བྷི། །སྐྱེས་པ་དང་འདྲ་བའི་ཕྱོགས་ལ་ཐུང་ངུ་དང་ནུ་ཨཱ་ག་མ་མི་འགྱུར་ཏེ། །དེས་ན་ཡི་གེ་་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །།གྲྭ་མ་སྐྱེ། །གང་གི་ཚེ་མ་་ནིང་ཡིན་པ་དེའི་ཚེ། །གྲྭ་མ་ཎི་ནེ། གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་ཎི་བྷུ། གྲྭ་མ་ཎི་་་ན༑གྲྭ་མ་ཎྱ། །གྲྭ་མ་ཎི་དྷཾ། གྲྭ་མ་ཎི་བྷུ། གྲྭ་མ་ཎི་ན། གྲ་མ་ཎྱ། །གྲ་མ་ཎ་ནོ། གྲྭ་མ་སྐྱོ། སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱོགས་ལ། གྲྭ་མ་ཧཱུཾ། གང་གི་ཚེ་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བ་མེད་པ་དེའི་ཚེ་ཐུང་ངུ་དང་ནུ་ཨཱ་་ག་་་མའང་སྟེ། ནུ་དང་བཅས་པའི་ཨཱམ་ལ་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །གྲ་འདི་མ་ཎི་ཧཱུཾ། སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱོགས་ལ། །ནི་ལས་ང་ཨཱམ་མོ་ཞེས་་་པས་ཨཱམ་མོ། །གྲྭ་མ་ཎྞཾ། ཐུང་དུར་བྱས་པ་ན་ཡུལ་གཅིག་ལ་རྣམ་པར་འགྱུར་བ་ནི་གཞན་མ་ཡིན་པ་དང་འདྲ་བར་འགྱུར་རོ་ཞེས་པ་ཨཱམ་དུ་ཐོབ་ཅིང་། ཨཱ་ག་མ་དང་།
