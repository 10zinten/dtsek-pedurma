སྔ་མ་བཞིན་དུ་ཨེ་ཉིད་དང་ཨ་བྷྱཱ་སཱ་དབྱི་བ་དག་གོ། །ོ། དཾ་ཤི་སཾ་ཛི་སྭཱ་ཛི་རལླི་་རྣམས་ཀྱི་ཨན་ལའོ། །དཾ་ཤ་སོས་ཟ་བར་བྱེད་པ་ལའོ། །ཎཾ་ཛ་འགྲོགས་པ་ལའོ། །དྷཾ་ཛ་ཡོངས་སུ་འཁྱུད་པ་ལའོ། །རཾ་ཛ་ཚོས་ལའོ་འདི་རྣམས་ཀྱི་ཨན་ཕ་རོལ་ལ་ཨ་ནུ་ཏཾ་ག་དབྱིའོ། །ད་ཤ་ཏི། ས་ཛ་ཏི། :བ་རི་ཥ་ཛ་ཏི༄༅། ::ར་ཛ་ཏི། ཨན་ལ་ཞེས་པ་ཅི་ཞེ་ན། །ཨུ་བ་དེ་ཤ། :བངྒ། བ་རཱི་ཎཾ་ག། རངྒ། དངོས་པོ་ལ་གྷཉའོ། །།།ཙ་ཛ་དག་གི་དྷུཊ་དང་གྷ་རྗེས་སུ་འབྲེལ་བ་ལ་ཀ་ག་དག་་ཅེས་པས་ཛའི་གའོ། །རཾ་ཛའི་ཨིན་ཕ་རོལ་ལ་:སྨིག་རྒྱུའི་དོན་ལ་ཨ་ནུ་ཏཾ་ག་་་་དབྱི་བར་བྱ་སྟེ། ར་ཛ་ཡ་ཏི་མྲ་ྋགཾ་། སྐོམ་དུ་འཇུག་གོ་ལ་ཨིནའོ། །ཨའི་་ཨུ་པ་དྷཱ་ཞེས་པས་རིང་པོའོ། ་། མ་རྗེས་སུ་འབྲེལ་བ་རྣམས་ཀྱི་ཐུང་ངུའོ། །།:；སྨིག་རྒྱུ་ཞེས་པ་ཅི་ཞེ་ན། རཉྫ་ཡ་ཏི་བ་སྔོ། །ཨིན་ཨིཙ་ཨཊ་རྣམས་ལ་ཨུ་པ་དྷཱའི་:ཨ་རིང་པོ་སྟེ་ནཱ་མི་རྣམས་ཀྱིའང་འཕེལ་བའོ། །ཨ་ཡིག་ཨུ་པ་དྷཱ་གྱུར་པ་རིང་པོ་དང་ན་མི་རྣམས་ཀྱི་ཡང་དེ་བཞིན་ཉིད་དུ་འཕེལ་བར་འགྱུར་ཏེ། །ཨིན་སྟེ་དང་ཨིཙ་དང་ཨཊ་ཕ་རོལ་ལའོ། །པཱ་ཙ་ཡ་ཏི། ཡཱ་ཛ་ཡ་ཏི། ཙཱ་ཡ་ཡ་ཏི། ནཱ་ཡ་ཡ་ཏི། ་ཧཱ་བ་ཡ་ཏི། པ་ཙ་འཚེད། ཡཛ་ལྷ། ཙིཉ་སོག། ཊཱིཉ་ཐོབ། ཧུ་སྦྱིན་པ། འཚེད་དུ་འཇུག་གོ། །མཆོད་དུ་འཇུག་གོ། །།དེ་བཞིན་དུ་སོག་ཏུ་འཇུག་གོ། ་། ཐོབ་ཏུ་འཇུག་གོ། །ལེན་དུ་འཇུག་གོ་ལ་ཨིནའོ། །པ་ཙ་དང་ཡཛ་དག་གི་:ཨུ་པ་དྷཱའི་ཨཱ་རིང་པོའོ། །།གཞན་རྣམས་ཀྱི་:ནཱ་མི་མཐའི་འཕེལ་བ་ཨི་དང་ཨོའོ། །ཨི་ཙ་ལ། །ཨ་པཱ་ཙི། །ཨ་ྋཡཱ་ཙི། ཨ་ཀཱ་རི། ཨ་དྱ་ཨིཏ་ནཱིའི་ཏའོ་། །དངོས་པོ་:དང་ལས་དག་སེལའང་ངོ༌། །ཞེས་པས་ཨིཙའོ། །ཨིཙ་ལས་ཏ་དབྱིའོ། །རིང་པོའོ། །འཕེལ་བའང་ངོ༌། །ཨཊ་ལ། ་པ་པཱ་ཙ། ཙི་ཙཱ་ཡ། ཛུ་ཧཱ་བི། ཙ་ཀཱ་ར། ཨའི་ཞེས་པ་ཅི་ཞེ་ན། བྷེ་ད་ཡ་ཏི། །
