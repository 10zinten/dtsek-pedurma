སྤྱལ་བསྟན་འགྱུར། སྒྲ་མདོ། ལི་ད་ཀ་ཎ་མུ་ཉྩ། ཨ་མུསྨིན། ་༼གང་མ་ལས་ཕ་རོལ་གྱི་ཡི་གེ་དཱི་མ་ཡིན་པའི་མ་དེ་ལ་དེ་་་ལྟར་བསྒྱུར་བའི་གནས་ཀྱི་མཐའ་ལའོ་ཞེས་པའི་ཚིག་ལས་སོ། །མ་ལས་ཞེས་པ་ཅི་ཞེ་ན། ་ཨ་མུ་ཀཱ་ཧཱུཾ། ་ཀ་ལས་ཕ་རོལ་གྱི་མི་འགྱུར་རོ། །ཌ། མང་པོ་ཉིད་ལ་ནི་་་ཨེཏཱ་ཨཱི་ཉིད་དོ། །p། །མ་ལས་ཞེས་པ་ཁོ་ནའོ། །ཨ་དསའི་མ་ཡིག་དེ་ལས་ཕ་རོལ་གྱི་མང་པོ་ཉིད་ཀྱི་ཡུལ་གྱི་ཨེ་ཡིག་ནི་ཨཱི་ཡིག་ཏུ་འགྱུར་རོ། །།།ཨུ་ཉིད་ཐོབ་པ་ལའོ། །ཨ་མཱི། ཨ་མཱི་བྷུ། ཨ་མཱི་ཥཱམཱི། མང་པོ་ཉིད་ལ་ཞེས་པ་ཅི་ཞེ་ན། ་་་ཨ་མུ། ཨ་དས། ཨོའོ། འོ་། ཏྱད་ལ་སོགས་པའི་ཨ་ཉིད་དོ། །།བུད་མེད་ལ་ཨ་ལས་་་ཨཱཔའོ༑། ཨོ་ཨཱིའོ། །།ཨའི་ཡི་གེ་ཨིའི་ཡི་གེ་ལ་ཨེའོ། །མ་ཉིད་དུ་བྱས་པ་ན་མ་ལས་ཨུ་ཉིད་དོ་ཞེས་པས་ཨུ་ཉིད་དོ། །ཨཔྲའི་བྷ་ལ་དའོ། །།།ཨཔྲ་ེཞེས་པ་་་་འདིའི་རྣམ་དབྱེའི་བྷ་ཡིག་ཕ་རོལ་ལ་ད་ཉིད་དུ་འགྱུར་རོ། །:ཨདྷི། ཨདྡྷི། བྷ་ལ་ཞེས་པ་ཅི་ཞེ་ན། འོ་ཨསླུ། །རྣམ་དབྱེ་སྨོས་པ་ཅི་ཞེ་ན། འི་ཆུ་རྣམས་ཀྱི་ཁུར་ནི༑་ཨབྷཱ་ཙར། /རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་རྣམས་ལ་ཨ་ན་ཌ་ཧ་ན་་ཧི་ཝནྣ་རྣམས་ཀྱིའང་ངོ༌། །ཚ་། གང་རྣམས་ཀྱི་དང་པོ་གསལ་བྱེད་ཡིན་པ་དེ་རྣམས་་ནི་གསལ་བྱེད་དང་པོ་རྣམས་སོ། །དེ་ནས་ཟླས་དབྱེ་:བའང་ངོ་གྱི། །ཨ་ན་ཌ་ཏེ། ཧ་དང་ན་ཧི་དང་ཝནྟི་ཞེས་པའང་ཟླས་དབྱེ་བའོ། ་། ཨ་ན་ཌ་ཉིཧའི་འང་ངོ་ཞེས་པས་ཨ་ན་ཌ་ཧ་ཞེས་པ་འདིའི་ཨུ་ཉིད་དོ། །འདིས་ཧའི་དའོ། །ཚིག་མཐའ་ལ་སྤྱི་དྷཊ་རྣམས་ཀྱི་ཞེས་པས་དའི་ཏའོ། །དེ་ནས་ལྔ་པའོ། །ཨ་ན་ཌ་ཉིཧ་དང་ནཧྲ་དང་ཝརྨ་ཞེས་པ་འདི་རྣམས་ཀྱི་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ལ་ད་ཉིད་དུ་་་་འགྱུར་རོ། །གྲོང་ཁྱེར་གང་ན་གླང་བཟང་པོ་ཡོད་པ་དེ་ནི། ཌ་་:སཱ་ན་ཌཱུཏྲ།  ོ་ཨ་ན་འུདྡྷམ། ཨ་ན་རུཏུ། །སཱ་ན་ཌཱུཏྲ་ཅེས་པ་འདིར་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ལ་བརྗོད་པ་ནི་ཞེས་པས་ཨུ་ཉིད་དོ། །།།:ཨུ་པཱ་ནཧཱུྃ། ཨུ་པཱ་ནཧྥུ། —\།
