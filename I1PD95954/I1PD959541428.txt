སྲི་ཏཱཏ། ས་སྲྀ་ཏཱི། ས་སྲ་ཏུ། ཀི་དང་ཀིཏ། ཀ་སྡེའི་ཙ་སྡེའོ། །ཙེ་ཀེ་ཏུ། ཙིཀི་ཏཱཏ། ཙི་ཀི་ཏཱཾ། ་ཙི་ཀྱ་ཏུ། ཙིཀི་ཧི། ཙི་ཀ་ཏཱཏ། །ཙིཀི་ཏཱཾ། ཌཱ་ཙི་ཀི་ཏ། ཙི་ཀི་ཡཱ་ནི། ཙི་ཀི་ཊྛཱཡཱ་བ། ཙི་ཀི་ཡཱ་མ། ཏུ་ར་་མགྱོགས་པ་ལའོ། །གཉིས་བརྗོད་དོ། །ཨ་བྷ་སའི་གསལ་བྱེད་དང་པོ་ལྷག་མར་བྱའོ༑ཏི་། ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཏུ་ཏུནྟུ། ཏུ་ཏུ་ཏྟཱཏ། ཏུ་ཏུཏྟཾ། ཏུ་ཏུར་ཏུ། ཏུ་ཏུནྟི། །ཙི་ཀེཏྟཱི། ཙི་ཀིཏྟཾཏེ། ཙི་ཀིཏྟཾ། ཙི་ཀི་ཏྟཱ། ཙི་:ཀི་ཊཱི། ཙི་:ཀི་ཏྟཱཏི། ཙི་ཀིཏྟཾ། ཙི་ཀིཏྟ། ཨ་བྱཱ་སྟའི་ཡང་ཨུ་པ་དྷཱ་ན་མིན་བྱིངས་ཐམས་ཅད་པ་:ཡོན་ཏན་ཅན་ལ་དབྱངས་ལའོ་ཞེས་པས་ཡོན་ཏན་བཀག་གོ། །ཙི་ཀི་ཏཱིནི། ཙིཀི་ཏཱ་བ། ཙི་ཀི་ཏཱ་མ། ཛ་ཛ་ན། ཛ་ཛཱ་ཏུ། །ཛ་ཛཱ་ཏཱཏ། ཛ་ཛཱ་ཏྭཾ། ཛ་ཛྷ་ཏུ། །ཛ་ཛཱ་ཧི། །ཛ་ཛཱ་ཏཱཏ། ཛ་ཛཱ་ཏཾ། ་ཛ་ཛཱ་ཏ། ཨ་བྱཱ་སྟའི་ཡང་ཨུ་པ་དྷཱ་ཞེས་པ་ལ་སོགས་པ་ཁྱད་པར་ནཱ་མི་གཟུང་བ་ལས་ནཱ་མིའི་རྒྱུ་མཚན་ཉིད་ཀྱིས་ཡོན་ཏན་ཉིད་དུ་མི་་འགྱུར་བ་ལ། འོ། གམ་ཧན་ཞེས་པ་ལ་སོགས་པས་ཨུ་པ་དྷཱ་དབྱི་བ་མེད་དོ། །སེཛ་ནཱ་ནི། ཛ་ཨིཛ་ནཱ་བ། ཛ་ཛ་ནཱ་མ། ༈ངེས་པ་མེད་པ་ལའང་ཨཱ་ག་ཏི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྱ་ཏི་ཛ་ཛཱ་ཏྭཾ། བྱ་ཏི་ཛ་ཛྷཱ་ཏྭཾ། །བྱ་ཏི་ཛ་ཛྷཱ་ཏྭཾ། ཨིཊ་ཛ་ན་དག་གི་ས་དང་དྷཱ་ལའང་ངོ་ཞེས་པས་་་་ཨིཊ་ཨཱ་ག་མའོ། ་བྱ་ཏི་ཛ་ཛྷི། བྱ་ཏི་ཛ་ཛྷཱ་ཐཾ། བྱ་ཏི་ཛ་ཛྷི་དྷཾ། བྱ་ཏི་ཛ་ཛ་ནི། ་བྱ་ཏི་ཛ་ཛ་ནཱ་བ་ཧི། བྱ་ཏི་ཛ་ཛ་ནཱ་མ་ཧི། ་ཎི་ཛིར། ནེནེཀྱུ། ནེ་ནིཀཱཏ༄༅། །ནེ་ནིཀྵི། ནེ་ནི་ཛནྟུ། ནེ་ནིཧི། ནེ་ནིཀྐཏེ། ༼ནེ་ནིཏྟཾ། ནེ་ནིཏྟ། ?ཨ་བྱཱསྟའི་ཡང་ཨུ་པ་དྷཱ་ཞེས་པ་ལ་སོགས་པས་ཡོན་ཏན་བཀག་གོ། །ནེ་ནི་ཛཱ་ནི། ནེ་ནི་ཛཱ་བ། ནེ་ནི་ཛཱ་མ། ནེ་ནིཏྟཾ?། ནེ་ནི་ཛཱ་ཏྭཾ། :·ནེ་ནི་།
