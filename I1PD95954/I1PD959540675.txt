ཀ་ཀ་ཎ'ཡ་མ་རྣམས་ཀྱི་ཚའོ། །པ་རསྨི་ཞེས་པ་མི་འཇུག་གོ། །གཙླི་འགྲོ་བ་ལའོ། །ཨི་ཏུ་འདོད་པ་ལའོ། །ཡ་མུ་ཉེ་བར་འཇིག་པ་ལའོ། །།།འདི་རྣམས་ཀྱི་ཨན་ཕ་རོལ་ལ་ཚ་ཉིད་དུ་འགྱུར་རོ། །གཙྪ་ཏི། སཾ་གཙྪ་ཏེ། ཨིཙྪ་ཏི། ཡཙྪ་ཏི། སཾ་སྔོན་མ་ཅན་གྱི་ག་མཱི་ཏུ་ཙ་སོགས་སོ། །བཱའི་བི་བའོ། །པ་འཐུང་བ་ལའོ་་་་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་བི་བ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །བི་བ་ཏི། བི་བ་ཏིའི་དབྱི་སྟེ་ཞེས་པའི་ཤེས་པར་བྱེད་པ་ལ་ཡོན་ཏན་མེད་དོ། ཤོ། གླུའི་ཛི་གྲུའོ། གྲྭ་ཏི་ལེན་པ་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་ཛི་སྒྲ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །ཛི་གྷ་ཏི། དྷཱའི་དྷ་མའོ། །དྨ་སྒྲ་དང་མེ་ཀུན་དུ་སྦྱོར་བ་དག་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་དྷ་མ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །དྷ་མ་ཏི། །ཌ་སྦྲའི་ཏིཥྛའོ། །ཥྛ་གནས་པ་ལའོ་ཞེས་པ་འདིའི་ཨན་པ་ཅིཕ་རོལ་ལ་ཏིཥྛརུ་བསྒྱུར་བར་འགྱུར་རོ། །ཏིཥྛ་ཏི། སྣའི་མ་ནའོ། །སྣ་གོམས་པ་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་མ་ན་རུ་བསྒྱུར་བར་འགྱུར་རོ། །མ་ན་ཏི། །དཱཎའི་ཡཙྪའོ། །ཞེ། དཱཎ་སྦྱིན་པ་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་ཡཙྪ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །ཡཙྪ་ཏི། །དཱཎའི་ཎ་་ཡིག་འདིའི་དོན་ཏོ། །དྲི་ཤིའི་པ་བྱའོ། །དྲི་ཤིར་ལྟ་ཙིབ་ལའོ་ཞེས་པ་འདིའི་་་་ཨན་ཕ་རོལ་ལ་པ་བྱ་རུ་བསྒྱུར་བར་འགྱུར་རོ། ༽། པ་བྱ་ཏི། ཨརྟིའི་རིཙྪའོ། །ར་འགྲོ་ཅིབ་དང་ཐོབ་པ་དག་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་རིཙྪ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །རིཙྪ་ཏི། སརྟིའི་དྲྭ་བའོ། །:དྲི་ཤིར་ལྟ་ཅིབ་ལའོ་ཞེས་པ་འདིའི་ཕ་རོལ་ལ་:ས་ཤ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །དྷཱ་བ་ཏི། དྷཱ་བ་བསྒྱུར་བ་ནི་འགྱོགས་ཀྱི་པར་འགྲོ་བ་ལ་འདོད་པའོ། །དེས་ན་འདིར་མི་འགྱུར་ཏེ། དྲྀཡཾ་ཨ་ནུ་ས་ར་ཏི། སུཾ་འགྲོ་བ་ལའོ། །དྷཱ་བུ་འགྲོ་བ་དང་དག་པ་ལའོ་ཞེས་པའི་་་བྱིངས་གཉིས་:ཀྱི། སྐྱ་སརྟི༄༅། དྷཱ་བ་ཏི། །ཞེས་པའི་སྦྱོར་བ་གཉིས་གྲུབ་པ་ལ་་་།
