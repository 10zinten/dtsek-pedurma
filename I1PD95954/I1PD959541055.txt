བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་ཏྟོsolooookmolerཀྲོ༑་ཀྲོཟླུ། ཀྲོཥྛ་བྷཱུམ། ཀྲོཥྛ་བྷུ། ༼ཀྲོཊོ། །ཀྲོཏྟུ། ཞེ་:ཀློབྦོ། ཀློབྦོ༑ཀྲོཏྟཱ་ནཾ། ཀྲོབྦྷ་ཏཱཾ། ཀྲོཊྚ། ཀྲོཌ་རི། ཀྲོཟློ། :：ཀྲོཟློ། ཀྲོབྷུ་ཥཱཀུན་དུ་བོད་པ་ལ་:ཧ་ཀྲོཊྚེ། ཧེ་ཀྲོཊྚརོ། ཧེ་ཀྲོཏྟཱར། །iHe། ::པཨུ་ཡིག་གི་མཐའ་ཅན་བརྗོད་པར་བྱའོ། །སོ་སོར་འབྱུང་ངོ་ཞེས་པས་པྲ་ཏི་སྔ་མ་ཅན་གྱི་བྷ་བ་ཏི་ལས་ཀྭིཔའང་ངོ་ཞེས་པས་ཀྭི་པའོ། །པ་ཡིག་ནི་བྱིངས་ཀྱི་མཐའ་པ་རྗེས་སུ་འབྲེལ་པ་ལ་ཏའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཀ་ཡིག་ནི་ཀ་ལ་ཡཎ་བཞིན་འང་སྟེ། ཡ་ལ་བརྗོད་པ་བོར་བའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨ་ཡིག་ནི་སྦྱར་བ་མིན་པའི་ཝའི་དབྱིའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །འདི་་ཉིད་ཀྱི་ཝ་:དབྱི་བ་རྟགས་ཀྱི་མིང་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །སི་ལ། ་པྲ་ཏི་བྷུ༑བྷུའི་ཝཱནྟ་བྷུ་དང་པུ་ནནྟཱ་མ་ཡིན་པའོ་ཞེས་པས་བྷུ་ནི་རྣམ་པར་དབྱེ་བ་དབྱངས་ལ་ཨུ་ཝ་རུ་འཐོབ་བོ། །པྲ་ཏི་བྷུ་ལོ། པྲ་ཏི་བྷ་ཝ།  ོཔྲ་ཏི་བྷ་ཝཾ། པྲ་ཏི་བྷུ་ཝོ། པྲ་ཏི་བྷ་ཝ། པྲ་ཏི་བྷ་ཝཱ། པྲ་ཏི་བྷཱ་བྷཱུམལ། པྲ་ཏི་:བྷུ་བྷིནྡྷེ། པྲ་ཏི་བྷུ་ཝེ། པྲ་ཏི་བྷཱ་བྷཱུམ། པྲ་ཏི་བྷུ་བྷུ། པྲ་ཏི་བྷ་ཝ། པྲ་ཏི་བྷུ་བྷཱུམ། པྲ་ཏི་བྷུ་བྷུཉྩི། པྲ་ཏི་བྷུ་ཝ། པྲ་ཏི་བྷུ་ཝོ། པྲ་ཏི་བྷུ་ཝཱམ། པྲ་ཏི་བྷུ་ཝི། པྲ་ཏི་བྷུ་ཝོ། །པྲ་ཏི་བྷཱ་ཏུ། ཀུན་དུ་བོད་པ་ལ། ་ཧེ་པྲ་ཏི་བྷཱཉྩ། ཧེ་པྲ་ཏི་་བྷུ་ལོ། ཧེ་པྲ་ཏི་བྷུ་ཝེ། དེ་བཞིན་དུ་ཀཊ་པཱུ་ནི་འདྲེས་པ་ལ་ཤེས་པར་བྱའོ། །ཟླ་བ་དང་:ཐུ་ལ་ཨ་ཏྲི་ནེ་ཏྲ་བྷུའོ། །མེ་ལ་དེ་བཞིན་དུ་ཧི་ར་ཙྩབྷཱུའོ། །ཨཀྵ་རཱུ་དང་སྨྲ་ཡང་བྱས་པ་ལ་ཡང་ངོ༌། །བྲཧྨ་སུ་བྷཱུ་ནི་རིག་བྱེད་ལའང་ངོ༌། །ཤྭ་ཥཱ་རབྷུ་ནི་ཕུར་པ་ལའོ། །ཨ་དྷི་བུ་འདྲེན་པ་ལ་རབ་ཏུ་བརྗོད་དོ། །བྲིཏྟི་བྷ་དེ་བཞིན་དུ་སྦལ་པ་ལའོ། །སཱ་བྷུ་ནི་ཁྱབ་འཇུག་དང་རྣམ་འཐོར་ལའོ། །།འདོད་་པ་དང་ཚངས་པ་ལའང་ཨཱཏྨ་:བྷཱའོ། །།ལུ་། །།།ལས་གཏོད་།
