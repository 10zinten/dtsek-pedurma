བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི::ཨ་རོ་ད། དེ་བཞིན་དུ་ཉི་ཥ་བ། ཥཱ་ས་ཞེས་པ་འདི་དག་དང་ཨན་ཡང་ངོ༌། །ཨ་ཡིག་གི་ཨཱ་ཡིག་གོ། ་། ཨཱ་ནཱིཏ། ཨ་ནཏ། ཨཱ་ནཱི་ཏཱཾ། ཌཨཱ་ནན། ཨཱ་ནཱི། ཨཱ་ནི། །དེ་ལ་ཡང་སྨོས་པ་ལས་ཨནྟིའི་མཐའི་ཡང་ངོ་ཞེས་པས་ཎ་ཉིད་དོ། །བླ་ཎྜིཏ། ་་:བཱ་ཎིདི། ཛཀྵ། ཨ་ཛཀཱིཏ། །aཨ་ཛཀཱཏ། །ཨ་ཛཀཱི་ཏཱཾ། ཛཀླུ་སོགས་ཀྱང་ངོ་ཞེས་པས་ཨ་བྱཱསྟའི་མིང་ངོ༌། ་། སི་ཙ་ཨ་བྱཱསྟ་བིད་ཨཱཏ་་རྣམས་ལས་བྷུ་མ་ཡིན་པ་ལས་ཨནའི་ཨུསའོ་ཞེས་པས་ཨན་གྱི་ཨུསའོ། །ཨ་ཛགྲུ། ཨ་ཛ༎་། :། ཨཀྵི། །ཛཱ་གྲི། ༼ཡོན་ཏན་ནོ། །དི་སི་དག་གི་དབྱིའོ། །ཚིག་མཐར་རེ་ཕའི་རྣམ་པར་བཅད་པའོ། །ཨ་ཛཱ་ག། ཨ་ཛཱ་ཤྲཱི་ཏཱཾ། ཨ་་བྱཱསྟ་རྣམས་ཀྱི་ཨུས་ལའོ་ཞེས་པས་ཡོན་ཏན་ནོ། །ཨ་ཛཱ་ག་རུ། ད་རི་དཱི། ཨ་ད་རི་དྷཱཏ། ཨ་ད་རི་དྲི་ཏྟཾ། ཨ་བྱཱསྟ་རྣམས་ཀྱི་ཨཱ་ཡིག་གོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། །ཨ་ད་རི་དྲིུ། ཙ་ཀཱ་སྲི༄༅། རྒྱལྟ་ནཱིའི་དི་ལ་སའི་ཏའོ་ཞེས་པས་་་སའི་ཏ་ཡིག་གོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དང་པོ་དང་གསུམ་པ་དག་གོ། །ཨ་ཙ་ཀཱཏ། ཨ་ཙ་ཀཱདེ། ཨ་ཙ་ཀཱནྟཾ། ཨ་ཙ་ཀཱ་སུ། །རྒྱས་ནཱིའི་སི་ལ་་ཡང་ན་བརྗོད་པར་བྱའོ་ཞེས་པས་ཡང་ན་སའི་ཏའོ། །ཨ་ཙ་ཀཱཏ། ཨ་ཙ་ཀཱད། ཨ་ཙ་ཀཱ། །ཤཱ་སུ། ཨ་ཤཱཏ། ཨ་ཤཱད། ཨ་ཤིཏྟཾ། ཨ་ཤཱ་སུ།  ོཨ་ཤཱཏ། ཨ་ཤཱད། ཨ་ཤཱ། ཙགླིང་། ཨཱ་ཙཊྚ། ཨཱ་ཙཀྵ་ཏྭཾ། ཨཱ་ཙཀ་ཏ། ཨཱ་ཙཥྛ*། ཨཱ་ཙཀྵ་བྷཾ། ཨཱ་ཙནྡྷཾ། ཨཱ་ཙཀྵ། ཨཱ་ཙཁཱ་ཧི། ཨཱ་ཙཀྵ་ཧི། ཨཱིར་་ར༑དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །།ཨིཏྟ། ཨིརཱ་ཏྭཾ། ཨིར་ཏི། ཨིནྡེ། ཨིརཱ་ཐཱ། ཨིནྣཾ། ཨི་རི། ཨིརྦ་ཧི། ཨིརྨ་ཧི། ཨཱི་ཌ༑ཨིཎྜ། ་ཨེ་ཌཱ་ཏྭཾ། ཨེ་ཌཏ། :·ཨེ་ཌཐཱ། ::ཨི་ཌ་ཐཱ། ཨཱིཊ་དབྱེཛན་དག་གི་ས་དང་དྷཱ་ལའང་ངོ་ཞེས་པས་ཨིཊའོ། །ཨེ་ཌི་ཧཱུྃ། ཨཱི་ཤ། ཨིཊྚ། །
