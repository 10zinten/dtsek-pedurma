རི་དཱ། ཨིཊ་ཨཱ་ག་མའོ། །ད་རི་དྭ་ཏིའི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལའོ། །ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ༄༅། ::། ད་རི་པྲ་ཏཱ་ཛི། ཨིའི་ཡི་གེ་ལས་ཤི་གླིང་ྋརྟིང་ཤིང་མ་ཡིན་པ་ལས་སོ་ཞེས་པས་ཤཱི་ཤྲཱི་རྟིང་ཤིང་བོར་བ་ཨིའི་ཡི་གེ་དབྱངས་གཅིག་པ་ལས་ཕ་རོལ་གྱི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་ཨིཊ་མེད་པ་ཉིད་དོ། །ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཛེ་ཏཱ་བི་པ་རཱ་དག་ལས་ཛིའོ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བེ་ཛི་ཏཱ། བཱ་རཱ་ཛེ་ཏཱ། ཀྱི་ཟད་པ་ལའོ། །ཀྱེ་ཏཱ༑ ི་རི་པི། །།རེ་ཏཱ། བྷཾ་འཛིན་པ་ལའོ། །ཨིཎི། །བྱ་ཡ་སྟོན་པ་བཞིན་བསྒྱུར་བས་ན་མི་མཐའི་ཡོན་ཏན་ནོ། །དྷཏྟཱ། ཨི་ཀ། ཨ་དྷེ་ཏཱ། ་སྨིང་། སྨེ་ཏཱ། ཨིང་། ཨ་ཧྥེ་®ཏཱ། ཙི་ཉ། ཙེ་ཏཱ། ཥི་ཉ། སེ་ཏཱ། །ཤི་ཉ། ཐེ་ཏཱ། བཱི། ་བེ་ཏཱ། ཉི་བྷི། བྷེ་ཏཱ། །ཧྲེ་ཏཱ། བཱི། བྲེ་ཏཱ། མཱིང་། པམེ་ཏཱ། པཱིང་། ྋཔེ་ཏཱ། ''ཎཱིཉ། ནེ་ཏཱ། འུ་ཀྲྀཉ། ཀྲེ་ཏཱ། ལྡིང་། པྲ་ཏཱ། ཊུ་ཨོ་ཤཱ། ཨིཊ་ཨཱ་ག་མའོ། །ཤུ་ཡི་ཏཱ། རྟིང་། ཌ་ཡི་ཏཱ༑ྋཤིང༌། mཤ་ཡི་ཏཱ། ཙི་རི། ཛི་རི། ཏོཙི་ར་ཡི་ཏཱ། ཛི་ར་ཡི་ཏཱ། དཱི་དྷིང་། །འབེ་བཱིང་། དཱི་ནི། བེ་བཱི་དག་གི་ཨི་ཡིག་ཡ་ཡིག་དག་ལའོ། །ཞེས་པས་མཐའ་དབྱིའོ། ལྔ་། དཱི་དྷི་ཏཱ། བེ་བི་ཏཱ། ཤུ། ཨུཏ་ཀྱི་ལས་ཡུ་རུ་ཆུ་་སྣུཀླུ་སྒྱུ་མ་ཡིན་པ་ལས་སོ། །ཞེས་པས་དབྱངས་གཅིག་པའི་ཨུཏ་མཐའ་ལས་ཡུ་རུ་ཆུ་སྣ་ཀླུ་སྒྱུ་བོར་བའི་ཕ་རོལ་གྱི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་ཨིཊ་མེད་དོ། །ཏུ། ཤུ། དུ་དུ། སོ་ཏཱ། ཤོ་*ཏཱ། དོ་ཏཱ། དྲོ་ཏཱ། བྷུ་གནས་པ་ལའོ། །བྷོ་ཏཱ། ཧུ། ཧོ་ཏཱ། ：་གུ་བཤང་བ་འདོར་བ་ལའོ། ་། དྷུ་འགྲོ་བ་དང་བརྟན་པ་ཉིད་དག་ལའོ། །།ཀུཊ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །།གུ་་པི་ཏཱ། དྷུ་ཏཱ། ་བླུང༌། བློ་ཏཱ། སྲུང་སྒྲ་ལའོ།  ོ། ཨོ་ཏཱ།  ོ་ཀུང་། བྷུ་།
