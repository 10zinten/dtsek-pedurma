ཊི་ཐ། བི་ཤ། བི་བི་ཤ། བི་བི་ཤ་ཏུ། སྲི་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ༑་། དེ་བཞིན་དུ། །དི་ཤ་། པི་ཊཱི། མི་ཧའོ། །ཎི་དི། ནི་ནིནྡ། ནི་ནིནྡ་ཏུ། ནི་ནིནུ། །གསལ་བྱེད་དང་པོ་ལ་ཨིཊའོ། །ནི་ནཱིནྡི་ཐ། ཊི་ཀྲྀཊི་ཊི་ཀེ། ཏི་པྲ། ཏི་ཏི་བེ། སྲི་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །ཏི་ཏི་པི་ཥ། ་ཏི་ད་ཞན་པ་ཉིད་ལའོ། །བིད་ཡོད་པ་ལའོ། །བི་ད་རྣམ་པར་དཔྱོད་པ་ལའོ། །།ཁི་ཁི་དེ། །བི་བི་དེ། བྷི་དིར། ་བི་བྷེ་སྐྱིད། །བི་བྷི་ད་ཏུ། བི་བྷི་དུ། བི་བྷི་དི་ཐ། བི་བྷི་དེ། ཙྪ་:དིར། །ཙིཏྟཱཙྪེད། ཙི་ཙྪེ་ད་ཏུ། ཙིཙྪ་དུ། :ཙིཙྪི་དི་ཐ། ཙི་ཙྪི་དེ༌། ཎི་ཙ། སི་ཥ་ཙ། སི་:ཏི་ཙུ་ཏུ། ཡིསི་ཁེ་ཙུ། སི་ཁེ་ཙི་ཐ། ་ལི་ཧ། ལི་ལེ་ཧ། །ལི་:ལེ་ཧ་་་ཏུ༑་ལི་ལི་ཧུ། ལི་ལེ་ཧི་ཐི། ལི་ལི་ཧེ། ལི་ལེ་ཧི་ལྷེ། ་་:དེ་བཞིན་དུ་དི་ཧའོ། །བིཊཱི། བི་བེ་ད། །བི་བིད་ཏུ། བི་བི་དུ། བི་བེ་དི་ཐ། དེ་བཞིན་དུ། ཀྱི་པ། ལོ་པའོ། །།ཏུད་ལ་སོགས་པ་ལ་གཉི་གའི་ཚིག་གོ། །ཙིཀྵེ་པ། ཙིཀཱི་པ་ཏུ། ཙིཀཱི་པུ། །ཙིཀེ་པི་ཐ། ཙིཀཱི་པེ། བི་ད་ལ་་སོགས་པ་ལ་ནི་པ་རསྨིའི་ཚིག་གོ། །ཀཱི་ཙེ། ཙི་ཀྲྀ་ཌ། ཙི་ཀཱི་སེཌ་ཏུ། ཙི:;ཀྲི་ཌི། གསལ་བྱེད་དང་པོ་ལ་ཨིཊའོ། ་། ཙི་ཀྲྀ་ཌི་ཐ། དེ་བཞིན་དུ། མཱི་ལ། མི་ལ། ཐི''ི་ལ། ཎི་ལ། པཱི་ལ། ཤི་ལ། ཀཱི་ལ། ཛཱི་བ། པཱི་བ། མཱི་བ། ཎི་བའོ། །།།མི་མཱི་ལ། མི་རྨི་ལ། རྒྱུ་མཚན་ལས་རྐྱེན་དང་རྣམ་འགྱུར་དང་ཨཱ་ག་མ་ལ་གནས་པའི་སའི་ཥ་ཉིད་དོ། །ཤཱི་ཀྲྀ། ཤི་ཤྲཱི་ཀེ། ཤི་ཤྲཱི་ཀཱ་ཏེ། ཤི་ཤྲཱི་ཀཱི་རེ། ཤི་བྷི་ཀི་ཏེ་ཊི''ཀྲྀ། ཊི་ཊཱི་ཀེ། ཀླུ་བྲི། གི་བེ། དཱི་:པེ། དཱི་དི་བེ། ::བྷཱུ་ཏིར། ཙུཤྩ་ཏ། །ཙ་བྷཱུ་ཏ་ཏུ། ཙུ་སྦྱོ་ཏི་ཥ། ཤུ་ཙ། ་ཤུ་ཤོ་ཙ། ་༌་ཤུ་ཤུ་ཙ་ཏུ། ཤུ་ཤུ་ཙུ། ཤུ་ཤོ་ཙི་ཐ། GU།
