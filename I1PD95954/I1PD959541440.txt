ཨེ་ཤཱ་ཏཱ། ཨི་ཤ་ཏི། ཨི་བྷཱ། ཨི་ཤཱ་ཛཾ། ཨེ་ཤི་དྡྷཾ། ཨཱ་ས། ཨ་་ཡིག་གི་ཨཱ་ཡིག་སྟེ་འཕེལ་བའི་དམིགས་བསལ་ལོ། །།ཨཱསྟ། ཨཱ་སཱ་ཏྭཾ། ཨ་ས་ཏ༑ཨཱང་ལས་ཤཱས་སུ། ཨཱ་ཤཱསྟ། :ཨཱ་ཤཱ་སཱ་ཏཱཾ། ཨཱ་ཤཱ་ས་ཏེ། བ་ས་ཀུན་ནས་:འགེབས་པའོ༄༅། །ཨ་བསྟ། ་ཨ་བ་སཱ་ཏྭཾ། ཨ་བ་ས་ཏ། ཎི་སི། ཨ་:ནི་སྟེ། །ཨ་ནི་སཱ་ཏཱ། ཨ་ནི་ཡིས་ཏ། ཏུང་སྲོག་ཆགས་མངལ་ནས་སྐྱེ་བ་ལའོ༑། ཨ་སཱུ་ཏ། ་ཨ་སུ་བཱ་ཏྭཾ། ཨ་སུ་བ་ཏ། ཤིང་། ཨ་ཤེ་ཏ། ཨ་ཤ་ཡཱ་ཏྭཾ། ཤི་ཏི་ལས:：ཨནྟིའི་དང་པོར་ཨིརའོ། །།ཨ་ཤེ་ར་ཏ། /ཨིང༌། དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའ:ོ། །ཨ་དྷི་ཏ། ཨི་ཡ་རུ་་བསྒྱུར་རོ། །།འཕེལ་བའོ། །།ཨ་དྷི་ཡཱ་ཏྭཾ། ཨ་དྷི་ཡ་ཏ། ཨ་དྷི་ཐཱི། ཨ་་དྷི་ཡ་ཐཾ། ཨ་དྷི་ཉྩཱདྷཾ། »ཨ་དྷི་ཡི། ་ཨ་དྷི་བ་ཧི། ཨ་དྷི་མ་ཧི། དྲི་དྷིང༌། ཨ་དཱི་དྷཱི་ཏ། ཨ་དཱི་དྷཱ་ནཱཾ། ཨ་དཱི་རྒྱ་ཏ། དྭི་དྷི། བེ་བཱི་དག་གི་ཨེ་ཡིག་ཡ་ཡིག་དག་ལའོ་ཞེས་པས་མཐའ་དབྱིའོ། །ཨ་དཱི་དྷི། དེ་བཞིན་དུ། བེ་བཱི་ངའོ། །ལྷུང་། །ཨ་ཧྥུ་ཏ། ཨ་དྷྱ་བཱ་ཏཱ། ཨ་ཧྥུ་བ་ཏ། དྭི་ཥ། །ཨ་དྭེཌ། ཨ་་དཱིཏྟཾ། སིཙ། ཨ་བྱཱསྟ་བིད་ཨཱཊ་རྣམས་ལས་བྷུ་མ་ཡིན་པ་ལ་ཨནའི་ཨུསའོ་་་ཞེས་པའི་ཚོགས་སུ་བྱས་པའི་རྟག་ཏུ་བ་མ་ཡིན་ནོ་ཞེས་པས་ཨནའི་ཨུས་སུ་ཡང་་ཡང་ཨིན་འགྱུར་རོ། །ཨ་དཱི་ཥཱཚེ། །ཨ་དཱི་ཥན། ཨ་དྭེཊ། ཨ་དྭེཌ། ཨ་་དྲིé༑?ཨ་དཱི་ཝ། ཨ་དྭེ་ཐཾ། ཨ་དཱི་། ཨ་དཱི། ཨ་དཱི། ཨ་དཱི་ཥཱ'ཏྭཾ། །ཨ་དཱི་ཥཏ། ཨ་ཏིཥྛ། ཨ་དཱི་ཥཱ་ཐཱ། ཨ་དཱི་ཥཱ་ནཱཾ། ཨ་དཱི་ཏི། ཨ་དཱི་ཤྲཱི་ཧི༑ཨ་དཱི༔་ཧི། དུ་ཧ། ད་དང་པོའི་གྷའོ་ཞེས་པའི་མཐའི་དྷཊ་ལ་ཧའི་་གྷའོ༑། ཕྱིས་པ་ལའང་དི་སི་དག་ལའོ་ཞེས་པས་དང་པོ་དའི་བཞི་པ་དྷ་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་གྷའི་དང་པོ་ཀ་དང་གསུམ་པ་དག་གོ། །ཨ་དྷོཀ། །
