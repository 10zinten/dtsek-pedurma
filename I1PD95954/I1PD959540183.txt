གང་རྔུ་ཀྲྀ་ཉ་ནི་བྱེད་པའོ། །ཁ་ན་ལ་ཨཊ་སྦྱར་བས་གོང་དུ་ཁ་གཉིས་པ་བྱས། ཁས་:ཚ་ཐོབ། ཁོ་ཚས་ཙ་ཐོབ། ཙ་ཁཱ་ནའོ། །།ཁ་ན་ནི་ངེས་པར་འཇོམས་་་པའོ༑། ན་ཀ་བ་ཏེཝེ་ཀྲི་ཡ་ཏི། ་ཀུང་གི་སྒྲ་ཞེས་བྱ་བ་འདིའི་དྷཱ་ཏུའི་ཕ་རོལ་ན་ཙེ་ཀྲི་ཡི་ཏ་ཡོད་ཀྱང་ཨ་བྷཱ་སའི་ཀ་ཙར་མི་འགྱུར་བ་སྟེ། ཀུ་ལ་ཡ་སྦྱར། གང་དུ་ཀུ་གཉིས་པ་བྱས། ཀུ་ཀོར་གྱུར། ༼ཨུ་ཨུར་གྱུར། མཐར་ཏེ་དང་། ཨེན་སྦྱར། ཡའི་ཨ་ཕྱི། ཨན་གྱི་ཨ་དང་སྦྱར། ཀོ་ཀུ་ཡ་ཏེའོ། །།ཧྲ་སཱ། གང་ཨ་བྷུ་ས་རིང་བ་དེ་ཐུང་བར་འགྱུར་བ་སྟེ། ང་དཱ་ལ་ཨཊ་སྦྱར་བས་གོང་དུ་དཱ་གཉིས་པ་:བྱས་ཏེ?། ད་བསྟུངས། །ཨཊ་ཀྱི་ཨོ་ཐོབ། ད་དོའོ། །པཱ་ལ་ཨཊ་སྦྱར་བས་གོང་མ་བཞིན། པ་པོའོ། །ཏུ་དཱཉ་ནི་སྦྱིན་པའོ། །།པཱ་ནི་ཆུའོ། །ཨ་ཀཱ་རརཾ་ཝཎ་སྱ། ཨ་ངྷ་སའི་རི་གང་ཡིན་པ་དེ་ཨར་དུ་འགྱུར་བ་སྟེ། དྲྀ་་ཏ་ལ་པ་རོཉགླུའི་ཨེ་སྦྱར་བས། བྲོ་གཉིས་པ་བྱས། མདོ་འདིས་རིས་ཨ་ཐོབ། བ་བྲྀ་ཏེའོ། །དྲྀ་ཏུ་ནི་འཚོ་བའོ་། :། །:བ་ཏེ་ཚན་པ། ཕྱི་པ་བ་ཁ་ལ་བློ་གཏེ། དཱིགྷ་ཨི་:ན་པ་རོཉཀྵ་ཡ་མ་གྲུ་ཅེཎེ། ཨི་ག་ཧ་ཞེས་བྱ་བ་གང་ཡིན་པ་འདིའི་ཨ་བྷཱ་ས་དེ་རིང་བར་འགྱུར་བ་སྟེ། །པ་:རོཉཁཱ་གུ་ཎ་མ་ཡིན་པའི་རྐྱེན་ཕ་རོལ་ན་ཡོད་ནའོ། ?། ཨིན་ཅེལ་པ་རོཉཀཱའི་ཨ་ཏུས་སྦྱར། སཱ་ར་དོ་ཨི་བཎ་ཨནྟ་ཨསྱའི་མདོས་ཨིས་ཨི་ཡ་ཐོབ། །ཡ་དང་ཨ་སྦྱར། མདོ་འདིས་ཨི་སྲིངས། :：ཨི་ཡཱ་ཏུའོ། །ཨི་ཎ་ལ་ཨུས་སྦྱར་བས་གོང་བཞིན། ཨི་ཡུའོ། །ཨ་མྱུ་:དེས་གྲྭ་ཏྲ། གང་ཨ་བྷཱ་སའི་ཨ་དེ་རིང་བར་འགྱུར་བ་སྟེ། །པ་རོཉཀྵ་སེཐམས་ཅད་ན་ཕ་རོལ་ན་ཡོད་ནའོ། །ཨཊ་ལ་པ་རོཉཁཱི། ཨཊ་སྦྱར། །མདོ་འདིས་ཨ་བྷཱུ་སའི་ཨ་སྲིངས་ཏེ་དྷཱ་ཏུའི་ཨ་དཱིགྷ་པ་ར་ལོ་པའི་མདོས་ཕྱི། ཨཱཊའོ། །ཨཊ་ལ་་་་ཨུཊ་སྦྱར་བས། ཨ་གཉིས་པ་བྱས་པ་ལ་སོགས་པས། །ཨཱཊ་ཏུའོ༄༅། །ཨཊ་ལ་—།
