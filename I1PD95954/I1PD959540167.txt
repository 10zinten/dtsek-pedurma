བསྟན་འགྱུར། སྒྲ་མདོ། ལིཨཎ་འབྱུང་བ་སྟེ། ཨ་པ་ལ་ཨ་བྱ་ཏ་ནིའི་ཏ་སྦྱར། བར་དུ་ཨཎ་སྦྱར་གོང་དུ་མ་:སྦྱར། །ཨ་དྱ་དེའི་མདོས་ཐ་ཐོབ། ས་དང་ཨ་དང་ཨཎ་གྱི་ཨ་སྦྱར་ཨ་གོང་མ་བསྲིངས་ཏེ། །པ་རི་དང་ཨི་:ཝརྞའོ། །ཝ་ཙ་ལ་དི་སྦྱར་བས་ཨོ་ཐོབ་པའོ། །ཨ་བོ་ཙཏའོ། །ཁྱུ་ལ་དི་སྦྱར། གོང་དུ་ཨཱའོ། །ཨ་ཁྱུ་ཏའོ། །ལི་པ་ལ་་དི་སྦྱར། གོང་དུ་:ཨའོ། །ཨ་ཁྱུ་ཏའོ། །ཨཱ་ལི་བཏའོ། —། སི་ཙ་ལ་དི་སྦྱར། གོང་དུ་ཨའོ། །ཨ་སི་ཙཏྲའོ། །།པུ་ཥཱ་དི་ཏྱཱ་ཏཱ་ཙྩཾ་ྋཀཱ་རཱ་ནུ་བནྡྷརྟི:;ཤརྟི་ཤསྟི་བྱཱཉྩ་པ་རསྨི། །པུ་ཥ་ནི་རྒྱས་པ་སྟེ། དེ་ལ་སོགས་པ་རྣམས་དང༌། །བྱུ་ཏ་ནི་་་འབར་བ་སྟེ། དེ་ལ་སོགས་པ་རྣམས་དང་དེ་བཞིན་དུ་ལིའི་མཐའ་ཅན་གྱི་དྷཱ་ཏུ་་རྣམས་དང་། དེ་བཞིན་དུ་ལོ་ནི་འགྲོ་བ་དང་རི་སྲ་ཡང་འགྲོ་བའི་དོན་ཅན་ནོ། །ཤཱ་སུ་ནི་རྗེས་སུ་སྟོན་པ་སྟེ། དེ་རྣམས་དང་ཕ་རོལ་ནི་བྱེད་པ་པོའི་ཚིག་ལ་ཨ་བྱ་ཏ་ནི་གཞན་གྱི་ཚིག་ཅན་ཡོད་ན་རྐྱེན་ཨན་འབྱུང་བར་འགྱུར་ཏེ། པུ་ཥ་ལ་་་སོགས་པ་རྣམས་ནི། །པུ་ཥ་ལ་དི་སྦྱར་བས་བར་དུ་ཨན་འབྱུང་སྟེ། །%ཥ་དང་། སྦྱར་གོང་དུ་ཨའོ། །ཨ་པུ་ཥཏའོ། །ཏུ་ཥ་ལ་དི་ལ་སོགས་པ་གོང་བཞིན། z་ཨ་ཏུ་ཥཏའོ། །ཤུ་ཥ་ལ་ཡང་གོང་བཞིན། ཨ་ཤུ་ཥཏའོ། །པུ་ཥ་ནི་རྒྱས་པའོ། །ཏུ་ཥ་ནི་དགའ་བའོ། །ཤུ་ཥ་ནི་སྐེམས་པའོ། །དྷཱུ་ཏ་ལ་སོགས་པ་རྣམས་ལ། དྷུ་ཏ་ལ་དི་སྦྱར་བས་བར་དུ་ཨན་སྦྱར། །གོང་དུ་ཨ་སྦྱར། ཨ་དྷུ་ཏཏྲའོ། །།རུ་ཙ་ལ་དི་སྦྱར་བ་གོང་བཞིན། །ཨ་རུ་:ཙན་ཏའོ། །ཏུ་ཏ་ནི་འབར་བའོ། །ལ་རྗེས་སུ་འབྲེལ་པ་རྣམས་ལ། གངྒྷཾ་ལ་དི་སྦྱར་བས། ལ་ཕྱི། བར་དུ་ཨན། ན་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨ་ག་མཏྲའོ། །ཨ་དི་ལ་དི་སྦྱར་བས། གྷསྶཾ་ཐོབ། གོང་དུ་ཨ་སྦྱར། ལ་ཕྱི། ཨན་དང་སྦྱར། ཨ་གྷ་སཏྲའོ། །གམ་་ལི་ནི་འགྲོ་བའོ། །ཨ་:རིའི་གྷསླུ་གཞག་ཨེཔ་སྟེ། ཨརྟི་སོགས་རྣམས་ལ། །རཾ་ལ་—།
