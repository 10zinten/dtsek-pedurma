དཱིཔཱི༑དྲི་དྷཱི་ཏི། ཨཱ་སི། རཱ་ཛི། ར་ཏི། ཁཱ་སྟེ། དཎི། ཀཾ་བི་དྷི། ཞེས་པ་ལ་སོགས་པ་རྣམས་སོ། །དང་པོ། དང་བ་དྷཱའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ན་དཱི་དང་འདྲ་བར་དྲང་བར་བྱ་ཞིང༌། འདི་ནི་ཁྱད་པར་ཏེ་ཨཱི་ཡིག་མཐའ་ལས་སི་དབྱི་བ་མེད་ལ། དབྱངས་དང་། པོ་ཅན་ལ་ཨུའི་ཡི་གེ་ལ་ཝའོ་ཞེས་པས་ཨུ་ཡིག་ཝ་ཡིག་ཏུ་ཐབ་པོ། །བ་། བ་དྷོ༑། ཟབ་དྷཱ། བ་དྷཾ། བ་དྷོ། བ་དྷཱ། བ་དྷཱ། བ་དྷཱ་ཧཱུཾ། མ་དྷུ་བྷི། བ་དེ། བ་དྷུ་བྷཾ། བ་དྷཱ་ཤྩ། བ་དྷཱ། བ་དྷཱ་ཧཱུཾ། །བ་དྷུ་བྷྱ། བ་དྷཱ། བ་དྷོ། བ་དྷཱ་ནཾ། བ་དྷི། བ་དྷོ། ཥ་ཉིད་དོ། །བ་དྷཱ་ཥཱ། ཀུན་དུ་བོད་པ་ལ་སི་དབྱི་བ་དང་ཐུང་ངུ་ཉིད་དོ། །ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། །།བ་བྷཱུའི་སྒྲ་ནི་ལྕིའི་སྒྲ་དང་འདྲ་བར་ཤེས་པར་བྱ་སྟེ། །དབྱངས་དང་པོ་ཅན་ལ་ནི་ཨུ་ཝ་རུ་འགྱུར་ཞིང་། བྷཱུ་བྱིངས་བཞིན་ནོ་ཞེས་པས་བྱིངས་དང་འདྲ་བར་ཨཱིཏ་ཨུཏ་དག་གི་ཨི་ཡ་ཨུ་ཝ་དག་དབྱངས་ལའོ་ཞེས་པ་ཁྱད་པར་རོ། །བྷུ། བྷུ་ཝོ། བྷ་ཝ། བྷུ་ཝཾ། བྷུ་ཝོ། བྷུ་ཝ། བྷུ་ཝཱ། བྷུ་བྷཾ། བྷུ་བྷི། བྷུ་ཝེ། བྷུ་བྷཾ། བྷུ་བྷུ། བྷུ་ཝ། ་བྷུ་བྷཾ། བྷུ་བྱཱ། བྷུ་ཝཱ། བྷུ་ཝོ། བྷུ་ཏཱཾ། བྷུ་ཝཾ། བྷུ་ཝོ། །ཥ་ཉིད་དོ། །བྷུ་ཥཱ། ངོཀུན་དུ་བོད་པ་ལ། ཧེ་ཧྥུ། ཧེ་བྷུ་ཝོ། ཧེ་བྷ་ཝ། །།ཟོ་། ::བ་གཅོད་པར་བྱེད་པས་ན་ལུ་སྟེ། །སཾ་པད་ལ་སོགས་པ་ཉིད་ལས་བུད་མེད་ལ་ཀྭིཔ་འགྱུར་ཞིང་པ་ར་མའང་འདི་ཡིན་ལ་ལུའང་ངོ་ཞེས། །གཞི་མཐུན་པ་ཡི་ཚིག་དག་ནི། །བློ། །ཞེས་པ་ལ་སོགས་པའི་ཚིག་སྡུད་དོ། །པ་ར་མ་ལུ། པ་ར་མ་ལུ་ཝོ། ལྔ་པ་ར་མ་ལུ་ཝ། ཞེས་པ་ལ་སོགས་པའོ། །གང་གི་ཚེ་མི་་ཟད་པ་ལྡན་པ་དེའི་ཚེ། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་།
