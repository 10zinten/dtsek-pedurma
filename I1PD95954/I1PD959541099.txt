བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་ཡང་། འཆི་མེད་ཕྲེང་བར་གསུངས་པ། ་ཀཾ་དང་:ཁཱ་ར་ནི་བུད་མེད་དེ། ཨཔ་ནི་མང་པོ་རྣམས་ལ་བུད་མེད་མ་ཡིན་པ་ལའོ་ཞེས་སོ། ལ་། འདིའི་དོན་འདི་་ཡིན་ཏེ་ཀཾ་ཞེས་པ་མ་ནིང་ངོ༌། །ཁཱར་ཞེས་པ་བུད་མེད་དོ། ་། ཨཱཔ་རྣམས་ནི་་་མང་པོ་རྣམས་ལ་བུད་མེད་ལའོ་ཞེས་པའོ། །རིན་ཆེན་མཛོད་དུ་ནི། མ་ནིང་་གི་ལེའུར་བཤད་དེ། པཱ་ཐོང་། ་ཨར་ཎ། །།ཏོ་ཡ། ནི་ར་ཀཾ། ཁྲི་:ཝར། ཨཔ་རྣམས་མང་པོ་ལ་སྟེ། བུད་མེད་ལའོ་ཞེས་སོ། །་། དེའི་ལུགས་ཀྱིས་མ་་ནིང་གི་སྐབས་སུ་དཔེར་བརྗོད་པ་བྱས་སོ། །ལ་མཐའ་ཅན་རབ་ཏུ་མ་གྲུབ་པོ། །ཝ་མཐའ་དི་ཝའི་སྒྲ་སྟེ། སི་ལ་ཨོ་ཞེས་པ་ཝ་ཡིག་གི་ཨོ་ཉིད་དང་ཡ་ེཉིད་དོ། །ས་ཡིག་ནི་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ། །བྱོ། ཨུ་ད་ལོ། ་་་་་ཝ༑ཨམ་ལ་ཡང་ན་ཨཱའོ་ཞེས་རྣམ་པར་བརྟག་ཅིཔས་ཨོཾ་ལ་ཝ་:ཡིག་ཨཱ་ཉིད་དོ། །:ཏྭཾ། ་དི་ཝམི། དི་ལོ། དི་ཝ། དི་ཝའི་གསལ་བྱེད་ལ་ཨུཏའོ་ཞེས་པས་གསལ་བྱེད་དང་པོ་ཅན་ལ་དི་ཝའི་ཝ་:ཨུཏ་དུ་ཨེ་འགྱུར་རོ། །ཏྭ་བྷཱུམ། སདྷུ་བྷིཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་དྲོ། །ཧེ་དི་ཝ། ཤ་་མཐའ་ཅན་:ད་ཤའི་སྒྲ་སྟེ། དེ་སྟོན་པར་བྱེད་དོ་ཞེས་པས་དི་ཤི་ལས་རི་ཏི་:ཤ་ད་དྷི་ཥ་སྲི་ཛ་དི་ཤ་ཞེས་པས་ཀྭིཔའོ། །ཙ་སྡེ་དང་:དྲི་ཤ་ལ་སོགས་པ་རྣམས་་ཀྱིའང་ངོ་ཞེས་པས་ག་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དང་པོ་དང་གསུམ་པ་དག་གོ། །དིཀ། དིག། དི་ཤོ། དི་ཤ། །དི་ཤཱི། དི་ཤོ། དི་ཤ། :དཱི་ཤཱི། །དིདྡྷཾ། དིགྷི། །ཞེས་པ་ལ་སོགས་པའོ། །བདུན་པའི་མང་པོའི་ཚིག་ལ་ག་ཉིད་དུ་བྱས་པ་ན་སྒྲ་མེད་པ་ལ་དང་པོའོ་ཞེས་པས་ཀ་ཉིད་ཀྱི་ཀྱང་ངོ༌། །དིཀླུ། དིཁྲུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་དིཀ། ཧེ་དིག། ཧེ་དི་ཤོ། ཧེ་དི་ཤ། །དེ་བཞིན་དུ་དྲི་ཤ་དང་བི་ཤ་ལ་སོགས་པ་རྣམས་སོ། །།
