ཛོ་ག། །ཛཾ་ག་ནཱ། ཙྪཱ། ཛཾ་གམྨ། ་ཡམ། ཡཱཾ་ཡ་མཱི་ཏི། ཡཾ་ཡནྟི། ཡོ་ཡ་ཏ། ཡཾ་ཡ་མ་ཏི། ཡཾ་ཡ་མཱི་ཏི། ཡཾ་ཡཾ་སི། ་ཡཾ་ཡ་ཐ། ཡཾ་ཡ་ཐ། ཡཾ་ཡ་མཱི་མི། །ཡཾ་ཡནྣཾ། ཡཾ་ཡ་ནཱ། །ཡཾ་ཡཙྪ། ཧ་ན། ཨ་བྱཱ་ས་ལས་ཀྱང་ངོ་ཞེས་པས་ཨ་ཧྥུ་ས་ལས་ཕ་རོལ་གྱི་ཧའི་གྷ་ཉིད་དོ། །ཛཾ་གྷ་ནཱི་ཏི། ཛཾ་གྷནྟི། ཛཾ་གྷ་ཏ། །སྔ་མ་བཞིན་ཨུ་པ་དྷཱ་དབྱིའོ། །ཛཾ་གྷཱ་ན་ཏི། ཛཾ་གྷ་ནཱི་ཏི། ཛཾ་གྷ་སི། ཛཾ་གྷ་ཐ། ་་ཛཾ་གྷ་ནི་མི། ་ཛཾ་གྷནྣཾ། ཛཾ་གྷ་ནཱ། ཛཾ་གྷརྨ། །ཤ་མུ། ཤཾ་ཤ་མི་ཏ། ་ཤོ་ཤནྟི། དྷུཊ་དང་པོ་ཡོན་ཏན་མེད་པ་ལ་སྔ་མ་་བཞིན་རིང་པོའོ། །ཤཾ་ཤཱནྟ། ཤཾ་ཤཱ་མ་ཏི། ཤཾ་ཤ་ཡཱི་ཎི། ཤཾ་ཤ་སི། ་ཤྲཱི་ཤཱཏྨ༑་ཤཾ་ཤཱནྡྲ། ཤཾ་ཤ་མཱི་མི། ་སྔ་མ་བཞིན་མའི་དབྱངས་དང་བཅས་་པའི་ན་ཡིག་གོ། །དེའི་ཨ་ཡིག་དབྱིའོ། །གཽ་ཤཱམྦྷི། །ཤོ་ཤནཱ། །ཤཾ་ཤམྨ། ད་མུ། ཏནྟ་མཱི་ཏི། ཏནྟཱནྟ། ཏནྟ་མ་ཏི། ཏནྟ་མཱི་ཎི། ཏན་སི། ཏནྟཱནྡྲ༑དནྟནྠ། ཏནྟ་མཱི་མི། ཏནྟནཱ། ཏནྟནྣ། :བྷནྡ། བི་བྷ་ནི་ཏི། ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། ་། ཁཾ་བྷཎྜི། ?ཐཾ་བྷ་ཎྞ། །ཌཐཾ་བྷ་་ཎ་ཏི། བི་བྷ་ཎི་ཎི། ཐཾ་བྷ་ཎིས། ཎ་ཉིད་ཀྱི་བར་ཆོད་པ་ལ་ཊ་སྡེ་ལས་:སའི་ཥ་ེཉིད་དོ། །ཐཾ་བྷནྟི། ཐཾ་བྷཎྞཉྫ། །ཐཾ་བྷཉྫི། ཐཾ་བྷ་ཎི་མི། ཐཾ་བྷཉྫལ། བོ་བྷ་ཎ། ཐཾ་བྷཉྫ། ར་ཎ། རཾ་ར་ཎཱི་ཏི། རཾ་རཎྜི། རཾ་རཎྞ། རཾ་ར་ཎ་ཏི། རཾ་ར་ཎ་སི། ཎ་ཉིད་ཀྱིས་བར་ཆོད་པ་ལ། རོ་ར་ཎི། རཾ་རཏྟཱ། རཾ་རནྟུ། །ར་ར་ཎཱི་མི། རཾ་རལླི། རཾ་ར་ཎི། རཾ་རཙྪ། ཛ་པ། ཛ་པ་སོགས་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཛ་པ་ལ་སོགས་པ་རྣམས་ཀྱི་ཨ་ཧྥུ་ལིསའི་མཐར་STCSSKIKEBIKENKERSKELIཛ་པཱི་ཏི། ཛཾ་ཛ་པིས། ཛཾ་ཛདྨ། ཀམཛཾ་ཛ་པཱི་མི། ཛེཛམྦྷི། ཛཾ་ཛ་སཱི། །།
