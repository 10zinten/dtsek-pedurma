བསྟན་འགྱུར། སྒྲ་མདོ། ཀིའི:དེ་ནས་ཨཱ་ཤི་སའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །ས་ཛུ་ཥ་དང་། ཨཱ་:ཤི་ཥ་དག་གི་རའོ་ཞེས་པས་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་དག་ལས་ས་ཡིག་ར་རུ་འགྱུར་རོ། །།ཨི་ཨུར་དག་གི་ཨཱིར་ཨུར་དག་གོ། །རེ་ཕ་དང་ས་དག་གི་རྣམ་་པར་གཅད་བྱའོ། །།ཨཱ་ཤི། ཨཱ་ཤི་ཏོ། །ཨཱ་ཤི་ཥ། ཨཱ་ཤི་ཎཾ། ཨཱ་ཤི་ཤོ། ཨཱ་ཤི་ཥཱ@། །ཨཱ་ཤི་ཥཱ། །ཨཱ་ཤིདྡྷཾ། ཨཱ་ཤིཙྪི། །ཞེས་པ་ལ་སོགས་པའོ། །བདུན་པའི་མང་པོའི་ཚིག་ལ་རྣམ་པར་གཅད་བྱ་བྱས་པ་ན་ཤ་དང༌། །ཥ་་་དང་:ས་ལའང་ཕ་རོལ་གྱི་གཟུགས་སུ་ཡང་ནའོ། །ཨཱ་ཤིཀྵ། རིཨཱ་ཤིཁྱུ། ཀུན་དུ་བོད་པ་ལ་ཁྱད་པར་མེད་དོ། །ཨུཏེ་སྔ་མ་ཅན་སྣེ་ཧི་ལས་རི་:ཏཱ་ཀི། ད་དྷ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཀྭིཔ་དང་ཉེར་བསྒྱུར་གྱི་མཐའ་དབྱི་བ་དང་ཥ་་ཉིད་ཚིག་ཕྲད་དུ་བྱའོ། །དེ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །ཙ་སྡེ་དང་དྲི་ཤ་ལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ག་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ན་སྟེ་དང་པོ་དང་གསུམ་པ་དག་གོ། །:：ཨུཙྪིཀ། ཨུཙྪིག། །ཨུཎྜི་ཧོ། །ཨུཎྜི་ཧེ། ཨུཎྜི་ཧཾ། ཨུཎྜི་ཧོ། ཨུཎྜི་ཏཿ། ཨུཎྜི་ཧཱ། །e:：ཨུཙྪི། ་ཨུཙྪིནྡྷི། ཞེས་པ་ལ་སོགས་པའོ། །སྣུ་ཧི་ལས་སམ་པད་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ཀྭིཔའོ། །:དེ་ནས་མི་ཏྲ་དྷུ་ཧའི་ྋསྒྲ་དང་འདྲ་:བར་བྱ་སྟེ། སྣུཀ། ལྕུག། སྣུ་ཧོ། སྐུ་ཧ༑སྣུ་ཧཾ། སྐུ་ཧོ། སྐུ་ཧ། །སྣུ་ཧཱ། Ti:སྣུདྡྷཾ། སྣུཊྚ། སྣུབྷི། སྣུབྷིཙྪེ༑་འཞེས་པ་ལ་སོགས་པའོ། ་། ཨུ་པ་སྔ་མ་ཅན་ན་ཧི་ལས་ཀྭི་པའོ། ་། ན་ཧི་བྲི་ཏི་སྲྀ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཚིག་སྔ་མའི་རིང་པོ་ཉིད་དོ། །དེ་ནས་སི་ལ་སོགས་པ་རྣམས་སོ། །རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་རྣམས་ལ་:ཨཎ་ཌཱ་ཧ་དང་ན་ཧི་དང་ཝནྟི་རྣམས་ཀྱི་:འང་ངོ་ཞེས་པས་རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ་ཧ་ཡིག་ད་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ན་སྟེ་དང་པོ་དང་གསུམ་པོ་དག་གོ། །_།
