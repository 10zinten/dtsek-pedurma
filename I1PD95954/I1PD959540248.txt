སྐྱེ་བྱིངས་ཀྱི་ཚོགས། ས་སྐྱ་ང་ད་སྦྱིན་པ་ལའོ། །།།:ཧ་ད་བཤང་བ་འདོར་བ་ལའོ། །ཥཱ་ད་དང་སཱ་ད་དང་སཱཌྷ་རོ་མྱང་བ་ལའོ། །།ཨུདྷ་ཚད་ལ་སྟེ་རྩེ་བ་ལ་ཡང་ངོ་། །ཀུནྡ་དང་ཁྲུནྡ་་་དང་:གུ་ད་རྩེ་བ་ཉིད་ལའོ། །ཨོཥཱ་ད་འགྱུར་བར་བྱེད་པ་ལའོ། །།ཧཱ་ད་མི་གསལ་བའི་སྒྲ་ལའོ། །གྷཱ་དཱི་ཨིབདེ་བ་ལ་ཡང་ངོ༌། །པཌྷ་སྨད་པའི་སྒྲ་ལའོ། །ཡ་་་ཏཱི་རབ་ཏུ་འབད་པ་ལའོ། བ་། ཡུ་ཏྲ་དང་ཛུ་ཏྲེ་འཆད་པ་ལའོ། །བི་ཤྲཱི་དང་བེ་ཤྲཱི་སློང་བ་ལའོ། །ནཱ་དྷཾ་དང་ནཱ་ཤྲཱི་ཉེ་བར་གདུང་བ་དང་དབང་ཕྱུག་དང་ཤིས་པ་རྣམས་ལ་ཡང་ངོ༌། །།ཤ་ཐི་བསྙེན་པ་ཉིད་ལའོ། །གྲ་ཐི་དང་བ་ཀི་གྱ་གྱུ་ལའོ། །ཀཏྠ་མངགས་པ་ལའོ། །ཤི་ཀྲྀ་འཁྱུང་བ་ལའོ། །ལོ་ཀྲྀ་དང་ལོ་ཙཱིལྟ་བ་ལའོ། །ཟློ་ཀྲྀ་སྡུད་པ་ལའོ། །དྲེ་*ཀྲྀ་དང་དྷེ་ཀྲྀ་སྒྲ་དང་སྤྲོ་བ་དག་ལའོ། །རེ་ཀྲྀ་དང་ཤ་ཀི་དོགས་པ་དག་ལའོ། །ཨ་ཀི་མཚོན་བྱེད་ལའོ། །མ་ཀི་རྒྱན་ལའོ། ་། ཀ་ཀ་་་ཆགས་པ་ལའོ། ཌ། ཀུ་ཀ་དང་བྲྀ་ཀ་ལེན་པའོ། །ཙ་ཀ་ཚིམ་པ་ལ་སྟེ་ཐོགས་པ་ལ་ཡང་ངོ༌། པ། །ཀ་ཀི་དང་བ་ཀི་དང་སཱ་ཀི་དང་ཏྲ་ཀྲི་དང་རྣོ་ཀྲྀ་དང་རྡོ་ཀྲྀ་དང་ཀྐ་དང་བསྐ་དང་མསྐ་དང་ཏི་ཀྲྀ་དང་ཊི་ཀྲྀ་དང་ཊཱི་ཀྲྀ་དང་སེ་ཀྲྀ་དང་གླེ་ཀླུ་དང་ཤ་ཀྲྀ་དང་ཧྥ་ཀི་དང་ར་གྷི་དང་ལ་གྷི་འགྲོ་བའི་དོན་ལའོ། །།ཨ་གྷི་དང་བ་གྷི་འགྲོ་བ་་་་འཕེན་པ་:ལའོ། །ཎིམ་གྷི་རྒྱུན་པོ་འགྱེད་པ་ལའོ། །རཱ་ཤྲཱི་དང་ལཱ་གྷི་ནུས་པ་ལའོ། །དཱ་གྷི་ཉོན་མོངས་པ་ལའོ། །ཤཱ་ྋགི་མངགས་པ་ལའོ། །ཥ་ཙ་འཁྲུད་པ་ལའོ། །ཤ་ཙ་གསལ་བར་བརྗོད་པ་ལའོ། །ཀ་ཙ་འཆིང་བ་ལའོ། །མ་ཙ་དང་མུ་ཙི་རྙོག་པ་ལའོ། ཟླ་། མ་ཙི་འཛིན་པ་དང་བསྒྲེང་བ་དང་མཆོད་པ་རྣམས་ལའོ། །པ་ཙི་་་གསལ་པོར་བྱེད་པ་ལའོ།  ོ། ཙ་རབ་ཏུ་དང་བ་ལའོ། །ཤཱ་*ཙ་དང་ཤཱ་ཙི་དང་ཨི་ཛི་དང་ཨི་ཛ་དང་རཾ་ཛ་འགྲོ་བ་ལའོ། །ཛྷ་ཛ་དང་བྷཾ་ཛི་རྗོད་པ་ལའོ། །ཙཙྪ་དང་ཀ་ཙི་དང་ཨེ་ཛི་དང་བྷཾ་ཛྲ་དང་བྷཱ་ཝིཛྲ་འབར་བ་ལའོ། །ཏི་ཛ་རྣོ་བ་ལ་།
