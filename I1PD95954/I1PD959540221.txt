ཁབསྟན་འགྱུར། སྒྲ་མདོ། ༼ལིའི:བ་ཁ་བྱ་སྟེ། ཨ་ཨ་ད་ལ་ཏི་སྦྱར་བས། །བར་དུ་ཨན་སྦྱར། ?མདོ་འདིས་ཕྱི། །རི་ཥ་ཏ་ཝརྒའི་མདོས། དས་ཏ་ཐོབ་སྟེ་སྦྱར་རོ། །ཨཏྟིའོ། ལྕི་ཁྲུ། །ཧ་ན་ལ་ཏི་་དང་ཨན་སྦྱར་བ་གོང་བཞིན་ནོ། །ཧིནྟའོས། ད་། །དྲེ་ཥ་ལ་ཏི་སྦྱར་བས་སྔ་མ་་་བཞིན། ་ཏིས་ཊི་ཐོབ་བོ། །དྭེབྦིའོ། །ཨ་ད་ནི་བཟའ་བའོ། །ཧ་ན་ནི་འཚེ་བའོ༑། དྭེ་ཥ་ནི་སྡང་བའོ་། :། །།ག་བཞི་ཡིན་པ། །པ་བྱ་ལ། ཛུ་ཧོ་ཏཱ་དེ། 《ཧུ་ནི་སྦྱིན་པ་དང་བཟའ་བ་སྟེ། དེ་ལྟ་བུ་ལ་སོགས་པའི་དྷཱ་ཏུའི་ཚོགས་ཀྱི་ཕ་རོལ་གྱི་བི་ཀཱ་ར་ཎ་ཕྱི་བར་བྱའོ། །ཧུ་ལ་ཏི་སྦྱར། བར་དུ་ཨན་སྦྱར། མདོ་འདིས། ཧུ་གཉིས་པ་བྱས། ：ཧས་ཛ་ཐོབ། །ཨུ་ཨོར་གྱུར། ཛུ་ཧོ་ཏིའོ། །བྷི་ལ་ཏི་དང་ཨན་སྦྱར་བས་སྔ་མ་བཞིན། བྷས་བ་ཐོབ། ཁོབི་བྷེ་ཏིའོ༑། ཧཱི་ལ་ཏི་དང་ཨན་སྦྱར། ཤྲཱི་གཉིས་པ་སོགས་སྔ་མ་བཞིན། །ཛི་ཧྲེ་ཏིའོ༑་། ཧུ་ནི་སྦྱིན་པ་དང་བཟའ་བའོ། །།ཉི་བྷི་ནི་:འཇིགས་པའོ། །།ཧྲི་ནི་་ངོ་ཚ་བའོ། །ཝཱ་ཙེ་ཀྲི་ཡི་ཏ་སྱ། །ཙེ་ཀྲི་ཡི་ཏའི་ཡ་ཕྱི་བའམོ། །༼བྷ་ལ་ཡ་་་སྦྱར་བ་ཕྱི་ནས། པྲ་ཏྱ་ཡ་ལུས་ཏམ། པྲ་ཏྱ་ཡ་ལ་ཀཱ་ཎའི་མདོས་བྷུ་གཉིས་པ་་་བྱས་བྷས་བ་ེཐོབ། ་ཨིཊ་སྦྱར། ཨུ་ཨོར་གྱུར། ཨོ་ཨཝ་དང་སྦྱར་རོ། །བོ་བྷ་ལྕི་ཏིའོ། །ར་ཊ་ལ་ཡ་སྦྱར་ནས་ཕྱི་བ་སོགས་སྔ་མ་བཞིན། རྭ་ར་ཌི་ཏིའོ། །ཡང་ན་བྷུ་ལ་ཡ་སྦྱར་བས། ཡ་མ་ཕྱི་སྟེ། ་བྷུ་གཉིས་པ་ལ་སོགས་པའོ། །བ་བོ་དྷུ་ཡ་ཏེའོ། །ར་ཊ་ལ་ཡ་སྦྱར་བས་སྔ་མ་བཞིན། ''དཱིགྷོ་ནཱ་ག་མནྟ་སྱའི་མདོས་ཨ་བྱཱ་ས་བསྲིངས་པའོ། །རཱ་ར་ཊུ་ཏེའོ། །:ཨིནྡྲ་ད་པ་ྋཝ་ཏི་བྷུ་:བྷུསི་ཙིཔ་རསྨི། ཨིན་ནི་འགྲོ་བ་དང་། སྦྲ་ནི་འགྲོ་བ་ལ་མི་འཇུག་པ་དང༌། དཱའི་མིང་ཅན་དང་། །པཱ་ནི་བཏུང་བ་དང༌། ་བྷུ་ནི་ཡོད་པ་ཞེས་བྱ་བ་འདི་་་་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་སི་ཙི་ཕྱི་བ་སྟེ། ཕ་རོལ་ན་གཞན་གྱི་ཚིག་ཅན་ཡོད་ནའོ། །།
