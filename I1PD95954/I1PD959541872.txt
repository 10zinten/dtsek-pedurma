ང་རྒྱ་ནག་n་ཟླ་བ་ལ་བྱེད་་་་་་པའི་ས་ཐོབ་པ་ཆེ་ཁག་ག་་ཀྱི་བར་གྱི་ཁ་ལ་རྒྱས་ཀྱི་བྱེད་པས་ཁ་ཆུ་བདེ་འག་ན་ད་ལྟའི་བར་ལ་གནའ་བོ་ནི། ::བ་་བར། :ཀ:་་པལོ་དོན། ༄༅། །སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུ་ཞེས་ཞུ་བ་པོའི་བ་བ་བཞུགས་པའི་བྱ་བའི་འགྲེལ་པ། བ་བ་ན་ད་ལྟ་ཀཚོན། པར་ཁ་དེ་འབའ་ཞིག་པ་བཏོན་པ་འདྲ་། ག་འདུ་ཤེས་བྱ་ལ་མ་གས་གང་གག་ནས་ཁཚོག་པ་འུ་ཤི་བ་དགའི་དགའ་བ་བཞི་ལ་ད་ཁོ་བྱ་བ་རི་ན་ཏེ་པ། གཏི་༄༅། །རྒྱ་གར་སྐད་དུ། ཝ་ཙ་ན་མུ་ཁ་ཡུ་དྷོ་པ་མ་ནཱ་མ་བྲྀཏྟི། བོད་སྐད་དུ། སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུ་ཞེས་བྱ་བའི་འགྲེལ་པ། སྨྲ་བའི་་མཆོག་ཐམས་ཅད་མཁྱེན་པ་ལ་ཕྱག་འཚལ་ལོ། །འདིར་བསྟན་བཅོས་རྩོམ་པར་་བཞེད་པས། བར་ཆད་སྤང་བར་བྱ་བ་དང༌། ཡ་རབས་ཀྱི་ཚུལ་དང་མཐུན་པའི་ཕྱིར། དང་པོར་ཕྱག་འཚལ་ཞིང༌། དགོས་པ་ཁོང་དུ་ཆུད་པར་བྱ་བའི་ཕྱིར། གང་ཞིག་བཤད་པར་བྱ་བར་དམ་བཅས་ནས། བསྟན་བཅོས་ཀྱི་་གཞུང་ཉིད་ལེགས་པར་བསྟན་པ་དང༌། བསྟན་བཅོས་བརྩམས་པ་ཡང་དོན་ཆེན་པོར་འགྱུར་བར་བྱ་བའི་ཕྱིར། དགེ་བའི་རྩ་བ་བསྔོ་བ་དང་བཅས་པ་སྟོན་པར་་་མཛད་དོ། །དེ་ལ་འདིའི་གཞུང་ཉིད་ཀྱང་ཡི་གེའི་ཚོགས་དང༌། མིང་གི་ཚོགས་དང་། ཚིག་གི་ཚོགས་རྣམས་སྟོན་ཏེ། དེ་དག་གི་ཡན་ལག་ཇི་སྐད་དུ་བསྟན་པ་རྣམས་ཀྱང་མདོར་བསྡུ་ན། དེ་དག་གི་རྒྱུ་དང༌། ཤརང་གི་ངོ་བོ་དང་། །
