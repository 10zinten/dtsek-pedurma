ཙའི་ཀི་ཉིད་དོ། །ཨི་ཡིག་བརྗོད་པ་ནི་རྣམ་དབྱེ་དང་བཅས་པ་བསྒྱུར་བ་སྟེ་་དབྱངས་མ་ཡིན་ནོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །བསྟི། བཀྟབ་ཙནྟི། གཞན་རྣམས་ཀྱི་བ་ཙནྟི་ཞེས་མངོན་པར་བརྗོད་པར་མི་བྱའོ་ཞེས་གསུངསི། བཀྲི། །བཀྑ། བཀྑ། བཙྪི། བཙྪ། བཙྪ༔རུ་དིར་མཆི་མ་འཛག་པ་ལའོ། །རུད་སོགས་ལས་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་གསལ་བྱེད་་དང་པོ་ཡ་ྋཡིག་མ་ཡིན་པའི་དང་པོར་ཨིཊ་ཨཱ་ག་མའོ། ་པ། །ཊ་ཡིག་ནི་ཨིཊ་ལ་འང་ཞེས་པ་ལ་སོགས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། ་། ཨུ་པ་ད་ཡང་བའི་ཡོན་ཏན་ནོ། །རོ་དི་ཏི། རུ་དི་ཏ། ་རུ་དནྟི། རོ་དི་ཎི། རུ་དི་ཐརུ་དི་ཐ། རོ་དི་མི། རུ་དི་བ། །ལྷོ་རུ་དི་མ། །ཨ་ནུ་རོ་དི་ཏི། །ཉིཥ་པ་ཉལ་བ་ལའོ། །:སྭཱ་པ་ཏི། སྭ་པ་ཏ་སྭ་:པནྟི། སྭ་པི་སི། སུ་པི་ཐ། སྭ་པི་ཐ། ་སྭཱ་པི་མི། སྭཱ་པི་བ། སཱ་པི་མ། ཤྭ་ས་སྲོག་ལའོ། །ཤཱ་སི་ཏི། །ཤཱ་སི་ཏཿཤཱ་སནྟི། ནི་ཤཱ་སནྟི། ནི་ཤཱ་སི་ཏི། 《ཨུཙྪཱ་སི་ཏི། ཨུཙྪཱ་སི་ཏི། ཨ་ན་ཡང་ངོ༌། །ཨ་ནི་ཏི། ་་ཨ་ནི་ཏ་ཨ་ནནྟི། པྲ་སྔོན་དུའོ། །མཚུངས་པ་འདྲ་བའི་ཡི་གེ་ལ་རིང་པོར་འགྱུར་ཏེ། ཕ་རོལ་ཡང་དབྱིའོ། །དེ་ལ་ཡང་སྨོས་པ་ལས་ཨ་:ནི་ཏའི་མཐའི་ཡང་ངོ་ཞེས་པས་ནའི་ཎ་ཉིད་དོ། །པྲ་ཎི་ཏི། ཛཀླུ་ཟ་བ་དང་དགོད་སུཔ་དག་ལའོ། །།ཛཀྱི་ཏི། ཛཀཱི་ཏ་ཛཀ་སོགས་ཀྱང་ངོ་ཞེས་པས་:ཨ་བྱཱ་སྟའི་མིང་ཅན་དུ་ཡོད་པ་ལ། །ཨ་བྱཱསྟ་ལས་ཨནྟིའི་ན་དབྱིའོ་ཞེས་པས་ན་དབྱིའོ། །ཛཀཱ་ཏི་ཞེས་པ་རུད་ལ་སོགས་པ་རྣམས་སོ། །ཛཱ་གྲི་གཉིད་སད་པ་ལའོ། །ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཛཱ་གཏི། ཛཱ་གྲི་ཝེཏའི་ཡི་གེ་རའོ་ཞེས་པས་རི་་ཡིག་གི་ར་ཉིད་དོ། །ཛཱ་གྲ་ཏི། ཛཱ་གནྟི། ཛཱ་གྲི་ཐ། ཛཱ་གྲི་ཐ། ཛཱ་གརྨི་ལོ། ཛཱ་གྲི་བ?ཏོཛཱ་གྲི་མད་རི་དྭ་ངན་འགྲོ་ལའོ། །ད་རི་དྭ་ཏི། ད་རི་དྭ་་།
