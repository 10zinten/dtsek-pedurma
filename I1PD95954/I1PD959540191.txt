བསྟན་འགྱུར། སྒྲ་མདོ། ལིའིགསྦྱར་བས་ར་བྷ་གཉིས་པ་བྱ་སྟེ་ཕྱི། གོང་དུ་ཨཱང་བྱས་ང་ཕྱི་བྷས་པ་ཐོབ་སྟེ་ར་ཨིའོ། །ཨ་་རིམྨ་ཏིའོ། །ལ་བྷ་ལ་སན་སྦྱར་བས་གོང་བཞིན་ལི་པས་ཏིའོ། །:：ཤཀླུ་ལ་སན་སྦྱར་བས་ཤཀ་གཉིས་པ་བྱ་སྟེ་ཕྱིས། ཌ་ཨི་ཐོབ་ནས་:ས་ཥའོ། །ཤིཀཱ་ཏིའོ། །པ་ཏ་ལ་སན་སྦྱར་:བ་སོགས་གོང་བཞིན། པིཏྠ་ཏེའོ། །ཨིན་དང་བཅས་པའི་སན་ལ་ནི་མི་འཐད་དེ། བཏ་ལ་སན་དང་ཨིན་སྦྱར་བས་ཨ་བྷུ་ས་དབྱི་བ་དང་། ཨིས་ཐོབ་པ་མ་བྱུང་བས་སོ། །པི་པ་ཏི་ཥ་ཏིའོ། །།ར་དྷ་་ནི་འཚེད་པའི་ཞེས་བྱ་བ་འདིའི་དབྱངས་ལ་སན་ཡོད་ན་ཨི་ཉིད་ཐོབ་པར་་་གསུངས་ཏེ་ར་དྷ་ལ་སན་སྦྱར་བས། གོང་དུ་པྲ་ཏི་བྱས་ཨ་ཧྥུ་ས་དབྱི་བ་དང་ཨིས་ཐོབ༑པྲ་ཏི་རི་ཏ་ཉེས་ཏིའོ། །ཨཱརྣོ་ཏེ་རཱི། ཥཨཔལཾ་ཁྱབ་ཅེས་པའི་དྷཱ་ཏུ་གང་ཡིན་པའི་དབྱངས་དེའི་ཕ་རོལ་ན་སན་ཡོད་ན་ཨི་ཉིད་ཐོབ་པ་དང༌། །ཨ་བྷུ་ས་ཡང་དབྱི་བའོ། །ཨཔ་སན་སྦྱར་ན་ཕྱི། གོང་དུ་ཨ་གཉིས་པ་བྱས་ཏེ་ཕྱི། ཨས་ཨི་ཐོབ་པའོ། །ཨིཔ་ས་ཏིའོ། །ཛྷ་བི་དང་རི་དྷའི་དབྱངས་ཀྱི་ཡང་ཕ་རོལ་ན་སན་ཡོད་ན་གོ་རིམས་ཇི་ལྟ་བ་བཞིན། ཨཱི་ཐོབ་པའམ། ཨཱིར་ཐོབ་པར་ཡང་གསུངས་ཏེ། ཛྷ་ལ་སན་སྦྱར་བས་གོང་དུ་ཛྷ་གཉིས་པ་བྱས་ཏེ་ཕྱི། Kས་ཨི་་་་ཐོབ་ཛཱི་པ་ས་ཏིའོ། །ཛྷཱནི་ཁོང་དུ་ཆུད་པའོ། །རཾ་དྷཱ་ལ་སན་སྦྱར་བ་གོང་དུ་རི་བྱས་ཏེ་ཕྱི། རཾས་ཨཱིར་ཐོབ། དྷས་:ཏ་ཐོབ་ཏེ་སྦྱར། ཨིཎྚཏིའོ། །རཾ་དྷུ་ནི་འཕེལ་བའོ། །དཾ་:བྷི་རི་རིཙྩ། གང་དག་དན་བྷིའི་དྷཱ་ཏུའི་དབྱངས་་་་འདིའི་ཕ་རོལ་ན་སན་ཡོད་ན་ཨི་ཉིད་དང༌། ཨཱ་ཉིད་ཐོབ་པ་དང༌། ་་ཨ་བྱཱ་ཀྱིས་ཡང་དབྱིའོ། །དན་བྷི་ལ་སན་སྦྱར་བས་ད་གཉིས་པ་བྱ་སྟེ་ཕྱི། དས་དྷ་ཐོབ། :བྷས་པ་ཐོབ། 《མདོ་འདིས་རིམ་པ་བཞིན། །དྷས་ཨི་དང༌། ཨཱ་ཙེཐོབ་པོ། །དྷིམྨས་ཏི། དྷིསྶ་ཏིའོ། །དི་གི་ད་ཡ་ཏེ་པ་:རོ་གྷ་ཡ་མ། དེང་རཀྵ་ཎེ་—།
