བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཔཉྩཱ་ཤ་ཏ། པཉྩཱ་ཤ་ཏ། །ཨེ་ཀ་པཉྩ་ཤ་ཏ། དྭ་པ་ཤ་ཏ། དྭི་པཉྩ་ཤ་ཏེ། ཏྲ་ཡཔཉྩ་ཤ་ཏ། ཏྲི་པཉྩ་ཤ་ཏ། ཙ་ཏུ་ཐཝཱ་ཤ་ཏ། པཉྩ་པཉྩཱ་ཤ་ཏ། ཥ༌༌ཊ་པཉྩ་ཤ་ཏ། སཔ་པཉྩཱ་ཤ་ཏི། ཨཊྚ་པཉྩཱ་ཤ་ཏི། ཨཏྟཱ་པཉྩ་ཤ་ཏའོ། །བློ། ཨུ་ན་ཥ་ཤྲཱིའི་སྒྲ་ནི་བི་ཤ་ཏའི་སྒྲ་དང་འདྲའོ། །དེ་བཞིན་དུ། ཨེ་ཀེ་ན་ཥཉྩི། །ཨེ་ཀཱནྣ་ཥཉྩི། ཨེ་ཀཱ་དྷུ་ཥ། ཥཊཱི། ཨེ་ཀ་ཥཊྚི་དྷཱ་ཥཊཱི། དྷཱི་ཥཥཱི། ཏྲ་ཡཥཊྚི། །ལ་ཏྲི་ཥ། ཙ་ཏིུཥཊྚི། པཉྩ་ཥཉྩི། ཥཊ་ཥཝི། སཔྟ་ཥཉྩི། ཨཥྛ་ཥཝི། ཨཊྚ་ཥཉྩི། ཨཱ་ན་:སཙྩི་ཏ། ཨེ་ཀཱནྣ་སསྟི། ཡི། ཨེ་ཀཱ་དྷུ་སཔཏི། སབྷ་ཏི། ཨེ་ཀ་སཔྟ་ཏི། དྭ་སཔྟ་ཏི། དྲི་སཔྟ་ཏི། ཏྲ་ཡསཔ་ཏི། ཏྲི་་སཔྟ་ཏི། ཙ་ཏུསཔྟ་ཏི། པཉྩ་སཔྟ་ཏི། ?ཥཊ་སབྷ་ཏི། སཥ་སབྷ་ཏི། ཨཊྚ་སཔ། ཨཊྚ་སཔྟ་ཏི། ཨུ་ན་ཤཱི་ཏི། ཨེ་ཀེ་ན་ཤཱི་ཏི། ཨེ་ཀཱརཱ་ཤཱི་ཏི། :：ཨེ་ཀཱ་དུ་ཤཱི་ཏི། ཨཱ་ཤཱི་ཏི། ཨེ་ཀཱ་ཤཱི་ཏི། དཱ༌ཤཱི་ཏི། དྲིཤཱི་ཏི། ཙ་ཏུར་ཤཱི་ཏི། :ཥ་ཊ་ཤཱི་ཏི། སཥ་ཤཱི་ཏི། ཨཏྟཱ་:ཤཱི་ཏི། ཨཏྟཱ་ཤཱི་ཏི། ཨུ་ཏི་ནན་ཝ་:ཏི། ཨེ་ཀེ་ནན་ཝ་ཏི། ་་ཨེ་ཀཱནྣ་ན་ཝ་ཏི། ཨ། ཨེ་ཀཱ:：དྲུ་ན་ཝ་ཏི། ན་ཝ་ཏི། ཨེ་ཀཱ་ན་ཝ་ཏི། དྷཱ་ན་ཝ་ཏི། དྲི་ེན་ཝ་ཏི། ཏྲ་ཡོ་ན་ཝ་ཏི། ཏྲི་ན་ཝ་ཏི། ཙ་ཏུར་ན་ཝ་ཏི། པཉྩ་ན་ཝ་ཏི། 《ཥཎྜ་ན་ཝ་ཏི། ་༼སལྟ་ན་ཝ་ཏི༑ཨཊྚ་ན་ཝ་ཏི། ཨཥྛ་ན་ཝ་ཏིའི་སྒྲ་རྣམས་སོ་། :། བྷང་། མཚོ་དེཊ་ཨུ་ན་ཤ་ཏའི་སྒྲ་ནི་མ་ནིང་གི་རྟགས་ཅན་ཏེ། ཝ་ནའི་སྒྲ་དང་འདྲའོ། །སྤུ་ན་ཤ་ཏཾ༄༅། གང་གི་ཚེ་ཚོགས་པ་གཉིས་ལ་སོགས་པ་བརྗོད་པ་དེའི་ཚེ་གཉིས་་ཀྱི་ཚིག་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །དྲེ་ཨུ་ན་ཤ་ཏེ། ཐ་:ཏྲེ་ཏྱཱ་ན་ཤ་ཏ་ནི་ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཨེ་ཀོ་ན་ཤ་ཏ། —ཨེ་ཀཱནྣ་ཤ་ཏ། ཨེ་ཀཱ་དྷུ་ཤི་ཏ༑ཤ་ཏ། ས་ཧ་སྲ་ཨ་ཡུ་ཏ། །ནི་ཡུ་ཏ། པྲ་ཡུ་ཏ་སོགས་རྣམས་སོ། །།
