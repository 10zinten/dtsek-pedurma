དེ་ཝཱ་ག་ཏཱ། བྷོ་ཡཱ་སི། བྷ་གོ་བྲ་ཛ། །ཞེ་ཨ་གྷོ་ཡ་ཛ། དྲང་སྲོང་མཆོད། དབྱི་བ་བཟུང་བ་ནི་ཡ་བ་ཞེས་པ་མི་འཇུག་པའི་དོན་ནོ། །ནཱ་མའི་ཕ་རོལ་དུ་རའོ། །ནཱ་མི་ཅན་གྱི་ཕ་རོལ་དུ་ཚེག་དྲག་རར་འཐོབ་སྟེ་ལྟོས་པ་མེད་པར་རོ། །སུ་པཱི། སུ་ཧཱུg?ཚིག་འདི་ཨི་རཱུ་རོའི་དོན། །ཁ་གས་། ་་་་་་་་་་་་ཡིད་དགའ་པ། དབྱངས་དང་འདྲ་བ་དང་དབྱངས་ཀྱི་ཕ་རོལ་དུའོ། །ནཱ་མི་དང་ལྡན་པའི་ཕ་རོལ་དུ་ཚེག་དྲག་དབྱངས་དང་འདྲ་བ་དང་། ་དབྱངས་ཀྱི་ཕ་རོལ་དུ་ར་རུ་འཐོབ་སྟེ༑ཨབྷིརྒ་ཏ། ཨགྷི་ར་ཏྲ། པ་ཊརྦ་ད་ཏི། པ་ཊུ་ར་ཏྲ། །་ཡོ། ་ཨུཏ་ད་ན་ཁོཅརང་བཞིན་གྱི་ར་ནི་ན་མི་མ་ཡིན་པའི་ཕ་རོལ་དུ་ཡང་ངོ༌། །རང་བཞིན་་གྱི་ར་ཡིག་གི་ཚེག་དྲག་ནི་ནཱ་མིའི་ཕ་རོལ་ལམ་ནཱ་མིའི་མ་ཡིན་པའི་ཕ་རོལ་དུ་་་་དབྱངས་དང་མི་འདྲ་བ་དང་དབྱངས་དང་འདྲ་བའི་ཕ་རོལ་དུ་ཡང་རར་འཐོབ་སྟེ། གཱིརྦ་ཏི། གཱིཔ་ཏི། དྷརྦ་ཏི། དྷཱ་པ་ཏི། །དབྱངས་དང་་་་དབྱངས་དང་འདྲ་བ་རྟག་ཏུ་སྟེ། པི་ཏ་ར་ཏྲ། །པི་ཏརྱ་ཏ། ར་ཡིག་མ་ཡིན་པའི་རང་བཞིན་དུ་ཡིན་ཡང༌། ཧེ་པྲ་ཙེ་ཏྲཿརཱ་ཛནི། ཧེ་པྲ་ཙེ་ཏོ་རཱ་ཛན་ཞེས་སོ། །ཨུ་ཥརྦ་དྷ། ཨནྣ་རེ་ཕ་མ་ཡིན་པ་ལས་ཏེ། ཨ་ཧརྦ་ཏི། ཨ་ཧཔ་ཏི། བ། །ཨ་ཧརྒ་ཎ། ཨ་ཧ་ར་ཏྲ། ར་ཡིག་ལ་ནི། ཨ་ཧོ་ར་ཏྲཨ་ཧོ་རུ་པཾ༑་ཨེ་ཥ་དང་ས་དག་གི་ཕ་རོལ་དུ་གསལ་བྱེད་ལ་དབྱིའོ། །ཨེ་ཥ་དང་ས་་་དག་ལས་ཕ་རོལ་དུ་ཚེག་དྲག་དབྱི་བར་བྱའོ། །ཕ་རོལ་གསལ་བྱེད་ལའོ། ་ ོ། ཨེ་ཥ་པ་ཙ་ཏི། པ་ཌི་ཀ་སྟེ། ཨེ་ཥ་ཤ་སྟི། ས་པ་ཙ་ཏི། ཕྱི་མ་ལས་སྔོན་མ་ལ་སྒྲུབ་པ་བཀག་པའོ། །ཨ་པའི་དབང་དུ་བྱས་པ་ལས་ཡང་ན་རང་བཞིན་དུ་གཟུང་བའི་ཕྱིར་རོ། །ཨེ་ཥ་ཀཀ་རོ་ཏིས་ཀཀོ་རོ་ཏི། ཨ་ནེ་ཁོ་གཙྪ་ཏི། དེ་མི་འགྲོ་ཨཀྲ་དང་། ན་ཧྲའི་བསྡུ་བ་ལ་ཚེག་དྲག་དབྱི་བར་མི་འགྱུར་རོ། །ཚེག་་་་—།
