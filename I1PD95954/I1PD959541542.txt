སོ༑ལ་། པི་ཎྞཾ། ལྷག་མའོ། །བི་བློ། ་ཡ་ཛ་སོགས་སོ། །བི་ཥིརབ་ཏུ་འཇུག་པ་ལའོ། །ཤིཊྚཾ། ཤིཥའཚེ་བ་ལའོ། །:ཤ་ཥཱི། ཏུ་ཥ། དུ་ཥ་ལྷག་མའོ། །ཏཱི་ཏི། འི་པུ་ཥཱ་ཏི་ཀྲྀཥི་ཤྲཱི་ཥྛ་ཏི་དྭི་ཥི་པི་ཏི་བི་ཥི་ཤི་ཥི་ཤུ་ཥི་ཏུ་ཎི་དུ་ཎི་ལས་ཥ་ལས་སོ། །ཞེས་པས་ཨིཊ་མེད་དོ། །ཏྭེཀྵ། པོཏྟཱ། སྔ་མ་་བཞིན་དབྱངས་ལས་ཕ་རོལ་ཏུ་ཡང་ན་ཨ་ཡིག་ཨཱ་ག་མའོ། །།ཀྲཥྛ། ཀ། :：@ཥྛ། དྭེཥྛ། ་པེཥྛ། །བེཥྛ། ཤེཥྛ། ཤོཏྟཱ། ཏོཏྟཱ། །དོཥཱ། བ་་ས་ངེས་པར་གནས་པ་ལའོ། སྔོ། བ་ས་ཏི་དང་གྷ་སི་ལས་ས་ལས་སོ། །ཞེས་པས་་ཨིཊ་མེད་དོ། །བལྟ། ད་ཧ་ལྷག་མའོ། །དི་ཧ། དུ་ཧི། ཡ་ཛ་སོགས་སོ། །མི་ཧ་ལྷག་མའོ། །གྱུ། རི་ཧ་མདོ་ལས་བྱུང་བ་དང་འཚེ་བའི་དོན་ནོ། །ཚེ། རུ་ཧ་ལྷག་་མའོ། འི། ལི་ཧ། ་ཡ་ཛ་སོགས་སོ། །ལུ་ཧ་མདོ་ལས་བྱུང་བ་དང་འཚེ་བ་ལའོ། །ཎ་ཧ། བ་ཧ། ཡ་ཛ་སོགས་སོ། །།ད་ཧི་དི་ཧི་དུ་ཧི་མི་ཧི་རི་ཧི་རུ་ཧི་ལི་ཧི་ལུ་ཧི་ན་ཧི་བ་ཧི་ལས་ཧ་ལས་སོ། །ཞེས་པས་ཨིཊ་མེད་དོ། །ད་དང་པོའི་གྷའོ། །ཞེས་པས་ད་དང་པོའི་གྷ་ཉིད་དོ།  ོ། ཏ་ཡིག་གི་དྷ་ཡིག་གོ། །།སྔ་མ་བཞིན་དྷཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་གྷའི་ག་ཉིད་དོ། །ད་། དེབྷཱ་ཧའི་རྙའོ། །ཏ་ཡིག་གི་དྷ་ཉིད་དོ། །སྔ་མ་བཞིན་ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། །ཌྷ་ལ་ཌྷ་དབྱིའོ། །མེ་ཎཱ། །༈རེ་ཆ་། རོ་ནཱ། ལེ་ཆ་། ལོ་ཞྭ། ན་ཧིའི་དྷའོ། །།ཞེས་པས་ཧའི་དྷ་ཉིད་དོ། །ཏ་ཡིག་གི་དྷ་ཉིད་དོ། །སྔ་མ་བཞིན་དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ། །ཞེས་པས་སྔ་མ་དྷ་ཡིག་གི་ད་ཉིད་དོ། །:། ནདྡྷཱ། ཧའི་ཆ་འོ། །ཏ་ཡིག་གི་དྷ་ཉིད་དོ། །སྔ་མ་་བཞིན་ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་:སྡེ་ཉིད་དོ། །་ལོ། ས་ཧི་བ་ཧ་དག་གི་ཨ་ཡིག་་གི་ཨོཏའོ། །།ཞེས་པས་ལྷ་ལ་ཌྷ་དབྱི་ཞིང་ཨ་ཡིག་གི་ཨོ་ཉིད་དོ། །་མ། བོ་ནི། །
