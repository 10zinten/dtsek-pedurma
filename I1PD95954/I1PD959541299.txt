བསྟན་འགྱུར། སྒྲ་མདོ། ཤིཀྱི་དང་དཀར་རྣམས་ཏེ། །སིའི་རྣམ་པར་གཅད་པའོ། །ཨེ་ཀ། ཨེ་ཀཾ། ཨེ་ཀ་ན། ཨེ་ཀ་སྨི། ཡོཨེ་:ཀ་སྣཏི། ཨེ་ཀ་སྱ། ཨེ་ཀ་༔སྨིན། ད་བུད་མེད་ལ། ཨེ་ཀཱ། མ་ནིང་ལ། ཨེ་ཀཾ། ཨ་ཀ་སྟེ། ཨེ་ཀ་ཀ། བུད་མེད་ལ། ཨེ་ཀི་ཀཱ། མ་ནིང་ལ། ཨེ་ཀ་ཀཾ། ཀིམ་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ཏེ། ཀིམ་ཀའོ་་ཞེས་པས་ཀ་རུ་བསྒྱུར་བའོ། །ཀ། །ཀོ། ཀེ། ཀཾ། ཀོ། ་ཀཱན། ཀེ་ན། ཀཱ་ཧཱུཾ། ཀི། ཀ་སྨི། ཀཱ་ཧཱུཾ། །ཀེ་བྷྱ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་་མེད་ལ། ཀཱ། ཀེ། ཀཱ། ཊང་ལྡན་གྱི་བརྗོད་པ་རྣམས་ཡི་ཡཱས་ཡཱས་ཡཱམ་་རྣམས་སུ། སུ་དང་ལྷན་ཅིག་འགྱུར་ཞིང་སྔ་མ་ཐུང་བའང་སྟེ། ཀ་སྤྱི། ཀཱ་ཧཱུཾ། ཀཱ་། ཞེས་པ་ལ་སོགས་པའོ། །མ་ནིང་ལ། །ཀཾ། ཀེ། ཀཱ་ནི། ཁྱད་པར་མེད་པའི་སྒྲུབ་པ་དག་ལ་རིན་དང་ཙིཏ་ཀྱི་རྐྱེན་དག་ནི་ཀིམ་གྱི་རྣམ་པར་དབྱེ་བའི་མཐའ་ལས་བསྒྲུབ་པར་བྱ་སྟེ། ཀ་ཤྲཱིཏ།  ོཀཉྩིན། ཀོ་ཙིན། ཀོ་ཙིནྟ། །ཀེ་ཙི་ཏ། ཀེ་ཙིན་ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། །ཀཱ་ཙིཏ། ཀཱ་ཙིན། ཀེ་ཙིཏ། ཀེ་ཅིན། ་ཀཱ་ཤྲཱིཏ། ཀཱ་ཕྱིན། །མ་ནིང་ལ། ཀིཾ་ཙིཏ། ཀི་ཅིན་ཞེས་པ་ལ་སོགས་པའོ། །མི་ཚེ། །ལྷ་བྱད་ལན་ཁང་ལ་སྐྱེ་ཚུ་ཥམ་ན་དཔེ་ལོདཱིའི་སྒྲ་ནི་གཉིས་ཀྱི་དོན་མངོན་པར་རྗོད་པར་བྱེད་པ་ཉིད་ཀྱི་ཕྱིར་གཉིས་ཀྱི་ཚིག་ཁོ་ནར་འགྱུར་ཏེ། །ཁྱད་ལ་སོགས་པའི་ཨ་ཉིད་དང་ཨོ་ཡིག་ལ་ཨོ་སྟེ་ཨོ་ཡིག་ལ་འང་ངོ་ཞེས་པས་ཨོ་རུ་འགྱུར་ཞིང་ཕ་རོལ་དབྱི་བའང་ངོ༌། །དྲོ། དྲོ། ཨ་ཡིག་སྒྲ་ལྡན་ལ་རིང་པོར་ཐོབ་སྟེ། དྭཱ་ཧཱུཾ། རྭ་ཧཱུཾ། །དྷཱ་། དྭ་ཡོ། དྭ་་ཡོ༑། བུད་མེད་ལ། ཁྱད་ལ་སོགས་པའི་ཨ་ཉིད་དུ་བྱས་པ་ན་ཨཱའི་རྐྱེན་ཏེ། ཨོ་ཨཱིའོ་ཞེས་པས་ཨོ་ཨཱི་རུ་འཐོབ་ཅིང་། །ཨའི་ཡི་གེ་ཨིའི་ཡི་གེ་ལ་ཨེའོ། །དྭེ། དྭེ༑དྭཱ་ཧཱུཾ། །དྭཱ་ཧཱུཾ། རྭ་། ཊཱ་ཨོས་དག་ལ་ཨེ་ཉིད་དེ། །ཨེ་ཨ་ཡའོ། །།
