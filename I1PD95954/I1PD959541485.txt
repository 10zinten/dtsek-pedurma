ལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་་ཊི་ཚི་དིར། ཨཙྪི་དཏ། ཨཙྪི་ནཱིཏ། Pཨཙྩི་ཏྟཾ། །ཨཙྪིཥྛ། ་ཨཙྪིཏྟི། ཨཙྪིཏྟཱ་ཏྭཾ། ཨཙྪིཏྟཏ། རི་ཙིར། ཨ་རི་ཙཏ། །ཙ་སྡེའི་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཀིའོ། །ཨ་རིཊྛིཏ། ཨ་རིཏྟཾ། ཤྲཱིཨ་རིཀྵ། ཨ་རིཀྵཏ། །:ཨ་རིགྱུཏ། ཨ་རིཀྵ་ཏྭཾ། ཨ་རིཀནྟ། །བི་ཙིར། ་ཨ་བི་ཙཏ། ཨ་བིཀྵིཏི། ཨ་བིཀཾ། ཨ་བིཀྵ། ཨ་བཏྟི། །ཨ་བཀཱི་ཏཱཾ། ཨ་བི་ཀཱཏ། ཀླུ་དིར། སྔ་མ་བཞིན་དང་པོའོ། །ཨཀོནཱིཏ། :：ཨཀྵ་ཏྭཾ། ཨཀོནྟུ། ཨགྷཱཏྟ། ཨ་བླུཏྟཱ་ཏྭཾ། ཨ་ཀླུཏྠཏ། ཡུ་ཛིར། ནཨ་ཡུ་ཛཏ། །ཨ་ཡོཀྵིཏ། ་ཌ་ཨ་ཡོཏྟཾ། ཨ་ཡོཀྵ། མུ་ཝཾ། ་ཨ་མུཀྟི། ཨ་མུཀྵ་ཏྭཾ། ཨ་མུཀྟཏ། །ཆ་ཨུཙྪི་དིར། ཨཙྩཾ་དཏ། སི་ཙའོ། །།ཨིཊའོ། །ནཱ་མི་ཨུ་བ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨཙྪརཱིཏ། ཨཙྪདིཏྟཾ། ཨཙྪཇི་ཥཱ། ཨཙྪདིཥཱ། ཨཙྪདི་ཥཱ་ཏྭཾ། ཨཙྪདིཥ་ཏ། །ཨ་ཀྲྀ་དིར། ཨ་ཀྲྀ་དཏ། ཨ་ཏི་རཱཏ། :：ཨ་ཏཇི། ཙུ་དིར། ་ཨ་ཙུ་ཏཏ། ཨ་ཙོ་ཏཱིཏ། ཨ་ཙོ་ཏིཏྟཾ། ཨ་ཙོ་ཏི་ཥཱ། བྷཱུ་ཏིར། །ཨ་བྷཱུ་ཏཏ། ཨ་ཙོ་ཏཱིཏ། སྤུ་ཊིར་འགྲོ་བ་ལའོ། །ཨ་སྤུ་ཊཏ། །ཨ་སྤོ་ཊཱིཏ། ཨ་སྤོ་ཊིཏྟཾ། ཨ་སྤོ་ཊི་ཥཱ། གྷུ་ཏིར་སྒྲ་ལའོ། །ཨ་བྷུ་ཥཏ། །ཨ་བྷིཀྵིཏེ། ཨ་གྷོ་ཥིཏྟཾ། ::ཏུ་ཧིར་དུ་ཧིར་འཚེ་བ་ལའོ། །ཨ་ཏུ་ཧཏ། །ཨ་ཏོ་ཧཱིཏ། །ཨ་ཏོ་ཧིཏྟཾ། ་་ཨ་དུ་ཧཏི། ཨ་་དོ་ཧཱིཏ། ཨ་དོ་ཧིཏྟཾ། སྐནྡ་ར། སྔ་མ་བཞིན་ཨ་ནུ་ཎཾ་ག་དབྱིའོ། །ཨ་སྐ་དཏ། :སི་ཙ་ལ། ཨའི་འང་རིང་པོའོ། །།ཨ་ཨཱཿཏི་ཏ། ཨ་སྐུ་ཏྟཾ། ཨ་སྐུདྡྷ། ཛྲ་ཤིར། ་ཨཏི། སརྟི་དག་གི་ཨཎ་ལའོ་ཞེས་པས་ཨ་ཡིག་གི་ཉེ་བར་མཚོན་པ་ཉིད་ཀྱི་ཕྱིར་ཡོན་ཏན་ནོ། །ཨ་དནཏི། ་ཨ་དན་ཏྭཾ། ཨ་དན་ན། སི་ཙ་་ལ་ནི། སྭ་ཛི་དྲི་ཤ་དག་གི། །དབྱངས་ལས་ཕ་རོལ་དྷུཊ་ལ་ཡོན་ཏན་དང་། འཕེལ་བ་གནས་:པ་ལ་ཨ་ཡིག་ཨཱ་ག་མའོ་ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་དུ་ཨ་་།
