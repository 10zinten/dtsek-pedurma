ཚོ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼མི་ཀ་ཁ་ཎ་སྤྱི་ལ་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་ཞེས་པས་ཨཱ་དབྱིའོ། །ཚི། འདི་ཀུན་དུ་གནོད་པར་བྱེད་དོ་ཞེས། ::ཨཱ་བ་དྷི། གྲ་ཧི་ཛཱ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །བྱ་བ་འདིས་འཇོམས་པར་བྱེད་དོ་ཞེས། བིགྷཱ༄༅། སྔ་མ་བཞིན་དུ་ཨུ་པ་དྷཱ་དབྱི་བ་དང་གྷ་ཉིད་ཀྱང་ངོ༌། །འདི་ཀུན་དུ་འཁྲུག་པར་བྱེད་དོ་ཞེས། ཨཱ་ཡུ་དྷི། ཏུ་རྗེས་སུ་འབྲེལ་པ་ལས་ཨ་ཐུའོ། །ཊུའི་སྒྲ་རྗེས་སུ་འབྲེལ་བའི་བྱིངས་ལས་ཨ་་ཐུའི་རྐྱེན་འགྱུར་རོ། །བེ་:ས་ཐིུ། ཀླུ་བ་ཐིུ། གཡོ་བར་བྱེད། སྒྲོགས་པར་བྱེད་ཅེས་ཚིག་རྣམ་པར་སྦྱར་རོ། །།ཊུ་པེ་སྲི་ཀ་པི་གཡོ་བ་ལའོ། །ར། ཊུ་ཀླུ་ཥ་སྒྲ་ལའོ། ༈། ཊའི་སྒྲ་ནི་འདིའི་དོན་ནོ། །ཏུ་རྗེས་སུ་འབྲེལ་པ་ལས་དེས་ཚར་བ་ལ་དྲི་མ་ཀའོ། །རྔུའི་སྒྲ་རྗེས་སུ་འབྲེལ་པའི་བྱིངས་ལས་བྱིངས་དོན་དེས་་་ཚར་བའི་དོན་ལ་དྲི་མ་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །།སྨིན་པས་ཚར་བ་ལ། པཀྴི་མོ༑དེ་བཞིན་དུ་བྱེད་པས་ཚར་བ་ནི། ཀྲྀ་ཏྲི་མཾ། ཌུ་པ་ཙཥ་འཚེད་པ་ལའོ། །འུ་ཀྲྀཉ་བྱེད་པ་ལའོ། །ཏུའི་སྒྲ་ནི་འདིའི་དོན་ཏོ། །::ཡཱ་ཙི་བིཙྪ་པྲ་ཙྪ་ཡ་ཛི་སཱ་པི་ར་ཀླུ་ཡཏ་རྣམས་ཀྱི་ནངའོ། །ཡཱ་ཙི་དང་བིཙྪ་དང་པྲཙྪི་དང་ཡ་ཛི་དང་སཱ་པི་དང་རཀྟི་དང་ཡཏུ་ཞེས་པ་འདི་རྣམས་ལས་ནང་ཕ་རོལ་ཏུ་འགྱུར་རོ། །སློང་བར་བྱེད། ་ཡཙྪཱམ། དེ་བཞིན་དུ། བིཤཱན། པྲབྷ། ཡཛྷ། སཱརཿ། རཀྵ། ཡནྡྷ། སྒྲ་ལྡན་དང་ཏི་དག་ལ་ཞེས་པས་ཨིཊ་མེད་དོ། །ཚ་བ་དག་གི་ལྔ་པ་ལ་ཞེས་པས་ཚ་ཡིག་གི་ཤའོ། །ཉེར་བསྒྱུར་ལ་:དཱ་ལས་ཀིའོ། །ཉེར་བསྒྱུར་ཉེ་བའི་ཚིག་ལ་དཱའི་མིང་ཅན་ལས་ཀི་རུ་འགྱུར་རོ། །སྦྱིན་པར་བྱ་་བ་དང་པོ་ལས་ཐོབ་པ་ནི། ཨཱ་དི། སྒྲུབ་པར་བྱེད་པ་ནི། བི་དྷི། ཨནྟར་གྱི་སྒྲ་ཉེ་བའི་ཚིག་ལས་ཀྱང་ཀི་འདོད་དེ། གཞན་འཛིན་ནོ། །ཨནྟ་དྷིགེ། ལས་ལ་གཞི་ལ་འང་ངོ༌། །ལས་ཉེ་བའི་ཚིག་:ལའང་དཱའི་མིང་ཅན་ལས་གཞིའི་—།
