ཡཱ་བ་བྷུ་བ། ཨི་ཨཱ་ཡའོ། །བི་བྷཱ་ཡ། ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །བི་བྷཱུ་ཏུ། བི་བྷུ། རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ། །ཞེས་པ་ལ་སོགས་པས་ཐལ་ལ་ཡང་ན་ཨི་ཊའོ། །བི་བྷ་ཡི་ཐ། བི་བྷེ་ཐ། བི་བླ་ཐུ། བི་བྷཱུ། བི་བྷཱ་ཡ། སྲི་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །བི་བྷི་བ། བི་ཕྱི་མ། ཇི་ཧྲ་ཡཱ་ཙ་ཀཱ་ར། ཛི་ཧྲ་ཡཱ་མཱ་ཡིས། ཛི་ཧྲ་ཡཱ་བ་བྷཱ་བ། ཛི་ཧ་ཡ། དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་པས་ཨི་ཡའོ། །ཛི་ཧྲ་ཡ་ཏུ། ཇི་ཧྲི་ཡུཉྩ། ཛི་ཧྲ་ཡི་ཐ། ཛི་ཧྲེ་ཐ། ཛི་ཧྲི་ཡ་ཐུ། ་ཛི་ཧྲི་ཡ། ཏ༄༅ཛི་ཧཱ་ཡ། བི་བྷ་རཱི་ཙ་ཀཱ་ར། བི་བྷ་རཱི་ཙ་ཀྲེ། བི་བྷ་རཱ་མཱ་ས། བི་བྷ་རཱ་བ་བྷུ་བ། བ་བྷ་ར། :བ་བྷཱ་བ། རིའི་ཡི་གེརའོ། །བ་བྷཱ་ཏུ། ་བ་བྷིུ། སྲོ་ལ་སོགས་པའི་མདོ་ཡིས་ཨིཊ་མེད་དོ། །བ་བྷུ་ཐི། བ་བྷུ་ཐིུ། པབ་བྷ། །བ་བྷཱ་སིར། བ་ཤྲྰི་ལིབ། བ་བྷཾ་མ། བ་བྷེ། བ་བྷཱ་ཏེ། ་བ་བྷི་རེ། བ་བྷི་ཥ། བ་བྷཱ་ཐེ༑ྋནཱ་མི་མཐའ་ཅན་ནའི་བྱིངས་ལས་ཨཱ་ཤཱི་ཥ་ཨཱ་རྱ་ཏ་ནི་པ་རོ་ཀླུ་རྣམས་ལ་ཉེ་དྷའི་ལྷའོ། །ཞེས་པས་དྷ་ཡིག་གི་ཌྷ་ཡིག་གོ། །བ་བྲྀ་ཌྷཝེ། བ་ཧྥེ། ་བ་བྷཾ་གབ་ཧེ། བ་ཤྲཱི་མ་ཧེ། །ཨོ་ཨ་བའོ། །ཛུ་ཧ་བཱཉྩ་ཀཱ་ར། ཛུ་ཧ་བཱ་མཱ་ས། ཛུ་ཧ་ཛྭ་ལེབ་བྷུ་བ། །ཨོ་ཨཱ་བའོ། །ཛུ་ཧཱ་བ། དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་་སོགས་པས་ཨུ་བའོ། །ཛུ་ཧུ་བ་ཏུ། །ཛུ་ཧུ་བུ། ་རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ་ཞེས་པ་ལ་སོགས་པས་ཐལ་ལ་ཡང་ན་ཨིཊའོ། །ཛུ་ཧ་བི་ཐ། ཛུ་ཧོ་ཐ། ཛུ་ཧུ་བ་ཐུ། ཛུ་ཧུ་བ། སྲི་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །ཛུ་ཧུ་བི་བ། ཛུ་ཧུ་བི་མ། ཉི་ཨིནྡྲི། ཨིནྡྲི། ཤཱནྡྲི། གནྡྲིཡཾ། །དནྡྲི་རྣམས་ཀྱི་པ་རོ་ཀླུ་ལ་ཡོན་ཏན་མེད་པ་ལའོ་ཞེས་པས་ཨ་ནུ་ཊཾཨག་དབྱི་བ་བརྗོད་པ་ལས་ཨོཾ་ྋམེད་དེ། ནཱ་མི་དང་པོ་ལྗི་བ་ལྡན་པ་རཾ་ཙྪ་མ་ཡིན་པ་ལས་སོ། །།ཞེས་པའོ། །མཚུངས་།
