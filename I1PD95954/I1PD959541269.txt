:བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའི་བ་དང་ཨུ་ཉིད་དོ། །ཨ་མུ་མུ་ཡང་། ཨ་མུ་མུ་ཡཉྙོ། ཨ་མུ་མུ་ཡཉྩ། ཨ་མུ་མུ་ཡཧཱུྃ། ཨ་མུ་མུ་ཡཉྙོ། །འདིར་ཡང་ཝ་ཉིད་དུ་ཁ་ཅིག་འདོད་པར་བྱེད་དེ། །ཨ་མུ་མུ་ཨཱི་ཙ༑༌་ཨ་དམཱི'ི། བུད་མེད་ལ། ་ཨ་མུ་མུ་ཨཱི་ཙཱི། ཨ་དམཱི་ཙཱ། མ་ནིང་ལ། ༼ཨ་མུ་མུ་ཡཀ། ཨ་མུ་མུ་ཡག། ཨ་མུ་མུ་ཨཱི་ཙཱ། ཨ་མུ་མམྨི་ཙཱ། ཨ་ད་མུཉྩི། ཡང་དག་པར་ན་སོ་སོར་གཉི་ག་དགག་སྟེ། །ཨ་དདྲུང་། ཝཨ་ད་རླཉྙོ༑ཨ་ད་དྲུཉྩ། ཨ་ད་དྲུཤྩ། ཨ་ད་དྨོ། ཨ་ད་དཱི་ཙཱ། །ཨ་ད་དཱི་ཙཱ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། ཨ་ད་དྲི་ཙཱ། ་མ་ནིང་ལ།  ོ་ཨ་་་་དྲུཀ། ཨ་ད་དྲུག་། ཨ་ད་དཱི་ཙཱ། ཨ་ད་དྲིཤྩ། སྤྱི་ལ་སོགས་པའོ། །ས་ཧཅུ་ཏི་ཞེས་པས་ཀྭིཔ་སྟེ། ས་ཧ་དང་། །སཾ་དང༌། །ཏི་རས་རྣམས་ཀྱི་ས་དྷི་དང་ས་མི་དང་། ཏི་རི་རྣམས་སོ་ཞེས་པས་ས་ཧའི་སྒྲ་ནི་ས་དྷི་ཞེས་པའི་ཚིག་ཕྲད་དོ། །ས་རྒྱུད། ་སངྒྷ༄། །ས་ཧྥུཝུ། སངྒྷ། ལོས་ཏྱཱཉྙོ། ཨ་དབྱི་བ་དང་རིང་་པོའང་སྟེ། །ས་དྷི་ཙ། ས་དྷི་ཙཱ། ས་དྷུབྷཾ། སཙྪིདྷི། ཞེས་པ་ལ་སོགས་པའོ༑། བུད་མེད་ལ་ས་དྷི་ཙཱ། མ་ནིང་ལ་སངྒྷཀ། སངྒྷག། ས་དྷཱ་ཙཱ། སངྒྷཉྩི༑ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ས་མཅུ་ཏི་ཞེས་པས་ཀྭིཔ་སྟེ། ས་མའི་སྒྲ་ནི་ས་མཱི་ཞེས་པའི་ཚིག་ཕྲད་དོ། །སམྱང་། :སམྱཉྙོ། སམྱཉྩ། སམྱཉྩ༑་སམྱ༄། བྷུཊ་མ་ཡིན་པའི་དབྱངས་ལ་ཨཉྩའི་ཨ་དབྱི་ཞིང་སྔ་མའི་ཡང་རིང་པོ་སྟེ། ས་མཱི་ཙ། ས་མཱི་ཙཱ། ས་མྱདྡྷཾ། །ས་མྱབྷི། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། ས་མཱི་ཙཱི། མ་ནིང་ལ། སམྱཀ། སམྱག། ས་མཱི་ཙཱ། ས་མཱིཉྩི། །ཏི་རོ་ཨཅུ་ཏི་ཞེས་པས་ཀྭིཔ་སྟེ། ཏི་ར་ནི་ཏི་རི་ཞེས་པའི་ཚིག་ཕྲད་དོ། །ཏིརླང་། ཏིརླཉྙོ། །ཏིམྨནྟུ། །ཏིཙྩཾ། ཏིརླཉྙོ། ཏིརླཉྩ་ནི་ཏ་ར་ཐུ་རུ་འགྱུར་ཏེ། ཏིརླ། ཏིརླ། །ཏིརླནྡྷཾ། ཏིརླདྷི། GLCALCGuns།
