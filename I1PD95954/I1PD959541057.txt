བསྟན་འགྱུར། སྒྲ་མདོ། ཞིའིདབྱི་བར་བྱ་བའང་མ་ཡིན་ནོ། །པི་ཏཱ། པི་ཏྲ་བྷཱུམ། པི་ཏྲེ་བྷི། པི་ཏེ་པེ། པི་ཏྲ་བྷཱུམ། ་པི་ཏྲ་བྷཱུ། རྀཏ་མཐའ་ལས་སྔ་མ་དང་བཅས་པའོ་ཞེས་པས་རཾ་མཐའ་ལས་ཕ་རོལ་གྱི་ང་སི་ངས་དག་གི་ཨ་ཡིག་སྔ་མའི་དབྱངས་དང་ལྷན་ཅིག་ཨུ་རུ་འགྱུར་རོ། །པི་ཏུ། པི་ཏྲ་བྷཱུམ། པི་ཏྲ་ཧྥུ། པི་ཏུ། པི་ཌོ། ཨཱ་མ་ལའང་ནུའོ། །ནུ་དང་བཅས་པའི་ཨཱ་མ་ལ་:རིང་པོའོ་ཞེས་པ་རཾ་ཡིག་་་རིང་པོའོ། །ར་ཥ་རའི་ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་སོགས་:པས་ན་ཡིག་ཎ་ཉིད་དོ༑། པི་ཙཱེཎཾ། ང་ལ་ཨརའོ་ཞེས་པས་རིཏ་མཐའ་ཇི་ཕ་རོལ་ལ་ཨར་དུ་་འགྱུར་ཏེ། པི་ཏ་རི། །པི་ཏྲིཥྛ། ཥ་ཉིད་དེ། པི་ཏྲི་ཥཱ། ཀུན་དུ་བོད་པ་ལ༑སཾ་བུདྡྷི་:ལས་ཨཱའང་མ་ཡིན་ནོ་ཞེས་བཀག་པའི་ཕྱིར་སི་ལ་ཨཱ་སྟེ་སི་དབྱི་་བའང་ངོ་ཞེས་པའི་ཨཱ་ཡིག་སཾ་བུདྷི་ལ་འགྱུར་བ་མ་ཡིན་ཞིང་། དོན་གྱིས་ན་གྷཊ་ལའང་ངོ་ཞེས་པ་ཨར་ཁོ་ནར་འགྱུར་རོ། གློ་། རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་་་་གཅད་བྱའོ། །།ཧེ་པི་ཏ། ཧེ་པི་ཏ་རོ། ཧེ་པི་ཏ་ར། དེ་བཞིན་དུ་:དྲྭ་ཚིགས་མི་བཟོད་པ་དང་ལྡན་པ་ལའོ། །ཛཱ་མཱཏྟ་ཅིབུ་མོའི་བདག་པོའོ། །ས་བྱཏྟཱ་བྷསྐ་ཅི་རེ་ཏུ་དེ་ཝཱ་དེ་ཝ་:ཏ་གོ་ཏ་མ་ཞེས་པ་ལ་སོགས་པ་རྣམས་སོ། །རྫིའི་སྒྲ་ནི་དྲུག་་་པའི་མང་པོའི་ཚིག་ལ། ཚྭ་ཡང་ནའོ་ཞེས་པ་རྣམ་པར་བརྟག་པས་རིང་པོ་ཉིད་་་་ཁྱད་པར་རོ། །ནཾ་ཏཱཾ། ཙཱ་ཎཾ། བླ་མ་རྗོད་པར་བྱེད་པའི་པྲ་ཤཱསྟྲཱིའི་ྋསྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །སི་ལ་སྔ་མ་བཞིན་དུ་ཨཱའོ། །པ་ཤ། །གྷཊ་ལ་སཱ་སྲི་ལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་ཞེས་པ་ཨརའོ༄༅། །པྲ་ཤཱསྣ་ྋརོ། པྲ་ཤཱསྟཱ་རསྨི། །པྲ་ཤས་ཙེརམ། །པྲ་ཤཱསྟཱ་རོ། ལྷག་མ་པི་ཚིགས་བཞིན་ནོ། །དེ་བཞིན་དུ་ལྷའི་་བཟོ་བོ་ལའང་ཏཱཤྩ་སྟེ། ཏུཏྟཱསི། ཁ་ལོ་བ་དང་ཉི་མ་དག་ལ་:ཀླུ་རྀ་ཥཱ་དེ་པྲ་ཏི་ཧཱ་ཋརེ་ཀླུ་:ཏྲ་ཡཱ་ལིཤུ་ཏྲ་ཛེ་སུ་ཏོཙེ། ཀྲོཊྚ་ནི་ཅེ་སྤྱང་ལའོ། །ན་ཛཱི་བུའི་བུ་ལའོ། །།
