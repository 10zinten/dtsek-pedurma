མེད་ཀྱི་རྟགས་ཅན་ཉིད་ཀྱི་ཕྱིར་ན་ཉིད་མེད་དོ། །སཱ་སཱི། སཱ་སཱ། སྲ་ྋསྲི་ཧཱུཾ། སྲ་སྲི་བྷིཞེས་པ་ལ་སོགས་པའོ། ་། ཀུན་དུ་བོད་པ་ལ། ཧེ་སཱ་སཱཙྪེ། ཧེ་སྭཱ་་སཱ་རོ། །ཧེ་སཱ་སཱ་ར། དེ་བཞིན་དུ་ནསྗེ་ལ་སོགས་པ་རྣམས་སོ། །དེ་སྐད་དུ་ནི༑་སྭ་སྲི་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ན་ལྟ་ཞེས་བྷ་ཥའི་འགྲེལ་པ་བྱེད་པའོ། །གཞན་རྣམས་ནི་སྤུ་སྲི་ལ་སོགས་པ་ཉིད་དུ་:འདི་ཉིད་དུ་འདི་མི་འདོད་དོ། །དེའི་ལུགས་ཀྱིས་ན་ད་ལ་སོགས་པ་ཉིད་ཀྱི་:ཕྱིར་ཨཱིའི་རྐྱེན་ནོ། །ནཙྭ། པོ་ཏཱི། རཱི་ཏའི་མཐའ་ཅན་དང་ལེའི་ཡི་གེ་མཐའ་ཅན་དག་རབ་ཏུ་གྲགས་པ་མེད་དོ། །ཨེ་ཡིག་གི་མཐའ་ཅན་དྲན་པའི་དགའ་མ་རྗོད་པར་བྱེད་པ་སེའི་སྒྲ་སྟེ། ཨི་དང་བཅས་པར་འཇུག་གོ་ཞེས་པ་སེའོ། །རྟགས་ཀྱི་མིང་ཅན་:ནོ། །སི་ལ་སོགས་་པ་རྣམས་སོ། །།།རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ། །སེ། དབྱངས་དང་པོ་ཅན་ལ་ཨེ་ཨ་ཡའོ། །ས་ཡོ། །ས་ཡ། ས་:ཡམ། ས་ཡོ༄༅། །ས་ཡ། ས་ཡཱ། སེ་ཧཱུཾ། སེ་བྷི། སེ་ཡེ། སེ་ཧཱུཾ། སེ་བྷུ། གཞན་རྣམས་ནི་ང་སི་ངས་དག་གི་ཨ་ཡིག་སྔ་མའི་གཟུགས་ཉིད་ལ། སེཞེས་སྒྲུབ་བོ། །ཁོ་བོ་ཅག་ནི་ང་སི་ངས་དག་གི་ཨ་དབྱི་བའང་ངོ་ཞེས་པ་འདིར། ཨའི་དབྱི་་བའང་ངོ་ཞེས་ཅི་ལྟར་མི་བྱེད། །ཁྱད་པར་དུ་བྱེད་པའི་རྗེས་སུ་འབྲེལ་པ་མེད་པ་རྣམས་དོར་ནས། ཡོདཨསྣ་ཏྲེ། ཏེ་འདིར་གཟུང་བར་བྱའོ། །ཛས་དང་། ཤས་ལའང་ཁྱད་པར་བྱས་པའི་ཕྱིར། །་་མི་ང་སི་ངས་ཁོ་ན་དག་འཛིན་པ་རྟོགས་་་་པར་བྱའོ་ཞེ་ན་བདེན་ཏེ། ཁྱོད་ང་སི་ངས་དག་ལ་ཞེས་ཉེ་བར་བླངས་པ་ནི་མཚོན་པར་བྱ་བ་གཞན་འཛིན་པ་གསལ་བར་བྱེད་དོ། ལ་། མདོའི་རྒྱུན་ཆེན་པོས་སློབ་་་དཔོན་རྣམས་ཀྱི་དགོངས་པ་མཚོན་པར་བྱའོ་ཞེས་པའི་རིགས་པ་ལས་སོ། །དེས་ན་སེ་ལ་སོགས་པ་རྣམས་ལས་ཕ་རོལ་གྱི་ང་སི་ངས་དག་གི་ང་ཨའི་དབྱི་་—།
