གྲ་ཧ། པྲ་ཙྪ། བྲཤྩ། བྷསྔ། གྲ་ཧི་ཛུ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་སཱ་ར་་་་ཎའོ༑། ཛ་རཱི་གྲོ་རྒྱ་ཏེ། པ་རཱི་པཱིཙྪ་ཏེ། །པ་རཱི་བྷིཛྫ་ཏེ། བྱ་ཙ། སྔ་མ་བཞིན་སཾ་པྲ་སཱ་ར་ཎའོ། །བེ་བི་ཙྩ་ཏེ། བིད། བེ་བི་ཅང་ཏེ། བྷི་དིར། །བེ་བྷི་རྱ་ཏེ། ཙྪ་དིར། །ཙེཙླི་རྱ་ཏེ། ཀྱི་བྲི། ཙེ་ཀླུ་བྱ་ཏེ། ལི་ཁ། ལེ་་ལི་ཁྱ་ཏེ། ཎི་དི། ནེ་ནི་བྱ་ཏེ། ཎི་ཛི། །ནེ་ནི་ཛེ་ཏེ། སི་ཛི་མི་གསལ་་བའི་སྒྲ་ལའོ། །:སེ་སི་ཛ་ཏེ། དེ་བཞིན་དུ། ཤིཀྵ། བིཀྐ། དཱི་པ། :མི་ལ་སོགས་པ་ལས་སོ། །དི་བུ། ན་མིནའི་ར་བ་དག་ཅེས་ཁྱེཔ་ལ་:སོགས་པ་རིང་པོའོ། །དེ་དཱི་བྱ་ཏི། ་ཊི་བུ། ཏི་བི། རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །སེ་ཁཱི་བྱ་ཏེ། ཙྪི་བུ། ཏྟི་བི་ལས་ཥ་ཡིག་ལས་ཋ་ཡིག་ཕ་རོལ་ལམ་ཡང་ན་ས་་ཡིག་སྐྱེལས་ཐ་ཡིག་ཕ་རོལ་ཏུ་འདོད་དོ། ་། ཊེཊྛི་བྱ་ཏེ། ཏེཊྛི་བྱ་ཏེ། ཡང་གི་་སྒྲ་མང་པོའི་དོན་ཅན་གྱི་ཕྱིར་རོ། །སི་ཙ། ཙེ་ཀཱི་ཡི་ཏ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །།སེ་སི་ཙ་ཏེ། ཨ་བྷི་སེ། སི་ཙུ་ཏེ།  ི་ཤཱ་ས། ཤཱ་སའི་ཨུ་པ་:ཨུ་པ་ཉྩཱདྷཱའི་ཨཎ་་་དང་གསལ་བྱེད་དག་ལ་ཨིཏའོ་ཞེས་པས་ཨུ་པ་དྷཱའི་ཨ་ཉིད་དོ། །ཤཱ་སི་བ་སི་་གྷ་སི་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཤེ་ཤི་ཥྛ་ཏེ། བྷུ་ཏིར། ཛོ་གླུ་བྱ་ཏེ༑ཏཱ་ཀྲུ་དྷ། ཙོ་ཀྲུ་རྒྱ་ཏེ། ཛྙཱ་པ། ཙོཙྪ་པྱ་ཏེ། གུ་ཛི། ཇོ་གུ་ཛ་ཏེ། གྷུ་ར། ན་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཛོ་གྷརྨ་ཏེ། །ཏུ་ར་དབང་ཕྱུག་ཉིད་དང་འབར་བ་དག་ལའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །སི་ཏཱུརྨ་ཏེ། གཞན་དོན་རྣམས་ལ་ཨུ་བ་དྷཱ་ཉེ་བར་བསྟན་པ་སེམས་པར་བྱེད་པ་་དེའི་ཚེ། སོ་སཱུརྨ་ཏེ། མཱཙྪ། མོ་མུཙྪ་ཏེ། ཀུ་ར་སྒྲ་ལའོ། །ཙོ་ཀུརྨ་ཏེ། ཁུ་ར་གཅོད་པ་ལའོ། །ཙོ་ཁུརྨ་ཏེ། ་མུ་ར། མོ་མུརླ་ཏེ། གྱུར་རྣམ་པར་རྐོ་བ་ལའོ། །ཙོ་ཁྱུརླ་ཏེ། བྱཱཏ། ཏྭ་ཏི་སྭཱ་པི་དག་གི་ཨ་བྷཱ་སའིའོ་ཞེས་པས་ཨ་—།
