ན་དྷ་ཡིག་གི་ཌྷ་ཡིག་གོ། །ཛ་ཧི་:དྷེ། ཛ་ཧི་ཌྷ?། དེང༌། །ད་ཡ་ཏིའི་པ་རོ་གླུ་ལ་:ང་དིའོ། །ཞེ། ཞེས་པས་དི་ངར་བསྒྱུར་ཞིང་ཨ་སྒྱུ་ས་དབྱིའོ། །ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །།ང་གྱེ། ང་གྱི་ལྷེ༄༅། ::དི་གྱི་ལྷེ། ཏྲིང་། :ཏ་ཏྲེ། །ཏ་ཏྲི་ཌྷི། ཏ་ཏྲི་དྷེ། ་ཕྱིང་། :པ་ཕྱེ། པ་ཕྱི་དྡྷེ། པ་ཕྱི་བྷེ། ཊུ་དཱཉ༄༅། །ད་དོ། ད་དཏུ། ད་དེ། །ཊུ་དྷ་ཉ། ད་དྷོ། ད་དྷ་ཏུ། །ད་དྷེ། ཙ་གླིང་། པ་རོ་ཀླུ་ལ་ཡང་ནའོ། །།ཞེས་པས་པ་རོཀྟའི་ཡུལ་ལ་ཡང་ན་ཁྱུ་ཉ་རུ་བསྒྱུར་རོ། །ཉ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་གཉི་གའི་ཚིག་གོ༑ཏུ་། ཨཱ་ཙ་ཁྱོ། ཨཱ་ཙ་ཁྱ་ཏུ། ?ཨཱ་ཙ་ཁྱེ། ཨཱ་ཙ་ཁྱི་:དེ། ་ཨཱ་ཙ་ཁྱི་ཞཱ?། ཨཱ་ཙ་ཙཀྵེ། ཋ་ཨཱ་ཙ་ཙཀྵ་ཏེ། གསལ་བྱེད་དང་པོ་ཞེས་པས་ཨིཊ་ཨཱ་ག་མའོ། །ཨ་ཙ་ཙ་ཀྱི་རེ། ཛི་རྒྱལ་བ་ལའོ། །ཛིའི་སན་པ་རོཀྵ་དག་ལ་གིའོ་ཞེས་པས་གི་་རུ་བསྒྱུར་རོ། །གིའི་གཉིས་བརྗོད་དོ། །ཨའི་ཨུ་པ་དྷཱ་ཞེས་པ་ལ་སོགས་པས་བྱིངས་ཀྱི་འཕེལ་བའོ། །ཨི་ཨཱ་ཡའོ། ::། ཛི་གཱ་ཡ། ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་པ་ལ་་སོགས་པས་དབྱངས་དང་པོ་ཡོན་ཏན་མེད་པ་ལ་ཡ་ཉིད་དོ། །ཛི་གྱ་ཏུ། ཛི་་གྱུ༑རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ། །དབྱངས་མཐའ་རྣམས་ཀྱི་དང་པྲ་ཛི་ཛྲ་ཤ་དག་གི་ཡང་ཐལ་ལ་ཨིཊ་ཡང་ནའོ། :། ཏྲི་ཙ་ལ་རྟག་ཏུ་ཨིཊ་མེད་པར་འགྱུར་་་་ཡང་ཛ་ཁོ་ད་བྲི་བྱེ་ཨད་རྣམས་ཀྱི་ཐལ་ལ་རྟག་ཏུ་ཨིཊའོ། །ཞེས་བརྗོད་པ་ལས་་་ཡང་ན་ཨིཊའོ། །ཛི་ག་ཡི་ཐ། །བི་པ་རཱ་དག་ལས་ཇི། །ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཏྨ་ནེའི་ཚིག་གོ། །བི་ཛི་གྷ་ཏེ། བི་ཇི་གྱི་རེ། བི་ཛ་གྲུ་ཐེ། སྔ་མ་བཞིན་ཡང་ན་དྷ་ཡིག་གི་རྣ་ཡིག་གོ། །བི་ཛི་གྱི་དྡྷེ། བི་ཛི་གྱི་ལྷེ། བི་ཛི་གྱེ། བི་ཛི་གྱི་བ་ཧེ། བི་ཛི་གྱི་མ་ཧེ། བ་རཱ་ཛི་གྱེ། རི་པི་འགྲོ་བ་ལའོ། །རི་རཱ་ཡ། རིང་པོ་བསྒྲུབ་པ་ལ་ཡང་ཡ་ཉིད་དོ། །དབྱངས་བསྒྱུར་བ་གནས་་་།
