ཡ་མ་ལ་ཡ་སྦྱར་བས། གོང་དུ་ཡ་མ་གཉིས་པ་བྱས། ཊཱི་མ་ཕྱི་མདོ་འདིས་གོང་་་མར་ཐིག་ལེ་ཐོབ། ཡཾ་ཡ་མྱ་ཏེའོ། ་ཞུ། བྷ་ཎ་ཀྱི་ལ་ཡ་སྦྱར་བས་གོང་མ་བཞིན། ཐཾ་བྷ་ནྱ་ཏེའོ། །ཡ་མ་ནི་ཉེ་བར་སྡོམ་པའོ། ་། ཡ་མ་ཨུ་པ་ར་མེའོ། །བྷ་ཎ་ནི་སྒྲའོ། འོ་། །ཨའི་མཐའ་ཅན་ཞེས་ཅིའི་ཕྱིར་སྨྲོས་ཤེ་ན། ཏི་མ་ལ་ཡ་སྦྱར་བས་་་་གོང་དུ་ཏི་གཉིས་པ་བྱས་ཏེ། ཨིའི་མཐའ་ཅན་ཡིན་པས་ཐིག་ལེ་མ་ཐོབ་པའོ། །ཏེ་ཏི་མྱ་ཏེའོ། །ཛ་པཱ་དི་:ནཾ་ཙི། ཛ་པ་ནི་ཡིད་ལ་:བཟླས་པའོ། །།ཛ་བྷ་་་དང་ཛཱི་བྷི་དང༌། ཛི་བྷི་ནི་ལུས་ཀྱིས་གཡོ་བའོ། །དན་ཤ་ནི་མཐུས་འཚེ་བའོ། །བྷཉ་ཇོ་ནི་ཀུན་དུ་འཇོམས་པའོ། །པ་ས་ནི་ཀུན་དུ་འགྲོ་བའོ། ་། :ཛ་ཡཱ་དཱི་ནཱཉྩ། ཛ་པ་ལ་སོགས་པ་དེ་རྣམས་ཀྱི་བྱ་སའི་ཕ་རོལ་ན། ཙེ་ཀླུ་ཏ་ཡོད་ན་རྗེས་སུ་་་འཇུག་པའི་དབྱངས་ཐོབ་པོ། །།ཛ་པ་ལ་ཡ་སྦྱར་བས་ཛ་པ་གཉིས་པ་བྱས་ཏེ། པ་ཕྱི། ཛ་ལ་ཐིག་ལེའོ། །།ཛཾ་ཛ་སྱ་ཏེའོ། །ཛ་:བྷ་ལ་ཡ་ཙྪེསྦྱར་བས་གོང་བཞིན། ཛཾ་ཛ་བྱ་ཏེའོ། ཊ། དན་ཤ་ལ་སྦྱར་བས་ན་ཕྱི། གོང་དུ་ད་གཉིས་པ་བྱས། ་གོང་བཞིན། དཾ་ད་བྱ་ཏེའོ། ?། བྷཉྫ་ལ་ཡ་སྦྱར་བས་གོང་བཞིན། ཐཾ་བྷ་ཛུ་ཏེའོ། །པ་ས་པ་ལ་ཡ་སྦྱར་བས་གོང་བཞིན། པཾ་པ་སྱ་ཏེའོ། །།ཙ་ར་ཕ་:ལོ་དུ་ཙཱ་པ་་ར་སྟ་སྱ། ཙ་ར་ནི་འགྲོ་བའི་དོན་ཏོ། །ཕ་ལ་ནི་ངེས་པར་རྫོགས་པའོ། །དེ་་རྣམས་ཀྱི་ཨ་ཧྥུ་ཅེས་ལ་རྗེས་སུ་འཇུག་པའི་དབྱངས་ཐོབ་པར་གྱུར་པ་སྟེ་ཕ་རོལ་གྱི་ཨ་ལ་ཨུ་ཡང་གཞག་པ་སྟེ། ཙེ་ཀྲི་ཡི་ཏ་ཕ་རོལ་ན་ཡོད་ནའོ། །ཙ་ར་ལ་ཡ་་་སྦྱར་བས་གོང་དུ་ཙ་གཉིས་པ་བྱས། མདོ་འདིས་ཐིག་ལེ་ཐོབ་པ་དང་། །་་་་་་འོག་མས་ཨུ་ཐོབ། །མདོས་བསྲིངས་པའོ། ་། ཙོ་ཙུརྱ་ཏེའོ། །ཕ་ལ་ལ་ཡ་་་སྦྱར་བས་གོང་མ་བཞིན། ཕོ་ཕུ་མྱ་ཏེའོ། །:རཾ་མད་ཏོ་རཱི། གང་ལ་རེ་ཡོད་པའི་དྷཱ་ཏུ་དེའི་ཕ་རོལ་ན་ྋཙེ་ཀྲི་ཏ་ཡོད་ན་ཨ་བྷཱུ་ས་ལ་རེ་འབྱུང་སྟེ། ཚྭ་། །
