རཱ་ཡ། རཱ་ཡམ། །རཱ་ཡོ། །རཱ་ཡ། ་རཱ་ཡཱ། རཱ་བྷཱུམ། རཱ་བྷི། རཱ་ཡེ། རཱ་བྷཱུམ། རཱ་ཤྩ། 《རཱ་ཡ། རཱ་བྷཱམ། རཱ་ཤྩ། རཱ་ཡ། རཱ་ཡོ། རཱ་ཡཾ། ་རཱ་ཡི། རཱ་ཡོ། །རཱ་སུ། །ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་རཱ། ཧེ་རཱ་ཡོ། ཧེ་རཱ་ཡ། །ཕྱོགས་པ། །ཟླགས་པ་འོ་ཞེ་ན་[prག་དབང་བློ་ཀྱང་། ལ། གོའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི་གྷཊ་ལ་གོའི་ཨོའོ་ཞེས་པས་གོའི་ཨོ་ཡིག་ཨོ་རུ་འགྱུར་རོ། །གོ། ཨོ་ཨཱ་ཝའོ། །གཱ་ཝོ། གཱ་ཝ། ཨམ་ཤས་་དག་གི་ཨཱའོ་ཞེས་པས་ཨཱ་རུ་འགྱུར་རོ། #། གཱ་མ། གཱ་ཝོ། །སྔ་མ་བཞིན་དུ་ཨཱ་རུ་འགྱུར་རོ། །གཱ། དབྱངས་དང་པོ་ཅན་ལ་ཨོ་ཨ་ཝའོ། །ག་ཝཱ། གོ་བྷཱུམ། གོ་བྷི། ག་ཝེ། གོ་བྷཱུམ། གོ་བྷུ། ང་སི་ངས་དག་གི་ཨ་དབྱི་བའང་ངོ་་ཞེས་པ་སྤྱིར་བཏང་ནས་གོ་ལས་ཀྱང་ངོ་ཞེས་པས་ང་སི་ངས་དག་གི་ཨ་དབྱིའོ། །གོ། གོ་བྷཱུམ། གོ་། z་གོ། ་ག་ཝོ། ག་ཝཾ། །ག་ཝི། ག་ཝོ། གོ་ཞུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་གོ། །ཧེ་གཱ་ཝོ། ཧེ་གཱ་ཝ། གོའི་སྒྲ་འདི་བུད་མེད་དང་སྐྱེས་པ་ལ་འཇུག་སྟེ་གཉི་ག་ལ་ཡང་ཁྱད་པར་མེད་དོ། །གོ། །གཱ་ལོ༑། གཱ་ཝ། དི་བ་མཚུངས་པའི་དོན་ཅན་བུད་མེད་ཀྱི་རྟགས་ལ་བྱོའི་སྒྲ་་་འཇུག་སྟེ། །ཞེདེ་འང་གོའི་སྒྲ་དང་འདྲ་བར་གཟུགས་སུ་གྲུབ་བོ་ཞེས་ཁ་ཅིག་སེམས་པར་བྱེད་པ་ནི། དཔེར་ན་རྒྱོ། རྭ་ལོ། ཏྭ་ཝ། །ཏྭ་ལོ། y། ཏྭ་ཝཱ། བྱོ་བྷཱུམ། བྱོ་བྷིཞེས་པ་ལ་སོགས་པའོ། །།དབང་ཕྱུག་གོ་ཆའི་བཞེད་པས་ནི་མཚན་ཉིད་ཡོངས་སུ་བཤད་ནས་རབ་ཏུ་སྦྱོར་བ་ཉིད་མེད་དེ། ཨམ་ལ་ཡང་ནའོ་ཞེས་པའི་མདོ་མཛད་པའི་ཕྱིར་རོ། །རྣམ་པ་གཞན་དུ་དི་ཞེས་པའི་དྲོའི་སྒྲ་ཉིད་མི་འགྲུབ་ཅིང་དོན་ཐ་དད་པ་ཡང་མེད་དོ་ཞེས་སོ། །གློའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། དང་པོའི་གཅིག་གི་ཚིག་ལ་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་་།
