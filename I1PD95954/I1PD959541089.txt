བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཨ་ཏི་སྦི། ན་པ་ཤ། ཨུ་ཡིག་གི་མཐའ་ཅན་དྷེ་ནུའི་སྒྲ་སྟེ། འདིའི་མཚན་ཉིད་ནི་:བྷུ་མིའི་སྒྲ་དང་འདྲའོ། །དྷེ་ནུ། ལ་དྷེ་ནུ། དྷེ་ན་ཝ། ་དྷེ་ཧཱུྃ། དྷེ་ནུ། དྷེ་ནུ། དྷེ་ནཱ། དྷེ་ནུ་ཧཱུཾ། །དྷེ་ནུ་བྷི། དྷེ་ན་ཝེ། །དྷེ་ནུ་ཧཱུཾ། དྷེ་ནུ་། །དྷེ་ནཱ། དྷེ་ནུ་བྷཾ། ་དྷེ་ནུ་བྷཱུ། དྷེ་ནཱ། །དྷེ་ནོ། དྷེ་་་ནུ་ཧཱུྃ། དྷེ་:ནཱི། དྷེ་ནོ། དྷེ་ནུ་ཞུ། ཊི་ཀུན་དུ་བོད་པ་ལ། །ཧེ་དྷེ་ནོ། ཧེ་དྷེ་:ནུ། ཧེ་དྷེ་ན་ཝ། །དེ་བཞིན་དུ། ཨིཀཱ། :ཨཱ་ཀུ། ཨི་རུ། ཨཱ་རུ། རུ་ཏུ༄༅། །ཀ་རེ་ཎུ། :：ཤ་ཏི་དུ། ཝཱ་སིན་དྷུ། ཀྱི་ཤ་ཏ་དུ། །ཀནྟུ། ཀཱ་ཀུ། པྲི་ཡཾ་གུ། ཀུ་ཧ་རེ་ཧིུ། རཛྫཱ། ཧ་ནུ་དེ་བཞིན་དུ། ཀུན་དུ་རུ། :མྲྀ་ཏུལ། ཨ་ལུ་ཞེས་པ་ལ་སོགས་པ་རྣམས་སོ། །ཨཱུ་ཡིག་གི་མཐའ་ཅན་བ་དྷཱའི་སྒྲ་སྟེ། འདིའི་བྱ་བ་ན་དཱི་དང་:འདྲའོ། །ཨཱི་ཡིག་གི་མཐའ་ེལས་སིའོ་ཞེས་པའི་སི་་དབྱི་བ་མེད་དོ། །རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ། །བདྷུ:：ཨུའི་ཡི་གེ་ཝ་ཞེས་པས་ཝ་ཉིད་དོ། །བདྷོ། བ་དྷཱ། བ་དྷཾ། བདྷོ། བ་དྷཱ། བ་དྷཱ། བ་དྷུ་བྷཾ། བ་དྷ་བྷི། བ་དེ། །བ་དྷཱ་ཧཱུཾ། བ་དྷཱ་ཤྩཎི། །བདྷཱ། བ་དྷཱ་ཧུཾ། །བདྷུ་བྱ། བདྷཱ། བ་དྷོ། བདྷཱ་ནཾ། བ་དྷི། བ་དྷོ། བ་དྷཱ་ཏེ། ཀུན་དུ་བོད་པ་ལ། ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། ཧེ་བ་དྷཱ། དེ་་བཞིན་དུ་ཏ་ནུ་ལུས་ལའོ། །དྷ་ནུ་མཚོན་ལའོ། ་། ཀརླུ་བ་ལས་བྱུང་བའི་དག་བྱེད་ལའོ། །:ཁསྡུ་ཀ་ཎ་ཏི་ལ་འདོད་དོ། །ཏརྡུ་བུད་མེད་ཀྱི་ལག་པ་ལ་་འགྱུར་རོ། །:：ཨནྡྷ་ར་ལ་འགྱུར་རོ། །:：ཀརྐན་དྷ་རྒྱ་ཤུག་སྩོལ་འདོད་དོ། །སརྫུ་དབང་དུ་བྱེད་པ་ལའོ། །ཛམྦུ་ར་ལ་ཙཱུ་ཏུལླི་ཀའི་འབྲས་བུའོ། །ཤཀྟི་ཞེས་བྱ་བའི་མཚོན་ཆ་ལ་ཀཱ་:སཱུའོ། །ཀཱནྡྷ་ཨེཕ་རོལ་གྱི་ནོར་འཕྲོག་ཨེཔ་ལའོ། །:པཱ་ཌུ་བུ་རྙིན། བནྡ་བྷཱུརྞཱི། བ་རུ་སྐྱོ་བ་ལ་ཙ་:མཱུའང་ངོ༌། །དྲི་རཱུ་ཌཱ་དི། དྷི་་།
