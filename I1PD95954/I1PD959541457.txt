::བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིཀྱི་འི་བིཊྚ༑། པ་ཨ་སྣ་བིཊྚ། ཨསྠཱ་བི་ཎཾ། ཨསྠཱ་བི་ཥ། །ཨསྣ་བི། ནཱ་མི་ཅིམཐའི་ཡོན་ཏན་ནོ། །ཨསྟོབྦ། ཨསྟོ་ཥཱ་ཏྭཾཨསྟོ་ཥ་ཏ། །ཨསྟོ་ཥྛ། ཨསྟོ་ཥཱ་ཐཱ། ཨསྟོ་བྷཾ། ཨསྟོ་ཏི། ་ཨསྟོ་ཥཱ་ཧི། ཨསྟོ་ཟླ་ཧི། ཨ་སཱ་བཱིཏ། ་ཨ་སོཊྚ༈:：ཨ་དྷཱ་བཱིདི། །ཨ་དྷོ། །སྦྱོང་། ༼ཨ་ཙུཊྚ། རླུང་། །ཨ་བློབྦི། ཡུ་སྤེལ་བ་ལའོ། །བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་གསལ་བྱེད་དང་པོ་ཡ་ཡིག་་དང་པོ་མ་ཡིན་པའི་དང་པོ་ཨིཊ་ཨཱ་ག་མའོ་ཞེས་པས་སི་ཙ་ལ་ཨིཊ་ཨཱ་ག་མའོ། །ཨ་ཡཱ་བཱིཏ། ་ཨ་ཡཱ་བཱིཏྟཾ། ཨ་ཡཱ་བི་ཥཱ། རུ་སྒྲ་ལའོ། །།ཨ་རཱ་པཱིཏ། །ཆུ་བསྟོད་པ་ལའོ། །ཨ་ནཱ་པཱིཏ། གླུ། ་ཨ་ཀྵ་པཱིཏ། ་། །ཨ་བྷཱ་བཱིཏ། བྷུ་ཨ་སྣ་བཱིཏ། །ཨུཙྪཱཉ། །སི་ཙའོ། །ཨིཊ་ཨཱ་ག་མའོ། །ཨུཎྞོ་ཏིའི་ཡང་ནའོ་་་ཞེས་པས་གཞན་གྱི་འདོད་པ་ཡིས་ཡང་ན་འཕེལ་བའོ། །འདིར་ནི་ཕྱོགས་གཅིག་ལ་འཕེལ་བ་ཉིད་ཚད་མའོ། ཀ། རྒྱལྟ་ནི་ལ་ཡང་ངོ་ཞེས་པ་འདིར་ཡང་:ཡི་གེ་མ་་་བརྗོད་པ་ེབསྡུས་པའི་དོན་ཉིད་ལས། ཨུཎྜོ་ཏིའི་ཡང་ན་ཡོན་ཏན་ནོ། །ཡོན་ཏན་མེད་པའི་ཕྱོགས་ལ་སྔ་མ་བཞིན་འཕེལ་བ་མེད་དོ། །སྔ་མ་བཞིན་ཨུབ་རུ་བསྒྱུར་རོ། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །ཨོཙྪཱ་བཱིཏ། ཨོཙྪཱ་བཱིཏ། ཨོ་ྩ་བཱིཏ། ་ཨོ་ཌཱ་བིཏྟཾ། ཨོཙྪཱ་བིཤྩ། ཨོཙྪཱ་བིཏྟཾ། ཨོཙྪཱ་བི་ཥ། ཨོཙྪཱ་བི་ཥཱ། ཨོཙྪཱ་བི་ཥ། །ཨོཙྪཱ་བིཊྚ། །ཨོཥྛ་བིཊྚ། ཨོཙྪཱ་བིཊྚ། ཨོཏྟཱ་བི་ཥཱ་ཏྭཾ། ཨོཎྜ་བི་ཥཱཏྭཾ། ཨོཙྪཱ་བི་ཥཱ་ཏྭཾ། ཨོཙྪཱ་བི་ཥཏ། ཨོཥྛ་བི་ཥཏ། ཨོཾནྡྷཱ་བི་ཥཏ། ཨོཙྪཱ་བིཏྟཱ། ཨོཥྛ་བི། །ཨོཾནྡྷཱ་བིཏྟཱ། ཨོཙྪཱ་བི་ཥཱ་ཐཱ། ཨོཎྜ་བི་ཥཱ་ཐཱ། ཨོཙྪཱ་བིཏྟཱཙྪཱཐཾ། སྔ་མ་བཞིན་དྷ་ཡིག་གི་ཌྷ་ཡིག་གོ། །ཨོཙྪཱ་བི་ལྷོ༑ཨོཎྜ་བི་དྷྭཾ། །ཨོཙྪཱ་བི་ཞྭ། ཨོཙྪཱ་བི་དྷཾ། ཨོཙྪཱ་བི་དྷཾ། ཨོཾནྡྷ་བི་དྷཾ༑ཀུང་སྒྲ་ལའོ། །ནཱ་:མི་མཐའ་ཡོན་ཏན་ནོ། །ཨ་ཀོ། །ཨ་ཀོ་®ཥཱ་།
