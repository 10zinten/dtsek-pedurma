དང་། འ་ར་བ་ན་བ་སོགས། །བ་བ། ད་། དེ་ལ་བྱེ་ཀ་ཁ་ཡི་ཞེ་ཆེ་ཡིན་། ཚོགས་ལ། ད་ལྟའང་ལག་ཚོ་ལོ::ཀ་བ་ཁོ་:ཤར། ཀ་ཁ་ག་བ་:::ཀ་བཤད། །ཀ་བ་བ་བ་ན་སྐྱོ་བ་འདུལ་བའི་ཁ་བ་མར་ཁp':ཀ་ཁ་༄༅། །སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུ་ཞེས་བྱ་བ། ཡོད་པ། གང་བ་(ਬਲਕਰਭੂਆਨੁਕਸਨਾਲ། ཁ་ཚེ་ན་འཚོ་བ་ན::ཀ། ཏཱ་#། ལྟ་བ་བྱེ་མའོ་ཎཏུ་གྱུར་༄༅། །རྒྱ་གར་སྐད་དུ། ཝ་ཙ་ན་མུ་ཁ་ཡུ་དྷོ་པ་མ་ནཱ་མ། བོད་་་སྐད་དུ། སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུ་ཞེས་བྱ་བ། སྨྲ་བའི་མཆོག་ཐམས་ཅད་་མཁྱེན་པ་ལ་ཕྱག་འཚལ་ལོ། །སྨྲ་མཆོག་གཙོ་ལ་ཕྱག་འཚལ་ནས། །སྨྲ་བའི་ཡན་ལག་རྫོགས་དོན་དུ། །ངག་གི་འདུ་བྱེད་གང་དེ་ཡི། །ཚོགས་པ་བདག་་གིས་བཤད་པར་བྱ། །ཁོག་པ་མགྲིན་པ་རྐན་དང་ལྕེ། །སྣ་དང་སྤྱི་བོ་སོ་དང་། མཆུ༑། ཕྱི་དང་རེ་རེའི་ཆ་དག་ལས། །ཕྲད་དང་ཅུང་ཟད་ཕྲད་པ་དང༌། །དོག་པ་དང་ནི་ཡངས་པ་ནས། :། དྲང་པོ་བསྒྲེང་དང་བཀུག་པ་དང༌། །བསྟོད་པ་དང་ནི་སྨད་པ་དང༌། །བསྒྲིམས་དང་གློད་པའི་ང་རོ་འབྱུང༌། །དྲང་པོ་་བཀུག་དང་སྨད་པ་དང་། ::། བསྒྲིམས་པ་དང་ནི་གློད་པ་ལ། །གྲྭར་དང་ཙཱང་་གིས་ཕྱེ་བ་དང༌། །བསྒྲེང་དང་བསྒྲེངས་ཏེ་བཀུག་པ་དང་། །བསྟོད་དང་་་བསྟོད་དེ་སྨད་པ་དང་། །སྣ་ལྡན་རྣམ་པར་གཅད་པ་ཡིས། །།ང་རོ་བཅུ་དྲུག་།
