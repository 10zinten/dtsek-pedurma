བསྟན་འགྱུར། སྒྲ་མདོ། ཤི་ཙེ་ཏ་ཏི། ཎི་དྷུ་འགྲོ་བ་ལའོ། །སེ་དྷ་ཏི། སྦྲ། སེ་ནི། སེ་དྷ་ཏི། སི་ཙ༑་སཉྫ། སཱནྡྲ། རྣམས་ཀྱི་ཨ་ཧྥུ་ཅིས་དང་གཞན་ལ་ཡང་ངོ་ཞེས་པས་སའི་ཥ་ཉིད་དོ། །པྲ་ཏི་ཥ་དྷ་ཏི། དེ་ལ་སེ་དྷ་ཏའི་འགྲོ་བ་མ་ཡིན་པ་ལའོ་ཞེས་་བརྗོད་པ་ལས་འགྲོ་བ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །པ་རི་སེ་དྷ་ཏི། མི་ཧ་འཁྲུད་པ་ལའོ༑། མེ་ཧ་ཏི། པ་རསྨིའི་ཚིག་གོ། །ཤཱི་ཏཱ་ཁ་དོག་ལའོ། །།ཨཱ་ཡིག་ནི་ཨཱ་ཏ་རྗེས་སུ་འབྲེལ་པ་ལས་ཀྱང་ངོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །ཕྱི། རུ་ཙ་སོགས་་་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་གོ། ་། ཤྭ་ཏ་ཏེ། ཉི་མི་ད་གཤེར་བ་ལའོ། །།ཉི་ཡིག་ནི་ཉ་རྗེས་སུ་འབྲེལ་པ་དང་འདོད་པའི་བློ་མཆོད་པའི་དོན་རྣམས་ལ་ལྟའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །མེ་ད་ཏེ། ཉི་སྤྱི་དཱ། ་གྲོལ་བ་ལའོ། །སྐྱེ་ད་ཏེ། ཨཏྨ་ནེའི་ཚིག་གོ། །ཏྲེ་ཥ་འབར་བ་ལའོ། །ཡ་ཛ་སོགས་སོ། །ཏེ་ཥ་ཏི། ཏྲེ་ཥ་ཏེ། ཛཱི་བ་སོག་འཛིན་པ་ལའོ། །ལྷག་མ་ཉིད་ལས་པ་རསྨིའི་ཚིག་གོ། །ཛཱི་བ་ཏི། ཤི་ལ་ཏིང་ངེ་འཛིན་ལའོ། །ཤི་ལ་ཏི། མཱི་ལ། མི་ལ། མིག་་་འཛུམས་པ་ལའོ། །མཱི་ལ་ཏི། :སཾ་མཱི་ལ་ཏི། ནི་མཱི་ལ་ཏི། ཞིརྨི་ལ་ཏི། ཨུཏ་ྋསྔོན་དུའོ། ཏོ། ལྔ་པ་:ལྔ་ལ་པ་རྣམས་སུ་སྟེ་ཡང་ན་མ་ཡིན་པར་གསུམ་པ་་་རྣམས་སུའོ་ཞེས་པས་ཏ་ཡིག་ན་ཡིག་གོ། །ཨུཎཱི་ལ་ཏི། །ཀཱི་ཌཱ་གནས་པ་ལའོ། །ཀཱི་ཌ་ཏི། ཨ་ནུ་པ་རི་དག་ལས་ཀྱང་:ཀྲི་ཌ་ལའོ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨ་ནུ་ཀྲུ་ཌ་ཏེ། པ་རི་ཀཱི་ཌ་ཏེ། ཨཱ་ཀཱི་ཌ་ཏེ། སམ་ལས་:ཨུ་ཀཱུ་ཛ་ཨིན་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །སཾ་ཀྲི་ཌ་ཏེ། ཀུ་ཛ་ལ་ནི། །སཾ་ཀྲྀ་ཌ་ཏི། སྒྲ་མི་གསལ་བར་བྱེད་དོ་ཞེས་པའི་་་དོན་ནོ། །པ་རསྨིའི་ཚིག་གོ། །ཙཱུ་ཏིར་བཀྲུ་བ་ལའོ། །ཨིར་ནི། ཨིར་རྗེས་སུ་འབྲེལ་པ་ལས་ཡང་ནའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །ཙོ་ཏ་ཏི། བྷཱུ་ཏིར། །
