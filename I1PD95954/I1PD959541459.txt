བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་ཨ་ཊི་ཨ་བྷུ་ཏ། ཨ་བྷུ་ཁཾ། ཨ་བྷུ་བ། །ཨ་བྷུ་མ། ངེས་པ་མེད་པ་ལའང་ཨཱ་ག་ཏི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །།སི་ཙའོ། །།ཨིཊའོ༑། /། ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཨོཾ་ཨ་བའོ། །བྱ་རྱ་བྷ་བིཥྛ། བྱ་རྱ་བྷ་བི་ཥ'ཏྭཾ། བྱ་ཏ་བྷ་བི་ཥ་ཏ། བྱ་རྱ་བྷ་བིཥྛ། བྱ་རྱ་བྷ་བི་ཥ་ཛཾ། སྔ་མ་བཞིན་ཡང་ན་དྷ་ཡིག་གི་ལྷ་ཡིག་གོ། །བྱ་རྱ་བྷ་བི་བྷཾ། བྱ་རྱ་བྷ་བི་དྷཾ། བྱ་་རྱ་བྷ་བི་ཏི། བྱ་རྱ་བྷ་བི་ཥ་ཧི། བྱ་རྱ་བྷ་བིཧྥ་ཧི། དེ་བཞིན་དུ་ཨས་ཡོད་པ་ལའོ་ཞེས་པ་འདིའི་:ཨས་ཏིའི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་བྷུའོ་ཞེས་པས་སྔ་མ་བྷུ་རུ་བསྒྱུར་རོ། །ཨ་སྟི་ལས་དི་སི་དག་ལའོ་ཞེས་པས་ཏི་པའི་བྱིངས་ལས་་དབྱངས་གཟུང་བའི་ཕྱིར་གནས་ཅན་བཞིན་འགྱུར་བ་ཡིས་ཀྱང་ཨིཊའོ། །སྨྲེ་སྔ་མ་བཞིན་རཾ་:ཡིག་གི་འཕེལ་བར་ཨཱར་ཡིག་གོ། །ཨ་སྨནྡྲིཏ། ཨ་སྨཎཱ་ཊཾ། ཨ་མཱནྟུ། ཨ་སྨནྟི། །ཨ་སྣ། ཨ་སྨ། །ཨ་སྨནྟཾ། ཨ་སྨནྟུ། །ཨ་སྨདྨ། ཧྭ?། །གྱ་གྱུ་ཉིད་ལའོ། །ཨ་ཧཱནྟིཏ། དྷཱི་གནོད་པ་ལའོ། །ཨ་:དྷཱནཱིཏ། སྲི་འགྲོ་བ་ལའོ། །ཨ་སཱནྟིཏ། རཾ་རབ་ཏུ་ཐོབ་པར་བྱེད་པ་ལ་ཡང་ངོ༌། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། ་། ཨཱརཱིཏ། །པུ་ཥ་སོགས་ཀྱུ་ཏ་སོགས་ལ་ཡིག་རྗེས་སུ་འབྲེལ་པ་ཨར་ཏི་སར་ཏི་ཤཱསྟི་རྣམས་ལས་ཞེས་པ་ལ་སོགས་་པ་དང་ལྷན་ཅིག་སྤྱོད་པ་ལས་བི་ཀ་ར་ཎ་ཉིད་འཇིག་པ་གཟུང་བའི་ཕྱིར་འདི་དག་ལས་ཨཎ་མེད་དོ། །སམ་ལས་ག་མི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ས་མཱམྦྷི། ས་མཱཎཱ་ཏྭཾ། ས་མཱནྟ་ཏ། ས་མཱམྦྷ། ས་མཱནཱ་ཐཾ༑སྔ་མ་བཞིན་དྷ་ཡིག་གི་ཌྷ་ཡིག་གོ། །།།ས་མཱནྡོལ། །ས་མཱནྟི། ས་མཱནྟུ་ཧི། ས་མཱདྨ་ཧི། སྲིང་། །རྀཏ། མཐའ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་ལ་ཨིཊ་མེད་པའི་སི་ཙ་ལ་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཐུང་ངུ་ལས་ཀྱང་ཨིཊ་མེད་།
