བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི?ཏི། །:ན་རི་ནིཏྟི། །ན་རཱི་:ནི་ཏཱི་ནིི། ནརྣནྟི། ནརྨི་ཏི་ཏི། ཨཱ་ག་མའི་ཕྱོགས་གཅིག་ལ་སྔ་མ་བཞིན་ཏིའི་རྐྱེན་ལ་ཨིཊ་མེད་དོ། །ན་རཱི་ནིཏྟ། ན་་རི་ནིཏྟ། :ན་རི་ནིཏྟ། ནནྡྲིཏྟ། ན་རཱི་ནི་ཏ་ཏི། ན་རི་ནི་ཏ་ཏི། ནནྡྲི་ཏ་ཏི། བྲྀ་དྷ། ཏ་ཐ་དག་གི་དྷའོ། །སྔ་མ་བཞིན་གསུམ་པ་ཉིད་ཀྱང་ངོ༌། །བ་རི་བནྟི། བ་རཱི:༔དྷི་ཏི། བ་རི་བདྷི། བ་རཱི་བྲི་དྷཱི་ཏི། བཧྲི་དྷི་ཏི། བཱ་རཱི་བྲིདྡྷ། །བ་རི་:བྲིདྡྷ། བཧྲིདྡྷི། བ་རཱི་བྲི་དྷ་ཏི། བ་རི་བྲི་དྷ་ཏི། བཏྲི་དྷ་ཏི། ཐཾ་དྷུ། ཤ་རཱི་ཤནྟི། ཤ་རཱི་ཤི་ནི་ཏི། 《ཀ་ཤ་ར་ཤདྡྷི། ཤ་རི་ཤི་་དྷཱི་ཏི། ཤནདྷི། ཤནི་དྷི་ཏི། ཀྲུ་ཎིཔཱུ། ཀྲི་པའི་རའི་ལའོ། །ར་ཐོས་པ་ལས་ལ་ཐོས་སོ། །ཙ་ལཱི་ཀཥྛནྟི། ཙ་ལི་ཀླུ་པཱི་ཏི་ཙ་ལི་ཀལླཱི། ཙ་ལི་ཀྲི་པི་ཏི། ཙ་ལཀམྤི། ཙ་ཀྱི་པཱི་ཏི། ཙ་ལཱ་ཀླy། ཙ་ལི་ཀློy། ཙ་ལི་ཀླསྟ༑ཎི་ཙ་ལི་ཀླུ་པ་ཏི། ཙ་ལི་ཀླུ་པ་ཏི། ཙ་ལི་ྋཀླུ་པ་ཏི།  ོསྲི་ཛི་དྲི་ཤ་དག་གི་ཨ་ཡིག་ཨཱ་ག་མ་ཞེས་བྱ་བ་ལ་སོགས་པས་ཙུ་ཊ་སོགས་ལ་ཡོན་ཏན་ཅན་ལ་་དབྱངས་ལས་ཕ་རོལ་དུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །རྀའི་ཡི་གེ་རའོ། །པྲ་ཛིའི་དྷུཊ་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །དྲི་ཤིའི་ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་་་ཉིད་དོ། །སཱ་རཱི་སྲཊྚ། སཱ་རཱི་སྲི་ཛི་ཏི། ས་རི་སྲཊྚ། ས་རི་སྲི་ཛི་ཏི། སནྣཏྠི༄༅༑་སནྡྲི་ཛི་ཏི། ས་རི་སྲིཋ། ་་:སངྷི་ཤྩ། ས་རཱི་སྲི་ཛ་ཏི། ་ས་རི་སྲི་ཛི་ཏི། སནྡྲི་ཛཱི་ཏི། ད་རི་དྲཊཱི། ་་:ད་རི་ཤཱི་ཏི། ད་རི་དྲིཋི། ཙ། ད་རི་དྲི་ཤཱི་ཏི། དནྡཏྠི། དར་དྲི་ཤཱི་ཏི། ད་རི་དྲིཋ། ད་རི་དྲིཋ། ད་ར་དྲིཋ། །ད་རཱི་ཥ་ཏི། ད་རི་དྲི་ཤ་ཏི། དནྡ་ཙྪཱཤ་ཏི། ཛྲ་ཛཱ། མཛྫའི་:མཱཛྫིའོ། །བྷི་ཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །མ་རཱི་མཱཙྪི། ཡོན་ཏན་མེད་པའི་དབྱངས་ལ་ཡང་ན་བརྗོད་པར་བྱའོ། །ཞེས་པས་མཱ་རྫིའོ། །མཱ་རཱི་མཱ་—།
