རི་དྲི་ཥྛ་ཏེ། ཇི་ཨིཙ་བཞིན་ཨིཊ་འགྱུར་བ་ལས་འཕེལ་བའོ། །ཛ་ཡི་ཥྛ་ཏི། ཤིཉ༑ཤྭ་ཡི་ཉྩ་ཏེ། ཎཱིཉ། ནཱ་ཡི་ཥྛ་ཏེ། ནེ་ཥྛ་ཏེ། དྲི་དྷིང༌། །བེ་བཱིང་། དྲི་དྷི་བེ་བཱི་དག་གི་ཨི་ཡིག་ཡ་ཡིག་དག་ལའོ། །།ཞེས་པས་མཐའ་དབྱིའོ། །དཱི་དྷི་ཥྛ་ཏི། བེ་བི་ཥྛ་ཏེ། །ཐུ། ཤཱ་བི་བྱེ་ཏེ། ཤོ་ཥཱ་ཏེ། །བྷུ། བྷ་བི་ཥྱ་ཏི། ་བྷ་བི་ཥྛ་ཏེ། ལྷུཉ། ལཱ་བི་ཥྛ་ཏེ། ལ་བི་ཥྛ་ཏེ། འུ་ཀྲྀཉ། ཀ་རི་བྱ་ཏེ། ཀ་རི་ཥྛ་ཏེ། བྲིང༌། བྲྀཉ། རྀཏི། ་བྲིང༌། བྲྀཉའི་ཡང་ཡང་ན་རིང་པོའོ། །པ་རོགཱ་དང་ཨཱ་ཤཱིཥ་དག་མ་ཡིན་པ་ལ་ཨིཊའོ། །པ་རསྨི་མ་ཡིན་པའི་སི་ཙ་ལ་རབ་ཏུ་བརྗོད་དོ། །ཞེས་སྦྱོར་བ་རྣམ་པར་ཕྱེ་བ་ལས་ཀྱང་ཨིཊའི་ཡེཡང་ན་རིང་པོའོ། །ཨི་ཙ་བཞིན་ཨིཊ་འགྱུར་ཞིང་དེ་མི་འགྱུར་བ་ཡང་ཐོས་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །བཱ་རི་ཥྛ་ཏི། བཱ་རི་ཥྛ་ཏེ། ཀྲྀ། ཏཱ་རི་ཥྛ་ཏེ། །ཏ་རི་བྱ་ཏེ། ཧ་ན། :ཨིཙ་ཉིབཞིན་འགྱུར་བ་ལས་ཨའི་ཨུ་པ་དྷཱ་རིང་པོའོ། །ཧའི་གྷ་ཉིད་འང་ངོ༌། །གྷཱ་ནི་ཥྛ་ཏེ། :ཧ་ནི་ཥྛ་ཏེ། གྲ་ཧ། །གྲ་ཧི་ལས་པ་རོག་་་མིན་པ་ལ་ཨིཊའི་རིང་པོའོ་:ཞེས་པས་ཨིཊའི་རིང་པོའོ། །ཨིཙ་བཞིན་ཨིཊ་འགྱུར་ཞིང་མི་འགྱུར་བ་ཡང་ཐོས་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །གྲྭ་ཧི་ཥྛ་ཏེ། གྲ་ཧཱི་ཥྛ་ཏེ། དྲི་ཤིརི། དནི་ཉྩ་ཏེ། དྲག་ཏེ། །པ་ཋི། ཨིཊ་ཨཱ་ག་མའོ། །པའི་ཙྪིཥྛ་ཏེ། པ་ཙ། པ་ཏེ། ལས་བྱེད་པ་པོ་ལས་བཞིན་ནོ་ཞེས་པས་ལས་བཞིན་འགྱུར་བ་ལས་ལས་བྱེད་པ་པོ་ལ་ཨཱཏྨ་ནེའི་ཚིག་གོ། །དབྱངས་དང་ཧ་ན་ལ་སོགས་པས་་་ཡང་ན་ཨིཙ་བཞིན་ཨིཊ་ལའང་ངོ༌། །ལཱ་བི་ཥྛ་ཏི། ལ་བི་ཥྛ་ཏི། ཀེ་དཱ་རཿསཱ་ཡ་མེ་བ། བྷིཥྛ་ཏེ། ཀུ་ཤུ་ལསཱ་ཡ་མེ་བ། པགྱུ་ཏེ། ཨོ་ད་ནསཱ་ཡ་མེ་བ། །ཀུ་ཊི། རཾ་ཛ་དག་གི་སྐྱ་ལ་སོགས་པའི་ཡུལ་ལ་ཡང་པ་ར་སྨིའི་ཚིག་མཐོང་་་བ་མེད་དོ། །ཀོ་ཎི་བྱ་ཏེ། པཱ་དསཱ་ཡ་མེ་བ། རཾ་གྲུ་ཏེ་བསྭོ་སཱ་ཡ་མེ་ག་།
