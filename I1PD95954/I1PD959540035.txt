ཕྱིས་པ་ལ་སླར་མཚམས་སྦྱོར་བ་མ་ཡིན་ནོ། །རའི་ར་ལ་དབྱི་ཞིང་དབྱངས་སྔ་མའང་རིང་པོའོ། །དབྱངས་ལས་ཕ་རོལ་ཚ་ཡིག་གཉིས་སུ་འགྱུར་རོ། །ཞེས་པ་མཚམས་སྦྱོར་ལས་རྐང་པ་ལྔ་པ་རྫོགས་སོ། །།།།།་་བློ། །ཁ་ཁ། བདེਉਹਦੇAMAR*ਜੋਲਕਸਰਸ:ਬਖਸਲਤਨਨੂੰབྱིངས་དང་རྣམ་དབྱེ་བོར་བའི་དོན་དང་ལྡན་པ་རྟགས་སོ། །དེ་ལས་ཕ་་་རོལ་རྣམས་རྣམ་དབྱེའོ། །་། སི་ཨོ་ཛས་དང་པོའོ། །ཨམ་ཨོ་ཤས་གཉིས་པའོ། །ཊཱ་བྷཱུམ་བྷིས་གསུམ་པའོ། །ངེ་བྷཱུམ་བྱས་བཞི་པའོ། །ང་སི་བྷཱུམ་བྱུས་ལྔ་པའོ། །ངས་ཨོས་ཨཱམ་དྲུག་པའོ། །།ང་ཨོས་སུབ་བདུན་པའོ། །དང་པོ་ལྔ་གླུ་ཊའོ། །མ་ནིང་ལ་ཛས་ཤས་དག་གོ། །བོད་པ་ལ་སི་སམྦུདྷིའོ། །ཨཱ་ག་མ་ཨུཏ་རྗེས་སུ་་་འབྲེལ་པ་དབྱངས་མཐའ་ལས་ཕ་རོལ་ཏུའོ། །གསུམ་པ་སོགས་ལ་ནི་ཕ་རོལ་གྱི་་་དང་པོར་རོ། །ཨིཏ་ཨུཏ་ཨབྷིའོ། །།ཨཱིཏ་ཨུཏ་བུད་མེད་བརྗོད་པ་དག་ན་དཱིའོ། །ཨཱ་ཤདྷཱའོ། །མཐའ་ལས་སྔ་མ་ཨུ་པ་དྷཱའོ། །གསལ་བྱེད་ལས་ན་ཨ་ནུ་ཥངྒའོ། །མཐར་གནས་དང་རྗེས་སུ་སྣ་ལྡན་མ་ཡིན་པའི་གསལ་བྱེད་དྷུ་ཊའོ། །ཨ་ཡིག་སྒྲ་་ལྡན་ལ་རིང་པོའོ། །ཛས་ལའོ། །ཤས་ལ་སྟེ་སའི་འང་ནའོ། །།ཨ་ཡིག་ལ་་་དབྱིའོ༑། བྷིས་ཨིས་ཡང་ནའོ། །དྷུཊ་མང་པོ་ཉིད་ལ་ནི་ཨེའོ། །ཨོས་ལ་འོང་། ངོ༌། །ང་སི་ཨཱཏའོ། །ངས་སྐྱའོ། ་། ཊཱ་ཨི་ནའོ། །ངེ་ཡའོ། །སརྦ་ནཱ་མ་་་ལས་:ནི་ངེ་སྨིའོ། །ང་སི་སྣཏའོ། །ང་ནི་སྨིནའོ། །པཱུརྦ་སོགས་ལས་ཁྱད་་་པར་དུ་བཤད་པར་བྱ་བ་དག་གོ། །ཐམས་ཅད་ལས་ཨོཾ་ལ་སུའོ། །ཛས་ཐམས་ཅད་ཨཱིའོ། །།།::ཨཱལྤ་སོགས་ལས་ཡང་ནའོ། །པཱུརྦ་སོགས་ལས་ཀྱང་ངོ༌། །ཟླས་དབྱེ་གནས་པ་ལས་ཀྱང་ངོ་། །གཞན་སརྦའི་མིང་ཅན་མ་ཡིན་ནོ། །གསུམ་པའི་ཚིག་སྡུད་ལའང་ངོ༌། ་། འབྲུ་མང་པོ་པ་ལའང་ངོ༌། །ཕྱོགས་རྣམས་།
