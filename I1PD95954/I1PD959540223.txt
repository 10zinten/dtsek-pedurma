ཁ་བསྟན་འགྱུར། སྒྲ་མདོ། ལི་དང་རྫོགས་སོ། །།།ཚེ་བ། །བལ། དང་དོ་་་ནིདེ་ཚོ་ལྷ་བླ་མ་ཞི་བ་འོད་ཀྱིས་ཞང་ཞུང་གི་ས་ཆ། ཐོ་ལིང་གསེར་གྱི་གཙུག་ལག་ཁང་དུ་བསྒྱུར་བའོ། །ཅི་། ང་ཚོའི། །ལྔ། །དེ་དལ་བའོ༽། གས་ཀྱི་དོ་ར་། ད་ལྟ་རྒྱ་བལ་པ་ཞི་བའི་རྒྱུ་དང་རྗེ་n:དབསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ཀ་ལཱ་པའི་འགྲེལ་པ་ཉུང་ངུ་ལས་སློབ་མ་ལ་ཕན་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །ཞེས་འཁོད། ཉི«པེ་»《སྣར་»དོ། «པེ་»ཐས། «སྣར་ཐསྲི། (ཆོས་ཚན་འདིའི་ནང་དུ་«པེ་»«སྣར་ལ་སྔགས་སྲོག་མེད་ཀྱི་རྟགས་ཅན་བསྡུར་་་་་་ཀྱི་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་པ། «པེ་»«སྣར་»དེ། «པེ་»«སྣར་»ཏི་ཎི་ཏི་ཎི། ༄༅«པེ་»«སྣར་དྷ། «པེ་»«སྣར་»+། ་་་་་་:བར་དུ་ཕྱོགས་ཀྱི་ནད་«པེ་»«སྣར་བྱ་བའི། ®«པེ་»«སྣར་»དྷ། «སྣར་»ཎ་མ། «པེ་»«སྣར་ཐོབ་པར་འགྱུར། «པེ་»ཏུད་དའི། «སྣར་»ཏྱད་དེའི། «པེ་»«སྣར་»ཙ། @«པེ་»«སྣར་»སྨི། «པེ་»«སྣར་+::པི། «པེ་»«སྣར་»དྷུ། «པེ་»«སྣར་»ཏོ། ་ཀྱི་གནས། «པེ་»«སྣར་»ཏདྲ། «སྣར་+ཏ། «པེ་»«སྣར་སྐ། «པེ་»«སྣར་སྒྲུབ། གྱེ«པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་དོན། «པེ་»«སྣར་»དྷ། ཨེ《སྣར་»ཏུཾ། «པེ་»«སྣར་ཝཱམ། དེཚེ«པེ་»《སྣར་འབྲེལ། ཀྱེ«པེ་»སྦྱར་རོ། །ཡུ་ཝཾ་པ་ཙ་ཐས། «སྣར་སྦྱོར་རོ། །ཡུ་ཝཾ་པ་ཙ་ཐས། «སྣར་ཡུ། བྱེ«པེ་»«སྣར་»+སི། «པེ་»«སྣར་ཐ། «པེ་»«སྣར་»། «པེ་»«སྣར་»ལས། (《སྣར་»ཕྱིས། «པེ་»«སྣར་»མིའོ། «པེ་»«སྣར་ཨ། ནི«པེ་»«སྣར་ཝ། འགྱི«པེ་»ཡཾ་པ་ཙཱ་མའོ། «སྣར་ཡཾ་པ་ཙཱ་མའོ། »«པེ་»«སྣར་ཙཱ། ་་་་་་་!/ི«པེ་»«སྣར་»ཙཱ། «པེ་»«སྣར་ཨ། 《པེ་»«སྣར་»པའང༌། «པེ་»«སྣར་དེ། «པེ་»«སྣར་»དྷཱ། «པེ་»«སྣར་»ཆ་ཉ། ༄༅«པེ་»«སྣར་»ཊ། «སྣར་དུ། «སྣར་»དེཔྲ། ཀྱེ《སྣར་»སྟེར། ཀྱེ«པེ་སྟེ། «སྣར་»ཏེ། ཉེ«སྣར་ཨཱི། «པེ་»«སྣར་»དྷཱི། ཨེ«པེ་»«སྣར་མ། །«སྣར་ཨན། «པེ་»«སྣར་བྲི། (སྐྱེ«པེ་»«སྣར་»། ཏེ«པེ་»«སྣར་ཀྲི། ་«པེ་»«སྣར་+པཱ་ཐར་བརྗོད་པ་ནི་དྷཱ་ཏུ། གྱི«པེ་»«སྣར་གྱི། ཀྱེ«པེ་»«སྣར་»། །འཉེ«པེ་»«སྣར་ཨཱན། «པེ་»«སྣར་། །D«པེ་»«སྣར་»སྨེ་ནཱ་ཏི། @«པེ་»«སྣར་+པྲ། «པེ་»«སྣར་བྱའོ། «པེ་»«སྣར་»ཀ།
