བསྟན་འགྱུར། སྒྲ་མདོ། ༼ལིའིསྤྱད་དང་། ཨིཊ་སྦྱར་བས་གོང་མ་བཞིན་ཨ་:ཀཱ་རི་ཏའོ། །བྷུ་ཏ་ཀ་ར་ཎ་ཞེས་བྱ་བ་ནི་ཨ་ཐོག་མར་བྱས་པའོ་། :་་བ་། །ཞི་བ་དང་ཕྱི་ལོ་ང་། ཎཱ་ལ་བྷ་བི་ཥྛ་ཏི་བྷ་བི་:ཥྛཏྭཱ་ཤཱིཤཱསྟནྱ། ཚོ་མ་འོངས་པའི་དུས་རྗོད་པའི་བྱེད་པའི་ཚིག་ནི། བྷ་བི་ཉྩན་དང༌། ཨཱ་ཤིཥ་དང༌། ཤཱསྟ་ནཱི''སྟེ། འདི་་་རྣམས་ནི་མ་འོངས་པའི་དུས་ཀྱི་དྷཱ་ཏུ་ལ་འབྱུང་བར་འགྱུར་རོ། །བྷ་བི་ཥྛནྟིའི་་་དཔེ་ནི། ཀྲི་ཨིལ་སུ་ཏི་དང་། ཨཱིཏ་སྦྱར་བས་རཾ་ཨཱར་དུ་འཕེལ་ཏེ་ར་དང་ཨི་་སྦྱར། སས་ཥ་ཐོབ་པའོ། །ཀ་རི་ཥྛ་ཏིའོ། །ཎྜིཨཱ་ཤིཋ་ཀྱི་དཔེ་ནི། བྷ་ལ་ཡཱཏ་སྦྱར་བའོ། །:བྷཱུ་ཡཱཏའོ། །ཤཱསྟ་ནཱིའི་དཔེ་ནི། །པ་ཐ་:ལ་ཏཱ་དང་ཨིཊ་སྦྱར་བའོ། །པ་:ཋི་ཏཱའོ། །ཏི་་ན་་ས་་བ་བདེ་བ་རྒྱུན་ཏུ་ཁོ་ཏཱ་སཾ་སཱ་སཾ་ཛྷ་བྷི་ཌཱ་ལ་པི་ཤེ་ཥ། པ་རོཉག་དང༌། ཧསྟ་:ནཱི་དང༌། ཀྲི་ཡ་ཏི་པཏྟི་དང༌། བྷ་བི་ནི་དང་། ཨཱ་ཤིཥ་དང་། ཤཱ་སྟ་ནཱི་སྟེ་འདི་རྣམས་ནི་རང་ཉིད་ཀྱི་མིང་དུ་བྱས་པའོ། །འདས་པ་དང་འབྱུང་བར་འགྱུར་བའི་ཁྱད་པར་གྱི་དུས་སུ་བྱའོ། །དེ་ལ་གང་མིག་ལ་སོགས་པའི་དབང་པོ་རྣམས་ལས་འདས་པ་དེ་ལ་ནི་པ་རོཉཀཱ་ཞེས་བྱའོ། །།ཏྲི་བུ་རཾ་དང་ད་ཧ་ལ་ཨཊ་སྦྱར་ཊ་ཕྱི། ཨ་དང་ཧ་སྦྱར། གཉིས་པ་དང་ད་ཧ་སྦྱར་ཏེ། ཧ་ཕྱི། :ཨ་སྱ་བ་དའི་་་མདོས་དཱ་ྋསྲིང་བའོ། །ཏྲི་བུ་རམ་ཅིད་དཱ་ཧ་རུ་:དྲའོས། །ཤིན་ཏུ་བསྙེན་པ་ལ་པ་རོཉགླུ་འབྱུང་བར་གསུངས་ཏེ༄༅། དམངས་རིགས་ཀྱི་ལག་ནས་མ་ཟོས་ཞེས་བསྙེན་པའོ། །བུ་ཛ་ལ་པ་རོཉཀྵ་ཨེ་སྦྱར། བུ་ཛ་གཉིས་པ་བྱ་སྟེ་ཛ་ཕྱི། བྷ་ས་བུ་ཐོབ་པའོ། །བུ་བྷུ་ཛེ། ཁར་ཙང་ནས་བརྩམས་པའི་འདས་པ་ནི་དེ་ལ་རྒྱལྟ་ནཱི་ཝའབྱུང་བའོ། །མྱ་ས་སཾ་ཧྲིལ་རྒྱལྟ་ནཱིདི་སྦྱར་བས་ཧའི་ཐོག་མར་ཨ་་སྦྱར། རཾ་ཨཱརྱ་དུ་འཕེལ་ཏེ། །ཨན་དང་སྦྱར་བའོ། །རྒྱ་ས་ས་ཧ་རཏྲའོ། །—།
