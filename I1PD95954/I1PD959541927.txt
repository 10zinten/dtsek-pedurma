བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི ོཐམས་ཅད་མཁྱེན་པ་དང༌། སྔོན་གྱི་མཁན་པོ་རྣམས་ཀྱི་རྗེས་སུ་སྨྲས་ཤིང་བཤད་པས་འགྲོ་བ་རྣམས་སྒྲ་དང་སྒྲའི་དོན་ལ་མཁས་པར་གྱུར་ཅིག་ཅེས་བྱ་བའི་དོན་ཏོ། །སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུའི་འགྲེལ་པ་རྫོགས་སོ། །་ཌི། ལའི་སྐེ་ལ་བ་་་་་་རྒྱ་གར་གྱི་མཁན་པོ་:དྲན་པའི་ཡེ་ཤེས་གྲགས་པས་བླ་མ་ནོར་བཟངས་ཀྱི་གྲགས་པ་ལས་ཐོས་ཏེ། སློབ་མ་གཞོན་ནུ་གྲགས་པའི་དོན་དུ་མིང་གི་སྦྱོར་་བ་ནི་ཅུང་ཟད་གཞག། ཕྲད་ཀྱི་སྦྱོར་བ་ནི་ཅུང་ཟད་བཤད་ནས་བསྒྱུར་ཅིང་ཞུས་་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཀྱི། །བརྟགས་པ་དང་། ད་ཆ་ཁཚོ། ཚོད་བློ་མེ་བྷ་ཝ། རབ་ཀྱི་བྱ་བ་གང་། །སྦྱོར་བ་དག་ལོ་བཞི་པར་རྟག་དབན་བསྡུར་མཆན། །ས་ཀྱི་བྱེ་བྲག་ཤེས་པར་འདོད་ན་ཡ་ལ་ལོང་། ཡིན་ན་དུས་ཏེཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་སྨྲ་སྒོ་མཚོན་ཆའི་འགྲེལ་པ་བཞུགས། ཞེས་འཁོད། ®«སྣར་བཞེན། «པེ་»«སྣར་བཙལ། «པེ་»ཚིགས། «པེ་»«སྣར་»ལྡན། @«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»བཙལ། «པེ་»«སྣར་»རིག་@«པེ་»«སྣར་བཙལ། @«པེ་»«སྣར་»བཅས་པ། ཉི«པེ་»《སྣར་»གི། ཚ«པེ་»«སྣར་རེའི། །ཚེ«པེ་»«སྣར་»ལྤགས། ཚེ«པེ་»«སྣར་རྒྱུད། ཚེ«པེ་»«སྣར་དུ། «པེ་»«སྣར་»སྤྱི། «པེ་»«སྣར་»དང༌། «སྣར་+མ། «སྣར་»ཤེས། «པེ་»«སྣར་དོག@«པེ་»«སྣར་»ལས། ༄༅«པེ་»«སྣར་ཀྱི། ཉི«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»འདི། གྱི«པེ་»«སྣར་»ཏེ། ཨེ«པེ་»«སྣར་»བསྒྲེང་བ། ཉེ«པེ་»«སྣར་»ཨི། «པེ་»«སྣར་»ནི། ཆེ«པེ་»«སྣར་»གློད། «པེ་»«སྣར་»+པ་བརྗོད། །«པེ་»«སྣར་»བས། ཚུ«པེ་»«སྣར་»དེ། «པེ་»«སྣར་»དེ་ནས། ཁ«པེ་»«སྣར་»དྲུག«སྣར་»བསྒྲིབས། «པེ་»«སྣར་»གློད་པ་དང་། «པེ་《སྣར་ལང་། ཚེ«པེ་»«སྣར་ན་བ། «པེ་»«སྣར་»དེ། «སྣར་སྒྲེའི། «པེ་»«སྣར་»ཧ་ས། «པེ་»«སྣར་»ནི། ཊི་ཨ«པེ་»«སྣར་»ཅན། «སྣར་»ༀང་། «པེ་»«སྣར་+དང༌། ཚེ«སྣར་»བཅོ། «སྣར་»ལྟར། ཆེ«པེ་»«སྣར་»གློད། «པེ་»«སྣར་»གྱི། །ཆེ«པེ་»«སྣར་ཕྱེ། «པེ་»«སྣར་བསལ། «པེ་»«སྣར་»གློད། «པེ་»«སྣར་»གྱི། ཉི«པེ་»«སྣར་བསལ། «པེ་»«སྣར་»ཚེ་ཇི་་།
