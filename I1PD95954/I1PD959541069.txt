བསྟན་འགྱུར། སྒྲ་མདོ། །ཤིའིཅན་རབ་ཏུ་གྲུབ་པ་མེད་དོ། །ཤ་མཐའ་ཅན་ལས་བི་ཤའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །ཧ་དང་ཤ་དང་ཥ་དང་ཚ་མཐའ་དང་ཡཛ་ལ་སོགས་པ་རྣམས་ཀྱིས་ཌའོ་ཞེས་པས་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ཅན་ལ་ཤ་ཡིག་ཌ་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དང་པོ་དང་གསུམ་པ་དག་གོ། །བི་ཊ། བིཌ། བི་ཤོ། བི་ཤ། བི་ཤཾ། བི་ཤོ། བི་ཤ། བི་ཤཱ། བིནྡྷམ། བིནྟིཞེས་པ་ལ་སོགས་པའོ། །ཏྲི་ཥའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །སྔ་མ་བཞིན་དུ་ཌ་ཉིད་ལ་སོགས་པའོ། །དྲིཊ། དྭིཌ། དྭི་ཥོ། དྭི་ཥ། དྲི་ཎཾ། དྲི་ཌོ། དྭི་ཥ། དྭི་ཥཱ། དྲི་ཟླུམ། དྭི་ཥཱིཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ༑ཧེ་དྲིཌ། ཧེ་ཧྲི་ཏོ། ཧེ་དཱི་ཥཉྩདེ་བཞིན་དུ་ཥ་མཐའ་ཅན་གཞན་རྣམས་ཀྱང་ངོ༌། །ས་མཐའ་ཅན་བི་དྷསའི་སྒྲ་སྟེ་ས་ལ་སོགས་པ་རྣམས་སོ། །ཌ། ཨནྡྷ་དང་ཨས་མཐའི་བྱིངས་མ་ཡིན་པའི་འང་སི་ལའོ་ཞེས་པས་ཨ་པ་དྷཱ་རིང་པོའོ། །གསལ་བྱེད་ལས་ཀྱང་ཀྱང་ངོ་ཞེས་པས་སི་དབྱིའོ། ་ཀ། རྫོགས་པ་དང་གསལ་བྱེད་དག་ལས་ཡིག་གི་རྣམ་པར་གཅད་བྱའོ། །བེ་དྷ། བེ་དྷ་སོ། བེ་དྷ་ས། བེ་དྷ་སཾ། བེ་དྷ་སོ། བེ་དྷ་ས། བེ་དྷ་སཱ། རྣམ་པར་གཅད་བྱར་བྱས་པ་ན། ་ཨ་དང་སྒྲ་ལྡན་དག་གིའང་ངོ་ཞེས་པས་རྣམ་པར་གཅད་བྱ་ཨུ་ཉིད་དོ། །ཨུའི་ཡི་གེ་ལ་ཨོའོ། །ཕ་རོལ་ཡང་དབྱིའོ། །བེ་དོ་བྷུམ། བེ་དྷ་བྷི། བེ་དྷ་ྋསེ། བེ་དྷོ་བྷཱུམ། །བེ་དྷོ་བྷྱ། བེ་དྷ་ས། བེ་དྷོ་བྷཱུམ། བེ་དྷོ་བྷུ། བེ་དྷ་ས། བེ་དྷ་སོ། བེ་དྷ་ཧཱུཾ། བི་དྷ་སི། བེ་དྷ་སོརྣམ་པར་:གཅད་བྱ་ནི་ཤ་ཥ་ས་ལ་ཕ་རོལ་གྱི་གཟུགས་སུ་ཡང་ནའོ་ཞེས་པས་རྣམ་པར་བརྟག་པས་ས་ཡིག་གོ། །བི་དྷསླུ། །བེ་དྷསིུ། ཀུན་དུ་བོད་པ་ལ། སཾ་བུདྡྷི་མ་ཡིན་པ་་་ཞེས་པའི་སྤྱིར་བཏང་ལས། སཾ་བུདྡྷི་ལ་རིང་པོར་འགྱུར་བ་མ་ཡིན་ནོ། །ཧེ་བེ་།
