ལས་ཕ་རོལ་གཉིས་སུ་བསྒྱུར་བར་འགྱུར་རོ་ཞེས་པའི་ན་ཉ་གཉིས་ཀྱི་རྟེན་ཡིན་པའི་ཕྱིར་ཡང་ན་གཉིས་ཉིད་དེ། ཀ་མུཀྐ། ཀ་མུཀྐ?། ཀ་མུ་ཙོ། ཀ་མུ་ཙི། ཀ་མུ་ཙོ། ཀ་མུ་ཙོ། ཀ་མུ་ཙ། ཀ་མུ་ཙཱ། ལ་ཀ་:མུདྡྷམ། ལཀ་མུདྡྷིཥྛ། ཀ་མུ་ཙེ། །ཀ་མུཀྵུམ། འཀ་མུཀྵུ། ཀ་མུ་ཙ། །ཀ་མུདྡྷམ། ༄༅མུདྡྷ། ཀ་མུ་ཙ། ཀ་མུ་ཙེ། ཀ་མུ་ཙཱ། ཀ་མུ་ཙི། ཀ་མུ་ཙོ། ག་ཉིད་དུ་བྱས་པ་ན་སྒྲ་མེད་ལ་དང་པོའོ་ཞེས་པ་ཀ་ཡིག་གོ། །ནཱ་མི་དང་ཀ་དང་ར་ལས་ཕ་རོལ་ཞེས་པ་ལ་སོགས་པས་ཥ་ཉིད་དོ། །ཀ་མུཀྟ། ཤཱི་ཊ་ལ་གཉིས་པའང་འདོད་དོ། །དེས་ན། ཀ་མུཀླུ་ཞེས་པའང་ངོ༌། །ཀུན་དུ་བོད་པ་ལ། ཧེ་ཀ་མུག། ཧེ་ཀ་མུག། ཧེ་ཀ་མུ་ཙོ། ཧེ་ཀ་ཙི༄༅། །དེ་བཞིན་དུ་ཨམྦྷོ་མུ་ཙ། ཛ་ལ་མུ་ཙ། པ་ཡོ་མུ་ཙ་ལ་སོགས་པ་རྣམས་སོ། །ཚའི་མཐའ་ཅན་རབ་ཏུ་གྲུབ་པ་མེད་དོ། །ཛའི་མཐའ་ཅན་བྷུ་བྷུ་ཛའི་སྒྲ་ནི། ས་གཞི་ལ་སྤྱོད་པར་བྱེད་དོ་ཞེས་པ་ཀྭིཔའང་ངོ་ཞེས་པ་ཀྭིཔའོ། ་། ཚིག་སྡུད་པ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །སི་ལ་སོགས་པ་རྣམས་སོ། །གསལ་བྱེད་ལས་ཀྱང་ངོ་ཞེས་པ་སི་དབྱིའོ། །ཙ་སྡེ་དང་དྲི་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་འང་ངོ་ཞེས་པ་སེ་རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ་ཛ་ཡིག་གི་ག་ཡིག་གོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པའི་ཕྱོགས་ལ་ཀ་ཉིད་དོ། །བྷུ་བྷུ་ཀ། བྷུ་བྷུ་ག་། བྷུ་བྷུ་ཛཾ། བྷུ་བྷུ་ཛ། །བྷུ་བྷུ་ཛཾ། བྷུ་བྷུ་ཛོ། །བྷུ་བྷུ་ཛ། བྷུ་བྷུ་ཛཱི། །བྷུ་བྷུབྷཱམེ། བྷུ་བྷུདྡྷིམྨེ། བྷུ་བྷུ་ཛེ། བྷུ་བྷུདྡྷམ། བྷུ་བྷུཤྩ། བྷུ་བྷུ་ཛ། དྷུ་བྷུདྡྷམ། བྷུ་བྷུཤྩ། བྷུ་བྷུ་ཛ། དྷུ་བྷུ་ཛེ། ་བྷུ་བྷུ་ཛཾ། བྷུ་བྷུ་ཛི། བྷུ་བྷུ་ཛེ། བྷུ་བྷུཀླུ། །བྷུ་བྷུཀླུ་པོ། །ཀུན་དུ་བོད་པ་ལ་ཁྱད་པར་མེད་དོ། །དེ་བཞིན་དུ། ཧུ་ཏ་བྷུ་ཛ། །བ་ལི་བྷུ་ཛ། ཨཱ་ྋཁུ་བྷུ་ཛ། པ་ཎ་ཛི༑འི་རཾ་ཏཱ་ཛི་ལ་སོགས་པ་རྣམས་སོ། །བི་:ཤྭ་སྲི་ཛི། པྲ་ཛ་:སྭ་ཛ་ཨཞེས་པ་།
