དོ་དུ་ཧུ། ཨ་དོ་དྷཀ?། ཨ་དོ་རྙོག་། ཨ་དོ་དུ་ཧཱི། ཨ་དོ་དུདྡྷཾ། ཨ་དོད་དུགྷ། །ཨ་དོ་དུ་ཧཾ། ཨ་དོ་དུ་ཧྭ། ཨ་དོ་དུདྨ། །ཎ་ཧ། ཚིག་མཐར་དྷུཊ་ལའང་ན་ཧིའི་ཌྷའོ་ཞེས་པས་ཧའི་ལྷ་ཉིད་དོ། །ཨ་ན་ནཱཏ། ཨ་ནཱ་ནད། ཨ་ནཱ་ན་ཤཱི་ཏ། ཨ་ནཱ་ནདྡྷཾ། ཨ་ནཱ་ན་ཧུ། ཚིག་མཐར་སི་ལ་ད་དྷ་དག་གི་རེ་ཕའི་རང་བཞིན་དག་ཏུ་འང་ཡང་ན་འགྱུར་རོ། །ཨ་ནཱ་ན། །ཨ་ནཱ་ནཏེ། ཨ་ན་ནད། ཨ་ནཱ་ན་ཧཱིཨ་ནཱ་ནདྡྷཾ། ཨ་ནཱ་ནདྡྷ། ་ཀྲུ་ཤ་། ཚིག་མཐར་དྷུཊ་ལ་ཙྪ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཤའི་ཥ་ཉིད་དོ། །ཨ་ཙོ་ཀྲོཊི། ཨ་ཙོ་ཀྲོ་ཌ། ཨ་ཙོ་ཀྲོ་ཤཱི་ཏི། ཥ་ལས་ཊ་སྡེའོ། །ཨ་ཙོ་ཀྲུཊྚཾ། ཨ་ཙོ་ཀྲུ་ཤུ། ཨ་ཙོ་ཀྲོཊ། ཨ་ཙོ་ཀྲོཌ། །ཨ་ཙོ་ཀྲུ་ཤྲཱི། ཨ་ཙོ་ཀྲུཝཾ། ཨ་ཙོ་ཀྲུé༄༅། ཨ་ཙོ་ཀྲུ་ཤཾ། ཨ་ཙོ་ཀྲུ་ཤཱ། ་/ཨ་ཙོ་ཀྲུ། བཱ་བྷི། སྔ་མ་བཞིན་པའི་བཞི་པ་བྷ་་ཉིད་དོ། །ཨ་བཱ་བྷཱཏ། ཨ་བཱ་བྷཱད། ཨ་བཱ་བཱ་དྷཱི། ཨ་བཱ་བཱ་དྷུ། ཚིག་་མཐར་སི་ལ་ད་དྷ་དག་གི་རེ་ཕའི་རང་བཞིན་དག་ཏུ་འང་ཡང་ན་འགྱུར་རོ། །ཨ་བཱ་བྷཱ། ཨ་བཱ་བྷཱཏ། ཨ་བ་བྷཱད། ཨ་བཱ་བཱ་དྷི། ཨ་བཱ་བཱ་དྷཾ༄༅། ཨ་བཱ་བཱདྷཱ༑ཨ་བཱ་བཱ་ཛྷེ། ཨ་བཱ་བཱ་དྷཱ། ་་:ཨ་བཱ་བཱ་དྷཱ། ཨ་བཱ་བཱ་དྷཱ། སྤཛྷ། ཨ་པ་སྤཏྟ། ཨ་པསྤརྡ། ཨ་པཱམྤནྣཱ་ཏི། ཨ་པཱལྤནྣཾ། ཨ་པསྤ་དྷུ། །རའི་ར་ལ་དབྱི་ཞིང་དབྱངས་སྔ་མའི་རིང་པོའོ་ཞེས་པས་རེ་ཕའི་རྣམ་པར་བཅད་པའོ། །ཨ་པཱཡཱ། ཨ་པཱ་སྤརྟ། ཨ་པཱསྤརྡ། ཨ་པཱ་སདྡྷི། ་ཨ་པཱ་སྤནྣཾ། ཨ་པཱ་སྤཛྷ། གྲོ་དྷུ་མངོན་པར་འདོད་པ་ལའོ། །སྔ་མ་བཞིན་ཨ་བྷཱ་སའི་རི་ལ་སོགས་པ་ཨཱ་ག་མ་རྣམས་སོ། །།དང་པོ་གའི་འང་བཞི་པ་གྷ་ཉིད་དོ། །ཨ་ཛ་རཱི་གྷརྟ། ཨ་ཛ་རི་་གྷརྡ༑ཨ་ཛ་རཱི་གི་དྷཱི་ཏ། ཨ་ཛ་རི་གྷརྡ། ཨ་ཛ་རི་གྲི་དྷཱིཏ། ཨ་ཛ་གྷརྟ། ཨ་ཛ་གྷརྡ། ཨ་ཛརྦ་དྷིཏ། ཨ་ཛཱ་རཱི་:གྲི་དྷོ། །ཨ་ཛརྦ་དྡྷཾ། ཨ་ཛ་རཱི་གྲི་དྷུ། །
