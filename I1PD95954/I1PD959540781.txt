གྱུར་ཕྱོགས་ཧཱ་ར། འབད་པར་བྱེད་དོ་ཞེས། ཨ་བ་སཱ་ཡཱི། དི་ཧི་ལི་ཧི:༔ཤྲཱི་ཥི་ཤཱ་སི་བྱ་དྷ་ཏི་ཨིཎ་ཨ་བ་ཤཱ་ཨཱཏ་རྣམས་ཀྱིའོ། །དིཧ་ཉེ་བར་སོགས་པ་ལའོ། །ལིཧ་རོ་མྱང་བ་ལའོ། །ཤིཥ་འཁྱུད་པ་ལའོ། །ཤཱ་ས་སྲོག་འཛིན་པ་ལའོ། །བྱ་དྷ་་་འཐེར་བ་ལའོ། ::། ཨིཎ་འགྲོ་བ་ལའོ། །ཨཏི་སྔོན་དུའོ། །བྱིང་འགྲོ་བ་ལའོ། །ཨ་བ་སྔོན་དུའོ། །འདི་རྣམས་དང་ཨཱཏྲ་མཐའ་རྣམས་ལས་ཀྱང་ཎའི་རྐྱེན་་་འགྱུར་རོ། །དི་ཧི་ལི་ཧི་ཐིRརྣམས་ལས་ནཱ་མི་ཨུ་པ་དྷཱ་ཞེས་པས་ཀའི་རྐྱེན་ཐོབ་པ་ལའོ། །།ཨ་བ་ཤུ་ཏི་ལས་ཉེར་བསྒྱུར་ལ་ནི་ཨཱཏ་ལས་ཌའོ་ཞེས་པ་འདི་ཐོབ་པ་ལའོ༑། ལྷག་མ་རྣམས་ལ་ཡང་ཨཙ་ཐོབ་པ་ལའོ། །དེ་ཧཚེ། ལེཧ། གླེཥf་ཤཱསི། བྱཱདྷཱི། ཨཏྟཱཡ། ཨ་:བཤུ་ཡ?། །དཱཡ། དྷཡི། སོག་པར་བྱེད་དོ། །རོ་མྱོང་བར་བྱེད་དོ། །འཁྱུད་པར་བྱེད་དོ། །སྲོག་་་འཛིན་པར་བྱེད་དོ། །འཐེར་བར་བྱེད་དོ། །འགྲོ་བར་བྱེད་དོ། །ངེས་པར་་འགྲོ་བར་བྱེད་དོ། །སྟེར་བར་བྱེད་དོ། །འཛིན་པར་བྱེད་དོ་ཞེས་པ་ལ་ཎའོ། །ཉེར་བསྒྱུར་མེད་པ་ཞེས་པ་ཅི་ཞེ་ན། པྲ་དི་ཧགྲཧི་ལས་ཡང་ནའོ། །གྲཧི་ལས་ཎའི་རྐྱེན་ཡང་ན་འགྱུར་རོ། །གཱཧ༄༅། །གྲཧ*། ཕྱོགས་གཅིག་ལ་ཨཙའོ། །འདིར་ཡང་ནའི་སྒྲ་རྣམ་པར་གཞག་པ་རྣམ་པར་བཤད་པ་བརྗོད་པ་སྟེ་དེས་ན་ཆུ་ལ་སྤྱོད་པའི་དོན་ལ། གྲྭཧཱལཾ། སྣང་བྱེད་ཅན་ལ་གྲཧའོ། །ཁང་པ་ལ་ནི་ཨཀའོ༑། གྲཧི་ལས་ཁང་པའི་དོན་ལ་ཨཀའི་རྐྱེན་འགྱུར་རོ། །ཁང་པ་ནི་་་ཁང་:པའམ་གྲི་ཧཾ་འཛིན་པར་བྱེད་དོ་ཞེས་ཨཀའོ་གྲཧི་ཛུ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ༑། །གྲོཧའི་སྒྲས་ཁྱིམ་ན་གནས་པའི་ཆུང་མ་རྣམས་བརྗོད་པར་བྱ་སྟེ། དཔེར་ན་པཉྩ་ཀྲོ་ཤནྟི་ཞེས་པ་བཞིན་ནོ། ་། བཟོ་བོ་ལ་བུཥའོ། །བྱིངས་ལས་་་བཟོ་བོའི་དོན་ལ་བུཥ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །:ཀྲྀཏི་ཁ་ནུ་ཝིརཉྫི་རྣམས་ཉིད་ལས་།
