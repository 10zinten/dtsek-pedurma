དྲྀ་ཤྱ་ཏི། ཀྲྀ་ཤ་ཕྲ་མོར་བྱེད་པ་ལའོ། །ཀྲྀ་ཤུ་ཏི། ཉི་ཀྲྀ་ཥ་སྐོམ་པ་ལའོ། །ཀྲྀ་་ཏི། གལ་ཏེ་ཨ་ཡིག་གི་མཐའ་གང་ཡིན་པ་དེའི་ཚེ་ཀྲྀ་ཥི་:ཏ་ཞེས་པ་མི་་འགྱུར་རོ། །རྣམ་པ་གཞན་དུ་ཀྲྀཥཱ་ཞེས་པར་འགྱུར་རོ། །ཏུཥ། ཧྲ་ཥ་་ངོམས་པ་ལའོ། །།ཏུ་ཥྛ་ཏི། པྲྀ་ཥྱ་ཏི། ཀུ་པ། ཀྲུ་དྷ། །རུ་ཥ་ཁྲོ་བ་ལའོ། །ཀྲུ་པྱ་ཏི། ཀུདྡྷཱ་ཏི། རུ་ཥྛ་ཏི། ཌི་པ་འཕེན་པ་ལའོ། །ཌི་པྱ་ཏི། བྷཱུ་པ་གྱེན་དུ་བསྒྲེང་བ་ལའོ། །:སྐུ་པྱ་ཏི། གུ་པ་རྣམ་པར་འཁྲུགས་པ་ལའོ། །གུ་པྱ་ཏི། ཡུ་པ། ལྟེ་རུ་པ། ལུ་པ་རྣམ་པར་རྨོངས་པ་ལའོ། ་། ཡུ་བྱ་ཏི། ཌ་རུ་པྱ་ཏི༑ཟླལུ་པྱ་ཏི། ལུ་བྷ་ཆགས་པ་ལའོ། །ལུ་བྱཱ་ཏི། གླུ་བྷ་ཀུན་དུ་གསལ་བར་བྱེད་པ་ལའོ། །ཀླུ་བྷ་ཏི། །།ཎ་བྷ། ཏུ་བྷ་འཚེ་བ་ལའོ། །ན་བྱ་ཏི། ཏུ་བྱཱ་ཏི། ཀླུ་དྷུ་གཤེར་བར་གྱུར་པ་ལའོ། །ཀླུ་ཅུ་ཏི། ཉི་མི་དཱ་གཤེར་བ་ལའོ༑་། མེ་དིའིའོ་ཞེས་པས་ཡོན་ཏན་ནོ། །མེ་བྱ་ཏི། ཉི་ཎི་དཱ་གྲོལ་བ་ལའོ། །སཱི་ཏྱ་ཏི། མ་རཱི་ཙེ་འཕེལ་བ་ལའོ། །རི་རྒྱ་ཏི། ་གྲི་དུ་མངོན་པར་འདོད་པ་ལའོ། ཨ། །གྲི་དྨ་ཏི། པ་རསྨིའི་ཚིག་རྣམས་སོ། །སྐྱུང་སྲོག་®ཆགས་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ང་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །།སཱུ་ཡ་ཏེ། ་སཱུ་ཡེ་ཏེ༑། སཱུ་ཡན་ཏེ། སཱུ་ཡ་སེ། །སཱུ་ཡ་ཐེ། སཱུ་ཡ་རྣོ། །སཱུ་ཡེ། སཱུ་ཡཱ་བ་ཧེ། སཱུ་ཡཱ་མ་ཧེ། དྲུང་ཡོངས་སུ་གདུང་བ་ལའོ། །དུ་ཡ་ཏེ། དཱིང་ཟད་པ་ལའོ། །དཱི་ཡ་ཏེ། རྙིང་མ་གུས་པ་ལའོ། །དྷཱི་ཡ་ཏེ། སྙིང་འཚེ་བ་ལའོ། །མཱི་ཡ་ཏི། པྲ་མཱི་ཡ་ཏི། རཱིང་ཉན་པ་ལའོ། །རཱ་ཡ་ཏེ། ལཱིང་སྦྱོར་བ་ལའོ། །ལཱི་ཡ་ཏེ། བཱི་ལཱི་ཡ་ཏེ། ་ཎིང་འགྲོ་བ་ལའོ། །།ཌི་ཡ་ཏེ། ་ཨུ་ཌཱ་ཡ་ཏེ། བྲིང་སྒྲིབ་པ་་་དཱ་མཱ་གཱ་:ཡ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཡན་ལ་ཨཱི་ཉིད་མ་ཡིན་ཏེ། དེ་ལ་རུས་ཀྱི་།
