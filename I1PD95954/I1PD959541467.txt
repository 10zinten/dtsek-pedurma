བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་ཏ་ཐ་དག་གི་གནས་ལ་དྷ་ཡིག་གོ༑། དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་བྷ་ཡིག་གི་བ་ཡིག་གོ། །ཨ་ལv། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་བྷ་ཡིག་གི་པ་ཡིག་གོ། །ཨ་ལཔ་ཏྭཾ། ཨ་ལམྨ་ཏ། ཨ་ལ་ྗ། ཨ་ལངྐཱ་ཐཱ། ཨ་ལདྡྷཾ། ཨ་ལཔི། ཨ་ལསྨཱ་ཧི། ཨ་ལཙྪ་ཧི། ྋར་བྷ། ཨཱ་རདྡྷ། ཨཱ་རཔ་ཏཱཾ། ཨཱ་རམྥ་ཏ། ཧ་ད་བཤང་བ་འདོར་བ་ལའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་ད་ཡིག་གི་ཏ་ཡིག་གོ། །།ཨ་ཧཏྟ། ཨ་ཧཏཱ་ཏཱ། ཨ་ཧཏྠཏ། སཱ་ཀི་རྒྱུན་ཆགས་པའི་བྱིངས་ཏེ་འགྲོ་བའི་དོན་ལའོ། །ཨ་ཧཱུཾ་ལྟ། ཨ་ཧཱུཾ་ཀཱ་ཏཱ། ཨ་མྲྀ་ཀ་ཏ། ར་མ། མ་ན་དག་གི་དྷཊ་ལ་རྗེས་་་སུ་ང་རོའོ་ཞེས་པས་མ་ཡིག་གི་རྗེས་སུ་ང་རོའོ། །།ཨ་རཾ་སྟ། ཨ་རཾ་སཱ་ཏྭཾ་ཨ་རཾ་ས་ཏ། བྱ་རཾ་སཱིཏ། ཨཱ་རཾ་སཱིཏ། པརྨ་རཾ་སཱིཏ། ་མ་ན་ཤེས་པ་ལའོ། །ཨ་མཾ་སྟ། ཨ་མཾ་སཱ་ཏཱ། ཨ་མཾ་ས་ཏ། ཡ་ཛ། དྷུཊ་ལས་ཀྱང་དྷཊ་ལའོ་ཞེས་པས་སི་ཙ་དབྱིའོ། །བྷཾ་ཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥའོ། ?། ཨ་《ཡཱཀྵིཏ། །་ཌ་ཨ་ཡཱཏྟཾ། ཨ་ཡཱཀླུ༄༅། ཨ་ཡཱཀཱི། ཨ་ཡཱཊཾ༄༅། ཨ་ཡཱཏྟ། ཨ་ཡཱཏྟཾ། ཨ་ཡཱཀྵ། ཨ་ཡཱཀྵ། ཨ་ཡཉྫ། ཨ་ཡཀྵ་ཧཱུཾ། ཨ་ཡཀྵ་ཏ། ཨ་ཡཏྟཱི། ཨ་ཡཀྵ་ཧཱུཾ། དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་ཥའི་ཌའོ། །ཨ་ཡཎྜིལ། ཨ་ཡཀྴི༑ཨ་ཡཀྵ་ཧི། ཨ་ཡཀྵ་ཧི། ཌུ་པ་ཙཥ། ཙ་སྡེ་འདྲ་བའི་ཡི་གེ་མ་་ཡིན་པ་ལ་ཀིའོ། །ཨ་པཱཀྵི་ཏ། ་ཨ་པཱཏྟཾ། ཨ་པཱཀྵ། ཨ་པཀྟ། ཨ་པཀྵ་ཧཱུཾ། ཨ་པགྷཱཏ། རཉྫ། ཨ་རཱ་གཱིཏ། ཨ་རཾ་ཀྲྀཏེ། ཨ་རཱི་གླུ། ཁྱད་པར་ནཱ་མི་གཟུང་བ་རྗེས་སུ་འཇུག་སྟེ། ནཱ་མི་སྨོས་པ་ཉིད་ཀྱིས། སི་ཙ་ཨཱ་ཤཱིཥེདག་་ལའང་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་ལ་ཨ་ནུ་ཎཾ་ག་དབྱི་བ་མེད་དོ། །—།
