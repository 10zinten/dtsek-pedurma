ཨེ་ཀཱ་ཨཱི་སཔྟ་ཏི་དྲུག་ཅུ་རེ་དགུ་དང༌། ཨེ་ཀཱནྣ་སཔྟ་ཏི་དྲུག་ཅུ་རེ་དགུ་དང༌། ཨེ་ཀོ་་་ནསཔྟ་ཏི། ༼རེ་དགུ། །ཨུ་ན་སཔྟ་ཏི། ་རེ་དགུ། སཔྟ་ཏི། བདུན་ཅུ། ཨེ་ཀ་སཔྟ་ཏི། བདུན་ཅུ་དོན་གཅིག། དྲི་སཔྟ་ཏི།  ོ་དོན་གཉིས། ཏྲི་སཔྟ་ཏི། །དོན་གསུམ། ཙ་ཏུསཥ་ཏི། དོན་བཞི། པཉྩ་སཔྟ་ཏི། དོན་ལྔ། ཥཊ་ཏི། ་༼དོན་དྲུག། སལྟ་སཔྟ་ཏི། དོན་བདུན། ཨཊྚ་སཔྟ་ཏི། དོན་བརྒྱད། ཨཥྛ་སཔྟ་ཏི། ཨེ་ཀཱ་ཨཱི་ཤཱི་ཏི་བདུན་ཅུ་དོན་དགུ་དང༌། ཨེ་ཀོནྣ་ཤི་ཏི་དོན་དགུ་དང་། སྤུ་ནཱ་ཤཱི་ཏི། དོན་དགུ། ཨ་ཤཱི་ཏི། བརྒྱད་ཅུ། ཨེ་ཀཱ་ཤཱི་ཏི། གྱ་གཅིག། གཉིས་པ་ལ་ཨཱ་ཉིད་མི་འགྱུར་རོ། །དྨ་ཤཱི་ཏི། གྱ་གཉིས། ཏྲ་ཡཱ་ཤཱི་ཏི། གྱ་གསུམ། ཙ་ཏུ་ར་ཤཱི་ཏི། གྱ་བཞི། པཉྩ་ཤཱི་ཏི། གྱ་ལྔ། ཥཌ་ཤཱི་ཏི། གྱ་དྲུག། སམྦྷ་ཤཱི་ཏི། གྱ་་བདུན། ་ཨ་ཉིད་རྣམ་པར་བརྟག་པ་ལ་ཡང་ངོ༌། ཨཏྟཱ་ཤཱི་ཏི། ་གྱ་བརྒྱད། ཨེ་ཀཱ་ཨཱི་ན་ཝ་ཏི་གྱ་དགུ་དང༌། །eཨེ་ཀཱནྡ་ཝ་ཏི་གྱ་དགུ་དང་། །ཌཨེ་ཀོནྣ་ཝ་ཏི་གྱ་དགུ་དང༌། ཨུནྣན་ཝ་ཏི། གྱ་དགུ། ན་ཝ་ཏི། དགུ་བཅུ། ཨེ་ཀ་ན་ཝ་ཏི། དགུ་བཅུ་གོ་གཅིག། སྔ་མ་བཞིན་དུ་ཨཱ་ཉིད་དང་། ཏྲ་ཡ་གསུམ་པ་ལ་ཡ་རུ་བསྒྱུར་བ་རྣམ་པར་བརྟག་པས་སོ། །དྲི་ན་ཝ་ཏི། གོ་གཉིས། དྲྭ་ན་ཝ་ཏི། ཏྲི་ན་ཝ་ཏི། གོ་གསུམ། །ཏྲ་ཡོ་ན་ཝ་ཏི། ཙ་ཏུན་ཝ་ཏི། གོ་བཞི། པརྩྭ་ན་ཝ་ཏི། གོ་ལྔ། ཥ་ཡིག་ནི་ཌ་ཉིད་དེ། །ཏ་དང༌། ཌ་དང༌། །་ཎ་ཕ་རོལ་ལ་ནི་ཎ་ཡིག་གོ་ཞེས་པས་ནའི་ཎ་ཉིད་དང་། །ལྔ་པ་ལ་ལྔ་པ་རྣམས་སུ་སྟེ། ་ཡང་ན་མ་ཡིན་པར་གསུམ་པ་རྣམས་སུའོ་ཞེས་རྣམ་པར་གཞག་པ་ལ་རྣམ་པར་བཤད་པས་རྟག་ཏུ་ལྔ་པའོ། །ཥཎྜ་ཝ་ཏི། གོ་དྲུག། སལྟ་ན་ཝ་ཏི། གོ་བདུན། ཨཥཱ་ན་ཝ་ཏི། གོ་བརྒྱད། ཨེ་ཀཱ་ཨཱི་ཤ་ཏཾ་གོ་དགུ་དང༌། ::ཨེ་ཀཱནྣ་ཤ་ཏཾ་གོ་དགུ། དང༌། ༼ཨེ་ཀོནྣ་ཤ་ཏཾ་གོ་དགུ་དང༌། སྤུ་ན་ཤ་ཏཾ། གོ་དགུ། ཤ་ཏཾ་བརྒྱ་ལ་སོགས་་པའང་ངོ་ཞེས་པས་མ་ནིང་ལ་ཁྱད་པར་བའི་རྟགས་སོ། །གཅིག་ཚིག་གི་ཡུལ་ལ་་ཡང་། །བྱེ་བ་ཀོ་ཎི་ཡངས་པ་རྣམས་ཏེ། བྱེ་བའི་སྒྲ་ནི་བུད་མེད་ཀྱི་རྟགས་སོ། །།
