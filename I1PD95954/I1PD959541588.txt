ཏི་ལ་སོགས་པའི་མཐའི་བྱ་བ་རྣམ་པར་དཔྱད་པ་ཞེས་བྱ་བ། །དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་གཉིས་བརྗོད་དོ། །ཨ་ཙི་ཀཱི་ཥ་ཏི། ཨ་ཙི་ཀླུ་ཏི། བྲཤྩབྲི་བྲཤྲཱི་ཥ་ཏི། བྲཤྩ་མསྔ་དག་གི་དྷུཊ་ལའོ་ཞེས་པས་མཐའ་དབྱིའོ༑་། བྷཱཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །བྲི་བྲཀྵ་ཏི། ཏྲཾ་ཧཱུ། །སྡེ་ཧུ༑འཚེ་བའི་དོན་ལའོ། །ཏི་ཏནི་ཥ་ཏི། ཧའི་ཆ་འོ། ་། ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཏི་ཀྲྀཀྵ་ཏི། ཏི་སྟརྟི་ཥ་ཏི། ཏི་པཱིཀ་ཏི། དྲྀ་ཧཱིུ། བི་བདྷི་ཥ་ཏི། བི་བྲྀཀྵ་ཏི། གཱ་ཧུ། ཛི་གཱ་ཧི་ཥ་ཏི། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་པས་དང་པོ་གའི་བཞི་པ་གྷ་ཉིད་དོ། »། ཛི་གཀྵ་ཏེ། །པ་ཏི་ནིཥྛ་ཥ་དག་མ་ཡིན་ཞེས་པའི་ཤེས་པར་བྱེད་པ་ལས་ནིར་སྔོན་དུའོ། །ཀུཥ་ལ་སྔ་མ་བཞིན་ཡང་ན་ཨིཊའོ། །།ཨིཊ་དང་བཅས་པའི་སན་ལ་ཡང་ན་ཡོན་ཏན་ཉིད་དོ། །ནིཤྩ་ཀོ་ཏི་ཥ་ཏི། :ནིཤྩ་ཀུཀླུ་ཏི། དྲྀ་ཏུ། ཨེཊ་ཨཱ་ག་མའོ། །བི་བརྟི་ཥ་ཏེ། ཐཾ་དྷུ། བི་བརྙི་ཥ་ཏེ།  ིབི་དྷུ། ཤི་ཤདྷི་ཥ་ཏེ། སྱཱནྟཱ། ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་ཨིཊའོ། །སི་པནྡི་ཥ་ཏེ། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །སི་སྱཱཏྨ་ཏེ། ཀྲྀ་པཱུ། ཙི་ཀལྤ་ཥ་ཏེ། ར་ཐོས་པ་ལས་ལ་ཐོས་ཏེ། རཾ་ཡིག་གི་ལ་་་་ཡིག་འཇིག་རྟེན་པའི་ཉེ་བར་སྤྱོད་པ་ལས་སོ། །དེའི་ཕྱིར་རཾ་ཡིག་གི་ཨ་ཡིག་གོ་་་ཞེས་པས་ཨ་ཧྥུ་སའི་ལི་ཡིག་གི་ཨ་ཡིག་གོ། །ཙི་ཀླདྨ་ཏེ། ཨཏྨ་ནེ་མ་ཡིན་པའི་ཚིག་མཐའ་ལ་ནི་བྲིཏ་ལ་སོགས་པ་ལས་ཨིཊ་སྐྱ་དང་སན་ལ་མ་ཡིན་ནོ། །བྲི་ཏུ། བྲི་དྷུ་བི་ཀྲྀཏྠ། ཏི། ་ཤྲདྷུ། ::：ཤི་ཐཾ། ཤི་ཐིཏྠ་ཏི། སྱཱནྟུ། སི་སུམྦྷ་ཏི། ཏོཀྲྀ་པུ། ཤཱསྟ་ནཱི་ལ་ཡང་སྲིད་པ་ལས་མ་ཡིན་པ་ཁོ་ནའོ་ཞེས་པས་ཨིཊ་མེད་དོ༑་། ཙི་ཀྑཾ་པ་ཏི། ཀྲྀ་ཏཱི་བསྐོར་བ་ལའོ། །ཙཱི་ཏཱི་འཚེ་བ་ལ་ཡང་ངོ༌། །ཨུཙྪི་དིར། ཨུ་ཏྲི་དིར། ཀྲྀ་ཏཱི། ཀྲྀཏ་ལ་སོགས་པ་ལས་ས་ལ་ཡང་ན་བ་།
