ཏི་ལ་སོགས་པའི་མཐའི་བྱ་བ་རྣམ་པར་དཔྱད་པ་ཞེས་བྱ་བ། །ཎི་ཛིར། ཙ་སྡེའི་ཀིའོ། །ནི་ནིz་ཏི། ནི་ནིཀྵ་ཏེ། བི་ཛིར། །བི་བིཀ་ཏི། བི་བིཀྵ་ཏེ། ཥཱཉྫ། སི་ཏྟཾ་ཀྵ་ཏེ། བ་རི་?#་ཀྵ་ཏེ། ཏུད། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཏུ་ཏུཏྠ་ཏི། ཏུ་ཏུཏྠ་ཏེ། ཁྱེད། །ནུ་ནུཏྠ་ཏི། ནུ་ནུཏྠ་ཏེ། ཀླུ་དིར། ཙུ་ཀླུཏྠ་ཏི། ཙུ་ཀླུཏྠ་ཏེ། ཎི་དཱ། །སི་སཱིཏྠ་ཏི། སྦེ་ད་ཏིའི་ནི། ལསི་སཱི་དི་ཥ་ཏེ། བིད་ཡོད་པ་ལའོ། །བི་བློ་ཐོབ་པ་ལའོ། །བིད་རྣམ་པར་དཔྱོད་པ་ལའོ། །བི་བིཏྠ་ཏི། བི་བིཏྠ་ཏི། ཙྪི་་དིར། ་ཙི་ཙྪིདྨ'ཏི། ཙི་ཙྪིཏྠ་ཏེ། །བྷི་དིར། བི་བྷིཏྠ་ཏི། ་བི་བྷིཏྠ་ཏེ། ཧ་ད། ཛིཧཏྠ་ཏི། ་ཤ་བློ། རྒྱུ་མཚན་མེད་པ་ལས་ཨཱཏྨ་ནེ་མ་ཡིན་པའི་ཚིག་གོ༑ཁོ་། ན་ཧི་ཤ་དི་ཉིད་ཨཱཏྨ་ནེའི་ཚིག་གི་རྒྱུ་མཚན་ལ་བརྗོད་པ་འོན་ཀྱང་ཤི་་་་ཤཏྠ་ཏི། ཥ་བློ། 《སི་སཏྠ་ཏི། ས་དིའི་པྲ་ཏི་མ་ཡིན་པའི་ཞེས་པས་ཥ་ཉིད་མ་ཡིན་ནོ། །ཨ་བྷི་སི་སཏྠ་ཏི། སྐནྡིར། །ཙི་ཎཾ་ཏྠ་ཏི། ཁི་ད་ཡོངས་སུ་་འཇོམས་པ་ལའོ། ?། ཙི་ཁཏྠ་ཝེཏི། ཁི་ད་ཞེན་པ་ཉིད་ལའོ། །ཙི་ཁཏྠ་ཏེ། རཱ་དྷཱ། རི་རཱཏྨ་ཏི། རུ་དྷིར། རུ་རུཏྠ་ཏི། རུ་རུཏྠ་ཏེ། ཀྲུདྷ། :ཙུ་ཀྲུཏྠ་ཏི༑ཙུ་ཀུཏྠ་ཏེ། ཀླུ་དྷ། ཙུ་ཀླུཏྠ་ཏི། །ཙུ་ཀླུཏྠ་ཏེ། བནྡྷ། དང་པོ་བཞི་པ་ཉིད་དོ་ཞེས་པས་བའི་བྷ་ཉིད་དོ། །བི་བྷཾཏཱ་ཏི། ་ཤུ་དྷ། ཤུ་ཤུཏྠ་ཏི། ཎི་དྷུ་ཡང་དག་པར་གྲུབ་པ་ལའོ། །སི་སིཏྠ་ཏི། ཎི་དྷུ་རྟོགས་པ་ལའོ་ཞེས་པ་འདིའི༑སི་སེ་དྷི་ཥ་ཏི། ས་སེ་དྷི་ཥ་ཏེ། ནི་ཏི་ཥ་དྷི་ཥ་ཏི། ནི་ཊི་ཥ་དྷི་་ཥ་ཏེ། བུ་དྷ་ཁོང་དུ་ཆུད་པ་ལའོ། །།དིབ་སོགས་སོ། །དབྱིངས་ཀྱི་དང་པོ་བའི་བཞི་པ་བྷ་ཉིད་དོ། །བུ་བྷཏྠ་ཏེ། བྷུ་སོགས་ཀྱི་ནི། བུ་བོ་དྷི་ཥ་ཏི། བུ་བོ་་དྷི་ཥ་ཏེ། བུ་དྷིར། སླར་ཡང་ཡཛ་སོགས་སོ། །ཡུ་དྷ། ཡུ་ཡུཏྠ་ཏི། བྱ་དྷ། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་ཞེས་པ་ལ་སོགས་པས་བ་ཡིག་དང་པོའི་བཞི་པ་བྷ་།
