བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་ཕྱོགས་གཅིག་ལ་ཨནའོ། །ཙུ་ཅུ་ཏི། ཙུ་ཎ་ཏི། ཙུ་ཏྱཱ་ཏི་:ཙུ་ཊ་ཏི། ཚུ་ཏྱཱ་ཏི། ཚུ་ཊ་ཏི། ཙཱུརྱ་ཏི། :ཙཱུ་ར་ཝེཏི། ཏུ་རྒྱུ་ཏི་ཏུ་ཊ་ཏི། སྤུ་ཌ་རྣམ་པར་རྒྱས་པ་ལའོ། །སྤུ་ཊ་ཏི། མུ་ཊ་ཀུན་དུ་འཕེན་པ་དང་རབ་ཏུ་ཉེད་པ་དག་ལའོ། །མུཊ་ཏི། ཏུ་ཌ་འཐབ་པའི་ལས་ལའོ། །ཏུ་ཊ་ཏི། ་ཛུ་ཊ་་་་འཆིང་བ་ལའོ། །ཛུ་ཌ་ཏི། ཀ་ཌ་རྒྱགས་པ་ལའོ། །ཀ་ཌ་ཏི། ཀྲྀ་ཌ་རྒོད་པ་ལའོ། ་་། ཀྲྀ་ཌ་ཏི། ་ཀུ་ཌ་བྱིས་པ་ཉིད་ལའོ། །།ཀུ་ཌ་ཏི། ་གྷུ་ཊ་སོ་སོར་་འཇོམས་པ་ལའོ། །སྒྱུ་ཊ་ཏི། ཏུ་ཌ་ཐུ་བ་ལའོ། །ཏུ་ཌ་ཏི། །ཐུ་ཌ་སྦུ་ཌ་བུ་ཌ་བྲུ་ཌ་ཀུན་ནས་སྒྲིབ་པ་ལའོ། །ཐུ་ཌ་ཏི། སྦུ་ཌ་ཏི། བུ་ཌ་ཏི། ཞོ་བྲུ་ཌ་ཏི། ཧུ་ཌ་ཚོགས་པ་ལའོ། །ཧུ་ཌ་ཏི། སྤུ་ར་འཕྲོ་བ་ལའོ། །སྤྲུལ་ཀུན་ནས་་སོགས་པ་ལའོ། །།སྤུ་ཌ་ཏི། པྲ་སྤུ་ར་ཏིཡཾ། །པ་རི་སྤུ་ར་ཏི། སྤུ་ལ་ཏི། ནི་ནིར་བི་རྣམས་ལས་སྤུ་ར་སྤུ་ལ་དག་གི་ཞེས་པས་ཡང་ན་ཥ་ཉིད་དོ། །ནི་སྤུ་ར་ཏི༑ནིསྤུ་ར་ཏི། བི་སྤུ་ར་ཏི། :ནི་སྤུ་ར་ཏི། :ནིཉྩ་ལ་ནི། ནིཤྩ་ལ་ཏི། བིཤྩ་ལ་ཏི། ཞུ་བསྟོད་པ་ལའོ། །།དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་་སོགས་པའི་ཨུབ་ཏུ་བསྒྱུར་རོ། །ནུ་བ་ཏེ། པྲ་ནུ་ལེབ་ཏི། ཀ་དྷུ་རྣམ་པར་་འདར་བར་བྱེད་པ་ལའོ། །དྷུ་ཅིབ་ཏི། གུ་བཤང་བ་འདོར་བ་ལའོ། །གུ་བ་ཏི། དྷུ་འགྲོ་བ་དང་བརྟན་པ་ཉིད་དག་ལའོ། །།དྷུ་བ་ཏི། པ་རསྨིའི་བཤད་པ་་རྣམས་སོ། །།གུ་རཱི་བརྩོན་པ་ལའོ། །རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །གུ་ར་ཏེ། །ཨུངྒཱ་ར་ཏེ། ་བྷཱ་རམ། ཨ་བ་ཅེར་ཏི། ་གོ་བ་ནི་ཡཾ། ཀུང་སྒྲ་་ལའོ༑། ང་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཨཱཏྨ་ྋནེའི་ཚིག་གོ།  ོ། ཀུ་བ་ཏེ། སྲིང་རྣམ་པར་འབད་པ་ལའོ། །ཨན་ཡོན་ཏན་མེད་པ་ལ་ཨིའོ་ཞེས་པས་རཾཏ་མཐའ་ཅི་ཨི་ཡིག་ཨཱ་ག་མའོ། །རིའི་ཡི་གེ་རའོ། །དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་་་།
