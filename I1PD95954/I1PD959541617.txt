བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིརཾ་བྷ་ཡ་ཏི། །ལཾ་བྷ་ཡ་ཏི། ཛཱ་:ཁཾ། ཛཱ་གིཏྟིའི་ཀཱ་རི་ཏ་ལའོ་ཞེས་པས་ཡོན་་་་ཏན་ནོ། །ཛཱ་ག་ར་ཡ་ཏི། གུ་ཧུ། གོ་ཧིའི་ཨུ་པ་:དྷཱའི་ཨུཏའོ། །ཧཱུཾ་ཧ་ཡ་ཏི༑དུ་ཥ། དུ་ཏིའི་ཀཱ་རི་ཏ་ལའོ་ཞེས་པས་ཨུ་པ་དྷཱའི་ཨུཏའོ། །དྷུ་ཥ་ཡ་ཏི། ཀཱ་རི་ཏ་ཞེས་པ་ནི་མིང་སྔོན་མ་ཅན་ཉིད་ལས་སེམས་ཆགས་པ་དང་བྲལ་བ་ལ་ཡང་ནའོ༑། ཙིཏྟཾ་དྷུ་ཥ་ཡ་ཏི། ཙིཏྟཾ་དོ་ཥ་ཡ་ཏི་པྲཛྙཱ་དུ་ཥ་ཡ་ཏི། པྲཛྷི་དོ་ཥ་ཡ་ཏི། གྷ་ཊ་བྱེད་པ་ལའོ། །གྷ་ཊ་སོགས་རྣམས་མ་རྗེས་སུ་འབྲེལ་པ་རྣམས་སོ། །མ་རྗེས་སུ་འབྲེལ་པ་རྣམས་ཀྱི་ཐུང་ངུའོ་ཞེས་པས་ཨུ་པ་དྷཱ་ཐུང་ངུའོ། །གྷ་ཊ་ཡ་ཏི། བི་གྷ་ཊ་ཡ་ཏི། བྱ་ཐ། ཎི་སྡུག་བསྔལ་དང་འཇིགས་པ་གཡོ་བ་དག་ལའོ། །བྱ་ཐ་ཡ་ཏི། པྲ་ཐ་རབ་ཏུ་བཤད་པ་ལའོ། །པྲ་ཐ་ཡ་ཏི། མ་ད་ཉེད་པ་ལའོ། །མ་ད་ཡ་ཏི། སྭ་:ད། སཱ་ད་ོརོ་མྱང་བ་ལའོ། །སཱ་ད་ཡ་ཏི། ཉི་ཏཱ་:རཱ། ཏཱ་རཱ་ཡ་ཏི། ཛཱ་ར། །ཛཱ་ར་ཡ་ཏི། ང་ག་ཌ་བཀྲུ་བ་ལའོ། །ག་ཌ་ཡ་ཏི། ཧེ་ཌ་བསྐོར་བ་ལའོ། །ཧེ་ཊ་ཡ་ཏི། ཏེ་གྷ་ཊ་ལ་སོགས་པར་:བཀླགས་པ་ལ་ཡོན་ཏན་མེད་དོ། །ལ་གེ་ཆགས་པ་ལའོ། །ལ་ག་ཡ་ཏི། ཀ་གེ་བརྗོད་པར་མི་བྱ་བ་ལའོ། །ཇི་སྲིད་འདིའི་དོན་བརྗོད་པར་མི་བྱ་ཞེས་པ་ནི་འདིའི་དོན་མི་སྟོན་པ་སྟེ་འདིའི་དོན་བྱ་བ་ཐུན་མོང་བའོ་ཞེས་སོ། ཀ། ཀ་ག་ཡ་ཏི། ཙ་ནའང་འཚེ་་་བའི་དོན་དང་བྱེད་པ་ལ་བརྗོད་པར་མི་བྱ་བ་ལའོ། །ཙ་ན་ཡ་ཏི། བ་ནུའང་་་བརྗོད་པར་མི་བྱ་བ་ལའོ། །ཀ་གེ་འགྱུར་བ་ལས་དོན་གཞན་མི་སྟོན་ནོ། །པྲ་་་བན་ཡ་ཏི། ཛྭ་ལ་འབར་བ་ལའོ། །ཛ་ལ་ཡ་ཏི། ཧྭ་ལ། ཧྥ་ལ་གཡོ་བར་་་བྱེད་པ་ལའོ། །པྲ་ཧྭ་ལ་ཡ་ཏི། ྋཔདྨ་ལ་ཡ་ཏི། སྨྲ་ཀློག་པར་བྱེད་པ་ལའོ། །དགའ་བའི་ཡུལ་དྲན་ནས་རྗོད་པར་བྱེད་དོ། །སྤྲོ་སེམས་པ་ལའོ། །སྨྲ་ར་ཡོ་ཏི༑ཨ་དྱ་ནུ་བི་སྨྲ་ལེར་ཡ་ཏི།  ོ། ཤཱ་ཙ་དྲན་པར་བྱེད་པ་ལའོ། །ཨ་ནུ་སྔོན་—།
