ཁ་པརབསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིaryང་ནའགྱུར་ཞིང༌། གྷཊ་ལའང་སཾ་བུདྡྷི་ན་ཡིན་པ་ལ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །པནྡྷ་ནོ། བསྲུབ་པ་དག། པནྣཱན། བསྲུབ་པ་རྣམས། པནྟྲ་ན་མ། པནྡྷ་ནོ། ་གྲུཊ་མ་ཡིན་པའི་དབྱངས་ལ་དབྱིའོ་ཞེས་པས་ཨི་ཡིག་དབྱི་ཞིང་། གསལ་བྱེད་ལའང་འདི་་རྣམས་ཀྱི་ནའོ་ཞེས་པས་ན་དབྱིའོ། །པ་ཐ། ་པ་ཐཱ། པ་ཐི་བྷཱུམ། པ་ཐི་བྷི། པ་ཐེ། པ་ཐི་བྷཱུམ། ་པ་ཐི་ཧྥུ། པ་ཐ། པ་ཐི་བྷཱུམ། པ་ཐི་ཧྥུ། པ་་་ཐ༑ྋཔ་ཐོ། ཡིཔ་ཐཱམ། པ་ཐི། པ་ཐོ། ་པ་ཐི་ཏུ། ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལའང་ངོ་ཞེས་པས་ཨེ་ཉིད་ཐོབ་མོད། ་ཕ་རོལ་ཉིད་ལས་ཨ་་ཉིད་ཁོ་ནའོ། །ཧེ་པ། །ཧེ་པནྡྷ་ནོ། ཧེ་པནྡྲ་ན། །དེ་བཞིན་དུ་མཉྫི་དང་རོ་བྷ་ཀླུའི་སྒྲ་ལ་ནི་ན་ཡིག་ཡོད་པ་ལས་ན་དབྱི་བ་མེད་དོ་ཞེས་པ་ཁྱད་པར་རོ། །ཨོཾ་ཤུའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི་ཨབྷིའི་སྒྲ་དང་འདྲ་ལ། མཚན་ཉིད་འདི་ནི་ཁྱད་པར་ཏེ། ཨོཾ་ཡིག་སྔ་མའོ་ཞེས་པས་ཨོ་ཡིག་སྔ་མའི་དབྱངས་ཉིད་སྟེ་ཨུ་་་་ཡིག་ཏུ་ཐོབ་པོ། །ཨི་ཨེཏ་ཨུ་ཨོཏ་ཛས་ལའོ་ཞེས་པས་ཨུ་ཨོཏ་དུ་འགྱུར་ལ། །ཨོ་ཨཝའོ། ཞ། ཤས་ཀྱི་ཨ་ཡིག་ཅེས་པ་ལ་སོགས་པས་ཤས་ཀྱི་ཨ་ཡིག་ཨུ་ཡིག་ཏུ་ཐོབ་པོ། །ང་སྔ་མ་དང་བཅས་པ་ཞེས་པས། སྔ་མའི་དབྱངས་ཉིད་དེ་ཨུ་ཡིག་དང་ལྷན་ཅིག་ང་ཨོ་རུ་འགྱུར་རོ། །ཨུའི་ཡི་གེ་ཝའོ་ཞེས་པས་ཝ་ཉིད་དོ། །ཨོཾ་ཤུ། ཨོཾ་ཤུ། ཨོཾ་ཤ་ཝ། ཨོཾ་ཤུམ། ཨོཾ་ཤུ། ཨོཾ་ཤུན། ཨོཾ་ཤུ་ནཱ། ཨོཾ་ཤུ་བྷཱུམ། །ཨོཾ་ཤུ་་་བྷི༑ཨོཾ་ཤ་ཝེ། ཨོཾ་ཤུ་བྷཱུམ། ཨོཾ་ཤུ་དྡྷ། ཨོཾ་ཤོ། ༀ་ཤུ་བྷཱུམ། ཨོཾ་ཤུ་་བྱཱ༑ཨོཾ་ཤོ། ཨོཾ་ཤཱ། ཨོཾ་ཤུ་ནཱམ། ཨོཾ་ཤོ། ཨོཾ་ཤྲཱི། ཨོཾ་ཤུ་ཥཱ། ཀུན་དུ་བོད་པ་ལ། ཧེ་ༀ་ཤོ། ཧེ་ༀ་ཤུ། ཧེ་ༀ་ཤ་ཝ། དེ་བཞིན་དུ་ཤཾ་བྷུ། །བི་བྷུ། །སྦྲ་ཆུ། །ཀ་རེ་ཏུ། ཀི་སེ་རུ། ཀཱ་རུ་རྣམས་དང་། ཨིཀྵ། བྷི། ཧི་ཚྭ་ཤུ། ཨོ་ཏུ། ཨཱ་ཡུ། གོ་མཱ་ཡུ། ་བཱ་ཡུ་རྣམས་དང༌། སཱ་ནུ། །
