བསྟན་འགྱུར། སྒྲ་མདོ། ཤི་ཤིཀཎ༑ཨ་ཙཱི་ཀ་ཎཏ། བྷ་ཎ། །ཨ་བི་བྷ་ཎཏེ། ཐ་ཎ་སྦྱིན་པ་ལའོ། །གྷ་ཊ་སོགས་དང་ཙུ་ར་སོགས་ལ་འང་ངོ༌། །།ཨ་ཤི་ཐ་ཎཏ། ཧེ་ཋ། ཨ་ཛཱི་ཧི་་་ཋཏ༑། ཁ་ལུམྦྷི།  ོཨ་ལུ་ལུ་པཏ། ཨ་བ་བྷཱ་ཛཏ། ཅེས་པ་ལ་སོགས་པ་ནི་་ལེགས་པར་ཏི་ལ་སོགས་པའི་མཐའི་ཕྱིར་བྱིངས་དོན་ལ་ཞེས་བྱས་པ་ལ་སླར་ཡང་རྒྱུ་ཞེས་པ་ལ་སོགས་པ་ཡིས་སོ། །ཥྛ། །གཉིས་བརྗོད་པར་བྱས་པ་ལ། །ཏིཥྛ་ཏིའི་ཨིཏའོ་ཞེས་པས་ཨུ་པ་དྷཱའི་ཨ་ཡིག་གི་ཨི་ཉིད་དོ། །ཨ་ཏིཥྛ་པཏ། གྷཱ། གཉིས་བརྗོད་པར་བྱས་པ་ལ་ཛི་གྲྭ་ཏིའི་ཡང་ནའོ་ཞེས་པས་ཨུ་པ་དྷཱའི་ཨ་ཡིག་གི་ཡང་ན་ཨི་ཉིད་དོ། །ཨ་ཛི་ཤྲི་པཏ། །ཨ་ཛི་གྲྭ་པཏ། ཤཱནྟུ། ཨ་ཤི་ཐཾ་སཏ། བྷནུ། ཨ་བི་བྷཾ་སཏ། །དྷནུ། ཨ་དི་དྷཾ་སཏ། ཨིན་དང་ཙཎ་ལ་རྀཏ་ཨུ་པ་དྷཱའི་ཕྱོགས་ལ་ཡང་ན་ཡོན་ཏན་མེད་དེ། མིང་སྔོན་མ་ཅན་གྱི་སྒྲུབ་པ་མི་རྟག་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །དྲྀ་ཏུ། ཨ་བི་ཀྲྀ་ཏཏ། ཨ་བ་བཏྟཱཏ། ཀྲྀ་པཱུ། ཨ་ཙཱི་ཤྲཱི་པཏ། ཨ་ཙ་ཀ་ལྤཏ། ཀྲྀ་ཏཱི། ་ཨ་ནི་ཚྭ་ཏཏ། ཨ་ན་ནནྟཏ། ཀྲྀ་ཥཱ། ཨ་བཱི་ཤྲཱི་ཥཏ། ་ཨ་བ་བརྙཏ། ཚྭ་:ཛུ། ཨ་མཱི། ཨ་མཱི་ཨམྲྀ་ཛཏ། ཨ་མཱ་མཎྜཏི། ཀྲྀཏ། ཡང་དག་པར་སྒྲོགས་པ་ལའོ། །ཙུ་ར་སོགས་སོ། །ཀཱི་རྟི་དང་ཨི་ཏི་དག་གི་ཏྟིའང་ངོ་ཞེས་ཤེས་པར་བྱེད་པ་མི་རྟག་པ་ཉིད་ཀྱི་ཕྱིར་རོ་ཞེས་་་པས་ཨིཎ་དང་ཙ་ཎའི་ཕྱོགས་ལ་ཡང་ན་རི་ཏའི་ཨིརའོ། །ཨ་ཙཱི་ཀྲི་ཏཏ། ཨ་ཙི་ཀཱིཏྟཏ། ཨུཎྚཉ། །དབྱངས་དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་ནུའི་སྒྲ་་གཉིས་བརྗོད་དོ། །།རེ་ཕའི་ནི་གཉིས་བརྗོད་པར་མི་བྱའོ། །སྔ་མ་བཞིན་་དབྱངས་དང་པོ་ཉིད་ལས་ཡང་བའི་རིང་པོ་མེད་དོ། །ཨོམྦྷ་ན་བཏ། ཧི། པྲ་ཛཱི་ཧ་ཡཏ། དི་བེ། ཨ་དཱི་དི་བཏ། ཥ་ཧ། ཨ་སཱི་ཥ་ྋཧཏ། སི། བྱ་ཏིའི་ཙཎ་ལ་མ་ཡིན་ནོ་ཞེས་བརྗོད་པ་ལས། པ་རི། ནི། །བི། །རྣམས་ལ།
