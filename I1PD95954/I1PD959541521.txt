བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི?པོའོ༑། ཨི་ཡིག་གི་ཡ་སྟེ། ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །བི་བྱ་ཏུ། སཾ་པྲ་སཱ་ར་ཎར་འགྱུར་བའི་ཕྱོགས་ཀྱི་ཡོན་ཏན་མེད་པ་ལ་ཡང་ཨ་བྷ་སའི་སཾ་པྲ་སཱ་ར་ཎའོ། །བྱ་བ་མེད་པ་ལ་ཡང་ན་དགོས་པ་མེད་པའི་ཕྱིར་རོ། །ཨེ་ཨ་ཡའོ། །བི་བྱ་ཡ་ཏུ། ་བི་བྱ་ཡུ། ་་:བྲི་བྱེ་ཨད་རྣམས་ཀྱི་ཐལ་ལ་རྟག་ཏུ་ཨིཊའོ། ཁོ། ཞེས་པས་རྟག་ཏུ་ཨིཊའོ། །བི་བྱ་ཡི་ཐ། :：བི་བྱཱ་ཐ། བི་བྱ་ཡ་ཐུ། བི་བྱཱ་ཡ། ཨཊ་ཕྱི་མ་ལ་ཡང་ན་འཕེལ་བ་སྔ་མ་བཞིན་ནོ། ::། བི་བྷ་ཡ། བི་བྱ་ཡ། བི་བྱ་ཡི་བ། བི་བྱ་ཡི་མ། བི་བྱེ། བི་བྷཱུ་ཐེ། བི་བྱི་དེ། བྲི་བྱ་ཡི་:ལྷེ། བི་བྱ་ཡི་དྷེ། ཧེ་ཉ། ཨ་བྷཱ་སྟའི་ཡང་ངོ༌། །ཞེས་པས་ཐམས་ཅད་དུ་སཾ་པྲ་སཱ་ར་ཎ་བྱས་པ་ལ་དེ་མཐའ་རིང་པོའོ། །ཞེས་པས་རིང་པོ་ཉིད་དང༌། །གཉིས་་་་་བརྗོད་དོ། །ཨཊ་ལ་འཕེལ་བའོ། །ཕྱི་མ་ལ་ཡང་ན་སྔ་མ་བཞིན་ནོ། །ཛུ་ཧཱ་བ༑་དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་པས་ཨུ་བའོ། །ཛུ་ཧུ་བ་ཏུ། ཛུ་ཧུ་བུ། སྔ་མ་བཞིན་ཐལ་ལ་ཡང་ན་ཨིཊའོ། །ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཨོ་ཨ་ཝའོ། །ཛུ་ཧ་བི་ཐ། ཛུ་ཧོ་ཐ། ཛུ་ཧུ་བ་ཐུ། ཛུ་ཧུ་བ། ཛུ་ཧཱ་བ། ཛུ་ཧ་བ། སྲྀ་ཚེལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །ཛུ་ཧ་བི་བ། ཛུ་ཧ་བི་མ༑ཛུ་ཧུ་བེ། ཛུ་ཧ་བི་ལྷེ། ཛུ་ཧ་བི་དྷེ། བ་ས་ངེས་པར་གནས་པ་ལའོ། །ཨུ་བ་ཏུ་བསྒྱུར་རོ། །ཤཱ་སི་བ་སི་གྷ་སི་རྣམས་ཀྱི་འང་ངོ་། །ཞེས་པས་རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །ཨུ་ཥ་ཏུ། ཨུ་ཥཱ། ཨུ་བ་སི་ཐ། ཨུ་བསྒྲ། ཨུ་ཥ་ཐུ། ཨུ་ཥ། ཨུ་བཱ་ས། ཨུ་བ་ས། ཨུ་ཨེཥི་བ། ཨུ་ཥི་མ། བ་ད་གསལ་བར་བརྗོད་པ་ལའོ། །ཨུ་བཱ་ད། ཨུ་ད་ཏུ། ཨུ་དུ། གསལ་བྱེད་དང་པོ་ལ་་ཨིཊ་ཨཱ་ག་མའོ། །ཨུ་བ་དི་ཐ། ཨུ་ད་ཐུ། ཨུ་ད། ཨུ་བཱ་ད། ཨུ་བ་ད། ཨུ་དི་བ། ཨུ་དི་མ། ཀྲ་དི། :：ཙ་ཀྲནྡ། ཙ་ཀྲནྡ་དུ། ཙ་ཀྲ་ན་དུ། །།
