བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་ཊིཙཀྲྀ་ཐ། ཙ་རཱི་ཀྲྀ་ཐ། ཙ་རི་ཀྲྀ་ཐ། ཙཀྲྀ་ཐ། ཙ་རཱི་ཀ་རཱི་མི། ཙི་རི་ཀརྨི༑་ཙ་རཱི་ཀ་རཱི་མི། ཙ་རི་ཀརྨི། །ཙཀ་རཱི་མི། ཙཀརྨི། ཙ་རི་ཀྲི་བ༑ཙ་རི་ཀྲི་བ། ཙཀྲྀ་བ། ༽ཙ་རཱི་ཀྲི་མ། ཙ་རི་ཀྲི་མ། ་ཙ་་ཀྲི་མ། དེ་བཞིན་དུ། :：ཁྲི་མྲྀ་སྲི་ལ་སོགས་པ་རྣམས་ལས་སོ། །རཾ། རཾ་ལྡན་པ་ཞེས་པ་འདིའི་རི་ཅན་འདི་དག་གི་ཡ་ཉིད་དུ་གཟུགས་མཚུངས་པ་ཉིད་དེ། ཨརྨ་རཱི་ཏི། ཨིརླནྟི། རེ་ཕ་ཅན་ལ་ཨ་ར་རཱི་ཏི། ཨ་རནྟི། ཨཊྚཾ་ཏ། ཨ་རྗེ་ཏ། ཌོ་ནང་གི་ཡན་ལག་ཉིད་ལས་རཱི་ཅན་ཞེས་པ་འདི་དག་གི་ཨིའི་ཡི་གེ་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དུ་བྱའོ། #། རིའི་ཡི་གེ་རའོ། །ཨནྣཱ་ཏི། རེ་ཕར་བྱས་པའི་བྱིངས་ཀྱི་དོན་ལ་ཡང་། །རའི་ར་ལ་དབྱི་ཞིང་དབྱངས་སྔ་མའང་རིང་པོའོ། །ཨཱ་ར་ཏི། ཨདྨ་རཱི་ཏི། ཨརྨ་རྙི། ཨ་ར་རཱི་ཏི། ཨ་ར་:ཆུ། ཨགྣི་ཐ། ཨནྣི་ཐ། ཨནྡྲི་*ཐ། ཨརྗེ་ཐ། ཨརྦ་རཱི་མི། 《ཨརླརྨི། ཨ་ར་རཱི་མི། །ཨ་:རསྨི། ཨནྡྲི་བ། །ཨཎི་བ། །ཨ་མ། ཨནྣི་མ། ཏཱཾ༑་ཏ་ཏཱི་ཞེས་གནས་པ་ལ། རཾ་ཡིག་གི་རིགས་གཞན་ཉིད་ལས་:རཾ་ས་སོགས་པ་མེད་དོ། །ཨཱ་ག་མ་མེད་པའི་རིང་པོའི་ཞེས་པས་རིང་པོའོ། །ཏཱ་:ཏ་རཱི་ཏི། ཏྲ་ོཏ་མཐའི་ཡོན་ཏན་མེད་པ་ལ་ཨིརའོ། །ནཱ་མི་ནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཏཱ་ཏཱནྟི། ཏཱ་ཏི་ར་ཏི། ཏཱ་༌༌རཱི་ཏི། ཏཱ་ཏནྟི༑་ཏཱ་ཏཱི། ཏཱ་ཏཱནི། །ཏཱ་ཏ་རཱི་མི། ཏཱ་ཏམྦྷི། ཏཱ་ཏཱིནྠ། ཏཱ་ཏཱསྨི༑ཀྲྀ། ཙ་ཀཱ་ཙེརཱི་ཏི། ཙཱ་ཀནྟི། ཙཱ་ཀཱིཏྟ། ཙཱ་ཀི་ར་ཏི། ཧཱུཾ། ཛཱ་ག་རཱི་ཏི། ཛ་གནྟི། ཛཱ་གཱནྟི། །ཛཱ་གི་ར་ཏི། ཤྲཱི། ༽ཤཱ་ཤ་རཱི་ཏི། ཤཱ་ཤཱནྟི༑ཤཱ་ཁཱིཏྟ། ཤཱ་བཱི་ར་ཏི། པྲི། 《ཨེཔཱ་པནྟི། མཆུར་གྱུར་ཨུ་པ་དྷཱའི་ཡང་ཨུརའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ལ་ཨུར་དུ་བསྒྱུར་རོ། །སྔ་མ་བཞིན་—།
