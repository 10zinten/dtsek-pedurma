ཁང་ལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་::ད་དག་གི་ནའི་ཨུ་པ་དྷའི་འོ་ཞེས་པས་རིང་པོ་ཉིད་དང་། མ་ན་དག་གི་དྷུཊ་ལ་རྗེས་སུ་ང་རོའོ་ཞེས་རྗེས་སུ་ང་རོའོ། །ཏེ་ཛཱ་སི། སླར་ཡང་། ཏེ་ཛ། ཏེ་ཛ་སཱི། ཏེ་ཛཱ་སི། ཏེ་ཛ་ས། །ས་ཡིག་ནི་རྣམ་པར་གཅད་བྱ་སྟེ། །ཨ་དང་སྒྲ་ལྡན་དག་གི་འང་ངོ་ཞེས་པས་ཨུ་ཉིད་དུ་འགྱུར་ལ། ་ཨུའི་ཡི་གེ་ལ་ཨོ་སྟེ་ཕ་རོལ་ཡང་དབྱིའོ། །ཏེ་ཛོ་ཧཱུཾ། ཏེ་ཛོ་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །།དེ་བཞིན་དུ། བཱ་སས། ལོ་མས། པ་ཡས། ཙེ་ཏས། ཤོ་ཏས། ཨོ་ཛས། །ཤི་རས། རགས། ས་རས། བ་ཤས། རེ་ཏས། ཏ་མསྲ། ས་མསྲ། སུ་མ་ནས། །མ་ནས། །བ་ཙས། ་ཨོཾ་གྷས། ཨུ་རས་ལ་སོགས་པ་རྣམས་སོ། །དེ་བཞིན་དུ་མ་ནིང་གི་རྟགས་ཅན་གཞན་རྣམས་ལ་ཡང་ཤེས་པར་བྱའོ་ཞེས་པ་མ་ནིང་གི་་རྟགས་ཅན་རྣམས་རྫོགས་སོ། །།།མང་ནའོ་དབྱེའི་འབྲོག་དཔང་བ་དང་། ਰੂਨਾਨਕ::ཀ་ར་སོགས་འདུ་ཤེས་དགའ་མ་ངོ་དེབ་ཀྱི་སུས་ཁོག་གཡོད་ནི་བུད་མེད་ཀྱི་རྟགས་ཅན་རྣམས་བརྗོད་པར་བྱ་སྟེ། །ཛ་ཡཱ་དང་། ཛ་རཱ་དང༌། ན་དཱི་དང་། ལབྷཱི་དང༌། ་ཤྲཱི་དང༌། སི་དང༌། བྷཱུ་མི་དང༌། བ་དྷུ་དང་། །བླུ་དང་། བུ་ནནྟུ་དང་། །དྷེ་ནུ་དང༌། སཱསྲི་དང་། མ་ཚིགས་རྣམས་དང་། ཝ་ཀྲ་དང༌། སྲོག་དང༌། བི་དྷུཏྲ་དང༌། དྲྀ་ཤད་དང་། ཀྲུ་དྷ་དང་། བཱ་མན་དང་། ཨས་དང༌། ཀ་ཀུབྷ་དང་། ཌ་གིར་དང་། དྷུར་དང༌། །དིཤ་དང་། པྲ་ཀྲྀ་ཥ་དང༌། སུ་མན་དང༌། ཟླ་ཨུཎྜི་ཧི་རྣམས་བུད་མེད་ལའོ། །དེ་ལ་ཛ་ཡའི་སྒྲ་རྟགས་ཀྱི་མིང་ཅན་ལ། བུད་མེད་ལ་ཨ་ལས་ཨཱའོ་ཞེས་པས་ཨཱའི་རྐྱེན་ཏེ། འོམཚུངས་པ་ལས་རིང་པོ་དང་རིང་པོ་ལས་ཕ་རོལ་་་དབྱི་བའང་ངོ༌། །ཨཱ་ཤཱངྒཱའོ་ཞེས་པས་ཤདྡྷའི་མིང་ཅན་ཏེ། ཤུདྡྷཱ་ལས་སཾ་དབྱིའོ་ཞེས་པས་སི་དབྱིའོ། །ཛ་ཡཱ། ཨོ་ཨཱིའོ་ཞེས་པས་ཨོ་ཡིག་ནི་ཨཱི་ཡིག་ཏུ་ཐོབ་ཅིང་། །
