བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི༽aཔ་ལཱ་ཡི་ཡི་ཥ་ཏེ། ཨ་ན། ཨ་ནི་ནི་ཥ་ཏེ། །དེ་ལ་ཡང་སྨོས་པ་ལས་ཨ་ནི་ཏིའི་མཐའིའང་ཞེས་པས་ཎ་ཉིད་དུ་བྱ་སྟེ་གཉིས་བརྗོད་དོ། །པྲ་ཎི་ཎི་ཥ་ཏི། ཨཱ་པ། ཨ་དྷ་སི་སི་ཥ་ཏི། ཨིཀྵ། ཨི་ཙིའི་ཥ་ཏེ། ཨཱི་ཧ། ཨཱི་ཛི་ཧི་ཥ་ཏེ༑སྤུ་ཧ། ཨུ་ཛི་ཧི་ཥ་ཏེ། ཨུཀྟ། ཌཨུ་ཙིཀཱི་ཥ་ཏི། རིཙྪ། ་ཙི་ཙྪ་ཥ་ཏེ། ཨུ་ཙི་ཙྪི་ཥ་ཏེ། རྒྱུ་མཚན་མེད་ན་རྒྱུ་མཚན་ཅན་གྱི་ཡང་མེད་དོ། །ཞེས་པ་འདི་རྟག་ཏུའོ། །ཨེ་དྷ། ཨེ་དི་དྷི་ཥ་ཏེ། ་ཨཱིར། །ཨཱི་རི་རི་ཥ་ཏེ། ཨིཊ༑ཨི་ཊི་ཊི་ཥ་ཏེ། ཨཱི་ཤ་། ཨཱི་ཤི་ཤི་ཥ་ཏེ། ཨུ་ཨཱི། ཨུ་ཡི་ཡི་ཥ་ཏེ། ན་བ་ད་ར་རྣམས་བརྩེགས་པའི་དང་པོ་རྣམས་ཡ་མིན་པ་ལ་མ་ཡིན་ནོ་ཞེས་པས་་་་བརྩེགས་པའི་དང་པོ་ན་བ་ད་ར་རྣམས་ཀྱི་གཉིས་བརྗོད་པ་མ་ཡིན་ནོ། །ཨནྡྷུ། ཨི་ཙི་ཥ་ཏི། ཨམྦྷི། སྨིང་། པུང་ཞེས་པ་ལ་སོགས་པ་ཨིཊའོ། །ཨཉྫི་ཛི་ཥ་ཏི། ཨ་ཧི་འགྲོ་བ་ལའོ། །་། ཨིཊ་ཨཱ་ག་མའོ། །ཨཎྜི་ཧི་ཥ་ཏེ། ་ཨུནྡྷ། ཨུ་མྷི་བྷི་ཥ་ཏི། ཨི་ཁི་འགྲོ་བའི་དོན་ལའོ། །པྲིཙྪི་ཁི་ཥ་ཏི། ༼ཨཱིཙྪ། ༀ་ཙི་ཙྪི་ཥ་ཏི། ཨུནཱི། ཨུནྡྲ་དི་ཥ་ཏི། ཨཱིནྡྲི། ཨིནྡི་དྷི་ཥ་ཏི། ཨུཥྚ། །ཨུདྡེ་ཛི་ཥ་ཏི། ཨཊྚ། ་ཏ་སྡེའི་ཙ་ཊའི་སྡེ་པ་སྦྱོར་བ་ཞེས་པ་ལ་སོགས་པས་ནང་གི་ཡན་་་ལག་སྔ་མའི་ད་ཡིག་གི་ཌ་ཉིད་དོ། །ཨཏ་ཀྱི་ཡི་གེ་ཉེ་བར་བསྟན་པའི་ཤུགས་ཀྱིཉིད་ལས་སོ། །ཨཎྜི་ཌི་ཥ་ཏེ། :·ཨུཛྷ། ཨུཎྜ། ཨུཎྜའི་བར་གྱི་ཨུ་པ་དྷཱའོ༄༅། །ཨུ་ཛི་ཛཱི་ཥ་ཏེ། :：ཨུཎྜི་ཌི་ཥ་ཏེ། འདིར་ནི་ཛྷའི་ཛ་ཨུ་བ་དྷཱའོ། །ཨུ་ཛི་ཛཱི་ཥཏེ༑ཨུཎྜི་''ཌི་ཥ་ཏེ། ཨཙྩ། ཨཙཱི་ཙི་ཥ་ཏེ། ཨདྷ། ཨ་རྫི་ཧི་ཥ་ཏི། རི་ཆུ། དེའི་ཡང་ཨུ་བ་དྷཱ་ནཱ་མིའི་ཡོན་ཏན་ནོ། །དེ་ནས་གཉིས་བརྗོད་དོ། །ཨནྣི་ནི་ཥ་ཏི། ཨ་ཥ་ཏེ། ་་བུ། ཕ་རོལ་ཏུ་ཥ་ཡིག་དང་ཁ་ཡིག་ཏུ་སྦྲ་ཡིག་ཕ་རོལ་ལ་ཡང་ན་འདོད་དོ། །ཨི་ཡིག་དང་རིདྷ་ཞེས་པ་ལ་སོགས་པས་ཡང་—།
