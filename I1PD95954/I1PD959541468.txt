ཨ་རཾ་ཀྵ། ཨ་རཾ་ཁཱ་ཏཱ། ཨ་རཾ་ཁཱ་ཏ། ཤ་པ། ཨ་ཤཱཔཱིཏ། ཨ་ཤཱཙཱ། ཨ་ཤཱམྦྷ། །ཨ་ཤY། ཨ་:ཤཱ་བཱ་ཊྛིཏྟཾ། ཨ་ཤབླ་ཏ། ཏུ་བ་པ། ཨ་བཱདྷཱིཏ། ཨ་བཱཙཱ། ༈ཨ་བཱམྦྷ། ཨ་བལྟ། ་ཨ་བསྨཱ་ཏཱཾ། ཌཨ་བཙྪ་ཏེ། ཎ་ཧ་ན་ཧིའི་དྷིའོ་ཞེས་པས་དྷུཊ་ལ་ཧ་ཡིག་གི་དྷ་ཡིག་གོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་དྷའི་ཏའོ། །ཨ་ནཱཧཱི་ཏ། ཏ་ཐ་དག་གི་དྷའོ༑། །ཨ་ནཱནྟཾ། ཨ་ནཱནྟུ། ཨ་ནདྡྷ། #ཨ་ནཏྟཱ་ཏཱ། ཨ་ན་ཏྠ་ཏ། ཨ་ནདྡྷི། བ་ཧ་ཧའི་ཌྷའོ་ཞེས་པས་དྷཊ་ལ་ཧ་ཡིག་གི་ཌྷ་ཡིག་གོ། །::ཥ་ལྷ་དག་གི་ས་ལ་ཀའོ། །ཨ་བཱཀྵི་ཏ། ཏ་ཐ་དག་གི་དྷའོ། །ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། །ས་ཧི་བ་ཧ་དག་གི་ཨ་ཡིག་གི་ཨོཏའོ་ཞེས་པས་ལྷ་ལ་ཌྷའི་དབྱི་སྟེ་སྔ་མའི་ཨ་ཡིག་གི་ཨོ་ཉིད་དོ། །ཨ་བོ་ལྷ། ཨ་བཀྵ་ཙིཏྟཾ། ཨ་བཀླུ་ཏ། བ་ད་གསལ་བར་བརྗོད་པ་ལའོ། །སི་ཙའོ། །ཨིཊ་ཨཱ་ག་མའོ། ་་། བ་དང་བྲ་ཛ་ར་ལ་མཐའ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་པ་རསྨིའི་ཚིག་ཕ་རོལ་ལ་ཨ་ཡིག་གི་རིང་པོའོ། །ཨ་བཱ་དཱིཏེ། ཨ་བ་དིཏྟཾ། ཨ་བཱ་དི་ྋཥཱ། །བྲ་ཛ་འགྲོ་བ་ལའོ། །ཨ་བཱ་ཛཱིཏ། ཙ་ར་འགྲོ་བའི་དོན་ལའོ། །ཨཱ་ཙཱ་རཱིཏ། འ་ཨུཏ་ལས་ལས་དང་བཅས་པའི་ཙ་ར་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨུ་ད་ཙ་རིཊྚ་ཀུ་རུམྨཾ། ཛཱ་ར། ཨ་ཛཱ་རཱིཏ། ཙ་ལ། ཨ་ཙཱ་ལཱིཏ། ཨ་ཙཱ་ལིཏྟཾ། ཨ་ཙཱ་ལི་ཥཱ། ཛྭ་ལ། །ཨ་ཛཱ་ལཱིཏ། ད་ལ་རྣམ་པར་འཇོམས་:པ་ལའོ། །ཨ་དཱ་ལཱིཏ། བ་་གྲྭ་འགྲོ་བའི་དོན་ལའོ། །ཤཱལྤ་མྱུར་དུ་འགྲོ་བ་ལའོ། །ཨ་ཡིག་མ་ཡིན་པའི་ར་ལ་མཐའི་ཞེས་པས་རིང་པོ་མེད་དོ། །གང་ཨའི་ཨུ་བ་དྷཱ་ཞེས་པ་དེ་འཇུག་གོ། །ཨ་བ་བྷིཏ། །ཨ་ཤཱལཱུ་ཏི། ་༼པ་ཟ། གསལ་བྱེད་དང་པོ་ཨིཊ་དང་བཅས་པ་རྣམས་ཀྱི་ནི། །ཨེཏ་ྋརྗེས་སུ་འབྲེལ་པ་དང་ཧ་དང་མ་དང་ཡའི་མཐའ་དང་ག་ཎ་དང་་།
