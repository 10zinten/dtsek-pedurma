བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིཏིལ་ཧའི་རྣའོ། །གྷ་ཛྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་ཏ་ཐ་དག་གི་དྷ་ཉིད་དོ། །ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། །ཌྷ་ལ་ཌྷའི་དབྱི་སྟེ་ཨུ་པ་དྷཱའི་རིང་པོ་ཡང་ངོ་ཞེས་པས་ཌྷ་དབྱི་ཞིང་ཨུ་པ་དྷཱའི་ཡང་ཇི་ལྟར་་་འབྱུང་བ་བཞིན་རིང་པོའོ། །མཚུངས་པའི་མཚུངས་པ་ཉིད་ཀྱི་རིང་པོ་ལ་སོགས་པ་ཐ་སྙད་དོ། །ལེ་རྨི། :ལཱི་ཌྷ། ལི་ཧནྟི། ཧའི་རྙའོ། སོ། ཥཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ལྷའི་ཀའོ། །།རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །ལེཀྐི། ལཱི་ཌྷ། ལཱི་ལྷ། ལི་ཉྩི། ལི་ཧཱ། ལི་ཧྥུ། ལཱི་བྷེ། :：ལི་ཧཱ་ཧེ། །ལི་ཧ་ཏེ། ལིཀྵེ། ལི་ཧཱ་ཐེ། ལི་ཞྭ། ལི་ཧེ། ལི་ཧྭ་ཧེ། ལི་ཧྥ་ཧེ། ཨུཎྜ་ཉ་ཀུན་དུ་འགེགས་པ་ལའོ། །ཉ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་གཉིས་ཀའི་ཚིག་གོ། །པ་རསྨིའི་གཅིག་བརྗོད་པའི་ཚིག་རྣམས་ལ་ཨུཎོ་ཏིའི་ཡོན་ཏན་ནོ། །།རྒྱལྟ་ནཱི་ལ་འང་ངོ་། ཞེས་པ་བརྗོད་པ་ལས་ཕྱོགས་གཅིག་ལ་གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་བྱིངས་ཐམས་ཅད་པ་ལ་ཨུཏའི་འཕེལ་བར་ཡང་ན་འགྱུར་རོ། །ཨུཎྜི་ཏི།  ོཨུཎྜི་ཏི། ཨུཎྜ་ཏཿདབྱངས་དང་པོ་ལ་ཨུ་བའོ། །ཨཱུརྞ་བནྟི། ཨུཎོ་ཏི། ཨུཎོ་ཎི། ཨུཎྜ་ཐ། ཨུཎ་ཐ། ཨཱུ་ཌོ་མི། ཨུཎ་བ། །ཨུཎ་མ། ཨུཎྜ་ཏེ། ཨུཎྚ་བཱ་ཏེ། ཨུཎྜ་བ་ཏེ། ཨུཎྚ་ཏེ། ཨུཆུ་བཱ་ཐེ། ཨུཎྱ་དྷེ། ཨུཎྜ་བེ། ཨུཎྱ་བ་ཧེ། ཨུཎྜ་མ་ཧེ། པྲ་སྔོན་དུའོ། །ཨུའི་ཡི་གེ་ལ་ཨོའོ། །ཕ་རོལ་ཡང་དབྱིའོ། །པྲཎོ་ཏི། པྲཎྜི་ཏི། ་པྲཎྜ་ཏེ། བྷུཉ་བསྟོད་:པ་ལའོ༄༅། །ས་་་ཉིད་ལ་སོགས་པའོ། །འཕེལ་བའོ། །སྟོ་ཏི། སྟུ་ཏ། །སྐུ་བནྟི། སྟོ་ཏི། སྟེ་ཐ། སྐུ་ཐ། སྟོ་མི། སྐུ་བ། སྐུ་མ། སྟུ་ཏེ། །སྐུ་བཱ་ཏེ། །སྟུ་བ་ཏེ༑སྐུ་ཥ། སྐུ་བཱ་ཐེ། སྐུ་དྷེ། སྐུ་བེ། སྡུ་བ་ཧེ། སྐུ་མ་ཧེ། །ཉེར་་བསྒྱུར་ལས་སུ་ནོ་ཏའི་ཞེས་པ་ལ་སོགས་པས་ཥ་ཉིད་དོ། །ཥ་ལས་ཊ་སྡེའོ། །—།
