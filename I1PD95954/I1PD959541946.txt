རྣམས་རང་གིར་བྱེད་པ་ནི། ཨཱ་ཀ་ལ་ཡ་ཏི་གཱཞེས་པའོ། །དམ་བཅའ་བ་ལ་འགྱུར་ཏེ། ཨཱ་ཤིཙཾ་ཙི་ཎྱ། ཨཱ་ག་མ་ཡ་ཏི་བ་དཾ། ཨ་པྲྀ་ཎ་ཏི། ཞེས་པའོ༑། གློག་པ་ལ་གྱུར་ཏེ། ཨཱསྣ་ཡ་ས་ཐ་ཏི་བྱཱ་ཀ་ར་ཎཾ། ་ཨཱ་ག་ཡ་མ་ཏི་བྱ་ཀ་ར་ཎཾ། །ཞེས་པའོ། །ཁྲོ་བ་ལ་འགྱུར་ཏེ། ལྷ་བྱིན་ཁྲོས་པ་ནི། ཨཱ་ཡ་སྟེ་དེ་བ་དཏྟ་ཨ་བ། ཞེས་པའོ། །འཚམས་ལ་འགྱུར་ཏེ། ཨ་པཱ་ཆུ་ནང་ཚུན་་་་ཆད་ཆར་བབ་ཊ་ལི་པུ་ཏཱདྲིཊོ་དེ་བཞེས་པའོ། །ཅུང་ཟད་ཀྱི་དོན་ལ་འགྱུར་ཏེ། ཅུང་ཟད་བཅོམ་པ་ནི། ︽ཨ་ཧ་ར། ཅུང་ཟད་བྱེད་པ་ནི། བ། ཨ་རྣམ་པ་ཀ་ར། ཅུང་ཟད་བྱ་བ་ནི། ཙཨཱ་རྣམ་པ་ཀྲྀ་ཏི། ཞེས་པའོ། །ནང་གི་དངོས་པོ་ལ་འགྱུར་ཏེ། །ལ་ཆུ་ཁོང་པར་འཇུག་པ་ནི། ཨཱ་པཱི་ཏ་མུ་ད་ཀཾ། ཨཱ་ཀྲནྟ་བཱ་སཞེས་པའོ༑། ཁྱབ་པ་ལ་འགྱུར་ཏེ། ཤིང་རྣམས་ཁྱབ་པར་བྱེད་པ་ནི། ཨཱ་ཀྲྀ་ནིཀྵེ་ཏཱ་ཎི། ཞེས་པའོ། །འགྲན་པ་ལ་འགྱུར་ཏེ། ཨཱ་ཧཱ་ཡ་ཏེ་མམྨོ་མཝཾ། །ཞེས་པའོ། །མངོན་དུ་ཕྱོགས་པ་ལ་འགྱུར་ཏེ། ཨཱ་འོང་གཙྪ་ཏི། ཞེས་པའོ། །སྟེང་གི་ལས་ལ་འགྱུར་ཏེ། །ཨཱ་རོ་ན་ཏེ་ཤིང་ལ་འཛེག་བྲྀཏྟཾ། :དེ་བ་ཙནམ་མཁའ་ལ་འཛེག། མཐོ་རིས་སུ་འཛེག་མ་རོ་ཧ་ཏི། ཞེས་པའོ། །ཤིན་ཏུ་བའི་དོན་ལ་འགྱུར་ཏེ། དུ་བས་ཤིན་ཏུ་གང་བ་ནི། །ཨ་དྷ་མ་ཏཾ། ་ཨཱ་ར་ཙི་མཎྜ་ལཾ་ཨཱ་དྷུ་མི་ཏ་ཤཾ་ཁཞེས་པའོ། །:སྐྱེས་པ་ལ་འགྱུར་ཏེ། བུ་སྐྱེས་བཙས་པ་ནི། ་ཨཱ་པརྣ་ས་ཏྭཱ་སྟི་ཅེས་པའོ། །འདུ་བ་ལས་འགྱུར་ཏེ། ཨཱ་ས་བ་ཨཱ་སོ་ན། །ཞེས་པའོ། །ངོ་མཚར་ལ་འགྱུར་ཏེ། "ཨཱན་ཙ། རྨོ་ཞེས་པའོ། །།དྲན་པ་ལ་འགྱུར་ཏེ། ཨཱ་བྷ་བ་ཏུ་བི་ཛྷཱ་ནཾ། དྲན་པར་བྱ། ཞེས་པའོ། །སྔགས་ཀྱི་བྱ་བ་ལ་འགྱུར་ཏེ། :ཨཱ་རབྷཱ་བ་ཤུཞེས་པའོ། །རབ་ཏུ་གནས་པ་ལ་འགྱུར་ཏེ། ཨཱ་སྱ་པ་དཾ། །ཞེས་པའོ། །ངེས་པར་བསྟན་པ་ལ་འགྱུར། ཨཱ་བྷིཞེས་པའོ། །རབ་ཏུ་གསལ་བ་ལ་འགྱུར། །
