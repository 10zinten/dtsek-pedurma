བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་ཀཨ་རྙོག། ཨ་དུ། ཨ་དུ་ཧན། ཨ་དྷོག། ཨ་དུཏྟཾ། ཨ་དུབྷ། ཨ་དོ་ཧཾ། ཨ་དུ་ཧཱ། ཨ་དུ་ཧྥ། ཨ་དུགྷ། ཨ་དུ་ཧཱ་ཏྭཾ། ཨ་དུ་ཧ་ཏ། ཨ་་དུབྷཱ༑ཨ་དུ་ཧཱ་ཤྲཱི། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་པ་དང་པོ་དའི་བཞི་པ་དྷ་ཉིད་དོ། །ཨ་དུཏྟཾ། ཨ་དུ་ཧི། ཨ་དུ་ཧྥ་ཧི༑ཨ་དུ་ཧྥ་ཧི། དི་ཧ། ཨ་དྷེཀ། ཨ་དྷེག། ཨ་དྷིདྡྷཾ། ཨ་དི་ཧན། ཨ་:དྷི་གྷེ། ཨ་དཱི་ཧཱ་ཏཱམ། ཨ་དི་ཧ་ཏ། ལིཧ། ཧའི་ཌྷའོ་ཞེས་པས་མཐའི་དྷཊ་ལའང་ཧའི་རྙའོལ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ལྷའི་དང་པོ་དང་། གསུམ་པ་ཊ་དང་ཌ་དག་གོ། །ཨ་ལེཊ། ཨ་ལེ་ཌ། ཌྷ་ལ་ཌྷའི་དབྱི་སྟེ་ཨུ་པ་དྷཱའི་རིང་པོ་ཡང་ངོ༌། །ཨ་ལཱི་ཏཱཾ། ཨ་ལི་ཧན། :：ཨ་ལཱི་ལྷ། །ཨ་ལི་ཧཱ་ཏཱ། ཨ་ལི་ཧ་ཏ། ཨུཥྞ་ཉ། རྒྱལྟ་ནཱི་ལ་ཡང་ངོ་ཞེས་པས་གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་ལ་རྟག་ཏུ་ཡོན་ཏན་ནོ། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །ཨོཎྞོཏ༑ཨོཙྪཱ་ཏྭཾ། ཨོཾནྡྷ་བན། ཨོཎྞོ། ཨོཙྪཱ་ཏྭཾ། ཨོཙྪཱཏ། ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཨོ་ཌིཨ་བའོ། ་། ཨོཎྜཱུ་པཾ། ཨོཾནྡྷཱ་བ། ཨོཙྪཱ་མ། ཨོཎྜུ་ཏ༑་ཨོཙྪཱ་བཱ་ཏྭཾ། ཨོཾནྡྷཱ་བ་ཏ། ་བྷུཉི། གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་བྱིངས་ཐམས་ཅད་པ་ལ་ཨུ་ཏའི་འཕེལ་བའོ། །ཨི། སྟོཏ། ཨ་སཱུ་ཏཱཾ། ཨ་སྐུ་བན། །ཨ་སྟོ། ཨ་སྟུ་ཏཾ། ཨསྟུ་ཏ། ཨ་སྟ་བཾ། /ཨ་སྐུ་བ། ཨ་སྐུ་མ། ཨ་སྟུ་ཏ། ཨ་སྐུ་བཱ་ཏཱ། ཨ་སྟུ་བ་ཏ། ཉེར་བསྒྱུར་ལས་སུ་ནོ་ཏིའི་ཞེས་པ་ལ་སོགས་པས་ཥ་ཉིད་དོ། །ཨ་ཧྥུ་ཊོཏེ། ཨ་བྱཱཏྟཱཏ། བ་རི་ནི། བི་རྣམས་་་ལས་སེ་བ་ཏིའི་ཞེས་པ་མཐའ་དབྱི་བ་རྣམ་པར་བཤད་པའི་ཡུལ་ལ་ཡང་སྟོ་ཏི་ཥཱ་ཏིལྔ་དག་གི་ཡང་ཞེས་པས་ཡང་ན་ཥ་ཉིད་དོ། །པདྨ་ཟློད། ་པརྨ་སྟོཏ?། པརླ་ཏ། པརྨ་སཱུཏ། དེ་བཞིན་དུ་ནི་དང་བི་དག་ལས་སོ། །བྷུཉ། བྲུ་།
