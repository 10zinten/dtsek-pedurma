གི་ར། གི་རཾ། གི་རོ། གི་ར། གི་རཱ། གིཙྩཾ། གིབྷི། ཞེས་པ་ལ་སོགས་པ་སྟེ། བདུན་པའི་མང་ཚིག་ལ་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱར་འགྱུར་བ་མ་ཡིན་ལ། །རྣམ་དབྱེའི་གསལ་བྱེད་ལ་རེ་པའི་མི་འགྱུར་རོ་ཞེས་པའི་ཚིག་ལས་སུ་པའི་ཥ་ཉིད་དོ། །གཱིནྡྷ། ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་གཱི། ཧེ་་གི་རོ། ཧེ་གི་རཱ། །e། སའི་ག་གུ་ཁོ་ལ་དང་འད་བ་མ་དྷུར་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ཏེ། འདི་ལ་ནི་ཨུར་རུ་འགྱུར་བ་ཁྱད་པར་རོ། །དྷུ། དྷུ་རོ། དྷུ་ར། །ཞེས་པ་ལ་སོགས་པ་སྟེ། དེ་་བཞིན་དུ་པུར་རོ། །བོ་། ལས་ཀྱི་བྱུང་བ། ན་ཏེ། བྱང་ཆུབ་དྷཱར་གྱི་སྒྲའི་ནི་དྭ། རྭ་རོ། དྷཱ་རཿ། ཞེས་པ་ལ་སོགས་པའོ། །།།བཱར་གྱི་སྒྲ་ནི་ཁ་ཅིག་བུད་མེད་ཀྱི་རྟགས་སུ་འདོད་དེ། དེ་ལྟར་ཡང་འཆི་མེད་ཕྲེང་བར། ཀཾ་དང་བཱར་ནི་བུད་མེད་བརྗོད་པ་དག་སྟེ། ཨཱ་པ་ནི་མང་པོ་རྣམས་ལ་བུད་མེད་དོ་ཞེས་གསུངས་སོ། །འདིའི་དོན་ནི་འདི་ཡིན་ཏེ། ཀཾ་ཞེས་པས་མ་ནིང་དང༌། བཱར་ཞེས་པ་བུད་མེད་དང་། ་ཨམི་མང་པོ་རྣམས་ལ་བུད་མེད་དོ་ཞེས་པའོ། །དེའི་ལུགས་ཀྱིས་བུད་མེད་ཀྱི་རྟགས་ཁོ་ནར་ཉེ་བར་བསྟན་ཏོ། །བཱ༑བཱ་རོ། བཱ་ར། །གཞན་དག་ནི་བཱར་གྱི་སྒྲ་མ་ནིང་དུ་འདོད་དེ། དེ་སྐད་དུ་ཡང་། འཆི་མེད་མཛོད་དུ། ཨ་པཉྩནྡྷ། ་བྷུ་མིར། བཱ་ར། བཱ་རི། ལཾ༑། ལོཀ་མ་ལཾ། ཛ་ལོ་ཞེས་གསུངས་ཏེ། z་བཱ་རི་ལ་སོགས་པ་རྣམས་ཀྱིས་ལྷན་ཅིག་སྤྱོད་པའི་ཕྱིར། བཱར་གྱི་སྒྲ་ལ་ཡང་མ་ནིང་ལས་སི་དང་ཨམ་དག་དབྱི་ཞིང་་དེ་ལ་བརྗོད་པ་ཡང་མ་ཡིན་ནོ་ཞེས་པས་སི་དང་ཨམ་དག་དབྱིའོ། །བཱ། ཨོ་ཨཱིའོ་ཞེས་པས་ཨོ་ནི་ཨཱི་རུ་ཐོབ་བོ། །།བཱ་རཱི། ཁྱད་པར་གཞན་རྣམས་མཚུངས་པའོ༑། བཱ་རཱ། བཱཧཱུྃ། བཱདྷི། ཞེས་པ་ལ་སོགས་པའོ། །།།།།
