བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི?སོགས་དང་ཀུཊ་སོགས་འང་ངོ༌། །ཀོ་ཏཱ། ཀུ་ཏཱ། ཥཱཉ། སོ་ཏཱ། སྐུཉ། སྐོ་ཏཱ། པུཉ། པོ་ཏཱ། བྷུཉ། སྟོ་ཏཱ། ཡུ་སྤེལ་བ་ལའོ། །རུ་ཆུ། ཨིཊ་ཨཱ་ག་མའོ། །།ཡ་བི་ཏཱ། ་ར་བི་ཏཱ། ན་བི་ཏཱ། སྐུ། སྐུ་ཀྲ། མཱི་དག་ལས་པ་རསྨི་ལའོ་ཞེས་པས་འདིའི་མདོས་ཨིཊའོ། །སྣ་བི་ཏཱ། ཁུགླུ། གླུ་བི་ཏཱ། སྒྲིུ་གླུ་བི་ཏཱ། ཨུདྡྷཱཉ། བྱ་སྟ་ནཱི་ལ་ཡང་ངོ༌། །།ཞེས་པས་ཡང་ཡིག་ནི་མ་བརྗོད་པ་སྡུད་པའི་དོན་ཉིད་ཀྱི་ཕྱིར་ཨིཊའོ། །ཡང་ན་ཡོན་ཏན་ནོ། །ཨུཎྜ་བི་ཏཱ། ཨུཙྪཱ་བི་ཏཱ། ཨུ་མཐའ་ལས་ཨིཊ་དང་བཅས་པའོ། །བྷུ། ་བྷ་བི་ཏཱ། ཥཱ་དེད་པ་ལའོ། །སཱ་བི་ཏཱ། ཆུ་བསྟོད་པ་ལའོ། །དྷཱ་གཡོ་བ་ལའོ། །ཀུཊ་་་སོགས་སོ། །ནུ་བི་ཏཱ། དྷུ་བི་ཏཱ། དྲུང༌། ད་བི་ཏཱ། ཨུང་སི། པ་བི་ཏཱ། ཨཱུཉ༑་ལ་བི་ཏཱ། ཀུཉ་སྒྲ་ལའོ། །::ཀ་བི་ཏཱ། དྲུཉ་འཚེ་བ་ལའོ། །དྲ་བི་ཏཱ། ལྟུང་། སཱ་ར་ཏི་སཱུ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཡང་ན་ཨིཊའོ། །སོ་ཏཱ། ས་བི་ཏཱ། བི་དྷོ་ཏཱ་ཞེས་:པ་ལའོ། །དྷཱཉ་གཡོ་བ་ལའོ། །ཞེས་པ་འདི་ཡིས་ནི། བི་དྷཱ་བི་ཏཱ། བྷུཉ་གཡོ་བ་ལའོ། །ཞེས་པ་གྲུབ་ནས་འདིར་རིག་པར་བརྗོད་དོ། །སྨྲ། རྀཏ་ལས་བྲིང་བྲོཉ་མ་ཡིན་པ་ལས་སོ་ཞེས་པས་རིཏ་མཐའ་དབྱངས་གཅིག་པ་ལས་བྲིང་བྲྀཉ་བོར་བ་ལས་ཕ་རོལ་གྱི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་ཨིཊ་མེད་དོ། །སྨཏྟཱ། དྷཱི། དྷཱཏྟཱ། པྲ། པཏྟཱ། ཨོ། སྲྀ། ་༼སཏྟཱ། རཾ། ཨཏྟཱ༑དྲྀཉ། དྷཏྟཱ། སྲིང༌། བྱཱ་བ་ཏྟཱ། ཧཱཉ། ་ཧནྟཱ། ་རུ་ཤྲཱིཉ། བྷཏྟཱ༑སྔོཉ་ཀུན་དུ་སྒྲིབ་པ་ལའོ། །སྟནྟཱ། ཆུ་ཀྲྀཉ། ཀཏྟཱ། ཛཱ་ཧཱུྃ། ཨིཊ་ཨཱ་ག་མའོ། །ཛཱ་ག་རཾ་ཏཱ། བྲིང་དྲྀཉ། རྀཏ་ཕྲེང་པྲཉ་ལས་ཀྱང་ཡང་ན་རིང་པོ་སྟེ་པ་རོཀྵ་དང་ཨཱ་ཤཱིཥ་དག་མ་ཡིན་པ་ལ་ཨིཊའོ། །།པ་རསྨིའི་ཋསི་ཙ་ལ་་་རབ་ཏུ་བརྗོད་པ་མ་ཡིན་ནོ། །ཞེས་སྦྱོར་བ་རྣམ་པར་ཕྱེ་བ་ལས་ཨིཊ་ཡང་ན་རིང་།
