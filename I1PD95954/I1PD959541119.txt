བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཡཙྩ༑གཉི་ག་དག་གི་མ་ཉིད་མེད་པ་ན། ཨ་ད་དྨཅུའོ། །དེ་བཤད་པ། འགའ་ཞིག་ཕ་རོལ་འདོད་པར་བྱེད་དོ། །འགའ་ཞིག་སྔ་མ་འདོད་པར་བྱེད་དོ། །ཁ་ཅིག་གཉི་ག་འདོད་པར་བྱེད་དོ། །ཁ་ཅིག་གཉི་ག་དག་གི་མི་འདོད་དོ། །བཞི་པོ་འདི་རྣམས་ཀྱང་ཨཅུའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱའོ། །ཨ་ད་མུ་ཡམ། ཨ་ད་མུ་ཡཉྙོ། ཨ་ད་མུ་ཡཙྩ®། ་ཨ་ད་མུ་ཡནྟཾལ། 《་ཨ་ད་མུ་ཡརྩོ། །སྔ་མ་བཞིན་དུ་ཨ་དབྱི་བ་དང༌། །རིང་པོ་ཉིད་དག་གོ་ཨ་ད་མུ་ཨཱ་ཙ། །ཁ་ཅིག་ནི་ཝ་ཉིད་འདོད་པར་བྱེད་དོ། །ཨ་ད་མཱི་ཙི། ཨ་ད་མུ་ཨཱི་ཙ། །ཨ་ད་མཱི་ཙ་ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། ཨ་ད་མུ་ཨཱི་ཙཱ། ཨ་ད་མཱི་ཙཱ། ་་མ་་་ནིང་ལ། །ཨ་ད་མུ་ཡཀ། ཨ་ད་མུ་ཡག། ཨ་ད་མུ་ཨཱི་ཙཱ་ཨ་ད་མུ་ཡརྩི་་་ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། །ཞེཨ་:མཱུ་དྨསྩི། ་་ཨ་མུ་མུ་ཡནྟུ། ཨ་ད་:དྨརྩའི་སྒྲ་རྣམས་རྟོགས་པར་བྱའོ། །ཨི་དམ་ནི། ཨི་ད་དྲུརྩཏེ། ཀི་མ་ནི༑ཀ་དཱུནྱི། །ས་ཧ་ཨཅུ་ཏི། སཾ་ཨརྩ་*ཏི། ཏི་རས་ཨཅུ་ཏི་ཞེས་་་པས་སྔ་མ་བཞིན་དུ་ཀྭིཔ་ལ་སོགས་པའོ། །ས་ཧ་དང་སཾ་དང་ཏི་རས་རྣམས་ཀྱི་ས་དྷི་དང་ས་མི་དང་ཏི་རི་རྣམས་སོ་ཞེས་པས་ས་ཧ་དང་སཾ་དང་ཏི་རས་ཞེས་པ་འདི་རྣམས་ཀྱི་གྲངས་ཇི་ལྟ་བ་བཞིན་དུ་ས་དྷི་དང༌། ས་མི་དང་ཏི་རི་ཞེས་པ་འདི་་རྣམས་སུ་བསྒྱུར་བར་འགྱུར་རོ། །ས་:ཏནྟུ། ས་མྱཙྩི། །གསུམ་པོ་འདི་རྣམས་པཱཉྩའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱའོ། ཨ། :ས་དྷྱང་། 'ས་དྷུཙྩོལ། ས་ཏྱཱཉྩ། ས་ཏྱཱཉྙཾ། །ས་ཏྱཱཉྙོ། སྔ་མ་བཞིན་དུ་ཨ་དབྱི་བ་དང་རིང་པོ་ཉིད་དག་གོ༑། སདྡྷི་ཙ། ས་དྷི་ཙཱ་ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ་ས་དྷི་ཙི། མ་ནིང་ལ། ས་རྒྱུཀ། ས་ཧྥུག། ས་དྷི་ཙཱི། ས་ཧྥུརྩི་ཞེས་པ་ལ་སོགས་པའོ། །ས་མྱང་། ས་མྱཉྙོ། ས་མྱཉྩ། ས་མྱཧཱུཾ། །ས་མྱཉྙོ། །ཞེ། ས་མཱི་ཙ། ༼ས་མཱི་།
