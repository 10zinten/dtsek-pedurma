ན་རིང་པོའོ། །བི་བ་རཱི་ཥ་ཏེ། བི་བ་རཱི་ཥ་ཏི། །བུ་བུཎྜ་ཏེ། ཨཱ་བི་བརྟུ་ཏི། ཨ་བུ་བུདྷུ་ཏི། ཨཱ་:བ་བེ་རཱི་ཥ་ཏེ། ང་། བི་བ་རི་ཥ་ཏེ། །བི་བ་རཱི་ཥ་ཏི༑བུ་བུ་རཱ་ཏི། བྷཾ། པི་པ་རཱི་ཥ་ཏི། པི་པ་རི་ཥ་ཏི། པུ་པུཎཱ་ཏི། སྲི༑སི་སཱ་རཱི་ཥ་ཏི། སུ་པཱུནྱ་ཏི། སྔོཉ། ཨཱ་ཏིསྟ་རཱི་ཥ་ཏི། ཨཱ་ཏིས་རིཥ་ཏི། ཨཱ་ཏིསྨིན་ཏི། ཨཱ་ཏིས་ྋརཱི་ཥ་ཏེ། ཨཱ་ཏིས་རཱི་ཥ་ཏིཡ། ཨཱ་ཏིཙྪི་ཏེ༑ཏཱཾ། ཏི་ཏ་རཱི་ཥ་ཏི། ཏི་ཏ་རི་ཥ་ཏེ། དེ་བཞིན་དུ། །རཱི། ཛཱི། སྲིཥ་ཧཱུཾ་སྭ་འགྲོ་བ་ལའོ། །དབྱངས་དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་གཉིས་་བརྗོད་དོ། །ཨཱ་རི་རཱ་ཨཻཥ་ཏི། རེ་ཕའི་གཉིས་བརྗོད་པ་མ་ཡིན་ནོ། །ཨཱིརྟི་ཥ་ཏི། ཀྲྀཉ། ཙི་ཀ་རཱི་ཥ་ཏི། ཙི་ཀཱ::དི་ཥ་ཏེ༄༅། ཙི་ཀིརཱ་ཏི། ཙི་ཀཱ་རི་ཥ་ཏེ། ་་གཾ། སྒྲ་ལའོ། །ཛི་ག་རི་ཥ་ཏི། ཛི་ག་རི་ཥ་ཏི། ཛི་གཱིནཱུ་ཏི། ཀྲྀ༑་:རྣམ་གཡེང་ལའོ། །གཽ། ངེས་པར་གནོན་པ་ལའོ། །ཞེས་པ་ཏུད་སོགས་ཅན་གང་ཡིན་པ་ལ་ནི། རྨིང་། པུང༌། །ཞེས་པ་ལ་སོགས་པས་ཨིཊའོ༑་། ཙི་ཀ་རི་ཥ་ཏི། ཙི་ཀ་རཱི་ཥ་ཏི། དབྱངས་ལ་ཡང་ནའོ་ཞེས་པས་་་རེ་ཕའི་ལ་ཉིད་དོ། །ནི་ཛི་ག་ལཱི་ཥ་ཏི། ཡུ་:ནེ་ཛི་ག་ལི་ཥ་ཏི། །ནི་ཇི་ག་གེརཱི་ཥ་ཏི། ནི་ཛི་ག་རཱི་ཥ་ཏི། ཛཱ་གྲི། ཨིཊ་ཨཱ་ག་མའོ། །ཇི་ཛ་གཱ་རི་ཥ་ཏི། ཙ་ཀཱ་སྲི། ཙི་ཙ་ཀཱ་སི་ཥ་ཏི། ་ག་ད། །ཛི་ག་དི་ཥ་ཏི། བྷཎ། བི་བྷཱ་ཎི་ཥ་ཏི། བྲཛ། བི་བྲ་ཛི་ཥ་ཏི། ཉི་ཏཱ་རཱ། ཏི་ཏཱ་རི་ཥ་ཏི། སྤནྣཱ། པི་སྤདྡྷིཥ་ཏེ༑སཱ་ཡཱི། ་་:པི་སྤུ་ཡི་ཥ་ཏེ། :：ཁྱལ། ཏི་ཏྱ་ལ་ཨིཥ་ཏི། བིད། བི་བ་ཏི་ཥ་ཏི། ལ་པ། ལི་ལ་པི་ཥ་ཏི། ཛ་ཡ། ཛི་ཛ་ཡི་ཥ་ཏི། པ་ཋ། བི་བ་ཟི་ཥ་ཏི། ཙ་ལ། །ཙི་ཙ་ལི་ཥ་ཏི། ཙ་ར། འིཙི་ཙ་རི་ཥ་ཏི། ་ཀ་ཎ། ཙི་ཀ་ཎི་ཥ་ཏི། ::ཀ་ཐ། ཙི་ཀ་ཤྲཱི་ཥ་ཏི། ཧ་ས། ཛི་ཧ་སི་ཥ་ཏི། རཀྵ། །
