བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཉིའིའི་ཤ་ཤཱ་པ། ཤེ་པ་ཏུ། །ཤེ་པི་ཐ། ཤ་ཤ་སྦ། ཤེ་པི་བ། ཤེ་པེ། །རཉྫ། ར་རཉྫ། ར་རཾ་ཛ་ཏུ། ར་རོ་ཛི་ཐ། །ར་རཾ་ཀྵ། ར་རཾ་ཛི་བ། ར་རཾ་ཛེ། བྷཉྫ༑བ་བྷཾ་ཛ། བ་བྷཾ་ཛ་ཏུ། བ་བྷོ་ཛི་ཐ། ་བ་བྷོ་ཛི་བ། སྐནྡིར། ཙ་སྐནྡ། ཙ་སྐནྡི་ཏུ། ཙ་སྐནྡི་ཐ། ཙ་སྐནྡ། ཙ་སྐནྡི་བ། བནྡྷ། །བ་བྷནྡྷ༈'བ་བྷནྡྷ་ཏུ། ་བ་བྷནྡྲི་ཐ། བ་བྷནྡྷ། བ་བྷནྡྷིབ། ད་ན། ད་དཾ་ཤ། ད་དཾ་ཤ་ཏུ། ད་:དཾ་ཤི་ལོཐ། ད་དཾཥྛ། ཌད་དཾ་ཤི་བ། ཥཱཉྫ། ས་ཧཱུཾ་ཛེ། ས་ཧཱུཾ་ཛི་ཏེ། ཁ་ཅིག་ཏུ། འདིའི་པ་རོག་ལ་རྣམ་པར་བཤད་པ་ཨ་ནུ་ཎཾ་ག་དབྱི་བ་འདོད་དོ། །རྟག་ཏུ་ཕ་རོལ་ལ་མ་ཡིན་ནོ། །ས་སྭ་ཛེ། པ་རི་ཥ་ཧཱུཾ་ཛེ། ཤནྟིུ། བྷནྡྲ། ཤ་ཐཾ་སེ། གསལ་བྱེད་དང་པོ་ལ་ཨིཊའོ། །ཤ་ཐཾ་སི་ཏེ། ་བ་ཤྲཱི་སེ། དྷཱནྩི། ད་དྷཱ་སེ། །ག་ད། །ཛ་:གཱ་དཏི། ཛ་ག་ད་ཏུ། ཛ་ག་དི་ཐ། ར་ཎ། ར་རཱ་ཎ། སྔ་མ་བཞིན་ཨེ་ཉིད་དེ་ཨ་ཧྥུ་ལིས་དབྱི་བ་ཡང་ངོ༌། །རེ་ཎ་ཏུ། རེ་ཎི་ཐ། དེ་ལ། །ཀྱང་ཡིག་ལས་རིང་པོ་ཡང་ཁ་ཅིག་ཏུ་གཟུང་བར་བྱ་སྟེ། རཱ་དྷ་འཚེ་བ་ལའོ། །།ཨ་བ་ར་རཱ་དྷ། ཨ་་་བ་རེ་དྷ་ཏུ། ཨ་བ་རེ་དྷུ། སྲི་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །ཨ་བ་རེ་དྷི་ཐ། འཚེ་བའོ་ཞེས་པས་འདི་མ་ཡིན་པའི་ནི། ཨཱ་:ར་རཱ་དྷ་ཏུ། ཨཱ་ཅིར་རཱ་དྷི་ཐ། དེ་བཞིན་དུ་རཱ་ཛི་བྷཱ་ཛི་བྷཱ་སི་:བྷཱ་སྨིརྣམས་ཀྱི་ཡང་ནའོ། །རཱ་ཛི་གསལ་བ་ལའོ། །ར་རཱ་ཛ། རེ་ཛ་ཏུ། ར་རཱ་ཛ་ཏུ། རེ་ཛུ། ་ར་རཱ་ཛུ། གསལ་བྱེད་དང་པོ་ལ་ཨིཊའོ། །རེ་ཛི་ཐ། ར་རཱ་ཛི་ཐ། ལོ་རེ་ཛེ། ར་རཱ་ཛེ། བྷཱ་ཛྲ་ཊཱ་བྷི་སྲི་ཊཱ་བྷཱ་སྲི་འབར་བ་ལའོ། །བྷེ་ཛེ། བ་བྷཱ་ཛེ། བྷེ་སེ། བ་བྷཱ་སེ། བློ་སེ། བ་བྷཱ་སེ། ཕ་ལ་རྫོགས་པ་ལའོ།  ོ་། ཉི་པ་ལཱི། བྷ་ཛ། ཏྲ་པུ་ཥ། ཤནྡྲ་གྲོལ་བ་དང་སོ་སོར་དགའ་བ་དག་ལའོ། །གྲནྠ་སྤེལ་—།
