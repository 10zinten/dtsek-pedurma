བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའི:ཡའི་ཡིན་ཨཱ་ཡི་རྣམས་ལའོ། །ཞེས་པས་ཡ་ཡིག་དབྱིའོ། །ན་དབྱི་བ་འང་ངོ་། ཞེས་པས་ཡིན་དང་ཨ་ཡི་དག་ལ་ནཱ་དབྱི་བ་ཉིད་ཀྱི་སྦྱོར་བ་བྱའོ། །མཁས་པ་འདོད་པར་བྱེད་པ་ནི། བི་ེདྭ་སྱ་ཏི། རཱ་ཛཱི་ཡ་ཏི། ལཤརྨི་ཡ་ཏི། པ་ཐི་ཡ་ཏི། ཨ་ནུ་ཏཾ་གའང་ཀྲུཉྩ་དང་ཨིཊ་མ་ཡིན་པའིའོ། །རྟགས་མཐའི་ན་ཡིག་གིའོ༑། གསལ་བྱེད་ལའང་འདི་རྣམས་ཀྱི་ནའོ་ཨེཞེས་པའོ། །།ཨ་ཧརླ་ཏི། རའི་རང་བཞིན་ཅན་ནི་ནཱ་མིའི་ཕ་རོལ་མ་ཡིན་པའང་ངོ༌། །ཞེས་པས་འང་གི་སྒྲའི་མང་པོའི་དོན་འཛིན་པ་ཉིད་ཀྱི་ཕྱིར་ནའི་ར་ཉིད་དོ། །རྩེ་བར་འདོད་པ་ནི། དི་བྱ་ཏི། དྲི་བྱ་ཏི། བཱ་ཙྪ་ཏི། དྲུ་ཧུ་ཏི། བི་གྷཱ་མྱ་ཏི། ཡོ་ཏི་སྐྱ་ཏི། མ་ཧ་ཏུ་ཏི། ཏ་དྱ་ཏི། ཡ་རྱ་ཏི། ཨེ་ཏ་དྱ་ཏི། །ོཨ་ད་སྱ་ཏི། བྷ་བ་རྐྱ་ཏི། ཏཱ་རྱ་ཏི། མ་རྱ་ཏི། ཡུཧྥ་རྱ་ཏི། །ཨསྨྲ་བྱ་ཏི། ཙ་ཏུརྱ་ཏི། ཙ་་ཏུ་རའི་ཝཱ་ྋསྒྲའི་ཨུ་ཉིད་དོ། །ཨ་ན་ཌུ་རྒྱ་ཏི། ཨ་ན་ཌ་ཧའིའང་ངོ༌། །ཞེས་་པས་ཝཱ་ེསྒྲའི་ཨུ་ཉིད་དོ། །གཱིརྱ་ཏི། དྷུརྱ་ཏི། ཨིར་ཨུར་དག་གི་ཨཱིར་མྱུར་དག་གོ། ་། པུཾ་སྱ་ཏི། པུ་མརྨའི་ཨནའི་སྒྲ་དབྱིའོ། །།།སྒྲ་བསྟན་པ་ཉིད་ཀྱི་ཕྱིར་ན་དབྱི་བ་འང་ངོ་ཞེས་པས་ནའི་ཨན་ཞེས་པའི་ངེས་པ་ལས་སོ། །སམྦི་ཊྛ་ཏི། ཨཙྪི་ཊྛ་ཏི། དྷ་ནུ་ཥཱ་ཏི། ན་ྋམི་དང་ཀ་དང་ར་ལས་ཞེས་པ་ལྕལ་སོགས་པས་་ཥ་ཉིད་དོ། :། ལྟོས་པ་མེད་པར་འགྱུར་བ་ཉིད་ཀྱི་ཕྱིར་ཕ་རོལ་གྱི་རྒྱུ་མཚན་ཉིད་ལས་སོ། །ན་ྋམིའི་མཐའ་རྣམས་ཀྱི་ཡན་ཨཱ་ཡི་ཡིན་ཨཱ་ཤི་ཥ་ཙེ་ཀྲི་ཡི་ཏ་རྣམས་ལ་ཡ་ལ་རིང་པོའོ། །ཞེས་པས་ནཱ་མི་མཐའི་རྣམས་ཀྱི་རིང་པོའོ། །ཨགྣི་ཡ་ཏི། པ་ཊུ་ཡ་ཏི། ཨོ་ཥ་དྷི་ཡ་ཏི། བ་སཱུ་ཡ་ཏི། ་རཾ་ཏའི་ཙཱི་ཙེ་ཀྲི་ཡི་ཏ་ཡིན་ཨཱ་ཡི་་རྣམས་ལ་ཨཱི་མཐའོ། །ཞེས་པས་རིཏ་མཐའི་ཨཱི་ཡིག་མཐའོ། :རྗེའི་ཡི་གེ་རའོ། །དཱ་ཏི་ཡ་ཏི། པི་ཏི་ཡ་ཏི། ་་:མ་ཏཱིཙེཡ་ཏི། བྷཱ་ཏི་ཡ་ཏི། སྭ་སཱི་ཡ་ཏི། །
