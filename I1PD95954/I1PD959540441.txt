ཆོས་རྒྱལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ལེའི་ལ་ཌ་ཎ་སྟེ་ཡིན་པ་དེ་རྣམས་ནི་རྣམ་དབྱེའི་མིང་ངོ༌། །དབྱངས་ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུའོ། །གང་གི་ཚོགས་ཀྱི་དང་པོ་དཱིའི་སྒྲ་ཡིན་པ་དེ་དྲི་སོགས་རྣམས་སོ། ཁ། །དྲི་ལ་སོགས་པ་མ་ཡིན་པ་ནི་དྲི་སོགས་མིན་པའོ། །ཀིམ་བ་ཧུ་ཞེས་པ་ཀུན་སྡུད་པའི་ཟླས་དབྱེ་བའོ༑། འདི་ལས་ཏེ་མདོ་འདི་ལས་ཕྱི་མར་བར་མ་ཆད་དུ་ཏས་ལ་སོགས་པ་གང་ཡིན་པའི་རྐྱེན་འཆད་པར་འགྱུར་བ་དེ་རྣམས་རྣམ་དབྱེའི་མིང་ཅན་དུ་ཤེས་པར་བྱ་ཞིང་རྟོགས་པར་བྱ་སྟེ་དེའང་དཱི་སོགས་མིན་པ་སྟེ་དཱི་སོགས་སྤངས་པ་ཉིད་ལས་སོ། །ཐམས་ཅད་པའི་མིང་ཅན་དང་ཀིམ་དང་བ་ཧུ་ལས་ཀྱང་ཕ་རོལ་དུ་འགྱུར་རོ། །ལ་ལ་། དྲི་སོགས་མ་ཡིན་པ་ཞེས་པས་དགག་པ་ཐོབ་པ་ལའོ། ཁོ། ཀིམའི་སྒྲ་ཐམས་ཅད་་་པའི་མིང་ཅན་ལས་གཞན་དུ་ཡང་ཉེ་བར་བླང་ངོ་། །རྣམ་དབྱེའི་མིང་གིས་གང་དུ་དགོས་ཤེ་ན། །ཁྱད་སོགས་རྣམས་ཀྱི་རྣམ་དབྱེ་ལ་ཨའོ་ཞེས་པ་ལ་སོགས་པརྣམས་སུའོ། །དེ་ལས། ཏ་ཏ། །དེ་ལ། ཏ་ཏྲ། ་ཁརྣམ་དབྱེའི་མིང་ཡོད་:པ་ན་ཏད་སོགས་ཀྱི་ཨ་ཉིད་དོ། །ཀིམ་དང་བ་ཧུ་ལས། ཅི་ལས། །ཨ་ཀུ་ཏ༑ཅི་ལི། ཀུ་ཏྲ། མང་པོ་རྣམས་ལས། བ་ཧུ་ཏ། མང་པོ་རྣམས་ལ༑བ་ཧུ་ཏྲ། དྲི་སོགས་མིན་པའི་ཞེས་པ་ཅི་ཞེ་ན། དཱ་བྷུམ། དྭ་ཡོ། ཏས་དང་ཏྲ་ཞེས་པ་འདི་དག་མི་འགྱུར་རོ། །ཐམས་ཅད་པའི་མིང་ཅན་ཞེས་པ་་ཅི་ཞེ་ན། བྲི་:ཀཱ་ཏི། བྲྀ་ཀེ། ཐམས་ཅད་པའི་མིང་ཅན་གྱི་བྱ་བ་ཁོ་ན་བྱས་པའི་ཏས་སོགས་རྣམས་ཀྱི་རྣམ་དབྱེའི་མིང་ཅན་དུ་བཤད་པར་:བྱ་བའི་བྱ་བ་བཞིན་ཏེ་མ་ཡིན་ནོ། །དེས་ན། ཏ་དཱ་ཏ?། ཀ་དཱ། ཞེས་པ་འདིར་ཀ་ྋཡིག་རིང་་པོའོ་ཞེས་པས་རིང་པོར་མི་འགྱུར་རོ། ཨ་། དཱི:༔ཡུཉྩཏ། །ཨསྨཏ། སྟེ། ཏུབྷ་ཝནྟ། ཀིམ་ཞེས་པ་དཱི་སོགས་སོ། །དེ་ལ་ཨི་དམའི་ཨིའོ། །:：དེ་ལ་ཨི་དམའི་སྒྲ་ནི་དེ་:ལས་ཏེ་རྣམ་དབྱེའི་མིང་ཅན་དེ་རྣམས་ཕ་རོལ་ལ་ཨི་ཡིག་ཏུ་བསྒྱུར་བར་འགྱུར་།
