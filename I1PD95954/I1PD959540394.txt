།གྱི་སྒྲའི་རྟགས་ལས་དྲུག་པ་དང་གསུམ་པ་དག་བཤད་པར་བྱ་སྟེ། ཉེ་བར་འཇལ་བའི་མཉམ་པའོ་ཞེས་པའི་ཏུ་ལྕའི་དོན་དོར་བ་ཉིད་ལའོ། ་། ཏུ་ལོ་དེ་བ་དཏྟ་སྱ། ཏུ་ལོ་དེ་བ་དཏྟེ་ནི། ས་དྲི་ཤོ་དེ་བ་དཏྟ་སྱ། ས་དྲི་ཤོ་དེ་བ་དཏྟེ་ནི། ཉེ་བར་འཇལ་བའི་ཏུ་:མྱ་འོས་སྦྱོར་བ་ལ་དྲུག་པའང་ངོ༌། །།ཏུ་ལཱ་:ཏུ་ཏྲ་ས་ེནཱ་ཏྲི་ཨུ་པ་མཱ་ཀྲྀཥྛ་སུ་ན་བི་དྱ་ཏེ། :རྒྱུའི་དོན་ལའོ། །རྒྱུའང་འདི་ཡིན་ལ་དོན་ཡང་ཡིན་པ་ནི་རྒྱུའི་དོན་ཏོ། །དེ། འདིར་གསུམ་པ་རྗེས་སུ་འཇུག་སྟེ། མདོ་ཕྱི་མ་གསུམ་རྣམས་ལའང་ངོ༌། །འབྲས་བུ་སྒྲུབ་ཀྱིཔ་དང་ལྡན་པའི་དངོས་པོ་ནི་རྒྱུའོ། །རྒྱུ་ལས་གསུམ་པར་འགྱུར་རོ། །།བི་ཏྲ་ཡཱ་ཡ་ཤ?། ཀ་ནྱ་ཡཱ་ཤོ་ཀི། རྒྱུ་ལའོ་ཞེས་མ་བྱས་པ་ནི་རྣམ་པ་སྣ་ཚོགས་པའི་དོན་ཏོ། །སྨད་པ་ལ་ཡན་ལག་ལའོ། །ཡན་ལག་ལ་སྨད་ཅིང་སྨོད་པར་བྱེད་པ་ལ་གསུམ་པར་འགྱུར་རོ། །ཨཀྵ་ཀཱ་ཎ། ཤི་ར་སཱ་ཁ@་ཊི། །ནཱ་ས་ཡཱ་ཙི་པ་ཊི། སྨད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། མུ་་ཁེ་:གོ་རP། འདིར་གཙོ་བོ་མ་ཡིན་པ་ལས་གསུམ་པ་བཤད་པར་བྱའོ། །དེས་ན་འདིར་མི་འགྱུར་ཏེ། །ཨཀྵི་ཀཱ་ཎ་དེ་བ་དཏྟ་སྱི། འདིར་ནི་ཨཀྟི་གཙོ་བོའོ། །ཁྱད་པར་དུ་བྱེད་པ་ལའོ། །:གང་གིས་ྋཁྱད་པར་དུ་བྱེད་པ་དེ་ཁྱད་པར་དུ་བྱེད་པའོ༑། ཁྱད་པར་དུ་བྱེད་པ་ལ་གསུམ་པར་འགྱུར་རོ། །ཛ་ཊཱ་བྷི་ལྟ་པ་ས་དྲྀབྦཚེ་ཁངྒེ་སེན་བྷ་ཊོ་དྲི། བྱེད་པ་པོ་ལའང་ངོ༌། ་། བྱེད་པ་པོ་ལ་གསུམ་པར་འགྱུར་རོ། །རུ་:དྲེ་ཎ་དབྷཾ་ཛྲ་པུ་རཾ། བིཥྞ་ནཱ་:བནྡྷོ་བེ་ལིས། དུས་དང་དངོས་པོ་དག་ལ་བདུན་པའོ། །དུས་དངོས་པོ་དག་ལ་བདུན་པར་འགྱུར་རོ། །རེ་ཞིག་དུས་ལ། །་ཤ་ར་དི་པུཥྛནྟི་:པ་སྟཙྪ་དེ། ་བརྟ་སུ་པུ་:པནྟི་ཀུཎྜ་ནི། དངོས་པོ་ལ་སྟེ་དངོས་པོ་ནི་ཡོད་པའོ། །དེ་ཝ་དཏྟེ་ག་ཏེ། ཡ་ཛཱ་དཏྟེ་བྷུཀྟི། ཨཧྥེ་ཧོ་ཏྲེ་ཧྲི་ཡཱ་མཱ་ནེ་ག་ཏཿདུས་དང་དངོས་པོ་དག་ལས་གང་གི་ཚེ་གཞན་་་—།
