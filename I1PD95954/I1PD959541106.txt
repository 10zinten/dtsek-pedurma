གྲྭ་མ་ཎི་བྷི། གྲྭ་མ་ལྕེ། གྲྭ་མ་ཎི་ནེ། གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་ཎི་བྷེ། གྲྭ་མ་ཎྱ། ''གྲྭ་མ་ཎི་ནོ། །གྲྭ་མ་ཎི་བྷཾ། གྲྭ་མ་:ཎི་བྷཱཙྪ། གྲྭ་མ་ཎྱ། གྲྭ་མ་ཎི་ན། གྲ་མ་རྒྱོ། གྲྭ་མ་ཎི་ནོ། གྲྭ་མདྡྷཾ། གྲྭ་མ་ཎྜི་ནཱི། ཎྜི་ལས་་དི་:རྣམས་ཨ་མའོ་ཞེས་པས་ཨཱ་མའོ། །གྲྭ་མ་ཏྱཾ། གྲྭ་མ་ཎི་ནི། གྲྭ་མ་སྐྱོ། གྲྭ་མ་ཎི་ནོ། ་གྲྭ་མ་ཎི་ཤིུ། །ཀུན་དུ་བོད་པ་ལ་ཝ་རི་དང་འདྲ་བའི་གཟུགས་ཏེ། །ཧེ་གྲྭ་མ་ཎེ། ཧེ་གྲྭ་མ་ཎི། ཧེ་གྲྭ་མ་:ཎི་ནི?། དེ་བཞིན་དུ། སེ་་ན་ནི། :གྲྭ་མ་ཨ་གྲ་ཎི། ཏ་ནཱི། ་་ཨ་གྲ་ནཱི་ལ་སོགས་པ་རྣམས་སོ། །མཛེས་པར་རམ་ལེགས་པར་སེམས་པར་བྱེད་དོ་ཞེས་པ། དྷ་:པྱ་དག་གིས་ཞེས་པ༑། ཀྭིཔ་ལ་སཾ་པྲ་སཱ་ར་ཎར་གྱུར་པ་ན་རིང་པོའོ། །ཡང་ན་སེམས་པར་བྱེད་པ་ནི་དྷི་སྟེ། སཾ་པད་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ཀཱི་པའོ། །།འདིའི་བློ་བཟང་པོ་ཡིན་པ་ཞེས་པ་འབྲུ་མང་པོ་པའོ། །སུ་དྷི། དབྱངས་དང་པོ་ཅན་ལ་སུ་དྷིའོ་་ཞེས་པ་འདིས་ཨི་ཡར་བསྒྱུར་བའོ། །སུ་དྷི་ཡོ། སུ་དྷི་ཡ། སུ་:དྷ་ཡཾ། སུ་དྷི་ཡོ། སུ་དྷ་ཎིཡ། །སུ་དྷི་ཡཱ། སུ་དྷི་ཧཱུཾ། སུ་དྷི་བྷིཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལའང་དེ་བཞིན་ནོ། །མ་ནིང་ལ་སཱ་མ་བཞིན་དུ་ཐུང་ངུ་ཉིད་དོ། །སུ་དྷི། སུ་ཙྪཱདྷི་ནཱི། སུ་དྷི་ནི། སླར་ཡང་། སུ་དྷི། །སུ་དྷི་ནི༄༅། །སུ་དྷ་མི་ནི༑ཀ་ཊཱ་ལ་སོགས་པའི་དབྱངས་དང་པོ་ཅན་ལ་སྔ་མ་བཞིན་དུ་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱོགས་ལ་ཐུང་ངུ་དང་ནུ་ཨཱ་ག་མ་དག་མི་འགྱུར་རོ། །སུ་དྷི་ཡཱ། སུ་དྷི་ནཱ། སུ་དྷི་ཧཱུཾ། སུ་དྷི་བྷིཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཨུ་བ་ཨཱ་མི། ར་ཛི་ཏ་ཤྲཱི། ཡ་བ་ཀཱི། བྱཀྟ་:ཧྲི། པྲ་བྷ་བྷ་ལ་སོགས་པ་རྣམས་སོ། །དེ་བཞིན་དུ་སུ་:བྷི། ཀནཱ་ནཱ་དཱི་ཝེལ་སོགས་པ་རྣམས་ཀྱང་སུ་དྷིའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱའོ།  ོ། མི་ཟད་པ་དང་བྱེད་པ་ཅན་དག་ཁོ་ན་ལས་སྒྲུབ་པ་འདིའོ་།
