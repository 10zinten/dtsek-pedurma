བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཊཛ་ཏྭཾ། ནེ་ནི། ནེ་ནི་ཛཱ་ཐཱ། །ནེ་ནིཤྩི། ནེ་ནི་ཛི། ནེ་ནི་ཛཱ་བ་ཧི། ནེ་ནི་ཛ་མ་ཧི། དེ་བཞིན་དུ་བི་ཛིར། བེ་:བིཀྟ། བེ་བིཀྐཱཏ། བེ་བིཏྟཾ། བེ་བི་ཛ་ཏུ། བེ་བིཏྟཱི༄༅། བེ་བིཏྟཱཏ། བེ་བིཏྟཾ། བེ་བིཀྟ། བེ་བི་ཛ་ནི། བེ་བི་ཛཱ་བ། བེ་བི་ཛཱ་མ། :བེ་བློ། བེ་བིབྷིུ། བེ་བིཥྛ། །བེ་བིཊྚཾ། བེ་བི་ཥ་ཏུ། བེ་བིནྟི། བེ་བིཊྚཏི། བེ་བིཤྩ། བེ་བིཤྩ། ་༼བེ་བི་ཥ་ཎི། བེ་བི་ཥ་བ། བེ་བི་ཥཱ་མ། བེ་བི་:ཥ་ཊཾ། །བེ་བི་ཥཱ'ཏྭཾ། :：བེ་བི་ཥ་ཧཱུཾ། བེ་བི་ཥ། བེ་བི་ཥ་ཛཾ། བེ་བི་དྷྭཾ། བེ་བི་ཥི། །་བེ་བི་ཥཱ་བ་ཧི། བེ་བི་ཥ་མ་ཧི། འུ་དཱཉ། ད་དཱ་ཏུ། ཨ་བྱཱསྟ་རྣམས་ཀྱི་ཨ་ཡིག་གོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ལ་ཨ་ཡིག་དབྱིའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་་་པོའོ༑། དཏྟཱཏ། དཏྟཾ། ད་ད་ཏུ། དཱ་ཨསྟི་དག་གི་ཨི་སྟེ་ཨ་ཧྥུ་ས་དབྱི་་བའང་ངོ་ཞེས་པས་མཐའི་ཨེ་ཉིད་དེ་ཨ་ཧྥུ་ས་དབྱིའོ། །དེ་ཧི། དཏྟཱཏ། དཏྟཾ། དཏྟ། ་ད་དྭ་ནི། ད་དྲ་བ། %ད་དཱ་མ། དཏྟཾ། ད་དཱཏཱཾ། ཌཱ་ད་ད་ཏྭཾ། དཏྟཱ༑ད་དཱ་ཐཱ། ད་དྡྷཾ། །མཚམས་སྦྱོར་གྱི་ཡི་གེ་ལ་ཡང་ངོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། །ད་དེ། རིང་པོ་ལས་ཕ་རོལ་དབྱིའོ། །ད་དཱ་བ་ཧི། །ད་དཱ་མ་ཧི༑རྔུ་དྷཱཉ། ཊུ་ཤྲཱིཉ། ད་དྷི་ཏུ། ད་དྷཱ་ཏིའི་ཏ་ཐཱ་དག་:ལའང་ངོ་ཞེས་པས་དྷཱའི་མཐའི་དང་པོ་དེའི་བཞི་པ་དྷ་ཉིད་དོ། །དྷཏྟཱཏི། ་དྷཏྟཾ། ད་་་དྷཏུ༑དྷེ་ཧི།  ེདྷ་ཏཱཏ། དྷཏྟཾ། དྷཏྟ། ད་དྷཱ་ནི། ད་དྷཱ་བ། །ད་དྷཱ་མ། །དྷཏྟཾ། །ད་དྷཱ་ཏྭཾ། :ད་དྷ་ཏཾ། དྷཧཱུ?། ད་དྷཱ་ཤྲཱི། དྷ་ཧཱུྃ། དྷ་ས་དྷི༑ད་དྷཱ་བ་ཧི། ད་དྷཱ་མ་ཧི། བི་ད་དྷཱ་ཏུ། བི་དྷཏྟཱཏ། བི་དྷཏྟཾ། བི་ད་དྷ་ཏུ། བྷོཉ་ཧཱང་མཱང་རྣམས་ཀྱི་ཨིཏའོ། །བི་བྷརྟུ། བྷཾ་ཏཱཏ། བི་ཤྲཱི་ཏཱཾ། བི་བྷ་ཏུ། བི་བྷཾ་ཧི། བི་བྷཾ་ཏཱཏ། བི་བྷཱི་ཏཾ། བི་བྷཱི་ཏ། བི་བྷཱ་སི་།
