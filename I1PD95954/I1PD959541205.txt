བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི?བན་དང་པུ་ཥན་དང་བྲཧྨ་ཧན་དང་ཧ་ལིན་དང་ཨུ་ཤ་ནས་དང་ཨ་ན་ཌཧྲ་དང་མ་དྷུ་ལིཧ་དང་ཀཥྛ་ཧཀྵ་ཞེས་པ་ལ་སོགས་པ་སྟེ། སི་ལ་སོགས་པའི་མཐའ་ཅན་་་རྣམས་བྱིས་པ་རྣམས་ཀྱིས་བྱེ་བྲག་ཏུ་རྟོགས་པར་བྱ་བའི་ཕྱིར་ཡང་དག་པར་བཤད་པར་བྱའོ། །དེ་ལ་སྐུ་རུའི་སྒྲ་ནི་བྱིངས་དང་རྣམ་དབྱེ་བོར་བའི་དོན་དང་ལྡན་པ་རྟགས་སོ་ཞེས་པས་རྟགས་ཀྱི་མིང་ཅན་ནོ། །དེ་ལས་ཕ་རོལ་རྣམས་རྣམ་དབྱེའོ་་་ཞེས་པ་ནི་དང་པོ་ལ་སོགས་པའི་རྣམ་དབྱེ་བདུན་པོ་རྣམས་ལ་འགྱུར་ཏེ། དོན་ཤིན་ཏུ་རྣམ་པར་འབྱེད་པའི་ཕྱིར་ཞེས་པས་རྣམ་དབྱེའོ། །དེ་ལ;སིཨོཛས་དང་པོའོ། །ཨམ:ཨོ:ཤས་གཉིས་པའོ། །ཊཱ་བྷཱུམ། །བྷིས་གསུམ་པའོ། །ངེ་བྷཱུམ། ས་བཞི་པའོ། །ང་སི། བྷཱུམ། བྷུས་ལྔ་པའོ། །ངསཨོསཨཱམ་དྲུག་པའོ། །ང་ཨོསསུཔྲ་བདུན་པའོ། །དེ་ལ་ཐམས་ཅད་དུ་སི་ལ་་སོགས་པ་རྣམས་ཐོབ་པ་ལ་རྟགས་ཀྱི་དོན་གྱི་ཚིག་ལ་རྣམ་དབྱེ་དང་པོའོ་ཞེས་པས་་དང་པོའི་རྣམ་དབྱེར་འགྱུར་རོ། །དེའང་དོན་གཅིག་ལ་གཅིག་གི་ཚིག་སིའོ། །རིམ་པ་ལས་མ་འདས་པར་ཕྲལ་བར་བྱའོ། ་། ཨི་ཡིག་ནི་འདིར་གང་རྗེས་སུ་འབྲེལ་པ་རྣམས་མི་སྦྱོར་རོ་ཞེས་པས་ཁྱད་པར་གྱི་དོན་ཏོ། ཌ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པ་སྤྱིར་བཏང་ནས་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ་ཞེས་པས་ས་ཡིག་ནི་རྣམ་པར་གཅད་བྱའོ། །སཱུརྱ། གཉིས་ཀྱི་དོན་དག་ལ་གཉིས་ཀྱི་ཚིག་ཨོའོ། །རིམ་པ་ལས་མ་འདས་པར་ཕྲལ་བར་བྱའོ། །།།ཨོ་ཡིག་ལ་ཨོ་སྟེ་ཨོ་ཡིག་ལའང་ངོ་ཞེས་པས་ཨོ་ཡིག་ཏུ་འགྱུར་ཏེ་ཕ་རོལ་དབྱི་བའང་འཐོབ་བོ། ་། གསལ་བྱེད་་དབྱངས་མེད་པ་ཕ་རོལ་གྱི་ཡི་གེར་དྲང་ངོ༌། །སཱུཚོ། མང་པོའི་དོན་རྣམས་ལ་མང་པོའི་ཚིག་ཛས་སོ། །ཨ་ཡིག་ཕྲལ་བར་བྱའོ། །ཛ་ཡིག་ནི་ཛས་ལའོ་ཞེས་་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨ་ཡིག་ལ་དབྱིའོ་ཞེས་པས་དབྱི་བ་ཐོབ་པ་ལ་ཨ་ཡིག་།
