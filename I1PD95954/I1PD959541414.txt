འབྲེལ་པ་ཉིད་ཀྱི་ཕྱིར་ན་ཨཱ་ག་མའོ། །ཀཾ་སཱིཏ། ་ཎི་སིའོ་བྱེད་པ་ལའོ། །ནི་་་སཱིཏ༈ཎི་ཛི་དག་པ་ལའོ། །:ནི་ཛཱིཏ། ཤི་ཛི་མི་གསལ་བའི་སྒྲ་ལའོ། །ཤི་ཛཱིཏ། ཤྲཱི་ཛཱའི་དོར་བ་ལའོ། ་། བྲི་ཛཱིཏ། །པྲ་ཙྪི་ཚོགས་པ་ལའོ། །པྲ་་་ཙཱིཏ། །སཾ་པྲ་ཙཱིཏ། ཥཱང་སྲོག་ཆགས་མངལ་ནས་སྐྱེས་པ་ལའོ། །སྔ་མ་་བཞིན་ཨུབའོས། །སུ་བཱིཏ། པྲ་སུ་བཱིཏ། ཤིང་ཉལ་བ་ལའོ། །ཤིངའི་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་ཡོན་ཏན་ནོ། །ཨེ་ཨ་ཡའོ། ་། ཤ་ཡཱིཏ། །་སཾ་ཤ་་པཱིཏ། །ཨ་ཏི་ཤ་ཡིཏ། ཨིང་ཀློག་པ་ལའོ། །དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་་་སོགས་པས་ཨི་ཡའོ། །་། ཨ་དྷི་པཱིཏ། དྲི་དྷིང༌། །བེ་བཱིང༌། །དཱི་དྷི་བེ་བཱི་དག་གི་ཨེ་ཡིག་ཡ་ཡིག་དག་ལའོ་ཞེས་པས་མཐའ་དབྱིའོ་ཨཱ་དཱི་དཱིཏ། །ཨཱ་བེ་་་བཱིཏ། སྒྲུང༌། །ཨུ་བའོ། །ནི་ནྱཱ་བཱིཏ། ཨཱཏྨ་ནེའི་ཚིག་གོ། །:དཱི་ཥ། དྲི་ཧེཏ། དྭི་ཥཱིཏ། དུ་ཧ། དུ་ཧཱུཏ། དུ་ཧཱིཏ། དི་ཧ། དི་ཧཱཏ། དི་ཧཱིཏ། །ལི་ཧེ། ལི་ཧཱུཏི། ལི་ཧཱིཏ། ཨུཙྪཱཉ། སྲོཙྪཱ་ཡཱཏ། ་པོཥྞ་པཱིཏ། །བྷུཉ། སྐུ་ཡཱཏ། ་སྐུ་བཱིཏ། བྷུཉ། བྲུ་ཡཱཏ། བྲུ་བཱིཏ། གཉི་གའི་་་ཚིག་:གོ༑། ཧུ་སྦྱིན་:པ་ལའོ། །སྔ་མ་བཞིན་གཉིས་བརྗོད་དོ། །ཨ་བྷཱུ་པེ་སའི་བྱ་བ་ཡང་ངོ༌། །ཛུ་ཧུ་ཡཱཏ། ངེས་པ་མེད་པ་ལའང་ཨཱ་ག་ཏི་ཞེས་པ་ལ་་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །།ཛུ་ཧོ་ཏིའི་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་པ་ཉིད་དོ། །བྱ་ཏི་:ཛུ་ཧཱི་ཙིཏ། :ཉི་བྷི། བྷིའི་ཡང་ནའོ་ཞེས་བརྗོད་པར་བྱའོ་ཞེས་པས་ཡང་ན་ཨི་ཉིད་དོ། །བི་བྷི་ཡཏ། བི་བྷི་ཡཱཏ། སྔ་མ་བཞིན་ཨཱཏྨ་ནེའི་ཚིག་གོ། །།ཨི་:ཡིག་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །བྱ་ཏི་བི་བྷིཏ། །ཧཱུཾ། ་་:ཛཱི་ཧཱི་ཙྪཱཡཱཏ། པྲི། པི། པྲ་ཡཱཏ། ཨོ་ཧཱཀ་། ཛ་ཧཱ་ཏིའི་སཔཱ་མཱི་ལ་དབྱིའོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། །ཛ་ཧཱཏ། །ཤྲཱིཙྩཾ། ཛགྷོ་S།
