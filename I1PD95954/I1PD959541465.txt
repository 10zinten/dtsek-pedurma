བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི༎ཀྱི་Onཀའོ༑། ཨ་དཱ་ཀཱིཏ། ཨ་དཾཏྟཾ། ཨ་ཛྭགླུ། བྷརྫ། བྲཱཛ་སོགས་རྣམས་ཀྱི་ཞེས་པས་བྗཛའི་ཉེ་བར་མཚོན་པ་ལ་ཚིག་ཕྲད་ཀྱི་འདོད་པའི་ཡུལ་ཉིད་ཀྱི་ཕྱིར་སཾ་པྲ་སཱ་ར་ཎ་ལས་གཞན་དུ་རྣམ་པར་བཤད་དེ་བཱི་ཛ་རུ་བསྒྱུར་རོ། །གསལ་བྱེད་མཐའ་རྣམས་ཀྱི་ཨིཊ་མེད་པའིའོ་ཞེས་པས་པ་རསྨིའི་ཚིག་ལ་འཕེལ་བའོ། །ཁ། བྲཱཛ་སོགས་རྣམས་ཀྱི་ཥའོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཨ་བྷཱཎྜིཏ། ཨ་བྷཱནྟཾ་:ཨ་བྷརྒྱུ༑:བརྩེགས་པའི་དང་པོ་ས་ཀ་དག་གི་མཐའ་ལའང་ངོ་ཞེས་པས་ས་དབྱིའོ༑། །ཨ་བྷཱཀཱིཏ། ཨ་བྷཱཝཾ། ཨ་བྷཱཀྵ། དགག་པ་ངེས་པར་བསྟན་པ་མི་རྟག་གོ་ཞེས་པས་སི་ཙ་ཨཱ་ཤཱིཥ་དག་ལའང་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་ལ་ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨ་བྷཀྵ༄༅། ཨ་བྷཎཱ་ཏྭཾ། ?ཨ་བརྒྱུ་ཏ། ཨ་བནྡྷེ། ཨ་:བརྐུ་ཐཱ། དྷཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་ཥའི་ཊའོ། །ཨ་བྷནྡྷོཨེ། ཨ་བྷཉྫ། ཨ་བྷཀྵ་ཏྭཾ། ཨ་བྷ་ཀཱ་སྟེཏ༑་་ཨ་བྷཥྛ། །ཨ་བྷགྷཱ་ཤྲཱ'། ཨ་བྷ་དྡྷཾ། པྲཙྪ། ཙྪ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །སྔ་མ་བཞིན་ཥའི་ཀའོ། །ཨ་པྲཀཱིཏ། མ་ཨ་པྲཎྞཾ། ཨ་པཱ། སམ་ལས་ག་མི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ས་མ་དྲེ། ས་མ་པྲཀྵ་ཏྭཾ། ས་མ་པྲཀཱ་ཏ། །འདྲི་བ་ལ་ནུའོ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །པྲཋེ། །བྷནྡོ། ཙ་སྡེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཀིའོ། །ཨ་བྷྱ་གླིཏ་ཨ་བྷཱ་བྷཱུཏྟཾ། ཨ་བྷཾ་གླུ། །རཱ་དྷ། སཱ་དྷ། ཨ་རཱཏཱིཏ། །ཨ་རཱདྡྷཾཀྱེ། ཨ་རཱནྟུ། ཨ་སཱཏཱིཏ། ཨ་སདྡྷཾ། ཨ་སཱཏྟུ། ཧ་ན། ཧནྟིའི་ཨེཨཱ་ཤཱིཥ་ལ་བ་དྷིའོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ་ཨ་བྱ::ཏ་ནི་ལ་ཡང་ངོ་ཞེས་པས་ཡང་ན་བ་དྷི་རུ་བསྒྱུར་རོ། །ན་མཐའ་ཉིད་མ་ཡིན་པའི་ཕྱིར། :ཧ་ནི་མ་ནྱ་ཏི་ལས་ན་ལས་སོ་ཞེས་པས་ཨིཊ་བཀག་པ་མེད་དོ། །—།
