རྣམས་ལ་ཡ་ལ་རིང་པོའོ། །ཞེས་པས་ཨཱ་ཤཱིཥའི་ཡ་ལ་རིང་པོ་ཉིད་དོ། །ཛཱི་ཡཱཏ། ཛཱི་ཡཱ་སིཾ། །ཛཱི་ཡཱ་སུ། །བི་པ་རཱ་དག་ལས་ཛིའོ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་་་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །བི་ཛེ་ཊཱི། བི་ཛེ་ཏཱི་ཡཱནྟཾ། བི་ཛེ་ཏཱི་རན། བི་ཛི་ཊཱིཥྛ། བི་ཛེ་ཏཱི་ཝིཡཱསྟྲཱི། ནཱ་མིན་ོམཐའ་ཅནའི་བྱིངས་ལས་ཨཱ་ཤཱིཥ་ཨརྱ་ཏ་ནཱི་པ་རོཀྵ་རྣམས་ལ་དྷའི་རྙའོ། །ཞེས་པས་དྷ་ཡིག་གི་ཌྷ་ཉིད་དོ། །བི་ཛེ་ཊཱི་ནཱཾ། །བི་ཛེ་ཊྛི་ཡ། བི་ཛེ་ཏཱི་པ་ཧི། །བི་ཛེ་ཏཱི་མ་ཧི། པ་ར་ཛེ་ཏཱིབྦསི། ་ཀྱི་ཟད་པ་ལའོ། །ཀྱི་ཡཱཏ། ཨི་འགྲོ་བ་ལའོ༑། ཨི་ཡཱཏ། །ཨུ་དི་ཡཱཏ། ཨི་ཀ་། ཨ་དྷི་ཡཱཏ། རི་བི། རཱི་ཡཱཏ། །དྷཱི་འཛིན་པ་ལའོ། །དྷཱི་ཡཱཏ། ཨིཎ། ་ཉེར་བསྒྱུར་མེད་པའི་ཨིཎའི་འོ། །ཞེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཨཱི་ཡཱཏ། ཉེར་བསྒྱུར་མེད་པ་ཉིད་ཀྱི་ཞེས་་པའི་ངེས་པ་ལས་ཉེར་བསྒྱུར་གྱི་ནཱ་མིན་མཐའ་ྋརྣམས་ཀྱི་ཡཎ། །ཨཱ་ཡི། །ཞེས་པ་ལ་སོགས་པས་ཡ་ལ་རིང་པོ་མེད་དོ། །ས་མི་ཡཱཏ། ཨུ་དི་ཡཱཏ། ཨ་བྷི་ཡཱཏ། པཱ་རཱི་ཡཱཏ། ཞེས་པ་མཚུངས་པའི་རིང་པོའོ། །རྨིང་། སྨོཏྟཱིཀྵ། :སྨེ་ཏཱིནྡྷཾ། ཨིང་། ཨ་རྒྱ་ཏིཥཱ། ཨཱིང་། ཨེ་ཏི། ཙིཉ། ཙཱ་ཡཱཏ། ཙེ་ཏི། དེ་བཞིན་དུ་ཥཉི། སཱི་ཡཱཏ། :སེ་ཊྛིཏྟཱ། དེ་བཞིན་དུ་ཤིཉའོ། །བཱི་རབ་ཏུ་སྐྱེ་བ་ལའོ། །བཱི་ཡཱཏ། །ཉི་བྷི། བྷི་ཡཱཏ། ཧཱི། ཧཱི་ཡཱཏ། དྷིང་། དྷེ་ཊཱི༑མཱིང་། མེ་ཏྟིཋ། །པཱིང་། པེ་ཏིཥྛ། །ཎཱིཉ། ྋཎྜིཡཱཏ། ནེ་ཏིཥྛ། ཌཱ་ཀཱིཉ། ཀྲྀ་ཡཱཏ། །ཀྱེ་ཏིཥྛ། བཱིཉ། པཱི་ཡཱཏ། པྲ་ཏིཥཱ། ལཱིང་། བི་ལེ་ཏི། ་ལ། བི་ཝེལཱ་སཱིཤྩ། འདིའི་ནི་ཡོན་ཏན་དང་འཕེལ་བའི་གནས་ལ་ཡ་ཛ་སོགས་སོ། །།ཤིཉ། ཤྲཱི་ཡཱཏ། ཨིཊ་ཨཱ་ག་མའོ། །ཤ་ཡི་ཊྛིཏྠ༑དེ་ལ་བྱིངས་ཀྱི་ཞེས་པའི་མིང་སྔོན་མ་ཅན་ཉིད་ལས་ཨིཊ་མེད་དོ། །དེ་།
