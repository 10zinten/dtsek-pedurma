དྷརྨུ༑། ད་ནི་དྷཱ་སྱ་ཏེ། རྒྱུ་བྷནྡྷ་བ་ནཱི་བྷཱ་སྱ་ཏེ། <ཀ་ས་འགྲོ་བ་ལའོ། །ཙ་་ནཱི་ཀ་ཅ་ཏེ། པ་ཏི། པ་ནཱི་པ་ཏ་ཏེ། པ་ད། པ་ནི'ེརྱ་ཏེ། ྋསྐནྡིར། ཨཱ་ཙ་ནི་སྐ་བྱ་ཏེ། རྗེས་སུ་སྣ་ལྡན་མཐའ་ཅན་གྱི་ཨཏའི་རྗེས་སུ་ང་རོའོ་ཞེས་པས་རྗེས་སུ་སྣ་ལྡན་མཐའ་ཅན་གྱི་བྱིངས་ལས་:ཙེ་ཀྲི་ཡི་ཏ་ལ་གང་ཨ་བྷཱ་ས་དེའི་ཨཏ་མཐའ་ལས་ཀྱང་སྟེ་ཨཏ་དུ་གྱུར་པ་ལ་རྗེས་སུ་ང་རོ་འགྱུར་རོ། །སྡེ་པ་ལ་སྡེ་པའི་ལྔ་པར་ཡང་ནའོ་ཞེས་པས་སྡེ་ཕ་རོལ་ལ་སྡེ་པའི་ལྔ་པར་ཡང་ན་འགྱུར་རོ། །ཀ་ཎ།  ོཙམ་ག་ཅུ་ཏེ། བྷ་ཎ། །ཐཾ་བྷ་ལྕེ་ཏེ། པ་ཎ། པཾ་པ་ཅུ་ཏེ། །ར་ཎ༑རཾ་ར་ཏྱཱ་ཏེ། ཀ་ཎ། ཙི་ཀ་ྋརྱ་ཏེ། ཏ་ན། ཏཾ་ཏ་ཙྪ་ཏེ། མ་ན། མཾ་མ་རྱ་ཏེ། །ཎཾ། ནཾ་ན་རྱ་ཏེ། པྲ་ཎཾ་ན་མྱ་ཏེ། ཡམ། ཡཾ་ཡ་མྱ་ཏེ། དེ་བཞིན་དུ་ར་མ། ཤ་མ། :：ཤ་མ། :ཏ་མ། ད་མ། ཙ་མ། །བ་མ་་་སོགས་ལས་སོ། །བྷཱ་མ། ་ཁྲོ་བ་ལའོ། །ཨཏ་ཅེས་པ་ཅི་ཞེ་ན། མཚན་་ཉིད་ཅན་ཉིད་ཀྱི་ཕྱིར་ཐུང་དུར་བྱས་པས་འགྱུར་བ་མ་ཡིན་པ་ལས་རྗེས་སུ་ང་རོ་ཨཱ་ག་མ་མི་འགྱུར་རོ། །།བཱ་བྷ་མྱ་ཏེ། གཞན་དུ་ཨཏ་ྋགསུང་སྟེ། བི་བྷཱ་མྱ་ཏེ། ཞེས་པ་འགྱུར་བ་ཉིད་དོ། །《། ཧ་ན། ཧན་ཏིའི་ཡང་ན་རྨི་རུ་བརྗོད་པར་བྱའོ་་ཞེས་པས་འཚེ་བ་ཡང་ན་གི་རུ་འགྱུར་རོ། །ཛེ་གྷི་ཡ་ཏེ། ཨ་བྷཱུ་ལས་ལས་་་་ཀྱང་ངོ་ཞེས་པས་ཧའི་ལྷ་ཉིད་དོ། །ཛཾ་གྷ་ནྱ་ཏེ། དེ་བཞིན་དུ་ཡང་ཏྭཾ་ར་ཛ་ན་ཙ་ཊ་ཀ་མ་པི་ན་ཛཾ་གྷ་ནྱ་སེ་ཞེས་རབ་ཏུ་སྦྱོར་བ་བལྟ་བར་བྱའོ། །གཞན་རྣམས་ཀྱིས་ནི་འདིར་རྟག་ཏུདྨེ་རུ་འགྱུར་བར་འདོད་དོ། ་། འགྲོ་བ་ལ་ནི་ཛི་གྷ་ནྱ་ཏེ་ཞེས་པ་ཉིད་དོ། །ཁ་ནུ། ཥ་ནུ། ཛ་ན། ཛ་ནི། ཁ་ནི་ས་ནི་ཛ་ནི་རྣམས་ཀྱི་དྷུཊ་ལའོ་ཞེས་སྤྱིར་བཏང་ནས། ་་ཡ་ལ་ཡང་ནའོ་ཞེས་པས་ལྔ་པའི་ཡང་ན་་་ཨང་ཉིད་དོ། །མཚུངས་པའི་རིང་པོ་ལ་སོགས་པའོ། །གཉིས་བརྗོད་དོ། །—།
