ཎི༑འབི་བྷཱ་བ། བི་བྷཱ་མ། བི་བྷཱི་ཏཱཾ། བི་བྷཱུ་ཏཾ། བི་བྷཱི་ཥཱི། བི་བྷཱ་ཐཾ། བི་བྷཾ་དྷཾ། བི་བྷ་རི། །:བ་བྷ་རཱ་བ་ཧི། བ་བྷ་རཱ་མ་ཧི། མཱང་ཚད་ལ་སྟེ་སྒྲ་ལའོ། །གཉི་ག་རྣམས་ཀྱི་དྭ་མ་ཡིན་པའི་གསལ་བྱེད་དང་པོ་ལ་ཨཱི་ཡིག་གོ། །མི་མཱི་ཏཱཾ། སྔ་མ་བཞིན་ཨཱི་ཡིག་དབྱིའོ། །མི་མཱ་ཏཱཾ། :་:མི་མ་ཏྭཾ། མི་མཱི་ཥ། མི་མཱ་ཤྲཱི། མི་མཱི་ནཱཾ། མི་མི་དྡྷཾ། ་མི་མི། མི་མཱ་བ་ཧི། མི་མཱ་མ་ཧི། །ཨོ་ཧཱང༌། ཛི་ཧཱི་ཏཱཾ། ཛི་ཧཱ་ཏྭཾ། ཇི་ཧ་ཏྭཾ། ཛི་ཧི་ཏྭཾ། དི་བུ། །དཱི་བྱ་ཏུ། དི་བྱ་ཏཱཏ། དཱི་བྱ་ཏྭཾ། དི་བྱནྟུ། ཨ་ཡིག་ལས་ཧནྟི་མ་ཡིན་པ་ལ་ཧིའིའོ་ཞེས་པས་ཧིའི་དབྱིའོ། །དཱི་བྱ། དཱི་བྱ་ཏཱཏ། དཱི་བྱ་ཏཾ། :：ད་བྱ་ཏཱ། དྨ་བྱ་ཏེ། དཱི་བྱཱ་ནི། །དཱི་བྱཱ་བ། དྭ་ེབ་མ། ཁྱུང་སྲོག་ཆགས་རབ་ཏུ་སྐྱེ་་བ་ལའོ། །སཱུ་ཡ་ཏཱཾ། ཌསཱུ་ཡེ་ཏྭཾ༄༅། སཱུ་ཡནྟཾ། སཱུ་ཡ་སཱ། :སཱུ་ཡེ་ཐཾ། ཌསཱུ་ཡེ་ཧཱུཾ།  ོསཱུ་ཡི། སཱུ་ཡཱ་བ་ཧི། སཱུ་ཡཱ་མ་ཧི། །ཥཱ་ཉ། སུ་ནོ་ཏུ། སུ་ནུ་ཏཱཏ། །སུ་ནུ་ཏཱ། སུ་ནཱནྟུ། ནུ་ལས་ཀྱང་བརྩེགས་པ་མ་ཡིན་པའི་བི་ཀ་ར་ཎ་ལས་སོ་ཞེས་པས་ཧིའི་དབྱིའོ། །ཧི་ལའང་ངོ་ཞེས་པས་དང་པོ་ཉིད་ནས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཧི་དབྱི་བ་ལ་ཡང་ན་རྐྱེན་དབྱི་བ་མཚན་ཉིད་ཀྱིས་སོ། །སུ་ནུ། །སུ་ནུ་ཏཱཏ། སུ་ནུ་ཏྭཾ། །སུ་ནུ་ཏ། སུ་ནུ་བཱ་ནི། ་སུ་ནཱ་བ། སུ་ནཱ་ལེམ། །སུ་ནུ་ཏྭཾ། སུ་ནཱ་ཏཱ། སུ་ནཱ་ཨེཏཾ། སུ་ནུ་ཥཱ། སུ་ནཱ་བྷཾ། །སུ་ནུ་དྡྷཾ། སུ་ནཱི༑：སུ་ནཱ་བ་ཧི། སུ་ནཱ་མ་ཧི། དྲྀཉ། དྲྀ་ཎོ་ཏུ། བྲོ་ཆུ་ཙིཏཱིཏ། བྲོ་ཆུ་ཏྭཾ། །དྲྀ་ཎནྟུ། བྲི་ཆུ། བྲྀ་ཊུ་ཏཱཏ། བྲི་ཆུ་ཏཾ། བྲྀ་ཊུ་ཏ། ཀྲྀ་ཊུ་ཏྭཾ། ཤྲཱི་ཎཱ་ཏྭཾ། ཤྲཱི་ཎཱ་ཏཱཾ། ཌ། ཨཱ་པྲི། །ཨཱ་མོ་ཏུ། ཨཱ་པཱུ་ཏཱཏ། ཨཱ་ཙཱུ་ཏཱཾ། ཨཱ་ཛྙཱ་བནྟུ༑། བརྩེགས་པ་མ་ཡིན་པའི་ཞེས་བཀག་པ་ལས། ནུ་ལས་ཀྱང་བརྩེགས་པ་མ་ཡིན་པའི་བི་ཀ་ར་ཎ་ལས་སོ་ཞེས་པས་ཧི་དབྱི་བ་མེད་དོ། །ཨཱ་རཱུ་ཧི། ཏུད། །
