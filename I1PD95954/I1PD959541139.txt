བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིགི་དང་པོ་དང་གསུམ་པ་དག་གོ་ཞེས་པས་ཏ་ཡིག་གོ། །།ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུ་ཉིད་དོ། །ཨརྟི་ཨིན་གྷས་དབྱངས་གཅིག་པ་ཨཱཏ་རྣམས་ཀྱི་ཝནྟི་ལ་ཨི་ཊའོ་ཞེས་པས་ཨི་ཊའོ། །བྱིངས་ཐམས་ཅད་མ་ཡིན་པ་ལ་ཨ་དབྱིའོ་ཞེས་པས་ཨ་ཡིག་་དབྱིའོ༑། ཏསྦི་ཝནྣ་ཞེས་གནས་པ་ལ་སི་སོགས་རྣམས་སོ། །༄༅། ས་མཐའ་དང་མ་ཧནྟ་དག་གི་ནའི་ཨུ་པདྷཱའིའོ་ཞེས་པས་གྷཊ་ལས་རིང་པོ་ཉིད་དོ། ཚུ། གསལ་་་བྱེད་ལས་ཀྱང་ངོ་ཞེས་པས་དབྱིའོ། །བརྩེགས་པའི་མཐའ་དབྱི་བའང་ངོ༌། །ཏསྦྲ་ཝཱན། ཏསྦི་ཝནྣོ༄༅། ཏསྦི་ཝནྣི། ་ཏསྦི་ཝནྣཾ། ཏསྦི་ཝནྣོ། །སྒུཊ་་མ་ཡིན་པའི་དབྱངས་དང་པོ་ཅན་ལ་ཝནྟི་ཨིཊ་དང་བཅས་པའི་འང་ཝ་སྒྲའི་ཨུ་ཉིད་དོ། །།ཥ་ཉིད་ཀྱང་ངོ༌། །ཏ་ཥཱི། ཏསྦུ་ཏཱ། ཨ་ནུ་ཎཾ'ག་དབྱིའོ། །རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་རྣམས་ལ་ཨ་ན་ཌ་ཧ་དང་ན་ཧི་དང་ཝནྟི་རྣམས་ཀྱི་འང་ངོ་ཞེས་པས་སཾའི་ད་ཉིད་དོ། །:ཏསྦ་ཝདྡྷཾ། ཏསྦི་ཝདྷི། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ་ན་ད་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨིའི་རྐྱེན་ནོ། །ཏསྦྲ་ཏཱི། མ་ནིང་ལ། སི་ཨོ་དག་དབྱི་སྟེ། །ཨ་ནུ་ཏཾ་ག་དབྱི་བའང་ངོ༌། །སྔ་མ་བཞིན་དུ། ས་ཡིག་གི་ད་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དང་པོ་ཉིདང་གསུམ་པ་དག་གོ། །ཏསྦི་ཝཱཏ། :ཏེསྦྲ་ཝད། ཏསྦྲ་ཝ་ཏཱི། ཏསྦྲ་ཝནྟི། དེ་་བཞིན་དུ། ཝ་ཡེ་ཡི་ཝནྣི། བ་པི་:ཝམྨ། །པི་ཙིཝནྣ། ཤེ་ཀཱི་ཝནྟ། ཨཱ་ཤི་ཥི་ཝརྨ། བི་དྭརྨ་སོགས་རྣམས་:སོ། །ཨ་ཡནྣའི་མཐའ་ཅན་བརྗོད་པར་བྱ་སྟེ། འདི་ནི་པྲ་ཤ་སྱ་སྟེ། །འདི་ནི་འདི་དག་ལས་ཕུལ་དུ་བྱུང་བས་ན་ཞེས་་པས་ཏ་མ་ལ་སོགས་པར་ཀློག་པའི་ཕྱིར་ཨི་ཡངྒའོ། ཁྲོ། པྲ་ཤ་སེའི་ཙྪཱཐའོ་ཞེས་པས་་ཤ་རུ་བསྒྱུར་བའོ། །།དེ་དང་འདྲ་བར་ཨིཥྛ་ནེ་ཡའང་མང་:པོར་རོ་ཞེས་པ་འདིས་ཡི་གེ་གཅིག་མ་ཡིན་པའི་མཐའི་དབྱངས་དང་པོ་དབྱི་བ་ལ་སོགས་པ་མེད་དོ། །།
