བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤེའི*གཅིག་ལ་རྣམ་པར་གྱུར་པ་ནི་གཞན་མ་ཡིན་པ་དང་འདྲའོ་ཞེས་པས་སྔ་མ་བཞིན་ཨཱ་ཉིད་དོ། །བ་ཧུ་རཱ་ཤྩ། བ་:ཧུ་རཱ་བྷིཥྛ། ་བ་ཧུ་རཱ་ཡེ། བ་ཧུ་རཱེཎེ་ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་སཱ་རི་ལ་སོགས་པ་རྣམས་སོ། །ཨོ་ཡིག་གི་་མཐའ་ཅན་ཨ་ཏི་བྱོའི་སྒྲ་སྟེ། ནམ་མཁའ་ལས་འདས་པ་ནི། ཨ་ཏི་བྱོ། བུད་མེད་དང་སྐྱེས་པ་དག་ལ་བྱོའི་སྒྲ་དང་འདྲའོ། ོ། ཨ་ཏི་བྱོ༄༅། །ཨ་ཏི་ཅུ་ལོ། ཨ་ཏི་བྱ་ཝཞེས་པ་ལ་སོགས་པའོ། །མ་ནིང་ལ་ཐུང་ངུ་ཉིད་དེ་ཝ་སྐུ་དང་འདྲའོ། །ཨ་ཏི་བྱུ། ཨ་ཏི་བྱུ་ནི། ཨ་ཏི་ཅུ་ནི། སླར་ཡང་དེ་བཞིན་ནོ། །ཊཱ་ལ་་སོགས་པའི་དབྱངས་དང་པོ་ཅན་ལ་ཡང་ན་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་རོ། །ཨ་ཏི་རྱ་ཝའི། ཨ་ཏི་བྱུ་ནཱཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། བི་ཤ་ལ་དྲོ། སུ་བྱོ་པ་ཐི་ཏ་སོ་ལ་སོགས་པ་རྣམས་སོ། །ཨོ་ཡིག་གི་མཐའ་ཅན་སུ་ནོའི་སྒྲ་སྟེ། བུད་མེད་དང་སྐྱེས་པ་དག་ལ་ནོའི་སྒྲ་དང་འདྲའོ། །ས་ནོ། སུ་ན་ལོ། སུ་ནཱ་ཝཞེས་པ་ལ་སོགས་པའོ། །མ་ནིང་ལ་དབྱངས་ཐུང་ངུ་ཞེས་པ་ལ་སོགས་་་པས་ཐུང་ངུ་ཉིད་དོ། །དེ་ནས་ཝ་སྟུ་དང་འདྲའོ། །སུ་ནུ། སུ་ནུ་ནི། སུ་ནུ་ནི། སླར་ཡང་དེ་བཞིན་ནོ། །གསུམ་པ་ལ་སོགས་པ་དབྱངས་དང་པོ་ཅན་ལ་ཡང་ན་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བའོ། །སུ་ནཱ་ཝཱ། སུ་ནུ་ནཱ། སུ་ནུ་བྷཾ། སུ་ནུ་བྷི། དེ་བཞིན་དུ་ཨ་ཏི་ནོ། བྷ་གཱ་ནོ། ་ན་མི་ཏ་གློའི་སྒྲ་རྣམས་སོ། །བ་ཧུ་རི་དང་། ཨ་ཏི་དྲོ་དང་། སུ་ནོ་ལ་སོགས་པ་རྣམས་ཀྱིས་སྐྱེས་པ་དང་འདྲ་བར་:འགྱུར་བ་ཅི་མི་འདོད་དོ་ཞེས་པ་ཁ་ཅིག་གོ། །དེའི་ལུགས་ལ་དབྱངས་ཞེས་པ་:ཁ་ཅིག་གོ། །དེའི་ལུགས་དབྱངས་ཞེས་པ་ཅི་ཨ་ཏི་ནུ་བྷཾ། གལ་ཏེ་སྐྱེས་པ་དང་འདྲ་བར་་་འགྱུར་བ་འབྱུང་ན། །ཨོ་ཡིག་གནས་པར་འགྱུར་ཏེ། །དཔེར་ན་ཨ་ཏི་ནཱ་ཝཱ། ཨ་ཏི་ནུ་ནཱ། ཞེས་ཊཱི་ཀཱར་བཤད་པ་ལ་འགལ་བར་འགྱུར་རོ། །དབྱངས་་།
