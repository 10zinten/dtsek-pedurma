ལ་ནི། ཧེ་པྲཀ། ཧེ་ཕྲག། ཧེ་པྲ་ཛོ། ཧེ་པྲ་ཛ། དེ་བཞིན་དུ། རུཀ་སྟེ༑ས་འདིའི་ཡང་ཙ་སྡེ་ཉིད་ལས་ག་ཉིད་དོ་ཞེས་པ་ཁྱད་པར་རོ། །།།རྩ་བ་བི་བྱུད་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ་རྫོགས་པ་ལ་ཏ་ཡིག་གམ་ད་ཡིག་དག་གོ། །བིདྡྷཱཏ། གློག། བིདྡྷད། བིདྱ་དོ། བིཏྟཱ་ད། བིདྱ་དཾ། བིདྱ་དོ། བིདྱ་ད། བིཅུ་དཱ། དྷུཊ་རྣམས་ཀྱི་གསུམ་པའོ་ཞེས་པས་ད་ཉིད་དོ། །བིཧཱུདྡྷཾ། བིདྡྷཱ་དྷི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། ས་རིཏ། ཏ་ཌིཏ། །ཡོ་ཥིཏ། ལ་སོགས་པ་རྣམས་སོ། །།།་བ་དྲྀ་ཤད་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། རྫོགས་པ་ལ་ཏ་ཡིག་གམ་ད་ཡིག་དག་གོ། །དྲི་ཤཏ། དྲོ་བ། དྲྀ་ཤད། དྲྀ་ཤ་དོ། དྲྀ་ཤ་ད། དྲྀ་ཤ་དཾ། དྲྀ་ཤ་དོ། དྲྀ་ཤ་ད། དྲྀ་ཤ་དཱ། དྲྀ་ཤདྡྷཾ། ཛཾ་ཤདྷི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། ཨཱ་པད། བི་པད། པྲ་ཏི་པད། ཤ་རད། བི་ཥད་ལ་སོགས་པ་རྣམས་སོ། །་@། ཀུ་བ་ནི་ཀྲུ་དྷའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དྷ་ཡིག་ནི་ཏ་ཡིག་གམ་ད་ཡིག་དག་ཏུའོ། །ཀུཏྲ། ཁྲོ་བ། ཀྲུད། ཀྲུ་དྷོ། ཀྲུ་དྷ། ཀྲུ་དྷཾ། ཀྲུ་དྷོ། ཀྲུ་དྷ། ཀྲུ་དྷཱ། །དྷུཊ་རྣམས་ཀྱི་གསུམ་པའོ་ཞེས་པས་དྷ་ཡིག་ནི་ད་ཡིག་ཏུའོ། །ཀུདྡྷཾ། །ཀུདྷིཞེས་པ་ལ་སོགས་པ་སྟེ། བདུན་པའི་མང་ཚིག་ལ་སྒྲ་མེད་པ་ལ་དང་པོའོ་ཞེས་པས་དྷ་ཡིག་ནི་ཏ་ཡིག་ཏུའོ། །ཀུཉྩ། དེ་བཞིན་དུ། ཀུདྷ། ས་མིདྷ། ཡུདྷ། ལ་སོགས་པ་རྣམས་སོ། །པཱ་མན་གྱི་སྒྲ་ལ་ནི། འགྲེལ་པར་གསུངས་པ། ན་མཐའ་ལ་ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཞེས་པ་སྤྱིར་བཏང་བ་ལ་ཨ་ནའི་མཐའ་ལས་ཀྱང་ཞེས་པའི་ཨཱིའི་རྐྱེན་བཀག་པའི་ཕྱིར་པཱ་མན་ཞེས་པ་ན་ཡིག་གི་མཐའ་ཅན་ཉིད་བུད་མེད་ཀྱི་རྟགས་།
