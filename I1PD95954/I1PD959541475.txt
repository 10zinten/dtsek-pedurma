བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་། ཨ་མྱུ་ཡི་ནཱཾ། ཨ་པྱ་ཡི་དྡྷཾ། ཀླུ་པ། གསལ་བྱེད་མཐའ་རྣམས་ཀྱི་ཨིཊ་མེད་པའོ་ཞེས་པས་པ་རསྨིའི་ཚིག་ལ་སི་ཙ་ལ་ནཱ་མིནའི་འཕེལ་བའོ། །དྷུཊ་ལས་ཀྱང་དྷཊ་ལའོ་ཞེས་པས་སི་ཙ་དབྱིའོ། །ཨཀྵི་པཱིཏ། ཀ་:ཨཀཱི་པ། ཨཀཱི་པཱུ། ཏུད་སོགས་སུ་གྲགས་པ་ནི་ཡ་ཛ་སོགས་སོ། ་། ཨཀྵ་པཱིཏ། སིཙ། ཨཱ་ཤཱིཥ་དག་ལ་ཡང་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཨཀིསྟ། ཨ་ཀྭིཔཱ་ཏཱ། ཁིད་ཡོངས་སུ་འཇོམས་པ་ལའོ། །ཨ་ཁི་ཏིཏ། ཨ་ཁེ་:ཏྭཾ། ཁིད་དམན་པ་ཉིད་ལའོ། །སྔ་མ་བཞིན་ཡོན་ཏན་མེད་པ་ཉིད་ལ། ཨ་ཁིཏྟ། ཨ་་ཁིཏྟཱ་ཏྭཾ་ཎི་དྷུ་འགྲོ་བ་ལའོ། །སི་ཙའོ། །ཨིཊ་ཨཱ་ག་མའོ། །ཨ་སེ་དྷཱ་ཏི། སྦ༑སེ་ནི། སེ་དྷ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཤ་ཉིད་དོ། །ནེ་ཥ་དྷཱི་ཏ། ཏི་དྷ། ཡང་ན་ཨིཊའོ། །ཨ་སེ་དྷཱི་ཏ། ཨ་སེ་དྷིཏྟཾ། ཨ་སིཧི་ཏ། ཨ་སིདྡྷཾ། ལི་པ། སི་ཙ། ཨ་སུ། ་བ་། :ཁཱ་ཏི། ལི་པི། སི་ཙ། ཧྭ་ལས་་ཨནའོ་ཞེས་པས་ཨནའོ། །ཨ་ལི་པཏ། ཨ་ལི་པ་ཏཱཾ། ཨ་ལི་པཎ། དེ་ལ་རྣམ་པར་གཞག་པ་ཡང་ན་སྤྱིར་བཏང་ནས་ལིཔ་ལ་སོགས་པ་རྣམས་ཀྱི་ཨཱཏྨ་་་ནེའི་ཚིག་ལ་ཡང་ན་ཨིཎའོ། །ཨ་ལི་པ་ཏེ། ཨ་ལི་བྷ། །ཨ་ལི་པ་ཏྭཾ། ཨ་ལིཔཱ་ཏྭཾ། ཨ་སི་ཙཏ། :：ཨ་སི་ཙཏྭཾ། ཨ་སི་ཙན། ཨ་སི་ཙཏ། ཨ་སི་ཀྟ། ཨ་སི་ཙི་ཏཱཾ། ཨ་སིཀྵ་ཏྭཾ། །དི་ཤ་ཨིཊ་མེད་པ་ཤིཊ་མཐའ་ནཱ་མི། ཨུ་བ་དྷཱ་ཅན་དྲི་ཤ་མིན་པ་ལས་སཎའོ་ཞེས་པས་སཎའི་རྐྱེན་ནོ། ཁོ་། ཎ་ཡིག་ནི་ཎ་ཡིག་རྗེས་སུ་་་འབྲེལ་པ་དང་ཙི་ཀྲི་ཡི་ཏ་དག་ལ་མ་ཡིན་ནོ་ཞེས་པས་ཡོན་ཏན་མེད་པའི་དོན་ནོ། །ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཤ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །ཨ་དཀྵིཏ། ཨ་དིཀྵ་ཏྭཾ། ཨ་དཀློན། ཨ་དིཀྐ། ཨ་དིཀྵ་ཏཾ། ཨ་དིཀ་ཏ། ཨ་དིཀྑཾ། ་ཨ་དིཀྵ་བའོ། །ཨ་དིཀྵ། ཨ་དིཀྵ་ཏ། །
