ར་བསྟན་འགྱུར། སྒྲ་མདོ། ཤིཀི་རི་རཀཱི་ཥ་ཏི། ཛམྦ། ཛི་ཛམྦི་ཥ་ཏི། །ཤནུ། ཤི་ཤྲཱི་སི་ཥ་ཏི། ཙོ་བྷནྡྲ། བི་བྷཾ་སི་ཥ:：ཏི། དྷརྨུ། དི་དྷཾ་སི་ཥ་ཏེ། བ་དི། བི་བནྡི་ཥ་ཏེ། :ཌཱ་ཡཱ་ཙེ༄༅། ཡི་ཡཱ་ཙི་ཥ་ཏེ། །ཡི་ཡཱ་ཙི་ཥ་ཏི། ཁཱ་དྲི། །ཙི་ཁཱ་དི་ཥ་ཏི། ཟླ་དཱི། ཨ་ཛི་གྷཱ་དཱི་ཥ་ཏི། སཱ་ད། སི་སྭཱ་དི་ཥ་ཏེ། ཀཱ་སི། ཙི་ཀཱ་སི་ཥ་ཏེ། བྷ་སི། བི་བ་སི་ཥ་ཏེ། ཥ་ཧ། སྔ་མའི་རྒྱུ་མཚན་ལས་ཥཎ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །སི་ས་ཧི་ཥ་ཏེ། སེ་ཐྲི། སི་སེ་བི་ཥ་ཏེ། ལོ་ཀྲྀ། ལོ་ཙཱི༄༅། ཨ་བ་ལུ་ལོ་ཀི་ཥ་ཏེ། །རྣོ་ཀྲྀ། :རྔུ་དྷོ་ཀི་ཥ་ཏེ། ཙུ་བི། ཙུ་ཙུམྤི་ཥ་ཏི། གནྡྲ། གཞུང་བསྒྲིག་པ་ལའོ། །ཛི་གྲནྡྲི་ཥ་ཏི། ཛཱི་བ། ཇི་ཛཱི་བི་ཥ་ཏེ། ༼ཀྲྀ་ཊཱི། ཙི་ཀཱི་ཊི་ཥ་ཏི། ལི་:ཥ། གསལ་བྱེད་ལ་སོགས་པས་བར་ཆོད་པར་གྱུར་པ་ལ་ཡང་ནའོ། །ཞེས་པ་འདིར་སན་ཡང་ནའོ་ཞེས་བརྗོད་པར་བྱའོ། །ཞེས་པས་་་སན་གྱི་ཡང་ན་ཡོན་ཏན་ཉིད་དོ། །ལི་ལི་ཁི་ཥ་ཏི། ལི་ལེ་:ཁི་ཥ་ཏི། ཁིཊ་་སྐྲག་ཅིཔ་ལའོ། །ཙི་ཁི་ཊི་ཥ་ཏི། ཙི་ཁེ་ཊི་ཥ་ཏི། ཏྲི་མ། ::སྔ་མ་བཞིན་གྱི་་ངེས་པ་ལས་ཥཎ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །ཏི་སྟི་མི་ཥ་ཏི། ་ཏི་སྟེ་མི་ཥ་ཏི། བྱུཏ༑ཤུ་བྷ། རུཙ། །དྷུ་ཏི། སཱ་པི་དག་གི་ཨ་ཧྥུ་སའིའོ་ཞེས་པས་སཾ་པྲ་་་སཱ་:ར་ཎའོ། །།དི་དྷུ་ཏི་ཥ་ཏེ། དི་ངོ་ཏི་ཥ་ཏི། ཤུ་ཤུ་བྷི་ཥ་ཏི། ཤུ་ཤོ་བྷི་ཥ་ཏེ། རུ་རུ་ཙི་ཥ་ཏེ། རུ་རོ་ཙི་ཥ་ཏེ། ཌཱ་གུ་པཱུ། ཨུད་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་ཨིཊའོ། །ཛུ་གོ་པི་ཥ་ཏེ། ཛུ་གུཔྤཱ་ཏི། ཨཱ་ཡའི་རྐྱེན་ལ་ནི། །ཛུ་གོ་པཱ་ཡི་ཥ་ཏེ། ཨིཊ་འགྲོ་བ་ལའོ། །ཨིཊ་ཨཱ་ག་མའོ། །ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །དབྱངས་དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་གཉིས་བརྗོད་དོ། །ཨི་ཀྱི་ཊི་ཊི་ཥ་ཏི། ཨི། ཨི་ཏི་ཥ་ཏི། པྲ་ཏི་ཥ་ཏི། །ཨུ་ཁ་འགྲོ་བའི་དོན་ལའོ། །ཨོ་ཙི་ཁི་ཥ་ཏི། དི་བུ། ཨི་ཡིག་དང་རིདྷ་ཞེས་པ་ལ་སོགས་པས་ཡང་ན་ཨིཊའོ། །།
