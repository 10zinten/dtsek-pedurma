བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་འགྱུར་བའི་ཕྱོགས་ལ། ་ན་དཱི་ལས་ཨེ་ཨཱས་ཨཱས་ཨོཾ་ཞེས་པ་ཨེ་དང་ཨཱས་ཨཱས་དང་་ཨོཾ་ཞེས་པ་འདི་རྣམས་སུ་འགྱུར་ཏེ། གཞན་ལ་ཨགི་དང་འདྲའོ། །བྷུ་མྱི་འི། བྷུ་མ་ཡེ། །བྷུ་མི་བྷཾ། ་་:བྷུ་མི་བྷུཉྩ། བྷུ་མྱུཡེ། བྷུ་མེ། བྷུ་མི་བྷཾ། ་མི་བྱཱ། བྷཱུ་མཱུ། བྷུ་མེ། བྷཱུ་མྱོ། །བྷཱུ་མི་ནཾ། བྷུ་མོ། བྷཱུ་མྱོ། བྷཱུ་མི་ཁུ། །་ཀུན་དུ་བོད་པ་ལ་ཨགྷི་དང་འདྲ་སྟེ། ཧེ་བྷུ་མེ། ཧེ་བྷཱུ་མཱི། ཧེ་བྷཱུ་མ་ཡ༄༅༈དེ་བཞིན་དུ་སྐུ་ཏི། :ཤི་རོ་དྷི། ཀ་རོ་ཊི། ཊ་ཀ་ཊི། སྦི་ཏི། དྷཱ་ར་ཎི། :ཨོ་ཥ་ཏི། པ་དྷ་ཏི། པོ་ཏྟི་རྣམས་དང༌། ::ཝ་ས་:ཏི། སྲིཎྜི། བི་སྲིདྡྷི། ཁ་ནི། ཀླུ་ཏི། ས་ར་ཎི། ཧཱ་ནི། ག་ཏི། ཀྱི་ཏི། །པཀྵ་ཏི། །རུ་ཙི། ན་ཏི། བི་ཀྲྀ་ཏི་དག་དང་། །ཨིཏྟི། ཀོ་ཊི། དཆི་བྷཾ་ཀུ་ཊི། ར་ཏི། བྷུ་ཏི། ཤནྟི། པ་ལི་རྣམས་ཀྱང་སྟེ། བི་མ་ཏི། ::::ཚིགས་ཏི། མ་སི། པྲ་བྷི་ཏི། བེ་དི། ཚོབྲྀ་ཏི། ་དུ་ལི། དུནུ་བིི། དྷུ་ལི་ནཱི་ཏི་ཞེས་པ་ལ་སོགས་པའོ། །།ཨཱི་ཡིག་གི་མཐའ་ཅན་ན་དཱིའི་སྒྲ་སྟེ། །ན་དའི་སྒྲ་ལས་ན་ད་སོགས་:ཨ་རྡུ་བཱ་ཙེཧ་ཨུ་ཨི་ཨནྟ་རཾ་ས་ཁི་ན་མཐའ་རྣམས་ལས་ཨཱིའོ་ཞེས་པས་ཨཱི་ཨེརྐྱེན་ནོ། །བུད་མེད་དུ་བྱས་པའི་ཨཱི་ཡིག་ལ་ཨ་དབྱི་བར་བྱའོ་ཞེས་པས་ཨི་ཡིག་དབྱི་བར་ཨེའགྱུར་རོ། །།ན་དཱི་ཞེས་པ་སེགནས་པ་ལ། །::སི་ལ་་སོགས་པ་རྣམས་སོ། །ཨཱི་ཡིག་གི་མཐའ་ལས་སིའོ་ཞེས་པས་སི་དབྱིའོ། །ན་དཱི། ཨིའི་ཡི་གེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཡའོ་ཞེས་པས་ཡ་ཉིད་དོ། །ན་བྱོ། །ན་༑་ཨམ་ཤས་དག་གི་དང་པོ་དབྱིའོ་ཞེས་པས། ཨམ་ཤས་དག་གི་དང་པ་དབྱི་བར་འགྱུར་རོ། །ན་དཱི། ནདྨོ། ན་དཱི། ནངྒལཾ། ན་དཱི་ཧཱུཾ། ན་དཱི་བྷི། །ན་དཱི་ལས་ཨི་ཨཱས་ཨཱས་ཨོའོ་ཞེས་པ་དེ་དང་ང་སི་དང་ངས་དང་ང་རྣམས་ཨི་ཨཱས་ཨཱས་:ཨོཾ་ཞེས་པ་འདི་རྣམས་སུ་བསྒྱུར་བ་རིམ་པ་བཞིན་དུ་འགྱུར་རོ། །།
