དགག་པ་ངེས་པར་བསྟན་པ་མི་རྟག་པ་ཉིད་ཀྱི་ཕྱིར་སྔ་མ་བཞིན་ཏི་ཉིད་དོ། །ཀ་ཉིད་དོ། །བྷཎིབྦི། བརྩེགས་པའི་དང་པོས་དབྱིའོ། །བྷསྔ་ལ་བྷིཛ་ཉེ་བར་མཚོན་པ་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །བྷཀྵིབྦི། །ཡ་ཛཱ་སཱ་པི་བ་ཙི་ཞེས་པ་ལ་སོགས་པས་ཡོན་ཏན་མེད་པ་ལ་སཾ་པྲ་སཱ་ར་ཎའོ། །ཨིཛཱཏ། བྷིཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །ཡགིཋ། རྔུ་མསྔོ། །མཛཱད། སྲིཛ། སཾ་ཛཱཏ། དིབ་སོགས་དང་རུ་ཙ་སོགས་སོ། །།སྔ་མ་བཞིན་ཥ་ཉིད་དོ། །སྲ་གླིཤྩ། །ཎི་ཛིར། ནི་ཛཱཏ། ཀི་ཉིད་དོ། །ལ་། ནིàཋ། ཥཱཉྫ། སྭ་གླིཋ། ཨད། །ཨདྷཱཏ། སི་ཏུད། །ཏུདྡྷཱཏ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤཱིཥམ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཏུནཱིཏྠ། ཏུཏ། ནུ་དྷཱཏ། ཀླུ་དིར། དྷཱཏ། དྷཱིཏྲ། ཤཱི་ཙྪེད། སཱིདྡྷཱཏ། བིད་ཡོད་པ་ལའོ། །བིཏིཥྛ། བི་དྷི། བི་དྷཱཏ། བི་་ཏིཥྛ། །བིད་རྣམ་པར་དཔྱོད་པ་ལའོ། ་། བིཏི། ཚི་དིར། ཚིདིཉྩ། བྷི་དིར། ་བི་ཉིདྷཱཏ། བྷིཏི། ཧ་ད། ཧཏི། །ཤ་དྲི། ཤདྡྷཱཏ། པད། །པཏིཥྛ། སྐནྡིརཏི། སྐནྡྷཱཏ། །ཁི་ད། །ཡོངས་སུ་འཇོམས་པ་ལའོ། །།ཁི་་དྷཱཏ༑། ཁི་ད་ཞེན་པ་ཉིད་ལའོ། །ཁིཏཱི། རཱ་དྷ། ་ར་དྷཱཏེ། རུ་དྷིར། རུ་དྷཱཏ། རུཧཱིཊྚ། ཀྲུ་དྷ། ཀྲུ་དྷཱཏ། གླུ་དྷ། །ཀླུ་དྷཱཏ། འབནྡྷ། ཌབནྡྷཱཏེ། ཤུ་དྷ། ཤུ་དྷཱཏ། ཎི་དྷུ་ཡང་དག་པར་གྲུབ་པ་ལའོ། །།སཱི་དྷཱཏ། བུ་དྷ་་་རྟོགས་པ་ལའོ། །དིབ་སོགས་སོ། །བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་་་་མཐའི་ཞེས་པ་ལ་སོགས་པ་དང་པོ་བའི་བཞི་པ་བྷའོ། །བྷུ་ཏིཥྛ། །་ཡུ་དྷ། ཡུཏིཥྛ༑་བྱཱ་དྷི། །གྲ་ཧི་ཛྷ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །བི་དྷཱཏ། སཱ་དྷ། སཱ་དྷཱཏ། ཧ་ན་ཧནྟིའོ༄༅། །ཨ་ཤིཥ་ལ་བ་དྷིའོ། །ཞེས་པས་བི་དྷི་་རུ་བསྒྱུར་རོ། །བ་དྷཱཏ། ཨཱད་ལས་ཡམ་དང་ཧན་དག་ཀྱང་ངོ༌། ?། ཞེས་པས་།
