ཏྲི་ཏ་ཡེ། ཏྲི་ཏ་ཡ། །ཕྱོགས་གཅིག་ལ་ནི་དཱི་དང་ཏྲི་དག་ལས་ཡའི་རྐྱེན་དང་། ཨིའི་ཡི་གེ་དང་ཨའི་ཡི་གེ་དག་ལ་དབྱིའོ་ཞེས་པས་ཨི་ཡིག་དབྱིའོ། །དྭ་ཡེ། དྭ་ཡཱ༑ཨོཏྲ་ཡེ། ཏྲ་ཡཱ། ཙ་ཏུཊྚ་ཡེ། བཞི་རྣམས། ཙ་ཏུཏྟཱ་ཡཱ། པཉྩ་ཏ་ཡེ། ལྔ་རྣམས། པཉྩ་ཏ་ཡཱ། །ཀ་ཏི་པ་ཡེ། དྲུག་རྣམས། ཀ་ཏི་པ་ཡཱ། ཨབྷེ། ཕྱེད་རྣམས། ཨཛྷ༑གྲངས་ལས་རྫོགས་པ་ལ་ཌ་མ་དག་གོ་ཞེས་པ་སྤྱིར་བཏང་ནས། དཱིའི་སྒྲ་ལས། དྲི་ཏད་ལས་ཏཱི་ཡའོ་ཞེས་པས་ཏཱི་དེ་ཕན་ཡ་རྐྱེན་དང་། ཏྲིའི་ཀྲྀ་ཡང་་ངོ་ཞེས་པས་ཏྲི་རུ་ཡང་བསྒྱུར་བར་འགྱུར་ཏེ། ཏཱི་ཡ་ལ་གཉིས་བརྗོད་པར་བྱའོ་་་ཞེས་པས་ངེ་སྨིར་འགྱུར་རོ། །དྲི་ཏཱི་ཡ་སྨི། གཉིས་ཀྱི་ཆེད་དུ། དྲི་ཏཱི་ཡཱ་ཡ། ཀྲྀ་ཏཱི་ཡསྨི། ཀྲྀ་ཏཱི་ཡཱ་ཡ། ་ཏཱི་ཡ་བྱིན་པ་ལ་ལུགས་གཉིས་བརྗོད་པར་བྱ་སྟེ། ཡང་ན་་ང་སི་དང་རི་སྨཏ་སྨིན་དག་གོ། :། དྲི་ཏཱི་ཡ་སྨཏེ། དྲི་ཏཱི་ཡཱཏ། ང་སི་ཨཏའོ། །ཏྭཾ་ཏཱི་ཡ་སྨཏ། ཏྲི་ཏི་ཡཱཏ། དྲི་ཏཱི་ཡ་སྨིན། དྲི་ཏཱི་ཡི། ཀྲྀ་ཏཱི་ཡ་སྨིན། ཀྲྀ་ཏཱི་ཡེ༑བུད་མེད་ལ། དཱི་ནག་ཏི་ཡ་དང༌། །ནྲྀ་ཏཱི་ཡ་དག་ལས་ཡང་ནའོ་་ཞེས་པའི་རྣམ་པར་བརྟག་ང་ལྡན་རྣམས་པས། ཡི་ཡཱས་ཡཱས་ཡཱམ་རྣམས་སུ་ཨཱ་ག་མ་དང་ལྷན་ཅིག་པ་འགྱུར་ཏེ་སྔ་མ་ཐུང་བ་འང་ངོ༌། །ཏཱི་ཏཱི་ཡ་སྤྱི། ངེ་། །དཱི་ཏཱི་ཡ་ཡི། ངེ་། ཀྲྀ་ཏཱི་ཡ་སྤྱི། ཀྲྀ་ཏཱི་ཡ་ཡི། དཱི་ཏཱི་ཡ་ང་སི་སྣུ། དཱི་ཏཱི་ཡཱ་ཡཱ། ང་སི། དྲྀ་ཏཱི་ཡ་སྣུ། ཀྲྀ་ཏཱི་ཡཱ་ཡཱ། དཱི་ཏཱི་ཡ་སྣུ། ངས། དཱི་ཏཱི་ཡཱ་ཡཱ། ངས། ཀྲྀ་ཏི་ཡ་སྣུ། །ཏྲི་ཏཱི་ཡཱ་ཡ། དཱི་ཏཱི་ཡ་སཱུ། བི། དཱི་ཏཱི་ཡཱ་ཡཱ། པི། ཀྲྀ་་་ཏཱི་ཡ་ཧཱུཾ། ཀྲྀ་ཏཱི་ཡཱ་ཡཱ། ལྷག་། །ལྔ་པ་ལས་ཁྱད་ལ་སོགས་པ་རྣམས་བརྗོད་པར་བྱ་སྟེ། ཁྱད་ལ་སོགས་པ་རྣམས་ཀྱི་་་རྣམ་དབྱེ་ལ་ཨའོ་ཞེས་པས་ཨ་ཉིད་དུ་འགྱུར་ལ། ཨ་ཡིག་ལ་དབྱིའོ། །ཨ་ཡིག་དབྱི་ཞིང་། སི་ལ་སའོ་ཞེས་པའི་རྗེས་སུ་འཇུག་བཞིན་པ་ལ། ཏའི་འང་ངོ་།
