:ལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི་༌། དང་ལས་ཕ་རོལ་ཨུ་ཉིད་དོ། །ཨ་མཱུ་ཧཱུཾ། ཏྱད་ལ་སོགས་པའི་ཨ་ཉིད་དུ་བྱས་པ་ན། དེ་ལ་བྷིས་བྷིར་རོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ཨ་དས་ལས་ཀྱང་ངོ་ཞེས་པས་བྷིར་་དུ་བསྒྱུར་ཞིང་། དྷཊ་མང་པོ་ཉིད་ལ་ནི་ཨེའོ་ཞེས་པས་ཨེ་ཉིད་དེ། མང་པོ་་ཉིད་ལ་ཨེཏ་ཨཱི་རུ་འགྱུར་རོ། །ཨ་མཱི་བྷི། ཏྱད་ལ་སོགས་པའི་ཨ་ཉིད་དུ་བྱས་པ་ན། སརྦ་ནཱ་མ་ལས་སྨིའོ་ཞེས་པས་སྣེ་རུ་འགྱུར་ཞིང༌། མ་ཉིད་དང་ཨུ་ཉིད་ཀྱང་ངོ༌། །ཨ་མུ་རྨི། ཨ་མཱུ་ཧཱུཾ། །ཨ་མཱི་བྷུ། ཨ་མུ་ཉྩ། ཨ་མུ་ཡོ། ཨེཏ་ཨཱི་རུ་འགྱུར་ཏེ། ཨ་མཱི་ཏཱཾ། ཨ་མུསྨིན། ཨ་མུ་ཡོ། ་ཨ་མཱི་ཏེ། བུད་མེད་ལ། །ཨ་སོ། །་གཉིས་ཀྱི་ཚིག་ལ་སོགས་པ་རྣམས་ལ། ་ཁྱད་ལ་སོགས་པའི་ཨ་ཉིད་དུ་བྱས་པ་ན། །རྒྱུ་མཚན་ལ་ཉེ་བར་སྐྱེའོ་ཞེས་བྱས་པ་ཉིད་ཀྱིས་བུད་མེད་ལ་ཨ་ལས་ཨཱ་སྟེ། ཨོ་ཨཱི་དང་། ་ཨའི་ཡི་གེ་ཨིའི་ཡི་གེ་ལ་ཨེ་སྟེ། ་ཕ་རོལ་དབྱི་ཞིང་། ད་ཡིག་ནི་མ་ཉིད་དང༌། མ་ལས་ཕ་རོལ་ཨུ་ཉིད་དོ། །ཨ་མཱུ།  ོཨ་་མཱུ༑ཨ་མཾ། ཨ་མཱུ། ཨ་མཱུ། ཊཱ་ཨོས་དག་ལ་ཨེ་ཉིད་དེ། ཨེ་ཨ་ཡ་པ་དང་མ་ཉིད་དོ། །དེ་ལ་ནང་གི་དེ་ལས་ཨ་ཡིག་ཐུང་ངུ་དང་། ཨུ་ཡིག་ཐུང་ངུའོ། །ཨ་མུ་ཡཱ། །རིང་པོ་སྟེ། ཨཱ་ཡིག་ཏུ་གནས་པ་ན། ཨུ་ཙྪཱཡིག་གོ། །ཨ་མཱུ་ཧཱུཾ། ཨ་མཱུ་བྷི། ང་ལྡན་གྱི་བརྗོད་པ་རྣམས་ནི་སུ་དང་ལྷན་ཅིག་ཡི་ཡཱས་ཡཱས་ཡཱམ་་རྣམས་སུ་འགྱུར་ཞིང་། སྔ་མ་ཐུང་བའང་སྟེ། མ་ཉིད་དང་ཨུ་ཉིད་ཀྱང་ངོ༌། །ཨ་མུ་བྱེ། ཨ་མཱུ་ཧཱུཾ། ཨ་མཱུ་ཧྥུ། ཨ་མཱུ་དྡྷེ། ཨ་མཱུ་ཧཱུཾ། ཨ་མཱུ་པཿཨ་མཱུ་དྡྷེ། ཨ་མུ་ཡོ། ཨ་མུ་ཏྟཾ། ཨ་མུ་ཧཱུཾ། ཨ་མུ་ཡོ། ཨ་མཱུ་ཏུ། མ་ནིང་ལ། །སི་དང་ཨམ་དག་དབྱི་བ་བྱས་པ་ན། དེ་ལ་བརྗོད་པ་བཀག་པའི་ཕྱིར། ས་ཡིག་ནི་རྣམ་པར་གཅད་བྱའོ། །།ཨ་ད། ཀ་ཏྱད་ལ་སོགས་པའི་ཨ་ཉིད་དེ། ཨོ་ཨཱི་དང༌། ཨའི་ཡི་གེ་ཨུའི་ཡི་གེ་ལ་ཨེའོ། །ཨ་མཱུ། ཤི་རུ་བསྒྱུར་ཞིང་དྷུཊ་་།
