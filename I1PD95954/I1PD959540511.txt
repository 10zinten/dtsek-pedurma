་་། ཀྲྀཉའོ༑། །ཨུདྡོ་ཙ་ཀཱ་ར། ཨུཌ་ཀུན་ནས་ཉེད་པ་ལའོ། །ཨཊའོ། །།ཨཱམའོ། །ནཱ་མི་དང་པོ་ཞེས་པ་ཅི་ཞེ་ན། ར་རཀྟ། ལྕི་བ་ཉིད་ནི་དེར་བརྩེགས་པ་ལ་ལྗི་་བའོ་ཞེས་པའི་ཚིག་ལས་སོ། ་། ལྗི་བ་ལྡན་པ་ཞེས་པ་ཅི་ཞེ་ན། ཨ་ཧ་མི་ཡ་ཛ། ཡ་ཛི་ལ་ཕྱི་མའི་གཅིག་ཚིག་ཨཊའོ། །གཉིས་བརྗོད་དོ། །ཨ་བྷ་སའི་པ་རོཉཀྵ་ཀྱི་ལ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ།  ོ། འོ་ན་འདིར་ཨཊ་ལ་ཕྱོགས་གཅིག་ལ་ཨའི་ཨུ་པ་དྷཱ་ཞེས་པས་རིང་པོའོ། །མིང་སྔ་མ་ཅན་གྱི་:སྒྲུབ་པ་མི་རྟག་ཅེས་པའི་ཚིག་་་་ལས་སོ། །འདིར་ཅིའི་ཕྱིར་མི་འགྱུར། །ཨི་ཡེ་ཥ། ཨི་ཏུ་འདོད་པ་ལའོ། །ཨཊའོ༑། གཉིས་བརྗོད་དོ། །།ནཱ་:མིའི་འང་ཨུ་པ་དྷཱའི་ཞེས་པས་ཡོན་ཏན་ནོ། །ཨ་ངྷ་སའི་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པའི་ཞེས་པས་ཨི་《ཡའོ་ཞེ་ན། །འདིར་བརྗོད་པར་བྱའོ་། ལ། ཉེ་བར་ལྷུང་བའི་མཚན་ཉིད་ཀྱིས་སྒྲུབ་པའི་རྒྱུ་མཚན་དེ་མི་རྟག་གོ་ཞེས་པའི་ཚིག་ལས་མི་འགྱུར་རོ། །རྐྱེན་དང་འབྲེལ་པ་ཅན་གྱི་གང་ཉེ་བར་ལྷུང་པས་བར་མ་ཆད་དུ་བྱིངས་ཀྱི་ཡོན་ཏན་བྱས་པ་ལ་ལྕི་བ་ཉིད་ཐོབ་པ་ལ་་་དེའང་ཐོབ་པ་མ་ཡིན་ཏེ། ་ཨཱི་གྱི་རྐྱེན་བྱས་ནས་བྱིངས་ཀྱི་རྐྱེན་བར་མ་ཆད་དུ་བྱས་པའི་ཕྱིར་རོ། །རིཙྪ་མ་ཡིན་པ་ཞེས་པ་ཅི་ཞེ་ན། །ཨཱ་ནནྡྷུ། ཨཱ་ནནྡྷུ་ཏུ། རིཙྪའི་ཨིན་འགྲོ་བ་ལའོ། །།།ཨཊ་དང་ཨ་ཏུསའོ། །ལན་གཉིས་བརྗོད་དོ། །རིའི་ཡི་གེ་ཨ་ཡིག་གོ། །དང་པོའི་ཨའི་ཐམས་ཅད་དུའོ་ཞེས་པས་རིང་པོའོ། །རཾ་ཡིག་ལའང་ངོ་ཞེས་པས་ན་ཨཱ་ག་མའོ། །རིཙྪའི་རིཏའི་འོ་ཞེས་པས་ཡོན་ཏན་ནོ༑། ཨུཎོ་་ཏིའི་དགག་པ་བཤད་པར་བྱ་སྟེ། བྲོརྫུ་ནཱ་བ། ཨུཎྱཉ་འགེབས་པ་ལའོ། །བྲ་སྔ་མར་རོ། །ཨཊའོསྟེ། །ཙཎ་པ་རོཉཀཱ་ཞེས་པས་ནུ་གཉིས་བརྗོད་དོ། །ན་བ་ད་ར་རྣམས་ཞེས་པས་རེ་ཕའི་མིན་ནོ། །ཨའི་ཨུ་པ་དྷ་ཞེས་པས་འཕེལ་བའོ། །ཨཱ་སི། ད་ཡི། ཨ་ཡི། :：ད་རི་དྲ་རྣམས་ལས་།
