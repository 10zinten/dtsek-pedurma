རྩ་ལྟ་བ་དི་སི་ཞེས་པ་ཅི་ཞེ་ན། ཨསྟི། ཏི་ངེས་པར་བསྟན་པ་ལ་བྷུ་རུ་བསྒྱུར་བར་བྱས་པ་ན་མི་འགྱུར་ཏེ། ཨ་བྷཱུཏ། »་:ཨ་བྷིུ། ཨི་ཎ་སྦ་དཱ་ཞེས་པས་སི་ཙ་འཇིག་གོ། ཚེ་། སི་ཙ་ལས་སོ། །སི་ཙ་ལས་ཕ་རོལ་གྱི་དི་སི་དག་ལས་དང་པོར་ཨཱིཊ་ཨཱ་ག་མ་འགྱུར་རོ། །:ཨ་ཀཱནཱ། ཨ་ལཱ་བྷིཏེ། ཨ་ལཱ་བྷི༄༅། ་རུད་སོགས་རྣམས་ལས་ཀྱང་ངོ༌། །མཚོན་པར་བྱ་བའི་ངོར་རུད་སོགས་:ལྔ་རྣམས་གཟུང་བར་བྱ་སྟེ། རུ་དིར་མཆི་མ་འཛག་:པ་ལའོ། །ཉི་སྭཱ་པ་ཉལ་བ་ལའོ། །ཤཱས་སྲོག་འཛིན་པ་ལའོ། ་། ཨ་ནའང་ངོ༌། །ཛ་ཀྵ་ཟ་བ་ལའོ། །འདི་རྣམས་ལས་ཕ་རོལ་གྱི་དི་སི་དག་ལས་དང་པོར་ཨཱིཊ་ཨཱ་ག་མ་འགྱུར་རོ། །རུད་སོགས་་་་ལས་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་ཨིཊ་ཐོབ་པ་ལའོ། :། ཨ་རོ་དཱི་:ཏ། ཨ་རོ་དཱི། ཨ་སཱ་པཱི་ཏ། །ཨ་སཱ་པཱི། ཨ་ཤཱ་སཱིཏ། །ཨ་ཤཱ་སཱི། :པྲ་ཎཱིཏ། པྲ་ཎི། །ཨ་ཛ་བྷཱི་ཏ། ཨ་ཛ་ཊཱི། ལྔ་རྣམས་སྨོས་པ་ལས་འདིར་མི་འགྱུར་ཏེ། ཨ་ཛཱ་གབྷ་བཱན། ཡོན་ཏན་ནོ། །གསལ་བྱེད་དི་སི་དག་ཅེས་པས་དི་དབྱིའོ། །ཨད་ལས་ཨཊའོ། །ཨད་ཟ་བ་ལའོ་ཞེས་པ་འདི་ལས་ཕ་རོལ་གྱི་དི་སི་དག་ལས་་་དང་པོར་ཨཱིཊ་ཨཱ་ག་མ་འགྱུར་རོ། །ཊ་ཡིག་ནི་རྣམ་པ་སྣ་ཚོགས་པའི་དོན་ཏོ། །:ཨཱ་ད་ཏ། ཨཱ་ད། ཨའི་ཡི་གེའི་:ཨཱ་ཡིག་གོ་ཞེས་པས་ཨཱ་ཡིག་གོ། །།རུད་སོགས་ལས་ཀྱང་འདོད་དེ། ་ཨ་རོ་དཏ། ཨ་རོ་ད། ཨ་སཱ་ཎིཔཏ། །ཨ་སཱ་་པ༑་ཨ་ཤཱ་སཏ། ཨ་ཤཱ་ས། །པྲ་ཎ་ཏ། པྲ་ཎ། ཨ་ཛ་ཀྑཏ?། ཨ་་ཛ་ཀྑ། སའི་ས་ལ་བྱིངས་ཐམས་ཅད་པ་མིན་པ་ལ་ཏའོ། །།སའི་ཏེ་ཨེས་ཡིག་དང་པོ་ཅན་བྱིངས་ཐམས་ཅད་པ་མིན་པ་ཕ་རོལ་ལ་ཏ་ཉིད་དུ་འགྱུར་རོ། །:བཉྩ་ཏི༑ཚེ་བས་གནས་པ། བི་བཏྠ་ཏི། སནའོ། །ཛིགྷཏྠ་ཏི། }ཨ་རིའི་སན་ཨ་རྱ་ཏ་ནཱི་དག་ལ་གྷསླའོ་ཞེས་པས་གྷསློའོ། །གཉིས་བརྗོད་དོ། །སའི་ས་ལ་་་—།
