།སྐུ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཟླ་ཞིང་། ས་ཡིག་ནི་རྣམ་པར་གཅད་བྱའོ། །ཨ་ཧ། ྋསྔ་མ་བཞིན་དུ། ཨ་བྷི། ཨ་ཧ་ནཱི། ཨ་ཧཱ་ནི། སླར་ཡང༌། ཨ་ཧ། ཨ་བྷི་ཨ་ཧཱ་ནི། །ཨ་དབྱིའོ། །ཨ་དྷཱ། ཨ་ཧན་གྱི་རེ་ཕ་མ་ཡིན་པ་ལའོ་ཞེས་གང་གསུངས་པ་ནི། རྣམ་དབྱེ་་གསལ་བྱེད་ལ་འགྱུར་བར་འགྲེལ་པས་བཤད་དེ། དེས་ན་ཨ་དང་སྒྲ་ལྡན་དག་གིའང་ངོ་ཞེས་པས་ཨུ་ཉིད་དུ་འགྱུར་ལ། ཨུའི་ཡི་གེ་ལ་ཨོ་སྟེ་ཕ་རོལ་ཡང་དབྱིའོ། །ཨ་ཧོ་ཧཱུཾ། ་ཨ་ཧོ་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །སྤྲིངས། །དེ་བས་བྱ། ང་། ཀརྨན་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ་སི་དང་ཨམ་དག་དབྱི་ཞིང༌། ན་དབྱི་བའང་ངོ༌། །ཀརྨ། ཨོ་ཨཱི་ཞེས་པར་བྱས་པ་ན། །ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལ་ཨན་གྱི་ཨ་དབྱི་ཞེས་སོགས་ལ། །ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཞེས་པ་བཀག་པའི་ཕྱིར། ང་དང་ཨཱི་དག་ལ་ཡང་ནའོ་ཞེས་རྣམ་པར་བརྟག་པས་ཨ་དབྱི་བ་མི་འགྱུར་ཞིང་། ར་དང་ཥ་དང་རིའི་ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་སོགས་པས་ཎ་ཉིད་དོ། །ཀརྨ་ཎཱི། ཀརྞ་ཎི། སླར་ཡང་། ཀརྨ། །ཀརྨ་ཎཱི། ཀརྨ་ཎི། གསུམ་པ་ལ་སོགས་པའི་དབྱངས་ལ་ཨ་དབྱི་བ་མེད་དེ། །ཝ་མ་་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཞེས་པ་བཀག་པའི་ཕྱིར་རོ། །ཀརྨ་ཎཱ། ཀརྨ་ཧཱུཾ། ཀརྨ་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། ཌ་སརྦན། ཙརྨན། ?ཝ་རྨན། ཡོདྷརྨན་ལ་སོགས་པའོ། །ལ་། བཎ་བོན་ཞིས་ཁོག་ཞུགས་བྱ་དྲོ་པ་ངསཟིས་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། །སི་དང་ཨམ་དག་དབྱི་ཞིང་་ས་ཡིག་ནི་རྣམ་པར་གཅད་བྱའོ། །སའི། མར། ཨོཨོ་ཨཱི་དང་། ཥ་ཉིད་དོ། །སཇི་ཏཱི། མར་དག། ཛས་ཤས་དག་ཤིའོ་ཞེས་པས་ཤི་རུ་བསྒྱུར་ཞིང༌། དྷཊ་དང་དབྱངས་ལས་གྱུཊ་ལ་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ།  ོ། ཥ་ཉིད་དེ། མ་གྲུབ་པ་ནི་བ་ཧིའི་ཡན་ལག་དང་མནྟའི་ཡན་ལག་གོ་ཞེས་མ་གྲུབ་པ་དང་འདྲ་བར་འགྱུར་བ་་།
