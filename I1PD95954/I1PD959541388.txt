སོགས་ལས་ཨུའོ། །ཨན་བི་ཀ་ར་ཎའི་དམིགས་བསལ་ལོ། །m། ནཱ་མི་མཐའི་ཡོན་་ཏན་ནོ། །།ཏ་ནོ་ཏི། ཏ་ནུ་ཏ། ་ཨུའི་ཡི་གེ་བའོ་ཞེས་པས་བ་ཉིད་དོ། །ཏ་ནཱནྟི?༑ཏ་ནོ་ཏི། ཏ་ནུ་ཐ། །ཏ་ནུ་ཐ། ཏ་ནོ་མི། ཨུ་ཡིག་བ་མ་དག་ལ་ཡང་ན་དབྱིའོ་ཞེས་པས་དམིགས་བསལ་མེད་པ་བཞིན་འགྱུར་བ་ཡིས་བི་ཀ་ར་ཎའི་ཨུ་ཡིག་ཡང་ན་དབྱིའོ། །ཏ་ནུ་བ། ཏ་ནཱ། ཏ་ནུ་མ། ་ཏམྨ། ཏ་ནུ་ཏེ། :ཏ་ནཱ་ཏེ། ཏ་ནཱ་ཏེ། ཏ་ནུ་ཥ། ཏ་ནཱ་ཐེ། ཏ་ནུ་དྷེ། ་ཏ་ནེ། ཏ་ནུ་བ་ཧེ། །ཏ་ནཱ་ཧེ། ཏ་ནུ་མ་ཧེ། ཏརྨ་ཧེ། །སཾ་ཏ་ནོ་ཏི། སཾ་ཏ་ནུ་ཏེ། བི་ཏ་ནོ་ཏི། བི་ཏ་ནུ་ཏེ། ཨཱ་ཏ་ནོ་ཏི། ཨཱ་ཏ་ནུ་ཏེ། བྱཱ་ཏ་ནོ་ཏི། བྱཱ་ཏ་ནུ་ཏེ། ཥ་ཎུ་སྦྱིན་པ་ལའོ། །བྱིངས་ཀྱི་དང་པོ་ཥའི་སའོ། །རྒྱུ་མཚན་མེད་་ན་རྒྱུ་མཚན་ཅན་གྱི་ཡང་མེད་དོ་ཞེས་པས་ཎའི་ན་ཉིད་དོ། །ས་ནོ་ཏི། ས་ནུ་ཏེ། ཀླུ་ཆུ་འཚེ་བ་ལའོ། །ཀྱི་ཎོ་ཏི། ཀླུ་རྟུ་ཏེ། རཾ་ཎུ་འགྲོ་བ་ལའོ། །རྀ་ཎོ་ཏི། རི་ཆུ་ཏེ། ཀྲྀ་*ཆུ་ཟ་བ་ལའོ། །ཀྲྀ་ཎོ་ཏི། ཀྲྀ་ཊུ་ཏེ། ཤྲཱི་རྔུ་འབར་བ་ལའོ། །ཤྲཱི་ཎོ་ཏི། ་གྲྭ་ཆུ་ཏེ། །ཏ་ན་སོགས་པ་ཉེ་བའི་མཚན་ཉིད་ལ་ཀ་རོ་ཏིའི་འོ་ཞེས་འདོད་དོ། །རི་ཆུ་ལ་སོགས་པའི་ཡང་རང་གི་བི་ཀ་ར་ཎ་ལ་ཡོན་ཏན་ནོ། །ཨཎྞོ་ཏི། ་ཨནྡྷ་ཏེ། གྷནྡྷོཏི། ་གྷནྡྷུ་ཏེ། ་ཌཱ་ཀྲྀཉ་བྱེད་པ་ལའོ། །།ཉ་་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །ཀ་རོ་ཏིའིའོ་ཞེས་པས་རང་གི་བི་ཀ་ར་ཎ་ལ་ཡོན་ཏན་ནོ། །བི་ཀ་:ར་ཎའི་ཨེཔ་རསྨིའི་ཚིག་ལ་གཅིག་གི་ཚིག་་རྣམས་ལ་སྔ་མ་བཞིན་ནོ། ། ོ། ཀ་རོ་ཏི། བྱིངས་ཐམས་ཅད་ཡོན་ཏན་མེད་པ་ཨའི་ཨུ་ཡིག་གོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ལ་ཨ་ཡིག་གི་ཨུ་ཡིག་གོ། །ཕྱོགས་གཅིག་རྣམ་པར་འགྱུར་བའི་གཞན་འགྱུར་བ་མ་ཡིན་པ་ཡང་ངོ༌། །སླར་ཡང་ཀ་རོ་ཏིའི་འོ་ཞེས་པས་ཨུ་ཡིག་གི་ཡོན་ཏན་མེད་པ་:ཉིད་དོ། །ཅིརང་གི་རང་བཞིན་c།
