ཏྭཾ༑ཨ་ཀོ་ཥ་ཏ།  ོ་གང་ཀུཊ་སོགས་ཡིན་པ་དེ་ལས་ནི་ཕ་རོལ་གྱི་སི་®ཙའི་ཡོན་ཏན་མེད་པ་ཉིད་དོ་ཞེས་པས་ཡོན་ཏན་མེད་དོ། །ཐུང་ངུ་ལས་ཀྱང་ཨིཊ་མེད་པའིའོ་ཞེས་པས་སི་ཙ་དབྱིའོ། །ཨ་ཀུ་ཏ། །ཨ་ཀུ་ཥཱ'ཏྭཾ། ཨ་ཀུ་ཥ་ཏ། ཨུང་སྒྲ་ལའོ། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །།ཨོཥྛ། ་ཨོ་ཥཱ་ཏྭཾ། །ཨོ་ཥ་ཏ། ལུཉ། སི་ཙའོ། །།ཨིཊ་ཨཱ་ག་མའོ། །ཨཱུ་ཡིག་གི་་འཕེལ་བ་ཨོ་ཡིག་གོ། །ཨ་ལཱ་བཱིཏ། ཨ་ལཱ་བིཏྟཾ། ཨ་ལ་བི་ཥཱ། ཨ་ལ་བི། ཨ་ལ་བི་ཥ་ཏྭཾ། ཨ་ལ་བི་ཥ་ཏ། པུཉ། ཨ་པཱ་བཱིཏ། ཨ་པཱ་བིཏྟཾ། ཨ་པཱ་བི་ཥཱ། ཨ་པ་བིཊྚ། །ཨ་པ་བི་:ཥཱ་ཏཱི། ཨ་པ་བི་ཥ་ཏ། དྲུཉ་འཚེ་བ་ལའོ། །ཨ་དཱ་བཱིཏ། ཨ་དཱ་བི། དྷཱཉ། ཨ་དྷཱ་:བཱིཏ། ཨ་དྷ་བཱིཊ། མུང་། ཨ་མ་བིཊྚ། དྲུང་། ཨ་ད་བིཤྩ། ་ཥཱ་དེད་པ་ལའོ། །ཨ་སཱ་བཱིཏ།  ི་ཨ་སཱ་བིཊྚཾ། ཨ་སཱ་བི་ཥྛ། །ཆུ་བསྟོད་པ་ལའོ། །།དྷུ་རྣམ་པར་འདར་བར་བྱེད་པ་ལའོ། །ཀུཊ་སོགས་སོ། །སྔ་མ་བཞིན་འཕེལ་བའོ། །ཨུ་བ་རུ་བསྒྱུར་རོ། །ཨ་ནུ་བཱིཊ༑ ོ་ཨ་ནུ་བིཙྪཱ། ་ཨ་ནུ་བི་ཥྛ། ཨ་དྷུ་བཱིཏ། ཨ་དྷུ་བིཏྟཾ། ཨ་དྷ་བཱི་། ལྟུང་། །སཱ་ར་ཏི། སཱུ་ཏི། ་སུ་ཡ་ཏི། ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ལས་སོ་ཞེས་པས་ཡང་ན་ཨིཊའོ། །ཨ་ས་བིཤྩ། ཨ་ས་བི་ཥཱ་ཏཾ། ཨ་ས་བི་ཥ་ཏ། ཨ་སོཊྚ། ཨ་སོ་ཥཱ་ཏཱ། ཁ་ཨ་སོ་ཥ་ཏ། བྷཱུ་ཡོད་པ་ལའོ། །ཨི་ྋཎི་སྦྲ་ཞེས་པ་་ལ་སོགས་པས་པ་རསྨིའི་ཚིག་ལ་སི་ཙའི་འཇིག་གོ། །བྷུ་ལས་སི་ཙ་འཇིག་པ་ལའོ་ཞེས་པས་ཨིཊ་མེད་དོ། །་། བྷཱའི་སི་ཙ་འཇིག་པ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་དོ། །ཨ་ཧྥུ་ཙིཏ། ཨ་བྷུ་ཏྭཾ། བྷུའི་མཐའ་པ་རོ་ཀླུ་དང་། །ཨ་བྱ་ཏ་ནཱི་དག་:ལ་བའོ་ཞེས་པས་དབྱངས་དང་པོའི་མཐར་བ་ཡིག་གོ། །དེ་ལ་བའི་མཐའ་ཞེས་བརྗོད་པ་ལས་ཨན་གྱི་ཨུས་སུ་བསྒྱུར་བ་མེད་དོ། །ཨ་བྷཱུ་བན། ཨ་བྷུ། །ཨ་བྷུ་ཏཾ། །
