།སྦྱོར་བ་བརྩམ་པར་བྱ་བ་མིན་ནོ་ཞེས་གསུང་ངོ༌། །ཤ་དིའི་ཤི་ཡའོ། །ཤདྲི་ཉམས་པ་ལའོ། །ཤཛྲ་:ལ་འཇིགས་པ་དང་འགྲོ་བ་ལ་འཇུག་པ་དང་སྦྱིན་པ་རྣམས་ལའོ་ཞེས་པས་ྋཡང་ནའོ། །:ཤ་དིའི་ཨན་ཕ་རོལ་ལ་ཤཱི་ཡ་རུ་བསྒྱུར་བར་འགྱུར་རོ། །།ཤི་ཡ་ཏེ། ཊི་ཤ་དིའི་ཨནྡྲི་ཀ་ར་ཎའི་ཡུལ་ལ་ཨཱཏྨ་ནེའི་ཚིག་བཤད་པར་བྱ་སྟེ། རུཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་རོ། །ཥ་དིའི་ཤཱིདའོ། །ཥབློ་འཇིགས་པ་ལའོ་ཞེས་པ་འདིའི་ཨན་ཕ་རོལ་ལ་ཤཱིད་དུ་བསྒྱུར་བར་འགྱུར་རོ། །རུ་ཏྲ་ཤཱི་ད་ཏི། ཛ་ནིའི་བི་ཀ་ར་ཎ་ལ་ཛཱའོ། ་། ཛ་ནི་རབ་ཏུ་སྐྱེ་བ་ལའོ་ཞེས་པ་འདིའི་བི་ཀ་ར་ཎ་ཕ་རོལ་ལ་ཛ་ར་བསྒྱུར་བར་འགྱུར་རོ། །ཛཱ་ཡ་ཏེ། ཡ་ཛཱ་ཡེ་ཏེ། ྋཛཱ་ཡནྟེ། དིབ་སོགས་ལས་ཡནའོ། །།ཛན་སྐྱེ་བ་ལའོ་ཞེས་པ་འདི་མ་སྨོས་པ་ནི་བི་ཀ་ར་ཎ་མེད་པའི་ཕྱིར་རོ། །ཡན་ལའོ་ཞེས་བྱ་བ་ལ་བི་ཀ་ར་ཎ་སྨོས་པ་ནི་ཕྱི་མའི་དོན་དུའོ། །ཛྷཱའི་ེའང་ངོ༌། །ཛྷཱ་རྟོགས་པ་ལའོ་ཞེས་པ་:：འདིའི་བི་ཀ་ར་ཎ་ཕ་རོལ་ལ་ཛ་ྋརུ་བསྒྱུར་བར་འགྱུར་རོ། །ཛ་ནཱ་ཏི། པཱུ་་་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ། ཡོ། པུང་དག་པ་ལའོ་ཞེས་པ་ལ་སོགས་པའི་བི་ཀ་ར་ཎ་ཕ་རོལ་ལ་ཐུང་ངུར་འགྱུར་རོ། །པུ་ནཱ་ཏི། ལུ་ནཱ་ཏི། :བསྔོ་ཎཱ་ཏིཏྟཾ། པཱུཉ་དག་པ་ལའོ། ཌི། །ལུཉ་ྋགཅོད་པ་ལའོ། །སྔོཉ་སྒྲིབ་པ་ལའོ། །ཨེཐུང་ངུའོ། །པུ་སོགས་རྣམས་བྲྀཏའི་མཐར་ཐུག་པའི་བར་རྣམས་སོ། །བི་ཀ་ར་ཎ་ལ་ཞེས་པ་ཅི་ཞེ་ན། །པུ་ཝིཡ་ཏེ། གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་གྱི་བྱིངས་ཐམས་ཅད་པ་ལ་ཧིཨུཏའི་འཕེལ་བའོ། ། ། །ཨུ་ཡིག་གི་འཕེལ་བར་འགྱུར་ཏེ། །གསལ་ཏེ་དང་པོ་ཡོན་ཏན་ཅན་ཏེ། ཡོན་ཏན་དང་བཅས་པའི་བྱིངས་ལིཐམས་ཅད་པ་ལ་ཡོན་ཏན་ཅན་ནོ། །ཡོ་ཏི། ཡིཡོ་ཏི། སྟོ་ཏི། སྟོཥི། ཡུ་སྤེལ་བ་ལའོ། །བྷུཉ་བསྟོད་པ་ལའོ། །།ཨནྡྲི་ཀ་ར་ཎའོ། །ཨད་སོགས་ལས་འཇིག་གོ། །།འདིས་—།
