བསྟན་འགྱུར། སྒྲ་མདོ། ཤི་། མང་པོའི་ཚིག་དང་རྟགས་ཅན་དང་། །:མཐའ་ཡི་ཚིག་དང་དབུས་ཀྱི་ཚིག། ཉེ་དང་རིང་ལ་སྙེག་པ་དང༌། །མངོན་དུ་བ་དང་ལྐོག་ཏུ་གྱུར། །རབ་ཏུ་ལྐོག་ལ་སྙེག་པ་དང་། །དྲི་དང་ལན་གྱི་ཚིག་དང་ནི། །བདག་དང་གཞན་དང་སྦྱོར་བའི་ཚིག། སུམ་ཅུ་ལྔ་ནི་བྱེད་ལྡན་དུས། །རིགས་དང་དུས་གསལ་བྱས་ནས་ནི། །བསྟན་བྱ་གང་ཞིག་སྟོན་པ་དེ། །དེ་ནི་ཆེད་དུ་འདོད་པ་ཡིན། །།ཆེད་ཀྱི་དབྱེ་བ་ཇི་སྙེད་པ། །དུས་ཀྱི་ཚིག་གིས་ཤེས་པར་བྱ། །ཇི་སྙེད་དོན་གྱི་སྦྱོར་བ་ནི་མི། །གང་ལ་འཇུག་པ་དེ་སྙེད་པ། །གང་གིས་སྟོན་པའི་ཡི་གེ་དེ། །དེ་ནི་མིང་དུ་་གདགས་པ་ཡིན། །སྒྲའི་བསྟན་བཅོས་སྨྲ་བ་ཀུན་ལ་འཇུག་པ་ཞེས་བྱ་བ་སྦྱོར་བ་བཞི་དང་ལྡན་པ་སློབ་དཔོན་ཆེན་པོ་སུ་བྷཱ་ཥ་ཀཱིརྟིས་མཛད་པ་རྫོགས་སོ། །།ཚེགོང་ཀའོ། །ཎ་གྱི་མི་ལང་བ་བློ་གཏཾ་ཏ་ਨੌਕਰੀਤੋਂਬਾਹਰਬਣੇਇਸਉ॥མགོ་བསྡུར་མཆན། ལཔལྷ་མོ། ཏི་ཀ་བ་ཡོད་པ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་སྨྲ་བ་ཀུན་་་ལ་འཇུག་པའི་སྒྲའི་བསྟན་བཅོས་སུ་བྷ་ཤིནྟི། ཞེས་འཁོད། «པེ་»«སྣར་»སབྦ་བྷ། «པེ་»«སྣར་»བྱེད། «པེ་»«སྣར་བརྗོད། «པེ་»«སྣར་»བརྙད། «པེ་»བརྗོད། ཨ«སྣར་པར། «པེ་»«སྣར་»གུ་ལུ་གུ། «པེ་»འདོམས་པ་གསལ། «སྣར་»འདོམ་པ་གསལ། @«པེ་»«སྣར་ནི། ཚེ«པེ་»«སྣར་»དེ། «པེ་»«སྣར་བྱེ། «པེ་»«སྣར་»བཞི། ཚ«པེ་»«སྣར་»བསྣན། «སྣར་»བསྡུད། «པེ་»«སྣར་»མཐའི། «པེ་»«སྣར་རྙེག«པེ་»«སྣར་»ན། &«པེ་»«སྣར་»བདགས། «པེ་»«སྣར་»ཀཱནྟིས། །ਉਸਤੋਂਬਲਬਨਕਾਬੂਕਰRਉਸਸਸਸਸਬsRਨਰਸਸਭਥੇਸਟਗਏਸਨਤੇਇਸਨੂੰਲਧਨਹਨਉਸ—།
