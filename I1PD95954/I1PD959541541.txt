བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའི་་ཊ། བློ་ལྷག་མ་ལོ། །ཨཱ་པི་ཏ་པི་ཏི་པི་ཥ་པི་བ་པི་ཤ་པིཙྪཱ་པིཀཱི་པི་ལི་པི་:ལུ་པི་སྲི་པི་ལས་པ་ལས་སོ། །ཞེས་པས་ཨིཊ་མི་འགྱུར་རོ། །ཨཱབྷཱ། ཏ། ཏེཔ། སཔ། བསྟཱ། ཤཔ། ཙྪཔཱ། ཀེདྷཱ། ལེམྦྷ། ལོཔྟ། །དེ་ལ་སྦྱོར་བ་ཐ་དད་པའི་ཕྱིར་སྤྲོ་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ནའོ། །ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །སྲསྨཱ། སཔ། ཡབྷ། ར་བྷ། །ཌཱ་ལ་བྷ་ཥ། ཡ་བྷི་ར་བྷི་ལ་བྷི་རྣམས་ལས་བྷ་ལས་སོ་ཞེས་པས་ཨིཊ་མེད་དོ། འོ། ཏའི་དྷ་ཉིད་དོ། །དྷཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་བྷའི་བ་ཉིད་དོ། །།ཡབྷཱ། རབྷཱ༑ལབྷཱ། ཡམ། རམ། ཎམ། གམྨཾ། ：ཡ་མི་ར་མ་ན་མི་ག་མི་་་ལས་མ་ལས་སོ། །ཞེས་པས་ཨིཊ་མེད་དོ། །།ཡནྟུ། །རནྟཱ། ནནྟུ། །གནྟུ། །རི་ཤ། །རུ་ཤ་། ཀྲུ་ཤ་ལྷག་མའོ། །ལི་ཤ་ཆུང་བའི་དངོས་པོ་ལའོ། །།རུ་ཙ་་་སོགས་སོ། །ལི་ཤ་འགྲོ་བ་ལའོ། །བི་ཤ། ལྷག་མའོ། །བི་ཤ་ཡ་ཛ་་་སོགས་སོ། །དི་ཤ། དྲི་ཤིར། སྤྲོ་ཤ། མྲྀ་ཤ། དན། ་པ་རསྨིའི་ཚིག་གོ༑། །རི་ཤི་རུ་ཤི་ཀྲུ་ཤི་ལི་ཤི་བི་ཤི་དི་ཤི་དྲི་ཤི་སྤྲོ་ཤི་མཾ་ཤི་དཾ་ཤི་ལས་ཤ་ལས་སོ། །ཞེས་པས་ཨིཊ་མེད་དོ། །ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །རེཥྛ། །རོཏྟཱ༑ཀྲོཏྟཱ། ལེཏྟཱ། བེཥྛ། དེཥྛ། །སྲི་ཛི་དྲི་ཤ་དག་གི་ཞེས་པ་ལ་སོགས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །ཏེ། དྲཥྛ། དེ་ལ་སྦྱོར་བ་ཐ་དད་པའི་ཕྱིར་སྤྲོ་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ནའོ་ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །སྤྲ་སྤནྣཱ། མུཏྟཱ། མནྟཱ?། དཾཥྛ། །ཏཱི་ཥ་འབར་བ་ལའོ། །ཡ་ཛ་སོགས་སོ། །པུ་ཥ་རྒྱས་པ་ལའོ། །དིབ་སོགས་སོ། །པ་་རསྨིའི་ཚིག་གོ། །ཀྲྀ་ཥ། བྷཱུ་སོགས་སོ། །་། པ་རསྨིའི་ཚིག་གོ། །།ཏུད་སོགས་ལ་ཡང་སྟེ་གཉི་གའི་ཚིག་གོ། །ཀྲྀ་ཥ་མཁྱུད་པ་ལའོ། །བློཥ། །ཡ་ཛ་སོགས་།
