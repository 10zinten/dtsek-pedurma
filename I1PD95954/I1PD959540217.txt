::བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཁྲི་། ་བ་པ་:གྲྭ་ཡཾ། ་༼ཨ་དའི་ཕ་རོལ་ན་པ་རོ་ཀླུ་ཡོད་ན་གྷསླུ་གཞག་པར་རུང་བ་སྟེ། ཨ་ད་ལ་པ་:རོ་ཀླུ་ཨཊ་སྦྱར་བས་གྷསློ་ཐོབ་ནས་གྷསྶཾ་གཉིས་པ་བྱས་ཏེ། རྨེ་ཕྱི། གས་ཛ་ཐོབ། ཨཱ་སྱ་པ་དཱིརྒྱུས་ཙྪེདྷཱ་ཏུ་བསྲིངས་པའོ། །ཛ་གྷ་སའོ༄༅། །ཨད་ལ་ཨ་ཏུས་སྦྱར་བས་གྷསླུ་ཐོབ། ལ་ཕྱི། ྋགྷནྡྷཾ་གཉིས་པ་བྱ་སྟེ་གོང་བཞིན། །ཨ་གྷོ་ཥ་ཥ་ཤིད་ཀྱི་མདོས་གྷས་ཀ་ཐོབ། །ཤཱ་སི་བ་སའི་སས་ཥ་ཐོབ་སྟེ་ཨ་དང་་་སྦྱར་རོ། །ཛ་ཀྵ་ཏུའོ། །ཡང་ན་ཨད་ལ་ཨཊ་སྦྱར་བ་རྣལ་མའོ། །ཨ་ཏུས་་་སྦྱར་བའ:ོ། པ། དེདཔཝེ་ཉཐུ་ཝ་ཡི། ཝེཉ་ནི་ཐགས་འཐག་པ་ཞེས་བྱ་སྟེ། དྷཱ་ཏུ་འདིའི་ཕ་་་རོལ་ན་པ་རོཀྵ་ཡོད་ན་ཝ་ཡ་གཞག་@པའམ། བེཉ་ཨཊ་སྦྱར་བས། མདོ་་་འདིས་ཝ་ཡ་ཐོབ་སྟེ། ཝ་གཉིས་པ་བྱས། ཝས་ཨུ་ཐོབ། ་ཨུ་ཝ་ཡའོ། །བེཉ་ལ་ཨ་ཏུས་སྦྱར་བས། ཝ་ཡ་ཐོབ། །ཝས་ཨུ་ཐོབ། ཨུ་གཉིས་པ་བྱས་ཏེ། ས་མཱ་:ན་དཱིགྷའོ༄༅། །ཨུ་ཡ་ཏུའོ། །ཡང་ན་ཝེཉ་ལ་ཨཊ་སྦྱར་བས། ཝ་གཉིས་པ་བྱས། ་ཨ་ཀཱ་ར་ད་ཏྲའི་མདོས་ཨཊ་ཀྱིས་ཨོ་ཐོབ་བོ། །ཝ་ཝོའོ། ཙྪཱ། ཝེཉ་ལ་ཨ་ཏུས་སྦྱར་བས། ཝ་གཉིས་པ་བྱས། ཝ་ཝ་ཏུའོ། །ཧནྟོརྦ་ེདྷི་རཱ་ཤི་ཎི། །ཧ་ན་འཚེ་བ་དང་འགྲོ་བ། །འདིའི་ཕ་རོལ་ན་ཨཱ་ཤཱིཥ་ཡོད་ན་ཧ་ན་ལ་བ་དྷི་་གཞག་པའོ། །ཧ་ན་ལ་ཡཏ་སྦྱར་བས་ཝ་དྷ་ཐོབ་སྟེ་ཡ་དང་སྦྱར། ཝ་:དྷུ་ཏའོ༄༅། །ཧ་ན་ལ་ཡཱ་ཏམ་སྦྱར་བས་སྔ་མ་བཞིན། ཝ་དྷཱ་ཏམའོ༄༅། །ཨ་བྱ་ཏ་རཱུཉྩ། །ཧ་ན་ཏེའི་ཕ་རོལ་ན་ཨ་དྱ་ནཱི་ཡོད་ན་ཝ་དྷི་གཞག་པ་སྟེ། ཧ་ན་ལ་དི་སྦྱར་བས། ཝ་དྷ་ཐོབ། གོང་དུ་ཨ་སྦྱར། སི་ཙ་དང་། །ཁེཨི་ཊ་སྦྱར་ཏེ། ཨི་ཏ་ཤཱ་ཨི་ཏི་སི་ཙ་ལོ་པ་བྱས་ནས། ཨཱི་དང༌། ཨཱི་ས་:མ་ན་དཱིརྒྱུའོ། །ཏེ་ང་། །ཁེ་བྱ་ཏེ་ཨ་ཝ་དྷི་ཙིཏའོ། །ཧ་ན་ལ་ཏམ་སྦྱར་བས། ?ཝ་དྷཱ་ཐོབ། ཨིཊ་དང་སི་—།
