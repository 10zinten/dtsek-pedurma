བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིབོ་བླུ་བུ། བོ་བྷ་བི་ཐ། བོ་དྷྭ་བ་ཐུ། བོ་:བྷ་བ། བོ་དྷི་བ། །བོ་ཧ་བ་བསོ། བོ་བྷ་བི་མི་ལི། པཱ་པཱ་ཙ། ཨད་ཨེ་ཉིད་དེ་ཨ་ཧྥུ་ས་དབྱི་བའང་ངོ་། །པེ་ཙ་ཏུ། པེ་ཙུ། པེ་ཙི་ཐ། པེ་ཙ་ཐུ། པ་པ་ཙ། པ་པཱ་ཙ། པ་པ་ཙ། :པེ་ཙི་བཅི༑པེ་ཙི་བ། པེ་ཅི་མ། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཏཱ་ཏྲའི་ལས་ལ་གཉིས་བརྗོད་པ་རྣམས་ཀྱང་ངོ༌། །ཏཱ་ཏྲེ། ཏཱ་ཏཱ་ཏྲེ། ཏཱ་ཏྲི་རེ། ཏཱ་ཏཱི་ཥ། ཏཱ་ཏཱིཐེ། ཏཱ་ཏྲི་དེ། ཏཱ་ཏྲེ། ཏཱ་ཏྲི་བ་ཧེ། ཏཱ་ཏྲི་མ་ཧེ། ཙ་ཙྩེ། ཙེ་ཙཱ་ཏེ། ཙེ་ཙཱ་རེ། ལོ་ལུ་བེ། ལོ་ལུ་བཱ་ཏེ། ལོ་ལུ་བི་རེ། ་བོ་བྷུ་བེ༑གཤཱསྟ་ནཱི། ཏཱ་ཏྲི་ཏཱ། ཙེ་ཙཱ་ཡི་ཏཱ། །ཙེ་ཙ་ཡེ་ཏཱ། །་་༄༅ལོ་ལཱ་བི་ཏཱི། བོ་བྷཱ་བི་ཏཱ། བོ་བྷཱ་བི་ཏཱ། ་:། །་ཡིན། ཁོ་tkz་ཀྱིན་ཅི་ལོག་ཏག་ལྟ་ཨཱ་ཤཱིཥ། ཨཱ་ཤཱིཥ་ལ་ཨེ་ཡིག་གོ་ཞེས་པས་དཱ་ལ་སོགས་པ་བཞི་པོ་རྣམས་ཀྱི་ཨེ་ཉིད་དོ། །དཱ་དེ་:ཡཱཏ། ད་བ་དང་དེ་པ་དག་གི་མི་འགྱུར་རོ། །དཱ་དཱ་ཡཱཏ༑གང་གི་དྷཱ་ཉིད་ནི། དཱ་དྷེ་ཡཱཏ། །བརྩེགས་པ་དང་པོའི་གནས་མ་ཡིན་པ་ལ་ཡང་ནའོ་ཞེས་བརྗོད་པར་བྱའོ་ཞེས་པས་ཡང་ན་ཨཱཏ་ཨེ་ཉིད་དོ། །ཏཱ་ཏྲེ་ཡཱཏ། ཏཱ་ཏཱ་ཡཱཏ། ཛཱ་གླེ་ཡཱཏ། །ྋན་མིའི་མཐའ་རྣམས་ཀྱི་ཡཎ་ཨཱ་ཡི་ཞེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཛེ་ཛཱི་ཡཱཏ། །ཙེ་ཙཱ་ཡཱཏ། ནེ་ནཱི་ཡཱཏ། །ཤོ་ཐུ་ཡཱཏ། ལོ་ལུ་ཡཱཏ། བོ་དྷཱ་ཡཱཏ། པ་པཱ་ཙུཏ། དངོས་པོ་དང་ལས་ལའང་། ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨིཊ་ཨི་ཙ་བཞིན་ཡང་ན་འགྱུར་རོ། །ཏཱ་ཏཱ་ཡི་ཊཱིཋ། ཏཱ། ཏྲི་ཏིཊྚེ། ཙེ་ཙཱ་ཡི་ཏྟི། ཙེ་ཙ་ཡི་ཏྟིབྷི། ལོ་ལཱ་བི་ཥིཏྠ། ལོ་ལ་བི་ཊཱི། བོ་བྷཱ་བི་ཎི། བོ་བྷ་བི་ཏི། པཱ་:པཱ་ཙ་ཥཱི། ་པཱ་པ་ཙི་ཥཱབྦི། །།བྷ་བི་ཥྛནྟི། ཏཱ་ཏྲི་ཥྛ་ཏི། །ཙེ་ཙ་ཡི་ཥྛ་ཏི། ཤོ་ཤ་བི་ཥྛ་ཏི། །ལོ་ལ་བི་ཥྛ་ཏི། བོ་བྷ་བི་ཥྛ་ཏི། པཱ་པ་ཙི་ཥྛ་ཏི། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་་་།
