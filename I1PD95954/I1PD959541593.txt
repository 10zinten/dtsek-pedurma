བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཉེའི?་ཎི་ཉིད་དོ། །བི་བྷཱུཏྠ་ཏི། སཱ་དྷ། ཥ་ཉེ་བར་བསྟན་པ་མེད་དོ། །སི་སཱཏྨ་ཏེ། ཧ་ན་ཧ་ན་དང་ཨིངའི་གཾ་དག་གི་ཨུ་པ་དྷཱའི་འོ་ཞེས་པས་ཨུ་པདྷཱའི་རིང་པོའོ། །ཨ་བྷཱ་ས་ལས་ཀྱང་ངོ་ཞེས་པས་གྷ་ཉིད་དོ། །ཛི་གྷོ་ས་ཏི། ་མན་ཤེས་པ་ལའོ། །མི་མཾ་ས་ཏེ། ཏ་པ། ཏི་ཏཔཱ་ཏི། ཏ་པ་དབང་ཕྱུག་ཉིད་ལ་ཡང་ནའོ། །ཏི་ཏཔ་ཏེ། ཏི་སྲི། ཏི་ཏིཙྪ་ཏི། ཁུ་བ་པ། བི་བདྨ་ཏི། བི་བསྶ་ཏེ། ཡ་:ཤ་པ༑ཤི་ཤ་པ་ཏི། ཤི་ཤདྨ་ཏེ། ཙྪཱ་པ། ཙུ་ཙྪཔྥ་ཏི། ཀླུ་པ། ཙི་ཀིམྨ་ཏི༑ཏུད་སོགས་ཀྱི་ནི་ཡཛ་སོགས་སོ། །།ལི་པ། ལི་ལིཙྪ་ཏི། ལི་ལིཙྪ་ཏེ། ལུཔྤེ༑ལུ་ལུདྨ་ཏི། ལུ་ལུདྨ་ཏེ། །གྲོགས་བློ། ཥ་ཉེ་བར་བསྟན་པ་མེད་དོ། །སི་སྲིདྨ་ཏི། ཡ་བྷ། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཡི་ཡཔྥ་ཏི༄༅། ཡ་མ། ཡི་ཡཾ་ས་ཏི། རམ། :རི་རཾ་ས་ཏི། ཊཱིབི་རི་རཾ་ས་ཏི། བི་རི་རཾ་ས་ཏེ། ཨཱ་རི་རཾ་ས་ཏི། ཎམ། ནི་ནཾ་ས་ཏི། རི་ཤ། རུ་ཤ། ཚ་ཤ་དག་གིའང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཥའི་ཀའོ། །རི་རིཀྵ་ཏི། རུ་རུག་ཏི༄༅། ཀྲུ་ཤ། ཙུ་ཀྲུཁཱ་ཏི། ལི་ཤ་ཆུང་ངུར་གྱུར་པ་ལའོ། །ལི་ལིཀ་ཏེ། ལི་ཤ་འགྲོ་བ་ལའོ། །ལི་ལིཀཱ་ཏི། བི་ཤ་། བི་བིཀྵ་ཏི། བི་བིཀྵ་ཏེ། དི་ཤ། དི་:དིཀྵ་ཏི། དི་དིཀྵ་ཏེ། དྲི་ཤིར། སནའི་མཐའ་རུ་ཙ་སོགས་སུ་་་ཤེས་པར་བྱའོ། །དི་དྲིཀྵ་ཏེ། སྤྲོ་ཤ། །པི་སྤྲིཀཱ་ཏི་མྲྀ་ཤ་། མི་མྲྀཀྵ་ཏི། དཾ་ཤ། དི་དཾཥྛ་ཏི། ཏཱི་ཥ། ཏི་ཏཱིཀྵ་ཏི། ཏི་ཏཱིཀྵ་ཏེ། ་པུ་ཥ། དིབ་་སོགས་སོ། །པུ་པུཀྵ་ཏི། བྷུ་སོགས་སོ། །ཨཱ་དིའི་ཡང་ངོ༌། །པུ་པོ་ཏི་ཥ་ཏི། པུ་པོ་ཏི་ཥ་ཏེ། ཀྲྀ་ཥ། ཙི་ཀྲྀཀྵ་ཏེ། ཏུད་སོགས་ཀྱི་ནི་ཡཛ་སོགས་སུའོ། །ཤྲཱི་ཥ་མཁྱུད་པ་ལའོ། །ཤི་ཤྲཱིཀཱ་ཏི། ཤྲཱི་ཥ་བསྲེག་པ་ལའོ་ཞེས་པ་འདིའི་ནི་ཤི་གླེ་ཏི་ཥ་ཏི། ཤི་ཤི་ཥི་ཥ་ཏེ། དྭི་ཥ། དི་དཱིཀྵ་ཏི། དི་དཱིཀྵ་ཏེ། པི་བློ། །
