པོ་ལ་ཨུ་བའོ། །ལྦུ་བ་ཧེ། ལྦུ་བ་ཏེ། རྩྭ་ཥ། སྦྲུ་བཱ་ཐེ། སྠཱ་ནཱཾ། ཧྥུ་བེ། ཅུ་བ་ཧེ། །ཧཱུ་མ་ཧེ། ནི་བྷུ་ཏེ། བི་བྷུ་ཏེ། ་ཨ་པ་ཧྥུཏེ། ཨཱཏྨ་ནེ་བཤད་དོ། །དྲི་ཥ་མི་དགའ་བ་ལའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །པ་རསྨིའི་ཚིག་ལ་གཅིག་གི་ཚིག་རྣམས་ལ་ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཥ་༌ལས་ཊ་སྡེའོ། །དེཊྛི། དཱིཥྛ། དྲི་ཥནྟི།  ོ་ཥ་བྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་་པས་ཥའི་ཀའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །དྭེ་ཀླུ། དྲིཊྛ། དཱི། དཱིཙྪི། དཱིཋ༑དཱི། །དཱིབྦེ། དྭི་ཥ་ཏེ། དྭི་ཥ་ཏེ། དི་གླེ། དྭི་ཥཱ་ཐེ། དྷཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་ཥའི་ཌའོ། །དྲི་ཌྷ། དྲི་༑དྭི་ཥ་ཧེ། དཱིཉྩ་ཧེ། དུ་ཧ་འཇོ་བ་ལའོ། །ད་དང་པོའི་གྷའོ་ཞེས་པས་དྷཊ་ལ་ཧའི་གྷའོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་་དྷའོ་ཞེས་པས་ཏ་ཐ་དག་གི་དྷ་ཉིད་དོ། །ག། དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལས་གསུམ་པའོ་ཞེས་པས་གྷའི་གའོ། །དོ་བྷི། དུདྡྷ། དུ་ཧནྟི། བྱིངས་ཀྱི་་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་དང་པོ་ས་དང་དྷ་དག་ལ་བཞི་པ་ཉིད་དོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་གྷའི་ཀའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །དྷོཥི། དུདྡྷ། དུགྷ། དོ་ཧྥ། ་དུ་ཧྥ། དུ་ཧྨ༑དུབྷེ། དུ་ཧཱ་ཏེ། :དུ་ཧ་ཏེ། དྷུཀྵེ། དུ་ཧཱ་ཐེ། དྷབྷེ། དུ་ཧེ། དུ་ཧུ་ཧེ། །དུ་ཧྥ་ཧེ། དི་ཧ་ཉེ་བར་སོགས་:པ་ལའོ། །དེརྨི། དིངྒྷ། དི་ཧནྟི། དྷེཤཱི། :：དིམྦྷ། དིག། དེ་རྨི། དི་ཧྭ། དི་རྨི། དིབྷེ། དི་ཧཱ་ཏེ། དི་ཧ་ཏེ། དྷིཀྵེ། །དི་ཧྭ་ཐེ། དྷིཀྵེ། དི་ཧེ། དི་ཧྭ་ཧེ། དི་ཧྥ་ཧེ། །པྲ་ཎི་དེདྡྷི། པྲ་ནི་དེངྒཱི། :：ལ་ཧི་རོ་མྱངས་པ་ལའོ། །ནང་གི་ཡན་ལག་ཉིད་ལས་སྔ་མ་བཞིན་ཡོན་ཏན་ནོ། །དེ་ལས་ཧའི་རྟའོ་ཞེས་པས་དྷུཊ་།
