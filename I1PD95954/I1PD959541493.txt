བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའིགྲོལ་བར་བྱེད་པ་དང་སོ་སོར་དགའ་བར་བྱེད་པ་དག་ལའོ། །།།གྲནྡྲི་སྤེལ་བ་ལའོ། །ཤཱནྡྲ། གནྡྲ། སྤེལ་མ་ལ་ཡང་ནའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་རྣམ་པར་རྟོག་པ་མཐའ་ཡས་སོ། །ཨ་བནྡྲིཏྠ་མེ་ཁ་ལཱ་སཱ་ཡ་མེ་བ། ཨ་གྲིནྡྲ་མེ་ཁ་ལཱ་སཱ་ཡ་མེ་བ། བྲུཉ༑དྲུའི་བ་ཙིའིའོ། །ཨ་བོ་ཙཏ་དྷརྨ་སྭ་ཡ་མེ་བ། ལྷས་བྱིན་གྱི་ངག་དང་གདུགས་ཀྱི་རྣམ་འགྱུར་ལ། དེ་བ་དཏྟཾ་བྱ་ཀྲྀ་ཏ་ཙྪཏྠུ། སཱ་ཡ་མེ་བ། ་སྐུ་ན་མ་དག་རང་ཉིད་ལ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ལས་བཞིན་འགྱུར་བའི་དམིགས་བསལ་ལོ། །པྲསྣོ་ཋ་གོ། སཱ་ཡ་མེ་བ། ལྷས་བྱིན་གྱིས་དཔྱའི་འདུད་པར་བྱེད་ཟིན་པ་ནི། དེ་བ་དཏྟ། ཨ་ནཾསྟ་དཎྞ། སཱ་ཡ་མེ་བ། ཤིཉ། ཀིར་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ལས་བཞིན་འགྱུར་་་་བའི་དམིགས་བསལ་ལོ། །ཨུ་ད་ཤི་ཤི་ཡཏ་དཎཱ་སཱ་ཡ་མེ་བ། །།འཚོ། །་་་་་གང་། ཞེས་པ་མཁན་པོ་སརྦ་དྷ་རས་བཀོད་པའི་ཏི་ལ་སོགས་པའི་མཐའི་བྱ་བ་ལ་ཨ་བྱ་ཏ་ནི་མདོར་བསྟན་པའོ། །ལོ་རཚོ་ཕྱི་ལ་གང་ཡོད་ཀྱི་བྱ་བ་ཁབ་ཀྱང་དེ་ལ་དེ། ་མ་ཟླ་ཧ་པ། ཞཎིན་པ་། བ་ཤོག༄༅། །དཔལ་ངག་གི་དབང་ཕྱུག་:རིག་འཛིན་ཆེན་མོ་སྒྲོལ་མ་དཀར་མོ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་པ་རོབྷཱའོ། །སྨའི་འདས་པ་ལའོ་ཞེས་སྤྱིར་་བཏང་ནས་པ་རོགཱའོ་ཞེས་པ་བྱིངས་ལས་འདས་པའི་དུས་ལ་པ་རོགཱའི་རྣམ་དབྱེ་་་འགྱུར་རོ། །དེ་རྣམས་ཀྱི་རང་མིང་གི་དུས་ཀྱི་ཁྱད་པར་རོ་ཞེས་པས་ལྐོག་ཏུ་གྱུར་པའི་འདས་པའི་དུས་ལ་པ་རོགའོ། །དབང་པོ་རྣམས་ལ་ཕ་རོལ་ལྐོག་ཏུ་གྱུར་པའི་དུས་ནི་དབང་པོ་རྣམས་ཀྱི་ཡུལ་མ་ཡིན་ཞེས་པའི་དོན་ནོ། འོ་། རྒྱལྟ་ནི་ཐོབ་པ་ལ་ཡང་པ་རོག་བསྒྲུབ་པར་བྱའོ། །བྷྱཱསྔ་མ་བཞིན་སྐྱེས་བུ་དང་པོ་ལ་སོགས་པ་རྣམས་ཏེ། དེ་བཞིན་དུ་གཅིག་ཉིད་ལ་གཅིག་གི་ཚིག་ཨཊ། ཊ་ཡིག་ནི་ཨཱ་ཡིག་།
