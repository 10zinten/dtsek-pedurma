བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིརྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་གཉིས་ཀ་ལ་དང་པོ་དང་གསུམ་པ་དག་གོ། །མི་ཏྲ་དྷཱཀ། མི་ཏྲ་དྲུག། མི་ཏྲ་དྷུཊ། མི་ཏྲ་དྷུཌ། མི་ཏྲ་དྲུ་ཧོ། མི་ཏྲ་དྲུ་ཧ༑མི་ཏྲ་དྲུ་ཧཾ། མི་ཏྲ་དྲུ་ཧོ། མི་ཏྲ་དྲུ་ཧ། མི་ཏྲ་དུ་ཧཱ། མི་ཏྲ་:དྷུཙྪཱམི༑ོམི་ཏྲ་དྷདྡྷཾ། མི་ཏྲ་:དྷཱནྟི®། མི་ཏྲ་དྷགྷི%ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་པུ་ར་དྲུཧ་ལ་སོགས་པ་རྣམས་སོ། བྱ་ཡཾ་ཀྵ་ཡིག་གི་མཐའ་ཅན་ཀཱཥྛ་ཏགྲུའི་སྒྲ་སྟེ། །ཤིང་འཇོག་པར་བྱེད་དོ་ཞེས་པས་ཏཀྱི་ལས་ཀྭིཔ་ལ་སོགས་པའོ། །སི་ལ་སོགས་པ་རྣམས་སོ། །སི་དབྱིའོ། ཞེF། བརྩེགས་པ་དང་པོའི་དྷུའིའོ་ཞེས་པས་ཀ་ཡིག་དབྱིའོ། །རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་དག་ལ་ནི་ཥ་ཡིག་གི་་སྔ་མ་བཞིན་དུ་ཌ་ཡིག་གོ། །རྫོགས་པ་ལ་ཡང་ནའོ། །ཀཱཥྛ་ཏཊ། ཀཱཥྛ་ཏཌ། ཀཱཥྛ་ཏགྷོ། ཀཥྛ་*ཏཀྵ། ཀཱཥྛ་ཏཏྟཾ། ཀཱཥྛ་ཏཀློ༄༅། ཀཱཥྛ་ཏཀྵ། ཀཱ་ཏཁཱ༑སྔ་མ་བཞིན་དུ་ཀ་དབྱི་བ་དང་ཌ་ཉིད་དོ། །་། :ཀཱཥྛ་ཏནྟཾཀཱཥྛ་ཏནྟིཞེས་པ་ལ་སོགས་པའོ། །སྐྱེས་པ་ལ་དབྱངས་ཀྱི་མཐའ་ཅན་དང་གསལ་བྱེད་ཀྱི་མཐའ་ཅན་རྣམས་ཀྱང་ཁྱད་པར་དང་བཅས་པ་བརྗོད་ཟིན་ཏོ། །སྐྱེས་པའི་་རྟགས་ཀྱི་སྐབས་ཏེ་དང་པོའོ:༌། །:། །ཅི་ཀ་སྡེ་ཚན་བ་ཏེ། ཞི་བ་པས་ད་ནི་མ་ནིང་གི་རྟགས་ཅན་རྣམས་བརྗོད་པར་བྱའོ། ་། ཝ་ན་དང་། ལ་རི་དང༌། ཨསྦི་དང་། ཝ་ཏུ་རྣམས་དང་།  ི་པི་པད་དང་། སཱ་ཙེམན་དང་། །ཨ་ཧན་དང་། དེ་བཞིན་དུ་ཀརྨན་དང་། ་འསཎིཥ་དང༌། །ཊོ་ཝ་པུཥ་དང་། ཙེ་ཏས་ཏེ། འདི་རྣམས་མ་ནིང་དུ་བརྗོད་དོ། །ཝ་ནའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །མ་ནིང་ལས་སི་ཨོ་དག་གི་དབྱི་བ་ྋསྟེ། དེ་ལ་བརྗོད་པ་ཡང་མ་་་ཡིན་ནོ་ཞེས་པ་རྗེས་སུ་འཇུག་བཞིན་པ་ལས། ཨ་ཡིག་ལ་སཾ་བུདྡྷི་མ་ཡིན་པ་ལ་།
