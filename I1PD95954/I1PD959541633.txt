ད་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི:དོ༑། །ཛུ་ཧཱ་བ་ཡི་ཥ་ཏི། ཀུ། ཙུ་ཀཱ་བ་ཡི་ཥ་ཏི། ཧི། ཧིའི་ཙ་ཎ་ལ་མ་ཡིན་ནོ་ཞེས་བརྗོད་པར་བྱ་བ་ལ་ཨིན་དང་ཙཎ་ལ་ཞེས་བཀག་པ་ལས་ཨིན་ཕ་རོལ་གྱི་ཧའི་གྷ་ཉིད་དོ། །པྲ་ཛི་གྷ་ཡ་ཡི་ཙེཏི། བྷུ་ཏིའི་རྒྱུ་ལ་ཨིན་མཐའོ། ཟེ་། ཏ་པ་:ཙུ་ར་སོགས་སོ། །བི་བྷ་བ་ཡི་ཥ་ཏེ། བྷི་ཏི། བི་བྷི་ཥ་ཡི་ཥ་ཏེ། །ཙཱ་ཡི༑ཁ་ཙི་ཙཱ་ཡ་ཡི་ཥ་ཏི། བསྟེན་དུ་འཇུག་པར་འདོད་པ་ལ། ཤཱ་ཡི་ཡ་ཡི་ཥ་ཏི༑ཌ་བསྲུང་དུ་འཇུག་པར་འདོད་པ་ལ། ཨཱ་བི་བ་ཡི་ཥ་ཏི། ་གནས་ཅན་་བཞིན་འགྱུར་བ་:ཐོབ་བོ་ཞེས་པས་ཡི་བ་དང་བི་གཉིས་བརྗོད་དོ། ཊ། འཕུར་བ་སྨྲ་རུ་འཇུག་པར་འདོད་པ་ལ། ་ཨོཥྛ་ཌ་ཡི་ཥ་ཏི། ་ཨུཎྜ་ཡི་བརྗོད་དུ་འཇུག་པར་འདོད་པ་ལ། ཨུཎྜ་ཌ་ཡི་ཥ་ཏི། གནས་ཅན་བཞིན་འགྱུར་བ་ལས་ནི་ཌ་གཉིས་བརྗོད་དོ། །ནཱ་ཡི། ནི་ནཱ་ཡ་ཡི་ཥ་ཏི། ཀཱ་རི། ཙི་ཀཱ་ར་ཡི་ཥ་ཏི། ཧཱ་རི། ཛི་ཧཱ་ར་ཡི་ཥ་ཏི། བཱ་དི། བི་བཱ་ད་ཡི་ཥ་ཏི། ་ཁཱ་དི། ཙི་ཁཱ་ད་ཡི་ཥ་ཏི། །གྷ་ཊི། ཛི་གྷ་ཊ་ཡི་ཥ་ཏི། ྋག་མི། ཛི་ག་མ་ཡི་ཥ་ཏི། ར་མི། རི་ར་མ་ཡི་ཥ་ཏི། རཀྵི། །རི་རཀྵ་ཡི་ཥ་ཏི། ::ཤིཀྵི། །ཤི་ཤིཀ་ཡི་ཥཏི། བེ་དི༑། བི་བེ་ད་ཡི་ཥ་ཏི། བོ་དྷི། བུ་བོ་དྷ་ཡི་ཥ་ཏི། ཌི་ཛ་དང་མཐར་གནས་དང་པ་སྡེ་ཕ་རོལ་ཅན་ཨུ་ཡིག་གི་ཨ་ཡིག་ལའོ་ཞེས་པས་སན་ཕ་རོལ་ལ་ཨ་བྷཱུ་ཏི། སའི་གང་ཨུ་ཡིག་དེའི་ཛ་ཡིག་དང་མཐར་གནས་དང་པ་སྡེ་ཕ་རོལ་གྱི་ཨ་ཡིག་ཕ་རོལ་ལ་ཨི་ཉིད་དོ། །ཛུ། མདོ་ལས་བྱུང་བའི་བྱིངས་སོ། །ཨིན་ལ་གང་བྱས་པ་དེ་་ཐམས་ཅད་དེ་གནས་ཅན་བཞིན་ནོ། །ཛི་ཛ་ཨེབ་ཡི་ཥ་ཏི། པུ། པི་པཱ་བ་ཡི་ཥ་ཏི། རུ། རི་རྭ་བ་ཡི་ཥ་ཏེ། ལུ་ཉ། ལི་ལཱ་བ་ཡི་ཥ་ཏི། བྷཱུ། བི་བྷཱ་བ་ཡི་ཥ་ཏི། ལྷུང་། །མི་མཱ་བ་ཡི་ཥ་ཏི། ་ཤུ། སྲུ། དུ། སྲུང༌། རླུང་། ཁྱུང་རྣམས་ཀྱི་ཡང་ན་བརྗོད་པར་བྱའོ་ཞེས་པས་:ཐུ་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ནའོ། །ཏེ།
