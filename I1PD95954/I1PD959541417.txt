ར:བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའི་་ཏུ་དེ་ཡཾ༄༅། ཏུ་དེ་བ། ་ཏུ་དེ་མ། ཏུ་དེ་ཏ། ཏུ་དེ་ཡཱ་ཏྭཾ། ཏུ་དེ་ར་ན། །ཏུ་དེ་ཐཱ། ་ཏུ་དེ་ཡཱ་ཐཱ། ཏུ་དེ་དྡྷཾ། ་ཏུ་དེ་ཡ། ་ཏུ་དེ་བ་ཧི། ཡི་ཏུ་དེ་མ་ཧི། ཎྞཏ། ནུ་དེཏ། རུ་དྷིར། རུ་དྷ་སོགས་ཀྱི་དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་སྒྲའོ། །རུ་དྷ་སོགས་ཀྱི་བི་ཀ་ར་ཎའི་མཐའ་དབྱིའོ། །རྗེས་སུ་ང་རོའོ། །སྡེ་པ་ལ་སྡེ་པའི་མཐའོ། །རུདྡྷཱཏ། :：རུདྡྷཱ་ཏྭཾ། རུདྡྷི། །ྋརུནྡྷ། རུདྡྷཱ་ཏྭཾ། རུདྡྷཱཏེ། །རུཙྪཱ༑རུདྡྷཱ་བ། རུདྡྷཱ་མ། རུནྡྲིཏ། རུནྡྲི་ཡཱ་ཏྭཾ། རུནྡྲི་རན། རུནྟཱི་ཐཱ། །རུནྡྲི་ཡཱ་ཐཱ། :རུདྡྷི་ཧཱུྃ། རུནྟི་ཡ། རུདྡྷི་པ་ཧི། །རུནྟཱི་མ་ཧི། བྷི་དིར། བྷིནྡྷཱ་ཏ། :·བྷི་ཏི། ཚི་དིར། ་་:ཚིནཏ། (ཚིནྡཏི། རི་ཙིར། རིཉྩ་ཏ། །རི་ཝཱིཏ། གཉི་གའི་ཚིག་གོ། ་། ཤི་མྨཾ། ཤི་ཏེ། བྷཉྫ། ན་སྒྲ་ལས་་ཀྱང་བི་ཀ་ར་ཎ་:ལས་སོ་ཞེས་པས་ཨ་ནུ་ཊཾ་ག་དབྱིའོ། །བྷཾ་ཛཱཏ། བྷུ་ཛི་ྋསྐྱོང་བར་བྱེད་པ་དང་བཟའ་བ་དག་ལའོ། །ཟླ་། བྷཾ་ཛཱཏ། ་བྷུ་ཛིའི་ཟས་ལའོ་ཞེས་པས་་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྷཾ་ཛཱིཏ། པ་རསྨིའི་ཚིག་རྣམས་སོ། །ཉི་ཨིནྟཱི། ཨིནྡྲིཏ། །ཁིད་ཚིཞན་པ་ཉིད་ལའོ། ་ཟླ། ::ཁིརཱིཏ། །བིད་ྋརྣམ་པར་་་དཔྱོད་པ་ལའོ། །བྷི་ནིཏ། ཨཱཏྨ་ནེའི་ཚིག་རྣམས་སོ། །ཏ་ནུ། ཏ་ན་སོགས་ལས་ཨུའོ། །ཏ་ནུ་ཡཱཏ། ཏ་ནུ་ཡཱ་ཏཱཾ། ཌི་ཏ་ནུ་ཡུ། ཏེ་ནུ་ཡཱ། ཏ་ནུ་ཡཱ་ཏཾ། ཏ་ནུ་ཡཱཏ། ཏ་ནུ་ཡཱ། ཏ་ནུ་ཡཱ་བ། ་ཏ་ནུ་ཡཱ་མ། ་ཨུའི་ཡི་གེ་ཝའོ། །ཏ་ནཱིཏ། ཏ་ནཱི་ཡཱ་ཏཱ། ཏ་རཱི་རན། ཏ་ནཱི་ཐཱ། ཏ་ནཱི་:ཡཱ་ཐཱ། ཏ་ནཱི་ཧཱུཾ། ཏ་ནཱི་ཡ། །ཏ་ནཱི་བ་ཧི། །ཏ་ནཱི་མ་ཧི། ཁུ་ཀྲྀཉ་ཀ་རོ་ཏིའིའོ་ཞེས་པས་བྱིངས་ཀྱི་ཡོན་ཏན་ནོ། །།བྱིངས་ཐམས་ཅད་པ་ཡོན་ཏན་མེད་པ་ལ་ཨའི་ཨུ་ཡིག་གོ། །ཡ་ལའང་ངོ་ཞེས་པས་བི་ཀ་ར་ཎའི་ཨུ་ཡིག་དབྱིའོ། །།ཀུར་མྱུར་དག་གི་མ་ཡིན་ནོ་་་ཞེས་བཀག་པ་ལས་ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོ་མེད་དོ། །།
