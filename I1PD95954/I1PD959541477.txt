པའི་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་:༌ཀྱི་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་རྟའོ། །དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་་པའོ་ཞེས་པས་གྷའི་གའོ། །ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། །ཌྷ་ལ་ཌྷའི་དབྱི་སྟེ་ཨུ་པ་དྷཱའི་རིང་པོ་ཡང་ངོ༌། །ཨུ་དྷཀྵཏ། ཌེ་ཨ་དྷཀྵ་ཏྭཾ། ཨ་དྷུཀྵན། ཨ་དྷཱཏ། །ཨ་དྷགྷཱི། ཨ་དྷཀྵ་ཐཱ། །ཨ་:དྷུ་དྷ། །་་ཨ་དྷུ་དྡྷཾ། ཨ་དྷབྷཾ༈། ཨ་དྷཀྵ་པ་ཧི། ཨ་དྷཁཱ་མ་ཧི། ཨ་དྷིཥྛ་ཏ། ཨ་དྷིགྷ། །ཨ་དྷིཀྵ་ཐཱ། ཨ་དྷིཥྛཱ། ཨ་དྷིཀྵ་ཧཱུཾ། ཨ་དྷིཤྩ། ཊཨ་དྷིཥཱ་བ་ཧི། ཚ:：ཨ་དྷིཀྵ་མ་ཧི། :ཨ་ལིཀྵ་ཏ། ཨ་ལཱི་ལྷ། ཨ་ལིཀ་ཐཱ། ཨ་ལི་ཞཱ། ༼ཨ་ལིཀྵ་ཧཱུཾ། ཨ་ལི་བྷཾ། ཨ་ལིཀྐཱ་བ་ཧི། །ཨ་ལིཀཱ་མ་ཧི། ཨ་བྷུཀྵ་ཏ། ཨ་གཱ་ཀླུ་ལྷ། །ཨ་བྷུཀྟ་ཐཱི། ཨ་བྷུཀྟ་ཋཱ། ཨ་གྷཀྵ་ཧཱུྃ། །ཨ་གྲྭ་རྣོ། ་ཨ་གྷཀྵ་བ་ཧི། ཨ་བྷུཀྵ་མ་ཧི། གོ་ཧིའི་ཨུ་པ་དྷཱའི་ཨཱུ་ཏའོ། །།ཨ་གཱ་ཧི་ཏ། ཨ་གཱ་ཧིཏྟཾ། ཨ་གཱ་ཧི་ཥཱ། ཨ་ཧྥུ་ཧི། ཨ་གཱ་ཧི་ཥཱ'ཏཱཾ། ཌཨ་གཱ་ཧི་ཞྭ། །ཨ་ཧྭ་ཧི་ཧཱུཾ། ཏི་སྲི་འགྱུར་བའི་དོན་ལའོ། །སི་ཙ། ཨཱ་བཱི་ཥ་དག་ལ་ཡང་ན་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་ལ་ཨ་ཏིཔ་ཨ་ཏི་པ་ཧཾ། །ཏུཏེ། གསལ་བྱེད་མཐའ་རྣམས་ཀྱི་ཨིཊ་མེད་པའོ་ཞེས་པས་པ་ར་སྨིའི་ཚིག་དང་། །སི་ཙ་ལ་འཕེལ་བའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཨ་ཏོཏིཏམེ། ཨ་ཏོཏྟཱལཾ། ཨ་ཏུཏྟ། ཨ་ཏུཏྠ་ཏྭཾ། ཆུད། ཨ་ནོ་ཏི་སཱིཏ། ཨ་ནོཏྟཾཙོ། བིད་ཡོད་པ་ལའོ༑། ཨ་བི་དཱིཏ། བྷུ་ཛ་སྐྱོང་བ་དང་བཟའ་བ་དག་ལའོ། ་། ཙི་ཉྩསྡེའི་འདྲ་བའི་ཡི་གེ་མིན་པ་ལ་ཀིའོ། །ཨ་བྷོཊྛིཏ། :：ཨ་བྷོ་ྋཀཾ། བྷུ་ཛིའི་ཟས་ལའོ་་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨ་བྷགྷེ། ཨ་བྷུཀྵ་ཧཱུཾ། ཀུ་པ། ：ཨ་ཀོཔཱིཏ། ་ཨ་:ཀོཔི་མི། །དི་བུ། སི་ཙའོ། །ཨིཊའོ། །ནཱ་མི་་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨ་དེ་བཱིཏ། ཨ་དེ་བིཊྚ། ཎི་བུ། ཨ་སེ་བཱིཏ། །
