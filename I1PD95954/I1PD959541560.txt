བ་དྷི་མ་ཡིན་ནོ། །ཨིཙ་བཞིན་འགྱུར་བ་ལས་ཨའི་ཨུ་པ་དྷཱ་རིང་པོའོ། །ཧའི་གྷ་ཉིད་དོ། །གྷ་ནི་ཌི། ་ཡང་ན། ་བདྷི་ཊཱི། གྲ་ཧ། །སྔ་མ་ལས་ཨིཊའི་རིང་པོའོ། །།ཨི་ཙ་བཞིན་གྱི་ཨིཊ་ལ་ནི་གྲྭ་ཧི་ཏིཥྛ། ་གྲ་ཧི་ཁཱིཥྛ། དྲྀ་ཤིར། དནྟི་ཏིཥྛ། དྲྀ་ཀཱི། །པ་ཋ༄༅། ཨིཊ་ཨཱ་ག་མའོ། །པ་ཋི་ཏཱི། ལས་བྱེད་པ་པོ་ལས་བཞིན་ནོ། །ལས་བཞིན་འགྱུར་བས་བྱེད་པ་པོ་ལ་ཨཱཏྨ་ནེའི་ཚིག་གོ། །དབྱངས་དང་ཧ་ན་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ན་ཨིཙ་བཞིན་ཨིཊ་ལའང་ངོ༌། །ལཱ་བི་ཥིཉྩ། ཡང་ན། ་ལ་བི་ཊྛིཏྠ། །ཚོཀེ་དཱ་རཿསཱ་ཡ་མེ་བ། པ་གླིཏྠ་སཱ་ཡ་མེ་བཱནྣཾ། ཀུ་ཋི་རཾ་ཛ་དག་གི་ཨཱ་ཤཱི་ཥ་དག་ལ་པ་རསྨིའི་ཚིག་ལ་མཐོང་ངོ༌། །ཀོ་ཥིཉྩ། པཱ་དས་ཡ་མེ་བ། :རཾ་ཁཱིཏྠ་བཙྩཾ་སྭ་ཡ་མེ་བ། ཞེས་པ་མཁན་པོ་ཆེན་པོ་སརྦ་དྷ་རས་བཀོད་པའི་ཏི་ལ་སོགས་པའི་མཐའི་བྱ་བ་ལས་ཨཱ་་ཤིཥ་མདོར་བསྟན་པའོ། །་ཌཊི་། ཏི། ཁོང་བ་ལ་རྒྱུ་nཚོ་ཕྱི། བློག་ལ་ཉེན་ཁ་བྱུང་ཏེ། ཁྲི་ཚུ་་། གས་ལ་ན་མ་ཤི་ཟླ། རང་ཁྲ་ཡི་སྤྱི་ཟླ་ལ་ཕྱིར། ཁོ་། །ཊི་དཔལ་ངག་གི་དབང་ཕྱུག་:རིག་འཛིན་ཆེན་མོ་ྋསྒྲོལ་མ་དཀར་མོ་ལ་ཕྱག་་འཚལ་ལོ། །ད་:ནི་བྷ་བི་ནཱིའོ། །།འབྱུང་འགྱུར་ལ་བྷ་བི་ཥཱནྟི་དང་ཨཱ་ཤཱིཥ་་་་དང་ཤཱསྟ་ནཱིའོ། །ཞེས་པས་འབྱུང་འགྱུར་གྱི་དུས་ལ་བྱིངས་ལས་བྷ་བི་ཊྛནྟཱིའི་་རྣམ་དབྱེ་འགྱུར་རོ། །དེ་རྣམས་ཀྱི་རང་མིང་རྣམས་ཀྱིས་དུས་ཀྱི་ཁྱད་པར་རོ། །ཞེས་པས་བྷ་བི་ཥྛནྟི་འབྱུང་འགྱུར་ཙམ་ལས་ཏེ། །འབྱུང་བར་འགྱུར་བའི་ཁྱད་པར་ལས་མ་ཡིན་ནོ། །དི་ྋརིང་དང་སང་ཡང་ན་འབྱུང་བར་འགྱུར་བའོ། །ཞེས་པས་བྱིངས་ལས་ཤཱསྟ་ཉྫཱནཱི་ཡང་ཡང་ནའོ། །སྔ་མ་བཞིན་སྐྱེས་བུ་དང་པོ་ལ་སོགས་པ་རྣམས་ཏེ། གཅིག་ཉིད་ལ་གཅིག་གི་ཚིག་སྱ་ཏི། གཉིས་ཉིད་ལ་་གཉིས་ཀྱི་ཚིག་སྣ་ཏས། །མང་པོ་ཉིད་ལ་མང་པོའི་ཚིག་སྱཱནྟི། །སྐྱེས་བུ་དབུས་མ་།
