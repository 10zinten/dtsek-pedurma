།བསྟན་འགྱུར། སྒྲ་མདོ། ལིཀེ ོ་ཀ་་གང་། ཁོ་ལ་སྤྱི་ཁཎྜ་ཟླཏེ་ལཊ་ཀྱི་ལྟ་རྒྱ་མི་ཨར་སྦྱོར་བ་བདུན་ཀྱི་བདེ་བ། རྒྱུས། རྗེ་བསྡུར་མཆན། །ད་ལྟ་ཡོད་པ་ཁྱད་མཚན་བྱང་འདི་«པེ་»«སྣར་»ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་དང་«སྡེ་»«ཅོ་»གཉིས་འགྱུར་མི་འདྲ་བས། །འདིར་«པེ་ལ་གཞི་བཟུང་ནས་《སྣར་»དང་དཔེ་བསྡུར་བྱས་ཡོད། བློ་«སྣར་ཨཱ། @《སྣར་པ། །ལས་ཀྱི་དཀའ་བ་ཁྱད་«སྣར་པོར། གང་དུ་སྐྱེ་བ་བྱ་མིན་དུ་ག་ན་ན་ག་ར་དཔའ་(«སྣར་»ལ། 《སྣར་»ལས། 《སྣར་»པའི། །«སྣར་»པར། :«སྣར་ཨ། «སྣར་པ། «སྣར་མིང་། ཁ་ ོ ེ«སྣར་»ལ། ®«སྣར་»ལ། «སྣར་པ་མིའི། «སྣར་ཤཐཱ། «སྣར་»ཀྲྀ། ཨེ«སྣར་རུ། ༄༅«སྣར་»དཎྜི། «སྣར་»ཧགཅིང་གོ། མ་རྒྱུ་ད་ཀ«སྣར་»དྷཱ། ་ཀྱི་ཀིརལླི་ཡི་གོ་བློ་ཀཱ་ར་དྷཾ། «སྣར་»ཏྭཾ། «སྣར་»ཀ་རཿཀླུའི་ཡི་གེཀཾ། །«སྣར་»ལི། ཕྱི《སྣར་»ཨོཾ། @སྣར་»རཾ། «སྣར་བས་རྣོ། 《སྣར་སནཙྩཾ། «སྣར་འཐོད། «སྣར་»ཏལྟ། «སྣར་པ། «སྣར་ཨརེ། «སྣར་བ་ཨོ། «སྣར་ཡནྟུ། ®«སྣར་བྲི། དཀྱིལ་::འིའི་བར་ེ«སྣར་ནི། «སྣར་ལུང་བ་ཝ་ནི། «སྣར་ན། ཀ་གང་དུ་འདུག་ནའང(ི«སྣར་མའི། «སྣར་ན། «སྣར་ཨོའི་ཡི་གེ། ཉེསྣར་»། «སྣར་བ། «སྣར་+བ། «སྣར་བྱིས། «སྣར་ནི། རབ་བ་ནཔཎྜི་(《སྣར་ཐཱ། 《སྣར་སྲངས་དྲུབ་ཏེ་ཏྠ། «སྣར་»དནཾ། «སྣར་»ཀྲྀཥྛ། །«སྣར་»ཝཱཉྩི། «སྣར་བྷ་ས་བྱར་སང་། «སྣར་»ཏི། དེ་་«སྣར་»ཙྪ། «སྣར་»—ལ། «སྣར་གྱིས། &«སྣར་»ཡ། ག་ལ་སྤྱིཔ་ཏ་«སྣར་ཐ། 《སྣར་སྦྱར། 《སྣར་»+ས། «སྣར་ཡིག«སྣར་»ཝཱཉྩ། ®«སྣར་»སཾ། འ«སྣར་»ཏཱ། &«སྣར་ཏུཾ། «སྣར་»ཙ། ཕྱི«སྣར་པས། «སྣར་»པའི།  ི«སྣར་»ཏཾ། «སྣར་བ་ཅམ་གིས། «སྣར་»—ལ། «སྣར་ན། ཀ་«སྣར་»འདུ། 《སྣར་»ཏ། «སྣར་ཨི། «སྣར་ཨི། 《སྣར་བསལ། «སྣར་»ཏི་ས། «སྣར་»རོ། །
