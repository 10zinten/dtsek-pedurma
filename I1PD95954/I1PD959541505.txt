:བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤེཏི་ཧའི་ཛའོ་ཞེས་པས་ཨ་ཧྥུ་སའི་ཧ་ཡིག་གི་ཛའོ། །ཛ་ཧྭ་ཐེ། ཛ་ཧྲི་ཌྷ། ཛ་ཧྲི་དྷེ། ཛ་ཧྲེ། ཛ་ཧྲི་བ་ཧེ། །ཛ་ཧྲི་མ་ཧེ། ཛྷཉ་འཛིན་པ་ལའོ། །ད་དྷཱ་ར༑ད་དྷཱ་ཏུ། ད་དྷེ། ཤྲཱིཉ་འཛིན་པ་ལའོ། །།བ་བྷཱ་ར། བ་བྷཱ་ཏིུ། བ་བྷུ། སྲི་ལ་སོགས་པའི་མདོས་ཨིཊ་མེད་དོ། །བ་བྷནྡྲ། བ་གྲྭ་ཐུ། བ་བྷ་ར། བ་བྷཱ་ར། བ་ཛཾ་བ། །བ་བྷི་མ། །བ་བྷེ། ཌབ་བྷཱ་ཏེ། །བ་ཤྲཱི་རེ། བ་བྷི་ཥ། བ་བྷཱ་ཐེ། བ་བྷི་ཝེ། བ་བྷེ། བ་གྷི་པ་ཧེ། ་བ་བྷཾ་མ་ཧེ། བྲྀཉ༑བ་བཱ་ར། བ་བྲ་ཏུ། །བ་བྲུཙྪི། དྲྀ་བྱེ་ཨད་རྣམས་ཀྱི་རྟག་ཏུ་ཐལ་ལ་ཨིཊའོ་ཞེས་པས་ཨིཊའོ། །བ་བ་རི་ཐ། ?བ་བྲི་ཐུ། བ་བྲ། བ་བཱ་ར། སྲི་ལ་སོགས་པའི་མདོས་ཨིཊ་མེད་དོ། །བ་བྲི་བ། བ་ཛྲ་མ། ་བ་བྲེ། བ་བཱ་ཏེ། བ་ཤྲཱི་རེ། བ་དྲྀ་ཏེ། བ་གྲྭ་ཐེ། ་བ་བྲི་། བ་བྲེ། བ་ཛྲ་ེབ་ཧེ། བ་ཛྲ་མ་ཧེ། ཌཱ་ཀྲྀཉ། ཙ་ཀཱ་ར། ཙ་ཀྲ་ཏུ། ཙཀྲུ། །ཀྲྀཉ་ལས་སུཊ་མེད་པ་ལས་སོ་ཞེས་པས་ཨིཊ་མེད་དོ། །ཙ་ཀནྡྲ། ཙཀྲ་ཐུ། zཙཀྲ། ཙ་ཀཱ་ར། ཙ་ཀ་ར། ཙ་ཀྲྀ་བ། །ཙ་ཀྲ་མ། །ཙཀྲེས། ཙ་ཀཱ་ཏེ། ཙ་ཀྲྀ་ཏེ། ཙ་ཀྲྀ་ཏེ། ཙ་ཀཱ་ཐེ། ལྷ་ཉིད་སྔ་མ་བཞིན་ནོ། །ཙ་ཀྲྀ་ཌྷ། ཙཀྲེ། ཙཀྲི་བ་ཧི༑ཙཀྲྀ་མ་ཧེ། སཾ་པ་རི་ཨུ་པ་ལས་རྒྱན་ལ་སུཊའོ། །ཞེས་པས་ཨ་ཧྥུ་སས་བར་ཆོད་པ་ལ་ཡང་སུཊ་འདོད་དོ། །སཾ་ཙ་སྐུ་ར། །བརྩེགས་པ་དང་པོའི་རྀཏའི་ཡང་ངོ་ཞེས་པས་པ་རོག་ཡོན་ཏན་མེད་པ་ལ་ཡོན་ཏན་ནོ། །གལ་ཏེ་གཉི་གའི་ཚིག་བསྟེན་པ་སུཊ་ཨཱ་ག་མ་ཕྱི་རོལ་གྱི་ཡན་ལག་གཅིག་གི་ཚིག་བསྟེན་པ་ཉིད་ཀྱི་ཡོན་ཏན་ནང་གི་ཡན་ལག་གོ། །དེ་བཞིན་དུ་ཡང་དེ་ལ་བྱེད་པ་གྲུབ་པ་ཕྱིའི་ཡན་ལག་དང་ནང་གི་ཡན་ལག་ཅེས་པས་སུཊ་མེད་པར་གྲུབ་བོ། །རིགས་པ་་་ཉིད་ཀྱི་རྟག་ཏུ་བྱིངས་ཀྱི་དང་པོར་ཉེར་བསྒྱུར་:ཀྱིས་འགྲེལ་པ་ཕྱིས་བྱེད་པ་ནི་ཉེ་་—།
