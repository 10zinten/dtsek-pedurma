ཨཱ་ཡིག་དབྱིའོ། ཊི། བྱ་ཏི་བྷེ། བྱ་ཏི་བྷ་བ་ཏེ། བྱ་ཏི་བྷཱ་མ་ཧེ། ཡཱ་ཐོབ་པ་་ལའོ། །ཡཱ་ཏི། །ཡཱ་ཏ་ཡཱནྟི། པྲ་ཡཱ་ཏི། ཨུང་ཡཱ་ཏི། པྲ་ཎི་ཡཱ་ཏི། བཱ་འགྲོ་བ་དང་དྲི་དག་ལའོ། །བཱ་ཏི། པྲ་བཱ་ཏི། པྲ་ཎི་བཱ་ཏི། །ཌཱ་གཙང་བ་ལའོ། །སྣ་ཏི། ཤཱ་འཚེད་པ་ལའོ། །ཤཱ་ཏི། དྭ་སྨད་པར་འགྲོ་བ་ལའོ། །དཱ་ཏི། ནི་རཱ་ཏི། པྲ་ཎི་དཱ་ཏི། པཱ་བསྲུང་བ་ལའོ། །པཱ་ཏི། པཱ་ཏཿཔཱནྟི༑་རཱ་ལཱ་ལེན་པ་ལའོ། །རཱ་ཏི། ལཱ་ཏི། དཱ་པ་གཅོད་པ་ལའོ། །པཱ་ཡིག་ནི། དཱ་པ་མིན་པའི་དཱ་དྷཱ་དག་དཱའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །དཱ་ཏི། ཁཱ་རབ་ཏུ་གཏམ་པ་ལའོ། །ཁཱ་ཏི། པྲ་ཁྱུ་ཏི། བྱཱ་ཀྵ་ཏི། པྲ་རྫོགས་པ་ལའོ། །པྲ་ཏི། ཚོམཱང་ཚང་པ་ལའོ། །མཱ་ཏི། ་་སཾ་མཱ་ཏི། བིང་ཤེས་པ་ལའོ། །གཅིག་གི་ཚིག་རྣམས་ལ་ཨུ་པ་དྷཱ་ནཱ་:མིནའི་འང་ཡང་བའི་འོ་ཞེས་པའི་ཡོན་ཏན་ནོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །བེཏྟི། བིཏྟ༔བི་དནྟི༑འབེཏི། བིཏྠབིཏྠ། བེདྨ། བིདྡབིདྨབྲུའི་ཨཱ་ཧ་ལས་ནི་ལྔ་རྣམས་ཀྱི་དང་དགུ་རྣམས་ཏེ། །བི་དེའི་ཡང་དེ་བཞིན་ནོ། ༌། ཨཊ་ལ་་སོགས་པའི་ཚིག་ཕྲད་རྣམས་སུ་བྱ་བ་ལ་ཏི་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ན་རིམ་པ་:ཇི་ལྟ་བ་བཞིན་དུ་བིད་ལས་ཕ་རོལ་གྱི་ཏཾ་ལ་སོགས་པ་དགུ་པོ་རྣམས། ཨཊ་ལ་་སོགས་པ་རྣམས་སུ་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་ཚིག་ཕྲད་དུ་བྱ་བར་རྣམ་པར་བཤད་དོ་ཞེས་ཁ་ཅིག་གོ། །ཏིའི་གནས་ལ་ཨཊ། ཏས་ཀྱི་ཨ་ཏུས། ཨནྟིའི་ཨུས། སིའི་ཐལ། ས་ཐས་ཀྱི་ཨ་ཐུས། ་་ཐའི་ཨ། མིའི་ཨཊ། ་བས་ཀྱི་བ། མས་་ཀྱི་མ། བེ་ད། བིད་ཏུ། །བི་དུབེཏྠ། བི་ད་ཐུབི་ད། བེ་ད། བིདྨ། བིདྨ༄༅། འདིར་ནི་ཏི་ལ་སོགས་པའི་མཐའི་གཟུགས་ཅན་འདི་རྣམས་་ཚིག་ཕྲད་དོ། །།སམ་ལས་གམ་ཞེས་པ་ལ་སོགས་པས་ལས་ཅན་གྱི་ནི་རུ་ཙ་སོགས་།
