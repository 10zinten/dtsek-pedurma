སོགས་སོ། །ཧ་ད། རུ་ཙ་སོགས་སོ། ་། ཤ་བློ། ཥ་ཛྲ། ལྷག་མའོ། །པ་ད། རུ་ཙ་སོགས་སོ། །སྐནྡིར་ལྷག་མའོ། །ཁི་ད་ཡོངས་སུ་གདུང་བ་ལའོ། །ལྷག་མའོ། །།ཁི་ད་ཞེན་པ་ཉིད་ལའོ། །རུ་དྷ་སོགས་སོ། །ཨ་དི་ཏུ་དི་ནུ་དི་་གླུ་དི་སཱི་ཏྲ་ཏི་བི་བྱ་ཏི་པིནྡ་ཏི་བི་ནཏྟི་ཚི་དི་བྷི་དི་ཧ་དི་ས་དི་པ་དི་སྐནྡི་ཁི་དི་ལས་ད་ལས་སོ། །ཞེས་པས་ཨིཊ་མེད་དོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཨཏྟཱ། །ཏོཏྟཱ། ནོཏྟཱ། །གློཏྟཱ། སྭཱཏྟཱ། བེཏྟཱ། ཚེཏྟཱ༑བྷེཏྟཱ། ཧཏྟཱ། ཤཏྟཱ། སཏྟཱ། པཏྟཱ། :སནྟཱ། ཁཏྟཱི། རཱ་དྷ་ལྷག་མའོ། །རུ་དྷིར། །ཡཛ་སོགས་སོ། །ཀྲུ་དྷ། ཀླུ་དྷ། བནྟ། ཤུ་དྷ། ཎི་དྷུ་ཡང་དག་པར་གྲུབ་པ་ལའོ། །པ་རསྨིའི་ཚིག་གོ། །བུ་དྷ་ཁོང་དུ་ཆུད་པ་ལའོ༑། །དིབ་སོགས་སོ། །།ཡུ་དྷ་ཀུན་ནས་འཁྲུག་པ་ལའོ། །ཨཏྨ་ནེའི་ཚིག་གོ། །བྱ་དྷ། །སཱ་དྷ་ལྷག་མའོ། །རཱ་དྷི་རུ་དྷི་ཀྲུ་དྷི་ཀླུ་དྷི་བནྟི་ཤུ་དྷི། སི་དྷ་ཏི་བུ་རྒྱ་་ཏི་ཡུ་དྷི་བྱ་ཏི་སཱ་དྷི་ལས་དྷ་ལས་སོ་ཞེས་པས་ཨིཊ་མེད་དོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་་་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་ཏ་ཡིག་གི་དྷ་ཡིག་གོ། །།དྷཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་དྷ་ཡིག་གི་ད་ཡིག་གོ། །:རཱདྷཱརོདྡཱ། བནྡྷ། ཀོདྷཱ། ཤོནྣཱ། སཙྪཱ། བོ། ཡོདྡྷཱ། བྱདྡྷཱ། སནྣཱ། ཀཱཧ་ན་ལྷག་མའོ། །མ་ན་ཤེས་པ་ལའོ། །རུ་ཙ་སོགས་སོ། །ཧ་ནི་མ་རྱ་ཏི་ལས་ན་ལས་སོ་ཞེས་པས་ཨིཊ་མེད་དོ། །ཧནྟཱ། མནྟཱ། ཌཨཱ་བྷཱི། ཏ་པ་ཀུན་ཏུ་གདུང་བ་ལའོ། །ལྷག་མའོ། །ཏ་པ་དབང་ཕྱུག་ཉིད་ལ་ཡང་ནའོ། །རུ་ཙ་སོགས་སོ།  ོ་། ཏི་ཕྲེ། ཉེ་པ་ལྷག་མའོ། །ཁུ་བ་པ། :ཤ་པ་ཡ་ཛ་་སོགས་སོ། །།།ཙྪ་པ་ལྷག་མའོ། །།ཀྲི་པ་དིབ་སོགས་དང་ལྷག་མའོ། །ཏུད་སོགས་དང༌། ཡཛ་སོགས་སོ། །ལི་པི། །ལུཔྤེ་གཉིས་ཀའི་ཙག་གོ། །ཟིགས། །
