འ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་ཨུཎྜ་བི། །ཨུཎྜ་བཱ་བ་ཧི། ཨཱུ་ཎྞ་བཱ་མ་ཧི། ་བྷུཉ། སྟོ་ཏུ། །སྟུ་ཏཱཏ། སྟུ་ཏཱ། སུ་བནྡཱ། སུ་ཧི། སྟུ་ཏཱཏ། སྐུ་ཏཾ། ཏུ་ཏ། ་་:སྟ་བཱ་ནི། སྐུ་བཱ་བ། ༽སྐུ་བཱ་མ། སྟུ་ཏྭཾ། སྐུ་བཱ་ཏཱཾ། སྡུ་བ་ཏྭཾ། སྐུ་། སྐུ་བཱ་ཐཱ། སྐུ་དྷཾ༑ཥ་སྟ་བི། སྟ་བཱ་བ་ཧི། སྟ་བཱ་མ་ཧི། །བྲུན། །གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་ལ་བྲུ་ལས་ངག་དང་པོར་ཨཱིཊའོ། །བྲ་བཱི་ཏུ། བྲུ་ཏཱཏ། བྲུ་ཏྭཾ། བྲུ་བནྟུ། བྷུ་ཧི། བྷུ་ཏཱཏ། །བུ་ཏཾ། བྷུ་ཏ། བྲ་བཱ་ནི། བྲུ་བཱ་བ། འབྲུ་བཱ་མ། བྷུ་ཏྭཾ། བྷུ་བཱ་ཏཱ། བྷཱུ་བ་ཏྭཾ། བུ་°། བྲུ་བཱ་ཐཱ། བྷུ་བྷཾ། བྲ་བི། །བྲ་བཱ་བ་ཧི། བྲ་བཱ་མ་ཧི། ཧུ། ཛུ་ཧོ་ཏུ། །ཛུ་ཧུ་ཏཱཏ། ཛུ་ཧུ་ཧཱུཾ། ཨ་བྱཱ་སྟ་ལས་ཨནྟིའི་ན་དབྱིའོ། །ོ། ཛུ་ཧྭ་ཏུ། ཌོཧུ་དྷུཊ་དག་ལས་ཧིའི་དྷིའོ་ཞེས་་པས་ཧིའི་དྷིའོ། །ཛུ་ཧུ་དྷི། ཛུ་ཧུ་ཏཱཏ། ཛུ་ཧུ་ཏཾ། ཛུ་ཧུ་ཏ། ཛུ་ཧུ་བཱ་ནི༑ཛུ་ཧུ་བཱ་བ། ཛུ་ཧུ་བཱ་མ། ཉི་བྷི། བེ་བྷེ་ཏུ། བི་བྷི་ཏཱཏ། ་་་བྷི་ཏཱཏ། །བི་བྷི་ཏཾ། བི་བྷཱི་ཏཱཾ། ཌ་བི་བྷུ་ཏིུ། བི་བྷི་ཧི། བི་བྷི་ཧི། བི་བྷི་ཏཱཏ༑བི་བྷཱི་ཏཱཏ། བི་བྷི་ཏཾ། བི་བྷཱུ་ཏཾ། །བི་བྷི་ཏ། བི་བྷཱི་ཏ། བི་བྷ་ཡཱ་ནི། བི་བྷ་ཡཱ་བ། བི་བྷ་ཡཱ་མ། ཧཱི། ཛི་ཧྲེ་ཏུ། ཛི་ཧཱི་ཏཱཏ། །ཛི་ཧཱི་ཏཱཾ། ཛི་ཧྲི་ཡ་ཏུ། ཛི་ཧི་ཧི*། ཛི་ཧཱི་ཏཱ་ཏ། ཛི་ཧཱི་ཏཱཾ། ཛི་ཧྲི་ཏ། ཛི་ཧྲ་ཙྪིཡཱ་ནི། ཇི་ཧྲ་ཡཱ་བ། ཛི་ཏྲ་ཡཱ་མ། པྲ?། པི་པརྟུ། །:པི་སྲི་ཙེཏཱཏ། :：པི་པྲ་*ཏྭཾ། པི་པྲ་ཏུ། པི་པྲ་ཧི། ་་:པི་སྲི་ཙྪཱཏཱཏ། པི་པྲ་ཏྭཾ། པི་པྲ་ཏ༑། པི་པ་རཱ་ཎི། ་་:པི་པ་རཱ་བ། །པི་པ་རཱ་མ། ཨོ་ཧཱ་ཀ་། །ཛ་ཧཱ་ཏུ། ཛ་ཧི་ཏཱཏ། ཛ་ཧཱི་ཏཱཏ། ཛ་ཧི་ཏྟཾ། ཛ་ཧཱི་ཏཱཾ། ཛ་ཧ་ཏུ། ཧི་ལའང་ཨཱ་་་ཉིད་དང་:ཨི་ཉིད་དང་ཨཱི་ཉིད་བཤད་དོ། །ཛ་ཧཱ་ཧི། ཛ་ཧི་ཧི། ཛ་ཧཱི་ཧི། རཾ་སུཾ། ་ཨི་ཡཆུ། ཨི་ཡཾ་ཏཱཏ། ཨི་ཡཾཏྭཾ། ཨི་ཡུ་ཏུ། །ས་སཆུ། ས་།
