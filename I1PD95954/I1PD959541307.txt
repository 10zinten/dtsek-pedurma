ས་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི༼ནང་ཏཱར། སྲི་རུ་འགྱུར་ལ། །སྔ་མ་བཞིན་དུ། །ཨཱིའི་རྐྱེན་མེད་ཅིང་། དེ་དག་དབྱངས་ལ་རའོ་ཞེས་པས་རིའི་ར་ཉིད་དོ། །ཙ་ཏ་སྲ། ཛས། ཙ་ཏ་སྲ། ཤས། ཙ་ཏ་སཾ་བྷི། ཙ་ཏ་སྭ་བྷུ། ལ་ནུ་ཨཱ་ག་མ་སྟེ། ནཱ་མ་ལ་རིང་པོ་མ་་ཡིན་ནོ་ཞེས་པས་བཀག་པའི་ཕྱིར། ཙ་ཏ་སཾ་ཏཱཾ། ཙ་ཏ་སཾ་ཥཱ། ་མ་ནིང་ལ་ནི། ཙ་ཏཱ་རི། ཛས། ཙ་ཏཱ་རི། ཤས། །ལྷག་མ་སྐྱེས་པ་དང་མཚུངས་སོ། །མང་ཚིག་ཁོ་ན་ལས། གཞན་པའི་ཚིག་གི་དོན་ལ་གང་གི་ཚེ། ཙ་ཏི་སྲི་འགྱུར་བ་དེའི་ཚེ། སྔར་བསྟན་པའི། པྲི་ཡ་ཏི་སྲི་བཞིན་ཏེ། པྲ་ཡ་ཙ་ཏ་སཱ། དགའ་བ་བཞི་པོ། པྲ་ཡ་ཙ་ཏ་སྲོ༑པྲ་ཡ་ཙ་ཏ་སླ། ཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་ཕྱི་ཡ་ཙ་ཏ་ས། རཾ་ཨར། །ཧེ་པྲ་ཡ་ཙ་ཏ་སྲོ། ཧེ་པྲ་ཡ་ཙ་ཏེ་སྲ། དེ་བཞིན་དུ་བུད་མེད་ལ་ཡང་ངོ༌། །།མ་ནིང་ལ་ནི། ནཱ་མའི་མཐའ་རྣམས་ཀྱི་ཙ་ཏཱར་ལ་ཡང་ཕྱོགས་གཅིག་ལ་བརྗོད་པ་བཀག་པའི་ཕྱིར། བྲི་ཡ་ཙ་ཏ་སཾ། ཕྱོགས་གཅིག་ལ་བསྒྱུར་བའི་ཕྱིར་ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲ་ཡ་ཙ་ཏུ། པྲ་ཡ་ཙ་ཏ་སྲི་ཎཱི། ཨོ་ཨཱི། པྲི་ཡ་ཙ་ཏ་སྭཱ་ཎི། ཛས་ཤི། གང་གི་ཚེ་ཙ་ཏ་སྲི་མི་འགྱུར་བ་དེའི་ཚེ། པྲ་ཡ་ཙ་ཏཱ་ར། པྲི་ཡ་ཙ་ཏཱ་རོ། པྲ་ཡ་ཙ་ཏཱ་ར། ཛས། པྲ་ཡ་ཙ་ཏཱ་རཾ། པྲ་ཡ་ཙ་ཏཱ་རོ། །ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲ་ཡ་ཙ་ཏུ་ར། །པྲ་ཡ་ཙ་ཏུ་རཱ། པྲ་ཡ་ཙ་ཏུདྡྷཾ། པྲ་ཡ་ཙ་ཏུབྷི། དྲུག་པའི་མང་ཚིག་ལ་ནུ་མེད་དེ། པྲ་ཡ་ཙ་ཏུ་རཱི། ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལ་གཉི་ག་ཐུང་བའོ། །།ཧེ་པྲ་ཡ་ཙ་ཏཱ། ཧེ་པྲ་ཡ་ཙ་ཏཱ་རོ། ཧེ་པྲ་ཡ་ཙ་ཏཱ་ར། དེ་བཞིན་དུ། བུད་མེད་ལ་ཡང་ངོ༌། །མ་ནིང་ལ་ནི། ལ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲ་ཡ་ཙ་ཏུ། པྲ་ཡ་ཙ་ཏུ་རཱི། པྲ་ཡ་ཙ་ཏཱ་རི། སླར་ཡང་། དེ་བཞིན་དུ་ཟླ་ལ་སོགས་པ་ལ་སྐྱེས་པ་དང་མཚུངས་སོ། །ཀུན་དུ་བོད་པ་ལ། །ཧེ་པྲ་ཡ་ཙ་ཏུ། ཧེ་པྲ་ཡ་ཙ་ཏུ་རཱི། ཧེ་པྲ་ཡ་ཙ་ཏཱ་རི། །ལ་ཏེ་། ་་་་་་་་་་་་་་མོ་བ་ག་ཏཱ། །
