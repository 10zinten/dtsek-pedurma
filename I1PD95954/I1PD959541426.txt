བེ་བྱཱ་ཏཱཾ། ཨཱ་བེ་བྱཱ་ཏཱཾ། ཨཱ་བེ་བཱི་ཥཱ། ཨཱ་བེ་བྱཱ་ཐཱ། ཨཱ་བེ་བཱི་ཧཱུཾ། ཨཱ་བེ་བྱི༑ཨཱ་བེ་བྱཱ་བ་ཧི།  ོཨཱ་བེ་བྱཱ་མ་ཧི། རླུང་། རྩྭ་ཏྭཾ། དབྱངས་དང་པོ་ཡོན་ཏན་མེད་པ་ལ་ཨུ་བའོ། །བྷུ་བཱ་ཏཱཾ་ཧྥུ་བ་ཏྭཾ། བྷུ་ཥཱ། །ཧྥུ་བཱ་ཐཾ། རྩྭ་ཧཱུཾ། ཡོན་ཏན་ནོ། །ཨོཾ་ཨ་བའོ། ་། སྣ་བི་སྣ་བཱ་བ་ཧི། དྷཱ་བཱ་མ་ཧི། དཱིཥ༑དྭིཤྩི། དཱིཏྟཱཏ། ་དཱིཏྟཾ། །:·དྭི་ཥནུ། ཥའི་གསུམ་པ་ཌའོ། །དཱིརྟི། དཱིཥྛ༑ཏ། དཱིཝཾ། ཧཱིཋ། དྭེ་ཏཱ་ཎི། ཏྲི་ཥཱ་བ། ཌདྭེ་ཥཱ་མ། དཱིཏྟཾ། དྲི་ཥ་ཏྭཾ། དྭི་ཥ་ཏྭཾ། དྲི་ཀླུ། དྲི་ཥཱ་བྷཾ། དྲིནྡྷུ། དྭེ་ཏི། དྭེ་*ཥཱ་བ་ཧི། དྭེ་ཥཱ་མ་ཧི། དུ་ཧ། :：དོཧྥུ། དུབྷཱཏ། དུདྡྷཾ། དུ་ཧན་དིུ། དུབྷི། དུབྷཱཏ། དུདྡྷཾ། །དུགྷི། །དོ་ཧཱ་ནི། དོ་ཧཱ་བ། ཁདོ་ཧཱ་མ། དུདྡྷཾ། དུ་ཧཱ་ཙེཏྭཾ། དུ་ཧ་ཏྭཾ། དྷཀྵ། དུ་ཧཱ་བྷཾ། དུཤྩ། དོ་ཧི། དོ་ཧཱ་བ་ཧི། དོ་ཧཱ་མ་ཧི། དི་ཧ། ་་:དེབྦྷ། དིབྷཱཏ། །དིདྡྷཾ། །དི་ཧནྟུ། དིབྷི། དིབྷཱུཏ༑དིཤྩ། དིརྒྱུ། དེ་ཧཱ་ནི། དེ་ཧཱ་བ། དེ་ཧཱ་མ། དིཤྩ། དི་ཧཱ་ཏྭཾ། དི་ཧ་ཏྭཾ། དི་རྒྱུ། །དི་ཧཱ་ཤྲཱི། དིགྷཾ། དེ་ཧི། དེ་ཧཱ་བ་ཧི། དེ་ཧཱ་མ་ཧི། ལི་ཧ། ལིབྦྷ། ལིངྒཱཏ། ལིནྣཾ། ལི་ཧནྟུ། ལིངྒཱི། ལིངྒཱཏ། ལིངྒྷཾ༑ ིལིངྒཱ། ལེ་ཧཱ་ནི། ལེ་ཧཱ་བ། ལེ་ཧཱ་མ། །ལིངྒྷཾ། ལི་ཧཱ་ཏཱ། ལི་ཧ་ཏཱཾ། ལིཀྵ། ལི་ཧཱ་ཤྲཱི། ལཱིནྡྷེདཾ། །ལེ་ཧི༄༅། ་ལེ་ཧཱ་བ་ཧི། །ལེ་ཧཱ་མ་ཧི། :ཨུཙྪཉ། ཨུཙྪཱཏིའི་ཡོན་ཏན་ནོ་ཞེས་པ་ཅིགསལ་བྱེད་དང་པོ་ཡོ་་་་ཏན་ཅན་ལ་ཡང་ཡོན་ཏན་ནོ། །ཕྱོགས་གཅིག་ལ་ཨུ་ཏའི་འཕེལ་བའོ། །།ཨུཎོ་ཏུ། ཨུཎྜི་ཏུ། :ཨུཥྞ་ཏཱཏ། །ཨུཎྜཱུ་ཏཾ། ཨུཥྛ་བནྟུ། ཨུཙྪཱ་ྋཧི། ཨུཎྜུ་ཏཱཏ། ཨུཎྜཱུ་ཏཾ། ཝི་ཨུཙྪཱ་ཏ། ཨུཙྪཱ་བཱ་ནི། ཨུཎྚ་བཱ་བ། །ཨུཙྪཱ་བཱ་མ། ཨུཥྞཏྭཾ། ཨུཥྛ་ེབཱ་ཏཱཾ། ཨུཥྞ་ང་ྋཏྭཾ། ཨུཥྛ་། །ཨུཎྜུ་བཱ་ཐཱ། ཨུཙྪཱཙྪཱནཾ། 。ཨུolcuGolol—།
