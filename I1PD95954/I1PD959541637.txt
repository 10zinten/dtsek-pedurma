ལཱ༑་ལ་སོགས་པ་ལས་སོ། །དཱཎ། འུ་དཱཉ། དེང༌། དེཉ། དོ། དཱ། མཱ༑གཱ་ཡ་ཏི། ཞེས་པ་ལ་སོགས་པས་ཨཱི་ཉིད་དུ་བྱས་ཏེ་གཉིས་བརྗོད་དོ། །།དེ་དཱི་ཡ་ཏེ། དི་པ་དང་དཱ་པ་དག་གི་ནི། དཱ་དཱ་ཡ་ཏེ། ཊུ་དྷཱཉ། དྷེཊི། དེ་དྷི་ཡ་ཏེ། མཱ། མེང་། མང་ལ། མེ་མཱི་ཡ་ཏེ། །གི། ཛེ་གི་ཡ་ཏེ། གཱང་གི་ནི། ཛཱ་གཱ་ཡ་ཏེ། པ་བཏུང་བ་ལའོ། །པེ་པཱི་ཡ་ཏེ། པི། པཱ་ཏི་དག་གི་ནི། པ་པཱ་ཡ་ཏེ། ཥཱ། ཏེཊྛི་ཡ་ཏེ། ཁོ། །ཨ་པ་སེ་ཌི་ཡ་ཏེ། ཨོ་ཧཱང་ཀ་ཛེ་ཧི་ཡ་ཏེ། ཨོ་ཧཱང་གི་ནི། ཛཱ་ཧཱ་ཡ་ཏེ། :གླུ། དྷཱ། གྷཱ། དྷཱ་དག་གི་ཨིའོ་ཞེས་པས་ཨཱི་ཉིད་དུ་བྱས་ཏེ་གཉིས་བརྗོད་དོ། །ཛེ་གྷི་ཡ་ཏེ། དེ་དྷཱི་ཡ་ཏེ། ཙཱ། ་་:གྲ་ཧི་ཛུ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་ས་ར་ཎའོ། །ནཱ་མི་མཐའ་རྣམས་ཀྱི་ཡཎ། །ཨཱ་ཡི་ཞེས་པ་ལ་སོགས་པས་རིང་པོའོ། །གཉིས་བརྗོད་དོ༑། ཛེ་ཛཱི་ཡ་ཏེ། བྱེཉ། སུ་པི་སྱ་མི་བྱེཉ། རྣམས་ཀྱི་ཙེ་ཀཱི་ཡི་ཏ་ལའོ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །བེ་བཱི་ཡ་ཏེ། ཧེཉ། :ཨ་བྱཱ་སྟའི་ཡང་ངོ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །།ཛོ་ཧུ་ཡ་ཏེ། ཙིཉ། ཙེ་ཙཱ་ཋཡ་ཏེ། ཊུ་ཨོ་ཤཱི། ཤཱི་?ཡ་ཏིའི་ཡང་ནའོ་ཞེས་པ་ཡང་ན་སཾ་པྲ་སཱ་ར་ཎའོ། །ཤོ་ཤུ་ཡ་ཏེ། ཤེ་ཧཱུཾ། ཡ་ཏེ། ཧི༑ཧིའི་ཙན་ལ་མ་ཡིན་ནོ་ཞེས་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨ་ཧྥུ་ས་ཕ་རོལ་གྱི་ཧའི་གྷ་ཉིད་དོ། །ཛེ་གྷི་ཡ་ཏེ། །ཎཱིཉི། ་ནེ་ནཱི་ཡ་ཏེ། འུ་ཀྲྀཉི། ་ཙེ་ཀཱི་་ཡ་ཏེ། ཥ་ཉི། རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །སེ་ཁཱི་ཥ་ཏེ། ་གླིང༌། སེ་ཧྥི་ལི་ཡ་ཏེ། ཤིང་ཡ་ལ་ཨཱི་ནི་ཨ་ཡིའོ་ཞེས་པས་ཨཱི་ཡིག་གི་ཨ་ཡ་རུ་བསྒྱུར་རོ། །གཉིས་བརྗོད་དོ། །ཤཱ་ཤ་སྤྱི་ཏེ། ཤུ། སྔ་མ་བཞིན་རིང་པོའོ། །ཤོ་ཤུ་ཡ་:ཏི། རླུང་། སེལ་བ་ལའོ། །ནི་ཛ་ཧྥུ་ཡ་ཏེ། །བྷུཉྩ། བོ་ངྷ་ཡ་ཏེ། ལྷུཉ། ལོ་ལུ་ཡ་ཏེ། ཏུཉ། ཥཱ་ཉིད་དོ། །རྒྱུ་མཚན་ལས་ཉིད་དོ། །སོ་:ཥཱ་ཡ་ཏེཧི། །
