ཙོ་ཀཱུ་ཛ་མ། ཙུ་ཥ། ཙོ་ཙཱུ་ཏཱི''ཏི། ཥ་ལས་ཊ་སྡེའོ། །ཙོ་ཙུཏྟི། ཙོ་་ཙུཋ༑ཙོ་ཙཱ་ཥ་ཏི། ཙོ་ཙུ་ཥི་ཏི། ཥེལྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཙོ་ཙུཀྟི། ཙོ་ཙུ། ཙོ་ཙུཥྛ། ཙོ་ཙཱ་བི་མི། ༼ཙོ་ཙུ་རྨི། ཙོ་ཙཱ་ཋ། ཙོ་ཙཱ་ཋཱ། བྷུ་མི། བོ་བྷུ་ཏཱི་ཏི། —་བོ་བྷུཉྩི། །བུ་བྷུ། བོ་བྷུ་ཥ་ཏི། །ཤུནི། ཤོ་ཤུ་ནཱི་ཏི། ་ཤོ་ཤུནྟ། ཤོ་ཤུ་ན་ཏི། ཀྲྀཥ། ་སྔ་་མ་བཞིན་དུ་ལེཨ་བྷཱུ་སའི་རི་ལ་སོགས་པ་ཨཱ་ག་མ་རྣམས་སོ། །སྤྲོ་ཥེལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་དེ་ལ་སྦྱོར་བ་རྣམ་པར་ཕྱེ་བ་ལས་ཙུ་ཊ་སོགས་པ་ལ་ཡོན་ཏན་ཅན་ལ་དབྱངས་ལས་ཕ་རོལ་དུ་ཡང་ན་ཨ་ཡིག་ཨཱ་ག་མའོ། །རེའི་ཡི་གེ་རའོ༑་། ཙ་རཱི་ཀཏྠི༄༅། ཙ་:རི་ཀྲནྟི། ཙ་རི་ཀྲི་*ཏི་ཏི། ཙ་རི་ཀཉྩི། ཙ་རི་ཀནྡྲི༈ཙ་རི་ཀྲི་ཏི་ཏི། ཙཎཊྚེ། ཙརྐནྡྲི། ་་:ཙཎི་ཥཱིཏི། ཙ་རི་ཀྲིཥྛཆ༑ཙ་རི་ཀྲིཋི། ཙ་རི་ཀྲིཥྛཝེ། ཙཀྲིཋ། ཙ་རཱིཀྲྀ་ཨེཥ་ཏི། ཙ་རཱི་ཀྲྀ་ཥ་ཏི། ཙཎི་ཥ་ཏི། པྲྀ་ཥཱ། བ་རཱི་བྲཊི། །བ་རཱི་བཛྫི། །བ་རཱི་བྲི་ཥི་ཏི། །བ་རི་བྲཏྟི། བ་རཱི་བནྟི། བ་རཱི་བྲི་ཏཱི་ཏི། བཛྲ། བརྦཙྪི། བཀྲི་ཏཱི་ཏི། བྲི་ཏུ། །བ་རཱི་བནྟི༄༅། བ་རཱི་བྲི་ཏཱི་ཏི། བརྡནྟིས། བནྡྲ་ཏཱི་ཏི། ཕྱོགས་གཅིག་ལ་རྣམ་པར་གཞག་ྋཔ་རྣམ་པར་བཤད་པའི་དོན་ལས་ཏིའི་རྐྱེན་ལ་ཨིཊ་མེད་དོ། །བ་རི་བྲིཏྟ། བ་རི་བྲིཏྟ། བཧྲིཏྟ། བ་རཱི་བྲི་ཏ་ཏི། བཀྲི་ཏ་ཏི། བ་རི་བནྟི། བ་རཱི་བྲི་ཏཱི་ཥི། །བ་རི་བནྟི། བ་རི་བྲི་ཏཱི་ཏི། བནྡནྟི། བཏྲི་ཏཱི་ཏི། བ་རི་བྲིཏྠ། བ་རི་བྲིཏྠ། བཤྲཱིཏྠ། བ་རཱི་བྲིཏྠ། བནྡྲིཏྠ། བ་རཱི་བཛྫི། ཟོབ་རི་བྲི་ཏི་མི། བ་རི་བཛྫི། བ་རི་བྲི་ཏི་མི། :བརྦ་དྷཱི། བརྗི་ཏི་མི། །བ་རི་བྲི་ཏཱ། བ་རི་བྲི་ཏཱ། :：བ་རི་བྲི་ཏཱ། ་བཀྲི་ཏཱ། བ་རཱི་བྲིཏྠ། བ་རི་བྲིpཚེ། བ་ཧྲིp། །ནྲྀ་ཏཱི། ན་རི་ནནྟི། ན་རཱ་ནི་ཏཱི་།
