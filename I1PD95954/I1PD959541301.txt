:ལ་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའི་ཀ་ཀ། ཡུ་བཱ་ཧཱུཾ། ཨ་བཱ་ཧཱུཾ། སྔ་མ་བཞིན་དུ་རིང་པོ་ཉིད་དེ། ཡུནྣཱ་བྷི། ཨསྨཱ་བྷི། གཅིག་ཉིད་དག་ལ་ཏཱན་མན་དག་སྟེ། དེ་ལ་ཏུ་དྡྷཾ་དང་མ་ཧཱུྃའོ་ཞེས་པས་ཏཱན་་དང་མན་དག་ནི་ཏུ་བློ་དང་མ་ཧུཾ་ཞེས་པ་འདི་དག་ཏུ་འགྱུར་རོ། །ཏུ་བྷཾ། །མ་ཧཱུཾ། ཡུ་བཱ་དྡྷཾ། ཨ་བཱ་བྷཾ། བྷུས་ནི་ཨ་དྷིའོ་ཞེས་པས་བྱས་ནི་ཨ་བྷཾ་ཏུ་འགྱུར་ཏེ། ཡུ་དྡྷཾ། ཨ་སྨདྡྷཾ། གཅིག་ཉིད་ལ་ཏཱཏ་མད་དག་སྟེ། ལྔ་པ་ནི་གཉིས་ཉིད་་མ་ཡིན་པ་ལ་ཨཱཏའོ་ཞེས་པས་ང་སི་ཨཱཏ་དུ་འགྱུར་རོ། །ཏཱཏ། །མཏི། ་ཡུ་བཱ་ཧཱུཾ། ཨཱ་བཱ་ཧཱུཾ། ལྔ་པ་ནི་གཉིས་ཉིད་མ་ཡིན་པ་ལ་ཨཏའོ་ཞེས་པས་ས་ནི་་་ཨཏ་དུ་འགྱུར་ཏེ། ཡུད་ཨསྨད་གཅིག་ཉིད་ལ་ཏཱཏ་མད་དག་སྟེ་ངས་ལ་ཏ་་་་དང་མ་མའོ་ཞེས་པས་ཏཱཏ་མད་དག་རྣམ་དབྱེ་དང་བཅས་པ་ཏ་བ་དང་མ་མ་ཞེས་པ་འདི་དག་ཏུ་འགྱུར་རོ། །ཏ་བ། མ་མ། ཡུ་ཝ་ཝོ། མ་བསྒྱུར་བ་ལ་ཨེ་་་ཉིད་དོ་ཞེས་པས་ཨེ་ཉིད་དེ། ཨེ་ཨ་ཡའོ། །ཚོ། ཡུ་བ་ཡོ། ཨ་བ་ཡོ། མཐའ་དབྱི་ཞིང༌། སུ་ཨཱ་ག་མ་སྟེ། སཾ་ནི་ཨ་ཀཾ་དུ་བསྒྱུར་རོ། ་། ཡུཏྟཱ་ཀཾ། ཨ་ཀཾ། གཅིག་ཉིད་ལ་ནི་ཏཱཏ་མད་དག་སྟེ། །མཐའ་དབྱི་ཞིང༌། །འོམ་བསྒྱུར་བ་ལ་ཨེ་་ཉིད་དང་ཨེ་ཨ་ཡའོ། །ཏྭཱ་ཡི། མ་ཡི། ཡུ་བ་ཡོ། ཨཱ་བཱ་ཡོ། ཌ་གསལ་བྱེད་དང་པོ་ཅན་ལ་ཨ་ཉིད་དོ་ཞེས་པས་ཨ་ཉིད་དེ། ཡུསྨཱ་སུ། ཨསྨཱ་སུ། །ཀ་བ། ༎་ལ་དཀྱུས་སུ་རབ་ཏུ་གཞག་པ་རྣམས་ལས་གཞན་པའི་ཚིག་དོན་ལ་གཅིག་ནི་་གཅིག་ཚིག་ཀྱང་འཆད་པ་ཚིག་སྡུད་ཅན་རྣམས་སོ། །ཏཱམ་ཏི་ཀཱནྟུ། མཱ་མ་ཏི་ཀཱནྟ། ཨ་ཏི་ལ་སོགས་པ་དང་ཀཱནྟི་ལ་སོགས་པ་དོན་ལ་གཉིས་པའི་ཆེད་དུའོ་ཞེས་པས་་ཚིག་སྡུད་དོ། །དེར་གནས་རྣམ་དབྱེ་རྣམས་དབྱི་བར་བྱའོ་ཞེས་པས་རྣམ་དབྱེ་དབྱིའོ། །གང་དང་གང་གི་རྟགས་སུ་གྱུར་པ་སྔ་མ་རྣམས་ནི་རང་བཞིན་ཏེ། དེ་དང་དེ་དག་ལ་སྦྱིན་པར་བྱ་བ་ནི་སི་ལ་སོགས་པ་རྣམས་སོ། །མིང་ལེ་གསུམ་པར་་།
