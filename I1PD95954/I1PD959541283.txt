:ལ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའི༌ཀ་པ་ལས་ཏ་མ་ལ་སོགས་པ་ཞེས་པ་འདིར་ཡོན་ཏན་ལས་ཨིཏྠ་དང་ཨི་ཡནྣ་དག་གོ་ས་ཞེས་པས་ཨིཊྚ་དང་ཨི་ཡརྨ་དག་གི་རྐྱེན་དུ་འགྱུར་བའི་མཚན་ཉིད་ཅན་ཉིད་ལས་་ཨིཊྚའི་རྐྱེན་ལའང་དཔེར་བརྗོད་པ་བྱ་སྟེ། །བསླུ་མནྟ་བིན་རྣམས་ཀྱིས་འཇིག་གོ་་ཞེས་པ་སྟེ། ཚོགས་པས་བཀག་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །སྐྱེས་པ་ལ། །པྲ་ཛི། པྲ་ཛིཏྟཾ། པྲ་ཛིཏྟཱ། བུད་མེད་ལ། པྲ་ཛིཏྟཱ། མ་ནིང་ལ། པྲ་ཛིཤྩ། ཕྲ་ཛིབྦེ༑ལ་པྲ་ཛི་ཋནི། ཨི་ཡནྣ་ལ་ནི་རིང་པོ་ཉིད་དེ། སི་དབྱི་བ་དང་བརྩེགས་པའི་མཐའ་དབྱི་བ་དག་གོ། །པྲ་ཛཱི་ཡཱན། པྲ་ཛཱི་ཡནྣཾ། ཌཱ་པྲ་ཛི་ཡཱམྨ། ཟ་ཞེས་པ་ལ་སོགས་པ་དང་། བུད་མེད་ལ་ཨཱི་རྐྱེན་ཏེ། པྲ་ཛཱི་ཡ་སཱི། ཋམ་ནིང་ལ། པྲ་ཛཱི་ཡ། པྲ་ཛཱི་ཡ་སཱི། པྲ་ཛི་ཡཱནྟི། དེ་བཞིན་དུ། ཚ་པྲ་ཤ་སྐྱ་ཞེས་པ་འདིར་ཡང་འདི་དག་ལས་ཕུལ་དུ་བྱུང་བས་པྲ་ཤ་སྱ་ཞེས་པ་སྟེ། པྲ་ཤ་སྱའི་སྒྲ་ནི་ཤ་རུ་བསྒྱུར་བའོ། །ཐེཥྛ། ཐེ་ཡཱན། ་བུད་མེད་ལ། ཐེཥྛ། །ཐེ་ཡ་སཱི། མ་ནིང་ལ། ཐེཏྟཾ། ཤེཥྛ། ཐེཥྛ་ནཱི། ཤེ་ཡ། ཐེ་ཡ་སཱི། ཎོ་ཐེ་ཡཱནྟི། པ་ཊུའི་སྒྲ་ལས་ཀྱང་ཨིཊྚ་དང་ཨི་ཡནྡ་དག་སྟེ། །ཨིན་ལ་རྟགས་ཀྱི་ཡི་གེ་གཅིག་མ་ཡིན་པའི་མཐའི་དབྱངས་དང་པོ་དབྱིའོ། །པ་ཊི་བྷི། པ་ཊི་ཡཱན། བུད་་མེད་ལ། །པ་ཊེ། པ་ཊཱི་ཡ་སཱི། མ་ནིང་ལ། པ་ཊིཊཾ། པ་ཊི་ཡ། །པ་ཊཱི་ཡ་སཱི། པ་ཊི་ཡཱནྟི། དེ་བཞིན་དུ། ལ་གྷིཥྛ། ལ་གྷི་ཡཱན། ཨ་ནི་ཀ་བཱ་ཊ་དག་ནེ་ད་དང་སཱ་དྷ་ཞེས་པས་ནེ་ད་དང་ས་དྷཱ་རུ་བསྒྱུར་བ་སྟེ། ནེ་དིཋ། ནེ་དཱི་ཡཱན། སཱ་དྷིཥྛ། ས་དྷཱི་ཡཱན། ཞེས་པ་དང༌། དེ་བཞིན་དུ། ཨ་་ཝིཊྛ༑ཨམྦི་ཡཱན། སྦ་བི་ཥྛ། །སྦ་བི་ཡཱན། ད་བིཥྛ། ད་བཱི་ཡཱན། ཀྱི་དྷི་ཥྛ། ཀྱི་དྷཱི་ཡཱན། །སྲི་ཡ་དང་། སྦྲ་དང་། སྤྱི་ར་དང་། ཨུ་རུ་དང༌། གླུ་རུ་དང་། བ་ཧུ་དང་། ཏྲཔྲ་དང་། །འདིརྒྱུ་དང༌། ཧྲ་སཱ་དང༌། །བྲོ་དྷ་།
