དེ་ན་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི། ཨ་ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཞེས་པ་ལ་སོགས་པས་ཨན་གྱི་ཨ་ཡིག་དབྱིའོ། །ཨནྣཱ༑་ཨམྦི་ཧཱུཾ། ཨ་སྦི་བྷི། ཨམྦྷེ། /ཨམྦི་ཧཱུཾ། ཨསྦི་ཧྥུ། ཨམྦྷ། ཨསྦི་ཧཱུཾ། ཨམྦི་། ཨམྦྷ། ཨམྦྷེ། ཨནྣཾ། བདུན་པའི་གཅིག་གི་ཚིག་ལ་ཨཱི་དང་ང་དག་ལ་ཡང་ནའོ་ཞེས་རྣམ་པར་བརྟག་པས་ཨ་ཡིག་དབྱིའོ། །ཨམྦྷི། ཨསྦྲ་ནི། ཨདྡྷ། ཥ་ཉིད་དོ། །ཨསྦི་ཏུ། ཀུན་དུ་བོད་པ་ལ། ཌཧེ་ཨསྦི། ཧེ་ཨསྦི་ནཱི། ཧེ་ཨསྠཱ་ནི། །དེ་བཞིན་དུ་ད་དྷི་ཤཤྲཱི། ཨཀྵ། ནེ་ཏྭཱ་ནི་རྣམས་དང༌། ད་དྷི། ད་དྷི་ནཱི། ད་དྷི་ནཱི། སླར་ཡང་། ད་དྷི། ད་དྷི་ནི། ད་དྷཱ་ནི། ད་། ཞེས་པ་ལ་ཨ་ཡིག་དབྱི་བ་བྱས་པ་ན། ་པོདྷུཊ་རྣམས་ཀྱི་གསུམ་པའོ་ཞེས་པས་དྷ་ཡིག་ཏུ་འགྱུར་བ་མ་ཡིན་གྱི། ::ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཨན་གྱི་ཨ་དབྱིའོ་ཞེས་པས་ཨ་ཡིག་ཞིག་པ་དང་འདྲ་བར་འགྱུར་རོ། །བསྟུའི་སྒྲ་ནི་བཱ་རིའི་སྒྲ་དང་འདྲ་བར་ཐོབ་བོ། །བསྟུ། བསྡུ་ནཱི། བསྐྱུ་ནི་་་ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཛཱ་ནུ། ཨམྦྷ། ཛནྟུ། ཨསྟུ། མ་དྷུ། མy༑ཞེས་པ་ལ་སོགས་པས་གཞན་རྣམས་ཀྱང་ངོ༌། །ཞེས་པ་མ་ནིང་གི་་རྟགས་དབྱངས་ཀྱི་མཐའ་ཅན་རྣམས་སོ། །་། ཕྲན་བུ་ཞི་བློ་སྒོ་ཡོན། /ན་ཆུང་དབང་་། ཁྲོ་ལྔ། ཚོད་ཚོ་བ་དེ་ནས་མ་ནིང་གི་རྟགས་གསལ་བྱེད་ཀྱི་མཐའ་ཅན་རྣམས་བརྗོད་པར་བྱ་སྟེ། ཛ་གདྲ་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། མ་ནིང་ལས་སི་ཨམ་དག་དབྱི་ཞིང་མ་ནིང་ལས་སི་ཨམ་དག་ཕྱིས་པ་ལ་ཡང་ཞེས་བསྒྲེ་བའི་སྟོབས་ལས་ཏེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཏ་ཡིག་གམ་ད་ཡིག་གོ། །ཛ་གཏ། ཛ་གད། ཨོ་ཨཱིའོ། །ཛ་ག་ཏཱི། ཤི་རུ་བསྒྱུར་ཞིང་དྷཊ་དང་དབྱངས་ལས་གྲུཊ་ལ་ནུའོ་ཞེས་པས་ནུ་ཨཱ་་ག་མའོ། །ཛ་གནྟི། སླར་ཡང་། ཛ་གཏ། ཛ་གད། ཛ་ག་ཏཱི། །ཛ་གནྟི། —།
