ལ་ཏུ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིན་:དིལྔའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། །དིཝའི་གསལ་བྱེད་ལ་ཨུཏ་དོ་ཞེས་པ་སྤྱིར་བཏང་ནས། སི་ལ་ཨོའོ་ཞེས་པས་ཝ་ཡིག་ཨོ་རུ་འགྱུར་རོ། །བློ། དི་ལོ། དི་ཝ། ཨམ་ལ་ཡང་ཨཱ་ནའོ་ཞེས་རྣམ་པར་བརྟག་པས་ཨ་ཡིག་གོ། །དྡྷཾ། དི་ཝཾ། དི་ལོ། དི་ཝ། དི་། གསལ་བྱེད་ལ་ནི་དི་ཝའི་གསལ་བྱེད་ལ་ཨུཏའོ་ཞེས་བཤད་པར་བྱའོ། །ཏྭ་ཧཱུཾ། །དྷུ་བྷི། །ཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་བྱོ། ཧི་དི་ཝོ། ཧི་དི་ཝ། ཞེས་པའོ། །དེ་སྟོན་པར་བྱེད་དོ་ཞེས་པ་དི་ཤ་ལས་ར་ཏི་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་ཀྭིཔ་སྟེ། དེ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི་གསལ་བྱེད་ལས་ཀྱང་ངོ་ཞེས་པས་སི་དབྱི་ཞིང༌། རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ་དྲི་ཤ་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ག་ཉིད་དེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཀ་ཡིག་གམ་ག་ཡིག་དག་གོ། །།དིཀ། དིག། དི་ཤོ། དི་ཤ། དི་ཤི། དི་ཤོ། དི་ཤ། དི་ཤཱ། དིདྡྷཾ་དིལླཱི། ཞེས་པ་ལ་སོགས་པ་སྟེ། སུཔ་ལ་ཥ་ཉིད་དུ་བྱས་པ་ན། །སྒྲ་མེད་པ་ལ་དང་པོའོ་ཞེས་པས་ཀ་ཡིག་གོ། །དིཀྵ། ཀུན་དུ་བོད་པ་ལ། ཧེ་དིཀ། ཧེ་དིག། ཧེ་དི་ཤོ། ཧེ་དི་ཤ། དེ་བཞིན་དུ། དྲྀ་ཤ་ལ་སོགས་པ་རྣམས་སོ། །།།པྲ་དྲྀཥཱའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ། ཧ་ཤ་དང་ཥ་དང་ཚ་མཐའ་དང་ཡ་ཛ་ལ་སོགས་པ་རྣམས་ཀྱི་ཌའོ་ཞེས་པས་ཌ་ཡིག་སྟེ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཊ་ཡིག་གམ་ཌ་ཡིག་དག་གོ། །པྲ་བྲིཊ། །པྲ་ཀྲྀཏྲ། པྲ་བྲི་ཏོ། པྲ་ཀྲྀ་ཥ། ཊཔྲ་ཤྲཱི་ཎཾ། ་༌་པྲ་ཀྲྀ་་ཎོ༑པྲ་དྲྀ་ཥ། པྲ་ཀྲྀ་ཥཱ། ཌ་ཉིད་དོ། པྲ་བྲིདྡྷཾ། པཱ་ཤྲིནྟི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། བི་ཀྲུཥ། ཀྲྀཀྵ། རུས་ལ་སོགས་པ་རྣམས་སོ། །།།གང་མིང་གཞིའོ། །ཁྱེའུ། མི་།
