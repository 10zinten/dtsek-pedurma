པ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་?དོར་ནས་ཨིཊ་དང་བཅས་པ་ཤིན་ཏུ་རྣམ་པར་བཤད་པ་ལ་གྲུབ་བོ། ཀེ། ཞེས་པས་་ཡང་ན་དྷའི་ལྷ་ཉིད་དོ། །:﹙ཤི་ཡི་ཊཱི་བྷཾ། ཤི་ཡི་ཊཱི་ཀྐི༄༅། ཤིང་། ཤ་ཡི་ཊྛིཏྠ། ང་། ཌ་ཡི་ཊཱི། དཱི་དྷིང་། བེ་བིང་། དྲི་དྷཱ། བེ་བཱི་དག་གི་ཨི་ཡིག་ཡ་ཡིག་དག་ལའོ། །ཞེས་པས་མཐའ་དབྱིའོ། །དཱི་དྷི་ཥིཉྩ། བེ་བི་ཥིཉྩ། ཐུ། ཐུ་ཡཱཏ། ཧུ། ཧཱུ་ཡཱཏ། ཙུང་། སྦྱོ་ཊཱིཋ། །ཕུང་། པོ་ཏཱི། རླུང༌། བློ་ཊཱི། ཨུང་། ཨོ། ཏིཥཱ། །ཥཱཉ། །སཱུ་ཡཱཏ། སོ་ཥིཉྩ། བྷུཉ། སྐུ་ཡཱཏ། སྟོ་ཏཱི། ཨུཎྚཉ། ཨུཙྪཱ་ཡཱཏ། །ཨིཊ་ཨཱ་ཅིག་མའོ། །རྒྱས་ནཱི་ལ་་་ཡང་ངོ་། །ཞེས་པས་ཡང་ཡིག་གིས་མ་བརྗོད་པ་བསྡུས་པའི་དོན་ཉིད་ལས་ཨིཊའོ༑། ཡང་ན་ཡོན་ཏན་ནོ། །ཨུཙྪཱ་བི་ཏི། ཨུཙྪཱ་བི་ཊཱིཋ། བྷཱུ། བྷཱུ་ཡཱཏ། ཏུ་དེད་པ་ལའོ། །སཱུ་ཡཱཏ། པུང་། པ་བི་ཊཱི། དེ་བཞིན་དུ་མུཉའོ༑། པཱུཉ། པཱུ་ཡཱཏ། པ་བི་ཥིཉྩོ་དེ་བཞིན་དུ་ལུཉ་དང་ཀུང་དང༌། རླུད་དྲུཉའོ། །ལྟུང༌། སཱ་ར་ཏི། པཱུ་ཏི། ཞེས་པ་ལ་སོགས་པས་ཡང་ན་ཨིཊའོ། སོཊྛིཏྠ། ས་བི་ཏཱིཤྩ། སྲི། ་ཡན་ཨཱ་ཤཱིཥ་དག་ལ་ཡ་ལའོ། །ཞེས་པས་ཡ་ལ་ཨི་ཡིག་མཐའོ། །རོའི་ཡི་གེ་རའོ། །ནཱ་མི་མཐའ་རྣམས་ཀྱི་ཡན་ཨཱ་ཡི་ཞེས་པ་ལ་སོགས་པས་རིང་པོ་མེད་དོ། །ཨ་ཡིག་ཨཱ་ག་མ་སྒྲུབ་པ་ཉིད་ལས་གཞན་དུ་ཡོན་ཏན་མེད་པ་ལ་ཨིརའོ། །།ཞེས་པས་འདིར་སྔ་མའི་མདོ་ལ་སོགས་པ་འཇུག་པར་འདོད་དོ། །སྲི། སྲི་ཡཱཊི། །ཕྲེ། ཤྲཱི་ཡཱཊི། དྷིང་། སིཙ་ཨཱ་ཤཱིཥ་དག་ལ་ཡང་ཨཱཏྨ་ནེ་ལའོ། །ཞེས་སྤྱིར་བཏང་ནས་རྀཏ་མཐའ་རྣམས་་་ཀྱིའང་ངོ་ཞེས་པས་ཨིཊ་མེད་པ་ལ་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཛཾ་ཏིཥྛ། དྲྀང༌། དྲྀ་ཏི། སྲིང་། བྱཱ་པྲ་ཏཱིཋ། སྲྀང་། ཨཱ་ཤཱིཥ་ཨརྱ་ཏ་ནཱི་དག་ལ་ཡང་་མྲངའི་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །མྲྀ་ཊྛིཏྠ། རྔུ་་་—།
