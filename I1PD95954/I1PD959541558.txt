ཨིཊ་ཨཱ་ག་མའོ། །དྷཱ་བི་ཊཱིཥྛ། ཌྷ་ཉིད་རྣམ་པར་བརྟག་པ་སྔ་མ་བཞིན་ནོ། །།།།དྷཱ་བི་ཌི་ཌྷ། དྷཱ་བི་ཥི་དྡྷཾ། ཨ་ས། འབར་བ་དང་ལེན་པ་དག་ལ་ཡང་ངོ༌། །ཨ་སཱུཏ། ཨ་སཱི་ཥིཉྩ། བུ་དྷིར། བོ་དྷི་ཥིཉྫ། གུ་ཧུ། །གུ་ཧཱཏ། །ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་ཨིཊའོ། །གོ་ཧིའི་ཨུ་པ་དྷའི་ཨུཏའོ། །།གོ་ཧིའི་ཏིཥྛ། གཱ་ཧི་ཁཱི་ཌྷགཱ་ཧི་ཁཱི་ཧཱུྃ། །ཌྷ་ཉིད་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །།གྷཱ་ཁཱི། བེན། བྱེཉ། ཧྭཉ།  ི་ཨ་ཉིད་དོ། །སྭཱ་པི་བ་ཙི་ཞེས་པ་ལ་སོགས་པས་ཡོན་ཏན་མེད་པ་ལ་སཾ་པྲ་སཱ་ར་ཎའོ། །རིང་པོ་ཉིད་སྔ་མ་བཞིན་ནོ། །།ཨུ་ཡཱཏ། ཏབ་སཱིཊ། སཱམཱི་ཡཱཏ། །སཾ་བྱཱ་སཱིཏྠ། ཧུ་ཡཱཏ། །ཧཱ་སཱིཥཱ། ཤཱ་སུ༑ཤཱ་སིའི་ཨུ་པ་དྷཱའི་ཨཎ་དང་གསལ་བྱེད་དག་ལ་ཨིཏའོ། །ཞེས་པས་ཨུ་པ་དྷཱའི་ཨི་ཉིད་དོ། །ཤཱ་སི་བ་སི་གྷ་སི་རྣམས་ཀྱི་ཡང་ངོ༌། །ཞེས་པས་ཥ་ཉིད་དོ། །ཤི་བྷཱུཏ། ཙཀླུང༌། ་ཙགླིངའི་ཁྱུཉའོ། །བརྩེགས་པ་དང་པོའི་གནས་མ་ཡིན་་་པ་ལ་ཡང་ནའོ། །ཞེས་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨ་ཡིག་གི་ཡོན་ཏན་མེད་པ་ལ་ཡང་ན་ཨེ་ཉིད་དོ། །ཨཱ་ཁྱེ་ཡཱཏ། ཨཱ་ཥཱ་ཡཱཏ། དི་བུ་ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །།དཱི་བྱཱཏ། ཨིཊ་ཨཱ་ག་མའོ། །དེ་བི་ཥིཉྩ། །ཥི་བུ། སཱི་བྱཱཏ། ་ཨུ་ཙྪི་དིར། :：ཨུ་ཏྲ་དིར། ཀྲི་ཏ་ལ་སོགས་པ་ལས་ས་ལ་བྱེད་ཡང་ན་བ་ཡང་སྟེ་སི་ཙ་ལ་མ་ཡིན་ནོ། །ཞེས་པས་ཡང་ན་ཨིཊའོ། །ཙྪརི་ཊྛིཏྠ། །ཙྪིནྡྲིཋ༄༅། །ཀྲྀ་དྷཱཏ། ཏཇི་ཏིཥྛ། དྲྀཏི། གྲ་ཧ། གྲ་ཧི་ཚྭ་ཞེས་པ་ལ་་་སོགས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །གྲི་བྱཱཏ། ཨིཊ་ཨཱ་ག་མའོ། །གྲ་ཧི་ལས་པ་་་་རོགཱ་མིན་པ་ལ་ཨིཊའི་རིང་པོའོ། །ཞེས་པས་ཨིཊའི་རིང་པོའོ། །གྲ་ཧཱི་ཊཱིཋ། །ཌྷ་ཉིད་རྣམ་པར་བརྟག་པ་སྔ་མ་བཞིན་ནོ། །གྲ་ཧཱི་ཏཱི་ནཱཾ། གྲ་ཧཱི་ཎཱི་ཧཱུྃ། །དེ་་ལ་རྣམ་པར་གཞག་ཅེཔ་ཡང་ན་སྤྱིར་བཏང་ནས། གཱ་པཱུ་ལ་སོགས་པ་རྣམས་ལས་—།
