:བསྟན་འགྱུར། སྒྲ་མདོ། ༽ཏི་རིང་པོའོ། །ཞེས་པ་འམ་ནཱ་མིའི་མཐའ་རྣམས་ཀྱི་ཞེས་པ་ལ་སོགས་པས་རིང་་་པོའོ༑། ཤུ་ཡཱཏ། ཛཱ། བ་ཤ་མཛེས་པ་ལའོ། །བྱ་ཙ། ཀུཊ་སོགས་སོ། །བཤྩ། གྲ་ཧི་ཛྙཱ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །ཛཱི་ཡཱཏ། ཨ་བྱཱཏ། བི་ཙཱཏ། དྲྀ་ཙཱཏ། ཨེ་དྷ། ཨིཊ་ཨཱ་ག་མའོ། །།ཨི་ཙྪཱདྷི་ཏི། སྤཛྷ། སྤདྷི་ཊཱིཊ། བཱ་དྷི། བཱ་དྷི་ཏིཥྛ། བ་དི། བནྟི་ཏཱི། ཨཱ་ས། ཨཱ་སི་ཊཱི། མུ་ད། མོ་དི་ཏཱིཋ། མེ་ད། མེ་དི་ཊཱིཋ། ས་ཏྱཱཏ། བྱོ་ཏི་ཥིཉྩ། །སྲནྟུ། སཾ་:སཱིཥྛ། བྷནྡྷལ། ཤྲཱི་སི་ཥིཉྩ། དྷཱནྟུ། དྷཾ་སི་ཏཱི། ས་ཧ། ས་ཧི་ཁཱི། ནཱ་མི་མཐའི་བྱིངས་ལས་ཞེས་པའི་མིང་སྔ་མ་ཉིད་ཀྱིས་རྟག་ཏུ་ཧ། ་ཡ། བ། ར། ལ། ན་རྣམས་ཀྱི་ཡང་ཨིཊ་དང་བཅས་པ་ལ་ཌྷ་ཉིད་རྣམ་པར་བརྟག་པའོ༑། སཾ་ཧི་ཏི་ནཱཾ། ས་ཧི་ཏི་ནཱཾ། ཨ་ཡ་པ་རཱ་སྔོན་དུའོ། །ཉེར་བསྒྱུར་གྱི་ཨ་ཡ་ཏི་ལའོ། །ཞེས་པས་རེ་ཕའི་ལ་ཉིད་དོ། ་། པ་ལཱ་ཡི་ཁཱི། པ་ལཱ་ཡི་ཏི་ཞྭ། པ་ལཱ་ཡི་ཏི་ནཱཾ། བ་ལ། བལླ་ཡང་ངོ༌། །སཾ་བ་ལི་ཁཱི། །སཾ་བ་་ལི་:ཏི་ནཱཾ། སཾ་བ་ལི་ཁཱི་ཧཱུཾ། ཨི་ལ། །སྲེ་ལི་ཊཱི། སྲེ་ལི་ཏི་ནཱཾ། :སྲེ་ལི་་ཏིཥཱ་དཾ། བྲྀ་ཏུ། བརྟི་ཏིཥྛ། དྲྀ་དྷུ། བདྷི་ཊཱིཋ། ཤྲདྷུ། ༽ཤདྷི་ཊཱི། སྱཱནྟུ། ་པནྡི་ཁཱི། ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ལས་ཡང་ན་ཨིཊའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །སཾཏྠིབྦཙེ། ཀྲྀ་གཱི། ཀྲྀ་པའི་རའི་མའོ་ཞེས་པས་རེ་ཕའི་ལ་ཉིད་དོ། །ཀལླི་ཙྪིཥིཉྩ། ཀྲི་པཱི། གཱ་ཧུ། གཱ་ཧི་ཏི། གཱ་ཧི་:ཁཱི་བྷཾ?། གཱ་ཧི་ཁཱི་ཧཱུཾ། ཧའི་ལྷའོ། །བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་པས་དང་པོ་གའི་བཞི་པ་གྷ་ཉིད་དོ། །ཀ་ཉིད་དང་ཥ་ཉིད་སྔ་མ་བཞིན་ནོ། །གཀྐི། ཨ་ཤུ། ཨ་ཤི་ཊཱི། ཚ་ཤ་དག་གི་འང་ངོ༌། །ཞེས་པས་ཥ་ཉིད་དོ། །ཨཀྵིཋ། །དྷཱ་བིུ། དྷཱ་བྷཱུཏི། །
