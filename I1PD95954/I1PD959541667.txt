བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི༎འི་ཨ་ཛ་རི་གྲི་དྷུ། ར་ལ་སོགས་པས་སྔ་མ་བཞིན་ནོ། །ཨ་ཛ་རཱི་གྷ། ཨ་ཛ་རི་གྷ༑:：ཨ་ཛརྒ། ?ཨ་ཛརྙ། ་ཨ་ཛ་རཱི་གྷརྟ། །ཨ་ཛ་རི་རྒརྟི། ཨ་:：ཛརྒ་ཏི། ཨ་ཛ་རཱི་གྷནྡ༄༅། ཨ་ཛ་རི་གྷརྡ། ཨ་ཛ་རི་གྲི་དྷཱི། ཨ་ཛ་རི་་གི་དྷཱི། ཨ་ཛ་རྦ་དྷི། །ཨ་ཛ་རཱི་གྲི་དྡྷཾ། ཨ་ཛ་རི་གྲི་དྡྷཾ། ཨ་ཛརྦ་དྷཾ། ཨ་ཛ་རཱི:;གྲི་ཌྷི། ཨ་ཛ་རཱི་:གྲི་ཌྷེ། ཨ་ཛ་ཧྲི་ཌྷེ། ཨ་ཛ་རཱི་གྲི་དྷཾ། ཨ་ཛ་རི་གྲི་དྷཾ། :ཨ་ཛ་རི་གྲི་གྷཾ། ཨ་ཛགྲིདྷཾ། ཨ་ཛ་རཱི་གྲི་དྷཱ། ཨ་ཛ་རི་གྲི་དྷ། ཨ་ཛགྲིརྡ༑ཨ་ཛ་རཱི་གྲིདྨ། ཨ་ཛ་རི་གྲི་དྨ། ཨ་ཛ་རྦདྡྷ། །ཀུརྡ། ནཱ་་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཨ་ཙ་ཀུནྟ། ཨ་ཙོ་ཀུརྡ། ཨ་ཙོ་ཀཱུརྡ་ཏི། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཨ་ཙོ་ཀཱུཏྟཾ། ཨ་ཙོ་ཀུརུཾ། ཨ་ཙོ་ཀཱུ། རྗེ་ཨ་ཙ་ཀུནྟ། ཨ་ཙོ་ཀུརྡ། །ཨ་ཙོ་ཀུརཱི། ཨ་ཙོ་ཀུ་ཏྟཾ། ཨ་ཙོ་ྋཀུཏྟཾ། ཨ་ཙོ་ཀུརུཾ། ཨ་ཙོ་ཀུརྡ། ཨ་ཙོ་ཀུརྨི། ཨ་ནཱ་ནཏ། ཨཨ་ནཱ་ན་རཱིཏ། །་ཨ་ནཱ་ནནྟཾ། ཀྵ་ཨ་ནཱ་ནརྡུ། ཨ་ནཱ་ནཱ། ༼ཨ་ནཱ་ནཏ༑ཨ་ནཱ་ནརྡ། །ཨ་ནཱ་ནདི། ཨ་ནཱ་ནནྟཾ། ཨ་ནཱ་ནནྟ། ཨ་ནཱ་ནརྡ༄༅། །ཨ་ནཱ་ནརྡ། ཨ་ནཱ་ན། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨ་པཱ་པ་ཙུ་ཏ། ཨ་ལོ་ལུ་ཡཏ། ཨ་བོ་ངྷ་ཡཏ། །འ?_ག་། །ཁ་ཏོ། །Pagབོདཨ་རྱ་ཏ་ནཱི། ཨ་ཙེ་ཙ་ཡིཏ། ཨ་ལོ་:བ་ལ་པཱ་ཏི། སྔ་མ་བཞིན་རང་བཞིན་གཟུང་ེབ་ལ་ཙེ་ཀྲི་ཡི་ཏ་ཞིག་པའི་མཐའི་འང་གཟུང་ངོ་ཞེས་པའི་ཚིག་གི་་ཨནའི་ཨུསའོ། །ཨ་བོ་བྷུ་ཏ། ཨ་བོ་བྷཱུ་ཏཱནྟི། ཨ་བོ་བླུ་བུ། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨིཙ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །།ཨ་ཏཱ་ཏཱ་ཡི། ལས་ལ་གཉིས་ཚིག་ལ་སོགས་པའང་ངོ་། །ཨ་ཏཱ་ཏཱ་ཡི་ཥཱ་ཏྭཾ། དབྱངས་གཅིག་པ་མ་ཡིན་པ་ལས་ཨིཊ་ཨཱ་ག་མའོ། །།ཨ་ཡིག་གི་དབྱིའོ་ཞེས་པས་ཀྱང་ཨཱ་དབྱིའོ་—།
