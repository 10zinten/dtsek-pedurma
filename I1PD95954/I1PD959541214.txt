སྭ་ཡཾ་བྷུ་ཝ། སྭ་ཡཾ་བྷུ་ཝོ། ་སྭ་ཡཾ་བྷུ་ཝམ། །སྭ་ཡཾ་བྷུ་ཝི། སྭ་ཡཾ་བྷུ་ཝོ། །སྭ་ཡཾ་བྷུ་ཥཱ། ཀུན་དུ་བོད་པ་ལ་ནི། །ཧེ་སྭ་ཡཾ་བྷུ། ཧེ་སྭ་ཡཾ་བྷུ་ལོ། ཧེ་སཱ་་་ཡཾ་བྷུ་ཝ། དེ་བཞིན་དུ་ཨཱཏྨ་བྷཱུ། །མ་ནོ་བྷུ། །པྲ་ཏི་ཧྥུ་ཞེས་པ་འདི་ལྟ་བུ་ལ་་་སོགས་པ་རྣམས་སོ། །ཁ་ལ་པཱུའི་སྒྲ་ལ་ཁྱད་པར་ནི། ཁ་ལ་པུ་དག་པར་བྱེད་པ་ལའོ་ཞེས་པས་ཀྭིཔ་སྟེ། ཉེ་བའི་ཚིག་གིས་ཚིག་སྡུད་དོ། །སི་ལ་ནི་ཁ་ལ་པཱུ། ཨུ་ཝ་ཐོབ་མོད། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དོ། །ཁ་ལ་ཧཱུཾ། ཁ་ལ་སཱ། ཁ་ལ་ཧཱུཾ། ཁ་ལ་ཧཱུཾ། ཁ་ལ་སཱ། ཁ་ལ་སཱ། ཁ་ལ་སུ་བྷཱུམ། །ཁ་ལ་སཱུ་བྷི། ཁ་ལ་སྭ། པོ་ཁ་ལ་སཱུ་བྷཱུམ། །ཁ་ལ་སཱུ་ཧྥུ། ཁ་ལ་སཱ༑ཀ་ཁ་ལ་སཱུ་བྷཱུམ། ཁ་ལ་སཱུ་ཧྥུ། ཁ་ལ་སཱ། ཁ་ལ་ཧཱུཾ། ཁ་ལ་སྭཱམ། ཁ་ལ་སཱི། ཁ་ལ་སྦོ། ་ཁ་ལ་པུ་ཥཱ། ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་ཁ་ལ་པཱུ། ཧེ་ཁ་ལ་ཧཱུཾ། ཧེ་ཁ་ལ་སཱ། །ཝནཱ་བྷཱུའི་སྒྲ་ལའང་ཁྱད་པར་ནི། བྷཱུའི་ཝནྟཱ་་་དང་པུ་ནར་བྷུ་མ་ཡིན་པ་ཞེས་པས། འདི་ཝནྟཱ་བྷཱུའི་སྒྲ་ལ་ནི་བཀག་པའི་ཕྱིར། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་དབྱངས་དང་པོ་ལ་ཝ་ཉིད་དོ། །ཝནྟཱ་བྷུཉྩ། ཝནྣཱ་བྷོ། ཝནྟ་བྷཱ། zaxi། དེ། སྐྱེད་ཀྱི་རྟ་ཀློག་ཀ་ཐོ་:པི་ཏྲཱའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ཏེ། དང་པོའི་གཅིག་ལ་སི་ལ་ཨཱ་སྟེ་སི་དབྱི་བའང་ངོ་ཞེས་པས་རཾ་ཡིག་ཏུ་འགྱུར་ཞིང་སི་ཕ་རོལ་ལ་སི་དབྱི་བའང་ངོ༌། །པཱི་ཏཱ། ?ང་ལ་ཨར་རོ་ཞེས་པ་སྤྱིར་བཏང་ནས། གྷཊ་ལའང་ངོ་ཞེས་པས་རཾ་ཡིག་ཨར་རོ། །བི་ཏ་རོ། པི་ཏ་ར། པི་ཏ་ར་མ། བི་ཏ་རོ། །ཤས་ལ་་ཨགི་བཞིན་ནོ་ཞེས་པས་ཨགྣི་དང་འདྲ་བར་འགྱུར་ཏེ། ཤས་ཀྱི་ཨ་ཡིག་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཤས་ཀྱི་ཨ་ཡིག་ར་ཡིག་ཏུ་ཐོབ་ཅིང་། སའང་ན་རུ་འགྱུར་ལ། མཚུངས་པ་ལས་རིང་པོ་སྟེ། རིང་པོ་ལས་ཕ་རོལ་དབྱིའོ། །པི་ཧྲིན། ར་ཉིད་_།
