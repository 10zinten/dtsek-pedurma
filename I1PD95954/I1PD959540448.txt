ཨཱ་རྱ་ཙ་ར། ''ལྷན་ཅིག་མ་ཡིན་པར་བརྗོད་པའི་དོན་ལ་ཨེ་ཀའི་སྒྲ་ལས་ཀ་དང་ཨ་ཀིན་ཞེས་པ་འདི་དག་འདོད་དེ། ཨེ་ཀ་ཀ། ཨེ་ཀཱ་:ཀཱི་ཀ། འདིར་དེའི་བརྗོད་པ་ཞེས་པའི་དོན་ལ་ཙུཉྩ་དང་ཙ་ན་དག་འདོད་དེ། སྐྲ་ལས་བརྗོད་པ་ནི། ཀེ་ས་ཙུཉྩ། ཀེ་ས་ཙ་ཎ། སྨད་པ་ལ་བ་ཤའི་རྐྱེན་འདོད་དེ། །ཝ་སྐྱེས་བུ་སྨད་པ། པུ་རུ་ཥ:：ཤཱ་ཤ། ཏ་ར་ཀ་ལ་སོགས་པ་རྣམས་ལས་ཡང་དག་པར་སྐྱེས་པ་ཞེས་པའི་དོན་ལ་འདིར་ཨི་ཏའི་རྐྱེན་འདོད་དེ། སྐར་མ་ཡང་དག་པར་་་་སྐྱེས་པ་དེ་འདིའི། །ཏཱ་ར་ཀི་ཏ་ར་ོན་བྷ། ཡབདེ་བ་ཡང་དག་པར་སྐྱེས་པ་དེ་འདིའི༑སུ་ཁི་ཏ། ་་:ཏ་ར་ཀ་དང་། སུ་ཁ་དང་། དུཁ་དང་། པུ་བྱ་དང༌། བི་ཙཱ་ར་དང༌། ཀུརྨ་ལ་དང༌།  ི་སྟ་བ་ཀ་དང༌། མུ་ཀུ་ལ་དང༌། ཀུ་སུ་:ལ་དང༌། 《ལ་ཀི་ས་ལ་ཡ་དང༌། ནི་དྭ་དང་། ་ཏནྡྲ་དང་། །བུ་་་་དང་། ཀླུ་དང་། །པི་པཱ་ཤཱ་དང་། ༀ་ཏྲ་དང༌། རོ་ག་དང་། །ཨུཀྐཎཱ་ཏི། དང་། ཤངྒཱ་དང༌། །བྱཱ་དྷི་དང༌། བྲ་ཎ་དང༌། གརྦ་དང་། ཀཎྜ་དང་། བེ་ཀ་སྟེ་འདི་རྣམས་ཏཱ་ར་ཀ་ལ་སོགས་པ་རྣམས་སོ། །ཚད་ཀྱི་དོན་ལ། །དགྷ་་་དང་། དྭ་ཡ་ས་དང་། མཱ་ཏྲཊ་ཅེས་པ་འདི་རྣམས་འདོད་དོ། །ཆུ་འདིའི་་་རླའི་ཚད་ནི། ཨུ་རུ་དནྣཾ། :：ཨུ་རུ་དནྣཾ། ཨུ་རུ་དྭ་ཡ་སཾ། ཨཱུ་རུ་མཱ་ཏྲ། །པུ་རུ་ཥ་དང་ཧ་སྟི་དག་ལས་ཨ་ཎའི་རྐྱེནའང་འདོད་དེ། སྐྱེས་བུ་འདིའི་ཚད་ནི། པོ་རུ་ཎཾ། དེ་བཞིན་དུ། ཧཱ་སྟི་ནཾ། འང་གི་སྒྲས། །པུ་རུ་ཥ་དནྣཾ། །ཅཞེས་སོགས་ཀྱང་འགྱུར་རོ། །ཚད་ཀྱི་དོན་ལ། ཡད་ཏད་ཨེ་ཏད་རྣམས་ལས་ཝནྟུ་་་འདོད་:དེ། ཨ་ཡིག་ཏུའང་བསྒྱུར་རོ། །གང་གི་ཚད། ཡཱ་ཝཱན། ཏཱ་ཝཱནི། ཨེ་ཏཱ་ཝཱན། ཨི་དམའི་སྒྲ་ལས་ཨི་ཡན་ཏུ་འདོད་དེ། ཨི་ཡིག་ཏུ་བསྒྱུར་བ་ཡང་ངོ༌། །འདི་ཡོངས་སུ་གཞལ་བ་ནི། ཨི་ཡཱན། —ཨི་ཡིག་ཏུ་བསྒྱུར་བར་བྱས་པ་.།
