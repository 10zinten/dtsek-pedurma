མ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་ཀྱ་་གན་པ་འདི་ལྟ་བུ་ལ་སོགས་པ་རྣམས་སོ། །ཕྱི་མ། བཀའི་ཞེ་འnde། ད་དོ། ཡུ་བན་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ཡུ་བཱ། ཡུ་བཱ་ནོ། ཡུ་བཱ་ན༑ཡུ་བཱ་ནཾ། ཡུ་བཱ་ནོ། གླུཊ་མ་ཡིན་པའི་དབྱངས་དང་པོ་ལ་ཨིཊ་དང་བཅས་པའི་འང་བནྡྷའི་བ་སྒྲའི་ཨུ་ཉིད་དོ་ཞེས་པ་སྤྱིར་བཏང་ནས། །་ཤཱན། ཡུ་བན། མ་གྷ་བན་རྣམས་ཀྱི་འང་ངོ་ཞེས་པས་ཝ་སྒྲའི་ཨུ་ཉིད་དོ། །མཚུངས་པ་ལས་རིང་པོ་དང་། རིང་པོ་ལས་ཕ་རོལ་དབྱིའོ། །ཡཱ་ན། ཡུ་ནཱ། པཡུ་བ་་བྷཱུམ། འོ་ཡུ་བ་བྷི། ་ཡུ་ན། ཡུ་བ་བྷཱུམ། །ཡུ་བ་། ཡུ་ན། ཡུ་བ་་་བྷཱུམ། ཡུ་བ་བྱཱ། ཡུ་ན། ཡུ་ནོ། །ཡཱ་ནཾ། ཡུ་ནི། ཡུ་ནོ། ཡུ་བ་སུ༑ཀུན་དུ་བོད་པ་ལ། ཧེ་ཡུ་བན། ཧེ་ཡུ་བཱ་ནོ། །ཧེ་ཡུ་བཱ་ན། དེ་བཞིན་དུ་ཤཱ། ཤཱ་ནོ། ཤཱ་ན། ཤཱ་ནཾ། ཤཱ་ནོ། གླུཊ་མ་ཡིན་པའི་་དབྱངས་ལ་མཚུངས་པ་ལས་རིང་པོའང་མེད་དོ་ཞེས་པ་ཁྱད་པར་རོ། །ཤུ་ན། ཤུ་ནཱ། ཤཱ་བྷཱུམ། ཤྭ་བྷི། ཞེས་པ་ལ་སོགས་པའོ། །།དེ་བཞིན་དུ། མ་གྷ་བཱ༑མ་གྷ་བྷ་ནོ། མ་གྷ་བཱ་ན། མ་གྷ་བཱ་ནཾ། མ་གྷ་བཱ་ནོ། ཨུ་ཉིད་དུ་བྱས་པ་ལ་ཨུའི་ཡི་གེ་ལ་ཨོ་སྟེ་ཕ་རོལ་ཡང་དབྱིའོ་ཞེས་པ་ཁྱད་པར་རོ། །མ་གྷོ་ན། མ་གྷོ་ནཱ། ་ལ་སོགས་པའོ། །ཕྱོགས་ལ་ནི། ནརྦ་མེད་པའི་ཨ་རྒན་ནི་སི་མ་་ཡིན་པ་ལ་ཨར་བནྟིའོ་ཞེས་པ་སྤྱིར་བཏང་ནས་སི་ལའང་མ་གྷ་ཝན་ནི་མ་གྷ་ཝནྟུ་ཡང་ནའོ་ཞེས་པས་མ་གྷ་བན་གྱི་སྒྲ་ནི་མ་གྷ་བནྟ་རུ་འགྱུར་ཞིང་། །ཨུ་ཡིག་ནི་ཨནྡྷ་དང་ཨས་མཐའི་ཡང་བྱིངས་མ་ཡིན་པའི་སི་ལའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ལ། འདི་ཁོ་ནས་རིང་པོ་ཉིད་དེ་གསལ་བྱེད་ལས་ཀྱང་ངོ་ཞེས་པས་སི་དབྱི་ཞིང་། བརྩེགས་པའི་མཐའ་ཞེས་པས་ཏ་དབྱི་སྟེ། ན་དང་བརྩེགས་པའི་མཐའ་དག་ནི་་ཆོ་ག་སྔ་མ་ལ་མ་ཕྱིས་པ་བཞིན་འང་ངོ་ཞེས་པས། །མ་ཕྱིས་པ་དང་འདྲ་བར་_།
