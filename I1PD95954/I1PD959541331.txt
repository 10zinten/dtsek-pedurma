བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིལས་མེད་པ་ཅན་ཞེས་སྤྱིར་བཏང་ནས། ལས་དང་བཅས་པའི་རུ་ཙ་སོགས་མ་ཡིན་པ་ཉིད་ལས། སཾ་ཤི་ཌོ་ཏི་གུ་རུ་བ་ཙ་ནཾ། ཥཱ། སྲུ། ་དུ། རྀཙྪ། གལླི༑:སྲ་མྲི་འགྲོ་བ་ལའོ། །བྱིངས་ཀྱི་དང་པོའི་ཥའི་སའོ། །"། བྱེད་པ་པོ་ལ་ཨན་བི་ཀ་ར་ཎའོ། །ཨན་ལའང་བི་ཀ་ར་ཎ་ལའོ་ཞེས་པས་ཡོན་ཏན་ཏེ་ཨུ་་ཡིག་གི་ཨ་ཡིག་གོ། །ཨོཾ་ཨ་བའོ། །ས་བ་ཏི། །ས་བ་ཏ། ས་བནྟི། འོས་བ་སི༑ས་བ་ཐ། ས་བ་ཐ། ས་བཱ་མི། ས་བཱ་བ། ས་བཱ་མ། །སྲ་བ་ཏི། དྲ་བ་ཏི། པྲཎྜཱུ་ཏི། སརྦ་ཏི། སཾ་སརྦ་ཏི། ་པ་རསྨི་བཤད་དོ། །་་་བརྟན་པ་ལའོ། །དྷ་བ་ཏི། ལྷག་མ་ལས་བྱེད་པ་པོ་ལ་པ་རསྨིའི་ཚིག་གོ། །ཙུང་། ཚུང་། :ངྷ། ཛུང་། རླུང་། སྲུང་། གཱང་། བྱིང་འགྲོ་བ་ལའོ༑། ང་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་གོ། ལྟ་། ཙྩ་བ་ཏེ། ཚུ་བ་ཏེ༑ཛུ་བ་ཏེ། ཛྷ་བ་ཏེ། ཕྲ་བ་ཏེ། སླ་བ་ཏེ། བི་སླ་བ་ཏེ། རུང་ཁྲོ་བ་ལ་ཡང་ངོ༌། །ར་བ་ཏེ། ཨུང་སྒྲ་ལའོ། །ཨ་བ་ཏེ། བྷཱུ་ཡོད་པ་ལའོ། །ལྷག་མ་ལས་བྱེད་པ་པོ་ལ་པ་རསྨིའི་ཚིག་གོ། །བྷ་བ་ཏི། བྷ་བ་ཏ། བྷ་བནྟི། བྷ་བ་སི། བྷ་བ་ཐ། ་ ིབྷ་བ་མི། བྷ་བ་མ། ཉེ། པྲ་བྷ་བ་ཏི་བ་རི་བྷ་བ་ཏི། བ་ར་བྷ་བ་ཏི། ཨ་ནུ་བྷ་བ་ནི། སཾ་སྔོན་དུའོ། །མ་གསལ་བྱེད་ལ་རྗེས་སུ་ང་རོའི་ཞེས་པས་མའི་རྗེས་སུ་ང་རོའོ། །།སྡེ་པ་ལ་སྡེ་པའི་ལྔ་པར་ཡང་་་ནའོ་ཞེས་པས་རྗེས་སུ་ང་རོའི་དེའི་སྡེ་པའི་ལྔ་པ་སྟེ་ཡང་ན་མ་ཡིན་ནོ། །སཾ་བྷ་པ་ཏི། སཾ་བྷ་བ་ཏི། བི་བྷ་བ་ཏི། ཨུད་སྔོན་དུའོ། །སྔ་མ་བཞིན་དང་པོ་་་དང་གསུམ་པ་དག་གོ། །ཨུད་བྷ་བ་ཏི། ཨ་བྷི་བྷ་བ་ཏི། བི་དང་ཨ་ཏི་སྔོན་་དུའོ༑་། བི་དང་ཨ་ཏིའི་ངེས་མེད་དུ་ཡང་མི་འགྲོ་བ་དང་མི་འཚེ་བའི་སྒྲའི་དོན་་ལ་ཧ་ས་ཞེས་པས། རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྱ་ཏི་:བ་བྷ་།
