རབསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཡང་སྟེ། །སིཙ་ལ་མ་ཡིན་ནོ་ཞེས་པས་ཡང་ན་ཨིཊའོ། །ཙི་ཀརྟི་ཥ་ཏི། ཙི་་ཀྲིཏྠ་ཏི། ཙི་ཙནྟི་ཥ་ཏི། ཙི་ཙིཏྠ་ཏི། ཙི་ཙྪཇི་ཥ་ཏི། ཙིཙྪཏྠ་ཏི། :ཏི་ཏཇི་ཥ་ཏི། ཏི་ཀྲྀཏྠ་ཏི། ནི་ནནྟི་ཥ་ཏི། ནི་ཀྲྀཏྠ་ཏི། གམྨཾ། གམ་ལས་ས་ལ་པ་རསྨི་ལའོ་ཞེས་པས་པ་རསྨིའི་ཚིག་ལས་ན་ལ་ཨིཊའོ། །ཛི་ག་མི་ཥ་ཏི། སཾ་ཛི་ག་མི་ཥ་ཏེ། ཨི་ཎ། ཨིང་། 《་སན་ལ་ཨིཎ་ཨིང་དག་གི་ག་མིའོ་ཞེས་པས་ག་མིར་བསྒྱུར་རོ། །ཇི་ག་མི་ཥ་ཏི། སཾ་ཛི་ག་མི་ཥ་ཏེ། ཧན་དང་ཨིངའི་གམ་དག་གི་ཨུ་པ་དྷཱའིའོ་ཞེས་པས་རིང་པོའོ། །ཨ་དྷི་ཛི་ནཱཾ་ས་ཏེ། ཨིཀ། ཨིཎ་་བཞིན་དུ་ཨིཀའི་འང་རོ་ཞེས་པས་སན་ལ་ཨིཎ་ཨིང་དག་གི་ག་མིའོ་ཞེས་པས་ག་མིར་བསྒྱུར་རོ། །ཨ་དྷི་ཛི་ག་མི་ཥ་ཏི། འདི་རྣམས་ཀྱིས་རྟོགས་པ་ལའོ། །ཨིནའི་ག་མིར་བྱས་པ་དེ་རྣམས་ཀྱི་ཇི་ལྟར་ག་མྱི་རྟོགས་པ་ལ་མངོན་པར་བརྗོད་་་པར་མི་བྱ་བ་དེ་བཞིན་དུ་འདིར་ཨིན་ཞེས་པའི་འང་ངོ༌། །པྲ་ཏཱི་སི་ཥ་ཏི། ཨ་པེ་སི་ཥ་ཏི། ཞེས་པའང་ངོ༌། །།ཨིན་འགྲོ་བ་ལའོ། །ཞེས་པ་འདིའི་དབྱངས་དང་པོ་ལ་གཉིས་པའི་གཉིས་བརྗོད་དོ། །སྟོ་ཏི། ཨིན་མཐའི་བྱིངས་དང་ལྷན་ཅིག་སྤྱོད་པའི་ཕྱིར་རོ། །།སྟོ་ཏི། ཨིན་ོམཐའ་ཉིད་དག་གི་ཥན་ལའོ་ཞེས་པའི་ངེས་པས་བྱིངས་ཀྱི་ས་ཡིག་ཉིད་ཀྱི་ཥ་ཉིད་ལྡོག་གོ། །།གྲ་ཧ། གུ་ཧི། གྲ་ཧ་གུ་ཧ་དག་གི་སན་ལའོ་ཞེས་པས་ཨིཊ་མེད་དོ། །གྲ་ཧི། །སཱ་པི། པྲཙྪ་རྣམས་ཀྱི་སན་ལའོ་ཞེས་པས་སམྤ་སཱ་ར་ཎའོ། །ཧའི་ལྷའོ། །སྔ་མ་བཞིན་དང་པོ་བཞི་པ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཛི་གྷིཀཱ་ཏི། ཛི་གྷིཀྵ་ཏེ། ཛུ་བྷུཀྵ་ཏི། ཛུ་བྷུཀྵ་ཏེ། །ཉི་ཥ་པ། ཨཱ་པི་ཏ་པི་ཞེས་པ་ལ་སོགས་པས་ཨིཊ་མེད་དོ། །པྲཙྪ། སྨིང༌། པུང་ཞེས་པ་ལ་སོགས་པས་ཨིཊའོ། །སྔ་མ་བཞིན་སཾ་པྲ་སཱ་ར་ཎ་བྱས་ཏེ་གཉིས་བརྗོད་དོ། ཁྱི། རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །།
