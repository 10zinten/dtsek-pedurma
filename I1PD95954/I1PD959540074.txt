ཁ་ནས་པར་བྱ་འདོད་ཅིང་། །གང་ཞིག་ཤེས་ན་མཁས་འགྱུར་བ། །འདི་དང་འདི་་་འདྲའི་གཞུང་མི་སློབ། །དེ་རྣམས་བློ་ནི་བེམ་པོའམ་ཅི། །འཕགས་བློ་མ་ཐོབ་བློ་གྲོས་སྐྱོན། །ར། མཉན་པ་མ་གྲུབ་སྐལ་བའི་སྐྱོན། །ཡང་ཡང་མ་བལྟས་བརྩོན་་འགྲུས་སྐྱོན། །དེས་སྐྱོན་ལྡན་ན་མཁས་བཟོད་མཛོད། །འདིར་འབད་དགེ་་་་བས་འགྲོ་བ་ཀུན། །རང་རང་སྐད་ལས་གསུང་རབ་དོན། །བརྗོད་མེད་དེ་ཉིད་རབ་རྟོགས་ནས། །ངག་གི་དབང་ཕྱུག་ཉིད་གྱུར་ཅིག། འགྲོ་བ་:དུ་མ་ལ་ཕན་པར་གྱུར་ཅིག། །བང་དུ་འགྱུར་བ་དང་རྒྱབ་སྐྱེས། །《འི་རྒྱུ་རེད། །ཁྱུང༌:དེ་དག་ལ་གནང་། ས་གནས་། ད། བདེ་བའི་ལས་བསྡུར་མཆན། ཀ་ང་ད་ན་ ེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ལུང་སྟོན་པ་ཀཱ་ལ་པའི་མདོ་ཞེས་བྱ་བ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་ཀུ། «པེ་»«སྣར་»པར་བརྗོད་པའོ། ®«པེ་»《སྣར་ཨི། །«སྣར་»ན། «པེ་»«སྣར་+རྣམས། ་་་་་་་(@«པེ་»«སྣར་ལྔའོ། (ཆོས་ཚན་འདིའི་ནང་དུ་«པེ་《སྣར་ལ་སྔགས་སྲོག་མེད་ཀྱི་རྟགས་ཅན་བསྡུར་མཆན་དུ་པ《བཀོད་མེད།  ི«པེ་»«སྣར་བ། ནི«པེ་»«སྣར་མིའོ། །(«པེ་»«སྣར་»དྷ། (《«པེ་ནིན། «སྣར་ན་ན། &«པེ་»«སྣར་ན། ང་(«པེ་»«སྣར་ལས། «པེ་»«སྣར་»བ། ་«པེ་»། «སྣར་བྷི། «པེ་»«སྣར་»ཤདྷཱའོ། ན«པེ་»«སྣར་»འང་། «པེ་»«སྣར་»ནི་ངེ་། ཚེ«པེ་»«སྣར་ཨི། «པེ་»«སྣར་»ཨལྤ་སོགས་ལ། །ི«པེ་»«སྣར་+ནི། ཉི«པེ་»«སྣར་ཡ། «པེ་»«སྣར་ལ། »«པེ་ཀྱིས་ཨ་མུའང༌། 《སྣར་»ཀྱིས་ཨ་ནུའང༌། 《སྣར་»དག«པེ་»«སྣར་ནོ། «པེ་»«སྣར་»ཤདྷཱ། ཉི«པེ་»«སྣར་»ཡང་། «པེ་»«སྣར་ནི། ཀྱེ«པེ་ཨམའོ། «སྣར་ཨམྲའོ། ་ཀྱི«པེ་»«སྣར་»སྟི་ན་དཱ། ་(«པེ་»«སྣར་»ལ། 《«པེ་»«སྣར་ན་ནིང༌། «པེ་»«སྣར་»སྲཎི་འི། «པེ་»«སྣར་»ལས། 《པེ་»«སྣར་ཝ། མེ་ལྟར་འབད་རྒྱུ་ནི། །ན«པེ་»«སྣར་»ཨུཏྲ། «སྣར་»དྡྷིའོ། ་«སྣར་»བྷུ། ®«པེ་»«སྣར་ལས། «པེ་»«སྣར་འོ། @«པེ་»«སྣར་»ཧཱུཾ། ༄༅«པེ་»«སྣར་རེའོ། «པེ་»«སྣར་»འོ། «པེ་»«སྣར་»སོགས། «པེ་»«སྣར་པའང༌། །«པེ་»«སྣར་པོ། །
