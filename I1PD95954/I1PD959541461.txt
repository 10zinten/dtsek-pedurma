བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤི་::ཊཨཱཏྨ་ནེའི་ཚིག་ལ་སྔ་མ་བཞིན་ཡང་ན་ཨིཊའོ། །།ཨིཊའི་རིང་པོའང་ངོ༌། །པྲ་་་བྲྀཏ༑པྲ་བ་རཱིཤྩ། པྲ་བ་རི། པྲ་ཀྲྀ་ཥཱ་ཏྭཾ། པྲ་བ་རཱི་ཥཱ་ཏྭཾ། པྲ་བ་་རི་ཥཱ'ཏྭཾ། པཱ་ཤྲཱི་ཥ་ཏ། པྲ་བ་རཱི་ཥ་ཏ། པཱ་པ་རི་ཥ་ཏ། སྔོཉ་འགེབས་པ་ལའོ། །།ཨཱ་སཱུ་རཱིཏ། ཨཱསྟནྣཾ། ཨཱསྟརྙ། ་ཨཏྨ་ནེའི་ཚིག་ལ་རཾཏ། བྲིང་བྲིཉ་རྣམས་ཀྱི་སན་ལ། ཨིཊ་ཡང་ན་འགྱུར་ཏེ་སི་ཙ་ཨཱ་ཤཱི་ཥ་དག་ལ་ཨཱཏྨ་ནེ་ལའོ། །བརྩེགས་པ་དང་པོའི་རིཏའི་བརྗོད་པར་བྱ་བ་སུཊ་ནི་ཕྱི་རོལ་དུ་གྱུར་་པར་གྲུབ་པ་མ་ཡིན་ནོ་ཞེས་པས་ཡང་ན་ཨིཊའོ། །ཨཱཙཱིཏ། །ཨཱ་སྟ་རབྦི༄༅། ཨཱ་སྦི་ཥཱ་ཏཱཾ། ཨཱསྟ་རི་ཥ་ཏྭཾ། ཨཱཙཱི་ཥ་ཏ། ཨཱསྟ་རི་ཥ་ཏ། རཾ་སུཾ། འགྲོ་བ་་་ལའོ༑། པུཥ་སོགས་བྱུ་ཏ་སོགས་ལ་ཡིག་རྗེས་སུ་འབྲེལ་པ། ་ཨརྟི། །་སརྟི། ཤཱསྟི་རྣམས་ལས་ཀྱང་པ་རསྨི་ལའོ་ཞེས་པས་ཨཎའོ། །ཨརྟི། སརྟི་དག་:གིས་་ཨན་ཚིལའོ་ཞེས་པས་ཡོན་ཏན་ནོ། །ཨ་ཡིག་གི་ཨཱ་ཡིག་གོ། །ཨཱ་རཏ། ཨཱ་ར་ཊཱ༑ཨཱ་རན། སམ་ལས་ག་མི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །ས་མཱཙྪ། ས་མཱནྟཱ་ཏྭཾ། ས་མཱནྟ་ཏ། །ཁ་ཅིག་ཨཱཏྨ་ནེའི་ཚིག་ལ་ཡང་ཨཎ་སྨྲའོ། །སམཱ་ར་ཏ། ས་མཱ་རེ་ཏྭཾ། ས་མཱ་རནྟེ། ཨ་ས་ར་ཏ། ་ཨ་ས་ར་ཧཱུཾ། ཨ་ས་རན། ཨ་ས་ར་ཏ། །ཨ་ས་རེ་ཧཱུྃ། ཨ་ས་རནྟ། ཏཱཾ། སི་ཙའོ། །ཨིཊ་ཨཱ་ག་མའོ། །འཕེལ་བ། ཨ་ཡིག་གི་ཨཱ་ཡིག་གོ། །ཨ་ཏཱ་རཱི་ཏི། ཨ་ཏཱ་རིཏྟཾ། ཨ་ཏ་རི་ཥཱ༑ཨ་ཏཱ་རཱི། ཨ་ཏཱ་རིé། ཨ་ཏཱ་རིཊྚ། ཨ་ཏཱ་རི་ཎཾ། ཨ་ཏཱ་རི་ཡ། ཨ་ཏཱ་རི། བྷཾ། ་ཨ་པཱ་རཱིཏ། ཀྲྀ་ངེས་པར་གནོན་པ་ལའོ། །དབྱངས་ལ་་་ཡང་ནའོ་ཞེས་པས་ཡང་ན་རེ་པའི་ལ་ཉིད་དོ། །ཨ་གཱ་རཱིཏ། །ཨ་གཱ་ལཱིཏ། ཨ་བ་ལས་གི་ར་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །རྀཏ། །
