བ་ལའོ། །ཤཱནྡྲ་གྲནྠ་སྤེལ་བ་ཞེས་པས་མཆོད་པ་ལ་སོགས་པ་རྣམས་ལ་རྣམ་པར་བརྟག་པ་ཡང་ནའོ། །ད་ནྡྷ། པ་:ཕྲ་ལ། ཕལ་བྷ་ཛ་ཏྲ་པ་ཤཱནྡྲི་གནྡྲི་་་དནྟི་རྣམས་ཀྱིའང་ངོ༌། །ཟླ་། ཞེས་པས་པ་རོགཱ་ཡོན་ཏན་མེད་པ་དང་ཐལ་ལའང་་་་ཨིཊ་དང་བཅས་པ་ལ་ཨཏ་ཨེ་ཉིད་དང་ཨ་བྷྱཱ་ས་དབྱི་བ་ཡང་ངོ༌། ་། ཕེ་ལ་ཏུ། ཕེ་ལུ། ཕེ་ལི་ཐ། ་བ་བྷཱ་ཛ། བྷེ་ཛ་ཏུ། བྷེ་ཛུ། རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ། །ཞེས་པ་ལ་སོགས་པས་ཐལ་ལ་ཡང་ན་ཨིཊའོ། །བྷེ་ཛི་ཐ། བ་བྷཀྵ། ལ་ལ་སོགས་པའི་ངེས་པ་ལས་ཨིཊའོ། །བྷེ་ཛི་བ། བྷེ་ཛེ། ཎཏྲེ་པེ། ཏྲེ་པཱ་ཏེ། ཏྲེ་པི་རེ། གསལ་བྱེད་དང་པོ་ལ་ཨིཊ་ཨཱ་ག་མའོ། །ཏྲི་པི་ཏེ། ཤ་ཐནྡྲ། ཨིནྟི། །ཤམྦི། གནྡྲི། དནྟི་རྣམས་ཀྱི་པ་ོརོགླུ་ལ་ཡོན་ཏན་མེད་པ་ལའོ༑། ཞེས་པས་ཨ་ནུ་ཎཾ༌ག་དབྱིའོ། །ཐེ་ཐ་ཏུ། །ལྟ་ཐེ་ཐུ། བཅས་ཏེ་་་སྤྱོད་པ་ལས་ཨ་ནུ་ཥ་ག་དབྱི་བར་མི་འགྱུར་རོ། །ཤ་ཐངྒི་ཐ། །ཐེ་ཐ་ཐུ། ཐེ་ཐ། ྋཤ་ཤནྡྲ། ?ཐེ་ཐི་མ། ཛ་གྲནྠ། ་གྲེ་ཐ་ཏུ། གྲེ་ཐུཉྩི། ཛ་་གྲནྡྲི་ཐ། ད་དནྟུ། དེ་བྷ་ཏུ། དེ་བྷུ། ད་དམྷི་ཐ། བྷ་མུ། བྱཱ་སཱི། སཱ་ན།  ོ་་ཕ་ཎ་འགྲོ་:བ་ལའོ། །སྱ་མུ་སྒྲ་ལའོ། །བ་བྷཱ་མ། ཛཱ། བྷ་མུ། ཏྲ་སི། སྭ་ན། ཕ་ཎ་སྱམ་རྣམས་ཀྱི་ཡང་ན་བར་བརྗོད་:པར་བྱའོ། །ཞེས་་པས་པ་རོག་ཡོན་ཏན་མེད་པ་ལ་དང་། ཐལ་ལའང་ཨིཊ་དང་བཅས་པ་ལ་ཡང་་ན་ཨཏ་ཨེ་ཉིད་དེ་ཨ་བྷུ་ས་དབྱི་བ་འང་ངོ༌། །བྷེ་མ་ཐེ་ཏུ། །བ་བྷཱ་མ་ཏུ། བྷེ་མུ། བ་བྷཱ་མུ། སྡེ་མི་ཐ། བ་བྷ་མི་ཐ། ཏ་ཏཱ་ས། ཏྲེ་ས་ཏིུ། ཏ་ཏྲ་ས་ཏུ། ཏྲེ་སུ། ཏ་ཏྲ་སུ། ཏྲེ་སི་ཐ། །ཏ་ཏྲ་སི་ཐ། སེ་སཱ་ན། །སྦེ་ན་ཏུ། ？ས་སཱ་ན་ཏུ། སྭེ་ནུ། ས་སཱ་ནུ། སྡེ་ནི་ཐ། ས་སྭ་ནི་ཐ། བི་ལས་ཀྱང་སཱ་ནའི་ཟས་ལ། ་ཞེས་པས་ནང་མའི་འང་ཥ་ཉིད་དུ་ཐོབ་པོ། །བི་།
