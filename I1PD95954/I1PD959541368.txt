བྷསྨི?༑བི་བྷཾ་བ། བི་བྷཾ་མ། བི་བྷཱི་ཏེ། བི་བྷཱ་ཏེ། བི་བྷུ་ཏེ། བི་ཤྲཱི་ཏེ། བི་བྷུ་ཐེ། བི་བྷཾ་དྷེ། བི་བྷེ། བི་བྷཾ་བ་ཧེ། བི་བྷཾ་མ་ཧེ། །:གཉི་ག་ལས་བཤད་པ་རྣམས་སོ། །མཱང་ཚད་ལ་སྟེ་སྒྲ་ལ་ཡང་ངོ༌། །གཉི་ག་་རྣམས་ཀྱི་དྷ་མིན་པའི་གསལ་བྱེད་དང་པོ་ལ་ཨི་ཡིག་གོ་ཞེས་པས་དཱ་མ་ཡིན་པའི་གསལ་བྱེད་དང་པོ་ལ་ཨ་ཡིག་གི་ཨཱི་ཡིག་གོ། །མི་མཱི་ཏེ། མི་མཱ་ཏེ། :མི་མ་ཏེ༑མི་མཱི་ཥ། །མི་མཱ་ཐེ། །མི་མཱི་བྷེ། མི་མེ། མི་མཱི་བ་ཧེ། མི་མཱི་མ་ཧེ༑པྲ་ཎི་མི་མཱི་ཏེ། ཨ་ནུ་མི་མཱི་ཏེ། ཨོ་ཧཱང་འགྲོ་བ་ལའོ། །ཇི་ཧཱི་ཏེ། ཛི་ཧཱ་ཏེ། ཛི་ཧ་ཏེ། ཛི་ཧཱི་ཏེ། ཛི་ཧཱ་ཐེ། ཛི་ཧཱི་བྷེ། ཛི་ཧེ། ཛི་ཧཱི་བ་ཧེ༑ཛི་ཧཱི་མ་ཧེ། ཌ་ཨུཏ་ྋསྔོན་དུའོ། །ཏ་ཡིག་ལ་དང་ཙཊའི་སྡེ་པ་རྣམས་ལ་ཕ་རོལ་གྱི་གཟུགས་སོ་ཞེས་པས་ཏ་ཡིག་གི་ཕ་རོལ་གྱི་གཟུགས་ཛ་ཡིག་གོ། །:：ཨུཛྫི་ཧི་ཏེ། ་ནིར་སྔོན་དུའོ། །རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ་ཞེས་་་པས་ར་ཡིག་གི་རྣམ་པར་གཅད་ཀྱི་བྱའོ། །སྒྲ་ལྡན་དང་དབྱངས་ཕ་རོལ་ལོ་ཞེས་་་པས་རྣམ་པར་:གཅད་བྱའི་རེ་ཕའོ། :：ནིརྫི་ཧཱི་ཏེ། ཛུ་ཧོ་ཏི་ལ་སོགས་པ་ཉི་ཤུ་རྩ་བཞི་པོ་རྣམས་སོ་ཞེས་པ་ཨཏ་ལ་སོགས་པ་བི་ཀ་ར་ཎ་འཇིག་པའི་ཚོགས་སོ། །དེ་བུ་རྩེ་བ་དང་བསྔོ་བར་འདོད་པ་དང་ཐ་སྙད་དང་སྡུ་གུ་དང་བསྟོད་པ་དང་་་མཛེས་པ་དང་འགྲོ་བ་རྣམས་ལའོ། །ཕོ། ལྷག་མ་ལས་བྱེད་པ་པོ་ལ་པ་རསྨིའི་ཚིག་་་གོ་:ཞེས་པས་པ་རསྨིའི་ཚིག་གོ། །།དིབ་སོགས་ལས་ཡནའོ་ཞེས་པས་ཡན་བི་ཀ་ར་ཎའོ། །ཡན་བི་ཀ་ར་ཎའི་དམིགས་བསལ་ལོ། །ན་ཡིག་ནི་ཤམ་སོགས་་རྣམས་ཀྱི་ཡན་ལ་རིང་པོའོ་ཞེས་པས་ཁྱད་པར་གྱི་དོན་ནོ། །བི་ཀེཀ་ར་ཎའི་:：ཨན་ཉིད་ལའོ་ཞེས་པའི་ངེས་པ་ལས་ཡོན་ཏན་མེད་དོ། ལ་། ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །དཱི་བྱ་ཏི། དཱི་བྱ་ཏི། དཱི་བྱནྟི། དྲི་།
