དང་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཤིའིཀ་ཊི་ར་ཎའི་:ནི་ནཱ་མཱིའི་མཐའ་མཚོན་པ་ལས་ཡོན་ཏན་ནོ། །མ་གསལ་བྱེད་ལ་རྗེས་སུ་ང་རོའོ་ཞེས་པས་མའི་རྗེས་སུ་ང་རོའོ། །བི་དཾ་ཀ་རོ་ཏུ། བྱིངས་ཐམས་ཅད་པ་ཡོན་ཏན་མེད་པ་ལ་ཨའི་ཨུ་:ཡིག་གོས། །བི་དཾ་ཀུ་རུ་ཏཱཏ། བི་དཾ་ཀུ་རུ་ཧཱུྃ། ཨུའི་ཡི་གེ་ཝའོ། །བི་དཱ་ཀུརྦནྟུ། ཨུ་ཡིག་ལས་ཀྱང་ངོ་ཞེས་པས་ཨུ་ཡིག་བི་ཀ་ར་ཅ་ལས་ཧི་དབྱིའོ། །ཏཱ་ཏཎའི་ནི་གནས་ཅན་བཞིན་མི་འགྱུར་བ་ལ་ཧི་གསལ་བ་ཉིད་ལས། བི་དཾ་ཀུ་རུ། བི་དྷྭཾ་ཀུ་རུ་ཏཱཏ། བི་དཾ་ཀུ་རུ་ཏཾ། བི་དཾ་ཀུ་རུ་ཏ། ར་ཥ་རིའི་ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་སོགས་པས་ཎ་ཉིད་དོ། །བི་དཱ་ཀ་ར་བཱ་ཎི། བི་དཾ་ཀ་ར་བཱ་བ། བི་དཾ་ཀ་ར་བཱ་མ། 《གང་གི་ཚེ་སྡེ་པ་ལ་དེའི་སྡེ་པའི་ལྔ་པར་ཡང་ནའོ་ཞེས་པས་རྗེས་སུ་ང་རོའི་དེའི་སྡེ་པའི་ལྔ་པ་ང་ཡིན་པ་དེའི་ཚེ་བི་དཾ་ཀ་རོ་ཏུ་ཞེས་པ་ལ་སོགས་པའོ། །སམ་ལས་གམ་མི་ཞེས་པ་ལ་སོགས་པས་ལས་ཅན་་གྱི་ནི་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །སཾ་བིཏྟཾ། སཾ་བི་དཱ་ཏཱ། དེ་ལ་བེཏྟིའི་ཡང་ན་བརྗོད་པར་བྱ་བ་ལ་ཨིར་ཏེ་ཨནྟིའི་ཞེས་པ་ལ་སོགས་པའོ། །སཾ་བི་ད་ྋཏྭཾ། སཾ་བི་ད་ཏཾ?། སཾ་བིཏྟཱ། སཾ་བི་དཱ་ཤྲཱི། སཾ་བི་ཉྙཱནཾ། སཾ་བི་དི༑སཾ་བི་དཱ་བ་ཧི། །སཾ་བི་དཱ་མ་ཧི། ཕྱོགས་གཅིག་ལ་སྔ་མ་བཞིན་ཨཱམ་ལ་སོགས་པའོ། །སཾ་:བི་དཱ་ྋཀུ་རུ་ཧཱུཾ། སཾ་བི་དཾ་ཀུ་རུནྟ། སཾ་བི་དཱ་ཀུརྦ་ཏྭཾ། སཾ་བི་དཾ་ཀུ་རུ་ཥ། སཾ་བི་དཾ་ཀུཊཱ་ཐཱ། ་སཾ་བི་དཾ་ཀུ་རུ་དྷཾ། །སཾ་བི་དཱ་ཀ་ར་བི། སཾ་བི་དཾ་ཀ་ར་བཱ་བ་ཧི། སཾ་བི་དཱ་ཀ་ར་བཱ་མ་ཧི༄༅། །རྗེས་སུ་ང་རོའི་དེའི་:སྡེ་པའི་ལྔ་པ་ལ་ང་ཡིག་གོ། །སཾ་བི་དཱརྐུ་རུ་ཏྭཾ། ཞེས་པ་ལ་སོགས་པའོ། །ཨས་ཡོད་པ་ལའོ། །ཨསྐུ། ཨསྟིའི་དང་པོའིའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ལ་ཨ་ཡིག་དབྱིའོ། །།སྟཏ། །ཏྭཾ། །སནྟུ། དཱ་ཨསྟི་དག་གི་ཨེ་སྟེ་ཨ་བྷུས་དབྱིའོ་་་་ཞེས་པས་ཨེ་ཉིད་དོ། །ཨསྟི་ལས་སོ་ཞེས་པས་ཧིའི་དྷིའོ། »། ཏཱ་ཏཎའི་ནི་གནས་།
