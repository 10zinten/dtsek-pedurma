བཱ་བ་ཤྩ། 《་ས་ཧ། སཱ་ས་ཧཱི་ཏི། ཨུ་:ཧིའི་རྙའོ༄༅། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ། །ཥ་དང་ཊ་སྡེ་ལ་ཏ་སྡེའི་ཊ་སྡེའོ། །ས་ཧི་བ་ཧ་དག་གི་ཨ་ཡིག་གི་ཨོཏའོ་ཞེས་པས་ཨ་ཡིག་གི་ཨོ་ཉིད་དོ། །ཌཱ་ལ་དྷའི་དབྱིའོ། །སཱ་སོ་རྙི༄༅། །སཱ་སོ་ཌྷ། །སཱ་ས་ཧ་ཏི། ་སཱ་སཱ་ཧཱི་ཏི། ཥ་ཌྷིདག་གི་ས་ལ་་་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །སཱ་སz། །སཱ་སོ་ཌྷ༑སཱ་སོ་རྣོ། །སཱ་ས་ཧཱི་མི། སཱ་ས་ཧྥི། སཱ་ས་ཧཱ། ཎསཱ་ས་ཧྥུ། དེ་བཞིན་དུ། བ་ཧ། བཱ་བ་ཧཱི་ཏི། བཱ་བོ་རྨི། །བཱ་བོ་དྷཥྛ། བཱ་བ་ཧ་ཏི། །བཱ་བ་ཧཱི་ཏི། བཱ་བ་ཀླུ། བཱ་བོ་ངྷ། བཱ་བོ་ལྷ། །བཱ་བི་ཧྥ་མི། ་བཱ་བ་རྨི། བཱ་བ་ཧཱ། བཱ་བ་ཧྨ། །:：ཐཾ་ཙི་སྲོ་སི་ཞེས་པ་ལ་སོགས་པ་ཡི་གེ་དང་པོ་རྣམས་ཀྱི་སྟེ། ཨ་བྱཱ་སའི་མཐར་ནཱིའོ།  ོ། བརྩིབ་ནི་བརྩི་ཏི། །ཙ་སྡེའི་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཀིའོ། །བ་ནཱི་པཾ་ཀི། ཡོན་ཏན་མེད་པ་ལ་ཨ་ནུ་ཥད་དབྱིའོ། །བ་ནི་བལྟ། བ་ནཱི་བ་ཙི་ཏི། བ་ནཱི་བཉྩི་:ཥ། བ་ཙྪཱནཱི་པཾ་ཀྵ། བ་ནི་:བ་ཀེ། བ་ནི་བཀྐི། བ་ནཱི་བཉྩི་མི། བ་ནཱི་བཉྩི། །ོབ་ནཱི་བ་ཙཱ། བ་ནཱི་བ་ཙྪ། སྲནྡྷཎ་ཌས་ནི་སཾ་སཱི་ཏི། ་ས་ནཱི་སྲཊཱི། ས་ནཱི་སྲཊྚ། ས་ནི་སྐྱེས་ས་ཏི། ས་ནཱི་སཾ་སཱི་ཎི། ས་ཥི་སྲཊྛ། ས་ནི་སུཏྟཱ།  ོས་ནཱ་སིསྲམ་སཱི་མི། ༼སཱ་ནཱི་ཧྲི་སྨི། ས་ནི་སྲ་སཱ། ས་ནི་སྲ་:མ། །དྷརྨུ། །ད་ནཱི་ཧཱུྃ:༔སྟི། ཏི། དཱ་ནཾ་ྋདཾ་ཧཱུཾ། ད་ནཱ་ཙྪཱདྷཱསྟ། ད་ནི་དྷཱ་ས་ཏི། བྷནྡྷ། བ་ནཱི་བྷཾ་གཱི་ཏི། ་དྷུ་ཊ་ལ་ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཥ་ལས་ཊ་སྡེའོ། །བ་ནི་བྷོ་བྷི། བ་ནཱི་བྷཋ། བ་ནི་བྷ་ས་ཏི། །:བ་ནཱ་བྷོ་སྲི་ཏི། ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ། །ེརྒྱུ་མཚན་་་་ལས་ཥ་ཉིད་དོ། །བ་ནི་ཧཱུཾ་ཀྱི། བ་ནི་བྷཥྛ། བ་ནཱི་བྷཀྵ། བ་ནཱི་བྷཾ་སཱི་:མཱི། བ་ནཱ་ཉྩཱཐཾ་སྨི། བ་ནཱི་བྷ་སཱབ་ནཱི་བྷ་སྨ། ཀ་ས་འགྲོ་བ་ལའོ། །ཙ་ནཱི་ཀ་སྨིGotKTO།
