ང་ལ་བསྟན་འགྱུར། །སྒྲ་མདོ། ༼ཤི། ཁ་དོ༑། པ་ར་མ་ལོ། །པ་ར་མ་ལཿལོ་ང། penཏུ་མི་ལའི་པུ་ནནྡྷའི་སྒྲའི་ཁྱད་པར་ནི། པུ་ནནྟུ། །ཨུ་ཝ་ཐོབ་པ་ལ་ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དེ། དེ་ཡང་བྷུའི་ཝནྟ་བྷུ་དང་པུ་ནནྟཱ་མ་་་ཡིན་པའོ་ཞེས་པས་དགག་པ་ཐོབ་པ་ལ་ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པས་ཝ་ཉིད་དོ། །པུ་ནབྷོ། པུ་ནབྷཱ། པུ་ནབྷོ། ་པུ་ནབྷཾ། པུ་ནནྡྷ། །པུ་ནནྟཱ། པུ་ནནྟབྷཾ༑པུ་ནནྟབྷི། ཨུ་ཝ་གནས་པ་ཉིད་མེད་པའི་ཕྱིར་ཐུང་ངུའི་ཡང་ང་ལྡན་ལའོ་ཞེས་རྣམ་པར་བརྟག་པས་ན་དཱི་དང་འདྲ་བར་འགྱུར་བ་མེད་དེ། དེས་ན་ཨཱིཏ་ཨུཏ་བུད་མེད་བརྗོད་པ་དག་ན་དཱིའོ་ཞེས་པས་ན་དཱིའི་མིང་ཅན་ཡིན་པས་་རྟག་ཏུ་ན་དཱིའི་བྱ་བའོ། །པུ་ནནྡྷེ། པུ་ནནྡྷ་བྷཾ། པུ་ནནྟ་བྷྱ། །པུ་ནནྡྷ། པུ་ནནྟཱ་ཧཱུཾ། པུ་ནནྡྷ་བྷུ། །པུ་ན་ནྡྷེ། པུ་ན་བྷཾ། པུ་ནདྡྷཾ། པུ་ན་བྷཱུ་ཎྞཾ༑པུ་ནདྡྷཾ། པུ་ནདྡྷོ། ཥ་ཉིད་དོ། །པུ་ནནྟཱ་ཏུ། །ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལ་ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུ་སྟེ། ཐུང་ངུ་ན་དཱི་ཞེས་པ་ལ་སོགས་པས་སི་དབྱིའོ༑། ཧེ་པུ་ནནྡྷ། ཧེ་པུ་ནབྷོ། ཧེ་པུ་ནབྷོ། །ཧེ་པུ་ནནྡྷ། །ས་ལ་ཚེ། །བརྟ་བྷཱུའི་སྒྲ་ནི་བུད་མེད་ཀྱི་རྟགས་ན་དཱིའི་མིང་ཅན་ཏེ། བྷུའི་ཝནྟཱ་བྷུ་དང་པུ་ནནྟུ་མ་ཡིན་པའོ་ཞེས་པ་འདིར་བརཱ་བྷུ་མ་ཡིན་པ་ཞེས་པས་བཀག་པའི་ཕྱིར། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དོ། ལ། བནྡྷ་བྷུ། བནྡྷ་བྷོ། བ་བྷཱ། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ། པྲ་ཏི་བྷུ་དང་བཱི་ར་བྷུ་ལ་སོགས་པ་རྣམས་སོ། །པོ། །བམ་དེ་ལ་བྱ་དང་འཚོ་བའི། །::དྷེ་ནུའི་སྒྲ་ནི་བྷུ་མིའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱའོ། །དྷེ་ནུ། དྷེ་ནུ། དྷེ་ན་ཝ། དྷེ་ཧཱུྃ། དྷེ་ནཱ། དྷེ་ནུ། དྷེ་ནཱ། དྷེ་ནུ་བྷཾ། དྷེ་ནུ་བྷི། དྷེ་ནཱི། དྷེ་ན་ཝེ། དྷེ་ནུ་བྷཾ། དྷེ་ནུ་བྷྱ། །དྷེ་ནཱ། དྷེ་ནོ། དྷེ་ནུ་ཧཱུཾ། །
