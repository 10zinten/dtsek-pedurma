བསྟན་འགྱུར། སྒྲ་མདོ། ཤིའིm/དེ་བི་ཏཱ། ནྲྀ་ཏཱི། ནནྟི་ཏཱ། ར་དྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོ༌། །ཞེས་པས་ཡང་ན་ཨིཊའོ།  ི། ར་དྷི་ཛ་བྷ་དག་གི་དབྱངས་ལའོ། །ཞེས་པའི་ན་ཡིག་ཨཱ་ག་མ་མེད་དེ། །་ར་དྷིའི་ཨིཊ་ལ་པ་རོ་གྷ་མིན་པ་ལ་མ་ཡིན་ནོ། ་་། ཞེས་བཀག་པ་ལས་སོ། །ར་དྷི་ཏཱ། ཏ་ཡིག་གི་དྷ་ཉིད་དོ། །གསུམ་པ་ཉིད་འང་ངོ༌། །རདྷཱ། །ཙཱ་པ། ཏརྦ་ཏཱ། །སྔ་མ་བཞིན་དབྱངས་ལས་ཕ་རོལ་ཏུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །ཏྲཔཱ༑ཏཏྟཱ། མུ་ཧ། དྲུ་ཧ། སྣུ་ཧ། །སྣེ་ཧ། །<མོ་ཧི་ཏཱ། དྲུ་ཧ། མུ་ཧ། སྐུ་ཧ། སྣེ་ཧ་རྣམས་ཀྱི་རྣམ་པར་བཤད་པ་བལྟ་བར་བྱ་སྟེ་ཧའི་གྷ་ཉིད་དོ༑འི། མོངྷ། མོནྟཱ། དྲོ་ཧི་ཏཱ། དྲོབྦྷ། །དྲོ་ནཱ། སྒོ་ཧི་ཏཱ། སྣོབྷཱ། སྣ་ནྟཱ། སྣེ་ཧི་ཏཱ། སྣེབྷཱ། སྣེ་ཝཱ། ཎ་ཤ། ན་ཤི་ཏཱ། མཉྫི་ན་ཤ་དག་གི་དྷུཊ་ལའོ་ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་ཡིག་ཨཱ་ག་མའོ། །མ་ན་དག་གི་དྷཊ་ལ་རྗེས་སུ་ང་རོའོ། །ཞེས་པས་ནའི་རྗེས་སུ་ང་རོའོ། །ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ནཾཏྟཱ། པྲ་ནཾཏྟཱ། རྟ་དེ་ལ་ཡང་སྨོས་པ་ལས་ལྷག་མ་ཥ་ཡིག་དང༌། །ཀ་ཡིག་མཐའི་ཞེས་བཀག་པ་ལས་ཉེར་བསྒྱུར་ལས་བསྡུ་བ་མེད་པ་ཉེ་བར་བསྟན་པའི་ཞེས་པས་ཎ་ཉིད་མ་ཡིན་ནོ། །ར་དྷ་ལ་སོགས་པ་བརྒྱད་རྣམས་པ་རསྨིའི་ཚིག་གོ། །དཱི་པཱི། དཱི་པི་ཏཱ། ་ཀླུ་ཤ་། ཀླུ་ཤི་ཏཱ། ཨཱ་ཤུ། ་ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ལ་ཡང་ན་ཨིཊའོ། །ཨཏྟཱ། ཨ་ཤི་ཏཱ། ཨཱཏྨ་ནེའི་ཚིག་རྣམས་སོ༑། བྲཤྩ། བཉྩི། མཎྜི་དག་གི་དྷཊ་ལའོ། །ཞེས་པས་མཐའ་དབྱིའོ། །བྷཾ་ཛ་སོགས་རྣམས་ཀྱི་ཥའོ། །བྲ། བྲཝཱི་ཏཱ། བྲི་ཧུ་བརྩོན་པ་ལའོ།  ི། བནྡྷ། བརྙི་ཏཱ། བྱ་ཙ། ཨིཊ་ཨཱ་ག་མའོ། །ཀུཊ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨིཊའོ། །།།དེའི་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །གྲ་ཧི་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་ས་ར་ཎའོ། །པི་ཙི་ཏཱ་པ་རསྨིའི་བཤད་དོ། །གུ་རཱི། གུ་རི་ཏཱ། ་ཨོ་བི་ཛཱི། བི་ཛིའི་ཨིཊའོ། །།
