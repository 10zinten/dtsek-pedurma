བས་བསྟན་འགྱུར། སྒྲ་མདོ། ཤི་་་་«ཅོ་རཱ། །བུད་མེད་རྣམ་དཀར་གྱི་ཉ། །ཁ་«ཅོ་»ཀཱི་པ། «ཅོ་»ཐོབ། བ་ཡུང་ལ་སོགས་པ་ནམས་ལ་ནི་བྱེ་ན་བ་ཕྲག་«ཅོ་ཐོབ། ཉི«ཅོ་»བྷཱུ། ལ་ཡོང་ཁྲོའོ་ཞེས་པ་པོ་ཀ་བདུན་«ཅོ་»ནབྷཾ། ::ཞེས་འབྲས་བཞིན་དུ་ཚིག་སྡུད་དེ། ཨེ་ཤ་«ཅོ་ནཱ། «ཅོ་»ཛི། ཚུ«ཅོ་པྲ་བྷི། ཡི་ག་འད། ཁྲོ་ཁོཔའི་ང་ཚོ་ལ་::་«ཅོ་»ཨདྷཱ། ཨི«ཅོ་»བྷའི། །«ཙོ་»གླུ་ཊ། «ཅོ་»ཀུ་བྷོ། «ཅོ་»དྲི། «ཅོ་»ཨང་། དལ་འབྱོར། །ཆཞུ་རྒྱུ་ཡི་ཕྱི་ལོ»ཞེalg«ཅོ་»ཤི་ཧྥུ། «ཅོ་»བྷེ། དུ་གསའ་«ཅོ་»ཏྲ། «ཅོ་»དེ། ག་་་པ། ན་ཚེ«ཅོ་»ལ། «ཅོ་»འཐུང་། ཚེ«ཅོ་»ཨའི། །སྐྱེ་བ་དེ་དང་པོ་བྱེ་བ་«ཅོ་»ཡཱ། :::་ལོ་། ཚེ«ཅོ་»ནི། «ཅོ་»ག། «ཅོ་»ཨ། «ཅོ་»ག། [E<<་ཚེ»J《་༈«ཅོ་ལྕེ། «ཅོ་»པཱུ། «ཅོ་»ཎེ། «ཅོ་»ཀཎྜིན། «ཅོ་»ཏྲིའི། འ། །པཚོང་དང་ལོག་འོང་བ་ག་སྟོན་«ཅོ་»དོ། («ཅོ་»བྷི། །།པོད། །དེ་སྐྱེ་ཚེ། ཚེ་དོ། «ཅོ་»ར་ྩ། «ཅོ་»རཱ་ཛཱ། «ཅོ་»བྲྀཛྫ་བློ་ན་་Sརྒྱུ་དང་། དངོ«ཅོ་ཧཱུཾ། ན། བདུན་པ། གང་དུ་བབ༌པ་ཚུ་གིས་་་་པ། «ཅོ་ཙཱ། Q«ཅོ་»པཱཏྱཱཉྩི། །བས་《། 《ཚོ»ཀྱི་ཚེ་ཚེ་ངོད་ཚེ«ཅོ་ཟན། དང་སྒྲ་རིག་པའི་དགེ་སྐྱོ་བོ་ཚ་བ་དང་(«ཅོ་»མུཉྩ། །པོ་འདི་ཁོ་ན། །བྱང་ཕྱོགས་ལ«ཅོ་»ད་མཱི། ཉི«ཅོ་»ཙཱི། ཉི«ཅོ་»མུམྨི། «ཅོ་»སpཀ[pཚེ»༽སྟོང་(«ཅོ་»ཀུཉྩ། «ཅོ་»+ཡ། །«ཅོ་»ཡུཉྩད། «ཅོ་»ཡུཉྩནྟི། «ཅོ་ཝནྟོ། ཚེ«ཅོ་»ཀི། །(gg:ཀྱིངོ(«ཅོ་»ཀཾ། ®«ཅོ་»ཏི། ནས་ལེགས་པར་བསྒྲུབ་(ི«ཅོ་»པནྟི། ®«ཅོ་བྷ། ®«ཅོ་»བྷཱསའོ། :::»Unཨེ། «ཅོ་»ཏོ། ཚེ«ཅོ་»ཀཾ། དེ་ཚོ་ཁྱིཚ་—།
