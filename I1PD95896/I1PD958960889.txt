ཚོ་ཁ་བ་ཌི་ལོ། «པེ་»ཀྱི། «པེ་»«སྣར་»ཀྱང་། ཉི«པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་»སྨོས། «པེ་»ཀུ་《སྣར་»—ཡང་། ཀློག་པ་དང་། ཕྱིན་རོགི་མ་ཎg《'(། ཁརྒྱུ@?ཀྱི«པེ་»«སྣར་བསྲུང་། «པེ་»«སྣར་བསྒོམ། «པེ་»«སྣར་པར། ཀ་ཅིp«པེ་»«སྣར་»+དུ། ཉི«པེ་»«སྣར་»ནི། །མ《་། ཏཅ}《་《ཉི(«སྣར་»ལྟ། ཉི«པེ་»«སྣར་»དོ། །འe«པེ་»«སྣར་ཞེས། །ལྟ་་(:ཁཚོ་པ་མན་«པེ་»«སྣར་དེ། ་་་་། ཀཚོ་ཅ་དང་། ཁྱg%:ཀྱི་འཚོ་བའི་ཚེ་ཟླ་བ་«པེ་»«སྣར་»+དོ། །ས། ས། ་ཁ། ༼ཡ། ་པ། ཀྱི། ཀྲྀ་ད་ཀར་པ་《་༽༈། «པེ་»སྣར་»ཚིག«པེ་»«སྣར་དེས། །དེ་ནས་། ཕྱི། ཁ་《ཁོ«པེ་»《སྣར་»—ནི། «པེ་»«སྣར་»—དེ། ང་ཚོའི་བྱེད་ན་ང་ཚོ་ཁ་ཕྱིར་ཕྱིན་«པེ་»«སྣར་»—དོ། «པེ་»«སྣར་»—སྤྱོད་པ་དེ་དག་མཚུར་ནང་རྒྱུ་དེ་མ་ཡང་«པེ་《སྣར་ལྟུང་། «པེ་»«སྣར་»འཕྲོགས། «པེ་»«སྣར་»འཕྲོགས་པར། «སྣར་»—སྟོང་པ་ཉིད། ནང་པ་ཡིན། སྐྱེ་བྱེད་བ་aཏུ་ན། ལན། བྱེད་པ། «པེ་»«སྣར་+ཐམས་ཅད། ཁྱི«པེ་»«སྣར་—རྣམས། ཁ་སང་། <<(འི། «པེ་»«སྣར་»གི། །ཉི«པེ་»«སྣར་»ཀྱི་སྙད། །@«པེ་»«སྣར་»རྟོགས། །་«པེ་»«སྣར་+ཀྱང་། ཚེ«པེ་»«སྣར་»རང་། །ཡེ་ཏེ། དེ་། ད་བར《ཡེ་་པེ། «པེ་»«སྣར་»ཡང་། ཏ། ཁ་ན་ཊ་ར་《(ཁོ་། ཁ:《བ《འི»ན«སྣར་པ། དེ་དང་མཀྱི་དབྱེ་ཁག་གི་མིང་། །བཀའ་བརྒྱུད་ཀྱི་«པེ་»«སྣར་»པའི། །།@«པེ་»«སྣར་»སེམས་ཅན། ན«པེ་«སྣར་སྐྱབས། （«པེ་»«སྣར་བསྲུང་སྟེ། ཚུ«པེ་»«སྣར་»རོ། ། ོ«པེ་»«སྣར་+རོ། ད་ཀྱང་ལམ་«པེ་»«སྣར་»འཁྲུག@«སྣར་»ཆལ། «པེ་»«སྣར་»+ཡང་། &«པེ་»«སྣར་+ཕྱིར། «པེ་»«སྣར་»འཁྲུག«པེ་»«སྣར་ངའི་ང་རྒྱལ་ལམ། «པེ་»«སྣར་+ཆོས་ཀྱི། «པེ་»:::«སྣར་+པ། །པ་དང་་ང་དོན། ཀ་པ་དང་། པར《»། @«པེ་»«སྣར་བསྟན། །@«པེ་»«སྣར་»སྙོམས། «པེ་»«སྣར་»དོ། «སྣར་པ། ང་«པེ་»«སྣར་བསྲུང་། ཉི«པེ་»སྣར་འགྲོའམ། །ཌ་ཀ་ལས་ད་ཀྱི་ཀྱི་«པེ་»«སྣར་བར། ཚེ«པེ་»«སྣར་སྒོམ། ་་་ཡེ་་་་«པེ་»«སྣར་+ཉིད། «པེ་»«སྣར་+དེ། «པེ་»«སྣར་»སྒོམ། «སྣར་»ཀྱིས། 《་ན་གྱི«པེ་»«སྣར་ལ། :«པེ་»«སྣར་དེ། །ཚེ«པེ་»«སྣར་—རྣམས། «པེ་»«སྣར་དེས། «པེ་»«སྣར་»+བཞི། ༈།
