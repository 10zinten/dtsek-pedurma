།སོ་སོར་བརྟགས་པ་མ་ཡིན་པས་འགོག་པ་ཞེས་བྱ་སྟེ། ཇི་སྐད་དུ། སོ་སོར་བརྟག་པས་འགོག་པ་གང་། །བྲལ་::ཀ་བའི་སོ་སོ་སོ་སོ་ཡིན། །སྐྱེ་ལ་གཏན་དུ་གེགས་བྱེད་པ། །འགོག་གཞན་སོ་སོ་བརྟགས་མིན་པས། །ཞེས་བཤད་པ་བཞིན་ནོ། །འཇིག་པའི་བདག་ཉིད་ཡིན་དུ་ཟད་མོད། །དེ་ལྟ་ན་ཡང་འཁྲུལ་པ་མ་ཡིན་ནོ། །ཞེས་བསྟན་་་་་པའི་ཕྱིར། རང་བཞིན་མེད་པའི་ཕྱིར་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། བཙལ་མ་ཐག་ཏུ་བྱུང་བ་མིན་ཞེས་བྱ་བ་ཀཀྱིས་དང་སྦྱར་རོ། །གང་ལས་ཞེ་ན། རང་བཞིན་མེད་པའི་ཕྱིར་ཏེ། ཇི་ལྟར་དངོས་པོ་ཡོད་པ་བཀག་པ་ཙམ་གྱི་་་་་ལྔ་པ་འཇིག་པ་ཡིན་ལ། ནམ་མཁའ་ཡང་རེག་པར་བྱ་བ་མེད་པ་ཙམ་ཡིན་ཏེ། དེ་ལྟ་ཡིན་དང་འཇིག་པ་འདི་གཉིས་་་ཀྱིས་ཀྱང་རང་བཞིན་མེད་པ་ཡིན་ནོ། །དེ་ཇི་ལྟར་འདི་དག་བཙལ་མ་ཐག་ཏུ་འབྱུང་བར་འགྱུར་ཏེ། «པེ་»«སྣར་»+ཆུ། «པེ་»«སྣར་»ནས། མུ«པེ་»《སྣར་»ཀྱིས། པོ«སྣར་»སྨ། ཚེ«པེ་»དང་«སྣར་»འགྱུར། ཉི«པེ་»《སྣར་»—གྱི། @«པེ་»《སྣར་བ། (ཚེ«པེ་»«སྣར་ཡི། འགྱེ«པེ་»《སྣར་དྲགསྐྱེ«པེ་»ཤིང་། ཅེ«སྣར་སྙོམ། ལོ«པེ་»«སྣར་»གོ། ཁྱེ«པེ་»《སྣར་བ། ༄༅«པེ་»:《སྣར་»+འདི། «པེ་»«སྣར་»དྲག་འཚེ«པེ་»«སྣར་—ནི། །(«པེ་»«སྣར་»མཐུར། @«སྣར་+པ། «པེ་»«སྣར་རྩི་དེའི། «པེ་《སྣར་+སྣུམ་གྱི། ::ཀྱི«པེ་»《སྣར་ཐམས་ཅད། «པེ་»«སྣར་»—པ། ཚེ«པེ་»«སྣར་ལས་གཡོས་སུ། །«སྣར་རབ་གས་སྐོལ།  ི«པེ་»«སྣར་»ཚྭར། @«པེ་»«སྣར་»དྲོ། «པེ་»«སྣར་»ནད། ནི། «པེ་»«སྣར་བཏང་བ། «པེ་»«སྣར་»ཚྭ། ༄༅«པེ་»«སྣར་»+རོ། «པེ་»«སྣར་བཏང་བ་ནི། :བ«པེ་»སྦྱར། 《སྣར་སྦྱོར། ཀྱེ«པེ་»«སྣར་བའི། ་འ«པེ་»«སྣར་རྩི། («པེ་»«སྣར་»དབང་རྣམས། ཁྱེ«སྣར་བ། ཅེ«པེ་»«སྣར་སྐཁ། ཨེ«པེ་»«སྣར་ཉུང་། ཚེ«པེ་»«སྣར་»—པ། ༄༅«པེ་»ད་དུ་《སྣར་»+དྲོ་བ་དང་། «པེ་»«སྣར་པ། «པེ་»«སྣར་སྤང་། །«སྣར་གསུང༌། «པེ་»«སྣར་»ཀྱི། @«པེ་»«སྣར་ལ། «པེ་»«སྣར་ཤེས། «པེ་»། ས་《སྣར་»བ། @«སྣར་གསན། @«པེ་»«སྣར་»དུ་ཡིན་པ་ལས། ཉི«པེ་»《སྣར་»+ཕྱིར། «པེ་»«པེ་»«སྣར་+སློབ་དཔོན་འདི་དག་གིས་བཤད་པ་མས་བཏང་བ་གཞན་དང་གཞན་དུ་བཤད་པའི་ཚུལ་གང་ན་ཡིན་པ་དེ་ཐམས་ཅད་ཚད་མས་གཞལ་བ་ཡིན་གྱི། །«པེ་»«སྣར་མ། «པེ་»《སྣར་+ནི། །།«སྣར་དཔག་མ། «པེ་»«སྣར་»དཔག་པས། ་ོ«པེ་»«སྣར་མ། «པེ་»«སྣར་ཡང་། «པེ་»«སྣར་གྱིས་ལ་ལ་ཙམ་གྱིས། «པེ་«སྣར་དཔྱིད། «པེ་»མ་ལེགས་པ། «སྣར་ལེགས་ཀ་པ། ་(ཀྱེ«པེ་»《སྣར་»མ། «པེ་»མས་བཏང་། 《སྣར་མ་བཏང་། རྩེ«པེ་»«སྣར་བཞག::ཀྱི། —།
