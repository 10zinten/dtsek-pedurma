སྦྲ་ན་སཱ་མི་ན། དེ་བ་ནཱ་གཡཀ་:དི་གནྡྷརྦ་བྷཱུ་ཏ། བྷཱུ་ཏ་ནི། ཌཱ་ཀི་ནི། །:སཏཱ་ཨནྡྲ་ཀྲྀ་ཡ་ཨ་པ་རཾ་བ་ག་ཚ་ཏྲ་སྲི་གྲོ་མེ་ཝནྟུ། ཨོ་ཧཱུྃ་ཕཊ། བསྐྲད་པའི་སྔགས་དང་ཕྱག་རྒྱའོ། །རྡོ་རྗེ་ས་གཞི་ར་བ་དང་། །དྲ་བ་དག་དང་བླ་རེ་དང་། །མདའ་ཡི་དྲ་བ་ཕྱི་རོལ་དུ། །མེ་རི་ཡང་ནི་བྱིན་གྱིས་བརླབས། །ཨོ་མེ་དཱི་ནཱི་བཛི་བྷ་བ་བཛྲ་བནྡྷ་ཧཱུྃ། ཨོཾ་བཛྲ་པྲ་ཀ་ར་ཧཱུྃ་ཁཾ་ཧཱུྃ། ཨོཾ་བཛྲ་པཉྩ་ར་ཧཱུཾ་ཡཾ་ཧཱུཾ། ཨོཾ་བཛྲ་བི་དྷ་ན་ཧཱུྃ་ཁཾ་ཧཱུྃ། ::ཨོཾ་བཛྲ་ས་ར་ཛ་ལ་ཏྲ་®སཾ་ཏྲཾ། ཨོཾ་བཛྲ་ཛྭ་ལ་ཨ་ན་་་ལཱརྐ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཨོཾ་སུཾ་བྷ་ནི་སུཾ་བྷ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་:གྲིཉྩ་གྲིཉྩ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་་:གྲོནྟཱ་པ་ཡ་གྲོནྣཱ་པ་ཡ་ཧཱུྃ་ཧཱུྃ་ཕཊ། །ཨོ་ཨཱ་ན་ཡ་ཧོ་བྷ་ག་ཝཱན་བཛྲ་རཱ་ཛཱ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་གྷ་གྷ་གྷ་ཏ་ཡ་གྷཱ་ཏི་ཡ། ་སརྦ་དུཏྟཱ་བིགྷཱ་ཧཱུྃ་ཕཊ། ཨོ་ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ་སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨ་ཙཱ་པ་ཡ་ཏི། སརྦ་བིདྡྷཾ་ཀཱ་ཡ་བཱཀྲིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཧཱུྃ་མུངྒ་ར་བཛྲ་ཀཱ་ལ་ཡ། ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཟ། སྔགས་ཚིག་གྲུབ་པ་འདི་རྣམས་ཀྱིས། །།འབར་བའི་མེ་ཡི་ཕྱི་རོལ་དུ། །ཁྲོ་བོའི་རྣམ་པའི་ཕུར་བུ་ནི། །ལུགས་སུ་འབྱུང་དང་ལུགས་མིན་པས། །བགེགས་རྣམས་ཀྱི་ནི་སྤྱི་བོར་གདབ། །ཕུར་པ་ཁྱིལ་སོགས་ཆོ་གའོ། །རྡོ་རྗེ་དྲིལ་བུ་ལ་སོགས་ཀྱི། །།དེ་ཉིད་རྣམས་དང་མཆོད་ཡོན་སོགས། །།ཆོ་ག་རྒྱས་པའི་དམ་ཚིག་ནི། །སྒོ་སྐྱོང་ལས་ཀྱི་རྡོ་རྗེས་གདབ། །སོར་མོ་བརྒྱད་པ་ནས་་་་བརྩམས་ཏེ། །ཉི་ཤུའི་བར་དུ་གྲངས་མཉམ་བྱ། །།གསེར་དང་འཇོན་དམར་་་སོགས་བྱུང་བའི། །རྡོ་རྗེ་རྣམ་པ་གསུམ་ནི་བཟུང་། །རྩེ་གསུམ་པ་དང་རྩེ་ལྔ་པ། ཌོ་། རྩེ་དགུ་པ་སྟེ་རྣམ་པ་གསུམ། ་། ཞི་དང་མ་ཞི་རབ་ཕྱེད་པས། །ཡང་ནི་རྣམ་པ་གཉིས་སུ་བརྗོད། །བཞི་པ་སྣ་ཚོགས་རྡོ་རྗེ་ཉིད། །རྩེ་མོ་ལྔ་པ་གདོང་་་བཞི་ཡིན། །དབུས་མ་གཟུང་བའི་གནས་ཡིན་ཏེ། །།མཐའ་མའི་རྩེ་མོ་དེ་ཡི་ཚད། །།
