།།ཆུ་ཁའུ་ཁོབ་ཎ་པའི་རྒྱུ། །ཀཎཁ་བཏུལ་བ། ཁ་ཎ་ཁ་ལོ་ཏpཁ་aཔ་ཏeཏི་ཏ་བློ་ལྔ། །མ་ཎgཁ་ཆཎ་ཀྱཁ་nཥ་བའི། ཀླུ་ང་ཀ་ཌཚོ་ཁ་ལ་སོགས་པ་ཞུ་ཁའཁྲུན་ཆད། །ཁཏྟ་སྤཁ་ཊིཌ་ཌཱ་ཀཚོར་ཉི་ཟླའ་ཁཊ་ཀྲྀ་ཟླ་ཚོ་ཀྲུང་། །ཨོ་ཁཟླ་་ཥ། ཟླ་བར་གྱི་ཀ་ལྔ་བ་དང་དུས་ད་ཚོ་། །ོ་ཏུ་ང་ཚོ་ར་ཀ་དུ་བྱ་སྤྱི་། །ཊ་ཁང་༄༅། །དཔའ་གཅིག་མའི་སྒྲུབ་ཐབས་བཞུགས། ཀ། གཔོ་དཔ། །ཅ/zཀ་བ་བལ་འཚུ་། ་་་་་་༌། དང་འཚོ་ਸੌਸ਼ਲੋਨਨੌਨਸ਼ੋਸ'Bgngਸਰਕਥਾਣਾਸmਸਮਵਸ(དེབས། །ཁཀྱང༌། ་་་ཆཎྜ་། །ཡ་ཆུང་བདག་༄༅། །དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་དབྱིངས་ཡེ་ཤེས་ལྷན་ཅིག་སྐྱེས། །ཞལ་གཅིག་ཕྱག་གཉིས་གཡོན་བརྐྱང་སྔོ། །རྒྱན་ལྔ་ཐོད་ཕྲེང་མགོ་སྐྱེས་སེར། །གྲི་གུག་ཐོད་ཕྱག་ཁྱོད་ལ་འདུད། །བློ་དམན་བུད་མེད་་རྣམས་ཀྱི་ཕྱིར། །མདོར་བསྡུས་རྒྱས་པ་མ་ཡིན་པ། །།བསྐྱེད་པ་དང་ནི་རྫོགས་་པ་ཡིས། །དཔའ་མོ་གཅིག་པ་བྲི་བར་བྱ། །དང་པོ་རེ་ཞིག་རྣལ་འབྱོར་པས། དབེན་པའི་གནས་སུ་སེམས་དཔའི་སྐྱིལ་མོ་ཀྲུང་གིས་གནས་ལ། ཇི་སྐད་་་་་་གསུངས་པའི་རིམ་པས་བླ་མ་མཆོད་ནས་ཚངས་པའི་སྒོམ་པ་བྱས་ལ། དེ་ནས་ཨོ་ཤུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། དེ་རྗེས་སྨོན་ལམ་སྟོབས་ལས་ནི། །ཆུ་སྐྱེས་དམར་པོ་ཉི་དབུས་སུ། །མ་སྐྱེས་མཆོག་གི་ས་བོན་ལས། །ནག་པོ་ཞལ་་་་གཅིག་ཕྱག་གཉིས་མ། །གཡས་པར་གྲི་གུག་འཇིགས་ཆེན་མ། །གཡོན་པ་ན་ནི་པདྨའི་སྣོད། །གཡས་པ་མིན་པར་ཁ་ཊཾ་ག། སྟག་གི་པགས་པས་སྨད་དཀྲིས་པ། །།
