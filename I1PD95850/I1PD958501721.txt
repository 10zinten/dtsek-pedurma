བསྟན་འགྱུར། རྒྱུད། ཏའིཚ་ཏེ་n@་ཆེར་གན་སོགས་བཀའ་ཁྲི་རྒྱལ་བའི་བྱ་ཆེན་དག་པ་རྒྱ་གར་། །འཕྱིz་ལ་ཡོད་ཀྱི་ཁ་ཡི་ཁྲུབ་པ་གཙོ་ཚོ་བ་ཎི་ཞུ་གཏུཊཏཊིས་ལ་དབང་ཆེ། གཟུགས་སུ་བsའི་བསྡུར་མཆན། ཁོཋབ་བཀྱིན་པ་ལྟ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། ཚེ«པེ་»«སྣར་དམར། «པེ་»«སྣར་»བརྒྱན། «པེ་»«སྣར་བྱ། །ཨུ་བབ། ཕྱི་ནས། །ཎ་ཊའི་ཚོང་པའི་བར་དུ། །གལ་པ་དཔལུ་ཁ། ཚpp@edelཎྜི་ཁྲོད་དགའ། །ཀ་ཁ་ག་ང་པ་ཚོ་ཁག་མ་པ་ཎ་ཁེཊ་ཀ་ཚོནོ་བསྡུར་འབྲས་རེའུ་མིག་ཆམ་ཏེ་ལོ་འགག་ཆg་། ་དོརྐཏེངསhཛོ་བརྒྱུད། （འanང་མ་རིག་པ་བའི་ཚོར་དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ༽ཅོ་ནེ། སྡེ་ཚན། རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། པོད་རྟགས། །ཚན་གྲངས། །།ལྡེབ། ན་བ་། ནབརབབ?ནབ་ཕྲེང་། །:ན། །ལ། ཚིག་རྐང་། །ཚེག་ཁྱིམ། །།།་། །བདངད་པ་། དེ་བའི་ནད་ལ་གཞུག་པར་བྱ། །དེ་ནས་བྲལ་གླིང་པས་བ་བ་རང་གི་ཁ་དོག་ན། ཕྱི་དང་དེ་བཞིན་དང་བདག་་བ་སློམ་པར་བྱ། །པོད་དུ་ལུ་སྨྲེ་བ་ཡིན་ན་མེ་ཝ་ས་མ་བམ་པ་བྱས་པ་ཡིས། །འཇིལ། ་རྟེན་པ། བུམ་པོད་ས་རྒྱལ་པོ་དབང་དུ་འགྱུར། །ཁྲི་ཡིད་འཇིག་རྟེན་།
