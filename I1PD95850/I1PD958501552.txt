སརྦ་ཨ་སུ་ར། སརྦ་མ་ནུ་ཥ། སརྦ་ཏིརུ་ཀ། སརྦ་སྦེ་ཏ་ཀ། སརྦ་ན་ར་ཀ། སརྦ་བྷཱུ་ཏི། པི་ཤཱ་ཙ། ཨུཉྩནྡ། །༼ཨ་པསྨ་ར་ཏ་ཡ། མོས་པའི་ཚུལ་དུ་སྤྱན་་དྲངས་མཆོད། །པདྨའི་ཕྱག་རྒྱས་བླངས་ལ་དབུལ། ་། པུཥྛ། དུ་པ། དཱི་པ། །གནྡྷ། ནི་བི་ཏེ། པཉྩ་ཨ་མྲྀ་ཏེ། མ་ཐ། དྲརྩ། བ་ཏ་ག་ཏེ། ཙ་ཏྲ་་ཙ་མ་ར། ས་ཡོ་ཐ། ་མ་ཙ་མ་ས། མ་ཧཱ་རཀྟ། །ན་ན་ཁཱ་ཧི། སཾ་པུཎྜི༄༅། ཨི་དཾ་བ་:ལིཾ་ཧཱུཾ་ཛ་ཐ། གྲོསྣ་ཐ་ས་མ་ཡ་རཀྟ་ཐ་མ་མ་སརྦ་ཨ་ཐེ། ་ཧ་ས་ཡཾ། ཀུ་རོ་ཐ། །ཨ་ཏྲ། སཱ་དྷ་ནི། །ཡི་ཐ་ཏྲཊྚ་ཤཾ། རཀྵ་སརྦ། ་ཨནྣཾ། །ཡ་ཐི་བོ༑། ༽ཡ་ཐེཊྚཾ། བྷཾ་ཛ་ཐ། ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། པི་བ་ཐ། མ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ་ས་པ་རི། ས་ཏ་སུ་ཁཾ་བི་ཤོ་དྷ་ན་ཡེ་སྭཱ་ཧཱ། ཡི་ཀ་བྷ་བནྟུ། །ཨོཾ་ཨཱཧཱུཾ། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། རོལ་མོའི་སྒྲ་དང་རྡོ་རྗེ་གླུ་བླངས་ཏེ། ལྷག་མའི་གཏོར་མ་དབུལ་བར་བྱའོ། །བདག་ཉིད་དགྱེས་རྡོར་རྣལ་འབྱོར་གྱིས། །ཆོས་འབྱུང་ནང་དུ་འདབ་བརྒྱད་གསུམ། །རིམ་པར་དཀར་ནག་དམར་པོའོ། །དེ་ཡང་ཡིག་འབྲུ་གསུམ་ལས་བསྒོམ། །ལྷ་དང་ཕྱོགས་སྐྱོང་ཀླུ་ཆེན་བརྒྱད། །ཨོ་ཨཱཧཱུྃ་ལས་པད་མདོག་བཞིན། །ཞལ་གཅིག་ཕྱག་གཉིས་རྡོར་དྲིལ་འཆང༌། །རང་གི་ཡུམ་ཡང་དེ་བཞིན་ཏེ། །བརྐྱང་བསྐུམ་ཚུལ་གྱིས་ཀུན་ཀྱང་གནས། །ས་བོན་ལས་ནི་འོད་ཟེར་སྤྲོ། །གཟུགས་དང་གཟུགས་མེད་ལྷ་བརྒྱད་ལ། །།འདོད་ཁམས་ལྷ་མི་ཕྱོགས་སྐྱོང་སྟེ། །ངན་སོང་གསུམ་དང་ལྷ་མ་ཡིན། །ཌ་རྒྱུ། ཀླུ་ཆེན་་་་་བརྒྱད་ལ་བསྟིམ་པར་བྱ། །བདག་གི་ཐུགས་ཀའི་ས་བོན་ལས། །ཕྱག་རྒྱ་བཞི་་ཡིས་སྐུ་གསུང་ཐུགས། །རང་བཞིན་གནས་ནས་སྤྱན་དྲངས་ཏེ། ཡི། ཕྱི་ནང་་་་གསང་བས་མཆོད་བྱ་སྟེ། །ལྷ་དང་ཕྱོགས་སྐྱོང་ཀླུ་བརྒྱད་ལ། ག་། རྣམ་སྣང་མི་་བསྐྱོད་མཐའ་ཡས་ཀྱིས། །རིམ་པ་ཇི་བཞིན་དབང་བསྐུར་བྱ། །།དེ་ལ་སྔགས་་་།
