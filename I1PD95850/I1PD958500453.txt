ཥ་སྐུ་ཞFབསྟན་འགྱུར། རྒྱུད། ཉྩི་ཀ་ཀོ་ཊིཀཤིང་དག་བྱེད་པ། །ལ་། དགེ་ཞིང་མུན་པ་རྣམ་སེལ་བ། །བདག་གི་དད་པས་་་འབུལ་ལགས་ཀྱིས། ་། བཞེས་ནས་བདག་ལ་དགྱེས་པར་མཛོད། །ལྔ་། ལྷན་ཅིག་སྐྱེས་དང་སྦྱར་བའི་དྲི། ་ཁ། ཤངས་ལ་ཚིམ་བྱེད་དག་པའི་དྲི། །བདག་གི་དད་པས་འབུལ་ལགས་ཀྱིས། ལ། །བཞེས་ནས་བདག་ལ་དགྱེས་པར་མཛོད། །།ལྷ་་་བཤོས་གཙང་མ་ཐམས་ཅད་ནི། །བཟའ་དང་བཅའ་བའི་བྱེ་བྲག་བཅས། །ཁ་དོག་དྲི་དང་རོར་ལྡན་པ། །འབུལ་གྱིས་སོ་སོར་བཞེས་སུ་གསོལ། ཝཱ་པི། ཨོ་ཀུ་་རུ་ཀུ་རུ་མ་ཧཱ་བྷཀྵ་བྷཾ་སྭཱ་ཧཱ། ལྷ་བཤོས་འབུལ་བའི་སྔགས་སོ། །།ཨོྃ་ཛ་སྭཱ་ཧཱ། འབྲས་བུ་ཐམས་ཅད་དབུལ་བའི་སྔགས་སོ། །ཨོཾ་ཤྲཱི་ཧཱུཾ་ཧཱུཾ་ཛྲ་སཱ་ཧཱ། ལྕགས་ཀྱུ་ལྔ་དབུལ་བའི་སྔགས་སོ། །སངས་རྒྱས་ཐམས་ཅད་རབ་བསྒྲུབ་པ། །མཆོད་པ་ཉི་ཤུ་བྱ་བ་ནི། །ཧཱུྃ་གི་དམ་ཚིག་ལས་བྱུང་བ། །རང་གི་ཕྱག་རྒྱས་ཐམས་ཅད་་དབུལ། །སྒེག་པའི་ལག་པས་སྒེག་པར་བྱ། །ཕྲེང་བ་ཕྲེང་བའི་མཁའ་འགྲོ་་་་མའི། །།།གླུ་ནི་ཧཱུཾ་གི་དམ་ཚིག་གིས། །གར་ནི་རྡོ་རྗེ་རྐང་པས་བྱ། །དེ་ལྟར་མི་ཝང་གླིང་བུ་སོགས། །མཆོད་པ་བློ་དང་ལྡན་པས་བྱའོ། །དེ་བཞིན་མེ་ཏོག་་་་མར་མེ་དང་། །སྤོས་སོགས་དངོས་མང་ཚོགས་རྣམས་དང་། །སྣ་ཚོགས་གོས་་དང་བླ་རེ་དང་། །དྲིལ་བུ་རྔ་ཡབ་མཛེས་པ་དང་། །བྲན་པོ་བྲན་མོ་:བཞོན་པ་དང་། ྋ། རྟ་དང་གླང་པོ་ལ་སོགས་དབུལ། །བདེ་གཤེགས་གསུངས་པའི་དགའ་སྟོན་ཆེ། །།དཀྱིལ་འཁོར་སྙིང་པོའི་ནང་ཕུལ་བས། །དེ་ཉིད་དོན་ལས་མ་འདས་ཤིང་། །དོན་ཤེས་པ་ཡི་འཁོར་ལོ་དགྱེས། །རྡོ་རྗེ་དྲིལ་བུ་མ་སྤངས་པར། །ཞོ། བསྟོད་བྱ་བཀྲ་ཤིས་གླུ་བླང་བྱ། །།ཧཱུཾ་ཧཱུཾ་ཧཱུྃ། དེ་བ་ེདྷ་རུ་སཾ་ས་ར། ་་དྷ་རུ་དྷ་་ན་དྷ་ལི་ག་ན། ཇོ་དྷ་རུ་ཧེ་བཛྲ་ཏུ་ཧཱུཾ། ཏེ་ན་ཧཱུྃ། ་ཏེ་ན་ཧཱུཾ། ཏེ་ན་ཏེ་ཏེ་ཧཱུྃ། ཕྱོགས་སུ་ར་ན་ར་བནྟི་ཏ་ཙ་ར་ཎ་དྷ་རུ་ཀུ་སུ་མ་བི་མན་མ་ཐ་རོ་ཥ་ཀ་རུ་ཧེ་བཛྲ་..།
