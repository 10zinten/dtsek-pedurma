་དཀར་བ་ཚོས་བོད་སྲིད་ལྟར། །བདག་ལ་ཁང་པ་བསྟན་དགའ་བ་རྒྱུན་ལས་དག་པ་། ཁ། །ཀན་ལཎཥཎ་དེའི་རྗེས་ཞུ་བོད་དུ་ཡི་ཁར་བཞེཀ་ཞ་བོ་ལ། ་་་་། ཁ་ཆེ། ཁ་ཞན་ཆ་ཡོད་པ་ལ་ཁར་ཇོ་ཁྱོ་ག་ཚེ་མི་ཚེ་ལ་བསམ། ཡང་དག་པའི་བདེ་བ་རྫོགས་:ཀན་ཞུ་རྒྱུ་བད་བ་ལུལྟར༌བྷ་དྲུག་སྟེ། །ཨ་ན་ཀ་ཁ་ག་བ་ག་ལ་པཏཏཱིཀ་དུས་བབ་དང་༄༅། །ཀྱེ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་ཀྱི་མདོར་བཤད་པ་པའི་བར་དུ་དག་པ་རྡོ་རྗེ་སྒྲོན་མ་ཞེས་བྱ་བ། །ཁག་ཏུ། རྗེ་བློ་བོ་ཀ་ཎཁྱོལ་བ་ཁ་འཁྱོག་གང་འདཞོ་ཟོ་ཏས་ཅིང་ཚེ་དབང་ཞུ་བ་པོ་དང་རྒྱུ་ཆའི་ཆེ་ཏི་ཊ་ཁོ་ཚོ་ཁྲི་ཤོ་ཀ་ཀ་ཚེ་ན་ལ་ཞབྱིན་:བུ་གླུ་དང་། ཟླ་བློ་ལ་ང་::ཀར་བྱ་བ་o@ན་དང་། །ཚོགས་བདག་ཚོག༄༅། །རྒྱ་གར་སྐད་དུ། ཧེ་བཛྲ་སཱ་དྷ་ན་སྱ་བཛྲ་པྲ་དཱི་པ་ནཱ་མ་ཊི་པི་:ཎཱི་ཤུདྡྷི། བོད་སྐད་དུ། ཀྱེ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་ཀྱི་མདོར་བཤད་པ་དག་པ་རྡོ་་རྗེ་སྒྲོན་མ་ཞེས་བྱ་བ། དཔལ་ཀྱེ་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །རྟོག་པ་མེད་པའི་་་རང་བཞིན་ཅན། །ཀྱེ་ཡི་རྡོ་རྗེའི་ཞབས་ཕྱག་འཚལ། །བསྐྱེད་པའི་རིམ་པ་ལ་་་བརྟེན་ཏེ། །རྣམ་པར་དག་པ་གསལ་བཤད་བྱ། །དང་པོར་རེ་ཞིག་སྒྲུབ་ཐབས་ཀྱི་མདོ་བརྗོད་པར་བྱ་སྟེ། དེ་ནས་མཆོད་པའི་འཁོར་ལོ་དང་། ཚངས་པའི་་་གནས་བཞི་དང་། མཆོག་དང་ལྷན་ཅིག་སྐྱེས་པ་དང་། རང་གི་འདོད་པའི་ལྷ་དང༌། །སྲུང་བའི་འཁོར་ལོ་དང༌། སྒོ་སྟོང་པ་ཉིད་དུ་ལྷག་པར་མོས་པ་དང་། དུར་ཁྲོད་དང༌། གཞལ་ཡས་ཁང་དང་། ་རྒྱུའི་དཀྱིལ་འཁོར་དང་། ཞུ་བས་བསྐྱེད་པ་དང་། བཞེངས་པ་དང་། དགོད་པ་དང་། ཞེ་སྡང་གི་བདག་ཉིད་དང༌། འབྱིན་པ་དང་། ཡེ་ཤེས་ཀྱི་འཁོར་ལོ་དང་། དབང་བསྐུར་བ་དང་། །།
