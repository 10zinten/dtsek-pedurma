བསྟན་འགྱུར། །ཨོ་རྒྱུད། ཏིའིམ་ལ་སོགས་པ་ཆེ་བར་བས། བཞད་ཀ་དེ་ཁྲི་ཚོ་འགལ་བ་ཆེ་བ་དེ་འཛིན་ད་པ་བློ་མ་གྱི་ཁྲོད་ན་མ་ལ་བགོ་བག་ག་ལེན། ཚེ་་ད་ཀི་ཡི་ཁ་ང་ན་ངོ་ཚ་། འོ་ན་གནས་སོགས་ཚོ་བསྡུར་འབྲས་རེའུ་མིགརྐོ་བ་ཁཚེག་ཁ་ཚོ་ལ་ཕྱི་རྒྱུད། （མ་ཏེ། དེའི་རྗེས་སུ་དུ་བ་ལས་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང༌། །ཅོ་ནེ། པ་གོ་སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། ས་བརྒྱུད། ཀུ་འདུ། བོད་རྟགས། །ཏེ། ་་འཚན་གྲངས། ཕྱི་ཡུ་ར་ཝཱ་ཡུps་་ཟླ་་་་་ལྡེབ། །བའབ་ན་བ། ན། ན*། བབ་ལ་ཕྲེང་། ::། ཚ་ཚེག་ཁྱིམ། །།།ཡོད་པ་དང་བག་ལ་བྱང་ཆུབ་ལ། ཕྱིས་ཕྱི་་་་་་་ན་བསག་་་་་་་་་་་་ཡིན་ན་ཟླ་འོ༄༅། །ལོ་ཀཅི@་ཀྱི་ཀློ༌་་་པོསདྲ་ར《་བྱ་ལ་བློ་ཁོ་ན་སྒྱུ་《་དDASདn*་ཨཱལ་n@་ཀྱི་འོ་ལགས་ཀྱི་་ཚོ་སྤྱིར་རྒྱུ་རལ་ནམས་ཀྱི་བདེ་བ་སྐྱེ་བ་ནས། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་པ་དག་པ་ཡིན་ནོ། །དེ་ཡིས་གོ་འབྱེད་སྙིང་པོ་བརྩམས་པ་ཡི། །རྣམས་གཞི་བདུན་པ་དེ་དག་ཐམས་ཅད་ཀྱིས། །དྲི་མེད་བདེ་བའི་རྩ་བ་གང་ཐོབ་པ། རྩོམ་པ་ལ་དུ་བ་བ་ནས་ཀྱང་། །བླ་མེད་བྱང་ཆུབ་ཐོབ་པར་ཤོག། བོད་བར་འབར་བྱེད་པའི་སྙིང་པོ་སློབ་དཔོན་ཆེན་པོ་བྱུང་བ་ཐ་མག་དོན་དང་། ད་བརྒྱུད་རིམ་པ་བདུན་དུ་ཕྱེ་ནས་ཆོས་:::དུས་ག་བ་ང་མར་འབོད་དོ། །+།
