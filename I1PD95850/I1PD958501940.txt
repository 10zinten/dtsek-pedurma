འདི་ནི་དཔལ་ཨུ་རྒྱན་རྡོ་རྗེ་ཆེན་པོའི་གནས་ནས་བྱུང་བ། དཔལ་དེ་ཁོ་ན་ཉིད་ཀྱི་བདུད་རྩི་མན་ངག་དཔལ་ཨིནྡྲ་བྷཱུ་ཏི་རང་ལས་བྱུང་བའོ། །སློབ་དཔོན་་་ཆེན་པོ་རྒྱལ་པོ་ཨིནྡྲ་བྷཱུ་ཏིའི་ཞལ་སྔ་ནས། དཔལ་དེ་ཁོ་ན་ཉིད་ཀྱི་བདུད་རྩིའི་མན་ངག། མཆོག་གི་དངོས་གྲུབ་བརྙེས་ནས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཤྲཱི་གུཔྟ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་་་ཆེན་རྒྱལ་མཚན་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཀྱི་::ཚེན༑། རྩ(《་བསྡུར་མཆན། ཎམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་དེ་ཁོ་ན་་ཉིད་ཀྱི་བདུད་རྩིའི་མན་ངག་བཞུགས། ཚེ«པེ་»«སྣར་»ཏཏྟཱ། Q«པེ་»«སྣར་»ཏོ། «པེ་»«སྣར་ནི། «པེ་»«སྣར་བཞིས། «པེ་»«སྣར་མ། @«པེ་»ཡིད་ཡང་། «སྣར་»ཡང་ཡང༌། «པེ་»«སྣར་བོ། «པེ་»«སྣར་གཉིས། «པེ་»«སྣར་བརྒྱད། :«པེ་»«སྣར་བསྙངས། «པེ་»«སྣར་འབར། @«པེ་»«སྣར་»ས། @«པེ་»«སྣར་»གི། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་རི། །ཚེ«པེ་»ལ་ཁ། «སྣར་ལ་ཁུ། ཚེ«པེ་»«སྣར་བཞག་པ་དེ«པེ་»«སྣར་»ལས། དོན་དུ་ད་བར་«པེ་»«སྣར་»སྒེག«པེ་»«སྣར་སྒོ། «པེ་»«སྣར་སྤྱོན། །ན་གབ་རུ་པ་ད་ལུ་བདེན་པ་བདེ་བར་ན་ཀ་བ་འདྲ་ཁ་པ་ཞེས་བྱ་འཁོར་ལོན་གསུངས། །ལོ་ད་ལོའི་ཁང་པ་ལ་མེད་དག་དག་འདུག་པའི་ས་ན་ལས་མེད་དེ། །བདེ་ལགས་ང་པ་ནས་སོ་དམ་འ་བ་ལ་སོགས་ལོ་སོར་གསུང་དག་ལ་ད་——།
