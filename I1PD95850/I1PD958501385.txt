དེ་ལPSབསྟན་འགྱུར། རྒྱུད། ཏ༽བྱེད་པ་ལ་ལོ་ང་ད་ན་ཀལ་ཀ་ཁ་ཚུ་མཁོ་མི་ཊོ་ཆེ་ཀpབྷ་ཀp/nཏོnཀ་ཁའི་ཚེ་ཀ་ཀ་ཀྐཎ་ཌི། s/pཀ་ར་ཀ་ཚཛོ་བ་ཉི་ཁpཏུ་ཁ་མི་ཁ་ལ་་་་རྒྱུ་བར་ལ་ལ་ལ་ལ་ཞུ་ཀ་ན་ཡོའ་བསྡུར་མཆན། ཅ་ཀ་ཁ་ཟླ་ཀཱ་ཊིཀ་ཊཏོ་༼བྱ་a'སྤྱད་པར་དུ་སྐྱོང་། །ཆུང་བ་ཚ་ཚེར་གྱུར། །ཨེམཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད། «པེ་»«སྣར་»དེའི། ཉི«པེ་»«སྣར་»—པོ། «པེ་»«སྣར་གདང་། ?༽། གྱི«པེ་»«སྣར་བསྟར། ་བྱས་དང་། ཚོགས་་་ਤੋਂਲਨਨਿੰਬਲਰਾਜਸੌਉਨਾਸਨਰਸੰਟਰਸཁ་བ་ཏོང་་ཏུ། །བསྡུར་འབྲས་རེའུ་མིགཔ་འོ། ཁྲོ་རྒྱུ་དང་། ་་་ཡེ་་་། སུ་པོ་ཏ་ལ་ཞོ། །༼བོ་རྒྱུད། »(་ཌ་འོས་ཀྱི་བ་དཎཎ་ཊ་ཏོག། །དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། ལ། ཅོ་ནེ། །སྡེ་ཚན། རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། །པོད་རྟགས། ་ཏ། ཞ་ལུ། ༼ཞག། ：ཏང་ཚན་གྲངས། ལྡེབ། བཙནན་—བ*རབ་—ན་བ་—ན་ཕྲེང་། །།།།།།།::ཀྱཚེག་ཁྱིམ། :། ::::། དེགས་ཡོང་པདྨ་དམར་པོ་འདབ་མ་བརྒྱད་པའི་སྟེང་། ད་དམ་པ་ལ་ཁྱིམ་དང་དེ་འོད་དུ་ཞུ་སྟེ་རིམ་པས། ཁ་དོག་དམར་པོས་པ་དངས་ཤིང་འདཔ་ལ་དེ་མ་མེད་པ། རོ་མཆོག་དང་ལྡན་པར་བསྒོམ། དེ་ནས་ཡང་ཟླ་བའི་ད་ལ་འཁོར་པ་བདུ། ཨི་དཀར་པོ་གཅིག་བསྒོམ། དེ་འོད་དུ་ཞུ་སྟེ་ཁྱིམ་པས་ལ་དང་བ་དོ་མགོ་པག་བརྒྱ་དང་ལྡན་པའི་བདུད་རྩིའི་རང་བཞིན་དུ་ད་པར་བ་དེ་ནས་བདུད་རྩི་ལས་འོད་འཕྲོས་པས། དེ་བཞིན་།
