བསྟན་འགྱུར། རྒྱུད། ཏིབསྐུར་ནས་རོ་ཁྲུས་བྱ། ལྷར་བསྐྱེད་ལ་ཡེ་ཤེས་པ་གཞུག། དེ་ལ་དབང་བསྐུར་དེ་་་དུས་རང་གིས་ཚིག་བཤད་བརྗོད། དེ་ཡང་དེའི་སློབ་མ་སྙིང་ཉེ་བས་བྱའོ། །རུས་པ་དེ་བུམ་ཆུས་ཞག་བདུན་ལ་ཉི་མ་རེ་རེ་ལ་ཁྲུས་འཁོར་གསུམ་གསུམ་དུ་བདུན་ཕྲག་་བདུན་དུ་བྱའོ། །ཡི་གེ་བརྒྱ་པ་བཟློས་ཤིང་བྱ་གསུངས། །་ཀg། ལྔ་ཥ། དལ་བ་''ཎ་ལ། པད་དཀར་་བློ་ཊོ་ཁཁ་ལ་ཀཾ། ཤ་ལ་ཀླུ་བ་་་ཁRབ་། ཚོ་me། །བ་ཏུ་དར་བ་ལ་བློ་དང་ཀོང་ਦੇARਰਸਨੌਂਰਸੌਲੁਕਨਸੌਰਸ਼ਲਸਟומימוןטיראןבייממימייןגייגיגיגןམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་རོ་བསྲེག་པའི་ཆོ་ག་་་་་ཞེས་བྱ་བ་བཞུགས། «པེ་»«སྣར་སྦྱིན། ཤ་»ཊིརེཤ་ཥཀ་ཁཊ་ཚོ°་ཁ་། །།།ཁ་ལྷོ་། །ལ་་ཎ་རཏ་ཁྲོ་ཚོ། ཁ་ཚོ་དཱི་ཀ་ཀྐཎྜིཏྟ་ངོ་། །ཁྱབྷPཊ་པར་དུ་བོད་དང་བསྡུར་འབྲས་རེའུ་མིག་ཅ། །།ར་བ་དཔག་པའི་ཚོ་བ་ཞེས་པ་ལ། རྒྱུད། （་ཏེ། པ་རps:འི། pལ་རདཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། པ་རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། །རྒྱུད། དེ་པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། ན་བན་ན། ན་ན་བ་ན་ཕྲེང་། ༼༽༼(ཡདེ་ཡང། ཀླུ་མ་ཚེག་ཁྱིམ། ད་ལ། །་འི་་ར་བ་གྲུ་ཁ་ལ་ཁ་pཚོ་འགྲོ་བ་ཀ་འཚོ་བ་ཡོད་ལ་ཁ་ག་ཁྱིའི་ཏེ། རིམ་ཐ་ཏུང་ཀྱིས་ཞིབ་འོ་མ་ཁ་ལ་ཁྱེར་དག་ཏུ་ཁག་ཏུ་མ་བ་སྟེ་ཟླ་ཆུ་དབཀ་འད་དུ་ཀ་ཀ་ཀྱི་ཚོགས་ཀ་ན་ཀ་པཱ་ཀ་སྐེ་བ་དང་ཁ་ཏུ་ཀུའི་།
