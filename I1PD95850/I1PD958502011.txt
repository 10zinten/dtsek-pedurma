བསྟན་འགྱུར། ་རྒྱུད། ཏིའིཔ་གང་རྒྱུད་པདྨ་ལ་དགའ་ཡིས་གསུངས་རྣལ་འབྱོར་མ་ལ་སོགས་ནི་རྨད་དུ་བྱུང་། །དེ་ཡི་ནང་ནས་ཤིན་ཏུ་རྣམ་དཔྱད་འདི་དྲངས་བླ་མ་ཀུན་གྱི་ཡང་དག་ནོར༑། འདིར་ནི་བདག་རང་གིས་ནི་ཅུང་ཟད་རང་བཟོ་མ་ཡིན་འོན་ཀྱང་འདུས་པ་བྱས། །དེ་ལས་གང་ཞིག་འབྲས་མཆོག་ཐོབ་པ་དེ་ནི་སེམས་ཅན་ཀུན་ལ་འཕེལ་བར་ཤོག། ::ཀྱིག་ལཌ་ད་བ། ཁ་ཕྱོགས་ལ་བློ་ཁོ་རྡོ་ཀ་གོ་ཁཚོ་ཁ་དོག་པ། ཁཡན་ལག་བརྒྱད་པའི་རིམ་པ་ཞེས་བྱ་བ། རྒྱ་གར་གྱི་སློབ་དཔོན་དཔལ་ཤེས་རབ་གསང་བའི་ཞལ་སྔ་ནས་མཛད་ནས་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པའོ། །།།།ཨ་ཊི་རྫོགས་སོ། །་། ལོག་ཎ་པ་ཏོ་ལ་ཁང་པ་ཁོ་ར་ས་བབའི་ན། བྱི་བ་ཀ་ལཱ་པ་གནས་ཀྱང་ཁོང་ཁྲོ་མ་བྱི་དང་ཁིའིལ་སོགས་པ་ད་ཁ་རྒྱུ་དེ་སུ་དང་། ལགཤོབ་ལ་ཟླ་མཚན་དང་ལས་བསྡུར་མཆན། ::དོ་དང་རྒྱ་གར་བ་ཕྱོགས་ཀྱི་འཚོ་ཆམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཡན་ལག་་་་་བརྒྱད་པའི་རིམ་པ་བཞུགས། «པེ་»རྡོ། «སྣར་རྡོ་རྗེ། «པེ་»«སྣར་བ། «པེ་»«སྣར་»—གང་། «པེ་»«སྣར་»—པ། «པེ་»«སྣར་»ཨོ་ཌཱཎ། «པེ་»«སྣར་»ལྷན་ཅིག@«པེ་»«སྣར་»གྱི། ཚ«པེ་»«སྣར་»+པའི། ཚེ་ང་ལ་མི་ཚེ«པེ་»«སྣར་བྲིས་སོ། «པེ་»«སྣར་ཀྱི། «སྣར་»འདི་ལྟར། «པེ་»«སྣར་བརླབ་སྟེ། «པེ་»«སྣར་པོར། «སྣར་»བེ། «པེ་»«སྣར་»ཧཱུྃ། «པེ་»«སྣར་»སིཏྟི། འ«པེ་»«སྣར་»ཧཱུཾ། «པེ་»《སྣར་»བཞགཚེ«པེ་»«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་»ར། ་་། ན་@«པེ་»«སྣར་འཇིགས། «པེ་»«སྣར་བཞགཉི«པེ་»«སྣར་»ནམ། «པེ་»«སྣར་»བཞག«པེ་»«སྣར་+ཀྱི། @«པེ་«སྣར་»—སྟེ། ༄༅«པེ་»«སྣར་»བཞི་པ་ནི་སློབ་མའི་སྟ་གོན། «པེ་»《སྣར་+དཀྱིལ་འཁོར་བའི། །«པེ་»«སྣར་+དང༌། «ཅོ་»འཕེན། «པེ་»«སྣར་»མའི། ,&«སྣར་མོ། @«པེ་»«སྣར་བཞགཚེ«པེ་»«སྣར་»འཇིག་རྟེན། ཚེ«པེ་»«སྣར་»བུ། ཚེ«ཅོ་»བོ། ཡིད་་གལ་་་་ཚུ།
