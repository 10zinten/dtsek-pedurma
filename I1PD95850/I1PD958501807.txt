བསྟན་འགྱུར། རྒྱུད། ༽མཛད་པ་རྫོགས་སོ། །།།།།ང་ལྟ་བ་བདེ་ཀློབ་གེ་ཟླ་བ་བོད་པ་སྤྱི་པ་རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ་དཱ་ན་ཤཱི་ལས་རང་འགྱུར་དུ་་་་མཛད་པའོ། བ་། ག་ཎ། དེ་ཉིད། ཁཀ་ཏོ་ཚང་ཁ་ཁཚོ་ཁར་ང་ཚོས་ད་ལྟ་ད་བརྒྱ། ཀག་ན་པ་ཏ་ལ་སོགས་པ་བར་གྲགས་ཚོད་ཀྱི་དེ་ལས་པ་ག་ལ་བཏང༌། དེ་ལ་ཀཁ་པཟོ་ག་ཏེ། སྤྱི་ཚོགས་འདུ་བའི་ང་ལ་བསྡུར་མཆན། ས་ག་ཚོས་ག་ཁཚོ་དེ་བལྟ་བ། བྱ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་འཕགས་པ་རྡོ་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་པ། །«པེ་»«ཅོ་»མ། ཉི@«པེ་ཨོཾ། «སྣར་ཨོཾ། «པེ་»«སྣར་བཛཱི། ཨུ«པེ་»«སྣར་རཏྟ། །«པེ་»«སྣར་+བཛྲ། «པེ་»«སྣར་པ། «པེ་»ཨོ། «སྣར་གིས། ®«པེ་»«སྣར་»གསོལ། a་ནས་བརྒྱུད་ལ་ཆུ་ན། ཀློ་པ་«པེ་»«སྣར་»+གནས་པའི། །«པེ་»«སྣར་»བཛྲ། *«པེ་»«སྣར་»པ་དེ། བ་ལ་ཁ་གཏེར་ཁ་ཀུན་ཏུ་«པེ་»«སྣར་བླ། ཉི«པེ་»«སྣར་པཾ། »/«པེ་»སྣར་»དྷ། «པེ་»《སྣར་»ཛཱ། @«པེ་»«སྣར་མ། «པེ་»«སྣར་»+ཀྱི། ༄༅«པེ་»«སྣར་གོང་། ཉི«པེ་»«སྣར་»ཡོངས་སུ་དག་«པེ་»«སྣར་»—སྒོ་བཞི་པ། ཉི«པེ་»«སྣར་»རྭ། «པེ་»«སྣར་»མ། «པེ་»མཆོག་«པེ་»«སྣར་བསྡིགས། ཚེ«པེ་»«སྣར་བསྟེན། རྒྱུ་«པེ་«སྣར་པ་བསྡིགས། ཚེ«པེ་»«སྣར་མཛད། «པེ་»«སྣར་འདུས། དེ་བས་ང་ལ་དགའ་བ་མ་ཤེས་པ་«པེ་»«སྣར་»«ཅོ་»ཤོ། «པེ་»«སྣར་ནཾ། ན«པེ་»«སྣར་»གསོལ། ཨེ«པེ་»《སྣར་པ། «པེ་»«སྣར་»ཞུགས། «པེ་»«སྣར་+ཏ། །«པེ་»«སྣར་ཧཱུྃ་ཨོཾ་ཛྲ་ཁཾ་ཧཱུཾ་ལཾ་མཾ་པཾ་ཏཾ་ཁཾ། བདག་ཚེ་ན་ད་ལྟ་བ་དང་སྐྱབས་«པེ་»«སྣར་བརྟེན། ཉི«པེ་»—ཞབས་གཉིས་ལ། ཉི«པེ་»«སྣར་»—ཕཊ། «པེ་»«སྣར་པས། «པེ་»«སྣར་»ནམ། «པེ་»«སྣར་ཀྱི། །«པེ་»གྱི། «པེ་»དུཋ་ན། «སྣར་»དུཋ་ན་ས། Q«པེ་»«སྣར་»ཚོ། «སྣར་བྷནྡྷ་ཡ་བྷནྡྷ། «པེ་»«སྣར་»ཤྲཱི། «པེ་»«སྣར་»སྒྲུབ། @«པེ་»«སྣར་»ཀི། «པེ་»«སྣར་»འབྱུང་། «པེ་»«སྣར་»+གི། «པེ་»།
