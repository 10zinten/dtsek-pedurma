རརབསྟན་འགྱུར། རྒྱུད། ཏི༽ད་པ་བྱེ་བ་ས་ཡ་བཏེ་སྐལ་གྱི་ཨ་མ་ཚེ་ལྷ། བུ་དད་པ་བྱེད་པ་མ་རང་དབང་། ཚོད་བདག་རྒྱལ་བཞི་པོ་གན། །ཁོ་བོས་གོ་མེད་གང་དུ་འཚོ་བྱ་ན་འགྲུད་ཏ་ཙམ་ང་། ང་ཚོ་ལ་དབ་ཀ་བསྡུར་མཆན། འབོད་། དག་པ་ནཥ་་པ་སྐབས་འཚོ་བ་ད་དུ་བསྡུ་བ་ན། །དགའ་བ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །དབང་གི་རིམ་པ་མདོར་བསྡུས་པ་བཞུགས། ཚེ«པེ་»«སྣར་»ཞི་བཱ། Q«པེ་»«སྣར་པའོ། «པེ་»《སྣར་ངོ་མ་གཉིས་དང་སྙུན། «ཅོ་སོམ་ཉི་དང་ནི་གཉིས། །ཨེ«པེ་»«སྣར་ཆུང་། ®«པེ་»«སྣར་ཡི། «པེ་»«སྣར་—བུམ་པའི་དབང༌། «པེ་«སྣར་ཡ། །«པེ་»«སྣར་»བསྐྱོད། སྤྱི་དོནལ་འགོ་བ་ལ། །བ། ::ཀྱི། །ངོ་། ཨེ«པེ་»«སྣར་»དབུ། «པེ་»«སྣར་ཆེ། «ཅོ་—བདགད་པར་དཀའ། ཏ་ཀ་བ་། :རྗོབསྡུར་འབྲས་རེའུ་མིགརྒྱུད། （པེ་ཅིན། སྣར་ཐང་། རྩ་ན། དཔེ་གཞི། །སྡེ་ཚན། །པོད་རྟགས། སྡེ་དགེ། ·རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། ཏཟgཏཚན་གྲངས། །།།ལྡེབ། །བ་ན། ནབབསན་བ་ན་༎ཕྲེང་། །།།།ཚེག་ཁྱིམ། །།༽། བབབབད་ངོ་། བར་པ་བཅུ་དྲུག་པོ། །ད་པས་ད་ན་བ། །དེ། །ག་ད་བ་ཚོ་དང་དོན་མེད། ཡི་ལད། ད་ལ་དད་པས་མེད་ཀྱིས་གྲྭ་བ་དུམ་བུ་ཤེས་རབ་—།
