བཛྲའི་སྒྲུབ་ཐབས་སློབ་དཔོན་བྱང་ཆུབ་སྙིང་པོས་མཛད་པ་རྫོགས་སོ། །།།::ཞུ་ལ་རྒྱ་གར་གྱི་མཁན་པོ་དཱི་པཾ་ཀ་ར་ཙནྡྲ་དང༌། །བོད་ཀྱི་ལོ་ཙཱ་བ་ལྷ་བཙན་པོ་མུ་ནི་རཱ་ཛས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པའོ། །ད་དེ། རྒྱས་པ་ག་ཚོད་མེད་སེར་པོ་ཞི་བ་ག་ལ་པོར་ཤེས་པར་བྱ། །ད་བ་་ལ་་ད་མང་པོར་རབ་ཏུ་འདུག་པས། །དེ་བཞིན་གསོད་་ལ་ནག་པོར་འདོབསྡུར་མཆན། དེ་ལས་ནི་དཔང་དུ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་ཀྱེ་རྡོ་རྗེ་་་ཞེས་བྱ་བའི་སྒྲུབ་པའི་ཐབས་བཞུགས། «པེ་»«སྣར་བདག«པེ་»«སྣར་»—ཀྱི་བདག་ཉིད། «པེ་»«སྣར་ཀྱི། ི«པེ་»སྣར་—ལཱི། «པེ་»«སྣར་»—ལྡན། @«པེ་»«སྣར་—ནི། །ཨེ«པེ་»«སྣར་རྗེ་ཡིས། «པེ་»«སྣར་གསོལ། «པེ་»«སྣར་ཡིས། «པེ་»«སྣར་»ལས། @«པེ་»«སྣར་»ཨཱ། །ཚེ«པེ་»«སྣར་»གཤེད། «པེ་རྣམ་པ། 《སྣར་»རྣམས། ང་དང་། །བ་«པེ་»«སྣར་ཌོམྦིའོ། ་་་པརཔ་དངདདརའི་Trཟླ་་ག་པའི་EAའབའི་ཁྱུ་y\། ཀཱ«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་»དེ་ཡིས་བློ་ཡིས། «པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་»ཀྱི། བ་གཞི། །ད་ཀྱི་འཁྲུ་བར་བྱ། ཁྲི་འབ་པ། །ཎ། ཁོ་ཊི་ཁ་ཊཀྱི་ཀུན་ཕུར་བུ་ཀྲ་བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (དཔེ་གཞི། །།སྣར་ཐང་། །ཅོ་ནེ། སྡེ་དགེ། རྒྱུད། པེ་ཅིན། རྒྱུད་འགྲེལ། སྡེ་ཚན། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང་། ཚེག་ཁྱིམ། །ནན་། ་། །༼ཀ་༽་ཟླ་pY»geན་—ན། །བ་བ་། ན་ན་། །
