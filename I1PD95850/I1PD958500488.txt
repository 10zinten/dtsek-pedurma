རྫོགས་སོ། །།པ་བ། དེ་ཁོ་ན་ཉིད་དད་པའི་རྒྱ་གར་གྱི་མཁན་པོ་ག་ཡ་དྷ་ར་དང་། ལོ་ཙཱ་བ་དགེ་སློང་ཤཱཀྱ་ཡེ་ཤེས་་་་ཀྱིས་བསྒྱུར་བའོ། །དང་། ཡུམ་གྱི་མ་ཡུམ་དུས་ཞེ་ན། དེ་ཕྱི་ཡིན་ལ། ཕྱིའི་མཚན་མས་ནང་ད་ན་མ་ར་བྱེད་པའོ། དེ་ཡང་དཔལ་ཤེས་པ་དང་། པོ། །སྤ་དུང་། ད་དུ། །བསྡུར་མཆན། སུམ་བརྒྱ་ཡན་དུ་བསྡུར་མཆན། །བས། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །འབྱུང་པོ་ཐམས་ཅད་པའི་གཏོར་མའི་ཆོ་ག་བཞུགས། ཚེ«པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་རིག®«པེ་»«སྣར་ཡང་དག་ཉི«པེ་»«སྣར་མ། ཚེ«པེ་»«སྣར་མཉེས་བྱེད་ཁྱོཟད། ་མ«པེ་»«སྣར་»ཧུཾ་ཨོཾ་ཛཾ། «པེ་»«སྣར་རཀྵ། «པེ་»«སྣར་སྭཱཛྫ། ཚེ«པེ་པ་ཏཱ། ལ། ཨཊྛ། «སྣར་པ་ཏཱ་ལ། ཨཊྚ། «པེ་»«སྣར་དྷ། ཨུ«པེ་»«སྣར་དྷ། «པེ་»«སྣར་ཁུ། ཚེ«པེ་»«སྣར་ཁུ། ཉི«པེ་»«སྣར་ཨཱ་ཀ@«པེ་»«སྣར་»སིདྷིམྨེ། ཁྱེ«པེ་»«སྣར་»—མུ། ཀྱེ«པེ་»«སྣར་»ལྷ་མཉེས་པ་དང་། །ཁ་ཏུ་ཀི་ན་བཙོ་བཀpཁྲུལ་པཎན་འ་ར་བ་ལྔ་པོ་ལ། །ཀུ་བ་བཏོཁཀ། །ཎ་ཊ་ཀ་ཊི་ཀ་། །།པར། དོ་་་ཤ་་། ང་ཚོ་ལཁྲི་ཀླུ་ནང་ལོ་ང་བཞིའོ། བསྡུར་འབྲས་རེའུ་མིག་ཁྲོ་བ་ཁག་ཀྱང་པར་ཁ་བ་བབ་ཀྱང་ཤེས་མ་རྒྱུད། （དཔེ་གཞི། སྡེ་དགེ། ཀྱིཔེ་ཅིན། སྡེ་ཚན། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བ?བ། བ་_བརབ་བསབ?བ། ཕྲེང༌། ཚེག་ཁྱིམ། །སྣར་ཐང༌། རྒྱུད། རྒྱུད། རྒྱུད། /།
