བསྟན་འགྱུར། རྒྱུད། ཏིན། མ་ཡ་ཀན་ཁ་དཀར་pte%། ་ལཚུཀ་ཀུ་ཏmpཀ་དེ་ཀ་ལ་། ཉ་ཎ་་་་ཏོ་ཀ་ཀ་རླཁ་སྤུའི་ཁ་ལ་ཁ་ཕྱོགས་ན་ཁཚོ་pའི་འོ་ཏཱ་ཎ་བློ་་་་་ས་ཁྱོན། །ལ་ལ་ལ་ཚོeབསྡུར་མཆན། ་ཏཥཱཀ་ཀ་ཁ་ཚོ་་་་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད། @«པེ་»«སྣར་པའི། །་་་་་nཥ་ཚེ་ས་ཁFཁ་ཁ་བ་བློD་དང་པ་ག་དབང་དུ་བསྡུས་ན་ཆུ་མི་ཡང་པ། གདེ་gypཙག་། ་་ཟླ་། ཁ་ཀའ་ye/wfབསྡུར་འབྲས་རེའུ་མིག་། གང་ལས་ཚོ་རང་ལ་རྒྱུ་ཁོ་རྒྱུད། (་ང་ར*#/ཀའ་འགོ། «ਤੋਂਲਵਾਲਓਟਾਲਸਰདཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། །ཅོ་ནེ། སྡེ་ཚན། རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན་—བ། །བན?བt_ན་བསན་། རྒྱུད། ཕྲེང་། ཚེག་ཁྱིམ། །མ། ར། ་ང་མང་དང་དམག་བྱ། །ཉི་མ་ཐོག་ཀུང་པའི་ཉེན་པ་ལེགས་པར་བསུ། །འོ་མ་དྲི་ཞིམ་གྱིས་ཕྲུས་བྱ། དེས་ནི་ངདེ་བས་པར་དྲི་ཞིམ་པོས་ཁྲུས་བྱ་བ་དང་། ཡོམ་ལ་སོགས་པ་ན་པ་དང་། དཀྱིལ་འཁོར་བྲི་བ་དང་། དཀྱིལ་འཁོར་པ་ལ་ས་ལ་བྱནས། ཁུ་དང་། དབུ་རྒྱན་དང་། རྡོ་རྗེ་དང་། བལོ་ན། བུམ་པ་ཨ་མས་ཀྱིས་དབང་བསྐུར་ནས་རང་མི་།
