ས་ཕྱོགས་བསྟན་འགྱུར། རྒྱུད། །ཏིའིབར་དང་ལུང་རིགས་ཀྱི་དབང་ཕྱུག་སྐད་གཉིས་སྨྲ་བའི་གཙུག་གི་ནོར་བུ། དཔལ་ལྡན་བྱང་ཆུབ་རྩེ་མོའི་བཀའ་དྲིན་ལས། །བརྡ་སྤྲོད་པའི་ཚུལ་ལ་བློའི་སྣང་བ་ཅུང་ཟད་ཐོབ་པ་ཤཱཀྱའི་བཙུན་པ་སྐྱོབ་པ་དཔལ་བཟང་པོ་ཞེས་བྱ་བས་བསྒྱུར་བའོ། །།།བ་ལ་རྒྱུད་རྒྱལ་རིན་ཆེན་འབྱུང་གནས་བསྲུབས་པ་ལས། །འབྱུང་བའི་ནོར་བུ་རིམ་གཉིས་འོད་དཀར་ཅན། །གང་འདིས་བོད་ཡུལ་གདུལ་བྱའི་ཀུ་མུད་ཚལ། །རྒྱས་པར་བྱ་ཕྱིར་སྐྱོབ་པ་དཔལ་བཟང་པོས། །བསྒྱུར་ལས་བྱུང་བའི་དགེ་བ་རབ་དཀར་བ། །དེ་ཡིས་འགྲོ་ཀུན་རྡོ་རྗེ་འཛིན་གྱུར་ཅིག་། ལྔ་བ་ཁག་ཅ་པཎྜི་ཏ་ལ་མ་གཏུགས་པ་དང་། རྒྱ་དཔེ་གཉིས་པའང་མ་རྙེད་པས་ནོངས་པའི་ཆ་མཆིས་པ་རྣམས་མཁས་པ་རྣམས་ཀྱིས་བཟོད་པར་མཛད་ནས་སླར་བཅོས་་ཤིང་ཉེས་པའི་ཆ་རྣམས་འདོར་བར་བྱའོ། །།།ལག་རྒྱུད་ཀྱི་ཕྱོགས་གལ་ཏཊ། བོད་པ་ལ་བྱ་བ་དེ་ང་ལ་ཁྲིམས་བཀོਅਫ਼ਸਲਮਕਰਾਸਨਲਬਬਕਮਾਲཁང་དུ་བསྡུར་མཆན། རྒྱུ་ལ་འོ་ཡུ་བ་ལ་བར་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ཀྱེའི་རྡོ་རྗེའི་མངོན་རྟོག་ཐིག་ལེ་བཞུགས། ཉི«པེ་»«སྣར་ཟླུམས། «པེ་»«སྣར་»པས། ཚུ«པེ་»«སྣར་ཁ་མ། ན་ཆུང་བྱིས་(«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»བདག«པེ་»«སྣར་—པ། «པེ་»«སྣར་»«ཅོ་»+ལ། («པེ་»«སྣར་མའི། ®«པེ་»«སྣར་པོའི། ཚེ«པེ་»«སྣར་བསྐྱེད། །པ་ས་ར་མ་ས་ག་«པེ་»«སྣར་»—དང་། @«པེ་»«སྣར་»གི། «པེ་»«སྣར་»«ཅོ་»རཾ། །«པེ་»«སྣར་པས། «པེ་»«སྣར་པ། O«པེ་»«སྣར་ཨོ། ་། «པེ་»«སྣར་ཉིད། «པེ་བསྐྱེད། «སྣར་བསྐྱད། «པེ་ནག《སྣར་སྔོན། «པེ་»«སྣར་+པ་དག@«པེ་»«སྣར་»རྟོགས། །«པེ་»«སྣར་བ། «པེ་»«སྣར་»ཀྱིས། །
