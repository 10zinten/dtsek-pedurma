།རྣམ་པར་དག་པའི་གཏེར་ཞེས་བཤད་པར་ཁས་བླངས་ནས། །བདག་གིས་བྱས་་པས་ཀྱི་བསོད་ནམས་ཉེར་བསགས་ངེས་པ་དེས་ནི་སྡུག་བསྔལ་རྣམ་བསལ་ནས། །འཇིག་རྟེན་གསུམ་གྱུར་སྐྱེ་བོ་མྱུར་དུ་ཕྱག་རྒྱ་ཆེན་པོའི་གནས་སུ་འགྲོ་བར་ཤོག། བདག་གིས་སྒྲུབ་ཐབས་བརྩམས་པ་ཡིས། །དགེ་བའི་རྩ་བ་གང་ཐོབ་པས། །ཁམས་གསུམ་སེམས་ཅན་ཐམས་ཅད་ནི། །བདེ་གཤེགས་སྐུ་གསུམ་ཐོབ་པར་ཤོག། དཔལ་ཀྱེ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་ཀྱི་རྒྱལ་པོ། སློབ་དཔོན་ཨ་བ་དྷཱུ་ཏཱི་པ་གཉིས་མེད་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ། །།།།་ཀྱི། ད་ལོ། དམ་པའི་ལས་ཀྱི་བྱེད་པ་་byuyo+་ཚེ་་་་དོང་། རྫོགས་པ་དང་། །དབྱེ་ན་སྤྱི་ལུ་མ་བྱང་ཀྱི་་བསྡུར་མཆན། །དེ་ལ་དང་བའི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །ཀྱེའི་་ཁ་ཟླ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་»གསུམ། ཚེ«སྣར་མོ། «པེ་»སྣར་བེཏྟཱ། ཆེ«པེ་»«སྣར་»མི་རུས། «པེ་»«སྣར་དེ་ནི། གང་«པེ་»«སྣར་»ཤིར་ཥའི། «པེ་»«སྣར་»ཏའི། །«པེ་»«སྣར་གིས། ཉི«པེ་»«སྣར་ཁ་དཏྟ། «པེ་»«སྣར་»གི་བི་ཧྭ། །«པེ་»«སྣར་»—པོ། @«པེ་»«སྣར་བེཏཱ། «པེ་»«སྣར་»གིས། ཉི«པེ་»«སྣར་+བྱུང་ནས། །«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་བསྡིགས། །«པེ་»«སྣར་དུད་པའི། །ཚེ«པེ་»«སྣར་ཞུགས། །«པེ་»«སྣར་»+དག་པར། «པེ་»«སྣར་ན། །«པེ་»«སྣར་ཀྱིས། །«པེ་»«སྣར་ཧཱུྃ་ཨོཾ་ཛི། ®«པེ་»«སྣར་»བསྐྱེད། «པེ་»«སྣར་ཧཱུྃ། «པེ་»«སྣར་»གིས། གྱི«པེ་»«སྣར་»རོའི། «པེ་»«སྣར་ཀྱི། ཉི«པེ་»«སྣར་རྗེ། ཉི«པེ་»«སྣར་སྐརྡོ་རྗེ། «པེ་»«སྣར་ཉི་མིའི། «པེ་»«སྣར་ལས། «པེ་»«སྣར་ཧཱུྃ་ཨོཾ་ཛི། @«པེ་»«སྣར་»+ཨེ་བོ་མཱ་ཡ་མཚམས་བཞིར་དགོད་དོ། །།
