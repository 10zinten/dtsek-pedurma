བསྟན་འགྱུར། རྒྱུད། ཏའིསྲོག་སོང་སྟེང་ན་གཡོན་བརྐྱང་གནས། །སྐྲ་སེར་གྱེན་འགྲེང་ཕྱག་རྒྱ་ལྔ། །ཐོད་ཕྲེང་ཐལ་བས་རྣམ་བརྒྱན་ལ། །དེ་རྗེས་ཡེ་ཤེས་སེམས་དཔའ་གཞག། སྐྱེ་་་མཆེད་དྲུག་དང་སྐུ་གསུང་ཐུགས། །དབང་བསྐུར་བདུད་རྩི་མཆོད་པར་བྱ། །ཨོ་ཨཱༀ་ཧཱུྃ་སྭཱ་ཧཱ། བཟླས་པའི་སྔགས་སོ། །ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨ་ཎཾ། ཨཱརྱ་་་་ནུཏྤནྣ་ཏྭཱཏ། ཨོཾ་ཨཱཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། གཏོར་མའི་སྔགས་སོ། །ལྷ་མོའི་ལྟེ་བར་ཟླ་རྒྱས་འབར༑། ཨོ་ཡིག་ཡེ་ཤེས་ལྔ་ཡི་དངོས། །སྲིད་དང་མྱ་ངན་འདས་པའི་རྒྱུ། །དེ་ཉིད་འཁོར་བ་ཞེས་བྱ་སྟེ། །དེ་ཉིད་མྱ་ངན་འདས་པ་ཡིན། །འཁོར་བ་ཡོངས་སུ་ཤེས་གྱུར་ན། །མྱ་ངན་འདས་ཞེས་དེ་ལ་བརྗོད། །ཇི་ལྟར་ནོར་བུ་རིན་པོ་ཆེ། །མ་ཕུག་པས་ནི་སྦྱོར་བྲལ་འགྱུར། །རྣམ་པར་ཕུག་པས་སྦྱོར་ལྡན་པ། །འཁོར་་བའི་རྣམ་པ་དེ་ལྟ་བུ། །བླ་མ་རྣམས་ཀྱིས་གསུངས་པར་འགྱུར། །་མི་དཔའ་གཅིག་མའི་སྒྲུབ་པའི་ཐབས་བསྐྱེད་པ་དང་རྫོགས་པར་ལྡན་པ་ཞེས་་བྱ་བ་སློབ་དཔོན་བསོད་སྙོམས་པ་སྒྲ་གཅན་འཛིན་གྱིས་མཛད་པ་རྫོགས་སོ། །།།་་ལ་ཚོ་རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཨིནྡྲ་རུ་ཙི་དང་། ་དགེ་སློང་ཤཱཀྱ་ཡེ་ཤེས་ཀྱིས་་་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།/། ཀ་ཋཏུ་ཏ་ལདབས་ཁོ་ཆ། །ཁ་ཏི་ཎ། །ཀ་ངོ་ཁཎྜ་ལ་ཀོ་ཀ༌། །ཀུ་ཊལ་བ་ཀོ་ཥ་་ཚོ་་༌། །p/:འོ་ན་ཁ་བ་ང་ཚོ་ད་ཀ༌་ཏe'ཏུ་ཆ། ཁ་ཏི་ཀ་ཎཊི༌ཁ་གྱི་བྱ་འིབ་པ་དང༌། །རྒྱུ་ཀཔཆི་ཁ་ལ་ཚ་། ཆུ་ལ་ས་ཏུང་། ཤzགྱི། །ཚེ་ཉཁ་ཕྱོགས་། ཆབ་འཚོཌ། ཁོ་བ་ལ་ད་ལྟ་ཎ་ཥ་ཎ་ཟླ་ལྔ་སྤྱི་བྱZAARA'Bਟੋ'ਸਰਮ'ਓਸਲཎི་ཎ་ལ་ཎ། བོ། །ང་ཁཚོ་ཁཝིཀ་ཏ་ཀའི་ར་བ་/། །།ཁ་pཊིཀ་ཁ་ཏུ་ཁ་ཚོན་། ༌་pཊཌ་ལ་ཀ་ལ། ཀ་ཏའུ། ཁ་ཏི་ཀ་ང་ཀ་ཎ/ཙི་ལ་/། །ཁ་དེ་ལ།
