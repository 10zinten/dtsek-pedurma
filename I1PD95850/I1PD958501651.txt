རབསྟན་འགྱུར། རྒྱུད། ཏི༽བསྟན་པ་བཞིན་བསྒྲུབ་པར་བྱའོ། །བུད་མེད་སོགས་ཀྱིས་རྒྱས་པ་དག་། ཕལ་ཆེར་གཟུང་བར་དཀའ་བསམས་ནས། །ཞི་བའི་ཞབས་སོགས་ཀྱིས་བསྟན་པའི། །རྣལ་འབྱོར་བསྡུས་པ་བདག་གིས་བྱས། །།་་་་། །ཚུལ་:བདག་མེད་མའི་སྒྲུབ་པའི་ཐབས་མདོར་བསྡུས་པ་སློབ་དཔོན་ཀྲྀཥཱ་པཎྜི་ཏས་མཛད་པ་རྫོགས་སོ། །བདག་ལ། ་ལ་དར་ཆ་ཡིན་པས་ཁོ་ཚེ་ང་ཚོ་བ་བདེ་བ་པའི་ལོ་དང་དེ་འ་ལ་བ་ལ་ཁྱོན་དུ་ང་ཚོ་ནས་ཡོད་པ་དང་ད་དུང་བན་བ་དང་། །ཁོ་བསྡུར་མཆན། ཝ་ཏ་::ཀྱི་། །ཏེ་ཁྲི་ཚོ་གignaronདུ་འོ་བརྒྱལ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་བདག་མེད་མའི་སྒྲུབ་་་པའི་ཐབས་བཞུགས། «པེ་»«སྣར་བྲིས། @«སྣར་དེ། །«པེ་»«སྣར་»བི་ར། ཉི«ཅོ་»དེའི། «པེ་»«སྣར་»བེཏྟཱ། པརད། བཞེས་ཞིང་པའི་«པེ་»«སྣར་»སྣང་བ། «པེ་»«སྣར་མར་རྫོགས། ང་ད་ན་དང་ན་བ་་«པེ་»«སྣར་དཀར། །ལ་ཁལ་ཁབ་ཟ་བ་ཚོས་ཁ་བ་བབ་པ། ་་་་་ལ་གྱིས། ཁྱོད་::ཁོ་ར་ག་ཏ་བ་ལ་བarས་ཏེ་་་་་་་་་་ཀུན་ཏུ་རི་ཀཉིད་དང་། བསྡུར་འབྲས་རེའུ་མིགਕਮਲਸਲਾਤਕམིག་གསལ་བ། ་་་་་ལ་་རྒྱུད། （འོ་ཏེ་པ་ཞེས་ཀྱི་ཚོན་གང་དཔེ་གཞི། །སྡེ་དགེ། །སྣར་ཐང་། །ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད་འགྲེལ། བོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན། བ་ན་ན་ནན། ན་བ་། ཕྲེང་། །，། །ཚེག་ཁྱིམ། །།།།།།པེ་ཅིན། རྒྱུད། རྒྱུད། །།
