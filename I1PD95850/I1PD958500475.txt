ཁ་n་༌་བསྟན་འགྱུར། རྒྱུད། ་ཌ་པཎ་ཧཱུྃ༑་སྐུ་ཁྲུས་གསོལ་བའི་སྔགས་སོ། །མ་ཤ་ཀ་ལས་མི་ཆུང་བའི། །གསེར་གྱི་ཐུར་མས་རྡུལ་ཚོན་ལས། ་། མིག་སྨན་མི་མང་བ་བླངས་ཏེ། །བ། འདི་དག་གིས་ནི་སྤྱན་དབྱེ་བྱ། ོ། །ཨོཾ་ཙཀླུ་ས་མནྟ་ཙཀླུ་བི་ཤོ་དྷ་ནི་སྭཱ་ཧཱ། ཆ་གོས་ཀྱིས་ཆུ་ནི་ཕྱི་བྱས་ཏེ། །གནས་གང་ཡིན་པར་བརྟན་བཞུགས་ལ། །ལྷ་ཡི་འཁོར་ལོ་རེ་རེ་ལའང་། །གོས་ནི་ཟུང་ཟུང་དབུལ་བར་བྱ། །ཨོཾ་བཛྲ་:ཝསྡུ་ཡེ་སྭཱ་ཧཱ། །གོས་དབུལ་བའི་སྔགས་སོ། །་། ཞབས་བསིལ་ཞལ་བསིལ་མཐོར་འཐུང་དང༌། །མཆོད་ཡོན་ཕུལ་ནས་དེ་རྗེས་སུ། །སྦྱང་བ་བྱས་པའི་མཐའ་མ་རུ། །མཆོད་པ་རྒྱ་ཆེན་བྱ་བ་ཡིན། །ཨེ་མ་པུཏྤ་ཤུ་བ་དྲ་བྱ། ་ཤུ་ཙ་ཡཾ་ཤུ་ཙ་ཡོ་ནཱ་མ། ཡ་མ་ནི་བི་ཏོ་བ་རྒྱུ་པྲ་ཏི་སྲྀཙཱ་པྲ་སིདྡྷི་མེ་བ་ནསྤ་:ཏི་ར་ས་ཧྲི་དོ་གནྡྷ་དྷུ་པ་མུ་དྷ་མ། ཡ་མ་ནི་བི་དི་ཏོ་བཀླུ། པྲ་ཏི་གྲོསྣ་པྲ་སད་མེ། རཱགྷོ་གྷ༔་པ་པི་ཏྲ། ་ཏ་མོ་བྷི་དྷ་མ་ན་སུ་བྷ། མ་ཡ་ནི་བི་དྷི་ཏོ་བྷཀྵ་པྲ་ཏི་ཤྲཱིཉྩ་པྲ་སིད་མེ། ས་ཧ་ཛ་སཾ་ཤོ་ག་ཀི་ཤུདྡྷཾ་གནྡྷོ་ཡཾ་གྷ་ཎ་ཌརྦ་ཎ། ཡཾ་མ་ཡ་ནི་བི་དོ་ཀྱོ་བཱ་གྲུ་པྲ་ཏི་སྲྀཧྥ་པྲ་ས་ད་མེ་ནི་བིདྱེ་སརྦ་སཾ་ཡུསྶཾ། ཀི་ཁ་ད་བྷོ། ་ཛ་ས་མན་ནི་ཏཾ། ཝརྞ་གནྡྷ་རྭ་སོ་བི་ཏཾ་ད་དྷ་མི་པྲ་ཏི་ཤྲཱིཧཱ་ཏཾ། ཨོ་ཛཾ་སཾ་སྭཱ་ཧཱ། འབྲས་བུ་ཐམས་ཅད་དབུལ་བའི་སྔགས་སོ། །ཨོཾ་པྲ་ཏཱིཙྪ་ཨགྷཾ་ཧཱུྃ་སྭཱ་ཧཱ། མཆོད་ཡོན་དབུལ་བའི་སྔགས་སོ། །རྡོ་རྗེ་དྲིལ་བུ་གསོར་བྱེད་པ། །ཕྱག་ན་རྡོ་རྗེ་མཚུངས་ཉིད་ཀྱིས། །བཀྲ་ཤིས་ཚིགས་བཅད་གླུ་བླངས་ནས། །ཅ་། སངས་རྒྱས་རྣམས་ལ་་་གཏོར་མ་དབུལ། །ཎ་མ་ཧཱུཾ་ཀཾ་ཨེར་ན་བུ་རུ། སེ་ཛི་བི་སདྡྷཱ་ཨུཏྟ་རུ། །ཏེ་ན་ཧཱུཾ་ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཏེ་ཏེ་ཧཱུྃ། དྷཾ་བ་ལ་སུ་སཾ་ཁཾ་ཏེ་ཧ་དྷ་རུ་ས་ར་ཡ་སུ་སོ་བྷེ་ཙནྡྷ་མུ་ཧཱུཾ། ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཧཱུཾ། ཏེ་ན་ཏེ་ཏེ་ཧཱུཾ། ་་:དན་བྷི་ལ་ཀ་ན་ཡོ་ཨུ་དྷ་རུ་གཎྜ་བཛྲ་་་་མུཏྟ་དྷ་རུ། ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཏེ་ཏེ་ཧཱུཾ། ་་:མཱ་ཡ་བི་དྷེ་ཧི་ན་ཛ་གུ། སོ་ཨི་ཨ་ཀ་ར་ན་རཱུ་པི། ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཧཱུྃ་ཏེ་ན་ཏེ་ཏེ་ཧཱུཾ། གླུ་དཀྱུགས་ལྔ་པ་བསྲེས་།
