བསྟན་འགྱུར། རྒྱུད། ཏི༽ནས་མེ་ཏོག་གཞག་པར་བྱའོ། །ཤར་དུ་ནི་ཧཾ་གོ་རཱི་ན་མཞེས་བྱ་བས་སོ། །ལྷོར་ནི་ཙོ་ཙོ་རཱི་ན་མཞེས་བྱ་བས་སོ། །ནུབ་ཏུ་ནི་ཐཾ་:བེ་ཏཱ་ལཱི་ན་མཞེས་བྱ་་་བས་སོ། །བྱང་དུ་ནི་གྷོ་གྷསྨ་རཱི་ན་མཞེས་བྱ་བས་སོ། །དབང་ལྡན་དུ་ནི་་་་་པུཀྐ་སཱི་ན་མཞེས་བྱ་བས་སོ། །མེའི་མཚམས་སུ་ཤཾ་ཤ་བ་རཱི་ན་མཞེས་པས་སོ། །བདེན་བྲལ་དུ་ནི་ཙི་ཙཎྜ་ལཱི་ན་མཞེས་བྱ་བས་སོ། །རླུང་མཚམས་སུ་ནི་ཌི་ཌོམྦི་་ཎཱ་ན་མཞེས་བྱ་བས་སོ། །ཨོ་དེ་བ་པི་ཙུ་བཛྲ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཞེས་་་་བས་བཅོམ་ལྡན་འདས་ཡང་དག་པར་མཆོད་ལ། བདག་མེད་མ་ལ་ཡང་། ཨོ་ཨ་ཨཱ། ཨི་ཨཱི། ཨུ་ཨཱུ། རྀ་རཱི། ལི་ལཱི། ཨེ་ཨཻ། ཨོ་ཨོ་ཨོཾ་སྭཱ་ཧཱ། ཞེས་བྱ་བའི་སྔགས་ཀྱིས་མཆོད་དོ། །ཡང་བཅོམ་ལྡན་འདས་ལ། ཨོ་ཨཏྟཱན་ནཱ་ཡ་ན་མ༑ཨོ་པིངྒ་ཨུཛྷ་ཀེ་ཤ་བརྟ་མ་ནེ་ན་མ། ཨོ་ཙ་ཏུར་བི་ཤ་ཏི་ནེ་ཏཱ་ཡ་ན་མ། ཨོ་ཊོ་ཌ་ཤ་བྷུ་ཛཱ་ཡ་ན་མ། ཨོ་ཀྲྀཥྛ་ཛི་མཱུ་ཏ་ཝ་པུ་ཥ་ན་མ། ཨོ་ཀ་པཱ་ལ་མཱ་ལཱ་ཨ་ནེ་ཀ་དྷ་རི་ཎེ་ན་མ། ཨོ་ཨདྷཱ་ཏ་ཀླུ་ར་ཙིཏྟཱ་ཡ་ན་མ། :ཨརྙན་དུ་ྋདཾཎྜི་ཎེ་ན་མ། ཞེས་བྱ་བའི་ཚིག་བརྒྱད་ཀྱིས་མཆོད་པར་བྱ་སྟེ། ཨོཾ་བཛྲ་པུཔྤེ་ཨཱཧཱུྃ༑ཨོཾ་བཛྲ་དྷུ་པེ་ཨཱཿཧཱུཾ། ཨོཾ་བཛྲ་གྷནྡྷེ་ཨཱཧཱུཾ། ཨོཾ་བཛྲ་ནི་བི་དྱེ་ཨཱཿཧཱུཾ། ཞེས་བྱ་བས་བྱིན་གྱིས་བརླབས་པའི་མེ་ཏོག་ལ་སོགས་པ་ཕུལ་ལ། ལན་གསུམ་གྱི་བར་དུ་སྐྱབས་གསུམ་གྱི་ཚིགས་སུ་བཅད་པ་བརྗོད་དོ། །ཡང་མཆོད་པ་བྱས་ནས་ཕྱག་བཙལ་ལ། གཏོར་མ་བྱིན་ནས་བཛྲ་མུཞེས་བྱ་བས་གཤེགས་སུ་གསོལ་བར་བྱའོ། །དེའི་རྗེས་ལ་སེམས་ཅན་ཐམས་ཅད་ལྷའི་བདེ་བ་ལ་དགོད་པར་འདོད་་པའི་བྱམས་པ་དང་། སྡུག་བསྔལ་ཐམས་ཅད་བསལ་བར་འདོད་པའི་སྙིང་རྗེ་་་་དང་། ཁ་ྋལྷའི་བདེ་བ་རྒྱུན་མི་འཆད་པར་ངེས་པར་འདོད་པའི་དགའ་བ་དང་། བླ་ན་མེད་པའི་ལམ་ཉེ་བར་དགོད་པར་འདོད་པའི་བཏང་སྙོམས་བསྒོམ་པར་བྱའོ། །།
