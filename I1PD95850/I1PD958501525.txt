ལ་བསྟན་འགྱུར། "རྒྱུད། ཏི༽གས་བས་ག་ཟླ་བ། ལྷ་?། ཁ་ཁ་ཤེས་མ། གྲོ་ཡེ་ཤེཁ་མ་བྲག་དང་། འ::ཀ་ཁ་འད། ཏོ་གཟེར་བ། །བོང་སྐྱེ་བ་བཞི་ན་ཁར་གཏཔཀུན་ཏུ་འབྲི་བར་བྱ་བ་དང་བསྡུར་མཆན། nཚོན་ན་བཀའ་བ་དུ་མ་དང་། ཡེ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»དཔལ་ཀྱེའི་རྡོ་རྗེའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་བཞུགས། །«སྣར་ཀྱེའི་རྡོ་རྗེའི་...ཆོ་ག་བཞུགས། །«པེ་»«སྣར་—པ། «པེ་»«སྣར་»གསོལ། །«པེ་»«སྣར་ལ། «པེ་《སྣར་རྡོ་རྗེ། «པེ་»«སྣར་»_ཡོན། ཀ་ཏཅཙྪ་ཚོ། པའི་བ་ཀུའང་བྱ། ཏཎ་ཀཞུ་ན་ཁ་ར་ཏ་ཞུམ་འགྲོ་བ། །རིགས་ཀྱི་ཁ་ཊོ་ཁོ་པ་དག་བྱ་བ་གཙོ་། །ས་བ་ཞའོངས་བསྡུར་འབྲས་རེའུ་མིགཔདང་ལུ་ཀ་ཀ་ཀོ་བ་པོ་ཤོག་། ་་་་རྒྱུད། (ཁ་ཏོན་ལོ། :བོ། དཔེ་གཞི། སྡེ་དགེ། ད་པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། སྡེ་ཚན། ༼རྒྱུད། རྒྱུད་འགྲེལ། །ས་རྒྱུད། །རྒྱུད། །པོད་རྟགས། །ཏ་། ཟན། ཟན། ཚན་གྲངས། ལྡེབ། ན་_བ་ན་_བ་ན་_བན་—བ་ཕྲེང་། ::ཚེག་ཁྱིམ། །།།།ཤིས་ཀྱི་ཚོང་ཁང་ཀ་བ་ཀ་བ་གས་ལ་ཁ་མ། ད་ལྟ་པ། ད་པ་བོར་ཏ་ཀོ་ཀ་ཌ་། ལྷ་སྐུ་ཆུ་ཁྱེར་བ། ཚོང་བདག་ཡོ་རོ་ཏི། སོ་ལ། ་་་་། ཁ་དེ་ན་ཏོ། །ཁ་། ་་་་་འཁོར་འདས་ཀྱིས་བབ་པའི་ཚོང་པ་བ་དང་བདག་ངོ་བོ། ནམ་པས་ཁོ་ལ་།
