བསྟན་འགྱུར། ་རྒྱུད། ཏིགས་པར་བྱས་ལ། དེ་ཡང་ད་བར་འོས་པའི་གནས་སྐབས་དང་སྦྱར་བ་ནི་རྗེ་སྒྲུབ་པ་བ་མ། སློབ་དཔོན་ལ། ཡི་ལ་ད་ལྟའི་སྡིག་སྒྲིབ་ཀུན། །བརྒྱ་པའི་དང་། དད་པ་དྲག་པ་བ་མ། བད་ནབསྡུར་འབྲས་རེའུ་མིག་བཟླས་པར་བྱ། བ་་། :། །རྒྱུད། （་ལ། །།znamnpa་་་རྒྱས་དོངོ་གསལ་ལ་རྒྱུས་གསརs/sའི་ཁུ་ཚོ་ཡིད་འདས་པ་ན་ལ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། །།རྒྱུད། །རྒྱུད། མན་པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། ，ནསན། བརབབའབ་། བན་ཕྲེང་། །ཚེག་ཁྱིམ། །།།།།བའི་ནང་དང་། དྲ་བ་ཀོ་ད་པ། །ནརབལ། ད་བར་དུ་མ་བུ་བ་དག་ལ། ཚོ་ལ། །ད་ལ་ཐ་དད་གང་ཡིན་ས་བད»ཚེ་ཀྱི་དུས་དཀྱིལ་ནི། ཀ་ཁྱུ《སྐྱིད། »D་བ་ལྔ་པའི་ཀྱི་ཆེད་་མ་་་་་་ཀྱི་་་ཀྲབ་། ཁ</aཔ་་ལཊ་ཀྱི་ལ°བ་བ་བ་ལ་དང་དགའ་བ་ཡོད་པས་་་་་་་་་་་བདག་ལ་བབབ། །Z་ཀ། བ་ནི་ཀ་ཁ། པ་གདོན་མ་ལས་བྱེད་པོ་ཡེ་ཤེས་སྙིང་པོའི་ཞལ་སྔ་ནས་མཛད་པ། དེབ་དེབ། འདུ་བ་ཡོད་པ་ངོ་། །།
