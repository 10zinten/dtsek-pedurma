ཐའ་ཁྱུ་ཆེ་བསྟན་འགྱུར། །རྒྱུད། ཏའི་ཌོ་སློབ་ཀརྨ་དང་མཚུངས་རེད་ལ་ལུས་སྦྲ་དང་། བ་གཉིས་་་དྲིལ་བུར་འབྲེལ། །ད་པ་ག་འདྲ་བ་ཡིན་པ་ཁྱད་པ། དུ་བ་པོ་ལ་དབུ་པ། པའི་ཕྱུག་བླ་མ་ལ། །ཁོད་་་རྒྱུ་སོགས་དོ་བསྡུར་འབྲས་རེའུ་མིགཔར་དབང་ཏེ། །ཀུན་བད（སྤྱི་ཟླ་པ::དམར་བྱང་ལ་p:དགའ་བ་བདེ་བ་ཞིག་ན་བ་aཀྱི་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ::ཅོ་ནེ། །སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། བོད་རྟགས། ཚན་གྲངས། ཞིབ་ལྡེབ། ་། ནབ། ནན། བ?བ་ནབཕྲེང་། །་། ཚེག་ཁྱིམ། །་། ན་ཟབ་ལུ་མ་་་་༌་་ལ་ལ། དེ་ལྟ་མང་པའི་ནང་། །་་་་སྤྲོ་བ་བར་བར་ཆད་ཞི། ང་བོད་པ་དང་་་་ལ་་་་་་པ་ར་ཡིག་བས་རྣམ་པར་བྲལ་བའི་མ་བ་འགྲོ་བའི་བའི་ཐབས་དང་པོའི་པ་བ་དང་བོད་ཀྱི་ལོ་བསོད་ནམས་རྒྱ་བ་དང་། ས་པད་མ་བདེ་བ་ན། །དད་དད་པ། དེ་ཉིད་།
