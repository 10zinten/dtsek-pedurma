རྒྱུད། པསམ་མིང་བྱང་ལ། ས་ཚ་གདབ་ཅིང་དབང་བསྐུར་བྱ། །ཞེས་འབྱུང་བས་ས་ཚའི་ཆོ་ག་ཇི་ལྟ་བ་བཞིན་བཏབ་པ་ལ་དཀྱིལ་འཁོར་དུ་དབང་བསྐུར་ཏེ། རྣམ་་་པར་ཤེས་པ་དགུག་པ་སྔོན་དུ་འགྲོ་བས། རང་གི་ལྷར་བྱས་ལ་སྡིག་པ་བཤགས་པ་ལ་སོགས་པ་བྱས་ཏེ། རིག་པའི་དབང་ལྔ་བརྟུལ་ཞུགས་དང་དྲུག་བསྐུར་རོ། །།སྡོམ་པ་འབོགས་པར་མི་བྱ་སྟེ། ཤེས་པ་ཀུན་དུ་ཟིན་ཀྱང་དབང་པོ་མེད་པའི་ཕྱིར་རོ༑། རོ་སྲེག་ཆོ་ག་བདག་གིས་ནི། །བྱས་པའི་དགེ་བ་གང་བསགས་པ། ་། དེས་ནི་འཇིག་རྟེན་ཐམས་ཅད་ཀྱིས། །དྲི་མེད་ཡེ་ཤེས་ཐོབ་པར་ཤོག། རོ་སྲེག་པའི་ཆོ་ག་སློབ་དཔོན་ནག་པོས་མཛད་པ་རྫོགས་སོ། །་བྷོ། །རྒྱ་གར་གྱི་མཁན་པོ་ནག་པོ་དང༌། འགོས་ལྷས་བཙས་ཀྱིས་བསྒྱུར་་་བའོ༑། །ཁྲོ་བ་རྒྱུ་ཆ་མི་མང་གོ་མ། ག་ད་ཀ་ཀུ་བྷ་་ལ་གཤིན་ཏེ། དེ་དག་བློ་ཁ་དང་། བྱ་བ་ਲਨਕਵਸਨੂੰਉਨ੍ਹਾਂਤੋਂਬਚਾਥੋਂਸਦਾབསྡུར་མཆན། མཁན་ཚོས་ཀྱང་། །བ་ཀྱིས་ང་ཚོའི་སྤྱི་ཁཀའ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སྦྱིན་སྲེག་གི་ཆོ་ག་བཞུགས། ®«པེ་«སྣར་+ལྔ། «པེ་+སྤྱི་བོར་རྩ་བའི་སྔགས་གདགས། «སྣར་»+སྤྱི་བོར་བའི་་་སྔགས་བདགས། །«པེ་»«སྣར་»ཀྱི། :«པེ་»«སྣར་»གཟུངས། «པེ་»«སྣར་»+ལྷ། «པེ་»སྣར་+འདབ་མ། ::«པེ་《སྣར་བཏང་བར། བཡན་གྲུ་ཆུ་ཚོ་ར་ཚེ་དབང་བས་«པེ་《སྣར་»—ལས། «པེ་»«སྣར་»དྲང་། @«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་བཞི། «པེ་»«སྣར་»ཧཱུྃ། ཚེ«པེ་»«སྣར་»+མཆོད་པ་ལྔ་དང་ནང་གི། །«པེ་»«སྣར་གསོལ། ཉི«པེ་»«སྣར་མནས། ་ཚོར་་་། དུབླ་བའི་«པེ་»«སྣར་གི་མཆོད་པས་མཆོད་དེ། «པེ་»«སྣར་ཚོགས་ཀྱི། ཉི«པེ་»«སྣར་»ཤ་རུ་དབུལ། ཆེ«པེ་»«སྣར་ནི་གཙོ་བོ། ཚེ«པེ་»«སྣར་པའི། ཚེ«པེ་»«སྣར་+གློ་བ་དང་རྒྱུ་མ་དང་། ཚེ«པེ་«སྣར་+དུའོ། །ལ་—།
