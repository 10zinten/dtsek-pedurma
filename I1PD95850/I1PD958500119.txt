སྤྱི་«པེ་»«སྣར་»སྒྲ། «པེ་»«སྣར་+དོ། །«པེ་»«སྣར་བསྐོར་བ། «པེ་»«སྣར་»«ཅོ་»བསྐྱེད། «པེ་»«སྣར་»ལྡོག«པེ་»«སྣར་ལ་རྙེས། «པེ་«སྣར་ཡི། བྱེ་བ། ཟླ་བ་གང་བའི་བར་ཟླ་སྔོན་མ་དག། ཁམཁོ་ཚེ། དྲག་ཏོ་ཀྱི་སྐོར། ད་ཁྱོད་མི་རབསྡུར་འབྲས་རེའུ་མིག་ཁ་དེ་ལ་འཁྲིདུ་བ་ལ་་་་རྒྱུད། (མ་མོལ:ཉེ་དང་། དཔེ་གཞི། །སྡེ་ཚན། །སྡེ་དགེ། རྒྱུད། པེ་ཅིན། སྣར་ཐང་། རྒྱུད་འགྲེལ། ད་རྒྱུད། །ཞབས་ཞན། །རྒྱུད། ད། པོད་རྟགས། ::་ཉ:ཚན་གྲངས། ལྡེབ། ་། ནབབཙབ་། བ་ན་། ན་བ་པ། ཕྲེང་། །ག་ག་ཚེག་ཁྱིམ། ་མ་?་བླ་་::gཔ། པ:ཀ་ཡོད་པ་ཅི་ཡིས་ལས་པ་བན་པ་བདེ་བས་ཕོ་བ་རླུང༌། ང་ལ་ཁ་གྲ་ལ་གྱུ་ཆ་ཆ་ལ་སྤྱི་ཁྲ་ཁྲ་ཞུ་་ལོ་ཀ་ཁ་ན་ཕྲེང་བ་བསྐྱོད་དེ་སངས་རྒྱས་ཀྱི་མཁྱེན་པའི་ཡེ་ཤེས་དྲན་པར་བྱས་ཏེ་དེ་ནས་ཡང་ལ། ལ། ད་ལ་ཡ། ཡོད་ན་དེ་ལ་ཆུ་གཏོར་བ་ལ་པར་བྱས་ཏེ་པwཝ་ལ་ཡུག་པ་དང་།
