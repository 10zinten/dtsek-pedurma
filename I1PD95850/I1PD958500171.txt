རབསྟན་འགྱུར། རྒྱུད། ཌབསྡུས་པ། སློབ་དཔོན་བོ་དྷི་གནྡྷས་མཛད་པ་རྫོགས་སོ། །།།།བ་མོ་བློ་ལ་རྒྱ་གར་གྱི་མཁན་པོ་ཀ་མཱ་ལ་ཤྲཱི་དང་། བྱ་དགེ་སློང་ཤེས་རབ་སྐྱབས་ཀྱིས་་་བསྒྱུར། །་ཚོ༽། ཀ་ཁ་ཎ་ཊི་ད་ལོ་ལོ་། བལན་ཚུ་PRབ་མ་ཎ་བླ*gཔ་ཟླ་་་་་་་་་ཊི་ཏཆ། ཆུ་ཀྵ་ཀ་ཁ། སྐྱོ་པ་ཨ་ལམ་ལ་ཁ་བསྡུར་མཆན། ་༌་དཀའ་ཚུ་བསོཀ་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་ཀྱེ་རྡོ་རྗེ་ལས་སྦྱིན་སྲེག་གི་ཆོ་ག་བཞུགས། «པེ་»«སྣར་»བཞགཀྱི་ཆུ་ཀའི། མན་པ་ཞུ་གཏཚེ་བློ་སྒོ་ཀཀྐཀྱི«པེ་»«སྣར་བཞག«སྣར་བཞག«པེ་»«སྣར་དམར་པོའི་ཆུ་ལ་སྦང་། &«པེ་»«སྣར་པའི། «པེ་»«སྣར་»མཆོད། «པེ་»«སྣར་»གསུང་ལྡིར། «པེ་»«སྣར་ཨཧྥེ། «པེ་»«སྣར་»ཀཱ་མཱརྦ། «པེ་»«སྣར་»ཎ་ཀྲི་ཏཱ། «པེ་»«སྣར་བུ། «པེ་»«སྣར་»མེ། /«པེ་»«སྣར་ལས་བཀུག@«པེ་»«སྣར་བེཏྟཱ། «པེ་»«སྣར་པུཎ། «པེ་《སྣར་བཛྲ་ཨ་ཨ་མུ་ཀམ«པེ་»«སྣར་»ཧཱ། «པེ་»«སྣར་+ཡེ། ཨ«པེ་»«སྣར་ཨཛྷཱ་ཡེ། །«པེ་»«སྣར་ཚིག་བཤད། «པེ་»«སྣར་གིས་ཚིག་བཤད། །བཞི་ཀྱི་«པེ་»«སྣར་»གིས། ་་་:prལ་བ་ཚོལ་དུ་བྱས་པ་ལགས་ཀྱིས་ཝི་ཏོ། :::བསྡུར་འབྲས་རེའུ་མིག་དང་བློ་ཁ་རྒྱུད། (ཕྱི་ལོ་སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། gi*རྒྱུན་དེས་ན་འདི་ག་:དཔེ་གཞི། །སྡེ་ཚན། །རྒྱར་རྒྱུད་འགྲེལ། རྒྱུད། འི་པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། །ན་ན?ཕྲེང་། །ཚེག་ཁྱིམ། །།ལོ%/་ཚེ། །རབ་བ*ན་ན་ན་རབདང་། །+།
