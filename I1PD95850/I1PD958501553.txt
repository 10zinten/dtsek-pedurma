བསྟན་འགྱུར། རྒྱུད། ཏིའདིས་རབ་ཏུ་འབུལ། །པདྨའི་ཕྱག་རྒྱས་བླངས་ནས་སུ། །ཨོ་ཨིནྡ་ཡ་མ་ཛ་ལ། ཡཀྵ་ཛཧཱུྃ་ཁཾ་ཧོབྷུ་ཏ་བསྟྲཱི། བཱ་ཡུ། ་རཀྑ། ཙནྡ། སུརྫ་མཱ་ཏ། ཊ་པཔྤ། ཏ་ལ། པཱ་ཏཱ་ལ། ཨ་ཋ་མིསཔྤ་སྭཱ་ཧཱ། ཨི་དཾ་བ་ལིཾ་ཧཱུཾ་ཛ། ཛི་གྷ་མཾ་ས་དྷུ་པོ་པུ་དཱི་པཾ། གནྡྷེ། པི་གྷ། ཨོཾ་བ། ཀཙྪ། ཁནྟི། ཁུ་ཎི། ཕེ་ཊ། ག་ད། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ། སརྦ་དྷརྨཱ་ཎཾ། ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ། ཨོཾ་ཨཱཿཧཱུྃ་སྭཱ་ཧཱ། རྡོ་རྗེ་གླུ་དང་རོལ་མོའི་སྒྲས། །མཆོད་དེ་ཉིན་མོ་ལན་གསུམ་གཏང་། །ཡི་དམ་ལྷ་ཡི་རྣལ་འབྱོར་གྱིས། །མཆོད་པའི་རྫས་ལ་བསང་གཏོར་བརླབ། །འདབ་་་བརྒྱད་དབུས་ཀྱི་ལྟེ་བ་ལ། །མཁའ་འགྲོ་འཁོར་བཅས་དམིགས་བྱས་ཏེ། །།བཛྲ་ཌཱ་ཀི་ནི། སཾ་ྋགི་ནི། བྱཱ་གྷི་ནི། ཛཾ་བུ་གི་ནི། ལུ་ཀི་ནི། ཌཱ་ཀི་ནི། དྲི་པི་ནི། ཙུ་པི་ནི། ་ཀཾ་པོ་ཛི་ནི། བརྗོད་པ་ཙམ་གྱིས་དམིགས་བྱས་ཏེ། །འདབ་བརྒྱད་ལ་ནི་མཁའ་འགྲོ་མ། །གྲུ་ཆད་ལ་ནི་ཁམས་གསུམ་གྱི། །གཙོ་བོ་ཐམས་ཅད་བསམ་པར་བྱ། །ཁྱམས་ལ་འཁོར་རྣམས་ཐམས་ཅད་དོ། །ཨོཾ་ཨཱཧཱུཾ། ཀུན་ཀྱང་སྐུ་གསུང་ཐུགས་དང་ལྡན། །བྷ་གའི་ཕྱག་རྒྱ་མགོ་སྟེང་བཅའ། །ཨོ་་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱ་ཎཾ། ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ། སྤྱན་དྲངས་བསྟིམས་ལ་བར་་་ཆོད་བགེགས། །འོང་བ་སྲུང་ཕྱིར་འདི་སྐད་དོ། །ཨོཾ་ན་མས་མནྟ་ཀཱ་ཡ་ཝཱཀ་་ཙིཏྟ་བཛྲ་ཎཾ། ན་མོ་བཛྲ་ཀྲོ་དྷ་ཡ། མ་ཧཱ་དཾབྷོ་ཏ་ག་ཊ། བྷི་ར་བཱ་ཡ། ཨ་་སི་མུ་ཤུ་ལ་པནུ་པཱ་ཤ་ཧསྨཱ་ཡ། ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི། ཁ་ཁ། ཁཱ་ཧི་ཁཱ་ཧི། ཏིཥྛ་ཏིཥྛ། བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན། མ་ཐ་མ་ཐ། གརྫ་གརྫ། སྤོ་ཊ་ཡ་སྤོ་ཊ་ཡ། བི་སྤོ་ཊ་ཡ་བི་སྤོ་ཊ་ཡ། ཌ་། སརྦ་བིགྷཱ་ན་བི་ནཱ་ཡ་ནཾ། མ་ཧཱ་ཀ་བི་དེ། བི་བི་ཏནྟ་ཀ་རཱ་ཡ། སརྦ་གོ་རཱི་ཨུ་པ་ཐ་བ་ས་མ་ནཾ། ཀུ་རུ་ཀུ་རུ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། དེ་ལྟར་བདུད་རྩིས་མཚམས་བཅད་ནས། །པདྨའི་ཕྱག་རྒྱས་།
