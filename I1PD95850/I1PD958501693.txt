བསྟན་འགྱུར། རྒྱུད། །ཏེརྒྱལ་བ་ཀུན་ཀྱང་ཕྱི་མོ་ཀྱང་ཁོ། ཊི་ཏན་ས་ཁ་ཁ་ཀར་བོ་པ། ཁྱི་ཁཆུ་ལ་ཎཎྜ་ཚོeཆ་ཀལོ་ལྔ་ཡི་ཚེ་ཆོ་སྤྱི་ལ་རྒྱུ་ཁྱབ་ཏུ་བསྐྲུཁ་བློ་ལpཏཁ་པར་་་། འི་ཁཚོ་ཁང་ཀྱི་འཁ་ཡོཀ་ཊབསྡུར་མཆན། འཁ་ཏུ་བོ་དྷིང་་ང་ད་པ་དང་། དྲོ་བ་དང་། བད་ཀན་དང་། ཁྱོས་ཁ་ཏོ་་n»ི་ཡ་པཚོ་ཏོ་བ་ར་ཎཱ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། ®«པེ་»«སྣར་—པ། enz་། །འོ་་་་་་ཁ་npའི་། རྒྱུ་ཚོགས་ཁ་གས་ཆ་ཆེ་བ་ལ་༄༅། ཁ་ཆཥ་:ཁ་ཀཱ་ར་་་། མ་ནས་བསད་པས། ང་བསྡུར་འབྲས་རེའུ་མིགP༼་ཀྱི་ཁ་ལ་ཁཚོ་དཀཱ༌ལ་ཌ། །ཌ་འཚོ་། pཚོ་ཛོ་རྒྱུད། (་ཌ་་ལུ་ཀ་ཀ་ཁ་ལ་ཀ་ཏོརབ། ར་ན༌་་་་་༌ང་ད་ར་ད་སྣར་ཐང་། ཅོ་ནེ། པེ་ཅིན། དཔེ་གཞི། །སྡེ་ཚན། །ངའི་པོད་རྟགས། སྡེ་དགེ། རྒྱུད། ཏ་རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། ཚན་གྲངས། ལྡེབ། བ་ཤན[བ—བར། ན་—བ་བ*ན་ཕྲེང་། ཚེག་ཁྱིམ། །་་་N། པ། ཡི་ཀྱ་འདུ་ད་པར་འགྱུ་བ་འདྲ། །དྲི་བཟང་པོའི་བདུག་སྤོས་ཀྱིས་བདག་སྐྱེ་བོ་ཐམས་ཅད་དགའ་བར་འགྲོ་བ་རོ། གང་གི་མིང་གིས་མེ་ཏེད་པ་ན་བ། སུ་པས་སྦྱིན་སྲེག་བྱས་ན་དེ་དབང་དུ་འགྱུར་རོ། །བཟའ་བ་དང་བཏུང་བ་བད་པ་དུ་བ་ལགས་ཏེ་བྱིན་ན་དེ་དབང་དུ་འགྱུར་རོ། །སྐྱེས་བུ་ཆེན་པོ་དབང་དུ་བྱེད་པ་ན། ཡུངས་ཀར་པ་སྦྱིན་སྲེག་བྱས་ནས། དེ་ནས་ཚེས་བླངས་རྒྱུ་སྐར་དབང་པོ་ལ་བཅོམ་ལྡན་འདས་མའི་མཆོད་པ་བྱས་ནས་ཁུ་བ་ཀུ་མ་དང་། དེ་ལ་ལབ་པ། དུ་པ་གོས་སུ་གྲོ་མ་འདུ་བྱ། ད་ལའི་འདབ་མ་།
