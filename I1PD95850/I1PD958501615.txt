བསྟན་འགྱུར། རྒྱུད། ཏ། གྷསྭཱཛྫ། མཱ་ད། པདྨ། ཏ་ལ་པ་ཏཱ་ལཱི། ཨཥྛ། སཔྤ་སྭཱ་ཧཱ། ཨི་དཾ། བ་ལཾ། བྷུཉྫ། ཛི་གྷ། ་ཁ་ཕུལླ། ་དྷུཔྤེ། པུ། ཚྭ་ས། །བིང་གྷ། ཨམྤ། སཊྚ། ཀཛྫ། སཱ་དྷ། _ཁནྟི། །ཁུ་ཎི། ཕེ་ཌ། །ག་ད། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨ་ཎཾ། ཨཱདྱ་ནུཏྤནྣ་ཏྭཱཏ། །ཨོཾ་ཨཱཿཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ང་། །།།ཚོ། མཁས་པ་ཆེན་པོ་དཔལ་མི་ཐུབ་ཟླ་བས་མཛད་པའི་བདག་མེད་མའི་སྒྲུབ་་ཐབས་རྫོགས་སོ། །n། །།མིང་སྦྱོ་ཥ་ཎི་ནང་ན་ས་ཁ་ལ་འོ་ཆེ་ལ་འོ་པ། རྒྱ་གར་གྱི་མཁན་པོ་ག་ཡ་དྷ་ར་དང་། ལོ་ཙཱ་བ་དགེ་སློང་ཤཱཀྱ་ཡེ་ཤེས་་་་ཀྱིས་བསྒྱུར་བའོ། །་ཁ་། ཀ་ལྔ་པོ་དེ་ག་ཁོ་ཤི་ཀ་ཁ་བ་ལོ་ཀ་ཁ་དབང་གཙོ་ཀ་བ་ཀ་རྒྱུ་འདུག་པ་ལ་ངོ་། །ཐ་ཚོང་བདེ་ང་ཚོ་ཐོག་ལ་ཡིབ་དུ་གཟི་འོ་ཏཁྲོ་བ་མ་གཏོགས་པ་དང་། བོ། །pཚོལ་བསྡུར་མཆན། །ཁ་ཚོད་ཏོག་ཏོག་འབར་གས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་བདག་མེད་མ་ཞེས་བྱ་བའི་སྒྲུབ་པའི་ཐབས་བཞུགས། «པེ་»«སྣར་»དཀྱིལ་འཁོར། །@«པེ་»«སྣར་བརྡུངས། ཡང་ཨེ«པེ་»«སྣར་—ཁ་དོག་ནག་མོ་དྲག་ཆེན་མོ། །«པེ་»སྣར་ལོའི། «པེ་»«སྣར་»ཛཀྵ། «པེ་»«སྣར་»ལཾ། «པེ་»«སྣར་»དྷུ་པ། «པེ་《སྣར་»ས༴། །ཉིནཔ་བྱེ་བ་རྗེ་ཡི། །ཕྱི་བ། པའོ། །ཀོ་ས་ལ་་བ་ཁལ་ཁའི་ལྔ་ལ་མ་འ། །ར་ལ་ས་བ་ཚོར་ཡང་། །འ་ན་ཏག་བ་སོགས་ཀྱི་ཚོགས་ཆེ། །::ཀ་འཚོ་བར་འགན་ཁཚོ། །འ་བ། ནད་ཀས་ཏང་བ་ལ་ཚོགས་ཏེ། །ས་བརྒྱད་དེ་ས་ཐོག་ལ་ཁང་ལ་རྗེ་འཚོལ་བ། །བ། ་་ར་ཎ་ཏ་འབུ་རྐོ་བ་དག་ལ་ཁ་ཏའ་བའི་དེ་ལས་ང་ལ་དེབ། །དེ་ཡིན་ཏེ། ཀྐཎོ། །མ་པང་ལ། ཎ་། །
