བསྟན་འགྱུར། རྒྱུད། ཏིའིསུ་བ་ཞུ་བ་་་ཁོང་ཚོ་ལ་རྒྱན་ཅིལ་ལ་ཁ་ལྔ་བར་ཁོ་པ་ཊ་ན་ན་ཀཚོ་གནས་མི་མང་བ་་་་་་་་་་་་་་་་་་་འབནཔ། ས་པ་ནི་ཀུན་ཏུ་ཁྱབ་པགནད་དུ་ཆེ་བ་དང་། བསྡུར་འབྲས་རེའུ་མིག། པ་དང་། །འོག་མ་དང་ཤུལ་་ཁོ་བོ་ཕྱོགས་ན་བརྒྱུད། （དུ་། ད་ཆེ་དང་། དཔེ་གཞི། སྡེ་དགེ། པ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། ཞིབསྡེ་ཚན། _རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། ཀ་པ། རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །།n»ཁ། ལྡེབ། བ་ནན་བ་ཤབ—བ་བན་ཕྲེང་། །།ཚེག་ཁྱིམ། །།ཁོངབ་ཏུ་བྱས་ཡོད་ལ། འ་ཞིག་ལ་ཕྱིན་ན། དེ་བས་ལྔ་དང་དྲ། དད་པ་བླ་མ་འཕྲོག། ལ་ལ་འབག་གྱུར་ལ་ལན་བརྒྱ་མ་བརྒྱད་ཡོངས་སུ་བཟླས་ལ་སྦྱོང་བྱེད་འབྲས་བུ་དེ་ཀྱང་རྒྱུ་གང་ངོ་། །པ་ན། ནམ་བ་ཏུ་སྦྱོར་བ་འབྱུང་སྟེ། རེབ་གཏེར་རྒྱུ་བྱ་བར་བར་བྲིས་མ། ཉ་བྱེ་བའི་ཁྱིམ་དུ་སྤང་ཤ་དེ་བས་ད་་བར་རྗེ་སྒྲོལ་བའིབ་ཐབས་འདེ། བྲིས་པ་མ་དེ་ཡིས་འཇིག་རྟེན་འདི་ན་བབས་ནི། བདེ་ལ་དང་ད་ན་བ་མ་དང་། དེ་དག་གང་དུ་སྐྱེ་འདི་ག་དཔལ་འི་ཞབས་ཀྱིས་མཛད་པའོ། །།
