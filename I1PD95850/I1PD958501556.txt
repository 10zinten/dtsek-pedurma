བ་ལུ་དེ་ཀ། མདུན་གྱི་འདབ་མ་བརྒྱ་བྱིན་བངས། །གཡས་སུ་པའི་དབོད་པ་བྱའོ། །ཆུ་ལྷ་ཁྱབ་ཀྱི་འདབ་མ་ལ། དེ་ལྟར་བ་ལ་་་་པོ་ད། ཞེ་ཡི་བསྡུར་མཆན། །བདེན་བྲལ་དུ་ད་ཡོད། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»གཏོར་མའི་ལས་ཀྱི་་རིམ་པ་བཞུགས། །«པེ་»«སྣར་བཞགཉི«པེ་»«སྣར་བནྡྷ། Q«པེ་»«སྣར་བར་བཞག«པེ་»«སྣར་ཧཱུྃ་ཨོཾ་ཛི། «པེ་»«སྣར་ཀྱི་འབུད། _@«པེ་»འགྲོས། «སྣར་བཀྲོད། «པེ་»«སྣར་»—ཧི། «པེ་»«སྣར་བཀྲོས། མི«པེ་»«སྣར་»ཀྱིས། ་པོ«པེ་»«སྣར་བྷི། ༄༅«པེ་»«སྣར་བྷེ། «པེ་»ཚེ«པེ་»«སྣར་པུར་ནཾ། «པེ་»«སྣར་ལི་ཧཱུཾ་ཚ། «པེ་»«སྣར་ཐེབྦཾ། &«པེ་»«སྣར་»ཡ།  ེ«པེ་»«སྣར་སུཛྫ། «པེ་ཨཥྛ། «སྣར་ཨཊྛ། མི«པེ་»«སྣར་སི། «པེ་»«སྣར་»—ན། «པེ་《སྣར་+ར། «པེ་»«སྣར་ཡེ། «པེ་»«སྣར་»དེ་པདྨེ། «པེ་»«སྣར་»ཏི། «པེ་»དེབད་《སྣར་སེཏྟ། «པེ་»«སྣར་»ཙ། »@«པེ་»«སྣར་»ཡཉྩ། «པེ་»«སྣར་སརྦ་ས་ཏཱ་ར་ཐ། སི་ཏི་དུ་ཏ་ཡ་ཐཱ་ནུ་ག་ག་ཛ་ཏ། པུ་ཏཾ་བི་ས་ཡེ། «པེ་»«སྣར་ན་མ་དུ། ཨེ«པེ་»«སྣར་འཕྱོང་ཞིང་འབུར། །::ཀྱི་འབཀའ་ལན་དེ་ཚོ་ཁྲི་ལ་ཁ་བཏང་མི་དང་བྱེད་ན་ཁ་བ་བསྡུར་འབྲས་རེའ་མིག་འབ:ཏོལས་ད་ཚོ། ང་ག་ལ་བརྒྱུད། (པེ་ཅིན། སྣར་ཐང་། །ཅོ་ནེ། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། ༽དཔེ་གཞི། སྡེ་དགེ། །།སྡེ་ཚན། རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། ན་—བ་ཕྲེང་། ཚེག་ཁྱིམ། T། །རང་ནབ་། ན་ན། ན་_བ་། །
