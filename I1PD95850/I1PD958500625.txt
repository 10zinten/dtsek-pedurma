ཚོད་ཡེངས་བསྒོམ་ལ་བརྩོན་ཞིང་རིམ་པར་སྟོན། །དེ་ལྟ་བུ་ལ་སྔགས་ཀྱི་དེ་ཉིད་ལེགས་པར་བསྟན་པར་བྱ། །ཡེ་ཤེས་ཀྱི་དེ་ཉིད་ཀྱི་རིམ་པ་སྟེ་གཉིས་པའོ། །ཌ་བའོ། །ਲਈਸਸਸਫ਼ਾਰਮਰਸਿੰਘਨਾਰੰਗਲਾਦਾཏེ། དེ་ནས་སྔགས་ཀྱི་མཚན་ཉིད་བཤད། །རྣལ་འབྱོར་པ་རྣམས་རྟག་ཏུ་བཟླ། །སྔགས་དང་ཡི་གེ་ཐ་མི་དད། །གཉིས་མེད་རྟག་ཏུ་བཟླ་བ་ནི། །།གལ་ཏེ་བཟླ་་་འདོད་རྣལ་འབྱོར་པས། །སྙིང་གིས་བསམ་དང་བསྒོམ་པ་སྟེ། །རྡོ་རྗེ་རྒྱུ་གྱུར་་ས་བོན་ནི། །།དེ་ཁོ་ན་ནི་གཟུགས་མེད་དོ། །རང་གི་སྙིང་གར་ས་བོན་ལ། །བསམ་གཏན་སྔགས་ཀྱིས་བསྐོར་ལ་བཟླ། །།།ཨོ་དེ་ཝ་པི་ཙུ་བཛྲ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། སྙིང་པོའོ། །ཨོཾ་ཨཏྟཱ་ཅིན་ནཱ་ཡ། པི་ག་ཨུདྷ་ཀེ་ཤ་བནྡྷ་ནེ། ཙ་ཏུཝཾ་ཤ་ཏི་ནེ་ཏཱ་ཡ། །ཁོ་ཌ་ཤ་བྷུ་ཛཱ་ཡ། ཀྲྀཥྛ་ཛཱི་མཱུ་ཏ་ཝ་པུ་ཥ། ཀ་པཱ་ལ་མཱ་ལཱ་ནེ་:ཀཱ་དྷཱ་རི་ཎི། ཨདྷཱཏ་ཀླུ་ར་ཙིཏྟཱ་ཡ། །ཨདྷེནྡྲ་དོཙྪཱེཎེ། །ཨོཾ་མཱ་ར་ཡ་མཱ་ར་ཡ། ཀཱ་ར་ཡ་ཀཱ་ར་ཡ། །ཞི་་:གཛྫ་ཡ་ག་ཡ། ཏཛྫ་ཡ་ཏཛྫ་ཨིཡ། ཤོ་ཥ་ཡ་ཤོ་ཥ་ཡ། །སཔ་སཱ་ག་རཱ་ན། བནྡྷ་བམ། ནཱ་ག་ཨཊྚ་ཀཱན། ་་:གྲོསྣ་གྲོསྣེ། སརྦ་དུཏྟཾ་ཤ་ཧཱུཾ། ཧ་ཧཱ་ཧི་ཧཱི་ཧུ་ཧུ་ཧེ་ཧི་ཧོ་ཧོ་ཧཾ་ཧཕཊ་སྭཱ་ཧཱ། རྩ་བའི་སྔགས་སོ། །ཨོ་བཛྲ་གླུ་ཛུ་སི་ཏ་པ་ར་མ་:ལོ་ཀེ་ཤྭ་རེ་ཀ་པཱ་ལ། མཱ་ལ་དྷ་རི་ཎི་རུ་ེདྷི་ར་བྲི་ཡེ་་ན་པ་སི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། གོ་རིའི་སྙིང་པོའོ། །ༀ་བཛྲ་ཙཎེ་ཤཱ་རི་ཁ་ཊཾ་གི་མ་ཧཱ་བཛྲ་ཀ་པཱ་ལ་མཱ་ལ་ཀུ་ཧོ། ཨཱ་ཀཟླུ། སརྦ་དུཏྟཾ་ཧྲི་ད་ཡ། ཨཱ་ཀཟླུ་ྋབྷུ་ཧཱུྃ་་་་ཕཊ་སྭཱ་ཧཱ། ཙོ་རིའོ༄༅། །ཨོཾ་བཛྲ་བེཏྟཱ་ལི་ཁ་ཁ་ཁཱ་ཧཱི་སརྦ་དུབྦཾ་བི་ཀ་ཊ། བེ་ཤེ་ཏ་རི་ནི་བི་:ཀྲྀ་ཏ་ཨཱ་ལཾ་ཀཱ་ར་བྷུ་ཥི་ཏ། ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། པི་ཨེལཾ་བ་ས་མ་ནུ་སྨྲ་ཙེར། པྲ་ཤ་ཡ་མཎྜ་ཚེལ་མ་ཏེ། ཨུ་ཏ་པ་ཡ་སརྦ་ཧཱུཾ་ཧཱུཾ་ཧཱུྃ་སྭཱ་ཧཱ། བེཏྟཱ་ལཱིའོ༄༅། །ཨོཾ་བཛྲ་མ་ཧེ་ཤྭ་རི་ཧཾ་ཧཾ་ཧཾ་ཧཾ་རུ་ལུ་སྦྱོ་ཧཱུྃ་ཕཊ། #་པ་ཁྲ་ཡཾ་_།
