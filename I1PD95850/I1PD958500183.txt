བསྟན་འགྱུར། རྒྱུད། f/འཁོར་གྱི་ས་བ་བྱུང་གིས་བྱུགས་ཤིང་ཡི་གེ་མུའི་མཐའ་ཅན་གྱི་ཆོ་གས་:བསང་་་གཏོར་བྱས་ལ་གཏོར་མ་སྦྱིན་པར་བྱའོ། ཞ། གཤེགས་སུ་གསོལ་བའི་དེ་ཁོ་ན་ཉིད་དེ་བཅུ་པའོ། །།།འཁོར་བའོ། །ཏང་ལ་ཏུ་དེ་ཉིད་བཅུ་པ་ངེས་བྱས་ལས། །བདག་གིས་བསོད་ནམས་གང་བསགས་པ། །།དེས་ནི་འགྲོ་བ་མི་ཤེས་པའི། །རབ་རིབ་རྣམས་ནི་ཞི་འགྱུར་ཤོག། སློབ་དཔོན་ཌོམྦི་པས་མཛད་པའི་དེ་ཁོ་ན་ཉིད་བཅུ་པ་རྫོགས་སོ། །༄༅། །གང་དེ་ཚོ་ཀློག་དེའི་རྒྱ་གར་གྱི་མཁན་པོ་དཔལ་མར་མེ་མཛད་ཡེ་ཤེས་ཀྱི་ཞལ་སྔ་ནས་དང༌། །ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་་་ཕབ་པའོ། །།།།ཁོ་མི་བྱ་བ་བྱེད་ཀྱི་ངpབ་ངན་ཁ་ཎ་། །།ཀ་ཏ་ཌ་ཎ་ཀོ་ལྔ། བར་འ་ཤེས་ཀུན་བསྡུར་མཆན། དང་དང་ྋམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དེ་ཁོ་ན་ཉིད་་་་ནས་ས་ག་ཟླ:བཅུ་པ་བཞུགས། «པེ་《སྣར་»ཧཱུྃ། «པེ་»«སྣར་»ཀྱི། །འདུ་བྱེད་པ། ཀུན་«པེ་»«སྣར་ནག་པོ། «པེ་»«སྣར་»+གཤིན་རྗེ་གཤེད་དང་། ཤེས་རབ་མཐར་བྱེད་དང་། པདྨ་མཐར་བྱེད་དང་། བགེགས་མཐར་བྱེད། །མི«པེ་»«སྣར་»མི་བསྐྱོད་པ་དང་། རྣམ་པར་སྣང་མཛད། མ«པེ་»«སྣར་ཡིན་ནོ། །གཏོར་བ་། «པེ་»«སྣར་—ཁྲོ་བོ་རང་འདྲ་བའི། «པེ་»སྣར་བས། Q«པེ་»«སྣར་ཤ། «པེ་»«སྣར་བྱའོ། །«པེ་»«སྣར་»—གཞན་ལ་དད་པ་མེད། «པེ་»«སྣར་»བཞི་པ། ༄༅«པེ་»«སྣར་། «པེ་»«སྣར་+ཆེན་པོ། །(«པེ་»«སྣར་+མདོག@«པེ་»«སྣར་—དང་། རོལ་མོ། ཚེ«པེ་»«སྣར་ཡ་མཎྜ་ལ་ཡ། «པེ་»«སྣར་»སིདྡྷི་ཧཱུཾ། «པེ་»«སྣར་»བཛཱི་བྷ་བ། ༄༅«པེ་»སྣར་»ད། «པེ་»«སྣར་»ཀྱིས། «པེ་»བཤག་:«པེ་»«སྣར་+ཉིད་ཀྱི་གསང་། «པེ་«སྣར་»དྲན་པས་རང་གི་རང་གི་ལྷའི། ཚེ«པེ་»《སྣར་»པ་མི། /«པེ་»«སྣར་»རཱིའི། «པེ་»«སྣར་ཁབ་མེའི། །།
