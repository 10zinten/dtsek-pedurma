རར་ཤེ་པ་ཀྲེ་ཐེ་ཏྲེ་ཤུ་རྒྱ་:ག་རེ་བི་ཤི་ཥཏ། བྷཱ་ཛ་ནེ་སྦྱ་ལག་ཏེ། མཱ་ཏ་དྷ་ན་བི་ཤེ་ཥ་ཏཀྲྀཥྛ་:རུ་དྲ་དེ་བ་དཏྟེ། ས་མཱ་ཤྲཱི་ཏ་ཀྲྀཥྛ་ཨཱ་ཀ་རཱ་ལ་བྷི་བཱ་ཙ་ནནྡ་ཏཱི་ཏ་བི་ནཱ་ཡ་ཀ་། ཙ་མུཎྜ་གོ་རཱི་བི་བྷཙྪ། ཨུ་མ་དེ་བཱི་ཙ་མ་ཏཱ་ར། ཛ་ཡཱན་ཙ། བི་ཛ་ཡཱ་ཎ་ཙེ་བ། ཌཨ་ཛི་ཏ། ཨ་པ་རཱ་ཛི་ཏཱ། བྷ་དྲ་ཀཱ་ལེ་མ་ཧཱ་ཀཱ་ལི་སྒྱུ་ལ་ཀཱ་ལི་ཏུ་ཛོ་གི་ནཱི་ནྡྲི། ཙནྡྲི་གོ་རཱི། དུཏྟཱ་ལཾ་བ་ཀི། ཏྲི་ད་ཤི་ཤྭ་རི་ཀཾ་བོ་ཙིད་ཀྱི་པཱི་ཙུ་པི་རྩྭ་གྲྭ་མཱ་ཝསྦི་ཏ་ཡོ་གི་ནི་:གྷོ་ར་ེརཱུ་པི། མ་ཧཱ་རཱུ་པི། :དཾ་ཥཱ་རཱུ་པིའོ། ཀ་པཱ་ལི་མཱ་ལཱ་མཱ་ལི་རྒྱ་ཁ་ཌཇི་ན། ཌ་མ་ཧཱ་དྷི་ཀཏ་། ཁངྒ་ྋཔ་ར་ཤུ་ཧ་ཏ་ཡ། བཛྲ་ཧསྟ། དྷ་ནུས་ཐཱ། མ་ཧཱ་སུ་ཁ་མ་ཧཱ་བཛྲ་སཉྫ་ཀརྨ་མཱ་ནུ་སཱ་དྷ་ཀ། ཡོ་་ག་མཎྜལ་མ་ཧཱ་རཱ་ཛཱ། ་བཛཱ་དྷ་ཨེཔ་ཏི་བྲ་བྷུ་ཐ། ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་:ལ་ལ་ནི་རཉྫ་ཡོ་གི་'ཁྲི་ཀཱ། ཨི་དཾ་མ་ཧཱ་སུ་ཁ་ཨ་ཛྷེ་ནཱ། ཨ་བ་ཧ་ནཾ་སཊྚ་སརྦ་ཎི། ཨོ་་ཀཀྐ་ལྷ་ན། བབྦནྡྷ་ན། ཁཁཱ་ནི། སཏྠ་དུཏྟཱ་ཎཱི་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཛིསཱ་ཧཱ། རྡོ་རྗེའི་སྟབས་དང་སྔོན་ཕྱག་རྒྱར། །།བཅས་པའི་སྔགས་ཀྱིས་སྤྱན་དྲང་སྟེ། །ཁམས་གསུམ་རྣམ་རྒྱལ་བཅིངས་ནས་ནི། །མཛུབ་གཉིས་ལུ་གུ་རྒྱུད་བྱས་ལ། །ཨོ་བྲྀཏྟི་ཧཱུཾ་ཧཱ་བྷཀྵ་བྷཀྵ་ཕཊ་སྭཱ་ཧཱ། སྔགས་དང་ཕྱག་རྒྱ་འདྲི་ཨམཉེས་བྱ། །ང་། སེ་་གོལ་གསུམ་གྱིས་འདིས་མཉེས་བྱ། །ཙནྡར་ཅིདམར་པོའི་ཆུ་ཡིས་ནི། །ཕྱི་རོལ་དཀྱིལ་འཁོར་གྲུ་གསུམ་བྱ། །དེར་ནི་མེ་ཏོག་དམར་པོ་དགྲམ། །ཉ་སྐམ་ཤ་་་་སྐམ་སྒོག་པ་ཁྲག། ཏིལ་ཕྱེ་བག་ཕྱེ་འབྲས་སྲན་ཆནི། །ཁྱད་པར་བདུད་རྩི་ལྔ་སོགས་དང་། །རྣམ་པ་སྣ་ཚོགས་བཟའ་བ་རྣམས། །ལྷོ་སྒོ་བ་ཡི་གཏོར་མ་ཡིན། །ཨོཾ་མ་ཎི་བྷ་དྲ། པུཎྜ་བྷ་དྲ། སུ་བྷ་དྲ། །ཛ་ཡ་བྷ་དྲ། ནནྡ་ཀཱི་ལ། མ་ཧཱ་ཀི་ལ། ཌསུ་ཀཱི་ལ། པདྨ་ཤི་ཀ། ཤཾ་ཁ་ཤི་ཀ་ཨེན་ཊ། མ་ཧཱ་ན་ཊ། བྷེ་ད་ནཱི། ་ཙྪེ་ད་ནི། པུ་ར་ནི། དྷ་ན་དྷ་སུ་དྷ་ན། སུ་རཱུ་པ། ཀེ་ལི་མ་ཧཱ་ཀེ་།
