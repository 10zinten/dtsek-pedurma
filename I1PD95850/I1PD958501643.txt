ཁ་བ་བསྟན་འགྱུར། །རྒྱུད། ཏཌ་་་་་ང་ཚོས་བོན་གདབ་བྱན་། ཁཀབའི་རྒྱུ་དཀའ་ཞིང་བསྐྱོད་་་E་ཀྱི་བཀར་གྱི་ཁང་ཡང་ན་ཡི་ག་་ཁཞ་ཏོ་རྗེའི། །བོའུ་ང་བ། བ་རི་བ་བསྡུར་འབྲས་རེའུ་མིགབཀའ་རྒྱུད་པ་ཚོལ་བ། ཞི་བ་ལ་འ་ལྔ་ཡི་གའི་རྒྱུད། （་་་་་་་་ཁབ་ལྔ་པ། པེ་ཅིན། །ཅོ་ནེ། །སྣར་ཐང་། རྒྱུད། རྒྱུད་འགྲེལ། ད་རྒྱུད། དཔེ་གཞི། སྡེ་དགེ། སྡེ་ཚན། རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན་བ་ཕྲེང་། ཚེག་ཁྱིམ། །ཤ་། ་་། བ་_བ། བབན?བ་བར་ལ་། བར། །།།པ་དུ་ཞུགས་པར་བལྟའོ། བདག་ཉིདམས་ཀྱི་ཕུང་པོ་ལ་སོགས་མང་དེ་དག་ནི་མེད། ཁ་བ་དང་བཞི་བཞི་དག་ནས་མི་ཚད་དུ་ཕྱིན་པ། །རྟག་ཏུ་གྱུར་པ་ལ་ཤིན་ཏུ་མང་བ་མར་དམ་པར་ལ་དེ་ལ་རང་གི་བུའི་ང་རྒྱལ་དུ་འབད་ན་ང་དེཀྱི་ལས་ལ་ཞེས་དང་་། །ར་རུ་འི། ་དག་ལ་ལ་ཡེ་ཤེས་ཀྱི་ངང་པ་ཞིག་དང་། ད་དང་དོན་ཀུན་འད། ད་དངོས་གྲུབ་རྗེས་སྐྱོབ། །ནས་དངས་བླངས་བ། ཤེར། །མ་གསལ་ཏེ། དེ་ན་ཡང་། ཤེ་ལག་པས་འབྱེད་ལ་། ཞེས་བརྗོད་པས་ལུ་གུ་ཞེས་བྱ་བས་ཀྱང་བསྐུལ་ལོ། །དེ་ནས། དམ་ཚིག་—།
