རྣམས། །བསྐུར་བ་བཅས་པས་བྱ་བ་ཡིན། །བཟའ་བ་ཕུད་དབུལ་བདུད་རྩི་མྱང་། །སྤྱོད་ལམ་རྣམ་བཞིར་ཅི་སྤྱོད་པ། །དེ་ཉིད་བཞི་ཡི་རིམ་པས་སྤྱད། །རྒྱུད་དང་བླ་མའི་མན་ངག་ལ་བརྟེན་ནས། །དགྱེས་རྡོར་རིམ་པ་བཞི་ཡི་སྒྲུབ་་་་ཐབས་བྱས། །འདི་ལས་དྲི་མེད་དགེ་བ་གང་བསགས་པས། །འགྲོ་ཀུན་དགྱེས་པའི་རྡོ་རྗེ་འགྲུབ་པར་ཤོག། སློབ་དཔོན་ཀརྨ་ལས་མཛད་པའི་དགྱེས་པ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་དེ་ཁོ་ན་ཉིད་བཞིའི་རིམ་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །།།པཎྜི་ཏ་བི་རོ་ཙ་ན་རཀྵི་ཏ་དང་ལོ་ཙཱ་བ་གླན་:དར་མ་བློ་གྲོས་ཀྱིས་བསྒྱུར་་་་ཞིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །《་:ཁོ་ར་། བོ་ད་ར་《་ཀ་ལ་་ད་བྱེད་ཀྱི་ཆུ། ཁོ་ཚོ་དང་མི་པོའི《། 《་ཀ《་། །ངོ་ཏིང་ག་བསྡུར་མཆན། །ངོ་བོ་ད་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བཅོམ་ལྡན་འདས་དགྱེས་པའི་རྡོ་རྗེ་སྒྲུབ་པའི་ཐབས་བཞུགས། «པེ་»«སྣར་ནས། 《་ཁ《《ཟླ་བ། ཀྱི། འི«པེ་»«སྣར་པ་ཡིས། །ཚོགས་ཀྱི། «པེ་»«སྣར་རཉྫ། «པེ་»«སྣར་»ལ་ཏི། «པེ་»«སྣར་»ཡག། «པེ་»«སྣར་རལ་གྲི་གཡས་ན་ཕུབ་ནི། །།@«པེ་»«སྣར་»གཡས་པ་བོ་དབྱུག་པ་དག། ®«པེ་»yn་ལ་《སྣར་»སྒྲུབ། «པེ་»སྣར་མཚན། །@«པེ་»ཏུ། 《སྣར་གྱུར། »བྱ་ཆ་ཆུ་ལྟ་«པེ་»《སྣར་ཤོར། «པེ་»«སྣར་བཅོ་ལྔའི། «པེ་»«སྣར་»དཏྟ་ཙ་ཏུརཱ། «པེ་»«སྣར་»དུ། @«པེ་《སྣར་ནི། »«པེ་»«སྣར་བཞི། ཏ་ནི་ཁ། །ཞི་ན་ཚེའི་ཁ་་«པེ་»«སྣར་»གདན། «པེ་»«སྣར་»ན། ཚེ«པེ་»«སྣར་»གྱི། 《༌དོ་«པེ་»«སྣར་འཚོག། གང། ཀ་ཊ་ཀྱི་སྒྲ་དང་ཤི་«པེ་»«སྣར་པ། «པེ་»«སྣར་»ཀྱི། །་་་དྲན་ཤེས་ཀྱི་ཚོ་པ་:ར་ཁ་«པེ་»«སྣར་སྐ། །ཚེ«པེ་»«སྣར་»ཏེ། ཁའི་ཀ་ཡ་་ཊཾ།
