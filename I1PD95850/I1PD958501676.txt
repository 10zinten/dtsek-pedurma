«པེ་»«སྣར་གཏེར། །«པེ་»«སྣར་དཔྱལ། ལས་བྱས་པ་ལ་བ་བྲལ་བས་བཀང་བ་བ་བ་བ་བབས་པ། ཁ་བ་བབས་བྱ། བསྡུར་འབྲས་རའི་མག་རྒྱུད། (ཙོ་། རྒྱུད། །།དཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། པོད་རྟགས། ཚན་གྲངས། །ཆོ་ག། །ལྡེབ། བབ་། ན་—ན། ན་ནབ་བ་ཕྲེང་། །ཚེག་ཁྱིམ། །བའཊ་ཀླུཊ། ང་ཚོ་བཀོད་བྱས་པ་ལ་་་་་བཀོད་ཀྱི་བྱ་བ་བ:ཁ་སང་ང་ལྟ་བ་ཕྱོགས་ཀྱི་བཀྲ་རྒྱུ་བ་གྲཊིའི་ཟླ་བ་དྲན་ཚོར། ཚོ་ལྔ་དང་། ཟླ་བ་གྲོ་ཤལ་ཁབ་ལྔ་པ་དང་གའོ་ཋཀ་ཁ་ག་ཆ་དང་ཁང་ལ་དྲི་བ། ངའ་བ་གྱུར་བཅུ་འཁོར་བ་ན་དུག་མ་ཡོང་བ་ཞི་ཁྲོ་བས་ངངཁ་ན་ཞིག་གང་དུ་འཚོགས་་་་་་་ཁོ་ནས་ལྷ་གནས་བོན་ཀྱི་བ་ལ་ཁ་འབབ་ལ་དབང་བ་ལ་སོ། །
