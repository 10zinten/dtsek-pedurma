རྒྱུད། བཞིས་བསྐོར། །ནང་དུའང་སོར་བཞི་དོར་བར་བྱ། །མངོན་སྤྱོད་ལ་ནི་སོར་་་གསུམ་སྟེ། #། ནང་དུའང་སོར་གསུམ་དོར་བར་བྱ། །དགུག་དང་བསྐྲད་ལའང་དེ་བཞིན་ནོ། །ཞི་བའི་ཐབ་ཁུང་དབུས་སུ་ནི། །རྡོ་རྗེ་དང་ནི་གྲི་གུག་བྲི། །ཁ་ཁྱེར་ལ་ནི་རྡོ་རྗེས་བསྐོར། །རྒྱས་ལ་དབུས་སུ་རིན་པོ་ཆེའོ། །@་། ཁ་ཁྱེར་ལའང་རིན་པོ་ཆེའོ། །།དབང་ལ་དབུས་སུ་པདྨས་མཚན། ཌ་་། ཁ་ཁྱེར་དེ་དང་འདྲ་བས་་བསྐོར། །མངོན་སྤྱོད་ལ་ནི་རལ་གྲིའོ། །ཁ་ཁྱེར་ལ་ཡང་རལ་གྲིས་བསྐོར། །དགུག་ལ་དབུས་སུ་ལྕགས་:ཀྱུ་བཞག། ཁ་ཁྱེར་དེ་དང་འདྲ་བས་བསྐོར། །བསྐྲད་ལ་དབུས་སུ་རྡོ་རྗེ་ནི། །རྩེ་གསུམ་པ་ནི་གཞག་པར་བྱའོ། །ཁ་ཁྱེར་ལ་་་ཡང་རྩེ་གསུམ་བྲི། །ཁ་། དབུས་སུ་གྲི་གུག་ཀུན་ལ་མཐུན། །ྋ། ཐབ་ཁུང་ནང་གི་་ཕྱོགས་བཞིར་ནི། །།ཁ་ཊཾ་ག་དང་བདེ་གཤེགས་ཀྱིས། ཏི། །ཕྱག་མཚན་ཕྱི་མ་་་རིམ་བཞིན་གཞག། ཐབ་ཁུང་ཀུན་ལ་ཁ་དོག་དང་། །མཐུན་པའི་དྲི་ཡིས་བྱུག་པར་བྱ། །ཞི་བ་ཕལ་ཆེར་སྲོད་ལ་བྱ། །རྒྱས་པ་སྔ་དྲོ་བྱ་བ་ཡིན། །དབང་ནི་དགོངས་ཀའི་དུས་སུ་བྱ། །དགུག་པ་ལ་ཡང་དེ་དང་འདྲ། །མངོན་སྤྱོད་་་་ནམ་ཕྱེད་ཉིན་ཕྱེད་བྱ། །བསྐྲད་པའང་དེ་དང་འདྲ་བ་ཡིན། །ཞི་བ་ཤར་དུ་བྱ་བར་བཤད། །ཨ་། རྒྱས་པ་ལྷོ་ཡི་ཕྱོགས་སུ་བྱ། །དབང་གི་ལས་ནི་ནུབ་ཏུ་བྱ། །དགུག་ལའང་ནུབ་ཏུ་བྱ་བ་ཡིན། །མངོན་སྤྱོད་བྱང་དུ་བརྩམ་པར་བྱ། །བསྐྲད་པའི་ལས་ཀྱང་དེ་དང་འདྲ། །ོ། ཚོན་གྱིས་རྗེས་མཐུན་བྲི་བ་འམ། །ཡང་ན་མེ་་་ཏོག་དག་ཀྱང་རུང་། །ཐབ་ཁུང་དག་ནི་དེ་ལྟར་བྱ། ཀ་ཁ།  ། ག་ང་ད་ན་བ་ན། །དོ་ཁོ་་་གང་། །ཏོg་ཀྱང་ཁོ་ཥཎ་ཊི་ད་ཀ་ཁཎྜ། །དུ་ངཀོ༌ཎཌ་ཎི་བ་ཀ་ཏ་ཁ་ཚེ་རྫས་ནི་འདི་ལྟར་ཤེས་པར་བྱ། །ཞི་བ་ལ་ནི་ཏིལ་གྱིས་སོ། །རྒྱས་པ་ལ་་ནི་ཞོ་ཟན་ནོ། །དབང་དང་དགུག་ལ་ཨུཏྤ་ལ། །མངོན་སྤྱོད་ལ་ནི་ཚེར་མས་སོ། །—།
