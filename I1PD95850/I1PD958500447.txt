རྒྱུད། ་་་ཁཡོདprཊྛི་བསྐོར་སྟབས་སེང་གེའི་མཆོང་སྟབས་སོ། །ཨོཾ་ཧཱུཾ་ཨཱཕཊ། ལྷོ་སྒོའི་འོ། །གཡས་བརྐྱང་བ་དང་གཡོན་བརྐྱང་དང་། །ས་ག་ཟླུམ་པོའི་རྐང་སྟབས་སོ། ཁོ། ཨོཾ་ཧཱུྃ་ཨོཾ་ཨཱ་ཧཱུཾ་ཕཊ། ནུབ་སྒོའི་འོ། །མཐུན་པའི་སྟབས་དང་མི་མཐུན་སྟབས། །གཅིག་བསྡུས་གཡོན་བརྐྱང་རྐང་སྟབས་སོ། །ཨོཾ་ཧ་ན་མཱ་ར་དུབྷཱུན། གྷ་ཏ་ཡ་ག་ཏ་ཡ་ཨ་ཨ་ཧཱུྃ་ཧཱིཕཊ། བྱང་སྒོའི་སློབ་དཔོན་གྱི་སྟོང་སྟབས་དང་སྔགས་སོ། །འདོད་པའི་དོན་གྲུབ་བྱ་བ་དང་། །བགེགས་ཀྱི་བར་ཆད་ཞི་བྱའི་ཕྱིར། །དྲི། སྔགས་དང་ཕྱག་རྒྱས་ལྷ་ལ་སོགས། །གཡས་བརྐྱང་བཀུག་ལ་གཏོར་མ་སྦྱིན། །རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་བཅིངས་ཏེ། མ། རྒྱབ་དང་རྒྱབ་ཏུ་སྦྱར་བར་བྱ། །ས་་་་གདུགས་དང་ནི་པདྨ་སྦྲེལ། ཁྱུ་། རོ་ཙ་ན་དང་དཀར་པོ་བརྐྱང་། །དཀར་པོ་བཀུག་ནས་འདྲེན་པ་ནི། །ལྷ་དང་ལྷ་མིན་གྲུབ་པ་དགུག། ཨོཾ་ཨཱབཱ་ཧ་ནི་མ་ཧཱ་བཱ་ཧ་ནི་སྲེ་ཏ་བཱ་ཧ་ནི་སཱུརྱ་ཙནྡྲ། སིདྷ་མུ་ཁི་ཛ་ཡ་རཱ་ཛཱི་སོ་མ་རཱ་ཛཱི། ཨུ་མ་དེ་བི་གངྒ་མ། སུ་རཱུ་པཱི་ཏཱ་ར་ཡ། སཾ་གྷ་གྲ་ཧཱ་ན་སཟེ། ག་ཎ་པ་ཏི། ཀཱ་རི་ཏི་ཀ་ཡ་ཤྲཱི་ས་སི་དེ་བི། ཨ་ག་ལོ་པ་ག་ལ། ཨ་སུ་རཱ་སུ་རཱ་ཛ་ཡོ། །པུཀྐ་སི་མ་ཧཱ་པུཀྐ་སི། ཧཱུཾ་ཧོ་ཧོ་ཨཱ་ཀརྟ་ཡ་ཨཱ་ཀརྟ་ཡ། ས་མ་ཡ་དྷི་པེ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཛ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའི་སྔགས་དང་ཕྱག་རྒྱའོ། །ཨོ་བ་ཏ་ལི་:བ་ཏ་ལི་མ་ཧཱ་བ་ཏ་ལི་ཧཱུཾ་ཧཱུཾ་ཧཱུྃ༑་བྷཀྵ་ཡ་བྷཀྵ་ཡ། ས་མ་ཡ་ཏིཥྛ་ཨིཛརྫུ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། སྔགས་འདིས་་་་སྤྱན་དྲང་ཕྱག་རྒྱ་ལས། །མཛུབ་བཅིངས་སེ་གོལ་གསུམ་བརྡབས་གཏང་། །མཉེས་པར་བྱེད་པའི་སྔགས་དང་ཕྱག་རྒྱའོ། །དྲི་ཡི་དཀྱིལ་འཁོར་གྲུ་བཞི་པ། །མེ་ཏོག་དཀར་པོ་དགྲམ་པར་བྱ། །ཤཱ་ལིའི་བཟའ་བ་རྒྱ་ཆེན་པོ། །བུ་རམ་ཞོ་་་དང་འབྲས་ཆན་མར། །མར་མེ་སོགས་མཆོད་རོལ་སྟབས་ཀྱིས། །ཤར་སྒོ་བ་ཡི་གཏོར་མ་སྦྱིན། །ཨོ་ཨེ་ཀ་:བཱི་ཀེ་སྨ་ཤཱ་ནེ་པ། །?སརྦ་ཏེ་ཀནྡ་རེ་གྲོ་ཧེ་གྲྭ་མ་པཱ་ནས_།
