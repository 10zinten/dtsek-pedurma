དཾ་བ་ལིཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཏི། བློ་སྦྱོ་བློ་བྷོ། ཛ་ཛ་ཛ་ཛ། ་བཛྲ་དྷ་རོཏྟཱཛྫཱ་པ་ཡ་་ཏི་སྭཱ་ཧཱ། ་ཨོ་ཨཱཧཱུཾ་ཧོ། སརྦ་དེ་བ། ནཱ་ག་། ཡཀྵ། རཱཀྵ་ས། མ་ནུ་ཥ། ཨ་མ་ནུ་ཥ། ཨ་སུ་ར། གནྟབྷ། :ཀཾ་བྷི་ར། །གྷི་དྷ་རི། ག་རུ་ཌ། །བྷཱུ་ཏ༑པྲ་ཏ། པི་ཤཱ་ཙ། ཨ་པསྨ་ར། ཨུ་མནྟ། ཚ་སརྦ་ལོ་ཀི་ཀ་ལོ་ཀོཏྟ་ར། །བཛྲ་ཌཱ་ཀ་ཌཱ་ཀི་ནི་ས་པ་རི་ཝཱ་རེ་བྷོ། ཨི་མཾ་སརྦ་པཱུ་ཛ་གནྟུ། །པུ། དྷུ་པ། །དཱི་པཾ། ནི་བི་ཧུཾ། ཁཨཀྵ་ཏཾ་ད་ད་མ་ཧི་ཏེ་ཙ། ཨ་གཏྟཱས་པ་རི་ཝཱ་རེ་བྱོ། །ཨི་དཾ་བ་ལིཾ་གྲོཏྟནུ། ཁ་དནུ། པི་བནྟུ། ::ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་རཀྵནྟུ། །མ་མ་སིདྡྷིམྨེ་པྲ་ཡ་ཙྪནྟུ། ཡ་ཐི་པཾ། ཡ་ཐེéབྷུཉྫ་ཧོ། མཱ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ་ཀཅུ་སད་སུ་ཁཾ། བི་ཤུདྡྷ་ཡེ་ས་ཧ་ཀ་ཡི་ཀ། བྷ་བནྟུ། ཧཱུྃ་ཧཱུྃ་ཕཊ། །བཛྲ་དྷ་རོ་ཨཱཛྫཱ་པ་ཡ་ཏི། ཨོཾ་ཨིནྡ། །ཛ་མ། ཛ་ལ། །ཛཀྑ། བྷཱུ་ཏ། བནྡི༑། བ་ཨུ། རཀྵ། །ཙནྡྲི༄༅། སུཛྫ། །མ་ཏེ། པཔྤ། །ཏ་ལ། པཱ་ཏཱ་ལ། ?ཨ་ཋ། སཔྥ། སཱ་དྷ། ཨི་དཾ་བ་ལཾ་བྷུཉྫ། ཛི་གྷཾ། པུཤྩ། དྷུ་པ། །དཱི་པཾ། ཚོམཱ་ས། ཨམྤ། །ཀཱཙྪ། སཱ་དྷ། ་ཁནྟི། ཁུ་ཎི། །ཕེ་ཊ། གཱ་ད། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨ་ཎཾ་ཨདྱ་ནུཏྤནྣ་ཏྭཱཏ། ཨོཾ་ཨཱཿཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་ཚིགས་སུ་བཅད་པ་འདིས་ཡང་བསྔོ་བ་ནི། །ཨོ་ལྷ་ཆེན་བརྒྱད་པོ་འཁོར་བཅས་དང༌། །འཇིག་རྟེན་སྐྱོང་བ་འཁོར་བཅས་དང༌། །ཀླུ་ཆེན་བརྒྱད་པོ་འཁོར་བཅས་དང༌། །གནོད་སྦྱིན་ཚོགས་རྣམས་མ་ལུས་དང་། །སྲིན་པོའི་ཚོགས་རྣམས་མ་ལུས་དང༌། །འབྱུང་པོའི་ཚོགས་རྣམས་མ་ལུས་དང་། །ཡི་དྭགས་ཚོགས་རྣམས་མ་ལུས་དང་། །ཤ་ཟའི་ཚོགས་རྣམས་མ་ལུས་དང་། །སྨྱོ་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང་། །བརྗེད་བྱེད་ཚོགས་རྣམས་མ་ལུས་དང་། །མཁའ་འགྲོའི་ཚོགས་རྣམས་མ་ལུས་དང་། །དེ་ལ་སོགས་པའི་འབྱུང་པོའི་ཚོགས། །།།
