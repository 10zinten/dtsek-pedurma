བ་མ་ཁག་ཏུ་དབྱེ་བ་གང་དུ་ཁ་མ་ཚོ་དག་པ་དང་འདི་གའི་འགོ་ཁ་ལ་ཞོན་དེ་འདྲ་ན་ད་གཟོད། དེ་ཎ་ཊི་ག་ག་ལྔའི་དཀའ་ག། ཤ་འོད་ཟེ། ག་ང་བ་། ལོ་ཁ་དང་བཀའ་སློབ་གནང་བ་ད་ལྟ་རྒྱུ་དེ་རྒྱུན་དུ། དུ་བ། གམ་༄༅། །བདག་མེད་མའི་མན་ངག་ཅེས་བྱ་བ། །དེ་ཊི་ན་ར་འཁྲུག་གང་བྱུང་བ་ལ། ་་་་་་་་་་་བའོ། །གནས་བཅུ་ན་བ། སྐྱེ་བ། །ཀློག་བཤད་ཁོ་ནའི་ཀ་བ། འཊའི་ཡུལ་ཚོ་ལ་འགན་འཁp:ཁ་སྟེལ་བྱེད་པ། བྱེད་པ་པོ། ན་ནང་ཁ་་་་གོངམ་དང་པོ་བང་། ས་བབ་༄༅། །རྒྱ་གར་སྐད་དུ། ནི་རཱཏྨ་ཀོ་པ་དེ་ཤ་ནཱ་མ། བོད་སྐད་དུ། བདག་མེད་མའི་མན་ངག་ཅེས་བྱ་བ། དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །གཡོན་གནས་མིང་གི་བརྐྱང་མ་ཆུ་འབབ་སྟོང་ཉིད་ཤེས་རབ་ཆ། །གཡས་གནས་མིང་གི་རོ་མ་ཁྲག་འབབ་སྙིང་རྗེ་དངོས་པོའི་བདག་ཉིད་ཅན། །དབུས་སུ་གནས་པ་ཀུན་འདར་མ་ནི་ཞུ་བ་འབབ་ཅིང་མི་ཕྱེད་གཟུགས་ཅན་གཉིས་མེད་པ། །དེ་ནི་རྩ་གསུམ་གཟུགས་ཀྱིས་བཅོམ་ལྡན་འདས་མ་ཁྱེད་སྲུངས་ཤིག། བདག་མེད་མས་ནི་ནམ་མཁའི་མཐར་ཐུག་ཁྱབ། །དཔལ་ལྡན་བདག་མེད་མ་ལ་ཕྱག་འཚལ་ནས། །ཡང་དག་བདེ་བའི་གནས་སུ་བདག་གིས་ནི། །བདག་མེད་མ་ཡི་མན་ངག་བཤད་པར་བྱ། །འདིར་བཟླས་པ་དང་བསྒོམ་པའི་དབང་དུ་བྱས་པ་སྙིང་རྗེའི་རོས་བྱུགས་པའི་སེམས་ཀྱི་རྒྱན་ཅན། བྱང་ཆུབ་ཀྱི་སེམས་རིན་པོ་ཆེ་ལེགས་པར་གཟུང་བ། བླ་མ་དང་བདག་མེད་མ་གཉིས་མི་ཕྱེད་པར་བསྙེན་།
