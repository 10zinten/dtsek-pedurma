རྒྱུད། ༼ཨུ༽ལ་འཕྲོག་པ། །སྒྱུར་བ་རྣམ་པ་སྣ་ཚོགས་པ། །བདེ་བ་ཆེན་པོ་གསལ་བར་བྱེད། །སྐུ་གསུམ་སྟོན་ཨཱ་ཀྱེ་ཀྱེ་གཤིན་རྗེ་གཤེད་ནག་པོ། །བླ་མ་སྲིན་པོའི་གཟུགས་ཅན་རང་བཞིན་ཁྱོད། །བདག་གིས་ཁྱོད་མཐོང་ན་ཤིན་ཏུ་འཇིགས་པ་ཡིས། །ཁྲོས་པའི་་་་རང་བཞིན་འདི་མཐོང་®མཛོད། །ས་འོག་ས་སྟེང་ས་བླར་བཅས། །ཁྱོར་གྱིས་གར་གྱིས་གཡོས་པར་མཛོད། །ཨུ་དྷེ་ཤ་ནི་དྷ་ཤ་ཨན་འཛིན་ནག་པོ་བཅག་པ་འདྲ། །ཁྲོ་བོས་རོ་ལངས་བཞིན་གར་མཛོད། །དབྱངས་ཅན་མ་རྣམ་པ་སྣ་ཚོགས་གཟུགས་སྤྲུལ་ཞིང་། །མདངས་ནག་ཚད་ནི་ཐུང་བ་ལ། །།བདེ་ཆེན་ངོ་བོ་ཁྱོད་གར་མཛོད་་་ཙམ། །རྡོ་རྗེ་དབྱངས་ཅན་མ་གསོལ་འདེབས། །ཧཱི་ཙྪི་སྔགས་ཀྱིས་དང་གར་་་་གྱིས་སྒྲས་ཟེར་མཛོད་ནི། །སྲིད་པ་གསུམ་གྱི་འཁྲུལ་པ་ཆོད། །སྙིང་རྗེས་ཁྲོས་ནས་གར་མཛད་པ། །།འགྲོ་བས་མཐོང་ན་ངོ་མཚར་འགྱུར། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་་པུ་ཛྙཱ་བཛྲ་སྭ་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། སྒྲའི་ཚོགས་རྣམས་ཐམས་ཅད་ནི། །བྲག་ཆ་འདྲ་བར་བལྟ་འགོའུ་རིམ་གྱིས་བྱས་ནས། །།དཀྱིལ་འཁོར་པས་ནི་གཙོ་བོ་བསྟོད། །གཙོ་བོ་དེས་ཀྱང་འཁོར་དེ་དག་དེ་བཞིན་ནོ། །གཤིན་རྗེ་གཤེད་པོ་རབ་འཇིགས་པ། །གཏི་མུག་རྡོ་རྗེའི་ངོ་བོ་ཁྱོད། །སྟོན་པ་སངས་རྒྱས་ཀུན་རང་བཞིན། །རྡོ་རྗེ་སྐུ་ལ་ཕྱག་འཚལ་བསྟོད། །གཤིན་རྗེ་གཤེད་པོར་@རབ་འཇོམས་པ། །ཕྲ་མ་རྡོ་་་རྗེའི་ངོ་བོ་ཁྱོད། །རྡོ་རྗེའི་ཐུགས་དང་རབ་མཚུངས་པ། །རྡོ་རྗེ་རིན་ཆེན་ཕྱག་འཚལ་བསྟོད། །འདོད་ཆགས་རྡོ་རྗེའི་ངོ་བོ་ཁྱོད། །འདོད་ཆགས་པད་མའི་ཆོས་འབྱུང་གནས་བརྙེས། །དབྱངས་མཆོག་ཀུན་གྱི་གཙོ་བོའི་གཙོ། །རྡོ་རྗེ་གསུང་ལ་ཕྱག་འཚལ་བསྟོད། །ཕྲག་དོག་རྡོ་རྗེའི་རང་བཞིན་ཁྱོད། །གཤིན་རྗེ་གཤེད་པོ་ལས་ཀུན་པ། །རྡོ་རྗེ་སྐུ་དང་རབ་ཏུ་མཚུངས། །ཕྱག་ན་རལ་གྲི་ཕྱག་འཚལ་་་བསྟོད། །སངས་རྒྱས་ཀུན་གྱི་དངོས་པོ་ཁྱོད། །སངས་རྒྱས་ཐམས་ཅད་གཅིག་།
