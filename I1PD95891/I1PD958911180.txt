རྗེད་རྒྱུན་དུ་བསྒྲུང་། །གནོད་སྦྱིན་མོ་མདུན་དུ་བཀུག་པ་དེ་ལ་སྦྱིན། །སྒྲུབ་ཐབས་འདི་ཉིད་ཐེ་ཚོམ་སྤངས་ནས་སུ། །ཞག་ཉི་ཤུ་རྩ་གཅིག་དེ་ནས་རྒྱུན་ཏུ་སྙན་གྲངས་ཚུལ་བཞིན་བྱས་གྱུར་ན། །ཡིད་བཞིན་ནོར་བུ་རིན་པོ་ཆེ་དང་འདྲ་ཡིད་ལ་ཅི་འདོད་ནོར་བུ་ལྟར། །ཇི་ལྟར་རང་གང་འདོད་པ་དམིགས་པ་ཙམ་གྱིས་འདོད་པ་རྫོགས་པར་འགྱུར། །སློབ་དཔོན་་་་་འཕགས་པ་ཀླུ་གྲུབ་ཀྱིས། །ཤ་ཟ་རིག་པ་འཛིན་པའི་རྒྱུད་ནས་ཕྱུང་ནས་མཛད་པ་རྫོགས་སོ། །།།གང་དུ་སུ་དང་དམར་པོ་དང་། །།།།།ཕྲ་ལ། མ་མ་བདེ་དེ་བྲལ། ་་་་་་་་་་་་་་དེའི་བསྡུར་མཆན། ཡི་ནང་ལོགབབ། །།དེ་མཐུན་འཇུག། ས་སུ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་བཤམས། «པེ་»བསྐུག«པེ་བཞགནས་བན་ཀྱི། །བཀའ་དང་ਸਿੰਮਲਰਲਨੋਂਨੂੰਸੁਲਇਸਲਲਲਲਲਨੂੰਪਲੌਟཁ་འབབ་བསྡུར་འབྲས་རེའུ་མིགagਖརྒྱུད། (ਮਨਪਸਲmਲਾਰਾਮཔ་དང་འགྲོ་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །།།སྡེ་ཚན། །པོད་རྟགས། །འོ་ཞུ་ཚན་གྲངས། ལྡེབ། །མ་་་་ནན་བ་ནང་ན་དང་། གང་ལ་འབད། ཕྲེང་། །།།།།::ཀྱི། ཚིག་རྐང་། །བ། ་པ་ནི། ཚེག་ཁྱིམ། །ས་ག་ཟླ་ཁ་ལོན་ན་ཏན་ཏྲ། །།།བ་ཡོང་ཚུན་དུ་ཁ་དཔལ་སྔགས། སྔགས། པོ་། །
