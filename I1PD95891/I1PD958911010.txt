ལས་ཀྱི་རྒྱུད་དང་གྲོགས་བསྟན་པའི་རྟོག་པ། ་ཡོངས་སུ་མྱ་ངན་ཞུས། འདས་པ་དེ་ནས། དེ་ལྟར་ན་མ་འོངས་པ། འདང་བ་ད་ན་དེ་སྐད་བདག་གིས་ཁྲོ་བ་མ་ལུས་པའི་གླེང་ག་རིམ་པས་སྐྱེ་བ་ད་ར་བཞིན་དུ་བ་དང་བསྡུར་མཆན། ་པ་བདེ་བ་སྟེ་རྒྱུ་བ། ་། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད་ཅིང༌། «པེ་འཁོར་ལོ་ཞེས་བྱ་བ་འཇམ་་་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །ཉི«པེ་»—དདྲུགཚེ«པེ་ཀྱིས། ་ཚེ«པེ་མང་ཟུང་། ཚེ«པེ་གསོད། མི«པེ་»དགྱིས། «པེ་»—དབཞི། ཚུ«པེ་»སྟན། ་ ོ«པེ་»སྨྱོས། «པེ་ཚོམ་གྱིས། «པེ་»གཏམ། @ལེ«པེ་»ཚོམ། «པེ་རིགས། «པེ་»ཙོམ། ་ ི«པེ་སམ། «པེ་»སྔགས། ཉི«པེ་བཀུར། ཚེ«པེ་»hཡིད་་ཆེས་པ་དང་། །::་བ་ལ་O«པེ་»པ། «པེ་གཞིར། «པེ་»བཛྲ། «པེ་+ཨེ་ཝཾ། ནད་འབུལ་བར་བགྲོད་་ང་ལ། དེར་མ་ཕྱེ་བའོ། །བསྐལ་བ་ས་:ས་རྒྱུ། འོན་ཀྱི། ཏུ་nfཞུ་ཁཌོ་ལན་ཊི་''དའཚོ་་་་་་་བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། （དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །ཅོ་ནེ། སྣར་ཐང༌། སྔགས། སྡེ་ཚན། །སྔགས་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ་་་་༼་་་་་་་་ན། བ་བ་ན་ཕྲེང་། ཚིག་རྐང་། །ཚེག་ཁྱིམ། ཆུ་ལས་ས་ང་ཀ་རྒྱུ་དེ་ན་་ཆུ་བཞི་བའི་ཁ་བྱའི་ཤ་ཉེ་ཁ་ད་དོར་སྐུ་ཁྱད་ཆོས་གལ་མ། ན་དང་་ཅགེ་སར། ན་ཚེ། ཁ་།
