།དྲན་པ་་དག་པ་ཀཤེན་བློ་ཀ་ཁ། །ག་ང་ད་ན་ཟླ་བློ་ཁ་དང་ཚོང་ཁང་། ਮੱਖਮਲੇਸਲਮਏਰੇਏਏਥੋਂਉਸੋਪਓਸਾਗਰਉਪਲਇਕਕਨਾਲਰਾའདོན་པར་སྐྱེ་ལ། །དེ་བ་ར་ན་ཞུ་ན། །འགྲོ་བ་ན་ས་ག་དང་༄༅། །གཤིན་རྗེ་གཤེད་དམར་ལྷ་ལྔའི་བར་དུ་ལོ་དོན་དེ་ནགས་ལ་ས་ཁ་ལོ་བས་མངོན་པར་རྟོགས་པ་དངོས་ལ་་་་་པར། ཤི་ཉེཁཚོ་ཁ་བ་གྲུབ་ཀྱི་སྙེ་མ་ཞེས་བྱ་བ། །ཁ་༌འགྲོ་ན་ཚེ་ཊ་ཀ་མང་བ། ཀ་ཁ་ག་ང་འ་ཁ། འོ་ཀོལ་། བལཟླ་ཁོ་འབའ་སྐྱོར་ཏཁཞུ་འོང་། །ཁ་ཆ་ཐོབ། །ཚད་བཀག་མི་བྱ་བའི་ཁ། ལམ། དུ་པ། ཁ་ཁ་ཁ་ཁ་ཉི་་ཏ། མ་། རྒྱུན་ཁྱཚོ་བོ་ཁོ་ཚོ་དང་། ཏི་༄༅། དཔལ་གཤིན་རྗེ་གཤེད་དམར་ལྷ་ལྔའི་མངོན་པར་རྟོགས་པ་དངོས་གྲུབ་ཀྱི་སྙེ་མ་ཞེས་བྱ་བ་བཞུགས་སོ། །༄༅། །རྒྱ་གར་སྐད་དུ། ::ཡ་མནྟ་ཀ་རཀྟ་དེ་ཝ་པཉྩ་བྷི་ས་མ་ཡཱ་སིད་དྷི་མཉྫ་རི་ནཱ་མ། བོད་སྐད་དུ། གཤིན་རྗེ་གཤེད་དམར་ལྷ་་ལྔའི་མངོན་པར་རྟོགས་པ་དངོས་གྲུབ་ཀྱི་སྙེ་མ་ཞེས་བྱ་བ། བཅོམ་ལྡན་འདས་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ལ་རེ་ཤིག་དཀྱིལ་འཁོར་འདིར་དབང་བསྐུར་བ་ཐོབ་ཅིང་དམ་ཚིག་དང་ལྡན་པའི་རྣལ་འབྱོར་པས། ཡིད་དང་འཐུན་པའི་གནས་སུ་སྟན་བདེ་བ་ལ་འདུག་ནས། བདུད་རྩི་རིལ་བུ་ལྕེ་ལ་བཞག་ཏེ། བདག་དང་འགྲོ་བ་ཁམས་གསུམ་ནམ་མཁའི་མཐའ་དང་མཉམ་པའི་སེམས་ཅན་ཐམས་ཅད་ཕྱོགས་བཅུ་དུས་གསུམ་གྱི་བདེ་བར་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྐུ་་་གསུམ་ོཐུགས་ཡོན་ཏན་འཕྲིན་ལས་ཐམས་ཅད་ཀྱི་ངོ་བོ། ཆོས་ཀྱི་ཕུང་པོ་སྟོང་ཕྲག་བརྒྱད་ཅུ་རྩ་བཞིའི་འབྱུང་གནས། འཕགས་པའི་དགེ་འདུན་ཐམས་ཅད་ཀྱི་།
