བསྟན་འགྱུར། རྒྱུད། ༼གུའིཟླ་བ་གསུམ་ན་འཆི། །༄༅། །མིག་གི་འོད་སྐོར་ལ་བརྟག་པའི་འཁོར་ལོ། །ཞག་ལ་ན་འཆི། ཟླ་བ་གཉིས་ནི་འཆི་གཞག་གསུམ་ན་འཚོ། སྟེང་དུ་མེང་ན། （སྟེང་དུ་མེད་ན། རྣ་བའི་ངོས་སུ་རླུང་སྣ་བུག་གཡས་ནས་རྒྱུ་བའི་མིག་གཡོན་པའོ། །རླུང་སྣ་བུག་གཡས་ན་རྒྱུ་བའི་མིག་གཡས་པའོ། །af.galgའ*ངོས་སུ་ཆུད་ན། ཏུ་འོང་མེད་ན། vའི་ངོས་སུ་མེང་ན། བྱ། མིག་གི་འོག་ད་མེད་ན། ཟླ་བ་གཅིག་ན་འཚོ། ཟླ་བ་དྲག་ན་འཆི་ཞག་གཉིས་ན་འཚོ། ཞག་བཅུ་ན་འཆི་ཅ༄༅། །འཕོ་བ་བཅུ་གཉིས་བརྟག་པའི་འཁོར་ལོ། །འཆི། ཅ་ལོ་བཞི་རུ་ལོ་གཅིག་མ་ཚང་བས་འཚི། འཆི། ང་ལོ་བཞི་རུ་ཟླ་བ་དགུ་མ་ཚང་བས་ག་ལོ་བཞི་རུ་ཟླ་བ་དྲུག་མ་ཚང་བས་རུ་ཟླ་བ་གསུམ་མ་ཚང་བས་འཆི། ཀ་ལོ་བཞི་ན་འཆི། ཁ་ལོ་བཞི་དགུར་ལོ་གཅིག་མ་ཚང་བས་འཆི། དྲུག་མ་ཚང་བས་འཆི། ཅ་ལོ་བཅུ་བས་འཆི། ག་ལོ་བཅུ་དགུར་ཟླ་བ་བཅུ་དགུར་ཟླ་བ་གསུམ་མ་ཚང་ཀ་ལོ་བཅུ་དགུ་ན་འཆི། ཁ་ལོ་ཅ་ལྔར་རྒྱུ་ན། ང་བཞི་རྩ་ལྔར་རྒྱུ་ན། གཅིག་ཁ་གཉིས་ག་གསུམ་། ནས་ཉི་མར་ཞག་ཀ་གཅིག་ཁ་གཉིས་ག་འདུས་ཏེ་ཀ་ཞག་། ང་བཞི་ཅ་ལྔར་རྒྱུ་ན། ཁ་གཉིས་ག་གསུམ་ག་གསུམ་ང་བཞི་གསུམ་ཅི་གཅིག་ཁ་གཉིས་། ལམ་དུ་ཞག་ཀ་པོ་རླུང་ཉི་མའི་དགུན་ཟླ་འབྲིང་ཟླ་ར་བའི་དགུན་། Jའདིར་ཉི་མར་དགུན་དེརའ་ཟླ་ར་བ་ཁ་གཉིས་ག་གསུམ་ང་བཞི་རུ་ལྔར་རྒྱུན། ཆུང་ཉི་མར་ཞག་ཀ་གཅིག་། ང་བ་སྒོ་ཐ་མ་ཚང་བས་འཆི། ཀ་ལོ་བདུན་ན་འཆི། བས་འཆི། ཅ་ལོ་བདུན་ལ་ལོ་གཅིག་མ་ཚང་བས་འཆི། ཚང་བས་འཆི། ང་ལོ་བདུན་ལ་ཟླ་བ་དགུ་མ་ཚང་ག་ལོ་བདུན་ལ་ཟླ་བ་དྲུག་མ་ཁ་ལོ་བདུན་ཟླ་བ་གསུམ་རྒྱུ་ན། ཇི་བཞི་ཅ་ལྔར་རྒྱུ་ན་སུ་ཚུwཁ་གཉིས་ག་གསུམ་གཅིགམར་ཞིག་ཀ་གཅིག་ཉི་མར་པོར་འདུས་ཏེ་ཉི་དཔྱིད་ཟླ་འབྲིང་:་སྦྱང་ཙསྟོན་ཟླ་འབྲིང་ཟླ་ར་པོ་ཉི་མར་ཀ་གཅིག་ཁ་གཉིས་ག་གསུམ་ང་བཞི་ཅ་ལྔར་རྒྱུ་ན། ཅ་ལོ་བཅུ་་་ཁ་ལོ་བཅུ་དྲུག་ཀ་ལོ་བཅུ་དྲུག་ན་འཆི། ཏུ་ཟླ་བ་གསུམ་མ་ཚང་བས་འཆི། ག་ལོ་བཅུ་དྲུག་ཏུ་ཟླ་བ་དྲུག་མ་ཚང་བས་འཆི། ང་ལོ་བཅུ་དྲུག་ཏུ་ཟླ་བ་དགུ་མ་ཚང་བས་འཆི། དྲུག་ཏུ་ལོ་གཅིག་མ་ཚང་བས་འཆི། གསུམ་ང་བཞི་ཅ་ལྔར་རྒྱུ་ནཀ་གཅིག་ཁ་གཉིས་ག་ཐ་ཆུང་འདུས། བ་ཉི་མར་ཞག་ཀ་གཅིག་་ཁ་གཉིས་ག་གསུམ་ང་བཞི་ཅ་ལྔར་རྒྱུ་ན། ཉི་མར་ཞག་ཀ་གཅིག་ཁ་གཉིས་ག་གསུམ་ང་བཞི་རྩ་དེར་དབྱར་ཟླ་གྱུར། འབྲིང་འདུས་ཏེ་[ཐ་ཚུད་ཀ་གཅིག་ཁ་གཉིས་ག་གསུམ་ང། བཞི་ཅ་ལྔར་རྒྱུ་ན། ཞེལྔར་རྒྱུ་ན། ཉི་མར་ཞིག་ཀ་གཅིག་ཁ་གཉིས་ག་གསུམ་ཀྱི་ཟླ་། །ང་བཞི་ཅ་ལྔར་རྒྱུ་ན། .ཀ་ལོ་བཅུ་ན་འཆི། ཁ་ལོ་བཅུར་ཟླ་བ་གསུམ་མ་ཚང་བས་འཆི། ག་ལོ་བཅུར་ཟླ་བ་དྲུག་མ་ཚང་བས་འཆི། ང་ལོ་བཅུར་ཟླ་བ་དགུ་མ་ཚང་། བས་འཆི། ཅ་ལོ་བཅུར་ལོ་གཅིག་མ་ཚང་བས་འཆི། །ཀ་ལོ་བཅུ་གསུམ་ན་འཆི། ཁ་ལོ་བཅུ་། གསུམ་དུ་ཟླ་བ་གསུམ་མ་ཚང་བས་འཆི། ག་ལོ་བཅུ་གསུམ་དུ་ཟླ་བ་དྲུག་མ་ཚང་བས་འཆི། ང་ལོ་བཅུ་གསུམ་དུ་ཟླ་བ་དགུ་མ་ཚང་བས་འཆི། ཙ་ལོ་བཅུ་གསུམ་དུ་ལོ་གཅིག་མ་ཚང་བས་འཆི། །
