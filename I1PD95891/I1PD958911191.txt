ཀྱག་པ། ཀ་བསྟན་འགྱུར། །རྒྱུད། aEpའི་ཆppསོ། །བརྒྱ་རྩ་བརྒྱད་དམ་སྟོང་ཚར་ཅི་ནུས་སོ་བཟླའོ། །ཚིག་བསྲེ་བའི་སྙིང་པོ་བརྗོད་པར་བྱ་སྟེ། ཨོ་དུ་ན་དུ་ན། བི་དུ་ན་བི་དུ་ན། ཙ་ལ་ཙ་ལ། ཙ་ལ་ཡ་ཙ་ལ་ཡ། :པྲ་བི་ཤ་ཡ་པྲ་བི་ཤ་ཡ། ས་མ་ཡ་མ་ནུ་པ་ལ་ཡ་བློ་སྦྱོ། ས་མན་ས་ན་པྲ་བི་ཤ་ཀྱི་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། འབྱུང་པོ་ཡབ་ཡུམ་གྱི་དམ་ཚིག་བསྲེ་བའི་སྔགས་ཏེ། འདི་་་བརྗོད་ན་འབྱུང་པོ་ཐམས་ཅད་དམ་ཚིག་ལས་མི་འདའོ། །།བར་ཆད་སེལ་བ་ཡང་འདིས་བྱའོ། །དངོས་གྲུབ་ནི་ཕྱོགས་གྲུབ་པའི་དུས་སུ་མཐའ་དག་ནས་འོང་སྟེ། ཤེས་རབ་ཆེ་བ་དང་། ཆོས་མཁས་པ་དང་། མི་བྱམས་པ་དང་། རྫས་དང་ལོངས་སྤྱོད་འདུ་བ་དང་། ཕ་རོལ་མིའི་ཡིད་འཕྲོགས་པ་སྟེ། ཐམས་ཅད་དབང་དུ་འདུ་ཞིང་མོས་པར་འདུན་པ་དང་། ནོར་སྟེར་བར་བྱེད་དོ། །གནོད་སྦྱིན་ཡིད་འཕྲོག་གི་་བསྒྲུབ་ཐབས་སློབ་དཔོན་ཆེན་པོ་ཨིན་དྲ་བོ་དེས་མཛད་པ་རྫོགས་སོ། །།།།་རྒྱ་གར་གྱི་མཁན་པོ་ཁ་ཆེ་པཎ་ཆེན་དང༌། བོད་ཀྱི་ལོཙྪ་བ་ཚུལ་ཁྲིམས་གཞོན་ནུས་བསྒྱུར་བའོ། །མངྒ་ལཾ། །ག་ད་བ་ས་དང་། :ལ་བཀpp//wཁ་དང་ན་ལ་་་་་་་་་་་་་་་་ལ་་་། ཁོ་ཁོ་ན་འཚེ་ཏ་ཏེད། ང་། ཝysཡིད་འོང་ལ། གནས་ཀT:ཚོ། །ཏིད་ལ་ཉར་ཚོལཀྱི། འ་དང་། ཡོ་ག་དང་གཞི་དུས་ཀྱི་ཊི་ཀ་ར་ག་ཏེ་པ་ར་ག་ཏ་བློའབར་བ་དང་། །དེ་ནས་ར་མོ་ར་ཀ་ཁ་དམ་པ་ནི་ཁ་བ་ཙམ་ཞིག་པ་དག་བདེན་པ་ཡིན་ཚེ་ལོ་པ་འ་ཆེམ། །ཁྱད་པར་དུ་ཡོད་ཅིང་དཀྱིལ་། །ག་ང་ད་ན་བ་དེར་ད་དུང་ཁ་ཅིག་ཏུ་ཁ་ཆེ་དེ་དག་ལ་Pਨਸ਼ੀਲn'ਸਨਨਨੜਨਾਲਟਸੌਟਨਇਸਨੂੰਸਾਲਾਨਾਏਤੇਕੌਸ਼ਾਲਸਮkਹਿਮਾਲਕདང་ད་བར་མ་སྐྱེད་ཀྱི་ཡོད་པ་ར་ཁམ་ཎཉིས་པ་དེ་དག་ནད་པ་ཁོ་འཚུབ།
