བསྟན་འགྱུར། ་རྒྱུད། བ་གདམས་པ་དང་། སློབ་དཔོན་རྣལ་འབྱོར་ཉིད་ལ་གདམས་པའོ། །་་་་་ཏུ་ལྟ་ན་པ་དང་། ང་། བ་ཡི་ལ་པ་ར་ལཱ་ག་ང། འཇུག་བསྡུ་བ་ལ་གསུམ་སྟེ། །གདམས་དེ་ལྟར་ཤིང་རང་བཟོ་སྤང་བ་དང་། དགེ་བའི་རྩ་ནམ་མཁའ་བར་སྔོ་བ་དང་། གཞུང་གི་ཚད་ཉོན་མོངས་ཡི་གེ་བསྟན་པའོ། །སློབ་དཔོན་ངག་གི་དབང་ཕྱུག་གྲགས་པས་མཛད་པའི་འཆི་བ་བླུ་བའི་བསྡུས་དོན་རྫོགས་སོ། །་རེ་བ། །འདས་པ་དང་། ཆུ་དང་ཀ་དག་གཡོ་བ་ལས་ལ་བར་ད་དུང་། ད་དེ་ག་ང་བ་མ་ཡིན་པའི་ཕྱིར་དུ་ཡ། ཤོ་བའི་མ་བསྡུར་མཆན། ལ་ལ་བྲོ་། །ཆཁའ་ལས་ད་ལྟོ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་བླངས་ཏེ་བཀོད། «པེ་»སོ། །«པེ་»པར། ༄༅«པེ་བརྟག«པེ་བསྟན། །«པེ་བརྟགས། Q«པེ་»བརྟན། །ི«པེ་»མཇུག«པེ་གཙོར། «པེ་སྲེགས། མི«པེ་»བསྔགས། དེ་དང་། ་ཡའོ་་་་བུ་«པེ་རིམ། ཨ་་་་།  ། །།།ཀ་ལ་པ་། ਤਾਲਬਪਤਕਾਲਸੈਂਸਰਇਸਲਾਸੁਣਲਾਓལུ་བསྡུར་འབྲས་རེའུ་མིག་ལ། །ང་བས་གནས་བོད་དུ་གོ་བ་རྒྱུད། （དང་། ཕྱག་འཚལ་བ་དང་དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། ང་སྡེ་ཚན། སྔགས། ::སྔགས། འདབ་པའི་པོད་རྟགས། ཚན་གྲངས། ན་_བབt_ན་ཕྲེང་། །ཚེག་ཁྱིམ། །ར།
