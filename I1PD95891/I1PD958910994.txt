ཡིལྟ་ཀྱི་བགྱི་སྒོ་འབྱེད་པ་སྟོན་་་་ལ་་་་་ਮੇਂਨੂੰਇਸਲਈਇਸਨੂੰਵਾਰੋ,ਸੜ༄༅། །དཔལ་ལྡན་གཤིན་རྗེ་གཤེད་ནག་པོའི་འཁྲུལ་དེས་གཡོ་འཁོར་བྲི་བའི་ཆོ་ག་རྒྱས་པའི་རྟོག་པ། ::ཀླུ་རྟ༼ལོ་ཚ:ཀ་ལ་ཅུ(ས་ཏ་ཀར་ལ་་ར་རུ་ཀ་ཙ་ལེ*མ་། །།། ། ཀ་ཆུ་ཁ་ཤ་ཚོ་ལ་བློན་ཀྱི་ཊཀཱ་ཀྱི་:ཀྱིལ་། ཀཱ་ཏ་ལ་འོ་ཀྱངོ་ར་ཌོ་ལར་༽ཡང་འོ་མ་དེ། །རྟོག་པ་གསུམ་པ་བཞུགས་སོ། །༄༅། །རྒྱ་གར་སྐད་དུ། ཀྲིབྷ་ཙིཡ་མ་ཨ་རི་ཙཀྲ་ན་མ། བོད་སྐད་དུ། གཤིན་རྗེ་གཤེད་ནག་པོའི་འཁྲུལ་འཁོར་་་ཞེས་བྱ་བ། ལོ་ཙཱ་བ་ཡིས་བཅོམ་ལྡན་འདས་གཤིན་རྗེ་གཤེད་པོ་ལ་ཕྱག་འཚལ་ལོ། །དོན་དམ་དུ་ཆོས་སྐུ་ཞི་བ་ཉིད་ལས་མ་གཡོས་ཀྱང་ཀུན་རྫོབ་དུ་སྤྲུལ་སྐུའི། །གང་འདིར་ཁམས་་གསུམ་ཆོགྲ་སེམས་ཅན་ངོ་བོ་ཕྱིར་ལྟར་རམ་དྲང་དོན་དུ་ཁམས་གསུམ་གྱི་སེམས་ཅན་གྱི་སྲོག་གནད་འདོར་བ་གཤིན་སྲོག་ལ་དབང་བྱེད་པས་རྗེ་དེའི་འདུལ་བྱེད་ལ་གཤེད་དམ་དགྲ་ཞེས་བྱ་བ་ནང་ལྟར་རམ། ངེས་དོན་དུ། གཤིན་ཁམས་གསུམ་གྱི་རྣམ་རྟོག་རྗེས་མ་རིག་པ། གཤེད་འཇམ་དཔལ་ཡེ་ཤེས་ཀྱི་སྐུའོ། །ཞེས་གྲགས་པ་ཡི། །འཁོར་ལོ་བསྐོར་བའི་ཆོ་ག་བཤད། །ཡོན་ཏན་ལྡན་པའི་ཐོས་བསམ་དབང་དམ་ཚིག་སྡོམ་པ་ཐོབ་པའི་སྔགས་པས་ནི། །ཆོ་ག་མདོར་བསྡུས་བཞིན་དུ་བསྙེན་པ་བྱས་པ་ཡིས་ཡིས། །གནས་དུར་ཁྲོད་དབེན་པ་རྒྱུད་རྒྱས་པར་བཤད་པའི་ལས་སྦྱོར་ལས་ཤེས་པ་གང་རུང་ལ་སོགས་པར། །མཎྜལ་ཁྲུ་དོའི་ཚད་དུ་གྲུ་བཞི་པ་བྱ། །དྲི་ལྔ་དང་བའི་རྣམ་ལྔ་དང༌། །བྱང་སེམས་དཀར་དམར་ཉི་ཟླ་་་།
