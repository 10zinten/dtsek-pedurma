ལ་་པ་ལ་ཡི་རང་བར་གྱུར་ནས་བསྟོད་པ་ལ་སོགས་པ་མཛད་དོ་ཞེས་འཆད་དོ། །བདག་འདྲ་སྡེབ་སྦྱོར་གསལ་ལྡན་མ་ཡིན་ཀྱང་། །དོན་ལ་མོས་པའི་གང་ཟག་རང་བས་དམན་པ་དང༌། །ཆོས་ཉིད་མཐོང་ལ་ལོག་པར་རྟོག་པ་ཅི་ཡོད་ཀྱི། །དོན་དུ་བདག་གིས་བྱས་པ་ཡིན། །དེ་ལས་བྱུང་བའི་དགེ་བ་དེས། །བདག་དང་མཐའ་ཡས་མ་རིག་པ། །མི་མཐུན་ཤེས་རབ་ཀྱིས་འཇོམས་ཤོག། དོན་ལ་མོས་པའི་མཐའ་གཉིས་མེད་པར་ཤོག། འཕགས་པ་འདའ་ཀ་ཡེ་ཤེས་ཀྱི་མདོའི་དོན་གསལ་བ་ཞེས་བྱ་བ། སློབ་དཔོན་ཞི་བ་ལྷས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་དྷརྨ་རཱ་ཛ་དང་། བོད་ཀྱི་:ལོ་ཙཱ་བ་འཕགས་པ་ཤེས་་་ਨਹBu'ਹਮੂਝ॥ਸ਼REਮਿਲਰਾਰਸਾਨਸਾਲਦੇਨਾਲਪੁਲਸ་་ཚོ་བ་དེ་དག་བབནབསྡུར་མཆན། ས་བར་བ་། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་འཕགས་པ་འདའ་ཀ་་ཡེ་ཤེས་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོའི་འགྲེལ་པ་ཞི་བ་ལྷས་མཛད་པ་བཞུགས་སོ། །ཞེས་དང་། 《སྣར་ལ་་འཕགས་པ་འདའ་ཀ་...མཛད་པ་བཞུགས། ཞེས་འཁོད།  ི«པེ་»《སྣར་པ། «པེ་»སྣར་བཟང༌། «པེ་»«སྣར་»ནི། «པེ་»བ། «པེ་»«སྣར་བཟང་ངོ༌། «ཅོ་»བཞགས། གྱི«སྣར་རྟོགས། །«པེ་»«སྣར་»དེ་ནས། «པེ་»«སྣར་དེས། ཚེ«པེ་»«སྣར་དེ། «པེ་»«སྣར་ལས། མོ«པེ་»«སྣར་བྱུང་། @«སྣར་»ངེས། ༄༅«པེ་»«སྣར་»ཡིས། «པེ་»«སྣར་འཁྱམས། «པེ་»སྣར་»དེ། ཁྱེ«པེ་»«སྣར་ནམས་ཆི་ཀ་མའི་ཚེ། འགྱེ«པེ་»«སྣར་»སྒོམ། །«པེ་པར་སྟན། «སྣར་པར་བསྟན། @«པེ་»«སྣར་»ད། ཉི«པེ་»«སྣར་»+ཀྱི། ཁ«པེ་»«སྣར་ལ། «པེ་»སྣར་ཀྱི། ༄༅«པེ་»«སྣར་རྣམ་དག@«པེ་»«སྣར་—ཡང་། «པེ་»《སྣར་»དམ། «པེ་»«སྣར་ལས་བལྟོས། ཁོ«པེ་»«སྣར་རྒྱ་ཆད་པ། ཀྱེ«པེ་»«སྣར་»སྤོང་། ཚེ«པེ་»«སྣར་ལས་བལྟོས། །ག་དོག:_།
