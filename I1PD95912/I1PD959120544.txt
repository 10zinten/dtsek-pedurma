བསོད་ནམས་ཐམས་ཅད་ལ་རྗེས་སུ་ཡི་རང་ངོ༌། །སངས་རྒྱས་ཐམས་ཅད་ལ་བསྐུལ་ཞིང་གསོལ་བ་འདེབས་སོ། །བདག་བླ་ན་མེད་པའི་ཡེ་ཤེས་ཀྱི་མཆོག་དམ་པ་ཐོབ་པར་གྱུར་ཅིག། མི་མཆོག་རྒྱལ་བ་གང་དག་ད་ལྟར་བཞུགས་པ་དང་། །གང་དག་འདས་པ་དག་དང་དེ་བཞིན་གང་མ་བྱོན། །ཡོན་ཏན་བསྔགས་པ་མཐའ་ཡས་རྒྱ་མཚོ་འདྲ་ཀུན་ལ། །ཐལ་མོ་སྦྱར་བར་:བགྱི་སྟེ་སྐྱབས་སུ་ཉེ་བར་མཆི། །ཞེས་པ་ནི་གོ་སླ་བས་མ་བཤད་དོ། །བྱང་ཆུབ་ཀྱི་ལྟུང་བ་བཤགས་པའི་འགྲེལ་པ་བྱང་ཆུབ་སེམས་དཔའི་སློབ་པའི་རིམ་པ་ཞེས་བྱ་བ། སློབ་དཔོན་:ཛེ་ཏཱ་རཱི་དགྲ་ལས་རྣམ་པར་རྒྱལ་བས་མཛད་པ་རྫོགས་སོ། །བལ་པོའི་ྋཔཎྜི་ཏ་དང་། ་་:ཆགས་ལོ་ཙཱ་མིབས་བསྒྱུར་བའོ། །:ལནབབབབབབཀྲ་ཤིས་ལན་ppདང་མའིཀྱི་བར་གྱུརཔར་གྱུར། །པ་/བ་བསྐྱེལ་བའི་བསྡུར་མཆན། གང་་། yoཡོང་བ་བ་ད། དལ་སློབ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་ལྟུང་བཤགས་ཀྱི་འགྲེལ་པ་དགྲ་ལས་རྒྱལ་བས་མཛད་པ་བཞུགས། ཞེས་འཁོད། «པེ་»«སྣར་སྤྱ། «སྣར་བྲིཏྟི། ལོ་«པེ་»«སྣར་ཀླུ། ལོ་རེའིལོ་དང་/ཨེ«པེ་»«སྣར་ལ། ཉི«པེ་»«སྣར་_བ། «པེ་»«སྣར་»ལྷས་བྱིན། «པེ་»«སྣར་»གཞིའི། །«པེ་»«སྣར་»དེ། «པེ་»«སྣར་ཏུ། «པེ་»«སྣར་»བ། ༄༅«པེ་»«སྣར་»ནི། ཨེ«པེ་»«སྣར་དཔའ། ཚེ«པེ་»«སྣར་དཀོན། །«པེ་»«སྣར་བསྲེག«པེ་»«སྣར་བསྒོམ། «པེ་པ་སྟེ། ཟད། «སྣར་པ་སྟེ། ཟག@«པེ་»«སྣར་»«ཅོ་+པ། ༄༅«པེ་»«སྣར་གྱི། འ«པེ་»མཁྲེགས་ཞིང་། 《སྣར་»མཁེགས་ཞིང་། «པེ་»«སྣར་སོ། (《སྣར་བཟངས། །ཚེ«པེ་»«སྣར་ཐོག«པེ་»«སྣར་ལ། ན«པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་»བར། «སྣར་ན། །O«པེ་»«སྣར་གནོན། «པེ་»«སྣར་»འདུལ། །«པེ་»«སྣར་པོའི། «སྣར་»བཞེས། «པེ་»«སྣར་»ནི། ཉི«པེ་»«སྣར་སོ། «པེ་»«སྣར་»འགྲོན། ཚེ«པེ་»_—།
