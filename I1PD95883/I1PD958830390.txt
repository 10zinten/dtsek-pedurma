འགྲོ་བ་དངོས་མེད་ཀྱི་རང་བཞིན་གཡུ་དུ་མ་དང་། སྤྱི་ལམ་ལ་བྱ་བ་འདོད་པ་ལ་ན་བཛྲ་སུ་བབ་པ་ཀོ་དབསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཁྲོ་བོའི་རྒྱལ་པོ་འཕགས་པ་མི་གཡོ་བ་ལ་བསྟོད་པ་བཞུགས། «པེ་»«སྣར་ཨཱ། «པེ་»«སྣར་»སྟོཏྟཱ། «པེ་»(«སྣར་བརྐུམ། «པེ་»«སྣར་ཟ། «པེ་»«སྣར་བཞིར། «པེ་»«སྣར་»འདུལ། «པེ་»«སྣར་ཆོད། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་སྟོབས། ®«སྣར་གནག་མ«པེ་»《སྣར་»འདུལ། ོ«པེ་«སྣར་བསྟོད། «པེ་《སྣར་»གསོལ། «ཅོ་»—ཀྱིས། ལོ་ལུ་དབང་བུའི་ལུས་ཅན་རེན་པོ་ཆེ་ཚོད་དུ་གད་པ་དང་། ན་བ་ག་རེ་ཅོད་པན་གྱི་ལག་ཚུལ་དུ་ལན་བ་། །ན། བདག་པ། བྱ་རྒྱུད་ཁཊོ་དེའི་བསྡུར་འབྲས་རེའུ་མིགལག་། ག་ན་ས་པ་ལ་ནབབབབནརྒྱུད། （། དཔེ་གཞི། སྡེ་དགེ། དཔེ་ཅིན། སྣར་ཐང༌། :ཅོ་ནེ། ་་་་་་་་་་་སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། ，་རྒྱུད། པོད་རྟགས། ཚན་གྲངས། ཀལྡེབ། བ—བ*། བ—བ་བབ་བནན། ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། །ਵਿਰਸਲਾ'ਸਮਰਥਨਕੀਕਸਮ:ཁ་བ་འབབ་པ་དེར་ཕྱི་མ་ཞན་པ་་་དུང་ས་པ། །ལམ་གྲོན་དེ་བ་ལ་ད་ནང་ལ། ਸੁਲਤਾਨਸਿੱਧਕਰਨਾਸਪਲਣར། _།
