རྒྱུད། ༼པུའུབདག་ཉིད་ཀྱི། །སྙིང་གར་བསླན་པ་འཁོར་ལོ་བཞིན་དུ་བསམ། །དེ་ལྟར་ཞག་གྲངས་ཉི་ཤུ་རྩ་གཅིག་ཏུ། །ཐུན་གསུམ་ཉིད་དུ་བསྒོམ་པ་གང་ཡིན་དེས། །བཀའ་དང་བསྟན་བཅོས་ཀུན་གྱི་ཚིག་དང་དོན། །བལྟས་པ་ཙམ་གྱིས་ཁོང་དུ་་་་ཆུད་པར་འགྱུར། །གཞན་ཡང་ཀུན་ལ་མི་འཇིགས་སྤོབས་པ་འཐོབ། །སྙན་དངགས་བསྟན་བཅོས་ཐོགས་མེད་རྩོམ་ནུས་ཤིང་། །དེ་ཡི་སྨྲ་བ་སུས་ཀྱང་་་་བཟློག་མི་ནུས། །དེ་ཕྱིར་འདི་འདྲ་གར་ཡང་རྙེད་པར་དཀའ། །འཕགས་པ་མི་གཡོ་བའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །།།རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ་དཱ་ན་ཤཱི་ལས་རང་འགྱུར་དུ་མཛད་་པའོ༑། །དབང་འབྲུག་པ་མེད། ཀྱི་ཡང་། མ་འོངས་སྟེ། །ཁོ་ན་མ་འོབསྡུར་མཆན། །ཁ་ན་ཏུ། ཞི་དེ། །།།།ཀྱི་དབང་ལ་དམ་པ༌:z་། ཊཱི་ཀཱ། ༈། །ཁ། ལྔ་པ་Cཚེ་ཡྻzཤད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་པ་མི་གཡོ་བའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་ཨཱཙཱ། ན«པེ་»«སྣར་»+སུ། «པེ་»«སྣར་། སྟེངས༑ཨ་༌་(འ«པེ་»«སྣར་བ། ་ར་་ན་། བྱས་ཏ། །ཙ་ལ་ཞུགས་(«པེ་»«སྣར་བརྩོམ། «པེ་《སྣར་ཡིས། «པེ་»«སྣར་བརྙེད། ਮਿਲਕਬੂਟਲਓਟਣਸਮਝ,ਆਸਮਸਅਲਮਸਾਣ་་འ་ཏུར་ཏུ་ངའ་ཁཚོ་ལཾ། །ཀ་ཁའི་ནང་ཀ་ཆ་། ཚོ་»འདི་ཚོ་ལ་འོ། །་་༌༌་་་ཏུཌ། ཆ་ཀྱི༌འོ་ཆུ་གློ་། ཀ་ཀ། ཏུ་ར། ཀ་ཆཊོར་AਲੋਂਸਾਧgਨੇਸਨਸਲਵਾਬਲੌਰਉਮਛਟਨਾਹੈਵਿਸਲਨਵੇਂਉAB/ਸBਨਸਥੋਂB%ERBApਰਇਮਾਰਤ—།
