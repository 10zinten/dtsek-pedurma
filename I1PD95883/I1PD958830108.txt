གསོལ་དམིགས་མེད་རྒྱས་བཏབ་ལ། །།དེ་ནས་དགའ་སྟོན་དབུགས་ཕྱུང་སྟེ། །།རིག་པའི་ལས་དང་གཟེངས་ཡང་བསྟོད། །དེ་ནས་གཏོར་མ་བྱིན་བརླབས་ཏེ། །མགོན་པོའི་གཏོར་མ་བཏང་བ་ནི། །སྔགས་ཀྱིས་སྤྱན་དྲངས་ཡེ་ཤེས་གཞུག་པོ::། པཾ་ལས་:གྱུར་པ་སོ་སོ་ཡི། །སྐུ་མདོག་ཕྱག་མཚན་སོ་སོར་དམིགས། །དེ་ནས་འཛབ་དང་བསྙེན་པ་འབུལ། །དེ་ཡི་སྔགས་ནི་འདི་དག་བཟླས། །རྡོ་རྗེ་ལྕེ་་་ལས་ཡང་དག་ཕྱུང༌། །ཨོཾ་ཨེ་ཧེ་ཧི་ཧ་ན་ཧ་ན་བྷིནྡ་བཛྲ་ཧཱུྃ་ཕཊ། ༀ་ཨེ་ཧེ་ཧི་བཛཱ་ན་ལ་བྷིནྡ་ོན་མཎྜ་ལ་བུདྡྷ་སརྦ་བྷསྨི་ཀུ་རུ་སརྦ་དུཙྪཱ ིཧཱུྃ་ཕཊ། ཨོ་ཨེ་བྱེ་ཧི་བཛཱ་ཀཱ་ལ་དཎྜ་ཧ་ན་ཧ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཧེ་ཧི་བཛྲ་ཀི་རི་:ཊི་སྟེ་རཀྵ་ས་དེ་བ་ཏ་ཡེ་སྭཱ་ཏཱ། ཨོཾ་ཨེ་ཧྱ་ཧི་ནཱ་ག་བཛྲ་ཨཱ་ན་ཡ་སརྦ་དྷ་ན་དྷ་ནྱ། ་སརྦ་ན་མ་ནི་མུཀྐཱ་ལཾ་ཀ་ར་དྷ་ར་སརྦ་ནཱ་ག་ན་ར་ནི་ཙ་བཛྲ་དྷ་ར་ཡ། ས་མ་ཡཱ་མ་ནུ་སྨ་ར་ཨ་:ཀཟླུ་ཨ་ཀཟླུ་ེབནྡྷ་བནྡྷ་ཧ་ར་ཧ་ར་:པྲ་ཎ་ན་མ་ཧཱ་:ཙྪེ་ད་ཧཱུྃ་ཕཊ། ཨོ་ཨེ་བྱེ་ཧི་བཛྲ་ཨ་ནི་ལ་མ་ཧཱ་བི་ག། ཨཱ་ན་ཡ་སརྦ་དུཏྟཾ''ཧཱུྃ་ཕཊ། ༀ་ཨེ་ཧེ་ཧི་ཀུ་བེ་ར་ཡཀྵ་ཏི་བ་ཏ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧེ་ཧི་དྷི་དྷུ་བྷི་ན་བཛྲ་ཤྭ་ལ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧེ་ཧི་ན་མས་མནྟ་:བུདྡྷཱ་ནཾ། ཨོ་ཕྲི་ཉི་སཱ་ཧཱ། ཨོཾ་ཨེ་ཧེ་ཧི་ན་མས་མནྟ་་་་་བུདྡྷ་:ན་མི། ཚེ་ཧེ་ཤོརྱ་པྲ་ཏོ་ཚོ་ཏི་ར་མཱ་ཡ་སྭཱ་ཧཱ། ལྷ་རྣམས་སོ་སོའི་ཕྱག་རྒྱ་ཡིས། །མཆོད་ཡོན་ལ་སོགས་གཏོར་མ་ལ། །སྔགས་ནི་བདུན་ཙམ་ཅིག་ཉིདག་གིས། །ངེས་པར་:འབུལ་དང་བསྟབས་པར་བྱའོ། །དེ་ནས་མགོན་པོ་མཉེས་པ་ཡི། །ཚིག་འདི་རྣམས་ཀྱང་བརྗོད་པར་བྱའོ། །ཧཱུཾ་ལྷ་དང་ལྷ་མིན་གནོད་སྦྱིན་ལྟོ་འཕྱེ་གྲུལ། ་། ནམ་མཁའ་ལྡིང་དང་འདབ་བཟངས་ལུས་སྲུལ་པོ། །དྲི་ཟ་སྲིན་པོ་གདོན་གྱི་རིགས་རྣམས་དང་། །གང་དག་ས་ལ་གནས་པའི་རྫུ་འཕྲུལ་ཅན། །བུ་སྨད་བཅས་ཤིང་གཡོག་འཁོར་བཅས་པ་ཡང་། །བདག་གི་པུས་མོ་ས་ལ་་་་—།
