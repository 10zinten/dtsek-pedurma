སརྦ་པྲ་བྷ་བི་ག་ཏེ། ཨཱ་ཀཱ་ཤེ་ག་ག་ན་ཏེ་ལེ། ཨཱ་ཀཱ་ཤ་བི་ཙ་རི་ཎི། ཛ་ལི་ཏ་ཤེ་་ཁ་རེ། ་མ་ཎི་མོ་ཁ་ཏི་ཀ། ཁ་ཙི་ཏ་མོ་ལི་ས་དྷ་རེ། །སུ་ཀེ་ཤེ། སུ་བྷེ་ཤེ། སུ་:བཀྲེ། སུ་བརྟེ། སུ་བཎ་་གོ་ར། ཨ་ཏི་ཏ། ཨ་ན་ག་ཏེ། པྲ་ཏཱུདྡནྡྷེ། ན་མཿསརྦ་ཏཱཾ་བྷུ་ཏ་ནཾ། ཛ་ལི་ཏི། ཏེ་ཛ་ས། །བུདྡྷེ། སུ་བུདྷེ༑@བྷ་ག་བ་ཏི། སུ་རཀྵི་མེ། །སུ་པྲ་བྷེ། ཎི་སུ་པྲ་ཏེ། སུ་ཏ་མེ། སུ་དཱནྟེ། བ་རེ་བ་ར་ཏེ། བྷ་ག་:བ་ཏི། བྷ་དྲ་བ་ཏི། བཛེ་སུ་བྷ་རེ། བི་་མ་ལེ། ཛ་ཡ་བྷ་དྲི་པྲ་ཙཎྜེ། ཙནྡྷེ། མ་ཧཱ་ཙནྡྷེ། ་་:བྷ་གོ་རི་བོ་རི། གནྡྷ་རི། གོ་བྷུ་རེ། ཙཎྜ་ལི། མཱ་ཏཾ་གི། །མ་ལི་ནི། བརྩ་ནི། སུ་མ་ཏི། པུཀྐ་སི། ཤ་བ་རི་ཤ་བ་རི། ཤཾ་ཀ་རི། ་དྲ་མི་ཌི། དྷ་མི་དྷི། རོ་དུ་ཏྲི་ནི། སརྦ་གྲ་ཧ་བི་དྷ་རི་ཎི། སརྦ་ཨརྦ་སཱ་དྷ་ནི། ཧ་ན་ཧ་ན་སརྦ::ཤཱ་ཏུཾ་ནཾ། ཌོད་ཧ་ད་ཧ་སརྦ་དུཏྟཾ་པ་ཙ་པ་ཙ་སརྦ་བྷཱུ་ཏ། སྲེ་ཏ་པི་ཤཱ་ཙཱ་ནཾ། །མ་ཐ་མ་ཐ་སརྦ་ཡཀྵ་་་་རཱཀྵ་ས། ཌཱ་ཀི་ནཱི་ཧཱུྃ། །མ་ནུ་། ་ཨ་མ་་་ནཱཉྩི། པྲ་མ་ཐ་པྲ་མ་ཐ། ཧྲི་ད་ཡཾ། ཏཱ་ས་ཡ། ཛཱི་བི་ཏཾ། གདོན་མ་རུངས་པ་ཐམས་ཅད་ཀྱི་སྲོག་གཞོམ་དུ་གསོལ། བཅོམ་ལྡན་འདས་མ་བདག་གི་སྡིག་པ་ཐམས་ཅད་གཞོམ་དུ་གསོལ། བདག་འཁོར་དང་བཅས་པ་ཐམས་ཅད་དུ་འཇིགས་པ་དང་། གནོད་པ་ལས་་་་་བསྲུང་དུ་གསོལ། །མ་རུངས་པ་ཐམས་ཅད་བཅིང་དུ་གསོལ། །སརྦ་ཀཱི་ལ་བི་ཤི་ནཱ་ཤ་ནི། :བ་ར་དནྟེ། སྲྀ་དུ་དནྟ། ཝ་ར་ཎི། མ་ར་བནྟེ། མ་ཎི་ས་ནི༑་་མ་ཧཱ་མ་ཎི༌ནི། ཙིཏི་ཙི་ཏི། བི་ཊི་བི་ཊི། །བི་མ་ལེ། ཊི་ཏི་ཊི་ཏི༑ོ་:ཏ་ཊ་ཏ་ཊ། ནི་རུ་ཊེ། ཤ་ནི་རི། ཤ་རི་ནི། གོ་བུ་རི། གོ་རི་ཎི༑p:པི་རི་ཎི། པྲ་བ་རེ། །ས་མ་རེ། །ཙཎྜ་ལི། མ་ཏཾ་གི། ་ཀཨ་ར་སི། ཏུ་བ་ཏ་སི། བརྩ་སི། །སུ་མ་ཏི། ཌཔུཀྐ་སི། ཤ་བ་རི་ཤ་བ་རི། ལཤངྐ་རི། །།
