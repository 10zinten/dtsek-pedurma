རྒྱུད། ་༼པུའིཏོཔདྨའི་འདབ་མ་དང་རྡོ་རྗེ་དག་གི་བར་དུ། ཨོ་ནི་སུཾ་བྷ་བཛཱི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཞེས་པའི་སྔགས་བརྒྱད་བྲི་བར་བྱའོ། །དེའི་ཕྱི་རོལ་དུ་རྒྱར་སོར་ཕྱེད་་དང་གསུམ་ལ་རིམ་པ་བཞིར་བསྐོར་ནས། ་་རིམ་པ་དང་པོ་ལ་ནི་ཡི་གེ་ཧྲི་ཉི་ཤུ་རྩ་ལྔའི་ཕྲེང་བ་བྲི་བར་བྱའོ། །།རིམ་པ་གཉིས་པ་ལ། ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི་སྭཱ་ཧཱ། ཨོ་ཨ་པ་རཱ་ཛི་ཏ་མ་ཏ་དེ་བི་ཏྲ། ཨོ་ཀཱ་ལ་ཀཎི་ཧཱུྃ། ཨོཾ་པདྨ་ཀུཎྜ་ལི་ཧཱུྃ་པཾ། ཨོ་སུ་ཤ་དནྟི་ནི་ཧཱུཾ། ཨོཾ་མ་ལ་ཨཾ་ཀུ་ཤ་ཧཱུཾ། ཨོཾ་བཛྲ་:ཤྲཱི་ཁ་ལེ་ཧཱུཾ། ་་ཨོ་སྦེ་ཏེ་ཧཱུཾ་ཐཾ༑ཨོཾ་མ་ཧཱ་སེ་ཏེ་ཧཱུཾ། ཨོཾ་མ་ཧཱ་:ཀཱ་ལི་ཧཱི། ཨོ་དྷུ་ཏི་ཏུ་ཊ། ཨོཾ་བཛྲ་:པཱ་ཎི་ཧྲ་། ཨོཾ་བཛྲ་པཱ་ཤ་ཧོ། ཨོཾ་བཛྲ་:པཱ་ཎི་པྲྀཏ་ཨོ་བཛྲ་པ་ལི་མ་ཧཱ་བ་ལི་དྷི། ཨོ་བཛྲ་མཱ་ལེ་མཾ། སྔགས་དེ་དག་གོ། །རིམ་པ་གསུམ་པ་ལ། ་ཁཚཨོཾ་མ་ཎི་དྷ::རེ་ན་མ་ཧཱ་པྲ་:ཏི་སཱ་རེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཙུནྡེ་ཧི། དོ་ཨོ་སུ་ཝཎ་་ཀེ་ཤི་ཀྲྀ་ཊ། ཨོ་:པི་ག་ལེ་ལཾ། ཁ་ཨོ་བི་རཱུདྨ་ལི་ནི་ཛ་ཧཱུྃ། ་ཨོ་ཤཱི་ཏི་ཧཱུཾ། ༀ་:པྲ་ཏི་བཛཱི་ཎི་ཧཱུྃ། ་ཨོ་བུ་དྷི་བཛྷ་ནི་ཧ། །ཨོཾ་སུཾ་བྷ་ཀ་རི་ཧཱུྃ། ཨོཾ་མ་ཎི་དྷ་རི་ཧཱུྃ། ཨོ་རཏྭཱ་མཱ་ལ་དྷ་རི་ཧཱི། ཨོཾ་མ་ཧཱ་པྲ་ཏི་སཱ་རེ་པུཾ། ཨོཾ་དྷུ་རུ་དྷུ་རུ། མུ་རུ་མུ་རུ་ཧཱུཾ། ཨོ་བ་ལ་བཌྷ་ནི་ཧིནྡྷེ། ཨོ་རུ་རུ་ཙ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ་་་ཕཊ་སྭཱ་ཧཱ། སྔགས་དེ་དག་གོ། །རིམ་པ་བཞི་པ་ལ། །ཨོ་མཀཾ་པ་ཀི་མཾ། ཨོ་ཤྲཱི་ཀུ་ཊེ་བཱི། ཨོ་ཧཱུྃ་ཨ། ཨོ་ཏཱ་ར་ཊཾ། ཨོ་ཨ། ང་ཨོ་རཀྟ་སྱ་ཀ་ཛ་ཌེ་ཛཾ། །ཨོ་བུདྡྷཱ་རཀྱི་ཧཱུྃ། ཨོ་ཧཀྵ་པཱ་ལི་ནི་བྷེ། :ཨོ་ལོ་ཀེ་ཤྭ་རི་ཁཾ། ཨོཾ་པུ་ས་པེ་ལཾ། ཨོ་ཧ་རི་ཏེ་ཀྵ་ཧཱུཾ། ཨོཾ་པཉྩི་ཀེ། བཻ་ལི་ཁི་པཾ་སྭཱ་ཧཱ། ཨོཾ་ཏྲེ་ཏ། ་ཨོ་་་བྷཀྵ་སྭཱ་ཧཱ། ཨོ་ཏི་ཡོ་ཀྲི་ཏེ་ཧཱུྃ། ཨོཾ་སྭ་ར་སྭཱ་ཏེ་ཧཱ། ཨོཾ་རཀྵ་རཀྵ་སརྦ་བྷ་ྋཡེ་ཤྩ། །ཨོ་དེ་པ་ཎི། །ཨོ་པདྨེ་སཾ། ཞེས་བྱ་བའི་སྔགས་སོ། །དེ་ཡང་ཕྱི་རོལ་དུ་རྒྱུར་སོར་གཉིས་པའི་རིམ་པ་གཅིག་མངོན་པར་བྲི་སྟེ། དེའི་ནང་དུ་རྩེ་མོ་གྱེན་དུ་བལྟས་—།
