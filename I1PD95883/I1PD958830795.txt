་བ་་་་་་་་་་་་་་་་་་་་་་་་་་བར་བར་༄༅། །གཟའ་དགུའི་མཆོད་པའི་ཆོ་ག་བཞུགས། བཏོཟླ་བ་Sབྱོན་ངའིཕུལ་ཞལ་ལྔ་ན་སུ་༄༅། །སངས་རྒྱས་ལ་ཕྱག་འཚལ་ལོ། །ཏིང་ངེ་འཛིན་གསུམ་གྱིས་་དེ་མ་ཐག་ཏུ་བསྒོམ་པ་དང་། དགུག་པ་དང་། མཆོད་ཡོན་སྦྱིན་པ་དང་། སྦྱང་བ་དང༌། བདུག་སྤོས་དང་། མེ་ཏོག་གིས་་མཆོད་པ་སྟེ། ཨོཾ་མ་ཎི་དྷ་རི་་་་བཛྲ་ཎི་མ་ཧཱ་པྲ་ཏི་ས་རེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་མཆོད་ཡོན་ནི། ཨོཾ་ཨཱརྱ་མ་ཧཱ་པྲ་ཏི་ས་རཱ་ཡི་ཨགྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། དེ་ནས་བཟླས་པ་ནི། ཨོཾ་མ་ཎི་དྷ་རི་བཛི་ནི་ཞེས་པ་ལ་སོགས་པའོ། །མཆོད་པ་ལྔས་མཆོད་དོ། །ཨོཾ་ཨཱཨཱརྱ་མ་ཧཱ་སཱ་་ཧ་སྲ་པྲ་མརྡ་རྩེ། ཨོཾ་ཨ་མྲྀ་ཏེ་ཝ་རེ་ཝ་ར། ཝ་ར་པྲ་ཝ་ར་བི་ཤུདྡྷེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་མཆོད་ཡོན་ནི། ཨོཾ་ཨཱརྱ་མ་ཧཱ་སཱ་ཧ་སྲ་པྲ་མརྡ་ནྱ་ཨགྷཾ་པྲ་་་ཏཱིཙྪ་སྭཱ་ཧཱ། དེ་ནས་བཟླས་པ་ནི། ཨོཾ་ཨ་མྲྀ་ཏ་ཝ་རེ་ཞེས་པ་ལ་སོགས་པའོ། །མཆོད་པ་ལྔས་མཆོད་དོ། །ཨོཾ་ཨཱཨཱརྱ་མ་ཧཱ་མཱ་:ཡཱ་རཱ་ཡི་ཨོཾ་ཨ་མྲྀ་ཏ་བི་ལོ་ཀི་ནི་་གཌྷ་སཾ་རཀྵ་ཎི་ཨཱ་ཀཆུ་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་མཆོད་ཡོན་ནི། ཨོཾ་།
