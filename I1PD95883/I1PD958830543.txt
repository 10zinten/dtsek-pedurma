རྒྱུད། ་་པུ། ཡང་གཞུག་པའི་སྟེང་དུ། ལ་དེའི་བུམ་པའི་ནང་དུ་ཤ་རི་རམ་གཅིག་བཅུག་ལ་་་དབྱུང་དུ་རུང་བར་བྱས་ཏེ་མཆོད་པ་ཟིན་ནས་ཕྱིར་ཡང་དབྱུང་བར་བྱའོ། །་་་་༌། མཆོད་རྟེན་གཉིས་གདབ་པའི་ཆོ་གའོ་། :། ཏ་ལ། །ཁོ་ཏི་ཀའོ་ཁ་ཁོ་ལ་སྐུར་བ་ཁོ་ਪਲੈਪਧਰਿਸਨੂੰਸਰਲਤਰਕਾਲਮਨར༄༅། །མཆོད་རྟེན་གསུམ་གདབ་པའི་ཆོ་ག་ལ། གླུ་ལས་གཞན་ནི་གོང་་་དང་འདྲ། །དཀྱིལ་འཁོར་གྱི་ཚད་ནི་ཆུར་ཁྲུ་གསུམ། ཞེང་དུ་ཁྲུ་གང་། འཕང་དུ་ཁྲུ་གང་བྱས་ལ། ་ཁྲུ་གང་གི་རེའུ་གསུམ་ཐིག་གིས་བསྙམས་ཏེ། དེ་གསུམ་གྱི་་་དབུས་སུ་པདྨ་འདབ་མ་་བརྒྱད་པ་རེ་རེ་བྲིས་ལ། །དེའི་སྟེང་དུ་མཆོད་རྟེན་་་་་གཞག། དེ་ལ་བཞག་པའི་གོ་རིམས་ནི། དང་པོར་དབུས་སུ་གཞག་གོ། ་ཁོ། མཆོད་རྟེན་གསུམ་གདབ་པའི་ཆོ་གའ:ོ། །ས་ཁག་ཚོ་ཁ་བ་གཏོར་མ་ཀ་་། ལག་ཏུ་བཀའ་ལ་ལག་པ་གང་བགལ་པོ་ལ་ནང་དང་། ༄༅། །མཆོད་རྟེན་ལྔ་གདབ་པའི་ཆོ་ག་ལ། དེ་ལས་གཞན་ནི་གོང་མ་་་དང་འདྲའོ། །དེ་ལ་དཀྱིལ་འཁོར་གྱི་ཚད་ནི་སྟེགས་བུ་ཁྲུ་གང་གྲུ་བཞི་པ་འཕང་དུ་ཡང་ཁྲུ་གང་བ་ཞིག་བྱས་ལ། དེའི་གྲྭ་བཞིར་ཡང་དེ་དང་འདྲ་བ་རེ་རེ་བྱས་ལ། །དབུས་ཀྱི་སྟེགས་བུ་གྲུ་བཞི་དང་མཐའ་མའི་སྟེགས་བུ་བཞིའི་གྲྭ་་བཞི་དང་སྦྱར་བར་བྱ། དེ་ནས་སྟེགས་བུ་དབུ་མའི་དབུས་སུ་པདྨ་འདབ་མ་བཅུ་དྲུག་པའམ། །བཅུ་གཉིས་པ་ཅིག་བྲིས་ཏེ་གྲྭ་བཞིའི་སྟེགས་བུའི་དབུས་ཀྱི་པདྨ་འདབ་མ་གྲངས་་བཅུ་དྲུག་པ་ཞིག་ན་ནི། པདྨ་འདབ་མ་བཅུ་གཉིས་པ་བྲི། །དབུས་ཀྱི་བཅུ་་་་་གཉིས་པ་ཞིག་ན་ནི། །མཐར་བརྒྱད་པ་བྲིས་ཏེ། དེའི་དབུས་སུ་མཆོད་རྟེན་གཞག། མཆོད་རྟེན་བཞག་པའི་གོ་རིམས་ནི། དང་པོ་དབུས་སུ་:བཞག་ནས། མཐའ་མ་བཞི་ནི་ཤར་ནས་ལྷོར་བསྐོར་ཐབས་སུ་གཞག་གོ། བཀ་། ཡང་རྣམ་པ་.།
