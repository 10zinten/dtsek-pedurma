ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཀྲྀ་ཏུ་དཎྜ་དྷ་རེ། ཨ་བྷ་ཡ་པྲ་དེ་། ཨུཏྟཱི་ཥ་བྱ་བ་ལོ་ཀི་ཏེ། ས་མནྟ་མུ་ཁེ། །ས་མནྟ་བྱ་བ་ལོ་ཀི་ཏེ། མ་ཧཱ་མཱ་ཡེ། མ་ཧཱ་པཱ་ཤ་དྷ་རེ། ་ཨ་མོ་གྷ་བི་མ་ལེ། ཨཱ་:ཀརྟ་ཡ། ཨཱ་:ཀརྟ་ཡ། ཨཱ་ཀཟླུ་ཡ་ཨཱ་ཀརྞཱི་ཡ། བྷ་ར་བྷ་ར། སཾ་བྷ་ར་སཾ་བྷ་ར། ཨིནྡྲ་ཡ། བི་ཤོ་དྷ་ནི། བྷུ་ཥི་ཏ་:བྷུ་ཛི། མ་ཧཱ་མུ་དཱ་བི་ལོ་ཀི་ཏེ། ཛ་ཡ་ཛ་ཡ། སིངྒེ་སིདྡྷེ། ་་:བུདྡྷེ་བུདྡྷེ། བོ་དྷ་ནི་་བོ་དྷ་ནི། སཾ་བོ་དྷ་ནི་སཾ་བོ་དྷ་ནི། ཤོ་དྷ་ནི་ཤོ་དྷ་ནི། སཾ་ཤོ་དྷ་ནི::སཾ་ཤོ་དྷ་ནི། བི་ཤོ་དྷ་ནི་བི་ཤོ་དྷ་ནི། ཧ་ར་ཧ་ར། མ་མ་སརྦ་པཱ་པ། སརྦ་ཏ་ཐཱ་ག་ཏ། ཀུ་ལ་བྷུ་ཛེ། ས་མ་ཡ་ཏིཥྛ། ་པྲ་ས་རནྟུ། མ་མ་པུཏྱཾ། བི་ན་ཊུ་ཏུ་པཱ་པཾ། སརྦ་ཀིཝི་ཥཧ་རེ་མ་ཎི་བི་ཤུདྡྷེ། ཌ་ཤོ་དྷ་ཡ། བི་མ་ལེ། ཎབི་ཀ་སི་ཏ་་་་པདྨེ། ཀ་བ་ཙི་ཏ་བྷུ་ཛེ། ཥཊཱ་ར་མི་ཏཱ། ་པ་རི་པཱུ་ར་ཎི། སརྦ་ཏ་ཐཱ་་ག་ཏོཏྟཱི་ཥ་བི་མ་ལེ་བི་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། སརྦ་ཏ་ཐཱ་ག་ཏ་གུ་ཧྱ། ཨ་:དྷིཥྛ་ན་ཨ་་དྷིཊྛི་ཏེ་སྭཱ་ཧཱ། ཨོ་ཨཱ་ཡུརྡ་དེ་སྭཱ་ཧཱ། །པུ་ནྱད་དེ་སྭཱ་ཧཱ། ་པུཉྩ་བི་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། པུརྱ་ཨ་བ་ལོ་ཀི་ཏེ་སྭཱ་ཧཱ། སྲྀཏུ་དཎྜི་སཱ་ཧཱ། ཡ་མ་དནྡྷེ་སྭཱ་ཧཱ། ་བ་ཡ་མ་རཱུ་ཏི་སཱ་ཧཱ། སཾ་ཧ་ར་ཎི་སྭཱ་ཧཱ། སཾ་བྷ་ར་ཎི་སྭཱ་ཧཱ། སནྡྷ་ར་ཎི་སྭཱ་ཧཱ། །པྲ་ཏི་ས་ར་ཎི་སྭཱ་ཧཱ། ལོ་:ཨོ་ཇོ་བ་ཏི་སྭཱ་ཧཱ། ཏེ་ཛོ་བ་ཏི་སྭཱ་ཧཱ། ཌ་་་ཛ་ཡ་བ་ཏི་སྭཱ་ཧཱ། པེསརྦ་ཏ་ཐཱ་ག་:ཏེ་མུ་དྲ་ཨ་དྷིཥྛ་ཎ་ཨ་དྷིཥྛ་ཏེ་སྭཱ་ཧཱ། མཆོད་རྟེན་དེ་དང་། གདུགས་དེ་ལས་ཡེ་ཤེས་ཀྱི་འོད་ཟེར་སྣ་ཚོགས་ཕྱོགས་བཅུར་འཕྲོ་ཞིང་འཁྲིགས་པ་:བྲེ། དེའི་གཞིའི་དབུས་སུ་དར་རམ་ཤོག་ཤོག་གང་འབྱོར་པ་ལས། སྲིད་དུ་མདའ་གང་ཙམ་ཞེང་དུ་ཁྲུ་གང་ལྷག་ཙམ་ལ། ་གཟུངས་གཉིས་ཀ་རྫོགས་པར་བྲི་ཞིང་འོད་ཟེར་འཕྲོ་བ་དང་ལྡན་པར་བྱས་ཏེ། དབུས་ཀྱི་མཆོད་རྟེན་རིང་བསྲེལ་ཅན་དེ་ལ་མི་རེག་ཏུ་གདགས། ལ་དེ་ནས་དཀྱིལ་འཁོར་དེའི་སྟེང་ལྕི་བ་་་Hr—།
