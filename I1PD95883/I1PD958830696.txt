ཨོ་བློ་བནྡྷ་བནྡྷ་བདག་ལ་རཀྵ་རཀྵ་སྭཱ་ཧཱ། ཨོཾ་ཧཱུྃ་ཏྲོ་བནྡྷ་བནྡྷ་བདག་ལ་རཀྵ་རཀྵ་སྭཱ་ཧཱ༑ཨོཾ་བཛྲ་པཱ་ཎི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་རཏྣ་ཏ་ཐཱ་ག་ཏོརྞཱི་ཥ་ཨ་ན་ཨ་བ་ལོ་ཀི་ཏེ་མཱུརྟ་ནི་ཏེ་ཛོ་:ར་ཎི་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་པདྨ་ཛྭ་ལ་ཛྭ་ལ། དྷ་ཀ་དྷ་ཀ། དྷ་ར་དྷ་ར། བི་ད་ར་བི་ད་ར། :ཚིནྡ་ཚིནྡ། བྷིནྡ་བྷིནྡ། །ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ཀརྨ་སརྦ་ཏ་ཐཱ་ག་ཏོརྞཱི་ཥ་སི་ཏཱ་ཏ་པ་ཏྲེ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། རིགས་ལྔའི་གཙུག་ཏོར་རོ༑། ཏདྱ་ཐཱ། ཨ་ན་ལེ་ཨ་ན་ལེ། ཁ་ས་མེ་ཁ་ས་མེ། བི་ར་བི་ར། སོ་མེ་སོ་མེ། སརྦ་བུདྡྷ། ཨ་དྷིཏྟཱ་ན་ཨ་དྷིཥྛ་ཏེ་སརྦ་ཏ་ཐཱ་ག་ཏོརྞཱི་ཥ་སི་ཏཱ་ཏ་པ་ཏྲེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཧཱུཾ་མ་མ་ཧཱུཾ་ནི་སྭཱ་ཧཱ། གཙུག་ཏོར་དཀར་མོའི་འོ། །ཨོ་:ཊཾ་ཤི་ག་ཎ་པྲ་ཤསྟེ་སརྦ་ཏ་ཐཱ་ག་ཏོརྞཱི་ཥ་སི་ཏཱ་ཏ་པ་ཏྲེ་ཧཱུཾ་ཧཱུཾ། ཛམྦྷ་ནི་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། མ་ཧཱ་བིདྱ། སཾ་བྷཀྵ་ཎ་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། པ་ར་བིདྨ་སཾ་བྷཀྵ་ཎ་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། །སརྦ་དུཏྟཱ་ནཱན། སྟམྦྷ་ནི་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། སརྦ་ཡཀྵ་རཱཀྵ་ས་གྲ་:ཧཱནཱན། བི་དྷཱན་ས་ན་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། ཙ་ཏུར་ཤཱི་ཏཱི་ནཱན། །གྲ་ཧ་ས་ཧ་སཱ་ཎཱ་མ་བི་དྷུན་ས་ན་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། ཨཊྚ་བི་ཤ་ཏཱི་ནཱན། ནཀྵ་ཏཱ་ཎཱ། པྲ་སཱ་དྷ་ན་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། ཨཏྟཱ་ནཾ་མ་ཧཱ་ག་ཧཱ་ཎཱི། བི་དྷཱན་ས་ན་ཀ་རཱི་ཧཱུཾ་ཧཱུཾ། པདྨ་འདབ་བརྒྱད་གདན་སྟེངས་སུ། །རིམ་བཞིན་གཙུག་ཏོར་བཞུགས་སུ་གསོལ། །གྲུ་ཆད་མཆོད་པའི་ལྷ་:མོ་བཞི། །སྒོ་བཞི་སྒོ་བ་ཕུར་བུ་བཞི། །རིག་སྔགས་ཕྲེང་བས་བསྐོར་ཏེ་གཟུགས། །སྒོ་བཞིར་རལ་གྲི་མེ་ལོང་དང་། །བུམ་པ་མདའ་དང་རིམ་བཞིན་བརྒྱན། །ཁ་དོག་ལྔ་ཡི་རྒྱན་རྣམས་དང༌། །ཡུངས་ཀར་ལ་སོགས་འབྲུ་ལྔ་བཀོད། །དེ་ནས། རིག་་་་་སྔགས་གཅོད་ཅིང་མཚམས་བཅད་དེ། །ཡན་ལག་བདུན་པའི་སྨོན་ལམ་གདབ། །ཕྱོགས་བཅུའི་བདེ་གཤེགས་ཐམས་ཅད་ལ། །བདག་ལ་དབང་བསྐུར་སྨོན་ལམ་་་་་གདབ། །གནང་བ་ཐོབ་ནས་སྔགས་འདི་བརྗོད། །ཨོཾ་མ་ཧཱ་ཤུ་ནྱ་ཏཱ་ཛྙཱ་ན་་་་་—།
