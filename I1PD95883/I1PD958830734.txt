ཨོཾ་བཛྲ་པཱ་ཤི་ཀྲྀ་ཏ་ཧྲི། །ཨོཾ་བཛྲ་:པཱ་ཤི་བྷཾ་་ཏ། ་ཨོཾ་བཛྲ་པཱ་ཎི་། མ་ཧཱ་བ་ལི་དྷི། ཨོ་བཛྲ་མཱ་ལེ་མཾ། ཨཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི་སྭཱ་ཧཱ། །ཨོ་ཛཱ་ལ་ཛཱ་ལ་ཧཱུྃ་ཕཊ་:སྭཱ་ཧཱ། །ཨོ་ཊཀྐི་ཧཱུཾ་ཛཉྷིསྭཱ་ཧཱ། རིམ་པ་གསུམ་ལ་ཨོཾ་བཛྲ་:ཧཱི། ཨོ་སུ་བཎ་ཀེ་ཤི་:ཀྲི་ཏ། ཨོ་བི་ག་ལེ་ལཾ། ཨོ་:པི་ཏུདྨ་ལི་ཛ*ཧཱུྃ། ་ཨོ་ཤཱནྟི་ཧཱུྃ། ་་ཨོཾ་པུཎྜི་བ་ལ་བརྡ་ནི་ཧ། ཨོཾ་སུཾ་བྷ་ཀ་རི་ཧཱུྃ། ཨོཾ་མ་ཎི་དྷ་རི་ཧཱུཾ། ཨོཾ་རཏྣ་མཱ་ལ་དྷ་རི་ཧི། ཨོ་་མ་ཧཱ་པྲ་ཏི་སཱ་རེ་པཾ། ཨོཾ་དྷུ་རུ་དྷུ་རུ། མུ་རུ་མུ་རུ། ཀུ་རུ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ། ཨོ་་བ་ལ་བནྡྷ་ནི་ཡོ་ཧ། ཨོཾ་རུ་རུ་ཙ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་མ་ཎི་དྷ་རི་བཛྲ་ཎི་མ་ཧཱ་པྲ་ཏི་ས་རེ་ཆེ་གེ་མོ་རཀྵ་རཀྵ་ཀུ་རུ། ཨོ་ཨཱཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། རིམ་པ་བཞི་པ་ལ། ༀ་ཏཱ་རེ་ཏཱཾ། ཌཨོཾ་ཨ་ར་སྱ་ཀ་ཛ་ཏི་ཛྙཱ། ཨོཾ་བུདྡྷ་ཀ་རི་ཏེ་ཧེ་ཧཱུྃ༔ ོཨོ་ཀ་པཱ་ལི་ནེ། །ༀ་ལཾ་ཀེ་ཤྭ་རི་ཁཾ། ཨོ་དྷ་ནྱ་ལཾ་ཀི་ཤི། ཌ་ཨོ་ལཾ་ཀི་་ཤཱ་རི་པུཾ། ཨོ་ཧ་རི་ཏྲེ་:ཧཾ་ཧཱུཾ། ཨོཾ་པཉྩི་ཀེ་ཀྲེ། ཨོ་ལི་ཁེ་སཾ་དུ། །ཨོ་་དནྟི་ནི་ཀུ་ཏ། །ཨོཾ་ཤྲཱི་ཡེ་དེ་བི་ཤཱི། ཨོཾ་:སྭ་ར་སྭ་ཏི་ཐཾ། ཨོཾ་རཀྵ་རཀྵ་ཆེ་གེ་མོ་སརྦ་བྷ་ཡེ་ཧྥུ་་་བཻ། ཨོ་དི་བ་པདྨེ་མཾ། །ཨོཾ་མ་ཎི་དྷ་རི་བཛྲ་ཎི་མཱ་མ་ཀཱི་མཾ། ཨོ་བྷཾ་་ཀུ་ཊི་:བྷཾ། བྷཱུ་ཧཱུཾ་ཨ། དེའི་ཕྱི་རོལ་དུ་སོར་གཉིས་ཚད་ཀྱི་རིམ་པ་གཅིག་བྱས་ལ། དེར་རྡོ་རྗེ་གྱེན་དུ་བལྟས་པ་སུམ་ཅུ་རྩ་གཉིས་བྲིས་ལ། རྡོ་རྗེ་གཉིས་ཀྱི་མཚམས་སུ་ནུབ་ཕྱོགས་ནས་རིམ་གྱིས་པུཥྛ། :：ལིཀཾ་ཀཱི་ཧཱུཾ། ཨི་ཧཱུཾ། ཧཾ། །ཀ་ཁཾ། ཧ་བོ། ཧྲི་ཛ་ཧཾ། ཧྲི་རི། ཧཱུྃ་ལི། ཧི་སུ། ཧཱུྃ་ཛ། ཐཾ་སུ། ཧོ་ཧཱུྃ༑ཨོཾ་ཕཊ། ཛཾ་སྨ། ཀཾ་ཐཾ། ༀ། སུཾ། ཨོ། བྷཾ། ཐཾ། ཨོ། ཏྲི༑ཤྲཱི། དྷི། །ཛཾ་རུ། རུཾ། ཨོ། སཾ། དེའི་ཕྱི་རོལ་དུ་སོར་་་་་གཉིས་ཀྱི་ཚད་ཀྱི་རིམ་པ་གཅིག་བྱས་ལ། དེར་སྡོང་བུ་གོས་དང་རས་ཀྱིས་དཀྲིས་པའི་པདྨ་ཁ་ཕྱེ་བ་བཞི་བཅུ་རྩ་གཉིས་བྲིས་ལ། དེར་ནུབ་ཕྱོགས་ནས་རིམ་གྱིས་འ__།
