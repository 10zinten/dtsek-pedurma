བྱ་ཁྲིའི་བསྡུར་མཆན། རོ། དྲི། aཚོགས་པ་ཚུ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཁྲོ་ཆེན་གཏུམ་པོའི་སྒྲུབ་ཐབས་བཞུགས་སོ། ། ོ ི«པེ་»«སྣར་ན། ལ༽/ཀྱི་བར་ལSལ་«པེ་»«སྣར་—ཉ། ་«པེ་»«སྣར་»བཞུགས། @«པེ་»«སྣར་»«ཅོ་»ཧཱུྃ། ༄༅«པེ་»«སྣར་གྱིས། གུ«པེ་གདུགས། «པེ་»«སྣར་མཐོ། «པེ་»«སྣར་པས། «པེ་»«སྣར་བསྐྲགཉེ«པེ་»«སྣར་བསོད། «པེ་»«སྣར་»གིར་ཏིས། «པེ་»«སྣར་རྗེས། ཀྱང་མ«པེ་《སྣར་བརྩོད་པ་སྙན་དང༌། སྤུས་དེ། །ངོ་ན་གང་དག་བྱེད་པ་དང་ཞpལོ་ਆਉਸਪਨਮਨਨੁਸਾਲਕੁਧਨཀལ། ཀོ་ལ་འབབ་པ་བསྡུར་འབྲས་རེའུ་མིག་ལ་ཁང་པ་ཐོག་ཉིས་བདག་གནང་བ་ལ་་བབསྐྱེབར་ཁ་ཆེ་བ༌རྒྱུད། （ས་འབབས། བདག་དང་འགྲོ་བ་པ་དང་། པའི་དཔེ་གཞི། །སྡེ་དགེ། ད་པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། །ལ་སྡེ་ཚན། རྒྱུད། །ཀ་རྒྱུད་འགྲེལ། རྒྱུད། །རྒྱུད། ང་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བ་—ན་བ་_ན། བ་ན?ནན་ཕྲེང༌། །།།ཚེག་ཁྱིམ། །།།་་་་་་་་་་་་་་་དང་། དྲང་བའི་དམ་པ་ཁྱོན་ན་འདས་པ་ལ་བ་དང་དབང་འབྱོབ་དང་ར་ལུག་དང་། བོན་གྱི་ལོ་འདོད་ལྟ་ལ་སོགས་པ་དག་ལ་ད་དང་། ་པ་པོ་དང་ཀློབ། ས་པའི་དྲ་བས་།
