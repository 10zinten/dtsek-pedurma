རྒྱུད། ༼པུཀྱང་དབུལ་བགྱི་ཞིང་། །དུས་ཀུན་དུ་ནི་བསྲུང་བར་མཛོད། །རྣམ་པ་ཀུན་ནི་་་་་བདག་དང་ནི། །འཁོར་བཅས་མ་ལུས་སེམས་ཅན་གྱི། །ཉེ་བར་འཚེ་ཉིད་ཐམས་་་ཅད་དང༌། །འཇིགས་པ་ཐམས་ཅད་ཉམས་པར་གྱིས། །དེ་བཞིན་རྟག་ཏུ་མྱུར་བར་ནི༑། འཆི་ལྟས་ཀུན་དང་ཆེར་འབབ་པའང་། །ལོ་བརྒྱ་རུ་ནི་འཚོ་གྱུར་ཅིག། རྟག་ཏུ་སྟོན་བརྒྱ་མཐོང་བར་ཤོག། རྟག་ཏུ་གྲུབ་པའི་སྔགས་ཀྱི་གཞི། །བདེ་་་་ལེགས་དང་ནི་གཞི་ཚུགས་གྱུར། །སླར་ཡང་མཆོད་དེ། མ་རྙེད་ཅེས་པ་ལ་སོགས་་པས་བཟོད་པར་གསོལ་ཏེ་གཤེགས་སུ་གསོལ་བར་བྱའོ། །བསྲུང་བ་ལྔའི་:མཆོད་པའི་ཆོ་ག་རྫོགས་སོ། །འོ་། ཀ་བ་ཏུ་ཞི་ཚོགས་པའི་ན་ཟླཏརྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ། ནགས་ཀྱི་རིན་ཆེན་གྱི་ཞལ་སྔ་ནས་དང་། གཞོན་ནུ་དཔལ་གྱིས་བསྒྱུར་བའོ། །ཊ། །འཁོལ་བཊིཀབ་ད་ཤ་བའཁ་ཡོད། སྐྱོ་ཊ། །ཁྲོ་རྒྱུ་དེ་ཁྲི་བློ་ཟློག་མའི་བྱ་བ་ན། །རྩེ་ཆཆོ་ག་སྦྱོང་ལ་བའི་འོག་དང་བྲག་ཀའབཟོ་ཁོ་རང་ਕਿਸੰਨਬਕਕਬੂਤਰਉਪਲਸਛਾਊਸਕਰਕੇའཚོ་བའི་འཛིན། །ང་རྒྱུ་དང་། ཎ་་་་ང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །བསྲུང་བ་ལྔའི་མཆོད་པའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་གནག«པེ་»«སྣར་—པ། «པེ་»साम«སྣར་མ། །«པེ་»«སྣར་»ཉ། @«སྣར་»ཏ། «པེ་»«སྣར་»བཱདྨ། «པེ་»«སྣར་བཟུང་། ་། («པེ་»སྣར་»ནས། «པེ་»«སྣར་»སཏྟཱ། «པེ་»«སྣར་»ཨཱ། «པེ་»«སྣར་»ནི་ཨ། །།ཚོཀྱེ«པེ་»«སྣར་ཡ།  ེ«པེ་»«སྣར་»ཏཱ། «པེ་»«སྣར་བདྡྷ། «པེ་»«སྣར་»དྲོ། ང«པེ་»«སྣར་»ཀྲྀཏྟིཾ། «པེ་»«སྣར་»ཡ། ་«པེ་»«སྣར་ཨཱ་དྲ། «པེ་»«སྣར་པུརྦ་ཕཱ་ལྷུ་ཙྩཾ། «པེ་»«སྣར་»ཕཱ་ལཱ་གུ་ཙི། ཁྱེ«པེ་»《སྣར་པུཏྲ། ེ«པེ་《སྣར་»ཎ། «པེ་»«སྣར་པུརྦ། «པེ་»«སྣར་»དྲ། ཨེ«པེ་»«སྣར་»ནི་རི། ་ལེ«པེ་»«སྣར་བ། སྐྱེ«པེ་»«སྣར་བྱེ། འགལ་བ་དང་དུ་བ། «པེ་»«སྣར་»ཁོར། Q«པེ་»«སྣར་»བསྒོམས། «པེ་»«སྣར་»ཧཱུཾ། «པེ་»«སྣར་ཁང་། །་«པེ་»«སྣར་»—ལ། «པེ་»«སྣར་»+ཕྱག«པེ་»«སྣར་པོ། སྤྱི«པེ་»«སྣར་པ། ་བ་།
