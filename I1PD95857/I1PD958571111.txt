བསྟན་འགྱུར། རྒྱུད། འ་པེ་»«སྣར་རྟག«པེ་»ཏིཥྛ། «སྣར་ཏིཥྛ། །པེ་《སྣར་བཟུང་། ས་གོ་བ་སྐྱེས། གཏོ་འདཞུ་ཀ་ད་ལ། །རྗེ་ཐག་་ན་ལ་སྤྱོམ་ཟླ། ཞེས་བ་གོ་ཀ་ཏཚོན་ན་ཆེས་ཁ༌པར་བྱེད། གང་ཚེ་འགན། བླ་as/ཡིག་ཁ་བ་དང་བུམ་ལ་འགལ་gམཁས༌པ་ཁ་ཚ་བསྡུར་འབྲས་རེའུ་མིག་བྦོ་་པ་དག་འགལ་ཟླ་དུས་དང་རྒྱུད། (ཆུ་ཀླུ་བ་ཀེ་རེ་ལ་ཆུ་གང་མང་། །དཔེ་གཞི། སྡེ་དགེ། །།པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། ་སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། དཔལ་པོད་རྟགས། ཚན་གྲངས། 。་ལྡེབ། །བསན། བ?ན་ཕྲེང་། །།།ཚེག་ཁྱིམ། །།།བ?ན། བ། ན། །བངས་ལ་ཁྲལ་མོ་ལུང་བ་བ་ན་ཚོང་བ་ཡང་བགྱི་བ་སུ་ལུ་ཁ་བལུགས། ཚོགས་པ་དང་བར་དང་། བལ། ལྔ་པ་བ་དང་། །བླ་་དབུ་། །།་ངག་དང་། དྲ་རྒྱའི་སྐྱེ། དང༌རྒྱུས་ཀྱི་ངོ་:་པལཀོསgs/z་ནིཔཉི་ཏུང་ཚོ་དདརz་པ%Ea་པས་འབབ་པ། ་་་་་་་་་་་བསམ་བཞག་ལ། ག་་་་་་ཡིན་པ་དང། སྐྱེ་བོ་དག་བསྡུ་ན། །ཀྱེ་་་་་་་་་་་་་་་ཞེ་ག་ཅི་དེ་་་་་་་་། །དག་པའི་སྡུག་བདེ་བ་དང་བྱ་བ་དེ་ལུང་པ་ཀུན་། jp། །བདག་དང་། གང་《་དབང་། དང་། ལོ་རྗེའི་བདག་ཉིད་དུ། །འ་དང་འགལ་བ་དང་། བདག་བཤད་། ཁོ་ར་དང་། །འཁྲུལ་བ་《»། papཆ་བསྒྱུ༌@་ཁ་ན།
