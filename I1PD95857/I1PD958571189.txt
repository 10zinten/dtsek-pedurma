བསྟན་འགྱུར། རྒྱུད། ༼འ་སུམ་དུ་བྱིན་གྱིས་བརླབས་པའི་གྲུབ་པའི་དབང་ཕྱུག། རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ་ཤྲཱི་བ་ན་རཏྣའི་:ཞལ་སྔ་ནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།བ། །་ཏཏོ། ད་པཎྜི་ཏ་ཆེན་པོ་དེ་ཉིད་དང༌། ་ཆོས་འཁོར་:ལོ་ཙཱ་བ་འཇམ་དཔལ་ཡེ་་་་ཤེས་ཀྱིས་འབྲས་ཡུལ་ནོར་བུའི་གླིང་དུ་བསྒྱུར་བ་ལས། ཕྱིས་མི་འཕྲོགས་པའི་་་་མཁྱེན་པ་དང་དད་པ་རྣམ་པར་དཀར་བའི་ལུགས་གཉིས་ཀྱི་དབང་ཕྱུག་དམ་པ་ཨ་མོ་གྷ་སིདྡྷིས་བཀས་གནང་བ་ལས། པཎྜི་ཏ་ཆེན་པོ་དེ་ཉིད་ཀྱི་བཀའ་དྲིན་གྱིས་འཚོ་བའི་དགེ་སློང་ཆོས་སྨྲ་བ་རྒྱ་མཚོའི་སྡེ་ཞེས་བྱ་བས་མི་བཅོས་སུ་མི་རུང་བ་་་་་་རྣམས་བཅོས་ཤིང་དོན་ལ་རྟོན་པ་ཉིད་དུ་གཏན་ལ་ཕབ་པ་འདི་ནི། ལྷོ་ཕྱོགས་ཀྱི་རི་བོའི་རྒྱུད་རིན་པོ་ཆེའི་གླིང་སྐལ་བཟང་དགའ་བའི་ཚལ་ཞེས་བྱ་བའི་་་་་་མདུན་ས་དམ་པར་གྲུབ་པར་བགྱིས་པ་:ཤུ་བྷམ་སྟུ་སརྦ་ཛ་ག་ཏཾ། ཀྱི་མིག། གཏི་པོའི། ན་པོ། བ་ད་ན་པ་ཏིང་ཏེ་རྗེལ་བར་དུ་བདག་ལ་དག་བུ་ཚོད་ཚེ། དེ་དག་གོ་བའི་ནད་ལ་མ་འབད་དེ་ན་ཡོད་པར་པ་སྐྱེ་བ་དང་བློ་འཚོ་བའི་བསྡུར་མཆན། གཏོ་སྐྱེལ། ས་སུ་ག་བ་ང་མ་ཁོ་ལ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»རྡོ་རྗེ་ཕག་་་མོའི་བསྟོད་རིམ་གཉིས་པ་དགེ། ་«པེ་»«སྣར་དྭ། «པེ་»«སྣར་»བདེ་བས། ཚ«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་བཞག(«པེ་《སྣར་པ། «པེ་»སྣར་ཕྱོགས། «པེ་《སྣར་མཁྱུད། «པེ་»«སྣར་རོའི། སུ«པེ་»«སྣར་»འདི། @«པེ་»«སྣར་འདྲའི། ཚེ«པེ་»སྣར་»པ་སྙམ། («པེ་»«སྣར་»འཆང་། «པེ་»«སྣར་ལྕི། «པེ་»«སྣར་འདི། དགྱེ«པེ་»«སྣར་ནས། ་ཀྱི«པེ་»«སྣར་ཡུལ། QQ«པེ་»«སྣར་»སྟོད། «སྣར་པས་ཅིལ། །«པེ་»《སྣར་ཟེའུ། «པེ་»«སྣར་»བསྐྱོད། «པེ་»«སྣར་»གི། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་བསན། དགྱེ«པེ་«སྣར་བུར། «པེ་»«སྣར་»དབང་། ཉེ«པེ་»«སྣར་»སྤོ། སྟེ«པེ་»«སྣར་»དོན། ཨེ«པེ་《སྣར་»མཚོར། «པེ་»«སྣར་»རིམ། །::::།
