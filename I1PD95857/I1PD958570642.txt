བརྗོད་ཅིང་། མཆོད་པའི་ཡོ་བྱད་ལྔ་དང་བཅས་པའི་གཏོར་མ་མཚན་ཕྱེད་ན་་་་་དབུལ་བར་བྱ་སྟེ། གཏོར་མ་བྱིན་པ་འདི་ཡིས་ནི། །རྟག་ཏུ་བདུད་སོགས་ཞི་བར་བྱེད། །ཉེ་བར་བསྟན་པ་ཇི་ལྟ་བ་བཞིན་དུ་གསང་ནས་མཆོད་པ་བྱས་ན་་རྡོ་རྗེ་རྣལ་འབྱོར་མ་འགྲུབ་པར་འགྱུར་རོ། །དེ་ལྟར་བསྙེན་པ་བྱས་ནས་དངོས་་་གྲུབ་གྲུབ་པའི་དུས་སུ་ནི་ཚོགས་པ་:ཚང་བ་ཅན་གྱིས་ཉེ་བར་བསྟན་པ་ཇི་ལྟ་བ་་་བཞིན་དུ་ཐམས་ཅད་བྱས་ན་དེ་ལ་རྣལ་འབྱོར་མས་བྱིན་གྱིས་བརླབ་ཀྱི་གཞན་དུ་ནི་མ་ཡིན་ནོ། །དོན་ཐམས་ཅད་གྲུབ་པའི་སྒྲུབ་ཐབས་སློབ་དཔོན་མཁས་པ་གྲུབ་པ་བརྙེས་པ་ཨ་བ་དྷཱུ་ཏཱི་པས་མཛད་པ་རྫོགས་སོ། །།།བལ་པོའི་ཨཱ་ཙཱརྱ་བྷ་རེནྡྲ་རུ་ཙི་དང་། :：ལོ་ཙཱ་བ་བློ་ལྡན་ཤེས་རབ་ཀྱིས་་བསྒྱུར་བའོ། །།།འད། །འོ་ཞེས་བྱ་བ། བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»རྡོ་རྗེ་དོན་ཐམས་ཅད་གྲུབ་པ་བཞུགས་སོ། །《སྣར་བསྐལ་པ། «པེ་»«སྣར་གྷགྷ། «པེ་《སྣར་»—བྷཱ་ཏ་ཡ། «པེ་»«སྣར་+། «པེ་»«སྣར་»ལྟར་སྣང་། ་«པེ་»«སྣར་»—ཞལ། «པེ་»«སྣར་»ཀྱི། །«པེ་བཞག«སྣར་ཧི། «པེ་»ཧཱི། 《སྣར་»ཧྲི། «པེ་»«སྣར་བཞག«པེ་»《སྣར་བྱ་བ། «པེ་»སྣར་»དུད། །«པེ་»«སྣར་བསངས། «པེ་»«སྣར་ཚངས་ཅན་གྱི། །《སྣར་བརླབས། «པེ་《སྣར་»ཚོ་ཡོད། །ཕྱི་ནང་གི་ལོ་དེ་དག་ལ། བ་དང་བརྒྱད། ་མ་ལ་བཏུད་ནས། །དཔག་མེད་བབས་པ་པ། འདས་པ་དང་དབང་བླང་བ་དང་འགལ་བ་དང་འགལ་བ་པས་སྐྱེད་བྱེད་ཚུལ་བཞིན་བདག་ལ་དབབ་པ་དང་། །
