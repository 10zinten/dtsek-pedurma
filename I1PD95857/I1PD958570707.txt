རྒྱུད། ༼ཝིཏི་བརྒྱད་ལ་གནས་པའི་ཕྱོགས་སྐྱོང་རྣམས་ལ། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་ལ་སོགས་པའི་་སྔགས་ཀྱིས་ཕྱོགས་ལ། གཡོན་སྐོར་མཚམས་རྣམས་ལ་གཡས་སྐོར་དུ་ཕུལ་བས། །དེ་རྣམས་ཀྱིས་བདུད་རྩི་མྱངས་པས། :ཀྱི་བདེ་བ་ཆེན་པོ་ཐོབ་པར་བསམས་ལ། དེ་ནས་མཆོད་པ་ལ་སོགས་པས་ཡང་དག་པར་མཆོད་པ་དང་། །རྐང་པ་བརྒྱད་་་པའི་སྔགས་ཀྱིས་བསྟོད་ལ་འདོད་པའི་དོན་ལ་གསོལ་བ་གདབ་པོ། །ལྷག་ཆད་ཁ་བསྐང་བར་བྱ་བའི་དོན་དུ་ཡི་གེ་བརྒྱ་པ་ལན་གསུམ་བརྗོད་དོ། ཟ་། དེའི་རྗེས་སུ་ཨོ་བཛྲ་མུཉྩ། ཞེས་བརྗོད་ལ་པདྨའི་ཕྱག་རྒྱ་བསྐོར་བས་དགྱེས་པར་བྱས་ལ། དེ་་་ནས་སེ་གོལ་གཏོགས་པ་སྔོན་དུ་འགྲོ་བས་ལྷག་མའི་གཏོར་མ་བསྡུས་པ་ལ། དམ་ཚིག་གིས་ནི་ཁ་བསྐང་ནས། །།འབར་བའི་ཕྱག་རྒྱས་བཅིང་བར་བྱ། །དྲག་པོའི་ཧཱུྃ་ཡིག་གདོན་བྱས་ནས། །ལྷག་མའི་གཏོར་མའི་སྣོད་དུ་དབུལ། ཎ། ཨོ་སཉྫ་བྷ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་ཨུ་ཙྪིཥྛ་བྷཀྵ་སྭཱ་ཧཱ། ཞེས་གཤེགས་སུ་གསོལ་ལོ། །སློབ་མ་རྗེས་སུ་:གཟུང་བ་ལས་གཏོར་མའི་ལས་ཀྱི་ཆོ་གའོ། །ག་། ་་་ནང་འིག་ག་་་ཊ་ཡོང་ཁག་དུ་དྲག་དེ་ནས་སྦྱིན་སྲེག་ཆོ་ག་བཤད། །རྡོ་རྗེ་རྣལ་འབྱོར་མ་མཆོད་པའི། །ཕྱི་དང་ནང་གི་དབྱེ་བ་ཡིས། །སྦྱིན་སྲེག་ཡང་ནི་གཉིས་སུ་འབྱུང་། །ཕྱི་ཡི་ལས་ཀྱི་དབྱེ་བ་ཡིས། །སྦྱིན་སྲེག་ཡང་ནི་བཞི་རུ་བཤད། །།ནང་གི་སྦྱིན་སྲེག་གཉིས་སུ་ཤེས༑། རོ་དང་ཚིམ་པར་བྱ་བའོ། །ཕྱི་ནི་འདི་རུ་བཤད་པར་བྱ། །རྡོ་རྗེ་་་རྣལ་འབྱོར་མ་སྒྲུབ་པའི། །དེ་བཞིན་ཆོ་ག་ལྷག་ཆད་ལ། །སྦྱིན་སྲེག་གིས་ནི་་་བསྐང་བར་བྱ། །།རྣལ་འབྱོར་ནུབ་ཏུ་ཁ་བལྟས་ལ། །རྒྱན་ཆ་དམར་པོ་ཡིས་་་་་བརྒྱན་ལ། །སྐྱེ་གནས་ལྟ་བུའི་ཐབ་ཁུང་དུ། །ཕྱེད་ནི་བསྐོར་ཏེ་དམར་པོར་བྱ། །སོར་གསུམ་དོར་བའི་དབུས་སུ་ནི། །རྡོ་རྗེ་དམར་པོ་འབར་བ་དགོད། །ཕྱི་།
