རྒྱུད། ཡོགཟི། ཟླ་ཚོ། ནས་ལས་ཀྱི་སློབ་དཔོན་གྱིས་ཚོགས་ཀྱི་སློབ་དཔོན་ལ་སོགས་པ་ཀུན་གྱི་མདུན་དུ་མཎྜལ་བྱ་སྟེ། ་རྡོ་རྗེ་ལག་སྐོར་དང་བཅས་པས་གར་བྱ་ཞིང་། །ཁྱེད་ནི་རྣལ་་་འབྱོར་དམ་པའི་མཆོག། བདེ་བ་ཆེན་པོའི་རོས་གང་བའི། །དེ་ཡི་རྐྱེན་དུ་་་་དབུལ་ལགས་ཀྱིས། །དགྱེས་པའི་མདངས་ཀྱིས་བཞེས་སུ་གསོལ། །དེ་ནས་་་བཟའ་བ་དང་བཏུང་བ་ལ་སོགས་པས་ཚིམ་པ་དང་ཁྱད་པར་དང་ཐུན་མོང་དུ་བྱས་ལ། །རྣལ་འབྱོར་ཕོ་མོ་ཀུན་ལངས་ལ། རྡོ་རྗེའི་གར་དང་གླུ་བླང་བར་བྱ། །།ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་པཱུ་ཛ་ཡཱ་མི། ཞེས་མཁའ་འགྲོ་མ་ལྔ་གླུར་བླང་ངོ༌། ལྔ་། གཞན་་་་ཡང་སྐད་ཅིག་གླུར་བླང་:བ་ནི། ཨེ་མ་ཧོ། ་འོ་དང་འཁྱུད་པ་ལ་སོགས་པ། །རྣམ་པ་སྣ་ཚོགས་སྐད་ཅིག་མ། ་། དགའ་བའི་ལོངས་སྤྱོད་རབ་ཏུ་འབུལ། ་་། རྣམ་པ་སྣ་ཚོགས་དུ་མའོ། །སྒེག་དང་ལྟ་བ་ལ་སོགས་པ། ་། རྣམ་པར་སྨིན་་་་པའི་སྐད་ཅིག་མ། །།མཆོག་དགའི་ལོངས་སྤྱོད་རབ་ཏུ་དབུལ། །རྣམ་པ་སྣ་་་ཚོགས་དུ་མའོ། །བདེ་བའི་སྟོན་མོ་ལ་སོགས་པ། །རྣམ་པར་སྨིན་མའི་སྐད་་་ཅིག་མ། །མཆོག་དགའི་ལོངས་སྤྱོད་རབ་ཏུ་དབུལ། ::། སྣམ་པ་སྣ་ཚོགས་དུ་་་མའོ༑་། །བདེ་བའི་སྟོན་མོ་ལ་སོགས་པ། །རྣམ་པ་མེད་དང་མཚན་ཉིད་བྲལ། །དགའ་བྲལ་ལྷན་སྐྱེས་ལོངས་སྤྱོད་དབུལ། །:རྣམ་པ་སྣ་ཚོགས་དུ་མའོ། ་། མ་ལུས་པ་ཡི་འཇིག་རྟེན་ཀྱང་། །ནམ་མཁའ་ཀུན་གྱི་རྣལ་འབྱོར་པ། ་། ནམ་་་་་མཁའི་མཚན་ཉིད་ལ་ཞུགས་ཤིང་། །ཐམས་ཅད་མཁའ་ལྟར་ལོངས་སྤྱོད་པའོ། །དེ་རྣམས་ཀླུར་བླང་བའོ། །དེ་ནས་ལྷག་མ་བསྡུས་ལ་མཁའ་འགྲོ་མའི་གཏོར་མ་་་དང་ལྷན་ཅིག་ཏུ་བཞག་ལ། ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་བ་ལཾ་པྲ་ཏཱིཙྪ་ཧོ། །ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་བྷུཉྫ་ཧོ། །ཞེས་ལྷག་མ་བསྡུས་ལ་གཏང་ངོ༌། བར། དེ་ནས་ཆོས་སྐྱོང་གི་གཏོར་མ་རང་གི་ཆོ་ག་ལྟར་གཏང་ངོ༌། །ཁྲེ་དང་། །གཏོར་མ་གསུམ་གཏོང་བའི་ཚེ། །
