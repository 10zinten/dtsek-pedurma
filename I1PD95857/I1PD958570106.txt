ཕཊ༑ཨོཾ་བཛྲ་རཱུ་པ་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པ་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤབྡ་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤབྷ་བཛྲ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་བཛེ་ལཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་བཛྲ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ས་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་རཱ་ས་བཛྲ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སྤར་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སྤཆུ་ཀྱི་བཛྲ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨ་དྷཱ་ཏུ་བཛེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨ་དྷཱ་ཏུ་བཛྲ་ཧུ་ཧཱུྃ་ཕཊ། ་ཨོ་བཛྲ་ཀྵི་ཏི་གཉྷི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྱི་ཏི་གནྡྷ་ཧུཾ་ཧཱུྃ་ཕཊ༈་ཨོཾ་བཛྲ་ཁ་གཌྷ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཁ་གཏྣ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་པཱ་ཎཱི་ཡེ་ྋཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་པཱ་ཎི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཀ་ནཱ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཀ་ནཱ་ཐ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སརྦ་ཨ་བ་ར་ཎ་ནིཏྟཾ་བྷི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སརྦ་ཨ་བ་ར་ཎ་ནིཏྟཾ་བྷ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ས་མནྟ་བྷ་དཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ས་མནྟ་བྷ་དྲ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཌོ་ལི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཏྟཱ་ལི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནི་རཱདྷུ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ::ཨོ་བཛྲ་ནི་རཱཏྨ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྲཱི་ཀུ་ཊི་ཀཱི''ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྲཱི་ཀུ་ཊི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཎ་ཤ་བ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཎ་ཤ་བ་རི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མཱནྟ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲཛྙཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲཛྙཱནྟ་ཀ་ཧུཾ་ཧཱུྃ་་་ཕཊ༑ཨོཾ་བཛྲ་པདྨནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པདྨནྟ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་བིགྷཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བིགྷཱནྟ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཙ་ལཱི་ཡེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཙ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནཱི་ལ་དཎྜཱི་ཡེཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ནཱི་ལ་དཎྜིཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐི་རཱ་ཛཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐི་རཱ་ཛ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་:བ་ལཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་།
