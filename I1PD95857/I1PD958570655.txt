བསྟན་འགྱུར། ་རྒྱུད། ཟི་་རྫོགས་སོ། །ཏག། བཀྱིགས་པར་བྱ་བ་་་་འ། །ཀ་པ་ལ་བལ་པོའི་པཎྜི་ཏ་ཨཱ་ཙཱརྱ་བྷ་རེནྡྲ་རུ་ཙི་དང་། ལོ་ཙཱ་བ་བློ་ལྡན་ཤེས་རབ་ཀྱིས་བསྒྱུར་བའོ། །།།་་་པ་ཞི་བྱཚོ་ཁྲི་ལ་བོའི་སྒོ་བ་ཁོ་ཚོਸਿਉਜੋਤਸਰਿਨਲਾਲਨੂੰਹਾਲੇਵਧਾਉਹੌਲਤੋਂਸཔ་འགའ། ང་རང་ནི། དབྱིན་་་་་་། །m་ཀྱི་ཊ་ཏི་དང་། སྤྱི་བསྡུར་མཆན། :?མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»རྡོ་རྗེ་ཕག་མོ་་དབུ་བཅད་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»《སྣར་གྱི། «པེ་»«སྣར་»དུ། ®«པེ་»«སྣར་»། +དང༌། «པེ་»«སྣར་»+། །ཀ། །ས། ག་ག་ཏཱ་ཡུ་ན། བྱ་དང་བྱཀློ་ཏུ་བོ/«པེ་»«སྣར་»ཀྱི། («པེ་»«སྣར་»—མ། «པེ་»«སྣར་+བ། «པེ་《སྣར་»པ། །«པེ་《སྣར་ཀཱི། «པེ་《སྣར་»—ཀྱི། «པེ་»«སྣར་བྱ་བའི་སྔགས། «པེ་»«སྣར་བྱ་བའི། «པེ་»«སྣར་ཕུལླི། ་@«པེ་ཤི། «སྣར་»ཤི། «པེ་»«སྣར་»ཀྱིས། («པེ་»སྣར་སརྦ། །«སྣར་»ཎི། འཁོ། «པེ་《སྣར་»ཙཱ། «པེ་»«སྣར་བཀྲ་ཤིས། <«པེ་»«སྣར་བར་དུ་ཐུབ་པས། «སྣར་ནི། ཀྱི«པེ་»《སྣར་ཙེ། «པེ་+བཀྲ་ཤིས། ཚེལ་པ། འབོ། །gགཡོན་ཀ་བྱེད། བསྐོར་ལྔ་བ་ནས་བོད་ཀྱི་སློབ་ཐོན་མོ་བྱ་ཁོ་ན་དེ་ཡིན། བསྡུར་འབྲས་རེའུ་མིགབོད་ཀྱི་ལོ་འཁོང་། །རྒྱུད། (དཔ། ཆང་ཉིད་བློས་གཏོང་གནང་དཔེ་གཞི། །སྡེ་དགེ། །སྣར་ཐང་། །ཅོ་ནེ། །པེ་ཅིན། ་། རྒྱུད་འགྲེལ། [བྱེད་སྡེ་ཚན། རྒྱུད། རྒྱུད། རྒྱུད། ་་། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནནན?ནཕྲེང་། ཚེག་ཁྱིམ། །བབོདན་ན། །།
