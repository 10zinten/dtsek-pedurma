རྒྱུད། ༼ཝིཏིངགཞན་ཡང་། དབང་པོ་རབ་མཆོག་ཏུ་གྱུར་པས། །ཟུང་འཇུག་ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ། །སྟོང་པའི་ངོ་བོ་བདེ་བ་ཆེ། །ལ་། ཇི་ལྟར་སྣང་བཞིན་བདེན་སྤངས་པ། :། མེ་ལོང་ནང་གི་གཟུགས་བརྙན་བཞིན། །གཟུང་དང་འཛིན་པ་་་མེད་འགྱུར་ཞིང་། །དམིགས་པ་ཐམས་ཅད་རྣམ་སྤངས་ཏེ། །སྟོང་དང་མི་སྟོང་ལས་འདས་པ། །དེ་ནི་བདུད་དང་ཉོན་མོངས་ཚོགས། །འཇིགས་མཛད་བདེ་བར་གཤེགས་པའི་འཁོར། །འབར་བའི་ཕྲེང་བ་འཁྲིགས་ཤིང་དྲག། ཧེ་རུ་་་་ཀར་གྲགས་ཆེན་པོར་བཞེངས། །བླ་མའི་དྲིན་ཐོབ་སྒོམ་ཕྱིར་དགའ། །ཤུགས་ཀྱིས་ཡེ་ཤེས་རབ་སྣང་བས། །ཕྱོགས་བཅུར་གསལ་བར་བྱེད་པར་ཤོག། རང་བྱིན་གྱིས་བརླབ་པའི་རིམ་པའི་མན་ངག་གི་སྒྲུབ་ཐབས་རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་ཀུ་མཱ་ར་ཤྲཱིས་དཔལ་ན་ལེནྡྲའི་གཙུག་ལག་ཁང་དུ་སཾ་ཀྲྀ་ཏའི་སྐད་དུ་་་་་མཛད་པ་རྫོགས་སོ། །ན། ་་་:ཀའི་་་་་་་་་་་་ན་མར་ཟོསྤྱི་དེ་ནས་ད་བར་། དཀའ་ངལ་ཚུ་འབད་ཚུལ་འཛིན་པ། མི་ག་ཏ་མ་ཀ་ཙཀ་བསྡུར་མཆན། དབུ་མ་ཚུགས་ཚོགས་ཀྱི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»བྱིན་གྱིས་བརླབ་པའི་མན་ངག་གི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»སཱ་དྷིཥྛ་ན་ཀྲ་མོ། «པེ་»«སྣར་»ས། «སྣར་»ནགནསབརླབས༑«པེ་»«སྣར་—། «པེ་»«སྣར་»—ཉ། ས་«པེ་»«སྣར་གིས། «པེ་»«སྣར་མཁྱུད། «པེ་»«སྣར་»+ཅན། «པེ་»《སྣར་»དཔྱངས། «པེ་»«སྣར་»མཁྱུད། «པེ་»«སྣར་»གི། །«པེ་»«སྣར་པ། «པེ་»«སྣར་མོ། ནོ《པེ་《སྣར་ལ། «པེ་《སྣར་»ཧཱུྃ། «པེ་»《སྣར་པར། «པེ་»«སྣར་གྱིས། «པེ་»《སྣར་བསྲུངས། །དེ་་་«པེ་»«སྣར་འཛིན། «པེ་»«སྣར་»བསྒོམ། «པེ་»«སྣར་»+ལ། «པེ་»«སྣར་»བརླབས། ::ཀྱིསུ《པེ་»«སྣར་ཤིས། ་«པེ་»«སྣར་»ལན་དའི། །་་་་་་:ངས་ཀ/བ། །
