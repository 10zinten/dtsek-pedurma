དྲག་ཏུ་འཁོར་བའི་འཁོར་ལོ་ལྟར། །ཕྱག་རྒྱ་གཉིས་ཀྱི་སྦྱོར་བ་ཡིས། །སོ་སོར་བསྡུ་བའི་རིམ་པའོ། །བླ་མའི་མན་ངག་ལམ་གྱིས་ནི། །རིམ་པ་རྒྱས་པར་ཤེས་པར་བྱ། །དེ་ལྟར་བསྟན་པའི་སེམས་ཀྱིས་ནི། །གྲུབ་པའི་རྟགས་ནི་འབྱུང་་་་བར་འགྱུར། །ཡེ་ཤེས་ཐོབ་པའི་སློབ་མ་ལ། །དེ་ཡི་ཚེ་ན་བ་སྤུ་ལྡང་། །མཆི་མ་བྱུང་དང་ཆུ་འཛག་དང་། །ཚ་བ་དང་ནི་གྲང་བ་དང་། །འདར་བ་དང་ནི་གཡོ་བ་དང་། །ཨོ་ཏིཥྛ'ིབཛྲ་ཞེས་བརྗོད་དེ། །བརྟན་པར་གྱུར་ཅིག་ཅེས་་་་་བརྗོད་དོ། །རང་བཞིན་དུ་ནི་གནས་པ་ལ། །བསྒོམ་པ་ཕྱི་ནས་བསྟན་པར་བྱ། །མཆོད་པ་གསང་སྔགས་ལྷ་མོ་ཡི། །ལྷན་ཅིག་སྐྱེས་པའི་བདག་ཉིད་དོ། །ཞེས་་སློབ་མ་རྗེས་སུ་གཟུང་བ་ལ་ཡེ་ཤེས་འབེབས་པའི་ཆོ་གའོ། །བར། །ངོ་pཔ་དང་། །ཐོ་རྗེ་ཚུལ་འབྱོར་མའི་འབསྡུར་མཆན། ཚུལ། བར་ང་ལ་ན་འ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «སྣར་»༄༅། «པེ་»«སྣར་གྱི་ཧ་བིདྷི། ︽«པེ་»«སྣར་བྱ། །ཚུ«པེ་»«སྣར་བཟུང་། @«སྣར་བསྔགས། «པེ་»«སྣར་»+ནི། «པེ་»«སྣར་ལོ། «པེ་»«སྣར་»—སྔོན་དུ། ︽«པེ་»«སྣར་»བོ། «སྣར་»ཨགྷཾ། _@«པེ་»«སྣར་བརླབས། «པེ་»«སྣར་»པ། ཉི«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་བྱུགགྱི«པེ་»«སྣར་»འཁར། ཀྱེ«པེ་»«སྣར་»གྱིས། །(«པེ་»«སྣར་བཞག«པེ་»«སྣར་»བཟུང་། «པེ་»«སྣར་། ︽«པེ་»«སྣར་གཞུག། གྱི«པེ་»«སྣར་»—ལྡན་པ་དང་། ཚེ«པེ་»«སྣར་»ཏེ། ཚེ«པེ་»«སྣར་འཁོར་ལོ་ལ་ཡེ་ཤེས། ཉི«པེ་»«སྣར་»—པ། «སྣར་ཨགྷཾ། དགྱེ«པེ་»«སྣར་བརླབས། འགྱེ«པེ་»«སྣར་པའི། །«པེ་»«སྣར་མ། ལྕེ«པེ་»«སྣར་བརླབས་དུ། མོད་། ་་་་་་ཞུ་«པེ་»«སྣར་གསོལ། ， ེ«པེ་»«སྣར་»གཟུང་། ནུ«པེ་»«སྣར་དད་པའི་ཆུ་ཡི། «པེ་»«སྣར་»གྱི། །«པེ་»«སྣར་བརླབས་བསྒྲུབས། པོ«པེ་»«སྣར་བཟུང་། ཚེ«སྣར་»—རས། «པེ་»«སྣར་ཨོཊྛན། «པེ་»«སྣར་+ནི། གྱི«པེ་»«སྣར་»སྟེངས། ཀྱེ«པེ་»«སྣར་»དེ། །།
