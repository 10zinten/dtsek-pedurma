།ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གིཎྜི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གིཎྜ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་་བཛྲ་ཙ་མ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙ་མ་ར་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མུ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མུ་ཥ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གརྡ་བྷི་ཙིཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་གརྡ་བྷ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷི་ཊཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷི་ཊ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཨ་ཛ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཛ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་ཨེ་ཌ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨེ་ཌ་ྋཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྭ་ནི་ཡེ་ལུ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྭ་ན་@ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སཱུ་ཀ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་སཱུ་ཀ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷཝཱི་ཙྪཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷལླི་ཧུ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་དཎྜ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་དཎྜི་ར་ཧུ་ཧཱུྃ་ཕཊ། ་ཨོ་་་བཛྲ་མོ་ཙཱ་ཀཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མོ་ཛཱ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བེ་ས་རཱི་ཡེ་ཨེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བེ་ས་ར་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ལཱ་སཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ལཱ་ས་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ར་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ར་ཎི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཻ་ཧ་ཤཱ་ནི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྲཱི་ཧ་ཤུ་ནི་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་དྲོ་ཎ་ཀཱ་ཀཱི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྲོ་ཎ་ཀཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། །ཨོཾ་བཛ::ཤ་ར་དྲུ་ལི་ཡེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ::ཤ་ར་རླུ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་་་བཛྲ་བྱ་ཏྲི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྱ་ཌ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙི་ཏྲི་ནི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙི་ཏྲི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨཱུ་ཌི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་:ཨཱུ་ཌ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ཀུ་ལཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ཀུ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྲི་:ཀཱ་ཡི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྲི་ཀཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་གུ་ཧཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གུ་ཧ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གྲྭ་མ་བ་སི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གྲྭ་མ་བ་སི་ྋཧུཾ་ཧཱུྃ་ཕཊ། །་་་། །་་་་་་།
