བ་དང་ཁ་ཏི་ཀ་ར་བསྟན་འགྱུར། རྒྱུད། ཝ་། ཁོང་དུ་ཆུ་ཚོར་བ། དམ་ཀན་ཨོ་བཛྲ་ཙ་ཏུརྨ་ཧཱ་རཱ་ཛ་ཀཱ་ཡི་ཀ་ཙཀྲ་ཝརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཏུརྨ་ཧཱ་རཱ་ཛ་ཀཱ་ཡི་ཀ་ཙཀྲ་ཝརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏྲ་ཡ་ཏྲི་ཤ་ཙཀྲ་ཝརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏྲ་ཡ་ཏྲི་ཤ་ཙཀྲ་ཝརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མི་་་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མི་ཙཀྲ་བརྟི་ཧུ་ཧཱུྃ་ཕཊ། ::ཨོ་བཛྲ་ཏུ་་ཥི་ཏ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏུ་ཥི་ཏ་ཙཀྲ་བརྟི་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ནིརྨ་ཎ་ར་ཏ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནིརཱ་ཎ་ར་ཏི་ཙཀྲ་བརྟི་ཀྱིཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་ར་ནིརྨི་ཏེ་བ་ཤ་བརྟི་ཙཀྲ་བརྟི་:ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །།ཨོ་་བཛྲ་པ་ར་ནིརྨི་ཙེཏ་བ་ཤ་བརྟི་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྲཧྨ་ཀཱ་ཡི་ཀ་ཙཀྲ་བརྟི་ནི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྲཧྨ་ཀཱ་ཡི་ཀ་ཙཀྲ་བརྟི་ྋཧུཾ་ཧཱུྃ་ཕཊ། །ཨོཾ་བཛྲ་བྲཧྨ་པུ་རོ་ཧི་ཏ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྲཧྨ་པུ་རོ་ཧི་ཏ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་བཛྲ་མ་ཧཱ་བྲཧྨ་ཎ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཏༀ་བཛྲ་མ་ཧཱ་བྲཧྨ་ཎ་ཙཀྲ་བརྟི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་ེརཱི་ཏཱ་བྷཱ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་རཱི་ཏཱ་བྷཱ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་པྲ་མཱ་ཎཱ་བྷཱ་ཨཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་པྲ་མཱ་ཎཱ་བྷཱ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཱ་བྷཱ་སུ་ར་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཱ་བྷཱ་སུ་ར་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:པ་རི་ཝེཏ་ཤུ་བྷ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་རཱི་ཏ་ཤུ་བྷ་་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་པྲ་མཱ་ཎ་ཤུ་བྷ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་ཨ་པྲ་མཱ་ཎ་ཤུ་བྷ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། རྗེ་ཨོཾ་བཛྲ་ཤུ་བྷ་ཀྲྀ་ཙཀྲ་བརྟི་ནི་་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤུ་བྷ་ཀྲྀཥྛ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ན་བྷ་ཀ་ཡི་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ན་བྱཱ་ཀ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་་བཛྲ་པུརྱ་པྲ་ས་བ་ཙཀྲ་བརྟི་ནཱིཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པུཎྱ་པྲ་ས་བ་ཙཀྲ་བརྟི་པོ་།
