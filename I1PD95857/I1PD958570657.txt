བསྟན་འགྱུར། རྒྱུད། ཟི་མཆོད་རྫས་དག་ལའང་བསང་གཏོང་བྱ། །སྔགས་དེ་ཉིད་ཀྱིས་ཁྲུས་བྱས་ཏེ། །དེ་ཡི་རྗེས་ལ་གོ་བགོའོ། །ཡན་ལག་ཉིང་ལག་ཐམས་ཅད་ལ། ་། གཡོན་གྱི་ལག་པ་ནས་བརྩམས་ནས། །ཨོ་ཐཾ། ཧཾ་ཡཾ། ཧྲི་མོ་ཧྲེ་ཧྲི། ཧཱུཾ་ཧཱུཾ། ཕཊ་ཕཊ། ཅེས་པའི་སྔགས་ཀྱི་ཡི་གེ་རྣམས། །ལག་པ་གཡོན་པའི་རྩེ་མོ་ཡིས། །དང་པོ་ལྟེ་བ་ནས་བརྩམས་ནས། །།བརྗོད་བྱ་དེ་ནས་དག་གསུམ་མོ། །།ཨོཾ་སྭ་བྷཱ་བ་ཤུདྡྷསརྦ་དྷརྨཱ?སྭ་བྷཱ་བ་ཤུདྡྷོཉ་ཧཾ། ཨོཾ་བཛྲ་ཤུདྡྷསརྦ་དྷརྨཱ་བཛྲ་ཤུདྡྷོཉ་ཧཾ། ཨོ་ཡོ་ག་ཤུདྡྷསརྦ་དྷརྨཱ་ཡོ་ག་ཤུདྡྷོཉཧཾ། དེ་རྗེས་བློ་ལྡན་གྱིས་ང་རྒྱལ། །བསྐྱེད་བྱ་་ཁམས་གསུམ་དག་པའི་ཕྱིར། །ཕྱོགས་མཚམས་མཉམ་པར་ལྡན་པ་ལ། །དུར་་་་ཁྲོད་བརྒྱད་པོ་དེ་བཞིན་དུ། །མྱུར་བའི་རྣམ་པའི་རྣལ་འབྱོར་གྱིས། །དེ་རྗེས་་་་་གསུམ་པོ་མཉམ་པར་གཞག། གནས་གསུམ་ལ་ནི་བསྒོམ་བྱའོ། །འོད་ཟེར་་་་རྣམ་པར་སྤྲོས་པ་ལས། །སྲིད་གསུམ་ལ་གནས་སེམས་ཅན་རྣམས། །གསལ་བར་་་་བྱས་ཏེ་དེ་ནས་ཡང་། །ཉེ་བར་བསྡུ་བའི་སྒོམ་པ་ལས། །སྣ་ཚོགས་ཀུན་དུ་སྟོང་པར་བྱའོ། །དག་པ་ཡིས་ནི་ཤེལ་ལྟ་བུར། །གཉིས་མེད་ཡེ་ཤེས་རང་བཞིན་ནོ། །དེ་རྗེས་སྟོང་སོགས་རང་བཞིན་གྱིས། །རྒྱལ་བའི་སྔགས་ཀྱི་བྱིན་གྱིས་བརླབ། །དེ་ནས་ནམ་མཁའ་བསྒོམ་བྱ་བ། །ཧཱིདམར་པོ་ལས་འདམ་སྐྱེས་སོ། །ཡང་ནི་དེ་སྟེང་བསྒོམ་བྱ་བ། །ཟླ་བ་ཉི་མ་མེ་དེར་བལྟ། །དེ་དཀྱིལ་ཡང་ནི་བསྒོམ་བྱ་བ། །བཻ་ཡིག་སྣ་ཚོགས་མདོག་ལྟ་བུ། །ོ། དེ་ནས་ཐཾ་སྐྱེས་བལྟ་བྱ་བ། །གྲི་གུག་ཐཾ་གྱིས་་བརྒྱན་པའོ། །གྲི་གུག་ཡོངས་གྱུར་བསྒོམ་བྱ་བ། R། སྤྲོས་ལས་བདག་ཉིད་རྣལ་་་འབྱོར་ནི། །རྡོ་རྗེ་ཡི་ནི་རྣལ་འབྱོར་མ། །གཟུགས་གསུམ་ཁ་དོག་གསུམ་ལྡན་མ། །ལྗང་གུ་དམར་སེར་སེར་མོ་སྟེ། །གྲི་གུག་ཐོད་པ་འཛིན་པའོ། །དབུ་སྐྲ་གྲོལ་་་་ཞིང་གཅེར་བུ་ལ། ཌ་། དམར་ཞིང་ཟླུམ་པའི་སྤྱན་གསུམ་པ། །ལང་ཚོ་གཞོན་ནུ་་—།
