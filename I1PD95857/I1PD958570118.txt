ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡཀྵ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷཱུ་ཏི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷཱུ་ཏ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲ་ཏ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཏ་ཧཱུྃ་་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀ་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨ་བཱི་ཙཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་བཱི་ཙིཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཏ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཏ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨནྟ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨནྟ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཀུནྣི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་:ཀུནྡྷ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་སྦི་:ཡཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་པུ་རུ་ཥ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཀཱ་ཝེལ་སཱུ་ཏཱི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ལ་སཱུ་ཏྲ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་ཀུ་ཀུ་ལཱི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཀུ་ཀུ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏ་པ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏ་པ་ན་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲ་ཏ་པ་ནཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཏ་པ་ན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རོ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་རོ་ར་བ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་རོ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་རོ་ར་བ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏི་ལ་པ་ཙཱི་ཡེ་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏི་ལ་པ་ཙ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དཱི་པརྦ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དཱི་པརྦ་ཏ་ཧཱུྃ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་དྭེ་ཏཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྭ་ཥ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མོ་ཧཱི་ཡེཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མོ་ཧ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨིནྡྲི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨིན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱ་གཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ག་ཧཱུྃ་ཧཱུྃ་ཕཊ༑ཏུ་ཨོ་བཛྲ་མ་ད་ན་:མཱ་ན་སཱ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ད་ན་:མཱ་ན་་སཱ་ར་ཡ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་སཱུ་ཏྲིཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སཱུ་:ཏྲི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་ཏ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་ཏ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སི་བཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སི་བཱ་ན་ཨེཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་—།
