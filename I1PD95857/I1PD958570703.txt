རྒྱུད། བཀྱིཔ་ཤེས་བྱས་ལ། །དེ་ཚེ་མཆོད་པའི་ཆོ་ག་བྱ། །ཉེ་བར་སྤྱོད་པ་ལྔ་ཡིས་མཆོད། །ཁྱད་པར་དུ་ནི་ཙནྡན་དང་། །གུར་ཀུམ་ལ་སོགས་ལྡན་པ་དང་། །མེ་ཏོག་ཕྲེང་བ་དམར་པོ་དང༌། །།མཆོད་པ་ལྔ་ཡིས་མཆོད་པར་བྱ། །ལས་ཀྱི་རྡོ་རྗེ་མཁས་པ་ཡིས། ལ། དང་པོ་ལྕགས་ཀྱུ་བཟུང་བྱས་ལ། །སྦྲང་ངམ་བུ་རམ་སྦྱར་བའམ། །ཇི་ལྟར་རྙེད་པ་བཟུང་བྱས་ལ། །ཆོས་འདི་ལ་ནི་ཁྱོད་ལྟོས་ལ། །མེད་པའི་་་་རྣམ་རྟོག་དོར་ནས་ནི། །ཁྱི་དང་བྲམ་ཟེ་གདོལ་པ་དང་། །རང་བཞིན་གཅིག་ཏུ་བཟའ་བར་གྱིས། །ཞེས་ཚིགས་སུ་བཅད་པ་འདིས་དབུལ་བར་བགྱིའོ། །རིན་ཐང་མེད་ལས་ལེགས་འོང་ཆོས། །འདོད་ཆགས་དྲི་མ་དག་པའི་ཕྱིར། །གཟུང་བ་དང་ནི་འཛིན་པ་སྤངས། །དེ་ཉིད་ཀྱི་ནི་ཕྱག་འཚལ་ལོ། པོ། ཞེས་་་དེས་བླང་བར་བྱའོ། །ལས་ཀྱི་རྡོ་རྗེ་མཁས་པ་ཡིས། །ཐུན་མོང་དུ་ནི་དབུལ་་་བར་བྱ། །བཟའ་བ་དང་ནི་བཏུང་བ་ཡང་། །ཉ་དང་ཤ་དང་དེ་བཞིན་དུ། །བཟའ་བ་ལ་སོགས་རྫོགས་པ་ལ། །སློབ་དཔོན་གྱིས་ནི་བྱིན་བརླབ་བྱ། །ཡི་གེ་གསུམ་གྱིས་བྱིན་གྱིས་བརླབ། ་ལོ། ཁ་དོག་དྲི་ཡང་དག་པར་བྱེད། །ཨོ་སཏྠ་བཛྲ་ཀཱ་མི་ཎི་སཉྫ་བྷཁཾ་བི་ཤོ་དྷ་ཡ་གུ་གྱ་བཛྲ་ཎི་ཧཱུྃ་སྭཱ་ཧཱ། ཞེས་པས་སོ། །དེ་ནས་མདུན་དུ་མཎྜལ་བྱས་ལ། །།པུས་མོ་ཕབ་ལ་ཐལ་མོ་སྦྱར། །སེམས་ཅན་ཀུན་གྱི་དོན་དུ་ནི། །དེ་ལ་སྨོན་ལམ་གདབ་པར་བྱ། །གདན་ལ་སོགས་པ་བཅུ་ཡིས་ནི། །ས་དང་ཕ་རོལ་ཕྱིན་པ་སོགས། །དཔའ་བོ་དཔའ་མོ་དབང་ཕྱུག་རྣམས། །གུས་པས་བདག་ནི་ཕྱག་འཚལ་ལོ། །ལྷ་མོ་ཚད་མ་དམ་ཚིག་ཚད། །སྟེ། །དེ་གསུང་ཅི་མཆོག་ཏུ་ཚད་མ་སྟེ། །བདེན་པ་དག་གིས་ཉྩེལྷ་མོ་ཡིས། །དེ་དག་བདག་གི་རྗེས་འཛིན་ཤོག། ཇི་སྙེད་རྡོ་རྗེ་མཁའ་འགྲོ་མ། །ཀུན་དུ་རྟོག་པ་འཆིང་གཅོད་ཅིང་། །འཇིག་རྟེན་བྱ་བ་རབ་འཇུག་མ། །ཇི་སྙེད་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །།
