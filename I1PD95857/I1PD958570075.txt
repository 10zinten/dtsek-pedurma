ཞུ་མཁོ་ཆས་ཚོང་པ་རྗེ་བསྟན་འགྱུར། རྒྱུད། ོཝ་སྐུ་བཅོཎ་། སྐྱ་ཚོ་དཀ་ཆུ་གླང་ཆེན་དང་། སེང་གེ་དང་། ཟླ་བ་དང་། རིན་ཆེན་འོད་ཟེར་དང་། སྒྲ་གཅན་སྦེད་དང༌། །ཡོན་ཏན་རྒྱ་མཚོ་དང་། འོད་ཟེར་ལྡན་པ་དང་། །རབ་ཏུ་ཞི་བའི་སྟོབས་དང་། འཇིག་རྟེན་བཟང་པོ་དང་། །མྱ་ངན་མེད་པ་དང་། ཤུགས་བཅུ་པ་དང་། སྟོབས་ལ་དགྱེས་པ་དང་། མཐུའི་དཔལ་དང༌། མཐུ་ཐོབ་པ་དང་། མཐུ་ཆེན་དང༌། ཡོན་ཏན་སྙིང་པོ་དང་། བདེན་པར་སྤྱོད་པ་དང༌། །དགེ་མཆོག་རྒྱལ་པོ་དང༌། །སྐར་རྒྱལ་དང་། འོད་ཟེར་ཆེན་པོ་དང༌། གློག་གི་འོད་དང༌། ཡོན་ཏན་རྒྱ་ཆེན་དང་རིན་པོ་ཆེ་དང༌། དཔལ་འོད་དང་། བདུད་འདུལ་བ་དང་། གོ་ཆ་བགོས་པ་དང༌། སེང་་་་གེའི་ཕྱག་དང་། མེ་ཏོག་བཟང་པོ་དང་། །རིན་ཆེན་མཐོན་པོ་དང་། རྒྱ་་་་མཚོ་དང་། ས་འཛིན་དང་། དོན་གྱི་བློ་མངའ་བ་དང༌། ཡོན་ཏན་བརྩེགས་པ་དང་། ཡོན་ཏན་གྱི་ཚོགས་དང༌། རིན་པོ་ཆེའི་མེ་དང་། འཇིག་རྟེན་ལས་འདས་པ་དང༌། འཇིག་རྟེན་གྱི་ཟླ་བ་དང༌། པ་དབྱངས་སྙན་དང༌། ཌཚངས་པའི་ཏོག་དང་། ཚོགས་ཆེན་གཙོ་བོ་དང་། སེང་གེའི་སྟོབས་དང་། །དྲག་ཅན་གྱི་བྱིན་པ་དང༌། ཆོས་ཀྱི་དབང་ཕྱུག་དང༌། གཟི་བརྗིད་འོད་དང༌། འོད་ཟེར་ཆེན་པོ་དང་། རིན་ཆེན་གྲགས་པ་དང་། །ཚོགས་ཅན་གསལ་བ་དང་། །གྲགས་པ་མཐའ་ཡས་པ་་དང༌། འོད་ཟེར་དོན་ཡོད་པ་དང་། ལྷའི་དྲང་སྲོང་དང་། །སྐྱེ་བའི་དབང་པོ་དང༌། ་དགེ་འདུན་བརྟན་པ་དང་། ཕྱོགས་་་་་བཟངས་དང་། །རྟོག་དང་། །ཡུལ་འཁོར་མེ་ཏོག་དང༌། །ཆོས་ཀྱི་བློ་གྲོས་་་དང༌། རླུང་གི་ཤུགས་ལྟར་གཤེགས་པ་དང་། ལེགས་པའི་གྲགས་པ་དང་། སྣང་ལྡན་དང་། ལྷའི་ཚོགས་དང་། ཡོན་ཏན་སྲུང་དང་། །དོན་གྱི་བློ་གྲོས་དང་། བསྙེངས་པ་མེད་པ་དང་། གྲགས་པ་བརྟན་པ་དང་། འོད་གནས་་་a།
