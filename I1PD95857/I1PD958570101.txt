རྒྱུད། ཟི་བར་ཏུ་ཁྱོ་བ། །ཆུ་། །ཀུན་ཞྭཕ་ར་ཕ་ར། ཕུ་རུ་ཕུ་རུ། །བ་ར་བ་ར། བུ་རུ་བུ་རུ། བྷ་ར་བྷ་ར། བྷུ་རུ་བྷུ་རུ། མ་ར་མ་ར། མུ་རུ་མུ་རུ། ཡ་ར་ཡ་ར། ཡུ་རུ་ཡུ་རུ། ར་ར་ར་ར། རུ་རུ་རུ་རུ། །ལ་ར་ལ་ར། ལུ་རུ་ལུ་རུ། ་ཝ་ར་ཝ་ར། ཝུ་རུ་ཝུ་རུ། :ཤ་ར་ཤ་ར། ཤུ་རུ་ཤུ་རུ། ཥ་ར་ཥ་ར། ཏུ་རུ་ཏུ་རུ། ས་ར་ས་ར། སུ་རུ་སུ་རུ། ཧ་ར་ཧ་ར། ཧུ་རུ་ཧུ་རུ། །ཀླུ་ར་ཀྟ་ར། །ཀླུ་རུ་ཀླུ་རུ། ཧཾ་ཧཱུཾ། ཧཱུཾ་ཧཱུཾ། ཧཱུཾ་ཧཱུཾ། ཧེ་ཧཾ། ཧོ་ཧོ། ཧཾ་ཧགབྷཾ་གནྡྷ?། :·ཛཱ་ཏི་ཛཱ་ཏི། དེ་བཱ་ནཱི༄༅། །དཱ་མ་ཡ་དཱ་མ་ཡ། སརྦ་ཤ་རཱི་རཱ་ཡ་ཧཱུཾ། ཛ་ཡ་ཛ་ཡ་ཧཱུཾ། བི་ཛ་ཡ་བི་ཛ་ཡ་ྋཧཱུཾ། །ས་ར་ས་ར་ྋཧཱུྃ། བི་ས་ར་བི་ས་ར་ཧཱུྃ། ཨག་ཡེ་ཙྪཱཧཱུྃ། བ་རུ་ཎེ་ཧཱུྃ༑ཨ་ནཱ་ལཱ་ཡ་ཧཱུྃ། །བཱ་:ཡ་ཝེ་ེཧཱུྃ། ཨཱ་ཀཱ་ཤཱ་ཡ་ཙྪཱཧཱུྃ། སརྦ་རོ་གོ་པ་ཧ་རཱ་ཡ་ཧཱུྃ། བི་ཥ་བྷ་ཀྵ་ཎཱ་ཡ་ཧཱུྃ། ཡནྡྲ་བཱ་ཧ་ནཱ་ཡ་ཧཱུྃ། ཀཱ་ལཱགྷ་ྋཡེ་ཧཱུྃ། དེ་བཱ་ཡ་ཙྪེཧཱུྃ། ཡ་མཱ་ཡ་ཧཱུྃ། ཀུ་བེ་རཱ་ཡ་ཧཱུྃ། དེ་ཏྭཱ་ཡ་ཙྪཱཧཱུྃ། ཨིནྡྲ་ཡ་ཧཱུྃ། བྷུ་ཏཱ་ཡ་ཧཱུཾ། སརྦ་སཏྭཱ་ཡ་ཙྪེཧཱུཾ། བརྟ་ཙཀྲ་ཡ་ཧཱུྃ། དྷརྨ་དྷ་ཏ་ཝེ་ཧཱུྃ། མུ་ཁཱ་ཡ་ཧཱུྃ། ནི་ཏཱ་ཡ་ཧཱུྃ། ཀ་པཱ་ལ་མཱ་ལི་ནེ་ཧཱུཾ། བྷུ་ཛཱ་ཡ་ྋཧཱུྃ། ཨམྦྷ་ཡ་ཧཱུཾ། ཙ་ར་ཎཱ་ཡ་ཧཱུཾ། ཛ་ཊཱ་མུ་ཀུ་ཊཱ་ཡ། མ་ས་མ་སི། ག་ས་ག་ས། །ར་ས་ར་ས་། བྷ་ར་བྷ་ར། མ་ཊ་མ་ཊི། ཏྲ་ཊ་ཏྲ་ཊ། གྷ་ཊ་གྷ་ཊ། །མུཉྩ་མུཉྩེ། གྷོ་རཱ་ན་དྷ་ཀཱ་ར་པྲ་ཡཱ་ཡ། ཕི་ཕོ། ཕུ་ད་ཕུ་ད། །མ་ར་མ་ར༈པ་ར་མཱརྦ་རཱུ་:པི་ཎེས། ཤཱནྟི་ཀ་རཱ་ཡེ། པུཊི་:བནྡྷ་ནཱ་ཡེ། ལོ་ཀ་སུ་པཱ་ལ་རཱུ་པི་ཎེ༄༅། སརྦ་ཤ་རཱི་ར་རཱུ་པཱ་ཡ། བྲྀཀྵ་གུམྨོ་ལ་ཏཱ'ིཡི། སསླ་རཱུ་པི་ཎེ་སརྦ་བསྟུ་ནེ་མཱ་ར་ཎཱ་ཡ། རཱུ་པཱ་ཨིཡ། བྷཀྵ་ཡ། བྷཀྵ་ཎ། །རཱུ་པཱ་ཡ། ནི་རོ་པ་མཱ་ཡ། ནི་རོ་པ་མ་མུ་ཁ་བེའོ། །ཧ་ས་ཧ་ས། བི་ལཱ་ས་བི་ལཱ་སི། ཀྲྀ་ཏྭཱ་པ་ཡ་ཚྭ་ཏྭཱ་པ་ཡ། དཱིནྡྲི་ཡ་ཡོ་གཱ་ཡ། ནིསྭ་བྷཱ་ཝ་མ་KU_།
