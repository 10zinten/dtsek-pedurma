རྒྱུད། ཁཟླ། ་ཚོjམ་ལུ་ས་ཀ། ཆུ་:ཀn《སྣར་བཛཱ། «པེ་»«སྣར་+hརྡོ་རྗེ་སྐུ་མ། «པེ་»«སྣར་»+hརྡོ་རྗེའི་གདན་མ། :ལོ«པེ་»«H««°««+dºws»«+bཁ་ཅིག་ན་མེད། །ཡོ::ཚེ་«པེ་»«སྣར་»+ཆགཉིས་སུ་མེད་པ། ::ཚེ«པེ་»«སྣར་+hརྣལ་འབྱོར་མ་ཆེན་མོ། །ལོ་འགེ«པེ་»«སྣར་»སིདྡྷེ། +དབཀླགས་པས་གྲུབ་པ། འ«པེ་»«སྣར་+ས། +hའཇིགས་མ། «པེ་»«སྣར་»ཧ་ས་ཧ་ས། ་+hརྒྱུད་གཞུང་དུ་ཧ་ས་ཧ་ས་ཞེས་པ་ཡོད་ཀྱང་འདི་ཡི་སྔགས་བཏུ་ལས་འདི་ཁོ་ན་བཞིན་ཡོད་དོ། །སུ«པེ་《སྣར་+ཆདཔའ་མོ། ན«པེ་»«སྣར་+དསྔགས་བཏུ་ལས་ཧ་ཡི་ས་བོན་ཡི་གེ་བཞི་ཧོ་གཅིག་ཧཱུཾ་གཉིས་བརྗོད་པར་བྱ་ཞེས་དང་འགྲེལ་པར་ཡང་དེ་དང་མཐུན་པས་གཞན་དང་མི་བསྲེའོ། །«པེ་«སྣར་»+hའཇིག་རྟེན་གསུམ་རྣམ་པར་འཇིགས་མ། འ«པེ་»«སྣར་རེ་ཏི། +ན་བརྒྱ་སྟོང་ཕྲག་བྱེ་བའི་དེ་བཞིན་གཤེགས་མ་་་་ཀཐམས་ཅད་ཀྱི་འཁོར་དང་ལྡན་མ། ::འཚེ«པེ་»«སྣར་»+ནསེང་གེའི་གཟུགས་ཅན་མ། །ནགེ«པེ་»«སྣར་»+ཇགླང་པོ་ཆེའི་གཟུགས་ཅན་མ། འཀྱེ«པེ་»《སྣར་»ལོཀྱོ། འགེ«པེ་»«སྣར་»+hའཇིག་རྟེན་གསུམ་གྱི་ལྟོ་བ་་་ཅན་མ། ཅེ«པེ་»«སྣར་»+hརྒྱ་མཚོ་ཆེན་པོའི་སྐ་རགས་ཅན་མ། «པེ་《སྣར་»བཱི། ནང་། «པེ་»《སྣར་»+ཆདཔའ་བོ་གཉིས་སུ་མེད་མ། «པེ་»《སྣར་ཧཱ། འཚེ«པེ་»«སྣར་+hཕྱུགས་ཆེན་མོ་རྨོངས་པར་མཛད་མ༑་བ་ཞབྱེ«པེ་»སྣར་»—མ་ཧཱ། །«པེ་»《སྣར་»+དརྣལ་འབྱོར་དབང་མོ་ཁྱོད། མགུ«པེ་《སྣར་»+ཇམཁའ་འགྲོ་མ། གྱི«པེ་»«སྣར་+hའཇིག་རྟེན་རྣམས་ཀྱི་འཆིང་བྱེད་མ། ཉེ«པེ་《སྣར་—པྲ་ཏྱ་ཡ་ཀ་་རི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཉེ《སྣར་»གླུ། ཚེ«པེ་»«སྣར་»+hའབྱུང་པོ་ཐམས་ཅད་སྐྲག་མཛད་མ། མསུ«པེ་»:འི《སྣར་»སིདྡྷེ། «པེ་»ཤོ། ཁྱི«པེ་《སྣར་+hརིག་པའི་དབང་ཕྱུག«པེ་《སྣར་སྐ། །ཚེ«པེ་»《སྣར་»+དརྒྱུད་ལས། མཆོད་དང་བསྟོད་སོགས་དུས་སུ་ནི། །ཐོག་མར་རྡོ་རྗེའི་མིང་ཤེས་བྱ། །དཔའ་བོ་རྣམས་ལ་དེ་བཞིན་ནོ། །ཞེས་དང་། རང་མིང་བརྗོད་པའི་སྔགས་ལ་ནི། །ཐོག་མར་ཕྱག་འཚལ་རབ་ཏུ་གྲགས། །ཧཱུྃ་ཧཱུྃ་ཕཊ་ཀྱི་ལོཡི་གེ་མཐར། །དེ་ལྟར་འཁོར་ལོ་ཀུན་ལ་བྱ། །ཞེས་གསུངས་པའི་དོན་ནོ། །དགྱེ«པེ་»«སྣར་»—ཡེ། +དམཁའ་་་འགྲོ་མ། བཀྱེ«པེ་»«སྣར་+kགཟུགས་ཅན་མ། མཉེ«པེ་»«སྣར་»ཡེ། +hའོ་བྱེད་མ། ་འཁྱེ«པེ་ནི། 《སྣར་»ནི། ཙཀྱི《པེ་»«སྣར་»ལི། +ཚེ«པེ་《སྣར་»+hཀེང་རུས་མ། མཉེ«པེ་»+hརྒྱལ་པོས་བསྐོར་བ། 《སྣར་»+hརྒྱལ་པོས་སྐོར་མ། ་འཁྱེ«པེ་»«སྣར་»ཧེ། +དདུམ་སྐྱེས་མ། ཚདུ«པེ་»《སྣར་»+གདུར་ཁྲོད་མ། ཨེ«པེ་»«སྣར་»+hརྣམ་པར་ཞུ་མ།
