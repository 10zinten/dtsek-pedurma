།ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྲཱི་ཧཏྤ་ལ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲྀ་ཧཏྤ་ལ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཧ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་བྲི་ཧ་ཙཀྲ་བརྟི་ེཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཏ་པ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། །ཨོ་བཛྲ་ཨ་ཏ་པ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་དྲི་ཤ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་སུ་དྲི་ཤ་ཙཀྲ་བརྟི་ེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་དན་ན་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །་ཨོ་བཛྲ་སུ་དན་ན་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཀ་་་ནིཥྛ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཀ་ནིཥྛ་ཙཀྲ་བརྟི་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨཱ་ཀཱ་ཤཱ་ནནྡྷ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཱ་ཀཱ་ཤཱ་ནནྡྲ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཛྷཱ་ནཱ་ནནྟཱ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་བི་ཛྷཱ་ནཱ་ནནྡྷཱ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཀིཉྩ་་རྩྭ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཀིཉྩ་ནྡྷ་ཡ་ཏ་ན་ཙཀྲ་བརྟི་པེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནི་བ་སཾ་ཛྙཱ་ནཱ་སཾ་ཛྷཱ་:ཡ་ཏ་ན་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནི་བ་སཾ་ཛྙཱ་ནཱ་སཾ་ཛྷཱ་:ཡ་ཏ་ན་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀ་་་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་་པྲ་ཏ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲ་ཏ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་་བཛྲ་ཏིརྱ་ཀ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏིརྱ་ཀ་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོཾ་བཛྲ་ཨ་སུ་ར་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སུ་ར་ཙཀྲ་བརྟི་ཧུཾ་ཧཱུྃ་་་ཕཊ༑རྒྱུ་ཨོ་བཛྲ་བི་མཱ་ཎ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་མཱ་ཎ་ཙཀྲ་་་་བརྟི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤ་རི་ར་བི་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤ་རི་ར་བི་ཙཀྲ་བརྟི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་།
