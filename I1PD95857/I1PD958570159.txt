ས་དུ་བགྲོད་པ་ན་ཕྱིས་ཀྱི་སྐྱེ་བྱིན་གྱིས་བརླབས་ཏེ་ལོ་པ་ཁ་བ་ཏི་བ་ཏི་སརྦ་གྲྭ་ས་དཔེ་ཏ་པི་ཤཱ་ཙ་ན་ད། ཨ་པ་ས་བབ་དང་ཀ་ལཱ་ཀི་ནྱ་ད་ཡ་པ་ལེ་ལ་བས། དུ་མི་བདེ། བླ་མ་ཞུ་ཡི་ཡ་ཐིཏྟཾ། ཤུ་ཐ། ཛི་གྷ་ཐ། དེ་ལབ་ན་དེ་མ་ཐ་མ་བཀྲ་དག་འདུ་བ་བྱི་ཁྲིད་ཡེ་མ་མུ་། ་ཏུ་མི་གནས་/:ཀྱི%་ཚེ། ང་ངོ་:ཞི་[སོཌ་་ང་ཀུ་ཎ/གྱིས༈((་པ་ལོ༄༅། །བླ་མའི་མཎྜལ་བཞུགས་སོ། །ད་ལ་ཐན་བཔས་ཞེན་པ་བདར་ན་ཀུན་ལ་འགྲོ་པ་ག་དེ་དག་ད་བར་ལོ་དར་དད་པ་ཡོད་ན་ངོ་བ་གང་། ཆུའི་ག་ཡི་གོང་བུ་ཡི་མགོ་ན་བ་ལ་ཁུརྩེ་དུ་་༄༅། །དཔལ་མཁའ་འགྲོ་རྒྱ་མཚོ་ལ་ཕྱག་འཚལ་ལོ། །བླ་མ་ཡོངས་་སུ་མཉེས་པ་ལ། །དངོས་གྲུབ་མ་ལུས་འབྱུང་བའི་ཕྱིར། །རྒྱ་མཚོ་ཉིད་ནས་་་གསུངས་པ་ཡི། །བླ་མའི་མཎྜལ་བཤད་པར་བྱ། །དེ་ལ་ཐོག་མར་བླ་མའི་་་་་མདུན་དུའམ་གཞན་དུ་ཡང་རུང་སྟེ་ཆུ་དང་མེ་ཏོག་ལ་སོགས་པ་ཚོགས་པར་བྱས་ལ་རྐང་པ་བཀྲུས་ཏེ། ཨོ་ཏིཿསྭཱ་ཧཱ། ཞེས་བྱ་བས་ལག་པ་བཀྲུ་བ་དང་། ཁ་་་་བཤལ་བར་བྱའོ། །ཨོཾ་ཧཱུཾ་ཞེས་བྱ་བས་བདག་དང་གནས་དང་རྣལ་འབྱོར་བསྲུང་བར་བྱའོ། །དེ་ཉིད་ཀྱིས་སྟན་བྱིན་གྱིས་བརླབ་་པ་ཡང་བྱའོ། །ཨོཾ་བཛྲ་བྷུ་མི་ཧཱུྃ་ཞེས་བྱ་བས་ས་གཟུང་བར་བྱའོ། །ཨོཾ་སརྦ་བིགྷཱན་ཨུཏ་ས་ར་ཡ་ཧཱུྃ་སྭཱ་ཧཱ། །ཞེས་བྱ་བས་ཆུ་དང་ལྕི་བ་སྲོག་ཆགས་མེད་པས་བྱུག་པར་བྱའོ། །ཨོཾ་ཨཱབཛྲ་རེ་ཁེ་ཧཱུྃ་ཞེས་བྱ་བས་ལག་པ་གཡས་པས་མེ་ཏོག་གཡོན་པའི་རྒྱབ་ཏུ་བཞག་ལ་མཎྜལ་དུ་་དོར་རོ། །ཨོ་སུ་རེ་ཁེ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་དྷི་ཏིཥྛནྟ་སྭཱ་ཧཱ། ཞེས་བྱ་བས་ལག་པ་།
