བསྟན་འགྱུར། རྒྱུད། ཝིཏི། ཕྱག་གཡས་པ་རྣམས་ནི་རིམ་པ་བཞིན་དུ་འཁོར་ལོ་དང༌། །རྡོ་རྗེ་དང་། ཌཔདྨ་དང༌། ཨུཏྤ་ལ་ཐོགས་པ། །གཡོན་པ་རྣམས་ཀྱིས་ཀྱི་མཆོད་པ་སྣ་ཚོགས་འཛིན་་པས་གཙོ་བོ་ལ་སྟེར་བ་སྟེ་རྒྱན་ལ་སོགས་པ་ཐམས་ཅད་སྔར་དང་འདྲ་བ། །ཐམས་ཅད་ཀྱང་སྒེག་ཅིང་རོལ་པའི་འདུག་སྟངས་ཅན་གསལ་བར་བསམས་ལ། s/ཨོཾ་བཛྲ་སཏཱ་གུ་རུ་པཱ་དེ་སརྦ་སཏྭཱ་པ་ཀ་ར་ཎཱ་ཡ་སྭཱ་ཧཱ། ཞེས་བྱ་བས་ཕྱག་བྱས་ཏེ་་་དབུལ་བར་བྱའོ། །དེ་ནས་ལྷ་མོ་རྣམས་སློབ་དཔོན་ལ་ཐིམ་པས་ཐུགས་དགྱེས་་་པར་བསམས་ཏེ། ཨོཾ་ཨཱཿཧཱུྃ་མུཞེས་བརྗོད་པས་གཤེགས་སུ་གསོལ་བར་བྱའོ། །དེ་ནས་མེ་ཏོག་བསྡུས་ཏེ་རང་གི་མགོ་བོར་བཞག་་ལ་ཇི་ལྟར་བདེ་བར་གནས་པར་བྱའོ༑། དཔལ་ལྡན་མཁའ་འགྲོ་རྒྱ་མཚོ་ཡི། །རྒྱུད་རྒྱལ་མན་ངག་གིས་་་་་བསྲུངས་་ནས། །མཎྜལ་རིན་ཆེན་ཕུལ་གང་དེས། །འགྲོ་ཀུན་ཆོས་རྒྱལ་འཁོར་སྒྱུར་ཤོག། རིན་པོ་ཆེ་མརྒད་མཁས་པ་ཛ་ཡ་སེ་ནའི་གསུང་། །དཔལ་ཡེ་ཤེས་་རྡོ་རྗེས་བསྒྱུར་བ་རྫོགས་སོ། །།།།།ཚད་མ་ཟླ་བ་དང་ཟླ་བ་ད་དང་ན་ལ་དེ་་་་་་་་ལ་་་་་ལ་་་སྤྲོ་བ་འམ་བློ་ལ་གང་དང་གང་ཁ་པ་ཆ་མཚོན་ན་ཤེས་པ། །སྟོང་པ་ཁོ་ན་བསུ་ན་ཁོ་ལ། ས་པ་ཁ་ཆུ་ཁ་ལོ་gਸਇਸਨੂੰਇਕਰਲਮਲਲਉਸਠੋਧਲਾਈਨཔར་འབད་པ་ལྷག་པ་དེ་ཆེན་པ་ཆོ་ག་ལས་ཁ་ཉན་པར་དགའ་བ་བདག་མཁོ་བས་གསུམཚོ་ཕྱིན་པ། ཚེ་དང་ན་བ་མ་དག། དག་ཆ་ཁག་ཆ་ཆུང་། ཤད་ལ་སྐུ་ལྔ་ཕྱི་བ་དེ་འགྲོ་བ་སོ་ཆོས་དང་ས་ཁོ་ཚོང་ལ་ཤ་བ་བྱེད། །ལྟ་ཀུཏ་ཆེ་ཀའོ་ཁ་ཏི་ཀ་ཏ། ༽ਇਸਤੋਂਪਤਾਲਾਸ਼ਛਾਲ।ਸਿੰਘਸਤਕਓਗੱਲਓਥੇਪੁਲਨੂੰ'ਦਾਲਾਲ।ਰਸਨੂੰਸ਼ਰਨਬਿਨਸੰਸਕਸਖ਼ਪਤམ། །ཆུ་བློ་ཀུན་ནས་བྱ་སྟེ། བོད་ཁ་རུང་། །ཁོ་དག་ཁ་ཚོ་དང་། གང་ཞེ་འཚོ་བས་འཁཏྟན་།
