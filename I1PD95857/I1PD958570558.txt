རྨོངས་པ་གསལ་བྱེད་ཊཱི་ཀཱ་འདི། །ཚིག་དོན་མ་ལུས་སྟོན་བྱེད་པ། །བརྩམས་་ཏེ་བསོད་ནམས་གང་བསགས་པ། །དེ་ཡིས་འཇིག་རྟེན་གྲོལ་བར་ཤོག། སློབ་དཔོན་ནག་པོས་མཛད་པའི། །སྒྲོན་མ་བཞི་ཞེས་བྱ་བའི་མན་ངག་གི་ཊཱི་ཀཱ་ལས། གསང་བའི་སྒྲོན་མ་སྟེ་བཞི་པ་རྫོགས་སོ། །རྒྱ་གར་གྱི་མཁན་པོ་ཆེན་པོ་ཤྲཱི་དྷ་ར་ཉིད་ཀྱི་ཞབས་ཀྱི་ཞལ་སྔ་ནས་ཀྱིས་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པའོ། །།།ཅིང་གཏན་ལ་ཕབ་པའོ། །ལེགས་ཀྱི་། །པ་ན། །བ་ཀུ་ལ་དམ་པ་དང་། དེ། །ཀྱིས་ཁ་ཡིན་དེ། །།།ཀ་འཚོ་བ་དང༌། ར་བའི་དང་། ཀྱིས་ད་ལྔ་ནི། མེ་དང་བསྡུར་མཆན། ག་ལ་བགས་ཀ་་་་་་་་་་་་་་་་་:། དེ་ལྟ་བས་ན་ཁ་་་་་དང་། རྒྱ་སྐྱེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་སྒྲོན་མ་བཞིའི་ཊི་ཀ་་་་་བཞུགས་སོ། །《སྣར་»སྒྲོན་མ་བཞིའི་ཊཱི་ཀ་བཞུགས། «པེ་»《སྣར་»ཏུ་རཱི། «པེ་»《སྣར་བརྡུས། «པེ་»«སྣར་»བརྡ། ཀཟླ《ལ་ཁྲིམས་ཀྱི་བླ་ཆེན་་་(«པེ་»«སྣར་»གསལ། «པེ་»«སྣར་»ཀ«པེ་»«སྣར་»+གི། ཆེ«པེ་《སྣར་པ། །«པེ་»《སྣར་བརྡའ། «པེ་»«སྣར་ཁྱབ། («པེ་»«སྣར་»—པས། («པེ་»《སྣར་གཙོ་བོར། («པེ་»«སྣར་»ལི། །«པེ་»«སྣར་»པ། «པེ་»«སྣར་བཟུང་། «པེ་དམ་པའི་དོན་ཆེན་པོ་ལ་དེས་རྒྱས་བཏབ་པའི་ཕྱིར་རོ། །(ས«པེ་《སྣར་»—པ། ་་་་་་་་་་་་་་་་་། («པེ་»«སྣར་སྐད། («པེ་»«སྣར་གིས། 《སྣར་གསོལ། ་། :་་་(«པེ་»《སྣར་བསྟན། ་(«སྣར་»ཝཾ། «པེ་»«སྣར་»བརྟེན། «པེ་»ཀྱིས་རྫས་སུཾ། «སྣར་»ཀྱི་ས་སར་རྫས་སིཾ། །«པེ་»སྣར་པ། །ལེགས་པ་ལ། །འོ་ནངས་་(«པེ་»«སྣར་བརྟེན། («པེ་»«སྣར་པ། ༽«པེ་»«སྣར་གནས། །ཤེས། «པེ་»«སྣར་ལ་་jp/::ཉམས་ཀྱི་སྟེ་པ། ་ཉི«པེ་»«སྣར་»ཉིད་ཀྱི་འཁོར་ལོ། ་པོ«པེ་»ཚིགས། @«པེ་»«སྣར་»དེ། «པེ་»«སྣར་ཕྱག་(«པེ་»«སྣར་»གི། («པེ་《སྣར་པར་ནཱ་ད། 《པེ་»«སྣར་ཟུངས། «པེ་»«སྣར་བར། །/«པེ་»«སྣར་པའི། །(«པེ་»«སྣར་»ཆོགཚེ«པེ་»«སྣར་»ཕྱི་རོལ། །།ས«པེ་《སྣར་»སྣ་བུག་གཡོན། «པེ་»《སྣར་—པར། །བྱེད་(«པེ་»«སྣར་»གཡས། Q«པེ་《སྣར་བསྐུལ། «པེ་《སྣར་»ཀྱིས། །®«པེ་»《སྣར་—ཤུ་ཀྲའོ། །།
