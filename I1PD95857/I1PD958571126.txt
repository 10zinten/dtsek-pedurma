ག་ཏུ་གྲག་ན་་ཁོ། འཁྲོང་། ཚོད་ག་ས་གས་ཁ་བ་ཀལ་ཁ་བ་བློ་ཁེ་སྐྱེ་ཁམིཞར་ལ་ཀློང་ལ་སྤྱན་མ་སྒྲིབ་བག་བབས་སོ། ཞི་བ་དང་ནད་འབྱོར༌ལ། ད་དུང་བག་བསུ་བ་བྲ་ན་མཚོ་ཁ་ཏོ་བ་ལྟ་ལ་འཁ་ཁ་སྨྲ་བ་ཁྱབ་བོདཞེས། །བགྲོད་ལpཝཱ་ཞུ་ག་ལ་བའི་འི་འཇོག་ག་འགྱུ་ལྟ་བུ་ག་ར་འོ་ན་ད་དོན་ཧ་ར། ཕྱིས་དེ་བ་ག་ཚོད་དག་ལ་བློ་ཁོ། ཁོ་ཆེ་བློ་དགའ་ཞིང་ཡིག་ཆ་དོད་གང་མང༄༅། །རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་པའི་ཐབས། ་ཆ་ཁྲི་ན། མཎྜཅིང་གཚོ་དམ་དང་ཚོགབ་ཞེན་པ་ལ། །ཀུ་བ་བ་འདཆེན་ཏེ་ཁ་བ་དང་ཚོ་བ་ཆཌ་བསྐྱོད་ངལ་བ་དང་དབྱིག་པ་དག་པ་དང་། ཚོལོ། ཁྱུང་པའི་ཁ་བའི་བྱིན་ཅན་བཞི། ས་པ་གས་ཏཊིཀ་ཁgདོ་ཟེ་འབྱེད་ཀྱི་ཀྱི་བང་ཏ་ཏཚུལ་ཁྲོ་བོ་འཁྲིག་དང་པར་དུ་ཁ་འབྱེད་ན་ཚོད་ལ་ཀ་བ་དང་། ཁོས་ཀྱི་ས༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་:བཱ་རཱ་ཧཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་པའི་ཐབས། དཔལ་རྡོ་རྗེ་ཕག་མོ་ལ་ཕྱག་འཚལ་ལོ། །དང་པོ་རྣལ་འབྱོར་པས་རང་གི་སྙིང་གར་ཨ་ལས་ཟླ་བའི་དཀྱིལ་འཁོར་དེའི་སྟེང་དུ་ཡི་གེ་ཧཱིsདེའི་འོད་ཀྱིས་ལུས་བདག་དང་གཞན་དང་ཞིང་སྦྱང་བར་བྱའོ། །དེ་་ནས་འོད་ཟེར་འཕྲོས་པས་རྡོ་རྗེ་ཕག་མོ་དང་། སངས་རྒྱས་དང་བྱང་ཆུབ་་་་་སེམས་དཔའ་རྣམས་སྤྱན་དྲང་བར་བྱའོ། །དེ་ནས་ཕྱག་བྱས་ཏེ། ནང་གི་་་་་མཆོད་པ་དང༌། གསང་བའི་མཆོད་པ་དང༌། ཕྱིའི་མཆོད་པའོ། །དེ་དག་གི་སྤྱན་སྔར་སྡིག་པ་བཤགས་པ་དང༌། རྗེས་སུ་ཡི་རང་བ་དང༌། གསོལ་བ་་་གདབ་པ་དང༌། དགེ་བའི་རྩ་བ་བསྔོ་བར་བྱ་སྟེ། དེ་ནས་སྐྱབས་སུ་འགྲོ་བ་་་བྱའོ༑། དེ་ནས་བྱམས་པ་དང་། ཚིག་སྙིང་རྗེ་དང་། །དགའ་བ་དང༌། བཏང་སྙོམས་བསྒོམས་ལ་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་པར་བྱའོ། །དེ་ནས་ཨོཾ་སྭ་བྷཱ་བ་་་།
