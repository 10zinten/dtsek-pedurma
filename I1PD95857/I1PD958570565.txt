རྒྱུད། ་་སྙིང་གི་པདྨར་ཟླ་བ་ཉི་མ་ཁ་སྦྱར་དབུས་གནས་པ། །ཕྲ་བའི་ཐིག་ལེ་གསལ་བའི་གཟུགས་ཀྱིས་གནས་པ་རྣམ་བལྟས་ཏེ། །དེར་ནི་ཡང་དང་ཡང་དུ་སྲོག་འགོག་་་་སྦྱོར་བ་ལས་འདས་པས། །»། མཁའ་མཉམ་དྲི་མེད་མཆོག་ནི་རྣལ་འབྱོར་པས་་་་་མཐོང་འགྱུར།  ོ། ལུས་ཀྱི་དབུས་གནས་སྐུད་པ་ལ། །ཕྲ་བའི་རྣམ་ཤེས་འགྲོ་འོང་བ། །།རྩེ་གཅིག་སེམས་ཀྱིས་གོམས་བྱས་ན། །རྟོག་མེད་ཡེ་ཤེས་ཡང་དག་་་་འཇུག། དཔལ་རྡོ་རྗེ་ཕག་མོ་སྒྲུབ་པའི་ཐབས། །ཨཱ་ཙརྱ་པྲཛྙཱ་བྷ་དྲས་མཛད་པ་རྫོགས་སོ། །་་། ་ཟཅ། ་ཚེ་རག་ཏཀ་ལས་ཏ་ཀྲྀ་ཌོ། །དོཔ་ལ་བལ་པོའི་པཎྜི་ཏ་སུ་མ་ཏི་ཀཱིརྟི་དང༌། བོད་ཀྱི་ལོ་ཙཱ་བ་མར་པ་ཆོས་ཀྱི་་་དབང་ཕྱུག་གིས་བསྒྱུར་བའོ། །ཚོ། ཆུ་ཀ། །ཥ་ཊ་ཋཀརླཁ་ཎའི་ཏུ་ཆpཁ་ཊ་ཌོཟང་། །ཏུ་ས་ཆ་ཁ་ཀ་། དར་ལར་ཌོ། །འཛོ་འཁོpབ་n/j་ཎི་ཀའི་ཁྲལ་ཀ་ལ་ལྟོས་པས་ན་དང་པ་ལ་རང་བློ་དང་གོ་ཀ་ག་ང་བ། བཀག་ལྟར་ན་ད་ལོ་ང་པ་གྲ་སྒོ་ཁཏོཥཱ་ཡ། མ་་བསྡུར་མཆན། ལྔ་ཕྲ་མོ་ལ་དgཔཏྟ་སྐྱ། /(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། *«པེ་»«སྣར་»རྡོ་རྗེ་ཕག་མོའི་གཡུ་སྒྲུབ་ཐབས་བཞུགས་སོ། །༼«པེ་»«སྣར་ལ། /Q«པེ་»«སྣར་བསྟན། ལ་ཚ་ཀྱི་འོ་«པེ་»«སྣར་མ། «པེ་»«སྣར་དེ། «པེ་»«སྣར་རྫོགས་པ། ༄༅«པེ་《སྣར་ཕཊ། ང«པེ་»«སྣར་མཛད་པའོ། །«པེ་»«སྣར་བལྟམས། ཚེ«པེ་»«སྣར་རྣམ། «པེ་»«སྣར་»«པེ་«སྣར་མོའི། «པེ་《སྣར་ཙ། （འདིའི་འོག་གི་ལོ་ཙཱ་ཞེས་པ་རྣམས་པེ་»«སྣར་»ལ་ལོ་ཙ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། ཀྱིས་བདག་ཅག་པ། ཚེ་དཆཚོ་#jpg#/Sཁ་སྔོགས་དང་། །དཀཆེ་ཆེ་དང་གཏི་ང་ཆེ་ང་ལ་ཁྲིད་དང། ཁ་འཁྱིལ་བའོ༈དཔལདའིལྔ་པ་ནི། པ་དོ་ཀ་ཏཁོར་ཆ་ལྔ་དnཀ་པཆོད་མེཏེརཏིཏཔ། ཧོར་གྱི་རལ་དང་པ་ཏཊིR། ཏ/ཏོ། ་བྱུ། ཏད། །ཆཀོ་ཏ་ཀྱི་ཨུ་ཏམཀརོབ་ཙྪ་།
