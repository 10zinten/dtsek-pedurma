ཁ་ཟླབསྟན་འགྱུར། རྒྱུད། སྐུ་ཚོ་ཁ་ཕཊ༑ཨོཾ་བཛྲ་ཙཀྲ་བཱ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤྲཱིཀ་ར་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོཾ་བཛྲ་ཤྲཀླུ་ར་ཎ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཀྐ་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཀྐ་བི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ལ་ཀཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ལ་ཀཱ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་བི་ལཱ་ཌི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ལ་ཌ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་ནཱ་ལ་གྲི་བཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་གྲི་བ་ྋཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སཱ་རི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སཱ་རི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སེ་ནཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་སེ་ན་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཧཱུྃ་ཀུ་མ་ལོ་ལཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྱི་ཀུ་མ་ལོ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱ་ཊི་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱ་ཊི་ར་ཧུ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཀཱ་ཀ་ཛཾ་གྷ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀ་ཛཾ་གྷ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ས་མཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ས་མ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལེ་ཧ་སྲིཥྛ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལེ་ཧ་སྲིཊྚ་ཨེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དད་ད་རཱི་་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་དད་ད་རི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་མྲ་གཱ་རི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་་ཕཊ༑ཨོཾ་བཛྲ་མྲ་གཱ་རི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སི་ག་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་སཾ་གྷ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷཱུ་ཤྲཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷ་གྲུ་ཧུཾ་ཧཱུྃ་་་ཕཊ༑ཨོ་བཛྲ་བྷིནྡྲི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷིནྡྷཱ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཤཱི་་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཤི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ག་ཛཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ག་ཛ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མྲ་གཱ་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་མྲྀ་ག་ཧུ་་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མརྫ་རི་ཀཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མརྫ་:རི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་གཱ་བ་ཎཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གཱ་ེབིཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧི་ཥཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧི་ཥ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏུ་ར་གཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཏུ་ར་ག་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་ཛཾ་བུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཛཾ་བུ་ཀ་—།
