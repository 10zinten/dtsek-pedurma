མི་གང་བསོད་ནམས་དམན་པ་ཡང་། །དེ་ནི་གྲུབ་འགྱུར་ཐེ་ཚོམ་མེད། །བླ་མ་སངས་རྒྱས་བླ་མ་ཆོས། །དེ་བཞིན་བླ་མ་དགེ་འདུན་ཏེ། །བླ་མ་་་་་དཔལ་ལྡན་རྡོ་རྗེ་ཅན། །འདིར་ནི་བླ་མ་བླ་མེད་པའོ། །སངས་རྒྱས་་་འབྲས་བུ་འདོད་པ་ཡིན། །དེ་ཕྱིར་བླ་མ་བསྙེན་བཀུར་བྱ། །དཔལ་དང་ལྡན་པའི་རབ་ཏུ་སྔགས་པའི་གནས་:ཨུ་རྒྱན་ནས་བྱུང་བའི་དགོངས་པ་བླ་ན་་མེད་པ་དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་གསང་ཆེན་:རྣ་བ་ནས་རྣ་བ་རུ་:ཞལ་ནས་ཞལ་དུ་བརྒྱུད་པའི་རིམ་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་བསོད་ནམས་འབྱུང་གནས་བཟང་པོའི་ཞལ་སྔ་ནས་་་་དང་། བོད་ཀྱི་:ལོ་ཙཱ་བ་རྨ་བན་ཆོས་འབར་གྱིས་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། (མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་རྡོ་རྗེ་རྣལ་་་་་འབྱོར་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་—ལ། «པེ་»«སྣར་སྙེད། Q«པེ་»«སྣར་བྱའོ། «པེ་《སྣར་ཧཾ། «པེ་»«སྣར་ཀྱིས། ས«པེ་»«སྣར་—ཉ། @«སྣར་སཊྚ། «པེ་»«སྣར་»—ཉ། ཏེ«སྣར་»སཉྫ། «པེ་《སྣར་—ཉ། «པེ་»«སྣར་བརྟག་སྟེ། «པེ་»《སྣར་»རིམས། «པེ་»《སྣར་པོ། «པེ་《སྣར་ཡིག«པེ་»«སྣར་»གྱིས། །«པེ་»«སྣར་ད། «པེ་《སྣར་»ཡི། «པེ་»«སྣར་»གནས། «པེ་»«སྣར་»ཊ་ཡི། «པེ་»《སྣར་»ཡིགའགྱེདཀྱེ«པེ་»«སྣར་ཡི། (《སྣར་ནས། «པེ་ཚེག《སྣར་བརྩེག་«པེ་»«སྣར་ཡིས། «པེ་»«སྣར་བསྒྲགས། «པེ་»«སྣར་ཡིག་ག«པེ་»《སྣར་ཡི། «པེ་»«སྣར་བཟུང༌། («པེ་»«སྣར་ཡིས། །«པེ་»«སྣར་ས་ཡའི། «པེ་»«སྣར་»སུམ། («པེ་»«སྣར་»ཛྷ། «པེ་»《སྣར་»སྤང་བ། «པེ་»«སྣར་བྱ། «པེ་»«སྣར་»བཙག། «པེ་»«སྣར་བྲི། །—།
