ཀུ་ཡ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཀུ་ཡ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ཛ་བྷ་ཊཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ཛ་བྷ་ཊ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཁཊྛི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཁཊཱི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏཱམྦོ་ལ་ཝ་ཀྲི་ཡཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་་་་བཛྲ་ཏཱ་ལ་ཝ་ཀྲི་ཡ་ཀ་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སོ་བཙྪཱ་ཀཱ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་སོ་བཎྜ་ཀཱ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ལ་ཨོ་བཛྲ་ལོ་ཧཱ་རཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཧཱ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཎི་ཧཱ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཎི་ཧཱ་ར་ྋཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་དཱ་པ་ཀཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དཱ་པ་ཀ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་:སླེ་ཙྪེ་ཏཱི་ཡེེཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་:མེ་ལྕེ་ཌ་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:པ་ཎི་ཛཱི་ཡེ་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:པ་ཎི་ཨཛ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་ཐ་ར་གཱདྷི་ྋཡེ་ཏེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་ཐ་ར་གཱདྷ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྲྀ་ཎི་ཀཱ་རཱི་ཡེ་ཨཻཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྲྀ་ཏི་ཀཱ་ར་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙརྨ་ཀཱ་རི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཙརྨ་ཀཱ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷ་དྲ་ཀལྤ་བུདྡྷེ་བྷོ་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་ཀཱ་ཀཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨུg་ཀཱ་ཀྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤཱ་ནཱ་སྐྱེ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། :ༀ་བཛྲ་སཱུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཡ་མ་དཱ་དྷཱི་ཡེ་ཧུཾ་ཧཱུྃ་་་ཕཊ༑ཨོཾ་བཛྲ་ཡ་མ་རཱུ་ཏཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་དཾཎྜི་ཎཱིཉྩཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཡ་མ་མ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ་། :། ད་། །མེ:ཀར་ཏེ༈ཏཨོ་བཛྲ་ཀིནྣ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀིནྣ་ར་ཨེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་ཧ་རི་ཎཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་ཧ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཧུ་ཏུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨེཏཨོ་བཛྲ་ཧུ་ཏུ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཊ་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཊ་བ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱི་ཎཱི་ཡེ་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱི་ཎ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཻ་ཤྲཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:བཻ་ཤི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་་།
