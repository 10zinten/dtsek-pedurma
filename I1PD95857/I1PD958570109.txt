རྒྱུད། ཝིཏིཏུ་ཚ་ཆུ་སྐྱེ་སྐྱོང་། །ཟླ་འི་Lབཛྲ་མུ་ཀུཎྜིཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་མུ་ཀུཎྜ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་བཛྲ་མུརྫི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མུརྫ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གགྷ་རཱི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་གངྒ་རཱི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་པཱི་ཡེ་%ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་པ་ཧུཾ་ཧཱུྃ་ཕཊ༑། ཨོཾ་བཛྲ་ཤི་ལེནྡྲ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ལེནྡྲི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་གཱི་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གཱི་ཏ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་ཀ་ར་ཊི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་ཀ་ར་ཊ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏ་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་་་བཛྲ་ཏ་མ་ཌ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནྲྀ་ཏྱཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནི་ཏྱ་ཧཱུྃ་ཧཱུྃ་་་ཕཊ༑ཨོཾ་བཛྲ་ལཱ་སཱུ་ཡི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་སྱ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་་་ཊཀྐི་ཡེ་®ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ལི་ཤ་ར་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཏ་ལི་ཤ་ར་ཎ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཧཱུྃ་དུ་བྷི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཧཱུྃ་དུ་བྷི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མོ་དཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མོ་དྲི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་མཱ་ཏུ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་མཱ་ཏུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་བཱ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌྷ་བ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:ཌྷ་བ་ཨིཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌ་མ་རཱི་ཡེ་®ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌ་མ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཊཱ་ཎ་ཌཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཊཱ་ཎ་ཊ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་ཀ་ཧཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་ཀ་ཧཱ་ལ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨོ་ར་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨོ་ར་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷ་ཀི་ཧཱུྃ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་གྷཎྜི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་གྷཎཱི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀིཾ་ཀི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀིཾ་ཀི་ཎི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གྷནྡྷུ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་བྷུརྒྱུ་རི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌ་ཀོ་ལི་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་།
