ཟླ་ཁ་སྦྲ་དང་རང་ལ་བ་ཀ་ཏུགཏང་བ་ན་ས་བ་༄༅། །དབང་གི་ཆོ་གའི་རིམ་པ་ཞེས་བྱ་བ། བ་ས་འཁྲུ་བ་ལ་ན་ཁ་རྗེ་ཁ་ན་ཀ་ལས་ཁ་ཅ་ག་ད་་་་་་ང་ལ་བ་དང་། ས་ལ་ས་འ་:ཡག་ਧਰਟਮਲਾਲg/ਬਟਨਸਨਪਰਸਲੋਂਨੇཀྱང་། །ཟླ་བ་ཡར་དུ་ཟློ། །དེའི་མདོནང་ལ་ལས་དང་། །བདག་བ། ༄༅། །རྒྱ་གར་སྐད་དུ། ཨ་བྷི་ཥེ་ཀ་བི་དྷི་ཀྲ་མ་ནཱ་མ། བོད་སྐད་དུ། དབང་གི་ཆོ་གའི་རིམ་པ་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །དབང་གི་བྱ་བ་འདྲི་བར་བྱའོ། །དང་པོ་སློབ་མས་ལན་གཉིས་ལན་གསུམ་དུ་་་་་གསོལ་བ་གདབ་པ་དང༌། རྡོ་རྗེ་སློབ་དཔོན་གྱིས་སློབ་མ་ལ་དབང་གི་ཡོ་བྱད་་་་ཐམས་ཅད་འབྱོར་པ་ཡོད་དམ་ཞེས་དྲིའོ། །།བདག་ལ་ཐམས་ཅད་ཡང་དག་པར་ཡོད་དོ་ཞེས་པ་དེ་ལ། ཇི་ལྟར་དེའི་ལས་ཀྱི་སྒྲིབ་པ་དང་ལྡན་པ་དེ་སློབ་དཔོན་གྱིས་རབ་ཏུ་འཇོམས་པའི་དོན་ནི། དེའི་སྙིང་གར་ཀཾ་གི་ཡི་གེ་ཞུ་བ་ལས། ལས་ཀྱི་སྒྲིབ་པའི་རང་བཞིན་གདུག་པ་ཅན་མ་ལུས་པ་རྣམ་པར་བསམ་མོ། །དེ་བཞིན་དུ་དེའི་་ལྟེ་བའི་གནས་སུ་རྡོ་རྗེ་རྩེ་ལྔ་པ་ཞུ་བ་ལས་ཡོངས་སུ་རྫོགས་པའི་རྡོ་རྗེ་གནོད་སྦྱིན་་་ཕྱག་གཉིས་པ་ཁ་དོག་ནག་པོ་མཆེ་བ་གཅིགས་པ་ཞལ་གཅིག་པ་བསྒོམས་ཏེ། གཡས་དང་གཡོན་ན་རྡོ་རྗེ་དང་ཞགས་པ་བསྣམས་པའི་ཁྲོ་བོ། དེ་ནས་ཡང་ན་མཆེ་།
