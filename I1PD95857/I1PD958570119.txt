རྒྱུད། བའི། སྤྱི་ཚོར་མྱུ་ཁྱོད་ན། ཆ་ཀཱནྡ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱནྡ་ཀི་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་དུར་བྷཀྵི་ཀཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དུར་བྷཀྵ་ཀ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རོ་ག་ཀན་རཱི་ཡེ་ཏེ། ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རོ་ག་ཀན་ར་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤསྡུ་ཀནྟཱ་རཱི་ཡེ་ཧཱུྃ་་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤསྡུ་ཀནྟཱ་ར་ཀ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པ་ཎི་ཀནྟཱ་རཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་པ་ཎི་ཀན་ར་ཀ་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སི་ན་ཁཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཨ་སི་ན་ཁ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཏཱོར་ཎི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཏཱ་སིར་ཎ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀླུ་ར་དྷཱ་རཱི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་བཛྲ་ཀླུ་ར་དྷཱ་ར་ཨེཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙཀྲ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙཀྲ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་བྷནྡྷི་ཨཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀཱི་བྷཎྜ་ཧཱུཾ་ཧཱུཾ་ཕཊ། །རྫ་ཊ་ཚེ། ཨོཾ་བཛྲ་པཱུ་ཛ་ཙིནྟི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:པཱུ་ཛ་ཙིནྟཱ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་བྷཀྵ་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷཀྵ་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་:ནི་དྲ་ཙིནྟཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་བཛྲ་ནི་དྲ་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཱ་ཙྪེལཱ་སྱ་ཙིནྟི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཱ་ཙྪེལཱ་སྱ་ཙིན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨ་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨ་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷ་བ་ན་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོཾ་བཛྲ་བྷ་བ་ན་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཀྲྀ་ཧ་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གྲི་ཧ་རིན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སྟི་ཙིནྟི་ཡེ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སྟི་རིན་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨརྦ་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨརྦ་ཙིནྟ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཡོ་ག་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཡོ་ག་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་པུ་ཏྲ་ཙིནྟི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་ན་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤོ་ཀ་ཙིནྟི་ཡེ་ཨཻཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤོ་ཀ་ཙིནྟ་ཨེཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་དྷཱ་ན་ཙིནྟི་ཡེ་ཤུ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་དྷཱ་ན་ཙིནྟ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མནྡཱ་ཛ་པ་ཙིནྟི%ཡེ་ཧཱུྃ་་—།
