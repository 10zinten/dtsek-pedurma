ཨོཾ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བྷི་ར་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁྲི་་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁཎྜི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཛ་:ཊཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་རུ་དཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོ་་བཛྲ་ཌཱ་ཀ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་མེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་ཤྭ་ཌཱ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཁཎྜ་རོ་ཧེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པདྨ་ཊཱ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ྋཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རཏྣ་ཊཱ་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པྲ་་་ཙཎྜི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཁཎྜ་ཀ་པཱ་ལི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙཎྜཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་ཀཾ་ཀ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་པྲ་བྷཱ་ཝ་ཏི་ཡེ་ཙྪཱཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཀཾ་ཀ་ལ་ཨེཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་ནཱ་སེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་་བཛྲ་བི་ཀ་ཊ་དཾབྱཱ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱི་ར་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་རཱ་:བཱི་རི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཁརྦ་རཱི་ཡེ་ཨཻཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨ་མི་ཏཱ་བྷ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཀེ་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་པྲ་བྷ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་དྷུ་ཙེམཙྪཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་དེ་ཧ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨེ་ར་ཝ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཨཾ་ཀུ་རི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་བྷི་ར་ཝཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་ཛ་ཊི་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཱ་ཡུ་བེ་གེ་ཧཱུྃ་་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་བཱི་ར་དྨེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་ར་བྷཀྵི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། །ཨོ་བཛྲ་ཧཱུྃ་ཀཱ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཤུ་མ་དེ་ཝཱི་ཡེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་བྷ་དྲ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་སུ་བྷ་དཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་བྷ་དྲ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་བཛྲ་ཧ་ཡ་ཀནྡྷེཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་བྷི་ར་ཝེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བཛྲ་ཁ་གཱ་ན་ནེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་བི་རཱུ་པཱཀྵི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཙཀྲ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བཛྲ་མ་ཧཱ་བ་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་ཁཎྜ་རོ་ཧེ་ཧུཾ་ཧཱུཾ་།
