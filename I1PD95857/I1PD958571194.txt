ཀྱི་བྲག་ལ་ཟླ་ཁ་སྒྱུར་མ་པོ་བྱེད་ཀཏུ་ཀ་ཀ་མ་འཁབ། ར་པོ། བལྟ་ན་ཁ་མི་ཆ་ཁ་རྒྱན་། ན་ཏེ་ཚོགས་པ་ཁགྱི་ཁཀྱི་གར་འཆར་རྒྱུག་བྱེད་དང་། རྒྱན་ཁོ་ལ་ས་བ་། ལ་ཁ་བའི་::ཀ་ཁ་ག་ཛིན་ཁ་ལ་ཀ་ཀོ་ཁ་བ་ཞུ་རྒྱུན་ལ་སོང་། ཁ་བ་གང་སྐུ་ས་ཀར་དགོས་པ་གཙོ་ཚོན་པར་ཆ་ཞོ་འམ། །ཁཎའི་ཀ་ཁ་འབྲི་ཚོ་ཁ་ཁཱ་ལོ་ཀ་ཁ་དང་ཟླ་པ་ན་ན་ལ། གཅན་ཅཊིཀའ་ཆེ། །འགྲོ་བ་ར་། ལྷ་ས་སོགས་བཀའ་མ་དང་། །གྲོ་ག་བཀ་ཡིན། ཕྱི་ལ་སོགས་པ༄༅། །རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་ཐབས། ཕྱོགས་པོ་ལ་ད་ར་དང་འཁོད་ང་ཚོ་ཕྱི་ལ་ཕྱི་རྒྱུའི་སྟེང་ངའི་གཏམ། དཀྱིལ་དུ་ལོག་པ་ན་ཁ་ཊ་ཚོ་བན་བཀལ་ཏེ་མེད་གཞན་ལ་དེ་ད་པ་དང་པ་འདུ་ཤོག་ལས་བ་ལ་ལ་ཞི་བ།  ། སྐྱེ་བ་གནང་བ་ཁ་སྐོང་དང་སྐྱོབ་ཀྱི་ཀོ་གྲུ་ག་ནི། ཚེ་དང་བ་ལ། དང་པོ་ལ་པ་ར་ཁས་ལ་གླག་པ་དང་བག་ལེབ། བན་་::་བ་ཁྱོད་ལ་ཀུན་༄༅༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་གྷོ་ན་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རྡོ་རྗེ་ཕག་མོའི་སྒྲུབ་ཐབས། རྗེ་བཙུན་རྡོ་རྗེ་ཕག་མོ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ཡང་དང་པོར་སྒྲུབ་པ་པོས་བླ་མའི་སྤྱན་སྔར་གདམས་པ་ཐོབ་པར་བྱས་ནས། ཡིད་དུ་འོང་བའི་གནས་སུ་བདེ་བའི་སྟན་ལ་འདུག་སྟེ། སྐྱབས་སུ་འགྲོ་བ་སྔོན་དུ་བཏང་ནས༑ལ་ཚངས་པའི་གནས་བཞི་བསྒོམ་པར་བྱའོ། །དེའི་རྗེས་སུ་ཆོས་ཐམས་ཅད་སྟོང་པ་ཉིད་དུ་བསམས་ནས། ཨོཾ་ཤུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ་་་ཞེས་བྱ་བ་འདིས་བྱིན་གྱིས་བརླབ་པར་བྱའོ། །ཨོཾ་ཧཱུཾ་གྷཱ་ཏ་ཡ་སརྦ་དུཏྟཱན་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ་ཞེས་པས་མདུན་གྱི་རྟེན་གྱི་ལྷ་ལ་གཏོར་མ་ཕུལ་ནས། ཨོཾ་ཨཱཿཧཱུྃ་ཧཱུྃ་ཧ་ཧཾཞེས་པ་འདིས་ཤ་ཆེན་གྱི་ཕྱེ་མའི་བདུག་སྤོས་ཕུལ་ཏེ། བསམ་གཏན་བསྒོམ་པར་བྱའོ༑། དེ་ནས་སྟོང་པའི་ངང་ལས་ཧཱིདམར་པོ་ཡོངས་སུ་གྱུར་པ་ལས། བདག་ཉིད་རྡོ་རྗེ་ཕག་མོ་ཕག་གི་གདོང་ཅན་ཞལ་གཅིག་ཕྱག་བཞི་མ། སྤྱན་་་་་།
