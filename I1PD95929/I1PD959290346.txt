བསྟན་འགྱུར། འདུལ་བ། །༼དེཀྱི་སྣོད་གསུམ་ལས་འདུལ་བར་གཏོགས་པ་ཐམས་ཅད་དང༌། ཁྱད་དུ་གསོད་པའོ། །འདི་ལ་ཡན་ལག་ལྔས་དངོས་གཞིའི་ལྟུང་བར་འགྱུར་ཏེ། ཁྱད་དུ་གསོད་པའི་་་་་བསམ་པ་དང་། ཚིག་ཏུ་སྨྲ་བ་དང་། འདུལ་བར་གཏོགས་པ་ཉིད་དང་། རང་གི་ཚིག་དང་། གཞན་གྱིས་དོན་གོ་བའོ། །དེ་དག་ལས་ཡན་ལག་མ་ཚང་ན་ཅི་རིགས་པར་ན་སྨད་དོ། །ཁྱད་དུ་གསོད་པའི་ལྟུང་བྱེད་དེ་བཅུ་པའོ། །།ལ་། ང་ཁྱེད་སྡོམ་ལ། ས་བོན་འཕྱ་བ་བསྒོ་བ་དང་། །བྱ། །ཁྲི་དང་གདིང་བ་སྐྱོད་པ་་དང༌། །ཕྱིས་གནོན་འབྱུང་བ་འདེབས་པ་དང་། །རིམ་པ་གཉིས་སུ་རྩིག་པའོ༑། ཞེས་བྱ་བ་ནི། བཅུ་པ་གཉིས་པ་སྟེ། །སྡོམ་ལ་ཞེས་བྱ་བའི་ཚིག་གི་་དོན་དང་། དགོས་པ་ནི་སྔ་མ་བཞིན་ནོ། །བཅུ་པ་གཉིས་པ་འདིས་ཀྱང་ལྟུང་བྱེད་ཀྱི་ཆོས་བཅུ་བསྟན་ཏེ། ས་བོན་དང་སྐྱེ་བ་འཇིག་པ་དང་། །འཕྱ་བ་དང་། བསྒོ་བ་རྣ་ལ་གཟན་པ་དང༌། གནས་མལ་དང་གདིང་བ་དང་། སྐྱོད་པ་དང་། ཕྱིས་གནོན་པ་དང་། རྩ་བ་འབྱུང་བ་ལ་འདུག་པ་དང་། སྲོག་ཆགས་དང་་་་བཅས་པ་ལ་སྤྱོད་པ་དང་། གཙུག་ལག་ཁང་གི་ལྟུང་བྱེད་དོ། །དེ་ལ་མཁན་དེ་ལ་ས་བོན་དང་སྐྱེ་བ་འཇིག་པའི་ལྟུང་བྱེད་ཀྱི་གླེང་གཞི་ནི། སངས་རྒྱས་བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན། རྒྱལ་བུ་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་་་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ་ན་བཞུགས་སོ། །དེའི་ཚེ་ན་དགེ་སློང་དྲུག་སྡེ་དགེ། རང་གི་ལག་གིས་ཤིང་ལྗོན་པ་གཅོད་དུ་འཇུག་པར་ཡང་བྱེད། རྩྭ་སྔོན་པོ་རྔར་འཇུག་པར་ཡང་བྱེད། །མེ་ཏོག་དག་འཐོག་ཅིང་འཐོག་ཏུ་འཇུག་པར་ཡང་བྱེད་་ནས། གཞན་མུ་སྟེགས་ཅན་དག། ཤེས་ལྡན་དག་ཤཱཀྱའི་བུའི་དགེ་སྦྱོང་། འདི་དག་ནི་འདི་ལྟར་བྱེད་དེ། བྲམ་ཟེ་དང་ཁྱིམ་བདག་འདི་དག་ཀྱང་དེ་ལྟར་་།
