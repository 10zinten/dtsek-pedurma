བྱེད་པས། །ཤཱཀྱའི་བུའི་དགེ་སྦྱོང་དང་། བྲམ་ཟེ་དང་། ཁྱིམ་བདག་འདི་དག་ལ་བྱེ་བྲག་ནི་ཅི་ཡོད། བསམ་པ་ནི་ཅི་ཡོདི། ཐ་དད་དུ་བྱ་བ་ནི་ཅི་ཡོད། སུ་ཞིག་དགེ་སྦྱོང་མགོ་རེག་འདི་དག་ལ་བསོད་སྙོམས་སྦྱིན་པ་དག་བྱ་བར་སེམས་་་ཞེས་སྨོད་པར་བྱེད། ཕྱར་ཀ་གཏོང་བར་བྱེད། ཁ་ཟེར་བར་བྱེད་པའི་སྐབས་དེ་དགེ་སློང་རྣམས་ཀྱིས་བཅོམ་ལྡན་འདས་ལ་གསོལ་ཏོ། །དེ་ནས་བཅོམ་ལྡན་་འདས་ཀྱིས་སྔ་མ་བཞིན་དུ། ས་བོན་གྱི་ཚོགས་དང༌། འབྱུང་པོའི་གནས་་འཇིག་གམ་འཇིག་ཏུ་འཇུད་ན་ལྟུང་བྱེད་དོ་ཞེས་བཅའ་བ་མཛད་དེ། ཀ་ཡུལ་ནི་མཉན་ཡོད། གང་ཟག་ནི་དྲུག་སྡེ། ཉོན་མོངས་པ་ནི་འདོད་ཞེན། ཉེས་པ་ནི་ས་བོན་གྱི་ཚོགས་དང༌། །འབྱུང་པོའི་གནས་འཇིག་པའོ། །དེ་ལ་ས་བོན་གྱི་ཚོགས་ཞེས་བྱ་བ་ནི་ས་བོན་ལྔ་སྟེ། རྩ་བའི་ས་བོན་དང་། སྡོང་བུའི་ས་བོན་་དང༌། །ཐོ་ྋགུའི་ས་བོན་དང་། འགས་པའི་ས་བོན་དང་། ས་བོན་གྱི་ས་་བོན་ཉིད་ནི་ལྔ་པའོ། །རྩ་བའི་ས་བོན་གང་ཞེ་ན། གླ་སྒང་དང༌། ཤུ་དག་དང༌། །ཡུང་བ་དང་། སྒེའུ་གཤེར་དང་། བོང་ང་དཀར་པོ་དང༌། གཞན་ཡང་རྩ་བ་བཏབ་པ་ལས་རྩ་བ་ཉིད་སྐྱེ་བ་གང་ཡིན་པའོ། །སྡོང་བུའི་ས་་་བོན་གང་ཞེ་ན། ཤིང་ཀན་ཙ་ན་དང༌། :：ཀ་ལི་ལ་ཧ་དང་། །ཨ་ཤྭ་ད་ག་དང་། :：ཨུ་དུམ་བཱ་ར་དང་། ནྱ་གྲོ་དྷ་དང་། གཞན་ཡང་སྡོང་བུ་བཏབ་པ་ལས་སྡོང་བུ་ཉིད་སྐྱེ་བ་གང་ཡིན་པའོ། །ཐོ་གུའི་ས་བོན་གང་ཞེ་ན། བུ་རམ་ཤིང་དང་། འདམ་བུ་དང་། །སྨུག་མ་དང་། རྒྱ་མཚོ་ནས་འབྱུང་བ་དང་། །གཞན་ཡང་་ཐོ་ྋགུ་བཏབ་པ་ལས་ཐོ་གུ་ཉིད་སྐྱེ་བ་ཉིད་གང་ཡིན་པའོ། །།འགས་པའི་ས་བོན་གང་ཞེ་ན་ཨརྫ་ཀ་དང་། སརྫ་ཀ་དང་། ཕ་ནི་ནརྫ་ཀ་དང་། ་བྷུ་སྟི་ན་་དང༌། །འཛམ་བི་ར་དང་། གཞན་ཡང་འགས་པ་བཏབ་པ་ལས་འགས་པ་་_།
