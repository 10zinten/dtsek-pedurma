#། དེ་ཉིད་ཀྱི་ཕྱིར། དངོས་གྲུབ་མཚན་ཉིད་དབྱེར་མེད་པ། བྲི། །ཕྱག་རྒྱ་རྣལ་འབྱོར་པ་ཡིས་འགྲུབ། ངོ་། མཚན་ཉིད་ཅེས་པ་ནི། མཚན་ཉིད་དེ་དེའི་དངོས་གྲུབ་ནི་དངོས་གྲུབ་མཚན་་་་ཉིད་དོ། །ཇི་ལྟར་ཞེ་ན་དབྱེར་མེད་པ་ལས་ཏེ། ཁ་དབྱེར་མེད་པའི་རྒྱུའི་མཚན་ཉིད་ཀྱི་དངོས་གྲུབ་བོ། །ཕྱག་རྒྱ་ནི་ཕྱག་རྒྱ་ཆེན་པོའོ། །དེ་དང་ལྷན་ཅིག་རྣལ་འབྱོར་པ་འདུས་པ་གང་ལ་ཡོད་པ་དེས་ནི་ངེས་པར་འགྲུབ་བོ། དེ་། ཡང་ན་ཕྱག་རྒྱ་ཡང་ཡིན་ལ་རྣལ་འབྱོར་ཡང་ཡིན་པས་ཕྱག་རྒྱ་རྣལ་འབྱོར་པ་སྟེ་བདག་མེད་མ་་དང་དབྱེར་མེད་པའི་རྡོ་རྗེ་འཛིན་ནོ། །གྲུབ་པ་ནི་རྫོགས་པའོ། །ཀུན་རྫོབ་ཏུ་ཡང་མི་ཕྱེད་པའི་མཚན་ཉིད་ཀྱི་དངོས་གྲུབ་བོ། །ཀུན་ཏུ་ཀུན་ནས་ཀྱི་རྣམ་་་པས་གྲུབ་པ་ནི་གྲུབ་པའོ། །དབྱེར་མེད་པ་ཉིད་ཀྱི་མཚན་ཉིད་ནི་དབྱེར་མེད་་པའི་མཚན་ཉིད་དོ། །ཇི་ལྟར་སྤྲིན་དང་སྒྱུ་མ་དང་རྨི་ལམ་ལ་སོགས་པའི་དཔེ་དེས་གང་གི་ཚེ་དབྱེར་མེད་པའི་མཚན་ཉིད་དུ་གྱུར་པ་དེའི་ཚེ་ཕྱག་རྒྱའི་རྣལ་འབྱོར་པར་འགྱུར་རོ། །དེ་ནས་གྲུབ་པར་འགྱུར་རོ། །སྟོན་པས་དཀྱིལ་འཁོར་འདི་སྐད་གསུངས། །རྒྱུ་དེ་ལྟར་སྟོན་པས་འགྲོ་བའི་དོན་གྱི་ཕྱིར་འཇིག་པས་་་་དཀྱིལ་འཁོར་བཀའ་སྩལ་ཏེ། ལ་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་མེད་ན་གང་ལ་ཕྱག་་་་རྒྱའི་དངོས་གྲུབ་ཡོད་པ་མ་ཡིན་ནོ། །དཀྱིལ་འཁོར་དེ་ཡང་སྤྲིན་དང་འདྲ། །ཞེས་བྱ་བ་ལ་སོགས་པའི་དཔེས་དབྱེར་མེད་པ་དེ་གྲུབ་པར་བྱེད་པ་ཡིན་ནོ། །དཀྱིལ་འཁོར་དེ་ཉིད་གསུངས་པ། གྲུ་བཞི་ཡང་དག་འབར་བ་ཉིད། །སྒོ་་་་བཞི་གཟི་བརྗིད་ཆེན་པོ་སྟེ། ད་། དྲ་བ་དྲ་བ་ཕྱེད་པས་བརྒྱན། ་་སོ་། ཕྲེང་བ་སྣ་་་ཚོགས་རྔ་ཡབ་ལྡན། །པ། ཀ་བ་བརྒྱད་ཀྱིས་ཉེ་བར་མཛེས། །རྡོ་རྗེའི་སྲད་ཀྱི་བུ་་་མཉམ་པར་ལྡན། རྗེ་ཋ། མེ་ཏོག་སྣ་ཚོགས་ཉེ་བར་བརྒྱན། མ། བདུག་པ་མར་མེ་དེ་།
