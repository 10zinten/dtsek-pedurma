གདོན་ནི་འདིར་སྣང་སྟེ། ཨ། ལྟེ་བའི་བུ་ག་འོག་བལྟས་པའོ། །རྐན་ནི་མཚམས་ཀྱི་མདུད་པ་རུ། །།བུམ་པ་ཧ་ཡི་རྣམ་པ་ཉིད། །།རྩ་གསུམ་གྱི་ནི་ནང་རྣམས་སུ། །རྡོ་རྗེ་བཟླས་པ་དེས་རྡོ་རྗེ། །ཡི་གེ་གསུམ་དང་སྐུ་གསུམ་ཉིད། །ཐིག་ལེ་བརྒྱད་ནི་ཐིགས་པ་ཅན། །ཐམས་ཅད་དུ་ནི་འགྲོལ་བར་བྱེད། ་། ཕྱོགས་བརྒྱད་བརྒྱད་ཀྱིས་དབྱུག་ཆོད་ནི། །དྲུག་ཅུ་རྩ་བཞི་དུས་ཆེན་པོ། །སྒྲ་ཡི་ཚངས་པ་བསྐུལ་བའི་ཕྱིར། །ཡི་གེ་སུམ་ཅུ་གཉིས་ཀྱི་རྩ། །ཀ་སོགས་མཐར་ཐུག་དེ་དེ་བཞིན། །དེ་རྣམས་དབུས་སུ་ཧ་ཡིག་ནི། །ཐམས་ཅད་དུ་ནི་དགྲ་ཡིས་དགང་། །དགའ་བ་བཅུ་དྲུག་རྐན་ཅན་གྱིས། །རྗེས་སུ་མཐུན་པ་ཡིས་ནི་བརྡུང་། །དེ་ཡང་་་་དབྱངས་སུ་ཡང་དག་བཤད། །སྲོག་ནི་བཅུ་དྲུག་པོ་རྣམས་ནི། །འཕོ་བ་བཅུ་་དྲུག་པོ་འདི་རྣམས། :། འགག་པའི་དུས་ཀྱི་མཚམས་སུ་སྟེ། །།དེ་རུ་འདས་པར་་ཡང་དག་བཤད། །ཐིག་ལེ་སྒྲ་ནི་མཚོན་པར་བྱེད། །ཐིག་ལེ་བརྒྱད་དུ་དེ་ཤེས་བྱ། །སྒྲོལ་མ་ལ་སོགས་བརྒྱད་དུ་གྲགས། །དེ་ལྟར་དྲིལ་བུ་ཁམསགསུམ་པ། །གཅིག་གི་གཟུགས་སུ་རྟག་ཅེས་བྱ། །སྒྲ་ནི་སྒྲ་ཡིས་བརྗོད་པའི་ཕྱིར། ་། ་་་་་ཚོགས་དོན་ནི་བྱེད་པ་ཉིད། །ཏོ། །སེམས་ཅན་གཟུགས་ཀྱང་དེ་ཉིད་དུ། །ཆོས་ཀྱི་་སྐུ་ཉིད་བརྗོད་མེད་ཕྱིར། །རྡོ་རྗེ་མི་ཕྱེད་བ་ཡི་གཟུགས། །རྡོ་རྗེ་སེམས་དཔའ་དྲིལ་བདག་ཉིད། །དེ་ལྟར་བསྟན་པ་ཐམས་ཅད་ནི། །རྣལ་འབྱོར་པ་ཡིས་རྟག་ཏུ་བསྒོམ། །བསམ་དུ་མེད་པའི་ཆོས་ཉིད་འདི། །རྟག་ཏུ་རྒྱུ་དང་འབྲས་བདག་ཉིད༑། འདི་བཞིའི་དེ་ཉིད་ཆེན་པོ་ནི། །བསྒྲུབས་ལ་བྱང་ཆུབ་སྟེར་བྱེད་པ། །བདག་གིས་མདོར་རུ་བསྡུས་ཏེ་བཤད། ལ། རྒྱས་པར་གཞན་དུ་ཤེས་པར་བྱ། །ཀ་། ཞེས་བྱ་ྋསྟེ་ངེས་པའོ། ཁ་སྦྱོར་ཐིག་ལེའི་རྒྱུད་ཀྱི་རྒྱལ་པོའི་རྟོག་པ་བརྒྱད་པའི་རྡོ་རྗེ་དྲིལ་བུ་ལ་སོགས་པའི་དེ་ཁོ་ན་ཉིད་ཀྱི་རབ་ཏུ་བྱེད་པའི་རྒྱ་ཆེར་འགྲེལ་པ་ཡང་།
