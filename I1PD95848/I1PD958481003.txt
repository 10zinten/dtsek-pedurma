འུ་ལོ་ཀ་་བསྟན་འགྱུར། རྒྱུད། ༼ཙ་ལ་སྤྱི་སོགས་ཆ་ལག་དང་བ་ཀ་བ་ཀ་དང་་ཀ་ཁ་ག་ཚོད་་སྒྲུབ་པ་དང་བཟོ་བོ་ཁབ་ཏུང་། རྗེ་ཁྲོ་བ་ཤཎཱཚོཊ་་ཏ་་་་ང་འདར་བྲིས་ཀྱི། ཏེ་ཁན་ཊ་ཏུ་བསྡུར་མཆན། །ལ་བ་མཆན་ཁ། ་་་་། ཀ་བ་བཞི། ལོ་ལ་ང་ལ་ཀ་ཀཱ་ར་ཡང་ད་མཚན་བྱང་འདི་གཞུང་ནས་བླངས་ཏེ་བཀོད། ཉི«པེ་»«སྣར་པའི། «པེ་»«སྣར་»རེ་ཞིག«པེ་«སྣར་སྲེག«པེ་»«སྣར་»རྡོ་རྗེ། «སྣར་ཀྱིས། «པེ་»«སྣར། །/ཀྱིས་ཀྱང་གྱི«པེ་»སྣར་+ལ། ་ཚེ«པེ་»«སྣར་»ལྔ་པོཀ་དང་། ཁང་ཆུང་ཀུཀ་ཁོག་ཏུ་«པེ་»«སྣར་»དཀའ། «པེ་»«སྣར་»གིས། Q«པེ་»«སྣར་»ལ། «པེ་»«སྣར་—ལ། ་«པེ་《སྣར་»འདི། 《པེ་»«སྣར་»ཏེ། «སྣར་བ་འདི། །ཁ«པེ་སྒྲུབ་པ། སྣར་»སྒྲུབས། །«པེ་»—དུ། «པེ་»«སྣར་ཞེས། @«པེ་»«སྣར་»+པོ། «པེ་»«སྣར་བུ། @«པེ་»«སྣར་བྱེད། :::ལ། །«པེ་»བྱ་བ། «སྣར་»—ནི་བྱ་བ། «པེ་»«སྣར་གྱི། «པེ་»«སྣར་»གསུངས་པ། «པེ་»《སྣར་དཀར། «པེ་»«སྣར་ལི། པོ«པེ་《སྣར་གསུངས་པའོ། …༄༅«པེ་»《སྣར་+ལག་«པེ་»«སྣར་སྐྱེ། «པེ་»«སྣར་གད། «སྣར་+ཡང་། །ཟླ་?་ཡང། «པེ་»«སྣར་+ཕྱིར། །«པེ་»«སྣར་བརླབས། «སྣར་»བརླབས། «པེ་»«སྣར་»—ཐུགས་རྡོ་རྗེ། «པེ་»«སྣར་»བརླབས། «པེ་»སྣར་»མའི། །བ་མ་ཙ་དང་ལང་«པེ་»《སྣར་»འཁོར། @«པེ་»«སྣར་»གྱི། @«པེ་»སྣར་དུ། ®«པེ་»«སྣར་—ཞེས། །བློ་SS*SSལ་བས་བསྡུས་པའི་དཀྱིལ་འཁོར་པ་ཉིས་བོད་པ་ལ་དཀྱིལ་འཁོར་གཉིས་པོ་མེད་པ་ཡི་གནད་དུ་དཀྱིལ་འཁོར་གྱི་སྒོ་སྲུང་དཔོདོད་པར་བྱ་བ་མ། སུངས། སྒོ་སྲུང་། བ་ཞེས་བྱ་བ་ལ་བོ་དམ། སུ་པའོ། །བླ་བང་པོ་ལ་རབ་ཏུ་གྲགས། །དེ་ནི་དེང་དུས་སུ་ངས་པའི་ཧེ་རུ་ཀ་ལ་ཕྱག་མཚན་པ། ས་བོན་མ་ར་་ཞེས་བྱ་བ་ནི་ཆོས་ཀྱི་འབྱ་བ་ཞུས་ཏེ། དེའི་མཁའ་ནི་དེ་མ་མ་ཁའི་།
