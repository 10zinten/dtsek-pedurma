མཚན་མོ་རྒྱུ་བས་ནི་རྒྱུ་མ་སྐྱེད་དོ། ལ། གྲུབ་སྟེར་གྱིས་ནི་གྲང་བ་སྐྱེད་དོ། །།ཨུ་ད་རི་ནི་དར་ནིས་ལྟོ་བ་སྐྱེད་དོ། །།།ལ་ལ་ནཱ་ནིས་མི་བསྐྱོད་སྐྱེད་དོ། ལ་། ཨ་ཝ་དྷུ་་་ཏཱིས་ནི་ཤེས་རབ་ཟླ་བ་རྒྱུ་བ་སྐྱེད་དོ། །ཁྲག་རྒྱུ་བ་ནི་རོའོ། །དྲང་པོས་ནི་་་་་བཤང་བ་སྐྱེད་དོ། ཁ། །མདོག་ནག་གིས་ནི་ཤ་དང་པགས་པའི་སྲུབས་སྐྱེད་དོ། །ཐུན་མོང་གིས་ནི་མཆིལ་མ་སྐྱེད་དོ། འོ་། གཞིར་སྟེར་གྱིས་ནི་རྣ་སྦབས་སྐྱེད་དོ། །རྣལ་ཅི་བྲལ་གྱིས་ནི་ཁྲག་སྐྱེད་དོ། །དགའ་བས་ནི་ཚིལ་སྐྱེད་དོ། །འོ། །་་། གྲུབ་པས་ནི་མཆིལ་མ་སྐྱེད་དོ། །རྣམ་སྨིན་གྱིས་ནི་མིག་རྣག་སྐྱེད་དོ། །མེ་ཏོག་གིས་ནི་་་སྣབས་སྐྱེད་དོ། །འ་། མཁྲིས་གསུམ་གྱིས་ནི་གཟུགས་སྐྱེད་དོ། །འདོད་མས་ནི་སྐྲ་་་སྐྱེད་དོ། །ཁྱིམ་གྱིས་ནི་དྲི་སྐྱེད་དོ། །་། ཙཎྜིས་ནི་རོ་སྐྱེད་དོ། །ཀ་ཏ་ཡས་ནི་རེག་བྱ་སྐྱེད་དོ། །བདུད་ཀྱི་བུ་མོས་ནི་ཆོས་ཐམས་ཅད་རང་བཞིན་མེད་པ་སྟེ། །དེ་དག་ནི་རྩ་སུམ་ཅུ་རྩ་གཉིས་ཀྱི་ྋསྐྱེ་བ་དང་བསྐྱོད་པའོ། །དེ་རྣམས་ཀྱི་རྣམ་་་པར་དག་པ་ནི་ལྷ་མོ་སྟེ་བདག་མེད་མ་དང༌། རྡོ་རྗེ་མ་དང་། གོ་རཱི་མ་དང༌། །བ་རི་དང་། རྡོ་རྗེ་མཁའ་འགྲོ་མ་དང་། པུཀྐ་སཱི་དང་། ཤ་ཝ་རཱི་དང་། ་ཙཎྜ་ལཱི་དང་། ཌོམྦི་དང་། །གོ་རཱི་དང༌། ཙོ་རཱི་དང་། ་་་པེ་ཏཱ་ཙྪཱལཱི་དང༌། །ཁོ་གྷསྨ་རཱི་དང། ས་སྤྱོད་དང། མཁའ་སྤྱོད་དོ། །དེ་ལྷ་མོ་དེ་རྣམས་རྩ་གཉིས་གཉིས་ལ་དབང་ཞིང་དག་པའོ། །དེ་ལས་ལ་ལ་ནཱ་དང་། ར་ཉིས་ནཱ་དང༌། ས་ཨ་ཝ་དྷུ་ཏཱི་ནི་བདག་མེད་པའི་རྣལ་འབྱོར་མའོ། །ལྷག་མ་ཉི་ཤུ་རྩ་དགུ་ནི་འབད་པས་སྤང་བར་བྱའོ།  ོའི་། ཅིའི་ཕྱིར་ན་དེ་དག་ནི་སུས་ཀྱང་མི་རྟོགས་ཤིང་། ཁ་གཞིའི་དོན་གྱི་བྱ་བ་མི་ནུས་པའི་ཕྱིར་རོ། །གསོལ་བ། ་་་བཅོམ་ལྡན་འདས་ཇི་ལྟ་བུ་ལགས། །བཀའ་སྩལ་པ། སྲིད་གསུམ་མ་ལུས་ཡོངས་བསྒྱུར་ཅིང་། །ཁ། གཟུང་དང་འཛིན་པ་རྣམ་སྤངས་པའོ། ཐ་། ཞེས་པ་ལ། སྲིད་པ་གསུམ་ནི་་་་་།
