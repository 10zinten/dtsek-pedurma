ཁ་ཏུ་བསྟན་འགྱུར། རྒྱུད། ༼ང། སྤྱི་ཟ་མ་Nལརང་གིས་འཕྲོག་པོ་རང་ཉིད་བྱེད། ལྟ་། །རང་ཉིད་རྒྱལ་པོ་རང་གཙོ་བོའོ། །རང་གིས་འཕྲོག་ནི་རང་རིག་བསྡུ་བའི་རང་བཞིན་ནོ། འོ་། །རང་ཉིད་བྱེད་པོ་ནི་་རང་ཉིད་སྤྲོ་བའི་རང་བཞིན་ནོ། ལ་ཁ། རང་ཉིད་རྒྱལ་པོ་ནི་རང་ཉིད་རབ་ཏུ་གསལ་བའི་རང་བཞིན་ནོ། །རང་ཉིད་གཙོ་བོ་ཞེས་པ་ནི་ཐམས་ཅད་ཀྱི་བདག་པོ་དང་ལྡན་པའི་ཕྱིར། །སྣ་ཚོགས་གཟུགས་ཅན་གྱི་ནོར་བུ་དང་འདྲ་བར་རབ་ཏུ་གསལ་བ་སྤྲོ་བ་དང་བསྡུ་བའི་རང་བཞིན་ནོ། ཁ་། །འཕྲོག་ཅིང་གི་ཚུལ་གྱིས་གཟུགས་ལ་་་་སོགས་པའི་ཡུལ་གྱི་རང་བཞིན་ནོ། །''། བྱེད་པོའི་ཚུལ་གྱིས་མིག་ལ་སོགས་པའི་་་་དབང་པོའི་རང་བཞིན་ནོ། །ཚ:། རྒྱལ་པོའི་ཚུལ་གྱིས་རྣམ་པར་ཤེས་པའི་རང་་་་་་བཞིན་ནོ། a། །གཙོ་བོའི་ཚུལ་གྱིས་ཤེས་པ་དང་དབང་པོ་དང་ཡུལ་དབྱེར་མེད་པ་་་ཉིད་ཡང་དག་པའི་ཡེ་ཤེས་ཡིན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར་ཡེ་ཤེས་རང་ཉིད་བྱེད་པོ་ཡིན་པའི་ཕྱིར། ཁ་འདོད་ཆགས་ཞེ་སྡང་ཕྲག་དོག་དང་། །དེ་བཞིན་གཏི་་་མུག་ང་རྒྱལ་ཉིད། ཀ་པ། །དེ་ཀུན་དེའི་གནས་ཉམས་དགའ་བར། །།ཆ། བཅུ་དྲུག་་་ཆར་ཡང་མི་ཕོད་དོ། །འདོད་ཆགས་དང་ཞེ་སྡང་ལ་སོགས་པ་དེ་ཐམས་ཅད་དོ། །དེའི་གནས་ནི་ཡང་དག་པའི་ཡེ་ཤེས་རྟོགས་པའོ། །ཅིས་ཁྱད་པར་འཕགས་ཤེ་ན༑ཙཉམས་དགའ་བ་སྟེ། །འོ་ན་འདོད་ཆགས་ལ་སོགས་པ་ནི་ཉམས་དགའ་་བར་གྱུར་པ་སྟེ། ཡེ་ཤེས་ལྔའི་རང་བཞིན་དེའི་བདག་ཉིད་ཀྱིས་འབྱུང་སྟེ་ཡེ་ཤེས་ཉིད་ཀྱིས་མ་ནིང་དུ་བསྟན་ཏོ། །ཐམས་ཅད་ཀྱི་བཅུ་དྲུག་ཆར་ཡང་མི་ཕོད་ནི། བཅོ་ལྔའི་ཆའི་བདག་ཉིད་ཀྱི་ཟླ་བ་ཉིད་ཡིན་གྱི། །བཅུ་དྲུག་པའི་ཆ་དེ་ནི་མེད་པ་ཉིད་དེ་བྱ་བ་ཅི་ཡང་མེད་པའི་ཕྱིར་རོ། །།དེ་མི་ཕོད་པ་ནི་མཆོད་པ་མི་བྱེད་པའོ། །རང་རིག་པའི་བདག་ཉིད་ཀུན་ཉིད་དོ། ལོ་། བཅུ་དྲུག་ནི་སྟོང་པ་ཉིད་བཅུ་དྲུག་གི་རྣམ་པར་དག་པའོ། །ཆ་ནི། ཀ་བདེ་བ་དེ་ལེན་ཅིང་འཛིན་པར་བྱེད་པས་་།
