ལྟ་བའི་དྲན་སྣང་བ། །བསྟན་པའི་ཚིག་གི་སྦྱོར་བ་ཡིས། །སྐལ་ལྡན་དདང་། བདུད་དོ་ཕྱེད་དོ། །ངས་པ་མ། བདུན་པ། བར་དངོས་རྒྱ་ལ་ཅན། རྣམ་པ་ད་པ་ཉིད་ཡིན། བསྡུར་འབྲས་རེའུ་མིག་ཕྱེ་བ། མངའ་བློ་ལས་ར་ཐགཚེ་ཏེ། །ཤེལ་རྒྱུད། （བོད་ན་མ། །མནར་མེད་བསྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། +རྒྱུད་འགྲེལ། རྒྱུད། དཔེ་གཞི། སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། དུp:རྒྱུད། ཅ:ཚ(འོག་མཚ(འོག་མལེའུ། བ། ལྡེབ། ནབ། བརབ་བབ་ནབ་རཕྲེང་། །ཚེག་ཁྱིམ། :ཁག་ཞིག་ད་བར་སླེབས་པ་དེ་དཞེབ་ས། །མང་མང་དུ་ཡོད་པ་ཀུན་པོ་འཁོར་དཎྜ་ལན་ལུ་ཚོ་བ་ལ་ཁྲོ་བ་མེད་པ་འདས་པར་བསམ་བློ། །ན་དང་ཚིག་དེ་མཚན་ཉིnང་ག་འདྲ་ཡིན་པ། ད་ཆ་བལ་པོ། ཏོ་ར་ན་::ཀ་བ་བདེ་བ་འཚོ་ད་པོ། །མེ་ཡི་མ་རེག་། ཚེ་བསྒྱུར་ཏེག་ཤ་བ། ནས་ཤེས་བྱ་སྒྲོས་ཤིང། དབང་་སྒོ་ཚོ། ཁང་རྒྱུ་ནི་ཆེ་མཆ་ལྔཁོད། ལག་ག་ལ་ཡ་ང་བ་ཁ་ཚོ། །བོད། ཞེས་ལ།  ། ཚེ་འཥྛན་ལོ་བདུ་ཚཟླ་་ཚོད་དེ་ཡང་བོད་གོས་པ། །པ་ཊ་pཚོང་འདུག་པ་དང་དད་དག་ལ་ཁྱོབ། །འད། ཊི་ཏ་བསྐོ་བློ་འབྲིདད་པ་ཆེ་དད་དོནཎ་ཁ་ལོ་ཙ་པར་ཀ་ཁོལ་འཁུ་བའི་།
