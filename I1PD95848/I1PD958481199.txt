།་ལ་གླུ་ཞིག་རྒྱུ། ཀ་ཁཌ་བསྟན་འགྱུར། །རྒྱུད། ༼:ང་ཚོར་དང་པོ་ཀའི་ལ་ཕྱོགས་ན་བཀའ་མནན། ལ་། རབ་བསྒྲུབས་སྟོབས་ཀྱིས་དེ་མི་འགྲོ། ༄༅། །དགའ་མ་བརྩེ་བརྟན་དབང་ཕྱུག་མ། ་། ཡང་དག་བཀུག་ནས་ཉེར་ལོངས་སྤྱོད། །དེ་སོགས་མཐའ་མེད་དང་པོ་ནི། །ཆོས་དབྱིངས་མཉམ་དང་མི་མཉམ་པ། །ནམ་མཁའི་ཁམས་ཀྱི་མཐར་ཐུག་པར། ཁོ། རྡོ་རྗེ་བུ་མོ་མཆོག་ནི་གནོན། ཚོ། ནམ་མཁའ་ཀུན་གྱི་སྐབས་ན་་་་་དཔལ། །རྡོ་རྗེ་སེམས་དཔའ་རབ་ཏུ་བཞུགས། །བསྒྲུབས་པའི་རྟ་ཡི་སྦྱོར་བ་ཡི༑། སྐྱེས་བུ་གཞོན་ནུའི་སྒྲུབ་ཐབས་ཀྱིས། ང་། །མ་ལུས་འཇིག་རྟེན་གསུམ་རྒྱལ་པོ༑། རྡོ་རྗེ་འཛིན་བཅས་ཡང་དག་ཐོབ། །བྱང་ཆུབ་སེམས་ནི་མཐའ་མེད་་་བརྩེ༑ཞ། བཙན་ཐབས་སྦྱོར་བས་བསྒྲུབ་པར་བྱ། །ཐམས་ཅད་བདག་ལ་མོས་པ་ཡིས། །བསྒྲུབས་པའི་སྟོབས་ཀྱིས་དེ་མི་འགྲོ། །ཕ་རོལ་སྟོབས་ནི་མནན་པ་ཡིས། །སངས་རྒྱས་བྱང་ཆུབ་རབ་བརྙེས་བྱས། །བདུད་མེད་དཀའ་སྤྱོད་ཀྱིས་བསྒྲུབ་ཅིང་། །སངས་རྒྱས་བྱང་ཆུབ་བདེ་ཆེན་པོ། །དེ་ཡི་སྦྱོར་བའི་བདག་་་་ཉིད་ཐོབ། །གོ་འཕང་མཆོག་ཏུ་རྟག་པའོ། །སེམས་ཅན་ཁམས་ནི་སྣ་ཚོགས་པ། །མ་ལུས་པ་ནི་མངོན་དུ་བྱ། ་། །བདུད་མེད་དཀའ་སྤྱོད་བྱང་ཆུབ་པ། །ང་། སེམས་་་ཅན་རྣམས་ལ་གནས་བརྟན་སྟེར། ཟླ་། མ་ལུས་ལུས་པ་མེད་པ་ཡི། །རྟོག་པ་་་་ཐམས་ཅད་དབང་དུ་འགྱུར། །རྟ་ཡི་མཆོག་གི་རྣམ་འཕྲུལ་གྱིས། །སེམས་ཅན་དོན་ནི་ཐམས་ཅད་བྱེད། །དེ་ཕྱིར་དཔལ་ལྡན་རྟ་སོགས་ཀྱིས། ལ། རྟོག་པ་ཀུན་ནི་བླ་མེད་པས། །།ནམ་མཁའི་ཁམས་ནི་མཐར་ཐུག་པར། །སེམས་ཅན་ཁམས་ནི་རབ་ཏུ་བསྒྲུབས། འོ། །ཞེས་པ་ནི་བཅོམ་ལྡན་འདས་ཀྱི་ངེས་པའོ། །།དེ་སོགས་་མཐའ་མེད་དང་པོ་ནི། །ཆོས་དབྱིངས་མཉམ་དང་མི་མཉམ་པའོ། །ཆོས་ཀྱི་ཡེ་ཤེས་རང་བྱུང་བ། །།།ནམ་མཁའ་ཀུན་གྱི་སྐབས་ན་དཔལ། །།།རྡོ་རྗེ་སེམས་དཔའ་རབ་ཏུ་བཞུགས། ྋ་། ཐམས་ཅད་བདག་ཉིད་རབ་ཏུ་འཇུག་པ་ཆེ། །དངོས་གྲུབ་་་་་།
