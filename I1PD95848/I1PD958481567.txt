།ཁ་པར་དང་ར་བསྟན་འགྱུར། རྒྱུད། ༼ཅ༽ཁྱོབ་ཀརྨ་པ་སྐྱོ་ག་ཞེས་པ་ནི་ཨིའོ། །བསྡུས་ན་ཏི་ཞེས་འགྱུར་རོ། །གཉིས་པ་ཡི་ནི་དང་པོ་ཞེས་པ་ནི་ཀའོ། །ཉི་ཤུ་བདུན་པ་ཞེས་པ་ནི་རའོ། །དེ་ལ་ཙོ་རི་ཞེས་པ་ནི་ཨེ་དང་ལྷན་ཅིག་གཉིས་ཀ་རེ་ཞེས་འགྱུར་རོ། །གཉིས་པ་ཡི་ནི་བཞི་ཞེས་པ་ནི་གྷའོ། །རྡོ་རྗེ་མཁའ་འགྲོ་མ་སྟན་བྱ་ཞེས་པ་ནི་ཨུའོ། །་། དེ་གཉིས་ཀ་གླུ་ཞེས་འགྱུར་རོ། །གསུམ་པ་ཡི་ནི་དང་པོ་ཞེས་པ་ནི་ཙའོ། །ལན་གཉིས་བཟླས་ཞེས་པ་ནི་སྒྱུ་ཙ་ཞེས་པའི་སྔགས་ཀྱི་ཚིག་ལན་གཉིས་སུ་བཟླས་ཞེས་པའི་དོན་ཏོ། །དེ་བཞིན་དུ་གཉིས་པ་ཡི་ནི་བཞི་པ་ཞེས་པ་ནི་གྷའོ། །རྡོ་རྗེ་མཁའ་འགྲོ་མ་དང་སྦྱར། །ཞེས་པ་ནི་ཨུ་དང་ལྷན་ཅིག་གླུ་ཞེས་འགྱུར་རོ། །ཏི༌། །ཡི་གེ་བཅུ་གཅིག་ཅེས་པ་ནི་ཊའོ། །།དབྱངས་ཡིག་གསུམ་པས་བྱེ་བ་ཞེས་པ་ནི་ཨིའོ། ཟླ་། དེས་ནི་ཊི་ཞེས་འགྱུར་རོ། །ཚ། ལྔ་པ་ཡི་ནི་ལྔ་པ་ཞེས་པ་ནི་ནའོ། །དེ་ལ་གོ་རི་སྦྱར་བར་བྱ་ཞེས་པ་ནི་ཨིའོ། །དེས་ན་གཉིས་ཀ་ནི་ཞེས་འགྱུར་རོ། །གཉིས་པ་ཡི་ནི་བཞི་པ་ཞེས་པ་ནི་གྷའོ། །རྡོ་རྗེ་མ་ཡིས་མཆོག་ཏུ་བརྒྱན། །ཞེས་པ་ནི::ཨཱའོ། །དེས་ན་གཉིས་ཀ་ག་ཞེས་འགྱུར་རོ། །དེ་ནས་བཅུ་དྲུག་པ་ཞེས་པ་ནི་ཏའོ། །ཉི་ཤུ་རྩ་དྲུག་ཅེས་པ་ནི་ཡའོ། །ལན་གཉིས་བཟླས་ཞེས་པ་ནི་དེ་ལྟ་གྷ་ཏ་ཡ་ཞེས་པ་ནི་སྔགས་ཀྱི་ཚིག་ལན་གཉིས་སུ་བཟླས་པར་འགྱུར། དེ་བཞིན་དུ་གཉིས་པ་ཡི་ནི་བཞི་པ་ཞེས་པ་གྷའོ། །པ་བོ། དབྱངས་ཡིག་ལྔ་པས་བརྒྱན་པར་བྱ། །ཞེས་པ་ནི་ཨུ་སྟེ། བློ་གཉིས་ཀ་གླུ་ཞེས་་འགྱུར་རོ། །བཞི་པའི་དང་པོ་གང་ཡིན་པ་ནི་ཊའོ། །གོ་རཱི་དང་ནི་ཡང་དག་་་ལྡན། །ཞེས་པ་ནི་ཨི་སྟེ་དེས་ནི་ཊི་ཞེས་འགྱུར་རོ། །ལྔ་པའི་ལྔ་པ་ཞེས་པ་ནི་ནའོ༑། དེ་ལ་གོ་རཱི་སྦྱར་བར་བྱའོ། །བསྡུས་ན་ནི་ཞེས་འགྱུར་རོ། །དེ་ལྟར་བསྡུས་ན་སྔགས་ཨོཾ་ཤཾ་ཀ་རི:༔ཤིནྟ་ཀ་རེ་གླུ་ཙ་གླུ་ཙ་ཧྥུ་ཊི་ནི་གྷཱ་:ཏ་ཡ་གྷ་ཏ་ཡ་གླུ་ཊི་ནི་སྭཱ་ཧཱ་ཞེས་པའོ། །རྡོ་རྗེ་སྲུང་བ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། རྡོ་རྗེ་སྲུང་་་—།
