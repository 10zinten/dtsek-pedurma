ལོ་ན་རྒྱ་གར་གྱི་མཁན་པོ་སརྦ་ཛྙཱ་དེ་བ་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་:གཙང་་དེ་བེནྡྲ་རཀྵི་ཏས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པཚེ། ན་བ་འབྱའོਕਸੈਮਸਲਨਲਾਮਸਿੰਘਸੱਸਲੈAA.བསྡུར་མཆན། ནང་པ་བ། ན་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་འཕགས་པ་་་གཞི་ཐམས་ཅད་ཡོད་པར་སྨྲ་བའི་དགེ་ཚུལ་གྱི་ཚིག་ལེའུར་བྱས་པའི་འགྲེལ་པ་འོད་ལྡན་ཞེས་བྱ་བ་སློབ་དཔོན་ཤཱཀྱ་འོད་ཀྱི་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། @«པེ་»«སྣར་»སརྦ་སྟི་བ་ད་ཤཱ། ཨི«པེ་»«སྣར་»ཀཱ་རི་ཀ«ཅོ་»ཀཱ་རི་ཀཱ། «པེ་»«སྣར་»གཞི། «སྣར་»ལེའུ། «སྣར་»། །༄༅«པེ་»«སྣར་»སོགས། ཆེ«པེ་»«སྣར་»མཚོ། ཨ«པེ་»«སྣར་»རྒལ། «པེ་»«སྣར་»སྨྲ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»སྨྲ། «པེ་»«སྣར་»། ཨི«སྣར་»+དེ༑«པེ་»«སྣར་»བ། ཚེ«པེ་»«སྣར་མཐོ། ་ཚེ«པེ་»«སྣར་རྫུན། «པེ་»«སྣར་»སོགས། «པེ་»«སྣར་»ཆེ། «པེ་»«སྣར་ནོར། ཚེ«པེ་»མལ་ཆེ། «སྣར་»མེལ་ཆེ། «པེ་»«སྣར་»སྒྲར། «པེ་»«སྣར་»ཕྱིར། །ཚེ«པེ་»«སྣར་»—ཟས་ནི་བཟའ་བ་སྟེ་མ་ནིང་གི་རྟགས་སུ་བསྟན་ཏོ། །«པེ་»«སྣར་»—པ་ཡིན། Q«པེ་»«སྣར་»གྱི། །«པེ་»«སྣར་རིམ་གྱིས། ཨེ«པེ་»«སྣར་»བཞགི«སྣར་སྲོགས། «པེ་»«སྣར་བཞག་«པེ་»«སྣར་པར། «པེ་»«སྣར་བཞག་གོ། «པེ་»«སྣར་བཞགཚེ«པེ་»«སྣར་»+དང་། «པེ་»«སྣར་»+པ། «པེ་»«སྣར་»དེ་ལ་སྲོག་གཅོད་པའི་...སྨྱན་ལ་སོགས་པ་སྟེ་དེ་བཞིན་དུ། ཚུ«པེ་»«སྣར་བརྐོ། «སྣར་»སྤངས། «པེ་»འཛེག《སྣར་»འཛག«པེ་»«ཅོ་»རྩ། «པེ་»«སྣར་»གསོག@«ཅོ་»བསླབས། ཉི«པེ་»«སྣར་པར། །«སྣར་བསམ་མོ། @«པེ་»«སྣར་བརྐོ། @«སྣར་»བཏེང་། «པེ་»«སྣར་»འཛེག«པེ་»«སྣར་»གསོགཔོ«པེ་»«སྣར་»གཏོགས། «པེ་»«སྣར་»སོགས། «པེ་»«སྣར་»ཞེས། «པེ་»«སྣར་»འདྲེགཆེ«པེ་»«སྣར་»ཡང་། ལོ་སོ་«པེ་»«སྣར་—པ་ཡིན། ་དང་། ད་«པེ་»«སྣར་»གནས་ལ་མ་གུས་པ་ཉིད་ནི་ཉམས་པའི་ཕྱོགས་དང་མཐུན་པ་ཉིད་ཀྱང་ཡིན་ནོ། །«པེ་»«སྣར་»བསྡམས། «པེ་»«སྣར་ཕྱིར། «པེ་»«སྣར་»བཞག། །
