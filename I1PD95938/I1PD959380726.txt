རྒྱ་གར་གྱི་མཁན་པོ་བུདྡྷ་ཤཱནྟི་དང༌། :：ལོ་ཙཱ་བ་དགེ་སློང་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་:པ། འདི་ལ་བློ་ཀ་སྟོང་ཕྲག་ཕྱེད་་དང་བཞི་ཡོད་དོ། །དོ། །དེ་ལ་:::དེ་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ཚིག་ལེའུར་་་་བྱས་པ་སུམ་བརྒྱ་པའི་རྣམ་པར་བཤད་པ་སློབ་དཔོན་དུལ་བའི་ལྷས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་»ཏྲ། «པེ་»«སྣར་»ཅིང་། «པེ་《སྣར་»ཏེ། ཚུ«པེ་»«སྣར་གཞན། «པེ་»«སྣར་»གྱི། @«པེ་»«སྣར་»+དོན་ཐོབ་པའི། «པེ་»«སྣར་ན་བྱུང་བ་ཞེས་བྱུང་ཏེ། «པེ་»«སྣར་བ་ལའམ། །«སྣར་བ་ལའམ། ར་་་«པེ་»«སྣར་བའི། «པེ་»«སྣར་»+བར། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་»ཉམ། «སྣར་བསྟན། །«པེ་»མིན། Q«པེ་»«སྣར་+ནོ། །ཚེ«པེ་»«སྣར་»ཅི། «པེ་»«སྣར་»མི་ཟ་བ། ཚེ«པེ་»《སྣར་»—ལ། «པེ་»«སྣར་»ནོ། །«པེ་»འཇིགས། O«པེ་»སྣར་»ཞེས། «པེ་»《སྣར་»སྤངས་པར། «པེ་»«སྣར་»—ཐར་པ། «སྣར་»ཡང་། «པེ་»«སྣར་»བུ། «སྣར་»ལ། «པེ་»«སྣར་གྱི། །«སྣར་»ཀྱིས། «པེ་»«སྣར་»+ལ། @«པེ་»«སྣར་གྱི། «སྣར་»འཇིགས། @«པེ་»«སྣར་བུའི། །«སྣར་ཅད། «སྣར་»ལྡག«པེ་»བཟུང་། «པེ་»«སྣར་རྫུན། ཚུ«པེ་»རྫུན། །ཚེ«པེ་»ཁྱེད་པར། 《སྣར་»ཁྱད་པར་དང་། ཚེ«པེ་»«སྣར་»གཞི། ཚི«པེ་»«སྣར་»དྲོ། «པེ་»«སྣར་»གྱི། «སྣར་»ལྡོགས། «པེ་»བའི། །«པེ་»«སྣར་»གིས། «པེ་»«སྣར་ཐབས། «སྣར་»པའི། ཆེ«པེ་»«སྣར་»—དེ། «པེ་»«སྣར་»+པ། «སྣར་»མོང་། ཚ«པེ་»«སྣར་»སྤངས་པར། །«པེ་»«སྣར་»དེ། ་ཚེ«པེ་»«སྣར་གྱུར། «པེ་»«སྣར་»དོན་དེ་བཞིན་དུ། ཆེ«པེ་»《སྣར་»བརྗོད། «སྣར་»བྱས། QQ&«པེ་»«སྣར་བཞག་ཚུ«སྣར་»གཅད། O«པེ་»《སྣར་»དེ་དང་། —།
