བསྟན་འགྱུར། འདུལ་བ། ༼གུ་ཐོབ་པ་དེ་ཡིས་ནི། །འཇིག་རྟེན་དགེ་བཤེས་བྱེད་པར་ཤོག། འདུལ་བ་དྲི་བའི་ཚིག་ལེའུར་བྱས་པ་ལྔ་པའི་གཏན་ལ་དབབ་པ་རྫོགས་ཏེ། འདུལ་བ་དྲི་་་བའི་ཚིག་ལེའུར་བྱས་པའི་གཏན་ལ་དབབ་པ། སློབ་དཔོན་མདོ་སྡེ་འཛིན་པ་་་ཀ་:ལྱཱ་ཎ་མི་ཏྲས་མཛད་པ་རྫོགས་སོ། །།།།གལ་ཏེ་བདག་ནི་རྨོངས་པ་ཡིས། །ལན་ཞིག་ལེགས་པར་མ་ཐེབས་ན། །འདི་ནི་ཕྱོགས་ཙམ་བསྟན་པར་བྱས། །མཁས་པས་བདེ་ལེགས་གདབ་ཏུ་གསོལ། །སྡུག་ཅིང་གདུང་བ་སེལ་བྱེད་པ། །ཇི་ལྟར་འདོད་པའི་འབྲས་བུ་སྟེར། །གྲིབ་་མ་སྟུག་པོའི་ཤིང་མཆོག་ལྟར། །གཞན་དོན་དགའ་བའི་ཕྱིར་འཁྲུངས་པ། །ཀུན་དགའ་གོ་ཆས་བདག་ལ་ནི། །ཤིན་ཏུ་བཅོལ་ཏེ་དེ་ཡི་ཕྱིར། །ཅི་དྲན་པ་་་ཞིག་འབད་དེ་བྱས། །ང་རྒྱལ་གྱི་ནི་ཕྱིར་མ་:ཡིན། །རྒྱ་གར་གྱི་མཁན་པོ་་་དྷརྨ་ཀ་ར་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་དཔལ་བརྩེགས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་:ཕབ་པ། བློ་ཀ་སྟོང་དྲུག་བརྒྱ་དྲུག་ཅུ་རྩ་བཞི་ཡོད་དོ། །།།བསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་ལ་འདུལ་བ་དྲི་བའི་རང་འགྲེལ་འགྲེལ་པ་ཀལྕན་མི་ཏྲའི་མཛད་པ་བཞུགས་སོ། །ཞེས་དང་། «སྣར་ལ་འདུལ་བ་དྲི་བའི་རང་འགྲེལ་་་་འགྲེལ་པ་ཀ་ལྕན་མི་ཏྲ། ཞེས་འཁོད། 《«པེ་»«སྣར་བ། «པེ་»«སྣར་དེ། ︽«པེ་《སྣར་»དབྱེ་བ། །«སྣར་»སྐྱེས་པས། «པེ་»«སྣར་»གི། «པེ་»«ཅོ་སྨོས། «པེ་»«སྣར་»གཉིས། «པེ་»《སྣར་»+པ། །«པེ་»«སྣར་བཏུང་བ། «པེ་»«སྣར་བསླང་། «པེ་»«སྣར་»བ། «པེ་»«སྣར་»སོ། །«པེ་འདི་སློབ། སྣར་»འདི་སོག@«པེ་»«སྣར་བསྟོན། «པེ་»«སྣར་»པར། «པེ་»《སྣར་»སྔོན། «པེ་»«སྣར་པ། («པེ་《སྣར་བཏུད། །འགྱེ《སྣར་བཞན། ་མུ«པེ་»།
