རྒྱུད། ་་་ཚོརྒྱ་གར་གྱི་མཁན་པོ་ཤྲདྡྷཱ་ཀ་ར་ཝརྨ་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཏ་ལྔ། ལྔ་ཁ་ཊ་རྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་ཛྙཱ་ནཱ་ཀ་ར་བས་བཤད་ནས། བོད་ཀྱི་ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་འགོས་ལྷས་བཙས་ཀྱིས་ཞུ་གཏུགས་གཡར་ཁྲལ་འཚལ་བའོ།  ོ། ཡང་སླད་ཀྱིས་རྒྱ་གར་གྱི་པཎྜི་ཏ་ཆེན་པོ་ནག་པོའི་ཞལ་སྔ་ནས་ཀྱིས་བཤད་ནས། ་བོད་ཀྱི་ལོ་ཙཱ་བ་འགོས་ལྷས་བཙས་ཀྱིས་ཡུལ་དབུས་ཀྱི་དཔེ་དང་གཏུགས་ཏེ་བཅོས་པའོ། །།ਭਰੀਕਰਨਵBERAਉਸੰਸਕਾਲਰ,ཞེས་བསྒྱུར་མཆན་ཝ་ཆུ་འབབ་པ་ਲੈਸੋਹਲਨੇਵਰਲਪੂਲਾਤੋਂਪਵਨམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད། Q«པེ་»«སྣར་»ཀ(Q«པེ་»«སྣར་»ཀྱིས། ཀྱ®«པེ་»《སྣར་ཕྱེས། «པེ་»«སྣར་པས། «པེ་»«སྣར་གྱུར། ༄༅«པེ་»《སྣར་འདུལ། «པེ་ལ། །«པེ་»«སྣར་»སྒྲུབ། «པེ་»«སྣར་ཡི། «པེ་»«སྣར་མཐར། ༄༅«པེ་»«སྣར་»པོས། །བས/འ«པེ་《སྣར་བཀའ། «པེ་《སྣར་བ་བྱ། གླེང་། @«པེ་»«སྣར་ཡི། ད་བ་ས་གྲ། ཏུ་(«སྣར་»+དོ། «པེ་»«སྣར་»པ་ཡིན། «པེ་»«སྣར་»གྱི། །ཟླ་ལ་འས་པ་བབལ་དང་«པེ་»«སྣར་»ཞུགས། «པེ་»«སྣར་དང༌། «པེ་»«སྣར་»རྟོགས། «པེ་»«སྣར་སོགས། «སྣར་ཟུང་། «པེ་»«སྣར་»རལ། «པེ་»«སྣར་»ཚིགས་མེད་འོད། ངའི་(«སྣར་པ། «པེ་»«སྣར་»ཤེས། ནི«པེ་»《སྣར་རིགས། ཆོ«པེ་»«སྣར་རིག། «པེ་«སྣར་བརྟགས། «པེ་»«སྣར་»པའི། །«པེ་»«སྣར་སྐྱེད། «པེ་»«སྣར་རྣམ་པ་བཞིའི། ང་བ་/.ཀཱ་«པེ་»«སྣར་བསླུ། <«པེ་《སྣར་»འདོད་ཆགས། «པེ་»《སྣར་»+། ®«པེ་»《སྣར་»མའོ། ཀྱི«སྣར་»ན། «སྣར་བྱིག་«སྣར་དྷིཀ«པེ་»«སྣར་ལིཀགྱི«པེ་རླབས། །«སྣར་»རླབ། «པེ་»«སྣར་»+དུ། «པེ་»«སྣར་»མའོ། «པེ་»«སྣར་»ཡིས། །ར་པ། །«སྣར་དཔའ། «སྣར་»དྷརྨཱ། «པེ་»«སྣར་»དུ། «པེ་»«སྣར་»ཏོ། ལས་«པེ་»«སྣར་ཡོང་། འ«པེ་»«སྣར་»དེ། «པེ་»«སྣར་བསྟན། «པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་སྲེད། Q«པེ་»«སྣར་དུ། «པེ་»«སྣར་»འཐོབ། «པེ་»«སྣར་»ལས། །«པེ་»སྣར་འཐོབ། །།
