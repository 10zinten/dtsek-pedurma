ཡ་གྷ་ཊེ་སྦྱོ་ན་མ། ཨོཾ་:བི་ཛ་ཡ་གྷ་ཊེ་གྷོ་ན་མ། ཨོཾ་སརྦ་ཙིནྡྷེ་ཙྪཱབྷོ་ན་མ། ཨོཾ་སརྦ་མུ་དྲྭ་བློ་ན་མ། ཨོ་ཙིན་ཏཱ་:མ་ཎ་ཡེ་ན་མ། ཨོཾ་དྷརྨ་ག་ཎ་ཊཱི་ན་མ། ཨོཾ་དྷརྨ་ཤཾ་ཁཱ་ཡ་ན་མ། །ཨོཾ་ཀལྤ་དྲྀ་གྷ་ཡ་ན་མ། ཨོཾ་ཨཀྵ་བྷྱཱ་ཡ་ན་མ། ཨོཾ་ཨ་མོ་གྷ་:སིད་དྷ་ཡེ་ན་མ། ཨོཾ་རཏྣ་སཾ་བྷ་བཱ་ཡ་ན་མ། །ཨོཾ་ཨ་མི་ཏཱ་བྷཱ་ཡ་ན་མ༑ཨོ་བི་རོ་ཙ་ནཱ་ཡ་ན་མ། ཨོ་ལོ་ཙ་ནཱ་ཡི་ན་མ། ཨོཾ་མཱ་མ་ཀྱི་་ན་མ། ཨོཾ་པཱཎྜ་ྋརཱ་ཡི་ན་མ། ་ཨོཾ་ཏཱ་རཱ་ཡི་ན་མ། །ཨོཾ་བཛྲ་དྷཱ་ཏཱི་ཤཱརྨེ་ན་མ། ཨོཾ་བཛྲ་པཱ་ཎ་ཡེ་ན་མ། །།ཨོཾ་ཁ་གནྟཱ་ཡ་ན་མ། ཨོཾ་ཀྵི་ཏི་གནྟཱ་ཡ་ན་མ། ཨོ་ལོ་ཀེ་ཤྭ་རཱ་ཡ་ན་མ། །ཨོཾ་སརྦ་ནི་བ་ར་ཎ་བི་སྐམ་བྷི་ནེ་ན་མ། ཨོཾ་ས་མནྟ་བྷ་དཱ་ཡ་ན་མ། ཨོ་གན་དྷ་བཛཱ་ཡི་ན་མ། ་ཨོ་རཱུ་པ་བཛཱ་ཡི་ན་མ། ཨོ་ར་ས་བཛཱ་ཡི་ན་མ། ཨོཾ་:སྤར་ཤ་བཛཱ་ཡི་ན་མ། ཨོཾ་ཤཔྟ་བཛཱ་ཡི་ན་མ། ཨོ་་་་དྷརྨ་དྷཱ་ཏུ་བཛཱ་ཡི་ན་མ། ཨོཾ་ཨུཏྟཱི་ཥཱ་་ཡ་ན་མ། ཨོཾ་བིགྷཱན་ཏ་ཀཱ་ཡ་ན་མ། ཨོཾ་པྲཛྙཱནྟ་ཀཱ་ཡ་ན་མ། །ཨོཾ་པདྨནྟ་ཀཱ་ཡ་ན་མ། ཨོཾ་ཡ་མཱནྟ་ཀཱ་ཡ་ན་མ། ཨོཾ་སྟམ་བྷི་རྩེ་ན་མཿ། ཨོཾ་མཱ་ནི་རྩེ་ན་མ། ཨོ་སྟོ་བྷི་རྩེ་ན་མ། ཨོཾ་ཨ་ཏི་བཱིནཱ་ཡི་ན་མ། ཨོཾ་ཨ་ཏི་ནཱི་ལཱ་ཡི་ན་མ། ཨོཾ་སརྦ་དྷཱ་རི་ཎཱི་བྷོ་ན་མ། ཨོཾ་:ཥ་ཌོ་གཱ་ཡ་ན་མ། །ཞེས་པ་ནི་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ལ་མཆོད་པའི་ཆོ་གའོ། །དེ་་ནས་གསུང་གི་དཀྱིལ་འཁོར་ལ་འདི་ལྟ་སྟེ། ཨོཾ་བཛྲ་ཙརྩི་ཀཱ་ཡི་ན་མ། ཨོཾ་བཛྲ་བཱ་རཱ་གྱི་ན་མ། ཨོཾ་བཛྲ་མ་ཧེ་ཚེ་ན་མ། ཨོཾ་བཛྲ་:ཨེ་ན་བྱེ་ན་མ། ཨོཾ་བཛྲ་པྲདྨ་རྩེ་ན་མ། ཨོཾ་བཛྲ་མ་ཧཱ་ལགླུ་ན་མ། ཨོཾ་བཛྲ་ཀོ་མཱརྨི་ན་མ། ཨོཾ་བཛྲ་བིཥྛ་བྱི་ན་མཿ། ཨོཾ་ཨཏྠ་ཨཊྚ་ཀེ་ན་ཙ་ཏུ། །ཥཊྛི་བཛྲ་ཡོ་གི་ནི་སྦྱོ་ན་མ། །ཨོཾ་ཥཊ་ཏྲི་*ཤཏ་ཨིཙྪཱ་བློ་ན་མ། ཞེས་པ་ནི་གསུང་གི་དཀྱིལ་འཁོར་ལ་མཆོད་པའི་ཆོ་གའོ། །དེ་ནས་སྐུའི་དཀྱིལ་འཁོར་ལ་འདི་ལྟ་སྟེ། ཨོཾ་བཛྲ་བིཥྛ་བེ་ན་མ། HOTOlc།
