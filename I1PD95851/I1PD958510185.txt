།ཀ་ལ་ཀ་ཁོ་ན་བསྟན་འགྱུར། རྒྱུད། ༼ཐིཀ་ལོ་ཀ་ཐོནཁར་པདྨའི་དབུས་སུ་རི་བོང་ཅན་དང་ཉི་མ་མེ་ཡི་སྟེང་དུ་རྡོ་རྗེ་ནི། །ཧཱུྃ་ཡིག་ཡེ་ཤེས་ལས་སྐྱེས་འཇིག་དུས་སྤྲིན་དང་མཚུངས་པ་རྩེ་ལྔ་འོད་ཟེར་དང་བཅས་དགོད་པར་བྱ། །དེ་ཡི་དབུས་སུ་ཛཡི་ལྕགས་ཀྱུའི་འོད་ཟེར་རྣམས་ཀྱིས་སྲིད་པ་གསུམ་པོ་མཐའ་དག་བཀང་བྱས་ནས། །རྣམ་གསུམ་སྲིད་པ་ལ་གནས་ཡེ་ཤེས་འཁོར་ལོ་བཀུག་ནས་རྡོ་རྗེའི་ལམ་དུ་རབ་ཏུ་གཞུག་བྱ་སྟེ། །ཐམས་ཅད་ཟླ་བ་ཞུ་བ་་་་དང་མཚུངས་རང་གི་རྡོ་རྗེའི་ཁ་ནས་ཡུམ་གྱི་པདྨར་སྤྲོ་བར་བྱ། །འོ། །དེ་ནི་ཉི་མར་་་རབ་ཏུ་ཞུགས་པ་རོ་མཉམ་དག་ཏུ་འགྱུར་ཏེ་ཨ་སོགས་ཀ་སོགས་རབ་ལྡན་པ། །དེ་ཡི་དབུས་སུ་ཞི་བ་ལ་སོགས་ལས་ཀྱི་ཡེ་ཤེས་ས་བོན་དུ་འགྱུར་རིགས་ཀྱི་དབང་་་་་གིས་སོ། ་། དེ་ཡིས་རྫོགས་པའི་ལྷ་མོའམ་ནི་ལྷ་ཡང་རྣལ་འབྱོར་པ་ལ་འབྲས་བུ་་་་སྟེར་བ་ཉིད་དུ་འགྱུར། །ཛཧཱུྃ་བཾ་ཧོརིམ་པས་ལྕགས་ཀྱུ་ཞེས་དང་རྡོ་རྗེ་་་་དང་ནི་རྡོ་རྗེ་ཞགས་པ་དྲིལ་བུ་སྟེ། །།ཨོཾ་ཨཱཿཧཱུཾ་ཧོདེ་བཞིན་དུ་གསུངས་ཟླ་བ་ཉི་མ་རྡོ་རྗེ་དེ་བཞིན་ཉིད་དུ་འགྱུར་མེད་དོ། ལོ་ག། ཨཱི་རཱི་ཨཱུ་ལཱ་དེ་བཞིན་ཉིད་དུ་རབ་་གསལ་ཡ་ར་ཝ་ལ་རླུང་མེ་ཆུ་དང་ས་རྣམས་སོ། a། །ཧཧཱུཾ་ཧཾ་ཕྲོ་ཉི་མ་དང་ནི་རྡོ་རྗེ་ཟླ་བ་དག་དང་གྲི་གུག་ཏུ་ཡང་དེ་བཞིན་གསུངས། །ཟླ་བ་ཉི་མའི་དབུས་སུ་ཨཱ་ལི་ཀཱ་ལི་དང་བཅས་ཆུ་ཡི་ས་བོན་བདག་ཉིད་ཆུ་སྐྱེས་ཏེ། །དེ་ལས་རྫོགས་པའི་ལྷ་མོ་ཟླ་བའི་མདོག་ཅན་ཞལ་གཅིག་མ་སྟེ་ཟུང་གི་ཕྱག་གི་པདྨ་ཅན། །གླང་ཆེན་དཀར་པོ་ལ་ཞོན་ཕྱག་ན་ཆུ་སྐྱེས་དཀར་དང་མི་འཇིགས་པ་དང་གོས་ནི་དཀར་པོ་དང་། །དཀར་བའི་རྒྱན་དང་ལྡན་ཞིང་རབ་ཏུ་བཞད་པའི་ཞལ་ནི་བསྒོམས་ནས་བསྒྲུབ་བྱའི་ཁྱིམ་དུ་བསྐུལ་བར་བྱ། །དེ་ནས་བསྒྲུབ་བྱ་བཟུང་ནས་སླར་ཡང་ཁྱབ་བདག་དཀྱིལ་འཁོར་དུ་ནི་ཡང་དག་རབ་ཏུ་གཞུག་པར་བྱའོ། །།གཙོ་བོའི་་་་བཀའ་ནི་རབ་ཏུ་ཐོབ་ནས་སྤྱན་མ་ལ་སོགས་རབ་དགྱེས་རྣམས་ཀྱིས་བདུད་རྩིའི་་་་།
