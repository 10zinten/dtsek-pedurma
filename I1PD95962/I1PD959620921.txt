བྱོར་ཁག་ག་བ་བློ་ལ་ཨིཊ་བྱིན་པས། ཧཱ་ཡ་ཡི་ཥྱ་ཏི། སོགས་དང༌། ཁྲིང་ལ། 《ཨ་ཧཱ་ཡ་ཡི་་་ཏེ༑སོགས་སོ། །།ཞེས་པ༈སྤུ་ན་ཡོངས་སུ་ཉམས་པ་ལའོ་ཞེས་པ་ལ། རྒྱུ་དང་བཅས་པ་སོགས་ཀྱིས་ཎིཙ་སྦྱར། ཨུ་ནི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ོལཊ'སྤུ་ན་ཡ་ཏི། ལིང་སྤུ་ན་ཡེཏ། ལོཊཨཱ་ན་ཡ་ཏུ། ལང་ཨོ་ན་ཡཏ། སོགས་སོ། །ལུང་ལ། ནོ་ན་ཡ་ཏི་དྷཱ་ན་ཡ་ཀྱི་ལ་པ་ཏྱརྡ་ཡ་ཏི་བྷཱི། ཨུ་ན་ཡ་ཏི་དྷཱ་ན་ཡ་ཏི་ཨེ་ལ་ཡ་ཏི་ཨརྡ་ཡ་ཏི་རྣམས་ལས་མ་ཡིན་ནོ། །ཨཱུ་ན་ཡོངས་སུ་ཉམས་པ། དྷ་ན་སྒྲ། ཨེ་ལ་དང་ཨརྡ་འགྲོ་བ་རྣམས་ཎིཙའི་མཐའ་ཅན་ཕ་རོལ་ལས་ཙྪི་ཙང་དུ་བསྒྱུར་བ་མ་་ཡིན་ཏེ། རྨི་སི་ཙར་བསྒྱུར་ཨིཊ་བྱིན་པས། ཨོ་ན་ཡཱཏ། ཨོ་ན་ཡི་སིཾ། ཨོ་་ན་ཡི་། །སོགས་སོ། །འགྲེལ་པར་སྐབས་འགའ་ཞིག་ཏུ་སྡེབ་སྦྱོར་སོགས་ཀྱི་དགོས་པ་ཡོད་ན་ཙང་དུ་བསྒྱུར་བར་བཤད་པས། ་ཙང་དུ་བསྒྱུར་བའི་ཕྱོགས་ལ་ཎིཙ་དང་བཅས་པ་གཉིས་བརྗོད། ་ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས། ཨོ་ནི་ནཏ། ཨོ་ནི་ན་ཏྭཾ། ཨོ་ནི་ནན། སོགས་སོ། །ལིཊ་ལ་ཨཱམ་སྦྱར་བས། ཨུ་ན་ཡཾ་ཙ་ཀཱ་ར། སོགས་སོ། །ལུཊ་ལ་ཨིཊ་བྱིན་པས། སྤུ་ན་ཡི་ཏཱ། སྤུ་ན་ཡི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཎི་ཕྱིས་པས། ཨཱི་སཱུཏི། ཨུ་་ཙཙཱ༑ཨུ་རྩྭ་སུ། །སོགས་སོ། །ཁྲིཊ་ལ་ཨིཊ་བྱིན་པས། སྐུ་ན་ཡི་'ཏི། སོགས་དང༌། དེ་བཞིན་དུ། རྀང་ལ། ཨོ་ན་ཡི་ཏ། སོགས་སོ། །༈དྷཱ་ན་སྒྲ་ལའོ་ཞེས་པ་ལའང་ཎིཙ་སྦྱར། དྷ་ནི་ཞེས་པ་ལ་ལཊ་སོགས་་སྦྱར་བས། ལཊ་དྷཱ་ན་ཡ་ཏི། ལིང་དྷ་ན་ཡེཏ། ལོཊ་དྷཱ་ན་ཡ་ཏུ། ལང་ཨ་དྷ་ན་ཡཏ། །སོགས་དང་། ལུང་ལ། ཙྪིཙང་དང་སིཙ་ཡང་ན་བར་བྱས་་་པས། སིཙའི་ཕྱོགས་ལ་ཨིཊ་བྱིན་པས། ཨ་དྷཱ་ན་པཱིཏ། ཨ་དྷཱ་ན་ཡིདྡྷཾ། །ཨ་—།
