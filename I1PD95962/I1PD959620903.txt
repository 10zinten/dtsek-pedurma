ཚོ་ཁ་ར་ཏི་ཀ་ཏོ་ར། ཙོ་ར་ཡི་ཏཱ་སི། སོགས་དང་། ཎིཙ་ལས་ཀྱང་ངོས། ཏང་ལ། ཙ་ར་ཡི་ཏཱ། ཙོ་ར་ཡི་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཙ་ཕྱིས་པས། ཙོནཱ་ཏ། ཙོན་ཧཱུཾ། ཙོནཱ་སུ། སོགས་དང༌། ཎི་ཙ་ལས་ཀྱང་ངོས་ཏང་རྣམས་སྦྱོར་ན། བལ་དང་པོའི་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཙོ་ར་ཡི་ཥིཉྩ། ཙོ་ར་ཡི་ཏཱི་ཡཱནྟཾ། ཙོ་ར་ཡི་ཁཱི་རན། །་སོགས་སོ། །ལོཊ་་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཙོ་ར་ཡི་ཥྛ་ཏི། སོགས་དང་། ་ཎིཙ་ལའང་ངོས། ཏང་ལ། ཙོ་རི་ཡི་ཥྛ་ཏེ། སོགས་སོ། ?། དེ་བཞིན་དུ། ལླིང་ལ། ཨ་ཙོ་ར་ཡི་ཥྱ་ཏི། སོགས་དང་། ཨ་ཙོ་ར་ཡི་ཥྱ་ཏི། སོགས་སོ། །།འགྲེལ་པར་་་་་ལུང་དྲངས་པ། ་ཙུ་རཱ་དེཎིཙ་བཱ་ཀེ་ཙིཏ། ཙུ་ར་ལ་སོགས་པ་རྣམས་ལས་ཎིཙ་འགའ་ཞིག་ཏུ་ཡང་ནའོ། །ཙུ་ར་ལ་སོགས་པ་རྣམས་ལས་སྐབས་འགའ་ཞིག་ཏུ་་་ཎིཙ་ཡང་ན་བར་བྱེད་དེ། བྱེད་པ་པོ་ལ་ཤཔ་སྦྱར་བས། ལཊ་ལ། ཙོ་ར་ཏི། སོགས་དང་། ལིང་ལ། ཙོ་རེཏ། སོགས་སོ། །ལོཊ་ལ། །འོ་ཙོ་ར་ཏུ། །སོགས་དང་། ལང་ལ། ཨ་ཙོ་རཏ། སོགས་སོ། ༈། ལུང་ལ། ཨིཊ་བྱིན་་་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཌ་ཨ་ཙོ་རཱིཏ། ཨ་ཙོ་རིཏྟཾ། ཨ་ཙོ་རི་ཥཱ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཙོ་རཱི། ཨ་ཙོ་རིཏྟཾ། ཨ་ཙོ་རི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཙོ་རི་ཎཾ། ཨ་ཙོ་རི་། ཨ་ཙོ་རི། ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙུ་ཙོ་ར། ཙུ་ཙུ་ར་ཏུ། ཙུ་ཙུ་རུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙུ་ཙོ་རི་ཐ། ཙུ་ཙུ་ར་ཐུ། ཙུ་ཙུ་ར། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙུ་ཙོ་ར། ཙུ་ཙུ་རི་བ། ཙུ་ཙུ་རི་མ། ལུཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཙོ་རི་ཏཱ། ཙ་རི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ་ཨུ་པ་དྷཱའི་འང་ངོས་།
