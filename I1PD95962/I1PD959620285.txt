ཚེས་འི་་་་་་དྷ་དུ་ཧ་ཨཱིར་ཨ་སཱུ་རྣམས་གང་ལ་སོ་སོར་ཁྲོ་བའི་དོན་ལའོ། །ཀྲུ་དྷ་སོགས་རྣམས་ཁྲོ་བའི་དོན་བྱ་བ་མ་ཡིན་པ་སོ་སོར་བརྗོད་པའི་སྦྱོར་བ་ལ་བཞི་པ་འགྱུར་ཏེ། ཧ་ར་ཡེ་ཀྲུ་རྒྱ་ཏི། འཕྲོག་བྱེད་ལ་ཁྲོ་བར་བྱེད། ཧ་ར་ཡེ་དྲུ་རྱ་ཏི། འཕྲོག་བྱེད་ལ་འཁུ་བར་བྱེད། ཧ་ར་ཡེ་ཨཱིརྱ་ཏི། འཕྲོག་བྱེད་ལ་ཕྲག་དོག་བྱེད། ཧ་ར་ཡེ་ཨ་པཱུ་ཡ་ཏི། འཕྲོག་བྱེད་ལ་ཕྲ་མ་བྱེད། ཀྲུ་དྷ་དྲུ་ཧོ་རུ་པ་སྲྀཏྠ་ཡོ། །གྲུ་དྷ་དྲུ་ཧ་དག་གི་ཉེ་བར་འོས་པ་དག་ལའོ། །ཀྲུ་དྷ་དང་དྲུཧ་དག་གི་སྦྱོར་བ་མི་འོས་པ་མ་ཡིན་པ་འོས་པའི་དོན་གྱི་སྦྱོར་བ་སྟེ་ཁྲོ་བའི་དོན་་མ་ཡིན་ན་བཞི་པ་འགྱུར་བ་མ་ཡིན་པས་གཉིས་པ་ཉིད་འགྱུར་ཏེ། །བྲཱཧྥ་ཎ་མ་་བྷི་ཀྲུ་དྷུ་ཏི། བྲམ་ཟེ་ལ་ཁྲོ་བར་མི་བྱེད། བཱཧྥ་ཎ་མ་བྷི་དྲུ་རྒྱ་ཏི། བྲམ་ཟེ་ལ་འཁུ་བར་མི་བྱེད། རཱ་་་དྷཱི་ཀློརླུ་སྱ་བི་པྲཉྫཱ། རཱ་དྷ་ཨཱིཀ་དག་ལས་གང་གི་འདྲི་བ་ལའོ། །རཱ་དྷ་དང་་་ཨཱིཀྐ་དག་འདྲེ་བའི་དོན་ལ་སྦྱོར་བ་ལ་བཞི་པ་འགྱུར་ཏེ། ཀྲྀཥཱ་ཡ་རཱ་དྷ་ཏེ། ཀྲྀཥཱ་ཡེ་ཀྵ་ཏེ། ནག་པོ་ལ་འདྲི་བར་བྱེད། པྲ་ཏྱངྒྷཾ་ཐུ་བ་པཱུརྦ་སྱ་ཀཏཱ། པྲ་ཏི་ཨཱང་དག་་་་ལས་ཐུའི་སྔ་མའི་བྱེད་པ་པོ་ལའོ། །ཉེར་བསྒྱུར་པྲ་ཏི་དང་ཨཱང་ཡོད་པའི་ཤུའི་སྔ་མའི་རྟགས་དེ་ལ་བྱེད་པ་པོའི་དོན་ལ་བཞི་པ་འགྱུར་ཏེ། བི་པཱ་ཡ་ཧཱུྃ་པྲ་ཏི་ཤི་ཎོ་ཏི༑བྲམ་ཟེའི་བ་ལང་སོ་སོར་ཐོས། བི་པྲ་ཡ་གཱ་མཱ་ཤི་ཌོ་ཏི། བྲམ་ཟེ་བ་ལང་ཀུན་དུ་ཐོབ། ཨ་ནུ་པྲ་ཏི་སྲྀ་ཎ། ཨ་ནུ་པྲ་ཏི་དག་གི་གྲི་ཎ་ལའང་ངོ༌། །ཨ་ནུ་དང་པྲ་ཏི་སྔ་མའི་གྲོ་ཎ་དང་སྦྱོར་བའི་རྟགས་སྔ་མ་དེ་ལའང་བྱེད་པ་པོའི་དོན་ལ་བཞི་པ་འགྱུར་ཏེ། ཧོ་ཏྲེ་ཨ་ནུ་:ཤྲཱི་ཎཱ་ཙིཏི། སྦྱིན་སྲེག་མཁན་ལ་སོ་སོར་རྗོད་པར་བྱེད། ཧོ་ཏྲེ་པྲ་ཏི་ཤྲཱི་ཎཱ་ཏི། སྦྱིན་སྲེག་མཁན་ལ་རྗེས་སུ་རྗོད་པར་བྱེད། པ་རི་ཀྲ་ཡ་ཎེ་སཾ་པྲ་དཱ་ན་མ་ནྱ་ཏ་ར་ཧཱུཾ། པ་རི་ཀྲི་ཡའི་་་ཡང་དག་པར་རབ་ཏུ་སྦྱིན་པ་ཡང་ནའོ། ་། :པ་རི་ཀྲི་་ཡའི་སྦྱོར་བ་ལས་བྱེད་པ་པོའི་དོན་ཡང་དག་པར་རབ་ཏུ་སྦྱིན་པའི་དོན་ཅན་ལ་བཞི་པ་ཡང་ན་འགྱུར་ལ་་ཡང་ན་གསུམ་པ་ཉིད་དེ། ཤ་ཏཱ་ཡ་པ་རི་ཀཱི་ཏོན་ནུ་བུ་ཧི། བརྒྱ་ཡིས་ཡོངས་སུ་ཉོས་པ་།
