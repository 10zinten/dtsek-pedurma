སྒྲ་མདོ། ཏི་ཁ་ཚོའི་དེ་ཉིད་ཚེ་་བློ་ཡཾ་ཙ་ཀཱ་ར། སོགས་དང་། ་ལུཊ་ལ་ཨིཊ་བྱིན་པས། ཙ་ཀཱ་ས་ཡི་ཏཱ། ཙ་ཀཱ་ས་ཡི་ཏཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས༑ཙ་ཀཱ་སཱུཏ། ཙ་ཀཱ་སཱཙཱ། ཙ་ཀཱ་སྟ་སུ། སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་པས། ཙ་ཀཱ་ས་ཡི་ཥྛ་ཏི། །སོགས་དང་། དེ་བཞིན་དུ། ལྡིང་ལ། ཨ་ཙ་ཀཱ་ས་ཡི་ཏ། །སོགས་སོ། །་་་ལུང། ཁ་ལ་ཅི་ཤི་སོགཏི་ཧ་ཊི་༈ཨུནྡི་གཤེར་བ་ལའོ་ཞེས་པ་ལ། རྒྱུ་དང་བཅས་པ་སོགས་ཀྱིས་ཎི་ཙ་སྦྱར་བས་ཨུཎྜི་ཞེས་གྲུབ་པ་ལ། ལཊ་སོགས་སྦྱར་བས། ལཊཨུནྡ་ཡ་ཏི། ལིང་ཨུནྡ་ཡེཏ། ་ལོཊཨུནྡ་ཡ་ཏུ། ལང་། ཨོརྡ་ཡཏ། །སོགས་དང་། ལུང་ལ༑ཎི་ཤི་སོགས་ཀྱིས་ཙི་ཙང་། ཙང་ལ་འོས་ད་ཎི་དང་བཅས་པ་གཉིས་བརྗོད། ཨ་ཙ་དང་པོའི་ཨཱཊ། །ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ་བྱས། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས། ཨོཎྜི་དཏ། ཨོཎྜི་ད་ཏྭཾ། ཨོཟིགས་དན། སོགས་སོ། །ལིཊ་ལ་ཨཱམ་སྦྱར་བས། ཨུནྡ་ཡཾ་ཙ་ཀཱ་ར། །སོགས་སོ། །ལུཊ་ལ་ཨིཊ་བྱིན་པས། །ཨུ་ཡི་ཏཱ། ཨུནྡ་ཡི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས། ཨུདྡྷཱཏ། ཨུནྡྷ་ཧཱུཾ། ཨུནྡྲ་སུ། སོགས་སོ། །ཁྲིཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཨུནྡ་ཡི་བྱེ་ཏི། སོགས་དང་། །དེ་བཞིན་དུ། ཁྲིང་ལ། །ཨོནྡ་ཡི་བྱེཏ། སོགས་སོ། །པ་གའི། །ད་བ་སྐྱེལ་བ་བ༈ཨུ༴་ལུགས་ལའོ་ཞེས་པ་ལ། རྒྱུ་དང་བཅས་པ་སོགས་ཀྱིས་ཎི་ཙ་སྦྱར་བས༑ཨུཎྜི་ཞེས་གྲུབ་པ་ལ། ལཊ་སོགས་སྦྱར་བས། ལོཨུཥྚ་ཡ་ཏི། ཨུ་ྗ་་་ཡཏ༑ཨུ་ྗ་ཡ་ཏུ། ཨོ༴་ཡཏ། སོགས་སོ། །ལུང་ལ། ཎི་ཤི་སོགས་ཀྱིས་ཙྪི་ཙང་། ཙང་ལ་འོས་ཛ་ཎི་དང་བཅས་པ་གཉིས་བརྗོད། ཎིའི་ཨིཊ་མེད་པ་་ལ་འོས་ཎི་ཕྱི་མ་ཕྱིས་པས། ་ཨོཌི་ཛཏ། །ཨོཌི་ཛ་ཊཱ། ཨོཌི་ཛན། སོགས་སོ། །རྒྱུ—།
