སྒྲ་མདོ། ཏི་ཨ་ཏིའི་སྟོན་་་་་་་པའི་ཚོ་པ༌ཨ་ཏང་ལ། བ་རི་ཏཱ། བ་རི་ཏཱ་སི། སོགས་སམ། ་བ་རཱི་ཏཱ། བ་རཱི་ཏཱ་སི། སོགས་སོ། །ཏང་ལ། བ་རི་ཏཱ། བ་རི་ཏཱ་སེ། སོགས་སམ། བ་རཱི་ཏཱ། བ་རཱི་ཏཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ཤ་ཡཀ། །ལིང་རྣམས་ལ་རིང་འོས། ་རིང་དུ་བསྒྱུར་བས། བྲེ་ཡཱཏ། :ཤྲཱི་ཡཱནྟཾ། ཤྲཱི་ཡཱ་སུ༑སོགས་སོ། །ཏང་ལ། ལིང་སིཙ་དག་གི་ཨཏྨ་ནེ་པ་ད་རྣམས་ལ་འོས། ཨིཊ་ཡང་ན་བར་བྱས་པས། བ་རི་ཊཱིཋ། །པ་རི་ཏི་ཡཱཙཱསི། བ་རི་ཁཱི་རན། སོགས་སམ། །བྲྀ་ཏིཥྛ། དྲྀ་ཏཱི་ཡཱནྡྷཾ། བྲྀ་ཏཱི་རན། སོགས་སོ། །ལྀཊ་ལ། བྲི་རཱིཏ་ལས་ཡང་ན་འོས། ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ་ཨ་ཏང་ལ། བ་རི་ཥྱ་ཏི། སོགས་སམ། བ་རཱི་ཥྛ་ཏེ། སོགས་སོ། །ཏང་ལ། །ཌབ་རི་ཥྱ་ཏེ༑སོགས་སམ། །པ་རཱི་ཥྱ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་གི་ཨ་་་ཏང་ལ། ཨ་བ་རི་ཏི། སོགས་སམ། ་ཨ་བ་རཱི་ཏི། སོགས་སོ། །ཏང་ལ། ཨ་བ་རི་ཏ། །སོགས་སམ། ཨ་བ་རཱི་ཏ། །སོགས་སོ། །༈དྷུཉ་འགུལ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་སྦྱོར་ཏེ། ལཊའི་ཨ་ཏང་ལ། དྷུ་ནོ་ཏི། སོགས་དང༌། ཏང་ལ། དྷུ་ནུ་ཏེ། ་སོགས་སོ༑། ལིང་གི་ཨ་ཏང་ལ། དྷུ་ནུ་ཡཱཏ། སོགས་དང་། ཏང་ལ། དྷུ་ནཱི་ཏ། སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། དྷུ་ནོ་ཏུ། སོགས་དང༌། ཏང་ལ། དྷུ་ནུ་ཏྭཾ༑སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ཨ་དྷུ་ནོཏ། །སོགས་དང༌། །ཏང་ལ༑ཨ་དྷུ་ནུ་ཏ། སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ། ་，་སུ་དྷུཉ་རྣམས་ལས་པ་རསྨི་པ་ད་རྣམས་ལ་འོས། ཨིཊ་བྱིན། སིཙ་ལ་འཕེལ་བ་སོགས་ཀྱིས་འཕེལ་བ་བྱས་པས། ཨ་དྷཱ་བཱིཏ། ཨ་དྷཱ་བིཏྟཾ། ཨ་དྷཱ་བི་ཥཱ། སོགས་དང༌། ཏང་ལ། ་་སཱ་ར་ཏི་སཱུ་ཏི་སཱུ་ཡ་ཏི་དྷུཉ་ཨུཏ་འགྲོ་བ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་རར་_—།
