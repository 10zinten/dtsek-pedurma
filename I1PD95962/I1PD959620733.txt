འ་བ། ཁོག་ནད་ཀྱི་འབད་ཁཎོ་བསྟན་འགྱུར། སྒྲ་མདོ། གཏི་ཏ་ཏི་ཎི་ནི། ལོ་ཆང་ཁོ་ཚོས་། དགའ་རི་ཀཔུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ད་དང་པོའི་གྷ་འོས་ཧ་གྷ། ཛྷ་ཥ་ལ་དྷཱ་མ་ཡིན་པའི་ཏ་ཐ་དག་གི་དྷ་འོས། ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་གྷ་གར་བསྒྱུར་བས། དོཥཱི། གཉིས་ཚིག་དང་མང་ཚིག་ལ། དུསཿ། དུ་ཧནྟི། སྐྱེས་བུ་དབུས་་མའི་གཅིག་ཚིག་ལ། ལ་སིཔ། །ད་དང་པོའི་གྷ་འོས་ཧ་གྷ། ཨཙ་གཅིག་་་སོགས་ཀྱིས་ད་དྷ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཁར་ལ་ཙར་འོས། གྷ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། །།གཉིས་ཚིག་དང་མང་ཚིག་ལ། དུམྦྷ། ཐས་དུགྷ། ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །དོ་རྨི། དུ་ཧྥ། དུ་། སྒྲུབ་ཚུལ་དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དུབྷེ། དུ་ཧཱ་ཏེ། དུ་ཧ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྷ་ཀྵེ། དུ་ཧཱ་ཐེ། ་དྷབྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་ཧེ། དུ་ཧྥ་ཧེ། །དུདྨ་ཧེ། ལིང་གི་ཨ་ཏང་ལ་ཧ་འདྲེན་པ་ཁོ་ན་སྟེ། དུ་ཧཱཏ། དུ་ཧྭ་ཏྭཾ། དུ་ཧུ། །སོགས་དང༌། ཏང་ལའང་ཨཏ་མཐའ་མ་ཡིན་པ་སོགས་་་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས་ནས་ཧ་དྲངས་པས། དུ་ཧཱི་ཏཾ། དུ་ཧཱི་ཡཱ་ཏཱ། དུ་ཧཱི་རན༑སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། ཨ་ཏང་གི་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དོགླུ། དུདྡྷཱཏ། དུདྡྷཾ། དུ་ཧནྟུ། །སྐྱེས་བུ་་་དབུས་མའི་གཅིག་ཚིག་ལ། ་ལ་སིཔ། སིའི་ཧི། ཧུ་ཛྷལ་ལས་ཧི་དྷི། ད་དང་པོའི་ཧ་གྷ། ཛྷལ་སོགས། གྷ་ག་དྲངས་པས། དུབྷི། ཡང་ན་ཧི་ཏཱ་་་ཏང་དུ་བསྒྱུར་བས། དུབྷཱཏ། །གཉིས་ཚིག་དང་མང་ཚིག་ལ། །དུགྷཾ། དུབྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དོ་ཧཱ་ནི། དོ་ཧཱ་བ། དོ་ཧཱ་མ། དེ་བཞིན་དུ། ཏང་ལ། དུདྡྷཾ། དུ་ཧཱ་ཏྭཾ། དུ་ཧ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྷོ། དུ་ཧཱ་བྷཾ། བྷུདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དོ་ཧི། དོ་ཧཱ་བ་ཧི། དོ་ཧཱ་མ་བྷཱ།
