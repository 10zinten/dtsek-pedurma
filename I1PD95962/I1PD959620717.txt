སྒྲ་མདོ། ཏི་ཧོཎི་ཞི་ཎ་འོ་ཎའི་བཀཚོ་ཁ་ཎར་མོ་བློ་ལ་ཙ་ཁྱི་མ་ཧེ། ཡང་ཙཀྐིང་ཀཱཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་ཅིང་ཨ་ཡིག་མཐའ་ཅན་ཡིན་པས་ཆོ་ག་རྣམས་ཁྱུཉ་དང་འདྲ་སྟེ། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙཀོ། ཙཀྐ་ཏུ། ཙཀླུ། སྐྱེས་བུ་་་དབུས་མ་གསུམ་ལ། ཙཀི་ཐ། ཙཀཱ་ཐ། ཙཀྐ་ཐུ། ཙཀྐ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙཀོ། ཙཀྐི་བ། ཙཀྐི་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་་་དང་པོ་གསུམ་ལ། ཙཀེ། ཙཀཱ་ཏེ། ཙཀྐི་རེ། པ་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙཀྐི་ཥ། ཙཀ་ཐེ། ཙཀི་དྷེ། ཙཀྐི་ཝེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙཀེ། ཙཀི་པ་ཧེ། ཙཀྐི་མ་ཧེ། ལུཊ་ལ་ཥཱཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། ཁཱ་ཏཱ། ཁཱ་ཏཱ་སི། སོགས་དང་། ཏང་ལ། བྱཱ་ཏཱ། ཁཱ་ཏཱ་སེ། སོགས་སོ། །ཀཱཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། ་ཀཱ་ཏཱ། ཀཱ་ཏཱ་སི། སོགས་དང་། ཏང་ལ། ཀཱ་ཏཱ། ཀཱ་ཏཱ་སེ། །སོགས་སོ། །ཡང་བོར་བའི་དོན་ལ་སྦྱོར་བའི་་དབང་དུ་བྱས་ན། ཙགླིང་ཉིད་དུ་གནས་པ་ལ་ལུཊ་ཏང་རྣམས་སྦྱར། བལ་དང་པོའི་ཨིཊ་བྱིན་པས། ཙཀྵི་ཏཱ། ཙཀྱི་ཏཱ་རོ། ?། ཙཀྐི་ཏཱ་ར། ཙཀྱི་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ་ཁྱཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ་གཞན་ཡང་བརྩེགས་པ་དང་པོ་སོགས་ཀྱིས་ཨེར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱ་སྟེ། ཥཱ་ཡཱཏ། ཥཱ་ཡཱནྟཾ། ཁཱ་ཡཱ་སུ། སོགས་སམ། ཁྱེ་ཡཱཏ། ཁྱེ་ཡཱཙཱ། ཁྱེ་ཡཱ་སུ། །སོགས་སོ། །ཏང་ལ་ཨིར་མི་བསྒྱུར་བས། བྱཱ་སཱིཥྛ།  ོ་ཥཱ་སཱི་ཡཱནྟཾ། ཥཱ་སཱི་རན། སོགས་སོ། །དེ་བཞིན་དུ། ཀཱཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ལའང་ཨ་ཏང་ལ་ཨཱ་ཉིད་དང་ཨེ་ཡང་ན་བར་བྱས་པས། ཀཱ་ཡཱཏ། ཀཱ་ཡཱནྟཾ། ་ཀཱ་ཡཱ་་སུ༑སོགས་སམ། ཀེ་ཡཱཏ། ཀེ་ཡཱནྟཾ། ཀེ་ཡཱ་སུ། །སོགས་སོ། །ཏང་ལ་ཨེར་མི་བསྒྱུར་བས། ཀླུ་སཱིཊྚ། ཀཱ་སཱི་ཡཱནྟཾ། ཀཱ་སཱི་རན། ་སོགས་སོ། །།
