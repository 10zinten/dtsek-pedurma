མ་འ:དེ་ལྟར་ན་ཤེས་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི༽ཀྱི་དེ་ཚོ་འཚོ་གནས་ས་ཁ་ན་ཟླ་བློ་ཙུ་འོས། ཁ་ཚ། ཨ་བྷུ་སའི་ཙར་འང་ངོས། ཚ་ཙ། རྗེས་སུ་སྣ་ལྡན་མཐའི་སོགས་ཀྱིས་ནུཀ་ཨཱ་ག་མ་བྱིན། །ཡང་གི་ཨཙ་ལ་འང་ངོས་ཡ་ཕྱིས་པས། །ཙི་ཁན། །ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར། །སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ།  ོཡང་ལས་ཡང་ན་འོས། ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ཙ་ཁ་ནཱི་ཏི། ཙཾ་ཁནྟི། གཉིས་ཚིག་ལ༑ཛན་ཁན། སན་སོགས་ཀྱིས་ཊི་ཨཱཏ་དུ་བསྒྱུར་བས། ཁཙོ་ཁཱ་ཏ། མང་ཚིག་ལ། ཨ་བྱཱསྟ་ལས་ཨཊ་འོས། ཛྷ་ཨཏི་དུ་བསྒྱུར། གམ་ཧན་ཁན་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཡང་ན་ཕྱིས་པས། ཙི་ཁ་ན་ཏི། ཙི་སྣ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། »ཙོ་ཁ་ནཱི་ཎི། ཙི་ཁརཱི། ཙཾ་ཁཱ་ཐ། ཙཾ་ཁཱ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙཾ་ཁ་ནི་མི། ཙི་ཁརྨི། ཙཾ་ཁ་ནཱ། ཙི་ཁམཿ། ལིང་ལ། ཙི་ཁ་དྷཱཏེ། །ཆ་ཙི་ཁ་རྩྭ་ཏྭཾ། ཙཾ་ཁ་ནྱ། ་སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། ཙཾ་ཁ་ནི་ཏུ༑ཙི་ཁནྟུ། ཙཾ་ཁཱ་ཏཱཏི། ཙཾ་ཁཱ་ཏྭཾ། ཙི་ཁ་ན་ཏུ། ཙི་སྣ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙཾ་ཁཱ་ཧི། ཙཾ་ཁཱ་ཏཱཏ། ཙཾ་ཁཱ་ཏཾ། །ཙཾ་ཁཱ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ནཙཾ་ཁ་ནཱ་ནི། ཙི་ཁ་ནཱ་བ། ཞཙོ་ཁ་ནཱ་མ། དེ་བཞིན་དུ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཙོ་ཁ་ནཱིཏ། ཨ་ཙོ་ཁན། ཨ་ཙི་ཁཱ་ཏྭཾ། ཨ་ཙི་ཁ་ནུ། ཨ་ཙི་ཟླུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཙི་ཁ་ནཱི། ཨ་ཙོ་ཁན། ཨ་ཙི་ཁཱ་ཏཾ། ཨ་ཙི་ཁཱ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཙི་ཁ་ནཾ། ཨ་ཙི་ཁ་ནཱ། ཨ་ཙ་ཁ་རྨ། ལ་ཡིག་ལྷག་མ་རྣམས་ལ་རིམ་པ་བཞིན༑ལུང་ཨ་ཙི་ཁ་ནཱིཏ། ཏེལིཊ་ཙི་ཁ་ནཾ་ཝཾ་ཀཱ་ར། ལུཊཙོ་ཁ་ནིཏཱ། སྨོན་ལིང་ཙོ་ཁ་སཱུཏི། ཁྲིཊཙཾ་ཁ་ནི་ཥྛ་ཏི། ཁྲིང་ཨ་ཙི་:ཁ་ནི་ཏ། སོགས་སོ། །༈དི་བུ་རྩེ་བ་ལའོ་ཞེས་པ་ལ། ་ཡང་བྱིན། གཉིས་བརྗོད། །ཡང་—།
