་ཁ་ལུ་པོ་ལྟ་བ་་་་་་་བསྟན་འགྱུར། སྒྲ་མདོ། བཏི། *pཁ་ཙིནྟི་ནི་ལྟ་ཁོ་ཎ་ཁཙོ་ཎཥ་ཎི་པ་བློ་གསལ་། ༈ན་དྷི་ལ་ཡཀ་སྦྱར་ཨིཏ་འགྲོ་བ་སོགས་ཀྱིས་ནུམ་བྱིན་པས། ན་ཏེ། ནནྡྷེ་ཏ། ་ནནྡྷཱ་ཏཱ། ཨ་ནནྡྷཱ་ཏ། །སོགས་སོ། ། ོ། ལུང་ལ། །ཨ་ནནྟི། ཨ་ནནྟི་ཥཱ་ཏཱ། ཨ་ནནྟི་ཥ་ཏ། སོགས་སོ། ་། ལིཊ་ལ། ན་ནནྡྷེ། ན་ནནྡྷ་ཏེ། སོགས་སོ། །།ལྷག་མ་རྣམས་ལ། །རང་ལུཊ་ནནྟི་ཏཱ། །ོསྨོན་ལིང་ནནྟི་ཊཱི། ཁྲིཊ་ནནྟི་ཉྩ་ཏེ། ཁྲིང་…ཨ་ནནྟི་་ཏ། སོགས་སོ། སྐྱོ་༌་། :::ཁ་ཀེ། ང་ཞུར། ཊ་༈ཡ་ཛ་ལ་ཡཀ་སྦྱར། བ་ཙི་སཱ་པི་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་་་་བས༑་ལཊཨི་ཛུ་ཏེ། ་་ལིང་…ཨི་ཛེ་ཏ། ཡོ། ལོཊ་ཨི་ཛུ་ཏྭཾ། ལང་ཨི་ཙུ་ཏ། སོགས་སོ། །ལུང་ལ། གཅིག་ཚིག་ལ་ཙི་ཙི་ཎར་བསྒྱུར་བས། ཨ་ཡཱ་ཛི། གཉིས་ཚིག་སོགས་ལ། ཛྷལ་དང་པོ་སོགས་ཀྱིས་སིཙ་དབྱི་བ་དང་། བྲཤྩ་བྦསྔ་སོགས་ཀྱིས་ཛ་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། །ས་ཨ་ཡ་ཀཾ་ཏཱཾ། ཨ་ཡ་ཀྵ་ཏེ། སོགས་སོ། །ལིཊ་ལ་གཉིས་རྣམས་ཀྱི་སོགས་ཀྱིས་ཨ་བྷཱ་ས་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར། བ་ཙི་སྭཱ་པི་སོགས་ཀྱིས་ཕྱི་མ་ཡང་སཾ་པྲ་སཱ་ར་ཎ་བྱས་པས། ཨཱི་ཛེ། ཨཱི་ཛཱ་ཏེ། སོགས་སོ། །ལུཊ་ལ། བྲཤྩ་བྷསྔ་སོགས་ཀྱིས་ཛ་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཡཏྟཱ། ཡཏྟཱ་རོ། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཛ་ཥ༑ས་ལ་ཥ་ཌྷ་དག་གིས་ཀ་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཡཀྴིཋ༑ཏོཡཀྴི་ཡཱཙཱ། སོགས་སོ། །།དེ་བཞིན་དུ། ཊ་དང་ཁྲིང་ལ། །ཡཀྵ་ཏེ། ཨ་ཡགྱིུཏ། སོགས་སོ། །།།།ཁྲི་བ་ཁའི་ཁཙིཏའི་༈ཤིང་ལ་ཡཀ་་། ཀ་འགྲོ་བ་ལ་ཨ་ཡང་ངོས་ཨ་ཡང་དུ་བསྒྱུར་བས། ལཊཤ་ཡྻ་ཏེ། ལིང་ཤ་སྤེ་ཏ། ལོཊཤ་ཡྻ་ཏྭཾ། ལང་ཨ་ཤ་ཡྻ་ཏ། །སོགས་སོ། །ལུང་གི་གཅིག་ཚིག་ལ། །ཙྪི་ཙི་ཎར་བསྒྱུར་བས། །ལ་ཨ་ཤཱ་ཡི། གཉིས་ཚིག་སོགས་ལ། ཨིཊ་བྱིན། ཙིཎ་བཞིན་ཡང་ན་བར་བྱས་པས། ཨ་ཤཱ་ཡི་ཥཱ'ཏྭཾ། །
