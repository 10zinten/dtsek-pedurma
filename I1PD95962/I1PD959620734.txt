ཧི༑ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ལ་ཏིཔ། ཨིཏ་་་ཀྱིས་འང་ངོས་ཨི་ཕྱིས། ད་དང་པོའི་ཧ་གྷ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཧལ་དཱིཔ་སོགས་ཀྱིས་ཏ་ཕྱིས། གནས་མེད་པ་ལ་ཡང་ན་འོས། གྲའམ་ཀྲར་་་བསྒྱུར་བས། ཨ་དོགཨ་དོཀ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། །ཨ་དུདྡྷཾ། ཨ་དུ་ཧན། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷོག། སའི་རྒྱུ་མཚན་ལས་ད་དྷ། ཨ་དྷོཀ་། ཨ་དུདྡྷཾ། ཨ་དུགྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དོ་ཧཾ། ཨ་དུ་ཧཱ། ཨ་དུ་ཧྥ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དུགྷ། །ཨ་དུ་ཧཱ་ཏཱ༑ཨ་དུ་ཧ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དུདྡྷེ། ཨ་དུ་ཧཱཐཱ། ཨ་དྷུདྡྷཾ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དུ་ཧི། ཨ་དུ་ཧཱ་ཧི། ཨ་དུ་ཧྥ་ཧི། ལུང་ལ། ཤལ་ཨིཀ་ཨུ་པ་དྷཱ་ཨིཊ་མེད་པ་སོགས་ཀྱིས་ཙྪིཀྐར་བསྒྱུར། ཀ་ཕྱིས། ད་དང་པོའི་ཧ་གྷ། ཨ་ཙ་གཅིག་པ་སོགས་ཀྱིས་ད་དྷ། ཁར་ལ་ཙར་འོས་གྷ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། ་ཨ་ཏང་ལ། ཨ་དྷུ་ག་ཏེ། ཨ་དྷུ་གྷ་ཏྭཾ། ་ཨ་དྷུ་ཀུན། སོགས་དང༌། ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། །དུ་ཧ། དི་ཧ། ལི་ཧ་སོགས་ཀྱིས་ཀླུ་དབྱི་བ་དང་མི་་་དབྱི་བ་ཡང་ན་བར་བྱས་པས། ཨ་དྷ་ཀླུ་ཏ། ཨ་དུགྷ། གཉིས་ཚིག་ལ། ཀའི་ཨཙ་ལ་འོས་སའི་ཨ་ཕྱིས་པས། ཨ་དྷུ་ཀ་ཏྭཾ། མང་ཚིག་ལ་ཨ་དབྱི་བ་་་ཡང་ན་བར་བྱས་པས། ཨ་དྷ་ཀཱནྟ།  ོཨ་དྷུ་ཀྵ་ཏེ། དེ་བཞིན་དུ། སྐྱེས་བུ་་དབུས་མའི་གཅིག་ཚིག་ལ། །ཨ་དྷུཀྟ་ཐཱ། ཨ་དྷཏྟཱ། གཉིས་ཚིག་ལ། ཨ་དྷ་གྷ་བྷཾ། མང་ཚིག་ལ། ཨ་དྷཀྵ་ཧཱུཾ། ཨ་དྷུདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དྷུཀྟི། །་ཨ་དྷུཀྟ་བ་ཧི། ཨ་དྷཀྵ་མ་ཧི། ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་བྱས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དུ་དོ་ཧ། དུ་དུ་ཧ་ཏུ། དུ་དུ་—།
