སོགས་སམ། ཨ་ཤ་ཡི་ཥ་ཏ། ་སོགས་སོ། །ལིཊ་ལ། །ཤི་བྱེ། ཤི་ཤུ་ཏེ། སོགས་སོ། །ལུཊ་ལ། 《ཨིཊ་བྱིན་ཙིཎ་བཞིན་ཡང་ན་བར་བྱས་པས། ཤཱ་ཡི་ཏཱ༑་སོགས་སམ། ཤ་ཡི་ཏཱ། སོགས་སོ། །དེ་བཞིན་དུ། །ལྷག་མ་རྣམས་ལ༑། ཤཱ་ཡི་ཏཱིཤྩ། སྨོན་པའི་ལིང་ཤ་ཡི་ཥིཉྫ། ཝིཊཤཱ་ཡི་བྱ་ཏེ། ཁྲིཊཤ་ཡི་་བྱ་ཏེ། ཁྲིངཨ་ཤཱ་ཡི་ཥྛ་ཏ། ཁྲིང་.…ཨ་ཤ་ཡི་ཥྛ་ཏ། སོགས་སོ། །ཞི་བ་༈ཏ་ནུ་ལ་ཡཀ་སྦྱར། ཏ་ནོ་ཏེརྨ་ཀི། །ཏ་ནོ་ཏིའི་ཡཀ་ལའོ། །།ཡཀ་ཕ་རོལ་ལ་ཏ་ནོ་ཏིའི་མཐའ་ཨཱཏ་དུ་ཡང་ན་བསྒྱུར་ཏེ། །ལཊ་ཏཱ་ཡ་ཏེ། ཌ་ལཊ་ཡཏ་ནྱ་ཏེ། ལིང༌ཏཱ་ཡེ་ཏ། ་ལིང་ཏ་རྩེ་ཏོ། །ལོཊཏཱ་ཡ་ཏྭཾ། ལོཊ་ཡཏཱ་རྱ་ཏྭཾ། ལང༌…ཨ་ཏཱ་ཡ་ཏི། ལང་ཨ་ཏ་ནྱ་ཏི། སོགས་སོ། །ལུང་ལ། ཨ་ཏཱ་ནི། ཨ་ཏ་ནི་ཥཱ་ཏཱ། སོགས་སོ། ་། ལིཊ་ལ། ཨ་བྷཱ་ས་དབྱི་བ་སོགས་ཀྱིས། ཏེ་ནེ། ཏེ་ནཱ་ཏེ། སོགས་སོ། །ལྷག་མ་རྣམས་ལ། །ལུཊ་ཏ་ནི་ཏཱ། སྨོན་ལིང་ཏ་ནི་་ཊྛིཏྟ། ཁྲིཊ་ཡཏ་ནི་ཉྩ་ཏེ། ཁྲོང་…ཨེ་ཏ་ནི་ཥྛ་ཏི། སོགས་སོ། ་། ་་་་་བ། དང་༈ཛ་ན་ལ་ཡཀ་། ཡེ་ལ་འདམ་ཀ་འོས། མཐའ་ཨཱཏ་དུ་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་འདམ་ཀ་ཅན་དུ་བྱས་པས། །ལཊཛཱ་ཡ་ཏེ། ལཊ་ཛ་ནེ་ཏེ། ལིཊཛཱ་ཡེ་ཏ། །ལིཊཛ་རྩེ་ཏ། ལོཊཛཱ་ཡ་ཏྭཾ། ་ལོཊཛ་ནྱ་ཏྭཾ། །ལང་ཨ་ཛཱ་ཡ་ཏི། ལང་…ཨ་ཛ་ནྱ་ཏ། སོགས་སོ། །ལུང་ལ་འཕེལ་བ་མི་བྱེད་པས། ་ཨ་ཛ་ནི། ཨ་ཛ་ནི་ཥཏྭཾ། ཨ་ཛ་ནི་ཥ་ཏ། །སོགས་སོ། །ལེཊ་ལ་གཉིས་བརྗོད༑ཌགམ་ཧན་ཛན་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཕྱིས། སྐུའི་ཤ་ཙུ་འོས་ན་ཉར་བསྒྱུར་བས། ཛ་ཛྷེ། ཛ་ཛྷཱ་ཏེ། ཛ་ཛཱི་རེ། སོགས་སོ། །ལྷག་མ་རྣམས་ལ་ཨིཊ་བྱིན་པས། ལུཊཛ་ནི་ཏཱི། སྨོན་ལིང་ཛ་ནི་ཏིཥྛ། ཁྲིཊ་…ཛ་ནི་བྱ་ཏེ། ཕྲེང་…ཨ་ཛ་ནི་ཥྛ་ཏ། ་སོགས་སོ། །དེའི་མ་མེ། པོད་ཁོ་པ་ལ་སྤྲོ་བོ་གང་ལང་བ་།
