པོ་རྣམས་ལ། །ཛྷལ་ལས་ཛྷལ་ལ་འོས། སིཙ་དབྱི་བ་དང༌། ཙུའི་ཀུ། ཁར་ལ་ཙར་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །་ཨ་ནིཀྵིཏ། །ཨ་ནིཏྟཾ། ཨ་ནི་ཀླུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ནིཊཱི། ཨ་ནིཏྟཾ། ཨ་ནིཏྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ནིཏྟཾ། ཨ་ནིབ། ཨ་ནིཀྵ། ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ནི་ནི་ཛ། ནི་ནེ་ཛ་ཏུ། ནི་ནི་ཛུ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནི་ནེཀྑ། ནི་ནེ་ཛི་ཐ། ནི་ནི་ཛུ་ཐུ། ནི་ནི་ཛ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནི་ནེ་ཛ། ནི་ནི་ཛི་བ། ནི་ནི་ཛི་མ། ལུཊ་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ནེཀྐཱ། ནེཀ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ནི་ཛཱ་ཏ། ནི་ཛཱནྟཾ། ནི་ཛུ་སུ། སོགས་སོ། །ཁྲིཊ། ཕུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཙུའི་ཀུའོས། ཛྲ་གྲ། ཁར་ལ་ཙར་འོས། གྲ་ཀྲ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཥར་བསྒྱུར་བས། ནེ་གླུ་ཏི། སོགས་དང༌། དེ་བཞིན་དུ། ལྀང་ལ། ཨ་ནེ་གྱུཏ། སོགས་སོ། །ས་༈བི་ཛིར་ཐ་དད་པར་གྱུར་པ་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་དང་ཏང་གཉིས་་་འཇུག་སྟེ། །གོ་ཤཔ་ལྷུར་བསྒྱུར་བ་སོགས་ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་་་་འདྲ་སྟེ། ལཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བེསྟི། །བེ་བིཀྐ། བེ་བི་ཛ་ཏི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བེཀྟི། བེ་བིཀྑ། བེ་བིཀྵ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བེÉབི་བི་ཛཱ། །བི་བིཙྪ། །ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བིཏྟེ། བེ་བི་ཛཱ་ཏེ། བེ་བི་ཛ་ཏེ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བིཀྑེ། །བི་བི་ཛཱ་ཐེ། བི་བིབྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཛེ། བེ་བི་ཛ་ཧེ། བེ་བི་ཛ་ཧེ། ལིང་གི་ཨ་ཏང་ལ། བེ་བི་ཛཱཏ། བེ་བི་ཛྙཱ་ཏཱ། བེ་བི་ཛུ། །སོགས་དང༌། ཏང་ལ། མཐའ་མ་ཡིན་པ་སོགས་—།
