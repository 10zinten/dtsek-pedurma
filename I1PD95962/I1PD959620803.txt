སྒྲ་མདོ། ༼ཏི༽་མ་ནི་་ཚོ་་བའི་ཏཚེའི་རིན་པ་ཐ་ནIམང་ཚིག་ལ། །ཨ་པདྷཱ་ཏྭཾ། ་་ཨ་པཏྠ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་པཏྟཱ༑ཨ་པཏྟཱ་ཐཱ། ཨ་པདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་པཏི། ཨ་པཏྟཱ་ཧི། ཨ་པདྨཧི། ལིཊ་ལ་ཨ་བྷཱུ་ས་དབྱི་བ་སོགས་ཀྱིས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། པེ་དེ། པེ་དཱ་ཏེ། པེ་དི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། པེ་དི་ཥ། པེ་དཱ་ཐེ། པེ་དི་དེ། པེ་དི་ཞྭ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། པེ་དེ། པེ་དི་བ་ཧེ། པེ་དི་མ་ཧེ། །ལུཊ་ལ་ཥར་ལ་ཙར་འོས། ད་ཏར་བསྒྱུར་བས། པཏྟཱ། པཏྟཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ་འང་པཏིཥྚི། པཏི་ཡཱ་ཧཱུཾ། པཧྥི་རན། སོགས་སོ། །ལྀཊ་ལ། པཉྩ་ཏེ། སོགས་དང༌། ཁྲིང་ལ། ་ཨ་པཉྩཏ༑སོགས་སོ། །ས་རི་ལ། །སོའུ་དེའོ། །པ་ཀུ། ལུབའི་ནང་༈བུ་དྷ་ཁོང་དུ་ཆུད་པ་ལའོ་ཞེས་པའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། ལཊ་ལ། བུ་དྷ་ཏེ། སོགས་དང༌། ལིང་ལ། བུ་དྷེ་ཏ། །སོགས་སོ། །ལོཊ་ལ། བུ་དྷུ་ཏྭཾ། །སོགས་དང་། ལང་ལ། །ཨ་བུ་དྨ་ཏ། སོགས་སོ། །ལུང་གི་་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། དཱིཔ་ཛན་བུ་དྷ་སོགས་ཀྱིས་ཙྪི་ཙིཎ་དང་སིཙ་་་ཡང་ན་བར་བྱས། ཙིཎའི་ཕྱོགས་ལ་པུཀ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། གཨ་བོ་ེདྷི། སིཅིའི་ཕྱོགས་ལ་ཛྷལ་ལས་ཛྷལ་ལ་འོས་སིཙ་ཕྱིས། །ཛྷཥ་མཐའ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་དྷ་དར་བསྒྱུར་བས། །ཨ་བུདྡྷ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨཙ་གཅིག་པ་སོགས་པ་བྷར་བསྒྱུར་བས། །ལཨ་བྷུཏྟཱ་ཏྭཾ། ཨ་བྷུཏྠ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བུདྡྷེ། ཨ་བྷཱཧ་ཧཱུཾ། ཨ་བྷུདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བྷུཏྟི། །ཨ་བྷུཏྟཱ་ཧི། ཨ་བྷུདྨ་ཧི། ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས། །བུ་བུ་དྷེ། །བུ་བུ་དྷཱ་ཏེ། བུ་བུ་དྷི་རེ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་་བུ་བུ་དྷི་ཥ། བུ་བུ་དྷཱ་ཐེ། །བུ་བུ་དྷི་དྷེ། བུ་བུ་དྷི་བྷེ། sབུདྷཱདྷི—།
