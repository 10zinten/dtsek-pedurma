སྒྲ་མདོ། །nཏི། མ་ཎི་ནི་བློའི་ཏཙལ་སྐྱ་ལ་ཀཀྐོའིདང་པོའི་ཨིཊ། ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། ཨ་བ་དྷཱི་ཏ། རྗེས་སུ་ཨ་ཕྱིས་པ་ཡིན་ལ། དང་པོ་ནས་ཧལ་མཐའ་མ་ཡིན་པས། བད་སོགས་ཀྱིས་འཕེལ་བ་འདིར་མི་བྱེད། ཨ་བ་དྷིཏྟཾ། ཨ་བ་དྷི་ཥཱ་སོགས་སོ། །ལིཊ་ལ་བྱིངས་གཉིས་བརྗོད། །ཀུ་ཧ་དག་གི་ཙུ་འོས་ཧ་ཛྷ། ཨ་་ངྷ་སའི་ཙར་འང་ངོས། ཛྷ་ཛ། ་ཎལ་གཉིས་ལ། ཧནྟིའི་ཧིཊྛིཏ་ནི་རྣམས་ལ་འོས་ཧ་གྷ། ཨུ་པ་དྷཱའི་འང་ངོས་འཕེལ་བ། <ཀིཏ་རྣམས་ལ་གམ་ཧན་སོགས་་་ཀྱིས་ཨུ་པ་དྷཱ་ཕྱིས། ཧནྟིའི་ཧི་སོགས་ཀྱིས་ཧ་གྷར་བསྒྱུར་བ་སོགས་བྱས་པས། ཛ་བྷཱ་ན། ཛ་བྷཱ་ཏུ། ཛམྦྷ། སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ཨ་ཧྥུ་་སཱཙཱ༑ཨ་བྷྱཱ་ས་ལས་ཀྱང་ངོ༌། ཨ་བྷཱ་ས་ལས་ཀྱང་ཕ་རོལ་གྱི་ཧན་གྱི་ཧ་ནི་གྷ་་རུ་བསྒྱུར་ཏེ། ཛ་གྷ་ནི་ཐ། ཛ་གྷནྡྲ། ་གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཛ་་ཐུ༑ཛ་། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཛ་གྷ་ན། ཛ་གྷ་ན། ཛ་རྨི་བ། ཛ་རྨི་མ། ལུཊ་ལ། ཧནྟཱ། ཧནྟཱ་རོ། ་ཧན་ར། ཧན་སི་སོགས་སོ། །ཧ་ནོ་བ་དྷ་ལི་ང་། ལིང་ལ་ཧན་གྱི་བ་དྷའོ། །སྨོན་པའི་ལིང་ཕ་རོལ་ལ་ཧན་ནི་བ་དྷ་ཞེས་པར་བསྒྱུར་བར་བྱ་སྟེ། ཧན་ལ། །སྨོན་ལིང་། མདོ་འདིས་བ་་་དྷར་བསྒྱུར། ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། བ་དྷཱཏ། བ་དྷཱ་སིཾ། བ་དྷཱ་་་སུསོགས་སོ། །ཁྲིཊ་ལ། །རྀཏ་ཧན་དག་གི་ས་ལ་འོས། །་ཨིཊ་བྱིན་པས། ཧ་ནི་བྱ་ཏི། ཧ་ནི་བྱ་ཏ། ཧ་ནི་ཉྩནྟི་སོགས་དང༌། དེ་བཞིན་དུ། ཕྲེང་ལ། ཨ་ཧ་ནི་ཏ། ཨ་ཧ་ནི་་ཏྭཾ། ཨ་ཧ་ནི་སྦྱིན་སོགས་སོ། །ཚན་/ཏོ༈བྱུ་མངོན་པར་འགྲོ་བ་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་འཇུག་ཅིང་ཨའི་སོགས་་་་ཀྱིས་ཤཔ་ཀྱང་འཇིག་གོ། །ཨུ་ཏོ་བྲྀདྷིརླུ་ཀི་ཧ་ལི། འཇིག་པ་ལ་ཧལ་ལ་ཨུཏི་ཀྱིས་འཕེལ་བའོ། །ཤཔ་འཇིག་པ་ལྟ་བུ་བར་རྐྱེན་འཇིག་པ་ལ་རྐྱེན་ཧལ་དང་པོ་པ་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨུཏ་ཀྱི་འཕེལ་བར་འགྱུར་ཏེ། །བྱུ་ལ་ལཊ་སྦྱར་་།
