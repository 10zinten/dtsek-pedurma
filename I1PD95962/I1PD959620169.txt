།དག་པའི་ཁ་ལ་བསྐོ་ཊ། བསྟན་འགྱུར། སྒྲ་མདོ། *ཏིའི་ཀ་བ་ཀ་ཏི་ཀཚོཥ་ཎ་ས་ཀ་ཊ། ཏང་ཤྲཱི་བྷུ། །ལྔ་པའི་ང་སི་ལ། །ཤི་ཡ། ཤི་ཡཱ། བྷཱུམ་བྷུལ། །ཤྲཱི་ཧཱུཾ། ཌཤྲཱི་བྷུ༑། དྲུག་པའི་ངས་ལ། ཤི་ཡ། ཤི་ཡཱ། ཨོས་ལ། ཤི་ཡོ། ཤྲཱི་ལ༑ཨཱམ། བཱ་མི། ་ཨཱམ་ལ་ཡང་ནའོ། །ཨིཡ་ཨུཝ་ཐོབ་པ་རྣམས་རྣམ་་་་དབྱེ་དྲུག་པའི་ཨཱམ་ཕ་རོལ་ལ་ན་དཱིའི་མིང་ཅན་ཡང་ནའོ། །ཤི་ཡཾ། ན་དཱིའི་མིང་ཅན་ལྟར་ན། ཐུང་ངུ་ན་དཱིའི་ཞེས་པས། ནུཊ། བཱི་ཎཾ། བདུན་པའི་དི་ལ། ཤྲི་ཡཾ། ཤི་ཡི༄༅། ཨོས་དང་སུཔ་ལ། ཤི་ཡོ། ཤྲཱི་ཥཱ། སུདྡྷི་བློ་བཟང་མ་ལ་སོགས་པ་ཡང་འདི་དང་འདྲའོ། །པྲ་དྷཱི་བློ་རབ་མ་ཞེས་པའི་སྒྲ་མོ་རྟགས་ལ་ཡང་ཆོ་ག་ཐམས་ཅད་སྐྱེས་་་་པའི་སྐབས་སུ་བཤད་པ་དང་ཁྱད་པར་མེད་དོ། །དེ་རྣམས་ཀྱིས་བུད་མེད་ཀྱི་རྟགས་ལས་ཨཱི་རིང་མཐའ་ཅན་གྱི་བྱ་བ་རྣམས་བསྟན་ཟིན་ཏོ། །རྒྱུ་དེ་ག་དུ། །ཀ༈ད་ནི་མོ་རྟགས་ཨུ་ཐུང་མཐའ་ཅན་རྣམས་བརྗོད་པར་བྱ་སྟེ། དྷེ་ནུ་་་ལ༑འཇོ་མ། སུ། དྷེ་ནུ། ཨོ་ལ། དང་པོ་དག་གི་ཞེས་པས། དྷེ་ནུ། དྷེ་ནུ། ཛས༑་ཛས་ལ་འང་ངོས། དྷེ་ན་བ། གཉིས་པ་རྣམས་ལ། དྷེ་ནུ། ཨམ། ཨམ་ལ་སྔ་མ་ཞེས་པས། དྷེ་ནཾ། དྷེ་ནུ། ཨོ། དྷེ་ནཱི། དྷེ་ནུ། ཤས༑དང་པོ་དག་གི་མདོས། དྷེ་ནུ། གསུམ་པ་རྣམས་ལ། དྷེ་ནུ་ཊ། ཊཱ་ནཱར་མི་འགྱུར། དྷེ་ེནཱ། དྷེ་ནུདྡྷཾ། དྷེ་ནུ་བྷི། བཞི་པ་ལ། བྷེ་ནུ། ངེ་། ང་འགྲོ་བ་ལ་ཐུང་ངུའི་ཡང་ན་འོས་། དྷེ་ནཱི། དྷེ་ན་ཝི། དྷེ་ནུ་ཧཱུཾ། དྷེ་ནུ་བླུ། །ལྔ་པའི་ང་སི་ལ། དྷེ་ནཱ། དྷེ་ནོ། བྷཱུམ་བྱུས་འདྲེ། དྲུག་པའི་ངས་ལ། དྷེ་ནཱ། དྷེ་ནོ། །ཨོས་ལ། དྷེ་ནཱི། ཨཱམ་ལ། །དྷེ་ནཱ་ནཾ། བདུན་པའི་ང་ལ། དྷེ་ནཱཾ། :དྷེ་ནོ། ཨོས་ལ། དྷེ་ནོ། དྷེ་ནུ་ཥཱ། བོད་ཚིག་རྣམས་ལ། ཧེ་དྷེ་ནོ། ཧེ་དྷེ་ནཱ། ཧེ་དྷེ་ན་ཝ་རྣམས་གྲུབ་སྟེ་ཨཙ་མི་གཅིག་པའི་འགྱུར་ཁྱད་ཕྲན་ཙམ་མ་གཏོགས་གོང་དུ་བཤད་པ་མ་ཏིའི་སྐབས་དང་་་་།
