བསྟན་འགྱུར། སྒྲ་མདོ། ཏིའི་ཀ་རེ་སྨོན་སྙན་ཆོ་ཏཚོ་ཚོར་ཎ་ཀ་ཀ༌ཎ་བློ་ནུ༑་སུ་ནུ་ཏཱཏ། སུ་ནུ་ཏཾ། སུ་ནུ་ཏེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སུ་ན་བཱ་ནི༑སུ་ན་བཱ་བ། སུན་བཱ་མ། །ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །་སུ་ནུ་ཏྭཾ༑སུ་ནཱ་ཏཱ། སུ་ནཱ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སུ་ནུ་ཥཱ། སུ་ནཱ་ཐཱ༑་་སུ་ནུ་དྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །སུ་ན་བ། སུ་ན་བཱ་བ་ཧི། སུ་ན་བཱ་མ་ཧིས། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སུ་ནོཏ། །ཨ་སུ་ནུ་ཏཱ། ཨ་སུ་ནཱན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སུ་ནོ། ཨ་སུ་ནུ་ཏཾ། །ཨ་སུ་ནུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་སུ་ན་ཁཾ། ཨ་སུ་ནུ་བ། ཨ་སུ་ནཱ༑བ་མ་སོགས་ཀྱིས་ཡང་ན་ཨུ་ཕྱིས་པའོ། །ཨ་སུ་ནུ་མ། ཨ་སུམྨ། བ་མ་སོགས་ཀྱིས་ཡང་ན་ཨུ་ཕྱིས་པའོ། །ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སུ་ནུ་ཏ། ཨ་སུ་ནཱ་ཏཱ། ཨ་སུ་ནཱ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སུ་ནུ་ཐཱ། ཨ་སུ་ནཱ་ནཱཾ། ཨ་སུ་ནུ་དྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །་ཨ་སུ་ནཱི། ཨ་སུ་ནུ་བ་ཧི། ཨ་སུ་ནཱ་ཧི། ཨ་སུ་ནུ་མ་ཧི༑ཨ་སུམྨ་ཧི། ལུང་གི་ཨ་ཏང་ལ། །སྡུ་སུ་དྷུཉ་རྣམས་ལ་སོགས་ཀྱིས་ཨིཊ་བྱིན། སིཙ་ལ་འཕེལ་བ་སོགས་ཀྱིས་འཕེལ་བ་བྱས་པས། ཨ་སཱ་བཱིཏ། ཡ་ཨ་་སཱ་བིཏྟཾ། ཨ་སཱ་བི་ཥཱ། སོགས་དང་། །ཏང་ལ། ་རུ་དུ་སུ་ནུ་སོགས་ཀྱི་ལུང་འདྲེན་གྱིས་ཨིཊ་ཡང་ན་བར་བྱས་པས། ཨིཊ་བྱིན་པའི་ཕྱོགས་ལ། །ཨ་ས་བི། ཨ་ས་བི་ཥཱ'ཏྭཾ། ཨ་ས་བི་ཥ་ཏ། སོགས་དང་། །ཨིཊ་མེད་པའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སོ། ་ཨ་སོ་ཥཱ་ཏྭཾ། ཨ་སོ་ཥ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སོ་ཥྛ། ཨ་སོ་ཥཱ་ཐཱ། ཨ་སོ་དྷཾ། ྋ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་སོ་ཎི། ཨ་སོ་ཥཱ་ཧི། ཨ་སོཧྥ་ཧི། ལིཊ་ལ་གཉིས་བརྗོད། །ཨཙ་དང་པོ་་ཅན་རྣམས་ལ། །ཨཙ་ལ་་བྱིངས་སོགས་ཀྱིས་ཨུ་བར་བསྒྱུར་བས། ༽ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །སུ་ཥཱ་བ། སུ་ཥཱ་བ་ཏུ། སུ་ཏུ་བུ། སྐྱེས་བུ་་་—།
