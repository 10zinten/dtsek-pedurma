བོད་ཀྱི་ཚོ་༼ཁ་བསྟན་འགྱུར། སྒྲ་མདོ། རྟའིཏི་ལོ་ར་ཁོ་ཚོ་དམག་དར་ཊིཀ་ཀ་ལྷོ་ཁ་བས་བའི་མིང་ཅན་ནོ། །ཎ་ཞེས་པ་ལྕེ་རྩ་ཅན་གྱི་མིང་ཅན་ནོ། །ཀླ་ཞེས་པ་མཆུ་་་ཅན་གྱི་མིང་ཅན་ནོ། །ཨོཾ་ཞེས་པ་རྗེས་སུ་ང་རོའི་མིང་ཅན་ནོ། །ཨཞེས་པ་་་་རྣམ་པར་བཅད་བྱའི་མིང་ཅན་ནོ། །བྲྀདྷི་རཱ་དེཙ། ཨཱཏ་ཨིཙ་འཕེལ་བའོ། །ཨཱཏ་དང་ཨིཙ་དག་འཕེལ་བའི་མིང་ཅན་ནོ། །ཨ་དེདྷུཎ། ཨཏ་ཨེང་ཡོན་་་ཏན་ནོ། །ཨཏ་དང་ཨེང་དག་ཡོན་ཏན་གྱི་མིང་ཅན་ནོ། །ཨི་ཀོ་གུ་ཎ་ཧྲིདྡྷི། ཨིཀ་གི་ཡོན་ཏན་དང་འཕེལ་བའོ། །གོང་དུ་སྨོས་པའི་འཕེལ་བ་དང་ཡོན་ཏན་་གྱི་མིང་ཅན་དེ་རྣམས་ཨ་མ་གཏོགས་ཨིཀ་སོ་སོར་བསྡུ་བའི་ནང་ནས་འབྱུང་སྟེ། ཨའི་འཕེལ་བ་ཨཱཏ། ཨི་འཕེལ་བ་ཨི། །ཨུའི་འཕེལ་བ་ཨོ། ་རི་ལི་གཉིས་ཀྱི་་་འཕེལ་བའང་ཨཱཏ་ཡིན་ལ། ཡོན་ཏན་ནི་ཨའི་ཡོན་ཏན་ཨ་ཉིད། ཨིའི་ཡོན་་་་ཏན་ཨེ། ཨུའི་ཡོན་ཏན་ཨོ། རཾ་ལཾ་གཉིས་ཀྱི་ཡོན་ཏན་ཡང་ཨཏའོ། །ཨ་དན་ནཾ་ལོ་པ། མི་མཐོང་བ་ཕྱིས་པའོ། ་། མདོ་གང་ལའང་དབྱིའོ་ཞེས་པ་བྱུང་ན་ཡི་གེ་གང་ཡིན་པ་དེ་མི་མཐོང་བར་དབྱི་དགོས་ཏེ་དེ་ལ་ཕྱིས་པའི་མིང་ཅན་ཞེས་་་ཟེར་རོ། །བྷུ་བཱ་ད་ཡོ་དྷཱ་ཏ་བ། བྷཱུ་ལ་སོགས་པ་རྣམས་བྱིངས་སོ། །བྷཱུ་ཡོད་པ་ལའོ་ཞེས་པ་ལ་སོགས་པ་རྣམས་ནི་བྱིངས་ཞེས་བྱའོ། །ཨུ་པ་སརྒ་ཀྲི་ཡཱ་ཡོ་གེ། བྱ་བ་སྦྱོར་བ་ལ་ཉེར་བསྒྱུར་རོ། །གང་བྱ་བའི་ཚིག་དང་སྦྱོར་བའི་དྲ་ལ་སོགས་པ་རྣམས་ལ་ཉེར་བསྒྱུར་ཞེས་ཟེར་རོ། །པྲ་ལ་སོགས་པ་ཅི་ཞེ་ན། པྲ། རབཔ་་རཱ༑མཆོགཨ་པ། པྲ་ལ་གནོད། སམ། རྫོགས་པ་ཀུན་ཏུ་ཡང་དག། ཨ་ནུ། རྗེས་སུ། ཨ་བ། ཀུན་་་ཏུ་ངེས་པར། ནིར། ངེས་པ་དགག་པ། དུར། ངན་པ་དཀའ་བ། བི། རྣམ་བྲལ་ལོག། ཀུན། …ཨཱ། བར། ནི༑བྲལ་བ། ཨ་དྷི། ལྷག་པར། ཨ་པི། ཀྱང་འང་ཡང༌། ཨ་ཏི། ཤིན་ཏུ་འདས། སུ། བཟང༌། མཛེས། བདེ། རག། ལེགས། ཤིན་ཏུ། ཨུཏ། ཤིན་ཏུ། ཨ་བྷི། མངོན་པར། པྲ་ཏི། སོ་སོ། པ་་་རི༑ཡོངས་སུ། ཨུ་པ་ཉེ་བ་ཕན་པ་སྟེ་ཉི་ཤུ་པོ་འདི་རྣམས་སོ། །བི་རཱ་མོ་ཨ་བ་སཱ་ནཾ། རྩེ་།
