རྟོགས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་pཎ་ཏི་མེ་ནི་མ་ཧོ་ཎཞི་ཅཚོཥ་ཎ་ཏའ་ཚཉིན་འཚོ་དང་ེ་ཐེ། ་ཨེ་དྷི་ཥྱ་དེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་དྷི་སྐྱེ། ་ཨེ་དྷི་ཥཱ་བ་ཧེ། ཨེ་དྷི་ཤྩ་མ་ཧེ། ཨེ་དྷ་ལ། བྲྀང་། ཨཙ་དང་པོའི་ཨཱཊ་འོས། བྱིངས་ལ་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ། སྱ་སྦྱོར་བ་སོགས་བྱས་པས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཨི་དྷི་ཥྱ་ཏ། ཨ་དྷི་ཙྪེ་ཏྭཾ། ཨི་དྷི་ན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨི་དྷི་ཥྱ་ཐཱ། ཨི་དྷི་ཥྱ་ཐཱ། ཨེ་དྷི་ཀྵ་ནཱཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨི་དྷི་སྐྱེ། ཨེ་དྷི་བྷ་བ་ཧི། ཨེ་དྷི་ཥཱ་མ་ཧི་སོགས་ཏེ། །ཆོ་ག་ཕལ་ཆེར་བྷུ་ཐོབ་པའི་སྐབས་་སུ་བཤད་པ་དང་འདྲའོ། །ལྔ་ང་ལ་་། བླ་མ་གཞུང་། །༈ཨིཀྵ་ལྟ་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ། ་ལཊ་སོགས་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱོར་བའི་ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་འདྲའོ། །་་་་་་། ༈ད་ད་སྦྱིན་པ་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་ཨཏྨ་ནེ་པ་ད་རྣམས་སྦྱར་བས་་ཆོ་ག་ཡང་ལིཊའི་སྐབས་མ་གཏོགས་ཨེ་དྷའི་སྐབས་དང་འདྲ་སྟེ། ལཊ་ལ། ད་ད་ཏེ། ད་ད་ཏེ། ད་དནྟེ་སོགས་སོ། །ལིང་ལ། ད་དེ་ཏ། ད་དེ་ལཱ་ཏཱ། ད་དེ་རན་སོགས་སོ། །ལོཊ་ལ། ད་ད་ཏྭཾ། ད་དེ་ཏཱཾ། ད་དནྟཾ་སོགས་སོ། །ལང་ལ། ་ཨ་ད་ད་ཏ། ཨ་ད་དེ་ཏཱཾ། ཨ་ད་དནྟ་སོགས་སོ། །ལུང་ལ། ཨ་ད་དིཥྛ། ཨ་ད་དི་ཥཱ་ཏྭཾ། ་་་ཨ་ད་དེ་ཥ་ཏ་སོགས་སོ། །ཁྲིཊ་ལ། བྱིངས་གཉིས་བརྗོད་པ་དང་ཝལ་དང་པོ་ལ་ཨིཊ་སྦྱོར་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་ད་དེ། ད་ད་དཱ་ཏེ། ད་ད་དི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་ད་དི་ཏེ། ད་ད་དཱ་ཐེ། ་ད་ད་དི་དེ། ་ད་ད་དི་ལྷེ། སྐྱེས་་བུ་ཕྱི་མ་གསུམ་ལ། །ད་ད་དེ། །ད་ད་དི་བ་ཧེ། །ད་ད་དི་མ་ཧི་སོགས་ཏེ། འདིར་ཤ་ཤ། ད་ད། བ་དང་པོ་སོགས་ཀྱི་མདོས། ཨ་བྷཱ་ས་དབྱིབ་དང་ཨེར་བསྒྱུར་བ་སོགས་མི་བྱེད་དོ། །།ལུཊ་ལ་བྱིངས་ཕྱེད་པའི་བལ་དང་པོ་ལ་ཨིཊ་འོས་་ལབ་མ་།
