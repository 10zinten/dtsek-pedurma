སྒྲ་མདོ། ཏི་ཁ་ཚེ་རིང་བ་ན་མཚོ་ཁ་བ་དང་ལུས་སོགས་བློ་ལ་ལུཊ་ལ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། །ཛྷཥ་སོགས་ཀྱིས་ཏ་དྷི། ཛྷ་ཤ་ལ་ཛུལ་རྣམས་ཀྱི་ཛ་ཤའོས་དྷ་དར་བསྒྱུར་བས། ཨ་ཏང་ལ། རོདྷཱ། རོདྡྷཱ་སི། སོགས་དང་། ཏང་ལ། ་རོདྡྷཱ། རོདྡྷཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། རུ་དྷཱཏ། ་རུ་དྷཱ་ཙཱ། རུ་རྒྱུ་སུ། སོགས་དང༌། ཏང་ལ། ཁར་ལ་ཙར་འོས་དྷ་ཏར་བསྒྱུར་བས། རུཏི། རུཏི་ཡཱ་ཙཱ། རུཏི་རན། སོགས་སོ། །ལྀཊ་ལ། ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཁར་ལ་ཙར་འོས་དྷ་ཏ་དངས་པས། ཨ་ཏང་ལ། རོཉྩ་ཏི། །སོགས་དང་། ཏང་ལ། རོཉྩ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་རོཉྩ་ཏེ། སོགས་དང་། ཏང་ལ། ཨ་རོཉྩཏ། སོགས་སོ། །ང་ནི། །ཁྱུང་པ་༈ཨུ་ཚྭ་དིར་འབར་བ་དང་རྩེ་བ་དག་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཟ་ལ་ཏིཔ། རུ་དྷ་སོགས་ཀྱི་སྒམ་བྱིན་་་ཁར་ལ་ཙར་འོས་ད་ཏར་བསྒྱུར་བས། ཚྭ་ཎཏྟི། །གཉིས་ཚིག་ལ། ཁྲ་ཏས། ཐམ༑ཤ་དང་མ་ཕྱིས། བྷ་ཨས་དག་གི་ཨཏ་དབྱི་འོས། ཀྲིཏ་ཕ་རོལ་ལ་་་་ནའི་ཨ་ཕྱིས། ཁར་ལ་ཙར་འོས་ད་ཏ། ཛྷར་འདྲ་བ་ལ་ཛྷར་རྣམས་ཀྱི་འོས་ཏ་གོང་མ་ཕྱིས། ་ཚོནྟ། མང་ཚིག་ལ། ཁྲ་ཛྷི། ཛྷའི་ཨནྟ་བྷ་ཨས་དག་གི་དབྱི་འོས་ཨ་ཕྱིས་པས། ཚིནྡནྟི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨོཾཎཏི། ཚིནྡྷ། ཨོཾནྡྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ༑ཚྭཎཏྠི། ཚིནྡྲ། ཚིནྡྷ། །ལིང་ལ། ལྷ་ཨས་སོགས་ཀྱིས་ཨ་ཕྱིས་་་་པས༑ཨོཾནྡྷཏ། ཚིནྡྷ་ཧཱུྃ། ཨོཾནྡྷ། ལོཊ་ལ། ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཚིཎཥྛ༄༅། ཚོནྟཱཏ། ཚོནཾ། ཚིནྡནུ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཚིདྡྷི། ཚོནྟཱཏ། ཀྲྀནྟཾ། ཨོཾནྟ། Mཏྭཾ།
