སྒྲ་མདོ། འདིའིཏི་ནི། ཆོས་ཀྱི་དབང་བྱས་པ་ལ་བློ་ལྡན་འགྱུར་བའང་མ་ཡིན་ཏེ། དུབྷ་བཱ་ནི། ངན་པར་གྱུར་ཅིག། བྷཱུ་ཐོབ་པ་ལའོ་ཞེས་པ་ལ་ཤཔ་བྱིན་ནས་ཡོན་ཏན་སོགས་བྱས་པས་བྷ་བ་ཞེས་བསྒྲུབ་པ་སོགས་སྔར་ལྟར་བྱས་ནས་ལོཊའི་ཨཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨཱ་མེ་ཏཨེཏ་ཀྱི་་་ཨཱམ་འོ། །ལོཊ་འགྱུར་བའི་རྣམ་དབྱེ་:རྣམས་ཀྱི་?མཐའི་ཨེ་ཡིག་རྣམས་ནི་ཨཱམ་དུ་འགྱུར་ཏེ། ་ཐོབ་པར་ཤོག། བྷ་བ་ཏྭཾ། བྷ་བ་ལ། ལོཊ་ལ་ཏ་རུ་འགྱུར་ཊ་འགྲོ་བ་ཞེས་པས་ཏའི་སྲོག་ཨ་ཨེ་རུ་འགྱུར། ་མདོ་འདིས་ཨེ་ཨཱམ་དུ་འགྱུར། ལོཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཐོབ་པ་དག་ཤོག། བྷ་བེ་ཏཱ། བྷ་བ་ལ། ལོཊ་ལ་ཨཱ་ཏཱཾ་ང་འགྲོ་བ་ལ་ཨཱཏ་ཀྱི་འོས། ཨཱ་ཨི་ཡ་འདིར་ང་འགྲོ་བ་མ་ཡིན་ཀྱང་ལོཊའི་ལང་འདྲ་བ་འོས། བྲ་ཡ་སོགས་ཀྱིས་ཕྲ་ཕྱིས། ་་ཊ་འགྲོ་བ་སོགས་ཀྱིས་ཨཱམ་ཨེ་ཨེཏ་ཀྱི་ཨཱམ་འོས་སླར་ཨཱམ་དུ་ལོག། བྷ་བནྟཾ། བྷ་བ་ལ། ལོཊ་ལ་ཛྷར་འགྱུར་ཛྷ་ཨནྡྷ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་ཨེཏ་ཀྱི་ཨཱམ་འོས་ཨེ་ཨཱམ་ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས་གཉིས་དག་གི་གཅིག་སྔ་མའི་གཟུགས་སུ་སོང་། :ལོཊའི་ཨཱཏྨ་ནེ་་པ་དའི་སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། །ས་བཱ་བྷཾ་བཱ་མོ། ས་བ་དག་ལས་བ་ཨམ་དག་གོ། །ལོཊ་འགྱུར་བའི་ས་ཡིག་དང་བ་ཡིག་གི་སྲོག་གི་ཨེ་ཡིག་ནི་རིམ་པ་བཞིན་པ་ཨམ་དག་ཏུ་འགྱུར་ཏེ། བྷ་བ་སཱ། བྷ་བ་ལ། ལོཊ་ལ་ཐཱས་སུ་འགྱུར་ཐཱས་ཀྱི་སེ་འོས་ཐཱས་སེ་རུ་འགྱུར། མདོ་འདིས་ཨེ་བ་སྲོག་བཅས་སུ་འགྱུར། སྲ་དྲངས་པའོ། །ལོཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དབུས་མའི་གཉིས་་ཚིག་དང་མང་ཚིག་ལ། ཐོབ་པ་དག་ཤོག། བྷ་བེ་ཐཱ། བྷ་བ་ལ། ལོཊ་ལ་ཨཱ་ཐཾ་དུ་འགྱུར་ཨཱཏ་ཀྱི་རིཏ་ལ་འོ་དང་ལོཊའི་འང་འདྲ་བ་འོས། ཨཱ་ཨི་ཡ་བ་ཡ་སོགས་ཀྱིས་ཡ་ཕྱིས་མཚམས་སྦྱར། ཊ་འགྲོ་བ་སོགས་ཀྱིས་ཨཱ་ཨེ་ཨཏ་ཀྱི་ཨཱམ་འོས་སླར་ཨཱམ་དུ་ལོག། བྷ་བ་དྡྷཾ། བྷ་བ་ལ། ལོཊ་ལ་དཾ་དུ་འགྱུར། ཊ་འགྲོ་བ་སོགས་ཀྱི་མཐའི་ཊི་ཨེ་ས་བ་དག་ལ་བ་ཨམ་དག་གོས་སླར་ཡང་ཨེ་ཨམ་དུ་འགྱུམའི་གཅིག་ཚིག་ལ། ཨེ་ཏཨི། ཨེཏ་ཀྱི་ཨིའོ། །ལོཊ་འགྱུར་བའི་སྐྱེས་བུ་ཕྱི་མའི་རྣམ་དབྱེ་རྣམས་ཀྱི་ཨེ་ཡིག་རྣམས་ནི་ཨི་རུ་འགྱུར་ཏེ། ཐོབ་པར་ཤོགབྷ་བི། བྷ་་བ་ལ། ལོཊ་ལ་ཨི་ཊར་འགྱུར་ཕྱི་མའི་ཨཊ་སྟེ་ཞེས་པས་ཨཱཊ་ཨཱ་ག་མ་ཊེ་གནས་དོན་ཨཀ་འདྲ་བ་ལ་རིང་པོ། ་མདོ་འདིས་ཨེ་ཨི་རུ་འགྱུར་ཨེ་ཙའི་འཕེལ་བ། ལོཊའི་ཨཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་ཕྱི་མའི་གཉིས་ཚིག་དང་མང་—།
