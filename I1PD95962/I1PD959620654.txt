ཥྛ་ཏི་སོགས་དང༌། ཨཱཏྨ་ནེ་པ་ད་ལ། ཧ་རི་ཥྛ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་པ་རསྨི་པ་ད་ལ། ཨ་ཧ་རི་ཏ་སོགས་དང་། ཨཱཏྨ་ནེ་པ་ད་ལ། ་ཨ་་་ཧ་རི་ཎྱ་ཏ་སོགས་སོ། །བ༈བྷཾཉ་འགེངས་པ་ལའོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་པས་ལཊའི་པ་རསྨི་པ་ད་ལ། བྷ་ར་ཏི་སོགས་དང་། །ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། །བྷ་ར་ཏེ་་སོགས་སོ། །དེ་བཞིན་དུ། ལིང་གི་ཨ་ཏང་ལ། བྷ་རེཏ་སོགས་དང༌། ཏང་ལ། ཨ་བྷ་རེཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། བྷ་ར་ཏུ་སོགས་དང་། ཏང་ལ། བྷ་ར་ཏཱ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ཨ་བྷ་རཏ་སོགས་དང་། ཏང་ལ། ཨ་བྷ་རཏ་སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ། སིཙ་ལ་འཕེལ་བ་ཞེས་སོགས་ཀྱི་མདོས་འཕེལ་བ་བྱས་པས། ཨ་བྷཱནྟིཏྲ། ཨ་བྷནྡྷཾ། ཨ་བྷཱརཱུསོགས་དང་། ་ལུང་གི་ཏང་ལ། རཾ་ལས་ཀྱང་ངོས། ཀིཏ་ཡིན་པས་ཡོན་ཏན་མི་བྱེད། གཟུགས་ཐུང་ངུ་ལས་སོ་ཞེས་པའི་མདོས་རྣམ་དབྱེ་ཛྷལ་དང་པོ་རྣམས་ལ་སིཙ་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། །ཨ་བྲྀ་ཏ། ཨ་བཱི་ཥཱ་ཏྭཾ། ཨ་པྲྀ་ཥ་ཏ་སོགས་སོ། །ལིཊ་ལ། ཀྲྀ་སྲི་བྷཾ་སོགས་ཀྱི་ཝལ་དང་པོ་རྣམས་ལ་ཨིཊ་མི་འཇུག་སྟེ། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་བྷཱ་ར། བ་བྷཱ་ཏུ། བ་དྷུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་བྷརྦ། བ་བྷཱ་ཐུ། བ་གྲྭ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་བྷཱ་ར། བ་བྷ་ར། བ་ཤྲཱི་བ། བ་བྷཾ་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑བ་ཧྥེ། བ་བྷ་ཏེ། བ་ཤཱི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་ཛྲ་ཥ། བ་བྷཱ་ཐེ། བ་ཤྲཱི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་ཧྥེ། བ་ཤྲཱི་བ་ཧེ། བ་་ཤྲཱི་མ་ཧེ། ལུཊ་ལ་འགྲེལ་པར་ཨིཊ་ཡང་ན་བར་བཤད་པས། ཨ་ཏང་ལ། བྷ་རི་ཏཱ། བྷ་རི་ཏཱ་སི་སོགས་སམ། བྷཏཱ། བྷཎཱ་སི་སོགས་སོ། །དེ་བཞིན་དུ། ——D?།
