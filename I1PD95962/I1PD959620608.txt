བུ་དབུས་མ་གསུམ་ལ། ཨེ་དྷ་སཱ། ཨེ་དྷ་ལ་ཤཔ་ལོཊ་ལ་ཐཱས་ཐཱས་ཀྱི་ས་འོས་ཐཱས་སེ་ས་ཝ་དག་ལ་བ་ཨམ་དག་གོས། ཨེ་ཝ། ཨེ་དྷེ་ཛཾ། ཨེ་དྷ་ལ་ཤཔ་ལོཊལ་ཨཱ་ཐཱ་ཨཱཏ་ཀྱི་རིཏ་ལ་འོས་ཨཱ་ཨིཡ་བཡ་དག་གི་ཞེས་པས་ཡ་ཕྱིས་ཨི་མཚམས་སྦྱར། ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། ཨེཏ་ཀྱི་ཨཱམ་འོས་ཨེ་སླར་ཨཱམ། ཨེ་དྷ་ཧཱུཾ། ཨེ་དྷ་ལ་ཤཔ་ལོཊ་ལ་ཚྭ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེས་བདག་ལ་བ་ཨོཾ་དག་གོས། ཨེ་སླར་ཡང་ཨམཅི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་དྷི། …ཨེ་དྷ་ལ་ཤཔ་ལོཊ་ལ་ཨིཊ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་ཨི་ཨེ་ཨེཏ་ཀྱི་ཨི་འོས་ཨེ་ཨི་རུ་བསྒྱུར་ཕྱི་མའི་ཨཱཊ་སྟེ་ཞེས་པས་ཨཱཊ་ཨཀ་འདྲ་བ་ལ་རིང་པོ་ཨེ་ཙའི་འཕེལ་བ་བྱས། ཨེ་དྷཱ་བ་ཧེ། ཨེ་དྷ་ལ་ཤཔ་ལོཊ་ལ་ཝ་ཧི་དང་མ་ཧིང་དུ་བསྒྱུར་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་ཨེཏ་ཀྱི་ཨི་འོས་ཨེ་ཨི་ཕྱི་མའི་ཨཱཊ་སྟེ་ཞེས་པས་ཨཱཊ་ཊ་གནས་དོན་ཨཀ་འདྲ་བ་ལ་རིང་པོ། ཨེ་དྷཱ་མ་ཧི། ཨེ་དྷ་ལ་ལང་སྦྱར་བའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨི་དྷ་ཏ། ཨེ་དྷ་ལ་ལང་། ལ་ཏ་ཨཙ་དང་པོ་རྣམས་ཀྱི་ཨཱཊ་འོས་བྱིངས་ལ་ཨཱཊ་ཊ་གནས་དོན། ཨཊ་ལས་ཀྱང་ངོས་འཕེལ་བ། ཨི་དྷེ་ཏྭཾ། ཨེ་དྷ་ལ་ཤཔ་ལང་བྱིངས་ལ་ཨཱཊ་དང་འཕེལ་བ་ལ་ཨཱ་ཏཱ་མ་ང་འགྲོ་བ་ལ་ཨཱཏ་ཀྱི་འོས་ཨཱ་ཨི་ཡ་ཝ་ཡ་དག་གི་ཞེས་པས་ཡ་ཕྱིས་མཚམས་སྦྱར། ཨེ་དྷནྟ། ཨེ་དྷ་ལ་ཤཔ་ལང་བྱིངས་ལ་ཨཱཊ་དང་འཕེལ་བ། ལཛྷ་ཛྷའི་ཨནྟ་འོས་ཛྷ་ཨནྟ། ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས་གཉིས་དག་གིས་གཅིག་ཕ་རོལ་གྱི་གཟུགས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨེ་ད་ཐཱ། ཨེ་དྷ་ལ་ཤཔ། ་ལང་ལ་ཐས་བྱིངས་ལ། ཨཱཊ་དང་འཕེལ་བ། །ཨེ་དེ་ཐཱ། ཨེ་དྷ་ལ་ཤཔ་ལང་བྱིངས་ལ་ཨཱཊ་དང་འཕེལ་བ་ལ་ཨཱ་ཐཱམ་ང་འགྲོ་བ་ལ་ཨཱཏ་ཀྱི་འོས་ཨཱ་ཨིཡ་བ་ཡ་དག་གི་བལ་ལ་འོས་ཡ་ཕྱིས་མཚམས་སྦྱར། །ཨེ་དྷ་ལ་ཤཔ་་ལང་བྱིངས་ལ་ཨཱཊ་དང་འཕེལ་བ་ལ་ཧཱུྃ་དུ་བསྒྱུར། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨི་དྷི། ཨེ་དྷ་ལ་ཤཔ་ལང་་བྱིངས་ལ་ཨཱཊ་དང་འཕེལ་བ་ལ་ཨིཊ་རུ་བསྒྱུར་མཚམས་སྦྱར། ཨེཨེ་དྷ་ལ་ཤཔ་ལང་བྱིངས་ལ་ཨཊ་དང་་འཕེལ་བ་ལ་ཝ་ཧི་དང་མ་ཧིང་དུ་བསྒྱུར་ཨཏ་ཀྱི་ཡཉ་ལ་རིང་པོ་འོས་རིང་པོ། ཨེ་དྷཱ་མ་ཧི། ཨེ་དྷ་ལ་ལུང་། ལུང་ལ་ཙླི་འོས། ཙྪི་བྱིན། ཙྪིའི་སིཙ་འོས་སི་ཙར་བསྒྱུར། །ཨཙ་དང་པོའི་ཨཱཊ་་འོས་བྱིངས་ལ་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ་དང༌། བྱིངས་ཕྱེད་པའི་་བལ་ལ་ཨིཊ་འོས། ཨིཊ་སྦྱོར་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨི་དྷིཥྛ། ལ་ཏི་རུ་བསྒྱུར། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ་ཥ་ཁུ་ལ་ཁ་ཊུ་འོས་ཏ་ཊ། ཨི་དྷི་ཥཱ་ཏྭཾ། ལ་ཨཱ''་ད་ནི་རྒྱ་/—།
