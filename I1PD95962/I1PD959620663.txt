སྒྲ་མདོ། ཏི་ཏིབྦ་ནི་ཝ་ཁོ་འཚོ་ཚོཥ་དང་པོ་ལ་ཁབློ་རྣལ་མ་གསུམ་ལ། ཨི་ཡཱ་ཛ། །ཨི་ཡ་ཛ། ཨཱི་ཛི་བ། ཨི་ཛི་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨཱི་ཛེ། ཨཱི་ཛཱ་ཏེ། ཨཱི་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨཱི་ཛི་ཏེ། ཨཱི་ཛཱ་ཐེ། ཨཱི་ཛི་དེ། །ཨཱི་ཛི་ཝཱི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨི་ཛེ། ཨཱི་ཛི་བ་ཧེ། ཨཱི་ཛི་མ་ཧེ། ལུཊ་ལ། ་བྲ༔་བྷསྔ་སོགས་ཀྱིས་ཛ་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་སོགས་ཀྱི་ཨ་ཏང་ལ།  ོཡཏྟཱ། ཡཏྟཱ་སི་སོགས་དང༌། ཏང་ལ། ཡཏྟཱ། ཡཏྟཱ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་་་ཏང་ལ། བ་ཙི་སཱ་པི་ཡ་ཛ་ལ་སོགས་པ་རྣམས་ཀྱི་ཀིཏ་ལ་འོས་སཾ་པྲ་ས་ར་ཎར་བསྒྱུར་བས། །ཨི་ཛཱཏ། ཨི་ཛཱནྟཾ། ཨི་ཛཱ་སུསོགས་དང་། ཏང་ལ། བཅུ་བྷསྔ་སོགས་ཀྱིས་ཛ་ཥ། ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས། ཥ་ཀས། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ཡཁཱི། ཡཀྴི་ཡཱནྟཾ། ཡཀྵེ་རན་སོགས་སོ། །ཀྲྀཊ་ལའང་ཛ་ཥར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ། །ཡཀྵ་ཏི་སོགས་དང་། ཏང་ལ། །ཡགྱུ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། །ཀྲྀང་གི་ཨ་ཏང་ལ། ཨ་ཡཀྵ་་ཏ་སོགས་དང་། ཏང་ལ། :ཨ་ཀྵ་ཙིཏ་སོགས་སོ། །འོ་ན་ཊི། རྒྱུ། :::འ༈ཊུ་བ་པ་ས་བོན་འདེབས་པ་ལའོ་ཞེས་པ་ལ་ཡང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། ལཊའི་ཨ་ཏང་ལ། བ་པ་ཏི་སོགས་དང་། ཏང་ལ། བ་པ་་ཏེ་སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། །བ་པེ་ཏ་སོགས་དང་། །ཏང་ལ། ་བ་པེ་ཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། ་བ་པ་ཏུ་སོགས་དང༌། ཏང་ལ། བ་པ་ཏྭཾ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ཨ་བ་པ་ཏ་སོགས་དང་། ཏང་ལ། ཨ་བ་པ་ཏ་སོགས་སོ། །ལུང་ལ་ཛྷལ་དང་པོ་ཅན་རྣམས་ལ། ཛྷལ་ལས་ཛྷལ་ལ་འོས་སིཙ་དབྱི་བ་སོགས་བྱས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བཔཱིཏ། ཨ་བཙཾ། ཨ་བསླུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བདྷཱི། ཨ་པཏྟཾ། །
