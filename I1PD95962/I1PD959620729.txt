་ན་བ་གནས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིཁ་ཆེ་དེ་དག་མ་ཚོགས་ལ་རིན་མ་བློ་གསུམ་ལ། ཨ་དྷ་ཀཱི་ཥྛ། ::ཨ་དྷ་ཀཱི་ཥཱ་ཐཱ། ཨ་རྒྱ་གཱི་ཏཱཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་རྒྱ་གཱི་ཏི། ཨ་རྒྱ་གཱི་ཥཱ་ཧི། ཨ་རྒྱ་གཱི་ཧི། གཱང་དུ་མ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཨ་ཙ་དང་པོའི་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ་བྱས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷིཥྛ། ་ཨ་དྷིཥཱ'ཏྭཾ། །ཨ་དྷི་ཥཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ངྷི་ཥྛ། ཨ་དྷི་ཥཱ་ཐཱ། ཨ་དྷི་ཧཱུཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དྷི་ཏི། ་ཨ་དྷི་ཥཱ་ཧི། ཨ་དྷིམྨེ་ཧི། གཱང་ལི་ཊི། ལིཊ་ལ་གཱང་འོ། ་། ལིཊ་ཕ་རོལ་ལ་ཨིང་ནི་གཱང་དུ་བསྒྱུར་བར་བྱ་སྟེ། །ཨ་དྷི་སྔོན་མའི་་་ཨིང་ལ་ལིཊ། མདོ་འདིས་གང་དུ་བསྒྱུར། གཉིས་བརྗོད། ཨིཊ་ལའང་ཨཱཏ་ཀྱི་དབྱི་འོས། ་ཨཱཏ་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཨ་དྷི་ཛ་གེ། ་ཨ་དྷི་ཛ་གཱ་ཏེ། ཨ་དྷི་ཛ་གི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷི་ཛ་གི་ཏེ། ཨ་དྷི་ཛ་གཱ་ཐེ། ཨ་དྷི་ཛ་གི་དྡྷེ། ཨ་དྷི་ཛ་གི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དྷི་ཛ་གེ། ཨ་དྷི་ཛ་གི་བ་ཧེ། ཨ་དྷི་ཛ་གི་མ་ཧེ། ལུཊ་ལ་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨ་དྭེ་ཏཱ། ཨ་དྷེ་ཏཱ་སེ། །སོགས་དང༌། དེ་བཞིན་དུ་སྨོན་པའི་ལིང་ལ། ཊཾ་ཨ་དེ་ཊཱི། ཨ་དྭེ་ཏཱི་ཡཱ་ཙཱ། ཨ་དྷེ་ཏཱི་ར་ན། ཆསོགས་སོ། །ལྀཊ་ལའང་ཡོན་ཏན་བྱས་པས། ་ཨ་དེ་བྱ་ཏེ། སོགས་སོ། །ལྀང་ལ། །ལུང་ལྡིང་དག་ལ་གདམ་ང་འོས། གཱང་དུ་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། གཱང་གི་ཕྱོགས་ལ། གླུ་མཱ་སྦྲ་གཱ་སོགས་ཀྱིས་ཨཱིཏ་དུ་བསྒྱུར་བས། །ཨ་དྷ་ཀཱི་ཥྛ་ཏེ། །ཨ་དྨ་གི་མྨེ་ཧཱུཾ། ཨ་རྒྱ་གཱི་ཥྛནྟ། སོགས་དང་། མ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཨཙ་དང་པོ་སོགས་ཀྱིས་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ་བྱས་པས། ཨ་དྷི་ཏ། ཨ་དྷི་བྱེ་ཏྭཾ༑ཨ་དྷི་ནྟ། སོགས་སོ། །།།།།ལྟག་རྒྱུ་དེ་བ་ཁ་མ་—།
