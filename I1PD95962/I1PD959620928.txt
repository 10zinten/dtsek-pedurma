དུ་། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས། ཨ་བི་བྷུ་ཛཏ། ཨ་བི་བྷུ་ཛ་ཧཱུཾ། ཨ་བི་བྷཱ་ཛན། སོགས་དང༌། ་ཐུང་ངུར་མ་བསྒྱུར་བའི་ཕྱོགས་ལ། །ཨ་བི་བྷཱ་ཛཏི༑༑ཨ་བི་བྷཱ་ཛ་ཏྭཾ། ཨ་བི་བྷཱ་ཛན། སོགས་སོ། །ལ་ཡིག་ལྷག་མ་རྣམས་ལ༑ལིཊབྷཱ་ཛ་ཡཾ་ཙ་ཀཱ་ར། ལུཊ་བྷཱ་ཛ་ཡི་ཏཱ། ་སྨོན་པའི་ལིང་བྷཱ་ཛཱཏ། ཁྲིཊ་བྷཱ་ཛ་ཡི་ཥྛ་ཏི། ཁྲིང་ཨ་བྷཱ་ཛ་ཡི་ཏ། སོགས་སོ། ་ཀྱི། ་ལྟ། སྐུ་ད་ན། ༈བྷ་ས་འབར་བ་ལའོ་ཞེས་པ་ལ་ཎི་ཙ་སྦྱར་བས་བྷཱ་སི་ཞེས་པ་ལ། ལཊ་སོགས་སྦྱར་བས། ་ལཊ༌བྷཱ་ས་ཡ་ཏི། །༼ལིང་བྷ་ས་ཡེཏ། ལོཊ་བྷཱ་ས་ཡ་ཏུ། ལང་ཨ་བྷཱ་ས་ཡཏ། །སོགས་དང་། །ལུང་ལ། ་བྷཱ་ཛ་བྷཱ་ས་སོགས་ཀྱི་མདོས། ཨུ་པ་དྷཱ་ཐུང་ངུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། །ཨ་བི་བྷཱ་་་སཏ། ་སོགས་སམ། ཨ་བི་བྷཱ་སཏ། །སོགས་སོ། །ལ་ཡིག་ཕྱི་མ་རྣམས་ལ། ལིང་བྷཱ་ས་ཡཾ་ཙ་ཀཱ་ར། །ལུཊ་བྷཱ་ས་ཡི་ཏཱ། སྨོན་པའི་ལིང༌བྷཱ་སཱུཏ། ཁྲིཊ༌བྷཱ་ས་་ཡི་ཥྛ་ཏི། ་ཕྲེང་ཨ་བྷཱ་ས་ཡི་ཏ། །སོགས་སོ། །ཁོ་ར་སློབ་ཞུ་བ་བྱུང་དྲུག་ལ། ༈དེ་བཞིན་དུ། བྷཱ་ཥ་བརྗོད་པ་ལ་ཎིཙ་སྦྱར། དེ་ལ་ལཊ་སོགས་སྦྱར་བས༑ལཊཡབྷཱ་ཥ་ཡི་ཏི། ལིང་བྷཱ་ཥ་ཡེཏ། ལོཊ་བྷཱ་ཥ་ཡ་ཏུ། ལོ་ལང་ཨ་བྷཱ་ཥ་ཡཏ། །སོགས་དང་། ལུང་ལ། བྷཱ་ཛ་བྷཱ་ཤ་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཐུང་ངུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ༼ཨ་བི་བྷ་ཥཏི། ་སོགས་སམ། ཨ་བི་བྷཱ་ཥཏ། སོགས་སོ། །།།ལ་ཡིག་ཕྱི་མ་རྣམས་ལ། ལིཊ་བྷཱ་ཥ་ཡཾ་ཙ་ཀཱ་ར། ལུཊ༌བྷཱ་ཥ་ཡི་ཏཱ། ༼སྨོན་པའི་ལིང་བྷཱ་བྷཱུཏ། །ཁྲིཊ'བྷཱ་ཥ་ཡི་ཥྛ་ཏི། ་ཕྲེང་…ཨ་བྷཱ་ཥ་་་ཡི་ཏ། སོགས་སོ། །བཀའ་བློ་ངོ་ལ། དཀར་བྱ་བ་ཞེས་བྱ་བ། ༈དཱི་པ་འབར་བ་ལ་ཎིཙ་དང་ལཊ་སོགས་སྦྱར་བས། ལཊ་དཱི་པ་ཡ་ཏི། ལིང༌ཡདཱི་པ་ཡེཏ། ལོཊ་དཱི་པ་ཡ་ཏུ། ལང་ཨ་དཱི་པ་ཡཏ། སོགས་དང་། ལུང་ལ། །།
