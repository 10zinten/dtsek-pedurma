བསྟན་འགྱུར། སྒྲ་མདོ། ཏི། ཉིན་མོ་ཁ་ཏཁྱོལ་བ་ཁ་ཏི་ས་བློ་རྣམ་དབྱེ་སོགས་ཕྱིས། འདིར་ཀུན་སྡུད་པ་མ་ཡིན་པས་གཅིག་ཚིག་མི་འཇུག། གངྒཱ་ཡ་མུ་ནེ། ཆུ་བོ་གངྒཱ་ཆུ་བོ་ཡ་ཡུ་ན་དགགྲོང་གི་སྒྲ་བརྗོད་པ་ཡིན་ན་ཀུན་སྡུད་པར་མི་འགྱུར་ཏེ་གཅིག་གཅིག་སྦྱོར་བ་ཉིད་དོ། །ཛམྦ་བ༔་ཤཱ་ལཱུ་ཀི་ནཱི་ཙ། ཛཱམྦུའི་གྲོང་དང་སཱ་ལུའི་གྲོང་། ཞེས་པ་ལ། ཛཱམྦ་བ་ཤཱ་ལཱ་ཀི་ཧཱུཾ། ཛམྦུའི་གྲོང་དང་སཱ་ལུའི་གྲོང་དག། ཀླུ་དྲ་ཛན་་་བ༑སྲོག་ཆགས་ཕྲ་མོ་རྣམས་སོ། །སྲོག་ཆགས་ཕྲ་མོ་རྣམས་ཀུན་སྡུད་པ་དོན་གཅིག་པར་འགྱུར་ཏེ། ཡུ་ཀཱ་ཙ་ལིཀྵ་ཙ། ཤིག་དང་སྲོ་མ་དང༌། ཞེས་པ་ལ། ཡུ་ཀཱ་ལིཀཾ། སྲོ་ཤིག། ཡེ་་ཙ་བི་རོ་དྷཤ་ཤཱ་ཏི་ཀ། གང་རྣམས་ཀྱི་གཏན་གྱི་དགྲ་རྣམས་ཀྱིའོ། །སྲོག་ཆགས་གང་དང་གང་གི་གཏན་གྱི་་དགྲ་རྟག་ཏུ་འཁོན་པ་་་་རྣམས་ཀུན་སྡུད་པ་དོན་གཅིག་པར་འགྱུར་ཏེ། ཨ་ཧིཉྩ་ན་ཀུ་ལཉྩ། སྦྲུལ་དང་སྲེ་མང་། ཞེས་པ་ལ། ཨ་ཧི་ན་ཀུ་ལཾ། སྤྲུལ་སྲེ་མང་ཅི། དེ་བཞིན་དུ། ཨ་ཤཱཉྩ་མ་ཧི་ཥ། རྟ་དང་མ་ཧེ་དང་། ཞེས་པ་ལ། ཨ་ཤྭ་མ་ཧི་ཧཾ། རྟ་མ་ཧེ། ཀཱ་ཀཉྩ་ཨུ་ལུ་ཀཉྩ། བྱ་རོག་དང་འུག་་པ་དང་། ཞེས་པ་ལ། ་ཀཱ་ཀོ་ལུ་ཀཾ། བྱ་རོག་འུག་པ། འགྲེལ་པར་གཏན་གྱི་དགྲ་ཡིན་་་ཀྱང་ལྷ་དང་ལྷ་མིན་ནི་ཀུན་སྡུད་པར་མི་འགྱུར་ཏེ། གཅིག་གཅིག་སྦྱོར་བ་ཉིད་དོ་ཞེས་བཤད་པས། ?དེ་བཉྩ་དེ་ཏུཉྩ། ལྷ་དང་ལྷ་མིན་དང་། ཞེས་པ་ལ། ཝདེ་བ་དེ་ཅོ༑ལྷ་དང་ལྷ་མིན་དག། ཤུ་དཱ་ཎ་མ་ནི་ར་བ་སི་ཏཱ་ནཾ། དམངས་རིགས་རྣམས་ཀྱི་མི་གནས་པ་མ་ཡིན་པ་རྣམས་ཀྱིའོ། །དམངས་རིགས་ཀྱི་རིགས་ཡིན་པའི་ལས་བྱ་བ་དེ་མི་གནས་པ་མ་ཡིན་པ་གནས་པའི་ལས་བཅས་དེ་ནི་ཀུན་སྡུད་པ་དོན་གཅིག་་པར་འགྱུར་ཏེ། ར་ཛ་ཀཉྩ་ཏནྟུ་བཱ་ཡཉྩ། རས་འཁྲུད་མཁན་དང་ཐ་ག་པ་དང༌། ཞེས་པ་ལ། ར་ཛ་ཀ་ཏནྟུ་བཱ་ཡཾ། རས་འཁྲུད་མཁན་དང་ཐ་ག་པ། མི་གནས་པ་ཡིན་ན། ཀུན་སྡུད་་་་པར་མི་འགྱུར་ཏེ། ཙཎྜ་ལ༔་མྲྀ་ཏ་ཡཉྩ། གཏུམ་པོ་དང་རོ་དང་། ཞེས་པ་ལ། ཙཱཎཱ''ལ་མྲྀ་ཏ་ཡོ། གཏུམ་པོ་དང་རོ། ག་བཱཉྩ་པྲ་བྷཱི་ཏི་ནི་ཙ། བ་ལང་ལ་སོགས་པ་རྣམས་་་—།
