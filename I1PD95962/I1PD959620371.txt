སྒྲ་མདོ། ༼ཏིཁ་ཆེ་ནི་ལྟ་རོ་ཚཚོ་ཁྱོ་ཎཝ་ད་བ་ཀ་ཛས། ཡཙུ་ཏཱ་ཤོ་ཀཱ། ཤིང་ཙུ་ཏེ་དང་ཤུག་པ་རྣམས། དེ་བཞིན་དུ། །སྒྲ་སྒྲོགས། རུ་རུ༔་པྲ་ཥ་ཏཉྩ༑ཡརི་དྭགས་རུ་རུ་དང་ཤ་བ། ཞེས་པ་ལ། སུ་ཡརུ་རུ་པྲ་ཥ་ཏཾ། རི་དྭགས་རུ་རུ་ཤ་བ། རྣམས། ཏཱ། ཛས། ཀུ་ཤ་་ཀཱ་ཤཤུ། རྩྭ་ངན་སེལ་དང་ཀཱ་ཤ། ཞེས་པ་ལ། སུ་ཡཀུ་ཤ་ཀཱ་ཤོ༑ཡངན་སེལ་ཀཱ་ཤ། རྣམས། ཤཱཛས། ཡ་བཤྩ་མཱ་ཥ། ནས་དང་སྲན་མ་དང༌། ཞེས་པ་ལ། སུ། ཡ་བ་མཱ་ཎཾ། ནས་སྲན། རྣམས། ཥཱཛས། ང་དྷི་ཙ་གཱི་ཏཾ་ཙ། ཞོ་དང་འོ་མ། ཞེས་པ་ལ། སུ། ཡད་ེདྷི་གཱི་ཏཾ། ཞོ་འོ་མ། ཨོ་ཤཱི། ཏེ། རྣམས། བ་ལང་དང་མ་ཧེ། གོཉྩ་མ་ཧི་ཥ། ཞེས་པ་ལ། སུ། གོ་མ་ཧི་ཧཾ། .བ་ལང་མ་ཧེ། ཨོཁོ། དག། ཤུ་ཀཉྩ་ཧངྒ། ནེ་ཙོ་དང་ངང་པ་དང༌། ཞེས་པ་ལ། སུ། ཀུ་ཤ་ཧཾ་སཾ། ནེ་ཙོ་ངང་པ། རྣམས། སཱཙྪཱཛས། ཨ་ཤཱཉྩ་བ་ཌ་་བཉྩ༑ཡརྟ་དང་རྒོད་མ་དང་། ཞེས་པ་ལ། སུ། ཨ་ཤྭ་བ་ཌ་ཁཾ། རྟ་རྒོད་མ། ཛས། བཱ། ཡརྣམས། པཱུརྦ་ཨ་པ་ར། ཤར་དང་གཞན། ཞེས་པ་ལ། པཱུརྦ་པ་རཾ། ཤར་གཞན། ཛས། རཱ། རྣམས། ཨ་དྷ་ར༔་ཨུཏྟ་ར། མཆུ་དང་བྱང་ངམ་དམའ་དང་བྱང་། ཞེས་པ་ལ། སུ། ཨ་དྷ་རོཏྟ་་་རཾ། མཆུ་བྱང༌། 《ཛས། རཱ། རྣམས། །བི་པྲ་ཏི་ཥིདྷཾ་ཙཱ་ན་དྷི་ཀ་ར་ཎ་བཱ་ཙི། འགལ་བ་རྫས་་མེད་པ་རྣམས་ཀྱི་བརྗོད་པ་ལའང་ངོ༌། །།འགལ་བ་རྫས་མེད་པ་རྣམས་ཀྱི་བརྗོད་པ་ལའང་ཀུན་སྡུད་པ་དོན་གཅིག་པར་ཡང་ན་འགྱུར་ཏེ། ཤི་་་ཏཉྩ་ཨུབྷཾ་ཙ། གྲང་བ་དང་དྲོ་བ་དང་། ཞེས་པ་ལ། ་སུ། ཤི་ཏོ་ཧཱུཾ། གྲང་དྲོ། ཨོ། བྷེ། དག། དེ་བཞིན་དུ། །ཀཱ་མཉྩ་ཀྲོ་དྷ། ལས་དང་ཁྲོ་བ་དང་། ཞེས་པ་ལ། ཀཱ་མ་ཀྲོ་དྷཾ། ལས་་ཁྲོ་བ། ཨོ་དྷོ། དག་། སུ་ཁཾ་ཙ་དུ༔ཁཾ་ཙ། བདེ་བ་དང་སྡུག་བསྔལ་དང༌། ཞེས་པ་ལ། སུ། སུ་ཁ་དུཁཾ། བདེ་སྡུག། ཨོཁེ། དག། ནི་ད་དྷི་པ་ཡཱ་དཱི་ནི། ད་དྷི་པ་ཡས་ལ་སོགས་པ་རྣམས་མ་ཡིན་ནོ། ལ་། ད་དྷི་པ་ཡས་ལ་སོགས་པ་རྣམས་ཀུན་སྡུད་པ་དོན་གཅིག་པར་འགྱུར་བ་མ་ཡིན་ཏེ། ད་དྷི་ཙ་པ་ཡ། ཞོ་དང་འོ་མ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། ད་དྷི་པ་ཡ་སཱི། ཞོ་འོ་མ་དག། དེ་བཞིན་དུ། ་ཨིནྡྷ་་་olisཞོ་འོ—།
