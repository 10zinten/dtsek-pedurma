མོ་ལྟ་ཀཁ་ཎཀློབ་མ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིཁ་ཚོRནི་ལྟ་ཏོ་མཚོ་ཨོཥ་ཎ་ཥ་༌་་་ཀRཀྟ་ལྟ་བ་ཏུ་ནི། ལྟ་དང་ལྟ་བ་ཏུ་དག་ཎིཏྟཱའོ། །།རྐྱེན་ལྟ་དང་ལྟ་བ་ཏུ་དག་ལ་ནིཥྛ་ཞེས་མིང་འདོགས་བྱེད་དོ། །དེ་གཉིས་ལས་ལྟ་ནི་བྱུང་བ་བརྗོད་པའི་དངོས་པོ་དང་ལས་ཀྱི་དོན་ལ་འགྱུར་ཞིང་ལྟ་པ་ཏུ་ནི་བྱེད་པ་པོ་བརྗོད་པ་ལ་འགྱུར་ལ། རྐྱེན་འདི་གཉིས་ཀྱི་ཀ་ཀིཏ་ཀྱི་དོན་དང་། ཨུ་སུ་ཊའི་དོན་ཏོ། །སྣ་དང་སྐུ་ལ་ལྟ་སྦྱར་བས། སྣ་ཏཾ་མ་ཡ། བདག་གིས་ཁྲུས་བྱས། སྟུ་ཏཾ་བུདྡྷ། ཁྱོད་ཀྱིས་སངས་རྒྱས་བསྟོད། ཏཱ་ཡཱ་ཀྲྀཉ་ལ་ལྟ་བ་ཏུ་སྦྱར་བས། །བིgབིཙྩཾ་ཀྲྀ་ཏ་བཱན། ཁྱབ་འཇུག་གིས་ཁྱབ་པར་བྱེད། ནིཥྛ་ཡཾ་ཙྪ་དརྗེ། ཙྩཏ་ཀྱི་དོན་ལ་ནིཥྛའོ། །གོང་དུ་བཤད་པའི་རྐྱེན་ཙུཏ་འཇུག་པའི་དོན་རྣམས་ལའང་ནིཏྟཱའི་མིང་ཅན་གཉིས་འགྱུར་རོ་ཞེས་པ་སྤྱིར་བཏང་ངོ༌། །ཀྱི་ཡོ་དཱིགྷཱཏ། ཀཱིའི་རིང་པོ་ལས་སོ། །ནིཥྛའི་རྐྱེན་ཕ་རོལ་ལ་གླིའི་ཨི་རིང་་་་་པོར་བསྒྱུར་ཞིང་རིང་པོའི་ཕ་རོལ་ལ་ལྟ་དང་ལྟ་བ་ཏུའི་ཏའང་ནར་བསྒྱུར་ཏེ། གི་ཎ་བཱན། ཟད་པར་བྱེད། གླི་ནཟད། བྱུ་ཀgཀི་ཏི། ཤི་ཨུཀ་ལས་ཀ་འགྲོ་བ་ལའོ། །བྱིངས་ཤི་དང་གཞན་ཡང་བྱིངས་ཨཙ་གཅིག་པ་རྣམས་དང་བྱིངས་ཨུཀ་མཐའ་་་ཅན་ལས་ཕ་རོལ་གྱི་ཀྲྀཏ་སོགས་ཀྱི་རྐྱེན་ཀ་འགྲོ་བ་རྣམས་ལ་ཨིཊ་མི་འགྱུར་ཏེ། ཤི་ཏ། གུས་པ་བྱས། ཤྲི་ཏ་བཱན། གུས་པ་བྱེད། བྷཱུ་ཏ། བྱུང༌། བྷཱུ་ཏ་བཱན། བྱུང༌། འགྲེལ་པར་ནིཥྛ་ཕ་རོལ་ལ་ཨུརྔུའི་ཐུང་ངུར་འགྱུར་རོ་ཞེས་བྱུང་བས། རྒྱུཨུཎྜ་ཏ༑བཀབ། ར་དཱ་བྷྱ་ནིཥྛ་:ཏ་ནཔཱུརྦ་སྱ་ཙ་ང། ར་ད་དག་ལས་ནིཥྛའི་ཏ་ན་སྟེ་སྔ་མའི་དའི་འང་ངོ༌། །བྱིངས་མཐའ་ར་ཡིག་དང་ད་དག་ལས་ཕ་རོལ་གྱི་ནིཥྛའི་ཏ་ནར་འགྱུར་ཞིང་དེའི་སྔ་མའི་དའང་ནར་འགྱུར་ཏེ། ཤྲཱི་ལཀྟ་སྦྱར་བྱིངས་ཀྱི་རཱིཏ་ཀྱི་ཨིཏ་འོས། ཛཱ་ཨིཏ། རིའི་ཨཎ་ལ་ར་ཕ་རོལ་ལོས་དྲ། ཧལ་སོགས་ཀྱིས་་་རིང་པོ་མདོ་འདིས་ཏ་ནར་བསྒྱུར་བས། །ཤཱཎ། འཚེ་བར་བྱས། བྷི་དིར་ལ་ལྟ་མདོ་འདིས་ཏ་དང་ད་གཉིས་ཀ་ནར་བསྒྱུར་བས། ༼བྷིནྡ། རྣམ་འཇོམས། ས་ཡོ་གཱ་དེ་རཱ་།
