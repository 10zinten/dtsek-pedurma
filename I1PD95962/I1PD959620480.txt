རཾ་ཥ་བྷ་ཏ་ར། ཁྱུ་མཆོག་གི་ལུས། རཱི། མ་དཱིཏ། ཀིམ་ཡཏྟ་དོ་ནིནྡྷཱར་ཎེ་དྭ་ཡོ་རེ་ཀ་སུ་་་ཌ་ཏ་ར་ཙ། ཀིམ་ཡཏ་ཏད་ལས་གཉིས་ལས་གཅིག་ངེས་པར་འཛིན་པ་ལ་ཌ་ཏ་་ར་ཙའོ། །ཀིམ་ཡད་ཏེད་རྣམས་ལས་གཉིས་ཀྱི་ནང་ནས་གཅིག་ངེས་པར་འཛིན་པའི་དོན་ལ་ཌ་ཏ་ར་ཙ་འགྱུར་ཏེ། ཀིམ་ལ་ཌ་ཏ་ར་ཙ་ཙ་དབྱངས་དོན་ཌི་ཏ་ལ་ཊི་ཕྱིས་ཌ་དེའི་དོན། ཨ་ན་ཡོ་ཀ་ཏ་རོ་པ་ཊུ། འདི་དག་ལས་མཁས་པ་སུ། »ཡད་ལ་ཌ་ཏ་རཙ། ཨ་ན་ཡོ་ཡ་ཏ་རོ་པ་་་ཊུ༑འདི་དག་ལས་མཁས་པ་གང་ཡིན། ཨ་ན་ཡོ་ཏ་ཏ་རོ་པ་ཊུ་རཱ་གཙྪ་ཏུ། འདི་དག་ལས་མཁས་པ་དེ་ཚུར་ཤོག། བཱ་བ་ཧཱུ་ནཱི་ཛཱ་ཏི་པ་རི་པྲ་བྷེ་ཌ་ཏ་མཙ། མང་པོ་རྣམས་ཀྱི་རིགས་འདྲི་བ་ལ་ཌ་ཏ་མཙ་ཡང་ནའོ། །མང་པོ་རྣམས་ཀྱི་ནང་ནས་གཅིག་ངེས་པར་བཀར་ཏེ་རིགས་འདྲི་བའི་དོན་ལ་ཌ་ཏ་མཙ་ཡང་ན་འགྱུར་ཏེ། །འདི་རྣམས་ཀྱི་ནང་ནས་བྲམ་ཟེ་སུ། ཨེ་ཊཱ་ཀ་ཏ་མོ་གྲྭ་ཧྥ་ཎ། ཀིམ་ལ་ཌ་ཏ་མཙ། འདི་རྣམས་ཀྱི་ནང་ནས་བྲམ་ཟེ་གང་། …ཨེ་ཊཱཡ་ཏ་མོ་བྲཱཧྥ་ཎ། ཡད་ལ་ཌ་ཏ་མཙ། །འདི་རྣམས་ཀྱི་ནང་ནས་བྲམ་ཟེ་དེ། .ཨེ་ཊཱ་ཏ་ཏ་མོ་བཱཧྥ་ཎ། ཏཏ་ལ་ཌ་ཏ་མཙ། ཡང་ན་ཨ་ཀཙ་སྦྱར་བས། འདི་རྣམས་ཀྱི་ནང་ནས་བྲམ་ཟེ་སུ། ཨེ་ཊཱ་ཀ་ཀཿབཱ་ཧྥ་ཎ༑ཡཀལ་ཨ་ཀཙ་ཊིའི་གོང་དུ། ཡལ་ཊིའི་གོང་དུ་ཨ་ཀཙ། ཨེ་ཊཱ་ཡ་ཀཿབཱ་ཧྥ་ཎེ། འདི་རྣམས་ཀྱི་ནང་། ནས་བྲམ་ཟེ་གང་། སལ་ཊིའི་སྔོན་དུ་ཨ་ཀཙ། …ཨེ་ཊཱ་ས་ཀཿབཱ་ཧྥ་ཎ། འདི་རྣམས་ཀྱི་ནང་ནས་བྲམ་ཟེ་དེ། ཨེ་ཀཱཙཱ་པཱ་ཙཱ། ཨེ་ཀ་ལས་ཀྱང་ཤར་པ་རྣམས་ཀྱིའོ། །བཤད་མ་ཐག་པའི་དོན་དེ་དག་ལ་ཨེ་ཀ་ལས་ཀྱང་ཌ་ཏ་རཙ་དང་ཌ་ཏ་མཙ་དག་འགྱུར་ཏེ། སྔོན་གྱི་ཤར་ཕྱོགས་ཀྱི་སློབ་དཔོན་རྣམས་ཀྱི་འདོད་པས་སོ། །།འདི་དག་ལས་གཅིག་བྲམ་ཟེ། ཨེ་ཏ་ཡོ་རེ་ཀ་ཏ་རོ་བི་པྲ། ཨེ་ཀ་ཌ་ཏ་རཙ། འདི་རྣམས་ཀྱི་ནང་ནས་གཅིག་བྲམ་ཟེ། ཨེ་ཥ་མེ་ཀ་ཏ་མོ་བི་༈ཨེ་ཀ་ལ་ཌ་ཏ་རཙ། ཨེ་ཀ་ཏ་ལ་ཊིའི་གོང་དུ་ཨ་ཀཙ། …ཨེ་ཥཱ་མེ་ཀ་ཀཿབི་པ། འདི་རྣམས་ཀྱི་ནང་ནས་གཅིག་བྲམ་ཟེ། འགྲེལ་པར་ལུང་དྲངས་པ། ཨ་ནྱ་ཤཔྟ་ཏ་ཌ་ཏ་མཙྪ། ་ཨ་ནྱ་གྱི་སྒྲ་ལས་ཌ་ཏ་མཙ་འང་ངོ༌། །ཨ་ནྱ་གྱི་སྒྲ་ལས་ཀྱང་ཌ་ཏ་མཙ་འགྱུར་ཏེ། ་ཨ་ནྱ་ལ་ཌ་ཏ་—།
