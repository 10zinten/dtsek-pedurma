བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བེ་ད། བི་བི་ད་ཏུ། བི་བི་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བེ་དི་ཐ། བི་བི་ད་ཐུ། བི་བི་ད། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བི་བེ་ད། བི་བི་དི་བ། བི་བི་དི་མ། ལུཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། བེ་དི་ཏཱ། བེ་དི་ཏཱ་རོ། བེ་དི་ཏཱ་ར། །བེ་དི་ཏཱ་སི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། །བི་དྷཱཏ། བི་དྷཱནྟཾ། བི་དྷུ་སུསོགས་སོ། །།ལྀཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། བེ་དི་ཉྩ་ཏི་སོགས་དང༌། དེ་བཞིན་དུ། ཕྲོང་ལ། ཨ་བེ་དི་ཏ་སོགས་སོ། །ས་ཁ། ༈ཨས་ཡོད་པ་ལའོ་ཞེས་པ་ལའང་པ་རསྨི་པ་ད་རྣམས་སྦྱོར་ཏེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨསྟི། །བྷ་སོ་ར་ཀློ་པ། ལྷ་ཨས་དག་གི་་་ཨཏ་དབྱིའོ། །རྐྱེན་བྷ་དང་ཨས་ཀྱི་ཨ་ཡིག་བྱིངས་ཐམས་ཅད་པ་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་དབྱི་སྟེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ་སྟསནྟི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སི། ཏཱས་ཨསྟི་དག་གི་ས་དབྱི་འོས་ས་་་ཕྱིས། སྦྲ། སྦྲ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨསྨི། སྭཨས་ལ་ལཊ་ཤཔ་འཇིག། ལ་བས་དང་མས་སུ་བསྒྱུར། བྷ་ཨས་དག་གི་ཨ་དབྱི་འོས། ཨ་ཕྱིས། ས་དྲངས། སྨ། ལ་ལིང་ལ་བྷྱ་ཨས་དག་གི་ཨཏ་དབྱི་འོས་ཨ་ཕྱིས་སྲ་ཡཱ་ལ་དྲངས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། སྙཏེ། །སཱུ་ཏྭཾ། །སྩ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སྣུ། སཱུ་ཏཾ། སཱུ་ཏི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཧཱུྃ། ས་ག་སྣ་མ། དེ་བཞིན་དུ་ལོཊ་ལའང་ཀ་ང་་་འགྲོ་བ་རྣམས་ལ་ཨ་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨསྟུ༑༑ཏ། ཙཱ། སནྟུ། གྷ་སོ་རེདྡྷཱ་བ་བྷཱ་ས་ལོ་པཉྩ། གླུ་ཨས་དག་གི་ཧི་ལ་ཨེ་སྟེ་ཨ་སྒྱུ་ས་དབྱི་བའང་ངོ༌། །བྱིངས་གླུའི་མིང་ཅན་རྣམས་དང་ཨས་ཀྱི་་ལོཊ་བསྒྱུར་བའི་ཧི་ཕ་རོལ་ལ་ཨེ་རུ་བསྒྱུར་ཞིང་གླུ་སོགས་ཀྱིས་ཨ་བྷྱཱ་ས་འང་དབྱི་སྟེ། —།
