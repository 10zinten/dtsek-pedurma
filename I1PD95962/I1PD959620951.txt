ཁ་བ་ཏི་བ་ཀ་དམ་བ་ཀྱི་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིན་ཉི་ཤུ་ནས་ཁོ་པའི་ས་ཁྱོལ་ཎགས་པ་ཀ་་སོགས་དང༌། ་ལུང་ལ། །ཨིར་ཡང་ན་བསྒྱུར་བས། ཨ་པི་པྲ་བཏ། སོགས་སམ། ཨ་པུ་པྲ་བཏ། སོགས་སོ། །ལིཊ་སོགས་ལྷག་མ་ྋརྣམས་ལ། །ལིཊཔཱ་བ་ཡཱ་ཙ་ཀཱ་ར། ལུཊ་པྲ་བ་ཡི་ཏཱ། སྨོན་ལིང་པྲ་བྱཱཏ། ་ཁྲིཊ་པྲ་བ་ཡི་ཥྛ་ཏི། ཕྲེང་ཨ་པྲ་བ་ཡི་ཏ། །སོགས་སོ། །ཁྱདྨ་འོད་ཀྱིས་་་་་་་གདན་ས་༈རླུ་འགྲོ་བ་ལ་ཎིཙ་སྦྱར་འཕེལ་བ་བྱས་པས། པཱ་བི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། །ལཊབླུ་བ་ཡ་ཏི། ལིང་བླུ་བ་ཡེཏ། ལོཊ་བླུ་བ་ཡ་ཏུ། ལང་ཨ་པ་བ་ཡཏ། སོགས་དང་། ལུང་ལ། ཨིར་ཡང་ན་བསྒྱུར་བས། ཨ་པི་བླ་བཏ། སོགས་སམ། ཨ་པུ་བླ་བཏ། །སོགས་སོ། །།ལིཊ་སོགས་ལྷག་མ་རྣམས་ལ། ལིཊ་བླ་བ་ཡཾ་ཙ་ཀཱ་ར། ་ལུཊབླ་བ་ཡི་ཏཱ། སྨོན་ལིང་སླུ་བྱཱཏ། ཝིཊ'【་བ་ཡི་ཥྛ་ཏི། ་ཕྲེངཨ་པ་བ་ཡི་ཏ། སོགས་སོ། བད། །ཤ་ཞི་བ་g༈བྱུང་འགྲོ་བ་ལའོ་ཞེས་པ་ལ་ཎིཙ་སྦྱར་འཕེལ་བ་བྱས་པས། ་་བྷ་བི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་བྱ་བ་ཡ་ཏི། ལིང་—བྱཱ་བ་ཡེཏ། ལཊ་བྱ་བ་ཡ་ཏུ། ལང་ཨ་བྱཱ་བ་ཡཏ། ལུང་ལ། ཨིར་ཡང་ན་བར་བྱས་པས། ཝཱ་ཨ་བི་བྱ་བཏ། སོགས་སམ། །ཨ་བུ་བྱ་བཏ། །སོགས་སོ། །ཁྲིཊ་སོགས་ལྷག་མ་རྣམས་ལ། ལིཊབྱཱ་བ་ཡཾ་ཙ་ཀཱ་ར། ལུཊབྱཱ་བ་:ཡི་ཏི། སྨོན་ལིང་བྱཱ་བྱཱཏ། ཝིཊ'བྱཱ་བ་ཡི་བྱ་ཏི། །ཕྲེང་…ཨ་བྱཱ་བ་ཡི་ཏ། སོགས་སོ། །ཌ་། ངའི་ཏ། པ་༈གུ་ཧྥུ་ཀུན་ནས་སྒྲིབ་པ་ལའོ་ཞེས་པ་ལ་ཎིཙ་སྦྱར། གོ་ཧའི་ཨུ་པ་དྷཱའི་ཨུཏ་འོས་རིང་པོར་བསྒྱུར་བས། །ཧཱུ་ཧི་ཞེས་གྲུབ་པ་ལ་ལཊ་ལ་སོགས་པ་རྣམས་་་སྦྱར་བས། ལཊ་ཧཱུཾ་ཧ་ཡ་ཏི། ལིང་ཧཱུཾ་ཧ་ཡེཏ། ལོཊ་གཱ་ཧ་ཡ་ཏུ། ལང་ཨ་ཧཱུཾ་ཧ་ཡཏ། །ལུངཨ་ཛྙཱ་གུ་ཧཏ། ལིཊ་ཧཱུཾ་ཧ་ཡཾ་ཙ་ཀཱ་ར། ༈ལུཊ་གཱ་ཧ་ཡི་ཏཱ། སྨོན་པའི་ལིང་གུ་ཧཱཏ། ཁྲིཊཧཱུཾ་ཧ་ཡི་ཉྩ་ཏི། ཁྲིང་ཨགུ་ཧ་ཡི་ཏ། སོགས་སོ། །།།།
