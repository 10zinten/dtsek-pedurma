སྒྲ་མདོ། ''འིཞི་ནི་ལྟ་བ་ཁགོ་ཁཚོ་ད་ཀ་ཀ་ལ་སྒོ་ཀའི་ཨཱཏྨ་ནེ་རྣམས་འཇུག་སྟེ། བྱིངས་ཐམས་ཅད་པ་བཞི་པོ་ལ། བྷཱ་ཤ། བྷཱ་ཤ། །བྷ་མུ་སོགས་ཀྱི་མདོས། བྱེད་པ་པོའི་དོན་ལ་ཤཔ་དང་ན་ཡང་ན་བར་བྱས་པས། ལཊ་ལ། བྷཱ་ཤ་ཏེ་སོགས་སམ། ཡང་ན། བྷ་བྱ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ་ལིང་ལ། བྷཱ་ཤེ་ཏ་སོགས་སམ། བྷཱ་བྱེ་ཏི་སོགས་སོ། །ལོཊ་ལ། ''བྷཱ་ཤ་ཏྭཾ། སོགས་སམ། བྷཱ་བྱ་ཏཱ་སོགས་སོ། །ལང་ལ། ཨ་བྷཱ་ཤ་ཏ་སོགས་སམ། ཨ་་བྷཱ་ཤུ་ཏ་སོགས་སོ། །ལུང་ལ་ཝལ་དང་པོའི་ཨིཊ་སྦྱིན་པས། །ཨ་བྷཱ་ཤིཏྠ་སོགས་སོ༑། ལིཊ་ལ་ཕཎ་སོགས་ཀྱི་མདོས་ཨ་བྷཱ་ས་དབྱི་བ་སོགས་ཡང་ན་བར་བྱས་པས། བྷེ་ཤེ། བ་གྲྭ་ཤེ། བྷེ་ཤཱ་ཏེ། བ་བྷཱ་ཤཱ་ཏེ། བྷེ་ཤི་རེ། བ་བྷཱ་ཤི་རེ་སོགས་སོ༑། ལུཊ་ལ་ཨིཊ་བྱིན་པས། བྷ་ཤིཏཱ་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། །བྷཱ་ཤི་ཏིཥྛ་སོགས་སོ། །ལྀཊ་ལ། ་བྷཱ་ཤི་ཥྛ་ཏེ་སོགས་སོ། །ཁྲིང་ལ། ཨ་བྷཱ་ཤི་་་་ཏ་སོགས་སོ། །དེ་བཞིན་དུ། བྷཱ་ཤི་འབར་བ་ཡང་ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་འདྲའོ། །།ཀྱི་ས་འགྲོདོཔ། སྐྱེས་པ་དེ་ཚོ་བ་དྲན་་་་:༈སཱ་ན་སྒྲ་ལའོ་ཞེས་པ་བྱིངས་པ་རསྨི་པ་ད་ཅན་ཡིན་པས་ལཊ་སོགས་པ་་རསྨི་པ་ད་རྣམས་སྦྱར་བས། ལཊ་ལ། སཱ་ན་ཏི་སོགས་སོ། །ལིང་ལ། ཞིསྭ་་ནེ་ཏ་སོགས་སོ། །ལོཊ་ལ། སཱ་ན་ཏུ་སོགས་སོ། །ལང་ལ། ཨ་སཱ་ནཏ་སོགས་སོ། །ལུང་ལ་འཕེལ་བ་ཡང་ན་བར་བྱས་པས། །ཨ་སཱ་ནཱིཏ། ཨ་སཱ་ནིཏྟཾ། ཨ་སཱ་ནི་སོགས་སམ། ཨ་སཱ་ནཱིཏ་སོགས་སོ། །ལོཊ་ལ། ཕ་ཎ་སོགས་ཀྱི་ཨ་བྷཱ་ས་དབྱི་བ་སོགས་ཡང་ན་བར་བྱས་པས། ས་སཱ་ན། སྭ་ན་ཏུ། །ས་སཱ་ན་ཏུ༑སེ་ནུ། ས་སཱ་ནུསོགས་སོ། །ལུཊ་ལ། ཝལ་དང་པོའི་ཨིཊ་སྦྱར་་་བས༑སཱ་ནི་ཏྭཾ་སོགས་དང༌། སྨོན་པའི་ལིང་ལ། ཨོ་སུ་ནྟིཏ་སོགས་སོ། །ཁྲིཊ་ལ། སཱ་ནི་''ཏི་སོགས་སོ། །ཚོུང་ལ། ཨ་སཱ་ནི་ཏ་སོགས་སོ། །གོང་།
