།ཆགསད་ངོ༌། ཝཾ་། ::དྲྀ་ཥཱ་ཀ་པིཨགི་ཀུ་སི་ཏ:ཞི་ཀུ་སཱི་ད་རྣམས་ཀྱི་ཨུ་དཱཏྟའོ། །མ་ནུ་ལས་ཨོ་ཡང་ནའོ། །དབྱངས་ཨ་ནུ་དཱཏྟ་ལས་ཨུ་པ་དྷཱའི་ཏ་ནའོ། །སཱུ:ཏི：ལ་ཨ་གy:མཉྩ་རྣམས་ཀྱི་ཨུ་པ་དྷཱའི་ཡའོ། ་། ཛཱ་ན་པ་ད：ཀུཎྜང་གོ་ཎ:སྦྲ་ལབྷཱ་ཛནཱ་ག་ཀཱ་ལནི་་་་ལ་ཀུ་ཤ:་ཀཱ་མུ་ཀི་ཀ་བ་ར་རྣམས་ལས། རྒྱུ་སྣོད༔གླང་་ཁལ་མ་བཅོས་པཀློ་ལོ་མ:ཏུ་སྦོམ་པམདོགཁེབས་མིན་་་་པགཁྱེམ：ང༌། འཁྲིག་འདོད:སྐུ་ཆས་རྣམས་ལའོ༄༅། །ཨུཏ་མཐའ་ལས་ཡོན་ཏན་བརྗོད་པ་ལས་སོ། །བ་ཧུ་སོགས་རྣམས་ལས་ཡང་ནའོ། །།ཨིཏ་ལས་་སྲོག་ཆགས་ཀྱི་ལུས་ལས་སོ། །ཀྲྀཏྲ་ཨི་ཡིག་ལས་ཀྟིན་མ་ཡིན་པ་ལས་སོ། །སྐྱེས་པའི་མིང་དང་ལྡན་པ་ལའོ། ལ། སཱུརླ་ལས་ལྷ་ལ་ཙཱཔའོ། ་། ཨིནྡྲ；བ་རུ་ཎཿབྷ་བ；ཤརྦརུ་དྲ:མྲྀ་ཌ::ཧི་མ;ཨ་ར་ཎྱ::：ཡ་བ:ཅེཡ་བ་ནམཱ་ཏུ་ལ(ཚུཨཱ་ཙཱརྨ་རྣམས་ཀྱི་ཨཱ་ནུཀའོ། །གཽ་ཏ་ལས་བྱེད་པ་སྔ་མ་ལས་སོ། པ། ཀྟ་ལས་ཆུང་ངུ་བརྗོད་པ་ལའོ། །འབྲུ་མང་པོའི་མཐའ་ལའང་ཨུ་དཱཏྟ་ལས་སོ། །བ་ཧུ་ནཉ:་སུ་ཀཱ་ལ་སུ་ཁ་སོགས་སྔ་མ་ལས་ཀྱང་མ་ཡིན་ནོ། །རང་ལུས་མ་ཡིན་པའི་ཚིག་སྔ་མ་ལས་ཡང་ནའོ། །རང་གི་ཡན་ལག་ལའང་ཉེར་བསྒྱུར་ལས་ཨུ་པ་དྷཱ་ཡང་དག་པར་སྦྱོར་བ་མེད་པ་ལས་སོ། །ནཱ་སི་ཀ། ཨུ་ད་ར་ཨོཥྛ:། ཛཾ་གྷདནྟ;ཀརྞ་ཤིངྒ་ལས་འང་ངོ༌། །ཉེ་བར་འཇལ་བ་ལས་བག་པུཙྪ་དག་ལས་སོ། །ཀྲོ་ཌ་སོགས་དང་བ་ཧཱུྃཙ་་ལས་མ་ཡིན་ནོ། །ས་ཧ་ནཉ:རིང་མཱ་ན་སྔ་མ་ལས་ཀྱང་ངོ༌། །བཱཧ་ལས་སོ། །ས་ཁི་ཨ་ཤི་ཤྲི་བརྗོད་པ་ལའོ། །ཡ་ཨུ་པ་དྷཱ་མ་ཡིན་པའི་རིགས་ལས་བུད་མེད་ཀྱི་ཡུལ་མ་ཡིན་པ་ལས་སོ། །པཱ་ཀ། ཀརྞ::པཆཔུ：—།
