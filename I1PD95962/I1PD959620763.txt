སྒྲ་མདོ། །ཁ་དེའི་བན་ཆོས་ཀྱི་ཆོ་ག་ལ་བ་དེར་བློ་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས་དྲངས་པས། བེ་བི་ཛཱི་ཏ། བེ་བི་ཛཱི་ཏཾ། བེ་བི་ཛཱི་རན། སོགས་སོ། ་། ལོཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བེཀྱུ། བེ་བིཏྟཱཏ། བེ་བིཏྟཾ། བེ་བི་ཛ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བིབྷི། བེ་བིགྷཱཏ། བེ་བིཏྟཾ། བེ་བིཀྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཛཱ་ནི། བྱིངས་ཐམས་ཅད་པ་ཨཙ་པིཏ་ལ་ཨ་བྷུལྟའི་མ་ཡིན་ནོས་ཡོན་ཏན་མི་བྱེད། བེ་བི་ཛཱ་བ། ་བེ་བི་ཛཱ་མ། བྱིངས་ཐམས་ཅད་པ་ཨཙ་པིཏ་ལ་ཨ་བྱཱསྟའི་མ་ཡིན་ནོས་ཡོན་ཏན་མི་བྱེད། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བིཏྟཾ། བེ་བི་ཛཱ་ཏཱ། བེ་བི་ཛ་ཧཱུྃ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བིཥྛ། བེ་བི་་་་ཛཱཐཱ༑བེ་བིབྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཛི། བེ་བི་ཛཱ་བ་ཧི། བེ་བི་ཛཱ་མ་ཧི། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བེ་བེག་་་། ཨ་བེ་བེཀ། ཨ་བེ་བིཏྟཾ། ཨ་བེ་བི་ཛུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ༼ཨ་བ་་་བེག། ཨ་བེ་བེཀ། ཨ་བེ་བིཏྟཾ། ཨ་བེ་བིཀྟ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བེ་བི་ཛཾ། ཨ་བེ་བི་ཛ། ཨ་བེ་བི་ཛྙཱ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བེ་བིཀྟ། ཨ་བེ་བི་ཛཱ་ཏཱ། ཨ་བེ་བི་ཛ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བེ་བིཀྐུ། །ཨ་བེ་བི་ཛ་བྷཾ། ཨ་:བེ་བིཏྟཾ༄༅། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བེ་བི་ཛི། ཨ་བེ་བི་ཛཱ་ཧི། ཨ་བེ་བི་ཛྙཱ་ཧི། ལུང་གི་ཨ་ཏང་ལ། ཨིར་འགྲོ་བ་ལ་ཡང་ན་འོས། ཙྪི་ཨང་དུ་བསྒྱུར་བ་དང་སི་ཙར་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ཨང་གི་ཕྱོགས་ལ། ་ཨ་བི་ཛཏ། ཨ་བི་ཛ་ཧཱུྃ། ཨ་བི་ཛན། སོགས་སོ། །།སིཙའི་ཕྱོགས་ལ། བད་བྲཛ་སོགས་ཀྱིས་འཕེལ་བ། ཛྷལ་དང་པོ་རྣམས་ལ། ཛྷལ་ལས་ཛྷལ་ལ་འོས་སིཙ་དབྱི་བ་དང༌། ཙུའི་ཀུ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བི་ཀཱིཏ། ཨ་བི་ཏྟཾ། ་་:ཨ་་བི་། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་:བི་ཥཱི༄༅། །ཨ་བི་ཏྟཾ། ཨ་བི་ཥ། །
