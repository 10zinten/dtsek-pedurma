ནས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི༽ཁ་ཚོཎི་ནི་ལ་ཁོ་བོའི་ཆོས་ཁ་བར་བློ་གཏད་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་ཏེ། ཧཱ་ཡི་ཙང་ཎི་ལ་ཙང་ཨུ་པ་དྷཱའི་ཐུང་ངུ་འོས་ཐུང་ངུར་བསྒྱུར༈མདོ་འདིས་སཾ་པྲ་སཱ་ར་ཎ་གཉིས་བརྗོད་ཡར་བའི་རིང་པོ། །འགྲེལ་པར་སཾ་པྲ་སཱ་ར་ཎའི་ཕ་རོལ་ལ་ཡུཀ་ཨཱ་ག་མ་དབྱིའོ་ཞེས་བཤད་པས་ཡུཀ་ཕྱིས། ཨཙའི་ཊྛིཏ་ལ་འོས་འཕེལ་བ། ཨེཙའི་ཨཡ྄་སོགས་ཀྱིས་ཨོ་ཨཱབ། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས། ཀ་ཎ་ར་ཎ་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཐུང་ངུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ཨ་ཛུ་ཧཱ་བཏ། །སོགས་སམ། ཨ་ཛྙཱ་ཧ་་བཏ༑སོགས་སོ། །ལིཊ་སོགས་ལྷག་མ་རྣམས་ལ། ལིཊ་ཧཱ་ཡ་ཡཾ་ཙ་ཀཱ་ར། ལུཊ'ཧཱ་ཡ་ཡི་ཏཱ། སྨོན་ལིང་ཧཱ་ཡཱཏ། ཁྲིཊ་…ཧཱ་ཡ་ཡི་ཏ། ་ཁྲིང་ཨ་ཧཱ་ཡ་ཡི་་་ཊྛིཏ༑སོགས་སོ། །ཀ་ཁ་ག་ང་། །པ་བ་:་མ་ཏེ་། །:༈བྱེཉ་ཀུན་ནས་སྒྲིབ་པ་ལ། ཎིཙ་ཨེཙའི་ཨཱཏེ། ཡུཀ་ཨཱ་ག་མ་བྱིན་་་་་པས། བྱ་ཡི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊབྱཱ་ཡ་ཡ་ཏི། ལིང་བྱཱ་ཡ་ཡེཏ། ལོཊབྱཱ་ཡ་ཡ་ཏུ། ལང་ཨ་བྱཱ་ཡ་ཡཏ། ་ལུངཨ་བ་བྱཱ་ཡཏ། ལིཊ'བྱཱ་ཡ་ཡཾ་ཙ་ཀཱ་ར། ལུཊབྱཱ་ཡ་ཡི་ཏཱ། སྨོན་ལིང་བྱཱ་ཡཱཏ། ཁྲིཊ་བྱཱ་ཡ་ཡི་་ཥྱ་ཏི། ་ཕྲེང་ཨ་བྱཱ་ཡ་ཡི་ཥྱ་ཏི། སོགས་སོ། །།/ཉི་ཁ་༈བེཉ་ཐགས་ཀྱི་རྒྱུ་ལའོ་ཞེས་པ་ལ། ::ཎིཙ་ཨེ་ཙ་ཨཱཏ་ཡུཀ་ཨཱ་གི་མ་བྱིན་པས། བཱ་ཡི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་བཱ་ཡ་ཡ་ཏི། ལིང་བཱ་ཡ་ཡེཏ། ལོཊབཱ་ཡ་ཡ་ཏུ། ་ལང་…ཨ་བཱ་ཡ་ཡཏ། ལུང་…ཨ་བ་བ་ཡཏི། ལིཊབཱ་ཡ་ཡཱ་ཙ་ཀཱ་ར། ལུཊ'བཱ་ཡ་ཡི་ཏཱ། སྨོན་ལིང་བཱ་ཡཱཏ། ་ཁྲིཊབཱ་ཡ་ཡི་་བྱེ་ཏི། ལོང་ཨ་བཱ་ཡ་ཡི་ཏ། །སོགས་སོ། །རྒྱལ་པོའི་ཤུལ་༈དེ་བཞིན་དུ། པཱ་བཏུང་བ་ལ་ཎི་ཙ་སྦྱར་ཡུཀ་ཨཱ་ག་མ་བྱིན་པས། པཱ་ཡི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་པཱ་ཡ་ཡ་ཏི། ་ལིང་…པཱ་ཡ་ཡེཏ། །—།
