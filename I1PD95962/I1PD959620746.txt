ཨ་ཀྱཱ་བི་ཥ། སོགས་སོ། །ལིཊ་ལ་ཎལ་གཉིས་དང་ཐལ་མ་གཏོགས་སཱཉྫ་སོགས་ཀྱིས་ཀིཏ་ཡང་ན་བར་བྱེད་དེ། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙུགཱ་བ། ཙུཀྟ་བ་ཏུ། ཙུགླུ་བ་ཏུ། ཙུཀྟ་བུ། ཙུཀླུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ༑ཙུཀྵ་བི་ཐ། ཙུཀྐོ་ཐ། ཙུཀྟ་བ་ཐུ། ཙུཀླུ་བ་ཐུ། ཙུཀྟ་བ། ཙུཀླུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙུཀྵ་བ། ཙུཀྟ་བ། ཙུཀྟ་བི་བ། ཙུཀླུ་བི་བ། ཙུཀྟ་བི་མ། ཙུཀླུ་བི་མ། ལུཊ་ལ་བལ་དང་པོའི་ཨིཊ་དང་ཡོན་ཏན་བྱས་པས། ཀླུ་བི་ཏཱ། གླུ་བི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་པས། སྐྱུ་ཡཱཏ། ཁྱུ་ཡཱ་ཏཱཾ། ཀྵ་ཡཱ་སུ། སོགས་སོ། །ཁྲིཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། གླུ་བི་ཁྱཱ་ཏི། སོགས་སོ། །དེ་་་བཞིན་དུ། ཕྲེང་ལ། ཨ་ཀྱ་བི་ཏ། སོགས་སོ། །༈རུ་སྒྲ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། ལཊའི་སྐྱེས་བུ་་་དང་པོའི་གཅིག་ཚིག་ལ། ནུ་རུ་སྐུ་སོགས་ཀྱིས་ཨིཊ་སྦྱོར་བ་དང་འཕེལ་བ་ཡང་ན་བར་བྱས་པས། ར་བཱི་ཏི། རོ་ཏི། གཉིས་ཚིག་དང་མང་ཚིག་ལ། རུ་ཏ། རུ་བནྟི། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་བཱི་ཏི། རོ་ཏི། རུ་ཐ། རུ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ར་བཱི་མི། རོ་མི། རུ་བ། རུ་མ༑ལིང་ལ། རུ་ཡཱཏ། རུ་ཡཱ་ཏྭཾ། རུ་ཡུ། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ར་བཱི་ཏུ། རོ་ཏུ། རུ་ཏཱཏ། རུ་ཏྭཾ། རུ་བནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། རུ་ཧི། རུ་ཏཱཏ། རུ་ཏཾ། རུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ་ར་བཱ་ཎི། ར་བཱ་བ། ར་བཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཨ་ར་བཱི་ཏ། ཨ་རོ་ཏ། ཨ་རུ་ཏཱ། ཨ་རུ་བན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ར་བཱི། ་ཨ་རོ། ཨ་རུ་ཏཾ། ཨ་རུ་ཏ། སྐྱེས་བུ་—།
