ག་ག་ན་སྐྱེ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་ཡུན་བློ་སྐྱ་བོ་ཁའི་མཚོ་ཁའི་མཚོ་ཀ་དེ་Cལཊའི་ཨ་ཏང་ལ། ཧྭ་ཡ་ཏི་སོགས་དང༌། ཏང་ལ། ཧཱ་ཡ་ཏེ་སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། ཧཱ་ཡཏ་སོགས་དང༌། ཏང་ལ། ཧྭ་ཡེཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། ཧཱ་ཡ་ཏུ་སོགས་དང༌། ཏང་ལ། ་ཧཱ་ཡ་ཏྭཾ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ཨ་ཧྭ་ཡཏ་སོགས་དང་། ཏང་ལ། །ཨ་ཧྭ་ཡཏ་སོགས་སོ། །ལི་བི་སི་ཧཱཤྩ། ལི་པི་སི་ཙི་ཧཱ་ལས་ཀྱང་ངོ༌། །ལི་བ་བྱུག་པ། ཎི་ཙ་འཛག་པ། ཧྭ་འགྲན་པ་རྣམས་ལས་ཀྱང་ལྕི་ནི་ཨང་དུ་འགྱུར་ཏེ། མྱི་ཨང་དུ་བསྒྱུར། ཨིཊ་ལའང་ཨཱཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་ཨ་ཏང་ལ། ཨ་ཧཱཏ། ་ཨ་ཧྭ་ཏྭཾ། ཨ་ཧཱན་སོགས་སོ། །ཏང་ལ་ཡང་ན་བར་བྱེད་དེ། ཨཱཏྨ་ནེ་པ་དེ་ཥཱ་ནྱ་ཏ་ར་ཧཱུཾ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ་ཡང་ནའོ། །ལི་པ། ཎི་ཙ། ཧཱ་རྣམས་ལས་ཨཱཏྨ་ནེ་པ་ད་་་རྣམས་ཕ་རོལ་ལ་ལྕི་ཨང་དུ་བསྒྱུར་བ་དང་། །སིཙར་བསྒྱུར་བ་ཡང་ན་བར་བྱ་སྟེ། ལྕི་ཨང་དུ་བསྒྱུར་བའི་ཕྱོགས་ལ་ཨུ་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་་ཧཱཏ༑ྋཨ་ཧྭ་ཏྭཾ། ཨ་ཧཱནྟ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་ཧཱཐ། །ཨ་ཧྲེ་ཐཱ། ཨ་ཧཱཧཱུཾ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ཧེ། ཨ་ཧཱ་བ་ཧི། ཨ་ཧཱ་མ་ཧི༑ཙྪི་སིཙ་རུ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཨེཙ་ཨཱཏ་དུ་བསྒྱུར་བ་ཁོ་ན་སྟེ། ཨཱ་ཧཱ་སྟ། ཨ་ཧཱ་སཱ་ཏྭཾ། །ཨ་ཧཱ་ས་ཏ་སོགས་སོ། །ཨ་བྱཱསྟ་སྱ་ཙ། ཨ་བྱཱསྟའི་འང་ངོ༌། །ལིཊ་ཕ་རོལ་ལ་ཧཱའི་ཨ་ཧྥསྟའི་སཾ་པྲ་སཱ་ར་ཎར་འགྱུར་ཏེ། ཧཱ་ལ་ལིཊ། ཨ་བྷཱུསྟ་སྟེ་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎ་བྱས། ཀིཏ་མ་ཡིན་པ་རྣམས་ལ་འཕེལ་བ་དང་ཡོན་ཏན། ཀིཏ་རྣམས་ལ། ཨཙ་ལ་༔་བྱིངས་སོགས་ཀྱི་ཨུཔ་བྱས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛུ་ཧཱ་བ། །ཛུ་ཧུ་བ་ཏུ། ཛུ་ཧུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛུ་ཧ་བི་ཐ། ཛུ་ཧོ་ཐ། ཛུ་ཧུ་བ་ཐུ། ཛུ་ཧུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཛུ་ཧཱ་བ། ཛུ་ཧ་བ། ཛུ་ཧུ་བི་བ། ཛུ་ཧུ་།
