དུ་བསྒྱུར་བ་སོགས་བྱས་པས་ལཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བྷནྟི། བི་བྷཾ་ཏ། བི་བྷཱུ་ཏི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བྷནྟི། བི་དྷྭཾ་ཐ། བི་བྷཾ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷརྨི། བི་བྷཾ་བ། བི་བྷཾ་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བྷཾ་ཏེ། བི་བྷཱ་ཏེ། བེ་བཱ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཀླ་བི་ཤྲཱི་ཥ། བི་བྷཱ་ཐེ། བི་བྷཾ་དྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བི་བྷེ། བི་བྷཱི་བ་ཧེ། བི་བྷཾ་མ་ཧེ། ལིང་གི་ཨ་ཏང་ལ༑་ལ་བི་བྷཾ་ཡཱཏ། བི་བྷཾ་ཡཱ་ཏྭཾ། །བི་བྷཾ་ཡུ། སོགས་དང༌། ཏང་ལ་མཐའ་མ་ཡིན་པ་སོགས་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས། ཨིཀ་གི་ཨཙལ་ཡཎ་བྱས་པས། བི་བྷི་ཙིཏ། བི་བྷི་ཡཱ་ཏྭཾ། བི་བྷི་རན། སོགས་སོ། །ལོཊའི་ཨ་ཏང་གི་་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །་བི་བྷརྟུ། བི་བྷཾ་ཏཱཏ། བི་བྷཾ་ཏཱཾ། བི་བྷཱ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བྷཾ་ཧི། བི་བྷཾ་ཏཱཏ། བི་བྷཱི་ཏཾ། བི་བྷཱི་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷ་རཱ་ཎི། བི་བྷ་རཱ་བ། བི་བྷ་རཱ་མ། ཏང་གི་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བྷཾ་ཏཱཾ། བི་བྷཱ་ཏྭཾ། བི་བྷཱ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །བི་བྷཾ་ཥཱ། །བི་བྷཱ་ཐཾ། བི་བྷཾ་དྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷ་རེ། བི་བྷ་རཱ་བ་ཧི། བི་བྷ་རཱ་མ་ཧི་ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཨ་བི་བྷ། ་ཨ་བི་བྷཾ་ཏཱཾ། འཨ་བི་བྷ་རུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བི་བྷ། །ཨ་བི་བྷཾ་ཏཾ། ཨ་བི་བྷཾ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བི་བྷ་རཾ། ཨ་བི་བྷཾ་བ། ཨ་བི་ཤྲཱི་མ། ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བི་བྷཾ་ཏ། ཨ་བི་བྷཱ་ཏཱ། ཨ་བི་བྷཱ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བི་ཤྲཱི་ཐཱ། ཨ་བི་བྷཱ་ཐཱ། ཏེ་་ཨ་བི་བྷཾ་དྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བི་བྷཾ། ཨ་བི་ཤྲཱི་བ་ཧི། ཨ་བི་བྷཾ་མ་ཧི། ལུང་གི་ཨ་ཏང་ལ་སིཙ་ལ་འཕེལ་བ་པ་རསྨི་པ་——།
