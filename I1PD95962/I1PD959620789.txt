ཏ་ཁ་ཁ་ཏི་ཀུ་ཊ་ཏ་ཀྱིས་དྷ་དར་བསྒྱུར་བས། །རདྷཱ། །རངྒཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ར་དྷཱཏ། ར་དྷཱཙཱ། ར་དྷསུ། སོགས་སོ། །མྲྀཊ་ལ། 《ར་དྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོ༌། །ཨིཊ་ཡང་ན་བར་བྱས་པས། །ཨིཊའི་ཕྱོགས་ལ། །ར་དྷི་ཥྛ་ཏི། །སོགས་དང་། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། །ཁར་ལ་ཙར་འོས་དྷ་ཏར་པ་བསྒྱུར་བས། རཉྩ་ཏི། སོགས་སོ། །དེ་བཞིན་དུ། །ཀྲྀང་ལ། ཨ་ར་དྷི་་་་ཏ༑སོགས་སམ། ་ཨ་རཉྫཱཏ། །སོགས་སོ། །ཀྐ། ་ཀྱི་ཁང་ཁོ། ཁ་ན་ཆ་རྐྱེན་༈ཀྲྀ་པ་ཚིམ་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ལཊ་ལ། ཀྲྀ་མྱ་ཏི། སོགས་དང་། །ལིང་ལ། ཀྲྀ་ཕྱེཏ། ་་སོགས་སོ། །།།ལོཊ་ལ། ཚིགས་བྱ་ཏུ། །སོགས་དང༌། །ལང་ལ། ཨ་ཀྲྀ་པྱཏ། །སོགས་སོ། །ཎ། ལུང་ལ། པུཥ་ལ་སོགས་པས། རྨི་ཨང་དུ་བསྒྱུར་བ་དང་སིཙར་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས༑ཨང་གི་ཕྱོགས་ལ། ཨ་ཏྲི་པ་ཏཾ། ་ཨ་ཀྲྀ་པ་ཏཱ། ཨ་ཀྲྀ་པ་ན། སོགས་དང༌། 《སིཙའི་ཕྱོགས་ལ། །ར་དྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོས། ཨིཊ་ཡང་ན་བར་བྱ་སྟེ། །ཨིཊའི་ཕྱོགས་ལ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས༑་ཨ་ཏཝཱིཏི། ཨ་ཏ་ཅི། ཨ་ཏམི་ཥཱ། སོགས་དང་། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། ཨ་ནུ་དཱཏྟའི་འང་རིཏ་ཨུ་པ་དྷཱའི་ཨམ་ཡང་ན་འོས། ཨམ་ཡང་ན་བར་བྱེད་དེ། །ཨམ་མ་བྱིན་པའི་ཕྱོགས་ལ། །སིཙ་ལ་འཕེལ་བ་དང་། ཛྷལ་དང་པོ་རྣམས་ལ་སིཙ་དབྱི་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཏཱབྷཱི་ཏ། ཨ་ཏཱནྟཾ། ཌཨ་ཏཱམྦྷ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་་ཨ་ཏཱསྟྲཱི། ཨ་ཏཱནྟཾ། ཨ་ཏཱནྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ཏཱམྲྀ། ཨ་ཏཱ༔། ་་་ཨ་ཏགླུ༈སོགས་སོ། །ཨམ་གྱི་ཕྱོགས་ལ། ཨིཀ་གི་ཨཙ་ལ་ཡ་ཎ་དང༌། དེ་ནས་ཨིཙ་ལ་འཕེལ་བ། ཛྷལ་དང་པོ་རྣམས་ལ་སིཙ་དབྱི་བ་སོགས་བྱས་པས། —.།
