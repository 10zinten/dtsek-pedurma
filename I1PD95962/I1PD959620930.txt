ས་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཐུང་དུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ཨ་པི་པི་ཌ་ཏ། སོགས་སམ། ཨ་པི་པཱི་ཌ་ཏ། སོགས་སོ། །ལ་ཡིག་ཕྱི་མ་རྣམས་ལ། ལིཊ་པཱི་ཌ་ཡཱ་ཙ་ཀཱ་ར། ལུཊ་པཱི་ཌ་ཡི་ཏཱ། སྨོན་ལིང་པཱི་ཙྪཱཏ། ཀྲྀཊ་པཱི་ཌ་ཡི་ཥྛ་ཏི། ཕྲེང་ཨ་པཱི་ཌ་ཡི་ཏ། སོགས་སོ། །ཀླུ་བོན་བ་བ་ཀྱི་དོ:། ཟླ་ན་ཊ། དང་༈་ཀ་ཎ་སྒྲ་ལའོ་ཞེས་པ་ལ་རྒྱུ་དང་བཅས་པ་སོགས་ཀྱིས་ཎིཙ་སྦྱར། །ཨུ་པ་དྷཱའི་འོས་འཕེལ་བ་བྱས་པས། །ཀཱ་ཎི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་ཀཱ་ཎ་ཡ་ཏཱི༄༅། །ལིང་ཀཱ་ཎ་ཡེཏ། ལོཊཀཱ་ཎ་ཡ་ཏུ། །ལང་ཨ་ཀཱ་ཎ་ཡཏ། སོགས་དང་། ལུང་ལ། ་ཀ་ཎ་ར་ཎ་བ་ཎ་བྷ་ཎ་ཐ་ན་ཧེཧྥ་པ་ལུ་པ་ལུ་ཋ་ཧེ་ཊཱ་ཙ། །ཀ་ཎ་ར་ཎ་བ་ཎ་བྷ་ཎ་ཐ་ཎ་ཧྭཉ་ལ་པ་ལུ་པ་ལུ་ཥཱ་ཧཱ་ཨི་ཥ་རྣམས་ཀྱིའང་ངོ༌། །ཀ་ཎ་ར་ཎ་བ་ཎ་བྷ་ཎ་་རྣམས་སྒྲ། ཤོཎ་འཚེད་པ། ཧྭཉ་འདྲེན་པ། །ལ་་་པ་ཡོངས་སུ་བརྗོད་པ། ལུ་པ་སྦེད་པ། ལུ་ཋ་སྲུབ་པ། །ལོ་ཨོ་ཧཱཀ་གཏོང་བ། ཨི་ཏུ་འདོད་པ་རྣམས་ཀྱིའང་ཎི་དང་བཅས་པའི་ཙང་ཕ་རོལ་ལ་ཨུ་པ་དྷཱ་ཐུང་ངུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱ་སྟེ། །ཐུང་ངུར་བསྒྱུར་བའི་ཕྱོགས་ལ་་་། སན་བཞིན་གྱི་བྱ་བ་འཇུག་ཅིང༌། ཥ་ཐུང་དུར་མ་བསྒྱུར་བའི་ཕྱོགས་ལ་སན་བཞིན་གྱི་བྱ་བའང་མི་འཇུག་སྟེ། ཨ་ཙཱི་ཀ་ཎཏ། སོགས་སམ། ཨ་ཙ་ཀཱ་ཎཏ། སོགས་སོ། །ལ་ཡིག་ལྷག་མ་རྣམས་ལ། ལིཊ་ཀཱ་ཎ་ཡཾ་ཙ་ཀཱ་ར། ལུཊཀཱ་ཎ་ཡི་ཏཱ། སྨོན་ལིང་ཀཱ་རྒྱུཏ༑ཕྲིཊ་ཀཱ་ཎ་ཡི་ཏ། ཕྲོང་ཨ་ཀཱ་ཎ་ཡི་ཏ། སོགས་སོ། །ནགས་༈རཎ་སྒྲ་ལའོ་ཞེས་པ་ལ་ཎི་ཙ་སྦྱར། ཨུ་པ་དྷཱའི་འཕེལ་བ་འོས་འཕེལ་་་བ་བྱས་པས། རཱ་ཎི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། །ལཊ་རཱ་ཎ་ཡ་ཏི། ལིང་རཱ་ཎ་ཡེཏ། ལོཊ་རཱ་ཎ་ཡ་ཏུ། ལང་ཨ་རཱ་ཎ་ཡཏ། སོགས་དང་། ལུང་ལ། ཀ་ར་ཎ་སོགས་ཀྱི་མདོས། ཨུ་པ་དྷཱ་ཐུང་དུར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་་་་་།
