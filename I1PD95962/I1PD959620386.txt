ཏྲ༑གྲོང་གི་ཕྲ་མ་མཁན་གྱི་བུ། ཡང་ན། གྲྭ་མ་ཎི་པུ་ཏྲཿ། གྲོང་གི་ཕྲ་མ་མཁན་གྱི་བུ། ཨེ་ཀཱ་ཏདྷི་ཏེ་ཙ། ཨེ་ཀཱའི་དེ་ཕན་ལའང་ངོ༌། །ཨེ་ཀཱ་ཞེས་པའི་མཐའ་ནི་དེ་ཕན་གྱི་རྐྱེན་ཕ་རོལ་ལ་ཐུང་ངུར་འགྱུར་ཏེ། ཨེ་ཀཱ་སྱ་བྷ་བ། གཅིག་གི་དངོས་པོ། བྱ་?བའི་དོན་དེ་ཕན་གྱི་རྐྱེན་ཏ་བྱིན། རྣམ་དབྱེ་ཕྱིས། ཨཱ་ཐུང་ངུར་གྱུར། ་ཎཱ་པ་ཨེ་ཀ་ཏཱ༑དངོས་པོ་གཅིག་མ། །དྨ་པོསཾ་ཛྙཱ་ཚནྡ་སོརྦ་ཧུ་ལཾ། དཱིཔ་ཨཱ་པ་དག་གི་མིང་སྡེབ་སྦྱོར་དག་ལས་མང་པོའོ། །དཱིཔ་ཀྱི་མཐའ་དང་ཨཱ་པ་ཀྱི་མཐའ་དག་མིང་དང་སྡེབ་སྦྱོར་སོགས་ལ་ཐུང་ངུར་མང་པོ་འགྱུར་ཏེ། རེ་བ་ཏྲཿམི་ཏྲ། སྐར་མའི་གཉེན། ཞེས་པ་ལ་རྣམ་དབྱེ་ཕྱིས། །རེ་བ་ཏཱི་ཐུང་ངུར་སོང་རེ་བ་ཏི་མི་ཏྲ། དེ་བཞིན་དུ། ཤ་ལཱ་ཡཱབཧ། ཞེས་པ་ལ། ཤེ་ལ་བ་ཧཾ། རྡོ་ཐོབ་པ། ཨི་ཀེ་ཏཱི་ཀཱ་མཱ་ལཱ་ནཱ་ཙི་ཏ་ཏུ་ལ་བྷཱ་རི་ཥཱ། ཨིཊྚ་ཀཱ་ཨཱི་ཏཱི་ཀཱ་མཱ་ལཱ་རྣམས་ཀྱི་ཙི་ཏ་ཀཱུ་་ལ་བྷཱ་ར་རྣམས་ལའོ། །ཨིཊྚ་ཀཱ་ལ་སོགས་པ་དེ་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་རིམ་བཞིན། ཙི་ཏ་ལ་སོགས་པ་དེ་་་རྣམས་བྱུང་ན། ཨིཊྚ་ཀཱ་ལ་སོགས་པ་དེ་རྣམས་ཀྱི་མཐའ་ཐུང་ངུར་འགྱུར་ཏེ། ཨིཊྚ་ཀཱ་ཡཱཙི་ཏ། སོ་ཕག་གི་རྩིག་པ། ཞེས་པ་ལ། །ཨིཊྚ་ཀ་ཙི་ཏཾ། སོ་ཕག་རྩིག་པ། དེ་་་བཞིན་དུ། ཨཱི་ཏཱི་ཀཱ་ཡཱཏུ་ལཾ། སོ་ཕག་མཉམ་པ་འམ་མཚུངས་པའམ་འདྲ་བ། ཞེས་པ་ལ། ཨཱི་ཥཱི་ཀ་ཏུ་ལཾ། སོ་ཕག་མཉམ་པའམ་མཚུངས་པའམ་འདྲ་བ། མཱ་ལཱ་ཙཱ་སོབྷ་རཱི་ཙ། ཕྲེང་བ་ཡང་འདི་ལྕི་བ་ཡང་ངོ༌། །ཞེས་པ་ལ། མཱ་ལ་བྷཱ་རཱི། ཕྲེང་བ་ལྕི་མོ། ཛོ་ཏིརྫ་ན་པ་ད་རཱ་ཏྲི་ནཱ་བྷི་ནཱ་མ་གོ་ཏྲ་རཱུ་པསྦྲ་ན་བཎ་བ་ཡོ་བ་ཙ་ན། བནྡྷཱ་ཏུ་ཛོ་ཏི་ཥ་ཛ་ན་པ་ད་རཱ་ཏྲི་ནཱ་བྷི་ནཱ་མ་གོ་ཏྲ་རཱུ་པ་སཱན་བཎ་བ་ཡས་བ་ཙ་ན་བནྡྷ་རྣམས་ལའོ། །ངོ་ཏི་ཥ་ལ་སོགས་པ་་་་འདི་རྣམས་ཕ་རོལ་ལས་མ་ནཱ་ཞེས་པའི་སྒྲ་ནི་ས་ཞེས་པར་འགྱུར་ཏེ། ་ས་མཱ་ནཾ་ཛོ་ཏིདྨ་ས་སོ། གང་གི་མཚུངས་པའི་འོད་འདི། ཞེས་པ་ལ་རྣམ་དབྱེ་ཕྲད་སོགས་ཕྱིས། ས་མཱ་ན༑ས་ཞེས་པར་འགྱུར། །ས་ཛྭ་ཏི། མཚུངས་པའི་འོད། དེ་བཞིན་དུ། ས་མཱ་་ས།
