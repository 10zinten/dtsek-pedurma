འཁྱུ་་་འུ་མ། ་་་གཉིས་ཚིག་ལ། བྷ་བི་ཊཱི་ཡཱནྟཾ། བྷུ་ལ་སྨོན་པའི་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ལྔ་ཨཱ་ཧཱུྃ་དུ་བསྒྱུར། ལིང་གི་སཱི་ཡུཊ་འོས་སཱི་ཡུཊ། ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས། ཨིཊ''ཨཱ་ག་མ་ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་ཨེ་ཙ་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཥཱ་ཡ་ཨཱ་ལ་དྲངས། ཏ་ཐ་དག་གི་སུ་ཊ་འོས་ཏྭཾ་དང་ཊའི་བར་དུ་སུཊ་ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་སྲ་ཏཱ་ལ་དྲངས། སྨོན་པའི་ལིང་གི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། །བྷ་བི་ཏི་རན། སྒྱུ་ལ་སྨོན་པའི་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ཁྲ་ཛྷར་བསྒྱུར། ཛྷའི་རན་འོས་ཛྷ་རན་དུ་བསྒྱུར། ལིང་གི་སཱི་ཡུཊ་འོས་སཱི་ཡུཊ་ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན། བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས། བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨཝ་ལྔ་ཨི་ལ་དྲངས། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། བ་ཡ་དག་གི་བལ་ལ་དབྱི་འོས་ཡ་ཕྱིས། སྨོན་པའི་ལིང་གི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྷ་བི་ཊྛ། བྷུ་ལ་སྨོན་པའི་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་ཐཱས་སུ་བསྒྱུར། ཏ་ཐ་དག་གི་སུཊ་ཊ་གནས་དོན་ཨུ་་་བརྗོད་དོན། ལིང་གི་སཱི་ཡུཊ་ཊ་གནས་དོན་ཨུ་བརྗོད་དོན། པ་ཡ་དག་གི་བལ་ལ་དབྱི་འོས་ཡ་ཕྱིས། བྱིངས་ཕྱེད་པའི་་་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ལྔ་ཨི་ལ་དྲངས་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་གཉིས་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས་ཐཱ་ཋཱ་ཥ་ཥཱ་ལ་དྲངས་ར་ས་རུ་རྣམ་བཅད། བྷ་བི་ཊི་ཡཱཤྩ། བྷུ་ལ། སྨོན་པའི་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ཁྲ་ཨཱ་བྷཾ་དུ་བསྒྱུར། ཏ་ཐ་དག་གི་སུ་ཊ་འོས། ཨཱ་དང་ཐཱ་གྱི་བར་དུ་སུཊ། ཊ་གནས་དོན་ཨུ་བརྗོད་དོན། སྲ་ཛྭ་ལ་དྲངས། ལིང་གི་སཱི་ཡུཊ་འོས། སཱི་ཡུཊ། ཊ་གནས་དོན་ཨུ་བརྗོད་དོན། བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས། བྱིངས་ལ་ཡོན་ཏན། ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ་ཝ་ཨི་ལ་དྲངས་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཥ། །ཡ་ཨཱ་ལ་དྲངས། །ཡབྷུ་ལ་སྨོན་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་དྷཾ་དུ་བསྒྱུར་ལིང་གི་སཱི་ཡུཊ་འོས་སཱི་ཡུཊ། ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་བ་ཡ་དག་གི་བལ་ལ་དབྱི་འོས་ཡ་ཕྱིས་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ལྔ་ཨི་ལ་་་དྲངས་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཥ། ཨིཊའི་གདམ་ང་འོས་དྷ་ཌྷར་འགྱུར་བ་དང་མི་འགྱུར་བ་འདམ་ཀ་བྱས་་་པའོ། །ཡབྷ་བི་ཥིཏྟཾ། སྨོན་པའི་ལིང་གི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བྷ་བི་ཥི་ཡ། བྷུ་ལ་སྨོན་པའི་ལིང་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་ཨི་ཊར་བསྒྱུར་ཨིཊའི་ཨཏ་འོས། ཨིཊ་ཨཏ་དུ་བསྒྱུར། ཏ་ཡིག་ཏ་ཡི་དེའི་དུས་ཀྱི་དོན། ལིང་གི་སཱི་ཡུཊ་འོས། སཱི་ཡུཊ། ཊ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ་ཝ་་ཨི་ལ་དྲངས་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཋ། ཡ་ཨ་ལ་དྲངས། བྷ་བི་ཊཱི་བ་ཧི། བྷཱུ་ལ་སྨོན་པའི་ལིང་རྗེས་་་—།
