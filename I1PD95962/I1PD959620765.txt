སྒྲ་མདོ། ཏི༽དྷིའི་འོ་མཚོ་ཚེ་ཚབསཊི་ནཀན་བེགྱུ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་བེགྱུཏ། སོགས་དང༌། ཏང་ལ། ཨ་བེ་གྲུ་ཏི། སོགས་སོ། ཥྛ་། ་ཟླ་བ་ལེགས་གནང་བ་ད༈བི་ཊཱི་ཁྱབ་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། །ཤཔ་ལྷུར་བསྒྱུར༑་གཉིས་བརྗོད། ནི་ཛ་སོགས་གསུམ་རྣམས་ཀྱིས་གླུ་ལ་ཡོན་ཏན་ནོས། ཨ་དྷྱས་ཡོན་ཏན་བྱེད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ་ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བེཤྲཱི། བི་བེཋ། བེ་བི་ཥ་ཏི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། RaACBỊTv":'sTENI'LLSTâyBa༑བེ་བིཥྛ། ལིང་ལ། བེ་བི་བྷཱུཏ། བེ་བི་བྷཱུ་ཏྭཾ། བེ་བི་ཤྩ། ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བེལ། བེ་བིཏྟཱཏ། བེ་བིཏྟཾ། །ཌ་བེ་བི་ཥ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བིནྟི། ཧུ་ཛྷལ་ལ་ཧི་དྷི། །མཐའི་ཛྷལ་སོགས་ཥ་། ཥ་ཆུ་ལ་ཁ་ཏུ་འོས། དྷ་བྷ། :：བི་བིབྷཱུཏ། བི་བིཊྚཾ། བི་བི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཥཱ་ཎི། བེ་བི་ཥཱ་བ། བེ་བི་ཥ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བེ་བེཌ། ཨ་བེ་བེཊ། ཨ་བེ་བིཏྟཾ། ཨ་བེ་བི་ཥཱ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བེ་བེཌ། ཨ་བེ་བིཊ། ཨ་བེ་བིཏྟཾ། ཨ་བེ་བིཊྚ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བ་བི་ཎཾ། ཨ་བེ་བི་ཥ། ཨ་བ་བི། ལུང་ལ། ལ་འགྲོ་བ་སོགས་ཀྱིས་ལྕི་ཨང་དང་ཀར་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ཨང་གི་་་ཕྱོགས་ལ། ཨ་བི་ཥཏ། ཨ་བི་ཥ་ཏྭཾ། ཨ་བི་ཥན། སོགས་དང་། ཀླུའི་་ཕྱོགས་ལ། ་ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས། ཥ་ཀ། །ཀུན་ནས་བསྒྱུར་བ་སོགས་་ཀྱིས་བསྒྲུབས་པས། ཨ་བི་ཀཱཏ། ཨ་བི་གྷ་ཏྭཾ། ཨ་བི་ཀླུ་ན། སོགས་སོ། །ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བེ་ཥ། །བི་བི་ཥ་ཏུ། བི་བི་ཥ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བེ། བི་བེ་ཏི་ཐ། བི་བི་ཥ་ཐུ། བི་བི་ཐ། །
