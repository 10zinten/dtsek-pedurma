ཨསྟ་ཐཱ། ཨསྡུ་བཱ་ཐཱ། ་་ཨསྟེ་དཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨསྐུ་བི། ཨསྟུ་བ་ཧི། ཨསྟུ་མ་ཧི། ལུང་གི་ཨ་ཏང་ལ། སྟུ་སུ་དྷཱཉ་རྣམས་ལས་པ་རསྨི་པ་ད་རྣམས་ལ་འོས་ཨིཊ་བྱིན། སིཙ་ལ་འཕེལ་བ་སོགས་ཀྱིས་འཕེལ་བ་བྱས་པས། ཨསྠཱ་བཱིཏེ། ཨསྠཱ་བིཏྟཾ། ཨསྠཱ་བི་ཥཱ། སོགས་སོ། ་། ཏང་ལ་བྱིངས་ཐམས་་་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཁ་ཨསྟོཋ། ་ཨསྟོ་ཥཱ་ཏྭཾ། ཨསྟོ་ཥ་ཏ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨསྟོཊྛ༑ོཨསྟོ་ཥཱ་ཐཱ། ཨསྟོ་དྷཾ། དྷ་ལས་ཀྱང་ངོས་ས་ཕྱིས། ཨསྟེ་ཞྭ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨསྟོ་ཏི། ཨསྟོ་ཥཱ་ཧི། ཨསྟོདྨ་ཧི། ལིཊ་ལ་བྱིངས་གཉིས་བརྗོད་ཤར་དང་པོ་སོགས་ས་ཕྱིས་བལ་དང་པོ་རྣམས་ལ། ཁྲི་དྲྀ་སྲྀ་སྐུ་སྲུ་དྲུ་སོགས་ཀྱིས་ཨིཊ་མི་འཇུག་། ཎལ་གཉིས་དང་ཐལ་མ་གཏོགས་གཞན་རྣམས་ལ། །སཱཉྫ་བརྩེགས་པ་དང་པོ་སོགས་ཀྱིས་ཀིཏ་ཡང་ན་བར་བྱས་པས། ཀིཏ་ཀྱི་ཕྱོགས་ཀྱི་ཨཙ་དང་པོ་རྣམས་ལ། །ཨཙ་ལ་གླུ་བྱིངས་སོགས་ཀྱིས་ཨུབ་དང་། ཀིཏ་མ་ཡིན་པའི་ཕྱོགས་ལ་ཡོན་ཏན་བྱ་སྟེ། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཏུཏྟཱ་བ། ཏུཋ་བ་ཏུ། ཏུg་བ་ཏུ། ཏུཋ་བུ། ཏུཉྩ་བུ། སྐྱེས་སུ་དབུས་མ་གསུམ་ལ༑ཏུབྦོ་ཐ། ཏུཥཱ་བ་ཐུ། ཏུ་བ་ཐུ། ཏུཥཱ་བ། ཏུཏྟཱ་བ། སྐྱེས་་བུ་ཕྱི་མ་གསུམ་ལ། ཏུཏྟཱ་བ། ཏུཥཱ་བ། ཏུཏྟོ་བ། ཏུཉྩ་བ། ཏུཏྠོ་མ། ཏུཉྩི་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཏུཏྠ་བེ། ཏུཏྟཱ་བེ། ཁ་ཏུཊྚ་བཱ་ཏེ། ཏུཉྩ་བཱ་ཏེ། ཏུབྦ་བི་རེ། ཏུཉྩ་བི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཏུཏྠོ་ཥ། ་ཏུཥྛ་ཥ། ཁ་ཏུཥཱ་བཱ་ཐེ། ཏུཏྟཱ་བཱ་ཐེ། ་ཏུཊོ་དེ། ཏུཥྛ་དེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཏུཥཱ་བེ། ཏུཏྟཱ་བེ། །ཏུབྦོ་བ་ཧེ། ཏུཏྟཱ་བ་ཧེ། ཏུཊོ་མ་ཧེ། ཏུg་མ་ཧེ། འགྲེལ་པར་ལུང་དྲངས་པ། ་ཊཱ་བྷིཥྛ་ཉོ—།
