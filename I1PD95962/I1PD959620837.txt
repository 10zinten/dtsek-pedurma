དེ་ལས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་ས་ཙིཏི་ནི་ནས་ཀྱིས་ཕྱི་ཕྱོགས་ལ་ལྟ་ཕྱོགས་དང་འོས༑ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས། །་བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་་་ཡོན་ཏན་བྱས་པས། ཀ་རི་ཥྱ་ཏི། སོགས་སམ། ཅ་ཀ་རཱི་ཎྱ་ཏི། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་ཀཾ་རི་ཥྱ་ཏི། སོགས་སམ། ་ཨ་ཀ་རཱི་ཥྛ་ཏ། སོགས་སོ། །ཀི་ར་ཏོ་ལ་བ་ནེ། ཀི་ར་ཏི་ལ་གཅོད་པར་བྱེད་པ་ལའོ། །ཉེར་བསྒྱུར་ཨུ་པ་ལས་ཕ་རོལ་གྱི་ཀི་ར་ཏི་ལ་གཅོད་པར་བྱེད་པའི་དོན་ལ་སུཊ་ཨཱ་ག་མ་འགྱུར་ཏེ། ཨུ་པ་ཀི་ར་ཏི། ཉེ་བར་གཅོད་པར་བྱེད། །ཨཌ་བྷཱ་ས་བྱ་བཱ་ཡེཉཔི། །ཨཊ་ཨ་བྷྱཱ་ས་དག་བར་ན་ཡོད་ཀྱང་ངོ༌། །ཨུ་པ་ལས་ཕ་རོལ་གྱི་ཀི་ར་ཏི་ལ་ཨཊ་ཨཱ་ག་མ་དང་ཨ་བྷཱ་ས་བར་ན་ཡོད་ཀྱང་ཀའི་སྔོན་དུ་སུཊ་ཨཱ་ག་མ་འགྱུར་ཏེ། །ཨཊ་་བར་ན་ཡོད་པ་ལ། །ལང་ཨུ་པཱཀི་རཏ། …ཉེ་བར་གཅོད་པར་བྱེད། ཨ་བྷཱུ་ས་བར་ན་ཡོད་པ་ལ། ལིཊ་ཨུཔ་ཙསྐུ་ར། ཉེ་བར་གཅོད། ལྟ་བུའོ། །ཧིད་ཡཾ་པྲ་ཏེ། འཚེ་བ་ལ། པྲ་ཏི་ལས་ཀྱང་ངོ༌། །ཀྲྀ་འཚེ་བའི་དོན་ལ་འཇུག་པ་ན་ཨུཔ་དང་:པྲ་ཏི་ཕ་རོལ་ལས་ཀྱང་སུཊ་ཨཱ་ག་མ་འགྱུར་ཏེ། ་ཨུ་པ་ཀི་ར་ཏི། ཉེ་བར་འཚེ་བར་བྱེད། པྲ་ཏི་ཀི་ར་ཏི། སོ་སོར་འཚེ་བར་བྱེད། །སོགས་སོ། །ན་འབད་བ་དཀཔོ་་་་མི་ལ་ཁྱེད་༈གཽ་གསལ་བར་བྱེད་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ཨ་ཙི་པི་བྷཱ་ཥཱ། ཨ་ཙ་ལ་གདམ་ངའོ། །རྐྱེན་ཨ་ཙ་དང་པོ་ཅན་ཕ་རོལ་ལ་གྲིའི་རྦ་ཡིག་ལ་ཡིག་ཏུ་བསྒྱུར་བ་དང༌། ར་ཉིད་དུ་གནས་པ་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། གྲྭ་ལ་ཤ། བྱིངས་ཀྱི་རྀཏ་ཀྱི་ཨིཏ་འོས། རཱི་ཨིཏ། རིའི་ཨཎ་ལས་ར་ཕ་རོལ་ལོས་ར་བྱིན། མདོ་འདིས་ལར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་འདམ་ཀ་ཅན་དུ་བྱས་པས༑ལཊ་ལ། གི་ར་ཏི། སོགས་སམ། གི་ལ་ཏི། སོགས་སོ། །དེ་བཞིན་དུ། །ལིང་ལ། གི་རེ་ཏ། སོགས་སམ། གི་ལེ་ཏ། སོགས་སོ། །ལོཊ་ལ། གི་ར་ཏུ། སོགས་སམ། གི་ལ་ཏུ། སོགས་སོ། །ལང་ལ། །ཨ་།
