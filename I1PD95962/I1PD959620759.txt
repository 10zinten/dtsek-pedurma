བསྟན་འགྱུར། སྒྲ་མདོ། ༼༽་་་་ཆོ་ཏོ་ཀ་བ་ལྟ༌བུ་ལ་ཨ་ཛ་ཧི་ཏཾ། ་ཨ་ཛ་ཧཱི་ཏཾ། ཨ་ཛ་ཧི་ཏ། ཨ་ཛ་ཧཱི་ཏེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཛ་ཧཾ། ཨ་ཛ་ཧི་བ། ་ཨ་ཛ་ཧཱི་བ། ཨ་ཛ་ཧི་མ། ཨ་ཛ་ཧཱི་མ། ལུང་ལ། ཡམ། རམ། ནམ། ཨཱཏ་མཐའ་སོགས་ཀྱིས་སཀ་དང་ཨིཊ་བྱིན་པས༑ཨ་ཧཱ་སཱིཏ། ཨ་ཧཱ་སིཏྟཾ། ཨ་ཧཱ་སི་ཥཱ། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད། ཨཱཏ་ལས་ཎལ་ཨོ་འོས། །རྣལ་གཉིས་ཨོ། །ཨིཊ་ལའང་ཨཱཏ་་་ཀྱིས་དབྱི་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛ་ཧོ། ཛ་ཧ་ཏུ། ཛ་ཧུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཛ་ཧཱ་ཐ། ་ཛ་ཧི་ཐ། ཛ་ཧ་ཐུ། ཛ་ཧ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛ་ཧོ༄༅། ཛ་ཧི་བ། ཛ་ཧི་མ། ལུཊ་ལ། ཧཱ་ཏཱ། ཧཱ་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ལིང་ལ་ཨེ་འོས་ཨེ་རུ་་་་བསྒྱུར་བས་ཧེ་ཡཱཏ། ཧེ་ཡཱནྟཾ། ཧེ་ཡཱ་སུ། སོགས་སོ། །ལྀཊ་ལ། །ཧཱ་སྱ་ཏི༑སོགས་དང་། ཁྲིང་ལ། ཨ་ཧཱ་སྱ་ཏ། །སོགས་སོ། །སྐྱེས་བུ་༈རྀ་འགྲོ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། རཾཔ། ཤཔ་ཀླུར་བསྒྱུར། ཀླུ་ལའང་ངོས་གཉིས་བརྗོད། ཨརྟི་པི་པརྟི་དག་གི་འང་ངོས། ་ཨ་བྷྱཱ་ས་ཨི། ཨ་ངྷ་སའི་འདྲ་བ་མ་ཡིན་པ་ལ་འོས། ཨི་ཡ་རུ་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨི་ཡརྟི། ཨི་ཡཾ་ཏ། ཨི་ཕྲ་ཏི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨི་ཡནྟི། ཨི་ཡི་ཐ། ཨི་ཡཾ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨི་ཡརྨེ། ཨི་ཡི་བ། ཨི་ཡི་མ། ལིང་ལ། ཨི་ཡི་ཡཱཏ། ཨི་ཡི་ཡཱ་ཏཱ། ཨི་ཛཾ་ཡུ། སོགས་སོ། །དེ་བཞིན་དུ། ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨི་ཡཆུ། ཨི་ཡཾ་ཏཱཏ། ཨ་མྲི་ཏྭཾ། ཨི་ཡ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨི་ཡི་ཧི། ཨི་ཡཾ་ཏཱཏ། ཨི་ཡཾ་ཏཾ། ཨི་ཡི་ཏ། སྐྱེས་བུ་ཕྱི་མ་་་གསུམ་ལ། ཨི་ཡ་རཱ་ཎི། ཨི་ཡ་རཱ་བ། ཨི་ཡ་རཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་།
