སྒྲ་མདོ། ཏི་། བློ་ནི་བ་སྐྱོ་ཁྲི་ཙེད་པསྶ་མཚོ་བི་བྷཱི་ཏ། མང་ཚིག་ལ་ཨཙ་གཅིག་མ་ཡིན་པ་སོགས་ཀྱིས་ཡ་ཎར་བསྒྱུར་བས། བི་བྷཱུ་ཏི། །དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བྷེ་ཏི། བི་བྷི་ཐ། བི་བྷི་ཐ། བི་བྷི་ཐ། བི་བྷི་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷེ་མི། བི་བྷི་བ། བི་བྷི་བ། བི་བྷི་མ། བི་བྷི་མ། ལིང་ལ། བྷིའི་ཡང་ན་འོས་ཨཱི་ཐུང་ངུར་ཡང་ན་བསྒྱུར་བས། བི་བྷི་ཡཱཏ། ཎབི་བྷི་ཡཱ་ཏྭཾ། བི་བྷི་ཡུ། །སོགས་སམ། བི་བྷི་ཡཱཏ། བི་བྷི་ཡཱ་ཏྭཾ། བི་བྷི་ཡུ། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བྷེ་ཏུ། :བི་བྷི་ཏཱཏ། བི་བྷཱི་ཏཱཏ། བི་བྷི་ཊཱི། བི་བྷཱི་ཏཱཾ། བི་བྷུ་ཏུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བྷི་ཧི། བི་བྷི་ཧི། བི་བྷི་ཏཱཏ། བི་བྷཱི་ཏཱཏ། བི་བྷི་ཏཾ། བི་བྷཱི་ཏཾ། བི་བྷི་ཏ། བི་བྷཱི་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷ་ཡཱ་ནི། བི་བྷ་ཡཱ་བ། བི་བྷ་ཡཱ་མ། དེ་བཞིན་དུ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བི་བྷེ་ཏ། ཨ་བི་བྷི་ཏཱཾ། ཨ་བི་བྷཱི་ཏཱཾ། ཨ་བི་བྷུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བི་བྷེ། ཨ་བི་བྷི་ཏཾ། ཨ་བི་བྷཱི་ཏཾ། ཨ་བི་བྷིཏ། ཨ་བི་བྷཱི་ཏ། སྐྱེས་བུ་ཕྱི་་མ་གསུམ་ལ། ཨ་བི་བྷ་ཡཾ། ཨ་བི་བྷི་བ། ཨ་བི་བྷི་བ། #ཨ་བི་བྷི་མ། ཨ་བི་བྷི་མ། ལུང་ལ། སིཙ་ལ་འཕེལ་བ་པ་རསྨི་པ་ད་རྣམས་ལ་འོས་འཕེལ་བ་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྷི་ཏ། ཨ་བྷིཏྟཾ། ཨ་བྷི་ཥཱ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བྷི་ཥཱི། ཨ་བྷིཏྟཾ། ཨ་བྷིཤྩ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་བྷི་ཁཾ། །ཨ་བྷི་ཥཱ། ཨ་བྷིཥྛ། ལིཊ་ལ། བྷི་ཧྲི་སོགས་ཀྱིས་་ཨཱམ་ཡང་ན་བར་བྱེད་དེ། ཨཱམ་གྱི་ཕྱོགས་ལ། བི་བྷ་ཡཾ་ཙ་ཀཱ་ར། །བི་བྷ་ཡཱ་བ་བྷུ་བ། བི་བྷ་ཡཱ་མཱ་ས། སོགས་དང༌། ཨཱམ་མ་སྦྱར་བའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བྷཱ་ཡ། བི་བྷུ་ཏུ། བི་བྷིུ། སྐྱེས་བུ་དབུས་མ་—།
