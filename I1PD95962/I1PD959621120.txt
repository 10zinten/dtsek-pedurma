བྷ་ས་ལམདི་ཀྱི་དི་ངྷི་ཥ། ཤིངསཱི་དི་མི་དི;ཀླུ་དི༔བྷཾ་ཥ་ལས་ནིཊྛཱའོ། །ཤིང་སོགས་འདི་རྣམས་ལས་ཕ་རོལ་གྱི་ནིཥྛ་ཨིཊ་དང་བཅས་པ་ཀ་འགྲོ་བ་མ་ཡིན་ཏེ། ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཤ་ཡི་ཏ། ཉལ། སྡེ་དི་ཏ། གཤེར་བ། མེ་དི་ཏ། གྲོགས། གླུ་དི་ཏ། །དྷནྟི་ཏ། cཨཱ་དི་ཀརྨ་ཎི་ཀརྟ་རི་ཙ། །དང་པོའི་ལས་ལཀྟསྟེ་བྱེད་པ་པོ་ལའང་ངོ༌། །དང་པོའི་ལས་ཏེ་བྱ་བ་རྩོམ་པ་ལ་འཇུག་པའི་ལས་དང་བྱེད་པ་པོ་དག་ལའང་ལྟ་འགྱུར་རོ། །བི་བྷཱ་ཥཱ་བྷཱ་བཱ་དི་ཀརྨ་ཎོ། དངོས་པོ་དང་དང་པོའི་ལས་ལ་ཡང་ནའོ། །དངོས་པོ་དང་དང་པོའི་ལས་དག་གི་དོན་དུ་བྱིན་པའི་ནིཥྛ་ལ་ཨིཊ་ཡང་ན་འགྱུར་ཏེ། པྲ་སྦེ་དི་ཏ། རབ་ཏུ་གཤེར། པྲ་སཱིནྡ། མེ་དི་ཏ། གྲོགས། མིནྣཾ། །སོགས་སོ། ་། མྲྀ་ཥ་སྟི་ཏིཥཱ་ཡཾ། ཀྲྀ་ཥ་ལས་བཟོད་པ་ལའོ། །།མྲྀ་ཥ་ལས་བཟོད་པ་ལ་འཇུག་པ་ན་ཨིཊ་དང་བཅས་པའི་ནིཏྟཱ་ཀ་འགྲོ་་་བའི་མིང་ཅན་མ་ཡིན་ཏེ། བཟོད་པ་བྱས། མནྟི་ཏ། །ཨུ་དུ་པ་དྷཱདྷཱ་བཱ་དི་ཀརྨ་ཎི་ར་རྒྱ་ཏ་རནྟཾ། ཨུཏ་ཨུ་པ་དྷཱ་ལས་དངོས་པོ་དང་དང་པོའི་ལས་དག་ལ་ཡང་ནའོ། །བྱིངས་ཨུཏ་ཨུ་པ་དྷཱ་ལས་དངོས་པོ་དང་དང་པོའི་ལས་ཀྱི་དོན་དུ་བྱིན་པའི་ནིཥྛ་་་་ཨིཊ་དང་བཅས་པ་ཀ་འགྲོ་བའི་མིང་ཅན་ཡང་སྟེ། །པྲ་དྲོ་ཏི་ཏ། རབ་གསལ། པྲ་དྷུ་ཏི་ཏ། སོགས་སོ། །ནོ་པ་ཏྠ་པཱནྟཱ་དཱ། ན་ཨུ་པ་དྷཱ་ཐ་ཕ་མཐའ་ལས་་་་ཡང་ནའོ། །བྱིངས་ན་ཨུ་པ་དྷཱ་ཐ་དང་ཕའི་མཐའ་ཅན་ལས་ཕ་རོལ་གྱི་ནིཥྛ་ཨིཊ་དང་བཅས་པ་ཀ་འགྲོ་བའི་མིང་ཅན་ཡང་ན་སྟེ། མནྡྲི་ཏ། སྲུབས། མ་ཐི་ཏ། ཧཱུྃ་ཕི་ཏ། སྐྱོད། གུ་ཕི་ཏཱ། །སོགས་སོ། ཥ། ནིཏྟཱ་ཡཱ་སེ་ཊི། ཨིཊ་དང་བཅས་་་་པའི་ནིཥྛ་ལའོ། །ཨིཊ་དང་བཅས་པའི་ནིཥྛ་ཕ་རོལ་ལ་ཎིཙ་དབྱི་སྟེ། །བྷཱ་བེ་ཏ༑བྱུང། བྷཱ་བི་ཏ་བཱན་སོགས་སོ། །།ཨཱ་དི་ཀརྨ་ཎི། །དང་པོའི་ལས་ལའོ། །ཤཀྱི་ལ་དང་པོའི་ལས་ཀྱི་དོན་དུ་བྱིན་པའི་ནིཥྛ་ལ་ཨིཊ་ཡང་ན་མི་འགྱུར་ཏེ། དྷཱ།
