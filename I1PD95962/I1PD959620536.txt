ཨིཏ་འགྲོ་བ་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་མིང་དུ་བསྒྲུབས་པའི་རྐྱེན་དང༌། ཚིག་ཏུ་་་བསྒྲུབས་པའི་རྐྱེན་གང་བྱུང་ཡང་བྱིངས་དེ་ལ་ནུམ་ཨཱག་མ་སྟེར་ཏེ། མ་གནས་་་དོན་དང་ཨུ་བརྗོད་དོན་ནོ། །ཤ་ཞེན་བག་ས་ག་ན་ག་::ཀོ་:༈བྱིངས་ཀུ་ཏི་འཚེ་བ་དང་ཉོན་མོངས་པ་དག་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བའི་ལཊ་ལ། ་ཀུནྡྲ་ཏི་སོགས་སོ། །ལིང་ལ། ་ཀུནྡྲི་་ཏ་སོགས་སོ།  ོ། ལོཊ་ལ། ་ཀུནྡྲ་ཏུ་སོགས་སོ། །ལང་ལ། ཨ་ཀུཉྫཏ་སོགས་སོ། །ལུང་ལ་ཨ་ཀནྟཱིཏ་སོགས་སོ། །ལིཊ་ལ། ཀུ་ཧོཀུ་ཧ་དག་གི་ཙུའོ། །།བྱིངས་གཉིས་བརྗོད་པའི་ཨ་བྷྱཱ་སའི་མིང་ཅན་དེ་ཀུ་སྟེ་ཀ་སྡེ་ཡིན་ན་གྲངས་མཐུན་་པའི་ཙ་སྔར་འགྱུར་ཞིང་ཧ་ཡིག་ཡིན་ན་ཛྷ་ཡིག་ཏུ་འགྱུར་ལ། །ཛྷ་དེ་ཡང་ཨ་བྷཱ་་་སའི་ཅརའང་ངོ་ཞེས་པས་ཛ་ཡིག་ཏུ་འགྱུར་ཏེ། ཙུ་ཀུནྡྲ། ཙུ་ཀུནྡྲ་ཏུཙུ་ཀུནྡྷསོགས་སོ། །ལུཊ་ལ། ཀུནྡྲི་ཏཱ་སོགས་དང་། སྨོན་པའི་ལིང་ལ་སྐབས་འདིར་ཨི་འགྲོ་བའི་ནུམ་ཡིན་པས་ན་མི་དབྱི། ་ཀུནྡྷོ་ཏ་སོགས་སོ། །ལྀཊ་ལ། །ཀུནྡྲི་་ཁྱེ་ཏི་སོགས་སོ། །ལྀང་ལ། ཨ་ཀུནྡྲི་ཥྱ་ཏ་སོགས་ཏེ་རིགས་འགྲེའོ། །།དྷཱ་ཏཱ་དེཥསབྱིངས་ཀྱི་དང་པོའི་ཥའི་སའོ། །ཎི་དྷུ་ལྟ་བུ་བྱིངས་ཀྱི་དང་པོའི་ཥ་ཡིག་ནི་ས་ཡིག་ཏུ་འགྱུར་ཏེ། ཤི། npo། ཀ་ཚེ་བ་ཀུན་ལ་གལ་ཏེལུR༈ཏི་དྷུ་འགྲོ་བ་ལའོ། །ཞེས་པའི་བྱིངས་ལ་ལཊ་སོགས་པ་རམྨེ་པ་དང་་་་རྣམས་སྦྱར་བའི་ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། སེ་དྷ་ཏི། །ཨུ་པ་་་སཧཱཏྨ་ནོ་ཏི་སུ་བ་ཏི་སྐྱ་ཏི་སྟོ་ཏི་སྟོ་དྷ་ཏིསྨཱ་སེ་ན་ཡ་སེ་དྷ་སི་ཙ་སཉྫ་སཱཤྩ། །ཉེར་བསྒྱུར་ལས། །སུ་ནོ་ཏི་སུ་བ་ཏི་སྐྱ་ཏི་སྟོ་ཏི་སྟོ་བྷ་ཏི་ཟླ་སེ་ན་ཡ་སེ་དྷ་སི་ཙ་སཉྫ་་སཱཉྫ་རྣམས་ཀྱིའོ། །ཉེར་བསྒྱུར་ས་ཡིག་ཥ་ཡིག་ཏུ་འགྱུར་བའི་རྒྱུ་མཚན་ཅན་སྔ་མ་ལས་བྱིངས་སུ་ནོ་ཏི་སོགས་ཏེ། ཏུཉེ་བསྐྱེད་པ། སྐྱུཉ་དེད་པ། །ཁོ་མཐའི་་་—།
