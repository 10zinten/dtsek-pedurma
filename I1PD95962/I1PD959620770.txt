མང་ཚིག་ལ། །ད་དྷ་ཏི། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ད་དྷཱ་སི༑དྷཏྠ། །དྷཏྠ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་དྷཱ་མི། ད་དྷཱ། ད་࿄༑དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །དྷཏྟེ། ད་དྷ་ཏེ། ད་དྷ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྷཏྠེ། ད་དྷཱ་ཐེ། ་དྷདྡྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་དྷེ། ད་དྷ་ཧེ། ད་དྨ་ཧེ། ལིང་གི་ཨ་ཏང་ལ། །་་སོགས་ཀྱིས་ཨཱ་ཕྱིས་པས། ད་དྷཱཏཔོ། ད་དྷ་ཏྭཾ། ད་དྡྷི། །སོགས་དང་། ཏང་ལ། མཐའ་མ་ཡིན་པ་སོགས་ཀྱིས་ས་ཕྱིས། བྷ་སོགས་ཨཱ་ཕྱིས་པས། ད་དྷཱི་ཏ། ད་དྷཱི་ཏཱཾ། །ད་དྷི་རན། སོགས་སོ། །ལོཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དྷཱ་ཏུ། དཏྟཱཏ། །དཏྟཾ། ད་དྷ་ཏུ། སྐྱེས་བུ་དབུས་མ་་་གསུམ་ལ། །་དྷེ་ཧི། དྷཏྟཱཏ། དྷཏྟཾ། དྷཏྟ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་དྷཱ་ནི། ད་དྷཱ་བ། ་ད་དྷཱ་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དྷཏྟཾ། ད་དྷཱ་ཏྭཾ། ད་དྷ་ཏྭཾ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྷཏྟཱ། ད་དྷཱཐཾ། དདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ་དདྷི། ་ད་དྷཱ་བ་ཧི། ད་དྷཱ་མ་ཧི། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ད་དྷཱཏ། ཨ་དྷཏྟཾ། ཨ་ད་དྷི། །སྐྱེས་བུ་་་དབུས་མ་གསུམ་ལ། ཨ་ད་དྷཱ། ཨ་དྷཏྟཾ། ཨ་དྷཏྟ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ༑ཨ་ད་དྡྷཾ། ་ཨ་ད་དྷ། ཨེདྨ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷཏྟ། །ཨ་ད་དྷཱ་ཏྭཾ། ཨ་ད་དྷ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་དྷཏྟཱ། ཨ་ད་དྷཱ་ཐཾ། ཌ་ཨ་དྷ,ི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ད་དྷི། ཨ་ད་དྷཱ་ཧི། ཨ་ད་དྷཱཧི། །ལུང་གི་ཨ་ཏང་ལ། ག་ཏི་སྦ་གླུ་སོགས་ཀྱིས་སིཙ་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷཱཏ། ཨ་དྷཱཏྟཾ། ཨ་དྷུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷཱ་ཨ་དྷཱ་ཏཾ། ཨ་དྷཱཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་དྷཾ། —།
