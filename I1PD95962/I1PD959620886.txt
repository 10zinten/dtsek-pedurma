ཀྱིས་མ་ཕྱིས་པས། སྐུ་དྡྷཱ་ཏ། སྐུ་བྷཱུཙཱ། སྐུ་དྡྷཱ་སུ། །སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་པས། ་སྐུམྷི་ཥྛ་ཏི། སོགས་དང༌། དེ་བཞིན་དུ། ་ཁཁྲིང་ལ། ཨསྐུརྞཱི་ཥཱཏ། །སོགས་སོ། །ཀ་བ་གང་། །:གས་པོ་ལ་ཚོ། །༈མཱིཉ་འཚེ་བ་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་སྦྱོར་ཏེ། ལཊའི་ཨ་ཏང་ལ། མཱི་ནཱ་ཏི། སོགས་དང་། ཏང་ལ། :མཱི་ནཱི་ཏེ། སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། མཱི་ནཱི་ཡཱཏ། །སོགས་དང༌། ཏང་ལ། མཱི་ནཱི་ཏ། །སོགས་སོ། །།ལོཊའི་ཨ་ཏང་ལ། ་མཱི་ནཱ་ཏུ། སོགས་དང་། ཏང་ལ། མཱི་ནཱི་ཏཱཾ། སོགས་སོ། །།ལང་གི་ཨ་ཏང་ལ། ཨ་མཱི་ནཱ་ཏ། །སོགས་དང་། ཏང་ལ། ཨ་མཱི་ནཱི་ཏ། །སོགས་སོ། །།ལུང་ལ། མཱི་ནཱ་ཏི། །::མི་ནོ་ཏི། དཱིང་རྣམས་ཀྱིས་ལྕཔ་ལའང་འོས། ཨཱཏ་དུ་བསྒྱུར། ་ཨ་ཏང་ལ། ཡམ་རམ་ནམ་ཨཱཏ་རྣམས་ཀྱིས་ཉིཀ་འང་ངོས། %སཀ་དང་ཨིཊ་བྱིན་པས། ཨ་མཱ་སཱི་ཏ། ཨ་མཱ་སིཏྟཾ། ཨ་མཱ་སི་ཥཱ། སོགས་སོ། །ཏང་ལ་ས་འདྲེན་པ་ཁོ་ན་སྟེ། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་མཱསྟེ། ་ཨ་མཱ་སཱ་ཏཱ། ཨ་མཱ་ས་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །་ཨ་མཱསྟྲཱི། ཨ་མཱ་སཱ་ཐཱི། ::ཨ་མཱ་ནཱཾ། སྐྱེས་བུ་ཕྱི་མ་་་་གསུམ་ལ། ཨ་མཱ་སི། ཨ་མཱ་སཱ་ཧི། ཨ་མཱ་སྨྲ་ཧི། །ལིཊ་ལ་ཎལ་གཉིས་དང་ཐལ་ལ། །མཱི་ནཱ་ཏི། མི་ནོ་ཏི་སོགས་ཀྱིས་ཨཱཏ་དུ་བསྒྱུར་ནས་གཉིས་བརྗོད་གཞན་རྣམས་ལ་གཉིས་བརྗོད་ཐུང་ངུ་འོས། ཐུང་དུར་བསྒྱུར། ཨཙ་གཅིག་མ་ཡིན་པ་སོགས་ཀྱིས་ཡཎར་བསྒྱུར་བས། །ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། མ་མོ། མི་མྱ་ཏུ། །མི་མྱུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །མ་མི་ཐ། མ་མཱ་ཐ། །མི་མྱ་ཐུ། མི་མྱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། མ་མོ། མི་མྱི་བ། མི་མྱི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །མི་མྱེ། མི་མཱ་ཏེ། མི་མྱི་རེ། །
