བ་། ་་་་་་་་་་་་བ་ཚོས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་ཙེ་ནི་འཚོ་བའི་མི་ཁྱོན་ལྔ་བློ་པོའི་ཚོགས། དམར་རུ་མགོ་ནག་གི་ཚོགས། གོཎྜི་ཀ། གུཉྫ་ལ་ཟཀ། སུ་ཁ་ཤཱ་ཡ་ལ་ཟཀསོ་ཁ་ཤཱ་ཡི་ཀ༑བདེ་བར་ཉལ་བའི་ཚོགས་སམ་དམར་སེར་གྱི་ཚོགས། སིངྒ་ལི་ཀ། སིངྒ་ལི་ལ་ཁཀ། པཱ་ཤཱ་དི་བློ་ཡ༑བཱ་ཤ་ལ་སོགས་པ་རྣམས་ལས་ཡའོ། །པཱ་ཤ་ལ་སོགས་པ་རྣམས་ལ་ཚོགས་པ་བརྗོད་པ་ལ་དྲུག་པའི་མཐའ་ལས་ཡ་འགྱུར་ཏེ། ཡ་པཱ་ཤྩ། ཞགས་པའི་ཚོགས། ཡ་བཱ་ཏུ༑ཡརླུང་གི་ཚོགས། །ཡ་ཊབ་སྦྱར་བ་སྟེ་རང་བཞིན་གྱི་མོ་རྟགས་སོ། །དྲྀ་ཏྱཱ། རྩའི་ཚོགས། ཨུ་ལོཉཔི༑ཨུ་ལ་འང་ངོ༌། །བ་ཏ་ལས་ཚོགས་པ་བརྗོད་པ་ལ་ཨུ་ལའང་འགྱུར་ཏེ། རླུང་གི་ཚོགས། བཱ་ཏུ་ལ། ཨུ་ལ། འགྲེལ་པར་ལུང་དྲངས་པ། ཁ་ལ་གོ་ར་ཐཱ་རྱ། ཁ་ལ་གོ་ར་ཐ་ལས་ཡའོ། །ཁ་ལ་སོགས་པ་འདི་རྣམས་ལས་ཚོགས་པ་བརྗོད་པ་ལ་ཡ་འགྱུར་ཏེ། ཡ་ཁ་ལྕེ། ཊབ་ཐ་ཆད་དམ་ངན་པའི་ཚོགས། ཡ་ག་བྱུ། ཊབ་བ་ལང་གི་ཚོགས། ཡ་ར་ཐཱ། ཊབ་ཤིང་རྟའི་ཚོགས། ཨི་ནི་ཏྲ་ཀ་ཏྱཱ་ཙ། ཨི་ནི་ཏྲཀ་རྱ་ཙ་རྣམས་ཀྱང་ངོ༌། །ཁ་ལ་གོར་ཐ་རྣམས་ལས་ཚོགས་པ་བརྗོད་པ་ལ་རིམ་པ་བཞིན་ཨི་ནི་ཏྲ་ཀ་བྱཙ་་་རྣམས་འགྱུར་ཏེ། ་ཐ་ཆད་དམ་ངན་པའི་ཚོགས། ཁ་ལི་ནཱི། ཁ་ལ་ཨི་ནི་ནའི་ཨི་བརྗོད་དོན་རྀཏ་ན་མཐའ་ཞེས་པས་དཱིབ། བ་ལང་གི་ཚོགས་སམ་རིགས་ཀྱི་ཚོགས། གོ་ཏཱ། གོ་ཏྲ་ལ་ཊབ། ་ར་ཐ་ལ་ཀ་རྱ་ཙ་ཙ་དབྱངས་དོན། ཡར་ཐ་ཀ་ཏྱ། ཤིང་རྟའི་ཚོགས། ཏ་ད་ཙྩཾ་པྲ་ཧ་ར་ཎ་མི་ཏི་ཀྲྀ་ཌཱ་ཡཱཎྞ། ་མཚོན་ཆ་ལས་དེ་འདི་ལ་རྩེ་བར་བྱེད་པ་ཞེས་པ་ལ་ཎའོ། །མཚོན་ཆ་ལས་དེ་འདི་ལ་རྩེ་བར་བྱེད་པ་ཞེས་པའི་དོན་ལ་དང་པོའི་མཐའ་བདུན་པའི་དོན་ཅན་ལ་ཎ་འགྱུར་ཏེ། དཎྜཔྲ་ཧ་ར་ཎ་མ་སྶཾ་གཾ་ཌཱ་ཡཾ། འདི་ལ་མཚོན་ཆ་དབྱུག་པས་རྩེ་བར་བྱེད། ཞེས་པའི་དོན་ལ། དཎྜ་ལ་ཎ། དབྱུག་པས་རྩེ། དཱཎཱ། ཊབ། དེ་བཞིན་དུ། མུཉྫི་ལ་ཎ་འཕེལ་བ་ཌཔ། མོཏྟཱ། ཁུ་ཚུར་གྱིས་རྩེ། བྷཉསཱ་གཱི་ཡེ་ཏི་ཉ། །དེ་འདི་བྱ་བ་ཞེས་པའི་གྷན་ལ་ཉའོ། །།བྱ་་བའི་ཚིག་སྔོན་ན་ཡོད་པ་གྷ་ཉའི་མཐའ་ལས་བདུན་པའི་དོན་ལ་ཉ་འགྱུར་ཏེ། ་བྱེ་ན་པཱ་ཏ་ལ་ཉ། བྱེ་ན་ཏི་ལ་སྱ་པཱ་ཏེ་ཉེ། བྱེ་ན་ཏི་ལའི་པཱན་ཉ་ལའོ། །པཱ་ཏ་m།
