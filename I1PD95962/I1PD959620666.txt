བརྟའི་ལྷ་ལ་དབྱི་འོས་ལྷ་གོང་མ་ཕྱིས། ས་ཧི་བ་ཧ་སོགས་ཀྱིས་ཨོཏ་དུ་བསྒྱུར་བ་་་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ། བོ་ཝཱ། བོ་ཞྭ་སི་སོགས་དང་། ཏང་ལ། ་བོ་ཎཱ། །བོ་ཞྭ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། བ་ཙི་སཱ་པི་ཡ་ཛ་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎ་བྱས་པས། །ཨུ་ཧཱུཏ། ཨུ་ཧནྟཾ། ཨུ་ཧུ་སུསོགས་དང༌། ཏང་ལ། ཧའི་ཌྷ་འོས། ཧ་ལྷ། ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས། ཌྷ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། ཝཀྐིཋ༑ཝཀྐི་ཡཱནྟཾ། ་བཀཱི་རན་སོགས་སོ། །ལྀཊ་ལའང་ཧ་ཌྷར་བསྒྱུར་བ་་་སོགས་ཀྱིས་བསྒྲུབས་པས། #ཨ་ཏང་ལ། ཝགྱུ་ཏི། ཝགྱུ་ཏ། བནྟི་སོགས་དང༌། །འོ་ཏང་ལ། ་བསྒྱུ་ཏེ། བཀྱེ་ཏི། བཉྩནྟེ་སོགས་དང༌། དེ་བཞིན་དུ། །སྲྀང་གི་ཨ་ཏང་ལ། ཨ་བསྒྱུ་ཏ། ཨ་བསྒྱུ་ཏཱ། ཨ་བགྱུན་སོགས་དང༌། ཏང་ལ། །ཨ་བགྱུཏ། ཨ་བཀྱེ་ཏྭཾ། ཨ་བ་གྱུནྟ་སོགས་སོ། །ན་ག༈བེཉ་ཐགས་འཐག་པ་ལའོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་པས་ལཊ་སོགས་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། ལཊའི་ཨ་ཏང་ལ། བ་ཡ་ཏི་སོགས་དང༌། ཏང་ལ། བ་ཡ་ཏེ་སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། བ་ཡེཏ་སོགས་དང༌། ཏང་ལ། བ་ཡེཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། བ་ཡ་ཏུ་་སོགས་དང་། ཏང་ལ། བ་ཡ་ཏྭཾ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། །ཨ་བ་ཡཏ་སོགས་དང༌། ཏང་ལ། ་ཨ་བ་ཡཏ་སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ། ཤ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་ཨེཙ་ཨཱཏ། །ཡམ། རམ། ནམ་ཨཱཏ་རྣམས་ཀྱི་ས་ཀ་འང་ངོས། ཌཨིཊ་དང་སཀ་བྱིན་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བཱ་སཱིཏ། ཨ་བཱ་སིདྡྷཾ། ་ཨ་བཱ་སི་ཥ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བཱ་་་སཱི། ཌཨ་བཱ་སིཏྟཾ། ཨ་བཱ་སི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བཱ་སི་ཎཾ། །
