ལས་ཅུའོ། །དི་ཏི་ཨ་དི་ཏི། ཨཱ་དི་ཏ་རྣམས་དང་པ་ཏི་ཚིག་ཕྱི་མར་འགྱུར་བ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་ཅུ་འགྱུར་ཏེ། རྒྱུ་ཎ་འཕེལ་བའི་དོན། དེ་ཏྲཿ། སྦྱིན་པའི་བུ། །རྒྱ་ཨ་དི་ཏྱ། མ་སྦྱིན་མའི་བུ། ་ཨཱ་དི་ཏྱ་ལ་རྒྱ་ཎ་འཕེལ་བའི་དོན་ཧལ་ལས་ཡམ་རྣམས་ཀྱི་ཡམ་ལ་དབྱི་འོས་ཡ་སྔོན་མ་ཕྱིས་པའོ། །ཨཱ་དི་ཏྱ། ཀུན་དུ་སྦྱིན་མའི་བུ། རྒྱ་པྲ་ཛཱ་པ་ཏྱཾ། སྐྱེ་རྒུའི་བདག་པོའི་བུ། རྒྱ་སེ་ནཱ་པ་ཏུཾ༑སྡེའི་བདག་པོའི་བུ། འགྲེལ་པར་ལུང་དྲངས་པ། ཡ་མཱཙྪ། ཡ་མ་ལས་ཀྱང་ངོ༌། །ཡ་མ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་རྒྱུ་འགྱུར་ཏེ། ཡཱ་ཏྱཾ། གཤིན་རྗེའི་བུ། ཡང་འགྲེལ་པར་ལུང་དྲངས་པ། དེ་བ་སྱ་ཡཉ་ཉོ། དེ་བའི་ཡཉ་ཨཉ་དག་གོ། །དེ་བ་ལས་བུ་བརྒྱུད་ལ་་ཡཉ་དང་ཨཉ་དག་འགྱུར་ཏེ། ཡཉ་…དེ་བྱ། ལྷའི་བུ། ཨནདི་བ། ལྷའི་བུ། ཡང་འགྲེལ་པར་ལུང་དྲངས་པ། བ་ཧི་ཥཊི་ལོ་པོ་ཡཎ། བ་ཧི་ཥ་ལས་ཡཎ་སྟེ་ཊི་དབྱིའོ། །བ་ཧིཥ་ལས་བུ་བརྒྱུད་ལ་ཡཎ་་འགྱུར་ཞིང་ཊིའང་དབྱི་སྟེ། བཱ་ཧུ། ཕྱིའི་བུ་འམ་འདི་ཕྱི་་རང་ལའང་འཇུག་གོ། །ཡང་འགྲེལ་པར་ལུང་དྲངས་པ། ཨཱི་ཀཀ། ཨི་ཀ་ཀའོ། །བ་ཧ་་་ཥལས་བུ་བརྒྱུད་ལ་ཨཱི་ཀ་ཀ་འགྱུར་ཞིང་ཊི་དབྱི་བ་འང་ངོ༌། །པཱ་ཧིཥ་ལ་ཨཱི་ཀ་ཀ་ཀྲ་འཕེལ་བའི་དོན་ཊི་ཕྱིས། བཱ་ཧཱི་ཀ། ཕྱིའི་བུ། ཨུཏྟཱ་དི་སྦྱོཉཉ། ཨུཏྤ་ལ་སོགས་པ་རྣམས་ལས་ཨཉའོ། །ཨུཏྤ་ལ་སོགས་པ་རྣམས་ལས་བུ་བརྒྱུད་ལ་ཨཉ་འགྱུར་ཏེ། ཨཉ་ཨོp། ཆུ་ལྷའི་བུ། ཨཉ་ཨོ་ད་:པཱ་ན?། བཏུང་བའི་ཆུའི་བུ། སི་པུཾ་སྭཱ་བྷཾ་ནཉ་སྣ་ཉོ་བྷ་བ་ནཱཏ། །ཙཱི་པཾ་ས་དག་ལས་དངོས་པོ་ལ་ནཉ་སྣཉ་དག་གོ། །སྟི་དང་པོ་ས་དག་ལས་དངོས་པོ་་་་བརྗོད་པ་ལ་རིམ་པ་བཞིན་ནཉ་སྣཉ་དག་འགྱུར་ཏེ། ན་སྡེ་ཎ། བུད་མེད། སྐྱེས་པ། པོ་སྣ༑ཡཔུཾ་ས་ལ་སྣཉ་འཕེལ་བ་ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས་ས་སྔོན་མ་ཕྱིས་པའོ། །དེ་རྣམས་བུ་་་བརྒྱུད་ཀྱི་རྐྱེན་རྣམས་སོ། །ཏེ་ན་རཀྟ་རཱ་གཱཏ། དེ་ཡིས་བཙོས་པའི་ཚོས་ལས་སོ། །ངུར་སྨྲིག་དང་གུར་ཀུམ་ལྟ་བུ་དེ་ཡིས་བཙོས་པའི་ཚོས་བརྗོད་པ་ལས་གསུམ་པའི་་་མཐའ་ལས་ཨཎ་འགྱུར་ཏེ། ཀ་ཥཱ་ཡེ་ན་རཀྟ། ངུར་སྨྲིག་གིས་བཙོས་པ། ཞེས་པའི་དོན་—རང་ལ་།
