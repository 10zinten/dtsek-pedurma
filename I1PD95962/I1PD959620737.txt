ཀྱི་ནས་བསྟན་འགྱུར། སྒྲ་མདོ། རྟའིཔ་མི་སྣ་ལོ་ཀ་ཚེར་ཎཡུལ་ཚོཌ་པ་བློ་•ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ལེ་ཟླུ། ལཱི་ཎཱཏ། ལཱི་ཏཱཾ། ལི་ཧནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ལཱི་རྟི། །ལཱི་བྷཱཏ། ལཱི་བྷཾ། ལཱི་ལྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ལེ་ཧཱ་ནི། ལེ་ཧཱ་བ།  ོལེ་ཧཱ་མ། དེ་བཞིན་དུ། ཏང་གི་་་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ལཱི་ཧཱུཾ། ལི་ཧཱ་ཏྭཾ། ལི་ཧ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་་་གསུམ་ལ། ལེ་གླུ། ལི་ཧཱ་ཐཱ། ལཱི་ནཱཾ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ལེ་ཧི། ལེ་ཧཱ་བ་ཧི། ལེ་ཧཱ་མ་ཧི། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་་ལེཌ༑ཨ་ལེཊ། །ཨ་ལཱི་ཏཱཾ། ཨ་ལི་ཧན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ལེཌ། ཨ་ལེཊ། ཨ་ལཱི་ནཱཾ། ཨ་ལཱི་ལྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ལེ་ཧི། ་ཨ་ལི་ཧྭ། ཨ་ལི་ཧྥ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ལཱི་ཌྷ། ཨ་ལི་ཧཱ་ཏྭཾ། ཨ་ལི་ཧ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ལཱི་ཌྷ། ཨ་ལི་ཧཱ་ཐཱ། ཨ་ལཱི་ནཱཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ལི་ཧི། ཨ་ལི་ཧཱ་ཧི། ཨ་་ལི་ཧྥ་ཧི། ལུང་གི་ཨ་ཏང་ལ། ་ཤལ་ཨི་ཀ་ཨུཔ་དྷ་ཨིཊ་མེད་པ་ལ་ཀླ་འོས། ཙི་ཀྐར་བསྒྱུར། ཧའི་ལྷ་འོས། །ཧ་ལྷ། ས་ལ་ཥ་ཌྷ་དག་གི་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། ཨ་ལི་ཀཱཏ། ཁྱི་ཨ་ལི་ཀཱ་ཏཱ། ཨ་ལི་་ཀན༑སོགས་དང༌། ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། དུ་ཧ། དི་ཧ། ལི་ཧ་སོགས་ཀྱིས་ཀླུ་དབྱི་བ་དང་མི་དབྱི་བ་ཡང་ན་བར་བྱས་པས། ཨ་ལི་ཀཱཏ། ཨ་ལཱི་ཌྷ། །གཉིས་ཚིག་ལ། །ཀིའི་ཨཙ་ལ་འོས་ཨ་ཕྱིས་པས། ཨ་་་ལིཀྵ་ཏྭཾ། མང་ཚིག་ལ་ཨ་དབྱི་བ་དང་མི་དབྱི་བ་ཡང་ན་བར་བྱས་པས། ་ཨ་ལི་གནྟ༑ཨ་ལི་ཀཱ་ཏ། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ཨ་ལི་ཀཱ་ཐཱ། ཨ་ལཱི་ཙྪིནྡྷ། གཉིས་ཚིག་ལ། ཨ་ལི་ཀཱ་ཐཱ། མང་ཚིག་ལ། ཨ་ལི་ཀཱ་ནཱཾ། ཨ་ལཱི་ནཱཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ལི་ཀཱི། ཨ་ལི་གླུ་བ་ཧི། —།
