རེ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནི་ནིནྟི་ཏེ། ནི་ནིནཱ་ཐེ། ནི་ནིརྨི་དེ། ནི་ནིནྟི་རྗེ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནི་ནིམྨེ། ནི་ནིནྟི། བ་ཧེ། ནི་ནིརྨི་མ་ཧེ། ་་ལུཊ་ལ་བལ་དང་པོའི་ཨིཊ་བྱིན་པས། །ནིརྨི་ཏཱ། ནིནྟི་ཏཱ་སེ། སོགས་སོ། །དེ་བཞིན་དུ། །སྨོན་པའི་ལིང་ལ་ཨིཊ་བྱིན་པས། ནི་ཊཱི། ནིནྟི་ཏི་ཡཱ་ཙཱ། ནིནི་ཏི་རན། སོགས་སོ། །ཁྲིཊ་ལའང་ཨིཊ་བྱིན་པས། ནིརྨི་ཉྩ་ཏེ། ནིརྨི་ཀྱེ་ཏེ། ནིརྨི་ཉྩནྟེ། སོགས་སོ། །ཕྲེང་ལ། ཨ་ནིནྡི་ཏ། ཨ་ནིཎི་བྱེ་ཏྭཾ། ཨ་ནིནྣི་བྷེན། སོགས་སོ། །ན་༈ཁྱུང་སྲོག་ཆགས་མངལ་ནས་སྐྱེ་བ་ལའོ་ཞེས་པ་ལའང་ཏང་རྣམས་སྦྱོར་ཏེ། བྱིངས་ཀྱི་དང་པོའི་ཥ་ས་དང་ཨ་ཙ་དང་པོ་ཅན་རྣམས་ལ། ཨ་ཙ་ལ་་སོགས་་་ཀྱིས་ཨུ་བ་རུ་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཡོསཱུ་ཏེ། །སུ་བཱ་ཏེ། སུ་བ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སཱུ་ཏེ། སུ་བ་ཐེ། ་སཱུདེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སུ་བེ། སཱུ་བ་ཧེ། ཧཱུཾ་མ་ཧེ། ལིང་ལ་ཨཏི་མ་ཡིན་པ་སོགས་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས། ཨུ་བ་སུ་བསྒྱུར་བས། སུ་བཱི་ཏ། སུ་བཱི་ཡཱ་ཏཱ། སུ་བཱི་རན། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་་་གསུམ་ལ། །སཱུ་ཏཱ། སུ་བཱ་ཏཱ། སུ་བ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སཱུ་༈་སུ་བཱ་ཐཱ། སཱུ་དྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སུ་བི། སུ་བཱ་བ་ཧི། སུ་བཱ་མ་ཧི། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སཱུ་ཏ། །ཨ་སུ་བཱ་ཏྭཾ། ཨ་སུ་བ་ཏ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སཱུ་ཐཱ། །ཨ་སུ་པཱ་ཐཱ། ་ཨ་སཱུ་དཾ༑་བྱེསྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ་་ཨ་སུ་བི། ཨ་སཱུ་བ་ཧི། ཨ་སཱུ་མ་ཧི། ལུང་ལ། སཱ་ར་ཏི། སཱུ་ཏི། སཱུ་ཡ་ཏི། དྷུཉ་ཨུཏ་འགྲོ་བ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་བྱ་སྟེ། ཨིཊ་བྱིན་པའི་ཕྱོགས་ལ། ཨ་ས་བིཊྚ། ཨ་ས་བི་ཥཱ'ཏྭཾ། jo།
