ལ༑་ཡུ་ཡཱ་བ། ཡུ་ཡ་བ། ཡུ་ཡུ་བི་བ། ཡུ་ཡུ་བི་མ། ལུཊ་ལ་བལ་དང་པོའི་ཨིཊ་སྦྱར་བས། ་ཡ་བི་ཏཱ། ཡ་བི་ཏཱ་རོ། ་ཡ་བི་ཏཱ་ར། ཡ་བི་ཏཱ་སི་་་་སོགས་སོ། །སྨོན་པའི་ལིང་ལ་ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་པས། ཡུ་ཡཱཏ། ཡུ་ཡཱནྟཾ། ཡུ་ཡཱ་སུསོགས་སོ། ་། ཁྲིཊ་ལ་ཨིཊ་བྱིན་པས། ཡ་བི་ཥྱ་ཏི་སོགས་དང་། དེ་བཞིན་དུ། ལིང་ལ། ཨ་ཡ་བི་ཏ་སོགས་སོ། །་ཁོཀྑ་་༈ཆུ་བསྟོད་པ་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་རྣམས་སྦྱོར། ཆོག་ཡུ་དང་འདྲ་སྟེ། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ནོ་ཏི། ནུ་ཏཿ། ནུ་བནྟི། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ནོ་ཎི། ་ནུ་ཐ། ནུ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནོ་མི།  ོ་ནུ་བ། ་ནུ་མ། །ལིང་ལ། ནུ་ཡཱཏ། ནུ་ཡཱ་ཏྭཾ། ནུ་ཡུསོགས་སོ༑། ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ནོ་ཏུ། ནུ་ཏཱཏ། ནུ་ཏྭཾ། ནུ་་་བནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནུ་ཧི། ནུ་ཏཱཏ། ནུ་ཏཾ། ནུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ན་བཱ་ནི། ན་བཱ་བ། །ན་བཱ་མ། ལང་གི་སྐྱེས་བུ་་་དང་པོ་གསུམ་ལ། །ཨ་ནོཏ། ཨ་ནུ་ཏཱ། ཨ་ནུ་བན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ནོ། །ཨ་ནུ་ཏཾ། ཨ་ནུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ནུ་ཐཾ། ཨ་ནུ་བ། ཨ་ནུ་མ། ལུང་ལ་ཝལ་དང་པོའི་ཨིཊ་དང་། སིཙ་འཕེལ་བ་པ་རསྨི་པ་ད་རྣམས་ལ་འོས་འཕེལ་བ་བྱས་པས། ཨ་ནཱ་བཱིཏ། ཨ་ནཱ་བིདྡྷཾ། ཨ་ནཱ་བི་ཥཱསོགས་སམ། །ཡང་ན་འགྲེལ་པར་ནུ་ལས་བལ་དང་པོ་ལ་ཨིཊ་ཡང་ན་མི་འགྱུར་རོ་ཞེས་བཤད་པས་ཨིཊ་མེད་པའི་ཕྱོགས་ལ་འཕེལ་བ་བྱས་པས། ཨ་ནོ་ཊྛིཏ། ཨ་ནོཏྟཾ། ཨ་ནོ་ཥཱསོགས་སོ། །ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ནུ་ནཱ་བ། ནུ་ནུ་བ་ཏུ། ནུ་ནུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནུ་ན་བི་ཐ། ནུ་ནུ་བ་ཐུ། ནུ་ནུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནུ་ནཱ་བ། ནུ་ན་—།
