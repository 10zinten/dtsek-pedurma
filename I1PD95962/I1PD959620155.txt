:་་་་་་་:བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཉི། ་ཡིན་པ་གོ་མཚོན་སློབ་རྒྱུ་ལ་དབྱེ་ན་བས་དགའ། ཀོKབསྒྲུབས་པས། ཀྲོཊྚ། རོར་འགྱུར་བའི་ཕྱོགས་ལ། རཾ་ཏ་ཨུར། །རཾ་ཏ་་་་ལས་ཨུརའོ། །རྟགས་རིཏ་མཐའ་ཅན་ལ་ཕ་རོལ་རྣམ་དབྱེ་ལྔ་པའི་ང་སི་དང་། དྲུག་པའི་ངས་དག་གི་ཨ་དང་རྟགས་མཐའི་རི་གཉིས་འདུས་པར་ཨུར་དུ་འགྱུར་ཏེ། ཀྲོཌཱ༑ཨུར་སྲ། ཞེས་པ་ལ། རཱཏྨ་སྱ། ར་ལས་སའིའོ། །ར་ཡིག་སྲོག་་མེད་ལས་ཕ་རོལ་གྱི་ས་ཡིག་སྲོག་མེད་ནི་དབྱི་བར་བྱ་སྟེ། ས་ཕྱིས། ར་རྣམ་བཅད་བྱས་པས། །ཀྲོཥྛ། བྷཱུམ་བྱས་ལ། ཀྲོཉྩ་ཧཱུཾ། ཀྲོལྦུ་བྷྱ། དྲུག་པའི་ངས་ཡང་བྱ་བ་ལྔ་པའི་ང་སི་དང་འདྲ་བས། ཀྲོཌོ། ཀྲོཟླུ། ་ཨོས་ལ། ཀྲོཟློ༑ཀྲོསྒྲོ། ཀྲོསྒྱུ་ལ་དྲུག་པའི་ཨཱམ་ཤཾ་བྷུ་ལྟར་བསྒྲུབས་པས། ཀྲོསྠཱ་ནཾ། ལུགས་གཅིག་ལ། ནུ་མ་ཙི་ར་ཏྲ་ཛཱ་དྷཱ་བེ་བློ་ནུཊ། ར་ཨཙ་ལ་ནུམ་ཏཱིཙ་བཞིན་འགྱུར་བ་རྣམས་ཀྱི་ནུཊའོ། །རྟགས་མཐའ་ར་ཡིག་ཨཙ་དང་ལྡན་པ་ནུམ་ཨཱ་ག་མ་ཐོབ་པ་རྣམས་དང་ཀྲོ་ཚིགསཙ་བཞིན་དུ་འགྱུར་བའི་ཕ་རོལ་རྣམ་དབྱེ་དྲུག་པའི་་་མང་ཚིག་ཨཱམ་ལ་ནུཊ་ཨཱ་ག་མ་ངེས་པར་འཇུག་སྟེ། ནུཊ་ཨཱ་ག་མ་བྱིན་པས། ཀྲོདྡྷཾ་ཏཱཾ། རྣམ་དབྱེ་བདུན་པའི་ང་ལ། ཀྲོཋ་རི། །ཀྲོཊོ། །ཨོས་ལ། ཀྲོཟློ། །ཀྲོཟློ༑སུཔ་ལ། ཀྲོབྷུ་ཥཱ། བོད་ཚིག། ཧེ་ཀྲོཌོ། ཧེ་ཀྲོཏྟཱ་རོ། ཧེ་་་་ཀྲོཏྟཱ་རཿ། གཉིས་པའི་ཤས་ལ་ཡང་ན་ཀྲོཙྪིན་ཞེས་པ་འང་འབྱུང་ངོ༌། །དེ་་་ཡན་གྱིས་སྐྱེས་པའི་རྟགས་ཀྱི་ཨུ་ཐུང་མཐའ་ཅན་སྤྱིར་བཏང་དང་དམིགས་བསལ་་་རྣམས་བཤད་ཟིན་ཏོ། །ཁོག་ཏུ༌བཞི་ཁ་བ་ཅན་ལ་སོགས་གྲགས་༈ད་ནི་སྐྱེས་པའི་རྟགས་ལས་སྤུ་རིང་མཐའ་ཅན་གྱི་བྱ་བ་རྣམས་སྟོན་པར་བྱེད་དེ། ཧུ་ཧུ་ཞེས་པའི་སྒྲ་ལ་དཔེར་བརྗོད་དུ་བྱ་སྟེ། དབྱངས་མེ་འདྲ་བའི་་འགྱུར་ཁྱད་ཕྲན་བུ་མ་གཏོགས་བཱ་ཏ་པྲ་མཱི་དང་འདྲ་བས། ཧུ་ཧུ་ལ། རྣམ་་་་དབྱེ་དང་པོ་རྣམས་བྱིན། གོང་ལྟར་གྱི་ཆོ་གས་བསྒྲུབས་པས། ཧུ་ཧཱུ། དྲི་ཟ། —བསལ་།
