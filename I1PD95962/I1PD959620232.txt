སུ་སཾ་སུ་དྡྷཾ་སུ་ཨ་ན་ཌཱུཧ་རྣམས་ཀྱི་ད་འོས་ས་ད། བི་དྭདྡྷཾ། བི་དྭབྷི། ངེ་ལ། བི་དུ་ཏེ། །བྷཱུམ་བྱས་འདྲ། །ང་སི་ལ། །བི་དུ་ཥ། བྷཱུམ་བྱུས་འདྲ། དྲུག་པ་རྣམས་ལ། བི་དུ་ཥ། བི་དུ་ཁོ། བི་དུ་ཏཱཾ། བདུན་པ་རྣམས་ལ། བི་དུ་ཏི།  ོབི་དུ་ཁོ། བི་དྭཏུ། བོད་ཚིག་རྣམས་ལ། ཧེ་བི་དཱན། །ཧེ་བི་དྷཱཎོ། །ཧེ་བི་དཱམྨ། བས་བ་ག་ཊ་ལཊ་ཀུ་བཏབ་དང་། དེ་:༈ས་མཐའ་ཨུ་རྗེས་འབྲེལ་ཅན། པེ་ཙི་ཝ་སུ་ལ། སུ། ཨུཀ་འགྲོ་བ་་་སོགས་ཀྱིས་ནུམ། སུ་ཕྱིས། །ཡང་དག་པར་སྦྱོར་བའི་མཐའ་ཕྱིས། འོསཾ་བུདྡྷི་་མ་ཡིན་པ་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་རིང་པོ། པེ་ཙི་ཝཱན། ཨོ་དང་ཛས་ལ། པེ་ཅི་ཝནྣཾ། པེ་ཙི་ཝནྣ། ་ཨམ་དང་ཨོ་ལ། :པེ་ཙི་ཝནྣཾ། པེ་ཙི་ཝནྣཾ། ཤས་ལ། ཝས་ཀྱི་སཾ་པྲ་སཱ་ར་ཎ་འོས། ལྔ་ཨུ། །སཾ་པྲ་སཱ་ར་ཎ་ལས་ཀྱང་ངོས། ཨ་སྔ་མའི་གཟུགས་སུ་སོང་། ཝས་ཨུ་རུ་འགྱུར་ན་ཨིཊ་འཇིག་གོ་ཞེས་པ་འགྲེལ་པར་བྱུང་་་བས༑ཙའི་སྲོག་གི་ཨི་ཕྱིས། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས། ས་ཥ། པེ་ཙུ་ཥེ། ཊཱ་ལ། པེ་ཙུ་ཥཱ། བྷཱུམ་བྷིས་ལ། ཝ་སུ་སཾ་སུ་སོགས་ཀྱིས་ས་ད། །པེ་ཙི་ཝདྡྷཾ། པེ་ཙི་ཝདྡྷི། ངེ་ལ། པེ་ཙུ་ཥ། སྦྲུམ་བྱས་འདྲ། ང་སི་ལ། པེ་ཙུ་ཥ། །བྷཱུམ་བྱས་འདྲ། །དྲུག་པ་རྣམས་ལ། པེ་ཙུ་ཥ། །པེ་ཙུ་ཊོ། པེ་ཙུ་ཏྭཾ། བདུན་པ་རྣམས་ལ། པེ་ཙུ་ཏི། པེ་ཙུ་ཁོ། པེ་ཙུ་ཝནྟུ། བོད་ཚིག་རྣམས་ལ། ཧེ་པེ་ཙི་ཝན། ཧེ་པེ་ཙི་ཝནྣཾ། ཧེ་པེ་ཙི་ཝནྣ། ་་དང་། འགྱུར། ༈ས་མཐའ་ཅན། ཛ་ག་མཱ་སུ། ལ་སུ། སྦོཙ། ་མ་བ་དག་ལ་འང་ངོ༌། །མ་ཡིག་དང་བ་ཡིག་དག་གི་རྐྱེན་ཕ་རོལ་ལ་བྱིངས་ཀྱི་མ་ཡིག་ནི་ན་རུ་་་་འགྱུར་ཏེ། །མ་ན། ཛ་ག་ནཱ་སུ། སུ། ཨུཀ་འགྲོ་བ་སོགས་ཀྱིས་ནུམ། ཨུ་དེའི་དོན། །ཧལ་སོགས་ཀྱིས་སུ་ཕྱིས། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་ཕྱིས། ICO།
