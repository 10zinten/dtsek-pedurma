མ་བཟང་མོམཱ་ཏྲ་ལས་ཨུཏའོ། །གྲངས་དང་སམ་དང་བྷ་དྲ་རྣམས་སྔ་མའི་མཱ་ཏཱཾ་ལས་བུ་་་བརྒྱུད་ལ་ཨཎ་འགྱུར་ཞིང་ཏྲའི་རཾ་ཨུཏ་དུ་འགྱུར་ཏེ། །དཱི་མཱ་ཏྲ་ལ་ཨཎ་དང་པོ་འཕེལ་བ་རཾ་ཨུཏེ། རྀའི་ཨཎ་ལས་ར་ཕ་རོལ་ལོས་ར་ཨཱ་གམ། དཱི་མཱ་ཏུ་ར། མ་གཉིས་ཀྱི་བུ། ཥཱཎཱ་ཏུ་ར། མ་དྲུག་གི་བུ། སཾ་མཱ་ཏུ་ར། ཀུན་དུ་མའི་བུ། བྷ་དྲ་མཱ་ཏུ་ར། མ་བཟང་མོའི་བུ། ཀ་རྩྭ་ཡཱཀ་ནི་ན་ཙ། ཀ་རྩྭ་གྱི་ཀ་ནཱི་ནའང་ངོ༌། །ཀ་རྩྭ་ལ་བུ་བརྒྱུད་ལ་ཨཎ་འགྱུར་ཞིང་ཀ་རྩྭ་ནི་ཀ་ནཱིན་ཞེས་པར་འགྱུར་ཏེ། ཀཱ་ནཱི་ན། བུ་མོའི་བུ་སྟེ་ནལ་བུ། སི་སྦྱོ་ཆ་ཀ། བུད་མེད་རྣམས་ལས་ཌྷ་ཀའོ། །བུད་མེད་ཀྱི་རྐྱེན་མཐའ་ཅན་རྣམས་ལ་བུ་བརྒྱུད་ལ་ཌྷ་ཀ་འགྱུར་ཏེ། ཙནྡྲ་བར། །དཱིཔ་ཨཱཔ་ཏི་ཨུད་ལས་སོ་ཞེས་པ་དང་གཅིག་གོ། །བི་ན་ཏཱ་ཡཱཨ་པ་ཏུཾ༑ཞེས་པ་ལ། །ཆ་ཀ། ཀ་འཕེལ་བའི་དོན། ཌྷ། ་ཨེ་ཡར་འགྱུར། བི་ན་ཏེ་ཡ། .འདུལ་བྱེད་མའི་བུ། དེ་བཞིན་དུ། མ་ཧི་ལ་ཞཀ། མཱ་ཧེ་ཡ། ས་ཡི་བུ་སྟེ་མིག་་་དམར། །ན་ཆུང་མའི་བུ། ཡོ་བ་ཏེ་ཡ། ཡུ་བ་ཏི་ལ་རྟཀ། ཤུ་བྷཱ་དི་བྷུཉྩ། ཤུ་བྷ་ལ་སོགས་་པ་རྣམས་ལས་ཀྱང་ངོ༌། །ཤུ་བྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་བུ་བརྒྱུད་ལ་རྙཀ་་་འགྱུར་ཏེ། ཤོ་བྷེ་ཡ། དཀར་པོའི་བུ། གངྒེ་ཡ། གང་གའི་བུ། རྟ་ལོ་པོཉཀ་། ཀ་དྷུ་མ་ཡིན་པའི་ལྷ་ལ་དབྱིའོ། །དེ་ཕན་གྱི་རྐྱེན་ལྷ་ཕ་རོལ་ལ་ཀ་དྷུ་མ་ཡིན་པའི་་་མཐའི་ཨུ་ཡིག་དབྱི་སྟེ། །མཎྜ་ལུ་ལ་ཤུ་བླ་སོགས་ཀྱིས་ལྷ་ཀ་མདོ་འདིས་ཨུ་ཕྱིས་རྟ་ཨེ་ཡ་སུ། ཀཱ་མཎྜ་ལེ་་ཡ་ཡ། རིལ་བ་ཅན་གྱི་བུ། ཀ་དྲུ་ལ་མི་དབྱི་སྟེ། ་ཀ་དྲུ་ལ་ཤུ་བྷ་སོགས་ཀྱིས་རྟཀ་འཕེལ་བ། རྟ་ཨེ་ལ་་ཨུཏ་ཡོན་ཏན། ཀཱ་དྲ་བེ་ཡ། གཟར་བུ་མའི་བུ། འགྲེལ་པར་ལུང་དྲངས་པ། བརྗོ༔་ན། པཎྜའི་འང་མ་ཡིན་ནོ། །ཌྷའི་རྐྱེན་ཕ་རོལ་ལ་པཎྜའི་ཨུ་ཡང་མི་དབྱི་སྟེ། པཎྜཱུ་ལ་ཤུབྦ་སོགས་ཀྱིས་རྟཀ་རྟ་ཨེ་ཡ་ཨུ་ཡོན་ཏན། པཱཎྜ་བེ་ཡ། སྐྱ་བོའམ་སྐྱ་སངས་ཀྱི་བུ། ཨི་ཏ་ནི་ཉཨིཏ་་་ལས་ཀྱང་ཨིཉ་མ་ཡིན་པ་ལས་སོ། །།ཨཏ་ལས་ཨིཉ་འོ་ཞེས་པ་ལྟ་བུའི་དེ་ཕན་གྱི་རྐྱེན་ཨིཉ་བྱིན་པའི་ཨིཏ་མཐའ་མ་ཡིན་པའི་ཨི་ཡིག་མཐའ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་་་།
