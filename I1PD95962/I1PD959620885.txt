སྒྲ་མདོ། ཏི་ཕྱིས་མི་ནོན་ཁོ་བོའིདངོས་ཟླ་བ་བློ་ཌདག་གི་ཙུ་འོས། ཀ་ཙར་བསྒྱུར་བས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་སྐམྦྷ། ཙ་སྐམྦྷ་ཏུ། །ཙ་སྐམྦྷ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་སྐམྷི་ཐ། ཙ་སྐམྷི་ཐུ༑ཙ་སྐནྡྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་སྐནྡྷ། །ཙ་སྐམྦྷི་བ། །ཙ་སྐམྷི་མ། །ལུཊ་ལ་ཨིཊ་བྱིན་པས། སྐབྷཱི་ཏཱ། སྐམྦྷཱི་ཏཱ་སི། །སོགས་སོ། །སྨོན་་པའི་ལིང་ལ། ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་མ་ཕྱིས་པས། སྐ་བྷཱུ་ཏ། སྐབྷཱུཙཱ༑སྐབྷཱ་སུ། །སོགས་སོ། །ཁྲིཊ་ལ་ཨིཊ་བྱིན་པས། སྐམྷི་ཥྛ་ཏི། སོགས་དང་། དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་སྐམྷི་ཥྱ་ཏ། །སོགས་སོ། །བེསྐ་བྷཱ་ཏེནི་ཏཾ། བི་ལས་སྐ་བྷཱ་ཏིའི་རྟག་ཏུའོ། །ཉེར་བསྒྱུར་བའི་ལས་ཕ་རོལ་གྱི་སྐམྦྷ་བྷཱའི་རྐྱེན་ཅན་གྱི་ས་ཡིག་ནི་རྟག་ཏུ་ཥ་ཡིག་ཏུ་འགྱུར་ཏེ། བིཥབྷཱུ་ཏི། སོགས་་དང༌། བྷཱའི་རྐྱེན་ཅན་མ་ཡིན་ན་མི་འགྱུར་ཏེ། བིསྐདྡྷོ་ཏི། སོགས་སོ། །།།༈ྋསྐུམྦྷ་སྒྲིབ་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། བྷ་དང་ཀླུ་ཡང་ན་བར་བྱས། །ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་མ་ཕྱིས་པས། ལཊ་ལ། སྐུ་བྷཱ་ཏི། སོགས་དང་། སྐུདྡྷོ་ཏི། སོགས་སོ། །ལིང་ལ། སྐུ་ཙྪིཡཱཏ། སོགས་སམ། སྐུ་་ཡཱཏ། །སོགས་སོ། །ལོཊ་ལ། སྐུ་བྷཱ་ཏུ། སོགས་སམ། སྐུ་དྡྷོ་ཏུ། སོགས་སོ། །ལང་ལ། ཨ་སྐུདྡྷཱ་ཏི། སོགས་སམ། ཨ་སྐུ་དྡྷོ་ཏ། །སོགས་སོ། །ལུང་ལ་ཨིཊ་བྱིན་པས། ཨ་སྐུམྦྷི་ཏ། ཨ་སྐུརྞཱི་ཙྪཱཏྟཾ། ཨ་སྐུམྷི་ཥཱ། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་ཤར་དང་པོ་སོགས་ཀྱིས་ས་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙུ་སྐུམྦྷ། ཙུ་སྐུམྦྷ་ཏུ། ཙུ་སྐུམྦྷ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙུ་སྐུམྷི་ཐ། ཙུ་སྐུམྦྷ་ཐུ། ཙུ་སྐུམྦྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙུ་སྐུམྦྷ། ཙུ་སྐུམྷི་བ། ཙུ་སྐུམྷི་མ། ལུཊ་ལ་ཨིཊ་བྱིན་པས། །སྐུརྞཱི་ཏཱ། སྐུམྦྷཱི་ཏཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་་དྷཱ།
