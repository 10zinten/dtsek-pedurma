དྷཱ་ཏཱཾ་ཏཱཾ། བདུན་པའི་ང་ལ་རྀཏ་ཀྱི་ང་ཐམས་ཅད་གནས་པ་ཞེས་པས་ཡོན་ཏན། དྷཱ་ཏ་རི། ྋདྷཱ་ཏྲི། །དྷཱ་ཏྲི་ཥཱ། །ཀུན་ཏུ་བོད་པའི་སུ་ལ། རྀཏ་ཀྱི་ང་ཐམས་ཅད་གནས་པའི་མིང་ཅན་ཞེས་པས་ཡོན་ཏན། ཧེ་དྷཱ་ཏ། གཉིས་ཚིག་དང་་་མང་ཚིག་དང་པོའི་སྐབས་དང་འདྲ། །ཧེ་དྷཱ་ཏཱ་རོ། ཧེ་དྷཱ་ཏཱ་ར། དེ་བཞིན་དུ། ན་བྷཾ། ཚ་བོ། ཧོ་ཏྲཿ། སྦྱིན་སྲེག། པོ་ཚིགས་ྋསྦྱིན་སྲེག་སོགས་མང་པོ་ཞིག་འདི་དང་འདྲའོ༑། པི་ཏྲི་ལ། ཕ། སུ། གོང་བཞིན་བསྒྲུབས་པས། པི་ཏཱ། :པི་ཏྲའི་སྐབས་འདིར་གཞན་ཆོ་ག་ཐམས་ཅད་སྔ་མ་དྷཱ་ཏྲིའི་སྐབས་དང་འདྲ་ལ། དང་པོའི་ཨོ་ནས་གཉིས་པའི་ཨོའི་བར་རྣམས་ལ་ཨུ་པ་དྷཱ་རིང་པོར་མི་འགྱུར་ཏེ། དྷཱ་ཏྲ་ཞེས་པ་རྐྱེན་ཏྲཙ་ཛྲ་མཐའ་ཅན་ཡིན་ལ། པི་ཏྲ་ཞེས་པ་ཙ་མཐའ་ཅན་མ་ཡིན་པས་སོ། །པི་ཏ་རོ། པི་ཏ་རཿལྟ་བུའོ། །གཞན་རྣམས་རིགས་འགྲེའོ། །ཛཱ་མཱ་ཏྲི། མག་པ། བྷཱ་ཏྲི་སྤུན་ཟླ་སོགས་ཀྱང་འདི་དང་འདྲའོ། །ཚྭ་ཙ། ནཱིའི་ཡང་ནའོ༑རྫོཊ་། ཚིགས་ྋམི་ཞེས་པའི་རྟགས་ལ་རྣམ་དབྱེ་དྲུག་པའི་ཨཱམ་མ་གཏོགས་པི་ཁྲིའི་སྐབས་དང་འདྲ། ཨཱམ་ལ་རྟགས་མཐའ་རིང་པོར་འགྱུར་བ་དང་མི་འགྱུར་བ་་་་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། ཚྭ་ཎཱ་འམ། ཡང་ན་ཚྭ་ཚྭ་ལྟ་བུའོ། །དེ་རྣམས་ཀྱིས་སྐྱེས་པའི་རྟགས་ལ་རཾ་ཡིག་མཐའ་ཅན་གྱི་བྱ་བ་རྣམས་བཤད་ཟིན་ཏོ། །དབང་། ཀ༈སྐབས་འདིར་རཱིཏ་དང་ལ༹ཏ་དང་ཨེཏ་རྣམས་ཀྱི་མཐའ་ཅན་གྱི་རྟགས་་་མེད་པར་གྲགས་སོ། །།bབྱ་བ་མང་པོ་ལ་ཟླ་བས། ད་༈ཨོ་ཡིག་མཐའ་ལས་གོའི་སྒྲ་ལ་དཔེར་བརྗོད་དུ་བྱ་སྟེ། །གོ་ཏོ་ཎིཏ། གོ་ལས་ཎིཏའོ། །ཨོ་མཐའ་ཅན་གོ་ལྟ་བུའི་ཕ་རོལ་གྱི་རྣམ་དབྱེ་དང་པོ་ནས་་་་གཉིས་པའི་ཨོཊའི་བར་སྔར་ཐམས་ཅད་གནས་པའི་མིང་ཅན་ཞེས་བརྡ་ཆད་བྱས་་་པ་དེ་རྣམས་ལ་སྐབས་འདིར་ཎ་འགྲོ་བའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱེད་དོ། །།
