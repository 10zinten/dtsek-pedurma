*པས༑་ལྷ་མགྷ་ཨཎི། མཱ་གྷ། མཆུའི་ཟླ་བ། ་མ་གྷ་ལ་ཨཎ་དཱིཔ། མཱ་གཱི། མཆུས་ཉ་བ། ཕལྷུ་ཎ་ལ་ཨཎ། དབོའི་ཟླ་བ། པཱལྷ་ཎ། ཨཎ་དཱིཔ། པཱལྷ་ཎཱི། དབོ་ཡིས་ཉ་བ། ཙི་ཏྲལ་ཨཎ། ཙི་ཏྲ། ནག་པའི་ཟླ་བ། ནག་པས་ཉ་བ། ཙི་ཏི། ཨཎ་དཱིཔ། ་ས་གའི་ཟླ་བ། བི་ཤྭ་ཁ། བི་ཤྭ་ཁ་ལ་ཨཎ། ས་གས་ཉ་བ། བི་་་ཤཱཁཱི༑ཡཨཎ་དཱིཔ། སྣོན་གྱི་ཟླ་བ། ཛིཥྛ། ཛི་ཏྟཱ་ལ་ཨཎ། སྔོན་གྱིས་ཉ་བ། ཛིཛིཏྟཱ་ལ་ཨཎ་དཱིཔ། ཆུ་སྟོད་ཀྱི་ཟླ་བ། ཨཱ་ཥཱ་ཌྷ། ཨ་ཥདྷ་ལ་ཨཎ། དཱིང་ཨ་ཥ་དྷ་ལ་ཨཎ། ཨཱ་ཥཱ་ཎཱི། ཆུ་སྟོད་ཀྱིས་ཉ་བ། གྲོ་བཞིན་གྱི་ཟླ་བ། ཤབ་ཎ། ཤྲ་བ་ཎ་ལ་ཨཎ་དཱིཔ། གྲོ་བཞིན་གྱིས་ཉ་བ། ཤྲ་བ་ཎི། ཤྲ་བ་ཎ་ལ་ཨཎ་དཱིཔ། ཁྲུམས་སྟོད་ཀྱི་ཟླ་བ། བྷ་དྲ་པ་ད། བྷ་དྲབ་ད་ལ་ཨཎ། ཁྲུམས་སྟོད་ཀྱིས་ཉ་བ། བྷ་དྲ་པ་དཱི། བྷ་དྲ་བ་ད་ལ་ཨཎ་་དཱིཔ། ཐ་སྐར་གྱི་ཟླ་བ། ཨཱ་ཤི་ན། ཨ་ཤི་ནི་ལ་ཨཎ། ཐ་སྐར་གྱིས་ཉ་བ། ཨཱ་ཤཱི་ནཱི། ཨ་ཤི་ནི་ལ་ཨཎ་དཱིཔ། སྨིན་དྲུག་གི་ཟླ་བ། ཀཱརྟི་ཀ། ཀྲྀ་ཏི་ཀ་ཨཎ་དཱིཔ། ཀཱརྟི་ཀཱི། སྨིན་དྲུག་གིས་ཉ་བ། །མགོའི་ཟླ་བ། མཱརྒ་ཤི་རཿ༈བློ་ག་ཤི་ར་ལ་ཨཎ་དཱིཔ། མྲྀ་ག་ཤི་ར་ལ་ཨཎ་དཱིཔ། མཱརྒ་ཤི་རཱི། མགོ་ཡིས་ཉ་བ། བྲྀཏྟཾ་སཱ་མ། མཉམ་པར་ལྟ་བ་ལའོ། །མཉམ་པར་ལྟ་བའི་དོན་ལ་གསུམ་པའི་མཐའ་ལ་ཨཉ་་་འགྱུར་ཏེ། ང་བ་ཤིཥྛ་ན་དྲྀཏྟཾ། གནས་འཇོག་གིས་བལྟ། ཞེས་པའི་དོན་ལ། ཨཉ་ཉ་འཕེལ་་་བའི་དོན། བཱ་ཤིཥྛ། གནས་འཇོག་གིས་བལྟ། པ་རི་ཧྲི་ཏོ་རཐ། ཤིང་རྟ་ཡོངས་སུ་དཀྲིས་པའོ༑g། ཤིང་རྟ་ཡོངས་སུ་དཀྲིས་པ་ཞེས་པ་འདིའི་དོན་ལ་བསྡུ་དང༌། ཙརྨ་ལ་ཨཉ་འགྱུར་ཏེ། །བསྲེ་ཎ་པ་རི་དྲྀ་ཏོ་ར་ཐ། ཤིང་རྟ་གོས་ཀྱིས་ཡོངས་སུ་དཀྲིས། ཞེས་པའི་་དོན་ལ། ་གོས་ཀྱིས་དཀྲིས། བཱསྟཱ། ཨཉ་ཉ་འཕེལ་བ། དེ་བཞིན་དུ། །ཙཱརྨ་ཎ། པགས་པས་དཀྲིས། ཏ་ཀྲོཎྜི༌ཏཱ་པཱ་ཏྲེ་བྷྱ། སྣོད་རྣམས་ལ་དེར་བླུགས་པ་ལའོ། །སྣོད་རྣམས་ལས་སྣོད་དེར་བླུགས་པ་ལ་བདུན་པའི་མཐའ་ལ་ཨཉ་འགྱུར་ཏེ། ས་ར་བེ་ཨུདྡྷཾ་ཏཾ། རྫ་ཕོར་དུ་བླུགས་པ། ཞེས་པའི་དོན་ལ། རྫ་ཕོར་དུ་བླུགས་པ། སཱ་ར་བ། ཨཉ་ཉ་འཕེལ་བ། སཾ་ཀྲྀ་ཏཾ་བྷཀྵ། ཟས་ལེགས་པར་སྦྱར་བ་རྣམས་སོ། །ཟས་ལེགས་པར་སྦྱར་བ་་་བརྗོད་པ་ལས་བདུན་པའི་མཐའ་ལ་ཨཉ་འགྱུར་ཏེ། :ཨཉ་འཕེལ་བ། བྷཱབྷཱབཙོས་པ། —།
