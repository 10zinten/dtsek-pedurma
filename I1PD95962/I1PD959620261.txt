ཀུ༌ཀ་ཊི་ལ་སྒོ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིཎི་སོ་ལྟ་ཁོ་མཆི་མ་ཎ་ཁ་ལ་བློ་ཁ་ཁ་ཕྱོགས་ཀྱི་པཎྜི་ཏ་རྣམས་ཨཱ་ཕྱིས་པའི་ཤུལ་ཡ་ཡིག་དང་ཀ་ཡིག་ཀ་དང་བཅས་པའི་་ཨཱབ་ཕ་རོལ་ལ་ཨ་ཨིར་འགྱུར་བ་དང་མི་འགྱུར་བ་འདམ་ཀ་ཅན་དུ་འདོད་པས། ཨཱཏྨ་ཀཱ། འཕགས་མ། ཨཱངྒཱི་ཀཱ། ཙ་ཊ་ཀ་ཀཱ། བྱེའུ་མོ། ཙ་ཊ་ཀི་ཀཱ། ཡང་འགྲེལ་པར། སཾ་ཀཱ་ཤྩ་ཀཱ་སོགས་ཨིར་འགྱུར་བ་ཁོ་ནའོ་ཞེས་པ་བྱུང་བས། སཾ་ཀཱ་བྱི་་་ཀཱ༑ཀཤྲཱི་བུད་མེད། ཨ་ཤྭ་ཀཱ། རྒོད་མ། ཤུ་བྷ་ཡི་ཀཱ། མཛེས་མ། སུན་ཡི་ཀཱ། ཤིན་ཏུ་མཛེས་མ། སུ་པཱ་ཀི་ཀཱ། གཡོས་བཟང་མ། »འདི་རྣམས་ལ་ཡང་ན་བ་མི་བྱེད་དོ། །བྷནྡཱི་ཥཱ་ཛཱ་་་ཙཱ་རཱ་སཱ་ནཉ་ོཔཱུཏྲཱ་ཎཱ་མ་པ། བྷསྔ་ཨེ་ཥཨ་ཛཛྷ་དྭ་སྭ་རྣམས་ལ་ནཉ་སྔ་མ་ལས་་་ཀྱང་ཡང་ནའོ། །བྷསྡུ་ལ་སོགས་པ་དེ་རྣམས་ལ་ནཉ་སྔོན་ན་ཡོད་མེད་གང་་་་ཡིན་ཀྱང་ཨ་ཨིར་འགྱུར་བ་དང་མི་འགྱུར་བ་འདམ་ཀ་ཅན་ནོ། །བྷནྡྲ་ཀཱ། ཁོལ་མོ། བྷཉྫི་ཀཱ། ཨེ་ཥ་ཀཱ། ཨེ་ཎི་ཀཱ། ་ཨ་ཛ་ཀཱ། ར་མ། ཨ་ཛི་ཀཱ། ཛཱཀཱ། ཤེས་་་མ༑ཛྷིཀཱ། དྭ་ཀཱ། གཉིས་མ། དྲི་ཀཱ། སཱ་ཀཱ། སི་ཀཱ། ཨ་བྷནྡྲ་ཀཱ། ཨ་བྷསྟི་ཀཱ། ལྟ་བུའོ། །ཡང་དེ་རྣམས་ཀྱི་ནང་ནས་ཀྱང་ནཉ་སྔ་མའི་ཨེ་ཥ་ཀཱ་ནི་ཨིར་མི་འགྱུར་རོ་ཞེས་འགྲེལ་པར་བྱུང་བས། ཨ་ནེ་ཥ་ཀཱ་ཞེས་པ་ཉིད་དུ་གནས་སོ། །འགྲེལ་པར། བྷསྔ་ཀཱ་ཞེས་པའི་སྔོན་དུ་ཉེར་བསྒྱུར་གང་བྱུང་ཡང་ཨིར་འགྱུར་བ་་་དང་མི་འགྱུར་བ་འདམ་ཀ་བྱེད་དོ་ཞེས་པ་བྱུང་བས། ནིབྷནྡྲ་ཀཱ། ནིབྷསྟི་ཀཱ། ཨ་བྷཱ་ཥི་ཏ་པུཾ་སྐུཙྪ། སྐྱེས་པ་ལས་མ་བཤད་པ་ལ་ཡང་ནའོ། །སྔར་སྐྱེས་པའི་་་རྟགས་ཀྱི་སྐབས་སུ་མ་བཤད་པ་རྣམས་ལ་ཡང་ཨ་ཨིར་འགྱུར་བ་དང་མི་འགྱུར་བ་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། གཾ་ག་ཀཱ། ཧཾ་གི་ཀཱ། ཨ་གཾ་ག་ཀཱ། ཨ་ཧཾ་གི་ཀཱ། པ་ར་མ་ག་ག་ཀཱ། པ་ར་མ་གོ་གི་ཀཱ། སྐྱེས་པའི་རྟགས་ཀྱི་སྐབས་སུ་བཤད་པ་རྣམས་ལ་ཨ་ཨིར་འགྱུར་བ་ཁོ་ན་བྱེད་ལ་ཡང་ན་མི་བྱེད་དེ། ཤུབྷི་ཀཱ་ལྟ་བུའོ། །བ་ཧུ་བཱི་ཧོ་ཏུ་ནི་ཏུཾ། འབྲུ་མང་པོ་ལ་ངེས་པའོ། །སྔར་སྐྱེས་པའི་རྟགས་ཀྱི་་།
