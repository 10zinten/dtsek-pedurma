།་མོ་་་་སྒྲཊིས་སྐུ་ལ་རྫོགས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི། ཀུ་ཝེ་ཏ་ལོ་ཝ་གཙོའམ་བློ་ལྟ༌སུ༑སོགས་སོ། །ཁྲིཊ་ལ། ཨིཊ་བྱིན་ཨ་ཕྱིས་པས། ཛི་གྷཏི་ཥྛ་ཏི། སོགས་དང་། དེ་བཞིན་དུ། རྀང་ལ། ་ཨ་ཛི་གྷཌི་ཏ། སོགས་སོ། སྐྱེ་བ། ༈དཱི་ད་ཟད་པ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར་གཉིས་བརྗོད་ཐུང་ངུར་བསྒྱུར་བས། དི་དཱི་ཥ་ཞེས་པ་ལ་ལཊ་སོགས་ཏང་རྣམས་སྦྱར་བས། ལཊ་དི་དཱི་ཥ་ཏེ། ལིང་ཡདི་དཱི་ཥ་ཏ། ལོཊ་དི་དཱི་ཥ་ཏྭཾ། ལང་…ཨ་དི་དཱི་ཥ་ཏེ། ལུང་…ཨ་དི་དཱི་ཎི། ལིཊདི་དཱི་ཏཱཾ་ཙ་ཀྲེ། ལུཊ་དི་དཱི་ཥི་ཏཱ། སྨོན་ལིང་དི་དཱི་ཏི་ཊཱི། ཁྲིཊ་དི་དཱི་་ཏི་ཥྛ་ཏེ། ཁྲིང་ཨ་དི་དཱི་ཥི་ཏ། སོགས་དང་། འགྲེལ་པར་ལུང་དྲངས་པ། དཱིངས་རྩྭ་ཏྭཾ་ཀཾ་ཙ་དཱ། དཱིང་གི་སན་ལ་ཨཱ་ཉིད་འགའ་ཞིག་ཏུ་ཡང་ནའོ། །དཱིང་གི་སན་ཕ་རོལ་ལ་ཨ་ཉིད་དུ་བསྒྱུར་བ་སྐབས་འགར་ཡང་ན་བར་བྱ་སྟེ། ་ཨཱཏ་དུ་བསྒྱུར་བའི་ཕྱོགས་ལ། །ཨ་བྷཱ་ས་ཐུང་ངུར་བསྒྱུར། སན་ལ་ཨིཏ་འོས་ཨིཏ་དུ་་་བསྒྱུར་བས། དི་དཱ་ས་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊདི་དཱ་ས་ཏེ། ལིང་དི་དཱ་སེ་ཏ། ལོཊ..དི་དཱ་ས་ཏྭཾ། ལང་ཨ་དི་དཱ་ས་ཏ། ལུང་…ཨ་དི་དཱ་སིཊྚ། ལིཊདི་དཱ་སིཾ་ཙཀྲེ། ལུཊ་དི་དཱ་སི་ཏཱ། ་སྨོན་ལིང་དི་དཱ་སི་ཁཱི། ཝིཊདི་དཱ་སི་བྱ་ཏེ། ་ལོང་…ཨ་དི་དཱ་སི་ཏ། །སོགས་སོ། ཚ་། །གང་༈དྷཱུཏ་འབར་བ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར། སན་ཡང་དག་ལ་འོས་གཉིས་བརྗོད། ཧཱུཏ་དང་སཱ་པི་དག་གི་སཾ་པྲ་སཱ་ར་ཎ་འོས། །ཨ་བྷཱ་ས་སཾ་པྲ་སཱ་་ར་ཎར་བསྒྱུར། བ་ལ་དང་པོའི་ཨིཊ་བྱིན་པས། དི་དྷཱུཏ་ཨི་སན་ཞེས་པ་ལ། །ར་ལོ་བྱུ་པ་དྷཱདྷ་ལཱ་དེ་སཾ། ཨུ་ཨི་ཨུ་པ་དྷཱ་ཧལ་དང་པོ་དང་ར་ལ་ལས་སན་འང་ངོ༌། །བྱིངས་ཧལ་དང་པོ་ཅན་ཨི་ཡིག་དང་ཨུ་ཡིག་གང་རུང་ཞིག་ཨུ་པ་དྷུ་ན་ཡོད་པ་དང་བྱིངས་ར་ཡིག་དང་ལ་ཡིག་མཐའ་ཅན་ལས་ཕ་རོལ་གྱི་སན་ཨིཊ་དང་བཅས་པ་འང་ཀ་འགྲོ་བའི་མིང་ཅན་དུ་ཡང་ན་བྱ་སྟེ། ཀིཏ་ཀྱི་ཕྱོགས་ལ་ཡོན་ཏན་མི་བྱེད། @་་།
