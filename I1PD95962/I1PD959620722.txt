ཨཱ་སི། ཨཱ་སི་ཏྭཾ། ཨཱ་སི་ཥ་ཏ། སོགས་སོ། །ལིཊ་ལ། ་ད་ཡ་ཨ་ཡ། ཨཱས་ལས་ཀྱང་ངོས་ཨཱམ་སྦྱར་བས། ཨཱ་ཧཱུཾ་ཙཀྲེ། ཨཱ་སཱ་བ་བྷུ་བ། ཨཱ་སཱ་མཱ་ས། སོགས་སོ། །ལུཊ་ལ་བལ་དང་པོའི་ཨིཊ་བྱིན་པས། ཨཱ་སི་ཏཱ། ཨཱ་སི་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་ལའང་ཨིཊ་བྱིན་པས། ཨཱ་སི་ཁཱིཊྚ། ཨཱ་སི་ཁཱི་ཡཱནྟཾ། ཨཱ་སི་ཁཱི་རན། །སོགས་སོ། །ཁྲིཊ་ལ། །ཨཱ་སྱ་ཀྵ་ཏེ། ཨཱ་སི་བྱེ་ཏེ། ཨཱ་སི་་ཤྱནྟེ། མི་སོགས་སོ། །དེ་བཞིན་དུ་ལྡིང་ལ། ཨཱ་སི་ཥྛ་ཏེ། ཨཱ་སི་བྱེ་ཏྭཾ། ཨཱ་སི་ཊྛནྟ། སོགས་སོ། །།།ལ་ཏེ་ཁ་། ་བྱ་ཀྱི་དབང་་ས་བོ་༈བ་ས་ཀུན་ནས་འགེབས་པ་ལའོ་ཞེས་པ་ལའང་ཏང་རྣམས་སྦྱར་ཏེ། ལ་ཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བསྟེ། ་བ་སཱ་ཏེ། བ་ས་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བསེ། བ་སཱ་ཐེ། ་བ་དེ། བདྡྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་སེ། བ་སཱ་ཧེ། བསྨྲ་ཧེ། ལིང་ལ། ཨཏ་མཐའ་མིན་པ་་་སོགས་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས། །བས་ཀྱི་ས་དྲངས་པས། བ་སཱི་ཏ། བ་སཱི་ཡཱ་ཏཱ། བ་སཱི་རན། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བནྡྷཾ། བ་སཱ་ཏྭཾ། བ་ས་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བསྭཱ། བ་སཱ་ཐཱ། ་བ་དྡྷཾ། ་བདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་སི། བ་སཱ་བ་ཧི། བ་ས་མ་ཧི། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བསྟ། ཨ་བ་སཱ་ཏཱ། ཨ་བ་ས་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བནྡྷ། ཨ་བ་སཱ་ཐཱ། ཨ་བ་ཏཱཾ། ཨ་བདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་་་གསུམ་ལ། ཨ་བ་སི། ཨ་བ་སྭཱ་ཧི། ཨ་བསྨ་ཧི། ལུང་ལ་བལ་དང་པོའི་ཨིཊ་བྱིན་པས། ཨ་བ་སིཊྚ། །ཨ་བ་སི་ཥཱ'ཏྭཾ་ཨ་བ་སི་ཥ་ཏ། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་བ་སེ། བ་བ་སཱ་ཏེ། བ་བ་སི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་བ་སི་ཏེ། བ་བ་—།
