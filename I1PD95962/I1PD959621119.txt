ཁ་བྱང་ཉམོ་ཙཱ་ཨ་ཎ་སྐུ་མ་རྫོགས་བསྟན་འགྱུར། སྒྲ་མདོ། པ་ཐེཙི་ནི་ད་ལོ་མཚོ་ཚོར་ཎ་༼jཀ་ཊབློ?odལནའོས་ཚ་ཕྱིས། མཏྟ། མྱོས། བིཏྟཾ་བྷོ་ག་པྲི་:ཏཱ་ཏ་ཡོ། ལོངས་སྤྱོད་དང་རབ་ཏུ་གྲགས་པ་དག་ལ་བིཏྟའོ། །བིཏ་ཤེས་པ་ལ་ལྟ་བྱིན་པས། བིཏྟཞེས་པ་ལོངས་སྤྱོད་་་དང་རབ་ཏུ་གྲགས་པ་དག་བརྗོད་པ་ལ་ཚིག་ཕྲད་དེ། བིཏྟཔྲ་ཏཱི་ཏ། གྲགས་པ་རིག། བིཏྟཾ་དྷ་ནཾ། ནོར་རིག། རཾ་ཎ་མཱ་ཙིདྷ་མཙྩེ། རཾ་ཎཾ་བུ་ལོན་ལྷག་པ་ལའོ། །རྀ་འགྲོ་བ་ལ་ལྟ་སྦྱར་བས། རཾ་ཎི་ཞེས་པ་ལྷག་པའི་བུ་ལོན་བརྗོད་པ་ལ་ཚིག་ཕྲད་དེ། ཨཱ་དྷ་མཎཾ། བུ་ལོན་ཁྱེར་བ། །ཨིཊ་ནིཏྟཱ་ཡཱ། ནིཥྛ་ལ་ཨིཊའོ། །ནིར་སྔ་མའི་ཀུ་ཥ་་༌་ལས་ཕ་རོལ་གྱི་ནིཥྛ་ལ་ཨིཊ་འགྱུར་ཏེ། ནི་ཌུ་ཥི་ཏ། ཕྲོགས། བ་ས་ཏི་གླུ་དྷོ་རིཊ། བ་ས་ཏི་གླུ་དྷ་དག་ལས་ཨིཊའོ། །བ་ས་ཏི་དང་ཀླུ་དྷ་དག་ལས་ཕ་རོལ་གྱི་་་ནིཥྛ་དང་ཁྭ་དག་ལ་ཨིཊ་རྟག་ཏུ་འགྱུར་ཏེ། པ་ཙ་སཱ་ལི་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་བས། ཨུ་ཥི་ཏ། གནས། ཀླུ་དི་ཏ། ལྟོགས། ཨནྡྷེ་པཱུ་ཛཱ་ཡཾ། ཨཉྩ་་ལ་མཆོད་པ་ལའོ། །།ཨཉྩ་ལས་མཆོད་པའི་དོན་ལ་ཎིཥྛ་དང་ཀྵ་དག་ལ་ཨཊ་་འགྱུར་ཏེ། ཨཉྩི་ཏདེ་བ། ལྷ་མཆོད། ཀླུ་ཤནིཥྛ་ཡོ། ཀླུ་ཤ་ལས་ནིཏྟཱ་ལའོ། །ཀླུ་ཤ་ལས་ཕ་རོལ་གྱི་ནིཥྛ་ལ་ཨིཊ་ཡང་ན་འགྱུར་ཏེ། ཀླུ་ཤི་ཏ། །གླི། བྲཅུ་བྷསྔ་སོགས་ཀྱིས་ཤ་ཥ། འགྲེལ་པར། བ་མ་དང་ཛ་པི་དང་། བི་དང་ཨཱང་སྔ་མའི་ཤུས་ལས་ནིཥྛ་ལ་ཨིཊ་ཡང་ན་འགྱུར་རོ་ཞེས་བཤད་པས། བ་མི་ཏ། སྐྱུགས། བནྟ། ཛ་པི་ཏ། ཟླས། ཛཔྟ། བི་ཤྭ་སི་ཏ། བི་ཤཱསྟ། ཨཱ་ཤཱ་སི་ཏ། ཨ་ཤཱསྟ། པཱུ་དུ། པུང་ལས་ཀྱང་ངོ༌། །པུང་ལས་ཕ་རོལ་གྱི་ནིཥྛ་དང་ལྟ་དག་ལ་ཨིཊ་་་ཡང་ན་འགྱུར་རོ། །པུངཀ་ཙ། པུང་ལས་ཀྟིའང་ངོ༌། །པུང་ལས་ཕ་་རོལ་གྱི་ཨིཊ་དང་བཅས་པའི་ནིཥྛ་དང་ཁཱའང་ཀ་འགྲོ་བའི་མིང་ཅན་མ་ཡིན་ཏེ། མདོ་སྔ་མས་ཨིཊ་ཡང་ན་བར་བྱས། ཨིཊའི་ཕྱོགས་ལ་མདོ་འདིས་ཡོན་ཏན་བྱས་པས། པ་བི་ཏ། དག་པའམ་གཙང་མ། པཱུ་ཏཿ། དག་པའམ་གཙང་མ། ནིཥྛཤིང་སཱི་དི་མི་།
