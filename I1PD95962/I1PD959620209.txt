་ལུ་ས་མ་ཡ་་ཟླ་ཀློ་ཏུ་ལ་སོགས་པ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིཨ་མི་ན་ཚེ་གང་བཙོད་ཀཎཀ་དག་བློ་ཅན་ཡིན་པས་ཕྱིས། །ལྷག་མ་ལ་དབྱི་འོས་ད་ཕྱིས། བཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས་་ཨ་གཉིས་སྔ་མའི་གཟུགས་སུ་སོང་། ཏ་ཝ། མ་མ། ཡུསྨད་ཨསྨད་ལ་དྲུག་་་པའི་ཨོས། གཉིས་ཚིག་ལ་ཡུ་ཝ་ཨཱ་ཝ་དག་གོས། ཡུ་ཝ་ཨད། ཨོས། 《ཨཱ་ཝ་ཨད༑། ཨོས། །ཨཙ་ལ་ཡ་འོས། །ད་ཡ། ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས། ཨ་་སྔ་མའི་གཟུགས་སུ་སོང་དྲངས་པས། ་ཡུ་ཝ་ཡོ། ཨཱ་ཝ་ཡོ། ཡུམྨད་ཨསྨད་ལ་དྲུག་པའི་ཨཱམ། ལྷག་མ་ལ་དབྱི་འོས། ད་ཕྱིས། ཨཱམ་ལ་སརྦའི་མིང་ཅན་རྣམས་ལ་སུཊ་འོས། །སུཊ་ཨཱ་ག་མ། ཡུ། །སཱམ། ཨསྨ། །སཱམ། སཱམ་ཨ་ཀཾ། སཱམ་གྱི་ཨཱ་ཀཾ་ངོ༌། །ཡུད་ཨསྨད་ལ་རྣམ་དབྱེ་དྲུག་པའི་ཨཱམ་སུཊ་དང་བཅས་པའི་སཱམ་ཞེས་པ་ནི། ཨཱ་ཀཾ་ཞེས་པར་འགྱུར་ཏེ། སཾ་ཨཱ་ཀཾ་དུ་བསྒྱུར། །ཨཀ་གི་འདྲ་བ་ལ་རིང་པོ་འོས། །ཡུཏྟཱ་ཀཾ། ཨསྨཱ་ཀཾ། ཡུཥྨད་ཨསྨད་ལ་རྣམ་དབྱེ་བདུན་པའི་ང་། །གཅིག་ཚིག་ལ་ཏཱ་མ་དག་གོས། ཏཱ་ཨད། ང་། འམ་་ཨད༑ང་། ཨཙ་ལ་ཡ་འོས། ད་ཡ། ཏཱ་ཡི། མ་ཡི། ཨོས་དྲུག་པའི་སྐབས་དང་འདྲ་སྟེ། ཨ་ཡུ་ཝ་ཡོ། །ཨཱ་ཝ་ཡོ། ཡུཥྛད་ཨསྨད་ལ། ལ་བདུན་་་་པའི་མང་ཚིག་སུཔ། ཡུཥྨད་ཨསྨད་དག་གི་ཀུན་ནས་མ་བསྒྱུར་བ་ལ་འོས། །ད་ཨཱ༑ཨཀ་གི་འདྲ་བ་ལ་རིང་པོ་འོས། ཡུབྷཱ་སུ། ཨསྣ་སུ། །ཁ་ལས་ལྟ་བུ། བ༈ཡང་ཡུདྨད་ཨསྨད་ཀྱི་ལུས་འགྱུར་གཞན་ནི་མི་ཟད་པའི་ཚིག་སྡུད་ཀྱིས་་རྣམ་དབྱེ་རྣམས་ཀྱི་གཅིག་ཚིག་བསྡུས་རྗེས་ཨ་ཏི་སྔ་མ་ཅན་ནི་རྣམ་དབྱེ་དང་པོའི་་གཅིག་ཚིག་དང་མང་ཚིག་དང་བཞི་པའི་ང་། དྲུག་པའི་ངས་རྣམས་མ་གཏོགས་ཕ་རོལ་དུ་རྣམ་དབྱེ་གང་བྱུང་ཡང་ཨད་བོར་བའི་ཡུལྦ་ནི་ཏཱ་དང་ཨད་པོར་བའི་་་ཨསྨྲ་ནི་མ་ཞེས་པ་ཁོ་ནར་འགྱུར་ཏེ། །དཔེར་ན། ཏཱཾ་བཱ་ཨ་ཏི། ཨཾ་བཱ་ཨ་ཏི་་་ཞེས་པ་རྣམ་དབྱེ་གཉིས་པའི་གཅིག་ཚིག་ཨམ། ཚིག་སྡུད་ཀྱིས་བསྡུས་ཨ་ཏི་སྔོན། —།
