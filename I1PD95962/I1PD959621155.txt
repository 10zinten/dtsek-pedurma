འཡ་་་ཟླ་བ་ཡུ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི༽*ཁ་དྲི་ཚོར་བྱེད་བཞི་ཚེས་ཟླ་ན་ཌ་《སྣར་བའོ། 《སྣར་ཝེ། «པེ་»«སྣར་»ཀཱ། «པེ་»《སྣར་མཐའ། ཉི«སྣར་»ཀྲི། «སྣར་རི། «སྣར་»། «སྣར་པུ། «སྣར་»ཨུ། «པེ་ཨཾ། «སྣར་»ཨི། །«པེ་»«སྣར་»+བ། «སྣར་»—ལ། །པ«སྣར་—ཨ། ཚོ་ངག་འགྲོས་མ་བག་«པེ་ཧ། «སྣར་རྣམ། «པེ་»«སྣར་»མཉྫི། «པེ་»«སྣར་རུ། «པེ་»སོ། «སྣར་བ། «པེ་《སྣར་སི། «པེ་»«སྣར་»—ལ་རྗེས་སུ་སྣ་ལྡན། «སྣར་»ཥརྨ༑@«སྣར་»གླུ། ཚེ«སྣར་»ས། །ངོ་། «སྣར་པཾ། «སྣར་»སྔན། «པེ་»«སྣར་ལོ། ་®«པེ་»«སྣར་»ཀྱིས་དྲ་ཚྭ། «པེ་»《སྣར་»ཛ། ངབད་་། ཀྱི། །།།::བཀ་ར%?། ཁ»། aཚུལ་《། །བས་བ་《འི་(:«སྣར་»ར། «པེ་»ཡཾ་ཀཱ་རོ། «སྣར་»ཡི་ཀཱ་རི། མི«པེ་»«སྣར་»ག་བསྒྱུར། «སྣར་»སོ། «སྣར་ཀ«སྣར་»མྨོ། «པེ་»«སྣར་»ཛྲ། །སྐྱེ་བ་དམན། བ་ལང་«པེ་»«སྣར་»ཏི། ཨུ«པེ་»«སྣར་»ཟླ། «པེ་»«སྣར་»ཀིནྡྷ། «པེ་»«སྣར་»ཀ«སྣར་»ཊུསཱ་ཨ། «པེ་»«སྣར་»ཥཀྵ། བ་n»། བཔ«པེ་»«སྣར་སཉྩ། «པེ་»«སྣར་ཏུ། ་ཚེ«པེ་»《སྣར་ལ། ༄«པེ་»«སྣར་»སམ། ་་། «པེ་»«སྣར་»ཚིནྡ། «པེ་»«སྣར་ར། «པེ་»«སྣར་»ཙྪིནྡ། «པེ་»«སྣར་»ཤཱམ། «པེ་»《སྣར་པོ། «པེ་ཙཱམཱ། «སྣར་ཨཱམཱ། @«སྣར་»ཀཾ། «སྣར་»སྐ། དppཊ་དང་། སྐྱེ«སྣར་»ཏེནུ། ®«པེ་ཡ། ག་ག་ན་«སྣར་»ཏོཤོ། «སྣར་»དབི། «སྣར་»ཏཾ། །(ི@«སྣར་ཨ། «པེ་»ཨ། «པེ་»«སྣར་ཡ། «སྣར་_ག«པེ་»«སྣར་ཝཏ། @«པེ་»《སྣར་ཨ། །ཅེ«སྣར་»སརྦ། «སྣར་»+ཇལ། «པེ་»«སྣར་པ། ཞེ་ཆེ«སྣར་སུང་། «སྣར་»ས། །ར»། །གནས་«པེ་»«སྣར་ཨེ། «པེ་»«སྣར་»སྦྱར། «སྣར་»སེ། «སྣར་»ལྟན། བ་»«པེ་»«སྣར་»སི། པ་ར་ག་ར་་་དག་ཁོམ་ལ། ཉེར་«པེ་»«སྣར་ཡོད། «པེ་»«སྣར་བྷིས། «སྣར་བྱཱམ། རིག་པའི་ཐུགས་དགྱེ(«སྣར་ཙུ་ཊུ། Q«པེ་»«སྣར་»སྔ། «པེ་《སྣར་»སཾ། «པེ་»«སྣར་བུདྡྷེ། <*#»༽«པེ་»«སྣར་»ཤ«པེ་»ག«སྣར་»ཡཱངྒ། «སྣར་ན། ་་་། 《ར། 《་(། ཟླ《བ། »»༽«སྣར་ཨ། «པེ་《སྣར་ཏུ། །་་་་་་»དཀའ་བས། ཊོ་ཁོg»ད ེ«པེ་»ར། «སྣར་»ར། །
