ས་འོས༑ལས་ཨ། ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས། སྔ་མའི་གཟུགས། ཨེ་ཙ་ལ་་་འཕེལ་བ་འོས། ཨ་དོ། ཨ་ད་སོརྡ་དུ་དོ་མ། ཨ་དས་ཀྱི་ད་ལ་ཨུ་སྟེ་དའི་་་་འང་མའོ། །ཨ་དས་ཞེས་པའི་ད་ནི་རྣམ་དབྱེ་དང་པོའི་གཅིག་ཚིག་སུ་མ་གཏོགས་རྣམ་དབྱེ་གང་མཐོང་ཡང་མ་ཡིག་ཏུ་འགྱུར་ཞིང་དའི་སྲོག་ཨཙ་རིང་པོ་གང་ཡོད་་་ཀྱང་སྤུ་རིང་དུ་འགྱུར། །ཐུང་ངུ་གང་ཡོད་ཀྱང་ཨུ་ཐུང་དུ་འགྱུར་ཏེ། ཨ་མཱུ། ཛས་ལ། །ཁྱད་སོགས་ཀྱིས་ཨ་ད། ཛས། ཨ་དས་ཀྱིས་ཞེས་པས། ད་མ། ཛས་ཀྱི་ཤཱི། ཨ་ལས་ཡོན་ཏན། ་ཨ་མེ། ཨེ་ཏ་ཨཱི་དྭ་ཧུ་བ་ཙ་ནེ། ཨེཏ་ཀྱི་་ཨིཏ་མང་ཚིག་ལའོ། །ཨ་དས་ཀི་ད་མར་གྱུར་པའི་རྣམ་དབྱེ་མང་ཚིག་ལས་བྱུང་་བའི་ཨེ་ནི་ཨཱི་རུ་འགྱུར་ཏེ། ཨེ་ཨཱི། ཨ་མཱི། ཨམ་ལ། ཨ་ཡུཾ། ཨོ་ལ། ཨ་མཱུ། ཤས་ལ། ཨ་མཱུན། ཊཱ་ལ། ཨ་མུ་ནཱ། བྷཱུམ་ལ། །ཨ་མཱུ་ཧཱུཾ། བྷིས་ལ། ཀ་མ་ཡིན་པ་དག་གི་ཨི་དཾ་ཨ་དས་དག་ལ་མ་ཡིན་ནོས། བྷིས་ཨིསར་མི་འགྱུར། །མང་ཚིག་ལ་ཛྷལ་ལ་ཨེཏ་འོས་ཨེ། ཨེཏ་ཀྱི་ཨཱིཏ་མང་ཚིག་ལ་འོས། ཨ་མཱི་བྷི། ངེ་ལ་ཨ་མུ། བྷཱུམ་ལ། ཨ་མཱུ་དྡྷཾ། བྷུས་ལ། ཨ་མཱི་ཐཱི། ང་སི་ལ། །་་ཨ་མུཉྩཱཏ། བྷཱུམ་བྱུས་འདྲ། །ངས་ལ། ཨ་མུ་ཉྩ། །ཨོས་ལ། ཨཱ་མུ་ཡོ༄༅། ཨཱམ་ལ། ཨ་མཱི་ཏཱཾ། །ཇི་ལ། ཨ་མུཙྪིན། ཨོས་ལ། ་ཨ་མུ་ཡོ༑སུཔ་ལ། ཨ་མཱི་ཏུ། ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ཏཱུརླིངྒ་ཧ་་་ལནྟཱས་མཱབྷཱ། ཞེས་:པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་སྐྱེས་པའི་རྟགས་ཧལ་མཐའ་ཅན་རྣམས་རྫོགས་སོ། །།།་ཁཥྛ་ཏེཀོ་དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་་་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་་་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས་སྐྱེས་པའི་རྟགས་ཧལ་མཐའ་ཅན་།
