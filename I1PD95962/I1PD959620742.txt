སོགས་སམ། ་ཨཱཆུ་བི་ཏཱ། ཨཱཆུ་བི་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་་ཨ་ཏང་ལ། །ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོར་བསྒྱུར་བས། ཨུཥྞ་ཡཱཏ། ཨུཎྜཱུ་ཡཱནྟཾ། ཨཱུརྞཱ་ཡཱ་སུ། སོགས་དང༌། ཏང་ལ་བལ་དང་པོའི་ཨིཊ། ཨུཎྜའི་ཡང་ན་འོས་ཡོན་ཏན་དང་། ཨུབ་ཡང་ན་བར་བྱས་པས། ཨུཎྜ་བི་ཥིཉྩ། ཨུཎ་བི་ཏཱི་ཡཱནྟཾ། ཨུཎ་བི་ཊཱི་རན། སོགས་སམ། ཨུཎྜ་བི་ཊཱི། སོགས་སོ། །ལིཊ་ལའང་ཨིཊ་བྱིན། ཡོན་ཏན་དང་ཨུབ་ཡང་ན་བར་བྱས་པས། ཨ་ཏང་ལ། ཨུཎ་བི་ཥྛ་ཏི། །ཨུཎ་བི་ཁྱཱ་ཏ། ཨུཎ་བི་ཥྛནྟི། སོགས་སམ། ཨཱརྟུབི་ཥྛ་ཏི། སོགས་སོ། །ཏང་ལའང༌། ཨུཎ་བི་ཏེ། ཨུཎ་བི་སྐྱེ་ཏེ། ཨུཎ་བི་ཤྱནྟེ། སོགས་སམ། ཨུཆུ་བི་ཁྱཱ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་གི་ཨ་ཏང་ལ། ཨོཎ་བི་ལྱཱཏ། ཨོཎ་བི་ཥྛ་ཏྭཾ། ཨོཎ་བི་ཤྱན། སོགས་སམ། ཨོརྟུ་བི་ཏ། སོགས་སོ། །ཏང་ལ། །ཨོཎ་བི་ཁྱཱ་ཏ། ་ཨོཎ་བི་ཥྛ་ཏྭཾ། ཨོཎ་བི་ཤྱནྟ། སོགས་སམ། ཨོཥྞ་བི་ཁྱཱ་ཏ། ་སོགས་སོ། །།༈བྷུཉ་བསྟོད་པ་ལའོ་ཞེས་པ་ལ་ཡང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། ནུ་རུ་སྐུ་ཤ་མྱ་མ། སརྦ་བྷཱ་ཏུ་ཀེ། ནུ་རུ་སྐུ་ཤམ་ཡམ་ལས་བྱིངས་ཐམས་ཅད་པ་ལའོ། ཏུ། ནུ་ཐ་བསྟོད་པ་རུ་སྒྲ་བྷུཉའི་བྱིངས་ཀྱི་དང་པོའི་ཥ་སར་བསྒྱུར་བས། སྐུ་བསྟོད་པ་། ཤ་མུ་ཉེ་བར་ཞི་བ། ཡ་མུ་ཉེ་བར་འགོག་པ་རྣམས་ལས་ཕ་རོལ་གྱི་བྱིངས་ཐམས་ཅད་པ་ཧལ་དང་པོ་ཅན་པ་འགྲོ་བའི་མིང་ཅན་ཡིན་པ་དེ་ལ་ཨཱིཊ་ཨཱ་ག་མ་ཡང་ན་འགྱུར་ཏེ། བྷུཉ་ལ་ལཊ་སྦྱར་བས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་་་་པོའི་གཅིག་ཚིག་ལ། །མདོ་འདིས་ཨཱིཊ་ཡང་ན་བར་བྱས། །ཨཱིཊའི་ཕྱོགས་ལ་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་དང༌། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། །ཨུཏ་ཀྱིས་འཕེལ་བ་སོགས་བྱས་པས། སྟ་བཱི་ཏི། ་སྟོ་ཏི། —།
