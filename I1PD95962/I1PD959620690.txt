སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཎཨཱ་ཡཾ། ཨི་བ། །ཨ་མ། ཨི་ཌོ་གཱ་ལུ་དི། ་ལུང་ལ་ཨིཎ་གྱི་གཱའོ། །ལུང་ཕ་རོལ་ལ་ཨིཎ་ནི་གཱ་ཞེས་པར་བསྒྱུར་བར་བྱ་སྟེ། ཨིཎ་ལ། །ལུང༌། །ཤ་མདོ་འདིས་གཱར་བསྒྱུར། །གཱ་ཏི། སྣ། གླུ་བཱ། །བྷཱུ་སོགས་ཀྱིས་སིཙ་འཇིག་པས། ཨ་གཱཏ། །།ཨ་གཱ་ཏྭཾ། ཨ་གུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་གཱ། ཨ་གཱ་ཏཾ། ཨ་གཱ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་གཽ། ཨ་གཱ་བ། ཨ་གཱ་མ། ལིཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། །ཨིཎ་ལ་ལིཊ་ཨ་བྷཱུ་ས་མེད་པ་སོགས་ཀྱིས་བྱིངས་གཉིས་བརྗོད། །ལ་ཏིཔ། །ཏིཔ་ཎལ། ཨཙའི་ཞྭ་ཎཱ་འགྲོ་བ་ལ་འོས་འཕེལ་བ་ཨེཙའི་ཨཡ་ཨ་བ་སོགས་ཀྱིས་ཨེ་ཙིཨཱཡ། ཨ་བྷཱ་སའི་འདྲ་བ་མ་ཡིན་པ་ལ་འོས་ཨི་གོང་མ་ཨི་ཡང་ང་དང་ཨ་འགྲོ་བ་རིམ་པ་་་བཞིན་དྲངས་པས་ཨི་ཡཱ་ཡ། དཱིགྷ་:ཨི་ཎཀི་ཏི། ཀ་འགྲོ་བ་ལ་ཨིཎ་གྱི་རིང་་་པོའོ༑། ལིཊ་ཀ་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨིན་གྱི་ཨ་ཧྥུ་ས་རིང་པོར་འགྱུར་ཏེ། སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ། །ཨི་གཉིས་བརྗོད་མདོ་འདིས་ཨ་བྷྱཱ་ས་རིང་པོ། །ཨིཎ་གྱི་ཡ་ཎ་འོས་ཨི་ཕྱི་མ་ཕ་དྲངས་པས། ཨཱི་ཡ་ཏུ། ཨཱི་ཡུ། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨི་ཡ་ཡི་ཐ༑ཨི་ཡེ་ཐ། ཨཱི་ཡ་ཐུ། །ཨཱི་ཡ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨི་ཡཱ་ཡ། ཨི་ཡ་ཡ། །པུཨི་ཡི་བ། །ཨི་ཡི་མ། །ལུཊ་ལ་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། །ཨེ་ཏཱ། ཨེ་ཏཱ་རོ། ཨེ་ཏཱ་རསོགས་སོ༑ས་། སྨོན་པའི་ལིང་ལ་ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས། །ཨཱི་ཡཱཏ། ཨཱི་ཡཱནྟཾ། ཨཱི་ཡཱ་སུསོགས་སོ། །ཨེ་ཏེ་རྨི་ང་། ཨེ་ཏིའི་ལིང་ལའོ། །ཉེར་བསྒྱུར་སྔོན་དུ་སྦྱོར་ན་ཨིཎའི་སྨོན་པའི་ལིང་ཕ་རོལ་ལ་ཐུང་ངུར་འགྱུར་ཏེ། ཨཱི་་་ཡཱཏ་ཅེས་གྲུབ་པའི་སྔོན་དུ་ཨ་ནུ་བཞག། མདོ་འདིས་ཐུང་ངུ་། ཨིཀ་གི་ཨཙ་་།
