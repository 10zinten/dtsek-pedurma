ཞེས་པ་དེ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་ཀའིནི་ཀོ་ལ་སྐྱོན་པ་ན་ལོ་ཧི་ཏཱ་ཡ་ཏ། ཨ་ལོ་ཧི་ཏཱ་པཱིཏ། ...ལུང་ཨ་ལོ་ཧི་ཏཱ་ཡིཉྩ། ལོ་ཧི་ཏཱ་ཡཱ་ཙ་ཀཱ་ར། .ལིཊ་ལོ་ཧི་ཏཱ་ཡཾ་ཙ་ཀྲེ། ལོ་ཧི་ཏཱ་ཡི་ཏཱ་སི། ...ལུཊ'ལོ་ཧི་ཏཱ་ཡི་ཏཱ་སེ། ལོ་ཧི་ཏཱ་ཡཱཏ། སྨོན་ལིང་ལོ་ཧི་ཏཱ་ཡི་ཏིཋ°། ལོ་ཧི་ཏཱ་ཡི་ཥྛ་ཏི། ཁྲིཊལོ་ཧི་ཏཱ་ཡི་བྱ་ཏེ། ཨ་ལོ་ཧི་ཏཱ་ཡི་ཏ། ...ཁྲོང་…ཨ་ལོ་ཧི་ཏཱ་ཡི་ཥྛ་ཏ། སོགས་སོ། །།།པ་སྐྱེ་བ་༈དེ་བཞིན་དུ། ཙནྡཱ་ཡ་ཏི། པགས་པ་བྱེད། ཙརྨ་ཡ་ཏེ། སོགས་དང་། ཌཱ་ཙའི་མཐའ་ལ། བ་ཊ་པ་ཊ་ཡ་ཏི། ཤིན་ཏུ་ལམ་པ་ལམ་པ་བྱེད། པ་ཊ་པ་ཊ་ཡ་ཏེ། ཤིན་ཏུ་ལམ་པ་ལམ་པ་བྱེད། སོགས་སོ། །ཀཥཱ་ཡ་ཀརྨ་ཎེ། ཀཋ་ལ་ལས་ལའོ། །ཀཏྠ་ལ་མི་དགེ་བ་ཆེད་དུ་བྱེད་པའི་ལས་ཀྱི་དོན་ལ་ཀྱང་འགྱུར་ཏེ། །ཀཏྟཱ་ཡ་ཏེ། ཚེགས་ཀྱི་་ཆེད་དུ་འགྲོ། སོགས་སོ། །ས་ཏྲ་ཀཀྵ་ཀཋ་ཀྲྀ་ཚྭ་ག་ཧ་ནེ་ཧྥུ། བཱ་པ་ཙི་ཀཱིནཱ་ཡཾ། ས་ཏྲ་ཀག་ཀཎྜ་ཀྲྀ་ཚྭ་ག་ཧན་རྣམས་ལས་སྡིག་པ་བྱེད་པར་འདོད་པ་ལའོ། །ས་ཏྲ། ཀཀྵ། ཀཋ། །ཀྲྀ་ཚྭ། །ག་ཧ་ན་རྣམས་ལས་སྡིག་པ་བྱེད་པར་འགྲོ་བར་འདོད་པའི་དོན་ལ་ཀྱཔ་འགྱུར་ཏེ། ས་ཏྭཱ་ཡ་ཏེ། རྒྱུན་ནམ་ཚུལ་ལ་འགྲོ། ཀཥཱ་ཡ་ཏེ། རྩྭའི་ཆེད་དུ་འགྲོ། ཀཥཱ་ཡ་ཏེ། ཚེགས་ཀྱི་ཆེད་དུ་འགྲོ། ཀྲྀ་ཚྭ་ཡ་ཏེ། ག་ཧ་ནཱ་ཡ་ཏི། ཐིབས་པོ་ལ་འགྲོ། སོགས་སོ། །ཀརྨ་ཎོ་རོ་མནྡཱ་ཏ་པོ་བྱཱ་བརྟི་ཙ་རོ། རོ་མནྡྲ་དང་་་ཏ་པས་དག་ལ་གཡོ་བའི་ལས་ལ་འཇུག་པ་ལའོ། ::། རོ་མནྡྲ་ལས་འགྲམ་པ་གཡོ་བའི་དོན་ལ་འཇུག་པ་དང་ཏ་པས་གཡོ་བའི་དོན་ལ་འཇུག་པ་ལས་ཀྱང་འགྱུར་ཏེ། རོ་མནྡྷ་ཡ་ཏེ་གོ། བ་ལང་སྐྱུགས་བལྡད་བྱེད། ཏི་བ་སླ་ཏེ། དཀའ་ཐུབ་ལ་འགྲོ། སོགས་ཏེ། :：འགྲེལ་པར་ཏ་པས་ཀྱང་གི་མཐའ་ཅན་ལ་ཨ་ཏང་རྣམས་འཇུག་པར་བཤད་དོ། །བཱཟློཀྵ་བྷྱཱ་མུ་དྭ་མ་ནེ། བཱ༔ཨུཥྛ་དག་ལས་སྟེང་དུ་འཕྱུར་བ་ལའོ། །བཱà་ཨུ་དག་ལས་སྟེང་དུ་འཕྱུར་བའི་དོན་ལ་ཀྱང་འགྱུར་ཏེ། བཱབྷཱ་ཡ་ཏེ། རླུང་འཕྱུར་བར་བྱེད། ཨུབྷཱ་ཡ་ཏེ། དྲོད་འཕྱུར་བར་བྱེད། སོགས་སོ། །ཕེ་ནཱཙཱ། ཕེ་ན་ལས་ཀྱང་ངོ༌། །ཕེ་།
