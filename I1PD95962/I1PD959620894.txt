ཤ་རཱི་ཏཱ། །ཤ་རཱི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བྱིངས་ཀྱི་རཱིཏ་ཀྱི་ཨིཏ་འོས། ་རོ་ཨིཏ། ཨུ་པ་དྷཱའི་འང་ངོས་རིང་པོར་བསྒྱུར་བས། བཱིནཱ་ཏ། ཤཱིརཱནྟཾ༑། ཤིན་སུ། །སོགས་སོ། །།ལྀཊ་ལ། བྲྀ་རྀཏ་དག་ལས་ཡང་ན་འོས། ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། །ཤ་རི་ཥྛ་ཏི། །སོགས་སམ། ཤ་རཱི་ཥྱ་ཏི། སོགས་སོ། །།དེ་བཞིན་དུ། ལིང་ལ། ཨ་ཤ་རི་ཏ། སོགས་སམ། ཨ་ཤ་རཱི་ཏ། སོགས་སོ། །་་་་༈རྀ་རྣམ་པར་འཇོམས་པ་ལ་ལོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། པུ་སོགས་ཀྱིས་ཐུང་ངུར་བསྒྱུར་བས། ལཊ་ལ། དྲྀ་ཎཱ་ཏི། :སོགས་དང་། ལིང་ལ། དྲྀ་ཎཱི་ཡཱཏ། །སོགས་དང་། ལོཊ་ལ། །དྲྀ་ཎཱ་ཏུ། །སོགས་དང་། ལང་ལ། །ཨ་དྲི་ཎཱཏ། །སོགས་སོ། །ལུང་ལ། ཨིཊ་བྱིན་འཕེལ་བ་བྱས་པས། ཨ་དཱ་རཱིཏེ། ཨ་དཱ་རིཏྟཾ། ཨ་དཱ་རི་ཥ། །སོགས་སོ། ས་བ། ལིཊ་ལ། ཀྲིཏ་རྣམས་ལ། ཤྲཱི་ཛཱ་པཱ་རྣམས་ཀྱིས་ཐུང་ངུར་ཡང་ན་བསྒྱུར། ཐུང་ངུའི་ཕྱོགས་ལ། །ཡཎ་དང༌། །མ་བསྒྱུར་བའི་ཕྱོགས་ལ། རཱིཙྪ་ཏི། ཛཱཏ་རྣམས་ཀྱི་འོས་ཡོན་་ཏན་བྱས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དཱ་ར། ད་དྲ་ཏུ། ་ད་ད་ར་ཏུ། ད་དྲུ། ད་ད་རུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་ད་རི་ཐ། ད་དྲ་ཐུ། ད་ད་ར་ཐུ། ད་དྲ། ད་ད་ར། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ད་དཱ་ར། །ད་ད་ར། ་ད་དྲི་བ། ད་ད་རི་བ། ད་དྲི་མ། ད་ད་རི་མ། ལུཊ་ལ། ཤྲཱི་རཱིཏ་དག་གི་ཡང་ན་འོས། ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ད་རི་ཏཱ། ད་རི་ཏཱ་སི། ་། སོགས་སམ། ད་རཱི་ཏཱ། །ད་རཱི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བྱིངས་ཀྱི་རྀཏ་ཀྱི་ཨིཏ་འོས། རྀ་ཨིཏ་ཨུ་པ་དྷཱའི་འང་ངོས་རིང་པོར་བསྒྱུར་བས། དཱིནཱ་ཏ། དཱིནཱ་ཙཱ། དཱིནཱ་སུ། སོགས་སོ། །ཁྲིཊ་།
