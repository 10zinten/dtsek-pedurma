སྒྲ་མདོ། བཏི:སྐོར་གྱི་དཀྱིལ་ཁུག་ནས་ཀྱང་བྱས་ཀྲྀ་ཎཱ་ཏི། སོགས་དང་། ཏང་ལ། ཀྲྀ་ཎཱི་ཏེ། །སོགས་སོ། །ལིང་གི་ཨ་་་ཏང་ལ། ཀྲྀ་ཎཱི་ཡཱཏ། །སོགས་དང་། ཏང་ལ། ཀྲྀ་ཎྜི་ཏ། །སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། ཀྲྀ་ཎཱ་ཏུ། །སོགས་དང་། ཡེ། ཏང་ལ། ཀྲྀ་ཎི་ཏྭཾ། སོགས་སོ༑། ལང་གི་ཨ་ཏང་ལ། ཨ་ཀྲྀ་ཎཱ་ཏ། །སོགས་སོ། །ཏང་ལ། *ཨ་ཀྲྀ་་ཎྜི་ཏ། །སོགས་སོ། ཆུ། ལུང་གི་ཨ་ཏང་ལ། ཨིཊ་བྱིན་སིཙ་ལ་འཕེལ་བ་པ་རསྨི་པ་ད་རྣམས་ལ་འོས། འཕེལ་བ་བྱས་པས། ཨ་ཀཱ་རཱིཏ། །ཨ་ཀཱ་རི་ཏཱཾ། ཌཨ་་ཀཱ་རི་ཥཱ། ཏང་ལ། བྲྀ་རྀཏ་ལས་ཡང་ན་འོས། ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས། བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨ་ཀ་རི། ཨ་ཀ་རི་ཥཱ་ཏྭཾ། ཨ་ཀ་རི་ཥ་ཏ། སོགས་སམ། ་ཨ་ཀ་རཱི། ཨ་ཀ་རཱི་ཥཱ་ཏྭཾ། ཨ་ཀ་རཱི་ཥ་ཏ། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་རིའི་ཨཏ་འོས། ་རཾ་ཨ་ཀུ་ཧ་དག་གི་ཙུ་འོས། ཀཙ་ཎ་ལ་གཉིས་ལ་འཕེལ་བ་དང༌། །གཞན་རྣམས་ལ། རིཙྪ་ཏི་རཱིཏ་རྣམས་ཀྱི་འོས་ཡོན་ཏན་བྱས་པས། ་་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཙ་ཀཱ་ར། ཙ་ཀ་ར་ཏུ། །ཙ་ཀ་རུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཀ་རི་ཐ། ཙ་ཀ་ར་ཐུ། །ཙ་ཀ་ར། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཀཱ་ར། ཙ་ཀ་ར། ཙ་ཀ་རི་བ། ཙ་ཀ་རི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཀ་རེ། ཙ་ཀ་རཱ་ཏེ། ཙ་ཀ་རི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙ་ཀ་རི་ཥ། ཙ་ཀ་རཱ་ཐེ། ཙ་ཀ་རི་དྷེ། ཙ་ཀ་རི་རྗེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཀ་རེ། ཙ་ཀ་རི་བ་ཧེ། ཙ་ཀ་རི་མ་ཧེ། །ལུཊ་ལ་བྲྀ་རྀཏ་ལས་ཡང་ན་འོས། ཨིཊ་་་དང་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ཨ་ཏང་ལ། ཀ་རི་ཏཱ། ཏཱ་ཀ་རི་ཏཱ་སི། སོགས་སམ། ཀ་རཱི་ཙིཏཱ། ཀ་རཱི་ཙིཏཱ་སི། སོགས་སོ། །ཏང་ལ། ཀ་རི་ཏཱ། ཀ་རི་ཏཱ་སེ། སོགས་སམ། ཀ་རཱི་ཏཱ། ཀ་རཱི་ོཏཱ་སེ། སོགས་སོ། །སྨོན་།
