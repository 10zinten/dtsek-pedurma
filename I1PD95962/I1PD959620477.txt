བུ་མོ། །མཁས་པ་ཀུན་སྐྱོ་:བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་ཁ་ཚོ་ལ་མཁོ་ཁྱོབ་པ་ཀ་ལ་ཀ༌ལ་རྟགས་ཨཙ་གཅིག་པ་མ་ཡིན་པའི་མཐའི་བུད་མེད་དུ་བརྗོད་པས་རྐྱེན་དཱིཔ་་་་དང་དཱིཥ་དག་ཐུང་ངུར་འགྱུར་ཏེ། བཱ་ཧྥ་ཎཱི་ལ། ཏ་ར་སོགས་སྦྱར་བས། བྲམ་ཟེ་མོ་མཆོག། བཱ་ཧྥ་ཎི་ཏ་རཱ། ཏ་རཔ་ཊཱ་པ། བཱ་ཧྥ་ཏ་མཱ། ཏ་རཔ་ཊཱ་པ། བྲམ་ཟེ་མ་མོ་བརྫུས་པ་མ༑བཱ་ཧྨ་ཎི་རཱུ་པཾ། རཱུ་པ་པ། ཅུང་ཟད་དམན་པའི་བྲམ་ཟེ་མོ། བཱ་ཧྥ་ཎི། ཀཝཱ། ཀལྤ་ཌ་པ། ཅུང་ཟད་དམན་པའི་བྲམ་ཟེ་མོ། བཱ་ཧྥ་ཎི་ཙེ་ལཱི། ཙེ་ལཊ་དཱིབ། ཅུང་ཟད་དམན་པའི་བྲམ་ཟེ་མོ། བཱ་ཧྥ་ཎི་བྲུ་་་བཱ༑བྲུ་བ་ཆ་པ། བྲམ་ཟེ་མོའི་རིགས། གྲྭ་ཧྥ་ཎི་གོ་ཏཱ། གོ་ཏྲ་ཊཱཔ། །ཅུང་ཟད་དམན་པའི་བྲམ་ཟེ་མོ། བཱ་བྷཱ་ཎི་མ་ཏྲ། མ་ཏ་ཊཔ། །ཅུང་ཟད་དམན་པའི་བྲམ་ཟེ་མོ། བཱ་ཧྥ་ཎི་ཧ་ཏཱ། ཧ་ཏ་ཊ་པ། སོགས་ཏེ་མདོ་འདིས་དཱི་མཐའ་ཐུང་ངུར་བསྒྱུར་བའོ། །ན་yཨ་ནྱ་ཏ་ར་ཧཱིུྃ། ན་དཱིའི་ཡང་ནའོ༑། དཱི་མཐའ་མ་ཡིན་ཀྱང་བཤད་མ་ཐག་པའི་རྐྱེན་རྣམས་ཕ་རོལ་ལ་ན་དཱིའི་་་མིང་ཅན་རྣམས་ཀྱིའང་ཐུང་ངུར་ཡང་ན་འགྱུར་ཏེ། བ་མོ་རུ་ཏ་རཱམཆོག་ཏུ་བརླ་མཛེས་མ། བ་མོ་རཱུ་ཏ་རཱ། ལྡི་ཏ་རཱ། བུད་མེད་མཆོག་མ། སི་ཏེ་རཱ། བུད་མེད་མཆོག་མ། སོགས་ཏེ་རྟགས་མཐའ་ཐུང་དུར་བྱེད་པ་དང་མི་བྱེད་པ་ཡང་ན་བ་སྟེ་རྐྱེན་གཞན་རྣམས་ལའང་རིགས་འགྲེའོ། །ན་དཱིའི་མིང་ཅན་ཡིན་ཀྱང་ལགཱི་དང་ཏ་རཱི་དང་ཏནི་་རྣམས་ནི་ཐུང་ངུར་མི་འགྱུར་ཏེ། ལགཱི་ཏ་རཱ། …དཔལ་མཆོག་མའམ་མཆོག་ཏུ་ཕུན་སུམ་ཚོགས་པ། ཏ་རཱི་ཏ་རཱ། གྲུ་མཆོག་མ། ཏནི་ཏ་རཱ། སྒྲ་སྙན་མཆོག་མི། སོགས་སོ། །ཨུ་གི་ཏ། ཨུཀ་འགྲོ་བ་ལས་ཀྱང་ངོ་། །རྟགས་ཨུཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་འགྲོ་བ་ལ་དཱི་་་བྱིན་པའི་མཐའ་ཡང་རྐྱེན་གྷའི་མིང་ཅན་སོགས་ཕ་རོལ་ལ་ཐུང་ངུར་ཡང་ན་འགྱུར་ཏེ༑བི་དྭ་སུ་ལ་ཥ་འཕྲོ་བ་གོ་ར་སོགས་ཀྱིས་དཱིཔ་ཨུ་འགྲོ་བ་བ་སུའི་སླར་ཡང་དག་པར་འགྱུར་རོ་ཞེས་པས་བ་ཨུ་མདོ་་་འདིས་ཡང་ན་ཐུང་ངུར་འགྱུར་བ་དང་མི་འགྱུར་བ་ཡང་ན་བ། བི་དུ་ཥི་ཏ་རཱ། མཆོག་ཏུ་མཁས་མ། བི་དུ་ཏཱི''ཏ་རཱ། མཆོག་ཏུ་མཁས་མ། ས་ཏྲི་ལ་ཥིཏ་གོ་ར་སོགས་ཀྱིས་དྲི་བ་རི་འགྲོ་བ་མདོ་འདིས་ཐུང་ངུར་འགྱུར་བ་དང་མི་་འགྱུར་བ་ཡང་ན་བ། ས་ཏི་ཏ་རཱ། མཆོག་ཏུ་ངང་ཚུལ་མ། ས་ཏཱི་ཏ་མཱ། མཆོག་ཏུ་ངང་ཚུལ་མ། པྲ་གི་།
