སྒྲ་མདོ། ཙཏི་་་་་ཚོ་་་་་་་་་ལ་བློ་ཁོག་བ་ཙཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བཀ། ཨ་བག། ཨ་བཏྟཾ། ཨ་བ་ཙན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བཀ། ཨ་བཀ། ཨ་བཏྟཾ། ཨ་བཀྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བ་ཙཾ། ཨ་བ་ཙཱ། ཨ་བཙྪ། ལུང་ལ། ཨ་སྱ་ཏི། བཀྟི། ཥཱ་ཏི་རྣམས་ལས་ཨང་ངོས་ཙྪི་ཨང་དུ་བསྒྱུར། ཨ་བཛྲ། ཨཏྲ་ཅེས་པ་ལ། ་བ་ཙ་ཨུམ། བ་ཙའི་ཨུམའོ། །བ་ཙའི་ཙྪི་བསྒྱུར་བའི་ཨང་། ཕ་རོལ་ལ་ཨུམ་ཨཱ་ག་མ་སྟེར་ཏེ། མ་གནས་ཀྱི་དོན། ཨ་ལས་ཡོན་ཏན་ནོས། མཚམས་སྦྱར་བས། ཨ་བོ་ཙད། དེ་བཞིན་དུ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨ་བོ་ཙ་ཏྭཾ། ཨ་བོ་ཙན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བོ་ཙ། ཨ་བོ་ཙ་ཏཾ། ཨ་བོ་ཙ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བོ་ཙོ༄༅། 《ཨ་བོ་ཙཱ་བ། ཨ་བོ་ཙཱ་མ། ལིཊ་ལ་ཎལ་གཉིས་དང་ཐལ་ལ་གཉིས་རྣམས་ཀྱི་སོགས་ཀྱིས་ཨ་བྷཱུ་ས་སཾ་པྲ་སཱ་ར་ཎ་དང་ཀ་ང་འགྲོ་བའི་མིང་ཅན་རྣམས་ལ། ལབ་ཙི་སྭཱ་པི་ཡ་ཛ་སོགས་ཀྱིས་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར། ཨག་འདྲ་བ་ལ་རིང་པོ་སོགས་་་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཨུ་བཱ་ཙ། ཨུ་ཙ་ཏུ། ཨུ་ཙུ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨུ་བ་ཙི་ཐ། སཨུ་བཀྑ། ཨུ་ཙ་ཐུ། ཨུ་ཙ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨུ་བཱ་ཙ། ཨུ་བ་®ཙ། ཨུ་ཙི་བ། ཨུ་ཙི་མ། ལུཊ་ལ། །བཀྟ། བཀྐཱ་རོ། བཀཱ་རཿ། བཀཱ་སི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། །བ་ཙི་སཱ་པི་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎ་བྱས་པས། ཨུ་ཙྪཱཏ། ཨུ་ཙཱཙཱ༑ཨུ་ཙཱ་སུསོགས་སོ། །ལྀཊ་ལ་ཙུའི་ཀུ་འོས་ཙ་ཀ། ཀུན་ནས་བསྒྱུར་་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། བཀྵ་ཏི་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ༑ཨ་བགྱུཏ་སོགས་སོ། །༈རུ་དིར་མཆི་མ་འཛག་པ་ལའོ་ཞེས་པ་ལའང་པ་རསྨི་པ་ད་རྣམས་་་།
