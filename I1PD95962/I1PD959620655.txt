སྒྲ་མདོ། རྟའི་ཀྱགས་པའི་བློ་ནི་ས་ཆ་ཉཁྱོལ་དཀ་ཀ་ལ་་བློ་ཏང་ལ། བྷ་རི་ཏཱ། བྷ་རི་ཏཱ་སེ་སོགས་སམ། བྷཏཱ། བྷཎཱ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ཤ་ཡཀ་ལིང་རྣམས་ལ་རིང་འོས་རི་རིང་དུ་བསྒྱུར་བས༑བྷི་ཡཱཏ་སོགས་དང༌། ཏང་ལ། རྀ་ལས་ཀྱང་ངོས་ཀིཏ་ཡིན་པས་ཡོན་ཏན་སོགས་མི་བྱེད་དེ། ཤྲཱི་ཊྛིཏྠ་སོགས་སོ། །།ཁྲིཊ་དང་ཚིགསང་ལ། །རྀཏ་་་ཧན་དག་གི་སྐྱ་ལ་འོས། ཨིཊ་བྱིན་པས། ཁྲིཊའི་ཨ་ཏང་ལ། བྷ་རི་ཥྛ་ཏི་སོགས་དང༌། ཏང་ལ། །བྷ་རི་བྱ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་གི་ཨ་ཏང་ལ། ཨ་བྷ་རི་ཏ་སོགས་དང་། ཏང་ལ། ཨ་བྷ་རི་ཥྱ་ཏ་སོགས་སོ། །༈དཱ་ན་ཀུན་ནས་འགེམས་པ་ལའོ་ཞེས་པ་ལ། །མཱན་བ་དྷ་དཱན་སོགས་ཀྱི་སན་དང་སན་ལ་ཨཏ་ཀྱི་འོས། ་ཨ་བྷྱཱ་ས་ཨིཏ་མཱན་བ་དྷ་དཱན་སོགས་ཀྱི་ཨཱིཏ་དུ་་་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། དཱི་དཱརྨ་ཞེས་གྲུབ་པ་ལ། ལཊ་སོགས་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱོར་ཏེ། བྱིངས་ཉིད་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་ཀྱང་སན་་་མཐའ་ལ་ཏང་རྣམས་འགྱུར་ཞེས་བཤད་དེ། ལཊ་ལ། དཱི་དཱནྣ་ཏེ་སོགས་སོ། །ལིང་ལ། དཱི་དཱནྡྷེ་ཏ་སོགས་སོ། །ལོཊ་ལ། དཱི་དཱནྣ་ཏྭཾ་སོགས་སོ། །ལང་ལ། ཨ་དཱི་དཱརྨ་ཏ་སོགས་སོ། །ལུང་ལ། ཚུཝལ་དང་པོའི་ཨིཊ་དང་ཨཊ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། ཨ་དཱི་དཱནྟིཏྠ་སོགས་སོ། །ལིཊ་ལ། ཀཱས་སོགས་ཀྱིས་ཨཱམ་སྦྱར་བས། དཱི་དཱནཾ་ཙཀྲེ། དཱི་དཱནཾ་བ་བྷུ་བ། དཱི་དཱནཱ་མཱས་སོགས་སོ། །ལུཊ་ལ་ཡང་ཨིཊ་བྱིན་པས། དཱི་དཱནྟི་ཏཱ་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། དཱི་དཱནྟི་ཏིཥྛ་སོགས་སོ། །ལྀཊ་ལ། དཱི་དཱནྟི་ཥྛ་ཏེ་སོགས་སོ། །ཚོུང་ལ། ཨ་དཱི་དཱནྟི་བྱ་ཏ་སོགས་སོ། །༈ཤཱན་རྣོ་བ་ལའོ་ཞེས་པ་ལ་ཡང་སན་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཤཱི་ཤཱནྣ་ཏེ་སོགས་ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དཱན་གྱི་སྐབས་སུ་བཤད་་་་།
