ཧྥ་བྷས་འདྲ། ་ངས་ལ། པྲ་ཡཙ་ཏུར། ཨོས་ལ། པྲ་ཡ་ཙ་ཏུ་རོ། །ཨཱམ་ལ། གྲངས་:བགྲོང་བའི་སྐབས་མ་གཏོགས་ནུཊ་མི་འཇུག། དྲི་ཡ་ཙ་ཏུ་རཱི། ཇི་ལ། པྲ་ཡ་ཙ་ཏུ་རི། ཨོས་ལ། པྲ་ཡ་ཙ་ཏུ་རོ། །པྲ་ཡ་ཙ་ཏུནྟུ། སཾ་བུདྡྷི་ལ་ཨོཾ། ཧེ་ཙ་ཏཱི། ཧེ་ཙ་ཏཱ་རོ། ཧེ་ཙ་ཏཱ་ར། མ་མཐའ་ཅན་ལ། པྲ་ཤཱམ་ལ་སུ། མོ་ནོ་དྷཱ་ཏོ། བྱིངས་ཀྱི་མའི་ནའོ། །བྱིངས་ཀྱི་མ་ནི་ཛྷལ་ས་་་སོར་བསྡུ་བའི་ཡི་གེ་རྣམས་ཕ་རོལ་དང་ཚིག་མཐའ་ལ་ན་རུ་འགྱུར་ཏེ། མན། སུ་ཕྱིས། པྲ་ཤཱན། ་ཨོ་དང་ཛས་ལ། པྲ་ཤཱ་མོ། །པྲ་ཤཱ་མ། གཉིས་པ་་་་རྣམས་ལ། །པྲ་ཤཱ་མཾ། པྲ་ཤཱ་མོ། པྲ་ཤཱ་མ། ཊཱ་ལ། ་པྲ་ཤཱ་མཱ། བྷཱུ་ལ། པྲ་ཤཱཙྪཱ། བྷིས་ལི། པྲ་ཤཱནྟི༄༅། །བཞི་པ་རྣམས་ལ། པྲ་ཤཱ་མེ། པྲ་ཤཱཙྪཱ། པྲ་ཤཱནྟུ། ང་སི་ལ། པྲ་ཤཱ་མ། བྷཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། པྲ་ཤཱ་མ༑པྲ་ཤཱ་མོ། སྲ་ཤཱ་ནཱཾ། བདུན་པ་རྣམས་ལ། པྲ་ཤཱ་མི། པྲ་ཤཱ་མོ། པྲ་ཤཱནྟུ། ཧེ་པྲ་ཤཱན། །ཧེ་པྲ་ཤཱ་མོ། ཧེ་པྲ་ཤཱ་མ། 'ཀིམ་ལ། ཅི། སུ། ཀིམཀ། ཀིམ་གྱི་ཀའོ། །ཀིམ་ཞེས་པའི་སྒྲ་ནི་རྣམ་དབྱེ་གང་མཐོང་ཡང་ཀ་ཞེས་པར་འགྱུར་རོ། ་། ཀ། ཀོ། ཀེ། སོགས་ཏེ་ཕོ་རྟགས་སརྦའི་སྐབས་་་་དང་འདྲའོ། །སརྦ་ནཱ་མའི་ནང་ཚན་ཨི་འདི་དམ་ལ། སུ། །ཨི་ད་མོ་མ། ཨི་དམ་གྱི་མའོ། །ཨི་དམ་ཞེས་པའི་མ་ནི་སུ་ཕ་རོལ་ལ་མ་ཉིད་དུ་གནས་ཏེ། ཏྱད་སོགས་རྣམས་ཀྱི་ཨའོ་ཞེས་པ་འགོག་པའི་དོན་ནོ། །ཨི་ད་མོ་ཡ་་སི། ཨི་དམ་གྱི་ཨཕྲ་སྐྱེས་པ་ལའོ། །ཨི་དམ་ཞེས་པའི་ཨི་དང་ད་སྲོག་མེད་ནི་སྐྱེས་པའི་་་རྟགས་ཀྱི་སྐབས་སུ་ཕ་རོལ་ལ་ཡང་ཨཡ་ཞེས་པར་འགྱུར་ཏེ། ཨ་ཡཾ། ཨི་དམ་ལ། ཨོ།  ི་ཏྱད་སོགས་རྣམས་ཀྱི་ཨ་འོས་མ་ཨ། །ཨ་ཏོ་གུ་ཎེ། །ཨད་ཀྱི་ཡོན་ཏན་ལའོ། །ཚིག་མཐའ་མ་ཡིན་པའི་ཨཏ་མཐའ་ལས་ཨི་དམ་གྱི་མ་ཨར་འགྱུར་་་་་།
