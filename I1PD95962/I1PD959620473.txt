བསྟན་འགྱུར། སྒྲ་མདོ། ་ཏི། ::ས་ཆེན་སྐྱོ་ཁབོད་ཁལ་ཁ་རྒྱལ་ཏི། བློ་གེ་ཡོད་ན་དབྱི་སྟེ། །རྟགས་ཀྱི་དང་པོའི་ཨཙ་དང་པོའང་ཡོན་ཏན་དུ་འགྱུར་ཏེ། སྦྲ་བིཥྛ། ཆེས་སྦོམ་པ། ཆེས་སྦོམ་པ། …སྦྲ་བཱི་ཡཱན། སྒྱུ་ལ་ལ་ཨི་སྐྲུན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡ་ཎའི་ཡི་གེ་ལ་ཕྱིས་ཨུ་ཡོན་ཏན་ཨིཙའི་ཨཱ་ཡ་སོགས། ཆེས་རིང་བ། ད་བིཥྛ། དྷུ་ར་ལ་ཨི་ཊྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ར་ཕྱིས་དྲུ་ཡི་ཨུ་ཡོན་ཏན་ཨེ་ཙའི་ཨ་ཡའོ།  ོ། ཆེས་རིང་བ། ད་བཱི་ཡཱན། ཚེས་དར་བབ། ཡ་་་བིཥྛ༑ཡུ་བ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡ་ཎའི་ཡི་གེ་ཝ་ཕྱིས་ཨུ་ཡོན་ཏན་ཨེ་ཙའི་ཨ་ཡ་སོགས། ཡ་བཱི་ཡཱན། ཆེས་དར་བབ། ཧྲ་སཱ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ཝ་ཕྱིས། ཧྲ་་་སིཥྛ༑ཡཆེས་ཐུང་བ། ཆེས་ཐུང་བ། ཧྲ་སཱི་ཡཱན། ཧྲ་སཱ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ཝ་ཕྱིས། ཀྱི་པྲ་ལ་ཨཥྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ར་ཕྱིས་ཨི་ཡོན་ཏན། ཀྵེ་པིཥྛ། ཆེས་་མགྱོགས་པ། ཀེ་པཱི་ཡཱན། ཀྱི་ཕྲ་ལ་ཨཊྚན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ར་ཕྱིས་ཨི་ཡོན་ཏན། ཆེས་ཕྲན་ཚེགས། ...ཀོ་དིཥྛ། .ཀླུ་དྲ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ར་ཕྱིས་ཨུ་ཡོན་ཏན། ཆེས་ཕྲན་ཚེགས། ཀོ་དཱི་ཡཱན། གླུ་དྲ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ར་ཕྱིས་ཨུ་ཡོན་ཏན། ཨི་མན་ནི་དྲི་ཐུ་ལ་སོགས་པའི་སོགས་ཁོང་ཧྲ་སཱ་དང་། ཀླུ་པྲ་དང་། གླུ་དྲ་་་གསུམ་ལ་འཇུག་པ་ཉིད་དེ་སྒྱུ་ལ་སོགས་འདི་རྣམས་ཐམས་ཅད་ལ་འཇུག་པ་མ་་་ཡིན་ནོ། །ཏྲ་སྭ་ལ་ཨི་མན་ཊི་ཕྱིས་མདོ་འདིས་ཡཎའི་ཡི་གེ་ཝ་ཕྱིས་པའོ། །ཧྲ་སི་མ། ཐུང་དུ་ཉིད། གྱི་པྲ་ལ་ཨི་མན། ཀེ་པི་མ། མགྱོགས་པ་ཉིད། ཕྲན་ཚེགས་ཉིད། ཀློ་དི་མ། གླུ་དྲ་ལ་ཨི་མན། པྲི་ཡསྦི་རསྤྱི་རོ་རུ་བ་ཧུ་ལ་གུ་རུ་ཤྲཱིདྡྷཱ་ཏྲཔཱ་དཱིགྷ་ཤྲཱིནྡ་ར་ཀཱ་ཏཱཾ་པྲསྦྲ་སྤ་བརྦ་ཧི་གརྦནྟི་ཏྲ་པ་དཱ་གྷི་ཧྲིན། པྲ་ཡ་སྦི་ར་སྤྱི་ར་ཨུ་རུ་བ་ཧུ་ལ་གུ་རུ་བྲིདྡྷཱ་ཏྲཔཱ་དཱིགྷ་བྲིན་ར་ཀ་རྣམས་་ཀྱི་པྲ་སྦྲ་སྤ་བ་ར་བི་ཧི་ག་ར་བནྟི་ཏྲ་པ་དཱ་གྷི་བྲིནྡ་རྣམས་སོ། །ཨིཥྛན་དང་ཨཱི་ཡ་་སུན་ཨི་མན་རྣམས་ཕ་རོལ་ལ་དྲི་ཡ་ལ་སོགས་པ་འདི་རྣམས་རིམ་པ་བཞིན་པྲ་ལ་་་སོགས་པ་འདི་རྣམས་སུ་འགྱུར་ཏེ། པྲི་ཡ་ལ་ཨིཏྟན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་སྲི་ཡ་པྲ་རུ་འགྱུར་བར་ཡོན་ཏན། ཕྲེཊྛ། ་མཆོག་ཏུ་དགའ་བ། པྲ་ཡཱན། ཕྱི་ཡ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་སྲི་ཡ་པྲ་རུ་འགྱུར་།
