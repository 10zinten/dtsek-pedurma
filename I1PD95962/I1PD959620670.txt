ཨ་བྷྱཱ་ས་སཾ་པྲ་སཱ་ར་ཎ་བྱས། ཨཙའི་ཉཎ་འགྲོ་བ་ལ་འོས་འཕེལ་བ་བྱས་པས། བི་བྱཱ་ཡ། ཎལ། འགྲེལ་པར་ལུང་དྲངས་པ། བྱེཉཀི་ཏི་ལི་ཊི་སཾ་པྲ་སཱ་ར་ཎཾ་བཱ། བྱེཉའི་ལིཊ་ཀིཏ་ལ་སཾ་པྲ་སཱ་ར་ཎ་ཡང་ནའོ། །བྱེཉའི་ཨ་ངྷ་སར་མ་ཟད་ལིཊ་ཀིཏ་ཕ་རོལ་ལ་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎར་ཡང་ན་བསྒྱུར་ཏེ། ་ཨ་བྷྱཱ་ས་ཉིད་སཾ་པྲ་སཱ་ར་ཎ་བྱས་པའི་ཕྱོགས་ཀྱི་རྣམ་དབྱེས་ཨཙ་དང་པོ་ཅན་རྣམས་ལ། །ཨེཙའི་ཨ་ཡ་ཨ་ཝ་སོགས་དང༌། གཉིས་ཀ་སཾ་པྲ་ས་ར་ཎའི་ཕྱོགས་ལ་ཨིཀ་གི་ཨཙ་ལ་ཡཎ་བྱས་པས། སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་གཉིས་ལ། བི་བྱ་ཡ་ཏུ། བི་བྱ་ཏུ། བི་བྱ་ཡུ། བི་བྱུ། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བྱ་ཡི་ཐ༑ཡཨཏྟི་ཨརྟི་བྱ་ཡ་ཏི་རྣམས་ཀྱི་ཨིཊ་འོས། ཐལ་ལ་ཨིཊ་ཡང་ན་བར་མི་བྱེད། བི་བྱ་ཡ་ཐུ། བི་བྱ་ཐུ། བི་བྱ་ཡ། བི་བྱ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷཱ་ཡ། །བི་བྱ་ཡ། བི་བྱ་ཡི་བ། བི་བྱི་བ། །བི་བྱ་ཡི་མ། བི་བྱི་མ། དེ་བཞིན་དུ། ཏང་ལ་ཡང་། བི་བྱ་ཡེ། ཏ། བི་བྱེ། བི་བྱ་ཡཱ་ཏེ། ཨཱ་ནཱི། བི་བྷཱུ་ཏེ། བི་བྱ་ཡི་རེ། ཛྷ། བི་བྱི་རེ་་་སོགས་སོ། །།།ལུཊ་ལ། ཤིཏ་མ་ཡིན་པ་སོགས་ཀྱིས་ཨེཙ་ཨཱཏ་དུ་བསྒྱུར་བས། ཨ་ཏང་ལ། ཌ་བྱཱ་ཏཱ། བྱཱ་ཏཱ་སི་སོགས་དང་། ཏང་ལ། བྱཱ་ཏཱ། བྱཱ་ཏཱ་སེ་་་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ཤིཏ་མ་ཡིན་པ་སོགས་ཀྱིས་ཨེཙ་ཨཱཏ། བ་ཙི་གྲྭ་པ་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎ་དང་། ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་པས། བཱི་ཡཱཏ། བཱི་ཡཱནྟཾ། བི་ཡཱ་སུསོགས་དང༌། ཏང་ལ། །བྱཱ་སཱི། བྱཱ་སཱི་ཡཱཙཱ། སོགས་སོ། །ལྀཊ་ལ་ཨེཙ་ཨཏ་དུ་བསྒྱུར་བས། ཨ་ཏང་ལ། བྱཱ་སྱ་ཏི་སོགས་དང་། ཏང་ལ། བྱཱ་སྱ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ཕྲེང་གི་ཨ་ཏང་ལ། ཨ་བྱཱ་སྱ་ཏ་སོགས་དང་། ཏང་ལ། ཨ་བྱཱ་སྱ་ཏ་སོགས་སོ། །༈ཧཱིཉ་འགྲན་པ་ལ་སྟེ་སྒྲ་ལ་ཡང་ངོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཏེ། —།
