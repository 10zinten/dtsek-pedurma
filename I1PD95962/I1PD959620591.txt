བ་བླ་ནག་པ་བར་མའི་བྱ་དང་་་་བསྟན་འགྱུར། སྒྲ་མདོ། དའིཌ་བྷི་ནི་ལོ་མཐོ་nཙོང་ཁ་བ་བློ་བློ་ལ་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨཱ་ར་ཏུ། ཨཱ་རུ། །སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ཨི་ཌཏྟརྟི་བྱ་ཡ་ཏཱི་ནཾ། ཨཏྟི་ཨརྟི་བྱ་ཡ་ཏི་ོརྣམས་ཀྱི་ཨིཊའོ། །།བྱིངས་ཨ་ད་ཟ་བ། རྀ་འགྲོ་བ། །བྱཉ་སྒྲིབ་པ་རྣམས་ལས་ཕ་རོལ་གྱི་ཐལ་ལ་ཨིཊ་རྟག་ཏུ་འགྱུར་ཏེ། ཨཱ་རི་ཐ། སྐྱེས་བུ་དབུས་མའི་གཉིས་་ཚིག་དང་མང་ཚིག་ལ། ཨཱ་ར་ཐུ། ཨཱ་རུ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨཱ་ར༑ཨཱ་རི་བ། ཨཱ་རི་མ། ལུཊ་ལ། ཨཏཱ། ཨཏཱ་རོ། ཨཏཱ་ར། སྨོན་པའི་ལིང་ལ། ཨརྟི་བརྩེགས་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨན་ཏི། ཨརཱ༑༑ཨན་སུསོགས་སོ། །ཁྲིཊ་ལ། རྀཏ་ཧན་སོགས་ཀྱིས་ཨིཊ་སྦྱར་བས། ཨ་རི་ཥྛ་ཏི། ཨ་རི་ཥྱ་ཏ། /ཨ་རི་ནྟི་སོགས་སོ། །དེ་བཞིན་དུ། ལྭང་ལ། ཨཱ་རི་ཏ། ཨཱ་རི་ཥྛ་ཏྭཾ། ཨཱ་རི་ན་སོགས་སོ། །ཐུ་བ། ཤྲཱི་ཙ། ཐུའི་ཤྲཱི་འང་ངོ༌། །བྱིངས་ཐུ་ཐོས་པ་ལའོ་ཞེས་པ་ལ་བྱིངས་ཐམས་ཅད་པ་རྣམས་ཕ་རོལ་ལ་བྱེད་པ་པོ་ལ་ཤཔ་ཀྱི་དོན་ལ་ལྕུ་འགྱུར་ཞིང་དེར་མ་ཟད་བྱིངས་ཐུ་ཉིད་ཀྱང་ཤྲཱི་ཞེས་པར་འགྱུར་ཏེ། ཤ་ཡིག་ནི་ཤ་འགྲོ་བའི་དོན་ནོ། །ལཊའི། སྐྱེས་བུ་་དང་པོའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ། ་ཤྲཱི་ཎོ་ཏི། ར་ཥ་སོགས་ཀྱིས་ན་ཎ་བྱིངས་བྱེད་པ་དང་བྱིངས་ཐམས་ཅད་པ་དག་གི་འོས་ཡོན་ཏན། བྲི་ཆུ་ཏ། ཧུབྷྱཱ་བོ། སཱརྦ་དྷཱ་ཏུ་ཀེ། ཧུ༔་དག་གི་བྱིངས་ཐམས་ཅད་པ་ལའོ། ་། བྱིངས་སུ་སྦྱིན་པ་དང་རྐྱེན་ལྕེ་དང་གཞན་ཡང་བྱིངས་ཨཙ་གཅིག་པ་དང༌། ཡི་གེ་བརྩེགས་པ་མ་ཡིན་པ་རྣམས་ཀྱི་་་མཐའི་ཨུ་ཡིག་ནི་རྣམ་དབྱེ་བྱིངས་ཐམས་ཅད་པ་ཨཙ་དང་པོ་ཅན་རྣམས་ཕ་རོལ་ལ་ཡ་ཎར་འགྱུར་ཏེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། ་ཤྲི་ཧྥནྟི། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྲྀ་ཌོ་ཎི། ཤྲཱི་ཆུ་ཐ། ཤྲཱི་ཎཱ་ཐ། སྐྱེས་བུ་ཕྱི་མའི་གཅིག་ཚིག་ལ། ཤྲཱི་ཎོ་མི།  ོལོ་པཉྩཱ་སཱུ་ནྱ་ཏ་ར་སྟི་བློ། །བ་མ་དག་ལ་འདིའི་ITTནྟི_—།
