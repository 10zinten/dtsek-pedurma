ཁྱོལ་དང་། ཀ་མཚོ་ཁག་མ་གསུམ་ལ། བྲུ་ཧི། བྷུ་ཏཱཏ། བྲུ་ཏཾ། བྷུ་ཏི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བྲ་བཱ་ཎི། བྲ་བཱ་བ། བྲ་བཱ་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བྷུ་ཏཱཾ། བྲུ་བཱ་ཏཱ། བྲུ་བ་ཏཱི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྲུ་ཥཱ༑བྲུ་བཱ་ཐཱ། བྷཱུ་དྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བྲ་བི། བྲ་བཱ་བ་ཧི། བྲ་བཱ་མ་ཧི། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྲ་བཱིཏ། ཨ་བྲུ་ཏཱ༑ཨ་བྲུ་བན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བྲ་བཱི། ཨ་བུ་ཏཾ། ཨ་བྷུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བྲ་ཁཾ། ཨ་བྲུ་བ། ཨ་བྲུ་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྲུ་ཏ། ཨ་བྲུ་བཱ་ཏཱ། ཨ་བྲུ་བ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བྲུ་ཐཱ། ཨ་བྲུ་བཱ་ཐཾ། །ཨ་བུ་དྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བྲུ་བི། ཨ་བྲུ་བ་ཧི། །ཨ་བྲུ་མ་ཧི། བྲུ་བོ་བ་ཙི། བྲུའི་བ་ཙིའོ༑། རྐྱེན་བྱིངས་ཕྱེད་པའི་མིང་ཅན་ཕ་རོལ་ལ་བྲུའི་བ་ཙི་ཞེས་པར་བསྒྱུར་ཏེ། ཨི་ཡིག་བརྗོད་པའི་དོན་ནོ། །བྲུ་ལ་ལུང༌། །མདོ་འདིས་བྲུ་བ་ཙིར་བསྒྱུར། ཨ་སྱ་ཏི། བཀྟི་སོགས་ཀྱིས་ཙ་ལི་ཨང་དུ་བསྒྱུར། །བཙའི་ཨུམ་འོས། ཨུ་ཨཱ་ག་མ་བྱིན། ཨཏ་ལས་ཡོན་ཏན་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ་ཨ་བོ་ཙ་ཏ། ཨ་བོ་ཙ་ཏྭཾ། ཨ་བོ་ཙ་ན། སོགས་སོ། །དེ་བཞིན་དུ། ཏང་ལ་ཨ་བོ་ཙ་ཏ། ཨ་བོ་ཙེ་ཏྭཾ། ཨ་བོ་ཙནྟ། སོགས་སོ། །ལིཊ་ལ། བྲུ་བ་ཙར་བསྒྱུར་ཀིཏ་མ་ཡིན་པ་རྣམས་ལ། ་གཉིས་རྣམས་ཀྱི་སོགས་ཀྱིས་ཨ་ཧྥུ་ས་སཾ་པྲ་སཱ་ར་ཎ་དང་ཀིཏ་རྣམས་ལ། །བ་ཙི་སོ་པི་སོགས་ཀྱིས་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་བས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨུ་བཱ་ཙ། ཨུ་ཙ་ཏུ། སྤུ་ཙུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨུ་བཀྑ། ཨུ་བ་ཙི་ཐ། ཨུ་ཙ་ཐུ། སྤུ་ཙ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨུ་བཱ་ཙ། ཨུ་བ་ཙ། ཨུ་ཙི་བ། ཨུ་ཙི་མ། དེ་བཞིན་དུ། ko—།
