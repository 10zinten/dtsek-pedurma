སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཋ་ཨ་བིཏྟཾ། ཨ་བིཀླུ། །ཨ་བིགྷཱ། ཏང་ལ་ལིང་སིཙ་དག་ལ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ་འོས་འཕེལ་བ་དང་ཡོན་ཏན་གང་ཡང་མི་བྱེད། ཛྷལ་དང་པོ་རྣམས་ལ། །ཛྷལ་ལས་ཛྷལ་ལ་འོས་སིཙ་དབྱི་བ་དང༌། ཙུའི་ཀུ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བིཀྟ༑ཨ་བིཀ་ཏྭཾ། ཨ་བིཀྵ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བིཀྐཱ། ཨ་བིགྷཱ་ཐཱ། ཨ་བིཤྩ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བིཀྵ། ཨ་བིཀྵ་ཧི། ཨ་བིཀྐཱ་ཧི། ལིཊ་ལ། ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བེ་ཛ། བི་བི་ཛ་ཏུ། བི་བི་ཛུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བཀྑ། བི་བེ་ཛི་ཐ། བི་བི་ཛ་ཐུ། བི་བི་ཛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བེ་ཛ། བི་བི་ཛི་བ། བི་བི་ཛི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛཱ་ཏེ། བི་བི་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བི་ཛི་ཥ། །བི་བི་ཛཱ་ཐེ་། བི་བི་ཛི་ྋདེ། བི་བི་ཛི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛི་བ་ཧེ། བི་བི་ཛི་མ་ཧེ། ལུཊ་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཙུའི་ཀུ་འོས་ཛ་ག། ཁར་ལ་ཙར་འོས་ག་་་ཀར་བསྒྱུར་བས། ཨ་ཏང་ལ། བེཀི། །བེཀ་སི། སོགས་དང༌། ཏང་ལ། བེཀ༑:བེཀསེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ་འདྲེན་པ་ཁོ་ན་སྟེ། བི་ཛཱཏ། བི་ཛཱ་ཧཱུཾ། བི་ཛཱ་སུ། ་སོགས་དང༌། ཏང་ལ། ཙུའི་ཀུ་འོས། ཛྲ་ག་། ཁར་ལ་ཙར་འོས། གྲ་ཀྲ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། བིཀྵིཀྵ། བིཤཱི་ཡཱནྟཾ། བི་ཀཱི་རན། སོགས་སོ། །།ཁྲིཊ་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཙུའི་ཀུ་འོས་ཛྲ་གྲ། ཥར་ལ་ཙར། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ཨ་ཏང་ལ། པཀྵ་ཏི། སོགས་དང་། ཏང་ལ། །
