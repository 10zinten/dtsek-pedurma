བེ་བིཥྛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཚྭ་མི། བེ་བི་གླི། །བེ་བི་ཤྭ། བེ་བི་ཚྭ། །བེ་བི་ཐཱི། ལིང་ལ། བེ་བི་ཚུཏ། བེ་བི་ཚྭ་ཏྭཾ། བེ་བི་ཚུ། སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་བི་ཛཱི་ཏུ། བེ་བིཥྛ། བེ་བིཏྟཾ། བེ་བི་ཚ་ཏུ། སྐྱེས་བུ་དབུས་མ་་གསུམ་ལ། །བེ་བི་བྷི?། བེ་བིཏྟཱཏ། །:བེ་བིཤྩ། བེ་བིཊྚི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བི་ཚྭ་ནི། བེ་བི་ཚྭ་བ། བེ་བི་ཚྭ་མ། དེ་བཞིན་དུ། ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བེ་བི་ཚིཏ། ་ཨ་བེ་བེཊ། །ཨ་བེ་བིཏྟཾ། ཨ་བེ་བི་ཚུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བེ་བི་ཛཱི། །།ཨ་བེ་བིཊ། ཨ་བེ་བིé༑་ཨ་བེ་བིཊྚ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བེ་བི་ཚེ། །ཨ་བ་བི་ཤུ། ཨ་བ་བི་ཚྭ། ཨ་བ་བི་། ལ་ཡིག་ལྷག་མ་རྣམས་ལ། ལུང་…ཨ་བེ་བི་ཛཱིཏ། ལིཊ་བེ་བི་ཚྭ་ཙ་ཀཱ་ར། ་ལུཊ་བེ་བི་ཚི་ཏཱ། སྨོན་ལིང་བེ་བི་ཚུཏ། ཁྲིཊ་བེ་བི་ཚི་ཥྛ་ཏི། ་ཕྲེང་…ཨ་བ་བི་ཚི་ཏ། སོགས་སོ། །ས་བརྗེ་བ་ག་ཚོད་དང་། །།།ལ་པོ་བ་དང༌༈ཡུ་བསྲེ་བ་ལའོ་ཞེས་པ་ལ། ཡང་སྦྱར། །་གཉིས་བརྗོད། ཡང་འཇིག་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས། ཡང་གི་ཨ་ཙ་ལའང་ངོས་ཡང་ཕྱིས་པས། ཡོ་ཡུ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཡང་ལས་ཡང་ན་འོས་ཨཱིཊ་ཡང་ན་བར་བྱས། །ཡོན་ཏན་བྱས་པས། །ཡོ་ཡ་བཱི་ཏི། ཡོ་ཡོ་ཏི། གཉིས་ཚིག་ལ། ཡོ་ཡུ་ཏ། མང་ཚིག་ལ། ཨ་བྷུས་ལས་ཨཏ་་འོས་ཛྷ་ཨཏ་དུ་བསྒྱུར། ་ཨཙ་ལ་ཟླ་བྱིངས་སོགས་ཀྱིས་ཨུ་བར་བསྒྱུར་བས། ཡོ་ཡུ་བ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཡོ་ཡ་བཱི་ཏི། ཡོ་ཡོ་ཏི། ཡོ་ཡུ་ཐ། ཡོ་ཡུ་ཐ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཡོ་ཡ་བི་མི། ཡོ་ཡོ་མི། ཡོ་ཡུ་བ། ཡོ་ཡུ་མ། །་ལྟ་ལིང་ལ་ཀྲྀཏ་མ་ཡིན་པ་—།
