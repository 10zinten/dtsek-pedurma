དེ་ན་་རིན་པའི༌སྐུ་བསྟན་འགྱུར། སྒྲ་མདོ། ཙཏིpལ་ནི་ཚེ་འདི་ལ་སྐྱོ་མཚོ་ོཎ་་ཉིན་ལྔ། ཀན་ཨ་ཏང་ལ། ལེཔཱ། ལེཔཱ་སི། །སོགས་དང༌། ཏང་ལ། ལེསྨཱ། ལེཔ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ལི་བྱཱཏེ། ལིཔཱ་ཏཱཾ། ལིཔཱ་སུ། སོགས་དང༌། ཏང་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ལེཔཱི་། ལེཔཱི་ཡཱ་ཙཱ། ལེངྒི་རན། ་སོགས་སོ། །མྲྀཊའི་ཨ་ཏང་ལ། ་ལེ་ྩ་ཏི། སོགས་དང་། ཏང་ལ། ལེ་ྩ་ཏེ། །སོགས་སོ། །ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་་ལེ་ྩཏ༑སོགས་དང་། ཏང་ལ། །ཨ་ལེ་ཏ། །སོགས་སོ། །ཏོ། །གང༌། ཁུ་ཚུད་༈ཏི་ཙ་འཛག་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། མུ་ཙ་ལ་སོགས་པ་རྣམས་ཀྱིས་ཤ་ལ་འོས། ནུམ་ཨཱ་ག་མ་བྱིན། བྱིངས་ཀྱི་དང་པོའི་ཥའི་ས་འོས། ཥ་སར་བསྒྱུར་བས། ལཊའི་ཨ་ཏང་ལ། སིཉྩ་ཏི། སོགས་དང་། ཏང་ལ། སིཉྩ་ཏེ། སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། སི་ཏ། །སོགས་དང༌། ཏང་ལ། །སིཉྩ་ཏ། སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། སིཉྩ་ཏུ། སོགས་དང་། ཏང་ལ། ?སིཉྩ་ཏྭཾ། སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ༑་ཨ་སིཉྩ་ཏ། །སོགས་དང་། ཏང་ལ། ཨ་སིཉྩ་ཏ། །སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ། ་ལི་པི་སི་ཙི་ཧཱ་ལས་ཀྱང་ངོས། ་ཨང་དུ་བསྒྱུར་བས། ཨ་སིཙ་ཏ། ：ཨ་སི་ཙ་ཏྭཾ། ཨ་སི་ཙན། །སོགས་སོ། །ཏང་ལ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ་ཡང་ན་འོས། ཙྪི་སིཙ་དང་ཨང་ཡང་ན་བར་བྱས་པས། ཨང་གི་ཕྱོགས་ལ། །ཨ་སི་ཙ་ཏ། ཨ་སི་ཙེ་ཏྭཾ། ཨ་སི་ཙནྟ། སོགས་དང་། སིཙའི་ཕྱོགས་ལ། ཛྷལ་དང་པོ་རྣམས་ལ་ཛྷལ་ལས་ཛྷལ་ལ་འོས། སིཙ་དབྱི་བ་དང་། ཙུའི་ཀུ་འོས་ཙ་ཀ། །ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སིཀྟ། ཨ་སིཀྵ་ཙཱ། ཨ་སིཀྵ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་སིཀྵ། ཨ་སིཀཱ་ཐཱ། ཨ་སིདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་སིཀྵི། ཨ་སིཀྵ་ཧི། ཝཱས།
