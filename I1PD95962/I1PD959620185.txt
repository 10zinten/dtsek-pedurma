ལ་། བ་འz་ཎཊིར་ཞུ་རཚོ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཏིའིཔ་ནས་ཆོས་དང་ཚོགས་ཚོaཁ་བྱིད་ཀ་ཡ་NoDesདྷདྡྷཾ། ཌདྷབྷཾ། ཡོ་བྷིས་ལ། དྷབྷཱི། དྷུནྟི། ངེ་ལ། དྲུ་ཧེ། བྷཱུམ་སྦྲུལ། དྷགྷཾ། ་དྷནྡྷཾ། །དྷསྒྱུ། དྷཟླུ། ང་སི་ལ། ་དྲུ་ཧ། བྷཾ་། འདྲ། །དྲུག་པ་རྣམས་ལ། དྲུ་ཧ། དྲུ་ཧོ། དྲུ་ཧཾ། །ང་ཨོས་ལ། དྲུ་ཧི། དྲུ་་ཧོ༑སུཔ་ལ། བྷུཀླུ། དྷཟླ། །སཾ་བུདྷི་རྣམ་དབྱེ་དང་པོ་དང་འདྲ། མུཧ། སྣུཧ༑སྣེཧ་རྣམས་ཀྱང་འདི་དང་འདྲ་འོ། །བི་ཤྭ་ཝཱ་ཧ་ལ་སུ། སྣ་ཚོགས་ཐོབ། བི་ཤྭ་ཝཱཊ། བི་ཤྭ་ཝཱཌ། ཨོ་དང་ཛས། བི་ཤྭ་ཝཱ་ཧོ། བིཤཱ་ཝཱ་ཧ། ་ཧེ་་་་བི་ཤྭ་ཝཱ་ཊ། ཧེ་བི་ཤྭ་ཝཱཌ། ་ཧེ་བི་ཤཱ་:བཱ་ཏོ། ཧེ་བི་ཤྭ་ཝཱ་ཧ། །ཨམ་ལ། བི་ཤྭ་ཝཱ་ཧཾ། ཨོཊ་ལ། བི་ཤྭ་ཝཱ་ཧོ། ཤས་ལ། ཝཱ་ཧ་ཨུཥ། ་ཝཱཧའི་་་་ཨུཋའོ༑::། ཝཱཧ་ཞེས་པའི་སྒྲ་འདིའི་ཝ་ནི་རྣམ་དབྱེ་གཉིས་པའི་ཤས་ནས་བཟུང་་་ཨཙ་ཐོག་མ་ཅན་རྣམས་ལ་ཨུཋ་ཞེས་པར་འགྱུར་རོ། །ཝ་ཨུ་ཨེ་ཏི་ཨེ་དྷ་ཏི་ཨུཋ་་རྣམས་ལའོ། །ཞེས་པས་འཕེལ་བ། །བི་ཤོ་ཧ། ཊཱ་ལ། བི་ཤོཧཱ། །ཧཱུཾ་་་བྷིས་ལ། བི་ཤཱ་བཱནྟུ། བི་ཤཱ་བཱནྟི། ངེ་ལ། །བིཤོཧེ། ཧཱུཾ་བྷུའདྲ། ང་སི་ལ། བི་གྷོ་ཧ། བྷཾ་རྒྱས་འདྲ། །ཌ་ངས་ལ། ::བི་ཤོཧ། ཨོས་ལ། བིགྷོ་ཧོ། ཨཱམ་ལ། བི་ཤྭཧཱུྃ། ཇི་ལ། བི་ཤོཧི། བི་ཤྭཧོ། སུབ་ལ། །བི་ཤྭ་བཱཤྩ། བྷུ་ཝཧ་ཡང་འདི་དང་འདྲའོ། །ཧ་ཡིག་མཐའ་ཅན་ཨ་ན་ཌཱཧ་་་། ལ༑"གླིང་ཁྱུབ། སུ། ཙ་ཏུ་ར་ན་ཊུ་ཧོ་རཱ་མུ་དཱཏྟ། ཙ་ཏུར་ཨ་ན་ཌཱཧ་དག་གི་ཨཱམ་ཨུ་དཱཏྟའོ༑། ཙཏུར་དང་ཨ་ན་ཌཱཧ་དག་ལ། །རྣམ་དབྱེ་ཐམས་ཅད་གནས་པའི་་མིང་ཅན་རྣམས་ཕ་རོལ་ལ་ཨཱམ་ཨཱ་ག་མ་སྟེར་ཏེ་དེ་ལ་ཨུ་དཱཏྟའི་མིང་ཅན་ཞེས་་་་མིང་འདོགས་བྱེད་དོ། །ཨཱམ་ཞེས་པའི་མ་ནི་མ་འགྲོ་བ་ཨཙའི་མཐའ་ལས་ཕ་རོལ་ཞེས་པའི་དོན་ནོ། །ཨན་ཊཧ། །སུ། སཱ་ཝ་ན་ཊུ་ཧ། ཨ་ན་ཌཱ་ཧའི་སུ་ལའོ། །ཨ་ན་ཌཱཧ་ཞེས་པའི་རྟགས་ལ་རྣམ་དབྱེ་དང་པོའི་གཅིག་ཚིག་སུ་དང་སཾ་བུདྡྷིའི་་་།
