བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཏིམ་རྨ་:ནི་སྔ་ལོ་མང་པོར་ཁ་ལ་རིང་བས་སྐྱོ་གར་་བིནྡ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བིནྡྷ། ཨ་བིན་ཐཾ། ཨ་བིདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་བིནྡི། ་ཨ་བིནྡྲ་ཧི། ཨ་བིནྡཱ་ཧི། ལུང་ལ་ཨིཊ་བྱིན་པས། ཨ་བེ་དི། །ཨ་བེ་དི་ཥཱ་ཏྭཾ། ཨ་བེ་དི་ཥ་ཏ། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བི་དེ། བི་བི་དཱ་ཏེ། བི་བི་དི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བི་དེ་ཏེ། བི་བི་དཱ་ཐེ། བི་བི་དི་དྷེ། བི་བི་དི་ཌྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བི་བི་དེ། བི་བི་དི་བ་ཧེ། བི་བི་དི་མ་ཧེ། ལུཊ་ལ་ཨིཊ་བྱིན། །པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། བེ་དི་ཏཱ། བེ་དི་ཏཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ་ཨིཊ་བྱིན་པས། བི་དི་ཊྛིཏྠ། །བི་དི་ཏི་ཡཱནྟཾ། བི་དི་ཏི་རན། སོགས་སོ། །།ཁྲིཊ་ལ་ཨིཊ་བྱིན་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། བི་དི་ཉྩ་ཏེ། སོགས་དང་། དེ་བཞིན་དུ། །ལྗིད་ལ། ཨ་བེ་དི་་ཏ། །སོགས་སོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ཏུ་དྷཱ་ད་ཡ། །ས་མཱཔ། །ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་རུ་དྷ་ལ་སོགས་པ་རྣམས་རྫོགས་སོ། །།ཤོག་དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་་་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས་རུ་དྷ་ལ་སོགས་པ་ཤཔ་ཀྱི་ཡུལ་ལ་ཐམ་འཇུག་པ་རྣམས་ཀྱི་ཆོ་ག་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པ་རགས་པ་རྣམས་རེ་ཞིག་རྫོགས་སོ། །།།།དགའ་བ་དང་། །སྐྱེ་བ་ཕྱོགས་དང་། ལ། ད་དུང་ཁ་གང་་བཞི། །ད་ནི་ཏ་ནུ་ལ་སོགས་པ་རྣམས་ཀྱི་ཆོ་ག་བཤད་པར་བྱ་སྟེ། ཏ་ནུ་རྒྱས་པ་་་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། བྱིངས་ཐམས་།
