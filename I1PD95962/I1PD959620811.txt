ཁོ་ས་སྒོ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི༽་་་་་་དོན་སྐྱོ་ག་ཎ་ཀ་དཀzབློ་ཨ་ཙེ་ཏི། ཨ་ཙེ་ཥཱ་ཧི། ཨ་ཙེ་ཧྥ་ཧི། བི་བྷཱ་ཥཱ་ཙེ། ཙིའི་འདམ་ཀའོ། །ཙིཉའི་ལེཊ་སན་ཕ་རོལ་ལ་གཉིས་བརྗོད་པའི་ཙ་ཀར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། །ཙིཉ་ལ་ལིཊ་སྦྱར། ཙ་ཀར་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་གི། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙི་ཀཱ་ཡ། ་ཙི་ཀྱ་ཏུ། ཙི་ཀྱུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙི་ཀེ་ཐ། ཙི་ཀ་ཡི་ཐ། ཙི་ཀྱ་ཐུ། ཙི་ཀྱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙི་ཀཱ་ཡ། ཙི་ཀ་ཡ། ཙི་ཀྱི་བ། ཙི་ཀྱི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙི་ཀྱེ། ཙི་ཀཱ་ཏེ། ཙི་ཀྱི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙི་ཀྱི་ཥ། །ཙི་ཀཱ་ཐེ། ཙི་ཀྱི་དེ། །ཙི་ཀྱི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་་གསུམ་ལ། ཙི་ཀྱེ། ཙི་ཀྱི་བ་ཧེ། ཙི་ཀྱི་མ་ཧེ། སོགས་དང༌། ཙ་ཀར་མ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། །ཙི་ཙཱ་ཡ། ཙི་ཙ་ཏུ། ཙི་ཙུ། སོགས་དང་། ཏང་ལ། ཙི་ཙེ། །ཙི་ཙཱ་ཏེ། ཙི་ཙི་རེ། *སོགས་སོ། ་། ལུཊ་ལ་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨ་ཏང་ལ། ཙེ་ཏཱ། ཙེ་ཏཱ་སི། སོགས་དང༌། ཏང་ལ། ཙེ་ཏཱ། ཙེ་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོར་བསྒྱུར་བས། ཙཱི་ཡཱ་ཏ། ཙཱི་ཡཱནྟཾ། །ཙཱི་ཡཱ་སུ། །སོགས་དང༌། ཏང་་ལ་ཡོན་ཏན་བྱས་པས། ཙེ་ཏི། ཙེ་ཏཱི་ཡཱནྟཾ། ཙེ་ཏི་རན། སོགས་སོ། །ཊ་ལའང་ཡོན་ཏན་བྱས་པས། ཨ་ཏང་ལ། ཥ་ཙེ་ཥྱ་ཏི། ་སོགས་དང༌། ཏང་ལ། ཙེ་ཥྛ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་ཙེ་ཏ། སོགས་དང་། ཏང་ལ། ཨ་ཙེ་ཥྱ་ཏི། སོགས་སོ། །དེ་ན་ཁྱཊ་ཏ་ཀ་༈སྭེཉ་ཀུན་ནས་འགེབས་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་་ཀ་འཇུག་སྟེ། ལཊའི་ཨ་ཏང་ལ། སྡེ་ཎོ་ཏི། སོགས་དང༌། ཏང་ལ། སྡེ་།
