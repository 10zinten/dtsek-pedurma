ནུསགཞུ། །སྡུག་བསྔལ། …ཨ་རུས་རྣམས་ལའོ། ་། དི་བཱ་ལ་སོགས་པ་འདི་རྣམས་་་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལས་ཊ་འགྱུར་ཏེ། དི་བཱ་ཀ་ར། ཉིན་བྱེད། ནས། སྡུག་བསྔལ་བྱེད། ཨ་རུསྐ་རའི་བར་རོ། །འགྲེལ་པར། ་ཀོ། ཡད；ཏདབ་ཧུ་རྣམས་ལས་ཕ་རོལ་གྱི་ཁྲི་ལས་ཡང་ན་ཨཙ་འགྱུར་རོ་ཞེས་བཤད་པས། ཀི་ཀ་རཿ༑ཡཅི་བྱེད། ཡཀྵ་ར། གང་བྱེད། ཏཏྟ་ར། དེ་བྱེད། བ་ཧུ་ཀ་ར། མང་པོ་བྱེད། ཀརྨ་ཎི་ཤྲཱི་ཏོ། ཀརྨ་ན་ལས་གཡོག་ལའོ། །ཀརྨན་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལས་གཡོག་པོ་བརྗོད་པ་ལ་ཊ་འགྱུར་ཏེ། ཀརྨ་ཀ་ར། ལས་བྱེད་དེ། བྲྀ་ཏི། གཡོག་གོ། །ན་ཤབ་བློ་ཀ་ཀེ་ལ་ཧི་གཱ་ཐཱ་བི་ར་ཙཱ་ཏུ་སཱུ་ཏྲ་མནྡྲ་པ་དེ་ཥཱ། ཤ：སྒྲ། བློ་ཀ། ཚིགས་བཅད། ཀ་ལ་ཧ；འཁྲུག་པ། གཱ་ཐཱ；ཡསྨྲ་བ། བི་ར；དགྲ། ཙཱ་རཿབརྫུན། སཱུ་ཏྲ；མདོ། མནྡྲ；སྔགས། ཚིག་པ་ད་རྣམས་ལ་མ་ཡིན་ནོ། །ཤརྡ་ལ་སོགས་པ་འདི་རྣམས་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལས་ཊ་འགྱུར་བ་མ་ཡིན་ཏེ། དེ་རྣམས་ལས་ཕ་་རོལ་གྱི་ཀྲྀ་ལ་ཨན་སྦྱར་བས། ཤརྡ་ཀཱ་རཿ། སྒྲ་བྱེད། ནས། ཚིག་བྱེད། པ་ད་ཀཱ་རཿཡི་བར་རོ། །སྟམྦ་ཤ་ཀྲྀ་ཏཱི་རིན། སྟམྦ་ཤ་ཀྲྀ་ཏི་དག་ལ་ཨིན་འོ། ༈། སྟམྦ་དང་། ཤ་ཀྲྀཏ་དག་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལ་རིམ་པ་བཞིན་འབྲུ་དང་བེའུ་བརྗོད་པ་ལ་ཨིན་འགྱུར་ཏེ། ན་དབྱངས་དོན་ཡོན་ཏན་བྱས་པས། །སྟམྦ་ཀ་རི། སྡོང་པོ་བྱེད་པ་སྟེ། བཱི་ཧི། འབྲུ། ཤ་ཀྲྀཥ་རི། ལན་ཅིག་བྱེད་པ་སྟེ། བཏྠ། བེའུའམ་མཉེས་གཤིན་པ། ཧ་ར་ཏེདྲྀ་ཏི་ནཱ་ཐ་ཡོཔ་ཤོ། ཧ་ར་ཏི་ལས་དྲི་ཏི། ནཱ་ཐ་དག་ལས་ཕྱུགས་ལའོ། །ཕྲོ་ཏི། ནཱ་ཐ་དག་གི་ཕ་རོལ་ལས་ཧ་ར་ཏི་སྟེ་ཧྲིཉ་ལས་ཕྱུགས་ཀྱི་བྱེད་པ་པོ་ལ་ཨིན་འགྱུར་ཏེ། པྲ་ཏི་ཧ་རི། པགས་ལྡན་གྱི་རི་དྭགས། པ་ཤུ། ཕྱུགས། ནཱ་ཐ་ཧ་རི༑འདམ་སེང༌། པ་ཤུ། ཕྱུགས། ཕ་ལེ་གྲ་ཧི་རཱཏྨ་བྷ་རི། ཕ་ལེ་གྲ་ཧི། ཨཱཏྨསྒྱུ་རིའང་ངོ༌། །ཕ་ལེ་སྔོན་མའི་གྲ་ཧི་དང༌། འབྲས་བུ་འཛིན། ཨཏྨ་སྔ་མའི་བྷིན་ལ་་་་་ལསབས་ཕ་།
