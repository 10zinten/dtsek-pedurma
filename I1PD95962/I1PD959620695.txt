སྒྲ་མདོ། ཏི༽ཏ་ཏ་ར་ལོ་ཏཁོ་ཏོག་མཚོ་ཆ་ཅད་པ་ལའོ། །ཀྲྀཉ་ཨུའི་མཐའ་ཅན་གྱི་ཨད་ནི་བྱིངས་ཐམས་ཅད་པ་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨུཏ་དུ་འགྱུར་ཏེ། །ལོཊའི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ། བི་དཾ་ཀུ་རུ་ཧཱུཾ། ཁ་བི་དཾ་ཀུཅནྟུ། དེ་བཞིན་དུ་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་དཾ་ཀུ་རུ། རྐྱེན་ཨུཏ་ལས་ཀྱང་བརྩེགས་པ་མ་ཡིན་པ་སོགས་ཀྱིས་ཧི་ཕྱིས། བི་དཾ་ཀུ་རུ་ཏཱཏ། བི་དཾ་ཀུ་རུ་ཏཾ། བི་དྷྭཾ་ཀུ་རུ་ཏེ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་དཱ་ཀ་ར་བཱ་ཎི། བི་དཱ་ཀ་ར་བཱ་བ། བི་དཱ་ཀ་ར་བཱ་མ། ཨམ་མ་སྦྱར་བའི་་་ཕྱོགས་ལ་བིཏ་རྣམས་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱེད་ཤཔ་འཇིག་པས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེཏཱུ། བིཏྟཱཏ། བིཏྟཾ། བི་དནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བིདྡྷི། གུ་ཛྷལ་ལ་ཧི་དྷི་འོས་ཧི་དྷིར་བསྒྱུར། བིཏྟཱཏ། །བིཏྟཾ། བིཏྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་དཱ་ནི། བེ་དཱ་བ། བེ་དཱ་མ། བིད་ལ་ལང་སྦྱར་བའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ། །ཨ་བེད། ་ཨ་བེཏ། ཨ་བིཏྟཾ། འགྲེལ་པར་ལུང་དྲངས་པ། བེ་ཏྟེརླ་ངོ་བྷེརྫུ་སྭཱ། བེ་ཏི་ལས་ལང་གི་ཛཱིའི་ཛུས་ཡང་ནའོ། །བིཏ་ལས་ཕ་རོལ་གྱི་ལང་བསྒྱུར་བའི་ཛྷི་ནི་ཛུས་སུ་ཡང་ན་འགྱུར་ཏེ། :ཨ་བི་དུ། ཨ་བི་དན། །སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། །ད། དའི་འང་ངོ༌། །བྱིངས་མཐའི་ད་སྲོག་མེད་ཕ་རོལ་ནས་སིཔ་ཕྱིས་པ་དེ་ནི་ཡང་ན་རུ་རུ་འགྱུར་ཏེ་ཨུ་བརྗོད་དོན་ནོ། །ཨ་བེ། ་ཨ་བེཏ། ཨ་བེད། སྐྱེས་བུ་དབུས་མའི་གཉིས་ཚིག་དང་མང་ཚིག་ལ། །ཨ་བིཏྟཾ། །ཨ་བིཏྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བེ་དཾ། །ཨ་བི་དྭ། ་ཨ་བིཏྠ། ་ལུང་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ཨ་བེ་དཱིཏ། ཨ་བེ་དིཏྟཾ། ཨ་བེ་དི་སོགས་སོ། །ལིཊ་ལ། ཨུ་ཥ་བིད་ཛཱ་ཤྲི་སོགས་ཀྱིས་ཡང་ན་ཨཱམ་སྦྱར་བས། བི་དཾ་བ་བྷུ་བ། བི་དཱ་ཙ་་ཀཱ་ར། བི་དཱ་མཱ་ས་སོགས་སམ། ཡང་ན་ལིཊ་ཉིད་སྦྱར། །བྱིངས་གཉིས་V།
