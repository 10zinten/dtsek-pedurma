ལང་ལ། ཨ་ད་ཧཏ་སོགས་སོ། །ལུང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨ་དྷཱ་ཀཱིཏ། ད་ཧ་ལ་ལུང་ཙྪསི་ཙར་བསྒྱུར་ཨི་དང་ཙ་ཕྱིས་ལ་ཏིཔ་དུ་བསྒྱུར་ཨིཊ་ཀྱིའང་ངོས་ཨི་ཕྱིས་ཨསྟི་སིཙ་ཞེས་པས་ཨཱིཊ་ད་ཧ་སོགས་ཀྱི་གྷ་འོས་ཧ་གྷ་ཛྷ་ཥའི་ཨཙ་གཅིག་པ་སོགས་ཀྱིས་ད་དྷ་བད་པྲཛ་སོགས་ཀྱིས་འཕེལ་བ་ཁར་ལ་ཙར་འོས་གྷ་་ཀ་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ་ཀ་ཥ་འདུས་པ་གླུ། ལུང་གི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་ལ། ཨ་དདྡྷཾ། ད་ཧ་ལ་ལུང་སིཙ་ལ་ཏས་ཏས་ཐས་སོགས་ཀྱིས་ཏས་ཏཱ་དཧ་སོགས་ཀྱི་གྷ་འོས་ཧ་གྷ་བད་བྲཛ་སོགས་ཀྱིས་འཕེལ་བ། ཛྷལ་ལས་ཛྷལ་ལ་འོས་སིཙ་ཕྱིས་ཛྷ་ཥ་ལས་ཏ་ཐ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷ་ཤལ་ཛྷལ་རྣམས་ཀྱི་ཛ་ཤ་འོས་གྷ་ག་དྲངས། སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། ཨ་དྷཀྵ། ད་ཧལ་ལུང་སིཙ་ལ་ཛཱི་ཛཱི་ཛུས་ཛ་ཕྱིས་ད་ཧ་སོགས་ཀྱི་གྷ་་འོས་ཧ་གྷ་ཛྷ་ཥའི་ཨཙ་གཅིག་པ་སོགས་ཀྱིས་ད་དྷ། བད་བྲཛ་སོགས་ཀྱིས་འཕེལ་བ། ཁར་ལ་ཙར་འོས་གྲུ་ཀྲ། ཀུན་ནས་་་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། ཀྲ་ཋ་འདུས་པ་ཁ་དྲངས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷཱཀྐི། ཨ་་་དཱགྷཾ༑ཨ་དཱགྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དཀཾ། ཨ་དྷཱཀྵ། ཨ་དྷཱ། ལིཊ་་་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དཱ་ཧ། དེ་ཧ་ཏུ། དེ་ཧུ། སྐྱེས་བུ་དབུས་མ་་་གསུམ་ལ། དེ་ཧི་ཐ། ད་དབྷ། དེ་ཧ་ཐུ། དེ་ཧ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་དཱ་ཧ། ད་ད་ཧ། དེ་ཧི་བ། དེ་ཧི་མ། ལུཊ་ལ། དབྦྷ། དབྦྷ་རོ། དབྦྷ་ར། སྨོན་པའི་ལིང་ལ། ད་ཧཱུཏྲ། མྲྀཊ་ལི། དྷཀྵ་ཏི། དྷ་ཀླུ་ཏཿ། དྷཀྵནྟི་་་སོགས་སོ། །ཕྲེང་ལ། ཨ་དྷགྱུཏ། ཨ་དྷཀྵ་ཏྭཾ། ཨ་དྷགྱུན་སོགས་སོ། །རོལ་༈ཙ་ཧ་ཡོངས་སུ་སྙོག་པ་ལའོ་ཞེས་པ་ལ། །ལཊ་སོགས་པ་རསྨི་པ་ད་་་་རྣམས་སྦྱར། བྱིངས་ཨིཊ་ཅན་ཡིན་པས་ཆོ་ག་གོ་བདེ་སྟེ། ལཊ་ལ། ཙ་ཧ་ཏི་སོགས་སོ། །ལིང་ལ། ཙ་ཧེ་ཏ་སོགས་སོ། །ལོཊ་ལ། ཙ་ཧ་ཏུ་སོགས་སོ། །ལང་ལ། ཨ་ཙ་ཧཏ་སོགས་སོ། །ལུང་ལ། ཨ་ཙ་ཧཱིཏ་སོགས་སོ། །ལིཊ་ལ། ཙ་ཙཱ་ཧ། ཙེ་ཧ་ཏུ། ཙེ་ཧུ། སོགས་སོ། །ལུཊ་ལི། ཙ་ཧི་ཏཱ་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཙ་རྒྱ་ཏ་སོགས་སོ། །ལྀཊ་ལ། ཙ་ཧི་ཥྛ་ཏི་སོགས་སོ། །ཁྲིང་ལ། ཨ་ཙ་ཧི་ཏ་སོགས་སོ། །བདག་ལ་དལ་སོགས་ཁ་བ་།
