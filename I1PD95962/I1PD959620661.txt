སྒྲ་མདོ། ཏི། ཁ་ཚེནི། ན་ལོ་པའི་ཁ་ཕྱོགས་པ་དབུས་མ་གསུམ་ལ། ཏི་ཏྲེ་ཎི་ཐ། ཐལ། ཏི་ཏཱི་ཥི་ཐ། །ཏི་ཏྲེ་ཀྵ། ཐལ། ཏི་ཏྲེ་ཥ་ཐུ༑ཏི་ཏཱི་ཥ་ཐུ། ཏི་ཏྲེ་ཥ། ཏི་ཏཱི་ཥ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཏི་ཏྲེ་ཥ། ཏི་ཏྲེ་ཁི་བ། ཏི་ཏཱི་ཥི་བ། ཏི་ཏཱི་བི་མ། ཏི་ཏཱི་ཏི་མ། ཏང་གི་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཏི་ཏྲེ་ཥ། ཏི་ཏཱི་ཥ། ཏི་ཏྲེ་ཥཱ་ཏེ། ཏི་ཏཱི་ཥཱ་ཏེ། ཏི་ཏྲེ་ཎི་རེ། ཏི་ཏྲེ་ཎི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཏི་ཏྲེ་ཥི་ཏེ། ཏི་ཏྲི་ཏི་ཥ། ཏི་ཏེ་ཥཱ་ཐེ། ཏི་ཏཱི་ཥཱ་ཐེ། །ཏི་ཏེ་ཏཱི་བྷེ། ཏི་ཏཱི་ཏཱི་དྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཏི་ཏྲེ་ཥ། ཏི་ཏཱི་ཥ། ཏི་ཏྲེ་ཏི་བ་ཧེ། ཏི་ཏཱི་ཥི་བ་ཧེ། ཏི་ཏཱི་ཏི་མ་ཧེ། ཏི་ཏཱི་ཏི་མ་ཧེ། ལུཊ་ལ། །པུཀ་མཐའ་སོགས་ཀྱི་ཡོན་ཏན་དང་། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས་ཏ་ཊར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ། ཏྲེཏྟཱ། ཏེཥྛ་སི་སོགས་དང་། ཏང་ལ། ཏཱིཥྛ། ཏེཥྛ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། །ཏཱི་ཝཱཏ། །ཏཱི་ཙཱ། ཏིཥྛསུསོགས་དང་། ཏང་ལ༑ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས། ་ཥ་ཀ་། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། ཏཱི་ཀྵི། ཏཱི་ཀཱི་ཡཱནྟཾ། ཏཱིཀཱི་རན་སོགས་སོ། །མྲྀཊ་ལའང་ཥ་ཀར་་བསྒྱུར་བ་སོགས་བྱས་པས། །ཨ་ཏང་ལ། ཏེ་ཀླུ་ཏི་སོགས་དང་། ཏང་ལ། ཏྲེ་གྲུ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། །ལྀང་གི་ཨ་ཏང་ལ། ཨ་ཏྲེ་གྱུཏ་སོགས་དང་། ཏང་ལ། ཨ་ཏྲེ་གྱུཏ་སོགས་སོ། །།།ས་དོང་བ་དེ་གཏོང་༈ཡ་ཛ་ལྷ་མཆོད་པ་དང་། འགྲོགས་པར་བྱེད་པ་དང་སྦྱིན་པ་རྣམས་་་ལའོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་པས། ཨ་ཏང་དང༌། ར་ཏང་གཉིས་ཀ་འཇུག་སྟེ། །ལཊའི་ཨ་ཏང་ལ། ཡ་ཛ་ཏི་སོགས་དང་། ཏང་ལ། །ཡ་ཛ་་ཏེ་སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། ཡ་ཛེ་ཏ་སོགས་དང་། ཏང་ལ། ་ཡ་ཛེ་ཏ་སོགས་སོ། །།ལོཊའི་ཨ་ཏང་ལ། ཡ་ཛ་ཏུ་སོགས་དང་། ཏང་ལ། ཡ་ཛ་—ur།
