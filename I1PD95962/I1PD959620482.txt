གྲངས་བཞི་ཀློག་པར་བྱེད། །ཨེ་ཀ་སྱ་ས་ཀྲྀ༔། །ཨེ་ཀའི་ས་ཀྲྀཏ་འང་ངོ༌། །ཨེ་ཀ་ལས་བྱ་བ་དང་འབྲེལ་པའི་ལན་གྲངས་བརྗོད་པ་ལ་སུཙ་འགྱུར་ཞིང༌། །སུཙ་ཕ་རོལ་ལ་ཨེ་ཀ་སུཙ་དང་བཅས་པས་ཀྲྀཏ་ཞེས་པར་འགྱུར་ཏེ། ས་ཀྲྀཏ། ལན་ཅིག། བ་ཧྭ་ལྕུནྡཙྪསྐུ་ར་ཀཱ་ད་ནྱ་ཏ་ར་ཧཱུཾ། མང་བ་དང་ཉུང་བའི་དོན་ལས་བྱེད་པ་ལས་ཤས་ཡང་ནའོ། །མང་བ་དང་ཉུང་བའི་དོན་བརྗོད་པ་ལས་བྱེད་པ་སྟེ་རྣམ་དབྱེ་བྱེད་པ་ཅན་རྣམས་་ཀྱི་དོན་ལ་ཤས་ཡང་ན་འགྱུར་ཏེ། ཛས་བ་ཧུ་ནི། མང་པོ་རྣམས། རྐྱེན་ཤས། བེ་ཧུ་ཤ༑མང་པོ་རྣམས། ་བྷིས་བ་ཧུ་བྷི། མང་པོ་རྣམས་ཀྱིས། ཤས་…བ་ཧུ་ཤ། མང་པོ་རྣམས་ཀྱིས། ཐུས་བ་ཧུ་བྷྱ། མང་པོ་རྣམས་ལས། ཤས་བ་ཧུ་ཤ། མང་པོ་རྣམས་ལ། སུཔ་བ་ཧུ་ཥཱ། མང་པོ་རྣམས་ལས། ཤསབ་ཧུ་ཤ། མང་པོ་རྣམས་ལ། དེ་བཞིན་དུ། བྷུ་རཱི་ཤ། མང་པོ་རྣམས་ལ། བྷུ་རཱིན་སོགས་དང་། །ཨལྤ་ཤ། ཐུང་ངུ་ལ། ཐུང་ངུ་ལ། …ཨཊྚཾ། སྟ་ཀ་ཤ། ཉུང་དུ་ལ༑...སྟཀཾ། ཉུང་དུ་ལ། སོགས་སོ། །བྱེད་པ་ཅན་ཞེས་པས་རྣམ་བཅད་ཀྱི་བྱེད་ཅན་མིན་ན་མི་འགྱུར་ཏེ། །བ་ཧཱ་ནཱ་སཱ་མཱི། མང་པོ་རྣམས་ཀྱི་རྗེ་བོ། སཾ་ཁྱི་ཀ་པ་ཙ་ནཱཙཱ་བིདྨ་ཡཾ༑གྲངས་གཅིག་ཚིག་ལས་ཐོབ་པར་འདོད་པ་ལའང་ངོ༌། །གྲངས་ཀྱི་སྒྲ་་་གཅིག་ཏུ་བརྗོད་པ་ལ་ཐོབ་པར་འདོད་པའི་དོན་ལའང་ཤས་འགྱུར་ཏེ། རྡོ་རྡོ་དེ་ཧི༑ཡགཉིས་གཉིས་དག་བྱིན་ཅིག། ཞེས་པའི་དོན་ལ། ཤས་དྲི་ཤདེ་ཧི། གཉིས་བྱིན་ཅིག། དེ་བཞིན་དུ། ཤས་ཏྲི་ཤདེ་ཧི། གསུམ་བྱིན་ཅིག། པྲ་ཏི་ཡོ་གེ་དཉྩམཱསྟ་སི། པྲ་ཏི་སྦྱོར་བ་ལ་ལྔ་པ་ལས་ཏ་སིའོ། །པྲ་ཏི་སྦྱོར་བ་ལས་རྣམ་དབྱེ་ལྔ་པའི་དོན་ལ་ཏ་སི་འགྱུར་ཏེ། ཁོ་བཱ་སུ་དེ་བཏྠ་ཏི། ནོར་ལྷའི་བུ་ལས་སོ་སོ། ཞེས་པའི་དོན་ལ། བཱ་སུ་དེ་བ་ཏ༑ཡཏ་སི་ཨི་བརྗོད་དོན། པྲ་ཏི། ནོར་ལྷའི་བུ་ལས་སོ་སོ། ཨཱ་དི་མ་དྷཏྟ་པྲྀཏྟཱ་པཱནཱ་དེ་ཉྩ། ཨཱ་དི་མ་རྒྱ་ལས་ཏ་ཏེ་པྲྀཏྟཱ་པཱནྟུ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོ༌། །ཨཱ་དི་དང་མ་རྒྱ་དག་ལས་ལྔ་པའི་དོན་ལ་ཏ་འགྱུར་ཞིང་དྲིཥྛ་པཱནཱ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་—།
