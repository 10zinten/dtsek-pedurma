སྒྲ་མདོ། ཏི་ཀུ་ཁ་མི་ཎི་ནི་སློབ་མ་ཨ་ཡུལ་བ་བློ་ཏཱིཏ། ལིཊ་བི་བྷརྫི་ཏཾ་ཙ་ཀཱ་ར། སོགས་སོ། །།།ཨིཊ་མེད་པའི་ཕྱོགས་ལ། བྷཉྫི་བྷརྫར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱེད། མ་བསྒྱུར་བའི་ཕྱོགས་ལ་བརྩེགས་པའི་དང་པོ་ས་ཀ་སོགས་ཀྱི་སྲ་ཕྱིས། ་་བྲཤྩ་བྷསྔ་སོགས་ཀྱིས་ཛ་ཥ། ས་ལ་ཥ་ཌྷ་སོགས་ཀྱིས་ཥ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། བི་བྷཀྵ། ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། །ལཊ་བི་བྷ་ཀྵ་ཏི། ལུང་ཨ་བི་བྷཱ་གཱི་ཏ། །ལིཊབི་བྷ་ཀྵ་ཙ་ཀཱ་ར། སོགས་དང་། བྷརྫར་བསྒྱུར་བའི་ཕྱོགས་ལ་གཞན་ཆོ་ག་འདྲ་བ་ལ། རུ་ཀའི་སྟེང་དུ་དྲངས་པས། ལཊ་བི་བྷཎཱ་ཏི། ལུང་ཨ་བི་བྷཎྜིཏ། ལིཊ་བི་བྷནྡྷོ་ཙ་ཀཱ་ར། སོགས་སོ། །།།།བ༈དམྦྷ་ངན་གཡོ་བ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར། ་ཨཱི་པ་མཐའ་རཾ་དྷུ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་བྱས། ཨིཊའི་ཕྱོགས་ལ།  ོལཊདི་དམྷི་ཥ་ཏི། ་ལུང་…ཨ་དི་དནྟི་ཏཱི་ཏ། ལིཊདི་དམྷི་ཊཱ་ཙ་ཀཱ་ར། །སོགས་དང༌། །ཨིཊ་མ་བྱིན་པའི་ཕྱོགས་ལ། དམྦྷ་ཨིཙྪ། དམྦུའི་ཨིཏ་འང་ངོ་ཚེ། །p/། ཨིཊ་མེད་པའི་སན་ཕ་རོལ་ལ་དམྦུའི་ཨཙ་ཨིཏ་དུ་བསྒྱུར་ཞིང་འང་གི་སྒྲས་ཙང་ཡང་ན་ཨཱིཏ་དུ་འགྱུར་ལ་་ཨ་བྷཱུ་ས་འང་དབྱི་སྟེ། ཧལ་མཐའ་ལས་ཀྱང་ངོས་སན་ཀིཏ་ཡིན་པས། ་ཀིཏ་ཕ་རོལ་ལ་ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་མ་ཕྱིས། མདོ་འདིས་ཨིཏ་དང་ཨཱིཏ་་་ཡང་ན་བར་བྱས། ཨ་ཧྥུ་ས་ཕྱིས། ཨཙ་གཅིག་པ་བྷ་ཥ་སོགས་ཀྱིས་ད་དྷ། ཁར་ལ་ཙར་འོས། ་བྷ་པར་བསྒྱུར་བས། །ལཊ་དྷིམྨ་ཏི། ལུང་ཨ་དྷིཔཱིཏ། ལིཊ་ཡདྷིཙཾ་ཙ་ཀཱ་ར། སོགས་སོམ། ལཊ་དྷིམྨ་ཏི། ལུང་ཨ་དྷཱིཔཱིཏ། ལིཊདྷཱིདྡྷཾ། ཙ་ཀཱ་ར། སོགས་སོ། །་༌། ་་༈ཤི་གུས་པ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར། ཨཱིཔ་མཐའ་སོགས་ཀྱིས་ཨིཊ་་་་ཡང་ན་བར་བྱས། ཨིཊའི་ཕྱོགས་ལ་ཡོན་ཏན་བྱས་པས། ལཊ་ཤི་ཤ་ཡི་ཥ་ཏི། —།
