ནས་ད་བར་ཏུ་སྐྱོ་བསྟན་འགྱུར། སྒྲ་མདོ། །''འི:ཁམ་ཏི་ཙྪ་ཚོ་ཎ་གཙོ་དག་ལ་མི་ལྟོས་དོན་སུ་སྲུ་བཱ་བ། ཨ་སུ་སྲུ་བཱ་མ། ལིཊ་ལ། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། སུ་སྭ་བ། སུ་སྲུ་བ་ཏུ། སུ་སུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་སུ་སྲོ་ཐ། སུ་སུ་བ་ཐུ༑་སུ་སྲུ་བ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སུ་སཱ་བ། ཎལ། སུ་སྲ་བ། སུ་སྲུ་བ༑་སུ་སྲུ་མ། །སྐབས་འདིར་ས་ཕྱི་མ་ཥར་མི་འགྱུར་ཏེ། །བྱིངས་འདི་དང་པོ་ནས་ས་ཡིག་ཡིན་པས་སོ། །ཚོ། ལུཊ་ལ། སྲོ་ཏཱ། སྲོ་ཏཱ་རོ། །སྲོ་ཏཱ་རཿསོགས་སོ། །།སྨོན་པའི་ལིང་ལ། ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་་་པས༑སཱུ་ཡཱཏ། སཱུ་ཡཱནྟཾ། །སཱུ་ཡཱ་སུསོགས་སོ། ་། །ལྀཊ་ལ། སྲོ་ཥཱ་ཏི། སྲོ་ཥྱ་ཏ། །སྲོ་ཉྩནྟི་སོགས་སོ། །ཁྲིང་ལ། ཨ་སྲོ་ཏ། ཨ་སྲོ་་ཏྭཾ། ཨ་་སྲོ་ཁྱེན་སོགས་སོ། །དགའ་བ་བྱེད་མ་དགཔ་གཙོ། ཞེཏུ་ལོ། །ཀ་ཁྱུ་༈དུ་འགྲོ་བ་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བས། །ལཊ་ལ། ད་བ་ཏི་སོགས་སོ། །ལིང་ལ། །ད་བེཏ་སོགས་སོ། །ལོཊ་ལ། ད་བ་ཏུ་སོགས་སོ། །ལང་ལ། །ཨ་ད་བཏ་སོགས་སོ། །ལུང་ལ། བྱིངས་རུ་དུ་སུ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་བྱས་པས། ཨ་དཱ་བཱིཏ། །ཨ་དཱ་བིཏྟཾ། ཨ་་དཱ་བི་ཥྱསོགས་སམ། ཡང་ན། ཨ་དོ་ཏཱིཏ། ཨ་དོཏྟཾ། ཨ་དོ་སོགས་སོ། །ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །དུ་དཱ་བ། དུ་དུ་བ་ཏུ། དུ་དུ་བུ། འགྲེལ་པར་ཨིཊ་དང་བཅས་པའི་ལིཊའི་འགྱུར་བ་རྣམས་ལ་ལར་ཀ་འགྲོ་བའི་མིང་ཅན་དུ་བྱ་ཞིང༌། ལ་ལར་པ་འགྲོ་བའི་མིང་ཅན་དུ་ཡང་བྱའོ་ཞེས་བཤད་པས། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །དུ་ད་བི་ཐ། དུ་དུ་བ་ཐ། ་དུ་དུ་བ་ཐུ། དུ་དུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་དཱ་བ། །དུ་ད་བ། དུ་དུ་བི་བ། དུ་། ད་བི་བ། དུ་དུ་བི་མ། དུ་ད་བི་མ། ལུཊ་ལ། བྱིངས་རུ་དུ་སུ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་བྱས་པས། ཨིཊའི་ཕྱོགས་ལ། །ད་བི་ཏཱ། ད་བི་ཏཱ་རོ། ལམ—།
