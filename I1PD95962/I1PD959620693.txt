བློབ་མ། ་བློ་གཏད་བུ་ཕྱི་མ་གསུམ་ལ། ཡ་ཡོ། ཡ་ཡི་བ། »ཡ་ཡི་མ། །ལུཊ་ལ། ཡཱ་ཏཱ། ་ཡཱ་ཏཱ་རོ། ཡཱ་ཏཱ་ར། ཡཱ་ཏཱ་སཱི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། འདིར་ཨེར་མི་འགྱུར་བས། ཡཱ་ཡཱཏ། ཡཱ་ཡཱནྟཾ། ཡཱ་ཡཱ་སུསོགས་སོ། །ལྀཊ་ལ། ཡཱ་སྱ་ཏི་སོགས་དང་། ཁྲིང་ལ། ཨ་ཡཱ་སྱཏ་སོགས་སོ། །བཱ་འགྲོ་བ་དང་དྲི་བ་དག་ལའོ་དང༌། དྭ་སྨད་པ་དྲྭ་ལ། སྨོན་པའི་ལིང་ལ། གཞན་ཡང་བརྩེགས་པ་སོགས་ཀྱིས་ཨེ་རུ་བསྒྱུར་བ་དང༌། མི་བསྒྱུར་བ་ཡང་ན་བར་བྱེད་པའོ། །ལའོ་གཉིས་ཀྱང་ཆོ་ག་ཐམས་ཅད་ཡཱ་དང་འདྲའོ། །།།ཁྱོ་གས་་་་:༈ཀྵ་གཏམ་ལའོ་ཞེས་པའང་བཤད་མ་ཐག་པ་རྣམས་དང་ཕལ་ཆེར་འདྲ་སྟེ། ལཊ་ལ། ཥཱ་ཏི། ཁྭ་ཏ། ཁཱནྟི་སོགས་སོ། །ལིང་ལ། ཁཱ་ཡཱཏ། བྱཱ་ཡཱ་ཙཱ༑ཥཱ་ཡུ༔སོགས་སོ། །ལོཊ་ལ། ཀྵ་ཏུ། ཥཱ་ཏཱཏ། བྱཱ་ཏཱ། བྱཱནྟུ་སོགས་སོ། །ལང་ལ། །ཨ་བྱཱཏ། ཨ་བྱཱ་ཏཱཾ། ཨ་ཁྱུ། །ཨ་ཁཱན་སོགས་སོ། །ཨ་སྱ་ཏི་བཀྟི་བྱཱ་ཏི་བྱོནང་། ཨ་སྱ་ཏི་བསྟི་བྱཱ་ཏི་རྣམས་ལས་ཨངའོ། །ཨ་སུ་འཕེན་པ། བ་ཙ་བརྗོད་པ་ཁྱུ་གཏམ་རྣམས་ལས་ལྕི་ནི་ཨང་དུ་བསྒྱུར་ཏེ། ཁྱུ་ལ། ལུང་། ལུང་ལ་ཙླི་འོས་ལྕི་བྱིན། །མདོ་འདིས་ལྕི་ཨང་། ཨིཊ་ལས་ཀྱ:ངཨཱཏ་ཀྱི་དབྱི་འོས་ཨཱ་ཕྱིས་པས། ཨ་ཁྱཏ། །ཨ་ཁྱ་ཏཱཾ། ་ཨ་ཁྱན་སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད། ཧལ་དང་པོ་ལྷག་པོ། ཀུ་ཧ་དག་གི་ཙུ་འོས། ཁ་ཚ། །ཨ་ངྷ་སའི་ཙར་འང་ངོས། ཚ་ཙར་བསྒྱུར་བ་སོགས་བྱས་པས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཁྱོ། ཙ་ཁྱ་ཏུ། ཙ་ཁྱུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙ་ཁྱི་ཐ། ཙ་ཁྱ་ཐུ། ཙ་ཏྱ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཁྱོ། ཙ་ཁྱི་བ། ཙ་ཁྱི་མ། ལུཊ་ལ། ཥཱ་ཏཱ། ཁྱཱ་ཏཱ་རོ། ཁྱཱ་ཏཱ་ར། བྱཱ་ཏཱ་སི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། གཞན་ཡང་བརྩེགས་པ་དང་པོ་སོགས་ཀྱིས་ཨེར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། ཁྱེ་ཡཱཏ། །ཁྱེ་ཡཱཙཱ། ཁྱེ་ཡཱ་སུསོགས་སམ། །
