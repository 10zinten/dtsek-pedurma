ཁ་ན་མོ་བསྟན་འགྱུར། སྒྲ་མདོ། ཙཏིཁ་དེཎྜི་ནི་ལོ་ཚཚོ་གཉོ་ཚཀུ་ཏོ་ལྟཁ་ཧུའང་བྷཱ་ཞེས་པར་འགྱུར་ཏེ། བ་ཧུ་ལ་ཨཱི་ཡ་སུན་མདོ་འདིས་ཨཱི་ཕྱིས་བ་ཧུ་སྒྱུར་འགྱུར། བྷཱུ་ཡཱན། ཆེས་མང་བ། བ་ཧུ་ལ་ཨི་མན་མདོ་འདིས་ཨི་ཕྱིས་བ་ཧུ་བྷུ་རུ་འགྱུར། བྷུམ། མང་བ་ཉིད། ཨིཎྜ་སྱ་ཡི། ཨིཥྛན་གྱི་ཡིཊ་འང་ངོ༌། །བ་ཧུ་ལ་ཨིཊྚན་སྦྱིན་པ་ན་ཨིཥྛན་གྱི་ཨི་དབྱི་ཞིང་བ་ཧུ་ལ་བྷུར་འགྱུར་ལ་ཨིཥྛན་ལ་ཡིཊ་ཨཱ་ག་མ་ཡང་འགྱུར་ཏེ། བྷཱུ་ཡིཊྛ། ཆེས་མང་བ། ཡུ་་་བཱལྤ་ཡོཀ་ནཱ་རྱ་ཏ་ར་ཧཱུཾ། ཡུ་བ་ཨལྤ་དག་གི་ཀ་ན་ཡང་ནའོ། །ཨིཥྛན་ཨཱི་ཡ་་་སུན་དག་ཕ་རོལ་ལ་ཡུ་བ་དང་ཨལྤ་དག་ནི་ཀ་ན་ཞེས་པར་ཡང་ན་འགྱུར་ཏེ། འ་ཡུ་བ་འམ་ཨལྤ་ལ་ཨིཊྚན་ཡུ་བའམ་ཨལྤ་ཀ་ནར་འགྱུར་ཊི་ཕྱིས། ཀ་ནིཥྛ། ཆེས་དར་བབ་བམ་ཆེས་ཉུང་བ། ཡུ་བའམ་ཨལྤ་ལ་ཨཱི་ཡ་སུན་རྟགས་ཀ་ནང་འགྱུར་ཊི་ཕྱིས། ཀ་ནཱི་ཡཱན། ཆེས་དར་བབ་བམ་ཆེས་ཆུང་བ། ཡང་ན། ཡུ་བ་ལ་ཨིཏྟན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་སྦྲུ་ལ་དྲུ་ར་སོགས་ཀྱི་མདོས་ཡཎའི་ཡི་གེ་བ་ཕྱིས་ཨུ་ཡོན་ཏན་ཨེ་ཙའི་ཨ་ཡ་སོགས། ...ཡ་་བིཥྛ༑ཆེས་དར་བབ། ཡ་བཱི་ཡཱན། ཡུ་བ་ལ་ཨིཏྟན་དང་ཨཱི་ཡ་སུན་ཊི་ཕྱིས་སྦྲུ་ལ་དྨ་ར་སོགས་ཀྱི་མདོས་ཡཎའི་ཡི་གེ་བ་ཕྱིས་ཨུ་ཡོན་ཏན་ཨེ་ཙའི་ཨ་ཡ་སོགས། ཆེས་ཉུང་བ། ཨཝིཊྛ། ཨིཊྚན། ཨཱི་ཡ་སུན། ཨམྦི་ཡཱན། ཆས་ཉུང་བ། བིརྨ་ཏོརླུཀ། བིན་མ་ཏུཔ་དག་གིས་འཇིག་གོ། །རྟགས་དེ་བིན་དང་མ་ཏུཔ་ཀྱི་མཐའ་ཅན་ཡིན་ལ་རྐྱེན་ཨིཊྛན་དང་ཨཱི་ཡ་སུན་ཕ་རོལ་ལ་བིན་དང་མ་ཏུཔ་འཇིག་སྟེ། བྱིན་གྱི་མཐའ་ཅན་སྲཁྲིན་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བྱིན་འཇིག། …སྲ་ཛི། ཆེས་ཕྲེང་བ་ཅན། ཆེས་ཕྲེང་བ་ཅན། སྲ་ཛཱི་ཡཱན། བིན་གྱི་མཐའ་ཅན་སྲསྤྲིན་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བིན་འཇིག། ཏཱ་ཛིཥྛ། མ་ཏུཔ་ཀྱི་མཐའ་ཅན་ཏཱཧྥ་ཏུལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་མ་ཏུཔ་འཇིག། ཆེས་པགས་པ་ཅན། ཆེས་པགས་ཅན། ཏཱ་ཛཱི་ཡཱན། མ་ཏུཔ་ཀྱི་མཐའ་ཅན་ཏཱགྲ་ཏུལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་མ་ཏུཔ་འཇིག། པྲ་ཤཾ་སཱ་ཡཾ་རཱུ་པ་ཡ། ཁ་བསྒྱུར་བ་ལ་རཱུ་པ་ཡའོ། །མི་མཁས་པ་ལ་མཁས་པར་བརྫུས་པ་ལྟ་བུའི་དོན་ལ་རྟགས་དང་ཏིང་མཐའ་ལས་ཀྱང་རཱུཔ་ཡ་འགྱུར་ཏེ། པྲ་ཤསྟཔ་ཊུ། རྫུས་མའི་མཁས་པ། ཞེས་པའི་དོན་ལ། མཁས་པའི་གཟུགས་སམ་རྫུས་མའི་མཁས་པ། པ་ཊུ་རཱུཔ། པ་ཊུ་ལ་རཱུ་པ་ཡ་ཡ་དབྱངས་དོན། དེ་བཞིན་དུ། པ་ཙ་ཏི་རཱུ་པཾ། འཚེད་།
