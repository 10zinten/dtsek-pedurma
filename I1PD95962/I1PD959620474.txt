བར་ཡོན་ཏན། སྦི་ར་ལ་ཨིཉྩན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་སྦིར་སྦྲར་འགྱུར་ཡོན་ཏན། མཆོག་ཏུ་བརྟན་པ། སྦེ་་་ཊྛ༑མཆོག་ཏུ་བརྟན་པ། སྦེ་ཡཱན། སྦི་ར་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་སྦིར་སྦྱར་འགྱུར་ཡོན་ཏན། མཆོག་ཏུ་འཕེལ་བ། སྤེཊྛ། སྤྱི་ར་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་སྤྱི་ར་སྦྱར་འགྱུར་ཡོན་ཏན། སྤེ་ཡཱན། མཆོག་ཏུ་འཕེལ་བ། ཨུ་རུ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་ཨུ་རུ་བ་རར་འགྱུར་ཊི་ཕྱིས། བ་རིཥྛ། …བདེ་ལེགས་པའམ་མཆོག། བ་རཱི་ཡཱན༑ཡཨུ་རུ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་ཨུ་རུ་བ་རར་འགྱུར་ཊི་ཕྱིས། བ་ཧུ་ལ་ཨིཏྟན་དང་ཨཱི་ཡ་སུན་མདོ་་་འདིས་བ་ཧུ་ལ་བི་ཧིར་འགྱུར་ཊི་ཕྱིས། ཐཾ་ཧིཥྛ། མཆོག་ཏུ་མང་བ། མཆོག་ཏུ་མང་བ། ཐཾ་ཧཱི་ཡཱན། བ་ཧུ་་་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བ་ཧུ་ལ་བོ་ཧི་ར་འགྱུར་ཊི་ཕྱིས། གུ་རུ་ལ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་གུ་རུ་ག་་་རར་འགྱུར་ཊི་ཕྱིས། ག་རིཥྛ། མཆོག་ཏུ་ལྕི་བ། མཆོག་ཏུ་ལྕི་བ། ག་རཱི་ཡཱན། གུ་རུ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་་་སུན་མདོ་འདིས་གུ་རུ་ག་རར་འགྱུར་ཊི་ཕྱིས། བྲྀདྡྷ་ཡ་ཨིཥྛན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བོདྡྷ་བནྟི་རུ་འགྱུར་ཊི་་་ཕྱིས༑པནྟིཥྛ། ཆེས་རྒས་པ། བནྟི་ཡཱན། བྲྀདྷ་ཡ་ཨིཏྠ་ན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བིནྡྷ་བིནྟ་རུ་འགྱུར་ཊི་་་ཕྱིས། ཡཀྲྀཔ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བྲྀཏྣ་ཏྲ་པ་རུ་འགྱུར་ཊི་ཕྱིས། ཏྲ་པིཥྛ། ཆེས་ཚིམ་པ། ཆེས་ཚིམ་་་པ། ཏྲ་པཱི་ཡཱན། དྭིཔཱ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་ཀྲྀཔ་ཏཱ་པ་རུ་འགྱུར་ཊི་ཕྱིས། དཱ་གྷིཥྛ། དཱིནྡྷ་ལ་་་ཨིཏྟན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་རྒྱུ་དཱ་གྷི་རུ་འགྱུར་ཊི་ཕྱིས། དྲྭ་གི་ཡཱན། ཆེས་རིང་བ། དཱིགྷ་ལ་ཨིཏྟན་དང་ཨཱི་ཡ་་སུན་མདོ་འདིས་དཱིགྷ་དཱ་གྷི་རུ་འགྱུར་ཊི་ཕྱིས། བྲིནྡྷ་ར་ཀ་ལ་ཨིཊྚན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བྲིནྡྷ་ར་ཀ་བྲྀནྡར་འགྱུར་ཊི་ཕྱིས། བྲིནྡིཥྛ༑ཡཆེས་ཚོགས་པ། བྲནི་ཡཱན། བྲྀནྡྷ་ར་ཀ་ལ་ཨིཉྩན་དང་ཨཱི་ཡ་སུན་མདོ་འདིས་བྲིན་ར་ཀ་བྲྀནྡར་འགྱུར་་ཊི་ཕྱིས། འདི་རྣམས་ཀྱི་ནང་ནས་སྲི་ཡ་ཨུ་རུ་བ་ཧུ་ལ་གུ་རུ་དཱིགྷ་རྣམས་བྲི་ཐུ་ལ་་་སོགས་པའི་ནང་ཚན་ཡིན་པས་ཨི་མན་འགྱུར་ཏེ། པྲི་ཡ་ལ་ཨི་མན་སྲི་ཡ་ཕྲ་རུ་འགྱུར་ཡོན་ཏན། པྲ་མ། དགའ་བ་ཉིད། ཨུ་རུ་ལ་ཨི་མན་ཨུ་རུ་བ་རར་འགྱུར་ཊི་ཕྱིས། བ་རི་མ། བྲང་ཉིད། མང་པོ་ཉིད། ཐཾ་ཧི་མ། བ་ཧུ་ལ། ཨི་མན་བ་ཧུ་ལ་བི་ཧིར་འགྱུར་ཊི་ཕྱིས། ལྕི་པ་ཉིད། ག་རི་མ། གུ་རུ་ལ་ཨི་མན་གུ་རུ་ག་རའི་འགྱུར་ཊི་ཕྱིས། །དཱིགྷ་ལ་ཨི་མན་དཱིགྷ་དཱ་བྷིར་འགྱུར་ཊི་ཕྱིས། དཱ་གྷི་མ། ཆེས་རིང་བ། བ་་་ཧོརླ་པོ་བྷུ་ཙ་བ་ཧོ། བ་ཧུ་ལས་དབྱི་སྟེ་བ་ཧུའི་བྷུའོ། །བ་ཧུའི་ཕ་རོལ་ཏུ་ཨཱི་ཡ་་་སུན་ཨིཥྛན་ཨི་མན་རྣམས་བྱུང་ན་རྐྱེན་དེ་རྣམས་ཀྱི་ཐོག་མའི་ཡི་གེ་དབྱི་ཞིང་་་་་—།
