ལ་ལྟ་བའི་ཟས་ཀ་པ་ཚོ་འབཏ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼བཏི༽། ་ནི་སྨྲ་བ་མཚོ། ཕྱི་པ་ཀ་བ་བློ། ཤཱིརྙའི་ཤྲིནྟན་ཡང་ནའོ། །།ཤིནྟ་ཞེས་པའི་རྟགས་ནི་རྣམ་དབྱེ་གསུམ་པའི་ཊཱ་ནས་བརྩམས་ཏེ་བདུན་པའི་སུབ་ཀྱི་བར་གང་མཐོང་ཡང་ཤཱིརྙན་ཞེས་པར་ཡང་ན་་་་འགྱུར་བས། ཊཱ་ལ། ༼ཤིནྟེ་ཎ། ཤྲཱིརྟན་དུ་བསྒྱུར་ན། ནའི་ཨཏ་དབྱི་འོས། ཨ་ཕྱིས། ཥ་ར་སོགས་ཀྱིས་ན་ཎ། ཤྲཱིཙྪཱབྷྱཱ་ཅེལ། ཤྲཱིནཱ་ཧཱུཾ། །ལུགས་་་་གཅིག་ལ། མིང་མཐའི་ན་དབྱི་འོས། ན་ཕྱིས། ཤིནྟ་བྷཾ། བྷིས་ལ། ཤིནྟི༑ཤིནྟ་བྷི། །ངེ་ལ། ཤྲཱིནཱ་ཡ། བྷིནྡྷེ། བྷཱུམ་འདྲ། བྱས་ལ། བྷིནྡྷེ་བྷྱ། ཤྲིནྟ་བྷྱ། ་་ང་སི་ལ། ཤྲཱིནཱཏ། ཤྲཱིཙྪཱ། བྷཱུམ་བྱས་འདྲ། ངས་ལ། ཤཱནྟ་སྱ། ཤཱིཙྪ། ཨོས་ལ། ཤཱནྟ་ཡོ། ཤཱིཙྪཱ། ཨཱམ་ལ། ཤཱིནཱ་ཎཱ། ཤཱིཙྪཱང་ལ། ་ཤཱནྟེ། ཤཱནྟི། ཨོས་འདྲ། སུཔ་ལ། བྷིནྡྷེ་བུ། ཤཱིན་སིུ། ཧྲ་སྔོན་པོ་ས་ཀེ་པྲ་ཏི་པ་དི་ཀ་སྱ། །མ་ནིང་ལས་མིང་གི་ཐུང་ངུའོ། །རྟགས་མཐའ་ཨཙ་རིང་པོ་རྣམས་ནི་མ་ནིང་གི་རྟགས་ཀྱི་སྐབས་སུ་ཐུང་ངུར་འགྱུར་ཏེ༑དཔེར་ན། ཤྲཱི་པཱ་དཔལ་སྐྱོང་ཞེས་པ་ཨཱ་རིང་མཐའ་ཅན་ཡིན་ཀྱང་བཤད་མ་ཐག་པའི་མདོས་ཐུང་ངུར་འགྱུར། །ཤྲཱི་པཾ། ཤྲཱི་པེ། ཤྲཱི་པཱ་ནི་ལྟ་བུ་སྟེ་ཨཱ་རིང་ཐམས་ཅད་ཨ་ཐུང་དུ་འགྱུར་ཏེ། རྣམ་དབྱེ་དང་ཕྲད་པའི་བྱ་བ་ཐམས་ཅད་ཨ་་་ཐུང་མཐའ་ཅན་སྤྱིར་བཏང་ཛྙཱ་ནའི་སྐབས་སུ་བཤད་པ་དང་འདྲའོ། །།དེ་རྣམས་ཀྱིས་མ་ནིང་གི་རྟགས་ལས་ཨཱ་རིང་ཨ་ཐུང་གི་བྱ་བ་རྣམས་བཤད་ཟིན་ཏོ། །།།ཎ་༈ད་ནི་མ་ནིང་གི་རྟགས་ཨི་རིང་ཐུང་གི་མཐའ་ཅན་རྣམས་བཤད་པར་བྱ་སྟེ༑བཱ་རི། ཆུ། ལ། སུ། སྭ་མོརྣ་་ས་ཀཱདི། །མ་ནིང་ལས་སུ་ཨམ་་་་དག་གིའོ། །མ་ནིང་གི་རྟགས་ལས་རྣམ་དབྱེ་དང་པོའི་སུ་དང་གཉིས་པའི་ཨམ་ནི་འཇིག་གོ། །ཞེས་པས་སུ་འཇིག། བཱ་རི། བཱ་རི་ལ། ཨོ། མ་ནིང་པ་འང་ངོས་ཨོ་ཤཱི། ཨི་ཀོ་ཅི་བི་བྷ་ཀྟ། ་རྣམ་དབྱེ་ཨཙ་ལ་ཨིཀ་གིའོ། །མ་ནིང་གི་་་་—།
