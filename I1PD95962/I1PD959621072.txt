བརྩི་ཥྛ་ཏེ། ལྷས་ཆར་འབེབས་པར་འགྱུར། གླི་པྲ་བ་ཙ་ནེ་ཁྲིཊ། མགྱོགས་པར་བརྗོད་པ་ལ་ཁྲིཊའོ། །མགྱོགས་པར་བརྗོད་པ་རྟོགས་པར་བྱེད་པའི་འབྱུང་འགྱུར་ལ་ལྲྀཊ་་ཉིད་འགྱུར་ཏེ། ཁྲིཊཨཱ་ཡཱ་སྱ་ཏི་ཤྲཱི་ཧཱུྃ། ཐོབ་པར་འགྱུར་འཕྲལ་ལ། ཨཱ་ཤཾ་སཱ་བ་ཙ་ནི་་ལིང་། ཐེ་ཚོམ་བརྗོད་པ་ལ་ལིངའོ། །ཐེ་ཚོམ་བརྗོད་པའི་འབྱུང་འགྱུར་རྟོགས་པར་བྱེད་པ་ལ་ལིང་ཉིད་འགྱུར་ཏེ། །གུ་རཱུ་པེ་ཡཱཏི། ཨུ་པ་སྔོན་མའི་ཨིཎ་འགྲོ་བ་ལ་ལིང་། བླ་མའི་ཉེ་བར་འགྱུར། འགྲེལ་པར་ཨུཏ་དང་ཨ་པི་དག་སྦྱོར་བ་ལ་ལིང་གི་ཡུལ་ལ་ལྡིང་འགྱུར་རོ་ཞེས་བྱུང་བས། ལྡིང་གི་ཨིཊ། ཨུ་ཏ་བྷོ་ྋཀྱེ། ཤིན་ཏུ་ཟ། ཨ་པ་བྷོ་ྋཀྱེཟས་ཀྱང༌། གཞྭ་ཡཾ་ལཉཌ་པི་ཛཱ་ཏཱི། ཨ་པི་ཛཱ་ཏུ་དག་ལས་སྨོད་པ་ལ་ལཊའོ། །།ཨ་པི་དང་ཛཱ་ཏུ་དག་དང་སྦྱོར་བ་ལ་སྨད་པ་རྟོགས་པ་ལ་དུས་གསུམ་ཀ་ལ་ལཊ་འགྱུར་ཏེ། ལཊཨ་པི་ཛཱ་ཡཱ་ཏུ་ཛ་སི། མནའ་མ་ཡང་འབོར་བར་བྱེད། ལཊཛཱ་ཏུ་གཱ་ཎི་ཀཱ་མཱ་དཏྟེ། ནམ་ཞིག་སྨད་འཚོང་མ་ལེན་པར་བྱེད། བི་བྷཱ་ཥཱ་ཀ་ཐ་མི་ལི༔་། ཀ་ཐམ་ལ་ལིང་འང་ཡང་ནའོ༑། ཀ་ཐཾ་གྱི་སྒྲ་སྦྱོར་བ་ལས་སྨད་པ་རྟོགས་པར་བྱེད་པ་ལ་ལཊ་འགྱུར་ཞིང་་ལིང་འང་ཡང་ན་འགྱུར་ཏེ། ་ལིང་ཀ་ཐཾ་དྷརྨ་ཏྱ་ཛེ། ཇི་ལྟར་ཆོས་བོར་བ། ལཊ་ཀ་ཐོ་དྷརྨ་ཏུ་ཛ་སི། ཇི་ལྟར་ཆོས་བོར་བ། འགྲེལ་པར་ཀ་ཐོ་གྱི་སྒྲ་ཡོད་པ་ལ་སྨད་པ་རྟོགས་པར་བྱེད་པ་ལ་འབྱུང་འགྱུར་ལ་རྀང་རྟག་ཏུ་འགྱུར་ཏེ། ལཊ་ཀ་ཐཾ་ནཱ་མ་ཏ་ཏྲ་བྷ་བཱན་དྷརྨ་ཨ་ཏུ་གྱུཏ། ཇི་ལྟར་མིང་དེ་ལ་ཁྱོད་ཚོས་བོར་བར་འགྱུར། ཀིམ་བྲིཏྟེ་ལིང་ཀྲྀ་ཊ། ཀིམ་འཇུག་པ་ལ་ལིང་ཁྲིཊ་དག་གོ། ་། ཀིམ་འཇུག་པ་སྟེ་ཀི་ཉིད་དང་ཀི་རྐྱེན་མཐའ་་་་ཅན་དང་སྦྱོར་བ་ལ་སྨད་པ་རྟོགས་པར་བྱེད་པ་ལ་ལིང་དང་ཁྲིཊ་དག་འགྱུར་ཏེ། ཀ་ཏ་རོ་བྷ་བཱན་ནི་དེཏ། ཅི་ཁྱོད་སྨད་པར་བྱེད། ལིང་ཀ་ཏ་རོ་བྷེ་བཱན་ནི་དི་ཥྛ་ཏི། ཅི་ཁྱོད་སྨད་པར་བྱེད། ཀིམ་ཀི་ལཱསྐྱརྗེ་ཥཱ་ཁྲིཊ། ཀིམ་ཀི་ལ་ཡོད་པའི་དོན་རྣམས་ལ་ཁྲིཊའོ། །ཀིམ་ཀི་ལའི་སྒྲ་དང་ཡོད་པའི་དོན་རྣམས་ལ་དུས་གསུམ་ཀ་ལ་ཁྲིཊ་འགྱུར་ཏེ། །
