སྒྲ་མདོ། ༼ཏིཁ་ཏི་ནི་ལྟ་ཁོ་པའི་བོད་པར་ཏེ། བློ་ལ་རྀཏ་མཐའ་ཅན་ཡིན་ན་ཨཱ་ནང་དུ་འགྱུར་ཏེ། ཧོ་ཏཱ་ཙ་པོ་ཏཱ་ཙ། …སྦྱིན་སྲེག་མཁན་དང་ཚ་བོ་དང་། ཞེས་པ་ལ་རྣམ་དབྱེ་སོགས་ཕྱིས། ་ཧོ་ཧྲི་ཞེས་པའི་མཐའ་ཨཱ་ནང་དུ་འགྱུར། ན་དང་ང་རྗེས་འབྲེལ། ཧོ་ཏཱ་པོ་ཏ་རོ། སྦྱིན་སྲེག་མཁན་ཚ་བོ་དག། དེ་བཞིན་དུ། མཱ་ཏཱ་ཙ་པི་ཏཱ་ཙ། མ་དང་ཕ་དང༌། ཞེས་པ་ལ། མཱ་ཏཱ་པི་ཏ་རོ། ཕ་མ་དག། དེ་བཞིན་དུ། པི་ཏཱ་ཙ་པུ་ཏྲ༄༅། ཕ་དང་བུ་དང༌། ཞེས་པ་ལ། ་པི་ཏཱ་པུ་ཊོ། .ཕ་བུ་དག། དེ་བ་ཏཱ་ཛྭ་ཏྭེ་ཙ༑ལྷ་ལའང་ཟླས་དབྱེ་ལའོ། །ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འདིར་ལྷའི་སྒྲ་ཕ་རོལ་ལའང་རྟགས་སྔོན་མ་དེའི་མཐའ་ཨཱ་ནང་དུ་འགྱུར་ཏེ། མི་ཏྲ༔་བ་རུ་་་ཎ། གྲོགས་དང་ཆུ་ལྷ་དང༌། ཞེས་པ་ལ། །མི་ཏྲ་བ་རུ་ཎོ། གྲོགས་ཆུ་ལྷ་དག། ཨཱི་དགེསོ་མ་བ་རུ་ཎཡོ། ཨགིའི་སོ་མ་བ་རུ་ཎ་དག་ལ་ཨཱིཏའོ། །ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འདིར་སོ་མ་དང་བ་རུ་ཎ་དག་ཕ་རོལ་ལ་ཨགྷའི་མཐའ་ཨཱིཏ་དུ་འགྱུར་ཏེ། ཨཀྵེསྡུཏྟོ་མ་སོ་མཱ། ཨགི་ལས་སྟུཏ་སྟེ་མ་སོ་མ་རྣམས་སོ། །།ཟླས་དབྱེའི་བསྡུ་་བའི་སྐབས་འདིར་ཨགྣིའི་ཕ་རོལ་ཏུ་སྟུཏ་སྟོ་མ་སོ་མ་རྣམས་བྱུང་ན། ལ་རྟགས་དེ་་་དག་གི་ས་ཡིག་རྣམས་ཥ་ཡིག་ཏུ་འགྱུར་ཏེ། ཨགྣིy༄༅། མ་དང་བསྟོད་པ་དང༌། ཞེས་པ་ལ་རྣམ་དབྱེ་སོགས་ཕྱིས། མདོ་འདིས་ས་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས། །ཏ་ཊ། ཨབྷིཥྛ་ཏ། མེ་བསྟོད་པ། དེ་བཞིན་དུ། ཨབྷི་སྟོ་མཤྩ། མེ་དང་མཆོད་སྦྱིན། །ཞེས་པ་ལ། ཨགི་ཌོ་མོ། མེ་མཆོད་སྦྱིན། དེ་བཞིན་དུ། ཨགྣིཉྩ་སོ་མཉྩ། མེ་དང་ཟླ་བ། ཞེས་པ་ལ་་་རྣམ་དབྱེ་སོགས་ཕྱིས། མདོ་སྔོན་མའི་ཨགི་མཐའ་ཨཱིཏ། ཨགྣི་ཁོ་མོ། མེ་ཟླ་བ་དག། འགྲེལ་པར་ཡང་ན་ཥ་དང་ཨཱི་གཉིས་ཀ་རུ་མི་འགྱུར་རོ་ཞེས་བཤད་པས། ཨགྣི་སོ་མོ། དེ་བཞིན་དུ། ཨགིཤྩ་བ་རུ་ཎ། མེ་དང་ཆུ་ལྷ་དང་། ཞེས་པ་ལ། ཨགྣི་བ་རུ་ཎོ། མེ་ཆུ་ལྷ་དག། ཨིཛྫིདྡྷོ། འཕེལ་བ་ལ་ཨིཏའོ། །གོང་དུ་བ་རུ་ཎ་ཕ་རོལ་ལ་་ཨངྒེའི་མཐའ་ཨཱིཏ་དུ་འགྱུར་བར་བཤད་ཀྱང་། ་་བ་རུ་ཎ་འཕེལ་བ་བྱས་པ་དག་།
