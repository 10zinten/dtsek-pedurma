སྒྲ་མདོ། ་ཏི*། །འོ་ན་ཉེས་པ་ཚུ་ཁ་:ས་སྐྱོ་ནདྨ་ཧི། ལིཊ་ལ་གཉིས་བརྗོད་ཀིཏ་རྣམས་ལ་ཨ་བྷཱ་ས་དབྱི་བ་དང་ཨེར་བསྒྱུར་བ་སོགས་ཀྱི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ན་ནཱ་ཧ། ནེ་ཧ་ཏུ། ནེ་ཧུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནེ་ཧི་ཐ། ན་ནདྷ། ཐལ། ནེ་ཧ་ཐ་ཏུ། ནེ་ཧ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ན་ནཱ་ཧ། །ན་ན་ཧ། ་ནེ་ཧི་བ། ནེ་ཧི་མ། ::དེ་་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ནེ་ཧེ། ནེ་ཧཱ་ཏེ། ནེ་ཧི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནེ་ཧི་ཏི། ནེ་ཧཱ་ཐེ། ནེ་ཧི་དེ། ནེ་ཧིནྡྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནེ་ཧེ། ནེ་ཧི་བ་ཧེ། །ནེ་ཧི་མ་ཧེ། ལུཊ་ལ། ན་ལས་ཧའི་དྷ་འོས། ཧ་དྷ། ཛྷཥ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་དྷ་དར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ། ནདྷཱ། ནདྡྷཱ་སི། ༽སོགས་དང༌། ཏང་ལ། ནདྷཱ། ནདྡྷཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ༑ན་ཧཱཏ། ན་ཧཱནྟཾ། ན་ཧུ་སུ། སོགས་དང་། ཏང་ལ། ་ན་ལས་ཧའི་དྷ་འོས། ་ཧ་དྷ། ཁར་ལ་ཙར་འོས་དྷ་ཏིར་བསྒྱུར་བས། ཊི་ནཧཱིཋ། ནཏི་ཡཱཙཱ། ནཏི་རན། །སོགས་སོ། །ཝིཊ་ལའང་ན་ལས་སོགས་ཀྱིས་ཧ་དྷ། ཁར་ལ་ཙར་དུ་བསྒྱུར་བས། ཨ་ཏང་ལ། ནཉྩ་ཏི། སོགས་དང༌། ཏང་ལ། དཉྩ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། འོ་ཕྲོང་གི་ཨ་ཏང་ལ། ཨ་ནཉྩཏ། སོགས་དང༌། ཏང་ལ། ཨ་ནཉྩ་ཏ། སོགས་སོ། །།ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ཏ་དི་བཱ་ད་ཡས་མཱམྦྷ། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་དི་བུ་ལ་སོགས་པ་རྣམས་རྫོགས་སོ། །།།།ཌ་༈དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་་་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས། དི་བུ་ལ་སོགས་པ་བྱེད་པ་པོ་།
