བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བཉྩ། ཨུཥྛ'ཏ། །ཨུཏྟཾ། ཨུ་་་ཤནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨུཎྜི། བ་ཤ་ལ་ཧི། ་གྲ་ཧི་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎ་བྲ་སོགས་ཀྱིས་ཤ་ཥ། ཧུ་ཛྷལ་སོགས་ཀྱིས་ཧི་དྷི། ཥ་ཊུ་ལ་ཁ་ཊུ་འོས་དྷ་བྷ། མཐའི་ཛྷལ་སོགས་ཀྱིས་ཁ་ཌ། ཨུཥྛཏ། ཨུཏྠཾ། འཨུཊྚ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བ་ཤཱ་ནི། ཨཱཊ། བ་ཤཱ་བ། ཨཱཊ། བ་ཤཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། བ་ཤ་ལ་ལང་། ལ་ཏི། ཨིཏ་ཀྱི་འང་ངོས་ཨི་ཕྱིས། ཚོ་ཧ་ལ་དཱི་བ་སོགས་ཀྱིས་ཏ་ཕྱིས། བྲཤྩ་སོགས་ཀྱིས་ཤ་ཥ། :མཐའི་ཛྷལ་རྣམས་ཀྱི་ཛ་ཤ་འོས། ཥཌ། གནས་མེད་པ་ལ་ཡང་ན་་་་འོས། །ཨ་བཊ། ཨ་བཊ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨོཏྟཾ། གྲ་ཧི་སོགས་ཀྱིས་སཾ་པྲ་ས་ར་ཎ་ཨཙ་དང་པོ་རྣམས་ཀྱི་ཨཱཊ་འོས། ་བྱིངས་ལ་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ། ཨོ་ཤན། དེ་བཞིན་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་བཌ། ་ཨ་བཊ། ཨོཏྟཾ། ཨོཊྚ༈། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བ་ཤཾ། ཨོ་ཤཱ། ཨོ་། ལུང་ལ། ཧལ་དང་པོ་ཡང་བ་སོགས་ཀྱིས་འཕེལ་བ་ཡང་ན་བ་དང༌། ཨིཊ་སྦྱར་བས། ཨ་བཱ་ཤཱིཏ། ཨ་བཱ་ཤིཏྟཾ། ཨ་བཱ་ཤི་ཥཱ། །སོགས་སམ། །ཨ་བཱ་ཤིཏ་སོགས་སོ། །ལིཊ་ལ་ཀིཏ་མ་ཡིན་པ་རྣམས་ལ། གཉིས་རྣམས་སོགས་ཀྱིས་ཨ་བྷཱ་ས། སཾ་པྲ་་སཱ་ར་ཎ་དང་ཀིཏ་རྣམས་ལ་གྲ་ཧ་སོགས་ཀྱིས་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར། ཨཀ་འདྲ་བ་ལ་རིང་པོ་སོགས་ཀྱིས་བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨུ་བཱ་ཤ། །ཨཱི་ཤ་ཏི། ཨུ་ཤུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨུ་བ་ཤི་ཐ། ཨུ་ཤ་ཐུ། ཨུ་ཤ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨུ་བཱ་ཤ། །ཨུ་བ་ཤ་། ཨུ་ཤི་བ། །ཨུ་ཤི་མ། ལུཊ་ལ། ་བ་ཤི་ཏཱ། བ་ཤི་ཏཱ་རོ། བ་ཤི་ཏཱ་ར། བ་ཤི་ཏཱ་སི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། གྲ་ཧ་ཙཱ་སོགས་ཀྱིས་སཾ་པྲ་སཱ་ར་ཎ་བྱས་་་པས། ང་། ཨུ་ཤཱཏ། །ཨུ་བྱཱ་ཙཱ། ་ཨུ་བྱ་སུ། ཨུ་ཐཱ། ཨུ་ཤཱནྟཾ་སོགས་སོ། །ས་།
