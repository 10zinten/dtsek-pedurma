མ་དྲུག་ཅུ་ལ་འབྲུ་སྨིན་པའི་ཞིང་། བི་བྷཱ་ཥཱ་ཏི་ལ་མཱ་ཊོ་མཱ་བྷངྒ་ཏུ་བྷ། །ཏི་ལ་མཱ་ཥ་ཨུ་མཱ་་་་བྷངྒ་ཨ་ཆུ་རྣམས་ལ་ཡང་ནའོ། །ཏི་ལ་ལ་སོགས་པ་འདི་རྣམས་ལས་དེ་ཡོད་པའི་ཞིང་བརྗོད་པ་ལ་ཡཏ་ཡང་ན་འགྱུར་ཏེ། །ཡཏ་ཏི་རུཾ། ཏིལ་གྱི་ཞིང་། ཁ་ཏི་ལི་་་ནཾ༑ཏིལ་གྱི་ཞིང༌། ཡཏམཱ་ཏྱཾ། སྲན་མའི་ཞིང་། ཁཉམཱ་ཏཱི་ནཾ། སྲན་མའི་ཞིང་། ཡཏ་ཨུ་་་མྱོ༑ཟར་མའི་ཞིང་། ཁཉ་ཨོ་མཱི་ནཾ། ཟར་མའི་ཞིང་། ཡཏ་…བྷཧཱུྃ། སོ་མ་ར་ཙའི་ཞིང༌། ཁཉ…བྷཱངྒཱི་་་ནཾ༑སོ་མ་ར་ཙའི་ཞིང་། ཡཏ…ཨ་ཎ་བྱ། ཕྲ་མོའི་ཞིང་། ཁཉ། ཨཱ་ཎ་བཱི་ནཾ། ཕྲ་མོའི་ཞིང་། ཏཏྟརཱ་དེཔ་བྷཾ་ག་ཀརྨ་པ་ཏྲ་པཱ་ཊཾ་བྷཔོ་ཏི། སརྦ་དང་པོའི་པ་ཐི་ཨོ་ག་ཀརྨ་པ་ཏྲ་པཱ་ཏྲ་་ལ་དེ་རྣམ་པར་ཁྱབ་པ་ལའོ། །སརྦ་དང་པོའི་པ་ཐི་ལ་སོགས་པ་འདི་རྣམས་ལ་དེ་་་རྣམ་པར་ཁྱབ་པའི་དོན་ལ་ཁ་འགྱུར་ཏེ། །ཤས་…སརྦ་པ་ཐཱན་བྱསྣ་ཏི། ལམ་ཐམས་ཅད་ཁྱབ་པ། ཞེས་པའི་དོན་ལ། ཁ་ཁ་ཨཱིན། སརྦ་པ་ཤྲཱི་ན། ལམ་ཐམས་ཅད་ཁྱབ་པ། དེ་བཞིན་དུ༑། ཁ་སརཱངྒཱི་ཎ། ལུས་ཐམས་ཅད་ཁྱབ་པ། ཁསརྦ་ཀརྨི་ཎ། ལས་ཐམས་ཅད་ཁྱབ་པ། ཁ་སརྦ་པ་ཏཱི་ཎ། འདབ་མ་ཐམས་ཅད་ཁྱབ་པ། ་ཁ་…སརྦ་པཱ་ཏཱི་ཎ། སྣོད་ཐམས་ཅད་ཁྱབ་པ། ཨ་དྷཱ་ནོ་ཡཏྠོ། ཨ་དྷཱན་ལས་ཡཏ་ཁ་དག་གོ། ་། ཨ་དྷཱན་ལས་མྱུར་བ་བགྲོད་པའི་དོན་ལ་ཡཏ་དང་ཁ་དག་འགྱུར་ཏེ། ཨ་དྷཱ་ན་མ་ལངྒཙྪ་ཏི། ལམ་མྱུར་བར་བགྲོད། ཞེས་པའི་དོན་ལ། ཡཏཨ་དྷཱ་ནྱ། ལམ་མྱུར་བར་བགྲོད། ཁ་ཨ་དྷཱ་ནཱི་ན། ལམ་མྱུར་བར་་་བགྲོད། ཏ་སྱ་པཱ་ཀ་མཱུ་ལེ་པཱི་ལཱ་དི་ཀཎཱ་དི་བྱཱཀུ་ཎ་ཌ་ཧ་ཙོ། པཱི་ལུ་ལ་སོགས་པ་་་རྣམས་ལས་སྨིན་པ་དང་ཀཎ་ལ་སོགས་པ་རྣམས་ལས་དེའི་རྩ་བ་ལ་ཀུ་ཎ་པ་ཛཱ་ཧཙ་དག་གོ། །པཱི་ལུ་ལ་སོགས་པ་རྣམས་ཀྱི་སྨིན་པ་བརྗོད་པ་ལ་ཀུ་ཎཔ་དང་ཀཎ་ལ་སོགས་པ་རྣམས་ཀྱི་དེའི་རྩ་བ་བརྗོད་པ་ལ་ཛཱ་ཧཙ་འགྱུར་ཏེ། པཱི་ལཱ་ནཾ་པཱ་་་ཀ༑སེ་ཡབ་སྨིན། ཞེས་པའི་དོན་ལ། པཱི་ལུ་ལ་ཀུ་ཎབ་པ་དབྱངས་དོན། པཱི་ལུ་ཀུ་ཎ། སེ་ཡབ་སྨིན་པ། དེ་བཞིན་དུ། ཁྱད་པར་གནས་སྨིན་པ། ཨ་ཤཱཏྠ་ཀུ་ཎ། ཀུ་ཎཔ། ་ཀུ་ཎཔ། ཀརྐནྡྷ་ཀུ་—།
