ལ་ཁོ་ཏ་ཀ། ཀ་ཎ་ཁྲིད་ཏུ་ང་ཚོ་mབསྟན་འགྱུར། སྒྲ་མདོ། ཏི༽་པོའི་ནི་ཁོ་ཚོ་དག་པར་བློ་ཐགཡི་གེ་ཕ་རོལ་ལ་ཨུ་རུ་འགྱུར་རོ། །སུ་དྷུ་བྷཾ། ་སུ་དྷུ་བྷི། ངེ་ལ། སུ་དི་ཝེ། ༔་བྱས་ལ། སུ་བྱུ་ཧཾ། སུ་ཅུ་། །ང་སི་ལ། སུ་དི་ཝ། ཧཱུཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། ་སུ་དི་ཝ། སུ་དི་ཝོ། །སུ་དི་ཝཾ། ང་ལ། སུ་དི་ཝི། ཨོས་ལ། སུ་དི་ཝོ། སུཔ་ལ། སུ་ཅུ་ཥཱ། བོད་ཚིག་ལ། །ཤེ་སུ་དྲོ། ཧེ་སུ་དི་ལོ། ཧེ་སུ་དི་ཝ། ་ར་མཐའ་ཅན་ལ་ཙ་ཏུར་བཞི་ཞེས་པར་མང་ཚིག་་་བདུན་པོ་སྦྱོར་ཏེ། ཙ་ཏུར། ཛས། ཙ་ཏུར་ཨ་ན་རུ་ཧ་ད་ག་གི་ཨཱམ་ཨུ་་་་དཱཏྟའོ་ཞེས་པའི་མདོས་ཨཱམ་ཨཱ་ག་མ། ཙ་ཏཱརཿ། ཤས་ལ། ཙ་ཏུ་རཿ། བྷིས་ལ༑ཙ་ཏུབྷི། བྱུས་གཉིས་ལ། །ཙ་ཏུཉྩ། ཨཱམ་ལ། ཥཊ་ཙ་ཏུཉྩ། ཥཊ་ཙ་ཏུར་རྣམས་ལས་ཀྱང་ངོ༌། །ཥཊ་དང་ཙ་ཏུར་ཞེས་པ་ལ། རྣམ་དབྱེ་་་དྲུག་པའི་ཨཱམ་ལ་ནུཊ་སྟེར་རོ། །ར་ཥ་སོགས་ཀྱིས་ན་ཎ། ཨཙ་ལས་ར་ཧ་ལ་་་ཞེས་པས་ཎ་གཉིས། ཙ་ཏུཙྪཱ། ?སུཔ་ལ། རོསླུ་པི། རའི་སུཔ་ལའོ། །ཙ་ཏུར་གྱི་ར་ནི་རྣམ་དབྱེ་བདུན་པའི་མང་ཚིག་སུཔ་ཕ་རོལ་ལ་རྣམ་པར་བཅད་པར་མི་འགྱུར་བར་ར་ཉིད་འདྲེན་པའོ། །ཙ་ཏུནྟུ། སྤྱིར་བཏང་ར་ཧ་ལ། ཞེས་་་པས་ཥ་གཉིས་སུ་བྱ་བ་ཐོབ་ཀྱང༌། ཤ་རོ་ཙི། ཨཙ་ལ་ཤར་གྱིའོ། ལ། ཤར་སོ་་་་སོར་བསྡུ་བའི་ཡི་གེ་ཨཙ་དང་ལྡན་པ་དག་ར་ཧ་ལ་གཉིས་སུ་མི་འགྱུར་རོ་ཞེས་པས་དམིགས་བསལ་གྱི་མདོས་གཉིས་སུ་མི་འགྱུར། ཧེ་ཙ་ཏཱ་ར། པྲ་ཡ་ཙ་་་ཏུར་དགའ་བ་བཞི་ཞེས་པ་ལྟ་བུ་རྣམ་དབྱེ་ཐམས་ཅད་འཇུག་སྟེ། སུ་ལ། བྲི་ཡ་ཙ་ཏཱ། ཨོ་དང་ཛས་ལ། པྲ་ཡ་ཙ་ཏཱ་རོ། པྲ་ཡ་ཙ་ཏཱ་ར། ཨམ་ལ། ྋདྲི་ཡ་ཙ་ཏཱ་རཾ། ཨོ་ལ། པྲ་ཡ་ཙ་ཏཱ་རོ། །ཤས་ལ། པྲ་ཡ་ཙ་ཏུ་རཱི། གསུམ་པ་རྣམས་ལ། པྲ་ཡ་ཙ་ཏུ་རཱ། པྲ་ཡ་ཙ་ཏུདྡྷཾ། པྲ་ཡ་ཙ་ཏུབྷི། །བཞི་པ་རྣམས་ལ། པྲ་ཡ་ཙ་ཏུ་རེ། པྲ་ཡ་ཙ་ཏུདྡྷཾ། པྲ་ཡ་ཙ་ཏུཙྪ། ང་སི་ལ། པྲ་ཡ་ཙ་ཏུ་རཿ། —།
