མྱི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །མི་མྱེ། །མི་མྱི་བ་ཧེ། མི་སྤྱི་མ་ཧེ། ལུཊ་ལ། མཱི་ནཱ་ཏི་མི་ན་ཏི་དཱིང་སོགས་ཀྱིས་ཨི་ཨཱཏ་དུ་བསྒྱུར་བས། ཨ་ཏང་ལ། མཱ་ཏཱ། མཱ་ཏཱ་སི། སོགས་དང་། ཏང་ལ། མཱ་ཏཱ། མཱ་ཏཱ་སེ། སོགས་སོད། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། མཱི་ནཱ་ཏི་སོགས་ཀྱིས་ཨི་ཨཱཏ། ལིང་ལ་ཨེ་འོས། ཨཱ་ཨེར་བསྒྱུར་བས། མེ་ཡཱཏ། མེ་ཡཱནྟཾ། མེ་ཡཱ་སུ། །སོགས་སོ། །ཏང་ལ། ཨི་ཨཱཏ་དུ་བསྒྱུར་བས། མཱ་སཱིཊ། མཱ་སཱི་ཡཱནྟཾ། མཱ་སཱི་རན། སོགས་སོ། །ཀྲྀཊ་ལའང་ཨི་ཨཱཏ་དུ་བསྒྱུར་བས་ཨ་ཏང་ལ། མཱ་ྋསྐྱ་ཏི། སོགས་དང༌། ཏང་ལ། མཱ་སྱ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ལྭང་གི་ཨ་ཏང་ལ། ཨ་མཱ་སྱཱཏ། སོགས་དང༌། ཏང་ལ། ཨ་མཱ་སྱ་ཏ། སོགས་སོ། །ན༈ཙིཉ་གསོག་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་སྦྱོར་ཏེ། ལཊའི་ཨ་ཏང་ལ། ་ཙི་ནོ་ཏི། སོགས་དང༌། །ཏང་ལ། ཙི་ནུ་ཏེ། སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། ་ཙི་ནུ་ཡཱཏ། སོགས་དང་། །ཏང་ལ། ཙི་ནཱི་ཏ། །སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། །ཙི་ནོ་ཏུ། སོགས་དང༌། ཏང་ལ། ཙི་ནུ་ཏྭཾ། སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ་ཨ་ཙི་ནོཏ། སོགས་དང་། ཏང་ལ་ཨ་ཙི་ནུ་ཏ། སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ༄༅། སིཙ་ལ་འཕེལ་བ་སོགས་ཀྱིས་འཕེལ་བ་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཙི་ཏྟིཏ། ཨ་ཙིཏྟཾ། ཨ་ཙི་ཥྛ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཙི་ཏི། ཨ་ཙིཏྟཾ། ཨ་ཙིཏྠ༑སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཙི་ཎཾ། ཨ་ཙི་ཏེ། ཨ་ཙི་། ཏང་ལ། བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། སྐྱེས་་བུ་དང་པོ་གསུམ་ལ། ཨ་ཙེ། ཨ་ཙེ་ཥཱ'ཏྭཾ། ཨ་ཙེ་ཥ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཙེ་། ཨ་ཙེ་ཥཱ་ཐཾ། ཨ་ཙེ་དཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །
