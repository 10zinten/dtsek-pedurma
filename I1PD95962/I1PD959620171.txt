ལ་བའི་ལྟ་བ་དང་། སྐུ་མ་རྫོགས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིའིཡི་གེ་འདི་དག་ཁོ་ཚོའི་ཀ་མང་བོ་བློ་གཞན་ཆོ་ག་ཐམས་ཅད་བ་དྷུ་དང་འདྲ། །དྲུག་པའི་ཨཱམ་ལ། ཨེ་ཀཱ་ཛུཏྟ་ར་བ་དེ་ཎ། རྗེས་མའི་ཚིག་ཨཙ་གཅིག་པ་ལ་ཎའོ། །།རྣམ་དབྱེ་ཚིག་སྡུད་ཀྱིས་་་་བསྡུས་པའི་མིང་སྔ་མར་ཥ་འམ་ར་ལྟ་བུ་ན་ཎར་འགྱུར་བའི་རྒྱུ་མཚན་ཅན་ཡོད་ན་དེའི་རྗེས་མའི་ཚིག་ཨཙ་གཅིག་པ་སྟེ་དབྱངས་གཅིག་ལས་མེད་པ་དེ་བར་ན་ཡོད་་་ཀྱང་ན་ཎར་འགྱུར་རོ། །པུ་ནིབྷཱུ་ཏཱཾ་ལྟ་བུའོ། །བནཱ་བྷུ་ལ། ལོ་དུས་སུ་འབྱུང་མ། སུ། བནཱ་བྷུ། །ཨོ་དང་། ཛས། བནཱ་བྷཱཉྩ། བནྡྷ་བྷུའི་འང་ངོ༌། །བནྡྷ་བྷུ་་་་ཞེས་པའི་སྒྲ་ནི་རྣམ་དབྱེ་དབྱངས་ཐོག་མ་ཅན་གང་མཐོང་ཡང་ཡཎར་འགྱུར་རོ། །བནྡྷ་བྷོ། བརཱ་བྷཱ། བོད་ཚིག་ལ། ཧེ་བནཱ་བྷཱ། ཧེ་བནཱ་བྷཱ། ཧེ་བནྟཱ་བྷཱ༑རྣམ་དབྱེ་གཞན་རྣམས་ལའང་ཨཙ་ཐོག་མ་ཅན་རྣམས་ལ་ཨུ་ཝར་འགྱུར། ང་འགྲོ་བའི་མིང་ཅན་རྣམས་ལ་ཨཱཊ་སྟེར། ཧལ་ཐོག་མ་ཅན་རྣམས་ལ་བྱ་བ་ལྷག་པོ་མེད་དོ། །སྭ་ཡཾ་བྷུ་ཞེས་པ་ལ། ང་འགྲོ་བའི་མིང་ཅན་རྣམས་ལ་ཨཱཊ་སྟེར། གཞན་སྐྱེས་པའི་རྟགས་ཀྱི་སྐབས་སུ་བཤད་པ་དང་ཁྱད་པར་མེད་དོ། ་། དེ་རྣམས་ཀྱིས་བུད་མེད་ཀྱི་རྟགས་སྤུ་རིང་རྣམས་ཀྱི་ཆོ་ག་བཤད་ཟིན་ཏོ། །ག། ད་ཞུ་བ་༈ད་ནི་དམིགས་བསལ་རིཏ་མཐའ་ཅན་ནི། །་པ་ན་ཥཊ་སྭ་སྭ་དི་བྷུ། སཱ་སྲ་སོགས་ཥཊ་རྣམས་ལས་མ་ཡིན་ནོ། །གྲྭ་སྲི་སོགས་ཥཊའི་མིང་ཅན་ཞེས་བརྡ་ཆད་བྱས་པ་དེ་རྣམས་ལ་བུད་མེད་དུ་བསྒྲུབས་པའི་རྐྱེན་དཱིབ་དང་ཊཱཔ་སོགས་མི་འཇུག། རྀཏ་ན་རྣམས་ལས་དཱིཔའོ་ཞེས་སྤྱིར་བཏང་ཐོབ་ཀྱང་མདོ་འདིས་་་་བཀག་པའོ། །ཥཊའི་མིང་ཅན་ནི་འདི་རྣམས་ཏེ། སཱ་སི། སྒྱུག་མོ། ཏི་་་སྲི༑གསུམ། ཙ་ཏ་སྲི། བཞི། ན་ནཾ་ཏྲ། …ཚ་མོ། དུ་ཧི་ཏྲ། བུ་མོ་འམ། ཡ་ཚིགས། གཞོན་ནུ་མ། མཱ་ཀྲྀ། མ། བདུན་པོ་འདི་རྣམས་ལ་ཥཊའི་མིང་ཅན་ཞེས་བྱའོ། །སཱ་སྲི་ལ། སྐྱུག་མོ། སུ། རྀཏ་ཨུ་ཤ་ནས་སོགས་ཀྱི་མདོས། རྀ་ཨ་ནང་། སཾ་བུདྡྷི་མ་ཡིན་་།
