དེsགྱི་དཱ་ནི་དད་ཅེས་པར་འགྱུར་ཞིང༌། ཡང་ན་ཛཾ་ཞེས་པར་འགྱུར་ཏེ། ཀིཾ་དཏྟ༑ཡཅི་བྱིན། ཀིཾ་ཐཱནྟ། ཨཙ་ཨུ་པ་སརྦ་ཏ། ཉེར་བསྒྱུར་ཨཙ་ལས་ཏའོ། །ཉེར་བསྒྱུར་ཨ་ཙ་མཐའ་ཅན་ཕ་རོལ་གྱི་དཱཉ་ནི་ཏ་དང་པོ་ཀ་འགྲོ་བ་ཕ་རོལ་ལ་ཏ་ཞེས་པར་འགྱུར་ཏེ། །:ཏའི་སྲོག་ཨ་བརྗོད་དོན་ནོ། །པྲ་ཨ་པ་སྔོན་མའི་དཱཉ་ལ་ཀྟ་མདོ་འདིས་ཏར་བསྒྱུར་བས། པྲཏྟ། རབ་བྱིན། ཨ་བཏྟཀུན་བྱིན། །སོགས་སོ། །འགྲེལ་པར་ཡང་ན་དད་དུ་བསྒྱུར་བར་བཤད་པས། ཨ་བ་དཏྟཾ། ཀུན་བྱིན། དྲ་དཏྟཾ༑རབ་བྱིན། སུ་ད་ཏྟཾ། ལེགས་བྱིན། སོགས་སོ། །དནྟི། དཱའི་ཏ་ལའོ། །ཉེར་བསྒྱུར་ཨིཀ་སོ་སོར་བསྡུ་བའི་མཐའ་ཅན་ནི་དཱ་བསྒྱུར་བའི་ཏ་ཕ་རོལ་ལ་རིང་པོར་འགྱུར་ཏེ། ནི་དང་སུ་སྔ་མའི་དྭ་ལ་ལྟ་བྱིན། ཉེར་བསྒྱུར་ཨཙ་ལས་ཏ་འོས། དཱ་ཏ། མདོ་འདིས་རིང་པོར་བསྒྱུར་བས། ནཱིཏྟཾ། དེས་བྱིན། །སཱུཏྟཾ། ལེགས་བྱིན། སོགས་སོ། །དྷེཊ་བཏུང་བ། གི་སྒྲ། བཱ་བཏུང་བ་རྣམས་ལ། ཀྟ:གླུ་མཱམྦྷ་སོགས་ཀྱིས་ཨིར་བསྒྱུར་བས། དྷཱི་ཏཾ། བཏུང་། །གཱི་ཏཾ། ཀླུ། པཱི་ཏཾ། བཏུང༌། སོགས་སོ༑། ཛན། སན། ཁན་རྣམས་ལ་ལྟ། ཛན། སན། ཁན་རྣམས་ཀྱི་ཛྷལ་ལ་འོས་ཊི་ཨཱཏ་དུ་བསྒྱུར་བས། ཛཱ་ཏཾ། སྐྱེས། སཱ་ཏཾ། བྱིན། བཱ་ཏཾ། རྐོས། ཨ་དོ་ཛབྷི་ལྕསྟི་ཀི་ཏི། ལྕཔ་ཏ་ཀ་འགྲོ་བ་ལ་ཨཏ་ཀྱི་ཛབྷིའོ། །།ཀྵ་བསྒྱུར་བའི་ཁྱཔ་དང་ཏ་དང་ཀ་འགྲོ་བ་ཕ་རོལ་ལ་ཨད་ཟ་བ་ནི་ཛཤྲཱི་ཞེས་པར་འགྱུར་ཏེ། ཨད་ཟ་བ་ལ་ཀྟ། མདོ་འདིས་ཨད་ཛབྷི་རུ་བསྒྱུར། ཨི་བརྗོད་དོན། ཛྷ་ཥ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷར་འདྲ་བ་ལ་ཛྷར་རྣམས་ཀྱི་འོས་དྷ་གོང་མ་ཕྱིས་པས། ཛམྦྷ། ཟོས། བཱ་ཀྲོ་ཤ་དི་ནྱ་ཡོ། ཁྲོས་ཚིག་དང་འོ་བརྒྱལ་བ་དག་ལ་ཡང་ནའོ། །ཀཱི་ཟད་པའི་ནིཥྛ་ཕ་རོལ་ལ་ཁྲོས་ཚིག་དང་འོ་བརྒྱལ་བ་དག་ལ་འཇུག་པ་ན་རིང་པོར་ཡང་ན་བསྒྱུར་ཏེ། ཀཱི་ཎཾ་མ་ན་ན། ཟད་པར་སེམས། ཀྱི་ཎཾ་མ་ནེ་ན། ཀཱི་ཎཱ་།
