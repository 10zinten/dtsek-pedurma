ཧེ་དེ་བ་དཏྟཉ་ཨ་ཏྲ། ཀྱེ་ལྷས་བྱིན་འདིར། ཧེ་ཡཛྙཱ་དཏྟཉཨི་དཾ། ཀྱེ་མཆོད་སྦྱིན་འདིར། ཨཱི་གླུ་དེ་ཏ་དཱི་བ་ཙ་ནཾ་པྲ་ཀྲྀ་ཏཾ། ཨཱིཏ་ཨུཏ་ཨེཏ་གཉིས་ཀྱི་ཚིག་རང་བཞིན་་་་ནོ༑། ཨཱིཏ་དང་ཨུཏ་དང་ཨེཏ་གཉིས་ཀྱི་ཚིག་ཅན་རྣམས་ཨཙ་ཕ་རོལ་ལ་མཚམས་སྦྱོར་བ་མ་ཡིན་གྱི་རང་བཞིན་དུ་གནས་ཏེ། ཨ་ཨ་ཏྲ། མེ་དག་འདིར། བཱ་ཡུ་ཨི་ཏི། རླུང་དག་ཅེས་པ། གཽ་གེ་ཨ་མཱུ་ཧཾ་གཱ་དག་འདིར་སོགས་སོ། །མ་ཎཱི་བཱ་དེན། མ་ཎཱི་ཨི་བ་སོགས་ཀྱི་མ་ཡིན་ནོ། །གཉིས་ཀྱི་ཚིག་ཅན་ཡིན་ཀྱང་མ་ཎི་ལ་སོགས་པ་རྣམས་ཀྱི་ཨིབ་ཕ་རོལ་ལ་རང་བཞིན་དུ་གནས་པ་མ་ཡིན་གྱི་མཚམས་སྦྱོར་བ་ཁོ་ན་སྟེ། མ་ཎི་ཨེ་བ་ཞེས་པ་ལ། མ་ཎིནོར་བུ་འདྲ་བ། དེ་བཞིན་དུ། དམྤ་ཏེ་པ། ཁྱོ་ཤུག་འདིའམ་བཟའ་མི་འདི། རོ་ད་སཱི་བ། གནམ་ས་འདི། ཛམྦ་ཏཱི་བ་འབྱོར་པའི་་་་་བདག་པོ་འདི་སོགས་སོ། །།ཨ་ད་སོ་མཱཏ། ཨ་དས་ཀྱི་མ་ལས་སོ། །ཨ་དས་ཀྱི་ད་མར་བསྒྱུར་བའི་མཐའི་ཨཱིཏ་དང་ཨུཏ་དག་ཨཙ་ཕ་རོལ་ལ་མཚམས་སྦྱོར་བ་མ་ཡིན་ཏེ། ཨ་མཱི་ཨཱི་ཤཱ། འདི་རྣམས་དབང་ཕྱུག། ཨ་མཱུ་ཨཱ་སཱ་ཏེ། འདི་དག་སྡོད་པར་བྱེད། ནི་པཱ་ཏ་ཨེ་ཀཱ་ཛ་ནཱང་། ་ཨཱང་མ་ཡིན་པའི་ནི་པཱ་ཏ་ཨཙ་གཅིག་པའོ། །ཨཱང་མ་ཡིན་་པ་སྟེ་ཅུང་ཟད་དོན་དང་མངོན་ཁྱབ་ལ་སོགས་དོན་དུ་མ་ལ་འཇུག་པའི་ཨཱང་བོར་་་བའི་ཨཙ་གཅིག་པ་ནི་པཱ་ཏའི་མིང་ཅན་རྣམས་ཨཙ་ཕ་རོལ་ལ་མཚམས་སྦྱོར་བ་མ་ཡིན་ཏེ། ཨ་ཨ་ོནནྟ། ཁྱབ་འཇུག་མཐའ་ཡས། ཨི་ཨིནྡྲ་པ་བྱ། འདིར་དབང་པོ་བལྟས། ཨུ་ཨུ་མེ་ཤ། དབང་ཕྱུག་ཨུ་མའི་བདག་པོ། ཨཱ་ཨེ་ཝཾ། ཇི་འདྲར་དེ་ཉིད། ཨཱང་ཡིན་ན་་་་་་མཚམས་སྦྱོར་ཏེ། ཨཱ་:ཨུཏྟ་མཞེས་པ་ལ། །ཨོཏྟ་མ། ཅུང་ཟད་མཆོག་སོགས་སོ། །ཨོཏ༑ཨོཏ་འོ། །ཨོཏ་མཐའ་ནི་པཱ་ཏའི་མིང་ཅན་རྣམས་ཀྱང་ཨཙ་ཕ་རོལ་ལ་མཚམས་སྦྱོར་བ་མ་ཡིན་ནོ། །ཨ་ཧོ་ཨི་མ་དབང་ཕྱུག། ཨཱི་ཤ། ཨ་ཐོ་དེ་ནས་འདི་སྐད་ཨེ་ཐཾ་སོགས་སོ། །འགྲེལ་པར་ཧེ་དང་ཧི་དག་ཀྱང་ཨཙ་ཕ་རོལ་ལ་མཚམས་མི་་་—།
