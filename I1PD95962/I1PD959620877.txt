སྒྲ་མདོ། ཏི༽པའི་ཟླ་འོ་མཁོའི་ཤཚོགས་ར་ཁ་བ་བློ་ཀའི་ཀཱ་ཏེ། ་ཙ་ཀྲྀ་རེ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཀྲྀ་ཥ། །།ཙ་གྲྭ་ཐེ། ཙ་ཀྲྀ་དྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཙ་ཀྲེ། ཙ་ཀྲྀ་བ་ཧེ། ཙ་ཀྲ་མ་ཧེ། ལུཊ་ལ། །བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨ་ཏང་ལ། ཀཏྟཱ༑ཀཎཱ་སི། སོགས་དང༌། ཏང་ལ། ་ཀཏཱ། ཀཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། །ཤ་ཡཀ་ལིང་རྣམས་ལ་རིང་འོས། རཾ་རིང་དུ་་་་བསྒྱུར་བས། ཀྲྀ་ཝིཡཱཏ། ཀྲྀ་ཡཱནྟཾ། ཤྲིཡཱ་སུ། སོགས་དང་། ཏང་ལ། ཀྲྀ་ཏཱི། ཀྲྀ་ཊཱི་ཡནྟཾ། ཀྲྀ་ཏི་རན། སོགས་སོ། །ལྀཊ་ལ། རྀཏ་ཧ་ན། དག་ལས་ས་འོས། ཨིཊ་བྱིན། བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་ཡོན་ཏན་་་བྱས་པས། ཨ་ཏང་ལ། ་ཀ་རི་ཥྛ་ཏི། སོགས་དང༌། ཏང་ལ། ཀ་རི་ཉྩ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་གི་ཨ་ཏང་ལ། ཨ་ཀ་རི་ཏ། སོགས་དང་། ཏང་ལ། ་ཨ་ཀ་རི་ཥྛ་ཏི༄༅། སོགས་སོ། །སཾ་པརླུ་པེ་བྷཱ། ཀ་རོ་ཏོ་བྷཱ་ཥ་ཎེ། སཾ་པ་རི་ཨུ་པ་རྣམས་ལས་ཀ་རོ་ཏི་ལ་རྒྱན་ལའོ། །ཉེར་བསྒྱུར་སཾ་པ་རི་ཨུ་པ་རྣམས་ལས་ཕ་རོལ་གྱི་ཀ་རོ་ཏིལ་རྒྱན་གྱི་དོན་ལ་སུཊ་ཨཱ་ག་མ་འགྱུར་ཏེ། ཊ་གནས་དོན་དང་ཨུ་བརྗོད་པའི་དོན་ནོ། །སཾ་སྐ་རོ་ཏི། རྒྱན་ལེགས་པར་བྱསི། རྒྱན་ཡོངས་སུ་བྱེད། པ་རིཥ་རོ་ཏི། བ་རི་ནི་བི་རྣམས་ལས་སོགས་ཀྱི་ས་ཥར་བསྒྱུར་བའོ། །ཨུ་པ་སྐ་རོ་ཏི། རྒྱན་ཉེ་་་བར་བྱེད། ༈པ་རི་སྔོན་མའི་ཁྲི་ལ་ལུང་ཨཊ་ཨ་རྒྱུས་བར་ན་ཡོད་ཀྱང་ངོས། ཀའི་སྔོན་དུ་སུཊ་བྱིན། སིབ་ལ་སོགས་པ་རྣམས་ཀྱིས་ཡང་ན་འོས། ས་ཀར་ཡང་ན་བར་བྱས། །པརྨཨཱཿནཱིཏ། རྒྱན་ཡོངས་སུ་བྱས། པརླུཥྛནཱི་ཏ༑༌༌ཨུ་པ་སྔོན་མའི་ཁྲི་ལ་ལིཊ་ཨཊ་ཨ་བྷྱཱས་བར་ན་ཡོད་ཀྱང་ངོས། ཀའི་སྔོན་དུ་སུཊ། ཨུ་པ་ཙམྦྷ་ར། སྔོན་རྒྱན་ཉེ་བར་བྱས། ཨུ་པ་དང་སཾ་སྔོན་མའི་ཁྲི་ལ། ལི་ཊ་ཨ་ཊ་ཨ་བྷྱཱ་ས་དང་བར་ན་ཡོད་ཀྱང་ངོས། སུཊ་བྱིན། ལ་ཏས་དང་ཛཱི་རུ་བསྒྱུར། ༈དེ་གཉིས་ཨ་ཏུས་དང་ཨུས་སུ་བསྒྱུར། སུཊ་ཀའི་སྟེང་དུ་བྱུང་བས། བརྩེགས་པའི་ཕ་རོལ་ལ་ལིཊ་རྣམས་ལྡིཏ་མ་ཡིན་པསི། བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པ་སྟེ། ཨ་ཏང་དང་ཏང་གཉིས་ཀའི་ལྷག་མ་ཀུན་ལ་རིགས་འགྲེའོ། །:ཨུ་པ་ཙ་སྐ་ར་ཏུ། སྔོན་རྒྱན་ཉེ་བར་དག་བྱས། སཾ་ཙསྐ་ར་ཏུ། སྔོན་རྒྱལ་་་—།
