ཀླཏ༑ཨ་:ཀྲྀ་ཀླུ་ཏྭཾ?། ཨ་ཀྲྀ་ཀུན། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཀྲྀ་ཀྵ། ཨ་ཀྲྀ་ཀཱ་ཏཾ། ཨ་ཀྲྀ་ཀཱཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀྲྀ་ཏྟཾ། ཨ་ཀྲྀ་ག་བ། ཨ་ཀྲྀ་གླུ་མ། ཀྲྀ་ཥ་ལ་ལིཌ་སྦྱར་བའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཀཆུ། ཙ་ཀྲྀ་ཥ་ཏུ། ཙ་ཀྲྀ་། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཀརྟི་ཐ། ཙ་ཀརྨ། ཙ་ཀྲྀ་ཥ་ཐུ། ཙ་ཀྲྀ་ཥ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཀཆུ། ཙ་ཀརྙི་བ། །ཙ་ཀརྟི་མ། ༼ཀྲྀ་ཥ་ལ་ལུཊ་སྦྱར། ཨ་ནུ་དཱཏྟའི་འང་རྀཏ་ཨུ་པ་དྷཱའི་ཡང་ནའོ་་་་ཞེས་པས་ཨམ་ཨཱ་ག་མ་ཡང་ན་བར་བྱས་པས། ཨམ་བྱིན་པའི་ཕྱོགས་ལ། ཀྲཏྟཱ། ཀྲཥྛ་རོ། ཀྲ་ཏྟཱ་རཿསོགས་སོ། །ཨམ་མ་བྱིན་པའི་ཕྱོགས་ལ། པུཀ་མཐའ་ཨུ་པ་དྷཱ་ཐུང་དུ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཀནྡྷ། ཀནྡྷ་རོ། །ཀནྟཱ་རཿསོགས་སོ། །སྨོན་པའི་ལིང་ལ། །ཀྲྀ་ཏ་སོགས་སོ། །ལྀཊ་ལ། ཨམ་ཨཱ་ག་་མ་ཡང་ན་བར་བྱས་པས། ཨམ་བྱིན་པའི་ཕྱོགས་ལ། ཀྲགླུ་ཏི། ཀྲཀྵ་ཏ། ཀྲགནྟི་སོགས་སོ། །ཨམ་མ་བྱིན་པའི་ཕྱོགས་ལ། ཀགླུ་ཏི། ཀགྱུ་ཏ། ཀpནྟི་་སོགས་སོ། །དེ་བཞིན་དུ། ལྭང་ལའང་ཨམ་ཡང་ན་བར་བྱས་པས། ཨམ་་་བྱིན་པའི་ཕྱོགས་ལ། ཨ་ཀྲཀྵ་ཏ། ་ཨ་ཀྲཀྵ་ཏྭཾ། ཨ་ཀྲགྲུ་ན་སོགས་སོ། །།ཨམ་མ་བྱིན་པའི་ཕྱོགས་ལ། །ཨ་ཀརྒྱུཏ། ཨ་ཀརྞ་ཏྭཾ། ཨ་ཀརྞ་ན་སོགས་སོ། །པོ་པ། ༈རུ་ཥ་འཚེ་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ། །ལཊ་སོགས་པ་རསྨི་པ་ད་་་རྣམས་སྦྱར་བའི་ཆོ་ག་རྣམས་ཀྱང་ལུཊའི་སྐབས་མ་གཏོགས་སྔ་མ་རྣམས་ཀྱིས་་་རིགས་འགྲེ་སྟེ། ལཊ་ལ། རོ་ཥ་ཏི་སོགས་སོ། ་། ལིང་ལ། རོ་ཥ་ཏ་སོགས་སོ། །ལོཊ་ལ། རོ་ཥ་ཏུ་སོགས་སོ། །ལང་ལ། ཨ་རོ་ཥཏ་སོགས་སོ། །ལུང་ལ། ཤལ་ཨི་ཀ་ཨུ་པ་དྷཱ་ཨིཊ་མེད་པ་ལས་ཀླ་འོས། ཙྪི། ༼ཀར་བསྒྱུར་བས། །ཨ་རུ་གཏ༑ཨ་རུཀྵ་ཧཱུྃ། ཨ་རུ་གྱུན་སོགས་སོ། །།ལིཊ་ལ། རུ་རོ་ཥ། རུ་རུ་ཥ་ཀྲ།
