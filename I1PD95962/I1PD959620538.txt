ནོ་ཏི་ལ་སོགས་པ་རྣམས་ཀྱི་ས་ཡིག་ནི་ཨཊ་ཨཱ་ག་མ་བར་ན་ཡོད་པ་ལས་ཀྱང་ཥ་་་ཡིག་ཏུ་འགྱུར་ཏེ། ལང་ལ། རྒྱ་ཥ་དྷ་ཏ་སོགས་སོ། །ལུང་ལ། །ནྱ་ཥ་དྷཱིཏ་་་སོགས་སོ། །།སྒྲ་དི་ཥཱ་བྷཱ་སེ་ན་ཙཱ་བྷཱ་ས་སྱ། སྦྲ་ལ་སོགས་པ་རྣམས་ལ་ཨ་བྷཱ་ས་་་ཡིས་ཀྱང་ཨ་ཧྥུ་སའི་འོ། ?། ཉེར་བསྒྱུར་རྒྱུ་མཚན་ཅན་སྔ་མ་ལས། སྒྲ་ལ་སོགས་པ་རྣམས་ཀྱི་བྱིངས་གཉིས་བརྗོད་པ་ཨ་ངྷ་སའི་མིང་ཅན་ཡོད་ན་ཨ་བྷྱཱ་ས་དེ་ཉིད་ཥ་ཡིག་ཏུ་འགྱུར་ཏེ། ལིཊ་ལ། ་ནི་ཏི་ཥ་དྷ། མདོ་འདིས་ས་སྔ་མ་ཥ་དང་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཕྱི་མ་ཥ། ནི་ཌི་ཥ་དྷ་ཏུནི་ཏི་ཎི་དྷུབརྩེགས་པ་མ་ཡིན་པའི་ཞེས་པས་འདིར་ལིཊ་ཀིཏ་ཡིན་པས་ཡོན་ཏན་མི་བྱེད། སོགས་སོ། །ལུཊ་ལ། ནི་ཥ་དྷི་ཏཱ་སོགས་སོ། །།སྨོན་པའི་ལིང་ལ་ནི་ཏི་དྷཏ་སོགས་སོ། །ཁྲིཊ་ལ་ནི་ཥ་དྷི་ཥྛ་ཏི་སོགས་དང་། ཁྲིང་ལ་རྒྱ་ཥ་དྷི་༄༅ཏ༑སོགས་སོ། །སེ་དྷ་ཏེརྒ་ཏོ། །སེ་དྷ་ཏི་འགྲོ་བ་ལའོ། ་། ཉེར་བསྒྱུར་་རྒྱུ་མཚན་ཅན་སྔོན་ན་ཡོད་ཀྱང་བྱིངས་ཎི་དྷ་འགྱུར་བའི་ས་ཡིག་ནི་འགྲོ་བའི་དོན་བརྗོད་པ་ལ་ཥར་འགྱུར་བ་མ་ཡིན་ཏེ། ནི་སྔ་མའི་ཎི་དྷ་འགྲོ་བ་ལ་ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བའི། ལཊ་ལ། ནི་སེ་དྷ་ཏི་སོགས་ཏེ། ངག་ཏུ་སྦྱར་ན། །ནི་སེ་དྷ་ཏི་ནཱཾ། …བ་ལང་རྣམས་ངེས་པར་འགྲོ་བར་བྱེད། ལིང་ལ་ནི་སེ་དྷེ་ཏ། སོགས་སོ། །ལོཊ་ལ། ནི་སེ་དྷ་ཏུ། སོགས་སོ། །ལང་ལ། རྒྱ་སེ་དྷ་ཏ་སོགས་སོ། །ལུང་ལ་རྒྱ་སེ་དྷཱི་ཏ། ·སོགས་སོ། །་། ལིཊ་ལ། ནི་སི་སེ་དྷ། ནི་སི་སི་དྷ་་་ཏུ…བརྩེགས་པ་མ་ཡིན་པའི་ཞེས་པས་འདིར་ལིཊ་ཀིཏ་ཡིན་པ་ཨུ་པ་དྷཱ་ཡོན་ཏན་མི་བྱེད། སོགས་སོ། །ལུཊ་ལ། ནི་སེ་དྷི་ཏཱ་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ནི་སི་དྷཱཏ། སྨོན་པ་ལ་ཀིཏ་འོས་ཨུ་པ་དྷཱ་ཡོན་ཏན་མི་བྱེད། སོགས་སོ། །ལྀཊ་ལ། ནི་སེ་བྷི་ཥྛ་ཏི་སོགས་སོ། །ལྀང་ལ། རྒྱ་སེ་དྷི་ཏི། སོགས་སོ། །ཏི་པ་བ་དང་ཁྱོད་ཀྱི་ཚེ་ལ་ཕྱི་ལ་༈ཎི་དྷཱ་བསྟན་བཅོས་ལ་སྟེ་བཀྲ་ཤིས་པ་ལ་འང་ངོ༌། །།ཞེས་པའི་་་—།
