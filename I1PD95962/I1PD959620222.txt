ICTདཀར་༈ཙ་ཡིག་མཐའ་ཅན། སུ་བྲིཉྩ་ལ། ་སུ། ཡང་དག་པར་སྦྱོར་བའི་་མཐའ་ལ་སྔ་མའི་སྲ་ཀྲ་དག་གི་འང་ངོས། ཤ་ཕྱིས། ་བྲཧྥ་བྷ་ལྔ་སོགས་ཀྱིས་རྩྭ་ཥ། །ཛྷཤ་ལ་ཛྷལ་རྣམས་ཀྱི་ཛཤ་རྣམས་སོས། ཥ་ཌ། །གནས་མེད་པ་ལ་ཡང་ན་འོས། སུ་བྲྀཊི། སུ་པྲྀཊ། ཨོ་དང་ཛས་ལ། སུ་བྲྀཏྟཾ། ལསུ་བྲོ། གཉིས་པ་རྣམས་ལ། སུ་བྲིཤྩ། སུ་བྲྀཏྟཾ། སུ་བྲི། ཊཱ་ལ། ་སྲ་བྲིཤྩ། བྷཾ་བྷིས་ལ། :སུ་བྲིདྡྷཾ། །སུ་ཤྲིནྟི། ངེ་ལ། སུ་བྲི། བྷཱུམ་རྒྱས་འདྲ། ང་སི་ལ། སུ་ཏྲཾ། ཧཱུཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། སུ་བྲི། སུ་པྲྀབྷོ། སུ་བྲྀཧཱུཾ། བདུན་པ་རྣམས་ལ། སུ་བྲེ། སུ་བྲི། །སུ་བྲྀཀྵ། བོད་ཚིག་རྣམས་ལ། ཧེ་སུ་དྲི་ཌ། ཧེ་སུ་དྲྀཊ། ཧེ་སུ་བྲྀཏྟཾ། ཧེ་སུ་བྲི། །།ལྷ་ཀ༈ཏ་ཡིག་མཐའ་ཅན། མ་ཧཏ་ཆེན་པོ་ལ། སུ། ཨུཀ་འགྲོ་བ་སོགས་་ཀྱིས་ནུམ། ས་མཐའ་མ་ཧཏ་དག་གི་ཡང་དག་པར་སྦྱོར་བའི་འོས་ཨུ་པ་དྷཱ་རིང་པོ། ཧལ་སོགས་ཀྱིས་སུ་ཕྱིས། །ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས། ཏ་ཕྱིས། མ་ཧཱན། ཨོ་དང་ཛས་ལ། མ་ཧཱནྟོ། མ་ཧཱནྟ། ཨམ་དང་ཨོ་ལ། མ་ཧཱནྟཾ། མ་ཧཱནྟཾ། ཤས་ལ། །མ་ཧཏ། །ཌཱ་ལ། མ་ཧ་ཏཱ། བྷཱུམ་དང་བྷིས་ལ། ཛྷཤ་ལ་ཛྷལ་རྣམས་ཀྱི་ཛཤ་རྣམས་སོས། ཏྲ་དྲ། མ་ཧདྡྷཾ། མ་ཧདྷི། ངེ་ལ། མ་ཧ་ཏེ། བྷཱུམ་བྷྱས་འདྲ། ང་སི་ལ། མ་ཧ་ཏ། བྷཱུམ་བྱུས་འདྲ། ་དྲུག་པ་རྣམས་ལ། མ་ཧ་ཏ། མ་ཧ་ཏོ། ་མ་ཧ་ཏྭཾ། བདུན་པ་རྣམས་ལ། མ་ཧ་ཏི། མ་ཧ་ཏོ། མ་ཧཏུ། བོད་ཚིག་ལ། ཧེ་མ་ཧན། ཧེ་མ་ཧཱནྟོ། །ཧེ་མ་ཧཱནྟ། ཏྟཱ་བྷ་ཝཏ་ལ་ཡང་ཨུ་པ་དྷཱ་རིང་པོར་མི་འགྱུར། །གཞན་ཆོ་ག་་་ཐམས་ཅད་མ་ཧཏ་དང་འདྲའོ། །དྷི་མཏ་ཅེས་པ་ལ་ཡང་། ཨཏ་ཨས་མཐའི་འང་བྱིངས་མ་ཡིན་པའིའོ་ཞེས་པས་སུའི་སྐབས་སུ་ཨུ་པ་དྷཱ་རིང་པོ་བྱེད། གཞན་_།
