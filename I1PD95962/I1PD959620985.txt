བ་ན་བསྟན་འགྱུར། སྒྲ་མདོ། །ཉིཔ་ཎིའི་ལྷོ་མ་གོ་ལ་གཏོཥ་ན་ཀ་ལ་ལྟོ་བ་ལ་བརྗོད༑ཧལ་དང་པོ་ལྷག་པོ་འོས། བ་ཕྱིས། མདོ་འདིས་ས་ཥར་མི་བསྒྱུར་བས། ལཊསི་སྭཱ་ད་ཡི་ཥ་ཏི། ་ལུང་…ཨ་སི་སྦེ་ད་ཡི་ཏཱིཏ། ལིཊ་སི་སྭཱ་ད་ཡི་ཊཱ་ཙ་ཀཱ་ར། སོགས་སོ། །།བདེ་དང་འཚེ་གོམ་ཁ་ཏོ་། ༈། །༈སཱ་ད་རོ་མྱང་བ་ལ། ་ཎིཙ་ཨུ་པ་དྷཱའི་འོས་འཕེལ་བ་བྱས་པས། སཱ་་དི་ཞེས་པ་ལ་སན་ཨིཊ་བྱིན། གཉིས་བརྗོད་པ་ཕྱིས། སན་ལ་ཨིཏ་འོས་ཨིཏ་དུ་བསྒྱུར༑མདོ་འདིས་ས་ཥར་མི་བསྒྱུར་བས། ལཊ…སི་སྭཱ་ད་ཡི་ཥ་ཏི། ལུང་ཨ་སི་སཱ་ད་ཡི་ཏཱི་ཏ། ལིཊ་སི་སྭཱ་ད་ཡི་ཏཱཾ་ཙ་ཀཱ་ར། སོགས་སོ། །མི་རང། །༈ས་ཧ་བཟོད་པ་ལ་ཎིཙ་འཕེལ་བ་བྱས། ༼སཱ་ཧི་ཞེས་པ་ལ་སན་སྦྱར། ཨིཊ་བྱིན། གཉིས་བརྗོད། སན་ལ་ཨི་ཏ་འོས་ཨིཏ་དུ་བསྒྱུར། མདོ་འདིས་ས་ཥར་མི་བསྒྱུར་བས། ་ལཊི་སི་སཱ་ཧ་ཡི་ཥ་ཏི། ྋལུང་ཨ་སི་སྭཱ་ཧ་ཡི་ཊྛིཏ། ལིཊསི་ས་ཧ་ཡི་ཌཱ་ཙ་ཀཱ་ར། སོགས་སོ། །ཀུན་ཁཀི། [@ན་གྱི་ཁྱོ་ག་༈འགྲེལ་པར་རང་དོན་ལ་བྱིན་པའི་སན་གྱི་མཐའ་ལ་འང་དགོས་པ་ཡོད་ན་སན་སྦྱོར་རོ་ཞེས་བཤད་པས། གུ་པཱུ་སྦེད་པ་ལ་གུཔ་ཏི་ཛ་ཀིཏ་རྣམས་ལ་སན་འོས༑ཏ་སན་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཛུ་གུཔ་ཞེས་པ་ལ་་་བྱིངས་ལས་ལས་དང་འདོད་པ་བྱེད་པ་པོ་མཉམ་པ་ལས་ཡང་ན་འོས། སླར་་་་ཡང་སན་བྱིན། ་གཉིས་བརྗོད་ཟིན་ཡིན་པས་སླར་མི་བརྗོད། ་བལ་དང་པོའི་ཨིཊ་ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། ཛུ་གུཔཱི་ཥ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊཛུ་གུཔཱི་ཥ་ཏི། ལིང་ཡཛུ་གུཔཱི་ཥིཏ། ལོཊ'ཛུ་གུཔྤི་ཏི་ཏུ། 《ལངཨ་ཛུ་གུཔྤི་ཥཏ༑ལུང་ཨ་ཛུ་གུཔཱིཏྟིཏ། ་ལིཊཛུ་གུཔྤི་ཏཱཾ་ཙ་ཀཱ་ར། ་ལུཊ་ཛུ་གུཔཱི་ཥི་ཏཱ། སྨོན་ལིང་ཛུ་གུཔྤེ་བྷཱཏ། ཁྲིཊཛུ་གུཔྤི་ཏི་ཥྛ་ཏི། ཁྲིང་ཨ་ཛུ་གུཔྤི་ཏི་ཥྛཏ། སོགས་སོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ནནྟས་མཱམྦྷ། །ཞེས་པ་པཱ་ཎི་ནི་ལས་།
