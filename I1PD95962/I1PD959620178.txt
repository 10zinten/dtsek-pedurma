རྟགས་ཨིཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོད་ལ་རྣམ་དབྱེ་ཨཙ་ཐོག་མ་ཅན་བྱུང་ན་་་ནུམ་ཨཱ་ག་མ་སྟེར་རོ། །བཱ་རི་ཎཱི། ༼ཛས་ལ། ཛས་ཤས་དག་གི་ཤི། རྣམ་དབྱེ་ཨཙ་ལ་ཞེས་པས་ནུམ། སམྦུདྷི་མ་ཡིན་པའི་ཞེས་པས་ཨུ་པ་དྷཱ་རིང་པོ། །་བཱ་རི་ཎི། །སཾ་བུདྷི་ལ། ཐུང་ངུའི་ཡོན་ཏན་ནོ་ཞེས་དང་། ཐུང་ངུ་དང་ཨེང་ཞེས་པས་སཾ་བུདྷི་ཕྱིས། ཧེ་བཱ་རེ། ཧེ་བཱ་རི་ཎི། ཧེ་བཱ་རཱི་ཎི། རྣམ་དབྱེ་གཉིས་པ་རྣམས་དང་པོ་དང་འདྲ། །ཊཱ་ལ། བཱ་རི་ཎཱ། བྷཾ་ལ། བཱ་རི་ཧཱུཾ། བྷིལ། བཱ་རི་བྷི། ངེ་ལ། བཱ་རི་ཎེ། ངྷ་འདྲ། ་བྷུས་ལ། བཱ་རེ་། ང་སི་་་ལ༑བཱ་རི་ཎ། ་བྷཾ་བྷུའདྲ། ངས་ལ། བཱ་རི་ཎ། ཨོས་ལ། བཱ་རི་ཎོ། །བཱ་རཱི་ཎཾ། ཇི་ལ། །བཱ་རི་ཎི། ཨོས་འདྲ། །སུཔ་ལ། བཱ་རི་ཥཱ། ཨ་ནཱ་དི་ཞེས་པའང་རྣམ་དབྱེ་ཊཱ་ཡན་ཆད་བཱ་རི་དང་འདྲ་སྟེ། ་དང་པོ་རྣམས་ལ། ཨ་་་ནཱ་དི། ཨ་ནཱ་དི་ནི། །ཨ་ནཱ་དཱི་ནི། བོད་ཚིག་ལ། ་ཧེ་ཨ་ནཱ་དེ། ཧེ་ཨ་ནཱ་དི་ནཱི། ཧེ་ཨ་ནཱ་དཱི་ནི། གཉིས་པ་རྣམས་ཀྱང་དང་པོ་དང་འདྲ། ཨ་ནཱ་དི། །ཊཱ། འི། ཀྲྀ་ཏཱི་ཡཱ་དི་ཥཱ་བྷཱ་ཥི་ཏ་པུ་སྐོ་་བངྒཱ་ལ་ཝ་སྱ། གསུམ་པ་སོགས་རྣམས་ལ་སྐྱེས་པར་བཤད་པའི་སྐྱེས་པ་བཞིན་ག་ལ་ཝའིའོ། །རྟགས་ཨིཀ་སོ་སོར་བསྡུ་བའི་མཐའ་ཅན་ལས་ཕ་རོལ་རྣམ་དབྱེ་གསུམ་པའི་ཊཱ་ནས་བདུན་པའི་སུཔ་ཀྱི་བར་གྱི་་་ཨཙ་ཐོག་མ་ཅན་རྣམས་ལ། སྔར་སྐྱེས་པའི་རྟགས་ཀྱི་སྐབས་སུ་བཤད་པའི་ཆོ་ག་རྣམས་དང་། །མ་ནིང་གི་རང་ལུགས་ལྟར་ཨཙ་ཐོག་མ་ཅན་རྣམས་ལ་ནུམ་སྟེར་བ་སྔོན་གྱི་དྲང་སྲོང་ག་ལ་བའི་ལུགས་སོ། །ཨ་ནཱ་དི་ནཱ་སྟེ་ལུགས་གཉིས་ཀ་གཟུགས་འདྲ། སྐྱེས་པ་ལྟར་ན་ཊ་ནཱ། །མ་ནིང་རང་ལུགས་ལྟར་ན་ནུམ་བྱིན་དྲངས་་་་པའོ། །བྷཾ་བྷིས་ལ། ་ཨ་ནཱ་ད་ཧཱུཾ། ཨ་ནཱ་དི་བྷི། ངེ་ལ། ་ཨ་ནཱ་ད་ཡེ། ཨ་ནཱ་དི་ནེ། བྷཱུམ་བྱས་འདྲ། ང་སི་ལ། །ཨ་ནཱ་དེ། "ཨ་ནཱ་དི་ན། ཧཱུཾ་བྱས་།
