།༈ཏི་བུ་ཐགས་ཀྱི་རྒྱུ་ལའོ་ཞེས་པ་ལའང་རྒྱུ་དང་བཅས་པ་སོགས་ཀྱིས་ཎིཙ་སྦྱར༑་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། སེ་བི་ཞེས་གྲུབ་པ་ལ་་་་ལཊ་སོགས་སྦྱར་བས། ལཊ་སེ་བ་ཡ་ཏི། ལིང་སེ་བ་ཡཏི། ལོཊ་སེ་བ་ཡ་ཏུ༑ལང་ཨ་སེ་བ་ཡཏ། སོགས་དང༌། ལུང་ལ། ཎི་ཤི་སོགས་ཀྱིས་ཙླི་་་ཙང་དུ་བསྒྱུར། ཙང་ལ་འོས་ངོས་བྱིངས་གཉིས་བརྗོད། ཨ་་ཡང་བའི་རིང་པོ་་་་འོས་རིང་པོར་བསྒྱུར། །ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཕྱི་མ་ཥ། ཎི་ལ་ཙང་ལ་ཨུ་པ་དྷཱའི་ཐུང་ངུ་འོས། ཨུ་པ་དྷཱ་ཐུང་དུར་བསྒྱུར། ་ཎིའི་ཨིཊ་མེད་པ་ལ་་་་་འོས། ཎི་ཕྱིས། པ་རི་སྔོན། སྟམྦྷ་སི་བུ་སོགས་ཀྱིས་ས་ཥར་མི་བསྒྱུར། མཚམས་སྦྱར་བས། པརླ་སཱི་ཏི་བཏ། པརྨ་སཱི་ཏི་བ་ཏྭཾ། པརྨ་སཱི་ཊི་བན། སོགས་སོ། །ལིཊ་ལ། ཨཱམ་སྦྱར་བས། སེ་བ་ཡཾ་ཙ་ཀཱ་ར། སོགས་དང༌། ལུཊ་ལ་ཨིཊ་བྱིན་པས། སེ་བ་ཡི་ཏཱ། སེ་བ་ཡི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ་ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎི་ཕྱིས་པས། ་་:སེ་བྷཱུ་ཏ། སེ་བྱཱ་ཙིནྟཾ། སེ་བྱཱ་སུ། སོགས་སོ། །།ལྀཊ་ལ། ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། །སེ་བ་ཡི་བྱ་ཏི། སོགས་དང་། ཁྲིང་ལ། ཨ་སེ་བ་ཡི་ཏ། །སོགས་སོ། །སྐྱ་༈ཥ་ཧ་བཟོད་པ་ལའོ་ཞེས་པ་ལ་ཎིཙ་སྦྱར། ཨུ་པ་དྷཱའི་འོས་འཕེལ་བ་་བྱས༑སཱ་ཧི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་སཱ་ཧ་ཡ་ཏི། ལིང་སཱ་་ཧ་ཡེཏ། ལོཊ་སཱ་ཧ་ཡ་ཏུ། ལང་ཨ་སཱ་ཧ་ཡཏི། སོགས་སོ། །ལུང་ལ། ཎི་ཤི་སོགས་ཀྱིས་སྨིཾ་ཙང་། ཙང་ལ་འོས་གཉིས་བརྗོད། ཨ་ཀ་མ་ཕྱིས་པ་ལ་་་ཡང་བ་ལ་ཙང་ཕ་རོལ་ལ་སན་བཞིན་ནོས། ཨ་ཨི་ཡང་བའི་རིང་པོ་འོས་དེ་རིང་པོ༑ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཕྱི་མ་ཥ། །ཎི་ལ་ཙང་ལ་ཨུ་པ་དྷཱའི་་་་་སམt།
