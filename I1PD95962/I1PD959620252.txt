རགས་པ་རྣམས་རེ་ཞིག་རྫོགས་སོ། །།།།།།།དབང་། །བྱེད་པ་དེ་ལྟར་རྟགས་དྲུག་ཚིག་ཏུ་བསྒྲུབ་པའི་ཆོ་ག་རྣམས་བཤད་ཟིན་ནས། །ད་ནི་མིང་གཅིག་གིས་དོན་དུ་མ་སྟོན་པ་ནི་པཱ་ཏའམ་མི་ཟད་པའི་སྡེ་རྣམས་སྟོན་པར་བྱ་སྟེ། །སཱ་རཱ་དི་ནི་པཱ་ཏམ་བྱ་ཡཾ། སྭ་ར་ལ་སོགས་པ་རྣམས་ནི་པཱ་ཏ་མི་ཟད་་་པའོ༑ཧི། །སཱ་ར་ལ་སོགས་པ་རྣམས་ལ་ཨ་བྱཱ་སྟེ་མི་ཟད་པ་ཞེས་སམ་ནི་པཱ་ཏ་ཞེས་་བྱའོ༑། སཱ་ར་ལ་སོགས་པ་ན་འདི་རྣམས་ཏེ། སཱར། མཐོ་རིས། །ཨོཾ་ཏར། བར་་་་མཚམས། པྲ་ཏར། སྔ་དྲོའམ་ཞོག་ཁ། པུ་ན་ར། སླར། ས་ནུ། ཉུང་ངུའམ་ཆུང་ངུ་། ཏར། བརྐྱལ། ཨུཙྪིས༑མཐོ་བའམ་རིང་པོ། ནིཙྪིས། དམའ་བ། ས་ནིས། ག་ལེའམ་དལ་པོ། ་་་དྷཀ༑གྲུབ་པའམ་འཛིན་པ། རཾ་ཏེ། སྤངས་པ། ཡུ་ག་པཏི། གཉིས། ཨཱ་རཱཏ། ཐག་རིང་ཐུང་གཉིས་ཀ་ལ་འཇུག། པྲ་ཐག། སོ་སོ། ཕྱིན་ནས། རྒྱས། ཁ་སང༌། སྭས། སོང་ཉིན། དི་ཝཱ། ཉིན་མོ། ར་ཏོ། ...མཚན་མོ། སཱ་ཡཾ། ...དགོངས་མོ། ཙེ་རཾ། ...མང་པོ། མ་ནཀ། ...ཡེ་ཤེས། ་ཨི་ཥཏེ༑ཆུང་དུ། ཛོ་ཁཾ། དགའ་བ། ཏུཏྟཱི། མི་སྨྲ་བ། བ་ཧིས། ཕྱི། ཨ་ཝས། ཞ་དང་་་་བསྲུང་བ་དང་རྒྱན་རྣམས་ལ། ས་མ་ཡཱ། དམ་ཚིག་གམ་ཉེ་བ། ནི་ཀ་ཥཱ། ཐག་ཉེ་བ། སཱ། ང་འམ་རང་། སྭ་ཡཾ། ཐཾ་ཐཱ། བརྫུན། ནཏྟཾ། མཚན་མོ། ནཉ། དགག་ཚིག། ཧེ་ཏོ། གཉེན། ཨིབྷཱ། …མ་་་འདྲེས་པ། ཨདྷཱ། མ་འདྲེས་པ། སཱ་མི། མཚུངས་པའམ་འདྲ་པ། ་་་ན། །ཅད་པ། ཀྱི་བཏྤ་ཏྱ་ཡཾ་ཏཾ་ཙ། བ་ཏུའི་རྐྱེན་མཐའ་རྣམས་ཀྱང་ངོ༌། །བྱིངས་་་་སམ་རྟགས་བ་ཏུའི་རྐྱེན་མཐའ་ཅན་རྣམས་ཀྱང་མི་ཟད་པའོ། །བྲཱཧྥ་ཎ་་་པཏ༑བྲམ་ཟེ་ལྟར་རམ་བྲམ་ཟེ་ཅན། །ཀླུ་ཏྲི་ཡ་བཏ། རྒྱལ་རིགས་ལྟར་རམ་ཅན། །ཅེས་པ་་་ལྟ་བུ་བ་ཏུའི་རྐྱེན་མཐའ་ཅན་དང་། གཞན་ཡང་། ས་ན། བདེན་པའམ་ངེས་པ། ཨུ་པདྷཱ། ཐག་ཉེ་བའམ་རིང་བ་མིན་པ། ས་ནཏ། རྙིད་པའམ་སྔོན་མ་འམ་སྔ་སོར། ས་ནཱ་་་།
