ཡ་ཏུ། ་ལང་ཨ་སྭ་བ་ཡཏ། །སོགས་དང་། །ལུང་ལ། །སྲ་བ་ཏི་ཤྲཱི་ཎོ་ཏི་དྲ་་བ་ཏི་བྲ་བ་ཏི་བླ་བ་ཏི་བྱ་བ་ཏཱི་ནཱ་བཱ། སྲ་བ་ཏི་ཤྲཱི་ཎོ་ཏི་དྲ་བ་ཏི་པྲ་བ་ཏི་བླ་བ་ཏི་བྱ་བ་ཏི་རྣམས་ཀྱི་ཡང་ནའོ། ་། སུ་འགྲོ་བ། ཐུ་ཉན་པ། དྲུ་འགྲོ་བ་སྲུ་དང་་་རླུང་དང་བྱུང་རྣམས་འགྲོ་བ་རྣམས་ཀྱི་སན་དང་ཎི་དང་བཅས་པའི་ཙང་ཕ་རོལ་ལ་ཨ་བྷྱཱ་ས་ཨིར་ཡང་ན་བསྒྱུར་ཏེ། ཨ་སི་སྲ་བཏ། སོགས་སམ། ་ཨ་སུ་སྲ་བཏ། སོགས་སོ། །ལ་ཡིག་ལྷག་མ་རྣམས་ལ། ལིཊ་སྭ་བ་ཡཾ་ཙ་ཀཱ་ར། ལུཊསྭ་བ་ཡི་ཏཱ། སྨོན་ལིང་ས་བྱཱཏ། ཁྲིཊ་སྭ་བ་ཡི་ཥྛ་ཏི། ཁྲིང་…ཨ་སྭ་བ་ཡི་ཏ། སོགས་སོ། ། ོ། །༈དེ་བཞིན་དུ། ་ཐུ་ཐོས་པ་ལ་ཎིཙ་སྦྱར་བས། ཤཱ་བི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊཡཐཱ་བ་:ཡ་ཏི། ལིང༌ཐཱ་བ་ཡེཏ། ལོཊཐཱ་བ་ཡ་ཏུ། ལང་ཨ་ཤྭ་བ་ཡཏི། སོགས་དང༌། ལུང་ལ། ཨིར་ཡང་ན་བསྒྱུར་བས། །ཨ་ཤི་ཐ་བཏ། སོགས་སམ། ཨ་ཤུ་ཤ་བཏ། སོགས་སོ། །ལིཊ་སོགས་ལྷག་མ་རྣམས་ལ། ལིཊཐཱ་བ་ཡཱ་ཙ་ཀཱ་ར། ལུཊ་ཡཐཱ་བ་ཡི་ཏཱ། །སྨོན་ལིང་ཐཱ་བྱཱཏེ། ཁྲིཊཤྭ་བ་ཡི་ཥྛ་ཏི། ཕྲེང་ཨ་ཤྭ་བ་ཡཏ། །སོགས་སོ། །ལྟ་ཀྱི། །འདས་:༈དྲུ་འགྲོ་བ་ལ་ཎིཙ་སྦྱར་བས། 《དྲྭ་བི་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་…དྲྭ་བ་ཅིཡ་ཏི། ལིང་དྲྭ་བ་ཡེཏ། ལོཊདཱ་པ་ཡ་ཏུ། ལང་ཨ་དཱ་པ་ཡཏ། ལུང་ལ། ཨིར་ཡང་ན་བསྒྱུར་བས། ཨ་དི་དྲ་བཏ། སོགས་སམ། ་ཨ་དུ་དྲ་བཏ༑/སོགས་སོ། །ལིཊ་སོགས་ལྷག་མ་རྣམས་ལ། ལིཊ་…དཱ་བ་ཡཱ་ཙ་ཀཱ་ར། ལུཊདཱ་བ་ཡི་ཏཱ། སྨོན་ལིང་དཱ་བྷཱུཏ། ཁྲིཊ་དཱ་བ་ཡི་ཥྛ་ཏི། ་ཕྲེང་ཨ་དྭ་བ་ཡི་་ཏ། སོགས་སོ། །ང་། །།།སྔགས་ཀྱི་ནད། །ཁ་༈ཕྲུ་འགྲོ་བ་ལ་ཎིཙ་སྦྱར། ་བི་ཞེས་:པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་པྲ་བ་:ཡ་ཏི༄༅། །ལིང་ཡཔྲ་བ་ཡེཏ། ལོཊ་…པཱ་བ་ཡ་ཏུ། ལང་ཨ་པྲ་བ་ཡཏ། Ich—།
