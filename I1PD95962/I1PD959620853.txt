སྒྲ་མདོ། །དེ་ལ་ཟླ་བོ་ཐ་nཚོ་ཎ་ཥ་ཚོགས་སྒོ་ཀན་རྣམས་སྦྱོར་ཏེ། སྲྀ་ཊྛིཏྲ། མྲྀ་ཏཱི་ཡཱ་སིཾ། ནྲྀ་ཏཱི་རན། སོགས་སོ། །ཁྲིཊའི་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། རྀཏ་ཧན་དག་ལས་སྐྱལ་འོས་ཨིཊ་བྱིན། བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་ཡོན་ཏན་བྱས་པས། །མ་རི་ཥྛ་ཏི། སོགས་དང་། དེ་བཞིན་དུ། རྀང་གི་ཨ་ཏང་རྣམས་སྦྱོར་བས།  ོཨ་མ་རི་ཥྱ་ཏ༑སོགས་སོ། །ཁྱབ། :ཎེ། །པོ་ཏ་ཁ་ཁཊ། ཁ་ཁ། ཁྲོ་༈ཨོ་བི་ཛཱི་འཇིག་པ་དང་གཡོ་བར་བྱེད་པ་དག་ལའོ་ཞེས་པ་ལ་ཏང་རྣམས་སྦྱོར་ཏེ། ལཊ་ལ། བི་ཛ་ཏེ། སོགས་དང་། ལིང་ལ། བི་ཛེ་ཏ། སོགས་སོ། །ལོཊ་ལ། བི་ཛ་ཊཱ། སོགས་དང་། ལང་ལ། ཨ་བི་ཛ་ཏ། སོགས་སོ། །བི་ཛ་ཨིཊ། བི་ཛ་ལས་ཨིཊའོ། །བི་ཛ་ལས་ཕ་རོལ་གྱི་ཉ་ཎ་འགྲོ་བའི་མིང་ཅན་མ་ཡིན་པས་རྐྱེན་ཨིཊ་དང་པོ་ཅན་ནི་ང་འགྲོ་བའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱེད་དེ། དགོས་པ་ནི་ཡོན་ཏན་དང་འཕེལ་བ་འགོག་པའི་དོན་ཏེ། ལུང་ལ། ཨ་བི་ཛི། ཨ་བི་ཛི་ཥཱ་ཏྭཾ། ཨ་བི་ཛི་ཥ་ཏ། །སོགས་སོ། །ལིཊ་ལ། གཉིས་བརྗོད་པ་སོགས་ཀྱི། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛཱ་ཏེ། བི་བི་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །བི་བི་ཛི་ཥ། བི་བི་ཛཱ་ཐེ། །བི་བི་ཛི་དྡྷེ། བི་བི་ཛི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛི་བ་ཧེ། བི་བི་ཛི་མ་ཧེ། ལུཊ་ལ་ཨིཊ་བྱིན། །བི་ཛ་ལས་ཨིཊ་འོས། རིཏ་ཡིན་པས་ཡོན་ཏན་མི་བྱེད་དེ། བི་ཛི་ཏཱ། བི་ཛི་ཏཱ་སེ། སོགས་སོ། །དེ་བཞིན་དུ། སྨོན་པའི་ལིང་ལ། བི་ཛི་ཏཱིཏྲ། བི་ཛི་ཌི་ཡཱནྟཾ། བི་ཛི་ཁཱི་རན། །སོགས་སོ། །ཁྲིཊ་ལའང་ཨིཊ་བྱིན། བི་ཛ་ལས་ཨིཊ་འོས། རིཏ་ཡིན་པས་་་ཡོན་ཏན་མི་བྱེད། བི་ཛི་ཉྩ་ཏེ། བི་ཛི་བྱེ་ཏེ། བི་ཛི་ཉྩནྟེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་བི་ཛི་ཥྛ་ཏི། ཨ་བི་ཛི་བྱེ་ཏྭཾ། ཨ་བི་ཛི་ཏྱནྟ། —།
