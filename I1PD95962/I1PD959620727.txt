སྒྲ་མདོ། །ཏི༽ཉིན་ནི་ད་ལྟ་གོ་ལའི་ན་ཚོད། ཕྱིས་ཀསྦྱོ་ཏ། ཨཱི་ཨི་ཡ་རུ་བསྒྱུར་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཤི་ཤི་ཡེ། ཤི་ཤི་ཡཱ་ཏེ། ཤི་ཤི་ཡི་ར། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཤི་ཤི་ཡི་ཥ། ཤི་ཤི་ཡཱ་ཐེ། ཤེ་ཤི་ཡི་དྷེ། ,ཤི་ཤི་ཡི་རྗེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཤི་ཤི་ཡེ། ཤི་ཤི་ཡི་བ་ཧེ། ཤི་ཤི་ཡི་མ་ཧེ། ལུཊ་ལ་བལ་དང་པོའི་ཨིཊ་དང་། བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཤ་ཡི་ཏཱ། ཤ་ཡི་ཏཱ་སེ། སོགས་སོ། །དེ་བཞིན་དུ། སྨོན་པའི་ལིང་ལ། ཤ་ཡི་ཊཱི། ཤ་ཡི་ཊཱིཡཱནྟཾ། ཤ་ཡི་ཊཱི་རན། སོགས་སོ། །ལྀཊ་ལའང་ཨིཊ་དང་ཡོན་ཏན་སོགས་ཡོན་ཏན་བྱས་་་པས། ཤ་ཡི་ཉྩ་ཏེ། ཤ་ཡི་བྱེ་ཏེ། ཤ་ཡི་ཉྩནྟེ། སོགས་སོ། །དེ་བཞིན་དུ་ལྡིང་ལ། ཨ་ཤ་ཡི་ཉྩ་ཏ། ཨ་ཤ་ཡི་བྱེ་ཏྭཾ། ཨ་ཤ་ཡི་ཉྩན། སོགས་སོ། ་། ཏེ། ཊ་ཏོག་༈ཨིང་ཚིག་འདོན་པ་ལའོ་ཞེས་པ་ལའང་ཏང་རྣམས་སྦྱོར་ཏེ། འགྲེལ་པར་ཨིང་གི་སྔོན་དུ་ཉེར་བསྒྱུར་ཨ་དྷི་རྟག་ཏུ་སྦྱོར་དགོས་ཏེ། དོན་གཞན་དུ་འགྱུར་་་བའང་མ་ཡིན་པས། ཨིང་གི་དོན་ཁོ་ནར་གནས་སོ། །ཞེས་བཤད་པས། །ཨ་དྷི་སྔོན་དུ་སྦྱར། ཨཙ་དང་པོ་ཅན་རྣམས་ལ་ཨཙ་ལ་ཟླ་བྱིངས་སོགས་ཀྱིས་ཨི་ཡར་བསྒྱུར་བ་སོགས་བྱས་པས། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷཱི་ཏེ། ཨ་དྷཱི་ཡཱ་ཏེ། ཨ་དྷཱི་ཡ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷཱི་ཏེ། ཨ་དཱི་ཡཱ་ཐེ༑། ཨ་དྷི་དྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དྷཱི་ཡེ། ཨ་དཱི་བ་ཧེ། ཨ་དྷི་མ་ཧེ། ལིང་ལ། ཨཏ་མ་ཡིན་པ་ལ་ལིང་གི་ས་དབྱི་འོས་ས་ཕྱིས། ་ཨཀ་འདྲ་བ་ལ་རིང་:པོར་གྱུར་དྷི་ན་ཕ་རོལ་དབྱི་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་་་་གསུམ་ལ། །ཨ་དྷཱི་ཏ། ཨ་དཱི་ཡཱ་ཏཱ། ཨ་དྷི་རན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷཱི་ཐཱ། ཨ་དྷཱི་ཡཱ་ཐཱ། ཨ་དྷི་ཧཱུཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། »ཨ་དྷི་ས། ཨ་དྷཱི་བ་ཧི། ཨ་དཱི་མ་ཧི། ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷཱི་ཏཱཾ། ཨ་།
