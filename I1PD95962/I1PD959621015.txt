སྒྲ་མདོ། ཏི་ཁ་ཏི་ནི་་་་་་གང་ནས་བློ་ལྡན་ལ་ལཊ་ལྟར་བསྒྲུབས་པས། དཾ་ད་ཤཱི་ཏུ། དཾ་དཾ་། དཾ་ད་བྷཱཏ། དཾ་ད་དྡྷཾ། དཾ་ད་ཤ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དཾ་དནྟི། དཾ་དཥྛ་ཏ། དཾ་དཾ་ཐཾ། དཾ་དཾ་། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དཾ་དཾ་ཤཱ་ནི། དཾ་དཾ་ཤཱ་བ། དཾ་དཾ་ཤཱ་མ། དེ་བཞིན་དུ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དཾ་ད་ཤཱིཏ། ཨ་དཾ་དན། ཨ་དཾ་དཏྟཾ། ཨ་དཾ་ད་ཤུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་དཾ་ད་ཤཱི། ཨ་དཾ་དན། ཨ་དཾ་དབྦཾ། ཨ་དཾ་དཥྛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དཾ་དཾ་ཧཱུཾ། ཨ་དཾ་ད་ཤཱ། ཨ་དཾ་ད༅། །ལ་ཡིག་ཕྱི་མ་རྣམས་ལ། ལུང་…ཨ་དཾ་དཾ་ཤཱི་ཏ། ལིཊ་དཾ་དཾ་ཧཱུཾ་ཙ་ཀཱ་ར། ལུཊ་དཾ་དཾ་བྷིཏཱ། སྨོན་ལིང་དཾ་ད་ཤཱཏ། ཝིཊ་དཾ་དཾ་ཤི་བྱ་ཏི། ་ཕྲེང་ཨ་དཾ་དཾ་ཤི་ཥྛཏ། སོགས་སོ། །༈ཉི་ཥ་པ་ཉལ་བ་ལའོ་ཞེས་པ་ལ་ཡང་སྦྱར། སཱ་པི་སྐྱ་མི་དག་གི་ཡང་ལ་འོས་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར། །གཉིས་བརྗོད། ?ཡང་འཇིག་པ་སོགས་ཀྱིས་ཡོན་ཏན༑ཡང་གི་ཨ་ཙ་ལ་འང་ངོས་ཡང་ཕྱིས་པས། སོ་ཥཱཔ་ཅེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨཱིཊ་ཡང་ན་བར་བྱས་་་་པས། སོ་ཥཱ་པཱི་ཏི། སོ་ཏོ། གཉིས་ཚིག་ལ། སོ་ཥཱ་ཏཿ། མང་ཚིག་ལ་ཛྷ་ཨཏ་དུ་བསྒྱུར་བས། སོ་ཥཱ་པ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སོ་ཥཱ་པཱི་ཎི། །སོ་ཥོ་སྨི། ་སོ་ཥཱཔྥ། སོ་:བླ། :：སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སོ་ཥཱ་པཱི་མི། སོ་ཥོ། སོ་ཥཱཔྤ། །སོ་ཥཱ། ལིང་ལ། སོ་ཥཱ་པཱཏ། སོ་ཥཱ་པཱཏྭཾ། སོ་ཥཱ་པཱུ། སོགས་སོ། %། ལོཊ་ལ་ལཊ་ལྟར་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། སོ་ཥཱ་པཱི་ཏུ། སོ་ཥཱy། སོཥཱ་པཱཏ། སོ་ཥཱཙཱ། སོ་ཥཱ་པ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སོ་ཥཱབྷི། སོ་ཥཔཱཏ། །སོ་ཥཱཏྟཾ། སོ་ཥཱཏ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ་་་་སོ་ཥཱ་པཱ་ནི། སོ་—།
