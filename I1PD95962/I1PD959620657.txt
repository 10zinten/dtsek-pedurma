སྒྲ་མདོ། ཏི༽གྱུར་བྱིན་ནི་སྐྱོ་ག་ས་སྐྱpཀུ་ཋབློ་ལ་གསུམ་ལ། བྷེ་ཛེ། བྷེ་ཛཱ་ཏེ། བྷེ་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྷེ་ཛི་ཏེ། བྷེ་ཛཱ་ཐེ། བྷེ་ཛི་དེ། བྷེ་ཛི་ཌྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བྷེ་ཛེ། བྷེ་ཛི་བ་ཧེ། བྷེ་ཛི་མ་ཧེ། ལུཊ་ལ། ཙུའི་ཀུ་འོས། ཛ་ག། ཁར་ལ་ཙར་འོས༑ག་ཀར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་ལ། ་བྷ། བྷཀྵ་སི་སོགས་དང་། ཏང་ལ། བྷཀྵ། ་བྷཀྵ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ་བྷ་ཛྷཱཏ། བྷ་ཛཱཙཱ། བྷ་ཙཱ་སིུསོགས་དང༌། ཏང་ལ། ཙུའི་ཀུ་འོས། ཛ་ག། ཁར་ལ་ཙར་འོས། །ག་ཀ། །ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། བྷཀྵི། བྷཀྵི་ཡཱནྟཾ། །བྷཀྵི་རན་སོགས་སོ། །ཝིཊ་ལའང་བཤད་མ་ཐག་པའི་ཆོ་ག་དང་མཐུན་པར་བསྒྲུབས་པས། ཨ་ཏང་ལ། བྷཀྵ་ཏི་སོགས་དང་། །ཏང་ལ། །བྷཀྵ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་བྷགྱུཏ་སོགས་དང་། །ཏང་ལ། །ཨ་བྷག་ཏ་སོགས་སོ། ་། ༈ཤིཉ་བསྟེན་པ་ལའོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་པས་ལཊ་་་སོགས་པ་རསྨི་པ་ད་དང་ཨཱཏྨ་ནེ་པ་ད་གཉིས་ཀ་འཇུག་སྟེ། ལཊའི་ཨ་ཏང་ལ། ཤ་ཡ་ཏི་སོགས་དང༌། ཏང་ལ། ་ཤ་ཡ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ལིང་གི་ཨ་ཏང་ལ། ཤ་ཡེཏ་སོགས་དང༌། ཏང་ལ། ཐ་ཡེ་ཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། ཤ་ཡ་ཏུ་སོགས་དང༌། ཏང་ལ། ཤ་ཡ་ཏྭཾ་སོགས་སོ། ལང་གི་ཨ་ཏང་ལ། ཨ་ཤ་ཡཏ་སོགས་དང་། ཏང་ལ། ཨ་ཤ་ཡཏ་སོགས་སོ། །ལུང་ལ། ཎི་ཤྲཱི་དྲུ་སོགས་ཀྱིས་ཙི་ཙང་དུ་བསྒྱུར། །་ཙང་ལ་འོས། བྱིངས་གཉིས་བརྗོད། དང་པོའི་ཧལ་ལྷག་པོ་འོས། ར་ཕྱིས། ཨ་ཙ་ལ་བྱིངས་་་་་སོགས་ཀྱི་ཨི་ཡ་སོགས་ཀྱིས་བསྒྲུབས་པས། །ཨ་ཏང་ལ། ཨ་ཤི་ཤི་ཡ་ཏ། ཨ་་ཤི་ཤི་ཡ་ཏྭཾ། ཨ་ཤི་ཤི་ཡན་སོགས་དང་། ཏང་ལ། །ཨ་ཤི་ཤི་ཡཏ། ཨ་ཤི་ཤི་།
