་བར་ཉི་ཟླ་པཎ་ཊོག་ཀྱ་བ་ཚོ་བསྟན་འགྱུར། སྒྲ་མདོ། ༼''་་ན་མོ་བའི་ཚེ་ཀ་དམདའ་ཀར་བློ་ཨོ་ལ། ཨ་ཛ་རེ། །ཨ་ཛ་ར་སཱི། ཛས་ལ་ཨ་ཛ་རས་སུ་མི་འགྱུར་བའི་ཕྱོགས་ལ། ཨ་ཛ་རཱ་ཎི། ལུགས་གཅིག་ལ། །ཨ་ཛ་ར་:ན་སི། ཨི་ཞེས་གནས་པ་ལ། །སིཾ་ཏ་མ་ཧ་ཏསཾ་ཡོ་ག་སྱ། ས་མཐའ་མ་ཧཏ་དག་གི་ཡང་དག་པར་སྦྱོར་བའིའོ། །ས་མཐའ་ན་དང་ཡང་དག་སྦྱོར་བ་སྟ་བརྩེགས་པ་དང་མ་ཧཏ་ལ་ནུམ་བྱིན་པའི་ན་ལྟ་བུའི་ཨུ་པ་དྷཱ་ནི། སཾ་བུདྡྷི་མ་ཡིན་པའི་རྣམ་དབྱེ་ཐམས་ཅད་གནས་པའི་མིང་་ཅན་རྣམས་ལ་རིང་པོར་འགྱུར་རོ། །ཨ་ཛ་རཱནི། སཾ་བུདྷི་ལ། །ཧེ་ཨ་ཛ་ར། ཧེ་ཨ་ཛ་རེ། ཧེ་ཨ་ཛ་སཱི། ཧེ་ཨ་ཛ་རཱ་ཎི། ཧེ་ཨ་ཛ་རཱནྟི། གཉིས་པ་རྣམས་དང་པོ་དང་འདྲ། གསུམ་པ་རྣམས་ལ། ཨ་ཛ་རེ་ཎ། ཨ་ཛ་ར་སཱ། ཨ་ཛ་རཱ་ཧཱུཾ། ཨ་ཛ་རི། །བཞི་པ་ལ། ཨ་ཛ་རཱ་ཡ། ཨ་ཛ་ར་སེ། ཨ་ཛ་རཱ་བྷཾ། ཨ་ཛ་རེ་བྷྱ། །ལྔ་པ་ལ། །ཨ་ཛ་རཱཏེ། ཨ་ཛ་ར་སམྦྷི། བྷཾ་བྱཱའདྲ། དྲུག་པ་ལ། ཨ་ཛ་ར་སྐྱ། ཨ་ཛ་ར་ས། ཨ་ཛ་ར་ཡོ། ཨ་ཛ་ར་སོ། །ཨ་ཛ་རཱ་ཎཾ། ཨ་ཛ་ར་སཾ། ་བདུན་པ་ལ། ཨ་ཛ་རེ། །ཨ་ཛ་ར་སི། ཨ་ཛ་ར་ཡོ༑་ཨ་ཛ་ར་སོ། །ཨ་ཛ་ར་ཥཱ། ཧྲ་ད་ཡ་ལ། སྙིང་། རྣམ་དབྱེ་དང་པོ་རྣམས་དང་བོད་ཚིག་རྣམས་དང་། རྣམ་དབྱེ་གཉིས་པའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་རྣམས་ཛྙཱ་ནའི་སྐབས་དང་འདྲ་སྟེ། ཧྲི་ད་ཡཾ། ཧྲི་ད་ཡེ། ཀྲྀ་ད་ཡ་ནི། ཧེ་ཧྲི་ད་ཡ། ཧེ་ྋཧྲི་ད་ཡེ། ཧེ་ཧྲི་ད་:ཡཱ་ནཱི། ཧྲི་ད་ཡཾ། ཀྲྀ་ད་ཡེ། གྲོགས་ད་ཡ་ལ། ་ཤས། ཤས་སོགས་རྣམས་ལ་ཞེས་པའི་མདོས་ཧྲི་ད་ཡ། ཧྲིད་དུ་་་འགྱུར༑ཛས་ཤས་དག་གི་ཤི། །མ་ནིང་གི་ཛྷལ་ཨཙའི་འོས་ནུམ། ཧཱིནྡི། ལུགས་གཅིག། ཀྲྀ་ད་ཡ་ནི། །ཊཱ་ལ། །ཏཱཾ་དཱ། ཧྲི་ད་ཡེ་ན། །བྱཱ་ལ། ཧྲིདྡྷཾ། ཧྲི་ད་ཡ་བྷཾ། །བྷིལ། ཧྲིདྡྷི། ཏྲཾ་ད་ཡི། དེ་ལ། །ཧཱུྃ་དེ། ཧྲི་ད་ཡཱ་ཡ། ཧཱུཾ་འདྲ། །སྦྲུལ། ཧྲིདྡྷ། ཧྲི་ད་ཡེ་ཤྩ། ང་སི་ལ། ཀྲྀ་ད། —།
