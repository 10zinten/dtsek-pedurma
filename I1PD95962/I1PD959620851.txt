སྒྲ་མདོ། དེའི་ཨ་སྤུ་ར་ཏ། །སོགས་སོ། །སྤུ་ར་ཏི་སྤུ་ལ་ཁྱོ་ནི་ནི་བི་བྷུ། ནིར་ནི་བི་རྣམས་་ལས་སྤུ་ར་ཏི་སྤུ་ལ་ཏི་དག་གིའོ། །ཉེར་བསྒྱུར་ནི་ར་དང་ནི་དང་བི་རྣམས་ལས་ཕ་རོལ་གྱི་སྤུ་ར་ཏི་དང་སྤུ་ལ་ཏི་དག་གི་ས་ཡིག་ནི་ཥ་ཡིག་ཏུ་ཡང་ན་བསྒྱུར་ཏེ། ནིསྤུ་ར་ཏི། ནིམྱུར་ཏི། ནི་ཀླུ་ར་ཏི། ནིམྦྷ་ར་ཏི། བི་ཧྥུ་ར་ཏི། བིནྡུ་ར་ཏི། ནིཀླུ་ལ་ཏི། ནི་ྩ་ལ་ཏི། སོགས་སོ། །ཏི་ལྔའི་བར་ཊིའི། ཁཊ་ཅཡོ་ཞུ་༈སྤུ་ལ་ཀུན་ནས་སོགས་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། སྨོན་པའི་ལིང་གི་སྐབས་སུ་རིང་པོ་མི་བྱེད་པ་ཙམ་མ་གཏོགས་ཆོ་ག་ཐམས་ཅད་སྤུ་ར་དང་འདྲ་སྟེ། སྤུ་ལ་ཏི། སོགས་སོ། །ཏི། པ་པ། །ཊ་ཏི་ལ་ཁོ་ལ་ཞ་བའི་༈རྟུ་བསྟོད་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ཎའི་ན་འོས་ཎ་ན། ཨ་ཙ་ལ་ཐཱུ་བྱིངས་སོགས་ཀྱིས་ཨུ་བར་བསྒྱུར་བས། ལཊ་ལ། ནུ་བ་ཏི༑སོགས་དང༌། ལིང་ལ། ནུ་བེ་ཏ། སོགས་སོ། །ལོཊ་ལ། ནུ་བ་ཏུ། །སོགས་དང་། ལང་ལ། ཨ་ནུ་བ་ཏ། །སོགས་སོ། །ལུང་ལ། ཨིཊ་བྱིན། །གཱང་ཀུཊ་སོགས་ཀྱིས་ཡོན་ཏན་དང་འཕེལ་བ་སོགས་མི་བྱེད། /ཨ་ཙ་ལ་སྣེ་བྱིངས་སོགས་ཀྱིས་ཨུ་བར་བསྒྱུར་བས། ཨ་ནུ་བཱི་ཏ། ཨ་ནུ་བིཏྟཾ། །ཨ་ནུ་བི་ཥཱ། སོགས་སོ། །།ལིཊ་ལ། གཉིས་་་བརྗོད་ཎལ་གཉིས་ལ་འཕེལ་བ་དང་། གཞན་རྣམས་ལ་ཨུ་བར་བསྒྱུར་བས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ནུ་ནཱ་བི། ནུ་ནུ་བ་ཏུ། ནུ་ནུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ནུ་ནུ་བི་ཐ། ནུ་ནུ་བ་ཐུ། ནུ་ནུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ནུ་ནཱ་བ། ནུ་ན་བ། ནུ་ནུ་བི་བ། ནུ་ནུ་བི་མ། །ལུཊ་ལ་ཨིཊ་བྱིན། གཱང་ཀུཊ་སོགས་ཀྱིས་ཡོན་ཏན་མི་བྱེད། ཨཙ་ལ་ལྷ་བྱིངས་སོགས་་་་ཀྱིས་ཨུ་བར་བསྒྱུར་བས། ནུ་བི་ཏཱ། ནུ་བི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་mds།
