བའི་ལྷ་ཁྲིད་ཨ་སྐྱོ་བ། བསྟན་འགྱུར། །སྒྲ་མདོ། *ཏིམེདི་་ནི་་སྐུ་ཉི་ཟླ་ཚོར་ཎ/ལས་nབི་པཱཀྵི། བྲམ་ཟེའི་མིག། ཨ་ཙ་ཏུ་ར་བི་ཙ་ཏུ་ར་སུ་ཙ་ཏུ་ར་སི་པུ་ས་དྷེ་ནཱ་ན་ཌཱ་ཧ་་་རཱིཀ་མ་བཱདྨ་ན་སཱགཱི་བྷུ་བ་དཱ་ར་ག་བོརྦཊྛི་བ་པ་དཎྜི་བ་ནཏྟཾ་དི་བ་རཱ་ཏཱིནྡི་བཱ་ཧཇི་བ་ས་ར་ཛ་ས་ནི་ཐེ་ཡས་པུ་རུ་ཥཱ་ཡུ་ཥ་ཏུ་ཡུ་ཥ་ཏྱཱ་ཡུ་ཥ་རོགླ་ཛུ་ཥ་ཛཱ་ཏོ་གླུ་མ་ཧོཀྵ་བྲིདྡྷོཀློ་པ་ཤུ་ན་གོཥྛ་ཤཱ། ཨཙ་ཏུ་ར་ཡཨཙ། བི་ཙ་ཏུ་ར་ཡཨཙ། སུ་ཙ་ཏུ་ར་ཡཨཙ། ཝཱི་པཾ་ས་ཨཙ། དྷེ་ནཱ་ན་ཌཱ་ཧ་…ཨཙ། །རཾཀ་སཱ་མ་ཨཙ། ཝཱདྨ་ན་ས་ཡཨཙ། ཨཀྟི་བྷུ་བ་ཨཙ། དྨ་ར་ག་པ། ཨཙ། ཨཱུརྦཊྛི་བ་ཨཙ། པ་དཊྛི་བ་ནཀྟ་དི་བ་ཡཨཙ། རཱ་ཏྲི་དི་བ་ཡཨཙ། ཨ་ཧཇི་བ་ཨཙ། ས་ར་ཛ་ས་ཡཨཙ། ནི་ཀྲེ་ཡ་ས་ཡཨཙ། །པུ་རུ་ཥཱ་ཡུ་ཥ་ཨཙ། དཱུ་ཡུ་ཥཡཨཙ། ཏྟཱ་ཡུ་ཥཡཨཙ། རྀགླ་ཛུ་ཥཨཙ། ཛ་ཏོཀྵ་ཡཾཨཙ། མ་ཧོ་ཡཨཙ། བྲིདྡྷོཀ་ཨཙ། ཨུ་པ་ཤུན་ཡཨཙགོཏྟཱ་ཤྭ་རྣམས་སོ། །ཨ་ཙ་ཏུ་ར་ལ་སོགས་པ་ཉི་ཤུ་རྩ་ལྔ་པོ་དེ་རྣམས་བསྡུ་བའི་མཐའ་ལ་ཨཙ་ཚིག་ཕྲད་ཀྱི་བྱིན་པ་སྟེ། ཨ་པ་ག་ཏཱ་ནི་ཙ་ཏཱ་རི་ཨ་སྱ། བཞི་སོང་བ་འདི། ཞེས་པ་ལ། ཨ་པ་ག་ཏའི་དོན་ཨ་་་སྔོན་རྣམ་དབྱེ་སོགས་ཕྱིས། ཨཙ་བྱིན། ཙ་རྗེས་འབྲེལ། ཨ་ཙ་ཏུ་རཿ། བཞི་་་མེད་དམ་སོང་། དེ་བཞིན་དུ། བི་ག་ཏཱ་ནི་ཙ་ཏྭཱ་རི་ཨ་སྱ། །ཞེས་པ་ལ། བི་ཙ་ཏུ་རཿ༑བཞི་བྲལ། སཥྛ་ཙ་ཏཱ་རི་ཡ་སྱ། བཟང་པོ་བཞི་གང་གི། ཞེས་པ་ལ། སུ་ཙ་ཏུ་རཿ༑ཡལེགས་པ་བཞི། དེ་གསུམ་པོ་འབྲུ་མང་པོའི་བསྡུ་བའི་མཐའ་ལ་ཨཙ་བྱིན་པའོ། །དེ་བཞིན་དུ། སི་ཙ་པཾ་མཱ་ནཉྩ། བུད་མེད་དང་སྐྱེས་པ་ཡང༌། ཞེས་པ་ལ། སི་པཾ་སོ། བུད་མེད་སྐྱེས་པ་དག། དྷེ་ནུཉྩ་ཨ་ན་ཌཱ་ཙ། འདོད་འཇོ་དང་གླང་ཡང་། ཞེས་པ་ལ། ་དྷེ་ནཱ་ན་ཌཱ་ཧོ༑བ་གླང་དག། རཾཀ་ཙ་སཱ་མ། སྙན་ཚིག་དང་ཚིག་རྐང་ཡང་། ཞེས་པ་ལ། རོཀ་སཱ་མེ༑སྙན་ཚིག་ཚིག་རྐང༌། ཝཱ་ཀ་ཙ་མ་ན༄། ངག་དང་ཡིད་ཀྱང་། ཞེས་པ་ལ། ཝཱདྨ་ན་སེ། ངག་ཡིད་དག། ཨཀྵི་ཎཱི་ཙ་བྷཱ་བོ་ཙ། མིག་དག་དང་ཟེ་མ་དག་དང་། ཞེས་པ་ལ། །ཨཀཱི་བྷུ་བོ༑ཡམིག་ཟེ་མ། དཱ་ར༔་གོ། བ་དང་གླང་དང༌། ཞེས་པ་ལ། དཱ་ར་ག་པཾ། བ་གླང༌། —།
