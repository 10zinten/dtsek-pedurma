སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། །ཐེ་གྲུ་ཏི། སོགས་དང༌། དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་གླེ་ཥྛཏ། །སོགས་སོ། །་་་་་༈ར་དྷ་འཚེ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ལཊ་ལ། ར་དྷ་ཏི། སོགས་དང་། ལིང་ལ། ར་དེ་ཏ། སོགས་སོ། །ལོཊ་ལ། ར་དྷ་ཏུ། །སོགས་དང༌། ལང་ལ། །ཨ་ར་དྷ་ཏ། །སོགས་སོ། །ལུང་ལ། པུཀ་མཐའ་སོགས་ཀྱིས་ལྕི་ཨང་དུ་བསྒྱུར། ར་དྷི་ཛ་བྷ་དག་གི་ཨཙ་ལ་འོས། ནུམ་ཨཱ་ག་མ་བྱིན་པས། ཨ་རནྡྷ་ཏ། །ཨ་རནྣ་ཏྭཾ། ཨ་རནྟན། སོགས་སོ། །ལིཊ་ལའང་གཉིས་བརྗོད། ་ར་དྷི་ཛ་བྷ་དག་གི་ཨཙ་ལ་འོས་ནུམ་བྱིན་པས། ར་རནྟ། ར་རནྟ་ཏུ། ར་རནྡྷ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ར་རནྟི་ཐ། ར་རནྟ་ཐུ། ་་ར་རནྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ར་རནྟ། ར་རནྟི་བ། །་ར་རནྟི་མ། ར་དྷཱ་དི་བྷུཉྩ། ར་དྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོ༌། །།ར་དྷ་ལ་སོགས་པ་སྟེ། ར་དྷ་འཚེ་བ། །ཎ་ཤ་མི་མཐོང་བ། ཀྲྀ་པ་ཚིམ་པ། དྲྀཔ་དགའ་བ། །དྲུ་ཧ་འཚེ་བ་བྱེད་དུ་འཇུག་པར་འདོད་པ། མུ་ཧ་ཐེ་ཚོམ། ་བྷུ་ཧ་སྐྱུག་པ། ལྕི་ཧ་དགའ་བ་རྣམས་ལས་ཕ་རོལ་གྱི་བྱིངས་ཕྱེད་པ་བལ་དང་པོ་་་་ཅན་ལ་ཨིཊ་ཡང་ན་འགྱུར་རོ། །ནེ་ཁྱུ་ལི་ཊི་ར་དྷེ། ར་དྷིའི་ལིཊ་མ་ཡིན་པའི་ཨིཊ་ལ་མ་ཡིན་ནོ། །གོང་དུ་ར་དྷི་དང་ཛ་བྷ་དག་ལ་ཨ་ཙ་ཕ་རོལ་ལ་ནུམ་ཨཱ་ག་མ་སྟེར་བར་བཤད་ཀྱང་དེའི་དམིགས་བསལ་ཨཙ་དང་པོ་ཡིན་ཀྱང་ལིཊ་ཕ་རོལ་གྱི་ཨིཊ་མ་ཡིན་པས། ཨིཊ་ཕ་རོལ་ལ་ར་དྷི་ལ་ནུམ་འགྱུར་བ་མ་ཡིན་ཏེ། ར་དྷ་ལ། ལུཊ་སྦྱར་བས། ར་དྷ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ངོས་ཨིཊ་ཡང་ན་བར་བྱས། མདོ་འདིས་ནུམ་ཨཱ་ག་མ་མི་སྟེར། ཨིཊའི་ཕྱོགས་ལ། །ར་དྷི་ཏཱ། ར་དྷི་ཏཱ་སི། སོགས་སོ། །ཨིཊ་མེད་པའི་ཕྱོགས་ལ། ་ཛྷཥ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་—།
