བ་ལུ་དང་། བ་::བསྟན་འགྱུར། སྒྲ་མདོ། ༼ཏིཁ་ཐོཎི་རོ་ལ་ཁོ་ཚnཙོ་ཎ་ལ་སྨRབྷཱhIམཧུ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དུ་དོ་ཧི་ཐ། དུ་དོགྷ། དུ་དུ་ཧ་ཐུ། དུ་དུ་ཧ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་དོ་ཧ། དུ་དུ་ཧི་བ། དུ་དུ་ཧི་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དུ་དུ་ཧེ། དུ་དུ་ཧཱ་ཏེ། དུ་དུ་ཧི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་དུ་དུ་ཧི་ཥ། །དུ་དུ་ཧཱ་ཐཱ། དུ་དུ་ཧི་དེ། དུ་དུ་ཧིནྡེ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་དུ་ཧེ། དུ་དུ་ཧི་བ་ཧེ། དུ་དུ་ཧི་མ་ཧེ། ལུཊ་ལ། ད་དང་པོའི་གྷ་འོས་ཧ་གྷ། ཛྷཥ་ལས་དྷུ་མ་ཡིན་པ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་གྷ་གར་བསྒྱུར་བས། ཨ་ཏང་ལ། དོདྡྷེ། དོགྷཱ་སི། སོགས་དང་། ཏང་ལ། དོངྷ། དོགྷཱ་སེ། །སོགས་སོ། སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། དུ་ཧཱུཏེ། དུ་ཧནྟཾ། དུ་ཧ་སུ། སོགས་དང༌། ཏང་ལ། ལིང་སིཙ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ་འོས་ཡོན་ཏན་མི་བྱེད། །ད་དང་་་པོའི་གྷ་འོས་ཧ་གྷ། ཁར་ལ་ཙར་འོས་གྷ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། དུགཱི་ཏ། :དུགཱི་ཡཱནྟཾ། །དུགཱི་རན། སོགས་སོ༑། ལྀཊ་ལའང་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན། ཧ་གྷ། །ཁར་ལ་ཙར། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། ཨ་ཏང་ལ། དོཀྵ་ཏི། ་དོགྲུ་ཏ། དོ་གྱུནྟི། སོགས་དང༌། །ཏང་ལ། དོཀྵ་ཏེ། དོཀྵེ་ཏེ། དོཀྵེནྟེ། །སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་གི་ཨ་ཏང་ལ། ཨ་དོགྱུཏ། ཨ་དོགྲུ་ཏཱཾ། ཨ་དོpན། སོགས་དང་། ཏང་ལ། ་ཨ་དོགྱ་ཏ། ཨ་དོགཱི་ཏཱཾ། ཨ་དོ་གྱུནྟ། སོགས་སོ། །དང་བ་འི་nSཆ་ཏུ་ད་༈དི་ཧ་ཉེ་བར་གསོག་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་་་འཇུག་ཅིང་ཡོན་ཏན་གྱི་སྐབས་ཨོ་དང་ཨེའི་ཁྱད་པར་ཙམ་མ་གཏོགས་ཆོ་ག་དང་་་སྒྲུབ་གཟུགས་ཐམས་ཅད་བཤད་མ་ཐག་པ་དུ་ཧའི་སྐབས་དང་འདྲ་སྟེ། དེབྷི། —།
