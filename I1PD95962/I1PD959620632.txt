དག་ལ་འོས་བྱིངས་གཉིས་བརྗོད། །སན་ལ་ཨཏ་ཀྱི་འོས་ཨ་ཨི། མདོ་འདིས་་་རིང་པོ། ཨཙ་གཅིག་པའི་བ་ཤའི་བྷཥ། ཛྷཥ། མཐའི་ས་དྷ་དག་ལ་འོས་པ་འོག་མ་བྷ། །ཁར་ལ་ཙར་འོས། །དྷ་ཏ་རུ་བསྒྱུར་བས། བཱི་བྷཏྠ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས་ལཊ་ལ། བཱི་བྷཏྠ་ཏེ་སོགས་སོ། །ལིང་ལ། བཱི་བྷ་ཏྲེ་ཏ་སོགས་སོ། །ལོཊ་ལ། བཱི་བྷཏྠ་ཏྭཾ་སོགས་སོ། །ལུང་ལ། ཝལ་དང་པོའི་ཨིཊ་དང་ཨཏ་ཀྱི་དབྱི་འོས། ཁོ་ཨ་ཕྱིས་པས། ཨ་བཱི་བྷཏིཏྠ་སོགས་སོ། །ལིཊ་ལ། བཱི་བྷནཾ་ཙཀྲེ། བཱི་བྷནྡྷཾ་བ་བྷུ་བ། བཱི་བྷཏྟཱ་མཱ་ས་སོགས་སོ། །ལུཊ་ལ། བཱི་་་བྷཏི་ཏཱ་སོགས་སོ།  ོ། སྨོན་པའི་ལིང་ལ། བཱི་བྷཏི་ཊཱིཋ་སོགས་སོ། །ལྀཊ་ལ། བཱི་བྷཏྟི་བྱེ་ཏ་སོགས་སོ། །།ཁྲིང་ལ། ཨ་བཱི་བྷཏི་ཥྛ་ཏ་སོགས་སོ། །།།བ། ༈དྷུ་ཏ་འབར་བ་ལ་འོ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་ལ། བྱོ་་ཏ་ཏེ་སོགས་སོ། །ལིང་ལ། བྱོ་ཏེ་ཏ་སོགས་སོ། །ལོཊ་ལ། བྱོ་ཏ་ཏྭཾ་སོགས་སོ། །ལང་ལ། ཨ་བྱོ་ཏ་ཏ་སོགས་སོ། །ཧཱུདྡྷོ་ལུ་དི། དྷཱུཏ་རྣམས་ལས་ལུང་ལའོ༑། དྱ་ཏ་ལ་སོགས་པ་རྣམས་ལ་ལུང་གི་པ་རསྨི་པ་ད་རྣམས་ཡང་ན་འགྱུར་ཏེ། པ་རསྨིའི་ཕྱོགས་ལ། ༼པུ་ཥ་སོགས་་ཏེ་སོགས་ཞེས་པའི་མདོས་ལྕི་ཨང་དུ་བསྒྱུར་བས༑ཨ་དྷུ་ཏཏ། ཨ་དྷུ་ཏ་ཏྭཾ། ཨ་དྷུ་ཏན་སོགས་སོ། །ཨཱཏྨ་ནེ་པ་དའི་ཕྱོགས་ལ། ཙྪི་སི་ཙར་བསྒྱུར། ཝལ་དང་པོའི་ཨིཊ་སྦྱར་བས། ཨ་བྱོ་ཏིཥྛ། ཨ་ཅོ་ཏི་ཥཱ'ཏྭཾཀྱེ་ཨ་ངོ་ཏི་ཥ་ཏ་སོགས་སོ། །དྷཱུ་ཏི་སཱཔྱོ། སཾ་པྲ་སཱ་ར་ཎཾ། དྷུ་ཏི་སཱ་པི་དག་གི་སཾ་པྲ་ས་ར་ཎའོ། །དྷུ་ཏ་འབར་བ། ཉི་ཥ་བ་ཉལ་བ་དག་གི་ལིཊ་ལ་བྱིངས་གཉིས་བརྗོད་པའི་དང་པོ་ཨ་བྱཱ་སའི་མིང་ཅན་དེ་སཾ་པྲ་སཱ་ར་ཎ་སྟེ་་་སླར་ཡང་དག་པར་བསྒྱུར་བར་བྱ་སྟེ། །དི་བྱུ་ཏེ། ཏི་ཏྱཱ་ཏཱ་ཏེ། དི་དྷཱུ་ཏི་རེ། དི་དྷུ་ཏི་ཥ། སོགས་སོ། །།ལུཊ་ལ། ཨིཊ་སྦྱར་བས། དྲོ་ཏི་ཏཱ་སོགས་སོ། །D—།
