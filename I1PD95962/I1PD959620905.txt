དང་ད་ལིང་ལ། ག་ཎ་ཡེཏ། །སོགས་སོ། །ལོཊ་ལ། ག་ཎ་ཡ་ཏུ། །སོགས་དང༌། ལང་ལ། ་ཨ་ག་ཎ་ཡཏ། སོགས་སོ། །ལུང་ལ། །ཨི་ཙ་ག་ཎ། ག་ཎའི་ཨཱི་འང་ངོ༌། །ཎི་ཙང་དང་བཅས་པ་ཕ་རོལ་ལ་ག་ཎའི་ཨ་བྷྱཱ་སའི་ཨཱི་ཡང་ན་སྟེ། ཎི་ཤི་སོགས་ཀྱིས་ལྕི་ཙང་དུ་བསྒྱུར། ཙང་ལ་འོས་བྱིངས་གཉིས་བརྗོད། ཀུ་ཧ་དག་གི་ཙུ་འོས། ག་ཛ་མདོ་འདིས་ཨཱིར་ཡང་ན་བསྒྱུར་ཎིའི་ཨིཊ་མེད་པ་ལ་འོས། ཎིཙ་ཕྱིས་པས། ཛཱི་ག་ཎཏ། ཛཱི་ག་ཎ་ཏྭཾ། ཛཱི་ག་ཎན། ་སོགས་སམ། ཛ་ག་ཎཏ། སྐབས་འདིར་ཨཀ་ཕྱིས་པ་ཡིན་པས་ཎི་ཙང་ལ། སན་བཞིན་གྱི་བྱ་བ་མི་འཇུག་གོ། །སོགས་སོ། །ལིཊ་ལ། ཨཱམ་སྦྱར་བས། ག་ཎ་ཡཱ་ཙ་ཀཱ་ར། །ག་ཎ་ཡཾ་བ་བྷུ་བ། ག་ཎ་ཡཱ་མཱ་ས། སོགས་སོ། །ལུཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ག་ཎ་ཡི་ཏཱ། ག་:ཎ་ཡི་ཙིཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཎིའི་ཨིཊ་མེད་པ་ལ་འོས་ཎིཙ་ཕྱིས་པས། ག་ཙྪཱ་ཏ། ག་ཙྪཱཙཱ། ག་ཙྪཱ་སུ། སོགས་སོ། །ཁྲིཊ་ལ། ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། ག་ཎ་ཡི་ཥྱ་ཏི། སོགས་དང༌། ཁྲིང་ལ། ?ཨ་ག་ཎ་ཡི་ཏ། སོགས་སོ། །།།ཏུ་ཉན་ཕྱི་༈ཀ་ཐ་དག་གི་རྒྱུན་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས། ཨ་ཕྱིས་པའི་རྒྱུ་མཚན་གྱིས་འཕེལ་བ་མི་བྱེད། ཤཔ་ལ་ཡོན་ཏན་བྱས་པས། ལཊ་ལ། ཀ་ཐ་ཡ་ཏི། །སོགས་དང༌། ལིང་ལ། ཀ་ཐ་་ཡེཏ༑སོགས་སོ། །ལོཊ་ལ། ཀ་ཐ་ཡ་ཏུ། སོགས་དང༌། །ལང་ལ། ཨ་ཀ་ཐ་ཡཏ། སོགས་སོ། །ལུང་ལ་འགྲེལ་པར་ལུང་དྲངས་པ། ཀ་ཐོཉཧྥུ་ས་སྱཱ་ཨཱི་ཏཱཾ་ཀེ་ཙིདཱ། །ཀ་ཐའི་ཨ་བྷཱུ་སའི་ཨཱི་ཉིད་འགའ་ཞིག་ཏུ་ཡང་ནའོ། །།ཎི་་་་ཙང་དག་ཕ་རོལ་ལ་ཀ་ཐའི་ཨ་ཧྥུ་སའི་ཨཱི་ཉིད་སྐབས་འགའ་ཞིག་ཏུ་ཡང་ན་བར་བྱ་སྟེ༑ཎི་ཤི་སོགས་ཀྱིས་ཙྪི་ཙང་དུ་བསྒྱུར། ཙང་ལ་འང་ངོས་བྱིངས་གཉིས་ལང་ལ། །
