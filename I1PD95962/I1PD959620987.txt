སྒྲ་མདོ། ཏིའི་སྤྱལ་བློཎི་ནི་ལ་སྐྱོ་བ་བཙོན་ཁབཀབློ་དང་ཥ་དྲི་རྣམ་པར་འགྲོ་བ། ཙ་ར་འགྲོ་བ། ཛ་པ་ཡིད། །ཛ་བྷ་ལུས་རྣམ་པར་་་ཉམས་པ། ད་ཧ་ཐལ་བར་བྱེད་པ། ད་ཤ་མཆེ་བ། གཽ་གསལ་བར་བྱེད་པ་རྣམས་ལས་སྨད་པ་ཉིད་ལ་འཇུག་པ་ལ་ཡང་འགྱུར་ཏེ་བྱ་བ་ཡང་དག་པར་བསྡུ་བ་ཙམ་ལ་ནི་མ་ཡིན་ནོ། །ཏ་ད་ན་དེབ་ཁཊིབ་ཀྱི་གཏོ་འབྱེད། ༈སཱུ་ཙ་ཕྲ་མ་བྱེད་པ་ལའོ་ཞེས་པ་ལ་ཡང་། ལྷོ་སན་ཡང་དག་ལ་འོས་གཉིས་བརྗོད། ཐུང་ངུ་འོས་ཐུང་ངུར་བསྒྱུར་བས། ་སུ་སཱུ་ཙ། །ཡང་ཞེས་པ་ལ། །གུ་ནོ་ཡང་ལུ་ཀོ། །ཡང་འཇིག་པ་དག་ལ་ཡོན་ཏན་ནོ། ་ཎ། །ཡང་གི་་རྐྱེན་འཇིག་པ་དང་མི་འཇིག་པ་དག་ཕ་རོལ་ལའང་ཨ་བྷཱ་ས་ཡོན་ཏན་དུ་བསྒྱུར་ཏེ། མདོ་འདིས་ཡོན་ཏན་བྱས་པས། །སོ་སཱུ་ཙྩ་ཞེས་གྲུབ་པ་ལ། །འ་ལཊ་སོགས་ཏང་རྣམས་སྦྱོར་ཏེ། །བྱེད་པ་པོ་ལ་ཤཔ་ཨཏ་ལས་ཡོན་ཏན་ལ་འོས་གཉིས་དག་གི་་་་གཅིག་སྔོན་མའི་གཟུགས་སུ་བསྒྱུར་བས། ལཊསོ་སཱུ་ཙ་ཏེ། །ལིང་སོ་སཱུ་ཙེ་ཏ། ལོཊསོ་སཱུ་ཙ་ཏྭཾ། ལང་ཨ་སོ་སཱུ་ཙ་ཏ། སོགས་དང༌། ལུང་ལ། ་ཡ་སྱ་ཧལ༑་ཡའི་ཧལ་ལས་སོ། །བྱིངས་མཐའ་ཧལ་ལས་ཕ་རོལ་གྱི་ཡའི་བྱིངས་་་་ཕྱེད་པ་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། མདོ་འདིས་ཡ་ཕྱིས། །བྱིངས་ཕྱེད་པའི་བལ་དང་པོ་ལ་ཨིཊ་འོས་ཨིཊ་བྱིན་པས། ཨ་སོ་སཱུ་ཙིཏྟཾ། ཨ་སོ་སཱུ་ཙི་ཥ་ཏྭཾ། །ཨ་སོ་སཱུ་ཙི་ཥ་ཏ། སོགས་སོ། ་། ལིཊ་ལ། ཀཱས་སོགས་ཀྱི་ཨཱམ་སྦྱར། ཡའི་ཧ་་་ལས་འོས་ཡ་ཕྱིས་པས། སོ་སཱུ་ཙཱ་ཙ་ཀྲེ། སོ་སཱུ་ཙཱ་བ་བྷུ་བ། སོ་སཱུ་ཙཱ་མཱ་ས། སོགས་སོ། །ལ་ཡིག་ཕྱི་མ་རྣམས་ལ་བལ་དང་པོའི་ཨིཊ་བྱིན། ་ཡའི་ཧལ་ལས་་སོས༑ཡ་ཕྱིས་པས། ལུཊསོ་སཱུ་ཙི་ཏཱ། སྨོན་ལིང་སོ་སཱུ་ཙི་ཥིཉྩ། ཝིཊསོ་སཱུ་ཙི་ཥྛ་ཏེ། ཁྲིང་ཨ་སོ་སཱུ་ཙི་ཏ། སོགས་སོ། །ལ་། ཀུ་ཊ་མ་ལ། །ཀུན་༈མཱུ་ཏྲ་རབ་ཏུ་འཛག་པ་ལ་ཡང་སྦྱར། གཉིས་བརྗོད། ཐུང་དུ་འོས་—།
