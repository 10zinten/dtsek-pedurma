ལིཊ་ལ་བྱིངས་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ་དི་དེ་བ། འདི་དི་བ་ཏུ། དི་དི་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དི་དེ་བི་ཐ། དི་དེ་བ་ཐུ། དི་དི་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དི་དེ་བ། དི་དི་བི་བ། དི་དི་བི་མ། ལུཊ་ལ། །ཨིཊ་བྱིན་པས། དེ་བི་ཏཱ། དེ་བི་ཏཱ་སི༑་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ་ཨུ་པ་དྷཱའི་འང་ངོས་རིང་པོར་བསྒྱུར་བས། དཱི་བྱཱཏ། དཱི་བྱཱ་སིཾ། དཱི་བྱཱ་སུ། །སོགས་སོ། །ལྀཊ་ལ། བལ་དང་པོའི་ཨིཊ་དང༌། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། དེ་བི་ཥྱ་ཏི། སོགས་དང༌། དེ་བཞིན་དུ། །སྲྀང་ལ། ཨ་:དེ་བི་ཏ། སོགས་སོ། །ཁཊི་༈ཥི་བུ་ཐགས་འཐག་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཛུག་སྟེ། ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་འདྲ་བས། ལཊ་ལ། སཱི་བྱ་ཏི། སོགས་དང་། །ལིང་ལ། སཱི་བྱེཏ། །སོགས་སོ། །ལོཊ་ལ། སཱི་བྱ་ཏུ། སོགས་དང་། །ལང་ལ། ཨ་སཱི་བྱཏ། སོགས་སོ། །ལུང་ལ། བལ་དང་པོའི་ཨིཊ་དང༌། །པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། /ཨ་སེ་བཱི་ཏ། ཨ་སེ་བཱིཏྟཾ། ཨ་སེ་བི་ཥཱ། སོགས་སོ། །།ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་བྱས་་་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །སི་ཥ་བ། སི་ཏི་བ་ཏུ། སི་ཥི་བུ༄༅། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ?སི་ཏི་བི་ཐ། སི་ཏི་བ་ཐུ། སི་ཊི་བི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །།།སི་ཥ་བ། སི་ཏི་བི་བ། སི་ཏི་བི་མ། ་ལུཊ་ལ། ཨིཊ་དང་ཡོན་ཏན་བྱས་པས། སེ་བི་ཏཱི། སེ་བི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ཨུ་པ་དྷཱའི་འང་ངོས་རིང་པོར་བསྒྱུར་བས། སཱི་བྱཱཏ། སཱི་བྱཱནྟཾ། སཱི་བྱཱ་སུ། ་སོགས་སོ། །།ཁྲིཊ་ལ། བལ་དང་པོའི་ཨིཊ་དང་། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། །སེ་བི་བྱེ་ཏི། སོགས་སོ། །།དེ་བཞིན་དུ། —།
