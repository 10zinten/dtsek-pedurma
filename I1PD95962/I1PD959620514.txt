གི་ཨེ་ཤ་ཨི་རེ་ཙའོ། །ལིཊ་འགྱུར་བའི་ཨཏྨ་ནེ་པ་དའི་ཏ་ཡིག་ནི་ཨེ་ཤ་དང་ཛྷ་་་ཡིག་ནི་ཨི་རེཙ་ཞེས་པར་འགྱུར་ཏེ། བ་བྷ་བེ། བྷཱུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་ཏ་རུ་བསྒྱུར། མདོ་འདིས་ཏ་ཨེ་ཤར་བསྒྱུར། ་མཐའི་ཧལ་འོས་ཤ་ཕྱིས། ལིཊ་ལ་བྱིངས་ཨ་བྷཱུས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད་ཐུང་དུ་འོས་ཨུ་ཐུང་དུར་བསྒྱུར། བྷ་བ་ཏིའི་ཨ་འོས་ཨུ་ཨར་སོང༌། ཨ་བྷཱུས་ལ་ཙར་འོས་བྷ་ཝར་བསྒྱུར། བྷཱུའི་བུཀ་འོས་བུ་ཀ་ཨཱ་ག་མ་བྱིན་ཀ་གནས་དོན་ཨུ་བརྗོད་དོན་ཝ་ཨེ་ལ་དྲངས། ལིཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་ལ། བ་བྷུ་བཱ་ཏེ། བྷ་ལ་ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་ཨཱ་ཧཱུྃ་དུ་བསྒྱུར། ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེར་འགྱུར། ལིཊ་ལ་བྱིངས་ཨ་བྷཱུས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད་ཐུང་ངུ་འོས་ཨུ་ཐུང་ངུར་བསྒྱུར། བྷ་བ་ཏིའི་ཨ་འོས་ཨུ་ཨར་བསྒྱུར་ཨ་བྷཱུས་ལ་ཙར་འོས་བྷ་ཝར་བསྒྱུར། བྷུའི་བུ་ཀ་འོས་བུ་ཀ་ཨཱ་ག་མ་བྱིན། ཀ་གནས་དོན་ཨུ་བརྗོད་དོན་ལྔ་ཨཱ་ལ་དྲངས། ལིཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། བ་བྷཱུ་བི་རེ། ...བྷཱུ་ལ་ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་ཛྷར་བསྒྱུར། ལིཊའི་ཏ་ཛྷ་དག་གི་ཨེ་ཤ་ཨི་རིཙ་འོས་ཛྷ་ཨི་རེ་ཙར་བསྒྱུར། མཐའི་ཧལ་འོས་ཙ་ཕྱིས་ལིཊ་ལ་བྱིངས་ཨ་བྷུས་མེད་པའི་བྱིངས་གཉིས་བརྗོད། ཐུང་ངུ་འོས་ཨུ་ཐུང་ངུར་བསྒྱུར། བྷ་བ་་་ཏིའི་ཨ་འོས་ཨུ་ཨར་བསྒྱུར། ཨ་བྷུས་ལ་ཙར་འོས། བྷ་བ། བྷཱུའི་བུ་ཀ་འོས་བུ་ཀ་ཨཱ་ག་མ་ག་གནས་དོན་ཨུ་བརྗོད་དོན་་ཡིན་པས་ཕྱིས། ཝ་ཨི་ལ་དྲངས། ལིཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་བྷཱ་བི་ཥ། བྷུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་ཐཱས་སུ་བསྒྱུར། ཐཱས་ཀྱི་སེ་འོས་ཐཱས་སེར་བསྒྱུར། ལིཊ་ལ་བྱིངས་་ཨ་བྷུས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད། ཐུང་ངུ་འོས་ཨུ་ཐུང་ངུར་བསྒྱུར། བྷ་བ་ཏིའི་ཨ་འོས་ཨུ་ཨར་སོང་ཨ་བྷཱུ་ས་ལ་ཙར་འོས་བྷ་ཝར་བསྒྱུར། བྷུའི་བུ་ཀ་འོས་བུཀ་ཨཱ་ག་མ་ཀ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་བྲ་ཨི་ལ་དྲངས། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱི་ས་ཥ། བ་བྷཱ་བཱ་ཐ། .བྷུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་་ནས། ལ་ཨཱ་ཐཱམ་དུ་བསྒྱུར། ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་ལིཊ་ལ་བྱིངས་ཨ་ཧྥུ་ས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད། ཐུང་ངུ་འོས་ཨུ་ཐུང་ངུར་བསྒྱུར། བྷ་བ་ཏི་ཨ་འོས་ཨུ་ཨ། ཨ་བྷཱུ་ས་ལ་ཙར་འང་ངོས་བྷ་བར་བསྒྱུར། བྷུའི་བུཀ་འོས་བུཀ་ཨཱ་ག་མ་བྱིན། ཀ་གནས་དོན་ཨུ་བརྗོད་དོན་ཝ་ཨཱ་ལ་དྲངས། བ་བྷཱ་བི་དྷེ། བྷཱུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་ཧཱུཾ་དུ་བསྒྱུར་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་ལིཊ་ལ་བྱིངས་ཨ་དྷྱ་ས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད། ཐུང་ངུ་འོས་ཨུ་ཐུང་ངུར་སོང་། བྷ་བ་ཏིའི་ཨ་འོས་ཨུ་ཨར་བསྒྱུར། ཨ་བྷཱ་ས་ལ་ཙར་འང་ངོས་བྷ་བར་བསྒྱུར། བྷཱའི་བུཀ་འོས་བུཀ་ཨཱ་ག་མ་་་བྱིན། ཀ་གནས་དོན་ཨུ་བརྗོད་དོན་བྱིངས་ཕྱེད་པའི་ཝལ་ལ་ཨིཊ་འོས་ཨིཊ་ཊ་གནས་དོན་ཨིཊའི་འདམ་ཀ་འོས་དྷ་ཌྷ་་་་འགྱུར་བ་དང་མི་འགྱུར་བ་ཡང་ན་བ་བྱས་པའོ། །བ་བྷཱ་བི་ཌྷ། བྷུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་ཧཱུྃ་དུ་བསྒྱུར་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་ལིཊ་ལ་བྱིངས་ཨ་ཧྥུ་ས་མེད་པའི་འོས་བྱིངས་གཉིས་བརྗོད། ཐུང་དུ་འོས་ཨུ་ཐུང་ངུར་སོང་། །
