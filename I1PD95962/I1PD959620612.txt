ཏཱཨིཊ་སྦྱར་བས་ད་དི་ཏཱ། ༽ད་དི་ཏཱ་རོ། ད་དི་ཏཱ་རཿ། ད་དི་ཏཱ་སེ། ད་དི་ཏཱ་སཱ་ཐེ། ་ད་དི་ཏཱ་དེ། ་ད་དི་ཏཱ་ཧེ། ད་དི་ཏཱ་སཱ་ཏེ། ད་དི་ཏཱ་སྨྲ་ཧེ། སྨོན་པའི་ལིང་ལ་འང་ཨིཊ་སྦྱར་བས། ད་དི་ཏྟི། ད་དི་ཏཱི་ཡཱནྟཾ། ད་དི་ཏཱི་རན་སོགས་སོ། ཡོ་། ཁྲིཊ་ལ། ཝལ་དང་པོའི་ཨིཊ་དང་སྱ་ཏཱས་ཞེས་པས་སྱ་སྦྱིན་པས། ད་དི་ཥྛ་ཏེ། ད་དི་བྱེ་ཏེ། ད་དི་ཉྩནྟེ་སོགས་སོ། །དེ་བཞིན་དུ་ལྡིང་ལ། ཨ་ད་དི་ཥྛ་ཏ། པ། ཨ་ད་དི་བྱེ་ཏཱཾ། ཌཱ་ཨ་ད་དི་བྱིན་སོགས་སོ། །ཥ་འགྲོ་བ་ལའོ་ཞེས་པ་ལ། །ལཊ་སོགས་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱར། ཆོ་ག་བཤད་མ་ཐག་པ་དང་འདྲ། །འདིར་སུཔ་ཀྱི་བྱིངས་སོགས་ཀྱི་བྱིངས་ཀྱི་དང་པོའི་ཥ་སར་མི་འགྱུར་ཏེ། །་ལཊལ། ཥཥ་ཏེ། ་་སྐེ་ཏེ། ཥནྟེ་སོགས་སོ། །ལིང་ལ། སྟེགས་ཏ། ལ་ཥཥཱ་ཡཱ་ཙཱ། སྐེ་རན་སོགས་སོ། །ལོཊ་ལ། Bཥ་ཏྭཾ། ཥཥཱ་ཏྭཾ། ཋཱནྟཾ་སོགས་སོ། །ལང་ལ། ཨ་ཥཏ། ཨ་ཀྵེ་ཏྭཾ། ཨ་ཥཱཀྐན་སོགས་སོ། །ལུང་ལ། བལ་དང་པོའི་ཨིཊ་སྦྱར་བས། །ཨ་ཥཱཀྐིཥྛ། ཨ་ཥཱཀྐི་ཥཱ་ཏྭཾ། ཨ་ཥཥཥ་ཏ་སོགས་སོ། །ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ལོ་ཥ་ཥཀྵེ། །ཥ་ཥཀྵ་ཏེ། །ཥ་ཥཥི་རེ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཥ་ཥཥི་ཏེ། ཥ་ཥཀྵ་ཐེ། ཥ་ཥཱཀྐི་དེ༑ཥ་ཥཞི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཥ་ཥཀྵེ། ཥ་ཥཞི་བ་ཧེ། ཥ་ཥཱཀྐི་མ་ཧེ། ལུཊ་ལ། ཨིཊ་སྦྱར་བའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཥཥི་ཏཱ། ཥི་ཏཱ་རོ། མི་ཥཥི་ཏཱ་ར། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཥཱཥི་ཏཱ་སེ། །ཞི་ཏཱ་སཱ་ཐེ། ཥཥི་ཏཱ་དེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཥཥི་ཏཱ་ཧེ། ཌ་ཥཥི་ཏཱ་སཱ་ཧེ༑་ཥཥི་ཏཱསྨྲ་ཧེ། སྨོན་པའི་ལིང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཥཱཥི་ཏིཥྛ། ཥཥིཏྟི་ཡཱནྟཾ། ཥཱཥི་ཏི་རན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཥཥི་ཊཱིཋཱ། ཋི་ཏཱི་ཡཱཧཱུཾ། ཥཥི་ཊཱི་ཧཱུྃ། །།ཥཱཥི་ཏཱི་ཌྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཥཥི་ཏི་ཡ། བལ་—།
