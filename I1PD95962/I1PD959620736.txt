དི༑དི་ཧནྟི། སོགས་སོ། །ཉེར་བསྒྱུར་ནི་དང་པྲ་ལྟ་བུ་སྔོན་དུ་སྦྱོར་ན། ག་ད། ན་ད། པ་ད་སོགས་ཀྱིས་ཎར་བསྒྱུར་བས། པྲ་ཎི་དེབྷི། སོགས་སོ། །མེད༈ལི་ཧ་རོ་མྱང་བ་ལའོ་ཞེས་པའང་གཉིས་ཀའི་ཚིག་ཅན་ཏེ། ལཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། །ལ་ཏིཔ། པུཀ་མཐའ་སོགས་ཀྱིས་་་ཡོན་ཏན། །ཧའི་ཞྭ་འོས། ཧ་ཌྷ། ཛྷཥ་སོགས་ཀྱིས་ཏ་དྷ། ་ཥ་ཊུ་ལ་ཥ་ཊུ་་འོས༑། དྷ་ལྷ། ཌྷ་ལས་དྷ་དབྱི་འོས། ལྷ་གོང་མ་ཕྱིས་པས། ལེ་བྷེ། གཉིས་ཚིག་ལ། ལ་ཏས། ཧའི་ལྷ། ཛྷཥ་སོགས་ཀྱིས་དྷ་ཌྷ། །ལྷ་ལས་ཌྷ་་་་ཕྱིས། །དྷར་ཕྱིས་པ་ལ་སྔོན་མའི་ཨ་ཎའི་རིང་པོ་འོས། རིང་པོར་བསྒྱུར་བས། །ལཱི་ཌྷ། མང་ཚིག་ལ་ཧི་འདྲེན་པ་ཁོ་ན་སྟེ། ལི་ཧནྟི། །སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ལ་སིཔ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ཧའི་ལྷ། ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས། ཌྷ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། ལེཀཱི། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ་ལ་ཐས་དང་ཐར་། བསྒྱུར། ཧའི་ཌྷ། ཛྷཥ་སོགས་ཀྱིས་ཐ་དྷ། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས། དྷ་ཌྷ། ཌྷ་ལས་ཌྷ་ཕྱིསི། ཌྷ་ར་ཕྱིས་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་པས། ལཱི་ཌྷ། །ལཱི་ཌྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ལེཊྚི། ལི་ཧཱ། ལི་ཧྥུ། ཆོ་ག་དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ལཱི་བྷེ། ལི་ཧཱ་ཏེ། ལི་ཧ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ལི་ཀྵེ། ལི་ཧཱ་ཤྲཱི། ལཱི་བྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ལི་ཧེ། ལི་ཧྭ་ཧེ། ལིཧྥ་ཧེ། །ལིང་གི་ཨ་ཏང་ལ་ཧ་འདྲེན་པ་ཁོ་ན་སྟེ། ལི་ཧཱཏ། ལི་ཧཱ་ཙཱ། ལི་ཧྥུ། སོགས་དང༌། ཏང་་་ལའང་ཨ་ཏ་མཐའ་མ་ཡིན་པ་སོགས་ཀྱིས་ས་ཕྱིས། ཧ་དྲངས་པས། ལི་ཧཱི་ཏ། །ལི་ཧཱི་ཡཱ་ཐཱི། ལི་ཧཱི་རན་སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། །
