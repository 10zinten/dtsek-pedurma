སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བེ་ཥ། བི་བེ་ཏི་བ། བི་བེ་ཏི་མ། ལུཊ་ལ། བེཏྟཱ། བེཥྛ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བི་བྷཱུཏ། བི་་ཙཱ། བི་བྷྱཱ་སུ། །སོགས་སོ། །ཁྲིཊ་ལ། པུཀ་མཐའ་སོགས་ཡོན་ཏན། ས་ལ་ཥ་དྷ་དག་གི་ཀ་འོས། །ཥ་ཀ་། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། བེགྱི་ཏི། སོགས་དང༌། དེ་བཞིན་དུ། ལྀང་ལ། ཨ་བེདྨེཏ། སོགས་སོ། །།དགའ་༈ཛན་སྐྱེ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། །ཤཔ་ལྷུར་བསྒྱུར་བ་སོགས་བྱས་པས། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཛ་ཛནྟི། ཛ་ན་ས་ན་ཁ་ནཱ་སན་ཛྷ་ལོ། ཛ་ན་ས་ན་ཁ་ན་རྣམས་ཀྱིས་ན་ཛྷལ་དག་ལའོ། །ཛ་ན་སྐྱེ་བ། ས་ན་དང་ཁ་ན་ཀུན་ནས་གུས་པ་རྣམས་:ཀྱི་ཊི་ནི་རྐྱེན་སན་དང་ཛྷལ་དང་པོ་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨཱཏ་དུ་འགྱུར་ཏེ། གཉིས་ཚིག་ལ། ཛ་ཛཱ་ཏ། །མང་ཚིག་ལ། །ཨ་བྱཱསྟ་ལས་ཨཏ་འོས། ཛྷ་ཨཏ་གམ་ཧན་ཛ་ན་་་་སོགས་ཀྱིས་ཨུ་བ་དྷཱ་ཕྱིས། ་ས་ཏུའི་ཤ་ཙུ་ལ་ཤ་ཙུ་འོས། ན་ཉ། ཛ་ཉ་འདུས་པ་ཛྷ་བྱས་པས། ་ཛ་ཛྷ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛ་ཛ་ནི་ཏི། ཨིཊ་ཛན་དག་ལས་དྷ་ལའང་ངོས། ཨིཊ། ཛ་ཛཱ་ཐ། ཛ་ཛཱ་ཐ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛ་ཛཎི། །ཛ་ཛ་ནཱ། ཛ་ཛམྨ། ལིང་ལ། ཡལ་འདམ་ཀ་འོས། ཊི་ཨཏྲ་དུ་ཡང་ན་བར་བྱས་པས། ཛ་ཛཱ་ཡཱཏ། ཛ་ཛཱ་ཡཱ'ཏྭཾ། ཛ་ཛཱ་ཡུ། སོགས་སམ། ཛ་ཛ་:ནྱ་ཏི། ཛ་ཛ་ནཱུ་ཏྭཾ། ཛ་ཛ་ནྟུ། །::སོགས་སོ། །།ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛ་ཛནྟུ། ཛ་ཛཱ་་་་ཏཱཏ༑ཡ་ཛ་ཛཱ་ཏྭཾ། ཛ་ཛྷ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཛ་ཛ་ཧི། །ཛ་ཛཱ་ཏཱཏ། ཛ་ཛཱ་ཏཾ། །ཛ་ཛཱ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛ་ཛ་ནཱ་ནི། ཛ་ཛ་ནཱ་བ། ཛ་ཛ་ནཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཛ་ཛན། —།
