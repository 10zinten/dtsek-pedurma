ཙ་ཀཱ་ར། ལུཊ་ཨ་ནི་ནི་ཥི་ཏཱ། །སྨོན་ལིངཨ་ནི་ནི་བྷཱུ་ཏ། ཁྲིཊ་ཨ་ནི་ནི་ཥི་བྱ་ཏི༑ལྡིང་…ཨ་ནི་ནི་ཥི་ཏ། སོགས་སོ། །ཨུ་བྷོ་སཱ་བྷཱ་ས་སྐྱ། སེ་ལ་ཨ་བྷཱ་སའི་གཉིས་དག་གིའོ། །ཨ་ནི་ཏིའི་སན་ཕ་རོལ་ལ་ཉེར་བསྒྱུར་ན་ཎར་བསྒྱུར་བའི་རྒྱུ་མཚན་ཅན་སྔོན་དུ་བྱུང་ན་ཨ་བྷྱཱ་ས་དང་ཨུ་པ་དྷཱའི་ན་ཡིག་གཉིས་ཀ་ཎ་ཡིག་ཏུ་བསྒྱུར་བར་བྱ་སྟེ། སཱ་ཎི་ཎི་ཥ་ཏི། སོགས་སོ། །ན་མ་ན་རེད། ཏེ་༈བྲྀ་ཏུ་འཇུག་པ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར། ཧ་ལནྡཱཙྪ། ཧལ་མཐའ་ལས་ཀྱང་ངོ༌། ྋ། བྱིངས་མཐའ་ཧལ་སོ་སོར་བསྡུ་བ་ཡིན་པ་ཨུ་པ་དྷཱ་ན་ཨིཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོད་པ་དེའི་ཕ་རོལ་གྱི་སན་ཨིཊ་མེད་པ་ནི་ཀ་འགྲོ་བའི་མིང་ཅན་ཏེ། འདིར་ཨཏྟ་རྣམས་ལ། བྲྀཏ་རྣམས་ལས་:བཞི་རྣམས་ལས་སོགས་ཨིཊ་མི་འཇུག། གཉིས་བརྗོད། རིའི་ཨཏ། དེ་སན་ལ་ཨིཏ་འོས་ཨིཏ་དུ་བསྒྱུར། ་མདོ་འདིས་ཀིཏ་ཡིན་པས་ཡོན་ཏན་སོགས་མི་བྱེད། །་བི་བྲྀཏྠ་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་ཨ་་་ཏང་རྣམས་སྦྱར་བས། ་་ལཊབི་བྲྀཏྠ་ཏི། ་ལིང༌བི་བྲྀཏྟེ་ཏ། ལོཊ་…བི་བྲྀཏྠ་ཏུ། ལང་ཨ་བི་ཀྲྀཏྠཏ། ལུང་ཨ་བི་ཀྲྀནི་ཏི། ལིཊབི་བྲྀགཾ་ཙ་ཀཱ་ར། ལུཊ་བི་བྲྀཏྟི་ཏཱ། སྨོན་ལིང་བི་བྲྀཏྟཱ་ཏ། ཁྲིཊབི་བྲྀཏི་ཥྛ་ཏི། ཁྲིང་…ཨ་བི་བྲྀཏྟི་ཏ། སོགས་སོ། །ཁྱིཁོ༈ཏང་རྣམས་ལ་བལ་དང་པོའི་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས། བི་བཏིཥ། ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊབི་བརྟི་ཥ་ཏེ། ་ལིང༌བི་བརྟི་ཥཏ། ལོཊ་བི་བརྟི་ཥ་ཏྭཾ། ལང་ཨ་བི་བརྟི་ཥཏ། ལུང་…ཨ་བི་བརྟི་ཏིཥྛ། ༼ལིཊ་བི་་་བརྟི་ཊཱ་ཙ་ཀྲེ། ལུཊ་བི་བརྟི་ཥི་ཏཱ། སྨོན་ལིང་བི་བརྟི་ཊཱིཋ། ཁྲིཊ་བི་བརྟི་ཏི་༄་ཏེ། ཁྲིང་ཨ་བི་བརྟི་ཏ། སོགས་སོ། །ཅི་ཕན་དེ་དག་ལ་ར༈ཚྭ་ཏི་ལུས་བསྐྱོད་པ་ལའོ་ཞེས་པ་ལ་སན་སྦྱར། ་ཀྲྀཏ་ཙཱཏ་ཀྲྀཏ་ཀྲྀཏ་་་ཀྲྀཏ་སོགས་ཀྱི་ས་ཨིཊ་ཡང་ན་བར་བྱས། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། སན་ཀིཏ་—།
