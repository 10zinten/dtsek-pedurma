ཀུ་་མ་་གས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིདེ་མ་ཆོག་གམ་ཚོ་ཚོས་དཀ་ཌ་བློ་ལྡན་དཱིདྡྷཾ་ཙ་ཀྲེ། དྲི་ཌྷ་བ་བྷུ་བ། །དཱི་ཧ་མཱ་ས། ་སོགས་སོ། །ལུཊ་ལ། ་བལ་་་དང་པོའི་ཨིཊ་ཡི་ཨི་ཡིག་དག་ལ་དཱི་དཱི་བེ་བཱི་དག་གི་འོས་ཨཱི་ཕྱིས་པས། དྲི་དྷི་ཧཱུཾ། དཱི་དྷི་ཏཱ་སེ། དེ་བཞིན་དུ། །སྨོན་པའི་ལིང་ལའང་ཨིཊ་སྟེར་བ་སོགས་བྱས་པས། །དཱི་དྷི་ཝ། དཱི་དྷི་ཊཱི་ཡཱནྟཾ། དྷི་དྷི་ཊཱི་རན། སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་ཨཱི་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། དྲི་དྷི་ཥྛ་ཏེ། དཱི་དྷི་ཥྛ་ཏེ། དཱི་དྷི་ཥྛནྟེ། སོགས་དང༌། དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་དཱི་བྷི་ཥྱ་ཏ། ཨ་དཱི་དྷི་བྱེ་ཏྭཾ། ཨ་དཱི་དྷི་ཥྱན། ་སོགས་སོ། །ལས་བཞི་ཆུ་བྱ་སྟེ་དོན་ཁག་ལ་༈བེ་བཱི་རྔན་པ་མཚུངས་པ་ལའོ་ཞེས་པ་ལའང་ཏང་རྣམས་འཇུག་ཅིང་ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་འདྲ་སྟེ། བེ་བཱི་ཏེ། །བེ་བྱཱ་ཏེ། བེ་བྱ་ཏེ། སོགས་སོ། །p:pབ་ས་ཀྱི་ག་ང། བློ་ར་བ་༈དྭི་ཥ་མི་དགའ་བ་དག་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་རྣམས་འཇུག་སྟེ། པ་འགྲོ་བའི་མིང་ཅན་རྣམས་ལ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་དང༌། །ཥ་ཊུ་ལ་ཥ་ཁུ་སོགས་བྱས་པས། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །དྭེཉྫི། དྲི། དྲི་ཥཱནྟི༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྭེ་ཀཱི། ས་ལ་ཥ་ལྷ་དག་གིས་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ཥ། དཱིཥྛ། དཱིཥཱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །དཱིརྨི། དྭི་ཥ། དཱི༑ལིང་ལ་ཥ་འདྲེན་པ་ཁོ་ན་སྟེ། དྭི་ཥྛཏ། དྲི་བྷཱ་ཏཱ། ལ་དྲི་རྒྱུ། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དྭེ་། དྭེཥྛ། །དྲི་ཏྟཾ། དྲི་ཥནུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དྲི་རྟི། ལ། སིཔ། སི་ཧི། ཛུ་ཧལ་ལ་ཧི་དྷི། མཐའ་ཛྷལ་སོགས་ཥ་ཌ་ཏེ་ཁུ་ལ་ཁ་ཏུ་འོག་དྷ་དྷ་དྲངས། དྲིཏྟཱཏ། དྭིཤྩ། དཱི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དྭེ་ཥཱ་ཎི། དྭེ་ཥཱ་པ། དྭེ་ཥཱ་མ། :ལང་གི་སྐྱེས་བུ་དང་པོའི་་་་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ། །ཨ་དྭེཊ། ལ་ཏི། ཨིཏའང་ངོས་ཨི་ཕྱིས། ཧལ་དིཔ་སོགས་ཀྱིས་—song།
