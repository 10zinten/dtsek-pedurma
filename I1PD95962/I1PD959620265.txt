སྒྲ་མདོ། །''་མཚོ་བའི་མ་ཎའ་ཉིན་བློ་ལྟར་མཆོད་སྡོང༌། ཨིུདྷ་སོ་ནང༌། །སྤིུས་ཀྱི་ཨ་ནང་ངོ༌། །ཨཱུ་དྷས་ཞེས་པའི་ཊི་ནི་བུད་མེད་དུ་བརྗོད་པའི་རྐྱེན་ཕ་རོལ་ལ་ཨ་ནང་དུ་འགྱུར་ཏེ། ཊི་ཨ་ནང་དུ་བསྒྱུར། མཐའི་ཧལ་འོས། ང་ཕྱིས། ནའི་ཨ་བརྗོད་དོན་ཨུདྡྷ་ན། བ་ཧུ་ཤྲཱི་ཧེ་རུ་དྷ་སོ་དཱིཥ། །འབྲུ་མང་པོ་ལ་ཨུ་དྷས་ལས་དཱིཥའོ། །འབྲུ་མང་གི་ཚིག་སྡུད་ཅན་ཨུ་དྷས་ལ་བུད་མེད་དུ་བརྗོད་པའི་རྐྱེན་དཱིཥ་འགྱུར་ཏེ། ་ནའི་ཨ་དབྱི་འོས་ཨུ་པ་་་དྷཱའི་ཨ་ཕྱིས། པཱི་ནོ་། ནུ་མ་འོ་མས་གང་བ། བུད་མེད་དུ་བརྗོད་པའི་རྐྱེན་མ་ཡིན་ན་ཊིཨ་ནང་དུ་མི་འགྱུར་རོ། །དཱ་མ་ཧཱ་ཡ་ནཾ་ཏཱཙྪ། །དཱ་མན་ཧཱ་་་ཡན་ལས་འང་ངོ༌། །གྲངས་སྔ་མའི་དཱ་མན་དང་ཧཱ་ཡན་ལས་ན་ཚོད་ལ་དཱིཥ་་་འགྱུར་ཏེ། དྲི་དཱམཱི། ཆུ་བོ་གཉིས་མ། དྲི་ཧཱ་ཡ་ནཱི། ལོ་གཉིས་མ། །ཡང་འགྲེལ་པར་ཧཱ་་་ཡན་ལ་ན་ཚོད་ལ་དཱིཥ་འགྱུར་ཞིང༌། །ཏྲི་ཙ་ཏུར་དག་སྔོན་དུ་བྱུང་ན་ནའི་འང་་་ཎའོ་ཞེས་པ་བྱུང་བས། །ཏྲི་ཧཱ་ཡ་ཎཱི། ལོ་གསུམ་མ། ཙ་ཏུནཱ་ཡ་ཎཱི། ལོ་བཞི་མ། ན་་་ཚོད་མིན་ན་དཱིཥ་མི་འགྱུར་ཏེ། ཏྲི་ཧཱ་ཡ་ནཱ་ཤཱ་ལཱ། ལོ་གསུམ་གྱི་མགྲོན་ཁང་། ཨནྟརྦཏྤ་་ཏི་བ་ཏོརྣཀ། ཨནྟརྦཏ་པ་ཏི་བཏ་ལས་ནུ་ཀའོ། །ཨནྟརྦཏ་དང་པ་ཏི་བཏ་དག་ལས་བུད་མེད་དུ་བརྗོད་པའི་རྐྱེན་ཕ་རོལ་ལ་ནུཀ་ཨཱ་ག་མ་འགྱུར་ཏེ། ཨནྟརྦདྡྷི། མངལ་ལྡན་མ། པ་ཏི་བཏི། བདག་པོ་ལྡན་མ་སྟེ་ཆུང་མ། པ་ཏྱརྣོ་ཡཛྷསཾ་ཡོ་གེ། པ་ཏིའི་ན་་་མཆོད་སྦྱིན་སྦྱོར་བ་ལའོ། །།པེ་ཏིའི་སྒྲ་ལས་མཆོད་སྦྱིན་སྦྱོར་བ་ཅན་ཏེ་བདག་པོ་དང་ལྡན་པའི་དོན་ལ་ཊི་ནར་འགྱུར་ཞིང་དཱིཥ་འཇུག་སྟེ། པཏྟཱི། བདག་ལྡན་མ། བི་བྷཱ་ཥཱ་ས་པཱུརྦསྱ། སྔ་མ་དང་བཅས་པའི་ཡང་ནའོ། ་། མིང་གཞན་སྔ་མ་དང་བཅས་པའི་པ་ཏིའི་སྒྲ་ན་རུ་འགྱུར་བ་དང་དཱིཥ་ཡང་ན་སྟེ། དྲྀ་ཌྷ་པནྟི། བསྟན་པའི་་བདག་མོ། དྲྀ་ཌྷ་པ་ཏིནི་ཏྱཾ་ས་པཏྟཱ་དི་ཥཱ། བསྟན་པའི་བདག་པོ། ས་པདྡྷི་སོགས་རྣམས་ལ་རྟག་ཏུའོ། །ས་མཱ་ན་ལ་སོགས་པ་རྣམས་ལས་ཕ་རོལ་གྱི་པ་ཏི་སྒྲ་རྟག་ཏུ་ན་དང་—དད་མངལ་།
