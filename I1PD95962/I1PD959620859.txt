བདེ་བ་སྐྱེ་བསྟན་འགྱུར། སྒྲ་མདོ། །ཉ་ཚོ་བོ་མཚོ་ཊཀ་ཅུས:དྲྀཎརྟི༑དྲྀནྟ། དྲྀནྡནྟི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྲྀ་ཎཏི། དྲྀནྡྷ། དྲྀནྡྲ༑སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བྲྀཎ། དྲྀནྡྲ། །ོདྲྀནྣ། ལིང་ལ། ཏཱཾནྡྷཱ་ཏ། :：དྲྀན་ཏཱ?། ཏཱཾནྡྷཱ་སུ། སོགས་སོ། །།ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བྲྀཎ་ཏུ། །ཏཱཾནྟཱཏ། ཀྲྀནཾ། ཏཱཾནྡནྟུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྲྀདྡྷི། དྲྀནྟཱཏ། ཀྲྀནྟཾ། དྲྀན། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཀྲྀ་ཎ་དཱ་ནི། ཀྲྀ་ཎ་དཱ་བ། ཀྲྀ་ཎ་དཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀྲྀ་ཎ་ཏ། ཨ་ཀྲྀ་ཎ་ད། ཨ་ཏྲིནྟཾ། ཨ་ཀྲྀནྡན། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཀྲྀ་ཎ། ཨ་ཀྲྀ་ཎ་ཏ། ཨ་ཀྲྀ་ཎ་ད། ཨ་ཏྲིནྟཾ། ཨ་ཙཱནྟ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀྲྀ་ཎ་དཾ། །ཨ་ཀྲྀནི། ཨ་ཀྲྀནྡྷ། ་ལུང་ལ་ཨིར་འགྲོ་བ་སོགས་ཀྱིས་ཙི་ཨང་དང་སིཙ་ཡང་ན་བར་བྱས་པས། ཨང་གི་ཕྱོགས་ལ། ཨ་ཀྲྀ་དཏེ། ཨ་ཀྲྀ་ད་ཏྭཾ། ཨ་ཀྲྀ་དན། སོགས་དང་། སིཙའི་ཕྱོགས་ལ་ཨིཊ་བྱིན་པས། ཨ་ཏརཱིཏ། ཨ་ཏརཱི་ཏཱཾ། ཨ་ཏཇི་ཥཱ། སོགས་སོ། །ཀྲྀཊ་ལ་གཉིས་བརྗོད་་པ་སོགས་ཀྱིས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཏ་ཏརྡ། ཏ་ཀྲྀ་ད་ཏུ། ཏ་ཀྲྀ་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཏ་ཏཇི་ཐ། ཏ་ཀྲྀ་ད་ཐུ། ཏ་ཀྲྀ་ད། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཏ་ཏརྡ། ཏ་ཀྲྀ་དི་བ། ཏ་ཀྲྀ་དི་མ། ལུཊ་ལ་ཨིཊ་བྱིན་་་་པས༑ཏཇི་ཏཱ། ཏཇི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བྲྀ་ཏྭ་ཏ། ཀྲྀ་དྷཙཱ། ཀྲྀ་ཅུ་སུ། སོགས་སོ། །ལྀཊ་ལ། ཀྲྀཏ་བྲྀཏ་ཚིཏ་ཏྲིད་སྲིད་ལས་སིཙ་མ་ཡིན་པའི་ས་ལ་འོས། ཨིཊ་ཡང་ན་བར་བྱས་པས། ཨིཊའི་ཕྱོགས་ལ། ཏཇི་སྱ་ཏི། སོགས་དང་། །ཨིཊ་མེད་པའི་ཕྱོགས་ལ། ཏིཥྛ་ཏི། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལའང་ཨིཊ་ཡང་ན་བར་བྱས་པས། ཨ་ཏཇི་ཏ༑སོགས་སམ། ཨ་ཏིཉྩ་ཏ། སོགས་སོ། །ཏཥ། —།
