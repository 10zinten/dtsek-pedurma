ཨཱཝ་རྣམས་སོས་ཨོ་ཨབ། བྲ་ཨི་ལ་དྲངས། ཀུན་ནས་བསྒྱུར་བ་དང་རྐྱེན་སོགས་ཀྱིས་ས་ཥ། བྷ་བི་ཥཱ་ཐ། བྷུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ཁྲ་ཐས་སུ་བསྒྱུར། སྐྱ་ཏཱས་དག་ཁྲི་ལུཊ་དག་ལ་འོས་སུ་སྦྱར། བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས། ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས། བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། བྷ་བི་ཥཱ་ཐ། བྷུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། །ལ་ཐ་རུ་་་བསྒྱུར། སྱ་ཏཱས་དག་ཁྲི་ལུཊ་དག་ལ་འོས་སུ་སྦྱར། བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་་་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། ཀྲྀཊའི་པ་རསྨི་པ་དའི་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བྷ་བི་བྷཱུ་མི། .བྷུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་མིཔ་དུ་བསྒྱུར། པ་པིཏ་ཀྱི་དོན། སྱ་ཏཱས་དག་ཁྲི་ལུཊ་དག་ལ་འོས་སྱ་བྱིན། ཡཉ་ལ་ཨཏ་ཀྱི་རིང་པོ་འོས། སྐྱ་རིང་་་་པོར་སོང་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་བྱིན་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་་་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། བྷ་བི་བྷཱུ་བ། བྷཱུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས་ལ་བས་དང་མས་སུ་བསྒྱུར། སྱ་ཏཱས་དང་ཁྲོ་ལུཊ་དག་ལ་འོས་སུ་སྦྱར། ཡཉ་ལ་ཨཏ་ཀྱི་རིང་པོ་འོས། སྐྱ་རིང་པོར་སོང༌། བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། བྲ་ཨི་ལ་དྲངས། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་སྲ་ཁྲ། བྷ་བི་བྷཱུ་མ། དེ་རྣམས་བྷཱུ་ལ་ཁྲིཊའི་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བའོ། །བྷུ་ཐོབ་པ་ལའོ་ཞེས་པའི་་་བྱིངས་ལ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱར་བའི་དཔེར་བརྗོད་ནི། བྲྀཊའི་ཨཱཏྨ་ནེ་པ་དའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བྷ་བི་བྱ་ཏེ། བྷུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ལ་ཏ་རུ་བསྒྱུར། ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། སྱ་ཏཱས་དག་ཁྲི་ལུཊ་དག་ལ་འོས། སྱ། བྱིངས་ཕྱེད་པའི་བལ་ཨིཊ་ཊ་གནས་དོན་བྱིངས་་་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས། བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། བྷ་བི་བྱེ་ཏེ། བྷུ་ལ། ལིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ཕྲ་ཨཱ་ཏཱ་དུ་བསྒྱུར། ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ༑སྱ་ཏཱས་དག་ལྀ་ལུཊ་དག་ལ་འོས་སུ་བྱིན་བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས། ཨིཊ་ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་ལ་ཡོན་ཏན་ཨེཙ་སོགས་ཀྱིས་ཨོ་ཨབ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། ད་འགྲོ་བའི་ཨཱཏ་ཀྱི་འོས་ཨཱ་ཏྭཾ་གྱི་ཨཱ་ཨི་ཡ། ་བ་ཡ་དག་གི་བལ་ལ་འོས་ཡ་ཕྱིས། ཨ་ལས་ཡོན་ཏན་ནོས་ཨེ། བྷ་བི་་ཊྛནྟེ༑བྷུ་ལ། ཁྲིཊ་རྗེས་འབྲེལ་ཕྱིས་ནས། ཡཛྷར་བསྒྱུར། ཛྷའི་ཨནྡ་འོས་ཛྷ་ཨནྡ་ར་བསྒྱུར། ད་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། སྱ་ཏཱས་དག་ཁྲི་ལུཊ་དག་ལ་འོས་སྱ་སྦྱར། ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས་གཉིས་དག་གི་གཅིག་སྔ་མའི་གཟུགས། བྱིངས་ཕྱེད་པའི་བལ་ལ་ཨིཊ་འོས་ཨིཊ་བྱིན། ཊ་གནས་དོན་བྱིངས་ཐམས་ཅད་པ་དང་། བྱིངས་ཕྱེད་པ་དག་གི་འོས་བྱིངས་—།
