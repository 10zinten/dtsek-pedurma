བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནིའི་གཞུང་འགྲེལ་གྱི་གོ་དོན་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ། ཥི་ཏཾ། བཱ་ཀྱ་མཱ་ཧ། ཚིག་སྨྲས་པ། ཨརྗེསཾ་ནི་བི་བྷ། ཨཇིའི་སཾ་ནི་བི་རྣམས་ལས་སོ༑། སཾ::ནི:བི་རྣམས་ལས་ཨཇི་ལས་ནིཥྛ་ལ་ཨིཊ་མི་འགྱུར་ཏེ། ས་མཎྞ༑མཉམ་པར་ཟོས། རྱཎྞ། ངེས་པར་ཟོས། བྱཎ། ལྷག་པར་ཟོས། ཨ་བྷེཝཱ་བི་དྷུམེ། ཨ་བྷི་ལས་ཀྱང་མི་རིང་བ་ལའོ། །ཨ་བྷི་ལས་ཀྱང་ཨརྡ་ལས་མི་རིང་བ་ལ་འཇུག་པ་ན་ནིཥྛ་ལ་ཨིཊ་མི་འགྱུར་ཏེ། ཨ་བྱཱཎྞ། ལྷག་པར་ཉེ་བར་ཟ། ཎེ་ར་རྒྱ་པ་ནེ་བྲྀཏྟཾ། ཎིའི་་ཀློག་པ་ལ་བྲྀཏྟཾའོ། །བྲྀ་ཏ་ཎི་ཙའི་མཐའ་ཅན་ལས་ལྟ་སྦྱར་བས་བྲྀཏྟཾ་ཞེས་པ་ཀློག་པ་ལ་འཇུག་པ་ན་ཚིག་ཕྲད་དོ། །བཱ་དཱནྟ་ཐཱནྟ་པུཎ་དཱཋ་སྤཏྠ་ཚནྣ་ཛྷཔ། དཱནྟ；ཡདུལ་བ། ཤཱནྟཾ；ཡཞི་བ། པཱུརྞ；གང་བ། དཱཋ：ཡམཆེ་བ། སྤཊྚ：གསལ་བ། ཚནྣ：སྒྲིབ་པ། ངེས་སྦྱོར། ཛྷཔ་རྣམས་སོ། །ད་མུ་དང་ཤ་མུ་ཞི་བ། པུ་རེ་ཚིམ་པ། ད་སུ་ཉེ་བར་འཇོག་པ། སྤ་ཤ་སྣོད། ཚ་ད་སྒྲིབ་པ། ཛྷ་ངེས་་་སྦྱོར་རྣམས་ཎིཙའི་མཐའ་ཅན་ལ་ལྟ་སྦྱར་བས། དཱནྟ་སོགས་ཡང་ན་ཚིག་ཕྲད་དོ། །ཡང་ན་ཨིཊ་བྱིན་པས། ད་མི་ཏ། ཤ་མི་ཏ། པུ་རི་ཏ། ད་སི་ཏ། སྤ་ཤི་ཏ། ཚ་དི་ཏ། །ཛྷཱ་ནསོགས་སོ། །རུ་ཥྱ་མ་ཏྲི་ར་སཾ་གྷོ་ཥཱ་སཱ་ནཾ། རུ་ཥ：:ཨ་མཏཱ་ར་སཾ་གྷོ་ཥ：ཨཱ་སཱན་རྣམས་ཀྱིའོ། །རུཥ་ལ་སོགས་པ་འདི་རྣམས་ཀྱི་ཕ་རོལ་གྱི་ནིཏྟཱ་ལ་ཨིཊ་ཡང་ན་མི་འགྱུར་ཏེ། རུཊྚ། ཁྲོས་པ། །ཨོཾ་ཏ། ནད་པ། ཏུཎ། མགྱོགས་པ། སཾ་བྷུཊྚ། ཀུན་ནས་བསྒྲགས། ཨཱ་སྭ་ཏ། ཀུན་དུ་བསྒྲགས། ཡང་ན་ཨིཊ་བྱིན་པས། རུ་ཥི་ཏ། ཨ་མི་ཏ། ཏཱ་རི་ཏ། སཾ་གླུ་ཥི་ཏ། །ཨཱ་སཱ་ནི་ཏ། ཀྲྀ་ཁེརློ་མ་སུ། པྲྀ་ཥ་ལ་བ་སྤུ་ལའོ། །ཤྲཱི་ཥ་ལས་་་ནིཥྛ་ལ་བ་སྤུ་བརྗོད་པ་ལ་ཨིཊ་ཡང་ན་མི་འགྱུར་ཏེ། །བྲྀཏྟ། བ་སྤུའམ་མ་གུ་པ། བྲྀ་ཏི་ཏ༑འགྲེལ་པར་ལུང་དྲངས་པ། བིསྨི་ཏ་པྲ་ཏི་བྷཱ་ཏ་ཡོ༄༅། ་བིསྨི་ཏ་དང་་པྲ་ཏི་བྷཱ་ཏ་དག་ཀྱང་ངོ༌། །བི་སྔོན་མའི་གླིང་ཅུང་ཟད་རྒོད་པ་དང༌། པྲ་ཏི་སྔ་།
