བ་་་་་་་་ག་རེ་བསྟན་འགྱུར། །སྒྲ་མདོ། ཏིཀ་ནི་ནི་ཀོ་མཚོན་གཙོས་ཡུ་ཁ་རི་དང་བློ་གཏད་ཨ་དྷ་བི་ཥ་ཏྭཾ། ཨ་དྷ་བི་ཥ་ཏ། །སོགས་སམ། ཨིཊ་མེད་པའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་དྷོ། ་ཨ་དྷོ་ཥཱ'ཏྭཾ། ཨ་དྷོ་ཥ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷོཥྛ། ཨ་དྷོ་ཥཱ་ཐཱ། ཨ་དྷོ་ཧཱུྃ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དྷོ་ཥི། །ཨ་དྷོ་ཥཱ་ཧི། །ཨ་དྷོ་ཧྥ་ཧེ། ལིཊ་ལ། གཉིས་བརྗོད་པ་སོགས་་་་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དུ་དྷཱ་བ། དུ་དྷུ་བ་ཏུ། དུ་དྷུ་བུསྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །དུ་དྷུ་བི་ཐ། དུ་དྷུ་བ་ཐུ། དུ་དྷུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་དྷཱ་བ། །དུ་དྷ་བ། དུ་དྷུ་བི་བ། དུ་དྷུ་བི་མ། ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དུ་དྷུ་བེ། དུ་དྷུ་བཱ་ཏེ། དུ་དྷུ་བི་རི། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །དུ་དྷུ་བི་ཥ། དུ་དྷུ་བཱ་ཐེ། དུ་དྷུ་བི་དྷེ། བུ་དྷུ་བི་རྗེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དུ་དྷུ་བེ། དུ་དྷུ་བི་བ་ཧེ། དུ་དྷུ་བི་མ་ཧེ། །ལུཊ་ལ། ཨུཏ་འགྲོ་བ་སོགས་ཀྱིས་ཨིཊ་ཡང་ན་བར་བྱས། ཡོན་ཏན་བྱས་པས། ཨ་ཏང་ལ། དྷ་བི་ཏཱ། དྷ་བི་ཏཱ་སི། སོགས་སམ། དྷོ་ཏཱ། དྷོ་ཏཱ་སི། སོགས་སོ། །ཏང་ལ། དྷ་བི་ཏཱ། དྷ་བི་ཏཱ་སེ། སོགས་སམ། དྷོ་ཏཱ། བྷོ་ཏཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། དུ་ཡཱཏི། དྷཱ་ཡཱཙཱ། དྷཱ་ཡཱ་སུ། སོགས་དང༌། ཏང་ལ་ཨིཊ་ཡང་ན་བར་བྱས་པས། དྷ་བི་ཏི། དྷ་བི་ཏི་ཡཱནྟཾ། དྷ་བི་ཏི་རན། སོགས་སམ། རྡོ་ཏཱིཋ༑སོགས་སོ། །ཁྲིཊ་ལའང་ཨིཊ་ཡང་ན་བར་བྱས་པས། །ཨ་ཏང་ལ། དྷ་བི་ཥྱ་ཏི། སོགས་སམ། བྷོ་ཥྱ་ཏི། སོགས་སོ། །ཏང་ལ། ་དྷ་བི་ཥྛ་ཏེ། སོགས་སམ། དྷོ་ཥཱ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་དྷ་བི་ཥྱ་ཏི། སོགས་སམ། །་ཨ་དྷོ་ཥྱ་ཏ། སོགས་སོ། །ཏང་ལ། །ཨ་དྷ་བི་ཏ། སོགས་སམ། ཨ་དྷོ་་ཏ། སོགས་སོ། །ཤ་། —།
