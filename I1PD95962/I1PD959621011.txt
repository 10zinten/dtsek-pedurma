སྒྲ་མདོ། ༼ཏིའི:ཀ་ཀེ་་ཟླ་སྒྲོན་མ་ཁྱོད་ལ་བྱིན། གློ་བ་བྱocསོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དེ་དི་བཱི་ཏིུ། དེ་བྱེ་ཏུ། དེ་དེ་ཏུ། དེ་ཙྭ་ཏྭཾ། དེ་དི་ཏཱཾ། དེ་དི་བ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དེ་ཅུ་ཏི། དེ་ཅུ་ཏཱཏ། དེ་ཅུ་ཏཾ། དེ་དི་ཏཾ། དེ་དྷུ་ཏ། །དེ་དི་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། དེ་དི་བཱ་ནི། །དེ་དི་བཱ་བ། དེ་དི་བཱ་མ། དེ་བཞིན་དུ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཨ་དེ་དི་བཱིཏ། ཨ་དེ་ཁྱོད། ཨ་དེ་དེཏ། ཨ་དེ་རཱུ་ཏཱ། ཨ་དེ་དི་ཏཱཾ། ཌ་ཨ་དེ་དི་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དེ་དི་བཱི། ཨ་དེ་བྱོ། ཨ་དེ་དེ། ཨ་དེ་རཱུ་ཏཾ། ཨ་དེ་དི་ཏཾ། ཨ་དེ་དྷུ་ཏ། ཨ་དེ་དི་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དེ་དི་པཾ། ཨ་དེ་ཅུ་བ། །ཨ་དེ་དི་བ། །ཨ་དེ་ཅུ་མ། ་ལུང་ལ་ཨིཊ་བྱིན་པས། ཨ་དེ་དི་བཱིཏ། །སོགས་དང༌། དེ་བཞིན་དུ། ལ་ཡིག་ལྷག་མ་རྣམས་ལ། ལིཊདེ་དི་བཱ་ཙ་ཀཱ་ར། ་ལུཊདེ་དི་བི་ཏཱ། སྨོན་ལིང་དེ་དཱི་བྱཱཏ། ཀྲྀཊདེ་་་་བི་ཥྛ་ཏི། ཁྲིང་ཨ་དེ་དི་བི་ཏ། སོགས་སོ། །ཚུ་། ་ད། RE:འོ་ཀ་བ་གཏན་འོ༈བི་ཚ་འགྲོ་བ་ལའོ་ཞེས་པ་ལ་ཡང་སྦྱར། ཁ་གཉིས་བརྗོད། ཡང་འཇིག་པ་སོགས་ཀྱིས་ཡོན་ཏན། ཡང་གི་ཨཙ་ལའང་ངོས་ཡང་ཕྱིས་པས། །བེ་བི་ཚ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཡང་ལས་ཡང་ན་འོས་ཨིཏ་ཡང་ན་བར་བྱས། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། བྲཧྥ་བྷསྔ་སོགས་ཀྱིས་ཚ་འམ་ཤ་ཥ། །ཥ་ཊུ་ལ་ཥ་ཊུ་འོས་ཏ་ཊ། ཡོན་ཏན་བྱས་པས། བེ་བི་ཛཱི་ཏི། བེ་བིཊཱི། གཉིས་ཚིག་ལ། བྲཤྩ་བྷསྔ་སོགས་ཀྱིས་ཚ་འམ་ཤ་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས་ཏ་ཊར་བསྒྱུར་བས། བེ་བིཊྚ། མང་ཚིག་ལ། ཨ་བྷཱ་ྋས་་ལས་ཨཏ་འོས་ཛྷ་ཨཏ་དུ་བསྒྱུར་བས། བེ་བི་ཚ་ཏི། དེ་རྣམས་དང་འདྲ་བར་་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་བི་ཛཱི་ཏི། བེ་བིཊྚི། བེ་བིཥྛ། །—།
