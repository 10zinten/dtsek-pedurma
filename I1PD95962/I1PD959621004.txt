བ་ཐུ། བོ་བླུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བོ་བྷ་བ། བོ་བྷ་བ། བོ་བྷུ་བི་བ། བོ་བྷེ་བི་མ། ལུཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། བོ་བྷ་བི་ཏཱ། བོ་བྷ་་བི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བོ་དྷཱ་ཡཱཏ། བོ་ངྷ་ཡཱནྟཾ། བོ་བྷཱུ་ཡཱ་སུ། སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་ཡོན་ཏན་བྱས་པས། བོ་བྷ་བི་ཥྛ་ཏི། སོགས་དང་། དེ་བཞིན་དུ་ལྡིང་ལ། ཨ་བོ་བྷ་བི་ཏ། སོགས་སོ། །༈བཉྩ་འགྲོ་བ་ལའོ་ཞེས་པ་ལ། ཡང་སྦྱར། གཉིས་བརྗོད། བཉྩ་གྲོགས་སུ་དྷཾ་སུ་བྷཾ་སུ་སོགས་ཀྱིས་ནཱིཀ་ཨཱ་ག་མ་བྱིན། ཡང་གི་ཨཙ་ལའང་ངོས་་ཡང་ཕྱིས་པས། བ་ནཱི་པཉྩ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཡང་ལས་ཡང་ན་འོས་ཨཱིཊ་ཡང་ན་བར་བྱས། ཨཱིཊ་མེད་པའི་ཕྱོགས་ལ། ཙུའི་ཀུ་བྱས་པས། བ་ནཱི་པཾ་ཙཱི་ཏི། བ་ནཱི་པཾ་ཀྵི། གཉིས་་ཚིག་ལ་ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་ན་ཕྱིས། ཙུའི་ཀུ་འོས་ཙ་ཀ་ར་བསྒྱུར་བས། བ་ནཱི་པཾ་ཀཿ། །མང་ཚིག་ལ། ཨ་བྱས་ལས་ཨཏ་འོས་ཛྷ་ཨཏ་དུ་བསྒྱུར་བས། བ་ནཱི་པཾ་ཙ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །བ་ནཱི་པཾ་ཙཱི་ཏི། བ་ནཱི་པཾ་ཀྵ། བ་ནཱི་བཀྑ། །བ་ནཱི་བཀྑ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་ནཱི་པཾ་ཙཱ་མི། བ་ནཱི་པཾ་ཙྪི། བ་ནཱི་པཾ་ཙཱ། བ་ནི་བཻ་ཙ་མ། །ལིང་ལ། ་ཨིཏ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་ན་ཕྱིས་པས། བ་ནི་བ་ཙཱཏམེ། བ་ནཱི་བ་ཙཱ་ྋཏྭཾ། བ་ནཱི་པཾ་ཙུ། སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་ནཱི་པཾ་ཙཱ་ཏུ། བ་ནཱི་པཾཀྱུ། ་བ་ནི་བཀྐཱ་ཏ། བ་ནཱི་བཀཾ། བ་ནཱི་བཙ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་བ་ནི་བབྷི། བ་ནཱི་བཀཱཏ། །ལབ་ནི་བཏྟཾ། །བ་ནི་་་་་བཀྟ༑སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་ནཱི་པཾ་ཙཱ་ནི། བ་ནཱི་པཾ་ཙཱ་བ། བ་ནཱི་པཾ་—།
