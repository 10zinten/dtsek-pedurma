དྷུ་ན་ཡི་ཏ། སོགས་དང༌། ནུཀ་མ་བྱིན་པའི་ཕྱོགས་ལ་འཕེལ་བ་བྱས་པས། དྷཱ་བི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊདྷཱ་བ་ཡ་ཏི། ལིང་དྷཱ་བ་ཡེཏ། ་ཁ་ལོཊ'དྷཱ་བ་ཡ་ཏུ། ལང་ཨ་དྷཱ་བ་ཡཏ། ལུང་ཨ་དུ་དྷ་བཏ། ལིཊ'དྷཱ་བ་ཡཱ་ཙ་ཀཱ་ར། ཌལུཊ'དྷཱ་བ་ཡི་ཏཱ། སྨོན་ལིང་དྷཱ་བྱཱཏ། ཁྲིཊ་དྷཱ་བ་ཡི་་བྱ་ཏི། ཁྲོང་…ཨ་དྷཱ་བ་ཡི་ཏ། སོགས་སོ། ་་། ཀྱི་༈པཱིཉ་ཚིམ་པ་ལ་ཎིཙ། ཉ་པཱིཉ་དག་གི་ནུཀ་འགའ་ཞིག་ཏུ་ཡང་ན་འོས༑ནུཀ་སྦྱར་བའི་ཕྱོགས་ལ། ལཊ་ཡཔཱི་ཎ་ཡ་ཏི། ལིང་པཱི་ཎ་ཡེཏ། ལོཊཔཱི་ཎ་ཡ་ཏིུ། །ལང་ཨ་པཱི་ཎ་ཡཏ། ལུང་ཨ་པཱི་ཎཏ། །ལིཊཔཱི་ཎ་ཡཾ་ཙ་ཀཱ་ར། །ལུཊཔྲ་ཎ་ཡི་ཏཱ། སྨོན་ལིང་པཱི་ཙྪཱཏ། ཁྲིཊ་…པཱི་ཎ་ཡི་བྱ་ཏི། ཁྲིང་…ཨ་པཱི་ཎ་ཏ། སོགས་དང༌། ་ནུཀ་མེད་པའི་ཕྱོགས་ལ། ཨཙའི་ཉཎ་འགྲོ་བ་ལ་འོས་འཕེལ་བ་བྱས་པས། པྲ་ཡི་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བསི། ལཊ་པྲ་ཡ་ཡ་ཏི། ལུང་ཨ་པི་པྲ་ཡཏི། སོགས་སོ། །ཀ་མ་ཏིའོ་་་་་་་༈ལཱིང་སྦྱོར་བར་བྱེད་པ་ལའོ་ཞེས་པ་ལ་ཎིཙ། བི་བྷཱ་ཥཱ་ལི་ཡ་ཏེ། ལི་ཡ་ཏིའི་འདམ་ཀའོ། །ལཱི་ཡ་ཏིའི་མཐའ་ཎིཙ་དང་ལྕཔ་ཕ་རོལ་ལ་ཨཱཏ་དུ་ཡང་ན་བསྒྱུར་བར་བྱ་སྟེ། ཨཱཏ་དུ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཨཏི་ཧྲི་སོགས་ཀྱིས་པུཀ་ཨཱ་ག་མ་སྦྱར་བས། ་ལཱ་པ་ཞེས་པ་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་ལཱ་པ་ཡ་ཏི། ལིང་ལཱ་པ་ཡེཏ། ལོཊ་ལཱ་པ་ཡ་ཏུ། ལང་ཨ་ལཱ་པ་ཡཏ། ་ལུང་…ཨ་་ལ་ལ་པཏ། ལིཊལཱ་པ་ཡཾ་ཙ་ཀཱ་ར། ལུཊལཱ་པ་ཡི་ཏཱ། སྨོན་ལིང་ལཱ་པཱུཏི། ཁྲིཊ་ལཱ་པ་ཡི་ཥྱ་ཏི། ཁྲིང་ཡཨ་ལཱ་ཡི་ཏ། །སོགས་སོ། །ལཱི་ཁྱོ་རྣཀ་ལུ་ཀཱ་བ་རྒྱ་ཏ་ར་ཧཱུཾ། ལཱི་ལཱི་དག་གི་ནུཀ་ལུཀ་དག་ཡང་ནའོ། །ལཱི་ཆུང་ངུར་གྱུར་པ་དང་ལི་སྦྱོར་བར་བྱེད་པ་དག་གི་སྣུམ་པ་བྱུག་པའི་དོན་ལ་རིམ་པ་བཞིན་ནུཀ་དང་ལུཀ་ཨཱ་——ལངས།
