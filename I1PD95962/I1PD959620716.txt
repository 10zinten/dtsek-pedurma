ཏང་ལ་ཙླི་སི་ཙར་བསྒྱུར་བའི་ས་འདྲེན་པ་ཁོ་ན་སྟེ། །སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀཱསྟ། ཟཨཀཱས་ཏྭཾ། ཨཀཱས་ཏ།  ི་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨཀཱསྟྲཱི། ཨཀཱ་སཱ་ཐཾ། ཨཀཱ་ནཱཾ། དྷ་ལའང་ངོས་ས་ཕྱིས། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨཀཱ་སི། ཨཀཱ་སཱ་ཧི། ཨཀྵ་སྨྲ་ཧི། ་ཡང་འགྲེལ་པར། ཙཀླུང་སྤང་བའམ་བོར་བའི་དོན་དང་འབྲེལ་པ་ན་ཁཱཉ་དང་ཀཱ་དག་ཏུ་འགྱུར་བ་མ་ཡིན་ཞིང་དེའི་ཚེ་ཏང་ཁོ་ན་འཇུག་གོ་ཞེས་བཤད་པས། ལྕཉ་སོགས་སུ་མ་བསྒྱུར་བའི་ཕྱོགས་ལ། བལ་དང་པོའི་ཨིཊ་སྦྱར་བས། ་ལུང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཙཀྐི། ཨ་ཙཀཱི་ཥཱ་ཏཱཾ། ཨ་ཙཀླུ་ཥ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཙཀྐི། ཨ་ཙཀཱི་ཥ་ཛཾ། ཨ་ཙཀྱི་དཾ། ཨ་ཙཀྐི་ནཱཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཙཀྐི་ཏི། ཨ་ཙཀྐི་ཥཱ་ཧི། ཨ་ཙཧྥ་ཧི། བཱ་ལི་ཊི། ་ལིཊ་ལ་ཡང་ནའོ། །ལིཊ་ཕ་རོལ་ལ་ཙཀྟིང་ཉིད་དུ་གནས་པ་དང་ཁྱུཉ་རུ་བསྒྱུར་བ་དང་ཀླུ་ཉ་རུ་བསྒྱུར་བ་རྣམས་་་ཡང་ན་བར་བྱ་སྟེ། ཙགླིང་ཉིད་དུ་གནས་པའི་ཕྱོགས་ལ་ཏང་རྣམས་སྦྱར་བས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཙཀྵེ། ཙ་ཙཀྵ་ཏེ། ཙ་ཙཀཱི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཙཀཱི་ཥ། ཙ་ཙཀྵ་ཐེ། ཙ་ཙཀྐི་དྷེ། ཙ་ཙཀྐིནྡྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཙ་ཙཀྵེ། ཙ་ཙཥཱ་བ་ཧེ། ཙ་ཙཀྐི་མ་ཧེ། ཙ་གླིང་ཁྱུཉ་རུ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་པས། །ཨ་ཏང་གི་་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཁྱོ། ཙ་ཁྱ་ཏུ། །ཙ་ཁྱུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཁྱི་ཐ། ཙ་ལྱཱ་ཐ། ཙ་ཁྱ་ཐུ། ཙ་ཏྱཱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ༑ཙ་ཁྱོ། ཙ་ཁྱི་བ། ཙ་ཁྱི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཁྱེ། ཙ་ཀྵ་ཏེ། ཙ་ཁྱི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙ་ཁྱི་ཥ། ཙ་ཁྱུ་ཐེ༑ཙ་ཁྱི་དེ། ཙ་ཁྱི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཁྱི། ཙ་ཁྱི་བ་ཧེ། །
