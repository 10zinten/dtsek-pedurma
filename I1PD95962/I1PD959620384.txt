སཾ་ཀཱ་མོ་ཡ་སྱ། གང་གི་ཡང་དག་འདོད། ཞེས་པ་ལ། །ཡང་དག་འདོད། ས་ཀཱ་མ། ཡང་ན། སཾ་ཀཱ་མ། གང་གི་ཡང་དག་འདོད། དེ་བཞིན་དུ། སཾ་མ་ནོ་ཡ་སྱ། གང་གི་ཡིད་ཡང་དག་པ། ཞེས་པ་ལ། ས་མ་ན། དང་བའི་ཡིད། ཡང་ན། །སཾ་མ་ན། .དང་བའི་ཡིད། ཨུ་ད་ཀ་མྱོ་དསཾ་ཛཱ་ཡཾ། ་མིང་ལ་ཨུ་ད་ཀའི་ཨུ་དའོ། །།མིང་གི་དོན་དང་འབྲེལ་པའི་ཨུ་ད་ཀ་ནི་ཨུ་ད་ཞེས་པར་འགྱུར་ཏེ། ཨུ་ད་ཀ་སོ་མེ་གྷ། ཆུ་ཡང་འདི་སྤྲིན་ཡང་ངོ༌། །ཞེས་པ་ལ། ཨུ་ད་མེ་གྷ། ཆུ་སྤྲིན། དེ་བཞིན་དུ། ལོ་ཧི་ཏ༔་སོ་ཨུ་ད་ཀ༄༅། དམར་པོ་ཡི་ཡང་འདི་ཆུ་ཡང་ངོ་། །།ཞེས་པ་ལ། ལོ་ཧི་ཏོ་ད། ཆུ་དམར་པོ། གྷི་ར༔་སོ་ཨུ་ད་ཀ༄༅། འོ་། མ་ཡང་འདི་ཆུའང་ངོ༌། །ཞེས་པ་ལ། ཀྟི་རོ་ད། འོ་ཆུ། པེ་ཥ་བཱ་ས་བཱ་ཧ་ན་དྷི་ཥཱ་ཙ། པེ་ཥ་བཱ་ས་བཱ་ཧ་ན་དྷི་རྣམས་ལ་ཡང་ངོ༌། །པེ་ཥ་ལ་སོགས་པ་འདི་རྣམས་དང་་་སྦྱོར་བའི་ཨུ་ད་ཀའང་ཡང་ན་ཨུ་ད་ཞེས་པར་འགྱུར་ཏེ། ཨུ་ད་ཀེ་ན་པེ་ཏཾ། ཆུ་ཡིས་ཕྱེ་མར་བྱས། ཞེས་པ་ལ། ཨུ་ད་པེ་ཊཾ། །ཡང་ན། ཨུ་ད་ཀ་བེ་ཧཾ། དེ་བཞིན་དུ། ཨུ་ད་ཀེ་བཱ་ས། ཆུ་ལ་གནས་པ། ཞེས་པ་ལ། ཨུ་ད་བཱ་ས། ཆུ་གནས། །ཡང་ན། །ཨུ་ད་ཀ་བཱ་ས། ཆུ་གནས། ་་ཨུ་ད་ཀ་སྲ་བཱ་ཧ་ན། །ཞེས་པ་ལ། ཨུ་ད་བཱ་ཧ་ན། ཆུ་ཐོབ་པ། ་ཡང་ན། ཨུ་ད་ཀ་བཱ་ཧ་ན། ཨུ་ད་ཀཾ་དྷི་ཡ་སྨིན། གང་ལ་ཆུ་གཏེར། ཞེས་པ་ལ། ཨུ་ད་དྷི། ཆུ་གཏེར། ཡང་ན། ཨུ་ད་ཀ་དྷི། ཆུ་གཏེར། ཨེ་ཀ་ཧ་ལཱ་ངོ་པཱུ་ར་ཡི་ཏ་བྱེཉརྱ་ཏ་ར་ཧཱུཾ། ་དང་པོར་ཧལ་གཅིག་པ་ལ་གང་བར་བྱེད་པ་ལ་ཡང་ནའོ༑:། གང་བའི་དོན་དང་འབྲེལ་པའི་ཨུ་ད་ཀ་དང་སྦྱོར་བའི་ཚིག་གྲོགས་དེ་ལ་ཚིག་དེའི་ཡི་གེ་དང་པོ་དེ་ཧལ་གཅིག་པ་སྟེ། ཡི་གེ་གཉིས་གསུམ་བརྩེགས་པ་མིན་ན་ཨུ་ད་ཀ་ཉིད་ཨུ་ད་རུ་ཡང་ན་འགྱུར་ཏེ། ཨུ་ད་ཀ་སྱ་པཱུ་ར་ཎི་ཀུནྣཾ། ཆུས་གང་བའི་་བུམ་པ། ཞེས་པ་ལ། ཨུ་ད་ཀུནྣཾ། ཆུ་བུམ། ཡང་ན། ཨུ་ད་ཀ་ཀུནྣཾ། ཆུ་བུམ། དེ་་བཞིན་དུ། ཨུ་ད་ཀ་སྱ་པཱུ་ར་ཎཾ་པཱ་ཏྲ། ཆུས་གང་བའི་སྣོད། ༼ཞེས་པ་ལ། །ཨུ་ད་བཱ་།
