སྒྲ་མདོ། ཏིཁ་དེ་ནི་ལྟ་བ་ང་ཚོཁ་ཎཀ་པེར་ན་བློ་ཁ་ཁ་སོགས་སོ། །ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དན། ད་དྲི་ཤ་ཏུ། ད་དྲྀ་ཤུ། བི་བྷཱ་ཥཱ་སྭ་ཛི་དྲི་ཤོསྲི་ཛི་དྲི་ཤ་དག་ལས་འདམ་ཀའོ། །སྭ་ཛ་རྣམ་པར་བཅད་པ་དང་དྲི་ཤིར་ལྟ་བ་དག་ལས་ཕ་རོལ་གྱི་ལེཊ་བསྒྱུར་བ་ཐལ་ལ་ཨིཊ་་་་སྟེར་བ་དང་མི་སྟེར་བ་གདམ་ང་ཅན་དུ་བྱ་སྟེ། ད་དཞི་ཐ། ད་དྲཥྛ། སྲི་ཛི་དྲྀ་ཤ་དག་ལ་ཛྷལ་ཀ་སོགས་ཀྱིས་ཨམ། བྲཉྫ་བྦསྔ་སོགས་ཀྱིས་ཤ་ཥ། ཁ་ཏུ་ལ་་འོས་ཐ་ཐ། སྐྱེས་བུ་དབུས་མའི་་གཉིས་ཚིག་དང་མང་ཚིག་ལ། ད་དྲི་ཤ་ཐུ། ད་དྲི་ཤ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ༑ད་དན། ད་དྲྀ་ཤི་བ། ་་_ད་དྲྀ་ཤི་མ། ལུཊ་ལ། སྭ་ཛི་དྲིཤ་དག་ལ་ཛྷལ་ཀ་འགྲོ་བ་མ་ཡིན་པ་ལ་ཨམ་འོས། ཨམ་ཨཱ་ག་མ་བྱིན་པས། དྲ། བྲལྔ་བྦསྔ་སོགས་ཀྱིས་ཤ་ཥ་ཥ་ཏུ་ལ་ཥ་ཏུ་འོས་ཏ་ཊ། དྲཏྟཱ་རོ། དྲཏྟཱ་རཿསོགས་སོ། ཊ། སྨོན་པའི་ལིང་ལ། དྲྀ་ཏ། དྲྀ་ཤཱནྟཾ། །ཛཾ་བཱ་སུསོགས་སོ། །ལྀཊ་ལ། ཨམ་ཨཱ་ག་མ་དང་། བྲཉྫ་བྷསྔ་སོགས་ཀྱིས་ཤ་ཥ། པོས་ལ་ཥ་ཌྷ་དག་གི་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ་སོགས་བྱས་པས། ་དྲpཏི། དྲགླུ་ཏ། དྲpནྟི་སོགས་སོ། །དེ་བཞིན་དུ། ལྡིང་ལ། ཨ་དྲགྱུཏ། ཨ་དྲགྲུ་ཏཱཾ། ཨ་དྲགླུན་སོགས་སོ། འི། ཡུན་པ། ༈དན་མཆེ་བ་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར། དན་སཉྫ་སཱཉྫ་སོགས་ཀྱིས་ཤཔ་ལན་ཕྱིས་པས། ད་ཤ་ཏི་སོགས་སོ། །ལིང་ལ། ད་ཤེཏ་སོགས་སོ། །ལོཊ་ལ། ད་ཤ་ཏུ་སོགས་སོ། །ལང་ལ། ཨ་ད་ཤཏ་སོགས་སོ། །ལུང་ལ། བྲཉྫ་བྷསྔ་སོགས་ཀྱིས་ཤ་ཥར་བསྒྱུར་བ་སོགས་བྱས་པས། ཨ་དཾ་གཱིཏ། ཨ་དཾ་ཏཱཾ། ཛྷལ་ལས་ཚུལ་ལ་འོས་སི་ཙ་ཕྱིས། ཥ་ཁུ་ལ་ཥ་ཏུ་འོས་ཏ་ཊ། ཁ་ཨ་དཾ་གླུསོགས་སོ། །།ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དན། ད་དན་ཏུ། ་ད་དནུ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་དནི་ཐ། ད་དཾ་། ད་དན་ཐུ། ད་དན། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ད་དན། ད་དནི་བ། ད་དནྟི་མ། —།
