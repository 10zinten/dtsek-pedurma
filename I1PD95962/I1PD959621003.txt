སྒྲ་མདོ། ཏི་pཀྱིན་མ་ཤེས་ན་བོད་ལ་ཤ་ས་བྷཱུ་ཏཱཏ། བོ་བྷཱུ་ཏྭཾ། བོ་ངྷ་བ་ཏུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བོ་ཧྥུ་ཧི། བོ་བྷུ་ཏཱཏ། བོ་དྷྭ་ཏྭཾ། བོ་བྷཱུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ་བོ་བྷ་བཱི་ནི། བོ་བྷ་བཱ་བ། བོ་བྷ་བཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ལུལ་ཨ་བོ་བྷ་བཱིཏ། ཨ་བོ་བྷོ་ཏ། ཨ་བོ་བྷཱུ་ཏྭཾ། ་་ཨ་བོ་བྷུ་བུ། སིཙ་ཨ་ལྟ་སོགས་ཀྱིས་ཛྷི་ཛུས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་ཨ་བོ་བྷ་བཱི། ཨ་བོ་བྷོ། ཨ་བོ་བྷཱུ་ཏཾ། ཨ་བོ་བྷུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བོ་བྷ་ཐཾ། ཨ་བོ་བླུ་བ། །ཨ་བོ་བྷུ་མ། ལུང་ལ། པགཱ་ཏི་སྦ་བྷཱུ་སཱ་བྷཱུ་སོགས་ཀྱི་སིཙ་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་་་་གསུམ་ལ། ཨ་བོ་བྷཱུ་ཏ། ཨ་བོ་བྷུ་ཏཱཾ། ཨ་བོ་བྷ་བུ། སིཙ་ཨ་བྷུསྟ་སོགས་ཀྱིས་ཛཱི་ཛུས། ཛུས་ལའང་ངོས་ཡོན་ཏན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ལྟཨ་བོ་བྷུ། ཨ་བོ་་་བྷུ་ཏཾ། ཨ་བོ་བྷཱུ་ཏི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བོ་བྷུ་ཁཾ། སྒྱུའི་ལུང་ལིཊ་་་དག་ལ་བྷའི་བུཀ་འོས་བུཀ་ཨཱ་ག་མ། ཨ་བོ་་བ། །ཨ་བོ་བྷུ་མ། ལོལིཊ་ལ་ཀས་་་སོགས་ཀྱིས་ཨཱམ་སྦྱར་བས། བོ་བྷ་ཛྭ་ཙ་ཀཱ་ར། བོ་བྷ་བཱ་བ་བྷཱ་བ། བོ་བྷ་བཱ་་མཱ་ས། སོགས་དང༌། འགྲེལ་པར་སྐབས་འགའ་ཞིག་ཏུ་ཡང་འཇིག་པ་ལ་ཨཱམ་མི་སྦྱོར་བ་ཡང་སྲིད་དོ་ཞེས་བྱུང་བས། ཨཱམ་མ་སྦྱར་བའི་ཕྱོགས་ལ། ྋ་ཨཙ་དང་པོ་རྣམས་ལ། བྷཱུའི་ལུང་ལིཊ་དག་ལ་བུཀ་འོས་བུཀ་ྋཨཱ་ག་མ་བྱིན་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བོ་བྷྱཱ་བ། བོ་ངྷ་བ་ཏུ། བོ་དྷྱ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བོ་དྷུ་བི་ཐ། བོ་དྷྭ་བ་ཐུ། བོ་དྷུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བོ་བྷཱ་བ། བོ་དྷྭ་བི་བ། བོ་བྷེ་བི་མ། ཡང་ན་བུཀ་མ་བྱིན་པར་ཎ་ལ་གཉིས་ལ་འཕེལ་བ་དང་། ཐལ་ལ་ཡོན་ཏན་གཞན་རྣམས་ལ། ཨ་ཙ་ལ་ཀླུ་བྱིངས་སོགས་ཀྱི་ཨུ་བར་བསྒྱུར་བས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བོ་བྷཱ་བ། བོ་བྷ་བ་ཏུ། བོ་ཐུ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བོ་བྷ་བི་ཐ། བོ་བྷུ་—R།
