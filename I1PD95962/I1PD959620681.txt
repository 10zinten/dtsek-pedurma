སྒྲ་མདོ། ༼ཏི་་ཏིཎི་ཡིས་ཁོ་ཚོ་ཁཚོཊ་ནང་ཏི་ལོ་རྟག་ཁྲིཊ་ལ་ཝལ་:དང་པོའི་ཨིཊ་བྱིན་པས། །བ་ཤི་ཥྛ་ཏི། ་་བ་ཤི་ཥྱ་ཏ། བ་ཤི་་ཉྩནྟི་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་བ་ཤི་་ཏ། ཨ་བ་ཤི་ཉྩ་ཏྭཾ། ཨ་བ་ཤི་སྦྱིན་སོགས་སོ། །ན་མ་སློབ་རྟག་ཏུ་ཤཔ། ས་ཀྱི་མཐུན་དུ་ཝ་ཏོ༈ཧ་ན་འཚེ་བ་དང་འགྲོ་བ་དག་ལའོ་ཞེས་པ་ལ་འང་། ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་འཇུག་ཅིང་ཤཔ་ཀྱང་འཇིག་སྟེ། ལ་ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཧནྟེ། །ཨ་ནུ་དཏྟོ་པ་དེ་ཤ་བ་ན་ཏི་ཏ་ནོ་ཏཱ་དཱི་ནཱ་མ་ནུ་ནཱ་སི་ཀ་ལོ་པ་ཛྷ་ལི་ཀཱི་ཏི། ་ཨ་ནུ་དཱཏྟ་:ཉེ་བའི་ཡུལ་བ་ནོ་ཏི། ཎཏ་ནོ་ཏི་ལ་སོགས་པ་་་་རྣམས་ཀྱི་རྗེས་སུ་སྣ་ལྡན་ཛྷལ་ཀ་ང་འགྲོ་བ་ལ་དབྱིའོ། །ཨ་ནུ་དཱཏྟ་ཉེ་བའི་ཡུལ་ཏེ། ཡ་མུ་ཉེ་བར་འགོག་པ། །ར་མུ་རྩེ་བ། ན་མ་རབ་ཏུ་འདུད་པ། །གཙླི་འགྲོ་བ། ཧ་ན་འཚེ་བ། མ་ནུ་ཤེས་པ་རྣམས་དང༌། བ་ནོ་ཏི་སྟེ། བ་ནུ་འཚེ་བ་དང་། ཏ་ནོ་ཏི་ལ་སོགས་པ་སྟེ། ་ཏ་ནུ་རྒྱས་པ། ཀླུ་ནུ་འཚེ་བ། རཾ་ནུ་འགྲོ་བ། ཤྲཱི་ཆུ་འབར་བ། བ་ནུ་སློང་བ། ཌམ་ནུ་རྟོགས་པ་རྣམས་ཀྱི་རྗེས་སུ་སྣ་ལྡན་གྱི་ཡི་གེ་རྣམས་རྐྱེན་ཛུལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་དང་པོ་ཅན་ཀ་དང་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། ་ོ་ཧན་ལ་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་སྦྱར་བས། ཧ་ཏ། མང་ཚིག་ལ། །གམ་ཧན་ཛན་སོགས་ཀྱིས་ཨུ་པ་དྷཱ་ཕྱིས། ཧནྟིའི་ཧིཊྛི་ཏ་རྣམས་ལ་འོས། ཧ་གྷར་བསྒྱུར་བས། གྷཱནྟི། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཧནི། ཧ་ཐ། ཧ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཧརྨི། ཧནྡ༑ཧམྨ། །བ་མོརྭ། བ་མོ་དག་ལ་ཡང་ནའོ། <། ཉེར་བསྒྱུར་ན་ཎར་བསྒྱུར་བའི་རྒྱུ་མཚན་ཅན་སྔོན་དུ་བྱུང་ན་ཧན་གྱི་ན་ཡིག་ནི་བ་ཡིག་དང་མ་ཡིག་་་དག་ཕ་རོལ་ལ་ཎ་ཡིག་ཏུ་ཡང་ནི་འགྱུར་ཏེ། །པྲ་ཧནྣཾ། རབ་ཏུ་འཚེ་བར་བྱེད། པྲ་ཧཎི། པྲ་ཧནྡ། པྲ་ཧཎྜསོགས་སོ། །ལིང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཧ་ནཱཏ། _།
