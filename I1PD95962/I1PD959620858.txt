སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཚོན་ནི། ཚོན་བ། ཚོན་མ། དེ་བཞིན་དུ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀྲྀ་ཎད། ཨ་མྲྀ་ཎཏ། ཨ་ཚིནྟཾ། ཨ་་ཚིནྡན༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཚོ་ཎ། *ཨ་ཚྭ་ཎཏ། *དེའི་འང་ངོས་ད་རུ་ར་རྣམ་བཅད། གནས་མེད་པ་ལ་ཡང་ན་འོས། ད་ཉིད་དམ་ཏ། ཨ་ཚེ་ཎ་ད། ཨ་ཙིནྟཾ། །ཨ་་ཚོནྟ༑ ིསྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཚེ་ཎ་དཾ། ཨ་ཛིན། ཨ་ཚི། ལུང་ལ༑ཨིར་འགྲོ་བ་སོགས་ཀྱིས་ལྕི་ཨང་དང་སིཙ་ཡང་ན་བར་བྱས་པས། ཨང་གི་ཕྱོགས་ལ། ཨ་མྲྀ་དཏ། ཨ་མྲྀ་ད་ཏཱ། ཨ་མྲྀ་ད་ན། སོགས་དང༌། སིཙའི་ཕྱོགས་ལ་ཨིཊ་བྱིན། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། །ཨ་ཚརཱི་ཏི། ་ཨ་ཚཇིཏྟཾ། ཨ་ཚཇི་། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་རེའི་ཨཏ་འོས་རི་ཨ་ཨ་བྷུ་སའི་ཙར་འང་ངོས། ཚ་ཙར་བསྒྱུར་བ་སོགས་ཀྱིས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་ཚརྡ། ཙ་ཚྭ་ད་ཏུ། ཙ་ཚོ་དུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙ་ཚཇི་་ཐ། ཙ་ཚོ་ད་ཐུ། ཙ་ཚིད། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙ་ཚརྡ། ཙ་ཚོ་དི་བ། ཙ་ཚྭ་དི་མ། ལུཊ་ལ་ཨིཊ་བྱིན། པུཀ་་་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཚཇི་ཏཱ། ཚྭཇི་ཏཱ་སི། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བྲྀཏྟཱཏ། ཀྲྀཏྭཱ་སྟི། ཚྭཅུ་སུ། སོགས་སོ། །ཁྲིཊ་ལ། ཀྲྀཏ་ཙཱཏ་ཀྲྀཏ་ཀྲྀཏ་ནཱིཏ་ལས་སིཙ་མ་ཡིན་པས་ས་ལ་འོས་ཨིཊ་ཡང་ན་བར་བྱས། པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཨིཊའི་ཕྱོགས་ལ། ཚཇི་ཥྛ་ཏི། སོགས་དང༌། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། ཚཉྩ་ཏི། སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་ལ། ཨ་ཚཇི་ཏ། །སོགས་སམ། ་ཨ་ཚརྒྱུཏི། སོགས་སོ། །དེ་ཞི༈ཨུ་ཏྲ་དིར་འཚེ་བ་དང་མ་གུས་པ་དག་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ཆོ་ག་ཐམས་ཅད་ཚོ་ད་དང་འདྲ་སྟེ། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། —།
