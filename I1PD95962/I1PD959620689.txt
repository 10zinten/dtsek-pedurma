སྒྲ་མདོ། །ཏི་ཁྱཁ་ཏི་ཙྪའི་ཀཏོག་ཀ་ན་ཀ་ཁ་ཁ་སུཥྛ་བུ། སུཉྩ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། སུཥཱ་བི་ཐ། །སུཥྛ་བ་ཐུ། །སུ་བ་ཐུ། སུཥཱ་བ། སུབྷུ་བ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སུཏྟཱ་བ། །སུཎྜ་བ༑་སུཥཱ་བི་བ། སུཉྩ་བི་བ། སུཥཱ་བི་མ། །སུམྦུ་བི་མ། ལུཊ་ལ་ཨིཊ་་་བྱིན་པས། སྣ་བི་ཏཱ། སྣ་བི་ཏཱ་སི་སོགས་སོ། །སྨོན་པའི་ལིང་ལ་ཀྲྀཏ་མ་ཡིན་པ་སོགས་ཀྱིས་རིང་པོ་བྱས་པས། སྣུ་ཡཱཏ། སྣུ་ཡཱནྟཾ། སྠཱ་ཡཱ་སུསོགས་སོ༑་། ལྀཊ་ལ་ཨིཊ་བྱིན་པས། སྣ་བི་ཥྛ་ཏི་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལའང་ཨིཊ་བྱིན་པས། ཨ་སྣ་བི་ཥྱ་ཏ་སོགས་སོ། །གནམ་ཀ་ར་བུ་ཊ་ཏུཎྜ་ཨིཎ་འགྲོ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། ་ཨ་དི་སོགས་་་ཀྱིས་ཤཔ་འཇིག། བྱིངས་ཀྱི་ཎ་རྗེས་འབྲེལ་བིཏ་ལ་ཡོན་ཏན་སོགས་བྱས་པས། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ། ་ཨེ་ཏི། ཨི་ཏ། ཨི་ཌོ་ཡཎ། ཨིཎ་གྱི་ཡཎའོ། །རྐྱེན་པི་ཏ་མ་ཡིན་པ་ཨཙ་དང་པོ་ཅན་ཕ་རོལ་ལ་ཨིཎ་ནི་ཡཎ་སོ་སོར་བསྡུ་བའི་ཡི་གེར་འགྱུར་ཏེ། སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། ཡནྟི༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨེ་ཎི། ཨི་ཐི། ཨི་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་མི། ཨི་བ། ཨི་མ། ལིང་ལ། ཨི་ཡཱཏ། ཨི་ཡཱ་ཏྭཾ། ཨི་ཡུསོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨེ་ཏུ། ཨི་ཏཱཏ། ཨི་ཏྭཾ། ཡནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨི་ཧི། ཨི་ཏཱཏ། ཨི་ཏཾ། ཨི་ཏ༑སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཡཱ་ནི། ཨ་ཡཱ་བ། ཨ་ཡཱ་མ། ལང་ལ། ཨཙ་དང་པོ་ལ་ཨཱཊ་འོས། བྱིངས་ལ་ཨཱཊ། ཨཱཊ་ལས་ཀྱང་ངོས་འཕེལ་བ་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨིཏི། ཨི་ཏཱཾ། ཨཱ་ཡན། ཨཱ་ལ་འཕེལ་བ་བྱས་པས་ཨི་ཞེས་པ་ལ་ལ་ཛྷ། ཛྷ་ཨནྟ། ཨིཏ་ཀྱིའང་ངོས་ཨི་ཕྱིས་བརྩེགས་པའི་མཐའ་དབྱི་འོས། ཏ་ཕྱིས། ་ཨེཙའི་ཨ་ལ་་་ཨཔ་སོགས་ཀྱིས་མཚམས་སྦྱར། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨི། ཨི་ཏཾ། ཨི་ཏ། —།
