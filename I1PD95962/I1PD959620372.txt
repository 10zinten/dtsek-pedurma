བདྷིཤྩ༑གསལ་བ་དང་སྒྲོ་མདོངས་དང་། ཞེས་པ་ལ། གསལ་བ་སྒྲོ་མདོངས་དག། ཨིནྡྷ་བནྟི་ཏཱི། བཱཀ་ཙ་མ་ན། ངག་དང་ཡིད་དང༌། ཞེས་པ་ལ། ངག་ཡིད་དག། ཝདྨ་ན་སོ། སྦྲང་རྩི་དང་མར་གསར་དང་། མ་དྷུ་ཙ་སཟིཉྩ། ཞེས་པ་ལ། མ་དྷུ་སརྦ་ཏཱི། སྦྲང་མར་གསར་དག། དད་ཅན་མ་དང་བློ་ལྡན་མ་དང་། ཤཱདྷཱ་ཙ་མེ་དྷཱ་ཙ། ཞེས་པ་ལ། ཐཱངྒཱ་མེ་བྷེ། དད་ཅན་མ་བློ་ལྡན་མ་དག། དཀར་ཕྱོགས་དང་ནག་ཕྱོགས་དང༌། ཤུ་ཀླ་ཀྲྀ༄༅། །ཞེས་པ་ལ། ཤུཀླ་ཀྲྀ། དཀར་ཕྱོགས་ནག་ཕྱོགས་དག། ཨཱ་རྱ་ཨ་བ་སཱ་ནཾ་ཙ། ཐོག་མ་དང་ཐ་མ་དང༌། ཞེས་པ་ལ། ཨཱ་ཏྱཱ་བ་སཱ་ནེ། ཐོག་མཐའ་དག། ཡ་ཛཾཙ་ཨ་ནུ་བཱ་ཀྱི་ཙ། མཆོད་པ་དང་རྗེས་སུ་བརྗོད་པ་དང་། ཞེས་པ་ལ། ཡ་ཛཱ་ནུ་པཱ་ཀྱེ། མཆོད་པ་རྗེས་སུ་བརྗོད་པ་དག་། ཨུ་ལུ་ཁ་ལཾ་ཙ་མུ་ཤ་ལཾ་ཙ། གཏུན་དང་གཏུན་་་་ཤིང་དང༌། ཞེས་པ་ལ། ངས་…ཨུ་་ལུ་ཁ་ལ་མུ་ཤ་ལེ། གཏུན་གཏུན་ཤིང་དག། འཕྲོག་བྱེད་དང་བརྒྱ་བྱིན་དང༌། ཧ་རིཉྩ་བ་ས་ཝཱཉྩ། །ཞེས་པ་ལ། ་ཨོ། ཧ་རི་བཱ་ས་ཝོ། འཕྲོག་བྱེད་བརྒྱ་བྱིནདག། བྲཧྨ་ཙ་པྲ་ཛཱ་པ་ཏི། ཚངས་པ་དང་སྐྱེ་དགུའི་བདག་པོ་དང༌། ཞེས་པ་ལ།  ོ་བྲཧྨ་པྲ་ཛཱ་པ་ཏཱི། ཚངས་པ་དང་སྐྱེ་དགུའི་བདག་པོ་དང༌། ཤི་བཉྩ་བི་ཤྲ་བ་ཎ། ཞི་བ་དང་རྣམ་ཐོས་སྲས་དང་། ཞེས་པ་ལ། ཤི་བ་བི་ཤ་བ་ཎོ། ཞི་བ་རྣམ་ཐོས་སྲས་དག། སོ་མཉྩ་རུ་དྲ། ཟླ་བ་དང་དྲག་པོ། ཞེས་པ་ལ། །ཟླ་བ་དྲག་པོ་དག། སོ་མ་རུ་དྲོ། ན་རཱ་ད༔་པརྦ་ཏ༄༅། དྲང་སྲོང་མི་སྦྱིན་དང་རི་བོ་དང་། ཞེས་པ་ལ། ན་ར་ད་པརྦ་ཏོ། མི་སྦྱིན་རི་བོ་དག། སཱུརྨཐུ་ཙནྡྲ་མ། …ཉི་མ་དང་ཟླ་བ་དང་། ཞེས་པ་ལ། སཱུརྱ་ཙནྡྲ་མ་སོ། .ཉི་ཟླ་དག། རཱ་མཐུ་ལཀྵ་ཎཉྩ། དགའ་བྱེད་དང་མཚན་ཉིད་ལྡན་དང་། ༽ཞེས་པ་ལ། རཱ་མ་ལགླུ་ཎོ། .དགའ་བྱེད་མཚན་ཉིད་ལྡན་དག། བྷི་་མ་་་ཨརྫུ་ན། འཇིགས་པ་དང་སྲིད་སྒྲུབ་དང་། ཞེས་པ་ལ། བྷི་མཱརྫུ་ནོ། འཇིགས་པ་སྲིད་སྒྲུབ་དག། ཀམྤ་ལཉྩ་ཨ་ཤྭ་ཏ་ར༄༅། སྣམ་བུ་དང་རྟ་དང་། ཞེས་པ་ལ། ཀམྤ་ལཱ་ཤྭ་ཏ་རོ། སྣམ་བུ་རྟ་དག། དེ་དག་ད་དྷིའི་སོགས་ཁོང་ཡིན་ནོ། །ཨཱ་ནང་རཾ་ཏཱཾ་ཏཱཾ་དྭེ། ཟླས་དབྱེ་ལ་རྀཏ་རྣམས་ཀྱི་ཨཱ་ནང་ངོ༌། ཞ། ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འདིར་ཚིག་སྔོན་མའི་མཐའ་—།
