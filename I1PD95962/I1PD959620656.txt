པ་དང་འདྲའོ། །ཡིན་ཝཱ། ༈བྷ་ཛ་བསྟེན་པ་ལའོ་ཞེས་པ་གཉིས་ཀའི་ཚིག་ཅན་ཡིན་པས་ལཊ་་་སོགས་པ་རསྨི་པ་ད་རྣམས་དང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་གཉིས་ཀ་འཇུག་སྟེ། ལཊའི་ཨ་ཏང་ལ། བྷ་ཛ་ཏི་སོགས་དང་། ཏང་ལ། བྷ་ཛ་ཏེ་སོགས་སོ། །དེ་བཞིན་དུ། ལིང་གི་ཨ་ཏང་ལ། བྷ་ཛེ་ཏ་སོགས་དང་། ཏང་ལ། བྷ་ཛེ་ཏ་སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། བྷ་ཛ་ཏུ་སོགས་དང༌། ཏང་ལ། །བྷ་ཛ་ཊཱ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ཨ་བྷ་ཛ་ཏ་སོགས་དང་། ཏིང་ལ། ཨ་བྷ་ཛ་ཏ་སོགས་སོ། །ལུང་ལ་ཙུའི་ཀུ་འོས། ཛ་ག་ཁར་ལ་ཙར་འོས། ་ག་ཀ། ཛྷལ་ལས་ཛྷལ་ལ་འོས་རྣམ་དབྱེ་ཛྷལ་དང་པོ་རྣམས་ལ་སིཙ་འཇིག། ཛྷལ་མ་ཡིན་པ་རྣམས་ལ་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས༑ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྷཱཀཱིཏ། བད་བླ་སོགས་ཀྱིས་འཕེལ་བ། ཨ་བྷཱཏྟཾ། ཨ་བྷ་ཀླུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །་ཨ་བྷཱ་ཨཱི། ཨ་བྷཱཏྟཾ། ཨ་བྷཱཀྟ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་བྷཱཏྟཾ། ཨ་བྷཱཀླུ། །ཨ་བྷཱ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྷཱཀྟ། ཨ་བྷཱཀྵ་ཏྭཾ། ཨ་་་བྷཱཀྵཏ༑སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བྷཱཀྐུ། །ཨ་བྷཱཀྐཱ་ཐཱ། ཨ་བྷཱཧཱུཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བྷཱཀྐི། ཨ་བྷཱཀྵ་ཧི། ཨ་བྷཱཀྵ་ཧི། ལིཊ་ལ། བྱིངས་གཉིས་བརྗོད་ཀིཏ་རྣམས་ལ། །ཏི་ཕ་ལ་བྷ་ཛ་ཏྲཔ་ལས་ཀྱང་ངོས་། །ཨ་་བྷཱ་ས་དབྱི་བ་དང་ཨེར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་་་དང་པོ་གསུམ་ལ། བ་བྷཱ་ཛ། བྷེ་ཛ་ཏུ། བྷེ་ཛུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བྷེ་ཛི་ཐ། བྷ་བྷཀྵ། བྷེ་ཛ་ཐུ། བྷེ་ཛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ༑་བ་བྷཱ་ཛ། །བ་བྷ་ཛ། བྷེ་ཛི་བ། བྷེ་ཛི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་——།
