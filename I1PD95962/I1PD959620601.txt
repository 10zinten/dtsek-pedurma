བསྟན་འགྱུར། སྒྲ་མདོ། ཏིའི། ིའི་རི་མོ་ངོ་ཚོཀ་ཀ་ལ་ཀ་ཏ་འོས༑ཨིཊ་དང་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ཏ་རི་ཥྱ་ཏི། ཏ་རི་ཥྛ་ཏ། ཏ་རི་ཥཱནྟི་སོགས་སམ། ཡང་ན། ཎཏ་རཱི་ཥྛ་ཏི་ཏ་རཱི་ཥྛ་ཏ། ཏ་རཱི་ཉྩནྟི་་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ། ཨ་ཏ་རི་ཏ། ཨ་ཏ་རི་་ཏྭཾ། ཨ་ཏ་རི་ཥྱན་སོགས་སམ། ཡང་ན། ཨ་ཏ་རཱི་ཥྛཏ། ་ཨ་ཏ་རཱི་ཥྱ་ཏཱཾ། ཨ་ཏ་རཱི་ཥྱན་སོགས་སོ། །དན་སཉྫ་སཱཏྟཾ་ཤ་པི། དན་སཉྫ་སཱཉྫ་རྣམས་ཀྱི་ཤཔ་ལའོ། །བྱིངས་དན་མཆེ་བ་ཥལྔ་ཆགས་པ། ་ཥཱཉྫ་ཡོངས་སུ་འཁྱུད་པ་རྣམས་ཀྱི་ན་ཡིག་ནི་རྐྱེན་ཤཔ་ཕ་རོལ་ལ་དབྱི་སྟེ། ལ་། ་་་་་་ཡརཾ། ཏ། dཎ་པ་ཀརྒྱུད་::༈ ིཥལྔ་ལ། ལཊ་ཤཔ་བྱིངས་ཀྱི་དང་པོའི་ཥ་ས་དང༌། །མདོ་འདིས་་ཉ་ཕྱིས་པས། ས་ཛ་ཏི་སོགས་སོ། །དེ་བཞིན་དུ། ལིང་ལ། ས་ཛེ་ཏ་སོགས་སོ། །ལོཊ་ལ། ས་ཛ་ཏུ་སོགས་སོ། །ལང་ལ། 《ཨ་ས་ཛཏ་སོགས་སོ། །ལུང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། བད་བྲཛ་སོགས་ཀྱིས་འཕེལ་བ། ཙུའི་ཀུ་འོས། །ཛ་ག། ཁར་ལ་ཙར་འོས། ་ག་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ་བྱས་པས། །ཨ་སཱི་ཀི་ཏ། གཉིས་ཚིག་ལ། ::ཙུའི་ཀུ་འོས། ཛ་ག་ཛྷལ་ལས་ཛྷལ་ལ་འོས། སིཙ་ཕྱིས། །ཁར་ལ་ཙར་བྱས་པས། ཨ་སཱཏྟཾ། མང་ཚིག་ལ། ཙུའི་ཀུ་དང་། ཁར་ལ་ཙར། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ཨ་སཾ་གླུ། ཆོ་ག་དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་་ཨ་སཾ་ཁཱི། ཨ་སྭཱ་ཏྟཾ། ཨ་སཾཀྟ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཧཱུཾ་ཧཱུཾ། ཨ་སཾ་གླུ། ཨ་སཾ། ལིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ས་སཾ་ཛ། ས་སཾ་ཛ་ཏུ། ས་སཾ་ཛུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ས་སཾ་ཛི་ཐ། ས་་་སཾཧྥ༑ས་སཾ་ཛ་ཐུ། ས་སཾ་ཛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ས་སཾ་ཛ། ས་སཾ་ཛི་བ། ས་སཾ་ཛི་མ། །ལུཊ་ལ། ཙུའི་ཀུ་དང༌། ཁར་ལ་ཙར་བྱས་པས། གབསམ་—།
