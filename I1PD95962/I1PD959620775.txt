ཟླ་ཚོ་ཚོ་ལ་ཎ། དམ་པ་ཚོས་ཨ་མཱ་ས་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་མཱམྦྷ། །ཨ་མཱ་སཱ་ཤྩ། མ་ཨ་མཱ་དཾ༑ཡདྷ་ལའང་ངོས་སིཙ་ཕྱིས། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་མཱ་སི། ཨ་མཱ་སཱ་ཧི། ཨ་མཱ་སྨ་ཧི། ལིཊ་ལ། །ཨིཊ་ལའང་ཨཱཏ་ཀྱིས་དབྱི་འོས་ཨཱ་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། མ་མེ། མ་མཱ་ཏེ། ་མ་མི་རེ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། མ་མི་ཏེ། མ་མཱ་ཐེ། །མ་མི་དྷེ། མ་མི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །མ་མེ། །མ་མི་བ་ཧེ། མ་མི་མ་ཧེ། །ལུཊ་ལ། མཱ་ཏཱ། མཱ་ཏཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། མཱ་སཱི། མཱ་སཱི་ཡཱནྟཾ། མཱ་སཱི་རན། །སོགས་སོ། །ཁྲིཊ་ལ༑མཱ་སྱ་ཏེ། སོགས་དང་། ཁྲིང་ལ། ཨ་མཱ་སྱ་ཏ། སོགས་སོ། །༈ཨོ་ཧཱང་འགྲོ་བ་ལའོ་ཞེས་པ་ལའང་ཏང་རྣམས་སྦྱོར་ཏེ། ཤཔ་བླུ་འོས། བྱིངས་གཉིས་བརྗོད། ་ཀུ་ཧ་དག་གི་ཙུ་འོས། ཧ་ཛྷ། ཨ་བྱཱ་སའི་ཙར་འང་་་ངོས་ཛྷ་ཛ། ཤྲཱིཉ་རྣམས་ཀྱི་ཨིཏ་འོས། ཨ་བྷྱཱ་ས་ཨིཏ། ཀ་ང་འགྲོ་བའི་མིང་་ཅན་ཧལ་དང་པོ་རྣམས་ལ། །སྒྱུ་མ་ཡིན་པས་ཧལ་ལ་ཨཱི་འོས། མཐའ་ཨཱི་དང་། ཨཙ་དང་པོ་རྣམས་ལ། །བྷཱ་ཨ་བྷཱསྟ་སོགས་ཀྱིས་ཨཱ་ཕྱིས་པས། ་ལཊའི་སྐྱེས་བུ་་་དང་པོ་གསུམ་ལ། །ཇི་ཧཱི་ཏེ། ཛི་ཧཱ་ཏེ། ཛི་ཧ་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛི་ཧཱི་ཥ། ཛི་ཧཱི་ཐེ། ་ཛི་ཧཱི་དེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛི་ཧེ། ཛི་ཧཱི་བ་ཧེ། ཛི་ཧཱི་མ་ཧེ། ལིང་ལ། །མཐའ་མ་ཡིན་པ་སོགས་ཀྱིས་སཱི་ཡུཊའི་ས་ཕྱིས། །བྷ་སོགས་ཀྱིས་ཨཱ་ཕྱིས་པས། ཛི་ཧཱི་ཏ། ཛི་ཧཱི་ཡཱ་ཏྭཾ། ཇི་ཧཱི་རན། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛི་ཧཱི་ཏཱཾ། ཛི་ཧཱ་ཏཱ། ཛི་ཧ་ཏྭཾ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛི་ཧཱི་ཥ། ཛི་ྋཧཱ་བྷཾ། ཛི་གྷི་དྷི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛི་ཧི། ཛི་ཧཱ་བ་ཧི། ཛི་ཧཱ་མ་ཧི། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཛི་ཧཱི་ཏ། ཨ་ཛི་ཧཱ་ཏྭཾ། ཨ་ཛི་ཧ་ཏེ། སྐྱེས་བུ་ཏྭཾ།
