ཨ་བཱ་རིཏ། ཨ་བཱ་རིཏྟཾ། ཨ་བཱ་རི་ཐུ། སོགས་སོ། །ལིང་སི་ཙོ་རཱཏྨ་ནེ་པ་དེ་ཥཱ། ཀ་ལིང་སིཙ་དག་གི་ཨཏྨ་ནེ་པ་ད་རྣམས་ལའོ། །ཕྲོང་ཀུན་ནས་གུས་པ་་་དང༌། བྲྀཉ་སྒྲིབ་པ་དག་དང་། རྀཏ་མཐའ་རྣམས་ལས་ཕ་རོལ་གྱི་སིཙ་དང་་སྨོན་པའི་ལིང་ཨཱཏྨ་ནེ་པ་ད་ཅན་ལ་ཨིཊ་ཡང་ན་འགྱུར་ཏེ། བྲྀཉ་ལ་ལུང་གི་ཏང་རྣམས་སྦྱར། ཨིཊ་བྱིན་པའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བ་རིཥྛ། ཨ་བ་རི་ཥཱ'ཏྭཾ། །ཨ་བ་རི་ཥ་ཏ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བ་རིཥྛ། ཨ་བ་རི་ཥཱ་ཐཱ། ཨ་བ་རི་ཧཱུྃ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བ་རི་ཏི། ཨ་བ་རི་ཥཱ་ཧི། པཨ་བ་རི་ཧྥ་ཧི། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། སིཙ་ཀིཏ་ཡིན་པས་ཡོན་ཏན་དང་འཕེལ་བ་གང་ཡང་མི་བྱེད། ཛྷལ་དང་པོ་རྣམས་ལ་གཟུགས་ཐུང་ངུ་ལས་སོས། སིཙ་ཕྱིས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བྲྀ་ཏ། ཨ་བྲོ་ཥཱ་ཏཱཾ། ཨ་པྲྀ་ཥ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཤྲཱི་ཐཱ། ཨ་བྲྀ་ཥཱ་ཐཱ། ཨ་མྲྀ་དྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀྲྀ་ཎི། ་ཨ་པྲྀ་ཥཱ་ཧི། ཨ་མྲྀ་ཧྥ་ཧི། ཡང་ན། ཁཤྲི་རིཏ་ལས་ཡང་ན་འོས། ཨཱིཊ་བྱིན་པས། ཨ་བ་རིབྦ༄༅། །ཨ་བ་རཱི་ཥཱ་ཏྭཾ། ཨ་བ་རཱི་ཥ་ཏ། །སོགས་ཀྱང་ལུགས་གཅིག་གོ། །ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། །ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་བཱར། ས་བ་བྲ་ཏུ། ་བ་བྲུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་བ་བརླ། །བ་བ་རི་ཐ། ་བ་བྲ་ཐུ། བ་བྲ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་བཱ་ར། བ་བ་ར༑་བ་བྲོ་བ། ་བ་བྲོ་མ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་བྲེ། བ་བཱ་ཏེ། །དབ་བྲི་རེ། སྐྱེས་བུ་དབུས་མ་ཅིབ་བྲྀ་ཏེ། བ་གྲྭ་ཐེ། བ་བྲིདྡྷེ། ཥསྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་བྲེ། བ་ཛྲ་བ་ཧེ། བ་ཛྲ་མ་ཧེ། ལུཊ་ལ། །བྲོ་རིཏ་ལས་ཡང་ན་འོས། །ཨིཊ་དང་ཨིཊ་ཡང་ན་བར་བྱས་པས། སམ་ས་མཔས། །
