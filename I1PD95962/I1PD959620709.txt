ལ་བར་བསྟན་འགྱུར། སྒྲ་མདོ། གཏི?ཀུན་ཏེ་སྐྱེ་བར་ཤོག་ང་ཚོས་ད་ནི་བང་ཅིན་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། ཛཱ་གརཱཏ། ཛུ་གནཱ་སིཾ། ཛུ་གཆུ་སུ། སོགས་སོ། །ཁྲིཊ་ལ། ཛཱ་ག་རི་ཥྛ་ཏི། སོགས་སོ། །ལྡིང་ལ། ཨ་ཛཱ་ག་རི་ཥཱཏ༑་སོགས་སོ། །དཔལ་གྱི་བ། དུ་བ་དང་ཁ་ར་ཤིང་དང་ས་ཏཱཏ། །ན་བཀར་༈ད་རི་དྭ་ངན་འགྲོ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་འཇུག་སྟེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ད་རི་དྭ་ཏི། ཨིནྡ་རི་དྲ་སྱ། ད་རི་དཱའི་ཨིཏའོ། །བྱིངས་ཐམས་ཅད་པ་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཧལ་དང་པོ་ཕ་རོལ་ལ་ད་རི་དཱའི་ཨཱཏ་ནི་ཨིཏ་དུ་འགྱུར་ཏེ། ལཊའི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་ལ། ད་རི་དྲི་ཏ། ་སྣ་བྱཱསྟ་ཡོ་རཱ་ཏ། སྣ་ཨ་བྱཱསྟ་དག་ཨཱཏ་ཀྱིའོ། །བྱིངས་ཐམས་ཅད་པ་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཨཙ་དང་པོ་ཕ་རོལ་ལ་རྐྱེན་ཟླ་དང་བྱིངས་ཨ་་་་བྱཱསྟའི་མིང་ཅན་རྣམས་ཀྱི་ཨཱཏ་དབྱི་སྟེ། །་ལཊའི་སྐྱེས་བུ་དང་པོའི་མང་ཚིག་ལ། ལ། ཛཱི། ཨ་བྱཱསྟ་ལས་ཨཏ་འོས་ཛྷ་ཨཏ། མདོ་འདིས་ཨ་ཕྱིས་པས། ད་་རི་དྲ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་རི་དཱ་སི། ད་རི་དཱི་ཐ། ད་རི་ངྒི་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་རི་དཱ་མི༑ད་རི་དྲི་བ། །ད་རི་དྲི་མ། །ལིང་ལ། ད་རི་དྲའི་ཨིཏ་འོས། ཨི་་རུ་བསྒྱུར་བས། ད་རི་དྲི་ཡཱཏ། ་་ད་རི་དྲི་ཡཱ་ཏྭཾ། ་ད་རི་དྲི་ཡུ། །སོགས་སོ། །ལོཊ་ལའང་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཧལ་དང་པོ་རྣམས་ལ་ཨཱཏ་ཨིཊ་བསྒྱུར་བ་་་དང་ཨ་ཙ་དང་པོ་རྣམས་ལ་དབྱི་བ་སོགས་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་རི་དྭ་ཏུ། ད་རི་དྲི་ཏཱཏ། ད་རི་དྲི་ཏཱཾ། ད་རི་དྲ་ཏུ། སྐྱེས་བུ་དབུས་མ་་གསུམ་ལ། ད་རི་དྲི་ཧི། ད་རི་དྲི་ཏཱཏ། ད་རི་དྲི་ཏཾ། །ད་རི་དྲི་ཏི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་རི་དཱ་ཎི། ད་རི་དྲྭ་བ། ད་རི་དེ་མ། དེ་བཞིན་དུ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ་ཨ་ད་རི་དྲིཏ། ་ཨ་ད་རི་དྲི་ཏྭཾ། །ང་ཨ་ད་་—།
