ལ་ད་ནི་ཨད་ལ་སོགས་པ་བྱེད་པ་པོའི་དོན་གྱི་ཤཔ་འཇིག་པའི་བྱ་བ་རྣམས་་བསྟན་པར་བྱ་སྟེ། ཨ་དི་པྲ་བྷཾ་ཏི་ཤ་པ། །ཨ་ད་ལ་སོགས་པ་རྣམས་ལས་་་་ཤཔ་ཀྱིའོ། །བྱིངས་ཨཏ་ཟ་བ་ལའོ་ཞེས་པ་ལ་ངོ་བོའི་རྐྱེན་ཨི་སྦྱར་བའི་ཨ་རི་ལ་སོགས་པ་རྣམས་ལ་བྱེད་པ་པོའི་དོན་གྱི་ཤཔ་འཇིག་སྟེ། །ས། །ཏ་ད་འབུལ། ༈ཨད་ལ་ལཊ་སོགས་ཨ་ཏང་རྣམས་སྦྱར། །བྱེད་པ་པོ་ལ་ཤཔ་འོས་ཤཔ་སྦྱིན༑། མདོ་འདིས་ཤཔ་ཕྱིས་པས་ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨཏྟི། ཁར་ལ་ཙར་འོས་ད་ཏ། ཨཏྟ། ཨ་དནྟི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨཏི། ཨཏྠ། ཨཏྠ༑སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨངྒི། ཨདཱ། ཨཏྨ། ལིང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བྱཱཏ། ཨ་ཏྭ་ཏྭཾ། ཨ་དྷྱ། ཨུས་ལ་ཚིག་མཐའ་མིན་པའི་འོས་གཉིས་དག་གིས་གཅིག་ཕ་རོལ་གྱི་གཟུགས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་དྷྱ། ཨ་ཏྭ་ཏྭཾ། ་ཨ་་ཏྭ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ཧཱུཾ། ཨ་དྷྱ་བ། ཨ་ཅུ་མ། །ལོཊ་་་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨཏུ། ཨཏྟཱཏ། །ཨཏྟཾ། ཨ་དནྟུ། སྐྱེས་བུ་་་དབུས་མའི་གཅིག་ཚིག་ལ། ཧུ་ཛྷ་ལྕོ་ཧེ་དྷི། ཧུ་ཛྷལ་རྣམས་ལས་ཧིའི་དྷིའོ། །ཧུ་སྦྱིན་པ་དང་བྱིངས་མཐའ་ཛྷལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་རྣམས་ལས་ཕ་རོལ་གྱི་ཧི་ནི་དྷི་རུ་འགྱུར་ཏེ། ཨདྷི། ཨད་ལ། ལཊ་ཤཔ། ཨ་དི་སོགས་ཀྱིས་ཤཔ་ཕྱིས། ལ་སི་པ་སིའི་ཧི་སྟེ་་་་འགྲོ་བའང་མ་ཡིན་ནོས། སི་ཧི་རུ་བསྒྱུར་མདོ་འདིས་ཧི་དྷི་ང་དྲངས། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨཏྟཾ། ཨཏྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་དཱ་ནི། ལ་མིཔ། མིའི་ནི་འོས་མི་ནི་ཕྱི་མ་ལ་ཨཱཊ་སྟེ་པིཏ་འང་ངོས་ཨཱཊ་སྦྱིན། ཨ་དཱ་བ། ལ་བས་དང་མས་སུ་བསྒྱུར། ལང་དང་འདྲ་བ་སོགས་ཀྱིས་ས་ཕྱིས། ཕྱི་མ་ལ་ཨཱཊ་སྟེ་བིཏ་འང་ངོས། ཨཱཊ་ད་དྲངས། ཨ་དཱ་མ། །ཨ་དསརྦ་ཏྭཾ། ་ཨཱད་ལས་ཐམས་ཅད་ཀྱིའོ། །ཨཱད་ལས་བྱིངས་ཐམས་ཅད་ཧལ་གཅིག་པ་ཨཔྲཀྟའི་མིང་ཅན་དེ་ལ་ཨཊ་ཨཱ་ག་མ་སྟེར་བ་སྔོན་གྱི་དྲང་སྲོང་ཐམས་ཅད་འདོད་དེ། ལང་གི་སྐྱེས་བུ་_།
