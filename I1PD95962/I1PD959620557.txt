སུ་མ་ནོན་ན་གཙོ་བསྟན་འགྱུར། སྒྲ་མདོ། *ཏིའི::ཀ་ཝི་ནི་ཤོ་ཎཙོ་ཎ་ཌnལུཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཛི་ཏཱ། ཨ་ཛི་ཏཱ་རོ། ཨ་ཛི་ཏཱ་རཿསྐྱེས་བུ་་་དབུས་མ་གསུམ་ལ། ཨ་ཛི་ཏཱ་སི། ཨ་ཛི་ཏཱ】ཨ་ཛི་ཏཱསྦྲ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཛི་ཏཱསྨིཾ། ཨ་ཛི་ཏཱ་སྭཨ་ཛི་ཏཱ་ཨ་ཛ་བཱི་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་ཏཱ། དེ་ཏཱརོ། །བེ་ཏཱརཿསྐྱེས་བུ་་དབུས་མ་གསུམ་ལ། བེ་ཏཱ་སི། བེ་ཏཱསྦྱ། །བེ་ཏཱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་ཏཱསྨི། བི་ཏཱ་སཱབེ་ཏཱདྲང་སྲོང་བྷཱ་ར་དྷཱ་ཛའི་འདོད་པ་ལྟར་ཨིཊ་་་སྦྱར་ན། བ་ཡི་ཏཱ། བ་ཡི་ཏཱ་རོ། བ་ཡི་ཏཱ་རཿསོགས་སོ། །ང་། སྨོན་པའི་ལིང་་ལ་བྱ་བ་ལྷག་པོ་མེད་དེ། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བཱི་ཡཱཏ། བཱི་ཡཱཙཱ། བཱི་ཡཱ་སུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བཱི་ཡཱབཱི་ཡཱནྟཾ། ཤྲཱི་ཡཱསྟ། ་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །བཱི་ཡཱ་སཾ། །བཱི་ཡཱ་སཱ། བཱི་ཡཱསྨ། ཨ་ཛ་བཱི་རུ་མ་བསྒྱུར་་་བའི་ཕྱོགས་ཀྱི་ཁྲིཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །་ཨ་ཛི་བྱ་ཏི། །ཨ་ཛི་ཥྱ་ཏཨ་ཛི་ནྟི། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཛི་ཏྱཱ་སི། ཨ་ཛི་བྱ་ཐ། ཨ་་ཛི་བྱ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཛི་བྱཱ་མི། །ཨ་ཛི་བྷཱ་བཨ་ཛི་བྷཱུ་མཨ་ཛ་བཱི་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བེ་ཥྛ་ཏི། བེ་ཥྱ་ཏབེ་ཊྛནྟི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བེ་ཀྵ་སི། །བེ་བྱ་ཐ། ་་བེ་ཥྱ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བེ་བྷྱཱ་མི། བེ་བྷྱཱ་བབེ་༔་མབྷ་ར་དྷྭ་ཛའི་་་འདོད་པ་ལྟར་ཨིཊ་སྦྱར་ན། བ་ཡི་བྱ་ཏི། བ་ཡི་བྱ་ཏ། བ་ཡི་ཥྛནྟི་སོགས་སོ། །ཨ་ཛ་བཱི་རུ་མ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཁྲིང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨཱ་ཛི་ཥྛ་ཏ། ཨཱ་ཛི་ཥྱ་ཏཱཾ། ཨཱ་ཛི་ཏྱཱ་ན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨཱ་ཛི་ཥྱ། །ཨཱ་ཛི་ཊུ་ཏཾ། ཨཱ་ཛི་ཥྛ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨཱ་ཛི་ཏྱཾ། ཨཱ་ཛི་ཉྩ་བ། ཨཱ་ཛི་བྷཱུ་མ། ཨ་ཛ་བཱི་རུ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་བེ་—།
