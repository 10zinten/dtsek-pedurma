སྒྲ་མདོ། ཏིས་བློཊྛི་ཏུ་སྐྱོ་མགོན་མཚོ་རྒྱལ་བ་ལྔ་ལ་ཁྲི་ལྔ་བློ་གཱ་སི་ཥཱ། ལིཊ་ལ། ཛ་གོ། ཛ་ག་ཏུ། ཛ་གུསོགས་སོ། །ཨེ་རྨི་ང༌། ལིང་ལ་ཨེའོ། །བྱིངས་གླུའི་མིང་ཅན་རྣམས་དང༌། མཱ་མཆོད་པ། ཀྵ་འགྲོ་བ་ལས་ལྡོག་པ། གི་སྒྲ། པཱ་བཏུང་བ། ཨོ་ཧཱ་ཀ་གཏོང་བ། ཁོ་མཐའི་ལས་་་རྣམས་ཀྱི་མཐའ་ནི་སྨོན་པའི་ལིང་ཀ་ང་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨེ་རུ་འགྱུར་ཏེ༑གི་ལ། སྨོན་པའི་ལིང༌། ༼ཤ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་ཨེཙ། ཨཱཏ། མདོ་འདིས་ཨེ། གེ་ཡཱཏ། གི་ཡཱནྟཾ། གེ་ཡཱ་སུསོགས་སོ། །ཁྲིཊ་ལ༑གཱ་སྱ་ཏི་སོགས་སོ། །ཁྲིང་ལ། ཨ་གཱ་སི་ཏ། སོགས་སོ། །ཡང་༈ཤྲྰི་སྒྲ་དང་ཀུན་ཏུ་འཚེ་བ་དག་ལའོ་ཞེས་པའི་བྱིངས་ལའང་ལཊ་་་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར། ཆོ་ག་སྔ་མས་རིགས་འགྲེ་སྟེ། སུཔ་ཀྱི་བྱིངས་ཉྩི། ་་ཥ་སོགས་ཀྱིས་ཥ་སར་མི་འགྱུར། བྷྱཱ་ཡ་ཏི་སོགས་སོ། །ལིང་ལ། །བྱཱ་ཡེཏ་སོགས་སོ། །ལོཊ་ལ། སྒྱུ་ཡ་ཏུ་སོགས་སོ། །ལང་ལ། ཨཥྛ་ཡཏ། སོགས་སོ། །ལུང་ལ། ཨཊྚ་སཱིཏ། ཨཥྛ་སིདྡྷཾ། ཨཥྛ་སི་སོགས་སོ། །ལིཊ་ལ། ཏཏྠོ། ལྕི་ལ། ལིཊ་ཤ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་མཐའ་ཨཱ་ཨ་དྷྱ་ས་མེད་པ་སོགས་ཀྱིས་བྱིངས་གཉིས་བརྗོད་ཤར་སྔོན་མའི་ཥ་ཡ་འོས་ཥ་དང་ཡ་ཕྱིས་རྒྱུ་མཚན་མེད་ན་རྒྱུ་མཚན་ཅན་ཡང་མེད་དོ་ཞེས་པས་ཥ་ཕྱིས་པས་ཋ་ཡང་ཐར་ལོག་ཨ་བྷཱུ་སའི་ཙར་འང་ངོས་ཐ་ཏལ་ཎལ་ཨཱཏ་ལ་ཎལ་གྱི་ཨོ་འོས་ཎལ་ཨོ་མཚམས་སྦྱར། ཏཥྛ་ཏུ། ཏཏྟུསོགས་སོ། ཡ་། ལུཊ་ལབྷཱུཏཱ༑བྷཱུཏཱ་རོ། ཏཱ་ར། སྨོན་པའི་ལིང་ལ། གཞན་བརྩེགས་པ་སོགས་ཀྱི་ཡང་ནའོ་ཞེས་པའི་མཐའ་ཨེ་རུ་འགྱུར་བ་དང་མི་འགྱུར་བ་ཡང་ན་བར་བྱས་པས། བྷཱུ་ཡཱཏ། བྷཱུ་ཡཱནྟཾ། བྷཱུ་ཡཱ་སུསོགས་སམ། ཡང་ན། ལྕེ་ཡཱཏ། ཏྠེ་ཡཱཙཱ། བྱེ་ཡཱ་སུསོགས་སོ། །ཁྲིཊ་ལ། སྒྱུ་སྱ་ཏི་སོགས་སོ། །ལྡིང་ལ། ཨ་སྱཏ་སོགས་སོ། །ཏཚོ༈བྲ་དག་པ་ལའོ་ཞེས་པའི་བྱིངས་ལའང་ལཊ་སོགས་པ་རསྨི་པ་ད་་། ཅུཏཱ།
