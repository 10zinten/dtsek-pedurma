ཙགྱི་ནོ་ལས་ཀྱང་ཊཙའི་རྐྱེན་འཇུག་སྟེ། །ཨདྷཾ་ནཱ་བ། ཞེས་པ་ལ། རྣམ་དབྱེ་བསྡུས་ཊཙ་བྱིན། ཨཌྷ་ནཱ་པཾ། གྲུ་ཕྱེད། ཁཱརཱཔཱ་ཙཱ། ཁཱ་རཱི་ལས་ཤར་པ་་་རྣམས་ཀྱིའོ། ་་། ཨརྡ་དང་གྲངས་སྔ་མའི་ཁཱ་རཱི་ལས་ཊཙ་འཇུག་པར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་རྣམས་ཡང་ན་འདོད་དེ། དྭེ་ཁཱཚོ། ཁལ་དག་གཉིས་དག། ཞེས་པ་ལ། རྣམ་དབྱེ་ཕྱིས་ཊཙ་ཡང་ན་བར་བྱས་པས། དྭེ་ཁཱ་རཾ། ཁལ་གཉིས། དྲི་ཁཱ་རཱི། ཁལ་གཉིས། དེ་བཞིན་དུ། ཨདྷཾ་ཁཱ་རཱི། །ཞེས་པ་ལ། ཨཌྷ་ཁཱ་རཾ། ཁལ་ཕྱེད། ཨཌྷ་ཁཱ་རཱི། ཁལ་ཕྱེད། དྲི་ཏྲི་བྷཱུ་མཉྫ་ལེ། དྲི་ཏྲི་དག་ལས་ཨཉྫ་ལི་ལས་སོ། །དྲི་ཏྲི་སྔ་མའི་ཨཉྫ་ལི་ལ་ཊཙ་ཡང་ན་འགྱུར་ཏེ། དྭེ་ཨཉྫ་ལཱི། སྙིམ་པོ་དག་དོ་དག། ཞེས་པ་ལ། རྣམ་དབྱེ་ཕྱིས། ་ཊཙ་ཡང་ན་བར་བྱས་པས། །དྲུཌཱ་ལཾ། སྙིམ་པོ་དོ། དྲུཉྫ་ལི། སྙིམ་པོ་དོ། དེ་བཞིན་དུ། །སྙིམ་པོ་རྣམས་གསུམ་རྣམས། ཏི་སྲཨཉྫ་ལ་ཡ། ཏྲི་ལ་ཛས། བུད་མེད་ལ་ཏྲེ་ཙ་ཏུར་དག་གི་ཏི་སྲིཙ་ཏ་སྲི་འོས་བསྒྲུབས་པའོ། །ཞེས་པ་ལ། །ཏུཉྫ་ལཾ། ཏུཉྫ་ལི། བྲཧྨ་ཎོ་ཛཱ་ན་པ་དཱ་ཀྵ་ཡཾ། །ལ་བྲཧྨན་ལས་ཡུལ་བརྗོད་པ་ལའོ། །ཡུལ་དང་འབྲེལ་པའི་་་་བྲཧྨན་ལ་ཊཙ་འགྱུར་ཏེ། །སཱ་རཱཟླབྲཧྨ། ཞེས་པ་ལ། །རྣམ་དབྱེ་ཕྱིས། ་ཊཙ་བྱིན་པས། སཱ་རཱཟླ་བྲཧྨ། རང་ཡུལ་ལ་ཚངས་པ། ཀུ་མ་ཧནྟུ་མ་རྱ་ཏ་ར་ཧཱུཾ། ཀུ་མ་ཧཏ་་དག་ལས་ཡང་ནའོ། །ཀུ་དང་མ་ཧཏ་སྔ་མའི་བྲཧྨན་ལ་ཊཙ་ཡང་ན་འགྱུར་ཏེ། ཀུ་བྲཧྨ༑ཡཚངས་པ་ཆུང་དུ། ཀུ་བྲཧྨ། ཚངས་པ་ཆུང་དུ་། དེ་བཞིན་དུ། མ་ཧཱ་བྲཧྨ། ཚངས་པ་ཆེན་པོ། མ་ཧཱ་བྲ། ཚངས་པ་ཆེན་པོ། ཏ་ཀྲོ་པ་པ་དཾ་སཔ་མིཙྩཾ། དེ་ལ་བདུན་པ་། གནས་པ་ལ་ཉེ་བའི་ཚིག་གོ། ་། །བསྡུ་བའི་བཞི་ཚིག་གཉིས་བསྒྲིགས་པའི་སྔ་མ་་་རྣམ་དབྱེ་བདུན་པ་གནས་པ་དེ་ལ་ཉེ་བའི་ཚིག་ཅེས་མིང་འདོགས་བྱེད་དོ། །ཨུ་་པ་པ་ད་མ་ཏིང་། ཉེ་བའི་ཚིག་ལ་ཏིང་མ་ཡིན་པ་ལའོ། །གོང་དུ་ཉེ་བའི་ཚིག་ཅེས་མིང་འདོགས་བྱས་པ་དེ་ལས་ཕ་རོལ་ཏུ་ཏིང་མཐའ་ཅན་མ་ཡིན་པའི་ཚིག་གང་།
