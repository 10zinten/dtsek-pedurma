ལ་་་་་ལས་པར་ཏའི་ཁ་ན་ཏརྟོགས་བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་མོ་ཆེ་ནི་གན་ངོ་མ་མི་ཏ་བྷོ་ཎགཉེན་ཡིན་ནོ་རེ་ཞིག་རྫོགས་སོ། །me། ཏི་ནས་འད། ཁོ་རང་་ཁ་བ་བབ་དང་མི་ཁོང་ར་་ང་བ་། »ལ་པཚོ་%ཆ་ཅུ་པ། ཏི་ཊི་ན་ད་ནི་ཧལ་གྱི་མཐའ་ཅན་ཏེ་གསལ་བྱེད་མཐའ་ཅན་ལས་དང་པོར་ཕོ་རྟགས་རྣམས་བཤད་པར་བྱའོ། །དེ་ཡང་དང་པོ་ཧ་མཐའ་ཅན་ལ། ལི་ཧ་མྱོང་ཞེས་པ་ལ༑ཧོ་ཌྷ། ཧ་ཡི་ཌྷའོ། །ཛྷལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཕ་རོལ་ན་ཡོད་་་་པའི་ཚུ་རོལ་གྱི་ཧ་སྲོག་མེད་དང་ཚིག་མཐའི་ཧ་སྲོག་མེད་ནི་རྟ་རུ་འགྱུར་རོ། །ན། །ལིཧྲ༑སུ། ཧ་ཡི་ཌྷ་འོས། ཧ་ལྷ། ཧལ་དཱིཔ་ཨཱཔ་རིང་པོ་རྣམས་ལ་ཞེས་པས། སུ་ཕྱིས། ་གནས་མེད་པ་ལ་ཡང་ན་འོས། ལིཌ། །ལིཊ། ཨོ་དང་ཛས་ལ། ཨ་ཙ་དམན་པ་ཕ་རོལ་དུ་རབ་ཏུ་སྦྱོར་རོ། །ཞེས་པས་དྲངས།  ོལི་ཧོ༑ལི་ཧ། གཉིས་པ་རྣམས་ལ། ལི་ཧཾ། ལི་ཧོ། ལི་ཧ། ཀྵ་ལ། ལི་ཧཱི། བྷཾ་ལ། ཧ་ཡི་ཌྷི། ཛྷཤ་ལ་ཛྷལ་རྣམས་ཀྱི་ཛཤ་རྣམས་སོས། རྡོ་ཌ་དྲངས། ལིཤྩ། བྷི། ལིནྟི། ངེ་ལ། ལི་ཧེ། ཧཱུཾ་བྷལ། ལིཙྩཾ། ལིནྡྷ། ང་སི་ལ། ལི་ཧ། བྷཾ་བྷུལ། ལིཤྩ། ལཟླུ། ངས་ལ། །ལི་ཧ༑ཨོས་ལ། ལི་ཧོ། །ཨཱམ་ལ། ལི་ཧཱུྃ། ཇི་ལ། ལི་ཧི། ཨོས་ལ། །ལི་ཧོ༄༅། སུཔ་ལ། ཧ་ཡི་ཌྷ་འོས་ཧ་ལྷ། ཁར་ལ་ཙར་འོས། །ཌྷ་ཊ་དྲངས། ལིg༑་བོད་ཚིག་རྣམས་ཀྱང་རྣམ་དབྱེ་དང་པོ་དང་འདྲ་བས། །ཧེ་ལིཊ། ཧེ་ལིཌ། ཧེ་ལི་ཧོ། ཧེ་ལིཧ། དུ་ཧ་འཇོ་བ་ཞེས་པ་ཅུང་མི་འདྲ་སྟེ། དུ་ཧ། སུ༑དྭ་དེདྷཱ་ཏོདྦྷ། ད་སྔ་མའི་བྱིངས་ཀྱི་གྷའོ། །ད་སྔོན་ན་ཡོད་པའི་་་་བྱིངས་ཀྱི་ཧ་ཡིག་སྲོག་མེད་ནི། ཛྷལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་གང་རུང་ཕ་རོལ་དུ་བྱུང་བ་དང་ཚིག་མཐའ་ལ་ཡང་གྷ་རུ་འགྱུར་ཏེ། ཧ་གྷར་འགྱུར། དུགྲྭ། ་སུ། །ཨེ་ཀཱ་ཙོ་བ་ཤོ་བྷཥ་ཛྷ་ཊཾ་ཏ་མྱནྡྷེ། ཨཙ་གཅིག་པའི་:བཤའི་བྷཥ་ཛྷཥ་མཐའི་ས་—།
