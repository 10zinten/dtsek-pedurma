སྒྲ་མདོ། ཏིལ་བི་ཎི་ནི་ས་མོ་གཉི་ག་སྦྱོར་ཀན་ལས་བློ་ལ་ལ། །བེ་བྷི་དཱ་ནི། བེ་བྷི་དཱ་བ། བེ་བྷི་དཱ་མ། དེ་བཞིན་དུ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བེ་བྷི་དཱིཏ། ཨ་བེ་བྷེཏ། །ཨ་བེ་བྷིཏྟཾ། ཛཱི་ཛུས་ཨ་་བེ་བྷི་དུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་བེ་བྷི་དཱི། ཨ་བེ་བྷིཏ། །ཨ་བེ་བྷིཏྟཾ༑་ཨ་བེ་བྷིཏྟ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་བེ་བྷི་དཾ། ཨ་བེ་བྷི་དྲོ། ཨ་བེ་བྷིདྨ། ན་དྷཱ་ཏུ་ལོ་བ་ཨཱཛྙཱ་ནྡྲིད་ཏུ་ཀེ། བྱིངས་ཕྱིས་པའི་བྱིངས་ཕྱེད་པ་ལ་མ་ཡིན་ནོ། །ཡང་ཕྱིས་པ་ལྟ་བུ་བྱིངས་ཀྱི་ཆ་ཤས་ཕྱིས་པའི་ཕ་རོལ་ཏུ་བྱིངས་ཕྱེད་པ་ཡོན་ཏན་དང་། འཕེལ་བའི་རྒྱུ་མཚན་ཅན་བྱུང་ཡང་བྱིངས་དེ་ལ་འཕེལ་བ་་་དང་ཡོན་ཏན་སོགས་མི་བྱེད་དེ། ་ལ་ཡིག་ལྷག་མ་རྣམས་ལ། ་༼ལུང་ཨ་བེ་བྷི་་་་དཱིཏ༑ལིཊ་བེ་བྷི་དཾ་ཙ་ཀཱ་ར། ལུཊ་བེ་བྷི་དི་ཏཱ། སྨོན་པའི་ལིང་བེ་བྷི་དྷཱཏ། ཁྲིཊ་བེ་བྷི་དི་ཥྱ་ཏི། ཁྲིང་ཨ་བེ་བྷི་དི་ཏ། སོགས་སོ། །བྲྀ་ཏུ་འཇུག་པ་ལའོ་ཞེས་པ་ལ། ཡང་སྦྱར། གཉིས་བརྗོད། །ཏ་ཕྱིས། རུ་གྲི་ཀོ་ཙ་ལུ་ཀི། འཇིག་པ་ལ་རུཀ་རིག་དག་ཀྱང་ངོ༌། །བྱིངས་རྀཏ་ཨུ་པ་དྷཱ་ན་ཡོད་པའི་ཨ་བྷྱཱ་ས་ལ་ཡང་འཇིག་པ་ཕ་རོལ་ལ་རུཀ་ཨཱ་ག་མ་དང་རིག་ཨཱ་ག་མ་དང་འང་གི་སྒྲས་རཱིཀ་་ཨཱ་ག་མ་རྣམས་འགྱུར་ཏེ། །གློ་བུར་བ་དེ་རྣམས་ཀྱི་ཀ་གནས་དོན་དང་རུཀ་གི་ཨུ་བརྗོད་པའི་དོན་ཏེ། བདྲྀཏ་འམ། བ་རི་བྲྀཏ་འམ། བ་རཱི་བྲྀཏ། ཅེས་པ་གསུམ་པོ་གང་རུང་སྒྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ་ཨཱིཊ་ཡང་ན་བར་བྱས་པས། བཛྲ་ཏཱི་ཏི། བརྔནྟིགཉིས་ཚིག་ལ། བཀྲྀཏྟ༑མང་ཚིག་ལ། །ཛྷ་ཨཏ་དུ་བསྒྱུར་བས། བཛྲ་ཏ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བཛྲ་ཏཱི་ཎི། བརྔནྟི༑་་བཛྲཏྠ། བཛྲ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བཛྲ་ཏཱི་མི། བཅཎྚི༑བཛྲ་ཏཱ། བཛྲp། ལིང་ལ། བཛྲ་ཏྭཱཏ། བཛྲ་ཏྭ་ཏྭཾ། བཛྲ་acesalcerD་。。།
