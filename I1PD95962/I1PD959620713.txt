སྒྲ་མདོ། ཏིལ་སྒོ་ཆེ་ནི་ལོ་དྷི་ཚེ་ཎཡའ་ལོ་བློ་ལ་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཤཱསྶེ། ཤྲཱི། ཤིཥྛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཤཱསྨི༑ཤི་ཥ། ཤིཀྵ། ལིང་ལ། ཤི་བྷཱུཏ། ཤི་བྷཱུ་ཏྭཾ། ཤི་ཉྩ། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཤཱསྟུ། ཤིཏྟཱཏ། ཤིཙྪཱ། ཤཱ་ས་ཏུ། ལོཊའི་སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ།  ི་ཤཱ་ཧོ། ་ཧི་ལ་ཤཱའོ། །ཤཱས་ནི་ལོཊ་བསྒྱུར་བའི་ཧི་ཕ་རོལ་ལ་ཤཱ་ཞེས་པར་འགྱུར་ཏེ། ཤཱ་སུ་ལ། ལོཊ་་་ཤཔ་འཇིག། ཁྲ་སིཔ། སི་ཧི་འོས། ་སི་ཧི། མདོ་འདིས་ཤཱས་ཤཱ་རུ་བསྒྱུར། འདི་ལ་གཞི་མཉམ་པ་ལས་མ་གྲུབ་པ་ཞེས་པས་སྔོན་མ་སའི་རྒྱུ་མཚན་ལས་ཧི་ཡང་ཧུ་ཛྷལ་ལ་ཧི་དྷི་འོས་ཧི་དྷིར་བསྒྱུར་བས། ་ཤཱ་དྷི། ཡང་ན་ཧི་ཏཱ་ཏང་དུ་བསྒྱུར་བས། ཤིཥྛ། གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཤིཊཾ། ཤི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ལ་ཤཱ་སཱ་ནི། །ཤཱ་སཱ་བ། ཤཱ་སཱ་མ། ་་ལེང་གི་སྐྱེས་བུ་དང་་་་པོའི་གཅིག་ཚིག་ལ། །ལ་ཏིཔ། ཨིཊ་ཀྱིའང་ངོས་ཨི་ཕྱིས། ཨསྟི་མ་ཡིན་པའི་ཏི་ལ་འོས། །སྲ་དྲ། ཧལ་དཱིཔ་སོགས་ཀྱིས་ཏྲ་ཕྱིས། །གནས་མེད་པ་སོགས་་་ཀྱིས་ཨ་ཤཱད། ཨ་ཤཱཏ། །གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨ་ཤིཏྟཾ། ཨ་ཤཱ་སུ། སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། སིཔ་ལ་བྱིངས་ཀྱི་རུ་ཡང་ན་འོས་སྲ་རུ་དང༌། ད་ཡང་ན་བར་བྱས་པས། ཨ་ཤཱ། ཨ་ཤཱཏ། ཨ་ཤཱད། །གཉིས་ཚིག་དང་མང་ཚིག་ལ། ཨ་ཤིཏྟཾ། ཨ་ཤིཊྚ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ྋཨ་ཤཱ་སཾ། ཨ་ཤི་ཥ། ཨ་ཤི། ལུང་ལ། །སརྟི་ཤཱསྟི་ཨརྟི་རྣམས་ལས་ཀྱང་ངོས། ཙྪི་ཨང་དུ་བསྒྱུར། ཤཱས་ཀྱིས་ཨིཏ་ཨང་ཧལ་དག་ལ་འོས། ཨུ་པ་དྷཱ་ཨིཏ་དུ་བསྒྱུར། །ཤཱ་སི། གྷ་སི་བ་སི་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། ཨ་ཤི་ཥཏ། ཨ་ཤི་ཥ་ཧཱུཾ། ཨ་ཤི་ཥ་ན། སོགས་སོ། །ལིཊ་ལ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཤ་ཤཱ་ས། །ཤ་ཤཱ་ས་ཏུ། །་ཤ་ཤཱ་སུ། སྐྱེས་བུ་oat།
