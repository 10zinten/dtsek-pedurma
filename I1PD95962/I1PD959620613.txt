སྒྲ་མདོ། ཏིའི་ཀ་པོན་ཌོ་ཎའི་ཚོར་ཎཁ་བློ་ཌཚོ་nཥཱཋི་ཏཱི་བ་ཧི། །ཥཥི་ཏི་མ་ཧི། པྲྀཊ་ལ། སྱ་དང་ཝལ་དང་པོའི་ཨིཊ་སྦྱིན་པས། ཥཱཀྐི་ཥྛ་ཏེ། ཥཥི་བྱེ་ཏེ། །ཥཱཥི་ཏྱནྟེ་སོགས་སོ། །དེ་བཞིན་དུ། ཨ་མྲིང་ལ། ཨ་ཥཥི་་ཏ། ཨ་ཞི་བྱེ་ཏྭཾ། ཨ་ཥཥི་ཏྱནྟ་སོགས་སོ། །དp:། ཞི་ཊཱིཀ༈རི་ཛ་འགྲོ་བ་ལ་སྟེ་གནས་པ་ལའང་ངོ་ཞེས་པ་ལ། ལཊ་སོགས་ཨཱཏྨ་་ནེ་པ་ད་རྣམས་སྦྱར་བའི་ཆོ་ག་སྔ་མ་དང་འདྲ་སྟེ། ལཊ་ལ། ཨརྫ་ཏེ། ཨརྗེ་ཏེ། ཨརྫནྟེ་སོགས་སོ། །ལིང་ལ། ཨརྗེ་ཏ། 《ཨརྗེ་ཡཱ་ཏྭཾ། པཨརྗེ་རན་སོགས་སོ། །ལོཊ་ལ། ཨརྫ་ཏྭཾ། །ཨརྗེ་ཏྭཾ། ་ཨརྗེ་ནཾ་སོགས་སོ། །ལང་ལ། ཨཱརྫ་ཏ། ་ཨཱཛེ་ཏཱཾ། ཨཱརྫནྟ་སོགས་སོ། །ལུང་ལ། བལ་དང་པོའི་ཨིཊ་དང་ཙྩི་སིཙ་རུ་བསྒྱུར་བ་སོགས་བྱས་པས། ་ཨཱཛིཋ། འོཨཱརྫི་ཥཱཏྭཾ། །ཨཱརྫི་ཥཏ་སོགས་སོ། །ལིཊ་ལ། བྱིངས་གཉིས་བརྗོད། རིའི་ཨཏ་འོས། རཾ་ཨ་དང་། དང་པོའི་ཨཏ་ཀྱི་རིང་པོ་དེ་ལས་ནུཊ་སྟེར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཁོ་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨཱ་ཚྭ་ཛེ། ཨཱ་ནཱི་ཛཱ་ཏེ། ཨཱ་ཚྭ་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨཱ་ཛྲ་ཛི་ཏེ། ཨཱ་ནཱི་ཛཱ་ཐེ། ཨཱ་ཚྭ་ཛི་དྡྷེ། ཨཱ་མྲ་ཛི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨཱ་ཚྭ་ཛེ། ཨཱ་ཚྭ་ཛ་བ་ཧེ། ཨཱ་མྲ་ཛི་མ་ཧེ། །ལུཊ་ལ། །ཝལ་དང་པོའི་ཨིཊ་དང་ཏཱས་སྦྱོར་བ་སོགས་བྱས་པས། ཨརྫི་ཏཱ། ཨརྫི་ཏཱ་རོ། ཨརྫི་ཏཱ་ར། ཨརྫི་ཏཱ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ::ཨརྫི་ཏི། ཨརྫི་ཏཱི་ཡཱནྟཾ། ཨརྫི་ཏཱི་རན་སོགས་སོ། །ལྀཊ་ལ། ཝལ་དང་པོའི་ཨིཊ་དང་སྐྱ་ཏཱས་དག་ཚིགས་ལུཊ་དག་ལ་འོས་སྱ་སྦྱར་བས། ཨ་བྱ་ཏེ། ཨརྫི་བྱེ་ཏྭཾ། ཨརྫི་ནེ་སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་ལ། ཨཱརྫི་་ཏ། །ཨཱརྫི་ེ་ཏྭཾ། ཨཱརྫི་བྱཱནྟ་སོགས་སོ། །།།།་ལ་ཁ་ཚུམདོས། ངའི་བུ་དབྱིག་དང་། ༈ཏི་ཛ་རྣོ་བ་ལ་སྟེ་བཟོད་པ་ལ་ཡང་ངོ་ཞེས་པ་ལ། གུསྟི་ཛ་ཀི་ཏ་རྣམས་—།
