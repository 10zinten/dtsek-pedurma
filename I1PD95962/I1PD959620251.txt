བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་འ་སྐྱོ་བའི་ཚོ་ཁྱ་བླ་མ་ཀརྣམས་ལ། སཱ་པི།  ོསྲ་པོ། སཱམྦྷ། བོད་ཚིག་རྣམས་དང་པོ་དང་འདྲ། །མེད་པ་༈ས་ཡིག་མཐའ་ཅན། དྷ་ནུས་ལ། …གཞུ། སུ། ་སུ་ཨམ་ཕྱིས། །ས་ས་ཛུཥ་སོགས་ཀྱིས་དྷ་ནུ། ཨོ་ལ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས། །ས་ཥ། དྷ་ནུ་ཏྟི། ཛས་ལ། །དྷ་:ཧཱུྃ་ཏིལ། །གཉིས་པ་རྣམས་ཀྱང་འདྲ། ཊཱ་ལ། །དྷ་ནུ་ཥཱ། བྷཱུམ་བྱུས་འདྲ། ས་ས་ཛུཥ་སོགས་ཀྱིས་ས་ར། དྷ་ནུཤྩ། དྷ་ནུབྷི། ངེ་ལ། དྷ་ནུ་ཥཱ། བྷཱུམ་བྱུས་འདྲ། ང་སི་ལ། ་དྷ་ནུ་ཥ། བྷཱུམ་བྱས་འདྲ། དྲུག་པ་རྣམས་ལ། །དྷ་ནུ་ཥ། །དྷ་ནུ་ཁོ། ༼དྷ་ནུ་ཏཱཾ། བདུན་པ་རྣམས་ལ། དྷ་ནུ་ཏི། དྷ་ནུ་ཥོ། །དྷ་ནུམྱུ། བོད་ཚིག་རྣམས་རྣམ་དབྱེ་དང་པོ་དང་འདྲ། ཧ་བིས་དང་པོ་ཡས་སོགས་ཀྱང་དྷ་ནུས་དང་འདྲའོ། །།བོད་ཀྱི་སྐྱཁན༈ས་ཡིག་མཐའ་ཅན། །ཨ་དེས་ལ། །སུ། མ་ནིང་ལ་སུ་ཨམ་ཕྱིས། ས་ས་ཛུཥ་དག་གིས་རུ། ཨ་ད། ཨོ་ལ། ཁྱད་སོགས་ཀྱིས་ཨེ། ཨ་ལ་་ཡོན་ཏན། ཨ་དེ། ཨ་ངས་ཀྱི་ང་ལ་ཨུ་སྟེ་དའི་འང་མ་འོས། ཨ་མཱུ། །ཛས་ལ༑་་ཨ་མཱུ་ནི། གཉིས་པ་རྣམས་ཀྱང་དང་པོ་དང་འདྲ། ཊ་ནས་བཟུང་་་རྣམ་དབྱེ་ལྷག་མ་རྣམས་ཀྱི་ཆོ་ག་ཐམས་ཅད་ཨ་ངས་སྐྱེས་པའི་རྟགས་ཀྱི་སྐབས་སུ་བཤད་པ་དང་འདྲའོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱཇ་པུ་ས་ཀ་ལིངྒ་ཧ་ལནྡྷས་མཱ་སྟཱ། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་མ་ནིང་གི་རྟགས་ཧལ་མཐའ་ཅན་རྣམས་རྫོགས་སོ། །་། །རྒྱུ་མ་དང་། །ཞུ་ཚ་བོ་R། ཁ་བ་། ད་དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་་་གཞུང་འགྲེལ་གྱི་གོ་དོན་དཔེར་བརྗོད་ཆོ་ག་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་་་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས་མ་ནིང་གི་རྟགས་ཧལ་མཐའ་ཅན་རྣམས་རྣམ་དབྱེ་དང་ཕྲད་པའི་ཆོ་ག་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པ་་་།
