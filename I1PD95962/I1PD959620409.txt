ཀ་བ་ཀ་ཀུ་བ་ས་་ཞབསྟན་འགྱུར། སྒྲ་མདོ། ཏིཁ་ཚའི་ཉི་ཟླས་ལོ་མང་ཀ་ཅོལ་ཆབྷ་བློ་སྒོ་ཆ་ཀ་འགྱུར་ཏེ། །ཨེ་ཏྲི་དང་ནི་རྟི་ལ་མདོ་འདིས་ཟཀ་འཕེལ་བ་ལྷ་ཨེ་ཡ་གང་གི་ཨིཏ་ལའང་ངོས་ཨི་ཕྱིས། ཨཱ་ཏྲེ་ཡ༑རྒྱུན་ཤེས་ཀྱི་བུ། ནི་དྷེ་ཡ། ངེས་འཛིན་གྱི་བུ། ཀ་ལྱཱ་ཙྪཱ་དཱི་ནཱ་མི་ནང་། ཀ་ལྱཱ་ཎཱི་་་ལ་སོགས་པ་རྣམས་ཀྱི་ཨི་ནང་ངོ༌། :། ཀ་ལྱཱ་ཎཱི་ལ་སོགས་པ་རྣམས་ཀྱི་བུ་བརྒྱུད་ལ་ཆ་ཀ་འགྱུར་ཞིང་ཀ་ལྱཱ་ཎི་སོགས་ཀྱི་མཐའ་ཨི་ནང་དུའང་འགྱུར་ཏེ། ཀ་ཝཱ་ཎཱི་དང་ཛུ་བྲི་རྟི་ལ་མདོ་འདིས་ཆ་ཀ་དང་མཐའ་ཨིཎ་དང་རྗེས་འབྲེལ་ནའི་ཨ་བརྗོད་དོན། ཀཱ་ལྱཱ་ཎི་ནེ་ཡ། དགེ་བ་མའི་བུ། ཇོ་བྲྀ་ཎི་ནེ་ཡ། རྒྱན་ལྡན་མའི་བུ། རྒན་མོའི་བུ། ཛིཊྛི་ནེ་ཡ། ཛཱིཥྛ་ལ་ཟླ་ཀ་མཐའ་ཨི་ནང་། བྲྀཏྟཱ་ག་སིན་ནེ་པཱུརྦ་པ་ད་སྱ་ཙ། ཀྲྀ་ད་བྷ་ག་སིནྡྷ་རྣམས་མཐའ་ལ་ཚིག་སྔ་མའི་འང་ངོ་། །དེ་ཕན་གྱི་རྐྱེན་འཕེལ་བའི་རྒྱུ་མཚན་ཅན་ཕ་རོལ་ལ་བྲྀཏ་སོགས་འདི་རྣམས་ཚིག་ཕྱི་མ་ཡིན་ཞིང་དེའི་སྔོན་དུ་ཚིག་གཞན་ཡོད་ན་ཚིག་སྔ་ཕྱི་གཉིས་ཀར་གྱིའང་ཨ་ཙ་་་དང་པོ་འཕེལ་བར་འགྱུར་ཏེ། ::སུ་ཧྲི་ད་ལ་དེའི་བུ་བརྒྱུད་ལ་འོས་ཨཎ་སྔ་ཕྱི་གཉིས་འཕེལ་བ། སོ་་་ཧཱརྡ༑ཡིད་གཅུགས་ཀྱི་བུ། སུ་བྷ་གི་ནི་ལ་བུད་མེད་རྣམས་ལས་རྡོ་ཀ་འོས་ཌྷ་ཀ་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་་་བ། སོ་བྷ་གི་ནེ་ཡ། ལེགས་པའི་སྲིང་མོའི་བུ། སུ་སིནྡྷལ་དེའི་བུ་བརྒྱུད་ལ་འོས་ཨཎ་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་བ། སོ་སིནྡྷ་བ། ལེགས་པའི་སིནྡྷའི་བུ། འགྲེལ་པར་ལུང་དྲངས་པ། ཨ་ནུ་ཤ་ཏི་ཀཱ་དཱི་ནཾ་ཙ། ཨ་ནུ་ཤ་ཏི་ཀ་ལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་། །ཨ་ནུ་ཤ་ཏི་ཀ་ལ་སོགས་པ་རྣམས་ཀྱིའང་དེ་ཕན་གྱི་རྐྱེན་འཕེལ་བའི་རྒྱུ་མཚན་ཅན་ཕ་རོལ་ལ་ཚིག་སྔ་ཕྱི་གཉིས་ཀར་གྱི་ཨཙ་དང་པོ་འཕེལ་བར་འགྱུར་ཏེ། ཨ་ནུཤ་རི་ཀ་ལ་དེའི་བུ་བརྒྱུད་ལ་འོས་ཨ་ཎ་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་བ། ་རཉྫི་ལ་ཀ་ལྱཱ་ཎི་ལ་སོགས་པ་རྣམས་ཀྱིས་ཨི་ནང་ངོས་ཧྥུ་ཀ་དང་མཐའ་ཨི་ནང་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་བ་རྟ་ཨེ་ཡ། པཱ་རསྟི་ཎེ་ཡ། ཕ་རོལ་གྱི་བུད་མེད་ཀྱི་བུ། ཀུ་ལ་ཊཱ་ཡཱ་བཱ། ཀུ་ལ་ཊཱའི་ཡང་ནའོ། །ཀུ་ལ་ཊཱ་ལ་བུ་བརྒྱུད་ལ་ཌྷ་ཀ་འགྱུར་ཞིང་ཀུ་ལ་ནཱའི་མཐའ་ཨི་ནང་དུ་ཡང་ན་འགྱུར་ཏེ། ཀོ་ལ་ཊི་ནེ་ཡ། འཕྱོན་མའི་བུ། ཀོ་ལ་ཊེ་ཡ། འཕྱོན་མའི་བུ། ཙ་ཊ་ཀཱ་ཡཱཨེ་རཀ། ཙ་ཊ་ཀཱ་ལས་ཨི་ར་ཀའོ། །ཙ་ཊ་ཀ་དང་། ཙ་——།
