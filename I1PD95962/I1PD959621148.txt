འགྱུར་ཏེ། ཎིཙབི་ག་ཎ་སྤྱ། རྣམ་པར་འཚོགས་འདོད། པྲ་ཎ་མ་སྤྱ། རབ་ཏུ་འདུད་འདོད། བི་བྷཱ་ཥཱ་ཡ། ཨ་ཡ་ཡང་ནའོ། །ཨ་བློ་ཎིཙའི་མཐའ་ཅན་ནི་ལྕཔ་ཕ་རོལ་ལ་ཨ་ཡར་ཡང་ན་འགྱུར་ཏེ། པཱ་པ་ཡ། རབ་ཏུ་ཁྱབ་པ། པཱ་པྱ། ཎི་ཨིཊ་མེད་པ་ལ་ཕྱིས་པའོ༑། འགྲེལ་པར་ཀྱི་ལྕཔ་ཕ་རོལ་ལ་རིང་པོར་ཡང་ན་འགྱུར་རོ་ཞེས་བཤད་པས། པྲ་ཀཱི་ཡ། རབ་ཏུ་ཟད། པྲ་ཀྲི་ཏྱ། མ་ཡ་ཏེ་རི་ད་རྒྱ་ཏ་ར་ཧཱུཾ། ་མ་ཡ་ཏིའི་ཨིཏ་ཡང་ནའོ། །མ་ཡ་ཏི་སྟེ་མེང་འཚེ་བའི་ཨཱ་ནི་ལྕཔ་ཕ་རོལ་ལ་ཨིཏ་དུ་འགྱུར་ཏེ། ཨ་བ་མི་ཏྱ། ཀུན་ནས་འཚེས། ཨ་བ་མཱ་ཡ། མྱ་པ་ཙ། ལྕཔ་ལ་འང་ངོ་། །བེཉའི་ལྕཔ་ཕ་རོལ་ལ་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་བ་མ་ཡིན་ཏེ། པྲ་བཱ་ཡ། རབ་ཏུ་རྒྱུད། །བི་བྷཱ་ཥཱ་པ་རེ། པ་རི་ལས་ཡང་ནའོ། །པ་རི་ལས་ཕ་རོལ་གྱི་བྱེཉའི་ལྕཔ་ཕ་རོལ་ལ་སཾ་པྲ་སཱ་ར་ཎར་ཡང་ན་འགྱུར་ཏེ། པ་རི་བཱི་ཡ། ཡོངས་སུ་སྒྲིལ། པ་རི་མྱུ་ཡ། ཨཱ་བྷི་ཀྵེ་ཎ་མུལྕ། ཡང་ཡང་ལ་ཎ་མུལ་འང་ངོ༌། །བྱེད་པ་པོ་གཅིག་གིས་བྱ་བ་སྔ་མ་ལ་ཡང་ཡང་གི་ཁྱད་པར་ལ་འཇུག་པ་ན་ཌ་འགྱུར་ཞིང་ཎ་མུལ་འང་འགྱུར་ཏེ། བྷུཀྟ་བྷུཀྟ་པྲ་ཛ་ཏི། ཐོས་ཤིང་ཟོས་ནས་འགྲོ་བར་བྱེད། བྷོ་ཛཾ་བྷོ་ཛཾ་བྲ་ཛ་ཏི། སོགས་སོ། །ཀརྨ་ཅུ་ལྟོ་ཤེ་ཀྲྀ་ཉཁ་མུཉ། ལས་ལས་ཀྲྀཉ་ལ་སྤྲོ་བ་ལ་ཁ་མུཉའོ། །ལས་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལས་སྤྱོ་བ་རྟོགས་པར་བྱེད་པ་ལ་ཁ་མུན་འགྱུར་ཏེ། ་ཁ་མུམ་གྱི་དོན་དང་ཉ་འཕེལ་བའི་དོན་ནོ། །།ཙོ་@རཾ་ཀཱ་རཾ། རྐུ་བའི་ལས་བྱེད། སྤྲོ་བ་མ་ཡིན་ན་ཀ་སྦྱར་བས། སཱ་དྷུ་ཀྲྀ་ཏྭཱ་སྟོ་ཏི། ལེགས་སོ་བྱས་ནས་བསྟོད། སཱ་དུ་མི་ཎ་མུལ། ཞིམ་པ་ལ་ཎ་མུལའོ། ་། ཞིམ་པའི་དོན་ཅན་གྱི་ཚིག་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ཎ་མུལ་འགྱུར་ཏེ། །སཱ་དུ་ཀཱ་རཾ་བྷུཀྟི། ཞེམ་པར་བྱས་ཏེ་ཟས། ཡང་ན་ཌ་སྦྱར་བས། སཱ་དུ་ཁྲི་ཏཱ་བྷུཀྟ། ཞིམ་པར་བྱས་ཏེ་ཟས། ཨ་ནྱ་ཐི་བི་ཀ་ཐ་མིཏྠཾ་བྷུཉྩ། ་ཨ་ནྱ་ཐཱཨེ་བིཀ་ཐཾ་ཨིཏྠཾ་རྣམས་ལས་ཀྱང་ངོ༌། །ཨ་ནྱ་ཐཱ། ཀ་ཐཾ། ཨིཏྠཾ་རྣམས་དང་རམངལ་།
