སྤྲོ་ཡུརྫུ་བྷ་ར་ཛྙཱ་པི་ས་ཏྭཾ། །ཨཱིཔ་མཐའ་དང་རཾ་དྷ་བྷསྔ་དམྦྷ་ཤི་སྤྲོ་ཡུ་ཨུཎྜ་བྷ་ར་ཙཱ་པི་སན་རྣམས་ཀྱི་སན་ལའོ། །སྒྱུ་། སུཔ་ཀྱི་བྱིངས་མོ་རྐྱེན་ཨཱིཔ་མཐའ་ཅན་རྣམས་དང་། རཾ་དྷུ་འཕེལ་བ། བྷསྔ་འཚེད་པ། དམྦྷ་ངན་གཡོ་བ༑ཀ་ཤི་གུས་པ། །སྤྲོ་སྒྲ། ::ཡུ་སྲེ་བ། ཌཨུཎྜཉ་ཀུན་ནས་འགེབས་པ། བྷ་ར་སྟེ། ཊུ་བྷོཉ་འཛིན་པ། ཛཱ་ཁོང་དུ་ཆུད་པ། ཥ་ཎ་སྦྱིན་པ་རྣམས་ལས་ཕ་རོལ་གྱི་སན་ལ་ཨིཊ་ཡང་ན་འགྱུར་ཏེ། །ཨིཊ་བྱིན་པའི་ཕྱོགས་ལ་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། དྷ་ཨིཊ་དང་བཅས་པ་གཉིས་བརྗོད། ཨ་ངྷ་སའི་ཙར་འང་ངོས། 《། དྷ་དར་བསྒྱུར། ཨཇི་དྷི་ཥ། ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་་་་བས། ལཊཨཇི་དྷི་ཥ་ཏི། ལུང་…ཨཱཇི་དྷི་ཏཱིཏ། ་ལིཊཨཇི་དྷི་ཊཱ་ཙ་ཀཱ་ར། །སོགས་དང་། ཨིཊ་མེད་པའི་ཕྱོགས་ལ། ཨཱ་པ་ཛྙཱ་མྦོ་དྷཱ་མཱིཏ། ཨཱཔ་ཛྙཱ་པི་རི་དྷ་རྣམས་ཀྱི་ཨཱིཏའོ། །ཨཱ་མྲྀ་ཁྱབ་པ། །ཛཱ་ཁོང་དུ་ཆུད་པ། རཾ་དྷུ་འཕེལ་བ་་་་རྣམས་ཀྱི་ཨིཊ་མེད་པའི་སན་ཕ་རོལ་ལ་ཨཙ་ཨཱིཏ་དུ་འགྱུར་ཞིང་ཨ་བྷཱུ་སའང་དབྱི་སྟེ། །ཨུ་པ་དྷཱའི་འང་ངོས། རྀ་ཨིཏ། མདོ་འདིས། དེ་ཨཱིཊ་དུ་བསྒྱུར། ལ་ཛའི་་ཨཎ་ལ་ར་ཕ་རོལ་ལོས་ར་བྱིན། ཨ་བྱཱ་ས་ཕྱིས། ཁར་ལ་ཙར་འོས་དྷ་ཏར་བསྒྱུར་་་་བས། ཨིནྡྲ་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊཨིནྡྲ་ཏི། ལིང་ཨཱིནྡྷེཏ། ལཊཨཱིནྡྲ་ཏུ། ལུང་ཨེཆ་ཏེ། ལུང་…ཨིཎཱིཏ། ལིཊ་ཨིནཾ་ཙ་ཀཱ་ར། ལུཊཨཱནྟི་ཏཱ། སྨོན་ལིང་ཨཱིནྡྷཏ། ཝིཊ་ཨཱི་ཎཱིཥྛ་ཏི། ཕྲེང་ཨིནྟི་ཏ། སོགས་སོ། །་ཀྱང་། ད་བར་༈བྷསྔ་འཚེད་པ་ལའོ་ཞེས་པ་ལ་སན། ཨིཊའི་ཕྱོགས་ལ། བྷཱསྔའི་བྷརྫ་ཡང་ན་འོས། བྷརྫར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་གཉིས་བརྗོད། སན་ལ་ཨིཏ་འོས་ཨིཏ་དུ་བསྒྱུར་བས། ལཊ་བི་བྷཌི་ཥ་ཏི། ལུང་ཨ་བི་བྷནྡི་ཏྟིཏ། ལིཊབི་བྷཉྫི་ཊཱ་ཙ་ཀཱ་ར། །སོགས་སམ། ལཊ་བི་བྷརྫི་ཥ་ཏི། ལུང་…ཨ་བི་བྷརྫི་，།
