གསུམ་ལ། །ཀཱི་ཎཱི་ཥཱ། ཀཱི་ཎཱ་ཐཱ། ཀཱི་ཎཱི་ཧཱུཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཀཱི་ཎ༑་པཀཱི་ཎཱ་བ་ཧི། ཀཱི་ཎཱ་མ་ཧི། ལང་གི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀཱི་ཎཱ་ཏི། ཨ་ཀཱི་*ཎྜི་ཏྭཾ། ་ཨ་ཀཱི་ཎན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཀཱི་ཎ། ཨ་ཀྲྀ་ཎཱི་ཏཾ། ཨ་ཀྲྀ་ཎཱིཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀཱི་ཎཾ། ཨ་ཀཱི་ཎཱི་བ། ཨ་ཀཱི་ཎཱི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀཱི་ཎཱི་ཏ། ཨ་ཀཱི་ཎཱ་ཏཱ། ཨ་ཀཱི་ཎ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨ་ཀཱི་ཎཱི་ཐཱ། ཨ་ཀཱི་ཎཱ་ཐཱ། ་ཨ་ཀཱི་ཎཱི་ཧཱུྃ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀཱི་ཎི། ཨ་ཀཱི་ཎཱི་བ་ཧི༑་ཨ་ཀཱི་ཎཱི་མ་ཧི། ལུང་གི་ཨ་ཏང་ལ། སིཙ་ལ་འཕེལ་བ་སོགས་ཀྱིས་འཕེལ་བ་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀྲི་ཏཱི་ཏ། །ཨ་ཀྲི་ཏྟཱ། ཨ་ཀྲི་། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཀྲྀཏྟི། ཨ་ཀྲིཏྠཾ༄༅། ཨ་ཀྲི། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀྲིཏྟཾ། ཨ་ཀྲི་། ་ཨ་ཀྲིཥྛ། ཏང་ལ་བྱིངས་ཐམས་ཅད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཀྲེཊྚ། ཨ་ཀྲེ་ཥཱ'ཏྭཾ། ཨ་ཀྲེ་ཥ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཀྲེཊྛ། ཨ་ཀྲེ་ཥཱ་ཐཱ། ཨ་ཀྲེ་ཧཱུྃ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཀྲེ་ཎི། ཨ་ཀྲེ་ཥཱ་ཧི། ཨ་ཀྲེཧྥ་ཧི། །ལིཊ་ལ། གཉིས་བརྗོད་འདི་བརྩེགས་པ་ཅན་ཡིན་པས་ཀ་ང་འགྲོ་བའི་མིང་ཅན་མ་ཡིན་པས་ཡོན་ཏན་དང་འཕེལ་བ་སོགས་བྱས་པས། །ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ་ཙི་ཀཱ་ཡ། །ཙི་ཀྲ་ཡ་ཏུ། :ཙི་ཏྲ་ཡུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙི་ཀྲ་ཡི་ཐ། ཙི་ཀྲེ་ཐ། ་ཙི་ཀྲ་ཡ་ཐུ། ཙི་ཀྲ་ཡ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། »ཙི་ཀཱ་ཡ། ཙི་ཀྲ་ཡ། ཙི་ཀྲ་ཡི་བ། ཙི་ཀྲ་ཡི་མ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙི་ཀྲ་ཡེ། ཙ་ཀྲ་ཡཱ་ཏེ། ཙི་ཀྲ་ཡི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙི་ཀྲ་ཡི་ཥ། །ཙི་ཀྲ་ཡཱ་ཐེ། ཙི་ཀྲ་ཡི་དྷེ། ་ཙི་ཀྲ་—།
