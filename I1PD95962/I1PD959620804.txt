རསྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བུ་བུ་དྷེ། །།བུ་བུ་དྷི་བ་ཧེ། བུ་བུ་དྷི་མ་ཧེ། །ལུཊ་ལ༑མི་པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། ། ོཛྷཥ་མཐའ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་དྷ་དར་བསྒྱུར་བས། བོདྷཱ། བོདྡྷཱ་སེ། སོགས་སོ། །སྨོན་པའི་ལིང་ལ། ་ཨཙ་གཅིག་པ་སོགས་ཀྱིས་བ་བྷ། །་ཥར་ལ་ཙར་འོས་དྷ་ཏར་་་བསྒྱུར་བས། བུདྷིཥྛ། བུདྷཱི་ཡཱནྟཾ། །བུཏི་རན། །སོགས་སོ། །།དེ་བཞིན་དུ། ཁྲིཊ་ལ། ་་:བྷོཉྩ་ཏེ། …པུཀ་མཐའ་སོགས་ཀྱིས་ཡོན་ཏན། སོགས་དང་། །སྲིང་ལའང་། ཨ་བྷོཉྩ་ཏ། སོགས་སོ། །་། །་ལོ་ཚོད་ལ་ཐོགས་ཏེ། །དང་པོ་ནད་དེ་དག་ཁོ་རེ༈ཎ་ཧ་འཆིང་བ་ལའོ་ཞེས་པ་ལ་ཡང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། ཎའི་ན་འོས་ཎ་ན། །དི་བ་སོགས་ལ་བྱིན། ཧ་དྲངས་པས་ལཊའི་ཨ་་ཏང་ལ། ན་ཧུ་ཏི། སོགས་དང༌། ཏང་ལ། ན་:རྒྱ་ཏེ༄༅། །སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། ན་བྱེ་ཏ། །སོགས་དང་ཏང་ལ། ན་བྱེ་ཏ། སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། ་ན་རྒྱ་ཏུ། །སོགས་དང་། ཏང་ལ། ན་ཧུ་ཏྭཾ། སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། །ཨ་ན་བྱཱཏ། །སོགས་དང་། ཏང་ལ། ཨ་་་ན་རྒྱ་ཏ། །སོགས་སོ། །ལུང་ལ། །ཛྷལ་དང་པོ་རྣམས་ལ་སིཙ་འཇིག་། ་་་་ལས་ཧའི་དྷ་འོས། །ཧ་དྷ། ཁར་ལ་ཙར་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ནཱནཱིཏི། བདེ་བྲཛ་སོགས་ཀྱིས་འཕེལ་བ། ཨ་ནཱདྡྷཾ། ཛྷཥ་མཐའ་སོགས་ཀྱིས་ཏ་དྷ། ཛྷལ་སོགས་ཀྱིས་དྷ་ད། ་། ཨ་ནཱནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ༑་ཨ་ནཱཏི། ཨ་ནཱནྟཾ། ཨ་ནཱདྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ནཱཏྟཾ། ཨ་ནཱདྷཱི། ཨ་ནཱཏྨ། དེ་བཞིན་དུ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ནདྡྷ། དང་པོ། ཨ་ནནྟཱ་ཏཱ། ཨ་ནཏྠ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ནདྷཱ། ཨ་ནནྟཱ་ཐཱ། ཨ་ནདྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ནཏི། ཨ་ནཏྟཱ་ཧི། །ཨ་།
