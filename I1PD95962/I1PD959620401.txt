ར་གྱི་མ་ལ་ཏོ་བསྟན་འགྱུར། སྒྲ་མདོ། ཏིན་ནི་་་་་༌ཡིད༌:བློ་ལ་སཱ་ཤ་ཡ། བདག་ཉལ། གྲྭ་མེ་བཱ་ས། གྲོང་ལ་བསྡད། གྲྭ་མ་བཱ་ས། གྲོང་ལ་བསྡད། གྲྭ་མེ་་བཱ་སི། གྲོང་དུ་བསྡད། གྲྭ་མ་བཱ་སཱི། འགྲེལ་པར་ལུང་དྲངས་པ། ཨ་པོ་ཡོ་ནི་ཛ་མ་ཏི་ཥཱ། ཨ་པ་ལས་ཡོ་ནི་ཛ་མ་ཏི་རྣམས་ལའོ། །ཨཔ་ཀྱི་ཕ་རོལ་གྱི་བདུན་པ་ནི་ཡོ་ནི་སོགས་འདི་རྣམས་ཚིག་ཕྱི་མར་གྱུར་པ་ལ་མི་དབྱི་སྟེ། ཨ་ྩ་ཡོ་ནི། ཆུ་རྣམས་ལ་སྐྱེས་པ། ཨཔྤཱ་ཛ། ཨུg་མ་ཏི། ཆུ་རྣམས་ལ་བློ་གྲོས། ཥབྷཱ་ཨཱ་ཀྲོ་ཤེ། །དྲུག་པ་ནི་ཁྲོས་ཚིག་ལའོ། །དྲུག་པའི་མཐའ་ནི་ཁྲོས་ཚིག་རྟོགས་པར་བྱེད་པ་ན་ཚིག་ཕྱི་་མ་ལ་མི་དབྱི་སྟེ། ཙོ་ར་སྱ་ཀུ་ལོ། ཆོམ་རྐུན་གྱི་རིགས། དྭ་ས་སུ་ཀུ་ལོ། བྲན་གྱི་རིགས། བཱཀ་དིཀྲ་ཤུདྡྷོ་ཡུཀྟི་དཎྜ་ཧ་རེ་ཞུ། ་ཝཱཀ་དི་ཤ་པ་ཏ་རྣམས་ལ་ཡུཀྟི་དཎྜ་ཧ་ར་རྣམས་ལའོ༑། ཝཱཀ་དི་ཤ་པ་བྱ་ཏ་རྣམས་ལས་ཕ་རོལ་གྱི་དྲུག་པ་ནི་རིམ་པ་བཞིན། ཡུཀྟི་དཎྜ་ཧ་ར་རྣམས་ཚིག་ཕྱི་མར་གྱུར་པ་ལ་མི་དབྱི་སྟེ། བཱ་ཀོ་ཡུཀྟི། ངག་གི་སྦྱོར་བ། དི་ཤོ་དཎྜ། …ཕྱོགས་ཀྱི་དབྱུག་པ། པ་བྱ་ཏོ་ཧ་ར། མཐོང་བའི་འཕྲོག་པ། ཤེ་ཕ་པུཙྪ་ལཱངྒཱ་ལེ་ཥཱ་ཤུ་ན། ཤཱན་ལས་ཤེ་ཕ་པུཙྪ་ལཱདྨ་ལ་རྣམས་ལའོ། །།ཤཱན་ལས་ཕ་རོལ་གྱི་དྲུག་པ་ནི་ཤེ་ཕ་སོགས་འདི་རྣམས་ཚིག་ཕྱི་མར་གྱུར་པ་ལ་མི་དབྱི་སྟེ། ་ཤུ་ནཤེ་་ཕ༑ཁྱིའི་རྟགས། ཤུ་ནིཔུཙྪ། ཁྱིའི་མཇུག་མ། ཤུ་ནལཱདྷུ་ལ། ཁྱིའི་མཛུབ་མོ། པུ་ཏྲེ་་་ཉནྱ་ཏ་ར་ཧཱུཾ། པུ་ཏྲ་ལ་ཡང་ནའོ། ::། པུ་ཏྲ་ཚིག་ཕྱི་མ་ལ་ཁྲོས་ཚིག་རྟོགས་པར་བྱེད་པ་ལ་དྲུག་པ་ཡང་ན་མི་དབྱི་སྟེ། དཱ་སྟ་པུ་ཏྲ། བྲན་མོའི་བུ། དཱ་སཱི་པུ་ཏྲ། བྲན་མོའི་བུ། དྲྀཥ་ལྟ་པུ་ཏྲཿདམངས་རིགས་མོའི་བུ། པྲྀ་ཥ་ལཱི་པུ་ཏྲཿ། །ཁྲོས་ཚིག་མ་ཡིན་ན་་་་དབྱི་བ་ཉིད་དེ། བྲཱཏྨ་དྷཱ་པུ་ཏྲ། བྲམ་ཟེ་མོའི་བུ། ཞེས་པ་ལ། བྲཱཧྥ་ཎི་པུ་ཏྲ། བྲམ་ཟེ་མོའི་བུ། ཛ་ཏོ་བི་དྷཱ་ཡོ་ནི་སཾ་བནྡྷེ་བྷུ། རྀཏ་ལས་རིག་པ་སྐྱེ་གནས་འབྲེལ་པ་རྣམས་ལས་སོ། །རྀཏ་མཐའ་ལས་རིག་པ་དང་འབྲེལ་པ་དང་སྐྱེ་གནས་དང་འབྲེལ་པ་ཅན་ལས་ཕ་རོལ་གྱི་དྲུག་པ་ནི་རིག་པ་དང་སྐྱེ་གནས་དང་འབྲེལ་པ་ཅན་བྲམ་—།
