ལས་སན་འོས། སན་སན་ཡང་དག་ལ་འོས་བྱིངས་གཉིས་བརྗོད། ཧལ་དང་་པོ་ལྷག་པོ་འོས། <ཛ་གོང་མ་ཕྱིས། ཙུའི་ཀུ་འོས། ཛ་ག་། ཁར་ལ་ཙར་་་འོས། པོག་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། ཀ་ཥ་འདུས་པ་ཀྵ། ཏི་ཏིཀྟ་ཞེས་བསྒྲུབས་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་ལ། ཏི་ཏིཥྛ་ཏེ། བྱེད་པ་པོ་ལ་ཤཔ་རྗེས་འབྲེལ་ཕྱིས་ནས་ཨཏ་ཀྱི་ཡོན་ཏན་ལ་འོས། གཉིད་དག་གིས་གཅིག་ཕ་རོལ་གྱི་གཟུགས། ཏི་ཏིཀྵེ་ཏེ། ཏི་ཏིཀྵནྟེ་སོགས་སོ། །ལིང་ལ། ཏི་ཏིཀྵེ་ཏ། ཏི་ཏིཥཱ་ཡཱ་ཏཱཾ། ཏི་ཏིཀྵེ་རན་་སོགས་སོ། །ལོཊ་ལ། ཏི་ཏིཀ་ཏྭཾ། ཏི་ཏིཀྵེ་ཏྭཾ། ཏི་ཏིཀྵནྟཾ་སོགས་སོ། །ལང་ལ། ཨ་ཏི་ཏིཀ་ཏེ། ཨ་ཏི་ཏིཀྵེ་ཏྭཾ། ཨ་ཏི་ཏིཀཱནྟ་སོགས་སོ། །ལུང་ལ་བྱིངས་ཕྱེད་པའི་ཝལ་དང་པོ་ལ་ཨིཊ་འོས། ཨིཊ་བྱིན། ཨཏ་ཀྱི་དབྱིའོ། །བྱིངས་མཐའི་ཨ་དབྱི་བ་སོགས་བྱས་པས། ཨ་ཏི་ཏིཥིཉྩ། ཨ་ཏི་ཏིལླེ་ཥཱ་ཧཱུཾ། ཨ་ཏི་ཏིཀཱི་ཥ་ཏ་སོགས་སོ། །ལིཊ་ལ། ཀཱས་རྐྱེན་སོགས་ཀྱི་བྱིངས་མཐར་ཨཱམ་སྦྱར༑ཨཱམ་ལ་ས་སོས། །ལིཊ་འཇིག། ཀྲྀཉ་སོགས་སྦྱར་བས། ཏི་ཏིངྒྷཾ་་་ཙཀྲེ། ཏི་ཏིཀཾ་བ་བྷཱུ་བ། ཏི་ཏིཀཱ་མཱ་ས་སོགས་སོ། །ལུཊ་ལ་ཝལ་དང་པོའི་ཨིཊ་དང༌། ཨཏ་ཀྱི་དབྱི་བ་སོགས་བྱས་པས། ་ཏི་ཏིཥི་ཏཱ། ཏི་ཏིལླི་ཏཱ་རོ། ཏི་ཏིཀྵི་ཏཱ་ར། ཏི་ཏིཀྵི་ཏཱ་སེ་སོགས་སོ། །སྨོན་པའི་ལིང་ལའང་། ཏི་ཏིལླི་ཊཱི། ཏི་ཏིཀྵི་ཏི་ཡཱནྟཾ། ཏི་ཏིཀྵི་ཏི་རན་སོགས་སོ། །ཁྲིཊ་ལའང༌། བྱིངས་ཕྱེད་པའི་བལ་དང་པོ་ལ་ཨིཊ་དང༌། སྱ་ཏཱས་དག་ཅེས་པས། མྱ་སྦྱར་བས། ཏི་ཏིཀཱི་ཥྛ་ཏེ། ཏི་ཏིཀྱི་བྱེ་ཏེ། ཏི་ཏིཀྱི་ཉྩནྟེ་སོགས་སོ། །དེ་བཞིན་དུ། ལླིང་ལ། ཨ་ཏི་ཏིཀཱི་་བྱ་ཏ། ཨ་ཏི་ཏིཥི་ེ་ཏྭཾ། ཨ་ཏི་ཏིའི་ཤྱནྟ་སོགས་སོ། །།།བྱེད་༈ཥཱཉྫ་ཡོངས་སུ་འཁྱུད་པ་ལའོ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱར། བྱིངས་ཀྱི་དང་པོའི་ཥའི་ས་འོས་ཥ་ས། ་དན། སཉྫ་སཱཉྫ་རྣམས་ཀྱི་ཤཔ་ལ་འོས་ན་དབྱི་་།
