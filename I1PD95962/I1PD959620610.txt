ཚིག་ལ། །།།ཨེ་དཾ་ཙ་ཀྲྀ་ཥ། ལ་ཐཱས་སུ་བསྒྱུར་ཐཱས་ཀྱི་སེ་འོས་ཐཱས་སེ་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥ། ཨེ་དཾ་ཙཀཱ་ཐེ། ལ་ཨཱ་བྷོ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། ཨིཀ་གི་ཨཙ་ལ་ཡཎ། སྐྱེས་བུ་དབུས་མའི་་་མང་ཚིག་ལ། ་ཨི་ཎཥྛ་དྷཾ་ལུང་ལི་ཊཾ་དྷོཉནྟཱ་ཏ། ཨིཎ་མཐའ་ལས་ཏི་དཾ་ལུང་ལིཊ་རྣམས་ཀྱི་དྷའི་འོ། །བྱིངས་ཨིཎ་སོ་སོར་བསྡུ་བའི་མཐའ་ལས་ཕ་རོལ་གྱི་སྨོན་པའི་ལིང་གི་སི་དྡྷཾ་དང་ལུང་ལིཊ་རྣམས་ཀྱི་དྷ་ཡིག་ནི་ཌྷ་ཡིག་ཏུ་འགྱུར་ཏེ། ཨེ་ཧཱུཾ་ཙ་ཀྲྀ་ཌྷ། ཀྲྀ་ལ་ལཊ་བྱིངས་གཉིས་བརྗོད་རིའི་ཨཏ་འོས། རཾ་ཨ་ཀུ་ཧ་དག་གི་ཙུ་འོས། ཀཙ། །ལ་ཛྷ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། མདོ་འདིས་དྷ་ཆུ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་དཾ་ཙཀྲེ། ལ་ཨིཊ་རུ་བསྒྱུར་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ། ཨེ་དཾ་ཙ་ཀྲྀ་བ་ཧེ། ལ་བ་ཧི་དང་མ་ཧིང་དུ་བསྒྱུར། ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ༑ཨེ་དཾ་ཙ་ཀྲྀ་མ་ཧེ། ཨེ་དྷ་ལ་ལུཊ་ཝལ་དང་པོའི་ཨིཊ་སྦྱར་བས། སྐྱེས་བུ་་་དང་པོ་གསུམ་ལ། ཨེ་དྷི་ཏཱ། ཨེ་དྷི་ཏཱ་རོ། ཨེ་དྷི་ཏཱ་ར། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨེ་དྷི་ཏཱ་སེ། ཨེ་དྷལ་ལུཊ''ས་ལ་ཐཱས་སུ་བསྒྱུར་ཐཱས་ཀྱི་སེ་འོས་ཐཱས་སེ་ཏཱས་ཨ་ཏི་དག་གིས་དབྱི་འོས་ཏཱས་ཀྱི་ས་ཕྱིས་ཝལ་དང་པོའི་ཨིཊ། ཨེ་དྷི་ཏཱ་སཱ་ཐེ། ལ་ཨཱ་ཐཱམ་དུ་བསྒྱུར། །ཊ་འགྲོ་བ་སོགས་་ཀྱིས་མཐའ་ཨེ། ཨེ་དྷི་ཏཱ་དེ། ལ་དྭོ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་མཐའ་ཨེ་དྷ་ལ་འང་ངོས་ཏཱས་ཀྱི་ས་ཕྱིས། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་དྷི་ཏཱ་ཧེ། ཨེ་དྷ་ལ། ལུཊ་ཏཱས་ལ་ཨིཊ་རུ་བསྒྱུར་ཊ་འགྲོ་བ་སོགས་ཀྱིས་ཨི་ཨེ་ཨེཏ་ལ་ཧ་འོས། ས་ཧ་བལ་དང་པོའི་ཨིཊ། ཨེ་དྷེ་ཏཱ་སཱ་ཧེ། ལ་བ་ཧི་དང་མ་ཧིང་དུ་བསྒྱུར། ་ཊ་འགྲོ་བ་སོགས་ཀྱིས་་མཐའ་ཨེ། ཨེ་དྷི་ཏཱསྨ་ཧེ། ཨེ་དྷ་ལ་སྨོན་པའི་ལིང་སྦྱར་བའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨེ་དྷི་ཊཱི། ཨེ་དྷི་ཥཱ་ཡ་ཙཱ། ཨེ་དྷི་ཊཱི་རན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨེ་དྷི་ཏི་ཥྛ། ཨེ་དྷི་ཊཱི་ཡཱཧཱུཾ། ཨེ་དྷི་ཥི་དཾ། ་ཨེ་དྷི་ཌི་ཌྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨེ་དྷི་ཏི་ཡ། …ལ་ཨིཊ་རུ་བསྒྱུར་ཨིཊའི་ཨཏ་འོས། ཨིཊ་ཨཏ། ཨེ་དྷི་ཊཱི་བ་ཧཾ། ཨེ་དྷི་ཊཱི་མ་ཧི། ཊི་ཨེ་དྷ་ལ་ཁྲིཊ་སྦྱར་བའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨེ་དྷི་ཥྱ་ཏེ། ཨེ་དྷི་ཥྛ་ཏེ་ཨེ་དྷི་ཥྛནྟེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཨེ་དྷི་ཥྱ་སེ། ཨེ་དྷི་——སྒྱུ།
