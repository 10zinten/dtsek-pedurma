་སྒྲ་མདོ། ཏིའི་དཔེཥ་ནི་སྨན་ཐོག་མཚོ་མོ་བཀལ་ཡ་ཀར་བློ་ཀྲྀ་ཎི་དག་གི་ཨའང་ངོས། སྣུའི་ཡུལ་ལ་ཨུབྲ་ཨ། ཨིཏ་འགྲོ་བའི་བྱིངས་ལ་ནུམ་་འོས། ནུམ་ཨཱ་ག་མ། ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། དྷི་ནུ་ཞེས་གྲུབ་པ་ལ་ལཊ་སོགས་སྦྱར་བས། ལཊ་ལ། དྷི་ནོ་ཏི། སོགས་དང་། ལིང་ལ། དྷི་ནུ་ཡཱཏ། སོགས་སོ། །ལོཊ་ལ། དྷི་ནོ་ཏུ། སོགས་དང༌། ་ལང་ལ། ་ཨ་་དྷི་ནོཏ། སོགས་སོ། །ལུང་ལ། ནུམ་དང་ཨིཊ་བྱིན་པས། ཨ་དྷི་ནཱི་ཏི། ཨ་དྷི་ནཱིཏྟཾ། ཨ་དྷི་ནཱི་ཥཱ། སོགས་སོ། །ལིཊ་ལ། ནུམ་བྱིན་གཉིས་བརྗོད་པ་སོགས་ཀྱི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། དི་དྷི་ནཱ། དི་དྷི་ནཱ་ཏུ། །དི་དྷི་ནུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། དི་དྷི་ནཱི་ཐ། ་་དི་དྷི་ནཱ་ཐུདི་དྷི་ནཱ། སྐྱེས་་བུ་ཕྱི་མ་གསུམ་ལ། དི་དྷི་ནཱ། དི་དྷི་ནི་བ། དི་དྷི་ནཱི་མ། ལུཊ་ལ། དྷི་ནཱི་ཏཱ། དྷི་ནཱི་ཏཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། དྷི་དྷཱཏ། དྷི་སྭཱ་ཧཱུཾ། དྷི་ཅུ་སུ། །སོགས་སོ། །ལྀཊ་ལ། དྷི་ནཱི་ཥྱ་ཏི། སོགས་དང༌། ཁྲིང་ལ། ཨ་དྷི་ནཱི་ཥྱ་ཏ། །སོགས་སོ། །།་ཏ་ཏེ་དཀྱི་༈དནྟུ་ངན་གཡོ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། ཨི་ཏྲ་འགྲོ་བ་མ་ཡིན་པ་སོགས་ཀྱིས་ན་ཕྱིས་པས། །ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་དྡྷོ་ཏི། ད་སྨཱ་ཏ། ད་་བནྟི། ཨཙཱ་ལ་ཀྵ་བྱིངས་སོགས་ཀྱི་ཨུ་བར་བསྒྱུར། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །དདྡྷོ་ཏི། ད་བྷྱཱ་ཐ། ད་བྷྱཱ་ཐ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་བྷོ་མི། ད་ལྦུ་བ། ད་བྷྱཱ་མ། ལིང་ལ། ད་་ཡཱཏ། ད་ལྕུ་ཡཱ་ཏཱ། ད་བྷྱཱ་ཡུ། སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ད་བྷོ་ཏུ། །ད་་ཏཱཏ། ད་་ཏྭཾ། ད་དྡྷ་བནྟུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ད་་ཧི། ད་་ཏཱཏ། ད་ལྕུ་ཏཾ། ད་ལྕུ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ད་བཱ་ནི། ད་བྷཱ་བཱ་བ། ད་བྷཱ་བཱ་མ། ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ད་ཙྪེཏ། —།
