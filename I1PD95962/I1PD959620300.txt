ཨ་དྷི་རཱི་ཤྭ་རེ། ཨ་དྷི་དབང་ཕྱུག་ལའོ། །ཨ་དྷི་སྦྱོར་བ་དབང་ཕྱུག་གི་དོན་བརྗོད་པ་ཅན་ལ་ལས་རབ་ཏུ་བརྗོད་པ་ཞེས་མིང་འདོགས་བྱེད་དོ། །ཡསྣ་ད་དྷི་ཀཾ་ཡ་མྱ་ཙེ་ཤྭ་ར་བ་ཙ་ནཾ་ཏ་ཏྲ་སཏ་མཱི། གང་ལས་ཨ་དྷི་གང་གི་དབང་ཕྱུག་བརྗོད་པ་དེ་ལ་བདུན་པའོ༑། གང་ལས་ཨ་དྷི་སྦྱོར་བ་གང་གིའང་དབང་ཕྱུག་གི་དོན་བརྗོད་པ་ལས་རབ་ཏུ་བརྗོད་པ་ཞེས་མིང་འདོགས་བྱས་པ་དེ་ལ་བདུན་པ་འགྱུར་ཏེ། ཨ་དྷི་བྷུ་བི་རཱ་་་མ། སའི་དབང་ཕྱུག་དགའ་བྱེད། ཥཊྛིཙཱ་ནཱ་ད་རེ། མ་གུས་པ་ལ་དྲུག་པ་ཡང་ནའོ། །གང་གི་བྱ་བའི་བར་གྱི་རྟགས་དེ་ལས་མ་གུས་པ་བརྗོད་པ་ལ་དྲུག་པ་དང་བདུན་པའང་ངོ་། །རུ་ད་ཏི་པཱ་བཱ་ཛཱི་ཏ། དུ་ཞིང་སོང་། རུ་ད་ཏཔ་བཱ་ཛཱི་ཏ། སཱ་མཱི་ཤྭ་རཱ་དྷི་པ་ཏི་དཱ་ཡཱ་ད་སཱཀྟི་པྲ་ཏི་བྷཱུ་པྲ་སཱུ་ཏི། སཱ་མཱི་ཨཱི་ཤྭ་ར་ཨ་དྷི་པ་ཏི་དཱ་ཡཱ་ད། སཱཀྐི། པྲ་ཏི་་བྷུ་པྲ་སཱུ་ཏེ་རྣམས་ཀྱིའང་ངོ༌། །སཱ་མཱི་ལ་སོགས་པ་འདི་རྣམས་དང་སྦྱོར་བའི་་་རྟགས་ལ་དྲུག་པ་དང་བདུན་པ་ཡང་ན་བ་སྟེ། དྷརྨ་སཱ་མཱི། ཆོས་ཀྱི་རྗེ། དྷརྨ་སྱ་སཱ་མཱི༑ཆོས་ཀྱི་རྗེ། དྷ་ན་ཨཱི་ཤྭ་ར། ནོར་གྱི་དབང་ཕྱུག། དྷ་ན་སྱཱ་ཨཱི་ཤྭ་ར། །དེ་ཤེ་ཨ་དྷི་པ་ཏི། ཡུལ་གྱི་བདག་པོ། དེ་ཤ་སྱ་ཨ་དྷི་པ་ཏི། བྷོ་གེ་དཱ་ཡཱ་ད། བྷོ་ག་སྱ་དཱ་ཡཱ་ད། ལོངས་སྤྱོད་བགོ་སྐལ། དྷ་ནེ་སཱཀྐི། ནོར་གྱི་དབང་པོ། ད་ན་སྱ་སཱཀྐི། དྷ་ནེ་པྲ་ཏི་བྷཱུ། ནོར་གྱི་གཉའ་བོ། དྷ་ན་སྱ་པྲ་ཏི་བྷཱ། ག་ཐཾ་པྲ་སཱུ་ཏ། བ་ལང་གི་བེའུ། གོ་ཥཱ་པྲ་སཱུ་ཏ། ཨཱ་ཡུལྟ་ཀུ་ཤ་ལཱ་བྷཱུ་བཱ་སེ་བཱ་ནཱཾ། ཨཱ་ཡུལྟ་ཀུ་ཤ་ལ་དག་གི་མཆོད་པ་ལ་ཡང་ནའོ། །ཨཱ་ལི་ཡུཀྟ་དང་ཀུ་ཤ་ལ་དག་གི་མཆོད་པའི་དོན་བརྗོད་པ་དང་། སྦྱོར་བ་ལ་དྲུག་པ་་་དང་བདུན་པ་ཡང་ན་སྟེ། ཨཱ་ཡུཀྟཧ་རི་པཱུ་ཛ་ནེ། འཕྲོག་བྱེད་ལ་མཆོད་པ་ཀུན་དུ་སྦྱོར་བར་་་བྱེད། ཨཱ་ཡུཀྟཧ་རི་པཱུ་ཛ་ན་སྱ། ཀུ་ཤ་ལཧ་རི་པཱུ་ཛ་ནེ། འཕྲོག་བྱེད་ལ་དགེ་བ་་་མཆོད་པར་བྱེད། ཀུ་ཤ་ལཧ་རི་པཱུ་ཛ་ན་སྐྱ། ཡ་ཏ་ནིནྟཱ་ར་ཎཱ། གང་ལས་ཀྱང་ངེས་པར་བཀར་བ་རྣམས་ཀྱིའོ། །རིགས་དང་ཡོན་ཏན་དང་བྱ་བ་ལ་སོགས་པའི་སྒོ་་་ས་HC།
