བུར་བྱེད་དོ། །ཞེས་པའི་དོན་ལ། །གཱརྒྱལ་ཙཱ། ཀ་རོ་ཏི། ཀྱ་ཙཱ་ཙ། ཀྱ་ཙཱི་ལ་་འང་ངོ༌། །ཆ། །རྐྱེན་ཀྱི་དང་ཙཱི་ཕ་རོལ་ལ་རྟགས་བྱིངས་ཀྱི་མཐའི་ཧལ་དང་བརྩེགས་་པའི་ཡ་ཡིག་ནི་དབྱི་སྟེ། གཱནཱི་ཀ་རོ་ཏི། གནཱའི་བུར་བྱེད། ཙཱ་ཙ། ཙཱི་ལ་འང་ངོ༌། །ཙཱི་ཕ་རོལ་ལ་རྟགས་མཐའ་ཨིཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡིན་ན་རིང་པོར་འགྱུར་ཏེ། ཤུ་ཙི་ལ་ཙཱི། ཤུ་ཙཱི་བྷ་བ་ཏི། གཙང་མ་བྱུང་། པ་ཊུ་ལ་ཙཱ། པ་ཀླུ་སྙཏ། མཁས་པར་གྱུར། །འགྲེལ་་་པར་ལུང་དྲངས་པ། ཨ་བྱ་ཡ་ཀྱེ་ཏུ་དཱིགྷོ་ནེ་ཏི་ཀེ་ཙི་ཏ། མི་ཟད་པའི་ཨི་རིང་་པོ་དག་ལ་ལར་མི་འགྱུར་རོ། །མི་ཟད་པའི་སྡེ་ཡིན་ན་ཙི་ཕ་རོལ་ལ་ཨ་ཨཱིར་འགྱུར་བ་དང་ཨི་ཀ་རིང་པོར་འགྱུར་བ་སོགས་ལ་ལར་མི་བྱེད་དེ། །ཙཱིདི་བཱ་སྙཏ། ཉིན་མོར་གྱུར། ་ཙཱསྭ་སྟི་སཱུཏ། བདེ་ལེགས་སུ་གྱུར་ཅིག། ཨ་རུཏྠ་ན་བྷ་ལྕེ་ཏོ་ར་ཧོ་ར་ཛ་ཧཱུཾ་ལོ་པཉྩ། ཨ་རུས་མ་ནས་ཙཀླུས་ཙེ་ཏས་ར་ཧས་ར་ཛས་རྣམས་ཀྱི་དབྱིའོ། །ཨ་་་རུས་སོགས་འདི་རྣམས་ཀྱི་མཐའི་ས་ཡིག་རྣམས་ཙཱི་ཕ་རོལ་ལ་དབྱེ་སྟེ། རྩྭ་ས་ཕྱིས་ཨིཀ་རིང་པོ། ཨ་རཱུ་ཀ་རོ་ཏི། རྨ་བྱུང་ངོ༌། །ཙཱི་ས་ཕྱིས་ཨིཀ་རིང་པོ། ཨ་རཱུ་བྷ་བ་ཏི། རྨ་བྱེད་དོ། །ཙཱི་ས་ཕྱིས་ཨིཀ་རིང་པོ། …ཨ་རུ་སྙཏ། རྨར་གྱུར། དེ་བཞིན་དུ། ཨུདྨ་ནཱི། སྨྱོ་བ། ཨུཙྪ་ཁྱུ། གྱེན་མིག། བི་ཙེ་ཏཱི། སེམས་ལོག། བ་ར་ཧཱི། ཚིགས་འདོད། བི་ར་ཛཱི། རྡུལ་མེད། སོགས་ཏེ་དེ་་རྣམས་ལ་ཨ་ཀ་རོ་ཏི་དང་བྷ་བ་ཏི་དང་སཱུཏ་སྦྱོར་བ་སོགས་རིགས་འགྲེའོ། །བི་བྷཱ་ཥཱ་སཱ་ཏི་ཀཱནྡྷེ། རྫོགས་པ་ལ་སཱ་ཏི་ཡང་ན་འོ། །རྫོགས་པའི་དོན་དང་འབྲེལ་་་པའི་ཙཱིའི་ཡུལ་ལ་སཱ་ཏི་ཡང་ན་འགྱུར་རོ། །སཱཏྠ་དཱ་ཏྱཾ། སཱཏ་དང་ཚིག་གི་དང་པོ་དག་གིའོ། །ས་ཥར་འགྱུར་བའི་རྒྱུ་མཚན་ཅན་གྱི་ཡི་གེ་སྔོན་ན་ཡོད་ཀྱང་ཚིག་ཁོ་རང་གི་ཆ་ཤས་མ་ཡིན་པའི་ཚིག་གཞན་གྱི་ཐོག་མའི་ས་ཡིག་དང་རྐྱེན་སཱ་ཏི་ས་ནི་ཥར་མི་འགྱུར་རོ། །ཚིག་གཞན་གྱི་ཐོག་མ་ལ་མི་འགྱུར་བའི་དཔེར་བརྗོད་ནི། ད་དྷི་སཉྩ་ཏི། ཞོ་ལྕགས་པར་བྱེད། ཀྲྀཙྩཾ་ཤཧཱུཾ་ཨགྣིསཾ་པ་རྱ་ཏེ་ཀ་རོ་ཏི། མེ་ཡི་མཚོན་ཆས་རྫོགས་།
