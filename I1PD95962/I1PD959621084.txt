ཙརྩཉྩ། ཡ་ཛ་ཡཱ་ཙ་རུ་ཙཔྲ་བ་ཙ:རཾཙའི་འང་ངོ༌། །ཡཛ་ལྷ་མཆོད་པ། ཡཱཙ་སློང་བ། རུཙ་འབར་བ། །པྲ་སྔོན་མའི་བ་ཙ་བརྗོད་པ། རཾཙ་མཆོད་པ་རྣམས་ཀྱིའང་རྒྱ་ཕ་རོལ་ལ་ཀ་སྡེར་འགྱུར་བ་མ་ཡིན་ཏེ། ཡཱ་ཛཾ། མཆོད་བྱ། ཡཱ་ཧཱུཾ། སློང་བྱ། རོ་ཙྩཾ། འབར་བྱ། པྲ་བཱ་ཙྩཾ། རབ་ཏུ་བརྗོད་བྱ། ཨརྩི། མཆོད་བྱ། འགྲེལ་པར་རྱ་ཛ་ལང་ཚོ་ཉམས་པ་དང་པཱུཛ་མཆོད་པ་དག་གིའང་ཅུ་ཕ་རོལ་ལ་ཀ་སྡེར་འགྱུར་བ་མ་ཡིན་ནོ་ཞེས་བཤད་པས། ཏཱ་ཚིགས། རྒན་པོ། པཱུ་ཛྭ། མཆོད་བྱ། བ་ཙོཉཤརྡ་སཾ་ཛཱ་ཡཾ། བ་ཙའི་སྒྲའི་མིང་མ་ཡིན་པ་ལའོ། །བ་ཙའི་སྒྲའི་མིང་སྟེ་བརྗོད་པའི་དོན་མ་ཡིན་པ་ལ་ཀ་སྡེར་མི་འགྱུར་ཏེ། བཱ་ཙྩཾ། མི་སྨྲ་བ། བརྗོད་པའི་དོན་ལ་ཀ་སྡེར་འགྱུར་བས། བཱ་ཀྱི། བརྗོད་བྱ། པྲ་ཡོ་ཛ་ནི་ཡོ་ཚོ་ཤ་ཀྵརྗེ། པྲ་་ཡོ་ཛ་ནི་ཡོ་ཚུ་དག་ནུས་པའི་དོན་ལའོ། །པྲ་དང་ནི་སྔོན་མའི་ཡུ་ཛིར་སྦྱོར་བ་ལ་ཏྱཱཏ་སྦྱར་བས། །པྲ་ཡོ་ཛུ་དང་ནི་ཡོ་ཚུ་དག་ནུས་པའི་དོན་བརྗོད་པ་ལ་ཚིག་ཕྲད་དོ། །ནུས་པའི་དོན་མིན་ན་ཛ་གར་བསྒྱུར་བས། པྲ་ཡོ་གྱོ། རབ་ཏུ་སྦྱོར་བྱ། ནི་ཡོ་ཧཱུྃ། ངེས་པར་སྦྱོར་བྱ། །འགྲེལ་པ། བྷུ་ཛའི་འང་ཟ་བའི་དོན་ལ་ཀ་སྡེར་མི་འགྱུར་རོ་ཞེས་བཤད་པས། བྷོ་ཛུ་མནྣཾ། བཟའ་བྱ་བས། །ཟ་བའི་དོན་མིན་ན་ཀ་སྡེར་འགྱུར་ཏེ། །བྷོ་གྱི། སྐྱོང་བྱ། །ཨོ་རཱ་བ་བྱ་ཀེ། །ཨུ་ལས་ངེས་པ་ཅན་ལའོ། །བྱིངས་ཨུ་ཡིག་མཐའ་ཅན་ལས་གདོན་མི་ཟ་བའམ་ངེས་པ་ཅན་གྱི་དངོས་པོ་རྟོགས་པར་བྱེད་པའི་དོན་ལ་ཅུཏ་འགྱུར་ཏེ། །ལུཉ་གཅོད་པ་ལ་རྱཏ། ཨཙའི་འང་་ཉ་ཎ་ཀྱི་འགྲོ་བ་ལ་འོས་འཕེལ་བ། ཡའི་རྐྱེན་ལ་བའི་མཐའ་ཅན་ནོས་ཨོ་ྋཨཱ་བར་བསྒྱུར་ཅིབས། ཨ་བ་བྱ་ལཱ་བྱ། ངེས་པར་གཅད་བྱ། སོགས་སོ། །ཨཱ་སུ་ཡུ་བ་པི་ར་པི་ལ་པི་ཏྲ་པི་ཙ་མ། ཨཱ་སུ། ཡུབ་པིར་པི;ལ་པི；ཏྲ་པི་ཙམ་ལས་ཀྱང་ངོ༌། །ཨཱང་སྔ་མའི་ཥཱཉ་མངོན་པར་སྐྱེ་བ། ་་ཡུ་སྤེལ་བ། །།
