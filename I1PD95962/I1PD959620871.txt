ནི་ལོ་ལོའི་ཚེ་ཀ་དམ་པ་བློ་ནི༑་ཨ་ས་ནི་ཥཱ་ཐཱ། ཨ་ས་ནི་དྡྷཾ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ས་ནི་ཏི། ཨ་ས་ནི་ཥཱ་ཧི། ཨ་ས་ནིཥྛ་ཧི། ལིཊ་ལ་གཉིས་བརྗོད། སྡིཏ་རྣམས་ལ་ཨ་བྷཱུ་ས་དབྱི་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ས་སཱ་ན། ས་ན་ཏུ། སེ་ནུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ལུས་ས་ནི་ཐ། སེ་ན་ཐུ། སེ་ན། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ས་སཱ་ན། ས་སན། སེ་ནི་བ། སེ་ནི་མ། ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །སེ་ནཾ། སེ་ནཱ་ཏེ། །སེ་ན་ཏེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ་སེ་ནི་ཥ། སེ་ནཱ་ཐེ། སེ་ནི་དྷེ། ་སེ་ནི་རྗེ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། སེ་ནེ། སེ་ནི་བ་ཧེ། སེ་ནི་མ་ཧེ། ལུཊ་ལ་ཨིཊ་་་བྱིན་པས། ཨ་ཏང་ལ། ་ས་ནི་ཏཱ། །ས་ནི་ཏཱ་སི། སོགས་དང་། ཏང་ལ། ས་ནི་ཏཱ། ས་ནི་ཏཱ་སེ། །སོགས་སོ། །སྨོན་པའི་ལིང་གི་ཨ་ཏང་ལ། ་ཡལ་་་འདམ་ཀ་འོས། ་ཊི་ཨཱཏ་དུ་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་ཡང་ན་བར་བྱས་པས། སཱ་ཡཱཏ། སཱ་ཡཱནྟཾ། སཱ་ཡཱ་སུ། སོགས་སམ། ་ས་ནཱཏ། ས་ནཱནྟཾ། ས་རྩྭ་སུ༑སོགས་སོ། །ཏང་ལ་ཨིཊ་བྱིན་པས། །ས་ནི་ཏཱི། ས་ནི་ཏཱི་ཡཱཙཱ། ས་ནི་ཏཱི་རན། སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་པས། ཨ་ཏང་ལ། ས་ནི་་ཏི༑སོགས་དང༌། ཏང་ལ། ས་ནི་ཉྩ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ཁྲིང་གི་ཨ་ཏང་ལ། ཨ་ས་ནི་་ཏ། སོགས་དང་། ཏང་ལ། ས་ཅིས་ནི་་ཏ༑སོགས་སོ། །:༈ཀྱི་ཆུ་འཚེ་བ་ལའོ་ཞེས་པ་ལ་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་སྦྱོར་ཏེ། འགྲེལ་པར་ལུང་དྲངས་པ། །ཨུ་པྲ་ཏྱ་ཡེ་ཏ་ནཱ་དེ་རྣོ་པ་དྷཱ་གུ་ཎ། ཨུ་རྐྱེན་ལ་ཏ་ན་ལ་སོགས་པའི་ཨུ་པ་དྷཱ་ཡོན་ཏན་མ་ཡིན་ནོ། །རྐྱེན་ཨུ་ཕ་རོལ་ལ་ཏ་ན་ལ་སོགས་པ་རྣམས་ཀྱི་ཨུ་པ་དྷཱ་རཾ་ཡིག་ཡིན་ན་ཡོན་ཏན་ཡང་ན་བར་བྱེད་ལ། —།
