གསུམ་ལ། བི་བྷ་ཡི་ཐ། བི་བྷེ་ཐ། བི་བྷུ་ཐུ། བི་ཤྩ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བྷཱ་ཡ། བི་བྷ་ཡ། བི་བྱི་བ། བི་བྱི་མ། ལུཊ་ལ་བྱིངས་་་་ཐམས་ཅད་པ་དང་བྱིངས་ཕྱེད་པ་སོགས་ཀྱིས་ཡོན་ཏན་བྱས་པས། བྷེ་ཏཱ། བྷེ་ཏཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བྷི་ཡཱཏ། བྷི་ཡཱནྟཾ། བྷི་ཡཱ་སུ། སོགས་སོ། །ཁྲིཊ་ལ་ཡོན་ཏན་བྱས་པས། བྷེ་ཥྱ་ཏི། སོགས་དང་། ཁྲིང་ལ། ཨ་བྷི་ཏ། སོགས་སོ། །་། །།ད། དལ་་བ་བྱ༈ཧི་ངོ་ཚ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཞིང་ཆོ་ག་ཕལ་ཆེར་་བཤད་མ་ཐག་པ་དང་འདྲ་སྟེ། ལཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛི་ཧྲི་ཏི། ཛི་ཧི་ཏ། ཛི་ཧྲི་ཡ་ཏི། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛི་ཧྲེ་ཏི། ཛི་ཧི་ཐ། ཛ་ཧི་ཐ། ་་སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛི་ཧི་མི། །ཛི་གྷི་བ། ཛི་ཧི་མ། ལིང་ལ། ཛི་གྷི་ཡཱཏ། །ཛི་ཤཱི་ཡཱ་ཏྭཾ། ཛི་གྷི་ཡུ། ་སོགས་སོ། །ལོཊའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛི་ཧྲེ་ཏུ། ཛི་ཧཱི་ཏཱཏ། །ཛི་ཧཱི་ཏཱཾ། ཛི་ཧྲི་ཡ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛི་ཧི་ཧི། ཇི་ཧཱི་ཏཱཏ། ཛི་ཧཱི་ཏཾ། ཛཱི་ཧི་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛི་ཏྲ་ཡཱ་ནི། ཛི་ཏྲ་ཡཱ་བ། ཛི་ཏྲ་ཡཱ་མ། ་ལང་གི་་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་ཛི་ཧྲེཏ། ཨ་ཛི་ཧཱི་ཏཱཾ། །ཨ་ཛི་ཧྲ་ཡུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཛི་བྲེ། ཨ་ཛི་ཧཱི་ཏཾ། ཨ་ཛི་ཧི་ཏ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། །ཨ་ཛི་ཧྲ་ཡཾ། ཨ་ཛི་ཧི་བ། ཨ་ཛི་ཧི་མ། ལུང་ལ། སིཙ་ལ་་་འཕེལ་བ་བྱས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་ཧྲི་ཏ། ཨ་ཧྲིཏྟཾ། ་ཨ་ཧྲེ་ཥཱ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཧྲི་ཏི། འ་ཨ་ཧྲིཝཾ། །ཨ་ཧྲིཥྛ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཧྲི་ཏཾ། ཨ་ཧྲི་ཁྱེ། ཨ་ཧྲི། །ལིཊ་ལ། ཨཱམ་ཡང་ན་བར་བྱས་པས། ཇི་ཧྲ་ཡཾ་ཙ་ཀཱ་ར། ཨ་ཧྭ་ཡཾ་བ་བྷུ་བ། ཇི་ཧྲ་ཡཱ་མཱ་ས། octong—།
