ཙོ་ཚོགས་པའི་ཕྱི་བའི་་་་ར་མོ་༽བསྟན་འགྱུར། སྒྲ་མདོ། ཏི་《། གཏི་ཉི་མ་ཆོས་ཀཚོ་དང་སྨན་རི་བ་བློ་བཟང་གོང་འོག་བརྗེ་བའང་ཡང་ན་བ་བྱའོ་ཞེས་བཤད་པས། པཱ་ད་སྱ་ཏ་ལཾ། ཞབས་ཀྱི་མཐིལ། ཞེས་པ་ལ། རྣམ་དབྱེ་ཕྱིས། །གོང་འོག་:བརྗེ་མི་བརྗེ་ཡང་ན་བར་་་་བྱས་པས། །པཱ་ད་ཏ་ལཾ། ཞབས་ཀྱི་མཐིལ། ཏ་ལ་པཱ་དཾ། དེ་བཞིན་དུ། །ཧསྟ་སྱ་ཨ་གྲོ༑ཕྱག་གི་རྩེ། ཞེས་པ་ལ། །ཧསྟ་གྲི། ཕྱག་གི་རྩེ། ཨ་གྲ་ཧནྟཾ། པཱཔ་པནྡྷེ་ཙ་དཱི་ཏཱི་ཡ་ཡཱ། པཱལྟ་ཨཱ་པརྣ་ལའང་གཉིས་པའིའོ། །པཱསྟ་དང་ཨཱ་པནྣ་གཉིས་དང་སྦྱོར་བ་ལས་གཉིས་པའི་མཐའ་དོན་གཅིག་པར་འགྱུར་ཞིང་རྟགས་གོང་འོག་བརྗེ་བའང་། ཡང་ན་བྱ་སྟེ། པཱསྟོ་ཛཱི་བི་ཀཾ། ཞེས་པ་ལ། །རྣམ་དབྱེ་བསྡུས་རྟགས་གོང་འོག་བརྗེ་བའམ་མི་བརྗེ་བ་ཡང་ན་བར་བྱས་པས། པྲཔཱ་ཛཱི་བི་ཀཱ། འཚོ་བ་ཐོབ། ཛཱི་བི་ཀཱ་པཱསྟཱ། འཚོ་བ་ཐོབ། དེ་བཞིན་དུ། ཨཱ་པརྣོ་ཛཱི་བི་ཀཾ། འཚོ་བ་ལ་ཞུགས། ཞེས་པ་ལ༑ཨཱ་པནྣ་ཛཱི་བི་ཀཱ། འཚོ་བ་ལ་ཞུགས། ཛཱི་བི་ཀཱ་ཨཱ་པནྣ། །ཀཱ་ལཱཔ་རི་མཱ་ཎི་ནཱ༑དུས་རྣམས་ཀྱི་ཚད་ལའོ། །དུས་ཀྱི་ཚད་བརྗོད་པ་ལ་དྲུག་པའི་མཐའ་དོན་གཅིག་པར་འགྱུར་ཏེ། མཱ་སོ་ཛཱ་ཏེ་སྱ། ཟླ་བ་སོང་། ཞེས་པ་ལ། རྣམ་དབྱེ་བསྡུས། མཱ་ས་ཛཱ་ཏ། སཔྟ་མཱི་ཤོཎྜི། །ཤོཎྜ་རྣམས་ཀྱི་བདུན་པའིའོ། །ཤོཎྜ་སོགས་་་རྣམས་དང་སྦྱོར་བ་ལས་བདུན་པའི་མཐའ་སུཔ་དང་སུཔ་༔དོན་གཅིག་པ་དེའི་སྐྱེས་བུའི་བསྡུ་བར་འགྱུར་ཏེ། ཨཀྵེ་ཥ་ཤོཎྞ། ཤོ་ལོ་རྣམས་ལ་མཁས་པ། ཞེས་པ་ལ། ་ཨ་་ཀྟ་ཤོཎྞ། ཤོ་ལོ་མཁན། དེ་བཞིན་དུ། །བྱ་བ་ཧ་རེ་ཎུ། ཚོང་ལ་རེམ་པ། བྱ་བ་ཧར་་་དྷཱ༑ཚོང་ལ་རེམ་པ། དེ་བཞིན་དུ་རྣམ་དབྱེ་བདུན་པའི་རྗེས་བསྡུས་པས། རི་ཡསཱ་ར་སཾ་བཱི་ཏ། དབྱངས་ལ་མཁས་པ། རི་…པ་ཀ་པྲ་བཱི་ཎ། འཚོང་མཁན། ཇི་…བི་དྱ་པཎྜི་ཏ། ཡོན་་ཏན་མཁས་པ། བསྟན་བཅོས་དགེ་བ། སཱ་སྭ་ཀུ་ཤ་ལ། ཇི་ནེ་ཏྲ་ཙ་པ་ལ། མིག་བྲེལ་པོ། སིདྡྷ་་་ཤུཥ་པ་ཀཱ་པནྟི། གྲུབ་པ་བསྐམས་པ་ཚོང་བ་འཆིང་བ་རྣམས་ཀྱིའང་ངོ༌། །གྲུབ་པ་སོགས་དེ་རྣམས་དང་སྦྱོར་བ་ལའང་བདུན་པའི་མཐའ་དོན་གཅིག་པར་་་་—།
