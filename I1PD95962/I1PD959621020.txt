སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙརྐ་རི་མི། ཙཀརྨི། ཙཀྲྀ་བ། ཙཀྲྀ་མ། ལིང་ལ། ་ཙཀྲྀ་ཡཱཏ། ཙཀྲྀ་ཡཱ་ཏྭཾ། ཙཀྲྀ་ཡུ། སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཙ་རཱ་ཏུ། ཙརྐཆུ༑ཀེཙཀྲྀ་ཏཱཏ། ཙཀྲྀ་ཏྭཾ། ཙཀྲ་ཏུ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཙཀྲྀ་ཧི། ཙཀྲྀ་ཏཱཏ། །ཙཀྲྀ་ཏཾ། ཙཀྲྀ་ཏ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཙརྐ་རཱ་ཎི། །ཙཀ་རཱ་བ། ཙརྐ་རཱ་མ། དེ་བཞིན་དུ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཨ་ཙརྐ་རཱིཏ། ཨཙརྐཨ་ཙཀྲྀ་ཏྭཾ། །ཨ་ཙ་ཏུ། སྐྱེས་བུ་དབུས་མ་་གསུམ་ལ། ཨ་ཙརྐ་རཱི། ཨ་ཙརྐ། ཨ་ཙཀྲྀ་ཏཾ། ཨ་ཙཀྲྀ་ཏ། ལ་སྐྱེས་བུ་་ཕྱི་མ་གསུམ་ལ། ཨ་ཙརྐ་རཾ། ཨ་ཙཀྲྀ་བ། ཨ་ཙཀྲྀ་མ། ལྷག་མ་རྣམས་ལ། ལུང་…ཨ་ཙཎྜི་ཏ། ཨིཊ་བྱིན་ཨིཀ་གི་ཨཙ་ལ་ཡཎ། ལིཊཙཎཾ་ཙ་ཀཱ་ར། ཨཱམ་སྦྱར་ཨིཀ་གི་ཨཙ་ལ་ཡ་ཎ། ལུཊཙཀྲི་ཏཱ། སྨོན་ལིང་ཙཎི་ཡཱཏ། ཤ་ཡིག་ལིང་རྣམས་ལ་རིང་འོས་རི་རིང་དུ་བསྒྱུར། ཁྲིཊཙཎི་ཥྛ་ཏི། ཁྲིང་ཨ་ཙཎི་ཥྛཏ། སོགས་སོ། །རིཀ་དང་རཱི་ཀ་གཉིས་ཀྱང་གློ་བུར་བའི་ཁྱད་པར་མ་གཏོགས་ལཊ་སོགས་དང་ཕྲད་པའི་ཆོ་ག་འདྲ་སྟེ། ཙ་རི་ཀ་རཱི་ཏི། ཙ་རི་ཀརྟི། སོགས་སོམི། ཙ་རཱི་ཀ་རཱི་ཏི། ཙ་རཱི་ཀརྟི། སོགས་སོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱཏ་ཡང་ལུ་གནྟུས་མཱམྦྷ། །ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་་ར་ཎའི་མདོ་ལས་ཡང་འཇིག་པའི་མཐའ་རྣམས་རྫོགས་སོ། །།།་་་་་དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས། །ཡང་འཇིག་པའི་མཐའ་ཅན་གྱི་ཆོ་ག་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པ་རགས་པ་རྣམས་རེ་་་ཞིག་རྫོགས་སོ། །་། ཏིནི་ནད། ཏ་། ཏེ་དམ་པ་ཕུལ་ལ་་་་་བྱ—།
