ལོ་ནས་ཏི་ཎའི་བསྟན་འགྱུར། སྒྲ་མདོ། །ཏི་མ་པོ་ནི་འོ་མ་ཁོ་ཚོ་ལ་ཎཝཀའ་བློ་ཁའི་ལིང་ལ། ཛགྲོ་དྷཱཏ། ཛགྲོ་དྷཱ་ཏཱ། ཛགྲོ་དྡྷ། །སོགས་སོ། །ལོཊ་ལ་ལཊའི་སྐབས་ལྟར་བསྒྲུབས་པས། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཛཀྲྀ་དྷཱི་ཏུ། ཛརྒཛྷ༑ཛགྲོ་དྷཱཏ། ཛགྲོ་དྡྷཾ། ཛགྲོ་དྷ་ཏུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཛགྲིདྡྷིཛཀྲྀ་དྷཱཏ། ཛ་ཛཾ་དྷཾ། ཛགྲིདྡྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཛགྲོ་དྷཱ་ནི། ཛགྲོ་དྷཱ་བ། ཛགྲོ་དྷཱ་མ། དེ་བཞིན་དུ་ལང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཨ་ཛཐཾ་དྷཱིཏ། ཨ་ཛརྒ། ཨ་ཛགཽ་དྡྷཾ། པཨ་ཛརྦ་དྷ། ་སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཛཊཾ་དྷཱི། ཨ་ཛརྒ། ཨ་ཛཤྲིདྡྷཾ། ཨ་ཛགྲིདྡྷ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཛཊཾ་དྷཾ། ཨ་ཛགྲོ་དྷ། ཨ་ཛགྲོ་དྨ། །ལྷག་མ་རྣམས་ལ། ལུང་…ཨ་ཛགྲོ་དྷཱིཏ། ལིཊ་ཡཛགྲོ་དྷཱ་ཙ་ཀཱ་ར། ་ལུཊཛགྲོ་དྷི་ཏཱ། སྨོན་ལིང་ཛགྲོ་དྷཱཏ། ཁྲིཊཛགྲོ་དྷི་ཥྛ་ཏི། ཁྲིཊ་…ཨ་ཛཀྲྀ་དྷི་ཏ། །སོགས་སོ། །ཡང་ན་རིཀ་འམ་རཱིཀ་ཨེཨཱ་ག་མ་སྦྱར་བས། །ཛ་རི་གྲོ་དྷཱི་ཏི། །ཛ་རི་གདྷི། སོགས་སམ། ཛ་རཱི་གྲོ་དྷཱི་ཏི། ཛ་རཱི་གདྷི། སོགས་སོ། །།།མཚོ་༈རྔུ་ཀྲྀཉ་བྱེད་པ་ལའོ་ཞེས་པ་ལ། ་ཡང་གཉིས་བརྗོད། རིའི་ཨཏ། ཀུ་ཧ་དག་གི་ཙུ། རྀ་ཏ། རྀཏ་ལས་ཀྱང་ངོ༌། །བྱིངས་རིཏ་མཐའ་ཅན་ལས་ཀྱང་ཡང་འཇིག་པ་ཕ་རོལ་ལ་ཨ་བྷཱ་ས་ལ་རུཀ་དང་རིག་དང་རཱིཀ་ཨཱ་ག་མ་རྣམས་འགྱུར་ཏེ། མདོ་འདིས་རུཀ་ཨཱ་ག་མ་བྱིན། ཡང་གི་ཨཙ་ལའང་ངོས་ཡང་ཕྱིས་པས། །ཙཀྲྀ་ཞེས་པ་ལ་ལཊ་སོགས་སྦྱོར་ཏེ། སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨཱིཊ་ཡང་ན་བར་བྱས་པས། ཙརྐ་རཱི་ཏི། ཙརྐཏི། གཉིས་ཚིག་ལ། ཙི་:ཀྲྀ་ཏ། མང་ཚིག་ལ། །ཨ་བྱཱསྟ་ལས་ཨཏ་འོས། །ཛྷ་ཨཏི། ཨིཀ་གི་་ཨཙ་ལ་ཡཎ་བྱས་པས། །ཙཀྲ་ཏི། དེ་རྣམས་དང་འདྲ་བར་བསྒྲུབས་པས། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙཀ་རཱི་ཎི། ཙརྐནྟི། ཙཀྲྀ་ཐ། ཙཀྲྀ་ཐ། །
