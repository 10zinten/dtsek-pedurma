མ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཞ་བརྩེགས་པའི་དང་པོ་ས་ཀ་སོགས་ཀྱིས་ས་ཕྱིས། བྲཤྩ་བྷསྔ་སོགས་ཀྱིས་ས་ཥ། །ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས་ཥ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། བྲཱིཀི། བྷཀཱི་ཡཱ་ཧཱུཾ། བྷ་ཀཱི་རན། སོགས་སོ། །།ཁྲིཊ་ལ་དེ་དང་འདྲ་བར་བསྒྲུབས་པས། ་བྷརྫར་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། བྷཀྵ་ཏི། སོགས་དང༌། ཏང་ལ། བྷཀྵ་ཏེ། སོགས་སོ། །བྷརྫ་མ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། བྷ་གྱ་ཏི། སོགས་་་་དང་། <ཏང་ལ། ་བྷ་ཀྵ་ཏེ། སོགས་སོ། །དེ་བཞིན་དུ། ལྀང་ལ། བྷརྫར་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། །ཨ་བྷརྒྱུ་ཏ། སོགས་དང་། ཏང་ལ། ཨ་བྷནྡྷཱ་ཏ། སོགས་སོ། །བྷརྫར་མ་བསྒྱུར་བའི་ཕྱོགས་ཀྱི་ཨ་ཏང་ལ། ཨ་བྷནྡྷཱཏ། །སོགས་དང་། །ཏང་ལ། །ཨ་བྷགྱུཏ། སོགས་སོ། །༈མུཙླི་ཐར་པ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་དང་ཏང་གཉིས་ཀ་འཇུག་སྟེ། ཤེ་མུ་ཙཱ་དཱི་ནཾ། ་མུཙ་ལ་སོགས་པ་རྣམས་ཀྱི་ཤ་ལའོ། །མུཙྪ་ཐར་པ། ལུབློ་གཅོད་པ། བིདྨོ་རྙེད་པ། ལི་པ་བྱུག་པ། ཎི་ཙ་འཛག་པ། ཀྲྀ་ཏཱི་གཅོད་པ། པི་ཤ་ཆ་ཤས། ཏི་ད་ཡོངས་སུ་འཇོག་པ་རྣམས་ནི། མུཙ་ལ་སོགས་པ་སྟེ་འདི་རྣམས་ལ་རྐྱེན་ཤ་ཕ་རོལ་ལ་ནུམ་ཨཱ་ག་མ་འགྱུར་ཏེ། །་ནུམ་བྱིན་པས། ལཊའི་་ཨ་ཏང་ལ། །མུཉྩ་ཏི། སོགས་དང༌། ཏང་ལ། མུཉྩ་ཏེ། སོགས་སོ། །ལིང་གི་ཨ་ཏང་ལ། མུཉྩཏ། སོགས་དང་། ཏང་ལ། མུཉྩེ་ཏ། སོགས་སོ། །ལོཊའི་ཨ་ཏང་ལ། མུཉྩ་ཏུ། སོགས་དང་། ཏང་ལ། མུཉྩ་ཏྭཾ། སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ་ཨ་མུཉྩ་ཏ། །སོགས་དང་། ཏང་ལ། ཨ་མུཉྩ་ཏ། སོགས་སོ། །ལུང་གི་ཨ་ཏང་ལ་པུཀ་མཐའ་སོགས་བྱཱཏ་སོགས་ཀྱིས་ཙྪི་ཨང་་་་བསྒྱུར་བས། ཨ་མུ་ཙ་ཏ། ཨ་མུ་ཙ་ཏཱ། ཨ་མུ་ཙན། སོགས་དང་། །ཏང་—།
