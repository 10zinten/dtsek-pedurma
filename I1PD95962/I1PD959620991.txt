སྒྲ་མདོ། ཏིའི་པ་བྱིན་ན་ང་ཚོ་ལ་མ་རྒྱུད་ཚོ་ལ་བློ་གཉིས་བརྗོད། །ཀུ་ཧ་དག་གི་ཙུ་དང་ཡོན་ཏན་བྱས་པས། ་ལཊ་…ཛེ་གཱི་ཡ་ཏེ། ལུང་…ཨ་ཛེ་གཱི་ཡི། ལིཊཛེ་གཱི་ཡཾ་ཙ་ཀྲེ། ལུཊཛི་གི་ཡི་ཏཱ། སོགས་སོ། །།༈ཨོ་ཧཱ་ཀ་གཏོང་བ་ལ་ཡང་སྦྱར། གླུ་སོགས་ཀྱིས་ཨཱ་ཨཱི། གཉིས་་་་བརྗོད། ཀུ་ཧ་དག་གི་ཙུ། །ཨ་ངྷ་སའི་ཙར་འང་ངོས། ཛྷ་ཛ། །ཡོན་ཏན་བྱས་པས། ལཊཛེ་ཧི་ཡ་ཏི། ལུང༌…ཨ་ཛེ་ཧཱི་ཡིཝ། ་ལིཊཛཾ་ཧཱ་ཡཱ་ཙ་ཀྲེ། ལུཊ…ཛི་ཧཱི་ཡི་ཏཱ། སོགས་སོ། ་། །ཁ་ན་ཆུ་དང་ཟླ་ཁ་ཚ་ཏ@aང་ག་་། ༈ཁོ་མཐའི་ལས་ལའོ་ཞེས་པ་ལ་ཡང་སྦྱར། ཤ་འགྲོ་བ་ཉེ་བའི་ཡུལ་མ་་ཡིན་པ་སོགས་ཀྱིས་ཨེཙ་ཨཱཏ། གླུ་སོགས་ཀྱིས་ཨིཏ། གཉིས་བརྗོད་ཡོན་ཏན་དང་ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། ལཊསེ་ཏི་ཡ་ཏེ། ལུང་ཨ་སཱི་ཏཱི་ཡི། ལིཊསེ་ཊི་ཨཾ་ཙ་ཀྲེ། ལུཊསེ་ཁཱི་ཡི་ཏཱ། སོགས་སོ། །ཁེ། ༈ཊུ་ཨོ་ཤཱི་འགྲོ་བ་དང་འཕེལ་བ་དག་ལའོ་ཞེས་པ་ལ་ཡང༌། ་ཤྲཱིའི་འདམ་ཀ་འོས། སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་བ་དང་མི་བསྒྱུར་བ་འདམ་ཀ་ཅན་དུ་བྱས་པས། བསྒྱུར་བའི་ཕྱོགས་ལ། ལཊཤོ་ཤུ་ཡ་ཏེ། ལུང་…ཨ་ཤོ་ཤུ་ཡི། ལིཊཤོ་ཤུ་ཡཾ་ཙ་ཀྲེ། ལུཊཤོ་ཤུ་ཡི་ཏཱ། སོགས་དང༌། མ་བསྒྱུར་བའི་ཕྱོགས་ལ། ལཊ་ཤེ་ཤཱི་ཡ་ཏེ། ལུང་…ཨ་ཤེ་ཤཱི་ཡིཋ། ་ལིཊ་ཤེ་ཤཱི་ཡཱ་ཙ་ཀྲེ། ལུཊཤེ་ཤཱི་ཡི་ཏཱ༑སོགས་སོ། །་ལོ་པ་ན་མ། ་། ཏ་ཏོ་ལ་སོགས་པར་དང་བབ། ༈སྨྲ་དྲན་པ་ལའོ་ཞེས་པ་ལ་ཡང་སྦྱར། གཉིས་བརྗོད། ཡང་འཇིག་པ་སོགས་:ཀྱིས་ཨ་ཧྥུ་ས་ཡོན་ཏན། ཧལ་དང་པོ་ལྷག་པོ་སོགས་ཀྱིས་མ་དང་ར་ཕྱིས༑ཀ་འགྲོ་བ་མ་ཡིན་པའི་རིང་པོ་འོས་རིང་པོར་བསྒྱུར། ཡང་ལ་འང་་་ངོས་ཡོན་ཏན་བྱས་པས། ལཊ་སཱ་སྨརྦ་ཏེ། །ལུང་…ཨ་སྦ་སྨ་རི། །ལིཊསཱ་སྨྲ་རོ་ཙ་ཀྲེ། ལུཊ'སཱ་སྨ་རི་ཏཱ། །སོགས་སོ། །སྐུ་ཅི་སྟེ། །ཁ་ར—།
