སྒྲ་མདོ། ཏི༽*ཁ་ཏིནི་ཁ་ཤོ་བའི་ཚོང་ཁང་ཀྱི་གཙོ་སོགས་བག་ས། ཨ་བཱ་སི། ་ཨ་བཱ་སི། ཏང་ལ་ས་འདྲེན་པ་མ་གཏོགས་བྱ་བ་ལྷག་པོ་མེད་དེ། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། །ཨ་བཱས། ཨ་བཱ་སཱ་ཏཱ། ཨ་བཱ་ས་ཏ། སྐྱེས་བུ་་་དབུས་མ་གསུམ་ལ། ཨ་བནྡྷ། ཨ་བཱ་སཱ་ཐཱ། ཨ་བཱ་ནཱཾ། དྷ་ལའང་ངོས་སིཙ་ཕྱིས་པའོ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཝཱ་སི། །ཨ་བཱ་སཱ་ཧི། ཨ་བཱསྨ་ཧི། ལིཊ་ལ། བེཉ་ཉིད་དུ་གནས་པ་དང་། བ་ཡི་རུ་བསྒྱུར་བའི་ཕྱོགས་གཉིས་འབྱུང་སྟེ་བེཉ་་་ཉིད་དུ་གནས་པའི་ཕྱོགས་ལ། བེཉ། བེཉའིའོ། །གོང་དུ་ལིཊ་ལ་གཉིས་་་རྣམས་ཀྱི་ཞེས་སོགས་ཀྱིས་བེཉ་བ་ཙའི་སོགས་ཁོངས་ཡིན་ཀྱང་དེའི་དམིགས་བསལ། ལིཊ་ཕ་རོལ་ལ་བེཉའི་སཾ་པྲ་སཱ་ར་ཎ་མི་བྱེད་དེ། །བེཉ་ལ་ལིཊ་སྦྱར་བས། ཨ་་་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བ་བོ། ཨཱཏ་ལས་ཎལ་གྱི་ཨོ་འོས། ཎལ་ཨོ། །་བ་བ་ཏུ། ས་ཨིཊ་ལས་ཀྱང་ཨཱཏ་ཀྱི་དབྱི་འོས། ཨ་ཕྱིས། བ་བུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་བི་ཐ། །བ་བཱ་ཐ། བ་བ་ཐུ། བ་བ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་བོ། །བ་བི་པ། བ་བི་མ། ཏང་ལའང་ཨིཊ་ལས་ཀྱང་ཨཱཏ་ཀྱི་དབྱི་འོས་ཨཱ་ཕྱིས་པས། སྐྱེས་བུ་་དང་པོ་གསུམ་ལ། བ་བེ། བ་བཱ་ཏེ། བ་བཱི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བ་བི་ཥ། བ་བཱ་ཐེ། ་བ་བི་དྷེ། བ་བི་ལྷེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བ་བེ། བ་བི་བ་ཧེ། བ་བི་མ་ཧེ། ཡང་ཕྱོགས་གཅིག་ལ། །བེ་ཉོ་བ་ཡི། བེཉའི་ཝ་ཡིའོ༑། བེ་ཉ་ནི་ལིཊ་ཕ་རོལ་ལ་བ་ཡི་ཞེས་པར་ཡང་ན་འགྱུར་ཏེ། །ལིཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ་བེ་ཉ། །ཝ་ཡི་རུ་བསྒྱུར་ཨི་བརྗོད་པའི་དོན་ཡིན་པས་ཕྱིས། །ཨ་བྷཱ་ས་མེད་པ་སོགས་ཀྱིས་བྱིངས་གཉིས་བརྗོད། ལིཊ་ལ་གཉིས་རྣམས་ཀྱི་སོགས་ཀྱི་མདོས་ཨ་བྷཱ་ས་སཾ་པྲ་ས་ར་ཎར་བྱས་པས། །ཨུ་བཱ་ཡ། གཉིས་ཚིག་ལ་བྱིངས་གཉིས་བརྗོད། ་བཙི་སཱ་པི་ཡཛ་སོགས་ཀྱི་མདོས། བ་གཉིས་ཀ་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར། ཨཀ་འདྲ་བ་ལ་རིང་པོ་འོས་རིང་པོར་།
