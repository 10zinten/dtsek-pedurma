ཥཱ་ཏ། ཨ་བེ་ཥྛ་ཏྭཾ། ཨ་བེ་ཌུ་ན། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །་ཨ་བེ་། ཨ་བེ་བྱ་ཏཾ། ཨ་བེ་ཥྱ་ཏ། །སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་བེ་ཏྱཾ། ཨ་བེ་བྷཱ་བ། ཨ་བ་བྷཱུ་མ། དྲང་སྲོང་བྷཱ་ར་དྷཱ་ཛའི་འདོད་པ་ལྟར་ཨིཊ་བྱིན་ན། ཨ་བ་ཡི་ཥྛ་ཏ། །ཨ་བ་ཡི་བྱཱ་ཏཱ། ཨ་བ་ཡི་བྱེ་ཏཾ༄༅། །སོགས་ཏེ་རིགས་འགྲེའོ། །ཀླུ་ཟད་པ་་་ལའོ་ཞེས་པའི་བྱིངས་ལའང་ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བའི་ཆོ་ག་རྣམས་ཀྱང་ཕལ་ཆེར་བཤད་མ་ཐག་པ་དང་འདྲ་སྟེ། ལཊ་ལེ། གྷ་ཡ་ཏི། སོགས་སོ༑e། ལིང་ལ། ཀ་ཡེཏ་སོགས་སོ། །ལོཊ་ལ། ཀླུ་ཡ་ཏུ་སོགས་སོ། །ལང་ལ༑ཨ་གྱུཏ། སོགས་སོ། །ལུང་ལ། ཨཀཱི་ཌི་ཏ། ཨ་ཀྱི་ཏྟཾ། ?ཨ་ཀླུ་སོགས་སོ། །ལིཊ་ལ། །ཙིག་ཡ། ཙིའི་ཡ་ཏུ། ཙིཏྟི་ཡུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། །ཙི་ཀ་ཡི་ཐ། ཙེ་ཀྵེ་ཐ། ཙིཀཱི་ཡ་ཐུ། ཙིཀྱི་ཡ། སྐྱེས་བུ་ཕྱི་མའི་གཅིག་ཚིག་ལ། ཙི་གྷཱ་ཡ། ཙི་ཀཱ་ཡ། གཉིས་ཚིག་ལ། །ཙི་ཀླུ་ཡི་བ། ཙི་ཀྲུ་བ། མང་ཚིག་ལ། ཙི་ཀ་ཡི་མ། ཙི་ཀྱི་མ། ལུཊ་ལ་ཨིཊ་ཡང་ན་བར་བྱས་པས། །ཊཀེ་ཏཱའམ། ལ་གླུ་ཡི་ཏཱ་སོགས་སོ། །ཨ་ཀྲྀཏྟཱརྦ་དྷཱ་ཏུ་ཀ་ཡོདྡྷིནྡྷུ། ཀྲྀཏ་མ་ཡིན་པ་བྱིངས་ཐམས་ཅད་པ་དག་ལ་རིང་པོའོ། །ཀྲྀཏ་སོགས་ཀྱི་ནང་ཚན་མ་ཡིན་པའི་རྣམ་དབྱེ་བྱིངས་ཕྱེད་པ་དང་བྱིངས་ཐམས་ཅད་པ་དག་ཕ་རོལ་གྱི་རྐྱེན་ཡ་ཡིག་ཀ་དང་ང་འགྲོ་བའི་མིང་ཅན་ལ་བྱིངས་མཐའི་ཨཙ་རིང་པོར་འགྱུར་ཏེ། སྨོན་པའི་ལིང་ལ། །ྋགི་ཡཱཏ། ཀཱི་ཡཱཙཱ། ཀཱི་ཡཱ་སུསོགས་སོ། །ལྀཊ་ལ་ཨིཊ་ཡང་ན་བར་བྱས་པས། གླེ་ཥྱ་ཏིའམ། ཀླུ་ཡི་བྱ་ཏི་སོགས་སོ། །ཕྲོང་ལའང༌། ཨ་ཀྵེ་ཏའམ། ཨ་ཀླུ་ཡི་ཏ་སོགས་སོ། །སྟེ། ་ལ༈ཀ་ཊེ་ཆར་དང་སྒྲིབ་པ་དག་ལའོ་ཞེས་པའི་བྱིངས་ལ། ལཊ་སོགས་པ་རསྨི་པ་ད་རྣམས་སྦྱར་བའི་བྱ་བ་རྣམས་ཀྱང་ལུང་གི་སྐབས་འཕེལ་བ་མི་བྱེད་པའི་་ངས—།
