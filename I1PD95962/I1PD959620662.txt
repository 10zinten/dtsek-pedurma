ཏྭཾ་སོགས་སོ། །ལང་གི་ཨ་ཏང་ལ། ་ཨ་ཡ་ཛ་ཏ་སོགས་དང༌། ཏང་ལ་ཨ་ཡ་་ཛ་ཏ་སོགས་སོ། །ལུང་གི་ཛྷལ་དང་པོ་རྣམས་ལ། །ཛྷལ་ལས་ཛྷལ་ལ་འོས། སིཙ་ཕྱིས། བྲཐུ་བྷསྔ་སོགས་ཀྱིས་ཛ་ཥ། ཥ་ཊུ་ལ་ཥ་ཊུ་འོས་ཏ་ཊར་བསྒྱུར་བ་སོགས་དང་ཛྷལ་དང་པོ་མ་ཡིན་པ་རྣམས་ལ། ས་ལ་ཥ་ཌྷ་དག་གི་ཀ་འོས་ཥ་ཀ། ཀུན་ནས་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ༑ཨ་ཡཱཀྵིཏ། བད་བྷཱཛ་སོགས་ཀྱིས་འཕེལ་བ། ཨ་ཡཱཀྵ། །ཨ་ཡཱཀྵ། །སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཡཱཀྐི། ཨ་ཡཱཏྟཾ། ཨ་ཡཱ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཡཱཀཾ། །ཨ་ཡཱཀྵ། ཨ་ཡཱ། ཏང་གི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་ཡ། ཨ་ཡཀྵ་ཧཱུྃ། །ཨ་ཡཀྵ་ཏ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨ་ཡཥྛ། ཨ་ཡཀྵ་ཐཱ༑་ཨ་ཡནྟིཾ། ཛྷལ་ལས་ཞེས་པས་སིཙ་ཕྱིས་མཐའི་ཛྷལ་རྣམས་ཀྱི་ཛ་ཤའོས་ཥ་ཌ་ཥ་ཊུ་ལ་ཥ་ཏུ་འོས་དྷ་ན། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། ཨ་ཡཀྵ། ཨ་ཡ་ཀླུ་ཧི། ཨ་ཡཀྵ་ཧི། ལི་ཁྱུ་ངྷ་ས་སྨྱོ་བྷ་ཡེ་ཊཱ། ལིཊ་ལ་གཉིས་རྣམས་ཀྱི་ཨ་ངྷ་སའིའོ། །ལིཊ་ཕ་རོལ་ལ་གཉིས་རྣམས་ཏེ་བ་ཙ་ལ་སོགས་པ་རྣམས་དང༌། གྲ་ཧ་ལ་སོགས་པ་རྣམས་ཀྱིས་ཨ་བྷཱ་སའི་སཾ་པྲ་སཱ་ར་ཎར་འགྱུར་ཏེ། ཡ་ཛ་བ་ཙ་སོགས་ཀྱི་ནང་ཚན་ཡིན་པས། ལིཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། ཨི་ཡཱ་ཛ། བ་ཙི་སཱ་པི་ཡ་ཛཱ་དཱི་ནཱི་ཀི་ཏི། བ་ཙི་སཱ་པི་ཡཛ་ལ་སོགས་པ་རྣམས་ཀྱི་ཀིཏ་ལའོ། །བ་ཙ་འཆད་པ། ཉི་ཥ་པ་ཉལ་བ་གཉིས་དང་ཡཛ་ལ་སོགས་པ་རྣམས་ཀྱི་ཀ་འགྲོ་བའི་མིང་ཅན་ཕ་་་་རོལ་ལ་ཨ་བྷཱུ་ས་ཡོད་ན་གཉིས་ཀ་དང་ཨ་བྷྱཱ་ས་མེད་ཀྱང་ཡ་ཎའི་ཡི་གེ་རྣམས་སཾ་པྲ་སཱ་ར་ཎར་འགྱུར་ཏེ། ལིཊའི་ཨ་ཏང་གི་སྐྱེས་བུ་དང་པོའི་གཉིས་ཚིག་དང་མང་་་་ཚིག་ལ། ཨཱི་ཛ་ཏུ། ཨཱི་ཛུ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། ཨི་ཡ་ཛི་ཐ། ཨི་ཡཥྛ། བྲ༔་བྦསྔ་སོགས་ཀྱིས་ཛ་ཁ་ཁ་ཆུ་ལ་ཥ་ཏུ་འོས་ཐ་ཐ། ཨཱི་ཛ་ཐུ། ཨི་ཛ། སྐྱེས་བུ་ཕྱི་།
