ནི་ལོ་ངོ་ལྔ་ཁྲི་ཚོ་ག་ཙམ་པོ་བ་ལ་བློ་དང་ཨཱ་ག་མ་བྱིན། །ཙུའི་ཀུ་འོས་ཛ་ག། ཁར་ལ་ཙར་འོས། ག་ཀ་། ཀུན་ནས་་བསྒྱུར་བ་སོགས་ཀྱིས་ས་ཥར་བསྒྱུར་བས། མཾ་ཀྵ་ཏི། སོགས་དང༌། ཟླ་དེ་བཞིན་དུ། སྤྲིང་ལ། ཨ་མཾ་གྱུཏ། །སོགས་སོ། པ་ལ། ཀ་ཀ། ཀ་ཚུ་ན་ཁ་ཏོ༈བིཙྪ་འགྲོ་བ་ལའོ་ཞེས་པ་ལའང་ཨ་ཏང་རྣམས་སྦྱོར་ཏེ། །འགྲེལ་པར་ལུང་དྲངས་པ། །བིཙྪེ་རཱ་ཡོ་བཱ། བེཙྪི་ལས་ཨཱ་ཡ་ཡང་ནའོ། །།གོང་དུ་ཀུ་པཱུ་དཔ་བིཙྪ། པ་ཎཔ་ན་རྣམས་ལས་ཨཱ་ཡའོ་ཞེས་བཤད་ཀྱང་། བིཙྪ་ལས་ཨཱ་ཡ་་ཡང་ན་བར་བྱ་སྟེ། །ལཊ་ལ། བིཙྪཱ་ཡ་ཏི། སོགས་སམ། བིཙྪ་ཏི། སོགས་སོ། །།ལིང་ལ། བིཙྪཱ་ཡེཏ། ་སོགས་སམ། །བིཙྩི་ཏ། །སོགས་སོ། །ལོཊ་ལ། བིཙྪ་ཡ་ཏུ། སོགས་སམ། །བིཙྪ་ཏུ། སོགས་སོ། །ལང་ལ། ཨ་བི་ཡཏ། །སོགས་སམ། །ཨ་བིཙྪ་ཏ། སོགས་སོ། །ལུང་ལ་ཨིཊ་བྱིན་་་་་ཨཏ་ཀྱི་དབྱི་འོས་ཨ་ཕྱིས་པས། ་ཨ་བིཙྪཱ་ཡཱཏ། ཨ་བིཙྪཱ་ཡིདྡྷཾ། ཨ་བིཙྪཱ་ཡི་ཤུ། སོགས་སམ། ་ཨ་བིཙྪིཏ། ཨ་བིཙྪིཏྟཾ། ཨ་བིཙྪི་ཥཱ། སོགས་སོ། །ལིཊ་ལ། ཨཱ་ཡ་སྦྱར་བའི་ཕྱོགས་ལ། °ཀཱས་རྐྱེན་མཐའ་སོགས་ཀྱིས་ཨཱམ་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། །བིཙྪཱ་ཡཱ་ཙ་ཀཱ་ར། བིཙྪཱ་ཡཾ་བ་བྷུ་བ། བིཙྪཱ་ཡཱ་མཱ་ས། །སོགས་སོ༑། ཨཱ་ཡ་མེད་པའི་ཕྱོགས་ལ་གཉིས་བརྗོད་པས་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བིཙྪ། བི་བིཙྪ་ཏུ། །བི་བིཙྪ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བིཙྪི་ཐ། བི་བིཙྪ་ཐུ། །བི་བིཙྪ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བིཙྪ། །བི་བིཙྪིབ༑་བི་བིཙྪི་མ། ལུཊ་ལ་ཨིཊ་ཡང་ན་བར་བྱས་པས། །བིཙྪཱ་ཡི་ཏཱ། བིཙྪཱ་ཡི་ཏཱ་སི། སོགས་སམ། །བིཙྪི་ཏཱ། བིཙྪི་ཏཱ་སི། །སོགས་སོ། །སྨོན་པའི་ལིང་ལ། བིཙྪཱཡཱཏ། བིཙྪཱཡཱ་ཧཱུཾ། བིཙྪཱཡཱ་སུ། །སོགས་སམ། བིཙྪཱཏ། བིཙྪཱཙཾ། བིཙྪ་སུ། །སོགས་སོ། །ལྀཊ་ལ་ཨིཊ་བྱིན་པས། བིཙྪཱ་ཡི་ཥྛ་ཏི། —།
