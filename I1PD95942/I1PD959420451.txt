ཞིང་། །ནང་གི་ཤེས་བྱའི་ངོ་བོ་ནི། །ཕྱི་རོལ་ལྟར་སྣང་གང་ཡིན་དེརཚེ། །དོན་ཡིན་རྣམ་ཤེས་ངོ་བོའི་ཕྱིར། །དེའི་རྐྱེན་ཉིད་ཀྱང་ཡིན་ཕྱིར་རོ། །གཅིག་ཆའང་མི་འཁྲུལ་ཕྱིར་ན་རྐྱེན། །ནུས་པ་འཇོག་ཕྱིར་རིམ་གྱིས་ཡིན། །ལྷན་ཅིག་བྱེད་དབང་ནུས་པ་ཡིས། །ངོ་བོ་གང་ཡིན་དབང་པོའང་ཡིན། །དེའང་རྣམ་་རིག་ལ་མི་འགལ། །དེ་ལྟར་ཡུལ་གྱི་ངོ་བོ་དང་། །ནུས་པ་ཕན་ཚུན་རྒྱུ་ཅན་་་དང༌། །:ཐོགས་མེད་དུས་ནས་འཇུག་པ་ཡིན། །དམིགས་པ་:བརྟགས་པ་སློབ་དཔོན་ཕྱོགས་ཀྱི་གླང་པོས་མཛད་པ་རྫོགས་སོ། །།།བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་དམིགས་པ་་་བརྟག་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་»ཨ་ལོ་པ་ཎ་པ་རིཀྐ། «པེ་»«སྣར་»པ། །དེ་མེད། «པེ་»«སྣར་གསལ་བ། ནད་དེ་«པེ་»«སྣར་པར་འགྱུར། «པེ་»«སྣར་དེ། «སྣར་»+པའི། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»ཡིན། «པེ་»«སྣར་ཐོག་མ་མེད་དུས། ཨ«པེ་»«སྣར་»བརྟག་པའི་ཚིག་ལེའུར་བྱས་པ། ན་་་་་བསྡུར་འབྲས་རེའུ་མིགཚད་མ། （དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། །ཚད་མ། །ཚད་མ་ཀུནུས། །མདོ། ཚད་མ། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན?ན་ནབ་ན་—བ?རབ*ན། ཕྲེང༌། །ཚིག་རྐང་། །ཚེག་ཁྱིམ། །ལོ་—།
