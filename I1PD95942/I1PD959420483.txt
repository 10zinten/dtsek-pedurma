བ་བརྒྱ་གར་ལས་རྒྱའི་ལོ་ཙཱ་བ་ཐང་སཾ་ཙང་གིས་བསྒྱུར་ཅིང་སླད་ཀྱིས་རྒྱའི་དགེ་བཤེས་སིང་གྱང་ཇུས་དང་། བོད་ཀྱི་དགེ་བའི་བཤེས་གཉེན་སྟོན་གཞོན་གྱིས་བོད་སྐད་དུ་བསྒྱུར་ཞིང་། ཕྱིས་གཞི་ཐམས་ཅད་ཡོད་པར་སྨྲ་བའི་སྡེ་པ་ལས་རབ་ཏུ་བྱུང་བ་རྒྱ་ནག་ཆེན་པོའི་ལྷའི་བཙུན་པ་ཆོས་ཀྱི་རིན་ཆེན་གྱིས། དཔལ་ལྡན་ས་སྐྱའི་གཙུག་ལག་ཁང་དུ་རྒྱའི་དཔེ་དང་བོད་ཀྱི་དཔེ་གཉིས་པོ་ལེགས་པར་གཏུགས་ཤིང་བཅོས་ཏེ་དག་པར་བསྒྱུར་བའོ། །།།།རྒྱའི་དཔེ་ལ་རིགས་པ་ལ་འཇུག་པ་ཞེས་སྣང་། དེང་སང་བོད་རྣམས་་རིགས་པའི་སྒོ་ཞེས་གྲགས་སོ། །།།བསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་རིག་པ་ལ་་་་འཇུག་པ་བཞུགས་སོ། །ཞེས་འཁོད། @«པེ་»«སྣར་»གཡེན། @«པེ་»«སྣར་རིག«པེ་«སྣར་»དཔལ། ཚུ«པེ་»སྒྲུབ། །ཚེ«པེ་»«སྣར་གཞི། «པེ་»«སྣར་སྤྱིའི། «པེ་»«སྣར་»—པ། «པེ་»«སྣར་པའི། «པེ་སྒྲུབ། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»གི། «པེ་»«སྣར་»པའི། «པེ་»«སྣར་»གི། @«པེ་»«སྣར་གྱི། «པེ་»«སྣར་པའི། ཚེ«པེ་»«སྣར་»—སྒྲ། «པེ་»«སྣར་»སྒྲུབ། ལོ«པེ་»«སྣར་»པའི། ཀྱེ«པེ་»«སྣར་»གི། གེ«སྣར་»ཅན། ཉེ«པེ་»«སྣར་»ལ། «པེ་»«སྣར་»བསལ། ཚེ«པེ་»«སྣར་»སྒྲུབ། «པེ་»ཏེ། «པེ་»«སྣར་མིག༄༅«པེ་»«སྣར་བསལ། «པེ་»«སྣར་བརྟེན། ཉི«པེ་»«སྣར་»འཇུག་ལ། «པེ་»སྣར་»—ལ། «པེ་»བྱར། W«པེ་»«སྣར་བཙལ། (ཆོས་ཚན་འདིའི་ནང་དུ་བརྩལ་ཞེས་པ་རྣམས་«པེ་»«སྣར་»ལ་་་བཙལ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་»སྐྱ། «པེ་»«སྣར་»+དང་། W«པེ་»«སྣར་»སྒྲུབ། །«པེ་+རེ། :«པེ་»«སྣར་»དེ། «སྣར་»+པ། «པེ་»«སྣར་པ། ཨེ«པེ་»«སྣར་གྱིས། ®«པེ་»«སྣར་རོ། »«པེ་»«སྣར་—པ། «པེ་»«སྣར་པ། «པེ་»།
