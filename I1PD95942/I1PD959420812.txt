དོར་ནས་ངས་བཤད་ངེས་པར་ནི། །རྣམ་དག་ཐུགས་དེའི་དགོངས་པའང་འདི་་ཉིད་དེ། །མི་གསལ་བློ་ཅན་སྐྱེ་བོས་རྟོགས་མ་ཡིན། །གཞན་གྱི་དོན་གྱི་རྗེས་་སུ་དཔག་པ་སྟེ། ལེའུ་གསུམ་པའོ། །།།ཚད་མ་རྣམ་པར་ངེས་པ་ཞེས་བྱ་བ། ཡུལ་ལྷོ་ཕྱོགས་ཀྱི་རྒྱུད་དུ་བྱུང་བ། གཞུང་ལུགས་ངན་པ་ཐམས་ཅད་ཆེས་ཤིན་ཏུ་འཁྲུལ་པར་འཇུག་པ། སྙན་པའི་གྲགས་པས་སའི་སྟེང་མ་ལུས་པར་ཁྱབ་པ། །འགྲན་ཟླ་མེད་པའི་མཁས་པ་ཆེན་པོ། དཔལ་ཆོས་ཀྱི་གྲགས་པས་མཛད་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་ཁ་ཆེའི་པཎྜི་ཏ་གཞན་ལ་ཕན་པ་བཟང་པོ་ལ་སོགས་པ་དང་། ལོ་ཙཱ་བ་བློ་ལྡན་ཤེས་རབ་ཀྱིས། ཁ་ཆེའི་གྲོང་ཁྱེར་དཔེ་མེད་དུ་བསྒྱུར་བའོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»«པེ་»«སྣར་»ལྟར་བཀོད། @«ཅོ་»བཟུང་། «པེ་»«སྣར་བསྟན། «པེ་»«སྣར་»རྒྱས། «པེ་»«སྣར་»སྔོན། «སྣར་»+པ། ༄༅«པེ་»སྣར་བྱའོ། «པེ་《སྣར་»དཔྱད། །@«སྣར་»ནི། «པེ་»«སྣར་»རོ། ཨི«པེ་»+མ། «སྣར་»+ཡང་མ། «པེ་»«སྣར་»—ཏུ། །«པེ་»«སྣར་»ལ། «པེ་»«སྣར་»དོན། «པེ་»«སྣར་»—ཏེ། «པེ་《སྣར་»དེ། «པེ་»«སྣར་»གཞན། «པེ་»«སྣར་»པའི། «པེ་»«སྣར་»གིས་སྒྲུབ་པ། ཉི«པེ་»«སྣར་»དང་། «པེ་»«སྣར་»དང༌། མ«པེ་»«སྣར་བསླུ། @«པེ་»«སྣར་»ནོ། ཉི«པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་བསལ། «པེ་»«སྣར་»མ། @«པེ་»«སྣར་»སྤོང་། ཚེ«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»ཆོད། @«པེ་»«སྣར་»+མི། །«པེ་»«སྣར་བསླུ། «པེ་»«སྣར་»པའི། «པེ་»«སྣར་+ཐམས་ཅད། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»འཁྲུལ། «ཅོ་གྱིས། «པེ་»«སྣར་པོ། «པེ་»«སྣར་»ཡང་། ®«པེ་»«སྣར་»རོ། «པེ་»།
