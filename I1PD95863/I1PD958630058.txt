ཝུ་ཁང་ཐལ་པའི་མཐར་ནི་རྣམ་པར་བརྟག། རིམས་ནད་མེད་ཅིང་ནད་ཀྱང་མེད། །དུག་ལ་སོགས་པས་མི་འཆི་ལ། །དེ་ནི་དུས་མིན་འཆི་མི་འགྱུར། །འཚེ་བ་ཐམས་ཅད་རྣམ་འཇོམས་པའོ། །རྡོ་རྗེ་ཕྲེང་བ་དཔལ་མཆོག་སྲུབ་ཤིང་གིས། །གསང་འདུས་སྦས་ཆུའི་རྒྱ་མཚོ་བསྲུབས་པ་ལས། །སྒྲུབ་པའི་ཡན་ལག་བདུད་་རྩི་གང་ཐོབ་དེས། །འཇིག་རྟེན་མ་ལུས་འཆི་མེད་མྱུར་ཐོབ་ཤོག། སྒྲུབ་པའི་ཐབས་མདོར་བྱས་པ་སློབ་དཔོན་:ཆེན་པོ་ྋཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་ཨཱ་:ཙཱརྱ་ཤྲདྡྷཱ་ཀ་ར་ཝརྨ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །།།།'::»སྟེ་དགེཎན་དཀ»མེ་བ། བཀད་བརྒྱུད་ཀྱི་ལུང༌། :/ཀྱི་ལོ་ང་ཡི་། མ་བྱེ་བ་བ་ཁས་། ཁ། དཀར་འོད་ཁག། ཁ་ར《་ཀྱི»དྲི། ཁརྒྱུ་ཀ་བ་ཀྱི་ཚོགས་ནས་དགེ་:::་་༼ཨཀྵ་བསྡུར་མཆན། ཁར་། ད༌། ་[། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»པིཎྜ་དྷི་ཀྲི་ཏ་སཱ་དྷ་ན། «སྣར་»དིཎྜ་བྷི་ཀྲི་ཏ་སཱ་དྷ་ན། །«སྣར་རྟོགས། «པེ་»«སྣར་»དོན། «པེ་»«སྣར་སྐྱེད། །(«པེ་»«སྣར་མགོན་པོ། ་«པེ་»«སྣར་»དཀྱིལ། «པེ་»བིདྱ་ར། སྣར་བིཏྟཱ་ར། «པེ་»n《སྣར་»དུཏྟཾ། ཡང་«༼པའི་ནང་འབད་ཀྱི«པེ་»«སྣར་»ལཱ།  ི«པེ་»«སྣར་»ཨངྷ། «པེ་»《སྣར་»བཛྲ། ®«པེ་»«སྣར་པ། །།«པེ་»«སྣར་»བཅུ། @«སྣར་»ཀྱང་། «པེ་»«སྣར་»ཧཱུྃ། ལྔ་ཀྱི། །བདག་༽(«པེ་»«སྣར་+ཕྱི་ཡི་དཀྱིལ་འཁོར་ཐིག་གདབ་བྱ། །«པེ་»«སྣར་བྱ་ནི། «པེ་»«སྣར་བུ། «པེ་»«སྣར་»གྱི། སུ«པེ་»«སྣར་ལོ། །(འ«པེ་»«སྣར་»བསམ། «པེ་»«སྣར་གཏོགས། །ས་བར། །བ་སྤྱི། «པེ་»«སྣར་»མཁའ། (Q«པེ་»«སྣར་གྱུར། «པེ་»«སྣར་ཨཏྨ། ཌཚེ«པེ་»སྣར་ཡི། «པེ་»བསྐྱོད་པས། «སྣར་སྐྱོད་པས། @«སྣར་»ཨིནྡྲ། «པེ་»«སྣར་»འདི། 《ཉི«པེ་»«སྣར་ཨ། ཕྱི་ཞུ/«སྣར་»སྐྱོད།  ེ«པེ་»«སྣར་»ཤིང་། «པེ་»«སྣར་ཨོཾ། «པེ་»«སྣར་བསམ། «པེ་»«སྣར་མེ། @«པེ་«སྣར་»ཚིག༄༅«པེ་»«སྣར་»ལོ། ཀའི་འདྲ་བའི་བྱེད་པ། །
