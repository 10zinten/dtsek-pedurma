ཚེས་དང་བརྒྱུད། （བབEབ་འདུ་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། །བརསྡེ་ཚན། རྒྱུད། བརྒྱུད་འགྲེལ། བརྒྱུད། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །དུ་རིམ་པར་ཕྱེ་བ། །སྒྲ་པལྔ། ལྡེབ། ནན་བན་ནབ་ནvན་ཕྲེང་། །，་ཚེག་ཁྱིམ། لولولولمللدولད་ཀོབ་འགཌ་ཏུ་ཟླ་བ་ལགས་ཀ་དང་། །ཌ་ན་ན་ཟླ་ཊདེ་བཞིཤོ་ཀའི་ང་། :གས་ཏེ་རྡོ་རྗེ། ངག་ད་བ་ས་ཀ་ཏོਪਰਕਲਬipਇੰਝਸਪਲgਓਲਓਪਨཤུལ་་ཁགེ་སུའི་ཁྱབ་ཁང་རྒྱལ་བཀོད་པ། པད་འདབ་བདེ་ཆེ་བ་ག་ཚོབློག་ཀླདང་ཀ་ཀྱོང་ཏ་ཁ་་གང་ཚོང་ར། »ཁ་ཊིཀ་ཆུ་དང་འཚོ་བ་དེ་ཡང་། ཀྲཀུནཁོ་ལ་/Aག་ཚོདཁཚུལ་དང་རྗེ་ཎི། །བསང་དུ་ཡོད་ཀྱང་དུ་བའི་ཀ་ཁ་སུ་ག་བ་ལ་ཁའི་ཞེན་པ་བཟླ་བ་ཟ་ཞ་་་ཎི་ཡ་ཤོ་ལའི་འབྱུང་ལྔ་ནི། ལ་པ། ཤ་ཥ་ཌཁ་ཐཝེ། ཀུའི་ངག་གྱི་ཋབྷོ། །ཁ་བ་དང་། ཚེ་ཏན་ཏན་ཀ་ཁཎྜི་ཏཁ་ལེན་པར་བསད་བཏོན་པ་ལ་ས་སྐྱ་བ། བཞིན། །ག་ཚོ་། །།་ཁྱཚོ། །བ། །ཌལྔཆེན་གཏན་ལ་དབབ། །ཀུན་ནས་ལུགས་།
