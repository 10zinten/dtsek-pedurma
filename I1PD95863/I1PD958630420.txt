།སྣོད་ཀྱི་སྙིང་མགྲིན་མགོ་བོ་ལ། །ཧཱུྃ་ཨཱ༔ཧཱཞེས་བྱ་བ་དགོད། །དེ་བཞིན་་་་་གཤེགས་པ་དབབ་དོན་དུ། །འབེབས་བྱེད་ཕྱག་རྒྱ་བཅས་ནས་ནི། །སྣོད་ཀྱི་སྙིང་གར་གཞག་བྱས་ཏེ། །གསང་སྔགས་འདི་ནི་བརྗོད་པར་བྱ། །།འདི་ཁྱོད་དམ་ཚིག་རྡོ་རྗེ་སྟེ། །རྡོ་རྗེ་སེམས་དཔའ་ཞེས་བྱ་བ། །འབབ་མཛད་དེང་དུ་་་ཁྱོད་ལ་ནི། ཞེ། རྡོ་རྗེའི་ཡེ་ཤེས་བླ་མེད་འབྱུང་། །ཨོ་བཛྲ་ཨཱ་ཝེ་ཤ་ཡཨཨཨེ། དེ་ལྟའང་གལ་ཏེ་མི་འབབ་ན། །།།གསང་སྔགས་བརྟན་པོ་འདི་བརྗོད་ཅིང་། །དེ་ནས་ཡི་གེ་ཨ་ཟེར་གྱིས། །ཀུ། ཡེ་ཤེས་ཆེན་པོ་བསྐུལ་བར་བྱ། །ཨཱ་ཝེ་ཤ་ཡ་སྟོ་བབྷ་ཡ་ར་ར་ར་ར་ཙཱ་ལ་ཡ་:ཧཱུྃ་ཧ་ཨ་དྷོ། །སྙིང་འདར་དགའ་བ་ཆེ་བ་དང༌། །་ཆ། བརྒྱལ་བ་དང་ནི་གར་བྱེད་དང༌། ་། འགྱེལ་དང་གྱེན་དུ་འཕར་བ་ནི། །བབས་པའི་མཚན་ཉིད་དུ་ཤེས་བྱ། །སྣོད་ནི་རྒྱས་པར་མཐོང་ནས་སུ། །ཨ་ཡིག་ལྕེ་ལ་དགོད་པར་བྱ། །ཡི་གེ་བརྒྱ་པས་བརྟན་བྱས་ཏེ། །སྔགས་ཀྱིས་མེ་ཏོག་ཕྲེང་བ་དོར། :། ཏིཥྛ་བཛྲ་དྲི་རྣོ་མེ་བྷ་:ཝ་ཧྲི་ད་ཡ་མེ་ཨ་དྷི་ཏིཥྛ་སརྦ་སིདྡྷི་མེ་པྲ་ཡཙྪ་ཧཱུཾ་ཧ་ཧ་ཧ་ཧ་ཧོ། །།ཨོ་:ཨ་ཁཾ་ལྕི་ར་ཧཱུྃ་པྲ་ཏཱིཙྪ་ྋཀུ་སུ་མ་ཨཉྫ་ལི་:ནཱ་ཐ་ཧོ༄༅། །རིགས་ཀྱི་དབྱེ་བ་མཁས་པ་ཡིས། །མེ་ཏོག་བབས་པ་ཤེས་གྱུར་ནས། །སྔགས་དང་བཅས་པར་བྱས་ནས་མགོར། །ཕྲེང་བ་དགོད་པ་སྦྱིན་པར་བྱ། །ཨོཾ་པྲ་ཏི་གྲིཉྩ་ནྡྷེཙཱ་ཨི་མཾ་སཏྭ་མ་ཧཱ་བ་ལ། ཨོཾ་གྱིས་མནན་པའི་མིག་ལ་ནི། །སློབ་མ་སྡིག་དང་བྲལ་བ་ལ། །བརྟུལ་ཞུགས་བཟང་པོས་སྔགས་འདི་ཡིས། མ་། གདོང་གཡོགས་རབ་ཏུ་དགྲོལ་བར་བྱ། །རྡོ་རྗེ་སེམས་དཔའ་རང་ཉིད་དེང་། ::། ཁྱོད་ཀྱི་མིག་ནི་འབྱེད་པར་བརྩོན། །རྡོ་རྗེའི་ྋམིག་ནི་བླ་མེད་པ། །ཐམས་ཅད་སྤྱན་གྱིས་འབྱེད་པར་མཛད། །དེ་རིང་དད་པའི་ཤུགས་ཀྱིས་ནི། །དཀྱིལ་འཁོར་་་་འདི་ལ་བལྟ་བར་གྱིས། །ཁྱོད་ནི་སངས་རྒྱས་རིགས་སུ་སྐྱེས། །གསང་སྔགས་་་—།
