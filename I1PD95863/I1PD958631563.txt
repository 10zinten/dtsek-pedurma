བསྟན་འགྱུར། རྒྱུད། ཅི«པེ་»«སྣར་གྱི། «པེ་»«སྣར་»ལ། @«པེ་»«སྣར་»པ། ཞེ«པེ་»«སྣར་»ཨའོ། «ཅོ་ཨཱའོ། «པེ་»«སྣར་»པར། ཚེ«པེ་»—རྐེད་པ་དང་། ཚེ«པེ་སླའོ། «སྣར་»སཱའོ། ཉི«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་བཞག«པེ་བཞུགས་པར་བྱའོ། «སྣར་བཞུགས་པར་བྱ་བའོ། །«པེ་»«སྣར་»རྟགས༑«པེ་»«སྣར་»བའི། @«པེ་»«སྣར་—ནི། «པེ་»«སྣར་»བ། ༄༅«པེ་»«སྣར་»བལྟོས། ཉི«པེ་»«སྣར་བྱིན། །པ་་་་་་«པེ་»«སྣར་ལིང་། «པེ་»«སྣར་»—པ། །་ཏེད་པ་དུར་གྱི་དབུས་སུ་བག་ཁྱད་པ་པའི་པ་དང་དགག་པ་སོགས་ཁབ་ལ་བྱའོ། །དེ་བྱ་བ་ལྟ་བུ་ནི་ཚེ་ལ་དདསཊིn'ཏཆཏཚོ་ཌི་ཚེe/S། ས་བསྡུར་འབྲས་རེའུ་མིག་ན་བྱེད་པ་དང་རྒྱུད། （ལོའི་ཟླ་བ་ང་}(ཙཱ་སུ་ཞུ་དང་བྲལ་བ་ཉིད:པེ་ཅིན། སྣར་ཐང་། རྒྱུད། དཔེ་གཞི། །སྡེ་དགེ། སྡེ་ཚན། །རྒྱུད་འགྲེལ། པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། བརབ་བན་བན། བ?བ། ཕྲེང་། །།རབཞི་ཚེག་ཁྱིམ། /ན。་། །་}། »y། །ཁ《་། ཀྱི། 《་། །པར་ཀ་《་དོན་}འདྲ་བ་ལས། མ་ལ་དང་། ད་ཡུ་ད་དང་། ལ་༼(?། །ཋ་ཕྱི་ལོ་ཙོའི་དབང་་་་རྒྱདགའ། །བོད་མང༌། ནང་བ་དང་བདེ་ལ། །d། །ད་དུང། མ་ppt?། །་ལ་སོགས་དང་། དར་ལ་བྱ་བ་རྒྱུད་ཀྱི། ཁལ་བས། ་་དཀྱི་།
