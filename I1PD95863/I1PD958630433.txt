ན་མེད་པ་གང་བསགས་པ། ་་་་་། དེས་ནི་སྐྱེ་ཚོགས་རྡོ་རྗེ་འཛིན་པ་མཆོག་ཉིད་དུ་་གྱུར་ཅིང༌། །དེ་ཉིད་ཀྱིས་ན་བདག་གི་གྲགས་དང་ལྷན་ཅིག་ཀླུ་ཡི་ནི། །བྱང་་་ཆུབ་ཀྱིས་བྱས་རིན་ཆེན་རྡོ་རྗེ་མཆོག་འདི་རིང་གནས་ཤོག། དཔལ་གསང་བ་་་་འདུས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་ཉི་ཤུ་པ་ཞེས་བྱ་བ་སློབ་དཔོན་ཀླུའི་བྱང་ཆུབ་་་་ཀྱིས་མཛད་པ་རྫོགས་སོ། །ཁ། ལན་དུ་ང་ལ་བལུལ་ཏཀམ་ཎ་བཀའི་བ་བྱ་སའི་གཙུག་ལག་ཁང་དུ་ཤུད་ཀྱི་སྟོན་པ་དར་མ་རྡོ་རྗེས་ཞུས་ནས་རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་ཆེན་པོ་ཏི་ལ་ཀ་ཀ་ལ་ཤ་ཞེས་བྱ་བ་ཕ་རོལ་ཏུ་ཕྱིན་པ་གསང་་་་སྔགས་ཀྱི་ཐེག་པ་ལ་སོགས་པ་མཐའ་དག་ཆུབ་པའི་ཞལ་སྔ་ནས་དང་། ཚེ་བོད་ཀྱི་སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་ཆེན་པོ་བནྡེ་པ་ཚབ་ཉི་མ་གྲགས་ཀྱིས་བསྒྱུར་བའོ། །།།ཏཔ། །དཞི། ན་བྱེད་ཀྱི་ཡོད་ཀ། །།ཁང་དང་ཟབ་ལ་བདའཁ་ལ་སློབཙཎྜ་ཀ། ཁཁྱོ་ཀ::ཀ་ཁ་ཁ་བྱེད་ས་དག་ཞུ་བའི་ངོས་གང་ཀ་ཁ་ན། བསྡུ་བའི་ཁོངས་ཀྱི་ཁཏུ་ཁྱར་། །༽ཏོ། །ཡན་ལ་༼ཀ་ཏུ:༌བསྡུར་མཆན། །འཁྲུལ་ངོ་ཁ་nཚོ་ཁར་བཏུས། པརང་ཚོ། ༈། དང་། གང་བ། །།མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»དཔལ་གསང་བ་འདུས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་ཉི་ཤུ་པ་ཞེས་བྱ་བ་སློབ་དཔོན་ཀླུའི་བྱང་ཆུབ་ཀྱིས་མཛད་པའོ། །།།:«པེ་»«སྣར་མ། «པེ་《སྣར་»ཀི། ༄༅«པེ་»«སྣར་ཏ།  ། གྲོ་ག་སྤྱི་བ་ཀ་ཁ་ངན་«པེ་《སྣར་དྲངས། «པེ་»«སྣར་ཆེན། «པེ་»«སྣར་བརྟགས། བ«པེ་»རིས། «སྣར་རིགས། «པེ་»«སྣར་»ཀུན། །«པེ་»«སྣར་སྦྱོང་། «པེ་»«སྣར་བཟང་བྱུགས། །་ལུ་དུ་བདག་ཏུ། །བདེw«པེ་»«སྣར་ནོམ། …«པེ་»«སྣར་རྩེ། འ«སྣར་གསལ། «པེ་»«སྣར་རཀྵ། སུ«པེ་»«སྣར་ཨོ། «པེ་»«སྣརཧལླ། «པེ་»«སྣར་+ཚ། «པེ་»«སྣར་ས། «པེ་»《སྣར་པོས། བོ«པེ་»«སྣར་དེང་པ་འདིའི། 《སྣར་འགགས། འབུ«པེ་»«སྣར་+འགྱུར། །ཀྱི«པེ་»«སྣར་གླལ། ནུ«པེ་»«སྣར་བསྐོར། །«པེ་»«སྣར་འཇིག་་«སྣརབརླབས། སླར་སུ«སྣར་འདྲེམ། «པེ་»«སྣར་»—འདུག«པེ་»«སྣར་»ཀྱི། །«པེ་《སྣར་བཞག«པེ་»«སྣར་»འཐོན། ༈«པེ་»«སྣར་»བན་བིརྱ། བ«སྣར་»འདི། ག་ག་ན་།
