ཕྱག་རྒྱ་:སྒྲུབ་པར་བཤད་པ་ཡིན། །མྱུར་དང་རྣམ་བསྡུས་དོལ་བ་ཅན། །སྔགས་སུ་ངེས་པར་སྦྱོར་བ་ཡི། །སློབ་དཔོན་གྱིས་ནི་འབད་པ་ཡིས། །སློབ་་་མའི་རྣ་བའི་དྲུང་དུ་བརྗོད། །དཔལ་གསང་བ་འདུས་པ་ཞེས་བྱ་བའི་རྣལ་འབྱོར་ཆེན་པོའི་རྒྱུད་ལས་དོན་དམ་པའི་དཀྱིལ་འཁོར་གྱི་རྣམ་པར་གཞག་་པའི་རིམ་པར་ཕྱེ་བ་སྟེ་བཞི་པའོ། །སྟེ། །།མི་ཅིག་ར་ཤིང༌། །བའི་ཡན་གྲུབ་པའི་སློབ་དཔོན་ཀླུའི་བློའི་ཞལ་སྔ་ནས་མཛད་པ་རྣམ་པར་གཞག་པའི་རིམ་པ་བཞི་པ་རྫོགས་སོ། །ལ། །《་ལྟ་ཀ་ཀམའད་་་རཀྱིཧེཀrtག/@paz་རg/pཀ་པ་སུ་བྷyg་། ཀ་བ་དང《ཙཅུ་་ར་་གཀཁོ《ང་་ཀཾཀྱི། བོrང་ོn་བདུར་མཆན། ལ《ཀྱི་ཚོགས་དང་ཞལོང《འི་དཔཉྩ་ལྔ་ལས་བར་དོའི་ལྟ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། »«པེ་»«སྣར་»འདུས་པའི་སྒྲུབ་བད»ཐབས་རྣམ་པར་བཞག་པའི་རིམ་པ་ཞེས་བྱ་བ་གྲུབ་པའི་སློབ་དཔོན་ཀླུའི་བློའི་ཞལ་སྔ་ནས་མཛད་པའོ། །《སྣར་»ལ༑ ི«པེ་»«སྣར་»བཞག«པེ་»«སྣར་ཧཱུཾ། ཊཀྐཎཔ་ཚོ། ང་དང་(«པེ་»«སྣར་»སྐྱེད། «པེ་»སྣར་གྱིས། 《སྣར་»བ། «པེ་བཞག་«པེ་»《སྣར་»ཀྱི༑@«པེ་»«སྣར་»ཕང་། ཚེ«པེ་»«སྣར་»ཏོ། །(«པེ་»«སྣར་»གི། ་«པེ་སྒྲུབ་ཅིང་སྐྱེད། 《སྣར་»སྒྲུབ་ཅིང་སྐྱེས། :འཁོ«པེ་»«སྣར་པ། Ppz་དང་། ལ། ཎདང《། «པེ་»«སྣར་»སྒོམ། «པེ་»«སྣར་»ལྕགས། «པེ་»«སྣར་མངོན། ®«པེ་»«སྣར་སྐྱེད། ཁe/Q«པེ་»«སྣར་མངོན། སྤྱི«པེ་»བརྫུས། ས་འབར། །ཁ་ཏུབ་ཀ་་ཀྱི་«པེ་»«སྣར་བརྫུས། «པེ་《སྣར་མ། འ«པེ་«སྣར་»—དུ། ︽«པེ་»«སྣརདཔྱོད། :། «པེ་»«སྣར་+བ། «པེ་»«སྣར་»ཀྱི། ༄༅«པེ་»《སྣར་འཁྲུང་ཞིང་། །་་«པེ་»«སྣར་པོའི། «པེ་»«སྣར་»གི། ཚེསྣར་»བོའི། །ོ«པེ་»«སྣར་»དོ་ན། །།@«ཅོ་དེ། ༼Q«ཅོ་སྲིད། «པེ་»«སྣར་—ཚིག་ལ་ཀྱི། ད་ལ། ་ཁ་སྒྱུ་«པེ་»«སྣར་»གཉི་གའང་མཆོག་ཏུ་དགའ་བའི་བདེ་བས་ཚིམ་པར་བྱའོ། །«པེ་»«སྣར་དང༌། །
