བཙུན་མོ་ལ་འཁྱུད་པའི་བདག་ཉིད་དུས་རྟག་ཏུ་འཛིན་པའོ། །དེ་ཡང་། །ལྷུན་གྲུབ་བདེ་བ་ཆེན་པོའི་སྐུ། །རང་སྣང་རིག་མས་ཉེར་བརྒྱན་པ། །འོག་་མིན་གནས་སུ་དུས་རྟག་ཏུ། །མཚན་དང་དཔེ་བྱད་བརྒྱན་པས་བཞུགས། །ཞེས་སོ། །མ་ཙིནྟ་མ་ཎི་སྟེ་ཡིད་བཞིན་གྱི་ནོར་བུ་ལྟར་འབད་པ་མེད་པར་་་་་བདག་དང་གཞན་གྱི་དོན་སྒྲུབ་པར་མཛད་པའོ། །དེ་ཡང༌། ཡིད་བཞིན་ནོར་ལྟར་བསམས་ཙམ་གྱིས། །འབད་མེད་བདག་དང་གཞན་གྱི་དོན། །ཆེ་བའི་་་་ཡོན་ཏན་:རྣམ་བརྒྱན་་སྐུ། །བདེ་བ་ཆེ་ལས་འབྱུང་བར་འགྱུར། །ཞེས་སོ། །དེ་ལྟར་ཚུལ་བཞིའི་བཤད་པའོ། །དཔལ་གསང་བ་འདུས་པའི་རྫོགས་རིམ་མཐར་ཕྱིན་པ་སློབ་དཔོན་ཨཱརྱ་དེ་བས་མཛད་པ་རྫོགས་སོ། །པ་ལུ་སྤྱོད་འབད་བའི་གནས་བབ་དང་ཉེ་བ་དང་ག་འགག་པ་ད་དུ་འདུས་པའི་སྒྲུབ་པ༌བསྡུར་མཆན། པ་ག་དེ་འབད་ར་བ་ཞེས་བྱ་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་»བཅོམ་ལྡན་འདས་གསུང་གི་དབང་ཕྱུག་འཇམ་དཔལ་ཡེ་ཤེས་སེམས་དཔའ་བཞུགས། །།།ཉེ«པེ་»«སྣར་—པར། «པེ་»«སྣར་»སཾ་དྷི་ཤྩ། ༽(«པེ་《སྣར་ཚེ། ་«པེ་»«སྣར་སྤྱོད། །«པེ་»«སྣར་ཨ། «པེ་»«སྣར་མའི། །(«པེ་»«སྣར་»སཾམྱཀ«པེ་»«སྣར་»ལ། «པེ་»«སྣར་། །ཏུ«པེ་»《སྣར་»ཤིག«པེ་»«སྣར་ཀྱང༌། །སེམས་གཟུང༌། «པེ་»«སྣར་»མཐུན། །་(«པེ་»«སྣར་»ཆག་«པེ་»«སྣར་—ལྟར་རམ་མུན་པ། «པེ་»«སྣར་དེའི། །༄༅«པེ་»«སྣར་»བྱ་བའོ། །«པེ་»«སྣར་ལ། «པེ་»«སྣར་བཤིག«པེ་»«སྣར་»དོན། «པེ་»«སྣར་»«ཅོ་»བདེན་པའི། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»བསྒྲུབས། ས«པེ་»«སྣར་ཀྱི། པོ«པེ་»«སྣར་པ། «པེ་»སྣར་ཐོག་མ། བོད་པ་ཚོས་«པེ་《སྣར་པོས། («པེ་《སྣར་དཔའ། །དང་ནང་དུ་«པེ་»«སྣར་—སྐུ། «པེ་»«སྣར་ལ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»སྟེ། ས«པེ་»«སྣར་»གི། «པེ་»«སྣར་པ། བར། །
