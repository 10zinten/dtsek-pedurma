བསྟན་འགྱུར། རྒྱུད། ཅིདེས་ནི་སེམས་ཅན་རང་རིག་པའི། །བདེ་བ་ཁྱེད་དང་འདྲ་བར་ཤོག། དད་པ་སྨྲ་བ་ཞེས་བྱ་བའི་བསྟོད་པ་རྫོགས་སོ། །རྗེའི་། །མ་རྒྱུད་ས་སྐྱ་ན་ཆོ་ཎྜ་ཀ་འཚོཝེ་རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་བཱི་གུཔ་དང་། སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་བནྡེ་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཊཱི་ཀོ། །ཀ་ཁ་ག་ང་དེ་ནི་ཁྱིམ་པ་ཐོབ། ཀུ་བའོཆེ་དང་། བྱ་པ་ཁོ་ཀ་མ་དང་། །ཀ་བ་ཁ་ཆེཎྜ་ཞེ་ཏུཏོཥ་ཎ། བསྡུར་མཆན། ཡ་ཎ་འད། །།།བཞི་མཆོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་དད་པ་སྨྲ་བ་ཞེས་བྱ་བའི་བསྟོད་པ་བཞུགས་སོ། །«པེ་»«སྣར་»ཤཱནྟ་པྲ་ལཱ་པ་སྟས། མི«པེ་»«སྣར་»ཀའི། ལ«པེ་»«སྣར་བསྟོད། །རS། :::ཀ་བྱ་བའི་བྱ་བ་(ི«པེ་»«སྣར་»བལྟ། «པེ་»«སྣར་བརྩོལ། ®«པེ་»«སྣར་»ཚོ། «པེ་»«སྣར་ཡིད། །«པེ་»སྐྱེད། «སྣར་»སྐྱེས། «པེ་»«སྣར་འགྱུར་བའི། ༄༅ི«པེ་»«སྣར་»ལ། «པེ་»《སྣར་ཡི། ཉི«པེ་»«སྣར་བ། «པེ་»«སྣར་རྩོལ་ཅིང་། «པེ་»«སྣར་འཕྲུལ། ཀ་དཔེ་»སྣར་པའི་མུན་བསལ། 《སྣར་གིས། «པེ་»«སྣར་རྙེས། ཉི«པེ་»《སྣར་»འཛིན། །«པེ་»«སྣར་»ཀྱི། «པེ་»《སྣར་མི་འགྲུབ་པའི། «པེ་»«སྣར་རུན། «སྣར་པ། ཚུ«པེ་»《སྣར་»རྣམས་པར་བྲལ། @«པེ་»«སྣར་»བགྱིས། «པེ་»《སྣར་»སྐྱོབ། «པེ་»《སྣར་རྟོག«པེ་»སྣར་བཞུགས། «པེ་»《སྣར་བཟུང་། «པེ་»«སྣར་»དད་པ་སྒོམ། @«པེ་»«སྣར་བགྱིས་སྙེད། ༄༅«པེ་»«སྣར་»འཐོ་བས། «པེ་»«སྣར་མེད་པ། «པེ་»«སྣར་སྣ་ཚོགས། «པེ་»«སྣར་དེ་སྔགས་ཀྱང་། «པེ་»«སྣར་»ནི། ཚེ«པེ་»«སྣར་པོ། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»སྟོད། «པེ་»«སྣར་»དྲང༌། (གྱེ«པེ་»«སྣར་»སྟོད། །(«པེ་»«སྣར་ཤི། «པེ་+མངྒ་ལཾ། «སྣར་»+སརྦ་མངྒ་ལཾ། །བཀྲ་ཤིས། །དགེའོ། །།།ཨེ་ཀ་་་ནཔཕྲིཙྪ། །པི་ཏ་ཀྱི་། ཚེ་ཚད་ལ་ཨ་ཏི། ་་་ཏདང་། །ད་བ་ས་ཀྱི་འོ་ཏེ་བ། ་་འཚོ་པ། ལོ་ག་བྱ་གཏ?''ད་རོཊ་ཁྲི་རིབ་ཁོ་ཚོ་བདཝ་ཏེ་འགྲོ་ཀོ་ཁྱོན། །ག་ད་བྱི་ཟླ་་་པའི་ཟླ་བ་མར་གྱི་ཁ་པར་མི་བ་ཁོ་པ། །གང་ག། །་ཚེ། ཀ་ཁར་།
