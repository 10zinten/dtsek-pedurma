རྒྱུད། ཝིཏི%ཁ་ཤེས་སྦྱོར་ལྡན་པས། །སྒྱུ་མ་ལྟ་བུའི་ལུས་ཀྱིས་ནི། །འཁོར་ལོ་སྒྱུ་མ་ལྟ་བུ་བྲི། །གྲོང་ཁྱེར་གྲོང་དང་གྲོང་རྡལ་དང༌། །ཀུན་དགའ་ཚོང་འདུས་གཙུག་ལག་ནི། ཎ་། ཁང་དུ་པུས་ནུབ་ཙམ་བརྐོས་ནས། །རིམ་གྱིས་ས་ནི་སྦྱང་བར་བྱ། ཌ་། ཤིང་དང་གྱོ་མོ་གསེག་མ་དང་། །དེ་བཞིན་རུས་པ་སོལ་བ་དང༌། ཀཱ་། སྐྲ་ལ་སོགས་པ་་་་བསལ་བྱས་ནས། ་། །དེ་ནས་ས་ནི་སྦྱང་བྱ་སྟེ། །ཀུན་ནས་སྟེང་ནི་ཡང་དག་པར༑་། བསྙམས་ཏེ་བ་ལང་ལྕི་བས་བྱུག། དྲི་ཡི་ཆུ་ཡིས་བསངས་ནས་ནི། །ཕྱི་ནས་དྲག་པོའི་སྦྱིན་སྲེག་བྱ། །སྔགས་པ་རྡོ་རྗེ་སེམས་ཟློས་ཤིང་། །རྡོ་རྗེའི་ལྟ་བས་བལྟ་བར་བྱ། །།དཀྱིལ་འཁོར་གྱི་ནི་ས་ཡི་ཆ། །་་། སྐྱོན་དང་ལྡན་ཡང་སྤྱོད་་་་པར་འགྱུར། །གསང་དབང་རྟོག་པ་ལ་གནས་ཤིང༌། ཁ་། བྱང་ཆུབ་སེམས་ནི་རྣམ་སྤངས་པའི། །སྔགས་པ་གང་ཞིག་ས་སྦྱང་བར། །བྱེད་པ་དེ་ཡིས་འགྲུབ་མི་་་འགྱུར། །ས་སྦྱང་བའི་ཆོ་གའོ། །ལ་ལ། ཚg/བོད་ས་བཀོལ་སའི་ནི་སྐྱེ་ཎ་གཡའ་ཞི་ཅིན་ཚོང་པ་ས་ཅཚོ་ཊ་ཚི། དཀྱིལ་བཅགདེ་ནས་བླ་རེ་སོགས་བརྒྱན་ཅིང་། །མཆོད་པ་ལྔ་དང་བཅས་པ་ཡི། ཀ་། འཁོར་ལོའི་ཁྱིམ་དུ་ཕྱིན་ནས་སུ། ཌ། ས་ནི་ཡོངས་སུ་གཟུང་བར་བྱ། །དཀར་ལ་མ་ཉམས་གོས་འཆང་ཞིང་། །གཙུག་ཏོར་དག་པ་ཡིས་བརྒྱན་ལ། །རྒྱན་རྣམས་ཐམས་ཅད་ཀྱིས་སྤུད་ཅིང་། ལ་། དྲི་:བཟངས་བྱུག་པའི་ཆལ་འཐོར་བས། །ཉི་མ་གསུམ་དུ་ཐུན་གསུམ་དུ། ་། ཆོ་ག་ཇི་བཞིན་གཏོར་མ་དབུལ། དེ་རིང་། དེ་ནས་བགེགས་རྣམས་ཞི་བར་ནི། ང་ཚོ། བྱ་ཕྱིར་ཕྱོགས་སྐྱོང་བཅུ་ལའོ། །བརྟུལ་་་་ཞུགས་བརྟན་པས་ཞི་བ་ཡི། །སྦྱིན་སྲེག་ཚུལ་བཞིན་སྔར་བྱས་ནས། ཏ། ཕྱི་་་ནས་དབང་བསྐུར་རབ་གྲུབ་ཕྱིར། །།སློབ་མས་ཀྱང་ནི་བྱེད་དུ་གཞུག། ས་ཡོངས་སུ་གཟུང་བའི་ཆོ་གའོ། ཆམ། །ཁ། །རྒྱུ་བ་ཁྲི་—།
