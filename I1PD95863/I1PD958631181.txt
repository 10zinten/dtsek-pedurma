བསྟན་འགྱུར། རྒྱུད། ༼ཅི་རྫོགས་སོ་ཞེས་པ་ནི་དོན་མཐར་ཕྱིན་པའོ། །སྨྲ་ཏི་ཛྙཱ་ན་ཀཱིརྟི་ཡིས། །བྱང་་་་་ཆུབ་སེམས་ཀྱི་འགྲེལ་བཤད་འདི། །བསྒྱུར་བས་འགྲོ་བ་ཐམས་ཅད་ནི། །བྱང་ཆུབ་སེམས་ནི་ཐོབ་པར་ཤོག། ལྷོ་བྲག་དྷརྨ་སེང་གེ་ཡིས། །ཆོས་རེའང་ལམ་རིངས་བགྲོད་ནས་ཞུས། །།།།་ཏི་ཀའི་ཀ་བ་ཁ་སོགས་ཉ་བའི་དུས་རྒྱ་གར་གྱི་མཁན་པོ་སྨྲེ་ཙེཏི་ཛྙཱ་ན་ཀཱིརྟིས་བསྒྱུར་ཅིང་བཤད་ནས་གཏན་ལ་་་ཕབ་པའི་བྱང་ཆུབ་སེམས་ཀྱི་འགྲེལ་པ་བཤད་པ་རྫོགས་སོ། །བློ་བ། ་་་་་::དཔའ་ལྔ་བ་དང་ད་ལྟ::ཀྱིས་བློ། ཏཊ། ཙོང་ཆུ་བཏུབ་ཟླ་བ་བསྡུར་མཆན། བ་བ་ན་བན་བྱ་བ་དེ་ཁ་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་བྱང་ཆུབ་སེམས་ཀྱི་་་་་འགྲེལ་པའི་རྣམ་པར་བཤད་པ་བཞུགས་སོ། །«པེ་»«སྣར་»ཊཱི། ན«པེ་»«སྣར་»ལས། ®«པེ་»«སྣར་»རྡོ་རྗེས། «པེ་»«སྣར་པའི། «པེ་»«སྣར་སྒོམ། «པེ་»«སྣར་»གསོལ་བའི། «པེ་»«སྣར་+དོན་ནི། :«པེ་»«སྣར་»བསྒོམས། «པེ་»«སྣར་»+ཀྱི། «པེ་»སྣར་སྒོམ༑ག«པེ་»«སྣར་ཐབས། @«པེ་»«སྣར་—པས། ༄༅«པེ་»«སྣར་པས། ཚེ«པེ་»《སྣར་བཞག@«པེ་»«སྣར་པ། དགྱེ«པེ་»«སྣར་»བ། ཀྱེ«པེ་»སྣར་»—དང་། «པེ་»«སྣར་»ཉན་ཐོས་དང༌། «པེ་»«སྣར་པ། «པེ་»«སྣར་»བསྒོམས། །«པེ་»《སྣར་»བ། «པེ་»«སྣར་»+དག་་་::://aང་«པེ་»«སྣར་བསལ། ཉི«པེ་»«སྣར་»བཞགཉི«པེ་»«སྣར་སྤངས་པ། ༄༅«པེ་»«སྣར་»པ། །«པེ་»《སྣར་བཞག«པེ་»«སྣར་»ཅན། «པེ་»སྣར་པ་བཞི། «པེ་»སྣར་»ཀྱི། «པེ་»《སྣར་»སོ། དགྱེ«པེ་»«སྣར་དམའ། ཉེ«པེ་»«སྣར་བཞག«པེ་《སྣར་པ་ནས། (ཅེ«པེ་《སྣར་པ། «པེ་»«སྣར་ཀྱིས། །མ་ལ་་རྗེ་པ་། བཀོ(«པེ་»«སྣར་བཞག«པེ་»«སྣར་»བ། «པེ་»«སྣར་+སྒྲོལ། ®«པེ་»«སྣར་པ། «པེ་»«སྣར་ལས། ་པོ«པེ་»«སྣར་»གཞོམ། ་«སྣར་»མུག་འ«སྣར་»ཤེས། «པེ་»སྣར་»ཀྱིས་སྦྱང་བ། («པེ་»«སྣར་»རྟགས། རྒྱུན་དང་:ན་ཀུན་འགྲུབ། «པེ་»«སྣར་རྟན། Q«པེ་»«སྣར་སྤངས། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་»—པའི། «པེ་»«སྣར་»པར། «པེ་»«སྣར་»པ། O«པེ་»«སྣར་»བརྟགས། །
