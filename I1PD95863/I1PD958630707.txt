བསྟན་འགྱུར། རྒྱུད། ༼རིཏརིམ་པ་རྣམ་པར་ཕྱེ་བ་ལས། །བདག་གིས་བསོད་ནམས་བསྒྲུབས་པ་གང་ཡོད་་་དེས། །འགྲོ་ཀུན་རྟག་ཏུ་དབང་དོན་ལྡན་པར་ཤོག། ཇི་ལྟར་མཁས་པ་དུ་མ་དག་གིས་མཛད་པ་ཡི། །དབང་གི་དོན་འབྱེད་ཆོ་ག་ཡོད་མོད་དེ་ལྟ་ནའང་། །ཀླུ་སྒྲུབ་ཞབས་ཀྱི་པདྨོའི་སྦྲང་རྩི་ཐོབ་པས་ན། །དེ་ཡི་དགོངས་པ་འདིར་ནི་བདག་གིས་གསལ་བར་སྤེལ། །ཟུང་དུ་འཇུག་པ་གསལ་བ་ཞེས་བྱ་བ་དབང་གི་ཆོ་ག་དཀྱིལ་འཁོར་ཆེན་པོའི་དངོས་གྲུབ་བརྙེས་པའི་མཁས་པ་ཆེན་པོ་སྒྲ་གཅན་འཛིན་་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །ཁ་པ། །བློ་ཁཊ་ན་ན་ཚོམཁས་པ་ཆེན་པོ་བསོད་སྙོམས་པ། ཤཱཀྱ་ཤྲཱི་བྷ་དྲའི་ཞལ་སྔ་ནས་དང་། །ཤཱཀྱའི་དགེ་སློང་ཀུན་དགའ་རྒྱལ་མཚན་དཔལ་བཟང་པོས་ཐོབ་སྟེ། ཐོས་ནས་་་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །འདི་ལ་བློ་ཀ་ཕྱེད་དང་བརྒྱད་བརྒྱའོ། །།།ཁ་ཆེའི་ལས་གཞིའིབ་་་་་་ཊ་། དྲོ་བྱེ་བ་ཌོ་ལན་ཊ། རྫོཊ་པའཚོན་ཟླ་བའི་བྱུགགྲོང་བ་ཚབསྡུར་མཆན། ཏུ་གློགལམ། ཥཊཁོ་Pageਸਰਲਨਲਹੈ_མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ...«པེ་»«སྣར་ཟུང་འཇུག་་་གསལ་པ་ཞེས་བྱ་བ་མཁས་པ་ཆེན་པོ་སྒྲ་གཅན་འཛིན་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པའོ། །།འ«པེ་»《སྣར་ཏཾ། ་«པེ་»«སྣར་ཀྲྀ། «པེ་»«སྣར་»རྟེན། །«པེ་»«སྣར་»གཞུང་། ངའི།  ། «པེ་»«སྣར་བའང་། z(«པེ་《སྣར་རྒྱུད། ཁོན«པེ་»«སྣར་གྱིས། ་༌«པེ་»«སྣར་»ཀྱི། ་ས«པེ་»«སྣར་གསང་། «པེ་»«སྣར་རྒྱུད། ་«པེ་»«སྣར་+དང༌། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་བཞག་《སྣར་»པར། ་༼«པེ་»«སྣར་»བཞག་ན«པེ་»སྣར་མཐོ། ཚེ«པེ་»«སྣར་ཞན། ངས«སྣར་པར། པོ«པེ་»«སྣར་»—ཡང་ན། «པེ་»«སྣར་ཡི། ་དང་«པེ་»«སྣར་པ། «པེ་»«སྣར་བྲི། །«པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་»ཡིས། །«པེ་»《སྣར་»དུ་བ། པ«པེ་»«སྣར་»ཚང་། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»ཀྱི། «པེ་»—།
