བསྟན་འགྱུར། རྒྱུད། རྩི་:»དེ་ལས་ཐོས་པ་རྫོགས་པར་བྱེད་པ་ཤེས་རབ་སྟེ། ཆུ་། བསམ་པ་དང་ནི་གཉིས་པོ་ཡོད་གྱུར་ན། །སྒོམ་པ་ལ་ནི་རབ་ཏུ་སྦྱོར་བ་ཡིན། །དེ་ལ་དངོས་གྲུབ་བླ་ན་་་་་མེད་པའོ། །བློ་དང་ལྡན་པས་འདི་ལྟར་བསམས་ནས་ནི། །ཚ། ཐོག་མ་ཐོས་པའི་་དོན་ལ་འབད་པ་ཡིས། པ། རྟག་ཏུ་བརྩོན་པར་བྱེད་པ་གང་ཡིན་པ། །དེ་ཡི་ལག་ཏུ་དངོས་གྲུབ་ལེགས་པར་འཐོབ། །གང་ཞིག་ཚིག་ལ་ཐེ་ཚོམ་ཟ་གྱུར་པ། །དེ་ཡིས་དོན་ནི་ཅི་ལྟར་རྟོགས་པར་འགྱུར། །དེ་རྣམས་རྨོངས་པ་རྣམ་པར་བཟློག་པའི་ཕྱིར། །འདི་ལ་བདག་གིས་རྣམ་པར་བརྩོན་པར་བྱས། །རྣམ་པ་ཀུན་དུ་བརྗོད་པར་ནུས་པ་ནི། །དེ་བཞིན་གཤེགས་པ་ཉིད་ཀྱི་སྤྱོད་པའོ། །འོན་ཀྱང་། བློ་གྲོས་དམན་པ་རྣམས་ལ་ནི། །རྣམ་པར་དབྱེ་བ་ཙམ་ཞིག་བདག་གིས་བྱས། །འཇིག་རྟེན་གང་དུ་ཉི་མའི་སྣང་བ་རྒྱ་ཆེན་མཆོད་པས་འགྲོ་ཀུན་གསལ་གྱུར་པ། །སྲིན་བུ་མེ་ཁྱེར་ལུས་ཀྱི་སྣང་བ་ཙམ་གྱིས་བྱ་བ་ཅི་ཞིག་བྱེད་པར་འགྱུར། །དེ་ལྟ་ན་ཡང་བློ་དང་ལྡན་པ་གཞན་དག་གིས་ནི་གསོལ་བཏབ་དཀའ་འགྲེལ་འདིས། །རིམ་པ་ལྔ་ཡི་དོན་ནི་གཏན་ལ་ཕབ་ཅིང་རྣམ་ཕྱེ་འཕགས་རྣམས་བཟོད་པར་བཞེས༑། ཐམས་ཅད་མཁྱེན་པའི་གསུང་རབ་བློ་དམན་རྣམས་ཀྱིས་བཤད་པར་ནུས་པ་མིན་པ་དང༌། །།མཁས་པ་རྣམས་ཀྱིས་གཞུང་འདི་རྣམ་ཕྱེ་མཐོང་མིན་་་་ཕྱོགས་ཙམ་བདག་གིས་མཐོང་བས་ན། འ། སེར་སྣ་དང་ནི་མཆོག་ཏུ་བྲལ་བའི་བློ་་་ཡིས་རྣམ་བཤད་གང་ཡིན་དེ་ཡིས་ནི། །དམན་པ་འགྲོ་དོན་ཡང་དག་བརྩོན་པ་བློས་ྋསྤྱན་རྒྱས་པ་རྣམས་ནི་དགྱེས་པར་མཛོད། ཌ། འདི་ལྟར་མན་ངག་གཅིག་་་་ནས་གཅིག་ཏུ་ཡོཥས་བརྒྱུད་རིམ་པ་ལྔ་པོ་རྣམ་བཤད་པས། །གང་ཞིག་བསོད་་་་ནམས་སྟོན་ཀའི་ཟླ་བ་དྲི་མེད་ལྟ་བུ་བདག་གིས་ཐོབ་གྱུར་པ། །དེ་ཡིས་སྲིད་་་་པའི་འཁོར་ལོར་ལྷུང་གྱུར་འགྲོ་རྣམས་རྫོགས་པའི་བྱང་ཆུབ་ཀྱི། །རྩེ་མོར་རིམ་་་།
