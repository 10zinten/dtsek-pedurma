རྒྱུད། ་༼འདི་ཡིན་པའི་ན་འགྲོ་བ་ལྔ་རྣམས་སུ་འཁོར་ལོའི་ཚུལ་གྱིས་འཁོར་བའི་ཕྱིར་རོ། །དེ་རྣམ་་་་པར་:སྤྱོན་པའོ། །དེ་རིང་བཤད་པའི་རིན་ཆེན་གྱིས་ནི་དེ་རིང་ནི་ཉི་མ་འདི་ལ་སྟེ་དང་ངོ༌། །བཤད་པའི་རིན་ཆེན་གྱིས་ནི་ཀུན་རྫོབ་ཀྱི་བདེན་པ་བཤད་པའི་་་རིན་ཆེན་གྱིས་བཀའ་དྲིན་མཛོད་ཅིག། བདག་ལ་ནི་བདག་གིའོ། །གཙོ་བོ་ནི་་ཁྱབ་བདག་གོ། །ཁྱོད་ཞབས་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །དེ་ལྟར་་་་་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་དེ་ལྟར་སློབ་མའི་ཚིག་དེ་ཐོས་ནས་ཅིར་འགྱུར་ཞེ་ན། །ཡང་དག་པར་གསོལ་བ་འདེབས་པའོ། །འཇིགས་མེད་གསོལ་འདེབས་པ་ནི་གཙོ་བོ་མཐར་ཐུག་པ་ལའོ། །སྙིང་རྗེ་ཞེས་པ་ནི་ཁོང་དུ་ཆུད་པར་སླའོ། །དེ་ལས་་་བདག་བྱིན་བརླབ་པ་བརྩམ། །ཞེས་པ་ནི། བདག་ལ་བྱིན་གྱིས་བརླབ་པ་་་་བཤད་པར་འདོད་པས་བླ་མས་རིམ་པ་ཉེ་བར་བྱའོ། །བདག་བྱིན་བརླབ་པ་་་་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་ཀུན་རྫོབ་བདེན་པ་བསྟན་པ་ནི་སྟེ། དེ་ཅི་ལྟ་ཞེ་ན། །ཤིང་དང་རི་དང་ཆུ་བོ་དང་རྒྱ་མཚོ་ལ་སོགས་པ་དངོས་པོའི་གྲོང་ཁྱེར་ནི་ཀུན་རྫོབ་བོ༑། འདི་དང་འབྲལ་བས་བདེན་པ་ཁོང་དུ་ཆུད་པར་མི་འགྱུར་བ་ཉིད་དོ། །དེ་བཞིན་དུ་སློབ་དཔོན་ཟླ་བ་གྲགས་པའི་ཞབས་ཀྱིས་ཀུན་རྫོབ་ཀྱི་བདེན་པ་ཉམས་པས་འཇིག་རྟེན་པའི་དངོས་གྲུབ་མེད་དོ་ཞེས་བཤད་དོ། །བདེན་པ་བསྟན་པ་་་་འདི་ནི་ཡེ་ཤེས་གསུམ་པོ་ཐ་དད་པ་ལས་འདས་པའི་ཡེ་ཤེས་ཀྱི་རང་བཞིན་གྱི་ལུས་བསྟན་པའི་བླ་མའི་ཞབས་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །གང་གིས་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་སྒྲུབ་པ་པོ་གང་གིས་བླ་མ་ལས་དངོས་སུ་བདག་བྱིན་གྱིས་བརླབ་པའི་རིམ་པའི་མན་ངག་གཉུག་མའི་ལུས་སུ་གནས་པ་རྙེད་པར་མི་འགྱུར་ཏེ་ཐོབ་་་པར་མི་འགྱུར་བའོ། །མདོ་སྡེ་རྒྱུད་དང་རྟོག་པ་ལ་སྟེ་ཆོས་ཀྱི་ཕུང་པོ་སྟོང་ཕྲག་་་བརྒྱད་ཅུ་རྩ་བཞི་རྣམས་སུ་རྣལ་འབྱོར་པ་དེའི་ངལ་བ་ནི་དུབ་པ་ཞེས་པའི་རྣམ་་་་—།
