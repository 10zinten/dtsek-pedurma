ཀབྱབསྟན་འགྱུར། རྒྱུད། རིའིཁ་ཀ། «པེ་»«སྣར་»ཙིནྟ། ་«པེ་»«སྣར་རྣམས་རྒྱན། ་་་་་་་་གི་ཀྱི་ཡོལ་བ་ཚོ་ཎཎོ། །པར་ཁོ་པ། །།འདུལ་བ་ཁ་ལྟཁའ་ལ་རྒྱུ་བ། །་་རྒྱུ་སྦྱོ། ངའ་བློnཀཀཱ་དུ་ལུ། ཁ་མ་གང་བར་དུ་དཔལ་དུ་འབར་གས། ཤིང་ཆ་དེ་ལ་པ་གང་།  ། ལས་ཀྱི་ཁ་པའི་སྐྱེ་བ་བཀག་ནས། དང་བསྡུར་འབྲས་རེའུ་མིགང་ཚོ་ཁྱབློ་དག། འདུག་། ་་་་་་་། །ས། །ཀ་ཁ་ག་ག༌༌་དཔེ་གཞི། སྡེ་དགེ། །བ། པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། ་ལྟ་ཡང་སྡེ་ཚན། །་རྒྱུད། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། ཀྱི་བདེ་ལྡེབ། བ?ནབ། བན་ནབནལ་པའི་རྒྱུད། (རྒྱུད། ཕྲེང་། །ཚེག་ཁྱིམ། ཏུ་འཕུར་བཛད། སྤྱི། འདོད་པ་འདུ་ཁང་། ལ། ཤོག་ལྡན་ཏེ། །།ཡང་སྟོན། དེ་བས། །།ཞེས་སོ། །དེ་དང་། ::ཌ་ཚོཝང་ཁ་ཀཀྐཌ་འི་ཁ་ཌ་ཚེ། །ཅཚོདྲགས་་ཀྱི»། ཨཏི་ལྔ་དངོpཏང་《ོ《འི་ཁང་ཆ}༼རྩེ་པར་ལ་་་་་་་་་་་་་་་་་་ད་བྲུཀཏི་ཀར་:འི་སྐད་ཀྱིས་སོ། །་་་་་་་། དེ་དག་ས་་་་་་༌་་་་ད་བྱེར་བ་དེ་དང་། ཏི་བྱེད་ན་མ་གྲགཔ་aངོ་། 《༈ར་!Fབང་ལ《/ལོ{TAསྔབབ་ལ་ཕྱིར་དཞིང་ཁྱད་པར། དད་པ་ལ་དོན་དེ་དང་དུ་ར་མ་ལ། །ང་དང་ཁྱི་མ། མེང་ལ་ཁྲི་དང་། དེ་བ་ཡོད་དུ་གོམ་པའི་མོས། །བགེགས་དང་བ་དང་ད་དེ་འདྲ་བ་ག་འདྲ་བ་འགའ་བ། ཀྱི་ན་སྤྱི་ད་པ། ད་ལྟ་བ་ལོ་བཀོད་པ་དང་། ཁྲིབ་ཏེ། ཞུ་བ། །དགེ་མང་དུ་འདུ་ན་བྷ་དང་། གྱི་ངང་། །མ་_།
