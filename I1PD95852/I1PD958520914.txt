རྡོ་རྗེའི་ཚིག་འབྱེད་པ་ཞེས་བྱ་བ། གནས་སྲུང་མི་བྱེད། །སངས་རྒྱས་རྒྱལ་མཚན་སུན་འབྱིན་ཅིང་། །།ཕས་ཕམ་ལ་སོགས་རབ་ཏུ་བྱེད། །གང་གིས་བཏུང་བའི་མཆོག་མི་ཤེས། ཀ། སཱ་ལུ་ལ་སོགས་བཅོས་སྦྱར་བའི། །མྱོས་པའི་ཁུ་བ་འཐུང་བྱེད་ཅིང་། །དེ་ཡིས་ཕུང་ཁྲོལ་རབ་ཏུ་བསྐྱེད། ཚེ། ཅེས་ྋརྒྱལ་པོ་ཟླ་བ་བཟང་པོས་གསུངས་སོ། །དེ་བས་ན་གང་དག་རྡོ་རྗེའི་ཚིག་མ་ཤེས་པས་ལོག་པའི་ལམ་དུ་ཞུགས་པ་དེ་རྣམས། ལོག་པའི་ལམ་ལས་་་བཟློག་པར་བྱ་བའི་ཕྱིར། རྒྱུད་དེ་དང་དེའི་ངེས་པ་དང་དྲང་བའི་དོན་བདག་་་་་་གིས་བླ་མ་དམ་པའི་ཞལ་ལས་ཐོས་ཤིང་ལེགས་པར་བསམས་པ་དང་། གཞན་་་་ཡང་གསང་སྔགས་ཀྱི་སྒོ་ནས་སྤྱད་པ་སྤྱོད་པའི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་ཇི་་ལྟར་སྤྱད་ཅིང་བསླབ་པར་བྱ་བའི་ལུང་དང་རིགས་པས། །དབང་རྣམ་པ་གཉིས་ཀྱི་རབ་ཏུ་བྱེད་པ་བསྡུས་པའི་རྒྱུད་བརྟག་པ་ལྔའི་དོན་གསུངས་པ། འདིའི་དྲང་བ་དང་ངེས་པའི་དོན་ཇི་ལྟ་བ་བཞིན་དུ་དགྲོལ་བར་བྱའོ། །མི་དབང་པད་དཀར་ཕྱག་ན་པདྨོ་འཛིན་པ་གང་། །དེ་ཡིས་བསྡུས་ལེགས་རབ་ཏུ་བཀྲལ་གྱུར་མོད། །ོ། འོན་ཀྱང་ཕྱི་མཁན་མཁས་པའི་རྗེས་ནས་མཆན་བུ་རྣམས། །གཟོང་དང་སྟ་རེ་ཐོགས་ནས་རྗེས་བཞིན་ལེགས་འབྲང་དང་། །མཆན་བུ་བདག་ཀྱང་མི་དབང་ཕྱ་མཁན་རྗེས་འབྲངས་ནས། །ཤེས་རབ་གཟོང་དང་སྟ་རེ་ལེགས་པར་རབ་བཟུང་ྋསྟེ། །ལུང་རིགས་བླ་མའི་གདམས་ངག་:བཟོ་བོའི་ཐིག་བཏབ་ནས། །ཐར་པའི་གྲོང་ཁྱེར་བརྩིགས་ཤིང་སྒྲ་ཡི་འགྲེལ་པ་བྱེད། །ཎ། འགྲེལ་འདི་མ་གཏོགས་རིན་པོ་ཆེའི། །རྣམ་གཉིས་དབང་གི་དེ་ཉིད་ནི། །གསལ་ཞིང་ཐེ་ཚོམ་མེད་པ་དག::། མཁས་་རྣམས་ཀྱིས་ནི་རྙེད་མི་འགྱུར། །ཀ་། དབབ་པ་སྤངས་པའི་ཕྱིར་ན་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། དབབ་པ་ཞེས་བྱ་བ་ནི་སྐྱེས་བུ་གང་འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་དབང་མངོན་པར་བསྐུར་བར་འདོད་པ་དེ་ལ། ཐུན་མོང་གི་དངོས་གྲུབ་ཀྱི་ས་།
