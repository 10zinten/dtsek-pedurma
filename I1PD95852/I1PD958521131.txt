བསྟན་འགྱུར། རྒྱུད། མི་ལ་ལས་ཀླུའི་འཁོར་ལོའོ། །ལྟེ་བར་ལ་ཡིག་ལས་འབྱུང་པོའི་འཁོར་ལོ་རབ་ཏུ་གཞུག་པ་ཞེས་བྱ་བ་ནི། གཤེགས་སུ་གསོལ་བའི་ཆོ་གའོ། 《་། མགོན་པོ་བྱིས་པ་བློ་་་་་དམན་རྣམས་ཀྱིས་དྲན་ཕྱིར། །དཔལ་ལྡན་འཇིག་རྟེན་མགོན་པོ་ཉིད་ཀྱི་སྒྲུབ་་་་ཐབས། །ལས་ནི་ཕྱུང་བས་དེ་ཡི་ཕྱིར་ན་ཡབ་དང་། །མཚུངས་པར་གང་ཞིག་མི་:གཅིག་ཕ་ོལ་བཟོད་པ་ལྟར་དུ། །བདག་ལ་རབ་ཏུ་བརྩེ་བས་ཐུགས་རྗེས་་་་བརླན་པ་བཟོད་པར་མཛོད་ཅིག། དེ་ལྟར་དཔལ་དང་ལྡན་པའི་འཇིག་རྟེན་་་་་དབང་ཕྱུག་གི་སྤྲུལ་པ། པདྨ་དཀར་པོས་ཡང་དག་པར་མཛད་པའི་དྲི་མ་མེད་པའི་འོད་ལས་ཕྱུང་བ། བཅོམ་ལྡན་འདས་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་སྒྲུབ་ཐབས་ཀྱི་ཆོ་གའི་རིམ་པར་ཕྱེ་བ་རྫོགས་སོ། །།ཁྲོཎི། ཞི་བློ་ལོ་ཏེ་རྒྱལ་པོ་ཟླཡེ། ཏིཊ་ཌོ་དུས་ཀྱི་འཁོར་ལོ་པ་ཆེན་པོ། པཎྜི་ཏ་སཱ་དྷུ་པུ་ཏྲས་པཎྜི་ཏ་ཆེན་པོ་:དྷརྨ་ཨཱ་ཀར་ཤཱནྟིའི་དོན་དུ་མཛད་པ། རྒྱ་གར་གྱི་པཎྜི་ཏ་ཆེན་པོ་ཝཱ་གཱི་ཤྭ་ར་གུསྟ་པ་དང༌། སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་རབ་བདག་གིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་་གཏན་ལ་ཕབ་པའོ། །།།ཅལྔ་ལྔ། །ཊ་ཋཡུ་ནའི་བློ་གའི་ཟླ་བ་གུར་བསྡུར་མཆན། མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»+ནི། «པེ་»«སྣར་དུས། །།ད། «པེ་»«སྣར་གྱིས། ཚེ«པེ་»«སྣར་»—པ། མི«པེ་»«སྣར་—དཔལ། «པེ་»«སྣར་+པ། གྱི«པེ་»«སྣར་»+དབང་པོ་རྣམས། ཁོ«པེ་»«སྣར་»ཧཱུྃ། ཁ«པེ་»«སྣར་»དྷི་པ་ཏི་མ་མ་ཀཱ་ཡ་བཱཀྑིཏྟ་ཏ་ཛྙཱ་ན་བཛཱི་བཛྲ་བཛཱ་མི། O«པེ་སྣར་ཐཾ། «པེ་»སྣར་»ཀནྡྷ། ཁྱི«པེ་»«སྣར་བྷ། ཨ༌ཀྱེ«པེ་«སྣར་བསམས། ཁྱིལེ་»«སྣར་རྗེས། །«པེ་»«སྣར་ཨོཾ། «པེ་»«སྣར་»དྷརྨ། «པེ་»«སྣར་»ཨ། ཁྱི«པེ་»«སྣར་ཐུགས་་་།
