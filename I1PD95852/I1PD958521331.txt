བསྟན་འགྱུར། རྒྱུདཅེས་བསྒྲུབ་པར་བྱ་བ་དང་། སྒྲུབ་པའི་ངེས་པ་བཅོམ་ལྡན་འདས་ཀྱིས་བསྟན་ཏོ། །སྦྱོར་བ་ཡན་ལག་དྲུག་ཅེས་བྱ་བ་གསང་བ་པའི་བདག་པོ་ས་བཅུའི་དབང་ཕྱུག་བྱང་་་་ཆུབ་སེམས་དཔའ་ཕྱག་ན་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ། །།།Fཞོ་ཅིགiཥ་ཎnཔ་རྒྱ་གར་གྱི་མཁན་པོ་ཆེན་པོ་དཔལ་རབ་འབྱོར་ཞི་བའི་ཞལ་སྔ་ནས་ལོ་ཙཱ་བ་ཅོག་རོ་དགེ་སློང་ཏིང་འཛིན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་བོ། །ད་ཚུ་གེ་ར་ཞིག་ཡིན་པའི་ཁབ་འདྲ་བ། མཚོ་ནག་པ་བློ་ཀཊིཀ་ཀུ་ཞ་འབིབྷིp/བྱེའི་ས་སྐྱེ་བ་ཁན་ཊ་ཊ་ཨང་་་་་་་ཁ་ཎ་nཤོག་གྱུར་ལ་ཁ་ཡ༌ཀྱང་ང་ལ་བདག་ཁ་པ་ཀུའི་ཆ་ཀློག་རྒྱ་ཁཚོ་རྟ་ཀོ་བསྡུར་མཆན། །ཀཱ་པ་ར་མི་ཏཱ་ཏིཥཀ་ཏི་ཀང་དང་ལུག་བདུད་ཡུལ་ས་གནས། བློ་ལ་ཆུད་པ་གནag::ཀ་ར་ཀ་བྱ་ཕྱོགས་ཀྱང། :མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»སྦྱོར་བ་ཡན་ལག་དྲུག་པ་ཞེས་བྱ་བ་ས་བཅུའི་དབང་ཕྱུག་ཕྱག་ན་རྡོ་རྗེས་མཛད་པ་བཞུགས་སོ། «པེ་»«སྣར་»ཀཾ། མི«པེ་»«སྣར་»+ཡན་ལག་ཉིད་e«པེ་»«སྣར་»ཅེས། «ཙོ་»ན། །p@ག་ཏུ་བཏི་པའི་ཁ་«པེ་»«སྣར་»དེ། །གང་བསང་དང་རྒྱལ་ཏུ་བྱ་ར་བ་ན་བསམ་ན་གླུ་བློ་n«པེ་»«སྣར་»ན། ཨི«པེ་»«སྣར་»སྟེ«པེ་»«སྣར་»ཏིར་། «པེ་»«སྣར་»ཨཱཋས། «པེ་»«སྣར་»བ། «པེ་»«སྣར་»མཁའ་སྣང་། «པེ་»«སྣར་»མའི། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»སྤངས«པེ་»«སྣར་»སུ། «པེ་»«སྣར་»བསྒོམ། ༄༅«པེ་»«སྣར་»+ལ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»སྒོམ། ཨི«པེ་»«སྣར་»སྒོམ། དེང་«པེ་»«སྣར་»སྒོམ། »pཀྱི་ཉི་སྤྲུལེ:ཀ་/ང་ngབ་འཚོ་ཚེ་དེབ། «པེ་»«སྣར་»ཆེན། «པེ་»«སྣར་»འགྱུར་བའོ། «པེ་»«སྣར་»«ཙོ་»བསྒོམ། «པེ་»འ«སྣར་»+ཀྱིས། «པེ་»«སྣར་»ཡང་། ༄༅«པེ་»«སྣར་»སྒྲུབ། «པེ་»«སྣར་»ཀྱིས་«པེ་»«སྣར་»+པ། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་»འགྲུབ། *ད་བྱའི«པེ་»«སྣར་»+ངེ་། «པེ་»པའོ། །དགེའོ། །«སྣར་»པའོ། །ན་ངམ་པའི། །ཡི་ག་་་་་་་་་་་་་ཁྱོ་ག་དང་ཁ་འཐོར་གྱི་བཀ་ཁའཁ་ཐག་པ་ཤ་བག་ལ་ཏེ་ཀྲེ་ལ་ཡུན་ལ་གཞིག་ནན་མ་འཚོལ་ཚེ། ཁཆོག་ལ་ཁྲི་བ་ཏ་ཁ་ཁ་ཁ་བ་ལ་།
