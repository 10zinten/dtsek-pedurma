རྡོ་རྗེའི་ཚིག་འབྱེད་པ་ཞེས་བྱ་བ། དེ་ཡི་འགྲེལ་པ་མཛད་པར་གྱུར། །ཐུགས་རྗེ་མངའ་བའི་རྐང་གཉིས་གཙོ་བོས་ལུང་བསྟན་པས། །དེ་རྗེས་འཇམ་པའི་དབྱངས་ཀྱིས་བསྡུས་པའི་རྒྱུད་འདི་བསྟན། །འགྲོ་རྣམས་ངོ་མཚར་བསྐྱེད་ཕྱིར་སྤྱན་རས་གཟིགས་དབང་གིས། །།འཇམ་པའི་་་་དབྱངས་དང་སྒྲོལ་མའི་སྲས་སུ་འགྱུར་ཞེས་བསྒྲགས། །དེས་ཀྱང་སྔོན་གྱི་དམ་་་་་བཅས་བཤོལ་བ་མི་མངའ་བས། ལྟ་། བསྡུས་རྒྱུད་མེ་ཏོག་ཕྲེང་འགྲེལ་དྲི་མ་མེད་འདི་མཛད། །རབ་འབྱེད་དབང་གི་འགྲེལ་པ་རྡོ་རྗེའི་ཚིག། ལེགས་པར་འབྱེད་བྱེད་བདག་བརྩམས་་དགེ་བ་གང་། །ཆུ་བོའི་རྒྱུན་ལྟར་རབ་འབབ་དགེ་བ་དེས། །ཆོས་འབྱུང་དྲི་མེད་གྲོང་ཁྱེར་མཆོག་ཐོབ་ཤོག། དབང་གི་རབ་ཏུ་བྱེད་པའི་འགྲེལ་པ། རྡོ་རྗེའི་ཚིག་འབྱེད་པ་ཞེས་བྱ་བ། དགེ་སློང་མཁས་པ་ཆེན་པོ་དཱ་རི་ཀ་པས་མཛད་པ། བོད་ཀྱི་ལོ་ཙཱ་བ་ཅདགེ་སློང་ཤེས་རབ་གྲགས་པས་བླ་མ་ལས་མཉན་་་་ནས་བསྒྱུར་བ་རྫོགས་སོ། །།ཚེ་ཚེ་རབས་། ་ཌན། ཛ་པ། སྐྱས་ཚོུ་ལྔའི་.jpgཀ་ཨུ་རngབསྡུར་མཆན། 《་:c(ི་@མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་དབང་གི་རབ་ཏུ་བྱེད་པའི་འགྲེལ་པ་རྡོ་རྗེའི་ཚིག་འབྱེད་ཅེས་བྱ་བ་བཞུགས་སྟོ། །ཉི«པེ་»«སྣར་»ཛཱ། ཉི«པེ་»«སྣར་»ཀྲྀ«པེ་»《སྣར་མ། /«པེ་»«སྣར་»རྡོ་རྗེ། ཕྱི་ཚེ«པེ་»«སྣར་+གྲུབ། «པེ་»«སྣར་»ལྟར། «པེ་»«སྣར་»དེ་རྗེས། «པེ་»«སྣར་»ཡིས། །སད་@«པེ་»«སྣར་»མོར། «པེ་»«སྣར་»འབེབས། །#«པེ་»«སྣར་»འབྱེད། ཨི«པེ་»«སྣར་»ཀྱེ«པེ་»«སྣར་»གི། «པེ་»«སྣར་རིག«པེ་»«སྣར་»འབྱེད། «པེ་»«སྣར་»༄༅། ༄༅«པེ་»«སྣར་»འཆན། ཉི«པེ་»«སྣར་»ཝ། «པེ་»ཏུ«སྣར་»གཟུང་། ཚེ«པེ་»«སྣར་»གཟུ་པོའི། །:«པེ་»«སྣར་»སྐྱེ་བོ། ཚུ«པེ་»«སྣར་»སྤོ། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་»བབས། འོ་།
