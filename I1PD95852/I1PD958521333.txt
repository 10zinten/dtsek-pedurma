བ་སྤུ་བ་བ་རྣམས་སྤྲུབ་བ་བ་བྱ་བ་དེ་དག་སྒྲུབ་ཀྱི་ཤེས་པ་བཅོམ་ལྡན་འདས་ཀྱི་དག་པས་ན་ཏོ། བློ་བག་ག་ག་དེ་ག་ང་བ་མ་གད་ནས་ཏྲེ་བ་ད་ལུ་པོ་ལ་ཟླ་བྱུང་བ་དག་བྱས་ཀ་ཡ། ད་ཕྱི་བར་དོ་བ་དང་བགྱུད་སྤྱི་བའི་་་་་་་་་་་རེད། ༄༅། །དཔལ་དུས་ཀྱི་འཁོར་ལོའི་སྙིང་པོའི་རྒྱན་ཞེས་བྱ་བའི་སྒྲུབ་ཐབས། ལས་ཀྱིོད་ཆེན་(། ༼འཚོ་དེང་རབས་སྤྱོད་ཀྱང་། ་དྲུག་པ་ལ། ༼ཁ་བཏོན་པ། རེས་༄༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་ཀཱ་ལ་ཙཀྲ་གནྡྷཱ་ལཾ་ཀ་ར་སཱ་དྷ་ནཾ་ནཱ་མ། བོད་སྐད་དུ། དཔལ་དུས་ཀྱི་འཁོར་ལོའི་སྙིང་པོའི་རྒྱན་ཞེས་བྱ་བའི་སྒྲུབ་ཐབས། བཅོམ་ལྡན་འདས་དཔལ་དུས་གསུམ་གྱི་བདེ་བར་གཤེགས་པ་ཐམས་ཅད་ཀྱི་མང་་་བདག་དཔལ་རྡོ་རྗེ་འཆང་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །དུས་འཁོར་བདེན་པ་གཉིས་ཀྱི་བདག། ཆོས་སྐུ་ཟབ་མོ་ལོངས་སྤྱོད་རྫོགས། །འགྲོ་བ་དད་མོས་སྤྲུལ་སྐུ་དང་། །མཉམ་པ་ཉིད་ལ་ཕྱག་འཚལ་ལོ། །ཀཱ་ནི་མི་འགྱུར་ཞི་བ་སྟེ། །ལ་ནི་འདི་ལ་སྟོབས་དང་ལྡན། །ཙ་ནི་སེམས་ཀྱི་གཡོ་བ་ཉིད། །ཀྲ་ནི་:རིགས་ཀྱིས་པ་འཆིང་བ་སྟེ། །ཡེ་ཤེས་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །ཐམས་ཅད་རྨི་ལམ་ཐུགས་ཆུད་ཅིང་། །མི་མཁྱེན་མི་མངའ་ས་ལེ་མཁྱེན། །ཐུགས་ཀྱི་ཆོས་དབྱིངས་ཉིད་ལས་་་་འབྱུང་། །ཡེ་ཤེས་ཐུགས་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ཀྱི་སྐུ་ཡི་གསུང་དང་ཐུགས། །ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་གསུང་ཐུགས། །སྤྲུལ་པའི་སྐུ་ཡི་སྐུ་གསུང་ཐུགས། །ཡེ་།
