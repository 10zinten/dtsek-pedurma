ཤཱཀྱ་ཤྲཱི་བྷ་དྲས་མཛད་པ་རྫོགས་སོ། །ཀ་ཀ། །ཎི་པའི་ཁཅཁ་ཎྜ། །ཚེ་ཀ་ཎའི་ཏི་པཎ་ཊི། །ཏེ་ཁ་ཟླ་བ་ཁཥ་བ་ཀཚོ། །་་་་་་ཌའི་ཆཏ་སྐཁབློ་བལྟ་བྱ་ਬੰਸਲਓRਇਆਕਰਨਾਲਾਤੋਂਲਸ਼ਸਪੈਸਲਿਸ਼། །ཚོ་ལ་ཏོ་ལ་ཁ་ཎ་ཁ་ཎབསྡུར་མཆན། ཁ་། ཏ་ཟི་ཁཚོ་ལ། །ཁzaཉམཚན་བྱང་འདི་ལྡེ་»«ཙོ་»ལྟར་བཀོད། ༄༅«པེ་»«སྣར་ངེས་པ། ཚ«པེ་»«སྣར་བྱས། #ཁྱག་ར་«པེ་《སྣར་»བགོད། «ཙོ་མ། །བྱ་བ་འདི་ཆ་e། །བླ་མ་བྱ་བ་ལ་«ཅོ་ཅུ། ཚེ«པེ་»«སྣར་བཞི། «པེ་»«སྣར་འགྲོ། བདུད་ཀཀ་ཡོS་བ་བཏུཟླ་%ཡ་Peཆུ་ཁ#། །ཁ་དང་གསང་ཨང་། ཁ་:བླ་ཚོལ། །ཏའིང་ཏོ''pད་རི་ཁ་རྒྱུ་ཎ་eཎ་ཆ། །eཚོ་ཁཚོགས་སུ་ཡབ་བསྡུར་འབྲས་རེའུ་མིག་ཅཡོང་ཚེ་Tལ་yppཀ་ཁ་ཀ་ད་བ་ཏེ། །ཟླ་བ། ལ་pin/''རྒྱུད། (ཀ་ཀ། ཏོར་ཆ་ར་ཎའཏིན་པ་[དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། །སྡེ་ཚན། _རྒྱུད། རྒྱུད་འགྲེལ། ::ཀྱིརྒྱུད། ཀརྒྱུད། བོད་རྟགས། རྒྱུ་པ་ཚོ། ང་༽:ང་དཔག་ཚན་གྲངས། ལྡེབ། བ་—ལྡན་བ། བ་བ་བ་བ་—ན་ཕྲེང་། ཚིག་རྐང་། །pང་བ་ཚེག་ཁྱིམ། །པས། །ནརྒྱུ་བ་བ་ཏེ་ཁའི་ནང་ཚན་ཁཊིཀ་ལ་ན་ཚ་གདོད་ཁ་ཡིན་ན་ན་ཁྱོད་ཀྱིས་ཀ་ཁ་ཁ་རྒྱུ་ནད་པའིsཎ་་་་་བག་ཞི་བ་ཞེ་ཟླ་ཏེ། དེ་ਦੇਕਦਾਲਸਟਰਾਅਬੇਗਮਾਲਸਨམགོ་གང་བ་ཞིག་ཆི་ཀ་ཏཆུ་ན་ཁ་ཁ་ཁའི་ཐུག་ན་ཕྱི་ལ་བཏོ་དཀློབ་འི་་་་།
