ཁ་དན་ཏཱ་དི་མ་ལཾ་བི་ཤོ་དྷ་ཡ་བི་ཤོ་དྷ་ཡ་སྭཱ་ཧཱ། ཞེས་བྱ་བའོ། །སློབ་མས་སོ་དག་བྱས་ནས་ནི། །དཀྱིལ་འཁོར་ལ་ནི་དོར་བར་བྱ། །མགོ་བོ་གང་གིས་གང་དུ་ལྷུང་། དེ་ལ་གནས་པའི་ལས་གྲུབ་པའོ། །དེ་ནས་སློབ་མ་བསྲུང་བའི་ཕྱིར། །སྤྱི་བོ་་་མགྲིན་པ་སྙིང་ག་དང་། །ལྟེ་བ་གསང་བ་གཙུག་ཏོར་དུ། །ཨོ་ཨཱནྟཧཱུྃ་:ཧོཧཾ་ནི། རིགས་དྲུག་པ་ནི་དགོད་པར་བྱ། །ཕྱི་ནས་དེ་ལ་སྔགས་འདི་ཡིས། །རྡོ་རྗེ་སེམས་དཔའ་བསྐུལ་བར་བྱ། །དེ་ལ་སྔགས་ཀྱི་རྐང་པ་རྣམས་ནི། །ལྟ་། ཨོ་ཨ་མིཨཱ་ཨོཾ་ཨཱ་་་བཛྲ་སཏཱ། མ་ཧཱ་སུ་ཁ། །བཛྲ་ཀཱ་ལ་ཙཀྲ། ཤི་་སྐྱ། ཨ་བྷི་མུ་ཁོ་བྷ་བ། ཉེན་སན་དུཊོ་ཝིབྷ་ཝ། བ་ར་དོ་བྷ་བ། །ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ་ཨ་དྷིཥྛ་ན་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་པའོ། །དེ་ནས་དཀྱིལ་འཁོར་དབུས་སུ་ནི། །བུམ་པ་རྣམ་པར་རྒྱལ་བ་་་་དགོད། །ཤར་དང་ནུབ་དང་ལྷོ་དང་བྱང་། །རྡོ་རྗེ་ཐིག་སྐུད་ས་ཚོན་སྣོད། །རྡོ་རྗེ་དྲིལ་བུ་རིམ་པས་དགོད། །དེ་ནས་བཟང་ངན་ཤེས་པའི་ཕྱིར། །ལྷོ་ཡི་ཕྱོགས་སུ་ཉལ་ས་ནི། །བྱས་ནས་བྱིན་གྱིས་བརླབ་པར་བྱ། །བྱིན་གྱིས་བརླབ་པའི་སྔགས་ནི། :ཨོཾ་ཧོིབཛྲ་ཀ་མ་ལ་ད་ལ་གབྷེ་:ཤ་ཡེ་བཛྲ་སུ་ཁ་མ་ཧཱ་ནི་:དཾ་ཀ་རོ་མི་ཡ་ཙྪཱཐཱ་ཏ་ཐཱ་ག་ཏ་ན་ཀྲྀ་ཏཱ་ཧིཧསྭཱ་ཧཱ་ཞེས་བྱའོ། །དེ་ནས་ནུབ་ཕྱོགས་མགོ་བོ་དང་། །ཤར་དུ་རྐང་པ་བསྟན་པར་བྱ། །e། བསྒོམ་པ་ལ་སོགས་སྟོབས་ཀྱིས་ནི། །རྨི་ལམ་་་བཟང་ངན་བལྟ་བར་བྱ། །གལ་ཏེ་སངས་རྒྱས་བྱང་ཆུབ་སེམས། །ལྷ་མོ་རྣམས་ནི་ཕྱག་མཚན་བསྣམས། །གླིང་བུ་ཌ་རུ་ལ་སོགས་ནི། །མཐོང་བ་དག་པར་འགྲུབ་པར་འགྱུར། །ཡང་ན་རྡོ་རྗེ་མཁའ་འགྲོ་དང་། ང་། གྲི་གུག་ཐོད་འཛིན་མཁའ་འགྲོ་མ། །བགེགས་དང་ལོག་འདྲེན་ལ་སོགས་བཅོམ། །།མཐོང་ན་དེ་བཞིན་བཟང་པོ་ཡིན། །དེ་ནས་ཅུང་ཟད་མི་མཐོང་ན། །དེ་ནི་འབྲིང་པོར་ཤེས་པར་བྱ། །དེ་ནས་བདུད་ཀྱི་རིགས་རྣམས་ནི། །ལྟོགས་དང་བཀྲེས་དང་ལུས་སྐམ་པ། ག་། ལག་།
