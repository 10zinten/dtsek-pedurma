འ་ཟླ་ཏི་ཤ་བསྟན་འགྱུར། ་རྒྱུད། འི༼་ཎས་རྒྱུ་ཁམ་ཎབཛྲ་བནྡྷེ་བནྡྷ་ཏིཥྛ་ཏིཥྛ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧུཾ་ཧཱུྃ་བཛྲ་ཀརྐོ་ཊ་ཀ་བཛྲ་པདྨོ་བཱ་མ་དཀྵི་:ཎ་ཀུཎྜ་ལ་ཡོཏིཥྛ་ཏིཥྛ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱ་བཛྲ་བཱ་སུ་ཀི་ཤི་ཁ་བཱ་ལ་ཧཱ་མ་དཀྵི་ཎ་རུ་ཙ་ཀ་ཡོཏིཥྛ་ཏིཥྛ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱ་བཛྲ་ཨ་ནནྟ་ཀུ་ལ་ཀོ་མེ་ཁ་ལ་ཀཎྜ་ཀ་ཡོ®ཏིཥྛ་ཏིཥྛ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོ་ཨཽ་ཧཱ་བཛྲ་ཏཀྵ་ཀོ་མ་ཧཱ་པདྨོ་བཱ་མ་དཀྵི་ཎ་པཱ་ད་ནུ་པུ་ར་ཡོཏིཥྛ་ཏིཥྛ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཅེས་པས་ང་རྒྱལ་གཟུང་བར་བྱའོ། །དབང་བསྐུར་བའི་《ཡན་ལག་སྟེ་བརྒྱད་པའོ། །སྤྱི་ཟླ་། །:ཀ་བའི་གླུའུ་ཏབ་པ། རྗེ་ཁ་ཊཱི་ཁ་ལ་བཏེག་ཏུ་། །སྔར་བཤད་ལྷ་མོ་བཅུ་གཉིས་ཀྱི། །སྔགས་བརྗོད་མཆོད་པ་དབུལ་་་་་བར་བྱ། །ཨོཾ་ཨཱཿཧཱུཾ་ཧཱིཕཊ། །སྔོན་དུ་མཆོད་ཡོན་དབུལ་བ་ནི། ཨོཾ་པྲ་བ་ར་་སཏྟ་རཾ་མ་ཧཱ་:ར་སྨིན་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། དེ་ནས་མེ་ཏོག་ལ་སོགས་དབུལ། །རིན་ཆེན་གསེར་གྱི་མེ་ཏོག་སྣ་ཚོགས་དྲི་མཆོག་སྤོས་སོགས་མར་མེ་དང་། །དྲིལ་བུ་མེ་ལོང་བླ་རེ་དང་ནི་སྣ་ཚོགས་འབྲས་བུ་བ་དན་གར་དང་རོལ་མོ་དང་། འོ། །སྣ་ཚོགས་མཆོད་པ་བྱས་ཏེ་གཞན་རྣམས་བཅུ་པོ་བདག་ནུས་ཇི་ལྟར་གསུངས་བཞིན་དབུལ། །བདུད་རྩི་སྔར་བཞིན་བྱིན་བཤད་པ། ཚོ། ཧཱུྃ་བྱུང་རྡོ་རྗེའི་ལྕེ་ཡིས་མྱང་། །།ཨོཾ་བུདྡྷ་ཡ་ན་མ། ཨོཾ་དྷརྨཱ་ཡ་ན་མ། ཨོ་སཾ་གྷ་ཡ་ན་མ། ཨོཾ་སྭ་བྷཱ་བི་ཀ་ཀཱ་ཡཱ་ཡ་ན་མ། ཨོཾ་དྷརྨ་ཝིཀཱ་ཡཱ་ཡ་ན་མ། ཨོཾ་སཾ་བྷོ་ག་ཀཱ་ཡཱ་ཡ་ན་མ། ཨོཾ་ནིརྨ་ཨིཎ་ཀཱ་ཡཱ་ཡ་ན་མ། ཨོ་ཙིཏྟ་བཛཱ་ཡ་ན་མ། །ཨོཾ་ཛྙཱ་ན་བཛཱ་ཡ་ན་མ། ཨོ་་་་བཱཀ་ཋབཛཱ་ཡ་ན་མ། :·ཨོ་ཀཱ་ཡ་བཛཱ་ཡ་ན་མ། ཨོ་ཤཱཀྱ་མུ་ན་ཡེ་ན་མ། །ཚེ་ཨོ་ཀཱ་ཚེལ་ཙཀཱ་ཡ་ན་མ། ཨོཾ་བཛྲ་ས་ཏྭཱ་ཡ་ཁྱེན་མ། དེ་བཞིན་ལྷ་ཡི་སྔགས་་་་རྣམས་ལ། །ན་མབཏགས་ལ་བདུད་རྩི་མྱང་། །སྲིན་ལག་དག་གིས་མཆོད་་་་པར་བྱ། །།དེ་ནས་སྙན་པའི་དབྱངས་ཀྱིས་བསྟོད། །གང་ལ་ཐོག་མཐའ་དབུས་།
