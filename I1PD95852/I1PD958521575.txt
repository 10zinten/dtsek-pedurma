ཁ་འུ་བསྟན་འགྱུར། རྒྱུད། མ་དག་མགོན་པོ་དེ་ལ་རྟག་ཏུ་བདག་གིས་བླ་མེད་བྱང་ཆུབ་ཆེན་པོར་སེམས་ནི་བསྐྱེད་་་་་བགྱིའོ། །ཀ་ཁ་ག་ང་འ་ཡང་། དབུས་དོན་དག་ལ་གྲོ་ནས་འབད་་་་ང་ཚོ་རྒྱ་གར་བ་ན་དག་དེ་ལྟར་བདག་གིས་རྒྱུན་ཆགས་བསྟོད་པ་བགྱིས་པ་ལས། །བསོད་ནམས་གང་དེས་དུས་མཐར་འགྲོ་བ་སྐྱེ་བ་ཡི། །འཁྲི་ཤིང་འདི་ནི་རྒྱལ་བའི་སྲས་ཀྱི་མེ་ཏོག་དང་། །བྱང་ཆུབ་ཅེས་བྱའི་འབྲས་བུར་འབྲས་དང་བཅས་གྱུར་ཅིག། དཔལ་དུས་ཀྱི་་་་་འཁོར་ལོའི་བསྟོད་པ་རྒྱུན་ཆགས་པ་ཞེས་བྱ་བ། ཀ་ལིངྐའི་རྒྱལ་པོའི་བླ་མ་སྡེ་སྣོད་་་་གསུམ་བཤད་པའི་མཁས་པ་ཆེན་པོ་ལེགས་པའི་གྲགས་པས་མཛད་པ་རྫོགས་སོ། །།།པ་ལ་རྒྱ་གར་གྱི་པཎྜི་ཏ་བི་ནཱ་ཡ་ཤྲཱི་མི་ཏྲ་དང་། པ་ལྡན་མ་ཚུལ་ཁྲིམས་སེང་གེས་ཆུ་མིག་རིང་མོར་བསྒྱུར་བའོ། །།།ཁབ་། །འ་དང་སྲིད་ཀྱི་ཀ་མ་ལ་ལ་རྩེ་ཁག་ཆེ་བསྒྲོན་གནད་མ་བ་བཞི་དང་། ག་ང་དེ་གཙོ་བོ་ཁག་རྒྱ་གར་དུ་དམ་པ་དག་ལ་སློབ་མ་དག་ལ་བདེ་བ་བྱེ་མ་ཤ་པ་ཁྲོ་ཁོ་སྐྱོ་བ་དཀའ་བ་དན་ལ་བློ་བཀལ་བ་ཁོ་ཚོའི་བྱ་བ་དང་། ཚེ་::ཀ་ཁ་ག་ང་གི་འདུ་ཁང་བ་ཏུ་མང་། མག་བཏེ་ད་བ་ས་གཙུག་ལབསྡུར་མཆན། འོ་ན་ཉེ་བ་ཡོད་ན་ཆེ་བའི་ཟླ་བ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»དཔལ་དུས་ཀྱི་འཁོར་ལོའི་བསྟོད་པ་རྒྱུན་ཆགས་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །མི«པེ་»«སྣར་»གཞན་ནི་གཞན། @«པེ་»+སུ། :«པེ་»«སྣར་»འགྱུ། མི«ཅོ་»རྒྱས། «པེ་»«སྣར་»སུམ། «ཅོ་»གུག། @«པེ་»«སྣར་»ས། «པེ་»«སྣར་དུང་། «པེ་»«སྣར་ཐོགས། «པེ་»«སྣར་»དེ་བཞིན། ཚེ«པེ་»«སྣར་»དུས་ཤིང་། གྱེ«པེ་»«སྣར་»སྤྱངས། «པེ་»«སྣར་»+ཕྱག་རྒྱ། @«པེ་»«སྣར་པ། ་ཉི«པེ་»«སྣར་བསྟན། «ཅོ་»བསྟོན། «པེ་»«སྣར་»ཡིན། #«པེ་»《སྣར་»ལྷ་དགྲ། མི«པེ་»«སྣར་པས་བརྒྱན་གྱུར། ༄༅«པེ་»«སྣར་སྦྱངས། ཁྱད་ཨེ«པེ་»«སྣར་»ས། «པེ་»«སྣར་»ཀྱི། མི«པེ་»«སྣར་»ན་རབ་གནས། «པེ་»«སྣར་»+དང་། འ«པེ་»«སྣར་པོ། «པེ་»«སྣར་»གོའུ། ®«པེ་»«སྣར་བྱུང་། @«པེ་»«སྣར་»སྤྱངས། «པེ་»«སྣར་»མོ་རབ། ཨི«པེ་»«སྣར་པོའོ། གྱི«པེ་»«སྣར་»སྤྱངས། «པེ་»«སྣར་»སྐྱེ་ངན་ལགས། ཉེ«པེ་»«སྣར་»ངོམས། «པེ་»བྱ།
