ཤུ་ཀ་རཱ་སྤྱི་ན་མཨོ་བྱ་སྨྲ་སྤྱི་ན་མgཨོཾ་ཛམྦྷ་ཀཱ་སྱེ་ན་མཨོ་ག་རུ་ཌ་གྱི་ན་མ་ཨོ་ཨུ་ལུ་ཀཱ་སྤྱི་ན་མཨོ་གྲི་ཞྭ་སྤྱི་ན་མཀ་ཨོ་ཀཱ་ཀཱ་སྤྱི་ན་མཨོ་སརྦ་བྷུ་ཏེ་བྷོ་ན་མ། ཞེས་པ་དུར་ཁྲོད་བརྒྱད་པོ་ྋརྣམས་ལའོ། །དེ་ནས་ཕྱི་རོལ་དུ་འཇིག་རྟེན་གྱི་ྋལྷ་རྣམས་ཀྱི། ཨོཾ་བཛྲ་ཙནྡྲ་ཡ་ན་མཿཨོཾ་བཛྲ་སཱུརྱ་ཝིཡ་་་་་མ། ཨོཾ་བཛྲ་མཾ་ཡིག་ལཱ་ཡ་ན་མཿཨོཾ་བཛྲ་བུདྷཱ་ཀིཡ་ན་མzཨོཾ་བཛྲ་བྲི་ཧསྤ་ཏ་ཡེ་ན་མཿཨོཾ་བཛྲ་ཤུ་ཀཱ་ཡ་ན་མཿཨོཾ་བཛྲ་:ཤནི༔་རཱ་ཝིཡ་ན་མཿཨོཾ་བཛྲ་ཀེ་ཏ་བེ་ན་མཨོ་བཛྲ་རཱ་ཧ་ཅབེ་ན་མ་ཨོཾ་བཛྲ་ཀཱ་ལཱགྷ་ཡེ་ན་མཿཨོཾ་བཛྲ་དྷ་ཏི་བེ་ན་མཿཨོཾ་:བཛྲ་གནྡྷཱ་ཝིཡ་ན་མ། ཨོཾ་སརྦ་ནཀྵ་ཏྲེ་སྦྱོ་ན་མ…ཨོཾ་དྭ་ཉིད་ཤ་རཱ་ཤི་བློ་ན་མ་ཨོ་ཏོ་ཌ་ཤ་ཀ་ལེ་སྦྱོ་ན་མ། ཨོཾ་ད་ཤ་:དི་ཀ་པཱ་ལེ་བློ་ན་མཨོ་བཛྲ་ནནྡྲི་ཝིཀེ་ཤཱ་རཱ་ཊིཡ་ན་མ་ཨོ་བཛྲ་མ་ཧཱ་ཀཱ་ལཱ་ཡ་ན་མཨོཾ་བཛྲ་:གྷནྡྷོ་ཀརྞ་ཁཡ་ན་མཿཨོཾ་བཛྲ་:བྷཾ་གི་ཙཎེ་ན་མཨོཾ་སརྦ་ཀྵེ་ཏྲ་པཱ་ལེ་ཧཱུཾ་ན་མཿཨོཾ་སརྦ་དྷུ་ཏཱི་བྷོ་ན་མཨོ་ཧཱ་ཙརི་ཏི་ན་མཿༀ་སརྦ་སིདྡྷི་བློ་ན་མ། ཨོཾ་དྷརྨ་ཙཀཱ་ཡ་ན་མཟླ་། ཨོ་བྷ་དྲ་གྷ་ཊཱ་ཡ་ན་མཿཨོ་དུརྗ་ཕྱི་ན་མཟླ་ཨོཾ་བོ་དྷི་བྲྀཀྵ་ཡ་ན་མཿཨོཾ་གུ་རུ་བུདྡྷཱ་བོ་དྷི་སཏྭ་བྷོ་ན་མོ་ན་མཞེས་པ་མཆོད་པའི་ཆོ་གའོ། །དེ་ནས་རིན་ཆེན་ཨིནྡྲ་ནཱི་ལ་ལ་སོགས་པ་རྣམས་དང་། གསེར་གྱི་མེ་ཏོག་རྣམས་དང་། །རྣམ་པ་མང་་པོའི་ཁ་དོག་ལྔའི་གོས་དང་དྲི་དང་བདུག་པ་དང་། མར་མེ་རྣམས་དང་དྲིལ་བུ་དང་མེ་ལོང་དང་བླ་རེ་རྣམས་དང་རྣམ་མང་འབྲས་བུ་རྣམས་དང་ཁ་དོག་ལྔའི་བ་དན་ས་བ་རྣམས་དང་གར་དང་རོལ་མོའི་སྒྲས་མཆོད་པ་བྱས་ཏེ། རྣམ་པ་བཅུ་ཡང་རྒྱུད་་་་གསུངས་པ་རྣམས་སོ། །དེ་ལྟར་སློབ་མ་བདག་ཉིད་ཀྱི་ྋནུས་པས་ཇི་ལྟར་གསུངས་་་པའི་མཆོད་པ་བྱས་ཏེ། དེ་ནས་སློབ་དཔོན་དག་གི་@ཞབས་ཀྱི་ཁརྩ་བར་མཎྜལ་བྱས་ནས་མཆོག་གི་བུ་ཡིས་ྋཡོན་ནི་དག་པའི་སླད་དུ་དབུལ་བར་བྱ། །དག་པ་གསུམ་གྱིས་།
