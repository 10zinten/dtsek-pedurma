བསྟན་འགྱུར། རྒྱུད། ༼པའིབཅུ་བ། ༼ཚེ་བ་ཏ་ལོ་ཁག་ཁ་གད་དུས་བདེ་བ་སྒྱུ་ཊི་ལ་ཁ་བ་འམ། མ་བ་བྱང་ལ་ཞུ་ཎཎ་ཊིཀ་ཀྱིང་ཀ་བློས་བྱ། །བལ་པོ་ཁ་ཆེ་བ་དང་ང་ཚོ་ཎི་ཀའི་ཁ་ལོ་བ། །བསྡུར་མཆན། ཁྲི་ཀའི་ཚ་བར་སྤྱི་ཟླ་ནང་ཁཚོམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་སྦྱོར་བའི་ཕྲེང་བ་བཞུགས་སོ། །༄༅«པེ་»ལོ«སྣར་»བྷུ་ཊ་མཱ་པ་ལ། ཨ«པེ་»«སྣར་པ། «པེ་»འཛིགས། «པེ་»«སྣར་»ལས། «པེ་»«སྣར་རིན་ཆེན། ཉི«ཅོ»སྦྱར། «པེ་»«སྣར་»འདུས། ་དེ«པེ་»«སྣར་ན། ཁ་ཞུང་རྟ་འཟློག། བརྒྱ་པའི་འཁོར་བ་མཁའ་འོ། །ཐ་ན་ཁ། སྐྱོ་%། ཚེ་༼བྱ་བྱ་Fpཚོ་ལས་། བསྟར་འབྲས་རེའུ་མིག་དུར། ད་པེ་ཅིན། རྒྱུད། ་དཔེ་གཞི། སྡེ་དསྣར་ཐང་། །ཅོ་ནེ། སྡེ་ཚན། རྒྱུད། རྒྱུད་འགྲེལ། བོད་རྟགས། ཚན་གྲངས། །ལྡེབ། བ་བ?བབ?བ་བ*བ་བཕྲེང་། །།།པ། ཚིག་རྐང་། ཚེག་ཁྱིམ། རར་ལྷ་མི་ལ་སྡུག་ངན་འགྲོ་ལས་པ་ཡི། གང་དག་པ་འོད་པས་གྲུབ་བར། །ད་པ་ནི་འདས་ད་ལྟ་ང་ཚོར་། །དང་ཚུལ་ཁྲིམས་ཀྱི་རྗེ་ཆེ། །ཐོག་མར་ནད་འབད་པ་ནི། །དེ་ལ་ཐོ་མས་པ་དང་། ད་བྱ། །འདི་ནི་རྒྱུད་རྣམས་་་མང་བ། མདོ་སྡེ་བ་ག་ནི་བྲམ་དུད་པས། དད་པ་བྱས་ན། གས་ས་་པོར་བ། དུ་བ། གདམ་པ་མི་:ག་པོ། །ན་བའི་སྤྲོས་པ་འདིས་།
