དྷརྨ་ན་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། །སྔགས་འདི་བཟླས་བརྗོད་བྱས་ནས་སླར་ཡང་དྲི་མ་་་་་སོགས་པས་མཆོད་པ་བྱས་ནས་སྔགས་འདི་བརྗོད་པར་བྱ་སྟེ། ༀ་ཧཱུྃ་ཀྲོ་དྷ་བཛྲ་སཏྠ་ས་ཏཱ་ཀ་རུ་ཎ་ཨཱཏྨ་ཀ་སརྦ་དྷརྨ་ཨེ་ཀ་སུ་ཁ་སྭ་བྷཱ་ཝ་སརྦ་དྷམ་བཛྲ་སུ་ཁ་སཱ་་་་བྷཱ་ཝཱན་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་བྱའོ། །དེ་ནས་དྲི་ལ་སོགས་པ་བྱིན་ནས་ལྷོ་་་ཕྱོགས་སུ་དགོད་པར་བྱའོ། །དེ་ནས་རྡོ་རྗེ་དྲིལ་བུ་ལ་མཆོད་པ་བྱས་ནས། །སྔགས་འདི་བརྗོད་པར་བྱ་སྟེ། ཨོ་ཧོ་བཛྲ་གྷཎྜེ་སརྦ་:དྷརྨ་ཨེ་ཀ་པྲཛྙཱ་དྷ་ནི་སྭ་བྷཱ་བེ་:སརྦ་ཀ་རེ་སརྦ་དྷརྨཱ་པྲ་བོ་དྷ་ནི་སརྦ་དྷརྨཱན་ནི་སྭཱ་བྷཱ་བེན་པྲ་བོ་དྷ་ཡ་སྭཱ་ཧཱ། ཞེས་བྱའོ། །དེ་ལྟར་བྱིན་གྱིས་བརླབས་ནས་བྱང་ཕྱོགས་སུ་དགོད་པར་བྱའོ། ་:། དེ་ནས་མཆོད་པ་ལ་སོགས་པ་དང་སླར་ཡང་གཏོར་མ་སྦྱིན་པར་བྱ། །དེ་ལ་རྒྱས་པའི་ཆོ་ག་ནི། འགྲེལ་པ་དྲི་མ་མེད་པ་ལས་ཤེས་པར་བྱའོ། །དེ་ལྟར་སྲུང་ཚིབ་ལ་སོགས་ནི། །བྱང་ནས་ཐིག་སྐུད་གདབ་པར་བྱ། །སློབ་དཔོན་དང་ནི་སློབ་མ་ཡིས། ཞ། ཤར་དང་ནུབ་ཏུ་མངོན་ཕྱོགས་སུ། །ལག་པ་གཡོན་དང་གཡས་པ་ཡིས། །ཐིག་སྐུད་བཟུང་ནས་སྔགས་འདི་བརྗོད། །ཨོཾ་ཨཱིཧཱུྃ་ཀཱ་ཡ་ཝ་ཅིཀ་ཙིཏྟ་ཏི་ཀ་བྷུ་ཏཿསརྦ་དྷཱརྨ་ཨེ་ཀཱ་ཀ་ྋརེ་ཎ་བཛྲ་སརྦ་སཏྭ་ཧཾ་བཛྲ་བྷུ་མྱི་སཱུ་ཏྲ་ཡཱ་མི་:ཧཱུྃ་ཨཱ*ཕཊ། ཅེས་བྱ་བ་ནི། ཐིག་སྐུད་རྒྱས་པར་བྱས་ནས་སྔགས་འདི་བརྗོད་པར་བྱའོ། །དེ་ནས་བཛྲ་སཱུ་ཨིཏཱ་ཨེ་ཀཱ་ཀཱ་ར་ས་རཱུ་པེ་ན་ཛgཛgཛསརྦ་དྷརྨཱན་སཱུ་ཏྲ་ཡ་ཨོ་ཨཱཧུཾ་ཧོཧཾ་ཀྵཕཊ། ཅེས་བྱ་བ་ནི་ཐིག་སྐུད་གདབ་པའི་དུས་སུ་བརྗོད་པར་བྱའོ། །དེ་ནས་སློབ་དཔོན་གྱིས་་་་གཡོན་ཕྱོགས་སུ་སྐྱིལ་མོ་ཀྲུང་ཕྱེད་བྱས་ལ། །རྐང་པ་གཡས་པ་ས་ལ་བཞག་པའི་་་་འདུག་སྟངས་དང་། སློབ་མ་དེ་ལས་བཟློག་པའི་:རིམ་པའི་འདུག་སྟངས་ཀྱིས་་་་ཚངས་པའི་ཐིག་གདབ་པར་བྱའོ། །།དེ་བཞིན་དུ་སློབ་དཔོན་ལྷོ་ནས་མངོན་པར་ཕྱོགས་པ་དང་། སློབ་མ་བྱང་ནས་ལྷོ་རུ་མངོན་པར་ཕྱོགས་པས་ཐིག་གདབ་པར་།
