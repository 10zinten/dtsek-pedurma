རྡོ་རྗེའི་ཚིག་འབྱེད་པ་ཞེས་བྱ་བ། སོགས་པས་ཡོངས་སུ་བཏུལ་བ་མ་བྱས་པའི་དུག་དང་གསེར་དང་དངུལ་གྱི་བཟའ་བཏུང་ནི་སྡུག་བསྔལ་གྱི་ཕུང་པོ་ཆེན་པོ་བསྐྱེད་པར་འགྱུར་བ་བཞིན་ནོ། །དེ་བས་ན་བྱིས་པའི་རང་བཞིན་ཅན་རྣམས་ལ་ནི་བཀག་ལ། །རྣལ་འབྱོར་པ་འགྱུར་བ་མེད་པ་ཐོབ་པ། ཆོས་ཐམས་ཅད་ནམ་མཁའི་དཀྱིལ་དང་མཚུངས་པར་མཐོང་བ་རྣམས་ལ་གནང་བ་ཁོ་ན་ཡིན་ནོ། །དེ་ལྟར་ལས་དང་པོ་པ་རྣམས་བརྟུལ་ཞུགས་ཉི་ཤུ་རྩ་ལྔ་བྱིན་པའི་རྗེས་ལ། བྱིན་པའི་རིགས་ཡོངས་སུ་བརྟག་པའི་དོན་དུ། དཔལ་ལྡན་སྔགས་:ཀྱིས་མངོན་པར་བསྔགས་པའི་པདྨའི་ནང་དུ་མེ་ཏོག་གཅིག་བྱིན་ཏེ། །ཞེས་བྱ་བ་སྐུ་གསུང་ཐུགས་ཀྱིས་བསྔགས་པའི་མེ་ཏོག་ལག་ཏུ་སློབ་དཔོན་གྱིས་སྦྱིན་ནོ། །དེ་ཡང་དཀྱིལ་འཁོར་དུ་ཞུགས་པའི་དོན་ཡིན་གྱི་%གཞན་ནི་མ་ཡིན་ནོ། །དང་་་་པོར་ས་ལན་གསུམ་བསྐོར་ལ་ཞེས་བྱ་བ་ནི་སློབ་མ་ལས་ཀྱི་རྡོ་རྗེ་དང་ལྡན་པས་སོ། །ལག་པ་པདྨ་ནས་ནི་མེ་ཏོག་དཀྱིལ་འཁོར་ལ་དོར་བྱ། །ཞེས་བྱ་བ་ནི་དཀྱིལ་འཁོར་ལ་བསྐོར་བ་ལན་གསུམ་བྱས་པའི་རྗེས་ལ། རིགས་ཀྱི་མེ་ཏོག་ཚོམ་བུ་འམ། བུམ་པ་ལ་སོགས་པ་ལ་དོར་བར་བྱའི། རྡུལ་ཚོན་ལ་ནི་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །གང་གི་གནས་སུ་མེ་ཏོག་ལྷུང་གྱུར་དེ་ནི་དེ་ཡི་རིགས་ངེས་མི་ཡི་བདག་པོ་ཀྱེ། །ཞེས་བྱ་བ་ནི། ཀློཕྱི་ནས་མེ་ཏོག་བབས་པའི་རིགས་ཀྱི་ལྷ་དང་རྗེས་སུ་མཐུན་པའི་ལས་ཀྱི་ཕྱག་རྒྱ་བཟུང་ན་ལས་ཀྱི་ཕྱག་རྒྱའི་དངོས་གྲུབ་ཐོབ་ཅེས་བྱ་བའི་དོན་ཏོ། །དེ་ནས་རིགས་ཀྱི་ལྷ་ལ་སོགས་པ་ཡང་བསྟན་པར་བྱའོ། །ཕྱི་ནས་དབང་བདུན་དང་ནི་རྣམ་གསུམ་བླ་མེད་ཆེན་པོ་འདིར་ནི་སྦྱིན་པར་བྱ། །ཞེས་བྱ་བ་ནི། སློབ་དཔོན་གྱི་ལས་ཀྱི་གོ་རིམས་ཡིན་ཏེ། དེ་ཡང་སློབ་དཔོན་གྱིས་ཀྱི་དཀྱིལ་འཁོར་གྱི་ཤར་གྱི་སྒོར་དཀྱིལ་འཁོར་གྱི་གཙོ་བོའི་ང་རྒྱལ་དང་ལྡན་པས། ལྷ་རྣམས་གཡས་བརྐྱང་པ་དང་། གཡོན་བསྐུམ་ཀྱི་བཞུགས་པ་དང༌། བཞེངས་བའི་རྗེས་སུ་མཐུན་པར་གནས་ཏེ། །།
