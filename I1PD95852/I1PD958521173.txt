བསྟན་འགྱུར། རྒྱུད། ཚེཁྱུ་མར་གའོ། །བཛྲ་པི་བི། བཛྲ་ནི་རི་ཏྭཱ་ཡ། བཛྲ་ཨག་ཡེ། བཛེད་དྷ་ཡེ། ::ཀ་བ། བཛེ་ནྡྷ་ཡ། བཛེ་ཤཱ་རཱ་ཡ། བཛྲ་བྲཧྨ་ཎེ། །བཛྲ་བི་ནཱ་ཡ་ཀཱ་ཡ། བཛྲ་ཀཱིརྟི་ཀེ་ཡཱ་ཡེ། བཛྲ་བཱ་ཡ་བིཡོ། བཛྲ་ཡ་མཱ་ཡ། བཛྲ་ཡཀྵ་བྷོ། བ་རུ་སྐྱེ། བཱ་ཡ་བྱི། ཡཱ་མི་:ཙུ། ཡཱ་ཀྱི་ྋསྐྱེ། མ་ཧཱ་པ་ལཱ་ཡ། ཨ་ཙ་ལཱ་ཡ། ཊཀྐི་རཱ་ཛཱ་ཡ། ནཱི་ལ་དཎཱ་ཡ། སུམྦྷ་རཱ་ཛཱ་ཡ། རོ་:དཱཀྵི་ཡེ། བཛྲ་ཤྲཱི་ཁེ་ལཱ་ཡི། །ཙུན་ཡིའི། །བྷཾ་ཀུ་བྱི། མཱ་རི་ཙི། པྲ་ཏྲེ་ཀ་མཱ་བྷེ་དེ་ན་:ཥཌི་ཨུཏྟ་ར་ྋཏཱི་ཤ་ཏ་བཛྲ་ཏི་ཐི་དེ་བཱི་:བྷུ། ཥཌ་ཏྲི་ཤད་པྲེཏཱིཙྪ་ཧཱུཾ། བཛྲ་ཛ་ཡཱ་ཡ། བཛྲ་ཀརྐ་ཊ་ཀཱ་ཡ། བཛྲ་བཱ་སུ་ཀ་ཡེ། བཛྲ་ཨ་ནནྟཱ་ཡ། བཛྲ་ཏཀྵ་ཀཱ་ཡ། བཛྲ་མ་ཧཱ་པདྨ་ཡ། བཛྲ་ཀུ་ལི་ཀཱ་ཡ། བཛྲ་ཤཾ་ཁ་པཱ་ལཱ་ཡ། བཛྲ་པདྨ་ཡ། བཛྲ་བི་ཛ་ཡཱ་ཡ། ཞེས་བྱ་བ་ནི་སྐུའི་དཀྱིལ་འཁོར་ལ་མཆོད་པའོ། །ཤཱ་ནཱ་སཱུ་ཡི། སུ་ཀཱ་ར་སྩ་ཡི། །བྱཱ་གྷཱ་སཱ་ཡི། །ཛཾ་བུ་ཀཱཉྩིསཱ་ཡི། ག་རུ་ཌཱ་སྩ་ཡི། །ཨུ་ལུ་ཀཱ་ྋས་ཡི། གྲོ་བྷ་ྋསྩ་ཡི། །ཀཱ་ཀཱ་སཱུ་ཡི། སརྦ་བྷུ་ཏེ་བྷོ། ::ཞེས་བྱ་བ་འདི་ནི། དུར་ཁྲོད་བརྒྱད་རྣམས་ལ་མཆོད་པའོ། །བཛྲ་ཙནྡཱ་ཡ། བཛྲ་སཱུརྱ་ཡ། བཛྲ་མཾ་ག་ལཱ་ཡ། བཛྲ་བུདྡྷ་ཡ། བཛྲ་དྲྀ་ཧ་སྤ་ཏ་ཡེ། བཛྲ་ཤུ་ཀཱ་ཡ། བཛྲ་ས་ནི་ྋཧྥ་རཱ་ཡ། ་བཛྲ་ཀེ་ཏ་པེ། བཛྲ་རཱ་ཧ་བེ། བཛྲ་ཀཱ་ལ་ཨགྷ་ཡེ། བཛྲ་དྷུ་བེ། བཛཱ་ག་སྐྱ་ཡ། སརྦ་ན་ཀླུ་ཏྲེ་བྷོ། །དྭཱ་ད་ཤ་རཱ་སི་བྷོ། ཁོཌ་ཤ་ཀཱ་ལཱ་བྷོ། ད་ཤ་:དིཀ་པཱ་ལེ་བྷོ། ནན་དི་ཀེ་ཤྭརཱ་ཡ། །བཛྲ་མ་ཧཱ་ཀཱ་ལཱ་ཡ། བཛྲ་གྷཎྜ་ཨེཀརྩྭ་ཡ། བཛྲ་ཧཱུཾ་གི་ནི། སརྦ་ཀྵེ་ཏྲ་པཱ་ལེ་བྷོ། སརྦ་དྷུ་ཏི་བྷོ། །ཧཱ་རཱི་ཏི། སརྦ་སིདྡྷི་བྷོ། ་་:དྷརྨ་ཙཀཱ་ཡ། བྷ་དྲ་ག་ཌཱ་ཡ། བྷ་བཛྲ་དུནྡྲ་བྷི། བོ་:དྷིབྲྀཀྵ་ཙིཡ། །གུ་རུ་བུདྡྷ་བོ་དྷི་སཏྭ་བྷོ་ན་མོ་ན་མཞེས་བྱའོ། །དེ་ལྟར་མཆོད་པའི་དྲིལ་བུ་དཀྲོལ་ཞིང་བསྟོད་པ་བྱའོ། ཁྱི། །ཤེས་རབ་ཀྱི་མེ་ཏོག་དང་བཅས་པའི་རང་གི་བྱང་ཆུབ་ཀྱི་སེམས་པདྨའི་སྣོད་དུ་བླུགས་སེལ་བདུད་རྩི་རུ་བྱས་པ་།
