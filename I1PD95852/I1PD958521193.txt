བསྟན་འགྱུར། རྒྱུད། ཁས་སོགས་པ་ལ་མེ་ཏོག་བྱིན་ནས། །་དེ་ནས་མིང་གི་དབང་བསྐུར་བ་ནི་གསོལ་བ་་་་གདབ་པ་དང་། ལས་ཀྱི་དབང་པོ་དང་། དེའི་ཡུལ་རྣམས་ཁྲོ་བོ་དང་ཁྲོ་མོར་་་བསྐྱེད་པས། ཨོཾ་ཧ་ཧཱ་ཡ་ཡཱ་:ར་རཱ་ཝ་ཝ་ལ་ལཱ་ཙ་ཏུར་བྲཧྨ་བི་ཧཱ་ར་བི་ཤུདྡྷ་སྭཱ་ཧཱ། ཞེས་བྱ་བའི་སྔགས་ཀྱིས་གནས་ལྔར་དབང་བསྐུར་ོལ། དེ་ནས་ལག་པ་དང་རྐང་པ་རྣམས་ལ་ལག་གདུབ་བམ། ཆ། ཝཾམེ་ཏོག་གི་ཕྲེང་བ་བཅིངས་ལ། །སྲིད་པའི་ངན་འགྲོ་ལས་བཏོན་ནས། །སྲིད་པ་མཆོག་ཏུ་ཞི་བྱའི་ཕྱིར། །རྡོ་རྗེ་སེམས་དཔའི་དེ་བཞིན་གཤེགས། །བདག་གིས་ཁྱོད་ནི་ལུང་བསྟན་བགྱི། །སརྦ་ཏ་ཐཱ་ག་ཏ་སིདྡྷཱ་ས་མ་ཡ་ཧཱུྃ་:བྷུ་བྷུ་ཝ་སཱཧི། ཞེས་བརྗོད་པ་ནི་མེ་ཏོག་ལྷུང་བའི་སྒོ་ནས་རང་གི་རིགས་ཀྱི་མིང་གིས་ལུང་བསྟན་པ་ནི། པ་:ར་ཨཱརྱ་དང་། ཨ་ནུ་ས་མ་དང་། གས་སུ་ར་ཏ་དང་། །སྐྱེས་མ་ར་ས་དང་། བི་ཛ་ཡ་དང་། ཨ་ལ་ལ་དང་། ས་ཏཱ་དང་། མ་ཧཱ་སུ་ཁ་དང་། ཨ་ནཱི་ག་དང་། ཛྙཱ་ན་དང་། རྗེས་ཧཱ་ཛ་བཛྲ་ཞེས་པ་རྡོ་རྗེ་སེམས་དཔའི་མིང་གི་བྱེ་བྲག་སྟེ། །༄༅མཐར་རྡོ་རྗེ་ཀུན་ལ་གོ་བར་བྱའོ། །མི་བསྐྱོད་པའི་ཧཱ་ས་དང་། བི་ལ་ས་དང་། ར་ཏི་དང་། ལི་ལཱ་དང༌། ལ་ལི་ཏ་དང་། སྭ་ཡཾ་བྷུ་དང་། །དྭེ་ཥ་དང་། ཧེ་དང་། ཀ་རུ་ཎ་དང་། ཙིཏྟ་དང་། ཧཱུྃ་ཀ་ར་དང་། ་ཨཀྵ་བྷུ་བཛྲ། ཞེས་པ་རྣམས་སོ། །དོན་ཡོད་གྲུབ་པའི། །ཨ་མོ་གྷ་དང་། བ་ར་མཱ་ཤྭ་དང་། ས་མ་ཡ་དང་། པཱ་ར་མི་ཏ་དང་། ་ཀརྨ་དང་། ཨ་ནི་ལ་དང་། སྤྱན་དང་། ཨནྟའོ། ཞེས་པ་རྣམས་སོ། །རིན་ཆེན་འབྱུང་ལྡན་གྱི། རནྡ་དང་། ནཱ་མ་ཎི་དང་། ཁ་ཁ་དང་། ག་ག་ན་དང་། ཏེ་ཛོ་དང་། མཱརྟ་ཎ་ཌ་དང་། བྷས་ཀ་ར་དང་། མཱ་མ་ཞེས་པ་རྣམས་སོ། །རྣམ་པར་སྣང་མཛད་ཀྱི། བི་རོ་ཙ་ན་དང་། ཨོཾ་ཀཱ་ར་དང་། ཤ་ཤཱ་ཏ་དང་། ཏ་ཐཱ་ག་ཏ་དང༌། མཱ་ཡཱ་དང་། བུདྡྷ་དང་། རཱུ་པ་དང༌། ཀཱ་།
