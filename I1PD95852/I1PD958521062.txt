ཨོ་ཏོ་ཌ་ཤ་ཀ་ལཱ་བྷོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ད་ཤ་དིཀ་པཱ་ལེ་གྷོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་་་་བཛྲ་ནནྡི་ཀེ་ཤྭ་རཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མ་ཧཱ་ཀཱ་ལཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་་་བཛྲ་གྷཎྚ་ཀཙྪཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཧཱུཾ་གི་ནེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཀྵེ་ཏྲ་པཱ་ལེ་སྦྱོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་དུ་ཏི་སྦྱོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ཧཱ་རཱི་ཎི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ༑ཨོཾ་སརྦ་སིདྡྷི་བྷོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་དརྨ་ཙཀཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བྷ་་དྲ་གྷ་ཊཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དུན་དུ་བྷོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བོ་དྷི་བྲཁཱ་ཡ་ཧུཾ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་གུ་རུ་བུདྡྷ་བོ་དྷི་ས་ཏྭེ་བྷྱ་ན་མོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ཉ་བཛྲ་ཉི་བཛྲ་ཤྲཱི་བཛྲ་ཧུ་བཛྲ་ཤྲཱི་བཛྲ་ཧཾ་བཛྲ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཛྷ་བཛྲ་ཛཱི་བཛྲ་ཛྙཱ་བཛྲ་ཛྙཱ་བཛྲ་ཛྙཱ་བཛྲ་ཛྷ་བཛྲ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ་ཏཨོཾ་ཛ་བཛ་ཛི་བཛྲ་ཛཾ་བཛྲ་ཛུ་བཛྲ་ཛཱི་བཛྲ་ཛོ་བཛྲ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ཚ་བཛྲ་ཚི་བཛྲ་ཙཱ་བཛྲ་ཚུ་བཛྲ་ཟླ་བཛྲ་ཚོ་བཛྲ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ཙ་བཛྲ་ཙི་བཛྲ་ཙཱ་བཛྲ་ཙུ་བཛྲ་ཙྪ་བཛྲ་ཙི་བཛྲ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། དེ་བཞིན་དུ་སྔར་བསྟན་པའི་ས་བོན་ཀུན་ལ་ཡང་སྦྱར་བར་བྱའོ། །ཨོ་པ་རུ་སྐྱེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བ་ཡ་བེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་:ཡ་མ་རྩེ་°ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཡཀྵ་རྒྱ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་ནཱི་ལ་དཎྜཱི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཊཀྐི་རཱ་ཛཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨ་:ཙ་ལཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་མ་ཧཱ་བ་ལཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སུཾ་བྷ་རཱ་ཛ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་བཛྲ་རོ་དཀླུ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཌ་ཨོ་བཛྲ་ཨཾ་ཁ་ལཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཙུནྡེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བྷེ་ཀི་རི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་:མཱ་རཱི་ཙྪི་ྋཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤཏ་ཏྲི་ཨིཙྪ་བྷོ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཛ་ལཱ་ཝིཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོ་་་བཛྲ་ཀརྐ་ཊཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བཱ་སུ་ཀི་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨ་ནནྟ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། gཨོ་བཛྲ་ཏཀྵ་ཀཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མ་ཧཱ་:པདྨ་ཡ་ྋཧཱུྃ་་ཕཊ་སྭཱ་ཧཱ། །ཆཨོ་བཛྲ་ཀུ་:ལི་ཀཱ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤཾ་ཁ་པཱ་ལཱ་ཡ་ཧཱུྃ་ཕཊ་c།
