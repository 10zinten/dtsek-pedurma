མས་ཆུ་ཚོད་བསྒྱུར། །དེ་ལ་མཁའ་དུས་བགོས་ཐོབ་པ། །ཆུ་ཚོད་གནས་སུ་གཞག་@པར་བྱ། །ཆུ་སྲང་དབུགས་དག་རིམ་པས་ཤེས། །དེ་ལྟར་ཐོབ་པ་དེ་རྣམས་་་ཀྱིས། །རྐང་བཟུང་ྋསྔོན་སྡོམ་ཕྱི་ཕྲི་ལ། །དལ་བ་གཅིག་ལ་སྦྱིན་དོར་ནི། །མིག་དམར་མྱུར་བ་ཇི་བཞིན་ནོ། །ཡིད་བཞིན་ནོར་:བུའི་རྩིས་འདི་ནི། །ཟབ་ཅིང་རྒྱ་ཆེ་དུས་འཁོར་རྒྱུད། །རིན་ཆེན་གླིང་མཆོག་ནང་ནས་ནི། །བླ་མའི་མན་ངག་ལ་་་བརྟེན་ནས། །ཤེས་རབ་མིག་གིས་ལེགས་དཔྱད་བྱས། །དྲན་ཤེས་བརྩོན་ལྡན་་་གུས་པས་བླངས། །དོན་ལྡན་ཚིག་གིས་སྲད་བུར་བརྒྱུས། །བློ་ལྡན་རྣམས་ཀྱི་་་མགུལ་རྒྱན་མཛོད། །དེ་ལས་བྱུང་བའི་བསོད་ནམས་ནི། །དྲི་མེད་ཉི་ཟླ་ལྟ་བུ་་དེས། །འགྲོ་བ་སེམས་ཅན་མ་ལུས་རྣམས། །ཀུན་མཁྱེན་རྒྱལ་པོ་ཐོབ་པར་ཤོག། གཟའ་ལྔ་སོ་སོའི་རྐང་རྩིས་ཀྱི་མན་ངག་། པཎྜི་ཏ་ཆེན་པོ་ཤཱཀྱ་ཤྲཱི་བྷ་དྲས་མཛད་་པ་རྫོགས་སོ། །།།་་དུ་བཙན། བ་བ་བལ་པ་ལ། ན་བ་དང་འདས་ནས་མ་འཁྲུ་བ་དང་བདེ་ཐང་། གྲོལ་གི་འབད་བའི་བབས་སུ་ག་བ་ཀུ་བསྡུར་མཆན། མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་གཟའ་ལྔ་སོ་སོའི་རྩིས་ཀྱི་མན་ངག་ཅེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»སྒྱུར། «པེ་»«སྣར་ཡིས་སོ། «པེ་»«སྣར་སྲེ། «པེ་»«སྣར་»«ཅོ་»ཡིན། «ཅོ་»ཡིན། «པེ་»«སྣར་»བྱ། ཉི«པེ་»«སྣར་བཞག«པེ་»«སྣར་བཞག«པེ་»«སྣར་»བཞག«པེ་»«སྣར་གཅིག«པེ་»«སྣར་»རྫུན་མ་ཡིས། «ཅོ་»སྔོན་མ་ཡིན། «པེ་»«སྣར་»བྱ་བཞག«པེ་»དྲུགམ«པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་བཞག་ན། «པེ་»«སྣར་»སྡུན། «པེ་པར། «པེ་»ལས། «པེ་»«སྣར་བཞག@«པེ་»«སྣར་བཞགཕྱི«པེ་»«སྣར་»སྡུན། ཨ«པེ་»«སྣར་»བཞག«པེ་»«སྣར་སྔོན། «པེ་»«སྣར་»སྒྱུར། «པེ་»«སྣར་བཞག«པེ་»«སྣར་ཟློག།
