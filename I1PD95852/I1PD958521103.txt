བསྟན་འགྱུར། རྒྱུད། །དྷ་དྷཾ་དྷཾ། ད་དི་ཛཾ་དུ་ཛྲ་དཾ། ཐ་ཐི་༔་ཐུ་བློ་ཐཾ། །ཏ་ཏི་ཎཱ་ཏུ་ཀྲྀ་ཏཾ། ་རྣམས་དང་རྡོ་རྗེ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་་་་གནས། །དེ་ལྟར་སྟོན་ཟླ་འབྲིང་པོའི་ཚེས་རྣམས་དང་ནི་:གཡོན་ན་རྣམ་པར་་དག་པའི་རང་བཞིན་ཅན། །འདི་རྣམས་དང་བཅས་ནུབ་སྒོའི་གཡས་སུ་སྣ་ཐག་ཅན་དང་ཆུ་སྐྱེས་དམར་པོའི་སྟེང་གནས་པ། །ཏཾ་ཡིག་དང་ནི་རྡོ་རྗེ་དག་ལས་་་བྱུང་བའི་དབང་པོ་སེར་པོ་ཕྱག་ན་རྡོ་རྗེ་དང་། ཁ། མདའ་དང་དྲིལ་བུ་གཞུ་རྣམས་འཛིན་པ་ཙ་རྩི་ཀཱ་དང་བཅས་པ་དེ་ལ་ཕྱག་འཚལ་ལོ། །རྐ་རྨི་ཚྭ་རྐུ་ཌོ་རྐོའི། །ཤི་ཤི་ཐི་ཤུ་ཤྲཱི་ཧཱུྃ། ཥ་ཎི་ཛཱི་ཥཱ་བློ་ཥ། སྐཧྥ་ཁི་ཧྥེ་གླུ་ཤྲཱི་ཧྥ། ཏི་ས་སི་སྔོ་སུ་སྶཾ་སཾ། རྣམས་དང་རྩེ་གསུམ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དགུན་ཟླ་ར་བའི་ཚེས་རྣམས་དང་ནི་ལོང་བུའི་ཚིགས་ཀྱི་རང་བཞིན་ཅན་དེ་འདི་རྣམས་དང་། ཀ། བཅས་པས་བྱང་སྒོའི་གཡོན་དུ་ཁྱུ་མཆོག་་་་དང་ནི་ཆུ་སྐྱེས་དམར་པོའི་སྟེང་དུ་ཡང་དག་གནས། །ཁ་། སཾ་ཡིག་དང་ནི་:རྩེ་་་་གསུམ་ལས་བྱུང་འཕྲོག་བྱེད་སྔོན་པོ་བརྩེ་གསུམ་མདའ་དང་ཁ་ཊཾ་ག། གཞུ་རྣམས་ཕྱག་གིས་འཛིན་དང་གཤིན་རྗེ་དང་མཚུངས་རི་ཡི་བུ་མོ་པང་གནས་ལ་ཕྱག་འཚལ། །སཱ་སཱི་ཙཱ་སཱུ་དྷྭཾ་སཱ། ལྷ་ཤཱི་ཝཱ་ཛྙཱ་ཝ། ཥ་ཊཱི་ཙཱ་ཏུ་ཛྙཱ་ཥཱ། ཤཱ་ཤྲཱི་ཛཱི་ཤུ་ཤྲཱི་ཤཱ༑ཎཱ་ཎཱི་ཎཾ་ཏྲུ་ཌཱ་ཎཱ། རྣམས་དང་དབྱུག་པ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དེ་ལྟར་དགུན་ཟླ་འབྲིང་པོའི་ཚེས་རྣམས་དང་ནི་རྐང་གཡས་ལོང་མོའི་ཚིགས་ཀྱི་རང་བཞིན་ཏེ། །དེ་རྣམས་་་་་དང་བཅས་ནུབ་སྒོའི་གཡོན་དུ་མིང་དང་ྋཉི་མའི་མདོག་ལྡན་སྟེང་གནས་པ། །ཆ་ཡིག་དཔལ་གྱི་དབྱུག་པ་དག་ལས་བྱུང་བའི་གནོད་སྦྱིན་ནོར་བུ་དབྱུག་པ་སྲེ་མོ་དང་། །པདྨ་རྣམས་ནི་ཕྱག་ན་འཛིན་པའི་ཁྱབ་འཇུག་དང་མཚུངས་ནོར་་་་་།
