སྦྱོར་བ་ཡན་ལག་དྲུག་པ། ལ་སྤྱི་བོ་རྗེ་དབང་གིས་དངོས་སུ་མཐོ་བ་མ། དེ་མདས་པ། དེ་ལ་ཡེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་སོ། །དེ་ནས་ལ་བྱོན་ན་འབངསལ་བར་གྱུར་གང་འདབསྡུར་འབྲས་རེའུ་མིགཛིན་་འང་ས་::ཀ་ཆེན་པོ་བླ་མ་དེའི་ཁྲོད་མ་རྒྱུད། （བ་ལ་ལོག་པར་མགུ་བར་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། བོད་རྟགས། །ལྡེབ། བ་ན་ན་ན་བ*ན་བན་ཕྲེང་། །ཚེག་ཁྱིམ། །།།།་ང་ད་ན་རྒྱུ་བོན་དང་སྒྲུབ་ཀྱི་གནས་ཚད་ཤིན་ཏུ་ཁག་པ་དང་རྒྱུད་འགྲེལ། ་ཚན་གྲངས། འདeཏུ་བpན་པ་དPPཁཀི་ཏྟ་བའི་ཉན་ཁ་བ་བོད་པ་ཀོ་ཀྱི་བློ་ཁ་ཕྱི་ལ་སྤྱོད་པ་རུ་ལ་སྐྱེ་བོ་མ་དད་པ་ད་ལྟ་བྱ་ལྟ་བ་ལྟ་བཅུ་བའི་ཁྲི་པ་ཏི་ལྔ་པོ་དེ་ཚོ་སྤྱི་ལ་སྨནྟོ་་་་་དེ་ག་ཏེན་ཥཊི་ལ་ཀོ་ཊའོ། །ཁོ་དར་བ་རྟ་ཁ་དགག་པ་བྱེད་པ་བདག་བ་nཀ་བ་བརྒྱ་པ་ད་ང་ད་ཡི་ཁ་ལོ་བ་སོགས་ཆེ་བ་དང་ཁོ་པ་བ་ཚོར་བ། །བློ་ཁ་བོ་ཆེ་ཚེ་དམན་ཀpཁ་དེབ་ལ་སྐྱོ་ཁཌ་་དེའི་་བའི། །།
