བ་ཁོ་ན་བཤད་པ་ཡིན་ནོ། །སྐུ་ཞེས་པ་ལ་སོགས་པ་རྣམས་ཀྱིས་ནི་དཀྱིལ་འཁོར་་་གསུམ་གསུངས་སོ། །གསང་བའི་དབང་པོ་ནི་གསུང་རྡོ་རྗེའོ། །ཟླ་བ་ཞེས་པ་ནི། ཟླ་བ་བཟང་པོ་ལ་བོད་པའོ། །དབང་མདོར་བསྟན་པའི་མདོར་བཤད་པ། ལ་། །བདག་གིས་བྲིས་ལས་དགེ་བ་གང་། །ཐོབ་དེས་རྡོ་རྗེ་སེམས་དཔའ་ཡི། །གོ་་་་འཕང་སྐྱེ་བོས་ཐོབ་པར་ཤོགR། སཱ་ཧུ་པུ་ཏྲ་དཔལ་འཛིན་དགའ་བས་མཛད་པའི་དབང་མདོར་བསྟན་པའི་མདོར་བཤད་རྫོགས་སོ། །་། གས་པ་ལ་རྒྱ་ཆེ་བ་ས་སྐpཚེཔོརྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ་གནས་བརྟན་བྱང་ཆུབ་སེམས་དཔའ་་་ནགས་ཀྱི་རིན་ཆེན་གྱི་ཞལ་སྔ་ནས་དང་། ལོ་ཙཱ་བའི་ཆ་ཤས་སུ་གྱུར་པ་གཞོན་ནུ་་དཔལ་གྱིས། ཆོས་འཁོར་དཔལ་བསམ་ཡས་ཀྱི་དབེན་གནས་དམ་པ་མཆིམས་ཕུའི་དབེན་གནས་སུ་བསྒྱུར་བའོ། །ལོ། །%ཞི་ཏེད་པའིརིམ། །ཁོ་བའི་ཀ་ཁའི་ན་གསོ་བའི་ཁྲོ་ལ་དཔེynnཚོ་ཎཁng@་ལེ་ལེ་ཞི་ཚོrjepའིནས་ལྔ་པའི་ཚེ་ན་དེ་ལRཊིཀ་ཁོ་ཚེ་ཁོ་ལ་ཁོརྟ་ལ་ཆགས་པ་དེས་བྱ་རྟོg་ཡོnwns་ཀྱི་ན་གསལ་ཏJ། ཏཔོ། བམ་ཀ་སྐུ#བསྡུར་མཆན། ག་ཨིཊྚ་སྐལྔ་ལ་གཝེཊ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»ལེགས་པའི་བུ་དཔལ་འཛིན་དགའ་བས་་་་་མཛད་པའི་དབང་མདོར་བསྟན་པའི་བཤད་པ་བཞུགས་སོ། །«པེ་»གིས། «པེ་»སོགས། «སྣར་»ལ་སོགས། /«སྣར་»ནི། «པེ་»«སྣར་»ཡོ«པེ་»«སྣར་+པ། \@«པེ་»«སྣར་བརྟག་/az་༼ཆུ་རྫོགས་«པེ་»«སྣར་»བསྒོམ། «པེ་»«སྣར་»པའི། «པེ་»«སྣར་»པ་ནི། «པེ་རྩ་ཡིས། ཚུ«པེ་»ལཱ ི«པེ་»དྨ། «པེ་»«སྣར་»བཞག«པེ་»«སྣར་»«ཅོ་»ཇི། ཨ«པེ་»«སྣར་+དུ། ཁྱེ«སྣར་»དབྱུངས་་ན་ཚRཕྱི་ལ«པེ་»«སྣར་»སོ། «པེ་»«སྣར་»+པ། ༄༅«པེ་»«སྣར་»པ། ::ཀྱི། «པེ་»«སྣར་»ཚེ། ཉི«པེ་»«སྣར་»+རྣམ་པ། «པེ་»+པ། «པེ་»«སྣར་»+པ། ཚ«པེ་»ཁ«སྣར་»མེད། «པེ་»«སྣར་»པ། ༄༅«པེ་མ་མཱ། ཁཱ་ད་ཁོའི་རྒྱུདSཀྱིལ་—།
