ཀ་ཁྱུ་ཀགྱི་''བོ་བསྟོན་འགྱུར། རྒྱུད། པའིཀྱི་ཁཟླ་ཌའི་ཌgaཔབཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཧུཾ་ཧོབཛྲ་ཀུ་མཱ་རཱི། ལཔདྨ། ཨ་ནཾ་གཱ། ་པe;ཀུ་མཱ་རཱིི། མྲྀ་ག་པ་ཏི་ག་མ་ནཱ། རཏྣ་མཱ་ལཱ། སུ་ནེ་ཏཱ། ཀྭིནྣམི། བྷ་རཱ་བྷུཉྩ། ཞེས་བྱ་བས་གཞོན་ནུ་མ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཧཱུྃ་ཧོ་:བཛྲ་ཨིནྡྲི། བཛཱ་བྷཱ། བཛྲ་གཱ་ཏཱ། ཝར་ན་ཀ་བ་ཏཱི། ཙོ་ར་བ་སི། ཁོཙིཏྲ་ལེ་ཁཱི། རམ་བྷཱ། ཧཝ། སུ་ཏཱ་རཱ་བྷྱ། ཞེས་བྱ་བས་དབང་མོ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཏྟཧཱུྃ་ཧོམ་ཧཱ་བྲ་དྷཱ་ཎི༄༅། སཱ་བི་ཏྲ། པདྨ་ནེ་ཏཱི། ཁ་ལུ་ཛ་ལ་ཛ་བི་ཏི། ་བུདྡྷཾ། ཝཱ་གཱི་ཤྭ་རཱི། དེ་གཱ་ཡ་ཏཱི། བི་བྷཱུ་ཏཱ། སྨྲ་ཏི་བྷུགེ། ཞེས་བྱ་བས་ཚངས་མ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཧཱུྃ་ཧོབཛྲ་མ་ཧེ་ཤྭ་རཱི། གོ་རཱི། སཾ་གཱ། ན་ནི་ཏཱ། སུ་པ་ར་མ་ཏུ་རི་ཏཱ། །ཏོ་ཏ་ལཱ། ་ཙལགླུ་ཎཱ། པི་ག་ལཱ། ཀྲྀཏྟཱ་p། བློ་ཞེས་བྱ་བས་དྲག་མོ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཧུཾ་ཧོ། །མ་ཧཱ་ལག། ཌཤྲཱི་ཤྭ་ཏཱ། པོཙནྡྲ་ལེ་ཁཱ། ཤ་ཤཱ་དྷ་ར་བ་དཱ་ནཱ། ཧཾ་ས་བ་ར་ཎཱ། ཨདྷཾ་ཏི། པདྨེ་ཤཱ། ཞིའིཏཱ་རཱ་ནེ་ཏཱ། བི་མ་ལ་ཤ་ཤཱ་དྷ་རཱ་བྷྱ། ཞེས་བྱ་བས་བཀྲ་ཤིས་ཆེན་མོ་འཁོར་དང་བཅས་པ་རྣམས་ལ་གཏོར་མ་དབུལ་བར་བྱའོ། ་་། ས་པ་རི་ཝཱ་རེ་ཞེས་བྱ་བ་ལ་་་་་་སོགས་པ་ནས་སྭཱ་ཧཱའི་བར་དུ་སྡེ་ཚན་རེ་རེ་ཞིང་ཡང་ཤེས་པར་བྱའོ། །དབྱུ་གུ་་་དྲུག་ཅུ་རྩ་བཞི་ཡི། ཀ་མ། གཏོར་མ་དྲུག་ཅུ་རྩ་བཞི་ཡིས། ཏུ་། ས་སྟེངས་གནས་ན་ཕོ་ཉ་རྣམས། །མཆོད་བྱས་དགོས་འདོད་སྒྲུབ་པར་བྱེད། །ཕྱི་དང་ནང་གི་ཆོས་ཤེས་པའི་སྔགས་པས་རྣལ་འབྱོར་མའི་ཚོགས་རྣམས་མཆོད་ཅིང་མཉེས་པར་་་་་བྱས། མཆོག་དང་ཐུན་མོང་གི་དངོས་གྲུབ་ཏུ་འགྱུར་རོ། །དཔལ་དུས་ཀྱི་འཁོར་ལོ་ལས་རྣལ་འབྱོར་མ་དྲུག་ཅུ་རྩ་བཞིའི་གཏོར་མའི་ཆོ་ག་བསྡུས་པ། རྟ་རྣལ་འབྱོར་མ་དྲུག་ཅུ་རྩ་བཞི་དབྱུ་གུ་དྲུག་ཅུ་རྩ་བཞིའི་གཏོར་མའི་ཆོ་ག་རྫོགས་སོ། །ས། །།
