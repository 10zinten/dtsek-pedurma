བསྟན་འགྱུར། རྒྱུད། ག་གནས། །ྋ། ལུས་ལ་འོད་འཕྲོ་ལོངས་སྤྲུལ་མཐོང་། ྋ་། །རྗེས་དྲན་ལ་ནི་སྙོམས་ཞུགས་ན༑། འོད་ཀྱི་དཀྱིལ་འཁོར་འབྱུང་བར་འགྱུར། །རྗེས་མེད་ལ་སོགས་སྐུ་ཤེས་བྱ། །དེ་བརྟན་བྱ་ཕྱིར་ཏིང་འཛིན་བསྒོམ། །བདེ་སྟོང་དབྱེར་མེད་དེ་ཉིད་ལ། །ཤིན་ཏུ་གོམས་པར་བྱ་བའོ། །ཐམས་ཅད་བདེ་སྟོང་ལ་སོགས་འབྱུང་། །ལྷུན་གྲུབ་ལ་སོགས་དགུ་རུ་འགྱུར། །།ཏིང་འཛིན་དབང་སྒྱུར་ཙམ་གྱིས་ནི། །སྒྲིབ་པ་ཐམས་ཅད་འབྱང་བར་འགྱུར། །དེ་གཉིས་ཐ་མར་དགེ་བའོ། །དེ་ལ་སྤྱོད་པ་གོམས་པ་ཡིས། །ཚེ་འདི་ཉིད་དུ་སངས་རྒྱས་ཐོབ། །མ་ཐོབ་བར་དོའི་སྦྱོར་དྲུག་བསྒོམ། །དེ་ནི་ཆགས་བྲལ་མན་ངག་སྟེ། །།ཆགས་ཅན་སྦྱོར་དྲུག་གཞན་དུ་ཤེས། །དེ་བྱས་དགེ་བས་བདག་དང་གཞན། །རྡོ་རྗེ་འཆང་ཆེན་གནས་ཐོབ་ཤོག། སྦྱོར་བ་ཡན་ལག་དྲུག་གི་མན་ངག་ཙེ་ལུ་པས་མཛད་པ་རྫོགས་སོ། །ཀ་ཁ་ག་ང་ཚོ་ལྔ་པཎྜི་ཏ་ས་མནྟ་ཤྲཱི་དང་། ལོ་ཙཱ་བ་ར་ཆོས་རབ་ཀྱིས་བསྒྱུར་བའོ། །།།::གཞན་ཆ་བྱེད་པ། བའི་ལ་གཊབཀག་འཁོར་བདེ་བ་ཎ་ཞུསུ་ག་བ་ད་ན་ལོཀ། །ནན་ཅུ་པ་ཚུལ་ཁཊ་པ་ཁོ་ན་ཟླ་ཞིཝ་ཎའཁྲི་ཚད་ཡོད་པ་ལྟར་ན་རཏུ་ན། ཁོ་། ཋཀ། །ད་འོ་ན་དེ་བཟོ་བར་སྡུག་ཏུ། །དཀའོ། །པ་ནི་རྒྱུ་ཁོ་ན་བཏང་བས། །ཡི་དབཏང་ནས་སློབ་ཞོ། དར་བ་ཁ་ང་ད་ན། ད་འོ་། སྤྲོ་ན་ཀུན་བཀའ་བཏང་བ་རྒྱ་འདི་ཅི་ད་འོབག་ཚུ་ལླ་ཁོལ་ལད་ལ། ད་ཚོང་ཁ་བ་དང་ཚ་བ་ཏན་ཏའི་ཝརྒྱུ། །པཆི་སྐྱོན་ཚད་ལྟོ་གླ་ན་གཐུམ་ལ་ཁྱབ་རྒྱུ་ལྟ་བུ་ལ་ཁྲོ་བ་ནས་ཆུང་ཁཊི་ཀ་བན་དང་པ་མི་བྱ་ཁྱབ་ཏཚོགཙུག་ཏུ་བོད་ཡིགདཔེ་འོད་ཁeནས་ཤ་ག་བ་ག་བ། །ཚེ་ཐ་ན་ཚོའི་་ཁཟླ་སྤྱི་པ་ཏ་ཁ་ཎིདེ་ག་ཉཞུང་འོག་། ་་ག་པ་་བ་འཆགྲོ་རྒྱུ་ཎས་རིག་ག་འདྲ་བ་བཏོན་།
