ཁར་བློ། བསྟན་འགྱུར། །རྒྱུད། ་་paཏྭཱ་བྱོན་ནས་ཀྱི་སངས་རྒྱས་ཀྱི་བསྟན་པ་དར་བར་མཛད་དོ། ལོ་ཆ། གནས་པ་རྩ་ཡི་དུས་་འཁོར་གྱིས། །ཕྱི་ནང་ཆོས་ཀུན་ཤེས་པར་བྱེད། །e། གཡོ་བ་རླུང་གི་དུས་འཁོར་གྱིས། །སྲིད་དང་ཞི་བ་ཁ་ལོས་སྒྱུར། །འགྱུ་བ་བྱང་སེམས་དུས་འཁོར་གྱིས། །ཕུན་སུམ་ཚོགས་པ་སྐྱེད་པར་བྱེད། །འཁོར་ལོ་གསུམ་གྱི་དོན་ཐོས་ནས། །འཁོར་ལོ་གསུམ་གྱི་དོན་བཀོད་པས། །ཁམས་གསུམ་གནས་པའི་སེམས་ཅན་རྣམས། །སྐུ་གསུམ་མངོན་དུ་རྟོགས་པར་ཤོག། དཔལ་དུས་ཀྱི་འཁོར་ལོའི་མན་ངག་། འཁོར་ལོ་གསུམ་གྱི་རིམ་པ་ཉི་མ་ཟླ་བ་སྒྲུབ་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །།།པའི་ལ་རྒྱ་གར་ཤར་ཕྱོགས་ཛ་ག་དྷ་ལའི་པཎྜི་ཏ་མ་ཧཱ་བི་བྷཱུ་ཏི་ཙནྡྲས། དཔལ་དིང་རི་གླང་འཁོར་དུ་སྒྲ་རང་འགྱུར་དུ་བསྒྱུར་ནས་གསུངས་པའོ། །Seeབྱང་ན་ཞུ་ཁ་ཎཏ། ཁ་ན་ལོ་ལྔའཏུ་དཀ་/sདང་། །ཚེ་ད་པ་ཆ་བློ་ལྔ་འོག་བཤགས་པས་གསལ་བ་ཁྱོད། ཆད་པ་དེ་མུར་ཁོ་ན། །epའ་བཀླགས་པ་ལ་བབསྡུར་མཆན། སྤྱི་ལོ་ལྟར་རྒྱུ་མ་བར་འདུམེ་ཆ་ཁོད། མཚོ་P། ་ལ་བཏཎཁ་དཆ། བ་ཏུ་ཀར་པཚོ་ཏོ་ནམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»དཔལ་དུས་ཀྱི་འཁོར་ལོའི་གདམས་ངག་ཉི་མ་ཟླ་བ་སྒྲུབ་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །®«པེ་»«སྣར་»ཨུ། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»སྒྲུབ། «པེ་»«སྣར་»བྷེ། ཁ། ང་༽°ཚོ«པེ་»«སྣར་སྐྱེད། ®«པེ་《སྣར་ནས། «སྣར་»དགཚེ་%s། །ལམ་མེ། ཟླ་«པེ་»«སྣར་རུང་། ®«པེ་»སྣར་»དང་། «པེ་»«སྣར་བྱེད། «པེ་»«སྣར་»བྱུག«པེ་»«སྣར་»སྐྲག«པེ་»«སྣར་བྱེས། «པེ་»«སྣར་བརྟགས། «པེ་»«སྣར་ཚེས། «པེ་»«སྣར་»ནས། «པེ་»«སྣར་»སྦྱང་། །སྐྱེ་«པེ་དྷ་ཏཱིར་བཟང་། «སྣར་དྷུ་ཏཱིར་གཟང་། «པེ་»«སྣར་»དྷ། «པེ་»«སྣར་»སྤོངས། །«པེ་»«སྣར་»དྷཱ། «པེ་»«སྣར་»+བ། «པེ་»«སྣར་»སྐྱེས། «པེ་»དྷ། «པེ་»«སྣར་»ནས། «པེ་»«སྣར་»ཧཱུྃ། «པེ་»«སྣར་པ། @«པེ་»«སྣར་»བ། ༄༅«པེ་»«སྣར་»མཐུན། །
