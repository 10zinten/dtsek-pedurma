ཨོ་ཧཾ་ཐཾ་པཉྩ་ཀ་ར་མ་ཧཱ་ཤུ་ནྱ་ཏཱ་སྭ་བྷ་བ་བཛྲ་ཤི་རོ་མ་ཎི་ཤི་ར་སི་:སརྦ་ཨོཾ་གེ་བྷསྨ་ཧུཾ་ཧཱུྃ་ཕཊིཕཊ། ་ཨོ་ཨོཾ་ཨཱདྷ་ན་ྋརཱི་ཤཱ་ར་བཛྲ་ཨཌྷ་ཙནྡ་བཛྲ་:པཊྚ་ཨུཎྷ་ཧཱུྃ་ཧཱུྃ་་ཕཊ་ཕཊ། ་་ཨོཾ་ཨ་ཨཱཉྩསཾ་སརྦ་སཱ་རི་ཀ་ཏྲི་སྭ་ར་སྭ་བྷཱ་བ་བཛྲ་ཡདྷོ་པ་བཱི་ཨེཏ་སྐནྡྷེ་ཧུཾ་ཧཱུྃ་ཕཊ་ཕཊ། ཋ་དེ་ནས་ཕྱག་རྒྱ་བཞི་བཅས་ལ། ་། ༀ་ཀྵདུར་དཱནྟ་བྷི་མ་ན་ཀུ་ཡཾ་ཀ་ར་བྱཱ་གྷཱ་ཙརྨ་ཀ་ཊོ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོ་ཀླུ་ཏྲི་ཤད་གྲྭ་ར་སཔྟ་ཏི་ཧྲ་སཱ་དཱིགྷ་བྱཉྫ་ན་སྭ་བྷཱ་བ་དི་ཏེ་ཀུ་ལ་ཤ་ཏ་སི་རོ་བྷིར་མུཎྜ་མཱ་ལ་སྐནྡྷེབ་ལཾ་བི་ནི་སྐནྡྷེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧ་ཧི་ཙྰ་ཧུ་བློ། པཉྩ་སྐནྡྷ་བི་ཤུདྡྷ་སྭ་བྷཱ་བ་ཤི་ར་སི་ཀ་པཱ་ལ་མཱ་ལེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ༀ་ཀྵ་ཀྵ་ཀྱི་གླུ་ཚཱ་ཙཱ་གླུ་ལྱཱ་ཎཾ་ཨཾ་ཀྵཀྵཧ་ཧཱ་ཧི་ཧཱི་ཙཱ་ཙཱ་ྋསྶཾ་དྷྭཾ་ཧཾ་ཧཱ། ཀྵ་། སརྦ་བིགྷཾ་བི་ད་རི་ཏ་ག་ཛ་ཙརྨ་པ་ཊ་ཨ་རཀྟ་ཐ་བ་ཏ་ཧཱུྃ་ཕོ་ཉཕཊ། ཀུ་དེ་ནས་གཡས་ཀྱི་ཕྱག་རྒྱ་བཅའ། །ཨོ་ཌཱཀྲོ་དྷ་བཛྲ་ཧཱུྃ་ཕཊ། ཨོ་ཥཱབཛྲ་ཁན་་་ཧཱུྃ་ཕཊ། །ཨོཾ་ཀྵབཛྲ་ཏྲི་ཤུ་ལཾ་ཧཱུྃ་ཕཊ། ༀ་ཀྵབཛྲ་ཀརྟ་རི་ཧཱུྃ་ཕཊ། ༀ་ཀྵབཛྲ་བཱ་ན་ཧཱུྃ་ཕཊ། ༀ་ཀྵ༔བཛྲ་ཨོ་ཉཀུ་ཤ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵབཛྲ་ཌ་མ་རུ་ཀ་ཧཱུྃ་་་ཕཊ། ཌཨོ་བྷཱབཛྲ་མུངྒ་ར་ཧཱུྃ་ཕཊ། ༀ་ཀྵབཛྲ་ཙཀྲ་ཧཱུཾ་ཕཊ། ཨོ་ཀཿབཛྲ་་་ཀུནྟ་ཧཱུྃ་ཕཊ། ཨོ་ཥཱབཛྲ་དཎྜ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵབཛྲ་པ་ར་ཤུ་ཧཱུྃ་ཕཊ། དེ་ནས་གཡོན་གྱི་ཕྱག་རྒྱ་བཅའ། །ཨོ་སྨྱོ་བཛྲ་གཎྜི་ཧོ་ཕྲོ་ྋཕཊ། ཨོ་ཙྪེ་བཛྲ་ཁེ་ཊ་ཧོ་ཕོ་ཕཊ། ཨོཾ་ཧཱུཾ་བཛྲ་ཁ་ཊཾ་ག་ཧོ་ཕྲོ་ཕཊ། ཨོཾ་ཧཱུྃ་བཛྲ་ཀ་པཱ་ལ་ཧོ་ཕོ་ཕཊ། ཨོ་སྒྱུ་བཛྲ་ཙཱ་པ་ཧོ་ཕྲོ་ཕཊ། ཨོཾ་ཧཱུཾ་བཛྲ་པཱ་ཤ་ཧོ་ཕྲོ་ཕཊ། ཨོཾ་ཧཱུཾ་བཛྲ་རཏྣ་ཧོ་ཕྲོ་ཕཊ། ཨོ་ཨཽ་བཛྲ་པདྨ་ཧོ་ཕྲོ་ཕཊ། ཨོཾ་ཧཱུཾ་བཛྲ་ཤཾ་ཁ་ཧོ་ཕོ་ཕཊ། ཨོ་ཨཽ་:བཛྲ་ཨཱ་་དན་་ཧོ་ཕྲོ་ཕཊ། :ཨོ་ཨཽ་བཛྲ་ཤྲཱི་ཁ་ལེ་ཧོ་ཕྲོ་ཕཊ། ཨོ་ཨཽ་ཙ་ཏུར་བེ་ཏ་མུ་ཁ་སྭ་བྷཱ་བ་བྲདྨ་ཤི་རོ་ཧོ་ཕྲོ་ཕཊ། དེ་ནས་ཀླུའི་ཕྱག་རྒྱ་བཅའ། ཨོ་ཧཱུྃ་བཛྲ་ཛ་ཡ་ནཱ་:ག་ཨིནྡྲ་བཛྲ་ཛ་ཊཱ་མུ་ཀུ་ཊ་བནྡྷེ་ཏིཥྛ་ཏིཥཱ་ཧཱུྃ་ཨེཧཱུཾ་ཕཊ་ཕཊ། ཨོཾ་ཧཱུཾ་བཛྲ་བི་ཛ་ྋཡ་ནཱ་གེནྡྲ་དབྱེ།
