མཐར་གྱུར་ཁྱོད་ལ་འདུད། ཚོ། སྟོང་ཉིད་སྙིང་པོ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་བྱང་ཆུབ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་ཆགས་པ་ཁྱོད་ལ་འདུད། །ཡ། །སངས་རྒྱས་འདོད་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་དགྱེས་པ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་རོལ་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་འཛུམ་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་བཞད་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་གསུང་ཉིད་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་དངོས་ལ་ཕྱག་འཚལ་འདུད། །མེད་པ་ལས་བྱུང་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་འབྱུང་བ་ཁྱོད་ལ་འདུད། །ནམ་མཁའ་ལས་བྱུང་ཁྱོད་ལ་འདུད། ལ། ཡེ་ཤེས་ལས་བྱུང་ཁྱོད་ལ་འདུད། །སྒྱུ་འཕྲུལ་དྲ་བ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་རོལ་སྟོན་ཁྱོད་ལ་འདུད། །ཐམས་ཅད་ཐམས་ཅད་ཁྱོད་ལ་འདུད། །ཡེ་ཤེས་སྐུ་ཉིད་ཁྱོད་ལ་འདུད། །།ཅེས་བྱ་བའོ། །དེ་ལྟར་ལན་གསུམ་བརྗོད་ནས་སུ། །འདོད་པའི་ལྷ་ཚོགས་གཤེགས་པར་བསམ། །དེ་ནས་བདུད་རྩི་མྱང་བ་ཡི། །ཆོ་ག་བཞིན་དུ་་་གཏོར་མ་ནི། །བསྒྲུབ་པ་བྱས་ནས་རྩེ་ལྔ་པའི། །ཕྱག་རྒྱས་སྔགས་འདི་སྦྱིན་པར་བྱ། །དེ་ནས་གཏོར་མའི་སྔགས་ནི། །:ཨོཾཨཱཧཱུྃ''ཧོ་ཨུཏྟཱི་ཥ། སུཾ་བྷ་ནི་སུམྦྷལ། བིགྷཱནྟ་ཀ་ནཱི་ལ་དཎྜ། པྲཛྙཱན་ཏ་ཀཊཱ་ཀི་རཱ་ཛ། པཏྟཱནྟ་ཙཀ་ཨ་ཙ་ལ། ཡ་མཱནྟ་ཀ:མ་ཧཱ་བཱ་ལེ་ས་པ་རི་:ཝཱ་རེ་ྋཨི་དཾ་བ་ལི་གན་དྷཾ། པུཏྠཾ:༑དྷཱ་པཾ། དཱི་པཾ། ཨཀྵ་ཏཾ། ད་དཱ་མ་ཧེ། ཏེ་ཛ་ཨཱ་ག་ཏེ་ས་པ་རི་ཝཱ་རི༑ཤྲཱི་གྷོ་ཨི་དཾ་བ་ལཾ། གྲིཙྪནྟུ་ཁ་དནྟུ། .པི་བནྟུ། ཛཧཱུྃ་ཏཁཾ་ཧོས་་་ཏྲིདྡྷསརྦ་སཏྭཱ་ནཾ་ཤཱནྟི། པུཎྜི། རཀྵ་ཨཱ་ཝ་ར་ཎ། གུསྟི། ཀུརྔན་ཏུ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་བཛྲ་དྷ་ར་ཨཱ་ཛྷཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ནི་ཁྲོ་བོའི་རྒྱལ་པོ་རྣམས་ཀྱི་གཏོར་མའི་སྔགས་སོ། །སྔགས་འདི་བརྗོད་ནས་གཏོར་མ་ནི། །།བྱིན་ནས་ཡིད་དུ་འོང་བ་ཡི། །དགོས་པ་བསྒྲུབ་པར་བྱ་པའི་ཕྱིར། #། དེ་ལ་སྔགས་འདིས་གསོལ་བ་གདབ། །།
