ཨོ་བྲཧྨ་ཎ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ?ས་མ་ཡོཉཧཾ། །ཨོ་བྲཧྨ་ཎི་ཡེ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟེ། ས་མ་ཡོཉཧཾ། །ཨོཾ་སཱ་ར་སཱ་ཏི་ཡེ་དྲི་ཤྱ་ཛཧཱུྃ་བོ་ཧོ༑་ས་མ་ཡ་ཧཱུཾ། ་ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་སྙིང་གར་བཅིངས་:ནས་ནི། །པདྨའི་སྣོད་བཞིན་རབ་ཏུ་དབྱེ། །ས་། ཨོཾ་རུ་ཏྲ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོཾ་མ་ཧེ་ཤུ་ར་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། །ཨོ་གཾ་གཱ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བསྡམས་པ་བརྟན་བཅིངས་ནས། །རྡོ་རྗེ་རྣམ་པར་གཟུང་བར་བྱ། །ཨོ་ཤ་ཀྲེ་དྲྀ་ཤྱ་ཛཧཱུཾ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ་ས་མ་ཡོཉཧཾ། ཨོ་ཨིནྡྲ་ནཱི་ཛ་བྱ་ཛཧཱུྃ་བཾ་ཧོ། །ས་མ་ཡ་ཙཱ། ས་མ་ཡོཉཧཾ། ཀློཨོ་ཏི་:ལོཏྟ་མ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བསྡམས་པ་བརྟན་བཅིངས་ནས། །དཀར་པོ་གི་ཝང་རྩེ་མོ་སྦྱར། །ཅུང་ཟད་པར་ནི་རྣམ་པར་དབྱེ། །སྤྱན་གྱི་ཕྱག་རྒྱའི་རྣམ་པའོ། །ཨོ་ཨིནྡྲ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། །ས་མ་ཡོཉཧཾ། ཨོཾ་ཤ་ཙ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཙཱ། ས་མ་ཡོཉཧཾ། ཨོཾ་རཾ་བྷ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སིཾ། ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བསྡམས་པ་ལེགས་པར་བཅིངས། །དཀར་པོ་གི་ཝང་རྩེ་མོ་སྦྱར། །རིན་པོ་ཆེ་ཡི་དབྱིབས་སུ་བྱས། །གནོད་སྦྱིན་སྲིན་པོའི་ཕྱག་རྒྱའོ། །ཨོཾ་ཀུ་བེ་ར་དྲི་བྱ་ཛཧཱུྃ་བོ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོཾ་བ་སུ་མ་ཏཱི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོ་ཧཱ་རཱི་ཏཱི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། :ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བཅིང་བ་བརྟན་བཅིངས་ནས། །དཀར་པོ་གི་ཝང་གྱེན་དུ་བསྒྲེང་། །བ་དན་རྣམ་པར་གཡོ་བར་བྱ། །འབྱུང་པོའི་རྒྱལ་པོའི་ཕྱག་རྒྱའོ། །ཨོཾ་བྷ་ཏ་ཨ་དྷི་པ་ཏི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ༀ་ལཾ་ཀེ་ཤྭ་:རཱི་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ::ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོ་རཀྵ་པྲ་ཡེ་དྲིབྱ་ཛཧཱུྃ་བཾ་ཧོ། །།
