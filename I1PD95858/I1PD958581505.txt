ཟླ་FEཀ། བསྟན་འགྱུར། རྒྱུད། རིཏོ'g''"aབདག་གི་དགེ་བའི་ལས་རྣམས་ཀྱིས། །སངས་རྒྱས་མྱུར་དུ་བསྒྲུབ་བྱས་ཏེ། །སེམས་ཅན་འཁོར་བའི་རྒྱ་མཚོ་ལས། །བླ་མེད་གནས་སུ་བསྒྲལ་བར་བགྱི། །ཡུད་ཙམ་སློབ་མ་ལྷར་བྱས་ནས། །།གནས་གསུམ་ཐིག་ལེ་བཀོད་པའི་སར། །རྡོ་རྗེས་མཚན་པའི་འབྲུ་གསུམ་བསྒོམ། །སྐུ་གསུང་ཐུགས་ཀྱི་མཚན་མར་གྱུར། །དེ་དཀྱིལ་འབྲུ་གསུམ་ནུས་མཆོག་གིས། །བཀུག་བསྟིམ་ལྷ་རུ་གསལ་བྱས་ལ། ཡི་། སྤྱི་བོར་མེ་ཏོག་མདུན་དུ་སྤོས། །དྲི་དང་མར་མེ་སྙིང་གར་ཏེ། །ལྷ་བཤོས་ཞོལ་དུ་སྙན་ལ་སྒྲའོ། །ཨུ་དུམྦ་ར་ཨཤྭ་ད་སོགས། །སོར་བཅུ་གཉིས་ཀྱི་ཚད་ཙམ་ལ། །སྐྱོན་བྲལ་མེ་ཏོག་རྩེར་བཅིངས་སྦྱིན། །ཤར་རམ་བྱང་དུ་ཁ་བལྟས་ཏེ། །མི་སྨྲ་བར་ནི་སོ་ཤིང་བཅའ། །བ་ཀོ་ཙམ་ལྕི་བྱུགས་པའི་སར། །།སོ་ཤིང་མདུན་ནི་དྲང་པོར་དོར། དེ་ཚེ་རང་ལ་བལྟ་བ་དང་། ::། རྩེ་མོ་གྱེན་དུ་:འགྲེང་གྱུར་ན། །དངོས་གྲུབ་མཆོག་ཏུ་དེ་བཤད་དོ། །མཚམས་དང་བྱང་དུ་བལྟས་པ་ན། །འཇིག་རྟེན་པ་ཡི་དངོས་གྲུབ་ཐོབ། ::། དེ་བཞིན་གཞན་དུ་ཁ་བལྟས་ན། །དངོས་གྲུབ་་་འབྲིང་པོ་ཉིད་དུ་འགྱུར། །སོ་ཤིང་བོར་བ་བརྒྱ་ཞིག་ལ། །གལ་ཏེ་ཐུར་དུ་ཁ་བལྟས་་ན༑། དེ་ནི་ངེས་པར་ས་འོག་གི། །དངོས་གྲུབ་དག་ལ་ཐེ་ཚོམ་མེད། །ཨོ་ཧིལ་སོགས་གསང་སྔགས་ཀྱིས། །སྐྱོར་མོ་བསྡམས་པའི་ཆུ་བླུད་བྱ། །འཇིག་རྟེན་དུ་ནི་ཀུན་མཁྱེན་ཀྱང་། །ཨུ་དུམྦྷ་རའི་མེ་ཏོག་ལྟར། །བརྒྱ་ལམ་ན་ནི་རེས་འགའ་ཞིག། འབྱུང་འགྱུར་ཡང་ནི་མི་འབྱུང་ངོ༌། །གསང་སྔགས་སྤྱོད་པའི་ཚུལ་འདི་ནི༑། དེ་བས་ཀྱང་ནི་རྙེད་དཀའ་སྟེ། །གང་གིས་སེམས་ཅན་མཉམ་མེད་དོན། །བླ་ན་མེད་པར་བྱེད་ནུས་པ། རྩེ་། གཉིས་མེད་ཟབ་མོའི་སྤྱོད་ཚུལ་འདིས། །སེམས་ཅན་སངས་རྒྱས་ཐོབ་བྱེད་པའོ། །དང་པོ་བསྐལ་པ་གྲངས་མེད་ནས། །བསགས་པའི་སྡིག་པ་གང་དང་གང་། །དེ་ཀུན་ཟད་པར་:འགྱུར་བ་ནི། །ཟབ་མོའི་དཀྱིལ་_།
