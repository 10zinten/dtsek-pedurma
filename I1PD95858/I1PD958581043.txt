བསྟན་འགྱུར། རྒྱུད། བཟང་པོ་ཞེས་བྱའོ། །དབུ་མ་། དགའ་aལ་རྒྱ་གར་གྱི་མཁན་པོ་སྨྲ་ཏི་ཛྙཱ་ན་ཉིད་ཀྱིས་བསྒྱུར་ཅིང་བཤད་ནས་གཏན་ལ་ཕབ་པའི་རྡོ་རྗེ་བདུད་རྩིའི་རྒྱུད་ཀྱི་རྣམ་པར་བཤད་པ་རྫོགས་སོ། །ཁ་བག་། ་་་་་ཁ་ཁ་ཁ་འཊ་གལ་བའི་ཚེ། ཁཚོ་ཁ་བ་དང་ཁྲུང་་བ་ཀྱི་ཀཀཀྱི་ཀཀ་ཁོ་ན་བ་དང་། ཚུལ་་ལ་ཁSཚོ་བ་དང་། ས་བ་བར་པ་ཚེ་ཏོ་རྗེ་བསྡུར་མཆན། །ཞེས་ཁཞང་ཀ་པ་ལ་ཁལ་དཔ་ད་ག་ར་ཀྱང་། གསང་བ་བ། །དེ་ལ་པོ་ངའི་མདུན་དུ་ཁྲn;ཞེང༌། ཁག། (མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»རྡོ་རྗེ་བདུད་རྩིའི་རྒྱུད་ཀྱི་འགྲེལ་པ་བཞུགས། །(«པེ་»«སྣར་»མི། «པེ་»«སྣར་»བ། ཆོ«པེ་»«སྣར་»དྲི། «པེ་»«སྣར་»འགྲེལ། ཨ@«པེ་»«སྣར་»སྐྱོད། »། ཁགཏཱཞེས:ར་ཨf_ཡོད་པས། ང་@«སྣར་»སྔོན། «པེ་»«སྣར་+པའི། །«པེ་»ཟེར་གྱི་ཀོ། «སྣར་ཟེར་གྱི་སྐྱོ། «པེ་»«སྣར་ནི། «པེ་»«སྣར་»ལ་འགལ་བ། «པེ་»«སྣར་»འདིར་ནམས། «པེ་»«སྣར་»ཞུགས། «པེ་»«སྣར་»སྲུངས། °«པེ་»«སྣར་»བ་སྟེགས་ཀྱིས། «པེ་»《སྣར་པུ། ：:་། ཀྱང་(«པེ་»«སྣར་»པ། «པེ་»«སྣར་»པ། །«པེ་»«སྣར་»གཤིས། «པེ་»«སྣར་»ཧཱུཾ། «པེ་»«སྣར་»བཞུགས། «པེ་»«སྣར་»དང་། «པེ་»«སྣར་»དབུས་ཀྱི་ཞལ་དང་སྐུ་མདོག་སྔོ། ...ཕྱག་དྲུག་ལ། ཨེ«པེ་»«སྣར་»ན། («པེ་»《སྣར་»པ། «པེ་»«སྣར་གཡས་གནག་གཡོན་དཀར། «པེ་»ཀུན། «པེ་»«སྣར་»པོ། ་(«པེ་»«སྣར་»གུཎེ། «པེ་»«སྣར་»ན།  ོ«པེ་»«སྣར་»དང་། «པེ་»«སྣར་»པ། ནས་མི«སྣར་+བོ། «པེ་»«སྣར་»བོ། ༄༅«པེ་»«སྣར་»མེ། །དལ་(«པེ་»«སྣར་»ཀྱིས། ལག་པའི། །ང་། ནང་ལོ(«པེ་»«སྣར་»+པ། ®«པེ་»«སྣར་»ཧཱུྃ། «པེ་རྗེ་ཡི། «སྣར་»རྗེས་ཡི། «པེ་»«སྣར་»+མཆོད་ལ་བསྟོད་དེ། «པེ་»«སྣར་»བྱ། །«པེ་»«སྣར་»གཙོ་བོར། («པེ་»«སྣར་»སྐྱེད། ན་ཏུ་«པེ་»«སྣར་»པའི། «སྣར་»འགྲེལ། «པེ་»«སྣར་»ཏེ། །«པེ་»«སྣར་»རྩི། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་པ་སྟེ། །«པེ་»«སྣར་+དང་། «པེ་»«སྣར་»བསྒྲུབ། །«པེ་»«སྣར་»ལྡན་པའི། «པེ་»«སྣར་+།
