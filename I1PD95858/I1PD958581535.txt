བསྟན་འགྱུར། རྒྱུད། རོ་་་་་་་་་་་་་་་་རེད། བསམ་ཁོ་ད་ང། བོཎཊ་ཏ་ཨཊྚ་ཀྱི་ཕྱག་ཏེ་པོ་ལྔའི་ཚུན་ལ་ཚོད་འདོད་ཀྱི་རི་བོའི་རྗེ། །་ཚེ། དང་བདག་སོགས་དབྱེ་བ་གས་ན་བསྡུར་འབྲས་རེའུ་མིགདྲི་ཏོ་ཏི་ཏི་ར་ཁོ་ན་དེ་་པ་དང་། ་རྒྱུད། (ཕྱི་བ་དང༌། རྟ་འོ། །ན་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། ་། རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། བ། །རྒྱུད། ཀླད་པ་དང་པོད་རྟགས། ཚན་གྲངས། [་། ལྡེབ། བ་—བ་། བན་ནན་བ་བ་ཕྲེང་། ཟླ་བ་། །་ཚེག་ཁྱིམ། ང་ཀ་ཊ་}Aག་ཆ་བ་དང་། འ:་༌་ད་ལུ་::་་་་་་་་་་་་་་། །།།་་ནགས་ཚུལ་ཁོང་གི་དཀྱི་་་་་་་མ་ཐོབ་པ་དེ་དང་། བ་གང་། །པ་ཐོབ་པ་བསྐྱེད་པ་གྲོ་ནང་དུ་ཡོད་པ་འདུག་་བ་ལ། །ཟབ་ལ་ན། ད་ག་ག་ན་ས་མ་ད་ལ་དབང་བའིབྱ་བ་ལྔའི་ངོ་བོ་བདུད་ལྔ་དེ་ལ་ས་བ་མ་ས་དང་དབལ་ལཡུལ་ལ་དབང་བསྐུར་རོ། །དབང་། བར་པ་ལ་དོན་དང་པོ། དཔག་བ་རྒྱུ་།
