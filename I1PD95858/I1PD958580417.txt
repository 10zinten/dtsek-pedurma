རྒྱུད། མི་ཏཱ་ཎ་ཨ་nཁ་དངོས། །།སློབ་དཔོན་བྱང་ཆུབ་རྡོ་རྗེ་ཉིད། །དངོས་རྣམས་སྒྱུ་མ་ལྟ་བུ་ལ། །སྤྲོ་བ་དང་ནི་བསྡུ་བ་མཛད། །རྣལ་འབྱོར་ཆེན་པོ་སྐལ་ལྡན་རྣམས། ལ་། སྡིག་པ་རྣམས་ནི་རྣམ་བྱང་ཞིང་། །བསོད་ནམས་རྣམས་ནི་འཕེལ་བ་རྣམས། །འོ་། །བདེ་བ་ཅན་དུ་འགྲོ་བར་ཤོགཚེ་། ཤིང་གཅིག་པའི་དཀའ་འགྲེལ་སློབ་དཔོན་འཕགས་པའི་ལྷའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །མི་དག་ལ་་། །ལ་ལ། །་་་་་་་་་་་་་ཀྱིཁ་ལ་ཀ་ཡི་འདབ་མ་བཅུ། གལ་ཏེ་ལུས་ཀྱི་ཚེ། །ཇི་ལྟ་ཁོའི་ལོ་དགའ་སྐད་ཁོ་ན་ལས་སྐྱོ་ག་བ་ག་བ་ག་བ་ག་བ་དན་དེ་ལ་་མཁོ། ངོ་ཡཏེ། །ཁ་ནས་དེ་བ་ག་བ་ད་ར་སོགས་འགའ་བསྡུར་མཆན། དར་ཡ་། །ཤི། རོཏུ་ག//པ་། :ས་དང་། འཕྲལ་ཀློང་དབྱར་གནས་པར་དུ་བ་འབད། །ང་ཁ་པ་མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»དཔལ་གདན་བཞི་པའི་ཟབ་དོན་འགྲེལ་པ་་་བཞུགས། «པེ་»«སྣར་»མའི། «པེ་»«སྣར་»འཐོབ། «པེ་»«སྣར་མར། «པེ་»«སྣར་»བཞིན། ་་:། :མི་གསལ། །ནང། བ་བློའིསྤྲོ་@«པེ་»«སྣར་»པས། @«པེ་»«སྣར་»ལ། །@«པེ་»«སྣར་རབ། «པེ་»«སྣར་»རིགས། །«པེ་»«སྣར་»ལས། @«པེ་»«སྣར་གའོ། «པེ་»«སྣར་»འདི། ཉི«པེ་»«སྣར་»དང་། «པེ་»«སྣར་»ཏྲ། ཉི«པེ་»«སྣར་»ཡིས། «པེ་»«སྣར་»བཞི་པ། °«པེ་»«སྣར་»བུ། «པེ་»ཡི། «པེ་»《སྣར་»སོར། «པེ་»«སྣར་ཚོགས་བཤད། «པེ་»«སྣར་»པར། «པེ་»ཚུ་ཀ«སྣར་»ལུང་རིག«པེ་»«སྣར་པར། དགྱེ«པེ་»«སྣར་»རྩི་ཡང་། ང་(«པེ་»«སྣར་»ས། ®«པེ་»«སྣར་»རྗེས། «པེ་»«སྣར་»རེ། «པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་»ཡི། «པེ་»«སྣར་»ལས། །«པེ་»«སྣར་གྱིས་རླབས་ལས་འབྱུང་། «པེ་»«སྣར་»དང་། ®«པེ་»«སྣར་»རྟོགས། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»སྐྱེར། པོ«པེ་»«སྣར་གྱིས། @«པེ་»«སྣར་»གསུམ་པོ། @«པེ་»«སྣར་»ཀྱི། ཁྱེཔེ་»«སྣར་»གི། ཀྱེ«པེ་»«སྣར་»+ཞིང་བརྒྱད་དག་ནི་གནས་རྣམས་ནི། །«པེ་»«སྣར་»བཀྲོལ། «པེ་»«སྣར་སན་དོ། «པེ་»«སྣར་གྱི། «པེ་»«སྣར་བཟུང་། གང་ལྟ། «པེ་»«སྣར་»ཊི། «པེ་»«སྣར་པར། །་་་་་་་་་་་དང་། ཚོགས་«པེ་»«སྣར་»སུམ། «པེ་»«སྣར་»བསྟན། «པེ་»«སྣར་»ཉིད། པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་རྒྱུས། ༄༅«པེ་»«སྣར་»མོ། «པེ་»«སྣར་»བརྟགQ«པེ་»«སྣར་»+དག་པ། —།
