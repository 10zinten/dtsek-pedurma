མངོན་བསྔགས་ལ། །ལྕེ་རྩེར་རྡོ་རྗེ་བསམ་བྱས་ཏེ། །བཀུག་སྟེ་དེ་ཡིས་ཚིམ་པར་བྱ། །ཨོཾ་ཨཱཧཱུྃ་གུ་རྒྱ་བྷཀྵ་བྷཀྵ་ཧཾེ། འདི་ནི་བདུད་རྩི་མྱང་བའི་སྔགས་སོ། །ཁྲུས་ཀྱི་དུས་སུ་སྔོན་བཞིན་དུ། །དབང་བསྐུར་ཆོ་ག་སྔོན་བཞིན་བསྒོམ། །ཐུན་བཞི་པོ་ནི་ཐམས་ཅད་དུ། །དེ་ལྟར་བསྒོམ་ཞིང་མཉམ་པར་གཞག། བདག་གིས་རྡོ་རྗེ་མཁའ་འགྲོ་ཡི། །སྒྲུབ་ཐབས་བྱས་པའི་དགེ་བ་གང་། །ཐོབ་བ་དེས་ནི་འགྲོ་བདག་ཉིད། །དེ་བཞིན་བྱེད་པས་མཛེས་གྱུར་ཅིག། སྒྱུ་འཕྲུལ་ཆེན་མོའི་རྒྱུད་ཀྱི་རྗེས་སུ་འབྲང་བའི་ཧེ་རུ་ཀའི་སྒྲུབ་ཐབས་:སློབ་དཔོན་ཀུ་ཀུ་རི་པས་མཛད་པ་རྫོགས་་་་བ་ཡ༼༈:ཊལ(ཎབ་ལ། དེ་བསྡུར་མཆན། ཨེམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»སྒྱུ་འཕྲུལ་ཆེན་མོའི་རྒྱུད་ཀྱི་སྒྲུབ་ཐབས་་་་་བཞུགས། ®«པེ་»«སྣར་སཱ། «པེ་»«སྣར་»ན། «པེ་»«སྣར་པར། ཚུ«པེ་»«སྣར་»ཐུན། ཕྱི«པེ་»«སྣར་གཞག@«པེ་»«སྣར་»སཱ། «སྣར་ཉིད། «པེ་»«སྣར་»བསམ། «པེ་»«སྣར་»བོ། «པེ་»«སྣར་»ལི་ཀི་ར་ན། །«པེ་»«སྣར་»ལ། པོ«པེ་»«སྣར་»ཧེ། «པེ་»«སྣར་»མརྐཏ། «པེ་»«སྣར་»ལོང་།  ི«པེ་»«སྣར་དྷ་དང་ལྕེ། «པེ་»«སྣར་»ཡེ་ཤེས་ཀྱི། «པེ་»«སྣར་ནི། «པེ་»«སྣར་»ཀ@«པེ་»«སྣར་བཀོད། ༄༅«སྣར་»ལས། «པེ་»«སྣར་»པའི་«པེ་»«སྣར་རྩི། «པེ་»«སྣར་བྱུགས། «པེ་»«སྣར་རྩི། «པེ་»«སྣར་»ཨ། «པེ་»《སྣར་»ཀྱི། དང་། །བདག་དང་མ་«པེ་»«སྣར་»ལ། «པེ་»«སྣར་མྱང་། «པེ་»«སྣར་»ཀྱིས། «སྣར་»པད་རྩའི་སྐུད་འདྲའི་རི་མོ་ནི། །ཚུ«པེ་»«སྣར་བདུན། ཚོ«པེ་»«སྣར་དེ་ལྟར། «པེ་»«སྣར་»། ན«པེ་»«སྣར་»བ་ལ། «པེ་»«སྣར་»དགེ། ཁྱེ«པེ་»«སྣར་»འགྲོ་དག«པེ་»«སྣར་»ཨཱ། —།
