བྱུང་། །ས་གདུགས་པདྨ་གྱེན་བསྒྲེང་སྟེ། །ཡེ་ཤེས་རྣམ་ཤེས་གྱེན་དུ་བསྒྲེང༌། །ཨོ་བྷཀྵེ་ཏྲ་པཱ་ལེ་བློ་དྲི་བྱ་ཛgཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། རབ་དང་མ་སོགས་ལྷ་མོ་རྣམས། །སེམས་མ་རྡོ་རྗེའི་ཆོ་གས་བྱ། །ཨོཾ་ཨཱཀྵ་ཙིཏྲ་ཨ་བསྦི་ཏ་ཡོ་གི་ནི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཉི་མ་ཟླ་བ་བརྐྱང་ནས་ཀྱང་། །འཇིགས་དང་རྣམ་ཤེས་ཡང་དག་སྦྱར། །ནག་པོ་དྲག་པོ་ལ་སོགས་པའི། །དམ་ཚིག་ཕྱག་རྒྱུར་འདི་གསུངས་སོ། །ཨོཾ་ཨཱལོ་ཀ་པཱ་ལེ་སྦྱོ་དྲི་ཤ་ཛཧཱུྃ་བཾ་ཧོ། །ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། །པདྨ་དང་ནི་ཡེ་ཤེས་སྦྱར། །ལྷག་མ་གྱེན་གཞག་མདུང་ཐུང་སྟེ། །ཟླ་བའི་ལག་པར་ཐོད་པ་གནས། །རྒན་བྱད་ལ་སོགས་རབ་ཏུ་གྲགས། །ཨོཾ་ཨཱཏྨ་ཤཱ་ན་ཨ་བསྦི་ཏ་ཡོ་གི་ནི་ཛི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟེ། ས་མ་ཡོཉཧཾ། དེ་ནས་ཉེར་སྤྱོད་ལྔ་རྣམས་ཀྱིས། །ཕྱི་རོལ་ལྷ་རྣམས་མཆོད་པར་བྱ། །སྒེག་མོ་ལ་སོགས་རྣམས་ཀྱིས་ཀྱང༌། །དེ་བཞིན་ཉི་ཤུས་མཆོད་པར་བྱ། །འདི་ཡང་སྔགས་ཀྱང་ཉི་ཤུ་སྟེ། །མཆོད་པའི་རྗེས་སུ་རིམ་པ་བཞིན། པི་ཝང་གླིང་བུ་ཟངས་དུང་དང་། །འཁར་རྔ་བསིལ་ཡབ་རྒྱལ་མཚན་དང་། །ཕ་ཊ་ཧ་དང་རྫ་རྔ་དང་། གདུགས་དང་བླ་རེ་བ་དན་དང་། །གར་དང་བཞད་དང་སྒེག་པ་་་དང་། འ། ཁྲོ་དང་དྲག་ཤུལ་འཇིགས་རུང་དང་། །སྙིང་རྗེ་དང་ནི་སྙིང་སྟོབས་རྒྱུན། །རྨད་བྱུང་ཆོས་ནི་བརྟན་པའོ། །སྔགས་འདིས་པི་ཝང་ལ་སོགས་མཚོན། །ཟླ་བ་གྲི་གུག་ལྟར་བྱས་ནས། །པི་ཝང་གྱེན་དུ་རྣམ་པར་བསྒྲེང་། ང་། ཉི་མའི་ལག་པས་ཅུང་ཟད་གསོར། །པི་ཝང་དེ་ནི་སྒྲ་ཡི་མཆོག། ཁ་ཡི་གཡས་ཀྱི་ངོས་སུ་ནི། །ཟླ་བ་ཉི་མའི་ལག་གཉིས་གནས། །དེ་དག་དབྱུག་བཞིན་གྱེན་དུ་གཞག། གླིང་བུ་མ་ཡི་མཚན་ཉིད་འགྱུར། །རྡོ་རྗེ་བཅིངས་པ་བརྟན་བྱས་ནས། །ས་་་གདུགས་པདྨ་རྩེ་མོ་སྦྱར། ལྕོ། ཁ་ཡི་འོག་ཏུ་གཞག་པར་བྱ། །འདི་ནི་ཟངས་དུང་—།
