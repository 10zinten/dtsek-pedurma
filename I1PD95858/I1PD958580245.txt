།ཀཀབསྟན་འགྱུར། རྒྱུད། ཀྐཱ ི་»pམུདང་ཏི་དེ་སེ་སུ་བྷུ་ཛ་ན། །་ཀུ་ཏི་མ་ར་ར་ཧོ། ཎ་ཌི་བུ་ཛི་ནི་བོ་ཏི་ཀཛྫ་ཏི་ཧི། དེ་ན་ཎ་ཧཱུཾ་ཧཱུཾ། ཁྲོ་ཏེ་ན་ན་ཏེ་ཏེ་ཧཱུཾ། ཨ་མ་ཏི། ཨ་ན་ག་གྷ། ཨ་སད་ཏི། བི་མུད་ཏི༑༼སུན་ན་བི་སུན་ན། ་ོནཾ་ས་ན་ས་ར་བཱི་ད་ནི་བཱི། རྒྱུ་ཏེ་ན་ན་ཧཱུཾ་ཧཱུཾ་ཏེ་ན་ན་ཧཱུྃ། ཏེ་ཏེ་ཧཱུཾ་མ་ར་ཕོ་:པ་ན་ཙེ་བ་ཧཱུྃ་བ་ཧ་དེ་ཧཱུྃ། ཚ་སུ་ན་སོན་ས་རེ་ཀ་ཏུ་ཧཱུྃ་ཧའུ་ཛ་ན་པ་ཏེ་ཏེ་ན་ན་ཧཱུཾ། ཚོ། ཏ་དེ་ནས་ཕྱི་རོལ་གྱི་རྣལ་འབྱོར་མ་དཔག་ཏུ་མེད་པས་ཀྱང་། ཅཎྜ་བའུ་ན་གོ་ན་དམ་ཧཱུྃ་ལི་ནི་ཤན་རྟེན་མ། ཌདྷ་བ་ལ་སང་ས་ག་ཚོཎ་ཀ་མ་ནི་ཏི་ན་མ། ཎ་ཨ་མི་ཏ་བི་བི་ཤུདྡྷཱ་སི་ཧི་དྷ་ནུ་:ཀན་ཙ་ཀན་ཙ༄༅། །ན་བ་ར་ཏ་དྷ་ཎཾ་ཀ་ཎ་ཏི་མ་ཧཱུྃ་ཨ་ཏི་བ་ས་གྷོ་རེ་མ་ཎ་ཌི་ཝེན་ཡི། སཛྫ་ཀ་བ་ལེ་སོ་ནི་བཀྟ་ཁེ། ་ཛཱ་ཡ་ཛཱ་ཡ། ་༼ཨུད་དྷ་རི་ཏེ་བྷ་བ་ཧཱུྃ། ཌ་ཌོམ་ར་ག་ཡ་རེ་ཡ། ཌ་སོ་རུ་ཡཧི་ཧཱུྃ། ཀིན་ཏུ་བ་ར་བྷི་ས་ན་གྷོ་རུ་:དན་ཌ། ་་་་་ས་ཛ་བ་ཀ་ལ་མ། །བ་སྱ་མི་ཏེ་བ་ཎ་མི་པ་ཧཱུྃ། ༽སོ་ཧི་ལི་ལུ་ཎི་ཚ་བིཛྫ་ལུ། ཌཁའི་མུ་ཏྲ་ཏྲི་པ། ད་པ་ཧཱུཾ། ་ཌོམཱ་ར་བི་ཤུདྷ་ར་ར་ན་ཀོ་ཏི་ཡོ་ན་མ་ཧཱ་མ་ཧཱ་ཧཱུྃ། རྔ་ཡི་ནི་པ་ཧཱུཾ། ཌཱ་རྡོ་རྗེའི་གླུ་ནི་དེ་ཉིད་དང༌། Pཚོ། རྡོ་རྗེ་དྲིལ་བུའི་རབ་མཆོད་ཅིང་། ་ཌང་། པཉྩ་མ་དང་ཀོ་:ཤེ་ཀར་ཌཎ་། བསྲེས་པའི་མཚན་ཉིད་་བཟང་པོས་བླང༌། ཌ་(། དེ་ནས་རིམ་པ་གསུམ་པ་འཇིག་རྟེན་པ་ལ་སོགས་པ་བསྟོད་་་པར་བྱེད་པ། དཔལ་ལྡན་དེ་ཉིད་རྒྱལ་པོའི་དབང་ཕྱུག་དམ་པ་ཅོད་པན་མཛེས། །གསང་བའི་གདན་ལ་སྐུ་མཆོག་འོད་འབར་དང་ལྡན་ཞིང་། ་ཌ། དཔལ་ལྡན་རྡོ་རྗེ་དྲིལ་བུ་བསྐྱོད་སྒྲས་རྣལ་འབྱོར་མཉེས། །ཁ་། :དཔལ་གྱི་ཡེ་ཤེས་དམ་པའི་དབང་ཕྱུག་ལོངས་སྤྱོད་རྣམས་ལ་སྤྱོད། །།དེ་ལ་སོགས་པའི་བསྟོད་པ་རྣམ་པ་སྣ་ཚོགས་པས་བསྟོད་པར་བྱའོ༑ཌ་ཌོ། རྡོ་རྗེ་ཐལ་མོ་སྦྱར་ནས་ནི། སྙིང་གའི་ཕྱོགས་སུ་གཞག་པར་བྱ། །ཁZ། ཨོཾ་ཤུདྡྷ་དྷཚྭ་ཧཱུཾ། ང་ཌཨོ་ཨ་ཨོ་ཨ་སྭཱ་ཧཱ། z་དེ་ལྟར་མཉེས་པར་བྱས་ནས། ཇི་སྐད་བཤད་པའི་ལྷ་མོའི་འཁོར་ལོ་ཐུན་བཞིའི་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་བསྒོམ་པར་་་་་(།
