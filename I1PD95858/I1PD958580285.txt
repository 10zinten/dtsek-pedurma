རྒྱུད། མིའིཌནི། །དྲི་ཡི་ཕྱག་རྒྱས་བྱུག་པར་བྱ། །ཨོཾ་སུ་:པཱི་ཡ་གནྡྷེ་ཧཱུྃ། གཡོན་པས་གྲི་གུག་ལྟར་བྱས་ནས། །པི་ཝང་རྣམ་པར་གྱེན་དུ་བསྒྲེང་། །ཉི་མའི་ལག་པས་ཅུང་ཟད་བསོརཚེ། །པི་ཝང་དེ་ནི་སྒྲ་ཡི་མཆོག། ཨོཾ་ཧཱུཾ་ཧོས། །ཉི་མའི་ཁུ་ཚུར་བརྟན་བྱས་ནས། །དཀར་པོ་དང་ནི་ཡེ་ཤེས་སྦྱར། །སོར་མོ་ལྷག་མ་གྱེན་དུ་བརྐྱང༌། །མེ་ཏོག་ཕྱག་རྒྱར་རབ་ཏུ་གྲགས། །ༀ་ཤྲཱི། ཉི་མའི་ཁུ་ཚུར་བརྟན་བྱས་ནས། །ཧཱུྃ་གི་ཡི་གེ་སྒྲོགས་བྱེད་ཅིང་། འོ། ཁ་ཡི་རྩེ་མོར་དཀར་པོ་གཡོ། །གླུ་མའི་ཕྱག་རྒྱར་རབ་ཏུ་གྲགས། །ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཉི་མའི་ཁུ་ཚུར་བརྟན་བྱས་ནས། །དུ་བ་གྱེན་གཞག་ཅིབདུག་སྤོས་མ། །ཨོ་ཨི། །གཡས་པའི་རྐེད་པའི་ཕྱོགས་སུ་ནི། །ལག་པ་གཡོན་པ་འཕང་པར་བྱ། །ཉི་མའི་ལག་པ་གྱེན་དུ་སྟེ། །གར་མའི་རྣམ་པར་་་་་ལེགས་པར་གནས། །ཨོ་སུ་ཀྲྀ་ཏེ་ཧཱུྃ། ལྔ་ཉི་མའི་ཁུ་ཚུར་གྱེན་དུ་གཞག་ཚ་། དེ་ཡི་མཐེ་བོང་གྱེན་དུ་བསྒྲེང༌། །འཇིག་རྟེན་ཀུན་ནས་རབ་གསལ་བྱེད། །མར་མེ་ཡི་ནི་ཕྱག་རྒྱུར་འགྱུར། ་ཀུན། ཨོ་ཀླུ། འཇིགས་པ་དང་ནི་རྣམ་ཤེས་སྦྱར། །ཟླ་བའི་ལག་པས་རྩེ་གསུམ་སྟེ། །ཀེང་རུས་འཇིགས་བྱེད་ལ་སོགས་པའི། །ལས་ཀྱི་ཕྱག་རྒྱར་རབ་ཏུ་གྲགས། བ་། ཨོཾ་ཀཾ་ཀ་ཐལ་མོ་གཉིས་ཀ་བཅིངས་ནས་ནི། །ཐོད་པ་བཞིན་དུ་རྣམ་པར་དབྱེ། །ལྷ་མོ་པདྨའི་རྣམ་པ་སོགས། །ཏི། ལས་ཀྱི་ཕྱག་་རྒྱར་ཡང་དག་བརྗོད། །ཨོཾ་ཀཾ་ཀཾ། ཨོ་ཁཾ་ཁཾ། ཨོ་གོ་གཾ། ཨོཾ་གྷོ་གྷོ། །ཨོ་ཙོ་ཙཾ། :ཨོཙྪ་ཙྪ༄༅། ཨོཾ་ཛ་ཛཾ། ཨོཾ་དྷཾ་དྷཾ། ཨོ་ཊི་ཊི། :ཨོ་ཊྛི་ཊཱི། ཨོ་ཌི་ཌ༑ཨོ་ཌྷ་རྣོ། ཨོ་ཏཾ་ཏཾ། ཨོ་ཐཾ་ཐཾ། །ཨོཾ་དཾ་དཾ། ཨོ་:དྷི་དྷི། ཨོ་པི་པི། །ཨོ་ཕོ་ཧཱུཾ། :ཨོ་བྲོ་བཻ། ཨོ་པཱི་བྷཱི། ཉི་མས་དབྱུག་པ་གྱེན་དུ་གཞག། གཡོན་པས་མདུན་ཉིད་དུ་ནི་སྦྱར། ༄༅། །ཨོ་ཁེ་ཤ་ཧཾ། ཐལ་མོ་སྦྱར་བ་བཅིངས་ནས་ནི། །པདྨ་བཞིན་དུ་རྣམ་པར་དབྱེ། །ༀ་ལཾ་ཐཾ། ཐལ་མོ་སྦྱར་བ་བཅིངས་ནས་ནི། །བསྒྲ།
