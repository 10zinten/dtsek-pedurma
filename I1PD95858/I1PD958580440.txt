བྱིང་ཚེ་བས་གནས་སུ་ཁྲོ་ཁ། ཅ་ལ་བློ་ཚ་བ་ཡིན། མེ་འོད་འོག་ནས་ད་ཡོད་པ་ཆ། ཡི། །ཀ་བ་ཆེ་བས་གྲུམ་པའི་ཁ་:ལ་མ་རམས་པ་ཁོག་པ་ལྟ་བུ་དོ་ཝཊ། ::ཀ༌ཞུ་བ་ཁག་ལ་ཚོད་ནི། །ཀ་པ་ག་བ་ང་མར་འབབ་དང་ཚ་མིང་ཁ་ཏུནོ་སྦེ་ལས་ཁགོ་ཁར་ནི་ཞི་བ་སྐྱེ་གནས་བར་བྱ། །བློ་དང་། ཀ་ཆ་མི་ནི་ཡ་ང་ཚོལས་༄༅། །དཔལ་གདན་བཞི་པའི་ཆུའི་TiếngvềvụPHỤpsTRsnཔ་ལ་འགྱུར། ལ་ལ་ཤི་ཡང་། །བྱ་དང་ད་ཚོད་ལུ་མ་དང་བཅས་པ་དང་། ཀླུ་ཀརྒྱུད། བ་ལས་༼བོད་གནང་པ་དང་ཁ་ཡི་ཆ་ལྟེ་གྲལ་ཀུན་ལ་རྒྱབ་ན་་་དཔོན་དང་། ཡན་། བྲོག་བྱེ་བ་ནས་ཁྱེད་ལ་དགའ་བ་སྐྱེ་སྐྱེ་བ་དང་ར་ཤི་ཀཔ་་་་༄༅། །བཅོམ་ལྡན་འདས་ཡེ་ཤེས་མཁའ་འགྲོ་མ་ལ་ཕྱག་འཚལ་ལོ། །ཆུ་ཡི་སྦྱིན་སྲེག་རིམ་པ་ཞིག། འགྲོ་བ་སེམས་ཅན་དོན་དུ་ནི། །།ཅུང་ཞིག་བསྟན་པར་བྱ་བ་ཡང་། །ཁྱད་པར་བསྒྲུབ་པའི་དོན་དུ་བཤད། །།ཆུ་ཡིས་སྦྱིན་སྲེག་ཇི་ལྟར་་་་འགྱུར། །སྒྲུབ་པ་པོ་ནི་ཐེ་ཚོམ་:སྐྱེས། །།མེད་ཀྱང་སྡིག་སྲེག་སྡིག་པ་སྦྱོང་། །:ཆུ་ཡང་དེ་བཞིན་འདྲ་བས་ན། །དེས་ན་སྦྱིན་སྲེག་ཅེས་བྱའོ། །ས་གཞི་ལ་སོགས་མཚན་ལྡན་པར། །ཤིན་ཏུ་ཡངས་ཤིང་གྲུ་བཞི་ལ། །ཀུན་ཏུ་དྲིས་བྱུག་མཎྜལ་་་སྟེང་། །བླ་རེ་ལ་སོགས་རྒྱན་རྣམས་བྲེ། །དབུས་སུ་ཆུ་ཡི་སྣོད་སོགས་ལ། །དྲི་ནི་དཀར་པོས་བྱུགས་ནས་ནི། །བུམ་པ་སྐྱོན་བྲལ་ཡོན་ཏན་ལྡན། །དབྱིབས་ནི་གྲུ་བཞིའམ་ཟླུམ་པོ་ལ། །ཐབ་ཁུང་ལྟ་བུའི་དབྱིབས་ཅན་དེ། །བཅུ་བཞི་ཁ་ཚད་སུམ་གཉིས་ཟབས། །ཆུ་དང་འོ་མ་བསྲེས་པ་ཡི། །དྲི་ཞིམ་ཆུ་ཡིས་དགང་བར་བྱ༑། ཁ་ནི་ཤིང་ལོ་སར་པས་བརྒྱན། །ཡང་ན་བུམ་པ་བརྒྱད་སོགས་ཀྱིས། །།
