རྒྱུད། ཡINདཔལ་རིན་ཆེན་འབྱུང་གནས་ཞི་བའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །ཡང་། །མོ་ལ་པཎྜི་ཏ་ཆེན་པོ་ཞི་བ་བཟང་པོ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་འགོས་ལྷས་བཙས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།་ཉི་ངོ། །།ཁྱོ་གས་ང་ན་ཚེ་ཁ་རགས་ས་ཚེ་ཎཊ་ཏའི་ཇི་ལ་༼z་ཁཛི་ཀློངདོད་དེ་འཚPSཁེ་ལ་ཀུཀ་ཀོཔ་ཚོཨོ་ཨཧཱི༔ཧཱུཾ་ཕཊ། བཟླས་པའི་སྔགས་སོ། །།ཤྲཱི་ཧེ་རུ་ཀ་མ་ཧཱ་བཛྲ་ཨཱ་གཙྪཡ། པྲ་བེ་ཤ་ཡ་པྲ་བེ་ཤ་ཡ།  ོ་བནྡྷ་ཡ་བནྡྷ་ཡ། ཏོ་ཥ་ཡ་ཏོ་ཥ་ཡ། སརྦ་ཌཱ་:ཀི་ནཱི་ནཾ། ཧྲི་ད་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་བུདྡྷ་ཌཱ་ཀི་ནཱི་པྲ་ཏཱིཙྪ་མཾ་བ་ལི་ཧཱུྃ་ཨོ། ཨོ་བཛྲ་ཌཱ་ཀི་ནཱི་པྲ་ཏཱིཙྪ་ཡཾ་བ་ལི་ཧཱུཾ་ཧཱུཾ། ་་ཨོཾ་རཏྣ་ཊཱ་ཀི་ནི་པྲ་ཏཱིཙྪ་མཾ་པ་ལི་ཧཱུྃ་སྭཱ། །ཨོཾ་པདྨ་ཊཱ་ཀི་ནི་པྲ་ཏཱིཙྪ་མཾ་བ་ལི་ཧཱུྃ་ཨཱ། ཨོ་བི་:ཤྭ་ཎཱཀི་ནི་པྲ་ཏཱིཙྪ་མཾ་བ་ལི་ཧཱུཾ་ཧཱ། །ཨོཾ་སརྦ་ཌཱ་ཀི་ནི་ཡོ་གི་ནི་པྲ་ཏཱིཙྪ་མཾ་བ་ལི་ཧཱུཾ་ཧ་ཧ་ཧ་ཧ་ཧི། དེ་ལྟར་ན་གཏོར་མའི་ཆོ་གའོ། །དཀྱིལ། །ཚ་ཙ་ར་མ་ཎཁྲོད་ཀྱི་ད་ནི། རྐང་ལག་སྟེ་ལྔ་ལ་རྗེ་བོད་། །ཏང་ང༌། །།ཚོ་་ཁཆུ་ཁའི་བློན་པཎ། ཙ་ཊ༌ཝའོ། །ཀ་ཁན་། ང་ཌར་}རྒྱལ་ཟླའི་ཉཀ་ལ»ང་ཁ་ངོ་བ། ཎངཁ། ངོ་ཊི་པ་། ལ་ལརྣང་ཁཆ/ཎ་ཚུལ་ཁ་ཁོབསྡུར་མཆན། ཁ་ཎི་དབྱི་དང་བློ་གྲོས་པ། དེ་ཚཚོ། །།སྒོ་གཔདེ། །ས་སྐྱཎྜལ་། མཚན་ཁ་ལ། དེ་དག་ཁྲོ། ངན་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང༌། «པེ་《སྣར་»དཔལ་སྒྱུ་འཕྲུལ་ཆེན་མོའི་སྒྲུབ་ཐབས་་་་་གསལ་བཞགས༌པེ་དརྗེ། «པེ་《སྣར་»བཟས། བ:ཚཁོ༌། (《པེ་《སྣར་»དང་། Y?(«པེ་《སྣར་»ལ། («པེ་»«སྣར་»དེ། ་《པེ་»ཤུ། (《པེ་»《སྣར་»ཨཏྨ་ཀོ་ཧཾ། པ《པེ་»ཉིད། ལམ་དེ་དེད། རྗེ་དྲི། །c༽་(«པེ་»«སྣར་»བསལ་བར། («པེ་»«སྣར་སཱ། །«པེ་»«སྣར་ཧ། «པེ་《སྣར་»ལྕེ། ་«པེ་《སྣར་»ཏིའུ། «པེ་《སྣར་»འདི་ཡིས། ་[ཎཎྜིང་ཊཏིཊ་ཀྱི*ང་རྗོད་རྒྱུ་ལས་ནང་«པེ་»«སྣར་ཀྱིས། («པེ་»«སྣར་འགྱུར། («པེ་»+དཔལ། »«པེ་»«སྣར་»དབུས། ས«པེ་»«སྣར་»དཀྲུགས། «པེ་»ཀྱི། («པེ་»«སྣར་»འདིར། །«པེ་»«སྣར་བྱ་བས། /《སྣར་བཞུག«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»ཀྱི། («པེ་»།
