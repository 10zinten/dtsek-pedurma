རྒྱུད། ༼ཧོ། ས་མ་ཡ་ཧཱུཾ། ་ས་མ་ཡོཉཧཾ། །།ཨོ་:ཆ་ཀ་ཀོ་ལཱ་ཡ་ཧ་ལ་དྲྀ་ཤྱ་ཛཧཱུྃ་བོ་ཧོ༑ཡ་ས་མ་ཡ་ཧཱུཾ། །ས་མ་ཡོཉཧཾ། །?། ཨོ་རྙཀྐ་ལཱ་ལ་སཱི་ཡེ་དྲི་བྱ་ཛཧཱུྃ་བོ་ཧོ༑་ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་་ཨོཾ་ཏཾ་ཀ་པྲ་ཡཱ་ཡ་དྲི་ཤུ་ཛཧཱུྃ་བཾ་ཧོ། །ས་མ་ཡ་ཧཱུྃ། ཅ་ས་མ་ཡོཉཧཾ། ཨོ་ཏ་ནུ་ཛཾ་གྷ་ནཱི་ཡེ་དྲྀ་ཤྱ་ཛཉྩཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟེ། ས་མ་ཡོཉཧཾ། ཨོ་ཐོ་ར་བཱ་ཧུ་ཀཱ་ཡ་ཛི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ཏཱ་ས་མ་ཡ་ཙཱ༑ས་མ་ཡོཉཧཾ། ཨོ་ཐཱ་:ར་ནཱ་དི་ནི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ་ཊི་ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་ཚེ་ཨོ་:དཾ་ཀ་རཱ་ལི་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ཊོ་ས་མ་་ཡོཉཧཾ། ཨོ་ད་ར་ད་ར་ཤ་ནཱི་ཡེ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོ་དྷི་ར་གྷོ་ཎ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟེ། རྒྱུས་མ་ཡོཉཧཾ། ཨོ་དྷི་ར་གྷ་ཏི་ནཱི་ཡེ་ཧཱུཾ་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་་ཨོ་:པིཎྜ་ཙ་ར་ཡ་དྲི་བྱ་ཛཧཱུྃ་བོ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་ཨོ་པི་ག་ལགྣི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཙཱ། ཟེ་ས་མ་ཡོཉཧཾ། ་ཨོ་ཕེ་ན་མུ་ཁཱ་ཡ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། །ས་མ་ཡོཉཧཾ། ཨོ་ཕུལླ་ན་ཡ་ནཱི་ཡེ་དྲི་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ?ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་ཨོ་བྲཧྨ་དྷཱ་ནཱི་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། །ཨོ་བན་དྷུ་:ན་གཱ་ཏྲི་ཡེ་དྲྀ་ཤྱ་ཛ་ཧཱུཾ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོ་བྷི་མ་ནཱ་ད་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་ཨོ་བྷ་མ་ར་ཀེ་ཤཱི་ནི་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། །ཌཱ་ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཌ་རྡོ་རྗེ་བསྡམས་པ་བརྟན་བཅིངས་ནས། །འཁོར་ལོའི་རྣམ་པར་བརྟན་པར་བྱ། །ཨོཾ་བཛྲ་ན་:ར་ཡཱ་ན་དྲྀ་ཤྱ་ཛཧཱུྃ་བོ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོཾ་བཛྲ་ལགཱི་ཡེ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། །ད་ཨོ་བཱ་རཱ་ཧཱི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ཊ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཐལ་མོ་སྦྱར་བ་བཅིངས་ནས་ནི། །མགོ་བོའི་སྟེང་དུ་གཟུང་བར་བྱ། །(།
