ཁ་སྤུ་དཀར་བསྟན་འགྱུར། རྒྱུད། །།:རྒྱས་པ་བདག་གིས་བྱས་པ་ཡིན། །བསོད་ནམས་འདི་ཡིས་འཇིག་རྟེན་རྣམས། །དགའ་བའི་བདེ་བ་ཀུན་རྙེད་ནས། །ཡེ་ཤེས་ལྡན་པའི་ས་གནས་སུ། །ཐམས་ཅད་དུ་ནི་འགྲོ་བར་ཤོག། སངས་རྒྱས་ཐོད་པའི་རྒྱུད་ཀྱི་དཀའ་འགྲེལ་ཡེ་ཤེས་ལྡན་པ་ཞེས་བྱ་བ་དངོས་གྲུབ་བརྙེས་པའི་སློབ་དཔོན་ཆེན་པོ་ས་ར་ཧའི་ཞལ་སྔ་ནས་ཀྱིས་་་་མཛད་པ་རྫོགས་སོ། །།།།།ཁྲི་ཤོམ་ཞུ་ལཊ་བཀཀ་ཀྱི་ཁ་ཕྱོགས་ཚེ་ཁཁོརྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་གཱ་ཡ་དྷ་ར་དང༌། བོད་ཀྱི་ལོ་ཙཱ་བ་གྱི་ཇོ་ཟླ་བའི་འོད་ཟེར་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །འའོ་ཏུ། །ང་ད་པ། ཚོགས་བསྐོ་བླ་མའི་ཞབས་ལ་གུས་བཏུད་གསོལ་བཏབ་སྟེ། །མན་ངག་འགྲེལ་པའི་དཀའ་འགྲེལ་ཡེ་ཤེས་ལྡན། །སྒྲ་དོན་སྡེབ་སྦྱོར་འབད་དེ་བསྒྱུར་བྱས་པས། །།འགྲོ་ཀུན་གཉིས་མེད་ཡེ་ཤེས་ལྷུན་གྲུབ་ཤོག། །བའི། འ་ཁྲི། ཋ་ཌ་ད་བྱ་དང་པས་མི་ག་འདུལ་བར་ཞུས། ཞེན་པ་ནི་བླ་མའི་ཆོ་གདང་། རས་དེ་རུ། །བླ་མར་དད་ཞལ་པའི་དྲིན་གྱིས་གས་བར་བྱེད་THUHOẠCHHOTBESUREITHEONHUTHUẬTTRỰCམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་དཔལ་སངས་རྒྱས་ཐོད་པའི་རྒྱུད་ཀྱི་དཀའ་འགྲེལ་བཞུགས་སོ། །«པེ་»«སྣར་པཎྜཱུ་ཀན«པེ་»«སྣར་»སྟེར། «པེ་»«སྣར་»བདུན། «པེ་»«སྣར་བརྗོད། །«པེ་»«སྣར་གྱི། «སྣར་»མེད། «པེ་»«སྣར་»གི། བ«པེ་»«སྣར་»ཁཾ། ཚུ«པེ་»«སྣར་»ཡ། ་«པེ་»«སྣར་སྦྱོར། ཏོ:ཀ:/རི་འཚོ་བྱེད་རྒྱུ། «པེ་»«སྣར་»དག་(«པེ་»«སྣར་སོ། «པེ་»«སྣར་བཟུང་། ར། ང་། ཀྱི«པེ་»«སྣར་»བལྟ། ®«པེ་»«སྣར་ཐམས་ཅད། «པེ་»«སྣར་»+ཀྱི། «པེ་»«སྣར་+ཡིན་ལ་ཆེན་པོ་ཡང་། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»དཔའ། «པེ་»«སྣར་»དང་། «པེ་»«སྣར་»ཅེས། ༄༅«པེ་»«སྣར་བཅུགས། །_།
