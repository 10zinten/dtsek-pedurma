རྒྱུད། མིའི་སངས་རྒྱས་ཀུན་གྱི་བདག་ཉིད་མཆོག། དབྱུག་པ་འཇིག་རྟེན་དབང་ཕྱུག་་་་མགོན། །གྱ་གྱུར་རྡོ་རྗེ་འཛིན་པའི་གཙོ། །།ཡན་ལག་དམ་ཚིག་རྡོ་རྗེ་སྟེ། །མགོ་ནི་རིན་ཆེན་རྒྱལ་པོ་ཡིན། །ཐིག་ལེ་སྒྲ་དང་བཅས་པ་ནི། །སངས་རྒྱས་གཙོ་བོ་ཞེས་བྱར་གྲགས། །ཟླ་བའི་ཟེར་ལྟར་དྲི་མེད་པའི། །རིགས་ཀྱིས་ཁྱབ་པའི་འོད་ཟེར་འཕྲོ། །གལ་ཏེ་སྣ་ཚོགས་སྒྲུབ་（འདོད་ན། ཚོ། ཤེས་རབ་ཅན་གྱིས་འདི་བསམ་བྱ། །དེ་ནས་དྲོད་ཀྱི་གསུམ་པ་དང༌། །བརྒྱད་ཀྱི་མཐའ་དང་བདུན་པ་ཡི། །དང་པོ་དེ་བཞིན་དྲོད་པའི་མཐའ། །སྒྲ་དང་བཅས་པས་ཡང་དག་བརྒྱན། །ཤར་གྱི་ཕྱོགས་སུ་མཁའ་འགྲོ་མ། །དེ་བཞིན་བྱང་དུ་དྲག་མོ་བསྒོམ། །ནུབ་ཀྱི་ཕྱོགས་སུ་རོ་ལངས་མ། །ལྷོ་ཕྱོགས་མཁའ་འགྲོ་གཏུམ་མོ་སྟེ། །སྔོན་དུ་བསྟན་པའི་མདོག་བཞིན་དུ། །ཕྱོགས་ཀྱི་འདབ་མར་བསྒོམ་པར་བྱ། །མཚམས་སུ་ཁ་དོག་འདྲེན་མ་རུ༑། བརྩེགས་པའི་ས་བོན་བཞི་པོ་བསམ། ར། ཡན་ལག་བརྒྱད་ཀྱི་གོ་ཆ་གང་། །།སྔགས་དེ་ཉིད་ཀྱིས་རྣལ་འབྱོར་ལྡན། །འོད་ཟེར་འབར་བའི་ཕྲེང་འཁྲུག་ཅཔའི། །སྔགས་དེ་དག་ནི་བལྟ་བར་བྱ། །ལས་བཞི་ཡི་ནི་གནས་སྐབས་ལ། ཟགྱི། རྗེས་སུ་མཐུན་པའི་མི་ཤིགས་བསྒོམསི། །དེ་བཞིན་ལས་བཞི་སྒྲུབ་བྱེད་པས། །ལུས་ཀྱང་ཁ་དོག་བཞི་རུ་དགོད། །ཞལ་གཅིག་ཕྱག་གཉིས་ལ་སོགས་པ། །ཞལ་ཕྱག་ཇི་སྙེད་འདོད་པར་བྱ། །:དྲག་ལ་ཁྲོ་བོར་བསྒོམ་པར་བྱ། །ཞི་བ་ལ་ནི་ཞི་བའི་ལུས། །རླུང་དང་མེ་ཡི་སྟེང་དུ་ནི། །ཉི་མ་ལ་གནས་དྲག་པོར་བྱ། །ས་ཆུ་ཟླ་བ་ལ་གནས་ཏེ། །།ཞི་བའི་ལས་ནི་བསྒོམ་པར་བྱ། །ཨོཾ་སུཾ་ཧཱུཾ་ཡཾ་ཧཱུཾ་ཧཱུཾ། ཨཽ་དྡྷཾ་སྨིཾ་ཧཱུཾ། །ཕུ་ཕོ་ཕྲོ་བྲི། དཀྱིལ་འཁོར་གྱི་ས་བོན་ནོ། །ཨོཾ་ཧཱུཾ་ཊ་ཊ་དྷ་ར་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕུ་ཕུ་ཕུ། ཨཱ་ཨཱ་ཨཱ་ཕཊ་ཕཊ་ཕཊ། ཀླུ་བརྒྱད་ཀྱི་དུག་དབྲོག་པའོ། །ཨོ་སི་ཀྲི་ཧཱུཾ་སྭཱ་ཧཱ། གྲོང་ཁྱེར་དཀྲུགས་པའིའོ། །ཨོ་བཏྟཱ་ལི་རཀྵ་རཀྵ་ཧཱུྃ་ཧཱུྃ་ཕུ་རུཾ་ཕཊ། དམ་ཚིག་སྲུང་བའི་——།
