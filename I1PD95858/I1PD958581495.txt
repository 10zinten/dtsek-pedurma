བསྟན་འགྱུར། རྒྱུད། རི་::ཀ་བ་ཟླ་ཤ་་་་་་་་་་ཀུ་བར་པ། །ཊི་ད་ག་ཤེས་ལ་འཁོབ་སྒོམ་པ། :ཀ་དང་ཏིག། དེ་ལ་ཏ་ཀ་བསྡུར་མཆན། ་་་་་་་་་་ཎ་ཊ་དེ་ཁང་པ་ར:ང་ད:ན་བ་མ་འུད་་་་SE། །ར་། །།པ་ག་ང་གང་དུ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་འབྱུང་པོ་ཐམས་ཅད་ཀྱི་གཏོར་ཆོག་བཞུགས་སོ༑། «པེ་»«སྣར་»གླུ། ཉེ«པེ་»«སྣར་»ལི་བི་དྷི། «པེ་»«སྣར་བརྟན། ཚ«པེ་ད«སྣར་སྤོང་། «པེ་»«སྣར་ན་གནས། ཚ«པེ་»«སྣར་བསམ། ་་། །འདུ་བ་ག་ནས་ཀློག་«པེ་»«སྣར་སུ། «པེ་»«སྣར་རྗེ། «པེ་»«སྣར་རྩི། «སྣར་བརླབ། «པེ་»«སྣར་གྱུར། «པེ་»དྲི་ཤྩ། «པེ་»«སྣར་»—ཨཨ། «པེ་»«སྣར་»སྤྲོས། «པེ་»«སྣར་»ལ། ཁྱེ«པེ་»«སྣར་»མཆེ། ཀྱེ«པེ་»«སྣར་»ཀྱི། ཉེ«པེ་»«སྣར་»གྱི། ་་་«པེ་»«སྣར་བསམ་སྟེ། «པེ་»«སྣར་ལི་གྲིཉྩ་གྲིཙྪ་གྲཉྫཱ་པ་ཡ་གྲི། «ཅོ་ལི་གྲོཙྪ། གྲོ༔་་་གྲིཉྩཱ་པ་ཡ་གྲིནྡྷུ། ཡེ«པེ་»«སྣར་ཙེ། ཚེ«པེ་»«སྣར་ན་སརྦ་མ་ར་ཡ་མཱ་ར་ཡ། འཁྱེ«པེ་»«སྣར་མ་ར་ལ་དུ། བཀག་ཟློས་ག་ཀཱདིང་རི་ཉིལ་«པེ་»བིནྡྷ། «སྣར་བིགྷཱ། Q«པེ་»«སྣར་རཀྵ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»ཛྷ་ན་ས་ར༑«ཅོ»ག་ཏེ་ན། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་»བྱུང་། «པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་»བདེར། «པེ་»«སྣར་གྲོལ། ཁྱེ«པེ་»«སྣར་བརྒལ། འགྱེ«སྣར་»བ། །ས་ཁང་དུ་སྐབས་བན་ཁོ། ས་ག་འདཔད་:::དག་བསྡུར་འབྲས་རེའུ་མིག་ནས་ཀྱ:། ལ་ལ་ཀོབ་ས་བ། བརྒྱུད། ་་་:/ཅཡོ་འཁཞ་དཀའོ་པེ་ཅིན། །དཔེ་གཞི། སྡེ་དགེ། སྣར་ཐང་། སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བབ་ན་ན་ན་ན་ན་ན?ཕྲེང་། །ལ། །།ཚིག་རྐང༌། །།ར་ཚེག་ཁྱིམ། །།།།།
