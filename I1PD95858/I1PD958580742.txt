བོན་ལ་བསྡུས་ཏེ། སྣང་སྟོང་དབྱེར་མེད་བྱང་ཆུབ་ཀྱི་སེམས་སུ་མྱོས་པར་བྱས་་་ལ་:མཉམ་པར་བྱའོ། །དེ་ལྟར་ཉིན་དང་མཚན་དུ་དངོས་གྲུབ་ཀྱི་མཚན་མ་ཉེ་་་་བར་ཐོབ་པ་དེ་སྲིད་ཀྱི་བར་དུ་བསྒོམ་པར་བྱའོ། །སྒྱུ་འཕྲུལ་ཆེན་མོའི་སྒྲུབ་གཞུང་གི། སངས་རྒྱས་མཁའ་འགྲོའི་སྒྲུབ་ཐབས་འདི། །བདག་གིས་དད་པས་བྱས་པ་ལ། །མཁས་པ་རྣམས་ཀྱིས་བཟོད་པར་མཛོད། །སྒྱུ་འཕྲུལ་ཆེན་མོའི་སྒྲུབ་པའི་ཐབས། །བྱས་ལས་བསོད་ནམས་གང་སྐྱེས་ལས། །བདག་དང་ཁམས་གསུམ་་་སེམས་ཅན་རྣམས། །རྡོ་རྗེ་འཛིན་གྱི་ས་ཐོབ་ཤོག། སངས་རྒྱས་མཁའ་འགྲོ་་་མའི་སྒྲུབ་ཐབས་སློབ་དཔོན་ཆེན་པོ་ནག་པོ་པས་མཛད་པ་རྫོགས་སོ། །།།གྱི་བསྡུར་མཆན། དེ་ལ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་སངས་རྒྱས་མཁའ་འགྲོ་མའི་སྒྲུབ་ཐབས་་་་བཞུགས། «པེ་»«སྣར་»ཀཱི་སཱ། «པེ་»«སྣར་དཔལ། ®«པེ་»«སྣར་»གིས། «པེ་»《སྣར་»ཨོ། དེ་ནས་པ། («པེ་»«སྣར་བ་ཨp། («པེ་»«སྣར་»བསྒོམས། («པེ་»«སྣར་»བསྲུང་། «པེ་»«སྣར་»བྱའོ༑«པེ་»«སྣར་བསམ། «པེ་»«སྣར་»དོན། «པེ་»«སྣར་ལྕེ། «པེ་»«སྣར་མོ། «པེ་《སྣར་པའོ། (གྱེ«པེ་《སྣར་»། «པེ་»«སྣར་ཏེ། །(«པེ་»«སྣར་»པ། ®«པེ་»《སྣར་»+ལ། «པེ་»«སྣར་»ཀཱི། «སྣར་»ཧྲི། «པེ་»《སྣར་བྱས། «པེ་»«སྣར་»ལས། (པེ་»«སྣར་+ཙམ། །(«པེ་»«སྣར་»མོས། «པེ་《སྣར་ཉལ་བར། «པེ་»«སྣར་+མ། «པེ་»«སྣར་གི། །བ«པེ་»སྣར་»+པའི། །ད་བཡོང་ལྡན་ལ། །སྒོ་ལ་འབྱོར་བས་པ་དེ་བྱས་པ་དང་། བདེ་ལ་སྦྱར་ཏེ་བའི་ལས་བྱའོདུས་ཁྱོད་ལ་——།
