མི་འཆད་པར་གྱིས་ཤིག་པའོ། །དེ་ཁོ་ན་ཉིད་ཀྱི་མན་ངག་བྲིས་པ་ལས། དགེ་བ་གང་ཞིག་ཡང་དག་པར་ཐོབ་པའི་རྩ་བ་བསྔོ་བ་ནི། དེ་ཉིད་བསྟན་པ་ལེགས་་་་བྲིས་ལས། །བདག་གིས་དགེ་བ་གང་བསགས་པ། །དེས་ནི་སེམས་ཅན་མ་ལུས་ཀུན། །དེ་ཉིད་རིན་ཆེན་རབ་གསལ་ཤོག། ཅེས་གསུངས་སོ། །དེ་ཁོ་ན་ཉིད་ཀྱི་མན་ངག་གི་འགྲེལ་པ། སློབ་དཔོན་བློ་བཟང་སྙིང་པོའི་ཞལ་སྔ་ནས་་་་་མཛད་པ་རྫོགས་སོ། །།།་་་་་་་་་་་་་་་་་་་་་རྒྱ་གར་གྱི་མཁན་པོ་ཀ་མ་ལ་གུཔྟ་དང་ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ནས་གཏན་ལ་ཕབ་པའོ། །་། ་་་་་་འིཁཀ《《། མ་ཎའི་རི་ཎའི༽༼། ཀུ་བ་ཀྱི་བློ་ལ་ནི། །།བསྡུར་མཆན། །བ་ལྔའི་ཎ་ཁ《་མ་་་རྒྱལ་པོཔོད་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»དེ་ཁོ་ན་ཉིད་ཀྱི་མན་ངག་གི་འགྲེལ་པ་་་་་བཞུགས། ®«པེ་»«སྣར་བྲི་ཏ་ཏི། «པེ་»«སྣར་»ལྟ། «པེ་»«སྣར་མཐའོ། མོ«པེ་»«སྣར་»ཞེས། ་«པེ་»«སྣར་»ངོ༌། «པེ་»«སྣར་ཆུས་སྐྱེད་དེ། ཉི«པེ་»«སྣར་»ཞེས། «པེ་ཀྱི། །«པེ་»«སྣར་»བར། ཕྱི«པེ་+དེ། «པེ་»«སྣར་ལྟ། @«པེ་»«སྣར་»ཡེངས་པར། ལྟ། ཁོ«པེ་»«སྣར་»ལྟ། ལུ«པེ་»«སྣར་»+ནི། ེ«པེ་»«སྣར་ཆད། རྩེ«པེ་»«སྣར་»བའི། །«པེ་»«སྣར་»མ། ཚེ«པེ་»«སྣར་དག་པར། «པེ་»«སྣར་པ། «པེ་»«སྣར་»+ནི། «པེ་»«སྣར་»དགའ། ི«པེ་»«སྣར་+ལ་སོགས་པ། ༄༅«པེ་»«སྣར་»འཕྱན། །«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་བསྐྱེད། Q«པེ་»«སྣར་པའོ། །«པེ་»«སྣར་པ་སྣང་། གྱི«པེ་»«སྣར་»ཞེ་ན། །««པེ་»«སྣར་»པའི་སྦྱོར་བ། ཉི«པེ་»«སྣར་གཞུགས། «པེ་»«སྣར་མཁར། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་»དབུས། @«པེ་»«སྣར་»+བ། «པེ་»«སྣར་»ཀྱི། «པེ་»ལོ་ལྔ《སྣར་ཀྱི། །ཁྱོ་ཀ་ཁ་ཏོག། བག་ཏུ་གནས་པར་བ་དང་། །བ་།
