།དང༌པ་ཚོ་སྐུ་དབྱེ་རྒྱུ་ཆཉམ་ཀ་མཚོབསྟན་འགྱུར། །རྒྱུད། མ་ཀཋ་ཌལ་ཀཚན་ཁཅུ་རྗེས་སུ་སྒྲུབ་པ་བཤད། །།།ཅེས་པ་ཡིད་བདེ་:བ་སྟེ་ལྷན་ཅིག་ྋསྐྱེས་དགའོ། །།ལྷ་མོ་ཀུན་ལ་བདེ་བླུགས་པས། །ཞེས་པའི་བློ་ཀ་གཅིག་གིས་དགའ་བྲལ་གྱི་བཅུད་ཀྱིས་ལེན་གྱི་བདེ་བ་སྟེ་དགའ་བྲལ་བཤད་དོ། ཟཞིཥ། དེ་དུས་ནམ་གྱི་དགའ་བྲལ་ཞེས་་་་་གདགས་ཤེ་ན། །ཐམས་ཅད་ལྡང་བ་ཀུན་འགྲོ་བ། འོ། ཞེས་པའི་བློ་ཀ་གཅིག་གིས་དགའ་བྲལ་ཞེས་པའི་སངས་རྒྱས་ཀྱི་བརྡ་བཤད་དོ། །འོ། གལ་ཏེ་དགའ་བྲལ་དུ་མ་སོང་ན་མཆོག་དགའ་ལྷན་ཅིག་སྐྱེས་དགའི་རྗེས་ལ་ཅི་ཞིག་སྒོམ་སྙམ་ན། །དངོས་གྲུབ་་་ཆེན་པོ་ཀུན་གྱི་མཆོག། ཅེས་པའི་བློ་ཀ་གཉིས་ཀྱིས་བདེ་བ་ཆེན་པོ་ཞེས་བྱ་བ་དེ་་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་བརྡས་དངོས་གྲུབ་ཆེན་པོ་དང་། ཀུན་གྱི་མཆོག་དང་། དབང་ཕྱུག་ཆེན་པོ་དང་། །བདེ་མཆོག་དང༌། མཉམ་སྦྱོར་དང༌། སྦྱོར་བའི་་་དབང་ཕྱུག་དང༌། བདེ་ཆེན་དང་། བླ་ན་མེད་པ་ཞེས་པའི་མིང་བརྒྱད་ཀྱིས་་་བསྟན་ཏོ། །དེ་ལྟར་ཚོགས་ཀྱི་འཁོར་ལོ་སྒོམ་པའི་རྣལ་འབྱོར་པས་ཡིད་ནང་དུ་་ཆུད་པའི་དགའ་བ་བཞི་དང་། །ཁོ་ཡིད་ཕྱིར་བྱུང་བའི་ཚེ་ལོངས་སྤྱོད་པའི་དགའ་བ་་བཞི་བསྒོམ་པ་ཡིན་པར་རྒྱུད་འདི་ཉིད་དང་། དགྱེས་རྡོར། ངེས་བརྗོད། རྡོ་རྗེ་མཁའ་འགྲོ་ཧེ་རུ་ཀ་འབྱུང་བ། ་མཁའ་འགྲོ་མ་ཀུན་སྤྱོད་གདན་བཞི་ལས་་་་་གསལ་བར་གསུངས་པ་ཡིན་ནོ། །༼། ཀུན་སྙིང་ནི་མ་རྟོགས་ནས་:བསྐྱེད་རིམ་གྱིས་ལྷ་་བསྐྱེད་པར་བཤད། །ལོངས་སྤྱོད་པའི་དགའ་བ་བཞི་སྒོམ་པ་ཡིན་པར་རྒྱུད་འདི་ཉིད་དང་། འཕྲོ་བ་འབྱུང་བ་དང་འདྲ་བ་ཡིན་ནོ། །ད་ནི་དེ་རྣམས་ཀྱི་ཚོགས་ཀྱི་འཁོར་ལོ་དང་། ཞི་བའི་བཏུང་བ་དང་། །ཕོ་ཉའི་:གཙང་སྦྲའི་ཀྱི་ཡན་ལག་ཏུ་གྱུར་པའི་པཱུ་ཛའི་རྫས་འདོད་པའི་ཡོན་ཏན་རྣམས་ལ་འཇིག་རྟེན་ན་ལེགྲགས་པའི་་་བརྡའི་སྟེང་དུ་ཟག་མེད་ཀྱི་བརྡ་དངོས་གྲུབ་ཀྱི་ཡན་ལག་ཏུ་གྱུར་པ་ནི། འཇིག་རྟེན་ན་གྲགས་པའི་ཁྲག་ལ་སངས་རྒྱས་དང༌། བྱང་ཆུབ་སེམས་དཔའི་བརྡས་མེ་ཏོག་་་།
