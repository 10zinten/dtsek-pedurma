རྒྱུད། ཡིག། བཞེས་ཤིག་དམ་ཚིག་མ། །གནས་ནས་དབང་དུ་མཛད་དུ་གསོལ། །ཏིཥྛ་སིདྡྷི་ཀུ་རུ་ཧུ་ཧུ་ཨོ་ཕྲོ་ཧ་གྲི་:ཧ་སརྦ་དུཥྛན་བྷག་ཕུ་ཕོ་ཕྲོ་ཕྲོ་ཕཊ་སྭཱ་ཧཱ། རྡོ་རྗེ་དམ་པ་་་བཅིངས་ནས་སུ། །དཀར་པོ་གི་ཝང་བསྒྲེང་ནས་སུ། །།གཉིས་ཀ་རྩེ་མོ་བཀུག་ནས་ནི། །ཁ་ཡི་འགྲམ་དུ་གཞག་པར་བྱ། །༄༅། །ལྷག་མའི་གཏོར་མ་བྱིན་གྱིས་བརླབ། །དུད་པ་སོལ་བ་ཞེས་བྱ་བའི་སྔགས་དང་ཕྱག་རྒྱའོ། །བཅོམ་ལྡན་འདས་རྣལ་འབྱོར་ནམ་མཁའ་དང་ཡེ་ཤེས་མཁའ་འགྲོ་མའི་སྒྲུབ་ཐབས། »ང་། སློབ་དཔོན་ཆེན་པོ་རྣམ་་་པར་རྒྱལ་བའི་དབང་པོའི་སྡེས་མཛད་པ་རྫོགས་སོ། །ལོ་ངོ་ཌ། ་་་བདག་་་པའི་ཚོ། །ཚོང་པོ། ཡཾ་ང་ན་ཊཀོ། དམ་པ་བདག་འབྱུང་ལོ་ཁ་སྐོང་སྐུ་སྐྱེན་ཀྱི་ཚོན་ཆ་དོར་བྱའོཞུ་ཀྱང་གལ་ཆེའི་ལེགས་བཤད་བྱེད་པ། བོད་པ་ཚོ། ལ་ས་ག་བ་བསྡུར་མཆན། ས་re། མོ། །Pag་ད་ཁ་ཡང་ཁྱོད་གང་དུ་ཆུ་དང་ག་ཤོmaཚོ་ལ་ཁpལཏུ་ཁོ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་རྣལ་འབྱོར་ནམ་མཁའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་བཅས། «པེ་»«སྣར་»+མ། «པེ་»«སྣར་»གི། «པེ་»«སྣར་བརླབས། «པེ་»«སྣར་+། ::«པེ་»«སྣར་རྒྱའི། །༼«པེ་»«སྣར་»པ། «པེ་»ཏིཙྪ། «པེ་»«སྣར་བཞག་་བ་«པེ་»«སྣར་བཞག«པེ་»«སྣར་»བསྒྲེང་བའི། «པེ་»«སྣར་བཞག«པེ་»«སྣར་དེ། («པེ་»«སྣར་»བཅིང་། «པེ་»«སྣར་བསྒྲེངས། «པེ་»«སྣར་»སྤྲོད། །«པེ་»«སྣར་བསྒྲེངས། «པེ་»«སྣར་བཟློས་ནས་རྩེར། འ«པེ་»«སྣར་»བཅིངས། &«པེ་:::《སྣར་»། «པེ་»«སྣར་བསྲུངས། «པེ་»«སྣར་པ། ༄༅«པེ་»«སྣར་བསྒྲེངས་ནས། 《ཅོ་»བསྒྲེང་ན། «པེ་»«སྣར་»ཨ། «པེ་»«སྣར་»། «པེ་»«སྣར་»མཉམི། «པེ་»«སྣར་+ཕྱག་རྒྱ་ཡང་། («པེ་»«སྣར་»ན། «པེ་»«སྣར་ཏ་ཙིརྨ་ཀཱ་ཡ་སི། །«པེ་»«སྣར་ཏ་ནིརྱ་ཏཱ་ཡ། /«པེ་»«སྣར་ར་ནཱ་ཡཱཏྨ་ཏ་ནིརྱ་ཏཱ་ཡ། །(«པེ་«སྣར་གཟུང་། «པེ་»«སྣར་བཞིག«པེ་»«སྣར་པོ་གཟུང་། «པེ་»«སྣར་»པོར། །«པེ་»«སྣར་»བུའི། «པེ་»«སྣར་»གཟུང་། «པེ་»«སྣར་»ཉ། «པེ་»«སྣར་བསད་པ། («པེ་»«སྣར་»ལས། «པེ་»«སྣར་»། «པེ་»ཤུདྡྷེ་བཛྲ་ཧཾ། «སྣར་བཛྲ་ཤུདྡྷེ་བཛྲ་ཧཾ། །
