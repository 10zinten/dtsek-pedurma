ཨོཾ་བྱུ་གྷ་ཌཱ་ཀི་ནི་དྲི་བྱ་ཛཧཱུྃ་བོ་ཧོས་མ་ཡ་ཧཱུཾ། ་ས་མ་ཡོཉཧཾ། ཕྱག་རྒྱ་དེ་ཉིད་བརྟན་བཅིངས་ནས། །ས་གདུགས་པདྨ་ལྕགས་སྒྲོག་བྱ། །མགོན་པོས་གསུངས་པའི་ཕྱག་རྒྱ་སྟེ། །ཅེ་སྤྱང་མ་ཡི་དམ་ཚིག་བཤད། །ༀ་ཛི་བུ་ཀཱ་ཌཱ་ཀཱི་ནཱི་དྲི་བྱ་་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བཅིང་བ་བརྟན་བཅིངས་ནས། །ལན་གསུམ་དུ་ནི་དགྲོལ་བར་བྱ། །ཨོ་ཨུ་:ལུ་ཀཱིཌཱ་ཀི་ནཱི་དྲྀ་ཤྱ་ཛཧཱུཾ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། རྡོ་རྗེ་བཅིང་བ་བརྟན་བཅིངས་ནས། །ཏེ། རྣམ་པར་ཕྱེ་སྟེ་ཁ་རུ་གཞུག། ཨོ་ཌཱ་ཀི་ནི་དྲིབྱ་ཛཧཱུྃ་བཾ་ཧོ། །་ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཐལ་མོ་སྤྱི་བོར་བཅིངས་པ་ནི། །མར་མེ་མ་ཡི་དམ་ཚིག་བཤད། །ཨོ་དཱི་པ་ནི་དྲི་ཤུ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཡི་དྭགས་ཐལ་མོ་ཁ་རུ་བཅུག། འཇིབ་བྱེད་མ་ཡི་དམ་ཚིག་བཤད། །ཨོཾ་ཙུ་:ཏི་ནི་དྲི་བ་ཧཱུྃ་བོ་ཧོ། ས་མ་ཡ་ཧཱུྃ། ༈ས་མ་ཡོ་ཉཧཾ། རྡོ་རྗེ་བསྡམས་པ་བརྟན་བྱས་ནས། །གཏུན་ཤིང་འཛིན་པ་ལྟ་བུར་སྦྱར། །རྟ་ཡི་སྐད་ནི་ལྟར་བྱེད་པ། །འཕྲོག་བྱེད་མ་ཡི་སྐད་ཚིག་བཤད། །ཨོཾ་ཀཾ་པོ་ཛི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་སིཾ། །ས་མ་་་ཡོཉཧཾ། ཉི་མ་དང་ནི་ཟླ་བ་འང་བརྐྱང་། །འཇིགས་པ་དང་ནི་རྣམ་ཤེས་སྦྱར། །སྤོས་མ་ལ་སོགས་ལྷ་མོ་གསུམ། །དམ་ཚིག་ཏུ་ནི་ཐུབ་བས་གསུང་། །ཨོ་པུཀྐ་སཱི་དྲི་ཤྱ་ཛ༔ཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ་ཨོ་:ཌོམྦི་ནཱི་དྲིབྱ་ཛgཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ༀ་ཙཎྜ་ལཱི་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཌོཟླ་བའི་ལག་པ་བརྐྱང་པ་ལ། །ཉི་མའི་ལག་པ་ཡང་དག་སྦྱར། །དྲག་མོ་ལ་སོགས་ལྷ་མོ་གསུམ། །དམ་ཚིག་ཏུ་ནི་འགྱུར་བའོ། །ཨོ་གོ་རཱི་ནཱི་དྲི་བྱ་ཛཧཱུྃ་བོ་ཧཾ། ས་མ་ཡ་ཧཱུཾ། ས་མ་ཡོཉཧཾ། ཨོཾ་རུ་དྷི་རཱ་ནཱི་དྲི་ཤུ་ཛཧཱུྃ་བཾ་ཧོ། །ས་མ་ཡ་ཧཱུཾ། »ས་མ་ཡོཉཧཾ། ཨོ་མོ་སཱི་དྲི་ཤུ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཧཱུཾ། ས་མ་—།
