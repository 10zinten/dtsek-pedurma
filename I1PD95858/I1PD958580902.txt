ལྷ་རྣམས་ལ་སྤྲུལ་པའི་སྐུ་བརྒྱ་བྱིན་ལྟ་བུས་ཆོས་སྟོན་པའི་སྐལ་བ་དང་ལྡན་པར་གྱུར་ཅིགདེ་ལྟར་ཁམས་གསུམ་གྱི་སྡུག་བསྔལ་ལས་གྲོལ་ཏེ། ཁྱད་པར་གྱི་ལམ་ལྔ་ལ་བརྟེན་ནས། སྒྲིབ་པ་གཉིས་སྤངས་ཏེ། ཚོགས་གཉིས་རྫོགས་ནས། །མཐར་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཕྱག་རྒྱ་ཆེན་པོ་མཆོག་གི་གོ་འཕང་ལྷུན་གྱིས་གྲུབ་སྟེ། འཁོར་བ་མ་:སྟོང་གི་བར་དུ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་རྒྱུན་མི་འཆད་་པར་་་་་སྤྱོད་པའི་སྐལ་བ་དང་ལྡན་པར་གྱུར་ཅིགདེ་ནས་དགེ་བའི་རྩ་བ་བསྔོ། །དེ་ནས་ཕྱིས་སྲེག་ཁང་དུ་འོ་མ་གཏོར་ཏེ། ཞི་བའི་སྔགས་བཟླས་ཏེ། རུས་བུ་བཏུས་ལ་སཱཙྪ་བཏབ་ནས། །རབ་ཏུ་གནས་པར་བྱའོ། །རོ་སྦྱང་བའི་སྒྲུབ་ཐབས་ཀྱི་ཆོ་གབླ་མ་དམ་པའི་གདམས་ངག་ལ་བརྟེན་ནས། གཞུང་ཉེ་བར་བསྡུས་པ་རྫོགས་སོ༑། །།དེ། །རྡོ་རྗེ་མཁའ་འགྲོ་མ། ་་་་ལ། ལ་། བ་ཀུ་ལ་ལས་ཀྱི། །ཞལ་གདོང་དམག་བསྡུར་མཆན། དེ་ལ་བདེན་པ་ཡི་ལུཔ། ས། །(མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»སྣར་རོ་བསྲེག་པའི་སྒྲུབ་ཐབས་ཀྱི་ཆོ་ག་བཞུགས་སོ། །གི«པེ་»«སྣར་»དྷ། «པེ་»«སྣར་»། «པེ་»«སྣར་»པ། ས«པེ་»«སྣར་གནས། «པེ་»《སྣར་བསྲེགཚེ«པེ་»«སྣར་»+ལ། «པེ་»«སྣར་སར། «པེ་»«སྣར་»པ། ་ལ་་«སྣར་»སྟེན། «པེ་»«སྣར་གྱིས། «སྣར་འདབ། «པེ་»«སྣར་ཁ། མ«པེ་»«སྣར་»ཀྱིས། «ཅོ་»གྱི། «པེ་»«སྣར་གྱིས། ཉི«པེ་»«སྣར་གྱི། «པེ་»«སྣར་»+ཤེས། དགྱེ«པེ་»«སྣར་»ལ། ༽ཀྱི«པེ་»«སྣར་»རིམས། («པེ་»«སྣར་པ་ཆེ། ཉེ«པེ་»«སྣར་»ཏཱའི། སྤྱི«པེ་»«སྣར་བསྲུངས་པའི། («པེ་»«སྣར་»ཅིག«པེ་»སྣར་»པའི་སྔར། «པེ་»«སྣར་པ་བསམས། «པེ་»《སྣར་སྦྱངས་པ་བསམས། «པེ་»《སྣར་བསྙོན་པར། «ཅོ་»བསྙོན། («པེ་»«སྣར་པར། «པེ་»«སྣར་»དང་བ། ༄༅«པེ་《སྣར་»པར། «པེ་»«སྣར་ན། «པེ་»«སྣར་»བའི། («སྣར་པ། «པེ་»«སྣར་»སུ། «པེ་»«སྣར་བས། «པེ་»«སྣར་»སྦྲའམ། ཁྱི«པེ་»«སྣར་བཅུས། «པེ་»«སྣར་»དགོད་དོ། «པེ་»«སྣར་པ། «པེ་»«སྣར་གྲུ་བཞི། «པེ་»«སྣར་པའི་བསྲེགས། །།
