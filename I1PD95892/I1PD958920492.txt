།ར་རྒོལ་ཆོ་བ་་་་ཋཥ་ཎ་བོ། ་་་་་་་་་ང་འདབ་ལ་ཎ་ཁཀ་ཁང་ཀ་བབ་ཅ། །»/mཚོ་དེའང་མཚོར། །azད་ਸਵਾਲਉਸਨੌਂਸ:ਰਸਨੂੰਗਰਮਕੈਲਸ਼ਾਦਾਪੁਨਰਸ਼ཆ། ་་ཁང་ལ། ཀ་ཁ་ཏ་མི་མ་ཆ་དེ༌ན། མཁ་ཁོ་ན་ལ་ཏུ་དུ་ཎ་ཊི་ཁཀྲ་བས་ད་ཁ་ཀ་ཁ་ག་གདpཟླ་ལབལ་་་་ཨང་ཡ༄༅། །དཔལ་ལྷ་མོ་ལ་བསྟོད་པ་ཞེས་བྱ་བ། ཚོང་མི་ལ་ཁ་ཎ། ཁ་ཁ་ཁ་ཁ། །ག་ང་ལ་ཁོང་ཚོ་ག་ཚོ་ཀ་ཁ་ROADBਇੰਨਸਾਸyਲਰਲਰ'ਨ।ਮਨੂੰਸ਼ਰ། ་་་ཎ་། །ར་མ་ང་ཅ་་ར་ལ། འགྲོ་བ་ཁོ་ན་བྱ་མདའ་ལ་ཁང་བ་་་་/pm#བྱ་བ། །ས་ལ་རང་དབང་། ང་དང་གང་Eན༌་་་་་ན་། :༄༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་དེ་ཝི་ཀཱ་ལི་སྟོད་ཏྲ་ནཱ་མ། བོད་སྐད་དུ། དཔལ་ལྷ་མོ་ལ་བསྟོད་པ་ཞེས་བྱ་བ། དཔལ་ལྡན་བླ་མ་དམ་པ་དང༌། །ཧེ་རུ་་ཀ་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་བསྟན་པ་རབ་ཏུ་རྒྱས་མཛད་ཕྱིར། །།ཡེ་་་་ཤེས་དབྱིངས་ལས་རང་བྱུང་དཔལ་ལྡན་མ། །དུས་གསུམ་རྒྱལ་བའི་ཡུམ་གྱུར་་་་དཔལ་ལྡན་མ། །སྲིད་གསུམ་བདག་པོས་ཕྱག་བྱས་ལྷ་མོ་བསྟོད། །གདུག་པ་་་སྐྲག་མཛད་ཤིན་ཏུ་རིད་པའི་སྐུ། །ནམ་མཁའི་མདོག་ཅན་སྤྲུལ་པའི་གཟའ་སྐར་འཕྲོ། །ཁྲོ་གཉེར་སྨིན་མ་རབ་བསྡུས་མཆེ་བ་གཙིགས། །བདུད་བཞི་ཚར་་་་་གཅོད་དཔལ་ལྡན་ལྷ་མོ་བསྟོད། །དུག་ལྔ་བྱང་ཆུབ་ལམ་མཛད་བོང་བུ་ཕྱིབས། །ཐབས་ཤེས་ཟུང་འབྲེལ་ཞབས་གཉིས་ལྕགས་སྒྲོག་མཛེས། །ཉི་ཟླ་རབ་རྒྱས་ཁྲག་་་ཞག་ཐིག་ལེས་བརྒྱན། །བདེ་མཆོག་འབྱུང་གནས་དཔལ་ལྡན་ལྷ་མོ་བསྟོད། །སྐུ་གསུམ་སྒྲིབ་མེད་སྤྱན་གསུམ་གློག་ལྟར་མྱུར། །ཁམས་གསུམ་སྐྱབས་མཛད་དར་།
