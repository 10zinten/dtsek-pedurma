བར་བྱ། །གྲུ་བཞི་རྡོ་རྗེས་མཚན་པ་ལ། །དང་པོའི་རྒྱ་གསུམ་ནང་དུ་བྲི། །དེ་ནས་རེ་ཁ་རྣམ་བཞི་ལ། །མུ་ཁྱུད་གསུམ་དང་ལྡན་པ་ལ། །ཡི་གེ་སུམ་བརྒྱ་རྩ་་གཉིས་ཀྱི། །རྒྱ་ཀྱི་རིམ་པ་བྲི་བར་བྱ། །རྒྱ་རྣམས་ཐམས་ཅད་གཡོན་བསྐོར་བྲི། །ནང་དུ་མིང་ཡིག་བྲི་བར་བྱ། །ཡི་གེ་སུམ་ཅུ་རྩ་བཞི་ལ། །ང་ཉ་ན་མ་བཞི་་་སྤངས་ནས། །གཞན་རྣམས་འཁོར་ལོར་དགོད་པར་བྱ། _། ཀ་ཁ་ག་གྷ་ང་། ཙ་ཚ་ཛ་ཛྷ་ཉ། ཊ་ཋ་ཌ་ཌྷ་ཎ། ཏ་ཐ་ད་དྷ་ན། པ་ཕ་བ་བྷ་མ། ཡ་ར་ལ་ཝ་ཤ་ཥ་ས་ཧ་ཀྵ། ཀ་ཀཱ་དྷ་ཡ་གཉིས། ཁ་ཁཱ་ཧི་ཁཱ་ཧི། ག་གཱརྫུ་ཡ་གཱཛྙཱ་ཡ། གྷ་གྷཱ་ཏ་ཡ། གྷཱ་ཏ་ཡ། ཙ་ཙཱ་ལ་ཡ་གཉིས། ཚ་ཚྭ་ཏ་ཡ་གཉིས། ཛ་ཛཱམྦྷ་ཡ། ཛྷ་ཛྷཱ་ཏ་ཡ་གཉིས། ཊ་ཊཱ་ལ་ཡ་གཉིས། ཋ་ཌཱ་ཀ་ཡ་གཉིས། ཌ་ཌ་བྷ་ཡ་་་གཉིས༑ཌྷ་ཝཱ་ཀ་ཡ་གཉིས། ཎ་ཎ་ཤ་ཡ་གཉིས། ཏ་ཏཱ་བྷ་ཡ་གཉིས། ཐ་ཐཱ་བྷ་ཡ་གཉིས། ད་དཱ་ར་ཡ་གཉིས། དྷ་དྷཱ་ར་ཡ་གཉིས། ན་ནཱ་ཤ་ཡ་གཉིས། པ་པཱ་ཏ་ཡ་གཉིས། ཕ་ཝཱ་ཏཱ་ཡ་གཉིས། བ་བཱནྡྷ་ཡ་གཉིས། བྷ་བྷེ་ད་ཡ་་་གཉིས༑མ་མཱ་ར་ཡ་གཉིས། ཡ་ཡོ་ཧ་ཡ་གཉིས། ར་རོ་ཧ་ཡ་གཉིས། ལ་ལོ་བ་ཡ་གཉིས། ཝ་ཝཱ་ཧ་ཡ་གཉིས། ཤ་ཤོ་ཥ་ཡ་གཉིས། ་ཥ་ཥཱཉྩི་ཡ་གཉིས། ས་སཱ་དྷ་ཡ་གཉིས། ཧ་ཧཱ་ན་ཡ་གཉིས། ཀླ་ཀློ་བྷ་ཡ་གཉིས། ཆེ་གེ་མོ་ཝ་ཤཾ་ཀྲྀཏྟ་ཝ་བྱ་ཀུ་རུ་ཧོ༔ཤཱནྟིརླུ་རུ། པུཊྛིརྐུ་རུ། ཀླུ་མོ་ཀུ་རུ། ཀློ་ཐཾ་ཀུ་རུ། མ་རཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་སིདྡྷི་སིདྡྷི་མ་ཧཱ་སིདྡྷི་པཱུ་ཛ་ཨགི་ཏ་བྷ་ན་ཏ་མཱ་ར་ཎི་ཀ་རི་ཎི་ཧཱུཾ། ཨོ་ཀུ་སུ་མ་མཱ་ལི་བྷི་ཧཱུཾ་ཛཨོ་བཛྲ་ཀུ་པ་ཛི་ཀཾ་བྷི་ཥཋཋཨོ་གྲོ་ག་མ་ཀྲིཏྟེ་སརྦ་ཤ་ཏུན་ནཱ་མ། རྩ་བའི་བྱེ་དང་ཤམ་བུའི་རྒྱ། །བསྒྱུར་དང་འདོན་དང་འཇུག་པ་་་གཅེས༑། དང་པོ་བསྙེན་ཏེ་སྒྲུབ་པ་བྱ། །།ལས་ལ་སྦྱར་ཏེ་མ་གྲུབ་ན། །འཁོར་ལོ་ཏིང་འཛིན་རྒྱ་ཀྱིས་སྤོགས། །ཌཱ་ཀི་མ་ཡི་རྫས་སྒྲུབ་འདི། །སྤོག་མི་་་་—།
