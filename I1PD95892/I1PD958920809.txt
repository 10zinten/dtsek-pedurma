པའི་བྱ་བ། །ར་་ང་། ཀའ་ཁ། ང་འ་ན། །ཆོ་། བཀྲ་རྒྱུ་ਲਗੇਨਸੈਸ਼ਨ'ਲੈਲਾLਨੂੰ'ਲgਝਾਨਸਲੋਂਕਾਰੂਫਰਰਰ'ਨੂੰਸਨਮਝਤੋਂਹੈ।ਨਰਲਡ'ਲੈਣਨਾਲ༄༅། །བགེགས་བདག་ཆེན་པོའི་སྐུ་བྲི་བའི་མན་ངག་ཡིད་བཞིན་ནོར་བུ་ཞེས་བྱ་བ། ਜੈਲਸਿੰਘਸਨ।ਲਲਕਨਰੇਡ'ਚੋਂਨੈਸ਼ਨਲ(། །ཏི་ཊ། ཙਏਆਉਹ། ༄༅། བགེགས་རྒྱལ་གྱི་བྲིས་སྐུའི་མན་ངག༄༅། །རྒྱ་གར་སྐད་དུ། མ་ཧཱ་བི་ན་ཡ་ཀ་རཱུ་པ་ཨུ་པ་དེ་ཤ་རད་ན་ནཱ་མ། བོད་སྐད་དུ། བགེགས་་་་བདག་ཆེན་པོའི་སྐུ་བྲི་བའི་མན་ངག་ཡིད་བཞིན་ནོར་བུ་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་མཁའ་འགྲོ་ལ་ཕྱག་འཚལ་ལོ། །འཁོར་ལོ་སྡོམ་པ་ལ་བཏུད་དེ། །དགེགས་བདག་ཆེན་པོའི་བྲིས་སྐུ་བྲི། །རོ་ལ་གཡོགས་པའི་རས་སཾ་ནི། །མཚོན་གྱིས་་་བསྣུན་པའི་རས་དང་ཡང་། །བུ་བཙས་ཤི་བའི་རས་དང་ནི། །བ་གླང་ལྤགས་སོགས་ཁྱད་པར་ལས། །ནག་པོའི་བརྒྱད་དང་བཅུ་བཞི་ལ། །འབྱུང་པོའི་་་དཀྱིལ་འཁོར་བསྒྲུབ་པར་བྱ། །རི་མོ་མཁན་ཡང་དམ་ཚིག་ལྡན། །སྒྲུབ་པོ་དག་གིས་བྲི་བར་བྱ། །གྲུ་བཞི་པ་ལ་སྒོ་བཞི་པ། །སྒོ་བསྲུང་དག་ནི་ཡང་དག་ལྡན། །དངོས་གྲུབ་ཀུན་གྱི་ཡོན་ཏན་གནས། །འདོད་པའི་ཡོན་ཏན་ལྔས་གང་བ། །མཆོད་པ་ལྔ་ཡིས་བརྒྱན་པ་ཡིས། །དཀྱིལ་འཁོར་ཉམས་དགའ་བྲི་བར་བྱ། །།—།
