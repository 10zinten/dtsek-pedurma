ཥ་ཥ་ཞེས་པས། དབུ་རྒྱན་ཅོད་པན་སྦྱིན་པར་བྱའོ། །རང་གི་ལྷག་པའི་ལྷར་་གསལ་བས། །ོ། འདི་དེ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། ཅ། །ཧེ་རུ་ཀ་དཔལ་དབང་བསྐུར་བ། །འགྲོ་བ་སྐྱོབ་ཕྱིར་གཟི་བརྗིད་ཅན། །དེ་རིང་ཁྱོད་ལ་དབང་བསྐུར་ཏེ། ལ་། ཨོ་ཤི་ཧེ་རུ་བཛྲ་སཏཱ་ཟེར་ཀ་ཨ་ཏི་པ་ཏི་སིཾ་བཛྲ་དྷིག་ཟེར་ཨ་བྷི་ཥིཉྩ་མཾ་ཏིཥྛ་བི་ར་ས་མ་ཡ་ཧཱུཾ། །སངས་རྒྱས་ཀུན་གྱི་རྡོ་རྗེའི་དབང་། །དེ་རིང་ཁྱོད་ལ་བསྐུར་བར་བྱ། །སངས་རྒྱས་ཉིད་དུ་རབ་བསྒྲུབ་ཕྱིར། །རྡོ་རྗེ་འདི་ནི་ཁྱོད་ཀྱིས་ལོང་། །རྡོ་རྗེ་ལག་པ་གཡས་སུ་སྦྱིན་པར་བྱའོ། ་ོ། འདི་ནི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །དྲིལ་བུ་དབྱངས་ཀྱི་རྗེས་འགྲོར་བཤད། །ཁྱོད་ཀྱིས་ཀྱང་ནི་རྟག་ཏུ་བཅང་། །བྱང་ཆུབ་མཆོག་དུ་རྒྱལ་བས་བཞེད། ཁ་ཨོ་བཛྲ་གྷནྡྷ་ཨ་ཏི་པ་ཏི་ཙཱ་ཙེཨ་བྷི་ཥིཉྩ་མཾ་ཏིཥྛ་བཛྲ་ས་མ་ཡ་ཧཱུཾ་ཞེས་པས་དྲིལ་བུ་སྦྱིན་ནོ། །དེ་ནས་མགོ་བོར་རྡོ་རྗེ་བཞག་སྟེ། ལཨོ་ཤྲཱི་བཛྲ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་ཏི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་མཾ་ཤྲཱི་ཧེ་རུ་ཀ་ན་མ་ཨ་བྷི་ཏེ་ཀ་ཏ། །ཀྱེ། དཔལ་གཞད་པའི་རྡོ་རྗེ་ལ་སོགས་པ་རིགས་ཀྱི་ལས་དང་རྗེས་་སུ་འཐུན་པའི་མིང་དེ་ཉིད་འདུས་པའི་རྒྱུད་རྡོ་རྗེ་གུར་དུ་གསུངས་པ་བཞིན་ཤེས་པར་བྱའོ། །སྐྱེས་པ་དང་བུད་མེད་ལ་དཔའ་བོ་རྣལ་འབྱོར་མའི་དོན་མངོན་དུ་་་བྱེད་པ་བསྟན་པའི་མིང་གིས། །གསང་བའི་མཚན་ཤེས་པར་བྱའོ། །་་་་་། བླ་མ་དེ་ནས་གཞན་དོན་ནུས་པའི་སློབ་མ་ལ། །རྡོ་རྗེ་སློབ་དཔོན་གྱིས་དབང་་་་བསྐུར་བར་བྱ་སྟེ། །མཎྜལ་སྟེགས་བུ་སེང་གེའི་ཁྲི་དང་བཅས་པ། གདུགས་་་་དང་། ྋརྒྱལ་མཚན་དང་། བ་དན་དང་བཅས་པས་མཆོད་པར་བྱ་ཞིང་། རྡོ་རྗེ་སེམས་དཔར་བསྐྱེད་བྱས་ལ། །རྡོ་རྗེ་དྲིལ་བུ་ཡང་དག་འཁྱུད། །གཉིས་མེད་བདེ་ཆེན་རྣལ་འབྱོར་བདག། ཕྱག་རྒྱ་ཆེན་པོར་བྱིན་གྱིས་བརླབ། །ལག་པ་་་གཡས་སུ་ཟླ་བ་ལ་གནས་པའི། །ཧཱུྃ་ལས་རྡོ་རྗེ་བསྐྱེད་དེ། ཐོག་མཐའ་མེད་པའི་།
