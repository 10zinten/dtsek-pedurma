རྒྱུད། བ་ན། ན་ཏོ། །སྡིགས་འཛུབ་ཅན། །མེ་རི་འབར་བས་བགེགས་ཐམས་ཅད་བསྲེག་པ་ཅིག་བསམས་ལ་སྔགས་བཟླས་འོད་དུ་མི་བཞུ་བར་དཀྱིལ་འཁོར་ཆེན་པོའི་ལས་མ་གྲུབ་ཀྱི་བར་དུ་བར་ཆད་བསྲུང་དུ་གསོལ་ཅེས་འཕྲིན་ལས་གཞོལ་ལོ། །གཙོ་བོའི་བུམ་པ་བསྐྱེད་པ་ཡང་ཕྱི་ནང་གི་ཡབ་ཡུམ་དུ་བསྐྱེད། ་ཤར། །ལྷོ། ནུབ། བྱང་གི་བུམ་པར་ཕྱོགས་ཕྱོགས་ཀྱི་ལྷ་རྣམས་བསྐྱེད་པར་བྱའོ། །དེ་ཡང་བུམ་པ་གཉིས་ལ་ནི་རྣམ་རྒྱལ་བུམ་པར་ལྷ་ཚོགས་རྣམས་བསྒྲུབ་ཐབས་བཞིན་བསྐྱེད་པར་བྱའོ། །ལས་བུམ་ལ་ནི་གོང་བཞིན་ནོ། །དེས་ནི་བུམ་པ་ལྷག་ནས་བསྟན་ནོ། །ཁོ་ਸੀਕਿ'BEASAਊ॥དེ་ནས་སྤྲོ་ན་ཙན་དན་གྱི་ཐབ་ཟླུམ་པོའམ། ཟླ་གམ་དུ། གྲོ་དང་ནས་དང་ཡུངས་ཀར་དང་དཀར་གསུམ་མངར་གསུམ་རྣམས་བདུད་རྩིར་བྱིན་གྱིས་བརླབས་ཏེ། ཡབ་ཡུམ་གྱི་ཞལ་དུ་ཕུལ་བས་ཟག་པ་མེད་ཀྱི་བདེ་བས་རྒྱུད་གང་བས་མཉེས་ཏེ། བདག་གི་སྡིག་སྒྲིབ་དང་ནད་གདོན་ཐམས་ཅད་ཞི་བ་དང་། ཚེ་དང་ལོངས་སྤྱོད་ཐམས་ཅད་རྒྱས་པར་བསམས་ལ་རྩ་སྔགས་ལ་ཞི་རྒྱས་ཀྱི་ཤམ་བུ་གདགས་པར་བྱའོ། །དེ་ནས་ཡེ་ཤེས་པ་གཤེགས་དམ་ཚིག་པ་མི་དམིགས་པར་བྱའོ། །སྦྱིན་བསྲེག་ལྷག་གནས་སོ། །::ཀྱིངབད་ལ་དཀྱིལ་figਨਸਨਨੂੰਨੂੰਸੌਗਨਨਾਏདེ་ནས་སའི་ལྷ་མོ་ནི་མཆོད་བསྟོད་གཏོར་མ་རྣམས་གོང་བཞིན་བྱས་ལ་མཆོད་པ་ལྔས་མཆོད་དེ། ལག་པས་སྤོས་ཐོགས་དཔུས་པུས་མོ་ས་ལ་རྩུགས་ལ། སྐྱོབ་པ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །སྤྱོད་པའི་ཚུལ་དང་ཁྱད་པར་ལམ། །ས་དང་ཕ་་རོལ་ཕྱིན་པ་ལ། །ལྷ་མོ་ཁྱོད་ནི་དཔང་དུ་གྱུར། །ཤཱཀྱ་སེང་གེ་སྐྱོབ་པ་ཡིས། །ཇི་་་་ལྟར་བདུད་ཀྱི་སྡེ་བཅོམ་པ། །དེ་བཞིན་བདག་གིས་བདུད་ཀྱི་སྡེ། །ཕམ་པར་བྱས་་།
