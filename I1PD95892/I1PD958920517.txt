རྒྱུད། ༼ཧུའིགྱིས་བསྒྱུར་བའོ། །།།་་་ཡོད་འདིའི་བརྒྱུད་པ་ནི། དངོས་གྲུབ་བརྙེས་པའི་རྣལ་འབྱོར་པ་ཆེན་པོ་དཔལ་ལྡན་ཤ་བ་རི་དབང་ཕྱུག། མི་ཏྲི་པ། ཨ་ཛུ་ག་ཏ་པ། ཨ་ཏ་ཡ་བཛྲ། ཁྱུང་པོ་་རྣལ་འབྱོར། རིན་ཆེན་བརྩོན་འགྲུས། ཆོས་ཀྱི་སེང་གེ། ཆོས་ཀྱི་ཤེས་རབ། བརྩོན་འགྲུས་སེང་གེ། ཚུལ་ཁྲིམས་འབུམ། རྒྱལ་མཚན་འབུམ། བྱམས་པ་དཔལ་བཟང་པོ། བསྒྲུབ་ཆེན་ཆོས་འབྱུང་རིན་ཆེན། གདན་ས་པ་ཀུན་དགའ་དཔལ་་བཟང་པོ། དེས་བདག་ལའོ། །ལ། །་་་་བལ་པོའི་པཎྜི་ཏ་དེ་བ་པུརྣ་མ་ཏི་དང་། ལོ་ཙཱ་བ་འགར་ཆོས་ཀྱི་བཟང་པོས་ཀྱང་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་ཅེས་བྱམས་པ་གསུང་ངོ༌། །འདིའི་བཀའ་ཡིག་མཆོག་སྒྲུབ་ཀྱི་བཀའ་ཆུང་བ་དང་འཐུན་པར་མཛད། །།།ཁྱོད་དང་དབྱེ་ਕਵੀਸੰਗਤਾਲ'ਇਕ'ਇਸਲਾਮ་། ་་་་་་་ལ་ལ། ལེ་བ་ལྟ་། ངོ་མ། ད་ཀྱི་་་་ཁ། པོ་ཀྱི་ད་ཚོ་པ་ན་དད་པ་དེ་ནི་བའི་བྱ་བ་སོགས་དེ་དང་། །བྱ་བའོ། །བཞི་བྱེད་ཅཏེདཔོ། །།བབཔ་ཀན་དག་པ་ལ། འོ་ན་ཁོ་ਇਸਤੋਂਸੋਧਤੇਲਉਤੇ་་་་བོད་ཀྱི་ཚ་བོ་ཁ་ནཏེཟླཏོགལ་བ་:བ་ཏུ་ཁཔར་ཏེ་།
