འདུལ་ད་ནས་ད་། ཁ་བ་དང་། ༄༅། །ནག་པོ་ཆེན་པོ་ལ་བསྟོད་པ། །དགྲན་ན་ལྟ་བཞིན་དུ། དུ་བ་ལས་བནིབདག་ལ་བ་དུ་མ་ལ་འབད་ག་ག༌་་་་་་༄༅། ནག་པོ་ཆེན་པོ་ལ་བསྟོད་པ་བཞུགས་སོ། །༄༅། །རྒྱ་གར་སྐད་དུ། །མ་ཧཱ་ཀཱ་ལ་སྟོད་ཏྲ། བོད་སྐད་དུ། ནག་པོ་ཆེན་པོ་ལ་བསྟོད་པ། ཧཱུྃ་ལས་་་རྣམས་ཀུན་བྱེད་མགོན་པོ་ནག་པོའི་སྐུ། །དམར་ནག་མེའི་འོད་ཀྱིས་ཀུན་ནས་་་བསྐོར༑། ཕྱག་གཡས་གྲི་གུག་འབར་བ་བསྣམས་པ་ནི། །དམ་ཉམས་དགྲ་་་་བོའི་སྲོག་རྩ་དམར་པོ་གཅོད། །གཡོན་པ་ཐོད་པ་དགྲ་བོའི་ཁྲག་བཀང་གསོལ། །ཤངས་ནས་རླུང་བྱུང་འཇིག་རྟེན་རླུང་ནག་འཚུབ། །ཞལ་ནས་ཁྲག་འཛག་འཇིག་རྟེན་ནད་ཀྱིས་འགེངས། །སྤྱན་རྩ་དམར་པོས་འཇིག་རྟེན་སྡིགས་པར་བྱེད། །སྐུ་གསུང་ཐུགས་ཀྱིས་སྟོང་གསུམ་ཟིལ་གྱིས་གནོན། །མོན་པ་ཕོ་མོ་ཐམས་ཅད་་་དབང་དུ་སྡུད་། །འཇིག་རྟེན་ལྷ་སྲིན་ཐམས་ཅད་དབང་བྱས་ནས། །བསྟན་་པ་ཉམས་པའི་དུས་ཀྱི་བྱ་ར་མཛད། །དམ་རྫས་བདུད་རྩི་འབར་བ་འདི་བཞེས་ལ། །ཐུགས་དམ་རྒྱུད་བསྐུལ་བཅོལ་བའི་ཕྲིན་ལས་མཛོད། །ནག་པོ་ཆེན་པོ་གཏོར་་།
