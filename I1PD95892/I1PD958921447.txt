བསྟན་འགྱུར། རྒྱུད། ཀགསོལ། །པྲ་ཏི་བེ་ཨ་གད་ཛ་ཡ་ཛ་ཛ་འཁོར་གྱི་ས་བདག་འོངས་པར་བསམ། །དེ་ནས་སྡུད་པ་འདོད་པ་འདི་སྐད་གཏོར་མ་སྦྱིན། །ཉོན་ཅིག་ས་བདག་ལྷ་ཀླུ་འཁོར་དང་བཅས། །སྒྲུབ་པ་ཆེན་པོ་དག་ལ་ངེད་འཇུག་ཕྱིར། །ས་ཀླུང་ཡིད་བཞིན་རིན་ཆེན་གཏོར་མ་འདི༑། ཁྱེར་ཅིག་ཁྱེར་ཅིག་ས་བདག་ཇི་སྙེད་ཚོགས། །ང་རྒྱལ་གདུག་རྩུབ་ལ་སོགས་ཀུན་སྤངས་ནས། །བདག་ཅག་རྣམས་ཀྱི་བསམས་པ་བསྒྲུབ་དུ་གསོལ། །དེ་ནས་ཚིག་གིས་རབ་བགོ་བ་བྱ་བ་ནི། །ལག་ན་རྡོ་རྗེའི་ཏིང་འཛིན་གྱིས། །རྡོ་རྗེ་དྲིལ་ཐོགས་ལ་ཚིག་གིས་བསྒོ་བ་བྱ་ཧཱུྃ་དེང་ནས་རྡོ་རྗེ་འཆང་ཆེན་ང༌། །གསང་བའི་དཀྱིལ་་་འཁོར་ངེད་སྒྲུབ་པས། །།བགེགས་ཚོགས་སའི་བདག་པོ་རྣམས། ་། ས་ཕྱོགས་་་་གཞན་དུ་སོ་སོར་དེངས། །གལ་ཏེ་འཁོར་བཅས་དེ་དག་ཀུན། །བསྒོ་བའི་་་བཀའ་རྣམས་མི་ཉན་ན། །ཕྱག་ན་རྡོ་རྗེ་མཐུ་སྟོབས་ཅན། ལྟོ། །འབར་བ་ཞལ་གྱི་བྱིན་རླབས་ཀྱིས། །མགོ་ལུས་ཚལ་པ་བརྒྱ་རུ་ཁོས་དལ་གྱི་མཐའ་མ་བསྐོར་རྡོ་རྗེ་འགྲོས་བྱ། །ཧཱུྃ་ཧཱུྃ་སའི་ཆོ་གའི་རིམ་པའོ། །དེ་ལ་འབབ་བྱེད་ཚུལ་ངོ་མ་ཞིག་ਨੂੰਵੀਨਸੀਲਿੰਦੂਆਂਨੂੰਪਿੰਡਸਾਰདེ་ནས་དལ་དུར་ཁྲོད་ར་ཐལ་ཁུང་གི་ས། །བླངས་སྟེ་སྟེགས་བུ་གྲུ་བཞི་བརྩིག། མཚལ་ཆེན་པདྨ་རཀྟ་མང་པོའི་ཆག་ཆག་གདབ། །གུ་གུལ་དྲི་གསོད་ཤིང་གི་དུ་བས་་་་བདུག། ཟིག་བ་སུ་ཏའི་ཞགས་པ་འིས། །སྔགས་དང་བྱིན་རླབས་མཐུན་ཚངས་པའི་ཅེས་སྤྱན་བཟློག་ཐིག་ནས་བརྩམས་ཏེ་ཡི་ཤེས་ཐིག་གདབ། །དམ་ཐིག་རྒྱུས་ཆེན་པོའི་བསྒྲིལ་མ་འིས། །བློ་ལྡན་གྱིས་ནི་ས་ཐིག་གདབ། །མཚོན་མཚན་དང་ལྡན་པའི་ཐལ་ཆེན་ལ། །ཁ་་དོག་ལྔའི་ཆུས་བསྒྱུར་ནས། །འབྲུ་ལྔར་རླབ་བྱིན་རླབས་རིན་ཆེན་ལྔ། །།བསམས་ཏེ་བྲི་བ་དབང་ལྡན་ཕྱོགས་ནས་བརྩམ། །།སྒྲུབ་པའི་དུས་སུ་དཀྱིལ་འཁོར་དག། ལྟེ་བ་མཐིང་རྩིབས་བཞི་མུ་ཁྱུད་ཡང་ན་ཕྱོགས་མཐུན་བཅས། །དེ་མཐའ་མཐིང་ཀ་ལ་རྡོ་རྗེ་དཀར་།
