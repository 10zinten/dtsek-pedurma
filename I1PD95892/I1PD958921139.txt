༼བསྟན་འགྱུར། རྒྱུད། གར་གྱིས་བཞུགས་པ་གྲི་གུག་དང་ཐོད་པ་འཛིན་པའོ། །ཟུར་བཞི་ལ་ཐོད་པ་་་་བཞིའོ། །སྒོ་བཞིར་ཁཱ་གདོང་མ་དང་། འུག་གདོང་མ་དང་། །ཁྱི་གདོང་མ་དང་། ཕག་གདོང་མ་དང་། ཟུར་བཞིར་གཤིན་རྗེ་བསྲེག་མ་དང༌། །་ཕོ་ཉ་་མ་དང་། །མཆེ་བ་མ་དང་། །འཇོམས་མ་རྣམས་བསྐྱེད་ལ། ཨོཾ་ཤྲཱི་བཛྲ་བྷ་ར་ཧི་ཨ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཞེས་པ་ནི་ཏེ་ལོ་པའི་ཕག་མོའོ། །རྗེ་བོ་དག་ལ་མ་གནང་བ་དང་། །རྒྱུས་ལ་དོ་དམ་ཁང་། ཚོ་ཁག་ཁ་རྗེ་ན་རོ་པའི་ཕག་མོའི་མངོན་པར་རྟོགས་པ་ནི། ཧཱུཾ་ལས་པདྨ་འདབ་མ་བརྒྱད་ཀྱི་སྟེང་དུ་རོའི་སྟེང་དུ་པཾ་ལས་རྡོ་རྗེ་ཕག་མོ་ཞལ་ཅིག་ཕྱག་གཉིས་མ་གྲི་གུག་དང་ཐོད་པ་ཁ་ཊཾ་ཁ་འཛིན་པའོ། །སྐུ་མདོག་དམར་མོ་ཕྱེད་སྐྱིལ་གྱིས་བཞུགས་པ་འཁོར་གྱི་ལྷ་མོ་བཅུ་གཉིས་སྔར་ལྟར་རོ། །ༀ་སརྦ་:བུད་དྷ་ཌ་ཀི་ནི་ཡེ་བཛྲ་ཝན་ནི་ཡེ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཞེས་པ་ནི་ན་རོ་པའིའོ། ་་། ་་་ཊ་ར་ཏུ་ཊ་ཅepਕਸਸਸਸਕਾਨਨੂੰਮਾਨਣཡང་ཧཱུྃ་ལས་རྡོ་རྗེ་རྣལ་འབྱོར་མ་དམར་མོ་ཞལ་གཉིས་གཡས་ན་ཕག་ཞལ་སྔོན་མོས་མཚན་པ། ཕྱག་གཉིས་གྲི་གུག་ཐོད་པ་ཁ་ཊཾ་ཁ་འཛིན་པ། འཁོར་ལྷ་མོ་བཅུ་གཉིས་སྔར་ལྟར་རོ། །ཨོ་ཨོ་ཨོཾ་སརྦ་བྷུད་དྷ་ཌཱ་ཀཱི་ནི་ཡེ་བཛྲ་ཝརྣ་ནི་ཡེ་བཛྲ་བི་རོ་ཙ་ན་ཡེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཞེས་བཟླས་ལ་དབང་བསྐུར་བ་ནི་ཞལ་གཉིས་མའོ། །ང་། ལ་ཁཁ་བཁ་ལོ་བའི་ཨའོ་ཏེ་སྐུ་ཚད་སྤྱི་ਮਾਰੂਅਸਲਮਾਲਕਬਲਵਲਵਲདོན་ཐམས་ཅད་གྲུབ་པའི་མངོན་པར་རྟོགས་པ་ནི། ཧཱུྃ་ལས་ཕོ་བྲང་ཕོ་་ལས་རོའི་སྟེང་དུ་དོན་གྲུབ་མ་དམར་མོ་ཞལ་ཅིག་ཕྱག་བཞི་མ་རྡོ་རྗེ་ལྕགས་ཀྱུ་་་ཞགས་པ་ཐོད་པ་ཁ་ཊཾ་ཁ་འཛིན་པ། འཁོར་ནི་ལྔར་རོ། ་། ཨོཾ་བཛྲ་བ་ར་ཧི་ཨ་་།
