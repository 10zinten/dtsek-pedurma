མདངས་སྟོན་སྒྲོགས་གཅིག་ཡིན། །ཨུས་ཎི་ཤ་རི་རབ་མའི་བསྒྲུབ་ཐབས། སློབ་དཔོན་ཤྲཱི་སིང་ངས་མཛད་པ་རྫོགས་སོ། །།།།ན་རྒྱ་གར་གྱི་པན་བྱི་ཏ་དེ་ཉིད་དང་། །བོད་ཀྱི་:ལོ་ཙ་བ་བི་རོ་ཙ་ནས་་་་་་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།རྣམ། དེ་ནས་རྒྱལ་མོ་གཡུལ་སྒྲ་ལ་བབས་སོ། །ཨི་ཐི། །མ་དང་དུ། བདེ་། ནང་དུ་གསན་པ་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»བསྒྲུབ། ་«པེ་»སྟེང་། ཉི«པེ་»བསྲེས། «པེ་རྒྱལ་རིགས་བཞི་རྣམ་པ། ཚེ«པེ་»གླུ་གནོད། །«པེ་རྐྱལ། ®«པེ་གཞུག། དེ་«པེ་»+ཕྱིར། །«པེ་སྡུས། Q«པེ་སྲུང་། «པེ་ཨཱ། །«པེ་»ཏ། «པེ་བཛེ། ཚེ«པེ་བསྔགས། «པེ་»ཨུ་ས། །ཆུ་མས༌བཔ«པེ་ཡ། «པེ་»+ཡ། «པེ་»ལོཙྪ། ་་བོད་དང་། ཚོགས་པ་ད་མོ་པ་དེ་ན་༌། ཏུ། །བསྡུར་འབྲས་རེའུ་མིག་དུ་བསརྒྱུད། （ནབབབཔ::། བཀའ་འདུ། བ་ལུ་། །་བར། དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། །སྡེ་ཚན། །སྔགས། སྔགས། པོད་རྟགས། ཚན་གྲངས། །ན?ན་བ?ན། ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ལ་:::། །།ལྡེབ། །།
