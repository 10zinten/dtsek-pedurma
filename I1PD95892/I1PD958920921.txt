ཉིཊ། །དེ་ནས་ད་བར་འཁྲུབ། །ཏིང་འདྲ་ན། ་་བ་་འདི་་་་་་ན། །:་གི༄༅། ཁྲོ་བོ་རེག་ཚིག་གི་བསྒྲུབ་ཐབས་རྣམས་བཞུགས། ༄༅། །རྒྱ་གར་སྐད་དུ། ཀབྷ་ཡ་ན་ཀྲོ་ཏ་གོས་ཙ་ན་ན་མ། བོད་སྐད་དུ། ཀཁྲོ་བོ་ཁརེག་་་ཚིག་གི་གབསྒྲུབ་པའི་ཐབས་ཞེས་བྱ་བ། ཡི་གེ་པའི་ཕྱག་ཁྲོ་རྒྱལ་རེག་ཚིག་ལ་ཕྱག་་་འཚལ་ལོ། །ཕྱག་གི་ཡུལ་གཏུམ་ཆེན་རབ་འབར་གདུག་པ་འཇོམས། །མུ་སྟེགས་ངན་སྔགས་སྦྱོར་འཇོམས་ལུས་ངག་ཡིད་གསུམ་གྱི་ཕྱག་འཚལ་ལོ། །ཕ་རོལ་གཅོད་པས་མཚོན་ཆ་རབ་ཏུ་འབར་བ་ཡིས། །ཆེད་དུ་བྱ་བ་ཕ་རོལ་སྦྱོར་བ་བཤིག་ཕྱིར་བྲི། །དགྲའམ་རང་་་བསད་བསྲུང་གནད་ཀྱི་མན་ངག་ཟབ་མོ་ནི། །བསྲུང་བཟློག་གསད་མདོར་བསྡུས་གསུམ་ཤེས་པར་བྱ། །དང་པོ་བདག་ལ་མཐུ་གཏད་ནས་སྡིག་པ་བསགས་པ་སྡུག་བསྔལ་རྒྱུ་སྤྱོད་པའི། །སེམས་ཅན་ཀུན་ལ་སྤྱི་ལ་སྙིང་རྗེ་བསྒོམ། །བྱེ་བྲག་དུ་རང་ལ་སྡང་བའི་སེམས་ལྡན་ཞིང་། །ངན་སྔགས་སྦྱོར་བྱེད་ཕོ་ཉ་སྦྱོད། །དམོད་པའི་སྦྱོར་བ་བྱེད་པ་ལ། །གདེངས་གསུམ་བླ་མའི་ཞལ་ལས་ཤེས་ཆེད་དུ་གཏང་བའི་སྙིང་རྗེ་བསྐྱེད། །བདག་གཞན་གཉིས་ཀ་བདེ་།
