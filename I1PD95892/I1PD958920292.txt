ཆོ་། །ཀུན་ཏེ་ཚེ་རབས་ཀྱི། ས་ཕྱོགས་དང་ད་དུས་ངན་པ་དང་། ༄༅། །འཕགས་མ་སྒྲོལ་མ་ལ་བསྟོད་པ་སྨོན་བྱེད་་/ལམ་གྲུབ་པ་ཞེས་བྱ་བ། །།།་ཏོ་ན་ཁ་པའི་ལམ་། ལ་ལ་་ཎ་ཌའeཀཔར་::དུས་དེ་ལས་ལ་ཚེBਨਸyਲਇਸਸੰਸਥਸਸਲੋਂਨਰབར་ཚོ་ཚུ་བ་་་་་ཁྲོཏྟ་ར་ན། གས་ལ་བལ་ཚོད་ཀྱི༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏ་ར་སྟོད་ཏྲ་པྲ་ནི་དྷ་ན་ན་མ། བོད་སྐད་དུ། འཕགས་མ་སྒྲོལ་མ་ལ་བསྟོད་པ་སྨོན་ལམ་གྲུབ་པ་ཞེས་བྱ་བ། རྗེ་བཙུན་མ་འཕགས་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །ཡི་གེ་ལྗང་ཀུ་ཨུད་པལ་ལས་བྱུང་བའི། །།སྐུ་མདོག་ལྗང་ཀུ་ཞལ་གཅིག་ཕྱག་གཉིས་མ། །ཕྱག་གཉིས་སྐུ་ལ་རིན་ཆེན་དུ་མས་སྤྲས། །ཡི་དམ་སྒྲོལ་མ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །དོན་གྲུབ་ལ་སོགས་རིགས་ལྔས་དབུ་བརྒྱན་ཅིང་། །བཅུ་དྲུག་ན་ཚོད་འཛུམ་པའི་བཞིན་དང་ལྡན། །གཟི་བརྗིད་བཟོད་དཀའ་དག་ཚོགས་རྟག་པར་འདུལ། །ཡི་དམ་སྒྲོལ་མ་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །ཕྱག་གཡས་མཆོད་སྦྱིན་ཕྱག་གཡོན་ཨུད་པལ་འཛིན། །སྣ་ཚོགས་པདྨ་ཉི་མའི་དཀྱིལ་འཁོར་ལ། །གཡས་བརྐྱང་་་་གཡོན་བསྐུམ་ཚུལ་གྱིས་རབ་ཏུ་བཞུགས། །ཐུགས་རྗེའི་སྤྱན་གྱིས་འགྲོ་ལ་རྟག་ཏུ་གཟིགས། །སྐུ་ལ་འཇིགས་པ་བརྒྱད་སྐྱོབ་སྒྲོལ་མ་འཕྲོ། །འཁོར་བའི་སྡུག་་་་།
