ལྷ་སྲིན་སྡེ་བརྒྱད་ཆད་པ་བཟློག། །།མ་ཧཱ་མང་ས་ལ་ཁཱ་ཧི། མ་ཧཱ་རཀྟ་ལ་ཁཱ་ཧི། མ་ཧཱ་ཙི་ཏ་ལ་ཁཱ་ཧི། མ་ཧཱ་གོ་རོ་ཙ་ན་ལ་ཁཱ་ཧི། མ་ཧཱ་བ་སུ་ཏ་ལ་ཁཱ་ཧི། མ་ཧཱ་་་ཞིང་ཆེན་ལ་ཁཱ་ཧི། ཨ་མྲི་ཏ་སྲོག་ཨངྷ་ལ་ཁཱ་ཧི། །།།་བདེ་བ་ཧཱུྃ་བྱང་ཕྱོགས་ལྗང་ནག་གྲུ་གསུམ་གཞལ་ཡས་ནས། །།ལྗང་ནག་མེ་དཔུང་ཀློང་དཀྱིལ་ནས། ཟོ། གནོད་སྦྱིན་ནག་པོ་བྱ་རོག་གདོང་། །སྐུ་མདོག་ལྗང་ནག་་དུག་ཆེན་འཁྱིལ་བ་འདྲ། །།ཁ་རླངས་སྔོ་དམར་ནད་ཀྱི་ན་བུན་འཁྲིགས། །མི་་བཟད་གདུག་པའི་ཚ་ཚ་བུ་ཡུག་འཚུབ། །ཁྲག་གི་རལ་པ་དམར་པོ་གྱེན་དུ་འཁྱིལ། །རལ་པའི་བསེབ་ནས་ཐོག་དང་སེར་བ་འབེབས། །།གཡས་ཀྱི་གྲི་གུག་དམ་ཉམས་སྲོག་རྩ་གཅོད། ::། གཡོན་གྱི་ཐོད་ཁྲག་དགྲ་བོའི་སྙིང་ཁྲག་འཐུང༌། །སྐུ་སྨད་་དཔའ་བོ་སྟག་གི་ཤམ་ཐབས་ཅན། །དགྲ་བགེགས་མ་ལུས་དབང་དུ་བསྡུད། །ལྗང་ནག་སྦྲུལ་གྱི་སྐེ་རགས་བཅིངས། །།གནོད་སྦྱིན་ནག་པོའི་ཚོགས་ཀྱིས་བསྐོར། །དམ་རྫས་དམར་གྱི་གཏོར་མ་འདི་བཞེས་ལ། །ལས་ཀྱི་རིགས་ཀྱི་འོད་སྟོང་ཐམས་ཅད་བཟློག། ཀྱ། སྡིག་ཅན་དམ་ཉམས་དགྲ་བོ་ལ། །ནད་དང་མཚོན་གྱི་ཆར་པ་ཕོབ༑། འོ་་་། བྱད་ཁ་ཕུར་ཁ་བཟློག་ཅིག་བྱ་རོག་གདོང་། །དུས་ལ་བབ་པོ་བྱ་་་་རོག་གདོང༌། །།།པདེ་ལ། བྱ། ཆུང་བ་ཁགཀ་ཧཱུཾ་སྟེང་གི་ཕྱོགས་ནས་བཟློག་པ་ནི། །ཁྲ་དང་བྱ་ཀླག་བཤོག་ལྡང་འཁྲིགས། །སྦྱོ་བྱོ་སྡང་བའི་དགྲ་ལ་སྦྱོ། །བཟློག་བཟློག་བྱད་ཁ་ཕུར་ཁ་བཟློག། ཤར་གྱི་་་ཕྱོགས་ནས་བཟློག་པ་ནི། །ཁྲོ་བོ་མཐིང་ནག་བྱ་རོག་གདོང་། །ཤ་ལོག་ཅོད་བྱེད་སྡང་མིག་ལྟ། །།བློ་སྦྱོ་སྡང་བའི་དགྲ་ལ་སྦྱོ། །།བཟློག་བཟློག་གནོད་པའི་བགེགས་ལ་བཟློག། ལྷོ་ཡི་ཕྱོགས་ནས་བཟློག་པ་ནི། །སེང་སྟག་དོམ་དྲེད་ཅེ་སྤྱང་གཟིག། སྤྱན་ཟན་མང་པོས་ས་གཞི་གང། །སྦྱོ་སྦྱོ་སྡང་བའི་དགྲ་ལ་སྦྱོ། །PIK།
