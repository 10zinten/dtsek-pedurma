།བསྟན་པ་བསྲུང་བ་དང༌། །སེམས་ཅན་ཀུན་ལ་སྨན་སླད་དུ། །བྱང་ཕྱོགས་གནས་ན་བཞུགས་པ་རྣམས། །ཡིད་བཞིན་སྤྲུལ་པའི་གཞལ་ཡས་འདིར། །འཕྲིན་ལས་མཛད་ཕྱིར་གཤེགས་སུ་གསོལ། །ཨོཾ་བྷི་ཤ་རཱ་མ་ན་ཡེ་ས་པ་རི་བ་རྭ། བཛྲ་ས་མ་ཡ་ཛ་ཛ། ཛgཧཱུྃ་བཾ་ཧོ། བརྡའ་རྒྱུད་ཕྱག་རྒྱ་བཅས་ནས་ནི། །བྷི་ཡིག་སུམ་ཅུ་རྩ་བདུན་སྲིང༌། །སྙིང་སྙིང་འདོད་དང་མགོ་སྦྲིད་དང༌། །རང་རང་སྔགས་ཀྱིས་མཆོད་པ་དབུལ། །དགུག་གཞུག་དབང་བསྐུར་བྱིན་རླབས་་་བཟླས༑་། ༀ་ནི་ལམ་བྷ་ར་དྷ་ར་བཛྲ་པ་ནི་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བྷི་ཤ་མ་ན་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཛཾ་བ་པ་ཛཱ་དྷ་ཡ་སྭཱ་ཧཱ། ཨོཾ་པུརྣ་བྷ་ཀྲ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་མ་ནི་བྷ་ཏྲ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཀུ་བེ་ར་ཡེ་སྭཱ་ཧཱ། ཨོཾ་སམ་པྲ་ཛ་ན་ཡེ་སྭཱ་ཧཱ། ཨོཾ་གུ་ཧས་ན་ཡེ་སྭཱ་ཧཱ༑ཨོཾ་པན་ཛི་ཏ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་པི་ཙི་ཀུན་དྷ་ལི་ཡེ་སྭཱ་ཧཱ། སརྦ་ཡག་གླུ། ན་ག། །ཨནཾ། དུ་པ། །པུས་པོ། དི་པཾ། གན་དྷཾ། ནི་བེ་ཏེ། ཤཔྟ། ཨོ་ཨོཾ་ཧཱུྃ། ཨི་དཾ་བྷ་ལི་གྷི་ཧ་ན་དུ། །ས་མ་ཡ་རཀྟ་དུ། །མ་མ་སིདྡྷི། མེ་པྲ་ཡད་ཙ༑སརྦ་ཀརྨ། སཱ་ཏ་ཨ་མོ་ཀ་སྭཱ་ཧཱ། རིན་ཆེན་བདུན་དང་བཀྲ་ཤིས་་་བརྒྱད། །བཟའ་བཏུང་ལོངས་སྤྱོད་ཇི་སྙེད་ཀུན། །ཚོགས་མཆོད་རྒྱ་ཆེན་དབུལ་བར་བྱ། །རྟ་དང་གླང་པོ་ཁང་བཟངས་དང༌། །གསེར་དངུལ་ལ་སོགས་ནོར་གྱི་དབྱིག། ཕྱི་ནང་གསང་གསུམ་རྟེན་རྣམས་ཀྱང༌། །མགོན་པོ་འཁོར་དང་བཅས་ལ་དབུལ། ཚོ་། ནེར་བྷན་ཛ་ན་སྲེག་པ་རླབ། །ལིང་ཁ་བཅས་པ་་་། དགུག་གཞུག་བྱ། །ཕུར་པས་གདབ་ཅིང་མཚོན་གྱིས་བཅད། །དེ་ནས་བཀའ་སྒོ་འཕྲིན་ལས་བཅོལ། །།།།ལ། བདག་ཏུ་ལྟ། དེ་རྗེས་ཧཱུྃ་བཞེངས་ཅིག་བཞེངས་ཅིག་སྐུ་བཞེངས་ཅིག། རི་རྒྱལ་ལྷུན་པོའི་བྱང་། ཕྱོགས་ནས། ་། གནམ་ལྕགས་འབར་བའི་གཞལ་ཡས་ཁང་། །དྲེགས་པ་སྣོལ་་་།
