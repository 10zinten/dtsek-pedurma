་་་ལ་ཙཀདེ་ཀོ་པན་འོང་ང་ལ་བདེ་བ། བོད་ཁོང་དང་། །མ་ཏི་ལོ་ཁོ་ལ་གདོན་དང་། དད་པ། །དེཔའི་ཁྲོད་དུ་བྱོན། 'ཏོང་པ་དང་། བུད་ངཏའི། །སྐྱོང་། །མ་བྱེད་རྒྱུ་ཙ་པ་དེ་བད་བློན་་་་་ར་བ་མ་ཚེ་ལས་ཀྱི་༄༅། །སྔགས་ཀྱི་རྒྱལ་པོའི་དམ་ཚིག་དངོས་གྲུབ་། སྒྲུབ་པའི་ཐབས་ཞེས་བྱ་བ། :། དེདེ་ཁྲི་བ་གཏོང་། བྱ་བྱེ་བ་ཆེ། །བདག་་་་་་་་་་་་་་::ཀྱི་ཁྲོད་དབབ། འཁ་བྲིས་བྱེད་པ། །»/ཏ་ཀ:་ཊ་ལྷོ་༄༅། སྔགས་ཀྱི་རྒྱལ་པོའི་དམ་ཚིག་དངོས་གྲུབ་སྒྲུབ་པའི་ཐབས་བཞུགས་སོ། །:༄༅། །རྒྱ་གར་སྐད་དུ། མནྡཱ་རཱ་ཛཱ་ས་མ་ཡ་སིདྡྷི་སཱ་དྷ་ནཱ་མ། བོད་སྐད་དུ༑སྔགས་ཀྱི་རྒྱལ་པོའི་དམ་ཚིག་དངོས་གྲུབ་སྒྲུབ་པའི་ཐབས་ཞེས་བྱ་བ། དེ་བཞིན་གཤེགས་པ་གདོན་མི་ཟ་བར་གྲུབ་པ་ལ་ཕྱག་འཚལ་ལོ། །བླ་མ་དང་ནི་ལྷག་པའི་ལྷ། །མཁའ་འགྲོའི་ཚོགས་ལ་ཕྱག་བཙལ་ནས། །ཐུན་མོང་སྒྲུབ་པའི་རིམ་པ་བྲི། །ཕྲིན་ལས་སྒྲུབ་པར་འདོད་པ་ཡིས། །གནས་ནི་ནགས་ཚལ་དུར་ཁྲོད་དམ། །རྩེ་མཐོ་བྲག་དམར་མེ་ཏོག་ལྗོངས། །དབེན་ཞིང་ཉམས་དགའ་སྒྲུབ་པའི་གནས། །དེ་རུ་འདུག་ལ་སྤྲོས་པ་བཅད། །ཤོག་བུ་གྲོ་ག་དུར་ཁྲོད་་་རས༑། དུད་དང་གུར་ཀུམ་སིནྡྷུ་ར། །དུར་ཁྲོད་སོལ་བས་སྣག་ཚའོ། །མུ་་་ཁྱུད་གསུམ་པ་ཆུ་དཀྱིལ་འཁོར། །་། དབུས་ཀྱི་རི་མོའི་མཚན་ཉིད་ནི། །ས་དང་་ཆུ་དང་མེ་དབྱེ་ཡིས། །རྡོ་རྗེ་བུམ་པ་ལྕགས་ཀྱུ་ཡི། །བསྣོལ་དང་བརྩེགས་པ་གོ་—།
