གྱིས་ལེགས་པར་གཙང་། །བསྟན་བཅོས་འདི་ཡིས་རྣམ་པ་ཀུན་དུ་ཡོངས་སུ་ངལ། །གང་ཡང་དེར་ནི་ཟབ་མོ་གསལ་བར་མི་འགྱུར་རོ། །ཇི་ལྟར་འདི་ནི་དེ་བཞིན་དུ། །ཚིག་དོན་བསྡུ་བཤད་ལ་སོགས་གསལ། །བསྙེན་པ་རྡོ་རྗེ་བཞི་པོ་ཡང་། །ཁོ་་་བོས་ཡང་དག་བསྟན་པ་ཡིན། །དཔལ་གསང་བ་འདུས་པའི་རྒྱན་ཞེས་བྱ་བ། །སློབ་དཔོན་དྲི་མེད་སྦས་པས་མཛད་པ་རྫོགས་སོ། །་::། ་་་་་པཎྜི་ཏ་སུ་ན་པ་ཤྲཱི་མི་ཏྲ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་དར་མ་་་་གྲགས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།། ། བྱུང་། །ཁཚུ་ག་ར་མ་མ། །བསྡུར་མཆན། ཟླ་བ་ཡིན་པ་དང་། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་མ། «པེ་ལ། །»«སྣར་»ལོ། «པེ་»«སྣར་»བརྙས། (ན«པེ་»«སྣར་འབྲེང༌། དེབ་དེའི་མེད«པེ་»«སྣར་པ། ས་། །སྡུག་ཁང་དུ་ཞི་བ། །ཁོར། །ཁ་མ་གང་བ་(«པེ་»«སྣར་པའི། །«པེ་»«སྣར་»པ། འ«པེ་»«སྣར་»—དང༌། «པེ་»«སྣར་»སོགས། (འདིའི་འོག་གི་སྩོགས་ཞེས་པ་རྣམས་«པེ་»《སྣར་ལ་སོགས་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་གསུམ། P«@«པེ་»«སྣར་སྐུར། དབང་འདུས་ཀྱི། །བདེ་བ་དང་། ཁ་སེང་(«པེ་»《སྣར་»རྟོག/«པེ་»«སྣར་ཡིས། །ཁ་ཙ། །།དོན་ངོ༌། །+(«པེ་»«སྣར་བྱེད། «པེ་»«སྣར་བསླུ། «པེ་»«སྣར་»«ཅོ་»བརྗོད། ༄༅«པེ་»«སྣར་»ཆོདང་། «པེ་»«སྣར་»དེའི། «པེ་»«སྣར་ལ། ཚེ«པེ་»«སྣར་བཞིན། :«པེ་»ཀྱི་«སྣར་ཀྱི། །དེ་ལ་བདག་པའི་ས་ལམའབས་ཀྱི་ཀྱི་(«སྣར་»གསུམ། 《ི«པེ་»«སྣར་»རྟོག་འ«པེ་»«སྣར་བྱས། ་«པེ་»«སྣར་»ཀྱི། «པེ་»འགྲེལ་པ། །«པེ་»«སྣར་»མཐའི། «པེ་»«སྣར་»དགོངས། །(«པེ་《སྣར་»འཕགས། («པེ་»«སྣར་»སྨྲ་བ། «པེ་»«སྣར་ལ། ®«པེ་»《སྣར་»བརྙས། ནོ«པེ་»«སྣར་»རིག@«ཅོ་»རྒྱུན། «པེ་»«སྣར་»དཀྲུབ། །ག་ད་བ་མའ། །(«ཅོ་»ཡིས། «པེ་《སྣར་»ནི། ་འ«པེ་»«སྣར་»རིགས། ?«པེ་»«སྣར་+པ་ཡི། །«སྣར་»།
