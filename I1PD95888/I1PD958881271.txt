བསྟན་འགྱུར། རྒྱུད། །།བ་རྫོགས་སོ། །དེ་བ། ་་་ལནདུའི་བློ་། དེ་ཚོ་ལ་རྟག་པབ་ལུ་ག་ན་པ་བ་བཀན། འདས། །ད་ཚུན་ཝ་བློ། ਨੂੰਕਰਨਲਈਇਕབསྡུར་མཆན། པ་གཡས་པས། བོད་བ། པ་པོ། འི་ཚོ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་བརྟན། ་«པེ་»གདབ། །འོ་ན་ཀྱི«པེ་»ཀྱིས། «པེ་»ཡི། «པེ་»འཕོངས། ཆེས«པེ་»ཥ་ན། «པེ་»བཛཱ། འདྲད་འདྲག་«པེ་»སྦྲུལ། «པེ་གཟུགམཆོད་ཡོལ་ཏེ། ར་འོང་ནིཏེ་ཁྱོད་་་་། ཀཚོ་ཚོ་ལ་བསྡུར་འབྲས་རའུ་མགབ་དང་བར་་་་་་་རྒྱུད། (ནི་དང་འགརབབར་Eད། ད་སྤྲོ་གར་དཔེ་གཞི། །སྡེ་དགེ། །།སྡེ་ཚན། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། །བནོའཚོ་ན་ན། བ་—ནཔེ་ཅིན། སྣར་ཐང་། རྒྱུད་འགྲེལ། ལྡེབ། །ཚིག་རྐང་། ཚེག་ཁྱིམ། པ་བ། ལ་དོ་ལསྐྱོང་ཚོང་ཚོ་དེ་ལ་བདག་དབང་། ཁ་བྱང་དང་གྲོག་མ་བྱེ་བ་ਲੋਕਇਸਨੂੰਮਾਰਕੇਪਿੰਡਭੂਸਉਏਇਸਕਿਸਮਨੂੰਦੋਗਈਐਡਹਿਨਬੋਲ།
