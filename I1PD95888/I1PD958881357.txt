རབསྟན་འགྱུར། རྒྱུད། ༼gདམ་གྱི། །བྱང་ཆུབ་སེམས་ནི་བསྐྱེད་པར་བྱ། །སྡོམ་པ་ཡི་ནི་དམ་ཚིག་དང་། དགེ་བའི་ཆོས་རྣམས་སྡུད་པ་དང༌། །སེམས་ཅན་དོན་བྱར་བཅས་པ་ཡི། །རྣམ་པ་གསུམ་དང་ལྡན་པར་བཤད། །རྡོ་རྗེ་གོ་ཆའི་ཕྱག་རྒྱར་བཅས། །ཨོ་་བཛྲ་ཀ་བ་ཙ་རཀྵ་ཨ་ཏི་ཏི་སྟ་ཧཾ། དེ་ནས་སྐུ་གསུང་ཐུགས་ཀྱིས་བརླབ། །ཀ་ཡ་ཨོ༑བག་ཨོཾ། ཙིཏ་ཧཱུྃ། །འཁོར་ལོ་ཕྱག་རྒྱ་ཏིང་འཛིན་སྔགས། །བར་དུ་གཅོད་པ་གསལ་བར་བྱ། །ཨོཾ་ཛྙཱ་ན་ཙཀྲ་ཕཊ། ང་ཡིས་བཤད་ཕོག་ཐམས་ཅད་ནི༑། དམ་པས་གསུངས་པའི་ཚིག་ཡིན་པས། །ཁྱོད་ཀྱང་དེ་ལས་མ་འདའ་་་ཞིག། ཨོཾ་བཛྲ་ས་མ་ཨ་ཧྲིཐ་ཨ། བྱང་ཆུབ་རིགས་ཅན་ཉིད་དུ་ནི། །སྔགས་ནི་བརྗོད་པས་གྱུར་པའོ། །ཨོ་བྷོ་དྷི་ཙིཏ་ཨ་སཾ་མེ། ཏྲི་ས་མ་མེ། མ་ཧ་ས་མ་ཡ་སྭཱ་ཧཱ། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། །་་། འ་འོ། །ག་བ::ཟླ་ཆུ། ལུང་པ་ན་བ་དང་། བ་ཏི་ཞུ་ནུས་གསུམ་པ་དཀྱིལ་འཁོར་མཇུག་གཞུག་པ། །སློབ་དཔོན་ཞུགས་དང་སློབ་མ་གཞུག། གཤེགས་དང་ལྷག་ཆད་ཁ་བསྐང་དང་། །མཐུན་པར་སླར་གཤེགས་་གསོལ་བ་དང༌། །ལས་བཞི་མཆོད་ནས་བདག་ཉིད་ཀྱིས། །དཀྱིལ་འཁོར་་་་་སེམས་ཀྱིས་ནམ་མཁའ་ལ། བཏེག་ནས་བྱང་ཆུབ་སེམས་ཀྱིས་ནི། །ཞུགས་པ་ཉིད་དུ་ཐུབ་པས་གསུངས། །བདུག་སྤོས་ལག་དུ་ཐོགས་ནས་སུ། །བློ་གྲོས་ལྡན་པས་ལྷ་རྣམས་ལ། །གུས་པས་གསོལ་བ་ལེགས་པར་གདབ། །དེ་འོག་སྒོ་བཞི་་་དབྱེ་བར་བཤད། །ཞུགས་ལྷ་གཉིས་སུ་མེད་བསྒོམས་ལ། །འགྲོ་བ་དྲང་བའི་་་ལས་བྱའོ། །བཅོམ་ལྡན་ནམ་མཁའི་སངས་རྒྱས་ཀུན། །ཐུགས་རྗེ་ཆེན་པོ་མཐུ་ལྡན་ལགས། །བདག་ཉིད་རབ་ཏུ་བསྒྲལ་བའི་ཕྱིར། །རང་བཞིན་ཏིང་འཛིན་གཟུགས་བརྙན་ལྟས། །བདག་ལ་དབང་མཆོག་འདིར་སྩོལ་ཅིག་། དཀྱིལ་་་།
