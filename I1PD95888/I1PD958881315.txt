བསྟན་འགྱུར། རྒྱུད། ཏ་ཀ་ཏ། ཙ་ར་ཙ་ར། བནྡ་བནྡ། ད་ཤི་ཏི་ད་ཤ་ན། སརྦ་ཏ་ཐཱ་ག་ཏ། ཨ་ནུད་གཉའ་དེ། །བྲ་བ་དར་མ་ལ་མ་ཏ། བྷི་ཛ་ཡ། །བ་ག་བ་ཏི། ་ཀུ་རུ་་བི་ཀུ་ལེ། ལེ་ལུ་པུ་རེ་སྭཱ་ཧཱ། དེ་ཡིས་བདུད་སྲིན་རོད་རོད་པོ། །རིགས་ལྔའི་འཕྲ་མེན་གཡོག་བཅས་ཀུན། །ཅི་བགྱི་ཞེས་སུ་བརྗོད་པ་དང༌། །ཁྱེད་ཀྱིས་་བགེགས་བསྐྲད་ཚར་གཅོད་དང༌། །ལས་ཀྱི་བདག་པོས་གྲོགས་བྱའོ། ་། དུས་་་གསུམ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །མངག་གཞུ་ཕོ་ཉའི་ལས་མཛད་པ། །ཀླུ། ཁྲོ་་་ཆེན་བདུད་རྩི་འཁྱིལ་པ་ཡིས། །བགེགས་ཀྱི་སྦྱོར་བ་རྒྱུན་ཆོད་ཅིག་། ན་མོ་བཛྲ་རཏྣ་ཏྲ་ཡ་ཡ། །ན་མ་ཙཎྜ་བཛྲ་པ་ན་ཡེ། མ་ཧ་ཡཀྵ་སེ་ན་པ་ཏ་ཡ། ན་མོ་བཛྲ་ཀྲོ་ཏ་ཡ། མ་ཧཱ་དང་ཤོད་ཀ་ཏ། ་ཀབེ་ར་བ་ཡ། ཨ་སི་མུ་ས་ལ། ཌ་པ་ར་ཡ། ་པ་ར་ཤུ་པ་ཤ་ཏཐཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཨ་མྲྀ་ཐ་ཀུཎྜ་ལི། འོཁ་ཁ་ཁ། ཁ་ཧི་ཁ་ཧི་ཁ་ཧི། ཏིཥྛ་ཏིཥྛ་ཏིཤྩ། བནྡྷ་བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན་ཧ་ན། གླི་ན་གླི་ན་གླི་ན། གརྫ་གརྫ་གརྫ། སྤོ་ཏ་ཡ་སྤོ་ཏ་ཡ་སྤོ་ཏ་ཡ། ་བི་སྤོ་ཏ་ཡ་བི་སྤོ་ཏ་ཡ་བི་སྤོ་ཏ་ཡ། སརྦ་བིགྷན། །མ་ཧཱ་ག་ན་བ་ཏི། ཛི་བི་ཏནྟ་ཀ་ར་སྭཱ་ཧཱ། ཁྱོད། ལས་ཀ་དང་། ཚོ་བ་ཁོ་ཚོ་འོ་་་ཀྱི་ཞེས་དེ་བཞིན་དུ་སུ་དང་ཨོཾ་དང་ཧས་བསྒྱུར་ཏེ། །རད་ན་པདྨ་ཀར་མས་་མཚམས་བཅད་དོ། །བསྲུང་བའི་ལས་དག་བྱ་བ་ཡང་། །ཌ། ཇི་བཞིན་སྟེགས་བུ་བརྩིགས་ནས་སུ། །ལུགས་མའི་ཕྱག་རྒྱའམ་བདག་གནས་ཏེ། །རྒྱུན་ལྡན་དེ་་་ཡིས་བདག་གཞན་བསྲུང་། །བསྲུང་ཆེན་མོ་དང་ཕྱག་རྒྱ་ལྔ། །།བརྟན་རྒྱལ་་་བརྗིད་ལེགས་བདེ་བར་གནས། ་ཚེ། བཛྲ་བཞི་ཡིས་བསྲུང་བར་བྱ། །ས་རྣམས་་་་གཞན་གྱིས་མ་གཙེས་དང་། །འབབ་ཆུ་རྒྱ་མཚོར་མ་གྱུར་ལ། །ཉི་ཟླའི་མིག་་་དང་ཡིག་འབྲུའི་ཚོགས། །སྦྱང་ཞིང་སྤེལ་བའི་ལས་བྱའོ། །ཀུན་རྫོབ་དོན་དམ་།
