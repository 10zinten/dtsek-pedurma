རྒྱུད། འོ་ཀྱི་ཆོས། །གཞན་ནས་ཡོད་འོང་མ་ཡིན་ཏེ། །ཐབས་ལ་བརྟེན་པའི་ཤེས་རབ་་་ཉིད༑། དེ་ལྟ་བུའི་ངང་དུ་བྱུང་ཞེས་པ་དང༌། ཡང༌། ས་རྒྱུ་འབྲས་ཡེ་ནས་དབྱེར་མེད་ལྔ་ཕྱི་མ་ཡིན་པས། །འབྲས་བུ་ལམ་དུ་གྱུར་ནས་དུས་གཅིག་མཉམ་་་པར་སྦྱོར། །ཅེས་པ་ལ་སོགས་པ་གསུངས་སོ། །ཀྱི་རྒྱུ་བ་ਲਈਸਸਸਸਕਰੰਸਨੂੰਸਦਾགསུམ་པ་ནི་དེ་དག་ཤེས་པ་དང༌། གྲུབ་པར་བྱེད་པའི་ལས་བཏགས་ཏེ། དེ་ཡང་བློའི་བྱེ་བྲག་གིས་ཐ་དད་པར་སྣང་བ་ཁོ་ནའོ། །།དེ་ལ་དོན་དྲུག་སྟེ། ངོ༌བོ་ངེས་ཚིག་དགོས་ཆེད་དང་། །འབྱུང་ཚུལ་གཉེར་བཤད་ཐབས་ཀྱིས་དབྱེ་ཞེས་པའོ༑། ངོ་བོ་ལ་གསུམ་སྟེ། ཕྱག་ན་རྡོ་རྗེས་དབང་བསྐུར་བ་ལ་སོགས་པ་བྱ་བའི་རྒྱུད། དེ་ཁོ་ན་ཉིད་ཐུབ་པ་ལ་སོགས་པ་ཐུབ་པའི་རྒྱུད་དང་། ཐབས་ཀྱི་ཐེག་པའི་རྒྱུད་དོ། །དེ་ལ་ཡང་གསུམ་སྟེ། དཔལ་གསང་བ་འདུས་པ་ལ་སོགས་པ་ཐབས་ཀྱི་རྒྱུད་དང་། དཔལ་བདེ་མཆོག་ལ་སོགས་པ་ཤེས་རབ་ཀྱི་རྒྱུད་དང་། དཔལ་འབུ་ཏ་འབྱུང་བ་ལ་སོགས་པ་མ་ནིང་གི་རྒྱུད་དོ། །དེ་ལ་དཔལ་གསང་བའི་སྙིང་པོ་འདི་ནི་ཐམས་ཅད་ཀྱི་དོན་དང་རྒྱུ་འབྲས་ཚང་ཞིང་འབྲེལ་པའི་ཕྱིར། རྒྱུད་ཐམས་ཅད་ཀྱི་སྤྱི་ཡིན་པར་གསུངས་སོ། །ན་ག་ཚོ་ལ་བཤད་ན་ད་ਲਉਉਲਟਸਿੰਘਨੂੰਵੀਸੋਸ਼ਲਮਸਊਸਾਂਝངེས་ཚིག་ནི། རང་བཞིན་འབྲས་བུ་རྣམ་གཉིས་ལས། །ཤེས་ཐོབ་བཏགས་པའི་ངེས་ཚིག་གཉིས། །རྒྱུ་རྐྱེན་ཐབས་འབྲེལ་རྒོས་རྫོགས་ཕྱིར། །རང་གི་ངེས་ཚིག་བཞི་རུ་འགྱུར་ཞེས་པ་གོ་སླའོ། །།ཁག་ཏུ་གཙོ་འམ་ན། ད་གད་རྒྱུ་ག་ན་འདུག་བ་ཚོ་ཆེ་ས་ཀ:རྒོས་ཆེད་ནི། །རང་བཞིན་གྱི་རྒྱུད་མི་འགྱུར་བའི་ཡོངས་སུ་གྲུབ་པ་སྟེ། —།
