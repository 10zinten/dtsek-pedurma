ཐུགས་ཀྱི་ཐིགས་པ་ཞེས་བྱའོ། །«པེ་ཙོམ། དཔལ་འདུ་བ་དང་ཁྲུང་། «པེ་»བརྟུལ། ®«པེ་»ཟློགདུ་མེད་པ་བསྐྱེད་«པེ་སྒྱུར། །«པེ་རུ། ®«པེ་སྒྱུར། །«པེ་སྤྱོད། ལུ་ད་པ་དང་། པ་དང་། ཡང། དག་པར་བརྗོད། ཟླ::ཀྱབསྡུར་འབྲས་རེའུ་མིགནི་རྒྱུད། (ས་གནས་ཅོ་ནེ། །སྡེ་དགེ། །།ལྟ་བ་དང་། པེ་ཅིན། །རྒྱུད་འགྲེལ། སྣར་ཐང་། རྒྱུད། །།དཔེ་གཞི། སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ལེའུ། །ལྡེབ། ཕྲེང་། །ནན་བ་བ་ཚིག་རྐང་། །ཚེག་ཁྱིམ། ::ཀྱིནས་::ཀྱིངབ། བཀྱི་བཀལ་བ། ཀ་པ། ་་/ར་ཎའི་་། ས་ལ་ཁཀོ་བ་དག་ལ་བདེན་པ་ཀ་ག་ཎ་པ་དང་པོ་གཏོ་གཞི་བྱའོ་ཏའོ་ལ་ད་པ་ཉིད་ཡི་ཤེས་ཕྱིར། བར་སྐབས་ནས་བྱེད་ན་བྱ་བ་ང་ལ་བདེ་ਰਸਮਸੇਸਜੇਸਥਾਸ਼ਰੇਤਈਲਸਲੋਂਨਵੇਂਸਨਉਸཁ་བ་ཕྱི་ལ་བཏོ་བོ་ཚུ་བཏྟ། བོད་སྐད་དུ། །ཏེ་དེ༌ཞི་བྱས་ན་ཀུན་ཀྱང་འོས་འཚོ་བའི་བམ་དེ་ན་ཁ་བའི་ན་ཎ་ན་ང་ལ་རྒྱུ་མ་གནང་། དགའ་བ་དག་བྱེད་རྒྱ། ང་ཡི་ཡོལ་བ། ཚཊ། ད་པ་དང་། དུ་བ་དང་་་་་ཚིག་དེ་། ནམ། ་བ། ར་_།
