བསྟན་འགྱུར། རྒྱུད། །བུའིདག་བཤད། །ལྷོ་ཕྱོགས་རྒྱས་པ་ནུབ་དབང་བྱ། །།བྱང་དུ་དྲག་པོ་ཧོ་དུ་བཤད། །དཀྱིལ་འཁོར་དྲུང་ངམ་ཡང་ན་ནི། །།གྲོང་ཁྱེར་ཕྱི་རོལ་རྗེའུ་རིགས་དང་། ཀྵ། རྒྱལ་རིགས་སྨད་འཚོང་སྨིག་མ་ཁྱིམ། །སྔགས་པ་ནུབ་དང་ལྷོ་ཤར་ལྷོ། །འདུག་ལ་མཆོད་པའི་དོན་དག་ནི། །གང་གི་གཙོ་དང་ཐབས་དང་ནི། །ཤེས་རབ་་ཚུལ་དུ་ལྡན་པའི་དོན། །ཧོ་མ་ཁུང་གི་བྱེ་བྲག་ཀྱང་། ཌ། དེ་ཅི་རིགས་པར་བྱ་་བར་བཤད། །དེ་ལ་སོགས་ཏེ་གཙོ་ཡུམ་ནི། །ཁུང་གི་མཐིལ་དུ་ལྷག་པར་དགོད། །མེ་འོད་ངོས་ལ་རྡོ་རྗེ་ཁ། །འཁོར་རྣམས་ཕྱི་རོལ་ཡང་དག་དགོད། །རོ་སྲེག་་་་མཎྜལ་ཁྲུ་བཅུ་གཉིས། །ཁྲུ་བཞི་པ་ན་ཁྲུ་དོ་ཟབས། ་་་། བརྐོས་པའི་དབུས་སུ་་འཁོར་ལོ་ནི། །རྩིབས་བརྒྱད་པ་ལའང་བྱང་ཆུབ་ཀྱི། །མཆོད་རྟེན་ཀུན་རིག་་་ཕྱག་རྒྱ་སོགས། །རིགས་བཞི་དང་ནི་ཡུམ་བཞི་ཡི། །དེ་མཐའ་རྡོ་རྗེ་རྩེ་ལྔས་་་བསྐོར༑། ཁ་ལ་མེ་འོད་བྲི་བར་བྱ། །།དེ་མཐའ་པདྨ་འདབ་མ་ལ། །སེམས་དཔའ་སྐལ་བཟང་ལ་སོགས་རྣམས། །་། བྲི་བྱ་གྲུ་ལ་ལྷ་མོ་བྲི། །འཇིག་རྟེན་པ་ཡི་ལྷ་རྣམས་ཀྱང་། །ཕྱི་རིམ་ཕྱོགས་ཕྱོགས་དག་དུ་བྲི། །་། གཞུང་དང་འགལ་བྱས་ཉེས་སྡིག་བཤད། ་། སྡིག་པ་ཆེན་པོ་བསྐུར་བཏབ་ཡིན། །སྟོན་པའི་བསྟན་པ་་་དགེ་བའི་ལས། །ཕ་མ་མཁན་པོ་སློབ་དཔོན་དང་། །མཆེད་དང་ལྕམ་དྲལ་་་རྣམ་ལྔ་ཡིས། །དགེ་བ་མི་དགེ་དགེ་མི་དགེ། །རྒྱས་པའི་ལས་ལ་བསྐུར་བ་ཡིན། །སྡིག་ཅན་དེ་ནི་མི་ཐར་ཞེས། །རྣམ་པར་འབྱེད་པའི་ཐུབ་པས་གསུངས། ་བ། དཀྱིལ་འཁོར་འཇུག་འཛུད་བྱེད་པའི་ཕྱིར། །རི་མོ་འམ་ནི་དེ་བཞིན་བཞན། །ཅི་བགྱི་ཚད་མེད་བཞི་བསྒོམས་ཏེ། །ཤ་དང་ཆང་དང་སྒོག་བཙོང་དང་། །རྐུ་དང་བུད་མེད་སྤྱད་བྱས་ན། །དེ་ནི་ཡན་ལག་བསྐུར་བཏབ་ཡིན། །ལ:། དེ་ལ་་་་་བཤགས་པ་བྱར་ཡོད་ཐབས། །ཅི་བཞིན་དམ་ལ་གནས་བྱས་སམ། །ཐེག་ཆེན་།
