་ལོ། །ཨོཾ་མ་ཧཱ་ཀྲི་ཏ་ཡ་ནུ་ཏན་ཛྷཱ་ན་་་་བཛྲ་སྭ་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་ཀཱ་ཡ་བཛྲ་སྭ་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་བཱ་ག་བཛྲ་སྭ་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། །ཨོཾ་སརྦ་ཏ་་ཐཱ་ག་ཏ་མ་ཧཱ་ཙི་ཏ་བཛྲ་སྭ་བྷ་བ་ཨད་མ་ཀོ་ཧཾ། །སྐུ་གསུང་ཐུགས་དང་ཡེ་ཤེས་ལྔ། །དབང་དང་བྱིན་བརླབས་ཀུན་བསྡུས་ཐིམ། ་། །དཀྱིལ་འཁོར་ཤིན་ཏུ་གསལ་བར་་་་འགྱུར། །ས་ལ་སོགས་པའི་ཕྱག་རྒྱ་བྱས། །རྣམ་གྲོལ་རིམ་པ་ཐོབ་པར་འགྱུར། །དག་པའི་ཡེ་ཤེས་ཀྱིས་སྦྱངས་ན། །སྨོས་ཅི་དགོས་ཏེ་དེ་ཉིད་ཡིན། །ཕྱག་རྒྱུའི་དཀྱིལ་འཁོར་གསལ་སྐྱེད་ལ། །དབང་སྐུར་བྱིན་བརླབས་རྫོགས་བྱས་ལ། །སྟོང་གསུམ་ཡུངས་འབྲུར་འཇུག་ཚུལ་དུ། རྒྱུ་ཀ། དབྱིངས་ནས་དཀྱིལ་འཁོར་སྤྱན་་་་དྲངས་མཆོད། །ཕྱོགས་དུས་དཀྱིལ་འཁོར་བདག་ཉིད་ཆེ། །མི་དམིགས་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ལས། །།དཀྱིལ་འཁོར་ཐམས་ཅད་སྤྱན་དྲང་ངོ༌། །རང་སྣང་་་དབྱེར་མེད་དཀྱིལ་འཁོར་ལ། །།འཇུག་པའི་མཚན་ཉིད་མཉམ་སྦྱོར་བ། །ཁ། ཕྱོགས་དུས་ཀུན་ནས་གཤེགས་པ་ཡི། །བསྙེན་པའི་དཀྱིལ་འཁོར་རབ་ཏུ་རྫོགས། །ཉེ་བར་གྱུར་པའི་དམ་ཚིག་མཆོག། ཨོཾ་ཨེ་ཧེ་ཧི་བྷ་ག་བཱན་མ་ཧཱ་ཀ་རུ་ནི་ཀ་ཏྲི་ཐ་ཧོ༑། ས་མ་ཡ་སྟོ། ཛཧཱུྃ་བཾ་ཧོ། ཨོ་བཛྲ་ས་མ་ཡ་ཨོ། ཨོཾ་བཛྲ་ས་མ་ཡ་ཨོཾ། ཧཱུྃ་བཛྲ་ས་མ་ཡ་ཧཱུྃ། སཱ་བཛྲ་ས་མ་ཡ་སཱ། ཧ་བཛྲ་ས་མ་ཡ་ཧ། ཨོཾ་བཛྲ་ས་མ་ཡ་ཧཱུཾ། ཨོཾ་བཛྲ་ས་མ་ཡ་སྟོ། ཨོཾ་བཛྲ་ས་མ་ཡ་ཧོ། ཛཧཱུྃ་བཾ་ཧོ། །མེ་ཏོག་དོར་ལ་དངོས་གྲུབ་གསོལ། །གཉིས་མེད་ཐིམ་བཞུགས་བརྙེས་པར་བྱ། །འོ། སྒྱུ་་་་འཕྲུལ་དྲ་བས་མངོན་རྫོགས་པའི། །དབང་པོ་ཡུལ་དུ་གང་གྱུར་པ། །པ། མཉམ་་་པའི་ངང་ཉིད་རྟོགས་གསལ་ན། །ཐམས་ཅད་བྱང་ཆུབ་མཆོག་གི་རྒྱ། །སྐྱོད་་་ཆེན་འཁོར་ལོའི་གཞལ་ཡས་སུ། །ཐབས་དང་ཤེས་རབ་ཐམས་ཅད་ཀུན། །།
