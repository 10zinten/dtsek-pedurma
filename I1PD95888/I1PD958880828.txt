རངྒ་ལྕེ་ཏྲ་ཨ་བི་ཤིན་ཙ་ཧཱུྃ། སངས་རྒྱས་རབ་གནས་ལྷ་མོའི་ཚོགས། །རོལ་་པའི་ཚུལ་དུ་གུས་སྟབས་བྱ། ལོ་། ༀ་བཟའ་དང་བཅའ་དང་བཏུང་དང་བགོ། །རང་སེམས་རྣམ་དག་བརྒྱན་པའི་ཕྱིར། །གཉིས་མེད་རོལ་པའི་དབང་བསྐུར་བས། །བདུད་རྩི་ཡིད་བཞིན་སྐུ་གྲུབ་ཤོག། ཨོཾ་བཛྲ་ཏོ་རུ་ཨ་བི་ཤིན་ཙ་ཧཱུཾ། སྦྱོར་བ་ལས་བྱུང་ལེབ་རྒན་ཧཱུཾ། །གང་ཡིན་རྣམས་སུ་འཁོར་ལོའི་ཁང་། །ཉི་ཟླའི་གདན་དུ་སེམས་བྱུང་ཧཱུཾ། །ལྔ་གཉིས་ཟུང་ལས་འཁྲིལ་སྦྱོར་སྐུལ། །ཡེ་ཤེས་སྤྱན་དྲངས་་་ཅུང་ཟད་མཆོད། །ཡོངས་གྱུར་དེའི་ཕྱག་རྒྱ་གསལ། །ཉན་བྱེད་འཁོར་ལོའི་གཞལ་ཡས་སུ། །གསལ་བའི་ཐིག་ལེ་ངོ་བོ་ཉིད། །ཏྭཾ་གི་ཕྱག་རྒྱའི་ཚོགས་མཆོག་ལས། །རིན་ཆེན་དཀྱིལ་འཁོར་གསལ་སྤྲོ་བསྟིམ། །ཏྲི་གསང་བའི་གསང་ཆེན་གསང་་་མཆོག་གང༌། །གསང་བ་ཀུན་ལ་སྙན་པར་བྱ། །གསང་བའི་དོན་ཉིད་བརྟག་པ་ལ། །གཞན་དུ་སྨྲ་བར་བྱ་བ་མིན། །གསལ་བ་འཁོར་ལོའི་གཞལ་ཡས་སུ། །གསལ་བའི་ཐིག་ལེའི་ངོ་བོ་ཉིད། །རིགས་གང་ཡི་གེའི་ཚོགས་མཆོག་ལས། །རིགས་ཀྱི་དཀྱིལ་འཁོར་གསལ་སྤྲོ་བསྟིམ། །ཨོཾ་དུས་གསུམ་རྒྱལ་བའི་སྲས་ཆེན་པོ། །སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་ཟུང༌། །རྒྱལ་བ་ཐམས་ཅད་མཉེས་མཆོད་ལ། །རྒྱལ་བ་ཀུན་དང་མཉམ་པར་སྦྱོར། །ལག་པ་འཁོར་ལོའི་གཞལ་ཡས་སུ། །གསལ་་་བའི་ཐིག་ལེའི་ངོ་བོ་ཉིད། ཁ། ཧཱུྃ་གི་ཕྱག་རྒྱུའི་ཚོགས་མཆོག་ལས། །ལས་ཀྱི་་དཀྱིལ་འཁོར་གསལ་སྤྲོ་བསྟིམ། །ཨོ་ཁྱེད་ནི་རྡོ་རྗེ་ལས་ཡིན་གྱིས། །ལས་རྣམས་མ་་་ལུས་ཀུན་བྱོན་ཞིག། ཕོ་ཉའི་འདུ་འཕྲོ་མང་པོ་ཀུན། །ཁྱོད་ཀྱི་དགོས་པ་བྱེད་པ་ཡིན། །སྨྲ་བྱེད་འཁོར་ལོའི་གཞལ་ཡས་སུ། །གསལ་བའི་ཐིག་ལེའི་ངོ་བོ་ཉིད། །ཧྲིའི་ཕྱག་རྒྱའི་ཚོགས་མཆོག་ལས། །ཆོས་ཀྱི་འཁོར་ལོ་གསལ་སྤྲོ་སྟིམ། །ཨོ་ཆོས་ཀྱི་རྒྱལ་པོ་བླ་མེད་ཀྱིས། ང་། དད་ལྡན་སྐྱེས་བུ་ངོམས་པར་བྱོས། །རབ་འབྲིང་ཐ་།
