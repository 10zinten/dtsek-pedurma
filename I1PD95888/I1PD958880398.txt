བསྒོམ་ཞེས་གསུངས་ཏེ། ཨེ་། ལྷ་སོ་སོ་ཐོག་མ་ནས་བསྒོམ་པའམ། ལ་འོ་སྒྱུར་ཏེ་་་་བསྒོམ་པའི་ཕོ་མོ་བཅུ་དབུས་སུ་རྡོ་རྗེའི་རིགས་ཡབ་ཡུམ་བཅུ་རུ་དགོད་དོ། ཎ། དེ་ལྟར་ཕྱོགས་བཞིར་རིགས་བཞི་ཡི་ཡབ་ཡུམ་བཞི་བཅུའོ། །དེའི་ཕྱི་རིམ་དུ་ཁྲོ་བོ་ཡབ་ཡུམ་ཉི་ཤུའོ། །ང་ཚོ། །་ཁ། ད་ཡོད་ས། ཤ་ཁྲོ་དང་ག་བ་ཏི་བ། དབུ་མ་ལ་ཁཚོ་༼ཚོམ་བུ་གསུམ་པ་བསྟན་པའི་ཕྱིར། དེ་བཞིན་རིགས་དང་རྡོ་རྗེ་རིགས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། ཁྲོ་བོ་ཡབ་ཡུམ་བཅུ་དྲུག་གོ། ག། འོད་དབང་འདུས་པ་ཚོམ་བུ་གཅིག་པ་བསྟན་པའི་ཕྱིར། རིགས་ཀྱི་རིགས་ཏེ་རིགས་ཀྱི་གཙོ། །ཐུགས་ཀྱི་ཐུགས་ཏེ་ཐུགས་ཀྱི་གཙོ། །།ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། རིགས་ལྔ་ནི་རྡོ་རྗེའི་རིགས་སུ་འདུས་པས། རྡོ་རྗེའི་རིགས་ནི་དཀྱིལ་འཁོར་གྱི་སྐབས་ཀྱིས་རིགས་ཀྱི་གཙོ་བོ་ཁོ་ནའོ། །ཨ། གཅིག་དང་ཐ་དད་ལས་གྲོལ་བས་རིགས་གྱུར་ཡང་། །ཡེ་ཤེས་མི་འགྱུར་བའི་ཕྱིར། ཐུགས་ཐམས་ཅད་ཡེ་ཤེས་ལྔར་འདུས་ལ། །ལྔ་ཡང་སྟོང་པ་ཉིད་ཁོ་ནར་འདུས་པས་དེ་ནི་ཐམས་ཅད་ཀྱི་མཆོག་གོ། །གཞན་ནི་གོ་སླའོ། ལ་ཁྲོ་བོ་ཁྲོ་མོ་བརྒྱད་དོ། །།ཁྲོ་བོ་ཁྲོ་མོ་ཚོགས་ཆེན་པོ། །ཕོ་ཉ་ཕྱག་བརྙན་བཀའ་ཉན་ཚོགས། ་འ། ཅི་བགྱི་ལ་སོགས་ཚོགས་རྣམས་ཀྱིས། ་ཨོ། །དངོས་གྲུབ་ལས་རྣམས་རྫོགས་པར་བྱེད། །ཅེས་བྱ་བ་ནི། ཁྲོ་བོ་ཁྲོ་མོ་ཚོགས་ཀྱི་དཀྱིལ་འཁོར་བསྒྲུབ་ན། འོ་ཁྲོ་བོ་ཡབ་ཡུམ་བཅུ་དབུས་སུ་དགོད། ཌ་དེ་བཞིན་དུ་རིམ་པས་གནས་ཡུལ་བསྒོ་མ་ཉི་ཤུ་དང་། ཅི་བགྱི་ཞེས་ལས་གྲུབ་པའི་དབང་ཕྱུག་མ་ཉི་ཤུ་རྩ་བརྒྱད་དང༌། བསྡུ་བའི་ཁྲོ་བོ་ཁྲོ་མོ་ཉི་ཤུ་སྟེ། །ཕོ་མོ་བདུན་ཅུ་རྩ་་་བརྒྱད་ཀྱིས་བསྒྲུབས་ན་ལས་དང་དངོས་གྲུབ་གཉི་ག་ཡང་འགྲུབ་བོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ལྟ་ཏེ་ཁ། །ཡp:ཀ་ལའོ། །ཞེས་སོགས་ནས་བྱ་ཚང་ཀླ་ཀ་ཀ་ཎ་། ?ཀྱི་རྣལ་འབྱོར་སྔགས་འཆང་དངོས་གྲུབ་ནི། །།ཞེས་བྱ་བ་ལ་སོགས་པ་ནི། _།
