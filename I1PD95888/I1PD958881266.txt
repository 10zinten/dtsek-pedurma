སྒྱུ་འཕྲུལ་དཔེ་ཆུང་རང་གནས་ཀྱི་རིམ་པ། །རང་གནས་ཀྱི་རིམ་པའོ། །སྒེག་པ་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ། །།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»ཚིགས། «པེ་རྒྱུད། «པེ་»བཞི། །«པེ་རྐྱལ། ཉི«པེ་སྒོམ། །དྲན་པ་དང་ད་བར་སློབ་«པེ་»དཀར། །«པེ་ཚོ། བག་་་་བས་གཡོ་བཚེ«པེ་རྒྱུད། «པེ་—ཏེ། ད་་་་་་་་་་་་་་་་་་་་«པེ་»ལས། «པེ་དབྱིབས། ཚེ«པེ་»ལ། །དེ་ལ་འོ། །།ད། ਨਰਸਰ/g/APਪੂਰਲਜੋਹੌਓਲਾཟླངཁ་ཏར། ་བསྡུར་འབྲས་རེའུ་མིག་ན། །བདག་རྒྱལ་བ་འོད་ལྔ་། ཡང་། རིན་འ། ཀུན་གར་ཀྱི་ལ་ཚེ་དཔེ་གཞི། སྡེ་དགེ། ད་༌༌པེ་ཅིན། ་༌་སྣར་ཐང་། །ཅོ་ནེ། །།།སྡེ་ཚན། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། པ། ལྡེབ། ་ཀྱན་ན་ནབ། རྒྱུད་འགྲེལ། ཕྲེང་། ，ཚིག་རྐང་། ཚེག་ཁྱིམ། S། །བཟློགབསཔ:བ་ན་ཏེ། ན་ཏ། ཟླ་ང་འོ་མ། །ཀན་ཀཀ་བ་ཀ་བ་ལ། །ཟླ་བས་ཁ་ལ་བབ་འa::། ་་་་་གནས་དང་གདུག་།
