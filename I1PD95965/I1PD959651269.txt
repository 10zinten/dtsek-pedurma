།::ཀ་བ་བསྟན་འགྱུར། སྣ་ཚོགས། ༼བོའི་ལ་བསློག་མཁོ། ཨམ་བྱེད་པར་བའི་མིང་། འདི་ནི་ས་ཧཱ་ཀ་ར་ལྷན་ཅིག་བྱེད་དང༌། །ཤིན་ཏུ་བཅུད་བཟང་ཞེས་པའོ། །ཨ་མྲའི་འབྲས་དང་། འབྲས་བུ་མངར། །ཀི་ཀ་ཤི་དང༌། ཀ་བ་ཀཱི༑། ས་ན་སེ་དང་། ཚེར་ཅན་འབྲས། །ཙནྡྲ་ཀཱ་ལུ། གཙུག་ཕུད་ལྡན། །མུ་ར་ཛ་འབྲས་ཞེས་པའོ། །དེ་ཡི་སྦུབས་ལ་ས་ན་སའི། །།ཏཱ་ལི་ཀཱ་ཞེས་བྱ་བའོ། །བ་ཀུ་ལ་དང་། ཤྲཱི་དྷུ་དང་། །དྲི་ལྡན་གེ་སར་ཅན་ཞེས་པའོ། །ཁའི་མྱ་ངན་མེད་ཀྱི་མིང་། མྱ་ངན་མེད་དང༌། པཉྩ་ལ། །ཆུ་ལ་གར་རྩེ། །རྐང་་་མཛེས་དང་། །དགའ་བ་སྦྱིན་ཞེས་བྱ་བའོ། །དང་། སྐྱ་ནར་གྱི་མིང་། སྐྱ་ནར་ཅན་ཤིང་དོན་ཡོད་དང༌། །།ཆུ་ལྡན་གནས་པ་པཱ་ཊ་ལ། །མེ་ཏོག་རྩ་ནག་ལུས་ངན་མིག། ཀཱ་ཙི་ཐང་གི་ཀོ། སྦྲུ་ལཱི། འབྲས་བུ་སྐྱེས། །།འདོད་པའི་ཕོ་ཉ་མཛོད་ཐ་མ། །ཟངས་ཀྱི་མེ་ཏོག་ཅན་མའོ། །།ཆེནgeནཱ་ག་གེ་སར་གྱི་མིང་། ཙམ་པ་སྐྱེས་དང་གེ་སར་ཅན། །ནཱ་ག་གེ་སར་རྐང་དྲུག་དགའ། །ནཱ་ག་པུཥྤ་ཞེས་ཀྱང་བྱ། །འདི་ནི་གསེར་གྱི་མེ་ཏོག་ཅན། །པུཏྤ་་རོ་ན། གསེར་འདྲ་དང་། ཀླུ་ཡི་མིང་ཅན་ཞེས་པའོ། །ན་བ་ཀཱ་ཏེ་ན་དུ་སྲུང་པའི་མིང་། ཧ་རི་དྲ་བ་འདིའི་ཕྱེ་མ། །དེ་བཞིན་ཀ་དི་ཀུ་སུམྦྷ་ལེ་བརྒན། །ང་། ཙམ་པ་ཀའི་མིང་། ཙམ་པ་ཀ་དང༌་། །ཝ་བསོད་ནམས་བསུང་། །མེ་ཏོག་་་་རྒྱལ་དང་ཙམ་པ་སྐྱེས། །མཆོག་ཐོབ་གསེར་གྱི་མེ་ཏོག་ཅན། །འདི་ཡི་འབྲས་བུ་ཀ་ལི་ཀཱ། །དྲི་ཡི་འབྲས་བུ་ཞེས་པར་འགྱུར། །གལ་ལ་སེ་འབྲུའི་མིང་། སེ་འབྲུ་ཤིང་དང་། མེ་ཏོག་གོ། །དྲི་མྲ་དང་སེ་འབྲུ། ཀ་ར་ཀཾ། །འབྲས་བུ་བྱ་བ། །སདྡྷ་རྨུ། །ནེ་ཙོའི་སྨྱུག་མ་ཚིགས་ཆེན་སྐྱེས། །།རབ་སྐྱེས་དང་ནི་གོང་བུ་ཅན། །འོ་མ་ཅན་དང་རྒྱལ་པོའི་ཟས། །ལྟ། །འབྲས་་་་བུའི་བདག་པོ་ཞེས་བྱ་བའོ། །ཀླུ་ཤིང་སྐྱེས་བུ་མཐོ་བ་དང་། །གེ་སར་ཅན་དང་། །།
