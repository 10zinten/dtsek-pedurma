སྣ་ཚོགས། ༼ནིའི་། ཡ་པོ་དཱིགྷ་བ་ཏི། ཛ་ལ་ཏུམྤི་ཀཱ་ཏྱཱ་ཡེ་ན་རེ་ཕ་གྱོནྡྷ་ག་མ་ནཾ། གོརླུ་གྱི་ཡ་ནི་གཉིས་སུ་འགྱུར་རོ། །ཆུ་ལ་ཀུ་བ་གཞག་པའི་རིག་པས་རེ་ཕ་སྟེང་དུ་འགྲོ་བ་ཡིན་ནོ། །དཀར་མོ་ཏྲ༑སྭ་ར་ཨིཏྱ་ནུ་བཏྟ་ཏེ། །ཨེ་བ་མ་ནྱ་ཏཱ་པི་ཡ་ཏྲ་ན་སཱུ་ཏྲ་ཀླུ་རི། འདིར། དབྱངས་འདི་རྗེས་སུ་དྲངས་པའོ། །འདི་ལྟར་གཞན་དུ་ཡང་གང་དུ་མདོའི་ཡི་གེ་རྣམས་ཀྱིས་བྱ་ཀཱདྨ་སིདྷི་སྔ་ཙིཏྲ་སརྦ་ཏྲ་སཱུ་ཏཱཾ་ཏ་རཱཏྠ་དཾ་ཏ་རཱ་ད་ནུ་བྲྀཏྟི་ཛཱི་ཏ་བུ། གནྡྲ་བྷཱུ་ཡལྟ་བྱེད་སྒྲུབ་པ་དེ་འདི་ཐམས་ཅད་དུ་མདོ་གཞན་དང་ཚིག་གཞན་རྗེས་སུ་དྲངས་ཏེ་ཤེས་པར་བྱའོ། །གཞུང་ཕ་བྷ་ཡཱནྟཱ་སྣ་བྷི་རྨི་ཁྱ་ཏེ། ན་ཧི་ཨ་ཏྲ། ན་ཧུ་ཏྲ། ཨུ་པཾ། ཨུ་བནྡྷོ་མངས་པའི་འཇིགས་པས་བདག་གིས་མ་བྲིས་སོ། །མེད་འདིར། འདིར་མེད། ཨུ་ཝའོ། །ཨུའི་ཡི་གེ་ལྔ་བ་ཏཱ་མཱ་པ་བྱ་ཏེ་སྤ་ར་པ་རེ། མ་དྷ་ཨ་ཏྲཿམདྷཱ་ཏྲཿརཾ་རཾ། རཾ་བནྡྷོ་ཉིད་དུ་འཐོབ་སྟེ་དབྱངས་ཕ་རོལ་ལའོ། །།སྦྲང་རྩི་འདིར། ་་་རི་རིའོ། ་། རིའི་ཡི་གེ་ར་ཏཱ་མཱ་བང་ཏེ་སྤ་རེ་པ་རེ། པི་ཏྲ་ཨརྦ། པིཏྠ་དྲྭ། ལོ་ལཾ། ལོ་པཎྞོ་ལ་དྲ་ཉིད་དུ་འགྲོ་སྟེ་དབྱངས་ཕ་རོལ་ལོ། །།ཁོ་། ཕའི་དོན། ཡི་དམ་ད་ལི་ཝའོ། །ལོ་ཡི་ཡི་གེ་ལ་ཏཱ་མཱ་པ་བྱ་ཏེ་སཱ་རེ་པ་རེ། ་ལཾ་ཨ་ནུ་བནྡྷལ་ནུ་བཨེ་ཨ་ཡི། ཉིད་དུ་འོངས་ཏེ་དབྱངས་ཕ་རོལ་ལོ། །ལི་ཡིག་རྗེས་འབྲེལ། ལ་ཡིག་འབྲི་བ་པོ། ཨེ་ཨ་ཕའོ། །ཨེ་ཀཱ་རོ་ཨ་ཕྲ་བྷ་བ་ཏི་སཱ་རེ་པ་རེ། ནེ་ཨ་ནཾ། ན་ཡ་ནཾ། ཨོ་ཨ་བྲ། ཨོ་ཀཱ་རོ་ཨ་ཨེ་ཡིག་ཉིད་ཨ་ཡ་རུ་འགྱུར་ཏེ་དབྱངས་ཕ་རོལ་ལོ། །འདྲེན་བྱེད་མིག། ཨོཾ་ཨ་བའོ། །ཨོ་ཡིག་ཉིད་ཨ་་བ་བྷ་བ་ཏི་སཱ་རེ་པ་རེ། ཨོ་ཨ་ཏི། བྷ་བ་ཏི། ག་བ་དེ་རྔ་བ་ཌཱ་ག་མོཉཀྵ་་པཝ་རུ་འགྱུར་ཏེ་དབྱངས་ཕ་རོལ་ལའོ། །གྱུར་ཅིག། ག་བ་ལ་སོགས་པ་ལ་ཨ་ཡིག་ཨཱ་ག་མ་དཀླུ་ལ་སོགས་་དོ་བཀྟ་བྱགོ་ཨཀྵག་བཱ་ཀྵ། གོ་ཨིནྡྲག་བེནྡགོ་ཨ་ཛི་ནཾ། པའི་དང་པོར་བརྗོད་པར་བྱའོ། །སྒེའུ་ཁུང་། ::ཁྱུ་བ། །བ་ལང་པགས་པའམ་་་།
