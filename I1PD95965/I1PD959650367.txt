སྣ་ཚོགས། །ནིདྭ་མ་ཨི་ཏི་ནི་ཥ་དྷཱ་ད་ལྗོ་བོ་ན་ཞེས་པས་དབྱེ་བར་མི་བྱའོ། །།སུ་བི་བ་ལི་ཏ་རྭ་དི་ཏི་ན་ཡིག་དབྱིའོ། །སུ་པི་ན་ལོ་པ་སྣ་སི་དྷ་ཛཱིདཱ་ཏཱ་མི་ཏཱཾ་ཙ་མ་ཡིན་ནོ། །ཨ་སཾ་བུ་དྷཱ་བི་ཏི་ནི་ཡ་མཱ་ཏ། སཾ་བོ་དྷ་ནེ་ནི། དྲི་རྒྱན་སཾ་བུ་དྷ་བ་ཞེས་པས་ན་དབྱི་བ་བཀག་:གོ། །བྲགྷཱ། བྲཧྨཱ་ཎོ། བྲགྷཱ་ཎཿབྲངྒཱཎཾ། །བྲཧྨཱ་ཎོ། བྲགྷཱ་ཎཿབྲགྷཱ་ཎཱ། །བྲ་དྷཱ་ཧཱུཾ། བྲཧྨ་བྷི། བྲཧྨཱ་ཎེ། བྲཧྨ་དྡྷཾ། བྲཧྨ་བྷུབྲཧྨ་ཎ་གྷབྲཧྨ་དྡྷཾ། བྲཧྨ་བྲཧྨ་ཎབྲཧྨ་ཎོབྲཧྨ་ཎཾ། །བྲཧྨ་ཎི། བྲཧྨ་ཎོབྲཧྨ་སུ། ཧེ་བྲཧྨ་ན། ཧེ་བྲཧྨཱ་ཎོ། ཧེ་བྲཧྨ་:ནོ། རཱ་ཛ་ནའི་སྒྲ་ལ། ཙུ་ཊུ་ཏུ་ལ་ཤ་བྱ་བཱ་ཡ་ཞེས་པས་ཎ་ཉིད་དུ་བཀག་གོ། །ཤ་བཱ་དཱ་བ་ཛ་དོ་ཏུ་ཨ་དབྱིའོ་ཞེས་པས་ཨ་ཡིག་དབྱིའོ། །ཏོ་ཤྩ་ཎྜ་བྷུ་ནཱ་བ་ཞེས་པས་ན་ཡིག་་་ལས་ཉ་ཡིག་གོ། །བདུན་པའི་གཅིག་ཚིག་ལ་ཛི་མྱོ་རྦ་ཞེས་པས་ཕྱོགས་གཅིག་གི་ཨ་དབྱིའོ། །ཨ་སཾ་བུ་དྷཱ་བ་ཞེས་པས་བོད་པ་ལ་རིང་པོ་མེད་དོ། །ན་སཾ་བུ་དྷ་བ་ཞེས་པས་ན་དབྱི་བ་བཀག་གོ་། :། རཱ་ཛཱ། རཱ་ཛཱ་ནོ། xརཱ་xཛཱ་ན། རྭ་ཛཱ་ནཾ། རཱ་:ཛཱ་ནོ། རཱ་ཛྷཱརཱ་ཛྷཱ་རཱ་ཛཱ་ཧཱུཾ། རཱ་ཛཱ་བྷི་རཱ་ཛེ། རཱ་ཛ་ཧཱུཾ། རྭ་ཛ་བྷུ། རཱ་ཛ་རཱ་ཛ་ཧཱུཾ། རཱ་ཛཱརཱ་དྷོ། རཱ་ཛཱི། རཱ་ཛཱི། རཱ་ཛ་ནི། རཱ་དྷོ། རཱ་ཛཱ་སུ། ཧེ་རཱ་ཛཱ་ན། ཏxཧེ་ཏུ་ཛཱ་ཎོ། ཧེ་རཱ་ཛཱ་ན། ཡུ་བ་ནའི་སྒྲ་ལས། རཱ་ཛ་ནའི་སྒྲ་ལྟར་རོ། །ཤ་སཱ་ཏུ་ཛཱ་དོ་ཏུ་ཉྩ་ཡུ་བ་ན་མ་བྱོ་ནཱ་བ་ཞེས་པས་ཨ་ཡིག་བཟུང་ནས་བ་ཡིག་ལ་ཨུའོ། །ཡ་ཎི་ཀ་ཞེས་པས་ཕ་རོལ་སྔོན་དུའོ། །ཨུ་ཡིག་དང་ཨུ་ཡིག་དག་ལ་བས་ཨ་ཀོ། ཀི་རིང་པོའོ་:༑། ཡུ་བཱ།  ོ་ཡུ་བཱ་ནོ། ཡུ་བཱ་ན་ཡུ་བཱ་ནཾ། ཡུ་བཱ་ནོ། ཡུ་ན། ཡུ་ནཱ། ཡུ་བཱ་ཧཱུཾ། ཡུ་བཱ་བྷི། །ཡུ་ནེ། ཡུ་བ་བྷཱལཾ། ཡུ་བ་བྱཱཡུ་ནཡུ་བཱ་ཧཱུཾ། ཡུ་བཱ་ཡུ་ན་ཡུ་ནོ། །ཡུ་ནཾ། ཡུ་ནེ། ཡུ་ནོ༔ཡུ་བ་—AUVO།
