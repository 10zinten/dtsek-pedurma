བསྟན་འགྱུར། །སྣ་ཚོགས། ༼ཤཱ་བ་ཡ་ཏི། ཀཱ་རི་ཏ་ཉིད་དོ་ཞེས་པའི་ཚིག་ལས་འདིར་མི་འགྱུར་རོ། །ཐེཊྛཐེ་མཱ༑། བྲེ་ཡཱན་ཞེས་པའོ་ཞེས་པ་མཁན་པོ་དགའ་བའི་གྲགས་པས་མཛད་པའི་ཚིག་གསལ་རྫོགས་སོ། །།།ངེ་བ། །གལ་ཏ་། ་་་་་་་་་ད། རང་འདི་ནི་:དཔྱལ་ལོ་ཙཱ་བ་དཔལ་ཆོས་ཀྱི་བཟང་པོའི་ཕྱག་དཔེའི་རྒྱ་དཔེ་ལས། ཤཱཀྱའི་དགེ་སློང་དྷརྨ་ཤྲཱི་བྷ་དྲས་མཁན་ཆེན་:ལོ་ཙཱ་བ་ཆེན་པོ་བུ་སྟོན་ཞབས་ཀྱི་སབཀའ་དྲིན་ལས་བོད་ཀྱི་སྐད་དུ་བསྒྱུར་བའི་གནས་ནི། །པཎྜི་ཏ་དཱི་པཾ་ཀཱ་ར་ཤྲཱི་ཛྙཱ་ནཱའི་བྱིན་གྱིས་བརླབས་པའི་ཆོས་གནས་ཆེན་པོ་རི་ཕུག་གོ། །ན་ཡི། །འབྱུང་བད་ག་མ་ལོ་མེད་ན་ཁྱོ་ཤ་ཚ་བ་དང་དགའ་བ་འདུག་པ་དེར་ད་ལྟ་འང་ཤོ་་་་་་བོད་པ་དེ་དག་ལ་གད་བདག་པོ། བ་མི་འདའ་བ་འ་འདས་ਸਸਸਸਸਉਸਤgਨਕਸ਼ਗਵਾਰਮਨਬਉਸ(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ཀྱི། འི«པེ་»《སྣར་»ངུའི། «པེ་»«སྣར་»ཀ་ཏིའི། ས«པེ་»«སྣར་བསྟེན། །:ཀྱི«པེ་»«སྣར་བརྟ། («པེ་»《སྣར་»ཏ། ་::་ན་མ་«པེ་»«སྣར་»བུ། «པེ་»ཏསྦ། «པེ་»«སྣར་ཨ། «པེ་»སྱ་དྷེ། «སྣར་སྱ་དྷི། «པེ་ཧེ། སྟུ་མ་ཏེ། «སྣར་ཧི་སྭཱ་མ་ཧི། པ«པེ་»«སྣར་»བརྟ། «པེ་»«སྣར་དི། («པེ་»དི། «པེ་《སྣར་»ཤཱསྟ། སུ«པེ་»«སྣར་ན། «པེ་»«སྣར་»རསྨི། (ཆོས་ཚན་འདིའི་ནང་དུ་རསྨིའི་ཞེས་པ་རྣམས་«པེ་»《སྣར་»ལ་རསྨི་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་་་བཀོད་མེད། «པེ་»«སྣར་ཨནྟེ། 《སྣར་»དབུ། «པེ་»«སྣར་»དེ། («པེ་ལྷམ། འད། ལ་ཁྲུ་«པེ་《སྣར་སྦྱོར། «པེ་པོ། «པེ་»«སྣར་—པ། «པེ་ཨན། «པེ་»འོ། («པེ་»གོ། <«པེ་《སྣར་བཅད་བྱ་རུ། «པེ་»«སྣར་གི། ®«པེ་»«སྣར་ཡ་ཐ། སུ«སྣར་»::བྲང་། «པེ་»གྱི། །ི«པེ་»«སྣར་»ཅེས། (ི«པེ་»«སྣར་གེའི། «པེ་»«སྣར་»དནྟའི། «པེ་»«སྣར་སཱི། «པེ་སྲི། །པོ«པེ་»«སྣར་»ཥའི། །«པེ་»《སྣར་ཨp། «པེ་+གོ། །《སྣར་ཐིག«པེ་»«སྣར་»ཨུད། །
