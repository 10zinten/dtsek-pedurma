བསྟན་འགྱུར། སྣ་ཚོགས། །ནོ༎ན་ཡིག་དབྱིའོ། །གཱ་དའི་ཨཱཏ་ཨོ་ཉིད་དོ། །བི་བྷུ་ཏིའི་ཡོན་ཏན་ནོ། །ཤུ་དྲ…དམངས་རིགས། རིགས་བཞི་པའོ། །།ཨུ་གྲ…རྩེ་མོ། བཟོད་དཀའོ། །བདེ་བྱེད་ལ་ཡང་ངོ་། །བཛྲ་རྡོ་རྗེའོ། །།བི་པྲ་བྲམ་ཟེའོ། ་། བྷ་དྲི་བཟང་པོ་དགེ་ལེགས་སོ། །གོ་རཿདཀར་པོའོ། །བུད་མེད་ལ། ན་ད་སོགས་ཉིད་ལས་ཨི་འོ། །གོ་རཱི་དཀར་མོའོ། །བྷ་ཡ་ཏི་ལའང་སྔ་མ་བཞིན་ཨཱིའི་རྐྱེན་ནོ། །བྷེ་རཱི་རྔ་བོ་ཆེའོ། །བུད་མེད་ལ་ཨ་ལས་ཨཱའོ། །ཨི་རཱ་སའོ། །ཀ་པི་ལི་ཀ་སོགས་ཉིད་ཀྱི་ཕྱིར་ལ་ཉིད་དོ། །ཨི་ཨཱ་ས་གཞི་དེ་ཉིད་དོ། །རྗེ་། བྲང་ཏེ་བྱ་བ་མང་བྲབྷི་ཀྲྀ་ཎི་དག་ལས་ཀི་ཀའོ། །བྲ་ལྕེ་གཅོད་པ་ལའོ། །ཀྲྀཥ་རྣམ་པར་རྐོ་བ་ལའོ། །འདི་དག་ལས་ཀི་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ནི་ཡིན་བཞིན་དུ་བྱ་བའི་དོན་ནོ། །དེས་ན་ཀྲ་ཧི་ཛུ་ཞེས་པ་ལ་སོགས་པ་བྲཤྩའི་སམྤ་སཱ་ར་ཎའོ། །དབྱངས་དང་བཅས་པའི་རེ་ཕའི་ར་ཡིག་གོ། །ཀཎྜ་ཏིའི་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །བྲཱཉྩི་ཀཿསྡིག་པའོ། །།ཀྲྀ་ཎི་ཀམངལ་གྱི་བྱིས་པའོ་། :། ིམཱུ་ཎི་ཀ་སཱི་མ་ཀ་དག་གོ། །འདི་དག་ལ་ཀི་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཚིག་ཕྲད་དུ་བྱའོ། །མུ་་ཥ་རྐུ་བ་ལའོ། །སྱ་མུ་སཱ་ན་དྷཱ་ན་སྒྲ་ལའོ། །མུ་ཏིའི་རིང་པོ་ཉིད་དོ། །སྱ་མའི་སམྲ་སཱ་ར་ཎ་དང་རིང་པོ་ཉིད་འང་ངོ༌། །མཱ་ཏི་ཀབྱི་བའོ། ་། པི་མི་ཀཤིང་གི་ཁྱད་པར་རོ:༌། །ཀཱི་ལས་ཨི་ཀའོ། །རྔུ་ཀྲྀ་ཉ་རྫས་རྗེ་བ་ལའོ། །འདི་དག་ལས་ཨི་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཡོན་ཏན་ནོ། །ཨེ་ཨ་ཡའོ། འ། ཀྲ་ཡི་ཀཉོས་པའོ། :། །ྋཤཱ་ཧྲ་ཉ་ཨ་བི་དྲུ་ད་ཀི་རྣམས་ལས་ཨི་ནའོ། །གཱ་ད་ཤིང་འགྲོ་བ་ལའོ༑། མཚམས་སྦྱོར་གྱི་ཡི་གེའི་མཐའ་རྣམས་ཀྱི་བི་ཀ་ར་ཎ་མ་ཡིན་པ་ལ་ཨཱ་་ཡིག་གོ། །དྲྀཉ་འཕྲོག་པ་ལའོ། །ཨ་པ་རཀྟ་སྐྱོང་བ་ལའོ། །དྲུ་དྲུ་འགྲོ་བ་ལའོ། །དཀླུ་འཕེལ་བ་ལའོ། །འདི་རྣམས་ལས་ཨི་ན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨའི་ཡི་གེ་།
