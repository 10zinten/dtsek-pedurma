ཁྱིམ་གྱི་ས་དང་གཞི། །འཇུག་དང་། འཇུག་བྱ། །སྐྱེ་བོའི་གནས། །རབ་བསྔགས་གནས་དང་ས་གཞི་མ། །རབ་ཏུ་དགའ་དང་། དགའ་བྱེད་དོ། །གྲི་ཧ། ཀྲྀ་ཧཾ། བེ་ཤ་དང་། །བེཟླ་དང་ནི་གནས་པ་ཡིན། །གོ་ཧཾ་ཀུན་་དགའི་གནས་དང་ནི། །གནས་ཁང་། ཕིབས་འོག་འདུན་གནས་དང་། ཚ་། མཚན་མོའི་ནང་གནས་སྒྲིབ་གཡོགས་དང༌། །ཁྱིམ་དང་བྱ་བའི་གནས་ཀྱང་ངོ༌། །ངེས་གནས། །གནས་དང་། ཀུན་ཏུ་གནས། །སྐྱབས་་འོག་དང་ནི། །ཁང་ཁྱིམ་དང་། །ཁང་ཆུང༌། །སྤྱིལ་པོ། འདུན་ས་དང་། །ལཾ་ལཱ། ཀ་རྒྱུ་་འཇུག་གནས་སོ། །ར་བ་བཞི་པ། ཀུན་འཇུག་གོ། །ཀུན་གནས་ཁྱིམ་་་དང༌། བཤེས་གཉེན་ཅན། །ཁང་བཟངས་ར་བས་ཡོངས་བསྐོར་དང་། །ཟླ་བའི་ར་བ། སྤྱི་བོའི་ཁྱིམ། །ཁང་བརྩེགས་དང་ནི། བ་ཌ་བྷི། །ད་དྷི་ཊཱ་དང་། ངེས་བྱས་ཁྱིམ། །།ཐུབ་རྣམས་ལོ་མའི་སྤྱིལ་པོ་དང་། །།མཆོད་སྤྱིལ་པོ་ཞེས་བྱ་བའོ། །བཟོ་བོ་དང་ནི་ཀུན་འཇུག་གནས། །རྟ་ཡི་ར་བ། རྟ་ཡི་ཁང༌། །འདབ་ཆགས་ར་བ། བྱ་ལྡན་ནོ། །ཆུ་ཁང་བཏུང་བའི་ར་་བའོ༑ད། རྒྱ་མཚོའི་ཁྱིམ་དང་རླུང་ཡོང་ཅན། །།ཆུ་ཡི་འཁྲུལ་འཁོར་གནས་ཡིན་ནོ་་བ་། བ་སི་ཏ་པ་ཡི་ལས་དང༌། །ཨ་བ་ས་ནི། །སྦྲ་ཡི་ཁྱིམ། །ཏིལ་མར་ར་བ་འཁྲུལ་འཁོར་ཁྱིམ། །དུད་ཁང་ཨ་ལནྡྷ་ཀཱའོ། །བཙོན་ར་དང་ནི་འཆིང་བའི་ཁྱིམ༑ལོ་། གསོད་པའི་མྱུ་གུ་བྱ་བའི་སྦྱོར། །དཀའ་ཚེགས་གནས་དང་བྱེད་པ་ཅན། ་་། ཚང་མའི་ཁྱིམ་དང་གཉྫའོ། །གྲྭ་པའི་ཁང་པ་ལ་སོགས་ལ། །གནྡྲི་དང་ནི། ་མ་ཋའོ། །སོ་སོའི་གནས་ནི་དེའི་ར་བ། །ཐུབ་པ་རྣམས་ཀྱི་གནས་དང་རྟེན། །གྲགས་པའི་ཕུང་པོ་ལྷ་ཡི་གནས། །མཆོད་རྟེན་དང་ནི། །མཆོད་སྦྱིན་གནས། །སྙིང་པོའི་ཁང་པ། ཀུན་སྤྱོད་དོ། །འདུན་ཁང་དང་ནི་སྐྱེ་།
