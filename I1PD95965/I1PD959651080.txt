རྣམས་ཀྱི་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཛཱི་ཡ་ཏེ། བྱ་དྷ། །བི་དྷ་ཏེ། སཾ་པྲ་སཱ་ར་ཎ་ལ་མ་ཡིན་ནོ་ཞེས་པའི་དགག་པ་ལས། སཾ་པྲ་སཱ་ར་ཎ་བྱས་པ་ལ་སླར་ཡང་སཾ་པྲ་སཱ་ར་ཎ་རུ་མི་འགྱུར་རོ། ་། བ་ཤ། ཨུ་བྱ་ཏེ། བྱ་ཙ་མི་ཙྩ་ཏེ། པྲཙྪ། ཕྲིཙྪ་ཏེ།  ོ་བཤུ། ཕྲིཉྩ་ཏི། སྔ། ཤྲཱི་ཛྙཱ་ཏེ། ཉི་ཥ་པ། །སྭ་བི་བ་ཙི་ཡ་ཛ་སོགས་་རྣམས་ཀྱི་ཡ་ཎ་པ་རཀོ། ་ཨཱ་ཤི་ཥ་རྣམས་ལའོ་ཞེས་པས་སཾ་པྲ་སཱ་ར་་་ཎའོ༑། སུ་པྱ་ཏེ། བྲུཉ། བྲུའི་བ་ཙིའོ་ཞེས་པས་བ་ཙི་རུ་བསྒྱུར་རོ། །བ་ཙ་འཆད་པ་ལའོ། །ཨུ་ཙ་ཏེ། ་ཡ་ཛབ་བ་བ་ཧ་རྣམས་ཉིད་དང་། བེཉ་བྱེཉ་དག་དང་ཧྭ་ཨ་ཏི་དེ་བཞིན། བ་ད། ་བ་ས། །དག་དང་ཤཱ་ཡ་ཏི། ཡ་ཛ་ལ་སོགས་དགུ་པོ་རྣམས་ཉིད་དོ། །ཡཛ། ཨི་ཛེ་་ཏེ། ་བ་ས། །ཌ་ཨུ་པྱ་ཏེ༑ལོ་བ་ཧ། ཨུ་རྒྱ་ཏེ། བེཉ། མཚམས་སྦྱོར་གྱི་ཡི་གེའི་མཐའ་ཉིད་ཀྱི་ཕྱིར་ཨཱ་ཉིད་དོ། །སཾ་པྲ་སཱ་ར་ཎའོ། །ནཱ་མིན་མཐའ་རྣམས་ཀྱི་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཨུ་ཡ་ཏ། བྱེཉ། བཱི་ཡ་ཏེ། ཧྭཉ། ཧུ་ཡ་ཏེ། ཨཱ་ཧཱུ་ཡ་ཏེ། །བ་དཱ། ཡུརྱ་ཏེ། །བས། ཤཱི་སི་བསི་གྷ་སི་རྣམས་ཀྱང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཨུ་་རྱ་ཏེ། ཊུ་ཨོ་ཤཱི། ཤུ་ཡ་ཏེ། ཨས་ཡོད་པ་ལའོ། །ཨ་སྟིའི་བྷུའོ་ཞེས་པས་་་བྷུ་རུ་བསྒྱུར་རོ། །བྷཱུ་ཡ་ཏེ། ཙཀཱི། ཙགླིངའི་ཁྱུཉ་རུ་བསྒྱུར་རོ། །ཁྱུ་ཡ་ཏེ། ཨཱ་ཁཱ་ཡ་ཏེ། ཨ་ཛ། ཨ་ཛིའི་བཱིའོ་ཞེས་པས་བཱ་་རུ་བསྒྱུར་རོ། །བཱི་ཡ་ཏེ། ཨུ་ཧ། ཀྵ་རྣམ་པར་རྟོག་པ་ལའོ། །སཾ་སྔོན་དུའོ། །།ཨུ་ཧིའི་ཉེར་བསྒྱུར་ལས་་་་ཐུང་ངུའོ་ཞེས་ཁ་ཅིག་གོ། །ས་མུ་ཉྩ་ཏེ། འདིར་བ་ཧིའི་ནཾ་རེབ་ཏུ་སྦྱོར་བ་ལའོ། །ཤནུ་དྷནུ་ཨིཏ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར་ཨ་ནུ་ཥཇ་དབྱིའོ། །ཤ། བྱ་ཏེ། དྷ་སྱ་ཏེ། ཀ་པི་བཱ་ཀ་ྩ་ཏེ། མ་དི། །མནྟ་ཏེ། པ་ཙི། རྒྱས་པར་བརྗོད་པ་ལའོ། །ཕྲ་པཉྩ་ཏེ། ཤཱ་སུ། རྗེས་སུ་སྟོན་པ་ལའོ། །ཤཱ''hཉེ+།
