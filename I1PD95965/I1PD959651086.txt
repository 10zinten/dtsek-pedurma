ཨཱི་ཤ་། ཨཱིཏྟཱ་མ། ཨཱི་ཤཱ་ཏཱ་མ། ཨཱི་ཤ་ཏཱ་མ། ཨཱི་ཤི་ཥ། ཨཱ་ས། ཨ་ལྟ་མ༑ ེཨཱ་སཱ་ཏཱ་མ། །ཨཱ་ས་ཏཱ་མ། ཨཱསཱ། ཨཱང་ལས་ཤཱ་ས། ་ཨཱ་ཤཱསྟ་མ། ཨཱ་ཤཱ་སཱ་ཏྭཾ། །ནཨཱ་ཤཱ་ས་ཏཱ། ཨཱ་༔ཤཱསཿབས། བསྟམ། །:ལྟུང་། པྲ་སུ་་ཏཱ་མ། །པྲ་སུ་བཱ་ཏཱ་མ། པྲ་སུ་བ་ཏཱ་མ། པྲ་སཱུ་། པྲ་སུ་བཱ་ཐཱ། པྲ་སཱུ་དྷཾ། སཱུ་ཏི་ལས་པཉྩ་མཱི་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །།།པྲ་སུ་བི། བྲ་སུ་བཱབ་ཏི། ་བྲ་སུ་བཱ་མ་ཧེ། དྲི་དྷིང་། ཨཱ་དཱི་དྷཱི་ཏཱམ། ཨཱ་དཱི་དྷཱཏཱམ། ཨཱ་དྲི་དྷ་ཏཱམ། ་་ཨཱ་དཱི་དཱི་ཥྛ། ཌཨཱ་དཱི་དྷཱ་ནཱཾ། ཨཱ་དཱི་དྷི་དྷཱམ། དྲི་དྷེ་བེ་བཱི་དག་གི་འང་ངོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཨཱི་དཱི་དྷི། །ཨཱི་་དྲི་རྒྱུ་བ་ཧི། ཨཱི་དྲི་དྷཱ་མ་ཧི'། དེ་བཞིན་དུ། ཨཱ་བེ་བཱི་ཏཱམ། ཨཱ་བེ་བྱཱ་ཏཱིམ། ཨཱ་བེ་བྱ་ཏཱམ༑ཨ་བེ་བཱི་ཥཱ། ཡུ་ཡོཏུ། ཏཱ་ཏཎ། །ཡུ་ཏཱཏ། །ཡུ་ཏཱམ། ཡུ་བནྟུ། ཡུ་ཧི། ཏཱ་ཏ་ཎ། ཡུ་ཏཱཏ། ་ཨིཎ། ཨེ་ཏུ། ཏཱ་ཏཎ། ཨི་ཏཱཏ། ཨཱི་ཏཱམ། ་ཡནྟུ། །ཨེ་ཧི། ཏཱ་ཏི་ཎ། ཨི་ཏཱ་ཏ། ཨི། །་ཨི་༔དྷཱི་ཏཱམ། ཨ་་དྷཱི་ཡཱ་ཏཱམ། ཨ་དྷི་མ་ཏཱཾ། ཨ་དྷཱ་། ཨི་ཀ། ཨ་དྷེ་ཏུ། ཏཱ་ཏ་ཎ། ཨ་དྷཱི་ཏཱཏ། །ཨི་དྷཱ་ནཱཾ། ཨ་དྷི་ཡནྟུ། ཤྲཱི་བེཏྟཱ། ཏཱ་ཏ་ཎ། བཱི་ཏཱཏ། བཱི་ཏཱམ། བཱི་ཡནྟུ། །བཱ་བཱ་ཏུ། ཏཱ་ཏེ་ཎ། བཱ་ཏཱཏེ། བཱ་ཏཱམ། བཱནྟུ། བཱ་ཧི། ཏཱ་ཏཎ༑ཋབཱ་ཏཱཏ། བི་ད། །བེཏྟཱ། བཱི་བཾ་ཀ་རོ་ཏུ། ཏཱ་ཏན། བིཏྟཱ་ན། བིཏྟཱཏ༑བིཏྟཱམ། བི་དནྟུ།  ོ་བི་དྷི། ཏཱ་ཏཎ། །བཱིཏྟཱཏ། བིཏྟཱ་་མ། བི་ཏྟ། བི་དཱ་ནི། ཨས་ཡོད་པ་ལའོ། །ཨསྟུ། ་ཏཱ་ཏཎ། ་སྟཏ། སྟམ། སནྟུ༑ཨསྟི་ལས་སོ་ཞེས་པས་དྷིའི་དངོས་པོའོ། །དཱ་ཨ་སྟི་ཨེ་སྟེ་ཨ་བྷཱ་ས་དབྱི་བ་འང་ངོ་ཞེས་པས་ས་ཡིག་གི་ཨེ་ཡིག་གོ། །ཨེ་དྷི། ཏཱ་ཏ་ཎ། ་སྟཏ། སྟམ། །སྟ༑། ཨཱ་སཱ་ནི། ཨཱ་སཱ་བ། ཨ་སཱ་མ། ཚྭ་ཛུ། མཱནྟུ། ཏཱ་ཏཎི། ་མྲྀ་ཏེ་IC:ALICU།
