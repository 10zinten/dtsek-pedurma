ལས་ཐོན་པ་དེ་ནི་སྒང༌། །བྱེ་མའི་ངོགས་དང༌། །བྱེ་མའི་ཐང་། །བཞི་མདོའི་ཕྱག་དར་ཞེས་པ་ཡང་། །བཱ་ལུ་ཀཱ། བཱ་ལི་ཀཱ། །འདབ་མ་དང་ནི། ཆུ་འགེབས་དང་། །རྫས་དང༌། ཨོཾ་ཀ། ཀརྦ་ཊ། །འདམ་དང་འཇིམ་རྫབ་བཱ་ད་དང་། །ཛམྦྷ་ལ་དང་འབྱིང་འབྱེད་དང༌། །ཛབྷ་དང་ཆུ་རྙོག་གོ། །གྲུ་དང༌། །སྒྲོལ་བྱེད། ནོ་རྣམས་སོ། །།ག་ད་མ་ནི་ང་། གས་དང་། པ་གྲུའི་མིང་། སྒྲོལ་བ་རྟ་རི། ནོ་ཀཱ་ནོ། ཤིན་ཏུ་རྒྱལ་བྱེད། སྒྲོལ་བ་པོ། །ཏ་རཎྜི་ཀཱ། ཆུའི་ཤིང་རྟ། ་། བཱ་ད་:ལི་ནི་ཞེས་བྱའོ། །རྒྱལ་དང༌། ཆུ་་་རྩལ་གཟིངས་དང་ནི། །བྷེ་ལ་ཀ་དང་སྒྲོལ་བའོ། །སྒྲོལ་བ་ཧེ་ཌ། མདུང་ཅན་་དང༌། །བཏུང་སྣོད། བྱེད་པ། གྲུ་སྐྱ་དང་། །གྲུ་ཡི་བཎ་ཊ་དག་གོ། །བཱ་ད་ར་བ། སོ་ལི་ཛཾ། །དེ་ཉིད་བར་གྱི་སྟན་ཡིན་ནོ། །གོགས་པཊྚ་ཏི་རཾ་ཏྲཾ་གཱ། གྲུ་སྐྱ་དང་ནི། གཡོ་བྱེད་དང་། །ཆུ་སྐྱོང་སྣོད་ལ་ཆགས་པ་དང༌། །སྒྲོལ་བའི་ཤིང་རྟ་རྣམས་ཡིན་ནོ། །གྲུ་སྐྱ་དང་ནི། སྐྱོང་བྱེད་དོ། །སྣོད་སྐྱོང་བ་དང་ལཱ་དྷ་ཊ། །གྲུ་སྦྱར་ཤིང་གི་ཏོག་ཙིའོ། །ཆུ་འཐོར་སྣོད་དང་། འཐོར་་་བྱེད་དོ། །ཤིང་གཞོང་དང་ནི་ཆུ་ཡི་སྣོད། །རྟིང་འཇུག་ཤིང་དང་བསྒྱུར་། བའི་ཤིང༌། །གྲུ་བཙས་དང་ནི། མཉན་པའི་གླ། །རྣ་བ་འཛིན་དང་། གདོས་པ་འཛིན། །གླིང་གཞན་རྒྱུ་དང༌། གྲུའི་ཚོང་པ། །གྲུ་འདྲེན་པ་ཞེས་བྱ་རྣམས་སོ། ་། གཤིན་རྗེའི་གནས་དང༌། གཏིང་ཟབ་དང་། །ཚོང་གནས་གཤིན་རྗེའི་འོག་མཐིལ་རེག། རྒྱལ་ཆེན་ཉལ་འཆིང་དྲ་བ་འཛིན། །ཆུ་ཡི་་འབངས་དང་ཉ་པ་དང༌། །།ཉ་རྒྱ་ལྡན་པ་རྣམས་ཡིན་ནོ། །ཚོགས་ཀྱི་སྲད་བུ། གཙང་བྱེད་དེ། །དྲ་བ་དང་ནི་ཉ་ཡི་དོལ། །འཆིལ་མ་དང་ནི། ཉ་འབིགས་སོ༑། རྒྱ་མཚོ་ཅན་དང་། །རྒྱ་མཚོ་བ། །གཏིང་ཟབ་དྭངས་དང་། གསལ་།
