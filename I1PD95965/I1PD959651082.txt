དབྱིའོ༑། ཛ་ཡ་ཏ། ཛ་ཡེ་ཏེ། ཡཱམ་ཡུས་དག་གི་ཨི་ཡམ་ཨི་ཡུས་དག་གོ་ཞེས་པས་ཨ་ཡིག་ལས་ཕ་རོལ་གྱི་ཡམ་དང་ཡུས་དག་གི་གནས་ལ་ཨི་ཡམ་ཨི་ཡུས་དག་་་ཏུ་འགྱུར་རོ། །ཛ་ཡེ་ཡུRཛ་ཡེ་ཛ་ཡེ་ཏེ། ཛ་ཡེ་ཏ། ཛ་ཡེ་བ་ཡ། ཛ་ཡེ་བ། །ཛ་ཡེ་མ། ཨཱཏྨ་ནེའི་ཚིག་ལ། བི་ཛ་ཨ་ཏ། བི་ཛ་ཡེ་ཡཱ་ཏཱི། བི་ཛ་ཡེ་ར་ན། བི་ཛ་ཡེ་ཐཱ། བི་ཛ་ཡེ་ཡཾ་ཐཱམ། བི་ཛ་ཡེ་དྷཱམ། བི་ཛ་ཡེ་ཡ། བི་ཛ་ཡེ་བ་ཧི། །བི་ཛ་ཡེ་མ་ཧི། དེ་བཞིན་དུ་ཐམས་ཅད་དུའོ། །ཨ་ད་ལ་་་སོགས་པ་ཨདྷཱཏ། །ཨདྱ་ཏཱ་མ། ་ཨངཿཧཙཱ་ཏྭཾ། ཧནྩཨཱཏྨ་ནེའི་ཚིག་ལ། ཨཱ་གཱི་ཧ། །ཁ་ཨ་ཧྥི་ཡཱ་ཏཱམ། ་ཨཱ་གྷི་རན། །པཱུ་ཡཱ་ཏ། སྟ་མཱི་ལའང་ངོ་ཞེས་་་་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །་། བི་ཏྱ་ཏ། སཾ་བི་དཱ་ཏི། སྱ་་ཏཱ། སྡུ་་ཏམ། ལ། སྱ་gཡང་གི་སྒྲའི་མང་པོའི་དོན་ཉིད་ལས་ཥ་ཉིད་དོ། །ཨ་བྷི་ཥྛ་ཏ། མྲ་ཛ། །པྲ་ཛྙཱ་ཏ། །བ་ཙུཏ། རུ་དྷཱཏ། །་སྭ་བྷཱུཏ། ཛ་གྲི་ཡཱཏ། །ད་རི་་ཏྲི་ཡཱཏ། ཙཀྵ་སྭཱཏ། །ཤི་བྷཱུཏ། པ་རསྨིའི་ཚིག་རྣམས་སོ། །ཙ་ཀློང་། །བྱུ་ཙཀྵི་ཏ། ས་མཱི་རཱི་ཏ། །ཨི་ཌི་ཏ། ཨི་ཤི་ཏ། ཨཱ་ཧཱུཾ་ཏ། ཨཱ་ཤཱི་སཱི་ཏི། ཤ་ཡཱི་ཏ། ཨ་དྷི་ཨཱི་ཏ། ནི་ཧུ་བཱི་ཏེ། ཨཏྨ་ནེའི་ཚིག་རྣམས་སོ། །དཱི་དྭ་་་ཏ། ཏེ་དྲི་བྷཱུ་ཏྭཾ། །དི༔དྲྀ་ཏཱི་ཏ། དྭི་ཥི་ཡཱ་ཏཱམ། པཱི་ཏཱི་ར་ན། དུ་ཧ། །དུ་ཧཱ་ཏི། དུ་ཧཱུ་ཏཱི། དུ་དྷུདུ་ཧཱི་ཏ། །དུ་ཧཱི་ཡཱ་ཏཱམ། ་་དུ་ཧཱི་རན། དི་ཧ། དི་དྷཱཏི། འདི་:ཧུ་ཏི། ལི་པ། དི་ཧྥུདི་ཧཱི་ཏ། དཱི་ཧཱི་ཡཱ་ཏྭཾ། དི་ཧཱི་:རི་མ་ལཾཧ། ལི་ཧཱཏ། ལི་ཧྥུ་ཏྭཾ། :ལི་ཧཱུ་ལི་ཧཱི་ཧ། ལི་ཧཱི་ཡཱ་ཧཱུཾ། ལི་ཧཱི་རིན། པྲནྡྷཡཱཏ། །འོ་པྲཆུ་ཡཱ་ཏཱམ། པྲནྡྷཱ་ཡུཔྲཛྙཱ་ཡཾ?། པོཎྜཱུ་ཏྲི་ཡཱ་ཏཱི། པྲཎྜཱ་བཱ་རིནི། པུ་ཡཱཏ། སྟུ་ཡཱ་ཏཱི་སྟེ་ཡུ། སྟེ་བཱི་ཏཱ༑ཏུབཱི་ཡཱ་ཏི། ་སྟེ་བཱི་རི་ན། ་བྲུ་ཡ་ཏ། བྷཱུ་ཡཱ་ཏཱལི། བྲུ་ཡུབྲུ་་ཧཱུཾfotonlo།
