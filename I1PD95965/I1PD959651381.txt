ད་ལྟར་འགྱུ་བ་ལས། བསྟན་འགྱུར། སྣ་ཚོགས། ༼ཝིའིནལ་ཁཚོཥ་ཎ་འོ། བ་དང་བསམ་ཡངས་དང༌། །མི་ཞན་པ་དང་མཐོན་པོ་དང༌། །བདག་ཉིད་ཆེ་བ་བསམ་པ་ཆེ། །སྐྱབས་གནས་རྒྱ་ཆེན་ཡོན་གྱི་གནས། །རྩོམ་པ་ཆེ་་་་དང་སྤྲོ་བ་ཆེ། །བརྩོན་ཆེན་མཚོན་ཐབས་བསླབ་ཅཔ་དང་། །བརྩེ་ལྡན་དང་ནི་སྙིང་རྗེ་ཆེ། །སྙིང་བརྩེ་ལྡན་དང་ངེས་པ་དང༌། །ཕུན་ཚོགས་ཅན་དང་་་མཚན་ལྡན་དང་། །རྨི་ལ་དང་དཔལ་ལྡན་དང༌། །དཔལ་དང་མཛེས་པ་ཡང་དག་ལྡན། །ལས་ལ་བྱང་དང་ཁ་བྱང་དང༌། །བསླབ་པ་དང་ནི་དགེ་དང་་་་མཁས། །མངོན་ཤེས་རྣམ་ཤེས་ངེས་པར་བྱང་། །ལེགས་ཤས་རྣམ་པར་ཤེས་པ་ཅན། །ཤེས་ལྡན་དང་ནི་གོ་བ་ཅན། །དེ་ནས་ནུས་ལྡན་སྟོབས་ལྡན་མཚུངས། །ལྷ་དང་རྗེ་བཙུན་མདོ་ལྡན་དང་། །འོད་ལྡན་ལེགས་ལྡན་མཆོད་བྱ་དང་། །ཞབས་དང་མཐར་འགྲོ་ཡོན་གནས་དང་། །མཆོད་འོས་སྦྱིན་འོས་:ཡོན་དབུལ་བྱ། །རབ་སྦྱིན་གཏོང་ཕོད་སྦྱིན་རླབས་ཅན། ཏུ་། མང་དུ་འགོད་དང་སྙན་སྨྲ་དང་། །སྦྱིན་དང་ཚུལ་ལྡན་དཔལ་སྟུག་དང་། །ཤིན་ཏུ་སྦྱིན་བྱེད་སྦྱིན་པ་པོ། །སྦྱིན་་་་བྱེད་གྲོལ་བ་སྣ་ཀ་ལ། །སྦྱིན་པོ། སྤྱོང་པོ་ཞེས་བྱ་དང༌། །ཡོངས་སུ་རྟོགས་་་དང་རྒྱུ་མཚན་ཅན། །མཆོག་སྦྱིན་དང་ནི་རབ་ཏུ་སྟེར། །ཡུན་རིང་འཚོ་དང་ཚེ་ལྡན་ནོ། །བཤེས་གཉེན་ཅན་དང་བསྟན་བཅོས་རིག། ཤེས་རབ་ཅན་དང་་་་ལེགས་སྦྱངས་ཤེས། །ཡིད་གཞུངས་པ་དང་བློ་གཙང་མཚུངས། །ཡིད་དགའ་་མགུ་དང་ཡིད་རབ་བདེ། །སེམས་ནི་རངས་པར་བྱེད་པའོ། །མིག་ངན་དང་ནི་ཡིད་ངན་དང་། །ཡིད་ནང་ཆད་དང་ཡིད་མི་བདེ། །བློ་ཆུད་ཟོས་དང་སེམས་མྱོས་པ། །སོཥཎྜི་དང་ཨུ་ངྐཎྜི། །ཐེ་ཚོམ་དང་ནི་སོམ་ཉི་དང༌། །ཡིད་་གཉིས་དང་ནི་མ་ངེས་པ། །ཡོན་ཏན་ནོར་ལྡན་མཚན་ཉིད་དང་། །ཕན་པར་བྱེད་པ་མཚན་ཉིད་ཅན། །ཤེས་ལྡན་རྒྱ་ཆེར་གྲགས་པ་ཅན། །རིགས་ལྡན་རྣམ་།
