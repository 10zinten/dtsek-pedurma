སཾ་བྱ་བ་ཧཱ་རཱ་ཡ་སཾ་ཀྲྀ་བྱ་ཏེ། ...ཡོངས་སུ་བཟུང༌། །ཨ་ཨི་ཨུ་རྀ་ལི་བཟོད་མིན། །དེ་ལ་དང་པོར་བརྡ་བྱེད་ཡོངས་སུ་བཟུང་བས་ཤིན་ཏུ་ལྷག་པར་བྱ་བ་བྱེད་དོ། །ཨ་ཨི་ཨུ་རི་ལ་ས་མཱ་ནཱ། ཏུ་ཨ་ནེ་ན་པྲ་ཏྭཱ་ཧཱ་ར་གྲ་ཧ་ཏཱ་ཡ་བནྡྷ། པ་རི་ག་མཚུངས་པ་རྣམས་སོ། །འདི་རྣམས་སོ་སོར་བསྡུ་བས་བཟུང་བའི་ཡི་གེ་ཡོངས་སུ་འགྲོ་བར་བྱེད་དོ། །དེ་རྣམས་ཏྱནྟེ་ཏཾ་ས་མཱ་ན་སཾ་ཛྙཱ་ཙ་བི་དྷི་ཡ་ཏེ། ནི་ཏེ་ཥཱ་སུ་ཏྲེ་ཥཱ་སནྟི་ར་ནུ་སནྟེ་ལ་མཚུངས་པ་ཞེས་པའི་བརྡ་ཆད་ཀྱང་རིག་པར་བྱའོ། །འདི་རྣམས་མདོར་མཚམས་སྦྱར་བ་རྗེས་སུ་མ་སྦྱར་བ་ཡོཉ་པི་བ་ཏ། བི་བཀྱི་ཏ་པ་སཾ་དྷི་བྷ་བ་ཏཱིཏི་ནི་ཡ་་སྟེ་མཚམས་སྦྱོར་བ་ནི་འདི་དག་སྟེ་ཉན་འདོད་ན་འདིར་ངེས་པར་མཚམས་ྋསྦྱོར་བའི་ལུགས་སུ་འགྱུར་བ་ལས་མཱ་ཏྲ། ཡ་ན་ལོ་ཀི་ཀ་པྲ་ཡོ་ག་ནིཐན་ཡེ་ས་མ་ཡ་མཱ་ཏྲ་ཏཱཙཱ། མི་འདོད་པས་སོ། །འཇིག་རྟེན་གྱི་ཡོངས་སུ་སྦྱར་བ་དང་ཡོངས་སུ་གྲུབ་པ་ཅུང་ཟད་ཙམ་དེ་ཡང་ངོ༌། །ཧྲ་སཱ་དཱིནྡྷ་པཱུ་ཏ་བྷ་དེསཌྷ་ཎྞ་ཨེ་ཏེ་ཏཱཾ་ཧྲ་ས་དཱིནྡྷཱ་ྋསླུ་ཏ་བྷེ་ཐུང་ངུ་དང་རིང་པོ་སླུ་ཏས་ཕྱེ་བ་རྣམས་འདྲ་བ་རྣམས་སོ། །འདི་རྣམས་ཀྱི་ཐུང་ངུ་དང་རིང་པོ་དང་པ་ལུ་དཱཔ་ར་སྤ་རཾ་ས་བྷ་ཙྪཱབྷ་ཏྱཾ་ཏེ། ཀེ་ཏཱ་ཤེའ་སྐྱ། ལོ་ཀཱཙྪེ་ཥ་སྱ་ལྷས་མ་སིདྡྷི་ར་ཏི་ཏའི་དབྱེ་བ་རྣམས་ཕན་ཚུན་དུ་འདྲ་བའི་ཡི་གེར་བརྗོད་དོ། །ལྷག་མ་འཇིག་རྟེན་ལས་གྲུབ་པ་འདིར་་་་་བསྒྱུ་ཏི། ་ག་ཏ་ཏོ་ལོ་ཀ་ཏ་ཨེ་བ་ཧྲ་སཱ་དི་སཾ་ཛྙཱ་ཙཱ་ཏ་བྷཱུ། ཨེ་ཀ་མཱ་བཤད་དོ། །དེ་ལ་འདི་ནི་འཇིག་རྟེན་ལས་ཐུང་ངུ་ལ་སོགས་པའི་བརྡ་ཆད་རིག་པར་བྱས་པ་སྟེ། ཚད་དྲོ་ཧྲ་སཱུལོ་བདྲི་མཱ་ཀྲོ་དཱིནྡྷཏྲི་མཱ་ཏྲཿརླུ་ཏཿབྱཉྫ་་་་་གཅིག་པ་ཐུང་ངུ་དང་། ཚད་གཉིས་པ་རིང་པོ་དང་། ཚད་གསུམ་པ་དྷཱུ་ཏ་སྟེ། གསལ་བྱེད་རྣམས་ཕྱེད་ནོ་ཙཱཌྷ་མཱ་ཏྲ་ཀཾ་། སི་ཡུ་དནྟ། ...ཨེ་ཥ་མ་རྩེ་གླུ་དཱནྟཱ་དི་པེ་དཱ་་སནྟི། །ཀྱི་ཚད་ཅན་ནོ། །འདི་ལས་གཞན་གསངས་མཐོན་པོ་ལ་སོགས་པའི་དབྱེ་བ་ཡོད་དེ། སྐད་གསངས་མཐོན་o。poloh།
