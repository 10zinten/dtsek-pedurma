།ད་ཡིག་དི་་ཏྱནྟཱ་དེ་བརླའོ། ་། ན་ཡིག་གི་ཨ་ཡིག་བརྗོད་པའི་དོན་ནོ། །སཱ་བ་སཾ་བུ་དྷ་བ་་ཞེས་པས་རིང་པོའམ། སུ་ཞེས་སུ་དབྱིའོ། །སཾ་བོ་དྷ་ནོ་ཏུ་སོ་པ་ར་ཏ་ཙོ་ས་ཞེས་པས་ཕྱོགས་གཅིག་ལ་ནངའོ། །ན་སཾ་བུ་དྷཱ་བ་ཞེས་པ་ན་མི་དབྱི:འོ། །ཨུ་ཤ། ཨུ་ཤ་ན་སོ། ་ཨུ་ཤ་ན་སཿཨུ་ཤ་ན་སཾ། །ཨུ་ཤ་ན་སོ། །ཨུ་ཤ་ན་ས་ཨུ་ཤཱ་ན་ཤཱ། ཨུ་ཤ་ནོ་ཧཱུཾ། ཨུ་ཤ་ནོ་བྷི་ཨུ་ཤ་ན་སེ། ཨུ་ཤ་ནོ་བྷཾ། ཨུ་ཤ་ནོ་ཨུ་ཤ་ན་སཨུ་ཤ་ན་ཧཱུཾ། ཨུ་ཤ་ནོ་བྷུཨུ་ཤ་ན་སཿཨུ་ཤ་ན་སོཨུ་ཤ་ནཱ་སཾ། ་ཨུ་ཤ་ན་སཾ། :ཨུ་ཤ་ན་སཾ། ཨུ་ཤ་ན་སི། ཨུ་ཤ་ན་སོཨུ་ཤ་ན་སླུ། ་ཨུ་ཤ་ན་སུ། །ཧེ་ཨུ་ཤ་ནན། ཧེ་ཨུ་ཤ་ན་སོ། ཧེ་ཨུ་ཤ་ན་ས། སུམྤའི་སྒྲ་ལ་སུམ་སོ། །སུ་དི་ཏི་སུ་ཊི་ས་ཡིག་ལ་ཨུ་སུ་ང་་་བསྟན་པར་བྱའོ། །ང་ཡིག་ནི་རི་ཏི་མཐའི་དོན་ནོ། །།ཨུ་ཡིག་བྱུ་ཁྱུ་གི་ཏི་ཞེས་པ་མཚན་མའི་དོན་ནོ། །ཨ་ནེ་ནི་བ་སུ་ཊ་ལ་ནུ་མའོ། །མ་ཡིག་མི་ཏ་་་ཙོཉནྟུད་པར་ཞེས་པས་མཚན་མའི་དོན་ནོ། །ཨུ་ཡིག་བརྗོད་པའི་དོན་ནོ། །སཱ་བ་སཾ་བྷ་དྷཱ་བ་ཞེས་པས་སཾ་བུ་རྣོ་སོ་བར་རཏྠོ་རིང་པོ་ཉིད་དོ། །སུ་ཞེས་པས་སུ་དབྱིའོ། །སཾ་ཡོ་ག་སྱ་པ་ད་མྱོ་ཏིས་ས་ཡིག་དབྱིའོ། །ས་ཡིག་ཕྱིས་པ་ལ་་་པཱུརྦ་ཏྲ་སི་དྷཱི་ཞེས་པས་ས་ཡིག་དབྱི་བར་གྲུབ་པའི་ཕྱིར། །སུ་པ་པྲ་ཀྲི་ཏེས་ན་དབྱི་བ་མ་ཡིན་ནོ། །ན༔་ནཾ་ཏུ་སྱ་ཛྷ་ལི་ཞེས་པས་ཛྷ་ལི་ནི་ཡིག་གོ། །མ་ཡིག་ལ་རྗེས་སུ་དབྱངས་སོ། །བ་ལ་དོ་སུ་པི་བ་ཏ་དྷི་དི་ཏི་ཏ་དྷཱ་དི་ཏི་ཏེ་ཤ་ལས་སྦྱོར་བ་་ལ་པ་ད་ལ་ཞེས་པས་ས་དབྱིའོ། །ཧ་ལི་མས་ཨི་ཡིག་ལ་རྗེས་དབྱངས་སོ། །ཨ་ནུ་སཱ་ར་སྱ་ཡ་ཡི་ཡ་ཞེས་དེ་ཡང་མ་ཡིག་གོ། །སུ་བི་བནྡྷ་པྲ་ཏྭཾ་བྷཱནྡྲ་ཡ་མཱ་ད་ཤའོ་། :། པུ་མཱན། པུ་ཚྭ་སོ། ་་:པུ་ཚྭ་ས། པུ་ཡཱཾ་སཾ། པུ་ཚྭ་སོ། པུ་མཱི་ས་པུ་སཱ། —།
