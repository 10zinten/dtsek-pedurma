མཱ་རཉྫ་ར། ས་འཛིན་འཇོམས། །སྨོང་སྐྱེས། །མཐའ་ཡས། གསེར་མདུང་ཅན། །མཁའ་ལ་གནས་དང་། ཞི་བར་གྲགས། །སྐྲ་གྲོལ་བརྟན་པ། འཛིན་བྱེད་དང་། ཁ། སེམས་སྐྱེས། ཆུ་སྦྱིན། བཀོད་སྦྱིན་དང་། །ཀོ་བ་སིནྡྷའི་འབྲས། ཆར་འབྲས། །དགྲ་བོ། མི་ཡི་སེང་གེ་དང༌། །གཙང་བྱེད། འཇོ་བ། ་ཨ་རིཥྛ། ་། སྐྲ་ཅན། རྡུལ་འཛིན་འཇོམས་བྱེད་དང་། །རྭ་གཞུ་ཅན་དང་། དུང་གི་ལག། འཕེལ་བྱེད་དང་ནི་མི་ཟད་པ། །རྟ་དང་ཀོ་སུ་བྷ་འཆད་འགྱུར། །རྩེ་བྱེད་དང་ནི་དགའ་བ་འཛིན། །རྭ་གཅིག་འཁྱིལ་པ་བརྒྱ་པ་དང་། །རྡུལ་ཕྲན་ལུས་དང༌། །མ་སེར་སྐྱ་དང་། །དགའ་བའི་དབང་པོ་་་་དཔུང་བཞི་པ། །རྒྱུ་། རྟག་པ་སྦྱོར་བ་བརྒྱ་ལྡན་དང་། ག། བ་བྷཱུ་སྨིན་ཅན་དད་གཤིན་རྗེ་ཡ་མ་ནའི་དགྲ། །ཆར་སྐྱེས། །མདའ་དང་། །གཏོང་བྱེད་དང་། །གཟུགས་མང་ཅན། དང་ཀུན་ལ་ལྟ། །ཌ། ཀུན་ཤེས་པ་དང་། ཐམས་་་་་ཅད་མིགད་། ལྔ་སྐྱེས་འཛིན་དང་། །རྒྱལ་བྱེད་དང་། །ཐོས་པའི་སྙིང་པོ་་་སྐྱེས་བུ་མཆོག། མི་ཡི་སེང་གེ། །མཆོད་སྦྱིན་སྐྱེས། །རྒྱ་མཚོའི་ཟླ་བ། རི་་་བོང་ཐིག། མིའུ་ཐུང༌། འཇུག་པ་བཅུ་པ་དང་། །དང་པོའི། །ཕག་དང་། དང་པོའི་ལྷའོ། །འདི་ཡི་ཕ་ནི་ནོར་ལྷ་ཁྱབ་འཇུག་གི་ཕའི་མིང་སོགས་སྟེ། །དུནུ་ར་དང༌། ༼ཨ་ན་ཀ། དུནྡ་བྷི་དང་། བྷུ་ཀཱའོ། །རྔ་བོ་ཆེ་དང་བུག་ཅན་ནོ། །ཤིང་ལྡན་་་་་དང་ནི། །ཤིང་རྟ་འདྲེན། །བློན་པོ། །རྒོད་དང་། རླུང་ནད་ཅན། །ཤི་ནིའི་ཚ་བོ་བདེན་པ་ཅན། །བྲག་སྐྱེས། ཞི་ཡུ་ཡུ་དྷཱ་ནའོ། ཁྱབ་འཇུག་་་འཁོར་ལོ་ལེགས་མཐོང་སྟེ། །བེ་ཅོན་ཀོ་མོ་ད་ཀིའོ་ས་བདག་བྱེད། །།གཞུ་ནི་རྭ་གཞུ་གྲི་དགའ་བྱེད། །དུང་ནི་ལྔ་སྐྱེས། །རྒྱལ་བའི་དབྱངས། །འདི་ཡི་ནོར་བུ་མཐའ་བྱེད་ལག། དེ་ནི་ཀོ་སྐུ་བྷ་རུ་བཤད། །ན་ཚོད་མེད་ལ་ལས་+—།
