རི་བཱ་མ་ཆུ་ཡི་གནས། །ནཾ་ཏ་མྲ་དང་། ཡོངས་སུ་བཀབ། །པཱཉྩ་ཛེ་རྒྱ་ཧྲི་?ཊཱི་ཀ་ཤ། དུང་དང་བཱཊ་ག་ལའོ། །བོ་རུ་ཥ་ཡ་སྐྱེས་བུའི་རྩལ། ཚིག་དབྱེ་ཚོགས་པ་གསོད་པ་དང༌། །སྐྱེས་བུས་བྱས་པ་རྣམས་ལའོ། །བི་ཏ་བཱ་ཡཾ་འཇིགས་པ་དང༌། །འཇིགས་རུང་པྲ་ཏི་ཐ་མ་ནི། །འདུན་ས་དང་ནི་གནས་ལའོ། །ཕ་ལོ་་ད་ཡ་་འབྲས་བུ་རྣམས། །འབྱུང་དང་རྙེད་དང་སུམ་རྩེན་ལ། །པི་ལེ་ཤ་ཡ་བྱིས་པ་་་སྤྲུལ༑ལ། བྷཱ་ག་དྷེ་ཡཾ་སྐལ་བ་དང་། །ལག་པ་དང་ནི་ཆ་ལའོ། །:བྷི་ཏནྡྲིཡ་བྱེད་པ་དང༌། །སྒྲ་དང་སྤྱོད་ཡུལ་འདུས་པ་ལ། །མ་ཧོ་ད་ཡྻ་ཀུན་འབྱུང་་་དང༌། །བྱང་གྲོལ་དུ་མོ་རྒུར་མོ་ལ། །ཡ་ཧཱ་ལ་མ་གཙུག་ལག་ཁང་། །།མུ་སྟེགས་ཁང་དང་མཆོག་བདག་ལ། ལོ། མ་ཧཱ་མཱུ་ལི་རིན་ཐང་ཆེ། །པདྨ་རཱ་ག་ལ་་་སོགས་ལ། །མཱཛཱ་རི་ཡ་དམངས་རིགས་དང་། །བྱི་ལ་བྱ་བ་སྤྱོད་པ་ལ། །:：ར་ཏེ་ཅུ་ཡ་རབ་འཕྱང་གསོད། །ལྷག་པ་དང་ནི་ནེའུ་ལེ། །ས་མུཙྪ་ཡ་སྦྲིང་། དང་འགོག། ས་མུ་དཱ་ཡ་ཚོགས་པ་དང་། །ཀུན་ཏུ་སྦྱོར་དང་ཉེ་བའི་རིམ། །ས་མཱ་དཱ་ཡ་ཚོགས་བྲིད་སྒྲ། །སམྦ་ར་ཡ་གཡུལ་འགྱེད་དང་། །རྒུད་དང་ཕྱི་མའི་དུས་དག་ལ། །ས་ལངྒ་ཡ་སྒྲ་དང་མིང༌། །རོལ་རྩེད་ཕྱུགས་དང་བྱ་དག་ལ། །སྒྱུ་ལོཙྪ་ཡ་མཐའ་དག་ནི། །ཤིང་དུམ་རྡོ་བ་ཟ་མ་ཏོག། གླང་པོ་རྣམས་ཀྱི་བར་མ་ལ། །ཨ་ཀུ་ཏོ་བྷ་ཡཾ་ཞེས་པ། །ལྷ་ཁང་ཨ་ཀཱིརྗེ་ཤཱིཀ། ཀཱ་ལཱ་ནུ་པཱརྨི་ནི་ཤིང་། །ཤི་ཤ་བ་དང་དུས་སྐྱེས་དང༌། །རི་ལས་སྐྱེས་པ་རྣམས་ལའོ། །དུགྷ་ཏཱ་ལཱི་ཡང་་ཞེས་པ། །ཨཱ་མའི་འོ་མ་འོ་ྋམའི་ས། །དབུ་བ་ཅན་དང་དེ་ནས་ནི། །འོ་མའི་ཙ་མ་ས་དང་ནི། །ཏ་ཏ་པཎྜ་ཀཱི་ཊ་འོ། །པྲ་བ་ཙ་ནཱི་ཡ་ཞེས་པ། །བརྗོད་་་བྱའི་རབ་དང་རྗོད་བྱེད་རབ། །དྲྀཥ་ཀ་བཱ་ཡཱ་ཞེས་པ། །དཔལ་མོ་གོ་རིམ་ཕྱེ་འཚོ་བྱེད་མ། །བརྒྱ་ལས་མཆོག་ཏུ་གྱུར་མའོ། །པ་ཁྱུངྒ་མ་ནཱི་ཡཾ་ནི། །ཉེ་།
