བསྟན་འགྱུར། སྣ་ཚོགས། ཕྱིའི་?ཧ་ན། ཛཾ་གྷ་ནྱ་ཏེ། ཧ་ནིའི་འཚེ་བ་ལ་གྱི་ཡང་ན་བརྗོད་པར་བྱའོ་ཞེས་པས་གྷིར་བསྒྱུར་རོ། །ཛེ་གྷི་ཡ་ཏེ། འགྲོ་བའི་དོན་ལས་གྱ་གྱུ་ཅན་ཉིད་ལའོ་ཞེས་པས་ཡ་རྐྱེན་ནོ། །ཀྲ་མུ། ་ཀུ་ཏི་ལཾ་ཀྲ་མ་ཏི། ཙི་ཀྲ་མྱ་ཏེ། དྲ་མ། དཾ་དྲ་མྱ་ཏེ། ལུ་བ་སོགས་ལས་གནྟི་ལས་སོ་ཞེས་པས་ཡ་རྐྱེན་ནོ། །གནྟིཏཾ་ལུམྦ་ཏི། ལོ་ལུ་བྱཱ་ཏེ། ཤ་བློ། ཤཱ་ཤ་བྱུ་ཏི། ཙ་ར་པ་ལ་དག་གི་ཕ་རོལ་གྱི་ཨཱའི་ཨུཊ་འང་ངོ་ཞེས་པས་རྗེས་སུ་ང་རོ་ཨཱ་ག་མའོ། །ཕ་རོལ་གྱི་ཨ་ཡིག་གི་ཨུ་ཉིད་ཀྱང་ངོ༌། །ར་པ་དག་གི་ཨུ་པ་དྷཱའི་རིང་པོའོ། །ཙི་ཙུརྱ་ཏེ། ་བི་ཕུ་ལྕ་ཏེ། ཛ་བ་སོགས་་རྣམས་ཀྱི་འང་ངོ་ཞེས་རྗེས་སུ་ང་རོ་ཨཱ་ག་མའོ། །ཛཾ་ཛ་བྱ་ཏེ། དཾ་ད་བྱ་ཏེ། དཾ་ད་བྱ་ཏེ། བྷནྡོ། །ཀུན་ཏུ་ཉེད་པ་ལའོ། །ཨིཊ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་རྣམས་ཀྱི་ཡོན་ཏན་མེད་པ་ལ་ཨ་ནུ་ཥརྒ་དབྱིའོ། །བྷཾ་བྷ་ཛུ་ཏེ། པ་ཤ། མདོ་ལས་བྱུང་བའི་བྱིངས་སོ། །པི་པ་བྱ་ཏེ། པཉྩི་སྲི་སི་དྡྷཾ་སི་ཧྥ་ཤི་ཀ་སི་པ་ཏི་པ་དི་སྐནྡ་རྣམས་ཀྱི་མཐའ་ནཱིའོ་ཞེས་པས་ནཱི་ཨཱ་ག་མའོ། །ཨ་ནུ་ཥརྒ་དབྱིའོ། །བ་ནི་བ་ཙུ་ཏེ། ས་ནཱི་སཱ་སྱ་ཏེ། ད་ནི་དྷཱ་སྱ་ཏེ། བ་ནཱི་བྷ་བྱ་ཏེ། ཙ་ནཱི་ཀ་་སྱ་ཏེ། པ་ནཱི་བ་བྱ་ཏེ། པ་ནཱི་བ་བྱ་ཏེ། ཙ་ནི་སྐརྱ་ཏེ། བྷི་ད། བེ་བྷི་བྱ་ཏེ༑་བུདྷ། བོ་བུ་དྷ་ཏེ། བྲྀ་ཏུ། ཤྲཱི་བྷུ། རོ་ལྡན་པའི་རིའོ་ཞེས་པས་རི་ཨཱ་ག་མའོ། །པ་རཱི་ཤྲཱི་ཏྱ་ཏེ། པ་རཱི་ཤྲཱི་རྒྱ་ཏེ། དྲྀ་ཤིར། ད་རི་དྲི་བྱ་ཏེ། ཛ་རཱི་ཛིཙྪཱཏེ། ས་རཱི་སྲི་བྱ་ཏེ། ནཱིརི་ནཾ་བྱ་ཏེ། ཤཱ་རཱི་ཤྲཱི་དྷ་ཏེ། ཛཱ་རཱི་་གྲྭ་དྷ་ཏེ། པ་རཱི་པྲ་ཉྩ་ཏེ། པ་རཱི་ཕྲིཙྪ་ཏེ། རྣོ་ཀྲྀ། ཌོ་ལྷོ་ཀྱ་ཏེ། རྡོ་ཀྲྀ། ཏོ་ཌོ་ཀྱ་ཏེ། སཱུཙ། སཱུ་ཏྲ། མཱུ་ཏྲ། ཙུ་ར་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨིནའོ། །རྀཏ་སོགས་རྣམས་ལས་ཀྱང་ངོ་ཞེས་པས་ཡ་རྐྱེན་ནོ། །ཀཱ་རི་ཏིའི་ཨཱམ་ཨིཊ་བི་ཀ་ར་ཎ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཀཱ་རི་ཏའི་དབྱིའོ། །སོ་སཱུ་ནྱ་ཏེ། སོ་སཱུ་རྱ་ཏེ། TAH+།
