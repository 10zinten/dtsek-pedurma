ལ། ་། སས་ར་ཊ་SICpཟླ་། ༽ས་ག་དེ་་?། ད་དུང༌་། ཛ་ནི་མ་ནི་ད་སི་རྣམས་ལས་ཡུའོ། །ཧི་ཀྲྀ་ཤེ་དག་ལས་ཀྭ་ཀ། ཨཱ་ནུ་ཀ་དག་གོ། །མ་ནི་ལས་ཀི་སྟེ་ཨ་ཏ་ཨུཏ་ཉིད་ཀྱང་ངོ་། །ཨི་ཧི་ཀམྤི་དག་གི་ན་དབྱི་བ་ཡང་ངོ༌། །བི་:བི་ས་བ་པི་རཱ་ཛི་ཏྲཾ་ཧ་ནི་ན་བྷི་ལས་ཨི་ཉའོ། །ཨ་ཛི་ཛ་ནི་ཨ་ཏི་ར་ཤི་བ་ནི་ལས་ཀྱང་ངོ༌། །བེ་ཉ་ལས་ཌིའོ། །ནི་བིའོ། །བ་ཁི་ལ་སོགས་པ་རྣམས་སོ། ་། ཌི་ཝ་པ་ལས་ཀྟའོ། །བ་ཧི་བ་སི་ཨ་མི་རྣམས་ལས་ཨ་ཏིའོ། །ཀྱི་བི་དྷུ་བི་ལི་ཁི་ལིངྒཱ་དྷ་མི་རྣམས་ལས་ཛུ་ཀའོ། །ཧཱའི་གཉིས་ཀྱང་ངོ༌། །ཁྲི་ཥིའི་འཕེལ་བ་ཡང་ནའོ། ཀྲྀཉ་ལ་སོགས་པ་རྣམས་ལས་ཝུའོ། །ཤ་མི་ལས་ཁའོ༑། མུ་ཏིའི་མྱུར་ཡང་ངོ༌། །ཤི་ཁཱའོ། །ག་མི་ལས་གའོ། །ཛ་ནི་ལས་་་གྷའོ༑། ཀ་ཙི་ལས་ཚའོ། །ཏཱཾ་ཏཱཾ་ཏཱཾ་བི་ལས་ཀཱི་ཊའོ། །ཤ་མི་ལས་རྙའོ། །སུ་ཙི་ལས་མའོ། །ཨོ་ཏ་ལ་བྷུ་:ལ་བ་ཅིའུ་ཏའོ། །རུ་ཧྲི་ཧྲི་ཤུ་རྣམས་ལས་ཨི་ཏའོ། །ཚྭ་བྷཾ་བཱ་ཧ་སི་ཨ་མི་ད་མི་ལུ་པཱུ་རྣམས་ལས་ཏའོ། །མཱཙྪ་ས་སི་རྣམས་ལས་ཡའོ། །སམྦྷ་ལ་སོགས་པ་རྣམས་སོ། །རུ་ཙི་བྷུ་ཛི་དག་ལས་ཀི་སྐྱེའོ། །མ་དི་ལས་སྐྱའོ། །མ་དི་ཨ་ཤི་བ་མི་རྣམས་ལས་ས་རའོ། །དེ་ཝི་བ་ཐི་ཧྥ་མི་བཱ་བི་རྣམས་ལས་ཨ་རའོ། །ཤྲཱི་ཎཱ་ཏི་ལས་ཀ་རའོ། །པུ་ཥ་ལས་ཡཎ་བཞིན་ནོ། །ཙཱ་ཎོ་:ཏའི་ས་ཎ་ཊའོ། །ཀ་ཐི་ཙ་ཀི་དག་ལས་ཨོ་རའོ། །གླུ་ཎི་ལས་ཊོ་རའོ། །ཌཱ་ན་འོ། །ཨུཏྟའོ། །ཙི་མི་དི་དག་ལས་ཏྲ་ཀའོ། །པཱུ་ངའི་ཐུང་ངུ་ཡང་ངོ༌། །སི་ལ་མན་མཐའ་ཡང་ངོ༌། །ཀུ་ཊི་ལས་མ་ལའོ། །བྲཕྱི་ལས་བ་ཀའོ། །བྷཾ་ཝ་མི་ཀུ་རྣམས་ལས་ཤ་་་་།
