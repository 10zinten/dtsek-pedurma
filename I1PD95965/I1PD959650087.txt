སྣ་ཚོགས། ནིཁོ་དྲི་འགྲོ་བ་ཐོགས་པ་ལའོ། །དོ་དྲི་འགྲོ་བ་བཞི་ལའོ། །ཏྠ་ར་ཟ་འགྲོས་ལའོ། །ཀླ་ར་གནོད་པ་ལའོ། །ཨ་བྷ་བ་བྷ་མ་བྷཱ་ཙ་ར་རྣམས་འགྲོ་བའི་དོན་ཏོ། །ཊྛི་བུ་ཀྱི་བུ་ངེས་པར་འཛིན་པ་ལའོ། །ཛི་རྒྱལ་བ་ལའོ། །ཛཱི་བ་སྲོག་འཛིན་པ་ལའོ། །།པཱི་བ་མཱི་བ་ནི་བ་ཏཱི་བ་སྦོམ་པོ་ཉིད་ལའོ། །ཨུཎྚི་ཏུའྲི་ཐུརཱི་དུཏྲི་དྷུབཱི་རྣམས་འཚེ་་་བའི་དོན་ཏོ། །མུརཱི་འཆིང་བ་ལའོ། ་། གུལྡི་བཅོམ་པ་ལའོ། །པུཏྤ་པཧྥ་མཉྫ་་གང་བ་ལའོ། ་། ཙནྡྲ་ཟ་བ་ལའོ། །ཀཏྤ་ཁཉྫ་གཉྫ་དྲེགས་པ་ལའོ། །ཨཏྤ་ཤཟླ་སཉྫ་བྷཉྫ་འཚེ་བ་ལའོ། །ཨཱི་བི་ཁྱབ་པ་ལའོ། །བི་བི་མི་བི་ནི་བི་འཐོར་བ་ལའོ། །ཧི་བི་རི་བི་དྷི་བི་རྣམས་མཛའ་བའི་དོན་ཏོ། །རི་བི་ར་བི་དྷ་མི་རྣམས་འགྲོ་བའི་དོན་ཏོ། །ཀྲི་བི་འཚེ་བ་ལའོ། །མ་བ་འཆང་བ་ལའོ། །།ཨ་བ་བསྲུང་བ་ལའོ། །ཨཱི་བ་ཨཁྱུ་ཁྱབ་པ་ལའོ། །ཏཁྱུ་ཏཱཁྱུ་ཕྲ་མོར་བྱེད་པ་ལའོ། །ཨུ་ཀླུ་འཐོར་བ་་་ལའོ༑་། རཀྵ་བསྲུང་བ་ལའོ། །ཎི་ཀླུ་འོ་བྱེད་པ་ལའོ། །ཀྲྀ་ཀླུ་སྡེ་གྷ་ཎ་ཀླུ་འགྲོ་བ་ལའོ། །བཀྟ་ཁྲོ་བ་ལའོ། ་། མྲ་ཀླུ་འདུས་པ་ལའོ། །ཏཱ་ཀྵ་ཤུན་པ་ལའོ། །ཏུརྐུ་མ་གུས་པ་ལའོ། །ཀཱ་ཀཱི་བཱ་ཀཱི་མཱ་ཀྱི་འདོད་པ་ལའོ། །།དཱ་ཀྱི་དྷཱ་ྋཀྱི་དྷཱ་ཀླུ་་་འཇིགས་སུ་རུང་བའི་སྒྲ་:སྒྲོགས་པ་ལའོ། །ཧཱུཾ་ཥ་འཐུང་བ་ལའོ། །ཏུ་ཥ་་་ཚིམས་པ་ལའོ། །པུ་ཥ་འཕེལ་བ་ལའོ། །མུ་ཥ་བརྐུ་བ་ལའོ། །ཥཱ་ཥ་རྩ་བ་ལའོ༑། བྷུ་ཥ་རྒྱན་ལའོ། །ཨུ་ཥ་ནད་ལའོ། །ཨཱི་ཤ་མཐོ་བ་ལའོ། །ཀ་ཥ་སི་ཥ་ཛ་ཥ་ཛྷ་ཥ་བ་ཥ་མ་ཥ་རུ་ཥ་རི་ཥ་ཡུ་ཥ་ཛུ་ཥ་འཚེ་བ་ལའོ། །བྷ་ཥ་གཙེ་བ་་ལའོ༑། ཨུ་ཥ་སྲེག་པ་ལའོ། །ཛི་ཥཱ་བི་ཥཱ་མི་ཥཱ་བྲན་པ་ལའོ། །པུ་ཥ་རྒྱས་པ་ལའོ༑། ཤི་ཥཱ་ཤྲཱི་ཏུ་པུ་ཥཱ་བླ་ཏུ་སྲེག་པ་ལའོ། ་། པྲ་ཏུ་བྲི་ཏུ་འཐོར་བ་ལའོ། །མྲྀ་ཥཱ་བཞད་གད་ལའོ། །ཤྲཱི་ཥཱ་འགྲན་པ་ལའོ། །ཧྲ་ཏུ་བརྫུན་ལའོ། །ཏུ་ཥ་ཧྲ་ཥ་ཧྥ་ཥ་ར་ཥ་སྒྲ་ལའོ། །ཛརྫ་ཙརྫ་ཛྷརྫ་ཡོངས་སུ་བཤད་པ་ལའོ། །སྒྱུ་ཎི་ར་སྒྲ་།
