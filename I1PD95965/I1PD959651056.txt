པ་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་?ཚིག་གོ། ་། སུ་ནོ་ཏེ། །སུ་ནཱཏེ། སུ་ནཱ་ཏེ། སུ་་ནུ་ཥ། སུ་ནཱ་ཐེ། སུ་ནུ་དྷེ། སུ་ནེ། སུ་ནཱི་ཧེ། སུ་ནུ་བ་ཧེ། སུ་རྨ་ཧེ། སུ་ནུ་མ་ཧེ། ཌི་ཉ་འཆིང་བ་ལའོ། །སི་ནོ་ཏི། སུ་ནུ་། སི་ནཱནྟི། སུ་ནུ་ཏེ། སི་ནཱ་ཏེ། སི་ནཱ་ཏེ། ཤི་ཉ་རྣོ་བ་ལའོ། །ཤི་ནོ་ཏི། ཤི་ནུ་ཏ། ཤི་ནནྟི། ཤི་ནུ་ཏེ། ཌཱ་བི་ཉ། རབ་ཏུ་འཕེན་པ་ལའོ། །མི་ནོ་ཏི། །མི་ནུ་ཏེ༑ཙ་ཉ། སོག་པ་ལའོ། །ཙི་ནི་ཏི། ཙི་ནུ་ཏ། ཙི་ན་ཏེ། བྲ་ཙི་ནོ་ཏི། བྲ་ཎི་ཙི་ནོ་ཏི། བྲ་ཎི་ཙི་ནུ་ཏེ། སྡིཉ་ཀུན་ཏུ་འགེབས་པ་ལའོ། །ར་ཥ་རཾ་ཡིག་རྣམས་ལས་མཐའ་མ་ཡིན་པའི་ནའི་ཎ་སྟེ་དབྱངས་དང་ཧ་དང་ཡ་དང་ཝ་དང་ཀ་སྡེ་ངང་པ་སྡེ་བར་ན་ཡོད་ཀྱང་ངོ་ཞེས་པས་ཎ་ཉིད་དོ། །སྟེ་ནོ་ཏི། སྟེ་ཆུ་ཏེ། ཀྲྀཉ། འཚེ་བ་ལའོ། ་། ཀྲྀ་:ནོ་ཏེ། ཀྲྀ་ནུ་ཏེ། བྲིཉ་སྒྲིབ་པ་ལའོ། །ཤྲཱི་ནོ་ཏི༑བྲོ་ནུ་ཏེ། དྷུ་ཉ་འདར་བ་ལའོ། །དྷུ་ན་ཏི། དྷུ་ནོ་ཏེ། གཉི་གའི་ཚིག་རྣམས་སོ། །ཊུ་དྷུ་ཉེ་བར་གདུང་བ་ལའོ། །ཌཱ་ནོ་ཏི་ཧི། འགྲོ་བ་ལ་སྟེ་འཕེལ་བ་ལ་ཡང་ངོ༌། །ཧི་ནོ་ཏི། བྲ་སྔོན་དུའོ། །ཧྲི་ནུ་མཱི་རྣམས་ཀྱིས་། ཞེས་པས་ཎ་ཉིད་དོ། །བྲ་ཧི་ནོ་ཏི། བྲྀ་དགའ་བ་ལའོ། །བྲི་ནོ་ཏི། སྨྲ་སྐྱོད་པ་ལའང་ངོ༌། །སྨྲེ་*ཎོ་ཏི། སྤྲི་ཆུ་ཏཿ། སྨྲ་ནཱ་ནི། སྤྱི་ནོ་ཏི། སྨྲ་ནུ་ཐ༑སྨྲ་ཆུ་ཐ། སྨྲ་ཎོ་མི། སྤྱི་ཆུ་བ། སྨྲ་ཎ། སྨྲ་ཆུ་མ། སྨིནྣེ། ཨ་མི་ཁྱབ་པ་ལའོ། །ཨཱ་སྣོ་ཏི། པཱ་སྣོ་ཏི། བྱཱ་སྣོ་ཏི། ཤ་ཤྲཱི་ནུས་པ་ལའོ། །ཤ་ཀོ་ཏི། ཤ་ལྷུ་ཏཿ། འདིར་སྔ་མ་བརྩེགས་པ་མིན་པའི་ཞེས་སྤྱིར་བཏང་བ་་་ལས་དི་ཀ་ཎའི་ནུའི་འང་ངོ་ཞེས་པས་ཨུབ་ཏུ་བསྒྱུར་བ་ཁོ་ན་འགྱུར་རོ། །པྲ་སྣ་བནྟི༑ཤཀཱུ་པནྟི། རཱ་དྷཱ་དང་སཱ་དྷ་ཡང་དག་པར་གྲུབ་པ་ལའོ། །རཱདྷཱ་ཏི། སཱངྒཱི་ཏི། ཏི་ཀ་དང་ཏི་ག་དང་ཥ་གྷ་དང་རིཀཱི་ཙེདང་ཀྲྀ་བི་དང་ཙི་རི་དང་ཛི་རི་་།
