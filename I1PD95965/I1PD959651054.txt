དྷ་ཏེ། ཨ་ནུ་ལ་རུ་དྷ། འདོད་པ་ལའོ། །ཨ་ནུ་སྔོན་མའི་རུ་དྷ། །འདོད་་་་པའི་དོན་ལ་འདི་པ་དང་པོ་ཅན་ནོ་ཞེས་པའི་དོན་ཏོ། །ཨ་ནུ་རུ་དྷ་ཏེ། མ་ན་ཤེས་པ་ལའོ། །མ་རྱ་ཏེ། ཨན་སྲོག་ལའོ། །ཨ་ནྱ་ཏེ། ཡུ་ཛ་ཏིང་ངེ་་འཛིན་ལའོ། །ཡུ་ཛུ་ཏེ། སྲི་ཛ་རྣམ་པར་གཅད་པ་ལའོ། །སྲ་ཚུ་ཏི། ཨུཏི་ཛུ་ཏེ། བི་སྲ་ཛ་ཏེ། ལ་ཤ་ཆུང་ངུར་གྱུར་པ་ལའོ། ་། ལི་བྱ་ཏེ། ཨཱཏྨ་ནེའི་ཚིག་རྣམས་སོ། །ཤ་ཀ། མྲྀ་ཥ། བཟོད་པ་ལའོ། །ཤ་ཀྱ་ཏི། ཤ་ཀྱ་ཏེ། མྲྀ་ཥྱ་ཏི། རྀ་ཥྱ་ཏི། བཟོད་པ་ལས་གཞན་ལས་སུ་སོགས་ཉིད་ཀྱི་་་་ཕྱིར་ནུའི་རྐྱེན་ནོ། །ཤཀོ་ཏི། ཚགས་སུ་བྱས་པའི་རྟག་ཏུ་བ་མ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར༑པ་རི་མི་ཥཱ་ཏི། པ་རསྨིའི་ཚིག་གོ། །དེ་བཞིན་དུ་པྲ་ཏི་མྲྀ་བྱ་བ་ཏི། ཨི་ཤུ་ཙི་ར། དག་པར་གྱུར་པ་ལའོ། །ཨི་ཡིག་ནི་ཌི་ཤཱི་ཨཱི་ཏེ་རྗེས་སུ་འབྲེལ་པ་དང་ཨི་ཏ་ཡང་ན་བ་རྣམས་ཀྱི་པ་ཏི་ནིཥྛ་ཥ་དག་གི་མ་ཡིན་པའི་མ་ཡིན་ནོ་ཞེས་་་་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨི་ཡིག་དང་རེ་ཕ་ཡང་ཨིད་རྗེས་སུ་འབྲེལ་པ་ལས་་་ཀྱང་ངོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཤུ་ཙུ་ཏི། ་ཤུ་ཙ་ཏི། ཎ་ཧ་འཆིང་བ་ལའོ༑། ན་རྒྱ་ཏི། ན་རྒྱ་ཏེ། སཾ་ན་རྒྱ་ཏི། སཾ་ན་རྒྱ་ཏི། ཡི་གེ་ཉམས་པ་ཉིད་ཀྱིས་ཨ་པིའི་ཨཏ་ཡང་ན་དབྱིའོ། །པི་ན་རྒྱ་ཏི། ཨ་པི་ན་རྒྱ་ཏེ། པི་ན་་བྱ་ཏི། ཨ་པི་ན་རྒྱ་ཏེ། པྲ་ཎི་ན་རྒྱ་ཏི། པྲ་ཎི་ན་རྒྱ་ཏེ། རཎྜ་ཚོས་ལའོ། །ཡནའོ༑། །ཨིཏ་རྗེས་སུ་འབྲེལ་བ་མ་ཡིན་པ་རྣམས་ཀྱི་ཡོན་ཏན་མེད་པ་ལ་ཞེས་་་པས་ཨ་ནུ་ཥངྒ་དབྱིའོ། །ར་ཛུ་ཏི། ར་ཛུ་ཏེ། ཤ་ལ་སྤྱོ་བ་ལའོ། །།ཤ་བྱ་ཏི། ཤ་བྱ་ཏེ། ཞ་རྣམ་པར་བཤད་པ་རྣམས་སོ་ཞེས་པ་ཡན་བི་ཀ་ར་ཎ་ལས་དི་བ་:ལ་་་་སོགས་པ་རྫོགས་སོ། །།ད་ནས་ད་པ། འབ་པ། འུ་ནང་། །་དག་དང་བདེ་བ་འདོད་པ་དང་། ད་—།
