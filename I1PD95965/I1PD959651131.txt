བྷཱབསྟན་འགྱུར། །སྣ་ཚོགས། ༼བི་དུ་ཧི་མི་ཧི་རི་ཧི་རུ་ཧི་ལ་ཧི་ལུ་ཧི་ན་ཧི་བ་ཧི་ལས་ཧི་ལས་སོ་ཞེས་པས་ཨིཊ་བཀག་གོ། །ད་ཧ་ད་དང་པོའི་གྷའོ་ཞེས་པས་གྷ་ཉིད་དོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་དྷ་ཉིད་དོ། །དྷུཊ་རྣམས་ཀྱི་གསུམ་པ་ཉིད་དོ། །དགྷ༑ད་ཧི། དེབྷ། དུ་ཧོ། དོགྷཱ། མི་ཧི། ཧའི་རྙའོ་ཞེས་པས་ལྷ་ཉིད་དོ། །ཊ་སྡེ་ཉིད་དོ། །ཌྷ་ལ་ཌྷ་དབྱིའོ། །མེ་ཆུ། རི་ཧ། རེ་ཆ་། རུ་ཧ༑། རོ་ཞྭ། ལི་ཧ། ལེ་ཞྭ། ལུ་ཧ། ལོ་ཆ་། རི་ཧི་ལུ་ཧི་དག་མདོ་་་ལས་བྱུང་བའི་བྱིངས་དག་གོ། །ཁ་ཅིག་འདི་དག་ཀློག་པར་བྱེད་པ་ཉིད་དོ། །ན་ཧི་དྷའོ་ཞེས་པས་དྷཱ་ཉིད་དོ། །ཏ་ཐ་དག་གི་དྷ་ཉིད་དོ། །དྷུཊ་རྣམས་ཀྱི་་་་་གསུམ་པ་ཉིད་དོ། །ན་ཧ། ན་དྷཱ། མ་ཧ། ས་ཧི། བ་ཧ་དག་གི་ཨ་ཡིག་གི་ཨོཏའོ་ཞེས་པས་རྟ་དབྱི་བ་དང་ཨ་ཡིག་གི་ཨོ་ཉིད་ཀྱང་ངོ༌། །པོ་ཎཱ། གཞན་ལ་ཨིཊ་དང་བཅས་པའོ། །ཕཀྐ། ཕཀེ་ཏཱ། ་ཤུ་ཙ། ཤོ་ཙ་ཏཱ། རཱ་ཛི། རཱ་ཛི་ཏཱ། བྷཱ་ཛ། བྷཱ་ཛི་ཏཱ། ཨ་ཎ། ཨ་ཎི་ཏཱ། ར་ཎ། ར་ཎི་ཏཱ། ཀ་ཎ༑ཀ་ཎི་ཏཱ། དི་བུ། དེ་བི་ཏཱ། ཤི་བུ། སེ་བི་ཏཱ། བྲྀ་ཏུ། ་བཏཱི་ཏཱ་ཧྲི་དྷ། བར་དྷི་ཏཱ། གྲ་ཧ། ཨིཊའི་རིང་པོ་གྲ་ཧི་ལས་པ་རོག་མ་་་ཡིན་པ་ལའོ་ཞེས་པས་རིང་པོའོ། །གྲ་ཧཱི་ཏཱ། ཨིཊ་རྣམ་པར་བཤད་པ་བརྗོད་་་པར་བྱའོ། །སྤྲོ། སྒྲ་དང་ཉེ་བར་གདུང་བ་དག་ལའོ། །ཏུ། སྲོག་ཆགས་་་མངལ་ནས་བཙའ་བ་ལའོ། །ཞུང་། །སྲོག་ཆགས་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཨཱུཏ་་རྗེས་སུ་འབྲེལ་པ་རྣམས། གཱ་ཧུ། རྣམ་པར་དཀྲུགས་པ་ལའོ། །ཏུ་གླུ་དང་ཏཱཀླུ་ཕྲ་མོར་བྱེད་པ་ལའོ། །སཱ་ར་ཏི་སཱུ་ཏི་སཱུ་ཡ་ཏི་ཨུཏ་རྗེས་སུ་འབྲེལ་པ་ལས་་་ཞེས་པས་ཡང་ན་ཨིཊ་ཨཱ་ག་མའོ། །སཱཏྟཱ། སཱ་རི་ཏཱ། པྲ་སཱིཏཱ། པྲ་ས་བི་ཏཱ༑སོ་ཏཱ། ས་བི་ཏཱ། ཨཱ་བ་ཀཱ་ཞཱ། ཨ་བ་གཱ་ཧི་ཏཱ། ཏ་ཐཱ། ཏ་:ཀྱི་ཏཱ། ཏཱ།
