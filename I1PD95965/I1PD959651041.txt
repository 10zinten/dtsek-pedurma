བསྟན་འགྱུར། སྣ་ཚོགས། ཨན་འཇིག་གོ། ་། ཨུ་བ་དྷཱ་ཡང་བའི་ཡོན་ཏན་:ནོ། །དང་པོའི་གྷའོ་ཞེས་པས་ཧ་ཡིག་གི་གྷ་ཡིག་གོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷ་མ་ཡིན་པ་ལ:ས་ཏ་ཐ་དག་གི་དྷ་་་ཉིད་དོ། ་། དྷུཊ་རྣམས་ཀྱི་གསུམ་པ་ཉིད་དེ། །གྷའི་གའོ། །དོབ་བྷི། །དུབྷ། དུ་ཧནྟི་བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་དང་པོ་ས་དྷ་དག་ལ་བཞི་པ་ཉིད་དོ་ཞེས་པས་ད་ཡིག་གི་དྷ་ཡིག་གོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཎི་ཁཱ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་གྷའི་ཀའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །དྷཀྵ། དུགྷ༑། དུནྟ། དེརྨི། དུ་ཧཱ། །དུཧྨ། དུབྷེ། དུ་ཧཱ་ཏེ། དྷུ་ཥི། དུ་ཧཱ་ཐེ། དུབྷེ། དུ་ཧེ། དུ་ཧཱ་ཧེ། དུ་ཧཱ་ཧེ། དི་ཧ་ཉེ་བར་གསོག་པ་ལའོ། །དིབྷི༑དིགྷི། དི་ཧནྟི། དྷིཥི། དིངྒེ། དིངྒེ། །དིརྨི། དི་ཧཱ། དི་ཧྨ། དིལླི། དི་ཧཱ་ཏེ། དི་ཧནྟེ། དྷིཀྵེ། དི་ཧཱ་ཐེ། དིགྷེ། དི་ཧྭ་ཧེ། དིཧྥ་ཧེ། ཨུ་བ་དེབྷི། །པྲ་ཎི་སྔོན་དུའོ། །པྲ་ཎི་དེབྷེ། ལི་ཧ་རོ་མྱང་བ་་་་ལའོ༑:། ཨན་འཇིག་གོ། །གཅིག་ཚིག་ལ་ཨུ་པ་དྷཱ་ཡང་བའི་ཡོན་ཏན་ནོ། །ཧའི་དྷའོ་ཞེས་པས་ཧ་ཡིག་གི་དྷ་ཡིག་གོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་ཏ་ཐ་དག་གི་དྷ་ཉིད་དོ། །ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། ཞེས་པས་དྷའི་རྨི་ཉིད་དོ། །ཌྷའི་ལྷ་ལ་དབྱི་ཞིང་ཨུ་པ་དྷཱའི་རིང་པོའང་ངོ་ཞེས་པས་རྟ་དབྱིའོ། །ཡོན་ཏན་མེད་པའི་ཚིག་ལ་རྣོ་དབྱི་བ་ཇི་ལྟ་བར་འབྱུང་བ་ཨུ་པ་དྷའི་རིང་པོ་འང་ངོ༌། །ལེ་དྷི། ལི་དྷ། ལཱི་ཧནྟི། ཧའི་རྙའོ། །ཥ་་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་དྷའི་ཀ་ཉིད་དོ། ་། རྒྱུ་མཚན་ལས་ཁོ་ཉིད་དོ། ་། ལེཀཱ༑ལཱི་ཌྷ། ལཱི་ཌྷ། ལི་རྨི། ལི་ཧཱ། ལི་ཧྥ། ལཱི་བྷེ། ལི་ཧཱ་ཏེ། ལི་ཧ་ཏེ། །ལི་ཀེ། ་ལི་ཧཱ་ཐེ། ལི་ཞྭ། ལི་ཧེ། ལི་ཧྭ་ཧེ། ལི་ཧྥ་ཧོ། ཨུཎུ་ཉ། ་ཀུན་ཏུ་འགེབས་པ་ལའོ། །ཨན་འཇིག་གོ། །ཨུཎྞོ'ཏིའི་ཡོན་ཏན་ནོ། །SUA—།
