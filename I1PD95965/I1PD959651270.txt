ལྷའི་ལྕུག་མ། །སེང་གེ་སྐྲོད་དང༌། ཀླུའི་མེ་ཏོག། ཀིརྫི་ར་དང་། གེ་་་་སར་དམར། །བུམ་ཅན་ལྷ་ཡི་འདབ་མ་ཅན། །ཀེ་ཏ་ཀ་དང༌། སོགས་ལེ་སྒྲིབ༄༅༑། ཚད་ལྡན་འདབ་མ་མེ་ཏོག་རྩེ། །རྩ་བ་མང་དང༌། ཛམྦུ་འཛིན། །ས་སྲུང་སྐྱེས་དང༌། ཀླུ་ཡི་ཚོན། །ཤུན་པ་དྲི་ཞིམ་འབབ་ཆུ་གནས། །ཀིརྨི་ར་དང༌། ཆུ་བོ་སྐྱེས། །ས་ཡི་ཛམྦུ་ཀ་རྣམས་སོ། །ཆུ་ལ་རོལ་དང༌། གཞོལ་བ་དང་། །།མདོག་མཛེས་:མ་དུ་ལངྐ་ལ། །::ས་ག་སྐྱུར་རྩི་ཆེན་པོ་མིང་། སྐྱུར་རྩི་ཆེན་པོ་སོ་རྩེད་དང༌། །སྐྱུར་མོ། །སྐྱུར་མོ། ཛམྦྷིར། སྐྱུར་མོ། ཛམྦྷ་ལ། །ལེགས་འཛིན། རབ་གནས་མེ་ཏོག་དང༌། །སྙིང་རྗེ་དང་ནི་སོ་དགའ་བྱེད། །རྣམ་པར་མཛེས་ཞེས་བྱ་བའོ། །རྒྱལ་པོའི་འོ་མ། སྤྲེའུ་་ཡི་སྤྲང་། །ལས་ཀྱི་ཚོན་དང༌། ལས་ཀྱི་ཆ། །རབ་མཆོད་དང་ནི་མ་ལ་ཀཾ། །བཏུང་བ་ནས་ཀྱི་བཏུང་བ་དང་། །མ་ལ་ཀི་ཞེས་བྱ་བའོ། །ད་རཾ། རཀྟ་བ་་ད་རཱ། །མལ་ཀཾ་ཞེས་པ་ལ་ཡང་དེའོ། །ཆུས་གང་བ་དང༌། །ས་བོན་གང་། །དེ་ཉིད་སྐྱུར་བའི་གེ་སར་ཅན། །མཱ་རུ་ལུམ་བཱི། །རུལ་འབྲས་མ། །ལྷ་ལྡན་་མ་དང་དྲི་བཟང་ཅན། །ལ་ཀུ་ཙ་དང་ནི་ཀུ་ལ། །ཤིང་སྐྱུར་དང་ནི་ཤིང་བྲན་དང༌། །ཕྲ་བའི་ས་ན་ས་ཞེས་སོ། །མེ་ལྡན་མ་དང༌། །ཏིནྟི་དཱི། །ཙིཉྩ། ཏིནྟི་རི་ཡི་ཤིང༌། །ཙུ་ཀཱ། སྤྲིན་གྱི་རྭ་ཅན་དང་། །ཤིང་སྐྱུར་ཡལ་ག་སྐྱུར་བ་དང༌། །ས་སྐྱེས་སྐྱུར་དང་། ཤཱ་ཀཱམ། །ཏིནྟི་ལི་དང་། དམར་པོས་གང་། ཏིནྟི་ཊཱི་ཀ། སྟར་བུ་ཤིང་། །དེ་ནས་བྱ་རྒོད་སེན་མོ་དང༌། ཀོ་ནི། རྒྱ་ཤུག་ཀརྐམྤ། །བི་ཅི་ལ་འདོག་འབྲས་ཞིམ་དང་། །འབྲས་ལྡུམ་ཅན་ཞེས་བྱ་བའོ། །ཀོ་ལ། ཤིན་ཏུ་དཔའ་སྐྱེས་དང༌། །རྒྱ་ཤུག་གྷོཎྜ། །ས་སྟོབས་དང༌། །ས་ལ་ངེས་གནས་ཞེས་པའོ། །འབྲས་བུ་ནག་མིན། བི་གུ་འབྲས། །སྡེ་བཟང་་་་ཏིནྟི།
