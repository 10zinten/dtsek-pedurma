བསྟན་འགྱུར། སྣ་ཚོགས། ༼ཕི༽ཏྟཱཔྲཛ༑སྲིཛེ་དྲི་ཤ་དག་གི་དབྱངས་ལས་ཕ་རོལ་དྷུཊ་ལ་ཡོན་ཏན་དང་འཕེལ་བ་གནས་པ་ལ་ཨཱ་ཡིག་ཨཱ་ག་མའོ་ཞེས་པས་ཨ་ཡིག་ཨཱ་ག་མའོ། །རཾ་ཡིག་རའོ་་་་ཞེས་པས་ར་ཉིད་དོ། །ཥ་ཉིད་དོ། །ཊ་སྡེ་ཉིད་དོ། །པྲཉྫ། ནིཛ། ནེ། བི་ཛིར། བཀྟ། སཱཉྫ། སཱདྡྷ། ཨ་དི་ཧུ་དི་ཀླུ་:དི་སཱི་བྱ་*ཏི་བི་དྱ་ཏི་བིནྟ་ཏི་བ་ནིཏྟཙྪི་དི་བྷི་དི་ཧི་རི་ཤི་དིས་དི་པ་དི་ཨིནྟ་ཁི་རི་ལས་ད་ལས་སོ་ཞེས་པས་ཨིཊ་་་་བཀག་གོ། །ཨ་ད། ཨཏྟཱ། ཏུ་ད། ཏོཏྟཱ། ཆུ་དང༌། ནོཏྟཱ། ཀླུ་དིར། ཀློཏྟཱ༑སཱི་ད་སྭཱཏྟཱ། བི་ད། ཡོད་པ་ལའོ། །བིདྲི། རྙེད་པ་ལའོ། །བི་ད་རྣམ་པར་དཔྱོད་པ་ལའོ། །བིཏྟཱ། བི་ད། ཤེས་པ་ལའོ། །ཨིཊ་དང་བཅས་པའོ༑། བེ་དི་ཏཱ། ཚི་དིར། ཚེཏྟཱ། བྷི་དིར། བྷིཏྟཱ། ཧད། ཧཏྟཱ། ཤཛཱ༑དམན་པར་བྱེད་པ་ལའོ། །ཤཏྟཱ། ཥ་བློ། ་རྣམ་པར་རྒྱུ་བ་དང་འགྲོ་བ་དང་སྤྲོ་བ་རྣམས་ལའོ། །།སཏྟཱ། པ་ད། སཏྟཱ། སྐནྟིར། སྐནྡྷ། ཁིད། ཞན་པ་ལའོ། །ཁིད། ་ཡོངས་སུ་འཇོམས་པ་ལའོ། །ཁེཏྟཱ། རཱ་དྷི། རུ་དྷི། ཀྲུ་དྷི། ཀྲུ་དྷི་པནྟི། སྲུ་ཤུ་དྷི། སིདྷུ་ཏི། བུ་དྷུ་ཏི་བྱ་དྷི་སཱ་དྷི་་་ལས་དྷ་ལས་སོ་ཞེས་པས་ཨིཊ་བཀག་གོ། །རཱ་དྷ། རཱ་དྷཱ། རུ་དྷིརི། རོདྡཱ། ཀྲུ་དྷ། ཀྲོངྷ། ཀླུ་དྷ། ཀོདྷཱ། བནྡྷ། བནྡྷ། ཤུ་དྷ། ཤོདྷཱ། སི་དྷུ། སིདྷཱ༑བུ་དྷ། །ཁོང་དུ་ཆུད་པ་ལའོ། །བོདྷཱ། །བུ་དྷུ་ཏི་ལས་ཞེས་ངེས་པར་བསྟན་པ་ལས་བོ་དྷ་ཏི་ལས་ཨིཊ་འགྱུར་རོ། །བུ་དྷིར། བོ་དྷི་ཏཱ། ཡུ་དྷ། ཡོདྡྷཱ་བྱ་དྷཱ། བྱདྷཱ། སཱ་དྷ། སཱདྷཱ། ཧ་ནི་མ་སྐྱེ་ཏི་ལས་ན་ལས་སོ་ཞེས་པས་་ཨིཊ་བཀག་གོ། །ཧན། ཧནྟུ། མ་ན་ཤེས་པ་ལའོ། །མནྟཱ། །གཞན་ལ། །མ་ནི་ཏཱ། ན་ལས་ཞེས་པའི་ཁྱད་པར་ལས་ན་མཐའ་མ་ཡིན་པ་ལས་ཨིཊ་དགག་པར་མི་འགྱུར་རོ། །ཨཱི་བ་ཏ་པི་ཏི་པི་ཥ་པི་བ་པི་ཤ་པི་ཙྪ་པི་ཀྱི་བི་ལི་བི་སྲི་བི་ལས་བ་་TOUSདྷཱ།
