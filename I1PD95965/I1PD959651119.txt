བསྟན་འགྱུར། །སྣ་ཚོགས། བ་ཎི་ཐ། སྱ་མ། ས་སྩ་མ། ཀྱེ་མ་ཏཿམྱེ་མུས་སྐྱ་མ་ཏཿས་སྐྱ་མུསྐྱེ་མི་ཐ། ས་སྐྱ་མི་ཐ། མང་པོ་རྣམས་ཀྱིས་ངེས་པར་ཨ་ནུ་ཥཇ་རྣམས་ཀྱིས་་་ལྷན་ཅིག་སྤྱོད་པའི་ཕྱིར་ཨ་ནུ་ཥང་ཅན་རྣམས་ཀྱི་མི་འགྱུར་རོ། །ཤ་ཐཾ་ཐི་ཐ། ཛ་གྲནྠ་ཐ། ད་དམྷི་ཐ། དྷཱ་ན། ་ད་དྷ་ན། ད་དྷཱ་ན་ཏུད་དྷ་ནུཤ་སུ་འཚེ་བ་ལའོ། །དད་བསྟན་པ་ལའོ། །།ཁུ་བ་མ། སྐྱུག་པ་ལའོ། །ཤ་ཤ་དད་པ་དང་པོ་དང་ཡོན་ཏན་ཅན་རྣམས་ཀྱི་མ་ཡིན་ན་ཞེས་པས་ཨེ་ཉིད་དང་ཨ་བྷཱ་ས་དབྱི་བར་ཡང་མི་འགྱུར་རོ། །བི་ཤ་ཤཱ་ས། བི་ཤ་ཤ་ས་ཏུབི་ཤ་ཤ་སུད་ད་དོ། །ད་དེ་དཱ་ཏེ། ད་ད་དི་རེ། ་བ་བཱ་མ། བ་བ་མ་ཏུ༼བ་བ་མུབ་བ་མི་ཐ། ་བ་ན། {བ་བཱ་ན། །བ་བཱ་ན་ཏུབ་བ་ནུབ་བ་ནི་ཐ། ཨཱ་ཡིག་ཨུ་པ་དྷ་རྣམས་ཊུ་ཡཾ་ཝཾ། ཡ་ཡཱ་ཙ། ཡ་ཡཱ་ཙེ། གཉི་གའི་ཚིག་དག་གོ། །ཁཱ་ཧཱུཾ། ?ཤུ་ཙྪཱབྷཱ་གྷི། ཨི་ཡིག་ཨུ་པ་དྷཱ་རྣམས། མི་ཧ། མི་མེ་ཧ། མི་མི་ཧ་ཏུ%མི་མི་ཏུ་མི་མེ་ཧི་ཐ། ཏི་ཙ་བི་ཏོ་ཙ། ཨི་ཡིག་ཨུ་པ་དྷཱ་རྣམས། ཀྲྀ་ཏཱི། གནས་པ་ལའོ། །ཙི་ཌི་ཀྲི་ཌི། ཙི་ཀཱི་ཌི་ཏུཙི་ཀཱི་ཌཱུཙི་ཀཱི་ཌི་ཐ། ཨུ་ཡིག་ཨུ་པ་དྷཱ་རྣམས། རུ་དྷིར། རུ་རོ་དྷ། རུ་རུ་དྷ་ཏུརུ་རུ་དྷུ༑:མུཙྪི། མུ་མོ་ཙ། མུ་མུ་ཙ་ཏུ་མུ་མུ་ཙུམུ་མོ་ཙི་ཐ། རུ་ཙི། རུ་རུ་ཙི། རུ་རུ་ཙཱ་ཏེ། རུ་རུ་ཙི་རེ། །ཨུ་ཡིག་ཨུ་པ་དྷཱ་རྣམས། །ཀཱུ་ཛ། །ཙུ་ཀཱུ་ཛ། ཙུ་ཀཱུ་ཛ་ཏུཙུ་ཀཱུ་ཛུཙུ་ཀཱུ་ཙི་ཐ། །རཾ་ཡིག་ཨུ་པ་དྷཱ་རྣམས། བྲྀ་ཏུ་འཇུག་པ་ལའོ། །བ་བྲི་ཏེ། །བ་བྲྀ་ཏ་ཏེ། བ་ཛྲ་ཏི་རེ། བཻ་དུ། བ་:བྲི་དྷི། བ་ཛྲ་དྷཱ་ཏ། ་བ་:ཛྲ་དྷི་ྋརེ། །ཤྲཱི་དྷུ། ་ཤ་ཤི་ེདྷེ། ་ཤིཤ་ཝཾ་དྷེ་རེ། ཀྲྀ་པཱུ། ར་ཐོས་པ་ལས་ལ་ཐོས་སོ། །ཙ་ཀླུ་པེ། ཙ་ཀྱི་པཱ་ཏེ། ཙ་ཀླུ་བི་རོ། ཊུ་པེ་པྲི། བི་བེ་པྲ། བི་བེ་པེ། བི་བི་པཱི་ཏེ། ཨཱ་བི་བེ་པི་རེ། །
