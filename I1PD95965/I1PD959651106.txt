སsབྷུ་གྷཱ་ཏམ། ཨ་གྷོ་ཥཱནྟ། ཨ་ཧྥུ་ནྣཱཨ་བྷུཀྵ་ཐ?ཨ་བྷུ་ཐཱམ། ཨ་་གླུ་རྙམ། ཨ་གླུ་ཆ་མ། ཨ་གྷཀྵ། ཨ་བྷུཀྵ་བ་ཧི། ཨ་གྷ་མ་ཧི། ད་ཧི་་ལས་ནཱ་མི་ཨུ་པ་དྷཱ་མ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར་སཎ་མི་འགྱུར་རོ། །སིཙའོ། །ཨ་་་དྷཱཀྐཱ་ཏ། ཨ་དཱགྷ་མི། ཨ་དྷཱཀྵ། :ན་ཧ་ནཱ་ཧའི་དྷའོ་ཞེས་པས་དྷ་ཉིད་དོ། །སྒྲ་མེད་པ་རྣམས་ལའིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པ་དང་པོ་ཉིད་དོ། །ཨ་ནཱདྷཱིཏ། :ཨ་ནུ་དྷཱམ། ཨ་ནཱདྷུཨ་སུ། ཨ་སུ་བ་ཙི། བྱཱ་ཏི་ལི་སི་ཙ་ཧྭ་ལས་ཨཎའོ་ཞེས་པས་ཨཎའི་རྐྱེན་ནོ། །ཨ་སྱ་ཏའི་མཐའ་ཐའོ་ཞེས་པས་ཐ་ཡིག་ཨཱ་ག་མའོ། །ཨཱམྦཏ། ཏེ་ཨཱ་སྦཏྭཾ། ཨཱ་སྦྲན། ཉེར་བསྒྱུར་ལས་ཨ་སྱ་ཏི་དང་ཨུ་ཧི་དག་ལ་ཡང་ནའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ས་མཱ་སྦྲ་ཏཱི། ས་མཱ་སྦི་ཏཱམ༑ས་མཱཙྪནྟ། བྲུའི་བ་ཙིའོ། །པ་ཙ་འཆད་པ་ལ་ཡང་ནའོ། །ཨ་ཎའོ། །ཨཎ་ལ་བ་ཙིའི་ཨུ་པ་དྷཱའི་ཨོཏའོ་ཞེས་པས་ཨི་ཉིད་དོ། །ཨ་པ་ཙ་ཏེ། ལི་པ། ཨ་ལི་པ་ཏ། ::སི་ཙ། ཨ་སི་ཙ་ད། ལི་པ་སོགས་རྣམས་ཀྱི་ཨཱཏྨ་ནེའི་ཚིག་ལ་་་ཡང་ནའོ་ཞེས་པས་ཨཎའོ། །བྱ་ལིཔ། བྱ་ལི་པ་ཏ། ཨ་སི་ཀྟ། ཨ་སི་ཙ་ཏ། པུ་ཥ་ལ་སོགས་པའི་མདོ་ཡིས་ཚིག་ལ་ཨཎའོ། ?། པུ་ཥ་སོགས་པ་ནི། གྲོ་དྷུ་མངོན་པར་འདོད་པ་ལའོ། །ཇི་སྲིད་བར་རོ། །པུ་ཥ། ཨ་པུ་ཥཏ། ཤུ་ཥ། ཨ་ཤུ་ཥ་ཏ། དུ་ཥ། ཨ་དུ་ཥཏེ། རྨི་ཥ། དཔེར་བརྗོད་པ་རྣམས་སོ། །སཱི་ད། ཨ་སྲི་དཏ། གླུ་དྷ། ཨཀྵ་དྷཏ། ཤུ་དྷ། ཨ་ཤུ་དྷཏ། ཎི་དྷ། ཨ་སི་དྷཏ། ར་དྷ། ཨ་ར་དྷཏ། ཏྲི་བ་ཏྲི་པ་དག་གིས་ེབསྟན་ཏོ། །ཨ་མུ་ཧཏ། ཨ་དྲུ་ཧཏ། ཨ་སྣུ་ཧཏི། ཨ་སྣེ་ཧཏ། ཎ་ཤུ། །ན་ཤུ། ན་ཤ་དང་ཤ་མི་དག་གི་ཨཎ་ལ་ནི་ཨེ་ཉིད་བརྗོད་པར་བྱའོ། །།ཨ་ནེ་ཤཱཏེ། །ཨ་ཤེ་མཏ༑པཨད་མཏ། །ཨཏི་མཏ། ཤ་མུ། ཨ་ཐ་མཏ། ཨ་བྷ་མཏ། ཨཀྵ་ཀྱི་།
