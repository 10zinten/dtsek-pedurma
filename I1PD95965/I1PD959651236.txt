མཱ་ཏ་ཞེས་བྱ། །སིལ་སྙན་དང་ནི་སིལ་ཁྲོལ་ལོ། །མཱ་ཧཱ་ཏི་དང༌། ཨ་མཱ་ན། །པ་ཊ་ཧ་ལ་སོགས་པའོ། །མ་ཧཱ་ན་དང་། མརྡ་ཀ། རྔ་ཟླུམ། རྫ་རྔ་ཨ་ན་ཀ། ཨ་རུ་ར་འདྲའི་རྔ་ཨངྐ་། །ནས་ཀྱི་དབུས་འདྲ་ཨུཌྷ་ཀ། ཨ་ལིངྒ་ནི་བ་ལང་གི། །མཇུག་མའི་དབྱིབས་འདྲ་འདི་དག་གི། །བརྨ་མུ་ར་ཛ་ཡིན་ནོ། །འཁར་རྔ། ཛྷདྷི་པ་དག་མཚུངས། ་། ཉཀྐ། ཆ་ལང་མརྡ་ལ། །ཀཱ་ཧ་ལ་་་དང་སྒྲ་ཆེན་དང་། །ཀཱ་ཧའི་སྒྲ་དང་། །སྦུབས་ཅན་དང་། །གཏུམ་པོའི་ཀུ་ཅོ་ཞེས་པ་རྣམས། །སིཉྫ་ལ་དང༌། ཀཱ་ཧ་ལ། །དེ་ནས་རྙཀྐ། གུཉྫ་དང༌། ཌྷཀྐ་རྣམ་རྒྱལ་མརྡ་ལ། །རྒྱང་གྲགས་སྒྲ་ཅན། རྒྱང་གྲགས་ཀྱི། །ཆ་ལང་། སྙན་པའི་པ་ཊ་ཧ། །གོ་བྱེད་པ་ཊ་ཧ་དང་ནི། །ལནྡཱ་པ་ཊ་ཧ་མིང་ཅན། །ཨཱ་ན་ཀ་དང་། བ་ཊ་ཧ། །གུཉྫ་པ་ཊ་ཧི་ཀཱ་ཡ། །ཨ་ན་ཀ་དང༌། མྲ་དི་ཀ། འཇིགས་བྱེད་རྔ་དང་རྔ་བོ་ཆེ། །རོལ་མོའི་རབ་དབྱེ་འདི་རྣམས་ཏེ། །མ་ཧ་ན་་་དང་། །ཌ་མ་རུ། །།མུརྨུ། ཌིཎྜི་མ། རྫ་རྔ་། །གླུ་རྔ་། འཁར་རྔ་། ཀ་ར་རྔུ། །ན་པ། །རངྒཱི་ཏེངྒ་རི། །གཡུལ་དུ་སིལ་སྙན་སྒྲ་དང་ནི། །རྔ་ཆུང་གཉིས་ཀ་ཌིཎྜི་མ། །གཡུལ་འཐབ་ཉིན་མོའི་སིལ་སྙན་ནོ། །མི་བདག་རྣམས་་་ཀྱིས་བཟའ་བའི་མཐར། །བཟའ་བའི་སིལ་སྙན་གང་ཡིན་དེ། །འཇིག་བྱེད་་་་ཟས་ཀྱི་སིལ་སྙན་ཏེ། །རླབས་ཀྱི་འགྲོས་ཅན་ཡིན་ཞེས་བྱ། །བྱིས་པའི་སིལ་་སྙན་གང་ཡིན་དེ། །།རབ་དགའི་མརྡ་ལ་ཞེས་སོ། །འཚོ་བ་བྱུང་བའི་གོ་བྱེད་པའི༑། །སིལ་སྙན་གཎྜའི་སིལ་སྙན་བརྗོད། །ཐིག་པའི་སིལ་སྙན་དེ་དག་དང་ནི། །ཤིང་རྟ་གླང་པོ་གཉིས་ཀ་ཡི། །ཌིཎྜི་ཀཱ་དང་། ཨུངྒ་ར། ་། གྲོང་ཁྱེར་དབྱངས་དང༌། འཁོར་བའི་རླབས། །མི་བཟད་པ་ཡི་སྒྲ་རྣམས་སོ། །ཆུ་ཚོད་བུམ་པ། །དབྱུ་གུའི་རྔ་། །མེལ་ཚེ་འཛིན་པ། ་མེལ་ཚེའི་སྒྲ། །ཐུན་གྱི་དབྱངས་དང༌། །
