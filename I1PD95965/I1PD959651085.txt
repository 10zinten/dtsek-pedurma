བསྟན་འགྱུར། །སྣ་ཚོགས། ༼བིཧ་ཏེ། ཧ་ཏ། ཧ་ནཱ་ནི། ་གཉིས་ཉིད་དང་མང་པོ་ཉིད་དག་ལའང་པ་རསྨིའོ་ཞེས་པ་འདི་ཡིས་པ་རསྨིའི་ཚིག་ལ། པཉྩ་མཱིའི་ཕྱི་མའི་ཡོན་ཏན་ཉིད་ཀྱི་ཕྱིར་ཨུ་་པ་དྷཱ་དབྱི་བར་མི་འགྱུར་རོ། །ཧ་ནཱ་བ། །ཧ་ན་མ། ལ་ཨཱཏྨ་ནེའི་ཚིག་ལ། ཨཱ་ཧ་ཏཱམ། ཨནྣཏཱམ། །ཨཱ་གྷཱནྟཱམ། ་ཨཱ་ཧཱ་སཱ། །ཝ་:ཨཱ་བྷཱ་ཐཱམ། ཨཱ་ཧ་་དྷཱ་མ། ་ཐམས་ཅད་ཀྱི་བྱིངས་ཐམས་ཅད་པ་ཨཱཏྨ་ནེ་ལ་པཉྩ་མཱིའི་ཕྱི་མ་མིན་པ་ལའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་ལ་པཉྩ་མཱིའོ། །།ཕྱི་མའི་ཡོན་ཏན་མེད་པ་ཉིད་མ་་་་ཡིན་པའི་ཕྱིར་ཨུ་པ་དྷཱའི་དབྱི་བར་མི་འགྱུར་རོ། །ཨཱ་ཧ་ནི། ཨཱ་ཧ་ནཱ་བ་ཧེ། ཨ་ཧ་ནཱའི་མ་ཧེ། ཌ་དྭེཥ། དྭེཉྩ། ཏཱ་ཏཎ། ་དྭེཥཱཏ། དཱིཏྟཱ་མ། དཱིཥནུ། ཧི་ལའང་ངོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །།དྷཱའི་དངོས་པོའོ། །དྷཊ་་་རྣམས་ཀྱི་གསུམ་པ་ཉིད་དོ། :། ཊ་སྡེ་ཉིད་ཀྱང་ངོ་། :། དྲི་ཎི། །ཏཱ་ཏཎ། དྲིཊྚ་ཏ། དྭིཋ་མ། །དྭེ་ཥཱ་ཎི། །དྭེ་ཥ་བ། དྭེ་ཥཱ་མ། དྭེཀྵ་པོ། དྭེ་ཥཱ་ཏམ༑དྭི་ཥཱ་ཏམ། དཱི་ཀྵ། དྭི་ཥཱ་ཐམི། དཱིབྷམ། དྭེ་ཥ། དྭེ་ཥཱ་བ་ཧི༑དེ་ཥཱ་མ་ཧི། དུ་ཧི། དོགླུ། དུབྷཱམ། དུ་ཧ་ནུ། དུབྷི། ཏཱ་ཏི་ཎ། དུབྷྱཱ་ཏ། དུགྷ་མ། དུབྷྱཱ། དུབྷྱཱ་མ། དུ་ཧ་ཏཱ་མ། དུ་ཧི་ཏཱ་མ། དུ་ཀཱ། ལི་ཧ། ལི་ཟླུ། ཏཱ་ཏ་ཎ། །ཞྭ་དབྱི་ཞིང་ཨུ་པ་དྷའི་རིང་པོ་ཉིད་ཀྱང་ངོ༌། །།ལཱི་ཎཱ་ཏི། ལི་ཆ་མ། ལཱི་ཧནྟུ། ལི་ཌྷ། །་ཏཱ་ཏཎི། ལཱི་ཌྷ། ལེ་ཧཱི་ན། ལེ་ཧཱ་བ། ལི་ཧཱ་མ། ལི་ཌྷ་མ། །ལི་ཏཱིཏཱམ། ་ལི་ཧཱ་ཏམ། ལིཀྵ༑དི་ཧ། དེསྒྱུ། ཏཱ་ཏ་ནོ། :དིབྷཱ་ཏ། དིབྷཱ་མ། དི་ཧནྟུ། དིཀྐི༑དེ་ཏཱ་ཧ་ཎི། དིགྷ་ཏ། དིགྷམ། ་་དི་གྷ། དིངྒཱ་མ། དི་ཧཱ་ཏཱ་མ། དི་ཧ་ཏཱ་མ། དིཀྵི། ཙ་ཀླུང༌། ་ཨཱ་ཙཋམ། ་ཨཱ་ཙཀཱ་ཏཱམ། ་ཨཱ་ཙཀ་ཊཾ༑ཨཱིཏྟ་མ། ཨཱིཏྟཱ་མ། ཨཱི་ཎཱ་ཏ་མ། ང་ཨཱི་ཌཱ་ཏཱ་མ། ཉི་ཨཱི་ཌ་ཥཱ། ji།
