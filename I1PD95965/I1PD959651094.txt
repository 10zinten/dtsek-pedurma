བྲུ་ལས་ཚིག་དང་པོའི་ཨཱིཊ་འོ་ཞེས་པས་ཨཱིཊའོ། །ཨ་བྲ་བཱིཏ། ཨ་བྷཱུ་ཏཱམ། ཨ་བྲུ་བན། །ཨ་བུ་ཏ། ཨ་བྷུ་པཱ་ཏཱམ། ཨ་བྲུ་པ་ཏ། ཏུ། །།ཨ་ཛུ་ཧོཏ། ཨ་ཛུ་ཧཱ་ཏཱམ། །ས་ཨ་ཛུ་ཧ་བུཉི་བྷི། ཨ་བི་བྷེཏ། ཨཱ་༔བིབྷེ་ཏྭཾ། ཨ་བྷི་བྷི་ཏཱམ། ་ན་ཨ་བི་བྷི་ཡུཨ་བྱཱ་་སྟ་རྣམས་ཀྱི་ཨུས་ལའོ་ཞེས་པས་ཡོན་ཏན་ནོ༑། ཏི་། ཧི། །ཞི་ཨ་ཛི་ཧྲེཏ། །ཨ་ཛི་ཧཱི་ཏཱམ། ་ཨ་ཛི་ཧྲ་ཡུསྲི། ཨ་པི་བཨ་བི་ཀྲྀ་ཏཱམ། ་ཨ་པི་བ་རུཨོ་ཧཱཀ་ན། ཨ་ཛི་་ཧཱ་ཏ། གསལ་བྱེད་དང་པོའི་བྱིངས་ཐམས་ཅད་པ་ལ་ཨི་ཉིད་ཡང་ན་བར་བརྗོད་པར་བྱའོ། །ཨ་ཛ་ཧི་ཏཱམ། ་། ཨ་ཛ་ཧུ་རྀ། ཨེ་ཡཨེ་ཡིར་ཏཱམ། ཨེ་ཡ་རུཎི་ཛིར། ཨ་ནེ་ནིཀ་། ཨཱ་ནེ་ནི་ཁཱམ། ་བཨ་ནེ་ནི་ཛུཨ་ནེ་ནིཀྟ། ཨ་ནེ་ནི་ཛཱ་ཏཱམ། ཞེཨ་ནེ་ནི་ཛ་ཏ། །ཆབི་ཛི་ར། ::ཨ་བེ་བེཀ་། ཨ་བེ་་བིཀྐཱམ། ་་་ཨ་བེ་བི་ཛུ། ཨ་བེ་བིཏྟ། ཨ་བ་བི་ཛཱ་ཏཱམ། ་ཨ་བེ་བི་ཛ་ཏ། བི་བྷཱི། ཨ་བེ་བེཊ། ཨ་བེ་བེཏྟཱ་མ། ཨ་བེ་བི་ཥཱཨ་བ་བིཊྚ། ཨ་བེ་བི་ཥཱ་ཏྭཾ། ཨ་བེ་བི་ཥ་ཏ། ཏུ་དཱཉ། །་ཨ་ད་དཱཏ། །ཨ་དཏྟཱམ། ཨ་ད་དུཨ་དཏྟ། ཆ་ཨ་ད་དཱ་ཏཱམ། །ཨ་ད་དཏ། དྷཱ། ཨ་དཱ་དྷཱཏ། དེ་དྷཱ་ཏིའི་ཏི་ཐ་དག་ལའང་ངོ་ཞེས་པ་ལ་སོགས་པས་བཞི་པ་ཉིད་དེ་ད་ཡིག་གི་དྷ་ཡིག་གོ། །།ཨ་དྷཱ་ཏམ། ཨ་ད་དྷུ་ཨ་དྷཏྟ། ཨ་ད་དྷ། ་ཏཱམ། ཨ་ད་དྷ་ཏ། ཊུ་ཝྃཉ། ཨ་བི་བྷ་ཨ་བི་བྷཾ་ཏཱམ། ་ཨ་བི་བྷ་རུ། ཨ་བི་བྷཾཏ། ཨ་བི་བྷ་ཏཱམ། ::ཨ་བྷི་ེབཱ་ཏ། མཱང་། ཨ་མི་མཱིཏི། ཨ་མི་མཱ་ཏཱཾ། ཨ་མི་མ་ཏ། ཨོ་ཧཱང་། ཨ་ཛི་ཧཱི་ཏ། ཨ་ཛི་ཧཱི་ཏཱམ། ཨ་ཛི་ཧ་ཏ། དེ་བཞིན་དུ་ཨད་ལ་སོགས་པའོ། །དི་བུ། །་ཨཱ་དཱི་བྱཏ། ཨ་དཱི་བྱ་ཏམ། ཨ་དཱི་བྱན། ཨ་དཱི་བྱཨ་དཱི་བྱ་ཏམ། ་ཨ་དཱི་བྱ་ཏ། ཨ་དཱི་བྱ་མ། ་ཨ་དཱི་བྱཱ་བ། །ཨ་དྷི་བྱཱ་མཥཱ་ཉ་ཨ་joDN།
