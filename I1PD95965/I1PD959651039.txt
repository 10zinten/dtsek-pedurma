བསྟན་འགྱུར། །སྣ་ཚོགས། ༼ཁ༽འགེབས་པ་ལའོ། །བ་སྟེ། བ་སཱ་ཏེ། བ་ས་ཏེ། བསྶེ། །བ་སཱ་ཐེ། །བ་ས་དེ། བ་སེ། བ་སཱ་ཧེ། བསྨ་ཧེ། ཀ་སི། འགྲོ་བ་དང་རྣོ་བར་བྱེད་པ་དག་ལའོ། །ཨི་ཊ་རྗེས་སུ་འབྲེལ་བ་ཉིད་ལས་ན་ཨཱ་ག་མའོ། །ཀཾ་སྟེ། ཀཾ་སཱ་ཏེ། །ཀཾ་ས་ཏེ། ཀཾསྶེ། ཀནཱ་ཏེ། སྡེ་པ་ལ་སྡེ་པའི་མཐའོ། །རྗེས་སུ་ང་རོའི་ན་ཡིག་གོ། །ཀནྡྷེ། ་ཀཾ་སེ། ཀཾ་སྭཱ་ཧེ། ཀཾ་སྨ་ཧེ། ཎི་སི། འོ་བྱེད་པ་ལའོ། །ཎའི་ནའོ་ཞེས་པས་ཎ་ཡིག་གི་ན་ཡིག་གོ། །ནཱི''སྟེ། ནཱི་སཱ་ཏེ། ནི་ས་ཏེ། ནིསྶེ། ནི་ས་ཐེ། །ནིདྷཱ། ནི་ས། ནི་སྭཱ་ཧེ། ནི་སྨྲ་ཧེ། པྲ་སྔོན་དུའོ། །དེ་ལ་ནི་སི་ནི་ཀྱི་ནནྡ་རྣམས་ཀྱི་ཡང་ནའོ་ཞེས་པས་ཨར་ཎ་ཎ་་་ཉིད་དོ། །པྲ་ནི་སྟེ། །བྲི་ནི་སྟེ། ཎི་ཛི་དག་པ་ལའོ། །ཨན་འཇིག་གོ། །ཙ་སྡེའི་འདྲ་བའི་ཡི་གེ་མིན་པ་ལ་ཀིའོ་ཞེས་པས་ཀ་ཉིད་དོ། །རྗེས་སུ་ང་རོ་དང་སྡེ་པའི་མཐའ་དག་གོ། །ནིངྒེ། ནིཥྛ་ཏེ། ནི་ཏེ། ཀི་ཉིད་དོ། །རྒྱུ་་་མཚན་ལས་ཥ་ཉིད་དོ། །།ནིངྒེ། ནིཉྫ་ྋཐེ། ནིདྡྷེ། ནིཊེ། ནིཛཱ་ཏེ། ནིཥྛ་ཧེ། ཤི་ཛི་མི་གསལ་བའི་སྒྲ་ལའོ། །ཤིནྟེ། ཤིཥྛ་ཏེ། ཤིཀྵ་ཏེ། པྲ་ཛཱི། འདོར་བ་ལའོ། །པྲབྷེ། པྲ་ཛཱ་ཏེ། པྲ་ཛ་ཏེ། པྲསྐེ། པྲ་ཛཱ་ཐ། པྲབྷེ། པྲ་ཛེ། པྲ་ཛཱ་ཧེ། པྲ་ཙྪ་ཧེ། པྲ་ཙཱ། ཚོགས་པ་ལའོ། །པྲྀཏྟེ། སཾ་པྲྀཏྟེ། ཁྱུང་སྲོག་ཆགས་མངལ་ནས་བཙའ་བ་ལའོ། །ཨན་འཇིག་གོ། །ཐམས་ཅད་ཀྱི་བྱིངས་ཐམས་ཅད་པ་ཨཱཏྨ་ནེ་ལས་པཉྩ་མིའི་ཕྱི་མ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཡོན་་་་ཏན་མེད་པ་ཉིད་དོ། །སཱུ་ཏེ། :ཨི་ཡིག་དང་ཨུ་ཡིག་གི་མཐའི་དབྱངས་དང་པོ་ལ་ཞེས་པས་ཨུ་བ་རུ་བསྒྱུར་རོ། །སུ་བཱ་ཏེ། སཱུ་བ་ཏེ། སཱུ་ཏེ། སཱུ་བཱ་ཐེ། སཱུ་དྷེ། སུ་བེ། སཱུ་བ་ཧེ། སུ་མ་ཧེ། པྲ་སཱུ་ཏེ། གླིང་། ཉལ་བ་ལའོ། །ཐམས་ཅད་ཀྱི་བྱིངས་ཐམས་ཅད་པ་ཨཱཏྨ་ནེ་ལ་པཉྩ་མའི་ཕྱི་མ་མ་ཡིན་པ་ལ་ཡོན་་་།
