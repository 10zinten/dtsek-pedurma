།སུ་པི་ཁ་རཾ་ཙ་ར་ཛྷ་ར་ཞེས་པས་སད་ལ་་ཏ་ཉིད་དོ། །ཏ་སྱ་ཙ་ཡཤ་རི་ཏཱི་ཏཱི་ཡ་ཞེས་པས་ཐ་ཡིག་གོ། །མ་རུ་ཏེ། །མ་རུ་ཏ་ད། །མ་རུ་ཏོ། །།མ་རུ་ཏ་མ་རུ་ཏཾ། མ་རུ་ཏོ། །མ་རུ་ཏམ་རུ་ཏཱ། མ་རུདྡྷཾམ་རུཏྟི། མ་རུ་ཏེ། མ་རུ་ཏཏྟཾ། །མ་རུདྡྷ་མ་རུ་ཏ། མ་རུདྡྷཾ། །མ་རུ་དྡྷམ་རུ་ཏ། མ་རུ་ཏོམ་རུ་ཏཱཾ། མ་རུ་ཏི། མ་རུ་ཏོ། མ་རུཏུ། མ་རུ་བྷྱ། བོད་པ་ནི་དང་པོ་བཞིན་ནོ་། :། ཀྲ་བྱཱ་ཏ་ཀྱི་སྒྲ་ལས་སུ་ཞེས་པས་སུ་དབྱིའོ། །ཐཱ་བི་རཱ་མ་ཞེས་པས་ད་ཡིག་ལ་གོ། །སྟ་ཡིག། སུ་བི་ཁ་རི་ཙ་ར་ཛྷ་ལ་ཨི་ཏི་ཏྲེ་ཀྲི་ཏེ་ཙ་ཡ༔ཤ་རི་དཱི་ཏཱི་ཡ་ཞེས་པས་ཐ་ཡིག་གོ:༔༈༄༅། །ཀྲ་བྷཱུ་ཏ་ཀྲ་བྱཱ་ད། ཀྲ་བྱཱ་དོ། །ཀ་བྱཱ་ད། ཀྲ་:བྱཱ་དཾ། ཀྲ་བྱཱ་དོ། ཀྲ་བཻ%ཀྲ་བྱཱ་དཱ། ཀྲ་བྱཱ་དྡྷཾ། །ཀྲ་བྱཱ་དྷི། ཀྲ་བྱཱ་དེ། ཀྲ་བྷཱུདྡྷཾ། ཀྲ་ཤུ་དྡྷ། ཀྲ་བྱཱ་དཀྲ་བྱཱ་དྡྷཾ། ཀྲི་བྱཱ་དྷཱཀྲ་བྱུ་དཀྲ་བྱཱ་དོཀྲ་བྱཱ་དཾ། ་ཀྲ་་བྱཱ་དཱི། ཀྲ་བ་དོ་ཀྲ་བྷུཉྩ། ཀྲ་བྷཱཤྩ། མི་ག་བི་དྷའི་སྒྲ་ལའང་མ་རུང་་་གི་སྒྲ་ལྟར་བྱའོ། །མི་ག་བི་ཏ། །མི་ག་བི་ད། མྲྀ་ག་བི་དྷོ། །མི་ག་བི་དྷམི་ག་བི་དྷཾ། མི་ག་བིདྷོ། །མི་ག་བི་དྷམི་ག་བི་དྷཱདུ་མི་ག་བི་དྡྷཾ། མི་ག་བི་དྷི་མི་ག་བི་དྷེ། མི་ག་བིདྡྷཾ། མི་ག་བི་དྡྷམི་ག་བི་དྷ་རྫི་ག་་བི་དྷ། »མི་ག་བི་རྒྱུ། །མི་ག་བི་དྡྷམི་ག་བྷི་དྷམི་ག་བི་ཛྷམི་ག་་བི་:དྲོ༑མི་ག་བི་དྷྭཾ་སྲོག་བི་གྷི་མི་ག་བི་དྷོ་མི་ག་བིཏྟུ། །མི་ག་བིབྷུ། བྲ་ཧྥ་ན་གྱི་སྒྲ་ལས། །སཱ་བ་སཱ་པ་སཾ་བྷུ་དྷ་བི་ཞེས་པས་རིང་པོའོ། །:། སུ་ཞེས་པས་སུ་དབྱིའོ། །སུ་པ་:པྲ་ཀྲི་ཏི་རྣོ་ལོ་པའོསྟེ། །སོ་ར་མྱ་ཙེཏྲ་སུ་ཏྲི་ཤི་སུ་ཌི་ཞེས་པས་རིང་པོའོ། །ར་ཥ་མོ་ཎ་ཨེ་ཀ་པ་དང་ཞེས་པས་ཎ་ཉིད་དོ། །ན་སཾ་ཡོ་ག་—།
