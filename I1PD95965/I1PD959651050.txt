མྱ་ཏི། བྲི་རྒྱུ། རྩོད་པ་ལའོ། །པྲྀ་་ཏི། ཌ་ལ་དག་གི་གཅིག་ཉིད་དོ་་་ཞེས་པ་ཡང་ན་ལ་ཉིད་དོ། །བྲི་མྱ་ཏི། ཨི་ཏུ་འགྲོ་བ་ལའོ། །པྲ་སྔོན་དུའོ། །སྲི་ཥྱ་ཏི། ཨ་ནཱི་ཥྱ་ཏི། ཥ་ཧ། ཀྵ་ཧ། །ནུས་པ་ལའོ། །ས་ཏྱཱ་ཏེ། སུ་རྒྱ་ཏེ༑ཛཱ་ཥ། ཛཾ་ཥ། ལང་ཚོ་ཉམས་པ་ལའོ། །རྀཏ་མཐའི་ཡོན་ཏན་མེད་པ་ལ་ཞེས་པས་ཨི་རའོ། །ར་པ་དག་གི་ཨུ་པ་དྷཱ་རིང་པོའོ། །ཛཱིརྱ་ཏི། །ཛཱིདྨ་ཏི། ཤོ་ཕྲ་མོར་བྱེད་པ་དག་ལའོ། །ཡན་ལ་ཨོ་ཡིག་གི་ཞེས་པས་ཨོ་ཡིག་དབྱིའོ། །བྱཱ་ཏི༑ནི་བྱ་ཏི། ་ཚེ་གཅོད་པ་ལའོ། །ཚུ་ཏི། བོ། མཐའི་ལས་ལའོ། །ཁོ་མྱ་ཏི། ་ཨ་ནུ་སྱ་ཏི། དོ། ཀུན་ཏུ་འགེམས་པ་ལའོ། །དྱ་ཏི། ཨ་བ་བྱ་ཏི༑པྲ་ཎི་ཏྱ་ཏི། རཱ་དྷ། །སཱ་དྷ། ཡང་དག་པར་གྲུབ་པ་ལའོ། །ར་དྷིའི་ལས་ཅན་ལས་དང་ལས་ཅན་མ་ཡིན་པ་ལ་འཇུག་བཞིན་པ་ལས་ར་དྷི་ལས་ཡན་་རྐྱེན་དུ་འགྱུར་རོ། །རཱ་དྷ་ཏི། །སཱ་དྷ་ཏི། གཞན་ལ་སུ་ལ་སོགས་པ་བཞིན་ནོ། །རཱ་དྷཱ་ཏི། ལོ་འཕེལ་བ་ཉིད་ལ་དྲི་བ་ལ་སོགས་པ་ཅན་འདིའོ། །གཞན་ལ་ནི་དིབ་དང་པོ་ཅན་ཞེས་པའི་ཚིག་ལས། ཨ་ཥ་ཉེ་བར་བསྟན་པ་འདི་འདོད་དོ། །བྱ་དྷ༑ལྔ་འཚོག་པ་ལའོ། །གྲ་ཧི། ཛཱ། བ་ཡི་བྱ་དྷི། བཉྩི། བྱ་ཙི། པཙྪིརྒྱ་:པྲཉྩི། བྷཉྫི་། རྣམས་ཀྱི་ཡོན་ཏན་མེད་པ་ལའོ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎ་ལ་ཨ་ཡིག་གི་གནས་ལ་ཨི་ཡིག་གོ། །སསྨྲ་སཱ་ར་ཎ་ལ་མ་ཡིན་ནོ་ཞེས་པའི་་དགག་པ་ལས་སླར་ཡང་སཾ་པྲ་ས་ར་ཎ་རུ་མི་འགྱུར་རོ། །བི་དྷུ་ཏི། པུ་ཥ་རྒྱས་པ་ལའོ། །པུ་ཊོ་ཏི། ཤུ་ཁོ་སྐྱེམས་པ་ལའོ། །ཤུ་ཥྛ་ཏི། དུ་ཥ་རྣམ་འགྱུར་ཉིད་་ལའོ༑། དུ་ཥྛ་ཏི། ཤྲཱི་ཥ་འཁྱུད་པ་ལའོ། །:གླེ་ཥྱ་ཏི། ཥ་དཱ། ལུས་རབ་ཏུ་འགྱུར་བ་ལའོ། །ཨ་ཡིག་ནི་ཨཱཏ་རྗེས་སུ་འབྲེལ་པ་ལས་ཀྱང་ངོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །སཱི་བྱ་ཏི། གླུ་དྷ་བཀྲེས་པ་ལའོ། །ཀླུ་དྷུ་ཏི། ཤུ་དྷ་དག་་།
