ཎ་ཉིད་དོ། །གྲོནྣ་ཏེ། གྲིནྡི་ཏེ། རྣམ་པར་བཤད་པ་རྣམས་སོ། ཟླ་། ཤྲཱི་དང་། སྦི་འཚེ་བ་ལའོ། །ཤྲཱི་ཎཱ་ཏེ། སྤྲི་ཎེ་ཏེ། པྲ་་སྐྱོང་བར་བྱེད་པ་དང་རྫོགས་པར་བྱེད་པ་ལའོ། །ཤྲཱི་ཎཱ་ཏི། བློ་དང་པྲི། སྒྲིབ་པ་ལའོ། །བི་ཎཱ་ཏི། ཤྲཱི་ཏཱ་ཏེ། བྷཾ། བརྒྱད་བཀག་ལའོ། །བྷཱི་ཎཱ་ཏི། ཛྭ། ་རྣམ་པར་འཇོམས་པ་ལའོ། །ཛཱ་ནཱ་ཏི། ཛཱ། ལང་ཚོ་ེཉམས་པ་ལའོ། །ཛཱི་ཎཱ་ཏི། ཛྲ་ཚུལ་ལའོ༑། ཎཱ་ཎཱ་ཏི། གྲོ། སྒྲ་ལའོ། ཤྲཱི་ཎཱ་ཏི། ཛུ། ལང་འཚོ་ཉམས་པ་ལའོ། །གྲ་ཧཱི་ཛཱ་ལ་སོགས་པའི་མདོ་ཡིས་སཾ་པྲ་ས་ར་ཎའོ། །མའི་མཐའི་རྣམས་ཀྱི་དེ་རིང་པོའོ་ཞེས་རིང་པོ་ཉིད་དོ། །པཱུ་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ་ཞེས་་པས་ཐུང་ངུ་ཉིད་དོ། །ཛི་ནཱ་ཏི། རཾ། འགྲོ་བ་ལའོ། །རཾ་ཎཱ་ཏི། རཱི། རཾ:;ཎ་ཏཱི། ལཱི། སྦྱོར་བར་བྱེད་པ་ལའོ། །@་ནཱ་ཏི། ལྕི་ཡང་ངོ༌། །ཕྱི་ནཱ་ཏི་བྲྀ་ཏའོ། །པུ་ལ་སོགས་པ་ཉི་ཤུ་རྩ་གཉིས་སོ། །།ལུ་ལ་སོགས་པ་ཉི་ཤུ་རྩ་་་་གཅིག་གོ། །བྲི་སྒྲིབ་པ་ལའོ། །བྲི་ཝེཎཱ་ཏི། བྷི། ་འགེངས་པ་ལའོ། བྷི་ནཱ་སྟེཏི༑་ཎ་བྷ་དང་ཏུ་བྷ་དང་ཀྱི་ཥ་འཚེ་བ་ལའོ། །ནིབྷཱུ་ཏི། ཏུ་ཏི། ཀཱིཥྛ་ཏི། ཛྙཱ༑་པཁོང་དུ་ཆུད་པ་ལའོ། །ཛྷཱའི་འང་ངོ་ཞེས་པས་ཛཱ་རུ་བསྒྱུར་རོ། །ཛཱ་ནཱ་ཏི༑ལནི་སྒ་བ་ལ་ཛཱའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཛཱ་ནཱི་ཏི། སཾ་དང་པྲ་ཏི་་་དག་ལས་མི་དྲན་པ་ལའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །སཾ་ཛཱ་ནཱི་ཏི། པྲ་ཏི་ཛཱ་་ནཱི་ཏེ། བནྡ་འཆིང་བ་ལའོ། །ནཱིའི་རྐྱེན་ནོ། །ཨ་ནུ་ཥརྒ་དབྱིའོ། །བ་དྷཱ་ཏི། ཤནྡྲི་་གྲོལ་བར་བྱེད་པ་དང་སོ་སོར་དགའ་བར་བྱེད་པ་ལའོ། །ཤཱ་ཐཱ་ཏི། ཤ་རཱ་ནཱ། ༼མནྡཱ། སྲུབ་ས་ལའོ། །མཐཱ་ཏེ། བྲ་མཐཱ་ཏི། གྲཉྫ། སྤེལ་བ་ལའོ། །གྲབྷཱ་ཏི། ཀུནྡྲ། ཀུན་ནས་ཉོན་མོངས་པར་བྱེད་པ་ལའོ། །ཀུབྷཱ་ཏི། མྲྀང༌། ཕྲ་མོ་ལའོ། །མྤནྣཱ་ཏི། མྲཌ། བདེ་བ་ལའོ། །མྲྀདྷཱ་།
