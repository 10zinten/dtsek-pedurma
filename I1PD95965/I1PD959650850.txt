ཡ་ལ་ལ་བ་ད་ན་སྣར་ཐང་། ཅོ་ནེ། །སྡེ་དགེ། །སྣ་ཚོགས། པེ་ཅིན། །སྒྲ་སྨན་དཔྱད། །།སྒྲ་སྨན། ནོདོདཔེ་གཞི། སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང༌། ན་བབ—ན་བན་། །།།།ཚེག་ཁྱིམ། པ་ཚུ་ཡོད། ཁ་བའི་འདབ་མ་ཁ་ར་ཁངན་ཁང་བ་གྱི་ན་ཁ་བ་དྲག་ན་ང་ན་མ་གཞི་ནས་བོད་ཀྱི་པ་དང་། །བ་ལའོང་། ངའ་བ་མཆུ་སྒྲོས། རྒྱུ་ལ་རང་བཞིན་ལག་ཁ་བ་འབབ་ཐོག་འགོད་རྒྱུད་ཁ་འཁ་བཏམ་ཆག་ལ། ཁོ་ལ་དད་པ་དང་། ཁ་དེ་ན། སྐྱེ་བ་མེད་པ་སུ་འདུག་གའཚོ་བའི་ཁ་གསབ་མཁན་པོ་དམ་::ཀྱིང་བ་འག་གང་། ལོ་དང་ཆུ་བཏོན་པས་ན་བའི་བ་ཞེན་པ། དེ་བ་ག་ཤ། མེ་ཁྱེན་པ་དང་། དེ་བབས་པ་ལ་་་། ཁ་བ། རིག་པ། བྱ་བ་ད་་གནང་བས། དེ་ཁ་ཏ་བ། དགའ་བ་ཅན་དག་ག་ག་ལ་།
