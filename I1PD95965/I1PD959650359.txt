སྣ་ཚོགས། །༼ནི༎ཧཱ་ཧཧཱ་ཧཱ། ཧཱ་ཧཱ། ཧཱུྃ། ཧཱ་ཧཱ་བྷིཧཱཧེ། ཧཱ་ཧཱ་ཧཱུཾ། ཧཱ་ཧཱ་ཧཱཏ། ཧཱ་ཧཱ་ཧཱུཾ། །ཧཱ་ཧཱ་བྷུཧཱ་ཧསྱ། ཧཱ་ཧོ། ཧཱ་ཧཱུྃ། ཧཱ་ཧེ། ཧཱུ་ཧོ༔ཧཱ་ཧཱ་སུ། ཧེ་ཧཱ་ཧཱཧེ་ཧཱ་ཧོ། ཧེ་ཧཱ་ཧཱ། ཨཉྫིའི་སྒྲ་ལས་དང་པོའི་སུ་ལ། །སུ་རུ་ཉིད་དེ། ཚེག་དྲག་གོ། །དང་པོ་དག་ལ་ཨ་ཙ་ལ་ཞེས་པས་རིང་པོའོ། །ཛ་ས་ལ་ཞེས་་པ་:ཨ་ཉིད་དེ། ཨེ་ཙ་ལས་ཨ་ཡ་དང་ཨ་བ་དང་ཨཱ་ཡ་དང་ཨཱ་བའོ། །ཨཱ་་མ་་ལ་སྔོན་མའོ། །ཏ་ཏ་ཤ་སོ་ན་བུམ་ཤ་ལ་ཞེས་པས་ན་ཡིག་གོ། །ཊོནྟི་ཡཾ་ན་ཞེས་པས་ནཱ་ཡི་དངོས་པོའོ། །ཨི་ཏྱ་ས་ཁྱུ་ཞེས་པས་སུ་པ་ལ་ཨི་ཡིག་ལ་ཨེ་ཉིད་དོ། །ཨེ་ཙ་ལས་ཨཱ་ཡ་དང་ཨ་བ་ཨཱ་ཡ་དང་ཨཱ་པའོ། །ང་སི་ང་སོ་ཞེས་པས་ཕ་རོལ་་སྔོན་མ་ཉིད་དོ། །ཨི་ཀ་ལས་ཡ་ན་ཨ་ཙ་ལ་ཞེས་པ་ཡན་བསྟན་པར་བྱའོ། །་ཟ། ཐུང་ངུ་ནུ་ཊའོ། །སྔར་ལྟར་རྗེས་སུ་འབྲེལ་པ་དབྱིའོ། །ནཱ་མྱ་ཏི་པྲ་ཙ་ཏ་སྲོ་་ཞེས་པས་རིང་པོའོ། །ཨེངོ་། །ཙྪ་ཞེས་པས་ང་ཨོ་ཡིག་གོ། །ཨེ་ང་ལས་ཀྱང་ཨ་ཏ་བསྟན་པར་བྱའོ། །ཏ་ཡིག་ཧྲི་ཡཾ་གནྡྷ་བ་ཊ་སྣ་བསྡུ་བའི་དོན་ཏོ། །ཨ་ཡིག་དང་ཨོ་ཡིག་དག་ལས་ཨེ་ཙལ་ཞེས་པས་ཨོ་ཡིག་གོ། །ཀོ་ཞེས་པས་་་་ཉིད་དོ། །སཾ་བྷོ་དྷ་ནེ་ཨི་ཏེ་ཏོ། །།རིང་ཞེས་པས་ཨེ་ཉིད་དོ། །ཨེ་ང་ཧྲ་སཱ་ཏ་སཾ་བུ་དྷཱ་བཏ་ཅེས་པས་སུ་དབྱིའོ། །ཨགྣི། ་ཨབྷཱི། ཨག་ཡཨབྷཾ། ཨགྣི༑ཨགྷཱན། ཨགྣི་ནཱ། ཨགྣི་ནཱ། ཨམྦྷི་ཧཱུཾ། ཨགྣི་བྷིཨག་ཡེ། ཨགྣི་༄༑ཨགྣི་བྷུ། ཨགེ། ཨགི་བྱཱ། ཨངྒེ། ཨཙྩོཨགྣི་ནཾ། ཨོསྒོ། ཨཙྩོཨགི༈ཧེ་ཨམྦྷེ། ཧེ་ཨམྦྷེ། ཧེ་ཨགྷཱ་ཡ། དེ་ལྟར་ཨི་མཐའ་ཅན་གཞན་་རྣམས་ལ་ཡང་མཚན་པའོ:༌། །ས་ཁིའི་སྒྲའི་དབྱེ་བ། རི་དུ་ཤ་ན་སྤུ་རུ་དཾ་སོ་ཉ་ནི་ཧ་སཱཉྩ་ནི་སཱ་བ་ཞེས་པས་སུའི་ཕ་རོལ་གྱི་ཨི་ཡིག་ལ་ཨ་ན་ང་བསྟན་པར་བྱའོ། །ང་ཡིག་ང་ཏ་ཞེས་པ་མཐར་བསྟན་པའི་ཕྱིར་རོ། །ན་ཡིག་གི་ཨ་ཡིག་་།
