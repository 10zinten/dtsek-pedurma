ཏི་ཨི་ངའི་ག་མི་རུ་བསྒྱུར་རོ། །ཧ་ན་དང་ཨིངའི་ག་མ་དག་གི་ཨུ་པ་དྷཱའི་འོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཨ་དྷི་ཛི་ནཱཾ་ས་ཏེ། ཧ་ན། ཨུ་པ་དྷཱ་རིང་པོ་ཉིད་དོ། །ཨ་བྷུས་ལས་ཀྱང་ངོ་ཞེས་པས་གྷ་ཉིད་དོ། །ཇིགཾ་ས་ཏཱི། ཨང་། །ཨ་དེའི་སན་ཨ་བྱ་ཏ་ནི་དག་ལ་གྷསླའོ་ཞེས་གྷནྡྷཾ་རུ་བསྒྱུར་རོ། །ས་ཡིག་གི་ཏ་ཡིག་གོ། །ཛི་གྷཏྠ་ཏི། ད་ཧ། ད་དང་པོའི་གྷའོ་ཞེས་པས་གྷ་ཉིད་དོ། །དང་པོ་བཞི་པ་ཉིད་དེ་ཀ་ཉིད་དོ། །དིདྷ་ཀཱ་ཏི། བི་ད་རྣམ་པར་དཔྱོད་པ་ལའོ། །པ་ད་ཡོད་པ་ལའོ། །བི་དྷི། རྙེད་པ་ལའོ། །པི་སིཏྠ་ཏི། བི་བིཏྠ་ཏི། རུ་དི། མཆི་མ་འཛག་པ་ལའོ། །རུ་ད་བ་དི་མུ་ཥ་རྣམས་ཀྱི་སན་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །རུ་རུ་དི་ཥ་ཏི། བི་ད། ཤེས་པ་ལའོ། །::དི་བི་དེ་ཥ་ཏི། མུ་ཥ། རྐུ་བ་ལའོ། །མུ་མུ་ཏི་ཥ་ཏེ། དྷུ་ཏི་སཱ་བི་དག་གི་ཨ་བྷྱཱ་སའིའོ་ཞེས་་་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །།།སན་ཡང་ངོ་ཞེས་བརྗོད་པར་བྱའོ་ཞེས་པས་ཡང་ན་ཡོན་ཏན་ནོ། །::། དི་དྷུ་ཏི་ཥ་ཏེ། དེ་བྱོ་ཏི་ཥ་ཏེ། དེ་བཞིན་དུ། རུ་རུ་ཙི་ཥ་ཏེ། རུ་རོ་ཙི་ཥ་ཏེ། ལི་ལི་ཏི་ཥ་ཏི། ལི་ལོ་ཏི་ཥ་ཏི། ་ལུ་ལུ་བྷི་ཥ་ཏི། །ལུ་ལོ་བྷི་་ཥ་ཏི། ཤུ་ཤོ་བྷི་ཥ་ཏི། ཤུ་ཤོ་བྷི་ཥ་ཏ། ཙུཀྵ་བྷི་ཥ་ཏེ། ཙུ་ཀྵ་བྷི་ཥ་ཏེ། ཨོ་པི་ཛཱི། འཇིགས་པ་དང་གཡོ་བ་དག་ལའོ། ་། བི་་ཇིའི་ཨིཊ་ལའོ་ཞེས་པས་་་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །བི་བི་ཛི་ཥ་ཏེ། ཀུཊ་སོགས་ཀྱི་ཨིན་ཨིཙ་ཨཊ་་་རྣམས་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཨུཙྪ་ཀུ་ཊི་ཥ་ཏི། ཨུཏྤ་པུ་ཊི་ཥ་ཏི། ་.མུ་ཧ། མུ་མུ་གྷ་ཏི། མུཙ། མུ་མུ་ཀྵ་ཏི། ཀིར་སོགས་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་གོ། ཌ་། ཨུ་ཙའི་ལས་ཅན་མ་ཡིན་པའི་ཨོཏ་ཡང་ན་་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨོ་ཉིད་ཡང་ནའོ། ་། ཨ་བྷྱཱ་ས་དབྱི་བ་ཡང་ངོ༌། །མོཀྵ་ཏེ༑་མུ་མུ་ཀྵ་ཏེ། བྲྀ་ཏུ། 《བྲོ་དྷ། ཀྲྀ་། སྱནྡྲ་འདི་རྣམས་ཀྱི་ཨཱཏྨ་ནེ་མ་།
