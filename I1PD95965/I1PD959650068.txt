ཨརྱ་ཨོམ། །ཨཔྱོ། །ཨུ་ཨོ། ཁོ་ཊི་ཨ་བཙྪ་ཨུ་བཙྪེ་པ་རེ་ས་ཧ་ལོ་བྷ་བ་ཏི། ་ཊ་དེ་རིང་བདག་ལོས་ཀྱང་། ཨུ་ལ་ཨོའོ། །ཨ་ཡིག་ཕ་རོལ་ཨོ་ཡིག་དང་འདུས་པ་ལས་ཡོངས་སུ་འགྱུར་རོ། །གངྒ་ཨུ་ད་ཀཾ།  ོ་གངྒཱ་ད་ཀཾ། ཛཾ་ཨང་། ་ཨ་བཙྪཱ་རོ་པཎྞཱ་པ་རེ་ས་ཧ་ཨ་རྦབྷ་བ་ཏི། །::ཀོ་གངྒའི་ཆུ། རི་ལ་ཨངའོ། །ཨ་ཡིག་ཕ་རོལ་རི་ཡིག་དང་ཚོགས་པ་ལས་ཨར་དུ་འགྱུར་རོ། །ཏ་བ་རིདྡྷིཏ་བདྡྷིམ་མ་རིདྡྷིམ་མདྡྷི་ཀཱ་ཙི་དཱར། ཁོ། ཁྱོད་ཀྱི་འབྱོར་པ། ཁྲི་་ལོ་་ས། བདག་གི་འབྱོར་པ། རེས་འགའ་ཨཱར་ཏུའོ། །ཨ་བནྡྷཱ་རཾ་བནྡྷེ་པ་རེ་ས་ཧ་ཀཱ་ཙི་དཱར་བྷ་བེ་ཏི། རཾ་ཎ་རི་ཎཾ། རཾ་ཎཎྞཾ། པྲ་ར་ཎི་པྲཎྞཾ། །ཨ་ཡིག་ཕ་རོལ་རི་ཡིག་དང་འདུས་པ་ལས་རེས་འགའ་ཨར་དུ་འགྱུར་རོ། །བུ་ལོན་རྙིང་པ། བསམ་ཤྲཱི་ཏ་རཾ་ཏཿཤཱི་ཏཱནྟ། དུཁ་རི་ཏཿདུཁཱཏྟ། ལོ་ཨཝ། ཨ་བཎྜ་ལོ་བརྞ་པ་དེ་ཨེ་ཧ་གྲང་བའི་དངོས་པོའམ། གྲང་བས་ཁྱབ་པ། ལ་ལ་ཨ་ལའོ། །ཨ་ཡིག་ཕ་རོལ་ལི་ཡིག་དང་ཕྲད་པ་ལས་ས་ལྔ་བྷ་བ་ཏི། ཏ་བ་ལོ་ཀཱ་རཿཏ་བ་ལྟ་རཾ། ཏ་བམྦྷ་ར། ར་ལ་ཡོསཱ་བཙྩཾ་་་། ཨལ་དུ་འགྱུར་རོ། ཀཀཱ་ ི། ཁྱོད་ཀྱི་ལི་ཡིག། ར་ལ་འདྲ་བས་ལ་ལ་ཡང་འོག་མ་ཉིས་བརྩེག། བཱ་བལྟ་བྱ། པ་རཾ་ཨོཾ་གཔ་ལཾ་ཨངྐཔཎྞཾ་ཀཿཔརྨ་ཀཧོ་ཀྲྀ་ལཱ་ཀཱ་རཿར་དང་ལ་དག་འདྲ་བའི་ཡི་གེར་ཡང་བཤད་དེ་ཡང་ནའོ་། །སྦྱིན་སྲེག་མཁན་གྱི་ལི་ཡིག་ཟེར་བ་དེའི་ངག་། ཧོ་ཏཱ་ཀཱི་རཿཧོ་ཏ་ལཱི་ཀཱ་རཿཨེ་ཨེ་ཨེ། ཨ་བཥྛ་ཨེ་ཀཱ་རེ་ཨེ་ཀཱ་རེ༔་པ་རེ། ་་ཝེརང་ལ་འཇུག་པ་ཡིན། ཨེ་ལ་ཨི་ལ་ཨེའོ། །ཨའི་ཡི་གེ་ཕ་རོལ་ལ་ཨེ་ཡིག་དང་ཨི་ཡིག་འདུས་པ་ལས་ས་ཧ་ཨེ་ཀཱ་རོ་བྷ་བ་ཏི། ་ཌོ་པ། །ཏ་བ་ཨེ་ཥཱ། ཏ་བི་ཥ། ་ཏ་བ་ཨེ་ཤཱÉ། ཨི་ཡིག་ཉིད་དུ་འགྱུར་རོ། །འདུས་པ་ལས་ཨི་དུ་འགྱུར་རོ། །།ཁྱོད་འདིར། །ཁྱོད་ཀྱི་དབང་ཏ་བི་ཤཱཎཾ། ཨོ་ཨོ་ཨོ། ཨ་བནྡྷུ་ཨོ་ཀཱ་རེ་ཨོ་ཀཱ་རེཙཱ་པ་ར་ས་ཧ་ཨོ་ཀཱ་རོ་བྷ་བ་ཏི། ཕྱུག་ཁ་། ཨོ་ཨོ་ལ་ཨོའོ། །ཨ་ཡིག་ཕ་རོལ་ལ་ཨོ་ཡིག་དང་ཨོ་ཡིག་འདུས་པ་ལས་ཨོ་ཡིག་ཏུ་འགྱུར་རོ། །ལལམ—}།
