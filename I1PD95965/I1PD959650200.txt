རྣམས་སོ། །ཀྲྀ་ཧྲི་ཏཱཾ་ཡ་མི་དཱ་རི་ཨརྫི་ལས་ཨུ་ནན་ནོ། །ཤ་ཀི་ལས་ཨུ་ནིའོ། །པྲི་པཱ་ཏ་ལི་ལས་པའོ། །སྟུའི་ཨུ་ཡང་ངོ༌། །ཡུ་ཀུ་སུ་རྣམས་ཀྱི་ཀིཏ་ཡང་ངོ༌། །བཱ༔་ལ་སོགས་པ་རྣམས་སོ། །སརྟི་ལས་ཨ་བ་སྟེ་སུ་ཀའང་ངོ༌། །བི་ཊ་པ་ལ་་་སོགས་པ་རྣམས་སོ། །རཱ་ཏི་ལས་ཨི་བའོ། །གུ་རི་ལས་ཕ་ཀའོ། །བྷཾ་གཾ་དག་་ལས་ཕའོ། །ཨ་ཤི་ལ་ཊི་ཀ་ཎི་ཁ་ཊི་པི་ཤི་ལས་ཀུན་འོ། ་་། ཤི་ཝ་ལ་སོགས་པ་་་རྣམས་སོ། །ཀཽ་ཤཱི་གདྷི་ལས་ཨ་བྷ་ཙའོ། །རྀ་ཊི་བྲི་ཏི་རཱ་སི་བལླི་ལས་ཀིཏ་དོ། །དྲིང་ལས་བྷའོ། །ཧཱུྃ་ལས་བྷ་ནའོ། །ཨིཎ་ལས་ཀིཏ་དོ། །གུ་དྷི་ལས་སྤུ་མའོ། །པྲ་ཐི་ཙ་རི་ལས་ཨ་མེ་ཙའོ། །།རཾ་སྐུ་སཱུ་ཧུ་དྷཱི་ཀྱི་གླུ་བྷ་ཡཱ་བ་དི་ཡཀྵི་ཎཱི་རྣམས་ལས་མ་ནའོ། །་། གྲ་སིའི་ཨཱཏ་ཡང་ངོ༌། །སཱུ་ཙི་ལས་སྨན་འོ། །ཡུ་དྷི་ཧི་ཨིནྟཱི་ཛ་ནི་བྱཱ་དྷུ་རྣམས་ལས་མ་ཀའོ། །བྷི་ལས་ཥཱ་ཀ་ཡང་ནའོ། །ཡུ་ཛི་རུ་ཛི་ཏི་ཛི་ལས་ཀུ་་ཡང་ངོ༌། །གྷརྨ་གི་ཨ་དྷ་མ་རྣམས་སོ། །ཏ་ནི་ལས་ཀ་ཡིན་ནོ། །།ཧྲི་ཉ་་ལས་དུ་ཀ་ཡང་ངོ༌། །མཱ་ཙཱ་ས་སི་སཱུ་རྣམས་ལས་ཡའོ། །ཛ་ཡ་ལ་སོགས་པ་་་རྣམས་སོ། །རུ་ཙི་བྷུ་ཛི་ལས་ཀི་ན་འོ། །མ་དི་ལས་མྱ་ནའོ། །།སྤྲི་ཏི་ལས་ཨཱ་ཡུའོ། །བྲིང་ལས་ཨེ་ནྱའོ། །ཨརྟི་ལས་ཨ་ཅུ་ཙའོ། །ཧཱའི་ཧིང་ཡང་ངོ་། །པརྫ་ཅུའོ། །པུ་ཎི་ལས་ཀྱན་ནོ། །ཤི་ཀྱ་དྷིཙྩཾ་མོ། །ཨུཎ་ལ་སོགས་པར་རྐང་པ་གཉིས་པ་རྫོགས་སོ། །།ཕྱི་དང་ཏ་ཀོ་བ་དང་། །དེ་ལ་འ་Eའིཔ་དེ་ཚོ་བོད་ཊི་པ་ན༽དང་བ་ཀ་ད། དུ་མ་དི་ཨངྷི་ཝ་སི་མ་ཐི་ཙ་ཏི་རྣམས་ལས་ཨུ་ར་ཙའོ། །མ་ཀུ་ར་དདྷུ་ར་ཤི་དྷུ་ར་རྣམས་སོ། །ཨ་སི་ལས་ཨུ་ར་ནའོ། །ཤཱ་ཤུ་རའོ། །ཏི་མི་རུ་དྷི་མ་དི་མཟིགས་་་ཙན་དི་བནྟི་རྣམས་ལས་ཀི་ར་ཙའོ། །སྦི་ར་ལ་སོགས་པ་རྣམས་སོ། །ཏཉྩ་བཉྩ་ཤ་ཀི་ཀྱི་པི་ཀླུ་དི་རུ་དི་མ་དི་མནྡ་ཙནྡེ་རྣམས་ལས་ར་ཀའོ། །ཤི་ཏི་ཀྲྀ་ཏི་ནི་བཱི་ཙྪ་—།
