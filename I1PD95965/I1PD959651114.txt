ཙིཀྵེ་ཐ། ཙིཀྵེ་ཡ་ཐུ་ཙིཀཱ་ཡ། ཙིཀྐཱ་ཡེ། ཙིཀཱི་ཡ། །ཙིཀྐི་ཡི་བ། ཙིའི་ཡི་མ། ལས་ལ། ཙིཀྵེ་ཡེ། ཙིའི་ཡཱ་ཏེ། ཙིཀཱི་རེ། གླིང༌། སིཀྐི་ཡེ། །སིལླི་ཡཱ་ཏེ། སིདྡྷི་ཡི་རེ། ཨཱི་ཡིག་མཐའ་རྣམས། ཎཱིཉ། ཐོབ་པ་ལའོ། །ནི་ནཱ་ཡ། ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་སོགས་ཀྱིས་ཡ་ཉིད་དོ། །ནི་ནྱ་ཏཿནི་རཱུནི་ནེ་ཐ། ནི་ན་ཡི་ཐ། ན་ནྱ་ཐུནི་ནྱ། ནི་ནཱ་ཡ། ནི་ན་ཡ། ནི་རྩི། ནི་རྩྭ་ཏེ། ན་རྩི་རེ། ཏཱིང་། ཌི་སྐྱེ། ཌི་ཊཱི་རེ། ་ལཱིང༌། ལི་ལྕ་ཏེ། ལི་ལི་རེ། འུ་ཀྲྀཉི། ཙི་ཀཱ་ལ། །ཨི་ཡ་རུ་བསྒྱུར་རོ། །ཙི་ཀྲི་ཡ་ཏུ་ཙི་ཀྲི་ཡུཧྲི། ཛི་ཧཱ་ཡ། ཛི་ཧྲི་ཡ་ཏུཛི་ཧྲི་ཡུཛི་ཧྲི་ཐ། ཛི་ཧྲ་ཡི་ཐ། དྲིཉ༑པི་པྲ་ཡ། ་པ་པྲ་ཡ་ཏཿཔི་སྲི་ཡུཔི་སྲི་ཐ། པི་པྲི་ཡི་ཐ། ཨུ་ཡིག་མཐའ་རྣམས། ཙུང་། ཙུ་ཙུ་པེ། ཙུ་ཙྪ་བི་རེ། ཙུ་ཙྩ་བི་ཥ། ཙུ་ཙུ་བི་དྷཱ། ཙུ་ཙུ་བི་ཝཱ། སྲུང༌། པྲ་བྲུ་བེ། པྲ་པུ་བཱ་ཏེ། པུ་པུ་རེ། ཡུ། སྤེལ་བ་ལའོ། །ཡུ་ཡཱ་བ། ཡུ་ཡུ་བ་ཏུཡུ་ཡུ་བུཡུ་ཡ་བི་ཐ། རུ་རཱ་བ། རུ་རུ་བ་ལཿརུ་རུ་བུརུ་རུ་བི་ཐ། ནུ་ནཱ་བ། ནུ་ནུ་བ་ཏུནུ་ནུ་བུ?སྣུ། སུབྷུ་བ། སུབྷཱུ་བ་ཏུསུབྷུ་བུཀྵ། ཙུ་བ། ཙུ་ཀླུ་བ་ཏུཙུ་གླུ་བུབྷྱཱ། ཙུ་བྷྱཱ་བ། ཙུ་སྨྱུ་བ་ཏུཙུ་ཀྵ་བུཀུ། སྒྲ་ལའོ། །ཙུ་ཀཱ་བ། ཙུ་ཀུ་བ་ཏུཙུ་ཀུ་བུཙུ་ཀོ་ཐ། ཙུ་ཀ་བི་ཐ། ཨུ་ཡིག་མཐའ་རྣམས། ལུཉ། ལུ་ལཱ་བི། ལུ་ལུ་བ་ཏུལུ་ལུ་བུ༔ལུ་ལ་བི་ཐ། བྷུཉ། པུ་པཱ་བ། པུ་བུ་བ་ཏུཔུ་པུ་བུལུ་ལུ་བེ། བུ་བུ་བེ། བྷུ་ཡོད་པ་ལའོ། །བྷུའི་མཐའ་རྣམས་པ་རོག་དང་ཨ་རྱ་ཏ་ནི་དག་ལ་བའོ་ཞེས་་་པས་བ་ཡིག་ཨཱ་ག་མའོ། །གཉིས་བརྗོད་དོ། །ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུ་ཉིད་དོ༑། བྷ་བ་ཏིའི་ཨ་འོ་ཞེས་པས་ཨ་ཧྥུ་སའི་ཨུ་ཡིག་གི་ཨ་ཉིད་དོ། །བ་བྷུ་བ། །
