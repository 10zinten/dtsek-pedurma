གཏོར་མའི་ཕྲེང་བ་ཞེས་བྱ་བ་བཞུགས་སོ། །ས་ཧི་ཏཾ་གྲིཉྩ་གྲིཙྪ། ཁ་ཁ། ཁཱ་ཧི་ཁཱ་ཧི། མ་མ་ན་སིདྡྷཾ་པྲ་ཡཙྪ་ཨོ་ཨཱཧཱུྃ་ཕཊ་སྣ་ཧཱ། ནཱ་ག་བ་ལི་ཀླུའི་གཏོར་མའོ། །ཨོཾ་ན་མཿཤྲཱི་མི་ཏྲེ་ཡ་ནཱ་ཐཱ་ཡ། བྷཱ་བི་བི་རཱུ་ཙ་ར་ཎ་སམྤནྣ་སམྱཀཾ་བུདྡྷཱ་《བྱམས་པ་མགོན་པོ་ལ། སྒོམ་པའི་རིག་པ་དང་ཞབས་སུ་ལྡན་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་སརྦ་ས་ཏྭཱ་ཏྲེ་ཎ་དྲ་བི་ཏ་ཙིཏྟཱ་ཡ། རྒྱས༑བཱ་བུ་དྡྷི་ཙཱ་ནུཏྟ་རཱ་ཡཾ་སམྱཀཾ། དུ་དྷཾ་ཏྭཱ་ནུཏྟ་རཱ་ལོ་སམྱཀཾ་ཟེར། སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་གྱིས་ཨི་དཾ་བ་ལིཾ་དི་བྱ་གནྡྷ་བརྨ་ར་སོ་པེ་དཾ་ཨུ་པ་བྷུ་ཛུ་ནཱཾ། སརྦ་སཏྭཾ་བརླན་པའི་ཐུགས་ལ་ཕྱག་འཚལ་ལོ། །ལྷ་རྫས་ཀྱི་དྲི་དང་ཁ་དོག་དང་རོ་དང་ལྡན་པའི་གཏོར་མ་འདི་བཞེས་ཏེ། ཉྩ་སཾ་སཱ་དུཁཱཎ་བཱ་ཊཾ་བོ་དྷོ་པྲ་ཏིཥྛ་པ་ཡི་ཥྛ་སི་ཨོཾ་ཨཱཿཧཱུྃ་ཕཊ་བདག་དང་སེམས་ཅན་ཐམས་ཅད་འཁོར་བ་སྡུག་བསྔལ་གྱི་རྒྱ་མཚོ་ལས་བསྒྲལ་ཏེ། རྫོགས་པའི་སངས་རྒྱས་སྭ་ཧཱ། ཕྱོགས་ག་བདུད་དུ་མི་ཏྲི་ཡ་བ་ལིངོས་ཀྱི་ཀྱི་ས་ལ་འཁོད་པར་མཛོད་ཅིག། བྱམས་པའི་གཏོར་མའོ། །བ། ཨོ་ཨིནྡྲ་ཛ་མ་ཛ་ལ་ཛ་ཛཀྵ་:བྷཱུ་ཏ་བབྷི་བཱ་ཨུ་རཀྵ་ཙནྡྲ་སུཛྫ་དབང་པོ། གཤིན་རྗེ། ཆུ་ལྷ། གནོད་སྦྱིན། འབྱུང་པོ། ：མེ་ལྷ། རླུང་ལྷ། སྲིན་པོ། ཟླ་བ། ཉི་མ། མཱ་ད་བཡྻ་ཏཱ་ལི་ཨཊྚ་སཱཔྤ་སྭཱ་ཧཱ། ཨི་དཾ་བ་ལི་བྷུཉྫ་ཛི་གྷ་ཕུ་སྨ་དྷཱ་པ་ཚྭ་ས་པི་གྷ་མྱོས་མ། མཐིལ། འོག་ན། སྦྲུལ་བརྒྱད། ：འདི་སྣོམས། མེ་ཏོག། བདུག་པ་ཤ། ཨམྨ་སཔྲ་སཱ་དྷ་ཁཱནྟི་ཁུ་ཎི་ཕེ་ཌ་གཱ་ད་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱ་ཎཾ། ཉ་ཤ་བདག་གི་ཐམས་ཅད་བསྒྲུབས་བཟོད་པར་གསོལ་གནོད་པ་སོལཔེ། བ་་་་་ཨཱ་དྱ་ནུཏྤནྣ་ཏྭཱ་ཏ་ཨོཾ་ཨཱཿཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཤྲཱི་ཧེ་རུ་ཀ་བ་ལི་ད་འདི་དཔེ་ཤས་ཆེར་མི་འདུག། དཔལ་ཧེ་རུ་ཀའི་གཏོར་མའོ། །ཨོཾ་ས་ཧ་ཛ་སམྦ་ར་ཨཱ་ཧཱུྃ་ཨཱ་གཙྪ་ཨཱ་གཙྪ། ཤྲཱི་གྷོ་ཨི་དཾ་བ་ལིཾ་གྲིཉྩ་གྲི། ལྷན་སྐྱེས་སྡོམ་ེཔ། ཡང་ན་མར་ཁ་བ་བབ་པ་བཞུགས་པ་གསོལ་ཞིང་འཐུང་སྟེ་—།
