ཉིད་དོ། །ན་སྒྲ་ལས་ཀྱང་བི་ཀ་ར་ཎ་ལས་ཀྱང་ངོ་ཞེས་པས་ཨནུ་ཥངྒ་དབྱིའོ། །བྷ་ནཏྟི། བྷཏྟ། བྷཉྫནྟི། ་བྷ་ནཀི། བྷཏྟ། བྷ་ནིཙྪ། བྷཉྫབྷཉྫ། བྷུ་་ཛ། །སྐྱོང་བར་བྱེད་པ་དང་ཟ་བ་དག་ལའོ། །ཁོ། བྷུ་ནཏི། བྷུརྞབྷུཉྫནྟི༑་བྷུ་ནཀྵ། །བྷུདྡྷ། བྷུདྡྷཱ། བྷུ་ནÉབྷུཉྫ། བྷུཉྫ། ཟས་ལ་བྷུ་ཛིའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྷུནྟེ། བྷུཉྫ་ཏེ། །བྷུཉྫ་ཏེ། གྲོགས་ཧ་དང་སི་ཧ་སི་འཚེ་བ་ལའོ། །ན་སྒྲའོ། །ཀྲྀ་ཧ་ལས་ཨིཊ་བི་ཀ་ར་ཎ་ལས་སོ་་་ཞེས་པས་ཨི་ཊ་ཨཱ་ག་མའོ། །ཊ་ཡིག་ནི་ཨཱ་ག་མའི་དོན་ཏོ། །ཧའི་རྙའོ་ཞེས་་་་པས་དྷ་ཉིད་དོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་དྷའོ་་་་ཞེས་པས་དྷ་ཉིད་དོ། །ཏ་སྡེའི་ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །ལྷ་ལ་ཌྷའི་དབྱི་ཞིང་ཨུ་པ་དྷའི་རིང་པོ་འང་ངོ༌། །ཨ་ཡིག་གི་ཨིའི་:ཡི་གེ་ལ་ཨེ་ཏེ་ཕ་རོལ་ཡང་དབྱིའོ། །ཀྲྀ་ཎེ་དྷི། ཏྲཱནྡྷ། །ཀྲྀ་ཧནྟི། ཀྲྀ་ཎེ་ཀྵ། ཀྲྀནྡྷ། ཏྲཱནྡྷ། ཀྲྀ་ཎེཛྫི། ཀྲྀ་ཧཱ། ཀྲྀཛྫཱ། ཧི་སི། ཨི་ཏ་རྗེས་སུ་འབྲེལ་བ་ཉིད་ཀྱི་ཕྱིར་ན་ཨཱ་ག་མའོ། །ན་སྒྲའོ། །ན་སྒྲ་ལས་ཀྱང་བི་ཀ་ར་ཎ་ལས་སོ་ཞེས་པས་ཨ་ནུ་ཥངྒ་དབྱིའོ། །ཧི་ན་སྟི། ་་:ཧི་སནྟི། ཧི་སནྟི། ཧི་ནསི། ཧི་ཧི་སྦྲ། ཧི་ནསྨིཾ། ཧི་ཧི་སྨ། ཨུནྡི། སླས་པ་ལའོ། །སྔ་མ་བཞིན་ན་དབྱིའོ། །ཨུ་ནཏྟི། ཨུཙྪཱཨུནྡནྟི། ཨུ་ནཉྩི། ཨུནྡྷཨུནྡྷ། ཨུ་ནངྒཱི། ཨུནྡཱ། ཨུདྡྷ། །ཨནུ༑གསལ་བ་དང་འཆབ་པ་དང་འགྲོ་བ་རྣམས་ལའོ། །བི་སྔོན་དུའོ། །བྱ་ན་སྟི། བྱཎྞ། བྱེཎྜིཏྟི། བྱ་ནཀྟི། བྱཎྚབྱཱཏྨ། །བྱ་ནཛྫི། བྱཥྛབྱམྦྷཏནྟུ། །བསྐུམ་པ་ལའོ། ་། ཏ་ནཀྟི། ཏཏྟཱཏཉྩནྟི། ཨོ་བི་ཛཱ། འཇིགས་པ་དང་གཡོ་བར་བྱེད་པ་དག་ལའོ། །བི་ནཀྟི། བིཏྟཱབཉྫནྟི། བི་ནཏྟི། བིནྡྷ། བིནྡྲ། བི་ནÉ། བིཥྛབིཥྛ། དྲྀ་ཛཱ་འདོར་བ་་་—ooOG།
