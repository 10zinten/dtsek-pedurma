སྣ་ཚོགས། ཏི་ཥཊཱི་ད་ནྟི། ནསཀྲ་ཚ་ཏེ། ནཱ་:རྡ་སྱ་པ་ད་སྱ་ཚ་ཏེ་པ་རེ་ས་གཱ་ག་མོ་བྷ་བ་ཏི། ཏབསྔལ་བྱེད། ན་ལས་ཚ་ཏ་ལ་ས་ཀའོ། །ན་མཐའི་ཚིག་ལས་ཚ་དང་ཏ་ཕ་རོལ་ལས་སཱ་ཀྲ་ཨཱ་ག་མ་རུ་འགྱུར་རོ། །ཊི་རྐཏཱི་བཱ་རྱ་ཏ་ཡོརྦ་ལྟ་བྱོ། མ་རཱ་ཛན་ཙི་ཏྲ། རཱ་ཛི་ཉིཏྲ། འབྲོང་ཊི་ཏྲ་ཀརྫི་དག་ཐོག་མ་དང་ཐ་མ་དག་ཏུའོ་ཞེས་བཤད་དོ། །གསལ་ཞིང་བཀྲ་བ་སྟེ། །བྷ་བཱན་ཏ་ནོ་ཏི་བྷ་བཱ་སྟ་ནོ་ཏི། ཤེ་ཙཀཱ། ་ནཾནྟ་སྱ་པ་ད་སྱ་ཤེ་པ་དང་ཏཁྱོད་ཐགས་ཐོགས་ཤིག་ཁྱོད་ཡངས་པར་གྱིས། ཤ་ལ་ཙ་ཀྲ་ཡང་ནའོ། །ན་མཐའི་ཚིག་ནི་ཤ་ཕ་རོལ་ལ་ཡང་རེ་བཱ་ཙ་གཱ་ག་མོ་བྷ་བ་ཏི། བྷ་བཱན་ཤུ་རཿབྷ་བཱིཉྩཱ་རཿབྷ་ཊཱཉྩཱ་ར། བྷ་བཱབྷཱུ་རཿན་ཙཀྲ་ཨཱ་ག་མ་རུ་འགྱུར་རོ། །ཁྱོད་དཔའ་བོ། ཀ་མ་ཁྱོད་དཔའ་བོ། ་་་། ན་ཌོ་ཧྲ་སཱ་དཱིསཱ་རེ། ང་ཀཱ་ར་ཎ་ཀཱ་ར་ན་ཀཱ་རཱ་ཧྲ་སཱ་རུཏྟ་རཱ་དཱིགྷ་བོ་ཏི་སཱ་རེ་བ་ཐུང་ངུའི་ང་ཎ་ན་དབྱངས་ལ་གཉིས་སོ། །ང་ཡིག་དང་ཎ་ཡིག་དང་ན་ཡིག་ཐུང་ངུ་དབྱངས་ཕ་རོལ་དང་ཚིག་རེ་པཱ་དཾ་ཏེ་ཙ། པྲ་ཏྱདྲ་ཨི་དཾ། པྲ་ཏཇི་དཾ། སུ་གཎ་ཨི་དཾ། སུ་གནྡྷི་དཾ། མཐའ་ལ་ཡང་ཕྱིས་གཉིས་སུ་འགྱུར་རོ། །སོ་སོར་མཆོད་པ་པོ་འདི། གཡོག་བཟང་འདི། །རཱ་ཛ་ན། ཨི་ཧ། རཱ་ཛནི་ཧ། ཚཧྲ་སཱ་དུཏྟ་རཱ་ཚ་ཀཱ་རོ་དྲིཌྷ་བ་ཏི། ཁ་སེ་ཙ་་་གསལ་བ་འདི། ཚ་ཡང་ངོ༌། །ཐུང་ངུའི་ཕྱི་མར་ཚ་ཡིག་གཉིས་སུ་འགྱུར་རོ། །ཁ་ས་ལ་ཛྷ་ས་རྣམས་བཛྷ་སཱ་ནཾ། ཛྷ་སཱ་ནཾ་ཁ་ས་བ་རེ་ཙ་པ་བྷ་བནྟི། ཏ་བ་ཚ་ཏྲ། ཏ་བཙྪ་ཏྭཾ། ཀྭ་ཁཙ་བའོ། །ཛྷ་ས་རྣམས་ཁ་ས་ཕ་རོལ་ལ་ཙ་སར་འགྱུར་རོ། །ཁྱོད་ཀྱི་གདུགས། འགའ་། ཁོ་ཙིདྡྷིརྒྱུ་ད་པ་བལྟ་བྱསླེ་ཚ། མེཙྪ། ཧྲི་ཚ། ཧཱིཙྪ། །མོ་ནུ་སཱ་རཿཞིག་ཏུ་རིང་པོ་ལས་ཀྱང་བཤད་དོ། །ཀླ་ཀློའོ། །།ངོ་ཚ་བྱེད། །མ་རྗེས་སུ་ང་རོའོ། །་་མ་ཀཱ་ར་སཱུ་ནུ་སཱ་རོ་བྷ་བཱ་ཏི་ཧ་སེ་པ་ར་པ་དཱ་ནྟེ་ཙ། ཏམ་ཧ་ས་ཏི། ཏཾ་ཧེ་ས་ཏི། ་མ་ཡིག་རྗེས་སུ་ང་རོ་རུ་འགྱུར་ཏེ་ཧ་སེ་ཕ་རོལ་ལ་དང་ཚིག་མཐའ་ལ་ཡང་ངོ༌། །ཁྱོད་དགོད་པར་བྱེད། །།
