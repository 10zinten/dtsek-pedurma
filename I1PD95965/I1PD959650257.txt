སྣ་ཚོགས། ན༽ཌི་རྒྱན་ལའོ། །ཨི་ཡིག་ཨི་དིཏ་ཀྱི་ནུ་མའོ་ཞེས་པས་རྟགས་དོན་ནོ། །ནུ་མའོ། །ཊུ་དྷཱཉ་འཛིན་པ་ལའོ། །ནི་སྔོན་དུའོ། །འདི་རྣམས་ཀུན་འགྱུར་རོ། །ཀ་་ཡིག་ཨཏ་ཨེང་འགེགས་པའི་དོན་ནོ། །ས་མེད་པའི་ཡུ་བུའི་ཨ་ན་ཨ་ཀ་དག་གོ་་་་ཞེས་པས་ཨ་ན་རུ་བསྒྱུར་རོ། །བྱིངས་ཀྱི་ཨིཏའོ་ཞེས་པས་ཨི་ཉིད་དོ། །རིག་གི་ཨཎ་ལས་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད་དོ། །ཎ་ཉིད་ཀྱང་ངོ་། །ཀི་ར་ན་:པྲ་བྷཱི་མཆུ་བྱུང་ལས་ཨུཏའོ་ཞེས་པས་ཨུ་ཉིད་དོ། །།ར་ཕ་རོལ་ཉིད་དོ། །བྱིངས་ཀྱི་ར་བ་དག་ལ་ཨཙ་མེད་པ་ལ་ཨིཀ་གི་རིང་པོའོ་ཞེས་པས་:རིཏྲ། པ་ེཉིད་དོ། །པཱུ་མ་ཧབྲི་ཛན། རྒྱན། ཨནྟ་རཱི་ཧཱུཾ་མཎྜ་ནཾ་:བྷུ་ཥ་ན་ན་དྷཱ་ན་གཏེར་ནི་དྷིདྲི་ཏིའི་དྷི་ཥ་ཡང་ངོ༌། །དཾ་ཥཱ་སྤྱི་བརྟོལ་ལའོ། །ཨཱ་ཡིག་ཨཱ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ནོ། །འདི་ལས་སྐྲུན་འགྱུར་རོ། །ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །དྷི་ཥ་ཎཱ་བུདྡྷིཧ་ནིའི་ཛ་གྷ་ཡང་ངོ༌། །ཧ་ན་འཚེ་བ་ལའོ། །འདི་ལས་ཀུན་འགྱུར་ཞིང་། ཛ་གྷ་ཡང་ངོ་ཞེས་པས་ཛ་གྷ་རུ་བསྒྱུར་རོ། །ཛ་གྷ་ན་ཀ་ཊི་པྲ་དེ་ཤོ་:བྲ་ཐ་སི་དྷ་པཱི་རྣམས་ལས་ནའོ། །ཁྲ་དྷ་ཞེས་པས་་་མདོའི་བྱིངས་སོ། །བ་ས་ངེས་པར་གནས་པ་ལའོ། །ཏུ་དྷཱཉ་འཛིན་པ་ལའོ། །པཱི་རྫོགས་པ་ལའོ། །འདི་རྣམས་ལས་ན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །བྲ་:དྷ་ཨ་དྷི་.ཉི་མའོ། །བསྨ.རིན། མཱུ་ཊཱཔའོ། །དྷཱ་ནཱ་སྤོས་ཡ་བ་བི་ཀ་རཿནས་ཀྱི་རྣམ་ཀྱི་འགྱུར། ཨཏ་ཨེང་ར་ཕ་རོལ་ཉིད་དོ། །ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །པཎྞཔ་ཏྲ.ལོ་མ། ་ཀྲྀ་བྲི་ཏཱཾ་སཱ་བི་སི་དྲུ་རྣམས་ལས་ན་ནའོ། །ཌཱ་ཀྲྀ་ཉ་་་བྱེད་པ་ལའོ། །བྲོཉ་སྒྲིབ་པ་ལའོ། ་། ཏཱཾ་རྐྱལ་བ་ལའོ། །སཱ་པ་ཉལ་བ་ལའོ། །ཎི་འཆིང་བ་ལའོ། །ཎྜི་བུ་དང་ཋཀ་མ་ཡིན་པའི་ཥའི་སའོ་ཞེས་པས་ས་ཉིད་དོ། །དྲུ་འགྲོ་བ་ལའོ། །འདི་རྣམས་ལས་ནན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་་་—།
