བསྟན་འགྱུར། །སྣ་ཚོགས། ༼ཙིའི་ཡང་དབྱངས་གཅིག་པའི་བྱིངས་རྣམས་སོ། །བྱིངས་ཀྱི་ཨཱཏའི་མཐའ་རྣམས་ཀྱི་་་ཡང་རྗེས་སུ་འབྲེལ་པ་བསྒྲུབ་པར་བྱའོ། །དཱ་རི་དཱ་ཏི། ད་རི་དྭ་ཏིའི་ཨི་ཡིག་གོ་ཞེས་པས་ཨི་ཡིག་གོ། །།ད་རི་ཏཱི་ཏ། ཌ་ཨ་བྷ་སྟ་རྣམས་ཀྱི་ཨཱ་ཡིག་གོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། །ད་རི་དྷཱ་ཏི། ད་རི་དཱ་སི། ད་རི་དཱི་ཐ། ད་རི་ངྒི་ཐ། ད་རི་དྲི་མི། །ད་རི་དྲི་བ། ད་རི་དཱི་མ། ཙ་ཀཱ་པྲི། འབར་བ་འགོདི། ཙ་་་ཀཱསྟི༑ཙ་ཀཱསྟི། ་ཙ་ཀཱ་ས་ཏི། ཙ་ཀཱསི། ཙ་ཀཱིསྦ༄༅། ་ཙ་ཀཱསྦྲ། ཙ་ཀཱསྨི། །ཙ་ཀཱསྦ། ཙ་ཀཱསྨཱ། ཤ་སུ། རྗེས་སུ་སྟོན་པ་ལའོ། །ཤསྟི། ཤ་སིའི་ཨུ་བ་དྷཱའི་ཨཎ་དང་གསལ་བྱེད་དག་ལ་ཨིཏའོ་ཞེས་པས་ཨཱ་ཡིག་གི་ཨི་ཉིད་དོ༑། ཤཱ་སི་བ་སི་གྷ་སི་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །ཤི་ཐཱ། ཤཱ་ས་ཏི་ཤཱསི། ལ་:ཤཱིཊྛ། བྷིཥྛ། ཏིསྨི། ཤྲཱིཥྛ། ཤྲཱི༑་བྲྀཏའོ། །ཛེ་གྷ་ལ་སོགས་པ་ལྔའོ། །།པ་རསྨིའི་བཤད་པ་རྣམས་སོ༑། ཙ་གླིང༌། གསལ་བར་བརྗོད་པ་ལའོ། ་། ང་ཡིག་ནི་ཨཱཏྨ་ནེའི་ཚིག་གི་དོན་ནོ། །ཨི་ཡིག་ནི་ཙཀྐིའི་ཁྱཉའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨན་་་འཇིག་གོ། །བརྩེགས་པ་དང་པོའི་ས་ཀ་དག་གི་མཐའ་ལའང་ངོ་ཞེས་པས་ཀ་ཡིག་དབྱིའོ༑པ། ཥ་ལས་ཌ་སྡེ་ཉིད་དོ། །ཨཱ་ཙཥཱ། ཨཱ་ཙ་གཱ་ཏ། ཨཱ་ཙ་གྷ་ཏེ། ཀ་ཡིག་དབྱིའོ། ་། ཥ་དྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །རྒྱུ་མཚན་ལས་རྐྱེན་དང་རྣམ་འགྱུར་དང་ཨཱ་ག་མ་ལ་གནས་པའི་སའི་ཥ་ཉིད་དོ། །ཨཱ་ཙ་གླེ། ཨཱ་ཙ་གྷཱ་ཐ། ཀ་ཡིག་དབྱིའོ། །ཥ་ལས་ཊ་སྡེ་ཉིད་དོ་ཞེས་པའི་དྷའི་རྙའོ། །དྷུ་ཊ་རྣམས་ཀྱི་གསུམ་པའོ་ཞེས་པས་ཥ་ཡིག་ཌ་ཡིག་གོ། །།ཨཱ་ཙནྡྷེ། །ཨཱ་ཙཀྵེ། ཨ་ཙཀྵ་ཧེ། །ཨཱ་ཙཀྵ་ཧེ། ཨཱིར་འགྲོ་བ་དག་ལ་སྟེ་མགུལ་བ་ལ་ཡང་ངོ་། །ཨན་འཇིག་གོ། །ཨིནྟེ། ཨཱི་རཱ་ཏེ། ་ཨཱི་ར་ཏེ། ཨཱི་ར་ཥ། ཨཱི་རཱ་ཐ། ཨཱིདྷེ། །
