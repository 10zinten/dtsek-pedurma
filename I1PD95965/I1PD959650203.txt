བསྟན་འགྱུར། སྣ་ཚོགས། །ནིའི་ཡང་ངོ༌། །ཤིང་ལས་ཕུཊ་ཡང་ངོ༌། །ཚ་དི་ལས་ནུཊ་ཡང་ངོ༌། །ཨ་མི་ལས་བྷུ་ཀ་ཡང་ངོ༌། །ཨརྟི་ལས་ཨུཏ་ཡང་ངོ༌། ་། ཤུ་ཊ་ཡང་ངོ༌། །ནུ་ཊ་ཡང་ངོ༌། །ཡུ་ཊ་ཡང་ངོ༌། །ཨུཎ་ལ་སོགས་པར་རྐང་པ་གསུམ་པ་རྫོགས་སོ། །འོ་ན། །ཟླ་ཨུཎ་ལ་སོགས་པའི་མདོ། །སློབ་དཔོན་ཆེན་པོ་ཙནྡྲ་གོ་མིས་མཛད་པ་་་རྫོགས་སོ། །།རོ། །།།།།།།།།།དེ་རྗེས་ཆུ་དཔལ་ཨེའི་ཆོས་གྲྭ་ཆེན་པོར་ལོ་ཙཱ་བ་ནམ་མཁའ་བཟང་པོས་བསྒྱུར་བའོ། །།།ਵੀਇਹਕਰੰਸਸਬਉਸਉਸਨੂੰਸਾਲਖਾਲਸਰདམ་པ་ཁག་ལ་བསྡུར་མཆན། །དེ་དདོ་ཤ(མཚན་བྱང་འདི་«སྡེ་»ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ཙནྡྲ་པའི་ཨུཎ་ལ་སོགས་པའི་མདོ་བཞུགས་པ་་་་་ལེགས་སོ། །ཞེས་འཁོད། «པེ་གྲོ། «པེ་»དི། «པེ་ཨུའོ། «པེ་»«སྣར་མའི། །«པེ་»«སྣར་»ཨུཏ། «པེ་»ཛི། འི«པེ་»«སྣར་ཐེན། །དང་པོ«པེ་»«སྣར་ཨཱུརྨི། @«པེ་»«སྣར་»སྤྱི། «པེ་»«སྣར་ཨཉྫི་ལ་ཨ་ལི། «པེ་»«སྣར་»དཾ། ་«པེ་ལས། &«པེ་བྲྀཏྲ། «སྣར་བྲིཏྲ། །«པེ་»བེ། ་ ོ«པེ་»«སྣར་»ཏནྡི། @«པེ་»མུཊ། «སྣར་»མུན། ཏེ«པེ་»ཌིའོ། «པེ་»«སྣར་»ཀྲ༑«པེ་»«སྣར་+པ། «པེ་»ཤི། ཉེ«པེ་»«སྣར་»ཀཱ། «པེ་»ཨུ། ལྕེ«པེ་པ་ཏི། ཀ་(«པེ་ནུན། «པེ་»«སྣར་»ནཱའོ། «པེ་»«སྣར་»ཝཱ། «པེ་»«སྣར་»ཏ། མུ«པེ་»«སྣར་»དོ༑«པེ་»«སྣར་གཾ། «པེ་»«སྣར་»ད། «པེ་»ཏན་ནོ། །«སྣར་»ཏན། «པེ་»«སྣར་རུ། :ཚེ«པེ་»«སྣར་»ར། །ཀྱེ«པེ་»«སྣར་»བྷཱི། །།(ལེ«པེ་»ཏི། ་ཉེ«པེ་»ཧཱུྃ། ལྕེ«པེ་»《སྣར་རཱ། །ཚེ«པེ་ནའོ། «སྣར་ནིའ། «པེ་»«སྣར་»དྷཱ། «པེ་»«སྣར་»འོ། ®«པེ་»མནྡི། །«པེ་»«སྣར་»ཀྲྀ། །«པེ་བྲི། «པེ་»«སྣར་»ཨཱ། «པེ་»«སྣར་མ། «པེ་《སྣར་ཨཱ། «པེ་»«སྣར་པཱུང་། ཚེ«པེ་»«སྣར་བུ། །«པེ་»«སྣར་ལྕི། «པེ་ཀྲི། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་»ཀའོ། །«པེ་»«སྣར་རད། རི«པེ་»སིའོ། «སྣར་»སིའོ། @«པེ་»«སྣར་»ཡས། «པེ་ད། །ལོ་ཀ་བ། ཏཊ་ཀ་ཁ། །ག་ང་བ་མ་འགྲོ་བ་༽—།
