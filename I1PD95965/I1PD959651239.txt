།དྲག་པོ་དྲག་ཤུལ། ཞེས་པ་དང༌། །མི་སྡུག་པ་དང༌། བྲི་བནྡྷ། །དངངས་སྐྲག་། ཉམ་ང་། བག་ཚ་དང༌། །།ཕམ་་་་་པར་བྱས་དང་འཇིགས་པའོ། །ཡིད་ཀྱི་རྣམ་འགྱུར་དགའ་བ་དང༌། །ཞུམ་དང་ཁྲོ་སོགས་ཀྱིས་བསྐྱེད་པ། །ཉམས་སུ་མྱོང་དང་རྣམ་འགྱུར་རྟོགས། །དེ་ཡང་་་ཉམས་མྱོང་ལོངས་སྤྱོད་རྫོགས། བ་། ཐོབ་པའི་བདག་ཉིད་ཅན་དུ་འགྱུར། །དེར་ནི་མིག་བཞི་ལས་བྱུང་བ། །སྨིན་མ་འདེགས་དང་འཛུམ་རྣམས་འཁོར། །བཀྲ་བར་ཉམས་མྱོང་དག་ལ་སོགས། །ལོངས་སྤྱོད་རྫོགས་པའི་བདག་ཉིད་འདོད། །མངོན་འདོད་དེ་བཞིན་བསམ་པ་དང༌། །རྗེས་སུ་དྲན་པའི་ཚོགས་སུ་འདོད། །དེ་བཞིན་སྐྱོ་དང་རྣམ་པར་སྨྲ། །སྨྱོ་བ་དང་ནི་ནད་ཉིད་དང་། །ཛ་བྷ་འདུལ་་བྱེད་ཀྱང་ཞེས་པ། །ཐོབ་པའི་བདག་ཉིད་བཅུ་ཡིན་ནོ། །དགོད་པ་དགའ་་་་བར་བྱེད་པ་སྟེ། །དེ་བཞིན་གཉི་ག་སྤྲོ་བའོ། །སྦེད་དང་ངོ་མཚར་ཞེས་བྱ་བ། །དངོས་པོ་བཞག་པར་རབ་ཏུ་བསྒྲགས། ྋ་། དབང་དུ་བྱེད་པ་རྣམ་ཤེས་ཀྱིས། །དངོས་པོའི་ལམ་ལ་སྐྱེས་པའི་ལས། །ཡིད་ཀྱི་མདུད་པ་ངར་སེམས་ཏེ། ང། ང་་་་རྒྱལ་སེམས་ནི་མཐོ་བའོ། །སྙེམས་དང་མངོན་པའི་ང་རྒྱལ་དང༌། །ངར་་འཛིན༑རྒྱགས་དང་། དྲེགས། མངོན་ཞེན། །བསྐྱིངས་་པ་དང་ནི་་་ཡང་དག་ཤུགས། །ཀུན་རྩོམ། ཀུན་འབེབས། ཡང་དག་བསྐོར། །རང་་་དབང་འཇུག་དང་། ཀུན་ནས་ལྡང༌། །མངོན་པར་ལྡང་དང་། ཞེ་སར་་་བྱེད༈། །གུས་བྱེད། གུས་པར་ཉན་པ་དང༌། །ཡོངས་སུ་སྦྱོང་དང་། །
