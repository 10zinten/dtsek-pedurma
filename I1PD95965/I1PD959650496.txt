ལ་ལ་ད་ན། །ནང་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། _ཅོ་ནེ། སྡེ་ཚན། །སྣ་ཚོགས། སྒྲ་སྐོར། པོད་རྟགས། །ནོཚན་གྲངས། སྐབས། །ལྡེབ། བ—ན། ན?བ་ནན་ཕྲེང་། ཚེག་ཁྱིམ། །་བར་བ་བར་བ། ་ཁབློ་དང་། ནང་གི་དུག་་་་ང་བ་གལ་བ་གསལ་་བ་་ངའ་བའི་གི་བྱ་བ་དང་ཁ་ལ་བདེ་བ་ཤོར་བ་ག་གའི་ཞལ་འཚོ་ཊ་བའི་ལུག་བབ་པ་ཡིན་ག་བགྱི་བ་ནི་ཁྲུས། །སྔོན་པ། །ཀ་ར་ཀྱང་དེ་ལ་རྒྱའི་ས་རྒྱུའི་གད་ཟཊ་བྱ་བ་མཁས་ག་ལ་བ་དེས་ང་ལ་པ་གས་པ། ཤར་ལ་བཀལ་བ་དང་གས་པ་ལས། དོན། ཚོ་ལ་མོ་རང་རང་སར་ཞི་བ་དམ་པ་ནི། བ། ལྷག་མ་དྲུག་དང་དེ་ཌཊ་པས་ད་བར་ཤ་ཁ་ལ་།
