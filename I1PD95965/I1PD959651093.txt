བསྟན་འགྱུར། སྣ་ཚོགས། ༼ཝི?ཙཆ་མ༑ༀ་ཌི། དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ་ཞེས་པས་་འཕེལ་བའོ། ་། ཨིཊྚ། ཨི་ཌ་ཏཱམ། ཨི་ཌ་ཏ། ་་ཨི་ཟླཨི་ཤ་། ཨིཊྚ། ཨེ་ཤཱ་ཏཱམ། ཨེ་ཤ་ཏི། :ཨིཥྛཨེ་ཥ་ཐཱམ། ཨི་ཤ་དྷཱམ། ཨཱ་ས་ཨཱ་སྟེ། ཨཱ་སཱ་ཏཱཾ། ཨཱ་ས་ཏ། ཨཱང་ལས་ཤཱ་ས། ཨཱ་ཤཱསྟ། ཨཱི་ཤཱ་སཱ་ཏཱམ། །ཨཱ་ཤཱ་ས་ཏཱ༑ཨཱ་ཤཱསྦ?བ་ས། ཨ་བསྟ། ཨ་བ་སཱ་ཏཱམ། ཨ་བ་ས་ཏཱ། ཁྱུང༌། ཨ་སཱུ་ཏ། ཨ་སུ་བཱ་ཏཱམ། །ཨ་སུ་བ་ཏ། ཤིང་། ཨ་ཤེ་ཏ། ཨ་ཤ་ཡ་་ཏཱམ། ཨ་ཤེ་ར་ཏ། །ཨིང་། ཨ་དྷི་ཏ། ཨ་དྷ་ཡཱ་ཏཱམ། ཨ་དྷཱ་ཡཱཏ། དྲི་དྷིང་། ཨ་དཱི་བྷཱི་ཏ། ཨ་དྷི་དྷཱ་ཏཱམ། ཨ་དཱི་དྨ་ཏེ། སྙིང༌། ཨ་ཧྥུ་ཏ། ཧྥུ་པཱ་ཏཱམ། ཨ་ཧྥུ་བ་ཏ། དྭི་ཥ། ཨ་དྭེ་ཊ། ཨ་དིཌ་ོམ། བི་ད་སོགས་ཉིད་ཀྱི་ཕྱིར་་་ཆོགས་སུ་བྱས་པའི་རྟག་ཏུ་མ་ཡིན་ནོ་ཞེས་པས་ཡང་ན་བར་འགྱུར་རོ། །ཨ་དྭེ་ཥ་ན། ཨ་དྭེ་ཏུ། ཨ་དྭཊ། ཨ་དཱིཋམ། ཨ་དཱི། ཨ་དྭེ་ཥམ། །ཨ་དི་ཥཱ༑ཨ་དཱི། ཨ་དཱིཀྵ། ཨ་དཱི་ཥཱ་ཏཱམ། ངོ་། ཨ་དཱི་ཥཏེ། ཨ་དཱིཥྛདུ་ཧ། །ཕྱིས་པ་ལའང་དི་སི་དག་ལའོ་ཞེས་པས་བཞི་པ་ཉིད་དེ་ད་ཡིག་གི་དྷ་ཡིག་གོ༑། ཨ་དྷོ་ཀ་། ཨ་དུ་གྷཱམ། ཨ་དུ་ཧན། ཨོ་དྷོཀ་་་། ཨ་དུགྷམ། ཨ་དོ་ཧམ། ཨ་དུ་ཧྭ། །ཨ་དུ་ཧྥ། ་ཨ་དུགྷ། ཨ་དུ་ཧཱ་ཏཱམ། །ཨ་་དུ་ཧཏ། ་ཨ་དུདྡྷེ། ཨ་དུ་ཧཱ་ཐཱམ། ཨ་དུབྷཱམ། འདི་ཧ། །ཨ་བྷེ་ཀ། ཨ་དིབྷཱམ། ཨ་དི་ཧན། ཨ་དིགྷ། ་ཨ་དི་ཧ་ཏཱམ། ཨ་དི་ཧ་ཏ། ལི་ཧི། ཨ་ལེཊ། ་ཌྷ་ལ་ཌྷ་དབྱིའོ། །ཨུ་པ་དྷཱ་རིང་པོ་ཉིད་ཀྱང་ངོ༌། :། ཨ་ལཱི་ནྣཱམ། ཨ་ལི་ཧན། །ཨ་ལི་ལྷ། ཨ་ལི་ཧཱ་ཏྭཾ། ?ཨ་ལི་ཧཱ་ཏ། རྒྱལྟ་ནཱི་ལའང་ངོ་ཞེས་པ་འཕེལ་བའི་དམིགས་བསལ་ལས་ཡོན་ཏན་རྟག་ཏུའོ་སྔོཎྜ་ཏ། པྲཎྜ་ཏཱམ། པྲཆུ་བན། སྲོརྟུསྲོརྫུ་ཏ། ་སྲོཌུ་བཱ། ཏཱམ། །སྲོརྫུ་བ་ཏ། བྲུཉ། །
