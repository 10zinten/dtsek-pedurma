བསྟན་འགྱུར། སྣ་ཚོགས། པིའིདང་དཱས་དང་དྲུ་ཧ་འཚོ་བ་ལའོ། །ཏིཀྐོ་ཏི། ཏིགཽ་ཏི། སམྦྷོ་ཏི། རིམྨོ་ཏི། ཨི་ཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ཀྱི་ཕྱིར་ན་ཨཱ་ག་མའོ། །དྷི་ནཱི་དང་ཀྲི་ཎི་དག་གི་དྷི་ཧ་དང་ཀྲྀ་ཡང་ཞེས་པའི་བཤད་པར་བྱའོ་ཞེས་པའི་ཚིག་ལས་ཁྲིའི་དངོས་པོའོ། །ཀྲྀ་ནོ་ཏི། ཙི་རི་ཎོ་ཏི། ཛི་རི་ཎོ་ཏི། དཱ་བསྣོ་ཏི། དྲུ་སྒོ་ཏི། ་ཉི་དྷཱི་ཏི། སྤྱི་བརྟོལ་་ལའོ། །བྷོཊྛོ'ཏི། དབྷཱུ། རངངན་གཡོ་ལའོ། ཨ་ནུ་ཥངྒ་་དབྱིའོ༑། ད་བྷཱ་ཏི། རཾ་དྷུ་འཕེལ་བ་ལའོ། །རི་བྷཱུ་ཏི། དྷི་བི་དང་དྲི་པ་མཛའ་བ་ལའོ། །ཨིཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ཀྱི་ཕྱིར་ན་ཨཱ་ག་མའོ། །དྷི་ནཱི་་དང༌། ཀྲྀ་ཎི་དག་གི་དྷི་དང་ཀྲྀ་་ཡང་ངོ་ཞེས་པས་དྷཱའི་དངོས་པོའོ། །།དྷི་ནོ་ཏི༑ཏྲི་སྣོ་ཏི། པ་རསྨིའི་ཚིག་གོ། །ཨ་ཤུ་ཁྱབ་པ་ལའོ། ཟླ་། ཨ་ལྦ་ཏེ། ཌི་གྷ་ཀུན་ནས་འགྲོ་བ་ལའོ། ལ་། ཋི་ཏེ། ཨཱཏྨ་ནེའི་ཚིག་དག་གོ། །།ཐུ་ཐོས་པ་ལའོ། །ཐུའི་ཤི་ཡང་ངོ་ཞེས་པས་ནུའི་རྐྱེན་དང་ཤི་རུ་བསྒྱུར་བ་ཡང་ངོ༌། །ཤི་ཎོ་ཏི། སོ་ལས་ག་མི་ཞེས་པ་ལ་སོགས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །སཾ་ཤི་རྐྱ་ཏེ༑ཨ་ཁྱུ་སོགས་པ་ལ་ཡང་ངོ༌། །སྦྱོར་བ་ཐ་དད་པ་ལས་སོ། །:ཨ་ཀཱ་ཏི་ལས་ཡང་ནའོ་ཞེས་པས་ཡང་ནུའི་རྐྱེན་ནོ། །ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །ནའི་ཎའོ། །ཨ་བྷྱཱ་ཏི། ཨ་ཀཱ་ཏི། ཏཁྱུ་དང་ཏཱཁྱུ་ཕྲ་མོར་བྱེད་པ་ལའོ། །ཏཁཱ་ཏི་ལས་ཀྱང་ངོ་ཞེས་པས་ཡང་ན་ནུའི་རྐྱེན་ནོ། །ཏདྡྷོ་ཏི། ནིར་སྔོན་དུའོ། །རའི་རྣམ་པ་བཅད་པའོ། །ཏའི་？ཐ་ལ་སའོ་ཞེས་པས་རྣམ་པར་བཅད་པའི་་་་ཡང་ས་ཡིག་གོ། །ནི་སྟ་ཀློ་ཏི། ནི་སྟ་ཀཱ་ཏི། ཕྲ་མོར་བྱེད་པ་ལ་ཞེས་པའི་་ཚིག་གི་ཕྱིར་གཞན་ལ་མི་འགྱུར་རོ། ་། སཾ་ཏ་ཀྲུ་ཏི་བགྷི། སློབ་མ་ལ་བརྒྱད་་བཀག་བྱེད་དོ་ཞེས་པའི་དོན་ནོ། །ཐུ་ལ་སོགས་པ་བྷུ་ལ་སོགས་པ་ལས་བྱུང་བ་་་འདི་རྣམས་ནུ་སྒྲུབ་པའི་སྐབས་ལས་འདིར་གསུངས་པའོ་ཞེས་པས་ནུ་བི་ཀ་ར་ཎ་་་OO།
