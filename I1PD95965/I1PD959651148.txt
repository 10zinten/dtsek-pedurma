ན་ཡིག་ཨུ་པ་དྷཱ་རྣམས། །སྲནུ། དྷཱནྟུ། སི་པྲ། སི་ཥ་ཏེ། དི་ངྷི་སི་ཥ་ཏེ། དུ་ཧ། །དུ་དྷུཀྐཱ་ཏི། དུ་དྷུ་གྷ་ཏེ། ལི་ཧ། ལི་ལིཀྵ་ཏེ། ་ཨཱི་ཤ། ཨི་ཤི་ཤི་ཥ་ཏེ། ཨཱས། ཨཱ་སི་སི་ཥ་ཏེ། དེ་བཞིན་དུ་དངོས་པོ་དང་ལས་དག་ལ་ཡཎའོ༑། ཁ། ཨའི་དབྱིའོ། །ཙི་ཀིརཱཏེ། ཛི་ཧཱིར་ཏེ། དེ་བཞིན་དུ་གཞན་ལ་་ཡང་བལྟ་བར་བྱ་བ་རྣམས་སོ། །སཔ་མཱི། ཙི་ཀཱིརྙེ་ཏ། པཉྩ་མཱི། ཙི་ཀཱིརྱ་ཏུ། རྒྱས་ནི། ཨ་ཙི་ཀཱནྟ་ཏེ། ཨ་བྱ་ཏ་ནི། །ཨ་ཙིཀཱིརྟི་ཏ། པ་རོཀྵ། ཙ་ཀཱ་ས་དང་ཀཱས་དང་རྐྱེན་གྱི་མཐའ་རྣམས་ལས་པ་རོག་ལ་ཨཱམའོ་ཞེས་པས་ཨཱམའོ། །།།ཨཱམའི་ཀྲྀཉ་རྗེས་སུ་སྦྱར་བར་བྱའོ་ཞེས་པས་ཀྲྀཉ་རྗེས་སུ་སྦྱར་བར་བྱའོ། །ཙི་་་་ཀིཙྩཾ་ཙ་ཀ་ར། ཤཱ་སྟ་ནཱི། ཙི་ཀཱིརྟི་ཏཱི། ་ཨཱ་ཤིཙི་ཀཱིརྟི་ཏི། ་བྷ་བི་ཥཱནྟི། ཙི་ཀཱིརྱ་ཥ་ཏི། ཀྲ་ཡཱ་ཏི་བཏྟིཨ་ཙི་ཀཱིརྱ་ཥྛ་ཏ། །དེ་བཞིན་དུ་ཡུལ་ཐམས་་་ཅད་ལ་བསྟན་པའོ། །ལ། །:ཙེ་ཀྲྀ་ཡི་ཏ་བརྗོད་པར་བྱའོ། །ཡཱ། ཐོབ་པ་ལའོ། །འདས་པའི་དོན་་ཐོབ་པར་བྱེད་པའམ་ཡང་ནས་ཡང་དུ་ཐོབ་པར་བྱེད་པའི་དོན་བརྗོད་པར་འདོད་པ་ལ། ་བྱིངས་ལས་བྱ་བ་ཡང་དག་པར་མངོན་པར་སྡུད་པ་ལ་ཡ་སྒྲ་སྟེ་ཙེ་ཀྲྀ་ཡི་་ཏའོ་ཞེས་པས་ཡ་རྐྱེན་ནོ། །ཙ་ཎོ་པ་རོཀ་ཞེས་སོགས་ཀྱིས་གཉིས་བརྗོད་དོ། །ཐུང་ངུའི་ཞེས་པས་ཐུང་ངུ་ཉིད་དོ། ་། ཨཱ་ག་མ་མེད་པའི་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ལཱ་ཡཱ་ཡ་ཞེས་པ་གནས་པ་ལ་དེ་རྣམས་བྱིངས་རྣམས་སོ་ཞེས་པས་་་བྱིངས་ཀྱི་མིང་ངོ༌། །ཙེ་ཀྲྀ་ཡི་ཏའི་མཐའ་ལས་ཀྱང་ངོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཡཱ་ཡཱ་ཡ་ཏེ། བཱ། བཱ་བཱ་ཡ་ཏེ། འུ་དཱཉ། དཱ་མཱ་གཱ་ཡ་ཏི་ཞེས་སོགས་ཀྱིས་ཨཱི་ཉིད་དོ། །།ཙེ་ཀྲྀ་ཡི་ཏ་ལ་ཡོན་ཏན་ནོ་ཞེས་པས་ཨ་བྷུསའི་ཡོན་ཏན་ནོ། །་། དེ་དཱི་ཡ་ཏེ། དེ་བཞིན་དུ་ཌཱ་དྷཱཉ། དེ་དྷཱི་ཡ་ཏེ། མཱ། མེ་མཱི་ཡ་ཏེ། གི། ཨཱ་།
