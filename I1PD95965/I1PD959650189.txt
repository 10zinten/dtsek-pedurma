བསྟན་འགྱུར། སྣ་ཚོགས། ནི༎་ཟད་ཁ་འཕངས་པས། །སྐྱེས་ཆེན་རྣམས་ཀྱིས་བསྒྱུར་བར་བརྩམས་ཡོད་ཀྱང་། །གངས་ཅན་ཁྲོད་འདིར་རྫོགས་པར་མ་འགྱུར་བའི། །ཨུཎ་སོགས་དུར་སིང་འགྲེལ་པ་རྫོགས་པར་འགྱུར། །བསྒྲུབ་ཚུལ་སྣ་ཚོགས་འོད་ཟེར་འབུམ་ལྡན་པ། །སྒྲ་རིག་སྡེབ་ལེགས་ཉིན་མོའི་དབང་ཕྱུག་ཤར། །།མུན་སྤྲུལ་རང་དགའི་ཀུ་མུད་ཟུམ་པ་དང་། །རྣམ་དཔྱོད་པད་ཚལ་མཉམ་དུ་རྒྱས་པར་གྱུར། །ཉུང་ངུའི་ཚིག་་་གིས་རྒྱ་ཆེན་དོན་སྟོན་པ། །རྣམ་དག་གཞུང་ལུགས་དགོས་འདོད་ཀུན་འབྱུང་་་་འདིས༑་། ལེགས་པར་སྦྱར་བའི་བསྟན་བཅོས་མཐའ་དག་ལ། །རིང་ནས་་་འདྲིས་པའི་ཡིད་ཀྱི་དགའ་སྟོན་འཕེལ། །མང་། དགེ་བའི་གར་ཞ་ཤ་། །ཀ་དེ་བདག་དང་འགར་བ་མ་པ་ཁ་དེ་དག་ང་ན། །བག་བ་མདའ་པ། བབ་བར་ན་འཁྲུབ་ཁང་ངོ་མ། ཁྱོན་བསྡུར་མཆན། ::ཀྱི། བུང་བའི་དབང་དག་ལ་་་་་་་་མཚན་མཚན་བྱང་འདི་«སྡེ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་ཨུཎ་སོགས་ཀྱི་འགྲེལ་པ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་»སྟེངས། །པ། («པེ་»དང་། «པེ་གྲགས། «སྣར་བྲགས། «པེ་»«སྣར་»ཀླགས། &«པེ་ལ། གྱི«སྣར་»འ+པ། «པེ་»«སྣར་»སྒྲུབ། ངའི་«པེ་»རོ། ཉི«སྣར་རྟགས། «སྣར་»ཨུན། «པེ་»«སྣར་»འདི་ཡི། འ«པེ་»«སྣར་»། །«པེ་»«སྣར་ཚར། །«པེ་»«སྣར་»པཱ། «པེ་»«སྣར་»སཱ། ཉེ«པེ་»—པར། «པེ་»«སྣར་»འོ། «པེ་»«སྣར་»བཅད། ་ཏུ། །«པེ་བྱེ། «པེ་དེས། «པེ་པ། «སྣར་»—དེ། «པེ་ཡིགམུ«པེ་»«སྣར་»བཅད། «པེ་»«སྣར་»སཱ། །«པེ་»«སྣར་ཨིཎ། «པེ་»«སྣར་»། «པེ་སྣོད། «པེ་»«སྣར་སྐ། ར་«པེ་»«སྣར་»ཤཱ། Q«པེ་《སྣར་ཐཱི། «པེ་»སྣར་»བྷཾ། «པེ་»«སྣར་ལས་སོ། །«པེ་»«སྣར་»རའོ། «པེ་ཨེ་ཨ་ཡའོ། «སྣར་ཨེ་ཡ་ཨའོ། «པེ་»«སྣར་»བོ། «པེ་»བ། «པེ་»—།
