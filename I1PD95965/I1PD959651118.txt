རཨ་པ་ར་རཱ་དྷུརཱ་ཛི། བྷ་ཛི། བྷ་ཤི། བྷཱུ་ཤི་རྣམས་ཀྱི་ཡང་ན་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨ་ཉིད་དང་ཨ་བྷྱཱས་དབྱི་བ་ཡང་ངོ༌། །ར་རཱ་ཛ། རེ་ཛ་ཏུརེ་ཛུར་རཱ་ཛུརེ་ཛི་ཐ། ར་རཱ་ཛི་ཐ། །བྷེ་ཛེ། བྷེ་ཛཱ་ཏེ། བྷཱ་ཛི་རེ། བ་བྷཱ་ཛེ། བ་དྷཱ་ཛཱ་ཐེ། ་བ་བྷཱ་ཛི་རེ། བྷེ་ཤེ། བྷེ་ཤཱ་ཏེ། བྷི་ཤིར། བ་བྷཱ་ཤ་ཏེ། བ་བྷཱ་ཤི་རེ། ཧྥེ་ཤེ། སྨེ་ཤཱ་ཏེ། བྷེ་ཤི་རེ། བ་ཟླ་ཤེ། བ་ཥཱ་ཤཱ་ཏེ། བ་བྷཱཤ་རེ། གཞན་ལ་མི་འགྱུར་རོ། །ཀཱ་ཤྲཱི། ཙ་ཀཱ་ཤེ། ཐལ་ལའང་ཨིཊ་དང་བཅས་པ་ལ་ཡང་དེ་བཞིན་ནོ། །ཏཱཾ་ཕ་ལ་བྷ་ཛ། ཏྲ་པ། ཤནྡྲ། གནྡྲི། དནྟི་རྣམས་ཀྱི་ཡང་ངོ་ཞེས་པས་ཨེ་ཉིད་དང་ཨ་བྷྱཱ་ས་དབྱི་བ་ཡང་ངོ་།  ོ་། ཏྭཾ། ཏ་ཏཱ་ར། ཏེ་ར་ཏུཏེ་རཿཏེ་རི་ཐ། ཕ་ཝཱ་ལ། ཕེ་ལ་ཏུཕེ་ལུཕེ་ལི་ཐ། བྷ་ཛ། བ་བྷ་ཛ། ་བྷ་ཛ་ཏུབྷེ་ཛུབ་བྷཀྵ། བྷེ་ཛི་ཐ། ཏྲ་པ། ཏྲེ་པེ། ཏྲེ་པཱ་ཏེ། ཏྲེ་པི་རེ། ཏྲེ་པི་ཎེ། ཤནྟྲ། ཤ་ཐནྡྲ། ཨིནྟི་ཐནྡྲི་གནྡྲི་དནྟི། རྣམས་ཀྱི་པ་རོག་ལ་ཡོན་ཏན་མེད་པ་ལའོ་ཞེས་པས་ཨ་ནུ་་ཥངྒ་དབྱིའོ། །ཐེ་ཐ་ཏཿཐ་ཐུདནྟུ། ད་དམྦྷ། དེ་བྷུ་ཏཿདེ་བྷཛཱ། ཛ་ཛཱ་ར། ཛམ། བྷ་མུ། ཏྲ་སི། སྭ་ན། པ་ཎ། སཱུ་མ་་རྣམས་ཀྱི་ཡང་ན་བར་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨེ་ཉིད་དང་ཨ་བྷྱས་དབྱི་བ་ཡང་་ངོ༌། །ཛེ་ར་ཏུཛོ་རཿཛ་ཛ་ར་ཏུཛ་ཛ་རཱུཛེ་རི་ཐ། ཛ་ཛ་རི་ཐ། བླ་མ། བ་བྷ་མ། བྷེ་མ་ཏཿབྷ་མུབ་བྷ་མ་ཏཿབ་བྷ་མུབྷེ་མི་ཐ། བ་བྷ་མི་ཐ། ཏྲ་སཱི། ཏ་ཏཱ་ས། ཏྲེ་ས་ཏཿཏྲེ་སུཏ་ཏྲ་ས་ཏུ་ཏ་ཏྲེ་སུ་ཏྲེ་སི་ཐ། ཏ་ཏྲ་སི་ཐ། །སྭ་ན། ས་སཱི་ན། ཧྲེ་ན་ཏུསྦེ་ནུས་སཱ་ན་ཏཿས་སུ་ནུསྭ་ན་ཐ། །ས་སཱ་ནི་ཐ། ཕ་ཎ། ཕ་ཝཱ་ཎ། ཕ་ཙེཎ་ཏུ་ཕེ་ནུཔ་ཕ་ཎ་ཏུཔ་ཕ་ཎྱཕེ་ཎི་ཐ། ཕ་ཕེ་་—།
