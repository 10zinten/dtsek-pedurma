བསྟན་འགྱུར། །སྣ་ཚོགས། ༼བིཔསCHCLSམཏ༑། ཨཀཱ་མཏ། ་ཨ་མ་དཏ། ཨ་སུ། དཔེར་བརྗོད་དོ། །།ཡ་སུ། ཨ་ཡ་སཏ། ཨ་ཏ་སཏ། ཨད་སཏ། བྷཱནྟུ། ཨ་བྷཱ་ཤཏ། དེ་ལྟ་བུ་ལ་་སོགས་པ་རྣམས་སོ། །དྷཱུཏ་ལ་སོགས་པ་འདི་ཉིད་པ་རས་སྨིའི་ཚིག་ལ་ཨཎ་སྒྲུབ་པའི་ཕྱིར་དྷུཏ་ལ་སོགས་པ་རྣམས་ཀྱི་པ་རསྨིའི་ཚིག་ལ་ཡང་ཡོད་དེ་ཞེས་སོ། །ཨ་དྷུ་ཏཏ། ཨ་ཤུ་བྷ་ཏི། ཨ་རུ་ཙཏ། །ཨ་ཤཱི་ཏཏ། །ཨ་ཀླུ་བྷཏ། ཨ་བྲྀ་་་ཏཏ༑། ཨ་སྱ་དཏ། ཀྲྀ་པཱུ། ར་ཐོས་པས་ལ་ཐོས་སོ། །།ཨ་ཀླུ་པ་ཏ། ཨཱཏྨ་ནེའི་ཚིག་ལ་ཨན་མེད་དོ། །བྱ་བྱོ་ཏིཋ། ཨ་བརྟི། །ཨ་བནྟིཏྠ། ལེཊ༄༅། །རྗེས་སུ་འབྲེལ་པ་རྣམས་གཡུཾ། ཨ་ག་མ་ཏ། པྲ་སིཾ། ཨ་སི་བཏ། །ཤཀྑཾ། ཨ་ཤི་ཀཏ། མུཙྪ། ཨ་མུ་ཙཏ། ཨ་དིའི་ས་ན་དང་ཨ་བྱ་ཏ་ནཱི་དག་ལ་གྷཱ་སློའོ་ཞེས་པས་གྷ་སློའོ། །ཨ་གྷ་ས་ཏ། ལུསྶཾ། ཨུ་ལུ་བཏ། བི་དྷྭཾ། ཨ་བི་དཏི༑སཾ་ལས་གམ་རིཙྪ་ཞེས་སོགས་ཀྱིས་ཨཱཏྨ་ནེའི་ཚིག་གོ། ་། སི་ཙ་ཨཱ་ཤི་ཥ་དག་གི་ག་མི་ལས་དེ་བཞིན་ནོ་ཞེས་པས་ཡང་ན་ལྔ་པ་དབྱིའོ། །ས་མ་གཾ་ཏ། ས་མ་གཾ་སྟེ། ས་མ་ག་སཱ་ཏཱ་མ། ས་མ་གཾ་སཱ་ཏཱམ། །ས་མ་ག་ས་ཏ། ས་མ་ཧཾ་ས་ཏ། པ། ཡ་པ་ཏིའི་བ་སྟིའི་ཞེས་པས་བ་སྟེ་རུ་བསྒྱུར་རོ། །ཨ་བཏྟཏ། ཨརྟི་སནྟི་དག་གིས་བསྟན་ནོ། ཏེད་། ཤཱ་བུ། ཤཱ་སིའི་ཨུ་པ་དྷཱའི་ཨ་ཎ་དང་གསལ་བྱེད་དག་ལ་ཨིཏའོ་ཞེས་པས་ཨི་ཉིད་དོ། །ཤ་སི་པི་སི་གྷ་སི་རྣམས་ཀྱིའང་ཞེས་་་པས་ཥ་ཉིད་དོ། །ཨ་ཤི་ཥཏ། ཛཱ་དང་ཧྲི་དག་གིས་བསྟན་ཏོ། །:སྟ་བྷུ་མདོ་ལས་བྱུང་བའི་བྱིངས་སོ། །ཛཾ་ཤྲཱི་སྟ་མྦྷམུ་ཙུ་མུ་ཙུ་རྣམས་ཀྱི་ཡང་ནའོ་ཞེས་རྣམ་པར་བཤད་པས་ཨཎའོ། །ཨ་ནུ་ཥངྒ་དབྱིའོ། །ཨསྟ་བྷཏ། བི་སྔོན་དུའོ། །པྲ་ཏི་ལས་ཀྱང་སྟམྦྷི་ལས་སོ་ཞེས་པས་ཥ་ཉིད་དོ། །བྱཤྩ་བྷཏ། བྱཧྥ་བྷ་ཏཱམ། བྱཊྚ་བྷན། སྤུ་ཙུ་དང་གླུ་ཙུ་དག་འགྲོ་བའི་དོན་དག་གོ། །ཨ་མྱུ་ཙཏ། ཨ་གླུ་on།
