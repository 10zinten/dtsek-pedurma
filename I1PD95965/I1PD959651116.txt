མ་ཙ་ཀཱ་ར། ཙ་ཀ་ར། ཙ་ཀཱ་རི་པ། ཙ་ཀ་རི་མ། ལས་ལ། ཙ་ཀ་རེ། ཙ་ཀ་རཱ་ཏེ། ཙ་ཀ་རི་ར། གིལ། ཛ་གཱ་ར་ཏཿཛ་ག་རུ:་དབྱངས་ལ་་་ཡང་ནའོ་ཞེས་པ་ལ་ཉིད་དོ། །།ཛ་གཱ་ལ། ཛ་ག་ལ་ཏུཛ་ག་ལུ་ལས་ལ། ཛ་ག་རེ། ཛ་ག་ལེ། ཤྲཱི། འཚེ་བ་ལའོ། །ཤ་ཤ་རཱ། །ཤ་ཤ་ར་ཏུ་ཤ་ཤ་རུ་དྲི་རྣམ་པར་འཇོམས་པ་ལའོ། །བི་དཱ་ར། བི་ད་ད་ར་ཏུ་བི་ད་ད་རུསུཾ། ནི་པཱ་པཱ་ར། ་ནི་པ་པ་ར་ཏུ་ནི་པ་པ་རུ། ཤཱ་ཛཱ་པྲྀ་་་་རྣམས་ཀྱི་ཐུང་ངུ་ཡང་ནའོ་ཞེས་ཁ་ཅིག་གོ། །བི་ཤ་ཤ་ཏུབི་ཤ་ཐུ་བི་ད་དྲ་ཏུ་བི་ད་དྲུནི་པྲ་ཏུ་ནི་པ་བྲུའདིར་ནི་ཤྲཱི་ཛཱ་བཱི་རྣམས་ཀྱི་་་དོན་ལ་ཤཱ་དཱ་པཱ་རྣམས་ཀྱང་འཇུག་པར་བྱེད་དོ། །།དེའི་ཚེ་བྱིངས་ཐམས་ཅད་པ་་མ་ཡིན་པ་ལ་ཨཱ་དབྱིའོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། ?། པ་རོཀྵ་ལ་ཨིཊ་མེད་པ་རྣམས་བརྗོད་པར་བྱའོ། །གྲོགས་བྲི་བྷཾ་སྐུ་དྲུ་ཤུ་ྋསུ་རྣམས་ཉིད་ལས་པ་རོགཱ་ལའོ་ཞེས་པས་ཨིཊ་མེད་པ་ཉིད་དོ། །སཾ། འགྲོ་བ་ལའོ། །ས་སཱ་ར། །ས་སྲ་ཏུ། ས་སྲུས་སནྡྲ༑ས་སྲ་ཐུ་ས་ས། ་ས་སཱ་ར། ་ས་ས་ར།  ི་ས་སྲ་བ། ?ས་སྲི་མ། བྲྀཉ༑བ་བཱ་ར། བི་བྲ་ཏུབྷཾ། བྱེ་ཨད་རྣམས་ཀྱི་རྟག་ཏུ་ཐ་ལ་ལ་ཨིཊ་འོ་ཞེས་པས་རྟག་ཏུ་ཨིཊའོ། །བ་བ་རི་ཐ། །བ་བྲེ། བ་བཱ་ཏེ། བ་བི་རེ། བ་ཤྲཱི་ཥ། བྷཱཉ། ་་བ་བྷཱ་ར།  ོ་་བ་བྷཱ་ཏུབ་བྷུ་བྷུཉ། ཏི་ཊ་ལས་ཕ་རོལ་གྱི་ཨ་གྷོ་ཥའོ་ཞེས་པས་ཏི་ཊའི་དབྱིའོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། ་། ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །ཉྙོ། ཏུབྷུ་བ། ་ཏུཉྩ་བ་ཏུ:ཏུནྟུ་བུཏུ་ཊོ་ཐ། དུ་དྲྭ་བ། དུ་དྲུ་བ་ལཿདུ་དྲུ་བུ༔དུ་ཁྲོ་ཐ། ཐུ། ཤུ་ཐཱ་བ། ཤུ་ཐུ་བ་ལཿཤུ་ཐུ་བུ་སུ་ཐོ་ཐ། ་སུ། །སུ་སཱ་བ། སུ་སྲུ་བ་ཏུསུ་སྲུ་བུ༔སུ་སྲོ་ཐ། རྔུ་ཀྲྀཉ། ཀྲྀཉ་ལས་སུཊ་མེད་པ་ལས་སོ་ཞེས་པས་ཨིཊ་མེད་པ་ཉིད་དོ། །ཙ་ཀཱ་—།
