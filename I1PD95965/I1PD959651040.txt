ཏན་མེད་པ་ཉིད་ཐོབ་པ་ལ། ཤིངའི་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་ཡོན་་་ཏན་ནོ། །ཤེ་ཏེ། ཤ་ཡཱ་ཏེ། ཤེ་ཏེ་ལས་ཨནྟེའི་རི་དང་པོའོ་ཞེས་པས་རེ་་ཕའོ༑། ཤེ་ར་ཏེ། ཤེ་ཥ། ཤ་ཡཱ་ཐེ། ཤེ་ཧྥེ། ་ཤེ་ཡེ། ཤེ་པ་ཧེ། ཤེ་མ་ཧེ། ཨཱིང་ཀློག་པ་ལའོ། །རྟག་ཏུ་ཨ་དྷི་སྔོན་དུའོ། ་། ཨན་འཇིག་གོ། །མཚུངས་པ་འདྲ་བའི་ཡི་གེ་ལ་རིང་པོར་འགྱུར་ཏེ། །ཕ་རོལ་ཡང་དབྱིའོ་ཞེས་སོ། །ཨ་དྷཱི་ཏེ། བྱིངས་ཀྱི་ཨི་ཡིག་དང་ཨུ་ཡིག་མཐའི་དབྱངས་དང་པོ་ལ་ཞེས་པས་ཨི་་ཡ་རུ་བསྒྱུར་རོ། །མཚུངས་པ་འདྲ་བའི་ཡི་གེ་ལ་རིང་པོ་ཞེས་པ་ལ་སོགས་པས་་་རིང་པོ་ཉིད་དོ། །ཨ་དྷཱི་ཡཱ་ཏེ། ཨཱ་དྷཱི་:ཡཱ་ཏི། ་ཨ་དྷི་ཥ། །ཨ་དྷཱི་ཡཱ་ཐེ། ཨ་དྷི་དེ། ཨ་དཱི་ཡེ། ཨ་དཱི་བ་ཧེ། ཨ་དཱི་མ་ཧེ། དྲི་དྷིང་། འབར་བ་་དང་རྩེ་བ་དག་ལའོ། །བི་བཱིང་། བྱར་དགའ་མཚུངས་པ་ལའོ། །འདི་དག་་སྡེབ་སྦྱོར་དག་ལའོ་ཞེས་ཁ་ཅིག་གོ། །།ཨཱང་སྔོན་དུའོ། །ཨཱ་དཱི་བྷཱི་ཏེ། སྔ་མ་་་བརྩེགས་པ་མ་ཡིན་པའི་ཨེ་ཡིག་གི་ཡ་སྟེ་ཡི་གེ་གཅིག་མིན་པའི་འོ་ཞེས་པས་ཡ་་་་ཉིད་དོ། །ཨཱི་དཱི་དྷཱ་ཏེ། ཨ་དཱི་དྨ་ཏེ། ཨཱ་བེ་བཱི་ཏེ། ཨ་བེ་བྱ་ཏེ། སྒྲུང་་་་སེལ་བར་བྱེད་པ་ལའོ། །】་ཏེ། སྣུ་བཱ་ཏེ། རྩྭ་བ་ཧེ། ནིཉྩ་ཏེ། ནིཉྩཱ་བཱ་ཏེ། ་ཨཱ་བཉྩི་ཏེ། ཨཱཏྨ་ནེའི་ཚིག་རྣམས་སོ། །།།།།།དྭི་ཥ་མི་དགའ་བ་ལའོ། ་། ཨན་འཇིག་གོ། །གཅིག་གི་ཚིག་ལ་ཨུ་པ་དྷཱ་་་ཡང་བའི་ཡོན་ཏན་ནོ། །ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །དྲིཌི། དྭི་ཥ། དྭི་ཥནྟི། ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ག་ཉིད་དོ། །དཱིཀྵ། དཱིཥྛ། དྲི། དྲིལླི། དཱི༑དཱི། དཱིཏྟེ། དྭི་ཥཱ་ཊེ། དྭི་ཥ་ཏེ། དྭི་གླེ། དྭི་ཥཱ་ཐེ། ཥ་ལས་ཊ་སྡེ་ཉིད་དོ། །དྷ་ཡིག་གི་ལྷ་ཡིག་གོ། །དྷུ་ཊ་རྣམས་ཀྱི་གསུམ་པ་ཉིད་དོ། །ཥའི་ཌའོ། །དཱིནྡྷེ། དྭེ་ཏེ། དྭི་ཥ་ཧེ། དཱིཉྩ་ཧེ། དུ་ཧ་འཇོ་བ་ལའོ། །།
