རྟགས་དོན་ནོ། །ས་ས་ཛུཥའི་ཞེས་པས་རུ་ཉིད་དོ། །ཨུ་ཡིག་ཨཏ་ལས་ཧཤ་ལའང་རུའི་འོ་ཞེས་པའི་རྟགས་དོན་ནོ། །རེ་ཕའི་རྫོགས་པ་ལ་རྣམ་པར་བཅད་་་པའོ༑། །གློཙནྡྲ་མཟླ་བའོ། །གྲུ། ནོ་ཛ་ལ་ཏ་ར་ཎི་ཆུ་སྒྲོལ་བ་ཨུཎ་ལ་་་སོགས་རྐང་པ་དང་པོའི་འགྲེལ་པ་རྫོགས་སོ། །།།།ལ་ཚུགས་པ་དང་པོ། ཁ་ཨིན་དྷཱི་ཀཱ་བ་ཤ་ལི་:མ་རྩི་་རྣམས་ལས་ཀནའོ། །ཨིཎ་བཱི་བཱ་འགྲོ་བ་ལའོ། །བྷི་འཇིགས་པ་ལའོ། ་། ཀི་གི་རི་སྒྲ་ལའོ། །ཤིཏ་མ་ཡིན་པ་ལ་ཨེཙའི་ཨ་ཉིད་དོ། །པཱ་བཏུང་བ་ལའོ། ་། ཤ་ལ་ཧྭ་ལ་པ་:རྨི་ལམ་ལ་འགྲོ་བ་ལའོ། །མནྡ་ཞེས་པ་མདོའི་བྱིངས་འདི་རྣམས་ལས་ཀ་ན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །བྱ་བའི་དོན་ལ་ཨི་་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཡིག་། སུའི་རུ་ཉིད་སོགས་སྔར་བཞིན་ནོ། །ཨེ་་་ཀགཅིག། ཨེ་ཀཱ་ཀཱི་བྷེ་ཀཿརྗེ་ཀསྦལ་པ། ཀཱ་ཀབྱ་རོག། བཱ་ཡ་སཔ་ཀཿབཱ་ལཿབསྐ་རཿཙུའི་ཀུའོ་ཞེས་པས་ཙ་ཡིག་གི་ཀ་ཡིག་གོ། །མརྐསཱུརྱཉི་མ། ་་ཡུཀ་ལ་སོགས་པ་རྣམས་སོ། །ཡུ་ཀ་ཨུཥ་ཨབྷ་ཀ་པྲི་ཐུཀ་བྷུ་ཀ་སྱ་མཱི་ཀ་ཨུ་ད་ཀ་བཱི་ཀ་བྷི་ཀ་ཧཾ་ཀ་སྒྲ་འདི་རྣམས་ཀན་ཕ་རོལ་ཚིག་ཕྲད་དུ་བྱའོ། །ཡཱ་ྋསྤེལ་བ་ལའོ། །རྀ་འགྲོ་བ་ལའོ། །པྲི་ཐུ་ཡངས་པ་ལའོ། །ཨུནྡྲི་ཀླས་པ་ལའོ། །ཁྱུ་འགུལ་བ་ལའོ། །སྱ་མུ་སྭཱ་ན་དྷཱ་སྒྲ་ལའོ། །ཨ་ཛ་བྲ་ཛ་འགྲོ་བ་ལའོ། །།ཨིཎ་་བཱི་བཱ་འགྲོ་བ་ལའོ། །བྷི་འཇིགས་པ་ལའོ། །ཧྲི་ངོ་ཚ་བ་ལའོ། །དེ་ལ་ཡོ་ཏིའི་རིང་པོ་ཉིད་དོ། །ཨ་ཛ་ལ་སོགས་པ་དང་ཨཏ་ལས་སོ་ཞེས་པས་ཆ་པྲའོ། །ཌ་་་ཡིག་ཌིཏ་ལ་ཀུ་ཥ་ངེས་པར་འགུགས་པ་ལའོ། །གི་སྒྲ་ལའོ། །ཤིཏ་མ་ཡིན་པ་ལ་ཨེ་ཙའི་ཨཱཏ་ལའོ་ཞེས་པས་ཨ་ཉིད་དོ། །རྀ་འགྲོ་བ་ལའོ། །འདི་རྣམས་་་ལས་ཐན་འགྱུར་རོ། །ན་ཡིག་དབྱངས་དོན་ནོ། །།ཡང་བ་མཐའ་ཉ་འི་འོ་ཞེས་།
