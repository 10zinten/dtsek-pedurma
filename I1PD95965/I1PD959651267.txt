།:་་་་བསྟན་འགྱུར། སྣ་ཚོགས། ཡིལ་རྒྱུན་ལསྒྲོལ་མ་ཁོ་ཁ་الالمཀུན་ལ་སྨན། །སྨན་ནི་འབྲས་བུ་སྨིན་མཐའ་དང་། །།འཁྲི་ཤིང་། གུ་ག། ལྕང་མ་དང་། །ལྕུག་མ་དང་ནི་ཚང་ཚིང་ངོ༌། །།ཨུ་པ་ས་རྩེ་བར་བསྡུད་པ། སྡོང་པོ་དང་། །།མགོ་ལིང་གནས་དང་གེལ་པ་ཅན། །།རྩ་བ་ཡལ་གའི་བར་གྱི་ཤིང་། །སྡོང་རྐང་དང་ནི་ཕུང་པོའོ། །ཡལ་ག་རྒྱུན་དང་འཁྲི་ཤིང་ངོ༌། །ཕུང་པོའི་་་་ཡལ་ག་མཐོ་བའོ། །རྩ་བ་དང་ནི་རལ་པའོ། །ཡལ་གའི་རལ་པ། ་ཀུན་་་་ནས་སྐྱེས། །རྩ་ནས་རྩེ་མོར་འཁྲི་ཤིང་སོན། །ཤིང་ལ་སོགས་པ་བཙུགས་པ་་་དང་། །བསྟོད་པ་བསྒྲུང་དང༌། བླང་བའོ། །།མགོ་དང༌། རྩེ་མོ། །ཅོ་ཏོའོ། །རྩ་བ། སྨད་དང་། རྐང་བའི་མིང་། །།གོང་བུ་དང་ནི། སྙེ་མའོ། །དེ་ཉིད་མཉྫི་བལླཾ་སྟེ་མ་གོང་བུ་དང་། གུནྡྷེ། །འདི་རྣམས་མཚུངས། །སྦལ་མིག་ཐོན་དང་ཛཱ་ལཾ་ཀཱ། ཀོ་མཱ་རི་ཀུ། ཛཱ་ལ་ཀཱ། །གཞོན་ནུ་གསར་བུ་ཚིགས་་་ཀཾས་པ། །ཁ་བྱེ་བ་དང༌། གྱེས་པ་དང་། །གུཙྪ། གླུ་པ། གུ་ལུཙྪ། །ཨཱ་ལིམ་ཞེས་བྱ་བའོ། །བུ་ཚབ་པ་དང༌། མུ་ཀུ་ལ། །མེ་ཏོག་རབ་སྐྱེས་སྐྱེས་པ་དང༌། །སུ་མ་ན་ས་ཡིད་དགའ། །ཡིད་འོང་དང་། །ཀུ་སུ་མ་མེ་ཏོག་ཞེས་བྱ་བའོ། །མེ་ཏོག་རྡུལ་དང་། ཡིད་འོང་རྡུལ། །མེ་ཏོག་རྩི་དང་། །མེ་ཏོག་རོ། །ལོ་མ་རྙིངས་པ། །ས་བ་ཀཾ། །འདབ་མ་ལོ་མ། སྒྲིབ་བྱེད་དང༌། །ད་ལི་འདབ་མ་པརྣོ་ཤིང་ལོ་ཙྪ་ལོ་འདབ་དའོ། །སྡོང་པོའི་སན་མ་བརྟཉྩ། །མགྲིན་པ་ཅན་དང་། ཨུཎི་ལ། །ཀ་ད་ར་དང་། ཤུ་ནི་དང་། །འདབ་མའི་རྩེ་དང་། དཀར་པོའི་རྩེ། །སྡོང་པོ་ཅན་ཞེས་བྱ་བའོ། །།མགོ་ལིང་། ཡལ་ཚོགས་ཀི་ཤ་ལཾ༑། རྩར་སྐྱེས་སྡོང་པོ་ལྷག་པ་དང་། །ཤིང་གི་བཅུད་དང་ཡངས་པ་དང་། །ཤིང་གི་བརྟུལ་ཞུགས་ཤེས་པའོ། །ལྕག་མ་དང་ནི་འཁྲི་ཤིང་དང༌། །སྡོང་་་་པོའི་འཁྲི་ཤིང་ཡཱ་བ་ཀ། །འདབ་མའི་ལང་ཚོ་ཤིང་གི་བརྟུལ། །འདག་ཁུ། །
