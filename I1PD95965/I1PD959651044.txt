བསྒྲུབ་པར་བྱའོ། །ཛུ་ཧཱ་ཏི། ཛུ་ཧོ་ཏི། ཛུ་ཧུ་ཐ། ཛུ་ཧུ་ཐ། ཛུ་ཧོ་མི། ཛུ་ཧོ་བ། །ཛུ་ཧུ་མ། ཉི་བྷི་འཇིག་པ་ལའོ། །ཉི་ཨིག་ནི། ཉི་རྗེས་སུ་་་་འབྲེལ་པ་ལ་འདོད་པ་བློ་མཆོད་པའི་དོན་རྣམས་ཀྟའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨན་འཇིག་གོ། །།གཉིས་བརྗོད་དོ། །ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུའོ། །གཉིས་པ་བཞི་པ་དག་གི་དང་པོ་གསུམ་པ་དག་གོ་ཞེས་པས་ཨ་བྷཱུསའི་བ་ཡིག་གི་བ་ཡིག་གོ། །ནཱ་མིན་མཐའི་ཡོན་ཏན་ནོ། །པི་བྷེ་ཏི། བྷིའི་ཡང་ན་བཤད་པར་བྱའོ་ཞེས་་་པས་གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་བྱིངས་ཐམས་ཅད་པ་ཡོན་ཏན་མེད་པ་ལ་་་རྣམ་པར་བརྟག་པས་ཨཱི་ཉིད་དོ། །བི་བྷཱི་ཏསྔ་མ་བརྩེགས་པ་མ་ཡིན་པའི་ཨིའི་ཡི་གེའི་ཡ་སྟེ་ཡི་གེ་གཅིག་མིན་པའི་འོ་ཞེས་པས་ཡ་ཉིད་དོ། །བི་བྷཱུ་ཏི། བི་བྷཱི་ཏི། བི་བྷི་ཐ། བི་བྷི་ཐ། བི་བྷི་མི། ཌི་བི་བྷི་བ། བི་བྷཱ་བ། བི་བྷི་མ། བི་བྷི་མ། ཧཱི། ངོ་ཚ་བ་ལའོ། །ཨན་འཇིག་གོ། །།གཉིས་བརྗོད་དོ། །ཨ་ངྷ་སའི་གསལ་བྱེད་དང་པོ་ལྷག་མར་བྱའོ་ཞེས་པས་ཨ་ཧྥུ་སའི་གསལ་བྱེད་དང་པོ་ལྷག་མར་བྱ་བ་སྟེ་བསྲུང་བར་བྱ་བར་འགྱུར་གྱི་དང་དབྱི་བར་བྱ་བ་མ་ཡིན་ནོ་ཞེས་པའི་དོན་ཏོ། །ཐུང་ངུའོ། །ནཱ་མིན་མཐའི་ཡོན་ཏན་ནོ། །ཛི་ཧྲེ་ཏི། ཛི་ཧཱི་ཏིསྔ་མ་བརྩེགས་པ་མ་ཡིན་པའི་ཞེས་པའི་ཚིག་ལས་སྔ་མ་བརྩེགས་པའི་ཡ་ཉིད་དུ་མི་འགྱུར་རོ། །ཨི་ཡ་རུ་བསྒྱུར་རོ། །ཇི་ཧྲི་ཡ་ཏི། ཛི་ཧྲི་ཏི། ཛི་གྷི་ཐ། ཛི་ཧྲི་ཐ། ཛི་ཧྲི་མི། ཇི་ཧྲི་བི། ཇི་ཧྲི་མ། པྲ་སྐྱོང་བ་དང་རྫོགས་པར་་་་བྱེད་པ་དག་ལའོ། །ཨན་འཇིག་གོ། །གཉིས་བརྗོད་དོ། །རཾ་ཡིག་གི་ཨ་ཡིག་གོ་ཞེས་པས་ཨ་ཧྥུསའི་ར་ཡིག་ཨཱ་ཡིག་གོ། །།ཨརྟི་པི་པརྟི་དག་གི་འང་ངོ་ཞེས་པས་ཨི་ཡིག་གི་ཨི་ཉིད་དོ། །ནཱ་མིན་མཐའི་ཡོན་ཏན་ནོ། །པི་བནྟི། ་པི་པྲ་ཏ། བི་པྲ་ཏི། བི་པནྟི། བི་པྲ་ཐ། །བི་པྲ་ཐ། བི་པརྨི། བི་བྲི་བ། བི་སྲིད་—།
