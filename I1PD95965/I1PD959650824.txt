ཏེ་ར་སྱ་ཀ་རུ་ཝ་བཱནྟི། རྩོ་ཧ་སྱ་བྷེན་སུ་ར་བཱནྟི། སཾ་ཥ་ནི། ཌ་ཀ་ནི་བཱནྟི། པ་ཏ་ཀླུ་ནི་བཱནྟི། བེ་ག་ན་བཱནྡྷ་ཏོ་བཱ་བཱ་ཨཱ་ད་མ་ཀི་དོ་ཧཱ་ཨི། བཱི་བི་ཝཱ་ཏ་མཱ་ཀི་དོ་ཧཱ་ཨི། སུ་ཝ་ར་མུ་ར་དཱ་ར། བི་བི་ཕ་ཏ་མ་ཀི་དྭ་ཕ་ན་གི་རེ། ན་ཝ་ལཱ་ཥ། ཨ་སི་ཧཱ་ཛཱ་ར། པི་ར་པེ་ཀམྤ་ར་བཱ་དྷི་ཌ་རི། ཨ་ཝ་དེ་ཏོ་རོ། མ་ཧཱ་མ་དཱ་པི་ར་ཏེ་རེ་མནྟཱ་ཀི་ཤཀྟི། ཨུ་ཏ་ར་ཀོ་ཨུ་ཏ་ར་བཱནྟི། དཀྵ་ཎ་ཀོ་དཀླུ་ཎ་བཱནྟི། པཉྩེ་མ་ཀོ་པཉྩེ་མ་བཱནྟི། པཱུརྦ་ཀོ་པཱུརྦ་བཱནྟི། ཀཱ་ཋ་ཀི་རོ་ཌི་བཱནྟི། ཕི་རེ་མ་ཧ་མེ་དཱ་པཱི་ར། སེ་ཥ་སི་ལཱ་ར་པ་དྷ་ཏ་མནྡཱ། བཱ་དྷི་ཊཱ་བཱི་མེ་རི་བཱསྟི། གུ་རུ་ཀཱི་ཤཀྟི་ཕུ་རོ་མནྡཱ་ཨི་ཤྭ་རཱི་བཱ་ཙ། སོར་མོའི་ནད་དང་ལྦ་བ་ལ་སྔགས་འདི་སུ་ཧུག་དམར་པོར་བྲིས་ཤིང༌། འབྲས་ཟན་དང་འོ་མར་ཡང་བཏབ་སྟེ། དེ་ནས་ཝ་བ་དང་སོར་མོ་གང་ཡིན་ལ་སུ་ཧྲུག་དེས་དཀྲི་ས་འོ་མ་དང་འབྲས་ཟན་བསྲེས་པ་དེས་བྲབ་ན་ཕན་ནོ། ༈། ཨོཾ་ན་མོ་ཨཱ་དེ་ཤ་གུ་རུ་ཧཱུྃ། ཨོཾ་ཀཱ་ར་བིནྡ་སཀྟ། ནི་བྱདྨ་ཡནྟི་ཡོ་གེ་ཤཱ་རཾ། ཀཱ་མ་རཿམོ་གླུ་དབྦེ་ཝ། ཨོ་ཀཱ་རཱ་ཡ་ན་མོ་ན་མན་མ་སྟེ་རཾ་ཥ་ཡ་དེ་ཝཱ་ན་མ་སྟེ་ཙ་སུ་ར་ག་ཎཱ། ན་རཱ་ནཱ་མ་སྐྱ། དེ་ཝ་སྱ་ན་ཀཱ་རཱ་ཡ་ན་མོ་ནཱ་མ༔མ་ཧཱ་དེ་ཝ། མ་ཧཱ་ཏ་མཱ་ནི། མ་ཧཱ་ཡོ་གི་མ་ཧེ་་་ཤྭ་ར། མ་ཧཱ་པ་ཏ་ཀ། ཧ་ར་དེ་ཝ་མ་ཀཱ་རཱ་ཡ་ན་མོ་ན་མཿཤི་ཝ་སཱནྟཾ་ཛ་གནྣཱ་ཐཾ། ལོ་ཀཱ་ནཾ་ཤཱནྟི་ཀ་ར་ཎཾ། ཤི་བ་མེ་ཀ་པ་རཾ་དེ་ཝཾ། ཤ་ཀཱ་རཱ་ཡ་ན་མོ་ན་མབཱ་ཧ་ནཾ་བྲི་ཥ་བྷཕྱི་ཝ་བཱ་སུ་ཀི་ཀཎྜ་བྷུ་ཤ་ནཾ། བཱ་མ་ཤཀྟི་དྷ་རོ་དེ་ཝཾ། བ་ཀཱ་རཱ་ཡ་ན་མོ་ན་མཡ་ཏྲ་ཏ་ཏྲ། ཏྲ་ཡ་དེ་ཝཱ་བྷ་བྷཱ་བིཥྛ། མ་ཧེ་ཤཱ་རཾ། ཡ་ག་ཏ་ཀཱ་ར་ཏཱ། ཛ་གན་ཐཾ་ཡ་ཀཱ་རཱ་ཡ་ན་མོ་ན་མཿཨོཾ་ཀཱ་རེ་མུ་ལ་ཙཀྲ། ན་ཀཱ་རེ་ནཱ་བྷི་མཎྜ་ལེ། མ་ཀཱ་རེ་ཧཱུྃ་ད་ཡ་སྦྲ་ན་ས་ཀཱ་རེ། ཀཎྜ་མེ་ཝ་ཙ། བ་ཀཱ་རེ་བ་ཀྲ་སྦྲ་ལ། ཡ་ཀཱ་རེ་པུ་ཛོ་ཏི་ཡ་ག་ཏྲཿཞེས་པ་ཡི་གེ་དྲུག་ལ་བརྟེན་པའི་སྔགས་འདི་སྟན་བདེ་བ་ལ་འདུག་སྟེ་རང་་་གིས་རྒྱུན་དུ་བཟླས་ན་ལུས་ཀྱི་ནད་ཐམས་ཅད་ལ་ཕན་ཞིང༌། ཁྱད་པར་རླུང་མཁྲིས་།
