ཀཱ་ཏྱཱ་ཡ་ན་མཆོག་སྲེད་དང་། ལ་། དཀར་མོ་:ཀཱ་ཏྱཱའི་བུ་མོ་དང་། །རྒན་པོའི་རྣམ་པ་ངུར་སྨྲིག་གོས། །།ཀུ་ཙནྡ་ནཾ། ཙནྡན་དམར། །འདབ་མ་རི་ལུས་དང་ཤིང་དབྱེ་ལ། །ཀུཎྜ་ལིརྨ་ཆུ་ལྷ་ཉིད། །རྨ་བྱ་རི་དྭགས་སྦྲུལ་རྣ་རྒྱན། །བུམ་པ་སྐྱེས་ནི་རི་བྱི་དང་། །སྲིད་སྒྲུབ་དང་ནི་ཕུར་བུ་ལ། །ཀེ་ཤ་རན་ནི། སེང་གེ། ཀླུ། །ནཱ་ག་གེ་སར་རྟ་རྣམས་ལ། །ཀྲོཉྩ་ད་ནཾ། གྷིཉྩ་ལི། །ནོ་ཏ་ཀ་པདྨའི་རྩ། །ཁ་ཀཱ་མ་ནི་ངེས་བརྟན་དང་། །སྤྱོད་བྱེད་མ་དང་བུད་མེད་གཞོན༑། ཁརྒ་དྷེ་ནུ་བུ་མོའི་གྲི། །དེ་བཞིན་སྡོང་པོ་ཅན་ལའོ། །གོ་ད་ཡིདྨ་སྨྲ་བྱེད་དང༌། །འདོད་དང་འདོད་ལྡན་རྣམས་ལའོ། །ག་བཱ་ད་ནཾ་ཆུ་ལྷའི་དབང་། །ལུས་ལྡན་དབྱངས་ཀྱི་རྟེན་ལའོ། །གྷ་ནཱ་གྷ་ན། །ཆར་སྤྲིན་དང་། །བརྒྱ་བྱིན་དང་ནི་གླང་པོ་མྱོས། ་། གྷོ་ཥ་ཡིཧྥུ་ཞེས་བྱ་བ། །ནེ་ཙོ་དང་ནི་བྲམ་ཟེ་ལ༑འ། ཙི་ཏྲེ་བྷཱ་ནུ་བྱ་དང་མེ། །ཙོ་ལ་ཀིན་ནི་ཞ་ཉེ་དང༌། །ལཾ་བ་བ་ལང་གི། དང་ནི་ཀིསྐུའི་ཚིགས། །།ཛ་ལཱ་ཊ་ན་འཇུག་པ་དང་། །ཀྲྀ་ཀ། ཀཾ་ཀ་ཁྲག་གི་་རྒྱུན། །ཛ་ལ་ཊ་ནི་ཆུ་འཁོར་ལ། །ཛ་ལཱ་ཊ་ནི་ཆུ་གནས་ལ། ་། ན་ལ་མཱི་ན་་ཉའི་བྱེ་བྲག་། ཨིཉྩཱ་ཀ་དང་བྱིས་པ་གསོད། །ཏ་པ་སཱི་ནི་དཀའ་ཐུབ་ཅན། །དཀའ་ཐུབ་དང་ནི་རྗེས་སུ་རྩེ། །ཏ་བོ་དྷ་ན་དཀའ་ཐུབ་ལ། །་། སྣག་ཅན་ཚ་བ་་་སྐྱེ་བ་ལ། །ཏ་བོ་དྷ་ན་མགོ་ཟླུམ་མ། །ནང་མ་ཁྱིམ་དང་དཔའ་བ་ལ། །དུཊཱ་མནྣི་འདམ་གྱི་ནི། །ཉ་ཕྱིས་དང་ནི་འབྲུམ་སྲན་ནོ། །དེ་བ་སེ་ནི་ལྷ་ཡི་དམག། གནས་དང་རྟོགས་པ་དག་ལའོ། །དྲི་ཛརྨ་ན། སོ། བྱ་བྲམ་ཟེ། །གླང་པོ་་ཐོབ། ཀླུ་ཤིང་ཀླུ། །ནི་དྷུ་བ་ནཾ། བསྐྱོད་དང་དགའ། །གསོད་དང་མཐར་བྱེད་ནགས་ཚལ་ལ། །ནིཊཱ་ས་ནི་གནས་མེད་དང་། །འཚེ་དང་འགོ་བའི་གནས་ལའོ། །ནིཧྥཏྠ་ནི་ངན་བྱེད་དང་། ་། ཨ་ལལྟ་ཀ་ཞེས་པ་ལ། །—།
