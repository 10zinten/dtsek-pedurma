སུ་བ་ཁ་རི་ཙ་སྔ་ལ་ཞེས་པས་ཊ་ལ་ཊ་བྱའོ། །བེ་ཊ། ་བི་ཌ། བི་ཤོ། བི་ཤ་ལ་བི་ཤཱི། བི་དྷི། བི་ཙྪི༑། བི་ཤེ། བིདྡྷཾ། བིནྡྷ། བི་ཤྭ་བི་ཙྪི། །བིནྡྷ། བི་ཤ། བི་ཤོ། །བི་ཤཱི། བི་ཤི། བི་ཤོབིཥྛ། ཧེ་བི་ཊ། ཧེ་བི་ཤོ། ཧེ་་བིཤདྲི་ཥའི་སྒྲ་ལས་སུ་དབྱིའོ། །ཚིག་མཐའ་ལ་བ་ལཱ་དོ་ཙས་ཥ་ཡིག་ལ་ཛ་སཾ་ཞེས་པས་ཊ་ཉིད་དོ། །བཱ་བི་རཱ་མ་ཞེས་པས་བི་ར་མོ་ཌ་ལ་རྙའོ། །སུ་པི་་ཁ་རི་ཙ་ར་ཛྷ་ལ་ཞེས་པ་ཌ་ལ་ཊ་ཡིག་གོ། །དྲི་ཊ་དཱི་ཊ། དྭི་ཥདྲི་ཏཱཾ། དྲི་ཏོ། དྭི་ཥདྲི་ཥཱ། དྭི་ཥ། དྲི་རྒྱུ། །དྭ་ནྟི། དཱི་པེ། དཱི་ཧཱུཾ། དཱིཙྩུདྭེ་ཥ། དཱིདྡྷཾ། དྲི་ཟླུདྭི་ཥ། དྭི་ཥོདྲི་ཏི། དྲི་ཏི། དྭི་ཥོདྲི་གླུ། ཧེ་རཱི་ཊ། ཧེ་དཱི་ཊ། ཧེ་དཱི་ཊ་ཏོ། ་ཧེ་དཱི་ཥབེ་དྷ་བའི་སྒྲ་ལ་ཨ་ཏཱ་རུ་ཞེས་པས་སཱ་བ་སཾ་བུ་དྷོ་ས་རིང་པོ་ཉིད་དོ། །སུ་ཞེས་པས་སུ་་་དབྱིའོ༑། ས་ཡིག་ལ་རུ་སྟེ་ཚེག་དྲག་གོ། །སུ་པི་བ་ལི་ཏ་དྷ་དི་ཏི་ད་དྷ་ད་ཏི་དེ་ཤ་ལས་ཤ་ས་ཛ་སོ་ཞེས་པས་རུ་ཉིད་དོ། །ཧ་སི་ཙ་ཏོ་རོ་ཞེས་པས་ཨུ་ཉིད་དོ། །དེའི་ཡང་ཨ་ཡིག་གིས་ལྷན་ཅིག་པར་ཨ་ཏ་དེ་ང་ཞེས་པས་ཨེ་ཡིག་གོ། །སུ་པི་རུ་བྱ་སྟེ་བཱ་ས་རི་ཞེས་པས་རེ་ཕ་ལ་ས་ཡིག་ཀྱང་རུང་ངོ༌། །རོ༔སུ་པི་་ཞེས་པས་རེ་བ་ལ་ཚེག་དྲག་གོ:：། །བེ་དྷ་བེ་དྷ་སོ། བེ་དྷེ་སབེ་དྷ་སཾ། བེ་དྷ་སོ། བེ་དྷ་སིདྷ་སཱ། བེ་དྷོ་ཧཱུཾ། བེ་དྷོ་བྷིབེ་དྷ་སེ། །བེ་དྷོ་ཧཱུཾ། བེ་དྷོ་བེ་དྷ་ས། བེ་དྷོ་ཧཱུཾ། བེ་དྷོ་། བེ་དྷ་སབེ་དྷོ་སེབེ་དྷ་སཾ། བེ་དྨ་སི། བེ་དྷ་སོ། །བེ་དྷ་སྐུ། ཧེ་བེ་དྷཧེ་་བེ་དྷ་སོ། ཧེ་བེ་དྷ་ས། ཨུ་ཤ་ནའི་སྒྲ་ལས་དང་པོའི་གཅིག་ཚིག་ལ་སོ་ཏྲི་དུས་ན་།
