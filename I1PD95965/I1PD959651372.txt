ལྗོན་དམར་ཨུ་:དུམྦ་ར་དང་། །དམར་པོ་ཀླ་ཀློའི་ཁ་གཉིས་བརྒྱད། །བི་རཀྵ་དང་ཐུབ་མཁྲིས་འཛིན། །ལྕགས་དམར་དང་ནི་ཡོང་སྐྱོབ་ཀྱང་། །།ར་གན། རིའི་ཀཱ་རི་མེད། །མཁྲིས་འཛིན་ཨཱ་ར་ཀུ་ཊ་དང་། །རི་ཏི་དང་ནི་ཞུ་བྱའོ། །རྒ་བ་འཁྱོག་དང་ལུས་འཇམ་དང་། །སུ་རེ་བྷ་དང་། པིཙྪ་ཊ། །ཏྲ་སུ་ཞེས་་བྱ་བ་རྣམས་སོ། །ཞ་ཉེ། སཱི་པ། ཀ་སྦྱོར་འདོད། །ཏྲ་པུ་དང་ནི་དེས་་འདོད་ཅན། །ཕྲིཙྪ་ཏ་དང་སཱི་པའི་བདག། བ་དྡྷཾ། འཇམ་པའི་གཟུགས་་ཅན་ནོ། །གསེར་གྱི་དགྲ་དང་། སྟོབས་མང་པོ། །མ་དྷུ་ཀཾ་དང་སྟོབས་པོ་ཆེ། །ལྕགས་དང་མཚོན་ཆ་ཅན་རྣོན་པོ། །གོང་བུ། ནག་པོ། ལྕགས་གོང་དང་། །རི་ཡི་སྙིང་པོ་རྡོ་ཡི་སྙིང༌། །ས་ར་ས་པ་ཞེས་པ་ཡང་། །ལྕགས་ཀྱི་རྣམ་འགྱུར་ཕྲ་མོ་འདི། །ཐམས་ཅད་ཀྱང་ནི་ལྕགས་སུ་འགྱུར། །ལྕགས་སྙིགས་སོ་ཐོག་ལྕགས་དྲེག་དང་། །ལྕགས་ཀྱི་དྲི་མ་ཚ་ཚའི་ཞགཁ་ད་། འཁར་བ་ཀཾ་སྱ་ཀཾ་ས་སྦི། །ཟངས་སྐྱེས་བསིལ་བའི་ཆུ་སྐྱེས་སོ། །ཞ་ཉེ་སྐྱེས་དང་ཟངས་སྐྱེས་དང་། །ས་རཱ་ཏཱི་ཀཾ་ཙ་ར་ཎཾ། །རྒྱུག་བྱེད་དངུལ་ཆུ་འདང་བྱེད་དང༌། །ཁུ་བའི་དབང་པོ་ཞེས་གྲགས་དང་། །འཕྲོག་བྱེད་ས་བོན་ཁམས་ཆེན་པོ། །དྲག་པའི་ཁམས་དང་སུ་ཏ་ཀ་ཡོ། །མཚལ་སྐྱེས་དང་ནི་ཧིང་གུ་ལ། །དཱ་ར་ད་དང་དམར་པོ་དང་། །མཆོག་སྦྱིན་ངང་པའི་རྐང་པའོ། །སིནྡྷཱིར་དང་བྱེ་མ་དམར། །ཕྱེ་མ་དམར་དང་རྡུལ་དམར་དང་། །དྲི་འཛིན་སྐྱེས་དང་ཞ་ཉེ་སྐྱེས། །།བ་བླ་ལ་ནི་ཀ་ལཾ། །བ་ལང་སོ་དང་གར་མིང་ཅན། །པིཉྫ་ར་དང་པཱི་ཏ་ནཾ། །ཏཱ་ལ་མཱ་ལམ། གར་གྱི་རྒྱན། །སེར་པོ། པཱི་ཏཱཾ། བྱི་ལའི་མིག་ཀ། རོ་མ་པྲ་ཏ་ནཾ་ཞེས་སོ། །ལྡོང་རོས་དང་ནི་གར་ངན་མ། །ཡིད་འབོད་མ་དང་ཀླུ་ལྕེ་མའོ། །བལ་མོ་ས་ཡི་གོར་མ་དང་། །གོ་ལ་དྲི་རྣོ་མུ་ཟིའི་དྲི། །དྲི་བཟངས་དང་ནི་མིག་ལ་ཕན། །།
