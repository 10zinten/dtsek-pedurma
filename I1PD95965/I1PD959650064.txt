ཨི་ཡི་སྭ་རེ། ཨི་བནྡྷོ་ཡ་ཏཱ་མཱ་པཏེ་སྭཱ་རེ་པ་རེ། ་ད་དྷ་ཡ་ཨཱ་ན་ཡ་ཨི་ཏི་ཏཱ་གོ། །ཨི་ཡི་དབྱངས་ལའོ། །ཨིའི་ཡི་གེ་ནི་ཕྲ་ཉིད་དུ་ཉེ་བར་འགྲོ་སྟེ་དབྱངས་ཕ་རོལ་སྦྱར་རོ། །ད་དྷ་ཡ་་བནྡྷ་བ་ཏི། ཊ་ཊ་ཧ་སྱཉརྙས་། སྭ་རཏྤ་རོ་རེ་ཕ་ཧ་ཀཱ་ར་ཨཱན་ཡ་ཞེས་པ་དེ་འདྲར་འགྱུར་རོ། །ཧ་ས་ལ་ར་ཧ་མ་ཡིན་པའི་ཧ་ས་རྣམས་སོ། །དབྱངས་ཀྱི་ཕ་རོལ་རེ་ཕ་བཛྫི་ཏོ་ཧ་ས་ཧ་སེ་པ་རེ་དཱིབྷ་བ་ཏི། ཨི་ཏི་དྷ་ཀཱ་ར་སྱ་རཱི་ཏཱཾ། ལ་སྨྲས་པ། དང་ཧ་ཡིག་སྤངས་པའི་ཧ་ས་ནི་ཧ་ས་ཕ་རོལ་ལ་གཉིས་སུ་འགྱུར་རོ། །ཞེས་པས་དྷ་ཡིག་གཉིས་ཉིད་དོ། །དཱི་ཏཱ་བི་དྷཱ་ན་ཏཱ་བེ་བ་ཤི་བྱེ་ཏི། །ཁ་མ་ཨ་རྩེ་ཧ་སོ་ལུཔྤནྟེ། ཊཛ་བྷེ་ཛ་གཉིས་ཉིད་བཞག་པ་གཉིས་པོ་འདི་གནས་ཀྱི། གཞན་གསལ་བྱེད་དབྱི་བར་བྱེད་དོ། །ཛྷ་བྷ་ལ་ཛ་བི་ཆབཱ་ཏིཛྷ་སཱ་ནཾ་ཛྷ་བྷེ་པ་རེ་ཛ་བཱ་བྷ་བནྟི། རྒྱུ་ཨི་ཏི་པཱུརྦ་དྷ་ཀཱ་ར་སྱ་ད་ཀ་རཿརྣམས་སོ། །ཛྷས་རྣམས་ཀྱི་ཕ་རོལ་ཛྷ་བྷ་ལ་ཛ་བ་ཏུ་འགྱུར་རོ། །ཞེས་པས་དྷ་ཡིག་སྔོན་མ་དའི་ཡི་གེའོ། །ས་བཎྜཏཱི་དྭགློ་བ་རྩེ་ཎ་ས་བཎྜ་ཨི་ཏི་པ་ཙ་ནཱ་ཏ། །ང་། ཡ་ཐཱ་སཾ་ཏུཾ་བཱ་བལྟ་བྱ། འདྲ་བའི་ཡི་གེ་ནི་སྡེ་པ་ལ་སྡེ་པའི་འདྲ་བའི་ཡི་གེ་ཞེས་བརྗོད་དོ། །ཇི་ལྟ་བར་གྲངས་མཉམ་པ་ལ་ཡང་བཤད། ཨནྟ་རངྒ་བི་རངྒ་མ་སིདྡྷཾ། ་ཚ་ཁ་ཨནྟ་རངྒེ་ཀཱརྱ་ཀྲི་ཡ་མཱ་ནེ་བི་རདཾ་ཨ་བརྗོད་པ། ནང་ན་གནས་པ་ལ་ཕྱི་རོལ་ལ་གྲུབ་པ་མ་ཡིན་ནོ། །ནང་ན་སྤྱོད་པའི་ལས་བྱེད་པར་གྱུར་་་་སིདྡྷཾ་ས་ཏྲ། ་ཏཱ་པ:། ལསཱ་ར་ཧཱི་ནཾ་བ་རེ་ཎ་སཾ་ཡོ་ཕྱི། ཀ་ལ་དདྡྷན་པ་ལ་ཕྱི་རོལ་འགྲོ་བས་གྲུབ་པ་མ་ཡིན་ནོ། །དབྱངས་མེད་པ་ཕ་རོལ་ལ་རབ་ཏུ་སྦྱར་བར་བྱའོ། །དངྒྷན་་་ཡ་ཨི་ཏི་སིདྡྷཾ། གོ་རཱི་ཨ་ཏྲ་ཨི་ཏིསི་ཏེ། ཊི་ཨརྟ་ཨི་ཏི་བེ་ཤེཥ་ཎཱནྣ་ཡ་ཞེས་པ་གྲུབ་བོ། །གོ་རཱི་ཨཏྲ་ཞེས་པ་གཞག་གོ། །།།ར་ཧ་མ་ཡིན་པ་ཞེས་པའི་ཁྱད་རྣམས་འོན་རེ་ཕ་སྱ་རཱི་ཏཱཾ། ཀིནྟུ། །རྭ་རྒྱ་བོ་དཱི་ལྔ་ཁསཱ་ར་པཱུཏྟཱ་ཇི་ལྡཏྲ་རོ་འོཀྱང་གི་རེ་ཕ་ནི་གཉིས་ཉིད་མ་ཡིན་ནོ། །ར་ལས་ཡ་ཡང་གཉིས་སོ། །དབྱངས་ཕ་རོལ་ཅན་རེ་ཕའི་ཕ་རོལ་HoleHole—།
