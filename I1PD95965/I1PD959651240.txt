མཆོག་ཏུ་བྱེད། །སྟེན་བྱེད། རབ་བསྟེན་དང་བ་དང་། །ཉེ་བར་བྱེད་དང་། ཉེ་བར་གནས། །ཀུན་ཏུ་བསྟེན་དང༌། མཉེན་བྱེད་དང༌། །ཉེ་བར་སྤྱོད་་་་དང༌། །ཨུ་པ་སཱ་ནཾ། །མ་གུས། ཡོངས་སྨོད། ཟིལ་གནོན་དང་། །རིང་བྱེད། མི་རྩི་ནི་ཀཱ་ར། །།བརྙས་དང་། ཀུན་བརྙས། ཞུམ་བྱེད་དང་། །གུས་མེད། ་བརྙས་བྱེད་སྨོས་པ་དང་། །པ་བ་ནི་དང་ཨ་བ་ལི། །ཨ་བྷི་བྷ་བཱ་ཞེས་རྣམས་སོ། །ཞུམ་བྱེད། ངོ་ཚ་འཛེམ་། མིག་དམའ། །ཧཱི། ན་ཏི་བི་ཤ་ཁྲེལ་ཡོད། །བཟོད་པ། གླུ་མ། ཏི་ཏི། །ས་ཧིཥྛ་ཏཱ་ཞེས་་རྣམས་སོ། །།གསོད་དང༌། རྣམ་པར་འཚེ་པའོ། །མི་བཟོད་པ་དང༌། ཕྲག་དོག་གོ། །ཡོན་ཏན་ལ་ནི་སྐྱོན་སྒྲོ་འདོགས། ་། ཨ་སཱུ་ཡ་ཞེས་བྱ་བའོ། །རབ་སྐྱོན་འགེབས་དང་། ་འཆབ་པའོ། །དོ་ཥ་དང་ནི་ཨཱ་དཱི་ན། །ལེ་ལོ། རྨུགས་པ། སྙོམ་་ལས་སོ། །ཡིད་ལག་ཅགས་དང་། རྗེས་སུ་གཅགས། །ཕྱི་ནས་གདུང་དང༌། རྗེས་སུ་གདུང༌། །བགོད་པའོ། །བརྣབ་སེམས་་གཞན་ནོར་ཆགས། །ཉེ་བར་ལོངས་སྤྱོད། །ནིརྡེ་ཤ། ཨ་བྱཱ་དྷཱ་ན། ཉེ་་་བའི་རིམས། །འཁོན་དང་། །འགྲས་དང་། །རྣམ་པར་སྣང་། །ཁོན་དུ་འཛིན་དང༌། དེར་འཛིན་པ། །ཡིད་གདུངས། མྱ་ངན་གདུང་བ་དང་། །ནིས་མའོ། །བརྩོན་འགྲུས་དང༌། །སོ་ཊི་རྱ་དང་མ་དང༌། །སུ་ཁཱི་ར་ཏཱ་་་ཞེས་པའོ། ་། མི་བརྗོད། ཁོང་ཁྲོ། ཁྲོ་བ་ཁྲོ། །རུ་རྟ་ཀུཏྟ། བྷཱ་མ་ཀོ་ས། །ཡོངས་སུ་གདུང་བ་ཨ་བ་ཐ། །འདུ་བྱེད་དང་ནི། བཱ་ས་ནཱ། །ངལ་བ་དང་ནི༑ཐང་ཆད་པ། །ཡ་མཚན་དང་ནི། ངོ་མཚར་དང༌། །དགེ་་མཚན་དང་ནི་ལྟད་མོ་ཆེ། །སྐྱོ་བ་སངས་པ་ཞེས་བྱའོ། །འདོད་རྒྱལ། རང་འདོད༑རྒྱུ་མཚན་མེད། །ཡིད་ཆེས་པ་དང་། དད་པའོ། །རབ་ཏུ་་་།
