འགལ་བ་དང་། བསླུ་བའོ། །རངྒི་ཎ་དང༌། འདྲེས་པ་མཚུངས། །ཤེས་རབ་གསལ་བ། གཉིད་སད་པ། །ཛཱ་ག་ར་ཞེས་པ་རྣམས་སོ། །གཉིད་དང༌། ཉལ་དང་། གཟིམས་པ་དང་། །རྨི་ལམ་ལེགས་ཕབ་ཅེས་ཀྱང་བྱ། །སྙོམ་ལས་དང་ནི་སྐྱིད་:སྙོམ་མོ། །གཉིད་ཡུར་བ་དང༌། མིག་འཛུམས་སོ། །སྨིན་བསྡུས། །བཞིན་བསྡུས། ཁྲོ་གཉེར་རོ། །ཁྲོ་བས་བལྟ་དང་། སྡང་མིག་གོ། །།ཕབ་ནས་བལྟ་དང༌། །ཀུན་ཏུ་བལྟ། །ཨིཀཱ་ཎཾ་དང་། དན་ནི། །ནིརྦ་ཎ་ན་ནི་བྷཱ་ལ་ནཾ། །རང་ཚུལ་དང་ནི་ངོ་བོ་ཉིད། །ཡང་དག་གྲུབ་དང༌། རང་བཞིན་དང༌། །ངེས་ཆགས་བདག་ཆོས། ངང་ཚུལ་དང་། །སྐྱེ་བ་ཞེས་ཀྱང་བྱ་བའོ། །སྡུག་བསྔལ་འཇིག་རྟེན། སྲིད་པ་དང་། །འཁོར་བ་རུལ་གནས་བཙོན་རའོ། །ཉིན་མཚན་ལོ་དང༌། ཁ་སྐད་ཅིག་དང༌། །དགོད་དང༌། དགའ་སྟོན་རྣམས་ལ་ནི་། :། ཞེས་པ་དཔལ་ལྡན་དཔལ་་འཛིན་སྡེ་ཡིས་བཀོད་པའི་མངོན་བརྗོད་མུ་ཏིག་གི་ཕྲེང་བ་མིང་གཞན་སྣ་ཚོགས་་་་གསལ་བ་ཞེས་བྱ་བ་ལས་མཐོ་རིས་ཀྱི་སྡེ་རྫོགས་སོ། །།བཀའ་ཚབ་ས་འོག་། རྐང་འོག། སྟོབས་ལྡན། ་གནས། །བཅུང་འོག། འོག་གནས། ཀླུའི་འཇིག་རྟེན། །བུ་ག་། ཁུང་བུ། པདྨའི་མིག། ནིཅུཐ་ནཾ། གསལ་མཐོངས་དང་། །བི་རོ་ཀ་དང༌། ད་རོ་དང་། །ཁོང་་་སྟོང་། ཙྪི་ཊཾ་ཀུ་ཧ་རཱི། །རནཾ་བ་པཱ། ཤུ་ཁིའོ། །ས་དོང་། ་གཏཱ་བཊ་དང་། །བུ་ག་དང་བཅས་ཤུ་ཊི་རཾ། །མུན་ནག། །སྣང་མེད། མུན་པ་ཅན། །མུན་པ་ཉིན་མོའི་སྒོ་ང་དང༌། །ནམ་མཁའ་འབྱོར་པ་ཉིན་མཐར་བྱེད། །ཕྱོགས་གོས། །z་ཨནྣཾ། །མཁའི་ལྤགས་པ། །མཚན་མོ་ཀླུ་གནས་སྒྲ་གཅན་།
