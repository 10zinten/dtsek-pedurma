བསྟན་འགྱུར། སྣ་ཚོགས། རྒྱལ་པོའི། །ཀ་ཀོ་བ་ཀུན་ལའོ༑། གུཉྫ། ཀཱ་ཀ་མོན་སྨན་ནུ་མ་ཙི་རི་དང༌། །པ་ཊ་ཧ་དང་སྒྲ་ཡི་ཆ། །དྲི་ཛ། བྲམ་ཟེ། སྒོང་སྐྱེས་སོ། །དྲི་ཛཱ་ངན་སྤོང༌། བུ་རྡུལ་བཟང་། །དྷཱ་ཛ༑་རྟགས་མཚན་ཁྲི་རྐང་དང་། །བ་དན་མཚན་རྟགས། ཤོཎྜི་ཀ་། ནི་་ཛ་བདག་པོ། རྟགས་པ་དང༌། །ཆུ༴་ན་བྱིས་པ་ཐོས་པར་བརྗོད། །རླུདྡོ་མཚོན་གྱི་ལས་ལ་འགྱུར། །ཀུཌ། འོག་ཏུ་ཕྱོགས་པ་ལ། །པིལྔ་གནོད་པ། པིཀྲྀ་གཡོ། །བིཥྞ་ཤིང་བལ། །ཡུང་བ་ལ། །པལྔ་འཁྲུགས་ལ་བརྗོད་བྱ་ལྡན། །པྲ་ཛཱ་རྒྱུད་དང་འཇིག་རྟེན་ལ། །པྲ་ཛཱ་འགྲོ་བ་ཚོགས་པ་དང༌། །ཚྭ་བ་ཀ་དང་ལམ་ལའོ། །བྷུ་ཛ་དང་ནི་བྷུ་ཛ་དག། དཔུང་་པ་དང་ནི་ལག་པ་དང་། །མ་དང་ཕ་དང་གཉིས་ཀ་ལའོ། །མརྫུ་བཙོ་བླག་པ་དང་ནི། །ཁྲུས་དང་དག་པ་་་་རྣམས་ལའོ། །རཛྫ་ལྷས་མ་ཡོན་ཏན་ཐགས་པ་དང་། །།གྲལ་དང་ཕྲེང་བ་རྩིག་པ་ལའོ། །རུ་ཙ་ནད་དང་ཨུངྒ་ལ། །ལཉྫ། །ད་ར་དང་མཐའ་ངོས་ལ། །།ལཱ་ཛཱ་འཛིན་དང་འབྲས་སོ་བ། །ལཱ་ཛ་གཤེར་བའི་འབྲས་ལའོ། །ལཱ་ཛི། པུ་ཤེལ་རྩེ་ལ་བརྗོད། །བཱ་ཛ་ཕྱོགས་ལ་ཡང་འགྱུར་ཏེ། །སྒྲ་དང་ཐུབ་པའི་བྱེ་བྲག་ལའང་། །བཱི་ཛཾ་མཆོད་སྦྱིན་དོན་དང་ལ། །མར་ལའོ་བཱི་ཛི་རྒྱུད་དང་ནི། །ཉེ་བར་ལེན་དང་ཁུ་བ་ལ། །བྱཱ་ཛ་ཀུན་ཏུ་འཛིན་དོན་དང་། །དེ་ཉིད་གཡོ་དང་་མི་སྟོན་པའོ། །བརྫུ་ཚོང་པ་དང་ནི་ཀློག། སཛྫ་གོ་བགོས་ཀུན་ཏུ་འབྱུང་། །སཉྫ་བདེ་འབྱུང་ཚོང་་པ་དང༌། །སཱ་ཛ་རྡུལ་དང་རང་སྐྱེས་དང༌། །ཁྲག་དང༌དེ་བཞིན་བུ་ལའོ། །ཨངྒ་ཛོ་ནི་སྒྲ་དང་དགའ། །།བུ་དང་བདེ་བས་དྲེགས་པ་་་དང༌། །ཡན་ལག་འཛིན་དང་གླང་པོ་ཆེ། །ཨངྒ་ཛ་ནི་ཁྲག་ལའོ། ་། ཨཎྜ་ཛ་ནི་བྱ་དང་ཉ། །རྩངས་པ་སྦྲུལ་དང་གླ་རྩི་དང༌། །སྒོང་སྐྱེས་རྣམས་ལའོ། །་། ཨམྦུ་ཛེ༑། མཚོ་སྐྱེས་དང་ནི་ནི་ཙུ་ལ། །ཆུའི་ལྕང་མ། ཀོལླི་ཛ་ནི་ཡུལ་གླང་པོ། །ཤཾ་།
