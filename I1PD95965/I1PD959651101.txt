བསྟན་འགྱུར། །སྣ་ཚོགས། པིཀྱི་ཨསྟ་རི་ཨཏྨ་ནེའི་ཚིག་ལ། རྀཏ་ལས་བྲིང་བྲེཉ་རྣམས་ཀྱི་ཞེས་པ་ལ་སོགས་པས་ཡང་ན་ཨིཊའོ། །ཨསྟ་རི། ཨསྟ་རཱི་ཥཱ་ཏཱམ། ཨཱ་སྟི་རཱི་ཥ་ཏ། ཨིཊ་མེད་པའི་ཕྱོགས་ལ་རྀཏ་མཐའ་རྣམས་ཀྱིའང་ངོ་ཞེས་པ་འདིར་རཾ་ཡིག་གི་ཉེ་བར་་་མཚོན་པ་ཉིད་ཀྱི་ཕྱིར་ཡོན་ཏན་བཀག་གོ། །རྀཏ་མཐའི་ཡོན་ཏན་མེད་པ་ལ་་་ཨིརའོ་ཞེས་པས་ཨིརའོ། །།།ནཱ་མིནའི་ར་པ་དག་ལ་ཞེས་པ་ལ་སོགས་པས་ཨུ་པ་་་དྷའི་རིང་པོ་ཉིད་དོ། །ཨབྷིནྡྷ། ཨབྷིནྡྷཏཱམ། །ཨམྦྷིནྡཏ། ཛཱ་ཤཱི་སྟནྡྷཱ་སྤུ་་་་་ཙམུ་ཙུ་གླུ་ཙུ་རྣམས་ཀྱི་ཡང་ངོ་ཞེས་པས་ཡང་ན་ཨཎའི་རྐྱེན་ནོ། །ཨརཱི་སཏྟི་་་་དག་གི་ཨཎ་ལའོ་ཞེས་པས་རཾ་ཡིག་གི་ཉེ་བར་མཚོན་པ་ཉིད་ཀྱི་ཕྱིར་ཡོན་ཏན་ནོ། །ཨ་ཛ་རཏ། ཨ་ཛཱི་རཱིཏ། གསལ་བྱེད་མཐའ་རྣམས། །བ་ཙ་ཨའི་ཡང་རིང་པོ་ཉིད་དོ། །ཨ་བཱ་ཀཱིཏ། ༌དྷུ་ཊ་ལས་ཀྱང་དྷུ་ཊ་ལའོ་ཞེས་པས་སིཙ་དབྱིའོ། །།།ཨཱ་པཱཀྐཱ་མ། ཨཱ་བཱ་ཨཱཏྨ་ནེའི་ཚིག་ལ། ཨ་པཉ། ་ཨ་པཀྵ་ཏཱམ། ་ཨ་པ་ཀྵེ་ཏ། ཨའི་ཡང་རིང་པོ་འོ་ཞེས་སྦྱོར་བ་ཐ་དད་པ་ལས་ཨུ་པ་དྷཱ་མ་ཡིན་པའི་་ཡང་རིང་པོ་ཉིད་དོ། །བྷཉྫ། ཨ་བྷཱ་དྷི་ཏ། བ་ས། ཨ་བཱ་སི་ཏ། བ་ཧ། ཧའི་རྙའོ་ཞེས་པས་ཌྷ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །ཨ་བཱ་ཀྲི་ཏ། །ཨའི་ཡང་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་བྱས་པ་ལ་ཧའི་རྙའོ་ཞེས་པས་རྟ་ཉིད་དོ། །།དྷུཊ་ལས་ཀྱང་དྷུཊ་ལའོ་ཞེས་་་པས་སི་ཙ་དབྱིའོ། །ག་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལས་ཏ་ཐ་དག་གི་་་དྷའོ་ཞེས་པས་དྷ་ཉིད་དོ། །ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ་ཞེས་པས་ཊ་སྡེ་ཉིད་དོ༑། །ས་ཧི་བ་ཧ་དག་གི་ཨ་ཡིག་གི་ཨོ་ཏའོ་ཞེས་པས་རྟ་དབྱི་ཞིང་ཨ་ཡིག་གི་ཨོ་ཉིད་ཀྱང་ངོ༌། །ཨ་བོ་ཞྭ་མ། ཨ་བཱཀྐུཨ་བཱཀྐིཨཱཏྨ་ནེའི་ཚིག་ལས། ཨ་བོ་རྟ། ཨ་བགྷཱ་ཏམ། ཨ་བཱག་ཏ། ཨ་བོ་ཆ་རུ་དྷི་ར། ཨིར་རྗེས་སུ་་ས་ང—།
