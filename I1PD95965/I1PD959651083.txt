བསྟན་འགྱུར། །སྣ་ཚོགས། ་༼པོའིབཱི་ཏཾ། །བྷུ་བཱ་ཡཱ་ཏཱ། བྷཱུ་བཱི་རི་ན། །རྣམ་པར་བཤད་པ་རྣམས་སོ། །།ཧུ་ཛུ་ཧུ་ཧུ་ཡ་ཏ། །བཱི་བྷི་ཡ་ཏ། ་བྷིའི་ཡང་ནའོ་ཞེས་བཤད་པར་བྱའོ་ཞེས་པའི་་་ཚིག་ལས་ཡང་ན་ཨི་ཉིད་དོ། །བི་བྷི་ཡཱ་ཏ། ཛི་ཧཱི་ཡཱ་?ཏ། པི་པི་་ཡཱ་ཏ། ཨོ་ཧཱཀ་། ཨི་ཉིད་ཐོབ་པ་ལ། ཛི་ཧ་ཏིའི་སཔྟ་མཱི་ལ་དབྱིའོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ༑། ཛོ་རྒྱུ་ཏ། ཇོ་ཧཱ་ཏཱ། ཛེ་ཧཱུརཾ། ཨིལླི་་ཡཱ་ཏ། ས་མིསྨཱ་ཏ། དཱི་དཱི་བྷཱུ་ཏི། :དཱ་དྷ་ཙཱ་ཏཱ'། ཛ་ཛ་རྩྭ་ཏ། པ་རསྨི་ཚིག་རྣམས་སོ། ཡི། ནེ་ནི་ཛཱ་ཏི། ནེ་ནི་ཛཱི་ཏ། བེ་བི་ཛཱ་ཏ། བེ་བི་ཛཱི་ཏ། བེ་བི་བྷཱུ་ཏ། བེ་བི་ཏཱི་ཏ༑རྔུ་དཱཉ། ད་:བྱ་ཏཱ། །ད་དཱི་ཏ། ད་:དྷུ་ཏཱ། དཾ་དྷཱ་ཏེ། བི་བྷཱི་ཡཱ་ཏ། བི་བྷཱི་ཏ། །གཉི་གའི་ཚིག་རྣམས་སོ། །མཱ་ད་མི་མཱི་ཏ། ཨོ་ཧཱང༌། ཛི་ཧཱི་ཏ། ཨཱཏྨ་ནེའི་ཚིག་དག་གོ། །དེ་བཞིན་དུ་ཨད་ལ་སོགས་པའོ། །།དི་བུ། བཱི་བྱཱ་ཏ། སཱི་བྱེ་ཏ། །ཥཱཉ། སུ་ནུ་ཡཱ་ཏ། སུ་ནཱི་ཏ། ཀྲྀ་ཊུ་ཡཱ་ཏ། །བྲི་ཎྜི་ཏ། གཉི་གའི་ཚིག་དག་གོ། །ཊུ་ཊུ། ཊུ་ནུ་ཡཱ་ཏ། ཧི་ནུ་ཡཱ་ཏ། ཨཱ་་སྣུ་ཡཱ་ཏ། དྷི་ནུ་ཡཱ་ཏ། :：ཀྲ་ཆུ་་ཡཱ་ཏི། ཀྲྀ་ཊུ་ཡཱ་ཏ། ཊུ་ཏུ་ད། ཡ། ཏུ་དེ་ཏ། རུ་དྷ་རི། རུནྣཱ་ཏ། རུནྡྲི་ཏ། །ཏ་ནུ། ཏ་ནུ་ཡཱ་ཏ། ་ཏ་ནིཏ། ཀླུ་ཀྲྀཉ༈ཡ་ལའང་ངོ་ཞེས་པས་ཨུ་ཡིག་དབྱིའོ། བྱེའོ། །ཀུམཱ་ཏ། ་ཀུཎྜི་ཏ། ཀྲྀ་ཎཱི་ཡཱ་ཏ། ཀྲྀ་ཎྜི་ཏ། ཙོ་ར་ཡ་ཏ། ངེས་པར་བོད་པ་ལ། ཨི་ཧ། བྷུཉྫི་ཏ༑ཀུན་ཏུ་བོད་པ་ལ། ་ད་ཨེ་ཧཱ་སཱི་ཏ། །གསོལ་བ་གདབ་པ་ལ། །མཱ་ཎ་པ་ཀཾ་མ་དྷ་བ་ཡེ་ཏ། ཡང་དག་པར་དྲི་བ་ལ། ཀིནྡྲ་ཁ་ལུ་བྷོ་བྱཱ་ཀ་ར་ཎ་མ་དྷཱི་ཡཱ་ཡ། ཨུ་ཏཙྪིནྡ་ཌ་དྷཱི་ཡཱ་ཡ། །དོན་དུ་གཉེར་བ་ལ། ལ་བྷི་ཡ་བྷིཀཱི། དངོས་པོ་དང་ལས་དག་ལ་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཛཱི་ཡེ་ཏ། ཨངྒེ་ཡེ་ཏ། སྦི་ཡེ་ཏ། ཤ་བྱེ་ཏ། བ་ཐོ་ཏ། ག་མྱེ་ཏ། ་དུ་བྱེ་ཏ། །དེ་བཞིན་དུ་གཞན་ལ་ཡང་བལྟ་བར་བྱ་བ་་་།
