སྣ་ཚོགས། ་ནིཤག་ཛ་པྲ་བོ་དྷ་ཀགླང་པོ་བསྐུལ་བྱེད། རཾ་མཉྫི་པི་ཡི་ཧ་ནི་ཨ་གི་རྣམས་་་ལས་ཨཱ་ནའོ། །རྀ་འགྲོ་བ་ལའོ། །མཉྫ་ཞེས་པ་མདོའི་བྱིངས་སོ། །པི་ཡ་་་ཞེས་པ་མདོའི་བྱིངས་སོ། །ཧ་ན་འཚེ་བ་ལའོ། །ཨ་གི་ར་གི་ལ་གི་འགྲོ་་་བའི་དོན་རྣམས་སོ། །འདི་རྣམས་ལས་ཨུ་ཥན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨ་རུ་ཥཙནྡྲ་མཱཟླ་བ། མཉྫ་ཥཀཱཋ་མ་ཡི་པི་ཡུ་ཥསྲྀ་ཌ་ཀ། ཧནུ་ཥབྱ་གྷ༌སྟག། ཨདྷུ་ཥདེ་ཝ་ག་མ་པྲ་ལས་ཀུ་ཥ་ནའོ། །གྲོ་ྋསྐྱོང་བ་ལའོ༑། འདི་ལས་ཀུ་ཥན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །པུ་རུ་ཥ.སྐྱེས་བུ། ན་རཿམི། །ཀྲྀ་ཊཾ་དག་ལས་ཨི་ཥནའོ། །ཧཱུཾ་རྣམ་པར་གཡེང་བ་ལའོ། །ཏཱཾ་རྒྱལ་བ་ལའོ། །འདི་དག་ལས་ཨཱི་ཥན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་རཱི་ཥགོ་མ་ཡཏ་རཱི་ཥས་མུ་དྲ་ཤི་རཱི་ཥ་ལ་སོགས་པ་རྣམས་སོ། །ཤི་རཱི་ཥཔུ་རཱི་ཥ་ཨམྦ་རཱི་ཥརཾ་ཛཱི་ཥ་ཞེས་སོགས་སྒྲ་འདི་རྣམས་ཨི་ཥ་ན་ཕ་རོལ་ཚིག་ཕྲད་དུ་བྱའོ། །ཤྲཱི་འཚེ་བ་ལའོ། །བཱི་ཥ་༌སྐྱོང་བ་ལའོ། །པྲཉ་སྒྲིབ་པ་ལའོ། །:། རཾ་ཛཱི་འགྲོ་བ་་་ལའོ༑། དེ་ལ་ཤྲཱི་ཎཱ་ཏིའི་ཡི་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །ཤི་རཱི་ཥབྲྀཀ་་ཛ་ཏིཤིང་གི་ཁྱད་པར། བཱི་ཎཱ་ཏིའི་ཨུ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །པུ་རཱི་ཥབཱིཥྛ'ཨཱད་སྔ་མའི་པྲ་ཎཱ་ཏིའི་ནུཀ་ཨཱ་ག་མའོ། །ཨམྦ་རཱི་ཥབྷབྦརཾ་:ཛི་ཥཱནཱ་ཡི་ཀཿདེ་བཞིན་གཞན་ཡང་བལྟ་བར་བྱའོ། །ཨ་བི་ལས་ཊི་ཥ་ཙའོ། །ཨ་བ་རཀྵ་བསྲུང་བ་ལའོ། །འདི་ལས་ཊི་ཥ་ཙའོ། །ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཨ་བི་ཥས་མུ་དྲརྒྱ་མཚོ། ཌིད་ཌྷ་ཨཎ་ཨ་ཉ་ལ་ཞེས་པས་དཱི་ཥའོ། །ཨ་བི་ཏི་ན་དཱི། ཆུ་ཀླུང༌། ཀིལླི་ཥ་ལ་སོགས་པ་རྣམས་སོ། །ཀིལླི་ཥརོ་ཧི་ཥ་ཏཱ་བི་ཥ་ཏཱ་བི་ཏི་བྱ་ཐི་ཥ་ཞེས་སོགས་སྒྲ་འདི་རྣམས་ཊིཥ་ཙི་ཕ་རོལ་ཚིག་ཕྲད་དུ་བྱའོ༑་། པི་ལ་བརྩེ་བ་ལའོ། །རུ་ཧ་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཏ་པ་ཀུན་ཏུ་་་།
