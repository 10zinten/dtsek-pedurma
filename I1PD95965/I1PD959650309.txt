སྣ་ཚོགས། ༼ནི་ལའོ༑། །དཱ་ཊཾ་དང་དྷཱ་ཌོ་འགྲོ་བ་ལའོ། །།བླ་ཊཾ་བསྔགས་པ་ལ་ཏི་པྲ་དང་ཏེ་པྲ་དང་བྦེ་པྲ་འགྱུར་བའི་དོན་ལའོ། །དྨེ་པྲ་ཞན་པ་ཉིད་ལའོ། །།ཏུ་བེ་པྲ་དང་ཀ་་པི་གཡོ་བ་ལའོ། །ཀེ་སྲི་དང་གེ་སྲི་དང་ཐེ་པྲ་ཡང་ངོ༌། །ཏྲ་པུ་ཥ་ངོ་ཚ་བ་ལའོ། །ར་བི་དང་ལ་བི་ཀུན་ཏུ་འབབ་པ་ལའོ། །་་། ཀ་བྲྀ་ཁ་དོག་ལའོ། །ཀྲྀ་དྲི་སྲུངས་ཆེ་བ་ལའོ། ་། ་བྲོ་རྒྱགས་པ་ལའོ། །ཤཝཱ་དང་ཤི་བྷཾ་:ར་བསྔགས་པ་ལའོ། །ཙཱི། ཤྲཱི་དང་ཙ་ར་བྷཾ་སྒྲ་ལའོ། ་། ཋ་བྷི་དང་སྟ་བྷི་དང་སྐ་བྷི་གེགས་ལའོ། །ཛ་བྷི་དང་ཛྲ་བྷི་ལུས་རྣམ་པར་ཉམས་པ་ལའོ། །བལྷ་ཟ་བ་ལའོ། །གལྷ་སྲུངས་ཆེ་བ་ལའོ། །བྷུ་བྷུ་རེངས་པ་ལའོ། །གྷི་ཎི་དང་གླུ་ཎི་དང་གྷཱ་ཎ་འཛིན་པ་ལའོ། །སྒྱུ་ཎ་དང་། ཎྞ་འཁྱམ་པ་ལའོ། །པ་ཎ་བསྟོད་པ་ལའོ། །པ་ཎ་ཐ་སྙད་ལའོ། །བྷ་མ་ཁྲོ་་བ་ལའོ། །ཀཱ་མཱུ་ཥ་བཟོད་པ་ལའོ། །ཨ་པ་དང་བ་པ་དང་མ་པ་དང་ཡ་པ་དང་ཙ་པ་དང་ཏ་པ་དང་ཎ་པ་དང་ར་པ་འགྲོ་བ་ལའོ། །ད་པ་བསྲུང་བ་ལའོ། །ཡི་ཐགས་འཐག་པ་ལའོ། །།པཱུ་ཡི་འགྲོ་བ་ལའོ། ་། ཀླུ་ཡི་:འགྲོ་བ་ལའོ། །སྒྱུ་ཡི་རྣམ་པར་འདར་བ་ལའོ། །སཱ་ཨཱི་དང་ཨོ་དང་ཨོ་སྟུ་ཡི་འཕེལ་བ་ལ་ཏཱ་པྲ་བརྒྱུད་པ་ལའོ། །།ཁ་ལ་གཡོ་བ་ལའོ། །བ་ལ་ཀུན་ཏུ་སྒྲིབ་པ་ལའོ། །མ་ལ་དང་མལླ་འཛིན་པ་ལའོ། །།བྷ་ལ་དང་བྷལླ་བརྒྱད་པ་ལའོ། །ཀ་ལ་གྲངས་ལའོ། །ཀ་ལ་མི་གསལ་བའི་སྒྲ་ལའོ། །ཏེ་བྲི་དང་ཏི་ཤྲཱི་རྩེ་བ་ལའོ། །ཤེ་བྲི་དང་ཥ་ཤྲཱི་དང་ཀེ་བྲོ་དང་ཁེ་བྲི་དང་གེ་བྲི་མེ་བྲི་མྨེ་བྲོ་དང་ཕྲོ་བྲོ་དང་བླ་བྲོ་དང་གླུ་བྲོ་བཀྲུ་བ་ལའོ། །རེ་བྲོ་སྐྱལ་བར་འགྲོ་བ་ལའོ། །དྷ་ཀླུ་ད་ང་དྷི་ཀ་ཡང་དག་པར་གསལ་བ་ལའོ། །བྲྀཀྵ་སྒྲིབ་པ་ལའོ། །ཤེཀྵ་རིག་པ་ཉེ་བར་ལེན་པ་ལའོ། ་། བྷིཀྵ་སློང་བ་ལའོ། །ཀླུ་བུ་གནོད་པ་ལའོ། །རཀྟ་འཕེལ་བ་ལའོ། །དིཀྵ་མགོ་ཟླུམ་ཉིད་ལའོ། །ཨཱིཀྐ་བལྟ་བ་ལའོ། །ཨི་ཥ་ཚོལ་བ་ལའོ། །བྷཱ་ཥ་བརྗོད་པ་ལའོ། །སྤྱན་གཤར་བ་།
