ཛ་སྭ་བྷཱ་བེ་ཨདྷུར་དརྗེ་ཤཱ་ཁཱ་ལ་ཏྲ་པུཥྤ་ཕ་ལཱ་དཱི་ནཱ་མ་བྷ་བོ་བྷ་བ་ཏི། ཀུ་ཏོ་མེད་ན་མྱུ་གུ་དང་སྡོང་བུ་དང་ཡལ་ག་དང་ལོ་མ་དང་མེ་ཏོག་དང་འབྲས་བུ་སོགས་མེད་པར་འགྱུར་ཏེ། ཅི་བཱི་ཛ་པཱ་བྷཱ་བཱཏ། ནོ་བྷ་ཡ་ཏ། ཨུ་བྷ་ཡ་དོ་ཥ་དུཥཱ་ཏྭཱཏ། ཏསྨཱ་ད་སིདྡྷལས་ཞེ་ན་ས་བོན་མེད་པའི་ཕྱིར་རོ། །གཉི་ག་ལས་ཀྱང་མིན་ཏེ། གཉིས་ཀ་སྐྱོན་གྱིས་སུན་ཕྱུང་བའི་ཕྱིར་རོ། །ཀཎྜ༑ཨི་ཏི་ཨཱི་ཤྭ་ར་ཀཚྭ་ཏཱ་ནི་རཱ་ཀྲྀ་ཏིཇི་ཙྩོ་རེ་ཀ་ཀཎྜི་ཏཱ་ནི་རཱ་ཀ་ར་ཎཾ་དེའི་ཕྱིར་བྱེད་པ་པོ་མ་གྲུབ་བོ། །ཞེས་པ་དབང་ཕྱུག་བྱེད་པ་པོ་ཉིད་སེལ་བ་དང་ཁྱབ་འཇུག་བྱེད་པ་པོ་གཅིག་ས་མཱསྟ་མི་ཏི། ཀྲྀ་ཏི་རི་ཡ་མཱ་ཙཱརྱ་ཤྲཱི་ནཱ་གཱརྫུ་ན་པཱ་དཱ་ནཱ་མི་ཏི། ཉིད་སེལ་བར་བྱེད་པ་སློབ་དཔོན་དཔལ་ལྡན་ཀླུ་སྒྲུབ་ཀྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།ཞ་ལུ་ལོ་ཙཱ་བ་དྷརྨ་པཱ་ལ་བྷ་དྲས་བསྒྱུར་བའོ། །ཏ་སྱ་ཤིཥྛ་ཎ་ལི་ཁི་ཏཾ། །བྱང་བདག་དུ་བསྡུར་མཆན། གོ་བ་དང་ང་མཚན་བྱང་འདི་《སྡེ་»«པེ་»«སྣར་ལྟར་བཀོད། «པེ་»«སྣར་»ཏིརྫི། «སྣར་»ཀཚྭ། ®«པེ་»དཱམྦྷ་ཛནྣཾ། །«པེ་»«སྣར་»སྡོམས། པོ«པེ་»ཀུ་རུ་ཏ། «སྣར་»ཀུ་རུ་ཏེ། «པེ་»«སྣར་»དཔ། །(«པེ་»«སྣར་»ནཱབྷེ། «པེ་»«སྣར་»—དའདི། ག་བ«སྣར་»དདཔེ།  ེ«པེ་»«སྣར་»hཤམ། «པེ་»«སྣར་»རཱན། «སྣར་»འབྲལ། །«པེ་»«སྣར་མིཋ། «པེ་»«སྣར་སྐུ། །«པེ་»«སྣར་»དགཉིས། «པེ་»«སྣར་»pཅིག«པེ་《སྣར་སྐྱི། །བདེ་དང་བདེ་རྒྱ་བྲམ་དུ་འབབ་པ་དང་། ལ་ཕྱི། བཞིན་དེ་ཡིས་ཡིག་བ་ནི་ཀུན་མཁྱེན་ཡེ་ཤེས་དའ་ལོ། །ཡང་དང་ཤེས་རབ་ཕ་རོལ་ལ་བདག་པ་སྒོམ་པར་བྱེད་འདོད་པ་ལས་དང་བདུད་དད་གུ་ཡི་དོན་ལ་ཡེ། །བབས་པར་བྱ་བ་བསྒྲུབ་པ་འོ་བྱེད་ལས་ས་བ་དབག་བྱུང་། །སྐྱེ་མཆེད་དྲུག་བདེ་ལ་དབང་པད་འབྱུང་བ་ཞེས་།
