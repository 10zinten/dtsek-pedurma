་པ་ག:བསྟན་འགྱུར། སྣ་ཚོགས། ༼ཝི་་་་་སྐྱོ་ཎ་ཡོན་ཤུ་བ། རྗེས་སྐྱེས་དང་། །ནག་པོ། །ཚངས་སྐྱེས། །མི་བཟད་མིག་ཟ། འོད་ཟེར་བདུན་པ། གོས་སྔོན་ཅན། །གཤིན་རྗེའི་བ་དན། ཉི་མའི་བུ། །སྒྲ་གཅན༑གཟའ་ཡི་རྣ་བ་འཛིན། ན། །མཐོ་རིས་སྣང་བྱེད། བྲ་ཉེ་སྐྱེས། །སེང་གེ་མོའི་བུ། མུན་པ་ཅན། །མགོ་ཟླུམ། ལམ་ནག། ཟླ་བ་འཇོམས། །ཉེ་བར་འཕར་འགྲོ། །སྤྲིན་ལས་ནི། །རྣམ་རྒྱལ་དང་ནི། ཤ་ཟའོ། །དུ་བ་མཇུག་རིངས་སྦྲུལ་ཅན་དང་། །ལ་། གཙུག་ཕུད་ཅན་དང་། སྐྲ་གཉིས་པ། །སྐག་ལས་སྐྱེས་དང༌། །ཚེ་སྟེང་འགྲོའོ། །དྲང་སྲོང་བདུན་ཏེ་འོད་ཟེར་ཅན། །རྒྱུན་ཤེས། འོད་ཡངས། །ཡན་ལག་སྐྱེས། །བུ་ལ་ཧ་ཡི་མཆོད་སྦྱིན་དང་། །གནས་འཇོག་། སྣ་ཚོགས། །་་གཙུག་ཕུད་ཅན། །།ཁྱིམ་རྣམས་འཆར་བ་དུས་་་སྦྱོར་དང་། །།ཁྱིམ་གྱི་སྒོ་སྟེ་དེ་དག་ནི། །ཉིན་དང་མཚན་དུ་ལོངས་སྤྱོད་པ། །ལུག་དང་གླང་སོགས་ཤེས་པར་བྱ། །གྲིབ་མའི་ལམ་དགུ་ཚིགས་འདྲའོ་དང་ལྷ་ཡི་ལམ། །གཟའ་ཡི་མུ་ཁྱུད་ནམ་མཁའི་གནས། །འོད་གཟུགས། ཉི་མ། འོད་འབྱིན་་དང་། །ཉིན་མོའི་ནོར་བུ། །ཉིན་མོའི་འདྲེན། །གསལ་བའི་ནོར་བུ། །ནམ་མཁའི་ནོར། །སོ་སོར་ཉིན་བྱེད། ཉིན་མོར་བྱེད། སྣང་བྱེད། འོད་ལྡན། བ་རུ་ཎཱ། །འོད་འགྲོ། རྒྱས་བྱེད། མཁའ་འགྲོ་དང༌། །རྣམ་བསྒྱུར། སྣང་བྱེད། པ་ཏི་བ། ་། ཆ་ལྡན། །ངེས་བསྲེག། འོད་ཟེར་ལྡན། །འོད་ཀྱི་རྒྱུན་དང་། ཨ་སྲི་ཕྱོགས་སྣང་བྱེད། ག་བ་སྟེ། །སྒྲོལ་བྱེད། སཱུ་ར། སཱུ་ར་བ། །གདུང་བྱེད། །ཚང་བྱེད། རྣམ་སྣང་བྱེད། །མུན་སེལ། ཧཾ་ས། མ་ག་དང་། །འོད་སྟོང་བྱེད་དང༌། འཛིན་པ་དང༌། །ལུས་སྐྱོབ་ཉིན་བྱེད་འདམ་སྐྱེས་་མཚན༑:། ཉིན་མོའི་བདག་པོ། །པདྨའི་བདག། ཕྱོགས་བདག། མཆོག་འདོད། ཆར་འབེབས་དང་། །མཆོད་ལྡན་བདུན་གྱི་བདུན་པ་དང་། །།
