ལོའི་རྒྱ་གར་གྱི་བྱ་བ་དང་། དམར་པོ་དག་ཁོ་ན་བྱུས་འོ་ཀ་ཁ་སློབ་པའི་ཟླ་བས་ན་ངས་དེ་དག་ལ་ད་པའི་ཁ་ཁ་སོས་པ་དང་། ཁ་བ་ནག་པ་བ་ཁྱོད་ལ་བདེ་བ་༄༅། །ལྷ་དབང་ཕྱུག་གིས་ཚོགས་བདག་ལ་བ་མོ་དང་དགའ་བས་ཡིད་མ་གནང་བའི་ས་མུ་དྲ་ཕ་ལའི་སྦྱོར་བའི་བབ་ཁ་བསྒྱུར་དང་བཅས་པ། ਸਨਲਲਈਸਾਬདད་པ། རྒྱུ་མེད་དོ། །པར་ནང་དུ་། །པ་ད་༄༅། །ལྷ་དབང་ཕྱུག་གིས་ཚོགས་བདག་ལ་གནང་བའི་ས་མུ་དྲ་ཕ་ལའི་སྦྱོར་བ་ཁ་བསྒྱུར་དང་བཅས་པ་ནི། དེ་ཡང་ཐོག་མར་ས་མུ་དྲ་ཕ་ལ་མང་ཉུང་ཅི་རིགས་པ་ལྔ་སྣོད་གཙང་མར་བླུག་ལ་ཞག་བདུན་གྱི་བར་རའི་ཆུས་སྦངས་ཏེ་ཁུ་བ་ཕོ། དེ་གྲིབ་བསྐམས་བྱས་ལ་ཞ་ཐག་ཆོད་པ་དང་། སླར་ཡང་བའི་ཆུས་ཞག་བདུན། བངྒ་རའི་ཁུ་བས་ཞག་བདུན། ནི་ར་གུཎྜའི་ཁུ་བས་ཞག་བཞི། ད་དྷུ་རའི་ཁུ་བས་ཞག་གསུམ། འབྲས་ཆུ་སྐྱུར་མོས་ཞག་གཅིག་རྣམས་སྤངས་་སྐེམས་གོང་ལྟར་རིམ་བཞིན་བྱས་ལ། དེ་ནས་ཞིབ་པར་བཏགས་ལ་ཐུན་ཚད་་ཉིས་འགྱུར་ཙམ་རེ་ནི་ར་གུཎྜའི་ཁུ་བས་རྟ་བྱས་བཏང་བས་མཛེ་རིགས་བརྒྱད་གང་གིས་ན་ཡང་རྩ་བ་ནས་ཐོན་ཞིང་ཁོང་སྐྲན་དང་ཡན་ལག་ཁོལ་བུར་སྐྲངས་པ་ཞི་བར་འགྱུར་རོ། །།རའི་ཆུས་རྟ་བྱས་ཁོང་དུ་བཏང་ཞིང་དཔྲལ་བར་བྱུགས་ན་མགོ་བོ་ན་བ་ལ་ཕན་ནོ། །རའི་ཆུར་བརྡར་ཏེ་མིག་ལ་བྱུགས་ན་ལིང་ཏོག་མེད་པར་་་།
