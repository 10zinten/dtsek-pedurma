།སྣ་ཚོགས། ཝི། ཨིཏ་རྗེས་སུ་འབྲེལ་པ་རྣམས་བརྗོད་པར་བྱའོ། །ཎི་དི། སྨད་པ་ལའོ། །ཨི་ཡིག་ནི་ཨིཏ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་རྣམས་ཀྱི་ཡོན་ཏན་མེད་པ་ཞེས་པའི་་་ཁྱད་པར་གྱི་དོན་ཏོ། །འདི་ཉིད་ཀྱིས་ན་ཡིག་ཨཱ་ག་མའོ། །རྗེས་སུ་ང་རོ་དང་་་སྡེ་པའི་མཐའ་དག་གོ། །ནིན་ཏི། པྲ་ཎིནྡ་ཏི། །པྲ་ནིནྡ་ཏི། བཱ་ཚི། འདོད་པ་ལའོ། །བཱཉྩ་ཏི། པ་རསྨིའི་ཚིག་དག་གོ། །།སྦ་དི་ཅུང་ཟད་གཡོ་བ་ལའོ༑། སྤྱནྡ་ཏེ། པྲ་སྤནྣ་ཏེ། པ་རི་སཱནྡ་ཏེ། ལ་བི་ཀུན་ཏུ་འབབ་པ་ལའོ། །ལམྦ་ཏེ། བྲ་ལམྦ་ཏེ། །ཨ་ལམྦ་ཏེ། ཁ་དི། གསང་པོར་སྨྲ་བ་དང་བསྟོད་པ་དག་ལའོ། །:བརྡ་ཏི། ཨ་པི་བནྡ་ཏེ། པ་ཤ་ཀི་དོགས་པ་ལའོ། །ཤངྐ་ཏེ། ཨཱི་ཤཱངྒ་ཏེ། མ་:ཐི། ཀ་ཐི་མྱ་ངན་ལའོ། །མནྡྲ་ཏེ། ཨུཏྟནྠ་ཏེ། ཛ་བྷ། ཛྲ་བྷི། ལུས་རྣམ་པར་ཉམས་པ་ལའོ། །།ར་དྷི། ཛ་བྷ་དག་གི་དབྱངས་ལའོ་་ཞེས་པས་ནཱ་ཡིག་ཨཱ་ག་མའོ། །ཛམྦྷ་ཏེ། ཛིམྨཱ་ཏེ། ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཊྛི་བུ། །ཀཱི་བུ། སེལ་བ་ལའོ། །བྱིངས་ཀྱི་དང་པོའི་ཥའི་སའོ་ཞེས་པའི་སར་་མི་འགྱུར་རོ། །ཎྜི་བུ། ཀླུ་མུ། ཨཱ་ཙ་མ་རྣམས་ཀྱི་ཨནའོ་ཞེས་བསྟན་པ་ལས་སོ༑། ཀླ་མུ་དམན་པ་ལའོ། །ཙ་མུ། ཚ་མུ། ཛྷ་མུ། ཛི་མུ། ཟ་བ་་་་ལའོ༑། ཊྛི་བུ། ཀླུ་མུ། ཨཱ་ཙ་མ་རྣམས་ཀྱི་ཨན་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ལྕི་བ་ཏི། །ནིཥྛ་བ་ཏི། ཀླུ་མ་ཏི། ཨཱ་ཙ་མ་ཏི། ཀྲ་མུ། རྐང་པ་བསྐྱོད་པ་ལའོ༑། ཀྲ་མའི་པ་རསྨི་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཀཱ་མ་ཏི། སཾ་ཀཱ་མ་ཏི། ཟ་ཨུ་བ་ཀཱ་མ་ཏི། ་ཉེར་བསྒྱུར་མེད་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ༑། ཀཱ་མ་སྟེ། །གམྨཾ། ་འགྲོ་བ་ལའོ། །ཨི་ཥཱ་འདོད་པ་ལའོ། ་། ཡ་མུ་ཉེ་བར་འགོག་པ་ལའོ། །གམ་ཨི་ཀྵ་ཡཾ་རྣམས་ཀྱི་ཚའོ་ཞེས་ཚ་ཉིད་དོ། །དབྱངས་ལས་ཕ་རོལ་ཚ་ཡིག་གཉིས་སུ་འགྱུར་རོ་ཞེས་པས་ཚ་ཡིག་གཉིས་སོ། ་འ། སྒྲ་མེད་—།
