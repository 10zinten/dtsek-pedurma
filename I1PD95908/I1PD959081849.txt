བསྟན་འགྱུར། དབུ་མ། ༼ཨ༽::(«པེ་»«སྣར་+hཆེན་པོ། (ཉེ«པེ་»«སྣར་+རྒསེམས་ཅན་རྣམས་བཞུགས་པའི་ཆེད་དུ། ཚེ«པེ་»«སྣར་+རྒསངས་རྒྱས་ཀྱི་བྱང་ཆུབ་སེམས་དཔའི་གནས་སྐབས་དེའི། «སྣར་»+ཆདང་རྒྱལ་རིགས། །ཚེ«པེ་»《སྣར་+རྐབཟང་བ། «པེ་»«སྣར་+ཆགསལ་བ། དུ«པེ་»«སྣར་+རྐབསླབ་པ་མཐའ་::དག་མི་བསྲུང་བ། ་འཁྱེ«པེ་»«སྣར་+ཆབསླབ་པ་མཐའ་དག་བསྲུང་བས། འགྱེ«པེ་»«སྣར་»+ཆདང་་་་ལྡག་པ་དང་བཏུང་བ་དང་འཇིབ་པ། འགྱེ«སྣར་»+hསྟོན་པ། ་འཉེ«པེ་《སྣར་»+ཆདེ་བཞིན་ཚུལ་ཁྲིམས་ནས་བཟོད་བརྩོན་ཏིང་ངེ་འཛིན་དང་། །ཤེས་རབ་སྤྱོད་ཚེ་སྔར་གྱི་གནས་སྐབས་གང་། །མ་ཚང་མེད་དང་ཉིད་ཀྱི་སྤྱོད་པ་ཀུན། །སྐུ་ཡི་བ་སྤུའི་ཁུང་བུར་གསལ་བར་སྟོན། །ཅེས་པ་ལྟར། གཉེ«པེ་»«སྣར་»+hརྫོགས་པའི་སངས་རྒྱས། འགྱི«པེ་«སྣར་སྤྱད། །ཕྱི་ཡིན་པ་ཚོད་དཀའ་འོ། །«པེ་《སྣར་པའི། :ི«པེ་»«སྣར་»+རྒསངས་རྒྱས་ཀྱི་གནས་སྐབས་ན། EnQ«པེ་»«སྣར་+ནསངས་རྒྱས་དུས་ན། «པེ་»«སྣར་+རྐབཤད་མ་ཐག་པ་ལ་སོགས། མེ«པེ་《སྣར་+དབྱང་་་་་ཆུབ་ཀྱི། ་«པེ་»«སྣར་གཟུང་། ཚ«པེ་»«སྣར་»+ཆ་སོ་སོ། «པེ་»«སྣར་»++སྣ་ཚོགས། ན་ཁོ་«པེ་《སྣར་+ཇམཐས། ཁྱེ«པེ་»«སྣར་+hསྟོན་པར་སྤྱོད། ཀྱེ«པེ་»«སྣར་བར། (འུ«པེ་»«སྣར་+hའགྱུར་བ། ཉེ«པེ་»«སྣར་+hཕྱོགས་མཐའ་ཡས་པའི། *ཁྱེ«ཅོ་པར། «པེ་པར། ཚེ«པེ་»སྣར་+དསྦྱིན་པ་དང་སྦྱིན་པ་ཆེན་པོ་དང་མཆོག་ཏུ་གྱུར་པའི་སྦྱིན་པ་ལ་སོགས། །ཁྱེ«པེ་»དེ་ལ་བདག་«སྣར་+ནསོ་སོ་སྐྱེ་བོ་དང་ཉན་ཐོས་དང་རང་སངས་རྒྱས་དང་། །ཏེ«པེ་»མོངས། 《ཚེ«པེ་»«སྣར་»+hསྣང་བ་རང་བཞིན་མེད་པ། འཚེ«པེ་»«སྣར་»+hསླར་ཡང་མི་གཡོའི་སྐུ་མངའ་ཁྱོད་ཀྱིས་སྲིད་གསུམ་ཁྱོན་ནས་སྤྲུལ་རྣམས་ཀྱིས། །གཤེགས་པ་དང་ནི་བལྟམ་དང་བྱང་ཆུབ་ཞི་བའི་འཁོར་ལོའང་སྟོན་པར་མཛད། །དེ་ལྟར་ཁྱོད་ཀྱིས་འཇིག་རྟེན་གཡོ་བག་སྤྱོད་ཅན་རེ་བའི་ཞགས་པ་ནི། །མང་པོས་བཅིངས་པ་མ་ལུས་ཐུགས་རྗེས་པ། མྱ་ངན་འདས་པ་དཀྲི་བར་མཛད། །ཅེས་པའི་འགྲེལ་པར། ་ན་«པེ་»«སྣར་»+hསྤྲུལ་པའི་སྐུ་འཆད་པར་་འགྱུར་ཏེ། ཉེ«པེ་»«སྣར་+དཔའི་ཉོན་མོངས་པ་བག་ཆགས་དང་བཅས་པ་སྤངས་པས་དེ། འཉེ«པེ་《སྣར་+རྒགདུལ་བྱ་རྣམས། ལ་ལ་དག་པ་am། ཁ་བཀད་པ་(«པེ་»«སྣར་»+hཔའམ་གཤེགས། «པེ་»«སྣར་»+hའདོད་ཆགས་ལ་སྤྱོད་པ་སོགས་དང༌། «པེ་»མོ་གྱིས། ︽«པེ་»«སྣར་+ཇལ་རེ་བ། ཚུ«པེ་《སྣར་+དམངོན་པར་རྟོགས་པའི་རྒྱན་དུ། «པེ་»བ་ཀྱི་«སྣར་»རྟོག་པའི་བརྒྱན། @«པེ་»«སྣར་+bརྒྱལ་བའི་ཡུམ་གྱི་རྒྱས་པ་དང་བར་པར་བཀའ་སྩལ་པ་ལྟར། །འ«པེ་《སྣར་»+གསེམས་དང་བར་བྱས་པ། «པེ་»«སྣར་པས། «པེ་»«སྣར་བསྟན། ཀྱེ«པེ་»«སྣར་པ་དེས། དགྱེ«པེ་»«སྣར་+hརང༌། ཉེ«པེ་»«སྣར་+ནངན་པའི། «པེ་»«སྣར་»མི་་པ་རྣམས། ཨུ«པེ་»«སྣར་»ལྟ་བར། ་«པེ་»«སྣར་»+དབྱང་ཆུབ་ཀྱི། («པེ་+སུ། ཏེ«པེ་»། 《སྣར་»+hགདུལ་བྱ་རྣམས་ཀྱང་། འཁྱེ«པེ་»«སྣར་»+hརྫོགས་སངས་རྒྱས། A''། དེ་«པེ་»«སྣར་»+ཆདབུ་མ་ལ་འཇུག་པའི་འགྲེལ་པར། «པེ་»«སྣར་»+hཡོན་ཏན། ཚུ«པེ་»«སྣར་»ཀྱི། ®«པེ་»«སྣར་»+kཤིང་མཐར་ཐུག་པར་གྱུར། །@«པེ་»«སྣར་+དཁྱད་པར་གྱི་ཆོས། །
