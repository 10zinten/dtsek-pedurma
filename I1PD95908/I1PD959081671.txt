བསྟན་འགྱུར། དབུ་མ། དོན་ཁོ་ནར་བྱེད་པ། O«པེ་»«སྣར་»+ཇལུས། »«པེ་»«སྣར་+ཆརྐུ་བའི། ་ཁྱོ«པེ་»«སྣར་»::+hཅི་ཞིགགྱེ«པེ་»«སྣར་+hཅིང་ལྷུར་ལེན། ཉེ«པེ་»«སྣར་གཡེང་བས། ་ཉེ«པེ་»«སྣར་»པའང་། རྩེ«པེ་»«སྣར་+ང་དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་བསྟན་པར། །རང་ཉིད་ལས་བྱུང་བ་དང་། རང་ཉིད་ཆུང་མ་ལས་བྱུང་བ་དང༌། ཆུང་མར་མི་རིགས་པ་ཆུང་མར་བྱས་པ་ལས་བྱུང་བ་དང་། ཆུང་མ་མ་ཡིན་པ་ལས་བྱུང་བ་དང་། སྐྱེས་པ་ལས་བྱུང་བ་དང་། དུད་འགྲོ་ལས་བྱུང་བ་དང་། ཁྱེའུ་དང་བུ་མོ་ལས་བྱུང་བའོ་ཞེས་བདུན་གསུངས་པ་འདིར་ནི། བགྲོད་མིན་བགྲོད་པར་འདོད་པ་ཡིས། །ལོག་པར་གཡེམ་པ་་་་རྣམ་པར་བཞི། །ཞེས་པ་ལྟར། «པེ་»«སྣར་»+གྱི། ཚེ«པེ་»«སྣར་+ཆགཉིས་ཀྱིས་གཉིས་སྦྱར་་་བའོ། །:ཁྱེ«པེ་《སྣར་»+hཤིང་གཞན་གྱི་མ་བྱིན་པ་ལ་རང་གི་གླ་རྔན་བྱིན། །ཏེ«པེ་»སྣར་»+ཆབགྲོད་བྱ་དང་འདྲ་བ་མ་ཡིན་པ་ལ་བགྲོད་བྱ་མ་ཡིན་པ་ཞེས། ་ཚེ«པེ་»«སྣར་དེ་དང༌། +ཚེ«པེ་»«སྣར་»དང་། +ཆབགྲོད་བྱ་མ་ཡིན་པ། ཚེ«པེ་»«སྣར་འགྲེལ། མེ«པེ་»«སྣར་»+ཆདགེ་་་་་་བཅུའི་ལས་ཀྱི་ལམ་བསྟན་པར་ཡང་། རང་གི་བདུན་བརྒྱུད་ལས་མ་འདས་པར་འབྲེལ་པར་གྱུར་པ་དང་ཞེས་་བཤད། དོན་འདི་ལ་འདུལ་བ་མདོར། ཕ་མ་དག་གི་བདུན་པ་ཚུན་ཆད་ལས་གང་ཡང་རུང་བ་ཞིག་ཏུ་སྔོན་ཆད་གྱུར་པ་ཡང་ཉེ་དུ་ཡིན་ནོ་ཞེས་གསུངས་ཏེ། དེའི་སྤྲོས་པ་རྒྱ་ཆེར་བམ་པོ་ལྔ་བཅུ་པར་ཤེས་པར་བྱའོ། །«པེ་»«སྣར་མ། ཉི«པེ་»«སྣར་གཟུང་བས། +ཆདངོས་སུ་ཆུང་མ་བླངས་པ་དང་། མ་བླངས་ཀྱང་བདག་གིར་བྱས་པ་དང་། གཞན་གྱི་སླ་སྔར་བྱིན་པའི་སྨད་མ་ནི་ཕ་རོལ་པོ། འ«པེ་»«སྣར་»+དགོ་ཏྲ། །«པེ་»སྣར་»+ཆགཞོན་ནུ་མ་སོགས་ཤེས་འཇུག་ཊཱིཀ་ཕ་མ་དང་འབྲེལ་པ། «པེ་»«སྣར་»+hརིམ་་་ནུ་བཞིན། «པེ་»«སྣར་+hའདི་དག་ལ་འགྲོ་ནའང་འདོད་ལོག་ཏུ་འགྱུར་ན་གཞན་གྱི་ཆུང་མ་སོགས་ལྟ་ཅི་སྨོས་ཤེས། ཚེ«པེ་»«སྣར་»འམ། ::«པེ་»«སྣར་»+hཕ་རོལ་པོའི་ཆུང་མ། «པེ་»«སྣར་»+དཆུང་མ། ཚེ«པེ་»«སྣར་+དམནར་བ། འགྱེ«པེ་»«སྣར་+hལྷས་བྱིན་གྱི་ཁྱིམ་ཐབ་ལ་བགྲོད་པར་འདོད་ཀྱང་མཆོད་སྦྱིན་གྱི་ཁྱིམ་ཐབ་ལ་བགྲོད་པ་ལྟ་བུ། མེ«པེ་»«སྣར་»+ལ། བརྩེ«པེ་»«སྣར་»སྟེ། །ལོ«པེ་»«སྣར་+hཕོ་མོའི། «པེ་»«སྣར་+སྟེ། +རྐབགྲོད་བྱ་མིན་པ། ཚེ«པེ་»«སྣར་»+hཡན་ལག་མ་ཡིན་པས། ཁྱེ«པེ་»«སྣར་»+ཆདགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་བསྟན་པར། འདོད་ལོག་བདག་ཉིད་མི་རིགས་པའི་ཡུལ་དྲན་ནས་དབང་པོ་ལས་རང་ངམ་གཞན་གྱིས་ཁུ་བ་འབྱིན་པར་བྱེད་པའོ་ཞེས་གསུངས་པ་ལྟར། ཉེ«པེ་»སྣར་+དམཛོད་འགྲེལ་དུ། ཁྱེ«པེ་»སྣར་+ཆསྤྱད་པར། འགྱེ«པེ་»«སྣར་+hའམ་མི་་་་འོས་པ། «པེ་»«སྣར་»མཛོད་འགྲེལ་རྒྱལ་སྲས་མར། སྦྲུམ་མ་ལ་ཉལ་པོ་བྱས་ན་ལྟོ་ན་བུ་འདུག་པ་ལ་་་བ། གནོད་པར་འགྱུར་རོ། །བུ་ནུ་ཞོ་འཐུང་བའི་གནས་སྐབས་སུ་བུ་ཡོད་པའི་བུད་མེད་ལ་མི་ཚངས་པར་སྤྱོད་ན་་་་:དེའི་ནུ་ཞོ་སྐམ་པའམ་དེའི་ནུ་ཞོ་མི་བསོད་པར་འགྱུར་བས་སོ་ཞེས་གསུངས། འགེ«པེ་»«སྣར་+ཆབང་མ་་་་སངས་པ། གཉེ«པེ་»«སྣར་»+ཆབསྙེན་གནས་བསྲུང་བ་ལ་སོགས། བློའི་པད་མའི་«པེ་»«སྣར་+ཆདུས་མ་ཡིན་པ། «པེ་»«སྣར་»+དམྱ་ངན་ལ་སོགས་པས། «པེ་«སྣར་»པར། «པེ་《སྣར་+དབསྙེན་གནས་བསྲུང་བ་ལྟ་བུ། ཚུ«པེ་»«སྣར་པ། «པེ་»«སྣར་»སྤྱོད་པར། །།
