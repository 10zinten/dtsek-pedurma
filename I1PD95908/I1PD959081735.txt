བསྟན་འགྱུར། དབུ་མ། ཆབ་ལ་ཁོང་ཁྲོ་བའི། «པེ་»«སྣར་+hཔ་ལ་འཕགས་པ། བཀྱེ«པེ་《སྣར་བསླུ། ༼ནགེ«པེ་»«སྣར་»::+ང་སྟོང་པའི་དྲི་མ་མེད་པ་ཅེས། ཀྱེ«སྣར་+ཆརྒྱུ། བ་དང་། །འདུསsp::ཀོད་ཁྱིམ་དུ་«པེ་»«སྣར་བཞག། @«པེ་»«སྣར་»+སྐདམིགས་པ་ལ། Q«པེ་»«སྣར་བཞག&«པེ་»:::::«སྣར་+པ། །@«པེ་《སྣར་+hསྐད་ཅིག་རེ་རེ་ལ། @«པེ་»«སྣར་»+ནསེམས། @«པེ་»TL«+«:"''°I'++ང་དེ་ཁོ་ན་ཉིད་ལ་མི་ཕྱོགས་པ་རྣམ་པར་གཡེངས་པ་ཅན། «པེ་»«སྣར་»+ལྷུར་ལེན་པའམ། (ཀྱེ«པེ་»ད་པ»«སྣར་བཞགགྱེ«པེ་»《སྣར་གིས། གེ«པེ་»«སྣར་+ཆགཡེང་བ་རྣམ་པར་བཞག་པ་སོགས་ཡོད། །སྟེ«པེ་»«སྣར་+བདག། སྐྱེ«པེ་»«སྣར་+ཇབདག་ཡོད་པ། «པེ་»«སྣར་»ཞེས་ལམ། ༄༅«པེ་»བློ::«སྣར་གེགས། ཁྱེ«པེ་《སྣར་+དབདག་ཏེ«པེ་»སྣར་+ཇབདག་ཡོད། འཚེ«པེ་»«སྣར་+དབདག་འཛིན་ལ་སོགས་པ། ་དགྱེ«པེ་»«སྣར་+ཆབདག་མེད་པ། ཀྱེ«པེ་»«སྣར་+bརྒྱུ་འགྱུར་བའི་་འའི་བདག་འཛིན་ལ་སོགས་པ། འགེ«པེ་»«སྣར་»+རྒསེམས། ཉེ«པེ་»«སྣར་པའི། པདཔར་དུ་(«པེ་»«སྣར་»+དངེས་པ་རེ་ཡོད་ཀྱང་རྟག་ཏུ་མེད་པ། ::«པེ་»«སྣར་སྒོམ། ལ«པེ་»«སྣར་ན། +དསེམས་སྐྱེ་བའི། །«པེ་»«སྣར་»+ངཤེས་བྱ་ཇི་སྙེད་པ། ཚུ«པེ་»«སྣར་»+རྣཁམས་བཅོ་བརྒྱད་པོ། THITHPTQKR»'®«»«KAI®«»«sai®«»«+:ངེ་འཛིན། ཀྱེ«པེ་»«སྣར་»འདུན་པ། འགྱེ«པེ་»«སྣར་»++སྒོམ་པ་ལ། ལུ«པེ་»«སྣར་+kའི་་་་པ་ཤིན་ཏུ་སྦྱངས་པ། །ཁྱེ«པེ་»«སྣར་»+ཆབཞི། ་ཚེ«པེ་»«སྣར་+དསྒོམ་བྱའི། ཚེ«པེ་»«སྣར་»+ཆགཉེན་པོ། །ཚེ«པེ་»«སྣར་»དང༌། །«སྣར་འཚོན། ཏེ«པེ་»སྣར་»+hའམ་རྟོག་པ། །་་འགྱེ«པེ་»«སྣར་»+སྒོམ་པ་ལ་སེམས་མངོན་པར་འདུ་བྱེད་པ། དེ་ལ་ཁེལ་བ་དང་། སྤྱོད་པ། («པེ་»«སྣར་»འགྱུར། ་ཏ་ཝ«པེ་»«སྣར་+ཏིང་ངེ་འཛིན་ལས་སུ་རུང་བ། «པེ་»«སྣར་+%TA«««+aKTVSIKI®R»¥®«»»+ཆདམིགས་པ་དེ་ཉིད་ལ། «པེ་»«སྣར་+hཏིང་ངེ་འཛིན་གྱི་དམིགས་པ་ལས། །«པེ་»སྣར་»འགྲོ། :::@«པེ་»«སྣར་+hཏིང་ངེ་འཛིན་གྱི་དམིགས་པ། ཨེ«པེ་»«སྣར་ཡང༌། གྱེ«པེ་»«སྣར་»+དསེམས། །(«པེ་»《སྣར་+hཡེང་བ་ལ་དགའ་བ། ཉེ«པེ་»«སྣར་+ཇམིའི་སྡུག་བསྔལ་ཙམ་དུ་མ་ཟད། ང་རང་། ང་། «པེ་»«སྣར་+དསྒོམ་པའི་དུས་སུ། :«པེ་»«སྣར་+རྒསྒྲེལ་བ། «པེ་»སྣར་+hའབྲེལ། ན«པེ་《སྣར་+དསྔ་མ། ཚུ«པེ་»«སྣར་ཇརྩོལ་བ་མང་། «པེ་»སྣར་»དོ། ཚེ«པེ་»་པོ«སྣར་»+hའང་གདམས་པ་དང་རྗེས་སུ་བསྟན་པའི་སྐབས་སུ། «པེ་»«སྣར་བསླན། «པེ་»KỸ+ẾAI®«»x+V/U®««»ỚI®«r(ཉེ«པེ་»«སྣར་+hལེའུ་དང་པོར། ཚེ«པེ་»«སྣར་+hཏིང་ངེ་འཛིན་གྱི། སྐྱེ«པེ་»«སྣར་»+hའི་་་ན་རྩོལ་བ། ང:། ་་་་་:བོད་ས་གནས་སུ། ཨེ«པེ་»«སྣར་གྱིས། «པེ་»སྣར་བསླན། @«པེ་»«སྣར་+ངསྒོམ་པའོ། «པེ་»«སྣར་»།
