དུ་དབྱེ་བ་ཁྲག་སུམ་ཅུ་རྩ་དྲུག་རང་རྒྱལ་དུ་གྱུར་ནས་འགོག་པ་ལ་ཞུགས་པའི་སྨོན་ལམ་གྱིས་ཀྱང་མྱ་ངན་ལས་འདའ་་《་བར་གསུངས་སོ། །འབྱེ«པེ་»«སྣར་»+hཏེ་དེ་དག་འགལ། «པེ་»«སྣར་+དསྒྲ་བརྗོད་པ་མེད་ཀྱང་། །:ལྟ་:ཨེ«པེ་»«སྣར་+hཆོས་སྟོན། ན་ཚ:། ཁ། ལ། །ཨ་མེ་ཤ་ར་བའི་ཡུལ། «པེ་»«སྣར་+hངག་ཏུ། «པེ་»«སྣར་མ། མེ«པེ་»སྣར་+hརྣམ་གཡེང་། «པེ་»::ས་། »«སྣར་+རྒཆོས་བསྟན་པས། མི«པེ་»«སྣར་»+hཉན་པ་པོ། །@«པེ་»«སྣར་»—ལ། ༄༅«པེ་»ppམ་《སྣར་»+hའི་གང་ཟག«པེ་»«སྣར་+ཀྱང་དོན་དེ་སྣང་ངོ་ཞེས་པས་སྒྲ་མེད་པར་བརྗོད། «པེ་»ཀྱི«སྣར་»«ཅོ་བྱའི། ་ཚེ«པེ་»«སྣར་»+ཆརྟོགས་པར་བྱས་པ། ཀྱེ«པེ་»«སྣར་+རྒནང་སེམས་ཀྱིས། །་་་ཚེ«པེ་《སྣར་+hཡིན་ཞིང་དེ་རྣམས་ཉིད། :«པེ་《སྣར་+hརྫོགས་པའི་བྱང་ཆུབ་ལས། «པེ་»#་«སྣར་»+རྒཉན་ཐོས་དང་རང་སངས་རྒྱས། ཚེ«པེ་»《སྣར་»+hཏིང་ངེ་འཛིན། ::ཚེ«པེ་+པ། ་་ཉོ་n:ཁྱེ«པེ་»སྣར་+hཡང་ཡིན་པ་དེ། ::བོད་ཀྱི་ས་གནས་::་ན་བྱ་ཚིག་དུས་ངན་«པེ་»«སྣར་»+དརྒྱུ་ལ་འབྲས་བུར་བཏགས་ནས། @«པེ་»«སྣར་»+hཐར་པ་ལ་བརྩོན་པའི་དགེ་བ་་་ཅ་ཏ། ཅན། /Q«པེ་»«སྣར་+མ་ཡིན་མི་སློབ་པ་ཡང་། «པེ་»«སྣར་»+ཆརྣམ་པར་གྲོལ་བ་ཅེས་པ་དེ་ལ། པ་ཁ་@«པེ་»《སྣར་+དེ། །@«སྣར་+པ། ༄༅«པེ་»«སྣར་+དམངོན་པ་མཛོད་དུའང་འགག་པར་འགྱུར་པར་ཡལ་བའི་ལམ་གྱི་ནི། དེ་ཡི་སྒྲིབ་པ་རབ་ཏུ་སྤོང་ཞེས་སོ། །«པེ་»སྣར་+mཉི་ཤུ་པོ། «པེ་»«སྣར་+སྐྱེ་གཕྱིར་མི་ལྡོག་པ། །།ཇི་སྲིད་ས་བརྒྱད་པའི་བར་ཐེག་པ་དམན་པས་དེ་རྣམས་འཇིགས་ཤེས་བཀའ་སྩལ་པ་་་་༽ལྟར། །ཚེ«པེ་»«སྣར་»+ངས་བརྒྱད་པ་ལས། དགྱེ«པེ་»«སྣར་མ་ཡིན་པ་ལ་དེའི་ཕྱིར། གྱེ«པེ་»པའི་དཀྱི《སྣར་+bབརྩོན་འགྲུས་དང་དད་པའི་དབང་པོ་རྟུལ་བས་ན། ཞི་བ་ལ། སྒྱུ་ཞི«པེ་»«སྣར་»+ང་དེ་ཁོ་ན་ཉིད་ལ་འཇུག་པར། དེ་ཉིད་ལ་ནི་ཆེ་རྒྱུ་ཅན། །ཇི་ལྟར་རིང་དུ་སོང་ལ་ཞུགས། །དེ་སྲིད་དེ་ནི་ཕྱིར་མི་ལྡོག།
