བསྟན་འགྱུར། དབུ་མ། ༼ག༽«པེ་《སྣར་»འདིར། «པེ་»སྣར་+hཉི་མའི་འོད་ཟླ་བའི་གཞལ་མེད་ཁང་ལ་ཕབ་པ་ན། ཚེ«པེ་»AII+«aI+«««+AI®«VIད་«པེ་»«སྣར་+hཟླ་བ་དང་སྐར་མ། @«པེ་»སྣར་+hརྒྱལ་ཆེན་རིས་བཞི་པ། ཉི«པེ་»«སྣར་»:ཀྱིནི། (ཉེ«པེ་»«སྣར་+hཀྱི་ཕོ་བྲང་ལྕང་ལོ་ཅན་ནོར་ལྷ་བརྒྱད་ཐ་སྐར་གྱི་བུ་གཉིས་དྲག་པོ་བཅུ་གཅིག་ཉི་མ་བཅུ་གཉིས་ལ་དབང་པོ་ལ་གཅིག་དང་ཉེ་དབང་སུམ་ཅུ་རྩ་གཉིས་ཏེ། སྐྱེ«པེ་»«སྣར་བསྲུང་། +དགི་NH"gia«»»+aBV®«+སྣ་ཚོགས་གཟུགས། སྐྱེ«པེ་»«སྣར་»+དགྲོང་ཁྱེར་དཔག་ཚད་བརྒྱ་ལྔ་བཅུའི་ཚད་ཅན། ་ ེ«པེ་»«སྣར་ན། :《པེ་《སྣར་+ཆསའི་དོས་བཞིར་གཉའ་ཤིང་འཛིན་གྱི་ངོས་ལ་གནས་པར་བཤད། (འགྱེ《སྣར་»ལགས། ཕྱི་ལོའི་:ཀ་རྒྱལ་ས་དང་། 《རང་བྱུང་。«པེ་»«སྣར་»ལྟ། ཉི«པེ་»«སྣར་»དཔངས་སུ། །ན«པེ་《སྣར་»+དངོས་རེ་ལ་དཔག་ཚད་ཉིས་་བརྒྱ་ལྔ་བཅུ་པ། «པེ་སུ་འདུས། «སྣར་སུ་འདུས་རྟོལ། «པེ་»«སྣར་+ནསའི་ངོས་གཞིར་་་་་དཔག་ཚད་ལྔ་བཅུ་ཐུར་དུ་ཟུག་པར་བཤད། «པེ་»«སྣར་»+ངས་ཆུ་སོགས། «པེ་《སྣར་»+སྒརང་གི་འདོད་པའི་ཡོན་ཏན་རྣམས་སྤྲུལ་ནས་དགའ་བར་བྱེད་པས། »«པེ་»«སྣར་»+ཆགཞན་གྱི་སྤྲུལ་པའི་ལོངས་སྤྱོད་ཀྱིས་དགའ་བར་བྱེད་པས། «པེ་»«སྣར་»+hའདོད་པའི། ︽«པེ་»«སྣར་+ཆདང་པོ། «པེ་《སྣར་»གྱིས།
