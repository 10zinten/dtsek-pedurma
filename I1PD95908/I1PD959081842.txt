པ་ལ་དང་ཀྱང་སྔོན་གྱི་འཕེན་པས་སྟོན་པས། ཚེ«པེ་《སྣར་+ཇལམ་ཡང་དག་པ་ལ། ་ཁྱེ«པེ་»«སྣར་»ལྟར་ཏེ། ག་n/pི་(«པེ་»སྣར་+nཐེག་ཆུང་བའི་རིགས་ཅན་རྣམས། ::«པེ་»«སྣར་»+རྒསེམས་ཙམ་པའི་ལུགས་དབང་ད་བཙན་དུ་བྱས་ན། གྱེ«པེ་»«སྣར་»+པ། ་གྱེ«པེ་»«སྣར་»། ་འཉེ«པེ་»«སྣར་»+hའདི་དབུ་་་་::《མའི་ལུགས་ལ་དགོངས་པ་ཅན་མིན་གྱི་ངེས་དོན་ནོ། ཞཉེ«པེ་»«སྣར་»+ཇམདོ་སྡེ་རྒྱན་གྱི་བཤད་པའི་སྐབས་སུ། ཚིགས་སུ་བཅད་པ་གཉིས་པའི་གཟུང་དྲངས་པ། སངས་རྒྱས་ཆོས་ལ་བརྙས་པ་དང་། །ལེ་ལོ་ཅུང་་་《ཟད་ཆོག་འཛིན་དང༌། །ང་རྒྱལ་འདོད་ཆགས་སྤྱོད་པ་དང་། །འགྱོད་དང་མ་ངེས་རྟོག་པ་ནི། །སེམས་ཅན་::ཀ་རྣམས་ཀྱི་སྒྲིབ་པ་སྟེ། །དེ་ཡི་གཉེན་པོར་ཐེག་མཆོག་གསུངས། །།ཤེས་ཏེ། །སྐྱེ«པེ་»«སྣར་+sཤཱཀྱ་ཐུབ་::པ་འདི་རྒྱལ་བ་རྣམ་པར་གཟིགས་པས་དམན་ནོ་སྙམ་པ་ལྟ་བུ། («པེ་»«སྣར་»—པ། ཚེ«པེ་»«སྣར་»+དརྒྱལ་བ་རྣམ་པར་གཟིགས་དང་ཤཱཀྱ་སེང་གེ་གཉིས། བ་«པེ་»«སྣར་+hཐེག་པ་ཆེན་པོ་སོགས། «པེ་»«སྣར་»+ཇརྣམས། «པེ་»«སྣར་»+ཆསྐྱེ་བར་:::་་གྱུར་ཅིག་ཅེས། «པེ་»«སྣར་»+རྐབདེ་བ་ཅན། ལང«པེ་»«སྣར་+དསྨོས་པའམ། «པེ་»ཀླུ་མོཎ་ཌ«སྣར་»+ངལ་དེ་ཙམ་ངས་ཀྱང་ནུས་སོ་སྙམ་པ། ༄༅«པེ་»«སྣར་+ཆབདེ་བ་ཅན་དུ་སྐྱེ་བ་དང་རྫོགས་པའི་བ་བྱང་ཆུབ་ཏུ་ངེས་པར་འགྱུར་རོ། །ཉི«པེ་»«སྣར་»+སྒྲིབ་པའི། «པེ་»«སྣར་»+hཚུལ་ཁྲིམས་ཙམ་་་བ་དམ་པ་གྱིས་ཆོག་པར་འཛིན་པ་སོགས་ཀྱི་གང་ཟག་(ཁྱེ«པེ་»«སྣར་»+hཏེ་སྦྱིན་པ་ཙམ་གྱིས་སམ་ཐོས་པ་སོགས་་་་་པོ་ཙམ་གྱིས། །ཀྱེ«པེ་»«སྣར་+རྣཁམས་གསུམ་པའི་ཕུན་ཚོགས་ལ་ལྷག་པར་ཆགས་པ། «པེ་»«སྣར་»ཆད་དེ་བ་+ལྷ་དང་ཀླུ་སོགས་ཁ་ཅིག་གཟུགས་དང་དབང་ཕྱུག་དང་མཐུ་ལ་སོགས་པས་ཁེངས་པ། «པེ་»«སྣར་པོ། ::ཀ་རྩེ«པེ་»«སྣར་»+རྒནི་ལྷ་ཀླུ་སོགས་ཀྱི་ཕུན་ཚོགས་ཀུན་ལས་ཁྱད་པར་འཕགས་ཤིང་བསམ་གྱིས་མི་ཁྱབ་ཅེས། ལུ«པེ་»«སྣར་»པ། ཚེ«པེ་»«སྣར་+hཡིས་དགེ་བ་ལ་མི་སྤྱོད་འགྱོད་པའི་གཉེན་པོ་ཡང་མི་བསྟེན་པ། །:ཀྱིDཁྱེ«པེ་»«སྣར་འགྲོ་བར། ཨེ«པེ་»«སྣར་+hཅིང་ཉན་ཐོས་ཀྱི་ཐེག་པས་མྱ་ངན་ལས་འདའ་བར་་་་་ཀྱིའདོད། འགེ«པེ་»«སྣར་»+ཆདམ་པའི་ཆོས་པདྨ་དཀར་པོ་སོགས་སུ། ནགེ«པེ་»«སྣར་+hཤཱ་རིའི་བུ་་ཀྱི་ལྟ་བུ། གྱེ«པེ་»«སྣར་+དསེམས་ཙམ་པའི་གྲུབ་མཐའི་དབང་དུ་བྱས་པར། བང་རྒྱུད་པ། །(@«པེ་»«སྣར་»+ཇམདོ་སྡེ་རྒྱན་དུ་ཉན་ཐོས་དང་ནི་ངོ་བོ་ཉིད། དེ་བཞིན་ཉེས་པ་གདུལ་བ་དང་། །རྗོད་ལོབྱེད་ཟབ་པ་ཉིད་ལ་ནི། །ལྡེམ་པོར་དགོངས་པ་རྣམ་པ་བཞི། །ཞེས་གསུངས་པས། «པེ་»«སྣར་»+དབི་སཱ་ར། །«པེ་»+དགཙོ་བོ། «སྣར་»+དཙོ་བ། །ཚ«པེ་»«སྣར་»+hའམ་ལེགས་པར། ང་ཚ«པེ་»སྣར་»+དབསྐལ་པ་གྲངས་མེད་པ་གསུམ་ལ་སོགས་པར། @«པེ་»«སྣར་+hའམ་སྣུམ་པའམ་རླན་པ། ་«པེ་»«སྣར་»+དཆུས་མཉེན་པས་མེ་ཏོག་དང་ནགས་སོགས་སྟོན་པ་ལྟར། ཉི«པེ་»«སྣར་»:ལོ་ཎ+དམ་ཆགས་པ་སོགས། «པེ་»«སྣར་+དམ་སྐྱེད་ན་བསྐྱེད་པ་དང་བསྐྱེད་ན་འཕེལ། ཚེ«པེ་»e«སྣར་+ཤདར་དང་རས་བལ་ལ་སོགས་རེག་པ་ཙམ་གྱིས་བདེ་བ་སྐྱེ་བ་ལྟར། འགྱེ«པེ་《སྣར་»+དམཉན་་ཚིག་པ་ཙམ་ཉིད་ན། །ཚེ«པེ་»«སྣར་+ཆདོན་ནི་ཡིད་ཀྱིས་ཤེས་ཀྱི་རྣ་བས་མ་ཡིན་པས་ན། སྟེ«པེ་»«སྣར་»ཙལོ་+ངས་མཚོན་པའི་ཚིག(«པེ་»«སྣར་+དགོང་མ་དང་བཅས་པ་ཉན་རང་བྱང་ཆུབ་སེམས་དཔའི་ཡིན་་་།
