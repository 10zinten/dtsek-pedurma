བསྟན་འགྱུར། དབུ་མ། ༼ཨི་བས། ཁྱེ«པེ་»«སྣར་»+ཡུལ་མང་བ། ་གྱེ«པེ་»«སྣར་»+དམངལ་དུ། «པེ་»«སྣར་བལྟམ། (ཉེ«པེ་»སྣར་+ཆབཙུན་མོའི་འཁོར། །ཁྱེ«པེ་»སྣར་»+རྒཁབ་ནས། «པེ་»«སྣར་+རྒཆོས་་་་གསུང་བར། ཚེ«པེ་»«སྣར་»+དརྟག་ཏུ་དེ་དག་གི་ནི་སྤྱོད་ཡུལ་ལ། །།སྒྱུ་མར་གྱུར་པའི་རྣམ་པར་སྟོབས་་ཀྱིས་འཇུག་བྱེ«པེ་»«སྣར་+བྱང་ཆུབ་སྤྱོད་པ་ཡོངས་སུ་རྫོགས་བྱེད་ཅིང་། ་ཏེ«པེ་»«སྣར་»:::+ཆདེས་ཤིང་ཕ་རོལ་ཕྱིན་ལ་རབ་བརྩོན་ཏེ། ཚེ«པེ་»«སྣར་»+རྐབཟང་པོ་སྤྱོད་པ་རྣམ་པར་སྦྱང་བ་ཡང་། །མ་འོངས་བསྐལ་པ་ཀུན་ཏུ་སྤྱད་པར་བགྱི། (འགེ«པེ་»སྣར་»ལས་ཀྱི། འགྱེ«པེ་»«སྣར་+ཆབདག་ལ་་་་ཕན་པར་འདོད་པའི་གྲོགས་པོ་དག། བཟང་པོ་སྤྱོད་པ་རབ་ཏུ་སྟོན་པ་དག། དེ་རྣམས་བདག་གིས་ནམ་ཡང་ཡིད་མི་དབྱུང་། །འབྱེ«པེ་»«སྣར་སྒྲུབ། །«པེ་»«སྣར་»+ཆབདེ་བ་དག་ལ་འགྲོ་བ་ཀུན་འགོད་ཅིང་། །འགྲོ་བ་ཐམས་ཅད་ལ་ནི་ཕན་པར་སྤྱད། །ཅེས་དང༌། སེམས་ཅན་རྒྱ་མཚོ་བདག་གིས་རྣམ་པར་འགྲོལ། །«པེ་»«སྣར་»+དབྱང་ཆུབ་གཉིས་ཀྱི། ཚེ«པེ་»སྣར་བཅད། «པེ་»«སྣར་+ཆདེ་ལྟར་མ་ལུས་ཕྱོགས་རྣམས་ཐམས་ཅད་དུ། །རྒྱལ་བ་་་་་རྣམས་ཀྱི་ཞིང་གི་བཀོད་ལ་འཇུགཨུ«པེ་»«སྣར་+རྣམྦྷེལ་ཞིར་བསྒྱུར་བའང་བྱུང༌། «པེ་»«སྣར་»+hསྤྱིའུ་ཚུགས་སུའང་བསྒྱུར། ༄༅«པེ་»«སྣར་+ལ། «པེ་»«སྣར་»—སྟེ། «པེ་»«སྣར་»དུ། ཁ་བ་རི་བ་(གྱེ«པེ་»«སྣར་+ཆརྡུལ་གཅིག་སྟེང་ན་རྡུལ་སྙེད་ཞིང་རྣམས་ཏེ། ཀྱེ«པེ་»«སྣར་+བ། «པེ་»«སྣར་+དཞིང་རྣམས་རྒྱ་མཚོ་རྣམ་པར་དག་བྱེད་ཅིང་། «པེ་»«སྣར་+དཞིང་དེ། ཀསྐྱེ«པེ་»«སྣར་»ཀྱིས། ཨེ«པེ་»«སྣར་+ཆགང་ཡང་དུས་གསུམ་གཤེགས་པ་རྒྱལ་བ་ཡི། །བྱང་ཆུབ་སྤྱོད་པའི་སྨོན་ལམ་བྱེ་བྲག་རྣམས། །བཟང་པོ་སྤྱོད་པས་བྱང་ཆུབ་སངས་རྒྱས་ནས། །དེ་ཀུན་མ་ལུས་བདག་གིས་རྫོགས་་་་་པར་བགྱི། །ཚེ«པེ་»«སྣར་+ཇལུས་དང་ངག་དང་ཡིད་ཀྱང་རྣམ་དག་ཅིང་། །སྤྱོད་པ་རྣམ་དག་ཞིང་་་་་རྣམས་ཡོངས་དག་དང་། །བསྔོ་བ་བཟང་པོ་མཁས་པ་ཅི་འདྲ་བ། ་། དེང་འདྲར་བདག་ཀྱང་དེ་དང་མཚུངས་པར་ཤོག(«སྣར་+hཀུན་ཏུ་དགེ་བའི་བསོད་ནམས་སྟོབས་དག་དང་། །ལེ«པེ་»«སྣར་+ཆསངས་རྒྱས་སྲས་ཀྱི་བསྐོར་བའི་མགོན་པོ་རྣམས། །མངོན་སུམ་རྟག་ཏུ་བདག་གི་རྒྱལ་བ་བལྟ། །ང་ཚེ«པེ་»«སྣར་+དསྤྱོད་པ་ཚད་མེད་པ་ལ་གནས་ནས་ཀྱང་། །དེ་དག་འཕྲུལ་པ་ཐམས་ཅད་འཚལ་བར་བགྱི། །«པེ་»«སྣར་+hའགྲོ་བ་ཀུན་ཏུ་སྐྱེ་བ་དྲར་པར་གྱུར། །:«པེ་»«སྣར་+hཀུན་ཏུ་མྱུར་བའི་རྫུ་འཕྲུལ་སྟོབས་རྣམས་དང༌། །ཆེ«པེ་《སྣར་»+ཆདུས་གསུམ་གཤེགས་པ་མི་ཡི་སེང་གེ་གང་། །དེ་དག་སྐད་ཅིག་་གཅིག་ལ་བདག་གིས་བལྟ། ཚུ«པེ་»«སྣར་+རྒཚེ་རབས་ཀུན་ཏུ་འཆི་འཕོ་སྐྱེ་བ་ན། །རྟག་ཏུ་བདག་གི་་རབ་ཏུ་བྱུང་བར་ཤོག། ཅེས་དང༌། སྲིད་པ་ཐམས་ཅད་དུ་ཡང་འཁོར་བ་ན། །བསོད་ནམས་ཡེ་ཤེས་དག་ནི་མི་ཟད་རྙེད། «པེ་»«སྣར་»+hཀུན་ནས་སྒོ་ཡི་ཐེག་པའི་སྟོབས་རྣམས་དང༌། ཚ«པེ་»«སྣར་+དབཟང་པོ་སྤྱོད་པའི་སྟོབས་ནི་རྫོགས་པར་བགྱི། །ཞེས་དང༌། བསྐལ་པ་རྒྱ་མཚོར་མི་སྐྱོ་སྤྱད་པར་གྱི། །མཚོཞེས་དང་། གང་ཡང་བསྐལ་པ་དུས་གསུམ་ཚད་དེ་དག་ཌ། སྐད་ཅིག་ཆ་ཤས་ཀྱིས་ནི་ཞུགས་པར་སྤྱད། །ཁོ་«པེ་»«སྣར་»+དཀུན་ཏུ་ཡོན་ཏན་སྤྱོད་པའི་སྟོབས་རྣམས་དང་། །«པེ་»«སྣར་»+ཇལུས་དང་ངག་་།
