བསྟན་འགྱུར། དབུ་མ། @འི་:ཀ་ཁ་གེ«པེ་»«སྣར་+ཇལྟ་བ་གཉིས་པོ། གྱེ«པེ་《སྣརེ་»པ། །ཕྱི་ལོ་::ཀ་དང་དཔེ་ཀྱེ«པེ་»«སྣར་»+དམཛོད་ཀྱི་གནས་ལྔ་པར། གོང་མ་ཐམས་ཅད་ལུང་མ་བསྟན། །འདོད་ན་འཇིག་་ཚོགས་ལྟ་བ་དང༌། །མཐར་འཛིན་ལྷན་ཅིག་མ་རིག་པའང་ཞེས་པས། «པེ་»«སྣར་»+ཆབསྒྲིབས་ལ། ལི«པེ་《སྣར་+hཉོན་མོངས་ཅན་ལྡང་མ། «པེ་»«སྣར་»+hཐེ་ཚོམ་ནི། ::«སྣར་བརྟེན། །«ཅོ་བཏན། བོ@«པེ་»སྣར་»+དང་རྒྱལ་ནི་ཁེངས་པས། །p/«པེ་»«སྣར་+དམཛོད་བཞི་པ། །O«པེ་»«སྣར་»+hཐར་པ་དམ་པའི་དོན་དུ་དགེ་རྩ་བ། །ངོ་ཚ་ཁྲེལ་ཡོད་བདག་ཉིད་ཀྱིས། །དེ་དག་དང་ལྡན་མཚུངས་ལྡན་གྱི། །བྱ་བ་ལ་སོགས་ཀུན་སློང་བས་ཤེས་སོ། །«པེ་»«སྣར་»+ཆགཞན་དང་མཚུངས་པར་ལྡན་པ་དང་ཀུན་ནས་སློང་བ་ལ་མི་ལྟོས་པས། 《(«པེ་»སྣར་+ཆདགེ་རྩ་གསུམ་དང་ངོ་ཚ་དང་ཁྲེལ་་ཡོད་པ། ཁྱེ«པེ་»«སྣར་+ང་དེ་དག་དང་མཚུངས་པ་མི་ལྡན་པ་དགེ་བ་མིན་པས། །།ཉི«པེ་»«སྣར་»+ཆདགེ་བའི་རྩ་བ་སོགས་དང་མཚུངས་པར་ལྡན་པའི་ཆོས། «པེ་»«སྣར་ལུས། ཁྱེ«པེ་»«སྣར་+འདསྙོམ་འཇུག་གཉིས། «པེ་《སྣར་+hའོ་མ་ཡང་དག་པར་འདུས་པ་བཞིན་ཟེར། །ཁྱེ«པེ་»«སྣར་»བོད་ཀྱི་དེ+དསྔ་མ་ཕྱི་མས་བཤད། ཁྱི«པེ་»«སྣར་+དརྣམས་གྲངས་ཀྱིས་བཤད་པ། ལེ«པེ་»«སྣར་»+ཆདོན་་ལ་གནས་པ་བཞིན། །དགྱེ«པེ་»«སྣར་»+hཉིད། དགྱེ«པེ་»«སྣར་»+རྒཁེངས་པའམ། «པེ་»«སྣར་+kའམ་རློམ་པ། ནཔཁོ:ཀན། ཡཾ:། ཟླ་་ཀྱི་འགྲོས་དག། «སྣར་»ཡས། ཉི«པེ་»«སྣར་»+དང་རྒྱལ། *@«པེ་»«སྣར་»ལས། «པེ་《སྣར་»+hའི་ཕ་་་རོལ་པོ། «པེ་«སྣར་+དར་རྟོག་པ། ཁྱི«པེ་»«སྣར་+སྒར་རྟོག་པ། ག«པེ་+དརང་དང་་་་པ་ཐོས་པ་ལ་སོགས་ཀྱི། «སྣར་»+སྒརང་དང་ཐོས་པ་ལ་སོགས་ཀྱིས། །PO«པེ་»«སྣར་+hརང་ཉིད། །«པེ་»«སྣར་+དཆོས་གང་བདག་ཉིད་ཁྱད་འཕགས་དང་། །མཉམ་པར་རློམ་པ་གང་ཡིན་ཏེ། །ལྷག་པའི་ང་རྒྱལ། ་ཁྱེ«པེ་»«སྣར་»+ཁྱད་འཕགས་པས། ཁྱད་པར་འཕགས་པ་རློམ་པ་དང་། །ཤིན་ཏུ་མཐོ་་་བས་མཐོ་སྙམ་པ། །ང་རྒྱལ་ལས་ཀྱང་ང་རྒྱལ་ཏེ། །ཞེས་པ་རང་བས། ཁྱེ«པེ་»«སྣར་+hཉེ་བར་ལེན་པ་ཞེས་བྱ་བ་ལྔ་པོ་སྟོང་པོ་དེ་རྣམས་ལ་རྨོངས་པས་ང་སྙམ་འཛིན་པ་གང་། དེ་ནི་ངའོ་སྙམ་པར་བཤད་ཅེས་་་་པས། ཚེ«པེ་»«སྣར་+hཟག་བཅས་ཀྱི་ཕུང་པོ། «པེ་《སྣར་ལ། ཁྱེ«པེ་»«སྣར་+སམ། །ལུ«པེ་»«སྣར་»+ང་ཕུང་པོ་ཟག་བཅས་སམ་ཉོན་མོངས་པ། ?ཁྱི«པེ་»«སྣར་»+ཆབདེན་པ། («པེ་»«སྣར་»+ཕུང་པོ། ཉེ«པེ་《སྣར་+ང་འབྲས་བུ་མ་ཐོབ་ཐོབ་སྙམ་པ། གང་ཡིན་མངོན་པའི་ང་རྒྱལ་སྟེ་ཞེས་པས། འཚེ«པེ་»«སྣར་»+ཇའམ་ཁེངས་པ། གྱི«པེ་»«སྣར་»+ང་སྡིག་ལས་བྱེད་ལས་བསྟོད་པ་ནི་་་མཁས་པས་ལོག་པའི་ང་རྒྱལ་རྟོགས། ་«པེ་»«སྣར་»+ཇམི་འདོད་པའི་འབྲས་བུ་ཅན། ག་ཡ་འགེ«པེ་»«སྣར་+ཆགཟེངས། འཉེ«པེ་《སྣར་+རྣཁེངས་པ། སྐྱེ«པེ་»«སྣར་+ཆབདག་དགོས་མེད་པ་ཉིད་དོ་ཞེས། བདག་ལ་དམོད་པ་གང་ཡིན་ཏེ། དམན་པའི་ང་རྒྱལ་ཞེས་བྱ་སྟེ། དེ་རྣམས་མདོར་བདུན་བཤད་ཡིན་ཅེས་སོ། །ལནཕྱིར་ཞིང་ན་བྱུང་། ཤོག་དང་། བཙོན་ཁང་དང་བར་བྱེད་«པེ་»«སྣར་བསྒྲུབ། ཉི«པེ་»«སྣར་»+དཆོས་ཀྱི་ངོ་བོ་ཉིད་ཀྱི་ཡུལ་ཅན་ཤེས་རབ་ཟག་མེད་ཀྱི། །།
