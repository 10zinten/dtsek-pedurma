བསྟན་འགྱུར། དབུ་མ། ཀྱི་::ཀྱི(བྱེ«པེ་»«སྣར་»+དཁ་དོག་དྲི་རོ་ཕུན་སུམ་ཚོགས་ཤིང་གཟེགས་མ་དང་ཕུབ་མ་མེད་པ་སོར་བཞི་ཙམ་ལ་་་THTHESTsvarII®««««»«KXIAI®««+®««»KETN*®»:། སྣར་»ས། ནེ«པེ་»«སྣར་»+ཆགཟེག་མ་དང་ཕུབ་མ་བྱུང་ཞིང༌། ན«པེ་»«སྣར་སྔར། ད་«པེ་《སྣར་ཤེས། །འ«པེ་»«སྣར་ལ། («པེ་《སྣར་+hཆགས་པས་སེམས་ཅན་རྣམས་ལ་མ་བགོས་པར་རྐུན་པོ་མང་དུ་གྱུར་ཏེ་དེ་རྣམས་ལ་མཚོན་གྱིས་བཏབ་པས། «པེ་»«སྣར་+ན། «པེ་»ལ«སྣར་»+hལ་བར་གྱི་བསྐལ་པ་བཅུ་དགུ་འདས། ::@«པེ་»«སྣར་»+དམི་དགེ་བ་བཅུ། «པེ་»ལམི《སྣར་ནོན། ལོ«པེ་»«སྣར་»+ཆསའི་དངོས་བཞིར་ཚེ་ལོ་སུམ་ཅུ་པ་ལ་མུ་གེ་ཉི་ཤུ་པ་ལ་ནད་བཅུ་པ་ལ་་་ལ། མཚོན་ཅེས། «པེ་»«སྣར་+ཆགང་ལག་ཏུ་བླངས་པ། ཁྱེ«པེ་»«སྣར་»གྱི་སྟེ། +རྐབསྐལ་པ། «པེ་»«སྣར་»+སྡིག་པ་ལ་ཆགས་པ་སོགས། «པེ་»«སྣར་+ཆདེ། «པེ་»«སྣར་»ཉམས། ཀྱི། སྐྱེ«པེ་»《སྣར་བ་ལས། «པེ་»《སྣར་»+དཔའམ་འདུས། ཁྱེ«པེ་»«སྣར་ཕྱུང་། «པེ་»དང་། ཁ་«སྣར་»+དམཛོད་འགྲེལ་དུ་དྲངས་པ་གང་གིས་ཉི་མ་འགའ་སྲོག་གཅོད་སྤངས་པ་དང༌། དགེ་འདུན་ལ་ཨ་རུ་པོར་འགའ་དང་བསོད་སྙོམས་འགའ་གུས་པར་བྱས་ཏེ་ཕུལ་བས་མཚོན་དང་། ནད་དང་། མུ་གེའི་བསྐལ་པ་་ལ་མི་སྐྱེའོ་ཞེས་བརྗོད་དོ་ཞེས་སོ། །རེ«པེ་»«སྣར་»+hཤར་ནུབ་ཀྱི། །ཆུ་བྱི་ཅ་མིང''#་«པེ་»«སྣར་+དམཚོན་གྱི་ལ། །(«པེ་»སྣར་»+དནད་ཀྱི་ལ། ::«པེ་»«སྣར་»ཉམས། །«པེ་»«སྣར་+དམུ་གེའི་ལ། ་:«པེ་»«སྣར་+hཚེ་ལོ་དཔག་མེད་འགྲིབ་པ་ནས་ཚེ་ལོ་བཅུ་ཡི་བར། །འོན༌པོ«པེ་»«སྣར་+hཟིན། ་«པེ་»«སྣར་»+དམཛོད་དུ། བར་གྱི་བསྐལ་པ་དཔག་མེད་ནས། །ཚེ་ལོ་བཅུ་ཡི་བར་དུའོ། །ཞེས་སོ། །«སྣར་པོ། «སྣར་»ཁྲི་རེ། ་དཔེ«པེ་»«སྣར་»+དབར་བསྐལ། ན། :(«པེ་»«སྣར་»+ཚེ། ་(«པེ་»«སྣར་»ཆགས། ://ཉེ«ཅོ་+དུ། «སྣར་»ན། 《སྣར་»སྐལ་བ། མི«པེ་»«སྣར་»ཀྱིས། ཁྱེ«པེ་»«སྣར་»+དསྙིང་རྗེ་པདྨ་དཀར་པོའི་མདོར། བསྐལ་པ་འདི་འཆགས་་་་:པའི་དུས་སུ་ཁམས་སྣ་ཚོགས་ཅན་གྱི་སྤྲིན་ལས་བྱུང་བའི་ཆུའི་ནང་དུ་གསེར་གྱི་པདྨ་འདབ་མ་སྟོང་དང་ལྡན་པ་་་སྟོང་སྣང་ནས་གཙང་མའི་ལྷ་རྣམས་ཀྱིས་ཅིའི་ལྟས་ཡིན་བརྟགས་པས་བསྐལ་པ་འདི་ལ་སངས་རྒྱས་སྟོང་འབྱུང་་་་ལ་བའི་མཚན་མར་ཤེས་ཏེ། བསྐལ་པ་འདི་ནི་བཟང་པོའོ་ཞེས་བསྒྲགས་སོ་ཞེས་ལེགས་པར་གསུངས་པས། ::དཀོན་མཆོག་སྤྲིན་དུ། དེ་ནི་བསྐལ་པ་གྲངས་མེད་པ་གསུམ་གྱིས་ཡང་དག་པར་སྒྲུབ་པར་མི་ནུས་ཀྱི་ནམ་བྱང་ཆུབ་སེམས་དཔའ་ཆོས་ཐམས་ཅད་མཉམ་པ་ཉིད་དུ་ཞུགས་པར་གྱུར་པ་དེའི་ཚེ་ནས་བརྩམས་ཏེ་བསྐལ་པ་བགྲང་བར་བྱའོ་ཞེས་ཀྱང་བཀའ་སྩལ། ཐེག་བསྡུས་སུ་སའི་ཐད་དུ། བཟང་དང་སྨོན་པའི་སྟོབས་ཅན་དང༌། །སེམས་བརྟན་ཁྱད་པར་འགྲོ་བ་ཡིས། །བྱང་ཆུབ་སེམས་དཔའ་གྲངས་མེད་གསུམ། །།ཀུན་ཏུ་རྩོམ་པ་མཛད་་་་ཅེས་བརྗོད། །ཅེས་སོ། །དང་《ལེན་པ། ས་མ་འབད་རོལ་(ི«པེ་»《སྣར་ནི། «པེ་»«སྣར་+ཆདངོས་པོའི་གནས་ལུགས། དོ«པེ་»«སྣར་པ་སྤྱོད། །::«སྣར་»སྐལ་བ། («པེ་»«སྣར་+པ། །@«པེ་»«སྣར་»+དམི། <«པེ་»《སྣར་»+hཡུལ། །
