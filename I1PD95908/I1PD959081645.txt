བསྟན་འགྱུར། དབུ་མ། ༼ཨ༽པ་སོགས། ཨེ«པེ་»«སྣར་+ར་སྡོམ་བརྒྱད་འགྲེལ་པར། «པེ་»«སྣར་»+kཉ་དང་གནམ་སྟོང་སོགས། «པེ་»«སྣར་རིམས་ཀྱིས། «པེ་»«སྣར་»+དཞེས་སོ། །ལེ«པེ་»«སྣར་+ཆདང་སྒྲ་མི་སྙན་པ་དང་ངན་སོང་། ཁྱེ«སྣར་པ། ཁྱེ«པེ་»«སྣར་»+རྣམི་སྐྱེ་བ། «པེ་»«སྣར་+དབུད་མེད། ད་བར་ཚེ«པེ་»«སྣར་+hཏེ་དངོས་པོར་ན། མཛོད་འགྲེལ་དུ་འོ་ན་མདོ་ལས་དགེ་སློང་རྣམས་སྒོ་ང་ལས་སྐྱེས་པའི་ཀླུ་ནི་ཟླ་བ་ཕྱེད་ཀྱི་ཚེས་བརྒྱད་ལ་གནས་ནས་བྱུང་ནས་ཡན་ལག་བརྒྱད་དང་ལྡན་པའི་བསྙེན་གནས་ལ་ཉེ་བར་་་་::གནས་སོ་ཞེས་གསུངས་པ་ཇི་ལྟ་བུ། དེ་ནི་དེ་དག་གི་ལེགས་པར་སྤྱད་པ་ཙམ་ཡིན་གྱི་སྡོམ་པ་ནི་མ་ཡིན་ནོ་ཞེས་གསུངས་སོ། །གྱེ«པེ་»«སྣར་»+ང་སྟེ་ཐ་དད་དུ་བཞག་པ་ནི། །ཀྱེ«པེ་»«སྣར་»+ང་སྡོམ་པ། ཉེ«པེ་»:::སྣར་+ཆདད་པ་དང་ལྡན་ཞིང་འཚམས་མེད་པ་ལ་སོགས་སྤངས་པས་ཤེས་སྡོམ་བརྒྱད་འགྲེལ་པར། འཉེ«པེ་»ལ། «སྣར་»+དསློབ་དཔོན་པུཙྪཱ་ཤྲཱི་ཡིས་དགེ་བསྙེན་སྡོམ་བརྒྱད་ཀྱི་འགྲེལ་པར་སྡོམ་གནས་ཞེས་པ་དགེ་བསྙེན་གྱི་་་སྡོམ་པ་ལ་སློབ་པ་རྣམས་ཡིན་གྱི་གཞན་ལས་མིན་ཏེ་མི་དབང་བའི་ཕྱིར་ཞེས་བཤད། །ཀ་ར་བསྐྱེ«པེ་»སྣར་»+དབླངས་པ་ལས། «པེ་»«སྣར་+རྣབླ་མ་ལ་ཕྱག་བྱས་ནས། /:«པེ་»«སྣར་+དམིང་ཆེན་གྱིས་་་་ཏེ། ཞུས་པའི་མདོར་གསུམ་ལ་སྐྱབས་སུ་སོང་སྟེ་དགེ་བསྙེན་དུ་གཟུང་དུ་གསོལ་ཞེས་སྨྲ་ན་དེ་ཙམ་གྱིས་དགེ་བསྙེན་དུ་འགྱུར་རོ་ཞེས་བཀའ་སྩལ་པས། ཁྱེ«པེ་»«སྣར་གཟུང་། །བློ་བོད་ཀྱི་<<ད་བྱེད་པ། @«པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་པས། ཚེ«པེ་»«སྣར་+དསུམ་བརྒྱ་པར། འབྲས་་བབབུ་ལ་སོགས་ཁུ་བ་བཅོས་པ་ནི། །བཅོས་ཆང་དགུན་ཆང་བུར་ཆང་འདྲར་བསྟན། །་ཆན་བཏགས་མནན་པ་་་བ་བྱས་པ་འབྲས་བུའི་ཆང་ཞེས་པ། »«པེ་»«སྣར་+རྒན་རི་ཀ་དང་མེ་ཏོག་དང་། ::«པེ་»«སྣར་»+དངམ་བཅོས་པ། །«པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་+ཆགང་ཟགཚེ«པེ་»སྣར་»+ང་གསུམ་ལ་སྐྱབས་སུ་འགྲོ་བའམ། ་འཁྱེ«པེ་»«སྣར་+ཏསྲོག་གཅོད་པ་སོགས་ལྔ་སྤོང་བའི། ཞེ«པེ་»«སྣར་+དསྐྱབས་ཀྱི། ཨེ«པེ་»«སྣར་+དསྐྱབས་སུ། ཨེ«པེ་»«སྣར་»+ཆསངས་རྒྱས་ལས། ཡེ་«པེ་»«སྣར་+ཡབདག་གི་སྟོན་པའོ་ཞེས། ་ཁྱེ«པེ་»«སྣར་»+དབཞི་པོ། ཁྱེ«པེ་《སྣར་+ཆགང་ཞིག་སངས་རྒྱས་སྐྱབས་འགྲོ་བ། དེ་ནི་ཡང་དག་དགེ་བསྙེན་ཏེ། །ནམ་དུ་ཡང་ནི་ལྷ་གཞན་ལ། །སྐྱབས་སུ་བབདེ་སོང་བ་མ་ཡིན་ནོ། །དམ་པའི་ཆོས་ལ་སྐྱབས་སོང་ན། །འཚེ་ཞིང་གསོད་པའི་སེམས་དང་བྲལ། །དགེ་་་་་འདུན་རྣམས་ལ་སྐྱབས་སོང་བ། །།མུ་སྟེགས་ཅན་ལ་ཕྱོགས་མི་བྱ། །ཞེས་པས་འཕགས་པ་མྱང་འདས་ཆེན་་་་པོར་གསུངས་སོ། །༼ཨེ«པེ་»«སྣར་+དབླང་བའི་ཡུལ་ནི་གནས་འཇོག་གིས་ཞུས་པའི་མདོར། དགེ་སློང་་་བྲམ་ཟེ་ཤེས་པ་གཅིག་གི་དྲུང་དུ་སོང་སྟེ་ཞེས་གསུངས་པ་དང་འདུལ་བ་ལུང་དུ། ཁྱིམ་བདག་མགོན་མེད་་་་་ཟས་སྦྱིན་གྱིས་འགྲོན་པོ་ལ་བསྙེན་གནས་ཕོག་པར་བཤད་པས་དགེ་བསྙེན་ཆོ་ག་ཤེས་པས་ཀྱང་རུང་བར་གསལ། ཚེ«པེ་»«སྣར་+དཔར་འདོགས་པ། ::འགྱེ«པེ་»«སྣར་+དམཛོད་དུ། དམའ་བར་འདུག་སྨྲས་རྗེས་་་་་པ། བཟླས་པས། མི་བརྒྱན་ནམ་ནི་ལངས་བར་དུ། བསྙེན་གནས་ཡན་ལག་ཚང་བར་ནི། །ནང་པར་གཞན་་་ལས་ནོད་པར་བྱ་ཞེས། །པ་ལྟར། ་«པེ་»སྣར་གཟུང་། 《མ་ན་ཁ་པ། ཨ་ན་ལ་བ་སུ་«པེ་»«སྣར་འཁྲུ་གང་ལས་ལྷག་པའི། <«པེ་»«སྣར་+ཆདམའ་ཡང་རིན་པོ་ཆེ་ལས་བྱས་པ། ་་།
