ཚུན་རྣམ་པར་འཁྱམ་བྱེད་པ། །གང་ཕྱིར་ལུས་དང་སྲོག་ལ་ཡང་། །འདོད་་་་མེད་སྙིང་བརྩེ་བདག་ཉིད་རྣམས། །དཀའ་བ་བརྒྱ་ཕྲག་དུ་མ་དག། བསྐལ་པ་དུ་མར་ཉམས་སུ་ལེན། །དེ་ནི་མཆོག་ཏུ་གསང་བ་སྟེ། །དེ་ནི་གོ་འཕང་མཆོག་ཡིན་ནོ། །ཕྱོགས་འདི་ངག་ཡུལ་ཐམས་ཅད་ལས། །འདས་ཡིན་དེ་ནི་འཇིགས་པ་མེད། །འདི་ལ་ཡང་དག་མོས་པ་ཡང་། །ངན་འགྲོ་དག་པར་མི་འགྲོ་ན། །སུ་ཞིག་འདི་ཕྱིར་བསྒྲིམ་བྱས་ནས། །མངོན་པར་བརྩོན་པ་སྨོས་ཅི་དགོས། །ཤེས་རབ་འཆལ་དང་བསོད་ནམས་ཆུང་། །དད་པ་ཙམ་ཡང་འདིར་མི་སྐྱེ། །གང་ཞིག་དགེ་བ་བསྩགས་ྋགྱུར་པ། །བློ་ལྡན་དེ་ནི་འདི་ལ་སྦྱོར། །སྐུར་་་འཇུག་སྒོ་ཡི་བསྟན་བཅོས་ནི། །བདག་གིས་ཅི་ནུས་བྱས་པ་ཡི། །དགེ་བ་་གང་ཡིན་དེ་ཡིས་ནི། །འགྲོ་བས་སྐུ་གསུམ་ཐོབ་པར་ཤོག། སྐུ་གསུམ་ལ་འཇུག་པའི་སྒོ་ཞེས་བྱ་བའི་བསྟན་བཅོས་སློབ་དཔོན་ཀླུའི་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཝརྨ་དང་། ་ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བན་དེ་ཡེ་ཤེས་་སྡེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།་ཆཎ་བྱིན་ད་ལན་ཊི་ཁོ་pཚོ། །ཏིང་། ཁོང་བ་ལ་བཎཏོང་བ། ?《། ལཚོ་ཁག་འོང་ཀ་ཀྱི་ཚབ་བྱས་པར་ཁྱོད་ཀཏོ་་ཁ་::ཀ་བ་བརྒྱན་»anཊཎ་ཀྱི་ལོ་ཙཱ་ཁག་ང་བློའི»སྐྱེ། །ཁ་:《་མི་ཚུ་::ཀྱིངབ། དེ་ལ་ཀཀྐོ་ལའི་དྲ་བ་བ་ག་ཚོ་མང་ཁཙོ་༄༅། །ཞི་ལྷན་དང་། 《ཀྱི། འི<<། །ཀ་ཙ་ཏི་ཙྪ་ཀོ་ཊི་། ཁབ་ཀྱི་རྒྱུཚེ། བ་་་་དང་་་། བོ་::དུ་བྱེད་པའི་ཁ་་ལོ། །p?ཏོར་བཅུ་གདུ—།
