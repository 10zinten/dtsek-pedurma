རརབསྟན་འགྱུར། དབུ་མ། ༼ག༽དུ། །བརྟག་པ་མ་བྱས་ཉམས་དགའ་བ། །བདག་ཏུ་སྔ་མ་སྔ་མ་ལ། །བརྟེན་ནས་ཕྱི་མ་ཕྱི་མ་ཡི། །འབྲས་བུ་དེ་འདྲ་འབྱུང་བ་ཡིན། །དེ་ཕྱིར་ཀུན་རྫོབ་རྒྱུ་མེད་ན། །རུང་མིན་ཞེས་པ་ལེགས་མ་ཡིན། །གལ་་ཏེ་འདི་ཡི་ཉེར་ལེན་ཡང་། །ཡང་དག་ཡིན་ན་དེ་སྨྲོས་ཤིག་°། ཅེས་གསུངས་སོ། །༼སྐྱེ«པེ་»«སྣར་»+ཆབདེན་པའི། །«པེ་»«སྣར་»+hརྒྱུ་བདེན་པ། གེ«པེ་»«སྣར་+hའབྲས་བུ་བརྫུན་པ། འབེ«སྣར་»དེ། མུ«པེ་»«སྣར་+དའམ་གྲུབ་པ་གཟུགས་ལ་སོགས་པ། ་གྱི«པེ་»«སྣར་»+hཐ་སྙད་པས་ཤེས་པས། བབའཁོ«པེ་»«སྣར་བཞག«པེ་»«སྣར་པ། དབྱེ«པེ་»སྣར་»བདེན་པའི། དབང་དང་། «པེ་»«སྣར་+གཞན། «པེ་»«སྣར་བསླུ། «པེ་»སྣར་+hསྤྲོས་བྲལ། «པེ་»«སྣར་»+ང་དོན་དམ། «པེ་»«སྣར་»+ཆབདེན་གཉིས་སུ། སྐྱེ་སོགས་བཀག་པ་དོན་དམ་དང་། །མཐུན་པའི་ཕྱིར་ཡང་འདོད་པ་ཡིན། ཅེས་དང་རྒྱན་དུ། དམ་པའི་དོན་དང་མཐུན་པའི་ཕྱིར། །འདི་ནི་་དམ་པའི་དོན་ཅེས་བྱ། ཁྱེ«པེ་»«སྣར་»+hཤེས་བརྗོད། «པེ་»«སྣར་»+ཆསྐྱེ་མེད་དང་ངེས་པ། ཞ་ཀྱི་ཨེ«པེ་»«སྣར་+ཆདོན་དམ་འདུས་མ་བྱས་ནི་སྐྱེ་བ་དང་སྐྱེ་མེད། ཨེ«པེ་»«སྣར་+དོན་དམ། ཁྱེ«པེ་»«སྣར་+ཆདབུ་མ་རྩ་ཤེའི་རབ་བྱེད་བཅོ་བརྒྱད་པར། ཨེ«པེ་»«སྣར་+གགྱིས་བསྟན་པ། ཚེ«པེ་»«སྣར་»རྟོགཡི་ཁྱེ«པེ་»«སྣར་+དརང་བཞིན་དང་བྲལ་བ། ཉེ«པེ་»«སྣར་+hངག་ཆེ«པེ་»«སྣར་+ངཤིང་བརྗོད་དུ་མེད། དགྱེ«པེ་»«སྣར་+ཏེ་ཐད། །ལ་འགྱེ«པེ་»«སྣར་+hཔ་སྟོང་པ་ཉིད་དུ་རོ་མཉམ་པ། །ལོ་འགེ«པེ་»སྣར་+དཀྱི། ་་་ཞཉེ«པེ་»«སྣར་»+ཛདོན་དམ་གཅིག་དང་ཐ་སྙད་ཀྱི་སྤྲོས་པ་ལས་འདས་པ། དགྱེ«པེ་»«སྣར་»+hཆོས་ཅན་ལ་ཆོས་དང་ཆོས་ལ་ཆོས་ཉིད་དུ་བརྗོད་པ་མི་འཐད་དོ་ཞེ་ན་དེ་ལྟར་བར་བྱས་པ་སྟེ། ཉེ«པེ་»«སྣར་བཞགེ«པེ་»«སྣར་»བསྒྲུབ། འགུ«པེ་《སྣར་+jསྒྲུབ་པ་དང་དགག་པ། །ན«པེ་»«སྣར་»+དསྒྲ་རྣམས་ལས་དེ་་་་རྟོགས་ཉིད་ཀྱིས། གཞན་བཟློག་རྟོགས་འགྱུར། ཞེས་རྣམ་འགྲེལ་དུ་གསུངས་པ་ལྟར། འགེ«པེ་»«སྣར་»མ། འགུ«པེ་»«སྣར་+དརྟོགས་པ། འགྱེ«སྣར་ནི། །«པེ་»«སྣར་+དབྱ་བ་རྫས་ཡོན་ཏན། «པེ་»«སྣར་+དརྣམ་འགྲེལ་དུ། རང་བཞིན་དེ་འཛིན་་་་མཆུ་རྣམ་རྟོག་བློ། །དོན་མེད་ཀྱང་དེའི་དོན་ཅན་འདྲ། །ཞེས་པ་ལྟར། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»«ཅོ་»མེད། ་ཨུ«པེ་»«སྣར་+ཆདོན་ཡུལ་དུ་བྱོས་སྙམ་པའི། :«པེ་»«སྣར་+ཆགཞན་སེལ་བ། ༄༅«པེ་»«སྣར་»—དུ། «པེ་»«སྣར་»+ཆརྣམ་འགྲེལ་ངེས་གཉིས་སུ། དེ་ཕྱིར་ཁྱད་པར་གང་་། ཞིག་ཆོས། །གང་གིས་གོ་བར་འགྱུར་བ་དེ། །དེ་ལས་གཞན་གྱིས་ནུས་མ་ཡིན། །དེས་ན་ཐ་དད་རྣམ་པར་་@༌ཚོ་གནས། །ཞེས་པ་ལྟར་དོན་དེའི། «པེ་《སྣར་»+hརྣམ་པ། ཚེ«པེ་»«སྣར་+དསྒྲ་བྱུང་བའི་ཡུལ། བལ ེ«པེ་»«སྣར་»+དསྒྲ་བྱུང་གིས་གང་རྟོགས་པའི། «པེ་»«སྣར་»+ཆརྣམ་རྟོགས་ཀྱིས་ངེས་པ། «པེ་»:::«སྣར་»+hཚད་མ་རྣམ་འགྲེལ་དུ། སྐྱེ«པེ་»«སྣར་+hཔའི། ཁྱེ«པེ་»«སྣར་»+ར་ཡིན་པ། །
