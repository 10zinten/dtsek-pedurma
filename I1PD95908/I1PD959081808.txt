ཐུབ་པའི་དགོངས་པའི་རྒྱན། བྷིའི་(«པེ་»«སྣར་»རྒྱལ། «པེ་»«སྣར་+ར་བྱེད་པའི་སྒོ་ནས། Q«པེ་»«སྣར་+མཐུའི་སྒོ་ནས། །ནས་«པེ་»«སྣར་+ཇའཇིག་རྟེན་པ་དང་དེ་ལས་འདས་པའི། «པེ་»«སྣར་+ངས་བཞི་པ། ཚེ«པེ་»:ཁ་《སྣར་+hངག་དང་ཡིད་ཀྱི་སྤྱོད་པ་ལས་འདས་པའི་ཆེད་དུ་རྒྱུན་མི་འཆད་པར་བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་་་་་རྣམས་ཀྱིས་གནས་པའི་ཕྱིར་ན། །་ཚེ«པེ་»«སྣར་»+དམི་གཡོ་བ། «པེ་《སྣར་»+ཆབདེན་པ་བཞི་་་་:artསྒོམ་པ་ན་འཁོར་བ་ལ་མི་ཕྱོགས་པ་དང་མྱ་ངན་ལས་འདས་པ་ལ་ཕྱོགས་པའི་སེམས་བཟློག་སྟེ་ཐབས་ཀྱིས་ཟིན་་་::ནང་པའི་བྱང་ཆུབ་ཀྱི་ཕྱོགས་སྒོམ་ནུས་པ་ནས་བཟུང་ནས་ས་དྲུག་པ་ཐོབ་ཀྱི་བར་དུ་སྟེ། དེར་ཆོས་དབྱིངས་རྒྱུད་::ཀ་ཁ་རྣམས་ཐ་དད་མེད་དོན་དང་། ཞེས་པར་རྟོགས་ཤིང་འཁོར་བ་ལ་ཕྱིར་ཕྱོགས་པ་དང་མངོན་དུ་ཕྱོགས་པ་དང་བའི་མངོན་དུ་ཕྱོགས་པ་ཡིད་ལ་བྱེད་པའི་ངང་མྱ་ངན་ལས་འདས་པ་ཕྱིར་ཕྱོགས་པ་དང་མངོན་དུ་ཕྱོགས་པ་ཡིད་ལ་་་:ས་དང་བྱེད་པའི་ཤིན་ཏུ་རྨོངས་པ་དང་། དེ་དག་གི་གནས་ངན་ལེན་སྤོང་ངོ༌། །«པེ་»«སྣར་རིག(ཁྱེ«པེ་»《སྣར་»+hཆོས་དབྱིངས་བསྟོད་པར། རིགས་དང་། རྒྱུ་བརྩལ་བཟོ་གནས་ཀུན། །བསམ་གཏན་རྣམ་པ་སྣ་ཚོགས་ཉིད། །ཉོན་མོངས་ཤིན་ཏུ་སྦྱང་དཀའ་ལས། །རྣམ་པར་རྒྱལ་བས་སྦྱང་དཀའོ། །ཞེས་སོ། །ཀྱེ«པེ་»«སྣར་»+hརིན་ཆེན་ཕྲེང་བར། ལྔ་པ་ཤིན་ཏུ་སྦྱང་དཀའ་བདུད། །ཀུན་གྱི་ཤིན་ཏུ་ཐུབ་དཀའི་ཕྱིར། །TT"INVES®«»+va"SU®«s"<%+"Tă'SIL®«»»»:::+དགཅིག་ཏུ་གྱུར་ནས་མང་པོར་གྱུར་པ་སོགས། ་:«པེ་»«སྣར་»+bརྒྱལ་སྲས། «པེ་»«སྣར་»::ཀྱི+དཞིང་བརྩོན། །ཁྱེ«པེ་»ཀྱིས། +hཡོངས་སུ་སྨིན་པའི་ཁྱད་པར་བྱེད་འདོད་པས་ན་དེའི། «སྣར་»+hཡོངས་སུ་སྨིན་པའི་ཁྱད་པར་བྱེད་འདོད་པས་ན་དེའི། Q«པེ་»«སྣར་+དཞུགས་པའི། འཚེ«པེ་»«སྣར་+དཀྱི་ལམ། «པེ་»«སྣར་»+ཆདེ་འདོད་པས། །ས་ལ་བགྲོ་«པེ་»«སྣར་ན། +ང་དོན་དང་བྱ་བ་རྗེད་པའི་གནོད་པའི་གཉེན་པོར། ཀློ«པེ་»«སྣར་+ཇལྟ་བ་་་.ངན་པ་དང་བློ་ཆུང་བས་གནོད་པ་སེལ་བའི། །«པེ་»«སྣར་+ན་ནོར་བླངས་པ་མ་སྦས་པའི་གཉེན་པོར། །«པེ་»«སྣར་»+hའདུ་བ་དང་དགོད་པས་རྨོངས་པའི་གཉེན་པོར། །«པེ་»«སྣར་+རྣམས། །ལ་#@«པེ་《སྣར་»+ང་དབུལ་བའི་གཉེན་པོར། «པེ་རྒྱུ། +ཆདང་མ་ནོར་བསྒྱུར་བའི་དཔྱད། «སྣར་+:::དདམ་ནོར་བསྒྱུར་བའི་དཔྱད། «པེ་»གནོད། «པེ་»«སྣར་དུགཁྱེ«པེ་»«སྣར་+hཟློས་་་་ཏའི་གར་ལ་སོགས་པ། «པེ་གཏན་རྒྱུ། ་ཁ«པེ་《སྣར་»+hཡིད་མི་བདེ་བའི་གནོད་པའི་གཉེན་པོར། །»«པེ་»«སྣར་»+ཆགནས་ལ་མངོན་པར་མི་དགའ་བའི་གནོད་པའི་གཉེར་པོར། ཚེ«པེ་»«སྣར་+དཆུ་*pཀླུང༌། །::«པེ་»«སྣར་»འཚེའུ། ཨེ«པེ་»«སྣར་+དབསད་པ་དང་འཆིང་བ་ཆད་པ་ལ་སོགས་པའི་་་་་གཉེན་པོར། །ཁྱེ«པེ་»«སྣར་མན། «པེ་བྱུང་ཁུངས། 《སྣར་བྱུང་ཁུང་། །ཚེ«པེ་»«སྣར་»a+ཇམི་མཐོང་བ་དང་རྣམ་སྨིན་ཐོབ་པའི་ཡན་ལག་གི་ཉེས་པ་ལས་བྱུང་བའི་གཉེན་པོར། «པེ་»«སྣར་»། ཤི»ཟན༑+ཆགཉའ་འཛིན། དགྱེ«པེ་»«སྣར་ལ། +དབྱའི་ལ་ཉེ་དང༌། «ཅོ་»ཡང༌། མཉེ«པེ་»«སྣར་+::འཇལག་རིས་ལྟ་བུ། འབྱེ«པེ་»«སྣར་»+hའཆལ་བའི་ཚུལ་ཁྲིམས་ཉེ་བའི་ཉོན་མོངས་པའི་གཉེན་པོ། «པེ་»:《སྣར་+hའདོད་པའི་འདོད་ཆགས་ཀྱི་གཉེར་པོར། འགུ«པེ་»«སྣར་+Eསྐྱབས་ངན་པའི་གཉེན་པོར། །
