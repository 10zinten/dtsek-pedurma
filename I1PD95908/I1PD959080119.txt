བཤེས་གཉེན་གྱིས་བྱས་སྐུ་གསུམ་འདི་དག་འགྲེལ་པ་བརྩེ་བ་ལྟ་བུའི་ཚུལ་གྱིས་བྱས། །སྐུ་གསུམ་གྱི་འགྲེལ་པ་སློབ་དཔོན་ཡེ་ཤེས་ཟླ་བས་བགྱིས་པ་རྫོགས་སོ། །་་། རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཝརྨ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བན་དེ་ཡེ་ཤེས་་་སྡེས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ་། :། དེ་ན་ང་ལ་བཅོས་མ་འདི་ལ་བློ་ཀ་ནི་བདུན་བརྒྱ་དྲུག་ཅུ་མཆིས་སོ། །བློ་གུ་ཁོ་བོས་འབད་པ་ལ། ཁོ་བདུན་པ་ནི། ད་བར་བྱེད་པོ་གཞུན་དད་ཀྱི་ཁ་བ་ལ་བསྡུར་མཆན། aཔ་སྟེ། ཚོ་::ཀྱིང་ཚོ་་་ནན་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་སྐུ་གསུམ་གྱི་་འགྲེལ་པ་ཡེ་ཤེས་ཟླ་བས་མཛད་པ་བཞུགས། ཞེས་འཁོད། «སྣར་»ལྟའོ། «པེ་《སྣར་»དེ། «སྣར་»ཀྱི། «སྣར་»—ཞིང༌། «པེ་»«སྣར་»+པའི། «པེ་»«སྣར་བརྟག«པེ་»《སྣར་བསྟན། «པེ་»«སྣར་ནི། མཉི«པེ་»«སྣར་བརྟན། ་་་་@«པེ་《སྣར་ཅན་གྱི། «སྣར་»གི། «པེ་»«སྣར་»ཆོས་ཀྱི་སྐུ། «པེ་»«སྣར་པ།  ེ«སྣར་ཉིད། Q«སྣར་+པ། Q«སྣར་རྟོགས། «སྣར་»བློ། མ«པེ་»«སྣར་+ནི། །«པེ་»«སྣར་+དང་། ་་་། «པེ་»«སྣར་ཀྱི། «སྣར་»དེ། «པེ་»«སྣར་+པ། «པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་ལ། «པེ་»«སྣར་»བལྟོས། 《པེ་»«སྣར་»+རྣམས། «པེ་»«སྣར་མར། «པེ་»«སྣར་»+རྣམ། །«པེ་»«སྣར་གེགས། @«པེ་»«སྣར་གདགས། ༄༅«པེ་»«སྣར་བསྒྲུབས། ཚེ«པེ་»«སྣར་»གི། «པེ་»«སྣར་+ནི། «པེ་»«སྣར་»སྩལ། «པེ་»«སྣར་+རྣམ་པར། &«པེ་»«སྣར་བྱ་བར། «པེ་»«སྣར་བསྒྲུབ། །(ི«པེ་»«སྣར་»ཞེ་ན། «པེ་»«སྣར་»རྟོགས། @«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་སྲུང་། «པེ་«སྣར་གྱིས་བརླབས། «པེ་»«སྣར་+དེ། «པེ་»«སྣར་»གྱིས་བརླབས། མི«སྣར་»དཔའི། །@«སྣར་»ད། ཚེ«པེ་»«སྣར་»བསྐལ་པ། །ཉི«པེ་»«སྣར་»—།
