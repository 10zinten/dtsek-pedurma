::@«པེ་»«སྣར་+དསྙིང་རྗེའི་སྒོ་ནས། །«པེ་»«སྣར་»+hཅིའོ། «པེ་»«སྣར་»གི། O«པེ་»«སྣར་+ཇཉོན་མོངས་པའི། «པེ་»«སྣར་གཟུང་། +ཇཞྭ་དེས་ལྷམ་ལ་བཀབ། (གྱེ«པེ་»«སྣར་»+ཇམི་དགེ་བ། བཀྱེ«པེ་»«སྣར་བྱུང་། དགྱེ«པེ་»«སྣར་+དསྡུག་བསྔལ་ཆུང་ཤོས། ནེ«པེ་»«སྣར་»+དམི་དགེ་བ། །འཉེ«པེ་»«སྣར་»+རྣམི་དགེ་བ་བཅུ་བྱེད་པ། མཉེ«པེ་»«སྣར་»+རྒརིན་ཆེན་ཕྲེང་བར། གསོད་པ་ཡིས་ནི་ཚེ་ཐུང་འགྱུར། །རྣམ་པར་འཚེ་བས་གནོད་པ་མང་། ཕྲ་མ་ཡིས་ནི་བཤེས་དང་འབྱེད་པ་དང་། «པེ་»སྣར་+དང་། འབུ«པེ་»སྣར་»+དརྩུབ་པོ་ཡིས་ནི་མི་སྙན་ཐོས། ་འཁྱེ«པེ་»«སྣར་+དམ་འབྲེལ་པ་ཡིས་ཚིག་མི་བཙུན། ནེ«པེ་»«སྣར་+ཆབརྣབ་སེམས་ཡིད་ལ་རེ་བ་འཇོམས། འགེ«པེ་»«སྣར་+ཆགནོད་སེམས་འཇིགས་པ་སྦྱིན་པར་བྱེད། འཚེ«པེ་»«སྣར་»+ཇལོག་པར་ལྟ་བ་ལྟར་ན་ཉིད་ཅེས་ཀྱང་གསུངས། །ར་ར་ཞིང་ཅེས་ཀྱང་གསུངས། ང་«པེ་»«སྣར་+དསེར་བ་དང་རྡུལ་འབབ་པ་དང་རྩྭ་སོགས་གང་གིས་འབྲུ་རྣམས་མ་རུངས་པར་འགྱུར་་་་་ཞེས་རྒྱལ་པོ་སྲས། «པེ་»«སྣར་+hཚིག་རྩུབ་ཀྱིས། ཉི«པེ་»«སྣར་+hཚིག་བཀྱལ་པས། «པེ་»«སྣར་+hཚ་བའི་དུས་སུ་ཧ་ཅང་ཚ་ཞིང་གྲང་བའི་དུས་སུ་ཧ་ཅང་གྲང་བ། ག«པེ་»«སྣར་»+hའམ་རྩུབ་པ། ་ ོ«པེ་»«སྣར་+ཆདེ་མཛོད་དུ་ཐམས་ཅད་བདག་པོར་རྒྱུ་མཐུན་དང༌། རྣམ་སྨིན་འབྲས་བུ་འབྱིན་པར་འདོད། །ཅེས་པས་ལྟ།
