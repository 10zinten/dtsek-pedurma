རྒྱ་གར་གྱི་མཁན་པོ་མཉྫ་ཤྲཱི་གབྷ་དང་། པྲཛྙཱ་ཝརྨ་དང་། ཞུ་ཆེན་་གྱི་ལོ་ཙཱ་བ་བནྡེ་ཡེ་ཤེས་སྡེ་དང་། བནྡེ་དཔལ་བརྩེགས་ཀྱིས་བསྒྱུར་ཏེ་ཞུས་ནས་་གཏན་ལ་ཕབ་པ@། ཁ་ས། །ཀུ་ལ་། ་ཆས། ཆ་ཀཀྱི་ལ་དབང་་བསྡུར་མཆན། བབབབཔ་་་་ཀི་བསད་ཀྱི་སྐྱེ། བདག་རྒྱུ། བོད་པ་སྤྲུའི་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་༄༅། །འཕགས་པ་ས་བཅུའི་རྣམ་པར་བཤད་པ་བཞུགས་སོ། །ཞེས་འཁོད།  ེ«སྣར་དྷཱ། Q«པེ་»«སྣར་བཅུའི། :ན་བ་«པེ་»«སྣར་»གྱིས་བརླབས། ༼ལུས་ཀྱི་བ་ལ། མཚོ་ཀྱི«སྣར་བསྒྱུར། «པེ་»«སྣར་གཅིག«པེ་»《སྣར་»+པ། «པེ་»«སྣར་»ཅིའི་་་བ་ཕྱིར། «པེ་»«སྣར་—གནས་པས། «སྣར་—ལ། @«པེ་»«སྣར་པར། «སྣར་བསྟོན། དེ«པེ་»«སྣར་»དཔའ། (ཚེ《སྣར་»གིས། །རབ་བབ་རྗེས་། ་«པེ་»«སྣར་»པ།  ེ«པེ་»«སྣར་»རྗེའི། «པེ་《སྣར་—དང་འདྲ་བས་རྡོ་རྗེའི། «སྣར་»ཤིག@«པེ་རབས། «པེ་»«སྣར་གི། །«པེ་»«སྣར་»ཀྱིས། ®<«པེ་»«སྣར་»གྱིས་བརླབས། «པེ་»«སྣར་+ཏེ། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་»གིས། Q«པེ་»སྣར་+གྱིས། «པེ་»གྱིས་བརླབས་ཀྱིས། 《སྣར་གྱིས་བརླབས་ཀྱི། ལེན@«པེ་»«སྣར་བརླབས། «སྣར་»—སྟེ། @«པེ་»«སྣར་པར། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་ཁོང་དུ། ཚེ«པེ་»«སྣར་ཐོས། ཀྱེ«པེ་»«སྣར་འགོད། ས་གནས་ཀྱི་«པེ་»《སྣར་རྟོག«པེ་»«སྣར་»བརླབས། «པེ་»«སྣར་»ཀྱིས། ལས་དང་ཨེ«པེ་»«སྣར་»རྟོགས། «པེ་གིས། «པེ་»«སྣར་པའོ། «པེ་»«སྣར་»—བྱེད་པས་ཆོས་རྟོགས་པར། (@«པེ་»«སྣར་»+ནི། «པེ་»«སྣར་»སྦྱོང་བས། ཁ@«པེ་»«སྣར་—གྱིས། «པེ་»ཀྱིས་ས་འཐོབ། «སྣར་ཀྱིས་ས་ཐོབ། «པེ་»«སྣར་བསྒོམས། (གྱེ«པེ་»«སྣར་གིས། །«པེ་»«སྣར་ནི། «པེ་»«སྣར་རྟེན། «པེ་»«སྣར་»—དེ། ལ་་་་་རྒྱེསྣར་རིག«པེ་《སྣར་ལ་ཡང་། །Q«པེ་»«སྣར་བརྩོམ། ༄༅«པེ་»«སྣར་—འཛིན་པ། «པེ་»«སྣར་པ། @«པེ་»«སྣར་རྟོག@«པེ་»«སྣར་»སྒོམ། །«པེ་»«སྣར་»གིས། «སྣར་འཐོབ། »«པེ་»«སྣར་»མི། &«སྣར་»ཀྱིས། «པེ་»གིས། «པེ་»«སྣར་འཐོབ། །«སྣར་»གིས། «སྣར་པར། །«པེ་པ་རིགས། 《སྣར་པ་རིག། «པེ་»«སྣར་པས་དྲིས་པ། ་ནིར་སྒོམ། །།
