པ་་་་་། ཀ་ཚེ་ཆུད། ད་ཆ/n། བར་དུ་སྤྱོད་ཀྱིས་ཀུན་ན་:་་། བལ། ༼ཀ༌ཀུདང་ལག་པ༌ཟླ་་བསྡུར་མཆན། །ནས་དང་། ས་ཁལ་ཡང་ཁོང་བར་བ། སྤྱིན་འད། ༽༼སྤྲོད། པར་མནཔཔ་ག་ཏཡལམ་དུ། །རྒྱུ(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»ལ་༄༅། །ཆོས་རྗེས་སུ་་དྲན་པའི་འགྲེལ་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་དྷརྨ། «པེ་»«སྣར་+པར། «པེ་»«སྣར་རིགས། འ«པེ་»«སྣར་»འཐོབ། །འོ་འོ་ལས་ས་ལ་«སྣར་མིན། «པེ་»«སྣར་སྤོང་བ། ཉེ«པེ་»«སྣར་བྱ་བའོ། &«པེ་《སྣར་དུ། «པེ་»《སྣར་»ཀྱིས། བ་བ་ན་དུ་བ་ན། །དོ། ,དབུས་ལ་། ས་སྐྱའི་ཡང་ਸਰੋਂਉਸਨੂੰਸਰੌਣབསྡུར་འབྲས་རེའུ་མིགམདོ་སྡེ། བདེ་སྐ་་་ངང་གྲོ། པ་དང་འགྲོ་དང། ན་དཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། །སྣར་ཐང་། །ཅོ་ནེ། :::::སྡེ་ཚན། །མདོ་སྡེ། དཀོན་མཆོག་རྗེས་དྲན། །།མདོ་སྡེ། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །ན_བ། བ—ནས། ན—ན། བ་ན་པའི་ཕྲེང་། །ཚེག་ཁྱིམ། །།།པ་ཞིག་བྱེད་པ། ད་ལོའི་བར་ན། ཁོམਉਸਨੂੰਪਿਆਰਨਾਲਮੇਲਖਾਨ，མདོ། ལ། ད་ལན་འདུག་པ་དེ་ནས་བྱ་བ་ནི་བསྟན་།
