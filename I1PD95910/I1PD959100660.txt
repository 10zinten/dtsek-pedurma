མེར་ཕྱི། རཔོ་་་་་་་་་་་་་་་་་་་་་་་ཏ། དང༌། ཀ་ལན་པ། །བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། །རྒྱལ་ཟླ་བ་བརྒྱ་བའི་ལམ་ལ་འཇུག་པ་ཞེས་བྱ་བ་པཎྜི་ཏ་བུདྡྷ་ཤྲཱི་ནས་བོད་དུ་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་»མཱཧཱ། ཚེ«པེ་»«སྣར་»སྤངས། «པེ་»«སྣར་»གི། །དྲ་བ་མང་པོའི་བདུད་བརྒྱ་གང«པེ་»«སྣར་རིགས། «སྣར་གྱུར། ཚེ«པེ་»«སྣར་མདོ་དང་། «པེ་»«སྣར་སྲིད། «པེ་»zfzje《སྣར་+རང་བཞིན་ཁ་ན་མ་ཐོར་བཅས། །«པེ་»«སྣར་རིམ། :བ་དང་སྐུཚེ«པེ་»«སྣར་རིངས། «པེ་»«སྣར་»ཡི། «སྣར་བྱུང་། «པེ་»«སྣར་པ། ག«སྣར་»/རིག«པེ་»«སྣར་»—སྟེ། ༄༅«སྣར་གཤེས། «པེ་»«སྣར་མཚོ། ས་ལ་རྒྱལ་བ་ད«པེ་»«སྣར་མཆེ། «པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་བ། «པེ་»«སྣར་»གསད་བྱ་ཡི། དpཚུ«པེ་»«སྣར་གནས། @«པེ་»«སྣར་པར། ་་་་་་་་་་། ཆུང་བ་བk«པེ་»«སྣར་རྣམས་དཔྱོད་མིད། «པེ་»«སྣར་རྫུན། «པེ་»«སྣར་ཡང་ནོད། ︽«པེ་»«སྣར་»ལྷུང་། «པེ་»«སྣར་བརྟན། «པེ་»«སྣར་»ཡོན། »ཟླ་《p«པེ་»《སྣར་+ཞེས་བྱ་བ། གནའ་དེང་གང་། །མ་སྐྱེས་དpz«པེ་»«སྣར་རྒྱུད། @«པེ་»«སྣར་»—ལས། «པེ་»«སྣར་»གསུང༌། «པེ་»«སྣར་པའོ། །«པེ་ཟད། «པེ་»«སྣར་»ཡོངས། ཚེ«པེ་»«སྣར་»དྲག«པེ་»«སྣར་»དེས། མཚོ«པེ་»«སྣར་»སེམས་རྣམས། «པེ་»«སྣར་»གསོལ། «སྣར་ཡི། «ཅོ་སྲིད། མ«པེ་»«སྣར་»«ཅོ་བཟུང་། «པེ་»«སྣར་བསྲུང་། ་ཀྱི་ཚེ་དང་རྒྱུ་»། «པེ་»«སྣར་»སྡུས། ®«པེ་»«སྣར་པས། «པེ་»«སྣར་»སྐྱེད། «པེ་»«སྣར་ནི། ::སྐྱེ་«པེ་《སྣར་པའི། ®«པེ་»«སྣར་བརྟན། ཚེ«པེ་»«སྣར་རིགས། «པེ་»«སྣར་»ཆེད། །«པེ་»། ཞི་«སྣར་»སྨིན། :་དེ་འགོ་ཚུགས་པ་དེས་ཁོ་ན་གཙོ་པ་དེ་«པེ་»«སྣར་པའོ། «པེ་»«སྣར་»«ཅོ་»པ། «པེ་《སྣར་»སྐྱོད། «པེ་»«སྣར་»བ། «པེ་»།
