ང་ར་རྗེ་གསོར་ནས་སུ། །སེམས་ཅན་བྱང་ཆུབ་ལ་ནི་དགོད། །རྣམ་པར་དག་པའི་་ཆོས་ཀྱི་དོན། །རྡོ་རྗེ་འཆང་གིས་གཟུང་བར་བྱའོ། །ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧོ་ཧོ་ཧོ། །དེ་་་ནས་རང་གི་དྲིལ་བུ་ནི། །ནང་གི་ནམ་མཁའི་ཁམས་དག་ནི། །སྟོང་པ་ཉིད་ཀྱི་ཡེ་ཤེས་ཡིན། །སྟེང་གི་པདྨ་འདབ་བརྒྱད་ལ། །ཕྱོགས་བཞིར་སངས་རྒྱས་སྤྱན་སོགས་བཞི། །ཕྲིན་ལས་བཞི་ཡི་རང་བཞིན་ཡིན། །མཚམས་བཞི་ལ་ནི་སྐུལ་་་བྱེད་དང་། །ལུག་རྒྱུད་ཁྲོ་གཉེར་རིན་ཆེན་སྒྲོན། །ལྟེ་བ་ཤེས་རབ་ཕ་རོལ་ཕྱིན། །ཀུན་བཟང་གཟུགས་རྫོགས་འོད་ལྡན་བསམ། །རྩེ་ལྔ་རྩ་བའི་འདབ་བརྒྱད་ལ། །སྔོན་བཞིན་སྒེག་མོ་ལ་སོགས་བརྒྱད། །རྩེ་ལྔ་ཉིད་ནི་ཡེ་ཤེས་ལྔ། །དེ་ལྟ་བུ་ཡི་་དྲིལ་བུ་ལ། །ང་། སྔགས་འདི་དག་ནི་བརྗོད་ནས་ཀྱང་། །སྤྱི་བོར་དཀྲོལ་ཏེ་དབང་བསྐུར་རོ། །དེ་ནས་ཕབ་ལ་ལན་གསུམ་དཀྲོལ། །ཨོཾ་བཛྲ་ར་ཎི་ཏ་དྷརྨ་ར་ཎི་ཏ་སམྤ་ར་ཎི་ཏ་སརྦ་བུདྡྷ་ཀྵེ་ཏྲ་པྲ་ཙ་ལི་ཏ་པྲཛྙཱ་པཱ་ར་མི་ཏཱ་ནཱ་ད་སྭ་བྷཱ་བེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧོ་ཧོ་ཧོ་སྭཱ་ཧཱ། ཤེས་རབ་ཆོས་ཀྱི་དྲིལ་བུ་འདི། །སངས་རྒྱས་ཐམས་ཅད་བླངས་་བྱས་ཏེ། ལ་། མི་ཤེས་འདམ་དུ་བྱིང་བ་ཡི། །སེམས་ཅན་སངས་རྒྱས་དོན་དུ་དཀྲོལ། །དེ་ནས་གསང་བའི་ཐབ་ཁུང་མི་དམིགས་པ་ལས། །ཡི་གེ་ཧཱུྃ། ལྷའི་ཆོས་ཀྱི་ཡི་་གེ་ལས་ཐབ་ཁུང་སྔར་བསྐྱེད་པ་བཞིན་བསྐྱེད་ལ། །དེ་ནས་ཧྲིལས་པདྨ་འདབ་མ་བརྒྱད་པ་བསྐྱེད་ལ། །རྒྱུ་དེའི་དབུས་ཀྱི་ལྟེ་བ་ལ་ཡི་གེ་རཾ་འབར་བ་ལས་གྱུར་པའི་གྲུ་གསུམ་བསྐྱེད་ལ། གྲུ་གསུམ་དེའི་ཤར་ལ་སོགས་པའི་ཟུར་ལས། །ཚོད་ཨོཾ་ཨཱཧཱུྃ་་་གསུམ་ཁམས་གསུམ་གྱི་བདག་པོར་བསྒོམས་པ་ལས། །འཁོར་ལོ་དང་པདྨ་དང་། རྡོ་རྗེ་རང་གི་ཡི་གེས་མཚན་པ་བསྐྱེད་ལ། དེའི་དབུས་སུ་རཾ་ལས་བྱུང་བའི་མེ་རཾ་དང་བཅས་པ་མཐོང་སྟེ། །གོང་གི་ཕྱག་མཚན་གསུམ་རཾ་དང་བཅས་པ་འོད་དུ་ཞུ་བ་ལས་མེ་འབར་བར་བསམ་ཞིང་། དམ་ཚིག་གི་གསང་སྔགས། ཧཱུྃ་བཛྲ་ཨ་ན་།
