བསྡུས་པ་ཞེས་བྱ་བ། ::༄༅། །རྒྱ་གར་སྐད་དུ། པྲ་ཏིཥྛ་བི་དྷི་སཾ་ཀྐིསྟེ། བོད་སྐད་དུ། རབ་གནས་ཀྱི་ཆོ་ག་མདོར་བསྡུས་པ་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །འཇམ་དབྱངས་ཆེན་པོའི་ཡོན་ཏན་ནི། །རིན་ཆེན་འོད་་་ཀྱིས་བརྒྱན་པ་ལ། །བཏུད་ནས་རབ་གནས་ཆོ་ག་ནི། །ལྷག་པར་འདིར་བསྡུས་བརྗོད་པར་བྱ། །:བདེ་གཤེགས་རྟོགས་མེད་གོ་འཕང་བརྙེས། །དྲི་མེད་་མཁའ་འདྲའི་སྐུ་མངའ་ཞིང༌། །བརྗོད་དུ་མེད་པ་དེ་དག་ལ། །རབ་གནས་་འཐད་པ་མ་ཡིན་ཏེ། །དེ་དག་སྐུ་ནི་ཀུན་རྫོགས་པ། །བརྩེ་བཟང་རོ་ལས་བྱུང་བ་ལྟར། །དེ་ལ་བརྟེན་ནས་དགེ་བ་ནི། །མཐའ་ཡས་སྟེར་ཏེ་འཐད་པའང་་་ཡིན། །རྣལ་འབྱོར་དེ་ཉིད་ཤེས་སྐྱོན་བཅོམ། །བརྩེ་བའི་རང་བཞིན་རིགས་་སུ་སྐྱེས། །མདངས་ལྡན་བསྒོམ་པ་ཤེས་རབ་ཅན། །ཀུན་ལ་མཁས་ཤིང་དད་པ་ཡིས། །ཆུ་ཀླུང་ཆུ་དག་བཏབ་པ་ཡིས། །སེམས་སྦྱངས་མཆོག་ནི་སློབ་དཔོན་།
