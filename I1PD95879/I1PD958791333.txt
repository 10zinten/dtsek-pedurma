བ་བྱས་ཏེ་བསྟན་འགྱུར། རྒྱུད། ཇུལ་ཀ་ཏའི་:::་་བསྡུར་འབྲས་རེའུ་མིག་ནི་དེ་:དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང༌། །ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །མ་རྒྱུད། རྒྱུད། །།།པོད་རྟགས། ཚན་གྲངས། སྐྱེ་མེད། ལྡེབ། །ནབ། བབས། ནབན?བ། ་་། བ་ཚེག་ཁྱིམ། །བབྱེད་ཕྲེང་། །ཏི་ན་་ས་་་མ་མ་ཁས་། བདེ་བ་དེ་བདེ་བ་དང་། །་་་་་་་་་་་་་་པ་ལས་།
