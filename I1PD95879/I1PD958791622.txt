::ཏེ། ཏཁྲུ་སྦྲུལ། ཆན་ལོག་བྱེད་པ་འཐེན་གྲོས་བཟང་བ་ཙཎྜ་ན་ཆོ་:སུ། །།epཀ་སྙིང་པོ། བླ་མ::དབང་ག་ལ་འོས་འཚོ་། །འཁོས་:བསྒོམཁ་ལ་ཉེ་ཁང་པ་ན་ལ་ཏུ། ལ་ལ་གདུཎཊ་ཏ་ཏ་ཏེའོ། ཞི་བའི་པ་འདྲ། ན་པ། ་ང་ལ་། ཁ་ཏིང་ནས་བཏོན་ཏེ་། ཏི། :ཀྱིཀྱི་་་་་:::ངབནཔ་། དེ་། ཏི::༄༅། །མི་འཁྲུགས་པའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། བླ་ན་དག་བྱེད་དག་ལ་ཁ་ཏི་ཞེས་པའི་ན་གར་ཞ་དང་། ཀ་བ། ཆོས། འོན་ཏེ་ཡང་སྤྱི་ལ་ཁ་འཆ། པ། ས། །ལ་ཁ་ཞུ་རྒྱུ་བཅཆ་ཁ་ཤས་བསད་པ་མཐོང་ཡང༌། །ཀ་ལ་རྗེ་ཏི་པ་གཏེ། ཡ་ཀ་པ། ད་པ། ན་ང་ར་བ་དང་་། ་་་་་། བ་བླངས། །ཁྱོད་ཚོ་ན་ཁྲི་བ་::། ཁུ་ཚོ་དང་ཁ་བ་༄༅། །རྒྱ་གར་སྐད་དུ། :：ཨཀོ་བྱ་སཱ་དྷ་ནཾ་ནཱ་མ། བོད་སྐད་དུ། མི་འཁྲུགས་པའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ་། བཅོམ་ལྡན་འདས་མི་འཁྲུགས་པ་ལ་་་་ཕྱག་འཚལ་ལོ། །མི་འཁྲུགས་པ་ལ་ཕྱག་འཚལ་ཏེ། །དེ་ཡི་སྒྲུབ་ཐབས་བྲི་བར་བྱ། །དལ་འབྱོར་ཐེག་ཆེན་རིགས་ཆེན་དང༌། །སྙིང་རྗེ་ཤེས་རབ་ཡོན་ཏན་དང་། །བྱམས་དང་སྙིང་རྗེ་ལྡན་པ་དང་། །དད་ཅིང་བླ་མ་ལ་གུས་པས། །སྐྱབས་སུ་སོང་ཞིང་སེམས་བསྐྱེད་པ། །དམ་ཚིག་སྡོམ་ལྡན་དབང་ཐོབ་པས། །ལས་དང་ཆོ་ག་ཤེས་ནས་ནི། ::། དགོན་པའི་གནས་དང་གྲོང་གི་མཐར། །ལས་་བྱ་བདེ་བའི་སྟན་འདུག་ནས། །ཁྲུས་བྱས་གཙང་མའི་གོས་ཅན་གྱིས། །རང་་་ལྷར་གསལ་བའི་ཐུགས་ཀ་རུ། །པདྨ་ཟླ་བའི་སྟེང་དུ་ཧཱུཾ། །དེ་ནས་འོད་ཟེར་་འཕྲོས་པ་ཡིས། །བླ་མ་དང་ནི་མི་བསྐྱོད་པ། །སྤྱན་དྲངས་མདུན་དུ་བཞུགས་་པ་ལ། །བདུན་པོ་རྣམ་པར་དག་པར་བྱ། །ཚད་མེད་བཞི་དང་ལྡན་པར་བསྒོམ། །—།
