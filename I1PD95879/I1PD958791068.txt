ཉིད་དུ་དམིགས་ཏེ། བཟའ་བ་དང་། བཏུང་བ་ལ་སོགས་པ་ཐམས་ཅད་སློབ་དཔོན་དང་། སློབ་མས་ཇི་ལྟར་དམ་ཚིག་ལས་འབྱུང་བ་བཞིན་དུ་མཆོད་པར་བྱ་ཞིང་བཟའ་བར་བྱའོ། །སློབ་དཔོན་རུ་ར་བྷ་དྲས། དཀྱིལ་འཁོར་ཐམས་ཅད་ཀྱི་སྤྱིའི་རིམ་པ་ཞལ་དང་འདྲ་བས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་དྷརྨ་ཤྲཱི་བྷ་དྲ་དང་། ་་:ལོ་ཙཱ་བ་དགེ་སློང་རིག་པ་་་་གཞོན་ནུས་བསྒྱུར་བ། །།།ཡིག་དང་། ཁ་ཏོ། །དེའི་འགོ་ཀྱི་རུ་ཁརང:ད་ཀྱི་བདུར་མཆན། །ཟླ་བ་བདེ་མ་ཀྱི་(?བར་་་་་། །།ཀ༌བ་བའིདེབ་སྐྱེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»《སྣར་»དཀྱིལ་འཁོར་::སྤྱིའི་རིམ་པ་སློབ་དཔོན་ལྷའི་བཟང་པོས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་»པཱ། «པེ་»«སྣར་པའོ། །།།«པེ་»«སྣར་བས། «པེ་»«སྣར་»ཞུགས། «སྣར་ལས། «པེ་《སྣར་གཙུགས། །«པེ་»«སྣར་བསླང༌། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་པ། «པེ་»སྣར་ཀྱིས། «པེ་»«སྣར་ནས། «པེ་»«སྣར་ན། ེ«སྣར་བསྒོམ། (ེ«པེ་»«སྣར་གི། «པེ་»«སྣར་»གྱིས། ཀྱི«སྣར་བསྒྲིབ། »«པེ་»«སྣར་+སུ།  ི«པེ་《སྣར་»དུ། ®«སྣར་»ན། «སྣར་»བསྲེག«སྣར་»སྐྲད། @«སྣར་»ཉིད། «པེ་»«སྣར་»ཚ། ་་་་་་ཞེས་«པེ་»«སྣར་»ཀྱིས། ་ང«པེ་»«སྣར་»ཟླ། «པེ་»«སྣར་བསམ་སྟེ། «སྣར་»+ལོ། ་ར་«པེ་»«སྣར་»བ། «སྣར་བལྟས་ལ། 《@«པེ་»«སྣར་»བ། «པེ་»«སྣར་»འཕོངས། ལོ་«པེ་»《སྣར་ལས། «པེ་»«སྣར་སྣུམ། («པེ་»«སྣར་པའི། ་(མེ«པེ་《སྣར་བཅག་ན། ཅིང་བཀང་། (སྟེ«པེ་»«སྣར་པ། («པེ་»«སྣར་»—སྐར་ཁུང་། །ནམ་ ི«པེ་»«སྣར་+དང་། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་བྱུང༌། །®«པེ་»«སྣར་+པ། བ་དང་། «སྣར་»འབབས། །«པེ་»«སྣར་»ཀྱིས། །ཚེ«པེ་»«སྣར་»བའི། («པེ་བཞུག«སྣར་»::ཀྱིངབབཞུགས། «པེ་»«སྣར་»—ས། («པེ་»«སྣར་»དུས། ་(«པེ་»«སྣར་+བྱས། །«པེ་»«སྣར་»ཁྱོད། («སྣར་བརླབས། །«པེ་《སྣར་ཟླའི། «པེ་གཟུང༌། «པེ་»«སྣར་རྗེ། «པེ་»«སྣར་»རྫོགས། @«པེ་»«སྣར་»དྲི་ཡིས། («པེ་»«སྣར་གྲུབ། །«པེ་»«སྣར་བའི། Q«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་»དཀྱིལ་ཅིང་། །«པེ་»《སྣར་»གྱིས། —།
