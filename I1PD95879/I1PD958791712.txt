རྒྱན་དམ་པ་སྣ་ཚོགས་པ་རིམ་པར་ཕྱེ་བ་བཀོད་པ་བཞུགས། མཐར་འབྱིན་པ། །དགོངས་པ་མཐའ་དག་རྫོགས་པའི་བཀྲ་ཤིས་གང་ཡིན་པ། །བཀྲ་ཤིས་དེས་ནི་དེང་འདིར་ཞི་བྱེད་ལེགས་གྱུར་ཤོག། ཡ་ན་མཾ་ག་ལན། ཤུ་་བྷ་ཤ་མནྟོ། ས་ཡ་ཛི་ན་ཏ་ཀ་ཤ་སྱེ་ཨ་ཀོ་བྱ་རཱ་ཛ། བི་ཤི་ན་ཏ་སྱེལ། སརྦ་ཀུན༑མཾ་ག་ལ་སྐྱེ། །ནཾ་མཾ་ག་ལ་ད། །བྷ་བ་ཏུ་ཤཱནྟེ། །ཀ་ས་པྲཛྷ་སྟ་་་། བནྟུ་ཧུཾ། ་ཡ་མཾ་ག་ལན། མ་ནི་ཀུ་ལན་ཧི་ན་བུ་ཧ་ར། །པ་བ་ཤུ་ཤ་ཤ་ཙི་ན་སེ་ཏ་ནོ་ཏོ་ནེ། ་ཁ་ལ་ན་པི་པི་པུ་ཏི་ན། ཏན་མཾ་ག་ལན། ་ཏན་མཾ་གཱ་ལན་བྷ་:བ་ཏུ་ཤཱནྟེ། །ཀ་ར་ན་བད་རྒྱ་ཡན་མཾ་ག་ལན། །ཨ་ག་ནི། །ཨཱ་ཡུར་བྱིད་ཏ་ས་ཀྱེ་མ་ནེ་བུ་ར་ཏ་ལི་ཡས་སྐྱེ། གཱརྣ་བཛྲ། པ་རི་ག་ཏེ་སྐྱེ། ཏ་ཐུ་ག་་ཏེ་སྨི། ཏན་མཾ་ག་ལན། བྷ་:བ་ཏུ་ཤཱནྟེ'། ་ཀར་ནརྱ་ཕཊ་ཏྲ། །ཡན་མཾ་ག་ལན། མ་མ་མུ་ནེ་པ་ར་ཤ་སྱེ། ་ཨ་མོ་གྷ་སིདྡྷི:༔སཏཱ་ཏཱ་ར་གར་མ་ན་ནེ། བྷུ་ན་ཤ་བ་ན། ཤ་ཏ་ལེ་ཀྱེ་ཡོ་ཙི་ཏ་མན་ས་ས་སྱ་མུ་ན་ས་སྐྱེ། བནྡྷ་བྷ་བྷ་མྱེ་བྱ། ཏན་མཾ་ག་ལན། བྷ་:བ་ཏུ་ཤཱནྟེ། རྣ་ཏ་ཏྭཾ། ཡན་མཾ་ག་ལན། ས་ཀ་:ལ་མཱ་ཏོར་ཏ་ས་སྐྱེ། བི་རོ་ཙ་ན་ཀྱེ། བེ་ར་ཏ་ཉི་མ། མ་ཧཱ་ར་རྣ་བ་སྐྱེ། མ་ཧཱ་ར་རྣ་བྷུ་ར་ར་རྣ། ཙ་ཏྲ་བྷུ་ཤེ་ཧེ་བནྟ་ཀ་ཕྱེ། །ཏན་མཾ་ག་ལན། བྷ་:བ་ཏུ་ཤཱནྟེ་ཀ་ར་ཏ་བཙྩཾ། །དེ་ལ་དེ་བཞིན་གཤེགས་གཙོ་དང་། །གང་། ཆོས་རྣམས་ཀུན་ལ་དབང་བསྐུར་རོ། །རྡོ་རྗེ་རིགས་གཙོ་ཆེན་པོ་དང༌། །ཁམས་གསུམ་བདག་པོར་དབང་བསྐུར་རོ། །རིན་ཆེན་རིགས་གཙོ་ཆེན་པོ་དང་། །ལྷ་མིས་མཆོད་པར་་དབང་བསྐུར་རོ། །པདྨའི་རིགས་གཙོ་ཆེན་པོ་དང༌། །ཆོས་ཀྱི་རྒྱལ་པོར་དབང་བསྐུར་རོ། །ལས་ཀྱི་རིགས་གཙོ་ཆེན་པོ་དང་། །ལས་བཞིས་གཞན་དོན་སྤྱོད་དབང་བསྐུར། །བཀྲ་ཤིས་བརྗོད་པའི་བྱིན་རླབས་ཀྱིས། །འཇིག་རྟེན་དག་ཏུ་མི་རྣམས་ཀུན། །གྲགས་པ་རྣམ་སྤེལ་བགེགས་ཀྱང་མེད། །སེམས་ཉིད་བདེ་་།
