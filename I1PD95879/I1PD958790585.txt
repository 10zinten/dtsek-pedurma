རྒྱུད། ཆུཚོ་མ་ཀ་ཚལ་ཀ་ཎ་ཡ། སྟོན་ཏོ། །དེ་ནས་འབྲུབ་ཁུང་རྣམས་གཅིག་ནི། །ཞེས་བྱ་བ་ནས། །རྡོ་རྗེ་་་འདོད་པ་རྟག་ཏུ་བཟླས། །དབང་གི་ལས་སོ་ཞེས་བྱ་བའི་གཞུང་གིས་སྦྱིན་སྲེག་གི་ཆོ་ག་སྟོན་ཏོ། །རྟོག་པ་ཐམས་ཅད་ཀྱི་དཀྱིལ་འཁོར་རྣམས་ལ་ཡང་མཆོད་པ་རྣམས་ཀྱིས་ཡང་དག་པར་མཆོད་དེ་བསྟོད་ནས་ཕྱག་རྒྱ་དགྲོལ་ཏེ་གཤེགས་སུ་་་གསོལ་བར་བྱའོ་ཞེས་བྱ་བའི་དོན་འདི་བསྟན་པར་བྱ་བའི་ཕྱིར། དེ་ནས་བརྒྱ་བྱིན་ལ་སོགས་ལྷ། །ཞེས་བྱ་བ་ནས་རྗེས་སུ་ཡི་རང་ངོ་ཞེས་བྱ་བའི་གཞུང་གིས་བསྟན་ཏོ། །རྡོ་རྗེ་ཉི་མ་མེའི་བསྡུས་པའི་དོན་ཏོ། །དེ་ལྟར་བསྡུས་པའི་དོན་བཤད་ནས། :། ཤཱཀྱ་ཐུབ་པ་ཉིད་དང་གཞན་གྱི་དོན་ཕུན་སུམ་ཚོགས་པའི་་དཀྱིལ་འཁོར་དང་། སྦྱིན་སྲེག་དང་རས་རིས་ལ་སོགས་པ་སྣང་བ་རྣམས་ནི་བརྗོད་པར་བྱ་བ་ཡིན་ལ། །དེ་སྟོན་པའི་གཞུང་གི་ཆོ་ག་ནི་རྒྱུད་ཡིན་ཏེ་རྒྱུད་དང་བརྗོད་པར་བྱ་བ་གཉིས་ཐབས་དང་ཐབས་ལས་བྱུང་བའི་མཚན་ཉིད་དུ་འབྲེལ་པའོ༑། །རྒྱུད་ལ་བརྟེན་ནས་ཤཱཀྱ་ཐུབ་པ་ལ་མངོན་པར་འདོད་པ་བསྐྱེད་པ་ནི་དགོས་པའོ། །མངོན་པར་འདོད་པ་བསྐྱེད་པའི་མངོན་སུམ་དུ་བྱས་པ་ནི་དགོས་པའི་ཡང་དགོས་པའོ། །ཡང་འདིའི་མཚན་ཉིད་དང་གཞན་གྱི་དོན་ཕུན་སུམ་ཚོགས་པའི་གླེང་གཞི་དང་རྩ་བའི་རྒྱུད་དུ་གཏོགས་པའི་ཀུན་:རིག་གི་དཀྱིལ་འཁོར་དང་། །རྒྱུད་ཕྱི་མར་གཏོགས་པ་གཙུག་ཏོར་དགུ་ནས་བརྩམས་ཏེ་བདེ་བ་ཆེན་པོའི་དཀྱིལ་འཁོར་གྱི་བར་དང་། ་རྒྱུད་ཕྱི་མའི་ཕྱི་མ་ཁྲོ་བོའི་དཀྱིལ་འཁོར་ཏེ་བསྡུས་པའི་དོན་རྫོགས་སོ། །ཁ། །།་་་་་ཁ་ཚོ། ད་ཆ་ཤོད་ས་མ་ཤོག་འགོ་ཁྲི་ལ་བྱ། དེབ། འཁོར་ལོ་དང་། ཀ་ལ་ཀ་ཁ་ག་མ་ག་ར་ག་ལ་བ་དག་ལ་ཁོང་ཁ་དེ་ལ་འདི་ནི་ངན་སོང་ཐམས་ཅད་ཡོངས་སུ་སྦྱོང་བ་གཟི་བརྗིད་ཀྱི་རྒྱལ་་་:：པོའི་བརྟག་པ་ཞེས་བྱ་བའོ། །ལྔ་། དེ་ལ་ངན་སོང་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་སེམས་།
