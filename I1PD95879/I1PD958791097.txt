བསྟན་འགྱུར། རྒྱུད། ཇུདགེ་ལྷ་འཁོར་དང་བཅས་པའིའོ། །ཀུ་བེ་ར་ཡཀྵ་ནི་སྭཱ་ཧཱ། གནོད་སྦྱིན་འཁོར་བཅས་ཀྱི་སྔགས་སོ། །ཨོཾ་ཨག་ཡེ་ཀུ་ཙི་ཀུ་ཙི་སྭཱ་ཧཱ། དྲང་སྲོང་རྣམས་ཀྱིའོ། །ཨོ་:ནི་རྗེ་ཏུ་རཱཀྵ་ས་སྭཱ་ཧཱ། སྲིན་པོ་འཁོར་བཅས་ཀྱིའོ། །ཨོཾ་བཱ་ཡ་བེ་ཨ་:སཱུ་ར་ག་ཎེ་བྱཱསཱ་ཧཱ། རླུང་ལྷ་འཁོར་དང་བཅས་པའིའོ། །།ཨོ་རུ་དྲ་:ཨཀྵ་་བྷཱུ་ཨིཏ་ག་ཎེ་བྷྱ་:སྭཱ་ཧཱ། དབང་ལྡན་འཁོར་བཅས་ཀྱིའོ། །ཨོ་:པྲི་སྦི་བྱཱསཱ་ཧཱ༑སའི་ལྷ་མོའིའོ། །ཨོཾ་གྲ་ཧེ་པྲ་བྷཱ་ཛོ་ཏི་ས་ཡ་ནཀྵ་ོཏྲ་ནི་:རྣ་ད་ནི་སྭཱ་ཧཱ། གཟའ་དང་རྒྱུ་སྐར་སྤྱིའི་སྔགས་སོ། །ཨོཾ་མ་ཧོ་ར་ག་ག་ཎེ་བྷུསྭཱ་ཧཱ། ལྟོ་འཕྱེ་་རྣམས་ཀྱིའོ། །ཨོཾ་:ཧཱུྃ་བྷཎྜ་བྷཱུཏ་ཨུརྨ་དཱ་པ་སྨ་ར་ཨ་མ་ཨོཏྟཱ་ར་ག་ཎེ་བྷུསྭཱ་ཧཱ། འདྲེ་སྤྱིའིའོ། །ཨོ་ག་རུ་ཌ་ག་ཎ་བྱསྭཱ་ཧཱ། ནམ་མཁའ་ལྡིང་སྤྱིའིའོ། །ཨོ་་་་ཨངྒེ་སཱ་ཧཱ། མེ་ལྷའིའོ། །ཨོཾ་བ་རུ་:ཎཱ་ཡ་སྭཱ་ཧཱ། ཆུ་ལྷའིའོ། །ཨོཾ་བཱ་ཡ་བེ་སྭཱ་ཧཱ། རླུང་ལྷའིའོ། །ཨོ་:པ་ཊ་ལ་སཱ་ཧཱ། ས་བདག་གོ། །ཨོཾ་སརྦ་དེ་་བ་ཏ་སྭཱ་ཧཱ། ཁྱིམ་ལྷ་ལ་སོགས་པ་ཐམས་ཅད་ཀྱིའོ། །ཨོཾ་ཧྲི་སྭཱ་ཧཱ། །མཁའ་འགྲོ་མའིའོ། །དེ་དག་ནི་སྔགས་སུ་གསུངས་པའོ། །ཏིང་འཛིན་སྔགས་དང་ཕྱག་རྒྱར་ལྡན། །ཉིན་རེ་ཏིང་འཛིན་གསུམ་པོ་ནི། །བདག་དང་ལྷ་སྦྱོར་བསྒོམ་པར་བྱ། །དེ་ནས་བདག་ཉིད་དེར་གསལ་ནས། །རྒྱས་འདེབས་བཞི་དང་ལྡན་པ་ལ། །དཀྱིལ་འཁོར་སྒོ་ནི་དབྱེ་བྱས་ནས། །དཀྱིལ་འཁོར་ཡོངས་སུ་བསྐྱེད་པར་བྱ། །ཨོཾ་སརྦ་བིད་བཛྲ་ཙཀྲ་ཧཱུྃ། འཁོར་ལོ་དང་ལྡན་ཕྱག་རྒྱ་བྱ། །དཀྱིལ་འཁོར་བསྐྱེད་སྔགས་ཕྱག་རྒྱའོ། །དེ་ནས་གཙོ་བོར་བདག་སྦྱོར་བས། །དཔུང་པར་སེ་གོལ་གདབ་བྱས་ལ། །ཛཧཱུྃ་བཾ་ཧོས་སྤྱན་དྲང་ངོ༌། །དེ་ནས་ཉེ་རེག་:འཐོར་འཐུང་དང༌། །ཞབས་བསིལ་བ་ནི་དབུལ་བར་བྱ། །མཆོད་པ་རྣམ་པ་སྣ་ཚོགས་མཆོད། །དེ་རྗེས་བསྟོད་པ་བྱ་བ་ནི། །མཉམ་:མེད་མི་གཡོ་་་།
