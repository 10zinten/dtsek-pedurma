རྒྱུད། ཆུཨོ་ཀརྨ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཨི། ཨོཾ་བུདྡྷཱ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཨོ། །ཨོཾ་བཛྲ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྭཱ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་པདྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧཱི། ཨོ་ཀརྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཨ། ཨོཾ་བུདྡྷ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་པདྨ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཧཱི། ་ཨོ་བི་ཤྭ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཨ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ་ཧཱུཾ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་དྷི་ཨེཔ་ཏི་་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་བཛྲ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོ་ཨོ་ཨོ། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། :དྷཱཏཾ་་ཏཱཾ་ཏཱཾ། ཧཱིཧཱིཧཱི། ཨཨཨ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་དྷ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོ་བཛྲ་དྷ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་དྷ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ༄༅། །ཨོཾ་པདྨ་དྷ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཧི༄༅། ་ཨོ་ཀརྨ་དྷ་:ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཨཝེ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་:གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་གུ་རྒྱ་ཨ་བྷི་ཥིཉྩ་ཏྭཾ༑ཨོཾ་པདྨ་གུ་རྒྱ་ཨ་བྷི་ཥིཉྩ་ཧཱིི། ཨོཾ་ཀརྨ་གུ་རྒྱ་ཨ་བྷི་ཥིཉྩ་ཨ། ཨོ་་་པྲཛྙཱ་པ་ཡ་ས་མ་ཡོ་ག་ཨ་བྷི་ཥིཉྩ་ཨོཾ་ཨཱཿཧཱུྃ། ཞེས་པ་ལ་སོགས་པའི་སྔགས་འདི་་་༌་ལྟར་དབང་བཅུ་ཡོངས་སུ་རྫོགས་པར་བསྐུར་བར་གསུངས་ཏེ། ཚེ་དེ་ལ་འདི་ནི་་་་བཤད་པའི་རྒྱུད་ཡིན་པའི་ཕྱིར་དང་། ཁྱབ་པར་འབྱུང་བའི་རྡོ་རྗེའི་ཏིང་ངེ་་་་་འཛིན་གྱིས་དཀྱིལ་འཁོར་ཐམས་ཅད་དུ་ཁྱབ་པར་གསུངས་པའི་ཕྱིར་འདི་གསུངས་པའི་སྔགས་དང་རྣལ་འབྱོར་དང་བྱ་བ་ལ་སོགས་པ་རྣམས་ནི་རྒྱུད་ཀྱི་རྒྱལ་པོ་ཆེན་་པོ་འདིར་གསུངས་པའི་དཀྱིལ་འཁོར་བཅུ་གཉིས་པོ་ཐམས་ཅད་ལ་ཁྱབ་པར་བྱ་བ་ཉིད་དོ། །དེ་ལ་དང་པོ་ནི་བདག་ཉིད་རྣམ་པར་སྣང་མཛད་དང་། ངན་སོང་སྦྱོང་བ་དང་། རིན་ཆེན་ཏོག་དང་། ཤཱཀྱ་ཐུབ་པ་དང་། མེ་ཏོག་ཆེན་པོ་་་་རྒྱས་པ་ལ་སོགས་པའི་རྣལ་འབྱོར་དང་ལྡན་པར་བྱས་ལ། ་འསློབ་མའི་དཔྲལ་བ་་་དང༌། ལྐོག་མ་དང་། སྙིང་ག་དང་། ལྟེ་བ་དང་། །གསང་བའི་གནས་་་་་_།
