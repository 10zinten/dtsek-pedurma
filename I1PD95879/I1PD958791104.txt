རྒྱ་གར་གྱི་མཁན་པོ་སྨྲ་ཏི་ཛྙཱ་ན་ཀཱིརྟིས་བསྒྱུར་ཅིང་བཤད་ནས་གཏན་ལ་ཕབ་པའོ༑། :པར། ་༌་་་་་་་་་་པའི་:ཁམང་། ཀྱི་ང་གཉིས་ཤོད། འབྲེལ་བ་ན་བསྡུར་མཆན། ད་ད་པ་རྒྱུན་འབབ་ཤེདདེ་དག་སྐུ་ལ་འོ་བ་ལ། །བདེ་ལ་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»«སྣར་»ངན་སོང་་་་་ཡེ་མ་ཐམས་ཅད་ཡོངས་སུ་སྦྱོང་བའི་དཀྱིལ་འཁོར་ཆེན་པོའི་སྒྲུབ་ཐབས་སློབ་དཔོན་ཀུན་དགའ་སྙིང་པོས་མཛད་པ་་་་བཞུགས་སོ། །་«པེ་»«སྣར་»+བ། །«པེ་»«སྣར་»—རྐང་པ་གསུམ་ལས་མ་བྱུང་དཔྱད། «སྣར་»ཚེ༽གཞུགས༑འ«པེ་《སྣར་བཏུས། @«པེ་»«སྣར་»དེང་། @«པེ་»«སྣར་སྒྲུབ། །་/:རེ། «པེ་»«སྣར་པ། Q«པེ་»«སྣར་ཡིས། «པེ་»«སྣར་»གི། «པེ་»«སྣར་ཧྲི། «པེ་»དང་། «སྣར་»སྟེ། «པེ་»«སྣར་ཟློས། @«པེ་»«སྣར་»—རྡོ་རྗེ་རི་ཡི་མཐར་ཐུག་པར། །«སྣར་བཀྲོལ། «པེ་»«སྣར་སྒྲུབ། «པེ་»«སྣར་བྱས། «པེ་»«སྣར་བྱའོ། «པེ་»«སྣར་»ཀྱིས། «སྣར་»+པ། «པེ་»«སྣར་»འགྱུར། ར@«པེ་»«སྣར་རིམ། ་ཉི«པེ་»«སྣར་བསམ། «པེ་»«སྣར་བསྒོམས། («པེ་»《སྣར་—དཀྱིལ་འཁོར་འདི་ནི་ཐམས་ཅད་དུ། །མི་(«པེ་»«སྣར་»གཙོ། Q《སྣར་»—ཆེན། «པེ་»«སྣར་ཤཱཀྱ། «པེ་《སྣར་བསྐྱོད། ཉི«སྣར་བྱིན། «སྣར་བར། &<«པེ་《སྣར་ན། །«པེ་»《《སྣར་བསལ། ཁ་བློ་ཕྱིར་དྲན་བྱེད། ::མར་མེ། ་་་་་་་་་་། ཀྱི«སྣར་དེ། «པེ་»«སྣར་བསྐོང་། །འ«པེ་»ཤིང་། ::«པེ་»སྣར་»འཛིན། (ི«སྣར་+པ། །།@«པེ་»«སྣར་སྔ། @«པེ་»«སྣར་»བར། «པེ་»«སྣར་»ཞམ། «པེ་»«སྣར་»བསགས། །E(@«པེ་»«སྣར་»འཇིག་?«པེ་»«སྣར་པར། («པེ་»«སྣར་སྡིགས། «པེ་»«སྣར་»བཞིན། ་་་།  ། །«པེ་《སྣར་གྲུབ། «པེ་»«སྣར་»འཁྱིལ་བ། ་@«པེ་»«སྣར་»སྲིད་པ་འཇོམས་མཛད་བེ་ཅོན་་་ང་སྔོ། »«སྣར་པདྨོ། ་«སྣར་»སོག་ལ་སོག་(«པེ་»«སྣར་སོགས། ཁོ་ལ་བདེ་ཀྱི«སྣར་»ཅན། («པེ་»«སྣར་»ལ། «པེ་»བིབྷུ་དང་། །བྷིང་གི་རཱི་ཊི། 《སྣར་»བིབྷུ་དང༌། །བྷིང་། རང་གི་རཱི་ཊི། «སྣར་བཀུར། །(《«སྣར་»ཀ་ར་མ་ཀོ། «པེ་»«སྣར་ཨུ། ༄༅«པེ་»«སྣར་གདན། །(ི«པེ་»«སྣར་ན་ཆུང་གནས་པ་ཡི། «པེ་»«སྣར་»འགུ། «སྣར་གཡོག་«པེ་»«སྣར་»ཀྱི། ལངྒའི། «པེ་»«སྣར་»བུ་སྣ། m@«སྣར་»སྟོད། «པེ་»«སྣར་རྩེག་«པེ་»«སྣར་»སྣ་་་་ཚོགས༑«པེ་»«སྣར་དང་། (གྱེ«པེ་»«སྣར་བཟང་། འགྱེ«པེ་《སྣར་འཇམ། ་འ«སྣར་»ཀབནདྲབཟའ༑སྟེ«པེ་»«སྣར་ནབས། ་«པེ་《སྣར་མགོ། ལགན་་་་་་་:དཏེ། །
