རྒྱུད། ་་དགྲ་སྟ་དང༌། །རལ་གྲི་ཤིན་ཏུ་འབར་བ་བྲི། །དེ་ཡི་ཕྱི་རོལ་ཀེང་རུས་ཏེ། །བདེ་གཤེགས་ལ་སོགས་ཕྱག་རྒྱས་མཚན། །ཞེས་གསུངས་པ་སྟེ། །དེ་ལ་ཐབ་་ཁུང་ནང་དུ་རྡོ་རྗེ་རྩེ་དགུ་པ་ཁ་དོག་ནག་པོ་གཅིག་བྲི་བར་བྱའོ། །ཎ་། དེའི་ཟུར་་་གསུམ་དུ་ནི་མཚོན་རྩེ་གསུམ་པ་གསུམ་བྲིའོ། །དེའི་ལོགས་གསུམ་ལ་ནི་རྡོ་རྗེ་རྒྱ་་གྲམ་གསུམ་བྲིའོ། །དེའི་མུ་ཁྱུད་:ནང་མ་ལ་ནི་དབྱུག་པ་དང་། ལ་:རྡོ་རྗེ་རྩེ་གསུམ་པ་དང་། དགྲ་སྟ་རྣམས་སོ། །དེའི་མུ་ཁྱུད་ཀྱི་སྟེང་དུ་ནི་རལ་གྲི་དང་མེ་འོད་ནག་པོ་བྲི་བ་དང་། དེའི་ཕྱི་རོལ་དུ་ནི་ཀེང་རུས་ཀྱི་སྟེང་དུ་རིམ་པ་གསུམ་་མམ་བཞི་:ལ་བདེ་བར་གཤེགས་པ་ལ་སོགས་པའི་ཕྱག་མཚན་རྣམས་རིམ་པ་དང་པོ་ལ་ལྔ་དང་། གཉིས་པ་ལ་བཅུ་དྲུག་ལ་སོགས་པ་བྲི་བར་བྱའོ་ཞེས་གསུངས་སོ། །མེ་ལྷ་འདི་ལྟ་བུ་བྱའོ་ཞེས་བསྟན་པའི་ཕྱིར། །དུད་སྲིན་མདོག་འདྲ་དྲང་སྲོང་་མཆོག། མི་རོའི་སྟེང་བཞུགས་རྩེ་གསུམ་དང་། །གཟར་བུ་ཕྲེང་བ་རིམ་བར་་་ལྡན། །དེ་ལྟའི་མེ་ལྷ་སྤྱན་དྲངས་ནས། །ལྷོར་བལྟས་ཁྲོས་པའི་སེམས་ཀྱིས་་་་བསྲེག། ཅེས་གསུངས་ཏེ། དེ་ལ་ལས་ཀྱི་མེ་ལྷ་ཁ་དོག་ནག་པོ་མི་རོ་ལ་ཞོན་པ་ཕྱག་བཞི་པ་རྩེ་གསུམ་པ་དང་། །གཟར་བུ་དང་རིལ་བ་ཐོགས་པ་སྤྱན་:དྲང་བ་གཞན་དུ་བསྟན་པ་ལྟར་ཡེ་ཤེས་ཀྱི་མེ་ལྷ་སྤྱན་དྲངས་པ་བཅུག་སྟེ། ཁྲོས་པ་དང་། བཅས་པས་དེ་ཉིད་ཀྱི་ྋརྣལ་འབྱོར་གྱིས་བསྲེག་པར་བྱའོ། །འོ། དེ་ཉིད་ཀྱི་རྒྱུ་འདི་ལྟ་བུས་བྱའོ་ཞེས་བསྟན་པའི་ཕྱིར། ཤིན་ཏུ་གདུག་ཅིང་ཚེར་མར་བཅས། །བསྲེག་ཤིང་ཡམ་ཤིང་བྱས་ནས་ཀྱང་། །ཤ་དང་ཁྲག་དང་ལྕགས་ཕྱེ་དང་། །རུས་ཕྱེ་དུག་དང་སྐེ་ཚེ་དང་། །དྲི་ཆེན་དྲི་ཆུ་ལ་སོགས་པའི། །རྫས་རྣམས་ཀུན་ལ་དུག་གིས་ནི། །བསྒོས་པས་ལན་གྲངས་གསུམ་བསྲེགས་ནས། །དེ་ཡི་ལོངས་སྤྱོད་འབྱོར་པ་དང་། །གཉེན་དང་འཁོར་དང་དྲན་རྣམས་དང༌། །བུ་་།
