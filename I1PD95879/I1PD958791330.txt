རྣམ་རྒྱལ་བྲི། །ཡང་ན་དཀྱིལ་འཁོར་ཚུལ་བྲིས་ལ། ་། མེ་ཏོག་ལ་སོགས་མཆོད་པ་བྱ། །སྤྱན་སྔར་བསྙེན་པ་འབུམ་བྱས་ན། །ལས་རྣམས་ཐམས་ཅད་བྱེད་པར་འགྱུར། །རྡུལ་:ཚོན་ཆུར་དོར་དབང་ཡོན་དབུལ། །ཡེ་ཤེས་གཤེགས་ལ་དམ་ཚིག་བསྡུ། །གཙུག་ཏོར་དགུ་པའི་ཆོ་ག་འདི། །སངས་རྒྱས་གསང་བས་ཕྱི་་རབས་དོན། །ངན་སོང་སྦྱོང་བའི་རྒྱུད་ལས་མཛད། །འགྲོ་ཀུན་ངན་སོང་ལས་གྲོལ་ཤོག། སློབ་དཔོན་སངས་རྒྱས་གསང་བས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་སངས་རྒྱས་གསང་བ་དང་། མཉྫ་བྲི་བརྨ་དང་། བྲན་ཀ་མུ་ཏིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།པ་བསྡུར་མཆན། ::ང་འ་ར་བ་པ་དང་། ནག་པ། ས་ར་ཡེ་བབས་ཀྱི་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ངན་སོང་་་་::ཀུདུ་ཐམས་ཅད་ཡོངས་སུ་སྦྱོང་བའི་དཀྱིལ་འཁོར་གྱི་ཆོ་གའི་རིམ་པ་ཞེས་བྱ་བ་སློབ་དཔོན་སངས་རྒྱས་གསང་བས་་་་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་བསྒྲུབ། «པེ་»«སྣར་»གོ་ཆ་བགོ། །(«པེ་»«སྣར་བྱའོ། «སྣར་»—ཧ། «པེ་»«སྣར་བིགྷཱན། «པེ་»«སྣར་བགེགས། །ག«སྣར་»བཅིང་། «པེ་»«སྣར་སྤྱན་གྱིས། «པེ་»«སྣར་བཅིང་། «པེ་»«སྣར་ཧུ་ལུ་ཧུ། ན་དུ་བ་བ«པེ་»སྣར་»བཅིང་སྟེ། («པེ་»«སྣར་མཐེབ། («པེ་»«སྣར་པཱ། (ཉེ«པེ་»«སྣར་རཊ། ཤ::«པེ་»«སྣར་བཅིང་བ། «པེ་»«སྣར་»ཀཱ། «པེ་»«སྣར་བཅིང་། («པེ་»«སྣར་+རི། འབྱེ«པེ་»«སྣར་»བརླབ་པ་བྱའོ། ་ཀྱི«པེ་»《སྣར་གསག(ཀྱེ«པེ་»«སྣར་བཤགས། ེ«པེ་»::ཀྱིང《སྣར་»གཟུང་། །ཞཀྱི«པེ་»«སྣར་བཅས། །འབྱེ«པེ་»«སྣར་»མའོ། །(སྐྱེད«པེ་»«སྣར་»ཀྲྀ་ཧྲིཨ། Q«པེ་»སྣར་ཨཉྩ། («པེ་《སྣར་»། །«པེ་»«སྣར་པས། :«པེ་»«སྣར་»བཞུག་«པེ་《སྣར་བསྐུར། །«པེ་པ་དུགས། «སྣར་པོ་གདུགས། །ན་ཀྱི:(«པེ་»«སྣར་བསྐལ་བཟང་། སྤྱི་ལོ་་«པེ་»«སྣར་»དྷི་ཏིཥྛ། ་«པེ་»«སྣར་»ཀྱིས། ན«པེ་»«སྣར་»བྲུ་ལས། ་«པེ་»«སྣར་»ལས། —།
