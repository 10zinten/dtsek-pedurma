བསྟན་འགྱུར། རྒྱུད། གི་ལེགས་པར་ཞུས་ཤིང་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།།སླར་ཡང་དཔལ་ལྡན་ཤོང་སྟོན་རྡོ་རྗེ་རྒྱལ་མཚན་གྱི་ལེགས་བཤད་ལ་་་་སོགས་པའི་མཐུ་ལས། བརྡ་སྤྲོད་པའི་ཚུལ་རིག་པའི་དཔང་ལོ་ཙཱ་བ་དཔལ་ལྡན་བློ་གྲོས་བརྟན་པས་བྱང་ཆུབ་སེམས་དཔའི་འགྲེལ་པ་སྐོར་གསུམ་ཀྱི་ཚུལ་ལ་ཤིན་ཏུ་དད་ཅིང་བློའི་སྣང་བ་རྒྱས་པའི་དགེ་བའི་བཤེས་གཉེན་ར་ལུང་པ་ཆོས་གྲགས་དཔལ་་་༌་བཟང་པོས། སློབ་དཔོན་ཞི་བ་འཚོའི་ཞབས་དཔོན་སློབ་ཀྱིས་མཛད་པའི་དབུ་མའི་གཞུང་ལུགས་ཆེན་པོ་དེ་ཁོ་ན་ཉིད་བསྡུས་པ་རྩ་འགྲེལ་གྱི་གླེགས་བམ་བྲིས་ཏེ་ཡོན་དུ་་་གནང་ནས། །ཡང་ཡང་དུ་བསྐུལ་བའི་ངོར་ལེགས་པར་བཅོས་ཏེ་བསྒྱུར་ཅིང་ཞུས་ནས་གཏན་ལ་ཕབ་པའོ། །།།།།་པ། ་་ལོ་ཀྱི། འདི་གའིਗੁਰਮੇਲਨਣਲਈਇਕਸਧਾਰਨਨੂੰཌ་ཌ་ཀ་ཊི་ལ་ཏུg:ཁ་བ་ལ་ཁ། པེང་། བ་བ་ན་མ་འོ་་ཏུ་ན་བི་ཡུལ་གྱི་བརྩོན་གྱི་ལོའི་ཁྲོ་ཁྱེད་ལ་ཀོཤེས་ཀྱི་ཀ་བ་ཁང་ཚོ་བླ་མ་བྱང་ཁོ་དང་བཀའ༌དང་བོ་ངོད་། p?ppནག་གཞུང་ལ་ཁོ་ཚོ་ལ་《ཉིལ་ཏེ། ཁོizz་། ཀ་ཌི། པ་དེ་ཕྱི་ལོ་འི་དང་ཚོ་ཚོ་:ཚ་བ་མདའི་ལན་དང་རེ་བ། འདམ་བག་མེད་དབང་བོ་རྒྱུད་དག་ལ་དེ་དང་། ད་དུ་རྒྱུད་དུ་གསུངས་པ་དབང་བསྐུར་དང་བ་སོལ་བ། འདས་ཁེ་ལྔ་དོན་ལ་གོ་པ་ལེན་ཁོ་ལ་ཀོབ་ག་བ་བོད་ཀྱཚོ་ཊ་རྗེ་དཔལ་མེད་དེ་ལོག་པ་བྱེད་ཀྱི་ཚུལ་་བཔལོ་འཁ་ཕྱོགས་ཚེ་ད་བར་_།
