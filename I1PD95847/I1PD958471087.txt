བསྟན་འགྱུར། རྒྱུད། །སྭཱ་ཧཱ། ཨི་དཾ་བ་ལི་བྷུཉྫ་ཛི་གྷིཕུལླ་དྷུ་པ་མཾ་ས་བི་གྷ་ཨམྦ་ཀཱ་ཛྫ་སཔྤ་ས་དྷ་ཁནྟི་ཁུ་ཎི་ཕེ་ཌ་གཱ་ཏ། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱ་ཎཾ་ཨཱ་དྱ་ནུཏྤནྣ་ཏྭ་ཏ་ཨོཾ་ཨཱཋཧཱུཾ་ཕཊ་སྭཱ་ཧཱ༑ཨོཾ་བཛྲ་ས་ཏྭ་ས་མ་ཡ་མ་ནུ་པཱ་ལ་ཡ། ཌཱ་དྷོ་མེ་བྷ་ཝ་སུ་པོཔྱོ་མེ་བྷ་བ། སུ་ཏོ་བྱོ་མེ་བྷ་བ། ཤཱ་ཝིཤཱ་ཏོ་མེ་ཅིབྷ་ཝ་ཨ་ནུ་རཀྟ་མེ་བྷ་བ། ཀྲྀ་ད་ཡོ་མེ་ཨ་དྷི་ཏིཥྛ་སརྦ་སིདྡྷི་མྨེ་པྲ་ཡ་ཙྪ། སརྦ་ཀཱརྨེ་ཥཱ། མེ་ཙིཏྟཾ་ཤེ་ཡཿཀུ་རུ་ཧཱུཾ་ཧཱུཾ། ཧ་ཧ་ཧ་ཧ་ཧོ། །བྷ་ག་ཝཱན། སརྦ་ཏ་ཐཱ་ག་ཏ་མཱ་མེ་མུཉྩ། བཛཱི་བྷ་བ་ས་མ་ཡ་སཏྭ་ཨཱམུཞེས་སོ། །དཔལ་ཀྱེའི་རྡོ་རྗེའི་རྒྱུད་ལ་རྡོ་རྗེའི་ཚིག་གི་སྙིང་པོ་བསྡུས་པའི་དཀའ་འགྲེལ་དུ་རྒྱུད་་་་ཐམས་ཅད་ཀྱི་ཕྱག་རྒྱ་བསྡུས་པའི་དོན་ཞེས་བྱ་བའི་ལེའུ་སྟེ་བཞི་པའོ། །།།།ཞུ་ཁང་ཚོ་ཁའི་མེ་་་་་བད་ནི་བརྟག་པ་གཉིས་པ་ལ་ཧེ་རུ་ཀ་མངོན་པར་འབྱུང་བ་ཞེས་བྱ་བའི་ལེའུ་ལྔ་པ་གསུངས་པ། ལཕྱག་ནི་བཅུ་དྲུག་ཞལ་བརྒྱད་པ། :་། ཞབས་བཞི་པ་དང་་་་འཇིགས་པའོ། །དཔའ་བོ་ཐོད་པའི་ཕྲེང་བ་ཅན། །བདག་མེད་མ་ཡིས་མགྲིན་པར་འཁྱུད། །ཕྱག་རྒྱ་ལྔ་ནི་འཛིན་ལྷ་ལ། །བདག་མེད་རང་ཉིད་ཀྱིས་ཞུས་པ། །བཅོ་ལྔས་ཡོངས་སུ་བསྐོར་བ་ཡི། །བདག་གི་འཁོར་ལོ་ཁྱོད་ཀྱིས་གསུངས། ཡང་བས། །ཁྱོད་ཀྱི་འཁོར་ལོ་ཇི་ལྟར་ལགས། །གཙོ་བོ་བདག་གིས་སྔར་མ་འཚལ། །བདག་མེད་ཙུམ་པ་ན་མཛད་ནས། །རྡོ་རྗེ་ཐོད་པར་སྩལ་ནས་ནི། །སྟ་ན་དག་ནི་་་་མཉེས་བྱས་ནས། །།དཀྱིལ་འཁོར་ཡང་དག་རབ་གསལ་མཛད། །འཁོར་ལོ་ཇི་་ལྟར་སྔར་གསུངས་བཞིན། །ཞེས་པ་ལ་འཁོར་ལོ་ནི་དཀྱིལ་འཁོར་གྱི་འཁོར་ལོ་སྟེ། གྲུ་བཞི་པ་ལ་སྒོ་བཞི་པ། །རྟ་བབས་བཞི་ཡིས་རྣམ་པར་བརྒྱན། །དྲ་བ་དྲ་བ་ཕྱེད་པར་ལྡན། །ཀ་བ་བརྒྱད་ཀྱིས་ཉེ་བར་མཛེས། །མེ་ཏོག་ཕྲེང་དར་ལྡ་ལྡི་ལྡན། །ཞེས་པ་ལ་སོགས་པ་དྲང་བའི་དོན་ཏོ། །ད་ནི་ངེས་པའི་དོན་གསལ་བར་བྱ་སྟེ། ཁ་།
