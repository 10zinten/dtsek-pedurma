ཚོ་ལ་བསྟན་འགྱུར། །རྒྱུད། ཨཛིཏིappགལྔ་སྨྲ་བར་འཁོད་པར་གྱུར་ཏོ་ཞེས་བྱ་བ་ནི་བཀའ་སྩལ་པའི་དོན་ལ་གནས་པར་འགྱུར་བའོ༑། ལེའུ་བཅོ་ལྔ་པའོ། །གལ་ཏོ་། །ཨ་ལ་ཁ་བའི་ཚེ་ཆོ་། ཀོ་ཁ་ང་ཚོ་ཁོ་པོ་གང་གང་དོན་དང་། ཆོ་ཏ་ཟེ་བླ་མ་དེ་ག་ད་བ་ས་བོན་ལ་གཡུར་ཏུམ་ཀ་ཏ་པ་ཞེས་ཁ་ཏ་ལ་ཁ་ཏུ་ལ། ད་ནི་ལེའུ་བཞི་པར་ཞུས་པའི་དཀྱིལ་འཁོར་བཤད་པར་བྱའོ་ཞེས་གསུངས་པ་ལ། ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ནི་དེ་ཉིད་དུ་བཤད་ནས་སྐུ་དང་གསུང་གི་དཀྱིལ་འཁོར་བསྟན་པའི་ཕྱིར། དེ་ནས་དེ་བཞིན་གཤེགས་པ་རྣམས་ཡང་འདུས་ནས་་་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། དེ་ལ་རྡོ་རྗེ་ཚིག་གིས་གསོལ་བ་བཏབ་ནས་་་ཞེས་བྱ་བ་ལ། ཇི་སྐད་དུ་གསོལ་བ་བཏབ་པ་ནི་ལེའུ་བཞི་པ་ཉིད་ལ་ལྟོས་པ་ཡིན་ནོ། །ཞ་། དེ་ནས་སྐུ་ཡི་དཀྱིལ་འཁོར་མཆོག་་། ཐུགས་ཀྱི་རྡོ་རྗེ་རབ་འདྲ་བ། །ཞེས་བྱ་བ་ནི་ཇི་སྐད་བསྟན་པའི་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་དང་བྱེ་བྲག་མེད་དོ་་་་་་་ཞེས་:བྱ་བའི་དོན་ཏོ། །ཁ་ཅིག་ན་རེ་འདི་ཉིད་སེམས་ཀྱིས་བསྒོམས་པ་དང་རྡུལ་ཚོན་དུ་བྱེ་བྲག་མེད་དོ་ཞེས་ཟེར་རོ། །།ཁྲུ་ནི་བཅུ་དྲུག་ཚད་དུ་བྱ། །ཞེས་བྱ་བ་ནི་འཁོར་ཡུག་ཏུའོ། །་་། སྔགས་རྣམས་ཀྱི་ནི་གསང་བའི་མཆོགརྫོགས་ནི། ཕྱག་ན་རྡོ་་་རྗེའི་གནས་སུ་བྱ། །ཞེས་བྱ་བ་ནི་ཇི་ལྟར་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ལ་ཕྱག་ན་རྡོ་རྗེ་བྲིས་པ་དེའི་གནས་སུ་འདིར་རྣམ་པར་སྣང་མཛད་བྲི་བའོ། །མི་བསྐྱོད་པ་ཡང་བྲི་བར་བྱ། །ཞེས་བྱ་བ་ནི་འདིར་ཤར་དུ་སྟེ། ་་་ཡང་ྋགི་སྒྲ་ནི་གཞན་བསྡུ་བ་་་་ཡིན་གྱི་དབུས་ཀྱི་གནས་ལ་བྲི་བ་ནི་མ་ཡིན་ནོ། ་། གཞན་ནི་ཇི་ལྟ་བ་བཞིན་དུ་བྲི་བར་བྱའོ། །དེ་ལ་མཆོད་པ་ནི། རྡོ་རྗེ་གསང་བ་ལས་བྱུང་བའི། ཎ། མཆོད་པ་སྔགས་ཤེས་པ་ཡིས་བྱ། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། །འདི་ནི་གསང་བ་གཉིས་སུ་མེད་པའི་སྦྱོར་བའི་མཆོད་པའོ། ༈། འི། བཤང་དང་གཅི་བ་ལ་སོགས་པ། ཚི། །བྱེ་བྲག་ཏུ་ནི་ངེས་པར་འབུལ། སྐྱེ། །ཞེས་བྱ་བ་ནི་བདུད་རྩི་ལྔའི་མཆོད་པ་སྟེ། །།
