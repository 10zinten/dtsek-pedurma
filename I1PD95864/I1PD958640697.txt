རྒྱུད། རྩི་༌། ཆཝཾ་བྲལ་བ་དང་། ཆགས་པ་བར་མའི་མཚན་ཉིད་ཀྱི་རང་བཞིན་བརྒྱ་དྲུག་ཅུའི་་་་་ཚོགས་ཏེ། །དེ་ལས་ཕ་རོལ་སྒྲོལ་བ་ནི་ཤེས་རབ་ཀྱི་དབང་དང་། རྫོགས་དེ་ཡང་དེ་་་བཞིན་བཞི་པའོ། །ཞེས་བྱ་བ་སྟེ། དེ་དག་ལས་རང་བཞིན་གྱིས་འོད་གསལ་བའི་དེ་ཁོ་ན་ཉིད་འབྱུང་བའི་ཕྱིར་རོ། །ཡང་ན་ཆགས་པ་ལ་སོགས་པ་དང་འོད་གསལ་བ་ཕན་ཚུན་འགལ་བའམ། དེ་རྣམས་དེའི་ངོ་བོ་ཡིན་ནོ་ཞེས་དགོངས་པའོ། །ད་ནི་ཆོས་སྦྱིན་པར་མི་བྱ་བ་དང༌། །ཟླ་སྦྱིན་པར་བྱ་བའི་སློབ་མ་གང་ཡིན་པ་དེ་་་དག་གི་སྐྱོན་དང་ཡོན་ཏན་དཔྱད་པར་བྱ་བའི་ཕྱིར། རིགས་གང་ཞིག་ང་རྒྱལ་གྱིས་་་་བརྩེགས་ཤིང་། ཌ་། ཡང་དག་རིམ་པ་སུན་འབྱིན་དང༌། །དེ་ཉིད་ཆོས་ལ་བརྙས་བཅས་པ། །དེ་ལ་དེ་ཉིད་བསྟན་མི་བྱ། །ཞེས་བྱ་བ་གསུངས་ཏེ། །བདག་་་་གིས་རྣམ་པ་མང་པོར་ཡང་དང་ཡང་དུ་དབང་བསྐུར་བ་ཐོབ་པ་ཡིན་ནོ་ཞེས་གང་ཞིག་ང་རྒྱལ་དང་ལྡན་པ་དེ་བཞིན་དུ་ཡང་དག་རིམ་པ་ནི་རྡོ་རྗེ་བཟླས་པ་ལ་སོགས་པ་སྟེ་དེ་ལ་སུན་འབྱིན་པ་དང་། དེ་བཞིན་དུ་དེ་ཉིད་ཆོས་ནི་མན་ངག་གཙོ་བོར་གྱུར་པའི་རྒྱུ་རྣམས་བདག་རང་:གིས་རྟོགས་ལ་བདག་ལ་བླ་མ་ཅུང་ཟད་ཀྱང་་་་་དགོས་པ་མ་ཡིན་ནོ་ཞེས་དེ་ལ་ལོ་དང་ལྡན་པ་དང༌། སློབ་མ་སྐྱོན་ཅན་གང་ཡིན་པ་དེ་ལ་བདག་ལ་བྱིན་གྱིས་བརླབ་པའི་རིམ་པ་བླ་མས་བསྟན་པར་མི་བྱའོ། །དེ་ནི་སྒྲ་ཇི་བཞིན་པའོ། །།ཡོན་ཏན་ཅན་བཤད་པར་བྱ་བའི་ཕྱིར། །བླ་མ་ལ་་གུས་བདེན་པར་སྨྲ་:བ་དང་། །རྣམ་པར་མཁས་ཤིང་རྒྱུད་གཅིག་ལ། །།།དམ་་ཚིག་སྤྱོད་པ་བསྲུང་བ་དག། རིམ་པ་དེ་ལ་བསྟན་པར་བྱ། ཚཊ། ཞེས་བྱ་བ་་་་་གསུངས་ཏེ། ཎསློབ་མ་འདི་ལྟ་བུ་གང་ཡིན་པ་དེ་ལ་རིམ་པ་བསྟན་པར་བྱའོ་ཞེས་་ཚིག་གི་འབྲེལ་པའོ། །དེ་གང་ཡིན་ཞེ་ན། གང་ཞིག་བདེན་པར་སྨྲ་བ་དང་། །རྡོ་རྗེ་སློབ་དཔོན་ལ་དད་པ་ཅན་དང་། དེ་བཞིན་དུ་རྒྱུད་གཅིག་པ་ནི་ལན་ཅིག་ལ་དད།
