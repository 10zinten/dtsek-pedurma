འོ་མས་སོ། །རྒྱས་པ་ལ་ཆུ་སེར་པོས་སོ། །དབང་ལ་ཡང་འོ་མས་སོ། །གསད་པ་ལ་ཁྲག་གིས་སོ། །དེ་ནས་ཐབ་ཁུང་གི་ཐལ་བ་དང་སོལ་བ་ལས་ཐམས་ཅད་་་པའི་བུམ་པར་བླུགས་ཏེ་ཆུ་བོ་ལ་སོགས་པར་དོར་རོ། །།འདིར་ནི་ལྷ་མཉེས་པར་་བྱ་བའི་མཚན་ཉིད་བསྟན་པའོ། །།།དང་སྦྱིན་སྲེག་ཆོ་ག་བྱས་པ་འདི་ཡིས། །དགེ་བ་དྲི་མེད་ཐོབ་པ་གང་ཡིན། །དེས་ནི་མྱུར་བར་སེམས་ཅན་ཐམས་ཅད། །བདེ་ཆེན་རྡོ་རྗེ་སེམས་དཔར་གྱུར་ཅིག། ཡེ་ཤེས་ཀྱི་མེ་སྦར་བ་ཞེས་བྱ་བའི་སྦྱིན་སྲེག་གི་ཆོ་ག་རྫོགས་སོ། །།།པཎྜི་ཏ་ཆེན་པོ་ས་མནྟ་བཱིས་མཛད་པའོ། །།།པཎྜི་ཏ་དེ་ཉིད་དང་ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་རབ་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ཁལ་ཕབ་པའོ། །།་་་་་་་་་་། དེ་བས་ད་ན་ལ་བཀྱི། །དྲ་བ་མང་། །aདགའ་ར་སྐྱེ/pབློ་ཀླུ་གསར་པ་་། pབ་དང་། དྲ་བ་ལས་ཀྱང་ཀཁར་་་། བསྡུར་མཆན། :::གན་»@་དང་། དྲལ་བ་ལ་བྱེད་་་ང་མ་ལ་སྤྱོད་པ་ལ། མ་བསྒྲགས། །ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཡེ་ཤེས་ཀྱི་མེ་»སྦྱོར་བ་ཞེས་བྱ་བའི་སྦྱིན་སྲེག་གི་ཆོ་ག་བཞུགས་སོ། །«པེ་《སྣར་+ནི། ཚེ«པེ་»«སྣར་»སྦྱོར། ང«པེ་»«སྣར་»ཆེན། ཚེ«པེ་»«སྣར་»ཚངས། «པེ་»«སྣར་»སྦྱོར། @«པེ་»«སྣར་»གི། «པེ་»«སྣར་བསྣུམ། ལྷ@«པེ་»«སྣར་པ། @«པེ་»«སྣར་»ལས། «པེ་»«སྣར་བསྣུམ། པ་འདྲ་བ་ལ་མྱོ«པེ་»«སྣར་+པ། «པེ་»«སྣར་+ལ། །ཚེ«པེ་»«སྣར་—བརྒྱན་ནོ། «པེ་《སྣར་»འཇམ། ་«པེ་»«སྣར་»རྗེ། ཨེ«པེ་»«སྣར་—མོ། <འགྱེ«པེ་»«སྣར་ནག་པོའི། (ལེ«པེ་»《་ཁ་《སྣར་རྒྱུ། རྩེ«པེ་»སྣར་+དང་པོ། ལོ«པེ་»«སྣར་»ནག་པོའམ་སོ་བའམ། ཚེ«པེ་»«སྣར་»བ་བ་ལ་ཀྱི་ལྐོགས་པས། ཚེ«པེ་»«སྣར་བཅས་ལ། ཁ་བ་《ཀྱི། །ར། «པེ་»«སྣར་གཅད། «པེ་»«སྣར་+པ། /Q«པེ་»«སྣར་དང་། «པེ་»《སྣར་»དུར། འིགྱི«པེ་《སྣར་»རྐྱེད་པ། ༄༅«པེ་»«སྣར་»འབི་ལོའི། ཨ«པེ་»«སྣར་»ཀ་ར་ནི། །ཉི«པེ་»«སྣར་»NHIEN«««TAIĐÁPTHITHPTPhan།
