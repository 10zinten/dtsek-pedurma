རབ་ཏུ་བསྟོད་པའི་ཡོན་ཏན་ཆ། །བདག་གིས་བསོད་ནམས་བཤད་མི་ཤེས། །བསོད་ནམས་དེ་ཡི་སྟོབས་ཀྱིས་ནི། །འཇིག་རྟེན་པ་ཀུན་དོན་འགྲུབ་ནས། །གྲོལ་བའི་བདེ་བ་མྱོང་བར་ཤོག། སྤྱན་རས་གཟིགས་ཀྱི་ཞབས་ཀྱི་མཐིལ། །དེ་ཡི་མཚན་མ་བཀྲ་ཤིས་ཀྱིས། །དེ་བཀང་ཡོད་པ་དེ་ལ་ནི། །བདེ་གཤེགས་སྲས་ཀྱི་ས་རིས་ཡིན། །བདེ་ནས་བདེ་རུ་འགྲོ་བ་ཀུན། །མྱུར་བར་སོང་ནས་བདེ་་་བར་ཤོག། མེ་ཏོག་ཕྲེང་བ་འཛིན་པའི་བསྟོད་པ་སློབ་དཔོན་ཐམས་ཅད་མཁྱེན་པའི་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །རི་དང་། །སྐུ་པཎྜི་ཏ་ཟླ་བ་གཞོན་ནུས་རང་འགྱུར་མཛད་པའོ། །།།ཚོ《བཙྪ《་། ཎ་བོ་དཔ《་། pན་ཚོམས། །དཔའ་བོ་བ། །ཕྱི་པ། ༽བསྡར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་མ་སྒྲོལ་མའི་མེ་ཏོག་ཕྲེང་བ་འཛིན་པའི་བསྟོད་པ་བཞུགས་སོ། །«པེ་»«སྣར་—རྗེ་བཙུན་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །@«ཅོ་པ། «པེ་»«སྣར་སྐྱབས་སམ། གྱི«པེ་»«སྣར་»སར། «པེ་»«སྣར་བརྒལ། «སྣར་ནུབ། <«པེ་»«སྣར་གྱི། «པེ་བརླངས། «ཅོ་»བ་ལངས། «པེ་»«སྣར་»ཀྱི། ཚེ«ཅོ་»ལ། «པེ་»«སྣར་ཀྱི། །@«པེ་»«སྣར་པོ་དུས་པའི་ས། «པེ་»«སྣར་སྤང་བར། «པེ་»«སྣར་»ཀྱི། &«པེ་»སྣར་»གི། མུ«པེ་»«སྣར་གཙུགས། «ཅོ་»གཙུབ། «པེ་»«སྣར་གྲགས། «པེ་»«སྣར་»མདངས། «ཅོ་བེམ། «པེ་»«སྣར་»སྐོར། དེ་བས་«སྣར་»«ཅོ་»བཏུད། «པེ་»«སྣར་»འོད། པེ་»«སྣར་ཕར་ལོག་བརྒལ། «པེ་»«སྣར་»པ། «པེ་»སྣར་གསད། «པེ་»ནག་གསེབ། «ཅོ་ནགས་ཁྲོད། ༄༅«པེ་»སྣར་»དང་གཟུང་། O«པེ་»«སྣར་སྐུ། «པེ་»«སྣར་པ། ཚེ«པེ་»སྣར་བས། ཀྱེ«པེ་»«སྣར་»ཁྱེད། ལེ«པེ་»«སྣར་ཕུང་། «པེ་»«སྣར་»གདེངས། ཁྱེ«པེ་»«སྣར་བསྐྱིགས། ཚེ«པེ་《སྣར་»པར། །«པེ་»«སྣར་»མགུལ། ཚེ«པེ་»«སྣར་དེས། :«པེ་»«སྣར་བསྐྱིགས། «ཙོ་»རྐེ། «པེ་»།
