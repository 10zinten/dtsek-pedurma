བསྟན་འགྱུར། རྒྱུད། ༼གསང་སྔགས་འདི་རྣམས་ཡང་དག་བརྗོད། །སྙིང་གི་དབུས་སུ་གཙོ་མོ་སྟེ། །:·མཚོག་ཁ་པོ་ལ་སོགས་གནས་རྣམས་སུ། །ལྷ་མོ་ཉི་ཤུ་རྩ་བཞི་བསྒོམས། །སློབ་མ་རྣམས་ལ་ངེས་པར་ནི། །ལྷ་རྣམས་ཀུན་གྱིས་བྱིན་གྱིས་རློབ། །ཨུ་རྒྱན་གྱི་་་སྒྲོལ་མའི་རིམ་པ་གསང་བ་ལས་ཀྱང་ཆེས་གསང་བ་རྫོགས་སོ། །།།པཎྜི་ཏ་སཾ་གྷ་ཤྲཱི་དང༌། :ལོ་ཙཱ་བ་འཇམ་དཔལ་སྒེག་པའི་རྡོ་རྗེས་་་བསྒྱུར་བའོ། །།།དེ་རྒྱུ། ཨོ་ཏཱ་རེ་ཏུཏྟཱ་རེ་:ཏུ་རེ་ཧཱུྃ་ཕཊ། ཨོཾ་སུ་ར་ཏེ་ཤྭ་རི་ཧཱུྃ་ཕཊ་ལ་སོགས་པའོ། །ཨོ་ཨ་ཤ་པུ་ཙ་ཧཱུྃ་ཕཊ་ཨོ་ཧཱུྃ་ལ་སོགས་པའོ། །ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ད་ཧ་ད་ཧ། པ་་ཙ་པ་ཙི། མ་ཐ་མ་ཐ། ཨི་མཾ་བ་ལི་གྲོཏྟ་གྲནྠ་མ་མ་སརྦ་ཀརྱ་ནི། སཱ་ད་ཡ། སཱ་ད་ཡ། ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། གཏོར་མའི་སྔགས་སོ། །རྒྱུ། །བ་ཌཱ་ཀི་ནི་རྡོ་རྗེ་ལྡན་མ། པཎྜི་ཏ་པ་ར་ཧེ་ཏ། གྲུབ་ཐོབ་ཝི་ནི་ཏ་བྷ་དྲ། ཨཁ་ཆེའི་པཎྜི་ཏ་སཾ་བྷ་མ་ཧེ་བྷ་དྲི། འཇམ་དཔལ་གསར་མ་དེས་ས་ཕུག་པ། །ཆཆ་ཎ། །།ਸਿਰਨੂੰਛੋਟਲੈਮਾਣਸਸ਼ਲੋਂ། ཏལ་བ་ལ་ཀ::ཊ་་༌་་ཚེ། ཏཔ་ཊ་ཊ་ཋ་ཌཨ་དང་ཁ་:ཚེ་། ཏ་ཀ་་། ཏད་ཀྱི་་་་་་དང་། དཀྱི། ཏབ་ཀྱི་་ཚེ་ང་ཊ་ར་ཚེ། ཏག་བློ་ཙ་ཊ་ཏ་པ་པའི་ཏའི་ཆ་། །ཏབ་རྗེ་གི་ཚོ། ཏ་ས་་། ཅི་ཟླསྐྱ་ཊ་ཡ་ཏུ་ཚོ། ཏ་རྒྱས་ཀྱང་། །ས་ར་ཏི་བླ་། ཏས་རྒྱུ་ནི་ང་ཟླ་། ཏས་་ཀ་ཏེ་ཏ་ཀ་། །ཏ་ད་ང་ཚུ%། ཏ་ཀ་་་། ་། ཏཥ་་ཏི་ལྔ་། ཏ་ད་ན་གྱི། ཏ་་་ཊཎ་། །ཏཁ་ཙོ་ཏི་རི་རི་ཎཾ། ཏ་ཀ་། ཏ་བའི་ཊི་ཁ་ཊ་ཏ་ལ་ཚེ་ཚེ༌ཚུདོ༼ཆི་ཚེ། ཏག་བློའི་ཀཏུ་་།
