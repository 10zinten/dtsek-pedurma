ལས་ནི་སྡུག་བསྔལ་མཐོང་། །འཚོ་བ་ལས་ནི་ཆོས་འགྱུར་མཐོང་། །སོམ་ཉི་མ་ལུས་དང་བྲལ་བསི། །འཆི་བ་བསླུ་ཅིབ་ཀུན་དུ་ྋསྤྱོད། །དེ་ལ་དང་པོར་བསམ་གཏན་བཅས། །བགེགས་དགྲའི་གསང་སྔགས་འབུམ་ཉིད་ནི། །ལེགས་བཟླས་སྦྱིན་སྲེག་ཁྲི་བྱས་པས། །བགེགས་ནི་ཐམས་ཅད་བཟློག་པར་བྱ། །ན་མས་མནྟ་ཀཱ་ཡ། །དེ་ནས་བཱཙིཏྟའི་ཚིག། བཛྲ་:ཎཾ་ན་མོ་བཛྲ༄༅། །ཀྲོངྷ་ཡ་ཞེས་ཚིག་བརྗོད་ལ། །མ་ཧཱ་དཾབྦོཥ་ཊའི་རྗེས། །བྷི་ར་བ་ཡ་ཨ་སི་བརྗོད། །མུ་ས་ལ་པ་ར་ཤུ་པ༄༅། །གྲི་ཧི་ཏ་ཧསྟ་ཡ་ཡིན། །ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི། །ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧིའི་ཚིག། ཏིཥྛ་ཏིཥྛ་ཚིག་དེ་ནས། །བནྡྷ་གཉིས་དང་ཧ་ན་གཉིས། །ད་ཧ་གཉིས་དང་གརྫ་གཉིས། ་། དེ་ནས་བི་སྤོ་ཊ་གཉིས་ཏེ།  ོ། སརྦ་བཛྲ་ཚིག་དང་ནི། །བིགྷཱ་བི་ནཱ་ཡ་ཀཱན་བརྗོད།  ོ། མ་ཧཱ་ག་ཎ་བ་ཏིའི་ཚིག། ཛི་་་་་ཏཱནྟ་ཀ་ར་ཝིཡ། ཊོ་། ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་ཡི། ཊ། སྔགས་འདི་བགེགས་རྣམས་ཟློག་་བྱེད་ཡིན། །ཡང་ན་འདི་ཡི་སྙིང་པོ་ཡིས། །བགེགས་རྣམས་ཐམས་ཅད་ཟློག་པ་པར་བྱེད༅། །སྔོན་བཞིན་བཟླ་དང་སྦྱིན་སྲེག་གིས། །འདི་ཡི་མཐུ་ཡང་དེ་འདྲ་ཡིན། །ཨོ་སྔར་ཧུ་ལུ་ཧུ་ལུ་དང་། །དེ་འོག་ཏིཥྛ་ཏིཥྛའི་ཚིག། ཧ་ན་གཉིས་དང་ད་ཧ་གཉིས། །གརྫ་གརྫ་དེ་འོག་ཏུ། །བི་སྤོ་ཁྱེཊ་ཡ་ལན་གཉིས་བརྗོད། །སརྦ་:བིགྷཱ་བི་ནཱ་ཡ་:ཀཱ་ནི། །ཨོཾ་ཨ་མྲྀ་ཏ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ་མ་མ་རཀ་ཀུ་རུ་སྭཱ་ཧཱ། །དེ་ལྟར་བདུད་དང་བགེགས་རྣམས་ནི། །མ་ལུས་རྣམ་བསལ་འཁྲུག་མེད་པས། ཎེ། ཇི་ལྟར་མངོན་འདོད་རྣལ་འབྱོར་ནི། །གཅིག་ཅིག་ཉིན་མཚན་་་་བསྒོམ་པར་བྱ། །།སྟུག་པོ་བཀོད་པའི་གནས་བཞུགས་ཏེ། །ཡོ། ཞི་བས་སངས་རྒྱས་མཛད་པ་མཛད། །པདྨ་ཟླ་བའི་གདན་:སྟེངས་སུ༄༅། །རྡོ་རྗེ་སྐྱིལ་ཀྲུང་བཅས་་་ནས་ནི། །བྱང་ཆུབ་མཆོག་གི་ཕྱག་རྒྱར་ལྡན། །འོད་ཟེར་དཀར་པོའི་ཚོགས་་་i།
