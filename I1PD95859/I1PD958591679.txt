རྒྱུད། ཀྱི་གིས་བསགས་པའི་བསོད་ནམས་ཀྱིས། །ཉོན་མོངས་ཐམས་ཅད་གཅོད་པར་ཤོག། འཇིག་རྟེན་འདས་པའི་གྲུབ་པ་འཐོབ། །དཔལ་ནག་པོ་ཆེན་པོའི་མངོན་པར་་་་་རྟོགས་པའི་རྣམ་པར་བཤད་པ། ལྷན་ཅིག་སྐྱེས་པའི་ཞབས་ཀྱི་སློབ་མ། །སློབ་དཔོན་ལི་ཀ་རིའི་ཞབས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།་་བ་་་ཁར་་་་་་རྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་ན་རེནྡྲ་བྷ་དྲ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ཨུ་རྒྱན་པ་རིན་ཆེན་དཔལ་གྱིས་རི་བོ་ཆེན་པོ་སྤུ་ཏྲའི་མགུལ་དུ། བདེ་ཆེན་གླིང་གི་ཕོ་བྲང་་ཆེན་པོར་བསྒྱུར་བའོ། །།།ཏ་ག་ལ་བ་ད་ན་པ་ཏི། ਲੋਕਸਿੰਘ,ਸੰਮਲਨੂੰsERਉਸਨੂੰਰੂਹਨੂੰ::ཀྱི་བ་གས་་དག་བསྡུར་མཆན། བ་ལང་ཚོ། བློབ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«པེ་»«སྣར་»ལ་མི་བཞུགས་པས་དཔེ་སྡུར་བྱུང་མེད། །ਮੇਰਸਮਸčਬੂSpਮਲਲੇਬਕੋਲབསྡུར་འབྲས་རེའུ་མིགਸ'རྒྱུད། (ཡོ་ག་དང་། གྲུབ་དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། །རྒྱུད། །སྡེ་ཚན། །པོད་རྟགས། །ཚན་གྲངས། ལྡེབ། རབཙབ་བ་བ་ཕྲེང་། །།ཚེག་ཁྱིམ། ਲਈਖਾਲਈਸਥਾਸਥਾਵਲਟར།
