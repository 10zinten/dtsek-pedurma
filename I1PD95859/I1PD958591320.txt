བད་རུང། འཁ་ཎརྒྱུ་ཙPཁ་སྐམ་ཀ་ཎ་ཡ་རབང་དང་། ཚོ་པ་ལ་ལ་ལ་ཏོཥ་ཏེ་བ་ཁ་ཁ། ཀ་ཀ་ཀོ་ལ་་་་ཁ་ལ་ཁ་ཏ་ཁ་ਸਸਸਸਿੰਬਲੌਤੋਂ।ਲੀਵਰਨੂੰਸਮਝਪੈਲੀਫ਼། #ཅཁ་ཉིjsཁོ་ཆེ་བ་དེ་ཎ་ཊས་ཏཁའ་ཎ་ཚོeཚsའི་ཕྱི་ཀ་དང་པས་ཀྱི་གྲོགས་ཀྱི་བ་དེ་པར་འབད་པའི་བར་་བ་::ཀྱི༄༅། །དབང་ཆོས་རྟེན་འབྲེལ་འགྲོ་ལྡོག་ལ་::ཁ་ལ་། ་བ། ཚད་ལ་བློ་འདིའི་ཁ་ན་། དབང་ཁྲིད་ཀྱི་བཀའ་::ཀའ་ཚལ་གྱི་ཚབ་ཏུ་ཁ་ཆཀྱི་ག་ང་ལ་བློ་ཁོག་ཆེ་ན་གནས་དང་ཁོ་ཅན། ་ཆེ་གཏན་པ་བp,་ནར་འགགས་བྲལ་དཀ་ཁའིབ་ཊི་བ། གད་པ། ཏཱ་ཏོ་ཚེ་་ར། ལ། ཤ། ༄༅། །རྒྱ་གར་སྐད་དུ། དྷརྨ་ཨ་བྷི་ཥི་ཀ་མ་ར་ག་ས་ཧི། བོད་སྐད་དུ། ་དབང་ཆོས་རྟེན་འབྲེལ་འགྲོ་ལྡོག། དཔལ་རྡོ་རྗེ་མཁའ་འགྲོ་མ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །བླ་མ་མཁའ་འགྲོ་རྡོ་རྗེ་འཆང་དབྱེར་མེད། །གར་བསྒོམ་གནད་ཀྱི་ཡུལ་དེར་འདུད། །བླ་མེད་རྒྱུད་རྣམས་བཅུད་བསྡུས་པའི། །དབང་ཆོས་རིམ་པ་བཤད་པར་བྱ། །དབང་བཞིའི་གདམས་པ་ཟབ་མོ་ཡིས། །རང་གཞན་ལྷན་ཅིག་སྐྱེས་པ་དེས། །གཞན་རྣམས་དབང་དུ་སྡུད་པ་ཡི། །སྲོག་སྔགས་ཟབ་མོ་འདི་ཉིད་ལ། །དབང་བཞིའི་གདམས་པ་མེད་པ་ཡིས། །རྣམ་རྟོག་དབང་དུ་རྣམ་རྟོག་སོང་། །རང་ཉིད་གཞན་གྱི་དབང་དུ་འགྲོ། །རང་སྲོག་གཅོད་པའི་་ལྟ་རེ་ཡིན། །དབང་པོ་གཅིག་དང་ཚེ་གཅིག་ལ། །གང་ཟག་བཞི་དང་རིགས་ཅན་བཞིན་ཡི། །མཆོག་དབང་བཞི་ཡིས་ཐེ་ཚོམ་མེད། །དུག་གསུམ་ལམ་དང་། རྣམ་རྟོག་ལམ། །མཆོག་ཐུན་དངོས་གྲུབ་གང་བསྒྲུབ་པ། །བསྙེན་སོགས་བཞི་།
