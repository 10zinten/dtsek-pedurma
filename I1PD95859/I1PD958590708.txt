ཧཱུཾ། །ཛཧཱུྃ་བཾ་ཧོ?། ཨཱ་ཀརྞཱ་ཡ། ཨཱ་ཀརྞཱ་ཡ། །པྲ་བེ་ཤ་ཡ་པྲ་བེ་ཤ་ཡ། བནྡྷ་ཡ་བནྡྷཱ་ཡ། ཏོ་ཥ་ཡ་ཏོ་ཥ་ཡ། ཚམོ་ཧ་ཡ་མོ་ཧ་ཡ། ཨོ་ཨེ་:ཨཱ་ར་ལི་ཧོ། ཧྲི་ཧྲི་ཧྲི། ཁཨོ་འཇིགས་བྱེད་ལྷ་ཆེན་དབང་ཕྱུག་དང་། །དེ་བཞིན་ཚངས་པ་བརྒྱ་བྱིན་དང༌། །ཁྱབ་འཇུག་འདོད་དགའི་དབང་ཕྱུག་དང་། །འཇིག་རྟེན་་་སྐྱོང་བའི་ལྷ་ཀླུ་དང༌། །གནོད་སྦྱིན་གཤིན་རྗེ་:མ་མོའི་ཚོགས། །མཚམས་ཀྱི་་དབང་ལྡན་ལ་སོགས་པའི། །»::། གལ་ཏེ་རྫུ་འཕྲུལ་མཐུ་མངའ་རྣམས། །།བྲན་ཚོགས་གཡོག་དང་ཆུང་མར་བཅས། །དཔའ་བོའི་སྤྱན་སྔར་དམ་བཅས་བཞིན། །རང་གི་དམ་ཚིག་རང་བྱོས་ལ། །ཚོགས་ཀྱི་དཀྱིལ་འཁོར་བསྲུང་བའི་ཕྱིར། །འཁོར་བཅས་གནས་འདིར་གཤེགས་སུ་གསོལ། །།ཨོཾ་ཨེ་ཧྱ་ཧི། རུ་ལུ་རུ་ལུ་ཛ®ཧཱུྃ་བཾ་ཧོ། "་ཨོ་བཛྲ་ཌཱ་ཀི་ནེ་བློ་ཨེ་ཨ་:ར་ལི་ཛཛ%། ::ཨོ་ཌཱ་ཀི་:ནཱི་ཨཱ་ཀཱནྟ་ཡ་:ཨཱ་ཀཱནྟ་ཀྱི་ཡ། པྲ་བེ་ཤ་ཡ་པྲ་བེ་ཤ་ཡ། །བནྡྷ་ཡ་བནྡྷ་ཡ། ཏོ་ཥ་ཡ་ཏོ་ཥ་ཡ། །མོ་ཧ་ན་མོ་ཧ་ན། །་སརྦ་ན་ཏ་ཀི་ནན། ཁཧྲི་ད་ཡ་ཧཱུྃ་ཕཊ་ཛ་ཛཧཱུྃ་བཾ་ཧོ། ཊ་ཀི་ནི་ཧཱུཾ་ཛ། ཊ་ཀི་ཕྱག་རྒྱས་དགུག་པའོ། །ཨོཾ་ཤྲཱི་ཧེ་རུ་ཀ། མ་ཧཱ་བཛྲ། སརྦ་དུཥཱན་ས་མ་ཡ་མུ་:དྲ་པྲ་པཉྩ་ྋཀྲ་ཧཱུཾ་ཕཊ། ཧེ་རུ་ཀའི་ཕྱག་རྒྱས་ཟིལ་གྱིས་མནན་པའོ། པ་། ཨོཾ་བཛྲ་བ་ཏ་མུ་ཁ། །:ཨཱ་ཀཱ་ར་ཥ་ཡ། ཟླ་:ཨཱ་ཀཱརྟ་ཝིཡ། །པྲ་བེ་ཤ་ཡ། པྲ་བེ་ཤ་ཡ། བནྡྷ་ཡ། བནྡྷ་ཡ། ཏོ་ཥ་ཡ། ཏོ་ཥ་ཡ། སརྦ་ཟླ་ཀི་ནན། ཧྲེ་ད་ཡ་ན། །ཧ་ཧ་ཧི་ཧི། ༀ་ཨཱཿཧཱུྃ། རྟ་མགྲིན་ཕྱག་རྒྱས་དབང་དུ་བསྡུ་བའོ། །་། རྡོ་རྗེ་ཕུར་པ་གདབ་བྱས་པས། །བགེགས་ཀུན་གཡོ་བ་མེད་པར་བྱས། །ཡེ་ཤེས་མེས་བསྲེགས་བདུད་རྩིས་གསོས། ::། སྦྱང་ཞིང་བྱིན་བརླབས་ལྷ་ཡི་སྐུར། །སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་དང༌། །རིགས་ཀྱི་བདག་པོའི་མཚན་མས་བརྒྱན། །སྔ་མའི་ཕྱག་རྒྱའི་ཆ་ལུགས་ལས། །འོད་མདངས་རྒྱན་་་།
