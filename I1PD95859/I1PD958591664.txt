བྷེ་ལུ་ནེ་ལུ་ནེ་སྭཱ་ཧཱ། ལ་སྔགས་འདི་རང་ཉིད་ཉལ་བའི་ཚེ་བུད་མེད་ཀྱི་མིང་ནས་་་བཟུང་ལ་བཟླས་པ་བྱས་ན། དཔག་ཚད་བརྒྱ་ན་ཡོད་ཀྱང་འོང་བར་འགྱུར་རོ། །མ་འོངས་ན་འཆི་བར་འགྱུར་རོ། །།དེ་ནས། ན་མསརྦ་བུདྡྷེ་སྦྱོ་ཛ་ཧི་བུདྡྷ་ཨ་ནོ་ཛ་ར་སྲེ་སི་དོ་མེ་བཱ་ཙ་ནཾ་ཀུ་རུ་སྭཱ་ཧཱ། དེ་ནས་གང་དུ་འདུག་པར་བཟླས་པ་བྱས་པས། ལ་ཇི་ལྟར་སྐྱེས་པའམ། དཔའ་མོའི་མིང་ནས་བཟུང་སྟེ་བཟླས་པས། ཐམས་ཅད་རང་བཞིན་གྱིས་དབང་དུ་འགྱུར་རོ། །ན་མལླི་པྲ་གྷ་མི་ཀུ་རུ་ཀུ་རུ་་བི་མ་ལ་ཡེ་སྭཱ་ཧཱ། །ཆུ་ལ་ལན་བདུན་བཟླས་ཏེ། སྐྱེས་པའམ་བུད་མེད་གང་གི་མིང་ནས་བཟུང་སྟེ་འཐུངས་ན་དབང་དུ་འགྱུར་རོ། ་། ན་མཀ་ཏྲ་ཀ་ཏྲི་གྲོ་ར་རཱུ་་པི་ཎི་ཆེ་གེ་མོ་མེ་བ་ཤ་མ་ནཱ་ཡ་སྭཱ་ཧཱ། ་ཁཟླ་བའི་དུས་སུ། དང་པོ་ཟས་ཁམ་གཅིག་ལ་གང་གི་མིང་ནས་བཟུང་སྟེ། ལན་བདུན་བཟླས་ནས་ཟོས་པས། །ཞག་པའི་དབང་དུ་འགྱུར་རོ། །ན་མཨཱརྱ་སཾ་བུདྡྷཱ་ཡ། ཨེ་ཧེ་ཧི་བྷུ་ཏ་པྲ་ཏྲ་དྷུ་ཏ་ཀཱ་རྱ་ཀུ་རུ་སྭཱ་ཧཱ། དེ་ནས་མིག་སྨན་ལ་བརྒྱ་ཕྲག་བརྒྱ་བཟླས། །རང་གི་མིག་ལ་བསྐུས་ན། ཐོག་མར་མཐོང་བ་དེ་དབང་དུ་འགྱུར་རོ། །དེ་ནས། ་ན་མཔྲ་ཁ་ཀ་ཡ་མཉྫ་གྷོ་ཥ་ཀཱ་ཡ་པྲ་ཁི་ཏོ་སི་ཛྷ་ཧི་ལཾ་ཀུ་རུ་སྭཱ་ཧཱ། །དེ་ནས་ཡུངས་ཀར་ལ་བཟླས་ལ། གང་གི་ཁང་པར་གཏོར་བ་དེ་དབང་དུ་འགྱུར་རོ། །ན་མོ་རདྨ་ཏྲ་ཡཱ་ཡ། ཀུ་བྷི་ནི་ཛྷ་བྷི། མ་ཧཱ་བེ་ཏཱ་ལཱི་སྭཱ་ཧཱ། དེ་ནས་རང་གི་ཁུ་བ་དང་། ག་བུར་ལ་བརྒྱ་རྩ་གཅིག་བཟླས་ཏེ། གང་གི་མིང་ནས་བཟུང་སྟེ་སྦས་པ་དེ་དབང་དུ་འགྱུར་རོ། །།ན་མོ་བུདྡྷཱ་ཡ་དྷརྨ་སཾ་གྷ་བཛྲ་པཱ་ཎི་བྷུཤྲཱི་ཏ་པི་ག་ལེ་ལ་ཧཱུཾ་ཀུ་རུ་ཀུ་རུ་ཛྭ་ལ་ཛྭ་ལ་སྭཱ་ཧཱ། གི་ཝང་ལ་བརྒྱ་རྩ་བརྒྱད་བཟླས་ཏེ་ཐིག་ལེ་བྱས་པས་ཇི་ལྟར་མཐོང་བ་ཐམས་ཅད་དབང་དུ་འགྱུར་རོ། །ཧཾ་བྷོ་བྲ་ཥ་ཀ་ཡ་ཏི་ཨ་མུ་ཀ་ཤ་མ་ནུ་གླི་ངས་མ་ནེ་སྭཱ་ཧཱ། དེ་མ་ཐག་ཏུ་མི་འོང་ན། དཔའ་མོའམ། སྐྱེས་པ་ཇི་་་།
