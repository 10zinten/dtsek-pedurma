བསྟན་འགྱུར། རྒྱུད། ཤ་་ཀྱི་ལྷ་མོ་སྒྲོལ་མའི་རང་བཞིན་རྣལ་འབྱོར་ལྡན། །རྡོ་རྗེ་གསང་བའི་རྒྱུད་དུ་སྒྲོལ་མ་ཡི། །སྒྲུབ་ཐབས་གོང་དུ་བྲིས་པ་རྡོ་རྗེའི་ཚིག། སེམས་ཅན་ཡེ་ཤེས་ཐོབ་བྱེད་གང་ཡིན་འདི། །མ་དད་:པའམ་མ་སྦྱིན་བྱ་བ་ངན་པ་དང་། །ཆགས་ཅན་ཁྲོ་བ་ཅན་མིན་སློབ་མིན་བུ་ལ་མིན། །སྡུག་བསྔལ་བློ་ཅན་ལ་མིན་སློབ་དཔོན་གྱིས། །གཞན་ལ་་སྒྲུབ་ཐབས་འདི་སྦྱིན་བྱང་ཕྱོགས་ཀྱི། །ཨུ་རྒྱུན་ནས་འོང་བྲམ་ཟེའི་གནས་ཀྱི་་་་ནང་། །ཞིང་གི་བུ་མོའི་རྣལ་འབྱོར་མ་ཚོགས་རྡོ་རྗེའི་ཚིག་དེ་གྲུབ། །བཅོམ་་་ལྡན་འདས་མ་ལྷ་མོ་སྒྲོལ་མ་ཨུ་རྒྱན་གྱི་རིམ་པའི་སྒྲུབ་ཐབས་རྫོགས་སོ། །ཞིང། །།།མཁའ་འགྲོ་མ་རྡོ་རྗེ་ལྡན་མས་གསུངས་པའོ། །།ཀློབ། དོལ་པོ་མ་ཧཱ་ཨཱ་ཙརྱ་རཏྣ་ཤྲཱིའི་སློབ་མ་པཎྜི་ཏ་པ་ར་ཧེ་ཏ་དང་། ལོ་ཙཱ་བ་ནག་་ཚོ་དགེ་སློང་ཤེས་རབ་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །ཞབ། །ཁ་ཁོམ་ལདེ་བ་ཕྱི་མ་ནི་[gz་བཟླ་གསབློ་ག། ཏ་ཚོ། །བ་འཁཡཏུཊ་རྒྱུ་ཁཀྱི་ཤོnn@་རྒྱུ་བ་ཚོ། བསྡུར་མཆན། ਲਾਵਰਕਸ਼ਕਰਸਰਲਧਨਕཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཨུ་རྒྱན་གྱི་རིམ་པའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་ག། ཚེ«པེ་»«སྣར་བི། ཚེ«པེ་»«སྣར་»ནཾ། ཚུ«པེ་»«སྣར་+། «སྣར་»+ལྷ། ༄༅«པེ་»སྣར་»སྙིང༌། ཉི«པེ་»«སྣར་དྲིས། །«པེ་»དགོད། ཉི«པེ་»«སྣར་—ཉ། ཉི«པེ་»«སྣར་ཐིམས། «པེ་»«སྣར་»སར། «པེ་ཁ༽《སྣར་པ། །།ངབས་དེ་བས་མང་བ་ན་ཏུ་::ཀློ་ཡུལ་«པེ་»«སྣར་རྗེས། «པེ་ཙཱ་ཨཱུ་ཨ་གོ་ར་དེ་མ་ཀ་ཨུ་ཏེ་ཀུ་ཀ་ལ་ཀ་ཧེ་བྲི་སྭཱ་ས་ན་སི་མ་ཀོ། «སྣར་ཝཱ་ཨུ་ཨ་གོ་ར་དེ་མ་ཀ་ཨུ་ཏྲེ་ཀུ་ཀ་ལ་ཀ་ཧེ་བྲི་སཱི་སོ་ན་སི་མ་ཀོ། «པེ་»«སྣར་_འདི། «པེ་»«སྣར་མ་ལ་ལུས། «པེ་»«སྣར་ཡུ་ར་ཤ་ཏི་ཤ«པེ་»«སྣར་ཏི་སོན་དྷ། @«པེ་»«སྣར་ཏི་ཙན་ཏྲ་ཀློའོ། །O«པེ་»«སྣར་ཏི། «པེ་»«སྣར་»ཏི་བཀྲི། ཁྱེ«པེ་»«སྣར་»ཏི། ཀྱེ«པེ་»«སྣར་»ཏི་གྲ་ཧ། «པེ་»«སྣར་»ཏི་མུཏ་མ། «པེ་འཕྲེང་། «པེ་»«སྣར་ཁཎྜི། ལ«པེ་»«སྣར་+པ།  ིཚེ«པེ་»«སྣར་»ཀ་ར། ཁྱེ«པེ་»«སྣར་ལི། «པེ་»«སྣར་»རོཏྲ་ཛ་དྲི་དེ་བཞིན་ཙནྡྲ་དང་། །མི་ཌི་བཀ་རྒྱུ་་།
