རྒྱུད། །ལ་ཁྲོད་ཆོས་ཉིད་རོ་མྱོང་བས། །ཤེས་རབ་མཚོན་ཆས་ཀུན་གཅོད་ཅིང་། །སྲེག་་བྱེད་རིགས་ཀྱི་ཕྱག་རྒྱ་འཆང་། :། རིགས་ཀྱི་ཚུལ་གྱིས་གར་བྱེད་པའོ། །ཧཱུྃ་བཛྲ་ཀི་ཤི་ཙྪིནྡ་ཕཊ། མ་ཧཱ་རུ་ཏྲ་ཧི། ཀུན་སྤྱོད་ཚུལ་རྣམས་ཀུན་གྱིས་རོལ། །དབྱིངས་དང་ཡེ་ཤེས་ཆགས་པའི་ལས། །སྟོང་གསུམ་མི་གཏོང་འཆིང་བྱེད་པའི། །འཇིགས་བྱེད་རྔམ་པའི་ཆ་ལུགས་ཅན། །།ཧཱུྃ་བཛྲ་པེ་མཱ་ལ་པཱ་ཤེ་ཕཊ། ?མ་ཧཱ་རུ་ཏྲ་ཧི། བྱམས་པའི་མདའ་འཛིན་སྒེག་པའི་ཚུལ། །མ་བཅོས་ཟུང་དུ་འཇུག་་་་པའི་གཞུ། ་ར། །མཉམ་ཉིད་འཕྲོ་བས་ཀུན་བརྩོན་པའི། །རྩེ་གཅིག་ཕྲིན་ལས་འགྱུར་བ་མེད། ་། ཧེ་བཛྲ་ར་ཏི་བི། མ་ཧཱ་རུ་ཏྲ་ཧི། ཐུགས་རྗེ་སྣ་ཚོགས་ཁྱབ་པ་ཡི། །སྒྲར་བཅས་སྐུ་གསུང་ཐུགས་རྒྱལ་བྱེད། །ཟླ་། གནས་གསུམ་ལས་རྒྱུད་དག་བྱེད་པའི། །ཡོན་ཏན་ཀུན་གནས་འཕྲུལ་པའི་མཆོག། ཐཾ་མ་ཧཱ་ར་ཏ་བཛྲ་ཧོ། མ་ཧཱ་རུ་ཏྲ་ཧི། དགའ་བྱེད་མྱུར་བའི་བརྟུལ་ཞུགས་འཆང་། །ལས་རྒྱུད་གཅོད་བྱེད་ཐེག་པའི་མཆོག། ཟབ་ཅིང་རྒྱ་ཆེ་ཀུན་ཚུལ་གྱིས། །ཡོན་ཏན་བརྒྱད་འཛིན་བྱེད་པའོ། །།ཨོཾ་བཛྲ་ལོ་ཙ་ནེ་ཡཾ་ཡཔོ། མ་ཧཱ་རུ་ཏྲ་ཧི། བཏང་སྙོམས་སྣ་ཚོགས་ཕྲིན་ལས་སྐུ། །གཟུགས་མཛེས་:སྤྱོད་པའི་སྒྲ་དང་བཅས། ་་། དག་་་་བྱེད་ཕྲིན་ལས་བསྙེངས་པ་མེད། །།ཐམས་ཅད་ཡེ་ཤེས་རྣམ་སྤྲོས་པའོ། །།རཾ་མ་་་ཧཱ་བཛི་ཧི༄༅། །མ་ཧཱ་རུ་ཏྲ་ཧི༄༅། །སྦྱིན་པའི་མཆོག་རྣམས་རྣམ་པར་གྲོལ། །གཟུགས་མཆོག་ལྷ་རྫས་མེ་ཏོག་རྒྱ། །ས་གཞི་མདུན་རོལ་གཡས་གཡོན་བསྟོད། །སྣ་ཚོགས་:སྤུངས་པ་འཐོར་ྋབྱེད་པའོ། །ཨ་བཛྲ་པུངྒེ། མ་ཧཱ་རུ་ཏྲ་ཧི། ཚུལ་ཁྲིམས་དྲི་མཆོག་ཚིམ་བྱེད་པའི། །བདུག་སྤོས་ལེགས་པར་སྦྱར་བའི་མཆོག། གཡས་སྐོར་བཏེག་སྟེ་ཕྱོགས་བཅུར་བསྒྱུར། །མདུན་དུ་ཕྱོགས་པར་སྔོ་བྱེད་པའོ། །ཨ་བཛྲ་དྷུ་པ། མ་ཧཱ་རུ་ཏྲ་ཧི། བཟོད་གསོལ་ཤེས་རབ་ཏིང་ངེ་འཛིན། །།
