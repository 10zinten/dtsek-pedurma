ཡང་ཏིག་ཁ་ཏ༌བ་ཉི་ཁ་གླུ་ཡི་ལ་ཏཁ། ཚིག་། ་་ལ། »ཎ་ཀ་ག་ང་། །ཅན་ཁ་རནས་་ཚེ་གང་སྐྱཁང་། ཁལའི་ཁྱི་ར་བོ། ་བྱོན། །ཆཁ་། ་་ག་པ་ཉིད་ཀཚོ་ཆེན་པོ་ཀ། པ་ཊི་ཀོ། ཏི་ཋ་ཌཱ། ཡོ་ཏ་ཀ། ཏུ། ཤང་ཁ་ཤོཁའ་འགྲོས་བྱ། ཎཥ་ཊི་ཀ་ཁ་བློ། །ཟླ་ཚོ་ཞ་བཝིམ་པས་ཚེ་བཅའ་ཁྲི་ཁར་ཁྲིད་དུ་མ་ན། །ཟླ་ལ་ངའོདལྔ་། xz་བ། ཁ་ག་ཏེ་ལྟ་བ་ལ་དག་པའི་འཚོ་བྱེད་ཀཀསྐྱེ་བ་ན། དོ་བ་། ༄༅། །ཐུགས་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། །:aལྟ་ཐེ་ཚོམ་བྱའོ། །ཁ་དོག་འབད་བ་་སྐྱེ་བ་ཡིན་པ་དང་། ས་པ་ཀ་་་། ཁ་ARਉਸਲgਨਮpagਸਵਸੇ।ਕਾਨੂੰਨਲਾਸਟ:ཁ་ཎ་ཁགཙོ་གཏད་པ་བཀའ་བ་དུ་མ་ཁྲི་ཁང་བ་དེགས་ཞུ་མི་ནང་ཁ་ཚང་ཚེ་དེང་བྱ་འོཊ། པ་ཡི་ཆར་༼ཏོ་ཀན་ཁ་རྒྱུ་བ་དབྱར། ::དTaཚོགས་འདུར་ཆག་གམ་ཚོ་འོ་༄༅། །རྒྱ་གར་སྐད་དུ། ཀཱ་རུ་ཎ་ཏཱ་ར་སདྷ་ནཾ། བོད་སྐད་དུ། ཐུགས་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། ཐུགས་རྗེ་ཉེ་འགྱུར་ཁྲོ་མོ་ལྷ་མོར་སྟོན། །མཐོ་རིས་ཐར་པའི་བགེགས་རྣམས་སྒྲོལ་མཛད་མ། །འཇིགས་པ་ཆེན་པོ་བརྒྱད་་སྐྱོབ་ཅིང༌། །འདོད་ཡོན་ལྔ་ཡི་གདུང་བ་སེལ། །།ཕྲིན་ལས་བཞི་ཡིས་འགྲོ་དོན་མཛད། །རྟག་ཏུ་སེམས་ཅན་མདུན་དུ་བཞུགས། །གསོལ་བ་བཏབ་པས་མི་བསླུ་བའི། །སྒྲོལ་མ་རལ་གཅིག་མ་ལ་འདུད། །བླ་མ་བརྒྱུད་པའི་བཀའ་དྲིན་་གྱིས། ང་། སྙན་བརྒྱུད་ཞལ་ནས་ལུང་བསྟན་པའི། ་༌། སངས་རྒྱས་ཉིད་ཀྱིས་མི་གསུངས་པ་ནི། །གཉིས་པོ་འབྲེལ་འདོད་དོན་དུ་བྲི། །གསང་བ་ལས་ཀྱང་་་ཆེས་གསང་བ། །ཟབ་པ་བས་ཀྱང་ཟབ་པ་ཉིད། །མཆོད་པ་བཞིན་དུ་འདོད་་པ་ཡིས། །ཐེག་ཆེན་རིགས་ལྡན་སྨིན་པ་དེས། །ལྷ་ཡི་མཚན་མ་དུར་ཁྲོད་དང༌། །ཁང་སྟོང་དབེན་ལ་བཀྲ་ཤིས་པར། །མཆོད་ཡོན་ཞལ་ཟས་ལྡན་པ་ལ། །བསྡམ་།
