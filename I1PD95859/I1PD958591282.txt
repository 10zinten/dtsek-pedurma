རྒྱ་གར་གྱི་མཁན་པོ་:ར་ཏ་དྷ་ར་དང་། ལོ་ཙཱ་བ་རྨས་བསྒྱུར་བའོ། །།།ལག་པ་ན་བྱ་བའོ། །དེ་ལ་འ་མེད་པ་ལ་ཡི་རི་ལ། དེས། །སྙིང་གའི་པགས་པར་གན་ད། ཁྱེད་ད་པ་བྱེད་འདང་པའི་བར། བསྡུར་མཆན། དེ་ལ་བཟླ་བ་དང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ®«པེ་»«སྣར་»+། ཉི«པེ་»«སྣར་སྩལྟ། «པེ་»«སྣར་སྦྱང་། «པེ་»«སྣར་»པོ་བློ་ཡིས་བྱ། «པེ་»«སྣར་རྒོད། @«པེ་»«སྣར་བརླབས། «པེ་བསྟད་ལ། «སྣར་བསྟོད་ལ། «པེ་»ཨོཾ་དྷི། «སྣར་ཨོཾ་དྷི། W«པེ་»«སྣར་ན་འགྲུབ། «པེ་»«སྣར་བསྐྱིལ། ༄༅«པེ་བཏུག«ཅོ་»དགུག«པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་»གདོན། «ཅོ་གནད། ཨེ«པེ་»«སྣར་»གྱི། «པེ་«སྣར་རནྟ། ཨེ«པེ་»«སྣར་»ཙ། །།ད་དེ་ཁྲི་བ་་་་་་་་་་་་། །བ་པ། འོའབ#ངན་ན་ད་ན་བྱ་བ་ལྔ་བ་དང་::ཀ་་་། བ། ཊོ་ཝ་ཊྚགའ་བསྡུར་འབྲས་རེའུ་མིགཚོ་བ་ར་བླ་ཀློ་བློ་:ལ་ར་ཎ། ཀཆོག། རྒྱུད། （ན་ན་བད་ཀྱི་ཚོགས་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། །སྡེ་ཚན། །བ་རྒྱུད་འགྲེལ། རྒྱུད། །བོད་རྟགས། ཚན་གྲངས། །ར། ལྡེབ། ན། _བ་བ_ན་ནབཙ། ནབ་ཕྲེང་། པ་དང་། ཚིག་རྐང་། །ཚེག་ཁྱིམ། །་་ད་ལྟ་འཕྱོགས་པར། །ཁ་ཚོད་བལྔ། ཡབྱཁན། གདུགས། །
