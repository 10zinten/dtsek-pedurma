#*་ཀཱ་•••••••བ••••••བརདང་བསྟན་འགྱུར། རྗེ་བཙུན་མ་འཕགས་མ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཉི་ཤུ་རྩ་གཅིག་པའི་ལས་ཀྱི་ཡན་ལག་དང་བཅས་པ་མདོར་བསྡུས་པ་ཞེས་བྱ་བ། ........བ....................དང་པོ། འཕགས་མ་སྒྲོལ་མ་མྱུར་ཞིང་རབ་ཏུ་དཔའ་མོ་ཞེས་བྱ་བའི་ལས་པ་བར། །ཀྱི་ཡན་ལག་ཕྱིར་ཟློག་པའི་ཆོ་ག་ཞེས་བྱ་བའི་ལེའུ། ....གཉིས་པ། འཕགས་མ་སྒྲོལ་མ་སྟོན་ཀའི་ཟླ་བ་ལྟར་དཀར་བ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་ཡམས་ནད་རབ་ཏུ་ཞི་བར་བྱེད་པ་ཞེས་བྱ་བའི་ལེའུ། ......ལེའུ་གསུམ་པ། རྗེ་བཙུན་འཕགས་མ་སྒྲོལ་མ་གསེར་མདོག་ཅན་གྱི་ལས་ནང་ཀྱི་ཡན་ལག་ཚེ་འཕེལ་བའི་ཆོ་ག་རྒྱས་པར་ཕྱེ་བ། ...............བབབབལེའུ་བཞི་པ། འཕགས་མ་སྒྲོལ་མ་དེ་བཞིན་གཤེགས་པའི་གཙུག་ཏོར་རྣམ་པར་རྒྱལ་མ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་དུག་གསད་པའི་ཆོ་ག....…..ལེའུ་ལྔ་པ། འཕགས་མ་སྒྲོལ་མ་ལ་ཧཱུྃ་གི་སྒྲ་རབ་ཏུ་སྒྲོགས་མ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་དབང་དུ་བྱེད་པའི་ཆོ་ག....................བབབབབབབ•བར::ཀྱིཀྱིས་ལེའུ་དྲུག་པ། འཕགས་མ་སྒྲོལ་མ་འཇིག་རྟེན་གསུམ་ལས་རྣམ་པར་རྒྱལ་མ་བར་འབ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་སྒྲིབ་པ་ཡོངས་སུ་སྦྱོང་བའི་ཆོ་ག............ལེའུ་བདུན་པ། འཕགས་མ་སྒྲོལ་མ་གཞན་འཇོམས་མ་ཞེས་བྱ་བའི་ལས་.རརརར་་པོད་ཀྱི་ཡན་ལག་གོང་དུ་འཕོ་བ་ཞེས་བྱ་བའི་ཆོ་ག........................ལེའུ་བརྒྱད་པ། འཕགས་མ་སྒྲོལ་མ་བདུད་ཐམས་ཅད་འཇོམས་ཤིང་དབང་ར། བ་མ་ས་མཆོག་སྟེར་མ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་རྫོགས་པའི་རིམ་པ། .........ལེའུ་དགུ་པ། འཕགས་མ་སྒྲོལ་མ་མཆོག་སྩོལ་བ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་རབ་ཏུ་གནས་པའི་ཆོ་ག...ལེའུ་དགུ་པ། འཕགས་མ་སྒྲོལ་མ་མྱ་ངན་ཐམས་ཅད་སེལ་བར་བྱེད་མ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་དཀྱིལ་འཁོར་དུ་འཇུག་པའི་ཆོ་ག.............ན། དེ་བབབབབར།
