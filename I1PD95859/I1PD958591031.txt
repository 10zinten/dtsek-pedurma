རྒྱུད། ཀརྟགས་ཅན་བདེ་གཤེགས་སྲས་པོའི་ས་གཞི་བདེ་བ་ཅན་ཞེས་བྱ་དེ་ྋམྱུར་:དུ་་་་་་བགྲོད་པར་ཤོག། ཕྲེང་བ་འཛིན་པའི་བསྟོད་པ་པཎྜི་ཏ་ཆེན་པོ་ཐམས་ཅད་་་་་མཁྱེན་པའི་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །།།།ར་དུག་ང་ལོ། ཁ་ཚེ་ད་རིས་དང་འཚོ་གནས་བྱ། བསྡུར་མཆན། ད་དུ་འོང་བ་ན། ན་བ་ད་ན་་་་་་་་་་་་་་་་་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»སྣར་»༄༅། །ཕྲེང་བ་འཛིན་པའི་བསྟོད་པ་བཞུགས། «པེ་»ཏུད། @«པེ་»《སྣར་བགྱིད། «སྣར་ཅིགཚུ«སྣར་»བ། «སྣར་བུར། །«པེ་»«སྣར་བརྟེན། «པེ་»«སྣར་ལས་རྣམས། «པེ་»«སྣར་བཟུང༌། @«པེ་»«ཅོ་»«སྣར་»—ཡང་། «པེ་»«སྣར་ཕྱེད། ཚེ«སྣར་མ། ཉེ«པེ་»«སྣར་སྐཁ་བའི། «པེ་སྐོམ། «སྣར་»སྒོམ། ཁྱེ«པེ་»«སྣར་གསུམ་གྱི། «ཅོ་རྣམས། «པེ་»«སྣར་ཀྱིས། ཉེ«པེ་»«སྣར་ཡི། ཁྱེ«པེ་»《སྣར་»ཅེས། ཨེ«པེ་»«སྣར་མཛད། ཚེ«སྣར་»འཐོན། ཁ«པེ་»«སྣར་ཅིང་། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»+དོན་གཉེར། «པེ་»«སྣར་བླགས། «པེ་»«སྣར་»དོར། «པེ་»«སྣར་འཁོན། ཁོ«པེ་»ཅིང་། «སྣར་»ཅིད། @«པེ་ངང་། སྣར་»དང་། མ«པེ་»«སྣར་»ཏུ། «པེ་»«སྣར་གྲགས་གད་བརྒྱངས། ཚེ«ཅོ་འཚུབ། ཨེ«པེ་»«སྣར་»ཁོངས། །ཞེ«པེ་»འཁར། «སྣར་མཁར། འཉེ«པེ་»«སྣར་»མེ་དཔུང་མེ་ལྕེ། ལ«པེ་»«སྣར་འབར་བྱེད་ཤུགས་དག་འཇུག་པའི། «པེ་»«སྣར་»ལ། ན་:། ༄༅«པེ་»«སྣར་»འགྱུར་བའི་བརྩེ། «པེ་ཀྱི། །«པེ་»«སྣར་»གླང་། «པེ་»མདུངས། ཚུ«པེ་»«སྣར་པོ་བསྣུན་གསད། «པེ་»«སྣར་ཁྲི། «པེ་»«སྣར་»གཙེས། ཉི«པེ་»«སྣར་བཟས། «པེ་»«སྣར་»ཁྱོག་ཚེ«པེ་»«སྣར་»སྐྱོན། །སྐྱེ«པེ་»«སྣར་»སྒོལ། ཉེ«པེ་»«སྣར་»མཆོད།  ེ«པེ་»«སྣར་»བ་འཁྱིལ། «པེ་»«སྣར་»གདས། «པེ་»«སྣར་བ། ང་༽ཨེ«པེ་»«སྣར་»འཕྱར་བས། «པེ་»སྣར་སྔགས། «པེ་»«སྣར་སྤྲུལ་པའི་ལས་ཀྱི། «པེ་»《སྣར་»དག་གི། མ«པེ་»«སྣར་ཐག་པའི། «པེ་»«སྣར་»གཞོན། «པེ་»«སྣར་»ཀྱིས། ཚ«པེ་»«སྣར་»གིས་རྗེས་འབྲེལ། «པེ་»«སྣར་»ཡི། ཚེ«པེ་»«སྣར་+གོམས་པར་བྱེད་ལ། དགྱེཔེ་«སྣར་པདྨོ། (གེ«པེ་»«སྣར་སློངས། «པེ་»«སྣར་»ཡི། «པེ་»སྣར་»འདུན། «པེ་»«སྣར་གྱི། ཁ«པེ་»«སྣར་»གིས། «པེ་_།
