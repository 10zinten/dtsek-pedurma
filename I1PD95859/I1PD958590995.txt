རྒྱུད། ང་བློ་མས༔ཀུན་དུ་འཕྲོ་ཞིང་ཀུན་བསྲེག་མཛད། །ཞལ་གཅིག་ཕྱག་བཞི་མཆུ་སྨིན་གཡོ། །རལ་གྲི་མདའ་དང་འཁོར་ལོ་གཞུ། །གཅོད་འབིགས་གཤག་ཅིང་བསྣུན་བྱེད་་་ལས། ་ཎ་། བགེགས་དགྲ་ཕྲ་རབ་རྡུལ་མི་གནས། །གསེར་མདོག་ནོར་བུ་ཨིནྡྲ་དང་། །སྦུར་ལོང་འཕགས་པའི་གདུང་ཐལ་དང་། །བྲི་ཧ་ཏི་ཅདང་གུ་གུལ་་་དང་། །བདུད་རྩི་ལྔ་སོགས་སྦྱར་བའི་སྤོས། །རྣལ་འབྱོར་ལྡན་པས་བདུག་བྱས་ན། །སྟོང་ཕྲག་བརྒྱད་ཅུའི་མ་ལུས་བགེགས། །ནགས་མེས་ཤིང་སྐམ་ཇི་་བཞིན་འཇོམས། །རིམས་སོགས་དུག་སོགས་དེ་བཞིན་ནོ། །བསྒྲུབ་བྱའི་གཟུགས་ལ་ལྷ་མོ་ཡིསི། །ང་རྒྱལ་གྱིས་བཀུག་རྩུབ་མོའི་ཤིང༌། །རྟེན་ཕྲལ་་་ལིང་ག་གཏུབས་ནས་བསྲེགས། །བསྙེན་བསྒྲུབ་ལས་ལ་ྋགོ་རིམས་ཀྱི་བཞིན། །འབུམ་ཕྲག་སྟོང་ཕྲག་བརྒྱ་རྩ་བརྒྱད། །བཟླས་ཏེ་ཁྲོ་:མོའི་ཧོ་ཉཁུང་ཞལ། །ཕུལ་ནས་ྋགསུམ་བདུན་ཉི་ཤུ་གཅིག་། ཞག་གིས་ཚངས་པར་གྱུར་ཀྱང་གསོད། །ཨོ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་དྷེ་མ་ེདྷ་རེས་ཉྙཱཧཱུྃ། སྤོ་ཝེཊ་བྷསྨི་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། རྗེ་བཙུན་མ་སྒྲོལ་མ་ཡོངས་སུ་སྨིན་པས་དགྲ་བགེགས་གཞོམ་པའི་སྒྲུབ་ཏེ་ཐབས་ྋརྫོགས་སོ། །།།།ཕྱོགས་ལ་ཁ་ཕྱོགས་ན་འབབ་ཁ་ཁ་ཁ་ཁ་ཁའི་ཝ་ཎབ་བ་ན་གོ་བ་བསྐྱེད་དོ། །ལུས་བའི་ཕྱག་འཚལ་ས་གཞིའི་ངོས་ལ་ཕྱག་གིསགི། །མཐིལ་ྋགྱིས་བསྣུན་ཅིང་ཞབས་ཀྱིས་བརྡུངས་ཚིམ། །ཁྲོ་གཉེར་གཡོ་བའི་ཡི་གེ་ཧཱུཾ་གིས། །རིམ་པ་བདུན་པོ་རྣམས་ནི་འགེམས་མ། །འགུགས་བྱེད་སྣུན་བྱེད་རྡུང་བྱེད་མ། །ཡིག་མཆོག་འཛིན་རིམ་གཉིས་བདུན་འགེམས། །ལྷ་མོ་ཉིད་བསྙེན་གུས་པ་ཡིསི། །ཞེ་་་་སྡང་དང་བྲལ་ཁྲོ་བས་བསྒྲུབ། །སྟོང་ཟིལ་གནག་མདོག་ཞལ་གསུམ་མ། ཁོ། གཡས་དཀར་གཡོན་དམར་ཁྲོས་པའི་ཞལ། །ཐོད་པའི་དོ་ཤལ་རྒྱུ་མ་གསོལ། །།
