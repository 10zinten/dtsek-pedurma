བསྟན་འགྱུར། རྒྱུད། ༼ཤ་སངས་རྒྱས་རྣམས་ནི་ཐམས་ཅད་ཀྱི། སོགས་། གསང་བ་གསུམ་ལས་བྱུང་བ་ཡི། །་་། ཁམས་གསུམ་ཀུན་གྱིས་ཕྱག་བྱས་པ། ་། དབང་བསྐུར་རྡོ་རྗེ་ཆེན་པོ་བྱིན། །ཅོད་པན་རྒྱལ་བ་ལྔ་བསྟིམས་ཏེ། །མགོ་བོའི་གནས་ལྔར་དབང་བསྐུར་རོ། »། ཨོཾ་བཛྲ་དྷཱ་ཏཱི་ཤྭ་རི་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་སརྦ་སཏྭཱ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཨི། ཨོ་དྷརྨ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧཱི། ཨོཾ་སརྦ་བི་ཤྭ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོ་་་ཧཱུྃ་:ཧཱུྃ་ཧཱུྃ་ཨསྔགས་འདི་དར་དཔྱངས་དཔྲལ་བར་བཅིངས། །རིན་ཆེན་འབྱུང་་ལྡན་གནས་པས་སོ། །འོད་དཔག་མེད་པར་གནས་པ་ཡིས། །།སངས་རྒྱས་ཀུན་གྱི་རྡོ་རྗེའི་དབང་། །ཚེ་། དི་རིང་ཁྱོད་ལ་བསྐུར་བར་བྱ། ་ཌ། འདི་ནི་སངས་རྒྱས་ཐམས་ཅད་དེ། ་། ཤིན་ཏུ་གྲུབ་ཕྱིར་རྡོ་རྗེ་ཟུང་། ཀ་། ཞེས་བརྗོད་གནས་གསུམ་རེག་སྟེ་སྦྱིན། །ཨོ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧྲི༄༅། རྟ་མཆོག་ཏུ་ནི་གནས་པ་ཡིས། །ཨོ་བཛྲ་གྷཎྜ་ཨ་ཨགྲི་ཧི་སོ་སེ་བྷ་ག་ཝ་ན་མ་ཡེ་སནྣ་ནི་ཧི་ཏོ་བྷ་བ་ཏི། གཡོན་པར་་ཡོངས་སུ་གཏད་པར་བྱ། །རྟག་པར་རྡོ་རྗེས་གནས་པ་ཡིས། །རྡོ་རྗེ་དྲིལ་བུ་་་མགོར་བཞག་སྟེ། །ཨོཾ་བཛྲ་སཏྭ་ཧཱུྃ་ཨ་བྷི་ཥིཉྩ་མི་བཛྲ་ཨ་བྷི་ཥེ་ཀ་ཏ། ཁྱོད་ནི་རྡོ་རྗེ་ཆེ་གེ་མོ། །མེ་ཏོག་ལྷུང་བའི་རྗེས་འགྲོར་བརྗོད། །སློབ་དཔོན་དབང་སོགས་བརྗོད་ན་ཡང་། །ཀྱེ་ཡི་རྡོ་རྗེ་ཇི་བཞིན་བརྗོད། །དཀྱིལ་འཁོར་དུ་དབང་བསྐུར་བའི་ཆོ་ག་བསྡུས་པ་པ་ར་ཧེ་ཏ་པཎྜི་ཏས་སྦྱར་བ་རྫོགས་སོ། །ལ་ཕྱི། པ། ཁ་ཆེའི་པཎྜི་ཏ་སཾ་གྷ་ཤྲཱི་བྷ་དྲ་དང༌། །འཇམ་དཔལ་སྒེག་པའི་རྡོ་རྗེས་་་་་བསྒྱུར་བའོ། །།།།ཅང་ཤིའ་ཏཱ་། །ཚེ་ཚཙོ། ་བག༽བཀའ་བ་དང་། ཙཀ་ཁྲོ་ཏུང་ལ། པོའི་བཀན་བྱེད་ཕྱོགས་ཀ་བ་ཞུ་བ་ཨོ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཙ་ལ་པྲ་ཙ་ལ་ཤཱི་ྋགྷཾ་གཱ་མི་ནི་དེ་བ་ཏཏྟ་ཨུཙྪ་ཏྲ་ཡ་ཧཱུཾ། འདི་བྱ་རོག་གི་:སྒྲོས་ལན་གྱི་སུམ་ཅུ་རྩ་གཅིག་བཟླས་ནས་བསྒྲུབ་བྱའི་སྒོ་ལ་སྦས་པས་།
