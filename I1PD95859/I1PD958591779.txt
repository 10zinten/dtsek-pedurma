ནང་བསྟན་འགྱུར། རྒྱུད། ༼ཤ། བྱའི་ཕྱིར། །ལྷ་ཆེན་སྐུ་ཡི་ཕྱེད་དུ་བཞག། རྫོང་དང་དཀའ་ཟློག་ནག་མོ་ཆེ། །ལྷ་དང་ལྷ་མིན་ཕྱག་བྱས་པ། །།མི་ནད་གང་ཡིན་ཁྱོད་ཀྱིས་ཤེས། ་ཁོy། དེ་ནི་་འཇིགས་པ་མེད་པར་མཛོད། །གཤིན་རྗེའི་ལྕམ་མོ་བདུད་ཀྱི་ཡུམ། །འདོད་་་ཁམས་དབང་ཕྱུག་མ་:ལ་བསྟོད། ་། ྋནད་དོ་ཅོག་གི་བདག་མོ་དཔལ་:ལྷ་མོ་ནག་མོ་ཆེན་མོ་ལ་བསྟོད་པ་བརྒྱད་པ་ཞེས་བྱ་བ་བྲམ་ཟེ་:མཆོག་སྲེད་ཀྱིས་མཛད་པ་་་་་རྫོགས་སོ། །།།འཁོར་བའི་ཆང་ཁང་། ཤོག་ཀྱང་ཁ་ཚོག་པས་ཕྱོགས་དང་ਪਲਲੈਸਉਸਰਮਲਸਿੰਸਸਕੂਟਰཁ་ལ་ལ་ལ་་་་བསྡུར་མཆན། ཁཎྜ་ཀ་ང»ཁ་ཁ་ཏུ་མི་ང་་། །*། དཔག་བ་Pr་མ་་གས་Tཁ་ལྟའི་༼ཁ་ས་ཚེ་། ze། །/ཀོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་ལྷ་མོ་་་་ནག་མོའི་བསྟོད་པ་བཞུགས་སོ། །ཚེ«པེ་»«སྣར་»བི། ཚེ«པེ་»«སྣར་+། «སྣར་སྒྲོག«པེ་»«སྣར་གཅིག་ཀྱི་ཡི་དར་ཚོ་པ་«པེ་»«སྣར་གསལ་ཞིང་ཞི་ལ། ་ཚེ«པེ་»«སྣར་»ཁྱོད་ནི། «པེ་«སྣར་»བོའི། ༄༅«པེ་«སྣར་»འཕྲེང་། «པེ་»«སྣར་བཤིབས། «པེ་པ། ༄༅«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»འགྲན། «སྣར་བསྐྱོང་། «པེ་»«སྣར་རྟབས། གྱི«པེ་»«སྣར་»མཐོན། ཉེ«པེ་»«སྣར་»འཇུག་བཏུལ། «པེ་»«སྣར་»མགོ་བོའི་གདེང་། ཁྱེ«པེ་»«སྣར་བན་དུས། «པེ་བཞགདེ་«པེ་»བཞུགས། «སྣར་»གཞུགས། «པེ་»«སྣར་»མིང་། «པེ་»«སྣར་»ལ་བསྟོད། །༼«པེ་»«སྣར་»—ལྷ་མོ། ཚེ«པེ་»«སྣར་»མཆོ་སྲེད་ཀྱི། འཁྲི་༼POལ་ཉེབའི་ཟླ་། ལ་ལ་ཁ་བ། འབྲི་ཏ། pལྷ་ཚདག་ཉིད་ཀྱི་ངང་ལོ་་ར་། དེ་ལ་:ཁ་ག་ཆ། དབྲ་ཡིས་དཀཚིnལོབར་དུ་བདག་ཀྱང་། པད་དཀྱིཅཏི་ཙ་ལ། ཙི་ཎིཔ་ས་ཆ་བློ་ལ་ཡོད་བོད། Sཏཱཀ་དང་རྗོཊ་ཚོད་ཙ་བྱི་ལ། ད་ལ་་སཏེ་ཚོང་པ་ཆེན་པོ་ཞུ་བ་ཙམ་དང་་་་་། ་་ན་ནི་ར་བ། བྱང་ན་བ་དང་འཆི་།
