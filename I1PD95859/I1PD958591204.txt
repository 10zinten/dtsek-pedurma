སེམས། །སྲོག་གི་ས་བོན་ཧཱུཾ་ཡིག་བལྟ། །སྣ་ཚོགས་འོད་ཟེར་འཕྲོ་འདུ་ཡིས། །འཕགས་ཚོགས་མཆོད་ཅིང་འགྲོ་དོན་བྱས། །བཅུ་གཉིས་མཛད་པ་མཐར་ཕྱུང་སྟེ། །གང་འདོད་ཞིང་དུ་འཚང་རྒྱའོ། །གསོན་ཚེ་སྲོག་རྩོལ་ལ་བརྟེན་ནས། །ཐིམ་པ་བདུན་ནི་རིམ་བཞིན་བསྒོམ། །འཆི་ཀར་དངོས་སུ་ཐིམ་འགྱུར་བས། །མ་བརྗེད་ཙམ་གྱིས་འགྲུབ་པར་བཞེད། །རྗེ་བཙུན་ཨཱརྱ་ཏཱ་རའི་མན་ངག་ལ་བརྟེན་ནས་འཆི་ཀ་མའི་ཉམས་ལེན་དཔལ་ཆོས་ཀྱི་རྗེ་སའི་སྟེང་ན་འགྲན་ཟླ་དང་བྲལ་བའི་ཁ་ཆེ་་་་ཤཱཀྱ་ཤྲཱི་བྷ་དྲའི་ཞལ་སྔ་ནས་ལོ་ཙཱ་བ་བྱམས་པའི་དཔལ་གྱིས་བསྒྱུར་བའོ། །།།ཉ་ཞེས་བརྗོད་པར་བྱའོ། །ག་དང་པ་པང་ག་་བ་འག་པ་ཉིད་ཀྱི་འོ་བ་ལ་::་པའི་ཡི་གེ་ར་་འོད་བསྡུར་མཆན། ན་བ། ལུད་པ་ད་ཡུ་ཏིལ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»«སྣར་བལྟོས། མི«པེ་»:«སྣར་ནས། ཁྱི«པེ་»«སྣར་བཏུལ། ཚེ«པེ་»«སྣར་»གི། །་«པེ་»«སྣར་»ཏཾ། %པེ་»ཙ། ་་་་་ཆ་བློ་ལ་དེབ་ཏུ་སpཔz་བ། ཚ་། ང་ལུགས་ལ་ཆ་ཆ། འཁཊི། ཏོབསྡུར་འབྲས་རེའུ་མིགབགོ་བག་མའི་ཀ་འཁྲུངོ་འབའ་གས་ག་ཟླ་སྒྲ། བློ་ར་བ་ཚོདཆརྒྱུད། (སར་ཐང་། རྩམ་སྣར་ཐང་། རྒྱུད། །།རྒྱུད། དདཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྡེ་ཚན། རྒྱུད་འགྲེལ། །པོད་རྟགས། ཚན་གྲངས། ::::ཙོ་ལྡེབ། ནབ་། བན་ཕྲེང་། །་་ཚིག་རྐང༌། ་། ཚེག་ཁྱིམ། ར། ན་བ། ན་བ་།
