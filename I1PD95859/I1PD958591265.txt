བསྟན་འགྱུར། རྒྱུད། ན་ཡོད་མེད་པ་དང་། ད་བར་དུ་བདག་ལ་ན་བྱས་པ་དང་། ཁྱད་པར་ཁོ་དང་ཁྱོད། །བྱ་བྱེད་དུས་པ་ཡིན་ལ། དྲ་བའི་པར་བསྡུར་འབྲས་རེའུ་མིགཚོད་འཚོ་བ་འཚོལ་རྒྱུད། （བུ། འདཌ:ལྟ་ཀདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། 《རྒྱུད། བརྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། བཙབས། ན་བ། ན་_ན་བ། བ། ཕྲེང་། ཚེག་ཁྱིམ། །།་། ་པ་ལས་བལ་ཡུལ་ནང་། བདེ་བ་དེ་ལ་ཡ་ཀྱི་བྱེད་པ་དང་དུས་དང་། ་་་་་་་་་པའི་གདུང་བ་རྒྱལ་བ་གཡུ་ཁྲི། དང་། མད:ཚུ་དེ་་བསམ་་བ་བའི། ་་དྲ་༼ཏེ། །aཎ་ཆp：་ར་་ཀྱང་《ཏུ་nདུཁ་བ་བབས་རྗེཡིན། ཀ་ལུ། བ་ཀ་དག་དང་། སྐྱོ་བ་དང་་་་་་། སླད་མི་ས་ཁ་ཡི། དྲ་བ་《་། ཁཀ་མ་་་་་་། འོད་ལ་ཁོང་ཁ་ཎ་ཊཱཎ་(སྤྱི་། ཟཎཁ་ཊི་ད་དཆུ་ཁཟླ་ཆ་ཎཱ་ཀའོ། །་ལྔ་ཎ་ན་ང་ཁ་ཎཙོ་འཁ་ཁ་ཁ་ཁྱུ་ཆ་།
