མཆོག་ཡིན་པས། །དེ་ཡིས་རྗེ་བཙུན་ཡུན་རིང་བསྟེན། །སྒྲུབ་ཐབས་བཀོད་པའི་ཚོགས་དེ་ཡིས། །དེ་ཉིད་མཐོང་བས་ཕྱག་བྱས་ཏེ། །བླ་མ་དམ་པ་ལྷན་་་ཅིག་སྐྱེས། །མཉེས་པར་བྱས་ཏེ་དེ་མཐོང་ཤོག། རྗེ་བཙུན་མ་འཕགས་མ་སྒྲོལ་མའི་ལྷ་བཅུའི་སྒྲུབ་པའི་ཐབས། སློབ་དཔོན་ཆེན་པོ་ལྷན་ཅིག་སྐྱེས་པའི་རོལ་་་་་པས་མཛད་པ་རྫོགས་སོ། །།།།།འབྲོག་མི་ལོ་ཙཱ་བས་བསྒྱུར་བའོ། པཉྩཀཟླ་ཁའིཎིཀྱིཥབསྡུར་མཆན། ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་མ་སྒྲོལ་མའི་དཀྱིལ་འཁོར་གྱི་སྒྲུབ་ཐབས་ཆོ་ག་བཞུགས་སོ། །ཉེ«པེ་»«སྣར་ཨཱརྱ། ཚེ«པེ་»«སྣར་བྱའོ། །«པེ་»«སྣར་བྷ། «པེ་»«སྣར་»། «པེ་»«སྣར་བྷ། བྱང་དུ། །ཨེ«པེ་»«སྣར་»རཾ་ཡཾ། «པེ་»«སྣར་»—། «པེ་»«སྣར་—ཞུ། «པེ་»«སྣར་»ཀཾ། «པེ་»«སྣར་»རེ་པོ། «པེ་»«སྣར་སྐོང་། །ཚེ«པེ་»ཡིས། @«པེ་»བསྒོམས། «པེ་»སྣར་+མ། «པེ་»«སྣར་—པར། ཚེ«སྣར་»རྒྱན། «པེ་»«སྣར་»གུ། «པེ་»«སྣར་བྱའོ། «པེ་»«སྣར་—ཉ། ཉི«པེ་»«སྣར་པུ། «པེ་»«སྣར་»—། «པེ་»«སྣར་»ཀྱིས། @«པེ་»«སྣར་ཏཱ། @«པེ་»«སྣར་མི། ཚེ«པེ་«སྣར་»—ཉ། ཉི«པེ་»ཏུ་སྟེ། «སྣར་»ཏུ། «པེ་»«སྣར་»ཉ། ཁྱེ«པེ་»«སྣར་ཨཱརྱ་བ། འགྱེ«པེ་»«སྣར་»དུན། ཉེ«པེ་»«སྣར་»ཀྲི། ཚེ«པེ་»སྣར་»བྷ། «པེ་»«སྣར་ཧཱ། ཚེ«པེ་»«སྣར་བན་དྷཱ། ཨེ«པེ་»«སྣར་»དུཋན། «པེ་»«སྣར་བྷཱ། ཚེ«པེ་»«སྣར་ཟན། ལས་ཀྱི་མ་ཡིན་ཏེ། ཚེ«པེ་»«སྣར་རིངས། «པེ་»«སྣར་»ཙ། །ཏིང་འཛིན་ཕྱག་རྒྱ་ཡང་དག་བྱབས་། འདི་ནི་ཐམས་ཅད་དེ་ལ་འདྲ་ཉེ་བར་ད་པ་མང་བ་དེ། །།
