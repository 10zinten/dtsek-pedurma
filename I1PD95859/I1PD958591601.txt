་བསྟན་འགྱུར། རྒྱུད། *ར་ཀའི་བརྗོད་ནས་ལས་ཐམས་ཅད་པའི་སྔགས་བཟླས་སོ། །བདུད་རྩི་ལྔ་དང་བཅས་་་་པའི་དྲིའི་ཆུས་དཀྱིལ་འཁོར་གྱི་ཕྱི་རོལ་རྣམས་ལ་གཏོར་ཏེ། ཨོཾ་ཨཱཿཧཱུྃ་བཛྲ་མུཞེས་པ་བརྗོད་ཅིང་ས་ཚོན་ལྷག་མའི་ཡེ་ཤེས་པ་གཤེགས་ནས་སྣོད་དུ་བླུགས་ཏེ་དབེན་པའི་གནས་སུ་གཞག་ྋགོ། །འདི་ནི་ཚོན་གྱི་ཆོ་གའོ་། :། གལ་ཏེ་ཁ་བ་བོད་ལོ་ཥ་ཎ་npen'bརྒྱུ་ཎ་བས་མས་་་དེ་ཁོ་ན་ཁ་དེ་ནས་སྔོན་དུ་ལྷག་གནས་བྱས་པའི་བུམ་པ་རྣམས་གཡས་:སྐོར་གྱིས་དགྲམ་པ་ནི་དེ་རྣམས་ཀྱི་ནང་ནས་གཙོ་བོ་དང་རྣམ་རྒྱལ་གྱི་བུམ་པ་གཉིས་དཀྱིལ་འཁོར་གྱི་འཁོར་ལོ་ལ་:གཡས་སྐོར་བྱས་ནས་མེ་རིའི་ཕྱི་རོལ་གྱི་ཤར་སྒོར་སྟན་ལ་་བག་ཙམ་གཡས་སུ་བཞག། གཡོན་དུ་རིམ་པ་བཞིན་དུ་གཏུམ་པའི་དབང་ཕྱུག་མ་དང་ལས་ཐམས་ཅད་པའི་བུམ་པའི་ཆོམ་རྐུན་མ་སྟེ་གཉིས་གཡས་སྐོར་བྱས་་ཏེ་གཞག་གོ། །ལྷོ་ཕྱོགས་སུ་གཡོན་དུ་ཙརྩི་ཀ་དང་གཡས་སུ་རི་མོ་ཅན་ནོ། །ནུབ་ཕྱོགས་ཀྱི་གཡོན་དུ་དུས་ཅན་མ་དང༌། ལ་ལ་གཡས་སུ་དགའ་བ་ཆེན་མོའོ། །བྱང་ཕྱོགས་ཀྱི་གཡོན་དུ་རིགས་ཀྱི་དབང་ཕྱུག་མ་དང་། གཡས་སུ་དགའ་བའི་་་དབང་ཕྱུག་མའོ། །ཆོམ་རྐུན་མ་ལ་སོགས་པ་བཞི་འགའ་ཞིག་གི་°མཚམས་སུ་་་འདོད་དེ། དེ་ལྟར་ན་མཚམས་བཞིའི་བུམ་པ་རྣམས་དང་། མདའ་བཞི་ལ་་་དར་བཏགས་པ་རྣམས་ལས་ཐམས་ཅད་པའི་སྔགས་ཀྱིས་གཟུགས་སོ། །འདི་ནི་་བུམ་པ་དགྲམ་པའི་ཆོ་གའོ། །།།sཝ་ཎ་པ་ཏཏད་དེ་ཡང་ལས་། ་ཚེ་ང་ཚོ་ཊཀ་དག་གཏོང་བ། བ་། བ་བློ་འཛིན་ཅཅུ་%peཀཆདེ་ནས་སློབ་དཔོན་དྲི་ཞིམ་པོས་ལུས་ལ་བྱུགས་ཏེ། ཌདར་གྱི་གོས་དང་་་་ཐོད་ལ་རྒྱན་:བཏགས་པ་གོན་ཁྱིལ། ་༼ལས་ཐམས་ཅད་པ་ཅིབཟླས་པའི་ཆུས་དཀྱིལ་འཁོར་བསྔགས་པ་དང་། ཡི་གེ་གསུམ་དང་ལས་ཐམས་ཅད་པའི་སྔགས་ཀྱིས་ཉི++།
