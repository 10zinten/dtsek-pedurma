ལ་ཏ། བཛྲ་ནཱ་ག་གུ་སུ། པེབཛྲ་ཡ་མ་:ཎི་བྷ་དྲ་ཡེ། བཛྲ་པུརྞ་:བྷ་དྲ་ཡེ༑་བཛྲ་ད་ན་དྷ་ཡ། བཛྲ་མ་ཧཱ་དྷཱ་ཛན། བཛྲ་ཛི་བི་ཀུ་ལནྡྷ་ཡ། བཛྲ་ཛོ་བྷ་ལ་མུ་ཁན་ཡ། བཛྲ་ཛཾ་བྷ་ལ་:ཛ་ལེནྡྲ་ཡ། བཛྲ་ཀི་ལི་མི་ལི་ནི། བཛྲ་ཨག་ཡེ་ ོ་་བཛྲ་ནིརྀ་ཏི། །ཎ་བཛྲ་བི་རཾ་བ། བཛྲ་ཨཱི་ཤ་ན་ཡེ། བཛྲ་ནཱ་ག་ཨུ་ལི་ཀ། བཛྲ་ནཱ་ག་པ་ལ་ཏི། བཛྲ་སུརྱ་བ། ་ཁྱབཛྲ་ཙནྡྲ་བ། བཛྲ་ཁཾ་ཙིཏྲི། བཛྲ་པྲ་ཏི་བི་དེ་བ། ་བཛྲ་ཌཱ་ཀི་ནཱི་ཀཱ་ལ་བཛྲ་བྷོ། བཛྲ་ཌཱ་ཀི་:ནཱི་ཀཱ་ལ་རཏྭཱ་བྷོ། བཛྲ་ཌཱ་ཀི་:ནཱི་ཀཱ་ལ་པདྨ་བྷོ། །བཛྲ་ཌཱ་ཀི་:ནཱི་ཀཱ་ཎིལ་ཀརྨ་བྷོ། ་ས་:པཱ་རི་ཝ་ཀིར། །ཨི་དཾ་དྷུ་པ། པུཥྚེ། །དཱི་པཾ། གནྡྷཾ། །་་:ནི་བི་ཏེ། དི་སན་ཀ་ཏཾ། པ་ལཱི་པྲ་ཧ་རཾ། པྲ་ཏིཙྪ། ཡུ་བ་བུ་ཛ། ན་མ་སརྦ་སཏྭཱ། །ནན་བ་ཏི། རཏྱཱི། སུ་བཎ། འཁདྷ་ན་དྷ་ནྱ། རྗེ་ཨ་ཡུ་ཡོ་བ་ན། ཨ་རོཀྱ། །སད་སུ་ྋཁཾ། ཨ་པ་ཧ་ར་ཀཾ། །་ཤཱནྟི་ཀུ་རུ། པུཎྚི་ཀུ་རུ། ་ཚཔ་ཤཾ་ཀུ་རུ༄༅། མཱ་ལིར་ཡ། སརྦ་ཀརྨ་ཨརྦ་སིདྡྷི་ཧཱུྃ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོ་ཐམས་ཅད་གཉིས་མེད་ཐུགས་རྗེའི་མཆོག། རྡོ་རྗེ་གསང་གསུམ་གནས་ལས་བྱུང་། །དམ་པའི་དཀའ་སྤྱོད་བྱང་ཆུབ་སེམས། །དྲག་པོའི་གཟུང་འཛིན་མཁའ་འགྲོའི་ཚོགས། །དམ་ཚིག་བསྟན་པ་གནས་བསྲུང་ཞིང༌། །རྣལ་འབྱོར་སྒྲུབ་པའི་ལས་མཛད་མ། །དོན་ཀུན་ཕྲིན་ལས་སྔགས་ཀྱི་་མཐུས༑ཁ། འགྲོ་བའི་གནས་ཀུན་རྣམ་པར་དག། བྱང་ཆུབ་སྤྱོད་པ་རྒྱ་མཚོའི་ཚོགས༑། ཀཟླ་། དམ་བཅས་ཕྲིན་ལས་སྣ་ཚོགས་ཀྱིས། ང། སངས་རྒྱས་ཐམས་ཅད་མཉམ་སྦྱོར་བའི། །མཁའ་འགྲོ་སྒྱུ་མའི་དོན་སྟོན་མཛད། །ཨོ་ཤིན་ཏུ་དྲག་ཆེན་སྤྲུལ་པའི་སྐུ། །ཁམས་གསུམ་གནས་ཀུན་གཡོ་མཛད་པ། །ཀཾ་ལས་སྤྲུལ་པའི་ལྷ་ཆེན་པོ། ཆ། དཀར་སེར་ཁྱུ་མཆོག་གདན་ལ་བཞུགས། །ཞལ་ཕྱག་མཚུངས་་པར་:ཏྲི་ཤུ་ལ། །།དགྲ་སྟ་རལ་གྲི་ཐོད་ཁྲག་ཤ། སྤྱན་གསུམ་རལ་པ་ཐོད་ཕྲེང་།
