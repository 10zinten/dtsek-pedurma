བསྟན་འགྱུར། རྒྱུད། ཤའི། ད་ཚེ་ཀ་མ་བ་དོན་དུ་ལོག། །།།ཟླ་བ་། ཁ་བས་བདེའོ་ཞཌ་ཌ་འི་ཊ་ཀ་ཏ། ཁ་་་་་། །aaz། ཌོ་ལར་ཌཆ་ཆ་ཁཚོ། །།།པ་དང་། བྱེ་བ་དང་བསྡུར་འབྲས་རེའུ་མིག་་་་ངོ་། །ལ་ལ་ལ་་་་བློ་རྒྱལ་བ་བྱ་རྒྱུ། །ཚེ་་་རྒྱུད། （#ཀ་འཊི། །ཀ༌ཁ་ཆ་ཎཚོ། དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། །ཙ་ནེ། ཀྱ་་སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། །ཀ་རྒྱུད། དུ་བརྒྱུད། ས་ལ༌པོད་རྟགས། །ཚན་གྲངས། །ལྡེབ། བརབ་། ནབསབཙནབ་—བ་ཕྲེང་། །ཚིག་རྐང་། །།།།ཚེག་ཁྱིམ། །།།།་ངsའཔགྲོ་བར་ག་དུས་བབ་ལ་། ཀུན་དུ་རུ་ཀྱི། །ཀོཀ་ཀ། ཀ་ཀློག:ཁོ་བཏུདེའི༌པ་ཚོ་ཀུ་ཁཆཅརྒྱང། ལ་དམ་དུ་གཏད་ནས་བར་ལོས་བྱེད་ཀྱང་། སྐྱེས་པའམ་:མ་་་་་་རྒྱསྐྱེ། prpན་མི་ཏུ་ངོབ་ལ། ནད་པ་བདག་ལ་དུ་བ་ལ་དབུལ། །དེ་ནས་ལ་བབས་འཁོར་ལོ་ལས། བ་ལུ་དང་། ལྟེ་བ་དང་དང་། དེ་དག་ལ་འབབ་ན་བས། །རྩིབས་པ་ཉིས་བྱེ་ལ་པ་དེས་ད་བར་ལ་ཁྲི་བ་འབབ་བམ། །འབད་བ་མེ་དབང་འོད་ཁོར་ལོ་ལ་ནན་གཟུར་པ་བབ་པས་གསོད་པར་ལས་འདེས་འཁོར་ལོ་བསྐོརད་ར་ད་ར། བ་ལ་ཁྱེད་དད་ལ། པོ་།
