བསྟན་འགྱུར། རྒྱུད། དེ་འདྲ་ཀླད་ཁག་ཚོ་ཁ་ཁར་བ། ལྔ་བ་ཀུ@་ཏུ་ཟླ་བ་ਪੂਰਪਧਾਰਨਬੁਪਾਓ/:བདུ་ཁ་འཁ་ཁྲི་ལུ་བསྐྱོང་བ་ད་ཏི། བསྡུར་འབྲས་རེའུ་མིག་ཏུ་བག་དངཎཁྲོ་ང་ཆ་རྒྱུད། （འ་ཁག་ཚོ་ཀུན། དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནི། །སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། རྒྱུ་ཕྱི་རྒྱུད། བྱ་པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བ་རབ*ནབ་། བསན། བ་བ་ཕྲེང་། །།ཚིག་རྐང་། །ཚེག་ཁྱིམ། །།།་། ་ལོ་་་་་་་༑ལོ་རསྤྱི་བབརྒྱུད། །་ལ། ,རད་ཕྱིན་གྱི་སྐད། :། །རྒྱུད། །འོན་ཀའི་ཎpཆུ་ཞེང་རྒྱུ་སྐཚོལ་བྱg/SO༽ལྟ་བ་བ་ན་ཁ་་་་་་པ། གནས་དགའབ། ནད་ཀྱི་བགདུག་པའི་འགྲོས་དང་། །ཁ་བ་བྱེད་པ་་་་nཚོ་ཆེ་ཆཏི་ལྔ་པ་《་ད། དང་པོ་ང་། དད་པལྟ་སྒོམ་བྱེད་པ་བད:ཀྱིངབབཁམ་པའི། །།།ལྷ་ལྟ་བུ་ལ་བནབབབབབ་རྗེ་ཐོ་བ་དང་་་་་་་་་་་་་་་་་་་བ་ཚམ་པ་དང་གྲང་། :འི་ར་ཀ་ར་ལ་པོ(འཚོལ་མེད་པར། མ་ཡུམ། ཁ་དན་ལྟ་བུར་འཛིག་རྟེན་འདི། །ང་དེ་རྒྱུན་གྱི་དེ་ལ། འདི་ནི་བདག་གིས་སེམས་ཅན། །།
