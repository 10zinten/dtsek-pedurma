རྒྱ་གར་གྱི་མཁན་པོ་ཨ་ཙཱརྱ་ཐདྡྷ་ཀ་ར་ཝརྨ་དང༌། དྷརྨ་ཤྲཱི་བྷ་དྲ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་་་ཕབ་པའོ། །དབ། །ག་ད་བ་དང་ཀ། དབང་བ་དང༌། །ཀ་བ་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་སངས་རྒྱས་མཉམ་སྦྱོར་གྱི་དཀའ་འགྲེལ་བཞུགས། «སྣར་»༄༅། །དཔལ་སངས་...བཞུགས་སོ། །གྱི«པེ་»«སྣར་»མཐཱ། ན«པེ་»«སྣར་དབྱེ། །«པེ་»«སྣར་ཀྱིས། ཉི«པེ་»«སྣར་ལོ། «པེ་»«སྣར་གཉིས། ®«པེ་»«སྣར་»སུམ་བརྒྱ། @«པེ་»«སྣར་བརྗོད། །ཅེས་དང་། ར་ཚེས་ཀྱི་མིང་ཡང་ཀྱི༽འུ(«པེ་»«སྣར་དཔས། «པེ་»«སྣར་དག«པེ་»«སྣར་བའོ། ཚུ་ངོསོ@«པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་ཀྱི། །བ། ཁོ་ཚོ། 《«པེ་»«སྣར་ཀྱིས། «པེ་»འདི་ཡིས། «པེ་»«སྣར་རྒྱུ། ཀྱི་«པེ་»«སྣར་+དགཚེ«པེ་»《སྣར་»སྤྱོད། །ལ་ལ་བྱས་ཀྱི%E། ལཀྱིRཚེ«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་+སོ། O«པེ་»«སྣར་བསྟན། །ད་ཀྱི་«པེ་»«སྣར་དང་ཀུན་ཏེ། «པེ་»«སྣར་བཞགའ«པེ་»«སྣར་བསྟན། «པེ་»གྱིས། མུ«པེ་»བསྒོམ་པ། «སྣར་བསྒོམ་པ་ནི། །བན་། བོད་ཀྱི་«པེ་ཐུགས། «པེ་ཀྱི། «པེ་ལོགས། «པེ་»ཉེས། «པེ་»རྟོགས། (ཚེ«པེ་»«སྣར་»ལོ། «པེ་»པོ་བ། «པེ་»གྱིས། «པེ་»སྣར་»དང་། «པེ་»《སྣར་»སྟོང་། «པེ་»«སྣར་བར། །ཚེ«པེ་»«སྣར་»+བདེ། ྋ@«པེ་»«སྣར་ཀྱི། «པེ་»གྱིས། ཚུ«པེ་»《བ། «སྣར་»དགོངས། དཔའ་«པེ་»«སྣར་»གི། «པེ་»«སྣར་»སྒྲ། ཉི«པེ་»གྱི། །«པེ་»«སྣར་»ལྟ། ཚུ«པེ་+དང་། །R《་«པེ་»གྱིས། «པེ་»«སྣར་—པ། ཉི«པེ་»སྣར་བསྒོམ། །དྲག་པ་༽«པེ་»«སྣར་—དེ། ཚེ«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་—སྐུ་དང་། &«པེ་»«སྣར་+སྐུ་དང༌། །
