ལུ་ཀ་ན་ཏོ་ཚ་ཉོ་སྐུ་ལ་mdsབྱ་བ། །ས་ག་དང་ནི་རྐང་གཅིག་ཡིན། ་། ཨོཾ་ཧཱུྃ་ཧཱ་མཱ་?ར་དུཏྟཾ''ག་ཏ་ཡ་གྷ་ཏ་ཡ། ཨ་ཨ་ཨ་ཧཱུྃ་ཧཱུྃཕཊ། །བྱང་ཕྱོགས་ཀྱི་སློབ་དཔོན་གྱི་སྔགས་སོ། ་་། ཕྱོགས་དེ་དག་ན་གནས་ནས་ནི། །བློ་དང་ལྡན་པས་སྔོན་བཞིན་བརྗོད། །:སྦྱང་བ་ལ་་སོགས་རིམ་གྱིས་ནི། །གཏོར་མ་ལྔ་ནི་དབུལ་བར་བྱ། །ཨོཾ་བཛྲ་ཨཱ་ར་ལི་ཧོ། ཛཧཱུྃ་བཾ་ཧོ། བཛྲ་ཌཱ་ཀི་ནཱི་ས་མ་ཡ་སྟོ། དྲྀ་ཤྱ་ཧོ། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་:ཁཱ་ཧི། སརྦ་:ཡཀྵ་རཀྵ་ས་བྷཱུ་ཏ་པྲ་ཏ་པི་ཤ་ཙ་:ཨུ་སྨནྡ་ཨ་པ་སྨ་རི། །ཚོ་ཊཱ་ཀི་ནྱ་ད་ཡ་ཨི་མཾ་བ་ལིཾ་གྲིཙྪནྟུ། ས་མ་ཡ་:སིདྡྷི་མེ་པྲ་ཡཙྪནྟུ། ཏུ་ཡ་ཐི་ྋཐཾ། ཡ་ཐེ་:བྷཾ། བྷུཉྫཐཱ༑། ཨ་བྷ་བནྟུ་མཾ། མ་ཏི་ཀྲ་མ་ཐཏི། ་མ་མ་:སརྦ་ཀ་ར་སཏྨ་ཁ་བི་ཤུདྡྷ་ཡ་ས་ཧ་ཡི་ཀ་བྷ་བནྟུ། །ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ལ་ཞེས་བྱ་བ་ནི་དཀྱིལ་འཁོར་གྱི་སློབ་དཔོན་གྱི་གཙོ་བོའི་གཏོར་མའི་སྔགས་སོ། །ཡ་། ཨོ་ཨཱཧཱུཾ་ཧོཙྪབྷ་མི་བསླུ་ཨ་་་་ཧི་:སཱ་རཱི་ཨིནྡྲ་ཨ་དྷི་ས་པ་:རི་བཱ་རི། །ཨི་དཾ་བཱ་ལི་གཎཾ་:པུཊཾ་དྷུ་པཾ་དཱི་པཾ་ཨཀྵ་ས་ཏཾ་དཱ་ད་མ་:ཧེ་དེ་བཱི་ཨཱ་གཙྪ། ཁ་ས་པ་རི་:བཱ་རའོ། །༼ཨི་དཾ་བ་ལིཾ་:གྲི་ཙྪནྟུ་ཁ་ད་ཆུ་པི་བནྟུ་ཛཧཱུྃ་བཾ་ཧོ༔ས་:ན་ཏྲི་པ་ཏ་སརྦ་སཏྭཱ་ནཾ། ་་བ། ཤཱནྟི་པུལླི་རཀྟ་ཚིབ་ར་ཎ་གུ་:ཧཱུཾཀུརྦནིུ། ཧཱུཾ་ཧཱུཾ་ཕཊ། སློབ་དཔོན་གཉིས་པའིའོ། །ཨོ་ཨཱིཧཱུྃ་ཧོརོ་ཧཱུྃ། ཀ་ར་ཨ་ཀི་ནི། ཙནྡྲ་:སཱུརྱ་བུདྡྷ་མཾ་ག་ལ་སུ་ཏྲ་བེ་ཧེ་ས་པ་ཏི་ཀི་དུ་ས་ནི་སཿཞེས་བྱ་བ་ལ་སོགས་པ་ལྷོ་ཕྱོགས་ཀྱི་སློབ་དཔོན་གྱི་སྔགས་སོ། ༉་། ཨོ་་ཨཱཧཱུྃ་ཧོ། ཛ་ཡ༄༅། །བི་ཛ་ཡ། པདྨ། ཀརྐ་ཊས། བ་སུ་ཀྱི། ་་:ཤཾ་ཁ་པ་པོ་ལ། །ཀུ་ལི་ཀ། ཨཱ་ནནྡ་། ཏཀྵ་ཀ་། མ་ཧཱ་པདྨེ་ཧྥུ། །ཞེས་བྱ་བ་ལ་་་སོགས་པ་ནི་ནུབ་ཕྱོགས་ཀྱི་སློབ་དཔོན་གྱི་ཀླུའི་གཏོར་མའི་སྔགས་སོ། ཁ། ཨོྃ་ཨཱ༔ཧཱུྃ་ཧོ༑བྷེ་ཝ་ད་ལ་ཀི་ཀེ་ཏ་མུ་ཤཱ་ཀ་ཀ་སུ་ཀ་ར་གྲེ་ཏ་བྱ་ཀླུ་ཨ་ལུ་ཀ་ཛཾ་བོ་ཀ་གྷ་རུ། དྷ་མོ་ཁྱེད་བྷོ། ཡོ་ཞེས་བྱ་བ་ནི་འབྱུང་པོའི་གཏོར་མའི་སྔགས་སོ། །།འ་། དེ་ནས། —།
