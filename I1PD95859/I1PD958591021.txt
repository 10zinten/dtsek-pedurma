རྒྱུད། ཤེ་ཙཱ་བ་བློ་ཚོ་ཏཔོ་ཁད། འབུ«སྣར་»ཟུར་ཁྲོ། ཨེ《སྣར་ན། ཨེ«སྣར་»ཟད། ཨི«སྣར་ཁྲོ་བས། ཚེ«སྣར་ཕྱག་ནི་གཡོན། །མ«སྣར་རྒྱུན། «པེ་»གིས། ཚེ«སྣར་པས། དགྱེ«པེ་ལ་གྲོལ«སྣར་ལས་ཐར། །ཚེ«སྣར་ལ་བ། སྦྲུལ་དང་། བར་མ་ལ་ལྟ་བའི་བདུད་། །ཆ་ཡིས་ཀྱི་ཁང་afd@reg་ཟླ་ཚོre«སྣར་འགུར་སྒྲུག་པ། «པེ་»«སྣར་དམ། @«སྣར་གྲུབ། «སྣར་ཀྱི། འདོད། «པེ་»གྲུབ་པར་མཛད། 《སྣར་བསྒྲུབ་པར་བྱེད། «སྣར་+མཆོག་གི་དངོས་གྲུབ་ཐོབ་པར་འགྱུར་ཞིང་། །ཚེ«སྣར་»ནས། «སྣར་»མ། «པེ་»འདྲ་བ། ཆེ«སྣར་»དེ། ཀྱེ«སྣར་བརྩོད། ཉེ«སྣར་»དང་བྲལ་ནས། །མྱུར་བར་རྫོགས་པའི་སངས་རྒྱས་ཐོབ་སྟེ། །སེམས་ཅན་མ་ལུས་པ་ནི་ཀུན་ལ། །«པེ་»འབྱུང་བའི་ལེའུ་ལྔ་བརྒྱ། «སྣར་བྱུང་བའི་རྒྱུད་ལེའུ་ལྔ་བརྒྱ། «སྣར་»ལས། ཚེ«སྣར་»—ལྷའི་རྣམ་་་་་གྲངས་བརྟན་པའི་ལེའུ་...ཉི་ཤུ་རྩ་གཅིག་ལ། ཚེ«སྣར་བཏུས། ཁྱེ«སྣར་»+ས་ཆེན་པོ་ལ་གནས་པའི། «སྣར་»ཉི་མ་སྦས་པའི་ཞལ་སྔ་ནས། ཚེ«པེ་»«སྣར་»—རྒྱ་གར་གྱི་མཁན་པོ་...ཞུས་ཏེ་གཏན་ལ་ཕབ། །དང་དེ་ཁ་རྒྱུན་འབབ་པ་བསྡུར་འབྲས་རེའུ་མིག་ཏུ་ཆུང་ང་ཡི་བརྒྱུད། （ཁ་གས། །ར་::དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། ་སྣར་ཐང་། /ཅོ་ནེ། སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། :རྒྱུད། །རྒྱུད། །བོད་རྟགས། །ཚན་གྲངས། །རྒྱུལྡེབ། །ནབ། ནཡབ། བ_བ། ན། རབ་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ਤਸਰਨਾਲਸ਼ਾਮਉਸਨKa®ਚ།
