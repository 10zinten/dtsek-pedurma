རྒྱུད། Imབདུན་རྒྱལ་པོ་ལྟར། །ལྷམ་མེ་ལྷང་ངེར་གསལ་བའི་བཀྲ་ཤིས་ཤོག། གཙང་་བདག་བདེ་དང་རྟག་པ་ཉིད་ཐོབ་པ། །སྐྱེ་རྒ་ན་འཆི་མེད་པ་ཆོས་ཀྱི་སྐུ། །བརྙེས་གྱུར་ལྷུན་གྱིས་གྲུབ་ཅིང་འདུས་མ་བྱས། །བསམ་ཞིང་བརྗོད་པའི་ཡུལ་ལས་འདས་གྱུར་ཀྱང་། །ཆུ་ནང་ཟླ་བ་ནམ་མཁའི་འཇའ་བཞིན་དུ། །སྤྲུལ་་་་སྐུས་འགྲོ་དོན་མཛད་པར་སྨོན་ལམ་འདེབས། །ཀླུ་སྒྲུབ་པདྨོའི་གེ་སར་ལས། །ཉི་མ་སྦས་པ་བུང་བ་:ཡིས། །ལེགས་བླངས་བདུད་རྩིའི་དགའ་སྟོན་གྱིས། །བུ་སློབ་སྦྲང་བུའི་ལུས་རྒྱས་ཤོག། ཁ་ཆེའི་སློབ་དཔོན་ཉི་མ་སྦས་པས་རྗེ་བཙུན་སྒྲོལ་མ་ལ་བསྟོད་པ་རྫོགས་སོ༑། །།ཁུ་ནུ་བླ་ཚོ་ལ་འཁྲིས་ན་བ་དང་། བས་རྒྱ་གར་གྱི་པཎྜི་ཏ་སཱ་དྷ་:ཀཱིརྟི་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ལྡན་མ་ཚུལ་་་་་ཁྲིམས་སེང་གེས་བསྒྱུར་བ་ལས་ཕྱིས་དཔལ་ས་སྐྱའི་གཙུག་ལག་ཁང་དུ་དག་པར་་་་བྱས་སོ། །ཆུ་འཐེན་ཚུན་དབར་འབཀོ་བའི། །དཀོབ་དང་། བདག་དང་འགན་དབང་བྱ་བ་ཁ། །ཡོད་དོན་མཚོ། ཁོབ་ས། བ་བྱ་བའི་འོས་མི་བྱ་ཚ་བ་དཔག་མེད་གཙོ་བོ:བ། ཀས་ཆ་ཚོ་དཙgའབྱཏི་ཀར་ཚོང་[འོ། །ནད་དོད་དང་། དག་པ་སུ་ཡང་། །ཚོར་བ་ཀ་ཁཚོནས་སྨད་པ་གཉིས། གནད་ཀྱི་བང་ཆ/wnཚོ། ཚོང་བདེ་།
