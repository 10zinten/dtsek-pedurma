དོན་གསལ་སྲིན་བུས་གྲོང་ཁྱེར་མིན། །དམན་སོགས་མ་འོངས་གཟུང་བྱའི་ཕྱིར། །སྒྲུབ་དོན་དམ་པའི་གཞུང་གསལ་བྱས། །ཁེ་གྲགས་ངན་རྟོག་བློས་མ་ཡིན། །ཐུགས་རྗེ་མཆོག་གི་དགོངས་མཛད་བཞིན། །ངལ་བའི་དགེ་སོགས་ཉི་ཟེར་གྱིས། །ཁམས་གསུམ་མུན་ཚོགས་བསལ་འགྱུར་ཞིང་། །ཡང་དག་གྲུབ་པའི་འབྲས་ཐོབ་ཤོག། དཔལ་ཧེ་རུ་ཀའི་སྒྲུབ་པའི་ཐབས་སློབ་དཔོན་ཆེན་པོ་ཧཱུྃ་གི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་བཛྲ་ས་ཧ་དང༌། སྒྲ་བསྒྱུར་:གྱི་ལོ་ཙཱ་བ་རྨ་རིན་ཆེན་མཆོག་གིས་བསྒྱུར་ཅིང་གཏན་ལ་ཕབ་པའོ། །སླད་ཀྱི་ལོ་ཙཱ་བ་ཕྱི་མ་དག་གིས་སུམ་ཐུག་ཏུ་ཞུ་ཆེན་བཅོས་པའོ། །།བདག་པ་ཀ། »འཁོདོl:ཀ་ཀཱ། ཀ་ཀ་བ་ག་བ་ག་བ་་་་་ད་དེ་ད་ལ། ལེགས། །ཁ་འདེགས། །བ་དང་དཀྱིལ་བྱེད་ཀྱི་གནས། »། ཐུབ་ཏུ་རྗེ་ལ། དཀྱིནལོའི་ཀ་བ་གསཏེ་བས་སྐྱེ་བསྡུར་མཆན། །དྲག་ཚོ་བཅགས་::་:་ན་་ཁ་ཨངག་ཁོ་ན་འཁྲུག་དྲག་པ་དང་། དེ་ཡམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པར་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་ཧེ་རུ་ས་ཊཀ་ཞེས་བྱ་བའི་སྒྲུབས་པའི་ཐབས་བཞུགས་སོ། །Q«པེ་»«སྣར་»ཚོགས། («སྣར་»གྲུབ། «པེ་»《སྣར་»ལ། །«པེ་»«སྣར་»དོན། «པེ་《སྣར་»འདུལ། དེའི་«པེ་»«སྣར་པར། «པེ་»«སྣར་»འཕུང་༌། ན«སྣར་བགོས། «པེ་《སྣར་སྦྱར། «པེ་ཡི། དང་། ཚོགས་«པེ་»«སྣར་བཞག@«པེ་»བསལ། «པེ་《སྣར་ལོ། «པེ་《སྣར་བས། «པེ་»ཆོས་《སྣར་»། །དང་། །ང་དང་། འོ་བྱེད་པ། འབ་པའི«པེ་»«སྣར་»། Q«པེ་»«སྣར་པ། («པེ་»«སྣར་»དབྱིབས། ཚེ«པེ་»གྷཾ། «པེ་»པ་«སྣར་»ཁམ། <«པེ་»«སྣར་རྗེའི། («པེ་»«སྣར་ས། གང་། ང་:ན་(«པེ་»«སྣར་འཁྲུལ། »«པེ་»«སྣར་»འདོད། ་«པེ་»«སྣར་»«ཅོ་»ཧཾ། «པེ་»ཡི། «སྣར་»བའི། ་གྱི«པེ་»《སྣར་གཟའ་ཀླུབས། »«པེ་《སྣར་བཏུགས། «པེ་《སྣར་»ཀུ་མ། («པེ་ཨཱཏྨ་ཀོ་ཧཾ། 《སྣར་ཨཏྨ་ཀོ་ཧཾ། །«པེ་»«སྣར་བརླབས། «པེ་དྷརྨ་དྷ། «སྣར་དྷརྨ་དྷ། ::ཀྱི་ཁཆོ«པེ་ཨཱཏྨ་ཀོ་ཧཾ། «སྣར་ཨཏྨ་ཀོ་ཧཾ། །«པེ་»«སྣར་ཡུལ། པོ«པེ་《སྣར་ཟེ། @«པེ་»—།
