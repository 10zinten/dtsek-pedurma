བསྟན་འགྱུར། རྒྱུད། ཀྱིས་རྫོགས་པའི་སངས་རྒྱས་ཐོབ་པར་གྱུར་ཅིག་བྱའོ། །དེ་ནས་སྤྱོད་ལམ་ཐམས་ཅད་སྒྲོལ་མའི་ང་རྒྱལ་གྱིས་གོམས་པར་བྱའོ། །ཌ། དཔའ་མོ་གཅིག་པའི་རྣལ་འབྱོར་མའི་སྒྲུབ་པའི་ཐབས་རྫོགས་སོ། །་། ཀ་ཆR'pཎ་ས་པ་ལ། ཁ་ས་པཎྜི་ཏ་པ་ར་ཧེ་ཏ་དང་། ཁབོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་ཤེས་རབ་རྒྱལ་་་མཚན་གྱིས་བསྒྱུར་པའོ། །།།::ཀྱི། ་་གླ་བ་ངོ་མ་རྗེ་ཏ་ཀོ་ਉੱਤੋਂਸੌਉਨਾਸਬਨੂੰਸੁਲਝAgབ་ཀྱི་ད་ལའི་བསར་མཚན་°ཁRལ་ཆུ་ཡ་ཊིཀལག་བྱེ་ཀཱ་། །ཆུ་སོགས་ལ་ཁ་། ་/་ཏུ་ཡབ༌ཀྱིས་ས་ཁྱོyyཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔའ་མོ་་་་གཅིག་པའི་རྣལ་འབྱོར་མའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་—ནི། «པེ་»«སྣར་»མོ། «པེ་གྱིས། ཚུ«པེ་»«སྣར་»ཉ། «པེ་»«སྣར་པོ་ཡོངས་སུ་གྱུར་པ་ལས་ཨུཏྤལ། །P«པེ་སྣར་»འདི། ཚེ«པེ་»«སྣར་»+སྟེ། ཚེ«པེ་»«སྣར་ཀཱ་པ། ་༼པེ་»«སྣར་—པ། @«སྣར་»ཊཾ། «པེ་»«སྣར་»རྒྱན། ཚེ«སྣར་»ཐོད། «པེ་»ཤམས། «པེ་»«སྣར་»ཀའི། ཁྱེ«པེ་»«སྣར་ན། ཚུལའཡ་ལ་་་། ཁོ་ཡི་ཆ་«པེ་»«སྣར་»ཙ། ད་བར་དེ་དེབསར་བབས་རྗེ་མིག་ལ་། Rཡི། རྒྱུད། (སྡེ་དགེ། པེ་ཅིན། །།དཔེ་གཞི། །སྡེ་ཚན། །བོད་རྟགས། སྣར་ཐང་། རྒྱུད། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། བཚན་གྲངས། ལྡེབ། ཕྲེང་། ཚེག་ཁྱིམ། །ན་—ན། བ་_བ་། བབ་། ན་—བ་། ༼།
