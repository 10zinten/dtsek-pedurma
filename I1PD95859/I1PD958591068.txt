ཚོ་ལ་ཁཌ་ར་ཀ་ཏཆུ་ཀྐཎི་ཡ་ཚེ་ཚུ་ག་བ་ག་རྫོང་དང་། ཡང་། །ཁ་ཏུའི་ལོ་འཁབ་ཀླུ་ཁང་ཊ། ཁ་ཥཎཥ་ཊིཀ་ལ་ཀ་ཀའི་ཎ་ཎཀ་ཡང་ཁཁ་ལོ་བ་ཊི་ཌཁ་བ་ཀ་ཎིའ་ལ་དང་བས་ཀུ་སྐྱའི་ཡར། ཊ་ཋpཀਪਾਣRaRਲਲਲਲੈਸਨਲབལ་ཁ་ག་ག་ཏ་བp::ཀའི་ཟླ་བ་po། ནས་ད་བར་ན་ཀ༄༅། །བྱང་ཆུབ་ཀྱི་སེམས་བདེ་བའི་མན་ངག་ਸਨੂੰਨਸ਼ਟਕਰਨਲਉਸERISEPਨੂੰਸਰਵਉਨਵੰreཁཊ་ཎཎཱ་ཏི་ཎཝོབ་ཁཏྟཁ། ཚེ་ཀུ་»ཏི་ཀ་བ་ལ་ཀ་ང་ཀ་ཀླུ་མཁ་ཁོ་ག་ཏ་ཅn་ཚོ་ཁཊ་ལ་ཁ་གཏོང་བ་ཀུན་ཏཚོ་ལྟ་nཚོ་ཁ་ཎ་ཊ་ཡ་༄༅། །ཀ། རྒྱ་གར་སྐད་དུ། ཥབོ་དྷི་ཙིཏྟ་མ་ཧཱ་སུ་ཁ་ཨོཾ་ན་ཡ། བོད་སྐད་དུ། བྱང་ཆུབ་ཀྱི་སེམས་བདེ་བའི་མན་ངག་། བདེ་བ་ཆེན་པོ་བྱང་ཆུབ་ཀྱི་་་སེམས་ལ་ཕྱག་འཚལ་ལོ། །གསང་རྒྱུད་མཐའ་དག་ལས་མན་ངག་གང་བའི་དོན། བྱང་ཆུབ་ཀྱི་སེམས་འདའ་ཀ་མའི་མན་ངག་འདི་ལྟར་ལགས་ཏེ། འཆི་འཕོ་དུས་སུ་འདི་ལྟར་རོ། ་་། ད་ལྟར་བདག་གི་ལུས་ཁྱབ་པར་གནས་པའི་སེམས་འདི་ནི་་་་་འུབས་ཀྱིས་འདུས་ནས། སྙིང་གའི་དཀྱིལ་ན་འོད་མཚོན་གང་བ་ཅིག་གནས་སོ། །ཡི་གེ་ཤེས་པས་འོད་དུ་མི་ལྟ་སྟེ། ཧཱུཾ་ཡུངས་འབྲུ་ཙམ་ཅིག་ལྷང་ངེར་གནས་སོ། །དེ་ནས་ཚོར་བ་དང་བྲལ་དྲོད་ཤོར་ཏེ། འཇིག་རྟེན་པའི་སྤྱོད་ཡུལ་དང་བྲལ་བའོ། །དེ་ཙམ་ན་འོད་མཐིང་ག་དེ་སྙིང་གི་དཀྱིལ་ནས། བྱང་ཆུབ་ཀྱི་སེམས་ཀྱི་གསང་་་་བའི་ལམ་གཡས་ཀྱི་སྲོག་རྩ་དར་དཀར་གྱི་སྐུད་པ་འདྲ་བའི་ནང་ནས་ཡར་ཡེངས་་་ཀྱིས་བྱུང་སྟེ། སྤྱི་བོའི་གཙུག་ཏུ་བྱུང་། རིལ་གྱིས་དྲིལ་ནས། ལྷང་གིས་ཤར་།
