བསྟན་འགྱུར། རྒྱུད། ཤ་འགྲུབ་པ་དང༌། །མི་འདོད་ཐམས་ཅད་མྱུར་བྲལ་ནས། ་་། སེམས་ཅན་ཀུན་གྱི་དོན་གྲུབ་ཤོག། །ཐུགས་རྗེ་ཆེན་པོ་སྤྱན་རས་གཟིགས། །རྗེ་བཙུན་སྒྲོལ་མའི་་་ཞལ་མཐོང་ཤོག། རལ་པ་གཅིག་པའི་དངོས་གྲུབ་གཉིས། །།བསམས་ཤིང་་་་་་བསྒྲུབ་པས་མྱུར་ཐོབ་ཤོག། རྗེ་བཙུན་སྒྲོལ་མའི་ནང་གི་གཏོར་མའི་ཆོ་ག། སློབ་དཔོན་ཙནྡྲ་གོ་མིས་མཛད་པ་རྫོགས་སོ། །།།།ཀ་ལིང་དང་ཀོབ་། ཆ་ལ་ཡུལ་བལ་པོ་རྫོང་དུ་བསྒྱུར་བའོ། །་བསལ་འཁ་བ་ཚ་བ་སྐྱ། ਮਰਨਨਾਲਸੌਦਾਲgਨ।ਨਰੋਸདེ་ཚོ་ལ་བསྒྱུར་མཆན། ཅང་ཁ་སྐྱ༌ཀ་ཊ་ཚོ་ཡཚོ་ལགས་ང་/ཀླུ་ཡབ། ཁ་ཏུཀ་རོ། །ཏཁལ་་ང་ཚོ་ཚེ་ཏཊི༌མ་སོགས་་ཨེམཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་《སྣར་»༄༅། །སྒྲོལ་མའི་ནང་གི་གཏོར་ཆོག་་་་བཞུགས་སོ། །«པེ་»«སྣར་»འདུན། ་«པེ་»སྣར་»+མ་ཨ་པྲ་ཏི་ཧ་ཏ་སྭཱ་སཱ་ནཾ། «པེ་«སྣར་གཱ། O«པེ་»«སྣར་»སཱཧཱ། ཚེ«པེ་»«སྣར་མཾ་སཾ་ལཾ་བྷ་ཏ། ༄༅«པེ་»«སྣར་»ཛ། ཉི«པེ་»«སྣར་»ལི། ཚེ«པེ་»«སྣར་—པ། ཁྱེ«པེ་»པ་དབུལ། «སྣར་ལ་དབུལ། ཀྱེ«པེ་»«སྣར་བརླག་སྟེ«པེ་»«སྣར་སྤོང་། «པེ་»«སྣར་རིགས། «པེ་»«སྣར་»གཏེར། @«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་»དང་། །«པེ་»«སྣར་དང༌། 《«པེ་»«སྣར་»འགྲོ། ་@«པེ་»«སྣར་འཇོ། «པེ་»«སྣར་»བཟང་། «པེ་»«སྣར་བ། གྱི«ཅོ་»ལོན་ཚེན་རིན། འགྱེ«པེ་»«སྣར་བདེ། «ཅོ་ལ། ལུ«པེ་»«སྣར་ཚིགལེ«པེ་»ངགས་མ། «སྣར་ངགས་མང་། །«པེ་»«སྣར་གྲུབ། ཚེ«པེ་»«སྣར་བསྒྲུབས། །«པེ་»«སྣར་+བ་ལགས་གསུང་ངོ་། སྐུ་ཚེ་ཡོਇਸਤੋਂਇਸਕੋਲਕਈਬੋਟਸਏਸੇਲਈཔ་བཞེ། །་བའམ་དེ་ཉིད་ཀྱིས་ང་ਕੋਈਸੂਬਿਆਸਤਸਿਆਸਡੇਕਸੋਸਲབབབ་་་་་་་་་་་་ལ་་བ་་ང་སོགས་ཆ། །དྲུག་པོ་ལ་སྤྲོ། །ਸਲੋਂབ།
