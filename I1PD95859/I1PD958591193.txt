བསྟན་འགྱུར། རྒྱུད། ཐམས་ཅད་ཀྱི་རྣམ་པའི་དྲིལ་བུ་ཤེས་རབ་ཀྱི་སྒྲའི་རྣམ་པ་དེའི་རྡོ་རྗེ་སེམས་དཔའི་སྙོམས་པར་འཇུག་པའི་རྡོ་རྗེ་དབྱིངས་ཀྱི་དབང་ཕྱུག་མའོ། །ཆོས་འབྱུང་བའི་་་་མཎྜལ་གྱི་ཞིང་དུ་གནས་པ་སྣ་ཚོགས་པདྨའི་སྟེང་དུ་གནས་པ། །སྐུ་མདོག་དཀར་མོ་བསྒོམ་མོ། །རྡོ་རྗེ་དང་དྲིལ་བུའི་མཚན་ཉིད་རྫོགས་སོ། །།།མ་འ། །ཀ་བ་ཀཔཎྜི་ཏ་པ་ར་ཧི་ཏ་དང་། ལོ་ཙཱ་བ་ནག་ཚོ་དགེ་སློང་ཤེས་རབ་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ། །བྱེལ། ད་ལཥ་ཁོས་ཁོ་ཚོ་དབྱེ་བ། ་་་་་ཏཚོཊ་ཏ་ཏཁ;ད་་ཅང་བློ་ཁོ་ན་དེ་ཡང་་ཀའ་རྒྱུད། ཤེལ་ཏོ། །བཛྫཡཏཙི་ཏི། ཕྱིའི་བང་བོའི་སྒོ་ཁབསྡུར་མཆན། བཏྟཱ་ལོ་ཀྱི་འཚོ། །བཅཁོ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་དྲིལ་བུའི་མཚན་ཉིད་བཞུགས་སོ། །«པེ་»«སྣར་བརྟན། @«པེ་»«སྣར་»དང་བཞི། «པེ་»«སྣར་དང་། @«པེ་»«སྣར་རང་གི། «པེ་»«སྣར་ཚེ། ༄༅«པེ་»«སྣར་+པ། ཉི«པེ་»«སྣར་»དང༌། «པེ་»«སྣར་»ཀའི། ཁྱེ«པེ་»«སྣར་»གི་སྒྲ། ་་་་་«པེ་»«སྣར་»+ཤའི། ®«པེ་»«སྣར་པས། «པེ་»«སྣར་བྲིབ། «པེ་»«སྣར་གཟུང་། ཚུ«པེ་»«སྣར་སྐབུ། «པེ་«སྣར་བྱས་པ་སྟེ། ཨ«པེ་»«སྣར་»དོ། ཚ་ཤN/@«པེ་»«སྣར་ཁོག:«པེ་»«སྣར་ལ། «པེ་»«སྣར་མའི། «པེ་»«སྣར་—པར། «པེ་»«སྣར་དྲན། «པེ་»«སྣར་དཔའ། ཚེ«པེ་»«སྣར་ཙ། །ཏེ། བྱ་ཚཎ། །ཞི་འཁོར་ལོའི་མི་ཁྲི་ཚོ་་ལ། ར་ལ་ས་བཅུ་མཔ་ཕྱོགས་ན། ཀ་ཏ་ན་པ་ཏོ་ཆུ་ལོ་ཅའོ་འོ་ཆུ་ཁ་ཤས་དེ་མ་བོད་ལ་ཁ་ཏ་བྱ་བ་ཤོ་རྒྱག་་་་ང་ཚོ་བསྐྱིལ་ལ་་་་ཡང་་བ་འབྱེད་ཀཱན་sཊི་ད་ལོའི་་་་། ཁ་བའི་མེ་ལོ་ཁོ་ཊི་ད་དཀྱིལ་ལ་ལ་་་་་། །ཆ།
