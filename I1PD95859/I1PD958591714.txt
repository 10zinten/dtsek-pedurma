ཐོད་པ་ཁ་སྦྱར་དུ་བཅུག་སྟེ་ཐལ་བས་བཀང་སྟེ། ཡང་ཐ་གུ་སྔོན་པོས་བཅིངས། དུར་ཁྲོད་དུ་ས་མཐོ་གང་ཙམ་དུ་བརྐོས་པས་སྦ་བར་བྱའོ། །དེའི་སྟེང་དུ་རང་་་་ཉིད་ནག་པོ་ཆེན་པོའི་རྣལ་འབྱོར་དང་ལྡན་པས། ཁ་ལྷོར་བལྟས་ཏེ་ཁྲོ་བ་དང་་་་བཅས་པས་སྔགས་འདི་བཟླས་པར་བྱའོ། །ༀ་མ་ཧཱ་ཀཱ་ལ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཨ་མུ་ཀཾ་མཱ་ར་ཡ་ཧཱུྃ་སྭཱ་ཧཱ། དེ་ལྟར་:ཉིན་རེ་བཞིན་བྱས་ན་ཞག་བདུན་ནས་འཆི་བར་འགྱུར་རོ། །གལ་ཏེ་ཤི་བར་མ་གྱུར་ན། འཁོར་ལོ་དེ་ཉིད་དུམ་བུ་དུམ་བུར་་་་་བྱས་ཏེ་ཐལ་བ་དེ་དང་། མིའི་ཤ་དང་ལྷན་ཅིག་ཏུ་ཆང་གིས་བྱུགས། དེ་་་་མངོན་ྋསྤྱོད་ཀྱི་ཐབ་ཁུང་དུ་སྦྱིན་སྲེག་བྱས་ན། གདོན་མི་ཟ་བར་འཆི་བར་འགྱུར་རོ། །དཔལ་ནག་པོ་ཆེན་པོའི་སྒྲུབ་ཐབས་བྱིན་རླབས་དང་བཅས་པ། སློབ་དཔོན་ཆེན་པོ་བསོད་སྙོམས་པས་མཛད་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་དཔལ་སུ་མཱིཏི་ཀཱིརྟི་དང་། ལོ་ཙཱ་བ་དགེ་སློང་པྲཛྙཱ་ཀཱིརྟིས་བསྒྱུར་བའོ། །།།ཡིས་ཀྱང་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་མ་ཧཱ། «པེ་»པོ་བསྒྲུབ་པའི། «སྣར་པོ་སྒྲུབ། «པེ་»«སྣར་པས། ཚ«པེ་»«སྣར་+། «པེ་»དུ། ཚེ«པེ་»«སྣར་འཆག«པེ་»«སྣར་»+ཐམས་ཅད། ཉི«པེ་»«སྣར་»—སྡུག་བསྔལ་དང་། «པེ་»«སྣར་ཆད། «པེ་»«སྣར་»གྱིས། ཚེ«པེ་»«སྣར་ཝ། ཚེ«པེ་»«སྣར་»། «པེ་»«སྣར་»གིས། ཉི«པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་»+། ཁྱི«པེ་»«སྣར་»+དང་། གྱེ«པེ་»«སྣར་»ཀཱ་པ། «པེ་»«སྣར་»བ། ཏེ«པེ་»«སྣར་»+དུ། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་ཡུངས། @«པེ་»«སྣར་»ལ། ཉི«པེ་»«སྣར་»སཱ་ས་ན་ཨུ་པ་ཧཱ་རི་ཎི་ཨེ་ཀཱ་པཉྩི་མ་ཧཱ། «པེ་»ཏི་རནྟཱ་ཏཱ། «སྣར་»ཏཾ་རཏྣ་ཏཱ། ཚེ«པེ་»«སྣར་»ད། @«པེ་»«སྣར་»དུཉྩ། ༄༅«པེ་»།
