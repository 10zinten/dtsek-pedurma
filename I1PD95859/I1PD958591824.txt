།འབོ༌ཨོ་ཙཎྜ་མ་ཧཱ་སེ་ནཱ་གྲ་ས་གྲ་ས་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི་ཤོ་ཥ་ཡ་ཤོ་ཥ་ཡ་མཱ་ར་ཡ་མཱ་ར་ཡ་ཧཱུྃ་ཕཊ། མྱོས་བྱེད་ཀྱི་ཨིསོར་བཞི་པའི་ཚད་ཀྱི་གཟུགས་བརྙན་བྱས་ནས་དེའི་སྙིང་གར་སྔགས་འདི་དག་དུར་ཁྲོད་ཀྱི་རས་ལ་དུག་དང་སྐྱེ་ཚེས་བྲིས་པ་་་་་བཅུག་སྟེ། སོར་གཅིག་པའི་ཚད་ཀྱི་རུས་པའི་ཕུར་བུས་གདབ་པར་བྱའོ། །དུར་ཁྲོད་དུ་:ཁུང་བུ་བརྐོས་ནས་ཁ་སྦུབ་ཏུ་བཞག་ན་ཉི་མ་བདུན་གྱིས་འཆི་:བ་་་བསྒྱུར་བར་ཐེ་ཚོམ་མེད་དོ། །དཎ་ཁང་དྲུག་པ་སྤྲོ། བོན་གཁྲོ་། དཔ། ནིམ་པ་དང་བྱ་རོག་གི་ཚང་བླངས་ནས་ཡི་དྭགས་ཀྱི་མེ་ལ་བསྲེགས་པའི་་་ཐལ་བ་ལ། ༀ་ཙཎྜ་མ་ཧཱ་རོ་ཥ་ཨཎ་ཆེ་གེ་མོ་ཨུཙྪ་ཊ་ཡ་ཧཱུྃ་ཕཊ་བརྒྱ་རྩ་བརྒྱད་དུ་བསྔགས་ནས་སྒོ་ལ་གཏོར་ན་ཁྱིམ་ནས་སྐྱོད་པ་ལ་ཐེ་ཚོམ་མེད་དོ། །འདུལ་བ་ལ་ཝེ་ཁྱི་འཐབ་པའི་ས་དེ་དག་གི་རྐུབ་ནས་བླངས་ལ། ་ཤ་ཕན་ཚུན་སྡང་བའི་་་གཟུགས་བརྙན་ལ་འཐོར་ན་འབྱེད་པར་བྱེད་པ་ལ་རྣམ་པར་བརྟག་མི་དགོས་སོ། །སྔགས་ནི། །ོཨོ་རྭ་ཥ་ཎེ་དྭེ་ཥ་བཛེ་ཆེ་གེ་མོ་པི་དྭེ་ཥ་ཡ་ཨོ་ཙཎྜ་མ་ཧཱ་རོ་ཥ་ཎ་ཧཱུྃ་ཕཊ༑ཅེས་པ་དབྱེ་བའོ། །ལྟ་། །ཀ་བ་དཔལ་བ་ལ་རྒྱུ་བྱུང་། ཨང་། །གྲོ་ག་ལ་བ་བླས་སོར་དྲུག་པའི་རུས་སྦལ་བྲིས་ནས་རྐང་པ་བཞི་ལ་ཧཱུྃ་་་ཁར་བློ་དབུས་ཀྱི་ལྟོ་བར་ཏཾ་གཉིས་བྲིས་ལ། །སྒྲུབ་པ་པོས་རྒྱབ་ཏུ་ཕྲེང་བའི་སྔགས་ཀྱིས་ཕྱི་རོལ་དུ་བསྐོར་བ་ལ། །མཆོད་པ་དང་བསྟོད་པ་ཡང་དག་པར་བརྩམ་ཞིང་། སོ་ཕག་གི་སྟེང་དུ་བསྐོར་རུས་སྦལ་གྱི་ཁོག་པས་བཀབ་སྟེ། སྲད་བུ་དམར་པོས་དཀྲིས་ནས་རྐང་པ་ཕྱིས་པའི་གོས་ཀྱིས་བཏུམས་ཏེ་རྐང་པ་གཡོན་པས་བརྡེག་པར་བྱ་ཞིང་། །aཆེ་གེ་མོ་ལེབ་ཤཾ་བྷ་བནྡྷ་ཞེས་པ་ལན་བདུན་གྱི་མཐར་ཐུག་པས་དགྲ་བོ་ཐམས་ཅད་ཀྱི་ཁ་བཅིང་བར་འགྱུར་རོ། །ཡིད་ལ༌ཁཙོ་སོ་ཁ་དེ་ལ་ཕྲེང་བའི་སྔགས་ནི་འདི་ཡིན་ཏེ། ༀ་ཙཎྜ་མ་ཧཱ་རོ་ཥ་ཎ་:ཧཱུཾ་ཧཱུཾ་ཏྲཾ་ཧྲི་།
