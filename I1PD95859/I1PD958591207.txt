བསྟན་འགྱུར། རྒྱུད། ཤ་འགྱུར་རོ། །།རིགས་པ་དང་ཡོན་ཏན་གྱི་ཚོགས་ཐམས་ཅད་སྡུད་པར་བྱེད་དོ། །སྒྱུ་མ་ལྟ་བུའི་བློས་ནམ་མཁའ་ལྟ་བུར་བསྒོམ་པ་ནི་རིམ་པ་བཞིན་ནོ། །།དེ་ལས་་་བྱུང་བའི་དགེ་བས་ཡོངས་སུ་བསྔོས་ནས་དགེ་བའི་སྤྱོད་པས་གནས་པར་བྱའོ། །e། རྗེ་བཙུན་འཕགས་མ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་སློབ་དཔོན་ཆེན་པོ་ཐམས་ཅད་མཁྱེན་་པའི་བཤེས་གཉེན་གྱིས་མཛད་པ་རྫོགས་སོ། །»། ཤི་ལ་ཁྲལཱ་ཎ་ཌོ་ནི་ཀར་ཌ'ਮਨੂੰਸੰਨਲਵੋਟਰਰਣਵੀਟਵੀਟਸਲਫན་ང་ལ་རྒྱའི་ཏི་ཤུ་ལྔ་བསྡུར་མཆན་་ལ་་་་༌་ཁྱོ་ཆུ་ER'Pਨਰੌਸਉਲਟਛੂਸਲਨੂੰਸੌਵਨੀਝਲਸਾਲམཚན་བྱང་འདིའི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་མ་་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་ར། @«པེ་»«སྣར་»ནཾ། «པེ་»«སྣར་ཏཾ། «པེ་»«སྣར་བསམ། ་ལགྱི་་ལ་ཡ/»ལྔ་ཀ་ང་ཀ་ཡ་eཚེ«པེ་»«སྣར་»གཟུང་། ཚ«པེ་»«སྣར་ཞེས། «པེ་»«སྣར་བྱ་སྟེ། ，«པེ་»«སྣར་»+༑«པེ་»«སྣར་»+། «པེ་»ཏཾ། «པེ་»«སྣར་+པ། :ཀྱིངབན། ར་ཐང་། །ཁ་ན་མ་ཁོ་ན་འགན་ལོ་བརྒྱ་ཁྲི་བ་ཁག་ད་བ་ཅིན་བསྡུར་འབྲས་རེའུ་མིག་དང་་བྱེjz་ཀྱི་མི་ཚེ། རྒྱུད། （་པའི་གས་ཁ་དེའི་སྐོpདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། ས་ཅོ་ནི། །སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། བོད་རྟགས། ཚན་གྲངས། ཀྱི་?ལྡེབ། བབ། ན་ན་བ?ན་::བན་ཕྲེང་། །ཚེག་ཁྱིམ། །།ར།
