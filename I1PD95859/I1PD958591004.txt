ཕོ་བྲང་དམ་པ་ཁྲ་འབྲུག་གི་གཙུག་ལག་ཁང་དུ་བསྒྱུར་བའོ། །།།།:hརྗེ་བཙུན་སྒྲོལ་མ། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ། དགྲ་བཅོམ་པ་ཉི་མ་གུང་པ། དགྲ་བཅོམ་པ་ཤ་ནའི་གོས་ཅན། དགྲ་བཅོམ་པ་ཀྲྀཥྛའི་གོས་ཅན། ཁ་ཆེ་ཉི་མ་སྦས་པ། རཱ་ཧུ་ལ་ཤཱི། བི་ན་ད་བི། པཎ་ཆེན་ཤཱཀྱ་ཤྲཱི། ལོ་ཙཱ་བ་བྱམས་པའི་དཔལ། བླ་མ་བསོན་ནམས་རྒྱལ་མཚན། རིན་པོ་ཆེ་བ་ཚད་་མའི་སྐྱེས་བུ། དེས་བུ་སྟོན་ལའོ། །།བ་ཕྱི་མ་ནི། །ད་བསྡུར་མཆན། ནད་ཀྱི་མན་ཨམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བཅོམ་ལྡན་་་་་འདས་མ་སྒྲོལ་མ་ལ་བསྟོད་པ་ཉི་ཤུ་རྩ་གཅིག་པའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«སྣར་»ཏ། «པེ་»བི་ཨེ་ག་:::«སྣར་སྟོད། @«པེ་བི་ཧཾ། «སྣར་བྱི་ཀཾ། «པེ་བསྐྱེ། «སྣར་»སྐྱེད། @«སྣར་བྱས། «པེ་»ཀྱི། «སྣར་+། ཚེ«སྣར་བའི། །།«སྣར་»ལྟ། «སྣར་»སྐྱེས། «སྣར་པར། ༄༅«པེ་»འབྱུང་། «པེ་»«སྣར་བསྟན། O«པེ་ཧི། །དེ་ཚེ«སྣར་»ཁར། «སྣར་»བཞི། «པེ་ཐེ། ཚེ«པེ་བཏང་། །ཀྱེ«སྣར་»སྣ་དང་། ཉེ«པེ་བདུད། མོ་སུ«པེ་»«སྣར་བྱ། ཚེ«སྣར་བསྲེགཨེ«སྣར་»ཏིས། ཚེ«པེ་བསྣགས། «སྣར་བརྣགས། 《སྣར་»བཏང་ཅེས། ་ ེ«སྣར་ཨོཾ། «པེ་+། གྱེ«སྣར་ཨོཾ། འགྱེ«སྣར་ཏུ། «སྣར་»བྷིནྡྷ་བྷིནྡྷ། «པེ་གྲིནྣ་གྲིནྣཾ། ཞཉེ་བསྐྱེ«པེ་དྷཱ་ན། «སྣར་དྷན་ས། ཨེ«སྣར་ཨམ། «སྣར་མ། «སྣར་»ཕཊ། «པེ་མ་རཾ། «སྣར་ར་ཀུལླ་སྦྱོ། «པེ་《སྣར་»—འཕགས་མ། «སྣར་»ལས་ཀྱི། «སྣར་འཛིན། @«སྣར་»མོ། ཉི@«སྣར་འཕྲིན། «པེ་»«སྣར་»དགར། ཀྱེ《སྣར་»—མཎྜལ། ཞུ་«སྣར་བྱཱ་དྷ། ཨེ«སྣར་ནི་ཤན་ཏི། ཚ་་་། །པད«སྣར་སེར། «སྣར་»ཀྱི། ༼@«སྣར་བྱས་ཏེ། །བརྒྱ། «པེ་»ཅིགཚུ«སྣར་དབྱིག་ཉི་ཚེ«སྣར་»སྦྱངས། «སྣར་སྲིངས་འཕྲིན། «སྣར་ཨ། ཆེ«སྣར་བྷི། གེ«པེ་ཡ་མ་པྲ་ཧི་ཡ་དེ་ཧི། «སྣར་མ་མཱ་ཕྲཧྨ་དྷ་ར་དེ་དྷ། ཨེ«སྣར་»ཀྱིས། «སྣར་»བྱ། ཁྱེ«སྣར་བ། ཨེ«པེ་ཀྱི། ཚེ«སྣར་»དུ། ཁ«པེ་ཆུས། ལོ་ཀྱི་ལོ་ཀེ་བ་ལུ།
