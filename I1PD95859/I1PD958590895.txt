རྒྱུད། ལ། དགའ་བའི་རྡོ་རྗེ་རོལ་པའི་ཚོགས། །།གཞན་ལ་སྣང་བའི་དཀྱིལ་འཁོར་རྣམ་པར་མཛེས༑། ཚངས་པའི་ཚོགས་རྣམས་ཟིལ་གྱིས་གནོན་མཛད་པའི། །རྟོག་པ་དུ་མའི་སྒྲོ་འདོགས་གཅོད་པར་མཛད། །གཉིས་མེད་ཡེ་ཤེས་གཉིས་ཅན་རྣམ་མཐའ་ཡས། ཨུ་། སྣང་སྟོང་བདེ་ཆེན་མཉམ་ཉིད་དབྱིངས་ལ་རོལ། །ཀུན་འབྱུང་ཡོད་་་ཏན་རྩོལ་མེད་ལྷུན་གྲུབ་སྣང་། །རྒྱན་གྱི་དཀྱིལ་འཁོར་རྣམ་བཀོད་རང་ལས་བྱུང་། །མཛད་པ་རྒྱ་ཆེན་འགོག་མེད་ལྷུན་གྲུབ་པས། །།ཚོགས་བཅས་རོལ་པས་བྱ་བ་ཀུན་རྫོགས་གནས། འ། སྐུ་གསུང་ཐུགས་མཆོག་ཡོན་ཏན་ཕྲིན་ལས་རྒྱན། བ་། དཀྱིལ་་་་འཁོར་མཛེས་པའི་ལྷ་ལ་ཕྱག་འཚལ་ལོ། །ཨོཾ་ཨཱཧཱུཾ། ཨ་ཏི་པུ་ཧོ། སརྦ་ཏ་ཐཱ་ག་:ཏ་བནྡྷ་ནཾ་ཀ་རོ་མི། ཨ་ལ་ལ་ཧོ། དཔལ་ལྡན་གཡོན་པ་འགྱིང་བའི་ཐབས། །གཡས་པ་བ་དན་གསུམ་པར་བྱ། །།སྲི་ཞུ་བྱས་ཤིང་བཏེགས་པའི་མཆོག། བདག་ཉིད་རྡོ་རྗེ་སེམས་དཔའ་དཔལ། །འཛུམ་དང་བཅས་པའི་མིག་གིས་བལྟ། །བསམ་པ་གཡོས་པས་ཕྱག་འཚལ་ལོ། །ཨོཾ་བདེ་བར་གཤེགས་སོ་བཅོམ་ལྡན་འདས། །རྒྱལ་བ་རྒྱ་མཚོ་ཡེ་ཤེས་ཀྱིས། །སྲིད་པའི་རྒྱ་མཚོ་མཐའ་ཡས་ཀུན། །བློ། །ཀུན་བཟང་མཆོད་སྤྲིན་མཆོག་ཏུ་རོལ། །ྋ། ཨོ་ན་མསརྦ་ཏ་ཐཱ་ག་ཏེ་བྷ། བི་ཤྭ་མུ་ཁེ་བྱ༈སརྦ་ཏཊཾ་ཨུངྒ་ཏེ་སྤ་ར་ཎ་ཧི་མཾ་ག་ག་ན་ཁཾ་སྭཱ་ཧཱ། ཚད་མེད་བཞི་ཡི་་་ཕྱག་རྒྱ་ཡིས། །མདུན་གྱི་དཀྱིལ་འཁོར་མཉེས་པ་དག། བློ་། དམ་ཚིག་དཀྱིལ་་འཁོར་ལ་གཞུག་ཅིང་། །མཉེས་ནས་དགེས་ཤིང་བརྟན་པར་མཛོད། །རྡོ་རྗེ་་་སེམས་དཔའི་ཏིང་འཛིན་མཆོག། ཕྱག་རྒྱ་ཆེན་པོའི་ཆ་ལུགས་ལྡན། །ཁམས་་གསུམ་ཀུན་དུ་རྒྱས་བཏབ་ཅིང་། །།རིགས་ཀྱི་དཀྱིལ་འཁོར་གསང་བའི་བདག། ཨོཾ་བཛྲ་སཏྭ་ཧཱུཾ། མ་ཧཱ་མུ་དྲ་། །ཆརྡོ་རྗེ་སེམས་དཔའ་སྐལ་བཟང་དཔལ་ཁྱོད་་་བདག། བདག་ཅག་སྒྲུབ་ཅེལ་རྟོགས་པར་མཛད་དུ་གསོལ། །རིགས་མཆོག་་་།
