ལ་ལ་༽བསྟན་འགྱུར། །རྒྱུད། ལ་་འོ་anདོན་གྱི་དབང་གིས་དགོངས་ནས་ཀྱང་། །འཁོར་བཅས་གཤེགས་ཤིང་བཞུགས་སུ་གསོལ། །ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ་ཧཱུྃ། །སྒྱུ་འཕྲུལ་དགའ་བའི་རྡོ་རྗེ་རོལ་པའི་ཚོགས། །གཞན་ལ་སྣང་བའི་དཀྱིལ་འཁོར་རྣམ་པར་མཛེས། །ཨ་ལ་ལ་ཧོ། ་་པ་ཚངས་པའི་ཚོགས་རྣམས་ཟིལ་གྱིས་གནོན་མཛད་ཅིང༌། །རྟོག་པ་དུ་མའི་སྒྲོ་འདོགས་གཅོད་པར་མཛད། རྟ་ཀ། ཨ་ལ་ལ་ཧོ། །གཉིས་མེད་ཡེ་ཤེས་གཉིས་ཅན་རྣམ་་་མཐའ་ཡས། ཅ། སྣང་སྟོང་བདེ་ཆེན་མཉམ་ཉིད་དབྱིངས་ལ་རོལ། །ཨ་ལ་ལ་ཧོ། ཀུན་འབྱུང་ཡོན་ཏན་རྩོལ་མེད་ལྷུན་གྲུབ་གང༌། ཁྱི། །རྒྱན་གྱི་དཀྱིལ་འཁོར་རྣམ་་་བཀོད་རང་ལས་བྱུང༌། །ཨ་ལ་ལ་ཧོ། །མཛད་པ་རྒྱ་ཆེན་འགོག་མེད་ཀུན་གྲུབ་པས༑། ཚོགས་བཅས་རོལ་པས་བྱ་བ་ཀུན་རྫོགས་པ། །ཨ་ལ་ལ་ཧོ། །ལ་སྐུ་གསུང་ཐུགས་མཆོག་ཡོན་ཏན་ཕྲིན་ལས་བརྒྱན། །དཀྱིལ་འཁོར་མཛེས་པའི་ལྷ་ལ་ཕྱག་འཚལ་ལོ། །ཨ་ལ་ལ་ཧོ། སྨིན་མ་གཡོ་བ་འདེགས་ཚུལ་གྱིས། །།རུ་ལུ་རུ་ལུར་བཅད་་དགོད་འདུད། །ཧཱུྃ་ཨ་ཏི་བྷཱུ་ཧོ། བན་ད་:ནཾ་ཀ་་རོ་ཡ་མི། ་གུ་རུ་པ་ཏི། ཨ་ལ་ལ་ཧོ། ལཨོ་འབྱུང་བ་ཆེ་ལས་ཐམས་ཅད་འབྱུང༌། །སྟོང་གསུམ་རྡུལ་སྙེད་མ་ལུས་པའི། །གཟུགས་སྒྲ་དྲི་རོ་རེག་བྱ་ཆོས། ་ཁ་། མེ་ཏོག་་བདུག་པ་མར་མེ་དང༌། ་་། དྲི་ཚོགས་ཀུན་གྱིས་མཆོད་པའི་ཕྱིར། །ོ། ལྷ་མོའི་་ཚོགས་རྣམས་མ་ལུས་སྤྲོ། །ཨོ་པ་ན་པེ་ན་པེ་ནུ་སུ་ར་ཏ་ཏི་བུ་པ་ར་མུ། སརྦ་ཏ་ཐཱ་ག་ཏ་ས་མ་ཡོ་ག་ཙིཏྟ་ཨཧཱུཾ། ཨོཾ་བཛྲ་པུཥྛ། ཨོཾ་བཛྲ་དྷུ་པེ། །བཛྲ་ཨཱ་ལོ་ཀེ། བཛྲ་གནྡྷེ། པྲ་ཏཱིཙྪ་ཧོ། ཤྲཱི་བྷ་ག་བཱན་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། རྡོ་རྗེ་དྲག་པོ་གད་་མོ་ཆེ། །མཁུར་བ་དབུགས་ཀྱི་རླུང་འབྱིན་ཅིང་། །།འབྲུག་གི་སྒྲ་ལྟར་ཟབ་པ་ཡི། །རུ་ལུ་རུ་ལུའི་སྒྲ་བརྗོད་དོ། །ཛཧཱུྃ་བཾ་ཧོ། །རྣལ་འབྱོར་དེ་ཉིད་བཅོམ་ལྡན་འདས། །རྡོ་རྗེ་སེམས་དཔའ་དེ་བཞིན་གཤེགས། །དམ་ཚིག་ལས་དང་རྣམ་།
