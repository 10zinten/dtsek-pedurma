རེབས་པའི་མི། །འདུ་བ་ར་གཞན་དུ་འགྲོ་བ་ཡི། །བསམ་པ་དཔག་ཏུ་ཅིག། ལུས་སེམས་བདེ་ལ་འབད་དེ་འབད་བཞིན། ལས་མི་བདེ་བ་དེ་བས། དཔལ་བསྡུར་མཆན། སྤྱི་དཔོན་པ་བྱས་འདྲ་བ་བན་ནུའི་ས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །དཔལ་ཨུ་་་་རྒྱན་གྱི་རྣལ་འབྱོར་མའི་རང་གི་ལོངས་སྤྱོད་ཀྱི་རྟོག་པ་བཞུགས་སོ། །«པེ་»«སྣར་+ཡེ། ཉི«པེ་»«སྣར་»ཡོ་ག་ནི་རྨ། «པེ་བ། ཚེ«སྣར་བས། «པེ་»«སྣར་གྱི། ཨ«པེ་མའི། །«པེ་»«སྣར་+པ། ཉི«པེ་»«སྣར་»ཀ«པེ་»རིག«པེ་»«སྣར་བྱེད་ཅིང་། «པེ་»དྷུ་ཏི། «སྣར་དྷཱ་ཏི། ༄༅«པེ་»«སྣར་ཕག་ཚ«སྣར་»བ། «པེ་»«སྣར་»ཤིང་ནི། «པེ་»«སྣར་རིག་པའི་དབང་དང་དད་ཨེ«པེ་»«སྣར་རིག«སྣར་»དགེས། «པེ་»«སྣར་རིག་«པེ་»«སྣར་»རིག@«སྣར་»གྱི། «པེ་»«སྣར་ཙ། མུ«པེ་»«སྣར་སྒྲོག་ཅེས། @«པེ་བཟློག་«པེ་ལྟ། ཉི«པེ་»«སྣར་རིག«པེ་»«སྣར་སྐཁ་ས་མ། «པེ་»«སྣར་དབུ་རྒྱན་གྱིས། ཚུ«པེ་»《སྣར་བར་བ། ་ཡaཚོད་དམ་ཀ་ཤ་ར་བ་ཚོས་ཁ་བ་བློ་དང་ཀ་མ་འཁྲུལ། བ་ནི་ར་ཚིཏེ་ཁོ། པ་ག་ཏཁཀར་ལ་ཁང་ནང་་ཀུན་ལ་འཚོ་བ་འད། ་ཁང་ཀཁོ་ར་ཁ་ཆོ་ཏཚོ་ཡན་བསྡུར་འབྲས་རེའུ་མིག་ཁ་གང་ན་ཁྲོ་བས་ངང་::ཀ་ལོ་ཁོག་དང་ཁག་ཏུ་ཚོoཔ་ན་རྒྱུད། （པར་དེ་ལྟ་བྱས་། དའ་ཅོ་ནེ། དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང༌། །སྡེ་ཚན། རྒྱུད། །རྒྱུད་འགྲེལ། །བོད་རྟགས། ཚན་གྲངས། ལྡེབ། བཙབ་། ན་_ན་ན། ན་བ—བ་ཕྲེང༌། ན་ཚེག་ཁྱིམ། །།།སེ། ན་། ཀ་ཡ་ས་ཁོང་རྒྱ་ཤོགཀ༄༅། །ལཚོ་ཁྱེད་ལ་ཁའི་ཆུ་ཆེ་ཀn།
