རྒྱུད། ༼ལ༽ཁ་ར་སྦྲང་རྩིས་བྱུགས་པ་ཡིས། །ཨ་མྲ་ལ་སོགས་ཚི་གུ་བྱུག་། ཧ་ས་ཀ་རི་་མར་ལ་སོགས། །མཉམ་པའི་དྲི་ཡིས་བསྐུ་བར་བྱ། །ཡུང་དང་རྒྱ་སྐྱེགས་ལ་སོགས་པའི། །།ཚོན་གྱིས་མཐུན་པར་ཁ་བསྒྱུར་ནས། །ཤུན་པ་གྲོ་ག་ལ་སོགས་བྱས། ཀླུ་། མཆོད་པའི་ཚེ་ན་དབུལ་བར་བྱ། །སངས་རྒྱས་རྣམས་ལ་བཤོས་དབུལ་བ། །།ཤ་དང་ཆང་ནི་རྣམ་པར་སྤངས། །གླུ་དང་སིལ་སྙན་གྲུབ་པ་དང་། །གདུགས་དང་རྒྱལ་མཚན་བ་དན་དང་། །ཁོ་། ཤ་དང་ཆང་དུ་བཅས་པ་ནི། །གཏོར་མའི་མཆོག་སྟེ་ཚོགས་པའོ། །བདག་གི་:ཕྱུག་སྟེ་ྋསྐལ་བཟང་པོ། །རྡོ་རྗེ་ལས་ནི་དཔལ་གྲུབ་པའོ། །བཤོས་དང་གཏོར་མ་དཀར་པོ་དང་། །པདྨ་མེ་་་་ཏོག་ལྡན་པ་ནི། །སྤྱན་རས་གཟིགས་དབང་གཙོ་བོ་ལ། །ཕུལ་ན་དགོས་པ་་་ཐམས་ཅད་འགྲུབ། །རྒྱལ་པོ་ཁ་ཟས་ལ་སོགས་པ། །ཤ་དང་ཆང་དུ་ལྡན་པ་་་དང་། །གླུ་དང་སིལ་སྙན་སྐལ་བཟང་དག། རྡོ་རྗེ་ཉི་མའི་འཁོར་ལ་འབུལ། །ཁ་ཟས་ཀུན་གྱི་རོ་ཀུན་དང་། །ཤ་དང་ཤ་ཡི་མཆོག་རྣམས་དང༌། །ཤ་ཡི་སྦྱོར་བ་ཤ་སྐམ་དང་། །བཤོས་དག་རྡོ་རྗེ་ལྷ་ལ་དབུལ། །གཞན་ཡང་བཟའ་བཏུང་ལོངས་སྤྱོད་ཚོགས། །གཡོས་ཀྱི་སྟོན་མོ་མང་དུ་བཤམ། །ཌ། དཀྱིལ་འཁོར་ཆེན་་་པོའི་ཕྱོགས་ཀུན་ནས། །སྤོས་ཆེན་སིལ་སྙན་ཕྲེང་ཐོགས་ཤིང་། །གར་དབྱངས་དྲོ་ཐབས་དྲུག་འགྱུར་རྩེ། །གཟིངས་ེབསྟོད་དགའ་བའི་བཞིན་རྒྱས་སྤྱོད། །ཧཱུྃ་འཕགས་མཆོག་ཐུགས་རྗེའི་དུས་ལ་བབ། །ོ། ཁྱད་པར་བསྟན་པའི་དུས་ལ་བབ། །དམ་ཚིག་བསྐང་བའི་དུས་ལ་བབ། །དངོས་གྲུབ་སྩལ་བའི་དུས་ལ་བབ། །མཐུ་ཆེན་མཐུ་སྟོབས་བཅོམ་ལྡན་འདས། །:བྱིན་ཆེན་བྱིན་རློབས་བཅོམ་ལྡན་འདས། །དབང་ཆེན་དབང་བསྐུར་བཅོམ་ལྡན་འདས། །དངོས་གྲུབ་མཆོག་སྩོལ་བཅོམ་ལྡན་འདས། །བདག་གི་ལས་རྣམས་གང་སྒྲུབ་ཅཔ། །འབྲས་བུ་དུས་འདིར་།
