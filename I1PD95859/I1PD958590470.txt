།སྣང་བ་ཡུལ་དུས་རྣམས། །དཀྱིལ་འཁོར་རོལ་པའི་ཚུལ་དུ་བརྟག། བཟའ་་་་བཏུང་བྱིན་བརླབས་མ་ཉམས་བཟླས། །སྒྱུ་མ་གཉིས་མེད་ཡེ་ཤེས་རོལ། །བྱེད་སྤྱོད་ཚོགས་གཉིས་རྒྱུ་རུ་བརྟག། ཐམས་ཅད་འགྲོ་བའི་དོན་ཕྱིར་སྤྱད། །གང་ཞིག་དང་པོའི་ལས་ཅན་རྣམས། །རབ་ཏུ་བཟུང་ཞིང་ཅུང་ཞིག་ཙམ། །བསྡུས་ལས་བྱུང་བའི་དགེ་བ་དེས། །།རྟ་མཆོག་དཀྱིལ་འཁོར་འགྲུབ་པར་ཤོག། དཔལ་རྟ་མཆོག་རོལ་པའི་གསང་བའི་དོན་རྣམ་པར་དགོད་པ་ཞེས་བྱ་བ། སློབ་དཔོན་:ཆེན་པོ་ྋཀུ་ཀུ་རཱ་ཛས་མཛད་པ་རྫོགས་སོ། །།།་་་་་་:རྒྱ་གར་གྱི་མཁན་པོ་བིདྱ་ཀ་ར་སིདྷ་དང་། ཞུ་ཆེན་གྱི་ལོ་:ཙཱ་བ་བནྡྷེ་ལྷ་རིན་པོ་ཆེས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།《ཀོཕྱིpnཚོགས་ཀྱིས་༄༅། pཟསར་མཉན། ལ་ལོ་ཁ་་ཀྱི་ཡོད། །།ཊ་ཏ་ཏི་ཊ་ཏོ་དང་སྒྱོུའི། ་་ཏ། ་དོ་་་:ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །རྟ་་མཆོག་རོལ་པའི་གསང་དོན་རྣམ་པར་དགོད་པ་བཞུགས་སོ། །«པེ་»«སྣར་གུ་ཧཱུནྡྲ། «པེ་»«སྣར་+དུ། «པེ་»«སྣར་»ཚོགས། «པེ་»«སྣར་འཇིགཨེ«པེ་»«སྣར་གྲུབ། «པེ་»«སྣར་རྫོགས་རིམ། «པེ་»«སྣར་བཟུང་། ®«པེ་»«སྣར་པས་བསྲུང་། «པེ་»«སྣར་ཀུན། @«པེ་»«སྣར་བཅུའི། ཚེ«པེ་»«སྣར་»། «པེ་»«སྣར་»ཧཱུཾ། «པེ་»«སྣར་ར་གཚེ«པེ་»«སྣར་ཧཱུྃ། འགྱེཉེ«པེ་»«སྣར་རྒྱུན། «པེ་»«སྣར་»ཕོ་བྲང་། རྩེ«པེ་»«སྣར་+དང་སྒོ། ཨེ«པེ་»«སྣར་»མཁར། ཚེ«པེ་»«སྣར་»འཆང་། ཁྱེ«པེ་»«སྣར་»—སྣ་ལྔའི་རྩིག་པའི་དྲ་བ་དང་། །ཨུ«པེ་»«སྣར་རིན་ཆེན། །ཨ«པེ་»«སྣར་ཡི། ®«པེ་»«སྣར་མངོན། Q«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་སྐབ། ཚུ«པེ་»ཨ། «པེ་»«སྣར་»འཁོར། ཚེ«པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་ངང་། «པེ་»«སྣར་»ཚིགཚེ«པེ་»«སྣར་»དུ། ཚེ«པེ་»«སྣར་གཏུག་བྱ་སྟེ། སྟེ«པེ་»སྣར་»ཀྱི། «པེ་»«སྣར་»ཏཱ། ཉི«པེ་ཨཏྨ་ཀོ། ཧཾ། «སྣར་ཨཏྨ་ཀོཉཧཾ། ན«པེ་»«སྣར་»བ། «པེ་ཀོ། །
