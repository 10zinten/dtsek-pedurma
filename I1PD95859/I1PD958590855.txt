རྒྱུད། ་ལ། །ཀ་ཅ་ཏན་ཏན་ག་ད་བ་ས་ན་མ་ཀརྨ་པ་སྐུ་ཕྱོགས་ལ་བ་སྐྱེད་མ་ཞེས་བྱ་བ། ་ཁ་ཆེའི་མཁས་པ་ཆེན་པོ་ཤེས་བྱ་མཐའ་དག་ལ་མཁས་པ་དཔལ་རིན་ཆེན་རྡོ་རྗེས་མཛད་པ་རྫོགས་སོ། །ཁ། །།་། ཡོར་ཁོ་ཚོ༌ཨ༌ཀེ་དཔལ་ལྡན་ཨུ་རྒྱན་དུ་སྦྱངས་པ། །མཁས་པ་ཡེ་ཤེས་རྡོ་རྗེ་དང༌། །སྐད་བསྒྱུར་གཉན་བན་བསོད་དེ་ཡིས། །བསྒྱུར་ཅིང་ཞུས་ནས་གཏན་ལ་ཕབ། །དཔལ་ལྡན་མཉམ་སྦྱོར་རྒྱུད་ཆེན་གྱིས། །དགེ་བ་ཤེལ་སྒོང་ལྟར་ཐོབ་པ། །རྡོ་རྗེ་འཛིན་པའི་སར་གནས་ཤོག། ཀཆ། །།སྐྱེ་བ་ཐོག་མ་བྱེ་བ་འབྱེད་དུ་དང་། ཙ་ལའོའི་ཁབ་ལས་ག་ཏའི་རྒྱུད༌ཁ་བའི་རོ་ལ་ཁ་དང་། ਨਿਰੋਗਤੋਂਉਸBBਸਕਰੀਨਸਲੀਲ། བྱ་བ་བྱེད་ཀྱང་བསྡུར་མཆན། ལ་དུས། དེ་ཚུ་ག་ར་ལོ་དྲག་པོ་བྱེད་ཀྱི། བཀའ་དང་དམ་པ་དེ། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །རྒྱུད་་ཀྱི་རྒྱལ་པོ་སངས་རྒྱས་ཐམས་ཅད་དང་མཁའ་འགྲོ་མ་བདེ་མཆོག་གི་དཀྱིལ་ཆོག་བཞུགས། «པེ་»«སྣར་»ཀཱི། ལ་ལ་ན«པེ་»«སྣར་བ། «པེ་»ཛཱ། ས«པེ་»«སྣར་»བི། «པེ་»སྣར་»གཏུང་། མཚན་«པེ་»«སྣར་གཟུང༌། «པེ་»«སྣར་གིས། །«པེ་»«སྣར་»དང་སྡང་ལྡན་པ། པེ་《སྣར་»བ། ས«པེ་»«སྣར་»+དང༌། «པེ་»སྔར། .《པེ་»«སྣར་»བཞུག་མོན། «པེ་»«སྣར་»པ། བ«པེ་»«སྣར་»གཟུགས་་པར། །ས«པེ་»«སྣར་དྲི་རུ་བརྗོད། ་པོ«པེ་»«སྣར་»དམང་། «པེ་»«སྣར་»ད། «པེ་»«སྣར་»ཤཱ་ཐ། «པེ་གྲག«པེ་»«སྣར་»ད། ་ས«པེ་»«སྣར་གཞག«པེ་»«སྣར་སྟངས། .«པེ་»«སྣར་ལཱ། ་«པེ་»སྐྱེར་ཤིང་། «པེ་»«སྣར་ལྟར། (《པེ་»«སྣར་»ཡིས། པོ«པེ་»«སྣར་»བཞག«པེ་»«སྣར་»བཤད། «པེ་»«སྣར་ཧྲི། «པེ་»《སྣར་»ཆུ། «པེ་»《སྣར་»ཕྱིས། 《སྣར་བཞག«པེ་»«སྣར་»པོའི་རུ། «པེ་»«སྣར་ཨ། ས«པེ་»《སྣར་ཧོ། པ«པེ་ཧཾ། 《སྣར་ཧཾ། «པེ་ཡིས། «པེ་»«སྣར་ཟད། «པེ་»«སྣར་ཡིས། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་»ག«པེ་»«སྣར་ཝ། «པེ་»«སྣར་ལེ། ལྕེ«པེ་»པའི《སྣར་»དཾ། ཨེ«པེ་»«སྣར་»དང༌། ལ«པེ་»«སྣར་»རཱི། «པེ་»«སྣར་»བྲི་གྲི༔་པ་ཚ་ཏུ། ཤེས་།
