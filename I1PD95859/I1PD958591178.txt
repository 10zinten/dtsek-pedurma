དེ་ཁོ་ན་ཉིད་ཀྱི་ཚིག་ལེའུར་བྱས་པའི་མན་ངག་གི་འགྲེལ་པ་ཞེས་བྱ་བ། པ་འདི་བརྗོད་པའོ། །དེ་ཁོ་ན་ཉིད་འདི་ཞེས་པ་ལ་སོགས་པ་ལེགས་པར་གོ་བ་་་རྫོགས་སོ། །།།པོ་དེ་ཁོ་ན་ཉིད་ཀྱི་མན་ངག་མཁས་པ་ཆེན་པོ་:སློབ་དཔོན་ཆེན་པོ་སློབ་དཔོན་སུ་ཛ་ཡ་ཤྲཱི་བྷ་དྲའི་ཞབས་ཀྱིས་མཛད་པའོ། །རྒྱ་གར་གྱི་མཁས་པ་རིན་དཔལ་གྱི་ཞབས་ལ་གཏུགས་པ། པཎྜི་ཏ་པ་ར་ཧེ་ཏ་དང༌། ལོ་ཙཱ་བ་ནག་ཚོ་དགེ་སློང་ཤེས་རབ་རྒྱལ་མཚན་གྱིས་བསྒྱུར་་། །ཀ<《༌དཀ»Pཔ་ས《་དཀ»། ལོ་བ་དང་ས《'»། ཀོ་ས་ལའི《་། papizz་པ་། ::ཀྱི། དྲ་པ《ཁམས་པའི་བར་ལ་ཟླ་D། s་Rg<<་ལོང་བསྡུར་མཆན། 《་《་་་་{D། ཁོཚོས《ཚུ་ཀྱི་ཁོ་ཚོ་མ་ཏོང་<a»ཚོང་སྒྱུ་《*དེའུདམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །དེ་ཁོ་ན་ཉིད་ཀྱི་ཚིག་ལེའུར་བྱས་པའི་མན་ངག་གི་འགྲེལ་པ་བཞུགས་སོ། །«པེ་»«སྣར་+། «པེ་»«སྣར་»བའི། «པེ་»«སྣར་པོའི། «པེ་»«སྣར་»ཅེས། ཚེ«པེ་»ལ་པདྨ་ནི། «སྣར་»ལ། ཚེ«པེ་»«སྣར་བཞི། «པེ་»«སྣར་+དེ། །«པེ་»«སྣར་+མེ། ཨི«པེ་»«སྣར་»འདི། ®@«པེ་《སྣར་ཅེས། @«སྣར་རྩི། ཚེ«པེ་»«སྣར་ནོ། ཉི«པེ་»«སྣར་»+དེ། «པེ་»«སྣར་»—པ། ཁྱེ«པེ་»«སྣར་»དོན། ཀྱེ«པེ་»«སྣར་ལ། ཨེ«པེ་»«སྣར་པ། ཉི«པེ་»«སྣར་»ཏུ། «པེ་»«སྣར་ནི། «པེ་»«སྣར་»བསྒོམ། «པེ་«སྣར་+ནམ་མཁའི་དཔེ་དང་། «པེ་»«སྣར་»ལྷུང་། @«པེ་»《སྣར་བསྒོམ། «པེ་»«སྣར་»—སུ། «པེ་»«སྣར་»བསྒོམ། «པེ་»«སྣར་+ནི། «པེ་»«སྣར་རྣམ། ཚེ«པེ་»«སྣར་»པདྨའི། «པེ་»«སྣར་»+ལ། @«པེ་»«སྣར་»—པ། O«པེ་»«སྣར་པའོ། «པེ་»«སྣར་+ཀྱི། ཚེ«པེ་»«སྣར་པས། «སྣར་»+དེ། ཚེ«པེ་»«སྣར་ཡི། «པེ་»《སྣར་»གིས། «པེ་»«སྣར་རྗེ། @«པེ་»«སྣར་»+རྣམ་པ། O«པེ་»«སྣར་མ། «པེ་»«པེ་»«སྣར་ཅེས། @«པེ་»«སྣར་སྔོན། «པེ་»«སྣར་»ཅེས། པེ་»«སྣར་རྣམ། ཚེ«པེ་»«སྣར་—དུ། +།
