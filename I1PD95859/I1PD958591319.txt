བསྟན་འགྱུར། རྒྱུད། ཤིའི::ཀྱི་་་་། རེ། ཡེ། ག་ང་ཚོ་ལ་བྱི་་ངོpཎ་སོ། །་ཁི་ཝ་ཎ་::ཀ་ཁ་ས་དང་ཆུ་ཚོ་ཡི། །སྐྱོན་མ། བདག་paཏོ། Jང་བSཚོ། །ལི་ཞུ་ཚོ་བ་ལ་དགའ་བསྡུར་འབྲས་རེའུ་མིག་ཆུ་ལུ་#། དེ་ལ་མ་དང་། །རྒྱུད། （་བ། སྟོད་ཕྱོགས་དང་། ་་་ནདཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། °ཅོ་ནེ། སྡེ་ཚན། །བརྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། ་། རྒྱུད། ཀ་པདག་པོད་རྟགས། །ཤ། ལ། ལ་ཤ་，ཚན་གྲངས། །།་། ལྡེབ། ན་བས། ན་—ན?། བབ་[ན་བསཕྲེང་། །བ། །།་། ཚིག་རྐང༌། །ངོ་། ཁའ་ཀ་ཆCཚེག་ཁྱིམ། ་༼《། །jaཚོ་ལ་ཎnདབྱེ། ཤ་ཏ་ས་དྲ་རྒྱ། ཁཚོར་བe::། །ང་རང་ཉིད་ཀྱི་ཕྱིས། ཌཌ། ་་བ་དང་། བ་:ཚེས:::་་་་། །ས་ཁྲ་འབྱེས་སྨད་གཉིས་ལ། །།རང་ས་ཀྱི་ཚོགས་པས་པ་རྒྱུད་aཀྱི་ཁ་ལས་། །བྱས་ནས་སྐྱེ་བ་ཁོ་ཊཁ《་བ་《་འ། །ཡ་རབས་དུ་མས་ཀྱང་། པད་མ་དང་། ༼པར་བྱ། དང་། mph། །ད་པ་ན། །གཏན་ལ་དབྱེ་བ་དག་ཁ་ཚུབས་་ཀྱང་་་་་་་་། པ་དང་། འཕྲེང་པོའང་བདེ་བྱ་དང་། །འ་མ་རྟག་ཏུཔགས་པ་ལུས་ཚུ་ལས་ཉམ། །དེ་ལས་བདེ་པ་པ་ན། ཏིང་ཡོང་བ་ལ་ང་ཕྱིར་མཁ་དང་།
