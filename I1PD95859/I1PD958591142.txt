ཁ་མཁས། ཁ་ལོ:་ཁྲིད། །ཚལ་བ་ཏེ་ད་བ་ས། །ཉལ་ཁང་། བཏགས་ཀ་ཁ་ཁྲོ། བཀར་ནང་དྲུང་ཨེ་ཏ་ཀ་ཅོ་ཀ་གས་པའི་ལོ་ཏེཞུ་བ། ཕྱིན་བད་ཀན་སྐྱུག་ཁ་ཏུ་སྒྲུབ་ཀྱི་གདུལ་བའི་༄༅། །ཨུ་རྒྱན་གྱི་རིམ་པའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། ཀཀཀཚེ་དེའི་ལས་བྱ་བ། ཤོག་ཁབ་ཀུན་pཏོ། ཀp:ཁ་པ་བསུ། །མཚོ་ཁང་བ་[P?nཏེ། གནའ་མཁར་ਮਲੋਮਲgਰਵਾਇਓਮਰਾਨགaཀཚོ། aང་pན་ཁ་n?p་། ཏུ་ཏ་ཀོ་བ་པ་ལ་རྒྱུ་བ་:༄༅། །རྒྱ་གར་སྐད་དུ། ཨོ་རྔུན་ཀ་མ་ཏཱ་རེ་དེ་བཱི་སཱ་དྷ་ནཾ་®ནཱ་མི། བོད་སྐད་དུ། ཨུ་རྒྱན་གྱི་རིམ་པའི་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། བཅོམ་ལྡན་འདས་མ་ལྷ་མོ་འཕགས་མ་སྒྲོལ་མ་གནས་ཆེན་གྱི་དབང་ཕྱུག་མ་ལ་ཕྱག་འཚལ་ལོ༑། གསང་བའི་རྒྱུད་དུ་རྡོ་རྗེའི་ཚིག་འདི་གསུངས། །དེ་ཁོ་ན་ཉིད་རྟོགས་་་དཀའ་མན་ངག་འདི། །སྤྲོས་པ་དང་བྲལ་རྣམ་པར་རྟོག་མེད་ལས། །དང་པོ་་ཉིད་རྗེས་ལངས་ཏེ་བཤད་པར་བྱ། །ཆཉམ་ང་སྤངས་པའི་སེམས་ཀྱིས་ས་གཞིར་་བྲིས། ཀ་། མཎྜལ་བྱས་ཏེ་བདུད་རྩི་ལྔས་གང་བའི། །ཐོད་པ་རླུང་མེས་དག་བསྒོམ་བདུད་རྩི་འབར་བ་ཡིས། །ལུས་ལ་ཁྱབ་བྱུགས་ཕྱི་ནང་དག་བྱས་ནས། །འབོལ་བའི་སྟན་ལ་རྡོ་རྗེ་སྐྱིལ་ཀྲུང་དང་། ཚོ། །ཡེ་ཤེས་ཕྱག་རྒྱས་བརྟན་པས་རབ་་་འདུག་ནས། ྋ་། སྟོང་ཀྱང་སྣང་བའི་འགྲོ་གསུམ་ཐམས་ཅད་ལ། །རྨི་ལམ་སྒྱུ་མ་་དྲི་ཟའི་གྲོང་འདྲར་བལྟ། །སྙིང་ནང་མཁའ་དབུས་ཡེ་ཤེས་དྲི་བྲལ་བའི། ་། ཟླ་་+།
