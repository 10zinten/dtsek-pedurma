བསྟན་འགྱུར། རྒྱུད། ཤ་ཅད་ཀྱི་ཆེད་དུ་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་ཡོངས་སུ་བསྔོ་བ་དང་། ་་་བཟོད་པར་གསོལ་ལ་ཡེ་ཤེས་སེམས་དཔའ་གཤེགས་པར་བྱའོ། ?། དུད་པའི་སོལ་བ་ནི། །ཆོ་ག་ཇི་ལྟ་བར་གཏོར་མ་དང་མཆོད་པ་དང་བདུད་རྩིས་ཚིམ་པར་བྱས་པས། ཡང་་་་དག་པར་མཉེས་པ་ལས། རང་གི་འདོད་པའི་དོན་གསོལ་བར་བྱའོ། །དེ་ནས་ཆོས་ཐམས་ཅད་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་ཙམ་པ་སྟོང་པ་ཉིད་ཀྱི་རང་བཞིན་དུ་མོས་པར་བྱས་ནས། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨ་ཎཾ་ཨདྱ་ནུཏྤནྣ་ཏྭཱཏ་ཅེས་བྱ་་་བས་སྤྲོས་པ་མེད་པའི་ངང་དུ་གནས་པར་བྱ་བ་འདི་ནི་དུད་པའི་སོལ་བ་མ་སྒྲུབ་་་པའི་ཐབས་ཡིན་ནོ། །དངོས་གྲུབ་བརྙེས་པའི་སློབ་དཔོན་ཆེན་པོ་དཔལ་ནགས་་ཀྱི་རིན་ཆེན་ཞབས་ཀྱིས་མཛད་པའོ། །།ལཊ་ལ་ཀ་ཀ་ལ་ཁ་ਨਵੰਲੋਂਸੈਨਨਸ਼ਰਧਾਨਲੌਨਟབ་དང་ས་གནས་ཁག་བ་བཅུར་མཁན་པ་དཀར་བའི་གདན་ཁྲིའི་ཚེ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། ཆོས་ཚན་འདི་«པེ་»«སྣར་»པ་ལ་མི་བཞུགས་པས་དཔེ་སྡུར་བྱུང་མེད། གལ་ཏེ་ད་པལར་བྱེད་པདོན་ཀུ་བ་ཀྱི་ལོའི་ཚེ་ཞབསྡུར་འབྲས་རེའུ་མིགཀ་ཀ་གྱི་ཆ་ཎི་བ་དང་འབད་འབད་པ་ལ་དད་པ་རྒྱུད། （རིག་པ་ཆགས་ཡོད་པ་མ་ར%erཔེ་ཅིན། །སྣར་ཐང་། རྩོ་ནེ། དཔེ་གཞི། །སྡེ་དགེ། ལི། སྡེ་ཚན། བོད་རྟགས། ཚན་གྲངས། ལྡེབ། ན་བཕྲེང་། ཚ་ཚེག་ཁྱིམ། ཚོ། བན་ངའི་ངོསཉི་་བ་དང་་བའི་་་་་་། །
