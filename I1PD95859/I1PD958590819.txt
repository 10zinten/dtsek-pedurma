རྒྱུད། ཚལ་ལ་ལས་གamyaཀྱི་ཁོ་ཚོ་ཆོས་ཀྱི་དོར་བ། །།རྡོ་རྗེ་ར་བའི་ཐིག་ཡིན་ཏེ། །དེ་ནས་ཆ་ཆུང་ལྔ་གཉིས་ནི། །པདྨའི་འདབ་མའི་ཐིག་ཡིན་ནོ། །།དེ་བཞིན་དུ་ནི་ལྟེ་བ་སྟེ། ཏ། ནང་གི་ཆ་གཅིག་གེ་སར་ཡིན༑། དེ་ལྟར་ཆ་ཆུང་ལྔ་གཉིས་ལ། །ཌ། འདབ་མའི་བར་ཡང་ཤེས་པར་བྱ། །དབང་ལྡན་མཚམས་ནས་བརྩམས་ནས་ནི། །གཡས་སྐོར་གྱིས་ནི་བསྐོར་བྱ་སྟེ། །དཀྱིལ་འཁོར་གྱི་ནི་ཐིག་རྣམས་ལ། །ཡེ་ཤེས་ཐིག་ནི་བསྟིམ་པར་བྱ། །དེ་ནས་དེ་རྗེས་ཉིད་དུ་ནི། །།།རྡོ་རྗེ་སློབ་དཔོན་མཉམ་གཞག་པས། །ཀ། ཧི་ཧིའི་སྒྲ་ནི་བརྗོད་བྱ་ཞིང་། །ཡིད་ཀྱིས་སྒོ་ནི་དབྱེ་བར་བྱ། །།ཨོཾ་ཧཱུཾ་ཏཱཾ་ཨ་ཁཾ་འདིས་ནི། །རྡུལ་ཚོན་ཀྱང་ནི་བསྒྲུབ་བྱ་སྟེ། །ཇི་ལྟར་ཚོགས་པ་རྙེད་གྱུར་ནས། །དག་པའི་དྲི་རྣམས་དབུལ་བར་བྱ། །གསལ་བཅས་ལྷ་ཡི་ཕྱག་རྒྱས་ནི། །།སྔགས་ནི་བརྒྱ་་རྩ་བརྒྱད་དུ་བཟླས། །ཟ། དྲིལ་དཀྲོལ་རྡོ་རྗེ་གསོར་བ་དང་། ཀྟི་། རྡོ་རྗེའི་གླུས་ནི་སྐུལ་བྱེད་ཅིང་། །དམར་བའི་གོས་བཅས་གཏོར་བྱིན་ཏེ། ལུ་ཁ། འདི་ལྟ་བུ་ནི་་བསམ་པར་བྱ། །འདི་ནི་ཆོས་དབྱིངས་དག་པ་སྟེ། །ཐམས་ཅད་ཁམས་ནི་རྣམ་དག་པ། །རང་ཉིད་སངས་རྒྱས་ཐམས་ཅད་དེ། ང། དེ་བཞིན་གཤེགས་པ་ཀུན་་་ནས་གནས། །གཡོན་པའི་ཁུ་ཚུར་གྱིས་ཐོག་མར། །ཚོ། དབང་ལྡན་ནས་ནི་བསྐོར་བྱ་སྟེ། །འཁོར་ལོའི་གཙོ་བོའི་རྗེས་འབྲངས་ནས། །།སྔགས་པས་རྡུལ་ཚོན་བྲི་བར་བྱ། །ཨོཾ་བཛྲ:；ཙི་ཏྲ་ས་མ་ཡ་ཧཱུྃ། ཕྱིས་ནི་སློབ་མ་:བཟང་པོ་ཡིས། །ཅི་རིགས་པར་ནི་བྲི་བར་བྱ། །དྲང་དང་འཁྱོག་དང་ཆད་པ་དང་། །ཕྲ་དང་སྦོམ་པ་ཉེར་མཚོན་ཏེ། །།སྦོམ་པོ་ཡིས་ནི་ནད་འགྱུར་ཞིང་། །་། ཕྲ་ན་ནོར་རྣམས་ཟད་པར་འགྱུར། །འཁྱོག་ོན་འཐབ་རྩོད་འབྱུང་འགྱུར་ཞིང་། ཞ་ཏེ། ཆད་ན་སློབ་་དཔོན་སློབ་མ་འཆི། །དེ་ལ་སོགས་པའི་སྐྱོན་སྤངས་ནས། ཁ་། རྡུལ་ཚོན་རི་མོ་དགྱེ་བར་བྱ། །ཤར་དུ་སྔོན་པོ་ཆེན་པོ་སྟེ། །ལྷོར་ནི་སེར་པོ་འབྱུང་བ་ཡིན། །།
