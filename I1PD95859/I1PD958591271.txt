བསྟན་འགྱུར། རྒྱུད། བཀོད༑། འགལ་གཟིགས་མཁས་རྣམས་བཟོད་པ་མཛད། །དད་པོའི་ལས་་་ཅན་དམན་དང་བར་མ་མཆོག་དག་ལས། །དངོས་གྲུབ་སྐལ་བ་:རང་རང་སོ་་་སོར་མོས་པ་རྣམས། །དད་ཅིང་མོས་པ་ཐམས་ཅད་མཐར་ཕྱིན་རྩེ་མོ་དགོད་པར་བྱ་ཕྱིར་བྲིས། །ཕན་ཡོན་དགོས་པ་འདི་ཡིས་གང་གིས་བསྒྲུབ་པ་མྱུར་དུ་འགྲུབ་་པར་ཤོག། ཐུགས་རྗེ་སྒྲོལ་མའི་སྒྲུབ་ཐབས་དངོས་གྲུབ་ཕྲེང་བ་ཞེས་བྱ་བ། །སློབ་དཔོན་རྟག་པའི་བཛྲས་མཛད་པ་རྫོགས་སོ། །།།།ཚེ་རིང་བ་དང་། རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་དོན་ཡོད་རྡོ་རྗེ་དང་། །བོད་ཀྱི་ལོ་ཙཱ་ཅི་་་་མཁས་པ་བ་རིས་རྒྱ་དཔེ་བོད་དུ་སྤྱན་དྲངས་ནས། །བ་ཕྲུག་རོང་གི་གསེར་ཁང་དུ་བསྒྱུར་:ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།་ལུལ་དེ་མ་འོ་ཅིཀ་ཀ་མ་ཏ་ལམ་ཀྱི་॥ਤਾਸਹਸਸਥਾਨ'ਚਰਨਪੈਮਾਹੌਲਥ'ਸੰਨད་་་་་་་བསམ་བ་ཀུན་ཏོན་ཏུ་བཤད༌::ཀྱི་རྗེ་པའིཀྱརཀྱི་ཡོད་ལ། ལྔ་གཅཏེ་བྱ་གང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཐུགས་རྗེ་་་་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ་བཞུགས་སོ། །«སྣར་»སཱདྷ། «པེ་»«སྣར་»ཞུགས། «པེ་»«སྣར་»རྒྱུད། «པེ་»«སྣར་»ཀྱི། ཚི«པེ་»«སྣར་བས། །«པེ་»«སྣར་»བསྡམ། «པེ་»«སྣར་»ཉན། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»བོ། ཡོ«པེ་«སྣར་»འཆང་། «པེ་»«སྣར་»གཉིས། ག«པེ་»«སྣར་བསྟོད། «པེ་»«སྣར་»ཀར། «པེ་»ཤེད། ཚེ«སྣར་བསྲེགའགྱེ«པེ་»«སྣར་»ཏྭཾ། («པེ་»«སྣར་»ཀྱི། ་«པེ་»«སྣར་པ། «པེ་»«སྣར་»ཞིང་། «པེ་»«སྣར་ཡིས། «པེ་»སྣར་ནས། །«པེ་»«སྣར་»ཚེར། @«པེ་»«སྣར་»བོར། @«པེ་»སྣར་བཏབས། ཉི«པེ་»«སྣར་ཆོད། «པེ་»«སྣར་ཀྱི། («པེ་»«སྣར་ན། ཀྱེ«པེ་»«སྣར་»ཟླས་ཀྱང་། ཉེ«པེ་»«སྣར་»སྤུ་རྩེ་མོ། ཉེ«པེ་»«སྣར་དུ། ཆེ«པེ་»«སྣར་»ཡོགས་དང་། «པེ་»«སྣར་»ངགས། «པེ་»«སྣར་དྲི། ཁྱེ«སྣར་»+པ། དང་། རྒྱ་བཟང་བརྒྱ་བ། ་དང་། །ཆོ་«པེ་»«སྣར་»རང་རང༌། «པེ་»«སྣར་བསྒྲུབས། Q«པེ་»«སྣར་ཙ། «པེ་»«སྣར་ཞིང་། ཚེ«པེ་»«སྣར་པའོ། ::ཀྱི།
