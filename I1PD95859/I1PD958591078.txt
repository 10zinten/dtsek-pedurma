རྒྱུད། །སྣར་སྐྱབསག་སྤྲིན་མེད་པའི་ཟླ་བ་ལྟ་ལུ་ལང་ཚོ་ལུ་བ་དང་། འདུ་པ། ཆུ་«པེ་»«སྣར་»ཀྲུ་ལོ་ཙ། །བག་ས་མཚབ་པ་གཡོན་པ་ན་ཀླུ་ལ་སོགས་་བོ་་དུ་མར་གྱེར་བ་འགྲོ་བ། །པོ་ཆེའི་རྒྱན་་ཚོགས་ཀྱི་སྐུ་ཚེ་དང་། ད་དུ། པདྨ་དང་ཟླ་བ་དང་འདུ་་བ་པོ། ས་བཞི་པ། ལ་པོ་ལ་འདུ་gyབ་༌། བསྡུར་འབྲས་རང་མགན། །དེ་དག་རྒྱུད། (:་པ། *གུ། །ཁ་པའི་སྤྲུལ་བ། །(ལུ་ང་དཔེ་གཞི། ༼སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། རྒྱུད་འགྲེལ། བོད་རྟགས། ཚན་གྲངས། །པ་་ལྡེབ། ན་ན་ན་ན་། ན། ན་ན་ཕྲེང་། །ཚིག་རྐང་། །ཚེག་ཁྱིམ། [། །ཏུ་མཁོ་བར་བ་ལག་བསྤྱི་ལེ་ཉིད་ཀྱིས་ང་ཚོ། །ཁ་ན་བ། མི་ག་ང་་:/དགས་པ་དེས་མི་བྱེད་པ་ཟླ་བ་བ་འདོ། །ཀ་ཐོག་མཁ་ཆཎ་ཊ་ཚོ་ཕྱིར། །ལྟ་བ་གང་ཀལ་ཝིཏ་ཁ་ར་ཎཝཾ་ཉཞུདཀ་ཡ་ཏཁྱེཔ་ཀ་ཏོ་དེ་ཁབ་མ་ཤེས་པ་ལོ་་་་་་::ཀྱིང་ལ་དཆཚོ་ལྔ་ལ་བཞy། ཅཁ་ས་ཁང་ཚོགས་ཁག་སོང་བ་ལ་འབྲི་ཀཚ་ཁཥལོ་ཏི་ཏི་ར་བ་གང་དེ་ལོ་ཀོ་པའི་ཀ་དང་ར་བ་བ་ན་ཕྱོགས་འདུ་གལ་ཁབཀལ་ལ་ལོ་ཡག་ཁོ་ཟ་བའི་རྒྱ་ཞུ་ཁྲིད་དང་ཕྱི་འཁྱི། །ན་པ། །དྡྷ་ཀ་རྒྱུ་bePཆེ་ཏsཏི་ཞི་ཁnཏུ་ཤོད་བོདབློག་དང་། ཁ་ཅཏུ་ཀ་ཀ་ཆ་ཡཾ། །ཏོཀ་ར། ''ཎཎྜི་ཊི་ལ་ཁ་བློ་བཏཱ་ཊ་ཀུ་ཊཀ་ཎྜཁ་ན་ཌར་་་་ཌོ་ཆ་ཎཱ་ལ་ཀྵ་ཏཚོ་ཎིད་འཊཾ་ཀ་ཊ་ཎི་་བློ་། ཆུ་ཏཁབ་ལ་ཀཀཀ་་།
