བསྟན་འགྱུར། སེམས་ཙམ། ༼ཕྱིའིམཐར་ཐུག་པའི་སྐབས་ཞེས་བྱ་བ་སྟེ། ལེའུ་ཉི་ཤུ་རྩ་གཅིག་པའོ། །ཡ་ཡ། །ཐེག་པ་ཆེན་པོ་མདོ་སྡེའི་རྒྱན་གྱི་བཤད་པ་རྫོགས་སོ། །་་འ། ང་རཏུ་རྒྱ་གར་གྱི་མཁན་པོ་ཤཱཀྱ་སིདྷ་དང༌། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་དཔལ་བརྩེགས་ལ་སོགས་པས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །དེ་ལ། ད་ཐེག་ཆེན་མདོ་སྡེ་ཀུན་གྱི་རྒྱན། །བཤད་པ་འདིའི་གཞུང་ཚད་ནི། །ཤ་ལོ་ཀ་སུམ་སྟོང་དག་དང་རྩ། །དྲུག་བརྒྱ་ལྔ་བཅུ་ཡིན་པར་བསྒྲགས། །།།ਸਲਾਲਾਲਾਸਨਸਿੰਘਕਾਕBAਲਾਲੈ་བསྡུར་མཆན། །སྟེ། རྒྱུ་དེ་ཡག་བྱས་ནཔ་དང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། ཐེག་པ་ཆེན་པོ་མདོ་སྡེའི་རྒྱན་གྱི་བཤད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་»ཏཱ་ལ་ཆ་ར་བྷ་ཎ། «པེ་»«སྣར་བྱེད། «པེ་བཙམས། གིུ«པེ་»«སྣར་»རྒྱན། ༄༅«སྣར་པ། །«པེ་»«སྣར་པས། «པེ་»«སྣར་»བའི། @«སྣར་»རྗེའི། «སྣར་བརྒྱན། «པེ་»«སྣར་»། «སྣར་པ། ཚེ«པེ་»«སྣར་རྣམས། ཉི«སྣར་»གརྡུངས། «སྣར་བྱེ། །(ཀྱེ«པེ་»«སྣར་དེ། ཚེ«པེ་»«སྣར་»པ། «པེ་»«སྣར་+དེ། «སྣར་»གིས། འ«སྣར་རྒྱན། «སྣར་»ཀྱི། མུ«སྣར་དགའི། །པོ«པེ་ཁྱད་པར། «སྣར་བྱད་པ་ཐ། ཨེ«པེ་»«སྣར་»གེར། ཉི«པེ་»«སྣར་བཀའ། ཚེ«པེ་»«སྣར་»གི། 《སྣར་»གཉིད།  ེ«སྣར་ཀྱི། «པེ་»སྣར་པ། ®«པེ་»སྣར་»སྙམ། །«པེ་»«སྣར་པར། «པེ་»འདི། «པེ་»«སྣར་+དེ། ︽«པེ་»«སྣར་དཀའ། ཚེ«པེ་»«སྣར་+འདི། «སྣར་བ། ་«པེ་»«སྣར་»བ། ®«པེ་»+གང་ཡིན་པ། «སྣར་»གང་ཡིན། «སྣར་ནི། «སྣར་»དུ། ཨེ«པེ་»«སྣར་»—ཞེ་ན། «པེ་»«སྣར་»+ཏེ། «པེ་»བར། སྣར་»བ། «སྣར་སྟོན། ཚེ«སྣར་»གྱིས། «པེ་»«སྣར་»+ཡང་། ༄༅«པེ་»«སྣར་»+ཉིད་མེད་པ། «པེ་»ན། ཚེ«པེ་»«སྣར་»—ཡང་། @«སྣར་དིང་། &«པེ་»«སྣར་»རྟེན། །ཚེ«པེ་»—།
