བསྟན་འགྱུར། སེམས་ཙམ། ༼ཝི། དཔོན་ཐོགས་མེད་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།་་་ཏ། །བྱ་བ། ཁ་དཔལ་གྲོང་ཁྱེར་ཆེན་པོ་དཔེ་མེད་ཀྱི་མཁས་པ་ཆེན་པོ་བྲམ་ཟེ་རིན་ཆེན་རྡོ་་རྗེའི་དབོན་པོ། པཎྜི་ཏ་མཁས་པ་ཆེན་པོ་སད་ཛ་ན་དང༌། ལོ་ཙཱ་བ་ཤཱཀྱའི་དགེ་སློང་བློ་ལྡན་ཤེས་རབ་ཀྱིས། གྲོང་ཁྱེར་དཔེ་མེད་དེ་ཉིད་དུ་བསྒྱུར་བའོ། །་་། ਰਸ,ਇਸੇਕਾਰਨਨਰਮਾਕਈਮੋਲਲਮਕਬਰਨੂੰབསྡུར་མཆན། །་་་ལུ་ཏཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་། ཐེག་པ་ཆེན་པོ་རྒྱུད་བླ་མའི་བསྟན་བཅོས་ཀྱི་རྣམ་པར་བཤད་པ་འཕགས་པ་ཐོགས་མེད་ཀྱིས་མཛད་པ་བཞུགས་སོ། །ཞེས་:འཁོད། 《«པེ་»«སྣར་ཥཱ། «པེ་»«སྣར་»གྱིས། «སྣར་»བཞི། མུ«པེ་»«སྣར་བསམ། «པེ་»«སྣར་+ནི། ༄༅«པེ་»«སྣར་»ན། །«པེ་»«སྣར་»+གྱིས། ཉི«པེ་»«སྣར་རྗེའི། «པེ་»«སྣར་»ཏེ། «སྣར་»གིས། «པེ་»«སྣར་བལྟ། @«པེ་»«སྣར་པ། །:གནས་པ་དང་ད ེ«སྣར་ཚིགས། «སྣར་—ལས། «པེ་»«སྣར་»—ཆོས་ཀྱི། «སྣར་འཕྲེང་། «པེ་»«སྣར་»གའི། @«སྣར་རྟོགས། ཚེ«པེ་»«སྣར་དེའི། ཉི«པེ་མཁས། «སྣར་མ། །«པེ་»«སྣར་ཀྱིས། «པེ་»«སྣར་རྣམ། «པེ་»«སྣར་»གིས། «སྣར་བཞིའི། «པེ་»«སྣར་»གཞི། «པེ་«སྣར་»ཤིན་ཏུ། «པེ་»«སྣར་+ལ། ཚེ«སྣར་ཀྱིས། «པེ་»«སྣར་_ལ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»དེ། «པེ་»«སྣར་»+མངའ་བ། «སྣར་»གྱིས། །®«པེ་»ཐུགས། «པེ་»«སྣར་བསྒྲུབས། «སྣར་སྟོད། «པེ་»«སྣར་»ཡོངས་སུ། «པེ་»«སྣར་»རྗེའི། «པེ་»སྣར་བསྒྲུབ། «སྣར་བཞིའི། «སྣར་»ལྟ། «པེ་»གཞིའི་ལེའུ་རྗེས་ཐུགས། «སྣར་བཞིའི་ལེའུ་རྗེས་ཐོགས། མུ«པེ་»«སྣར་»གྱིས། ཚོ«པེ་»«སྣར་ཟས། ལས་«པེ་»«སྣར་»བའི། «སྣར་»རྣམས། @«པེ་»«སྣར་»པ། «པེ་»«སྣར་གྱུར། །«སྣར་»ཕྱི། @«པེ་»«སྣར་—མི། ཚེ«པེ་»«སྣར་པའོ། «པེ་»«སྣར་རྣམས། །ཚེ«པེ་+སྙིང་། «སྣར་འཐར། «པེ་»«སྣར་»རྣམ་པར། «པེ་»«སྣར་བརྟུལ། ཚུ«པེ་»།
