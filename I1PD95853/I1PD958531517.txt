བསྟན་འགྱུར། རྒྱུད། བཧུཾ་ཧཱུྃ་ཕཊ། ༀ་དྲུ་མཙྪཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཕཊ་ཕཊ། ཨོཾ་གུ་རི་ཀ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཨེ་རཱ་བ་ཏཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ད་ཧ་ད་ཧ་བཛྲ་ཛ་ཊི་ལ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་མ་ཧཱ་བྷི་ར་ཝཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་པ་ཙ་པ་ཙ་མ་ཧཱ་བཱི་ར་ཧུ་ཧཱུྃ་ཕཊ། ༀ་བཱ་ཡུ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོ་བྷཀྵ་བྷཀྵ་བཛྲ་ཧཱུྃ་ཀ་ར་བ་ས་རུ་དྷི་རཱན་ཏྲ་མཱ་ལཱ་བ་ལཾ་བི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་སུ་ར་བྷཀྵི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་གྲོཏྟ་གྲིཉྩ་སུ་བྷ་དྲ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བྷུ་ཛཾ་ག་སཟླ་བཱ་ཏརྫ་ཡ་ཏརྫ་ཡ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བྱཱ་མ་དེ་ཝཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཱ་ཀརྫུ་ཨཱ་ཀརྫུ་བཛྲ་བྷི་ར་བ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་སུ་བྷ་དྲེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧྲི་ཧྲི་མ་ཧཱ་བྷི་ར་བ་ཧུཾ་ཧཱུྃ་་ཕཊ༑ཨོཾ་ཧ་ཡ་ཀརྞ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་:ཛྙཱ་ཙྩཾ་བི་རཱུ་པཱཀྵ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་ཁ་གཱ་ན་ནེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་གྷཾ་གྷཾ་མ་ཧཱ་བ་ལ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་ཙཀྲ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཾ་ཧཾ་རཏྣ་བཛྲ་ཧུ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་རོ་ཧི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཧི་ཧི་ཧ་ཡ་གྲྭ་བ་ཧུ་ཧཱུྃ་ཕཊ། །ཨོ་:ཤོཎ་ཌི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཨཱ་ཀཱ་ཤ་གབྷ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་ཀི་ལི་ཀི་ལི་ཧེ་རུ་ཀ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་སུ་བཱི་རེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་སི་ལི་སི་ལི་པདྨ་ནརྟེ་ཤྭ་རི་ཧཱུྃ་ཧཱུྃ་ཕཊ༑ཨོཾ་མ་ཧཱ་བ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཧི་ལི་ཧི་ལི་བི་རོ་ཙ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་དྷི་ལི་དྷི་ལི་བཛྲ་ས་ཏྭ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཙཀྲ་བརྨི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་མ་ཧཱ་བཱིརྱ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་ལཱ་མྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་རོ་ཧི་ནི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་རུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀཱ་ཀཱ་ཀྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཨུ་ལུ་ཀཱ་སྱེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་ཤཱ་ནཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཤུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་རཱ་དྷཱི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་རཱུ་ཏི་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་ཡ་མ་དཾབྷཱི་ཎི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐཱ་ནི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏི་བཛྲ་བཱ་རཱ་ཧཱི་བཻ་ཧུ་ཧཱུྃ་ཕཊ། ༀ་ན་མ་ཨཱརྱ་ཨ་པ་རཱ་ཛི་ཏེ་ཏྲི་ལོ་ཀྱ་མཱ་ཏེ་མ་ཧཱ་བི་དྱེ་ཤྭ་རཱི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ན་མསྶརྦ་བྷུ་ཏ་བྷ་ཡཱ་བ་ཧེ་མ་ཧཱ་བཛྲ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་ན་མོ་བཛྲ་ཨཱ་།
