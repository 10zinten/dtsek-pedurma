སོགས་བསྟན་འགྱུར། ་་རྒྱུད། ཞེ་ཡོད། དེ་ལ་། འདས་པའི་སངས་རྒྱས་རྣམས་ཀྱིས་གསུངས། །མ་འོངས་རྣམས་ཀྱང་གསུང་འགྱུར་ལ༑། ད་ལྟར་བྱུང་བ་རྫོགས་སངས་རྒྱས། །ཡང་དང་ཡང་དུ་གསུངས་པ་གང་། །རྒྱུད་ཆེན་སྒྱུ་འཕྲུལ་དྲ་བ་ལས། །རྡོ་རྗེ་འཆང་ཆེན་གསང་སྔགས་འཆང་། །དཔག་མེད་རྣམས་ཀྱི་བཀའ་བཞིན་དུ། །གླུར་བླངས་གང་ཡིན་འདིར་བཤད་པ། །མངོན་པར་རྟོགས་པར་བྱ་བའི་ཕྱིར། །སྒྲུབ་པའི་ཐབས་འདི་བདག་གིས་བྱས། །དེ་དགེས་སེམས་ཅན་མ་ལུས་པ། །མི་ཤེས་ཐེ་ཚོམ་ཀུན་བསལ་ནས། །ཡེ་ཤེས་སྐུ་དེ་མྱུར་ཐོབ་ཤོག། འཕགས་པ་འཇམ་དཔལ་གྱི་མཚན་ཡང་དག་པར་བརྗོད་པའི་མངོན་པར་རྟོགས་པ་ཞེས་བྱ་བ། བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་་་ཐུགས་རྗེ་ཆེན་པོ་སྤྱན་རས་གཟིགས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།བ། བྱབསྡུར་མཆན། བསོ::ཀྱིངབདས་པོ་་ས་་་་(མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»《སྣར་»འཇམ་དཔལ་མཚན་བརྗོད་ཀྱི་མངོན་རྟོགས་བཞུགས་སོ། །«པེ་《སྣར་»ཏྭཱ། «པེ་»«སྣར་»འདུལ། «པེ་»«སྣར་»སྤངས། ཀྱི«པེ་»«སྣར་+ནི། Q«སྣར་བསྟན། «པེ་»«སྣར་»པ། «སྣར་ལེན། །@«པེ་»«སྣར་ཐ། «པེ་»«སྣར་»ལ། @«པེ་»གཱིརྟི་དང་། ནིར། «སྣར་»གི་ཏི་དང་། ནིར། །®«པེ་»《སྣར་བསླང་། །«པེ་»«སྣར་»ཅི་བགྱིས་པ། «པེ་»«སྣར་»གི། «པེ་»«སྣར་+གང་། «པེ་»«སྣར་»སྐོར། «པེ་»«སྣར་»གི། «པེ་»«སྣར་འཕངས། ཆོ«པེ་»«སྣར་བཏང་། །མ«པེ་»«སྣར་»ལ། «པེ་»«སྣར་གཤེས། «པེ་»«སྣར་»གོ་ཏལ། «པེ་»དཔའ། «པེ་»«སྣར་ནུབ། ཡིན་པ། «སྣར་གསོད། @«པེ་»«སྣར་»སེམས་ཅན་ཐམས་ཅད། «པེ་»«སྣར་སྒྲོལ། ཉི«པེ་»«སྣར་»བཏང་། «པེ་»«སྣར་»+དགའ་བའི། «པེ་»«སྣར་»གིས། 《ཅོ་»འཐོབ། (《སྣར་མངོན། «པེ་»«སྣར་»སྐྱུགས་སུ། «པེ་»«སྣར་»ཨཱཏྱཱ། «པེ་»བསག་«པེ་»«སྣར་»+གང་། ཚེ«སྣར་དེ་བདེ། །«ཅོ་»།
