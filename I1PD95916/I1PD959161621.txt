བསྟན་འགྱུར། སེམས་ཙམ། མིའིཔ་དང་ནི་མང་པོ་དང༌། །ཆོས་ཚོལ་བ་དང་བཤད་པ་དང་། །དེ་བཞིན་དུ་ནི་སྒྲུབ་པ་དང༌། །ཡང་དག་གདམས་ཤིང་རྗེས་སུ་བསྟན། །།ཞེས་བྱ་བ་ལ། ཇི་ལྟར་བསླབ་པར་བྱ་བའི་སྐབས་ལས་འདི་ཡན་ཆད་དུ་མོས་པ་མང་བ་དང་། ཆོས་ཚོལ་བ་དང༌། ཆོས་བཤད་པ་དང་། དེ་བཞིན་དུ་བསྒྲུབ་པ་དང་། ཡང་དག་པར་གཏམས་ཤིང་རྗེས་སུ་བསྟན་པ་སྟེ། དེ་རྣམས་བསྟན་ཏོ་ཞེས་བྱ་བའི་་་དོན་ཏོ། །གདམས་ཤིང་རྗེས་སུ་བསྟན་པའི་སྐབས་བཤད་པ་རྫོགས་སོ། །།།ਤੇਸਨਕਿཤེས་དཔར་འགན་བསྡུར་མཆན། དེ་ལ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»ཏཱ་ཀཱ། «པེ་»«སྣར་བྷ། ®«པེ་《སྣར་བ། །«སྣར་འབྲེལ། «པེ་»«སྣར་»སྡེ། @«པེ་»《སྣར་དཔའ། །(«པེ་»«སྣར་»སེམས། «པེ་»«སྣར་»སྟོན། ཉེ«པེ་»«སྣར་བསྟན། «པེ་»«སྣར་བསྡུས། «སྣར་བསྒྲུབས། «པེ་»བསྒྲུབས། «པེ་»«སྣར་པ། «པེ་»«སྣར་པ། «པེ་»《སྣར་»—པ། (ཁྱེ«པེ་»«སྣར་རྟོགས། («སྣར་བསྒྲུབས། ལྕེ«པེ་»«སྣར་—པ། ་འབུ«པེ་»་་་པ་ཆ་པ་《སྣར་ལས་བལྟོས། ་(«པེ་»«སྣར་»སོ། «པེ་»«སྣར་»བ། «པེ་»«སྣར་རིགས། «པེ་ལ། ་གྱི«སྣར་རིགས། «པེ་《སྣར་»རྩེ། «པེ་»ན། «སྣར་»སོ། «པེ་»«སྣར་—དག། «པེ་《སྣར་མ། «པེ་»«སྣར་བཞག«པེ་»«སྣར་»«ཅོ་»བསྐྱེད། །(«ཅོ་»བསྐྱེད། «པེ་»སྣར་མ། «པེ་»«སྣར་པས། «པེ་»«སྣར་»+ཕྱིར། ལས་(ི«པེ་»«སྣར་+ནི། «པེ་»«སྣར་བྱའོ། «པེ་»«སྣར་»+བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི། «པེ་»«སྣར་»+ལ། །«པེ་»«སྣར་»མདོ། །(«པེ་»«སྣར་»མདོ།  ེ«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་ལས། ་མ«པེ་»«སྣར་རྟོགས། པོ«པེ་»«སྣར་+མ། @«པེ་»«སྣར་པ། «པེ་»«སྣར་»+ཇི། «པེ་»«སྣར་—ལ། ཁྱེ«པེ་»«སྣར་པར། «པེ་《སྣར་»—པ། «པེ་»«སྣར་»+དུ། «པེ་»«སྣར་བརྩལ། «པེ་»«སྣར་»ནི། །
