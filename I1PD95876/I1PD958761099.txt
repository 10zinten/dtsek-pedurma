ཀ་བ་ཀ་ཨཀྵབསྟན་འགྱུར། རྒྱུད། ཀུའི་་ཀལ་ལ་བློ་ལོ་འཁོར་ཆེན་པོའི་ཆོ་ག། རྡོ་རྗེ་ཐམས་ཅད་འབྱུང་བ་ཞེས་བྱ་བ། རྡོ་རྗེ་སློབ་དཔོན་ཆེན་པོ་ཀུན་དགའ་སྙིང་པོའི་ཞལ་སྔ་ནས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།ལས་ྋརྒྱ་གར་གྱི་མཁན་པོ་ཨཱ་ཙཱརྱ་བུདྡྷ་ཤྲཱི་ཤཱནྟི་དང༌། །སྐྱེ་ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།:གདུག། ཀཥ་བ:དཔལ་འབྱས། པ་དེ་ལས་འགྲོ་འོང་དང་ཚོ་བའི་རྒྱུ་བ་ན། །ਤਾਜੋਫਸਲ'ਲੋਕਲਸੁਜਰਧਲਕੇਜੋਖੁਸਸਸਸਰਕਾਰཡོན། ཚོ་ཚུལ་བསྡུར་མཆན། །ཀྱི་བྱ་བ་ཁོ་བ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»«སྣར་»+པའི། «སྣར་»འགགྱིས། «པེ་»«སྣར་ཧྲི། («པེ་»བརླབས། «པེ་»«སྣར་བསམ་ཞིང་། *«པེ་»«སྣར་གྷིནྣཾ། «པེ་»《སྣར་»ཧཱུྃ། (འདིའི་འོག་གི་ཧཱུྃ་ཞེས་པའི་ཡི་གེ་རྣམས་«པེ་»«སྣར་»ལ་ཟླ་ཚེས་ཡོད་མེད་ཅི་རིགས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། («པེ་»སྣར་རིན་ཏི་རའི། «པེ་བཅིངས་པར། @«པེ་»«སྣར་བཅིངས། @«པེ་»《སྣར་རིན་ཏི། དགྱེ«པེ་»«སྣར་མཾ། དཀྱེ«པེ་»«སྣར་བཞག(འདིའི་འོག་གི་གཞག་ཅེས་པ་རྣམས་་་་་«པེ་《སྣར་»ལ་བཞག་ཅེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»རིན་ཏི་རིའི། «སྣར་རིན་ཏི་རའི༑་«པེ་འཕྲེང་། ཨེ«པེ་»«སྣར་—ཁུ་ཚུར། ལོ«པེ་»«སྣར་བསྲེག«པེ་»«སྣར་བྷཱཉྩ། འབྱེ«པེ་»«སྣར་བྱ་སྟེ། ཁྱེ«པེ་»«སྣར་»བསྒྲེང་། ཀྱེ«པེ་»«སྣར་»ཏྲི། བབ་(«པེ་»བྷཱ་བ་རཀྟ། 《སྣར་བྷ་རྦཀྵ། (འ«པེ་《སྣར་སྤྱན། ༼«པེ་《སྣར་ལོ། །«ཅོ་བ། «པེ་»«སྣར་བྲི། །«པེ་»«སྣར་»གདན། «པེ་»«སྣར་»པཱ་ཏཾ་ཀེ་པཱ་ཏཾ་གེ་ནི་ར་ཏྲཊ། «པེ་»ལས། འཛུབ། 《སྣར་»ལས། མཛུབ། «པེ་»«སྣར་»གཞིབས། «པེ་»«སྣར་»ཀྱིས། །«པེ་»སྙིང་གར། །«སྣར་ཉིད་གར། @«པེ་《སྣར་—ཕྱོགས། «པེ་《སྣར་ཀྱིས། ®«པེ་«སྣར་»ཨོ། «སྣར་»ཀྱི། «པེ་»«སྣར་»ཏེ། ཧཱུྃ། ༄༅«སྣར་»འཁྲིག་«པེ་»«སྣར་»ཀྱིས༑«པེ་»«སྣར་»སྦྲེལ། ཚེ«པེ་»རྗེ། འགྱེ«པེ་»«སྣར་པོ། «པེ་»«སྣར་»བཞག་ལ་བལྟ། :ཀྱི«པེ་»«སྣར་»དང་། སྐྱེ«པེ་»«སྣར་»མི། །«པེ་»«སྣར་»པུ། «པེ་»«སྣར་»མཾ་ཞེས་པ། «པེ་»«སྣར་»བར། «པེ་»«སྣར་ཨཱཏྟ་མཱ། «པེ་»«སྣར་»མཾ། «པེ་»«སྣར་ཨཱཏྨ། ༄༅«པེ་»«སྣར་མཾ། «པེ་»སྤྱི་བོ་ས་ལ། «པེ་»«སྣར་»ཡ་ཨཱཏྟཱ། དགྱེ«པེ་»«སྣར་»ཀ་རུ་མཾ། «སྣར་རིགས། མེ«པེ་»«སྣར་»—དང་། །།
