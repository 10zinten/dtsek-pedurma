དང་། །ཞབས་བསིལ་བསྒྲུབ་དང་དབང་བསྐུར་དང༌། །དམ་ཚིག་བསྒོ་དང་མཆོད་ཡོན་མཆོད། །བསང་གཏོར་འཐོར་འཐུང་སྲེག་བླུགས་དང་། །མེ་་བརྟག་བསང་དང་དགང་བླུགས་དང༌། །ཀུན་བསྲེགས་བསང་གཏོར་གསོལ་་བ་དང༌། །ཐབ་དང་མེ་ལྷ་དགོད་པ་དང༌། །བསྡུ་དང་བསྟོད་དང་དགུག་པ་དང་། །སྔོན་བཞིན་ཕྱག་རྒྱ་དབང་བསྐུར་དང༌། །མཆོད་དང་མཆོད་ཡོན་་འཐོར་འཐུང་བསང༌། །དགང་བླུགས་ཅིག་ཤོས་ལས་བསྐུལ་དང་། །ལྷ་རྣམས་གཟི་བྱིན་བསྐྱེད་པ་དང༌། །མེ་ལྷ་དགང་བླུགས་མཆོད་བསང་དང༌། །འཐོར་འཐུང་མཆོད་པ་གཤེགས་གསོལ་ལོ། །སྤྱན་དབྱེ་བའི་ཆོ་ག་སློབ་དཔོན་ཀུན་་་་དགའ་སྙིང་པོས་མཛད་པ་རྫོགས་སོ། །།།བ་ད་ན་བ་བལདེའི་པ། ་གགཔ་དེ་ཕྱིན་་བའི་ཆོ་ག། །འཛུམ་དང་བསྡུར་མཆན། དེ་ལས་ཕྱིར། །ེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །སྤྱན་དབྱེ་བའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་»བིདྡྷི། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་གོས། «པེ་»«སྣར་»+སུ། «པེ་《སྣར་»+པ། «པེ་»«སྣར་»ལོ། «པེ་»«སྣར་+པ། །།།།«པེ་»«སྣར་—བུམ་པ་རིགས་ལྔའི། «པེ་»«སྣར་»ལ་གཞག་གོ། སྐྱེ«པེ་»«སྣར་»—སྨན་ལྔ་དང་། «པེ་»«སྣར་»ཅིག«པེ་»«སྣར་»བུ། @«པེ་བསམ། @«པེ་»«སྣར་བཛྲ། ཟ«པེ་»«སྣར་+དག«པེ་»«སྣར་»«ཅོ་གི། །«པེ་»«སྣར་རིག ེ«པེ་པོ། «པེ་»«སྣར་»ཛ། ཚེ«པེ་《སྣར་ནས། «པེ་»«སྣར་ཨཱརྦ། «པེ་»«སྣར་ག་ཏ་ཡ་གྷཱ་ཏ་ཡ། «པེ་»«སྣར་»ཊེ་བཱིརྱ། ཉི«པེ་»«སྣར་»ལིཀའབྲ«པེ་»«སྣར་»སིདྡྷི་ཀ་གྱི«པེ་»«སྣར་བཅིང་བ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»+དུ། སྟེ«པེ་»«སྣར་གཏང་། སྐྱེ«པེ་»«སྣར་དགོད་པར་བྱ་བའི་ཕྱིར་ནི་རིགས་ཀྱི་རིགས་པས་སོ། །ལུ«པེ་»«སྣར་པ། །
