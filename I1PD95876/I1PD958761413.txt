རཁོ་ཚོ་བསྟན་འགྱུར། རྒྱུད། ཀུའི་ཏུ་:::《སྣར་གྱི་བསྒྲུབ། «པེ་»«སྣར་»གི། @«པེ་»«སྣར་བར། («པེ་《སྣར་བརླབས། «པེ་»《སྣར་མཉན། «པེ་»ཀྱི། །@«པེ་《སྣར་བས། &«སྣར་ཀྱི། ཨུ«པེ་»«སྣར་»དཀྱིལ། «པེ་»«སྣར་»འགྲོ། @«པེ་»«སྣར་གྱི། «པེ་»སྣར་པར། «པེ་»སྣར་»ཀྱི། །«པེ་»«སྣར་»ཀྱིས། «པེ་བྱས། «@«པེ་»《སྣར་»གྱིས། ®«པེ་»སྣར་»ཅིག། «པེ་»:དཥ་《སྣར་གཞུག«པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»+ས། »། གང་། དང་་་་་་་་ཀམ་«པེ་»«སྣར་»འདི་བསྒོམས། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»དཾ། «པེ་»«སྣར་ཨོ་ཨོ། །«པེ་»«སྣར་བྷུ་བྷིཤྩ། «པེ་»«སྣར་ཏྲ། «པེ་»«སྣར་»ཨཞེས་བྱ་བ། «པེ་«སྣར་»གྱིས། «པེ་»«སྣར་»ཞུགས། འགྱོ«པེ་»«སྣར་»ལྔས། འགྱེ«པེ་»«སྣར་»ཀྱི། འི་ནད་ཀྱི་བཏང་«པེ་»«སྣར་པོའི། «པེ་»«སྣར་»བ། «པེ་»«སྣར་»ཀྱི། ︽«པེ་»«སྣར་རྗེས། ག«པེ་»ཟླ་བ་《སྣར་ཐདྡྷ། «པེ་《སྣར་»ཙ། @«པེ་《སྣར་པའོ། འབྱུང་《བདུད་འདུལ་བ་བྱེད་པར་::ཀྱིབ་དང་། ངབདངབན་བསྡུར་འབྲས་རེའུ་མིག་བདུ། བར་ན་རྒྱུད། （དཀྱིན་དང་། ལ། པུརྒྱུ《°ja::ཀྱི་དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། དང་། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། ཀྱི། །དོམརྒྱུད། །པོད་རྟགས། །ཀུ་བའི་ཤིས་པ། :ཤི། །།།ཀུ། སྐྱེཚན་གྲངས། ལྡེབ། །ན། _བབ་_ན?བ_བ། ནབ་ཕྲེང༌། །་། ་ཚེག་ཁྱིམ། །།།།དང་་་་་་་་་་་་་་་་༌་་། ཁྱོད་༌་།
