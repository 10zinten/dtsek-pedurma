རྡོ་རྗེ་སེམས་དཔའི་སྒྲུབ་ཐབས། བ་དང་། དབང་དུ་བྱའོ། །དེ་ནས་དེ་དག་གི་ཕྱག་རྒྱ་རྡོ་རྗེ་ཁུ་ཚུར་ལས་བྱུང་བ༑མཛུབ་མོ་ལྕགས་ཀྱུ་:བཅིང་བ་ནི། །མཐེའུ་ཆུང་ལས་ཏེ་ལྕགས་ཀྱུ་ཆེ། །དེ་ནི་རྡོ་རྗེ་ལྕགས་ཀྱུའོ། །།།མཁྲིག་མ་མདུད་པ་ཞགས་པའོ། །མཛུབ་གཉིས་་་མདུད་པ་ལྕགས་སྒྲོག་གོ། །རྒྱབ་གཉིས་ཀྱང་ནི་:མཉམ་སྟེ་མནན། །དེ་ནི་རྡོ་རྗེ་འབེབས་པའོ། །དེ་ནས་དམ་ཚིག་གི་ཕྱག་རྒྱ་བཅིངས་ལ་བཛྲ་སཏྭ་དྲི་བྱ་ཞེས་བྱ་བ་བརྗོད་དེ་བལྟས་ལ། ཡེ་ཤེས་སེམས་དཔའ་དེ་མཐོང་ནས། །རང་གི་ལུས་ལ་རྣམ་ཤེས་བསྒོམས། །རྡོ་རྗེ་སེམས་དཔའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་དེ་ཉིད་དང་། །ཛཧཱུྃ་བཾ་ཧོཞེས་བྱ་བ་འདིས་དགུག་ཅིང་གཞུག་ལ་བཅིང་བ་དང་། དབང་དུ་བྱས་ཏེ་བསྒྲུབ་པར་བྱ། དེ་ཉིད་དང་ས་མ་ཡ་ཧཱུཾ་ཞེས་བརྗོད་པས། རྒྱབ་ཏུ་ཟླ་བ་གྲུབ་ནས་ནི། །ས་མ་ཡ་ཧཱུྃ་ཨ་ཧཾ་བརྗོད། །དེར་བདག་སེམས་དཔའི་ལུས་སུ་བསྒོམ། །བཛྲ་སཏཱ་ཨོཾ་བཛྲ་སཏྭ་ཧཱུཾ། །ཞེས་བརྗོད་དོ། །བཞད་པ་དང་་་་བཅས་སྐུ་མདོག་དཀར། །ཟླ་བའི་གདན་དང་ཟླ་བའི་འོད། །སངས་རྒྱས་ལྔ་ཡི་ཅོད་པན་ཅན། །སེམས་དཔའི་སྐྱིལ་མོ་ཀྲུང་གིས་བཞུགས། །རབ་ཏུ་དགྱེས་་་པས་སྤྱན་གདངས་པ། །མགོན་པོ་རྡོ་རྗེ་དྲིལ་བུ་བསྣམས། །ཕྱག་རྒྱ་ཆེན་པོ་རྣམ་པར་བཞུགས། །བཛྲ་དྲིཌི་མ་ཊ་ཞེས་བྱ་བ་མིག་ལ་བཀོད་དེ། ཡང་་་སེམས་མ་རྡོ་རྗེ་སྐྱེལ་སོགས་པས། སྙིང་ག་དང༌། མཛོད་སྤུ་དང་། ལྐོག་མ་་་དང༌། སྤྱི་བོར་བྱིན་གྱིས་བརླབ་པར་བྱ་སྟེ། སཏཱ་བཛྲ་ཨ་དྷི་ཏིཥྛ་ཚྭ་ཞེས་བྱ་བ་དང༌། སེམས་མ་རྡོ་རྗེ་མས་སྙིང་གར། རཏྠ་བཛྲ་ཨ་དྷི་ཏིཥྛ་ཚྭ་ཞེས་པ་དང་། །རིན་ཆེན་རྡོ་རྗེ་མས་དཔྲལ་བར། དྷརྨ་བཛི་ཨ་དྷི་ཏིཥྛ་ཧཱུཾ་ཞེས་བྱ་བ་དང་ཆོས་ཀྱི་རྡོ་རྗེ་མས་ལྐོག་མ་དང་། ཀརྨ་བཛྲ་ཨ་དྷི་ཏིཥྛ་ཚྭ་ཞེས་བྱ་བ་དང༌། ལས་ཀྱི་རྡོ་རྗེ་མས་སྤྱི་བོར་རོ། ་ཁ། སེམས་མ་རྡོ་རྗེ་མ་ནི་རྡོ་རྗེ་བསྡམས་པ་ལས་གུང་མོ་མྱུ་གུ་་་།
