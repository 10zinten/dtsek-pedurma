རྒྱུད། ཀི་ཚུ་ཀ་བློ་ལོ་པདྨོ་ཧཾ། ཁཌ་གོ་ཧཾ། ཙཀྲ་མ་ཧཾ། ཛི་ཧཱ་ཧཾ། ཀརྨ་བཛྲ་ཧཾ། བ་ར་མོ་ཧཾ། དཾཏྠོཧཾ། མུཉྫི་ར་ཧཾ། པ་བཛྲ་དྭ་ཡ་མ་ཧཾ། ཀ་རཏྣ་མཱ་ལ་ཧཾ། ང། བཱི་ན་ཧཾ། །བཛྲ་:ནི་ཏྱ་ཀ་ར་པ་སྨྲ་བོ་ཧཾ། དྷུ་པོ་ཧཾ། པུཊྚ་ཧཾ། དཱི་པོ་ཧཾ། གྷནྡྷོ་ཧཾ། ༀ་ཀུ་ཤོ་ཧཾ། པཱ་ཤོ་ཧཾ། སྤོ་ཊ་ཧཾ། གྷཌོ་ཧཾ། ་་་ཕྱག་རྒྱའི་གཟུངས་དེ་ཡང་རང་རང་གི་ལྷའི་སྐུར་བསྒོམ་པར་བྱ་སྟེ། །ལ་བཛྲ་རཱ་ཛོ་ཧཾ། ཁ་བཛྲ་རཱ་གོ་ཧཾ། བཛྲ་སཱ་དྷུ་ར་ཧཾ། བཛྲ་གནྡྷོ་ཧཾ། །བཛྲ་པྲ་བྷོ་ཧཾ། བཛྲ་ཡཉྫི་ར་ཧཾ། བཛྲ་པྲི་ཏི་ར་ཧཾ། བཛྲ་ཎེ་ཏྲེ་ཧཾ། བཛྲ་བུདྡྷི་ར་ཧཾ། བཛྲ་མཎྜ་ཧཾ། བཛྲ་བཱ་ཙོ་ཧཾ། བཛྲ་བི་ཤྭ་ཧཾ། བཛྲ་བཻ་རོ་ཧཾ། བཛྲ་ཙཎྜ་ཧཾ། བཛྲ་མུཉྫི་ར་ཧཾ། ་བཛྲ་ལཱ་སྱེ་ཧཾ། བཛྲ་མཱ་ལེ་ཧཾ། བཛྲ་གཱི་ཏི་ཧཾ། བཛྲ་ནཱིརྟི་ཧཾ། བཛྲ་དྷུ་པོ་ཧཾ། བཛྲ་པུཊྚ་ཧཾ། བཛྲ་དཱི་པོ་ཧཾ། བཛྲ་གནྡྷོ་ཧཾ། བཛྲ་ཨཾ་ཀུ་ཤོ་ཧཾ། བཛྲ་པཱ་ཤོ་ཧཾ༑་བཛྲ་སྤོ་ཊ་ཧཾ། བཛྲ་གྷཎོ་ཧཾ། དེ་ནས་རྡོ་རྗེ་ལྕགས་ཀྱུ་ལ་སོགས་པའི་་ལས་ཀྱི་ཕྱག་རྒྱ་དང། སྙིང་པོ་འདི་རྣམས་ཀྱིས་འདི་ཉིད་དུ་དགུག་པ་དང་། གཞུག་པ་དང་། བཅིང་བ་དང་དབང་དུ་བྱ་བའོ། །བཛྲ་ཨཾ་ཀུ་ཤ་ཛབཛྲ་པཱ་ཤ་ཧཱུྃ། བཛྲ་སྤོ་ཊ་ཐཾ། བཛྲ་གནྡྷེ་ཨ། དེ་ནས་རང་རང་གི་དམ་ཚིག་གི་ཕྱག་རྒྱ་བཅིངས་ལ། སྙིང་པོ་འདི་རྣམས་ཀྱིས་བལྟ་བར་བྱ་སྟེ། བཛྲ་རཱ་ཛེ་དྲི་བྱ། བཛྲ་རཱ་ག་དྲི་བྱ། བཛྲ་སཏྭ་དྲི་བྱ། བཛྲ་རཏྣ་དྲི་བྱ། བཛྲ་ཏེ་ཛ་དྲི་བྱ། བཛྲ་ཀེ་ཏུ་དྲི་བྱ། བཛྲ་ཧཱ་ས་དྲི་བྱ། བཛྲ་དྷརྨ་དྲི་བྱ། བཛྲ་ཏཱིཀྵ་དྲི་བྱ། ་བཛྲ་ཧེ་ཏུ་དྲི་བྱ། །བཛྲ་བྷཱ་ཥ་དྲི་བྱ། ་བཛྲ་ཀརྨ་དྲི་བྱ། བཛྲ་རཀ་དྲི་བྱ། བཛྲ་་་ཡཀ་དྲི་བྱ། བཛྲ་སནྣི་དྲི་བྱ། །བཛྲ་ལཱ་སེ་དྲི་བྱ། བཛྲ་མཱ་ལེ་དྲི་བྱ། ་བཛྲ་གཱི་ཏི་དྲི་བྱ། བཛྲ་:མྲི་ཏི་དྲི་བྱ། བཛྲ་དྷུ་པེ་དྲི་ཤྩ། བཛྲ་པུཥྛ་དྲི་བྱ། བཛྲ་ཨཱ་ལོ་ཀེ་དྲི་བྱ། བཛྲ་གནྡྷེ་དྲི་བྱ། ་བཛྲ་ཨཾ་ཀུ་ཤ་དྲི་བྱ། བཛྲ་པཱ་ཤ་དྲི་བྱ། པ་—།
