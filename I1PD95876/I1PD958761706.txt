བྱ་བ་བསྡུས་པ་ཞེས་བྱ་བ། ཐུགས་ཏུ་གཞུག་པར་བྱའོ། །ལག་པ་དག་རྡོ་རྗེ་བཅིངས་ནས་རྡོ་རྗེའི་སེ་གོལ་གྱིས་རྣམ་་་པར་གྲོལ་བས་སོ། །དཔལ་རྡོ་རྗེ་སེམས་དཔའི་རྣལ་འབྱོར་གྱི་བདག་ཉིད་སངས་རྒྱས་འདུས་པར་བྱའོ། །གཡོན་པའི་ཐལ་མོའི་སྒྲ་དང་ཐལ་མོ་མཉམ་པས་བསྒྲུབ་པར་བྱ་སྟེ། གཡས་པའི་ཐལ་མོར་གསུངས་པ་ནི་གདབ་པར་བྱ་བ་གཉིས་ཀས་་་་ཀྱང་ངོ་ཞེས་པ་ནི་རྡོ་རྗེའི་སྒྲ་འདུས་པའི་ཕྱག་རྒྱའོ། །དེ་ནས་རྡོ་རྗེ་ལྕགས་ཀྱུ་ལ་་་སོགས་པ་རྣམས་ཀྱིས་དགུག་པ་དང༌། །གཞུག་པ་དང་། བཅིང་བ་དང་། དབང་དུ་བྱ་བ་རྣམས་བྱས་ནས་མཆོད་ཡོན་ལ་སོགས་པ་ཕྱིའི་མཆོད་པ་རྣམས་ཀྱིས་ཡང་དག་པར་མཆོད་ནས་དམ་ཚིག་ཀྱང་བལྟ་བར་བྱའོ། ::། ཨོ་བཛྲ་དྷཱ་ཏུ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་སཏཱ་དྲི་བྱ། ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་རཏྣ་བཛྲ་དྲྀ་ཤྱ་ཛཧཱུྃ་ཁཾ་ཧོས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་དྷརྨ་བཛྲ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ་ཨོ་ཀརྨ་བཛྲ་དྲི་བྱ་ཛཧཱུཾ་བཾ་ཧོས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་སཏཱ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། །ཨོཾ་བཛྲ་རཱ་ཛ་དྲི་བྱ་ཛgཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་རཱ་ག་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་སཱ་དྷུ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་རཏྣ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། །ཨོ་བཛྲ་ཏེ་ཛ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་ཀེ་ཏུ་དྲི་ཤུ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་ཧཱ་ས་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། །ཨོཾ་བཛྲ་དྷརྨ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཨོ་བཛྲ་ཏཱིག་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོ། ས་མ་ཡ་སྟི་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་ཧེ་ཏུ་དྲི་བྱ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་བྷཱ་ཥ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཌཨོཾ་བཛྲ་ཀརྨ་དྲྀ་ཤྱ་ཛཧཱུྃ་བཾ་ཧོས་མ་ཡ་ཙཱ་ས་མ་ཡ་ཧཾ། ཨོཾ་བཛྲ་རཀྵ་།
