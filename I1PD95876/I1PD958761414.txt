ཁྱོ་ཆཇིའི་དབུ་ཁག་ལ་དུ་བ་འདུག་པ། །ལོ་ཏཁྲོ་རྗེ་ཁ་ཊི་ཀ་བ་འདྲེན་པ་ཀུན་ལ་ཁ་སྒྲོལ་ཕྱུག་འཚོ། ཁ་བ་ག་བ་ག་བ་ག་བ་ག་དེ་འབོད་ཕྱོགས་ཀྱི་ཀ་ཀྱངཊམ་འོང་ཀ་ལའོ་ནདམ་ལྟའི་ཆར་བ་ཤོ་པ་བ་ཀློག་དང་ཁ་ལོ་བའི་ཟ་ཁང་དང་། པའི་ཁྲི་ལོ་དང་ཁ་བྲལ་བ་ལ་བྱ་བ་གཙོ་བོ་དང་དུ་བ་དག། དེ་པ་ཡི་ག་་ཁ་ག་ད་བ་མ་འ་ས་ག་ཟླ་བ་ག་བ་ག་བ་ག་བ་ག་བ་ག་བ་ག་བ་ང་ཀའོ་ཀ་བ་༄༅། །སྦྱིན་སྲེག་གི་ཆོ་ག་དང་། །ད་ལྟ་ਵਿੱਚਸਾਲਲੰਘਲਹੌਲਸੌਂਸਲੂਸਨੂੰਤੋਂਕਿཀཥདགག་དང་རིགས་པས་ཁོ་ཁང་བ་འགའ་ཞིག་ཁ་ཏ་:ལོ་ལ་བབ་ཚོ་་་་་ཀྱི་འགལ་བའི་སྐལ་བ་བྱེ་བ་བྱེ་བ་ཀུན་ཞི་བར་བྱ་བ་དང་ལ་སྔོན་པོ། དེ་འདྲ་ཚང་གཙོ་ཀ་བ་ག་བ་ཏོ། །འོ། །ན་དང་བ་ད་༄༅། །རྒྱ་གར་སྐད་དུ། ཧོ་མ་བི་དྷི། བོད་སྐད་དུ། སྦྱིན་་་་སྲེག་གི་ཆོ་ག། ་། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་གང་ཞིག་དྲན་པ་ཡིས། །དངོས་གྲུབ་མི་:འཇིགས་ཐོབ་གྱུར་པ༄༅། །།རྡོ་རྗེ་ཅན་ལ་གུས་བཏུད་ནས། །སྦྱིན་སྲེག་ཆོ་ག་གསལ་བར་བཤད། །གང་དག་ཏུ་་ནི་རང་ཉིད་ཀྱིས། །ལས་ཀྱི་ཆོ་ག་རྣམས་མེད་པ། །དེར་ནི་སྤྱི་རྒྱུད་ལས་་་་གསུངས་པའི། །ཆོ་ག་མཁས་པས་བསྟན་པར་བྱ། །དེ་ལ་ཞི་བ་ལ་ནི་ས་གཞི་དཀར་པོ་ལ་རྒྱར་ཁྲུ་གང་ཙམ་ལ། ཟབས་སུ་སོར་བཅུ་གཉིས་པའི་ཐབས་ཁུང་ཟླུམ་པོ་བརྐོས་ལ། དེའི་དབུས་སུ་པདྨའི་སྟེང་དུ་རྡོ་རྗེ་རྩེ་ལྔ་པ་སོར་བརྒྱད་པར་བྱ། ཐབ་ཁུང་གི་སྟེང་དུ་སོར་བཞི་བར་དུ་དོར་ལ། །དཔངས་དང་རྒྱུར་་་་སོར་བཞིའི་ཁ་ཁྱེར་ཀུན་ནས་བྱས་ལ། མཚམས་རྣམས་སུ་རྡོ་རྗེ་རྩེ་གསུམ་པས་མཚན་པ་དང་། ཐབ་ཁུང་གི་ས་གཞི་དེ་བ་ལང་གི་ལྕི་བ་དང༌། ་ས་དཀར་པོས་།
