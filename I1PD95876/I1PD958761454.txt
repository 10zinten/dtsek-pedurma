སྒོ་བཞི་པ། །རྟ་བབས་བཞིས་ནི་མཛེས་བྱས་ལ། ང་། ཐིག་བཞི་དང་ནི་ཡང་དག་ལྡན༑། དར་དང་མེ་ཏོག་ཕྲེང་བས་བརྒྱན། །གྲྭ་ཡི་ཕྱོགས་ནི་ཐམས་ཅད་དང༌། །སྒོ་ཁྱུད་ཀྱི་ནི་མཚམས་རྣམས་སུ། །རྡོ་རྗེ་རིན་ཆེན་སྤྲས་པར་ནི། །དཀྱིལ་་་འཁོར་ཕྱི་རིམ་ཐིག་གདབ་བོ། ་ཡ། །དེ་ཡི་རིམ་པའི་ནང་ཞུགས་ལ། །སྦྱིན་སྲེག་ཐབ་ནི་བྲི་བར་བྱ། །ཀུན་ནས་ཟུར་ནི་བཞི་པ་ལ། །ཁྲུ་གཉིས་པར་ནི་ཡོངས་སུ་བསྒྲགས། ཚི། ཟབས་སུ་ཁྲུ་གང་ཙམ་ལ་ནི། །།ཁ་ཁྱེར་བཞི་ནི་བསྐོར་བའོ། །དབུས་སུ་གཙུག་ཏོར་དགོད་བྱ་ཞིང༌། །ཚེ། །དེ་ཡི་ཤར་དུ་ལྷུང་བཟེད་དོ། །ལྷོ་་་ཕྱོགས་ན་ནི་རིན་ཆེན་ཏེ། །ནུབ་ཀྱི་ཕྱོགས་སུ་པདྨ་བྲི། །བྱང་དུ་རྡོ་རྗེ་ཧཱུྃ་མཛད་ནི། །གདུག་པ་ཐམས་ཅད་ཟློག་པའོ། །མཚམས་ཀྱི་ཆ་ནི་ཐམས་ཅད་དུ། །གསང་བའི་མཆོད་པ་བསྟན་པ་དགོད། །དེ་ལྟར་ཐབ་ཁུང་ངེས་བྱས་ནས། །ཁ་ཁྱེར་ལ་ནི་འཇུག་པར་བྱ། །དང་པོ་ལ་ནི་སེམས་དཔའ་ཆེ། །བཅུ་དྲུག་ཉིད་དུ་བསྒྲགས་པ་ཡིན། །དེ་ཡི་རྒྱར་ནི་སོར་བརྒྱད་ལ། །དཔངས་སུ་སོར་ནི་བཞི་པའོ། །ཁ་ཁྱེར་གཉིས་པར་གྱུར་པ་དག། དེ་ཡང་རྒྱར་ནི་སོར་བརྒྱད་དེ། །དཔངས་སུ་སོར་ནི་བརྒྱད་པར་བཤད། །དེ་དགོངས་བདག་གིས་བཤད་པར་བྱ། །ཤར་་ཕྱོགས་ནས་བརྩམས་རིམ་བཞིན་དུ། །པདྨ་ནགས་ཀྱི་མཇུག་མ་དང༌། །བགྲང་ཕྲེང་རིལ་བ་སྤྱི་བླུགས་དང་། ལ། ཨ་ཛེ་ན་ཡི་ཕྱག་རྒྱ་ནི། །ཏྲི་ཤུ་ལ་དང་ཟླ་བ་ཕྱེད། །རལ་པ་མིག་དང་རྒྱལ་མཚན་ཆེ། །མཛེས་ཤིང་བདེ་བའི་ཕྱག་རྒྱ་ཆེ། །ལྷོ་ཡི་ཕྱོགས་སུ་གཞག་པར་བྱ། །དུང་དང་འཁོར་ལོ་བེ་ཅོན་དང་། །རལ་གྲི་་་དང་ནི་དེ་བཞིན་ཕུབ། །ཕག་དང་སེང་གེ་གཤིབས་པ་ནི། །།ནུབ་ཀྱི་ཕྱོགས་སུ་བྲི་བར་བྱ། །མདུང་དང་དྲིལ་བུ་བ་དན་དང་། །རྨ་བྱ་དེ་བཞིན་ར་སྐྱེས་དང་། །ཁྱིམ་བྱ་ལང་ཚོས་མྱོས་པ་ནི། །།བྱང་གི་ཕྱོགས་སུ་བྲི་བར་བྱ། །།ཁ་ཁྱེར་གསུམ་།
