བར་བྱའོ། །ཅུང་ཟད་དགྱེས་པ་དང་མངར་བ་དང་། ལན་ཚྭ་བ་དང་་་ལྡན་པར་བྱ་བ་ནི་དགེ་བར་བྱེད་པའི་ཞལ་ལོ། །ཚ་བ་དང༌། །ཆུ་ཁ་བ་དང་། བསྐ་བ་དང་། །ཚེ་སྐྱུར་བ་དང་། ཁྲོ་བའི་ཞལ་དང་ལྡན་པ་ནི་མི་དགེ་བར་་་་བྱེད་པའོ། །བསྟན་བཅོས་ཀྱི་མཚན་ཉིད་ལས་གཞན་པའི་ཞལ་གྱི་བྱེད་པས་་་སྦྱིན་བདག་མི་འཚོ་བར་བྱེད་པ་དང་བསྟན་བཅོས་ཀྱི་མཚན་ཉིད་ཀྱི་ཞལ་གྱི་བྱེད་་་པས་སྐྱེ་བོ་དང་ནོར་དང་རྣམ་པར་རྒྱལ་བ་དང་ཚེ་དང་ཕུན་སུམ་ཚོགས་པ་རྣམ་་་་པར་འཕེལ་ལོ། །།།།བ་ལང་ལ་དཀར་བ་ལ་སྐྱོད་ལ་ཚོད་ཀྱིས་ཀྱང་ཞི་བའི་བྱད་ནི་ལུས་ཀྱི་མཚན་ཉིད་བརྗོད་པར་བྱ་སྟེ། །དེ་ལ་རྐེད་པའི་འོག་ནས་ལྟེ་བའི་མཐར་ཐུག་པའི་བར་ལ་མཐོ་དོའོ། །ལྟེ་བའི་འོག་ནས་གསང་བའི་རྩ་བའི་་་བར་ལ་མཐོ་གང་། ནམ་:ཚོང་སྟེང་མོ་ནས་འཕྲེད་དུ་ལོགས་དག་ཏུ་ནུ་མའི་བར་དག་དང་མགྲིན་པའི་ལོགས་དག་ནས་དཔུང་མགོའི་རྩེ་མོ་དག་དང་། །།དེ་རྣམས་ཀྱི་ཚད་དུ་མཐོ་རེ་རེའོ། །ནུ་མ་དང་ལྟེ་བའི་བར་དུ་ཆ་བཅུ་བཞིའོ། །ཐལ་གོང་དག་དང་ནམ་ཚོང་སྟོད་མོ་དང་ལག་མ་རྣམས་ཐོག་མཉམ་པའོ། །དཔུང་པ་དག་གི་རྒྱུར་བར་དུ་གདོང་གསུམ་མོ། །མཆན་ཁུང་རྩིབ་ལོགས་དང་ནུ་མ་དག་གི་བར་ལ་ཆ་དྲུག་གོ། །ནུ་མའི་དཀྱིལ་ལ་ཆ་གཉིས་སོ། །ནུ་མའི་ཟླུམ་ལ་ནས་་་་གཉིས་སོ། །ལྟེ་བའི་དཀྱིལ་འཁོར་ལ་ནས་གསུམ་མོ། །གཡས་སུ་འཁྱིལ་ཞིང་ལྟེ་བ་དམའ་བའོ། །གསང་བའི་གནས་ལ་ཆ་གསུམ་མོ། །ལོགས་དག་ཆ་གཉིས་སོ། །འབྲས་བུ་ལ་ཆ་བཞི་སྟེ། །མཛེས་པ་རྒྱས་པ་ཟླུམ་པ་དག་གོ། །ཕི་ཙ་དག་གི་་་་སྟེང་དུ་ཆ་བརྒྱད་དོ། །ལྟེ་བ་དང་གསང་བ་དག་གི་དབུས་སུ་བརླའི་རྩ་བའོ། །།བརླ་དང་བྱིན་པ་འདི་གཉིས་ལ་ཚད་མཐོ་དོའོ། །པུས་མོའི་ལྷ་ང་ལ་བཞིའོ། །ལོང་བུ་དག་ལ་ཆ་གཉིས་སོ། །རྟིང་པ་དག་ལ་ཆ་བཞིའོ། །།རྐང་པ་དག་གི་་་།
