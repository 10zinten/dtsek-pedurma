རབསྟན་འགྱུར། ་རྒྱུད། [བཀུ་ལྷག་མ་སྔོན་བཞིན་ནོ། །དེ་བཞིན་དུ་དེའི་གཡོན་ལོགས་ན་གནས་པ་ལ། །ཨོ་མ་ཎི་ཀ་བ་ཙ་བནྡྷ་ཡ་ཧཱུྃ། འདིས་ལན་ཉེར་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་བཞིན་ནོ༑། །དེ་བཞིན་དུ་དབང་ལྡན་གྱི་མཚམས་ཀྱི་ལྷོར་གདོང་མངོན་པར་ཕྱོགས་པའི་ཁང་མིག་གི་སྟེང་གི་དབུས་སུ་གནས་པའི་རྒྱལ་མཚན་ལ།  ོ་ཨོ་ཛྷཱ་ན་བི་ཤྭ་ཤྭ་ར་ཧཱུྃ། འདིས་ལན་ཉེར་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་བཞིན་ནོ། །ཎ་། །དེ་བཞིན་དུ་དེའི་་་གཡས་ལོགས་སུ་གནས་པ་ལ། །ཀ། ཨོ་མ་ནི་ཀརྨོཏྟ་མ་བཛྲ་རསྨ་ཡ་མ་ནུ་སྨ་ར་སརྦ་་་བུདྡྷ་ནཱ་ཀཆུ་ཡ་པྲ་བེ་ཤ་ཡ་གྲ་ཧ་ཡ་སརྦ་ཀརྨ་ཎི་ཀུ་རུ་སྭཱ་ཧཱ། འདིས་ལན་ཉེར་་་་་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་བཞིན་ནོ། །དེ་བཞིན་དུ་དེའི་གཡོན་ལོགས་སུ་གནས་པ་ལ། །ཨོཾ་ན་མསརྦ་ཏ་ཐཱ་ག་ཏོརྞཱི་ཥ་ཨ་བ་ལོ་ཀི་ཏ་མུཌྷ་་ནི་ཧཱུཾ། ཛྭ་ལ་ཛ་ལ། ་དྷ་ཀ་དྷ་ཀ། བི་དྷ་ར་བི་དྷ་ར། ་ཙྪིནྡ་ཙྪིནྡ། བྷིནྡྷ་བྷིནྡྷ། ཧཱུྃ་ཧཱུྃ་་་ཕཊ༑། འདིས་ལན་ཉི་ཤུ་རྩ་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་མ་བཞིན་ནོ། །དེ་བཞིན་དུ་དེའི་ནུབ་ཏུ་གདོང་མངོན་པར་ཕྱོགས་པའི་ཁང་མིག་གི་སྟེང་གི་དབུས་སུ་གནས་པའི་རྒྱལ་མཚན་ལ། །ཨོ་བཛོཎྞི་ཥ་ཧཱུཾ་ཕཊ། ཟའདིས་ལན་ཉི་ཤུ་རྩ་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་མ་བཞིན་ནོ། །དེ་བཞིན་དུ་དེའི་གཡས་ལོགས་སུ་གནས་པ་ལ། །ཨོཾ་མ་:ཎི་ཙིནྟཱ་ཀཱ་ཤ་གབྷ་:སིདྡྷཱ་སིདྡྷ་བྷ་ག་བཱན་ཧཱུྃ། འདིས་ལན་ཉི་ཤུ་རྩ་གཅིག་བཟླས་ནས་ལྷག་མ་སྔོན་མ་བཞིན་ནོ། །དེ་བཞིན་དུ་དེའི་གཡོན་ལོགས་སུ་གནས་པ་ལ། །ཨོཾ་རཏྣ་ཀུ་རཱ་ཀཆུ་ཡ་སརྦ་ན། ཨཱ་ན་ཡ་ཨཱ་ན་ཡ་ཤྲཱི་ཤྲཱི་ཏ་ཐཱ་ག་ཏ་ས་ཏྱ་མ་ནུསྨ་ར་ཧཱུྃ། འདིས་ལན་ཉི་ཤུ་རྩ་གཅིག་མངོན་པར་ཡོངས་སུ་་་་བཟླས་ན་ལྷག་མ་རྣམས་སྔོན་མ་ཇི་ལྟ་བ་བཞིན་དུ་བྱའོ། །།དྲི་གཙང་ཁང་དང་་་་ཁང་མིག་དང་འདུ་ཁང་རྣམས་ཀྱི་ཤར་དུ་རྣམ་པར་སྣང་མཛད་ཀྱི་གཟུགས་ཀྱི་་་་སྔགས་ལན་ཉི་ཤུ་རྩ་གཅིག་བཟླས་སོ། །ལྷོ་རུ་རྒྱལ་མཚན་རྩེ་མོའི་དཔུང་རྒྱན་གྱི་་།
