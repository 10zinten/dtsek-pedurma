དེ་བཞིན་སེམས་ཅན་དམན་པ་ཡང༌། །ཟབ་ཅིང་བླ་མེད་རྒྱ་ཆེ་བ། ཌ། ཡོན་ཏན་བགྲང་འདས་འབྱུང་བའི་གནས། །སངས་རྒྱས་ཉིད་ལ་རེ་བའང་མེད། །གཞལ་ཡས་རྨད་བྱུང་བསམ་མི་ཁྱབ། །སྣང་བ་ཐམས་ཅད་ཟིལ་གནོན་པ། །རྫོགས་སངས་ཡོན་ཏན་རིན་ཆེན་ལ། ཡོ། སུ་ཞིག་ཡིད་ཀྱིས་སེམས་མི་བསྐྱེད། ་ཚོ། གཞན་དོན་བྱེད་པ་དཔའ་བ་དང་། །སྐྱེས་བུ་ཆེན་པོའི་ཚུལ་ལྡན་པ། །སུ་ཞིག་འཁོར་མཐའ་མེད་པར་ཡང་། །ཉིན་གཅིག་ལྟ་བུར་གནས་མི་བྱེད།  ོ་། མཁས་པ་གང་ཞིག་འདས་གྱུར་ཀྱང་། ཉི་དང་། གང་གིས་གཏམ་གཞི་འཕེལ་འགྱུར་བ། །འཕགས་པ་གཞན་དོན་མཛད་རྣམས་ཀྱི། །ལམ་གྱི་རྗེས་སུ་འཇུག་མི་བྱེད། ་་། སུ་ཞིག་ཆུ་ཤིང་དབུ་བ་ལྟར། །སྙིང་པོ་མེད་པའི་ལུས་འདི་དག། གཞན་ཕན་རྐྱེན་དུ་སྒྲུབ་བྱེད་པས། །ལྷུན་པོ་སྙིང་པོར་སྒྲུབ་མི་བྱེད། །ཡ་རབས་སྙིང་རྗེ་ལྡན་པས་ནི། །སྐད་ཅིག་རེ་རེ་རྒ་ཤི་དང༌། །ནད་ཀྱི་གཞིར་གྱུར་ལུས་དག་ནི། །གཞན་ལ་བདེ་སྐྱིད་ྋགཞིར་བྱེད་དོ། །དམ་ཆོས་སྒྲོན་མ་དང་བཅས་ཡོལ། །མི་ཁོམ་བརྒྱད་སྤངས་ཁོམ་པ་འདི། །སྐྱེས་བུ་ཆེན་པོའི་སྤྱོད་པ་ཡིས། །འབྲས་་བུ་ཡོད་པར་བྱ་བའི་རིགས། །ང་། དབུ་མའི་སྙིང་པོ་ལས་བྱང་ཆུབ་ཀྱི་ཨསེམས་མི་་་གཏོང་བའི་ལེའུ་སྟེ་དང་པོའོ། །ཀ་། ར་ལོན་ཆེད་དུ་ན་བ་དང་གས་པའི་བྱ་བ་ཁོ་ཌ། ཁྲ། ཏི་རུ་སྤྱི་ལོཚེ་ཎ་ཚོ། ང་ཚོའི་ས་གདན་ལ་ཚོང་ཁ་པོ་བྱས་འ#དེ་ལྟར་འགྲོ་ལ་ཕན་པའི་ཕྱིར། །བྱང་ཆུབ་ཆེན་པོར་བརྟན་བླངས་པ། །ཚུལ་དང་མཐུན་པའི་ལམ་ཉིད་དུ། །རབ་ཏུ་ཞུགས་པ་རྣམ་པར་མཛེས། །དེ་ནི་བྱམས་པ་གནོད་མེད་དང༌། །སྙིང་རྗེ་ཕན་པ་བསྒྲུབ་པ་ཡིས། །།།དམ་ཆོས་་་རྟོགས་པ་ངོམས་མེད་དང༌། །ཆོས་སྦྱིན་པ་ལ་སེར་སྣ་མེད། །དྲང་དང་་་་་་རྒྱལ་བཅོམ་པ་དང༌། །ཡིད་ཀྱིས་དེ་ཉིད་བལྟ་བ་དང་། །གཞན་སྐྱོན་རྟོག་ོལ་།
