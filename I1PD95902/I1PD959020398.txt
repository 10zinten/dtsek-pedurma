རྟེན་མ་ལུས་པ། །ལྷ་དང་མི་དང་ལྷ་མིན་ཐམས་ཅད་དགེ་བར་ཤོག། དངོས་་འཛིན་གདོན་གྱིས་བརླམས་པ་ཡི། །རིམས་ནད་རྟག་ཏུ་ཟློག་བྱེད་ཅིང་། །།འཇིག་རྟེན་འདི་ཡི་མུན་སེལ་བ། །དྲི་མེད་མི་གཡོ་རིགས་འདི་ཡིན། །།།རྒྱ་གར་གྱི་མཁན་པོ་ཛྙཱ་ན་གབྷ་དང༌། ལོ་ཙཱ་བ་དཔལ་བརྩེགས་རཀྵི་ཏས་བསྒྱུར་ཏེ་གཏན་ལ་ཕབ་པ། །།།་་་་་་་་་་བསྡུར་མཆན། ད་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་རྩོད་པ་བཟློག་པའི་འགྲེལ་པ་སློབ་དཔོན་ནཱ་གཱ་རྫུ་ནས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། ཉི«པེ་»«སྣར་»ནི། «པེ་»«སྣར་ལ། ®«པེ་»«སྣར་+དེ། །«པེ་»«སྣར་—མ། «པེ་»«སྣར་»—རྩེ། «པེ་»«སྣར་ལ། «པེ་»«སྣར་བཟློག«པེ་སྣར་མི་ནུས་ལ། «པེ་《སྣར་བཅད། ༄༅«པེ་»«སྣར་བཟློགཉེ«པེ་»«སྣར་»ཀྱི། །O«སྣར་+ཞེ་ན། Q«པེ་»«སྣར་»+ཀྱི། ཞེ«པེ་《སྣར་ཚིགས། «པེ་»«སྣར་ལ། སྟོང་པའི་ཕྱིར་དེས། «པེ་»«སྣར་»+མ། @«པེ་»«སྣར་+ནོ། @«པེ་»«སྣར་—ནི། «པེ་»«སྣར་ན། གང་གི། «པེ་»«སྣར་»«ཅོ་བསྡུས། ་«སྣར་»ཁ་ཅིག་ནི་སྟོང་། «པེ་《སྣར་བསྟན། «པེ་《སྣར་པ་ན་སྒྲ་དེས་སྒྲ་དེ་བཟློག«པེ་»«སྣར་བཟློག། «པེ་《སྣར་»—སྙམ་ན། «པེ་»«སྣར་»དེ། ཚེ«པེ་»«སྣར་»—ནི། ཉི«པེ་»«སྣར་—སྒྲ། «པེ་»«སྣར་ན་ཡང་། །«པེ་»«སྣར་འདི། 《«པེ་»«སྣར་+དེ་ལ། Q«པེ་»«སྣར་བཟློག«པེ་«སྣར་»གྱིས། །«པེ་《སྣར་བཟློག@«པེ་»«སྣར་ན། ཚེ«པེ་»«སྣར་—ནོ། «པེ་»«སྣར་—མངོན་སུམ་གྱི། «པེ་»«སྣར་»སེམས། «པེ་»«སྣར་བཟློག(སྣར་»སྙོམ། ་ཉེ«པེ་»«སྣར་»གི་སྒྲུབ། «པེ་《སྣར་སྒྲུབ། «པེ་《སྣར་—མ། («པེ་《སྣར་དང་། @«པེ་»«སྣར་»དགུ། «པེ་《སྣར་དཔའ། «པེ་»འབྲེལ། «སྣར་»འབྲལ། @«པེ་»«སྣར་»སྤྲོས་པ། «སྣར་»གནོས། «པེ་《སྣར་པ་་་མེད་པ་དང༌། མི་རྩོལ་བ། །
