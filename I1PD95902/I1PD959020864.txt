གསུམ་ཡན་ལག་ལྔ། །ཉི་མ་ཇི་བཞིན་འགྲོ་བཞིན་ཏེ། །ཤིང་རྟའི་འཕང་ལོ་བཞིན་དུ་འཁོར། །ཉེས་པ་བཅུ་གཉིས་སྲིད་རྩ་རྣམས། །དང་པོ་བཀག་པས་ཀུན་བཀག་ཅིང་། །རྒྱུ་འབྲས་ཡང་དག་བདེན་ཤེས་བྱ། །ཡང་དག་གྱུར་པ་བཅུ་གཉིས་རྙེད། །ཐར་པའི་ལམ་ལ་དེ་གནས་ཕྱིར། །ཡོན་ཏན་དཔག་མེད་ཐོབ་པ་ཡིན། །འདི་ལ་དགེ་བ་གང་ཡོད་དེས། །ལམ་ལྔར་འཁོར་བའི་སེམས་་ཅན་རྣམས། །ཆོས་ཉིད་ཤེས་ནས་ཉེས་སྤངས་ཏེ། །བདེ་བའི་ལམ་ལྔར་བགྲོད་པར་ཤོག། འཁྲུལ་པ་བཟློག་པའི་རིགས་པ་གཏན་ཚིགས་གྲུབ་པ་ཞེས་བྱ་བ། སློབ་དཔོན་འཕགས་པ་ལྷས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་སརྦ་ཛྙཱ་དེ་བ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་དཔལ་་བརྩེགས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།གིགག་ག་ਲਈਉਮਰਕੇਸਰੀਟਾਸਕབསྡུར་མཆན། ས་ཀྱི་ས་བབ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»«སྣར་ལ་འཁྲུལ་པ་་་བཟློག་པའི་རིགས་པ་གཏན་ཚིགས་གྲུབ་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་《སྣར་མ་དྷ། «ཅོ་»+སུ། ༄༅«པེ་»«སྣར་ལས། «པེ་»«སྣར་བསྒྲུབ། །«སྣར་《ཅོ་»ལྟ། («པེ་»«སྣར་རིག@«པེ་»«སྣར་གཟངས། «པེ་《སྣར་ཡང་། «པེ་»«སྣར་རྩོས། «པེ་»«སྣར་མཐོར། ན«པེ་»«སྣར་གནས། &«པེ་»«སྣར་ནི། །བློ«པེ་»«སྣར་—དུ། «པེ་»«སྣར་འདྲ་བ་བདག་«པེ་དེས་གནས། «པེ་»«སྣར་གྲུབ། «པེ་»«སྣར་»ཅི། «པེ་»«སྣར་ལས། «པེ་»《སྣར་»དབྱུ། 《སྣར་དབྱུ། ཚེ«པེ་པའང་སྤྱིའུ་གཙུག་པུད། «སྣར་པའམ་སྤྱིའུ་གཙུག་ཕུད། Q«པེ་»«སྣར་གྲངས། ན«པེ་»།
