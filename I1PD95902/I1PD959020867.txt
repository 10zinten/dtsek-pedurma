ཆགས་ལ་སོགས། །བདེ་བར་སྦྲུལ་གྱིས་སྐྲག་བཞིན་སྤོང་། །འཇིག་རྟེན་པ་ཡི་དོན་རྟོགས་པས། ལ་། འཇིག་རྟེན་བཞིན་དུ་བསྒྲུབ་པར་བྱ། །ཀུན་ནས་ཉོན་མོངས་སྤོང་འདོད་པས། །དམ་པའི་དོན་གྱིས་བཙལ་བར་བྱ། །རབ་ཏུ་བྱེད་པ་ལག་པའི་ཚད་ཀྱི་ཚིག་ལེའུར་བྱས་པ། སློབ་དཔོན་འཕགས་པ་ལྷས་མཛད་པ་རྫོགས་སོ། །།།།བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་རབ་ཏུ་་བྱེད་པ་ལག་པའི་ཚད་ཀྱི་ཚིག་ལེའུར་བྱས་པ་འགྲེལ་པ་དང་བཅས་པ་བཞུགས་སོ། །ཞེས་འཁོད། «སྣར་»པ། «པེ་»«སྣར་བཏགས། བསྡུར་འབྲས་རེའུ་མིགདབུ་མ། (པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། སྡེ་དགེ། དབུ་མ། །།།དབུ་མ། མདོ། །དབུ་མ། དཔེ་གཞི། །སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང༌། །ཚིག་རྐང༌། །ཚེག་ཁྱིམ། །བ?བན་—བ། བབ་། བན་འོང་འབབ་།
