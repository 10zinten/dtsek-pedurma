པ:：དྭ་ཛ་བྲྀཀྵ་དང་། དུ་ར་དུ་ཧ་དང་། དིརྒ་དནྟ་དང༌། མ་:ཧཱ་ཕ་ཅེལ་དང༌། དེ་དག་ནི་ཤིང་ཏ་ལིའི་མིང་:ཡིན་ནོ། །ཀྲྀཥྛ་ཨ་ཀ་རུ་དང༌། :ཛོར་གཱ་ཀཱ་དང་། :：ཤིར་ཤ་ཀ་དང་། ན་ནི་ཏྲ་ཧྲ་དང་། ལོ་ཧ་དང་། ལ་:གྲི་མི་ཛག་དྷ་དང༌། བ་ར་དྷ་མ་དང་། དེ་དག་ནི་ཨ་ག་རུ་ནག་པོའི་མིང་ངོ༌། །པ་ལ་ཤ་དང་། ཀི་ཤུ་ཀ་དང་། པལླི་དང་། ཡ་ན་ཡ་དང་། རཀྟ་:པུ་བྱ་དང་། །ཀླུ་ར་ཐེཏྠ་དང་། ་་:བ་ཏ་བྷུ་ཙིཐ་དང་། །བྲཧྨ་བྲྀཀྵ་དང་། ས་་་སྐྱིད་དྷ་ར་དང་། དེ་དག་ནི་རྒྱ་སྐྱེགས་ཤིང་གི་མིང་ཡིན་ནོ། །ས་ལ་དང་། སོརྫ་ས་ས་དང་། །ཀཎྜ་དང་། གསྨྱུ་ནི་དང་། ཤི་བ་ར་དང་། :དིན་བྲི་ཐུ་དང་། ཀནྡི་བྲྀཀྵ་དང༌། ར་ས་ནིརྱ་ས་དང་། དེ་དག་ནི་ས་ལའི་ཤིང་གི་མིང་ཡིན་ནོ། །ཀ་བུ་ཀ་དང༌། པུ་ཀ་དང་། ཀ་ཤ་ཡ་ཕ་ལ་དང་། དག་ཤ་ཡ་:ཕུ་བྱ་དང༌། ཀྲི་མི་ཀ་དང༌། ཨ་ཏ་བུ་ཤ་དང་། ཀ་ཤ་ཡ་དུ་ར་དང་། དེ་དག་ནི་ཀྲ་མུ་ཀའི་མིང་ཡིན་ནོ། །ཨ་ཛ་ཀཎ་དང་། པས་ཀཎ་་་དང་། ཀ་ཤང་ཡ་དང་། རཉྫ་ན་དུ་ཁྱིམ་དང་། ཀུ་ཤ་དི་ར་དང་། སཱ་་ད་ཧ་ར་དང་། ལ་ཏ་བྲྀཀྵ་དང༌། སརྫ་ཀ་དང་། དེ་དག་ནི་ཨརྫུ་ཀཎའི་མིང་ཡིན་ནོ། །:ཀར་ཤ་དང༌། གོ་ཤི་དང་། :དྷི་ར་ག་པ་ཏྲ་བྷ་བ་དང༌། ཨ་རན་མྱ་དང་། :·དིརྒ་བྲྀཀྵ་དང་། ས་སམྦྷ་ར་ན་དང་། དེ་དག་ནི་ཨ་ཤྭ་ཀཎའི་མིང་ཡིན་ནོ། །བ་ར་ན་དང་། ་་:ཤྭ་ཏ་བུ་ཤ་ཅི་དང༌། ཏིག་ཏ་དང་། ཤ་ཁ་དུ་ཙམ་དང་། །རོ་ད་བྲྀཀྵ་དང༌། ཏ་མ་ལ་དང་། :མར་ཏ་ལིབ་ཧ་དང་། དེ་དག་ནི་བ་རིའི་མིང་དུ་བཤད་པ་ཡིན་ནོ། །:ཀིར་ན་པུ་ཤ་དང་། པི་ལ་པ་ཏྲ་དང༌། མ་:དུ་སྲ་མཱུ་ལ་དང་། :：དིརྒ་མཱུ་ཙེལ་དང༌། པ་རུ་ཤུ་དང་། ཤ་ཀྱི་ར་མོ་ད་དང་། དེ་དག་ནི་རོ་མ་ཧའི་མིང་ངོ༌། །ཨ་ཛ་ཤིང་དང༌། །
