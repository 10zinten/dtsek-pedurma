བར་དོན་ལེགས་པར་རྟོགས་པ་དགེ་སློང་ཨ་མོ་གྷའི་གསུང་གིས་བསྐུལ་ནས། མང་དུ་་་ཐོས་པའི་ལོཙྪ་བ་ཤཱཀྱའི་དགེ་སློང་དཔལ་ཉི་མ་རྒྱལ་མཚན་བཟང་པོས་འགྲེལ་པ་་་་གསུམ་དང་བསྟུན་ཏེ་བསྒྱུར་ཞིང་ལེགས་པར་གཏན་ལ་ཕབ་:པ། །།གསོ་དཔྱད་གཞུང་ལུགས་བརྒྱ་ཕྲག་དུ་མའི་གནས། །སྣ་ཚོགས་མངོན་་་་བརྗོད་ཐ་དད་སྡེབ་སྦྱོར་ཅན། །ཡ་མཚན་བརྒྱན་ལ་ལེགས་སྦྱར་ཚིག་གིས་སྤེལ། །གང་འདི་བསྒྱུར་དགེས་ནད་རྣམས་ཞི་བར་ཤོག། ::ཀྱི་:ཀྱི་བ་ས་་་་དེ་དག་གབ་པ་དང་བསྡུར་མཆན། མཚན་བྱང་འདི་«སྡེ་»«ཅོ་»ལྟར་བཀོད། «པེ་»«སྣར་»སྤྱད། «པེ་»«སྣར་»བཅིངས་པར། «པེ་»«སྣར་»ས། ཚུ«པེ་»«སྣར་»ཏིཀྟ་པརྦ་ཊ། བཱ་ས། @«པེ་»«སྣར་»པསྐོར། @«པེ་»«སྣར་»«ཅོ་»ཀ::(ཆོས་ཚན་འདིའི་ནང་དུ་དང་ག་དང་དོང་ག་ཞེས་པ་རྣམས་«པེ་»«སྣར་»«ཅོ་»ལ་དང་ཀ་དང་དོང་ཀ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་»རིམ། ་་་་་་་«པེ་»«སྣར་བསླབ་བཙོལ། @«པེ་»《སྣར་»ལ། «པེ་»«སྣར་»ཀཱ། «པེ་»«སྣར་»དྲོ། ::ཚེ«པེ་»«སྣར་»ཆབ། «པེ་»«སྣར་»ཊ། «པེ་»«སྣར་»ཏིཏྠ། «པེ་»རིམ། «པེ་»སྤྱད་དོ། །《སྣར་»སྤྱད་དོ། །།།ཆེ«པེ་»བི་ཤ་ལ། «སྣར་»བི་ཤཱ་ལཱ། འགུ«པེ་《སྣར་»དཱི། «པེ་»«སྣར་བདེ། །འཉི«སྣར་ཟེར། ་ཅི«པེ་»«སྣར་»ལ། པེ«པེ་»«སྣར་»རིང་། འ«པེ་»«སྣར་»མཉམ། །(«པེ་»«སྣར་རྒུན། ཉེ«པེ་»«སྣར་པགས། ཚེ«པེ་»«སྣར་»ན། དགྱེ«ཅོ་»འགུལ། ་ཚེ«པེ་»«སྣར་»པུནན་བཱ། ནགེ«པེ་»«སྣར་»བླ། ཉེ«པེ་»«སྣར་»གྲུ། དགྱེ«པེ་»«སྣར་»དནྟི། གྱི«པེ་»«སྣར་»བི་ཏ་གྲོ། ཀྱི«པེ་»པུ། «སྣར་»ཕུ། ཚེ«པེ་»«སྣར་»དྲོ། ་.«པེ་»«སྣར་»ཨ་རཾ་བཱ། ཆེ«པེ་»«སྣར་དྲོ། ::ཀྱི«པེ་»«སྣར་»ཙམ། །:«པེ་»«སྣར་»བཱ་ཏ། ༄༅«པེ་»«སྣར་»བཱ། «པེ་《སྣར་»ལ། «པེ་»:«སྣར་»ཆུན། ཁྱེ«པེ་»བྷགི། «ཅོ་»བྷཱཧཱ། ཀྱེ«པེ་»«སྣར་»བི། ཉེ«པེ་»ན། སྟེ«པེ་»«སྣར་»ཀ། «པེ་»«སྣར་»སིནྟ། ཨེ«པེ་»«སྣར་»བཏུང་། ཚེ«པེ་»«སྣར་»ཙ་ཆ་སྙམ་བྱེད། «པེ་»«སྣར་»ཁྲུས། །ོ«པེ་»པ《སྣར་»བཙེགས། དདབ:::ངེད་«པེ་»སི། །《སྣར་»སཱརྩི། «པེ་»«སྣར་ཤིང་ནི། «པེ་»«སྣར་»ཊ་ཚྭ། «པེ་»«སྣར་»ཀ་་་།
