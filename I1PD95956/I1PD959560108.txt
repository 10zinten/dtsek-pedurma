་དང༌། ཨ་གྲ་གནྡྷ་དང་། ཇོ་:བྷུབ་པ་དང་། བྷུ་ཏག་དང༌། ཚི་ཚ་ཀཎྜདང་། ར་ྋས་ཡ་ན་བ་ར་དང༌། དེ་དག་ནི་སྒོག་སྐྱའི་མིང་ངོ༌། །གྲིན་ཛ་ནི་དང་། ཧར་ཤ་ན་དང༌། དྷ་ད་བ་ཀ་དང་། བ་རཱ་རི་ག་དང་། །ཕ་ར་ནི་དང་། ཕ་ར་ན་དང་། ཀུང་ཀུ་མ་:སང་ཀྲི་ཀ་དང༌། གྲིན་ཛ་སེ་ད་དང་། ས་ལ་ན་དྷ་དང་། ཡ་བ་ནི་ཤཱ་དང་། དུར་གན་དུ་དང་། མུ་ཁ་དུ་ཤ་ཁ་་་་དང་། ལ་པ་དར་ཀ་དང་། ཏ་ཐ་:ཏྲི་ཤ་དང༌། དེ་དག་ནི་པ་ལ་ན་དའི་མིང་ངོ༌། །གྲུ་ཤ་དང༌། བྷི་:ཤག་སེ་ན་དང༌། །ཀཱནྟཱ་དང༌། བ་ར་་་ཧི་དང༌། གྷི་སྟི་ཀ་དང་། བཱ་ར་དང་། །མ་རྭ་བ་དང༌། སོ་ཀ་རི་དང༌། ཀ་ཙི་དང་། ཀསྟི་དང་། བཱ་ནཱ་བཱ་ལི་ནི་དང་། དེ་དག་ནི་སུ་ཀ་རིའི་མིང་ངོ༌། །:（མན་པ་རོ་ཧི་ནི་དང་། ཨ་ཏི་རུ་ཧ་དང་། :བྷིད་ཏ་དང༌། ཙརྨ་ཀ་ཤ་དང་། བི་ཀ་ས་དང༌། མན་ས་ར་ཧ་དང་། རོ་ཧ་དང༌། པི་ཤི་ཏ་རོ་ཧི་ནི་དང༌། དེ་དག་ནི་མཉྫ་རོ་ཧི་ནིའི་མིང་ངོ༌། །ཛ་རུ་ཀ་དང༌། ར་ཛ་དང་། ཀྲི་ཤན་དང་། རཉྫ་ནི་དང༌། ཛ་ན་ནི་དང་། །ཨསྟི་རོ་ཧི་དང༌། ས་ཧ་ཤ་་་་དང་། ཛ་ཟླ་ཀ་དང་། ཙ་ཀྲ་:བར་དི་ནི་དང༌། དེ་དག་ནི་ཨ་ཧི་རོ་ཧི་ནིའི་མིང་ངོ༌། །ཙཀླུ་ཤ་དང་། ཙི་བི་ཏ་དང༌། ཀུནྡྷ'ཀ་རི་དང༌། ཀུ་སྟེལན་ཐི་ག་དང་། །ཀུ་ལ་ལི་དང་། ཨུ་རན་རྒྱ་ཀུ་:ལངྒཱི་ཀ་དང་། དེ་དག་ནི་རྒྱ་སྲན་ནག་པོའི་མིང་ངོ༌། །:སུ་བྱ་ེན་ཛ་ན་དང་། ་་:དུ་ཤེ་ཀེ་ལུ་དང་། ཀོ་ས་མཾ་དང༌། ཀུ་སུ་མ་ཛ་ན་དང༌། ་་:རི་ཏི་དང་། རི་ཏི་ཀ་སུ་མ་དང་། རི་ཏི་:པུ་བྱ་དང་། བ་སྤྱི་དང་། དེ་དག་ནི་ར་གན་གྱི་དུད་པའི་མིང་ངོ་། །ཨི་བི་གི་དང༌། ཛོད་ག་ཀོ་དང༌། ཛི་ར་ན་བ་ལུངྐ་དང་། ་ཨནྡ་ཀོ་ཐ་ཀ་པུ་་་ཤིང་དང་། ཨ་ཛནྡ་དང་། ཚ་གནྡ་དང༌། དེ་དག་ནི་བྷིནྡྷ་དྷ་རུའི་མིང་།
