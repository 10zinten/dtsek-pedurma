སྨན་དཔྱད་ཡན་ལག་བརྒྱད་པའི་:སྙིང་པོའི་འགྲེལ་པ་མཛད་པའི་སློབ་དཔོན་ཟླ་བ་ལ་དགའ་བས་ཡན་ལག་བརྒྱད་པ་ལ་སོགས་པའི་སྨན་གྱི་མིང་གི་རྣམ་་གྲངས་མཛད་པ་རྫོགས་སོ། །།།བའི་སྐབས་ས་བ་་་་དུ་བསྡུར་མཆན། མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་སྨན་དཔྱད་ཡན་ལག་བརྒྱད་པའི་སྨན་གྱི་་་མིང་གི་རྣམ་གྲངས་སློབ་དཔོན་ཟླ་བ་ལ་དགའ་བས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «ཅོ་པ། «པེ་»«སྣར་—གྱི། «པེ་»སྣར་»ད། །«ཅོ་»ཐཱ། «པེ་»«སྣར་པིཎཱ། «པེ་《སྣར་བ། «པེ་»«སྣར་ལ་དུ། «ཅོ་»ལཱ་དུ། @«པེ་»«སྣར་»ཡཉྩི་མ་དྷུ་ཀ་དང་། མ་དྷུ་ཀ་དང་། ཨཤྭ། «པེ་»«སྣར་»ཀ་དང་། ཀླུ་ཏ་ཀ་«པེ་»«སྣར་ག་རྩ་ཀ་ལྡན་བུ་ནི་དང་། ལཝ་དང། ཏུཾ་བི་དང་ཨཱི་ཤྭ་ཀུ་དང་། མ་ཧ་བ།  ེ«པེ་»《སྣར་»ཏིཏྟ་ལཱི་བུ། «པེ་»«སྣར་རིཏྠ་དང༌། །སུ་བྷ་དྲ། «པེ་»«སྣར་»ཏིཀྟ། (ཆོས་ཚན་འདིའི་ནང་དུ་ཏིག་་ཏ་ཞེས་པ་རྣམས་«པེ་»«སྣར་ལ་ཏིཀྟ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་བྷ་དྲ་་་དང༌། བྲ་བྷུ་ཏྲ། པོ«པེ་《སྣར་ཤུགླུ་ལ། «པེ་»«སྣར་»བིམྤ། «པེ་《སྣར་»ཏུནྟ་ཀི་ར་དང་། ཏུནྟི་དང་། ཨོ་ལྟ་ཕཱ་ལ་དང་། དནྟི་དང་། རཀྟ་ཕ། ཨ«པེ་»«སྣར་ཐིམྦའི། «པེ་»«སྣར་བི་ཤ་ལ་དང་། མེ་གེཅུ། «པེ་»«སྣར་ཕ། མུ«པེ་»«སྣར་བཀྲི་དང༌། ཨི་ཏྲཱ་བ་རུ་ཎཱ། ལེ«པེ་»«སྣར་»བི། ཚེ«པེ་»«སྣར་»སཥ། ལོ«པེ་»«སྣར་»སི་དནྡྲི་དང་། ཅི་ཏྲ་ཏེ། ཨེ«པེ་»«སྣར་བི། འ«པེ་»«སྣར་»ཏྲི་པུ་ས་ཁ། ཏེ«པེ་»«སྣར་»ཏིཀྐ། «པེ་»«སྣར་»ཚརྟ་པཎྜི་དང་། མུ་དྲ། ན«པེ་»«སྣར་ཡ། ཀྱེ«པེ་»«སྣར་ཊ་ཛ་དང་། པཉྩ་དང་། པཙྪ་ཀ་དང་། ཏཱ་བིཉྩ་དང༌། ཏུ་ཙྪ་དང་། པྲིཀྵ། འགེ«པེ་»«སྣར་མ། བ་འབྱེ«པེ་»«སྣར་བྲིཀྵ། །འོ་ན་ད་(«པེ་»«སྣར་ཡུངས་ཀྱི། «པེ་»«སྣར་»བཙྪ་ཀཾ་དང་། ཀོ་ཊ། ན«པེ་»《སྣར་»སཱ་ཧཱ། «པེ་»འི::«སྣར་ཡུངས་ཀྱི། («པེ་»ལླཱཧཱ། «སྣར་མུནྡ། «པེ་»«སྣར་»རཱ་སཱ། ཨི«སྣར་»པཎྜི། ་«པེ་»《སྣར་མཱུ་ཏྲཱའི། «ཅོ་བུ། (གྱེ«པེ་བྲི་ཏྲ་ཀོ་ཥ་དང༌། གྷ་རག《སྣར་བྲི་ཏྲ་ཀོ་ཥ་དང་། ལྷ་ར་གི། འགྱེ«པེ་»«སྣར་བཱི་ཊིཀ(མུ«པེ་»«སྣར་ཙིལླ། (ཉེ«པེ་»«སྣར་ཙི་ཏྲ་ཏཆུ། ཚེ«པེ་»«སྣར་»ཛཾ་དུ་གྷ། སྐྱེ«པེ་»«སྣར་»ཏ། ཁྱེ«པེ་»«སྣར་»བི། སྐྱེ«པེ་»«སྣར་»དཱིན། «པེ་»«སྣར་»གནྡྷ། «པེ་«སྣར་»ཀ་སཾ་པྲི། དགྱེ«པེ་»«སྣར་»«ཅོ་»ནཱ། ཀྱེ«པེ་»《སྣར་»ས། འགེ«པེ་»«སྣར་དུམྤ། །འཉེ«པེ་»རྒྱལ་བའི་།
