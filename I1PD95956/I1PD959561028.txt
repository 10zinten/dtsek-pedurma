བསྟན་འགྱུར། གསོ་བ་རིག་པ། ༼ཧེ༽::ན་འབྲས་རཀྟ་ཤཱ་ལི་ཞེས་བྱ་བ་ཕྱི་ཤུན་དམར། འབྲས་ཀྱི་ཁོང་དཀར་མདོག་ཅུང་ཟད་ཙམ་རྩ་སྣར་བ་ཞིག་ཡོད་པའི་མིང་ནི་སཱ་ལུ་རྩ་སྣར་ཞེས་བྱ་བ་དང༌། མ་:ཧཱ་ཤཱ་ལི་ཞེས་བྱ་བ་འབྲས་དཀར་ལ་ཆེ་ཞིང་རིང་བ་ཞིག་ཡོད་དོ། །མིང་ནི་སཱ་ཏི་ལུ་ཆེན་པོ་ཞེས་བྱ་བ་དང༌། འབྲས་ཀ་ལ་:མཆོག་ཅེས་བྱ་བ་དྲི་རབ་ཏུ་ཞིམ་ལ་ཁོང་དཀར༑དེའི་མིང་ནི་འབྲས་མཆོག་ཅེས་བྱ་བ་དང༌། འབྲས་ཏུན་ཀ་ཞེས་བྱ་བ་ས་བོན་བཏབ་ནས་ཞག་དྲུག་ཅུས་སྨིན་ཏེ། །བཟར་རུང་བ། ཏུརྣ་ཀ་ཞེས་་་་་ཀྱང་བྱ་སྟེ། འབྲས་ཀྱི་ལྡང་འདབ་ཕུ་དུང་འདུངས་པའི་ནང་ནས་སྨིན་པ་རབ་ཏུ་བཟང་སྟེ། མཆུ་ཐུང་ལ་ཟླུམ་རྩ་བ་ཀྱེརབ་ཏུ་དཀར་བ་ཡིན། །རོ་ཞིམ་པ་ཉིད་ཀྱང་ཡིན་ཏེ། དེའི་མིང་ནི་ཞག་དྲུག་ཅུས་སྨིན་པའམ། མྱུར་དུ་སྨིན་་པའི་འབྲས་ཞེས་བྱའོ། །འབྲས་ཀྱི་ས་བོན་བྱས་ཁྱེར་ཏེ་འོངས་པ་ཞེས་བྱ་བ་རིད་་པ་ཡང་མ་ཡིན། ཟླུམ་པ་ཡང་མ་ཡིན། ཡིད་འཛུམ་འཇོང་ལྟར་འདུག་པའི་མིང་ནི་བྱས་བླངས་པ་ཞེས་བྱ་བ་དང་། འབྲས་ཤ་ར་མུ་ཁ་ཞེས་བྱ་བ་འབྲས་ཀྱི་དབྱིབས་འགྲོན་བུ་འདྲ་ལ་དཀར་བ་ཞིག་ཡོད་པ་དང་། །འབྲས་:ཇི་ར་ཀ་ཤུ་ཀ་ཞེས་བྱ་བ་མཆུ་རིང་ལ་ཐོར་ཏོ་རྣོན་པོར་འདུག་པ་དང་། འབྲས་རོ་དྷ་ཨེཀ་ཞེས་བྱ་བ་སྨན་རོ་དྷའི་འབྲས་བུ་འདྲ་ལ་རྩེ་མོ་རྣོན་པོར་འདུག་པ་དང་། འབྲས་བུ་གནྡྷ་ཀ་་ཞེས་བྱ་བ་དྲི་རབ་ཏུ་ཞིམ། ་མཐུག་ྋཆེ་ལ་ལེབ། མཆུ་ཐུང་ལ་དཀར་བ་ཞིག་་་ཡོད་པ་དང་། འབྲས་པ་:ཏམ་ག་ཞེས་བྱ་བ་རྩ་སེར་བ་ཞིག་ཡོད་པ་དང་། འབྲས་འདི་རྣམས་དང་། །གཞན་ཡང་རུང་སྟེ། འབྲས་ནི་རོ་མངར། །ཟོས་ནས་ཞུ་བའི་རྗེས་ནི་མངར་ཞིང་སྣུམ། རླུང་ནད་སེལ། བཅུད་དུ་འགྱུར། རོ་ཙ་སྟེ་བུད་མེད་ལ་ཡོད། མཐུར་དུ་བཤང་བ་འོང་བ་ནི་ཆུང་ཞིང་སླ་མོ་མ་་ཡིན་ཏེ། ཟླུམ་པོར་འབྱུང་ངོ༌། །འབྲས་ཆན་ཟོས་ཏེ་མིད་པ་དང་ཁ་ན་རོ་བྲོ་བའི་།
