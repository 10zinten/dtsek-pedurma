།སོ་བི་ར་དང་། ཕི་ནི་ལམ་དང་། ཀོ་ཧ་མ་དང༌། དེ་དག་ནི་རྒྱ་ཤུག་གི་མིང་ངོ༌། །ཀརྐ་ད་ཧ་ས་ར་ཏ་དང༌། སོ་ཏུ་དང་། :：སིན་ཙི་ཏི་ཀ་ཕ་ལ་དང་། དེ་དག་ནི་:སིན་ཛ་ཏའི་འབྲས་བུའི་མིང་ངོ༌། །ཕ་བ་ལམ་བ་དང་། རེ་བ་བ་ཏི་ཀ་དང་། ཨ་རི་:བ་ཏི་དང་། མ་ཧ་ཅིབ་ལི་བ་ཏ་དང་། སྲོ་་དུ་ཤ་ད་ཀ་དང་། དེ་དག་ནི་པུ་ཤིའི་མིང་ངོ༌། །ཨ་རུ་ཀ་མ་དང་། བི་ར་སེ་ན་དང་། ར་ཙི་ན་རི་ཀ་དང་། དེ་དག་ནི་ཁམ་བུའི་མིང་ངོ༌། །།བ་ཙ་ནི་དང་། ཧ་བྱམ་དང༌། བཱ་ལེ་པཱ་དང༌། བྷ་བ་ནི་དང་། བྷག་ཤོ་ད་ན་དང་། :པིད་ཚི་ལ་པི་ཛ་དང༌། རོ་མ་ཤ་ཕ་ལ་དང་། དེ་དག་ནི་བྷ་བ་ནའི་མིང་ངོ༌། །རུ་ལ་དང་། རུ་ད་དང༌། ཡུ་བ་དང༌། ཀྲ་མུ་ཀ་དང་། ྋབྲ་ཧྥ་ཀ་ཤ་དང་། :：སྦྲ་སྨད་ར་དང་། བྲ་ཧྥ་:ད་ནི་ཀླུ་ཞྭ་དང་། ཧྥ་ན་དང༌། བྲ་ཧྥ་བ་རི་ན་དང་། དེ་དག་ནི་སུ་ལ་འོ་སེའི་མིང་ངོ༌། །ཨ་ཤོ་ཏ་དང་། ་བར་བྷ་དེ་ཡ་དང་། ཕ་ལ་:སྣམ་ཧ་དང༌། ་་:གླུ་ཧ་ད་ཤ་ཡ་དང་། ཀི་རི་ཤ་དང་། ཀན་ད་ར་ལ་དང་། སྭ་དུ་མཉྫ་དང༌། བྷི་ཏི་རནཱ་དང་། དེ་དག་ནི་སྟར་གའི་་མིང་ངོ༌། །ཨ་མ་ལ་ཙ་ཀམ་རི་དང་། སུ་ནི་ཤན་ཀ་པ་ཏྲ་ཀ་དང༌། གླུ་ཏ་ཨམི་དང་། ཙ་དུ་ཤཎི་དང་། ཙུག་ཏྲ་དང་། :དན་ན་ཤ་ཐ་དང་། ཨམྲི་ཀ་དང་། དེ་དག་ནི་ཙུ་ཀ་ཀའི་མིང་ངོ༌། །ཨམ་ཀ་དང༌། ཙུ་་་ཀྲི་ཀ་དང༌། ཤུཀྟི་དང་། །ཙནྡ་དང༌། ཙུག་ཨིཀྲི་དང་། ཐུགཱིཨིཀ་དང་། པི་ཏ་དང་། ཏིནྟ་ཨེནི་ཀ་དང༌། ཨམ་དང་། ཏིཏྟི་ཏོད་ཀ་དང་། སུ་ཏིད་ལ་དང་། དེ་དག་ནི་ཙུ་ཀྲི་ཀ་རྣམ་གཅིག་གི་མིང་ངོ༌། །གླུ་ཙེཏ་པ་དང་། ཏིཥྛ་ཅེས་རུ་དང་། ཧོ་དཱ་ཤ་ན་དང༌། །སོ་ཅེཕ་ལ་དང་། ཀ་ཁྱུ་ཏིག་ཏ་་་་དང༌། ཀྲ་ཤ་ར་:སི་ག་བྷཱ་ཀཎྜ་ཅིཀ་དང༌། དེ་དག་ནི་ཀ་རི་རའི་མིང་ངོ༌། །།
