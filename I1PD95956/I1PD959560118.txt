་དང་། བྷ་:ན་སོ་གནྟི་ག་དང་། སོ་:གནྟི་དང་། གནྡྷ་ཀུ་དང་། གནྡྷོ་དང་། གནྡྷ་ཨ་:མ་དང་། པི་ད་གནྟ་ཀ་དང་། དེ་དག་ནི་མུ་ཟིའི་མིང་ངོ༌། །ཨ་བྲ་གྲ་དང༌། བི་མ་ལ་དང་། །ཤུ་བྷུ་དང༌། བ་ར་པི་ཏ་དང་། པི་ཏ་ཀ་་དང་། ནིརྨ་ལ་དང་། :བར་བྷ་ཏོ་བྷུ་ཏ་དང་། ཨ་བྲ་པ་ཏ་ལ་དང་། དེ་དག་ནི་ལང་ཚེར་བའི་མིང་ངོ༌། །བ་ལུ་ཀ་དང༌། ་་:སྲིད་ཏ་དང༌། སི་ར་དང་། ལོ་བྷ་མ་ཏ་དང༌། པྲ་ལོ་བི་ནི་དང་། ནཱ་ནཱ་ད་ཏུ་མ་དང༌། :·སུག་སྨྲ་དང་། སོ་:བར་ན་སཾ་བྷ་བ་དང་། དེ་དག་ནི་གསེར་ལ་སོགས་པ་་འབྱུང་:ཁུང་གི་མིང་ངོ༌། །ཏ་པྱ་དང་། མ་ཀྱི་ཀ་:དྷ་ཏུ་དང་། ཏ་བྱ་ཛ་་་དང༌། ཏ་བྷ་ཌ་མ་ཀྲི་ཀ་དང་། ཨ་བརྟ་དང༌། བྷ་མ་དྷུ་ད་ར་དང༌། ཏི་བི་ས་མུདྡྷ་དང་། དེ་དག་ནི་བུ་ལེན་ཏེ་དུབ་མར་རྫོང་གི་མིང་ངོ༌། །ཤཾ་ཁ་དང༌། བྷི་རི་བྷ་དང༌། ཀུསླུ་དང༌། ཛ་ལ་ཛ་ལ་དང་། །དྷིརྒ་ནི་ས་་་་་དང་། སུ་སྲ་རོ་དང༌། དྷིན་བ་དང་། དྷ་བ་ལ་དང་། སྲི་བྷི་བྷུ་ཤ་ན་དང༌། དེ་དག་ནི་དུང་གི་མིང་ངོ༌། །ས་མུ་དྲ་ཕེ་ན་དང༌། :：ཕེ་ན་དང་། ཤུ་ཀླུ་ཤུ་ཀླུ་དང༌། བ་ཡུ་ཏི་ཛ་དང༌། སོ་:དྷ་ནི་ཕེ་ན་དང་། ས་ཚིག་ར་མ་ལ་དང་། དེ་དག་ནི་རྒྱ་མཚོ་དབུ་བའི་མིང་ངོ༌། །གོ་རོ་ཙནྡ'དང་། པི་ཅིག་ལ་དང་། རུ་ཙི་ར་དང་། རུ་ཙི་དང་། མེ་ད་དང་། གོ་རི་དང་། གོ་ཏ་མི་དང་། བནྡ་ནི་ཡ་དང་། གྲ་ཧ་ཕ་ཧ་དང་། དེ་དག་ནི་གི་ལོཝང་གི་་མིང་ངོ༌། །སིནྡྷཱ་ཅིར་དང་། རཀྟ་རེ་ན་དང༌། ནཱ་ག་རཀྟ་དང་། ནཱ་ག་ཛ་དང་། སྲི་ག་པུ་ཤ་ན་དང་། སྲི་མནྟ་དང་། བ་སནྟུད་པ་ཙ་མནྡཱ་དང་། དེ་དག་ནི་ལི་ཁྲིའི་མིང་ངོ༌། །:：ཀཾ་ཀུ་ལིཤ་ཏ་དང་། བཱི་རངྒ་དང་། །རངྒ་ན་ག་དང་། རོ་ཙ་ན་དང་། མཱུ་ཙལ་ཀ་དང་། ཏྲ་ས་དང་། ཤོ་དྷ་དང་། ཀ་།
