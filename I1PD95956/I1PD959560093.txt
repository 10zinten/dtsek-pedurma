ལི་ཀ་དང༌། ་ཨུནྡ་ར་ཀཎ་ཀ་དང་། དེ་དག་ནི་མུ་ཤ་ཀ་ཀཎིའི་མིང་ངོ༌། །ཀླུ་པ་ཀ་དང༌། :ཏིཥྛ་ན་དང་། ཀྲུ་ར་དང་། ཨུ་དྭ་ཛ་ན་དང་། ཀློ་ད་ཏི་ཛ་ན་དང་། ་་:ར་ཛ་གློ་བ་དང་། ཀློ་ཏྲ་བོ་ད་ན་དང་། དེ་དག་ནི་ཀླུ་པ་ཀའི་མིང་ངོ༌། །ཀ་:པིད་ཏ་པ་ཏྲ་དང་། བ་རི་ཤི་དང་། གིར་ཛ་ར་དང་། ཙེ་ཏ་པ་ཏྲ་དང༌། བི་རུ་པ་དང་། ཀུ་ལ་ཛ་དང་། བ་ན་ཛ་ན་དང་། པ་ཤཱ་པ་ཏྲི་ཀ་དང་། །དེ་དག་ནི་ཤིང་ཀ་:པིད་པའི་ལོ་མའི་མིང་ངོ༌། །པྲ་ཙི་བ་ལ་དང་། ན་ཛི་ཀཎྜ་དང༌། ཛ་ལ་ཛ་དང༌། རཀྟ་མཉྫ་རི་དང་། ནིསྣ་ཏེཛ་རུ་དང་། ཛ་ཡ་ད་ནི་དི་ཀ་མུ་ཁ་དང༌། དེ་དག་ནི་རཀྟ་མཉྫ་རིའི་མིང་ངོ་། །ཀ་ཀ་མ་ཛི་དང༌། །བ་ཧུ་བ་ལ་དང་། ཀ་མ་ཏ་དང་། ཛ་ག་གི་ཕ་ལ་དང༌། ད་ཀླུ་མ་ཙི་དང་། གུ་ད་པ་ལ་དང་། ཀུབྷ་ཅིག་ནི་དང་། བ་ཡུ་སི་དེ་དག་ནི་མོན་སྣེའུ་:མེ་རུའི་མིང་ཡིན་ནོ། །ཀུ་ལ་ཧ་ལ་དང༌། ལ་བུ་ས་དང༌། :：གུད་ཚ་ལ་དང་། ཁ་ཤ་ཁ་དང་། ནི་ཙ་དམྤ་དང་། བེ་ཏ་དང་། །སྟ་:བི་་ར་དང༌། བུ་ཀ་ཏམ་དང༌། དེ་དག་ནི་ཀུ་ལའི་མིང་ཡིན་ནོ། །བེ་ཤ་མུ་སྟི་དང༌། གློ་པ་མུ་སྟི་དང་། སུ་མུ་སྟི་དང་། བི་ཤ་ན་དང་། ཨུ་ཡུ་ཏ་མུ་ཏི་དང༌། ཀློ་མ་ཧ་དང་། ཙ་ཀ་མུ་དང་། :：ཤིང་དང་དེ་དག་ནི་ཤ་མུ་་་་སྟིའི་མིང་ངོ༌། །ནུ་སྲི་ན་དང༌། གུ་ལ་བི་ཙ་དང་། སུ་གནྡྷ་དང་། སུ་ན་དྷཱ་ན་ཤ་ན་དང་། མ་ལ་ཏྲི་ན་དང་། :：ཤིང་གའི་ཛ་ཅདང༌། བ་ཏི་ར་དང་། །ཤརྡ་རོ་ད་ན་དང་། དེ་དག་ནི་བུ་སྲི་ནའི་མིང་ཡིན་ནོ། །ནི་རྒུན་ཐི་དང༌། བ་ན་ཀ་དང་། །ཨིནྡྲ་ནི་དང་། སྲིནུ་ཅིབ་ར་ཚ་དང་། ནཱི་ལ་མཉྫ་རི་དང༌། མ་རི་ཙ་དང༌། བུ་ཏ་བེ་ཤི་དང་། སུ་གནྟི་ག་དང་། དེ་དག་ནི་ནི་་་རྒུན་ཐིའི་ྋམིང་ཡིན་ནོ། །མུཀྐ་ཀ་དང་། ཏིཥྛ་ཀ་དང་། ཤ་རི་ཀ་དང་། །
