རབསྐྱེད་པའི་:སྐྱེས་པའི་རབས་སོ། །སྐྱེས་པའི་རབས་ཀྱི་རྒྱ་ཆེར་བཤད་པ་སློབ་དཔོན་དྷརྨ་ཀཱིརྟིས་མཛད་པ་་་རྫོགས་སོ། །།།།།རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་ཛ་ནཱར་དྷ་ན་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་ཨི་་བུ་དགེ་སློང་ཤཱཀྱ་བློ་གྲོས་ཀྱིས་བསྒྱུར་བའོ། །།།ལེ་བ་དག་འོག། བསྡར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། །དཔའ་བོའི་སྐྱེས་རབས་ཀྱི་རྒྱ་ཆེར་བཤད་པ་བཞུགས་སོ། །ཞེས་འཁོད། Q«པེ་《སྣར་»ཛ། ན«པེ་»ཀ«སྣར་»ལ། «པེ་«སྣར་ཕན། «པེ་»«སྣར་»ཚིག@«པེ་»«སྣར་»གིས། ་«པེ་»«སྣར་»+ཞེས་བྱ་བའི་བར་དུ་སྟེ། ཉི《སྣར་»སོ། «སྣར་+སུ། «པེ་»«སྣར་པའོ། «པེ་»«སྣར་»ཏེ། ཁ༼?་་་་་་«པེ་»«སྣར་པ་དང༌། «པེ་»«སྣར་བསྔོ། «པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་བྱ་བའི། [ལོ་ཚེ«པེ་»«སྣར་—བཞིན་དུ། «པེ་»སྨྲས། «སྣར་»སྨྲོས། ༄༅«སྣར་པར། ་་་་་་་«སྣར་»ན། «པེ་»«སྣར་+སྤྱོད་པ། «པེ་»«སྣར་མི། «པེ་《སྣར་+དང་། «པེ་»«སྣར་»སྦྱོར། @«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»དོགས། ཉི«པེ་»ཚིག་བཅད། །«སྣར་ཚིག་་་བཤོགཆད། «པེ་»«སྣར་ཀྱིས། ད་«པེ་ལ་བརྩམས། 《སྣར་»ལ་བསམས། 《སྣར་བསྟེན། ་«པེ་《སྣར་»ཀྱིས། «པེ་»་གི་«སྣར་པ། O«པེ་»«སྣར་གྱིས། ་ོ«པེ་»«སྣར་»+ཡང་། ༄༅«སྣར་ཤེས། «པེ་»«སྣར་»བས། «པེ་»«སྣར་ནི། གྱེ«པེ་»«སྣར་ན། ཀྱེ«པེ་»«སྣར་གྱུར། ་ཀྱི་འོད་«པེ་»«སྣར་འགྱུར་བའི། «པེ་»«སྣར་པ། «པེ་»«སྣར་གྱི་དོན་དུ། ®«པེ་»«སྣར་བྱ་བ། །«པེ་»«སྣར་+ལ། @«པེ་»«སྣར་གྱུར་པ། @«པེ་»«སྣར་»—དང་། «པེ་»«སྣར་»ཏུ༑«པེ་»«སྣར་རྣམ་པ། (༌«པེ་»«སྣར་དེ། ཀྱེ«པེ་»«སྣར་»གི། ཡེ«པེ་»«སྣར་»་པ། :/ཉེ«པེ་»«སྣར་གསུང་པའོ། །རིག་པ་ནི་རིགཁྱེ«པེ་»«སྣར་རིགསྐལ་«པེ་»«སྣར་བའོ། «པེ་»«སྣར་པར། «པེ་བྱའོ། «པེ་»«སྣར་བྱ་བའོ། «པེ་»ཀྱིས་《《སྣར་»གཅོད། 《@«པེ་»«སྣར་བའོ། །་་་་་།
