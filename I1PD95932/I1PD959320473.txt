ཆསོར་ཐར་པའི་བཤད་སྦྱར་འདི། །འདུལ་བ་རྒྱ་མཚོ་ལས་བཏུས་པའི། །བསོད་ནམས་ཅུང་ཟད་གང་བསྐྱེད་དེས། །འགྲོ་རྣམས་སངས་རྒྱས་འགྲུབ་པར་ཤོག། འཕགས་པ་ཐམས་ཅད་ཡོད་པར་སྨྲ་བའི་རྩ་བའི་དགེ་སློང་མའི་སོ་སོར་ཐར་པའི་་མདོའི་འགྲེལ་པ་རྫོགས་སྙོ། །།།བ་བསྡུར་མཆན། :མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། («པེ་《སྣར་»བིཀྵ་ནི། «པེ་»ཀློཏོགས༑་བར་་ས་ལ་ན་འཁོར་ལོས་བྱ་བ་གནང་། དེ«པེ་»ཀྱིས། «པེ་»འདུག་«པེ་»«སྣར་བརྟུལ་པོ་རྣམས་ཀྱི། «པེ་»«སྣར་»། སུ«པེ་»སྡུས༑། «པེ་»«སྣར་»རིམ། (ཆོས་ཚན་འདིའི་ནང་དུ་རིམས་ཞེས་པ་རྣམས་«པེ་»«སྣར་»ལ་རིམ་ཞེས་་་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་བསྟན། ་་་་་་་་(ི«པེ་»«སྣར་པར། Q«པེ་པས། «པེ་»སྣར་བྱ། «པེ་»«སྣར་+སྡེ། 《སྣར་»—འདི། («པེ་»«སྣར་འཆད། ་ ེ«པེ་《སྣར་—ན། «པེ་»སྣར་བཞག(ཆོས་ཚན་འདིའི་ནང་དུ་་་གཞག་ཅེས་པ་རྣམས་«པེ་»«སྣར་»ལ་བཞག་ཅེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། ®«པེ་»«སྣར་»བའི༑། «པེ་»«སྣར་བསྒྲུབ། ་་་་་་(«པེ་»འབེལ་བའི་གཏམ། «སྣར་འབྲེལ་བའི་གཏམ། །«པེ་»«སྣར་བ། །(«པེ་《སྣར་བསྐྱེད། 《སྣར་བུ། «པེ་»«སྣར་པར། ®«པེ་»«སྣར་+མ། «པེ་»《སྣར་»ལ། «པེ་《སྣར་བསླད། «ཅོ་པར། །:«པེ་»«སྣར་བསྐྱེད། «པེ་»《སྣར་»པ། འ«པེ་»《སྣར་»ཡོངས། ®«པེ་»«སྣར་»གཞན། །«པེ་»《སྣར་མ། «པེ་»«སྣར་བ། །(«པེ་»«སྣར་—པ། «པེ་《སྣར་»བ། «པེ་»«སྣར་དེས། ®«པེ་《སྣར་»བྱས། «པེ་»«སྣར་»པོའི། 《«པེ་《སྣར་པར། ་«པེ་»«སྣར་ཀྱིས། །«པེ་»«སྣར་པར། ་«པེ་»«སྣར་»པ། ་ན«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»པའི། ་«པེ་»《སྣར་+བ། «པེ་»«སྣར་སྦྱོང་། @«ཅོ་»བྱ་བའི། «པེ་《སྣར་བརྟན། «ཅོ་ནི། །(ི«པེ་»«སྣར་+པ། «པེ་»«སྣར་»ཀྱིས། །«སྣར་བདེ། ®«པེ་»«སྣར་—པ། ་་་་་(«སྣར་»སྤོང་བའི། «ཅོ་སྤངས་པའི། «པེ་»«སྣར་»པོའི། །«པེ་»«སྣར་»ཤེས་བྱས་པས། །
