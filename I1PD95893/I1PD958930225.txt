བའབpeབ་དང་། ལུགས་གང་། །ལྟ་བུ། །དང་༄༅། །འཆི་མེད་ཀྱི་དངོས་གཞི་བཅུ་བདུན་གྱི་པ། པའི། །རྣམ་གཞག་བཞུགས་སོ། །ལ་དལ་ལེ་ཟླ་བ་ཚོ་དང་། ::ཀྱིདང་མ་བསང་། །ཁས་། ས། ག་དཔོ་ཞེ་དྲག་ཨང་། དེ་རྗེ། །བྱ་༄༅། །དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་ལོ། །དངོས་གཞི་་དང་པོའི་རྣམ་གཞག་ནི། །བུམ་པ་ཅན་ལ་བསླབ་བྱ་སྟེ། །སྔོན་འགྲོ་དངོས་་གཞི་རྗེས་གསུམ་མོ། །དང་པོ་སྔོན་འགྲོའི་རིམ་པ་ནི། །ཞབས་གཉིས་པདྨའི་སྐྱིལ་ཀྲུང་བཅའ། །ལག་གཡོན་གྱེན་ལ་བསྒྲེང་བྱ་སྟེ། །ཁུ་ཚུར་དག་ནི་བཅང་བྱས་ནས། །ལག་གཡས་སྟོབས་བསྐྱེད་ཐལ་མོ་ཡིས། །གཡོན་པའི་ཕྲག་གདོང་དག་ལ་ནི། །ལན་བདུན་བར་དུ་མནན་བྱས་ཤིང་། །གཡས་པ་ལ་ཡང་དེ་བཞིན་ནོ། །ཞབས་གཉིས་པདྨའི་སྐྱིལ་ཀྲུང་བཅའ། །ལག་གཉིས་རྒྱབ་ཏུ་་་བསྣོལ་བྱས་པས། །ཟེ་ཤ་གཉིས་ནི་དྲག་ཏུ་བསྡམ། །ཡར་གྱི་ཚིགས་པ་བདུན་པ་རུ། །ལག་གཉིས་བསྣོལ་བའི་ལག་ངར་གྱིས། །ལན་བདུན་བར་དུ་མདུན་་དུ་དབུལ། །ཞབས་གཉིས་པདྨའི་སྐྱིལ་ཀྲུང་བཅའ། །སྒལ་ཚིགས་བསྲང་ལ་མགྲིན་པ་དགུག། ལག་གཉིས་ཁུ་ཚུར་བཅང་བྱ་སྟེ། །མདུན་དུ་རབ་ཏུ་སྟོབས་།
