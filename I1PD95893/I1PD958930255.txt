བུ་མི་འི་བསྟན་འགྱུར། རྒྱུད། Pཆེ་ནི་པད་མའི་སྐྱིལ་ཀྲུང་བཅའ། །ལག་གཉིས་བསྣོལ་བས་རྗེ་ཤ་བསྡམ། །གྱེན་ལ་ཕྱར་བའི་སྦྱོར་བ་ཡིས། །མདུན་དང་གཡས་གཡོན་བརྡེག་པར་བྱ། །་ཏrpཚོགཉིས་པ་དངོས་གཞིའི་རིམ་པ་ནི། །ཞབས་ནི་པད་མའི་སྐྱིལ་ཀྲུང་བཅའ། །པད་མ་གྱེན་ལ་བསླང་བྱ་སྟེ། །ཚིགས་པའི་མར་མཚམས་བརྟན་བྱས་ཅིང་། །ལག་གཉིས་སྟོབས་བསྐྱེད་སྦྱོར་བ་ཡིས། །སྒྱིད་ཁུགས་གཉིས་སུ་གཞུག་བྱས་ཤིང་། །ཁུ་ཚུར་རབ་ཏུ་བཅང་བྱས་ཏེ། ཡོ་། གཤིབས་ཏེ་ནམ་མཁར་ལྡིང་བྱས་ཤིང་། ་ཁ། འོག་རླུང་དྲག་པོར་ཐེན་བྱས་ཏེ། །སྟེང་རླུང་དྲག་པོར་མནན་བྱས་ནས། །།སྤྲུལ་པ་དག་ཏུ་བསྐྱིལ་བྱ་སྟེ། །སྟོབས་བསྐྱེད་རབ་ཏུ་བསྒོམ་པར་བྱ། །ཀ་ཁ་nynགསུམ་པ་རྗེས་ཀྱི་རིམ་པ་ནི། །ཞབས་གཉིས་པད་མའི་སྐྱིལ་ཀྲུང་བཅའ། །ལག་གཉིས་བསྣོལ་བས་རྦ་ཤ་བསྡམ། །གྱེན་ལ་བཏེག་པའི་སྦྱོར་བ་ཡིས། །ཡན་ལག་དང་པོས་གཞུག་བྱས་ཤིང༌། །སྤྱི་གཙུག་ཕྱར་ཞིང་བརྡེག་པར་བྱ། །ཚེ་། །°བར། །ཁྲོ་ཉིཏྟཀ་pa། །ན་/sཆཆ། &ཚnFEETTv'sỜITASa::སྔོན་འགྲོ་དངོས་གཞི་རྗེས་གསུམ་མོ། །དང་པོ་སྔོན་འགྲོའི་རིམ་པ་ནི། ལ་། གན་་རྐྱལ་དག་ཏུ་རྙེས་བྱས་ཤིང་། །ལག་གཉིས་ས་ལ་བཙུགས་བྱས་པས། །ལུས་པོ་ནམ་མཁར་ལྡིང་བྱས་ཤིང་། །རྐང་གཡས་མདུན་དུ་ལན་བདུན་སྤྲུག། གཡོན་པ་ཡང་ནི་དེ་བཞིན་ནོ། །༌་ཚོ་་༽པས། །ཎ་ར། ཡུ་པ་བ་ཁ་པ་ལ་གཉིས་པ་དངོས་གཞིའི་རིམ་པ་ནི། །ཞབས་གཉིས་བྲང་པས་ས་ལ་བཙུགས། །རྟིང་པ་གཉིས་ནི་སྤྲད་བྱས་ཤིང་། །ཙོག་གུའི་ཚུལ་གྱིས་འདུག་བྱས་ཏེ། །།ལག་གཉིས་སྟོབས་བསྐྱེད་གྱེན་ལ་བསྒྲེང་། །ཁུ་ཚུར་དག་ནི་བཅང་བྱས་ནས། །ལུས་སྤྱི་སྟོབས་ཀྱིས་བརྐྱང་བྱ་སྟེ། །ལན་བདུན་བར་དུ་ལྡང་བར་བྱ། །ལོ། །་པ་ཚོ་།
