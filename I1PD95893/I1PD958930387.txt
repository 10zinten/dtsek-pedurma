བསྟན་འགྱུར། སྔགས། Pསྟོབས་བསྐྱེད་མདུན་དུ་བརྐྱང་བར་བྱ་ཞེས་པས་ཐིག་ལེ་མ་ཁྲེམས་པའི་སྐྱོན་སེལ་་ལོ༑། སྟོབས་:བསྐྱེད་རབ་ཏུ་བསྡམ་པར་བྱ་ཞེས་པས་མཆན་ཟུར་ན་རྩ་བྲན་ཏ་ལའི་ལོ་མ་ལྟར་གནས་པ་ལ་རླུང་ཤོར་བའི་སྐྱོན་སེལ་ལོ། །སྟོབས་བསྐྱེད་རབ་ཏུ་བསྒོམ་པར་བྱ་ཞེས་པས་རྐང་པ་གཉིས་ཀྱི་རྩ་རྣམས་འཁུམས་པའི་སྐྱོན་སེལ་བར་འགྱུར་རོ། །སློབ་དཔོན་ཆེན་པོ་འབིར་ཝ་པས་མཛད་པའི་རླུང་སྐྱོན་སེལ་བའི་གོ་རིམས་ཡི་གེར་བཀོད་པའོ། །:། ཡུ་ཡཱ་ཏུ། ལ་པོ། བོད་ཡུལ་དུ་རྣལ་འབྱོར་གྱི་དབང་ཕྱུག་ཆེན་པོ་ཨ་མོ་ཀ་བཛྲས་རང་འགྱུར་དུ་མཛད་ཅིང་གསུངས་སོ། །སྟི། །ལུས་པ་མེད། །་དང་ཟློས་སྐྱོན་པ་དེ་། ང་ཁཡོལ་སཔའིའོད་ཀྱང་། ང་འ་ན། །:ཀ་:སྤྱི་:ཀྱི་ཡོད་པ་བསྒྱུར་མཆན། ཀ། ད་པ་འ་ལ་འ་མོ་ཁ་ཨེམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་སྐྱའཐིང་། «པེ་»—། @e«པེ་»ཀར། «པེ་»འཐིང་། @ཚེ«པེ་»འཚམས། @«པེ་»འཐིང་། པ་ཁྱེ«པེ་མར། །@@«པེ་»ཀར། ཚེ@«པེ་»འཐིང་། «པེ་»ཀར། ག«པེ་»འཐིང་། «པེ་»ཀར། «པེ་སྐྱའཐིང་། «པེ་»—པ། «པེ་»ཙུག«པེ་»བསྔོ། །«པེ་»ཙིབ། «པེ་སྲག«པེ་»འགོ། །ཚེ«པེ་»སྐེད། «པེ་»+སོང་། «པེ་»ཞེས། «པེ་»དྲང་། «པེ་»རྒྱས། «པེ་འགུལ། ཀྱི«པེ་»ཟློག«པེ་བའི། ཉི«པེ་རྒྱུ། «པེ་»སྤྲུག་ཚུ«པེ་»སྐྱེད། @«པེ་སྐབསྐྱེད། ཨེ«པེ་»མའི། «པེ་»སྨད། «པེ་སྐྱེད། «པེ་ཉམ། །«པེ་»གསང་། «པེ་»དཀྱིལ་དཀྲུང་། «པེ་»འགུལ། «པེ་»+ཞེས། ཚེ«པེ་རྩེ་བོ། «པེ་དམིགས། «པེ་»སྐྱེད། «པེ་»མཆོག་པར། «པེ་+རླུང་། «པེ་»སྐྱེད། ༼@«པེ་སྐྱེད་རབ། /O«པེ་»སྐྱེད། ཁ་མ་གང་བཏུད་ན་ལྷོ་ཉིན་ཏེ། དེ་ལ་ལོ་ངོ་ལྔ། ::ཀྱིངབའི་ཡེ་།
