་:་་་་་་་་་་་་་། དེ་དག་ལ་བྱས་ན་འདི་པ་ཀློ་གྲ་ཡི། །ཤིག་ལ་དབུ་མjpgདེ་དུ༄༅། །འབྲས་བུ་ཡེ་ཤེས་ས་གཅིག་དང་གང་::::བོད་དུ་ས་འདིའི་བྱ་ཆོས་ཀྱི་དབྱིངས། ངས་ད་སྐྱེ། ག་ག་ང་བགགས་འགའ་དང་་་༌་་་་་་ཀྱི་མིན། ༄༅། །རྒྱ་གར་སྐད་དུ། ཕ་ལ་ཛྷ་ནཱ་དྷརྨ་དྷ་དྷུ། བོད་སྐད་དུ། འབྲས་བུ་ཡེ་ཤེས་ས་གཅིག་ཆོས་ཀྱི་དབྱིངས་འདི་ལ་སྐབས་བརྒྱད་དེ། སྐྱེ་མེད་་བློ་འདས་ཀྱི་སྐབས། བྱ་མེད་རྩོལ་བྲལ་གྱི་སྐབས། སྐུ་བཞི་ཆིག་ཆོད་སྐབས། བགྲོད་མེད་རྫོགས་པའི་སྐབས། །སྟོབས་ལྡན་མཆོག་གི་སྐབས། དེ་ཉིད་དཔེ་དོན་སྐབས། དྲང་བ་དོན་གྱི་སྐབས། ཐུགས་རྗེ་གདམས་པའི་སྐབས་སོ། །སྐབས་དང་པོ་ལ་གསུམ་སྟེ། སྦྱོར་བ་དགོད་པའི་གདམ་ངག་དང༌། གདེང་ཆེན་དགུའི་གདམ་ངག་དང་། མ་སྐྱེས་ཅེས་སྐྱེ་མེད་བྱ་བརྩལ་བྲལ་བའོ། །སྦྱོར་བ་དགོད་པ་ལ་གསུམ་སྟེ། གདོད་ནས་ཆོས་ཅན། ཐོབ་བསྒྲུབ་དམ་བཅའ། རེ་དོགས་གཏན་ཚིགས་སོ། །གདེང་ཆེན་དགུ་ནི། སྐྱེ་མེད་ཅེས་དོན་གྱི་གདེང༌། བརྗོད་མེད་ཅིས་ཚིག་གི་གདེང་། དེ་ཉིད་དེ་ཁོ་ན་ཉིད་སྟོང་པའི་གདེང་། མཐའ་བྲལ་ཅེས་ལྟ་བའི་གདེང་། གཉིས་མེད་ཅེས་བསྒོམ་པའི་གདེང་། དོན་དམ་ཀློང་—།
