རབསྟན་འགྱུར། རྒྱུད། P༎ཞི་བའོ། ཁོ་ཌ་ཀ་ཌ་ཀྱི་ས་ཁོ་ན་དང་། ་་ལྔ་ལ་གངས་ཁོganལ་ད་ན་ནི། །ས་ག་ཟློབཞུགས། pལ། བསྡུར་མཆན། གག་ཏེ་ལམ་ཤབ་མ་མི་མགོ་ད་དེ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཁོར་བ་རྩད་ནས་གཅོད་པ་གསེར་གྱི་ཐིགས་པའི་མན་ངག་ཅེས་བྱ་བ་བཞུགས་སོ། །@«པེ་»གདམས། «པེ་»—དབརྡའ། །»«པེ་གདམས། ང་«པེ་»གསད། ཚེ«པེ་སྟེར། «པེ་»མེད། «པེ་རྟོག«པེ་»ཆོས། ཉི«པེ་། །༎«པེ་། ཡོད་ན་མ་བུ་པ་མི་འབར་ཞིང་མི་འཁྲིགས་སོ། །ཁ་བྱང་གཏེར་གྱི་བ་གམ་གཞི་། S་ཁཀྐ། །ན་ཁ''འི་ཁོབསྡུར་འབྲས་རེའུ་མིག་། ད་ལེང་ཁ་ཤིའ་ཤ་ཚོ་ག་རྒྱུད། （ཟླ་ལུ་བྱ་བ་དེ་ན་སྡེ་དགེ། །པེ་ཅིན། ག་སྣར་ཐང་། སྔོགས། སྔགས། དཔེ་གཞི། སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། བ་བ། པ་:ཕྲེང་། ཚིག་རྐང་། ཚེག་ཁྱིམ། —།
