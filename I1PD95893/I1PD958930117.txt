རྒྱུད། Pགདམ་ངག་དང་། ལ་བཟླ་ཆེན་པོའི་གདམ་ངག་དང་། ཥདང་པོའི་དེ་ཕྱིར་ཅེས་པ་གཅིག་གཏན་ཚིགས་མཚམས་སྦྱོར་གྱི་མན་ངག་གོ། །གོང་གི་དེ་ཕྱིར་གཉིས་མེད་བདེ་ཅེས་པ་དང་ཐ་མའི་རྒྱ་ཆད་ཕྱོ་དང་གཉིས་སོ། z། ལ་བཟླ་ཆེན་པོའི་གདམ་ངག་ལ་བདུན་ཏེ། མུ་སྟེགས་ཀྱི་རྟག་ཆད་ཅེས་པ་དང་། རྟག་ཆད་གཉིས་གཉིས་སུ་མེད་ཅེས་པ་མེད་དུ་ལ་བཟླ་བ་དང༌། ཉན་རང་། གི་གཟུང་འཛིན་གཉིས་སུ་མེད་བཟུང་འཛིན་གཉིས་མེད་དུ་ལ་བཟླ་བ་དང༌། དབུ་མའི་ཀུན་རྫོབ་དོན་དམ་བདེན་པ་གཉིས་མེད་དུ་ལ་བཟླ་བ་དང་། ་ཀྲི་ཡོག་བླང་དོར་ཅེས་པ་དང་གི་་་་སྤྱོད་པ་བླང་དོར་གཉིས་གཉིས་སུ་མེད་ཅེས་པའི་མེད་དུ་ལ་བཟླ་བ་དང་། མ་ཧཱའི་སྤྱོད་པ་སྤྱོད་འབད་གཉིས་སུ་མེད་པ་ཅེས་འབད་རྩོལ་གཉིས་མེད་དུ་ལ་བཟླ་བ་དང༌། བླ་ཨ་ནུའི་དབྱིངས་དབྱིངས་དང་ཡེ་ཤེས་གཉིས་མེད་དུ་ལ་བཟླ་བ་དང༌། འཁོར་ཐམས་ཅད་གཉིས་སུ་་་ཅེས་འདས་གཉིས་མེད་རྫོགས་པ་ཆེན་པོར་ལ་བཟླ་བའོ། །ཀ་་་nཀཀ་། pསྐབས་བདུན་པ་ལ་གསུམ་སྟེ། ལྷུན་གྲུབ་ཆེན་པོའི་འགྱུར་བ་མ་ཡིན་གདམ་ངག་དང༌། རང་འབྱུང་ཅིས་པ་གཉིས་ཡེ་ཤེས་ཀྱི་གདམ་ངག་དང་། བདེ་ཆེན་ཚིག་ཙམ་ཅེས་པ་རང་ཤར་གྱི་གདམ་ངག་གོ། z:བབབབབབབབཔལ་སྐབས་བརྒྱད་པ་ལ་གཉིས་ཏེ། ཟབ་མོའི་གདམ་བདེ་ཆེན་ཅེས་ངག་དཔེའི་སྒོ་་་ནས་གསང་བ་དང༌། སྣོད་གང་ལ་གསང་འདོད་པ་མ་ཅེས་བ་བསྟན་པའོ། །ཡེ་་་་ཤེས་གསལ་བ་ཆོས་ཀྱི་དབྱིངས། །བསྒོམ་པ་ཤེས་པའི་ཤེས་བྱ་ལ། །སྐབས་བརྒྱད་ཚེག་བར་བདུན་པའོ། །ཤད་བར་བརྒྱད་དང་བཞི་ཡིས་ཚང་། །ཡང་་་་གསང་ཐུགས་ཀྱི་ལྡེ་མིག་ཅེས་བྱ་བ། སློབ་དཔོན་ཤྲཱི་སིང་ངས་མཛད་པའོ། །རྫོགས་སྙོ། །།་་་་་་་་་་་་་་་་་་་་་་འཚོ་ཊ། ད་པ་དང་། བྱང་། ད་ལོ་ལོ་ན་བས་དགའ་བའི་ས། །པར། དེ་བློས་གར། ད་དུང་བག་མ་དེ་—།
