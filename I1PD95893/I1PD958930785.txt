རྒྱུད། ་ས་པཎྜི་ཏ་ཆེན་པོ་དེ་ཉིད་དང་། འགོས་ལོ་ཙ་བ་དགེ་སློང་གཞོན་ནུ་དཔལ་གྱིས་བསྒྱུར་བ་ལས། སླད་དུ་རྗེ་བཙུན་གྱི་ཞལ་སྔ་ཉིད་ལ་ལེགས་པར་མཉན་ཞིང་དོན་ཁོང་དུ་ཆུད་པར་བགྱིས་ནས། །དགེ་སློང་བསོད་ནམས་རྒྱ་མཚོའི་སྡེ་ཞེས་བྱ་བས་ཅུང་ཟད་བཅོས་ཏེ་གཏན་ལ་ཕབ་པའོ། །དེ་། ཁྱའོ། དེ་ལ་ཏེ། མ་བ། །::::ད། ད::ཎ་་་་ན་བག་བཤ་ཁྲགས། ཀཚོ། ཎ་ཀྱིལ་བ་བསྡུར་མཆན། །།ག་ལ་ཁྱgy/ལས་འགན་ད། ::ཚོ་བ་མཚན་བྱང་འདི་«པེ་»«སྣར་»ལྟར་བཀོད། མི«པེ་»གེ། «པེ་ཀྱེ། ཚེ«པེ་དྲེག《ਸਮਐਸ«པེ་ཕྱུང་། གདོང་ལ་བུད་པ་དང་། དེ་བ། སྤྱི་རེབསྡུར་འབྲས་རེའུ་མིག་གི་བ་དང་། དེའི་འོད་ལ་བརྒྱུད། （དབ་ས་བ་སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། ཅོ་ནེ། དཎའི་སྔགས། སྔགས། དཔེ་གཞི། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། བབབཆུང་ལྡེབ། ན། བ་། བཙབ་ཕྲེང་། བ。ཚིག་རྐང་། །ཚེག་ཁྱིམ། །
