རྒྱུད། ༼གི༽ཀ་ཎ་ཡ་དང་། པདྨ་དང་། གཞུ་དང་། ཞགས་པ་རྣམས་སོ། །སྟོབས་དང་ལྡན་པའི་གྱད་གཉིས་ཀྱིས་ཞབས་ནས་ཡོངས་སུ་བཏེག་པ། རྒྱན་རྣམ་པ་སྣ་ཚོགས་པ་དང་། གོས་ཀྱི་ཁྱད་པར་སྣ་ཚོགས་པ་འཆང་བ། གཡོན་གྱི་ཕྱོགས་ནས། ནོར་གྱི་དབང་མོ་སེར་མོ་ནོར་བུ་དང༌། འབྲས་ཀྱི་སྙེ་མ་འཛིན་པས་འཁྱུད་པ། འདབ་མ་རྣམས་ལ་རང་རང་གི་མིང་ཡིག་དང་པོ་ལས། གཏེར་གྱི་བདག་པོ་བསྒོམ་པར་བྱ་སྟེ། ཤར་དུ་ནི་གང་བ་བཟང་པོ་སྔོན་པོ། ལྷོར་ནི་ནོར་བུ་བཟང་པོ་སེར་པོའོ། །ནུབ་ཏུ་ནི་ནོར་སྦྱིན་དམར་པོའོ། །བྱང་དུ་ནི་ཛམྦྷ་ལ་ལྗང་ཁུའོ། །ལྷོར་ནི་འཇམ་པོ་འཁྱིལ་བའོ། །ལྷོ་ནུབ་ཏུ་ནི་ཀི་ལི་མཱ་ལིའོ། །ནུབ་བྱང་དུ་ནི་ཞལ་གྱི་དབང་པོའོ། །བྱང་ཤར་དུ་ནི་སྤྱོད་པའི་དབང་པོའོ། །དེ་དག་ནི་ཁ་དོག་སེར་སྐྱའོ། །ཐམས་ཅད་ཀྱང་ཕྱག་བཞི་པ་ནོར་བུ་འཛིན་པའི་མཆོག་སྦྱིན་པ་དང༌། ག་ཏ་དང་། ན་གུ་ལི་དང་། པདྨ་འཛིན་པ། རང་རང་གི་གནོད་སྦྱིན་མོ་ལ་རབ་ཏུ་འཁྱུད་པ། ཕྱི་རོལ་ཏུ་ནི་ཤར་དང་། ལྷོ་དང༌། ནུབ་དང་། བྱང་རྣམས་སུ། ཚོགས་ཀྱི་དབང་ཕྱུག་བཞི་པོ་རྣམས། དཀར་པོ་དང་། སེར་པོ་དང༌། དམར་པོ་དང་། ལྗང་ཁུའི་མདོག་ཅན། ལག་ན་དབྱུག་པ་ཐོགས་པ། སྣང་བ་ཐམས་ཅད་འཇོམས་པར་བྱེད་པའོ། །དེ་ནས་ལུས་ངན་པོ་སྤྱན་དྲངས་པ་བྱ་སྟེ། ན་མོ་བི་ཤ་བ་ཎཱ་ཡ། མ་ཧཱ་ཙཎྜེ་ཤྭ་རཱ་ཡ་ཨ་ཀནཱ་ཡ། ཨ་པ་ར་མི་ཏ། ཙཎྜི་ཤྭ་ར་མ་ཧཱ་ཀ་རུ་ཎི་ཀ། སརྦ་སཏྭཱ་ཧི་ཏ་མ་ན། དྷ་ན་བནྡྷ་བཛྷི་ཤ་སྭཱ་ཡ་མཱ་ཀནྟ་ཡ་སྭཱ་ཧཱ། ཅེས་པའི་སྔགས་ཀྱིས་སོ། །དེ་ནས་ནི་ཡི་གེ་གསུམ་གྱིས་བྱིན་གྱིས་བརླབ་པར་བྱ་སྟེ། དབང་བསྐུར་ནས་མི་བསྐྱོད་པས་དབུ་ལ་བརྒྱན་པར་བྱའོ། །ཨོ་བི་སྭཱ་ཧཱ། ཉེ་བའི་སྙིང་པོའོ། །ཨོ་བི་ཤ་བ་ཎཱ་ཡ་ཧཱུྃ་སྭཱ་ཧཱ། སྙིང་པོའོ། །ཨོཾ་བི་ཤ་བ་ཎ་ཡ་ཚིནྡྷ་ཚིནྡྷ་བྷིནྡྷ་ཧཱུྃ། །རྩ་བའི་སྔགས་སོ། །—།
