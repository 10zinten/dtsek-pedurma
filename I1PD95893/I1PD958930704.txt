འགྲེལ་པ་རྫོགས་སོ། །།ཟག་ལ་ཁབབ། e། ཀ་ངན་པ་ཆ་ཤས་དང་འཆི་བ་ལ་བཀའ་བཏང་པ་དེ་ཀ::་་་་ཏུ༄༅། །ཀ་ལ་ཀ་ལའི་དོ་ཧ་འགྲེལ་པ་དང་བཅས་པ་བཞུགས་སོ། །༄༅། །གུ་རུ་ཀ་ལ་ཀ་ལའི་ཞལ་ནས། ཆོས་དབྱིངས་རྣམ་དག་མཁའ་དཀྱིལ་དུ། །སྐྱེ་མེད་་་འབྲུག་སྒྲ་ལེགས་བསྒྲགས་པས། །སྣང་སྲིད་འཁྲུལ་པའི་ཆོས་རྣམས་ཀུན། །སྐུ་གསུམ་ཡེ་ཤེས་འབྲས་བུས་མཛེས། །ཅེས་སོ། །བློཏཏྟ་ཁ་དག་ཏུ་ཁའདིའི་དཔེ་ནི། དཔག་བསམ་གྱི་ཤིང་ལྟ་བུ་ལ་བཞག་སྟེ། དེ་ཡང་ཤིང་སྣ་ཚོགས་འབྲས་བུས་བསྐྱེད་པར་བྱེད་པ་ནི། །ནམ་མཁའ་ལ་སྤྲིན་དང་། གློག་དང༌། འབྲུག་གི་སྒྲ་ལས་ཆར་འབབ་པས་ན། ཤིང་ཐོག་གང་ལས་སྐྱེ་བ་བསྐྱེད་པར་བྱེད་པ་དེ་བཞིན་དུ། ཆོས་དབྱིངས་རྣམ་དག་མཁའ་དཀྱིལ་དུ། །ཅེས་པ་་ཆོས་ཀྱི་དབྱིངས་མ་བཅོས་པ་ཡེ་ནས་རྣམ་པར་དག་པ་རང་བཞིན་ནམ་མཁའི་དབྱིངས་ལྟ་བུར། ཆོས་ཉིད་སྐྱེ་བ་མེད་པའི་འབྲུག་སྒྲ་བསྒྲགས་པས། ཟག་མེད་ཡེ་ཤེས་ཀྱི་ཆར་ཆེན་པོ་བབ་སྟེ། དེ་ལ་འབྲས་བུ་གང་སྨིན་བྱས་པ་ལ། སྣང་སྲིད་འཁྲུལ་པའི་ཆོས་རྣམས་ཀུན་ཅེས་པའོ། །དེ་ཡང་སྣང་བ་ནི་འཁྲུལ་པའི་སྣང་བ་ཡིན་ལ། སྲིད་པ་ནི་འཁོར་བའི་ཆོས་ཐམས་ཅད་སྲིད་པའོ། །རྣལ་འབྱོར་པ་ཀ་ལ་ཀ་ལའི་རྟོགས་བརྗོད་དང་འགྲེལ་པ་རྫོགས་སོ། །།།་ཏུ་::ཀྱི། ༄༅། །ཀནྟ་ལའི་དོ་ཧ་འགྲེལ་པ་དང་བཅས་པ་བཞུགས་སོ། །༄༅། །གུ་རུ་་ཀན་ལའི་ཞལ་ནས། བླ་མ་དམ་པའི་ཁབ་དག་དང་། །སྙིང་རྗེ་ཡི་ནི་སྐུད་པ་ཡིས། །ཁམས་གསུམ་མ་ལུས་གོས་སུ་བཙེམས། །གོས་ཀྱི་རྣལ་འབྱོར་ངོ་་་་མཚར་ཆེ། །རྒྱ་མཚོ་ལྟ་ཀྱིཆེཚོད་གང་།
