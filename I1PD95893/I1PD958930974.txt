རུ་ཀཾ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཌཱ་ཀི་ནི་ཛཱ་ལ་སམྦ་ར་སྭཱ་ཧཱ། ཨོཾ་ཧྲིཧ་ཧ་ཧཱུྃ་ཧཱུྃ་ཕཊ། བཅོམ་ལྡན་འདས་ཀྱི་སྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོའོ། །ༀ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ། བཛྲ་བརྞ་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། དེ་་ནས་གཏོར་མ་འབུལ་བར་བྱ་སྟེ། མདུན་དུ་ཡཾ་དང་རཾ་ལས་བྱུང་བའི་རླུང་དང་་་་མེའི་དཀྱིལ་འཁོར་ལ། ཡི་གེ་ཨ་ཡོངས་སུ་གྱུར་པ་ལས། པདྨའི་སྣོད་ཀྱི་ནང་དུ་བདུད་རྩི་ལྔ་དང་། སྒྲོན་མ་ལྔ་རྣམས། ཨོཾ་ཏཱཾ་ཛྲ་ཁཾ་ཧཱུཾ། ཐཾ་ལཾ་མཱི་པཾ་ཏྭཾ་གྱི་ཡི་གེ་རྣམས་ལས་སྐྱེད་ཅིང་། དེ་དག་ཉིད་ཀྱིས་མཚན་མ་བལྟས་ལ། །ཡོངས་སུ་་་གདུངས་པའི་ཞུ་བའི་རྫས་ཀྱི་སྟེང་དུ་ཧཱུྃ་དཀར་པོ་ལས་བྱུང་བའི་ཁ་ཊཾ་ག་དཀར་པོ་ཞུ་བས་རོ་གཅིག་ཏུ་གྱུར་པ་དང་། ཨཱ་ལི་ཀཱ་ལི་ཡོངས་སུ་གྱུར་པ་ལས་བྱུང་བའི་ཡི་་གེ་གསུམ་གྱིས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཡེ་ཤེས་ཀྱི་བདུད་རྩི་བཀུག་སྟེ་ཡི་གེ་གསུམ་ལ་ཞུགས་པ་དང་། །འ་དེ་དག་ཀྱང་རིམ་གྱིས་འོག་ནས་འོག་ཏུ་དམ་ཚིག་གི་བདུད་རྩི་དང་གཉིས་སུ་མེད་པར་བསམ་ནས། ཇི་སྲིད་འདོད་པར་ཡི་གེ་གསུམ་གྱིས་བྱིན་གྱིས་བརླབ་བོ། །དེ་ནས་བཅོམ་ལྡན་འདས་དང་། །ལབཅོམ་ལྡན་་་འདས་མ་ལ། ཨོཾ་ཨ་རལླི་ཧོཛཧཱུྃ་བཾ་ཧོ། བཛྲ་ཌཱ་ཀཱི་ནི་ས་མ་ཡ་སཱུ་དྲི་བྱ་ཧོཞེས་བྱ་བས་བསྟབས་ཏེ། །ཧཱུྃ་ལས་བྱུང་བའི་རྡོ་རྗེ་དཀར་པོའི་ལྕེས་དྲངས་ཏེ་་གསོལ་བས་ཚིམས་པར་གྱུར་པར་བལྟའོ། །དེ་ནས་དུར་ཁྲོད་ན་གནས་པ་རྣམས་ལ༑་ཤར་ལ་སོགས་པའི་ཕྱོགས་སུ་གཡོན་སྐོར་གྱིས། ཚོ། ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་ཡཀྵ་རཀྵ་ས། བྷུ་ཏ་པྲ་ཏ། པི་ཤཱ་ཙ་ཨུཎཱ་ད། ཨ་པསྨ་ར་ཌཱ་ཀ་ཌཱ་ཀི་ནྱ་ད་ཡ། ཨི་མཾ། བཱ་ལི་གྲིཉྩནྟུ། ས་མ་ཡ་རཀྵནྟུ། མ་མ་སརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪནྟུ། ཡ་ཐི་པཾ་ཡ་ཐིཊྚཾ། བྷུཉྫ་ཐ། པི་པ་ཐ། ཛི་གྷཱ་ཐ། ་མ་ཏི་ཀྲ་མ་ཐ། མ་མ་སརྦ། ཀཱརྟ་ཡ་སཏ་སུ་ཁཾ། བི་ཤུད་དྷ་ཡེ། ས་ཧཱ་ཡི་ག་བྷ་ཝཱནྟུ་ཧུཾ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཞེས་།
