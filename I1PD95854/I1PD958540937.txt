མཁྱེན་དེ་ཀའོ། སྒོ་། ཝ་ཁ་ཁ་ཚོ་བན་རྟོཎ་། ཀ་བ་ཀan་་བ་ཚེ་ལུགས་པ་དག། ཁ་བ་ཁྲ་འཁྲེང་ཏུ་བློ་ལ་ང་ཚོ་ཚན་aཎྜིད་པ་ཁྱོད་ལྡཁཆ། མྱཁྱིལ:::་་་བསྡུར་འབྲས་རེའུ་མིག་ཏུ་ཁ་བ་བཟླ། བླཐིའིརྒྱུད། (རྒྱུད། །རྒྱུད། རྒྱུད། །དཔེ་གཞི། །སྡེ་དགེ། །དུས་པེ་ཅིན། སྣར་ཐང༌། སྡེ་ཚན། །རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། ནན་བན་། ན་བ?ནན་། ཕྲེང་། །ཚིག་རྐང་། ཚན་པ་ལ་ཆུ་ལེ་ཁགྲོ་ཏི་ཏོཀ་ཊི་ཀབཀྵ་ཀའི་ཀཏུ་ཁཆཆཚེག་ཁྱིམ། ང༌ཡི་ཊཀྱིཀྱིལཀཨཏོ། 《《་《། ཌོ་ལ《་ཀྱི»བར་བློ་ཁ་བྲིད་པ་དང་BaSàiG། །འཚོ་བ་དང་གན་རྒྱ་རས་ལ་གལ་གནད་ཁག་ག་བ་བླ་iaHPhạcKhoa་་ར་བྱའོ། །དམ་ཚིག་གི་ཕྱག་རྒྱ་ནི་་་་་་་་ཚོན་ན་བོད་ལེན་པ་ལ་དབྱེ་ན་ག་དུ་བསྟན་པས་འདིར་དེ་ལ་ལོབས་པ་ན་དེ་བ་མ་བཏོ། དེ་ནས་སུམ་ཅུ་རྩ་བརྒྱུད་པས་ཤི་ཡང་རྣལ་འབྱོར་པ་ཉིད་ས་ལ་དོགས་པ་དེ་དོན་ཁྲོམ་ས་པར་བྱ་བའི་ཕྱིར་སྤྲོས་མེད་དང་། ཤིན་ཏུ་ད་པ་རྒྱས་པར་བསམ་ཏོ། །དེ་ལ་རྣལ་འབྱོར་མ་བ་ཀུམ་གྱི་ཐུན་མོང་པའི་འདབ་ཀྱི་མཚན་ཉིད་ལུང་ཆུ་ཆུ་དགུ་པ་ནས་བཞི་བཅུ་རུ་པ། ཉིས་བར་ད་ལ་མདོ་དང་རྒྱས་པ་བསྟན་པའི་སྒོ་ནས་ཁྱེའུ་ལྔ་བཅུ་།
