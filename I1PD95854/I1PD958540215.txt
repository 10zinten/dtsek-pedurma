བསྟན་འགྱུར། རྒྱུད། མ་༼ཀྱི་བསལ་བར་བྱ་ཞིང་རང་ཉིད་ཀྱིས་བཟའ་བ་དང་བཅའ་བ་ལ་སོགས་པ་ཡང་སྤྱད་པར་བྱའོ་ཞེས་བྱ་བ་ནི་མན་ངག་གོ། །གཞན་དག་ནི་གོ་སླའོ། །།འཁོར་ལོ་འདི་ཉིད་ལག་པའི་མཐིལ་དུ་བསམས་ལ་བསྒྲུབ་པར་བྱ་བའི་མགོ་ལ་བཞག་ན་ཡང་སྔ་མ་བཞིན་དུ་འགྱུར་རོ། །ལེའུ་བཞི་བཅུ་གསུམ་པའོ། །།ཁོ་དང་། ས་བདག་བ་དན་ཁྱཚོ་ལྷོ་ལུ་བ་དང་། ་་་་་་་་:ཚུ་ཁོ་་བཀག་པའི་ད་བ་མ་འ་ར་བ་ན་ལ་ཏུ། །ཆཟླ་དྲི་བ་ཀུན་འདུ་བའི་ས་་ལ། གཙང་མའི་འོ་མ་ཞེས་བྱ་བ་ནི་བའི་འོ་མའོ། ཝཱia་བྱ་ལ་ལེའུ་བཞི་བཅུ་བཞི་པའོ། །།།་་། པལ། །ཞ་ཟགས་ཁུལ་གྱི་ཏོག་ཁའི་ཚིག་ཎའི་ཀཏུ་ལྟ་ཊི་ཀངའི་་ན་ཏ་ཁ་ཁ་ཁ་ཚཀའི་བྱབོད་ཀྱི་དང་ག་ལ་ཚ་བ་བྱུང་གསུངས་དང་ཁ་ན་ཊ་ཀ་ཡིན་པ་དེ་དག་ཁ་ནི་རྟ་གསོད་པ་ནི་ཀ་ར་བཱི་རའོ། ་། སྐྱེ་གནས་རྒྱ་བཅིངས་པ་ཞེས་བྱ་བ་ནི་ཕྱིའི་ལྷ་མོ་དང་ཁ་སྦྱོར་གྱི་སྦྱོར་བ་བྱ་བའོ། །སྤྲ་ཚིལ་གྱིས་ཀྱང་རབ་ཏུ་བྱུག་ཡ། ཅེས་བྱ་བ་ནི། འདིར་སྤྲ་ཚིལ་ལས་གཟུགས་བརྙན་བྱས་ལ། ་སྐུ་ཙནྡན་དམར་པོའི་ཕྱེ་མ་དང་། །ཚ་བ་གསུམ་གྱི་ལྡེ་གུས་བྱུག་ལྟོ། །ཅེས་བྱ་བ་སྦྱར་རོ། །ར་ཁ། ལེའུ་བཞི་བཅུ་ལྔ་པའོ། །ཁོ། །སྤྱི་ཁྱབ་འགྲོ་ཀྲེཊ་ནི་ཏེ་ཐག་པས་དགོ་པ་ཅིགང་འི་སྤྲུལ་བ་བཏོདུ་བྱུང་བའི་ཆ་ལ་ན། མཁ་ལ་ལ་ནི་ཁ་བ་མར་ཁཎྜབར་ཤ་ཁྱོ་ག་ཟེར་བའི་དེ་ཡི་གེ་ཧ་ནི་ལྔ་པོ་དང་། །ལག་པ་ཀུན་དུ་མཉེས་ནས་ནི། །ཞེས་བྱ་བ་ནི་གང་ཞིག་གི་རྐང་རྗེས་ཀྱི་རྡུལ་ཁྲག་དང༌། །དུག་དང་། ལསྐེ་ཚེས་ལག་པ་གཉིས་ཀྱི་མཐིལ་བསྐུས་ལ། ::དེའི་སྟེང་དུ་བ་བླའི་སྣག་ཚ་དང་བྱ་རོག་གི་སྒྲོའི་སྨྱུ་གུས་ཡི་གེ་ལྔ། ཧེ་:གྷ། །ཧོ་ཧོ་བྲིས་ལ་བསམ་གཏན་གྱིས་ཀྱང་དེ་ལྟར་བསམས་ལ་སེང་ལྡེང་གི་མེ་མདག་གམ། རོ་བསྲེགས་པའི་མེ་མདག་ལ་ལག་པ་གཉིས་ཉེད་ཅིང་སྦྱིན་སྲེག་བྱས་ན་དགྲ་བོའི་ཁ་ནས་ཁྲག་འཛག་ཅིང་མོད་ལ་འཆི་བར་འགྱུར་རོ། །ཐོད་པ་ལ་་་་སོགས་པ་ལ་ཡང་སྦྱོར་བ་དེ་བཞིན་དུ་བྱའོ། །:བྱི་ལ་ལ་སོགས་པའི་ཤ་ཁ་ན་མ་ཐོ་བ་ལས་བ།
