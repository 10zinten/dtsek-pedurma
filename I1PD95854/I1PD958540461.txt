བསྟན་འགྱུར། ་རྒྱུད། a། ཚ་མ་ཧཱ་:དྷཱམྦ་ཨནྡྷ་ཀཱ་ར་ཝ་པུ་ཥཱ་ཡ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་ཀ་པཱ་ལི། མ་ཧཱ་ཀཾ་ཀ་ལ་བི་ཀ་ཊ་དཾབྱཱཊ་སུ་རཱ་་བི་རི་ཎ་ཨ་མི་ཏཱ་བྷཱ་་བཛྲ་པྲ་བྷ་བཛྲ་དེ་ཧ་ཨཾ་ཀུ་རི་ཀ་བཛྲ་ཛ་་ཊི་མ་ཧཱ་བཱིར་བཛྲ་ཧཱུྃ་ཀཱ་ར་སུ་བྷ་དྲ་མ་ཧཱ་བྷི་ར་བ་བི་:རུ་པཱཀྵ་མ་ཧཱ་བ་ལ་རཏྣ་་་་བཛྲ་:ཧ་ཡ་གྲྭ་ཝ་ཨཱ་ཀཱ་ཤ་གཛྷ་ཤྲཱི་བཛྲ་ཧེ་རུ་ཀ་པདྨ་ནརྟེ་ཤྭ་ར་བི་རོ་ཙ་ན་བཛྲ་སཏྭ་ཧཱུྃ་ཧཱུྃ་ཕཊ། དཔའ་བོའི་སྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོ་བསྡུས་པའོ། ་ར་། ཨོ་ཀ་ར་ཀ་ར་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་པྲ་ཙཎྜ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཀུ་རུ་ཀུ་རུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཙཎྜ་ཨཀྵི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་བནྡྷ་བནྡྷ་ཧུ་ཧཱུྃ་ཕཊ། ༀ་པྲ་བྷཱ་བ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཏཱ་ས་ཡ་ཏཱ་ས་ཡ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་མ་ཧཱ་ནཱ་སེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་གློ་བྷ་ཡ་ཀྵ་བྷ་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ལ་ཨོ་བཱི་ཎིརམ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཁརྦ་རཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཧྲཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ལཾ་ཀི་ཤཱ་རཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོ་ཕོ་ཕོ་ཧུ་ཧཱུྃ་ཕཊ། །ཨོ་དུ་མཙྪཱ་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཕཊ་ཕཊ་ཧུཾ་ཧཱུྃ་ཕཊ། པ་ཨོ་ཨེ་རཱ་པ་ཏཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། །ཨོ་ད་ཧ་ད་ཧ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་མ་ཧཱ་བྷི་ར་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་པ་ཙ་པ་ཙ་ཧུ་ཧཱུྃ་ཕཊ༑ཨོ་བཱ་ཡུ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། པ་ཨོ་བྷཀྵ་བྷཀྵ་བ་ས་རུ་:དྷི་ར་ཨནྜ་མཱ་ལ་:ཨ་བ་ལམྦི་ནི་ཧུཾ་ཧཱུྃ་ཕཊ། ཁ་ཨོ་སུ་རཱ་ཙྪིབྷཀཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། འ་ཨོ་:གྲིནྣ་གྲིནྣ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་ཉྫཱབྷུ་ཛཾ་གཾ་སརྦ་ཝཱ་ཏེརྫ་ཡ་ཏིརྫ་ཡ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་བྱཱ་མ་དེ་བཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཨཱ་ཀཥྨ་ཨཱ་ཀཟླུ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་སུ་:བྷ་དྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲི་ཧྲི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧ་ཡ་ཀནྡྷེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཛྷ་ཛྷ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་ཁ་གཱ་ན་ན་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཧྥུ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ༑ཨོཾ་ཙཀྲ་པེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་རོ་ཧི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཧི་ཧི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤོཎྜི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཙཀྲ་ཝརྨི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཀི་ལི་ཀི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སུ་བཱི་རེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་སི་ལི་སི་ལི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་མ་ཧཱ་བ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་:ཧི་ལི་ཧི་ལི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཙཀྲ་ཝརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ___།
