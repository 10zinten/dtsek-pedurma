བ་ཏི་སྐྱ་བསྟན་འགྱུར། རྒྱུད། Jའི་ཀ་ཁཏུ་ཀ་ཆུ་མཆོག་ཡེ་ཤེས་སེམས་དཔའ་གཞུག་པ་དང་། གོ་ཆ་བསྐོན་པ་དང་། ལོདབང་་་་བསྐུར་བ་དང་། ཚོ། བསྒོམ་པའི་ཚེ་དེའི་ཐུགས་ཀར་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་དབུས་སུ་ཕྱིར་བཟློག་ཆེན་མོའི་འཁོར་ལོ་བཀོད་པ་ཤིན་ཏུ་དད་པས་མཉམ་པར་བཞག་?ནས་་་བསྒོམ་མོ། །དེ་ནས་འཆི་བ་འཚོ་བའི་སྔགས་བཟླས་པར་བྱ་སྟེ། སྔགས་ནི་རྒྱུད་ན་གསལ་ལོ། །བསྙེན་པ་བྱེ་བ་བྱས་ལ་ལས་ལ་སྦྱོར་བའི་ཚེ། ཁྲི་བཟླས་ན་འཆི་བ་ཐམས་ཅད་འཚོ་སྟེ། ལ་ཁ་དུས་མ་ཡིན་པར་འཆི་བ་ལས་ངེས་པར་རྒྱལ་ལོ། །༄༅། །།དེ་ནས་རྣལ་འབྱོར་མ་སུམ་ཅུ་རྩ་བདུན་དགྱེས་ནས་འཆི་བ་བླུ་བའི་སྔགས་གསུངས་པ་ནི། འཆི་བདག་གི་བདུད་ཀྱི་གཟུགས་བརྙན་བྱས་ལ། དེའི་སྙིང་གར་གསོད་པའི་་་་འཁྲུལ་འཁོར་བཅུག་ནས་མཁའ་འགྲོ་མ་སྙིང་གནས་སྤྲུལ་ནས་ཧི་ཧི་འདོན་པ་དང་། ལཱ་མ་སྤྲུལ་ནས་ཧ་ཧ་འདོན་པ་དང་། ཡང་དུམ་སྐྱེས་མས་ཧོ་ཧོ་ཞེས་བརྗོད་པ་་དང་། གཟུགས་ཅན་མས་ཧེ་ཧེ་ཞེས་བརྗོད་པ་དང་། །རབ་གཏུམ་མས་ཧཱུཾ་ཧཱུཾ་ཞེས་བརྗོད་པ་དང་། གཏུམ་མིག་མས་ཧཱུྃ་ཧཱུྃ་ཞེས་བརྗོད་པ་དང་། འོད་ལྡན་མས་ཕཊ་ཕཊ་ཅེས་བརྗོད་པ་དང་། ཞི་སྣ་ཆེན་མས་ཧཱུྃ་ཧཱུྃ་ཞེས་བརྗོད་པ་དང་། དཔའ་་་་བོའི་བློ་མས་ཛ་ཛ་ཞེས་བརྗོད་པ་དང་། མིའུ་ཐུང་མས་ས་ས་ཞེས་བརྗོད་པ་དང་། །ལང་ཀའི་དབང་ཕྱུག་མས་སཱ་བྷུཞེས་བརྗོད་པ་དང་། །ཤིང་གྲིབ་མས་ཧཱུྃ་བོ་ཞེས་་བརྗོད་པ་དང་། །ས་སྲུང་མས་ཧཱུྃ་བུ་བཞི་དང་། ལཧཱུྃ་གསུམ་བརྗོད་པ་དང་། ལ་དཔའ་མོ་ཆེན་མོས་ཧ་ཧ་ཞེས་བརྗོད་པ་དང་། ཚོ་རླུང་ཤུགས་མས་ཧཱུཾ་ཧཱུཾ་བརྗོད་པ་་་དང་། ལྷ་དགྲ་མས་ཀྲྀ་ཏྲ་ཞེས་བརྗོད་པ་དང་། །སྔོ་བསངས་མས་ཧུཾ་ཧཱུྃ་ཞེས་་་བརྗོད་པ་དང་། རབ་བཟང་མས་སཾ་སཾ་ཞེས་བརྗོད་པ་དང་། ཁཱ་གདོང་མས་ལཾ་ལོ་ཞེས་བརྗོད་པ་དང་། འཁོར་ལོའི་ཤུགས་ཅན་མས་དྷི་དྷི་ཞེས་བརྗོད་པ་དང་། ཁ་དུམ་སྐྱེས་མས་:ཛི་ཛི་ཞེས་བརྗོད་པ་དང་། གླང་སྣ་མས་ཧཱུྃ་ཧཱུྃ་ཞེས་བརྗོད་པ་དང་། །
