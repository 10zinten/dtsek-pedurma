སོ་སོར་བྱས་པས་འཁྲུལ་འཁོར་སྔགས་དང་ལྡན་པས་འགྲུབ་ལ། དེ་ལ་ཁྱད་པར་དུ་རྩ་བའི་རིག་པ་ནི་འདི་ཡིན་ཏེ། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ། བཛྲ་བཱ་རཱ་ཧི་བཾ། ཨཱཅུ་པ་རཱ་ཛི་ཏེ། ཏྲི་ལོ་ཀྱ་མཱ་ཏེ་མ་ཧཱ་བི་དྱེ་ཤྭ་རཱི། སརྦ་བྷཱུ་ཏ་བྷ་ཡཱ་བ་ཧེ་མ་ཧཱ་བཛེ། །བཛྲ་ཨཱ་ས་ནེ། ཨ་ཛི་ཏེ། ཨ་པ་རཱ་ཛི་ཏེ། བ་:དྷཱ་རི་ཎི་ནེ་ཏྲ་བྷ་མ་ཎེ། ཤོ་ཥ་ཎི། རོ་ཥ་ཎི་ཀྲོ་དྷ་ཀ་རཱ་ལི་ནི། སཾ་ཏྲ་ས་ནི། མ་ར་ཎི་སུ་པྲ་བྷེ་ད་ནི། །:པ་ར་ཛ་ཡེ་བི་ཛ་ཡ། ཛམྦྷ་ནི། སྟམྦྷ་ནི། མོ་ཧ་ནི། བཛྲ་བཱ་རཱ་ཧི། མ་ཧཱ་ཡོ་གི་ནི། ཧེ་ཀ་མེ་ཤཱ་:རཱི། ཁ་གེ། ཏདྱ་ཐཱ། ཨོཾ་ཤྲཱི་པྲ་ཏངྒེ། པྲ་ཏངྒེ། ཧ་ན་ཧ་ན་པྲཱ་ཎཱཾ། ཀིཾ་ཀི་ནི། ཁེ་ཁི་ནི། དྷུ་ན་དྷུ་ན། བཛྲ་ཧ་སྟེ། ཤོ་ཥ་ཡ༑ཤོ་ཥ་ཡ། བཛཱ་ཀ་པཱ་ལ། ཁ་ཊཾ་ག་:དྷ་རི་ཎེ་མ་ཧཱ་བི་:ཤི་ཏོ། །ཚྭ་ས་ར་ཤ་ནི་མ་ནུ་:ཉྩནྟ་པྲ་བྲྀཏྟི། སནྣ་དྷ་ན་ར་ཤི་རོ་མཱ་:ལ་གྲེནྡྲི་ཏ་དྷ་ར་ཎི་སུམྦྷ་ནི་སུསྙེ༄༅། ཧ་ན་ཧ་ན་པཱ་ཏཱཾ། སརྦ་པཱ་པཾ་ས་ཏྭཱ་ནཾ། སརྦ་པ་ཤུ་ནཱི། མ་ཧཱ་མཱ་སཙྪེ་ད་ནི། སྨྲ་ཏཱི་ཀྲོ་དྷ་མུརྟེ། དཾཥྛ་ཨིཀ་རཱ་ལི་ནི། མ་ཧཱ་མུ་དྲེ། ཤྲཱི་ཧེ་རུ་ཀ་དེ་བ་སྱ༑ཨ་གྲ་མ་ཧི་ཥི། ས་ཧ་སྲ་ཤི་རེ། ས་ཧ་སྲ་བཱ་ཧ་བེ། ཤ་ཏ་ས་ཧ་ས། ཨཱ་ན་ནེ་ཛཱ་ལི་ཏ་ཏེ་ཛ་སེ། ཛྭ་ལ་མུ་ཁེ། པཾ་ག་ལ་ལོ་ཙ་ནེ། བཛྲ་ཨཱ་ས་ནེ། བཛྲ་ཤ་རཱི་རེ། མི་ལི་ཏི། ཙི་ལི་ཏ། ཧེ་ཧེ་:ཧཧེ། ཧཱུཾ་ཧཱུཾ། ཁ་ཁ་དྷ་དྷུ་རུ་རུ་དྷུ་རུ་དྷུ་རུ། མུ་རུ་མུ་རུ། མུཉྫ་མུཉྩ། ཨ་དྭེ་ཏེ། ཨ་དྷ་རི་ཎི་ཤེ་བྷ་མ་ཎ། མ་ཧཱ་ཡོ་གི་ནི་པ་ཐི་ཏ་:སིདྷེ། དྲོ་དྷཾ་དྲོ་དྷཾ། ཧེ་གྲོ་གྲོ་ཧེ་ཧེ་ཧཧ། བྷི་མེ། ཧ་ས་ཧ་ས་བཱི་རེ། ཧཱ་ཧཱ་ཧོཧོ། ཧཱུཾ་ཧཱུཾ། ཏྲི་ལོ་ཀྱ་བི་ནཱ་ཤ་ནི། ཤ་ཏ་ས་ཧ་སྲ་ཀོ་ཊི། ཏ་ཐཱ་ག་ཏ་པ་རི་ཝཱ་རེ་ཧཱུྃ་ཧཱུཾ་ཕཊ། སིཾ་ཧ་རུ་པེ་ཁག་ཛ་རཱུ་བེ་ག། ཏྲི་:ལོ་ཀྱོ་ད་་རེ་ས་མུ་དྲ་མེ་ཁ་ལེ། གྲ་ས་གྲ་ས་ཧཱུཾ་ཧཱུཾ་ཕཊ། བཱི་རཱ་:དྭེ་ཏེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། །:ཧཧམ་ཧཱ་པ་ཤུ་མོ་ཧ་ནི། མ་ཧཱ་ཡོ་གི་ཤྭ་རཱི་ཏཱཾ། བཛྲ་ཌཱ་ཀི་ནཱི། སརྦ་ལོ་ཀཱ་—།
