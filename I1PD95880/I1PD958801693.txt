བསྟན་འགྱུར། རྒྱུད། ༼སྔགས་ལས་འབྱུང་བ་རེ་རེའོ། །དོན་རྣམས་ཐམས་ཅད་གྲུབ་པར་བྱེད། །ཅེས་པ་ཡི་གེ་ཧེ་ལ་བྱ་སྟེ་དེ་ཡང་སྔགས་ཀྱི་རང་བཞིན་ལས་བྱུང་བའི་ཡི་གེ་འབྲུ་གཉིས་་་པའི་ཟུང་རེ་རེའོ། །འཁོར་ལོའི་རྩིབས་བཅུ་གཉིས་པོའི་གནས་ཀྱི་མིག་ཏུ་བྲིས་པ་དེ་ནི། ལས་འདི་དག་ཐམས་ཅད་འགྲུབ་པར་འགྱུར་རོ། །བཅུ་གཉིས་དག་ཏུ་གསུངས་པའི་སྔགས། ཞེས་པ་ཡི་གེ་ཧེ་ཧེ་དྲུག་ལ་བཅུ་གཉིས་ཟུང་། །།ਰਿਹਾਕਰਨਨੂੰਤਿੰਨਸਾਲਾਂਵਾਲੀਆཁྱོ་ག་པ་་་་་་་་་་་་་ལ་་་་་་་དད་པས་མ་དེ་ལ་བྱ་བ་ཁབསྡུར་མཆན། ས་ཀུན་པསཔོ། ཁྱོད་ཀྱི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»། །ཁྲོ་བོ་འཁོརྣམ་པར་རྒྱལ་བའི་རྟོག་པ་གསང་བའི་རྒྱུད་ཀྱི་རྒྱ་ཆེར་བཤད་པ་བཞུགས་སོ། །«པེ་»«སྣར་གླུ་བྱ་ཏནྟ། །«པེ་《སྣར་ཝན། «པེ་《སྣར་»གློ། «པེ་»«སྣར་»ཝན། («པེ་»«སྣར་»བློ། འདུ་«པེ་»«སྣར་»བི། «པེ་»སྣར་རིས། །«པེ་»«སྣར་»སྐུལ། «པེ་»སྣར་པ་དེ། ལོ་དེ་ལ། །ང་ལས་འབྱུང་བ་ལས་ནམ་«པེ་»སྣར་»མའི། «པེ་»«སྣར་»མང་། «པེ་»«སྣར་ཝན། «པེ་»«སྣར་»གློ། །«པེ་»«སྣར་པའི། «པེ་»«སྣར་»ལན། «པེ་»«སྣར་པའི། «པེ་»«སྣར་»ནས། («པེ་»གིས། «པེ་»«སྣར་གྱིས། «པེ་»གིས། «པེ་»《སྣར་པའི། «པེ་»རྗེ། པ་ད@«པེ་»«སྣར་དགའ་ཟློག་ཅེས་ཀྱང་བྱ། །ན་ནས་པ༌དབང་དང་དངོས་ལ་ལ་ད་(ི«པེ་»«སྣར་རྗེའི། «པེ་《སྣར་»ཤུ། «པེ་»«སྣར་པ། «པེ་»སྣར་»—ཅིང་། སུ«པེ་»«སྣར་རྗེའི། «པེ་»«སྣར་»ཅིང་། །«པེ་»«སྣར་བས། 《«པེ་《སྣར་འཁྲུག་«པེ་»ཀྱིས། «པེ་《སྣར་—དེ། སོ«པེ་»《སྣར་གསོལ། པོ«པེ་»གཞན་ནི་མཇལ། 《སྣར་བཞན་ནི་མཇལ། «པེ་»«སྣར་པ། ། ེ«པེ་»«སྣར་»ཏོ། («སྣར་མེ། «པེ་»«སྣར་མཇལ་བར་བས། «སྣར་»སྐྲག། «པེ་»«སྣར་རྩོལ། @«པེ་»«སྣར་—ལྷ། @«པེ་»«སྣར་བཟུང་། 《སྣར་»བའི། «པེ་»«སྣར་»གི༑«པེ་»«སྣར་»བོད། (《སྣར་ཟད། འགྱུར་བ། ཟླ་དང་པ་འབད«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»ཞུ་བའི། «པེ་»«སྣར་ཟད། «པེ་»«སྣར་བསྲུང་། འ«པེ་《སྣར་—ཡང་། «པེ་»«སྣར་དོང་ཏེ་བྱང་ཆུབ། «སྣར་»སན་ཅན། («པེ་»།
