ལ་འཚོ་རབསྟན་འགྱུར། རྒྱུད། ༼ཏུ?ཉིཀཀྱི་འཆང་གིས་དེ་ལྟར་ནི། །མཆོད་པ་དེ་ལྟར་ལེགས་པར་བྱ། ་། སྣོད་ནི་གསེར་་་་སོགས་རིན་ཆེན་དུ། །ཚོན་རྩི་ལེགས་པར་བསྡུ་བར་བྱས། །གཙང་མའི་ཆུ་བོ་ལ་སྩོགས་བོར། །ཕྱི་ནས་ཀྱང་ནི་རྣམ་ལྔས་བྱུག། ཚུལ་ཆེན་རྒྱུད་ལས་གསུངས་པ་ཡི༑:། དཀྱིལ་འཁོར་འཇུག་པའི་ཆོ་ག་བྱས། །ཅུང་ཟད་དགེ་བ་གང་ཐོབ་་་་འདིས༑། འཇམ་དཔལ་ཁྲོ་བོའི་གནས་ཐོབ་ཤོག། །འཕགས་པ་འཇམ་དཔལ་གསང་བའི་རྒྱུད་ཀྱི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག། སློབ་དཔོན་དོན་ཡོད་པའི་ཞབས་་་་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།་་མལ་པ་ཁའི་ཡི། མི་ལ་འཁོད་པ། དེ་ཚོང་ཚོ་བྱ་རྗོད་བྱེད་ད་པ་བ་དང་བདག་དེས་::ཀ་ཆེ་ཁག་ལ་་་་་་་ཁྱབ་ཀུང་སྐྱེད་ཚེ། ནད། པ་ལ་བདུད་དམག་འབརབ་བསྡུར་མཆན། ཚེ་དའི་ཚོ་བ་དེ་གང་ན་གནས་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །།འཕགས་པ་འཇམ་དཔལ་གསང་བའི་རྒྱུད་ཀྱི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་བཞུགས་སོ། །།«པེ་»«སྣར་»—། །«པེ་»«སྣར་པའི། ®«པེ་《སྣར་རྙིང༌། «པེ་»«སྣར་»འམ། དེ། །«པེ་»«སྣར་»ས། «པེ་»«སྣར་བར། །ཡང་ན། «པེ་»བརྟགས། «པེ་»«སྣར་»སྐྱོབས། ས«པེ་»«སྣར་ལ། «པེ་»«སྣར་བྱ། «པེ་»«སྣར་རིགས། «པེ་»«སྣར་མོ། «པེ་»བ་ཀུན་ཀྱི《སྣར་རིགས་འཛིན་ཅན། གླིང་བ། ཚོའི་ང་ཚོ། («སྣར་»གཅིགའ«པེ་»«སྣར་»ན། ན«པེ་»«སྣར་»ག«པེ་»«སྣར་»སོགས། ས《སྣར་འབྱུང་། «པེ་»«སྣར་»ཏུ་སྤྱངས་པ་བརྩལ། @«པེ་»«སྣར་»སྡུག། ཏེ«པེ་»«སྣར་»གདུག«པེ་ཆོག«སྣར་»ཡིས། «པེ་»«སྣར་»སོགས། (འདིའི་འོག་གི་སྩོགས་རྣམས་«པེ་«སྣར་ལ་སོགས་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། ཀྱི«པེ་»«སྣར་ལྷུང་། །《པེ་»«སྣར་བཞག«པེ་»«སྣར་དྲང༌། «པེ་»«སྣར་བཟུང་། སོ«པེ་»«སྣར་བརྒྱན། ཚ«པེ་»«སྣར་»གི། (འ«པེ་»«སྣར་བསྙུང་བ། («པེ་»«སྣར་»བརླབ། «པེ་»«སྣར་»སྐྱབ། ས«པེ་《སྣར་གསང་། «པེ་»གྱི། @«པེ་»«སྣར་»འདིས། ཏུ«པེ་»«སྣར་དཔོགས་ཏེ། «པེ་»«སྣར་ཀྱི། («པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་གྱིས། །།
