སངས་རྒྱས་རྣམས་ཀྱི་ཆོས་ཀྱི་དབྱིངས། །རྣམ་དག་འདི་ནི་ཡང་དག་བརྗོད། །ཡོངས་སུ་དག་པའི་མཚན་ཉིད་ཇི་སྐད་བཤད་པས་སངས་རྒྱས་རྣམས་ཀྱི་ཆོས་ཀྱི་དབྱིངས་སུ་ཡོངས་སུ་དག་པ་གང་ཡིན་པ་འདི་ནི་ཡང་དག་པར་བརྗོད་ཟིན་པར་ཚིགས་སུ་བཅད་པ་བཞིས་ནི་གོ་རིམས་བཞིན་དུ་རང་བཞིན་གྱི་མཚན་ཉིད་དང་རྒྱུ་དག་དང་ཕྲིན་ལས་དང་ལྡན་པ་དང་འཇུག་པའི་དོན་གྱི་ས་སངས་རྒྱས་ཀྱི་སའི་་་རྣམ་པར་དབྱེ་བ་དེ་མདོར་བསྡུས་ཏེ་སྟོན་པར་བྱེད་དོ། །མདོ་སྡེ་རྣམ་པར་བཤད་བྱས་ནས། །བསོད་ནམས་བདག་གིས་གང་བསྐྱེད་པ། །དེས་ནི་འཇིག་རྟེན་སངས་རྒྱས་སི། །དམ་པ་མཆོག་ནི་ཐོབ་པར་ཤོག། སློབ་དཔོན་ངང་ཚུལ་བཟང་པོས་མཛད་པ་རྫོགས་སོ། །།།ར། ས་དང་ལྔ་ཀྱིདབང་་་་མ་་ག་::སས། ::::བསྡུར་མཆན། ་་་་་་བསྐྱོད་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་འཕགས་པ་ས་ད་སངས་རྒྱས་ཀྱི་སའི་རྣམ་བཤད་ངང་ཚུལ་བཟང་པོས་མཛད་པ་བཞུགས་སོ། །ཞེས་འཁོད། «པེ་»«སྣར་། །«པེ་»«སྣར་བསྐྱེད། «པེ་རྣམ། «པེ་»«སྣར་»བསྒྲུབ། «པེ་»«སྣར་»ལོ། @«པེ་»«སྣར་+ན། ཚེ«པེ་»«སྣར་པའི། «པེ་»«སྣར་»གཅིགཚེ«པེ་»::ཀྱིང«སྣར་སྐ། །«པེ་»«སྣར་པ་སྟེ། དེ་ན་་་་་«པེ་»«སྣར་»ལྟ། «པེ་»«སྣར་པས། ཚེ«པེ་»«སྣར་»ཚོམ་དུ་འགྱུར་བས། ཁོ«པེ་»ང«སྣར་འབོག་«པེ་»«སྣར་»ལྟ། །O«པེ་»«སྣར་»གི། ཉི«པེ་»«སྣར་ཤེས། མི«པེ་»«སྣར་པོ། «པེ་»«སྣར་པ། །«པེ་»«སྣར་ཉིད། @«པེ་»«སྣར་སུ། «པེ་»«སྣར་ཏེ། «པེ་»«སྣར་»པོ། «པེ་»«སྣར་ཞ་འབྲིང་པ། ་་་་་ཚེ«པེ་»སྣར་པར། «པེ་»«སྣར་རྣམས། «པེ་»སྣར་»ཤེས། ཤིང་བྱ་པར། «པེ་»«སྣར་»རོ། @«པེ་»«སྣར་འདུ། «པེ་»«སྣར་»ཚད། «པེ་»«སྣར་»+མ། དེ་།
