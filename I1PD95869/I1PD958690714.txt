བྱས་པས། །བསོད་ནམས་དེ་ནི་གང་རྙེད་པ། །དེ་ཡིས་འཇིག་རྟེན་བསོད་ནམས་ཤོག། བདག་གི་སྡུག་བསྔལ་བྲལ་གྱུར་ཅིག། དཔལ་གྱི་སྒྲུབ་ཐབས་འདི་བསྡུས་པ༑། རྡོ་རྗེ་འཇིགས་བྱེད་རྟོག་པ་ནི། །སུམ་བརྒྱ་པ་སྟེ་འཇམ་དཔལ་གྱི། །རྒྱུད་ལས་ཅུང་ཟད་ཙམ་ཞིག་ཕྱུང་། །མ་ཚང་ལྷག་པ་གང་ཡིན་པ། །བདག་་ལ་ལེགས་པར་བཟོད་པར་མཛོད། །སྔོན་དུ་བདག་གིས་གང་ལས་མཐོང་། །དེ་ནི་མཉེས་པར་བྱ་བ་ཙམ། །དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་སྒྲུབ་པའི་ཐབས་སློབ་དཔོན་ལ་ལི་ཏ་བཛྲའི་ཞབས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ:༔ཤྲཱི་དེ་བ་དང༌། བོད་ཀྱི་ལོ་ཙཱ་བ་ཅོག་གྲུ་བནྡེ་ཤེས་རབ་བླ་མས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།།རྗེ་འཇིགས་བྱེད་ཀྱི་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»༄༅། །དཔལ་རྡོ་རྗེ་་་་འཇིགས་བྱེད་ཀྱི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་དཀའ། ་«པེ་»«སྣར་བཏུས། «པེ་»«སྣར་»སྟན། «པེ་»«སྣར་གཞག་བསྒོམས། 《སྣར་བཤག«པེ་»«སྣར་»ཡང་། སུ«པེ་»«སྣར་བཟངས། «སྣར་བྱས། «པེ་»《སྣར་»སཏྠ། «པེ་»«སྣར་»ཉ། «པེ་»《སྣར་»ཏ། ེ«པེ་»«སྣར་—ཉ། །(«པེ་»«སྣར་»—། «པེ་»«སྣར་»ཅིག«པེ་»《སྣར་ཁྱད། «པེ་»«སྣར་ཕུགས། «པེ་»«སྣར་བཞགའ«པེ་»«སྣར་རྒྱ། «པེ་»«སྣར་པགས། «པེ་»«སྣར་གནོད། («པེ་»«སྣར་»གཞད། «པེ་»«སྣར་»ཏོག«པེ་»«སྣར་+ནི། ནས«པེ་»«སྣར་_གོ། «པེ་»«སྣར་ནུབ། «པེ་»«སྣར་»ཨོཾ་ཛི། «པེ་»«སྣར་»དཾ་ཙི། ::«པེ་»«སྣར་»པཾ། ཀྱེ«པེ་»《སྣར་ན། །«པེ་»«སྣར་ཡང་། «པེ་《སྣར་བཞུག། «པེ་《སྣར་»དུ། འ«པེ་»«སྣར་»བསྒྲེང་ཅན། །འཇིགས། —།
