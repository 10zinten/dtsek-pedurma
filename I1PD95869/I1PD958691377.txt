བསྟན་འགྱུར། རྒྱུད། བརྩི། གཞན་གྱིས་མི་ཐུབ་པའི་ཕྱག་རྒྱའོ། །ཕྱག་རྒྱ་དེ་ཉིད་ཀྱི་གུང་མོ་རྩེ་མོ་སྤྲད་ལ་་་ཕྱིར་བསྐྱེད་ནས་མཛུབ་མོ་གཉིས་བསྐུམ་པ་ནི་མི་ཕམ་པའི་ཕྱག་རྒྱའོ། །དེ་ཉིད་རང་ལ་མཛུབ་མོ་འཁྱིལ་པར་བྱས་ལ། མཐེ་ཆུང་གཉིས་ཕྱིར་བསྐྱེད་་ལ་ཐ་དད་དུ་གཞག་པ་ནི་གང་བའི་ཕྱག་རྒྱའོ་ཞེས་སོ། །ས་མ་:པ་ཐཱི། །།།་་་་་་་། དང༌། བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»ཧི། «པེ་»སྣར་ལ། ཆོ«པེ་»«སྣར་»—གི། །«པེ་»«སྣར་པོར། «པེ་»«སྣར་»། །«པེ་»«སྣར་སྒྲུབ། «པེ་»«སྣར་»—མོ། «པེ་»«སྣར་བཀྱེད། «པེ་»«སྣར་པཱ་ཐ། །།དབང་་་་་༌་་་་བན། །བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (ཅོ་ནེ། །།བར་རྒྱུད། ཅིདཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། ་་ལྡེབ། ན་བ?ནབ་ནབ་ཕྲེང་། །ཚེག་ཁྱིམ། །།།ན་བ?།
