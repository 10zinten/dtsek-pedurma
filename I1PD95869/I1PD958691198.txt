ཞེས་པས་སྤྱན་དྲང་ངོ༌། ་། ཨོ་ཧྥི་:སྦིཧ་ཞེས་པས་ཡོན་ཆབ་བོ། ཎ། ཨོཾ་:ཨཱཿཧཱུྃ་ཕཊ་ཅེས་པས་ཞལ་བསིལ་ལོ། །ཨོཾ་དུཥཱ་སཏཱ་ད་མ་ཀ་གཞེས་པས་དྲི་ཆབ་ཀྱི་སྔགས་སོ། །ཨོཾ་བཛྲ་:བྷཱུ་མི་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་གོ་མ་ཡ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛོད་ཀེ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུདྨེ་ཧཱུྃ་ཕཊ། ཨོཾ་བཛྲ་རེ་ཁེ་ཧཱུྃ་ཕཊ། ཅེས་པའི་སྔགས་་་་རྣམས་ཀྱིས་མཎྜལ་བྱི་དོར་བྱ་བའི་ཆོ་གའོ། །ཡང་ལག་པ་གཡོན་པས་མེ་ཏོག་་་་བླངས་ལ། ཨོ་སུ་རེ་ཁེ་སརྦ་ཏ་ཐཱ་ག་ཏཱ་དྷི་ཏིཥྛནྟུ་སྭཱ་ཧཱ། ཞེས་པས་མཎྜལ་དུའོ། །ཕྱི་ནས་ཐོ་བ་གཤིན་རྗེ་གཤེད་ཀྱི་ས་བོན་གྱིས་བཀུག་སྟེ། མཎྜལ་གྱི་དབུས་སུ་་་་་བསམས་ནས། །ཏྰ་བཅོམ་ལྡན་འདས་མཆོད་པར་བྱའོ། །ལོ་་། ཨོ་བྷ་ག་:ཝཱན་ཤཱི་མོ་རཀྟ་ཡ་མཱནྟ་ཀ་བཛྲ་པུཥྛ་ཨཱཧཱུྃ་པྲ་ཏིཥྛ་སཱ་ཧཱ། ་ཨོ་བྷ་ག་:ཝ་ཏི་ཤྲཱི་མོ་བཛྲ་བེ་ཏཱ་ལི་བཛྲ་པུངྒེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་ཀྱི་སྐུར་ེབསྒོམས་ནས་མཆོད་པ་བྱ་བ་ནི། །ཨོ་རཱ་གཱནྟ་ཀྲྀཏ་བཛྲ་པུཔྤེ་ཨཱཿཧཱུྃ་པྲ་ཏིཙྪ་སྭཱ་ཧཱ། །ཨོ་:མོ་ཧཱནྟ་ེབ་་་་བཛྲ་པུངྒེ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་མ་ཧཱནྟ་ཀྲྀཏ་བཛྲ་པུཔྤེ་ཨཱཿཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོ་པི་ཤུ་ནཱནྟ་ཀྲྀཏ་བཛྲ་པུབྦེ་ཨཱཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་དྭེ་ཥཱནྟ་:ཀྲྀཏ་བཛྲ་པུཥྤ་ཨཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་ཨིནྡྷ་ར་:ཏཱི་བཛྲ་པུཏྠཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ་ཨོ་མོ་ཧ་ར་ཏཱི་བཛྲ་པུབྦཾཔྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་དྭེ་ཥ་ར་ཏཱི་བཛྲ་པུཏྠཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོ་ཨིནྡྷ་:ར་ཏཱི་བཛྲ་་་པུཏྠཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་དཎྜ་ཡ་མཱནྟ་ཀ་བཛྲ་པུཔྤེ་ཨཱཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་མུངྒ་ར་ཡ་མཱནྟ་ཀ་བཛྲ་པུཔྤེ་ཨཱཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོཾ་པདྨ་ཡ་མཱནྟ་ཀ་བཛྲ་པུཥཱི་ཨཱཿཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨོ་ཁརྒ་ཡ་མཱནྟ་ཀ་བཛྲ་པུཥཱི་ཨཱཧཱུྃ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ་ཨོ་ཀུ་:མཱ་ར་རཱུ་པི་ཎི་ཛཛ༔ཧཱུྃ་ཕཊ། ་མེ་ཏོག་དབུལ་བའི་སྔགས་སོ། །ཨོཾ་དཱིཔ་ལོ་ཙ་ནི་བི་ཀྲྀཏཱ་ན་ན་མ་ཧཱཊྛཊྚ*། ཧཱ་ས་ནཱ་དི་ནི་དཱིཔཱ་དཱིཔ་ཡ་སྭཱ་ཧཱ། སྒྲོན་མའི་སྔགས་སོ༑། ཨོ་ཧཱིཧཧེཕཊ། ་བདུག་སྤོས་ཀྱི་སྔགས་སོ། །ཨོ་ཀཱ་ལ་:རཱུ་པ་སརྦ་་་_།
