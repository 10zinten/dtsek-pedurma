བསྟན་འགྱུར། ་རྒྱུད། མི་:་རྣམས་སོ། །མགོ་ལ་ཨ་དང་ཨ་ཡིག་རྣམས་ཞེས་པ་ནི། ཨ་ཡིག་ཡོངས་སུ་གྱུར་་་པའི་ཟླ་བའི་དཀྱིལ་འཁོར་ལ། །ཨ་ཡིག་གནས་ཞེས་བྱ་བར་འབྲེལ་ཏོ། །དེ་དག་ནི་དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་རྣལ་འབྱོར་གྱི་རྒྱུད་ལས་རྟོག་པ་བདུན་པའི་བཤད་་་པའོ༑། ་་། །བ་བར་བོའི་། ཆུརྡོ་རྗེ་འཇིགས་བྱེད་རྒྱུད་ལས་ནི། །མདོ་མདོར་བཤད་པ་གསལ་བར་བྱས་པ་ཡིས། །བདག་གིས་བསོད་ནམས་གང་ཐོབ་པ། །དེ་ཡིས་འགྲོ་རྣམས་རྒྱལ་བར་ཤོག། དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་རྣལ་འབྱོར་གྱི་སྐབས་མདོ་མདོར་བཤད་པ་རྫོགས་སོ། །།།བའི་ནང་གང་་::ཀྱི་འབར་སློབ་དཔོན་དཔལ་འཛིན་གྱི་རྗེས་སུ་འབྲངས་པའི་སློན་དཔོན་དཔལ་ཆེན་པོ་རྡོ་རྗེ་གྲུབ་པས་མཛད་པ་རྫོགས་སོ། །།ལ་དོལ་པོ་བ་ཡིན་ཏེ། བ་མ་ཀ་བ་བུ་ཆེན། རྗེ་ཁ:::དབང་པོ་བདེ་བར་འཁ་ཁ་པ། རིག་གནས། བྱ་ཁང་དམ། བཀའི་པ་དང་། དེ་དབྱེ་བསྡུར་མཆན། དར་.་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་མདོར་བཤད་པ་སློབ་དཔོན་ཆེན་པོ་རྡོ་རྗེ་གྲུབ་པས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་སྱ་ཏི། «སྣར་ཡིས། «སྣར་—བྲི། །(«སྣར་»ག་གྱིས། «པེ་»«སྣར་»གསད། («པེ་»«སྣར་སྲེག་®«པེ་»«སྣར་པ། «སྣར་བབ༑རྗེ་.ཀྱེ«པེ་»«སྣར་»བསམ་གཏན་བཞིའི་རྣམ་པར་དག་པའོ། །«པེ་»«སྣར་»—པ། «པེ་»《སྣར་»+ནི། «པེ་»འཛག«སྣར་»འཇག་«པེ་»《སྣར་པ་ཞེས། @«པེ་»«སྣར་ན། ི«པེ་《སྣར་»(«པེ་»«སྣར་གིས། ཀྱེ«པེ་»«སྣར་ཡིས། «པེ་»«སྣར་གསད། «པེ་»«སྣར་»ལས། «པེ་»«སྣར་ནི། Q«པེ་»«སྣར་»ལ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་པའི། «པེ་།
