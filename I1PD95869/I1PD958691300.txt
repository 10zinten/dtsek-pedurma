ཏ་ཤི་ཀ་པ་ག་བ་ད་དཀའ་དེ་བབལ་བ། ཁ་རེ་བ་ཡོད་ལ། ལ་ཁབ་ཏེང་ལ་བྱ་བདེ་བ་ལ་ਕੌਰਬਰੋਥਲਨੂੰਨਵੇਂਸਬੰਸਸ਼ੇਲਸਣབརྗོད་ན་མི་དགའ་བ་བདེ་བའི། བོ་དེ་དག་འགལ་བ་གཙོ། ༄༅། །བདུད་རྩི་བྱིན་གྱིས་བརླབ་པ་ཞེས་བྱ་བ། བཀུར་འདས་ག་ལ་བ་ཏ་ཀ་་་་་་་་་་་་་་་་་་་་་་་་་བཀའ་བཞི་པའོ་ལྔ་ལ་མ་བབ་པའི་གནད་འགག་ག་་ཁྱ་འཁོད་པ། གཙོ་པོ་དང་། ག་ན་བ་མ་འ་ར་བ་ན་བཞུགས་པ་འགའ་བའི་ਭਰਾਵਣਨੂੰਸਰਰਰਸਲਨਕਉਂਸਲਰਬਣ::༄༅། །རྒྱ་གར་སྐད་དུ། ཨ་:མྲྀ་ཏ་ཨ་དྷིཥྛ''ན་ནཱ་མ། བོད་སྐད་དུ། བདུད་རྩི་བྱིན་གྱིས་བརླབ་པ་ཞེས་བྱ་བ། འཇམ་པའི་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །གང་ཞིག་རྣལ་འབྱོར་གཉིས་བརྟེན་ནས། །གཤིན་རྗེ་གཤེད་དཔལ་གཉེར་བྱེད་ན། །དེ་ཡི་བསྟེན་བྱ་དམ་ཚིག་ནི། །བྱིན་གྱིས་བརླབ་པའི་རིམ་པ་བཤད། །དེ་ལ་་་རྣལ་འབྱོར་པས་དབེན་པའི་གནས་རིའི་ཕུག་གམ་དུར་ཁྲོད་ལ་སོགས་པ་མིའི་སྐྱེ་་་བོས་དབེན་པར་བྱས་ལ། བཅུད་མ་ཉམས་པའི་དྲི་ཆེན་དང་། གྷཉྫི་དང་། བ་རུ་ར་དང་། ཨ་རུ་ར་དང༌། བླ་བ་དང༌། མར་དང་། །སྦྲང་རྩི་་་་དང་། བུ་རམ་དང་། སོ་གའི་དུས་མ་ཡིན་པའི་དུས་སུ་འདུས་བྱས་པའི་པ་ར་ན་དང༌། །ལེགས་པར་སྦྱར་བའི་རིལ་བུ་ཨུ་རིན་དྷ་བའི་སྣོད་དང༌། །སྐུ་ལཱ་བ་་་་དམར་པོའི་ན་བཟའ་དང་བཅས་པ་སྟ་གོན་དུ་བྱས་ལ། །ཇི་སྐད་བཤད་པའི་རིམ་པ་གཉིས་སྔོན་དུ་འགྲོ་བས་མཆོད་པ་དང་གཏོར་མའང་ཕུལ་ཞིང་། སློབ་དཔོན་།
