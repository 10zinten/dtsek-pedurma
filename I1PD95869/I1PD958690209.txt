༌ཀཀ:༌་ཡོ་་་་ཤོག་ལ་ཁའོ་ཀྲང་ཀ་ཌ་ཌRཁ་ཊཾ་ཊn::ཀར་དེབ་ར་བ། འདེམེད་བྱེད། ་དང་ནལུས་ཀྱི་ར་བ་ནས་འབབ། །ད། ལེ་བ། ཕྱུར་པ་བླ་། དེ་ལ། །ཡཾ། །བ་ཤིང་མེ་བསད་པ་དང་༄༅། །གཤིན་རྗེའི་གཤེད་ནག་པོ་སྒྲུབ་པའི་ཐབས། །ཡི། ཀ་ཏཾ༄༅། །རྒྱ་གར་སྐད་དུ། ཀྲྀཥཱ་ཡ་མཱ་རི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། གཤིན་རྗེའི་གཤེད་ནག་པོ་སྒྲུབ་པའི་ཐབས། འཕགས་པ་འཇམ་དཔལ་གཤིན་་་་་རྗེའི་གཤེད་ལ་ཕྱག་འཚལ་ལོ། །སྙིང་རྗེས་སེམས་ཅན་ཕན་པ་མཛད་ཅིང་དཔག་བསམ་ཤིང་དང་འདྲ་བ་ཡི། །དཔལ་ལྡན་གཤིན་རྗེའི་གཤེད་ཀྱི་མདངས་འབར་་་ཆར་སྤྲིན་གཞོན་འདྲ་ཡིད་འོང་བ། །རྟོག་པའི་རླབས་བྲལ་ལྷན་ཅིག་སྐྱེས་པའི་འོ་མ་ལས་འཕྲོས་དེ་ལ་ཕྱག་འཚལ་ནས། །འདི་ཡི་དཀྱིལ་འཁོར་བསྒྲུབ་བྱ་དང་་་་བཅས་དཔལ་ལྡན་བླ་མའི་བཀའ་ཡིས་བཤད་པར་བྱ། །དང་པོར་སྔགས་པས་ཕྱི་ནང་གི་རྣམ་པར་གཡོ་བ་མ་ལུས་པ་ཡོངས་སུ་བཏང་ལ། བློ་དང་ལྡན་ཞིང་སྙིང་་རྗེ་དང་ལྡན་པས་རང་གི་སྙིང་གར་ཐུགས་ཀྱི་ས་བོན་གྱི་སྔགས་འོད་ཟེར་གྱི་རིགས་ཀུན་དུ་འཁྲིགས་པ་བལྟས་ལ། འོད་འཕྲོས་པ་དེ་ལས་ཕྱོགས་སྐྱོང་བ་བརྒྱད། །ཤིང་བརྒྱད་ཀྱིས་བརྒྱན་ཅིང་དྲག་པོའི་དུར་ཁྲོད་དོ། །དེའི་དཀྱིལ་དུ་འཇིགས་པ་།
