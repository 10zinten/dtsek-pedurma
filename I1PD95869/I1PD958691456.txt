རྒྱས་དང་བསྡུས་ཀྱི་པ་ལས། །དགེ་བ་གང་ཡིན་སེམས་ཅན་དོན་དུ་བསྔོ། །དཔལ་གཤིན་རྗེ་ཚེ་བདག་ནག་པོའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་དྲན་པ་གསལ་བ་ཞེས་བྱ་བ༑། དཔལ་ནག་པོ་ཞབས་ཀྱིས་མཛད་པ་རྫོགས་སོ། །ན་པ། པ་པཎྜི་ཏ་ཆེན་པོ་ནག་པོའི་ཞབས་དེ་ཉིད་དང་། ལོ་ཙཱ་བ་གཉན་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།གང་བ་མ་འ་ར་ལ་ས་དང་::པར་དང་ཡི་གེ་། ལེ་དེ་ད་ཚུན་དང་ནི། །བསྡུར་མཆན། བའི་ཚེ། དག་པ་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»《སྣར་»༄༅། །ཚེ་བདག་ནག་པོའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་དྲན་པ་གསལ་བ་ཞེས་བྱ་བ་ནག་པོས་མཛད་པ་པཎྜི་ཏ་དེ་ཉིད་དང་ལོ་ཙཱ་བ་དར་མ་གྲགས་ཀྱི་འགྱུར། «པེ་»«སྣར་—། «པེ་བསྐྱེད། «སྣར་ཕྱགས་ཙལ། །«པེ་»«སྣར་དང་བ། «པེ་»«སྣར་»བསྒོམས་པས། །རྒྱ་ཆེ་བ། ཀ་ཀྱི«པེ་»«སྣར་གཏོང་། «པེ་»«སྣར་བཟུང་། «པེ་»«སྣར་ཚོགས། «པེ་»«སྣར་»+བྱས་པས། །«པེ་»«སྣར་»ཆགས། »«པེ་《སྣར་བྲེ། («སྣར་བརྒྱན། «པེ་»«སྣར་»::ཉ། ་《སྣར་བའི། (《པེ་»«སྣར་»། «སྣར་»འཐོད། གེ«པེ་»«སྣར་»གདུང་། :(«པེ་»«སྣར་+རྒྱས་པའི། «པེ་»《སྣར་+ཕྱོགས་བཞི་གནག་ལ་མཚམས་བཞི་དམར། །(«པེ་»《སྣར་ཟླའི། «པེ་»«སྣར་སྟན། («པེ་»«སྣར་ཟློག་་ཀྱི་རྟོ«པེ་»«སྣར་བརྟུན། «པེ་《སྣར་རྔམ། «པེ་»«སྣར་ར། «པེ་»《སྣར་»ལོ། «པེ་»«སྣར་»+ཡབ། «པེ་»《སྣར་»ལྗང་། «པེ་»«སྣར་»ས། «པེ་»«སྣར་»སུ་བྱིན། «པེ་»《སྣར་དྲང་། ཤི། ་(«པེ་»«སྣར་»གཟའ། (ི«པེ་»སོགས། ་«པེ་《སྣར་འདྲི། «པེ་»«སྣར་ཡི། །«པེ་»པོ《སྣར་»དང༌། «པེ་»《སྣར་»ཏེ། «པེ་»«སྣར་»—པར་བྱ། །ས་དང་། མར་་་་(«པེ་»《སྣར་»གྱིས། :«པེ་»«སྣར་»ཉིད་མཉེས། («པེ་»«སྣར་»ཞིག་«པེ་»«སྣར་»བཟུང་། «པེ་»《སྣར་»སྤང་། «པེ་»«སྣར་+དཔའ། །ཁོ(«པེ་»«སྣར་»ཀྱི་བཟུང༌། («སྣར་»རྡོ་རྗེ་དེ། «པེ་»གཟུང་། «པེ་»«སྣར་བསད། །«པེ་»《སྣར་གྱིས་རྩར་གྱུར་པའི། པ«པེ་»«སྣར་»འདུལ། «པེ་»«སྣར་རྟོགས། «པེ་»«སྣར་»།
