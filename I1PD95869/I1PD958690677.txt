དུ་དུ་བྱང་། དོ་བདག་ས་ལ་་་་་་ཚེ་ཀྵ་ས་་་་ཁ་བ། ན་བསྡུར་མཆན། །ག་ང་བ་འ་། །རལ་གསུམ་བར་དང་། དགཔདམ་པ་ཆ་མང་། པདེལ་(མཚན་བྱང་འདི་《སྡེ་»«ཅོ་»ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །བཅོམ་ལྡན་འདས་ཞེ་སྡང་གཤིན་རྗེ་གཤེད་ལ་་་ཕྱག་འཚལ་ལོ«པེ་»«སྣར་»སྐྱེད། འ«པེ་»«སྣར་མུན་ཏྲི། «པེ་»«སྣར་»ཛ་མྱོས། ཁ་«པེ་《སྣར་བསྣམས། «པེ་《སྣར་དུ། «པེ་»«སྣར་བསྣམས། «པེ་»ཤོདྷེ། «སྣར་»ཤོད་དྷེ། ས«པེ་»«སྣར་མདུན། «པེ་»«སྣར་དྨར་པཉྩ་ཙ་རྒྱས། Q«པེ་»«སྣར་_དང༌། «པེ་《སྣར་ཙ། །B%ਲੂਣਦਾਸਮਾਸਧਣਲਈབབཀོལ་བ། ད་ལྟ་ད་་མ་་་་་་་་བ་དམ་པ་དང་འགལ་བ་བསྡུར་འབྲས་རེའུ་མིག་དང་། དེ་དག་བ་ལ་རྒྱུད། (ལ་གྱི་དོསྡེ་དགེ། ལ། ་རྒྱུད། པེ་ཅིན། སྔགས། སྣར་ཐང་། །བསྔགས། ས་། རྒྱུད། དཔེ་གཞི། སྡེ་ཚན། །པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཕྲེང༌། ཚིག་རྐང་། །ཚེག་ཁྱིམ། །རབབརན་བ་ནན། །བ་ན་ལེན་། ་་་ང་ལ་བྱ་བ་བྱས། །།
