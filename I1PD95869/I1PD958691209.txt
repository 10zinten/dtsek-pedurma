བསྟན་འགྱུར། རྒྱུད། ༼ཝིཏི་་་་་་་་་་་་་་་་་། མ་དག་སྐྱོང་། pལ་འབད་འབད་པ་གནང་བ་དང་བློ་Bਨਹੌਸਲਖੁਸ'ਮਮਮਭਾਵਬੈਂਕਿੰਗਸਬਲਸਲੇਲੈਣན་་་་བསྡུར་མཆན། གལ་སྒྲིག་བྱེད་ཀྱི་ཁ་ལོན་པ་བྱཁང་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །གཤིན་རྗེ་གཤེད་དམར་པོའི་ཏིང་ངེ་འཛིན་གྱི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་མཱ། «པེ་»«སྣར་»། «པེ་»«སྣར་པ། («པེ་»《སྣར་ལ། ༄༅«པེ་»«སྣར་»—པ། ་་་་་་::::། །«པེ་»«སྣར་»རྩིབས། «པེ་»«སྣར་»—དེ། «པེ་»«སྣར་»མེ་ཡི་སྲེགས། «པེ་»《སྣར་བིགྷཱནྟ་ཀྲྀ་ཏཱ། «པེ་»«སྣར་བྲབས་པ། «པེ་»«སྣར་ཐོན། «པེ་»«སྣར་བྲབས། «པེ་»《སྣར་»+ཡང་། «པེ་»«སྣར་»སྭཱ་ཧཱ། །«པེ་《སྣར་གསོལ། «པེ་《སྣར་བདུན་གྱིས། །«པེ་»«སྣར་བཞུགས། «པེ་»《སྣར་»+ཞལ་ནས། «པེ་»སྣར་པའི། ...༄༅«པེ་»《སྣར་»ཡིན། «པེ་»«སྣར་»ཁྱོད། «པེ་»«སྣར་པ། «པེ་»«སྣར་»ཁྱེད། «པེ་»«སྣར་མགོ། «པེ་»«སྣར་»ཏྭཾ། ཨ་བྷི་ཥིཉྩ། «པེ་»«སྣར་པ་ཏེ། ཏེ་ན། ཏིབྷི། «པེ་»«སྣར་»ཡི། «པེ་ཧཱུྃ། «སྣར་»ཧཾ། «པེ་»«སྣར་པའི། «པེ་»«སྣར་+དང་། «པེ་»«སྣར་བྱས་པའི། «པེ་»«སྣར་སུ། ਦਾਬਲੋਚਹੈਕRRਕਰਨ।ਮਨਬੰਸབསྡུར་འབྲས་རེའུ་མིགང་ཚོ་བོད་རྒྱུད། (འདུ་བྱས་ལཙ་ནེ། དཔེ་གཞི། །སྡེ་དགེ། །།པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། _རྒྱུད། རྒྱུད་འགྲེལ། །རྒྱུད། དེས་རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། བ་—ན། །ན་བ*བ་—ན། བབཕྲེང་། །ལ། ཚེག་ཁྱིམ། །།།།།
