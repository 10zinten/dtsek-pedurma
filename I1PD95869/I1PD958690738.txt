འོ་བ་༼ཏོ་་་་་་་། ་བར་ནས། ས་ག་ཟླ་བདུན་་པ་་་་་་་་་གནས་་་༄༅། །དྲག་པོའི་ལས་ཀྱི་བྱ་བ་བདུན་པ། འགན་ཚེ། །ཏེ་པ་ལ་གོ་བ་དང་གནས་ਨਿਬੌਸ'ਦਾਕੋਲਾਸਫ਼ਰਕਛਤਵਾਲਸਨਾਯਨੂੰ,ਕਲਸੀਕਿਉਸਲਈਇਸਸਮਲਾ::ཀ་དག་པ་འདྲ་བ་དང་གནག་བག་མིམ་་བསམ་དང་བྱང་༄༅། །དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ལ་ཕྱག་འཚལ་ལོ། །དྲག་པོའི་ལས་་ཀྱི་བྱ་བ་བདུན་བཤད་པར་བྱ་སྟེ། དེ་ལ་དང་པོར་མཆོད་སྦྱིན་བྱ་སྟེ། འཕགས་པ་མཆོད་པ་དང༌། འགྲོ་:བ་ལ་་སྦྱིན་གཏོང་བཏང་བ་དང་། བླ་མ་ལ་སོགས་པ་སྤྱན་དྲངས་ཏེ། ཇི་ལྟར་འབྱོར་པའི་ཡོན་ཏན་ཕུལ་ནས། བཀའི་གནང་བ་དང༌། སྨོན་ལམ་ནོད་པར་བྱའོ། །མཆོད་སྦྱིན་བྱ་བའི་ཆོ་ག་སྟེ་དང་པོའོ། །།ལོ་ད་ནི་གཏོར་མའི་ལས་བཤད་དེ། དེ་ནས་ཉི་མ་གཉིས་པ་ལ་མེ་ཏོག་ནག་པོ་དང༌། ཤ་ཆེན་པོའི་བདུག་སྤོས་དང་། ཏིལ་ཆེན་པོའི་མར་མེ་དང་། བདུད་རྩི་རྣམ་ལྔ་དང་ལྡན་པའི་དྲི་ཆབ་དང༌། བཤོས་:གཙང་ཀྱང་་མི་ལ་སོགས་པའི་གདོང་པ་ཅན་གྱི་ཤ་དང་། རྟ་དང་བོང་བུ་དང་། རྔ་མོ་ལ་སོགས་པ་ཕྱུགས་སྣ་ཚོགས་ཀྱི་ཤ་དང་། །ཁྱི་དང་ཝ་ལ་སོགས་པ་གཅན་གཟན་སྣ་ཚོགས་ཀྱི་ཤ་དང༌། །
