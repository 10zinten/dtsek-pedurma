།བསྟན་འགྱུར། རྒྱུད། རྩི། ཨིནྡྷ་ར་ཏི་ཧཱུཾ་ཛ། །དཔའ་མོ་དབང་ཕྱུག་མ་བཞིའི་སྔགས་སོ། །དང་པོ་མི་མགོ་གསུམ་:གྱི་སྟེང་དུ་པདྨའི་སྣོད་དུ་བསྐལ་པའི་མེའམ། རྡོ་རྗེ་ལྟ་བུར་འབར་བ་ཅན། དེའི་ནང་དུ་ཟན་དང་ཆུ་བསྲེས་ལ། གོ་ཀུ་ད་ཧ་ན་:འོ་མ་མར་དང་རྒྱ་སྐྱེགས་འདྲེས་མའི་ཁུ་བ་ལྟ་བུ་ཡང་དག་པར་བལྟས་ལ་བསིལ་བར་གྱུར་པ་་་་་་བསམས་ལ། ཛཧཱུྃ་བཾ་ཧོ༔སྔགས་འདིས་དཀྱིལ་འཁོར་གྱི་འཁོར་ལོ་སྤྱན་་་་དྲངས་ཏེ། དེ་ནས་མཆོད་ཡོན་སོལ་སོགས་པ་ཕུལ་ནས། བསིལ་བར་གྱུར་པའི་གཏོར་མ་དབུལ་བར་བྱའོ། །།།གང་། ཁྱ་ཞེ་ཏུན་ཨོ་རཀྟ་མུ་ཀུ་ཊ་ཡ། རཀྟ་ཛཱ་ཊཱ་ཡ། རཀྟ་མུ་ཁཱ་ཡ། རཀྟ་ཤ་རཱི་རཱ་ཡ༑རཀྟ་བྷུ་ཛཱ་ཡ། རཀྟ་ཀ་མུཎྜ་མཱ་ལཱ་བྷ་ར་ཎཱ་ཡ། རཀྟ་ར་ཐ་ས་མ་རུ་ཞྭ་ཡ། །རཀྟ་པདྨ་ྋས་ནཱ། རཀྟ་བྷ་ར་ཎ། བྷུ་ཥ་ཎཱ་ཡ་། :ཨི་དཾ་བ་ལཾ། གྲིཉྩ་གྲིཉྩ། མ་མ་བིགྷཱནི། ཧ་ར་ཧ་ར། ཙ་ཏུམྨཱ་ར་ཎ། བི་བཱ་ར་ཡ། བྷ་མ་ཡ་བྷ་མ་ཡ། ཙྪིནྡ་ཙྪིནྡ། བྷིནྡ་བྷིནྡ། ནཱ་ཤ་ཡ་ནཱ་ཤ་ཡ། ཤོ་ཥ་ཡ་ཤོ་ཥ་ཡ། ཙྪེ་ད་ཙྪེ་ད། བྷེ་ད་བྷེ་ད་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་སོ་སོ་ལ་་་་དབུལ་བར་བྱ་སྟེ། ཨོཾ་ཨིནྡྲ་ཡ་ན་མོ་ཧཱི། ཡ་མཱ་ཡ་ན་མཛྫི། བ་རུ་ཎཱ་ཡ་ན་མོ་ཝི། ཀུ་བེ་རཱ་ཡ་ན་མཀྲྀ། ཨཱི་ཤཱ་:ནཱ་ཡ་ན་མཏ། ཨག་ཡེ་ན་མ་ཨ། །ནིརྗེ་ཏ་ཡེ་ན་མོ་ན། བ་ཡ་བེ་ན་མོ་ན། ཙནྡྲ་ཡ་ན་མོ་ཧཱུྃ། ཨཱཎཱ་ཡ་ན་མོ་ཧཱུྃ། བྲཧྨ་:ཎ་ཡེ་ན་མ་ཕཊ། བ་སུཾ་:དྷཱ་ར་ཡི་ན་མ་ཕཊ། བེ་མ་ཙི་ཏྲ་ཎི་ན་མས། སརྦ་བྷུ་ཏེ་བློ་ན་མོ་ཧཱ། :·ཧ་ཧ་ཧཱི་ཧཱུྃ། ཧཱུྃ་ཧཱུྃ་ཕོ་ཕོ་:སྭཱ་ཧཱ། ཨོཾ་མམྨེ་མམྨེ་བི་ག་ཏེ་ཧཱུྃ། བཛྲ་ཙརྩི་ཀ། ཧ་ར་ཧ་ར། །བྷུཀྐོ་མཱ་ར་ཡ་མཱ་ར་ཡ། །ཨ་མུ་ཀཾ་ཛཛ། ཨོ་གྷོ་ཎེ་སུ་གྷོ་ཎེ། བཛྲ་མཱ་མ་ཀཱི། བྷ་ར་བྷ་ར། ་སམྦྷ་ར་སམྦྷ་ར། :：བིགྷ་ན་ཛཛ། ཨོ་མརྨེ་མརྨེ་། ཀློ་བྷ་ཎི་བཛྲ་ས་ར་སྭ་ཀྱི༌། ཧ་ར་ཧ་རི། །
