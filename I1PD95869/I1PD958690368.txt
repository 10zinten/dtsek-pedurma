ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོཉཧཾ། སེམས་དེ་ཉིད་ནི་དག་བྱས་ནས། །ཨ་ལས་བྱུང་བའི་ཉི་དཀྱིལ་ཏེ། །དེ་ལ་ཧཱུཾ་སྐྱེས་ཁ་དོག་སྔོ། །འོད་ཟེར་འཕྲོས་པས་ཀུན་དུ་ཁྱབ། །བསྲུང་བའི་དོན་དུ་རྡོ་རྗེ་བསྒོམ། །ཀུན་དུ་རྩེ་ནི་ལྔ་ཡིན་ཏེ། །རྡོ་རྗེ་ས་གཞི་་དེ་ལས་བྱུང་། །དེ་བཞིན་གུར་དང་ར་བའོ། །སངས་རྒྱས་ཞིང་དག་དོན་དུ་ནི། །ཡང་དག་བསམ་གཏན་ཀུན་:བརྩམས་ཏེ། །དེ་ལ་སྣ་ཚོགས་པད་ཟླ་ལ། །རྟག་རྡོར་ཡན་ལག་ལས་བྱུང་ངོ༌། །དཀར་པོ་གྲུ་གསུམ་ཆོས་་་་་དབྱིངས་ཏེ། །དབུས་སུ་སྣ་པད་རྡོ་རྗེའི་གདན། །ནམ་མཁའ་དང་མཉམ་་་་་འགྲེང་བ་བསམ། །དཀྱིལ་འཁོར་དབུས་སུ་གནས་པའོ། །ཧཱུཾ་རྣམས་འཁོར་ལོ་སྟེ་དབུས་སྐྱེས། །དག་པའི་ཡན་ལག་ལས་བྱུང་སྟེ། །ཟུར་བཞི་དང་ནི་སྒོ་བཞི་པ། །རྟ་བབས་བཞིས་ནི་བརྒྱན་པའོ། །དྲ་བ་དྲ་ཕྱེད་ཀྱིས་ནི་སྤྲས། །མཆོད་རྣམས་ལྷ་མོ་ཡང་དག་བརྒྱན། །སྣ་ཚོགས་རྩེ་མོ་རླུང་གིས་འཕྱར། །བ་དན་དང་ནི་དྲིལ་བུ་ལྡན། །སྣ་ཚོགས་རྡོ་རྗེ་ཀུན་དུ་མནན། །ཟླ་བ་ཉི་མ་རྒྱལ་་བའི་སྟན། །ཤར་དུ་དཀར་པོ་ཆེན་པོ་སྟེ། །ལྷོ་རུ་སེར་པོ་ཡང་དག་ལྡན། །ནུབ་ཀྱི་ཆ་ནི་དམར་པོ་ཡིན། །བྱང་དུ་ལྗང་གུའི་ཁ་དོག་གོ། །ཨིནྡྲ་ནཱི་ལའི་འོད་དང་ལྡན། །དབུས་ཀྱི་ས་ནི་རྣམ་བསྒོམས་ཏེ། །འདི་ལྟར་ཕྱོགས་ཀྱི་དབྱེ་བ་ཡིས། །སྣ་ཚོགས་རྡོ་རྗེ་བསྒོམ་བྱའོ། ་། བྲིས་སྐུ་ལ་ཡང་དེ་བཞིན་བསྒོམ། །རབ་ཏུ་་་་གཟུགས་ཀྱི་རིམ་ཇི་བཞིན། །རྣམ་ལྔ་བསམ་པར་བྱས་ནས་ནི། །སངས་རྒྱས་་་གཟུགས་བརྙན་རྣམ་བསྒོམ་མོ། །རྣམ་ལྔ་དེ་ནི་དཔལ་དང་ལྡན། །དེ་རྗེས་གང་བར་བདུད་སེལ་བྱེད། །དེ་ནི་སངས་རྒྱས་གཟུགས་བརྙན་གྲགས། །དེ་ཕྱིར་་དེ་ལ་དེ་གསུངས་སོ། །ལྟེ་བར་སྣ་ཚོགས་པད་ཟླ་ལ། །དབྱངས་ཡིག་ཉིས་སྐོར་དགོད་པར་བྱ། །དེ་ཡིས་ཟླ་བ་རྫོགས་པ་སྟེ། །ཀ་ཡི་ཐོག་མར་ཉིས་སྐོར་བྱ། །—།
