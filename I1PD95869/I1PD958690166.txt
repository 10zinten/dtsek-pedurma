.བདེ་བ་བདེ་བ་ལ་རྒྱབ་ཀྱི་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །གཤིན་རྗེ་དགྲའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་གཤིན་རྗེ་འབྱུང་བ་ཞེས་བྱ་བ་བཞུགས་སོ། །ན་«པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་ནཱཾ། «སྣར་»ཆགས། «པེ་»«སྣར་»སྐྱེ་བོ། ས་«པེ་»«སྣར་བའི། «པེ་»«སྣར་»—ལུས་ངག་ཡིད་ཀྱི་དགེ་བ་གང་། །«པེ་»«སྣར་བགྱིད། ︽«པེ་»《སྣར་བ་ཨཱཏྨ་ཀ་ཧཾ། «པེ་»«སྣར་»། «པེ་»«སྣར་ཨ། །«པེ་»«སྣར་ཁཾ། «པེ་»«སྣར་བྲུམ་ཟེར། «པེ་»«སྣར་»ཟླའི། «སྣར་བཞིའོ། །«པེ་»《སྣར་སྟེང་། «པེ་»«སྣར་»ཨ། ཚེ«པེ་《སྣར་»ཀྱིས། «པེ་»«སྣར་ཀོ་ཏཾ། ་«པེ་»«སྣར་བན་དས། ཀྱེ«པེ་《སྣར་»ཡིས། «པེ་»«སྣར་»ཀ«པེ་»«སྣར་»བོར་བདུན། «པེ་»«སྣར་གཏོང་། «པེ་»«སྣར་»ཀླུབས། སུ«པེ་»«སྣར་»—ཉ། «པེ་《སྣར་བརླབས། །«པེ་»«སྣར་»ཝིལས། «པེ་»«སྣར་»རྭ་ཉིད་ལ། །འ«པེ་»«སྣར་ལིཀ«སྣར་»དྷིཀ«པེ་»«སྣར་»ཀ«པེ་《སྣར་»བཞག«པེ་»«སྣར་ར། བག་མའི་«པེ་»«སྣར་»བན་དའི། «སྣར་འཁྱུགས། «པེ་»«སྣར་བན་ད། ®«པེ་»«སྣར་སྐུ། སུ«པེ་»«སྣར་»སྟེང་དུ། པོ«པེ་»«སྣར་»ཀ«པེ་»«སྣར་»—ཉ། ་«པེ་»«སྣར་»ཀ་དེ་«པེ་»«སྣར་»ཉ། «པེ་»«སྣར་ལ། འ«པེ་»«སྣར་»+བ། ༄༅«པེ་《སྣར་»འོངས། «པེ་»། །«སྣར་»ཀ«པེ་《སྣར་བསམ། «པེ་»གཙིགས། «པེ་《སྣར་ཕྱེད། «པེ་《སྣར་»བཟུང་། ཁྱེ«པེ་»སྣར་»ཕྱོགས་སུ་དུང་ནི་གསུམ་པ་ལ། །དེ་(«པེ་»«སྣར་»—། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་ཀླུ། «པེ་»«སྣར་»པོར་མི་ཐུང་་། གཤིན་དགྲས་གནག«པེ་»«སྣར་»པཱ། «པེ་»«སྣར་»ཉ། ལ་་་་་་་་(«སྣར་»ཡཾ། «པེ་»ཡཾ་གྱིས་བྱིན་བརླབས། 《སྣར་»ཡཾ་གྱིས་བྱིན་བརླབ། «སྣར་»འདི། «སྣར་»མཐོང་། །«སྣར་བསམ། «པེ་»«སྣར་བརྡུབས། གནད་(«པེ་»«སྣར་གསལ། །《སྣར་བསམ། «པེ་»«སྣར་མཐར། དེ་བས་ན་དེ་བས་ཀྱི་༽«པེ་»«སྣར་»སྔོ། «པེ་»«སྣར་ན། འ«པེ་»«སྣར་བསམ། ཆེ«པེ་《སྣར་»ཡིས། «པེ་»《སྣར་མིན། །ཀ་བ་ཀ་བ་ད༽(«པེ་»«སྣར་»གདབ། «པེ་»«སྣར་»དགང༌། «པེ་»«སྣར་ནི་བཛྲ་བྷཱ། «སྣར་+ཀུན། །།
