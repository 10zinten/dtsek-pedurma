བསྟན་འགྱུར། །རྒྱུད། མིདང་། ལན་ད་ཀ་ག་ཟོ། དུར་ཁ་ལུ་ཨ་ལོ་ཀ་འི། ་་ཁ་ཆུ་ར་ཏའཆ། ਸਲਾਸ਼ਲ'ਮੈgਲੋਂਛੂਟਸੌnཀུན་འདུས་ལ་བ་ཆ་མ་འོ་བསྡུར་མཆན། ང་ལ་རྫོགས་ཀྱང་དུ་ཀཊཱིཀམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །གཤིན་རྗེ་གཤེད་་་་ནག་པོའི་སྒྲུབ་ཐབས་བཞུགས་སོ། །(«པེ་»«སྣར་ར་སཱི། ན«པེ་»«སྣར་བ། «པེ་《སྣར་ཡིད། «པེ་»«སྣར་»བསྒེགས། ་«པེ་»«སྣར་»བརླབས། ་ན«པེ་»«སྣར་མའི་བསྙེན་པ། «སྣར་+ས༑ས«པེ་»«སྣར་»+ཡེ་ཤེས། «པེ་»«སྣར་»ཀིརྟིས། ་་ལ་བ་དང་། ངས་ང་ལ་ལན་ཁ་ཕྱོགས༌ལ་ལ། བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (ངས་ཀྱི»། རྒྱུད། །ནས་བསནས། །དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། རྒྱུད་འགྲེལ། རྒྱུད། ::། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། །།།::ཀ་ལྡབ། ནདབར་། ན་ན?བ་ན་ཕྲེང་། །ཚིག་རྐང་། ཚེག་ཁྱིམ། ་་སད་པ་་་་་་་་་་་་་་་ལོ། བྱང་ཆུབ་ཀྱི་ལག་ལེན་བའི་ལ་བ་ཞུ་བའི་ཐིག་ལེ་བྱང་ཆུབ་སེམས། །བབས་པ་འམ་དཔལ་བས་རྗེ་གཤེད་ཡིན་པ་དབང་དང་མང་། (ག་མོ་་་པ་།
