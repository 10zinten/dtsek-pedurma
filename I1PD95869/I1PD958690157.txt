རྒྱུད། མིས་ཁོ་ན་པོ་འདི་ལ་ལྟོས། །ད་ནི་དད་པ་རབ་ཏུ་བསྐྱེད། །སངས་རྒྱས་གདུང་གི་རིགས་སུ་འཁྲུངས། ་། སྔགས་དང་ཕྱག་རྒྱས་བྱིན་གྱིས་བརླབས། །འགྲོ་བ་ཕུན་ཚོགས་དངོས་གྲུབ་ཀྱང་། །ཁྱོད་ལ་མངོན་དུ་ཕྱོགས་པར་འགྱུར། །དངོས་གྲུབ་ཐོབ་ཕྱིར་དམ་ཚིག་སྐྱོངས། །གསང་སྔགས་ལ་ཡང་བརྩོན་པར་གྱིས། ་་། རལ་གྲི་་་གཤིན་དགྲ་ནས་བརྩམས་ནས། །ཇི་སྲིད་གཙོ་བོའི་མཐར་ཐུག་པར། །བསྟན་པའི་དཀྱིལ་འཁོར་འཇུག་པ་ཡིན། །།སྤྱི་བོར་ཆུ་ཡིས་གཏོར་བྱ་ཞིང་། །བཛྲ་ཨ་བྷི་ཥིཉྩ་བརྗོད། །དབུ་རྒྱན་དབང་ཡང་དེ་བཞིན་ཏེ། །དེ་ནས་རྡོ་རྗེ་སྦྱིན་པར་བྱ༑་། སངས་རྒྱས་ཀུན་གྱི་རྡོ་རྗེའི་དབང་། །དེ་རིང་ཁྱོད་ལ་སྦྱིན་པར་བྱ། །འདི་དེ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །རྡོ་རྗེ་:ཟུངས་ཤིག་རབ་གྲུབ་ཕྱིར། །རྡོ་་རྗེ་དྲིལ་བུ་འཁྱུད་བྱས་ནས། །བདག་པོའི་དབང་ཡང་བསྐུར་བར་བྱ། །ཨོཾ་བཛྲ་ཨ་དྷི་པ་ཏི་ཏཱཾ། ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། སྤྱི་བོར་གདབ་ཚུལ་མིང་དབང་བསྐུར། །ཏིཥྛ་བཛྲ་ས་མ་ཡ་ཧཱུཾ། ཨོཾ་བཛྲ་སཏྭ་ཏྭཾ་ཨ་བྷི་ཥིཉྩ་མི། །བཛྲ་ནཱ་མ་ཨ་བྷི་ཥེ་ཀ་ཏ། སློབ་དཔོན་གང་དང་གང་དག་གི། །ལྷག་པའི་ལྷ་ནི་གང་གང་གི། །དེ་ནས་དེ་ཡི་མཚན་དག་ནི། །དཔལ་དང་ཀྱེ་ཡིས་མཛེས་པར་བྱ། །།རིག་པའི་དབང་ལྔ་བྱིན་ནས་སུ། །རིག་སྦྱོར་བརྟུལ་ཞུགས་ལུང་བསྟན་དང་། །དབུགས་དབྱུང་་་་་དང་ནི་གསུམ་དུ་འདོད། །ཧཱུཾ་གིས་རྡོ་རྗེ་དེ་བསྐྱེད་ལ། །འདི་ནི་སངས་རྒྱས་ཐམས་ཅད་དེ། །ཕྱག་ན་རྡོ་རྗེའི་ལག་ན་འདུག། ཁྱོད་ཀྱིས་རྟག་ཏུ་གཟུང་བར་བྱ༑། ཕྱག་ན་རྡོ་རྗེ་བརྟུལ་ཞུགས་བརྟན། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་སིདྡྷི་བཛྲ་ས་མ་ཡ་ཏིཥྛ་མཾ། ཨེ་ཥ་ཙཱ་དྷ་ར་ཡཱ་མི་བཛྲ་སཏྭ་ཧི་ཧི་ཧི་ཧི་ཧཱུཾ། ཐོག་མ་ཐ་མ་མེད་པ་ཡི། །རྡོ་རྗེ་སེམས་དཔའ་དགའ་བ་ཆེ། །ཀུན་དུ་བཟང་པོ་ཀུན་བདག་ཉིད། །རྡོ་རྗེ་སྙེམས་བདག་བདག་མཛོད་ཅིག། ཆོས་ཀྱི་དྲིལ་བུ་གཟུང་བར་བྱ། །འདི་—།
