།རྒྱུད། མིའིབློ་ཅན་ཐམས་ཅད་ཀྱིས། །སངས་རྒྱས་ཞིང་ཁམས་དག་ཐོབ་ཤོགབ་། །གཤིན་རྗེ་་་གཤེད་ནག་པོའི་:སྒྲུབ་ཐབས་ཀྱི་མཎྜལ་གྱི་ཆོ་ག་ཞེས་བྱ་བ་རྫོགས་སོ། །དེ། །ན། ཟངས་གླིང་རྣལ་འབྱོར་ནཱ་ག་པུ་ར་ཡིས། །རྣལ་འབྱོར་:ཀ་མ་རཀྱི་ཏ་་་ཉིད་ལ། །:གུ་ཚ་རྒྱལ་བུའི་བུ་ཉིད་བཏད་ནས་ཀྱང་། །ཏ་རུ་པ་རི་དུར་ཁྲོད་ཚལ་ཉིད་དུ། །རྣལ་འབྱོར་འདུས་པའི་ཚོགས་ཀྱི་འཁོར་ལོ་ནང་། །གསོལ་་་བཏབ་རྣལ་འབྱོར་རྣམས་ཀྱི་ཞལ་ངོ་རུ། །མཎྜལ་ཆོ་ག་བསོད་སྙོམས་པ་ཡིས་བྱས། །དེང་སང་རྣལ་འབྱོར་ལྟ་སྤྱོད་ལྡན་པ་ཡིས། །དུས་རྣམས་རྟག་ཏུ་མཎྜལ་འབད་་་་པར་རིགས། །དེ་བ། །ཚོར་བ། །ཁ་་nfཁྲིད་ཀྱི་པ་དང་། བློ་ཆུའི་ཅང་དབྱི་ཟླ་བ་པ་དར་བའིབསྡུར་མཆན། ཁོང་སྤྱི། །སྤྲོ་དགའ་པོ་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»«སྣར་»༄༅། །གཤིན་རྗེ་གཤེད་ནག་པོའི་སྒྲུབ་ཐབས་ཀྱི་མཎྜལ་གྱི་ཆོ་ག་པདྨ་སྲུང་བས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་»སྤུས། «པེ་»«སྣར་གཞག་«པེ་»«སྣར་»ལའི་ལོ། «པེ་»«སྣར་བསགས། པི«པེ་»«སྣར་»དུས་དང་བསྲུང་། གྲོ་མ་དང་: ེ«པེ་»《སྣར་ནས། «པེ་»«སྣར་པས། «སྣར་»ཏཱི། ®«པེ་»སྣར་»+ཧཱུྃ། «པེ་»《སྣར་»ཛིཀ«པེ་»«སྣར་»དྷིཀ«པེ་»«སྣར་»ལིག«པེ་»དྷཱིཀ་སྭཱ་ཧཱ། «སྣར་»དྷིག་སྭཱ་ཧཱ། །«པེ་»«སྣར་ར། འགྱེ«པེ་»«སྣར་བཤགས། «པེ་»«སྣར་»མཎྜ་ལ་ཧཱུཾ། ཨོཾ་བཛྲ་རེ་ཁེ་ཧཱུྃ། བར་«པེ་»«སྣར་ནང་། «པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»དབུལ། «པེ་»«སྣར་»མོངས། «པེ་»«སྣར་གིས། ::ངོག་་རྒྱལཀྱིཀྱིའི«པེ་»«སྣར་»—སྒྲུབ་ཐབས་ཀྱི། «པེ་»«སྣར་པ། ::«པེ་《སྣར་མ་ཀ་ར་ཏྲི། ཆེ«པེ་»«སྣར་བུ་ཚ་རྒྱལ་བའི། །«པེ་»«སྣར་ཀྱིས། པ་དང་། དེར་དབྱེ་བའང་། རྩེ་དྲཚོ། པ་::ཀྱིབདས་པ། པཁོ། དུ། ཉགྲོ་བ་ག་ཚོཊྚ། ལ། ་་འགན་སློབ་ཏུ་། དག་་::::རྒྱལ་། [དོན་དེ་ལ་ཁག་ད་བ་།
