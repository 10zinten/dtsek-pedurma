མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། «པེ་»«སྣར་»ཡིས། «པེ་»《སྣར་པས། ་ཡོ«པེ་»«སྣར་འབུལ། «པེ་»«སྣར་»བོས། @«པེ་»«སྣར་རྗེས། བོད་«པེ་»«སྣར་གྱི་རྩ། :<«པེ་»«སྣར་བསྟན་པ་ཉམ། «སྣར་»གྱིས། »«པེ་»«སྣར་། སུ«པེ་»«སྣར་ཡ། @«པེ་»«སྣར་»སྒོ། ་@«པེ་»«སྣར་ཡིས། («པེ་»«སྣར་མནའ། «པེ་»«སྣར་རྟོག་པ། ད་པ་ཉིད་ཀྱིས་ན་རྒྱུ་མའོ། །བྱེད་པ། ལལ་ལེ་བ། །དཀྱིལ་ཀུ། །་བྱེད་པལ་པ་པོ་ཕྱི་བསྡུར་འབྲས་རེའུ་མིགབ་ཀྱིས་ད་དུང་ཁྲོད་ན་དེ་དག་རྒྱུད། (སྣར་ཐང་། ཅོ་ནེ། པེ་ཅིན། རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། /དཔེ་གཞི། སྡེ་དགེ། སྡེ་ཚན། །རྒྱུད། པོད་རྟགས། མི་ཚན་གྲངས། །ལྡེབ། བན་ཕྲེང་། །ཚིག་རྐང་། །ཚེག་ཁྱིམ། ནཔན་—བནབ་བ?ན?ད་སན་་། :ལ་ཁྲི་འདུ་ཤེས་འགྱུ་དོ་ལས་ཀ་ལ་དཔལ་ན་ལོ་འཁ་ཏཁོད་ཉི་དབང་པོ་ཞུ་འོ་ཊི། །ལ་ན་་:་་ས། ་་་་་། བྱང་དོན་དོར་བ་གཙི་ནད་:ཌོཀ་སོགས་སུ་དེ་ནི། །ལོག་འཚོ་བ་དང་རྒྱལ་པའི་ཀ་བ་དང་། དབོ་བ་དཀ་ཁ་ཁ་ཐུག་འགན་གྱི་བྱ་བ་།
