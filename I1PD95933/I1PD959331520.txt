བསྟན་འགྱུར། འདུལ་བ། ་ལུ་ལུགས་དད་མོས་་མེད་པ་དར་ལས་བསྡུར་མཆན། ཚེས་་་་་བབདེ་བ(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་ལ་༄༅། །ལས་བརྒྱ་རྩ་སློབ་དཔོན་ཡོན་ཏན་འོད་ཀྱིས་མཛད་པའོ། །ཞེས་འཁོད། «ཅོ་ལས། «པེ་»«སྣར་»སྔོན་དང་དགེ་བསྙེན་བསླབ་པ་དང་། །ཞུ་དང་མཁན་པོ་དང་པོ་དང་། །རབ་བྱུང་དགེ་ཚུལ་གྲིབ་ཚོད་དང་། །བསླབ་པ་བརྗོད་པ་ཐ་མ་ཡིན། །དང་པོར་བྱ་དང་་་མཁན་པོ་དང་། །ཆོས་གོས་གཉིས་དང་ལྷུང་བཟེད་དང་། །སྤྲོ་བྱ་གསོལ་བ་གསང་སྟོན་་་་དང་། །ཞུ་དང་གསོལ་དང་གསོལ་བ་དང་། །དྲི་དང་ལས་དང་གྲིབ་ཚོད་་་་དང་། །ཉིན་མཚན་དུས་ཚོད་གནས་རྣམས་དང༌། །ལྟུང་དང་དགེ་སྦྱོང་མཆོག་འདོད་དང་། །ཚུལ་ཁྲིམས་མཉམ་དང་ཚུལ་འབྲེལ་དང༌། །དུལ་དང་དགོས་དང་མ་བརྗོད་དང་། །གུས་པ་དང་ནི་བསྒྲུབ་བྱའོ། །«པེ་《སྣར་གཟུང་། ཚེ«པེ་»《སྣར་གཅུག། པས། །ཤེད་«པེ་»«སྣར་—འདི། ཀྱེ«པེ་»«སྣར་དྲི། འི«པེ་»སྣར་»ཞིག«ཅོ་»བཟུང་། «པེ་»གཟུང་། «པེ་»«སྣར་»བླང་བས། «པེ་»«སྣར་»«ཅོ»བཟུང་། ཕྱིལོ། བྱ་བ་«སྣར་»གིས་སླབ། «པེ་»«སྣར་རྫུན། «པེ་»«སྣར་བསླབ། བདག་«ཅོ་བར། «པེ་»«སྣར་གཎྜི། «པེ་ནས་ཁོད། «སྣར་+པ། །«པེ་»«སྣར་དེ་ནི། «པེ་《སྣར་»གྱིས། «པེ་»«སྣར་བྲེག་«པེ་བྲེག། «པེ་»«སྣར་»ཞིག«པེ་»«སྣར་»མཚན། («པེ་»«སྣར་»སྤོང་། «པེ་»«སྣར་»འབྱུང་། «པེ་»«སྣར་བླང་བས། ®«སྣར་»སྐྱོར་བ། །«པེ་»«སྣར་»—བདག《སྣར་ཚིགས། «པེ་»གྱིས། «པེ་»«སྣར་»སྦྱིན། །«པེ་»«སྣར་རྫུན། །(«པེ་»«སྣར་རྫུན། «པེ་»«སྣར་»ལ། «པེ་《སྣར་»གིས། «པེ་»«སྣར་བསྒྲུབ། «པེ་»《སྣར་»འདུ། «པེ་《སྣར་བསྡམས། ཀྱི«པེ་བཏག«པེ་»«སྣར་»བརླབ། «པེ་《སྣར་»ཁོབ། «པེ་»«སྣར་པོ། །(«པེ་བཟུང་། 《སྣར་བ་བཟུང་། «སྣར་བར། «སྣར་བར། «སྣར་པོའི། («པེ་»«སྣར་གཟུང་། «པེ་»«སྣར་»—འདི། («སྣར་»བ། ®«པེ་»«སྣར་»བ། —།
