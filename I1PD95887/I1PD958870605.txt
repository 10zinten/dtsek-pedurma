་བོ་བསྟན་འགྱུར། རྒྱུད། ཝུ་བོད་ཀྱི་པའི་རྒྱུད་སྙིང་རྡོ་རྗེའི་ཚིག། གསལ་བར་སྟོན་བྱེད་རྒྱ་ཆེར་འགྲེལ། །བྱས་པའི་བསོད་ནམས་གང་ཐོབ་པའི་ཤིན་ཏུ་དྲི་མེད་རྒྱ་ཆེན་པོ། ་པད་མ་དཀར་པོ་ལྟ་བུ་དེས། ཁ་། རྡོ་རྗེ་འཛིན་གྱི་ལམ་རྙེད་ནས། །འགྲོ་ཀུན་རྫོགས་སངས་རྒྱས་་་གྱུར་ཅིག། །།དཔལ་དུས་ཀྱི་འཁོར་ལོའི་རྒྱུད་ཀྱི་སྙིང་པོའི་རྒྱ་ཆེར་འགྲེལ། དྲི་མ་མེད་་་་པའི་འོད་ཅེས་བྱ་བ། འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ། །།།།སམ་པ་ལའི་ཡུལ་བྱོན་ཞིང་། །མངོན་པར་ཤེས་པ་མངའ་བརྙེས་པ། །དུས་ཀྱི་འཁོར་ལོ་པ་ཞེས་པ། །དེའི་རིམ་ནི་བརྒྱུད་པ་དང༌། །ཨུ་རྒྱན་ཡུལ་སྐྱེས་བྲམ་ཟེའི་རིགས། །ན་རོ་པ་ཞེས་བྱ་བ་ནི། །ནུས་པ་སྐྱེས་ཤིང་ལྷས་གནང་བ། །དེའི་རིགས་རྒྱུད་ཐོས་པའི། །བླ་མ་ལྷ་ཡིས་ལྷས་བསྟན་པ། ་། དེ་མཉན་་་་་འགྲེལ་འབད་བསྒྱུར་བས། །མི་འཇིགས་བདེ་བའི་གནས་ཐོབ་ཤོག། །།རྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་བྷ་དྲ་བོ་དྷིའི་ཞལ་སྔ་ནས། ་བོད་ཀྱི་ལོ་ཚ་བ་གྱི་ཅོ་བན་དེ་ཟླ་བའི་འོད་ཟེར་གྱིས། ཡོན་བདག་ཞང་བཙུན་དགེ་སློང་གི་ཕྱིར་བསྒྱུར་ཅིང་ཞུས་པའོ། །པ་དང་པ་དེ་བདེན་པ་ང་ལ་མི་ཁ་དགོ་བ། ཁ་ཕྱོགས་ཁ་ལ་Sབྱའོ། །ཁྲི་་གནས་་་་:སྐྱེའབདལ་ཞེ་ན། ས་་་ལོའི་་་་་་ལ་ཁ་འགྲོ་པའི་ཁ་ན་མེན་པའི་ན་—།
