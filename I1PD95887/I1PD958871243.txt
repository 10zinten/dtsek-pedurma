རབསྟན་འགྱུར། རྒྱུད། [ཞུབ་སན་ཏ་ཏི་ལ་ཀ་ལས། གནས་ལ་སོགས་པའི་ས་བཅུའི་ཡུལ་གྱི་དབྱེ་བ་ངེས་པར་བསྟན་པ་སྟེ་ལྔ་པའོ། །ཁ་ཊི། ངས་པ་དང་ད་ག་བ་ལྟར་དུ་བདག་དུ་མ་ཎག། དགོན་ས་བྱེད་དུ་ཁཆུབ་བདེ་ལ་ཁ་མ་གཏོང་ཀོ། དེ་ནས་གཞན་ཡང་དེ་བཞིན་གཤེགས་པའི་སྐུའི་གནས་ཇི་ལྟ་བ་བཞིན་དུ་་་སོན་པའི་རྩའི་འཁོར་ལོ་བསྟན་པར་བྱ་སྟེ། སྙིང་གའི་དབུས་གྱུར་པདྨ་ནི། །ལ་། འདབ་མ་བརྒྱད་དང་གེ་སར་བཅས། །དེའི་དབུས་སུ་གནས་པའི་རྩ། །།མར་མེ་ལྟ་བུའི་གཟུགས་ཅན་ཏེ། །ཆུ་ཤིང་གི་ནི་མེ་ཏོག་ལྟར། །རྟག་ཏུ་འཕྱང་ཞིང་་་འོག་ཏུ་ལྟ། །དེའི་དབུས་བཞུགས་དཔའ་བོ་ནི། །ཉུངས་ཀར་སྦོམ་པོའི་ཚད་་་ཙམ་སྟེ། །ཧཱུྃ་གི་ས་བོན་མི་ཤིགས་པ། །ཁ་བ་ལྟ་བུར་འབབ་པ་ཡིན། །བ་སན་ཏ་ཞེས་བྱ་བ་སྟེ། །ལུས་ཅན་རྣམས་ཀྱི་སྙིང་དགའ་བྱེད། །རྒོད་མའི་ཁ་ནི་མེའི་གཟུགས། །ཕག་མོ་ཏི་ལ་ཀ་ཞེས་བརྗོད། །ལས་ཀྱི་ཀླུང་གིས་བསྐྱོད་ནས་སུ། །ལྟེ་བའི་དཀྱིལ་འཁོར་ནས་ནི་འབར། །བ་སན་ཏ་དང་ཕྲད་ནས་མཉེས། །སྙོམས་པར་འཇུག་པས་ཡང་དག་བཞུགས། །འདི་ནི་ཤྲཱི་ཧེ་རུ་ཀའི། /། བ་སན་་ཏི་ལ་ཀ་ཞེས་བརྗོད། །རྣལ་འབྱོར་མའི་གཟུགས་བཟུང་ནས། །རྒྱུ་དང་མི་རྒྱུ་་ཡང་དག་བཞུགས། །སྐུ་གསུང་ཐུགས་ཀྱི་བྱེ་བྲག་གིས། །སྒོ་གསུམ་ནས་ནི་ཡང་དག་རྒྱུ། །འགྲོ་བ་དང་ནི་འོང་བ་རུ། །དེ་དག་ཐམས་ཅད་ལུས་ལ་གནས། །ལྟེ་བར་ཨའི་གཟུགས་ཀྱིས་ནི། །ཐུང་ངུ་ཞེས་ནི་བྱ་བར་གྲགས། །སྙིང་གར་་་་་གནས་པའི་ཡི་གེ་ཧཱུཾ། །།རིང་པོ་ཆ་གཉིས་ཚད་དུ་བརྗོད། །མགྲིན་པར་ཨོ་་་་གཟུགས་ཀྱིས་ཏེ། །ཆ་གསུམ་འདྲེན་པར་ཡང་དག་བརྗོད། །དཔྲལ་བར་གནས་པའི་ཡི་གེ་ཧཾ། །སྒྲ་དང་ཐིག་ལེའི་མི་ཤིགས་པ། །འཁོར་ལོ་བཞིའི་བྱེ་་་བྲག་གིས། །ས་སྩོགས་འབྱུང་བ་ཆེན་པོའི། ར་། འཚམས་བཞི་རུ་ནི་ཡང་དག་—།
