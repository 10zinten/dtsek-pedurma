ཁེ་སྤྱི་བསྟན་འགྱུར། རྒྱུད། ཌོ། བ་ནི། པྲ་ཏི་པོ་ཏངྒི་སྒྲ་བརྗོད་པས། མངོན་པར་ཕྱག་འཚལ་བའི་ལན་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ལུ་མུའི་སྒྲ་འདི་ཡང་མ་ཡིན་ཏེ། །ལུམྦའི་སྒྲའོ། །ཁ་་་ཅིག་ན་རེ་མན་ངག་ལས་ལུམ་བུ་ཞེས་ཟེར་རོ། །འོང་བ་ཞེས་འདི་སྐད་དོ་ཞེས་བྱ་བ་ནི། འོང་བ་ཞེས་འདི་སྐད་ཅེས་བྱ་བའི་དོན་ཞེས་བྱ་བའི་དོན་ཏོ། །སློང་བ་ཞེས་བྱ་བ་ནི། བྱིན་ཅིག་ཅེས་བྱ་བའི་དོན་ཏོ། །ཅེ་ཊཱ་ཀི་ནི་ཁྱེར་ཅིག་པའོ་ཞེས་བྱ་བ་ནི། སྣོམ་ཞིག་ཅེས་བྱ་བའི་དོན་ལ་ཙ་ཊུ་ཞེས་བྱ་བའི་སྒྲའོ། །ཧྲི་ད་ཡོ་ནི་དཔའ་བོ་ཞེས་བྱ་བ་ལ། བྲྀ་ད་ཡོ་ནི་དཔའ་བོ་ཞེས་བརྗོད་དོ། ::། ཀོ་ར་བི་ནི་གསད་པའོ་ཞེས་བྱ་བ་ནི། ཀོ་ར་ཁཾ་གསད་པ་ཞེས་བརྗོད་དོ་ཞེས་གསུངས་སོ། །ཀཎྜི་ཀ་ནི་དྲིལ་བུའོ་ཞེས་བྱ་བ་ནི། ཀཉྩི་ཀ་ཞེས་བརྗོད་པར་བྱེད་པའོ། །ཨ་ལི་ཀ་ཀ་ར་ཎ་ནི་སྤྱི་བོ་ཞེས་བྱ་བ་ནི་མགོ་བོ་ཞེས་བྱའོ། །བ་ར་ཧསྐྲའོ་ཞེས་བྱ་བ་ནི་སྐྲའི་མིང་ངོ༌། །མནྡྷ་ནི་བདུད་བརྩིའོ་ཞེས་བྱ་བ་ནི། བདུད་རྩི་མནན་ཞེས་་བརྗོད་དོ། །ས་མཱ་ག་མ་ནི་ཞེས་བྱ་བ་ནི་སྐྱེས་པའི་ཞེས་བྱ་བའོ། །ཏཱ་ལི་ཀཱ་ནི་མཁའ་འགྲོ་མའོ་ཞེས་བྱ་བ་ནི་མཁའ་འགྲོ་མའི་མིང་ངོ༌། །པའོ། ་ཝ་ཌ། ་དམིགས་གནོད་པ་ཞེས་བྱ་བ་ནི་འཇིག་པ་སྟེ། འགང་ལ་བདུག་པ་ཐམས་ཅད་་་གནོད་པ་དེ་ནི་འཁྲུལ་འཁོར་གྱི་ལེའུ་འོ་ཞེས་བྱ་བ་སྒྲས་འདིའི་དུས་ཀྱིས་ཞུ་བ་་་རྫོགས་པ་སྟེ། འཁྲུལ་འཁོར་གྱི་ལེའུ་བསྐྱེད་དུ་གསོལ་ཞེས་བྱ་བར་སྦྱར་རོ། །དངོས་གྲུབ་གྱིའང་བསྐྱེད་པ་ཡིན་ཏེ། འན་གང་གི་ཕྱིར་བདག་ཉིད་མངོན་དུ་བྱེད་པའི་འང་བསྐྱེད་པ་ནི། དཔེར་ན་བུ་བསྐྱེད་པ་ནི་མ་བྱུང་བར་ཡང་བསྐྱེད་དོ། །དཔེར་ན་བུ་སྐྱེད་པ་ལྟ་བུར་འདིར་བདག་ཅག་རྣམས་ཀྱི་འཁྲུལ་འཁོར་གྱི་ལེའུ་་་བདག་གི་མངོན་དུ་མཛད་ཅིག་ཅེས་བྱ་བའི་དོན་ཏོ། །དེའི་ཕྱིར་འཁྲུལ་འཁོར་གྱི་།
