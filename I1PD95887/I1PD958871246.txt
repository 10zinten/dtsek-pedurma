མའི་དབང་ཕྱུག་མཆོག་། འདུ་ས་ཞེས་ནི་བྱ་བར་བཤད། །མཁའ་འགྲོ་མ་དང་སྦྱོར་བའི་མའོ། །སོ་རཋར་གླང་སྣ་མ། །རྟག་ཏུ་ཁྲག་ནི་གནས་གྱུར་དང༌། །གསེར་གླིང་ལ་ནི་རབ་རྟེན་པའི། །འཁོར་ལོས་སྒྱུར་མ་ཞེས་བྱ་གང༌། །འབར་བས་ཀུན་དུ་འཁྲིགས་པ་ནི། །རྡུལ་ནི་རབ་ཏུ་འབབ་པའི་རྩ། །ཉེ་བའི་འདུ་ས་ཞེས་བཤད་པའི། །འོ་། མཁའ་འགྲོ་མ་དང་སྦྱོར་བའི་མའོ། །གྲོང་ཁྱེར་ཤིན་ཏུ་་དཔའ་མོའི་ནི། ་ྋ། ཚིལ་དང་སྦོས་མོའི་ལུས་ཅན་དང་། P། མྱ་ངན་དང་བཅས་་་་མཆི་འཛག་པའི། ཎ། སིན་དུ་སྟོབས་ཆེན་ཞེས་བྱ་བ། ཚོ། །དུར་ཁྲོད་དུ་ནི་གནས་པ་ཡི། །མཁའ་འགྲོ་མ་དང་སྦྱོར་བའི་མའོ། །མ་དོར་འཁོར་ལོས་སྒྱུར་མ་ནི། །།ལུད་པ་སྐམ་པོ་འབབ་པ་དང་། །ཀུ་ལུ་ཏར་ནི་བརྩོན་ཆེན་མ།  ། ར། སྣ་ཆུ་འབབ་་པ་རྣམ་པ་གཉིས། }༌། ཉེ་བའི་དུར་ཁྲོད་ལ་གནས་པའི། །མཁའ་འགྲོ་མའི་སྦྱོར་བའི་མའོ། ་། སྐུའི་འཁོར་ལོའི་མཁའ་འགྲོ་མ། །ཕྱི་རོལ་ས་འོག་གནས་པ་ཡིན། །སྔར་བསྟན་ངག་གི་སྒོ་བཞི་ལ། །།མཁའ་འགྲོ་མའི་རྣམ་པ་བཞི། །དེའི་མཚམས་ཀྱི་རྩ་བཞི་ལ། །གཞད་དག་བཞི་པོ་བརྟེན་ཏེ་གནས། ཊ། རྩ་བའི་སྒོར་ནི་ཁཱ་་་་གདོང་མ། ཎཚ་། གཉིས་པ་འུག་པའི་གདོང་ཅན་བཞུགས། །གསུམ་པ་ལ་ནི་ཁྱི་་གདོང་སྟེ། །བཞི་པ་ཕག་གི་གདོང་ཅན་ནོ། #། །མཚམས་རྣམས་གཤིན་རྗེ་བརྟན་མ་དང་། ་་། དེ་བཞིན་གཤིན་རྗེ་ཕོ་ཉ་མ། །་། གཞན་ཡང་གཤིན་རྗེ་འཇིགས་ཆེན་དང་། །གཤིན་རྗེ་མཆེ་བ་ཞེས་བཤད་དང་། །བཞི་པ་ཞེས་བྱ་འཇིགས་བྱེད་མ། །གཤིན་རྗེ་འཇོམས་མ་དག་ཏུ་འདོད། །འདི་དག་རྡོ་རྗེ་མཁའ་འགྲོ་མ། །རྩའི་་གཟུགས་ཀྱིས་ཡང་དག་གནས། །སེམས་ཀྱི་རྡོ་རྗེ་རྣལ་འབྱོར་མ། །སུམ་ཅུ་རྩ་་བདུན་དག་ཏུ་འདོད། །ཤེས་རབ་ཕ་རོལ་ཕྱིན་མ་སྟེ། །ཎ། དེ་དག་ལུས་ལ་ཡང་་་དག་གནས། ཌ། ཌ་ཟ། ཀ་ཎིཊ། ཏེ་དུ་ཏོ་ཅ་ལྟོ་ཏྲ་ཊཱི་ཀ་་་ཁརྒྱུ་འཁ་ཁ༌ར++།
