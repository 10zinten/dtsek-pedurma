ས་ག་ས་ག་ལ། །ང་རྒྱལ་རྡོ་རྗེས་དབུ་རྒྱན་དེ་ལས་དུ་བ་མི་མང་བ་ང་ལ་ཕྱག་འཚལ་ལོ། །ཤེས་རབ་ཕུན་སུ་་རེ། དེ་དག་ལས༌སྐྱེ་བ། ཤོ་མ་ཆགས་དོ་རྗེས་བརྒྱན། །སོ་ལས་ནད་ལ་ཕྱག་འཚལ་ལོ། །ར་ཤུའིཀྱི་ཡོད། ང་དང་ལྷ་ན་བག་ལ་ལྟ་བ་༄༅། །བདག་མེད་མའི་སྟོབས་ཀྱི་འཁོར་བ་པ། །ལོ་སྒྲུབ་པའི་ཐབས། ས་འོག་གི་འབྲི་ན་དེ་ན་ལུས་པ་མཁན་མ། །ཚེ་ཆུ་ཁོ་ར་ཁྲོད་འཇིག་པར་པས་བརྒྱན། །ཤིབ། རབ་:P། ཨུ་ལ་ཀཱ་ལི་ཟླ་བ་ཉི་མ་ལ་སློང་མ་ངམ་ད་ལྟགླིང་། ཤིན་ཏུ་༄༅། སྟོབས་ཀྱི་འཁོར་ལོ་བསྒོམ་པའི་ཐབས་བཞུགས་སོ། །༄༅། །ཨོཾ་སྭཱ་སྟི། ་་རྒྱ་གར་སྐད་དུ། ནི་རཱཏྨ་བ་ལ་ཙཀྲ་སཱ་དྷ་ན། བོད་སྐད་དུ། བདག་མེད་མའི་་སྟོབས་ཀྱི་འཁོར་ལོ་སྒྲུབ་པའི་ཐབས། རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་ལོ། །སྒོམ་པའི་རིམ་པ་རྒྱས་པ་འམ། །སྐད་ཅིག་རྣམ་པས་རབ་འགྲུབ་སྟེ། །ཡང་ན་འཁོར་ལོ་རྫོགས་པའི་ལྷ། །གཞན་དུ་ཀྱང་པའི་སྦྱོར་བས་ཏེ། མ། ལྷན་ཅིག་སྐྱེས་་དགའི་གཟུགས་ཅན་མ། །སྐྱེ་གནས་འཁོར་ལོ་རྩིབས་དྲུག་པ། །ལྟེ་བའི་གནས་སུ་འབད་པས་བསྒོམ། །དབུས་སུ་བདག་མེད་རྣལ་འབྱོར་མ། །དེ་ཡི་སྙིང་གར་ཨོཾ་ཡིག་གོ། །ཨ་ཨི་ཨུ་ཨེ་ཨོ་ཨོཾ་ལས། །།ཤར་སོགས་གཡས་སུ་བསྐོར་བ་ཡི། །གོ་རཱི་རྩོ་རཱི་དེ་བཞིན་དུ། །བེ་ཏ་ལི་དང་གྷསྨ་རཱི། །སྦོས་མོ་རི་ཁྲོད་མ་ཡང་སྟེ། །རྣལ་འབྱོར་མ་དྲུག་འགོད་པ་ཡིན། །ནག་དང་དམར་ལ་སོགས་དེ་བཞིན། །རང་གི་ས་བོན་དག་གིས་མཚན། །བདག་ཉིད་ཉི་མའི་ལུས་དེ་བཞིན། །ཟླ་་་་།
