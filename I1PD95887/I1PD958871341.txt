སྐྱབསྟན་འགྱུར། །རྒྱུད། ཞུསཀ་ཁའི་བཛྲ་ལི་ལ་ཡ་བཛྲ་ཁང་གསར་བ་སློབ་དཔོན་ལོ་ཙ་བ་དགེ་བསྙེན་གྲགས་པ་རྒྱལ་མཚན་བདག་གིས་ནོས་སོ། །འདི་མན་་་་ཆད་མེད་པས་གཏུབ་ཤཱཀྱ་རཀྴི་ཏ། ཀ་ན་ཀ་ཤི༄༅། །ཛ་ཡ་ལེ་ན། ཕྱིས་རྒྱ་གར་དུ་་་་བྱོན་ནས་བླ་མ་ཨ་ཉིད་ལ་གསན་ཕྱིས་མི་ཁྲི་བ་ཉིད་ཀྱི་སློབ་མ་ཀ་མ་ལ་བཛྲ་བྱ་བ་ལ་བླ་མ་ཁང་གསར་བས་་་་་གསན་ནོ། །་། ་ལྟ་བ་ག་བ་ག་བ་ག་བའི་ས་ཡའོ། །་་་་་་ལ་བ་ན་དཔའ་དར་བའི་དུ་དེད་ཁ་ཁ་ཁུབ་ཡོང་། །ཏག་ལ་དག་འདི་ན་བསྡུར་མཆན། ང་བལྟ་ཚོ་ཁ་ཧྥ་་་་མཚན་བྱང་འདི་«པེ་»«སྣར་ལྟར་བཀོད། «པེ་བཅས། «པེ་ནས། །«པེ་»སྦྱོར། །«པེ་ཤུདྷཱ། «པེ་»ཏིཙྩཾ། «པེ་»ཤི། »ཐོན་པ་། aཊི་ཀ་བ་གྱུཎི་པ་ཕྲག་གི་ར་བ། ག་ད་བ་མ་འ་ལ་བ་ལས་པའི་ཕྱག་མཚན་བཅིང་བ་ག་ཎ༌ཥ་ས་::ཁ༌ཁ། །།ཟླ་འོད་བློ་ཁ་བྱང་གསར་པ་ཞབསྡུར་འབྲས་རེའུ་མིག། བརྒྱུན་བཞི་བཞི་རུང་གློག་ཤ། རྒྱུད། （སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། སྔགས། སྔགས། ཕུདཔེ་གཞི། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། །ཕྲེང་། ཚེག་ཁྱིམ། །བར་ནན་ནན། الدمདེ་བས་ད་ཚེ་ཀ་ཏ་དཀྱིལ་བ་སོགས་བཅད་ལའོ། །དེ་བག་པ་དང་ན་བ་དང་། གབསྒོལྟེ་ཀ་:འང་ཁ་པ། །བྱོར་པ་ཁོ་གད་པར་སྤྲོ་དེ་།
