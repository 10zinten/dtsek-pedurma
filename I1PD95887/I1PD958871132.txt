ཀ་ལ་འཁ་ས་ཁ་ཁ། །ཎིང་ཆ་ཅཚོཊ་ཋ་ng་ཊིཀ་ཁ། །ཁཞུ་། །ཆོ་ཏུ་ཁ་ཚོ་ཁཁ་ཊི་ཟླ་ལ། ང་དམུ་ད་ཀ་ཏཁབླ་བློ་ཁྲོ། །enp། ཁ་འཁྱོ། ཁ་ཆུ། །ཎཊིཀ་ལོ་ཀོཏུ་འཁ:ཞེ་ཌའོ། །ཀ་ད་ཀ་ཎ་སྐྱཁ་ཁ་བའིਸ਼ਸਵਾਸੈਲਇੰpਨਲ'ਚਉਸཁྱའི་p། p@་རཀ་དོང་པ། །།ཀལ་ཀ༌བླ་ནཔཀ་༄༅། །དཔལ་འཁོར་ལོ་སྡོམ་པ་ལ་བསྟོད་པ། z། ཁ་ཀ་ང་འོ། །ཏཊི་ཏ་ཚེ་ཎཆུ་ཁཚོ་ཏང་ཀ་ར། །ཊཏུ་ཁ་ཆུ་ཁ་བ་ཀ་ཏན:/ཀ་ཀ་ཀ་ར་ཎཌ། བབ་བླ་དང་རྒྱུ་ཆ། །།གpཔ་ཁས་ཀྱང་ང་ལ་ཁ་RABਇਸੌਬਸਇਮੌਲੁgਸ਼ਕ।ਉਸ། ཆོས་བརྒྱུའི་རྒྱུ་སུ་༼པཎཱ། ག་ཎཊོ་ལCཀླ་ཀ་ཁ་ཆ། ༄༅། དཔལ་འཁོར་ལོ་སྡོམ་པའི་བསྟོད་པ་བཞུགས་སོ། །༄༅། །རྒྱ་གར་སྐད་དུ། ཤི་ཙཀྲ་པོ་བྷ་ར་སྟོད་ཏྲ། བོད་སྐད་དུ། དཔལ་འཁོར་ལོ་སྡོམ་པ་ལ་བསྟོད་པ། དཔལ་འཁོར་ལོ་སྡོམ་པ་འཁོར་དང་བཅས་པ་ལ་ཕྱག་འཚལ་ལོ། །གང་ཞིག་རྒྱལ་བ་ཀུན་གྱི་དངོས། །ཧེ་རུ་ཀ་ཡི་དབང་ཕྱུག་གཙོ། །བཅོམ་ལྡན་མཁའ་འགྲོ་་ཀུན་གྱི་དབུས། །རྟག་ཏུ་བཞུགས་པ་ཁྱོད་ཕྱག་འཚལ། །གང་སྐུ་ཨུད་པལ་སྔོན་པོ་འམ། །འཐོན་འཐིང་བཻ་ཌཱུརྱ་འདྲ་བ། །བདུད་བཞི་འཇོམས་ཕྱིར་ཞལ་བཞི་སྟེ༑། རྩ་བའི་ཞལ་ནི་ནམ་མཁའི་མདོག། གཡས་ཞལ་གསེར་སྦྱངས་གཙོ་མ་འདྲ། །གཡོན་ཞལ་སྲི་ཡི་ཀུ་ལྟ་བུ། །རྒྱབ་ཞལ་པདྨ་ར་གའི་མདོག། རྩ་ཞལ་སྒེག་པའི་ཞལ་དང་གཡས་ཞལ་རྒོད་པའི་ཞལ། །རྒྱབ་ཞལ་དཔའ་བའི་ཞལ་དང་གཡོན་ཞལ་་་འཇིགས་རུང་ངོ༌། །གར་དགུའི་ཉམས་ཀྱི་རབ་ལྡན་པ། །མགོ་ཕྲེང་ལྔ་བཅུ་རྩ་གཅིག་འཆང་། །སྟག་གི་ལྤགས་པ་གྲོལ་ཚུལ་འཛིན། །ཐོད་པ་རིན་ཆེན་དབུ་་།
