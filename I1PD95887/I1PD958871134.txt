དཔལ་འཁོར་ལོ་སྡོམ་པ་ལ་བསྟོད་པ། །ལ་བས་པར་བྱེད་པ་དུ་བསྲུང་ན་དངོས་གྲུབ་འགྱུར་བ་སྟེ། བདུད་པས་ཆུས་འགྲུབ་པོ། ལོ་བཅུ་་གྲི་ཏ་ད་ག་བ། བོད་པར་བ་གང་དུ་ཕྱིན་ཡོད། དེ་ནས་དམ་ཚིག་བབསྡུར་མཆན། ས་མང་། སློབ་དཔོན་བདཔོ། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཉི«པེ་»སྦྱང་། «པེ་འཇིག་«པེ་»ཡིས། ་«པེ་»ཀྱིས། ཉི«པེ་སྒོ། པ། ལྔ་པ་རབ་བ། །།ར། །།།ས་ལ་དབབ་བྱ་ཞིང་། །འཁོར་ལོ་བ་དུ་མ་བདལ་པ་ཀུན། །རྗེ་འ་དང་བྲལ་བ་ག་བྱ། །སོ་བསྡུར་འབྲས་རེའུ་མིག་སྟེ་བདྲ་བ་ངོ་ཁྲི་ཕཊ། ཨི་ཏི་མ་མ་གྲོ་རྒྱུད། (ཀྱི་དེབ་དུ་དག་ལ་ཡིག་མ་དེ་ནི་རྟ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། ཅོ་ནེ། །སྡེ་ཚན། འདི་སྔགས། ་སྔགས། ་་་ན། བོད་རྟགས། །་ནས་། པུཔུ་བྱ་ཚན་གྲངས། ལྡེབ། །ནབ་བན་ཕྲེང་། ་ཚིག་རྐང་། །ཚེག་ཁྱིམ། །སྡེ་སྔགས། སྔགས། །ལུང་ཤོག་དང་། །ཁབ་མ་འ་ན་དེ་ལྟ་བཌ་ད་བ་མ་འ་དག་པ་སོགས་ལ། ཕྱིན་། །།་་་་་འོག་དོ་དམ་བྱ་བ་དེ་ཡུལ་ཨ་མ་ནས་འདས་ན་བུ་ལྟ་པཎ་ཆུ་འི། །དབང་ཆེན་ལ་ཁ་ཏི། །ཀཀཀུ་བྱ་བ་ག་ན་པ་བ་བ་བལྟ་མཁ་བབ་དོ་བ་ལ་ཞག་ཞེ་པོའཁྱིངས། །ཉི་ཚེ་བ་ཏུན། དེབ་མ་ཤེས་ཕྱོགསལ་བ་་་་་་།
