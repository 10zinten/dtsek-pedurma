རག་ང་ཁྲི་ཁང་བ་ཁང་དང་་བ་ལ་གོ་གང་བསྟན་འགྱུར། ་རྒྱུད། ཀ་དམ་བ་ན་ལ་སྒྲོ་ཆུམ། །བའ་ཞazཚོགས་། དུས་ད་ག་ན་ནང་ཁོང་གི། སྡུད་ལ་གས་ཁག་བསྡུར་མཆན། ཁ་ས་ཉི་ཁའི་ཏི་ཞུ་ཁག་ཀྱི་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»ཀ«པེ་»སྨྲད། «པེ་མོའི། «པེ་»སྐུར། «པེ་སྐྱོད། །«པེ་ཀྲི། །«པེ་»ཤ་«པེ་—ནི། «པེ་»ཙཱ། ་་བ་ལ་ཞི་བ། འཁཚོ། ལྟུན། ང་བར་ਸਤਨਤਾਂਕਿਕਾਇਆਕਸਾਨਕਰਨਦੇਤਸਕਰਮਨབོདྲ་བར་བསྡུར་འབྲས་རེའུ་མིགན། ་་་རྔ་འོ། ::ཀ་ཁག་ཁ་རྒྱུད། （གཞུང་ཞིག་བཀོ་བ་ཀ་ལ་དཔེ་གཞི། །སྡེ་དགེ། ་པེ་ཅིན། །སྣར་ཐང་། ཅོ་ནེ། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བསབནབ། ཕྲེང་། །རར་དུ་བ་། ཚེག་ཁྱིམ། །།སྔགས། སྔགས། E་་་་་་་ཀྱི་དོ་སྐྱོ་བོ་དང་། །དངོས་ཀྱི་ཁྲའོ་དེ་ཁ་བ་ན་། ཀ་ཊའཁོ་བ་བ་། ལུང་བ་ཁ་བ་ཟླཞུ་ཀ་ཏྱཱ། ཟླ་བ་ཁྱོདདེ་ནས་དབང་བསྒྱུར་བ་ལ། དེ་ལ་ཁྱེད་།
