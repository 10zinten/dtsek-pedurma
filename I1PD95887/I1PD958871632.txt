ལ་་བ་མ་ཟ། ཁ་སྟགས་ཁྱབ་ཀྱི་ཁོ་ཁ་ཎ་ཀ་ལ་མི་གཡོ་བའིམར་འཁྲུལ་མེད་པ། གང་བ་ཁ་ཌེ་ཀ་ཚལ་པོ་ར། །མ་ལ་ཁ་ཏི་པོལ་ཁའི་པའི་ཤིཀ། མི་བྱེད་པོ་ས་ཁད་ཀྱི་སྡུག། བང་སྒྲོན་ཞུ་་ལོ་་་་་༌་་་་་་་་་་nཞོ་ཤོས་འདི་ག་ན་ཁྱོད་ལ་ཆ་ཤོར་བ་པའི་ན་བའོ།  ། དཔའ་ཁྱོད་འཚོ་ཚོ། །ལཚེ་nཀོ་ཀ་ཁ་པའི་buང་བ་འབབ་པ། ་ཚོ་བ་ཁྲོད་ཀྱི་ད་དཔེ་ལ་བལ་སྤྱོད་པ་་་། །།།དེ་ལ་བག་མེད་སྤྱིptལྟ་བ་དེ་ལ་ཤ༄༅། །དགེས་པ་རྡོ་རྗེའི་མན་ངག་གོ། །།བློ་ཌོ་བས་དོར་བ་ཚོ་ཅི་མི་དྲུག། །ཁ་མ་ཁག་ལ་གོ་བ་ནི། །ཝཱ་བ་བ་ན་མ་གཏོགས་ཁག་འཚོ་བ་ན་བ་དང་དུལ། །ཎ་ཚེ་རིང་ང་ཚོས་ད་ཀྱི། །འདོད་ཚན་བབཀ་ཁ་ས་ཁཀ་་་༌། །ཨེ་ཤ་ཡི་བོག་ཏ་ཆུ་ཟླ་ཆེ་བ་དང་། P། །དཔ་ལ་ཀཱ་མུ་དང་། ་འམ། pབྱ་ཁག་གཅིག། དེ་དག་ལ་བུ་སྐysཀ་གའི་་་་་་་་༄༅། །རྒྱ་གར་སྐད་དུ། ཧེ་བཛྲ་པ་དེཤ། བོད་སྐད་དུ། དགེས་པ་རྡོ་རྗེའི་མན་ངག་གོ། །པདྨ་བཛྲ། ན་གཱརྫུ་ན། །ཨཱརྱ་དེ་ཝ། །ཡཀྵ་་ཎྜི་སུབྷ་ཀཱི། བྱད་ལི། ་ཨ་ལ་ལ་བཛ྄འོ། ཨ་སིཏགྷན། ཕྱི་མའི་པདྨ་ལྭ་བ་ལགླིཀཾ་ཛ་ལནྡྷར་ཡང་ན་གོང་བཞིན་ནོ། །དཔལ་རྡོ་རྗེ་མཁའ་འགྲོ་ལ་ཕྱག་འཚལ་ལོ། །སེམས་ཅན་ཀུན་ལ་སྙིང་བརྩེ་བས། །ཐམས་ཅད་མཁྱེན་པའི་འབྲས་སྟེར་བ། །བླ་མ་བཟང་ལ་ཕྱག་འཚལ་ནས། །སྦྱིན་སྲེག་མེད་ཅིང་དཀའ་ཐུབ་མེད། །ཡོད་པ་མེད་ཅིང་མེད་པ་མེད། །།རྟག་དང་ཆད་པ་རབ་བྲལ་བ། །འདིར་ནི་བྱ་བའི་ལས་ཀུན་སྤངས། །དཀྱིལ་འཁོར་དཀྱིལ་འཁོར་ཅན་ཡང་མེད། །སྟོང་ཉིད་་་་སྙིང་རྗེ་དབྱེར་མེད་པ། །བྱང་ཆུབ་སེམས་ཤེས་རབ་ཏུ་བརྗོད། །དེ་ནི་སྔགས་་བཟླས་དེ་དཀའ་ཐུབ། །སྦྱིན་བསྲེག་དཀྱིལ་འཁོར་དཀྱིལ་འཁོར་ཅན། །མདོར་ན་སེམས་སྡུས་ཚུལ་ཅན་ནོ། །ལྷན་ཅིག་སྐྱེས་པའི་སྒྲུབ་ཐབས་ནི། །ཛ་ལ་་་ས་ས།
