ཀ་་ལབསྟན་འགྱུར། རྒྱུད། ཞུའི་ཚོ་nཀ་སྤྲོ་ཁྲོ་གསལ་མཛེས་མའོ། ་། ཤེས་རབ་མཛེས་མ་ལ་ཆགས་སྦྱོར་བ་ཡིས། །འདུས་བྱས་་་འདུས་མ་བྱས་པ་ལས་གྲོལ་བ། །བསྐལ་པའམ་བསྐལ་པ་སྟོང་གི་བར་དུ་ཡང་། །ཡེ་ཤེས་འདི་སྦྱོར་བསྒྲུབ་པས་གནས་པར་བྱོས། །ཨ་རེ་ཨཱ་ཧོ། རིམ་པར་བྱེད་པ་དང་་པོ་རྫོགས་སོ། །ལྟ་བ། ཁ་ཏུ་ལྟ་སྐྱོང་ཁྱབ་བཀོ། འབྲུ། རྗེ་འཁ་གས་པོ་ཏོ་བ་པོ་པ་ལ་ས་ཁྲའོ་བྱ་བ་ལ་ཆེ་བ་ལ་འཁོ་ཧ་ཏེ། ཏ་ད་ན་བོ་ཚོའི་་་གསོལ་པ། རྗེ་བཙུན་ལྷའི་རྣལ་འབྱོར་ནི་ཇི་ལྟ་བུ། །སྔགས་ཀྱི་ཚུལ་ནི་་་གང༌། ལྷན་ཅིག་སྐྱེས་པ་ཡང་ཅི། ཟབ་པ་དང་རྒྱ་ཆེ་བ་གང་ལགས། བཅོམ་ལྡན་འདས་རྡོ་རྗེ་སློབ་དཔོན་གྱིས་བཤད་དུ་གསོལ། ་་་བཅོམ་ལྡན་འདས་ཀྱིས་་བཀའ་སྩལ་པ་ཀླུ་ཡི་གདན་དང་ལུས་བསྲངས་བཅས། །།།རྡོ་རྗེ་ཐལ་མོ་གསུམ་་་བརྡབས་པས། །སྒྲ་ཙམ་ཉིད་དུ་རྣམ་པར་བརྟགང་། སེམས་དང་རང་བཞིན་ལྷ་བདེ་བ། །ཨོཾ་སྭ་བྷཱ་བ། ༀ་བཛྲ། ཨོ་ཡོ་ག། ང་རྒྱལ་བྱ་སྟེ་འདི་རྣམས་ནི་་་སྔགས་སོ། །རང་བཞིན་མཚན་ཉིད་འབྲས་སྟོང་པ། །སྟོང་པ་མཚན་མ་མེད་པ་དང༌། །གང་ཡང་སྨོན་པ་མེད་པ་ནི། །།རང་བཞིན་སྤྲོས་དང་བྲལ་བ་བསྒོམ། །ཨོ་ཤུ་ནྱ་ཏ། ཨ་ན་མི་ཏ། ཨ་པྲ་ཎི་ཧི་ཏ། ཞེས་པའི་སྔགས་སོ། །རླུང་ནི་་ཡང་དག་བསྐུལ་བ་ལས། །སྣང་བ་མེད་པར་རབ་ཏུ་བརྟག། །སླར་ཡང་རླུང་གི་སྦྱོར་བ་ལས། །སྒྱུ་མ་ཙམ་དུ་རྣམ་པར་བསྒོམ། །ཡིད་ཀྱིས་རྫོགས་པ་སངས་་་་རྒྱས་པ་ཀུན། །སྐད་ཅིག་ལྷ་མོའི་རང་བཞིན་བརྟག། སངས་རྒྱས་ཟླ་བའི་ཁུ་བ་རྣམས། །ནམ་མཁའི་དབྱིངས་སུ་རབ་ཏུ་བསྒོམ། །རྣམ་པར་ཤེས་པ་རླུང་་་དང་བཅས། །ཁཾ་ཡིག་ཁུ་བ་གཉིས་མེད་གྱུར། །དེ་ཚེ་བྷ་ག་ཆོས་ཀྱི་དབྱིངས། །ཕྱི་རོལ་སྲུང་བ་ཡང་དག་བཅས། །ཧཱུྃ་ལྷ་མོ་རྡུལ་དུ་གྱུར་པའི་དབུས། །ཟླ་བ་ཁུ་རྡུལ་གཉིས་མེད་ཐིམ། །དེ་ལས་བཱ་རཱ་ཧཱི་དབང་ལང་ཚོ་མ། །འཆར་ཀའི་སྣང་།
