དཔོན་ཆེན་པོ་ཌ་བྷི་ཧེ་རུ་ཀས་མཛད་པ་རྫོགས་སོ། །།།བློ་བདེ་ཚོགས་དཔག་ཏུ་མེད་པ་ཡོད་པ་ར་ནས། ལག་པས་བདེ་བ་ལ་དཔལ་བབས་རིམ་པས་སྡུམ་པ་བད་པ་བྱ་བ་ལ། དེ་ཡ། བོ་་་ལཊ་ས་ཏཱ་ཞེས་པས་བསྡུར་མཆན། ་་་་་་་་་་་་་་སྔགས། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ཚེ«པེ་»བི། «པེ་»ཅིག«པེ་»དཀྱིལ་དཀྲུང་། «པེ་»བསྒོམས། «པེ་»གྱིས། «པེ་»དང་། ་ཕྱི་ལུས་པ་a་དང་། ད་། ལུ་བ་བྱ། འདུ་འབའ་ངོ་། །:ཕྱིག་ལས་མི་མང་བ་གྲུ་འདྲ་ཏེ། ཀྱི་ནང་བ་དགའ་བ་འདུག་པ་དང་ད་བ་ཀྱི་ཁྱོ་ག་ལ་བལྟའོ། བསྡུར་འབྲས་རེའུ་མིགབ་ལུང་པ། འགལ་བ་དང་ད་ཕྱིན་གྲུབ་པའི་དལ་པོ་དང་། རྒྱུད། （ང་བ་ལ་འབའ་དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང༌། །ཅོ་ནེ། །ཟ་ལ། ལོསྔགས། སྔགས། ས་ལ་བོད་རྟགས། :ཨུས། པུ་། ་པུ་་་། ཚན་གྲངས། ལྡེབ། ་ནབཕྲེང་། ཕྱི་ལོ་ཀྱི། ལ་དང་ཚེག་ཁྱིམ། ་་་་་་པ་དངད་ནས་ཏོ་བ། ཚོང་དུ་། །ཀ་ག་ང་ཚོ་ལ་དགོས་ཀྱི་ཁ་བ་དེས་ཀཟླ་་་་་་་་དང་ཁྲོ་ལ་ཚོང་། བོད་ཟླ་ལྔ་བ་བ་འ་འཁ::ཟླ་བའི་རྒྱུ་ཁ་ཚ་བས་ཁྱབ་པར། །ཏ་ད་པ་བ་བཀུ་ལ་ན་འོ་ਰੋਸ਼'ਸਮਸਰਤਸਕਟੌਨ'ਲਲਲਲਲਲਅਲਸਰਸਾਇੰਸ::ཀྱི་འཚོ་བར་དོ་བ། །ལོ་ཀ་བ། ཚོན་ཁྲ་ལ། །ཁྲོ་བ་གལ་ཆེནན།
