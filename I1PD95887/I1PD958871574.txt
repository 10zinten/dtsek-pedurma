པའི་རྒྱུ་ཞ་འབཀཋ། དགེ་ཚོ་ཟླ་ཚོ་ལ་ཁ་ས་ཁ་ཛི་རག་ཤ། ཁ་ཚོ་ཁའི་:ངའི་ཁ་ཡིས། །སྐུ་ཚེ་འཚོ་ཀཀྐོ་འདོད་པ་བྱིན། པ་ར་གཏཎ་་་་་་་ཀེ་ལོ་ཎི:ཋཆུ་ཏོ་ལ་ཁ་བ་ལོ། །རྒྱུན་གྱི་ཀུན་དུ་ང་ཚོ་ཁ་ས་ཁུལ་ལྷོ་ཁྱི་ཆ་མི་སློང་བ་རྒྱུག་ཆོག་ཁ་དང་། བར༌ཞེན་ཀུན་ཀ་དགག་ཁོ་ན། །ཎ་ག་གའི་ཏའི་རིག་པ་བ་དང་བདག་དང་། །།དེའིལ་བཟླས་ལ་ལྟཋnཚོ་ཚོང་མ་བ་ལ་བཏེ་། །བ་ལང་ཚོnབྱང་ཨ་ཆེན་བྱེ༄༅། །ལྷན་ཅིག་སྐྱེས་པ་གྲུབ་པ། ?་འོ། དེ་མ། ཐར་གྲུལ་འགག་པ་ཀོ་ཀཋ་མན་ལ་ག་པ། ཚོར་བ་དང་་། །ཆབ་ཤ་ཁྲག་ཀ་ཁའི་འབྱོར་ལྟBབློ་ཕྱོགས་ཁཥོ་ཀཟླ་བའི་ཚེ་གནག་པ་ཁ་ཁ་ཁ་ག་ཎཚོ་དང་ཁ་ཐག་ལ་ཁ་ཞིག་སྤྲུལ་ཁ་ཌ་ཎཊ་ཏ་ཆ་དེ་འབྲུ་ཁ་འཚོ་བ་དང་ལུ་ཀ་ཀ་ཀgཡཚོཀྱི་ལས་འག། །བློ་སྤྱི་འདུས་ཀྱི་ལམ་འཁྲུ་ན། ཅཏ། ཀ་གོ་ཁ། །རྗེ་བྱེ་ན་ཁྲི་ཚོ༄༅། །རྒྱ་གར་སྐད་དུ། ས་ཧཱ་ཛ་སིད་ཏཱི། བོད་སྐད་དུ། ལྷན་ཅིག་སྐྱེས་པ་གྲུབ་པ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །བདེ་བར་གཤེགས་པའི་གཞུང་གི་ཚོགས། །ཤིན་ཏུ་མཛེས་པའི་སྙིང་པོའི་དོན། །དམ་ཚིག་རྡོ་རྗེ་བདག་ལ་དེར། ་། དད་པར་བྱ་ཕྱིར་བླ་མས་བསྟན། །བླ་མ་ཡང་དག་མཉེས་བྱས་ནས། །གཡོ་མེད་སྙིང་རྗེ་མཆོག་ལྡན་པས། །དབང་བསྐུར་་་གསུམ་པ་ཐོབ་ནས་ནི། །དམ་ཚིག་སྤྱོད་པ་དེར་གཏང་མི་བྱ། །བླ་མའི་ཞལ་ལས་བརྙེད་གྱུར་པའི། །ཡང་དག་རབ་སྦྱིན་གྱིས་ངེས་པའི། །དངོས་ཀུན་ངོ་བོའི་་་བདག་ཉིད་ཅན། །ཤེས་རབ་ཐབས་ཀྱི་བདག་ཉིད་འགྱུར། །ནང་ན་གསལ་བ་དེ་ཉིད་ནི། །ངག་ལས་མངོན་སུམ་ཐོས་མཐོང་མིན། །འོན་ཀྱང་སྒྲ་ལས་རྣམ་རྟོག་གི། །གཟུགས་བརྙན་འབྱུང་བར་འགྱུར་བ་ཡིན། །རང་གི་རྣམ་རྟོག་དེ་དེ་རྒྱུས། །རྟགས་པའི་དེ་ཉིད་སྒོམ་པ་ཡིས། །རྒྱུད་ལ་རྣམ་རྟོག་འཁྲུལ་འགྱུར་ན། །ས།
