།རྒྱུད། །ཛིའིཙ་ཊི་ཌོ་བྱི། །ཀུན་ན། །གང་ལ་ན། །སུ་ལ་ཁྱིན་། །བདག་ད་ནི་ལེའུ་བཅུ་བདུན་པར་མདོར་བསྟན་པ་རྒྱས་པར་བསྟན་པ་ནི། དེ་་་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །བཅུ་བདུན་པར་ནི་ཁྱོད་གསུངས་པ་་་ཞེས་བྱ་བ་ནི། །ཁྱོད་ཉིད་ཀྱིས་བཤད་དོ་ཞེས་བྱ་བར་དགོངས་པའོ། །དེ་ཡང་་རྫས་ལས་སྐྱེས་པའི་ཕྱི་དང་ནང་གི་དབྱེ་བས་ཕྱེ་བ་བཤད་པ་ཡང་གོང་མ་ཉིད་དོ། །བཅུད་ཀྱིས་ལེན་གྱི་རྫས་ཀྱི་སྡེ་ཚན་ཏེ་དེའི་ལྷག་པར་བྱེད་པ་ནི་གནས་སྐབས་སོ། །དེ་ཉིད་ཀྱི་ཆོ་ག་ཞེས་བྱ་བ་ནི། །ཆོ་གར་བྱེད་པའོ། །གང་ལ་རྗོད་པར་བྱེད་པ་དེ་ཡང་འདིའི་ལེའུ་ཞེས་བྱ་བ་དེ་ནི་དེ་བཞིན་ཏེ། བཞི་བཅུ་རྩ་བདུན་པའོ། །།།རྡོ་རྗེ་མཁའ་འགྲོའི་རྣམ་པར་འགྲེལ་པ་ལས་ལེའུ་བཞི་བཅུ་རྩ་བདུན་་་པའོ༑། །།།ཁ་ལ་ཁ་ག་ང་ཅ་ཏེ་ཁ་ནས་་་་་། ཏབ་ཏེ། ལོ་ਸਮTARਰਨਸਿੰਪਲਸਕਮਰਕਲਸ਼ਲਆਮསྦྱིན་སྲེག་གི་ཐབ་ཁུང་གི་ཚད་ལ་སོགས་པ་བསྟན་པའི་ཕྱིར། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །།གང་གི་ལས་ཇི་ལྟ་བུའི་ཐབ་ཁུང་གི་ཚད་དང་་་གདན་དང་། ཡམ་ཤིང་ལ་སོགས་པ། དེ་ངེས་པར་བྱེད་པའི་དོན་གྱི་ལེའུ་འདི་གོ་སླ་བའི་ཕྱིར་རྣམ་པར་མ་ཕྱེའོ། །ལས་རྣམས་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་ཞི་བ་ལ་སོགས་པའི་རྫས་རྣམས་ཀྱི་གསང་བའི་དོན་གྱི་ཡུལ་ཏེ་ཐབ་ཁུང་ལ་སོགས་པའི་་་མཚན་ཉིད་དེ། དེས་སྦྱིན་སྲེག་གི་ཆོ་གའོ། ་་། དེ་རྗོད་པར་བྱེད་པའི་ལེའུ་སྟེ། བཞི་བཅུ་རྩ་བརྒྱད་པའོ། །།།ཚོ་ཚོ་ཁ་པར་ན་ཞི་བ་ལ་གླུ་བ་ལྷག་རྡོ་རྗེ་མཁའ་འགྲོའི་རྣམ་པར་འགྲེལ་པ་ལས་ལེའུ་བཞི་བཅུ་རྩ་བརྒྱད་་་་པའོ༑། ་་། ཟ་བྱེད་དེ་བ་ར་ཤཥི་ཁོམ་ཁཎ་ཊ་ཁོ་། ས་ལ་འདབ་ན་དའ་བ། །དང་། །ད་བ་ས་གཡོ་བ་—།
