པ་ག། །སྡེ་དགེ། སྣར་ཐང་། སྔགས། སྔགས། །།དཔེ་གཞི། །སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ཚན་ན་། ན་—བ། ཕྲེང་། ཚེག་ཁྱིམ། འདི་ཁ་ནས། སུ་གཡོ་བ་བྱ་བ་ཞིག་བཅུ་ན་ཁོག་ལ་སྐྱོན་འདུག་པ་བགྱིད་དང་བཟོ་ཁང་ཆ་ཁཱ་ཎ་ཏུ་སྐྱ་ཁྲི་ཞུབ་ལཝ་ད་ཆ་ཁ། ན། ཏེལ་བཅོལ་ཀ་བ་བརྟན་རྒྱུ་སྐྱེལ་མྱོང་བ་ཚབ་ཀྱིས་བླ་::འི་ཁོ་རེ་ཚོ། །ཅར་(ཁ་ཏོ་ལུག་ལགེག་ཡེ་ཤེ་རིབ་བྱས་ཚེ་རྒྱུད་ཕྱག་:ཞིཊ་ཏུ་བསྒྱུར་བ་འབག་བཙོག་དང་ས་སྒྲོག་བྱེད་མ་ཁ་ཎ་པ་ཏི་པ་ལ་བ་ར་མ་ར་ལ་ར་བ་དང་དམ་བཞག་སྐྱོང་། མཉིད་ཏཏ་ལའི་བླང་།
