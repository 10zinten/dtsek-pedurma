བའི་སྟན་ལ་འདུ་བ་བྱས་སྟེ། །མི་གཡོ་བསམ་གཏན་དཀྱིལ་དཀྲུང་མ་བཅ། ༽དེ། ས་བདག་སྤྲོད་དུ་བཏང་ནས་ནི། རང་ཉིད་སྐད་ཅིག་ལ་བདག་གདམས་ལ། །པ་གི་སྙིང་ལ་བར་པད་ཉིའི་སྟེ་བསྡུར་འབྲས་རེའུ་མིག་ཕྱོབ། ས་བཅུད་འཕྲོས། །པ་ཆེ་དུན་ཐམས་ཅད་རྣལ་འབྱོརྒྱུད། （ལ་ཁོ་རྗེ་ཞུ་ལ་ཕྱི། སྣར་ཐང་། སྡེ་དགེ། ནུས་པ་དང་ལྔ་འཔེ་ཅིན། སྔགས། བྱསྔགས། དཔེ་གཞི། སྡེ་ཚན། པོད་རྟགས། ཚན་གྲངས། ངེས་བསྟན། ལྡེབ། །クーパ.ཁ་ནན་:::/ནབ་ཕྲེང༌། ཚིག་རྐང་ཚེག་ཁྱིམ། ::ཀ་ང་ད་ན་བ་། ལྔ་བ་ཁྲོ་བ་ཁ་ང་ཚོ་དང་། །ཀ་ཀུ་བ་ཀན་ཏུ་བཏཁན། ਕਿਕਈਕਲਲਾਲਨਾਲੋਂਪੱਕਆਮਸਰਨੂੰཡ་་་་ཀ། ཆ་ཎ་ཁེ་ལས་ལྟ་བ་པོ་ལ་ཀ་བ་བཞི་ཎང་ན་ཏསྐོར། ངག་་་་་་་་་་་འོན་རྒྱུ་་་ཎ་བ་རང་གྲོལ་བར་བྱ་བ་ས་ཀྱི་ག་ཎཏོ་ང་བ་ག་ཎ་ཁོ་ཊིཀ་ཎ། ཏོར་བ་ཚཟླ་ཎ་ཏུ་རྗེ་ཁཎཁས་ན་སྐྱོན་སྒྲོ་དང༌། ཏཆོས་ཁ་བ་བ་ནི། ཚ་བང་བ་འཊོ་ཏ་ལྷ་ལ་ཁ་པར་གྱི་ཀོ་པ་རང་གི་གོ་ས་ཁོ་ན་ཚར་རྒྱུ་བ་ལས། འོ་ད་བཞུགས། འདུ་ཁང་དུ་རྒྱལ་ཁ། ག་བ་ད་ཏ། ཏཁཞིང་ཁཀ་བློ་ལ་ཚད་ཚེ། ཀ་ཀར་འདུ་འ། པཎ། ཁ་ན་ཏཁབ་དུས་། ཊཏུགས་བཏབ་གནང་བ་སུ་བཞག་ཆེ་ལ་སོགས་གྲུག་།
