ལྷོ་ཁོ་པ། paraགས་ཀྱི་སྣོད་དུ་རུང་བར་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་བསྟན་པར་བྱའོ། །དཀྱིལ་འཁོར་འདིར་ནི་ཡང་དག་པར། ལ་། བདག་གིས་སློབ་མ་བཞུག་པར་བྱ། །དེ་ལྟར་བསོད་ནམས་དེ་བཞིན་དུ། །ལྷ་རྣམས་རིགས་ཀྱི་རིམ་པར་ཤོག་སོགས། འདིར་ནི་དངོས་གྲུབ་ཇི་ལྟར་འབྱུང་། །གང་གི་རིགས་ཀྱི་སྣོད་འགྱུར་བ། །ཀs:/། ཇི་ལྟར་་བསོད་ནམས་ནུས་མཐུས་ཀྱང་། །དཀྱིལ་འཁོར་དག་ཏུ་དེ་བཞིན་ཤོགཚོ། །དེ་ནས་མེ་ཏོག་གི་ཕྲེང་བ་དེ་ཧཱུྃ་ཡིག་དང་ཀྲྀ་ཡིག་གིས་ཡང་བཟླས་ལ་དཀྱིལ་འཁོར་དུ་དོར་བར་བྱའོ། །།།པྲ་ཏཱིཙྪ་བཛྲ་ཧོ་ཞེས་བྱ་བ་འདིས་སོ། །དེ་ནས་མེ་ཏོག་བབས་པས་དམན་པ་དང་མཆོག་ལ་སོགས་པ་འདིས་དངོས་གྲུབ་མཚོན་པར་བྱ་སྟེ། ལྷ་རྣམས་ཀྱི་དབུའམ་མཛོད་སྤུའམ། ་ཌ་:གཙུག་ཏོར་ལ་བབས་ན་དེ་ཚེ་ཕྱག་རྒྱ་ཆེན་པོའི་དངོས་གྲུབ་བོ། །སྤྱན་དང་ཞལ་ལ་སྔགས་ཀྱི་དངོས་གྲུབ་བོ། །སྐུ་མཆོག་་་ལ་མཆོག་གོ། །སྐུའི་དབུས་ལ་འབྲིང་ངོ༌། །།སྨད་ཀྱི་ཆ་ལ་ཐ་མའོ། །སྐུ་ལས་་རིང་ན་ཡུན་རིང་པོ་ནའོ། །ལྷའི་གནས་ཀྱི་རེ་མིག་ལ་སོགས་པ་ཐམས་ཅད་ལ་་་ཤེས་པར་བྱའོ། །དཀྱིལ་འཁོར་གྱི་ཕྱི་རོལ་དུ་ལྷུང་ན་སླར་ལན་གསུམ་དུ་དོར་བར་བྱའོ༑ཁོ་། སླར་ཡང་དེ་ཉིད་དུ་བབས་ན་དངོས་གྲུབ་མེད་དོ། །ཞེས་བྱ་བ་ནི་མེ་ཏོག་འདོར་བའི་ཆོ་གའོ། །།།བར་ང་ལ། ཞེད་། ད་བར་དེ་བ་ཐོབཀ%ཁབ་ཆེ་བ་དང་། འོ་ད་ཡིན་ན་བོད་པའི་བག་ག་དོར་བའི་མེ་ཏོག་གི་ཕྲེང་བ་དེ་བཟུང་ལ། སློབ་མའི་མགོ་བོ་ལ་བཅིང་བར་བྱ་སྟེ། སེམས་དཔའ་སྟོབས་པོ་ཆེ་འདི་བཟུང་སྟེ། །ཞེས་བརྗོད་དོ། །དེ་ནས་མིག་གཉིས་ལ་ཨོཾ་ཡིག་གཉིས། །གཟི་བརྗིད་འབར་བ་བསམས་ལ། །རྡོ་རྗེ་་་སེམས་དཔའ་དེང་ཁྱོད་ཀྱི། །།མིག་འབྱེད་པ་ནི་བརྩོན་པར་མཛད། །མིག་ནི་་ཐམས་ཅད་འབྱེད་མཛད་དེ། །རྡོ་རྗེའི་མིག་ནི་བླ་ན་མེད། །ཅེས་བརྗོད་ལ་་།
