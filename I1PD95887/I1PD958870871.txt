ཁོ་མཁ་བbན་ཏུ་བསྟན་འགྱུར། རྒྱུད། དུའིལ་ཊི་པ། བཀགས་བསོ་བྱེད་ཅེས་གྲགས་ཤིང་། ཡུལ་དེ་ཡོ་བྱད་ཤིན་ཏུ་མང་ངོ༌། །ནང་ལྟར་ན་ནུ་མས་ཀྱང་གཞན་བསོ་ཞིང་མི་མཛད་པའོ། །༄༅་། ད་ཆ་ཁ་ཊོ་ཊ་ཁ་ལ་ཁ་ཅཁ་བསྒྱུར་ཞིང་ཞེས་དཔའ་བོ་དང་རྣལ་འབྱོར་མ་རྒྱུན་ཏུ་འཚོགས་པའོ། །ཉོ་། ཁ་ཅདལ་ད་ལ། ཁ་དེ་བ་ལ་སྐྱོ་དང་འབྲང་ཆུང་བ་རྟག་ཏུ་མི་ཚོ་ག་ཏ་། ལོ་ད་ཅཏྲི་ཤ་ཀུ་ཎ་ཞེས་གསུམ་འདུས་ཅེས་པ་སྟེ། ཏུ་རུ་ཀའི་ཡུལ་ན་ཆུ་བོ་གང་གཱ་དང༌། པཀླུ་དང་། སིན་དྷུ་གསུམ་འདུས་པའི་ས་ན་རྡོའི་ཆོས་འབྱུང་ལྟ་བུ་ཅིག་ཡོད་ལ། དེ་ལ་བྱ་ངང་པ་དང་། ཁྲུང་ཁྲུང་དང༌། ས་ར་ས་དང༌། བཞད་པ་དང་། ནེ་ཙོ་ལ་སོགས་པ་མང་པོ་སྐད་སྙན་པ་སྒྲོགས་ཅིང་ཉམས་དགའ་བ་་་ཅིག་ཡོད་ལ། མིང་ནི་གསུམ་འདུས་ཅེས་གྲགས་སོ། །ནང་ལྟར་ན། ལྟེ་བར་རྩ་གསུམ་འདུས་པས་སོ། །་་། ་༌་་་་་་བ་ལ་ཁའི་ཁ་ཚོ། ་བར་ལང་ཆ་ལ་ཀོ་ས་ལ་ནི་ནུབ་ཕྱོགས་རྒྱལ་པོ་སས་རྒྱལ་གྱིས་རྒྱལ་ཁམས་ནས་རྒྱང་གྲགས་་ཅིག་ན་བསྒྲུབ་གནས་ན་སྣ་ལྟ་བུའི་ལིང་ཁ་གཅིག་ཡོད་དེ། དེ་ལ་མི་རྒྱུད་བྱེད་ཅེས་སོ། །ནང་ལྟར་ན་སྣ་རྩེར་སེམས་རྒྱུད་པའོ། །རེས་འགའ་འཚོགས་པས་ཉེ་བའི་ཞིང་ངོ་། །འོང་བ། འབྱུང་ལ་ཀུན་ངོ་མཚོ། །ཁ་ཁ། ཁ་ལ་ཁ་ན་བདཥ་ཎ་ཀ་ལིང་ཀ་ནི་ཟད་བྱེད་ཅེས་ཏེ། རྡོ་རྗེ་གདན་ནས་དཔག་ཚད་བཅུ་གཉིས་ཀྱི་ཕ་རོལ་ན་ཡུལ་གླང་པོས་འཛིན་ཅེས་བྱ་བ་ན། ཁ་ལྟ་བུའི་བསྒྲུབ་གནས་སྤང་དང་ནགས་ཀྱི་མཚམས་ལེགས་པོ་ཅིག་ཡོད་ལ། དེ་ནི་ཟད་བྱེད་ཅེས་སོ། །ཡུལ་་དེར་ཡི་དྭགས་མང་བས་རོ་བསྐྱལ་བ་ཙམ་གྱིས་མེད་པར་བྱེད་དོ། །ནང་ལྟར་ན་་ཁས་ཀྱང་ཟས་མྱུར་དུ་མེད་པར་བྱེད་པས་ནའོ། ་བ། ཚོ། །ཏེ་ད་ད་ལཾ་པ་ཀ་ནི། འཕྱང་བ་ཅན་ཞེས་ཏེ། གར་ལོག་གི་རྒྱབ་ན་བསྒྲུབ་གནས་།
