ཡེ་ཤེས་སེམས་དཔའ་སྤྱན་དྲངས་ཏེ་ལྔ་པའོ། ན་། ད་དེ་རྫོབ་བས། ངས་ལོ་རྒྱུའི་བསམ་པ་ཐམས་ཏུ་ཁྲོཔ་སི་ཁྲོན་པ། རང་བཞིན་དག་པ་དག་པས་མེད་པ། །ར་ཏེདེ་ནས་དབང་བསྐུར་བ་ནི་འོད་ཟེར་གྱིས་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་དང་། རིག་པའི་ལྷ་མོ་དང་། ཁྲོ་བོ་རྣམས་སྤྱན་དྲངས་ལ། ཕྱི་དང་་ནང་གསང་བའི་མཆོད་པས་མཆོད་ལ། བདུད་རྩི་ལྔས་གང་བའི་བུམ་པས་དབང་བསྐུར་ཏེ། རྣམ་པར་སྣང་མཛད་ཀྱིས་སྐུ་རྡོ་རྗེ་ལ་དབུ་བརྒྱན་པར་བྱའོ། །སྣང་བ་མཐའ་ཡས་ཀྱིས་གསུང་རྡོ་རྗེ་ལ་དབུ་བརྒྱན་པར་བྱའོ། །མི་བསྐྱོད་པས་ཐུགས་རྡོ་རྗེ་ལ་དབུ་བརྒྱན་པར་བྱའོ། །དབང་བསྐུར་བ་སྟེ་དྲུག་པའོ། ་ཌ། བཀྱི་བདག་གིས་ཀས། །ཊ་་ཏེ་ཞུ་བ་ཚོའི་རྒྱུར་བ་ན། ར་བ་རབ་:། ཁ་དེ་ནས་གཏོར་མའི་བསྐྱེད་པ་ནི། །པདྨའི་ལྟེ་བར་ཡཾ་ལས་རླུང་གི་དཀྱིལ་་་འཁོར་གཞུའི་དབྱིབས་འདྲ་བ། དེའི་སྟེང་དུ་རཾ་ལས་མེའི་དཀྱིལ་འཁོར་གྲུ་གསུམ་དམར་པོ། །དེར་ཨ་ལས་པདྨ། དེའི་ནང་དུ་ཧཱུཾ་ལས་གསང་བའི་སྨན། །་ཨོ་་་ལས་རྡོ་རྗེ་ཆུ། ཛི་ལས་རང་བྱུང་། ཁཾ་ལས་ཤ་ཆེན་པོ། ཧཱུྃ་ལས་བྱང་ཆུབ་་་སེམས་དབུས་སུ་སྟེ། དེ་རྣམས་ཀྱི་ས་བོན་གྱིས་མཚན་པ་སྟེ། དེའི་ནང་དུ་ཨ་ལས་ཟླ་བའི་དཀྱིལ་འཁོར་དེའི་སྟེང་དུ་ཧཱུྃ་ལས་རྡོ་རྗེ་ཧཱུཾ་གིས་མཚན་པའོ། །དེའི་སྟོབས་ཀྱིས་བུད་པ་དང་འབར་བར་བཞུ་ནས་བྱང་ཆུབ་ཀྱི་སེམས་སུ་གྱུར་པར་་་བསམ་མོ། །བདུད་རྩིར་སྦྱར་བ་སྟེ་བདུན་པའོ། ལེ། ''ཎཎ། ཐི། ངོ་དེ། །ཏེ་ཏ་བeང་མི་ལུ་བར་ཆ་ལྔ་བ། ཚོང་ཚ་ཁབོ་ཚེp@་ཎ་པ་ཏོག། པཎདེ་ནས་ཡེ་ཤེས་ཀྱི་བདུད་རྩི་དགུག་པ་ནི་པདྨའི་སྟེང་དུ་ཨོ་བསམ་ལ་དེ་ལས་འོད་ཟེར་འཕྲོས་པས་རང་བཞིན་གྱི་གནས་ནས་ཡེ་ཤེས་ཀྱི་བདུད་རྩི་དགུག་ལ། དམ་ཚིག་ལ་བཞུག་པར་བསམ་པར་བྱའོ། །ཡེ་ཤེས་ཀྱི་བདུད་རྩི་བཀུག་པ་སྟེ་།
