ཁྲག་ཀྱང་མྱུར་དུ་དབྱུང་བར་བྱའོ། །སྨན་ནི་ལི་ནི་བསྐོལ་བའི་ནང་དུ་ཤི་ཤ་དང༌། རྒྱ་ཤུག་གི་ཕྱེ་མས་བཏབ་སྟེ་སློན་དུ་གཞུག་གོ། །ཀོ་ཤ་ཏ་ཀི་དང་། ཤི་ཤ་དང་། ཛི་ལུ་ཏ་གའི་འབྲས་བུ་དང༌། པོ་སོན་ཆ་དང༌། དེ་རྣམས་ཕྱེ་མར་བྱས་ཏེ། ཞོ་དང་སྦྱར་བ་བླུད་ནས་ྋསློན་དུ་གཞུག་གོ། །ཡང་ན་ཤུ་དག་དང༌། པོ་སོན་ཆ་དང་། ཛི་མུ་ཏ་དང༌། རུ་རྟ་དང་། དེ་རྣམས་བ་གཅིན་དང་བཏེགས་ཏེ། གོང་མ་བཞིན་དུ་བྱས་ལ་བླུད་ན་བྱི་བའི་དུག་འཇོམས་སོ། །བཀྲུ་སྨན་དུ་:ནི་ལི་ནི་དང་། དུར་བྱིད་དང་། འབྲས་བུ་གསུམ་:གྱི་ལྡེ་གུས་འདོད་པ་ཡིན་ནོ། །མགོ་ཁྲུས་ལ་ནི:༔ཤི་ཤ་དང༌། འབྲས་བུ་འདོད་པ་ཡིན་ནོ། །མིག་སྨན་དུ་བའི་ལྕི་བ་བཙིར་བའི་ཁུ་བའི་ནང་དུ་རྩ་བ་གསུམ་གྱི་ཕྱེ་མས་བཏབ་པ་སྦྱར་རོ། །བ་ལང་གི་ལྕི་ཁུའི་ནང་དུ་ཀ་:པིད་ཏའི་ཕྱེ་མ་དང་། སྦྲང་རྩི་དང་སྦྱར་བས་ཀྱི་ལྡེ་གུར་བཟའོ། །མོན་སྣེའུ་དམེ་རུའི་རྩ་བ་མར་དང་ལྷན་ཅིག་བསྐོལ་ཏེ་བཏུང་བར་བླུད་ན་ཕན་ནོ། །སེར་པོ་རྣམ་གཉིས་དང་། གེ་རི་ཀཎི་ཀ་དང་། བཙོད་དང་། ཤིང་མངར་དང༌། སླེ་ཏྲེས་དང་། །དེ་རྣམས་མར་དང་སྦྱར་་་་བའི་བཏུང་བ་སྦྱར་ནས་ཡང་བཟའོ། །ཡང་ན་ཐལ་ཏྲེས་ཀྱི་རྩ་བ་མར་དང་བསྐོལ་བའམ། ཡང་ན་ཀ་པིད་ཏ་རྣམ་ལྔ་མར་དང་ལྷན་ཅིག་བསྐོལ་བ་གང་ཡང་རུང་བ་བཏུད་པར་བླུད་ན་ཕན་ནོ། །:：སིནྡྷ་ར་དང༌། རྒྱ་སྤོས་དང༌། ཤི་ཀྲུ་དང་། བིལ་བའི་རྩ་བ་དང་། བ་སྤུ་བ་དང་། ཤུ་དག་དང་། གཟེ་མ་དང་། :ཛི་མུ་ཏ་ྋརྣམས་བསྐོལ་བའི་ཁུ་བའི་ནང་དུ་སྦྲང་རྩི་དང་ལྡན་པར་བླུད་དོ། །ཁ་ཟས་སུ་ནི་འབྲས་སཱ་ལུའི་ཆན་ཞོ་དང་ལྷན་ཅིག་ཟོས་ན་བྱི་བའི་དུག་གིས་ཉམ་ཐག་པ་སེལ་ཏོ། །ཤ་ར་ཕུང་ཀའི་:རྩ་བ་ཀྱིས་བོན་བཏགས་པའི་ཕྱེ་མ་དར་བ་དང་སྦྱར་བ་བཏུང་ངོ༌། །ཨ་ལའི་རྩ་བ་བཏགས་པར་ཐུག་གི་གཅིན་དང་སྦྱར་་།
