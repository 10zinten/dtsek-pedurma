དང་། ཁ་རན་ས་དང་། བེ་ཤ་ལ་དང༌། མ་ཧ་ཀི་ཤ་དང༌། ཉེའུ་ཤིང་་བ་དང་། དེ་རྣམས་བཏགས་ཏེ་རིག་སྔགས་ཀྱིས་བཏབ་ནས་གྲོ་བའི་ནང་དུ་་་བཅུག་སྟེ། བྱིས་པ་བྱང་ཕྱོགས་སུ་ཁ་ལྟར་བཅུག་ལ་མགུལ་དུ་གདགས་སོ། །དེ་དག་མ་འབྱོར་ན་ཟར་མའི་སྐུད་པ་སྲ་བ་ལ་མདུད་པ་བོར་ཏེ་སྔགས་བཟླས་པར་བྱའོ། །བྱིས་པ་བསྲུང་བའི་ཕྱིར་དང་པོར་རང་བྱུང་ཆེན་པོ་ལ་ཕྱག་བྱ་སྟེ། སྔོན་གྱི་དྲང་་སྲོང་ཆེན་པོ་རྣམས་ཀྱིས་ཞི་བར་བྱས་པ་ཡིན་པས་གདུ་བུའི་སྲུང་བའི་སྐུད་པ་་་གདགས་པར་བྱའོ། །ཨིན་དྲ་པྲ་ཏི་ས་རམ་དུ་བྱཱི། ཡེ་ཙ་ཏ་ས་ནུ་ཙ་རི་ཎ། བཛྲ་རཱ་ཤ་ནི་དྷ་ར་བྷཱུ་ཏ་མེ་གྷ་བིན་དཱ་ནུ་སཱ་རི་ཎ་ཨི་ར་བ་ཎ། སྐན་དྷ་ག་ཏམ། ཡེ་ན་ཡ་ཏི་པཱུ་ར་ཎ་ད་རམ། ཡ་མ་བྷི་བ་ས་ཏོ་རཱ་ཛ་ཡེ་ཛ་ཏ་སཱུ་ནུ་ཙ་རི་ཎིི། མེ་:ཙ་སུཾ་ཀ་རི་སྲིང་ཧོ། ག་ཛ་བ་ཛི་སཱ་མཱ་ནཱ་ནཱ། ཀཱ་:ལཀྵ་ཀཱིལ་ཡ་སི་པཱ་ཤི་སྟ་མ་དཀྵ། ཤྲཱི་ཀུ་ཊི་མུ་ཁ་ཤག་ཏི་མུངྐ་ར། ཧསྟེ་ཙ་ཁ་ད་ག་པྲ་སཱ་དྷཱ་ར་ཤུ་ཡེ། པཱ་ཤ་རཏྣ་ཀ་རོ་རཱ་ཛ། ཏོ་བྲིད་སཱ་ག་རཱ་ལ་ཡ། ཤ་རད་:ཕྲག་ཎ་ཀ་ར། ཡོ་ཙ་ཏ་སྤྱན་ནུ་ཙཱ་རེ་ཎ། མ་ཧོ་ར་རྩི་མན་ཏམ་བེ་གེ་ན། གློ་བ་ཡན་ཏི། སེ་རམ་པ་ཧི་ེད་ནཱ་དཱི་པོ་བྷེ་ཤ་བྷ་ན་སརྦ་ཡཀྵ་དི་པོ། བུ་བི་ནཱ་ནཱ། ཀ་རཤུ་ཡེ། བྷཱུ་ཏ་སྐྱེ་ཅིབ་ནུ་ཙཱ་རཱ་སཱ་དཱ། :·ཨཱི་དིཉྩ་བྷ་ས་བྷོ་རུ། དྲ་མ་ཏུ་ཏྲ༔་ཤི་ནཱ་བཱ་པི་ཏེ༑སརྦ་ན་མ་སཱ་ཏ་མ། ཨི་མམ་པྲ་ཏི་ས་ཀྲི་ཡམ། དཱིར་གཱ་མཱ་ཡཱི། ར་རོ་གན་ཙཱ། ད་བཱ་དཱ་སན་ཏུ། སན་ཏི་དེ་བྷ་ག་བཱན་བྲ་ཧྥ་སྭ་སྟི་ན་རཏྟི། པ་ར་བྷ་ཨུ་སྭ་སྟི་བེ་དཾ༔་ཡག་རྒྱ་ཤཱིཨིག་ནི་ཤ་མ་ཨུ་ཙ། སཱསྟི་ཏི་ཤི་ཌི''ཏེ་བྷི་ར་ཧི་ལ། ཤྲཱི་མེ་དྷ་ལྟ་མ་རྱ་ཏི། སྭསྟི་དུཋཤུལི། བ་སུ་ཧཱི་བྷུད་དྷི་རེ་བ་ཙ། སྭ་སྟི་རུ་ཏྲི་བ་ཤ་བ་ཀཱ་ཤུ། སྐན་དྷ་ཤ་ཀ་ཏི་དྷ་ར་བྷཱུ་ཏི། སཱསྟི་དཱ་ཏ་ཙ། སུ་བཎ་བུ་མ་ཧཱ་ལ། སཱསྟི་དེ་བཱཉྩ་རཀཉྩ། མ་ཙཱ་དེ་བ་པཱ་མ་ག་མ། དུར་བྱ་།
