ད་ཏྲ་སརྦ་ཀུ་མཱ་ར་ན་མ་སཾ་ག་ཏ་གན་མཱ་ད་ནེ། རཀ་ཏེརྦ་ཙ་པ་ཏ་རཀྟ་ཤུ་ག་ནི་བ་ས་ནི། ཀུ་མཱ་ར་སྱ་ཁ་ཡམ་ཏ་ཏྲི། ཡཱ་ཏ་བ་ལམ་བི་མུན་ཙ་ཏ་ཤན་མུ་ཁ་དྷ་ཤ་ཀྵ། བེ་ཏྲ་སི་ཀ་ལེ་དོ་བར། །བི་ར་ཀུ་མཱ་རི་དི་བ་ཏི། སྟེ་བ་པུསྟི་བི་ཤི་བྱ་ཏུམ་བྲ་:ཧ་མ་ནོ་བྲ་ཧྥ་ནོ་བྲ་ཧྥ་ན་ནན་ཏུ། སརྦ་པེ་དི་པ་ད་མཱ་པར། ་་ཡཀྵ་་རཱཀྵ་ས་ནཱ་:གཱ་ནཾ་གན་དྷ་ར་བཱ་ཨ་སུ་ར་ཡོ་ར་པི། པི་ཤཱ་ཙ་བྷཱུ་ཏ་སཾ་ག་ནཱ་མ། བྱ་ད་མི། ཙཱ་མཱ་རི་ཤཱ་རཱ། ཨག་རྒྱ་པ་ཡ་ཏི་སྐན་དྷོ། ག་མྱ་ཏཱ་མཱ་ག་ནད་མ་ད་:ནཱ་མི། གྲོ་ཧོ་ཡ་ཀོ་པ་ཡོད་ད་ཉམ། ཀུ་མཱ་ར་སྭཱ་ཡ་ཤསྦི་ནི། རྟན་ཙུ་ཏ་ས་:བྲེ་བྲི་མ་བྷ་ཡ་པ་རི་ཤ་དྷོ་བྷ་བེད། དེ་ལྟར་གཏོར་མ་བཏང་བའི་འོག་ཏུ་མེ་ལ་སྦྱིན་སྲེག་གིས་མཆོད་པར་བྱ་སྟེ། དེ་ཡང་བཅའ་བ་དང༌། བཟའ་བ་དང་། མེ་ཏོག་ལ་སོགས་པ་སེར་སྐྱ་དང་། དམར་པོ་ལ་སོགས་པ་རྣམ་པ་སྣ་་་ཚོགས་དང་། དྲི་རྣམ་པ་སྣ་ཚོགས་དང་། །གུ་གུལ་གྱི་བདུག་པ་དང་། སེང་ལྡེང་ལ་སོགས་པའི་ཡམ་ཤིང་བྱའོ། །ཡང་ན་ཙན་དན་གྱི་ཤིང་གིས་བྱ་སྟེ། དེ་ཡང་གྲངས་བརྒྱ་རྩ་བརྒྱད་སྦྱིན་སྲེག་བྱའོ། །གདོན་ཐམས་ཅད་ལས་གྲོལ་བར་་བྱེད་པའི་མེ་དབྱུག་པ་ལྟ་བུ་ཞེས་བྱ་བའི་རིག་སྔགས་བརྗོད་པར་བྱ་སྟེ། ཨོཾ་བྲ་་ཧྥ་བྱ་ཏ་ན་ལམ་དེ་བམ། ཧ་སྱག་བྱ་པ་ཏི་ཙམ་ཀམ། དེ་བ་ན་ས་པྲ་མུ་ཁ་མ་སཱ་ཏི་སྭཱ་ཧཱ། ཨ་བ་:ཧཱ་ར་ཧཱུ་ཛམ། ས་ཧི་བད་ཏམ་ཧ་ན་ཏན་སི། པཱ་དཱ་ཀ་ས་བཀྵ་ན། ཏཱམ་ཡ་ག་ཉཉྩ། ཡུ་པཉྩ། ཏཱམ་ཧི་དྷ་ར་མ་བ་ཏང༌། ག་ཏི་ཧི་རན་ཅུ་ཙིརཱ་ཏཱ་དེ་བ་སྣུམ། ཏཱམ་ཧི་བེ་དང་བི་དང་ག་ཏི། དུར་ཁྲི་ཙིཏཱ་ནཱམ། བི་ནཱ་ཡ་ཤ་ཡཱ་ཏཱམ། ཧི་ཨུད་པ་ད་ཏི་སུ་རེ། ད་ཧ་ས་གྲ་ཧ་དེ་བ་ཏཱམ། ཨི་དམ་དུཋ་ས་བྲི་གྲ་ཧ། ཨག་ན་ཡེ་སྭཱ་ཧཱ། ཏཱམ་ཏི་ཤཱན་ཏི་ཀ་རོ་བ་ན་ནི། སྟམ་མ་རུ་ཏ་རོ་དན། ཧ་རི་པྲ་དཱ་ན་གྷ་དི་རི་ཤི། དཱི་མཱམ་ནི་གི་རྒྱ་གུ་ཧུམ་་་།
