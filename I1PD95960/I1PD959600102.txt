།སུ་ཁ་བི་པཱ་ཀཾ། ར་ཎི་པྲ་ཏུཏྤནྣ་ཁ་མཱ་ཡ་ཧཱུྃ་དུཁ་བི་:པཱ་ཀི། །ཁ་ལོ་ལའང་རྣམ་པར་སྨིན་པ་བདེ་བ། ད་ལྟར་བྱུང་བ་ལའང་སྡུག་བསྔལ་ལ་ཕྱི་མ་ལའང་རྣམ་པར་སྨིན་པ་སྡུག་བསྔལ་བ། ཀnཌ་(བློ་ད་དཀ་ཁ་ཁ་nཔ་ཊ་ཀང། བོད་ཏི་ར་ཀ༌ཌ་Pཁོནོར་བདུན་གྱི་མིང་ལ། བོན་ལ་དུང་་:ཤ་དྷཱིདྷ་ནཾ། ཤཱི་ལ་དྷ་ནཾ། ཧཱི་དྷ་ནཾ། ཨ་པ་:ཏཱ་བྱ་དྷ་ནཾ། ཐུ་ཏ་དྷ་ནཾ། དད་པའི་ནོར། ཚུལ་ཁྲིམས་ཀྱི་ནོར། ངོ་ཚ་ཤེས་པའི་ནོར། ཁྲེལ་ཡོད་པའི་ནོར། ཐོས་པའི་ནོར། །ཏྭཱ་ག་དྷ་ནཾ། པྲ་ཛྙཱ་དྷ་ནཾ། །་་་་་་་སJཊ་ཋpན་དང་ལ་། བ་་འོད་འབར། །ཁྱjང་གྲ་ནང་གཏོང་བའི་ནོར། ཤེས་རབ་ཀྱི་ནོར། །་་ལ་་པ་བླ་ན་མེད་པ་དྲུག་གི་མིང་ལ། ཞི་ར་བ་གང་། གདན་ས་བ་::ཀ་ཀ། དན་ནཱ་ནུཏྟནྣཾ། ཐ་བ་ཎཱ་ནུཏྟམཾ། ལཱ་བྷཱ་ནུཏྟམཾ། :ཤི་ཀླུ་ནུཏྟམཾ། མཐོང་བ་བླ་ན་མེད་པ། ཐོས་པ་བླ་ན་མེད་པ། རྙེད་པ་བླ་ན་མེད་པ། བསླབ་པ་བླ་ན་མེད་པ། བ་རི་ཙནྡཱ་ནུཏྟནཱཾ། ཨ་ནུ་སྨྲ་ཏྱ་ནུཏྟམཾ། །རིམ་གྲོ་བྱ་བ་བླ་ན་མེད་པ། རྗེས་སུ་དྲན་པ་བླ་ན་མེད་པ། །དུག་ཀུན། སྡོམ་བཙུན། ན་མེད་ཡ། མ་ཉམས་པར། འདས་་་་་པ། བྱིན་གྱིས་བརླབས་པ་བཞིའི་མིང་ལ། །ལ་ལས་དམ་པ་ས་ཏྱཱ་དྷིཥྛ་ནཾ། ཁྲོ་::ཏྭཱ་གཱ་དྷིཥྛ་ནཾ། ::::ཨུ་པ་ཤ་མཱ་ཏི་བདེན་པའི་བྱིན་གྱིས་བརླབས། གཏོང་བའི་བྱིན་གྱིས་བརླབསི། ཉེ་བར་ཞི་བའི་བྱིན་གྱིས་་་་དྷིཥྛ་ནཾ། པྲ་ཛྙཱ་དྷིཥྛ་ནཾ། །་པའི་དཀའ་དང་བར་་བ:་། ར་ལ་ས་བརླབས། ཊིཤེས་རབ་ཀྱི་བྱིན་གྱིས་བརླབསི། ་ཆུང་། དང་བའི། །བྱ་དང་བ། །—།
