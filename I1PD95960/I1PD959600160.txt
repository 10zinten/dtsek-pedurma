་ག་ཎ་ཀ། ཁྱོ་ཏི་ཥ། བྷི་ཥཀྲི། བིདྱ། ཏ་ལ་:བརྐ། བཱ་རི་དྷཱ་ཛི་ཀ། རྩིས་མཁན། སྐར་མཁན། སྨན་པ། སྨན་པ། སྐུ་སྲུང་ཚེ། རྒྱལ་མཚན་ཐོགས་པ། ཙྪཏྟཱ་དྷ་ར། ཙིནྡྷ་དྷ་ར། ཙཱ་མ་རི་ཀི། ཀཱ་ར་བཱ་ལི་ཀ། ཁཱརྨི་ཀ། གདུགས་ཐོགས་པ། རྟགས་ཐོགས་པ། རྔ་ཡབ་ཐོགས་པ། རལ་གྲི་ཀྱོག་པོ་ཐོགས་པ། རལ་གྲི་པ། པཱ་སི་ཀ། པཱནཱ་དྷི་ཀ། ཙཱ་ཀྲི་ཀ། ཕཱ་རི་ཀ། ཙི་ཏྲ་དཎྜི་ཀ། ཡཱཏྟཱི་ཀ། མདུང་པ། དགྲ་སྟ་ཐོགས་པ། འཁོར་ལོ་ཐོགས་པ། ཕུབ་ཐོགས་པ། དབྱིག་གུ་ཐོགས་པ། དབྱུག་པ་ཐོགས་པ། རྭ་ར་པཱ་ལ། དོ་བཱ་རི་ཀ། ཤེ་ཥ་བ་ཏི༄༅། སཱུ་ཙ་ཀ། དཎྜ་བཱ་སི་ཀ། སེ་བ་ཀ། སྒོ་སྲུང་། །་སྒོ་བ། གཉེར་པ། ལོག་པའམ་འདམ་པོ་པ། ཡུལ་སྲུང་། ཞམ་འབྲིང་བ། བྷཊྚ། །བའི་ཨི་ཤཱསྨཱ་ཙཱརྱ། །ནཱ་སི་ར། ནི་ག་མ། ཛ་ན་པ་ད། ཤོར་བའམ་བཀྲབས་པ། འཕོང་གི་སློབ་དཔོན། སྔ་ཚོམས། གྲོང་མི། ཡུལ་གྱི་མི། དེ་བ་ཀུ་ལི་ཀ། མཱ་ཡཱ་ཀཱ་ར། ཨརྱ། མནྡཱ་བཱ་དཱི། ཀྲྀ་ཡ་བཱ་དཱི། ཁ་རྒྱ་བཱ་དཱི། ལྷ་ཁང་སྲུང་བ། སྒྱུ་མ་མཁན། ཇོ་བོའམ་རྗེའོ༄༅། གསང་སྔགས་སྨྲ་བ། བཅོམ་སེམ་བྱེད་པ། ཏེར་ལྟ་ཤེས་པ། དྷཱ་ཏུ་བཱ་དཱི། ལུབྷཱ་ཀི། མཱཏྨ''ཀ། ཤཱ་ཀུནྟི་ཀ༄༅། ཨོ་ར་ཤྲཱི་ཀ་ཁཊཱི་ཀཚེ། ནོར་སྒྱུར་མཁན། རྔོན་པ། ཉ་པ། །བྱ་བ། ཤན་པ། སོ་ཀ་རི་ཀ། གོ་བྷཱ་ཏ་ཀ། །བཱ་གཱ་རི་ཀ། ཀོཀཱུ་ཊི་ཀ། ནཱ་ཅིག་བནྟ་ཀ། ཕག་འཚོང༌། བ་ལང་གསོད་པ། ་རྒྱས་འཚོང་བ། བྱ་གག་པིའཚོང་བ། གླང་པོ་ཆེ་འཛིན་པ། ནཱ་ག་མཎྜ་ལི་ཀཿ། ཨཱ་ཧི་ཏུཎྜི་ཀ། བྷཾ་ག་རི་ཀ། སཱུ་པ་ཀཱ་ར། ར་ཛ་ཀ། སྦྲུལ་ཚོགས་ཅན། སྦྲུལ་ཁས་བསྡིགས་པ། ཕྱག་ཚང་ཤ་སྐྱེམས་མལ་བ། ཕྱག་ཚང་ཚང་མང་བ། བཙོ་བླག་ཉམཁན། སྦྲ་པ་ཏི། ཏ་ཀླུ་ཀཔ་ལ་གཎྜཔཌྷ་ཀཿ། སཱུ་ཏྲ་དྷཱ་རཿ། ཡཱ་བ་སི་ཀ། ཀཥྛ'ཧཱ་ར་ཀ། ཕྱ་མཁན། ཡཕ་གུ་བརྩིགས་མཁན། ཤིང་མཁན། ཐིག་མཁན། རྩྭ་འཚོང་། ཤིང་ཐུན། །ཅི་ཇིའི—།
