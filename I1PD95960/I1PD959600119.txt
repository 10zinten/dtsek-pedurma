ས་དཱ་ནུ་བདྷོ་བྷ་བ་ཏི། ་པྲ་ཏི་སནྟིམ། སཾ་སཱ་ར། སནྡྷིམྨཱསྣ་ཏི། རྟག་ཏུ་རྗེས་སུ་འབྲང་བར་འགྱུར། ཉིང་མཚམས་སྦྱོར་བ། འཁོར་བ། འབྲེལ་པར་བྱེད་པ། ས་མ་ནུ་བནྡྷ། ཙིཏྟཱ་ནུ་པ་རི་བཏཱི། ཨ་ནུ་ས་ཧི་ཏཾ། ས་མ་གྲ་སཱ་མ་གི། རྗེས་སུ་འབྲང་བ༄༅། སེམས་ཀྱི་རྗེས་སུ་འབྲང་བ། རྗེས་སུ་འབྲེལ་པ། འདུ་བ་ཚོགས་པ། པྲ་ཏི་སམྦ་དྷཱ་ཏི། ཨ་པྲ་ཏི་སནྟི། ཨ་ནུ་པརྟ་ནཾ། སནྟཱ་ན་བཏཱི། ཉིང་མཚམས་སྦྱོར་བའམ་འབྲེལ་བར་བྱེད་པ། མཚམས་སྦྱོར་བ་མེད་པ། རྗེས་སུ་འཇུག་པ། རྒྱུ་ལ་ཨེའབྲས་བུ་ཡོད་པ། ...རྒྱུད་ལ་ཡོད་པ་སྐད་དོད་དང་འདི་འགྲིག་ཀྱང་དོན་ཐོབ་ཀྱིས་དཀྱུས་བཞིན་འགྱུར་བཅོས་པའང་མཚར། ཨཱ་ནུཙྪ་བི་ཀ། ཨ་ནུ་སནྣི། སཱ་ནུ་ཙཱ་ར། :པྲ་བཱ་ཧི། ཨ་ནུ་བནྡྷ། འཕྲོད་པའམ་འབྱོར་པ། མཚམས་སྦྱོར་བ། འཁོར་དང་བཅས་པ། རྒྱུན། རྒྱུན་ཆགས་པ། ཨ་ཛ་སྲམ། ་་་་་་་ཨ་ནུ་ག་ཏ། ་་་་་་ཨ་བྷཱུ་ཏྭཱ་བྷཱ་བ། རྒྱུན་མི་འཆད་པའམ་ཉཁོར་མོར༄༅། རྗེས་སུ་སོང་བའམ་རྗེས་སུ་ཞུགས་པའམ་རྗེས་སུ་མཐུན་པ། མ་བྱུང་བ་ལས་བྱུང་བ། ཡ་ཐཱ་ཡོ་གཾ། ཨ་སཾ་ཀྲྀ་ཏཾ། པྲ་ཏི་སཾ་ཥཱ་ནི་རོ་དྷ། ཨ་པྲ་ཏི་སཾ་ཊུ་ནི་རོ་དྷ། ཅི་རིགས་པར། འདུས་མ་བྱས། སོ་སོར་བརྟགས་པའི་འགོག་པ། སོ་སོར་བརྟགས་པ་མ་ཡིན་པའི་འགོག་པ། སུ་ྋསྐྲྀ་ཏ། སཱ་ས་ཅེབ། ་ཨ་ནཱ་སྲ་བ། ལ་བཀོད། ཁཊཱི། ལཀྱིའདུས་བྱས། ཟག་པ་དང་བཅས་པ། ཟག་པ་མེད་པ། །།ག་ཏེ། ཆགས་པ་དང་ཞེན་པའི་རྣམ་པའི་མིང་ལ། །མ་ལ་་འར་ཎ། ཁ་་ཁ་་ཁ་ཁོདང་རཀྟ། སཀྟ། ཤྲིདྡྷ། གྲོ་ཐི་ཏཾ། མུཙྪི་པེཏཾ། ཨ་དྷི་པབ་སི་ཏ། ཆགས་པ། ཞེན་པ། ་ཆུམས་པའམ་མཁྲོན་པ། མདུད་པ་བོར་བ། བརྒྱལ་བ། ལྷག་པར་ཆགས་པ། ཨ་རྒྱ་བ་སཱ་ན་མཱ་པནྣི། ཨ་བྱ་ེབ་སཱ་ནཾ། བི་ནི་བནྡྷཾ། པ་རི་གྲེ་དྷ། སཾ་རཱ་ག། ལྷག་པར་ཆགས་པར་གྱུར་པ། ལྷག་པར་ཞེན་པ། རྣམ་པར་བཅིངས་པ། ཡོངས་སུ་ཞེན་པ། ཀུན་དུ་འདོད་ཆགས། །
