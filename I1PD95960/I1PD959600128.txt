ཁ་ལ་ཨ་ཤ་མ་ཏྱཾ། ཨ་པི་ཊཾ་ཙཱ། ཨ་མཱ་ཊཾ་ཛྷཱི། ཨ་དྷརྨ་ཀ་མ། པཱ་པ་དྷརྨ། དགེ་སྦྱོང་དུ་མི་འཛིན་པ། ཕར་མི་འཛིན་པ། མར་མི་འཛིན་པ། ཆོས་མ་ཡིན་པ་འདོད་པ། སྡིག་པའི་ཆོས་ཅན། ཛཱ་ཏྲ་ཀཱ་མ། བ་ཧུ་ཀྲྀཏུཾ༌། བ་ཧུ་ཀ་ར་ཎཱི་ཡ། པྲ་ཛྙཱ་:དོརྔ་། དུཤི་ལ། དུབྦ་ཛྷཱ། ཤེས་ཀྱི་ཁེ་འདོད་པ། བྱ་བ་མང་བ། བྱེད་པ་མང་བ། ཤེས་རབ་ཞན་པ། ཚུལ་ཁྲིམས་འཆལ་པ། ཤེས་རབ་འཆལ་པ། མནྡ་པྲ་ཛྙཱ། དུབློ་ཥ་ཏཱ། དུབྷ་ར་ཏཱ། སཾ་ཀཱིཎོ་བི་ཧ་ར་ཏི། ཨཱ་མི་ཥ་ཀིཾ་ཙིངྐ་ཋཧེ་ཏོ༄༅། ཤེས་རབ་ཞན་པ། གསོ་དཀའ་བ། དགང་དཀའ་བ། འདྲེས་པར་གནས་པ། ཟང་ཟིང་ཅུང་ཟད་ཀྱི་ཕྱིར། དུཐུ་ཏཾ། པ་ད་པ་ར་མ། ཨ་སཾ་པྲཀྲ་ཁྱུ་'ེནཾ། ཨ་བྷཱ་བི་ཏ་ཀཱ་ཡ། མཉན་ཉེས་པ། ཚིག་ལ་འཆེལ་པ། མི་གསལ་བ། ལུས་མ་བསྒོམས་པའམ་ལུས་མ་སྦྱངས་པ། པྲ་གམྦྷེདནྣཱི། མུ་ཁ་ར། ཨ་ཀཱ་དྷཱཏྟ། གྷསྨ་ར། མཱཏྨ་ར། སྤྱིས་བརྩན་པའམ་སྤྱི་རྟོལ་ཆེ་བ། མུ་ཅོར་སྨྲ་བའམ་ཁ་དྲག་པ། རྒྱན་པོ་ལ་དོར་བ། ལྟོ་ཆེ་བ། སེར་སྣ་ཅན། ཀ་ད་ཡཾ། ཁ་ལེ། ཤ་ཋ། དྷཱནྟིབི་ཊ། མཱ་ཡཱ་བཱི། ཀུ་ར་ཀུ་ཙིདམྦྷཀུ་སེཧ་ནཾ། འཇུངས་པ། མི་སུན་པ། གཡོ་སྒྱུ་ཉེཅན། གཡོ་ཅན། སྒྱུ་བྱེད་པ། ངན་གཡོའམ་ཚུལ་འཆོས་པ་ངན་པ། ལ་བ་ནཾ། ནི་མིཏྟི་ཀ་ཏྭཾ། ནི་སྐྱེ་ཤི་ཀ་ཏྭཾ༄༅། ལཱ་བྷེ་ན་ལཱ་བྷ་ནིཥྛ་ད་ནཱ། ཁ་གསག་་། ་གཞོགས་སློངས། ཟླ་ཐོབ་ཀྱིས་འཇལ་བ། རྙེད་པ་ལས་རྙེད་པ་སྒྲུབ་པ། ས་ཏཱ་བི་ཀྲ་ཡ། ཤསྡུ་བི་ཀྲ་ཡ། བི་ཥ་བི་ཀྲ་ཡ། མཾ་ས་བི་ཀྲ་ཡ། མ་ཏྲ་བི་ཀྲ་ཡ། སེམས་ཅན་འཚོང་བ། མཚོན་ཆ་འཚོང་བ། དུག་འཚོང་བ། ཤ་འཚོང་བ། ཆང་འཚོང་བ། དུརྨནུ་ཥདྷོ་གཱ་ནཱ་མ་པཱི་ཡ་སྦྲ་ནཱ་ནི། མ་ཏྲ་པཱ་ནཾ། དྷུ་ཏཾ། །བི་ཀཱ་ལ་ཙནྟཱ། གནོང་མི་བསྐུར་བ། ལོངས་སྤྱོད་བྲི་བའི་གནས་དྲུག། ཆང་འཐུང་བ། རྒྱན་པོ། མཚན་མོ་འཕྱན་པ། བཱ་ེབ་མི་ཏྲ་ཏཱ། ས་མཱ་ཛཱ་དན་ནཾ། །ཨཱ་ལ་སྩོ༄༅། །ཁོ་ལ་བསལ་བར་འདོད་པ་སྡིག་པའི་གྲོགས་པོ་བསྟེན་ཨེཔ། །འདུས་པ་ལ་ལྟ་བ། ལེ་ལོ། ་་་་་—།
