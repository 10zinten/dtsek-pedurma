འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོའི་རྒྱུད་ཀྱི་མིང་ལ། །མ་ཧཱ་སམྨ་ཏ། རོ་ཙ། ཀ་ལྱཱ་ཎ། བ་ར་ཀ་ལྱཱ་ཎ། :：ཨུ་པོ་ཥ་དྷ། མུངྒཱ''ཏ། མང་པོས་བཀུར་བ། འོད་མཛེས། དགེ་བ། དགེ་མཆོག། གསོ་སྦྱོང་འཕགས། སྤྱི་བོ་སྐྱེས། །མཱནྟཱ་ཙེཏ། ཙཱ་རུ། ཨུ་པ་ཙཱ་རུ། ཙཱ་རུ་མཾ་ཏི། མུ་ཙི། མུ་ཙི་ལིནྡ། ཤ་ཀུ་ནི། ང་ལས་ནུ། མཛེས་པ། ཉེ་མཛེས། མཛེས་ལྡན། བཏང་བ། ་བཏང་བཟུང་། ཤ་ཀུ་ནི། མ་ཧཱ་ཤ་ཀུ་ནི། ཀུ་ཤ། ཨུ་པ་ཀུ་ཤ། མ་ཧཱ་ཀུ་ཤ། སུ་དན་ནི། མ་ཧཱ་སུ་དན་ན། ཤ་ཀུ་ནི་ཆེན་པོ། ཀུ་ཤ། ཉེ་བའི་ཀུ་ཤ། ཀུ་ཤ་ཆེན་པོ། ལེགས་མཐོང་། ལེགས་མཐོང་ཆེན་པོ། བཱ་མ་ཀ། ཨོ་གི་རཿ། དྷྭཾ་གིུ། མེ་རུ། ཙྩཾ་ཀུ། པྲ་ཎཱ་ད། མ་ཧཱ་པྲ་ཎཱ'ིད། གནོད་སེལ། ཤེས་ལྡན། ངན་སྤོང་། ལྷུན་པོ། ངེས་འགྲོ། རབ་ཏུ་སྒྲ་བསྒྲགས། རབ་ཏུ་སྒྲ་བསྒྲགས་ཆེན་པོ། ཤཾ་ཀ་ར། དི་ཤམྦ་ཏི། རེ་ཎུ། བྷ་ར་ཏ། མ་ཧཱ་དེ་བ། ནེ་མི། བྷི་མ། བདེ་བྱེད། ཕྱོགས་བདག། རྡུལ་བཟང༌། རྒྱས་བྱེད། ལྷ་ཆེན་པོ། མུ་ཁྱུད། འཇིགས་བྱེད། བྷི་མ་ར་ཐ། ཤ་ཏ་ར་ཐ། ད་ཤ་ར་ཐ། པཉྩ་ལ་རཱ་ཛཱ། ཀ་ལཾ་ག་རཱ་ཛཱི། འཇིགས་བྱེད་ཤིང་རྟ། ཤིང་རྟ་བརྒྱ་པ། ཤིང་རྟ་བཅུ་པ། ལྔ་འཛིན་གྱི་རྒྱལ་པོ། ཀ་ལིང་ཀའི་རྒྱལ་པོ། ཨསྨ་ཀ་རཱ་ཛཱ། ཀོ་ར་བ་རཱ་ཛཱ། ཀ་པཱ་ལ་རཱ་ཛཱ། ་གེ་ཡ་རཱ་ཛཱི། ཨསྨ་ཀའི་རྒྱལ་པོ། སྒྲ་མི་སྙན་གྱི་རྒྱལ་པོ། ཐོད་པ་ཅན་གྱི་རྒྱལ་པོ། ཀླུ་དབྱངས་ཀྱི་རྒྱལ་པོ། མ་ཨེག་དྷ་རཱ་ཛཱ། །:ཏ་མཱ་ལིཔཱ་ཀ་རཱ་ཛཱི། ཨི་ཀྵ་ཀུ་ནཱ་མ་ེརཱ་ཛཱི། པི་རཱུ་ཌྷ་ཀ། མཱ་ག་དྷའི་རྒྱལ་པོ། ཏམ་ལི་སྟའི་རྒྱལ་པོ། རྒྱལ་པོ་བུ་རམ་ཤིང་སྐྱེས། འཕགས་སྐྱེས་པོ། སིཾ་ཧ་ཧ་ནུ། ཤུདྡྷོ་ཉིད་ན། ཤུཀློ་ད་ན། དྲོ་ཎོ་ད་ན། ཨ་མྲྀ་ཏོ་ད་ན། སིདྡྷརྦ། སེང་གེ་འགྲམ། ཟས་གཙང་མ། ཟས་དཀར། བྲེའོ་ཟས། བདུད་རྩི་ཟས། །དོན་གྲུབ། ནནྡ༑ཏིཥྛ། བྷ་དྲ་ཀ། མ་ཧཱ་ནཱ་མ། ཨ་ནི་རུདྷ། ཨ་ནནྡ། དགའ་བ། རྒྱལ། བཟང་ལྡན། མིང་ཆེན། །མ་འགགས་པ། །ས་ལ་ཀུན་དགའ་བོ། །སྒྲ—།
