སྤྱཡུ་བཱ། ཁ?pད་ཧྲ། མི་ཡུ་བ་ཏི། ང་ཀུ་མཱ་རི། །ལང་ཚོ་ལ་བབ་པ། དར་ལ་བབ་པའམ་སྐྱེས་ཕྲན། ན་ཆུང་། གཞོན་ནུ་མ། རྒས་པ་དང་ན་བའི་རྣམ་པའི་མིང་ལ། ཀུཊྚ། :ཛཱི་ཥྛ་ཨིཏཱ། བ་ལི་པྲ་ཙུ་ར་ཏཱ། པཱ་ལི་ཏྱཾ། དཎྜ་ཨ་བཏྟཾབྷ་ན་ཏཱི། སྒྱུར་པོ། འཁོགས་པ། གཉེར་མ་མང་བ། མགོ་སྐྱ། འཁར་བ་ལ་བརྟེན་པ། ཁཱ་ལི་ཏུཾ། ཀཱ་སི། ཁུ་རུ་ཁུ་རཱ་བ་སཀྟ་ཀཎྞ༄༅། པཱབྷཱ་རེ་ཎ་ཀཱ་ཡེ་ན་དཎྜ་མ་སྤྱི་ཐེར། ལུད་པ་ལུ་བ། མགུལ་སྒྲ་ངར་ངར་པོས་འགགས་པ། ལུས་སྒུ་བས་འཁར་བ་ལ་བཋབྷ༈དྷ་མ་ནི་སནྟ་ཏ་གཱ་ཊཾ། ཛཱིནྡྷེ། པྲིདྡྷི། མ་ཧམ་ཀ། བརྟེན་པ། ལུས་རྩ་ཕྱིར་འཐོན་པས་ཁེབས་པ། རྒས་པའམ་རྙིང་པའམ་རྙིས་པ༄༅། རྒན་པོ། ནས་ཀྱི་གཟེར་བ། ག་ཏ་ཡོ་བ་ནི། བ་ལི་ནི་ཙི་ཏ་ཀཱ་ཡ། བི་བྷཱུསྠཱ། ཨཱ་ཏུ་རཿ༄༅། །དར་ཡོལ་བའམ་ཕྱེད། ལུས་གཉེར་མས་གང་བ། སྒྱུར་བ། ནད་པའམ་ན་བ། ་ག་བའི་བ་ལང་རི། པ་ཁགྱི་དབང་ཕྱི་ཁ་ཡུལ་གྱི་མིང་ལ། །དེ་བཱ་བ་ཏཱ་ར། བཱ་རཱ་ཎ་སཱི། བི་ཤཱ་ལཱི། པ་རི་ནིཊཱ་ཎཾ། རཱ་:ཛ་གྲི་ཧཾ། ལྷ་ཡུལ་ནས་བབས་པ། ཡུལ་བཱ་རཱ་ཎ་སཱིའཉེ། ཡངས་པ་ཅན། ཡོངས་སུ་མྱ་ངན་ལས་འདས་པ། རྒྱལ་པོའི་ཁབ། བེ་ཆུ་བ་ནཾ། ཀོ་ཤམྦྷིའོ། ཤཱ་བསྟི། ཨ་ནཱ་ཐ་པཎྜ་ཨིད་སཱུ་རཱ་མ། ཛེ་ཏ་འོད་མའི་ཚལ། ཀོ་ཤམྦི། མཉན་ཡོད། མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་ར་བ། རྒྱལ་བུ་རྒྱལ་བ་ནཾ། པུཎྜ་བནྡྷ་ནཾ། བོ་དྷི་མཎྞ། :གྲིདྡྷ་ཀུ་སིཊ་པརྦ་ཏི། ག་ཡ་ཤྲིནྟཾ། བྱེད་ཀྱི་ཚལ། ལི་ཁ་ར་ཤིང་འཕེལ། བྱང་ཆུབ་སྙིང་པོ། བྱ་རྒོད་ཕུང་པོའི་རི། ་ག་ཡ་མགོའི་རི། སྒྱུ་ཎོ་པ་སྒྱུ་ཎོ་གྲྭ་མོ། ཀ་པི་ལ་བསྟུ་ན་ག་རཾ། མ་ཐུ་རཱ། ཤྲཱི་ན་ལནྡཱ། ཀ་བ་དང་ཉེ་བའི་ཀ་བའི་གྲོང་། སེར་སྐྱའི་གནས་ཀྱི་གྲོང་། ཌབཙོ་བླགན། དཔལ་ན་ལེན་ད། །—།
