ད་རིས་མ་བཅུ་དྲུག་པ། བབ་ཊི་དམ་པའི་ཆོས་ཀྱི་རྣམ་གྲངས་ཀྱི་མིང་ལ། ་གང་། ལེགས་ཀྱི། །།:། ་་་་་་་་དང་སཱུ་ཏྲ། བགེ་ཡམ། ན་བྱཱ་ཀ་ར་ཎཾ། ནད་གཱ་ཐཱ། ག་པ། ང་མདོའི་སྡེ། དབྱངས་ཀྱིས་བསྙད་པའི་སྡེ། ལུང་དུ་བསྟན་པའི་སྡེ། ཚིགས་སུ་བཅད་པའི་སྡེ། ཨུ་དཱ་ནཾ། ནི་དཱ་ནམ། ཨ་བ་དཱ་ནམ། ::ཨི་ཏི་བྲྀཏྟ་ཀམ། ??ཛཱ་ཏ་ཀམ། ཆེད་དུ་བརྗོད་པའི་སྡེ། གླེང་གཞིའི་སྡེ། རྟོགས་པ་བརྗོད་པའི་སྡེ། དེ་ལྟ་བུ་བྱུང་བའི་སྡེ། སྐྱེས་པ་རབས་ཀྱི་སྡེ། བི་པུ་། ཨདྷུ་ཏ་དྷརྨ། ཨུ་པ་དེ་ཤ༄༅། བྲ་ཧྥ་:ཙམྨཾ། ཨཱ་དོ་ཀ་ལྱཱ་ཎཾ། ཤིན་ཏུ་རྒྱས་པའི་སྡེ། རྨད་དུ་བྱུང་བའི་སྡེ། གཏན་ལ་དབབ་པའི་སྡེ། ཚངས་པར་སྤྱོད་པ། ཐོག་མར་དགེ་བ། མ་དྷེ་ཀ་ལྱཱ་ཎཾ། པརྱ་བ་སཱ་ནེ་ཀ་ལྱཱ་ཎཾ། སཱཐཾ། སུ་:བྱཉྫ་ནཾ། བར་དུ་དགེ་བ། ཐ་མར་དགེ་བ། དོན་བཟང་པོ། ཚིག་འབྲུ་བཟང་པོ། ཀེ་བ་ལཾ། པ་རི་པཱུརྞཙེ། པ་རི་ཤུདྡྷཾ། མ་འདྲེས་པ་སྐབས་གཞན་དུ་འབའ་ཞིག་ཏུའང་ཅིགདགས། ཡོངས་སུ་རྫོགས་པ། ཡོངས་སུ་དག་པ། པརླ་བ་དཱ་ཏཾ། ཨ་བྷི་བ་ད་མཱ་ནཱ་ཨ་བྷི་བ་དནྟི། སཱ་ཥཱ་ཏོ་བྷ་ག་བ་ཏོ་ཡོངས་སུ་བྱང་བ། མངོན་པར་བརྗོད་པས་མངོན་པར་བརྗོད་དོ། །བཅོམ་ལྡན་འདས་ཀྱི་ཆོས་ནི་ལེགས་དྷསཾ་པྲཉྫི་ཀ། ནིརྩྭ་རཿ། ཨཱ་ཀཱ་ལི་ཀཿ། ཨུ་པ་ནཱ་ཡི་ཀ། པར་གསུངས་པ། ཡང་དག་པར་མཐོང་བ། ནད་མེད་པ། དུས་ཆད་པ་མེད་པ། ཉེ་བར་གཏོད་པ། ཨེ་ཧི་པ་བྱི་ཀཿ། པྲ་ཏྱཱཏྨ་བེ་ད་ནཱི་ཡོ་བི་ཛྷི༄༅། ་སྭཱ་ཌུ་ཏོ་བྷ་ག་འདི་མཐོང་བ་ལ་དོན་ཡོད་པ། མཁས་པ་རྣམས་ཀྱིས་རང་གིས་ཚེརིག་པར་བྱ་བ། བཅོམ་ལྡན་འདས་ཀྱི་བ་ཏོ་དྷརྨ་བི་ན་ཡསུ་པྲ་བེ་དི་ཏ། མ་ནེནཱ་ཎི་ཀསཾ་བོ་དྷི་ལེགས་པར་གསུངས་པའི་ཆོས་འདུལ་བ་ནི་ལེགས་པར་བསྟན་པ། ངེས་པར་འབྱུང་བ་རྫོགས་པའི་བྱང་་་་།
