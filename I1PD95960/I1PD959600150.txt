ཀཱ་ལ་ཀ། ཡ་མ། ཤ་མ་ཎི། མཎྜཱུ་ཀ། མ་ཎི་ཙཱ་ཌ། ཨ་མོ་གྷ་དན་ན། ནག་པོ། གཤིན་རྗེ། དགེ་སྦྱོང་། སྦལ་པ། གཙུག་ན་ནོར་བུ། དོན་ཡོད་མཐོང་། ཨི་ཤཱ་དྷཱ་ར། :：ཙི་ཏྲ་(སེ་ན། མ་ཧཱ་པཱ་ཤ༄༅། ཀྵ་མཾ་ཀ་ར། མ་ཧཱ་ཕ་ཎ་ཨེཀ། གཤོལ་མདའ་འཛིན། སྣ་ཚོགས་སྡེ། ཞགས་པ་ཆེན་པོ། བདེ་བྱེད། གདེངས་ཀ་ཆེན་པོ། གམྦྷི་ར་ནིགྷོ་ཥ། མ་ཧཱ་ནི་ན་དཱི། པི་ནཇི་ཏི། མ་ཧཱ་བི་ཀྲ་མ། བྷུ་ཛཾ་ག་མ། དབྱངས་ཟབ། ཆར་སྒྲོགས། རྣམ་པར་དགྲ་འབྱིན། རྩལ་ཆེན། ལག་འགྲོ། མ་ཧཱ་:བ་ལེ། བིསཱུ་རྫི་ཏཿ། བི་སྤུ་ཊ་ཀི། པྲ་ཡཱ་ཊ་ཀ། མེ་གྷ་སཾ་བྷ་བེ། སྟོབས་ཅན། མཐུ་བསྐྱིངསེ། རྣམ་པར་འཇོམས། རབ་འཇོམས། སྤྲིན་བྱུང༌། སྭ་སྟི་ཀ། བཅུ་དྷཱ་ར། མ་ཎི་ཀཎྚི། སུ་པྲ་ཏིཥི་ཏ། ཤྲཱི་བྷ་དྲ། བཀྲ་ཤིས་ལྡན། ཆར་པའི་རྒྱུན། མགུལ་ན་ནོར་བུ། ཤིན་ཏུ་བརྟན་པ། དཔལ་བཟང་། མ་ཧཱ་མ་ཎི་ཙཱུ་ཌི། :：ཨི་རཱ་པ་ཎ། མ་ཧཱ་མཎྜ་ལི་ཀ། ཨིནྡྲ་ཝིཡུ་དྷ་ཤི་ཁཱི། གཙུག་ན་ནོར་བུ་ཆེན་པོ། ཅེས་སྲུང་བགི་བུ། ་དཀྱིལ་འཁོར་ཆེན་པོ་ཅན། འཇའི་གཙུག་ཕུད་ཅན། ཨ་བ་བྷཱ་ས་ན་ཤི་ཁཱི། ཨིནྡྲ་ཡཉྩི། ཛམྦུ་དྷཱ་ཛཱ། བྷཱི་ཏི་ཛཱ། ཤ་ཤི་ཏ་སྣང་ལྡན་གཙུག་ཕུད་ཅན། དབང་པོའི་མཆོད་སྡོང་། ཛམ་བུའི་རྒྱལ་མཚན། དཔལ་གྱི་གཟི་བརྗིད། ཟླ་བའི་ཛཱ། ཙཱུ་ཌ་ོམ་ཎི་ཝིདྷ་ར། ཨིནྡྲ་དྷཱ་ཛཱི། ཚོ་ཏཱིར་ས། སོ་མ་དན་ནོ། །གཟི་བརྗིད། གཙུག་ན་ནོར་བུ་ཐོགས་པ། དབང་པོའི་རྒྱལ་མཚན། སྐར་མ་ལ་དགའ་བ། ཟླ་མཐོང་། །པ་གནོད་སྦྱིན་གྱི་དབང་པོའི་མིང་ལ། ་། བི་ཤྲ་བ་ཎ། མ་ཧཱ་གྷོ་ཥ་ཤྭ་ར། དྷ་ར་ཎི་སུ་རོ་དྲྭ་ཡུ་དྷ། མ་ཧཱ་མ་ཏི། རྣམ་ཐོས་བུ། སྒྲ་ཆེན་པོའི་དབང་ཕྱུག། ས་སྟེང་མཚོན་ཆ་མཆོག་ཏུ་མི་ཟད་པ། བློ་གྲོས་ཆེན་པོ། །ཨརྩི་ནེ་ཏཱ་དྷི་པ་ཏི། བཛྲ་དྲྀ་ཌྷ་ནེ་ཏྲཿ། ཨ་གྲ་བཱི་ར་:བཱ་ཧེུ། མ་ཧཱ་སེ་ན་བྱུ་ཧ་པ་འོད་འཕྲོ་མིག་བདག། རྡོ་རྗེ་ལྟར་སྲ་བའི་མིག། མཆོག་ཏུ་དཔའ་བའི་ལག་པ། དམག་ཆེན་པོའི་—།
