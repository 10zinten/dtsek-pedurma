གྱིས་མདོར་བསྡུས་པ་དང༌། བྱེད་པའི་ཚིག་དྲུག་གིས་མདོར་བསྡུས་པ་དང༌། ཁྱད་པར་གྱི་མདོར་བསྡུས་པའོ། །དེ་རྣམས་ནི་ཚིག་མདོར་བསྡུས་པ་ཙམ་གྱི་སྦྱོར་བ་ཡིན་ནོ། །བྱེ་བྲག་བསྡུ་བ་ནི་གཉིས་ཏེ། ཚིག་སྔ་མ་བྱེ་བྲག་ཏུ་སྦྱོར་བ་དང༌། ཚིག་ཕྱི་མ་བྱེ་བྲག་ཏུ་སྦྱོར་བའོ། །དེ་དག་ནི་དངོས་པོ་ཀུན་སྤྱི་དང་བྱེ་བྲག་དང་ལྡན་པའི་ཕྱིར་་་སྤྱིར་བྱེ་བྲག་བསྡུས་པའི་སྦྱོར་བ་ཡིན་ནོ། །ཡན་ལག་བསྡུ་བ་ནི་བཞི་སྟེ། འབྲེལ་པ་དང༌། གོ་རིམ་དང༌། དོན་དང༌། གྲགས་པའི་ཡན་ལག་བསྡུ་བའོ། །དེ་རྣམས་ནི་དངོས་པོ་ཀུན་གཙོ་བོ་དང་ཡན་ལག་དང་ལྡན་པའི་ཕྱིར་གཙོ་བོ་ལ་སོགས་པའི་ཡན་ལག་ཅི་རིགས་སྡུད། «སྣར་བསྡུ་བ་དང་། ...ཟླས་དབྱེ་བ་བསྡུ་གཅིག་བརྟན་པས་གྲུབ་པ་...ཡན་ལག་ཅི་རིགས་སྡུད། «པེ་»«སྣར་གསལ་བ་དང༌། མི་གསལ། ཡོ«པེ་»སོར། །«པེ་»«སྣར་»བུ། «པེ་»«སྣར་བསྡུས་པ། «པེ་»«སྣར་»འཐུན་པ་དང་། «པེ་»«སྣར་»ཏེ། «པེ་»«སྣར་»ཡི། «པེ་»«སྣར་»དི། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»+གྱུར་པ། མི«པེ་»《སྣར་»+ངག་དང་མིག་དང་ཆུ་དག་དང་། «པེ་»«སྣར་པ། @«པེ་»«སྣར་»དྲངས། «པེ་»«སྣར་»སྒྲུབ། «པེ་»«སྣར་»པར་སྒྲུབ། «པེ་»«སྣར་»+ཆོས་གཅིག་དགག་པའི་སྒོ་ནས་ནི་མ་ཡིན་པ་བསྒྲུབ་པའོ། །འགྱེ«པེ་»«སྣར་»ལས། །«པེ་»«སྣར་»གྱིས། ཚེ«པེ་»«སྣར་+གཅིག་ལ། W«པེ་»«སྣར་»གཉིས་ཀ«པེ་»«སྣར་»གྲུབ། «པེ་»«སྣར་»གཉིས་ཀ«པེ་»«སྣར་»ལ་ནི། «པེ་»«སྣར་»+པ། «པེ་»«སྣར་»+ཡིན་ལ་ཁྱད་པར་མ། ཆེབཀྱེ«པེ་»«སྣར་»འབྱུང། མེ«པེ་»«སྣར་»+ཅན། «པེ་»«སྣར་»སྒྲའི། «པེ་»«སྣར་»རྟེན། «པེ་»«སྣར་»ཕྱི། «པེ་»«སྣར་»+གི། ཚེ«པེ་»«སྣར་དྡྷཾ། །«པེ་»«སྣར་»སྒྲ་གཅིག་ཤེས་པ་དང་། །ཚེ«པེ་»«སྣར་»གཅིག«པེ་»«སྣར་»+ཤེས། ཚེ«པེ་»«སྣར་»+སྒྲ་ཇི་སྙེད་པ་ཤེས་པས་ཕྱོགས་ཀུན་ཤེས་པ་དང༌། @«པེ་»«སྣར་»བསྒྲུབ། ཚ«པེ་»«སྣར་»པའོ། «པེ་»«སྣར་བའང་། «པེ་»«སྣར་»དབྱིངས་ཁོང་དུ། «པེ་»«སྣར་»སྒྲ་བོ། («པེ་»«སྣར་»+མིང་གི་ཚིག་ནི་འཐུན་པ་དང་། ཚེ«པེ་»«སྣར་»ཡང་། …«པེ་»«སྣར་པ་དང་། ཨེ«པེ་»«སྣར་»འཐུན། «པེ་»«སྣར་»གི། «པེ་»«སྣར་གྲུ། «པེ་»«སྣར་»+ཆེན། གས་བ་བ་ན་བ་མ་ས་དང་ཟེ་ར་ཁོ་གུ་པ་པོ་ཀཀགཞི་བདག་བནབ་་pད་ན་ཁ་ལོ་བ་ཚོས་ཀུན་ན་བདེ་བར་གྱུར་ཏེ་ཀ་ཀྐ་ཙིp?nང་ཚོਇਸਐriਸਨਉਹਕਿਸੇਸਬਸnBਮਤੋਂਵਾਰਵnrਈਉਸནས་ཁོ་ཚོ་དགའ་བ་སྤྱོད་པའི། བོད་ཀྱི་་་་ང་མ་བའི་། །བདུད་ཁཆཥ་ཎ་བར་ཀརཎི། ་་་་་འཁྲུ་ར་ཎཏ་ཏན་ད་པ་ནས་བར་དོ་བ་ལ་བྱེ་བྲག་ཇི་།
