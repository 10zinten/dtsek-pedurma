ཏ་ཀེ་ཤ༄༅། ནཱ་ག་ཀེ་ཤ། །ཧ་རི་ཀེ་ཤ། ཀ་པི་ལ་ཀེ་ཤ། ,ཨ་ཀེ་ཤ་ཀ། བ། སྒྲ་བ་ལང་གི་སྤུ་འདྲ་བ། སྐྲ་སེར་སྐྱ། ཊསྒྲ་སེར་སྐྲ་མེད། གྷ་ཊཱ་ཤི་ར། ཁ་བཏྟཱ་ཤི་རཱི། ཀེ་ས་ཨ་ཏི་སྒྱུ་ལ། །རྗེ་ཨ་ཏི་ཧྲ་སཱ། མགོ་འབར་འབུར་ཅན། །གླང་པོ་ཆེ། ཧ་ཅང་སྦོམ་པ། །དེ་ཧ་ཅང་ཐུང་བ། ཨ་ཏི་དཱི། ཀྲྀ་ཤཱ་ལ་ཀ། བི་ཀ་ཊ་ཀ། ནཱི་ལཙྪ་བི་བཙྪཱ། པཱི་ཏཙྪ་བི་ཧ་ཅང་རིང་བ། སྐེམ་པོ། ལུས་ཉམས་པ། ཤ་མདོག་སྔོ་ཆེས་པ། ཤ་མདོག་སེར་བཙྪ༑ལོ་ཧི་ཏཙྪ་བི་བཎྞ། ཨ་བ་དཱ་ཏཙྪ་བི་བཙྪ། :：བེ་པཱ་ཊ་ཀ། ཁ་ཆེས་པ། ཤ་མདོག་དམར༔ཤས་ཆེ་བ། ཤ་མདོག་དཀར་ཞེས་པ། ལུས་སླེབས་པ་འདྲ་བ། མགོ་ར་ཤིནྟ། སཱུ་ཀ་ར་ཤྲཱིརཿ། དཱི་ཤྲིནྟ། ཤཱ་ཁཱིརཿ། ཨ་ཤཱནྟ་ཀཿ༄༅། ཧསྟི་བོང་བུའི་ལྟ་བུ། མགོ་ཕག་གི་ལྟ་བུ། མགོ་གཉིས། མགོ་ཁྱིའི་ལྟ་བུ། མགོའུ་ཆུང་། རྣ་བ་གླང་པོ་ཆེ་ཀཎ། ཨ་ཤྭ་ཀཎྞ། གོ་ཎ་ཀཎྞི། མརྐ་ཊ་ཀཎྞི། ཁ་ར་ཀཎྞ། སཱུ་ལྟ་བུ། རྣ་བ་རྟའི་ལྟ་བུ། རྣ་བ་གླང་ཅགི་ལྟ་བུ། རྣ་བ་སྤྲེའུའི་ལྟ་བུ། རྣ་བ་བོང་བུའི་ལྟ་བུ། རྣ་བ་ཀ་ར་ཀཎྞ། ཨེ་ཀ་ཀཎྞ། ཨ་ཀཎྞ། ལོ་ཧི་ཏཱ་ཀྵ། ཨ་ཏི་བནྡྷ་ཀྵ། ཙུམྦྷ་ཝིཀྐ། ཕག་པའི་ལྟ་བུ། རྣ་གཅིག་པ། རྣ་མེད་པ། མིག་དམར་བ། མིག་ཧ་ཅང་ཆེ་བ། མིག་ཆུང་ལ། ཨ་ཏི་པིངྒ་ལཱ་། ཀཱ་ཙཱ་ཀྵ། སྐནྡྷ་ཀྵ། བུདྨ་དཱ་ཀྵ། ཨེ་ཀཱ་ཀྵ། ཨ་ན་ཀླུ་ཀཉྩི། མིག་ཧ་ཅང་སེར་ཅིབ། ཤེལ་མིགཕྲག་མིག། མིག་ཆུ་བུར་ལྟ་བུ། མིག་གཅིག་པ། མིག་མེད་པ། ཨ་ཤཱ་ནཱ་ས། ཧསྟི་ནཱ་ས། གོ་ཎ་ནཱ་ས། :མརྐ་ཊ་ནཱ་སི། ཁ་ར་ནཱ་ས། སྣ་རྟའི་ལྟ་བུ། སྣ་གླང་པོ་ཆེའི་ལྟ་བུ། སྣ་བ་གླང་གི་ལྟ་བུ། སྣ་སྤྲེའུའི་ལྟ་བུ། སྣ་བོང་བུའི་ལྟ་བུ། སཱུ་ཀ་ར་ནཱ་ས། ཨེ་ཀ་ནཱ་ས། ཨ་ནཱ་ས། ཧསྟི་ཇོ་ཌ། ཨ་ཤྭ་ཇོ་ཌ། སྣ་ཕག་གི་ལྟ་བུ། སྣ་སྦུབས་གཅིག་པ། སྣ་མེད། ཀོསྐོ་གླང་པོ་ཆེའི་ལྟ་བུ། ཀོསྐོ་ཉེརྟའི་ལྟ་བུ། གོ་ཎ་ཛོ་ཌ། མརྐ་ཊ་ཛོ་ཌ། ཁ་ར་ཛོ་ཌ། སཱུ་ཀ་ར་ཇོ་ཌ། ཀོསྐོ་ཉེབ་ལང་གི་ལྟ་བུ། ཀོསྐོ་སྤྲེའུའི་ལྟ་བུ། ཀོསྐོ་ཉེབོང་བུའི་ལྟ་བུ། ཀླུ་ཀོསྐོ་ཕག་གི་ལྟ་བུ། —།
