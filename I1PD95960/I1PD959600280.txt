དང་པ་ཨ་བྷི་ཀླུ། །ཛ་དོག་ལཔ་ད་དྲཨ་ཤ་མ་ཎ། ལས་འགན་དགེ་སློང་དུ་མི་རུང་བའམ་དགེ་སློང་མ་ཡིན་པ། དགེ་སྦྱོང་དུ་མི་རུང་བའམ་དགེ་སྦྱོང་མ་ཡིན་པ། ཨ་ཤཱཀྱ་པུ་ཏཱི་ཡ། ཝཱ་ཏི་ཀྲ་མ་དྷཱ་སྱ་ཏེ་བྷིཀྐུ་བྷྱཱ་བཱཏྲ། ཤཱཀྱའི་བུར་མི་རུང་བའམ་ཤཱཀྱའི་བུ་མ་ཡིན་པ། དགེ་སློང་གི་དངོས་པོ་ལས་ཉམས་པར་འགྱུར། ཧ་ཏ་མ་སྐྱ་བྷ་བ་ཏི་:ཤཱ་མ་ཙྩཾ༄༅། །ཀུན་ད་དྷཱ་སྟ། ་མ་ཐི་ཏེ༄༅། དེའི་དགེ་སྦྱོང་གི་ཚུལ་ཞིག་པར་འགྱུར། ཉམས་པ། བཅོམ་པ། པ་ཏི་ཏཾ། པ་རཱ་ཛི་ཏཾ། ཨ་པྲ་ཏྱཱདྷཱརྨ་མ་སྐྱ་བྷ་:བ་ཏི་ཤཱ་མ་ཏྱཾ། ཏདྱ་ཐཱ་ཏཱ་ལྟུང་བ། ཕམ་པར་གྱུར་པ། དེའི་དགེ་སློང་གི་ཚུལ་ཕྱིར་བསླང་དུ་མེད་པར་འགྱུར། དཔེར་ན་ཤིང་ཏ་ལེ་ེམས་ཀཱཙྪིནྡ་ཨ་བྷ་བྱོ་ཧ་རི་ཏ་ཏཱ'ཡ། འདུཤཱི་ལ། པཱ་ལའི་མགོ་བཅད་པ་བཞིན་སྔོན་པོར་འགྱུར་དུ་མི་རུང་བའམ་མི་བཏུབ། ཚུལ་ཁྲིམས་འཆལ་པ། སྡིག་པ་དྷམྨ། ཨནྟིཔཱུ་ཏིིཨ་པ་ཐུ་ཏ། ཀ་ཤམྦ་ཀ་ཛཱ་ཏ། ཤཎ་སྣ་ར་ས་པའི་ཆོས་ཅན། ཁོང་མྱགས་ཤིང་ཕྱིར་འཛག་པ། ཤིང་རུལ་པ་ལྟ་བུར་གྱུར་པ། ལུས་བོང་ལྟར་མཱ་ཙཱ་། ཨ་ཤ་མ་ཎཤ་མ་ཎ་པྲ་ཏི་ཛཱི། ཨ་བྲ་ཧྥ་ཙཱ་རི་བྲ་ཧྥ་ཙཱ་རི་པྲ་སྤྱོད་པ། དགེ་སྦྱོང་མ་ཡིན་པར་དགེ་སྦྱོང་དུ་ཁས་འཆེ་བ། ཚངས་པར་སྤྱོད་པ་མ་ཡིན་པར་ཚངས་པར་སྤྱོད་ཏི་ཛྙཱ། ཤཱི་ལ་བི་པནྣ། བྲྀཏྟི་བི་པནྣ། ཨཱ་ཙཱ་ར་བི་པནྣ། ཨཱ་ཛཱི་བ་བི་པནྣ། པར་ཁས་འཆེ་བ། ཚུལ་ཁྲིམས་ཉམས་པ། ལྟ་བ་ཉམས་པ། ཆོ་ག་ཉམས་པ། འཚོ་བ་ཉམས་པ། ཤིཀཱ་བ། །མི་ནི་སྐུ་:ཚེ་ལ་ཁ་བདེ་བ་བསླབ་པ་ཞིག་པ། ཚnཥ། དབཏབློའི་ཁ་བ་འབབ་ས་ན། མ，་འཆལ་པ། —།
