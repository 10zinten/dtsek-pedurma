དྷརྨི་ཀཿ། །ཛེ་ནོ་ར་ས། དྷརྨ་ཏོ་ནིརྒ་ཏ། ནིརྫི་ཏ། མུ་ཁ་ཏོ་ཛཱ་ཏཿ། །ཆོས་དང་ལྡན་པ། རྒྱལ་བའི་ཐུགས་ཀྱི་སྲས། ཆོས་ལས་བྱུང་བ། ་ངེས་པར་རྒྱལ་བ། ཞལ་ནས་སྐྱེས་པ། །ICN::བྱང་ཆུབ་སེམས་དཔའ་སོ་སོའི་མཚན་ལ། ཁྱེད་ཀྱི་འདུན་པ་བར་གྱི་དབྱེན་ཀ་བ་དང་ཚོ་བོད་ཨ་བ་ལོ་ཀི་ཏེ་ཤྭ་ར། མི་ཏྲེ་ཡ། ཨཱ་ཀཱ་ཤ་གབྷ། ས་མནྟ་བྷ་དྲ། སྤྱན་རས་གཟིགས་དབང་ཕྱུག། བྱམས་པ། ནམ་མཁའི་སྙིང་པོ། ཀུན་དུ་བཟང་པོ། བཛྲ་པཱ་ཎི@། མ་ཤྲཱི་ཀུ་མཱ་ར་བྷཱུ་ཏ། སརྦ་ནི་བ་ར་ཎ་བི་སྐམྦྷི། ཀྲི་ཏི་གབྷ། ལག་ན་རྡོ་རྗེ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ། སྒྲིབ་པ་ཐམས་ཅད་རྣམ་པར་སེལ་བ། སའི་སྙིང་པོ། མ་ཧཱ་སྠཱ་མ་པྲསྟ། རཏྣ་ཀེ་ཏུ། རཏྣ་པཱ་ཎི། རཏྭཱ་མུ་དྲྭ་ྋཧསྟ། རཏྣ་མཐུ་ཆེན་ཐོབ། རིན་པོ་ཆེའི་ཏོགལག་ན་རིན་ཆེན། ལག་ན་ཕྱག་རྒྱ་རིན་པོ་ཆེ། ཅོད་པན་མུ་ཀུ་ཊ། :རཏྣ་ཙཱུ་ཌ། རཏྣ་ཀུ་ཊ། རཏྟཱ་ཀ་ར། རཏྣ་ཤི་ཁ་ར། རིན་པོ་ཆེ། གཙུག་ན་རིན་པོ་ཆེ། རིན་ཆེན་བརྩེག་པ། དཀོན་མཆོག་འབྱུང་གནས། རིན་པོ་ཆེའི་རྩེ་མོ། རནྣཱ་དྷཱ་ཛི། བཛྲ་གབྷ། སུ་བཎྜ་གབྷ། རནྣཱ་གབྷ། ཤྲཱི་གབྷ། རིན་ཆེན་རྒྱལ་མཚན། རྡོ་རྗེའི་སྙིང་པོ། གསེར་གྱི་སྙིང་པོ། རིན་པོ་ཆེའི་སྙིང་པོ། དཔལ་གྱི་སྙིང་པོ། ཤུ་བྷ་གབྷ། ཤུ་བྷ་བི་མ་ལ་གབྷ། ཏ་ཐཱ་ག་ཏ་གདྷ། ཛྙཱ་ན་གབྷ། དགེ་བའི་སྙིང་པོ། དགེ་བ་དྲི་མ་མེད་པའི་སྙིང་པོ། དེ་བཞིན་གཤེགས་པའི་སྙིང་པོ། ཡེ་ཤེས་ཀྱི་སྙིང་པོ། སཱུརྨ་གབྷ། ས་མཱ་དྷི་གནྡྷ། པདྨ་གནྡྷ། བི་མུཀྟི་ཙནྡྲ། ས་མནྟ་ནེ་ཏྲ། ཉི་མའི་སྙིང་པོ། ཏིང་ངེ་འཛིན་གྱི་སྙིང་པོ། པདྨའི་སྙིང་པོ། རྣམ་པར་གྲོལ་བའི་ཟླ་བ། ཀུན་དུ་མིག། པདྨ་ནེ་ཏྲཿ། བི་མ་ལ་ནེ་ཏྲ། བི་ཤྭ་ལ་ནེ་ཏྲཿ། ས་མནྟེན་པ་ཐ། ས་མནྟ་:པྲ་པདྨའི་མིག་། དྲི་མ་མེད་པའི་མིག། ཡངས་པའི་མིག། ཀུན་དུ་སྤྱོད་ལམ། ཀུན་ནས་་་སཱ་དི་ཀཿ། ཙཱ་ན་བ་ཏཱི། ན་ས་མནྟ་ཙཱ་རི་ཏྲ་མ་ཏི། ཏ་ཛ་ཡ་མ་ཏི། མཛེས། ་་་ཡེ་ཤེས་ལྡན་པ། །ལ་ཀུན་དུ་སྤྱོད་པའི་བློ་གྲོས། རྒྱལ་བའི་བློ་གྲོས། །
