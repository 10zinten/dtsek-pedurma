རཾ་ཥ་བྷ། ནཱ་ཡ་ཀཿ། པ་རི་ཎཱ་ཡ་ཀཿ། བི་ནཱ་ཡ་ཀཿ། ཨ་དྭ་ཡ་ཁྱུ་མཆོག། འདྲེན་པ། ཡོངས་སུ་འདྲེན་པ། རྣམ་པར་འདྲེན་པ། གཉིས་སུ་བཱ་དཱི། སོ་དྷོ་ད་ནི། ད་ཤ་པ་ལི། མཱ་ར་ཛིཏྲ། ཤཱཀྱ་པུངྒ་བ། མེད་པར་གསུང་བ། ཟས་གཙང་གི་སྲས། སྟོབས་བཅུ། བདུད་ཐུལ། ཤཱཀྱ་སྐྱེས་མཆོག། གོཔ༑མ་ཧཱདྷཱ། བི་ཛ་ཡཱི། བི་བྷུ། བི་ཤཱནྟ་ར། སྲུང་མཛད། བདག་ཉིད་ཆེན་པོ། རྣམ་པར་རྒྱལ་བ་མངའ། ཁྱབ་བདག། ཐམས་ཅད་སྒྲོལ། སརྦ་:དྷརྨེ་ཤཱ་ར། བི་རཱཊི། དྷི་ར། གུ་ཎ་སཱ་ག་རཿ། ཤ་ར་ཎྱ། ཆོས་ཐམས་ཅད་ཀྱི་དབང་ཕྱུག། རྣམ་པར་མཛེས། བརྟན་པ། ཡོན་ཏན་རྒྱ་མཚོ། སྐྱབས་འོས། ཤ་ར་ཎམ། བཱ་དི་སིཾ་ཧ། ན་རོཏྟ་མ། མཱ་རཱ་བྷི་བྷུ། ཨ་པྲ་ཏི་པུངྒ་ལ། བཱནྟདོ་སྐྱབས། སྨྲ་བའི་སེང་གེ། མི་མཆོག། བདུད་ཟིལ་གྱིས་གནོན། གང་ཟག་ཟླ་མེད་པ། སྐྱོན་བསལ་ཥ༑ཧ་ཏ་བི་ཥ། ཨ་ནངྒ་ཛིཏྲ། ཥ་ཌ་བྷི་ཛྷ། བྷ་བཱནྟ་ཀྲྀཏྲ། བ། དུག་བཅོམ་པ། ལུས་མེད་ཐུལ། མངོན་པར་ཤེས་པ་དྲུག་དང་ལྡན་པ། སྲིད་མཐར་བྱེདི། ཨ་གྷ་ཧནྟ། སིངྒྷནྡ༄༅། ཤཱཀྱ་སིཾ་ཧ། བ་རཱདྷི། བ་ར་ད། བཱི་རཿ། ཤ་མི་སྡིག་འཇོམས། དོན་གྲུབ། ཤཱཀྱ་སེང་གེ། མཆོག་དོན་འོས། མཆོག་སྦྱིན། དཔའ་བ། ཞི་བར་ཏཱ༑ཤཱནྟ་:པཱ་པི། ཤཱི་ཏཱི་བྷཱུ་ཙིཏ། ཤི་བི་ཀ་རི། ནིརྡནྡྲ། ནིརྨ་མ། མཛད་པ། སྡིག་པ་ཞི་བ། བསིལ་བར་གྱུར་པ། ཤིས་མཛད། རྩོད་མེད། བདག་གི་བ་མེད་པ། ནེ་ཏཱ། ནི་ར་བ་རྱ། ནིབྷ་ཡ། དེ་བཱི་ཏ་ཀྲྀཏྟཱཙེ། ཀུ་ནི་རཱ་་་་མཚོན་པ། ཁ་ན་མ་ཐོ་བ་མི་མངའ་བ། བསྙེངས་པ་མི་མངའ་བ། ་སྲེད་པ་དང་བྲལ་བ། ལེན་པ་མི་དཱ་ན། བི་ཐུ་ཏ། ཤུ་བྷ་དྷརྨཱ་ཀ་ར། ་ཤུ་ཙི། ཨ་ནུ་པ་མ། མངའ་བ། རྣམ་པར་གྲགས་པ། དགེ་བའི་ཆོས་ཀྱི་འབྱུང་གནས། གཙང་བ། དཔེ་མེད་པ། །
