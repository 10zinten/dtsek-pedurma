བྱེ་བྲག་ཏུ་རྟོགས་པར་བྱེད་པ། དེ་ནཱ་མ་ས་མཱ་དྷི། ས་མཱ་དྷི་ས་མ་ཏཱ་ནཱ་མ་ས་མཱ་དྷི། ཨ་ར་ཛོ་བི་་་་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། ཏིང་ངེ་འཛིན་མཉམ་པ་ཉིད་ཅེས་བྱ་བའི་ཏིང་ངེ་འཛིན། རྡུལ་མེད་ཅིང་ར་ཛོ་ན་ཡ་ཡུཀྟ་ནཱ་མ་ས་མཱ་དྷི། བལ་ཨ་ར་ཎ་ས་མ་བ་ས་ར་ཎོ་རྡུལ་དང་བྲལ་བའི་ཚུལ་དང་ལྡན་པ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། ཉོན་མོངས་པ་མེད་པར་ཡང་དག་པར་ནཱ་མ་ས་མཱ་དྷི། ཨ་ར་ཎ་ས་ར་ཎ་སརྦ་མིས་མ་བ་ས་འདུ་བ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། ཉོན་མོངས་པ་མེད་པ་དང་ཉོན་མོངས་པ་དང་བཅས་པ་ཐམས་ཅད་ར་ཎ་ནཱ་མ་ས་མཱ་དྷི། ཨ་ནི་ལཾ་བྷ་ནི་ཀེ་ཙེཏ་ནི་ཀ་ཏོ་ནཱ་མ་ཡང་དག་པར་འདུ་བ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། གནས་ལ་བརྟེན་པ་མེད་པ་ལ་བརྩོན་པ་ཞེས་བྱ་བའི་ས་མཱ་དྷི། ཏ་ཐ་ཙིཏཱ་སྦི་ཏ་ནིཤྲཱིཏྟཾ་ནཱ་མ་ས་མཱ་དྷི། ཀཱ་ཡ་ཀ་ལི་སཾ་ཏིང་ངེ་འཛིན། དེ་བཞིན་ཉིད་ལ་གནས་ཤིང་སེམས་མེད་པ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། ལུས་ཀྱི་སྐྱོན་པྲ་:མརྦ་ནོ་ནཱ་མ་ས་མཱ་དྷི། ཏིཏབཱཀྐ་ལཾ་བི་དྷྭཾ་ས་ན་ག་ག་ན་ཀལྤ་་་་་ཡང་དག་པར་སེལ་བ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། ངག་གི་སྐྱོན་རྣམ་པར་འཇིག་པས་ནམ་མཁའ་ལྟར་གྱུར་པ་ཞེས་བྱ་བའི་ནཱ་མ་ས་མཱ་དྷི། ཨཱ་ཀཱ་ཤཱ་སཾ་ག་བི་མུཀྟི་ནི་རུ་:པ་ལེ་པོ་ནཱ་མ་ས་མཱ་དྷི། །།།ཏིང་ངེ་འཛིན། ནམ་མཁའ་ལྟར་ཆགས་པ་མེད་པས་རྣམ་པར་གྲོལ་ཞིང་གོས་པ་མེད་པ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན། དར་བྱང་ཆུབ་སེམས་དཔའ་སྤྱིའི་མཚན་ལ་སོགས་པའི་མིང་ལ། བྱང་ཆུབ་སེམས་དཔའ་སྤྱིའི་མཚན་ལ། །བོ་དྷི་ས་ཏྭ། མ་ཧཱ་ས་ཏྭ། དྷི་མཱན། ཨུཏྟ་མ་དྷུ་ཏི། ཛི་ན་པུ་ཏྲ། བྱང་ཆུབ་སེམས་དཔའ། སེམས་དཔའ་ཆེན་པོ། བློ་ལྡན། གསལ་མཆོག། རྒྱལ་བའི་སྲས། ཛི་ནཱ་དྷཱ་ར། བི་ཛེ་ཏཱ༄༅། །ཛི་ནཾ་ཀུ་རཿ། བི་ཀཱནྟེ། པ་ར་མཱརྨི། རྒྱལ་བའི་གཞི། །རྣམ་པར་རྒྱལ་བྱེད། རྒྱལ་བའི་མྱུ་གུ། རྩལ་ལྡན། འཕགས་མཆོག། སཱརྦ་བཱ་ཧ། །མ་ཧཱ་ཡ་ཤཱ། ཀྲྀ་པཱ་ལུ། མ་ཧཱ་པུ་:ཎྱ། ་ཨཱི་ཨེཤཱ་རཿ། དེད་དཔོན། གྲགས་ཆེན། སྙིང་རྗེ་ཅན། །བསོད་ནམས་ཆེ། དབང་ཕྱུག། །
