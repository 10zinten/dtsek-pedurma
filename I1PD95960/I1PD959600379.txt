ཡང་རྣམ་པ་གཅིག་ཏུ་ན་་་་་སརྦ་ཀཱ་ལ་སུ་ཁ་ཙ་:ཏཱ་ར་ཨ་ཀཱ་ལི་ཀཱ་ཞེས་བྱ་སྟེ། སྤྱོད་ལམ་བཞི་པོའི་དུས་ཀུན་ཏུ་བྱ་བར་སླ་བས་དུས་ཆད་པ་མེད་པ་ཞེས་བྱ། ཨོ་པ་ན་ཡི་ཀཱ་ཞེས་བྱ་བ་ཨ་ཤེ་ཥ་དུཁ་ནིཅུ་ཎ་ཨུ་པ་ན་ཡ་ནཱད་ཨོ་པ་ན་ཡི་ཀཱ་ཞེས་བྱ་སྟེ། སྡུག་བསྔལ་མ་ལུས་པ་ལས་འབྱུང་བ་ཉེ་བར་སྤྱོད་པའི་ཕྱིར་ཉེ་བར་གཏོད་པ་ཞེས་བྱ། ཨི་ཧི་བ་་་་ཀ་ཞེས་བྱ་བ་ཨ་སཱ་:དྷཱ་ར་ཎ་ཨཏཱད་ཨེ་ཧ་པ་བྱི་ཀ་ཨི་ཧ་པ་བྱེཎྜ་བྷ་བ་ཞེས་བྱ་སྟེ། ཆོས་འདི་མཐོང་བ་རྣམས་ལ་ཡོད་པས་མུ་སྟེགས་ཅན་དང་ཐུན་མོང་མ་ཡིན་པའི་ཕྱིར་འདི་མཐོང་བ་ལ་ཡོད་པ་ཞེས་བྱ། ཡང་རྣམ་པ་གཅིག་ཏུ་ནི། ཙྪ་ཨཱ་ཙཱརྱ་མུཉྫི་ར་ཧི་ཏ་ཏཱད་ཨི་ཧ་པ་བྱ་ཏི་པྲ་བཱ་ར་ཎ་ཡོ་གཱད་ཅེས་བྱ་སྟེ། སློབ་དཔོན་་་འཆབ་པ་མེད་པས་འདི་ལ་ལྟོས་ཤིག་ཅེས་སྐབས་འབྱེད་པ་དང་ལྡན་པའི་ཕྱིར་འདི་ལྟོས་ལྡན་ཞེས་ཀྱང་བྱ་སྟེ། ཚིག་རྣམ་པ་གཉིས་སུ་དྲང་དུ་རུང་གིས་ཀྱང་དོན་མེ་སྔོན་མ་དང་སྦྱར་ཏེ་བཤད་ན་བཟང་ཞིང་ཚིག་ཀྱང་བདེ་སྟེ། འདི་མཐོང་བ་་་་ཡོད་པ་ཞེས་བཏགས། པྲ་:ཏྱཱཏྨ་བེ་ད་ནི་ཡོ་བི་ཛྷ་ཞེས་བྱ་བ་ཨ་ཏརྐྱ་ཏཱད་པྲ་་་་ཏཏྠ་བེ་ད་བི་ཛཱི་ཞེས་བྱ་སྟེ། རྣམ་པ་གཅིག་ཏུ་ན་རྟོག་གེའི་ཡུལ་མ་ཡིན་ཏེ། ཤེས་རབ་ཅན་རྣམས་ཀྱིས་རྟོགས་ཀྱི་པར་བྱ་བ་ཡིན་པ་དང༌། ཡང་རྣམ་པ་གཅིག་ཏུ་ན་ཨ་ན་བྷི་ལ་བྱ་ཏཱད་པ:ྲཏདྨ་བེད་ནི་ཨིཡ་ཞེས་བྱ་སྟེ། བརྗོད་དུ་མེད་པའི་་་་ཕྱིར་:སོ་སོར་རང་གི་རིག་པར་ཟད་པས་མཁས་པ་རྣམས་ཀྱིས་རིག་པར་བྱ་བ་ཞེས་བྱ། སཱ་ལྱཱ་ཏོ་བྷ་ག་བ་ཏོ་དྷརྨ་བི་ན་ཡ་:སུ་པྲ་བེ་དི་ཝིན་ཏ་ཞེས་:པ་ཙཱ་ཙེཡ་ཏཱད་ཨ་བི་པ་:རཱི་ཏཟླ་ཏྲེ་ན་དྷརྨ་ཀླུ་ཤ་བི་ན་ཡ་ཞེས་བྱ་སྟེ། བཅོམ་ལྡན་འདས་ཀྱིས་་་ལེགས་པར་གསུངས་པའི་ཆོས་འདུལ་བ་གཞན་ཕྱི་རོལ་པ་དང་མི་འདྲ་བར་ལེགས་པར་བསྟན་ཅིང་གོ་བར་བྱས་ཏེ། རིགས་ཅིཔ་དང་ལྡན་པས་ན་ཆོས། ཉོན་་་—།
