ཀ་ར་བྷ། མ་ཧཱ་ཀ་ར་བྷ། ཨིནྡ། མ་ཧེནྡེ། ས་མཱy། ་མ་ཧཱ་ས་འོད་མཛེས། འོད་མཛེས་ཆེན་པོ། དབང་པོ། དབང་པོ་ཆེན་པོ། ལེགས་ཕྱིནི། །ལེགས་ཕྱིན་མཱསྟ༑ག་ཏི། མ་ཧཱ་ག་ཏི། ནིལླ་ར་ཛ། མ་ཧཱ་ནིམ་ར་ཛ། མུ་དྲེ། མ་ཆེན་པོ། རྟོགས་འགྲོ། རྟོགས་འགྲོ་ཆེན་པོ། འབྱིང་རྡུལ། འབྱིང་རྡུལ་ཆེན་པོ། རྒྱ་རྟགས། རྒྱ་ཧཱ་མུ་དཱི། བ་ལཾ། ཀ་མ་ཧཱ་བ་ལཾ། །སཾ་ཛཱ། མ་ཧཱ་སཾ་ཛཱ། བི་བྷཱུ་ཏ། རྟགས་ཆེན་པོ། སྟོབས་འཁོར། སྟོབས་འཁོར་ཆེན་པོ། །བརྡ་ཤེས། བརྡ་ཤེས་ཆེན་པོ། རྣམ་འབྱུང་། མ་ཧཱ་བི་བྷཱུ་ཏ། བ་ལཱཀྵ། མ་ཧཱ་བ་ལཱགྷཾ། ཨ་སཾ་ཏཱཾ། ཨ་པྲ་མཱ་ཎཾ། ཨ་པྲ་རྣམ་འབྱུང་ཆེན་པོ། སྟོབས་མིག། སྟོབས་མིག་ཆེན་པོ། གྲངས་མེད་པ། ཚད་མེད་པ། དཔག་ཏུ་མེ་ཡཾ། ཨ་པ་རི་མི་ཏཾ། ཨ་པ་རི་མཱ་ཎཾ། ཨ་ཏུལླཾ། ཨ་མཱ་པཾ། ཨ་ཙིནྟཾ། །།།མེད་པ། བསྐྱོད་དུ་མེད་པ། དཔག་ཡས། གཞལ་དུ་མེད། གཞལ་གྱིས་ཉམི་ལང་བ། བསམ་གྱིས་མི་ཁྱབ་པ། ཡེ། ཥ་ཊ་ཀྱི་བདེ་འཇིག་རྟེན་པའི་གྲངས་ཀྱི་མིང་ལ། །ཨེ་ཀཾ། ད་ཤ། ཤ་ཏཾ། ས་ཧ་སྲྀ། ཨ་ཡུ་ཏཾ། ལགྷཾ། ནི་ཡུ་ཏཾ། ཀོ་ཊཱི། ཨརྱ་དཾ། གཅིགབཅུ། བརྒྱ། སྟོང་། ཁྲི། འབུམ། ས་ཡ། བྱེ་བ། བྱེ་བ། རྒྱཆུ་དཾ། པÉི། ཁཐཾ། ནི་ཁཐཾ། མ་ཧཱ་:པཎཾ། ཤངྐུ། ས་མུ་་དུང་ཕྱུར། ཐེར་འབུམ། ：ཤུ་རྡོགས། མཆོག་ཉལི། སྐྱང་འཕྱེས། བྱེ་མ་ནབ་ནུབ། ཐིག་འབུམ་མམ་དྲོ། །མ་དྷཾ། ཨཏྟ། པ་རཱདྷ། ཨེ་ཀ། :：དྲོ། ཏཱི་ཎི༄༅། ཙ་ཏཱ་རི། པཉྩ། ཥཊ། མཚོ་ཡས། ལྷབས། སྦྱོར། རྫི། ལྔ་གཅིག། གཉིས། གསུམ། བཞི། ལྔ། དྲུག། སསྟ༑ཨཏྠོ། ན་བི། ད་ཤ། ཨེ་:ཀ་ད་ཤི། དྭཱ་ད་ཤ། ཏྲ་ཡོ་ད་ཤ*། བདུན། བརྒྱད། དགུ། བཅུ། ཀླུབཅུ་གཅིག། བཅུ་གཉིས། ་བཅུ་གསུམ། ཙ་ཏུརྡ་ཤི། ་པཉྩ་ད་ཤ། ཁོ་ཌ་ཤ། སལྟ་ད་ཏི། ཨཏྟཱ་ད་ཤམེ། བཅུ་བཞི། །འཁྲིnབཅོ་ལྔ། །བཅུ་དྲུགལ་། བཅུ་བདུན། བབཅོ་བརྒྱུད། །
