གཉིས་རྣམ་པར་དབྱེ་བ་བྱས་པས། བསོད་ནམས་བསགས་པ་ཡོངས་སུ་བསྔོ་བའི་ཕྱིར། །བདེན་པ་གཉིས་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཡེ་ཤེས་སྙིང་པོ་་ནི་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཡིན་ནོ། །བདེན་པ་གཉིས་རྣམ་པར་འབྱེད་་་་པའི་དཀའ་འགྲེལ་སློབ་དཔོན་ཞི་བ་འཚོས་མཛད་པ་རྫོགས་སོ། །།ཚ་ཁོལརྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཝརྨ་དང༌། །ཙཱ་ན་གབྷ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་བནྡེ་ཡེ་ཤེས་སྡེ་ལ་སོགས་པས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །་། ਨੇਨੂੰਸਬਸਿਡਸੋਸਨੂੰਰਲਾਸਿੰਗਲਵੀਰདགའ་བ་ས་ལ་བློ་ཁ་འགས། །ཁ་སོགས་ལ་ལད་ཤོད་ཀྱི་ཁ་དོག་ངོ་་བཀང་བའི་བསྡུར་མཆན། རྗེ་ལོ་ངོ་ལོ་ཀོ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། ®«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་བསྡུས། ༄༅«པེ་»«སྣར་པའི་ཕྱིར་རོ། །«པེ་»«སྣར་པར། «པེ་»«སྣར་ནི། @«པེ་»«སྣར་རོ། །«པེ་»«སྣར་+པོ། ཚེ«པེ་»«སྣར་»ཏོ། །@«པེ་»ཀྱིས། ཨ«པེ་བརྟན། ང་དེར་མི་བསང་ཨེ«པེ་»«སྣར་»ན། «པེ་»«སྣར་+སེམས་དཔའ། «པེ་»«སྣར་»དེ། «པེ་»«སྣར་བརྟགས། «པེ་《སྣར་ཀྱིས། ®«པེ་»«སྣར་»—ལ། «སྣར་བཟུང༌། «པེ་»«སྣར་གཞགཚུ«སྣར་»ལྟར། «པེ་»«སྣར་»ཞེ། «པེ་»«སྣར་»ཏེ། བཀའ་བར་མཉྫིལ་རྫོགས་པ་ལ་«པེ་»«སྣར་»—མི། «པེ་»«སྣར་»འབྲངས་པས། «པེ་»«སྣར་+བ། «པེ་»«སྣར་»གི། ཉི«པེ་»«སྣར་»+སྨོས་པ། «པེ་»ཀྱི་ཐུགས། «པེ་»+དང་གཞན་གྱི། ཀུན་། «པེ་»«སྣར་རྗོད། ་་ས་་ས་བྱ་བ་ལསབབག་ཡ་དང་སྐལ་པ༌«པེ་»«སྣར་བརྣན། «པེ་»«སྣར་བྱུང་། «པེ་»«སྣར་བརྣན། «པེ་»«སྣར་»པ། @«པེ་»«སྣར་རིག་པས་སྤྱད། ༄༅«པེ་»«སྣར་བརྣན། «པེ་»«སྣར་བསླུ། （ཆོས་ཚན་་་། འདིའི་ནང་དུ་སླུ་ཞེས་པ་རྣམས་«པེ་»«སྣར་»ལ་བསླུ་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་ཏུ་བཀོད་མེད། ོབ་ཀྱི་པེ་»«སྣར་»ཅི། (ཆོས་ཚན་འདིའི་ནང་དུ་ཇི་ལྟར་ཞེས་པ་རྣམས་«པེ་»«སྣར་»ལ་ཅི་ལྟར་ཞེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། ཚེ«པེ་»«སྣར་ཞེས། «པེ་»«སྣར་»པར། «པེ་»«སྣར་»འགྲེལ། «པེ་»«སྣར་»གི། «པེ་«སྣར་»སྨོས་པ། །
