ཉིན་མའི་ཆུ་ཟར་ལས་འབྲས་བུ་དང་བཅས་པར་བྱ་བའི་ཕྱིར་དགེ་བའི་རྩ་བ་བསྔོ་བ་བསྟན་ཏོ། །བསྟན་པ་རྙེད་དང་ཞེས་བྱ་བའི་ཚིག་གཉིས་ཀྱིས་བསྟན་པ་ཡུན་རིང་དུ་གནས་པར་བྱ་བའི་ཕྱིར་བསྔོ་བ་བསྟན་ཏོ། །གང་གི་དྲིན་གྱིས་དགེ་བློ་འབྱུང༌། །།ཞེས་བྱ་་་བའི་ཚིགས་སུ་བཅད་པ་གཅིག་གིས་དྲིན་གཟོ་བའི་སྒོ་ནས་རང་གི་དགེ་བའི་བཤེས་གཉེན་ལ་ཕྱག་འཚལ་བ་བསྟན་ཏོ། །བསྟན་བཅོས་ཀྱི་མིང་གི་དོན་ནི་ལེའུ་དང་་་་པོར་བཤད་པ་བཞིན་ནོ། །ཡོངས་སུ་བསྔོ་བ་ཞེས་བྱ་བ་སྟེ་ལེའུ་དགུ་པའི་བཤད་པ་རྫོགས་སོ། །།།དེ་ན་ཁའ་་་ཀུ་བ་དེ་དག་གི་ཡན་ལག་གི་དོན་ཤིན་ཏུ་གོ་སླ་བའི་ཕྱིར་རྣམ་པར་མ་ཕྱེའོ། །བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་ལ་འཇུག་པའི་རྣམ་པར་བཤད་པ་རྫོགས་སོ། །།།ਕਿਇਈਏਖਤਨਕਰਸਨਕਿਅਗਰਓਨਮੋਨੂੰਨਾབསྡུར་མཆན། ར་བ། རྟ་ར་གན་ན་བོད་ལ་(མཚན་བྱང་འདི་《སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»ལ་༄༅། །སྤྱོད་འཇུག་ཤེར་ལེའི་བསྔོ་བའི་་་དཀའ་འགྲེལ་མཛད་པ། ཞེས་འཁོད། །ཚེ«པེ་»«སྣར་བརྒྱ། Q«པེ་»«སྣར་སྡོགས། དེ་(«པེ་»«སྣར་»ན། «པེ་»«སྣར་གང་ལས། «པེ་»«སྣར་»བུ། «པེ་»«སྣར་དེ། («པེ་+མ། «པེ་»«སྣར་རིག༄༅«པེ་»«སྣར་བློའི། «པེ་»«སྣར་རྟགས། །(«པེ་»«སྣར་གིས། «པེ་《སྣར་»སྐྱེད། ཚེ«པེ་»«སྣར་གསལ། །ཨེ«པེ་»«སྣར་—པ། ཉི«པེ་»«སྣར་ཀྱི་གོ་རིམ། ཚེ«པེ་»«སྣར་པའི། ཆེ«པེ་«སྣར་བ། «སྣར་ཡིན། «པེ་»སྣར་བར་རོ། «པེ་»«སྣར་»ཏོ། «པེ་»«སྣར་རིག«པེ་»«སྣར་གཙང་བ་ལ། («པེ་»«སྣར་གཞི། །ཚེ«པེ་»«སྣར་གཞི། ཉི«པེ་»«སྣར་གཞུགཚེ«པེ་»—མ། «པེ་《སྣར་གྱི། མུ«སྣར་+པ། ༄༅«པེ་»«སྣར་རྒྱུ། །«པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་»+དེ། «པེ་»«སྣར་»ཚེ། «པེ་»«སྣར་»བ། །«པེ་»«སྣར་རིངས། «པེ་»—པོ། «སྣར་གཅད། ︽«པེ་»«སྣར་པ། མ«པེ་»«སྣར་»..་ཀཞེ་།
