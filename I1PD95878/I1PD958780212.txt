སོང་སྦྱོང་བ་ཡི། །ཕྲིན་ལས་ཁྱོད་ཀྱིས་མཛད་ལགས་ཀྱིས། །རང་གི་ཡུལ་དུ་་་གཤེགས་ནས་ཀྱང་། །སླར་ཡང་བྱོན་པར་མཛད་དུ་གསོལ། །ཨོཾ་བྷ་བ་ཀ་ཨག་ཡེ་ཨེ་བྱེ་ཧི་ས་ཧཾ་ད་དཾ་ཙ་ཏི་དྷརྨ་ཙ་ཡ་བཛྲ་མུ། ཞེས་པས་ཡེ་ཤེས་པ་གཤེགས། དམ་ཚིག་གི་མེ་ལྷ་ཐ་མལ་ལྟར་བསྡུའོ། །ཚེ་འདས་སྒྲིབ་པ་སྲེག་པའི་ཐབས། །ཆོ་ག་བཟང་པོ་བྱས་པ་ལས། །བསགས་པའི་དགེ་དེས་འགྲོ་བ་ཀུན། །བདེ་བ་ཅན་དུ་སྐྱེ་བར་ཤོག། སློབ་དཔོན་འཇམ་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པའི་་འཇམ་དཔལ་གྱི་རོ::སྲེག་པའི་ཆོ་ག་:རྫོགས་སོ། །།།།རྒྱ་གར་གྱི་མཁན་པོ་སུ་བཛྲ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ཆོས་ཀྱི་ཤེས་རབ་་་་ཀྱིས་བསྒྱུར་བའོ། །།།ཀལ་ལ་ལ། །བསར་མཆན། :(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་འཕགས་པ་འཇམ་དཔལ་གྱི་མཚན་ཡང་དག་པར་བརྗོད་པའི་སྡིག་པ་སྦྱོང་བའི་སྦྱིན་སྲེག་གི་ཆོ་ག་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»མཐོང་། («པེ་»«སྣར་ཤུགས། «པེ་»«སྣར་»—གང་དག«པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་—དང་། ས«པེ་»«སྣར་»ཤ་ལ་སོགས་པ། «པེ་»«སྣར་བའི། «པེ་»«སྣར་—ཞོ་དང༌། «པེ་»«སྣར་དུར། «པེ་»«སྣར་»—ནས་དང་། ཁྱེ«པེ་»«སྣར་+མ་ཤ་དང༌། «པེ་»«སྣར་»—སྲན་མ་ཤ་དང༌། འབྲས་སོ་བ་དང་། «པེ་»«སྣར་»སྟག་གེ། «པེ་»«སྣར་+པ། «པེ་»«སྣར་བ། ལུ«པེ་»སྣར་ལྕགས། ཁྱེ«སྣར་མ། བྱེ«པེ་《སྣར་»+ནི། «པེ་»«སྣར་ལོས། («པེ་»«སྣར་པ། «པེ་»«སྣར་པོ། «པེ་»«སྣར་བྱུང༌། «པེ་»«སྣར་»པ། །«པེ་»«སྣར་གསུམ། «པེ་»«སྣར་ལ་གུན་དེ། @«པེ་བསྲེག«སྣར་བསྲེགས། «པེ་»«སྣར་རི༑«སྣར་—། «པེ་»«སྣར་»ཧོ་ཏི། །(«པེ་»«སྣར་»ཀ་བ། «པེ་»«སྣར་བྷིམྨཾ། («པེ་»«སྣར་»དུé«པེ་»«སྣར་ཨཱ། (ས«པེ་»བ་ན་ཡ་མ་ཧཱ་བྷཱུ་ཏ། «སྣར་བ་ན་ཡ་མ་ཧཱ་བྷུ་ཏ། «པེ་»«སྣར་བྷསྨི། «པེ་»«སྣར་»དུཋན། —།
