རིན་ཆེན་རིགས་ནི་ཐོབ་པ་དང་། །དོན་དམ་རྟོགས་པ་པདྨའི་རིགས། །ལས་་ཀྱི་རིགས་ཀྱི་དོན་ཆེན་བྱེད། །ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་གྱི་རིམ་པ་ལས། བཅོམ་ལྡན་འདས་འཇམ་དཔལ་གྱི་སྒྲུབ་པའི་ཐབས་སློབ་དཔོན་སོ་མ་་་ཤྲཱིས་མཛད་པ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁས་པ་ཆེན་པོ་པཎྜི་ཏ་མ་ཧཱ་ཙཱ་ན་དང༌། བོད་ཀྱི་:ལོ་ཙཱ་བ་དགེ་སློང་གཞོན་ནུ་འོད་ཀྱིས་བསྒྱུར་བའོ། །།།་བོད་::འ་བ་ན་གྲོགས་་་་་་་་་་་་་་་་ནི་་་་་ལ་་ང་ཡི་བདེ་ཀྱིས་ངེས་བར་བདུན་བར་དུ་མཁོ་བསྡུར་མཆན། ལས་ལ་ཁ་བ་བབས། རྒྱ་གར་བ་དེ། ཁ་བ་ཚ་བོ་དེ་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །འཕགས་པ་འཇམ་དཔལ་སྒྲུབ་ཐབས་སློབ་དཔོན་སོ་མ་ཤྲཱིས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་འདུད། «པེ་»«སྣར་བརྟེན། ་ཚེ«པེ་»《སྣར་»ཡི། ང་(«པེ་»«སྣར་+དེ། ་པ། («པེ་»«སྣར་»སྒྲུབ། («པེ་»«སྣར་»ཀྱང་། «པེ་»«སྣར་—དེ། «པེ་»«སྣར་»ཡི། «པེ་»«སྣར་དཔས། («པེ་»«སྣར་»རྩི། «པེ་»《སྣར་དང་། «པེ་»ཆེ། (ི«སྣར་+ནས། ་་་་་་་་་་་་་་་་་་་་་་་་«པེ་»«སྣར་»—རྣམ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»+དང་། ®«པེ་»«སྣར་»གུས་་་པས་གླུ་ནི་ལེན་པ་ཡི། །་ས«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་གུས་པས་གླུ་ནི་ལེན་པ་ཡི། །«པེ་»:ང་《སྣར་»རྗེ། ར་:འི་ཚོང་། :ཀ་ར་དང་གླང་སྤྱོད། ་ན་་ས་ན་ད་«པེ་»«སྣར་»ལས། «པེ་»«སྣར་བསྒོམས་པས་བརྟན། །«པེ་»«སྣར་ལྟས། །«པེ་»《སྣར་ཀྱི། །«སྣར་བསྟན། «པེ་《སྣར་ཡི། ་«པེ་»«སྣར་བཟླས། «པེ་»«སྣར་»ཀ«པེ་»«སྣར་བཀྱེ། (ཁྱེ«པེ་»«སྣར་»སོདྲུདྡྷ། །(«པེ་»«སྣར་མུ། །(«པེ་»«སྣར་དྷ། མ་ནཱ་བིཏྟི། སྟེ«པེ་»«སྣར་ཡཎྚིཏྟཾ། «པེ་《སྣར་ན། («པེ་»«སྣར་བ་བ་ནིརྱ་ཏ་ཡ་མེ། ཨིཔེ་»::《སྣར་པྲ་ཏེ་གྲིབྷཾ་ཏཱ། ་«པེ་»«སྣར་པུནྱ། «པེ་《སྣར་བ། «པེ་»«སྣར་»ཏ། ནེ«པེ་»«སྣར་ཙིཏྟཾ་པཱ་ད་མུ་པ་ཏ་ཡ། ་«པེ་»«སྣར་»ཏེ་ས། འགེ«པེ་»«སྣར་»སིདྷ་ཡ་སཾ་པཉྩནྟི། ་ན«པེ་»ཀྱི་《སྣར་»ཏ་ཤ་ནས«པེ་»«སྣར་—། ན«པེ་»《སྣར་»ཨདྨ། བཀག་།
