གི་དམ་ཚིག་གི་ཕྱག་རྒྱ་ཉེ་བར་བསྟན་ལ་རང་གི་དབང་བསྐུར་བས་དབང་བསྐུར་ཏེ། ལག་པ་གཉིས་ཀྱིས་ཁུ་ཚུར་བྱས་ལ་མཛུབ་མོ་བརྐྱང་སྟེ། མདུད་པ་བྱ་བ་ནི་རྡོ་རྗེ་བཅིང་བའི་ཕྱག་རྒྱར་འགྱུར་ཏེ། འདི་དང་བཛྲ་མཉྫི་བོ་ཞེས་བྱ་བའི་སྔགས་་་་འདིས་ཀྱང་དཀྱིལ་འཁོར་པའི་ལྷ་རྣམས་བཅིངས་ནས་རང་གི་ཏིང་ངེ་འཛིན་གྱི་་་་ལས་ལ་ཞུགས་པ་ལ་སྒེག་མོ་ལ་སོགས་པས་ཡང་དག་པས་མཆོད་ལ། ཡང་དྲི་ལ་སོགས་པས་མཆོད་པར་བྱའོ། །དེ་ལ་རས་ཟུང་འབུམ་མམ་སྟོང་ངམ་བརྒྱའམ་རེ་རེ་ཞིང་ཟུང་རེའམ། ཐམས་ཅད་ལ་རས་རེ་རེའམ། བླ་རེ་དང་རྒྱལ་མཚན་་་་དང༌།  ོ་བདུགས་དང༌། ཀློབ་དན་སྣ་ཚོགས་པའི་ཡི་གེ་ཧཱུྃ་དང་། བདག་པོའི་སྔགས་བཟླས་པ་དང་། །བཛྲ་སྤ་ར་ཎ་ཁཾ་ལས་བྱ་བ་འདིས་དབུལ་བར་བྱའོ། །དེ་བཞིན་དུ་ཨོ་སཊྚ་ཏ་ཐཱ་ག་ཏ་གནྡྷེ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྤ་ར་ཎ་ས་མ་ཡེ་ཧཱུྃ་ཞེས་བྱ་བའི་སྔགས་ཀྱིས་སྔར་སྦྱར་བའི་ག་པུར་དང་། ཨ་ཀ་རུ་དང་། ་་:ཏུ་རུ་སྐ་ལ་་སོགས་པའི་དྲི་ཙནྡན་ལ་སོགས་པ་དང་བསྲེས་པ་དབུལ་ལོ། །དེ་བཞིན་དུ་ཨོ་སཊྚ་ཏ་ཐཱ་ག་ཏེ་པུབྦ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྤ་ར་ཎ་ས་མ་ཡེ་ཧཱུྃ་ཞེས་བྱ་བ་དང་། །མེ་་་ཏོག་གི་ཤིང་བརྒྱའམ། ཤིང་བཞིའམ། །མེ་ཏོག་ཐམས་ཅད་དབུལ་ལོ། །ཨོ་་སརྦ་ཏ་ཐཱ་ག་ཏ་དྷུ་པ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྤ་ར་ཎ་ས་མ་ཡེ་ཧཱུྃ་ཞེས་བྱ་བ་དང་སྤོས་་་སྣོད་སྟོང་ངམ་བརྒྱ་དབུལ་བར་བྱ་སྟེ། །བཅུ་ལས་ཉུང་བར་ནི་མི་བྱའོ། ་། ཨ་་་་ཁར་ཞེས་བྱ་བ་ལ་སོགས་པ་དང་། སྭསྟི་ཀ་ལ་སོགས་པར་བྱ་བའི་བཟའ་བ་དང་བཅའ་བ་དང་མྱང་བ་དང་། གཞིབ་པ་དང་། བཏུང་བའི་བཤོས་ཀྱི་རྣམ་པ་སྣ་ཚོགས་པ་གྲངས་སྟོང་སྙེད་དབུལ་ལོ། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་དཱི་པ་པཱུ་ཛ་མེ་གྷ་་ས་མུ་དྲ་སྤ་ར་ཎ་ས་མ་ཡེ་ཧཱུྃ་ཞེས་བྱ་བ་དང་། མར་གྱི་མར་མེ་ལ་སོགས་པ་འབུམ་ཕྲག་བརྒྱ་དང་མར་མེ་ཆེན་པོའི་སྣོད་བརྒྱའམ། མར་མེ་ཆེན་པོ་བཞི་དབུལ་ལོ། །—།
