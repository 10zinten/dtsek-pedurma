།་རྒྱུད། །དུ་ཀྱང་ན་ཀཀནརྒྱ་དེ་ནས་འཁོར་ལོ་བསྟན་པར་བྱ། །བཅུ་དྲུག་ཕྱེད་ཕྱེད་ཐིག་ལེ་ཅན། །ཡན་ལག་མེད་པའི་རྩིས་ལས་འདས། །བསམ་གཏན་བཞི་པའི་རྩེ་མོ་ཅན། །ཞེས་པ་འདི་ཡི་དོན་བཤད་ན། །ལུས་ལ་འཁོར་ལོ་བཞི་གནས་ཏེ། །ལྟེ་བར་ཤེས་རབ་འཁོར་ལོ་སྟེ། །བཅུ་གཉིས་རྩིབས་དང་ལྡན་པ་ལ། །ཟླ་། ཨ་ཨཱ་ཨོཾ་ཨཱ?རིམ་བཞིན་དགོད། །དབུས་ན་ཡི་གེ་ཨ་མཆོག་གོ། །སྙིང་གར་ཤེས་རབ་འཁོར་ལོ་སྟེ། །བཅུ་གཉིས་རྩིབས་དང་ཡང་དག་ལྡན། །དེ་ལ་སྔགས་རྣམས་དགོད་པ་ནི༑། ཨ་ཨཱ། ཨི་ཨཱི། ཨུ་ཨཱུ། ཨེ་ཨཻ། ཨོ་ཨཽ། ཨོ་ཨཞེས་པ་འདི་་་་དག་རིམ་བཞིན་དུ། །འཁོར་ལོའི་ལྕེ་ལ་དགོད་པར་བྱས། །མགྲིན་པར་ཤེས་་་་་རབ་འཁོར་ལོ་ནི། །བཅུ་གཉིས་ལྡན་པའི་རྩིབས་ལྡན་ལ། །ཨོཾ་ཨ་ར་པ་ཙ་ན། བཱ་གཱི་ཤྭ་ར་པྲཛྙཱ་ན་མཞེས་པ་རྣམས། །རིམ་བཞིན་ཡང་དག་དགོད་པར་བྱ། །དེ་ནས་མཚོག་མར་བསམ་པ་ནི། །བཅུ་གཉིས་ལྡན་པའི་ལྕེ་ཅན་ལ། །སྐུ་ཡི་སྔགས་ཀྱི་ཕྲེང་བ་དགོད། །།ཨོ་ཨཱ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཧྲི་ད་ཡ། །ཧ་ར་ཧ་ར། ་ཨོ་ཧཱུྃ་ཧཱིབྷ་ག་བཱན་ཛྷཱ་ན་མཱནྟི་བཱ་གཱི་ཤྭ་ར། མ་ཧཱ་པཱ་ཙ་སརྦ་དྷརྨཱ་ཌིག་ག་ན། ཨ་་མ་ལ་སུ་པ་རི་ཤུདྡྷ་དྷརྨཱེདྷཱ་ཏུ་ཛྙཱ་ན་གཉྷཾ་ཨཱ། །ཞེས་པའི་སྔགས་ཀྱི་ཕྲེང་བ་འདི། །ཤེས་རབ་འཁོར་ལོ་གྲགས་པ་ཡིན། །འདི་ལྟར་རྟག་ཏུ་གང་བསྒོམ་པ། །གཞོན་ནུ་གཟུགས་ཅན་བདེ་བ་འཐོབེ། །གྲགས་འཛིན་ལུས་དང་བདེ་ལྡན་ལུས། །་་། དེ་དག་སྤངས་ནས་གཞན་དོན་དུ། ་། རབ་དགའི་ལུས་སུ་བསྐྱེད་པར་ངེས། །།དེ་ནི་བཅུ་གཉིས་ས་རྫོགས་ཏེ། །ཀུན་ཏུ་འོད་ཀྱི་ས་ཆེན་པོ། །དེ་ནི་ཉི་མའི་དཀྱིལ་འཁོར་འདྲ། །བདུད་རྩི་འོད་ནི་གཉིས་པ་སྟེ། །ཤིན་ཏུ་འོད་གསལ་ཟླ་བ་འདྲ། །ནམ་མཁའི་འོད་ནི་གསུམ་པ་སྟེ། །།ནམ་མཁའི་བཞི་བཞིན་སོ་སོར་གསལ། །རྡོ་རྗེའི་འོད་ནི་བཞི་པ་སྟེ། །ས་དེ་ཉིད་ནི་བཞི་པའོ། །བསམ་པ་ནང་གི་བརྟན་_།
