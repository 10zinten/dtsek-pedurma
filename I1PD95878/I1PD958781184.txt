ཐཾ་ལ་མངོན་ཕྱོགས་བྱེད། ཀ། བྱང་དུ་རྟ་མཆོག་སྐུ་མདོག་ལྗང་། །ནམ་མཁའ་ལྡིང་གི་གདན་ལ་བཞུགས། ཡོ། །ཨ་བྱུང་སྣ་ཚོགས་རྡོ་རྗེ་དང་། །དྲིལ་བུའི་སྒྲ་ཡིས་ཕྱོགས་ཀུན་ཁྱབ། །།སཾ་གཾ་ཁཾ་མཾ་གྷོ་ས་བོན་རྣམས། །ཨོཾ་ཧཱུཾ་ཏཱཾ་*ཧིལྡན་པ་ལས། །གདན་འདི་དག་ནི་གོ་རིམས་བཞིན། །མཁས་པས་གྲངས་བཞིན་ཤེས་པར་བྱ། །ཟླ་བའི་གདན་གནས་འདི་རྣམས་ཀྱི། །དང་པོའི་སངས་རྒྱས་གོང་མ་བཞིན། །སྲིད་པའི་འཁོར་ལོ་ཚངས་པ་སྟེ། །སྔ་མ་ལས་ཁྱད་མེད་ཕྱིར་རོ། །།།ཞི་བའི་འཁོར་ལོ་རྩིབས་བརྒྱད་པ། །མི་བསྐྱོད་སོགས་རིགས་རིམ་བཞིན་དུ། །ཨོ་དུཁ་ཙྪེ་ད་ལ་སོགས་དགོད། །རིགས་ནི་ཇི་ལྟར་དེ་ན་གནས། །ན་མ་ཞེས་བྱ་རྩིབས་གཅིག་ཅེལ། འོ། མཁས་པས་ཀུན་ལ་ཤེས་པར་བྱ། །སྡུག་བསྔལ་གཅོད་པ་རྡོ་རྗེ་་་དང༌། །རིན་པོ་ཆེ་ནི་གསུང་དབང་ཕྱུག། རྡོ་རྗེ་རྣོན་པོ་པདྨ་དང་། །ཡེ་ཤེས་སྐུ་ཅན་ཞེས་བྱ་བ། །།སྣ་ཚོགས་རྡོ་རྗེ་གཡོན་པ་ལ། །གཡས་པ་རལ་གྲི་ཁྱད་པར་མེད། །ཀུན་ཀྱང་ཟླ་བའི་གདན་ལ་བཞུགས། །ཟླ་བ་ལྟར་འཇམ་ཆ་བྱད་ཅན། །དང་པོའི་དབྱངས་བྱུང་བཟླས་པ་ཡང་། །སྙིང་པོ་ཉེ་སྙིང་རིམ་པ་བཞིན། །དེ་ལ་ལྔ་པོ་རིམ་བཞིན་དུ། །སྙིང་པོ་དང་ནི་ཉེ་སྙིང་པོ། །ཨོཾ་:པྲཛྙཱ་ཛྷཱ་ན་མཱུརྟ་ཡེ་ནཱ་མ༑ཨོ་དུ༔ཁ་ཙྪེ་ད་ཡ་ཏེ་ནཱ་མ། ཨོ་བཱ་གི་ཤྭ་ར་ཡ་ཏེ་ནཱ་མ། །ཨོཾ་བཛྲ་ཏཱིཀྵ་ཡ་ཏེ་ནཱ་མ། ཨོཾ་ཛྙཱ་ན་ཀཱ་ཡ་ཡ་ཏེ་ནཱ་ལེམ། །་ཉེ་བའི་སྙིང་པོ་ཐུན་མོང་སྟེ། །ཨོཾ་སརྦ་དྷརྨ་ཨ་བ་བ་སྭ་བྷཱ་བ་བི་ཤུདྡྷ་བཛྲ་ཨ་ཨཱ་ཨོཾ་ཨ། པྲ་ཀྲྀ་ཏ་པ་རི་ཤུདྡྷ། །སརྦ་དྷརྨཱ་ཡ། ཎ/། དུ་ཏ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཛྙཱ་ན་ཀཱ་ཡ་མཉྫ་ཤྲཱི་པ་རི་ཤུདྡྷི་ཏ་མུ་:པཱ་དཱ་ཏི་ཡེ་ཏི་ཨ་ཨཱཉྩསརྦ་ཏ་ཐཱ་ག་ཏ་ཧྲི་ེད་ཡ། ཧ་ར་ཧ་ར།  ོཨོ་ཧཱུཾ་ཧཱིབྷ་ག་བཱན་ཙྪཱཛཱ་ན་མཱུརྟི་བཱ་གཱི་ཤཱ་ར་མ་ཧཱ་ཝ་ཙ་སརྦ་དྷརྨ་ག་ག་:ན་ཨ་མ་ཨེལ་ས་པ་རི་ཤུདྡྷ་དྷརྨ་དྷཱ་ཏུ་ཛྙཱ་ན་གབྷ་ཨཱི།  ོ་འདི་རྣམས་ཀྱང་ནི་རིམ་བཞིན་དུ། །ཤར་ལ་སོགས་པར་+།
