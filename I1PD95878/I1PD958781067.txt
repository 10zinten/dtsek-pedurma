ཁ་པ། བོན་ཏེ། །རྡོ་རྗེ་སྐྱིལ་མོ་ཀྲུང་དུ་བཞུགས། བློ། འདབ་བཞི་ཟླ་བའི་སྟེང་ན་ནི། །ཨ་ཨཱ་ཨོཾ་ཨིཡུམ་བཞི་བསྐྱེད། །ཡེ་ཤེས་གཉིས་མེད་བསྟིམ་བྱས་ནས། །བྱིན་བརླབ་དབང་བསྐུར་རྫོགས་པ་དང་། །མཚན་ནི་ཡང་དག་བརྗོད་པར་བྱ། །དེ་ནས་བསྟོད་པ་འདིས་བསྟོད་དོ། །ཨོཾ་ཛྙཱ་ན་ཀཱ་ཡ། ་་:མཱ་ཡ་ཛཱ་ལ་ན་མ་སྐུ་ཧཱུཾ། ནཱ་མ་སྟེ་:བུདྡྷ་ནཱ་ཊ་ཀ། ན་མ་སྟེ་:སརྦ་སརྦ་ཧཱུཾ། ཛྙཱ་ན་ཀཱ་ཡ་ན་མ་སྟུ་ཏེ། ཨོ་ཨཱཧཱུཾ་མུཉྩ། ་ཞེས་སྤྱོད་ལམ་གྱིས་གནས་པར་བྱའོ། །དྲི་མེད་ཡེ་ཤེས་ལས་་་བྱུང་བ། །དགེ་བ་དྲི་མེད་གང་ཐོབ་དེས། །འགྲོ་རྣམས་སྲིད་པ་ལས་གྲོལ་ནས། །བླ་མེད་བྱང་ཆུབ་ཐོབ་པར་ཤོག། དེ་ཁོ་ན་ཉིད་ཀྱི་སྒྲུབ་ཐབས་སློབ་དཔོན་གཉིས་སུ་མེད་པ་སྦས་པས་མཛད་པ་རྫོགས་སོ། །་་ག། པཎྜི་ཏ་དྷརྨ''ཤྲཱི་མི་ཏྲ་དང་། :：ལོ་ཙཱ་བ་དགེ་སློང་ཆོས་ཀྱི་བཟང་པོས་་་་བསྒྱུར་བའོ། །།།གང་མ། ས་ཆུ་ཁྱི་ཟླ། །ས་ར་བའི་ཚེ་ན་བ་ཏི་ཏོ། །ན། བདག་དངོས་ཀྱི་ཆུ་ཏ་ན་པ། དར་གྱི་ལོ་བསྡར་མཆན། དེ་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»༄༅། །འཕགས་པ་འཇམ་དཔལ་ཡེ་ཤེས་ཀྱིས་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་»ཏཏཱ། «པེ་»«སྣར་»ན། «པེ་»«སྣར་»ཤིགས«པེ་»«སྣར་»ཉེ།  ི«པེ་»«སྣར་»སྟོད། «པེ་»«སྣར་»ཏ། འི«པེ་»《སྣར་ཨp། «པེ་»«སྣར་མ། པེ་«པེ་»«སྣར་»ཀའི། །ར་། ་ཀྱི་(«པེ་»«སྣར་ན། «པེ་»«སྣར་»བུདྡྷ། «པེ་»«སྣར་མོ། སུ«པེ་»«སྣར་»སྟི། «པེ་»《སྣར་»ཏ་ག་བ། «པེ་»«སྣར་བ་ཏ། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་_ལ། དགྱེ«པེ་»《སྣར་པད། འགྱེ«པེ་»«སྣར་ཨ། «པེ་»«སྣར་»། ལོ«པེ་»«སྣར་ཝ་གི། «པེ་བུ་དྭ་::ག་ན་མོ་སྲེ་སྟུ། བུ་དྭ་ར་ག་ནཱ་མོ་ནཱ་མ། བུ་དྭ་པྲ་ཏི་ནཱ་མ། བུ་དྭ་བྱི་བུ་ཏ་ན་མ། «སྣར་བུ་དཱ་ག་ན་མོ་སྲ་སྟུ༑...བུ་ཧ་ན་མ། ལེ«པེ་»«སྣར་»སྨོན། ལོ«པེ་»«སྣར་»ཨ་ཨཾ་ཨ། དབྱེ«པེ་»«སྣར་»+བོ། «པེ་»«སྣར་དྲང༌། «པེ་»«སྣར་»བྱིན་བརླབ་དབང་བསྐུར་མཚན་གདོན་ནོ། །དགྱེ«པེ་»ཏིཥྛ། །
