བསྟན་འགྱུར། རྒྱུད། ༼ཅུའི་་དཀར་རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་ཀ་ར་ཝརྨ་དང༌། ཞུ་ཆེན་གྱི་:ལོ་ཙཱ་?བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །ཟེར་། ་ན་ཊ་ལ་ਰਕਮਲੈ/ਸਪੈਸੇਉਸਨੂੰਹਰਇੱਕੌਰਰਾਮਨੇਇਸਤੋਂਕੀਸਿਰਦਸਸਗਲਲਸਤਰੰਸਇਸ਼ਲਿਆਸਰੂਪਭੂਤਾਕਰਸ਼ਕਨੇਉਸਪਾਣੀའབད་་་། བ་ཀུ་ལ། །སྐྱེ་ཀུན་སྐྱེ་བ་དང་། །བླ་མ་སྐུར་བ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»༄༅། །དཔལ་ཐམས་ཅད་གསང་བའི་རྒྱུད་ཀྱི་བཤད་སྦྱར་གསང་བའི་སྒྲོན་མ་ཞེས་བྱ་བ་སློབ་དཔོན་རིན་ཆེན་འབྱུང་་་་་་གས། གནས་ཞི་བས་མཛད་པ་བཞུགས་སོ། །«པེ་《སྣར་བནྡྷེ། «པེ་《སྣར་མའོ། །༄༅«པེ་»«སྣར་ལས། 《སྣར་སྤྲོས་པ། «སྣར་»ཀྱིས། @«པེ་»«སྣར་»གསུངས་པ། །«པེ་»«སྣར་»རྟག«པེ་»«སྣར་བསྒོམ། ོན«པེ་»«སྣར་»རབ་ཀྱི། «པེ་»«སྣར་»གི། (མ«པེ་»«སྣར་»—ཁྱད་པར་དུ། @«པེ་»«སྣར་»དེ། ཚེ«པེ་ཡེ་ཤེས། «པེ་»«སྣར་སོ། «པེ་»«སྣར་»—ཏེ། ད་ནས་ཟླ་རབབནདགུན་ཟླ། :(«པེ་《སྣར་རྫུན། «པེ་»«སྣར་»ཡིན། འ«པེ་»«སྣར་»འདུས། «པེ་»«སྣར་»+གི། །དང་། གྲོ་«པེ་»«སྣར་པོ། «པེ་»«སྣར་»བྱ་བར། །«པེ་»«སྣར་བར། །༼«པེ་»«སྣར་པར༑«པེ་»«སྣར་»བ། ས་རྒྱུ། གླུ། ༼ལེན་པ་དག་དང་། ད་ནyཏུ་བྱེད་(«པེ་»«སྣར་+པ། «པེ་»«སྣར་»འཐོབ། «པེ་»«སྣར་པའི། «པེ་»«སྣར་»ཀྱི། ས«པེ་《སྣར་»ཀྱི། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»དུ། «པེ་»《སྣར་བཟུང༌། །བྱེ་བ། གྱི་ག་ལ་བུ་སྐྱེ་བ་ ེ«པེ་»«སྣར་»ནི། «པེ་»«སྣར་དཀར། «པེ་《སྣར་ཀྱི། ︽«པེ་»«སྣར་»གི། ས«པེ་»«སྣར་»པའི། «པེ་»«སྣར་»+ནི།  ོ ི«པེ་»«སྣར་པར་དང་སྤངས་པར། 《སྣར་»དེ་ནས༑«པེ་»«སྣར་_བ། «པེ་《སྣར་བ། འོང་འབབ། ་་་་་ས་བྱ་«པེ་»«སྣར་ལྟ། «པེ་»«སྣར་—བ། «པེ་»«སྣར་པས། «པེ་»«སྣར་ཐོག་སུ«པེ་《སྣར་»བསྐྱེད། «པེ་»«སྣར་པ། «པེ་»«སྣར་»སྟེ། «པེ་»«སྣར་ཅིག་པ་ན། «པེ་《སྣར་སོགས། ::ཀྱིངང་ཚོ་ཁྱུའི་འ༼འོ། །@«པེ་»«སྣར་—པ། «པེ་»«སྣར་པའི། །ན«པེ་»«སྣར་གོང་། «པེ་»«སྣར་»+པར༑«པེ་»«སྣར་»+རྒྱུའི་རྣམ་འབྱོར་ནི་མ་ཡིན་ནོ། །«པེ་»«སྣར་»བསྒོམ། །«པེ་»བགེགས། 《སྣར་»ཏེ། «པེ་»ནི། «པེ་»«སྣར་»སུ། །«སྣར་ནོ། །།
