།ལ་རང་བཞིན་མེད་པ་ཐོབ། །འབྲས་བུ་ཆོས་སུ་ཐོབ་པའོ། །དེ་ནས་སྐྱོ་བ་བྱུང་བ་ནི། །ཨོཾ་ཨཱཿཧཱུྃ་མིུགཤེགས་སུ་གསོལ། །ཐམས་ཅད་བསྡུས་ནས་ལྷ་ཉིད་གནས༑། །ངག་ཏུ་བཟླས་པ་ནུས་ཚད་བཟླས། །ཞི་ལ་གནས་ཤིང་སྙིང་རྗེ་་་་་བསྐྱེད། །སེམས་ཅན་དོན་ལ་དགའ་སེམས་བསྐྱེད། །གཏོར་མ་རྟག་ཏུ་སྦྱིན་་པར་བྱ། །སྟན་ལ་གཅིག་པུ་འདུག་ནས་སུ། །ལམ་ལྔ་ཉམས་སུ་མྱང་བར་བྱ། །ཚོགས་ལམ་བསྐྱེད་པའི་རིམ་པའོ། །སྦྱོར་ལམ་བཟླས་:པར་བྱེད་པ་ནི། །མཐོང་ལམ་མཚན་ཉིད་བརྗོད་པ་སྟེ། །བསྒོམ་ལམ་འཕྲོ་འདུ་དག་ལ་གནས། །མཐར་ཕྱིན་སྟོང་ཉིད་སེམས་དཔའོ། །དགེ་མས་བརྩམས་པའི་སྒྲུབ་ཐབས་ལ། །འགལ་འཁྲུལ་མཆིས་ན་མཁས་བཟོད་མཛོད། །བརྩམས་པའི་དགེ་རྩ་ཡོད་སྲིད་ན། །སེམས་ཅན་མོས་དང་ཕྲད་པར་ཤོག། དགེ་བསྙེན་མ་སྙན་གྲགས་དགེ་མས་མཛད་པ༑འཕགས་པ་འཇམ་དཔལ་གྱི་སྒྲུབ་པའི་ཐབས་རྫོགས་སོ། །།།།བར་ས་ལ་འངན་མེདཔོ་ལ་མགོ་བར་འབད་བདེན་པ་དང་འདྲ་བལས། ད་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»འཕགས་པ་་་་་འཇམ་དཔལ་གྱི་སྒྲུབ་ཐབས་བཞུགས་སོ། །«པེ་»«སྣར་སྒྲུབ། «པེ་《སྣར་»བླའོ། ་«པེ་»《སྣར་»གཞིར། །«པེ་»«སྣར་དྲ། །«པེ་»«སྣར་»+གྱི། «པེ་»«སྣར་བསྔོ། ནུ«པེ་»«སྣར་»སྒོམ། །«པེ་«སྣར་»+མ་ལུས། «པེ་»«སྣར་ཡང་དག«པེ་»«སྣར་བྱའོ། «པེ་»«སྣར་»བསྒྲལ། །«པེ་»«སྣར་»མཉམ་པ། ཕྱི«པེ་»«སྣར་»རབ་ཏུ་སྦྱོར། @«པེ་»«སྣར་རིག«པེ་»《སྣར་དྷརྨ། @«པེ་»«སྣར་བཟུང་། ཚེ«པེ་»«སྣར་བྱེད། འབྱེ«པེ་»«སྣར་གྱི་དོན་དབང་སྒྱུར། །ལེ«པེ་»«སྣར་ཁྱད། «པེ་»«སྣར་»+ས། སྐྱེ«པེ་»«སྣར་»འདུས། །«པེ་»«སྣར་»«ཅོ་རྒྱན། «པེ་»«སྣར་»སྟེངས། །«པེ་»«སྣར་རྒྱ་གཡས་བཞི། «པེ་»—།
