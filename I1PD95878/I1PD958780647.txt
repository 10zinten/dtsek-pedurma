།རྒྱུད། །ཌ༼དུ། ཀ་ཀ་དཆུའི་བpps:ཆ་ཁཧྲི་དེ་ནས་བདག་པོ་ཉིད་དུ་དབང་བསྐུར་ཏེ། དེ་བཞིན་དུ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་་་དྷརྨཱ་ཎཾ་ཞེས་བྱ་བས་དྲིལ་བུ་བྱིན་ལ་དེ་ཁོ་ན་ཉིད་བསྟན་པར་བྱ་སྟེ། བསམ་པ་ཐམས་ཅད་གྲུབ་པས་ན། །དོན་གྱི་བསམ་གྲུབ་བླ་ན་མེད། །དེ་ཡང་འདོད་པ་རབ་གྲུབ་ཕྱིར། །འདོད་དང་ཐར་བའི་བདེ་ཆེན་པོ། །།དྷི། དེ་ནས་མིང་གི་དབང་བསྐུར་བ་བྱིན་ལ། །དེ་བཞིན་དུ་བཞག་ནས་དམ་ཚིག་བསྟན་པར་བྱ་སྟེ། འདོད་དང་བདེ་བ་ཐམས་ཅད་དང་། །མི་མཐུན་པ་ཡང་སྦྱིན་བྱ་ཞིང་། ད། བདག་ཉིད་དང་ནི་ཕ་རོལ་གྱི། །བསམ་པ་ཐམས་ཅད་རྫོགས་པར་གྱིས། །།བཱིསི། དེ་ནས་དེ་བཞིན་དུ། ཨོ་ཛཧཱུྃ་བཾ་ཧོམ་ཧཱ་སུ་ཁ་བཛྲ་རཏྣ་ཨ་བྷི་ཥེ་ཀ་ཧཱུཾ། ཨ་་བྷི་ཥིཉྩ་མི། །སརྦ་རཏྣ་ཨ་དྷི་པ་ཏི་ཙཱ་དྲི་དྷོ་མེ་བྷ་བ། །སིདྡྷ་ན་མ་ཏེ། བྷ་ག་ཝཱན༑མ་ཧཱ་རཏྣ་ས་མ་ཡ་ཧཱུྃ། །ཨ་ཧཾ་བཛྲ་རཏྟཱ། ཨོཾ་བྷུར་གྷོ་བ་སྭཞེས་བྱ་བ་འདིས་ལག་པ་གཉིས་ཀྱིས་རིན་པོ་ཆེ་བྱིན་ལ། དེ་བཞིན་དུ་ཡང་དག་པར་མཆོད་དེ༑རང་གི་མཚན་བརྒྱ་རྩ་བརྒྱད་པས་བསྟོད་ལ། དམ་ཚིག་གི་ཕྱག་རྒྱ་ཡང་ཉེ་བར་བསྟན་ནས། །སེམས་ཅན་ཀུན་ལ་ཕན་ཕྱིར་དང་།  ེ་། སངས་རྒྱས་དོན་ལ་སྦྱོར་བ་ཡིས། །ཕ་རོལ་ནོར་རྣམས་ཕྲོགས་ཀྱང་ནི། །སྡིག་གིས་གོས་པར་མི་་་་འགྱུར་ཞེས། །གསང་བ་བརྗོད་ནས་དེ་བཞིན་དུ། །རིན་ཆེན་ཁྱོད་ལ་གསོལ་བ་ནི༑། ནོར་གྱི་ཚོགས་ནི་བཤད་དུ་གསོལ། ལ། སངས་རྒྱས་ཀུན་གྱི་དབང་གི་་་མཆོག་། མདོར་ན་འཇིག་རྟེན་ཐམས་ཅད་དུ། །སེམས་ཅན་ཀུན་ལ་ཕན་པའི་ཕྱིར། །ཇི་ལྟར་འདུལ་བ་སྣ་ཚོགས་པའི། །ནོར་བུའི་འཁོར་ལོ་རབ་ཏུ་བསྐོར། །ཞེས་བརྗོད་ལ། རྒྱལ་བར་གྱུར་ཅིག་རྒྱལ་བར་གྱུར་ཅིག། རིན་པོ་ཆེའི་རིགས་ཀྱི་བདག་པོ་རིན་པོ་ཆེའི་འཁོར་ལོ་བསྐོར་ཅིག་ཅེས་བརྗོད་དོ། །།དེ་ནས་རྡོ་རྗེ་་་ཆོས་སུ་རྣམ་པར་བསམས་ལ། དེའི་ཕྱག་རྒྱས་བྱིན་གྱིས་བརླབས་ནས། །དེ་་་་།
