རྗེ་པདྨ་སོགས་ཀྱིས་བརྒྱན། །བུམ་པ་དང་ནི་ལྡན་པ་རུ། །བླུགས་པས་དངོས་་གྲུབ་སྟེར་བར་བཤད། །འདི་ནི་དྲིལ་བུའི་མཚན་ཉིད་དོ། །ནམ་མཁའ་དྲི་མེད་དཀྱིལ་འཁོར་ལེགས་བྲིས་པའི། །རྣམ་པར་དཀར་བའི་དགེ་བ་གང་ཡིན་དེས། །འགྲོ་ཀུན་ལྟ་ངན་སྤྲིན་ཚོགས་རྣམས་བསལ་ཏེ། །ཕྱོགས་རྣམས་ཀུན་དུ་ཉི་ཟླ་ལྟར་གྱུར་ཅིག། །།།་ཁ་་་་མ་་་་་་། ཎ་ཊཀཱ་ཁྲོ་བོ་དག་ཀྱང་ཞི་བ་བྱེད་པ། ཚོགས་པ་::དདདདརྗེ་བ། ཁའ། དར་བ་དང་། ཤེས་པ་ནས། །དུག། ཁ་བའི་ལངཔ་ལ་བརྒྱུད། དེས་རྒྱལ་བ་བསྡུར་མཆན། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་《སྣར་འཇམ་དཔལ་ཆོས་དབྱིངས་གསུང་དབང་གི་དཀྱིལ་ཆོག་བཞུགས་སོ། །«པེ་»«སྣར་»༄༅། །རྒྱ་གར་སྐད་དུ། ག་ག་ནཱ་མ་ལ་སུ་པ་རི་ཤུདྡྷེ་དྷརྨ་དྷཱ་ཏུ་ཛྙཱ་ན་གབྷ་ནཱ་མ། བོད་སྐད་དུ། ནམ་མཁའ་དྲི་མ་མེད་པ་ཤིན་ཏུ་ཡོངས་སུ་་་་དག་པ་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་ཀྱི་སྙིང་པོ་ཞེས་བྱ་བ། ཉི«པེ་»«སྣར་བཞི། «སྣར་»ཡིས། «པེ་»«སྣར་»ཞུགས། «པེ་»«སྣར་པ། «པེ་»«སྣར་པོ་བྲི། ནང་། «པེ་»«སྣར་བྲི། «པེ་»«སྣར་པོས། ་«པེ་»«སྣར་»སྲུངས། «སྣར་རྒྱན། «པེ་»«སྣར་པ། «པེ་»«སྣར་སྲུངས། ཚེ«སྣར་གཟུངས། «སྣར་དུ་བྲི་བར། «པེ་»«སྣར་»ཀྱི་བྱུང་གནས། དགྱེ«པེ་»«སྣར་སྲུངས། ེ«པེ་»«སྣར་ཡང་། ་་་་་་པོ་བདེ་«པེ་»«སྣར་»+བར་བྱ། །«པེ་»«སྣར་བསྲུང་། «པེ་»«སྣར་བཞག་«པེ་»«སྣར་»ཆེའི། »། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་གཉེས་པས་རྟེན། །«པེ་»«སྣར་—པ། «པེ་»«སྣར་ན། །«སྣར་རྗེ། «པེ་»«སྣར་ན། «པེ་»«སྣར་»+རྣམས་«པེ་»«སྣར་»+པར། [P«པེ་»«སྣར་+ན། «ཅོ་»གཞུགས། «པེ་»«སྣར་རིམ། «པེ་»«སྣར་པ། གུ«པེ་»«སྣར་»དང་། 《«པེ་»«སྣར་+དང༌། :ཁ་:དྲ་བ། («པེ་»《སྣར་+བ། «པེ་《སྣར་ན། «པེ་»«སྣར་+རྒྱ། «པེ་»《སྣར་པ། «པེ་»། «སྣར་+པར། དུར་་་༌་ན་་ཁཁོའི«པེ་»«སྣར་བོ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»དང་། «པེ་»«སྣར་སྲུང་། །«པེ་»པ་ལ《སྣར་»ཆ་ནི། @«པེ་»«སྣར་མ། ཊི་ཟླ་པའི་ལ(«པེ་»«སྣར་སྦྱང་། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་པས། །།«པེ་»«སྣར་+རྩེ། །
