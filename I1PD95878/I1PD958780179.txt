མཁའི་ཞེ་བསྟན་འགྱུར། རྒྱུད། དུའི་ཁམ་zpདམ་:པ་བརྒྱུད་པའི་བླ་མའི་བདུད་རྩི་འཐུངས། །འགྲོ་བའི་སྡུག་བསྔལ་ནད་་་ཀྱིས་གཟིར་གྱུར་ལ། །དུས་གསུམ་སངས་རྒྱས་རྣམས་ཀྱི་བསྟོད་ཚིག་འདི། །སྨན་པའི་རྒྱལ་པོ་བཞིན་དུ་བདག་གིས་བཤད། །འཕྲོད་པའི་སྨན་བཞིན་ཀུན་ལ་ཕན་གྱུར་ནས། །མི་རྟོག་ཡེ་ཤེས་འཆི་མེད་ཐོབ་པར་ཤོག། འཕགས་པ་འཇམ་དཔལ་གྱི་མཚན་ཡང་དག་པར་བརྗོད་པའི་མདོ་དོན་མདོར་བསྡུས་པའི་ཆོ་ག་སྒྱུ་་་་འཕྲུལ་དྲ་བའི་རྒྱུད་ལེའུ་ཁྲི་དྲུག་སྟོང་པ་ལས་རྡོ་རྗེ་གླུར་བླངས་པའི་ལེའུའི་སྒྲུབ་ཐབས་མདོ་དང་བསྲེས་པ། སློབ་དཔོན་འཇམ་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པ་:རྫོགས་སོ། །།།དང་ཚེ་སྤོ་ཞེ་ན་རྒྱུད་པའི་བeརྒྱ་གར་གྱི་མཁན་པོ་སུ་བཛྲ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ཆོས་ཀྱི་ཤེས་རབ་་་ཀྱིས་བསྒྱུར་བའོ། །དཀྱིལ་འཁྲོན་ནི་་་་་་་ཚབ་དག་་བོན་ཏ་ར་ཡེ་ཁོ་ར་གངབད་ཁ་ལས་གཞི་དེ་ན་རེ། ཡ་ག་་་་་་དང་། ་དང་། མ་ཕྱི་ཀ་དག་ཁྱབ་ཀར་བསྡུར་མཆན། ད་བདུན་ཁག་དང་། གང་བྱུང་བ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །འཕགས་པ་་་་འཇམ་དཔལ་གྱི་མཚན་ཡང་དག་པར་བརྗོད་པའི་ཆོ་ག་མདོ་དང་བསྲེས་པ་བཞུགས། «པེ་»《སྣར་པས། («པེ་»«སྣར་ཅན། («པེ་»«སྣར་»+ཆུ། «པེ་»«སྣར་པས། «སྣར་སྡིགཆ《སྣར་»«ཅོ་ཧཱུྃ། «སྣར་ནས་མཉེན༑མཁས། །།།འཚོ། «པེ་»«སྣར་+ལ། «པེ་»«སྣར་བྱ་བ་བསྟན་ཏེ། «པེ་»«སྣར་»—མཆོད་པ་བྱེད་པའི། «པེ་»«སྣར་པ་ལ། སྐྱེ«པེ་»«སྣར་པས། «པེ་»+ཏེ། 《སྣར་»+པ་སྟེ། «པེ་»«སྣར་བ། ཀྱི་ཡོད་ཀྱི་«པེ་《སྣར་རླབས། «པེ་《སྣར་དང་། «པེ་《སྣར་ཐམས་ཅད། པེ་»«སྣར་»+ལ། «པེ་»སྣར་»—སྤྱད་པ། ་་«པེ་»«སྣར་»ལྟར། «པེ་»«སྣར་»མཉམ་པར་འདོན་ནམ། འ«པེ་»«སྣར་»སུ། «པེ་»+ལ༑།
