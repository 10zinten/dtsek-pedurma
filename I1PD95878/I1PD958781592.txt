བྱེད་ཅིང་། །གོས་དཀར་བགོས་ཤིང་མ་ཡེངས་པར། །དཀྱིལ་འཁོར་གྱི་ནི་ནང་དུ་འགྲོ། །ལས་ཀུན་བྱེད་པའི་གསང་སྔགས་ཀྱིས། ་་། ཐོག་མ་ཉིད་དུ་བསང་གཏོར་བྱ། །ཚོན་སྣ་ལྔ་ཡི་སྲད་བུ་ལ། །དེ་དག་ཉིད་ཀྱིས་བརྒྱར་བཟླས་ཏེ། །སར་པ་ཤིན་ཏུ་གཙང་བ་དང་། །མཛེས་ཤིང་ཤིན་ཏུ་རན་པ་དང་། །དྲི་ཞིམ་་རེག་ན་བདེ་བ་དང༌། ་། ལེགས་པར་བཀལ་དང་སྲ་བ་དང་། །སྦོམ་ཕྲ་ལ་་་་སོགས་བཀལ་ཉེས་དང་། །འཇུར་མདུད་མེད་པར་གྱུར་པ་ཡི། །ོ། ཐིག་སྐུད་མེ་་ཏོག་སྤོས་མཆོད་པས། ལོ་། ཤེས་རབ་ཅན་གྱིས་ཐིག་གདབ་པོ། །བཟང་པོའི་་་་་རྟགས་་ནི་བརྟགས་ནས་སུ། ཀོ། མཁས་པས་དཀྱིལ་འཁོར་ཐིག་གདབ་བོ། ཀ་། དཀར་པོ་དམར་པོ་དེ་བཞིན་སེར། །སྔོན་པོ་དང་ནི་གནག་དང་ལྔ། །ཚོན་ེསྣ་ལྔ་ཡི་སྲད་བུ་བཞིན། །།།ཕྱེ་མའི་ཚོན་རྩི་དེ་བཞིན་བྱ། །དཀོན་མཆོག་གསུམ་ལ་མཆོད་:བྱས་ཤིང་། །ལྷ་རྣམས་ལ་ཡང་ཕྱག་བྱས་ནས། །མཁས་པས་སྲད་བུ་་བྲེ་བར་བྱ། །ནང་པར་ཉི་མ་ཤར་བའི་ཚེ། །བློ་ཡིས་ཕྱོགས་ཆ་ལེགས་བརྟགས་ལ་་ལྕེ། དང་པོར་ཤར་ནས་:བརྩམས་ཏེ་གཟུང་། །སློབ་དཔོན་བྱང་དུ་ཁ་་་་་བལྟས་ཏེ། །སྲད་བུ་འཛིན་པ་དེ་ལ་བལྟ། །དེ་ནས་ཕྱོགས་བསྐོར་བྱས་ནས་ནི། །བདག་ཉིད་ཀྱིས་ནི་བྲི་བར་བྱ། །།།དེ་ཡི་འོག་ཏུ་བྱང་ཕྱོགས་སུ། །མཁས་པས་་སྲད་བུ་བྲེ་བར་བྱ། །དང་པོ་བདག་ཉིད་མེ་ཕྱོགས་སུ། །འདུག་ནས་ཕྱོགས་་་གཉིས་བརྟག་པར་བྱ། །དེ་ཡི་འོག་ཏུ་རླུང་ཕྱོགས་ནས། །།ཕྱོགས་གཉིས་སུ་ནི་སྲེད་བུ་བྲེ། ཆ། བདེན་བྲལ་དབང་ལྡན་གྲྭ་གཉིས་ནས། །སྲད་བུ་འཛིན་པ་:བལྟ་བར་བྱ། །གྲྭ་བཞི་ལྷམ་པ་བསྙམས་ནས་སུ། །།རྒྱ་གྲམ་དུ་ནི་སྲད་བུ་བྲེ། །རྒྱ་གྲམ་ཐམས་ཅད་བསྙམས་པར་བྱ། །།།དེ་ལྟར་སྲད་བུ་གཏན་ལ་དབབ། །།དེ་ནས་དབུས་ཀྱི་དཀྱིལ་འཁོར་དུ། ཕུར་བུ་གཅིག་ཀྱང་གདབ་པར་བྱ། །གྲྭ་་་་་།
