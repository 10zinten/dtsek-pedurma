བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམ་པ་དོན་བཅུ་གཉིས་བསྟན་པ། གཞན་དོན་སེམས་ཤིང་དོན་ཡོད་མྱུར་དུ་མཐུ་དང་ལྡན་པར་ཤོག། བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམ་པ་དོན་བཅུ་གཉིས་བསྟན་པ་ལུང་གི་རྗེས་སུ་འབྲངས་པའི་སིང་ག་གླིང་གི་སློབ་དཔོན་འཇམ་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པའི་ཚིགས་བཅད་བཞི་བཅུ་པ་རྫོགས་སོ། །།།རང་པར་བྱ་བ་བསྡུར་མཆན། ལོ་ཁ་ནས་་་་་་་་་་་་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«པེ་»«སྣར་བྱང་ཆུབ་ཀྱི་་སེམས་བསྒོམ་པ་དོན་བཅུ་གཉིས་བསྟན་པ་སློབ་དཔོན་འཇམ་དཔལ་བཤེས་གཉེན་གྱིས་མཛད་པ་བཞུགས་སོ། །(«པེ་《སྣར་བྷ། «པེ་»《སྣར་»ཤ་ནི། «སྣར་»རྟགས། ་ཁ༽༽ད་(«སྣར་ཐད། ཀ་ཋ་་ཆ་མ། པ་དང་། །གལ་(«པེ་»—ཕན། «པེ་»«སྣར་»གཞོན། ཆེ་སྐྱེ༌ཀཱ་(《སྣར་གང༌། «སྣར་གཏོགས། «པེ་»«སྣར་ལྡན། །ཡ་ང་(«པེ་»«སྣར་དེ། ་འ«སྣར་འགྲོ། །ནང་་་་«པེ་»«སྣར་ན། (Q«པེ་»«སྣར་»ཀྱིས། «སྣར་རྟགས། «པེ་《སྣར་»མ། །«པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་དཔའ། «པེ་»«སྣར་བསྒོམས་པར་བྱ་བ། སློབ་མ་སྐྱེལ་(«པེ་»«སྣར་»གཞན། («པེ་»«སྣར་རྟོགས། («པེ་»«སྣར་»«ཅོ་ཡུལ། །༄༅«སྣར་རྟག་«པེ་»«སྣར་པར། །ད་བ་ས་ཀྱི། («པེ་»«སྣར་»དཔྱོད། ®«པེ་»«སྣར་པའི། («སྣར་»རྟགས། «པེ་»«སྣར་+ལ། །«པེ་《སྣར་བརྟགས། «པེ་»«སྣར་པ་དང་། «པེ་»«སྣར་»ཀྱིས། ︽«པེ་»«སྣར་»+པའི། ::ཀ་རོ«པེ་»«སྣར་རྟོགས། @«པེ་»«སྣར་»+བྱ། @«པེ་»«སྣར་+ལ། («པེ་»«སྣར་ཀྱི། «པེ་»བརྟགས། «སྣར་»རྟགས། (《སྣར་»རོ། «པེ་»«སྣར་རྟོགས། ཁ་ར་ཁོ་«པེ་»«སྣར་པའི། («པེ་»«སྣར་»འགའ། བོད་ཀྱི་མ། སྐར་མ་«ཅོ་»+བ། «པེ་»«སྣར་ཅེས། «པེ་»སྣར་»གྱིས། ®«པེ་»«སྣར་»གིས། །(«པེ་»«སྣར་»ལྟས། «པེ་»«སྣར་»རྟས། «པེ་»«སྣར་པར། ®«པེ་»«སྣར་བཏགས། «པེ་»«སྣར་དེ། «པེ་»«སྣར་སྒྲིབས། «པེ་»«སྣར་»འདུས་བྱས། ཉི«པེ་»«སྣར་རྒྱུན། (ི«པེ་»«སྣར་གཟུང་། (Q«པེ་»«སྣར་རྟས། བྱ།  ། ཡང་རྒྱལ་ཁབ་།
