རྒྱུད། ་༼དུའིསྤྱོད་པ་དགེ་བ་གང་མོས་བྱ། །དད་ཡོད་ཁྲོ་མེད་སྦྱིན་པར་བྱ། །ལུས་ངག་་་་ཡིད་ཀྱི་བྱ་བ་ཀུན། །བདེ་བ་ཆེ་ལ་ངང་གནས་བྱ། །མ་བརྗེད་པ་ཡི་བསམ་་་གཏན་བསྟེན། །དེ་ལྟར་བྱས་ན་བླ་མེད་འགྱུར། ་། །གཞོན་ནུའི་སྐུ་འཆང་ཐོབ་་་པར་འགྱུར། །ཚེ་འདི་དག་ལ་བདེ་བར་འགྱུར། །།བདག་གིས་བསགས་པའི་ཆོ་ག་ཡིས། །འགལ་བ་མ་མཆིས་ལྷ་ལ་བཤགས། །དགེ་བའི་རྩ་བ་ཡིད་འོང་ཀུན། །ཐམས་ཅད་འཇམ་དཔལ་ཐོབ་པར་ཤོག་་། འཕགས་པ་འཇམ་དཔལ་རིན་པོ་་་ཆེའི་སྒྲུབ་ཐབས། དགེ་བསྙེན་མ་གྲགས་སྦྱིན་མས་མཛད་པ། པ་སྨྲ་ཏི་ཛྙཱ་ན་་་་ཀཱིརྟིས་བསྒྱུར་བ་རྫོགས་སོ། །།།།་་་་ཡ་བ། ནོར་བ་ཤེས་བ་མཁཀོ་ཀོ་དང་ལས་ਮਝੈਲ'ਸੇਧਨੋਸ਼ਸੋਸਪਦਉਨ੍ਹਾਂਕਿਹਾਇਧਮਾਚਕ,ਨੂੰལ་ས་། ཀ་ཀ་ཁབྱེས་འཚོ་བ་བཞི་པར་ཞུ་ཞེབ་བྱས་བསྡུར་མཆན། :གནས་བབ། འབྱུང་ཆེ། :དང་ག:བཞེས་པ་ས་ཁག་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»«སྣར་འཕགས་པ་་་་འཇམ་དཔལ་རིན་པོ་ཆེའི་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་པ། «པེ་»«སྣར་»སྟོང་། །(«པེ་»«སྣར་ཨཱ། Q«པེ་»«སྣར་»དྷརྨ། «པེ་»«སྣར་ཏཱཏ། «པེ་»«སྣར་གེས། །«སྣར་»གཙོ༑«ཅོ་»བརྩོ། ཚེ«པེ་»«སྣར་སྟོང་། ཚེ«པེ་»«སྣར་»སྦྱངས། «པེ་»«སྣར་»དྷུ། «པེ་«སྣར་དུ། ེ«པེ་»«སྣར་»རྗེ། ་་་་བ་འག་པ་::ཀ་སྐུལ་ཁཡག་པ་(«པེ་»«སྣར་ཅེས། «པེ་»«སྣར་»ལས། «པེ་»«སྣར་ནི། «པེ་»«སྣར་མཚོན། །«པེ་»«སྣར་བའོ། @«པེ་»«སྣར་»དྷརྨ་ཨ་བྷ། «པེ་»«སྣར་པ། ་ཉི«པེ་»«སྣར་དྷརྨ། ཨ་«པེ་»«སྣར་»ཏ།  ིཚེ«པེ་»ད་ཡེ་ཏི། «སྣར་»བཱ་ཡེ་ཏི། '(«པེ་»«སྣར་»ཏ། (འ«པེ་»«སྣར་པ། སྟེ«པེ་»«སྣར་དྷརྨ། སྐྱེ«པེ་»«སྣར་གསལ། སྤྱི«པེ་»«སྣར་པ། «པེ་»《སྣར་གདོན། «པེ་»«སྣར་པ་ལས། «པེ་»«སྣར་»བཅུ་དྲུག་ཕྱེད་བརྒྱུད་དེ་ཕྱེད་དེ། «པེ་»«སྣར་»ཡེ་ཤེས། ་༼ག«པེ་»«སྣར་»ཀྱིས། ་པོ«པེ་»«སྣར་བཀོད། ་ ོཚེ«པེ་»«སྣར་ཨ། («པེ་»«སྣར་—ཨ་ཨཱ། ཨི་ཨཱི། ཨུ་ཨཱུ། ཨེ་ཨཻ། ཨོ་ཨཽ། ཨཾ་ཨ། «པེ་»«སྣར་ནས། ས་བདག་ཀྱི«པེ་»«སྣར་»—ཨ་ཨཱ། སརྦ་ཏ་ཐཱ་ག་ཏ་ཧྲི་ད་ཡ་ཧ་ར་ཧ་ར་ཨོཾ་ཧཱུྃ་ཧཱི། བྷ་ག་ཝཱན་ཛྷཱ་ན་མཱནྟི། ཝཱ་གཱི་ཤྭ་ར་མ་ཧཱ་ཝཱ་ཙ། སརྦ་དྷརྨ་ག་ག་ན། ཨ་མ་ལ་སུ་པ་རི་ཤུདྡྷ། དྷརྨ་དྷཱ་ཏུ་ཛྙཱ་ན་གབྷ་ཨཱ། «པེ་»«སྣར་»+སྟོང་་་—།
