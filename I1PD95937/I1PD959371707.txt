ཎགཎྜི། 《སྣར་གཎྜི། །།།་«པེ་»གཎྜི་བརྡུང་། «སྣར་»གཎྜི་བདུང་། «པེ་འཇུག་ཙེ་ལུ། «སྣར་མཇུག་ཙེ་ལུ། «པེ་»པ་འ་འཇུག«པེ་»«སྣར་»གཡོག་(«པེ་»«སྣར་རྩ། པོ«པེ་»《སྣར་ཕྱག@«པེ་ཕུག་ཅིང་ཕྱི་བོད་ཀྱི་གདར་བྱ། 《སྣར་ཕྱག་ཅིང་ཕྱི་བདར་ཕྱ། །ཉི«པེ་»«སྣར་བས། ཚེ«པེ་»«སྣར་»རྟགས། ཀྱེ«པེ་»«སྣར་བསྒོ། («པེ་»«སྣར་རྟགས། «པེ་»《སྣར་»—ཆད་པ་པའི།  ེ«པེ་»«སྣར་»ལོ་དག་གིས་དགེ་སློང་གི། ཨུ«པེ་»«སྣར་»དྲོ་ལ། དུ་བ་«པེ་»«སྣར་»དགེ་སློང་གིས། «པེ་»«སྣར་»«ཅོ་»གིས། «པེ་»སྣར་བསྐུས་ནས་སྣུམ། «པེ་»«སྣར་»གིས། «པེ་»«སྣར་+མ། «པེ་»《སྣར་ཉིད། «པེ་»«སྣར་»ན། མ«པེ་»«སྣར་པ། «པེ་»«སྣར་»གིས། (གྱེ《སྣར་+པ། ་ ེ«པེ་+པ། འགེ«པེ་《སྣར་»། ངནཀྱིས༑ཉེ«པེ་»སྣར་དེ། རྩེ«པེ་»«སྣར་»གིས། ་་་་་་་་་་་་་«པེ་»«སྣར་བར། «པེ་»«སྣར་འཆད། «པེ་»«སྣར་»ཚང་། «པེ་»གིས། །«པེ་»«སྣར་ཡན། @«པེ་»«སྣར་»གྱིས་སྤངས་པ། ་ཚེ«པེ་སངས། «སྣར་སངས་རྒྱས། «པེ་»《སྣར་ན། «པེ་སྤངས་པའི། (ཚེ«པེ་»«སྣར་བགོས། ་(«པེ་»«སྣར་གྱི། ེ«པེ་»:《《སྣར་»སྤངས་པ། («པེ་»«སྣར་གྱི། ཨེ«པེ་《སྣར་བཀླག། སྤང་བ། ལོ«པེ་»«སྣར་སྤང་། །«པེ་»+དེ་། «པེ་»རྣམ། ་«ཅོ་»གྱིས། ®«པེ་»«སྣར་+མ། «པེ་ཁས། «པེ་»:《སྣར་»སྤང་། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་+པ། «པེ་《སྣར་»—པ། «པེ་»«སྣར་»གཅིག་(«པེ་»«སྣར་»དགེ་འདུན་གྱིས་ཤེས་ན། («པེ་»«སྣར་»གྱི། ལེ«པེ་»«སྣར་»གྱི། «པེ་»«སྣར་པ། ་ ེ«པེ་»«སྣར་ན། — ི«པེ་_བ། «པེ་»«སྣར་»—འདི་ལ་བྱ། །«པེ་»«སྣར་»ན་ཆད་པ་བྱས་པར་...སོ་སོར་ཐར་པ་དང་བཅས། «པེ་ཐོགས། «པེ་»ལ་«སྣར་»བྱས། «པེ་»«སྣར་»«ཅོ་དེར། ་མ«པེ་»+དེར། @«པེ་»«སྣར་ལོངས། ༄༅«པེ་»ལ《སྣར་»—ཚེ། ཉི«པེ་»«སྣར་»ཅིག«པེ་»གཅད། (ཁྱེ«པེ་»«སྣར་སྙེད། («པེ་»«སྣར་»«པེ་ཟེར། «པེ་»«སྣར་མ། ::@«པེ་»གིས། «པེ་»གཏང་། «པེ་《སྣར་»། ཀ་པ། ་«སྣར་»«ཅོ་གཏང་། ::@«སྣར་»གང་། «པེ་»ཕྱི། «པེ་»«སྣར་»འདས། «སྣར»གཟུང་། ་དགེ«པེ་ལམ། «པེ་»«སྣར་»། ་ཉེ«པེ་»འཇུགལོ《སྣར་ཞེ། བོ། ངེ«པེ་»«སྣར་»—དེ། ཨེ«པེ་»གཡོག་ཏུ་གཅིག་པ། 《སྣར་»ཏུ་གཡོག་ཏུ་གཅིག་པ། །དེ་ལ་དོ་«པེ་»བཟླ། «པེ་གཅིག་《སྣར་གཅིག་འདས་སོ་ཞེས་མ། «པེ་»+མ་བརྗོད་པ་ལས། ::ཀྱིང་®«སྣར་»བཞིའི། ཉི《སྣར་འབྲིའོ། @«སྣར་པ། @«པེ་»«སྣར་དེ། «པེ་»«སྣར་»::ཀྱིབརྟག@«པེ་»«སྣར་དེ། །ཁྱེ«པེ་»«སྣར་ཤུད། །ཀྱེ«པེ་《སྣར་བརྟགས། («པེ་»《སྣར་»«ཅོ་ཟླ། སྟེ«པེ་»«སྣར་»གི། ་འ::འ་བརྒྱུད་།
