སངས་རྒྱས་ཀྱིས་གསུངས་པ་ཉེས་བྱས་ཙམ་གྱི་ཚོད་ཀྱང་མི་གནང་ཞེས་འབྱུང་། ཡུལ་གྱི་:རྣམ་པ་སྦྱར་ཏེ། སེམས་ཅན་གྱི་མིག་འགག་པར་འགྱུར་ན་ཉེས་བྱས་ཅི་ཙམ་བཏང་ལ་སེམས་ཅན་གྱི་ཡིད་བཞིན་དུ་བདེ་བར་བྱས་ཀྱང་ཉེས་པ་མེད་དོ། །མལ་སྟན་གྱི་དངོས་པོ་སྟེ། ཕྲན་ཚེགས་སུ་གཏོགས་པའོ། །རྫོགས་སོ། །།།འདུལ་བའི་མདོའི་རྩ་བའི་འགྲེལ་པ་ཆུང་བ་སློབ་དཔོན་མང་དུ་ཐོས་་ཤིང་ཡོན་ཏན་དང་ལྡན་པ་ཡོན་ཏན་གྱི་འོད་ཐམས་ཅད་ཡོད་པར་སྨྲ་བ་པས་་བྱས་པའོ། །།།།བ་ཡོདའདོན་བྱེད་ན་ང་ཚོ་ལ་འབད་པའི་བབསྡར་མཆན། ::་་་་་་་ནས:མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་《སྣར་»ན། །«པེ་»«སྣར་»གཅིག«པེ་»«སྣར་»ཁོངས། «པེ་»«སྣར་»ཅེས། «པེ་»«སྣར་ཐོབ། བ«པེ་»«སྣར་»གའི། ༄༅«པེ་»བ་ར་ཎཱ་སྲེར། «སྣར་»བ་ར་ཎཱ་སེར། །«པེ་»«སྣར་»འབོག@«པེ་»«སྣར་»བླངས། །«པེ་»«སྣར་»བཟླ། «པེ་»«སྣར་གཅིག«པེ་»«སྣར་དྲིས། «པེ་»«སྣར་»བཏང་། «པེ་»«སྣར་དྲི། འདྲིར་བཏང་། «སྣར་»ན། ཚ«པེ་»«སྣར་»བ། ཉི«པེ་»«སྣར་»འདུན། «པེ་»«སྣར་»+པར་གནས། ཁ«པེ་»«སྣར་»འདྲི། «པེ་»«སྣར་»ལྟག། «པེ་»«སྣར་བྲེག«པེ་»«སྣར་»བཞགདེས་ནི། མི«པེ་»«སྣར་ཐེག«པེ་»«སྣར་»ན་བཞག(ཆོས་ཚན་འདིའི་ནང་དུ་གཞག་ཅེས་པ་རྣམས་«པེ་»«སྣར་»ལ་བཞག་ཅེས་འཁོད་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»«སྣར་»+ནི། @«པེ་»«སྣར་»བླང་། «ཅོ་»རྟགས། ཉི«པེ་»«སྣར་»འབྱུང༌། ཚེ«པེ་»«སྣར་»རིམས་སུའོ། ་ཚེ«ཅོ་»ཆགས། གྱེ«པེ་»«སྣར་གཉིས། ་་་་་་འདོད་@«པེ་»«སྣར་»སྦྱོང་། «པེ་»«སྣར་»སློབ། «པེ་»«སྣར་»འབོག་ཆ«པེ་»«སྣར་»གྲུབ། པོ«པེ་»སྣར་+ཀྱང་། བདག་«པེ་»«སྣར་»ཚིགས། «པེ་»«སྣར་»བཅས། «པེ་»«སྣར་རེང་། «པེ་»«སྣར་»ད་་་་་་འགྱུར། །དེ་ཡང་བ་དང་། «པེ་»«སྣར་པོ། @«པེ་»«སྣར་»+བྱ། «པེ་»«སྣར་»ནས། ད་།
