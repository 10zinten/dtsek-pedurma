སོགས་པ་བརྗོད་པ་ཡིན་ནོ། །དེ་ལྟར་བསྟན་པ་མཛད་ནས་གསུང་གི་རྡོ་རྗེ་ལ་་་་བསྟོད་པར་མཛད་པ་རྡོ་རྗེ་ཆོས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྟོན་ཏོ། །རྡོ་རྗེ་ཆོས་ཞེས་བྱ་བའི་ཚིག་ནི་ཕྱག་ན་རྡོ་རྗེ་ལ་བྱ་བ་ཡིན་ནོ། །སྒྲོན་མ་གསལ་བར་བྱེད་པའི་ལེའུ་བཅུ་བདུན་པའི་དཀའ་བ་བཏུས་པའི་བཤད་པའོ། །།།:pལུ་བྱ་བའི་ཚོསློབ་དཔོན་ལེགས་ལྡན་བྱེད་ཀྱིས་མཛད་པ་སྒྲོན་མ་གསལ་བར་བྱེད་པའི་:རྡོ་རྗེའི་ཚིག་གི་དཀའ་བ་བཏུས་ཏེ་བཤད་པ་ཞེས་བྱ་བ་རྫོགས་སོ། །།།རྒྱ་གར་གྱི་མཁན་པོ་རྒྱལ་བ་མཆོག་གི་ཞལ་སྔ་ནས་དང༌། བོད་ཀྱི་ལོ་་ཙཱ་བ་དགེ་སློང་དཔལ་ཤཱཀྱ་བརྩོན་འགྲུས་ཀྱིས་མཉན་ནས་བསྒྱུར་ཞིང་:ཞུས་ཏེ་བ་པ། །ཀྱང་། ་»(དྲ་བ་་ཡས་ཀྱི་། །།བ་སྤུ་འདུ་བསྡུར་མཆན། གཏེ། །དབང་། ཀུན་འཚེ་་ཚུལ་དང་བར་ད་ལྟ་འོཀ་ཀྱg/?གྱི། །བདག་དང་། །་ཚེས་ཚེ་དབང་་་་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»སྒྲོན་མ་གསལ་་:བར་བྱེད་པའི་འགྲེལ་པ་བཞུགས། «པེ་»《སྣར་དཱི། «པེ་»ཀླགས། «པེ་»གྱིས། ཀྱི་«པེ་»«སྣར་+ཤེས་རབ་དང༌། «པེ་»བ། «པེ་»སྣར་»ནི་ཚིགཅེ«སྣར་»བསྒྲུབ། འ«པེ་»སྣར་གིས། ས་གནས་y«པེ་»«སྣར་»ཡང་། ་@«པེ་»«སྣར་ནམ། «པེ་»«སྣར་+པས། པན༽«པེ་»«སྣར་འཛིན་གྱིས་བཞུགས། ཉི«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»སྐྱེས། ན་«པེ་»«སྣར་»སྟོང་། ི«པེ་»«སྣར་བསྲུང༌། །«པེ་»«སྣར་བདུན། «པེ་»《སྣར་»ལ། ：Aཚུ«པེ་»«སྣར་»ཉིད། @«པེ་»«སྣར་བཞུགས། །«པེ་»«སྣར་»ཝ། »ཚེ«པེ་»གྱིས། །«པེ་»སྣར་བདེ། 《་ཀ་ན། མ་ཏི་། །དབང་་ར《བནི(DPPདང་་་ཚེས་«པེ་»«སྣར་»—པ། (Q«པེ་»དེའི། @«པེ་«སྣར་»«ཅོ་སྲོགཚེ«པེ་»«སྣར་»པ། གྱི«པེ་»«སྣར་»—ནི། ཚེ«པེ་»རླབ། ༄༅«པེ་»«སྣར་པར། ཌ«པེ་»«སྣར་»པ། _@«པེ་»—།
