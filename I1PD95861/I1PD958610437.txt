།བ་། བསྟན་འགྱུར། རྒྱུད། ༼གཁ་དཆ་མཎོམ་:འགྱུར་བའི། །ཉེས་དེ་མཁས་པ་རྣམས་ཀྱིས་བཟོད་མཛད་ཅིང་། །ཕྱིས་འབྱུང་སྐད་གཉིས་སྨྲ་མཁས་བྱུང་གྱུར་པའི། །ཡ། །སྒྲ་དོན་བརྟགས་ནས་ལེགས་པར་འཆོས་པ་དང་། ཊི་ལོ། གཟུར་གནས་དོན་གཉེར་མཁས་པའི་སྐྱེས་བུ་རྣམས། ཚོ། མཉེས་བྱས་གཞུང་འདི་ཕྱོགས་བཅུར་རྒྱས་གྱུར་ཅིག། ལི་། ཅེས་པ་འདིའི་ཡི་གེ་པ་ནི་དགེ་སློང་རིན་ཆེན་རྒྱལ་མཚན་ནོ། །འདིས་འཁོར་བ་སྡུག་བསྔལ་གྱི་རྒྱ་མཚོ་་་ཆེན་པོ་མྱུར་དུ་སྐེམས་པར་གྱུར་ཅིག། ་གཏུ། ཁོང་ལ། ཁ་ཌོ་ཀ་བོའིལ་གླ་བ་ལ་ལན་དེ་འདྲ་ལ་དེ་ལ་རྫོགས། དེ་བ། །ཟླ་ཏིཥཱ་ཡོ་ཟེ་བ་ཅད་ns་་བགས་སུ་ཀ་བ་ལྔ་བ་དང་། ན་བསྡུར་མཆན། ཉི་འབུམ། ཚུལ་ཁཚོམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། ྋ«པེ་»«སྣར་»སྒྲོན་མ་གསལ་བར་བྱེད་པའི་དཀའ་འགྲེལ་་་པ་སློབ་དཔོན་ཐུགས་རྗེའི་དཔལ་གྱི་ཞབས་ཀྱིས་མཛད་པ་བཞུགས་སོ། །«པེ་»«སྣར་»པཱཉྫི། «པེ་»«སྣར་»ལ་«པེ་»«སྣར་པའོ། «པེ་»«སྣར་བྱ་བ། བ«པེ་»«སྣར་»འདི། ®«པེ་»«སྣར་པར། Q«པེ་»«སྣར་པར། e«པེ་»«སྣར་རྣམ་པ། པར། «པེ་»《སྣར་»འབྲེལ། ཚེ«པེ་»«སྣར་+པ། @«པེ་»«སྣར་»ངང་། །བསོདs@་གཏོ«པེ་《སྣར་—སྒྲའི། ®«པེ་»«སྣར་བར། «པེ་《སྣར་ནི། «པེ་《སྣར་»གིས། །་་་ལ། དེས«པེ་»«སྣར་»ཤེས། «པེ་»«སྣར་རྣམས། །«པེ་»«སྣར་»ཞེས། «པེ་»«སྣར་ན། —«པེ་»«སྣར་»གྱི། °«པེ་»«སྣར་འགྲེལ། ལས་«པེ་»«སྣར་རྒྱུད། «པེ་»«སྣར་མདོ། ཚེ«པེ་»«སྣར་པ་རུ། ་ཉི«པེ་»«སྣར་»+ཀྱི། «པེ་»«སྣར་རྒྱུད། «པེ་»«སྣར་»གྱི། ཀྱེ«པེ་»«སྣར་ཡི། «པེ་»«སྣར་»འབྱམས། ®«པེ་»«སྣར་»དོ། །ཉི«པེ་»«སྣར་»འབྱམས། «པེ་»«སྣར་»རིམ་གྱིས། ::ཀྱིང་། «པེ་»«སྣར་»བསྐྱེད་པའི་རིམ་པ་ཞེས་བྱ་བ་ལ་འཁོད་པར་བྱ་སྟེ། ཚན་པ་རགས་པ་དང་ཚེ་«པེ་»«སྣར་»གཟུང་། «པེ་»«སྣར་»བརླབས། Q«པེ་»«སྣར་ཉིད་ཀྱི། «པེ་པར། «པེ་»ཅན་འདི་ཉིད། མི«པེ་»«སྣར་»སྟོང་། །འོ«པེ་»«སྣར་བརྟགས། །ཉི«པེ་»«སྣར་+བ། མི«པེ་»«སྣར་བརྟགས། ཁྱི«པེ་»«སྣར་བརྟན། དང་མི«པེ་»«སྣར་པ། །།སསུག་པ་་་་་་་་། །ཞེས་པ་ར་བ་བློ་«པེ་»«སྣར་བལྟོས། Q«སྣར་»པ། «པེ་»«སྣར་—ཀྱི། «པེ་»«སྣར་གསུངས། «པེ་གསུང༌། «སྣར་ཞེས། ཚེ«པེ་»«སྣར་སྲེད། ཉི«པེ་»«སྣར་»འགྲུབ། «པེ་»«སྣར་།
