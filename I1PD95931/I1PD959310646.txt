དགེ་སློང་རྣམས་བག་བྱེད་པར་སྟོན་པ་སྟེ། །ལྟུང་བ་སྔ་མ་བཤགས་ཤིང་ཕྱི་མ་མི་་བྱ་བར་དམ་བཅས་པས་ན། དེ་ཚུལ་ཁྲིམས་ལ་གནས་ཤིང་བག་བྱེད་པ་ནི་ཇི་ལྟར་གཡག་གི་རྔ་མ་ཤིང་ལ་ཐོགས་ན་བག་བྱེད་པ་བཞིན་ནོ། ལ་ལ། སོ་སོར་ཐར་པ་་:བཏོན་པ་ཡིས། །བསོད་ནམས་ཅུང་ཟད་གང་ཡོད་པ། །དེས་ནི་འཇིག་རྟེན་མ་ལུས་པ། །ཐུབ་དབང་གོ་འཕང་ཐོབ་པར་ཤོག། ཚིགས་བཅད་:འདིས་བསྡུས་ཅིཔའི་བསྔོ་བའོ། །སོ་སོར་ཐར་པའི་མདོ་རྒྱ་ཆེར་འགྲེལ་པ་རྫོགས་:སོ། །ག་ང་། བབབབ་ག་ད་བ་ས་ཀྱི་གཏུམ་སྤྱི་དང་་དབྱེ། །དེ་ཡིན་དང་། ཁོང་དེ་དག་བསྡུར་མཆན། དེ་དག་བསྐྱོ་ཏོ་བ་བམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། «པེ་»«སྣར་ལ་སོ་སོར་ཐར་པའི་མདོའི་འགྲེལ་པ་བཞུགས་སོ། །ཞེས་འཁོད།  ེ«པེ་པྲ་ཏི་མོ་ཀླུ་སུ་ཏྲ་སྲི། «སྣར་པྲ་ཏི་མོ་ཀླུ་སཱུ་ཏྲ་པྲི། «པེ་»《སྣར་»+རུ། གནའ་བོ། ཟླ་བ་དང་ཟླ་བ་འབར་བ་དང་་། བྱ་བ་pf ེ«པེ་»«སྣར་»+ལུས་དང་། «པེ་»«སྣར་+གཟུགས་མེད་པའི་རིགས་ན། «པེ་»«སྣར་»ཅི། «པེ་»«སྣར་བྱའི། «པེ་»«སྣར་+ཡིད་ལ། ཕྱི«པེ་»«སྣར་+ན། ཚེ«པེ་»«སྣར་»ཅི། «པེ་»བར་«སྣར་»+པ། «པེ་»«སྣར་»ལྡན། ཁྱེ«པེ་»«སྣར་པ་དམ་པའི། «པེ་»«སྣར་»+ནི། «པེ་»《སྣར་གྱིས། «པེ་»བརྙེད། །«པེ་《སྣར་+ལ། «པེ་»«སྣར་»+སྟེ། Q&«པེ་»«སྣར་པར། «པེ་»«སྣར་»ཉིད། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་—དུ། «པེ་»ལོགས། «སྣར་»གྱིས། ཁྱེ«པེ་»«སྣར་»། ཀྱེ«པེ་»«སྣར་»+མེད་ཀྱི། རེ«པེ་»«སྣར་པ། ་་་་«པེ་»«སྣར་»པོ་ཆེའི། «པེ་»བརྙེད། ་«པེ་»«སྣར་»—པ། «པེ་»«སྣར་+པ་ཟད། @«པེ་»«སྣར་»འདིས། «པེ་»«སྣར་»+འདི། །«པེ་»«སྣར་»+ལ། དགྱེ«པེ་»«སྣར་»ན། འབྱེ«པེ་»གཏུགས་སྟེ། «སྣར་»གཇུགས་སྟེ། མུ«པེ་»«སྣར་»+ཀྱི། ཉེ«པེ་»«སྣར་ནམ། «པེ་»«སྣར་རུང་ཞེས། ལེ«པེ་»«སྣར་བྱར། ཀྱི«པེ་བཙོ། «སྣར་གཙོ། «པེ་»«སྣར་སྡེ་ལས། «པེ་»«སྣར་»+བྲལ། ︽«པེ་»«སྣར་»+—།
