དེ་ཅིའི་ཕྱིར་ཞེ་ན། དེའི་རང་བཞིན་ནི་དེ་ཡིན་ནོ། །དངོས་པོ་མེད་པའི་ངོ་བོ་ཉིད་སྟོང་པ་ཉིད་ཀྱི་བར་དུ་ཐེར་ཟུག་ཏུ་གནས་པ་མ་ཡིན་ཏེ། དེ་ཅིའི་ཕྱིར་ཞེ་ན་དེའི་རང་བཞིན་ནི་དེ་ཡིན་ནོ། །ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་སུམ་ཅུ་རྩ་བདུན་ཐེར་ཟུག་ཏུ་གནས་པ་མ་ཡིན་འཇིག་པ་མ་ཡིན་ཏེ། དེ་ཅིའི་ཕྱིར་ཞེ་ན་དེའི་རང་བཞིན་ནི་དེ་ཡིན་ནོ། །སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་བཅོ་བརྒྱད་ཀྱི་བར་དུ་ཐེར་ཟུག་ཏུ་གནས་པ་མ་ཡིན་འཇིག་པ་མ་ཡིན་ཏེ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན་དེའི་རང་བཞིན་ནི་དེ་ཡིན་ནོ། །ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དགེ་་་བའི་ཆོས་དང་མི་དགེ་བའི་ཆོས་དང་། འདུས་བྱས་ཀྱི་ཆོས་དང་འདུས་མ་བྱས་ཀྱི་ཆོས་དང་ཟག་པ་དང་བཅས་་པའི་ཆོས་དང་ཟག་པ་མེད་པའི་ཆོས་ཐེར་ཟུག་ཏུ་གནས་པ་མ་ཡིན་འཇིག་པ་མ་ཡིན་ཏེ་དེ་ཅིའི་ཕྱིར་ཞེ་ན་དེའི་རང་བཞིན་ནི་དེ་ཡིན་ནོ། །«པེ་»«སྣར་དོ། །«སྣར་དེ། །ེQ«སྣར་དག«པེ་»«སྣར་དོ། ::ཀྱིང::་བ་དད«པེ་»«སྣར་»ལ། Q«སྣར་»—པ། «པེ་»«སྣར་སྤེལ། «པེ་»«སྣར་སྨྲས། ®«པེ་»«སྣར་+ན། «པེ་»«སྣར་»+ན། ཀྱི«པེ་»«སྣར་+སུ། «སྣར་དག«པེ་»«སྣར་པའི། །ག་:བྱ་བ་@«པེ་»«སྣར་བ། «སྣར་+མ། «པེ་»«སྣར་གིས། «པེ་»«སྣར་»འདི། ༄༅«པེ་»«སྣར་»—ཡོངས་སུ། «པེ་»«སྣར་བགྱིད་པར། ®«པེ་»«སྣར་»ལ། «པེ་»«སྣར་རྟོགས་པ་ལ། «སྣར་»རྣམས། «པེ་»«སྣར་+དོ། །ཚེ«པེ་»«སྣར་»ཕྱིར། །«པེ་»«སྣར་»—འདི་ལྟར། «སྣར་»གི། །«པེ་»«སྣར་»སོ། «པེ་»«སྣར་»_པ། །«སྣར་མ། «པེ་»«སྣར་སོ། །«པེ་»«སྣར་»ནི། །«པེ་»«སྣར་དེ་ཅིའི། «པེ་»«སྣར་+བར་གྱི། «པེ་»«སྣར་—སྟོང་པ་ཉིད། «ཅོ་»རེའི། ཨེ«ཅོ་དེ་ཅིའི། «པེ་»«སྣར་»སྟོང་པ་ཉིད། །«པེ་»«སྣར་སྐུ། །«པེ་»«སྣར་+བྱ། ཉི«པེ་»«སྣར་»ནི། «པེ་«སྣར་+ཡང་། «སྣར་»དང་། །«པེ་»《སྣར་བ། «སྣར་»མ། «པེ་»«སྣར་»«ཅོ་»མི། «པེ་»«སྣར་+པ་མེད། «པེ་»«སྣར་»+ནི། «པེ་»«སྣར་དེ། «པེ་《སྣར་»+ནི། །
