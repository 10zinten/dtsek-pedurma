སྐྱེ་མཆེད་དང༌། རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བ་དང༌། རྣམ་པ་ཐམས་ཅད་་་མཁྱེན་པ་ཉིད་ཀྱི་བར་གྱི་འཇིག་རྟེན་བསམ་གྱིས་མི་ཁྱབ་ཅེས་བྱ་བར་སྟོན་ཏོ། །དེ་བཞིན་དུ་རྣམ་པར་དབེན་ཞེས་བྱ་བ་དང༌། ཤིན་ཏུ་སྟོང་ཞེས་བྱ་བ་དང་། རང་བཞིན་སྟོང་ཞེས་བྱ་བར་སྟོན་ཏོ། །རབ་འབྱོར་གཞན་ཡང་ཤེས་རབ་ཀྱི་ཕ་་་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་ནི་དེ་བཞིན་གཤེགས་པ་ལ་འཇིག་རྟེན་ཞི་ཞེས་བྱ་བར་སྟོན་ཏོ། །ཕུང་པོ་ལྔའི་འཇིག་རྟེན་ཞི་ཞེས་བྱ་བར་སྟོན་ཏོ། །དེ་བཞིན་དུ་་་ཁམས་དང་། སྐྱེ་མཆེད་དང་། རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བ་དང་། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་བར་གྱི་འཇིག་རྟེན་ཞི་ཞེས་བྱ་བར་སྟོན་ཏོ། འོ་། རབ་འབྱོར་དེ་ལྟར་ན་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་དེ་བཞིན་གཤེགས་པ་སྐྱེད་པར་བྱེད་པའོ། །འཇིག་རྟེན་འདི་སྟོན་པར་བྱེད་པའོ་ཞེས་བྱ་བ་ནི་བསམ་གྱིས་མི་ཁྱབ་པ་དང་ཞི་བ་ཉིད་ཤེས་པ་ཡིན་ནོ། :། །།་༌བསམ་། ལས་རབ་འབྱོར་གཞན་ཡང་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་དེ་བཞིན་གཤེགས་པ་ལ་འཇིག་རྟེན་སྟོང་པ་ཉིད་ཅེས་བྱ་བར་སྟོན་ཏེ། ཕུང་པོ་ལྔའི་འཇིག་རྟེན་སྟོང་པ་ཉིད་ཅེས་བྱ་བར་སྟོན་ཏོ། ་། དེ་བཞིན་དུ་ཁམས་དང༌། སྐྱེ་མཆེད་ཀྱི་འཇིག་རྟེན་སྟོང་པ་ཉིད་ཅེས་བྱ་བར་སྟོན་ཏོ། ཁ། རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བ་དང་། རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བའི་ཡན་ལག་གི་འཇིག་རྟེན་སྟོང་པ་ཉིད་ཅེས་བྱ་བར་སྟོན་ཏོ། །ཁ་། དེ་བཞིན་དུ་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་བར་གྱི་འཇིག་རྟེན་་སྟོང་པ་ཉིད་ཅེས་བྱ་བར་སྟོན་ཏོ། །རབ་འབྱོར་དེ་ལྟར་ན་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དེ་བཞིན་གཤེགས་པ་སྐྱེད་པར་བྱེད་པའོ། །འཇིག་རྟེན་འདི་སྟོན་པར་་་བྱེད་པའོ། ཁྲོ་ལ་། རབ་འབྱོར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་ཡང་དེ་་་་བཞིན་གཤེགས་པ་ལ་འཇིག་རྟེན་འདིའི་འདུ་ཤེས་དང་། འཇིག་རྟེན་ཕ་རོལ་གྱི་།
