བསྟན་འགྱུར། ཤེས་ཕྱིན། ངའི་«པེ་《སྣར་»—ལ། ཨེ«པེ་»«སྣར་»—རྣམས། «ཅོ་»དང་སྨོན་པ་མེད་པ་རྣམས། *#《ལྷ་ལ་«པེ་«སྣར་འདི། «སྣར་»གྱིས། སྟོང་ནག་པ། རང་རྒྱུད་ཊ་པོའི»ས་སྐྱ། །«པེ་སྐ། །།«པེ་»«སྣར་»ཤིང་ངམ། འ@«པེ་»《སྣར་»གྲུ། «པེ་»«སྣར་»རབ་འབྱོར། གོ་གནས། །དེ་ཚེ། །སྒྱུར། @«པེ་»«སྣར་+ཏེ། །ཀ་ར་འོ་ཀྱི། །དམ་པའི་ལ་དོན་གྱི་(ཆུ་འབགས་རིགས་ཀྱི་«པེ་»«སྣར་པའི། «པེ་《སྣར་»དད་པ། ར《ཀ་བ་ཀ་། །པནྟི་ཏིར་གྱུར་«པེ་»«སྣར་»—ཀྱི་ས། Q«པེ་»«སྣར་བྱ་སྟེ། «སྣར་»—པ། ཚེ«པེ་»«སྣར་། ༎@«པེ་»«སྣར་»བྱ་བར། །།《ང་། 《་#»ཀོད«སྣར་»འཚོའམ། ཉི«པེ་»«སྣར་»འཚེའུ་འམ། «པེ་»«སྣར་»ཆུ། ®«པེ་»«སྣར་གྱིས། «པེ་»«སྣར་། །དབྱེ། །།།དང་། »༽ལྟ་«པེ་»«སྣར་ཀྱིས། ཚེ«པེ་《སྣར་བཏོང་། དང་《f/gདེ་། ཏམ་དང་། ཟླ་བ་བདུན། %ཚུལེ་དུག་ཨེ«སྣར་»གྱིས། །:་ཏུཀུ་བ། ་བ་རྒྱུ《d»། དང་འདྲ་ལ»/aད༈། «པེ་»«སྣར་། །«པེ་»«སྣར་བརྟེན། «པེ་»«སྣར་—སུ། «པེ་»«སྣར་»གིས། «ཅོ་འཆི་བས། ཚེ«ཅོ་འཆི། དང་པད་འབྱིས། །་་་་་ས་བར། :ཅད་ཀད་པ་《ི«པེ་»«སྣར་»འཆལ། «པེ་»«སྣར་བསྟེན་ཏེ། །Q«པེ་»གིས་གཟུང་གིས། «སྣར་»གིས་བཟུང་། བ་…གིས། «ཅོ་»གི་བཟུང་གི། །།འབར་བ་ཁོ་ན་དེ་ཁཆཆ་རགས་ཨgp?མེ་ཀོང་ཏཏི«པེ་»«སྣར་བརྡེག«པེ་»«སྣར་»—སྟོང་པ་ཉིད། །མཁྲུད:《ཀ་གསང་ཏེ་ཅn@«པེ་»«སྣར་སྩལྟ། @«པེ་»«སྣར་བསྒོམ། །«པེ་»«སྣར་བརྩོམ། «པེ་»«སྣར་ལྟ། །།«པེ་»«སྣར་»བསྲུང་། ཚེ«ཅོ་»ནོ། «པེ་»«སྣར་»བརྩོམ། ་ང་<འི་བྱང«པེ་»སྣར་ཤེས་རབ་རློམ་སེམས་སུ་བྱེད། @«ཙོ་ཚུལ། Q«པེ་»«སྣར་དཔའ། «པེ་»«སྣར་»བསྒོམ། «པེ་»«སྣར་བརྩོམ། «པེ་»«སྣར་—དེ། ཥཊ་ཌ་ལོ»ཀྱི། །«པེ་བསྒོམ། ཨི«པེ་»«སྣར་»+རློམ་སེམས་སུ་བྱེད་དེ། ་«པེ་»«སྣར་»དེ། «ཅོ་+མི། །ཨེ«ཙོ་»ཚུལ། ཚེ«པེ་»«སྣར་»ཏེ། །«སྣར་»ན། «པེ་»«སྣར་གིས། ངོནཀྱི་ཚོགས་«པེ་《སྣར་ལྷུང་པའི། །«པེ་»སྣར་ལྷུང་། «པེ་»«སྣར་བསྒོམ། «པེ་»«སྣར་ལྷུང་། མེ«པེ་»བརྩོམ། @«པེ་»«སྣར་»ལྷུང་། «པེ་»«སྣར་»དེ། «པེ་»«སྣར་»དག«པེ་»《སྣར་»ནི། གྱེ«པེ་«སྣར་ལས། Q«སྣར་བསྒོམ། བློ། (:་ལོཚེ་«པེ་»«སྣར་བྱ་བར། ཉི«སྣར་ཀྱིས། «པེ་»«སྣར་པར། «པེ་»«སྣར་ཡང་། «ཅོ་»ཚུལ། བ་ར་ཚེ«པེ་《སྣར་»དེ་ལྟར་ན། །ཏིང་བར་བྱ་ཡི། དེ་ལྟpa:::«པེ་»«སྣར་»སྟེ། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱང་། @«པེ་»«སྣར་སྐ། །།།«སྣར་བརྟན། །«པེ་»«སྣར་»+དེ། མི«ཅོ་བསྔོས། ལ«པེ་»«སྣར་»«ཅོ་བསྡོས། །
