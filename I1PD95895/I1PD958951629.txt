བསྟན་འགྱུར། །ཤེས་ཕྱིན། ༼«པེ་»«སྣར་+ཤེས་རབ་ཀྱི། ®《སྣར་»སྙེན། ན«པེ་»«སྣར་»སྙེན། རྒྱ་ནག་གི«པེ་»«སྣར་»—གང་། ཚེ«པེ་ཚེ། ཚེ«པེ་»མོ། ཚེ«པེ་»«སྣར་»གི། ་ཏ། བ༌་«པེ་《སྣར་»སྟོང་བ་ཉིད། @«པེ་»«སྣར་སྤངས། །《འི་ཚེ། །གང་《་དོན༽«པེ་»«སྣར་+དང་། ཉི«སྣར་གཟུང་། «པེ་《སྣར་»བཀླག། །འདྲ་པ《»༈«སྣར་»མི། «པེ་»མི་འབྲེལ། «སྣར་མི་འབྲལ། ་༼ི«པེ་»«སྣར་སླད་དུ། &«པེ་+མ། ???«པེ་»«སྣར་»འཇིག་རྟེན་དུ། «པེ་»«སྣར་བསྒྱུར། །ཀཀྐཎོ་ཊ་《:«པེ་»«སྣར་»—དེ་བཞིན་དུ། བྱ:་། :ལྔཎ་ཀྱི་ཁའང་ཆེ་ལ། ཁཝ་ཎ་་་་་་་དང་། །«པེ་»རྟེན། «པེ་»《སྣར་»འཇིག་རྟེན་དུ། ་༌ཀ་་་་དགེ་བླའི་«པེ་»«སྣར་པ། «པེ་《སྣར་སྐྱབས་པར་བྱ་བ་དང༌། «པེ་»«སྣར་»གང་། «པེ་»སྐྱབས། «སྣར་སྐྱབ། :«པེ་《སྣར་»ཀྱི། ཀློག་པ་པོ་ཀ་ཁ་《ཁོ«པེ་»«སྣར་བྱ་བ། ®«ཅོ་པོར་སྐྱེད། «པེ་»«སྣར་»—དུ། «པེ་»«སྣར་དེ་ནི། «པེ་»«སྣར་། །།དང་། ་་་༌་ཨོ། ངས། །ཆོས་པར་ཁང་«པེ་»«སྣར་»—དེས། །།ས་། ཀར་ཏ་། #pཊ་། ན@«པེ་བཟུང་། «པེ་»«སྣར་དང་། །ལོ་ཁ་ཤིམ་དག་འདུལ་བ། ཎཏོག@། «སྣར་གཟུང་། «པེ་»«སྣར་བྱ་བ། «པེ་»«སྣར་»གྱིས། «པེ་»《སྣར་»ཀྱི། «པེ་»《སྣར་»དབྲལ། ་ཏུ། ལོ་གྱི། དེ་ལས་ye། ་::ཀ་ག་ཆ་ཁཆུ་ཀ་ངོད་ཁཀྱང་ཀ་ཎཀྱིrཔའི་ནང་། «པེ་»«སྣར་བརྩོད། «པེ་»«སྣར་ཕྱིར། ་ཉི«པེ་»«སྣར་»ལོ། ་ཤིང་། ཀར་ཎཉི་འཆི་«པེ་»«སྣར་། །«པེ་《སྣར་ཟས། «པེ་»སྣར་བསྙེགས། «པེ་»«སྣར་»ག། «པེ་»«སྣར་»རྩིའི། «པེ་»མཐུ། ་ཉི«པེ་»«སྣར་ཤ། །།ཡང་ན་དེ་ལ་པེ༽?«པེ་»«སྣར་སྤྱོད། །གང་། གང་བ། །སྒྲུང་《་༽?«པེ་»«སྣར་—དེ། «པེ་»«སྣར་»«ཅོ་+སྟོང་པ་ཉིད། «པེ་»«སྣར་»+པོ། ་ཚེ༽«པེ་»«སྣར་+ཡང་། «པེ་»«སྣར་སྤངས་པ། །།།ན། ད་ར《། »@༈༽»«པེ་»སྣར་»ཡང་། QQ«སྣར་ཡང་། @@@«པེ་»«སྣར་བཀྱལ། @《སྣར་»—ཡང་། @@《སྣར་»ཡང་། ®«པེ་»«སྣར་»ཅིང་། 《«པེ་»+ཡང་། _@«པེ་»«སྣར་+ཡང༌། @«པེ་»«སྣར་»+རབ་ཏུ། ང་། ཕྱིར་བྱེད། །དབྱེ། ༽ ེ«སྣར་»—ཡང་། @«སྣར་བ། ཡོ@«སྣར་»ཡང་། ཚེ«པེ་»«སྣར་—རབ་ཏུ། «པེ་+ཡང་། «པེ་»«སྣར་»—རབ་ཏུ། འགྱེ«པེ་»«སྣར་+ཡང་། ཉེ«པེ་»«སྣར་»ཡང་། །༽«པེ་»《སྣར་»+རོ། @«པེ་»«སྣར་བསྒོམ། ཉི«སྣར་ཡང་། «སྣར་»—ཡང་། «པེ་»«སྣར་—ཡང་། «པེ་»«སྣར་+ཡང་། «པེ་»«སྣར་+ཡང་། ་«པེ་»«སྣར་_ཡང་། 《«པེ་»«སྣར་»།
