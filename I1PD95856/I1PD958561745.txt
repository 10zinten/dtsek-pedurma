རྒྱུད། ཞི་ལ་བསྟན་པ་དང་དབང་བསྐུར་བར་བྱའོ། །དེ་ལྟར་ཡང་སངས་རྒྱས་ཐོད་པའི་རྒྱུད་ལས་ཤེས་རབ་པདྨའི་ནང་དབུས་སུ། །དཔག་ཚད་བྱེ་བའི་དཀྱིལ་འཁོར་ལ། །དཀྱིལ་འཁོར་དེ་ཡི་དབུས་སུ་ནི། །མཁས་པས་སློབ་མ་དབང་བསྐུར་བྱ། །ཞེས་པ་དང་གཞན་དུ་ཡང་གསུངས་པ་སྟེ། ཡིད་ཀྱིས་འགའ་ཞིག་ཉིད་དུ་ནི། །རྒྱལ་་བའི་དཀྱིལ་འཁོར་བྱ་བར་བཞེད། །ཅེས་གསུངས་སོ་ཞེས་པ་ནི་ཡིད་ཀྱི་དཀྱིལ་་་་འཁོར་གྱི་ཆོ་གའོ། །སློབ་དཔོན་ཆེན་པོ་ཨ་བྷ་:ཡཱ་ཀ་་ར་གུཔས་མཛད་པའི་དབང་བསྐུར་གྱི་རབ་ཏུ་བྱེད་པ་རྫོགས་སོ། །།།ཤ་ལ་དགའ་པཎྜི་ཏ་དེ་ཉིད་དང་མཁས་པ་བརྗེ་ཤེས་རབ་དཔལ་གྱིས་བསྒྱུར་བའོ། །།དལ་འབབ་ཁ་བ། ང་ཚོ་ཁ་བའི་དྲན་ਮਹਾਵਾਰਵਾਟਮਾਲਕਾਦੇਵਰਲੋੜਕੋਸਰੇਟཀྱི་བར་བསྡུར་མཆན། ལོ་བབ་པ་ཚོ་དང་གོ་བ། མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»དབང་གི་བྱེད་པ་བཞུགས། «པེ་»《སྣར་»ཆུང་། «པེ་»《སྣར་»པ། «པེ་»«སྣར་»«ཅོ་»ཧཱུྃ། «པེ་»«སྣར་»ཧཱུྃ། «པེ་»པཱ། «པེ་»«སྣར་»ཧཱུྃ། ས«པེ་»ཧོ༑པ«པེ་»ར། «པེ་»«སྣར་»ནཱ་ས་ཡ།  ི«པེ་»«སྣར་»ཧཱུྃ། «པེ་»«སྣར་»ལཱ། །«པེ་»བིཧཱུཾ་བནྡྷ་བནྡྷ། «སྣར་»བནྡྷཾ་བནྡྷ་བནྡྷ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»སྔགས། «པེ་»«སྣར་»པཾ། །«པེ་»«སྣར་»«ཅོ་»ཧཱུྃ། «པེ་»«སྣར་»+གླུ་ཊ་ནི། ་«པེ་»ཏཱཀི། །«པེ་»«སྣར་»བྷུ་ཊི་ནི། «པེ་»«སྣར་»ཨེ། །«པེ་»«སྣར་»ཕྱི། «པེ་»«སྣར་བམ། «པེ་»«སྣར་ཆུང་། «པེ་»«སྣར་»ལྔ། ཀྱི«པེ་»«སྣར་»བྲོ། འ«པེ་»«སྣར་»ནི། «སྣར་»ཧཱུྃ། ཅེ«པེ་»«སྣར་»བསྟིམ། སོ《སྣར་»རོ། «པེ་»«སྣར་»རྣམས། /«པེ་»«སྣར་པས། «པེ་»«སྣར་»དེ་བཞིན་གཤེགས་པ་ཀུན་ཞི་བ། །,:ན་ག་འདྲ་«པེ་»«སྣར་»ལོ། «པེ་»«སྣར་»ལྟེ། ་.«པེ་»རྡོ་རྗེའི། «པེ་»«སྣར་»རྗེད། ཀྱི«པེ་»སྣར་»བོ། «པེ་»«སྣར་»ལ། «པེ་»ཧཱུྃ། «པེ་»«སྣར་»ཀྱིས། །«པེ་»ལི། །
