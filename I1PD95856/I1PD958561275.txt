བསྟན་འགྱུར། ་::::་རྒྱུད། ཞི་ས་ལགས་་་བསྡུར་འབྲས་རེའུ་མིག/ན་་་་་་་མེས་རྒྱུད། (ཏི་ཡེ། རྒྱལ་བ། དུ། རྩེ་བ་དང་། གང༌ངསྡེ་ཚན། །རྒྱུད། །དཔེ་གཞི། །སྡེ་དགེ། ་། ་་པེ་ཅིན། །།སྣར་ཐང་། ཀྱི་རྒྱུད། །རྒྱུད་འགྲེལ། བརྒྱུད། །ཀྱི་པོད་རྟགས། ཚན་གྲངས། །།།ནན་ལྡེབ། ནན?བ་—ན་ནན། ན°ན། །ཕྲེང་། །།་། །ཚེག་ཁྱིམ། །།་ལོར་ད་ལ། ར་འན་འང་ཡིན། ད་，བྱ་བ་དགཀཾ། །ཉོ་བས་《་《འི་ག་ཚགླི། ཆཔདྨ་བ་::། ཁ་ཀོ་ཟླ། དོ་དང་ས་ལྔ་རྫོགས་པ། དོ་ལ་ཁབ་ཀྱང་། དེ་ཚོ་འབར་བ་དང་བདག་སྒྲ་སྟེ། །:.་ཀ། །མ°»jayཏོར་སྐྱ། རིགས་ར། ཀ་ལོཌ་ཚོ་འཚོ་མ། རྫོགས་པ་ཡོ་ག་ལ་བས་མ་ཁྲི་ནི་མ་དང་རྒྱུ་མཚན་དང་བག་མ་མ་ཀུན་དུ་་་་་་་ལོ་་་་་་མི་ཞིག་ལ་ཁོ་ཚོ་ནི་སྨང་ར་ཇོ་ཁྱི་ར་བ་ལྔ་བ་ཆེན་ཁག་ལ་ལོག་བ་ལའི་འདས་ལོ་བས་ཏེ། དག་པའི་སྟེ་དེ་དག་ལ་བཟློ་དགའ་བ་ལ་པད་བས་པར་ཁང་ན་ཡེ་ཤེས་ཀྱི་མེ་འབར་པ་དང་དེའི་བར་བྱས་ནབད་པ་བདུད་བཞིས་ལེ་འབབ། བ་ལ་འབད་པ་བྱས་པ་ལས་ལ། ཏད་པ་བྱེད་པ་བྱུང་+།
