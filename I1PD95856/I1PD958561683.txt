བསྟན་འགྱུར། རྒྱུད། ཞི་བདག་ཕཊ། ༀ་དྲུ་མཙྪཱ་ཡི་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་ཕཊ་ཕཊ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨ་ར་བ་ཏཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་ད་ཧ་ད་ཧ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་མ་ཧཱ་:བྷི་ར་བཱི་ཡེ་ཧུ་ཧཱུྃ་ཕཊ། ཨོཾ་པ་ཙ་པ་ཙ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་བཱ་ཡུ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་བྷཀྵ་བྷཀྵ་བ་ཤ་རུ་དྷི་:རཱནྟཱ་མ་ལཱ་བ་ལམྦི་ནེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་སུ་རཱ་བྷཀྵི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་གྲིཉྩ་གྲིཉྩ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བྷུ་ཛཾ་ག་:སརྦནྡཱ་ཏརྫ་ཡ་:ཏརྫ་ཡ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཤུ་མ་དེ་བཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཨཱ་ཀནྡྷ་ཨཱ་ཀཟླུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་:སུ་བྷ་དྲེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧྲི་ཧྲི་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧ་ཡ་ཀཎེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་དྷཾ་ཛྷི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁ་གཱ་ན་ནེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་:ཀཾ་གྷཾ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཙཀྲ་བེ་གེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་རོ་ཧེ་ཧཱུཾ་ཧཱུཾ་ཕཊ༑ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་ཤོཎྜི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་:ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། །ཨོཾ་ཙཀྲ་བརྨི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཀི་ལི་ཀི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སུ་:བཱི་རེ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་སི་ལི་སི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་བ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཧི་ལི་ཧི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། །ཨོཾ་ཙཀྲ་བརྟི་ནི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་དྷི་ལི་དྷི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་བཱི་རྩེ་ཧཱུཾ་ཧཱུཾ་ཕཊ༑ཨོ་ཀཱ་ཀཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཨུ་ལུ་ཨེཀཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་ཤཱ་ནཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ༑ཨོཾ་སཱུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་དཱ་དྷཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་རཱུ་ཏཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་དཾབྷཱི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ་ཀཨོ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་ལཱ་:མཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ༀ་ཁཎྜ་རོ་:ཧཱ་ཡི་ཧུཾ་ཧཱུྃ་ཕཊ། ཨོ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུྃ་ཕཊ། ཅེས་སོ། །དེ་ནས་ལྡང་བར་འདོད་ན་བདག་ཉིད་ཀྱི་དཔྲལ་བ། །སྤྱི་བོ། རྣ་བ་གཡས། ་ལྟག་པ། རྣ་བ་གཡོན། སྨིན་མཚམས། །མིག་གཉིས། ཕྲག་པ་གཉིས། མཆན་ཁུང་གཉིས། ནུ་མ་གཉིས། །ལྟེ་བ། སྣ་རྩེ། ཁ། རྟ་མགྲིན་པ། སྙིང་གདཔྱི་གཉིས། །མཚན་མ། །བཤང་ལམ། བརྒྱ་གཉིས། བྱིན་པ་གཉིས། རྐང་སོར། བོལ་གོང་། པུས་།
