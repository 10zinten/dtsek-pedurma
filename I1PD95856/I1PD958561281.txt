བསྟན་འགྱུར། རྒྱུད། ཞིལས་ལ་་་་་་ཁོང་ལ་བ་དང་དེ་ལ་ཁ་ཡ་པ་རབས་དའི་དད་འདགས་པ་དང་། ས་ག་ཟླ་བབ་ད་/། ང་ཚོ་དང་དེ། འོ་ན་ད་ལ། འདུ་བ་རྒྱུ་ལ་གཏུགས་ཁ་ཁ་བྱེ་མཚན་བྱང་འདི་《སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་དཔལ་འཁོར་ལོ་སྡོམ་པའི་ཞི་བའི་ཆོ་ག་་་་་་བཞུགས། («པེ་»«སྣར་»བོར་སོ་མ་གཏང༌། («པེ་《སྣར་»པ། «པེ་»«སྣར་བཤམ། དབང་དང་«པེ་»«སྣར་ནས། །«པེ་»«སྣར་»པ། «སྣར་»བིཧཱུཾ། «པེ་»«སྣར་པ། ཚ«པེ་»«སྣར་མི་ཏ། །«པེ་»ད། ལ་«སྣར་»ཤ«པེ་《སྣར་དཔྱད། རྒྱ་བ་སོགས་ਗੁਲਸਮਾਂਸਾਲਦਾਸਿੰਗਾਰਨਾਲਉਸApਨਲRAਲས་པ་གས་ལ་དང་་་་ཁག་ལྔ་༼འ་»ན་ཚོ། བསྡུར་འབྲས་རེའུ་མིགBI(govàTHPརྒྱུད། རྒྱུད། དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། །སྣར་ཐང་། །ཅོ་ནེ། སྡེ་ཚན། རྒྱུད་འགྲེལ། །རྒྱུད། །པོད་རྟགས། ཚན་གྲངས། །::། ་། ལྡེབ། །བ—ན། ན_བ་། བན། བ་—ནཕྲེང་། །ཚེག་ཁྱིམ། :::ཀ་སུའང་དེ་བས་ལོ་ལོཡ་ཁ། །ད་ལོ་ཀན་དག་ལ་དག་བ་ག་བ་ག་བ་ག་ད་ན་བ་མ་ས་ཁག་ਉਸਵਲਈਹਲੇਸਨ।ਇਸਲਈਸਿੰਧਬRਨགནས་མང་དུ་ས་བབས་ལུ་བཀྲ་ངན་པ་གྱུར་པ། ས་བདག་བ་ཏུས། ཚོ་བ་དང་ནི་དེ་དང་། །།
