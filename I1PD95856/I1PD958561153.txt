བསྟན་འགྱུར། རྒྱུད། ༼ཞ། ཆུབ་རྡོ་རྗེ་ཡི། །གོ་འཕང་མཆོག་ནི་ཐོབ་པར་ཤོག། ཀུན་སྤྱོད་མངོན་འབྱུང་་་་་ཞེས་བྱ་མ་ལུས་པ་དང་མཁའ་འགྲོ་ངེས་བརྗོད་བླ་མ་དང་། །ལྐུ་ཡི་བ་ལ་སོགས་པ་ཡིས་ནི་མཛད་པར་བལྟས་ནས་ཀྱང་། །འཁོར་ལོའི་གཙོ་བོ་དཔལ་དག་སྐྱེ་བ་གཅིག་ཏུ་ཡོངས་སུ་གོམས་མཛད་ནས། །དཔལ་ལྡན་སམྦ་ར་རོལ་སྒྲུབ་ཐབས་བྱང་ཆུབ་གང་ཡིན་དེ་ལ་ཕྱག་འཚལ་ལོ། །སློབ་དཔོན་བྱང་ཆུབ་རྡོ་རྗེའི་ཞལ་སྔ་ནས་ཀྱིས་་་་མཛད་པ་སམྤ་ར་ཤེས་པར་བྱེད་པ་ཞེས་བྱའི་འགྲེལ་པ། དཔལ་བདེ་མཆོག་གི་མངོན་པར་རྟོགས་པ་རྫོགས་སོ། །།།།།:རྒྱ་གར་གྱི་མཁན་པོ་བྷ་དྲ་བི་བོ་དྷི་དང༌། ལལོ་ཙཱ་བ་དགེ་སློང་དགེ་བའི་བློ་་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ། །བོདོ། སྐྱོ་བ་སྐྱེ་བ། །བ། ཁོང་ཁྲོ་དང། /ཕྱོགས་ག་ལ་སྐུ་བགག་ག་བྱ་ང་ཚོ་ལ་འཁྲིག་ནས་བསྡུར་མཆན། ཅི༌དགར་བག་ག་ཝན། །མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་བདེ་མཆོག་ཤེས་པར་བྱེད་པ་ཞེས་བྱ་བ་བཞུགས་སོ། །«པེ་»«སྣར་»མ། «པེ་»སྣར་པར། «སྣར་པོ། @«པེ་»«སྣར་ཨོཾ། ཕྱི«པེ་»《སྣར་པོ། །«པེ་»《སྣར་པོ། «པེ་»«སྣར་རིམ། «པེ་»«སྣར་»+དེ་བཞིན། «པེ་»«སྣར་»ཧཱུཾ། །@«པེ་»«སྣར་ལས། ༄༅«པེ་»«སྣར་ཏྲ། «པེ་《སྣར་ཨོཾ། «པེ་»«སྣར་»མཾ། «པེ་»《སྣར་»། འགྱེ«པེ་»«སྣར་ཡཾ། «པེ་«སྣར་»མ། «པེ་»«སྣར་»ཁཾ་ལས། སྐྱེ«པེ་»«སྣར་»དང༌། ཏེ། «པེ་»«སྣར་»ཏེ། ི«པེ་»«སྣར་དྷརྨཱ་པྲ་ཀྲི། «པེ་»«སྣར་»—ཉ། ་༄༅«པེ་བསམ། །ཨ་མ«པེ་»«སྣར་ཧཱུྃ་ཧཱུྃ། (འདིའི་འོག་གི་ཧཱུཾ་ཧཱུཾ་ཞེས་པ་རྣམས་མང་ཆེ་བ་«པེ་»«སྣར་ལ་ཧཱུྃ་ཧཱུྃ་ཞེས་འཁོད་་་་ཀྱང་བསྡུར་མཆན་དུ་བཀོད་མེད། «པེ་»ཀྲྀཏྟཱ་པ་ཡ་གྲིཙྪ། ༄༅«པེ་»«སྣར་»གྷགྷ། ཉེ«པེ་»བིགྷཾ། 《སྣར་»བིག། ་«པེ་»«སྣར་»ལི། ་ཚེ«པེ་»«སྣར་»པ། བཀྱེ«པེ་»«སྣར་»དུབྦ་བིགྷཱ། ཏེ«པེ་བ། «སྣར་»ལཱ། སྟེ«པེ་《སྣར་བལྟས། «པེ་»«སྣར་ནི་བཛྲ་བྷཱ། ཀ«པེ་པཱཀསྣར་»།
