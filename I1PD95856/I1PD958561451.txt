བསྟན་འགྱུར། རྒྱུད། ཞདང་། རི ེ«པེ་»«སྣར་»ཡི། «པེ་»«སྣར་»—། «པེ་»«སྣར་»དངོས་གྲུབ་སྩོལ། །«པེ་»«སྣར་གང་། ཉི«པེ་»«སྣར་»+ཕྱོགས་དུས་གནས་སྐབས་ཐམས་ཅད་དུ། བཀྲ་ཤིས་དང་བདེ་ལེགས་ཆེན་པོས་ཁྱབ་པར་གྱུར་ཅིག། སྤྲོལ། །།།བདག་ལྟ་བུ་དང་ང་། །་་་་་་་་་་་་་་་་་་་་་་་་་་བན་པ་ཡི། །།ཤཔའི《བསྡུར་འབྲས་རེའུ་མིག་ཊཱ་བར་ཐཔའི་n/རྒྱུད། （pp:། ་་ཁོང་:/?pཀི《མའི་དཔེ་གཞི། །སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། ས་སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། །རྒྱུད། <རྒྱུད། ཀ»པོད་རྟགས། ཞལཔཁོ<'པ། ལ། ཁཞ་། པེད། ཚན་གྲངས། ་་ང་། །ལྡེབ། བ?བ?ན༽བ་ནབ། བན་ཕྲེང་། །ངོ་་། །ཚེག་ཁྱིམ། الوداعلولاهཁདང་། »བློ་བཟང་ཁང་ཀྱི་/p:ས་བྱེད་པ། མེད། ::ཀ། རྫོགས་བ་ཀྱི་(ཁམཀྱི་《ལོ། བྱ་བ་མང་བ་ཁོར་དང་མྱ་ཞི《ཀྱིསཤ་ཁོག་བསྒྲ《་དོ། //?། ཚེ། །༼《ཁ་ལྔ་པ་ང་་བ་་་་་་། ང་བའི་བོལ་ལ་སྤྱི་རྒྱ་མཚོ་ལོ་ལ་༽ཕྱི། ན། ་བམི་དུ་་་ཀྱི་་་བརབྱབཞིཀྱི་ཞུ། བའི་འི་པོ་《སྐྱེས་རྒྱུ་ཚུ་།
