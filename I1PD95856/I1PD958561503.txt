བསྟན་འགྱུར། རྒྱུད། དེ་ན་ང་nཆདང་། ཏི་ལ་ན། ལ་དེ། བུའི་བྱེད་ལྟ་བ་དང་ས་བྱ། སྐྱེ་བ་མི་དང་། ལུ་མཁོ་འད། །::དང་། དེའི་ནང་དུ་བསྡུར་མཆན། །་་། ཏི་ཀ། དང་། །་་་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»འཁོར་ལོ་སྡོམ་པའི་རབ་གནས་བཞུགས། «པེ་»«སྣར་»སུ། «སྣར་སྦྱང་བའམ། ཚ«པེ་»«སྣར་བཅས། །«པེ་»«སྣར་»+པ། _@«པེ་»«སྣར་»པདྨའི་སྟེང་དུ། «སྣར་»བརྟེན། ®«པེ་»སྣར་»ག་དེ། ཕྱག་«པེ་»«སྣར་པ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»སུ་སྣ་སྟ་པྲ་ཏི་ཤ་ཡེ། ︽«པེ་»«སྣར་»འ་ཏ། («པེ་»«སྣར་»ལ། «པེ་»«སྣར་»པ། འགར་«པེ་»«སྣར་»+ལ། p། རྒྱུ་དང་། བ་ཆུང་། //«པེ་»«སྣར་»+གྱི། «པེ་»«སྣར་པའོ། «སྣར་རིག《སྣར་»ངས། «སྣར་»ཀར། ་@«སྣར་པ། ་«པེ་»«སྣར་»དི། ::ཀྱི«པེ་»«སྣར་»དི་བྱ་ཛྷཱ་ན། «པེ་»«སྣར་»བྱ། ་ཚུ《སྣར་»བརྒྱལ། པོ«པེ་»«སྣར་»དི། ༄༅«པེ་»«སྣར་སརྦ་ཀི་ཤ་བ་ད་ནི་བྷི། «པེ་»«སྣར་»ལོ། «པེ་»«སྣར་»དོ། ཚེ«པེ་»«སྣར་»+ལ། ཀྱེ«པེ་»«སྣར་»རཏེ། «པེ་»«སྣར་»དི་བྱ་བི་སི། «པེ་»«སྣར་པུ་ཀ་ས་བཱི། @«པེ་»«སྣར་ཏྲ་ཏྲེ་ཤ«པེ་»《སྣར་»ཎ་ཤི། «པེ་»«སྣར་བརླབ་བྱའོ། པོ«པེ་»«སྣར་»ཨ་བ་ཡུ་བུ་ག་ས་པ་རི་ཀ་ར་མེ། «པེ་»«སྣར་»གི། «པེ་»«སྣར་»སིལ་བ་ལྟར། «པེ་»«སྣར་»འབུལ། «པེ་»«སྣར་»དི། «སྣར་»འདི། །(ི«སྣར་བཞག«པེ་»«སྣར་»རབ་གནས་ཀྱི། ད་ལག་པས་བབ་བཅས་པ། དེ་བས་ལས་བྱེད་པ་དང་ཁྱབ་བྱེད་ཀྱི་དོན་ཡོངས་རྫོགས་དང༌། ཕྱི་བདེ་བ་དེ་བཞེས་སུ་ལ་དོལ། མ་ཡོངང་དུ་མཉེས་གྱུར་ན་ནམ་ཏུ་འགྲུབ་པ་འདུ་བ་དང་། པོ་དང་། གང་བྱེད་དུས་པ་ཡིས། ་ཡོང་མ་འདུ་བྱེད་པ་ཞེས་བ་བར་བྱཁོར་བར་བ་བ་ན་བ་དང་། །ཁྱེར་བའི་ལ་ད་བར་དུ། །།
