བསྟན་འགྱུར། རྒྱུད། ཝ«པེ་»«སྣར་»རྟག(«པེ་»«སྣར་»ཙརྱ། ་«པེ་»+མངྒལཾ། «སྣར་+བཀྲ་ཤིས། །བ་དེ། །བར་བ། །ས་ཉིད་་་བ་ཏི་ཤད་བྱ། བར་དུ་མི་བྱེད་བ། ཤ། ་་གྲོལ་བར་འདྲ་བ། བྱ་མེད་ན་། དེ་ལས་ད་། ཡི་གེ་བསྡུར་འབྲས་རེའུ་མིག་ས་འབད་པར་བྱས་པ་ཀུ་རྒྱུད། （ལ་བར་ན་་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། ད་སྣར་ཐང་། ཅོ་ནེ། །ས། ༽པོ། རྒྱུད་འགྲེལ། པོད་རྟགས། ཚན་གྲངས། །ལྡེབ། བ་—ན་བ་བ?ན། བ?ན་ཕྲེང་། །N::། །ཚེག་ཁྱིམ། །»། བོག་པ་/neསྡེ་ཚན། རྒྱུད། རྒྱུད། །ཚེ་ཚེ་ལས་ལོ་ཐོག་ཟླ་ཀྱིབར་་་་་། ར《《། ཆ《ལg《'ཏེ《གས་དང་། ཀྱིམེ་ལོང་དང་། (ལ་པའིPར་་མ་ངོ་། ན་བ་པ་དང་《འོག་མ་བ་་་བ་ཀྱི:དྲ་བ་གསང་བ་། དོན་དང་། དཔལ《འིgའི་རྒྱཁོ་བབབབབདང་འོ་་་་། །ཁོ་པཉྩ《འིཞ།
