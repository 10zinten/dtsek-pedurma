བསྟན་འགྱུར། རྒྱུད། ཞིཔ་ཚོ་ལ་ང་ཚོ་སྤྱི་ལོ་:ཀཀ''བོ། ཀོགཀན་:*ལ་བུ་%dབདེ/n*ཚོཊོ་ཡེ་***ཁ་ཁོཊ་ཀཱ་ཊིས་ཀ/རལ་པ༌ཀ་བ་བོ་དང་དཀྱིལ་ོབསྡུར་མཆན། ་་gཏུ་gq*'། འི་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»དཔལ་འཁོར་ལོ་སྡོམ་པའི་ཕྱི་རོལ་གྱི་མཆོད་པའི་་ཆོ་ག་བཞུགས་སོ། །«པེ་»«སྣར་»མཉྫ་རི། «པེ་»«སྣར་»ལྟ། ®«པེ་»«སྣར་རྟེན། Rཊ«པེ་»«སྣར་གཞགཡི«པེ་»སྣར་»+བ། /«པེ་»«སྣར་»གཞག་སྟེ་པུ་ཛ། ་་་གཏིང«པེ་《སྣར་གཞག«པེ་གྱི། «པེ་《སྣར་རྗེའི། «པེ་»«སྣར་ཚམས། པ་«པེ་»«སྣར་»ཀྱིས། །ངས་པ། །དེ་མིང་ཀླུ་བདག་བ་དན་ ེ«པེ་»«སྣར་»ཏི། །བསྡུར་འབྲས་རེའུ་མིགརྒྱུད། (རྒྱུད། །་དཔེ་གཞི། སྡེ་དགེ། །པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། རྒྱུད། པོད་རྟགས། །ཚོན་གྲངས། །ནས་ལྡེབ། བ—བ་། བསབ་བབ། བ—བ་ཕྲེང༌། ཚེག་ཁྱིམ། །་་་་་་་་་་་ལ་ལ་ལ་ལ་ལོ་བད་པ། དེ་་པ་དེ་ནས་་་་་་བདམས་ཏེད་པ་བྱེད་དཔའ་བ་ལ། པ་གད་པ་དུ་བ་བ་ན་རྗེས་མ་ད་པའི་བདག་པ་ནས་འི་ལ་བ་དང་། ས་བ་དེ་ལ་འབད་པ་པོ་བ་བོ་མེད་པ་ལ་ལུང་དཔལ་ནའག་དང་པ་ཀློག་པ་དང་མ་བ། བཀད་པ། ལ་དོ། ལ་ལ་བདེ་+།
