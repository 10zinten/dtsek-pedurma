ར་བ་དང་། འགྲུལ་ཏུ་བཏུ་བསྟན་འགྱུར། རྒྱུད། ཞི?ཌཅུ་པོ་ལུས་ཀྱི་བདག་ལ་བློ་ཁོ་ནས་གྱི«པེ་»«སྣར་»—ཧཱུྃ། «པེ་»བྷུ་ཏ་ཏྲ། 《སྣར་»བྷུ་ཏ་ཏཱ། ཚེ«པེ་»བི། མཉེ«པེ་»སིདྷེ། ཚེ«པེ་»«སྣར་»+ཕཊ། ཚུ«པེ་»བནྡྷ་དྷ། «སྣར་»དྷ་བནྟདྷ། ?《་། «པེ་»«སྣར་»རནྟཱ། «པེ་»«སྣར་»དྷཛྷ། གླུ་། «པེ་»«སྣར་»ཧི་ཧི། «པེ་»«སྣར་དང་། «པེ་»«སྣར་དུ། «པེ་»«སྣར་ག་ཏ་ཨ་བྷཱ་ཏ་ཡ། །༄༅«པེ་»«སྣར་»ཀགྲོ་འོ། པའི་«པེ་»«སྣར་+ཕཊ། /《,་ཀྱི་ཀྱི་རྟོག་«པེ་»«སྣར་»ཚིམ། ®«པེ་»«སྣར་»རེ། «པེ་»«སྣར་»+ཛ«པེ་»ལཾ་བ་རཱ་དྷ་རཱ་ཡ། «སྣར་»ལཱ་བ་ར་དྷ་རཱ་ཡ། («པེ་»«སྣར་»ཀོ་ས་སམ་བུདྡྷཱ་ཀི་ཊི། «པེ་»ཏཾ་དཾཏྟཱ་ཀ་ལ་རཱུ་པ། །«སྣར་»ཏཾ་དཾཥྛ་ཀ་ལ་རཱུ་པཱ། «པེ་»སི་ན་ཡཱ་ན་ཡ་ཛ་ལི་ཏི་བི་ཥ་ས་སཱ། «སྣར་»སི་ན་ཡཱ་ན་ཡཱ་ཛི་ལི་ཏ་བི་ཥ་ས་སཱ། «སྣར་»ན་ད། «པེ་»ན་དཱ། «པེ་»«སྣར་»ནཱ་ག་རཱ་ཛཱ། ཚེ«སྣར་»། ཚེ«པེ་«སྣར་»ཛ་གཱ་ཡ་ཡརྣོ་བ་བཱི། ཉེ«པེ་»«སྣར་»ལེ་ཏ་བྷུ་ཛ། «པེ་»«སྣར་པུ་རི་ཏ་ཊ་་་་་བཱ། ཅེ«པེ་»«སྣར་»ཀྲུ་ཙིཏ་བྷུ་ལ་ཏཱ། ཨེ«པེ་»«སྣར་»ཥ་ཎ་རོ་ཥ་ཎཱ། «པེ་»«སྣར་»དྷུ་ཛ་ཧ་ཡ་མ་ཧོ་ར་ག་བཀ། «པེ་»«སྣར་པཎ་ག་ན་ག་ར། ཏེ«པེ་»«སྣར་»+ཡ། འཚེ«པེ་»«སྣར་»ཤཱི་ཏི་ཡོ་ཛ་ན་སྭཱ་ཧཱ་ས་ག་བྷི་ར་སཱ་ག་ར་ཛ་མུ་མ་ན་ད་ལཱ་ཡ། ཙགྲྭ་དི་ཏྱཱ་ཀཾ་དྷ། །་དགེ«པེ་»«སྣར་»ར་ལཱ་ཡ་ཧ་སི་ཏྲ་ཌཱ་ཊྚ་ཧཱ་ས་ན་ད་ནཱ་ཡ། ནགེ«པེ་»«སྣར་བྷ། མཉེ«པེ་»«སྣར་»ར་ཎཱ། འཉེ«པེ་»«སྣར་»+ལ། འགྱི«པེ་»:ཀྱི«སྣར་»སཾ་གཱ་ནི་བ་ལེ། ཀ་ས་མེ་བྷུདྡྷེ། གཉི«པེ་»«སྣར་»བི་བྷུ་ཥི་ཧཾ། མནྟཱ་མ་བཱ་ཏྲ་ཡི་་མེ། བྷོ་བྷོ་བཱནྟཱ། ན《སྣར་»རཾ། དབྱེ«པེ་»«སྣར་»ཡ། ཞཉེ«པེ་»«སྣར་»མཱ་ཀུ་ལོ་ནཱ། ལོའི«པེ་»«སྣར་»ཧཱ་བ་ལོ་ཀ་པཱ་། ཀན་ལ། ནཱི་ཛ་ཧ་ར་ཛ། ་ནགེ«པེ་»«སྣར་»+སརྦ། ཀྱེ«པེ་»«སྣར་»ཤཱིརྱ། འབེ«པེ་»«སྣར་»ཀྲྀ་ད་ཡ་ནི་བབབཊ་ཡ། གཉེ«པེ་»གཱ་ནཱ། འགྱི«པེ་»«སྣར་»+པི་ཤིཙྩཾ་དུ་ནཱ་ག་ཤ་རཱི་རཱ་ཎི། མ«པེ་»«སྣར་»+ཧུཾ། ན«པེ་»«སྣར་»ནཱ། འགེ«པེ་»«སྣར་»ཚིནྡ། མཉེ«པེ་»«སྣར་»ལ། «པེ་»ར་གྲྭ། «སྣར་»ར་ཀྲ། ®«པེ་»«སྣར་»ཏཱ་ས་ནཱ་ཡ་ཕཊ། «པེ་»«སྣར་»ཀུམྨཊོནཱ་ད་ན་ཀ་ལ་རཱ། ཅེ«པེ་»«སྣར་»ཡ། །«པེ་»«སྣར་»ཨ། «པེ་»«སྣར་»བིནྡྷ། ?༄༅«པེ་»«སྣར་རོ་བཛྲ་པ་ཎཱི་རཱ་ཙཱ་པ། ་«པེ་»«སྣར་»པ། «པེ་»«སྣར་»གོ། ཚེ«པེ་»«སྣར་»རུ། ཀྱེ«པེ་»«སྣར་»དཔའ་བོ། ཉེ«པེ་»ཀྲི། «པེ་»«སྣར་»+བཅོམ་ལྡན་རྡོ་རྗེ་མཁའ་འགྲོའི་རྒྱུད་ལ་རྡོ་རྗེ་མཁའ་འགྲོ་དང་མཚུངས་ཤིན་ཏུ་ལེགས་པར་མཁྱེན། །མཁས་པ་ཆེན་པོ་བླ་མ་མར་པ་ལོ་ཙཱས་བློ་ལྡན་བདག་ལ་གནང་ནས་བདག་གིས་བྲིས། །།་བས་ཀྱི་དར་ཀྱིཉིས་བ་《»། ཀློང་།
