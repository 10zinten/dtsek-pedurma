རི་ཎི། བྱཱ་གྷཱ་ཛི་ནཱཾ་བ་ར་དྷ་རཱ་ཡ། མ་ཧཱ་དྷུམྦྷནྡྷ་ཀ་ར་བ་པུ་ཥཱ་ཡ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཧྰུཾ་ཧྰུཾ་ཕཊ་ཕཊ། ཀ་ར་ཀ་ར། :ཀུ་རུ་ཀུ་རུ། བནྡྷ་བནྡྷ། པ་ཙ་པ་ཙ། ཧི་ཧི། ཧཾ་ཧཾ། ད་ཧ་ད་ཧ། ཀུ་རུ་ཀུ་རུ། ཏཱ་ས་ཡ་ཏྲཱ་ས་ཡ། བ་བྷཀྵ་བྷཀྵ། ཧཱུཾ་ཧཱུཾ། ཀི་ལི་ཀི་ལི། བ་ས་རུ་དྷི་ར། ་ཨནྡྲ་མཱ་ལཱ་བ་ལཾ་བི་ནེ་གྲིཙྪ་གྲིཉྩ། ཀློ་བྷ་ཡ་ཀྵ་བྷ་ཡ། ཨཱ་ཀྐཎྜུ་ཨཱ་ཀཟླུ། ཙིལི་ཙི་ལི། སི་ལི་སི་ལི། ་་ཀསཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བྷུ་ཛཾ་ག་སརྦཝཱ་ཏརྫ་ཡ་ཏརྫ་ཡ་ཏྲཾ་ཧྲི། ཕོ་ཕོ། ཤྲཱི་ཧཱུྃ། དྷི་ལི་དྷི་ལི། སྔགས་ཀྱི་ཚིག་ནི་ཀུན་ལ་ཡང་། །ཨོ་ནི་ཐམས་ཅད་གསལ་བྱེད་དེ། །ཧཱུྃ་ཧཱུྃ་ཕཊ་ནི་མཐའ་རུ་སྦྱར། །རབ་གཏུམ་ལ་སོགས་མཁའ་འགྲོ་མ། །རང་མིང་བརྗོད་པའི་སྔགས་པ་ནི། །དང་པོར་ཨོ་དང་སྦྱར་བྱ་སྟེ། །ཧཱུཾ་ཧཱུཾ་ཕཊ་ནི་མཐའ་རུ་གནས། །ཨོཾ་ཤྲཱི་བཛྲ་ཧེ་ཧེ་རུ་རུ་ཀཾ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཌཱ་ཀི་ནཱི་ཛཱ་ལ་སམྦ་རཾ་སྭཱ་ཧཱ། །སྙིང་པོའོ། །ཨོཾ་ཧཱིིཧ་ཧ་ཧཱུྃ་ྋཧཱུྃ་ཕཊ། ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་སུམྦྷ་ནི་སུམྦྷ་ཧཱུྃ་ྋཧཱུྃ་ཕཊ། ཨོ་གྲིཅུ་གྲིཙྪ་:ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོ་གྲོནྟཱ་པ་ཡ་གྲིདྡྷཱ་:པ་ཡ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ༀ་ཨཱ་ན་ཡ་ཧོ་བྷ་ག་བཱན་བིཏྟཱ་:རཱ་ཛ་ཧཱུྃ་ཧཱུྃ་ྋཕཊ། སྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ་བཛྲ་བརྞ་ནཱི་ཡེ་:ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། སྙིང་པོའོ། །ༀ་བཛྲ་བི་རོ་ཙ་ནི་ཡེ་ཧཱུཾ་ྋཧཱུྃ་ཕཊ་ཕཊ། ཉེ་བའི་སྙིང་པོའོ། ་། ཨོ་བཱི་ར་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཇིསྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོའོ། །རྗེ་བཙུན་མའི་སྔགས་ཀྱི་ཚིག་གོ། །ཨོཾ་ཧ། ན་མ་ཧི། སཱ་ཧཱ་ཧུ། བོ་ཥཌ་ྋཧེ། ཧཱུཾ་ཧཱུཾ་ཧོ། ཕཊ་ཧཾ༄༅། དེ་དག་ནི་དཔའ་བོ་དྲུག་གི་གོ་ཆའོ། །ཨོ་ཐཾ། ཧཾ་ཡཾ། ཧྲི་རྨོ་། ཧྲེ་ཧཱུྃ། ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོ་ནི་ཀུན་ལ་སྦྱིན་པར་བྱ། །རྣལ་འབྱོར་མ་དྲུག་གོ་ཆ་ཡིན། །ཨོཾ་ཨ་ཨཱ། ཨི་ཨཱི། ?ཨུ་ཨཱུ། རྀ་རཱི། ལི་ལཱི། ཨེ་:ཨཻ། ཨོ་ཨོོ། ཨོཾ་ཨ་ཀ་ཁ་ག་གྷ་ང་། ཙ་ཚ་ཛ་ཛྷ་ཉ། ཊ་ཋ་ཌ་ཌྷ་ཎ། ཏ་ཐ་ད་དྷ་ན། པ་ཕ་བ་བྷ་མ། ཡ་ར་ལ་ཝ། ཤ་།
