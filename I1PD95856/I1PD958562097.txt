།རྒྱུད། ཞི་པར་བྱའོ། །ཞེས་སོ། །བདེ་མཆོག་སྒྲུབ་ཐབས་བསྡུས་པ་ལས། །དྲི་མེད་བསོད་ནམས་འཁྲུངས་པ་དེས། །མཐའ་ཡས་སྐྱེ་བོ་འདི་རྣམས་ནི། །རྡོ་རྗེ་དབང་ཕྱུག་མྱུར་ཐོབ་ཤོག། ཅེས་སོ། །་། དཔལ་སྡོམ་པ་འབྱུང་བའི་སྒྲུབ་ཐབས་:རྫོགས་སོ། །ས། །་་༌་༌་ཏུ་ཀ་ལ་དང་བ་དན་གྱིས་བློ་་ཟླ་་་་་་རྒྱ་བལ་པོའི་པཎྜི་ཏ་ཆེན་པོ་གྷཱནྟི''ཤྲཱིའི་ཞལ་སྔ་ནས་་བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་་་ཤོང་བློ་བརྟན་གྱིས་བལ་པོའི་ཡུལ་གོ་:ཏ་མ་བྷི་ཧ་རིར་བསྒྱུར་བའོ། །།།ਲਨੂੰਖਾਰਜਇੰਸਥਾਸਕਵਰਪਲਵਿਚਬਾਲਨਾ་་༌་་་ལོ་་་་་། །བསྡུར་མཆན། ང་ལ་བལ་བྱས་ཀྱི་ਟམཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་»«སྣར་»དཔལ་སྡོམ་པ་འབྱུང་བའི་སྒྲུབ་ཐབས་་་་་་བཞུགས་སོ། །«པེ་»«སྣར་»སཾ་པ་རོ་དཱ་ཡཱ། ན«པེ་»«སྣར་»འབྱུང་བ། «པེ་བསྲུང་། །«སྣར་བསྲུངས། «སྣར་»ཡིས། «པེ་སྒོམ། ༄༅«པེ་»«སྣར་བསྟན་པ། «པེ་»«སྣར་»སྒོམ་པ། ས:::་པ་:a«པེ་»«སྣར་»+པ་རྣམ། 《«པེ་»«སྣར་»ཧཱུཾ། ་«པེ་»«སྣར་»ཧཱུྃ་ཧཱུྃ། «པེ་»«སྣར་»བཞག@«སྣར་རྗེའི། «པེ་»«སྣར་རྣམས། «པེ་»«སྣར་»ཡི་གེ། ::། ཚར་བ་ལ་ང་«པེ་»«སྣར་»བྷ། «པེ་»«སྣར་དྷཆ་། «པེ་»«སྣར་»ཉ། «པེ་»«སྣར་»ཨུ། «པེ་»«སྣར་བྷ། «པེ་»«སྣར་»ཉ། «པེ་»«སྣར་»གྱིས། «པེ་»«སྣར་»བཞག་«པེ་»«སྣར་བཞག་«པེ་»སྣར་»ལ། «པེ་»«སྣར་པོ། «པེ་»«སྣར་རློན། «པེ་»«སྣར་གཡོན་པར། །«པེ་»«སྣར་»གྱི། «པེ་»སྣར་»ཀྱིས་བཅོམ་ལྡན་འདས་ཀྱི་བརླ་གཉིས། «པེ་»«སྣར་»ཎི། «སྣར་སེན། «སྣར་»བཞི། «པེ་»དྷ། ་«པེ་»«སྣར་»དྷུ། «པེ་»«སྣར་»དྷིཤྩ་ཎྜི། ༼བ«པེ་»«སྣར་»མཱ། «པེ་»《སྣར་»གྱིས། །་:འཁྲི«པེ་»«སྣར་»། འ«པེ་»«སྣར་»ནཾ། «པེ་»«སྣར་+ཀཱ་ཡ་བཱཀ«པེ་»«སྣར་»ཉ། «པེ་»«སྣར་དྷརྨ། «པེ་»«སྣར་»ཉ། «པེ་»«སྣར་»པ། «པེ་»«སྣར་»སྙིང་པོ། «པེ་»«སྣར་»+མིག་གཉིས་དང་། «པེ་»«སྣར་ཡ་ཧྲི། «པེ་《སྣར་ཧྲེ་ཧྲི། «པེ་»ག་།
