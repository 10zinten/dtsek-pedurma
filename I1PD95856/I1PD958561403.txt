བསྟན་འགྱུར། ་རྒྱུད། ཞཀྱི་གོ་ਇਲਾਵਾਮਲਥਪਾਵਰਪਲਨਰਕཔ་ལ་དང་འཚོ་བ་ལ་ང་ཚོ་སུ་ཡང་སྤྱིར་བསྡུར་མཆན། །ཀ་ང་ད་ན་བ་མཚན་བྱང་འདི་གཞུང་ལས་བླངས་ཏེ་བཀོད་ཅིང་། «པེ་དཔལ་འཁོར་ལོ་སྡོམ་པའི་དཀྱིལ་འཁོར་གྱི་བཀྲ་:དོ་ཤིས་ཚིགས་སུ་བཅད་པ་བཞུགས། «པེ་»«སྣར་འཁྱུད། «པེ་»«སྣར་སྤྱོད། །ཀྱང་ཡང་ཚེ«པེ་»«སྣར་རྗེའི། «པེ་»«སྣར་»སྟེང་། «པེ་»«སྣར་ལ། ཀྱེ«པེ་»«སྣར་ནི། ཉི«པེ་»«སྣར་ཀྱི། «པེ་»«སྣར་ཐེ། «པེ་»«སྣར་»+ཀྱི། མ«པེ་»«སྣར་བསྲུང༌། །ཀུག། ༽བྱང་ਪਰਵਾਨਵਰਗਨੂੰਵਰਤਕਰਲਓབ་ལ་རྒྱུས་ལ་བསྡུར་འབྲས་རེའུ་མིག་ལ་བརྒྱུད། (བྱཱ་ལ་ས། །ཀུན་་ཉིན་T/དཔེ་གཞི། སྡེ་དགེ། པེ་ཅིན། སྣར་ཐང་། སྡེ་ཚན། :རྒྱུད། སྤྱི་རྒྱུད་འགྲེལ། རྒྱུད། རྒྱུད། ་དང་འཔོད་རྟགས། ཞrrisཔགས་པདཔལ་གྱི་བྱ་ཚན་གྲངས། །ལྡེབ། ནནན?ན?བ—བ་། ནན་ཕྲེང༌། །ཚིག་རྐང༌། །་་:༼ན་/s། ཚེག་ཁྱིམ། ལབདེ་བ་བདེ་བ་དམ་པ་དང་ལ་ལ་ཤིས་ཤོང་བ་ཤིན་ན་བདུ་པ། ས། །མཛེས་པའི་དེ་དག་ས་དག་ལ་འབད་པས་འབར་ཞིང་། །ད་གི་འབད་པའི་དྲིལ་བུ་ད་པ་ལན་་་་་་་བན་མ་ར་ལ་ལ་ད། །ན་བ་མ་ཡིན་བྱ་བ་ད་།
