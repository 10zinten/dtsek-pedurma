ལྔ་པ་བསྟན་འགྱུར། རྒྱུད། ཞི་༌པ་ཀོ་ཝ་པར་གསོལ་བ་བཏབ་ནས། ཨོ་ཡོ་ག་ཤུདྡྷསརྦ་དྷརྨ་ཡོ་ག་ཤུདྡྷོཉཧཾ། ཞེས་བརྗོད་ཅིང་པད་ཀོར་གྱི་ཕྱག་རྒྱས་དགྱེས་པར་བྱས་ནས་ཕྱག་རྒྱ་དེ་བཏུལ་ལ་འཁྱུད་པའི་་་་སྟངས་སྔོན་དུ་འགྲོ་བས་རྡོ་རྗེའི་ཚོགས་ཌི་ཀས་དེ་ཁོ་ན་བཞིན་དུ། ཨོ་བཛྲ་མུཉཞེས་པ་གཤེགས་སུ་གསོལ་ཏེ། འཁོར་ལོ་དེ་བདག་ཉིད་ལ་རབ་ཏུ་གཞུག་པར་བྱའོ། །དེ་དང་དེའི་ས་ལ་གནས་པའི་མྱོས་བྱེད་ཀྱང་མིང་མེད་ཀྱིས་བླངས་ཏེ། སྙིང་དང་། །ལྕེ་དང་། སྤྱི་བོར། ཧཱུྃ་ཨཱཉཨོ་གྱི་ཡི་གེ་བརྗོད་པ་སྔོན་དུ་འགྲོ་བས་བྱུགས་ནས་ལག་པ་དེ་ལ་གནས་པའི་ལྷའི་འཁོར་ལོ་ཡང་བདག་ཉིད་ལ་རབ་ཏུ་ཞུགས་པར་བལྟ་བར་་་་བྱའོ༑། །ཨོཾ་ན་མོ་བྷ་ག་བ་ཏཱི་བཛྲ་བཱ་:རཱ་ཧཱི་ཐཾ། ཨཱརྱ་ཨ་པ་རཱ་ཛི་ཏི་ཏྲེ་ལོཀྱ་མཱ་ཏེ་མ་ཧཱ་:བིདྱེ་ཤྭ་རི་ཡིསརྦ་བྷཱ་ཙིཏ་བྷ་ཡཱ་པ་ཧེ་མ་ཧཱ་བཛེ། བཛེ་ས་ནེ་ཨ་ཛི་ཏེ། ཨ་པ་རཱ་ཛི་ཏེ་བ་ཤྱ་ཀ་རི་ནི་ནོ་ཏྲ་བྷ་མ་ཎི། ཤོ་ཥ་ཎི། རོ་ཥ་ཎི། ཀྲོ་དྷ་ཀ་རཱ་ལི་ནི༑ཏྲ་ོས་ནི། མཱ་ར་ཎི། བྲ་བྷེ་ད་ནི། །ཨ་པ་རཱ་ཎིཛ་ཡེ། བི་ཛ་ཡེ། །ཛམྦྷ་ནི། སྟམྦྷ་ནི། མོ་ཧ་ནི། །བཛྲ་བཱ་རཱ་ཧཱི། མ་ཧཱ་ཡོ་གི་ནཱིཏྟཾ། ཀཱ་མེ་ཤྭ་རི་ཁ་གེ། ཏདྱ་ཐཱ། པྲ་ཏང་གེ། པྲ་ཏང་གེ། ཧ་ན། ཧ་ན། པྲ་ཎཱ་ཀིཾ་ཀི་ནི། ཁེ་ཁི་ནི། དྷུ་ན། དྷུ་ན། བཛྲ་ཧཱ་སྟེ། ཤོ་ཥ་ཡ། ཤོ་ཥ་ཡ། བཛྲ་ཁ་ཊཱ་ག་ཀ་པཱ་ལ། མཱ་ལ་ང་རི་ཎི། མ་ཧཱ་པི་ཤི་ཏ་མོ་ས་ཤི་རོ་མ་ནུ་:བྷཱནཾ་པཱ་པྲྀཏྟེ། སཾ་ནི་རྒྱུ་ཅན་ར་ཤི་རོ་མཱ་ལ་གྲོ་ཐི་ཏ་དྷཱ་རི་ཎི། སུམྦྷ་ནི། སུམྦྷི། ཧ་ན་ཧ་ན། པྲ་ཎཱ་ན། སརྦ་པཱ་པཾ་ས་ཏཱ་ཝེནཾ་སརྦ་པ་ཤུ་ཝེནཾ་མ་ཧཱ་མཾ་སཙྪེ་ད་ནི། ཀྲོ་དྷ་མུརྟིས། །དཾབྷཱ་ཀ་རཱ་ལི་ནི། མ་ཧཱ་མུ་དྲེ། ཤྲཱི་ཧེ་རུ་ཀདེ་བ་སྱ་ཨཱ་གྲ་མ་ཧི་ཏི་ས་ཧ་སྲ་ཤི་རེ། ས་ཧ་སྲ་བཱ་ཧ་བེ། ཤ་ཏ་ས་:ཧ་སྲ་ཙིཨཱ་ན་ནི༄༅། ཛཱ་ལི་:ཏ་ཏེ་ཛ་སེ། ཛྭ་ལ་མི་མུ་ཁི་པི་ག་ལ་ལོ་ཙ་ནེ་བཛྲ་ཤ་རཱི་རེ། །བཛྲ་ས་ནི། མི་ལི་:མི་ལི། ཏི་མི་ལི། །ཏི་མི་ལི། ཙིལི་ཏ། ཧེ་ཧེ། ཧ་ཧ། ཧཱུཾ་ཧཱུཾ། ཁ་ཁ། དྷ་དྷུ། རུ་།
