བསྟན་འགྱུར། རྒྱུད། ཞི་ཉིན་མོ་མཛད་ཀྱིས་མཛད་པ་རྫོགས་སོ། །་། དེ་རྗེས་དུ་མ་ད་རྒྱ་གར་གྱི་མཁན་པོ་ཤྲཱི་སུ་:བྷཱུ་ཏི་ཤཱནྟི་དང་། །ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་དགེ་བའི་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ། །།།ད་དུང་དེ་དག་གི་བྱ་བ་དང་། ཚོ་བ་ཁ་ཤོར་ལ་དལ་ཏི། ལ་དགབལ་ཏི། །erས་མ་ཆུ་བོ་བྱ་བ་དང་། །ག་དང་འོད་ཀྱི་བསྡུར་མཆན། ་་་ནའི་ཁ་ཎ་ཁོ་ན་བགྱིམཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད་ཅིང་། «པེ་»«སྣར་»འཇམ་དཔལ་རྡོ་རྗེའི་སྒྲུབ་ཐབས་བཞུགས། «པེ་»«སྣར་»ཀྱི། «པེ་»«སྣར་»ཀྱིས། «པེ་»«སྣར་»+དེ། «པེ་»«སྣར་»ཧུཾ་ཧཱུཾ། @«པེ་»«སྣར་»ཧཱུྃ་ཧཱུྃ། ༄༅«པེ་»«སྣར་ཧཱུྃ་ཧཱུྃ། «པེ་»«སྣར་ཧཱུྃ། ལོ་«པེ་»«སྣར་»ཨ་ལས།  ི«པེ་»«སྣར་»ཏཱ་ན་བིབྷཱ་བི་ཡ། :«པེ་»«སྣར་»ཀྱི། «པེ་«སྣར་»ཉ། «པེ་»«སྣར་»ཀྱིས། །«སྣར་»ལ། «པེ་»«སྣར་»གྱི། «པེ་»«སྣར་བའི། «པེ་»«སྣར་»བའི། «པེ་»«སྣར་»+པ། །«པེ་»«སྣར་»+པ། «པེ་»«སྣར་»ཞལ་ཞི་བ། «པེ་»«སྣར་མ། «པེ་»«སྣར་»ཛ། ཉི«པེ་སྣར་»ཨུཏྟན། «ཅོ་»ཨོཏྟན། «པེ་»«སྣར་»ལྟོ། ®«པེ་»འི་«སྣར་»ནི། «པེ་»«སྣར་རི། @«པེ་»«སྣར་»བི་ཀོ་ཊི། «པེ་《སྣར་»ནི། «པེ་«སྣར་»དྲེ། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་»ཀཉྩ། «པེ་»«སྣར་»ལྒངས། «སྣར་»མ་ལཱ། «པེ་»«སྣར་»མིག་::«པེ་»«སྣར་»པུ། @«པེ་»ཏ་ར། «སྣར་»ཏཱ་ར། »@«པེ་»«སྣར་ཧ། «པེ་»«སྣར་སོའུ་རབྦ། «པེ་»«སྣར་»ནི། འགུ«སྣར་»ཁྲབ། «པེ་»«སྣར་»ལོ། «པེ་»«སྣར་»ཏར། Q«པེ་»«སྣར་བར། »«པེ་»«སྣར་»སྤྱོད་པས། མ«པེ་»«སྣར་མོ། གང་ལྟར་༼ང་ཀྱི་(«པེ་»«སྣར་»གཤིན་རྗེ་མཆེ་བ་མ་དང་། «པེ་»«སྣར་»ཁྱི། «སྣར་»རོ། །ཀྱི«པེ་»«སྣར་»ནི། «པེ་»«སྣར་»ནི། ཚེ«སྣར་གིས། «པེ་»«སྣར་»མ། «པེ་»«སྣར་»ལ། «པེ་»«སྣར་»+ལྐོག་མ་ལ། ༄༅«པེ་»«སྣར་»ཧས། «པེ་»བི། \།
