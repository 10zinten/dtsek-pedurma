རྒྱུད། ་ནུའི་བྱ་ཧ་ཧ་ཧ་ཧ་ཧོབྷ་ག་བཱན་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་མཱ་མེ་མུཉྩ་བཛཱི་བྷ་བ་མ་ཧཱ་ས་མ་ཡ་སཏཱ་ཨབདག་ཅག་སེམས་ཅན་དོན་ཀུན་མཛད། ་། ཇི་ལྟར་རྗེས་རྟོགས་དངོས་གྲུབ་སྩོལ། །སངས་རྒྱས་ཡུལ་དུ་གཤེགས་པ་དང་། །སླར་ནི་བྱོན་པར་མཛད་གྱུར་ཅིག་། ཨོ་མུསྭཱ་ཧཱས་གཤེགས་པར་བྱའོ། ་། སྨོན་ལམ་གྱི་བྱེ་བྲག་ཀྱང་གདབ་པར་བྱའོ། །དུས་གསུམ་བསེགས་དང་ཡོད་པའི་དགེ་བ་ཡིས། །བདག་ཅག་སྲིད་ཀུན་ཕ་རོལ་ཕྱིན་ནས་ནི། །སེམས་ཅན་ཀུན་བཅས་རིན་ཆེན་རྟོགས་ལྡན་པའི། །བླ་ན་མེད་པའི་འབྲས་:བུ་ཐོབ་པར་གྱུར། རྗེ། འཕགས་པ་ཨ་ར་པ་ཙ་ནའི་མཆོད་པའི་ཆོ་ག་ཞེས་བྱ་བ་རྫོགས་སོ་། :་ཁེ། ཚུ་ཁེ་ལུང་དུ་སྟོན་པའི་བསྟན་བཅོས་ཀྱི་སྐད་ཅུང་ཟད་རྟོགས་པའི། །ཤཱཀྱའི་་དགེ་སློང་ཉི་མ་རྒྱལ་མཚན་དཔལ་བཟང་པོས། གྲུབ་པའི་ཞིང་དམ་པ་མང་་་་ཡུལ་སྐྱིད་གྲོང་འཕགས་པ་འཇིག་རྟེན་དབང་པོའི་གཙུག་ལག་ཁང་ལས། :ཅུང་ཟད་མི་རིང་བའི་ཆེན་པོ་རྣམས་ཀྱི་གདན་ས། དཔལ་:ཤེར་གྱི་གཙུག་ལག་ཁང་དུ་བསྒྱུར་བའོ། །།།བདུན་པ་བདག་བཟང་། མི་འགལ་བའམ་མི་མེད་ན་ਉਸਸਸਲੇਮਸਮਸਆਪਰਸਨ'ਚਕੀਤਾਸੀ।་་་་་་་་་་་་་་་་་་་་་་་:༌་་་་ཁྲི་མཁན་པོ་དཀའ་བཞི་པོ་མརྗེ་བཙ་ཊ་ལུ་འབག་ཊི་ཀ་ག་ང་ཚོ་ཁོ་ན་བར་དུ། སྐུ་ལ་ཟླ་བ་ལྟ་བུ་ཁོ་ལ། །བདག་ཆགས་དང་། །ཆ་ཚོང་དབང་ལ་པ་དང་། ཏ་ཟིག་ན་དག་གཙོ་བསཞུང་:། ཀ་བ་ཁྲུའི་ན་རོ་འཚོ་བ་ལ་ང་ཁ་པ་དང་། ཁ་ཤས་བྱ། །ཀ་ར་སློབ་།
