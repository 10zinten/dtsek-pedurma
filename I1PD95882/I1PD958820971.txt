བསྟན་འགྱུར། རྒྱུད། ༼ནུ་་ན་བ་དང་ལུང་ལ་སྐྱོན་བཀག་པས་འདུ་བ་ང་ཚོ་རྒྱའི་མི་ཁྲི་ཁ་ཕྱོགས་དེར་ཁ་གས་པ། ལྔ་པ་དེ་ཁོ་མ་དེ་། གྲོ་། ཡི་ག་་འོ་དང་། ཏེ་ཛོ་ཀ་བ་མ་ཀཏུ་གྱུར་ན་ཊ་ཀྱི་བུ། གལ་འ་བསྡུར་མཆན། དུ་བབ། །ད་ལྟ༌ng(མཚན་བྱང་འདི་«སྡེ་»«ཅོ་ལྟར་བཀོད། འ«པེ་»«སྣར་»+རྒྱ་གར་སྐད་དུ། བཛྲ་པ་ཎི་ཨ་བ་བ་ལ་ནི་བི་དྷི། བོད་སྐད་དུ། ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་གཏོར་མའི་ཆོ་ག། བཅོམ་ལྡན་འདས་ཁྲོ་བོའི་རྒྱལ་པོ་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་ལ་ཕྱག་འཚལ་ལོ། །ཆུ་གཏོར་བྱེད་པར་འདོད་པས་དང་པོ་གཏོར་མའི་སྣོད་ཅི་འབྱོར་བའི་ནང་དུ་ཆུ་གཙང་མ་སྲོག་ཆགས་མེད་པས་ཁེངས་པར་བཀང་། དེའི་ནང་དུ་ཕྱེད་དང༌། རིན་པོ་ཆེ་ལྔ་དང་། སྨན་དང་། མེ་ཏོག་ལ་སོགས་པ་ཅི་འབྱོར་པ་བཏབ་ལ། («པེ་»«སྣར་+། «པེ་»《སྣར་ཤུདྷོ་ཧཾ། «པེ་»«སྣར་ཧཱུཾ། ་ག«པེ་ན་མ། «སྣར་»ཎཾ། «པེ་»«སྣར་»ཏཱཏ། «སྣར་»བརླབས་ཏེ། དེ་ལ་དབྱེ་Rਅਮਵਿਲਕਦੇਨੇ«པེ་—། 《སྣར་+། «པེ་»ཤ_《སྣར་»ཤོ། ®«པེ་»། །«པེ་»ཎི་ཨ་མི་ཏ། «སྣར་»ཎཾ་ཨ་མྲྀ་ཏ། པོ«པེ་པ་བྱའོ། «སྣར་བྱས། «པེ་སིདྷི་མ། 《སྣར་»སིདྡྷཾ། འ«པེ་»སརྦ། ་《སྣར་སརྦ། «པེ་»«སྣར་བརླབས། ལོ«པེ་»ཏེ། «སྣར་»—རོ། (《སྣར་སིདྡྷ། །«པེ་»«སྣར་»ཡེ། །ཅེ«པེ་»—ལ། «སྣར་པ། «པེ་»«སྣར་། «པེ་»«སྣར་»བཟང་། ཚེ«སྣར་»ཛོ། སོཔེ་»«སྣར་»སྒྱུར། ་་བབ་མར་ཡིག་རྗེ། །ལང་ਨੂੰਸਵਾਲਕਰਰਮਲੁਲਨਾਲਤੇਲབསྡུར་འབྲས་རེའུ་མིགརབ་གནས་རྒྱུད། (རྒྱུད། དཔེ་གཞི། །སྡེ་དགེ། །པེ་ཅིན། །སྣར་ཐང་། །སྡེ་ཚན། །རྒྱུད། །།རྒྱུད་འགྲེལ། རྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། །བན་། ནན་། ནན་ན་—ན། ཕྲེང་། །།།ཚེག་ཁྱིམ། །།་ན་པསཔལ་ར།
