པ་ཤེས་རབ་ཀྱིས་གསན་ནས་ཞུས་པའོ། །།།་་་་་:འི་འད། །ཆ་ར་གའི་°(། ར་ར། ཀ་ཀྱི་ལས་དྲང་ཁ། ཀུ་ཚོ་ལས༌དང་བར་་་་་། །།་ར། ཎ་ཊ་ཀྱི། དེ་ལ་པ། །ཀུན་དང་། ར་ལ་ཤ་ས་དྲུག། བསྔགས་པ་། དྲ་མང་བ་ནིབདུད་མཆན། ་(། །ཆུ་། ཤ། །e:ཀ༽མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། །«སྣར་རྡོ་རྗེ་རྣམ་པར་་་་་:དུ་འཇོམས་པའི་གཟུངས་ཀྱི་སྒྲུབ་ཐབས་ཆོ་ག་ཞིབ་མོ་དང་བཅས་པའོ། །་ཁ«པེ་»«སྣར་»ཎི། «པེ་»《སྣར་པའོ། །«པེ་»《སྣར་རྗེ། །《སྣར་རྗེ། «པེ་བརྙེས་བྱིན་བརླབས། 《སྣར་མཉེས་་བྱིན་བརླབས། གསང་རྒྱུད་ཀྱི་བ། :ད་། པ། ཁ་བ་བབས་(པེ་»«སྣར་»ན། །(«པེ་»པ་བསམ། 《སྣར་པའི་བསམས། 《སྣར་དང་། &«སྣར་»ནི། །::ཙཊི་(། 《སྣར་»ཀྱིས། «པེ་《སྣར་ལ། @《པེ་»《སྣར་»ནི་སྤྱི།  ི«པེ་《སྣར་རྡོ་རྗེའི། «པེ་»《སྣར་+རབ་ཏུ། «པེ་»བསྒྲུབ། (《སྣར་_དང༌། །«པེ་»«སྣར་—མན་ངག་སྟེ《སྣར་»ཏེ༑«པེ་《སྣར་གཙང་ཞིང༌། «པེ་»«སྣར་»སྒྲུབ། <«པེ་»«སྣར་»ལ། («པེ་»། །ཁཀོ«སྣར་བརྒྱན། ངོ«པེ་གཅད། །དུར། ་«སྣར་བཅད། །དུར། ེ«པེ་»«སྣར་གྲུབ། བར་བ་འགྱེ«པེ་»«སྣར་དུ། ཁ་བ་སྐྱེ། །ཆུ་ཏྲེ་ཏ་་ཀྱི་(«པེ་གྱོན། «པེ་བྱས། «པེ་»《སྣར་རྐུ། «པེ་གནོལ། 《སྣར་རྣམས། «པེ་»བཏོན། ཁོ«པེ་»《སྣར་»ར། «པེ་»«སྣར་»ཀྱིས། «པེ་བྱང་། ེ«པེ་»«སྣར་»སྟེང་། «པེ་» ེ«སྣར་»པས། «པེ་»«སྣར་རྗེའི། «སྣར་»ཀུམ། ®«པེ་»འཇིག«པེ་»སྡིགས། «སྣར་»ཀྱི། «སྣར་»སར། «པེ་»པས། «སྣར་ལ། ཚེ«སྣར་མདོ་བ། །«པེ་»«སྣར་»ཀི། «པེ་»སྡིགས། «སྣར་»+སྔོ་སྐྱ་དེ་བཞིན་འཇིགས་པ་ལ། །རྡོ་རྗེས་མཚན་པའི་ཐོ་བ་སྟེ། །«པེ་»«སྣར་»སྤྲིན། «པེ་»«སྣར་རྔམ། «པེ་»«སྣར་»དབང་པོ། «པེ་»«སྣར་གཅད། «པེ་ཉེ་བར་བསྟབ། «པེ་སྐྱོང་འཁོར། «སྣར་བསྐྱོང་འཁོར་གྱི། ཁྱོ«པེ་»《སྣར་བརྟན། 《སྣར་བརྒྱས། སྟེ«པེ་»གཉིས་ཀྱིས། «པེ་»གྱིས། ཨེ«པེ་»«སྣར་»+བྱང་ཆུབ། «པེ་»«སྣར་རྣམས། «པེ་»«སྣར་»འདུལ། །།(ི«སྣར་ཡིས། འ«པེ་»«སྣར་»ལོ། །གནད། «སྣར་འཇོམ། ཚེ«པེ་»«སྣར་མཛད་་་་ཕྱིར། །«པེ་»«སྣར་»འདུལ། «སྣར་ནས། «པེ་《སྣར་»སྐུའི། ཉི«པེ་»རླབས་ལག་པ་ལེགས་བཟུང་། 《སྣར་བརླབས་ལག་པ་ལེགས་བཟུང་། 《སྣར་མ། (ཁྱེ«པེ་»«སྣར་»སྦྱང་། །
