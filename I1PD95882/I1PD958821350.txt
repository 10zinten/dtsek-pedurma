གང་གིས་ཡེ་ཤེས་འདི། །འདིར་བཞུགས་རྣམས་ཀྱིས་ཐོབ་པར་ཤོག། རྡོ་རྗེ་་རྣམ་པར་འཇོམས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་རིན་ཆེན་སྣང་བ་ཞེས་བྱ་བ་སློབ་་་་་དཔོན་ཆེན་པོ་ས་ཝ་རི་པས་མཛད་པ་རྫོགས་སོ། །ལ། བལ་པོའི་བླ་མ་དེ་བ་པུརྞ་མ་ཏི་དང་། ལོ་ཙཱ་་བ་དགེ་སློང་ཆོས་ཀྱི་བཟང་པོས་བསྒྱུར་བའོ། །་་་་། དེངང་པོའི་ཆུ་ལོ་ཊ་ཏུ་ཊ་ཐོབ། ་་་་་་་ནང་ཁབ་ཀྱི་ཀྱི་། །པོ་་་ནང་ད་བར་སྤྲོ། །གཡོ་རྒྱུ་དེ་ཚོའི་འཚེའིམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །རྡོ་རྗེ་རྣམ་་་འཇོམས་པའི་གཟུངས་ཀྱི་དཀྱིལ་འཁོར་ཀྱི་ཆོ་ག་རིན་ཆེན་སྣང་བ་ཞེས་བྱ་བ་བཞུགས། སྣར་»རྡོ་རྗེ་རྣམ་་་་འཇོམས་ཀྱི་གཟུངས་ཀྱི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་རིན་ཆེན་སྣང་བ་ལ་སོགས་པ་རྣམ་འཇོམས་ཀྱི་བསྐོར་བཞུགས་སོ། །(«པེ་»«སྣར་»དྷ། «པེ་»མ་དྷ་ར་ནི། «སྣར་མ་དྷ་ར་ཎི། ®«པེ་»«སྣར་དུ། «པེ་»ཏི། །ཁགཡོཊཁོ«པེ་»«སྣར་པའི། «པེ་»«སྣར་ཡི། «སྣར་བརྩམས། །(«པེ་»+། 《སྣར་སྒོ། །«པེ་བཟུང་། ་«པེ་གདུགས་དབུབ་པོ། «སྣར་གང་དུ་བུའོ། ཀས《སྣར་»སོགས། །པོ«པེ་»གྱི་ལས་བྱ་སྟེ། «སྣར་གྱིས་ལས་བྱས་ཏེ། 《སྣར་བསྟན། «པེ་»དེའི《སྣར་རྣམས། ་«པེ་བཤམ། 《སྣར་སཞྭ། («པེ་»«སྣར་བྲོ། ལོ་བློ་(«པེ་—། «པེ་»«སྣར་ཏྲ། «པེ་»གི། ས«པེ་《སྣར་ཊཾ། «པེ་»ཏཾ། «སྣར་»ཏྲ། ཏེ། «པེ་»«སྣར་»ཏཾ། །«པེ་ཏེ། 《སྣར་གིས། «པེ་»ཀའོ། «པེ་མིན། །«པེ་»hཀ《སྣར་དྲང་། («སྣར་—སྐུ་གསུང་ཐུགས་བྱིན་བརླབ་པ་ནི། །ཨོཾ་ཨཱ་ཧཱུཾ། ་«པེ་སརྦ། ལུ«པེ་»(«པེ་»ཀྲྀ་རྀ་ཏཾ། «སྣར་»ཊཾ་རཾ་ཏཾ། «པེ་»བསྒྲུབ། «པེ་ན། «པེ་»གྷཎ་གྷཎ། བྷུརྞ་པ་ཡ། གྷཎ་པཱ། «སྣར་གྷཱཎྞ་གྷཱཥྛ་བྷཱུརྞ་པ་ཡ། བྷཱུརྞ་པ། ས«པེ་《སྣར་»ཡ་བོ་དྷ་ཡ། «པེ་བུ་དྷ་བོ་ཊི་བྷེ་དྷ«པེ་»«སྣར་»ཏཾ། («པེ་སརྦ་བི་ཧཱུཾ་བཛྲ་བཛྲ། སྤོ་ཊ་ཡ་བཛྲ་བཛྲ། ཀ་ཊ་ཡ་བཛྲ་བཛྲ། མ་་ཊ་བཛྲ་བཛྲ། མ་ཐ་བཛྲ་བཛྲ། བཛྲ་ཨ་ཊ་ཧ་ས་ནཱི་ལ་བཛྲ་སུ་བཛཱ་ཡ། ཧཱུཾ་དྷི་ཧཾ་རཾ་ཏཾ། ་་«པེ་»ཡང་། «པེ་རྩིའི། (ེ《སྣར་དུ། (བུ«སྣར་པོ། (ལེ«པེ་»«སྣར་བརླབས། 《སྣར་»ལོ་ཡིས། 《སྣར་+བྱེ། «པེ་»གླིན། 《སྣར་གྲི་ཧ། «པེ་རུ་བི་ཛ་ཡཱ་ཡ་སྭཱ་ཧཱ། «པེ་»ནཱ་ཡ། ་(«པེ་»ཚིན་དྷ་།
