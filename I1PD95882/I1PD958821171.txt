བསྟན་འགྱུར། །རྒྱུད། ༼ནུ༽མ་བློཏོ«པེ་»ཨོཾ་བཛྲ་ཨ། 《སྣར་ཨོ་ཨ། «པེ་《སྣར་—། 《སྣར་»ཀྱིས། «སྣར་དྷརྨ། །«པེ་སྒྲེང༌། «པེ་སྲུང་། «པེ་《སྣར་བ། «པེ་»«སྣར་པ། «པེ་»«སྣར་»གི་ལས་མཛད་དུ་གསོལ། ་《པེ་《སྣར་»+ལོ། །་《སྣར་པར། «པེ་»《སྣར་»མཛད་དུ་གསོལ། 《སྣར་གིས། ལ«པེ་»«སྣར་བས། ཀང«པེ་»པ། བའི་ནང་། དང་། པ་ལ། ཚ་«པེ་»བསྒྱུར། «པེ་»«སྣར་དབུལ། ::«པེ་»གཙོ་མོ། «པེ་ནམ་མཁའ་ལྡིང་དང༌། །«སྣར་ལས། ་«པེ་+རང༌། 《སྣར་—ལ། །།«པེ་ཀྱི་ལྷ་མོ། «སྣར་»ཀྱི་ལྷ། «པེ་ཟངས་ཀྱིས། «སྣར་ཟངས་ཀྱི། «པེ་»«སྣར་»::ས་སྤོས། 《སྣར་དབུལ། ས«པེ་པེམ། «སྣར་»—པོ། «པེ་»«སྣར་+འདས། །«པེ་»ལྔའོ། །ཚོགས་བསག་ཐུགས་རྗེས། ཨོཾ་ཨཱཿཧཱུྃ་པ་ན་པེམ་པེན་ཀུ་རག་ཥ་ཁཱ་ཧི། རྡོ་རྗེ་ཕུར་བུ་བེ་་་ཅོན་ཐོ་བ་རྣམས་འགྲེའོ། །ཨོ་ཨཧཱུྃ། 《སྣར་»ལྔའོ། །ཚོགས་བསགས་ཐུགས་རྗེས། ཨོཾ་ཨཱཿཧཱུྃ་པ་ན་པོ་པེན་ཀུ་རུ། རཀྟ་ཁཱ་ཧི། རྡོ་རྗེ་ཕུར་བུ། བེ་ཅོན་ཐོ་བ་རྣམས་འདྲེའོ། །ཨོ་ཨཧཱུྃ། :《སྣར་ཀྱིས། 《སྣར་བཞི། «པེ་»«སྣར་དབུལ། «པེ་»སྣར་བསག«པེ་《སྣར་འགྱུར། །པ་«པེ་རཀྵ། «པེ་རྩ་བརྒྱད་སྔགས་ཀྱི་སྐྱལ། འོསྣར་བརླང་། «པེ་—པ། «སྣར་»སྒྲུབ། «པེ་»བསྒྲུབ༑«པེ་《སྣར་—ཀྱི། 《སྣར་+བསྐྱེད། «པེ་»«སྣར་བཞི། །«པེ་བསྐུད། «པེ་»«སྣར་བརླབས། «པེ་»《སྣར་»དོ། །«པེ་《སྣར་འགྱུར། ས《པེ་»ཏྲ། ཧཱིཨ༑《སྣར་»ཀྲྀ་ཧྲིཨ། 《པེ་ནི། 《སྣར་»མཾ། «པེ་»«སྣར་»ཏྲ། «པེ་»ཀརྨ་བཛཱི། «སྣར་»ཀརྨ་བཛྲ། 《སྣར་གྱིས། «པེ་»གྱིས། 《སྣར་»+དེ་ནས། །«པེ་»«སྣར་»ཏེ། «སྣར་རྟོགས། ་«པེ་»«སྣར་+པར། «པེ་»«སྣར་རྗེ། «པེ་»《སྣར་»སྟི། «པེ་《སྣར་»ཡ། «པེ་སཏཱ། «པེ་»སྣར་»ནས་ཞལ་ཟས་ལ། «པེ་»《སྣར་—མོ། 《པེ་»གསུམ་པའོ། །།《སྣར་གསུམ་མོ། །ད་«སྣར་—ཆེན་པོ། «པེ་《སྣར་»ཙ། ::ཀྱི་འཚོ་བར་འབད་བའི་གང་་་ང་བ། །，།
