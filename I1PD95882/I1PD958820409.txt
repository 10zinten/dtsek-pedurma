རྒྱུད། ༼ནུའི་ਸཤར་ཇི་བཞིན་དྭངས་པར་མཛོད། །དཔལ་ལྡན་མཁས་པ་རྣམས་ཀྱི་ནང་ནས་་་་རྒྱལ་མཚན་རྩེ་མོ་ལྟར་ཤིན་ཏུ་མཐོ་ཞིང་གསལ་བར་གྱུར་པའི་པཎྜི་ཏ་ཆེན་པོ་མ་་་་ཏིས་འཇམ་པའི་དབྱངས་ཀྱི་སྒྲུབ་ཐབས་དང་བསྟུན་ཏེ། ཌ་བསྐྱེད་པའི་རིམ་པའི་་གསལ་འདེབས་བདེ་བའི་ཏིང་ངེ་འཛིན་གྱི་བསྟོད་པ་རྫོགས་སོ། །ང་ལྔ་ཁ། །ཁ་:། །པཎྜི་ཏ་ཤཱཀྱ་ཤྲཱི་བྷ་དྲ་དང་། བོད་ཀྱི་ལོ་ཙཱ་ཁྲོ་ཕུ་བ་གནུབས་བྱམས་པའི་་་དཔལ་གྱིས་སྲིན་པོའི་རིར་བསྒྱུར་བའོ། །།།ཁ་དང་ལ་ཁ་རྗེ་བས་སྐཁོ་ཊ། ਰਦਾਸ'ਲਈਨਿਆਰੀਸੌਸབལ་ཏུ་ཁཚཁྱའི༌ལ་བྱ་ཚོ་ལ་ཁདང་། ཀོ་ཁང་ཟླ་བ་བསྡུར་མཆན། ཆུ་། །:ན། །འོད་དང་ད་ཀ་ལ་་། །དེ། ད་དུང་བཅch?z་། z༽། །་། །མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང༌། ཆོས་ཚན་འདི་«པེ་»«སྣར་»ལ་མི་བཞུགས་པས་《ཅོ་དང་དཔེ་བསྡུར་བྱས་ཡོད། «ཅོ་»པོ། «ཅོ་ཐིགས། ཟང་། ་་་གཙོ་བོ་བམ། ཁ། །བོད་། ལ། ལ་། བྱང་འ་ཀྱགས་ནི། །འོ་པཎ་ག་ལ་འོད་དང་། ལན་དུ་མའམ་བསྡུར་འབྲས་རེའུ་མིག་ཆ་མ་ཕྱེད་པས་ཀྱི་ཁ་ལ་བྱང་། ཆུ་འདོན་རྒྱུད། (་་་་་་་་ལས། །jp/jའི་པེ་ཅིན། །སྣར་ཐང༌། ཅོ་ནེ། ཞི་། ནང་ཀྱི། དཔེ་གཞི། །སྡེ་དགེ། དང་། སྡེ་ཚན། མ་རྒྱུད། པོད་རྟགས། བགས་དང་ནུས་པ་ཚན་གྲངས། ལྡེབ། །བ—ན་ཕྲེང་། ་ཚིག་རྐང་། ཀུན་ནས་། ཚེག་ཁྱིམ། བ་རན་། __།
