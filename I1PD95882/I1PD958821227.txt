:ལན:ག་་་་། །བུ་རྒྱ་ཁྲི་ག་ག་ལ་བས་ནག་པོད་བྱབ། །པོ་སུ་འདུ་བ་ལ་འབྲས་ཀྱི་དེ། །:རྨ་་་་་་ན་འདབ་༼ག་དཀའ་བ་སྟེ། །དེ་་་་་་་༌་་འཁྲི་བ་ཀ་ཚེ་བའི་ས་ཁ༼ངོཔག་བྱས་པ་དང་། ཚོང་ཟླ་༽ཀླ་e༄༅། །རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་ལས་བཞི་པ། སྒྲུབ་པའི་སྦྱིན་སྲེག་གི་ཆོ་ག་ཚོང་བ་ལ། །ཁོང་བྱེད་པ། ལ། ཁྱོཌ་་་ཙ་ན་ཀ༌ཁ་དང་ཟླབ་ལ་ནི། །ཁྲོ་ཁོའི་བདག། ་་་::ཀྱིངབད་་་་་་ཁོ་དང་ར་བ་h@་《ག། G། །༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་བི་དཱ་ར་:ཎ་ཀརྨ་ཙ་ཏུ་སཱ་དྷ་ན་་ཧོ་མ་བི་དྷི། བོད་སྐད་དུ། རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་ལས་བཞི་སྒྲུབ་པའི་སྦྱིན་སྲེག་གི་ཆོ་ག། རྡོ་རྗེ་རྣམ་པར་འཇོམས་པ་ལ་ཕྱག་འཚལ་ལོ། །ཞེ་སྡང་ཆེན་པོ་ཁྲོ་བོའི་སྐུས། །འགྲོ་བའི་ཞེ་སྡང་གདུག་པ་འཇོམས། །རྡོ་རྗེ་ཅན་ལ་ཕྱག་་་་་་འཚལ་ཏེ། །སྦྱིན་སྲེག་ལས་ཀྱི་རིམ་པ་བཤད། །བསྙེན་པས་ལྷ་རྣམས་ཉེ་བར་བྱེད༑། སྦྱིན་སྲེག་འདོད་པའི་འབྲས་བུ་སྟེར། །དེ་བས་གང་བསྒྲུབ་འབྲས་་་ཡོད་པར། །འདོད་པའི་སྦྱིན་སྲེག་ལས་ལ་འབད། །་། ཞི་བས་ནད་དང་གདོན་་་བགེགས་ཞི། །རྒྱས་པས་ཚེ་སྟོབས་ལོངས་སྤྱོད་སྤེལ། །དབང་གིས་:བདག་་་་གཞན་དབང་དུ་བྱེད། །དྲག་པོས་གདུག་པ་ཚར་གཅོད་པའོ། །བགེགས་མེད་དངོས་གྲུབ་འབྱུང་བ་ཡི། །གནས་སུ་ཞི་སོགས་བརྩམས་པ་དང་། །།གང་ན་་མ་རུངས་བགེགས་གནས་པ། །དེར་ནི་དྲག་པོ་བྱ་བ་ཉིད། །དཀར་བའི་་་་་།
