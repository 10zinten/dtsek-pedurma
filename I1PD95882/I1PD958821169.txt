བསྟན་འགྱུར། རྒྱུད། ༼ནུ། །ཎ་ཡོ་བློ་ཡིས་བསོད་ནམས་གང་ཐོབ་པ། །རྣམ་འཇོམས་སྐུ་ཡི་ཞལ་མཐོང་ཤོག། དཀྱིལ་འཁོར་གྱི་ཆོ་ག་དགའ་བའི་སྒྲོན་མ་ཞེས་བྱ་བ་རྒྱ་གར་གྱི་སློབ་དཔོན་:ཆེན་པོ་རྡོ་རྗེ་གོ་ཆས་མཛད་པ་རྫོགས་སོ། །།་་་་་བོད་ཀྱི་ལོ་ཙཱ་བ་སྤུ་རངས་བསོད་ནམས་བློ་གྲོས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་་་གཏན་ལ་ཕབ་པའོ། །།།ཁྱོད་དང་། ག་དང་ཁ་ཡོར་ཀ་ལི་བ་དག་ལ་ਸਾਹਿਬਨਲੂ'ਪਲਵੀਨਸਲੋਂਸ਼ਰਨབྱེད་་བསྡུར་མཆན། བྱ་བ་ལ། ཁ་་གང་། །ཆུ་ཏཱཀཱཏོམཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། _«པེ་»༄༅། །རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་བཞུགས། «སྣར་»༄༅། །རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་དཀྱིལ་འཁོར་ཆོ་ག་རྡོ་རྗེ་གོ་ཆས་མཛད་པའོ། །«པེ་»བི་ད་ར། «སྣར་»བིདྷ་རཱ། ་«སྣར་»པ། «སྣར་»+ཧཱུྃ༑(ས«པེ་»«སྣར་། «པེ་—ལ། @«པེ་ཨོཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཨོཾ་ཏྲི་ཧཱུྃ་ཧྲི་ཧཱུཾ། ཧ་ཧཱུཾ། 《སྣར་ཧཱུཾ་ཧཱུཾ་ཏྲཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ཧ་ཧུཾ། 《སྣར་»འགྱུར། «པེ་»«སྣར་»—ག«སྣར་བསག«པེ་»《སྣར་»ཛོ། 《སྣར་»རཱ། «སྣར་»སp། ས«པེ་དྷུ་པེ་ཧཱུྃ། «སྣར་»དྷུ་པེ། པོ«པེ་»ཏྭཾ། «པེ་»«སྣར་བྲི། རཔེ་ནི། "«སྣར་རྣམས། ་གྱཔེ་»«སྣར་+ག་ག་ན། «པེ་《སྣར་»+ཧཱུྃ། «པེ་»བི་ད་ར་བཛྲ། 《སྣར་»བི་དྷ་ར་བཛྲ། «པེ་»«སྣར་»—ཨོ། «པེ་»«སྣར་»། «པེ་»«སྣར་»—ཎ། «པེ་»«སྣར་»བཅོམ་ལྡན་རྣམ་འཇོམས་འཁོར་་་་་བཅས་ལ། །དད་པས་དབུལ་ལོ་བཞེས་སུ་གསོལ། །ལག་ཏུ་རི་བོ་«པེ་《སྣར་»བཅོམ་ལྡན་རྣམ་འཇོམས་འཁོར་བཅས་ལ། །དད་པས་དབུལ་ལོ་བཞེས་སུ་གསོལ། །«པེ་»དུ་འོང༌། «པེ་»—། «པེ་»«སྣར་ཎ། «པེ་»བ་གན་དྷ་རད་ན། སྣར་»བ་གྷཎྜ་ར་ཏན། «པེ་བསག་«སྣར་—དང། འོ«པེ་བསྒྲལ། 《སྣར་རྒལ། «པེ་»—པ། «པེ་བྲལ། «པེ་»«སྣར་ཨ། «པེ་བསྟན་པ་བཛྲ་ནབ། 《སྣར་བསྟན་པ། འབྱེ«པེ་བྲལ། ༌«པེ་བསགསུ«པེ་»«སྣར་»བིད། །«པེ་»«སྣར་ཧཱུཾ། «པེ་»ཀུ། «སྣར་»སྡིག«པེ་སྒྲེང༌། ས《སྣར་»སྟེང་དུ། 《སྣར་སྟེ༑«སྣར་བུ། 《སྣར་ལ། «པེ་»སཾ། ལ༌«པེ་ལ་སེ། «པེ་»ཏྲང་། «སྣར་»ཏྭཾ། །«པེ་»«སྣར་»གཱི་ཏི་ཧྲིགཱི་ཏི། «པེ་ནིར་ཏི། «པེ་པ། ཀྱི«པེ་»+ཀར། ་འ«པེ་»«སྣར་»ཏྭཾ། «པེ་»«སྣར་»ཧྲི། «པེ་»«སྣར་»«ཅོ་»ཊཾ། །«པེ་»_།
