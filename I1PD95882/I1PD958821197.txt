བསྟན་འགྱུར། རྒྱུད། ནུ་་་་་་་་་་་བརྒྱད་དང་དུ་བ་ཀ་ཏ། དབློ་དམན་པ། །རྒྱུ་དང་བླ་མ་ਨੂੰਮਲਮਸਲਾਗਲਾਸGEYAAਨਸਲ་བསྡུར་མཆན། །ངོ་བ་་་ཁ་་་ཌ་། །ཆོ་གབབནབ་་་། (མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་རྫོགས་བྱང་ལས་བླངས་ཏེ་བཀོད། 《སྣར་བསམས། ་«པེ་འགྱུར། 《སྣར་»གཟེར། «སྣར་རྗེ། «པེ་»འདི། ་ས«པེ་»བསྒྲུབ། །དེਨੂੰ་་་་ད་འོ་ਸਨਤਨਪਾਸਟਰਨབསྡུར་འབྲས་རེའུ་མིག་ལ་བདུ་ཀ། ཏུཏྟ། ་ཀ་ལ་ན། རྒྱུད། ་(ང༌། ཁབ་བྱ། ཀ་ཅ་ལ་པེ་ཅིན། རྒྱུད། དེས་དཔེ་གཞི། །སྡེ་དགེ། །།སྣར་ཐང༌། །སྡེ་ཚན། །རྒྱུད། །རྒྱུད་འགྲེལ། ཀྱིརྒྱུད། པོད་རྟགས། ཚན་གྲངས། ལྡེབ། ནབ་ནབ་བའ—ན་ཕྲེང་། །།།ཚིག་རྐང་། །།།ཀྱ་བངཚེག་ཁྱིམ། །།།ན་_བ་གནས་ཡོད། འོ་རྗེ་བདུད། དུ། ཉིད་ལ་འདུས། བ་ནག་བྱ་བད་པ་དུ་མ་པ། །དེ་རྗེ་ཉི་མའི་པ་དང་ལུང་བ་མེ་དཔུང་འབར། །།
