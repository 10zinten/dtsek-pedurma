རྩེ་མོའི་ཆ་ལས་བྱུང་བ་མ་ཤིང་གོ་བ་ནི་བཅུ་པ། ཉེས་ལ། །བ། ན་ཞིང་འདབ་མ་བཏུས་པ་དང་མ་དང་བླ་མས་མེད་པ་སྟེ། །ད་བ་ས་ལ་བའི་ལས་ཤིང་ལ། ་བསྡུར་མཆན། དུམ་དག་པས་ན། །བདྲིར་འབྲས་བཅུ་མགནཔ་དང་། ད་བར་མ་མ་དུལ་ན་མ་མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །ཁྲོ་བོ་རྡོ་རྗེ་་་འཛིན་པའི་སྒྲུབ་པའི་ཐབས་བཞུགས་སོ། །་འ«པེ་»«སྣར་ན། །«སྣར་བསྲུབ་པའི། «པེ་»སྒྲུབ། །«པེ་»འགྲོས། «པེ་»«སྣར་»སྦྱང་བ། ༄༅«པེ་»སརྦ། «སྣར་»ག་བ། «པེ་《སྣར་»དྷ༑«པེ་»«སྣར་+ཤབྷ་ཧཱུྃ། །་«པེ་《སྣར་»དང་། «སྣར་»སྦྱང༌། «པེ་»«སྣར་རྡོ་རྗེའི། ®«པེ་《སྣར་པའི། །«སྣར་»གིས། པོ@«པེ་»«སྣར་»། «པེ་»། «པེ་»ཤ(ཀྱེ《སྣར་+ཧེ། རྗེ་དྲིལ་«པེ་《སྣར་བློ་བྷོ། «པེ་»«སྣར་»རཏ་ན། «སྣར་»—། «སྣར་»ཉིད་ནི། །«སྣར་»འབྲེལ༑«པེ་ཏ། །འོ་ན་གདུང་རིལ་ཉོ་བྱེད་པ་ཁྱོད་དང་། ལ་དོན་དག་ཡོད་པ་ལ། ::@་པས(EP:/pབོན་ན། །དབང་། དཔ། ཏུ་པའི། །།།།ས་བང་བ་ཟླ་ངོ་ཁ་རྒྱུད། (ཡོ་ཡས་ཀཀཀཀློགgན། །།དཔེ་གཞི། སྡེ་དགེ། སྣར་ཐང་། སྡེ་ཚན། །རྒྱུད། རྒྱུད་འགྲེལ། པོད་རྟགས། །ནུཚན་གྲངས། དུ་་བ་། :། a:ནང་ལག་ལ་ལྡེབ། བབ་ན་ནནནབ་བཕྲེང་། །།ཚིག་རྐང་། །ཚེག་ཁྱིམ། །ཀྱི་ཨ༌་འོད་་་༌ཀ་ག་བའི་ལས་ཀ་བྱེད་ཀའི། །དད་པ། ཁ་ཁེ་གའི་ང་ཚོ། ་་་་་ཀ་པ་ਕੇਮੈਸੁਇੱਸਨੂੰਦੂਸਰਿਸਨਾਲਯਤཔེ་ཅིན། །རྒྱུད། །
