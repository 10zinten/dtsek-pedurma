རྒྱུད། ཀུནུ་ཁ་ལཁ། ཁཀྱི་ལ་སྒོ་ལ་སོགས་པ་བྱའོ། །འདིར་གང་དག་དེ་སྙེད་ཀྱི་ཕྱག་རྒྱ་དག་དམིགས་པར་མི་ནུས་པས་ཕྱག་བརྒྱད་པའམ། ཞལ་གཅིག་ཕྱག་བཞི་པའམ། ་་:ཞལ་གཅིག་ཕྱག་གཉིས་པ་ཡང་བསམ་ཞིང་ཕྱག་གཉིས་པ་པདྨ་དང་མཆོག་སྦྱིན་པར་བལྟའོ། ཀ་ཀ་། སྒྲུབ་ཐབས་ཆོ་ག་བཀོད་པ་ཡིས། །དགེ་བ་གང་ཞིག་་དྲི་མེད་ནི། །བདག་་་་གིས་བསགས་དེས་འགྲོ་བ་ཀུན། །སྲིད་མཚོའི་ཕ་རོལ་འགྲོ་བར་ཤོག། འཕགས་པ་སྤྱན་རས་གཟིགས་:དབང་ཕྱུག་ཞལ་བཅུ་གཅིག་པའི་སྒྲུབ་ཐབས། དགེ་སློང་མ་་་དཔལ་མོས་མཛད་པ་རྫོགས་སོ། །།ཁྲོ་། ཀ་བ་ལྟ་དང་། །སྐྱེ་བ་ཀ་བ་ཀྱ་དང་རྒྱ་གར་གྱི་མཁན་པོ་དཱི་པཾ་ཀ་ར་ཤྲཱི་ཛྙཱ་ན་དང༌། ལོ་ཙཱ་བ་རིན་ཆེན་བཟང་པོས་བསྒྱུར་བའོ། །།།ཀྱི། །།།ན་ཁ་བྱ། ཀཀྐཁམ་བློ། ::ཀྱིངཎ་ਮਿਲਾਕਿਕਲਰਕਮਰਸਤੁਮਾਝਨਾਕਿ། :་:བསྒྱུར་མཆན། འཁལ་ཁབཀ་འཁ་ཏོ། །བྱུག་པ་ལ། དབྱ::ཚར། ཁཱ་ཀཱཀའི་རྒྱུད་ངན་(མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད་ཅིང་། «པེ་»༄༅། །སྤྱན་རས་་་གཟིགས་ཞལ་བཅུ་གཅིག་པའི་སྒྲུབ་ཐབས་བཞུགས། 《སྣར་»༄༅། །འཕགས་པ་སྤྱན་རས་གཟིགས་ཞལ་བཅུ་་་་གཅིག་པའི་སྒྲུབ་ཐབས་དགེ་སློང་མ་དཔལ་མོས་མཛད་པ་བཞུགས། «པེ་»ཀྵ་ར། «སྣར་»ཎཱ་ར། ྋ«པེ་::《སྣར་ཀི། «པེ་»《སྣར་»ད་ཤ(«སྣར་པགས་པ། «པེ་»«སྣར་»གཏུགས། :«སྣར་ཤིག«པེ་《སྣར་ཞིང་། «པེ་《སྣར་»+ལ་ལར། «པེ་《སྣར་—ཉེ་བར། །«སྣར་»ཀུན་ནས། «པེ་»«སྣར་པ་ལྟའོ། «པེ་»«སྣར་+པྲ་ཏཱིཙྪ། «སྣར་»སྐྱེ། «སྣར་»བ། པ«པེ་ས། «སྣར་སྣ། /«པེ་»«སྣར་»ཉ། ་《སྣར་»—རྣམས། །རྟག་«པེ་»«སྣར་—བསམ་པར། «པེ་»《སྣར་—དང༌། 《སྣར་ལས། «པེ་»«སྣར་»འཕགས་པ། ས«པེ་《སྣར་མ་ཤར་བ། «པེ་»«སྣར་བའི་སྐུ། «པེ་»«སྣར་འཁྲུགས། «པེ་»བལ《སྣར་»སྐོར། 《སྣར་བྲལ། «པེ་»«སྣར་པ། /«པེ་»«སྣར་ཕྱེ་བ། «པེ་»«སྣར་འདས། རྣམ་པར། པེ་《སྣར་འགྲེང༌། །།།།སྐུ་སྐུ་ཀ་ཀ། ཀ་ཀུE«པེ་»«སྣར་པས། 《སྣར་ཐུགས། ་«སྣར་དེ། «པེ་《སྣར་འགྲེལ་པ། ས《སྣར་ཐཾ། པའ«པེ་»«སྣར་»ལ་མ། «པེ་»ཡ། 《སྣར་ཨ། «པེ་《སྣར་»རྟེན། ་་་་་་—.།
