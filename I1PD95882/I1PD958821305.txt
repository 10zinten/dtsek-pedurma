བསྟན་འགྱུར། རྒྱུད། །༼ནུདོག་ནག་མོ་ཁྲོ་གཉེར་བསྙེར་བ་སྨད་གཡོགས་ནག་པོ་གྱོན་པ་འཇིག་རྟེན་གྱི་་་ཁམས་རང་གི་འོད་ཀྱི་སྣང་བས་འབར་བར་བྱེད་པ་དབྱུག་པ་ཐོགས་པའོ། །ཙ་བ་སཀྵ། ཕྱག་རྒྱ་ནི་ཐལ་མོ་ཁོང་སྟོང་དུ་སྦྱར་བ་ལས། མཛུབ་མོ་གཉིས་དང་གུང་མོ་གཉིས་ནང་དུ་བཀུག་སྟེ་མཐེ་བོ་གཉིས་དང་སྲིན་ལག་གཉིས་མཐེའུ་ཆུང་་་གཉིས་བསྒྲེང་བའོ། །སྔགས་ལ། ན་མསརྦ་ཏ་ཐཱ་ག་ཏེ་བྷྱོ། D་སརྦ་:བྷ་་ཡ་བི་ག་ཏེ་བྷོ། བི་ཤྭ་མུ་:ཁེ་བྷུ། སརྦ་:ཐཱ་ཧཾ་ཁཾ་རཀྵ་རཀྵ་མ་ཧཱ་བ་ལེ། སརྦ་ཏ་ཐཱ་ག་ཏེ་སྤུ་:ཎ་ནི་རྫ་ས་ཏེ་ྋཧཱུཾ་ཧཱུཾ། ཏྲ་:ཊ་ཏྲ་ཊ་ཨ་པྲ་ཏི་ཧ་ཏེ་སྭཱ་ཧཱ། རྡོ་རྗེ་ཁུ་ཚུར་གྱི་སྔགས་ལ། ཨོཾ་བཛྲ་ད་མ་ད་མ་ཧཱུྃ་སྤོ་:ཊ་ཡ་བྷསྨི་ྋཀུ་རུ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ༑བསྐྲད་པ་དང་། བདུག་པ་དང་། ཆུའི་ཁྲུས་དང་། བྱབ་པ་དང༌། །བསྲུང་བ་རིམ་པ་བཞིན་དུ་བྱའོ། །ཁྲུས་ཀྱི་ཆོ་གའི་རིམ་པ་རྫོགས་སོ། ཥ་ཌ་:ོ། །་་་་་་་་་་་ཚོན་པར་བས་པ་བློ་དེར་པ་དང་བསྒྱུར་མཆན། །ཏ་དང་བརྗེ་བར་.མཚན་བྱང་འདི་《སྡེ་»«ཅོ་»ལྟར་བཀོད། «པེ་ནི། «པེ་»གསལ། «པེ་《སྣར་བརླབས༑ས«པེ་ད་ར་ད། «སྣར་བྱ་བ། «སྣར་»མོའི། །ཀྱི«པེ་ཏ་བི་པི་ཏ་ས་པ། «སྣར་»ཊ་ཀུ་བི་ཏ་སངས། «པེ་»«སྣར་»+ཕཊ། «པེ་»+། «པེ་«སྣར་»དྷ། ས«པེ་»ར་བྱ་ཡཱ། །«པེ་»བིགྷ་ནཎ། 《སྣར་»བིབྷཱ་ནཾ། «པེ་»«སྣར་»ཀ་(«པེ་《སྣར་ཐཱ་མ་ཐཱ། «པེ་»དུཏྟན། «སྣར་དུདྡན། («པེ་བྷས་རྨི། <<པེ་»«སྣར་དུཋན། མེ«པེ་»བིགསྣེ༑«སྣར་»བིགྷཱ་ནཾ། ེ«པེ་»བྷུ་ཏི་། （སྣར་བྱ་ཏི་ཐཱི། «པེ་»«སྣར་»། །«པེ་སྲིད་ཏུ་བེ་། «སྣར་མི་ཏྱཱ་བྷུ། ་«པེ་»གསྣ་། «པེ་རཾ་དྭ་བྱ། «པེ་»སཱ་བ་ནི་། *«པེ་དུར་དུ། «སྣར་དུར་ནི། «པེ་། འཀྱེ«པེ་»། «སྣར»ཏ་བྱ། འབུ«པེ་»། ནའི«པེ་ན་ཡེ། འ«པེ་»བིག་སྣེ། 《སྣར་བིགྷཱ་ནཾ། ནས་«པེ་»བྷུ་ཏི། «སྣར་བྱ་ཏི། ནང་འཁོ«པེ་»—། འ«པེ་»ཡེ། «པེ་བིག་ནན་རཀྵ་རཀྵ། 《སྣར་»བི་ག་ཏེ་ན་རཀྵ་རཀྵ། འགྱི《སྣར་»། །ཀྱེ«པེ་བྷུ་ཏི་བྱུ། «སྣར་»_།
