རྒྱུད། ༼ན༽#ལ་ཡ་བཛྲ་པུལླེ་སཾ་སྭཱ་ཧཱ། ཚ་གཡས་སུ་ཨོ་ཀེ་ཤི་རྩེ་བཛྲ་པུངྒེ་ཙཾ་སྭཱ་ཧཱ། །གཡོན་དུ་ཨོ་་ཨུ་པ་ཀེ་ཤི་རྩེ་བཛྲ་པུལླེ་ནཾ་སྭཱ་ཧཱ། རང་རང་གི་སྔགས་འདིས། ་མེ་ཏོག་དང༌། །བདུག་སྤོས་དང་། མར་མེ་དང་། དྲི་མཆོག་དང༌། ཞལ་ཟས་ལ་སོགས་པ་རྣམས་ཀྱིས་ཡང་དག་པར་མཆོད་ནས། གཏོར་མའི་སྔགས་ཀྱིས་གཏོར་མ་དབུལ་བར་བྱ་སྟེ། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷརྨཱ་ཎཾ་ཨཱདྱ་ནུཏྤནྣ་ཏྭད་ཨོཾ་ཨཱཿཧཱུྃ་ཧོgའདི་་་་དག་གིས་གཏོར་མ་སྦྱང་བར་བྱའོ། །དེ་ནས་ཨོཾ་ཨ་ར་པ་ཙ་ནཱ་ཡ། ་་ཀུ་མཱ་རི་ད་ཧ་ན་དཀཱ་ཡ། ཙནྡྲ་ཀཱནྟི་མ་ཎི་པྲ་བྷཱ་ཡ། ཁངྒ་པུསྟི་ཀ་བྱཱ་གྲྭ་ཡ། ཧསྟ་ཡཾ་མ་བ་ནི་ཝ་ར་པྲ་དཱ་ཡ། ཨི་མཾ། བ་ལིཾ། གནྡྷཾ། ་པུཎྞཾ། །དྷཱུ་པཾ། དཱི་པཾ། མ་དྷཱ། ཤ་ཀ་རྭ་རི། ས་ཧི་ཏཾ། གྲིཉྩ་གྲིཉྩ། ཁ་ཁ། །ཁཱ་ཧི་ཁཱ་ཧི། མ་མ་ཤཱནྟི། པུཎྚི། རཀྟ་བ་ར་ཎ་གུཔྟ་ཀུ་རུ་ཨོ་ཨཱདྷིཧཱུྃ་སྭཱ་ཧཱ། ཀ་གཏོར་མའི་སྔགས་སོ། །དེ་ནས་མངོན་པར་བསྟོད་པར་བྱའོ། །དེ་ནས་ཚིགས་སུ་བཅད་པ་འདི་བརྗོད་པར་བྱ་སྟེ། །རྨོངས་པའི་བློ་ཡིས་བདག་གིས་ཅི་ནས་ཀྱང༌། །ནོངས་པ་བགྱིས་དང་བགྱིས་སྩལ་གང་ཅིའང་རུང༌། །ོ། གང་ཕྱིར་ལུས་ཅན་ཀུན་གྱི་སྐྱབས་ཡིན་ཕྱིར། །དེ་ཀུན་མགོན་པོ་ཁྱོད་ཀྱིས་བཟོད་པར་འཚལ། །དེ་ནས་ཚིགས་་་བཅད་འདིས་སྨོན་ལམ་བྱ་སྟེ། དགེ་བ་འདི་ཡིས་བདག་ཉིད་ནི། །སངས་རྒྱས་ཉིད་དུ་མངོན་གྱུར་ནས། །འགྲོ་ཀུན་གཏིང་མཐའ་མེད་པ་ཡི། །སྲིད་པའི་མཚོ་ལས་སྒྲོལ་གྱུར་ཅིག། དེ་ནས་ཚིགས་བཅད་འདིས་གཤེགས་སུ་གསོལ་བར་བྱ་སྟེ། ཁྱེད་ཀྱིས་སེམས་ཅན་དོན་ཀུན་མཛད། །ཇི་ལྟའི་རྗེས་མཐུན་དངོས་གྲུབ་སྩོལ། །སངས་རྒྱས་ཡུལ་དུ་གཤེགས་ནས་ཀྱང་། །སླར་ཡང་འབྱོན་པར་མཛད་དུ་གསོལ། །ཚུལ་ཁྲིམས་ཙནྡན་གྱིས་ལུས་བྱུགས། །བསམ་གཏན་དག་གི་གོས་གྱོན་ཞིང་། །བྱང་ཆུབ་ཡན་ལག་མེ་ཏོག་བཀྲམ། །།ཇི་ལྟར་བདེ་བར་བཞུགས་སུ་གསོལ། །།དེ་།
