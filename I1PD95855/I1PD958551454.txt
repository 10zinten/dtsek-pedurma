ཨོཥྛཎ་དུ་ལོངས་སྤྱོད་དོ། །:ཨཆུ་དར་ཐིམ་མོ། །ཛ་ལནྡྷ་རར་བདག་ཏུ་བྱེད་དོ། །པུཎ་གི་རིར་ཁྱབ་བདག་ཉིད་དོ། །ཨཅུ་དར་ལོངས་སྤྱོད་དོ། །པུཎ་གི་རིར་ཐིམ་མོ༑། ཨོཥྛཎ་དུ་བདག་ཏུ་བྱེད་དོ། །ཛཱ་ལནྡྷ་རར་ཁྱབ་བདག་ཉིད་ཅེས་བྱའོ། །དེ་བཞིན་དུ་གོ་:དཱ་པ་རཱིར་ལོངས་སྤྱོད་དོ། །།རཱ་མེ་ཤཱ་རིར་ཐིམ་མོ། །མཱ་ལ་བར་བདག་ཏུ་བྱེད་དོ། །།དེ་བཱི་ཀོ་ཊིར་ཁྱབ་བདག་ཉིད་དོ། །རཱ་མེ་ཤྭ་རར་ལོངས་་སྤྱོད་དོ། །དེ་བཱི་ཀོ་ཊིར་ཐིམ་མོ། །:གོ་དཱ་ཝ་རཱིར་བདག་ཏུ་བྱེད་དོ། །།མཱ་ལ་ཝར་ཁྱབ་བདག་ཉིད་དོ། །དེ་ཝཱི་ཀོ་ཊིར་ལོངས་སྤྱོད་དོ། །མཱ་ལ་ཝར་ཐིམ་མོ། །ར་མེ་ཤྭ་རར་བདག་ཏུ་བྱེད་དོ། །གོ་:དཱ་བཱ་རིར་ཁྱབ་བདག་ཉིད་དོ། །མཱ་ལ་བར་ལོངས་སྤྱོད་དོ། །གོ་ད་བ་རིར་ཐིམ་མོ། །དེ་ཝཱི་ཀོ་ཊིར་བདག་ཏུ་བྱེད་དོ༑། །རཱ་མེ་ཤྭ་རར་ཁྱབ་བདག་ཉིད་ཅེས་བྱའོ། །དེ་བཞིན་དུ་ཀཱ་མ་རུ་པར་ལོངས་སྤྱོད་དོ། །ཨོ་ཊེར་ཐིམ་མོ། །ཀོ་ས་ལར་བདག་ཏུ་བྱེད་དོ། །ཏྲི་ཤ་ཀུ་ནར་ཁྱབ་བདག་ཉིད་དོ། །ཨོ་ཊིར་ལོངས་སྤྱོད་དོ། །ཏྲི་ཤ་ཀུ་ནར་ཐིམ་མོ། །ཀཱ་མ་རཱུ་པར་བདག་ཏུ་བྱེད་དོ། །ཀོ་ས་ལར་ཁྱབ་བདག་ཉིད་དོ། །ཏྲི་ཤ་ཀུ་ནར་ལོངས་སྤྱོད་དོ། །ཀོ་ས་ལར་ཐིམ་མོ། །ཨོ་ཊེར་བདག་ཏུ་བྱེད་དོ། །ཀཱ་མ་རཱུ་པར་ཁྱབ་བདག་ཉིད་དོ། །ཀོ་ས་ལར་ལོངས་སྤྱོད་དོ། ་། :ཀ་མ་རཱུ་པར་ཐིམ་མོ། །དྲི་ཤ་ཀུ་ནར་བདག་ཏུ་བྱེད་དོ། །ཨོ་ཊེར་ཁྱབ་བདག་ཉིད་ཅེས་བྱའོ། །།དེ་བཞིན་དུ་ཀ་ལིངྒར་ལོངས་སྤྱོད་དོ། །:ལམ་ཀར་ཐིམ་མོ། །ཧི་མཱ་ཅེལ་ཡར་བདག་ཏུ་བྱེད་དོ༑། ཀཱཉྩིར་ཁྱབ་བདག་ཉིད་དོ། །ལམཱ་ཀར་ལོངས་སྤྱོད་དོ། །ཀཱཉྩིར་ཐིམ་མོ། །ཀ་ལིངྒར་བདག་ཏུ་བྱེད་དོ། །ཧི་མཱ་ལ་ཡར་ཁྱབ་བདག་ཉིད་དོ། །ཀཱཉྩིར་ལོངས་སྤྱོད་དོ། །ཧི་མཱ་ལ་ཡར་ཐིམ་མོ། །ལམྤ་ཀར་བདག་ཏུ་བྱེད་དོ། །ཀ་ལིངྐར་ཁྱབ་བདག་ཉིད་དོ། །ཧི་མཱ་ལ་ཡར་ལོངས་སྤྱོད་དོ། །ཀ་ལིངྒར་ཐིམ་།
