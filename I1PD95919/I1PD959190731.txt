བསྟན་འགྱུར། སེམས་ཙམ། ༼ཞིཔའི༑@«ཅོ་»ངན། ཚེ«པེ་»«སྣར་བཅུ་དྲུག་གི། «ཅོ་བཅུ་དྲུག«པེ་»«སྣར་_གཅིག་དང་གསུམ་གྱི་ཕྱོགས་སོ། །...དུ་ཞིག་འདུས་བྱས་ཡིན་ཞེ་ན། སྨྲས་པ། «པེ་»«སྣར་—དབང་པོ། རང་«ཅོ་»ཀྱིས། «ཅོ་»ནི། ༄༅«པེ་»«སྣར་»—ཡོངས་སུ་རླན་པར་བྱེད། །«པེ་《སྣར་»དུ། «པེ་»《སྣར་»ཤེས། དེནས་འབར་བའི་མེ་ཡོས། («པེ་»«སྣར་གཉིས། «པེ་»«སྣར་གཉིས། «པེ་»«སྣར་»ཡིན། «པེ་»«སྣར་»—ཡིན་ན། མུ@«པེ་»«སྣར་»ནི། །«སྣར་ལས། འ«པེ་《སྣར་»—དུ། «པེ་+དང་། «སྣར་ལྟར། «པེ་»«སྣར་—ལས། «པེ་《སྣར་གོ། «པེ་»«སྣར་+དག་ལ་གསོན། རྡོའི་མིང་«ཅོ་ནང་། «པེ་»«སྣར་—དང༌། «ཅོ་བར། «སྣར་རྣམ། སུ«པེ་»«སྣར་»མ༑@«པེ་»«སྣར་»+ཚང་བ་རྣམ་སྤོང་བར་བྱེད་ཅིང་མ་ཚང་བ་རྣམས་འཐོབ་པར་བྱེད་པ་དང་། དེ༌(«པེ་《སྣར་—གསུམ། «པེ་»«སྣར་—དག«པེ་《སྣར་»—ས། «སྣར་སྦྱོང་། «པེ་»«སྣར་»+དོན་གྱིས། «པེ་»«སྣར་ཕོ་དང་མོའི་དབང་པོ་ཕན་ཚུན། «པེ་»«སྣར་པ། ®«པེ་+དབང་པོ། །«པེ་»«སྣར་»ཡིད། པོ«སྣར་དྲན། «པེ་»«སྣར་»འཐོབ། ན་ ེ«པེ་»«སྣར་བདུན། «པེ་»«སྣར་ལས། «སྣར་པོ། «པེ་»«སྣར་ཡང་། །«པེ་《སྣར་ནུ།  ི«ཅོ་»—པ། ་། «པེ་»«སྣར་གཏོགས། «སྣར་»གྱིས། །«པེ་»«སྣར་»གཏོགས། «པེ་»«སྣར་»དག་«པེ་《སྣར་—དེ་དག་ཉིད་ད་ལྟར་གྱི་དག་ཡིན་ལ། དོན་ཡང་ད་ལྟར་བྱུང་བ་དག་ཡིན་པར་ཡང་བལྟ་བར་བའོ། །«པེ་《སྣར་གྱི། «སྣར་+དང་། «པེ་»«སྣར་+ལྡན་་པ་དང་། «པེ་»—ཡིན་པ། ་«པེ་»«སྣར་»«ཅོ་བསྟན། «ཅོ་»—པ། «པེ་པ། ནང་«པེ་»«སྣར་བསྒོམ། «པེ་»«སྣར་བསྒོམ། (འ«པེ་»«སྣར་བསྒོམ། ་®«པེ་»«སྣར་བསྒོམ། «པེ་བར་དུ། «སྣར་རབ་དུ། «པེ་»«སྣར་»གྱི། །«པེ་«སྣར་»སྒྱུར། «པེ་»«སྣར་ལས། «པེ་»«སྣར་»སུམ། «པེ་»«སྣར་»བའི། མེ«པེ་པ༑«པེ་»བོ། །«པེ་»«སྣར་»བརྟེན། «སྣར་»པའི། «པེ་»—དུས་ཀྱི། །དེནས་:(ི«པེ་»སྤོང་བ། «པེ་»«སྣར་»—དུ། «ཅོ་»ཀྱི། ︽«པེ་»«སྣར་»ཅན། «ཅོ་ཆེན། སུ«པེ་»བསྒོམ། པོ«པེ་《སྣར་བཅས། «པེ་»«སྣར་»མཐོར། ་ན<«པེ་»«སྣར་ལས། དགྱེ«པེ་»«སྣར་»རྒྱུ་སྟེ། ཀྱེ«ཅོ་»ལ། (ཉེ«པེ་»«སྣར་»ཤེས། «པེ་»«སྣར་ལས་ཏེ། །་་་་་«པེ་»«སྣར་པུ། «སྣར་»རྟེན། («པེ་»«སྣར་བཤམ། «པེ་»«སྣར་ལས། «ཅོ་»དོ། གི«པེ་»«སྣར་ཀྱི་ནི། ར་་—།
