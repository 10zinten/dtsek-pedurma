བསྟན་འགྱུར། རྒྱུད། བཛྲ་ཀྲོ་དྷ་ས་མ་ཡ་ཕཊ། ཨ་ལི་ཨུ་ལི། ཏ་ལི། ཏ་པ་ལི། དབྦཾ་ཀོ་ན་རོ་ཏྲི། ཁ་རཾ་ཡོ་ཀི་ནི། ཁ་ཧི་ཧོ། རྡོ་རྗེ་བསྟིམ་གླུ་དྲུག་བསྟོད་བྱ། ཧཱུྃ་ཧ་ཧེ་ཕཊ་བློ་རཾ། ཀ་ཡ་བཱཀ་ཙིཏྟ་ཏིཥྛ་བཛྲ། ཨོ་ཤི་ཧེ་རུ་ཀ་མ་ཧཱ་བཛྲ་སརྦ་དུཥན་ས་མ་ཡ་ཛཾཊཾ་ཨ་བ་སྟེ་པ་ཡ་ཀ་ཧཱུཾ། ཨོཾ་བཛྲ་གླུ་ཧེ་ཏྲི་ཤ་ས་མ་ཡ་ཧཱུཾ། ཨོཾ་བཛྲ་གླུ་ཧེ་ཏྲི་ཤ་ས་མ་་ཡ་ཧོ། ཨོ་བཛྲ་གྷ་ཧེ་ཏྲི་ཤ་ས་མ་ཡ་ཕཊ། ཨོཾ་ཤྲཱི་ཧེ་རུ་ཀ་མ་ཧཱ་བཛྲ་སརྦ་དུཥྚ་། ས་མ་ཡ་མུ་ཏྲ་མ་རཾ་པྲ་བན་ཙཀྲ། བག་ཤ་ཡ་བག་ཤ་ཡ། བི་ཏཱ་ས་ཡ་བི་དོ་ས་ཡ། །བྱ་བྷ་ག་བཱན། འབུ་དྷ་ག་ཐེ་ཏྲི་པྲ་ཏི་ཙཱ་པ་ཡ་ཀ་ཧཱུཾ་ཕཊ། བདག་མེད་ཕྱོགས་་བྲལ་དོན་དམ་ཐུགས་ཆུད་ཅིང་། །བྱམས་དང་སྙིང་རྗེ་དགྲ་ཟིན་མི་མངའ་ཡང་། །གདུགས་པ་འདུལ་ཕྱིར་དུས་ལས་མི་འདའ་བའི། །འཇིགས་བྱེད་ཧེ་རུ་ཀ་ལ་ཕྱག་འཚལ་ལོ། །ཁྲོ་མོ་དབང་ཕྱུག་སྐུ་ལ་གཉེར་གནས་ཤིང༌། །ཐབས་དང་ཤེས་རབ་འབྲེལ་བའི་དཀྱིལ་འཁོར་ནས། །མཁའ་འགྲོ་མི་ཟད་འབར་བའི་སྤྲིན་་་འཕྲོ་བའི། །གདུག་ཅཔ་འདུལ་མཛད་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །ནམ་མཁའི་ལུས་ཅན་གར་ཡང་ཐོགས་:མེད་ཅིང་། །འདོད་དགུར་བསྒྱུར་བའི་གཟུགས་ཅན་རྫུ་འཕྲུལ་ཅན། །མ་སྲིང་བྱམས་མའི་ཚུལ་གྱིས་གྲོགས་མཛད་མ། །ཕྱི་ནང་མཁའ་འགྲོའི་ཚོགས་ལ་ཕྱག་འཚལ་ལོ། །ཨ་ཏི་པུ་ཧོ། པྲ་ཏི་པུ་ཧོ། མཉམ་་པའི་གྲུབ་རྫས་ཆེན་པོ་དག། བདག་དང་མཉམ་པའི་ཚོགས་ལ་དབུལ། །།ཨོ་ཨོཾ་ཧཱུཾ་སུ་ར་ཏོ། ཀ་མ་ལ། །མ་ཧཱ་སུ་ཀ་། མ་ལེ། མ་ནུ་ས་མ་ན། ཛ་ལི་ནི་ད་ན༑ཨ་པ་ཡོ་ཁ་ཧི། ཨ་ལ་ལ་ཧོ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་པཉྩ་ཤྲཱི་ཧེ་རུ་ཀ་མནྟ་ཛ་ལ་མ་ཧཱ་མན་ལ། ཀ་ཡ་བཱ་ཀ་ཙིཏྟ། ཨོ་ཨོ་ཨོ། །ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཨོཾ་ཨཾ་ཨོཾ། ཧཾ་ཧཾ་ཧཾ། བཛྲ་མ་ཧཱ་ཀ་པ་ལ། ཁར་ཀ་ད་ན་ད་ན་ཛཱ་ལ་སིདྡྷི། མ་ཧཱ་ཤྲཱི་ཧེ་རུ་ཀ་ཨ་མེ་དུ་མེ་སིད་ཏོན་བྷ་བྷེ། །སཾ་ཧ་ར་ན་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། ་མ་ཧཱ་ཡོ་ག་།
