།།།སན་འདོད་པ་གཉིས། །བཀའ་ཡི་ཐ་རམས་དམ་པོ་དེ། །ཙ། ཀུན་འཆར་ཀུན་གྱི་མཐར་མི་གནས། བ། །ལྔ་བཞི་ལྡེ་མིག་ཐབས་ཀྱིས་དབྱེ། པ། ཐུགས་ཀྱི་སྒོ་ལྕགས་དམ་པོ་འགྲོལ། །ཐུགས་ཀྱི་སྒོ་ལྕགས་དམ་པོ་ལས། །ལུས་ཀྱི་བཅའ་བཞིའི་ལེའུ་སྟེ་དང་པོའོ། །ཨ་། པ་:ཀྱི་རྒྱལ་བ་དང་། ཚོ་ད་པ་ཚོ། ད་བ་མའི་ལུས་ཀྱི་དང་པོ་འདིར་བསྟན་པ། །མུན་པ་ཅན་དང་ལོག་སྲེད་ཅན། །མུན་པའི་འདོད་པ་གཉིས་ཡིན་ཏེ། །ཕྱལ་བ་དང་ནི་རྒྱང་ཕན་ནོ། །ཕྱལ་བ་ལ་ཡང་གཉིས་ཡིན་ཏེ། །དམངས་རིགས་དང་ནི་རིགས་མ་ངེས། །དམངས་་རིགས་སྒྲུབ་བྱེད་རིགས་པ་ནི། །བསྟན་བཅོས་བློ་ལ་མི་སྣང་བར། །རང་འཐད་འདོད་པ་ཚད་མར་བྱེད། །རང་བཞིན་ལོག་པ་ཞེས་སུ་བསྟན། །མ་ངེས་པ་ལ་གཉིས་ཡོད་དེ། །གང་ལ་གྲངས་པར་འཇུག་པ་དང༌། །བདག་འཛིན་རྫས་སུ་་འདོད་པའོ། །བདག་འཛིན་པ་ཡི་རིགས་པ་ནི། །རྒྱུ་རྐྱེན་ལས་ལས་མ་གྲུབ་་་པར། །དངོས་པོའི་ཆོས་རྣམས་ཐམས་ཅད་ལ། །བྱེད་པ་པོ་ཡིས་བྱས་པར་འདོད། །བརྒྱང་འཕན་འདོད་པ་འདིར་བསྟན་པ། །རིགས་ཆད་པ་དང་འདུ་ཤེས་ཅན། །ཞི་། རིགས་ཆད་བསྒྲུབ་པའི་རིགས་པ་ནི། །དགེ་སྡིག་ལས་ཀྱི་ཆོས་རྣམས་ལ། །རྒྱུ་དང་འབྲས་བུ་མ་ངེས་ཏེ། །།མེ་ཤི་ཤིང་ཟད་ལྟ་བུར་འདོད། །འདུ་ཤེས་ཅན་ལ་གསུམ་ཡོད་དེ། །སྤྱོད་ཡུལ་ཚད་མར་བྱས་ནས་ནི། །ལོངས་་སྤྱོད་ཡང་དག་འཛིན་པའོ། །དེ་ནི་གཡེང་བའི་འདུ་ཤེས་ཅན། །ཕྱི་པར་ལོངས་སྤྱོད་འབྱུང་བར་འདོད། །གསང་ཚིག་རྟེན་ཅིང་ལྷ་ལ་མཆོད། །བགེགས་ལ་་་ཡང་དག་ཚད་མར་བྱེད། །དེ་ནི་ས་ར་པ་ཡི་འདུ་ཤེས་ཅན། །ལྷ་མཆོད་དངོས་གྲུབ་འབྱུང་བར་འདོད། ་། ཕྱི་ཡིས་བསྟན་ཅིང་ཡང་དག་འཛིན། །དམིག་པ་་།
