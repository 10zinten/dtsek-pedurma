རྡོ་རྗེ་ཁུ་ཚུར་གྱི་རྡོར་དྲིལ་ཐབས། །གར་ལྷ་མོ་བརྒྱད་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །དཔྱི་གཉིས་བཞག་ཅིང་མགོ་གཡོན་འདུད། །།དེ་ནི་སྒེག་མོའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །དཔུང་པ་གཉིས་སུ་བཟུང་བར་བྱ། །དེ་ནི་ཕྲེང་བའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་ཞུ་གཡས་ཐད་དྲངས་ལ། །དྲིལ་བུ་དཔུང་གཡོན་ཐད་ཀར་བཟུང་། །དེ་ནི་པི་ཝང་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་གཡོན་གྱི་གྲུབ་མོར་བརྟེན། །།དྲིལ་བུ་གྱེན་དུ་བཟུང་བར་བྱ། །དེ་ནི་གར་མའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །ཐུར་དུ་ཕུབ་བམ་འདེག་པའི་ཚུལ། །དེ་ནི་བདུག་པའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །ཐུར་དུ་ཕུབ་ཅིང་གྱེན་ལ་གཟིང་། །དེ་ནི་མེ་ཏོག་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །མདུན་དུ་གཞིབས་ཏེ་བསྒྲེང་བར་བྱ། །དེ་ནི་མར་མེའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་སྟེང་དུ་བཟུང་བྱས་ནས། །དྲིལ་བུ་འོག་ཏུ་བཟུང་བར་བྱ། །དེ་ནི་དྲི་ཆབ་རྡོར་དྲིལ་ཐབས། །གར་xསྒོ་བ་བཞིའི་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །བསྣོལ་ཏེ་ནུ་མའི་ཐད་བཟུང་ལ། །དེ་ནི་ལྕགས་ཀྱུའི་རྡོར་དྲིལ་ཐབས། །གར་xརྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །བསྣོལ་ཏེ་རྒྱབ་ནི་སྦྱར་བར་བྱ། །དེ་ནི་ཞགས་པའི་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །སྤྲད་དེ་གྱེན་དུ་བསྟེན་པར་བྱ། །དེ་ནི་ལྕགས་སྒྲོག་རྡོར་དྲིལ་ཐབས། །གར་རྡོ་རྗེ་དང་ནི་དྲིལ་བུ་གཉིས། །སྦྱར་ཞིང་བརྟན་པར་བཟུང་བར་བྱ། །དེ་ནི་དྲིལ་བུའི་རྡོར་དྲིལ་ཐབས། །དེ་་རྣམས་ལྷ་རྣམས་སོ་སོ་ཡི། །རྡོ་རྗེ་དྲིལ་ཐབས་ཡིན་པར་བཤད། ་། ལྷ་ཀུན་ཐུན་མོངས་དྲིལ་ཐབས་བཤད། །རྡོ་རྗེ་རྩེ་མོའི་ཐུགས་བརྟན་ཤིང་། །དྲིལ་བུ་བཀྲོལ་བའི་ཐབས་རྣམས་བྱ། །།གཡས་བརྟན་པའི་ཐབས་དང་། ཡོན་བརྟན་—།
