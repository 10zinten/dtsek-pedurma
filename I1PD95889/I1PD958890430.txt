སྒྱུ་འཕྲུལ་ཁྲོ་བོའི་དབང་བསྐུར་བ་དཀྱིལ་འཁོར་རྡོ་རྗེ་ལས་ཀྱི་རིམ་པ་ཞེས་བྱ་བ། པའི་འཕྱལ་བ་དག། རྨོངས་ལ་འཕྱར་བས་གལ་ཏེ་བྱིན། །དུས་མིན་སྲོག་ཟད་སྦྲེགས་བསྲེགས་བལྟུང་། །ཆགས་པས་རང་དགར་སྤྱོད་འགྲུས་དང་། །ཚུལ་སྡོམ་རྟོག་པས་བཅིངས་པ་དང༌། །ཉིང་ཁུ་མེད་པར་སྒྲ་བརྟེན་སྤྱོད། །ཨེ་མ་ཧོ། སྙིང་རྗེས་སེམས་འཕྲོགས་ནས། ཐབས་ལམ་མཆོག་འདི་རབ་བགྱིས་པས། །ཆགས་པས་མི་བལྟུང་བདེ་ཐོབ་ཤོག། སྒྱུ་འཕྲུལ་དྲ་བ་ཁྲོ་བོའི་དཀྱིལ་འཁོར་དབང་རབ་གནས་ཀྱི་བསྒྲུབས་ཐབས། ཁྲོ་བོ་རྡོ་རྗེ་ལས་རིམ། བུདྷ་གྷ་བྱས་་མཛད་པ་འདི་ཡོངས་སུ་རྫོགས་སོ། །།་ཊ་་་ཅེབྱེད་བསྡུར་མཆན། མཚན་བྱང་འདི་ཆོས་ཚན་འདིའི་མཚན་བོད་སྐད་དུ་སྨོས་པ་ལྟར་བཀོད། «པེ་»མནྟ། «པེ་»བཞིར། །«པེ་རློབས། ®«པེ་»བརླབ། «པེ་»ལིམ། ཨེ«པེ་བསྔོས་པས། «པེ་སྲོག། ཨ«པེ་བསྟན། «པེ་བསྡུས། «པེ་»གིས། &«པེ་པས། «པེ་བསྐོ། «པེ་»སྦྱངས། «པེ་མེ། «པེ་བྱ། །«པེ་རྗེས། །«པེ་»ཁྱམས། @«པེ་བཟླས། «པེ་དྲི། «པེ་»ཡི། «པེ་»ཆུས། «པེ་»འཇིག&«པེ་རིགས། ཚུ«པེ་»ཐིམས། ༄༅«པེ་»སྤུན། །ཚེ«པེ་»བོ། ཉི«པེ་»དྲང་། «པེ་»གཟུང་། &«པེ་»ཡིས། «པེ་»སྨོན། ®«པེ་»བརྒྱས། «པེ་གྲང་། «པེ་མཛེས་པ་ལ། «པེ་དང༌། ཉི«པེ་»གྱི། @«པེ་པར། །«པེ་བཞུགས། ད་དུང་བདག་«པེ་སྡུས། «པེ་ཁམ། «པེ་»ཚལ་བུ་སྤྱི་གཏུགས། །@«པེ་སྐྱོད། །
