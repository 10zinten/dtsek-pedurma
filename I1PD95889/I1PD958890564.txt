ད་དགེ་བ་ས་ཀྱི་བཀའ། །ཁ་ཏ༄༅། །ལྷའི་སྐུ་སྒོམ་པའི་མན་ངགདེ༌ཡཡང་འཁར་རྒྱུག་ਹੈਬਸਸਕਿਉਨਲਸਸਤਲੂਨਾਨੀਵੀਗਲਨੂੰਸਬਤੋਂਉB%ਮਹਲਨਕੋਇ॥༄༅། བི་མ་ལའི་ལྷ་སྐུ་སྒོམ་པའི་མན་ངག་གོ། ༄༅། །རྒྱ་གར་སྐད་དུ། དེ་བ་ཀ་ཡ་མུ་ཏྲ་ས་མ་ཏ་བ་ཏྲེ་ན་མ། བོད་སྐད་དུ། ལྷའི་སྐུ་སྒོམ་པའི་མན་ངག། དཔལ་རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །དམིགས་དང་བདག་ཉིད་རྫོགས་་གསལ་བ། །རྣལ་འབྱོར་རིམ་པ་བཤད་པར་བྱ། །བསམ་གཏན་དགྲ་ལས་རིང་གྱུར་པའི། །བཀྲ་ཤིས་གནས་སུ་བསྡད་པར་བྱ། །བསྡུ་བའི་རྒྱན་གཉིས་་་འཚོགས་བྱས་པས། ་ཎ། བསྒོམ་ཐབས་རིམ་པ་གཉིས་ཡོད་དེ། །ཐུན་མོང་ཁྱད་པར་དབྱེ་བའོ། །རབ་གནས་བྱས་པའི་གཟུགས་བརྙན་ལ། །ཀུན་ཏུ་བརྟག་ཅིང་དམིགས་པ་བསྡུ། །མི་ཡི་མིག་གིས་བལྟར་ཡོད་པར། །ཤིན་ཏུ་འདྲིས་པ་་ཉིད་དུ་བྱ། །ས་ཕྱོགས་སྟོང་པ་ཐམས་ཅད་དག། གཟུགས་བརྙན་ཉིད་དུ་གསལ་མཐོང་ན། །ཐུན་མོང་ཞི་གནས་གསལ་བའི་ཚད། །དམིགས་པ་བསྡུ་ཞིང་ཡུལ་ལ་བརྟག། དེ་ལ་ཤིན་ཏུ་འདྲིས་པར་བྱ། །བདག་ཉིད་ཕྱག་རྒྱ་ཉིད་་།
